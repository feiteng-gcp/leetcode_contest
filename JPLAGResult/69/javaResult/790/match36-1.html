class Solution {
    public boolean isIdealPermutation(int[] A) {
        for(int i = 0; i < A.length - 1; i++) {
            if(A[i] > A[i + 1]) {
                int tmp = A[i]; //A[i] =
                A[i] = A[i + 1];
                A[i + 1] = tmp;
                i = i + 1;
            }
        
}

        for(int i = 0; i < A.length - 1; i++) {
            if(A[i] > A[i + 1]) {
                return false;
            }
        }
        return true;
    }
}
