class Solution {
    public int numJewelsInStones(String J, String S) {
        
        int res = 0;
        ArrayList<Character> ar = new ArrayList<>();
        
        for ( char c : J.toCharArray())
        {
            ar.add(c);
        }
        
        
        for ( int i = 0;i < S.length(); i++)
        {
            if (ar.contains(S.charAt(i) ) )
              res++ ;
            
        }
        
        return res;        
    }
}
