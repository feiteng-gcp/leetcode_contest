class Solution {
    public int numJewelsInStones(String J, String S) {
        //Map<Character, Integer> freq = new HashMap<>();
        Set<Character> jews = new HashSet<>();
        for(Character c: J.toCharArray()){
            jews.add(c);
        }
        
        int cnt = 0;
        for(Character s : S.toCharArray()){
            if(jews.contains(s)) cnt++;
        
        }
        return cnt;
    }
}
