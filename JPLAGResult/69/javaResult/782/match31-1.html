class Solution {
    public int numJewelsInStones(String J, String S) {
        if (J == null || J.length() == 0 || S == null || S.length() == 0) {
            return 0;
        }
        int res = 0;
        
Set<Character> dic = new HashSet<>();
        for(int i = 0; i < J.length(); i++) {
            dic.add(J.charAt(i));
        }
        
        
for(int i = 0; i < S.length(); i++) {
            if (dic.contains(S.charAt(i))) {
                res++;
            }
        }
            
        return res;
    }
}
