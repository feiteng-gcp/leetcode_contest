class Solution {
    public int numJewelsInStones(String J, String S) {
       Set<Character> set = new HashSet<>();
        for (Character character : J.toCharArray()) {
			set.add(character);
		}
        int count = 0;
        for (Character character : S.toCharArray()) {
        	if (set.contains(character)){
        		count++;
        	}
		}
        return count; 
    }
}
