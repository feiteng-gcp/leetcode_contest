class Solution {
    public int numJewelsInStones(String J, String S) {
        Set<Character> hs = new HashSet<>();
        for (int i = 0; i < J.length(); i++) {
            hs.add(J.charAt(i));
        }
        
        int res = 0;
        for (int i = 0; i < S.length(); i++) {
            if (hs.contains(S.charAt(i))) {
                res++;
            }
        }

        return res;
    }
}
