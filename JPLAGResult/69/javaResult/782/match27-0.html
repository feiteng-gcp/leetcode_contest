class Solution {
    public int numJewelsInStones(String J, String S) {
        // for each char in J
            // for each char in S
                // if S[i] matches J[k], replace J[i] with nothing and increase num
        // return num
        int num = 0;
        for (int i = 0; i < J.length(); i++) {
            for (int k = 0; k < S.length(); k++) {
                if (J.charAt(i) == S.charAt(k)) {
                    num++;
                }
            }
        }
        return num;
    }
}
