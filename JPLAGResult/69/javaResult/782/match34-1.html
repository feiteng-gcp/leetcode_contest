class Solution {
    public int numJewelsInStones(String J, String S) {
        int ret = 0;
        
        Set<Character> s = new HashSet<Character>();
        for (int i = 0; i < J.length(); i++) {
            s.add(
J.charAt(i));
        }
        for (int i = 0; i < S.length(); i++) {
            if (s.contains(S.charAt(i)))
                ret++;
        }
        
        return ret;
        
    }
}
