class Solution:
    def numJewelsInStones(self, J, S):
        """
        :type J: str
        :type S: str
        :rtype: int
        """
        dic=dict()
        for i in range(len(J)):
            if not dic.get(J[i]):
                dic[J[i]]=1

        count=0
        for i in range(len(S)):
            if dic.get(S[i]):
                count+=1 
        return count
