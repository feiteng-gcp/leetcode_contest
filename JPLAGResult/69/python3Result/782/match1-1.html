class Solution:
    def numJewelsInStones(self, J, S):
        """
        :type J: str
        :type S: str
        :rtype: int
        """
        jewel_dict = {}
        if len(J) == 0:
            return 0
        
        if len(J) == 1 and J != S:
            return 0
        
        
for i in range(len(J)):
            jewel_dict[J[i]] = 1
        
        count = 0
        for k in range(len(S)):
            if S[k] in jewel_dict:
                count += 1
        
        return count
