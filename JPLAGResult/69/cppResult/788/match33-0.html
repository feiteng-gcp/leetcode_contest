class Solution {
public:
    vector<double> dis;
    int check(double d, int k) {
        for (int r = 0; r < dis.size(); r++) {
            int cnt = floor((dis[r]/d));
            k -= cnt;
            if (k < 0) return 0;
        }
        return 1;
    }
    
    double minmaxGasDist(vector<int>& st, int K) {
        sort(st.begin(), st.end());
        double maxd = 0, mind = INT_MAX;
        for (int i = 1; i < st.size(); i++) {
            double dd = st[i] - st[i-1];
            dis.push_back(dd);
            if (dd > maxd) maxd = dd;
            if (dd < mind) mind = dd;
        }
        
        if (K == 0) return maxd;
        mind = mind/double(K);
        double eps = 1e-6;
        
        while (maxd - mind > eps) {
            double mid = mind + (maxd - mind) * 0.5;
            int ok = check(mid, K);
            
            if (ok) {
                maxd = mid;
            } else {
                mind = mid;
            }
        }
        return maxd;
    }
};
