const double eps = 1e-7;
class Solution {
    bool solve (double mid, int k, vector<double> & dist) {
        for (double item : dist) {
            if (k  < 0 || fabs(item-mid) < eps || item < mid) break;
            int tmp = item / mid;
            if (item / tmp < mid || fabs(item/tmp-mid) < eps) {
                k -= (tmp-1);
            }
            else k -= tmp;
        }
        if (k >= 0) return true;
        return false;
    }
public:
    double minmaxGasDist(vector<int>& stations, int K) {

        vector<double> dist;
        for (int i = 0; i+1 < stations.size(); i++) {
            dist.push_back(stations[i+1]-stations[i]);
        }
        sort(dist.begin(), dist.end(), greater<double>());
        double right = dist[0], left = 0;
        while (right-left > eps) {
            double mid = (left+right) / 2;
            if (solve (mid, K, dist)) right = mid;
            else left = mid;
        }
        return left;
        
        
    }
};
