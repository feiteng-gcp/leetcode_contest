class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        int *A = stations.data();
        int n = (int)stations.size();

        double l = 1e-6, r = 2e8;
        for (int _ = 0; _ < 50; _++) {
            double mid = (l + r) * 0.5;
            long long sum = 0;
            for (int i = 1; i < n; i++) {
                long long tmp = (long long)((A[i] - A[i - 1]) / mid);
                sum += tmp;
            }
            if (sum > K) {
                l = mid;
            } else {
                r = mid;
            }
        }
        return l;
    }
};
