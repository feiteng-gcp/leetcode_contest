class Solution {
public:
    priority_queue<double,vector<double>> hp;
    
    double minmaxGasDist(vector<int>& stations, int K) {
        double f=0,g=2000.0;
        double ans;
        sort(stations.begin(),stations.end());
        while (fabs(g-f)>1e-7) {
//        	printf("%f\n",ans);
            ans = (f+g)/2.0;
            int kk = 0;
            for (int i=0;i<stations.size()-1;++i) {
                if ((stations[i+1]-stations[i])>ans) {
                    kk += (stations[i+1]-stations[i])/ans;
                }
            }
            if (kk<=K) {
                g = ans;
            } else {
                f = ans;
            }
        }
        return ans;
    }
};
