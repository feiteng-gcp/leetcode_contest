class Solution {
public:
    bool checkOk(vector<int>& s, double m, int K)
{
    int cnt = 0, len = (int)s.size();
    double lastP = s[0];
    for(int i = 0; i < len; i++)
    {
        if(s[i] - lastP > m)
        {
            cnt += floor((s[i] - lastP) / m);
            lastP += m * floor((s[i] - lastP) / m);
            i--;
            if(cnt > K) return false;
        }
        else lastP = max(lastP, 1.0 * s[i]);
    }
    return true;
}

double minmaxGasDist(vector<int>& stations, int K)
{
    int len = (int)stations.size();
    sort(stations.begin(), stations.end());
    double r = 0;
    for(int i = 1; i < len; i++)
        r = max(1.0 * (stations[i] - stations[i - 1]), r);
    double l = 0;
    while(r - l > 5e-7)
    {
        double m = l + (r - l) / 2;
        if(checkOk(stations, m, K)) r = m;
        else l = m;
    }
    return l;
}
};
