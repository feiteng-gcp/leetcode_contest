class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        sort(stations.begin(),stations.end());
	int n=stations.size();
	double l=0.0,r=1.0*(stations[n-1]-stations[0])/(K+1);
	while(r-l>=1e-6)
	{
		double m=(l+r)/2;
		int need=0;
		for(int i=1;i<n;i++)
		{
			if(stations[i]-stations[i-1]<=m)continue;
			double now=1.0*(stations[i]-stations[i-1])/m;
			need=need+(int)now-1;
			if(now-(int)now>1e-8)need++;
			if(need>K)break;
		}
		if(need>K)l=m;
		else r=m;
	}
	return l;
    }
};
