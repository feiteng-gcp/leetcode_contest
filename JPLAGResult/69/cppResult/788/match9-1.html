class Solution {
public:

    double minmaxGasDist(vector<int>& stations, int K) {
        double begin = 0.;
        double end = 1e8;
        
        while (abs(end - begin) > 1e-7) {
            double mid = begin + (end - begin) / 2;
            
            int need = 0;
            for (int i = 0; i < stations.size() - 1; i++) {
                double d = stations[i+1] - stations[i];
                
                need += (d < mid + 1e-7) ? 0 :  d / mid;
                
                if (fabs(d - int(d/mid) * mid) < 1e-7) {
                    --need;
                }
            }
            

            if (need > K) {
                begin = mid;
            } else {
                end = mid;
            }
        }
        
        return begin;
    }
};
