class Solution {
public:
    double eps = 1e-7;

    bool possible(vector<int>& stations, int K, double dist) {
        int cnt = 0;
        for (int i = 0; i + 1 < stations.size(); i++) {
            cnt += (stations[i+1] - stations[i] - eps) / dist;
        }
        return cnt <= K;
    }
 
    double minmaxGasDist(vector<int>& stations, int K) {
        sort(stations.begin(), stations.end());
        double min = 0;
        double max = stations[stations.size() - 1] - stations[0];
        while (max - min > eps) {
            double mid = (min + max) / 2;
            if (possible(stations, K, mid)) {
                max = mid;
            } else {
                min = mid;
            }
        }
        return min;
    }
};
