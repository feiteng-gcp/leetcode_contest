class Solution {
    vector<int> s;
    int k;
    bool check(double d)
    {
        int left = k;
        for (int i=0; i<s.size()-1; ++i) {
            left -= int(ceil((s[i+1]-s[i])/d)+0.1)-1;
            if (left < 0) {
                return false;
            }
        }
        return true;
    }
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        s = stations;
        k = K;

        double l = 0.0, r = 1e8;
        while (r-l > 1e-6) {
            double m = (l+r)/2;
            if (check(m)) {
                r = m;
            } else {
                l = m;
            }
        }
        return l;
    }
};
