class Solution {
public:
    bool solve(double m, const vector<int>& stations, int K) {
        int n = stations.size();
        for (int i = 0; i < n - 1 && K >= 0; i++) {
            double d = stations[i + 1] - stations[i];
            if (d > m) {
                K -= ((int) ceil(d / m));
                K += 1;
            }
        }
        
        return K >= 0;
    }
    
    double minmaxGasDist(vector<int>& stations, int K) {
        double l = 0;

        double r = 2e8;
        for (int i = 0; i < 80; i++) {
            double m = (l + r) / 2.0;
            if (solve(m, stations, K)) {
                r = m;
            } else {
                l = m;
            }
        }
        return l;
    }
};
