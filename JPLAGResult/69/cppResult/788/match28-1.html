class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        double min_dist = 0, max_dist = INT_MIN;
        for (int i = 1; i < stations.size(); ++i) {
            double dist = stations[i] - stations[i-1];
            max_dist = max(max_dist, dist);
        }
        while (max_dist - min_dist > 0.000001) {
            double m = (min_dist + max_dist) / 2;
            int r = stations_required(stations, m);
            printf("[%lf, %lf] : %d\n", min_dist, max_dist, r);
            if (r > K) min_dist = m;
            else max_dist = m;
        }
        return (max_dist + min_dist) / 2;
    }
    
    int stations_required(vector<int>& stations, double dist) {
        int ans = 0;
        for (int i = 1; i < stations.size(); ++i) {
            auto n = static_cast<int>((stations[i] - stations[i-1]) / dist);
            ans += n * dist == (stations[i] - stations[i-1]) ? n - 1 : n;
        }
        return ans;
    }
};
