int pow(int a,int b){
    int ans=1;
    while(b){
        if(b&1)ans=ans*(ll)a%mod;
        a=(ll)a*a%mod;b>>=1;
    }
    return ans;
}

class Solution {
public:
    double minmaxGasDist(vector<int>& a, int K) {
        sort(all(a));
        int n=SZ(a);
        double l=0,r=1e8,eps=1e-9;
        while(l<=r){
            double m=(l+r)/2;
            int s=0;
            rep(i,0,n-1){
                int d=a[i+1]-a[i];
                if(d>m){
                    s+=ceil(d/m)-1;
                    if(s>K)break;
                }
            }
            if(s<=K)r=m-eps;
            else l=m+eps;
        }
        return (l+r)/2;
    }
};
