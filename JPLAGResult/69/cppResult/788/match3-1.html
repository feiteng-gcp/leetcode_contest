class Solution {
public:
    double minmaxGasDist(vector<int>& sta, int K) {        
        unordered_map<int,int> len2cnt;

        for(int i = 1; i < sta.size(); i++) {
            len2cnt[sta[i]-sta[i-1]]++;
        }
        double start = 0.0, end = 1.0e8;
        while(abs(start - end) > 1.0e-6) {
            double mid = start + (end - start) / 2.0;
            int bgt = K;
            bool done = false;
            for(auto entry : len2cnt) {
                double len = (double)entry.first;
                int cnt = entry.second;
                if(len > mid) {
                    int need = (int)ceil(len / mid) - 1;
                    bgt -= cnt * need;
                }
                if(bgt < 0) {
                    start = mid;
                    done = true;
                    break;
                }
            }
            if(!done) end = mid;
        }
        return start;
    }
};
