class Solution {
public:
    double minmaxGasDist(vector<int>& sta, int K) {
        if (sta.size() < 2) {
            return 0;
        }
        sort(sta.begin(), sta.end());
        double L = 0.00000001, R = sta.back() - sta[0];
        while (R - L > 0.00000001) {
            double M = (L + R) / 2;
            int count = 0;

            for (int i = 1; i < sta.size(); i++) {
                count += ceil((sta[i] - sta[i-1]) / M) - 1;
            }
            if (count <= K) {
                R = M;
            } else {
                L = M;
            }
        }
        return L;
    }
};
