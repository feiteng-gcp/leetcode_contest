class Solution {
public:
  double minmaxGasDist(vector<int>& stations, int K) {
    double eps = 1e-7;
    double l = 0.0, r = 100000000.0;
    while (r - l > eps) {
      double d = (l + r) / 2;
      int sum = 0;
      for (int i = 1; i < (int)stations.size(); ++i) {

        sum += (int)ceil((stations[i] - stations[i - 1]) / d - 1);
      }
      if (sum > K) {
        l = d;
      } else {
        r = d;
      }
    }
    return l;
  }
};
