const double eps = 0.0000001;

class Solution {
    
    bool ok(vector<int> & ss, int k, double d)
    {
        double cur = ss[0];
        int a = 0;
        for(int i = 1; i < ss.size(); ++i)
        {
            if (ss[i] - cur < d)
            {
                cur = ss[i];
                continue;
            }
            
            while(ss[i] - cur > d)
            {
                cur += d;
                ++a;
            }
            
            cur = ss[i];
        }
        
        return a <= k;
    }
    

public:
    double minmaxGasDist(vector<int>& stations, int K) {
        sort(stations.begin(), stations.end());
        int n = stations.size();
        double low = 0, high = 0;
        for(int i= 1; i < n; ++i)
        {
            if (high < stations[i] - stations[i-1])
                high = stations[i] - stations[i-1];
        }
        
        while(high - low > eps)
        {
            double mid = (low + high) / 2.0;
            if (ok(stations, K, mid))
                high = mid;
            else
                low = mid;
        }
        
        return high;
    }
};
