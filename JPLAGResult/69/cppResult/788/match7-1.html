class Solution {
public:

bool can(vector<int>& a, int k, double d) {
    int n = a.size();
    for (int i = 0; i < n - 1; i++) {
        int c = floor((a[i + 1] - a[i]) / d);
        k -= c;
    }
    return k >= 0;
}
double minmaxGasDist(vector<int>& a, int k) {
    int n = a.size();
    sort(a.begin(), a.end());

    int d = 0;
    for (int i = 0; i < n - 1; i++) {
        d = max(d, a[i + 1] - a[i]);
    }
    double left = 0, right = d;
    while(abs(left - right) > 1e-7) {
        double mid = (left + right) / 2;
        //cout << left << " " << right << " " << mid <<endl;
        if(can(a, k, mid)) right = mid;
        else left = mid;
    }
    return right;
}
};
