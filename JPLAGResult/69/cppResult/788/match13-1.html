class Solution {
public:
    double minmaxGasDist(vector<int>& a, int K) {
        double low = 0, high = 0;
        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());
        int n = a.size();
        for (int i = 1; i < n; ++i) high = max(high, (double)a[i] - a[i - 1]);
        while (high - low > 1e-7) {
            double mid = (low + high) / 2;
            int sum = 0;
            for (int i = 1; i < n; ++i) {
                int cur = (int)((a[i] - a[i - 1] + mid - 1e-7) / mid);
                sum += cur - 1;

            }
            //cout << mid << " " << sum << endl;
            if (sum > K) {
                low = mid;
            } else {
                high = mid;
            }
        }
        return high;
        
    }
};
