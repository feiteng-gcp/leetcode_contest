class Solution {
private:

    bool satisfy(vector<int>& stations, int K, double len)
    {
        int num = 0;
        for (int i = 0; i < stations.size() - 1; i++)
        {
            int distance = stations[i + 1] - stations[i];
            num += std::ceil((double)distance / len) - 1;
        }
        return num <= K;
    };
    
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        // sort stations
        std::sort(stations.begin(), stations.end());
        int size = stations.size();
        double minlen = (stations[size - 1] - stations[0]) / (double)(size + K - 1);
        int maxlen = minlen;
        for (int i = 0; i < size - 1; i ++)
            maxlen = max(maxlen, stations[i + 1] - stations[i]);
        
        double left = minlen;

        double right = maxlen;
        
        while (right - left > 0.000001)
        {
            // cout << left << ":" << right << endl;
            double mid = left + (right - left) / 2.0;
            if (satisfy(stations, K, mid))
            {
                right = mid;
            }
            else
            {
                left = mid;
            }
        }
        
        return left;
    }
};
