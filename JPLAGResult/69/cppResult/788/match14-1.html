class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        double l = 0, r = 1e8;
        int n = stations.size();

        for (int i = 0; i < 200; i++) {
            double mid = (l + r) / 2;
            int ans = K;
            for (int j = 0; j < n - 1; j++) {
                double dis = stations[j + 1] - stations[j];
                ans -= int(ceil((dis) / mid)) - 1;

                if (ans < 0) { break; }
            }
            if (ans >= 0) {
                r = mid;
            } else {
                l = mid;
            }
        }
        return l;
    }
};
