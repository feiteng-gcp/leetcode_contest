class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        int n = stations.size();
        sort(stations.begin(), stations.end());
        double eps = 1e-8;
        double l = eps, r = stations[n - 1] - stations[0];
        for (int pc = 0; pc < 200; pc++) {
            double dis = (l + r) / 2;
            long long cnt = 0;
            for (int i = 0; i < n - 1; i++) {
                double len = stations[i + 1] - stations[i];
                cnt += (int)((len / dis) - eps);
            }
            cnt <= K ? r = dis : l = dis;
        }
        return (l + r) / 2;
    }
};
