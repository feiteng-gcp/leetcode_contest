class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        sort(stations.begin(), stations.end());
        double l = 0, r = INT_MAX;
        for (int i = 0; i < 100; i++) {
            double mid = (l + r) / 2;
            if (check(stations, mid, K)) {
                r = mid;
            } else {
                l = mid;
            }
        }
        return (l + r) / 2;
    }
    bool check(vector<int>& stations, double mid, int K) {
        int sz = stations.size();
        for (int i = 0; i < sz - 1; i++) {
            if (stations[i+1] - stations[i] > mid) {
                K -= ceil((stations[i+1] - stations[i]) / mid) - 1;
            }
        }
        return K >= 0;
    }
};
