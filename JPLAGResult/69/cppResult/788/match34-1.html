struct P {
    int dist;
    int div;
    double key;
    void compute_key() { this->key = 1. * dist / div; }
    bool operator<(const P& p) const {
        return key < p.key;
    }
};
class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        vector<int> a;
        for(int i = 1; i < stations.size(); i++) {
            a.push_back(stations[i] - stations[i - 1]);
        }
        double l = 0, r = 1e8;
        while(r - l > 1e-7) {
            int k = K;
            double mid = (l + r) / 2.0;
            bool ok = true;
            for(auto s : a) {
                if(s <= mid) continue;
                int need = ceil(s / mid);
                need--;
                if(need > k) {
                    ok = false;
                    break;

                } else {
                    k -= need;
                }
            }
            if(ok) {
                r = mid;
            } else {
                l = mid;
            }
        }
        return l;
    }
};
