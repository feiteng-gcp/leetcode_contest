class Solution {
public:
    double minmaxGasDist(vector<int>& ss, int k) {
        int n = ss.size();
        sort(ss.begin(), ss.end());
        double l = 0, r = ss.back() - ss.front();
        while (r - l > 1e-8) {
            //cout << l << " " << r << endl;
            double v = (r + l) / 2;
            long long res = k;
            for (int i = 0; i < n - 1; ++i) {
                long long needed = floor((ss[i+1] - ss[i]) / v);
                //cout << ss[i] << " " << ss[i+1] << " " << needed << endl;

                res -= needed;
                //cout << res << endl;
                if (res < 0)
                    break;
            }
            if (res < 0)
                l = v;
            else
                r = v;
        }
        return l;
    }
};
