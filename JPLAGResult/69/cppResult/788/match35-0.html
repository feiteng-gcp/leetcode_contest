class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        int n = stations.size(), x, h[2005];
        double l = 0, r = 0, mid;
        for (int i = 1; i < n; i++) h[i] = stations[i] - stations[i - 1], r = fmax(r, h[i]);
        while (r - l > 1e-6)
        {
            mid = (l + r) / 2;
            x = 0;
            for (int i = 1; i < n; i++) x += ceil(h[i] / (mid + 1e-7)) - 1;
            if (x > K) l = mid;
            else r = mid;
        }
        return (l + r) / 2;
    }
};
