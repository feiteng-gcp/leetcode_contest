class Solution {
public:
    int N, M;
    vector<int> sts;
    
    bool ok(double dis)
    {
        int cnt = 0;
        for (int i = 0; i < N - 1 && cnt <= M; ++i)
            cnt += (sts[i + 1] - sts[i]) / dis;
        return cnt <= M;
    }

    double minmaxGasDist(vector<int>& stations, int K) {
        sts = stations;
        N = sts.size();
        M = K;

        double lo = 0.00000001, hi = 10e9;
        while (hi - lo > 0.00000001)
        {
            double mid = (hi + lo) / 2;
            if (ok(mid))
                hi = mid;
            else
                lo = mid;
        }
        return lo;
    }
};
