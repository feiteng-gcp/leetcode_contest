class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        priority_queue<pair<double,int>> gap;
        vector<double> dist;
        for (int i = 1; i < stations.size(); i++) {
            dist.push_back(stations[i] - stations[i - 1]);
            //gap.push({stations[i] - stations[i - 1], 1});
        }
        
        double left = 0, right = stations[stations.size() - 1] - stations[0];
        double eps = pow(10, -6), mid;
        while (right - left > eps) {
            mid = (right + left) / 2;
            double count = 0;
            for (auto d : dist) {
                count = count + max(0.0, ceil(d / mid) - 1);
            }
            if (count > K) {
                left = mid;
            } else {
                right = mid;
            }
        }
        return mid;
        
        /*
        while (K-- > 0) {
            auto cur = gap.top();
            gap.pop();
            double dist = cur.first * cur.second / (cur.second + 1);
            gap.push({dist, cur.second + 1});
        }
        
        return gap.top().first;
        */
    }
};
