class Solution {
public:
    double minmaxGasDist(vector<int>& st, int K) {
        double l=0,r=-1e15,mid;
        for (int i=0;i<st.size()-1;i++) r=max(r,1.0*st[i+1]-st[i]);
        while (r-l>1e-7){
            mid=(l+r)/2;
            int tot=0;
            for (int i=0;i<st.size()-1;i++){
                int len=st[i+1]-st[i];
                if (len<mid) continue;
                int t=ceil(len/mid)-1;
                tot+=t;
            }
            if (tot<=K) r=mid; else l=mid;
        }
        return l;
    }
};
