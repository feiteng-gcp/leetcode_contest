class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        vector<double>dist;
        sort(stations.begin(),stations.end());
        
        for(int i=0;i+1<stations.size();i++){
            dist.push_back(stations[i+1]-stations[i]);
        }
        
        double lo=0;
        double hi=(1e+9);
        
        for(int caso=0;caso<100;caso++){
            double me=(lo+hi)/2;
            
            int cant=0;
            for(int i=0;i<dist.size();i++){
                int val=dist[i];
                cant+=(int)(ceil(val/me))-1;
            }
            
            if(cant>K){
                lo=me;
            }else{
                hi=me;
            }
        }
        
        return lo;
    }
};
