class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        // how about binary search?
        sort(stations.begin(), stations.end());
        double left = 0, right = 0;
        for (int i=1; i<stations.size(); i++) {
            double diff = stations[i] - stations[i-1];
            if (diff > right)
                right = diff;
        }
        
        // double ans = right;
        while (abs(right-left) > 10e-7) {
            // cout << left << ", " << right << endl;
            double mid = (left + right) / 2;
            int count = 0;
            for (int i=1; i<stations.size(); i++) {
                double diff = stations[i] - stations[i-1];
                int num = int(diff / mid);
                if (num * mid == diff)
                    count += num - 1;
                else
                    count += num;
            }
            if (count <= K)
                right = mid;
            else
                left = mid;
        }
        return right;
    }
};
