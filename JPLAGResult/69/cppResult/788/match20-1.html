bool check(double x, vector<int>& a, int k) {
    int p = 0;
    for (int i = 1; i < a.size(); i++) {
        p += int((a[i] - a[i-1]) / x);
    }
    return p <= k;
}


class Solution {
public:
    double minmaxGasDist(vector<int>& stations, int K) {
        double le = 0, ri = 100000000.0;
        for (int _ = 0; _ < 50; _++) {
            double mid = (le + ri) / 2;
            if (check(mid, stations, K)) ri = mid;
            else le = mid;
        }
        return le;
    }
};
