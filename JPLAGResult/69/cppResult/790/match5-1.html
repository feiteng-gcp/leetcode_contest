const int N = 6000;
int c[N];

int query(int x) {
    int ret = 0;
    for (; x; x -= x & -x) {
        ret += c[x];
    }
    return ret;
}
void add(int x) {
    for (; x < N; x += x & -x) {
        c[x]++;
    }
}
class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        int ret = 0, n = A.size();
        for (int i = 1; i < n; ++i) {
            if (A[i - 1] > A[i]) ++ret;
        }
        memset(c, 0, sizeof(c));
        int sum = 0;
        for (int i = 0; i < n; ++i) {
            int cur = n - A[i];
            sum += query(cur);
            add(cur);
        }
        ret -= sum;
        return ret == 0;  
        
    }
};
