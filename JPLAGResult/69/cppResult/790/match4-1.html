class Solution {
    int N;
    vector<int> foo;
    int lowbit(int x) {return x & -x;}
    void add(int x) {
        while (x <= N) {
            foo[x]++;
            x += lowbit(x);
        }
    }

    int sum(int x) {
        int res = 0;
        while (x) {
            res += foo[x];
            x -= lowbit(x);
        }
        return res;
    }
public:
    bool isIdealPermutation(vector<int>& A) {
        N = A.size();
        foo.assign(N + 1, 0);

        int res1 = 0;
        for (int i = N - 1; i >= 0; --i) {
            res1 += sum(A[i]);
            add(A[i] + 1);
        }
        int res2 = 0;
        for (int i = 0; i < N - 1; ++i) {
            if (A[i] > A[i + 1]) ++res2;
        }
        return res1 == res2;
    }
};
