class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        int n = A.size();
        
        vector<int> minval(n);
        minval[n-1] = A[n-1];
        
        for (int i = n-2; i >= 0; --i) {
            minval[i] = min(A[i], minval[i+1]);
        };
        
        for (int i = 0; i < n-1; ++i)
            if (minval[i] < A[i]) {
                if (A[i+1] != minval[i+1]) return false;
                if (minval[i+1] != A[i]-1) return false;                
            };
            
        return true;
        
        
    }
};
