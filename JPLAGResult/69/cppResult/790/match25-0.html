class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        int n = A.size();
        int max_so_far = INT_MIN;
        for (int i = 0; i + 2 < n; ++i) {
            max_so_far = max(max_so_far, A[i]);
            if (A[i + 2] < max_so_far) return false;
        }
        return true;
    }
};
