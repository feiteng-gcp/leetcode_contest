class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        int num_of_glo = 0;
        int num_of_loc = 0;
        for(int i = 0;i < A.size();i++){
            for(int j = i + 1;j < A.size();j++){
                if(A[i] > A[j]) num_of_glo++;
                if(A[i] > A[j] && j == i + 1) num_of_loc++;
                if(num_of_glo != num_of_loc) return false;
            }
        }
        if(num_of_glo == num_of_loc) return true;
        else return false;
    }
};
