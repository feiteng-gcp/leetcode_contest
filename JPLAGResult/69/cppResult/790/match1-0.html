class Solution {
public:
    int c[5005];
int lowbit(int x)
{
	return x&-x;
}
void add(int x,int y)
{
	while(x<=5000)
	{
		c[x]=c[x]+y;
		x=x+lowbit(x);
	}
}
int sum(int x)
{
	int ans=0;
	while(x>0)
	{
		ans=ans+c[x];
		x=x-lowbit(x);
	}
	return ans;
}
    bool isIdealPermutation(vector<int>& A) {
        int n=A.size();
	memset(c,0,sizeof(c));
	add(A[0]+1,1);
	int ans1=0,ans2=0;
	for(int i=1;i<n;i++)
	{
		if(A[i]<A[i-1])ans2++;
		int cnt=sum(A[i]+1);
		ans1=ans1+i-cnt;
		add(A[i]+1,1);
	}
	//printf("%d %d\n",ans1,ans2);
	return ans1==ans2;
    }
};
