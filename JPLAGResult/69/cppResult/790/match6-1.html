class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        // the number of global inversion >= local inversion
        if (A.size() < 3)
            return true;
        int min_val = INT_MIN;
        for (int i=1; i<A.size(); i++) {
            if (A[i] < min_val)
                return false;
            min_val = max(min_val, A[i-1]);
        }
        return true;
    }
};
