class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        int sz = A.size();
        for (int i = 0; i < sz - 1; i++) {
            if (A[i] > A[i+1]) {
                if (A[i] != A[i+1] + 1 || A[i] != i+1)
                    return false;
            }
        }
        return true;
    }
};
