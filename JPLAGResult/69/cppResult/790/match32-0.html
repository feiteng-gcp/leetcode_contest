class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        int n = A.size();
        for (int i = 0; i < n; i++) {
            if (A[i] == i) { continue; }
            if (i == n - 1) { return false; }
            if (A[i] == i + 1 && A[i + 1] == i) {
                i++;
            } else {
                return false;
            }
        }
        return true;
    }
};
