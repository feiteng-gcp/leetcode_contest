class Solution {
public:
    int bit[5555];

    void add(int p) {
        while(p <= 5000) {
            bit[p] += 1;
            p += p & (-p);
        }
    }
    int query(int r) {
        int ans = 0;
        while(r) {
            ans += bit[r];
            r -= r & (-r);
        }
        return ans;
    }
    bool isIdealPermutation(vector<int>& A) {
        int local = 0, global = 0;
        for(int i = A.size() - 1; i >= 0; i--) {
            A[i] += 1;
            if(i != A.size() - 1 and A[i] > A[i + 1]) local++;
            global += query(A[i] - 1);
            add(A[i]);
        }
        return global == local;
    }
};
