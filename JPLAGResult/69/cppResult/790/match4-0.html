class Solution {
public:
  const int N = 10000;
  bool isIdealPermutation(vector<int> &A) {
    tr.resize(N);
    int local = 0;
    for (int i = 1; i < A.size(); ++i) {
      local += A[i - 1] > A[i];
    }
    int global = 0;
    for (int i = A.size() - 1; i >= 0; --i) {
      int x = A[i];
      ++x;
      global += get(x);
      add(x);
    }
    return global == local;
  }

  vector<int> tr;
  void add(int x) {
    for (; x < N; x += x & -x) {
      ++tr[x];
    }
  }

  int get(int x) {
    int s = 0;
    for (; x; x -= x & -x) {
      s += tr[x];
    }
    return s;
  }
};
