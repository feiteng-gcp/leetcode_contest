class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {

        for(int i=0;i+1<A.size();i++){
            if(A[i]>A[i+1]){
                swap(A[i],A[i+1]);
                i++;
            }
        }
        for(int i=0;i+1<A.size();i++){
            if(A[i]>A[i+1]){
                return false;
            }
        }
        return true;
    }
};
