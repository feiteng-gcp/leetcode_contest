class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        if(A.size()<=2) return true;
        int bench = A[0];
        for(int i=2;i<A.size();i++){
            if(A[i]<bench) return false;
            bench = max(bench,A[i-1]);
        }
        return true;
    }
};
