class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        int local = 0;
        for (int i = 0; i < A.size() - 1; i++) {
            if (A[i] > A[i + 1]) local ++;
        }
        int global = 0;
        global = mergeSort(A, A.size());
        
        if (global == local) return true;
        return false;
    }
private:
    int mergeSort(vector<int>& arr, int n) {
        int *temp = (int *)malloc(sizeof(int) * n);
        return M(arr, temp, 0, n - 1);
    }
    int M(vector<int>& arr, int temp[], int l, int r)
    {
        int mid, inv_count = 0;
        if (r > l) {
            mid = (l + r)/2;
            inv_count  = M(arr, temp, l, mid);
            inv_count += M(arr, temp, mid+1, r);
            inv_count += merge(arr, temp, l, mid+1, r);
         }
         return inv_count;
    }
    int merge(vector<int>& arr, int temp[], int l, int mid, int r) {
        int i, j, k;
        int inv_count = 0;
        i = l;
        j = mid;
        k = l;
        while ((i <= mid - 1) && (j <= r)) {
            if (arr[i] <= arr[j]) {
            temp[k++] = arr[i++];
            }
            else {
                temp[k++] = arr[j++];
                inv_count = inv_count + (mid - i);
            }
         }
         while (i <= mid - 1)
             temp[k++] = arr[i++];

         while (j <= r)
             temp[k++] = arr[j++];
         for (i = l; i <= r; i++)
             arr[i] = temp[i];
         return inv_count;
    }
};
