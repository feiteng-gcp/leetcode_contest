class Solution {
public:
    int n,x,t[5001];
    void upd(int idx, int val) {
        while (idx <= n) {
            t[idx] += val;
            idx += idx & -idx;
        }
    }

    int sum(int idx) {
        int mx = 0;
        while (idx > 0) {
            mx += t[idx];
            idx -= idx & -idx;
        }
        return mx;
    }

    
    bool isIdealPermutation(vector<int>& A) {
        n = A.size();
        int inv = 0;
        for(int i=1;i<=A.size();i++){
            t[i] = 0;
        }
        
        for(int i=A.size()-1;i>=0;i--){
            inv += sum(A[i]+1);
            upd(A[i]+1,1);
        }
        
        int x = 0;
        for(int i=0;i<A.size()-1;i++){
            if(A[i]>A[i+1]) x++;
        }
        // cout<<inv<<" "<<x<<endl;
        return (x==inv);
    }
};
