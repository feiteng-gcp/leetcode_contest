class Solution {
public:
    bool isIdealPermutation(vector<int>& A) {
        vector<int> S = A;
        sort(S.begin(), S.end());
        for(int i = 1; i < A.size(); ++i){
            if(A[i] < A[i-1]){
                if(A[i-1] == S[i] && A[i] == S[i-1])
                    continue;
                else
                    return false;
            }
        }
        return true;
    }
};
