class Solution {
    public:
            int numJewelsInStones(string J, string S) {
                set<char> stones;
                map<char,int> toFind;


                for(char stone:J){
                    stones.insert(stone);
                }

                int jew=0;
                for(char s:S){
                    if (stones.count(s)) {
                        ++jew;
                    }
                }

                return jew;

                        
                    
            }
};
