class Solution {
public:
    int numJewelsInStones(string J, string S) {

        unordered_set<char> ss;
        for (char ch : J) ss.insert(ch);
        
        int res = 0;
        for (char ch : S) {
            if (ss.find(ch) != ss.end()) ++res;
        }
        
        return res;
    }
};
