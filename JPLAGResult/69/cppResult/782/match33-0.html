class Solution {
public:
    int numJewelsInStones(string J, string S) {
        
        unordered_map<char, int> freq;
        for(char & ch : S){
            ++freq[ch];
        }
        int ans = 0;
        for(char & ch : J){
            ans += freq[ch];
        }
        return ans;
    }
};
