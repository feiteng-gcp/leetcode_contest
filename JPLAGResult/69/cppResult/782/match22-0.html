class Solution {
public:
  bool je[80];
    int numJewelsInStones(string J, string S) {
      memset(je,0,sizeof(je));
      for (int i=0; i < J.length(); ++i) {
        je[J[i]-'A']=true;
      }
      int ans = 0;
      for(int i = 0; i < S.length(); ++i) {
        if (je[S[i]-'A']) ans++;
      }
      return ans;
    }
};
