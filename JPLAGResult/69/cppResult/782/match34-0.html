class Solution {
public:
    int numJewelsInStones(string J, string S) {
        set<char> tab;
    int lenJ = (int)J.size(), lenS = (int)S.size();
    for(int i = 0; i < lenJ; i++) tab.insert(J[i]);
    int ans = 0;
    for(int i = 0; i < lenS; i++)
        if(tab.find(S[i]) != tab.end()) ans++;
    return ans;
    }
};
