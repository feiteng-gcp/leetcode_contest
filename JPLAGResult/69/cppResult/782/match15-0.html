class Solution {
public:
    int numJewelsInStones(string J, string S) {
        int lenJ=J.length();
        int lenS=S.length();
        int res=0;
        for(int i=0;i<lenS;i++){
            for(int j=0; j<lenJ; j++){
                if(J[j]==S[i]){
                    res++;
                    break;
                }
            }
        }
        return res;
    }
};
