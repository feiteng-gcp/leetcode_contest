class Solution {
public:
    int numJewelsInStones(string J, string S) 
    {
        unordered_set<char>Set;

        for (int i=0; i<J.size(); i++)
            Set.insert(J[i]);
        
        int result=0;
        for (int i=0; i<S.size(); i++)
            if (Set.find(S[i])!=Set.end())
                result++;
        return result;
    }
};
