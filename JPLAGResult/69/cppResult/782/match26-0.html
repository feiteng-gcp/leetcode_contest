class Solution {
public:
    int numJewelsInStones(string J, string S) {
        set<char> dp;
        for(char ch:J){
            dp.insert(ch);
        }
        int res = 0;
        for(char ch:S){
            if(dp.find(ch)!=dp.end()) res++;
        }
        return res;
    }
};
