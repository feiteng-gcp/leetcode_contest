class Solution {
public:
    int numJewelsInStones(string J, string S) {
        if(J.empty()||S.empty()) return 0;
        unordered_set<char> name_of_gems;
        for(char c:J) name_of_gems.insert(c);
        int num_of_gems = 0;
        for(char c:S){
            if(name_of_gems.count(c)==1) num_of_gems++;
        }
        return num_of_gems;
    }
};
