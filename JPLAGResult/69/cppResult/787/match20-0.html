class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        string cur = toStatus(board);
        string target = "123450";
        if (cur == target) return 0;
        queue<string> q;
        int level = 1;
        unordered_map<string, bool> um;
        um[cur] = true;
        q.push(cur);
        
        while (!q.empty()) {
            int cursize = q.size();
            for (int i = 0; i < cursize; i++) {
                string tmp = q.front();
                q.pop();
                vector<string> children = getchild(tmp);
                for (string c : children) {
                    if (um.find(c) == um.end()) {
                        um[c] = true;
                        if (c.compare(target) == 0) return level;
                        q.push(c);
                    }
                }
            }
            level += 1;
        }
        if (q.empty()) return -1;
    }
private:
    string toStatus(vector<vector<int>>& b) {
        string res;
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                res += b[i][j] + '0';
            }
        }
        return res;
    }
    vector<string> getchild(string& cur) {
        vector<string> vstr;
        int blank = 0;
        while (blank < 6 && cur[blank] != '0') blank += 1;
        if (blank != 0 && blank != 3) {
            swap(cur[blank-1],cur[blank]);
            vstr.push_back(cur);
            swap(cur[blank-1],cur[blank]);
        }
        if (blank != 2 && blank != 5) {
            swap(cur[blank+1],cur[blank]);
            vstr.push_back(cur);
            swap(cur[blank+1],cur[blank]);
        }
        if (blank - 3 >= 0) {
            swap(cur[blank-3],cur[blank]);
            vstr.push_back(cur);
            swap(cur[blank-3],cur[blank]);
        }
        if (blank + 3 < 6) {
            swap(cur[blank+3],cur[blank]);
            vstr.push_back(cur);
            swap(cur[blank+3],cur[blank]);
        }
        return vstr;
    }
};
