class Solution {
    const array<array<int, 2>, 4> dirs = {{{-1, 0}, {1, 0}, {0, 1}, {0, -1}}};
    
    int myHash(vector<vector<int>> &board) {
        int val = 0;
        for (int i=0; i<board.size(); i++) {
            for (int j=0; j<board[i].size(); j++) {
                val = 10 * val + board[i][j];
            }
        }
        return val;
    }
    
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        // memorization, BFS
        unordered_set<int> visited;
        unordered_map<int, int> cache;
        int target = 123450;
        visited.insert(myHash(board));
        
        queue<vector<vector<int>>> mqueue; // not efficient
        queue<pair<int, int>> indices;
        mqueue.push(board);
        
        for (int i=0; i<board.size(); i++) {
            for (int j=0; j<board[i].size(); j++) {
                if (board[i][j] == 0) {
                    indices.push(make_pair(i, j));
                    break;
                }
            }
        }
        
        int ans = 0;
        while (!mqueue.empty()) {
            queue<vector<vector<int>>> new_queue;
            while (!mqueue.empty()) {
                auto bo = mqueue.front();
                auto p  = indices.front();
                mqueue.pop();
                indices.pop();
                if (myHash(bo) == target)
                    return ans;
                for (int i=0; i<4; i++) {
                    int x = p.first + dirs[i][0], y = p.second + dirs[i][1];
                    if (x >= 0 && x < board.size() && y >= 0 && y < board[0].size()) {
                        bo[p.first][p.second] = bo[x][y];
                        bo[x][y] = 0;
                        int code = myHash(bo);
                        if (visited.find(code) == visited.end()) {
                            visited.insert(code);
                            new_queue.push(bo);
                            indices.push(make_pair(x, y));
                        }
                        bo[x][y] = bo[p.first][p.second];
                        bo[p.first][p.second] = 0;
                    }
                }
            }
            ans++;
            mqueue = new_queue;
        }
        
        return -1;
    }
};
