class Solution {
public:

    int serialize(vector<vector<int>> board){
        int res = 0;
        for(int i =0 ;i<2;i++) for(int j=0;j<3;j++) {res*=10; res+=board[i][j];}
        return res;
    }
    vector<int> getadj(int x){

        vector<vector<int>> board(2,vector<int>(3,0));
        int ix, iy;
        for(int i =1 ;i>=0;i--) 
            for(int j=2;j>=0;j--) {
                board[i][j] = x%10; x=x/10; 
                if(board[i][j]==0){ ix = i; iy = j;}
            }
        vector<int> adj;
        int dx[3] = {1,0,-1};
        for(int i = 0;i<3;i++){
            for(int j =0;j<3;j++){
                if((ix+dx[i]>=0) && (ix+dx[i]<2) && (iy+dx[j]>=0) &&(iy+dx[j]<3) && (abs(dx[i])+abs(dx[j])==1)){
                    auto temp = board;
                    swap(temp[ix][iy], temp[ix+dx[i]][iy+dx[j]]);
                    adj.push_back(serialize(temp));
                }
            }
        }
        return adj;
        
    }
    int slidingPuzzle(vector<vector<int>>& board) {
        
        unordered_map<int,int> visited;
        int begin = serialize(board);
        
        deque<pair<int,int>> Q;
        Q.push_back(make_pair(begin,0));
        visited[begin]=1;
        int goal = 123450;
        
        
        while(!Q.empty()){
            auto p = Q.front();
            Q.pop_front();
            int depth = p.second;
            int x = p.first;
            if(x==goal) return depth;
            auto adj = getadj(x);
            for(int a:adj){
                //cout<<a<<" ";
                if(visited.count(a)==0){
                    visited[a]=1;
                    Q.push_back(make_pair(a,depth+1));
                }
            }
            
        }
        
        return -1;
        
        
        
    }
};
