int dp[1000010];
int ta[2][3];
bool check(int x, int y) {
    if(x >= 0 && x < 2 && y >= 0 && y < 3) return 1;
    return 0;
}
int dx[] = {0, 0, 1, -1};
int dy[] = {1, -1, 0, 0};

int f() {
    int s = 0;
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            s = s * 10 + ta[i][j];
        }
    }
    return s;
}
class Solution {
public:

int slidingPuzzle(vector<vector<int>>& a) {
    memset(dp, -1, sizeof dp);
    int s = 0;
    int e = 123450;
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            s = s * 10 + a[i][j];
        }
    }
    queue<int> q;
    q.push(s);
    dp[s] = 0;
    //cout << s << endl;
    while(!q.empty()) {
        int u = q.front(); q.pop();
        //cout << u << endl;
        if(u == e) {
            return dp[e];
        }
        int t = u;
        for (int i = 1; i >= 0; i--) {
            for (int j = 2; j >= 0; j--) {
                ta[i][j] = t % 10;
                t /= 10;
            }
        }
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                if(ta[i][j] != 0) continue;
                for (int k = 0; k < 4; k++) {
                    int cx = i + dx[k], cy = j + dy[k];
                    if(check(cx, cy)) {
                        swap(ta[i][j], ta[cx][cy]);
                        int td = f();
                        //cout << td << endl;
                        swap(ta[i][j], ta[cx][cy]);
                        if(dp[td] == -1 || dp[td] > dp[u] + 1) {
                            dp[td] = dp[u] + 1;
                            q.push(td);
                        }
                    }
                }
            }
        }
    }
    return dp[e];
}
};
