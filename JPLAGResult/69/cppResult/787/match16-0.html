class Solution {
private:
  int getIndex(vector<vector<int>>& board) {
    return board[0][0] * 1 + board[0][1] * 6 + board[0][2] * 36 +
      board[1][0] * 216 + board[1][1] * 1296 + board[1][2] * 7776;
  }
  vector<vector<int>> direction = {{-1, 0}, {0, -1}, {0, 1}, {1, 0}};
public:
  int slidingPuzzle(vector<vector<int>>& board) {
    queue<vector<vector<int>>> q;
    vector<vector<int>> vgoal = {{1, 2, 3}, {4, 5, 0}};
    int goal = getIndex(vgoal);
    vector<int> dist(50000, -1);
    dist[getIndex(board)] = 0;
    q.push(board);
    while (!q.empty() && dist[goal] == -1) {
      vector<vector<int>> h = q.front();
      q.pop();
      int x, y;
      for (int i = 0; i < 2; ++i) {
        for (int j = 0; j < 3; ++j) {
          if (h[i][j] == 0) {
            x = i; y = j;
            break;
          }
        }
      }
      for (auto& d: direction) {
        int tx = x + d[0], ty = y + d[1];
        if (tx >= 0 && tx < 2 && ty >= 0 && ty < 3) {
          vector<vector<int>> next = h;
          swap(next[x][y], next[tx][ty]);
          if (dist[getIndex(next)] == -1) {
            dist[getIndex(next)] = dist[getIndex(h)] + 1;
            q.push(next);
          }
        }
      }
    }
    return dist[goal];
  }
};
