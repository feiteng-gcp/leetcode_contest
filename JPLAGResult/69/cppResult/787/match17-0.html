class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        set<vector<vector<int>>> v;
        v.insert(board);
        queue<vector<vector<int>>> q;
        q.push(board);
        vector<vector<int>> target = {{1,2,3},{4,5,0}};
        int dx[] = {1,-1,0,0};
        int dy[] = {0,0,1,-1};
        for (int i = 0; !q.empty(); i++) {
            for (int k = q.size(); k > 0; k--) {
                auto bod = q.front();
                q.pop();
                if (bod == target) {
                    return i;
                }
                for (int m = 0; m < 2; m++) {
                    for (int n = 0; n < 3; n++) {
                        if (bod[m][n] == 0) {
                            for (int p = 0; p < 4; p++) {
                                int m2 = m + dx[p];
                                int n2 = n + dy[p];
                                if (m2 < 0 || m2 > 1 || n2 < 0 || n2 > 2) {
                                    continue;
                                }
                                auto newbod = bod;
                                swap(newbod[m][n], newbod[m2][n2]);
                                if (v.count(newbod) == 0) {
                                    q.push(newbod);
                                    v.insert(newbod);
                                }
                            }
                            goto out;
                        }
                    }
                }
                out: continue;
            }
        }
        return -1;
    }
};
