class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        int ret=0;
        unordered_set<int> used;
        int init=board[0][0]*100000+board[0][1]*10000+board[0][2]*1000+board[1][0]*100+board[1][1]*10+board[1][2];
        vector<int> todo;
        todo.push_back(init);
        used.insert(init);
        while(!todo.empty()){
            vector<int> todo_next;
            while(!todo.empty()){
                int curr=todo.back();
                if(curr==123450){
                    return ret;
                }
                todo.pop_back();
                board[1][2]=curr%10;
                curr/=10;
                board[1][1]=curr%10;
                curr/=10;
                board[1][0]=curr%10;
                curr/=10;
                board[0][2]=curr%10;
                curr/=10;
                board[0][1]=curr%10;
                curr/=10;
                board[0][0]=curr%10;
                int x=-1;
                int y=-1;
                for(int i=0; i<=1; i++){
                    for(int j=0; j<=2; j++){
                        if(board[i][j]==0){
                            x=i;
                            y=j;
                            break;
                        }
                    }
                }
                // to up
                if(x!=0){
                    swap(board[x][y], board[x-1][y]);
                    int next=board[0][0]*100000+board[0][1]*10000+board[0][2]*1000+board[1][0]*100+board[1][1]*10+board[1][2];
                    if(used.find(next)==used.end()){
                        todo_next.push_back(next);
                        used.insert(next);
                    }
                    swap(board[x][y], board[x-1][y]);
                }
                // to down
                if(x!=1){
                    swap(board[x][y], board[x+1][y]);
                    int next=board[0][0]*100000+board[0][1]*10000+board[0][2]*1000+board[1][0]*100+board[1][1]*10+board[1][2];
                    if(used.find(next)==used.end()){
                        todo_next.push_back(next);
                        used.insert(next);
                    }
                    swap(board[x][y], board[x+1][y]);
                }
                // to left
                if(y!=0){
                    swap(board[x][y], board[x][y-1]);
                    int next=board[0][0]*100000+board[0][1]*10000+board[0][2]*1000+board[1][0]*100+board[1][1]*10+board[1][2];
                    if(used.find(next)==used.end()){
                        todo_next.push_back(next);
                        used.insert(next);
                    }
                    swap(board[x][y], board[x][y-1]);
                }
                // to right
                if(y!=2){
                    swap(board[x][y], board[x][y+1]);
                    int next=board[0][0]*100000+board[0][1]*10000+board[0][2]*1000+board[1][0]*100+board[1][1]*10+board[1][2];
                    if(used.find(next)==used.end()){
                        todo_next.push_back(next);
                        used.insert(next);
                    }
                    swap(board[x][y], board[x][y+1]);
                }
            }
            todo=todo_next;
            ret++;
        }
        return -1;
    }
};
