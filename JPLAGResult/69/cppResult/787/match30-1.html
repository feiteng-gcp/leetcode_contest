class Solution {
public:
    vector<vector<int>> swp =
    {
        {1, 3},
        {0, 2, 4},
        {1, 5},
        {0, 4},
        {1, 3, 5},
        {2, 4}
    };

    int slidingPuzzle(vector<vector<int>>& board) {
        vector<int> init = { board[0][0], board[0][1], board[0][2], board[1][0], board[1][1], board[1][2] };
        vector<int> tar = {1, 2, 3, 4, 5, 0};
        set<vector<int>> dp;
        dp.insert(init);
        queue<pair<int, vector<int>>> que;
        que.push({0, init});
        while (!que.empty())
        {
            int step = que.front().first;
            vector<int> cur = que.front().second;
            que.pop();
            if (cur == tar)
                return step;

            for (int i = 0; i < 6; ++i)
            {
                if (cur[i] != 0)
                    continue;
                for (int j = 0; j < swp[i].size(); ++j)
                {
                    vector<int> nxt = cur;
                    swap(nxt[i], nxt[swp[i][j]]);
                    if (!dp.count(nxt))
                    {
                        dp.insert(nxt);
                        que.push({step + 1, nxt});
                    }
                }
            }
        }
        return -1;
    }
};
