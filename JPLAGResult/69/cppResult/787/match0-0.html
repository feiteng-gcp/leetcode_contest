class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        int steps = 0;
        const vector<vector<int>> target = {{1,2,3}, {4,5,0}};
        
        set<vector<vector<int>>> visited;
        visited.insert(board);
        queue<vector<vector<int>>> q;
        q.push(board);
        
        while (!q.empty()) {
            int size = q.size();
            for (int index = 0; index < size; ++index) {
                const vector<vector<int>>& b = q.front();
                
                if (b == target) return steps;
                
                // 定位0之所在
                int i = 0, j = 0;
                for (int ind0 = 0; ind0 < 2; ++ind0) {
                    for (int ind1 = 0; ind1 < 3; ++ind1) {
                        if (b[ind0][ind1] == 0) {
                            i = ind0; j = ind1; break; // !
                        }
                    }
                }
                
                if (i-1 >= 0) {
                    vector<vector<int>> b2 = b;
                    swap(b2[i][j], b2[i-1][j]);
                    if (visited.find(b2) == visited.end()) {
                        visited.insert(b2);
                        q.push(b2);
                    }
                }
                
                if (i+1 <= 1) {
                    vector<vector<int>> b2 = b;
                    swap(b2[i][j], b2[i+1][j]);
                    if (visited.find(b2) == visited.end()) {
                        visited.insert(b2);
                        q.push(b2);
                    }                    
                }
                
                if (j-1 >= 0) {
                    vector<vector<int>> b2 = b;
                    swap(b2[i][j], b2[i][j-1]);
                    if (visited.find(b2) == visited.end()) {
                        visited.insert(b2);
                        q.push(b2);
                    }                    
                }
                
                if (j+1 <= 2) {
                    vector<vector<int>> b2 = b;
                    swap(b2[i][j], b2[i][j+1]);
                    if (visited.find(b2) == visited.end()) {
                        visited.insert(b2);
                        q.push(b2);
                    }                   
                }
                
                q.pop();

            }
            
            ++steps;
        }
        
        return -1;
    }
};
