const int N = 6 * 6 * 6 * 6 * 6 * 6 + 5;
int vis[N], dis[N], st = 11190;
queue<vector<vector<int>>> que;
int dx[] = {0, -1, 0, 1};
int dy[] = {1, 0, -1, 0};

int encode(vector<vector<int>>& b) {
    int ret = 0;
    for (int i = 0; i < 2; i++) for (int j = 0; j < 3; j++) ret = ret * 6 + b[i][j];
    return ret;
}

class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        memset(vis, 0, sizeof(vis));
        memset(dis, 0, sizeof(dis));
        while (!que.empty()) que.pop();
        vis[encode(board)] = 1;
        dis[encode(board)] = 0;
        que.push(board);
        while (!que.empty()) {
            vector<vector<int>> cur = que.front(); que.pop();
            if (encode(cur) == st) return dis[encode(cur)];
            int cx, cy;
            for (int i = 0; i < 2; i++) for (int j = 0; j < 3; j++) if (cur[i][j] == 0) cx = i, cy = j;
            for (int i = 0; i < 4; i++) {
                int nx = cx + dx[i], ny = cy + dy[i];
                if (0 <= nx && nx < 2 && 0 <= ny && ny < 3) {
                    vector<vector<int>> now = cur;
                    swap(now[cx][cy], now[nx][ny]);
                    if (!vis[encode(now)]) {
                        vis[encode(now)] = 1;
                        dis[encode(now)] = dis[encode(cur)] + 1;
                        que.push(now);
                    }
                }
            }
        }
        return -1;
    }
};
