typedef vector<vector<int>> Grid;
class Solution {


    public:
        int slidingPuzzle(vector<vector<int>>& board) {

            Grid dest={{1,2,3},{4,5,0}};

            map<Grid,int> reached;
            reached.insert({board,0});
            pair<int ,int >  zeroPos;
            bool foundZero=false;
            for(int y=0;(y<2)&&!foundZero;++y){
                for(int x=0;x<3;++x){
                    if (board[y][x]==0) {
                        zeroPos.first=y;
                        zeroPos.second=x;
                        foundZero=true;
                        break;
                    }
                }
            }

            int currentMin=INT_MAX;

            recuFind(board,dest,zeroPos,reached,0,currentMin);
            
            if(currentMin==INT_MAX) return -1;
            return currentMin;

        }

        bool inRange(int y,int x){
            return (x>=0) && (y>=0) && (x<3) && (y<2);
        }

        void recuFind(Grid& from,const Grid& dest,const pair<int,int> zeroPos,map<Grid,int>& reached, const int currentStep, int & currentMin){

            if (from==dest) {

                currentMin=min(currentMin,currentStep);

                return;
            }

            if(currentStep>currentMin){
                return;
            }



            int offset[4][2]={
                {0,1},{0,-1},{1,0},{-1,0}
            };

            for(int i=0;i<4;++i){

                pair<int,int> swapPos= {zeroPos.first+offset[i][0],zeroPos.second+offset[i][1]};

                if (!inRange(swapPos.first,swapPos.second)) {
                    continue;
                }


                Grid swapped (from);

                swap(swapped[zeroPos.first][zeroPos.second],swapped[swapPos.first][swapPos.second]);

                int thisSteps=currentStep+1;
                auto pos =reached.find(swapped);
                if((pos!=reached.end())&& (pos->second<thisSteps)){
                    continue;
                }

                reached[swapped]=thisSteps;

                recuFind(swapped,dest,swapPos,reached,currentStep+1,currentMin);
            }
        }
};
