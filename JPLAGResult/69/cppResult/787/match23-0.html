int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};

class Solution {
    long long kk(vector<vector<int>> & g)
    {
        long long k = 0;
        
        for(int i = 0; i < g.size(); ++i)
        {
            for(int j = 0; j < g[0].size(); ++j)
            {
                k = k * 10 + g[i][j];
            }
        }
        
        return k;
    }
    
    pair<int, int> tog(long long x, vector<vector<int>> & g)
    {
        pair<int, int> z;
        for(int i = g.size() - 1; i >= 0; --i)
        {
            for(int j = g[0].size() - 1; j >= 0; --j)
            {
                g[i][j] = x % 10;
                x /= 10;
                
                if (!g[i][j])
                    z = {i, j};
            }
        }
        return z;
    }
    
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        unordered_map<long long, int> dist;
        queue<long long> q;
        
        long long x = kk(board);
        if (x == 123450) return 0;
        
        q.push(x);
        dist[x] = 0;
        
        while(!q.empty())
        {
            auto y = q.front(); q.pop();
            
            vector<vector<int>> tt(2, vector<int>(3));
            
            auto p = tog(y, tt);
            for(int i = 0; i < 4; ++i)
            {
                int nx = p.first + dx[i], ny = p.second + dy[i];
                if (nx < 0 || nx >= 2 || ny < 0 || ny >= 3)
                    continue;
                
                vector<vector<int>> xx = tt;
                int z = xx[nx][ny];
                xx[nx][ny] = 0;
                xx[p.first][p.second] = z;
                
                long long nk = kk(xx);
                
                if (nk == 123450) return dist[y] + 1;
                
                if (dist.count(nk)) continue;
                dist[nk] = dist[y] + 1;
                q.push(nk);
            }
        }
        
        return -1;
    }
};
