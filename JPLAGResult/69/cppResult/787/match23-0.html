class Solution {
public:
	int slidingPuzzle(vector<vector<int>>& board) {
		string s = to_string(board[0][0]) + to_string(board[0][1]) + to_string(board[0][2]) + to_string(board[1][0]) + to_string(board[1][1]) + to_string(board[1][2]);
		queue<pair<string, int>> q;
		set<string> found;
		int a, b, i;
		string s2;
		vector<vector<int>> v = { { 1,3}, {0,2,4}, {1,5}, {0, 4}, {1,3,5}, {2,4} };

		q.push({ s, 0 });
		found.insert(s);

		while (!q.empty()) {
			s = q.front().first;
			a = q.front().second;
			q.pop();
			if (s == "123450") return a;
			for (i = 0; i < 6; i++) {
				if (s[i] == '0') {
					for (auto p : v[i]) {
						s2 = s;
						s2[p] = '0';
						s2[i] = s[p];
						if (found.find(s2) == found.end()) q.push({ s2, a + 1 });
						found.insert(s2);
					}
				}
			}

		}
		return -1;
	}
};
