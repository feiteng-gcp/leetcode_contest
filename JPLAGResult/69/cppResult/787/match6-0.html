class Solution {
public:
    vector<vector<int>> dict4={{1,0},{-1,0},{0,1},{0,-1}};
    int slidingPuzzle(vector<vector<int>>& board) {
    queue<int> q;
    int code=getHash(board);
    unordered_set<int> hset;
    int finalCode=123450;
    if(code==finalCode)
        return 0;
    int ans=1;
    q.push(code);
    hset.insert(code);
    int x=0,y=0;
    int thislevel=1;
    int nextlevel=0;
    while(!q.empty()){
        if(thislevel==0){
            ++ans;
            thislevel=nextlevel;
            nextlevel=0;
        }
        code=q.front();
        q.pop();
        --thislevel;
        getBoard(board,code);
        for(int i=0;i<2;++i){
            for(int j=0;j<3;++j){
                if(board[i][j]==0){
                    x=i;
                    y=j;
                    break;
                }
            }
        }

        for(int i=0;i<4;++i){
            int newx=x+dict4[i][0];
            int newy=y+dict4[i][1];
            if(newx<0||newx>=2||newy<0||newy>=3)
                continue;
            swap(board[x][y],board[newx][newy]);
            code=getHash(board);

            if(code==finalCode)
                return ans;
            if(hset.count(code)==0){
                hset.insert(code);
                q.push(code);
                ++nextlevel;
               // cout<<code<<endl;
               // show(board);
            }
            swap(board[x][y],board[newx][newy]);
        }
    }
    return -1;
}
    int getHash(vector<vector<int>>& board){
    int code=0;
    for(int i=0;i<2;++i){
        for(int j=0;j<3;++j){
            code=code*10+board[i][j];
        }
    }
    return code;
}
void getBoard(vector<vector<int>>& board,int code){
    for(int i=1;i>=0;--i){
        for(int j=2;j>=0;--j){
            board[i][j]=code%10;
            code/=10;
        }
    }
}
};
