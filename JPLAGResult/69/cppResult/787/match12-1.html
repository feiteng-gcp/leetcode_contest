class Solution {
    map<int, int> st;
public:
    int fold(vector<vector<int>>& board) {
        int res = 0;
        for(auto u : board) {
            for(auto v : u) {
                res *= 10;
                res += v;
            }
        }
        return res;
    }
    typedef vector<vector<int> > vvi;
    int dx[4] = { 0, 1, 0, -1 };
    int dy[4] = { 1, 0,-1,  0 };
    int slidingPuzzle(vector<vector<int>>& board) {
        st[fold(board)] = 0;
        queue<vvi> q;
        q.push(board);
        while(!q.empty()) {
            auto t = q.front(); q.pop();
            int thash = fold(t);
            int x, y;
            for(int i = 0; i < 2; i++) for(int j = 0; j < 3; j++) if(t[i][j] == 0) { x = i; y = j; }
            for(int i = 0; i < 4; i++) {
                int nx = x + dx[i], ny = y + dy[i];
                if(0 <= nx and nx < 2 and 0 <= ny and ny < 3) {
                    swap(t[x][y], t[nx][ny]);
                    int hash = fold(t);

                    if(st.find(hash) == st.end()) {
                        q.push(t);
                        st[hash] = st[thash] + 1;
                    }
                    swap(t[x][y], t[nx][ny]);
                }
            }
        }
        if(st.find(123450) != st.end()) {
            return st[123450];
        }
        return -1;
    }
};
