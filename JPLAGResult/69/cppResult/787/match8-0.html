typedef vector<vector<int>> item;
const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        queue<item> Q;
        map<item, int> dp;
        Q.push(board);
        dp[board] = 0;
        int n = 2, m = 3;
        while (!Q.empty()) {
            auto a = Q.front();
            int cur = dp[a];
            Q.pop();
            int x, y;
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    if (a[i][j] == 0) {
                        x = i;
                        y = j;
                    }
                }
            }
            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0];
                int ny = y + d[k][1];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
                swap(a[x][y], a[nx][ny]);
                if (!dp.count(a)) {
                    dp[a] = cur + 1;
                    Q.push(a);
                }
                swap(a[x][y], a[nx][ny]);
            }   
        }
        item target = {{1, 2, 3}, {4, 5, 0}};
        if (dp.count(target)) return dp[target];
        return -1;
        
    }
};
