map<vector<vector<int> >, int> mp;

int neigh[4][2]={{0,1},{1,0},{0,-1},{-1,0}};

class Solution {
public:
    int slidingPuzzle(vector<vector<int> >& board) {
      int i,j,k,b1,l,o,p,x,y,z;
      vector<vector<int> > a;
      vector<int> tmp;
      vector<vector<vector<int> > > q;
      tmp.clear();
      tmp.push_back(1);
      tmp.push_back(2);
      tmp.push_back(3);
      a.push_back(tmp);
      tmp.clear();
      tmp.push_back(4);
      tmp.push_back(5);
      tmp.push_back(0);
      a.push_back(tmp);
      mp.clear();
      mp[a]=0;
      q.clear();
      q.push_back(a);
      for (i=0;i<q.size();i++)
      {
        a=q[i];
        z=mp[a];
        b1=0;
        for (j=0;j<2;j++)
        {
          for (k=0;k<3;k++)
          {
            if (a[j][k]==0)
            {
              b1=1;break;
            }
          }
          if (b1==1) break;
        }
        for (o=0;o<4;o++)
        {
          x=j+neigh[o][0];
          y=k+neigh[o][1];
          if ((x>=0)&&(x<2)&&(y>=0)&&(y<3))
          {
            swap(a[j][k],a[x][y]);
            if (mp.find(a)==mp.end())
            {
              mp[a]=z+1;
              q.push_back(a);
            }
            swap(a[j][k],a[x][y]);
          }
        }
      }
      if (mp.find(board)==mp.end()) return -1;
      else return mp[board];
    }
};
