class Solution {
public:
    struct fff
{
	int a[6];
};
queue<fff> q;
int steps[6][6][6][6][6][6];

    int slidingPuzzle(vector<vector<int>>& board) {
        fff news;
	int cur=0;
	for(int i=0;i<2;i++)
	{
		for(int j=0;j<3;j++)
		{
			news.a[cur++]=board[i][j];
		}
	}
	memset(steps,-1,sizeof(steps));
	steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]=0;
	q.push(news);
	while(!q.empty())
	{
		if(steps[1][2][3][4][5][0]!=-1)return steps[1][2][3][4][5][0];
		fff news=q.front();
		q.pop();
		int cur=steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]];
		int p;
		for(int i=0;i<6;i++)
		{
			if(!news.a[i])
			{
				p=i;
				break;
			}
		}
		if(p%3<2)
		{
			int tmp=news.a[p];
			news.a[p]=news.a[p+1];
			news.a[p+1]=tmp;
			if(steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]==-1)
			{steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]=cur+1;
			q.push(news);}
			tmp=news.a[p];
			news.a[p]=news.a[p+1];
			news.a[p+1]=tmp;
		}
		if(p%3)
		{
			int tmp=news.a[p];
			news.a[p]=news.a[p-1];
			news.a[p-1]=tmp;
			if(steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]==-1)
			{steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]=cur+1;
			q.push(news);}
			tmp=news.a[p];
			news.a[p]=news.a[p-1];
			news.a[p-1]=tmp;
		}
		if(p<3)
		{
			int tmp=news.a[p];
			news.a[p]=news.a[p+3];
			news.a[p+3]=tmp;
			if(steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]==-1)
			{steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]=cur+1;
			q.push(news);}
			tmp=news.a[p];
			news.a[p]=news.a[p+3];
			news.a[p+3]=tmp;
		}
		if(p>=3)
		{
			int tmp=news.a[p];
			news.a[p]=news.a[p-3];
			news.a[p-3]=tmp;
			if(steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]==-1)
			{steps[news.a[0]][news.a[1]][news.a[2]][news.a[3]][news.a[4]][news.a[5]]=cur+1;
			q.push(news);}
			tmp=news.a[p];
			news.a[p]=news.a[p-3];
			news.a[p-3]=tmp;
		}
	}
	return -1;
    }
};
