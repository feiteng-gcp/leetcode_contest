class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& b) {
        queue<vector<vector<int>>> q;
        unordered_set<string> v;
        string s1 = b2s(b);
        
        vector<vector<int>> b0 = {{1, 2, 3}, {4, 5, 0}};
        string s0 = b2s(b0);
        if(s0 == s1) return 0;
        
        q.push(b0);
        v.insert(s0);
        
        int d = 1;
        int direct[4][2] = {{1, 0}, {-1, 0}, {0, -1}, {0, 1}};
        
        while(!q.empty()) {
            int len = q.size();
            for(int i = 0; i < len; ++i) {
                auto cur = q.front();
                q.pop();
                
                
                int x = 0, y = 0;
                

                for(int e = 0; e < 2; ++e) {
                    for(int f = 0; f < 3; ++f) {
                        if(cur[e][f] == 0) {
                            x = e;
                            y = f;
                        }
                    }    
                }
                
                for(int k = 0; k < 4; ++k) {
                    int xx = x + direct[k][0];
                    int yy = y + direct[k][1];
                    if(xx < 0 || xx >= 2 || yy < 0 || yy >= 3) continue;
                    auto next = cur;
                    swap(next[xx][yy], next[x][y]);
                    string s = b2s(next);
                    if(s == s1) return d;
                    if(v.find(s) == v.end()) {
                        v.insert(s);
                        q.push(next);
                    }
                }
                
            }
            ++d;
        }
        return -1;
    }
    
    string b2s(vector<vector<int>>& b) {
        string res;

        for(int i = 0; i < 2; ++i) {
            for(int j = 0; j < 3; ++j) {
                res.push_back(b[i][j] + '0');
            }
        }
        return res;
    }
};
