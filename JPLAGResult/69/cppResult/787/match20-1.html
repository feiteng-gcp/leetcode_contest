class Solution {
    unordered_set<int> dp;
    int convert(vector<vector<int>>& board) {
        int tmp = 0;

        for (int i = 0; i < 2; ++i) {
            for (int j = 0; j < 3; ++j) {
                tmp = tmp * 10 + board[i][j];
            }
        }
        return tmp;
    }

    int dx[4] = {0, 0, -1, 1};
    int dy[4] = {1, -1, 0, 0};
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        int terminal = 123450;
        queue<pair<vector<vector<int>>, int>> qq;
        qq.push({board, 0});
        while (!qq.empty()) {
            auto cur = qq.front();
            qq.pop();

            vector<vector<int>> vv = cur.first;
            int id = convert(vv);
            if (id == terminal) return cur.second;
            if (dp.find(id) != dp.end()) {
                continue;
            }
            dp.insert(id);
            int row, col;
            bool found = false;
            for (int i = 0; i < 2; ++i) {
                for (int j = 0; j < 3; ++j) {
                    if (vv[i][j] == 0) {
                        found = true;
                        row = i;
                        col = j;
                        break;
                    }
                }
                if (found) break;

            }

            for (int i = 0; i < 4; ++i) {
                int x = row + dx[i];
                int y = col + dy[i];

                if (x < 0 || x >= 2 || y < 0 || y >= 3) continue;
                swap(vv[row][col], vv[x][y]);
                qq.push({vv, cur.second + 1});
                swap(vv[row][col], vv[x][y]);
            }
        }

        return -1;
    }
};
