class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        set<int> store;
        vector<vector<vector<int>>> curr, next;
        curr.push_back(board);
        int step = 0;
        while (!curr.empty()) {
            for (auto i : curr) {
                if (correct(i))
                    return step;
                neighbor(i, next, store);
            }
            curr = next;
            next.clear();
            step++;
        }
        return -1;
    }
    
    int comHash(vector<vector<int>>& board) {
        int res = 0;
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                res = res * 10 + board[i][j];
            }
        }
        return res;
    }
    bool correct(vector<vector<int>>& board) {
        if (comHash(board) == 123450) {
            return true;
        } else {
            return false;
        }
    }
    void neighbor(vector<vector<int>>& board, vector<vector<vector<int>>>& next, set<int>& store) {
        store.insert(comHash(board));
        int i, j;
        for (i = 0; i < 2; i++) {
            for (j = 0; j < 3; j++) {
                if (board[i][j] == 0) {
                    int row = i ^ 1;
                    swap(board[i][j], board[row][j]);
                    if (store.find(comHash(board)) == store.end()) {
                        next.push_back(board);
                    }
                    swap(board[i][j], board[row][j]);
                    
                    if (j > 0) {
                        int col = j - 1;
                        swap(board[i][col], board[i][j]);
                    if (store.find(comHash(board)) == store.end()) {
                        next.push_back(board);
                    }
                    swap(board[i][col], board[i][j]);
                    }
                    
                    if (j < 2) {
                        int col = j + 1;
                        swap(board[i][col], board[i][j]);
                    if (store.find(comHash(board)) == store.end()) {
                        next.push_back(board);
                    }
                    swap(board[i][col], board[i][j]);
                    }
                    
                    break;
                }
            }
        }
    }
        
        
};
