class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        int di[4] = {0, -1, 0, +1};
        int dj[4] = {-1, 0, +1, 0};
        map<vector<vector<int>>, int> dis;
        queue<vector<vector<int>> > que;
        que.push(board);
        dis[board] = 0;
        while (!que.empty()) {
            auto tbl = que.front(); que.pop();
            for (int i = 0; i < 2; i++)
            for (int j = 0; j < 3; j++) if (tbl[i][j] == 0) {
                for (int d = 0; d < 4; d++) {
                    auto nxt = tbl;
                    int ii = i + di[d];
                    int jj = j + dj[d];
                    if (ii >= 0 && ii < 2 && jj >= 0 && jj < 3) {
                        swap(nxt[i][j], nxt[ii][jj]);
                        if (!dis.count(nxt)) {
                            dis[nxt] = dis[tbl] + 1;
                            que.push(nxt);
                        }
                    }
                }
            }
        }
        vector<vector<int>> t = {{1, 2, 3}, {4, 5, 0}};
        return dis.count(t) ? dis[t] : -1;
    }
};
