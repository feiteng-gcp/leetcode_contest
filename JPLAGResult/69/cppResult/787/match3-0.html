class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        unordered_set<string> visited;
        queue<vector<vector<int>>> q;
        int x, y;
        int step = 0;
        
        for (int i = 0; i < board.size(); ++i) {
            for (int j = 0; j < board[0].size(); ++j) {
                if (board[i][j] == 0) {
                    x = i;
                    y = j;
                } // if
            } // for
        } // for
        
        vector<int> dx = {0, 0, -1, 1};
        vector<int> dy = {-1, 1, 0, 0};
        
        q.push(board);
        while (!q.empty()) {
            int qSize = q.size();
            for (int idx = 0; idx < qSize; ++idx) {
                vector<vector<int>> brd = q.front();
                q.pop();
                
                string key = "";
                for (int i = 0; i < brd.size(); ++i) {
                    for (int j = 0; j < brd[0].size(); ++j) { 
                        key += to_string(brd[i][j]);
                        if (brd[i][j] == 0) {
                            x = i;
                            y = j;
                        } // if
                    } // for
                } // for
                
                if (visited.find(key) != visited.end())
                    continue;
                if (key == "123450") 
                    return step;
                
                visited.insert(key);
                
                for (int k = 0; k < 4; ++k) {
                    int xNew = x + dx[k];
                    int yNew = y + dy[k];
                    if (xNew < 0 || xNew == board.size() || yNew < 0 || yNew == board[0].size())
                        continue;
            
                    swap(brd[x][y], brd[xNew][yNew]);
                    q.push(brd);
                    swap(brd[x][y], brd[xNew][yNew]);
                } // for
            
            } // for
            
            ++step;
        } // while
        
        return -1;
    }
};
