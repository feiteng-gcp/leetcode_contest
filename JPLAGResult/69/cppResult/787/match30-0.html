class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        int src = encode(board);
        vector<vector<int>> targetBoard = {{1, 2, 3}, {4, 5, 0}};
        int target = encode(targetBoard);
        const vector<int> DIR_X = {0, 1, 0, -1};
        const vector<int> DIR_Y = {1, 0, -1, 0};
        queue<int> q;
        unordered_map<int, int> dists;
        unordered_map<int, bool> visited;
        q.push(src);
        dists[src] = 0;
        visited[src] = true;
        while (!q.empty()) {
            int u = q.front();
            auto board = decode(u);
            q.pop();
            int x = -1, y = -1;
            for (int i = 0; i < 2; ++i) {
                for (int j = 0; j < 3; ++j) {
                    if (board[i][j] == 0) {
                        x = i;
                        y = j;
                    }
                }
            }
            for (int k = 0; k < 4; ++k) {
                int tx = x + DIR_X[k];
                int ty = y + DIR_Y[k];
                if (0 <= tx && tx < 2) {
                    if (0 <= ty && ty < 3) {
                        swap(board[x][y], board[tx][ty]);
                        int v = encode(board);
                        if (dists.count(v) == 0 || dists[v] > dists[u] + 1) {
                            dists[v] = dists[u] + 1;
                            if (!visited[v]) {
                                visited[v] = true;
                                q.push(v);
                            }
                        }
                        swap(board[x][y], board[tx][ty]);
                    }
                }
            }
            visited[u] = false;
        }
        return dists.count(target) == 1 ? dists[target] : -1;
    }

private:
    int encode(vector<vector<int>>& board) {
        int state = 0;
        for (size_t i = 0; i < board.size(); ++i) {
            for (size_t j = 0; j < board[i].size(); ++j) {
                state = state * 6 + board[i][j];
            }
        }
        return state;
    }

    vector<vector<int>> decode(int state) {
        vector<vector<int>> board(2, vector<int>(3));
        for (int i = 1; i >= 0; --i) {
            for (int j = 2; j >= 0; --j) {
                board[i][j] = state % 6;
                state /= 6;
            }
        }
        return board;
    }
};
