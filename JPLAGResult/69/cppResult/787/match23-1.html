class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        string s = "000000";
        for (int i = 0; i < 2; ++i)
            for (int j = 0; j < 3; ++j)
                s[i*3+j] = '0' + board[i][j];
        if (s == "123450")
            return 0;
        set<string> seen;
        seen.insert(s);
        vector<pair<string, int> > v;
        v.push_back(make_pair(s, 0));
        int idx = 0;

        vector<vector<int>> m = {{1, 3}, {0, 2, 4}, {1, 5}, {0, 4}, {1, 3, 5}, {2, 4}};
        while (idx < v.size()) {
            s = v[idx].first;
            for (int i = 0; i < s.length(); ++i) {
                if (s[i] != '0')
                    continue;
                for (int j = 0; j < m[i].size(); ++j) {
                    string ns = s;
                    ns[i] = ns[m[i][j]];
                    ns[m[i][j]] = '0';
                    if (ns == "123450")
                        return v[idx].second + 1;
                    if (seen.find(ns) == seen.end()) {
                        seen.insert(ns);
                        v.push_back(make_pair(ns, v[idx].second + 1));
                    }
                }
            }
            idx++;
        }
        return -1;
    }
};
