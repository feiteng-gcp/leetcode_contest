class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        string curr;
        queue<string> myQue;
        for (int i = 0; i < board.size(); i++)
            for (int j = 0; j < board[0].size(); j++)
                curr.push_back(board[i][j]+'0');
        
        string s = "123450";
        if (curr == s)
            return 0;
        
        myQue.push(curr);
        int count = 0;
        unordered_set<string> mySet;
        while (!myQue.empty()) {
            count++;
            int currS = myQue.size();
            for (int i = 0; i < currS; i++) {
                curr = myQue.front();
                myQue.pop();
                mySet.insert(curr);
                vector<string> result = swap(curr);
                for (string temp:result) {
                    if (temp == s)
                        return count;
                    if (!mySet.count(temp))
                        myQue.push(temp);
                }                    
            }            
        }
        return -1;        
    }
    
    vector<string> swap (string curr) {
        vector<string> result;
        vector<vector<int>> pos{{1,3},{0,4,2},{1,5},{0,4},{3,5,1},{2,4}};
        for (int i = 0; i < 6; i++) {
            if (curr[i] == '0') {
                for (int j = 0; j < pos[i].size(); j++) {
                    mySwap(curr,i,pos[i][j]);
                    result.push_back(curr);
                    mySwap(curr,i,pos[i][j]);
                }
                break;
            }
        }
        return result;
    }
    
    void mySwap(string &curr, int i, int j) {
        char c = curr[i];
        curr[i] = curr[j];
        curr[j] = c;
    }
};
