class Solution {
    int encode(vector<vector<int>>& board) {
        int ret = 0;
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                ret = ret * 10 + board[i][j];
            }
        }
        return ret;
    }
    vector<vector<int> > decode(int val) {
        vector<vector<int> > ret(2, vector<int>(3, 0));
        for (int i = 1; i >= 0; i--) {
            for (int j = 2; j >= 0; j--) {
                ret[i][j] = val % 10;
                val /= 10;
            }
        }
        return ret;
    }
    void move(queue<int> &q, vector<vector<int>>& tmp, unordered_set<int> & vis, int i, int j, int xi, int xj) {
        swap(tmp[xi][xj], tmp[i][j]);
        int val = encode(tmp);
        if (vis.find(val) == vis.end()) {
            vis.insert(val);
            q.push(val);
        }
        swap(tmp[xi][xj], tmp[i][j]);
    }
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        int start = encode(board);
        queue<int> q; q.push(start);
        unordered_set<int> vis; vis.insert(start);
        int end = 123450;
        int ret = -1;
        while(!q.empty()) {
            int sz = q.size(); ret++;
            for (int k = 0; k < sz; k++) {
                int cur = q.front(); q.pop();
                if (cur == end) return ret;
                auto tmp = decode(cur);
                int i = -1, j = -1;
                for (int a = 0; a < 2 && i==-1; a++) {
                    for (int b = 0; b < 3 && i==-1; b++) {
                        if (tmp[a][b] == 0) {
                            i = a, j = b;
                            break;
                        }
                    }
                }
                if (i+1 < 2) move(q, tmp, vis, i, j, i+1, j); 
                if (i-1 >= 0) move(q, tmp, vis, i, j, i-1, j);
                if (j+1 < 3) move(q, tmp, vis, i, j, i, j+1);
                if (j-1 >= 0) move(q, tmp, vis, i, j, i, j-1);
            }
        }
        return -1;
        
        
        
    }
};
