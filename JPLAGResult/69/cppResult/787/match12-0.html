class Solution {
public:

    int encode(const vector<vector<int>>& a) {
        int ret = 0;
        for (int i=0; i<2; ++i)
            for (int j=0; j<3; ++j) 
                ret = ret*10 + a[i][j];
        return ret;
    }
    
    vector<vector<int>> decode(int x) {
        auto ret = vector<vector<int>>(2, vector<int>(3, 0));
        for (int i=1; i>=0; --i)
            for (int j=2; j>=0; --j) {
                ret[i][j] = x % 10;
                x /= 10;
            }
        return ret;
    }
    
    vector<int> dx = {-1, 0, 1, 0};
    vector<int> dy = {0, 1, 0, -1};
    
    int slidingPuzzle(vector<vector<int>>& board) {
        queue<pair<int,int>> q; // state, dis
        int target = encode(vector<vector<int>>{{1,2,3},{4,5,0}});
        int start = encode(board);
        if (target==start)
            return 0;
        q.push({start, 0});
        unordered_set<int> vis;
        vis.insert(start);

        
        while (!q.empty()) {
            auto state = decode(q.front().first);
            int dis   = q.front().second;
            q.pop();
            
            int x, y;
            for (int i=0; i<2; ++i)
                for (int j=0; j<3; ++j)
                    if (state[i][j]==0) {
                        x = i, y = j;
                        break;
                    }
            
            for (int dir=0; dir<4; ++dir) {
                int tx=x+dx[dir], ty=y+dy[dir];
                if (tx>=0 && tx<2 && ty>=0 && ty<3) {
                    auto after = state;
                    swap(after[x][y], after[tx][ty]);
                    int nxt = encode(after);
                    // cout << nxt << endl;
                    if (vis.find(nxt)==vis.end()) {
                        q.push({nxt, dis+1});
                        vis.insert(nxt);
                        if (nxt==target) 
                            return dis+1;
                    }
                }
            }    
        }
        
        return -1;
    }
};
