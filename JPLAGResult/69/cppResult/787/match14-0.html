const int n = 2;
const int m = 3;

using board = vector<vector<int>>;

board target = {{1,2,3},{4,5,0}};
bool solved(const board& a) {
    return a == target;
}

int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};
class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& a) {
        set<board> visited;
        deque<pair<board, int>> qu = {{a,0}};
        while (!qu.empty()) {
            auto a = qu.front().first;
            int d = qu.front().second;
            qu.pop_front();
            if (visited.find(a) != visited.end()) continue;
            visited.insert(a);
            if (solved(a)) return d;
            int x, y;
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    if (a[i][j] == 0) {
                        x = i;
                        y = j;
                    }
                }
            }
            for (int k = 0; k < 4; ++k) {
                int nx = x+dx[k];
                int ny = y+dy[k];
                if (nx >= 0 and nx < n and ny >= 0 and ny < m) {
                    auto b = a;
                    swap(b[x][y], b[nx][ny]);
                    if (visited.find(b) == visited.end()) {
                        qu.push_back({b, d+1});
                    }
                }
            }
        }
        return -1;
    }
};
