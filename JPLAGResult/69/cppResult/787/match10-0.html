class Solution {
public:
    int slidingPuzzle(vector<vector<int> > & board)
{
    int ori = 0, bit = 4, r = 2, c = 3;
    int mask = (1 << bit) - 1;
    int ans = 1 | (2 << bit) | (3 << (bit * 2)) | (4 << (bit * 3)) | (5 << (bit * 4));
    for(int i = 0; i < r; i++)
        for(int j = 0; j < c; j++)
            ori |= board[i][j] << (bit * (c * i + j));
    map<int, int> tab;
    tab[ori] = 0;
    queue<int> q;
    q.push(ori);
    
    const int dx[4] = {0, 0, 1, -1}, dy[4] = {1, -1, 0, 0};
    while(!q.empty())
    {
        int cur = q.front(); q.pop();
        if(cur == ans) return tab[cur];
        int x = 0, y = 0;
        for(int i = 0; i < r * c; i++)
            if(((cur >> (bit * i)) & mask) == 0)
            {
                x = i / c;
                y = i % c;
            }
        for(int i = 0; i < 4; i++)
        {
            int nx = x + dx[i], ny = y + dy[i];
            if(nx >= 0 && nx < r && ny >= 0 && ny < c)
            {
                int nxt = cur, n = (cur >> (bit * (c * nx + ny))) & mask;
                nxt &= ~(mask << (bit * (c * nx + ny)));
                nxt |= n << (bit * (c * x + y));
                if(tab.find(nxt) == tab.end())
                {
                    tab[nxt] = tab[cur] + 1;
                    q.push(nxt);
                }
            }
        }
    }
    return -1;
}
};
