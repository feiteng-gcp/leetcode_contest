class Solution {
public:
    unordered_map<string, int> m;
    string convert(vector<vector<int>>& board) {
        string res = "";
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                res.push_back('0' + board[i][j]);
            }
        }
        return res;
    }
    void swap(int& a, int& b) {
        int temp = a;
        a = b;
        b = temp;
    }
    void update(vector<vector<int>>& board, int depth) {
        string s = convert(board);
        if (m.find(s) == m.end()) {
            m[s] = depth;
        } else if (m[s] > depth) {
            m[s] = depth;
        } else {
            return ;
        }
        int ii;
        int jj;
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                if (board[i][j] == 0) {
                    ii = i;
                    jj = j;
                }
            }
        }
        if (ii > 0) {
            swap(board[ii][jj], board[ii - 1][jj]);
            update(board, depth + 1);
            swap(board[ii][jj], board[ii - 1][jj]);
        }
        if (ii < 1) {
            swap(board[ii][jj], board[ii + 1][jj]);
            update(board, depth + 1);
            swap(board[ii][jj], board[ii + 1][jj]);
        }
        if (jj > 0) {
            swap(board[ii][jj], board[ii][jj - 1]);
            update(board, depth + 1);
            swap(board[ii][jj], board[ii][jj - 1]);
        }
        if (jj < 2) {
            swap(board[ii][jj], board[ii][jj + 1]);
            update(board, depth + 1);
            swap(board[ii][jj], board[ii][jj + 1]);
        }
    }
    int slidingPuzzle(vector<vector<int>>& board) {
        vector<vector<int>> b = {{1,2,3},{4,5,0}};
        update(b, 0);
        if (m.find(convert(board)) == m.end()) return -1;
        return m[convert(board)];
    }
};
