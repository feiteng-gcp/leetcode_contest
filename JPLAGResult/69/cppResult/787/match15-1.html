class Solution {
public:
    int encode(vector<vector<int>>& board) {
        return board[0][0] + board[0][1]*6 + board[0][2]*36 + board[1][0]*36*6 + board[1][1]*36*36 + board[1][2]*36*36*6;
    }
    vector<vector<int>> decode(int x) {
        vector<vector<int>> v(2, vector<int>(3, 0));
        v[0][0] = x % 6;
        x /= 6;
        v[0][1] = x % 6;
        x /= 6;
        v[0][2] = x % 6;
        x /= 6;
        v[1][0] = x % 6;
        x /= 6;
        v[1][1] = x % 6;
        x /= 6;
        v[1][2] = x;
        return v;
    }
    int slidingPuzzle(vector<vector<int>>& b) {
        unordered_set<int> used;
        used.insert(encode(b));
        

        queue<pair<int, int>> q;
        for(int i = 0; i < 2; i++) {
            for(int j = 0; j < 3; j++) {
                if(b[i][j] == 0) {
                    q.push({encode(b), i*3+j});
                }
            }
        }
        vector<vector<int>> t = {{1,2,3}, {4,5,0}};
        int d = 1, tg = encode(t), dr[4] = {-1,1,0,0}, dc[4] = {0,0,-1,1};
        if(tg == encode(b)) return 0;
        while(!q.empty()) {
            int qsize = q.size();
            while(qsize-- > 0) {
                int curr = q.front().first;
                int pos = q.front().second;
                q.pop();
                auto v = decode(curr);
                int r = pos / 3, c = pos % 3;

                for(int i = 0; i < 4; i++) {
                    int rr = r + dr[i], cc = c + dc[i];
                    if(rr >= 0 && rr < 2 && cc >= 0 && cc < 3) {
                        swap(v[r][c], v[rr][cc]);
                        auto next = encode(v);
                        if(!used.count(next)) {
                            if(next == tg) return d;
                            used.insert(next);
                            q.push({next, rr*3+cc});
                        }
                        swap(v[r][c], v[rr][cc]);
                    }
                }
            }
            d++;
        }
        return -1;
    }
};
