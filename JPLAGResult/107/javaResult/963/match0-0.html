class Solution {
    public int[] threeEqualParts(int[] A) {
        int N = A.length;
        int i = 0;
        while (i < N && A[i] == 0) {
            i++;
        }
        if (i == N) {
            return new int[] { 0, N - 1 };
        }

        int len = (N - i) / 3;
        int s = i;
        for (int l = 1; l <= len; l++) {
            s = i;
            int j = i + l;
            int p1 = j;
            while (j < N && A[j] == 0) {
                j++;
            }
            int k = j + l;
            int p2 = k;
            while (k < N && A[k] == 0) {
                k++;
            }
            boolean found = true;
            for (int t = 0; t < l; t++) {
                if (k >= N) {
                    found = false;
                    break;
                }
                if (A[s] != A[j] || A[j] != A[k]) {
                    found = false;
                    break;
                }
                s++;
                j++;
                k++;
            }
            if (!found) {
                continue;
            }
            if (k == N) {
                return new int[] { p1 - 1, p2 };
            }
        }
        return new int[] { -1, -1 };

    }
}
