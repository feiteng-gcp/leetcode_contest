import java.util.*;


class Solution {
    public int[] threeEqualParts(int[] A) {

        int total = 0;
        for (int a : A) { total += a; }
        if (total == 0) { return new int[]{0,2}; }
        if (total % 3 != 0) { return new int[]{-1,-1}; }

        // Count lagging zeros.

        int i = A.length-1;
        while(A[i] != 1) { i--; }

        int laggingZeros = A.length - i - 1;

        // Advance to the last position of the first part.
        int count = 0;
        int pos1 = 0;
        count += A[pos1];
        while(count < total/3) {
            pos1++;
            count += A[pos1];
        }
        pos1 += laggingZeros;

        // Advance to the last position of the second part.
        count = 0;
        int pos2 = pos1+1;
        count += A[pos2];
        while(count < total/3) {
            pos2++;
            count += A[pos2];
        }
        pos2 += laggingZeros;


        // Check equality
        int idx3 = A.length-1;
        int idx2 = pos2;
        int idx1 = pos1;

        while(idx1 >= 0 && idx2 > pos1 && idx3 > pos2) {

            if (A[idx1] != A[idx2]) { return new int[]{-1,-1}; }
            if (A[idx2] != A[idx3]) { return new int[]{-1,-1}; }

            idx1--;
            idx2--;
            idx3--;
        }

        return new int[]{pos1, pos2+1};


    }
}
