class Solution {
    public int[] threeEqualParts(int[] A) {
        if( A == null) {
            return new int[]{-1,-1};
        }
        
        int n = A.length;
        int s = 0;
        while(s < n && A[s] == 0){
            s++;
        }
        if(s == n){
            return new int[]{0, 2};
        }
        ArrayList<Integer> cur = new ArrayList<>();
        
        for(int i = s; i < n; i++) {
            cur.add(A[i]);
            int len = i - s+1;
            if((n - i -1)/2 >= len) {
                Integer sc = helper(cur, i+1, n, A);
                if(sc != null) {
                    return new int[]{i, sc};
                }
            } 
        }
        
return new int[]{-1,-1};
    }
    
    private Integer helper(ArrayList<Integer> cur, int j , int n, int[] A) {
        
int s = j;
        //System.out.format("j = %d\n", j);
        while(s < n && A[s] == 0){
            s++;
        }
        if( s== n) {
            return null;
        }
        j = s;
        for(Integer match : cur) {
            if(j >= n || match != A[j++]){
                return null;
            }
        }
        int res = j;
        //System.out.format("res = %d\n", j);
        s= j; 
        
while(s < n && A[s] == 0){
            s++;
        }
        if( s== n) {
            return null;
        }
        j=s ;
        
for(Integer match : cur) {
            if(j >= n || match != A[j++]){
                return null;
            }
        }
        return j==n ? res : null;
    }
}
