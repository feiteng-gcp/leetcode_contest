class Solution {
    public int[] threeEqualParts(int[] A) {
        int one = 0;
        for (int x : A) {
            one += x;
        }
        if (one % 3 != 0) return new int[] {-1, -1};
        if (one == 0) return new int[] {0, A.length - 1};
        one /= 3;
        int cc = 0;
        int[] pos = new int[3];
        int idx = 0;
        for (int i = 0; i < A.length; i++) {
            
if (A[i] == 1 && cc % one == 0) {
                pos[idx++] = i;
            }
            cc += A[i];
        }
        int len = A.length - pos[2];
        if (pos[1] < pos[0] + len || pos[2] < pos[1] + len) return new int[] {-1, -1};
        for (int i = pos[0], j = pos[1], k = pos[2]; k < A.length; i++, j++, k++) {
            
if (A[i] != A[j] || A[i] != A[k]) return new int[] {-1, -1};
        }
        return new int[] {pos[0] + len - 1, pos[1] + len};
    }
}
