class Solution {
    public int[] threeEqualParts(int[] A) {
        int count1=0;
        int index1=-1;
        int last1=0;
        List<Integer> pos1=new ArrayList<Integer>();
        for(int i=0;i<A.length;i++){
            if(A[i]==1){
                count1++;
                pos1.add(i);
                if(index1==-1){
                    index1=i;
                }
                last1=i;
            
}
        }
        int countLast0=A.length-1-last1;
        if(count1%3!=0){
            return new int[]{-1,-1};
        }
        if(count1==0){
            return new int[]{0,2};
        }
        int min=pos1.get(pos1.size()/3-1)+countLast0;
        int max=pos1.get(pos1.size()/3);
        for(int i=min;i<=min;i++){
            int j=isMatch(A,0,i+1,i+1);
            if(j==-1){
                continue;
            }
            int k=isMatch(A,0,i+1,j);
            if(k==A.length){
                
return new int[]{i,j};
            }
        }
        return new int[]{-1,-1};
    }
    public int isMatch(int[] A,int s1,int e1,int s2){
        if(s2==A.length){
            return -1;
        }
        int ori=s2;
        boolean started=false;
        while(s1<e1&&s2<A.length){
            if(A[s1]==1){
                if(!started){
                    started=true;
                    while(s2<A.length&&A[s2]==0){
                        s2++;
                    }
                    if(s2==A.length){
                        return -1;
                    }
                }
                else{
                    if(A[s2]!=1){
                        return -1;
                    }
                }
                s2++;
                s1++;
            }
            else{
                if(!started){
                    s1++;
                }
                else{
                    if(A[s2]!=0){
                        return -1;
                    }
                    s2++;
                    s1++;
                }
            }
        }
        /*
        if(!started){
            if(s2<A.length&&A[s2]==0){
                s2++;
            }
            else{
                return -1;
            }
        }
        */
        return s2;
    }
}
