class Solution {
    public int[] threeEqualParts(int[] A) {
        int count = 0;
        for (int a: A) {
            count += a;
        }
        if (count % 3 != 0) {
            return new int[]{-1, -1};
        }
        if (count == 0) {
            return new int[]{0, 2};
        }
        count /= 3;
        int i = 0;
        int tail = 0;
        for (int j = A.length - 1; A[j] == 0; j--) {
            tail++;
        }
        List<Integer> pattern = new ArrayList<Integer>();
        while (A[i] == 0) {
            i++;
        }
        while (count > 0) {
            pattern.add(A[i]);
            if (A[i] == 1) {
                count--;
            }
            i++;
        }
        int k1 = i - 1;
        int t = 0;
        while (A[i] == 0) {
            i++;
            t++;
            if (t == tail) {
                k1 = i - 1;
            }
        }
        if (t < tail) {
            return new int[]{-1, -1};
        }
        for (int j = 0; j < pattern.size(); j++) {
            if (A[i] != pattern.get(j)) {
                return new int[]{-1, -1};
            }
            i++;
        }
        int k2 = i;
        t = 0;
        while (A[i] == 0) {
            i++;
            t++;
            if (t == tail) {
                k2 = i;
            }
        }
        if (t < tail) {
            return new int[]{-1, -1};
        }
        for (int j = 0; j < pattern.size(); j++) {
            if (A[i] != pattern.get(j)) {
                return new int[]{-1, -1};
            
}
            i++;
        }
        return new int[]{k1, k2};
    }
}
