class Solution {
    public int[] threeEqualParts(int[] A) {
        int num = 0;
        if (A.length < 3) {
            return new int[] {-1, -1};
        }
        
        
for (int j: A) {
            num += j;
        }
        
        if (num == 0) {
            return new int[] {0, 2};
        }
        
        if (num % 3 != 0) {
            return new int[] {-1, -1};
        } 
        
        int target = num / 3;
        int curr = 0;
        int[] result = new int[] {-1, -1};
        
        int leadingZeros = 0;
        boolean notOne = true;
        int i = A.length - 1;

        for (; i >= 0 ; i--) {
            if (curr == target) {
                break;
            }
            if (A[i] == 0 && notOne) {
                leadingZeros++;
            }
            if (A[i] == 1) {
                curr += A[i];
                notOne = false;
            }
        }
        
        int midZeros = 0;
        while (i >= 0 && A[i] == 0) {
            midZeros++;
            i--;
        }

        if (midZeros < leadingZeros) {
            return new int[] {-1, -1};
        }
        
        result[1] = i + leadingZeros + 1;
        
        curr = 0;
        for (; i >= 0 ; i--) {
            if (curr == target) {
                break;
            }
            curr += A[i];
        }
        
        int headZeros = 0;
        while (i >= 0 && A[i] == 0) {
            headZeros++;
            i--;
        }
        
        if (headZeros < leadingZeros) {
            return new int[] {-1, -1};
        }
        
        result[0] = i + leadingZeros;
        
        int index1 = result[0];
        int index2 = result[1] - 1;
        int index3 = A.length - 1;
        curr = 0;
        while (true) {
            curr += A[index1];
            if (curr == target) {
                return result;
            
}
            if (!((A[index1] == A[index2]) && (A[index1] == A[index3]))) {
                return new int[] {-1, -1};
            }
            index1--;
            index2--;
            index3--;
        }
        
    }
}
