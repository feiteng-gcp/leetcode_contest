    boolean debug = false;
    
    public int[] threeEqualParts(int[] A) {
        
        
        
        int n = A.length;
        int leftJ = 2, rightJ = n - 1;
        while(rightJ - leftJ > 1) {
            int middle = (leftJ + rightJ) / 2;
            int i = getIndex(A, middle);
            if(debug) {
                System.out.println(String.format(
                    "getIndex(A, %s)=%s", middle, i
                ));
            }
            if (i==-1) {
                // j==middle is too small, i.e., A[j..] is too large
                leftJ = middle;
            } else {
                rightJ = middle;
            }
        }
        if (getIndex(A, leftJ) != -1) {
            rightJ = leftJ;
        }
        int j = rightJ;
        int i = getIndex(A, j);
        if(debug) {
            System.out.println("Checking i=" + i + ", j="+j);
        }
        
        if (!checkEquals(A, 0, i+1, j, n)) {
            return new int[] {-1, -1};
        }
        
        if(debug) System.out.println("Checkpoint 1 done.");
        
        if (!checkEquals(A, i+1, j, j, n)) {
            return new int[] {-1, -1};
        }
        
        if(debug) System.out.println("Checkpoint 2 done.");
        
        return new int[] {i, j};
    }
    
    
    /*
    return the smallest value of i (>=0), so that
        (1) Binary Number of A[0..i] >= Binary Number of A[j..]; and
        (2) Binary Number of A[i+1..j-1] >= Binary Number of A[j..]; and
    or return -1 if such an i does not exist
    */
    private int getIndex(int[] A, int j) {
        int n = A.length;
        int firstOneAtJ = j;
        while(firstOneAtJ < n && A[firstOneAtJ] == 0) {
            firstOneAtJ += 1;
        }
        if (firstOneAtJ == n) {
            return 0;
        }
        int firstOne = 0;
        while(firstOne < j && A[firstOne] == 0) {
            firstOne += 1;
        }
        if(firstOne==j) {
            return -1;
        }
        
        int i = firstOne + n - firstOneAtJ - 1;
        for(int k = 0; k < n - firstOneAtJ; k++) {
            int tmp1 = A[firstOne + k], tmp2 = A[firstOneAtJ + k];
            if (tmp1 != tmp2) {
                if (tmp1 < tmp2) {
                    i += 1;
                }
                break;
            }
        }
        // now we have got the smallest i so that A[0..i] >= A[j..]
        
        if (i+1>j-1) {
            return -1;
        }
        
        // now check if A[i+1 .. j-1] >= A[j..]
        int firstOneAtI = i+1;
        while(firstOneAtI<j && A[firstOneAtI]==0) {
            firstOneAtI += 1;
        }
        if (firstOneAtI==j) {
            return -1;
        }
        if (j-firstOneAtI > n-firstOneAtJ) {
            return i;
        } else if(j-firstOneAtI < n-firstOneAtJ) {
            return -1;
        }
        for(int k = 0; k < n - firstOneAtJ; k++) {
            int tmp1 = A[firstOneAtI + k], tmp2 = A[firstOneAtJ + k];
            if (tmp1 != tmp2) {
                if (tmp1 < tmp2) {
                    return -1;
                }
                break;
            }
        }
        
        return i;
    }
    
    private boolean checkEquals(int[] A, int i, int ibar, int j, int jbar) {
        while(i<ibar && A[i]==0) {
            i += 1;
        }
        while(j<jbar && A[j]==0) {
            j += 1;
        }
        if (jbar-j != ibar-i) {
            return false;
        }
        int k = ibar-i;
        for(int a=0;a<k;a++) {
            if(A[i+a] != A[j+a]) {
                return false;
            }
        }
        return true;
    }
    
}
