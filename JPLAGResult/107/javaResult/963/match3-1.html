class Solution {
    public int[] threeEqualParts(int[] A) {
        int len = A.length;
        int z = 0;
        while(z < len && A[z] != 1) {
            z++;
        
}
        // A[z] is the first 1
        if(z == len) {
            return new int[]{0, len - 1};
        }

        for(int i = z; i < len - 2; i++) {
            // [0, i]  [0 ~ z ~ i ]
            int j = len - 1 - (i - z);
            // [j, len - 1] should be the same with [z, i]
            boolean flag = false;

            for(int k = 0; k <= i - z; k++) {
                if(A[len - 1 - k] != A[i - k]) {
                    flag = true;
                    break;
                }
            }

            if(flag) {
                continue;
            }

            // [j, len - 1] is the same with [z, i]
            //[z, i] should be the same with [i + 1, j - 1]
            int k = i + 1;
            while(k <= j - 1 && A[k] == 0) {
                k++;
            }

            //[z, i] [k, j - 1] j - 1 might be zero
            if(i - z > j - 1 - k) {
                continue;
            }
            boolean found = true;
            int x = 0;
            for(; x <= i - z; x++ ) {
                if(A[z + x] != A[k + x]) {
                    found = false;
                    break;
                }
            }

            if(found) {
                found = true;
                for(int p = k + i - z + 1; p <= j - 1; p++) {
                    if(A[p] != 0) {
                        found = false;
                        break;
                    }
                
}
                if(found) {
                    return new int[]{i, k + i - z + 1};
                }

            
}
        }

        return new int[]{-1, -1};
    }
}
