class Solution {
    public int[] threeEqualParts(int[] A) {
        int count = 0;
        for (int i = 0;i < A.length;i++) count += A[i]
;
        if (count%3 != 0) return new int[]{-1, -1};
        if (count == 0) return new int[]{0,A.length - 1};
        
        count /= 3;
        int last = A.length - 1, num = count;
        for (last = A.length - 1;last > 1;last--){
            if (A[last] == 1) num--;
            if (num == 0) break;
        }
        int first = 0;
        while (A[first] == 0) first++;
        for (int i = 0;i < last - first;i++){
            if (A[first + i] != A[last + i]) return new int[]{-1, -1};
            if (last + i == A.length - 1){
                first += i;
                break;
            }
        }
        
        // check mid
        int p = first + 1;
        while (A[p] == 0) p++;
        num = count;
        for (
int i = 0;i < A.length - last;i++){
            if (A[p] != A[last + i]) return new int[]{-1, -1};
            p++;
        }
        
        return new int[]{first, p};
    }
}
