class Solution {
  public int[] threeEqualParts(int[] A) {
        int count1 = 0;
        for (int i = 0; i < A.length; i++) {
            if (A[i] == 1) {
                count1++;
            }
        }

        if (count1 == 0) {
            return new int[]{0, 2};
        }

        if (count1 % 3 != 0) {
            return new int[]{-1, -1};
        }
        int num = count1 / 3;

        int num31 = 0;
        int index3 = -1;
        for (int i1 = A.length - 1; i1 >= 0; i1--) {
            if (A[i1] == 1) {
                num31++;
            }
            if (num31 == num) {
                index3 = i1;
                break;
            }
        }

        int j = index3;
        int index2 = -1;
        boolean first = true;
        int temp = A.length - 1;
        for (int i1 = index3 - 1; i1 >= 0; i1--) {
            if (A[i1] == 1) {
                first = false;
            }
            if (A[i1] == A[temp]) {
                temp--;
            } else if (A[i1] == 1) {
                return new int[]{-1, -1};
            } else if (first) {
                j--;
            } else {
                return new int[]{-1, -1};
            }
            if (temp < index3) {
                index2 = i1;
                break;
            }
        }

        int i = index2 - 1;
        int index1 = -1;
        first = true;
        temp = A.length - 1;
        for (int i1 = index2 - 1; i1 >= 0; i1--) {
            if (A[i1] == 1) {
                first = false;
            }
            if (A[i1] == A[temp]) {
                temp--;
            } else if (A[i1] == 1) {
                return new int[]{-1, -1};
            } else if (first) {
                i--;
            } else {
                return new int[]{-1, -1};
            }
            if (temp < index3) {
                index1 = i1;
                break;
            }
        }
        index1--;
        while (index1 >= 0) {
            if (A[index1] == 1) {
                return new int[]{-1, -1};
            }
            index1--;
        }

        return new int[]{i, j};
    }
}
