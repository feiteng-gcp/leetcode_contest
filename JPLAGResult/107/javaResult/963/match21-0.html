class Solution {
    public int[] threeEqualParts(int[] A) {
        int count = 0;
        for (int v : A) {
            count += v;
        }
        if (count % 3 != 0) {
            return new int[]{-1, -1};
        }
        if (count == 0) {
            return new int[]{0, 2};
        }
        int avg = count / 3;
        int a = 0;
        int n = A.length;
        for (int i = 0; i < n; i++) {
            if (A[i] == 1) {
                a = i;
                break;
            }
        }

        int i = a;
        for (int j = 0; i < n; i++) {
            if (A[i] == 1) {
                j++;
            }
            if (j == avg) {
                break;
            }
        }
        int a1 = i;
        i++;

        //find b
        int b = 0;
        for (int k = i; k < n; k++) {
            if (A[k] == 1) {
                b = k;
                break;
            }
        }

        int kk = b;
        for (int m = a; m <= a1; m++, kk++) {
            if (kk >= n || A[m] != A[kk]) {
                return new int[]{-1, -1};
            }
        }
        int b1 = kk - 1;

        int c = 0;
        for (int dd = kk; dd < n; dd++) {
            if (A[dd] == 1) {
                c = dd;
                break;
            }
        }

        int jj = c;
        for (int m = a; m <= a1; m++, jj++) {
            if (jj >= n || A[m] != A[jj]) {
                return new int[]{-1, -1};
            }
        }

        System.out.println(a);
        System.out.println(b);
        System.out.println(c);


        int leftZeros = n - jj;
        System.out.println(leftZeros);
        if (c - b1 - 1 < leftZeros) {
            return new int[]{-1, -1};
        }
        if (b - a1 - 1 < leftZeros) {
            return new int[]{-1, -1};
        }


        return new int[]{a1 + leftZeros, c - (c - b1 - 1 - leftZeros)};
    }
}
