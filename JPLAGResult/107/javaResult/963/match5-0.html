class Solution {

    public int[] threeEqualParts(int[] arr) {
        // all 0's case
        boolean zero = true;
        for (int i : arr) {
            if (1 == i) {
                zero = false;
                break;
            }
        }
        if (zero) {
            return new int[]{0, 2};
        }

        for (int i = 0; i < arr.length; i++) {
            int[] three = three(arr, i);
            if (three[0] >= 0) {
                return three;
            }
        }
        return new int[]{-1, -1};
    }

    private int[] three(int[] arr, int i) {
        int start = 0, n = arr.length;
        while (start < n && arr[start] == 0) {
            start++;
        }
        int b = i + 1;
        int[] impossible = new int[]{-1, -1};
        while (b < n && arr[b] == 0) {
            b++;
        }
        for (int a = start; a <= i; a++, b++) {
            if (a >= n || b >= n) {
                return impossible;
            }
            if (arr[a] != arr[b]) {
                return impossible;
            }
        }
        int c = b;
        while (c < n && arr[c] == 0) {
            c++;
        }
        for (int a = start; a <= i; a++, c++) {
            if (a >= n || c >= n) {
                return impossible;
            }
            if (arr[a] != arr[c]) {
                return impossible;
            }
        }
        if (c < n) {
            return impossible;
        }
        return new int[]{i, b};
    }
}
