	class Solution {
	    public int[] threeEqualParts(int[] A) {
	        int s = 0;
	        for(int v : A)s+=v;
	        if(s % 3 != 0)return new int[]{-1, -1};
	        if(s == 0){
	        	return new int[]{0, 2};
	        }
	        int x = 0;
	        int[] ret = new int[]{-1, -1};
	        int tail = 0;
	        for(int i= A.length-1;i >= 0;i--){
	        	if(A[i] == 1){
	        		tail = A.length-1-i;
	        		break;
	        	}
	        }
	        
	        for(int i = 0;i < A.length;i++){
	        	x += A[i];
	        	if(x == s/3 && ret[0] == -1){
	        		ret[0] = i+tail;
	        	}
	        	if(x == 2*s/3 && ret[1] == -1){
	        		ret[1] = i+1+tail;
	        	}
	        }
	        int n = A.length;
	        if(ret[1] >= n)return new int[]{-1, -1};
	        
	        int ct = 0;
	        for(int i = 0;i <= ret[0];i++)ct += A[i];
	        if(ct != s/3)return new int[]{-1,-1};
	        	
	        ct = 0;
	        for(int i = ret[0]+1;i < ret[1];i++)ct += A[i];
	        if(ct != s/3)return new int[]{-1,-1};
	        
	        for(int i = n-1, j = ret[1]-1, k = ret[0];i >= ret[1] && j >= ret[0]+1 && k >= 0;
	        		i--,j--,k--){
	        	int u = A[i]+A[j]+A[k];
	        	if(!(u == 3 || u == 0))return new int[]{-1, -1};
	        }
	        
	        return ret;
	    }
	}
