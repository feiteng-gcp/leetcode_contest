class Solution {
    public int minFlipsMonoIncr(String S) {
        int len = S.length();
        int[] preOne = new int[len];
        int[] preZero = new int[len];
        for (int i=0; i<len; i++) {
            if (S.charAt(i) == '0') {
                preZero[i] = i==0?1:preZero[i-1]+1;
                preOne[i] = i==0?0:preOne[i-1];
            }
            if (S.charAt(i) == '1') {
                preOne[i] = i==0?1:preOne[i-1]+1;
                preZero[i] = i==0?0:preZero[i-1];
            }
        
}
        // all One
        int ans = preZero[len-1];
        for (int i=0; i<len; i++) {
            ans = Math.min(ans, preOne[i]+preZero[len-1]-preZero[i]);
        }
        return ans;
    }
}
