class Solution {
    public int minFlipsMonoIncr(String S) {
        char[] input = S.toCharArray();
        int[] oneCounts = new int[input.length];
        int[] zeroCounts = new int[input.length];
        oneCounts[0] = input[0] == '1' ? 1 : 0;
        zeroCounts[0] = input[0] == '0' ? 1 : 0;
        
        for(int i=1; i<input.length; ++i) {
            oneCounts[i] = oneCounts[i - 1] + (input[i] == '1' ? 1 : 0);  
            zeroCounts[i] = zeroCounts[i - 1] + (input[i] == '0' ? 1 : 0);                
        }
        int total = input.length;
        int totalOne = oneCounts[input.length - 1];
        int totalZero = zeroCounts[input.length - 1];
        
        int best = totalZero; // zeroCount == 0
        for(int zeroCount=1; zeroCount<=input.length; ++zeroCount) {
            int lastZeroIndex = zeroCount - 1;
            
            int oneToZero = oneCounts[lastZeroIndex];
            int zeroToOne = totalZero - zeroCounts[lastZeroIndex];
            
if(oneToZero + zeroToOne < best) {
                best = oneToZero + zeroToOne;
            }
        }
        return best;
    }
}
