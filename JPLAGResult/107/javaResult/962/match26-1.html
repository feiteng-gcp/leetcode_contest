class Solution {
    public int minFlipsMonoIncr(String S) {
        if(S.length()==1){
            return 0;
        }
        
        int[] oneNumber = new int[S.length()];
        int[] zeroNumber = new int[S.length()];
        
        if(S.charAt(0)=='1'){
            oneNumber[0]=1;
        }else{
            oneNumber[0]=0;
        }
        for(int i=1;i<S.length();i++)
{
            if(S.charAt(i)=='1'){
                oneNumber[i]=oneNumber[i-1]+1;
            }else{
                oneNumber[i]=oneNumber[i-1];
            }
        }
        
        if(S.charAt(S.length()-1)=='0'){
            zeroNumber[S.length()-1]=1;
        }else{
            zeroNumber[S.length()-1]=0;
        }
        for(int i=S.length()-2;i>=0;i--)
{
            if(S.charAt(i)=='0'){
                zeroNumber[i]=zeroNumber[i+1]+1;
            }else{
                zeroNumber[i]=zeroNumber[i+1];
            }
        }
        
        int ans = Math.min(zeroNumber[0], oneNumber[S.length()-1]);
        for(int i=1;i<S.length();i++){
            ans = Math.min(ans,oneNumber[i-1]+zeroNumber[i]);
        }
        
        
        return ans;
    }
}
