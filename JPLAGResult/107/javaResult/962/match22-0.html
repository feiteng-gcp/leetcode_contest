class Solution {
    public int minFlipsMonoIncr(String S) {
        int len = S.length();
        int[] zeros = new int[S.length()], ones = new int[S.length()];
        int cntZero = 0, cntOne = 0;
        for (int i = 0, j = S.length() - 1; i < S.length() && j >= 0; i++, j--) {
            if (S.charAt(i) == '1') {
                cntOne++;
            }
            
            if (S.charAt(j) == '0') {
                cntZero++;
            }
            zeros[i] = cntOne;
            ones[j] = cntZero;
        }
        
        int min = Math.min(zeros[len - 1], ones[0]);
        for (int i = 0; i < S.length() - 1; i++) {
            min = Math.min(zeros[i] + ones[i + 1], min);
        }
        return min;
    }
}
