class Solution {
    public int minFlipsMonoIncr(String S) {
        int n = S.length();
        int[] left = new int[n + 1];
        int[] right = new int[n + 1];
        
        
        
for (int i = 0; i < n; i++) {
            left[i + 1] = left[i];
            if (S.charAt(i) == '1') {
                left[i + 1]++;
            }
        }
        
        
for (int i = n - 1; i >= 0; i--) {
            right[i] = right[i + 1];
            if (S.charAt(i) == '0') {
                right[i]++;
            }
        }
        int ans = n;
        for (int i = 0; i < n; i++) {
            int cur = left[i] + right[i + 1];
            ans = Math.min(ans, cur);
        }
        return ans;
    }
}
