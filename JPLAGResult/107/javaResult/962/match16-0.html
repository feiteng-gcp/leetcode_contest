class Solution {
    public int minFlipsMonoIncr(String S) {
        int[] leftZero = new int[S.length() + 1];
        int[] rightOne = new int[S.length() + 1];
        for(int i = 0 ; i < S.length() ; i++){
            if(S.charAt(i) == '0') leftZero[i + 1] = leftZero[i];
            else leftZero[i + 1] = leftZero[i] + 1;
        }
        for(int i = S.length() - 1 ; i >= 0 ; i--){
            if(S.charAt(i) == '1') rightOne[i] = rightOne[i + 1];
            else rightOne[i] = rightOne[i + 1] + 1;
        }
        int min = Integer.MAX_VALUE;
        for(int i = 0 ; i < leftZero.length ; i++) min = Math.min(min, leftZero[i] + rightOne[i]);
        return min;
    }
}
