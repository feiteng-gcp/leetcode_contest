	class Solution {
	    public int minFlipsMonoIncr(String S) {
	        char[] s = S.toCharArray();
	        int n = s.length;
	        int[] ct = new int[n+1];
	        int x = 0;
	        for(int i = 0;i <= n;i++){
	        	ct[i] += x;
	        	if(i < n && s[i] == '1')x++;
	        }
	        x = 0;
	        for(int i = n;i >= 0;i--){
	        	ct[i] += x;
	        	if(i > 0 && s[i-1] == '0')x++;
	        }
	        
	        int ret = 999999999;
	        for(int i = 0;i <= n;i++){
	        	ret = Math.min(ret, ct[i]);
	        }
	        return ret;
	    }
	}
