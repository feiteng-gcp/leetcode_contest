class Solution {
    public int minFlipsMonoIncr(String S) {
        if (S.length() <= 1) {
            return 0;
        }
        int zero = 0;
        int one = 0;
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            int tempZero = zero;
            int tempOne = one;
            if (c == '0') {
                one = Math.min(tempZero, tempOne) + 1;
            } else {
                zero = tempZero + 1;
                one = Math.min(tempZero, tempOne);
            }
        }
        return Math.min(zero, one);
    }
}
