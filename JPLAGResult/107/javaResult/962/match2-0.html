class Solution {
    public int minFlipsMonoIncr(String s) {
        int n = s.length();
        int[] dp0 = new int[n + 1]; // dp[i] how many before i exclusive
        dp0[0] = 0;
        for (int i = 1; i <= n; i++) {
            dp0[i] = dp0[i - 1];
            if (i < n && s.charAt(i - 1) == '1') {
                dp0[i]++;
            }
        }
        int[] dp1 = new int[n + 1]; // dp[i] how many after i inclusive
        dp1[n] = 0;
        for (int i = n - 1; i >= 0; i--) {
            dp1[i] = dp1[i + 1];
            if (s.charAt(i) == '0') {
                dp1[i]++;
            }
        }
        int min = s.length();
        for (int i = 0; i <= n; i++) {
            min = Math.min(min, dp0[i] + dp1[i]);
        }
        return min;
    }
}
