class Solution {
    public int minFlipsMonoIncr(String S) {
		if (S == null || S.length() == 0) {
			return 0;
		}
		int[] flipsEndWithOne = new int[S.length() + 1];
		int[] flipsEndWithZero = new int[S.length() + 1];
		for (int i = 0; i < S.length(); i++) {
			if (S.charAt(i) == '0') {
				flipsEndWithZero[i + 1] = flipsEndWithZero[i];
				flipsEndWithOne[i + 1] = Math.min(flipsEndWithOne[i],flipsEndWithZero[i]) + 1;
			}
			else {
				flipsEndWithZero[i + 1] = flipsEndWithZero[i] + 1;
				flipsEndWithOne[i + 1] = Math.min(flipsEndWithOne[i],flipsEndWithZero[i]);
			}
		}
		return Math.min(flipsEndWithOne[S.length()],flipsEndWithZero[S.length()]);
	}
}
