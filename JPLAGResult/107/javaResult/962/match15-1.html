class Solution {
    public int minFlipsMonoIncr(String S) {
        int[] minZeroCost = new int[S.length()];
        int[] minOneCost = new int[S.length()];
        int minZero = 0;
        int minOne = 0;
        for (int i = 0; i < minZeroCost.length; i++) 
{
            if (S.charAt(i) == '1') {
                minZero++;
            }
            minZeroCost[i] = minZero;
        }
        
for (int i = minOneCost.length - 1; i >= 0; i--) {
            if (S.charAt(i) == '0') {
                minOne++;
            }
            minOneCost[i] = minOne;
        }
        int minCost = Integer.MAX_VALUE;
        for (int i = 0; i < S.length() - 1; i++) 
{
            if (minZeroCost[i] + minOneCost[i + 1] < minCost) {
                minCost = minZeroCost[i] + minOneCost[i + 1];
            }
        }
        return Math.min(minCost, Math.min(minZeroCost[S.length() - 1], minOneCost[0]));
    }
}
