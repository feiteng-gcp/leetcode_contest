class Solution {
    public int minFlipsMonoIncr(String S) {
        int n = S.length();
        if (n < 2) return 0;
        char[] chs = S.toCharArray();
        int[] l = new int[n], r = new int[n];
        l[0] = chs[0] == '0'? 0:1;
        r[n-1] = chs[n-1] == '1'? 0:1;
        for (int i = 1; i < n; i++) {
            l[i] = chs[i] == '0'? l[i-1] : l[i-1]+1;
        }
        for (int i = n-2; i >= 0; i--) {
            r[i] = chs[i] == '1'? r[i+1] : r[i+1]+1;
        }
        int min = Math.min(l[n-1], r[0]);
        for (int i = 0; i < n-1; i++) {
            min = Math.min(min, l[i]+r[i+1]);
        }
        return min;
    }
}
