class Solution {
    public int minFlipsMonoIncr(String S) {
        int[] a = new int[S.length()];
        int[] b = new int[S.length()];
        
        for(int i = 0; i < S.length(); i++){
            if(i == 0){
                if(S.charAt(i) == '1') a[i] = 1;
                else a[i] = 0;
                continue;
            }
            if(
S.charAt(i) == '1') a[i] = a[i-1]+1;
            else a[i] = a[i-1];
            
        }
        
        
        
        for(int i = S.length()-1; i >=0 ; i--){
            if(i == S.length()-1){
                if(S.charAt(i) == '0') b[i] = 1;
                else b[i] = 0;
                continue;
            }
            if(S.charAt(i) == '0') 
b[i] = b[i+1]+1;
            else b[i] = b[i+1];
            
        }
        
        int ans = 111111;
        
        for(int i = 0; i < S.length(); i++){
            int num = 999999;
            if(S.charAt(i) == '0'){
                 num = (i+1>=S.length()?0:b[i+1] )+ a[i];
            }else{
                 num = b[i] + (i-1<=0? 0 :a[i-1]);
            }
            if(num < ans) ans = num;
        }
        return ans;
    }
}
