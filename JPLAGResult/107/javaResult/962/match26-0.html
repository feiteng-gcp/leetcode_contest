class Solution {
    public int minFlipsMonoIncr(String s) {
        int l = s.length();
        int[] left = new int[l];
        int[] right = new int[l];
        
        if(s.charAt(0) == '0'){
            left[0] = 1;
        }
        
        for(int i=1; i < l; i++){
            if(s.charAt(i) == '0'){
                left[i] = left[i-1]+1;
            }else{
                left[i] = left[i-1];
            }
        }
        
        if(s.charAt(l-1) == '0'){
            right[l-1] = 1;
        }
        
        for(int i=l-2; i >= 0; i--){
            if(s.charAt(i) == '0'){
                right[i] = right[i+1] + 1;
            }else{
                right[i] = right[i+1];
            }
        }
        
        int ans = right[0];
        for(int i=0; i < l; i++){
            int ones = (i+1)-left[i];
            int zeros = 0;
            if(i+1 < l){
                zeros = right[i+1];
            }
            ans = (int)Math.min(ans, ones+zeros);
        }
        return ans;
    }
}
