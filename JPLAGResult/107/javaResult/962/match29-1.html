class Solution {
    public int minFlipsMonoIncr(String S) {
        int n = S.length(); 
        int[][] dp = new int[n][2];
        char[] cs = S.toCharArray();
        for (int i = 0; i < n; i++) {
            Arrays.fill(dp[i], 1 << 30);
        }
        if (cs[0] == '0') {
            dp[0][1] = 1; 
            dp[0][0] = 0;
        } else {
            dp[0][0] = 1;
            dp[0][1] = 0;
        }
        for (int i = 1; i < n; i++) {
            if (cs[i] == '0') {
                dp[i][0] = dp[i - 1][0];
                
dp[i][1] = Math.min(dp[i - 1][1] + 1, dp[i - 1][0] + 1);
            } else {
                dp[i][0] = dp[i - 1][0] + 1; 
                dp[i][1] = Math.min(dp[i - 1][0], dp[i - 1][1]);
            }
        }
     // for (int i = 0; i < n; i++) {
     //        System.out.println(i + " 0:" +  dp[i][0] + " 1:" + dp[i][1]);
     //    }
        return Math.min(dp[n - 1][0], dp[n - 1][1]);
    }
}
