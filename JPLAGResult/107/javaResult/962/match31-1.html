class Solution {
    public int minFlipsMonoIncr(String S) {
        if (S == null || S.length() == 0) return 0;
        S = "0" + S + "1";
        int n = S.length();
        int[] fromLeft = new int[n];
        int[] fromRight = new int[n];
        int flip = 0;
        for (int i = 0; i < S.length(); i++) {
            if (S.charAt(i) == '1') {
                flip++;
            }
            fromLeft[i] = flip;
        }
        flip = 0;
        
for (int j = n - 1; j >= 0; j--) {
            if (S.charAt(j) == '0') {
                flip++;
            }
            fromRight[j] = flip;
        }
        int min = Integer.MAX_VALUE;
        // System.out.println(Arrays.toString(fromLeft));
        // System.out.println(Arrays.toString(fromRight));
        for (int i = 1; i < n; i++) {
            if (fromLeft[i - 1] + fromRight[i] < min) {
                min = fromLeft[i - 1] + fromRight[i];
            }
        }
        return min;
    }
}
