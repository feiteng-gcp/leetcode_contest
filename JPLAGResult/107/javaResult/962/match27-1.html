class Solution {
    public int minFlipsMonoIncr(String S) {
        char[] ch = S.toCharArray();
        
int res = Integer.MAX_VALUE, count = 0;
        for (char c : ch) {
            if (c == '1')
                count++;
        }
        res = count;
        for (int i = ch.length-1; i >= 0; i--) {
            if (ch[i] == '0')
                count++;
            else
                count--;
            res = Math.min(res, count);
        }
        return res;
    }
}
