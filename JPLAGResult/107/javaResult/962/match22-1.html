class Solution {
  public int minFlipsMonoIncr(String S) {
    // use a dp approach (O(n))
    
    // leftAllZero[i]: flip required to make S.substring(0, i + 1) all zero
    int[] leftAllZero = new int[S.length()];
    for (int i = 0; i < S.length(); ++i) {
      if (S.charAt(i) == '1') {
        // have to flip
        leftAllZero[i] += 1;
      }
      if (i > 0) {
        // sum prev
        leftAllZero[i] += leftAllZero[i - 1];
      }
    }
    
    // rightAllOne[i]: flip required to make S.substring(i, S.length()) all one
    int[] rightAllOne = new int[S.length()];
    for (int i = 
S.length() - 1; i >= 0; --i) {
      if (S.charAt(i) == '0') {
        // have to flip
        rightAllOne[i] += 1;
      }
      if (i < S.length() - 1) {
        // sum prev
        rightAllOne[i] += rightAllOne[i + 1];
      }
    }
    
    int res = Math.min(leftAllZero[
S.length() - 1], rightAllOne[0]);
    for (int i = 0; i < S.length() - 1; ++i) {
      res = Math.min(res, leftAllZero[i] + rightAllOne[i + 1]);
    }
    return res;
  }
}
