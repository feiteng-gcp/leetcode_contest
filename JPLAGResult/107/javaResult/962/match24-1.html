class Solution {
    public int minFlipsMonoIncr(String S) {
        char [] cs = S.toCharArray();
        int L = S.length();
        int [][] dp = new int[L+1][2];
        
        for (int k = 1; k <= L; k++) {
        	int n = cs[k-1] - '0';
        	if (n==1) {
        		dp[k][0] = dp[k-1][0] + 1;
        		dp[k][1] = Mat
h.min(dp[k-1][1], dp[k-1][0]);
        	} else {
        		dp[k][0] = dp[k-1][0];
        		dp[k][1] = Math.min(dp[k-1][1] + 1, dp[k-1][0] + 1);
        	}
        	//System.out.println(dp[k][0] + ", " + dp[k][1]);
        }
        return Math.min(dp[L][0],dp[L][1]);
    }
}
