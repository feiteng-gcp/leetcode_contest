class Solution {
    public int minFlipsMonoIncr(String S) {
        
        // find the number of zeros before and after each boundary
        int n = S.length();
        int[] before = new int[n+1];
        int[] after = new int[n+1];
        
        before[0] = 0;
        for (int i = 1; i < n+1; ++i) {
            
            if (S.charAt(i-1) == '0') before[i] = before[i-1] + 1; 
            else  before[i] = before[i-1];
        }
        
        after[n] = 0;
        for (int i = n-1; i >= 0; --i) {
            if (S.charAt(i) == '0') after[i] = after[i+1] + 1;
            else 
after[i] = after[i+1];
        }
        
        int min = n;
        
        for (int i = 0; i < n+1; ++i) {
            
            // the number of 1s before i
            int tmp1 = i - before[i];
            // the number of 0s after i
            int tmp2 = after[i];
            if (tmp1 + tmp2 < min) min = tmp1+tmp2;
            
        }
        return min;
    }
}
