class Solution {
    public int minFlipsMonoIncr(String S) {
        int n = S.length();
        int[] oneCount = new int[n+1];
        int[] zeroCount = new int[n+1];
        for (int i = 1; i <= n; i++) {
            oneCount[i] = oneCount[i-1];
            zeroCount[i] = zeroCount[i-1];
            if (S.charAt(i-1) == '1') {
                oneCount[i]++;
            } else 
{
                zeroCount[i]++;
            }
            //System.out.println(zeroCount[i] + " " + oneCount[i]);
        }
        
        int res = Integer.MAX_VALUE;
        for (int i = 1; i <= n; i++) {
            int zeroCost = oneCount[i] + Math.min(oneCount[n] - oneCount[i], zeroCount[n] - zeroCount[i]);
            int oneCost = zeroCount[n] - zeroCount[i-1] + Math.min(zeroCount[i-1], oneCount[i-1]);
            //if (i == 2) System.out.println(zeroCost + " " + oneCost);
            res = Math.min(res, Math.min(zeroCost, oneCost));
        }
        return res;
    }
}
