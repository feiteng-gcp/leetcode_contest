class Solution {
    public int minFlipsMonoIncr(String S) {
        int len = S.length();
        if (len <= 1) return 0;
        int []oneOnLeft = new int[len];
        int []zeroOnRight = new int[len];
        int totalOnes = 0;
        //remember to set all to be zero or one!!!
        for (int i = 0; i < len; i++) { // set current position to be zero
            if (S.charAt(i) == '1') totalOnes++;
            oneOnLeft[i] = (S.charAt(i) == '1') ? 1 : 0;
            if (i > 0)
                oneOnLeft[i] += oneOnLeft[i-1];
        }

        for (int i = len - 2; i >= 0; i--) {
            zeroOnRight[i] = zeroOnRight[i + 1] + (S.charAt(i + 1) == '0' ? 1 : 0);
        }

        int totalZero = len - totalOnes;
        
int best = Math.min(totalOnes, totalZero);

        for (int i = 0; i < len; i++) {
            best = Math.min(best, oneOnLeft[i] + zeroOnRight[i]);
        }

        return best;
    }
}
