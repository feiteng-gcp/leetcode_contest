class Solution {
    public int minFlipsMonoIncr(String S) {
        int n = S.length();
        int[] left = new int[n+1];
        int[] right = new int[n+1];
        left[0] = 0;
        for (int i = 0; i < S.length(); i++) {
            left[i+1] = left[i] + (S.charAt(i) == '1'?1:0);
        }
        right[n] = 0;
        for (int i = n-1; i>=0; i--) {
            right[i] = right[i+1] + (S.charAt(i) == '0'?1:0);
        }
        int ans = n;
        for (int i = 0; i <= n; i++) {
            if (left[i] + right[i] < ans) {
                ans = left[i] + right[i];
            }
        }
        return ans;
    }
}
