class Solution {
    public int minFlipsMonoIncr(String S) {
        int n = S.length();
        int[] prefOne = new int[1 + n];
        int[] suffZero = new int[1 + n];
        for (int i = 1; i <= n; i++) {
            prefOne[i] += prefOne[i - 1];
            prefOne[i] += S.charAt(i - 1) - '0';
        }

        for (int i = n - 1; i >= 0; i--) {
            suffZero[i] += suffZero[i + 1];
            suffZero[i] += '1' - S.charAt(i);
        
}

        int res = Integer.MAX_VALUE;
        for (int i = 0; i <= n; i++) {
            int tmp = prefOne[i] + suffZero[i];
            res = Math.min(res, tmp);
        }

        return res;
    }
}
