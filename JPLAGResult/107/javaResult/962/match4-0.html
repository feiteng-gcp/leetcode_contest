class Solution {
    public int minFlipsMonoIncr(String S) {
        int N = S.length();
        int[][] dp = new int[2][N + 1];
        for (int i = 0; i < N; i++) {
            char ch = S.charAt(i);
            if (ch == '0') {
                dp[0][i + 1] = dp[0][i];
                dp[1][i + 1] = Math.min(dp[1][i] + 1, dp[0][i] + 1);
            }
            else {
                dp[0][i + 1] = dp[0][i] + 1;
                dp[1][i + 1] = Math.min(dp[0][i], dp[1][i]);
            }
        }
        return Math.min(dp[0][N], dp[1][N]);
    }
}
