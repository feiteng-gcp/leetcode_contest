class Solution {
    public int minFlipsMonoIncr(String s) {
        if (s == null || s.length() <= 1) {
            return 0;
        }
        
        int res = Integer.MAX_VALUE;
        int[] ones = new int[s.length() + 1];
        int[] zeros = new int[s.length() + 1];
        ones[0] = 0;
        zeros[1] = 0;
        // int oneCount = 0;
        // int zeroCount = 0;
        // boolean hasOne = false;
        for (int i = 1; i <= s.length(); i++) {
            if (s.charAt(i - 1) == '1') {
                ones[i] = Math.min(ones[i - 1], zeros[i - 1]);
                zeros[i] = zeros[i - 1] + 1;
            } else {
                ones[i] = Math.min(ones[i - 1], zeros[i - 1]) + 1;
                zeros[i] = zeros[i - 1];
            
}
        }
        
        return Math.min(ones[s.length()], zeros[s.length()]);
    }
}
