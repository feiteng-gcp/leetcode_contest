class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int len = name.length();
        int tlen = typed.length();
        if(tlen < len) {
            return false;
        }
        
        int i = 0;
        int idx = 0;
        while(i < len) {
            if(idx >= tlen) {
                return false;
            }
            int j = i;
            int k = idx;
            char c = name.charAt(i);
            
            char c2 = typed.charAt(idx);
            
            // 
            while(j + 1 < len && name.charAt(j + 1) == c) {
                j++;
            }
            while(k + 1 < tlen && typed.charAt(k + 1) == c2) {
                k++;
            }
            
            if(c != c2 || k - idx < j - i) {
                return false;
            }
            
            i = j + 1;
            idx = k + 1;
        }
        
        return idx == tlen;
    }
}
