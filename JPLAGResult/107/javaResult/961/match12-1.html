class Solution {
    public boolean isLongPressedName(String name, String typed) {
        List<Pair> l1 = findPairs(name);
        List<Pair> l2 = findPairs(typed);
        if (l1.size() != l2.size()) {
            return false;
        }

        for (int i = 0; i < l1.size(); i++) {
            Pair p1 = l1.get(i);
            Pair p2 = l2.get(i);
            if (p1.c != p2.c) {
                return false;
            }

            if (p1.cnt > p2.cnt) {
                return false;
            }
        }

        return true;
    }

    List<Pair> findPairs(String str) {
        int n = str.length();
        List<Pair> res = new ArrayList<>();
        int i = 0;
        while (i < n) {
            int j = i;
            while (j < n && str.charAt(j) == str.charAt(i)) {
                j++;
            }

            res.add(new Pair(str.charAt(i), j - i));
            i = j;
        }

        return res;
    }
}


class Pair {
    char c;
    int cnt;
    public Pair(char c, int cnt) {
        this.c = c;
        this.cnt = cnt;
    }
}
