class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int m = name.length(), n = typed.length();
        int i = 0, j = 0;
        char pre = name.charAt(0);
        for (i = 0; i < n; i++) 
{
            if (j < m && typed.charAt(i) == name.charAt(j)) {
                pre = name.charAt(j);
                j++;
            }
            else if (typed.charAt(i) != pre) {
                break;
            }
        }
        return j == m && i == n;
    }
}
