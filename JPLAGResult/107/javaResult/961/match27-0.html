class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int n = typed.length();
        int m = name.length();
        if(m > n)
            return false;
        char last = 0;
        int i = 0;
        int j = 0;
        for(; i < n;){
            if(j < m && typed.charAt(i) == name.charAt(j)){
                last = typed.charAt(i);
                i++;
                j++;
            }
            else if(typed.charAt(i) == last){
                i++;
            }
            else
                return false;
        }
        return (j == m)? true : false;
    }
}
