class Solution {
    public boolean isLongPressedName(String name, String typed) {
        if (typed == null || name == null) {
            return false;
        
}
        int i = 0, j = 0;
        while (i < name.length() && j < typed.length()) {
            if (name.charAt(i) != typed.charAt(j)) {
                return false;
            
}
            int iStart = i, jStart = j;
            while (i < name.length() && name.charAt(i) == name.charAt(iStart)) {
                i++;
            }
            // i = len or name[i] != name[iStart]
            while (j < typed.length() && typed.charAt(j) == typed.charAt(jStart)) {
                j++;
            }
            if (j - jStart < i - iStart) {
                return false;
            
}
            //System.out.println(i + " " + j);
        }
        return i == name.length() && j == typed.length();
    }
}
