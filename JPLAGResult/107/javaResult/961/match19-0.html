class Solution {
  public boolean isLongPressedName(String name, String typed) {
    // 1. typed.length() >= name.length()
    if (name.length() > typed.length()) {
      return false;
    }
    
    // 2. typed must have all characters in name
    int nameIdx = 0;
    int typedIdx = 0;
    
    while (nameIdx < name.length() && typedIdx < typed.length()) {
      if (name.charAt(nameIdx) != typed.charAt(typedIdx)) {
        return false;
      }
      
      ++nameIdx;
      ++typedIdx;
      // name.charAt(nameIdx) == typed.charAt(typedIdx)
      if (nameIdx == name.length() || name.charAt(nameIdx) != name.charAt(nameIdx - 1)) {
        // name have new character
        // or consumed all name
        // consume all redundant typed
        while (typedIdx < typed.length() && typed.charAt(typedIdx) == typed.charAt(typedIdx - 1)) {
          ++typedIdx;
        }
      }
    }
    
    return nameIdx == name.length() && typedIdx == typed.length();
  }
}
