class Solution {
    public boolean isLongPressedName(String name, String typed) {
                int n = name.length();
        int m = typed.length();
        if (n > m) {
            return false;
        }

        int i = 0;
        int j = 0;

        for (; i < n && j < m; i++, j++) 
{
            if (name.charAt(i) != typed.charAt(j)) {
//                System.out.println(i);
//                System.out.println(j);
//                System.out.println(2);
                return false;
            }
            int tempI = i;
            int tempJ = j;
            while (i + 1 < n && name.charAt(i + 1) == name.charAt(tempI)) {
                i++;
            }
            
while (j + 1 < m && typed.charAt(j + 1) == typed.charAt(tempJ)) {
                j++;
            }
            if (i - tempI > j - tempJ) {
//                System.out.println(i - tempI);
//                System.out.println(j - tempJ);
//                System.out.println(1);
                return false;
            }
        }
        return i == n && j == m;
    }
}
