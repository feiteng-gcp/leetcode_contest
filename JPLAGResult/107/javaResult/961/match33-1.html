class Solution {
    public boolean isLongPressedName(String name, String typed) {
        char[] ch1 = name.toCharArray(), ch2 = typed.toCharArray();
        int index1 = 0, index2 = 0;
        while (index1 < ch1.length && index2 < ch2.length) {
            int start1 = index1++, start2 = index2++;
            while (index1 < ch1.length && ch1[start1] == ch1[index1])
                index1++;
            while (index2 < ch2.length && ch2[start2] == ch2[index2])
                index2++;
            if (ch1[start1] != ch2[start2] || index1 - start1 > index2 - start2)
                return false;
        }
        return index1 == ch1.length && index2 == ch2.length;
    }
}
