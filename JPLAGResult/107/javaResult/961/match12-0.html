class Solution {
    public boolean isLongPressedName(String name, String typed) {
        List<Pair> nameList = helper(name);
        List<Pair> typedList = helper(typed);
        
        if (nameList.size() != typedList.size()) {
            return false;
        }
        
        for (int i = 0; i < nameList.size(); i++) {
            Pair np = nameList.get(i);
            Pair tp = typedList.get(i);
            if (np.c != tp.c) {
                return false;
            }
            
            if (np.count > tp.count) {
                return false;
            }
        }
        
        return true;
    }
    
    private List<Pair> helper(String str) {
        if (str == null || str.length() == 0) {
            return new ArrayList<Pair>();
        }
        List<Pair> list = new ArrayList<>();
        list.add(new Pair(str.charAt(0), 1));
        for (int i = 1; i < str.length(); i++) {
            char c = str.charAt(i);
            Pair prev = list.get(list.size() - 1);
            if (c == prev.c) {
                prev.count++;
            } else {
                list.add(new Pair(c, 1));
            }
        }
        return list;
    }
    
    class Pair {
        char c;
        int count;
        public Pair(char c, int count) {
            this.c = c;
            this.count = count;
        }
    }
}
