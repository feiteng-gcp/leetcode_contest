class Solution {
    public boolean isLongPressedName(String name, String typed) {
        if (name.length() > 0 && typed.length() == 0) {
            return false;
        }
        if ((name == null || name.length() == 0) && typed.length() > 0) {
            return false;
        }
        char[] a = name.toCharArray();
        char[] b = typed.toCharArray();
        int index1 = 0;
        int index2 = 0;
        while (index1 < name.length()) {
            if (index2 >= typed.length()) {
                return false;
            }
            if (a[index1] != b[index2]) {
                if (index2 == 0) {
                    return false;
                } else if (b[index2] != b[index2 - 1]) {
                    return false;
                } else {
                    index2++;
                }
            } else {
                index1++;
                index2++;
            }
        }
        if (index2 == typed.length()) {
            return true;
        }
        
        while (index2 < typed.length()) {
            if (b[index2] != a[index1 - 1]) {
                return false;
            }
            index2++;
        }
        return true;
    }
}
