class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int i = 0, j = 0;
        char prev = ' ';
        while (i < name.length() || j < typed.length()) {
            char cn = i < name.length() ? name.charAt(i) : '+', ct = j < typed.length() ? typed.charAt(j) : '-';
            if (cn == ct) {
                prev = cn;
                i++;
                j++;
            }else if (prev == ct) {
                j++;
            }else return false
;
        }
        
        return i == name.length() && j == typed.length();
    }
}
