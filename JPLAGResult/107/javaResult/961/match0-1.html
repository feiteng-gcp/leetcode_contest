class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int N = name.length();
        int M = typed.length();
        if (N > M) return false;
        
int i = 0, j = 0;
        while (i < N && j < M) {
        	if (name.charAt(i) != typed.charAt(j)) {
        		return false;
        	}
        	int k = i;
        	while (k < N && name.charAt(i) == name.charAt(k)) {
        		++k;
        	}
        	int l = j;
        	while (l < M && typed.charAt(j) == typed.charAt(l)) {
        		++l;
        	}
        	if (k - i > l - j) {
        		return false;
        	}
        	i = k;
        	j = l;
        }
        return i == N && j == M;
    }
}
