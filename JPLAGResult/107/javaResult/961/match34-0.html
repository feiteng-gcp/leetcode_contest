class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int i1=0, i2=0;
        while (i1<name.length() && i2<typed.length()) {
            if (name.charAt(i1) != typed.charAt(i2)) return false;
            char c = name.charAt(i1);
            while (i1<name.length() && i2<typed.length() && name.charAt(i1)==c && typed.charAt(i2)==c) {
                i1++;
                i2++;
            }
            while (i2<typed.length() && c == typed.charAt(i2)) {
                i2++;
            }
        }
        return i1==name.length() && i2==typed.length();
    }
}
