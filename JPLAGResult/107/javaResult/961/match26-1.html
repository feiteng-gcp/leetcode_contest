class Solution {
    public boolean isLongPressedName(String name, String typed) {
        if(name == null || typed == null) {
            return false;
        }
        char[] nn = name.toCharArray();
        char[] tt = typed.toCharArray();
        int n = nn.length;
        int t = tt.length;
        if(t < n) {
            return false;
        }
        
int i = 0;
        int j = 0;
        
        while(i < n) {
            char c = nn[i];
            int s = i;
            while(i < n && c == nn[i]) {
                i++;
            }
            if( j >= t || tt[j] != c) {
                return false;
            }
            int sj = j;
            while(j < t && c == tt[j]) {
                j++;
            }
            
            if(j-sj < i -s) {
                return false;
            }
        }
        
        return j == t;
        
    }
}
