class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int n = name.length();
        int m = typed.length();
        int i = 0;
        int j = 0;
        while (i < n && j < m) {
            if (name.charAt(i) != typed.charAt(j)) {
                return false;
            }
            i++;
            int count0 = 1;
            while (i < n && name.charAt(i) == name.charAt(i - 1)) {
                count0++;
                i++;
            }
            j++;
            int count1 = 1;
            while (j < m && typed.charAt(j) == typed.charAt(j - 1)) {
                count1++;
                j++;
            }
            if (count0 > count1) {
                return false;
            }
        }
        return i == n && j == m;
    }
}
