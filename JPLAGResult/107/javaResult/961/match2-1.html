class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int N = name.length();
        int M = typed.length();
        int i = 0;
        int j = 0;
        while (i < N) {
            char ch = name.charAt(i);
            int s = i;
            while (i < N && name.charAt(i) == ch) {
                i++;
            }
            int len = (i - s);
            if (j >= M) {
                return false;
            }
            char ch2 = typed.charAt(j);
            int s2 = j;
            while (j < M && typed.charAt(j) == ch2) {
                j++;
            }
            int len2 = (j - s2);
            
if (ch != ch2 || len2 < len) {
                return false;
            }
        }
        return true;
    }
}
