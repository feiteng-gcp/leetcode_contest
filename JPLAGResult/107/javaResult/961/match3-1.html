class Solution {
    public boolean isLongPressedName(String name, String typed) {
        char[] cn = name.toCharArray(); 
        char[] ct = typed.toCharArray(); 
        int i = 0; 
        int j = 0;
        while (i < cn.length && j < ct.length) {
            int count = 1; 
            char c = cn[i];
            while (i < cn.length - 1 && cn[i] == cn[i + 1]) {
                i++;
                count++; 
            }
            if (ct[j] != c) {
                return false; 
            }
            int count2 = 1;
            while (j < ct.length - 1 && ct[j] == ct[j + 1]) 
{
                count2++; 
                j++; 
            }
            if (count2 < count) {
                return false; 
            }
            i++; 
            j++; 
            
        }
        if (i != cn.length || j != ct.length) {
            return false;
        }
        return true; 
        
    }
}
