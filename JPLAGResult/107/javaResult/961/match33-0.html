class Solution {
    public boolean isLongPressedName(String name, String typed) {
        char[] arr1 = name.toCharArray();
        char[] arr2 = typed.toCharArray();
        
        int fir = 0;
        int sec = 0;
        
        while (fir < arr1.length && sec < arr2.length){
            char c1 = arr1[fir];
            char c2 = arr2[sec];
            int count1 = 0;
            int count2 = 0;
            
            while (fir < arr1.length){
                if (arr1[fir] == c1) {
                    count1++;    
                } else {
                    break;
                }
                fir++;
            }
            
            while (sec < arr2.length){
                if (arr2[sec] == c2) {
                    count2++;    
                } else {
                    break;
                }
                sec++;
            }
            
            if (c1 == c2 && count2 >= count1){
                continue;
            } else {
                return false;
            }
        }
        
        return fir == arr1.length && sec == arr2.length;
    }
}
