class Solution {
    public boolean isLongPressedName(String name, String typed) {
        int n = name.length(), m = typed.length();
        int i = 0, j = 0;
        while (i < n || j < m) {
            if (i < n && j < m && name.charAt(i) == typed.charAt(j)) {
                i++;
                j++;
            }
            else if (j > 0 && j < m && typed.charAt(j) == typed.charAt(j-1)) j++;
            else return false;
        }
        return i == n && j == m;
    }
}
