class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int n = graph.length;
        int m = initial.length;
        Arrays.sort(initial);
        int min = n;
        int ret = 0; 
        for (int i = 0; i < m; i++) {
            int target = initial[i]; 
            int count = 0;
            
boolean[] vis = new boolean[n];
            for (int j = 0; j < m; j++) {
                 if (i == j || vis[initial[j]]) {
                     continue; 
                 }
                 
                 Queue<Integer> q = new LinkedList<>();
                 q.add(initial[j]);
                 vis[initial[j]] = true; 
                 
while (!q.isEmpty()) {
                     int tmp = q.poll();
                     count++; 
                     for (int k = 0; k < n; k++) {
                         if (k != target && graph[tmp][k] == 1 && !vis[k]) {
                             vis[k] = true; 
                             q.add(k);
                         } 
                     }
                }
            
}
            if (min > count) {
                min = count;
                ret = target;
            }
        }
        return ret;
    }
}
