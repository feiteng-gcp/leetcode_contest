class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int ans = 0;
        int total = -1;
        Arrays.sort(initial);
        for (int i = 0; i < initial.length; i++) {
            Queue<Integer> queue = new LinkedList<>();
            boolean[] visited = new boolean[graph.length];
            
for (int j = 0; j < initial.length; j++) {
                if (j != i) {
                    queue.add(initial[j]);
                    visited[initial[j]] = true;
                }
            }
            while (!queue.isEmpty()) {
                int cur = queue.poll();
                for (int j = 0; j < graph[cur].length; j++) {
                    if (graph[cur][j] == 1 && j != initial[i]) {
                        if (!visited[j]) {
                            queue.offer(j);
                            visited[j] = true;
                        }
                    }
                }
            }
            int tmp = 0;
            for (int j = 0 ;j < visited.length; j++) if (!visited[j]) {
                tmp++;
            }
            
if (tmp > total) {
                ans = initial[i];
                total = tmp;
            }
        }
        return ans;
    }
}
