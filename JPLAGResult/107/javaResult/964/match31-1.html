class Solution {
    int[] deps;
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int n = graph.length;
        
        deps = new int[n];
        for(int i =0; i < n; i++) {
            deps[i] = i;
        }
        
        for(int i = 0; i < n; i++) {
            for(int j = i+1; j < n; j++) {
                if(graph[i][j] == 1) {
                    connect(i, j);
                }
            }
        }
        Arrays.sort(initial);
        Map<Integer, Integer> gc = new HashMap<>();
        Map<Integer, Set<Integer>> gx = new HashMap<>();
        Set<Integer> set = new HashSet<>();
        for(int bad : initial){
            set.add(bad);
        }
        for(int i = 0; i < n; i++) {
            int f = getf(i);
            if(gc.get(f) == null) {
                gc.put(f, 0);
            }
            gc.put(f, gc.get(f)+1);
            
            if(set.contains(i)) {
                if(gx.get(f) == null) {
                    gx.put(f, new HashSet<>());
                }
                gx.get(f).add(i);
            }
        }
        /*
        System.out.println("Group");
        for(int key : gc.keySet()) {
            System.out.format("father = %d, family size = %d\n", key, gc.get(key));
        }
        
        System.out.println("Infect count");
        for(int key : gx.keySet()) {
            System.out.format("father = %d, infected size = %d\n", key, gx.get(key).size());
        }
        */
        int save = -1;
        int target = Integer.MAX_VALUE;
        for(Integer key: gx.keySet()) {
            if(gx.get(key).size() == 1) {
                if(gc.get(key)>save || (gc.get(key)==save && gx.get(key).iterator().next() < target)) {
                    target = gx.get(key).iterator().next();
                    save = gc.get(key);
                    //System.out.format("save = %d, target = %d\n", save, target);
                }
            }
            else {
                //more than 1 
                
                if(gc.get(key)>save) {
                    Set<Integer> xset = gx.get(key);
                    for(
Integer ex : xset) {
                        int temp = 0;
                        Set<Integer> connected = new HashSet<>();
                        Queue<Integer> queue = new LinkedList<>();
                        for(Integer infact : xset) {
                            if(infact != ex) {
                                queue.offer(infact);
                                connected.add(infact);
                            }
                        
}
                        
                        while(!queue.isEmpty()) {
                            int pro = queue.poll();
                            for(int i = 0; i < n; i++) {
                                if(i != ex && graph[pro][i] == 1 && !connected.contains(i)) {
                                    queue.offer(i);
                                    connected.add(i);
                                }
                            }
                        }
                        
                        temp = connected.size();
                        //System.out.format("I am here %d\n",gc.get(key));
                        //System.out.format("ex = %d, temp= %d\n", ex, temp);
                        if(gc.get(key) - temp > save || (gc.get(key) - temp == save && ex < target)) {
                            target = ex;
                            save = gc.get(key) - temp;
                        }
                    }
                    //System.out.format("save = %d, target = %d\n", save, target);
                }
            }
        }
        
        return save == -1 ? initial[0] : target;
    }
    
    private void connect(int i, int j) {
        int jf = getf(j);
        int ifa = getf(i);
        deps[jf] = ifa;
        return;
    }
    
    private int getf(int i) {
        if(deps[i] == i) {
            return i;
        }
        else return getf(deps[i]);
    }
}
