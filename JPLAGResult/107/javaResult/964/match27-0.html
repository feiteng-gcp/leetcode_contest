class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        
        int n = graph.length;
        List<Integer>[] neighbors = (List<Integer>[])new List<?>[n];
        
        for (int i = 0; i < n; ++i) {
            neighbors[i] = new ArrayList<Integer>();
            for (int j = 0; j < n; ++j) {
                if (j == i) continue;
                if (graph[i][j] == 0) continue;
                
                neighbors[i].add(j);
            }
        }
        
        Arrays.sort(initial);
        int min = bfs(neighbors, initial, initial[0]);
        int res = initial[0];
        
        for (int i = 1; i < initial.length; ++i) {
            int tmp = bfs(neighbors, initial, initial[i]);
            if (tmp < min) {
                min = tmp;    
                res = initial[i];
            }
        }
        
        return res;
    }
    
    private int bfs(List<Integer>[] neighbors, int[] initial, int ex) {
        // return the number of infected nodes
        
        List<Integer> queue = new LinkedList<Integer>();
        Set<Integer> visited = new HashSet<Integer>();
        
        for (int i: initial) {
            if (i == ex) continue;
            queue.add(i);
            visited.add(i);
        }
        
        while (!queue.isEmpty()) {
            
            int m = queue.size();
            for (int i = 0 ; i < m; ++i) {
                int cur = queue.remove(0);
                // find the neighbors
                for (int next: neighbors[cur]) {
                    if (next == ex || visited.contains(next)) continue;
                    queue.add(next);
                    visited.add(next);
                }
                
            }
        }
        return visited.size();
        
    }
}
