class Solution {
    
    public void DFS(Graph g, int i, int n, boolean[] visited){
        visited[i] = true;
        
        List<Integer> edges = g.V.get(i);
        
        for(int k=0; k < edges.size(); k++){
            int v = edges.get(k);
            if(!visited[v]){
                DFS(g, v, n, visited);
            }
        }
    }
    
    
    public int minMalwareSpread(int[][] graph, int[] a) {
        int n = graph.length;
        Graph g = new Graph(n);
        
        for(int i=0; i < n; i++){
            for(int j=0; j < n; j++){
                if(i == j){
                    continue;
                }
                if(graph[i][j] == 1){
                    g.addEdge(i, j);
                }
            }
        }
        Arrays.sort(a);
        
        int l = a.length;
        int answer = -1;
        int soln = -1;
        for(int j=0; j < l; j++){
            boolean[] visited = new boolean[n];
            visited[a[j]] = true;
            
            for(int i=0; i < l; i++){
                if(!visited[a[i]]){
                    DFS(g, a[i], n, visited);
                }
            }
            
            int count = 0;
            for(int i=0; i < n; i++){
                if(!visited[i]){
                    count++;
                }
            }
            
            if(count > answer){
                soln = a[j];
                answer = count;
            }
        }
        return soln;
    }
}

class Graph {
    List<List<Integer>> V;
    public Graph(int n){
        V = new ArrayList<>();
        for(int i=0; i < n; i++){
            V.add(new ArrayList<>());
        }
    }
    
    public void addEdge(int u, int v){
        V.get(u).add(v);
    }
}
