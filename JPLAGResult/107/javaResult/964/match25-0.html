class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Map<Integer,Set<Integer>> map = new HashMap<>();
        for (int i = 0; i < graph.length; i++){
            map.put(i, new HashSet<>());
            int[] temp = graph[i];
            for (int j = 0; j < temp.length; j++){
                if (temp[j] == 1){
                    map.get(i).add(j);
                }
            }
        }
        
        int len = graph.length;
        int[] arr = new int[len];
        Arrays.sort(initial);
        int res = initial[0];
        int total = getTotal(map, initial, -1);
        //System.out.println(total);
        int maxSave = 0;
        
        for (int num: initial){
            int save = total - getTotal(map, initial, num);
            //System.out.println(save);
            if (save > maxSave){
                maxSave = save;
                res = num;
            }
            //System.out.println(num + " res " + res);
        }
        
        return res;
    }
    
    private int getTotal(Map<Integer,Set<Integer>> map, int[] initial, int remove){
        int[] values = new int[map.keySet().size()];
        //0 not visited / not affected, 1 affected
        
        Queue<Integer> queue = new LinkedList<>();
        for (int num: initial){
            if (num != remove){
                values[num] = 1;
                queue.offer(num);
            }
        }
        
        while (!queue.isEmpty()){
            int cur = queue.poll();
            for (int next: map.get(cur)){
                if (values[next] == 0 && next != remove){
                    values[next] = 1;
                    queue.offer(next);
                }
            }
        }
        
        int count = 0;
        for (int num : values){
            if (num == 1) count++;
        }
        
        // for (int num : values){
        //     System.out.print(num + " ");
        // }
        // System.out.println();
        
        return count;
    }
}
