class Solution
{
    public int minMalwareSpread(int[][] graph, int[] initial)
    {
        if (initial.length == 1) return initial[0];
        Arrays.sort(initial);
        int min = Integer.MAX_VALUE, minNode = initial[0];
        for (int toRemove : initial)
        {
            HashSet<Integer> visited = new HashSet<>();
            Queue<Integer> queue = new LinkedList<>();
            visited.add(toRemove);
            for (int init : initial)
            {
                if (!visited.contains(init))
                {
                    queue.add(init);
                    visited.add(init);
                }
            }
            while (!queue.isEmpty())
            {
                int h = queue.poll();
                for (int next = 0; next < graph[h].length; next++)
                {
                    if (graph[h][next] == 1 && !visited.contains(next))
                    {
                        queue.add(next);
                        visited.add(next);
                    }
                }
            }
            if (min > visited.size())
            {
                min = visited.size();
                minNode = toRemove;
            }
        }
        return minNode;
    }
}
