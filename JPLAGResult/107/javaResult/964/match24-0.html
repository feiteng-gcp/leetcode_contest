class Solution {
    
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Map<Integer, Set<Integer>> myGraph = new HashMap<>();
        for (int i = 0; i < graph.length; i++) {
            myGraph.put(i, new HashSet<>());
            for (int j = 0; j < graph[0].length; j++) {
                if (graph[i][j] == 1) {
                    myGraph.get(i).add(j);
                }
            }
        }
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        for (int num : initial) {
            visited.add(num);
            queue.offer(num);
        }
        while (!queue.isEmpty()) {
            int curr = queue.poll();
            for (int neighbor : myGraph.get(curr)) {
                if (!visited.contains(neighbor)) {
                    visited.add(neighbor);
                    queue.offer(neighbor);
                }
            }
        }
        int smallest = Integer.MAX_VALUE;
        int key = -1;
        Map<Integer, Set<Integer>> badGraph = new HashMap<>();
        for (int num : visited) {
            badGraph.put(num, new HashSet<>());
            for (int j = 0; j < graph[num].length; j++) {
                if (graph[num][j] == 1 && visited.contains(j)) {
                    badGraph.get(num).add(j);
                }
            }
        }
        for (int num : initial) {
            int level = getLevel(badGraph, num, initial);
            if (level < smallest) {
                smallest = level;
                key = num;
            } else if (level == smallest) {
                if (num < key) {
                    key = num;
                }
            }
        }
        return key;
    }
    
    private int getLevel(Map<Integer, Set<Integer>> graph, int delete, int[] initial) {
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        for (int num : initial) {
            if (num != delete) {
                visited.add(num);
                queue.offer(num);
            }
        }
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int curr = queue.poll();
                for (int neighbor : graph.get(curr)) {
                    if (!visited.contains(neighbor) && neighbor != delete) {
                        visited.add(neighbor);
                        queue.offer(neighbor);
                    }
                }
            }
        }
        return visited.size();
    }
}
