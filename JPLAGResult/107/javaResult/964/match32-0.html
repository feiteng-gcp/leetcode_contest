class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        HashSet<Integer> inf = new HashSet<Integer>();
        Arrays.sort(initial);
        int min = Integer.MAX_VALUE;
        int minn = 0;
        for(int i = 0; i < initial.length; i++){
            inf.add(initial[i]);
        }
        
        for(int i = 0; i < initial.length; i++){
            int infect = countInf(new HashSet<Integer>(inf), initial[i], graph, initial);
            if(infect < min){
                min = infect;
                minn = initial[i];
            }
        }
        return minn;
    }
    
    int countInf(HashSet<Integer> inf, int removed, int[][] graph, int[] initial){
        Deque<Integer> s = new ArrayDeque<Integer>();
        int count = 0;
        for(int i = 0; i < initial.length; i++){
            if(initial[i] != removed){
                s.push(initial[i]);
            }
        }
        while(!s.isEmpty()){
            int cur = s.pop();
            count++;
            for(int i = 0; i < graph[cur].length; i++){
                if(i != cur && graph[cur][i] == 1 && !inf.contains(i)){
                    s.push(i);
                    inf.add(i);
                }
            }
        }
        return count;
    }
}
