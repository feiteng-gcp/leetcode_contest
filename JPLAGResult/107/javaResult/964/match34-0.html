class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int n = graph.length, index = n, max = 1;
        HashSet<Integer> initialSet = new HashSet<>();
        boolean[] visited = new boolean[n];
        Queue<Integer> q = new LinkedList<Integer>();
        for (int i : initial){
            initialSet.add(i);
            index = Math.min(index, i);
        } 
        for (int i : initial){
            if (!visited[i]){
                boolean dup = false;
                int count = 1;
                q.offer(i);
                visited[i] = true;
                // connectedGraph.add(i);
                // HashSet<Integer> connectedGraph = new HashSet<>();
                // connectedGraph.add(i);
                HashSet<Integer> malwareNode = new HashSet<>();
                malwareNode.add(i);
                
                while (!q.isEmpty()){
                    int cur = q.poll();
                    for (int j = 0; j < n; j++){
                        if ((!visited[j]) && (graph[cur][j] == 1)){
                            if (initialSet.contains(j)){
                                dup = true;
                                malwareNode.add(j);
                            }
                            q.offer(j);
                            visited[j] = true;
                            count++;
                        }
                    }
                }
                
                if ((!dup) && (count > max)){
                    max = count;
                    index = i;
                }
                
                if (dup){
                    for (int possibleNode : malwareNode){
                        boolean[] v = new boolean[n];
                        int savedNode = 1;
                        for (int j = 0; j < n; j++){
                            if ((!v[j]) && (graph[possibleNode][j] == 1) && (j != possibleNode)){
                                savedNode += bfs(v, possibleNode, j, initialSet, n, graph);
                            }
                        }
                        // System.out.println(possibleNode);
                        // System.out.println(savedNode);
                        if (savedNode > max || ((savedNode == max) && (possibleNode < index))){
                            max = savedNode;
                            index = possibleNode;
                        }
                    }
                }
            }
        }
        return index;
    }
    
    int bfs(boolean[] v, int possibleNode, int initialNode, HashSet<Integer> initialSet, int n, int[][] graph){
        
        Queue<Integer> q = new LinkedList<Integer>();
        q.offer(initialNode);
        v[initialNode] = true;
        boolean encounterMal = false;
        if (initialSet.contains(initialNode)){
            encounterMal = true;
        }
        int count = 1;
        
        while (!q.isEmpty()){
            int cur = q.poll();
            for (int j = 0; j < n; j++){
                if ((j != possibleNode) && (!v[j]) && (graph[cur][j] == 1)){
                    if (initialSet.contains(j)){
                        encounterMal = true;
                    }
                    q.offer(j);
                    v[j] = true;
                    count++;
                }
            }
        }
        //System.out.printf("init: %d, poss: %d, count: %d  %b\n", initialNode, possibleNode, count, encounterMal);
        if (encounterMal){
            return 0;
        } else {
            return count;
        }
    }
}
