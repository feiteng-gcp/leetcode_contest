class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
    List<List<Integer>> nodes = new ArrayList<>();
        for (int i = 0; i < graph.length; i++) nodes.add(new ArrayList<>());
        for (int i = 0; i < graph.length; i++) {
            for (int j = 0; j < graph[i].length; j++) {
                if (i != j && graph[i][j] == 1) nodes.get(i).add(j);
            }
        }
        
        int remove = 0, affectMax = Integer.MAX_VALUE;
        Arrays.sort(initial);
        for (int i : initial) {
            int cnt = getCount(nodes, i, initial);
            if (cnt < affectMax) {
                remove = i;
                affectMax = cnt;
            }
        }
        return remove;
    }
    
    private int getCount(List<List<Integer>> nodes, int remove, int[] initial) {
        Queue<Integer> queue = new LinkedList<>();
        boolean[] visited = new boolean[nodes.size()];
        int cnt = 0;
        for (int i : initial) {
            if (i != remove) {
                queue.offer(i);
                cnt++;
                visited[i] = true;
            }
        }
        while (!queue.isEmpty()) {
            int cur = queue.poll();
            for (int next : nodes.get(cur)) {
                if (!visited[next] && next != remove) {
                    visited[next] = true;
                    queue.offer(next);
                    cnt++;
                }
            }
        }
        return cnt;
    }
}
