class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int total = getNum(graph,initial,-1);
        
        int max = 0;
        int index = 0;
        for(int i=0;i<initial.length;i++){
            int num = getNum(graph,initial,initial[i]);
            if(total-num>max){
                max=total-num;
                index=initial[i];
            }else if(total-num==max){
                if(index>initial[i]){
                    index=initial[i];
                }
            }
        }
        
        
        return index;
    }
    
    public int getNum(int[][] graph, int[] initial, 
int removed) {
        HashSet<Integer> visited = new HashSet<Integer>();
        Queue<Integer> queue = new LinkedList<Integer>();
        
        visited.add(removed);
        for(int i=0;i<initial.length;i++){
            if(!visited.contains(initial[i])){
                visited.add(initial[i]);
                queue.offer(initial[i]);
            }
        }
        
        int ans = 0;
        
while(!queue.isEmpty()){
            int one = queue.poll();
            ans++;
            for(int i=0;i<graph.length;i++){
                if(graph[i][one]==1){
                    if(!visited.contains(i)){
                        visited.add(i);
                        queue.offer(i);
                    }
                }
            }
        }
        return ans;
    }
}
