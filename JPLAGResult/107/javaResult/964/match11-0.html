class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        if (initial.length == 1) {
            return initial[0];
        }
        int n = graph.length;
        int min = Integer.MAX_VALUE;
        int ans = -1;
        for (int i : initial) {
            int sum = bfs(graph, initial, i);
            if (min == sum) {
                ans = Math.min(i, ans);
            } else if (min > sum) {
                min = sum;
                ans = i;
            }

        }
        return ans;

    }

    private int bfs(int[][] graph, int[] initial, int m) {
        int n = graph.length;

        for (int i = 0; i < graph.length; i++) {
            if (graph[i][m] == 1) {
                graph[i][m] = -1;
                graph[m][i] = -1;
            }
        }
        
        Queue<Integer> queue = new LinkedList<>();
        Set<Integer> set = new HashSet<>();
        for (int j : initial) {
            if (j != m) {
                queue.offer(j);
                set.add(j);
            }
        }



        while (!queue.isEmpty()) {
            int cur = queue.poll();
            for (int i = 0; i < n; i++) {
                if (i == cur) {
                    continue;
                }
                if (graph[i][cur] == 1 && set.add(i)) {
                    queue.offer(i);
                }
            }
        }

        for (int i = 0; i < graph.length; i++) {
            if (graph[i][m] == -1) {
                graph[i][m] = 1;
                graph[m][i] = 1;
            }
        }

        return set.size();
    }
}
