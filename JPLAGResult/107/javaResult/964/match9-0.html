class Solution {
  public int minMalwareSpread(int[][] graph, int[] initial) {
    int N = graph.length;
    Arrays.sort(initial);
    // 1. build adjacency list
    List[] adjList = new ArrayList[N];
    for (int node = 0; node < N; ++node) {
      adjList[node] = new ArrayList<Integer>();
      for (int nextNode = 0; nextNode < N; ++nextNode) {
        if (nextNode != node && graph[node][nextNode] == 1) {
          adjList[node].add(nextNode);
        }
      }
    }
    
    // 2. dfs to get total infected
    int totalInfCount = 0;
    boolean[] visited = new boolean[N];
    for (int node : initial) {
      totalInfCount += dfs(node, adjList, visited);
    }
    
    int res = initial[0];
    int bestSavedCount = 0;
    // 3. dfs to get effect of removing each
    for (int initToRemove : initial) {
      int newInfCount = 0;
      visited = new boolean[N];
      visited[initToRemove] = true;
      for (int node : initial) {
        newInfCount += dfs(node, adjList, visited);
      }
      int savedCount = totalInfCount - newInfCount;
      if (savedCount > bestSavedCount) {
        res = initToRemove;
        bestSavedCount = savedCount;
      }
    }
    
    return res;
  }
  
  // returns total visited node
  int dfs(int node, List<Integer>[] adjList, boolean[] visited) {
    if (visited[node]) {
      return 0;
    }
    
    visited[node] = true;
    int res = 1;
    for (int nextNode : adjList[node]) {
      res += dfs(nextNode, adjList, visited);
    }
    return res;
  }
}
