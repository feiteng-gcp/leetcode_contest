class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int minInfected = Integer.MAX_VALUE;
        int minId = Integer.MAX_VALUE;
        for (int num: initial) {
            int infected = numOfInfected(graph, initial, num);
            if (minInfected >= infected) {
                if (minInfected > infected || minId > num) {
                    minInfected = infected;
                    minId = num;
                }
            }
        }
        return minId;
    }
    
    private int numOfInfected(int[][] graph, int[] initial, int removeInx) {
        int n = graph.length;
        Queue<Integer> queue = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        for (int num: initial) {
            if (num != removeInx) {
                queue.add(num);
                visited.add(num);
            }
        }
        while (!queue.isEmpty()) {
            int cur = queue.poll();
            for (int j = 0; j < graph[0].length; j++) {
                if (graph[cur][j] == 0 || j == removeInx) {
                    continue;
                }
                if (visited.contains(j)) {
                    continue;
                }
                queue.add(j);
                visited.add(j);
            }
        }
        return visited.size();
    } 
}
