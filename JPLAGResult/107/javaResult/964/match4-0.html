class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int bestDead = 1234567890;
        int bestNode = 1234567890;
        for (int avoid : initial) {
            int[] infected = infect(graph, initial, avoid);
            int dead = 0;
            for (int val : infected) {
                dead += val;
            }
            // System.out.format("avoid = %d, dead = %d\n", avoid, dead);
            if (dead < bestDead || (dead == bestDead && avoid < bestNode)) {
                bestDead = dead;
                bestNode = avoid;
            }
        }
        return bestNode;
    }
    
    public static int[] infect(int[][] graph, int[] initial, int avoid) {
        final int N = graph.length;
        Queue<Integer> q = new ArrayDeque<>();
        int[] seen = new int[N];
        for (int v : initial) {
            if (v == avoid) {
                continue;
            }
            seen[v] = 1;
            q.offer(v);
        }
        while (!q.isEmpty()) {
            int u = q.poll();
            for (int v = 0; v < N; ++v) {
                if (graph[u][v] == 0 || seen[v] != 0 || v == avoid) {
                    continue;
                }
                seen[v] = 1;
                q.offer(v);
            }
        }
        return seen;
    }
    
    public static class Node extends ArrayList<Node> {
    }
}
