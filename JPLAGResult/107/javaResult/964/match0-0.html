class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int min = Integer.MAX_VALUE, res = Integer.MAX_VALUE;
        Arrays.sort(initial);
        for (int i: initial) {
            int tmp = helper(graph, i, initial);
            // System.out.println(tmp);
            if (tmp < min) {
                min = tmp;
                res = i;
            }
        }
        return res;
    }
    private int helper(int[][] g, int ban, int[] init) {
        int res = 0;
        Queue<Integer> q = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        for (int i: init) {
            if (i != ban) q.add(i);
            visited.add(i);
        }
        while (!q.isEmpty()) {
            int cur = q.poll();
            res++;
            for (int i = 0; i < g.length; i++) {
                if (i != ban && g[cur][i] == 1 && !visited.contains(i)) {
                    q.add(i);
                    visited.add(i);
                }
            }
        }
        return res;
    }
}
