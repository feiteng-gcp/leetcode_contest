class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Arrays.sort(initial);
        List<Integer> list = new LinkedList<>();
        for(int i : initial) list.add(i);
        List<List<Integer>> edges = new ArrayList<>();
        for(int i = 0; i < graph.length; i++) {
            edges.add(new ArrayList<>());
            for(int j = 0; j < graph[0].length; j++) {
                if(graph[i][j] == 1) {
                    edges.get(i).add(j);
                }
            }
        }
        int num = infect(edges, list, graph.length);
        int mn = num-1;
        int ans = initial[0];
        // Remove each infected node in the initial array
        for(int i = 0; i < list.size(); i++) {
            int tmp = list.remove(i);
            num = infect(edges, list, tmp);
            if(num < mn) {
                ans = tmp;
                mn = num;
            }
            list.add(i, tmp);
        }
        return ans;
    }
    // BFS to find how many nodes are infected
    private int infect(List<List<Integer>> edges, List<Integer> initial, int removed) {
        Queue<Integer> queue = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        for(int i : initial) {
            queue.add(i);
            visited.add(i);
        }
        while(!queue.isEmpty()) {
            int cur = queue.poll();
            if (cur == removed) continue;
            for(int next : edges.get(cur)) {
                if (next == removed) continue;
                if(!visited.contains(next)){

                    queue.add(next);
                    visited.add(next);
                }
            }
        }
                    // System.out.printf("(%d, %d)", removed, visited.size());

        // for (int x : visited) {
        //     System.out.println(x);
        // }
        return visited.size();
    }
}
