class Solution {
public:
    int zeros[20010];
    int minFlipsMonoIncr(string s) {
        if(s[0] == '0') zeros[0] = 1;

        else zeros[0] = 0;
        int n = s.length();
        for(int i = 1; i < n; i++)
            zeros[i] = zeros[i-1] + (s[i] == '0');
        int ans = zeros[n-1];
        for(int i = 0; i < n; i++) {
            int t = (i+1-zeros[i]) + (zeros[n-1]-zeros[i]);
           // cout << "i = " << i << "t = " << t << endl;
            ans = min(ans,t);
        }
        return ans;
    }
};
