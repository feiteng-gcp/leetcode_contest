class Solution {
public:
    int minFlipsMonoIncr(string S) {
              vector<int> zero( S.size(), 0 );
      vector<int> one( S.size(), 0 );

      if( S[0] == '0' ) {
        zero[0] = 0;
        one[0] = 1;
      } else {
        zero[0] = 1;
        one[0] = 0;
      }

      for( int i = 1; i < S.size(); i++ ) {
        if( S[i] == '0' ) {
          // it wants to stay as 0? if previous op result is 0, don't move, if previous op result is 1, not allowed to stay 0
          zero[i] = zero[i-1] + 0;
          // it wants to flip to 1? if previous op result is 0, then add 1, if previous op result is 1, also add 1
          one[i] = min( zero[i-1] + 1, one[i-1] + 1 );
        } else {
          // it wants to flip to 0?
          zero[i] = zero[i-1] + 1;
          // it wants to state as 1? choose the smaller resulted #of op resulted from previous states
          one[i] = min( zero[i-1] + 0, one[i-1] + 0 );
        }
      }
      return min( zero.back(), one.back() );

    }
};
