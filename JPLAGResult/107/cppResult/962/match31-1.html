class Solution {
public:
    int minFlipsMonoIncr(string S) {
        vector<int> ones;
        int tmp =0;
        for(int i=0;i<S.size();i++)
        {
            if(S[i]== '1')
            {
                tmp++;
            }
            ones.push_back(tmp);
        }
        if(tmp==0) return 0;
        if(tmp == S.size()) return 0;
        int min = tmp<(S.size()-tmp)?tmp:(S.size()-tmp);

        for(int i=0;i<S.size();i++)
        {
            
            int tmin = ones[i] + (S.size()-i-1-(tmp-ones[i]));
            if(tmin<min)
            {
                min = tmin;
            }
        }
        return min;
    }
};
