class Solution {
public:
    int minFlipsMonoIncr(string S) {
        vector<vector<int>> f(S.length(), vector<int>(2));
        if (S[0] == '0') {
            f[0][0] = 0; f[0][1] = 1;
        }else{
            f[0][0] = 1; f[0][1] = 0;
        }
        for (int i = 1;i<S.length();i++) {
            if (S[i]=='0') {
                f[i][0] = f[i-1][0];
                f[i][1] = f[i-1][1] + 1;
            }else{
                f[i][1] = min(f[i-1][0], f[i-1][1]);
                f[i][0] = f[i-1][0] + 1;
            }
        }
        return min(f[S.length()-1][0], f[S.length()-1][1]);
    }
};
