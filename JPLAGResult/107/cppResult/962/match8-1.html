class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n = S.length();
        vector<int> zero(n, 0);
        vector<int> one(n, 0);
        for (int i = 0; i < n; ++i) {
            if (i > 0) {
                zero[i] = zero[i - 1];
                one[i] = one[i - 1];
            }
            if (S[i] == '0')
                zero[i]++;
            else

                one[i]++;
        }
        int ans = zero[n-1] - zero[0];
        for (int i = 1; i < n; ++i) {
            ans = min(ans, one[i-1] + zero[n-1] - zero[i]);
        }
        return ans;
    }
};
