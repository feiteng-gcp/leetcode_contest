class Solution {
public:
    int minFlipsMonoIncr(string S) {
        vector<int> f(S.size()+1);
        vector<int> g(S.size()+1);
        f[0] = 0;
        for(int l = 1; l <= S.size(); ++l) {
            f[l] = f[l-1] + (S[l-1] == '1');
        }
        
        g[S.size()] = 0;
        for(int l = S.size()-1; l >= 0; --l) {

            g[l] = g[l+1] + (S[l] == '0');
        }
        int best = 0x7fffffff;
        for(int i = 0; i <= S.size(); ++i) {
            best = min(best, f[i] + g[i]);
        }
        return best;
    }
};
