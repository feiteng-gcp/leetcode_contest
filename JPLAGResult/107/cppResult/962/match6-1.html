class Solution {
public:
    int minFlipsMonoIncr(string S) {

        vector<int> ones(S.size() + 1);
        vector<int> zeros(S.size() + 1);
        ones[0] = 0;
        for (int i = 1; i <= S.size(); ++i) {
            if (S[i - 1] == '1') {
                ones[i] = ones[i - 1] + 1;
            } else {
                ones[i] = ones[i - 1];
            }
        }
        zeros[S.size()] = 0;
        for (int i = S.size() - 1; i >= 0; --i) {
            if (S[i] == '0') {
                zeros[i] = zeros[i + 1] + 1;
            } else {
                zeros[i] = zeros[i + 1];
            }
        }
        int min = S.size();
        for (int i = 0; i <= S.size(); ++i) {
            int t = ones[i] + zeros[i];
            if (t < min) {
                min = t;
            }
        }
        return min;
    }
};
