class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n = S.length();
        vector<int> need_flip_1(n + 2, 0);
        for (int i = 1; i <= n; ++i) {
            need_flip_1[i] = need_flip_1[i - 1] + (S[i - 1] == '1');
        }
        need_flip_1[n + 1] = need_flip_1[n];
        
        vector<int> need_flip_0(n + 2, 0);
        for (int i = n; i >= 1; --i) {
            need_flip_0[i] = need_flip_0[i + 1] + (S[i - 1] == '0');
        }

        need_flip_0[0] = need_flip_0[1];
        
        int ret = INT_MAX;
        for (int i = 0; i <= n + 1; ++i) {
            ret = min(ret, need_flip_1[i - 1] + need_flip_0[i]);
            // cout << i << " " << need_flip_1[i - 1] << " " << need_flip_0[i] << endl;
        }
        return ret;
        
    }
};
