const int INF = 1000000000;


class Solution {
public:
  int minFlipsMonoIncr(string S) {
    int n = S.size();

    vector<int> zero;
    zero.push_back(0);

    for (int i = 0; i < n; ++ i) {
      if (S[i] == '0') {
        zero.push_back(zero.back() + 1);
      } else {
        zero.push_back(zero.back());
      }
    }


    int ans = INF;
    for (int i = 0; i <= n; ++ i) {
      int left = i - zero[i];
      int right = zero[n] - zero[i];
      ans = min(ans, left + right);
    }
    return ans;
  }
};
