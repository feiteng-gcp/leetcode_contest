class Solution {
public:
    int minFlipsMonoIncr(string S) {
      int n = S.size();
      vector<int> f(n+1, 0);
      for (int i = 0; i < n; ++i )
        f[i+1] = f[i] + (S[i] == '1');
      int ans = n;
      for (int i = 0; i <= n; ++i ) {
        int a = f[i];
        int b = (n-i) - (f[n]-f[i]);
        ans = min(ans, b+a);
      }
      return ans;
    }
};
