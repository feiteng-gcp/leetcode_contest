class Solution {
public:
    int minFlipsMonoIncr(string S) {
        vector<int> v0(S.size()), v1(S.size());

        for (int i = S.size() - 1; i > 0; i--) {
            if (S[i] == '0') {
                v0[i-1] = v0[i] + 1;
            } else {
                v0[i-1] = v0[i];
            }
        }

        for (int i = 0; i + 1 < S.size(); i++) {
            if (S[i] == '1') {
                v1[i + 1] = v1[i] + 1;
            } else {
                v1[i + 1] = v1[i];
            }
        }
        int ans = INT_MAX;
        for (int i = 0; i < S.size(); i++) {
            ans =  min(ans, v0[i] + v1[i]);
        }
        return ans;
    }
};
