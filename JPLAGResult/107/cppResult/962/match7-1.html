class Solution {
public:
    int minFlipsMonoIncr(string s) {
        int n = (int)s.length();
        vector<int> front1(n, 0), back0(n + 1, 0);
        for (int i = 0; i < n; ++i) {
            front1[i] = (i ? front1[i - 1] : 0) + (s[i] == '1');

        }
        back0[n] = 0;
        for (int i = n - 1; i >= 0; --i) {
            back0[i] = back0[i + 1] + (s[i] == '0');
        }

        int res = back0[0];
        for (int i = 0; i < n; ++i) {
            res = min(res, front1[i] + back0[i + 1]);
        }

        return res;
    }
};
