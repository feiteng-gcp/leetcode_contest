class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int len = S.length();
        if(len<=1) return 0;
        vector<int> a(len,0);

        vector<int> b(len,0);
        int k=0;
        for(int i=0;i<len;i++)
        {
            if(S[i]=='1')
                k++;
            b[i] = k;
        }
        k=0;
        for(int i=len-1;i>=0;i--)
        {
            if(S[i]=='0')
                k++;

            a[i] = k;
        }
        int res = min(a[0],b[len-1]);
        for(int i=1;i<len;i++)
        {
            res = min(res, a[i]+b[i-1]);
        }
        return res;
    }
};
