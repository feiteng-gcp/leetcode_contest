class Solution {
public:
    int minFlipsMonoIncr(string S) {
        vector<int> v0(S.size() + 1), v1(S.size() + 1);
        for (int i = 0; i < S.size(); i++) {
            v0[i + 1] = v0[i] + (S[i] == '1' ? 1 : 0);
        }
        for (int i = S.size() - 1; i >= 0; i--) {
            v1[i] = v1[i + 1] + (S[i] == '0' ? 1 : 0);

        }
        // for (int i : v0)
        //     cout << i << ' ';
        // cout << endl;
        // for (int i : v1)
        //     cout << i << ' ';
        // cout << endl << endl;
        int mn = INT_MAX;
        for (int i = 0; i <= S.size(); i++) {
            mn = min(mn, v0[i] + v1[i]);
        }
        return mn;
    }
};
