class Solution {
public:
    int minFlipsMonoIncr(string S) {
        
        vector<int> zeros;
        int sum = 0;
        for (int i = 0; i < S.size(); ++i) {
            if (S[i] == '0') {
                sum += 1;
            }
            zeros.push_back(sum);

        }
        
        int min = -1;
        for (int i = 0; i < S.size(); ++i) {
            int x = sum - zeros[i];
            if (i > 0) {
                x += i - zeros[i - 1];
            }
            if (x < min || min == -1) {
                min = x;
            }
        }
        
        return min;
    }
};
