class Solution {
public:
int minFlipsMonoIncr(string S) {
    int one = 0;
    vector<int> ones(S.size(),0);
    for(int i = 0; i < S.size(); i++) {
        if(S[i] == '1')
            one++;

        ones[i] = one;
    }
    
    int m = INT_MAX;
    for(int i = 0; i <= S.size(); i++) {
        int leftOne = i >= 1 ? ones[i-1] : 0;
        int rigtOne = one - leftOne;
        int temp = leftOne + (S.size()-i-rigtOne);
        m = min(m,temp);
    }
    return m;
}
};
