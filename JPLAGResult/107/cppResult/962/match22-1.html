class Solution {
 public:
  int minFlipsMonoIncr(string S) {
    int n = S.size();
    vector<int> left(n + 1), right(n + 1);
    for (int i = 0; i < n; ++i) {
      left[i + 1] = left[i] + (S[i] == '1');
    }
    for (int i = n - 1; i >= 0; --i) {

      right[i] = right[i + 1] + (S[i] == '0');
    }
    int ans = n;
    for (int i = 0; i <= n; ++i) {
      ans = min(ans, left[i] + right[i]);
    }
    return ans;
  }
};
