class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n=S.size();
        vector<int> zeros(n+1,0);
        vector<int> ones(n,0);
        for (int i=1;i<=n;i++) {
            zeros[i]=zeros[i-1]+(S[i-1]=='1');
        }
        ones[n-1]=(S[n-1]=='0');
        for (int i=n-2;i>=0;i--) {
            ones[i]=ones[i+1]+(S[i]=='0');
        }
        int ret=INT_MAX;
        for (int i=0;i<n;i++) {
            ret=min(ret,zeros[i]+ones[i]);
        }
        ret=min(ret,zeros[n]);
        return ret;
    }
};
