class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n = S.length();
        vector<int> left(1+n);
        vector<int> right(1+n);
        
        left[0] = 0;
        for(int i = 1; i <= n; ++i) {
            if (S[i-1] == '1') {
                left[i] = 1 + left[i-1];
            }
            else {
                left[i] = left[i-1];
            }
        }
        
        right[n] = 0;
        for (int i = n-1; i >= 0; --i) {
            if (S[i] == '0') {
                right[i] = 1 + right[i+1];
            }
            else {
                right[i] = right[i+1];
            }
        }
        
        int ans = INT_MAX;
        for(int i = 0; i <= n; ++i) {
            ans = min(ans, left[i] + right[i]);
        }
        return ans;
    }
};
