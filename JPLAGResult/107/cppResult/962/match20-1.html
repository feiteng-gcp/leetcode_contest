class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n = S.size();
        vector<int> dp0(n, 0);

        vector<int> dp1(n, 0);
        for(int i=0; i<n;i++){
            if(i>0){
                dp0[i] = dp0[i-1];
            }
            if(S[i]=='1'){
                dp0[i] += 1;
            }
        }
        for(int i=n-1; i>=0;i--){
            if(i<n-1)

                dp1[i] = dp1[i+1];
            if(S[i]=='0'){
                dp1[i] += 1;
            }
        }
        int res = INT_MAX;
        for(int i=0; i<n-1;i++){
            res = min(res, dp0[i]+dp1[i+1]);
        }
        res = min(res, min(dp0[n-1], dp1[0]));
        return res;
    }
};
