class Solution {
public:
int minFlipsMonoIncr(string str) {
    int n = str.length();

    vector<int> dp1(n, 0);  // starting with 0
    vector<int> dp2(n, 0);  // starting with 1
    
    if (str[n - 1] == '0') {
        dp1[n - 1] = 0;
        dp2[n - 1] = 1;
    } else {
        dp1[n - 1] = 1;
        dp2[n - 1] = 0;
    }
    
    
    for (int i = n - 2; i >= 0; i--) {
        if (str[i] == '0') {
            dp1[i] = min(dp1[i + 1], dp2[i + 1]);
            dp2[i] = 1 + dp2[i + 1];
        } else {
            dp1[i] = 1 + min(dp1[i + 1], dp2[i + 1]);
            dp2[i] = dp2[i + 1];
        } 
    }
    
    return min(dp1[0], dp2[0]);
}
};
