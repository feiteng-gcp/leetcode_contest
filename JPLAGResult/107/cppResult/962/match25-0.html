class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n = S.size();
        vector<int> c(n);
        c[0] = (S[0] == '1');
        for (int i = 1; i < n; i++)
            c[i] = c[i - 1] + (S[i] == '1');
        int res = n - c.back();
        for (int i = 0; i < n; i++)
        {
            res = min(res, c[i] + (n - i - 1 - (c.back() - c[i])));
        }
        return res;
    }
};
