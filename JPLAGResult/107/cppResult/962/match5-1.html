class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n = S.length();
        
        vector<int> sum_l(n+1, 0);

        for (int i = 0; i < n; i++) {
            if (S[i] == '1') {
                sum_l[i+1] = sum_l[i] + 1;
            } else {
                sum_l[i+1] = sum_l[i];
            }
        }
        
        vector<int> sum_r(n+1, 0);

        for (int i=n-1; i >= 0; i--) {
            if (S[i] == '0') {
                sum_r[i] = sum_r[i+1] + 1;
            } else {
                sum_r[i] = sum_r[i+1];
            }
        }
        
        int mmin = n;
        for (int i = 0; i <= n; i++) {
            mmin = min(mmin, sum_l[i] + sum_r[i]);
        }
        return mmin;
    }
};
