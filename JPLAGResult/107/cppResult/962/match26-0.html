const int N = 20005;

class Solution {
public:
    int minFlipsMonoIncr(string s) {
        int n = s.size();
        int a[N];
        a[0] = (s[0] == '0');
        for (int i = 1; i < n; i++) a[i] = a[i - 1] + (s[i] == '0');
        int ans = min(n - a[n - 1], a[n - 1]);
        for (int i = 1; i < n; i++) {
            int tmp = i - a[i - 1] + a[n - 1] - a[i - 1];
            ans = min(ans, tmp);
        }
        return ans;
    }
};
