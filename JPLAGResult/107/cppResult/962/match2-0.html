class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int len=S.size(),ans=INT_MAX,n0=0,n1=0;
        vector<int> v0(len,0),v1(len,0);
        for(int i=0;i<len;i++){
            v1[i]=n1;
            n1+=S[i]=='1';
        }
        for(int i=len-1;i>=0;i--){
            v0[i]=n0;
            n0+=S[i]=='0';
        }
        for(int i=0;i<len;i++){
            ans=min(ans,v0[i]+v1[i]);
        }
        return ans;
    }
};
