class Solution {
public:
    int minFlipsMonoIncr(string S) {
        int n = S.size();
        vector<int> l(n, 0), r(n + 1, 0);
        for (int i = 0, c = 0; i < n; ++i) {
            if (S[i] == '1') ++c;
            l[i] = c;
        }
        for (int i = n - 1, c = 0; i >= 0; --i) {
            if (S[i] == '0') ++c;
            r[i] = c;
        }
        int ans = r[0];
        for (int i = 0; i < n; ++i) {
            ans = min(ans, l[i] + r[i + 1]);
        }
        return ans;
    }
};
