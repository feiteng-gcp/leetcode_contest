class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
	int N = A.size();
	int one = count(A.begin(), A.end(), 1);
	if (one == 0) {
	    return {0, 2};
	}
	if (one % 3) {
	    return {-1, -1};
	}

	int tail = 0;
	REP (i, N) if (A[i] == 1) tail = N - i - 1;

	int cnt = 0;
	int x = -1, y = -1;
	REP (i, N) {
	    if (A[i] == 1) cnt++;
	    if (cnt * 3 == one) {
		x = i + tail + 1;
		break;
	    }
	}
	cnt = 0;
	for (int i=x; i<N; i++) {
	    if (A[i] == 1) cnt++;
	    if (cnt * 3 == one) {
		y = i + tail + 1;
		break;
	    }
	}
	if (x == -1 || y == -1) {
	    return {-1, -1};
	}

	VI a[3];
	a[0].assign(A.begin(), A.begin()+x);
	a[1].assign(A.begin()+x, A.begin()+y);
	a[2].assign(A.begin()+y, A.end());
	REP (i, 3) {
	    reverse(a[i].begin(), a[i].end());
	    while (!a[i].empty() && a[i].back() == 0) {
		a[i].pop_back();
	    }
	}
	
	if (a[0] != a[1] || a[1] != a[2] || a[0] != a[2]) {
	    return {-1, -1};
	}
	return {x-1, y};
    }
};
