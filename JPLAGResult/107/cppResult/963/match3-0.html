class Solution {
public:
    bool valid0(const string &s, int start, int end){
        for(int i = start; i <= end; i++)
            if(s[i] != '0')
                return false;
        return true;
    }
    
    vector<int> threeEqualParts(vector<int>& A) {
        if(A.size() < 3)
            return {-1, -1};
        string s;
        int num1 = 0;
        for(int i : A){
            num1 += (i == 1 ? 1 : 0);
            s.push_back(i+'0');
        }
        
        if(num1 % 3 != 0)
            return {-1, -1};
        if(num1 == 0)
            return {0, A.size() - 1};
        
        int part1Count = num1 / 3;
        int i = -1, j = s.length(), oneRight = 0;

        while(oneRight < part1Count){
            j--;
            if(s[j] == '1')
                oneRight++;
        }
        string suffix = s.substr(j);
        int first = s.find(suffix);
        if(first < 0 || !valid0(s, 0, first - 1))
            return {-1, -1};
        
        i = first + suffix.length() - 1;
        int second = s.find(suffix, i+1);
        if(second < 0 || second + suffix.length() > j  || !valid0(s, i+1, second-1))
            return {-1, -1};
        
        if(!valid0(s, second + suffix.length(), j-1))
            return {-1, -1};
        return {i, second + suffix.length()};
    }
};
