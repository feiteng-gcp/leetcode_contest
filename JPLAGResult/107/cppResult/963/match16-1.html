class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int n = A.size();
        int i = 0;
        while (i < n && A[i] == 0)
            ++i;
        if (i == n)
            return {0, 2};
        int i0 = i;
        auto check = [&](int s0, int e0, int& s1) {
            while (s1 < n && A[s1] == 0)
                ++s1;
            if (s1 + e0 - s0 > n)
                return false;
            while (s0 < e0) {
                if (A[s0++] != A[s1++])
                    return false;
            }
            return true;
        };

        for (; i < n-2; ++i) {
            int i1 = i+1;
            if (!check(i0, i+1, i1))
                continue;
            int j = i1;
            if (check(i0, i+1, j) && j==n)
                return {i, i1};
        }
        return {-1, -1};
    }
};
