class Solution {
    int help(vector<int>& A, int start, int end, int j) {
        while (A[j] == 0) {
            j++;
        }        
        while (start <= end) {
            if (j == A.size()) {
                return -1;
            }
            if (A[start] == A[j]) {
                start++;
                j++;
            } else {
                return -1;
            }
        }
        return j;
    }
public:
    vector<int> threeEqualParts(vector<int>& A) {

        int one = 0;
        int start = 0;
        for (int i = A.size() - 1; i >= 0; i--) {
            if (A[i] == 1) {
                one++;

                start = i;
            }
        }
        
        if (one == 0) {
            return {0, A.size()-1};
        }
        
        if (one % 3 != 0) {
            return {-1, -1};
        }
        
        one = one / 3;
        
        int i = start;
        while (i < A.size() && one >= 0) {
            if (A[i] == 1) {
                one--;
            }
            
            if (one == 0) {
                int j = help(A, start, i, i+1);
                if (help(A, start, i, j) == A.size()) {
                    return {i, j};
                }
            }

            i++;
        }
        
        return {-1, -1};
    }
};
