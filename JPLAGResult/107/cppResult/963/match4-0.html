const ll MOD=1e9+7;
class Solution {
public:
  bool good(vector<int>& a, int i,int j,int l){
    if(i+l-1>=j) return 0;
    for(int k=0;k<l;k++){
      if(a[i+k]!=a[j+k]) return 0;
    }
    return 1;
  }

  vector<int> threeEqualParts(vector<int>& a) {
    int N=a.size();
    int cnt=0;
    int f=-1;
    int tr=0;
    for(int i=0;i<N;i++){
      if(a[i]==1){
        cnt++;
        if(f==-1) f=i;
      }
    }
    if(cnt==0) return {0,2};
    if(cnt%3) return {-1,-1};
    cnt/=3;
    int j=N-1;
    while(a[j]==0){
      j--;
    }
    tr=N-1-j;
    int x=0;
    while(x<cnt){
      if(a[j]==1){
        x++;
      }
      j--;
    }
    j++;
    int f1=f;
    x=0;
    while(x<cnt){
      if(a[f1]==1){
        x++;
      }
      f1++;
    }
    f1+=tr;
    int ln=f1-f;

    int k=j-1;
    while(a[k]==0){
      k--;
    }
    k+=tr;
    int k1=k+1;
    x=0;
    while(x<cnt){
      if(a[k]==1){
        x++;
      }
      k--;
    }
    k++;

    if(k1-k!=ln || N-j!=ln) return {-1,-1};
    if(!good(a,f,k,ln) || !good(a,f,j,ln)) return {-1,-1};

    return {f1-1,k1};

  }
};
