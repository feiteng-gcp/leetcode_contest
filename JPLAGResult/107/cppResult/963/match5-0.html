class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        
        unordered_map<int, int> M; // number of ones, index
        
        int ones=0;
        for (int i=0; i<A.size(); ++i) {
            if (A[i]) {
                ++ones;
                M[ones]=i;
            }
        }

        if (ones%3!=0) return {-1, -1};
        ones=ones/3;
        if (ones==0) return {0, 2};
        
        int first_one_idx = M[1];
        int second_one_idx = M[ones+1];
        int third_one_idx = M[ones*2 +1];
        
        //cout << first_one_idx << ", " << second_one_idx << ", " << third_one_idx << endl;
        for (int i=0; third_one_idx+i<A.size(); ++i) {
            if (A[i+third_one_idx] != A[i + first_one_idx] ||
                A[i+third_one_idx] != A[i + second_one_idx]) return {-1, -1};
        }
        int len = A.size()-third_one_idx;
        // no need to check zeros
        return {first_one_idx + len - 1, second_one_idx + len};
    }
};
