class Solution {
public:
    vector<int> geterr()
    {
        vector<int> rt;
        rt.push_back(-1);
        rt.push_back(-1);
        return rt;
    }
    vector<int> threeEqualParts(vector<int>& A) {
        vector<int> rt;
        int ones =0;

        for(int i=0;i<A.size();i++){
            if(A[i]==1)
                ones++;
        }
        if(ones == 0){
            rt.push_back(0);
            rt.push_back(2);
            return rt;
        }
        if(ones % 3 != 0){
           
            return geterr();
        }
        ones = ones /3;
        int tmp1=0;
        int onetmp =0;
        for(int i=A.size()-1;i>=0;i--){
            if(A[i]==1)
            {
                onetmp ++;
                if(onetmp == ones)
                {
                    tmp1 = i;
                    break;
                }
            }
        }

        int b1 =0;
        for(int i=0;i<A.size();i++)
        {
            if(A[i]==1)
            {
                b1=i;
                break;
            }
        }
        for(int i=tmp1;i<A.size();i++)
        {
            if(A[i] != A[b1+i-tmp1])
                return geterr();
        }
        int b1end = b1+A.size()-tmp1;
        rt.push_back(b1+A.size()-tmp1-1);
        int b2 =0;
        for(int i=b1end;i<A.size();i++)
        {
            if(A[i]==1)
            {
                b2=i;
                break;
            }
        }
        for(int i=tmp1;i<A.size();i++)
        {
            if(A[i] != A[b2+i-tmp1])
                return geterr();
        }
        rt.push_back(b2+A.size()-tmp1);
        return rt;
    }
};
