class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int ones = 0;
        vector<int> res(2);
        for(int i(0);i<A.size();i++){
            if(A[i]==1)
                ones ++;
        }
        
        res[0]=res[1]=-1;
        if(ones%3!=0)
        {
            return res;
        }
        if(ones==0)
        {
            res[0]=0;
            res[1]=A.size()-1;
            return res;
        }
        int a0(0),a1,b0,b1,c0,c1(A.size()-1),count(0);

        for(int i(0);i<A.size();i++){
            if(A[i]==1)
            {
                count ++;
                if(count == 1)
                    a0 = i;
                if(count == ones/3+1)
                        a1 = i-1;
                
                if(count == ones/3+1)
                    b0 = i;
                
                if(count == ones*2/3+1)
                    b1 = i-1;
                
                if(count == ones*2/3+1)
                    c0 = i;
            }
        }
        cout <<ones<<' '<< a0 <<' '<<a1 <<' '<<b0 << ' '<<b1<<' '<<c0<<' '<<c1<<endl; 
        if(a1<a0+c1-c0 || b1<b0+c1-c0)
            return res;
        a1 = a0+c1-c0,b1=b0+c1-c0;
        for(int i(0);i<=c1-c0;i++){
            if(A[a0+i]!=A[b0+i] || A[a0+i]!=A[c0+i])
                return res;
        }
        res[0] = a1;
        res[1] = b1+1;
        return res;
    }
};
