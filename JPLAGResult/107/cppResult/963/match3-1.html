class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {        
        vector<int> loc;
        for(int i=0;i<A.size();i++){
            if(A[i]==1)
                loc.push_back(i);

        }
        if(loc.empty())
            return {0, 2};
        if(loc.size()%3 != 0)
            return {-1, -1};
        
        int n = loc.size();
        
        int a = loc[0];
        int b = loc[n/3];
        int c = loc[n/3*2];
        
        for(int i=0;i<loc.size()/3;i++){
            if(loc[i]-a != loc[i+n/3]-b)
                return {-1, -1};
            if(loc[i]-a != loc[i+n/3*2]-c)
                return {-1, -1};
        }

        int numzero = A.size()-loc.back()-1;
        cout<<numzero;
        if(loc[n/3] - loc[n/3-1] < numzero+1)
            return {-1, -1};
        if(loc[n/3*2] - loc[n/3*2 - 1] < numzero+1)
            return {-1, -1};
        
        return {loc[n/3-1]+numzero, loc[n/3*2-1]+1+numzero};
    }
};
