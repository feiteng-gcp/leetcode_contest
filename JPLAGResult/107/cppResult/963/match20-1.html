class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int bit_count = 0;
        int N = A.size();
        int first_one = -1;
        for (int pos = 0; pos < A.size(); ++pos) {
            int bit = A[pos];
            if (bit && first_one < 0) {
                first_one = pos;
            }
            bit_count += bit;

        }
        if (bit_count % 3) {
            return {-1, -1};
        }
        if (bit_count == 0) {
            return {0, 2};
        }
        int curr = 0;
        int part_bc = bit_count / 3;
        bit_count = 0;
        while (bit_count < part_bc) {
            bit_count += A[curr++];
        }
        int i = curr - 1;
        int j = N - 1;
        int zero_end = 0;
        while (A[j] == 0 && i < j) {
            zero_end++;
            if (A[++i] != 0) {
                return {-1, -1};
            }
            j--;
        }
        int len = i - first_one + 1;
        if (i >= j) {
            return {-1, -1};
        }
        int k;
        for (k = 0; k < len; ++k) {
            if (A[first_one + k] != A[N - len + k]) {
                return {-1, -1};
            }
        }
        j = N - len;
        int mid_zero_count = 0;
        while (A[--j] == 0) {
            mid_zero_count++;
        }
        if (mid_zero_count < zero_end) {
            return {-1, -1};
        }
        j = N - len - (mid_zero_count - zero_end);
        for (k = 0; k < len; ++k) {

            if (A[N - 1 - k] != A[j - 1 - k]) {
                return {-1, -1};
            }
        }
        return {i, j};
    }
};
