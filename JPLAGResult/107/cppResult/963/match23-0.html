class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        if(A.size()<3)return {-1,-1};
        int c1=0;
        vector<int> p1;
        for (int i=0;i<A.size();++i){
            if (A[i]==1) {
                ++c1;
                p1.push_back(i);
            }
        }
        if (c1==0) {
            return {0,2};
        }
        if (c1%3)return {-1,-1};
        int i0=p1[0],i1=p1[c1/3],i2=p1[c1*2/3];
        int j0=1,j1=c1/3+1,j2=c1*2/3+1;
        while (j0<c1/3) {
            if(p1[j1]-i1!=p1[j0]-i0
                    || p1[j1]-i1!=p1[j2]-i2)return {-1,-1};
            ++j0,++j1,++j2;
        }
        int t0=A.size()-1-p1[c1-1];
        if (i2-p1[c1*2/3-1]-1<t0||i1-p1[c1/3-1]-1<t0)return {-1,-1};
 //       eprintf("i0:%d,i1:%d,i2:%d,t0:%d,p1.back():%d,p1[%d]:%d\n", i0,i1,i2,t0,p1.back(),c1-1,p1[c1-1]);
        return {p1[c1/3-1]+t0,p1[c1*2/3-1]+t0+1};

    }
};
