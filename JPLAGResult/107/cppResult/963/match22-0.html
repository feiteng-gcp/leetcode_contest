class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int ones = accumulate(A.begin(), A.end(), 0);
        int s1 = 0;
        int s2 = 0;
        int s3 = 0;
        
        int m = 0;
        int p = 0;
        int q = 0;
        
        if (ones % 3) return {-1, -1};
        if (ones == 0) return {0, 2};
        
        vector<int> prefix_sum(A.size(), 0);
        
        for (int i = 0; i < A.size(); i++) {
            prefix_sum[i] = A[i] + (i > 0 ? prefix_sum[i-1] : 0);
            //cout << prefix_sum[i] << endl;
        }

        
        for (int i = 0; i < A.size(); i++) {
            if (prefix_sum[i]) {
                s1 = i;
                //cout << "s1" << s1 << endl;
                break;
            }       
        }
        
        
        for (int i = 1; i < A.size(); i++) {
            if (prefix_sum[i] > ones / 3) {
                s2 = i;
                //cout << "s2" << s2 << endl;
                break;
            }       
        }
        
        
        for (int i = 1; i < A.size(); i++) {
            if (prefix_sum[i] > 2 * ones / 3) {
                s3 = i;
                //cout << "s3" << s3 << endl;
                break;
            }       
        }
        
        
        
        vector<int> part3 = vector<int>(A.begin()+s3, A.end());
        int len = part3.size();
        //cout << "len" << len << endl;
        vector<int> part1 = vector<int>(A.begin()+s1, A.begin()+s1+len);
        vector<int> part2 = vector<int>(A.begin()+s2, A.begin()+s2+len);
        
        if (s1 + len > s2 || s2 + len > s3) return {-1, -1};
        //cout << "flag1" << endl;
        for (int i = s1 + len; i < s2; i++) if (A[i]) return {-1, -1};
        for (int i = s2 + len; i < s3; i++) if (A[i]) return {-1, -1};
        //cout << "flag2" << endl;
        for (int i = 0; i < len; i++) {
            //cout << i << " " << part1[i] << " " << part2[i] << " " << part3[i] << endl;
            if (part1[i] != part2[i] || part1[i] != part3[i]) return {-1, -1};
            
        }
        
        return {s1+len-1, s2+len};
        
        
    }
};
