class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        vector<int> pos = {};
        for (int i = 0; i < A.size(); i++) if (A[i] == 1) pos.push_back(i);
        if (pos.size() % 3 != 0) return {-1, -1};          
        if (pos.size() == 0) return {0, 2};
        for (int i = 0, j = pos.size() / 3, k = pos.size() / 3 * 2; i < pos.size() / 3; i++, j++ ,k++) {
            if (i == 0) continue;
            int dif = pos[i] - pos[i - 1];
            if (dif != pos[j] - pos[j - 1] || dif != pos[k] - pos[k - 1]) return{-1, -1};
        }
        int x = pos[pos.size() / 3] - pos[pos.size() / 3 - 1];
        int y = pos[pos.size() / 3 * 2] - pos[pos.size() / 3 * 2 - 1];

        int z = A.size() - pos[pos.size() - 1];
        printf("%d %d %d\n\n", pos[0], pos[pos.size() / 3], pos[pos.size() / 3 * 2]);
        printf("%d %d %d\n", x, y, z);
        if (z <= x && z <= y) return { pos[pos.size() / 3 - 1] + z - 1, pos[pos.size() / 3 * 2 - 1] + z};
        else return {-1, -1};            
    }
};
