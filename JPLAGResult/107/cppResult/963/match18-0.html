class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int oneNum = 0;
        for (auto num : A) {
            if (num == 1) oneNum++;
        }
        if (oneNum == 0) return vector<int>{0, 2};
        if (oneNum % 3 != 0) return vector<int>{-1, -1};
        
        int count = oneNum / 3;
        int thirdNumStart = A.size() - 1;
        while (count > 0) {
            if (A[thirdNumStart] == 1) count--;
            thirdNumStart--;
        }
        thirdNumStart++;
        int secondNumEnd = thirdNumStart - 1;
        //cout << "secondEnd" << secondNumEnd << endl;
        int ii = A.size() - 1;
        int jj = secondNumEnd;
        bool oneFlag = false;

        while (ii >= thirdNumStart) {
            if (A[ii] == 1) {
                if (!oneFlag) {
                    oneFlag = true;
                    while (A[jj] != 1) {
                        jj--;
                        secondNumEnd--;
                    }
                    //cout << "jj" << jj << endl;
                } else {
                    if (A[ii] == A[jj]) {
                        ii--;
                        jj--;
                    } else {
                        return vector<int>{-1, -1};
                    }
                }
            } else {
                if (A[ii] == A[jj]) {
                    ii--;
                    jj--;
                } else {
                    return vector<int>{-1, -1};
                }
            }
        }

        //cout << "secondEnd" << secondNumEnd << endl;
        int kk = 0;
        ii = thirdNumStart;
        while (A[kk] == 0) kk++;
        while (ii < A.size()) {
            if (A[kk] == A[ii]) {
                kk++;
                ii++;
            } else {
                return vector<int>{-1, -1};
            }
        }
        return vector<int>{kk-1, secondNumEnd + 1};
    }
};
