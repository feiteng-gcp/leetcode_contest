class Solution 
{
public:
    vector<int> threeEqualParts(vector<int>& A) 
    {
        vector<int> amt1;
        int len = A.size();
        int aveAmt1;
        if(len < 3)
            return vector<int>{-1,-1};
        amt1.push_back(A[0]);
        for(int i = 1;i < len;i++)
            amt1.push_back(amt1[i-1] + A[i]);
        if(amt1[len-1] == 0)
            return vector<int>{0, len-1};
        if(amt1[len-1] % 3 != 0)
            return vector<int>{-1,-1};

        aveAmt1 = amt1[len-1] / 3;
        int p0, p1, p2, p3, p4, p5;
        p0 = p1 = p2 = p3 = p4 = p5 = -1;
        for(int i = 0;i < len;i++)
        {
            if(amt1[i] == 1 && p0 == -1)

                p0 = i;
            if(amt1[i] == aveAmt1 && p1 == -1)
                p1 = i;
            if(amt1[i] == aveAmt1+1 && p2 == -1)
                p2 = i;
            if(amt1[i] == aveAmt1*2 && p3 == -1)
                p3 = i;
            if(amt1[i] == aveAmt1*2+1 && p4 == -1)
                p4 = i;
            if(amt1[i] == amt1[len-1] && p5 == -1)
                p5 = i;
        }
        if(!(p1-p0 == p3-p2 && p3-p2 == p5-p4))
            return vector<int>{-1,-1};
        int partLen = p1 - p0 + 1;
        for(int i = 0;i < partLen;i++)
        {
            if(!(A[p0+i] == A[p2+i] && A[p2+i] == A[p4+i]))
                return vector<int>{-1,-1};
        }
        int zeroAmt = len - 1 - p5;
        if(zeroAmt > p4-p3-1 || zeroAmt > p2-p1-1)
            return vector<int>{-1,-1};
        return vector<int>{p1+zeroAmt, p3+zeroAmt+1};
    }
};
