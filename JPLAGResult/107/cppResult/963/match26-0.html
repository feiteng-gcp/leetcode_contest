class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        vector<int> idx;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] == 0)
                continue;
            idx.push_back(i);
        }
        if (idx.empty())
            return {0, 2};
        if (idx.size() % 3 != 0)
            return {-1, -1};
        int third = idx.size() / 3;
        for (int i = 1, j = third + 1; i < third; i++, j++) {
            if (idx[i] - idx[i - 1] != idx[j] - idx[j - 1])
                return {-1, -1};
        }
        // for (int i : idx)
        //     cout << i;
        // cout <<endl;
        // trailing 0's
        int trl = A.size() - 1 - idx.back();
        // cout << trl <<  ' ' << idx[third] - idx[third - 1] << idx[third * 2] - idx[third * 2 - 1] << endl;
        if (trl < idx[third] - idx[third - 1] && trl < idx[third * 2] - idx[third * 2 - 1])
            return {idx[third - 1] + trl, idx[third * 2 - 1] + 1 + trl};
        else
            return {-1, -1};
    }
};
