class Solution {
public:
vector<int> threeEqualParts(vector<int>& vec) {
    int n = vec.size();
    reverse(vec.begin(), vec.end());
    int total = accumulate(vec.begin(), vec.end(), 0);
    if (total % 3) {
        return {-1, -1};
    }
    
    if (total == 0) {
        return {0, 2};
    }

    total = total / 3;
    
    int i = 0;
    int cnt = 0;
    while (i < n && cnt != total) {
        if (vec[i] == 1) {
            cnt++;
        }
        i++;
    }
    i--;
    
    string myStr = "";
    for (int i = 0; i < vec.size(); i++) {
        if (vec[i] == 1) {
            myStr += "1";
        } else {
            myStr += "0";
        }
    }
    
    string pattern = myStr.substr(0, i + 1);
    int left = i + 1;
cout << pattern << endl;   
    int index = myStr.find(pattern, left);
    if (index == -1) {
        return {-1, -1};
    }
    
    int left2 = index + pattern.length();

    int index2 = myStr.find(pattern, left2);
    if (index2 == -1) {
        return {-1, -1};
    }
    
    return {n-1-index2, n-1-index+1};
}
};
