class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int cnt=0;
        vector<int>b;
        for(int i=0;i<A.size();i++){
            if(A[i]==1) cnt++;
        }
        if(cnt==0){
            b.push_back(0);b.push_back(A.size()-1);
            return b;
        }
        if(cnt%3!=0){
            b.push_back(-1);b.push_back(-1);
            return b;
        }
        int cc=0;
        int i0=0,i1=0,j1=0,i2=0,j2=0,i3=0;
        for(int i=0;i<A.size();i++){
            if(A[i]==1){
                i0=i;
                break;
            }
        }
        for(int i=0;i<A.size();i++){
            if(A[i]==1){
                cc++;
                if(cc==cnt/3)     i1=i;
                if(cc==cnt/3+1)   j1=i;
                if(cc==cnt*2/3)   i2=i;
                if(cc==cnt*2/3+1) j2=i;
                if(cc==cnt)       i3=i;
            }
        }
        //cout<<i0<<" "<<i1<<" "<<j1<<" "<<i2<<" "<<j2<<" "<<i3<<endl;
        int kk=0;int flag=1;
        if(i1-i0!=i2-j1||i1-i0!=i3-j2||j1-i2!=j2-i3)
            flag=0;
        while(kk<i1-i0&&kk<i2-j1&&kk<i3-j2){
            if(A[i0+kk]!=A[j1+kk]||A[i0+kk]!=A[j2+kk]||A[j1+kk]!=A[j2+kk]){
                flag=0;
                break;
            }
            kk++;
        }
        int z1=j1-i1-1,z2=j2-i2-1,z3=A.size()-i3-1;
        //cout<<z1<<" "<<z2<<" "<<z3<<endl;
        int ans1=0,ans2=0;
        if(z3>z2||z3>z1) flag=0;
        else{
            ans1=i1+z3;
            ans2=i2+z3+1;
        }
        if(flag==1){
            b.push_back(ans1);
            b.push_back(ans2);
            return b;
        }
        else{
             b.push_back(-1);b.push_back(-1);
            return b;
        }
        
        
        
    }
};
