class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int n = A.size();
        int lead0[n + 1];
        lead0[n] = 0;
        for (int k = n - 1;k >= 0;k --) {
            if (A[k] == 0) {
                lead0[k] = lead0[k + 1] + 1;
            } else {
                lead0[k] = 0;

            }
        }
        if (lead0[0] == n) {// All zero
            return {0, 2};
        }
        
        int first = lead0[0];
        for (int second = first + 1; second < n; second ++) {
            int j = second + lead0[second];
            int i = first;
            while (i < second && j < n && A[i] == A[j]) {
                i ++;
                j ++;
            }
            if (i == second && j < n) {
                i = i + lead0[i];
                int third = j;
                j = j + lead0[j];
                while (j < n && A[i] == A[j]) {
                   i ++;
                   j ++;
                }
                if (i == third && j == n) {
                    return {second - 1, third};
                }
            }
        }
        return {-1, -1};
    }
};
