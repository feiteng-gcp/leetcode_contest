class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int ones = 0;
        for (auto& num: A) {
            if (num) {
                ++ones;
            }
        }
        if (ones % 3 != 0 || A.size() < 3) {
            return {-1, -1};
        }
        if (ones == 0) {
            return {0, A.size() - 1};
        }
        int third = ones / 3;
        int j = A.size() - 1;
        int zeros = 0;
        ones = 0;
        while (A[j] == 0) {
            ++zeros;
            --j;
        }
        int p1 = j;
        while (ones < third) {
            if (A[j]) {
                ++ones;
            }
            --j;
        }
        int i = j;
        int z2 = 0;
        while (A[i] == 0) {
            ++z2;
            --i;
        }
        if (z2 < zeros) {
            return {-1, -1};
        }
        ones = 0;
        int p2 = i;
        while (ones < third) {
            if (A[i]) {
                ++ones;
            }
            --i;
        }
        int i2 = i;
        z2 = 0;
        while (A[i2] == 0) {
            ++z2;
            --i2;
        }
        if (z2 < zeros) {
            return {-1, -1};
        }
        int p3 = i2;
        ones = 0;
        for (int i = 0; ones < third; ++i) {
            if (A[p1 - i]) {
                ++ones;
            }
            if (A[p1 - i] != A[p2 - i] || A[p1 - i] != A[p3 - i]) {
                return {-1, -1};
            }
        }
        return {p3 + zeros, p2 + zeros + 1};
    }
};
