class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int n=A.size();
        int cnt=0;
        for (int i=0; i<n; i++) {
            if (!A[i]) cnt++;
        }
        if (cnt==n) return {0,2};
        int l=0;
        while (!A[l]) l++;
        // cout << l << endl;
        for (int i=l;i<=n-3;i++) {
            int j=check(A, l, i, i+1);
            // if (i==4) cout << j << endl;
            if (j==-1||j==n) continue;
            int k=check(A, l, i, j);
            if (k!=-1&&k==n) return {i,j};
        }
        return {-1,-1};
    }
private:
    int check(vector<int>& A, int l1, int l2, int r1) {
        int n=A.size();
        while (r1<n&&!A[r1]) r1++;
        if (r1==n) return -1;
        for (int i=l1; i<=l2; i++) {
            if (r1>=n) return -1;
            if (A[i]!=A[r1]) return -1;
            r1++;
        }
        return r1;
    }
};
