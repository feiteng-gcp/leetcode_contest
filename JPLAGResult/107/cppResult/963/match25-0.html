class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int n = A.size();
        
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (A[i] == 1) {
                cnt += 1;
            }
        }
        
        bool suc = true;
        if (cnt == 0) {
            vector<int> ret = {0, 2};
            return ret;
        } else if (cnt % 3 == 0) {
            cnt /= 3;
            int ncnt = 0;
            vector<int> idx_vct;
            for (int i = 0; i < n; i++) {
                if (A[i] == 1) {
                    ncnt += 1;
                    if (ncnt == 1) {
                        idx_vct.push_back(i);
                    }
                    ncnt %= cnt;
                }
            }
            //printf("%d %d %d\n", cnt, ncnt, idx_vct.size());
            
            int st_a, ed_a, st_b, ed_b, st_c, ed_c;
            assert(idx_vct.size() == 3);
            st_a = idx_vct[0];
            st_b = idx_vct[1];
            st_c = idx_vct[2];
            ed_a = st_b - 1;
            ed_b = st_c - 1;
            ed_c = n-1;
            
            printf("%d %d %d %d %d %d\n", st_a, ed_a, st_b, ed_b, st_c, ed_c);
            for (int i = 0; i <= ed_c - st_c; i++) {
                if (st_a + i > ed_a or st_b + i > ed_b) {
                    suc = false;
                } else if (A[st_a+i] != A[st_b+i] or A[st_b+i] != A[st_c+i]) {
                    suc = false;
                }
            }
            
            if (suc) {
                int diff = ed_c-st_c+1;
                vector<int> ret = {st_a+diff-1, st_b+diff};
                return ret;
            }
        }
        vector<int> ret = {-1, -1};
        return ret;
    }
};
