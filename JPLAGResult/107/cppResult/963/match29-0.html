class Solution {
public:
    int n = 0;
    vector<int> one;

    bool check(vector<int>& A, int p1, int p2, int p3, int l) {
      for (int i = 0; i < l; ++i) {
        if (A[p1+i] != A[p2+i] ||
            A[p1+i] != A[p3+i] ||
            A[p2+i] != A[p3+i])
          return false;
      }
      return true;
    }

    vector<int> threeEqualParts(vector<int>& A) {
      n = A.size();
      for (int i = 0; i < n; ++i)
        if (A[i]) one.push_back(i);

      vector<int> ret;

      if (!one.size()) {
        if (n >= 3) {
          ret.push_back(0);
          ret.push_back(2);
        } else {
          ret.push_back(-1);
          ret.push_back(-1);
        }
        return ret;
      }

      int len = (n - one.front()) / 3;

      for (int l = len; l; --l) {
        int b1 = one.front()+l-1;
        auto p2 = upper_bound(one.begin(), one.end(), b1);
        if (p2 == one.end()) continue;
        int b2 = (*p2)+l-1;
        if (b2 >= n) continue;
        auto p3 = upper_bound(one.begin(), one.end(), b2);
        if (p3 == one.end()) continue;
        int b3 = (*p3)+l-1;
        if (b3 != n-1) continue;

        bool flag = check(A, one.front(), *p2, *p3, l);
        if (!flag) continue;

        ret.push_back(b1);
        ret.push_back(b2+1);
        return ret;
      }

      ret.push_back(-1);
      ret.push_back(-1);
      return ret;
    }
};
