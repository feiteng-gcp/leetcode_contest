class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int c = 0;
        for (auto a: A) if (a == 1) c++;
        if (c % 3 != 0) return {-1, -1};
        if (c == 0) return {0, A.size() - 1};
        int _c = c / 3;
        int tailing_zero = 0, j;
        for (j = A.size() - 1; j >= 0 && A[j] == 0; --j) {
            tailing_zero++;
        }
        vector<int> pattern;
        for (; j >= 0 && _c > 0; --j) {
            pattern.push_back(A[j]);
            if (A[j] == 1) {
                _c--;
            }
        }
        int tz2 = 0;
        for (; j >= 0 && A[j] == 0; --j) {
            tz2++;
        }
        if (tz2 < tailing_zero) return {-1, -1};
        int ans2 = j + tailing_zero + 1;
        for (int i = 0; i < pattern.size(); ++i, --j) {
            if (j < 0 || pattern[i] != A[j]) return {-1, -1};
        }
        int tz3 = 0;
        for (; j >= 0 && A[j] == 0; --j) {
            tz3++;
        }
        if (tz3 < tailing_zero) return {-1, -1};
        int ans1 = j + tailing_zero;

        for (int i = 0; i < pattern.size(); ++i, --j) {
            if (j < 0 || pattern[i] != A[j]) return {-1, -1};
        }
        return {ans1, ans2};
    }
};
