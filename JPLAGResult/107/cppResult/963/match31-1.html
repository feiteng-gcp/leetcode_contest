class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        int n=A.size();
        vector<int> pos;
        for(int i=0;i<A.size();i++){
            if(A[i]==1)
                pos.push_back(i);
        }
        if(pos.size()==0){
            return {0,2};
        }
        if(pos.size()%3 != 0)
            return {-1,-1};
        int k = pos.size()/3; 
        int right_zero = n-pos.back()-1;
        if(pos[k]-pos[k-1]-1 < right_zero || pos[2*k]-pos[2*k-1]-1 <right_zero )
            return {-1,-1};
        for(int i=0; i<k-1;i++){
            if(pos[i+1]-pos[i] != pos[i+k+1]-pos[i+k] || pos[i+2*k+1]-pos[i+2*k] != pos[i+1]-pos[i])
                return {-1,-1};
        }
        return {pos[k-1]+right_zero, pos[2*k-1]+right_zero+1};
    }
};
