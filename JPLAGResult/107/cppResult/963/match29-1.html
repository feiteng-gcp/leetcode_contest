class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        vector<int> ans={-1,-1};
        int sum=0;
        for(int item:A) sum+=item;
        if(sum==0){
            ans={0,A.size()-1};
            return ans;
        }
        int start=0,l=A.size();
        while(A[start]==0) start++;
        // cout << start << endl;
        for(int k=1;k<=(l-start)/3;++k){
            int start2=k+start;
            // cout << start2 << endl;
            while(start2<l && A[start2]==0) start2++;
            if(start2>=l) continue;
            int start3=k+start2;
            // cout << start3 << endl;
            while(start3<l && A[start3]==0) start3++;
            if((start3+k)==l){
                // cout << start2 << " " << start3 << endl;
                if(Compare(A,start,start2,start3,k)){
                    ans={start+k-1,start2+k};
                    return ans;
                }
            }
        }
        return ans;
    }

    bool Compare(vector<int>& A, int a, int c, int d, int k){
        for(int iter=0;iter<k;++iter){
            if(A[a+iter]!=A[c+iter] || A[d+iter]!=A[c+iter]) return false;
        }
        return true;
    }
};
