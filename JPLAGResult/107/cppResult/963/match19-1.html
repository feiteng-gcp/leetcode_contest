class Solution {
public:
bool canMathch(vector<int> & nums1, vector<int> & nums2, int pos) {
    int i = pos;
    int j = 0;

    for(; i < nums1.size() && j < nums2.size(); i++,j++) {
        if(nums2[j]!= nums1[i])
            return false;
    }
    
    return j == nums2.size();
}
vector<int> threeEqualParts(vector<int>& A) {
    int one = 0;
    vector<int> ones(A.size(),0);
    for(int i = 0; i < A.size(); i++) {
        if(A[i] == 1)
            one++;
        ones[i] = one;
    }
    
    if(!one)
        return {0,2};
    
    if(one%3)
        return {-1,-1};
    
    int k = one/3;
    int temp = 0;
    vector<int> target;
    for(int i = A.size()-1; i >=0; i--) {
        target.push_back(A[i]);
        if(A[i] == 1)
            temp++;
        
        if(temp == k)
            break;
    }
    
    reverse(target.begin(), target.end());
    int a = -1;
    int b = -1;
    
    for(int i = 0; i < A.size(); i++) {
        if(A[i] == 1) {
            if(canMathch(A, target, i)) {
                a = i + target.size()-1;
                break;
            } else {
                return {-1,-1};
            }
        }
    }
    
    for(int i = 0; i < A.size(); i++) {
        if(ones[i] == k+1) {
            if(canMathch(A, target, i)) {
                b = i + target.size();
                break;

            } else {
                return {-1,-1};
            }
        }
    }
    
    return {a,b};
}
};
