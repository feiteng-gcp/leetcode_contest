class Solution {
    vector<int> A;
    vector<int> cnt1;
public:
    uint64_t hash(int b, int e) {
        return cnt1[b]-cnt1[b];
    }
    bool cmp(int i, int j, int L) {
        if (hash(i,i+L)!=hash(j, j+L)) return false;
        for (int k=0; k<L; k++) {
            if (A[i+k] != A[j+k]) return false;
        }
        return true;
    }
    vector<int> threeEqualParts(vector<int>& A) {
        //printf("===\n");
        int i=0;
        this->A = A;

        vector<int> next1index(A.size()+1);
        next1index[A.size()] = -1;
        cnt1.resize(A.size()+1);
        for (int k=A.size()-1;k>=0; k--) {
            next1index[k] = (A[k]==1) ? k : next1index[k+1];
            cnt1[k] = cnt1[k+1] + A[k];
        }
            
        while (A[i]==0 && i<A.size()-3) {
            i++;
        }
        if (i==A.size()-3) {
            if (A[i]==A[i+1] && A[i]==A[i+2]) return {0,2}; else return {-1,-1};
        }
        for (int L=1; L<=(A.size()-i)/3; L++) {
            if (cmp(i,A.size()-L, L)) {
                int n = next1index[i+L];
                //printf("i=%d L=%d  n=%d  %d >? %d\n",i,L,n, n+L, A.size()-L);
                if (n<0 || n+L > A.size()-L) {
                    return {-1,-1};
                }
                int c1 = cnt1[n+L]-cnt1[A.size()-L];
                //printf("c1=%d\n",c1);
                if (c1 == 0 && cmp(i,n,L)) {
                    return {i+L-1, n+L};
                }
            }
        }
        return {-1,-1};
    }
};
