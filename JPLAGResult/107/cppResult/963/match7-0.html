class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        vector<int> ones;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] == 1) {
                ones.push_back(i);
            }
        }
        if (ones.size() == 0) {
            return {0, 2};
        }
        if (ones.size() % 3 != 0) {
            return {-1, -1};
        }
        int k = ones.size() / 3;
        for (int i = 1; i < k; i++) {
            if (ones[i] - ones[i - 1] != ones[i + k] - ones[i + k - 1] || 
                ones[i] - ones[i - 1] != ones[i + 2 * k] - ones[i + 2 * k - 1]) {
                return {-1, -1};
            }
        }
        int tail = A.size() - 1 - ones.back();
        int i = ones[k - 1] + tail;
        if (i >= ones[k]) {
            return {-1, -1};
        }
        int j = ones[2 * k - 1] + tail + 1;
        if (j > ones[2 * k]) {
            return {-1, -1};
        }
        return {i, j};
    }
};
