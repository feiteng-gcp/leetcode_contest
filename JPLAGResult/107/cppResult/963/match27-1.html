#include <vector>
#include <iostream>
#include <map>
using namespace std;

class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
        if (A.size() < 3)
            return {-1, -1};

        int ones = 0;
        int back_zeroes = 0;
        for (auto n: A) {
            if (n) {
                ones ++;
                back_zeroes = 0;
            } else
                back_zeroes ++;
        }

        if (ones % 3)
            return {-1, -1};

        if (ones == 0)
            return {0,2};

        std::vector<int> first, second, third;

        auto process = [&A, ones, back_zeroes](std::vector<int>& trg, int startidx) -> int {
            int added0 = 0;
            int added1 = 0;
            int i = 0;
            for (i = startidx; i < A.size(); i++) {
                if (A[i] == 0 && !added1) continue;

                if (A[i] == 1) {
                    added1++;
                    added0 = 0;
                } else {
                    added0++;
                }

                trg.push_back(A[i]);

                if (added1 == ones / 3 && added0 == back_zeroes)
                    break;
            }

            return i+1;
        };


        int addedZeroes, addedOnes = 0;


        int a = process(first, 0);
        int b = process(second, a);
        int c = process(third, b);

        if (first == second && first == third && c == A.size())
            return {a-1, b};
        else
            return {-1, -1};
    }
};
