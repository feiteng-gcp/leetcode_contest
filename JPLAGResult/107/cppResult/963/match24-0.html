class Solution {
public:
    vector<int> threeEqualParts(vector<int>& a) {
        string s;
        int cnt_1 = 0;
        vector<int> pos;
        for (int i = 0; i < a.size(); i++) {
            if (a[i]) {
                cnt_1++;
                pos.push_back(i);
            }
        }
        int cnt_0 = a.size() - cnt_1;
        if (cnt_0 == a.size()) {
            return {0, 2};
        }
        if (cnt_1 % 3 != 0) {
            return {-1, -1};
        }
        int target = cnt_1 / 3;
        vector<int> gap;
        for (int i = 1; i < target; i++) {
            gap.push_back(pos[i] - pos[i-1]);
        }
        int j = 0;
        for (int i = target + 1; i < 2 * target; i++) {
            int temp_gap = pos[i] - pos[i-1];
            if (temp_gap != gap[j++]) {
                return {-1, -1};
            }
        }
        j = 0;
        for (int i = target * 2 + 1; i < 3 * target; i++) {
            int temp_gap = pos[i] - pos[i-1];
            if (temp_gap != gap[j++]) {
                return {-1, -1};
            }
        }
        int end_0_cnt = a.size() - pos.back() - 1;
        if (pos[target] - pos[target-1]  - 1>= end_0_cnt && pos[2*target] - pos[2*target-1]  - 1>= end_0_cnt) {
            return {pos[target-1] + end_0_cnt, pos[target*2 - 1] + end_0_cnt + 1};
        }
        return {-1,-1};
    }
};
