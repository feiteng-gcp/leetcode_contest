class Solution {
public:
    vector<int> threeEqualParts(vector<int>& A) {
     //   reverse(A.begin(), A.end());
      //  cout << endl ;
        int total = 0;
        int totalzero = 0;
        int n = A.size();
        for (int k = 0; k < A.size(); k++) {
            if (A[k] == 1)
                total++;
            else

                totalzero++;
        }
        if (total % 3 != 0)
            return vector<int>{-1, -1};
        if (total == 0 || total == A.size())
            return vector<int>{n/3-1, n/3*2};
        totalzero = totalzero/3; //  upper limit for trailing zeros
        total = total / 3;
        int j = n - 1;
        while (j >= 0 && A[j] == 0) j--;
        

        int i = 0;
        int firstone = 0;
        while (firstone < total) {
            if (A[i] == 1)
                firstone++;
            i++;
        }
        i = i + (n - j - 1) - 1; // [0, i], the first part
        cout <<" first part: "<< 0 <<", " << i << endl;
        // now check the second part
        int t = i + 1;
        int u = 0;
        while (t < A.size() && A[t] == 0) t++;
        while (u <= i && A[u] == 0) u++;
        int s = u;
        for (; u <=i; u++) {
            if (A[u] != A[t])
                return vector<int>{-1, -1};
            t++;
        }
        cout <<"second part:" << i+1 <<", " << t-1 << endl;
        
        // now check third part
        j = t;
        while (j < A.size() && A[j] == 0) j++;
        for (; s <= i; s++) {
            if (j == A.size() || A[s] != A[j] )
                return vector<int>{-1, -1};
            j++;
        }
        if (j != A.size())
            return vector<int>{-1, -1};
        vector<int> ans = {i,t};
        return ans;
        
        
    }
};
