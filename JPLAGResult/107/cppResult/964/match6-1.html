class Solution {
public:
    int minMalwareSpread(vector<vector<int>> &graph, vector<int> &initial) {
        sort(initial.begin(), initial.end());
        int re = initial[0];
        int re_val = INT_MAX;
        for(int i=0; i<initial.size(); ++i){

            int tmp = del(graph, initial, i);
            if(tmp<re_val){
                re = initial[i];
                re_val = tmp;
            }
        }
        return re;
    }
    
    int del(vector<vector<int>> &graph, vector<int> &initial, int node){
        int cnt = 0;
        vector<int> visited(graph.size(),0);
        queue<int> q;
        for(int i=0; i<initial.size(); ++i)

            if(i!=node) {
                q.push(initial[i]);
                visited[initial[i]] = 1;
                cnt++;
            }
        while(!q.empty()){
            int f = q.front();
            q.pop();
            for(int e=0; e<graph[f].size(); ++e){
                if(graph[f][e]==1 && e!=initial[node] && !visited[e]){
                    q.push(e);
                    visited[e] = 1;
                    cnt++;
                }
            }
        }
        return cnt;
    }
};
