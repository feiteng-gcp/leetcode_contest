class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int count = 0, minCount = INT_MAX, res = -1;
        for(int i = 0; i < initial.size(); ++i) {
            int count = simulation(graph, initial, i);
            // cout << count << endl;
            if(count <= minCount) {
                if(count == minCount && initial[i] < res) res = initial[i];
                else if(count < minCount) {
                    minCount = count;
                    res = initial[i];
                }
            }
        }
        return res;
    }
private:
    int simulation(vector<vector<int>>& graph, vector<int>& initial, int removed) {

        int count = 0;
        queue<int> que;
        vector<bool> visited(graph.size(), 0);
        for(int i = 0; i < initial.size(); ++i) {
            if(i == removed) continue;
            // cout << initial[i] << " ";
            que.push(initial[i]);
            visited[initial[i]] = true;
            count += 1;
        }
        // cout << endl;
        int removeNode = initial[removed];
        // cout << removeNode << " is removed" << endl;

        while(que.size()) {
            int tmp = que.front();
            // cout << "tmp is " << tmp << endl;
            que.pop();
            for(int i = 0; i < graph[tmp].size(); ++i) {
                if(i == tmp || i == removeNode) continue;
                if(graph[tmp][i] && visited[i] == false) {
                    que.push(i);
                    visited[i] = true;
                    count += 1;
                }
            }
        }
        return count;
    }
};
