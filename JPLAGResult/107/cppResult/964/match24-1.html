class Solution {
public:
    int n;
    int rm(int id, vector<vector<int>>& g, vector<int>& ini) {
        vector<int> vis(n, 0);
        queue<int> que;
        int cnt = 0;
        for (auto v : ini) {
            if (v == id) continue;
            que.push(v);
            vis[v] = 1;

            cnt++;
        }
        while (que.size()) {
            int cur = que.front();
            que.pop();
            for (int v = 0; v < n; v++) {
                if (g[cur][v] == 0 || v == id) continue;
                if (!vis[v]) {
                    vis[v] = 1;
                    cnt++;
                    que.push(v);
                }
            }
        }
        return cnt;
    }
    int minMalwareSpread(vector<vector<int>>& g, vector<int>& ini) {
        n = g.size();   

        int best_res = INT_MAX;
        int best_id = 0;
        for (auto v : ini) {
            int tmp = rm(v, g, ini);
            if (tmp < best_res) {
                best_id = v;
                best_res = tmp;
            } else if (tmp == best_res) {
                if (v < best_id) best_id = v;
            }
        }
        return best_id;
    }
};
