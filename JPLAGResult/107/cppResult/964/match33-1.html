class Solution {
public:
    int bfs(unordered_map<int, vector<int>>& graph, unordered_set<int>& virus, int erased) {
        unordered_set<int> com = virus;
        queue<int> q;
        for (auto v: virus) q.push(v);
        while (q.size()) {
            int size = q.size();
            while (size--) {
                int cur = q.front(); q.pop();
                for (auto next: graph[cur]) {
                    if (next == erased) continue;
                    if (com.count(next)) continue;
                    com.insert(next);
                    q.push(next);
                } 
            }
        }
        return com.size();
    }
    

    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        unordered_map<int, vector<int>> g;
        for (int i = 0; i < graph.size(); ++i) {
            for (int j = 0; j < graph[0].size(); ++j) {
                if (graph[i][j]) g[i].push_back(j);
            }
        }
        unordered_set<int> virus(initial.begin(), initial.end());
        int ans = -1, value = -1;

        for (auto i: initial) {
            virus.erase(i);
            int tmp = bfs(g, virus, i);
            if (ans == -1 || tmp < value || tmp == value && i < ans) {
                ans = i;
                value = tmp;
            }
            virus.insert(i);
        }
        return ans;
    }
};
