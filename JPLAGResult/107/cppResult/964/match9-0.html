class Solution {
    int n;
    vector<int> infected;
    void flood(vector<vector<int>> const &g, int cur, int rm) {
        if (cur == rm) {
            return;
        }
        if (infected[cur]) {
            return;
        }
        infected[cur] = 1;
        for (int nxt = 0; nxt<n; nxt++) {
            if (nxt==cur || !g[cur][nxt]) {
                continue;
            }
            flood(g, nxt, rm);
        }
    }
    int sum() {
        int rt = 0;
        for (int x : infected) {
            if (x) {
                rt += 1;
            }
        }
        return rt;
    }
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        n = graph.size();
        int mi = n+1;
        int ans = -1;
        for (int rm : initial) {
            infected.assign(n, 0);
            for (int x : initial) {
                if (x==rm) {
                    continue;
                }
                flood(graph, x, rm);
            }
            int res = sum();
            //cout << "try: " << rm << " -> " << res << endl;
            if (res<mi) {
                mi = res;
                ans = rm;
            } else if (res == mi) {
                ans = min(ans, rm);
            }
        }
        return ans;
    }
};
