template <typename T>
void print_seq(T&& v) {
    cout << "[ ";
    for(auto& x : v) {
        cout << x << " ";
    }
    cout << "]\n";
}

class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int M = INT_MAX;
        int idx = INT_MAX;
        
        //
        //cout << nums(graph, initial) << endl;
        //
        
        
        vector<vector<int>> graph_copy;
        vector<int> initial_copy;
        
        for (int i = 0; i < initial.size(); ++i) {
            int u = initial[i];
            graph_copy = graph;
            for (int i = 0; i < graph_copy.size(); ++i) {
                graph_copy[i][u] = 0;
                graph_copy[u][i] = 0;
            }
            initial_copy = {};
            for (int j = 0; j < initial.size(); ++j) {
                if (j != i) {
                    initial_copy.push_back(initial[j]);
                }
            }
            
            /////////////////
            //print_seq(initial_copy);
            //for(auto& v: graph_copy) {
            //    print_seq(v);
            //}
            /////////////////
            
            int MM = nums(graph_copy, initial_copy);
            
            //cout << u << ": " << MM << endl;
            
            if (MM < M) {
                M = MM;
                idx = u;
            }
            else if (MM == M) {
                idx = min(idx, u);
            }
        }
        
        return idx;
    }
private:
    int nums(vector<vector<int>>& graph, vector<int>& initial) {
        unordered_map<int, bool> visited;
        int count = 0;
        for (int& u : initial) {
            dfs(graph, u, visited, count);
        }
        return count;
    }
    
    void dfs(vector<vector<int>>& graph, int u, unordered_map<int, bool>& visited, int& count) {
        if (!visited[u]) {
            visited[u] = true;
            ++count;
            for (int v = 0; v < graph[u].size(); ++v) {
                if (!visited[v] && v != u && graph[u][v] == 1) {
                    dfs(graph, v, visited, count);
                }
            }
        }
    }
};
