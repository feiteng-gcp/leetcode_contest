class Solution {
public:
    void remove_n(vector<vector<int>>& g, int n) {

        for(int r=0; r<g.size(); ++r) {
            g[r][n] = 0;
        }
        for(int c=0; c<g[0].size(); ++c) {
            g[n][c] = 0;
        }

    }
    void dfs(vector<vector<int>>& g, unordered_set<int>& visited, int n) {
        if(visited.count(n)) return;
        visited.insert(n);
        for(int i=0; i<g[n].size(); ++i) {
            if(i != n && g[n][i] == 1) {
                dfs(g, visited, i);
            }
        }
    }
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        if(graph.empty() || graph[0].empty()) return 0;
        
        int virus = graph.size();
        int best_i = initial[0];
        for(auto init: initial) {
            vector<vector<int>> g = graph;
            remove_n(g, init);
            /*
            for(auto r: g) {
                for(auto n: r) {
                    cout << n << ' ';
                }
                cout << endl;
            }
            */
            unordered_set<int> visited;
            for(auto i: initial) {
                if(i != init) {
                    dfs(g, visited, i);
                }
            }
            if(visited.size() < virus) {
                best_i = init;
                virus = visited.size();
            }
            else if(visited.size() == virus) {
                best_i = min(init, best_i);
            }
        }
        
        
        return best_i;
    }
};
