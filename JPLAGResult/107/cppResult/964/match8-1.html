class Solution {
public:
    vector<vector<int> > _graph;
    int num(int x,vector<int> ini){
        int cnt[305];
        int vis[305];
        fill(vis,vis+305,0);
        fill(cnt,cnt+305,0);
        queue<int> q;
        vis[x]=1;
        for(auto it:ini){
            if(it==x)
                continue;
            vis[it]=1;
            for(int j=0;j<_graph.size();j++){
                if(_graph[it][j]){
                    cnt[j]++;
                }
            }
        }
        for(int i=0;i<_graph.size();i++){
            if(!vis[i]&&cnt[i]>=1){
                q.push(i);
                vis[i]=1;
            }

        }
        while(!q.empty()){
            int x=q.front();
            q.pop();
            for(int i=0;i<_graph.size();i++){
                if(_graph[x][i]){
                    cnt[i]++;
                    if(cnt[i]>=1&&vis[i]==0){
                        q.push(i);
                        vis[i]=1;
                    }
                }
            }
        }
        int tot=0;
        for(int i=0;i<_graph.size();i++)
            tot+=vis[i];
        return tot;
    }
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        sort(initial.begin(),initial.end());
        _graph=graph;

        int ans=1e9,ansnum=0;
        for(int i=0;i<initial.size();i++){
            int x=num(initial[i],initial);
            if(x<ans){
                ans=x;
                ansnum=initial[i];
            }
        }
        return ansnum;
    }
};
