class Solution {
public:
    int n;
    vector<int> visited;
    int dfs(int x, vector<vector<int>>& graph, int t){
        visited[x] = 1;
        int ret = 1;
        for(int i=0;i<n;i++){
            if(i==t)
                continue;
            if(graph[x][i]==1 && visited[i] == -1){
                ret += dfs(i, graph, t);
            }
        }
        //cout<<x<<":"<<ret<<' ';
        return ret;
    }
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        n = graph.size();
        sort(initial.begin(), initial.end());
        
        int idx = -1;
        int m = n;
        
        for(int t=0;t<initial.size();t++){
            visited = vector<int>(n, -1);
            int cnt = 0;
            for(int i=0;i<initial.size();i++){
                if(i==t)
                    continue;
                if(visited[initial[i]]==-1){                    
                    cnt += dfs(initial[i], graph, initial[t]);                    
                }                
            }
            //cout<<cnt<<endl;
            if(m > cnt){
                idx = t;
                m = cnt;
            }
        }
        
        return initial[idx];
    }
};
