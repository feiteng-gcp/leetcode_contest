class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n = graph.size();
        int minRet = n;
        int minRemoved = 0;
        sort(initial.begin(), initial.end());
        for (int i = 0; i < initial.size(); i ++) {
            int removed = initial[i];
            queue<int> q;
            bool visited[n];
            memset(visited, false, sizeof(visited));
            for (const auto init : initial) {
                if (init == removed) continue;
                q.push(init);
                visited[init] = true;
            }
            int curRet = 0;
            while (!q.empty()) {
                int cur = q.front();
                q.pop();

                curRet ++;
                for (int next = 0;next < n; next ++) {
                    if (next == removed) continue;
                    if (visited[next]) continue;
                    if (graph[cur][next] == 0) continue;
                    q.push(next);
                    visited[next] = true;
                }
            }
            if (curRet < minRet) {
                minRet = curRet;
                minRemoved = removed;
            }
        }
        return minRemoved;
    }
};
