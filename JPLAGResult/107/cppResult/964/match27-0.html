class Solution {
    vector<int> bfs(vector<vector<int>>&graph, int x, int b)
    {
        queue<int>q;
        int n = graph.size();
        vector<int>visit(n, 0);
        visit[x] = 1;
        q.push(x);
    
        while(!q.empty())
        {
            int curr = q.front();
            q.pop();
            
            for(int i = 0; i < n; i++)
            {
                if(i == b || graph[curr][i] == 0 || visit[i])continue;
                visit[i] = true;
                q.push(i);

            }
        }
        return visit;
    }
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int ans = INT_MAX;
        int ind;
        for(int i = 0; i < initial.size(); i++)
        {
            vector<int>arr(graph.size(), 0);
            for(int j = 0; j < initial.size(); j++)
            {
                if(i == j)continue;
                vector<int> temp= bfs(graph, initial[j], initial[i]);
                for(int k = 0; k < temp.size(); k++)
                {
                    arr[k] |= temp[k];
                }
            }
            int c = 0;
            for(int j = 0; j < graph.size(); j++)
                c += arr[j];
            if(ans > c)
            {
                ans = c;
                ind = initial[i];
            }
            else if(ans == c && ind > initial[i])
                ind = initial[i];
        }
        return ind;
    }
};
