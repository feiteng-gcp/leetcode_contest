class Solution {
public:
    int ms(vector<vector<int>>& graph, vector<int>& initial, int sp) {
        int result = 0;
        vector<int> affected;
        for (int i = 0; i < graph.size(); i++)
            affected.push_back(0);
        
        vector<int> newaffected;
        
        int newaffectedindex = 0;
        for (int i = 0; i < initial.size(); i++) {
            if (initial[i] == sp)
                continue;
            affected[initial[i]] = 1;
            for (int j = 0; j < graph[initial[i]].size(); j++) {
                if (graph[initial[i]][j] == 1 && j != sp) {
                    if (affected[j] == 0) {
                        newaffected.push_back(j);
                        affected[j] = 1;
                    }
                }
            }
            
        }
        while (newaffectedindex < newaffected.size()) {
            for (int j = 0; j < graph[newaffected[newaffectedindex]].size(); j++) {
                if (graph[newaffected[newaffectedindex]][j] && j != sp) {
                    if (affected[j] == 0) {
                        newaffected.push_back(j);
                        affected[j] = 1;
                    }
                }
            }
            newaffectedindex++;
        }
        
        for (int i = 0; i < graph.size(); i++) {
            if (affected[i] == 1)
                result++;
        }
        return result;
    }
    
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int result = 0;
        int maxnode = -1;
        int wholenode = ms(graph, initial, -1);
        cout << wholenode << endl;
        for (int i = 0; i < initial.size(); i++) {
            int tempnode = ms(graph, initial, initial[i]);
            cout << tempnode << " ";
            if (wholenode - tempnode > maxnode) {
                result = initial[i];
                maxnode = wholenode - tempnode;
            }
            else if (wholenode - tempnode == maxnode && initial[i] < result)
                result = initial[i];
                
        }
        return result;
    }
};
