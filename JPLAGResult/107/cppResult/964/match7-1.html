class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int min_aff = graph.size()+1;
        int res;
        sort(initial.begin(), initial.end());
        int n = graph.size();
        vector<vector<int>> graph_list;
        for(int i=0;i<n;i++){
            vector<int> cur;
            for(int j=0;j<n;j++){
                if(i!=j && graph[i][j]==1)
                    cur.push_back(j);
            }
            graph_list.push_back(cur);
        }

        for(int i=0;i<initial.size();i++){
            int aff = calcaff(graph_list, initial, initial[i]);
            // cout<<initial[i]<<" "<<aff<<endl;
            if(aff<min_aff){
                min_aff = aff;
                res = initial[i];
            }
        }
        return res;
    }
    
    int calcaff(vector<vector<int>>& graph, vector<int>& initial, int removed){
        int n = graph.size();
        vector<int> affected(n, 0);
        queue<int> q;
        for(int i=0;i<initial.size();i++)
            if(initial[i]!=removed && affected[initial[i]]==0){
                q.push(initial[i]);
                while(!q.empty()){
                    int cur = q.front();
                    q.pop();
                    if(affected[cur]==1)
                        continue;
                    affected[cur] = 1;
                    for(int j=0;j<graph[cur].size();j++)
                        if(graph[cur][j]!=removed && affected[graph[cur][j]]==0)
                            q.push(graph[cur][j]);
                }
            }
        int res = 0;
        for(int i=0;i<n;i++)
            if(affected[i]==1)
                res++;
        return res;
    }
};
