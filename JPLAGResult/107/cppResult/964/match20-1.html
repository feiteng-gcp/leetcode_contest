class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        n = graph.size();
        int ans = n+1;
        int ansnum = n+1;
        int cur;
        for (int i = 0; i < initial.size(); i++) {

            cur = bfs(graph, initial[i], initial);
            if (ans == n+1 || cur < ansnum || cur == ansnum && initial[i] < ans  ) {
                ans = initial[i];
                ansnum = cur;
            }
        }
        return ans;
    }
    
    
    int bfs(vector<vector<int>>& graph, int remove, vector<int>& initial) {
        queue<int> q;
        vector<int> visited(n, false);
        int infect = 0;
        visited[remove] = true;
      //  cout << remove << endl;
        for (int i = 0; i < initial.size(); i++) 
            if (initial[i] != remove) {
                infect++;
                q.push(initial[i]);
                visited[initial[i]] = true;
            }
        int cur;
        while (!q.empty()) {

            cur = q.front();
            q.pop();
            for (int i = 0; i < n; i++)
                if (!visited[i] && i!=cur && graph[cur][i] == 1) {
                    visited[i] = true;
                    q.push(i);
                    infect++;
                }
        }
       // cout <<" remove:" << remove << " infect:" << infect << endl;
        return infect;
    }
    int n;
};
