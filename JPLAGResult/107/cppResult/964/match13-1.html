class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n = graph.size();
        vector<int> infected(n, 0);
        for (int m : initial)
            infected[m] = 1;
        int res = 400, minNum = n;
        for (int m : initial)
        {
            queue<int> q;
            vector<int> inf = infected;
            for (int p : initial)
                if (p != m) q.push(p);
            int c = q.size();

            while (!q.empty())
            {
                int u = q.front();
                q.pop();
                for (int i = 0; i < n; i++)
                {
                    if (i == m) continue;
                    if (!graph[u][i]) continue;
                    if (inf[i]) continue;
                    inf[i] = 1;
                    q.push(i);
                    c++;
                }
            }
            if (c < minNum)
            {
                minNum = c;
                res = m;
            }
            else if (c == minNum) res = min(res, m);
        }
        return res;
    }
};
