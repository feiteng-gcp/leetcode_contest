const ll MOD=1000000007LL;

const ll LINF=1LL<<60;

class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& G, vector<int>& ini) {
        int V=G.size();
        int N=ini.size();
        sort(all(ini));
        int ans=-1;
        int ansx=-1;
        REP(hoge,N){//except x
            int x=ini[hoge];
            queue<int> que;
            vector<int> vis(V);
            REP(i,N){
                if(ini[i]==x){
                    continue;
                }
                que.push(ini[i]);
                vis[ini[i]]=true;
            }
            while(que.size()){
                int t=que.front();que.pop();
                REP(i,V){
                    if(i!=x&&G[t][i]&&!vis[i]){
                        vis[i]=true;
                        que.push(i);
                    }
                }
            }
            int cnt=0;
            REP(i,V){
                if(!vis[i]){

                    cnt++;
                }
            }
            if(cnt>ans){
                ans=cnt;
                ansx=x;
            }
        }
        return ansx;
    }
};
