const int INF = 1000000000;


class Solution {
public:
  int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
    int n = graph.size();
    int m = initial.size();
    int ans = INF;
    int ans_x;
    for (int skip : initial) {
      int cur = connect(graph, initial, skip);
      if (cur < ans) {
        ans = cur;
        ans_x = skip;
      } else if (cur == ans && ans_x > skip) {
        ans_x = skip;
      }
    }
    return ans_x;
  }
private:
  int connect(const vector<vector<int>>& g, const vector<int>& initial, int skip) {
    int n = g.size();
    vector<bool> good;
    good.resize(n);
    for (int i = 0; i < n; ++ i) {
      good[i] = true;
    }

    queue<int> q;
    for (int x : initial) {
      if (x == skip) continue;
      q.push(x);
      good[x] = false;
    }


    while (!q.empty()) {
      int x = q.front(); q.pop();
      for (int i = 0; i < n; ++ i) {
        if (i == skip) continue;
        if (good[i] && g[x][i]) {
          good[i] = false;
          q.push(i);
        }
      }
    }

    int ret = 0;
    for (int i = 0; i < n; ++ i) {
      if (!good[i]) ++ ret;
    }
    return ret;
  }

};
