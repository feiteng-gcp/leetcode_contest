class Solution {
public:
    int dfs(vector<vector<int>>& graph, vector<int>& visited,int idx){
        if(visited[idx]) return 0;

        visited[idx]=1;
        int num=1;
        for(int i=0;i<graph.size();i++){
            if(!visited[i]&&graph[idx][i])
                num+=dfs(graph,visited,i);
        }
        return num;
    }
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        sort(initial.begin(),initial.end());
        int ans=initial[0],num=INT_MAX;
        for(int i:initial){
            int tmp=0;
            vector<int> visited(graph.size(),0);
            vector<vector<int>> tmpv=graph;
            for(int j=0;j<graph.size();j++) tmpv[j][i]=tmpv[i][j]=0;
            for(int j:initial){

                if(i!=j) tmp+=dfs(tmpv,visited,j);
            }
            if(tmp<num){
                num=tmp;
                ans=i;
            }
        }
        return ans;
    }
};
