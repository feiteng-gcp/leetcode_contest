class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& g, vector<int>& f) {
        int n = g.size();
        int opt = n + 1;
        int ans = -1;
        for (auto& b : f) {
            vector<bool> s(n, false);
            queue<int> q;
            for (auto& x : f) {
                if (x != b) {
                    s[x] = true;
                    q.push(x);
                }
            }
            int res = 0;
            while (q.size()) {
                int x = q.front();
                q.pop();
                ++res;
                for (int i = 0; i < n; ++i) {
                    if (i != b && !s[i] && g[x][i]) {
                        s[i] = true;
                        q.push(i);
                    }
                }
            }
            if (res < opt || res == opt && b < ans) {
                ans = b;
                opt = res;
            }
        }
        return ans;
    }
};
