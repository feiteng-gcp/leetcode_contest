class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        sort(initial.begin(), initial.end());
        
        int n = graph.size();
        int min_cnt = 10000000;
        int ans = -1;
        
        for (const int & p: initial) {
            
            vector<bool> visited(n, false);
            queue<int> que;
            for (const int & q: initial) {
                if (p != q) {
                    visited[q] = true;
                    que.push(q);
                }
            }
            visited[p] = true;
            
            int affected_cnt = 0;
            while (!que.empty()) {
                int x = que.front();
                que.pop();
                for (int i = 0; i < n; i++) {
                    if (visited[i]) continue;
                    if (graph[x][i]) {
                        visited[i] = true;

                        affected_cnt++;
                        que.push(i);
                    }
                }
            }
            
            // cout << affected_cnt << " " << p << endl;
            
            if (affected_cnt < min_cnt) {
                min_cnt = affected_cnt;
                ans = p;
            }
        }
        return ans;
    }
};
