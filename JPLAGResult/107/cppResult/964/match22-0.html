class Solution {
    int bfs(const vector<vector<int>>& g, const vector<int>& a, int d) {
        int n = g.size();
        vector<bool> f(n);
        deque<int> que;
        for (int x : a) if (x != d) que.push_back(x), f[x] = true;
        while (!que.empty()) {
            int x = que.front();
            que.pop_front();
            for (int y = 0; y < n; ++y) if ((y != d) && g[x][y] && !f[y]) {
                f[y] = true;
                que.push_back(y);
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) if ((i != d) && f[i]) ++ret;
        return ret;
    }
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n = graph.size(), ret = initial[0], val = 1000010000;
        sort(initial.begin(), initial.end());
        for (int i = 0; i < initial.size(); ++i) {
            int x = bfs(graph, initial, initial[i]);
            // cout << i << " " << x << endl;
            if (x < val) ret = initial[i], val = x;
        }
        return ret;
    }
};
