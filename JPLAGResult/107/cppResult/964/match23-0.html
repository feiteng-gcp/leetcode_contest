class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        sort(initial.begin(),initial.end());
        int maxNum = INT_MAX;
        int res = graph.size()-1;
        
        for(int leftOut=0;leftOut<initial.size(); ++leftOut){
            unordered_set<int> visited;
            for(int i = 0; i<initial.size();++i){
                vector<vector<int>> tempG(graph);
                for(int row = 0; row <graph.size(); ++row){
                    tempG[row][initial[leftOut]] = 0;
                }
                for(int col = 0; col <graph.size(); ++col){
                    tempG[initial[leftOut]][col] = 0;
                }
                
                if(i!=leftOut)sink(initial[i],tempG,visited);
            }
            //cout<<initial[leftOut]<<":" <<visited.size()<<endl;
            if (visited.size()<maxNum){
                maxNum = visited.size();
                res = initial[leftOut];
            }
        }
        return res;
    }
    //unordered_set<int> visiting;
    
    void sink(int i, vector<vector<int>>& tempG, unordered_set<int>&visited){
        if (visited.count(i)) return;
        visited.insert(i);
        for(int j = 0; j < tempG[i].size(); ++j){
            if(tempG[i][j])sink(j,tempG, visited);
        }
    }
};
