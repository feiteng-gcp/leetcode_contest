class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        sort(initial.begin(), initial.end());
        int nNodes = graph.size();
        int result = initial[0];
        int minAffected = INT_MAX;
        for (int& removed : initial) {
            int affected = 0;
            queue<int> bfsQ;
            vector<bool> visited(nNodes, false);
            for (int& start : initial) {
                if (start != removed) {
                    bfsQ.push(start);
                    visited[start] = true;
                    affected++;
                }
            }
            while (!bfsQ.empty()) {
                int node = bfsQ.front();
                bfsQ.pop();
                for (int neighbor = 0; neighbor < nNodes; neighbor++) {
                    if (neighbor == removed || graph[node][neighbor] == 0 || visited[neighbor]) {
                        continue;
                    }
                    affected++;
                    visited[neighbor] = true;
                    bfsQ.push(neighbor);
                }
            }
            if (affected < minAffected) {
                result = removed;
                minAffected = affected;
            }
        }
        return result;
    }
};
