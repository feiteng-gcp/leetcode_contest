class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int m = initial.size();
        int n = graph.size();
        int minInfected = 1000000000;
        int nodeMinInfected = 1000000000;
        
        
        for (int x = 0; x < m; x++) {
            int rem = initial[x];
            
            vector< int > a[n];
            for (int i = 0; i < n; i++) {
                if (i == rem) continue;
                for (int j = i+1; j < n; j++) {
                    if (j == rem) continue;
                    if (graph[i][j]) {
                        a[i].push_back(j);
                        a[j].push_back(i);
                    }
                }
            }
            
            queue<int> q;
            vector<bool> dd(n, false);
            for (int i = 0; i < m; i++) {
                if (i == x) continue;
                q.push(initial[i]);
                dd[initial[i]] = true;
            }
            int infected = m-1;
            
            while (!q.empty()) {
                int u = q.front();
                q.pop();
                for (int i = 0; i < int(a[u].size()); i++) {
                    int v = a[u][i];

                    if (!dd[v]) {
                        dd[v] = true;
                        q.push(v);
                        infected++;
                    }
                }
            }
            
            if (infected < minInfected) {
                minInfected = infected;
                nodeMinInfected = rem;
            }
            else if (infected == minInfected) {
                if (rem < nodeMinInfected) {
                    nodeMinInfected = rem;
                }
            }
            
        }
        return nodeMinInfected;
    }
};
