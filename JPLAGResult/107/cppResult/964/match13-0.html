class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int ret = 0; int global_cnt = INT_MAX;
        int n = graph.size();
        sort(initial.begin(), initial.end());
        for (int i = 0; i < initial.size(); i ++){
            int rm = initial[i]; int cnt = 0;
            queue<int> node_queue; vector<bool> visited(n, false);
            for (int j = 0; j < initial.size(); j ++){
                if (i == j) continue;
                node_queue.push(initial[j]); visited[initial[j]] = true;
                cnt ++;
            }
            while(!node_queue.empty()){
                int cur = node_queue.front(); node_queue.pop();
                for (int d = 0; d < n; d ++){
                    if (d == rm) continue;
                    if (graph[cur][d] == 0) continue;
                    if (visited[d] == true) continue;
                    node_queue.push(d); visited[d] = true;
                    cnt ++;
                }
            }
            if (cnt < global_cnt) {
                global_cnt = cnt;
                ret = rm;
            }
        }
        return ret;
    }
};
