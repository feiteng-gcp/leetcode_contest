#include <queue>

const int N = 305;
int f[N];

int bfs(vector<vector<int>>g, vector<int> a, int k) {
    int n = g.size();
    queue<int> q;
    memset(f, 0, sizeof(f));
    for (int i = 0; i < a.size(); i++) {
        int u = a[i];
        if (u == k) continue;
        f[u] = 1;
        q.push(u);
    }

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int i = 0; i < n; i++) {
            if (f[i]) continue;
            if (i == k) continue;
            if (!g[u][i]) continue;
            f[i] = 1;
            q.push(i);
        }
    }
    int ans = 0;
    for (int i = 0; i < n; i++) ans += f[i];
    return ans;
}



class Solution {
public:
    int minMalwareSpread(vector<vector<int>>g, vector<int> a) {
        int n = g.size(), p = -1, num = n, tmp;
        for (int i = 0; i < a.size(); i++) {
            tmp = bfs(g, a, a[i]);
            if (tmp < num) num = tmp, p = a[i];
            else if (tmp == num && a[i] < p) p = a[i];
        }
        return p;
    }
};
