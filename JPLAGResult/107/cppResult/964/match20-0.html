class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
         sort(initial.begin(), initial.end());
        int ans = -1;
        int best = graph.size() + 1;
        for (auto &node : initial) {
            int cur = bfs(node, graph, initial);
            if (cur < best) {
                best = cur;
                ans = node;
            }
        }
        return ans;
    }
    int bfs(int forbid, vector<vector<int>>& graph, vector<int>& initial) {
        queue<int> q;
        int n = graph.size();
        vector<bool> vst(n, false);
        int ret = 0;
        for (auto &node : initial)
            if (node != forbid) {
                q.push(node);
                vst[node] = true;
                ret++;
            }
        while (!q.empty()) {
            int x = q.front();
            q.pop();
            for (int i = 0; i < n; i++)
                if (i != forbid && graph[x][i] == 1 && !vst[i]) {
                    q.push(i);
                    vst[i] = true;
                    ret++;
                }
        }
        return ret;        
    }    
};
