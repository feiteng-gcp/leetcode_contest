#include <vector>
#include <queue>

using namespace std;

class Solution {
public:

    int find(int x, vector<int> &uset) {
        return uset[x]==x?x:uset[x]=find(uset[x], uset);
    }

    void merge(int x, int y, vector<int> &uset) {
        int px = find(x, uset);
        int py = find(y, uset);
        if(px < py)
            uset[py] = px;
        else
            uset[px] = py;
    }

    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n = graph.size();
        int ansmi, ansidx;
        ansidx = ansmi = 2100000000;
        for(const int vi: initial) {
            int start = vi;
            std::queue<int> que;
            vector<int> vs(n, 0);

            for(const int u: initial) {
                if(u != start) {
                    que.push(u);
                    vs[u] = 1;
                }
            }
            int s = 0;
            while(!que.empty()) {
                int u = que.front(); que.pop();
                s++;
                for(int j=0; j<n; j++) {
                    if(j!=start && graph[u][j] && !vs[j]) {
                        que.push(j);
                        vs[j] = 1;
                    }
                }
            }
            if(s < ansmi) {
                ansidx = start;
                ansmi = s;
            } else if (s == ansmi) {
                if(start < ansidx)
                    ansidx = start;
            }
        }
        return ansidx;
    }
};
