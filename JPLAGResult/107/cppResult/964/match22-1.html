class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& g, vector<int>& s) {
        int mi = 1003;
        int res = -1;
        int n = g.size();
        sort(s.begin(), s.end());
        for(int x : s){

            queue<int> que;
            vector<bool> used(n, false);
            for(int y : s) if(x != y) que.push(y), used[y] = true;
            while(!que.empty()){
                int now = que.front(); que.pop();
                for(int to = 0; to < n; ++to) if(to != x && g[now][to] && !used[to]){
                    used[to] = true;
                    que.push(to);
                }
            }
            int cnt = 0;
            for(int i = 0; i < n; ++i) cnt += used[i];
            if(mi > cnt){
                mi = cnt;
                res = x;
            }
        }
        return res;
    }
};
