class Solution {
public:
    
    vector<vector<int>> G;
    vector<bool> seen;
    vector<int> init;
    
    int dfs(int v) {
        if(seen[v]) return 0;
        seen[v] = true;
        int ans = 1;

        for(int i = 0; i < G.size(); ++i) {
            if(G[v][i])
                ans += dfs(i);
        }
        return ans;
    }
    
    int find(int v) {
        seen = vector<bool>(G.size(),0);
        seen[v] = true;
        int tot = 0;
        for(int x : init) {
            tot += dfs(x);
        }
        return tot;
    }
    
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        G = graph;
        sort(initial.begin(), initial.end());

        init = initial;
        int best = 0x7fffffff;
        int node = -1;
        for(int i = initial.size()-1; i >= 0; --i) {
            int q = find(initial[i]);
            //cout << initial[i] << " --> " << q << endl;
            if(q <= best) {
                node = initial[i];
                best = q;
            }
        }
        return node;
    }
};
