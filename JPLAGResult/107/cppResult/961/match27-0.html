#include <string>
#include <iostream>

using namespace std;

class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int len1 = name.size();
        int len2 = typed.size();
        if(len1 == 0)
            return len2 == 0;
        int i=0, j=0;
        while(i<len1 && j<len2) {
            char x1 = name[i];
            int c1 = 0;
            while(i<len1 && name[i] == x1) {
                c1++; i++;
            }
            char x2 = typed[j];
            int c2 = 0;
            while(j<len2 && typed[j] == x2) {
                c2++; j++;
            }
            if(x1 != x2 || c1 > c2)
                return false;
        }
        return i==len1 && j==len2;
    }
};
