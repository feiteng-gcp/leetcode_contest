class Solution {
public:
    bool isLongPressedName(string n, string t) {
        int i = 0, j = 0;
        while (i < n.size () && j < t.size ()) {
            if (n[i] == t[j]) ++i, ++j;
            else {
                if (j - 1 >= 0 && t[j] == t[j-1]) {
                    int x = j - 1;
                    while (j < t.size() && t[j] == t[x]) ++j;
                } else return false;
            }
        }
        return i == n.size() && j == t.size() || (j < t.size() && i == n.size() && all_of(t.begin() + j, t.end(), [c = t[j-1]](auto e){return e = c;}));
    }
};
