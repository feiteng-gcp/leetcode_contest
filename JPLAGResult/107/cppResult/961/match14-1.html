class Solution {
public:
    bool isLongPressedName(string name, string typed) {

        int i = 0, j = 0;
        while (i < name.length() && j < typed.length()) {
            if (typed[j] == name[i]) {
                i ++; j ++;
            } else if (i > 0 && typed[j] == name[i-1]) {
                j ++;
            } else {
                return false;
            }
        }
        if (i == name.length()) return true;
        else return false;
    }
};
