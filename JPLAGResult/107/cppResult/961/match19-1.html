class Solution {
public:
    bool isLongPressedName(string s, string t) {
        int n = s.length(), m = t.length();
        int j = 0;
        for (int i = 0; i < n; i ++){
            if (s[i] != t[j]) return false;
            j++;
            if (i == n - 1 || s[i] != s[i+1]){
                while (j < m && t[j] == s[i]) j++;
            }
        }
        return j == m;
    }
};
