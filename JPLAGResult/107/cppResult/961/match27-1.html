class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int i=0;

        int j=0;
        bool p = true;
        while(i<name.size() && j<typed.size()){
            char c = name[i];
            int ii = 0;
            while(i < name.size() && name[i] == c) {
                i++;
                ii++;
            }
            char d = typed[j];
            int jj = 0;
            while(j < typed.size() && typed[j] == d) {
                j++;
                jj++;
            }
            if(c!=d) return false;
            if(ii>jj) return false;
        }
        if(i!=name.size() || j!=typed.size()) return false;
        return true;
    }
};
