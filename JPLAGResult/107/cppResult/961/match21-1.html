class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        if (name.empty() || typed.empty()) return false;
        int i = 0;
        if (name[0] != typed[0]) return false;

        for (int j = 1; j < typed.length(); ++ j) {
            if (i < name.length() - 1 && name[i + 1] == typed[j]) {
                ++ i;
            } else if (name[i] != typed[j]) {
                return false;
            }
        }
        return i + 1 == name.length();
    }
};
