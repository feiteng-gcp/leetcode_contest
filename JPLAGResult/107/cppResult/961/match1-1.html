class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int pos = 0;
        for (char& c : name) {
            if (pos >= typed.length()) {
                return false;
            }
            if (typed[pos] == c) {
                ++pos;
            }
            else {
                if (pos == 0) {
                    return false;
                }
                while(typed[pos] != c && pos < typed.length()) {
                    if (typed[pos] != typed[pos-1]) {
                        return false;
                    }
                    else {
                        ++pos;
                    }
                }
                if (pos >= typed.length()) {
                    return false;
                }
                ++pos;
            }
        }
        if (pos == typed.length()) {
            return true;
        }
        else {
            char c = typed[pos];
            while (pos < typed.length()) {
                if (typed[pos] != c) {
                    return false;
                }
                ++pos;
            }
            return true;
        }
    }
};
