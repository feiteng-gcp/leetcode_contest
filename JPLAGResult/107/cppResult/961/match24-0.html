class Solution {
public:
    // alex, aaleex
    bool isLongPressedName(string name, string typed) {
        if (typed.size() < name.size()) return false;
        if (typed.empty() || name.empty()) return false;
        
        if (name[0] != typed[0]) return false;
        
        int i=1;
        int j=1;
        while (j<typed.size()) {
            if (i<name.size() && name[i] == typed[j]) {
                ++j;
                ++i;
            } else if (typed[j] == name[i-1]) {
                ++j;
            } else {
                return false;
            }
        }
        return j==typed.size() && i==name.size();
    }
};
