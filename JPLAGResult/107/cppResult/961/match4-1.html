class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int n = 0, t = 0;
        while (n < name.length() && t < typed.length()) {
            if (name[n] == typed[t]) {
                n++;
                t++;
            } else if (n > 0 && name[n-1] == typed[t]) {
                t++;
            } else {
                return false;
            }
        }
        return n == name.length();
    }
};
