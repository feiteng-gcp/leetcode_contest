class Solution {
 public:
  bool isLongPressedName(string name, string typed) {

    vector<pair<char, int>> a = Solve(name), b = Solve(typed);
    if (a.size() != b.size()) {
      return false;
    }
    for (int i = 0; i < a.size(); ++i) {
      if (a[i].first != b[i].first || a[i].second > b[i].second) {
        return false;
      }
    }
    return true;
  }

 private:
  vector<pair<char, int>> Solve(string& s) {
    vector<pair<char, int>> res;
    for (char c : s) {
      if (res.empty() || res.back().first != c) {
        res.emplace_back(c, 1);
      } else {
        ++res.back().second;
      }
    }
    return res;
  }
};
