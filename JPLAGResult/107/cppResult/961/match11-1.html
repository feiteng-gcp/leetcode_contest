class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int i = 0, j = 0;

        int n = name.size(), m = typed.size();
        while(i < n && j < m) {
            if(name[i] != typed[j]) return false;
            int i2 = i;
            while(i2 < n && name[i] == name[i2]) i2++;
            int j2 = j;
            while(j2 < m && typed[j2] == typed[j]) j2++;
            
            if(j2 - j < i2 - i) return false;
            i = i2;
            j = j2;
        }
        return i == n && j == m;
    }
};
