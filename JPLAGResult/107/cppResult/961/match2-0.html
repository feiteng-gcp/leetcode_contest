class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        if(typed.size()<name.size()) return false;
        if(name.size()==0) return false;
        int index1=0,index2=0;
        while(index1<name.size()&&index2<typed.size()){
            if(name[index1]==typed[index2]){
                index1++;
                index2++;
            }else{
                if(index1==0) return false;
                if(name[index1-1]==typed[index2]) ++index2;
                else return false;
            }
        }
        if(index1<name.size()) return false;
        while(index2<typed.size()){
            if(typed[index2]!=name[name.size()-1]) return false;
            ++index2;
        }
        return true;
    }
};
