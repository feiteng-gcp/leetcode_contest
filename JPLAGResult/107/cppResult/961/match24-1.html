class Solution {
public:
    bool isLongPressedName(string name, string typed) {

        if(name.size() > typed.size()) return false;
        int i=0;
        int j=0;
        while(i < name.size() && j < typed.size()) {
            //cout << "i = " << i << ", j= " << j << endl;
            if(name[i] == typed[j]) {
                ++i; ++j;
            }
            else if(j > 0 && typed[j-1] == typed[j]) {
                ++j;
            }
            else {
                return false;
            }
        }
        while(j < typed.size()) {
            if(typed[j] != typed[j-1]) {
                return false;
            }
            ++j;
        }
        return i == name.size();
    }
};
