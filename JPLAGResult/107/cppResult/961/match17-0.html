class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        if (name.size() == 0) return true;
        if (name.size() != 0 && typed.size() == 0) return false;
        int nameIndex = 0;
        int typedIndex = 0;
        while (typedIndex < typed.size()) {
            if (nameIndex < name.size() && name[nameIndex] == typed[typedIndex]) {
                nameIndex++;
                typedIndex++;
            } else if (typedIndex >= 1 && typed[typedIndex] == typed[typedIndex - 1]) {
                typedIndex++;
            } else {
                return false;
            }
        }
        if (nameIndex == name.size()) return true;
        else return false;
        
    }
};
