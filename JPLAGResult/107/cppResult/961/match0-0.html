class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int i=0;
        for(int j=0;j<typed.size()&&i<name.size();){
            if(name[i]==typed[j]){
                i++;
                j++;
            }
            else j++;
        }
        return i==name.size();
    }
};
