class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        vector<pair<char, int>> n = analyse(name);
        vector<pair<char, int>> t = analyse(typed);
        if (n.size() != t.size()) {
            return false;
        }
        for (int i = 0; i < n.size(); ++i) {
            if (n[i].first != t[i].first || n[i].second > t[i].second) {
                return false;
            }
        }
        return true;
    }
private:
    vector<pair<char, int>> analyse(string s) {
        vector<pair<char, int>> res;
        for (char c : s) {
            if (res.empty() || res[res.size() - 1].first != c) {
                res.push_back(make_pair(c, 1));
            } else {
                ++res[res.size() - 1].second;
            }
        }
        return res;
    }
};
