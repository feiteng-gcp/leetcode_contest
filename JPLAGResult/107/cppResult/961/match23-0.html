class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int p1 = 0;
        int p2 = 0;
        while (p1 < name.size()) {
            if (p2 == typed.size()) {return false;}
            if (name[p1] == typed[p2]) {
                p1++;
                p2++;
            } else if (p2 > 0 && p2 < typed.size() && typed[p2] == typed[p2-1]) {
                p2++;
            } else {
                return false;
            }
        }
        while (p2 < typed.size()) {
            if (typed[p2] != typed[p2-1]) {return false;}
            p2++;
        }
        return true;
    }
};
