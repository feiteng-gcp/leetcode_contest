class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int now = 0;
        for (int i = 0; i < typed.length(); ++i) {
            if(typed[i] == name[now]){
                now ++;
            }
            if(now >= name.length())
                return true;
        }
        return false;
    }
};
