class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int n = name.size(), m = typed.size();
        int j = 0;
        for(int i = 0;i<n;++i){
            if(j==m||typed[j]!=name[i])return 0;
            ++j;
            if(i==n-1||name[i]!=name[i+1])
                while(j<m&&typed[j]==name[i])
                    ++j;
        }
        return j==m;
    }
};
