class Solution {
public:
    bool isLongPressedName(string name, string typed) {
        int i = 0, j = 0;
        while (i < name.size() && j < typed.size()) {
            if (name[i] == typed[j]) {
                i += 1;
                j += 1;
            } else {
                while (j > 0 && j < typed.size() && typed[j] == typed[j-1]) {
                    j += 1;
                }
                if (name[i] == typed[j]) {
                    i += 1;
                    j += 1;
                } else {
                    return false;
                }
            }
        }
        if (i == name.size()) {
            while (j > 0 && j < typed.size() && typed[j] == typed[j-1]) {
                j += 1;
            }
            if (j == typed.size()) {
                return true;
            } else {
                return false;
            }
        } else {
            return false;
        }
    }
};
