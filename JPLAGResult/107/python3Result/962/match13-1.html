class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        n = len(S)
        mem = [[0, 0] for _ in range(n+1)]
        
        for idx, b in enumerate(S):
            if b == '0':
                mem
[idx+1][0] = mem[idx][0]
                mem[idx+1][1] = min(mem[idx][1] + 1, mem[idx][0])
            elif b == '1':
                mem[idx+1][0] = mem[idx][0] + 1
                mem[idx+1][1] = min(mem[idx][1], mem[idx][0])
        # print(mem)
        return min(mem[len(S)])
