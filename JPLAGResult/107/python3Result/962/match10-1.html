class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        pre_1 = [0 for i in range(len(S) + 1)]
        after_0 = []

        cnt = 0
        after_0.append(cnt)
        for c in S:
            if c == '1':
                cnt += 1
            after_0.append(cnt)

        cnt 
= 0
        for i in range(len(S)-1, -1, -1):
            if S[i] == '0':
                cnt += 1
            pre_1[i] = cnt


        # print(after_0)
        # print(pre_1)

        ans = [pre_1[i] + after_0[i] for i in range(len(S) + 1)]
        return min(ans)
