class Solution:
    def minFlipsMonoIncr(self, S):
        ls=len(S)
        if ls==0:
            return 0
        fo,lz=0,ls-1
        qo=[0]*ls
        hz=[0]*ls
        for i in range(1,ls):
            qo[i]=qo[i-1]
            if S[i-1]=='1':
                qo[i]+=1
        for i in range(ls-2,-1,-1):
            hz[i]=hz[i+1]
            if S[i+1]=='0':
                hz[i]+=1
        ans=20000
        #print(qo)
        for i in range(ls):
            ans=min(ans,qo[i]+hz[i])
        return ans
        
        """
        :type S: str
        :rtype: int
        """
