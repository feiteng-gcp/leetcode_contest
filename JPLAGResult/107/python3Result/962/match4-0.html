class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        n = len(S)
        left1s = 0 # num of 1s in S[:i]
        right0s = S.count('0') # num of 0s in S[i:]
        #print(left1s, right0s)
        best = left1s + right0s
        for i in range(1, n + 1):
            if S[i-1] == '0':
                right0s -= 1
            else: # 1
                left1s += 1
            #print(left1s, right0s)
            best = min(best, left1s + right0s)
        return best
