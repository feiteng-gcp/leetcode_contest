class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        N = len(S)
        left, right = [0] * N, [0] * N
        for i, c in enumerate(S):
            left[i] = left[i-1] + (c == '1') if i > 0 else int(c == '1')
        for i, c in list(enumerate(S))[::-1]:
            right[i] = right[i+1] + (c == '0') if i < N-1 else int(c == '0')
        ans = min(S.count('1'), N - S.count('1'))
        for i in range(N):
            ans = min(ans, left[i] - (S[i] == '1') + right[i] )
        return ans
