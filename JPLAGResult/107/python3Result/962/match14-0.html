class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        S = [c for c in S]
        N = len(S)
        l0, r0 = [0]*(N), [0]*(N)
        l1, r1 = [0]*(N), [0]*(N)
        
        for i in range(N):
            if S[i]=='0':
                l0[i] = (l0[i-1] if i-1>=0 else 0)+1
                l1[i] = (l1[i-1] if i-1>=0 else 0)
            else:
                l0[i] = (l0[i-1] if i-1>=0 else 0)
                l1[i] = (l1[i-1] if i-1>=0 else 0)+1
        
        for i in range(N-1, -1, -1):
            if S[i]=='0':
                r0[i] = (r0[i+1] if i+1<N else 0)+1
                r1[i] = (r1[i+1] if i+1<N else 0)
            else:
                r0[i] = (r0[i+1] if i+1<N else 0)
                r1[i] = (r1[i+1] if i+1<N else 0)+1
            
        # print(l0)
        # print(l1)
        # print(r0)
        # print(r1)
        
        ans = min(l0[N-1], l1[N-1])
        for i in range(N):
            ans = min(l1[i]+(r0[i+1] if i+1<N else 0), ans)
        return ans
