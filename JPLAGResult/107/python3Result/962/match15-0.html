class Solution:
    def minFlipsMonoIncr(self, S):
        import collections
        """
        :type S: str
        :rtype: int
        """
        if not S or len(S) < 1:
            return 0
        i = 0
        while S[i] == '0':
            i += 1
        if i == len(S):
            return 0
        s = S[i:]
        #print(s)
        counter = collections.Counter(s)
        count0 = counter['0']
        count1 = counter['1']
        if count0 == 0 or count1 == 0:
            return 0
        #print(count0)
        #print(count1)
        dp = [0] * (len(s) + 1) # num of 0 to the left
        dp[0] = 0
        res = len(s)
        for i in range(1, len(s) + 1):
            dp[i] = dp[i - 1] + (1 if s[i - 1] == '0' else 0)
            
        #print(dp)
        for i in range(len(s) + 1):
            cur = (i - dp[i]) + (count0 - dp[i])
            #print(cur)
            res = min(res, cur)
        return res
