class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        n = len(S)
        left = [None]*(n+1)
        left[0] = 0
        for i in range(n):
            left[i+1] = left[i] + int(S[i]=='1')
            
        right = [None]*(n+1)
        right[n] = 0
        for i in range(n-1, -1, -1):
            right[i] = right[i+1] + int(S
[i]=='0')
            
        ans = n+2
        for i in range(n+1):
            ans = min(ans, left[i]+right[i])
        return ans
