class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        if len(S) == 1:
            return 0
        lis = [(0,1)]
        a1 = 0; b1 = 0
        for i in range(1, len(S)):
            if S[i] == S[i-1]:
                a1 = lis[-1][0]
                b1 = lis[-1][1] + 1
            elif S[i] == '1' and S[i-1] == '0':
                a1 = min(lis[-1][0],lis[-1][1])
                b1 = lis[-1][0] + 1
            else:
                a1 = lis[-1][1]
                b1 = lis[-1][0] + 1
            lis.append((a1,b1))
        return min(lis[-1][0],lis[-1][1])
