class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        N 
= len(S)
        left = [0] * (N + 2)
        right = [0] * (N + 2)
        for i in range(N):
            if S[i] == '1':
                left[i + 1] = left[i] + 1
            else:
                left[i + 1] = left[i]
        left[N + 1] = left[N]
        for i in range(N - 1, -1, -1):
            if S[i] == '0':
                right[i + 1] = right[i + 2] + 1
            else:
                right[i + 1] = right[i + 2]
        # print(left)
        # print(right)
        right[0] = right[1]
        ans = float('inf')
        for i in range(N + 1):
            ans = min(ans, left[i] + right[i + 1])
        return ans
