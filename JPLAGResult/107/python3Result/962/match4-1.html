class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        ones = 0
        for ch in S:
            if ch == '1':
                ones += 1
        n = len(S)
        best = min
(ones, n - ones)
        flips = 0
        for i in range(n-1, -1, -1):
            if S[i] == '0':
                flips += 1
            else:
                ones -= 1
            best = min(best, ones + flips)
        return best
