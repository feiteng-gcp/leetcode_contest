class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        if not S:
            return 0
        flips = [0]*len(S)
        count1 = 0
        for i in range(len(S)):
            if S[i] == '1':
                count1 += 1
            flips[i] = count1
        count0 = 0
        for i in range(len(S))[::-1]:
            if S[i] == '0':
                count0 += 1
            flips[i] += count0
        print(flips)
        return min(flips)-1
