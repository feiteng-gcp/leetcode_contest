class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        n = len(S)
        ones = [0] * (n + 1)
        zeros = [0] * (n + 1)
        
        for i in range(n):
            ones[i + 1] = ones[i] if S[i] == '0' else ones[i] + 1
        
        for i in range(n-1, -1, -1):
            zeros[i] = zeros[i + 1] if S[i] == '1' else zeros[i + 1] + 1
        
        min_change = n
        for i in range(n + 1):
            min_change = min(min_change, zeros[i] + ones[i])

        return min_change
