class Solution:
    def minFlipsMonoIncr(self, S):
        """
        :type S: str
        :rtype: int
        """
        right_zero = S.count("0")
        left_one = 0
        m 
= right_zero + left_one
        for c in S:
            if c == "0":
                right_zero -= 1
            else:
                left_one += 1
            m = min(m, right_zero + left_one)
        return m
