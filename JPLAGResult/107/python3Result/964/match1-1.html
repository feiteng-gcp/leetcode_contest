class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        n 
= len(graph)
        neb = [[] for _ in range(n)]
        for i in range(n):
            for j in range(n):
                if graph[i][j] == 1:
                    neb[i].append(j)
                    neb[j].append(i)
        def compute(k):
            q = []
            infected = [0] * n
            for i in initial:
                if i != k:
                    infected[i] = 1
                    q.append(i)
            while q:
                curr = q.pop()
                for target in neb[curr]:
                    if target != k and infected[target] == 0:
                        q.append(target)
                        infected[target] = 1
            return sum(infected)
        _min = n
        minsave = initial[0]
        for save in range(len(initial)):
            ret = compute(initial[save])
            # print(initial[save], ret)
            if ret < _min:
                _min = ret
                minsave = initial[save]
            elif ret == _min:
                minsave = min(minsave, initial[save])
        # print(_min)
        return minsave
