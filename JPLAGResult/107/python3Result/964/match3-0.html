import queue

class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        n = len(graph[0])
        def traverse(remove_i):
            q = queue.Queue()
            visit = set()
            for i in initial:
                if i != remove_i:
                    q.put(i)
                    visit.add(i)
            while not q.empty():
                i = q.get()
                for node in range(n):
                    if node != remove_i and graph[i][node] == 1 and node not in visit:
                        q.put(node)
                        visit.add(node)
            return len(visit)
        node, min_num = 0, 500
        for remove_i in initial:
            num = traverse(remove_i)
            if num < min_num or (num == min_num and remove_i < node):
                min_num = num
                node = remove_i
        return node
