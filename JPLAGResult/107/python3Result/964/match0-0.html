class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        initial.sort()
        adj_list = collections.defaultdict(set)
        for i, row in enumerate(graph):
            for j, val in enumerate(row):
                if not val: continue
                adj_list[i].add(j)

        def bfs(v0, u):
            seen, level = set(), set(v0)
            while level:
                seen |= level
                level = {nb for x in level for nb in adj_list[x] if nb not in seen and nb != u}
            return len(seen)
        
        M = [None] * len(initial)
        for i in range(len(initial)):
            M[i] = bfs([x for j, x in enumerate(initial) if j != i], initial[i])

        return initial[M.index(min(M))]
