class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        def dfss():
            minCnt = float('inf')
            minCnti = -1
            for i in initial:
                
                # dfs from all initial but i
                cnt = 0
                beenInStack = [False] * n
                for j in initial:
                    if j == i:
                        continue
                    if beenInStack[j]:
                        continue
                    st = [j]
                    while st:
                        u = st.pop()
                        cnt += 1
                        # go through neigs
                        for v in neigs[u]:
                            if beenInStack[v] or v == i:
                                continue
                            st.append(v)
                            beenInStack[v] = True
                #print(i, cnt)
                if cnt < minCnt:
                    minCnt = cnt
                    minCnti = i
            #print(minCnt)
            return minCnti
        
        initial.sort()
        n = len(graph)
        # build adj list
        neigs = [[] for i in range(n)]
        for u in range(n):
            for v in range(u+1, n):
                if graph[u][v]:
                    neigs[u].append(v)
                    neigs[v].append(u)
        return dfss()
