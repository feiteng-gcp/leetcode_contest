class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """

        def helper(graph, initial, badpoint):
            V = len(graph)

            visited = [False for i in range(V)]
            q = []
            for i in initial:
                if i == badpoint: continue
                visited[i] = True
                q.append(i)

            while q:
                v = q.pop(0)
                for w in range(V):
                    if (not visited[w]) and (w != badpoint) and (graph[v][w] == 1):
                        visited[w] = True
                        q.append(w)

            count = 0
            for i in visited:
                if i: count += 1

            return count

        min_count = len(graph) + 1
        index 
= -1;

        for badpoint in initial:
            count = helper(graph, initial, badpoint)

            if (count < min_count) or (count == min_count and badpoint < index):
                min_count = count
                index = badpoint

        return index
