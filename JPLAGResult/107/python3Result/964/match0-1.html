class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """

        
        initial = sorted
(initial)
        adj_list = collections.defaultdict(set)
        for i, r in enumerate(graph):
            for j, v in enumerate(r):
                if not v: continue
                adj_list[i].add(j)
        
        def bfs(v,re):
            visited, level = set(), set(v)
            while level:
                visited |= level
                new_level = set()
                for x in level:
                    for j in adj_list[x]:
                        if j not in visited and j != re:
                            new_level.add(j)
                level = new_level
            return len(visited)

        M = [None] * len(initial)
        for i in range(len(initial)):
            ip = []
            for l, x in enumerate(initial):
                if l != i:
                    ip.append(x)
            M[i] = bfs(ip, initial[i])
        
            

        return initial[M.index(min(M))]
