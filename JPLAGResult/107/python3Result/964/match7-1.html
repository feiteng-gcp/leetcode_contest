class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        
        n = len(graph)
        initial.sort()
        
        ans = (math.inf, -1)
        
        def getFather(father, x):
            if father[x] != x:
                father[x] = getFather(father, father[x])
            return father[x]
        
        for removed in initial:
            father = [i for i in range(0, n)]
            
            
for i in range(0, n):
                if i == removed:
                    continue
                for j in range(0, n):
                    if j == removed:
                        continue
                    if graph[i][j]:
                        f1 = getFather(father, i)
                        f2 = getFather(father, j)
                        father[f2] = f1
            
            final = [False] * n
            
            for x in initial:
                if x != removed:
                    final[getFather(father, x)] = True
            
            c = 0
            for i in range(0, n):
                if i != removed:
                    if final[getFather(father, i)]:
                        c += 1
            #print(removed, father, final, c)
            if c < ans[0]:
                ans = (c, removed)
                
        return ans[1]
