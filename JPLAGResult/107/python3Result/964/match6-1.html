import heapq
import math
from collections import defaultdict
class Solution:
    def minMalwareSpread(self, graph, initial):
        
        def calculateAffect(ignore):
            l = len(graph)
            d = [0]*l
            size = [1]*l
            
def find(x):
                #print(i,d[i])
                if d[x] == x:
                    return x
                d[x] = find(d[x])
                return d[x]
            for i in range(l):
                d[i] = i
            for i in range(l):
                if i == ignore:
                    continue
                for j in range(i+1,l):
                    if j == ignore:
                        continue
                    if graph[i][j] ==1:
                        ri = find(i)
                        rj = find(j)
                        if ri!=rj:
                            d[ri] = rj
                            size[rj] = size[ri]+size[rj]
            #print("size",size)
            v = {}     
            res = 0
            for i in range(len(initial)):
                if initial[i] == ignore:
                    continue
                ri = find(initial[i])
                if ri in v:
                    continue
                v[ri] = True
                #print(initial[i],ri,size[ri])
                res+=size[ri]
            return res
        origin = calculateAffect(-1)
        res = 0
        best = initial[0]
        initial = sorted(initial)
        for i in range(len(initial)):
            r = calculateAffect(initial[i])
            if origin - r > res:
                res = origin -r
                best = initial[i]
            #print(initial[i],origin,r,res)
        return best
