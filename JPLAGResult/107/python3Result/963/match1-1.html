class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        n1 = 0
        if len(A)<3: return [-1,-1]
        
        for n in A:
            if n == 1:
                n1 += 1
        
if n1%3 != 0:
            return [-1,-1]
        if n1 ==0:
            return [0,2]
        
        endzero = 0
        j = len(A)-1
        while A[j]==0:
            endzero += 1
            j -= 1
        
        id1 = 0
        id2 = 0
        id3 = 0
        cn1 = 0
        each1 = int(n1/3)
        for i in range(len(A)):
            if A[i] == 1:
                cn1 += 1
                if cn1 == 1:
                    id1 = i
                elif cn1 == each1 + 1:
                    id2 = i
                elif cn1 == 2*each1 +1:
                    id3 = i
        
        l1 = j-id3
            
        if id2-id1-l1-1< endzero or id3-id2-l1-1< endzero:
            return [-1,-1]
        
        for k in range(l1):
            if A[id1+k+1]!=A[id2+k+1] or A[id1+k+1]!=A[id3+k+1]:
                return [-1,-1]
        return [id1+l1+endzero, id2+l1+endzero+1]
