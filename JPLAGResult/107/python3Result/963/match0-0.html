class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        
        one_index = []
        
        for i, c in enumerate(A):
            if c == 1:
                one_index.append(i)
        
        N = len(one_index)
        if N == 0:
            return [0, len(A) - 1]
        if N % 3 != 0:
            return [-1, -1]
        
        tail_zero = len(A) - one_index[-1] - 1
        
        n = N // 3
        
        if (one_index[n] - one_index[n-1] - 1) < tail_zero:
            return [-1, -1]
        elif (one_index[2*n] - one_index[2*n-1] - 1) < tail_zero:
            return [-1, -1]
        else:
            x = one_index[n] - one_index[0]
            y = one_index[2*n] - one_index[0]
            
            for i in range(1, n):
                a = one_index[i]
                b = one_index[i+n]
                c = one_index[i+2*n]
                if ((b - a) != x) or ((c - a) != y):
                    return [-1, -1]
        return [one_index[n-1] + tail_zero, one_index[2*n-1] + tail_zero + 1]
