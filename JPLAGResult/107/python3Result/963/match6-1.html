class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """

        def deleteLeft0s(s):
            i = 0
            while i < len(s) and s[i] == 0:
                i += 1

            return s[i:]
        count_index = [-1]
        # count_index[5] = 30  means A[30] is 5th (start from 1st) '1'

        for (i, c) in enumerate(A):
            if c == 1:
                count_index.append(i)

        N = len(count_index) - 1

        if N % 3 != 0: return [-1, -1]
        if N == 0: return [0, 2]

        p = count_index[N // 3]
        q = count_index[N // 3 * 2]
        r = count_index[N]

        right_move = len(A) - 1 - r
        if count_index[N // 3 + 1] - p <= right_move: return [-1, -1]
        if count_index[N // 3 * 2 + 1] - q <= right_move: return [-1, -1]

        p += right_move
        q += right_move

        sp = A[:p+1]
        sq 
= A[p+1:q+1]
        sr = A[q+1:]


        sp = deleteLeft0s(sp);
        sq = deleteLeft0s(sq);
        sr = deleteLeft0s(sr);

        if sp == sq and sq == sr:
            return [p, q+1]
        return [-1, -1]
