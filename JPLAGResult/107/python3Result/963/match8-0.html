class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        n = len(A)
        one_count = A.count(1)
        if one_count % 3 != 0:
            return [-1, -1]
        elif one_count == 0:
            return [0, n-1]
        idxs = []
        ones = one_count // 3
        # print(ones)
        i = 0
        count = 0
        while i < n:
            if A[i] == 1:
                if not count:
                    idxs.append(i)
                count += 1
            if count == ones:
                idxs.append(i)
                count = 0
            # print(i, count)
            i += 1
        # print(idxs)
        if len(idxs) == 6:
            binary = A[idxs[4]:]
            length = n - idxs[4]
            if A[idxs[0]:idxs[0]+length] == binary and idxs[0]+length >= idxs[1] and \
                A[idxs[2]:idxs[2]+length] == binary and idxs[2]+length >= idxs[3]:
                return [idxs[0]+length-1, idxs[2]+length]
        return [-1, -1]
