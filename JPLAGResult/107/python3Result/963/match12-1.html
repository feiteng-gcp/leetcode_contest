class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        
        totalOne = sum(A)
        n 
= len(A)
        
        if totalOne % 3 != 0:
            return [-1, -1]
        
        if totalOne == 0:
            return [0, 2]
        
        tailingZero = 0
        k = n - 1
        while True:
            if A[k] == 1:
                break
            else:
                tailingZero += 1
            k -= 1
        
        c = totalOne // 3
        i = 0
        while True:
            if A[i] == 1:
                c -= 1
                if c == 0:
                    break
            i += 1
        
        c = totalOne // 3 * 2
        j = 0
        while True:
            if A[j] == 1:
                c -= 1
                if c == 0:
                    j += 1
                    break
            j += 1
            
        i += tailingZero
        j += tailingZero
        
        begin1 = 0
        end1 = i + 1
        begin2 = i + 1
        end2 = j
        begin3 = j
        end3 = n
        
        def isValid(a, begin, end):
            n = len(a)
            return begin >= 0 and end <= n and begin < end
        
        print(begin1, end1, begin2, end2, begin3, end3)
        
        if not isValid(A, begin1, end1):
            print("d1")
            return [-1, -1]
        if not isValid(A, begin2, end2):
            print("d2")
            return [-1, -1]
        if not isValid(A, begin3, end3):
            print("d3")
            return [-1, -1]
        
        def areSame(a, begin1, end1, begin2, end2):
            n1 = end1 - begin1
            n2 = end2 - begin2
            nMax = max(n1, n2)
            for i in range(0, nMax):
                i1 = end1 - nMax + i
                i2 = end2 - nMax + i
                x1 = a[i1] if i1 >= begin1 else 0
                x2 = a[i2] if i2 >= begin2 else 0
                if x1 != x2:
                    return False
            return True
        
        if not areSame(A, begin1, end1, begin2, end2):
            print("s12")
            return [-1, -1]
        if not areSame(A, begin2, end2, begin3, end3):
            print("s23")
            return [-1, -1]
        
        return [begin2 - 1, begin3]
