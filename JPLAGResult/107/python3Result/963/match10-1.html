class Solution:
    def get_last_zero(self, A):
        i = len(A) - 1
        while i>=0:
            if A[i] == 0:
                i -= 1
            else:
                return len(A) - i - 1
        return len(A) 
        
    def get_first_letter(self, A):
        sum_one 
= 0
        for i in A:
            sum_one += i
            
        if sum_one % 3 != 0:
            return (-1,-1,-1)
        
        temp_one = 0
        res = []
        for idx, num in enumerate(A):
            if num == 1 and temp_one % (sum_one / 3) == 0:
                res.append(idx)
            temp_one += num
        return (res[0], res[1], res[2])
    
    def get_pre_zeros(self, A, pos):
        i = pos - 1
        sum_zero = 0
        while i >= 0:
            if A[i] == 0:
                sum_zero += 1
            else:
                return sum_zero
            i -= 1
    
    def judge_mid_zero(self, A, zeros, pos1, pos2):
        mid_zero1 = self.get_pre_zeros(A, pos1)
        mid_zero2 = self.get_pre_zeros(A, pos2)
        return mid_zero1, mid_zero2
        
    def threeEqualParts(self, A):
        zeros = self.get_last_zero(A)
        if zeros == len(A):
            return [0,2]
        i, j, k = self.get_first_letter(A)
        # print(i,j,k)
        if i==-1:
            return [-1, -1]
        zero1, zero2 = self.judge_mid_zero(A, zeros, j, k)
        
        if zero1<zeros or zero2<zeros:
            return [-1,-1]
        
        length = 0  
        while k + length < len(A):
            if A[i+length] != A[j+length] or A[j+length] != A[k+length]:
                return [-1, -1]
            else:
                length += 1
        return [j-(zero1-zeros)-1, k-(zero2-zeros)]
        """
        :type A: List[int]
        :rtype: List[int]
        """
