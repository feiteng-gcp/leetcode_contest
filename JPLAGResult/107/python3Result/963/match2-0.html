class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        N = sum(A)
        if N == 0: return [0, 2]
        if N % 3 != 0: return [-1, -1]
        j, cnt, trail = len(A) - 1, 0, 0
        while cnt < N // 3:
            cnt += A[j]
            if cnt == 0: trail += 1
            j -= 1
        
        j0 = j
        while A[j] == 0: j -= 1
        j += trail
            
        i, temp = j, len(A) - 1
        while temp > j0:
            if A[i] != A[temp]: return [-1, -1]
            i, temp = i - 1, temp - 1
            
        i0 = i
        while A[i] == 0: i -= 1
        i += trail
            
        k, temp = i, len(A) - 1
        while temp > j0:
            if A[k] != A[temp]: return [-1, -1]
            k, temp = k - 1, temp - 1
        
        return [i, j+1]
