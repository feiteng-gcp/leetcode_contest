class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        if not any(A):
            return [0, len(A) - 1]
        
        def b_to_d(A, start, end):
            base = 2 ** (end - start - 1)
            result = 0
            for a in A[start:end]:
                result += a * base
                base //= 2
            return result
        import math
        index = [i for i, v in enumerate(A) if v == 1]
        #print(index)
        length = len(A)
        index_length = len(index)
        if index_length % 3 == 0:
            last_start = index[index_length // 3 * 2]
            second_start = index[index_length // 3]
            last = b_to_d(A, last_start, length)
            first = b_to_d(A, 0, second_start)
            second = b_to_d(A, second_start, last_start)
            #print(first, second, last, second_start, last_start)
            if first % last == 0 and second % last == 0:
                return [second_start - int(math.log2(first // last)) - 1, last_start - int(math.log2(second // last))]
        return [-1, -1]
