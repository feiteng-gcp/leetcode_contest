class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        pos = []
        
        for i, x in enumerate(A):
            if x == 1:
                pos.append(i)

        n = len(pos) # number of ones
        
        if n == 0:
            return [0, len(A) - 1]
        
        if n % 3 != 0:
            return [-1, -1]
        
        m = n // 3 # number of ones in each part
        
        tail_1 = pos[m - 1]
        tail_2 = pos[m * 2 - 1]
        tail_3 = pos[m * 3 - 1]
        
        k = len(A) - 1 - tail_3 # number of 0s in tail
        
        while k:
            tail_1 += 1
            tail_2 += 1
            
            if A[tail_1] == 1 or A[tail_2] == 1:
                return [-1, -1]
            
            k -= 1
        
        head_1 = 0
        head_2 = tail_1 + 1
        head_3 = tail_2 + 1
        
        #s1 = A[head_1: tail_1 + 1]
        #s2 = A[head_2: tail_2 + 1]
        #s3 = A[head_3: ]
        
        while A[head_1] == 0:
            head_1 += 1
        
        while A[head_2] == 0:
            head_2 += 1
            
        while A[head_3] == 0:
            head_3 += 1
        
        if not (head_1 - tail_1 == head_2 - tail_2 == head_3 - len(A) + 1):
            return [-1, -1]
        
        s1 = A[head_1: tail_1 + 1]
        s2 = A[head_2: tail_2 + 1]
        s3 = A[head_3: ]
        
        for x,y,z in zip(s1, s2, s3):
            if not (x == y == z):
                return [-1, -1]
        
        return [tail_1, tail_2+1]
