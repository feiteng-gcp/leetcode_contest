class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        one_num = sum(A)
        if one_num % 3 != 0:
            return [-1, -1]
        if one_num == 0:
            if len(A) < 3:
                return [-1, -1]
            else:
                return [0, 2]
        N = len(A)
        one_num_in_digit 
= one_num // 3
        # print(one_num_in_digit)
        ends_one_idx = []
        i = N - 1
        count = 0
        while i >= 0:
            if A[i] == 1:
                # print(i, count, one_num_in_digit)
                if count % one_num_in_digit == 0:
                    ends_one_idx.insert(0, i)
                count += 1
                    
            i -= 1
        # print(ends_one_idx)
        tail = N - 1 - ends_one_idx[2]
        ends_one_idx = [each + tail for each in ends_one_idx]
        list1 = A[:ends_one_idx[0] + 1]
        list2 = A[ends_one_idx[0] + 1:ends_one_idx[1] + 1]
        list3 = A[ends_one_idx[1] + 1:ends_one_idx[2] + 1]
        num1 = self.list2str(list1)
        num2 = self.list2str(list2)
        num3 = self.list2str(list3)
        # print(num1, num2, num3)
        if num1 == num2 and num2 == num3:
            return [ends_one_idx[0], ends_one_idx[1] + 1]
        else:
            return [-1, -1]
        
        
    def list2str(self, l):
        s = ''
        for each in l:
            s += str(each)
        ans = int(s, 2)
        return ans
