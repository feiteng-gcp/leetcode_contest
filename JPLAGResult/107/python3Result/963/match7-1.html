class Solution:
    def threeEqualParts(self, A):
        l = len(A)
        sum = A.count(1)
        if sum % 3 != 0:
            return [-1, -1]
        if sum == 0:
            return [0, 2]
        num = [0 for i in range(sum + 1)]
        n = 0
        for i in range(len(A)):
            if A[i] == 1:
                n += 1
                num[n] = i
        n1 = l - num[-1] - 1
        n2 = num[2 * sum // 3 + 1] - num[2 * sum // 3] - 1
        n3 = num[sum // 3 + 1] - num
[sum // 3] - 1
        if A[num[1]:num[sum // 3]] != A[num[sum // 3 + 1]:num[sum * 2 // 3]] or A[num[sum*2//3+1]:num[-1]] != A[num[sum // 3 + 1]:num[sum * 2 // 3]]:
            return [-1,-1]
        if n2 >= n1 and n3 >= n1:
            return [num[sum // 3] + n1, num[sum * 2 // 3] + n1 + 1]
        return [-1, -1]
