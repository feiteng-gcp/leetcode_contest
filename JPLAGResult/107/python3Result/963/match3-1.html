class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        
        # we need to find out three parts with same amount of ones
        n = len(A)
        ones = []
        for i in range(n):
            if A[i] == 1:
                ones.append(i)
        num_ones 
= len(ones)
        if num_ones == 0:
            return [0,2]
        if num_ones % 3 != 0:
            return [-1, -1]
        
        split1 = num_ones // 3
        split2 = 2 * split1
        
        # test the last zeros
        # print(split1)
        # print(split2)
        # print(ones)
        last_zeros = n - ones[-1]
        if (split1 > 0 and ones[split1] - ones[split1 - 1] < last_zeros) or (split2 > 0 and ones[split2] - ones[split2 - 1] < last_zeros):
            return [-1, -1]
        i_o, j_o = 0, 0
        if split1 > 0:
            i_o = ones[split1-1] + last_zeros - 1
        if split2 > 0:
            j_o = ones[split2-1] + last_zeros
        
        i, j, k = 0, split1, split2
        for i in range(split1 - 1):
            dif = ones[i + 1] - ones[i]
            if dif != ones[j + 1] - ones[j] or dif != ones[k + 1] - ones[k]:
                return [-1, -1]
            j += 1
            k += 1
        
        return [i_o,j_o]
