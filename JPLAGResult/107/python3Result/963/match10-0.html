class Solution:
    def threeEqualParts(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        A = ''.join(map(str, A))
        n = len(A)
        count = 0
        for num in A:
            count += (num == '1')
        if count % 3:
            return [-1,-1]
        ones = count // 3
        cur = 0
        for i in range(n-1, -1, -1):
            if A[i] == '1':
                cur += 1
            if cur == ones:
                break
        pos = i
        target = int(A[pos:], 2)
        i, cur = 0, 0
        while i < pos:
            cur = cur*2 + int(A[i])
            i += 1
            if cur > target:
                return [-1, -1]
            elif cur == target:
                break
        first = i - 1
        cur = 0
        while i < pos:
            cur = cur*2 + int(A[i])
            i += 1
            if cur > target:
                return [-1, -1]
            elif cur == target:
                break
        if cur < target:
            return [-1,-1]
        return [first, i]
