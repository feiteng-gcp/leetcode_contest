class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if name == typed:
            return True
        elif not name:
            return False
        i, j = 0, 0
        while i < len(name) and j < len(typed):
            if name[i] == typed[j]:
                i += 1
                j += 1
            elif i > 0 and name[i-1] == typed[j]:
                j += 1
            else:
                return False
        else:
            if i == len(name):
                while j < len(typed) and name[-1] == typed[j]:
                    j += 1
                else:
                    return j == len(typed)
            else:
                return False
