class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        
        def phase(s):
            ans = [[None, 0]]
            for ch in s:
                if ch == ans[-1][0]:
                    ans[-1][1] += 1
                else:
                    ans.append([ch, 1])
            return ans
        a, b = phase(name), phase(typed)
        
if len(a) != len(b):
            return False
        for x, y in zip(a, b):
            if x[0] != y[0] or x[1] > y[1]:
                return False
        return True
