class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        m = len(name)
        n = len(typed)
        i = 0
        j = 0
        while i < m and j < n:
            if name[i] == typed[j]:
                i += 1
                j += 1
            else:
                if i > 0 and typed[j] == name[i-1]:
                    j += 1
                else:
                    return False
        if i < m:
            return False
        while j < n:
            if name[-1] == typed[j]:
                j += 1
            else:
                return False
        return True
