class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        i,j = 0,0
        while i<len(name) and j<len(typed):
            ii = i
            while ii<len(name) and name[ii]==name[i]: ii += 1
            jj = j
            while jj<len(typed) and typed[jj]==typed[j]: jj += 1
            if name[i]!=typed[j] or jj-j<ii-i: return False
            i,j = ii,jj
        if i<len(name) or j<len(typed): return False
        return True
