class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if name == typed: return True
        N1, N2 = len(name), len(typed)
        if N1 > N2: return False
        i1 = i2 = 0
        while i1 < N1 and i2 < N2:
            
if name[i1] == typed[i2]:
                i1 += 1
                i2 += 1
            elif i2 > 0 and typed[i2] == typed[i2 - 1]:
                i2 += 1
            else:
                return False
            
        # print(i1, i2, N2 - 1)
        if i1 == N1 and i2 <= N2 - 1:
            c = typed[i2 - 1]
            # print(c)
            while i2 < N2:
                # print(c, typed[i2], i2)
                if typed[i2] != c:
                    return False
                i2 += 1
        if i1 <= N1 - 1 and i2 == N2:
            return False
        
            
        return True
