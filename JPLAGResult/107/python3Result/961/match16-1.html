class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        
        
        alex
        i
        aaleex
        j
        
        """
        def shrink(name):
            outName = []
            outCnt = []
            for c in name:
                if not outName or outName[-1] != c:
                    outName.append(c)
                    outCnt.append(1)
                else:
                    outCnt[-1] += 1
            return outName, outCnt
        nameS, nameC = shrink(name)
        typeS, typeC = shrink(typed)
        if nameS != typeS:
            
return False
        for i in range(len(nameC)):
            if nameC[i] > typeC[i]:
                return False
        return True
