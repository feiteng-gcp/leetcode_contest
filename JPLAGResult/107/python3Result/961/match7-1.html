class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if name == "":
            return typed == ""
        if typed == "":
            return False
        def run(s):
            result = [[s[0], 1]]
            last = s[0]
            for i in range(1, len(s)):
                if last == s[i]:
                    result[-1][1] += 1
                else:
                    result.append([s[i], 1])
                    last = s
[i]
            return result
        nrun = run(name)
        trun = run(typed)
#        print(nrun, trun)
        if len(nrun) != len(trun):
            return False
        return all(n[0] == t[0] and n[1] <= t[1] for n, t in zip(nrun, trun))
