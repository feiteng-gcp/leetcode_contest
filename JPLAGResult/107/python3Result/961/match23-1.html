class Solution:
    def uniq(self, s):
        v = []
        c = []
        for e in s:
            if len(v) == 0:
                v.append(e)
                c.append(1)
            elif e == v[-1]:
                c[-1] += 1
            else:
                v.append(e)
                c.append(1)
        return v, c

    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        v1, c1 = self.uniq(name)
        v2, c2 = self.uniq(typed)

        # print(v1, c1)
        # print(v2, c2)


        if len(v1) != len(v2):
            return False
        
if any(p != q for p, q in zip(v1, v2)):
            return False

        for i in range(len(v1)):
            if c1[i] > c2[i]:
                return False
        return True
