class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        
        ch1 = []
        ch2 = []
        
        pre = None
        cnt = 0
        
        for c in name:
            if pre is None or pre != c:
                if pre:
                    ch1.append((pre, cnt))
                cnt = 1
                pre = c
            else:
                cnt += 1
        ch1.append((pre, cnt))
        
        pre = None
        cnt = 0
        
        for c in typed:
            if pre is None or pre != c:
                if pre:
                    ch2.append((pre, cnt))
                cnt = 1
                pre = c
            else:
                cnt += 1
        ch2.append((pre, cnt))
        if len(ch1) != len(ch2):
            return False
        
        for p1, p2 in zip(ch1, ch2):
            if p1[0] != p2[0] or p1[1] > p2[1]:
                return False
        return True
