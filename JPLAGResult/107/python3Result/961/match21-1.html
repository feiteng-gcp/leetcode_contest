class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        i, j = 0, 0
        while i<len(name) and j<len(typed):
            ci, cj = 1, 1
            while i+1<len(name) and name[i] == name[i+1]:
                i += 1
                ci += 1
            while j+1<len(typed) and typed[j] == typed[j+1]:
                j += 1
                cj += 1
            if ci > cj:
                return False
            i += 1
            j += 1
        return i == len(name) and j == len(typed)
