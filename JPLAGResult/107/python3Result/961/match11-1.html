class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """

        ii = len(name)
        jj = len(typed)

        i = 0
        j = 0
        while i <= ii and j < jj:
            if i < ii and name[i] == typed[j]:
                i += 1
                j += 1
            elif i > 0 and name[i-1] == typed[j]:
                j += 1
            else:
                break

        return i == ii and j == jj
