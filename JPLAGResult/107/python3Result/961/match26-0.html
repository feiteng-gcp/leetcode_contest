class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        def to_list(name):
            last = "#"
            count = 0
            result = []
            for c in name:
                if c == last:
                    count += 1
                else:
                    result.append((last, count))
                    count = 1
                    last = c
            result.append((last, count))
            return result
        n_list = to_list(name)
        t_list = to_list(typed)
        #print(n_list)
        #print(t_list)
        if len(n_list) == len(t_list):
            for i in range(len(n_list)):
                n = n_list[i]
                t = t_list[i]
                if n[0] == t[0] and n[1] <= t[1]:
                    continue
                else:
                    return False
            return True
        else:
            return False
