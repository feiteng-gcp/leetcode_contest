class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        ptr_name, ptr_typed = 0, 0
        while ptr_name < len(name) and ptr_typed < len(typed):
            if name[ptr_name] == typed[ptr_typed]:
                ptr_name += 1
                ptr_typed += 1
            elif ptr_name > 0 and name[ptr_name - 1] == typed
[ptr_typed]:
                ptr_typed += 1
            else:
                return False
        if ptr_name != len(name):
            return False
        while ptr_typed < len(typed):
            if typed[ptr_typed] != name[-1]:
                return False
            ptr_typed += 1
        return True
