class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if len(name) > len(typed):
            return False
        chars1, count1, chars2, count2 = [], [], [], []
        pre = None
        count = 0
        for char in name:
            if char == pre:
                count += 1
            else:
                chars1.append(pre)
                count1.append(count)
                count = 1
            pre = char
        chars1.append(pre)
        count1.append(count)
        pre = None
        count = 0
        for char in typed:
            if char == pre:
                count += 1
            else:
                chars2.append(pre)
                count2.append(count)
                count = 1
            pre = char
        chars2.append(pre)
        count2.append(count)
        
        if chars1 != chars2:
            
return False
        for i in range(len(count1)):
            if count1[i] > count2[i]:
                return False
        return True
