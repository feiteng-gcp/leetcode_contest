class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        from itertools import groupby
        K1=[]
        L1=[]
        K2=[]
        L2=[]
        for k,g in groupby(list(name)):
            K1.append(k)
            L1.append(len(list(g)))
        for k,g in groupby(list(typed)):
            K2.append(k)
            L2.append(len(list(g)))
        if K1!=K2:
            return False
        for i in range(len(L1)):
            if L1[i]>L2[i]:
                return False
        return True
