class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if len(name) > len(typed):
            return False
        i, j, lastname = 0, 0, ''
        while i < len(name) and j < len(typed):
            if name[i] == typed[j]:
                lastname = name[i]
                i += 1
                j += 1
            elif typed[j] == lastname:
                j += 1
            else:
                return False
        if i < len(name):
            return False
        while j < len(typed):
            if typed[j] == lastname:
                j += 1
            else:
                return False
        return True
