class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if len(name)>len(typed):
            return False
        i=0
        j=0
        curr_n=None
        curr_t
=None
        while i<len(name) and j<len(typed):
            curr_n=i
            while i+1<len(name) and name[i+1]==name[i]:
                i+=1
            count_n=i-curr_n+1
            
            curr_t=j
            while j+1<len(typed) and typed[j+1]==typed[j]:
                j+=1
            count_t=j-curr_t+1
            
            if name[curr_n]!=typed[curr_t] or count_n>count_t:
                return False
            i+=1
            j+=1
        return i==len(name) and j==len(typed)
