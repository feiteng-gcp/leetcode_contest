class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        
        lis1 = [0]
        lis2 = [0]
        for i in range(len(name)):
            if i == 0:
                lis1[-1] = 1
            elif name[i] == name[i-1]:
                lis1[-1] += 1
            else:
                lis1.append(1)
        for i in range(len(typed)):
            if i == 0:
                lis2[-1] = 1
            elif typed[i] == typed[i-1]:
                lis2[-1] += 1
            else:
                lis2.append(1)
        if len(lis1) != len(lis2):
            return False
        for i in range(len(lis1)):
            if lis1[i] > lis2[i]:
                return False
        return True
