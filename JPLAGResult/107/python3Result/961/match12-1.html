class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if name[0] != typed[0] or len(name) > len(typed):
            return False
        idx_true = 1
        idx_typed 
= 1
        while idx_true < len(name) and idx_typed < len(typed):
            if name[idx_true] == typed[idx_typed]:
                idx_true += 1
                idx_typed += 1
            elif name[idx_true - 1] == typed[idx_typed]:
                idx_typed += 1
            else:
                return False

        return idx_true >= len(name)
