class Solution:
    def isLongPressedName(self, name, typed):
        
        def f(s):
            ans = []
            for c in s:
                if ans and ans[-1][0]==c:
                    ans[-1][1] += 1
                else:
                    ans.append([c, 1])
            return ans
        
        st1 = f(name)
        st2 = f(typed)
        
        if len(st1)!=len(st2): return False
        for i in range(len(st1)):
            if st1[i][1]>st2[i][1]: return False
        return True
