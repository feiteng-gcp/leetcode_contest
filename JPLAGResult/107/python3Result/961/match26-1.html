class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        def collect(s):
            ret = []

            old = None
            count = 0
            for c in s:
                if c == old:
                    count += 1
                else:
                    if old != None:
                        ret.append((old, count))
                    old 
= c
                    count = 1
            ret.append((old, count))

            return ret

        collect_name = collect(name)
        collect_typed = collect(typed)

        if len(collect_name) != len(collect_typed): return False

        N = len(collect_name)
        for i in range(N):
            if collect_name[i][0] != collect_typed[i][0]: return False
            if collect_name[i][1] > collect_typed[i][1]: return False
        return True
