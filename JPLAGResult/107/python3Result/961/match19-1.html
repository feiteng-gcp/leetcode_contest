class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        i, j = 0, 0
        while i < len(name) and j < len(typed):
            if i == len(name) and typed[j] != name[-1]:
                return False
            
if name[i] == typed[j]:
                i += 1
                j += 1
            elif i > 0 and name[i-1] == typed[j]:
                j += 1
            else:
                return False
        while j < len(typed):
            if typed[j] != name[-1]: return False
            j += 1
        return i == len(name)
