class Solution:
    def isLongPressedName(self, name, typed):
        """
        :type name: str
        :type typed: str
        :rtype: bool
        """
        if len(typed) < len(name): return False
        
        j = 0
        for i in range(len(name)):
            if j >= len(typed): return False
            if name[i] == typed[j]:
                j += 1
            elif j>0 and typed[j]==typed[j-1]:
                while j<len(typed)-1 and typed[j]==typed[j-1]:
                    j += 1
                if name[i] == typed[j]:
                    j += 1
                else: return False
            else: return False
            
        if j!= len(typed):
            for c in typed[j:]:
                if c != typed[j-1]:
                    return False
        return True
