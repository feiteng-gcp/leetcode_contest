class Solution {
    public int numSteps(String s) {
        // remove
        
int step = 0;
        while (!s.equals("1")) {
            if (s.charAt(s.length() - 1) == '1') {
                // + 1
                s = addOne(s);
            } else {
                // remove 0
                s = removeZero(s);
            }
            step++;
        }
        return step;
    }
    
    private String addOne(String s) {
        // find the zero
        Integer pos = null;
        for (int i = s.length() - 1; i >= 0; i--) {
            if (s.charAt(i) == '0') {
                pos = i;
                break;
            }
        }
        StringBuilder sb = new StringBuilder();
        if (pos == null) {
            sb.append("1");
            for (int i = 0; i < s.length(); i++) {
                sb.append("0");
            }
        } else {
            // has pos
            sb.append(s.substring(0, pos));
            sb.append("1");
            for (int i = pos+1; i < s.length(); i++) {
                sb.append("0");
            }
        }
        return sb.toString();
    }
    
    private String removeZero(String s) {
        return s.substring(0, s.length() - 1);
    }
}
