import java.math.BigInteger;

class Solution {
    public int numSteps(String s) {
        BigInteger b = new BigInteger(s, 2);
        int ss = 0;
        
BigInteger two = new BigInteger("2");
        BigInteger one = BigInteger.ONE;
        while (!b.equals(one)) {
            if (b.getLowestSetBit() == 0) {
                b = b.add(one);
            }// odd
            else {
                b = b.divide(two);
            }
            ss++;
        }
        return ss;
    }
}
