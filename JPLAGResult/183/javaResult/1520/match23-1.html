class Solution {
    public int numSteps(String s) {
        char[] chs = s.toCharArray();
        int id = chs.length - 1;
        int res = 0;
        while(id > 0) {
            char x = chs[id];
            
if(x == '0') {
                ++res;
                --id;
            } else {
                while(id >= 0 && chs[id] == '1') {
                    ++res;
                    --id;
                }
                ++res;
                if(id >= 0) chs[id] = '1'
;
            }
        }
        return res;
    }
}
