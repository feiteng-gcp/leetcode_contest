import java.math.BigInteger;
class Solution {
    public int numSteps(String s) {
        int n = s.length(), res = 0;
        BigInteger num = new BigInteger("0"), base = new BigInteger("1");
        BigInteger one = new BigInteger("1"), two = new BigInteger("2");
        for (int i = n - 1; i >= 0; i--) {
            char c = s.charAt(i);
            BigInteger cur = new BigInteger("" + c);
            num = num.add(cur.multiply(base));
            base = base.multiply(two);
        }
        
        
        //System.out.println(one.equals(new BigInteger("1")));
        
while (!num.equals(one)) {
            res++;
            if (num.mod(two).equals(one)) {
                num = num.add(one);
            } else {
                num = num.divide(two);
            }
            //System.out.println(num);
        }
        return res;
    }
}
