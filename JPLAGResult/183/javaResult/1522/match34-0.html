class Solution {

	
	private int[] dp;	
	
    public String stoneGameIII(int[] stoneValue) {
    	int i , j , n = stoneValue.length;
    	dp = new int[n];
    	Arrays.fill(dp , Integer.MIN_VALUE);
    	int ans = dfs(stoneValue , 0 , n);
    	if (ans == 0) {
    		return "Tie";
    	} else if (ans > 0) {
    		return "Alice";
    	} else {
    		return "Bob";
    	}
    }
    
    private int dfs(int[] stoneValue , int current , int n) {
    	if (current == n) {
    		return 0;
    	} else {
    		if (dp[current] != Integer.MIN_VALUE) {
    			return dp[current];
    		} else {
    			int ans = Integer.MIN_VALUE , sum = 0;
    			for (int cnt = 1;cnt <= 3 && current + cnt <= n;cnt ++) {
    				sum += stoneValue[current + cnt - 1];
    				int result = - dfs(stoneValue , current + cnt , n) + sum;
    				ans = Math.max(ans , result);    				
    			}
    			return dp[current] = ans;    			
    		}    		
    	}    	
    }
}
