class Solution {
    public String stoneGameIII(int[] stone) {
        int n = stone.length;
        
int[] s = new int[n + 2];
        for(int i = 0; i < stone.length; i++) {
            s[i] = stone[i];
        }
        
        int[] dp = new int[n + 3];
        for(int i = n - 1; i >= 0;i--) {
            dp[i] = Math.max(s[i] - dp[i + 1], Math.max(s[i] + s[i + 1] - dp[i + 2], s[i] + s[i + 1] + s[i + 2] - dp[i + 3]));
        }
        
        
if(dp[0] > 0) {
            return "Alice";
        } else if(dp[0] < 0) {
            return "Bob";
        } else {
            return "Tie";
        }
    }
}
