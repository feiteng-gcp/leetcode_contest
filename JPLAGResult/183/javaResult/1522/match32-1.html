class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        int[] dp = new int[n];
        boolean[] visit = new boolean[n];
        
        helper(stoneValue, dp, visit, 0);
        
        if (dp[0] > 0) return "Alice";
        else if (dp[0] < 0) return "Bob";
        else return "Tie"
;
    }
    private int helper(int[] stoneValue, int[] dp, boolean[] visit, int pos) {
        if (pos >= stoneValue.length) return 0;
        if (visit[pos]) return dp[pos]
;
        int res = stoneValue[pos] - helper(stoneValue, dp, visit, pos+1);
        if (pos+1 < stoneValue.length) {
            res = Math.max(res, stoneValue[pos] + stoneValue[pos+1] - helper(stoneValue, dp, visit, pos+2));
        }
        if (pos+2 < stoneValue.length) {
            res = Math.max(res, stoneValue[pos] + stoneValue[pos+1] + stoneValue[pos+2] - helper(stoneValue, dp, visit, pos+3));
        }
        visit[pos] = true;
        dp[pos] = res;
        return res;
    }
}
