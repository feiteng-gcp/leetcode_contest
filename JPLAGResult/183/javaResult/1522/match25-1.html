class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        // f[i] 表示 i ~ n-1 的棋子, 先手能拿到的最大和
        int[] sum = new int[n + 1];
        int[] f = new int[n + 1];
        Arrays.fill(f, Integer.MIN_VALUE);

        f[n] = 0;
        f[n - 1] = stoneValue[n - 1];
        sum[n - 1] = stoneValue[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            sum[i] = stoneValue[i] + sum[i + 1];
            for (int j = 1; j <= Math.min(3, n - i); j++) {
                f[i] = Math.max(f[i], sum[i] - sum[i + j] + sum[i + j] - f[i + j]);
            }
        
}

        if (f[0] * 2 == sum[0]) {
            return "Tie";
        } else if (f[0] * 2 > sum[0]) {
            return "Alice";
        } else {
            return "Bob";
        }
    }
}
