class Solution {
   public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        if(n==1)
{
            if (stoneValue[0] > 0) {
                return "Alice";
            }
            if (stoneValue[0] < 0) {
                return "Bob";
            }
            return "Tie";
        }

        int[] ans = new int[n + 1];
        ans[n - 1] = stoneValue[n - 1];
        ans[n - 2] = Math.max(stoneValue[n - 2] - ans[n - 1], stoneValue[n - 2] + stoneValue[n - 1]);

        for (int i = n - 3; i >= 0; i--) {
            ans[i] = Math.max(stoneValue[i] - ans[i + 1], stoneValue[i] + stoneValue[i + 1] - ans[i + 2]);
            ans[i] = Math.max(ans[i], stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] - ans[i + 3]);
        }

        
if (ans[0] > 0) {
            return "Alice";
        }
        if (ans[0] < 0) {
            return "Bob";
        }
        return "Tie";
    }
}
