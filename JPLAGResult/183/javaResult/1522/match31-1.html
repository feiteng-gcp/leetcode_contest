class Solution {
    int[][] memo;
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        memo = new int[n][2];
        int[] postSum = new int[n+1];
        for(int i=n-1; i>=0; i--){
            postSum[i] = stoneValue[i]+postSum[i+1];
        }
        int a = calc(stoneValue, 0, 0, postSum);
        int b = postSum[0]-a;
        // System.out.println(a+","+(postSum[0]-a));
        if(a==b){
            return "Tie";
        }
        if(a<b){
            return "Bob";
        }
        return "Alice";
    }
    
    
    public int calc(int[] s, 
int cur, int p, int[] postSum){
        int n = s.length;
        if(cur>=n){
            return 0;
        }
        if(memo[cur][p]!=0){
            return memo[cur][p];
        }
        int max = Integer.MIN_VALUE;
        for(int i=0; i<3; i++){
            if(cur+i>=n){
                break;
            }
            
max = Math.max(max, postSum[cur]-calc(s, cur+i+1, (p+1)%2, postSum));
        }
        memo[cur][p] = max;
        return max;
    }
}
