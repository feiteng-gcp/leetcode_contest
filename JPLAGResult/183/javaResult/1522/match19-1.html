class Solution {
    private int[] dp;
    public String stoneGameIII(int[] stoneValue) {
        dp = new int[stoneValue.length];
        Arrays.fill(dp, -1);
        int g = helper(stoneValue, 0);
        return g > 0 ? "Alice" : g < 0 ? "Bob" : "Tie";
    }
    
    private int helper(
int[] stoneValue, int i) {
        if (i == dp.length) return 0;
        if (dp[i] != -1) return dp[i];
        int res = stoneValue[i] - helper(stoneValue, i + 1);
        if (i + 1 < stoneValue.length) {
            res = Math.max(res, stoneValue[i] + stoneValue[i + 1] - helper(stoneValue, i + 2));
        }
        if (i + 2 < stoneValue.length) {
            res = Math.max(res, stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] - helper(stoneValue, i + 3));
        }
        dp[i] = res;
        return res;
    }
}
