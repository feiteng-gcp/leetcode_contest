class Solution {
    Map<Integer, Integer> mem;
    
    public String stoneGameIII(int[] stoneValue) {
        mem = new HashMap<>();
        
        int bestDiff = bestDifference(stoneValue, 0);

        //System.out.println(mem);
        //System.out.println(bestDiff);

        if(bestDiff > 0) {
            return "Alice";
        }
        
        if(bestDiff < 0) {
            return "Bob";
        }
        
        return "Tie";
    }
        
    private int bestDifference(int[] stoneValue, int start) {
        if(start == stoneValue.length) {
            return 0;
        }
        
        if(start == stoneValue.length - 1) {
            return stoneValue[start];
        }
        
        if(mem.containsKey(start)) {
            return mem.get(start);
        }
        
        int out = -2000;
        
        int sum = 0;
        
        for(int i = start; i <= Math.min(start + 2, stoneValue.length - 1); i++) {
            sum += stoneValue[i];
            out = Math.max(out, sum - bestDifference(stoneValue, i+1));
        }
        
        //System.out.println(start + " " + out);
        
        mem.put(start, out);
        return out;
    }
}
