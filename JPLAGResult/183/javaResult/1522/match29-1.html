class Solution {
    int sum = 0;
    int[] dp;
    boolean[] viewed;
    int[] pre;
    public String stoneGameIII(int[] arr) {
        for(int num: arr) sum += num;
        dp = new int[arr.length];
        viewed = new boolean[arr.length];
        
pre = new int[arr.length];
        pre[arr.length-1] = arr[arr.length-1];
        for(int i=arr.length-2; i>=0; i--) {
            pre[i] = arr[i] + pre[i+1];
        }
        int max = helper(arr, 0);
        if(max > sum-max) return "Alice";
        else if(max == sum-max) return "Tie";
        return "Bob";
    }
    
    
private int helper(int[] arr, int idx) {
        if(idx >= arr.length) return 0;
        if(viewed[idx]) return dp[idx];
        int max = Integer.MIN_VALUE;
        int cur = 0;
        for(int i=0; i<3 && idx+i<arr.length; i++) {
            cur += arr[idx+i];
            int next = (idx+i+1 >= arr.length ? 0: pre[idx+i+1]-helper(arr, idx+i+1));
            max = Math.max(max, cur+next);
        }
        viewed[idx] = true;
        dp[idx] = max;
        return max;
    }
}
