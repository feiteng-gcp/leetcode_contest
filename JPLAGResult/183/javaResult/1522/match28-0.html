/*
1. def: int[] dp, dp[i] := max value diff to get when starting at index i
2. base case: dp[len] = 0
3. induction rule: dp[i] = max(v[i : k] - dp[k + 1]) for k = i, i + 1, i + 2
*/
class Solution {
    public String stoneGameIII(int[] val) {
        int len = val.length;
        int[] dp = new int[len + 1];
        Arrays.fill(dp, Integer.MIN_VALUE);
        dp[len] = 0;
        for (int i = len - 1; i >= 0; i--) {
            int cur = 0;
            for (int k = i; k <= Math.min(i + 2, len - 1); k++) {
                cur += val[k];
                dp[i] = Math.max(dp[i], cur - dp[k + 1]);
            }
        }
        if (dp[0] == 0) {
            return "Tie";
        }
        return dp[0] > 0 ? "Alice" : "Bob";
    }
}
