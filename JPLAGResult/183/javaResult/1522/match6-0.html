class Solution {
    public String stoneGameIII(int[] piles) {
        int[] sum = new int[piles.length];
        int accum = 0;
        for (int i = piles.length - 1; i >= 0; i--) {
            accum += piles[i];
            sum[i] = accum;
        }
        int[][] dp = new int[piles.length][4];
        int ret = dfs(piles, 0, 3, dp, sum);
        //System.out.println("ret = " + ret);
        if (ret * 2 < accum) {
            return "Bob";
        } else if (ret * 2 > accum) {
            return "Alice";
        } else {
            return "Tie";
        }
    }

    private int dfs(int[] piles, int s, int m, int[][] dp, int[] sum) {
        if (s >= piles.length) {
            return 0;
        }
        /*
        if (s + m >= piles.length) {
            return sum[s];
        }
        */
        if (dp[s][m] != 0) {
            return dp[s][m];
        }
        int res = Integer.MIN_VALUE;
        for (int i = s; i < s + m; i++) {
            res = Math.max(res, sum[s] - dfs(piles, i + 1, 3, dp, sum));
        }
        dp[s][m] = res;
        return res;
    }
}
