class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int l = stoneValue.length;

        
        int[] dp = new int[l];
        Arrays.fill(dp, Integer.MIN_VALUE);
        int val = helper(stoneValue, 0, dp);
        if (val == 0) {
            return "Tie";
        
}
        return val > 0 ? "Alice" : "Bob";
        
    }
    
    private int helper(int[] nums, int index, int[] dp) {
        int l = nums.length;
        if (index >= l) {
            return 0;
        }
        if (dp[index] != Integer.MIN_VALUE) {
            return dp[index];
        
}
        int res = Integer.MIN_VALUE, sum = 0;
        for (int i = index; i < Math.min(l, index + 3); i++) {
            sum += nums[i];
            res = Math.max(res, sum - helper(nums, i + 1, dp));
        }
        dp[index] = res;
        return res;
    }
}
