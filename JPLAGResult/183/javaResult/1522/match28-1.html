class Solution {
   public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        int[] dp = new int[n + 1];
        Arrays.fill(dp, -1000 * 50001);
        int[] sum = new int[n + 1];
        for (int i = n - 1; i >= 0; i--) {
            sum[i] = sum[i + 1] + stoneValue[i];
        }
        dp[n] = 0;
        for (int i = n - 1; i >= 0; i--) {
            for (int k = 1; k <= 3 && i + k <= n; k++) 
{
                dp[i] = Math.max(dp[i], sum[i] - dp[i + k]);
            }
        }
        if (dp[0] *2 == sum[0]) {
            return "Tie";
        }
        return dp[0] * 2 > sum[0] ? "Alice" : "Bob";
    }
}
