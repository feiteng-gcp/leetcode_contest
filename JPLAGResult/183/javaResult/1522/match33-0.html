class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        int[] res = new int[stoneValue.length+1];
        res[n] = 0;

        for(int i = n-1; i>= 0; i--){

            int use = 1;
            int max = Integer.MIN_VALUE;
            while(i+use <= n && use<=3){
                int uc = 0;
                for(int j = 0; j < use; j++){
                    uc+= stoneValue[i+j];
                }
                max = Math.max(max,uc- res[i+use]);
                use++;
            }

            res[i] = max;
           // System.out.println("i:" + i + "max:" + max );

        }



            if(res[0] == 0){
                return "Tie";
            }else if(res[0] > 0){
                return "Alice";
            }else{
                return "Bob";
            }

    }
}
