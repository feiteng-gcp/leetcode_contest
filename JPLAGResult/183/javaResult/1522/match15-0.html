class Solution {
    
    Integer[] M;
    
    public String stoneGameIII(int[] stoneValue) {
        int N = stoneValue.length;
        int[] prefixSum = new int[N];
        M = new Integer[N];
        
        for (int i = 0; i < N; i++) {
            prefixSum[i] = (i > 0 ? prefixSum[i - 1] : 0) + stoneValue[i];
        }

        int total = prefixSum[N - 1];
        int gain = dp(prefixSum, 0);

        if (total - gain == gain) return "Tie";
        else if (gain > total - gain) return "Alice";
        else return "Bob";
    }
    
    int dp(int[] sums, int i) {
        int N = sums.length;
        if (i >= N) return 0;
        if (M[i] != null) return M[i];
        
        int max = Integer.MIN_VALUE;
        for (int j = 0; j < 3 && i + j < N; j++) {
            int gain = sums[i + j] - (i > 0 ? sums[i - 1] : 0);
            int enemyGain = dp(sums, i + j + 1);
            int meGain = sums[N - 1] - sums[i + j + 1 - 1] - enemyGain;
            max = Math.max(max, gain + meGain);
        }
        
        M[i] = max;
        return max;
    }
}
