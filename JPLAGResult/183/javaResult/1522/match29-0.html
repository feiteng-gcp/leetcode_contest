class Solution {
    
    int dp[] = new int[50001];
    
    int mx(int i,int[] s){
        if(i==s.length) return 0;
        if(dp[i]!=0) return dp[i];
        int sum = 0;
        int r =-10000000;
        for(int j=i;j<i+3&&j<s.length;++j){
            sum += s[j];
            int bg = mx(j+1,s);
            int mx = sum+suf[j+1]-bg;
            r = Math.max(r,mx);
        }
        return dp[i] = r;
    }
        
        
    int t = 0;
    int suf[];
    public String stoneGameIII(int[] s) {
        suf = new int[s.length+1];
        suf[s.length-1] =s[s.length-1];
        for(int i=s.length-2;i>=0;--i){
            suf[i] = suf[i+1]+ s[i];
        }
        int na = mx(0,s);
       // System.out.println(suf[0]);
        if(na>suf[0]-na){
            return "Alice";
        }else if(na==suf[0]-na){
            return "Tie";
        }
        return "Bob";
    }
}
