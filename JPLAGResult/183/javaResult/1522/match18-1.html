class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int sum = 0;
        
int[] sums = new int[stoneValue.length];
        for (int i = 0; i < stoneValue.length; i++) {
            sum += stoneValue[i];
            sums[i] = sum;
        }
        Integer[] dp = new Integer[stoneValue.length];
        int aBest = dfs(0, stoneValue, dp, sums);
        // for (int i : dp) System.out.print(i + ",");
        if (aBest > sum - aBest) return "Alice";
        else if (aBest < sum - aBest) return "Bob";
        return "Tie";
    
}
    
    private int dfs(int cur, int[] stoneValue, Integer[] memo, int[] sums) {
        if (cur >= stoneValue.length) return 0;
        
        if (memo[cur] != null) return memo[cur];
        int best = Integer.MIN_VALUE, n = stoneValue.length;
        // take 1
        best = Math.max(best, stoneValue[cur] + sums[n - 1] - sums[cur] - 
dfs(cur + 1, stoneValue, memo, sums));
        // take 2
        if (n - cur >= 2) {
            best = Math.max(best, stoneValue[cur] + stoneValue[cur + 1] + 
                            sums[n - 1] - sums[cur + 1] - dfs(cur + 2, stoneValue, memo, sums));
        }
        
        if (n - cur >= 3) {
            best = Math.max(best, stoneValue[cur] + stoneValue[cur + 1] + stoneValue[cur + 2] +
                            sums[n - 1] - sums[cur + 2] - dfs(cur + 3, stoneValue, memo, sums));
        }
        
        memo[cur] = best;
        return best;
    }
}
