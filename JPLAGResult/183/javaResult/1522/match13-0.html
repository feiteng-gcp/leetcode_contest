class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        int[] preSum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            preSum[i + 1] = preSum[i] + stoneValue[i];
        }
        int[][] dp = new int[n + 3][2];
        for (int s = n - 1; s >= 0; s--) {
            for (int k = 0; k < 2; k++) {
                dp[s][k] = Integer.MIN_VALUE; 
                for (int e = s + 1; e <= s + 3; e++) {
                    dp[s][k] = Math.max(dp[s][k], -dp[e][1 - k] + preSum[Math.min(e, n)] - preSum[s]);
                }
            }
        }
        if (dp[0][0] == 0) return "Tie";
        return dp[0][0] > 0 ? "Alice" : "Bob";
    }
}
