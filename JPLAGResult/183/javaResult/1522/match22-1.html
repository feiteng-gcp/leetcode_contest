class Solution {
    Map<Integer, Integer> map = new HashMap<>();
    public String stoneGameIII(int[] A) {
        int res = helper(A, 0);
        
if (res > 0) {
            return "Alice";
        } else if (res == 0) {
            return "Tie";
        } else {
            return "Bob";
        }
    }
    
    private int helper(int[] A, int index) 
{
        if (index >= A.length) {
            return 0;
        }
        if (map.containsKey(index)) {
            return map.get(index);
        }
        int cur = 0;
        int max = Integer.MIN_VALUE;
        cur += A[index];
        int m1 = cur - helper(A, index + 1);
        max = Math.max(max, m1);
        if (index + 1 < A.length) {
            cur += A[index + 1];
            int m2 = cur - helper(A, index + 2);
            max = Math.max(max, m2);
        }
        if (index + 2 < A.length) {
            cur += A[index + 2];
            int m3 = cur - helper(A, index + 3);
            max = Math.max(max, m3);
        }
        map.put(index, max);
        return max;
    }
}
