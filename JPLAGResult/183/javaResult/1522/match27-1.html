class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int N = stoneValue.length;
        
        int[] values = new int[N+3];
        for(int i = 0; i < N; i++)
            values[i] = stoneValue[i];
        
        long[] arr = new long[N+3];
        
        for(int i = N-1; i >= 0; i--) {
            long valA = values[i];
            
            long scenarioA = arr[i+1];
            
            long valB = valA + values[i+1];
            long scenarioB = arr[i+2];
            
            long valC = valB + values[i+2];
            long scenarioC = arr[i+3];
            
            long diffA = valA - scenarioA;
            long diffB = valB - scenarioB;
            long diffC = valC - scenarioC;
            
            long max = Math.max(diffA, Math.max(diffB, diffC));
            
            arr[i] = max;
        
}
        
        if(arr[0] == 0) {
            return "Tie";
        }
        else if(arr[0] > 0) {
            return "Alice";
        }
        
        return "Bob";
        
    }
}
