class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int total = 0;
        for (int i : nums) {
            total += i;
        }
        List<Integer> res = new ArrayList<>();
        double target = (total * 1.0) / (2 * 1.0);
        int temp = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            temp = temp + nums[i];
            res.add(nums[i]);
            if (temp > target) {
                return res;
            }
        }
        return res;
    }
}
