class Solution {
    
    int s = 0;
    int n = 0;
    int cur = 10000;
    int tp = 0;
    
    List<Integer> r = null;
    
    void go(int i,int nums[],int sum,LinkedList<Integer> li){
        if(sum>s-sum){
            
            if(li.size()<cur){
                tp = sum;
                cur = li.size();
                r = new ArrayList(li);
            }else if(li.size()==cur&&sum>tp){
                tp = sum;
                cur = li.size();
                r = new ArrayList(li);
            }
            
            return;
        }
        if(i==nums.length) return;
        
        //go(i+1,nums,sum,li);
        li.add(nums[i]);
        go(i+1,nums,sum+nums[i],li);
        li.removeLast();
        
        
        
    }
    
    public List<Integer> minSubsequence(int[] nums) {
        for(int a:nums) s+= a;
        n = nums.length; 
        Arrays.sort(nums);
        for(int i=0,j=n-1;i<j;i++,j--){
            int t = nums[i];
            nums[i] = nums[j];
            nums[j] = t;
        }
        go(0,nums,0,new LinkedList<Integer>());
        return r;
    }
}
