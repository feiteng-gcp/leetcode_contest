	class Solution {
	    public List<Integer> minSubsequence(int[] nums) {
	        Arrays.sort(nums);
	        long s = 0;
	        for(int v : nums){
	        	s += v;
	        }
	        List<Integer> ret = new ArrayList<>();
	        for(int i = nums.length-1;i >= 0;i--){
	        	ret.add(nums[i]);
	        	s -= 2*nums[i];
	        	if(s < 0)break;
	        }
	        return ret;
	    }
	}
