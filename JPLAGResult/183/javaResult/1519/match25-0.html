import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

class Solution {

	public List<Integer> minSubsequence(int[] nums) {
		int copy[] = new int[nums.length], sum = 0, curr = 0;
		for (int i = 0; i < nums.length; i++) {
			sum += nums[i];
			copy[i] = nums[i];
		}
		Arrays.sort(copy);
		ArrayList<Integer> answer = new ArrayList<>();
        
        int i = nums.length;
		for (i--; i >= 0; i--) {
			answer.add(copy[i]);
			curr += copy[i];
			if (curr > sum / 2) {
				break;
			}
		}
		return answer;
	}
}
