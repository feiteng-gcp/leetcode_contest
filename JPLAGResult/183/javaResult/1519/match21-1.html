class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int total_sum = 0;
        for (int i = 0; i < nums.length; i++) total_sum+=nums[i];
        
        List<Integer> list = new ArrayList<>();
        int acc_sum = 0;
        
for (int i = nums.length-1; i >= 0; i--) {
            acc_sum+=nums[i];
            total_sum-=nums[i];
            list.add(nums[i]);
            if (acc_sum > total_sum) {
                return list;
            }
        }
        return list;
    }
}
