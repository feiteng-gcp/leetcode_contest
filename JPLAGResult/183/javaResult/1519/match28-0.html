class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int n=nums.length;
        int total=0;
        for(int i=0;i<n;i++){
            total+=nums[i];
        }
        int sum=0;
        List<Integer> result=new ArrayList<Integer>();
        for(int i=n-1;i>=0;i--){
            sum+=nums[i];
            result.add(nums[i]);
            if(sum>total-sum){
                break;
            }
        }
        return result;
    }
}
