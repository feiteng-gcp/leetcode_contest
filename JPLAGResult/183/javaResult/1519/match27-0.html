class Solution {
    public List<Integer> minSubsequence(int[] nums) {
		List<Integer> ans = new ArrayList<Integer>();

		int sum = 0;
		for (int i = 0; i < nums.length; i++) {
			sum += nums[i];
		}
		Arrays.sort(nums);
		int now = 0;
		for (int i = nums.length - 1; i >= 0; i--) {
			now += nums[i];
			ans.add(nums[i]);

			if (now * 2 > sum) {
				break;
			}
		}
		return ans;
	}
}
