class Solution {
  public List<Integer> minSubsequence(int[] nums) {
    Arrays.sort(nums);
    int sm = 0;
    for(int i=nums.length-1; i>=0; i--) {
      sm += nums[i];
    }
    int sm1 = 0;
    List<Integer> res = new ArrayList<>();
    
for(int i=nums.length-1; i>=0; i--) {
      res.add(nums[i]);
      sm -= nums[i];
      sm1 += nums[i];
      if(sm1 > sm) {
        return res;
      }
    }
    return res;
  }
}
