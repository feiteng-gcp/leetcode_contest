class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        List<Integer> temp = new ArrayList<>();
        // Arrays.sort(nums, (i, j) -> j - i);
        int sum = 0;
        for (int num : nums) {
            sum += num;
            temp.add(num);
        }
        Collections.sort(temp, (i , j) -> j - i);
        List<Integer> result = new ArrayList<>();
        
int cur = 0;
        for (int num : temp) {
            cur += num;
            result.add(num);
            if (cur > sum/2) break;
        }
        return result;
    }
}
