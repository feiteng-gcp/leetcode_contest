class Solution {
    //就是拍序好比较下
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for(int num : nums) sum += num;
        
        List<Integer> res = new ArrayList<>();
        int sum2 = 0;
        for(int i=nums.length-1;i>=0;i--){
            sum2 += nums[i];
            res.add(nums[i]);
            if(sum2 > sum - sum2) break;
        }
        return res;
    }
}
