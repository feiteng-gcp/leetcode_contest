class Solution {
    public String longestDiverseString(int a, int b, int c) {
        ArrayList<Integer> list = new ArrayList<>();
        
        
int[] count = {a, b, c};
        
        boolean valid = true;
        
        while(valid) {
            int max = 0;
            int index = -1;
            
            boolean setQuarantine = false;
            int which = -1;
            if(list.size() >= 2 && list.get(list.size() - 1) == list.get(list.size() - 2)) {
                which = list.get(list.size() - 1);
                count[which] = -count[which];
                setQuarantine = true;
            } 
            
            for(int i = 0; i < 3; i++) {
                
if(count[i] > max) {
                    max = count[i];
                    index = i;
                }
            }
            
            if(max == 0) {
                break;
            }
            
            if(setQuarantine) {
                count[which] = -count[which];
            }
            
            list.add(index);
            count[index]--;
        }
        
        StringBuilder sb = new StringBuilder();
        for(int i : list) {
            char ch = (char) ('a' + i);
            sb.append(ch);
        }
        
        return sb.toString();
    }
}
