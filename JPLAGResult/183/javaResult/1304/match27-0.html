class Solution {
    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb=new StringBuilder();
        int[] count=new int[3];
        count[0]=a;
        count[1]=b;
        count[2]=c;
        int pre=-1;
        int next=0;
        for(int i=0;i<3;i++){
            if(count[i]>count[next]){
                next=i;
            }
        }
        if(count[next]>1){
            count[next]-=2;
            sb.append((char)(next+'a'));
            sb.append((char)(next+'a'));
            pre=next;
        }else{
            count[next]--;
            sb.append((char)(next+'a'));
            pre=next;
        }
        while(true){
            next=(pre+1)%3;
            for(int i=0;i<3;i++){
                if(i!=pre&&count[i]>count[next]){
                    next=i;
                }
            }
            // System.out.println(pre+";;;"+next);
            // System.out.println(sb.toString());
            if(count[next]==0)break;
            
            if(count[next]<count[pre]){
                count[next]--;
                sb.append((char)(next+'a'));
            }else{
                for(int i=0;i<2&&count[next]>0;count[next]--,i++){
                    sb.append((char)(next+'a'));
                }
            }
            
            pre=next;
        }
        return sb.toString();
    }
}
