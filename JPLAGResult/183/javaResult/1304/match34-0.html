class Solution {
    public String longestDiverseString(int a, int b, int c) {
        String s = "g";
        int cc = 1;
        String[] arr = {"a","b", "c"};
        int [] count = {a,b,c};
        String result = "";
        while(true) {
            int max = 0;
            int index = 0;
            for(int i=0;i<3;i++) {
                if(cc>1 && arr[i] == s) {
                    continue;
                }
                if(count[i] > max) {
                    index = i;
                    max = count[i];
                }
            }
            if(max == 0) {
                break;
            }
            if(s == arr[index]) {
                cc++;
            } else {
                cc = 1;
                s = arr[index];
            }
            count[index] -= 1;
            result += arr[index];
        }
        return result;
    }
}
