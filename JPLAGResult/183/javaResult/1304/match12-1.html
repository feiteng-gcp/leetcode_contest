class Solution {
    public String longestDiverseString(int a, int b, int c) {
        
PriorityQueue<int[]> pq = new PriorityQueue<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] a, int[] b) {
                return b[1] - a[1];
            }
        });
        
        if (a != 0) pq.offer(new int[]{0, a});
        if (b != 0) pq.offer(new int[]{1, b});
        if (c != 0) pq.offer(new int[]{2, c})
;
        
        StringBuilder sb = new StringBuilder();
        while (!pq.isEmpty()) {
            int[] curr = pq.poll();
            char currChar = (char)(curr[0] + 'a');
            int l = sb.length();
            if (sb.length() > 1 && sb.charAt(l - 1) == sb.charAt(l - 2) && sb.charAt(l - 1) == currChar) {
                if (pq.isEmpty()) break;
                int[] next = pq.poll();
                char nextChar = (char)(next[0] + 'a');
                sb.append(nextChar);
                if (--next[1] > 0) pq.offer(next);
                pq.offer(curr);
            } else {
                sb.append(currChar);
                if (--curr[1] > 0) pq.offer(curr)
;
            }
        }
        
        return sb.toString();
    }
}
