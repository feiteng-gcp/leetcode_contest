import java.util.*;
import java.io.*;


class Solution {
  public String longestDiverseString(int a, int b, int c) {
    PriorityQueue<Char> pq = new PriorityQueue<>((x, y) -> (y.num - x.num));
    if (a > 0) pq.add(new Char('a', a));
    if (b > 0) pq.add(new Char('b', b));
    if (c > 0) pq.add(new Char('c', c));
      
    // System.out.println(pq);

    StringBuilder sb = new StringBuilder();
    int count = 0;

    
while (!pq.isEmpty()) {
      Char cur = pq.remove();

      // can not add current
      if (sb.length() > 0 && sb.charAt(sb.length() - 1) == cur.c && count == 2) {
        if (pq.isEmpty()) {return sb.toString();}
        else {
          Char next = pq.remove();
          sb.append(next.c);
          next.num--;
          if (next.num > 0) {pq.add(next);}
          count = 1;
          pq.add(cur);
        }
      }

      else {
        sb.append(cur.c);
        cur.num--;
        if (cur.num > 0) {pq.add(cur);}
        count += (sb.length() == 1 || sb.charAt(sb.length() - 2) == cur.c) ? 1 : 0;
      }
    }

    
return sb.toString();




  }


  class Char {
    char c;
    int num;
    Char(char c, int num) {this.c = c; this.num = num;}
    public String toString() {return "(" + c + ", " + num + ")";}
  }
}
