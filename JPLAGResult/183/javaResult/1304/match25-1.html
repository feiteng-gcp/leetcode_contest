class Solution {
    public String longestDiverseString(int a, int b, int c) {
                
PriorityQueue<int[]> queue = new PriorityQueue<>((o1, o2) -> o2[1] - o1[1]);
         if (a > 0) queue.add(new int[]{'a', a});
        if (b > 0) queue.add(new int[]{'b', b});
        if (c > 0) queue.add(new int[]{'c', c});
        StringBuilder builder = new StringBuilder();
        int count = 0;
        Character currentChar = null;
        while (!queue.isEmpty()) {
            int[] first = queue.poll();
            char temp = (char) first[0];
            if (currentChar == null || currentChar.charValue() == temp) {
                if (count == 2) {
                    if (queue.isEmpty()) break;
                    int[] second = queue.poll();
                    char x = (char) second[0];
                    currentChar = x;
                    count = 1;
                    builder.append(x);
                    second[1]--;
                    if (second[1] > 0) queue.add(second);
                    queue.add(first);
                } else {
                    currentChar = temp;
                    builder.append(temp);
                    count++;
                    first[1]--;
                    if (first[1] > 0) queue.add(first);
                }
            } else {
                currentChar = temp;
                count = 1;
                builder.append(temp);
                first[1]--;
                if (first[1] > 0) queue.add(first)
;
            }
        }
        return builder.toString();

    }
}
