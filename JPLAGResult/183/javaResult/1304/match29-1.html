class Solution {
    public String longestDiverseString(int a, int b, int c) {
        Queue<Point> q = new PriorityQueue(Collections.reverseOrder());
        
if (a > 0)q.add(new Point(a, 'a'));
        if(b>0)q.add(new Point(b, 'b'));
        if (c> 0)q.add(new Point(c, 'c'));
        
        StringBuilder sb = new StringBuilder();
        
        char let = ' ';
        int count = 0;
        while (q.size() > 0){
            Point p = q.poll();
            if (p.l == let && count == 2){
                if (q.size() == 0) break;
                else {
                    Point p2 = q.poll();
                    sb.append(p2.l);
                    p2.count--;
                    if (p2.count > 0) q.add(p2);
                    q.add(p);
                    let = p2.l;
                    count = 1;
                }
            }
            else {
                p.count--;
                if (p.count > 0) q.add(p);
                sb.append(p.l);
                if (p.l == let) count++;
                else {
                    let = p.l;
                    count = 1;
                
}
            }
        }
        return sb.toString();
        
    }
}


class Point implements Comparable<Point> {
    int count;
    char l;
    
    public Point(int c, char l){
        count = c;
        this.l = l;
    }
    
    public int compareTo(Point p){
        return this.count - p.count;
    }
}
