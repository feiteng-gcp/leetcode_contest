class Solution {
    public String longestDiverseString(int a, int b, int c) {
         StringBuilder sb =new StringBuilder();
        int last = -1;
        PriorityQueue<int[]> pq =new PriorityQueue<int[]>((x,y)->y[1]-x[1]);
        if(a>0)
        pq.add(new int[]{0,a});
        if(b>0)
        pq.add(new int[]{1,b});
        if(c>0)
        pq.add(new int[]{2,c});
        while (!pq.isEmpty()&&last!=sb.length()){
            int[] m1 = pq.poll();
            char c1 = (char)('a'+m1[0]);
            int n =sb.length();
            last = sb.length();
            if (n<2||(sb.charAt(n-1)!=c1||sb.charAt(n-2)!=c1)){
                sb.append(c1);
                if (m1[1]>1)
                pq.add(new int[]{c1-'a',m1[1]-1});
            }
            else {
                if(pq.isEmpty()) return sb.toString();
                int[] m2 = pq.poll();
                char c2 = (char)('a'+m2[0]);
                sb.append(c2);
                pq.add(m1);
                if (m2[1]>1) pq.add(new int[]{c2-'a',m2[1]-1});
            }
            //System.out.println(sb);
        }
        return sb.toString();
    }
}
