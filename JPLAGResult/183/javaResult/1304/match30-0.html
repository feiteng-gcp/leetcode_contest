class Solution {
    public String longestDiverseString(int a, int b, int c) {
		return slove("", a, b, c);
	}

	public String slove(String now, int a, int b, int c) {
		int n = now.length();
		if (n >= 2 && now.charAt(n - 1) == now.charAt(n - 2)) {
			if (now.charAt(n - 1) == 'a') {
				if (b + c == 0) {
					return now;
				} else {
					if (c >= b) {
						return slove(now + "c", a, b, c - 1);
					} else {
						return slove(now + "b", a, b - 1, c);
					}
				}
			} else if (now.charAt(n - 1) == 'b') {
				if (a + c == 0) {
					return now;
				} else {
					if (c >= a) {
						return slove(now + "c", a, b, c - 1);
					} else {
						return slove(now + "a", a - 1, b, c);
					}
				}
			} else {
				if (a + b == 0) {
					return now;
				} else {
					if (b >= a) {
						return slove(now + "b", a, b - 1, c);
					} else {
						return slove(now + "a", a - 1, b, c);
					}
				}
			}
		}

		if (a + b + c == 0) {
			return now;
		}

		if (a >= b && a >= c) {
			return slove(now + "a", a - 1, b, c);
		}

		if (b >= a && b >= c) {
			return slove(now + "b", a, b - 1, c);
		}

		return slove(now + "c", a, b, c - 1);
	}
}
