class Solution {
    public String longestDiverseString(int a, int b, int c) {
        Pair A = new Pair('a', a);
        Pair B = new Pair('b', b);
        Pair C = new Pair('c', c);
        
        PriorityQueue<Pair> pq = new PriorityQueue<>((x, y) -> {
           return y.num - x.num; 
        });
        if (A.num != 0) {
            pq.offer(A);
        }
        if (B.num != 0) {
            pq.offer(B);
        }
        if (C.num != 0) {
            pq.offer(C);
        }
        // pq.offer(A);
        // pq.offer(B);
        // pq.offer(C);
        StringBuilder sb = new StringBuilder();
        
        char prev = 'x';
        
while (pq.size() > 0) {
            Pair i = pq.poll();
            
            if (pq.size() == 0) {
                if (prev != i.c) {
                    for (int j = 0; j < Math.min(2, i.num); j++) {
                        sb.append(i.c);
                    }
                    break;
                    
                } else {
                    break;
                }
            } else {
                if (prev == i.c) {
                    Pair next = pq.poll();
                    
                    sb.append(next.c);
                    prev = next.c;
                    if (next.num == 1) {
                        pq.offer(i);
                    } else {
                        pq.offer(i);
                        next.num -= 1;
                        pq.offer(next);
                    }
                } else {
                    prev = i.c;
                    for (int j = 0; j < Math.min(2, i.num); j++) {
                        sb.append(i.c);
                    }
                    
                    if (i.num > 2) {
                        i.num -= 2;
                        pq.offer(i);
                    } else {
                        
                    }
                
}
            }
            //System.out.println(sb);
        }
        return sb.toString();
        
    }
}


public class Pair {
    char c;
    int num;
    
    public Pair(char c, int num) {
        this.c = c;
        this.num = num;
    }
}
