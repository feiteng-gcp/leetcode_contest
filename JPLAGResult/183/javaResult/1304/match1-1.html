class Solution {
    public String longestDiverseString(int a, int b, int c) {
        
        int d[][] = new int[3][2];
        d[0][0] =a;
        d[1][0] =b; d[1][1] = 1;
        d[2][0] =c; d[2][1] = 2;
        
        
        
        int lst = -1;
        int cnt = 0;
        StringBuilder sb = new StringBuilder();
        
        while(true){
            boolean ok =false;
        Arrays.sort(d,(x,y)->{
            return x[0]-y[0];
        });
        for(int i=2;i>=0;--i){
            int cur = d[i][1];
            if(d[i][0]==0) continue;
            
            if(cur==lst&&cnt+1<=2){
                sb.appendCodePoint(cur+'a');
                d[i][0]--;
                cnt++;
                ok = true;
                break;
            }else if(cur!=lst){
                sb.appendCodePoint(cur+'a');
                d[i][0]--;
                ok = true;
                cnt = 1;
                lst = cur;
                break;
            }
        }
            if(!ok){
                break;
            }
        }
        return sb.toString();
        
        
        
        
    }
}
