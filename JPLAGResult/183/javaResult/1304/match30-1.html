class Solution {
    //只包含a，b，c，组成一个不包含连续aaa，bbb，ccc的最长string，限制了a，b，c的个数
    //是由最长那个限制的，比如aa得配一个其他的b或者c，所以最长那个-2-剩下两个和*2，
    //如果<0，都能用
    //如果>0，多余的不能用
    public String longestDiverseString(int a, int b, int c) {
        return helper(a, b, c, '0');
    }
    private String helper(int a, int b, int c, char pre){
        // 贪婪，最大那个总是上两个
        if(a >= b && a >= c){
            if(a == 0) return "";
            else if(a == 1){
                if(pre != 'a'){
                    return "a" + helper(a-1,b,c, 'a');
                } else {
                    if(b >= c){
                        if(b == 0) return "";
                        else return "b" + helper(a,b-1,c,'b');
                    } else {
                        if(c == 0) return "";
                        else return "c" + helper(a,b,c-1,'c');
                    }
                }
            
} 
            else{
                if(pre != 'a'){
                    return "aa" + helper(a-2,b,c, 'a');
                } else {
                    if(b >= c){
                        if(b == 0) return "";
                        else return "b" + helper(a,b-1,c,'b');
                    } else {
                        if(c == 0) return "";
                        else return "c" + helper(a,b,c-1,'c');
                    }
                }
            } 
        } else if(b >= a && b >= c){
            if(b == 0) return "";
            else if(b == 1){
                if(pre != 'b'){
                    return "b" + helper(a,b-1,c, 'b');
                } else {
                    if(a >= c){
                        if(a == 0) return "";
                        else return "a" + helper(a-1,b,c,'a');
                    } else {
                        if(c == 0) return "";
                        else return "c" + helper(a,b,c-1,'c');
                    }
                }
            
}
            else{
                if(pre != 'b'){
                    return "bb" + helper(a,b-2,c, 'b');
                } else {
                    if(a >= c){
                        if(a == 0) return "";
                        else return "a" + helper(a-1,b,c,'a');
                    } else {
                        if(c == 0) return "";
                        else return "c" + helper(a,b,c-1,'c');
                    }
                }
            }
        } else {
            if(c == 0) return "";
            else if(c == 1){
                if(pre != 'c'){
                    return "c" + helper(a,b,c-1, 'c');
                } else {
                    if(a >= b){
                        if(a == 0) return "";
                        else return "a" + helper(a-1,b,c,'a');
                    } else {
                        if(b == 0) return "";
                        else return "b" + helper(a,b-1,c,'b');
                    
}
                }
            } else{
                if(pre != 'c'){
                    return "cc" + helper(a,b,c-2, 'c');
                } else {
                    if(a >= b){
                        if(a == 0) return "";
                        else return "a" + helper(a-1,b,c,'a');
                    } else {
                        if(b == 0) return "";
                        else return "b" + helper(a,b-1,c,'b');
                    
}
                }
            } 
        }
    }
}
