class Solution {
    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb = new StringBuilder();
        
        int[] counts = new int[]{a,b,c};
        
        int prev = -1;
        while(true){
            // find max
            int index = maxIndex(counts, prev);
            if(counts[index] == 0){
                break;
            }
            int count = 0;
            if(counts[index] == max(counts)){
                count = Math.min(2, counts[index]);
            }else{
                count = 1;
            }
            for(int i =0;i<count;i++){
                sb.append((char)('a'+index));
            }
            counts[index] -= count;
            prev = index;
        }
        return sb.toString();
    }   
    
    private int max(int[] counts){
        
int max = 0;
        for(int i =1;i<=2;i++){
            if(counts[i] > counts[max]){
                max = i;
            }
        }
        return counts[max];
    }
    
    private int maxIndex(int[] counts, int exclude){
        int max = -1;
        
for(int i =0;i<=2;i++){
            if(i != exclude && (max == -1 || counts[max] < counts[i])){
                max = i;
            }
        }
        return max;
    }
}
