class Solution {
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<Count> queue = new PriorityQueue<>((i, j) -> j.count - i.count);
        if (a > 0) queue.add(new Count ('a', a));
        if (b > 0) queue.add(new Count ('b', b));
        if (c > 0) queue.add(new Count ('c', c));
        StringBuilder sb = new StringBuilder();
        while (!queue.isEmpty()) {
            Count temp = queue.poll();
            if (sb.length() < 2) {
                // just append
                sb.append(temp.c);
                temp.count--;
                if (temp.count > 0) queue.add(temp);
                continue;
            }
            // Make sure prev 2 chars are not the same
            char cc = temp.c;
            int len = sb.length();
            if (sb.charAt(len-1) == cc && sb.charAt(len-2) == cc) {
                // Append a new character if possible
                if (queue.isEmpty()) break;
                Count temp2 = queue.poll();
                sb.append(temp2.c);
                temp2.count--;
                if (temp2.count > 0) queue.add(temp2);
                queue.add(temp);
                continue;
            } else {
                // just append
                sb.append(cc);
                temp.count--;
                if (temp.count > 0) queue.add(temp);
                continue;
            }
        }
        return sb.toString();
    }
    
    class Count {
        char c;
        int count;
        public Count(char c, int count) {
            this.c = c;
            this.count = count;
        }
    }
}
