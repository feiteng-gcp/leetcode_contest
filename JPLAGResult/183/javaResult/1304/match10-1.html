class Solution {
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((x, y) -> (y[1] - x[1]));
        
if(a > 0) {
            pq.add(new int[]{0, a});
        }
        if(b > 0) {
            pq.add(new int[]{1, b});
        }
        if(c > 0) {
            pq.add(new int[]{2, c});
        }
        StringBuilder sb = new StringBuilder();
        while(!pq.isEmpty()) {
            int[] cur = pq.poll();
            char letter = (char)(cur[0] + 'a');
            // System.out.println(letter);
            int len = sb.length();
            int pre0 = len - 2;
            int pre1 = len - 1;
            if(pre0 >= 0 && sb.charAt(pre0) == sb.charAt(pre1) && sb.charAt(pre1) == letter) {
                if(pq.isEmpty()) {
                    return sb.toString();
                } else {
                    int[] next = pq.poll();
                    sb.append((char)(next[0] + 'a'));
                    int num = next[1] - 1;
                    if(num > 0) {
                        pq.add(new int[]{next[0], num});
                    }
                    pq.add(cur);
                }
            } else {
                sb.append(letter);
                int num = cur[1] - 1;
                if(num > 0) {
                    pq.add(new int[]{cur[0], num});
                
}
            }
        }
        return sb.toString();
    }
}
