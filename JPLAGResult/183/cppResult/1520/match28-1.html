class Solution {
public:
    int numSteps(string s) {
        int i = 0;
        for (;s != "1";i++) {
            //cout << s << "\n";
            if (s.back() == '0') {
                s = s.substr(0, s.length()-1);
            } else {

                bool found = false;
                for (int j = s.length()-1; j >= 0; j--)  {
                    if (s[j] == '0') {
                        s[j] = '1';
                        found = true;
                        break;
                    } else {
                        s[j] = '0';
                    }
                }
                if (!found) {
                    s = "1"+s;
                }
            }
        }
        return i;
    }
};
