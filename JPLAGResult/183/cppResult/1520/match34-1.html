class Solution {
public:
    int numSteps(string s) {
		int Result = 0;

		while(s.size() > 1)
		{
			Result++;

			if(s.back() == '0')
			{
				s.pop_back();
			}
			else
			{
				int i;
				
				s.back() = '0';
				
				for(i = (s.size() - 2); i >= 0; i--)
				{
					if(s[i] == '1')
					{
						s[i] = '0';
					}
					else
					{
						s[i] = '1';
						break;
					}
				}
				
				if(i == -1)
				{
					Result += s.size();
					break;
				}
			}
			
			//cout << s << " " << Result << endl;
		}
		
		return Result;
    }
};
