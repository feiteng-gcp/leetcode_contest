class Solution {    
public:
    int numSteps(string s) {
        reverse(s.begin(), s.end());

        int ans = 0;
        
        // 1011
        // 111
        // 0001
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '1' && i+1 < s.size()) {      
                if (i+1 < s.size()) {
                    ++s[i+1];
                    s[i] = '0';
                } else {
                    s += '1';
                    s[i] = '0';
                }
                ++ans;
            }
            
            // carry
            for (int j = i; j < s.size(); ++j) {

                if (s[j] == '2') {
                    if (j+1 < s.size()) {
                        ++s[j+1];
                        s[j] = '0';
                    } else {
                        s += '1';
                        s[j] = '0';
                    }
                }
            }
            
            ++ans;
            // move bit
        }
        
        return ans-1;
    }
};
