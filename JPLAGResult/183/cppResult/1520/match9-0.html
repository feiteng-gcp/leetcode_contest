class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
    while(s != "1")
    {
        if(s.back() == '0') s.pop_back();
        else
        {
            int id = (int)s.size() - 1;
            while(id >= 0 && s[id] == '1')
            {
                s[id--] = '0';
            }
            if(id < 0) s = "1" + s;
            else s[id] = '1';
        }
        ans += 1;
    }
    return ans;
    }
};
