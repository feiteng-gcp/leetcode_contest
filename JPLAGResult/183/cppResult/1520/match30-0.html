class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while(s != "1") {
            ans++;
            if(s.back() == '0') {
                s.pop_back();
            }
            else {
                int n = s.size();
                bool flg = true;
                for(int i = n-1; i >= 0; i--) {
                    if(s[i] == '1') {
                        s[i] = '0';
                    }
                    else {
                        s[i] = '1';
                        flg = false;
                        break;
                    }
                }
                if(flg) {
                    s = "1"+s;
                }
            }
        }
        return ans;
    }
};
