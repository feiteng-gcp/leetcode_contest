class Solution {
public:
    int numSteps(string s) {
      int ans = 0;
      while (s != "1") {
        ++ans;
        if (s.back() == '0') {
          s.pop_back();
        } else {
          int p = s.size() - 1;
          while (p >= 0 && s[p] == '1') {
            s[p--] = '0';
          }
          if (p == -1) {
            s = '1' + s;
          } else {
            s[p] = '1';
          }
        }
      }
      return ans;
    }
};
