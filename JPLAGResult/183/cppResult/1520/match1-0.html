class Solution {
public:
    int numSteps(string s) {
        int ans = 0, t, i, l;
        while (s.size() > 1) {
            if (s.back() == '0') {
                s.pop_back();
            } else {
                l = s.size();
                s[l - 1] += 1;
                for (i = l; i > 0; i--) {
                    if (s[i] == '2') {
                        s[i] = '0';
                        s[i - 1]++;
                    }
                }
                if (s[0] == '2') {
                    s[0] = '0';
                    s = "1" + s;
                }
            }
            ans++;
        }
        return ans;
    }
};
