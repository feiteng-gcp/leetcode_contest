class Solution {
public:
    int numSteps(string s) {
        int c = 0;
        int n = 0;
        for (int i = s.size() - 1; i >= 0; --i) {
            if (s[i] == '0') {
                if (c == 0) {
                    n++;
                } else {
                    n += 2;
                    c = 1;
                }
            } else {
                if (c == 0) {
                    if (i == 0) break;
                    n += 2;
                    c = 1;
                } else {
                    n++;
                    if (i == 0) break;
                }
            }
        }
        return n;
    }
};
