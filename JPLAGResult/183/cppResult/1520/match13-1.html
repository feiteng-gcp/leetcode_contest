class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while (s.size() != 1) {
            ++ans;
            if (s[s.size() - 1] == '0') {
                s = s.substr(0, s.size() - 1);
            } else {
                int pos = s.size() - 1;
                while (pos >= 0 && s[pos] == '1') s[pos--] = '0';

                if (pos < 0) {
                    s = "1" + s;
                } else {
                    s[pos] = '1';
                }
            }
        }
        return ans;
    }
};
