class Solution {
public:
    
    string AddOne(string& s) {
        string ans = s;
        int carry = 1;
        for (int i = s.size() - 1; i >= 0; --i) {
            if (carry == 0) {
                break;
            }
            if (ans[i] - '0' + carry > 1) {
                ans[i] = '0';

                carry = 1;
            } else {
                ans[i] = '1';
                carry = 0;
                break;
            }
        }
        if (carry > 0) {
            ans.insert(ans.begin(), '1');
        }
        
        return ans;
    }
    
    int numSteps(string s) {
        if (s == "1") {
            return 0;
        }
        int count = 0;
        while (s != "1") {

            if (s.back() == '1') {
                s = AddOne(s);
            } else {
                s.pop_back();
            }
            ++count;
        }
        return count;
    }
};
