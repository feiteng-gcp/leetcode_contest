class Solution {
public:
  int numSteps(string s) {
    int ans = 0;
    while (s != "1") {
      ans++;
      if (s.back() == '0') s.pop_back();
      else {
        int carry = 1;
        for (int i = s.size() - 1; i >= 0; i--) {
          if (s[i] == '0') {
            s[i]++;
            carry = 0;
            break;
          } else {
            s[i] = '0';
          }
        }
        if (carry) s.insert(0, "1");
      }
    }
    return ans;
  }
};
