class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        int carry = 0;
        for(int i=s.length()-1;i>=0;i--){
            if(i==0&&carry==0) break;
            int cur = carry+(s[i]-'0');
            carry = cur/2;
            cur = cur%2;
            if(cur==1){
                carry=1;
                ans++;
            }
            ans++;
        }
        return ans;
    }
};
