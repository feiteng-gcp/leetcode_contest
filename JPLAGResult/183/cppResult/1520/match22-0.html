class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while(s.size() > 1){
            if(s.back() == '1'){
                s = addOne(s);
            }else{
                s.pop_back();
            }
            ans++;
        }
        return ans;
    }
private:
    string addOne(string& s){
        string ans(s.begin(), s.end());
        int add = 1;
        for(int i = ans.size() - 1; i >= 0; i--){
            if(ans[i] == '1'){
                ans[i] = '0';
                add = 1;
            }else{
                ans[i] = '1';
                add = 0;
                break;
            }
        }
        if(add == 1){
            ans.insert(ans.begin(), '1');
        }
        return ans;
    }
};
