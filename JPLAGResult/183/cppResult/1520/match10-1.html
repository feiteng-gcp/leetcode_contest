class Solution {
public:
    int numSteps(string s) {

        int res = 0;
        int c = 0;
        for (int i = s.length() - 1; i >= 0; i--) {
            int b = (s[i] - '0') + c;
            
            if (i == 0 && b == 1) break;
            
            if (b % 2 == 0) {
                res ++;
                c = b / 2;
            } else {
                res += 2;
                c = 1;
            }
        }
        
        return res;
    }
};
