class Solution {
public:
    int numSteps(string s) {
        int ans =0;
        while(s!="1")
        {
            //cout<<ans<<" "<<s<<"\n";
            int sz = s.size();
            if(s[sz-1]=='0')
                s= s.substr(0,sz-1);

            else
            {
                int idx = sz-1;
                while(idx>=0 && s[idx]=='1')
                {
                    s[idx]='0';
                    idx--;
                }
                if(idx==-1)
                    s='1'+s;
                else
                    s[idx]='1';
                    
            }
            ans++;
        }
        return ans;
        
    }
};
