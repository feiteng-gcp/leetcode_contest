class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while(s.size() > 1) {
            if(s[s.size() - 1] == '0') s.pop_back();
            else {
                s[s.size() - 1] = '0';
                int j = (int)s.size() - 2;

                for(; j >= 0 && s[j] == '1';) {
                    s[j --] = '0';
                }
                if(j == -1) s = "1" + s;
                else s[j] = '1';
            }
            ans ++;
        }
        return ans;
    }
};
