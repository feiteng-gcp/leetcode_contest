class Solution {
public:
    string add(string &s){
        int ls = s.size();
        reverse(s.begin(), s.end());
        int e=1;
        for(int i=0; i<ls; i++){
            if(s[i] == '1'){
                s[i] = '0';
            }else{
                s[i] = '1';
                e = 0;
                break;
            }
        }
        if(e == 1){
            s.push_back('1');
        }
        reverse(s.begin(), s.end());
        return s;
    }
    int numSteps(string s) {
        int step = 0;
        while(s != "1"){
            int ls = s.size();
            if(s[ls-1] == '1') {
                s = add(s);
            }else{
                s.pop_back();
            }
            step++;
        }
        return step;
    }
};
