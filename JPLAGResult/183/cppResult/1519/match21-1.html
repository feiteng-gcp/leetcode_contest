class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        reverse(nums.begin(), nums.end());
        int sum = 0;
        for(int i : nums) sum += i;
        int sum1 = 0;
        vector<int> ans;

        for(int i : nums){
            sum1 += i;
            sum -= i;
            ans.push_back(i);
            if(sum < sum1){
                break;
            }
        }
        return ans;
    }
};
