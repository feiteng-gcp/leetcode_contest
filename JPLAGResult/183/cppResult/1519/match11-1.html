class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int n = nums.size();
        int all = 0;
        for (int i = 0; i < n; i++)
            all += nums[i];
        int cur = 0;
        sort(nums.begin() , nums.end(), greater<int>() );
        vector<int> ans = nums;

        for (int i = 0; i < n; i++) {
            cur += ans[i];
            if (cur > all - cur)
            {
                ans.resize(i + 1);
                return ans;
            }
        }
        return ans;
    }
};
