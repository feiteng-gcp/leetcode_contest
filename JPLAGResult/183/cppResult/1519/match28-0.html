class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        sort(nums.begin(), nums.end(), [](int a, int b) {
            return a > b;
        });
        vector<int> result;
        int s = 0;
        for (int num : nums) {
            result.push_back(num);
            s += num;
            if (s > sum - s) {  //!1 sum
                break;
            }
        }
        return result;
    }
};
