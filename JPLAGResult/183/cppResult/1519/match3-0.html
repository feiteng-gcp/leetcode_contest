class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int sum = 0;
        for(int num : nums){
            sum += num;
        }
        sort(nums.begin(), nums.end());
        vector<int> ans;
        int curSum = 0;
        for(int i = nums.size() - 1; i >= 0; i--){
            ans.push_back(nums[i]);
            curSum += nums[i];
            if(curSum * 2 > sum){
                break;
            }
        }
        return ans;
    }
};
