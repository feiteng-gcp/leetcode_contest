class Solution {
 public:
  vector<int> minSubsequence(vector<int>& nums) {
    sort(nums.rbegin(), nums.rend());
    int sum = 0;
    for (int num : nums)
      sum += num;
    vector<int> res;
    int presum = 0;
    for (int num : nums) {
      presum += num;
      sum -= num;
      res.push_back(num);
      if (presum > sum) {
        break;
      }
    }
    return res;
  }
};
