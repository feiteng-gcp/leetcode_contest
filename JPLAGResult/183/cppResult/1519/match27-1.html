class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<int> answer;
        
        int sum = 0;
        for(int i = 0; i < nums.size(); i++) {
            sum += nums[i];
        }

        int sum2 = 0;
        for (int i=nums.size() -1; i>= 0; i--) {
            sum2 +=nums[i];
            sum -= nums[i];
            answer.push_back(nums[i]);
            
            if(sum2 > sum) return answer;
        }
        return answer;
    }
};
