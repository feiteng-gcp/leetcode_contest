class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        vector<int>ans;
        int s=0;
        for(int i=nums.size()-1;i>=0;--i){
            s+=nums[i];
        }
        int t=0;
        for(int i=nums.size()-1;i>=0;--i){
            ans.push_back(nums[i]);
            t+=nums[i];
            if(2*t>s)return ans;
        }
        return ans;
    }
};
