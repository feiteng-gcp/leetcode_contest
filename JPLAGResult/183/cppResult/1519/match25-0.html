class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int s = 0;
        sort(nums.begin(), nums.end());
        reverse(nums.begin(), nums.end());
        for(int a : nums) s += a;
        vector<int> res;
        int c = 0;
        for(int i=0; i < nums.size(); i++){
            c += nums[i];
            res.push_back(nums[i]);
            if(c > s-c) break;
        }
        return res;
    }
};
