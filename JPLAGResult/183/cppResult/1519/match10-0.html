class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int i, sum = 0, t = 0;
        vector<int> ans;
        sort(nums.begin(), nums.end(), greater<int>());
        for (i = 0; i < nums.size(); i++) {
            sum += nums[i];
        }
        for (i = 0; i < nums.size(); i++) {
            t += nums[i];
            ans.push_back(nums[i]);
            if (t > sum - t) {
                return ans;
            }
        }
        return ans;
    }
};
