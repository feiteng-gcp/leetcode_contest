class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int sum = 0;
        for (auto x: nums) sum += x;
        int cur = 0;
        vector<int> ans;
        for (int i = nums.size() - 1; i >= 0; --i) {
            cur += nums[i];
            ans.emplace_back(nums[i]);
            if (cur > sum - cur) break;
        }
        return ans;
    }
};
