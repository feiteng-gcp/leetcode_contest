class Solution {
public:
    string stoneGameIII(vector<int>& stones) {
        int n = stones.size();
        reverse(stones.begin(), stones.end());
        vector<int> a(n + 1), s(n + 1);
        for (int i = 1; i <= n; i++) {
            a[i] = stones[i - 1];
            s[i] = s[i - 1] + a[i];
        }

        vector<int> dp(n + 1);
        dp[1] = a[1];
        for (int i = 2; i <= n; i++) {
            dp[i] = INT_MIN;
            for (int j = 1; j <= 3 && j <= i; j++) {
                dp[i] = max(dp[i], s[i] - dp[i - j]);
            }
            // cout << dp[i] << endl;
        }
        if (dp[n] > s[n] - dp[n]) return "Alice";
        else if (dp[n] == s[n] - dp[n]) return "Tie";
        else return "Bob";
    }
};
