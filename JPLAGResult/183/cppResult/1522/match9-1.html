class Solution {
public:
    string stoneGameIII(vector<int>& a) {
        int n = a.size();
        a.resize(n + 10);
        vector<int> prefix_sum(n + 10);
        for (int i = n - 1; i >= 0; i--) {
            prefix_sum[i] = a[i] + (i + 1 < n ? prefix_sum[i + 1] : 0);
        }
        const int inf = 1e9;
        vector<int> dp(n + 10, -inf);

        for (int i = n; i < dp.size(); i++) dp[i] = 0;
        for (int i = n - 1; i >= 0; i--) {
            int sum = 0;
            for (int j = 0; j < 3; j++) {

                sum += a[i + j];
                dp[i] = max(dp[i], sum + prefix_sum[i + j + 1] - dp[i + j + 1]);
            }
        }
        int alice = dp[0];
        int bob = prefix_sum[0] - alice;
        if (alice > bob) return "Alice";
        else if (bob > alice) return "Bob";
        else return "Tie";
    }
};
