class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int sz = stoneValue.size();
        vector<int> dp(sz, -60000000);  // dp[i]：i - end() 先手最多拿多少 先手胜利1, 先手失败-1，平局 0
        if(sz == 1) {
            if(stoneValue[0] > 0) return "Alice";
            else if(stoneValue[0] == 0) return "Tie";
            else return "Bob";
        }
        int s = 0;
        dp[sz - 1] = stoneValue[sz - 1];

        s += stoneValue[sz - 1];
        for(int i = sz - 2; i >= 0; --i) {
            s += stoneValue[i];
            for(int j = min(i + 2, sz - 1); j >= i; --j) {
                int other = 0;
                if(j < (sz - 1))
                    other = dp[j + 1];

                dp[i] = max(dp[i], s - other);
            }
        }
        if(dp[0] > s - dp[0])
            return "Alice";
        else if(dp[0] < s - dp[0])
            return "Bob";
        else
            return "Tie";
    }
};
