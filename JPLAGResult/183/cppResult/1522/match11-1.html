class Solution {
public:
    string stoneGameIII(const vector<int>& val) {
        vector <int> prefix_sum(val.size() + 1, 0);
        for (int i=val.size()-1; i>=0; --i){
            prefix_sum[i] = prefix_sum[i+1] + val[i];
        }
        vector<int> dp(val.size()+1,INT_MIN);
        dp[val.size()] = 0;

        for(int i=val.size()-1; i>=0; --i){
            int cand_sum = 0;
            for(int j=0; j<=2; j++){
                if(i+j <val.size()){
                     cand_sum += val[i+j];

                     int curr_sum = cand_sum + (prefix_sum[i+j+1]-dp[i+j+1]);
                     dp[i] = max(dp[i],curr_sum);
                }
            }
        }
        if(dp[0] == prefix_sum[0] - dp[0])
            return "Tie";
        else if(dp[0] > prefix_sum[0] - dp[0])
            return "Alice";
        else
            return "Bob";
        
    }
};
