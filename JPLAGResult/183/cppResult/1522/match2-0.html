class Solution {
public:
    int get(int st, int n, vector<int>& stoneValue, vector<int>& dp) {
        if (st == n) {
            return 0;
        }
        if (dp[st] > -500000000) {
            return dp[st];
        }
        if (st < n) {
            dp[st] = max(dp[st], stoneValue[st] - get(st + 1, n, stoneValue, dp));
        }
        if (st + 1 < n) {
            dp[st] = max(dp[st], stoneValue[st] + stoneValue[st + 1] - get(st + 2, n, stoneValue, dp));
        }
        if (st + 2 < n) {
            dp[st] = max(dp[st], stoneValue[st] + stoneValue[st + 1] + stoneValue[st + 2] - get(st + 3, n, stoneValue, dp));
        }
        return dp[st];
    }
    string stoneGameIII(vector<int>& stoneValue) {
        int ans = 0;
        int i, n = stoneValue.size();
        vector<int> dp(n, -500000000);
        ans = get(0, n, stoneValue, dp);
        if (ans > 0) {
            return "Alice";
        } else if (ans < 0) {
            return "Bob";
        }
        return "Tie";
    }
};
