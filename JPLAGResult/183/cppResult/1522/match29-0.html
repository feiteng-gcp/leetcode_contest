class Solution {
public:
    int ret[50005];
    
    string stoneGameIII(vector<int>& v) {
        int n = v.size();
        int sum = 0;
        for (int i = n - 1; i >= 0; --i) {
            sum += v[i];
            ret[i] = v[i] + (sum - v[i] - ret[i + 1]);
            if (i + 1 < n) ret[i] = max(ret[i], v[i] + v[i + 1] + (sum - v[i] - v[i + 1] - ret[i + 2]));
            if (i + 2 < n) ret[i] = max(ret[i], v[i] + v[i + 1] + v[i + 2] + (sum - v[i] - v[i + 1] - v[i + 2] - ret[i + 3]));
        }
        if (ret[0] + ret[0] == sum) return "Tie";
        if (ret[0] + ret[0] > sum) return "Alice";
        return "Bob";
    }
};
