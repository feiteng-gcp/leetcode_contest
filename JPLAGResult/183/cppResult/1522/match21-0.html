class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();

        int dp[n+1][2];
        dp[n][0] = dp[n][1] = 0;
        for (int i = n - 1; i >= 0; i--) {
            int s = 0;
            dp[i][0] = -6e8, dp[i][1] = 6e8;
            for (int j = i + 1; j <= n && j <= i + 3; j++) {
                s += stoneValue[j-1];
                dp[i][0] = max(dp[i][0], dp[j][1] + s);
                dp[i][1] = min(dp[i][1], dp[j][0]);
            }
        }

        int sum = 0;
        for (int a : stoneValue) {
            sum += a;
        }

        return dp[0][0] > sum - dp[0][0] ? "Alice" : dp[0][0] == sum - dp[0][0] ? "Tie" : "Bob";
    }
};
