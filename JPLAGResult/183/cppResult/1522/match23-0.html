class Solution {
public:
    vector<int> dp;
    string stoneGameIII(vector<int>& s) {
        int n=s.size();
        dp=vector<int>(n, INT_MAX);
        int res=dfs(s, 0);
        // cout<<res<<endl;
        if(res>0) return "Alice";
        else if(res<0) return "Bob";
        else return "Tie";
    }
    
    int dfs(vector<int>& s, int i){
        if(i>=s.size()) return 0;
        if(dp[i]!=INT_MAX) return dp[i];
        int tmp=s[i]-dfs(s, i+1);
        for(int l=1;l<3 && i+l<s.size();l++){
            tmp=max(tmp, accumulate(s.begin()+i, s.begin()+i+l+1, 0)-dfs(s, i+l+1));
        }
        dp[i]=tmp;
        return tmp;
    }
};
