class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> d[2];
        d[0].resize(n+1,0), d[1].resize(n+1,0);
        d[0][n] = 0;
        d[1][n] = 0;
        for(int i=n-1;i>=0;i--){
            int sum = stoneValue[i];
            d[0][i] = -1e9;
            d[1][i] = 1e9;
            for(int j=i+1; j <= n && j <= i+3; j++){
                d[0][i] = max(d[0][i], sum + d[1][j]);
                d[1][i] = min(d[1][i], d[0][j] - sum);
                if(j < n)
                    sum += stoneValue[j];
            }
            //cout << i << ' ' << d[0][i] << ' ' << d[1][i] << endl;
        }
        if(d[0][0] > 0) return "Alice";
        if(d[0][0] == 0) return "Tie";
        return "Bob";
    }
};
