class Solution {
public:
    int N;
    vector<int> dpA,dpB;
    int dfsA(vector<int>& stoneValue,int index,int sub){
        if(index>=N){
            return sub;
        }
        if(dpA[index]!=INT_MAX){
            return sub+dpA[index];
        }
        int res=INT_MIN;
        int osub=sub;
        for(int i=index;i<index+3;i++){
            if(i>=N){
                break;
            }
            sub+=stoneValue[i];
            res=max(res,dfsB(stoneValue,i+1,sub));
        }
        dpA[index]=res-osub;
        return res;
    }
    int dfsB(vector<int>& stoneValue,int index,int sub){
        if(index>=N){
            return sub;
        }       
        if(dpB[index]!=INT_MAX){
            return sub+dpB[index];
        }
        int res=INT_MAX;
        int osub=sub;
        for(int i=index;i<index+3;i++){
            if(i>=N){
                break;
            }
            sub-=stoneValue[i];
            res=min(res,dfsA(stoneValue,i+1,sub));
        }
        dpB[index]=res-osub;
        return res;
    }
    string stoneGameIII(vector<int>& stoneValue) {
        N=stoneValue.size();

        dpA=vector<int>(N,INT_MAX);
        dpB=vector<int>(N,INT_MAX);
        int res=dfsA(stoneValue,0,0);
        if(res>0){
            return "Alice";
        }
        if(res<0){
            return "Bob";
        }     
        return "Tie";
    }
};
