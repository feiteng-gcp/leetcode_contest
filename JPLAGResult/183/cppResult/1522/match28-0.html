class Solution {
public:
  string stoneGameIII(vector<int>& stoneValue) {
    vector<int> A = stoneValue;
    reverse(A.begin(), A.end());
    vector<int> dp(A.size());
    for (int i = 0; i < A.size(); i++) {
      int cur = 0;
      int best = -0x3f3f3f3f;
      for (int k = 1; k <= 3; k++) {
        if (i - k + 1 >= 0)  cur += A[i-k+1];
        int off = 0;
        if (i - k >= 0) off = dp[i-k];
        best = max(best, cur - off);
      }
      dp[i] = best;
    }
    if (dp.back() > 0) return "Alice";
    if (dp.back() < 0) return "Bob";
    return "Tie";
  }
};
