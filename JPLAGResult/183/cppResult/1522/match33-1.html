class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int N = stoneValue.size();
        vector<int> outcome(N);
        for(int i = N-1; i >= 0; i--)
        {
            int best = INT_MIN;
            int r = 0;
            for(int j = i; j < min(i+3, N); j++)
            {
                r += stoneValue[j];
                int temp = (j+1) < N ? r - outcome[j+1] : r;
                best = max(best, temp);
            }
            outcome[i] = best;
            //cout << "outcome " << i << ": " << outcome[i] << endl;
        }
        
        if(outcome[0] == 0)
        {
            return "Tie";
        }
        else
        {
            if(outcome[0] < 0)
            {
                return "Bob";
            }
            else
            {
                return "Alice";
            }
        }
    }
};
