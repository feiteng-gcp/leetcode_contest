class Solution {
public:
  string stoneGameIII(vector<int>& v) {
    int N=v.size();
    vector<int> z(N+1), s(N+1);
    for(int i=N-1;i>=0;i--) s[i] = s[i+1] + v[i];
    for(int i=N-1;i>=0;i--){
      z[i] = s[i]-z[i+1];
      if(i+2<=N) z[i]=max(z[i], s[i]-z[i+2]);
      if(i+3<=N) z[i]=max(z[i], s[i]-z[i+3]);
    }
      // cout<<z[0]<<endl;
    if (z[0]*2 > s[0]) return "Alice";
    if (z[0]*2 < s[0]) return "Bob";
    return "Tie";
  }
};
