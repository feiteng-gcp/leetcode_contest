class Solution {
public:
    vector<int> sum;
    vector<int> dp;
    int get(int n, int len) {
        if (len == n) return 0;

        if (len - n == 1) return sum[n];
        if (dp[n] != INT_MIN) return dp[n];
        
        int res = INT_MIN;
        for (int i = 1; i <= 3 && i + n <= len; i++) {
            res = max(res, sum[n] - get(n + i, len));
        }
        
        dp[n] = res;
        return res;
    }
    
    string stoneGameIII(vector<int>& stoneValue) {
        sum = vector<int>(stoneValue.size(), stoneValue.back());
        for (int i = stoneValue.size() - 2; i >= 0; i--) 

            sum[i] = sum[i+1] + stoneValue[i];
        
        dp = vector<int>(stoneValue.size(), INT_MIN);
        
        int res = get(0, stoneValue.size());
        
        cout << res << " " << sum[0] - res << endl;
        if (sum[0] - res < res) return "Alice";
        else if(sum[0] - res > res) return "Bob";
        else return "Tie";
    }
};
