class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> ps(n+1);
        ps[n] = 0;
        for (int i = n; i-->0;) {
            ps[i] = stoneValue[i] + ps[i+1];
        }
        vector<int> opt(n+1);
        opt[n] = 0;
        if (n >= 1) {
            opt[n-1] = ps[n-1];
        }
        if (n >= 2) {
            opt[n-2] = ps[n-2] - min(opt[n-1],opt[n]);
        }
        for (int i = n-2; i-->0;) {

            opt[i] = ps[i] - min({opt[i+1], opt[i+2], opt[i+3]});
        }
        if (2*opt[0] > ps[0]) {
            return "Alice";
        } else if (2*opt[0] == ps[0]) {
            return "Tie";
        } else {
            return "Bob";
        }
    }
};
