class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int s = stoneValue.size();
        vector<pair<int,int>> score(s,make_pair(INT_MIN,INT_MAX));
        score.push_back(make_pair(0,0));
        score.push_back(make_pair(0,0));
        score.push_back(make_pair(0,0));
        stoneValue.push_back(0);
        stoneValue.push_back(0);
        for(int i=s-1;i>=0;i--){
            int now=0;

            for(int j=0;j<3;j++){
                now+=stoneValue[i+j];
                if(now+score[i+j+1].second>score[i].first){
                    score[i].first = now+score[i+j+1].second;
                    score[i].second = score[i+j+1].first;
                }
            }
        }
        //cout<<score[0].first<<endl;
        if(score[0].first>score[0].second)
            return "Alice";
        else if(score[0].first<score[0].second)
            return "Bob";
        else
            return "Tie";
    }
};
