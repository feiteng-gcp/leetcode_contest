class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        vector<pair<int, int>> dp(1, make_pair(0, 0));
        int n = stoneValue.size();
        for (int i = 1; i <= n; i++) {  // 当有i堆石子，stoneValue[n-i, n-1]
            int maxV = -0x3f3f3f3f;
            int maxP = -1;

            int tot = 0;
            for (int j = 1; j <= 3 && j <= i; j++) {
                tot += stoneValue[n-i+j-1];
                if (maxV < tot+dp[i-j].second) {
                    maxV = tot+dp[i-j].second;
                    maxP = i-j;
                }
            } 
            dp.push_back(make_pair(maxV, dp[maxP].first));
        }
        if (dp[n].first == dp[n].second)
            return "Tie";
        else if (dp[n].first < dp[n].second)
            return "Bob";
        else
            return "Alice";
    }
};
