class Solution {
public:
    string stoneGameIII(vector<int>& a) {
      int n = a.size();
      vector<int> dp(n + 1, -1e9);
      dp[n] = 0;
      int s = 0;
      for (int i = n - 1; i >= 0; --i) {
        s += a[i];
        for (int j = 0; i + j < n && j < 3; ++j) {
          int p = i + j;
          int v = s - dp[p + 1];
          dp[i] = max(v, dp[i]);
        }
      }
      if (dp[0] + dp[0] > s) {
        return "Alice";
      } else if (dp[0] + dp[0] < s) {
        return "Bob";
      } else {
        return "Tie";
      }
    }
};
