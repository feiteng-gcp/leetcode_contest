class Solution {
public:
    int len;
    vector<int> mem[2];
    vector<int> sum;
    int recursiveSolve(int i, int who) {
        if (i >= len) {
            return 0;
        }
        if (mem[who][i] != INT_MAX) {
            return mem[who][i];
        }
        int res = recursiveSolve(i+1, who^1);
        res = min(res, recursiveSolve(i+2, who^1));
        res = min(res, recursiveSolve(i+3, who^1));
        res = sum[i] - res;
        mem[who][i] = res;
        return res;
    }
    string stoneGameIII(vector<int>& stoneValue) {
        len = stoneValue.size();
        sum = vector<int>(len, 0);
        sum[len-1] = stoneValue[len-1];
        for (int i = len-2; i >= 0; --i) {
            sum[i] = stoneValue[i] + sum[i+1];
        }
        mem[0] = vector<int>(len+1, INT_MAX);
        mem[1] = vector<int>(len+1, INT_MAX);
        int val = recursiveSolve(0, 0);
        if (val * 2 > sum[0]) {
            return "Alice";
        }
        if (val * 2 < sum[0]) {
            return "Bob";
        }
        return "Tie";
    }
};
