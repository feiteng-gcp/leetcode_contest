class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int len = (int)stoneValue.size();
    vector<int> dp(len + 1, -50000000), sum(len + 1, 0);
    for(int i = len - 1; i >= 0; i--)
        sum[i] = sum[i + 1] + stoneValue[i];
    dp[len] = 0;
    for(int i = len - 1; i >= 0; i--)
    {
        int s = 0;
        for(int j = i; j < i + 3 && j < len; j++)
        {
            s += stoneValue[j];
            dp[i] = max(dp[i], s + sum[j + 1] - dp[j + 1]);
        }
    }
    if(2 * dp[0] > sum[0]) return "Alice";
    else if(2 * dp[0] < sum[0]) return "Bob";
    else return "Tie";
    }
};
