class Solution {
public:
    string stoneGameIII(vector<int>& a) {
        int n = a.size();
        vector <int> f;
        f.resize(n + 1);
        f[n] = 0;
        for (int i = n - 1; i >= 0; i--) {
            f[i] = -1e9;
            int s = 0;
            for (int j = i; j < n && j < i + 3; j++) {
                s += a[j];
                f[i] = max(f[i], s - f[j + 1]);
            }
        }
        if (f[0] > 0) {
            return "Alice";
        } else if (f[0] == 0) {
            return "Tie";
        } else {
            return "Bob";
        }
    }
};
