class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size(), i, sum = 0;
        vector<int> sum_vec;
        for (i = n - 1; i >= 0; --i) {
            sum += stoneValue[i];
            sum_vec.push_back(sum);
        }        
        reverse(sum_vec.begin(), sum_vec.end());
        sum_vec.push_back(0);
        vector<int> dp(n + 1, INT_MIN);
        dp[n] = 0;
        for (i = n - 1; i >= 0; --i) {
            int cur = 0;
            for (int j = i; j < n && j <= i + 2; ++j) {
                cur += stoneValue[j];
                // cout << cur << " " << j << " " << sum_vec[j + 1] << " " << dp[j + 1] << endl;
                dp[i] = max(dp[i], cur + sum_vec[j + 1] - dp[j + 1]);
            }
            // cout << i << " " << dp[i] << endl;
        }
        if (dp[0] > sum - dp[0]) return "Alice";
        else if (dp[0] < sum - dp[0]) return "Bob";
        else return "Tie";
    }
};
