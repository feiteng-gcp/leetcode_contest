class Solution {
public:
    vector<int> sum;
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        sum = vector<int>(n+1, 0);
        for(int i=n-1; i>=0; i--){
            sum[i] += stoneValue[i];
            if(i<n-1) sum[i] += sum[i+1];
        }

        vector<int> dp(n+1, 0);
        dp[n-1] = stoneValue[n-1];
        for(int i=n-2; i>=0; i--){
            dp[i] = INT_MIN;
            int s = 0;
            for(int j=i; j<i+3 && j<n; j++){
                s += stoneValue[j];
                dp[i] = max(dp[i], s + sum[j+1] - dp[j+1]);
            }
        }

        int s1 = dp[0], s2 = sum[0] - dp[0];
        if(s1 > s2) return "Alice";
        else if(s1 < s2) return "Bob";
        else return "Tie";
    }
};
