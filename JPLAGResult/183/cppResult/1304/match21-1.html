class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res;
        priority_queue<pair<int, char>> pq;
        if(a) pq.push({a, 'a'});
        if(b) pq.push({b, 'b'});
        if(c) pq.push({c, 'c'});
        while(pq.size()) {
            int n;
            char ch;
            tie(n, ch) = pq.top();
            pq.pop();
            int s = res.size();
            if(res.size() > 1 and ch == res[s - 1] and ch == res[s - 2]) {
                if(pq.size()) {
                    int _n;
                    char _ch;

                    tie(_n, _ch) = pq.top();
                    pq.pop();
                    res.push_back(_ch);
                    if(_n > 1) {
                        pq.push({_n - 1, _ch});
                    }
                    pq.push({n, ch});
                }
            } else {
                res.push_back(ch);
                if(n > 1) {
                    pq.push({n - 1, ch});
                }
            }
        }
        return res;
    }
};
