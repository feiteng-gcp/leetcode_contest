class Solution {

        if(a < 0 || b < 0 || c < 0) return "";
        string res = "";
        if(a == 0 && b == 0 && c == 0) return res;
        if(a >= b && a >= c){
            res += 'a';
            a--;
        }
        else if(b >= a && b >= c){
            res += 'b';
            b--;
        }else{
            res += 'c';
            c--;
        }
        if(a == 0 && b == 0 && c == 0) return res;

            res += 'a';
            a--;

        else if(b >= a && b >= c){
            res += 'b';
            b--;
        }else{
            res += 'c';
            c--;

            if(a == 0 && b == 0 && c == 0) return res;
            if(res[res.size()-1] == 'a' && res[res.size()-2] == 'a'){
                if(b <= 0 && c <= 0) break;
                if(b >= c){
                    res += 'b';
                    b--;
                }else{
                    res += 'c';
                    c--;
                }
            }
            else if(res[res.size()-1] == 'b' && res[res.size()-2] == 'b'){
                if(a <= 0 && c <= 0) break;
                if(a >= c){
                    res += 'a';
                    a--;
                }else{
                    res += 'c';
                    c--;
                }
            }
            else if(res[res.size()-1] == 'c' && res[res.size()-2] == 'c'){
                if(a <= 0 && b <= 0) break;
                if(a >= b){
                    res += 'a';
                    a--;
                }else{
                    res += 'b';
                    b--;
                }
            }else{
                if(a >= b && a >= c){
                    res += 'a';
                    a--;
                }
                else if(b >= a && b >= c){
                    res += 'b';
                    b--;
                }else{
                    res += 'c';
                    c--;
                }
            }
        }
