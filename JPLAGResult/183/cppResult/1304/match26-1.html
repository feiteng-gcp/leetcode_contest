class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res = "";
        int size = 0;
        while (a + b + c > 0) {
            if (c >= a && c >= b) {

                if (size >= 2 && res[size - 1] == res[size - 2] && res[size - 1] == 'c') {
                    if (a >= b && a > 0) {
                        res.push_back('a');
                        --a;
                    } else if (b >= a && b > 0) {
                        res.push_back('b');
                        --b;
                    } else {
                        break;
                    }
                } else {
                    res.push_back('c');
                    --c;
                }   
            } else if (a >= b && a >= c) {

                if (size >= 2 && res[size - 1] == res[size - 2] && res[size - 1] == 'a') {
                    if (b >= c && b > 0) {
                        res.push_back('b');
                        --b;
                    } else if (c >= b && c > 0) {
                        res.push_back('c');
                        --c;
                    } else {
                        break;
                    }
                } else {
                    res.push_back('a');
                    --a;
                }                 
            } else {

                if (size >= 2 && res[size - 1] == res[size - 2] && res[size - 1] == 'b') {
                    if (a >= c && a > 0) {
                        res.push_back('a');
                        --a;
                    } else if (c >= a && c > 0) {
                        res.push_back('c');
                        --c;
                    } else {
                        break;
                    }
                } else {
                    res.push_back('b');
                    --b;
                }                
            }
            ++size;
        }
        return res;
    }
};
