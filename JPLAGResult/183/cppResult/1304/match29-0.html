class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res = "";
        
        priority_queue<pair<int, char>> pq;
        if (a > 0)    pq.push({a, 'a'});
        if (b > 0)    pq.push({b, 'b'});
        if (c > 0)    pq.push({c, 'c'});
        
        while (!pq.empty()) {
            vector<pair<int, char>> temp;
            while (!pq.empty() && !valid(res, pq.top().second)) {
                temp.push_back(pq.top());
                pq.pop();
            }
            
            if (pq.empty())    break;
            auto out = pq.top();    pq.pop();
            int cnt = out.first;
            char c = out.second;
            // cout<<c<<endl;
            res += c;
            cnt--;
            // cout<<cnt<<endl;
            if (cnt > 0)    pq.push({cnt, c});
            
            for (auto t : temp)
                pq.push(t);
        }
        
        return res;
    }
    
private:
    bool valid(string res, char c) {
        if (res.length() < 2)    return 1;
        if (res[res.length() - 1] == c && res[res.length() - 2] == c)
            return 0;
        return 1;
    }
};
