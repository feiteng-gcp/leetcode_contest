class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res;
        while (a != 0 || b != 0 || c != 0) {
            int n = res.size();
            int forbidden = -1;
            if (res.size() >= 2 && res[n - 1] == res[n - 2]) {
                forbidden = res[n - 1] - 'a';
            }

            int index = findNext(a, b, c, forbidden);
            if (index == 0) {
                res += 'a';
                a -= 1;
            } else if (index == 1) {
                res += 'b';
                b -= 1;
            } else if (index == 2) {
                res += 'c';
                c -= 1;
            } else {
                return res;
            }
            // cout << res << ", "<< a << ", " << b << ", " << c << endl;
        }
        return res;
    }
    int findNext(int a, int b, int c, int forbidden) {
        if (forbidden == -1) {
            if (a >= b && a >= c) return 0;
            if (b >= a && b >= c) return 1;
            if (c >= a && c >= b) return 2;
        }
        if (forbidden == 0 && b == 0 && c == 0) return -1;
        if (forbidden == 1 && a == 0 && c == 0) return -1;
        if (forbidden == 2 && a == 0 && b == 0) return -1;
        if (forbidden == 0) {
            if (b >= c) {
                return 1;
            } else {
                return 2;
            }
        } else if (forbidden == 1) {
            if (a >= c) {
                return 0;
            } else {
                return 2;
            }
        } else {
            if (a >= b) {
                return 0;
            } else {
                return 1;
            }
        }
        return 0;
    }
};
