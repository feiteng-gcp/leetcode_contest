class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ans = "";
        int a_long = 0;
        int b_long = 0;
        int c_long = 0;
        
        string before = "";
        
        while(true)
        {
            before = ans;
            
            if(a_long == 2)
            {
                if(b >= c && b > 0)
                {
                        ans += "b";
                        b-=1;
                        a_long = 0;
                        b_long++;
                        c_long = 0;
                }
                else if(c >= b && c > 0)
                {
                        ans += "c";
                        c-=1;
                        a_long = 0;
                        b_long = 0;
                        c_long++;
                }
            }
            else if(b_long == 2)
            {
                if(a >= c && a > 0)
                {
                        ans += "a";
                        a-=1;
                        a_long++;
                        b_long = 0;
                        c_long = 0;
                }
                else if(c >= a && c > 0)
                {
                        ans += "c";
                        c-=1;
                        a_long = 0;
                        b_long = 0;
                        c_long++;
                }
            }
            else if(c_long == 2)
            {
                if(a >= b && a > 0)
                {
                        ans += "a";
                        a-=1;
                        a_long++;
                        b_long = 0;
                        c_long = 0;
                }
                
                else if(b >= a && b > 0)
                {
                        ans += "b";
                        b-=1;
                        a_long = 0;
                        b_long++;
                        c_long = 0;
                }
            }
            else
            {
                if(a >= b && a >= c && a > 0)
                {
                    ans += "a";
                    a-=1;
                    a_long++;
                    b_long = 0;
                    c_long = 0;
                }
                else if(b >= a && b >= c && b > 0)
                {
                    ans += "b";
                    b -= 1;
                    b_long++;
                    a_long = 0;
                    c_long = 0;
                }
                else if(c >= a && c >= b && c > 0)
                {
                    ans += "c";
                    c -= 1;
                    a_long = 0;
                    b_long= 0;
                    c_long++;
                }
            }
            if(before == ans) break;
        }
        
        return ans;
    }
};
