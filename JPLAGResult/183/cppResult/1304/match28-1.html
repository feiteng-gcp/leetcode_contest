class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res;
        vector<pair<int,char>> t={{a,'a'},{b,'b'},{c,'c'}};
        sort(t.begin(),t.end());
        int s=t[0].first+t[1].first;
        t[2].first=min(t[2].first,2+2*s);

        while(t[2].first>2){
            res+=string(2,t[2].second);
            t[2].first-=2;
            if(t[1].first>t[0].first){
                res+=t[1].second;
                --t[1].first;
            }
            else{
                res+=t[0].second;
                --t[0].first;
            }
            while(t[1].first>t[2].first){
                res+=t[1].second;
                --t[1].first;
            }
            while(t[0].first>t[1].first){
                res+=t[0].second;
                --t[0].first;
            }
        }
        res+=string(t[2].first,t[2].second);
        res+=string(t[1].first,t[1].second);
        res+=string(t[0].first,t[0].second);
        return res;
    }
};
