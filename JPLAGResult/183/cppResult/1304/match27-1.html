class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res = "";
        while (a+b+c) {
            if (a >= b && a >= c) {
                if (a == 1 || res.size() > 0 && res[res.size()-1] == 'a') {
                    res += 'a';
                    --a;
                    continue;
                }
                res += "aa";
                a -= 2;
                if (b == 0 && c == 0) {
                    break;
                }

                if (b >= c) {
                    res += 'b';
                    --b;
                } else {
                    res += 'c';
                    --c;
                }
            } else if (b >= a && b >= c) {
                if (b == 1 || res.size() > 0 && res[res.size()-1] == 'b') {
                    res += 'b';
                    --b;
                    continue;
                }
                res += "bb";
                b -= 2;
                if (a == 0 && c == 0) {
                    break;
                }

                if (a >= c) {
                    res += 'a';
                    --a;
                } else {
                    res += 'c';
                    --c;
                }
            } else {
                if (c == 1 || res.size() > 0 && res[res.size()-1] == 'c') {
                    res += 'c';
                    --c;
                    continue;
                }
                res += "cc";
                c -= 2;
                if (b == 0 && a == 0) {
                    break;
                }

                if (b >= a) {
                    res += 'b';
                    --b;
                } else {
                    res += 'a';
                    --a;
                }
            }
        }
        return res;
    }
};
