class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ans = "";
        while (a > 0 || b > 0 || c > 0) {
            if (a > 0 && a >= b && a >= c && (ans.size() < 2 || ans.back() != 'a' || ans[ans.size() - 2] != 'a')) {
                ans.push_back('a');
                a--;
            }
            else if (b > 0 && b >= a && b >= c && (ans.size() < 2 || ans.back() != 'b' || ans[ans.size() - 2] != 'b')) {
                ans.push_back('b');
                b--;
            }
            else if (c > 0 && c >= a && c >= b && (ans.size() < 2 || ans.back() != 'c' || ans[ans.size() - 2] != 'c')){
                ans.push_back('c');
                c--;
            }
            else if (ans.back() == 'a' && ans[ans.size() - 2] == 'a') {
                if (b > 0 && b >= c) {
                    ans.push_back('b');
                    b--;
                }
                else if (c > 0) {
                    ans.push_back('c');
                    c--;
                }
                else break;
            }
            else if (ans.back() == 'b' && ans[ans.size() - 2] == 'b') {
                if (a > 0 && a >= c) {
                    ans.push_back('a');
                    a--;
                }
                else if (c > 0) {
                    ans.push_back('c');
                    c--;
                }
                else break;
            }
            else if (ans.back() == 'c' && ans[ans.size() - 2] == 'c') {
                if (a > 0 && a >= b) {
                    ans.push_back('a');
                    a--;
                }
                else if (b > 0) {
                    ans.push_back('b');
                    b--;
                }
                else break;
            }
            else {
                break;
            }
        }
        return ans;
    }
};
