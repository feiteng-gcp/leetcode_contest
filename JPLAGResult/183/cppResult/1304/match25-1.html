class Solution {
public:
    
    //return lognest string over alphabet (a,b,c) withuot "aaa" "bbb" or "ccc" and at most a a, b b, c c
    string longestDiverseString(int a, int b, int c) {
        string res = "";
        if (a > b and a > c) {
            res.push_back('a');
            a--;
        } else if (b > c) {
            res.push_back('b');
            b--;
        } else {
            res.push_back('c');
            c--;
        }
        return aux(a,b,c,res);
    }
    
    bool canAdd(const string& res, char c) {
        int n = res.size();
        char l = res[n-1];
        if (l != c) return true;
        if (n == 1) return true;
        if (res[n-2] != c) return true;
        return false;
    }

    string aux(int a, int b, int c, string& res) {
        while (true) {
            // cout << a << b << c <<endl;
            while (true) {
                // cout << a << b << c << " 2" <<endl;

                while (true) {
                    // cout << a << b << c << " 3" <<endl;
                    if (a > 0 and (a >= b and a >= c and canAdd(res, 'a'))) {
                        res.push_back('a');
                        a--;
                    }
                    else if (b > 0 and (b >= a and b >= c and canAdd(res, 'b'))) {
                        res.push_back('b');
                        b--;
                    }
                    else if (c > 0 and (c >= b and c >= a and canAdd(res, 'c'))) {
                        res.push_back('c');
                        c--;
                    } else break;
                }
                if (a > 0 and ((a >= b or a >= c) and canAdd(res, 'a'))) {
                    res.push_back('a');
                    a--;
                }

                else if (b > 0 and ((b >= a or b >= c) and canAdd(res, 'b'))) {
                    res.push_back('b');
                    b--;
                }
                else if (c > 0 and ((c >= b or c >= a) and canAdd(res, 'c'))) {
                    res.push_back('c');
                    c--;
                } else break;
            }
            if (a > 0 and (canAdd(res, 'a'))) {
                res.push_back('a');
                a--;
            }

            else if (b > 0 and (canAdd(res, 'b'))) {
                res.push_back('b');
                b--;
            }
            else if (c > 0 and (canAdd(res, 'c'))) {
                res.push_back('c');
                c--;
            } else break;
        }
        return res;
    }
};
