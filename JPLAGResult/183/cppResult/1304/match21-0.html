class Solution {
public:
  string longestDiverseString(int a, int b, int c) {
    priority_queue<pair<int, char>> pq;
    if (a > 0)
      pq.push({a, 'a'});
    if (b > 0)
      pq.push({b, 'b'});
    if (c > 0)
      pq.push({c, 'c'});
    string res = "";
    while (!pq.empty()) {
      vector<int> freqs(3, 0);
      auto [freq, ch] = pq.top();
      pq.pop();
      int n = res.size();
      if (n >= 2 && res[n - 1] == ch && res[n - 2] == ch) {
        if (pq.empty()) {
          break;
        } else {
          auto [freq2, ch2] = pq.top();
          pq.pop();
          res += ch2;
          if (freq2 != 1) {
            pq.push({freq2 - 1, ch2});
          }
          pq.push({freq, ch});
        }
      } else {
        res += ch;
        if (freq != 1)
          pq.push({freq - 1, ch});
      }
    }
    return res;
  }
};
