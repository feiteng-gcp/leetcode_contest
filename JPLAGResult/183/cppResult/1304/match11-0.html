class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string result;
        if(a>=(b+c)*2+2){
            a=(b+c)*2+2;
        }
        if(b>=(a+c)*2+2){
            b=(a+c)*2+2;
        }
        if(c>=(a+b)*2+2){
            c=(a+b)*2+2;
        }
        int ac=0,bc=0,cc=0;
        while(a+b+c>0){
            if(ac+bc+cc<2){
                if(a>b){
                    if(a>c){
                        result+='a';
                        a--;
                        ac++;bc=0;cc=0;
                    }else{
                        result+='c';
                        c--;
                        cc++;ac=0;bc=0;
                    }
                }else{
                    if(b>c){
                        result+='b';
                        b--;
                        bc++;ac=0;cc=0;
                    }else{
                        result+='c';
                        c--;
                        cc++;ac=0;bc=0;
                    }
                }
            }else{
                if(ac==2){
                    if(b>c){
                        result+='b';
                        b--;
                        bc++;ac=0;cc=0;
                    }else{
                        result+='c';
                        c--;
                        cc++;ac=0;bc=0;
                    }
                }else if(bc==2){
                    if(a>c){
                        result+='a';
                        a--;
                        ac++;bc=0;cc=0;
                    }else{
                        result+='c';
                        c--;
                        cc++;ac=0;bc=0;
                    }
                }else if(cc==2){
                    if(a>b){
                        result+='a';
                        a--;
                        ac++;bc=0;cc=0;
                    }else{
                        result+='b';
                        b--;
                        bc++;ac=0;cc=0;
                    }
                }
            }
        }
        return result;
    }
};
