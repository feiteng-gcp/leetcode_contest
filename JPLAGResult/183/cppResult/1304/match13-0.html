class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int,char>>pq;
        if(a)
        pq.push({a, 'a'});
        if(b)
        pq.push({b, 'b'});
        if(c)
        pq.push({c, 'c'});
        string ans;
        while(!pq.empty())
        {
            pair<int,char>temp = pq.top();
            pq.pop();
            int n = ans.size();
            if(n >= 2 && ans[n-1]==temp.second && ans[n-2] == temp.second)
            {
                if(pq.empty())
                    break;
                 pair<int,char>temp2 = pq.top();
                 pq.pop();
                pq.push(temp);
                ans.push_back(temp2.second);
                temp2.first--;
                if(temp2.first)
                    pq.push({temp2.first,temp2.second});
                 
                
            }
            else
            {
                ans.push_back(temp.second);
                temp.first--;
                if(temp.first)
                    pq.push({temp.first,temp.second});
            }
        }
        return ans;
        
    }
};
