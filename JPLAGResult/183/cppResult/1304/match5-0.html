class Solution {
public:
    bool insert(string &s, char a){
        int ls = s.size();
        for(int i=1; i<ls; i++){
            if(s[i]!=a && s[i-1]!=a){
                s.insert(s.begin() + i, a);
                return true;
            }

            if(s[i]==a && s[i-1]!=a){
                if(i==ls-1 || s[i+1]!=a){
                    s.insert(s.begin() + i, a);
                    return true;
                }
            }

            if(s[i]!=a && s[i-1]==a){
                if(i==1 || s[i-2]!=a){
                    s.insert(s.begin() + i, a);
                    return true;
                }
            }
        }
        if(s[0] != a || ls == 1 || s[1] != a){
            s.insert(s.begin(), a);
            return true;
        }

        if(s[ls-1] != a || ls == 1 || s[ls-2] !=a){
            s.insert(s.end(), a);
            return true;
        }

        return false;
    }

    string longestDiverseString(int a, int b, int c) {
        string res = "";
        bool f = true;
        while(f){
            f = false;
            if(a>0 && insert(res, 'a')){
                f = true;
                a--;
            } 

            if(b>0 && insert(res, 'b')) {
                f = true;
                b--;
            }

            if(c>0 && insert(res, 'c')) {
                f = true;
                c--;
            }
        }

        return res;
        
    }
};
