class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        stoneValue += [0] * 3
        dp, s = [0] * (n + 3), [0] * (n + 3)

        s[n - 1] = stoneValue[n - 1]
        for i in range(n - 2, -1, -1):
            s[i] = s[i + 1] + stoneValue[i]
        # print(s)

        for i in range(n - 1, -1, -1):
            dp[i] = float('-inf')
            for j in range(3):
                dp[i] = max(dp[i], sum(stoneValue[i: i + j + 1]) + s[i + j + 1] - dp[
i + j + 1])
        # print(dp)
        if dp[0] > s[0] - dp[0]:
            return 'Alice'
        elif dp[0] == s[0] - dp[0]:
            return 'Tie'
        else:
            return 'Bob'
