from functools import lru_cache
import sys
sys.setrecursionlimit(500001)
class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        
        n = len(stoneValue)
        @lru_cache(None)
        def expected(i):
            if i>= n:
                return 0
            res = - 1000000000
            if i<n:
                res = max(res, stoneValue[i] - expected(i+1))
            if i<n-1:
                res = max(res, stoneValue[i]+stoneValue[i+1] - expected(i+2))
            if i<n-2:
                res = max(res, stoneValue[i]+stoneValue[i+1]+stoneValue[i+2] - expected(
i+3))
            return res
        res = expected(0)
        if res>0:
            return 'Alice'
        elif res<0:
            return 'Bob'
        else:
            return 'Tie'
