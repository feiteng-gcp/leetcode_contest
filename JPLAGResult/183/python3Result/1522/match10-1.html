class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        arr = [[0, 0], [0, 0], [0, 0]]
        stoneValue.extend([0, 0])
        for i in range(len(stoneValue) - 3, -1, -1):
            diff1 = stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] - arr
[2][0] + arr[2][1]
            diff2 = stoneValue[i] + stoneValue[i + 1] - arr[1][0] + arr[1][1]
            diff3 = stoneValue[i] - arr[0][0] + arr[0][1]
            
            if diff1 >= diff2 and diff1 >= diff3:
                arr = [[stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] + arr
[2][1], arr[2][0]]] + arr
            elif diff2 >= diff3:
                arr = [[stoneValue[i] + stoneValue[i + 1] + arr[1][1], arr[1][0]]] + arr
            else:
                arr = [[stoneValue[i] + arr[0][1], arr[0][0]]] + arr
            arr.pop()
        
        diff = arr[0][0] - arr[0][1]
        if diff > 0:
            return 'Alice'
        elif diff < 0:
            return 'Bob'
        else:
            return 'Tie'
