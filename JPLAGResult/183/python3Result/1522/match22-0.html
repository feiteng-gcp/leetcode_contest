class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        seen = {}

        def maxDiff(idx):
            if idx == len(stoneValue):
                return 0
            if idx in seen:
                return seen[idx]
            
            myStone = 0
            res = float('-inf')
            for i in range(idx, min(idx+3, len(stoneValue))):
                myStone += stoneValue[i]
                res = max(res, myStone - maxDiff(i+1))
            seen[idx] = res
            return res
        
        res = maxDiff(0)
        if res > 0:
            return 'Alice'
        elif res < 0:
            return 'Bob'
        return 'Tie'
