class Solution:
    def stoneGameIII(self, s: List[int]) -> str:
        n = len(s)
        dp = [float('-inf')] * (n + 1)
        dp[-1] = 0
        for i in range(n - 1, -1, -1):
            t 
= 0
            for j in range(1, 4):
                if i + j > n:
                    break
                t += s[i + j - 1]
                dp[i] = max(dp[i], t - dp[i + j])
        return 'Alice' if dp[0] > 0 else 'Tie' if dp[0] == 0 else 'Bob'
