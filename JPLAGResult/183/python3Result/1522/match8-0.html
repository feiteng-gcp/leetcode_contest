class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        dp = {}
        def count(i):
            if i >= len(stoneValue):
                return 0
            if i in dp:
                return dp[i]
            m = -float('inf')
            for j in range(3):
                m = max(m, sum(stoneValue[i: i + j + 1]) - count(i + j + 1))
            dp[i] = m
            return m
        res = count(0)
        return "Alice" if res > 0 else "Bob" if res < 0 else "Tie"
