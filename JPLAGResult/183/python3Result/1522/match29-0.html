class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        # dp
        n = len(stoneValue)
        dp = [-float('inf') for _ in range(n)] + [0]  # dp[i] is optimal value when one starts pick from index i
        
        # cumulative sum from i to end
        right_sum = [0 for _ in range(n+1)]
        tmp = 0
        for i in range(n-1,-1,-1):
            tmp += stoneValue[i]
            right_sum[i] = tmp
            
        # loop
        for i in range(n-1,-1,-1):
            cur = 0
            for k in range(3):
                if i+k < n:
                    cur += stoneValue[i+k]
                    dp[i] = max(cur + right_sum[i+k+1] - dp[i+k+1], dp[i])
                    
        if dp[0] > right_sum[0] - dp[0]: return 'Alice'
        elif dp[0] == right_sum[0] -dp[0]: return 'Tie'
        else: return 'Bob'
