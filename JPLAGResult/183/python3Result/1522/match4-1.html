class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        import functools
        @functools.lru_cache(None)
        def dp(k):
            if k>=len(stoneValue):
                return 0
            res = -float('inf')
            tmp = 0
            for i in range(k,min(len(stoneValue),k+3)):
                tmp += stoneValue[i]
                res = max(res,tmp-dp(
i+1))
            return res
        res = dp(0)
        if res>0:
            return 'Alice'
        elif res == 0:
            return 'Tie'
        else:
            return 'Bob'
