class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        dp = [[0] * (n+1) for i in range(2)]
        # dp[i] = max(arr[i]-dp[i+1], sum(arr[i:i+2]), dp[i+2], sum(arr[i:i+3]), dp[i+3])
        for i in range(n)[::-1]:
            #if i == n-1:
            #    dp[0][i] = dp[1][i] = stoneValue[-1]
            #    print('dp i', i, ' = ', dp[0][i])
            #    continue
            for j in range(2):
                dp[j][i] = stoneValue[i]-dp[1-j][i+1]
                if i + 2 <= n:
                    dp[j][i] = max(dp[j][i], sum(stoneValue[i:i+2])-dp[1-j][i+2])
                if i + 3 <= n:
                    dp[j][i] = max(dp[j][i], sum(stoneValue[i:i+3])-dp[1-j][i+3])
                # print('dp i', i, ' = ', dp[j][i])
        if dp[0][0] > 0:
            return "Alice"
        elif dp[0][0] < 0:
            return "Bob"
        return "Tie"
