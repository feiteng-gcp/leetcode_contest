class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        s = stoneValue
        n = len(s)
        f = [0] * (n + 1)
        #g = [0] * n
        total = 0
        for i in range(n-1, -1, -1) :
            total += s[i]
            f[i] = total - f[i+1]
            for j in range(1, 3) :
                k = i + j + 1
                if k <= n :
                    f[i] = max(f[i], total - f[k])
        delta = f[0] - (total - f[0])
        if delta > 0 :
            return 'Alice'
        elif delta < 0 :
            return 'Bob'
        else :
            return 'Tie'
