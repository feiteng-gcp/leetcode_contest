class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        stoneValue.append(0)
        stoneValue.append(0)
        stoneValue.append(0)
        stoneValue.append(0)
        ans = [0 for i in range(n + 10)]
        for i in range(n - 1, -1, -1):
            t1 = stoneValue[i] - ans[i + 1]
            t2 = stoneValue[i] + stoneValue[i + 1] - ans[i + 2]
            t3 = stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] - ans[i + 3]
            ans[i] = max
(t1, max(t2, t3))
        if ans[0] < 0:
            return 'Bob'
        elif ans[0] == 0:
            return 'Tie'
        else:
            return 'Alice'
