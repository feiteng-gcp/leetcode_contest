class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        #dp: how many scores I can get mox starting at position i
        dp = [0] * (len(stoneValue) + 1)
        
        for pos in range(len(stoneValue) - 1, -1, -1):
            cur = 0
            res = -100000000000
            for i in range(pos, min(pos + 3, len(stoneValue))):
                cur += stoneValue[i]
                res = max(res, cur - dp[i+1])
            dp[pos] = res
        
        # print(dp)
        if dp[0] > 0:
            return "Alice"
        elif dp[0] == 0:
            return "Tie"
        else:
            return "Bob"
