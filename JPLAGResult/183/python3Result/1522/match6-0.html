class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        d = {len(stoneValue): 0}
        for i in reversed(range(len(stoneValue))):
            d[i] = max(sum(stoneValue[i: i + j + 1]) - d[i + j + 1] for j in range(min(3, len(stoneValue) - i)))
        if d[0] < 0:
            return "Bob"
        elif 0 < d[0]:
            return "Alice"
        else:
            return "Tie"
