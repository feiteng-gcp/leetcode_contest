class Solution:
    def stoneGameIII(self, s: List[int]) -> str:
        l = len(s)
        x = [0] * (l+1)
        import math
        for i in range(l):
            j = l-i-1
            a = 0
            m = -math.inf
            for k in range(1, 4):
                if j+k > l:
                    break
                a 
+= s[j+k-1]
                m = max(m, a-x[j+k])
            x[j] = m
        # print(x)
        if x[0] > 0:
            return "Alice"
        if x[0] < 0:
            return "Bob"
        return "Tie"
