class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        stoneValue+=[0,0,0]
        n=len(stoneValue)
        dp=[[0]*2 for _ in range(n)]
        for i in range(n-4,-1,-1):
            a=dp[i+1][0]+stoneValue[i]
            b=dp[i+2][0]+stoneValue[i]+stoneValue[i+1]
            c=dp[i+3][0]+stoneValue[i]+stoneValue[i+1]+stoneValue[i+2]
            dp[i][1]=max(a,b,c)
            if dp[i][1]==a:
                dp[i][0]=dp[i+1][1]
            elif dp[i][1]==b:
                dp[i][0]=dp[i+2][1]
            else:
                dp[i][0]=dp[i+3][1]
        num1=dp[0][1]
        num2=sum(stoneValue)-num1
        if num1>num2:   return "Alice"
        elif num1<num2:  return "Bob"
        else:    return "Tie"
