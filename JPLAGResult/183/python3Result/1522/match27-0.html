from functools import lru_cache
class Solution:
    def stoneGameIII(self, s: List[int]) -> str:
        n = len(s)
        su = sum(s)
        pre = [0]
        for x in s:
            pre.append(pre[-1] + x)
        @lru_cache(None)
        def helper(i = 0):
            if i == n:
                return 0
            ans = -math.inf
            for j in range(3):
                if i + j <= n - 1:
                    ans = max(ans, sum(s[i:i + j +1]) + su - pre[i + j+1] - helper(i + j + 1))
            return ans
        a = helper()
        if 2*a > su:
            return 'Alice'
        elif 2*a == su:
            return 'Tie'
        return 'Bob'
