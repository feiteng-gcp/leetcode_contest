class Solution:
    def stoneGameIII(self, stones: List[int]) -> str:
        total = sum(stones)
        @functools.lru_cache(None)
        def helper(idx):
            if idx >= len(stones):
                return 0
            ans = float("-inf")
            for le in range(1, 4):
                cur_sum = sum(stones[idx: idx + le])
                ma = helper(idx + le)
                ans = max(
cur_sum - ma, ans)
            return ans
        ans = helper(0)
        if ans > 0: return "Alice"
        elif ans == 0: return "Tie"
        return "Bob"
