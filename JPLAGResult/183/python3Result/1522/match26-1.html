import functools
class Solution:
    '''
    f(0,i) = max{
      A[0], sum(A[1:]-f(1,i))  get 1
      A[0,1], 
      A[]
    }
    '''
    
    def stoneGameIII(self, A: [int]) -> str:
        n = len(A)
        cum = []
        for x in A[::-1]:
            cum.append(x if not cum else x+cum[-1])
        cum = cum[::-1]
        @lru_cache
        def g(x):
            return cum[x]
        @lru_cache
        def f(x):
            if x>= n: return 0
            if x == n-1: return A[-1]
            else:
                return max(g(x)-f(x+1),
                          g(x)-f(x+2),
                           g(x)-f(x
+3),
                          )
        total = sum(A)
        gain = f(0)
        if gain == total / 2:
            return "Tie"
        if gain > total /2:
            return "Alice"
        return "Bob"
