class Solution:
    def stoneGameIII(self, stoneValue) -> str:
        n = len(stoneValue)
        dp = [None] * (n+1)
        dp[n] = 0

        for i in range(n-1, -1, -1):
            s = 0
            dp[i] = -10**10
            for k in range(3):
                if i + k >= n:
                    break
                s += stoneValue[i+k]
                    
                dp[i] = max(dp[i], s - dp[i+k+1])
        
        if dp[0] > 0:
            return "Alice"
        if dp[0] == 0:
            return 'Tie'
        if dp[0] < 0:
            return 'Bob'
