class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        
        @functools.lru_cache(None)
        def check_first_best(start_pos=0) :
            if start_pos >= n :
                return 0, 0
            if start_pos == n-1 :
                return stoneValue[-1], 0
            
            max_receive = -1e9
            other_receive = None
            for i in range(1, min(3, n-start_pos)+1) :
                now_take = sum(stoneValue[start_pos:start_pos+i])
                other, mine = check_first_best(start_pos+i)
                if mine + now_take > max_receive :
                    max_receive = mine + now_take
                    other_receive = other
            return max_receive, other_receive
        
        # print(check_first_best(n-1))
        alice, bob = check_first_best()
        # print(alice, bob)
        if alice > bob :
            return 'Alice'
        elif alice < bob :
            return 'Bob'
        else :
            return 'Tie'
