class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        l = len(stoneValue)
        stoneValue.reverse()
        mx = [None] * l
        sm = 0

        for i in range(0, l):
            sm += stoneValue[i]
            for j in range(3):
                if i - j < 0: continue

                prev_mx = 0 if i-j-1<0 else mx[i-j-1]
                cur_mx = sm - prev_mx
                if mx[i] is None or cur_mx > mx[i]: mx[i] =
 cur_mx

        res = mx[-1] - (sm - mx[-1])
        if res > 0:
            return 'Alice'
        elif res == 0:
            return 'Tie'
        else:
            return 'Bob'
