class Solution:
    def stoneGameIII(self, nums: List[int]) -> str:
        # m = [0] * (len(nums) ** 2)
        m = {}
        
        def getScore(nums,l,r): #获得我和对手的最大分数差
            if l > r: return 0
            if l == r: return nums[l] #当数组只有一个数的时候，分数差为这个数
            
            if (l , r) in m: return m[(l, r)] #如果这个组合以及出现过的话，直接调用，剪枝
            
            first = nums[l] - getScore(nums,l + 1,r)
            second = nums[l] + nums[l + 1] - getScore(nums, l + 2, r) if l + 1 <= r else -float('inf')
            third = nums[l] + nums[l + 1] + nums[l + 2] - getScore(nums, l + 3, r) if l + 2 <= r else -float('inf')
            
            m[(l , r)] = max(first, second, third) #我所选的数减去对手所能选的最大数 中取最大值，在递归过程中会不断的换边
            return m[(l ,r)]

        if getScore(nums,0,len(nums)-1) >0:
            return "Alice"
        elif getScore(nums,0,len(nums)-1) == 0:
            return "Tie"
        else:
            return "Bob"
