class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        
        @functools.lru_cache(None)
        def dp(index):
            if index == len(stoneValue):
                return 0
            res = []
            res.append(stoneValue[index] - dp(index+1))
            if index <= len(stoneValue)-2:
                res.append(stoneValue[index] + stoneValue[index+1] - dp(index+2))
            if index <= len(stoneValue)-3:
                res.append(stoneValue[index] + stoneValue[index+1] + stoneValue[index+2] - dp(index+3))
            return max(res)
            
            
        res = dp(0)
        if res == 0:
            return 'Tie'
        if res > 0:
            return "Alice"
        return 'Bob'
