class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        s = list(reversed(stoneValue))
        f = [0] * n
        g = [0] * n
        f[0] = s[0]
        g[0] = 0
        
        if n > 1:
            f[1] = max(s[0] + s[1], s[1])
            g[1] = min(0, f[0])
        
        if n > 2:
            f[2] = max(s[0] + s[1] + s[2], g[0] + s[1] + s[2], g[1] + s[2])
            g[2] = min(0, f[0], f[1])
        for i in range(3, n):
            f[i] 
= max(s[i] + g[i - 1], s[i] + s[i - 1] + g[i - 2], s[i] + s[i - 1] + s[i - 2] + g[i - 3])
            g[i] = min(f[i - 1], f[i - 2], f[i - 3])
        sst = sum(s)
        sa = f[-1]
        sb = sst - sa
        if sa == sb:
            return 'Tie'
        elif sa > sb:
            return 'Alice'
        else:
            return 'Bob'
