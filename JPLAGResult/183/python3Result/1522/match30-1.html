import functools

class Solution:
	def stoneGameIII(self, stoneValue: [int]) -> str:
		pf = [stoneValue[0]] * len(stoneValue)
		for i in range(1, len(stoneValue)):
			pf[i] = pf[i-1] + stoneValue[i]

		def calc(start: int, end: int) -> int:
			if start == 0:
				return pf[end]
			return pf[end]-pf[start-1]
		
		@functools.lru_cache(None)
		def rec(start: int) -> int: 
			if start >= len(stoneValue):
				return 0
			if start == len(stoneValue)-1:
				return stoneValue[
start]
			
			ans = float('-inf')
			sm = 0
			for i in range(start, min(start+3, len(stoneValue))):
				sm += stoneValue[i]
				sumRemaining = calc(i+1, len(stoneValue)-1)
				bob = rec(i+1)
				ans = max(ans, sm + (sumRemaining - bob))
			return ans
		
		alice = rec(0)
		if alice > pf[-1] - alice:
			return "Alice"
		elif alice == pf[-1] - alice:
			return "Tie"
		return 'Bob'
