class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        nums.sort(reverse=True)
        half = sum(nums) // 2
        res = []
        cur = 0
        for num in nums:
            cur += num
            res.append(num)
            if cur > half:
                return res
        return []
