class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        if len(nums) == 1:
            return nums
        
        nums
.sort(reverse=True)
        
        s = sum(nums)
        
        cur = 0
        for i in range(len(nums)):
            cur += nums[i]
            
            if cur > s - cur:
                return nums[:i+1]
