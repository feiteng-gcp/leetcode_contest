class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        ans = ''
        pos = [[a, 'a'], [b, 'b'], [c, 'c']]
        while pos[0][0] + pos[1][0] + pos[2][0] > 0:
            pos = sorted(pos, reverse = True, key = lambda x : (x[0], -ord(x[1])))
            if len(ans) < 2 or ans[-2] != ans[-1] or ans[-1] != pos
[0][1]:
                ans += pos[0][1]
                pos[0][0] -= 1
            elif pos[1][0] > 0:
                ans += pos[1][1]
                pos[1][0] -= 1
            else:
                break
        return ans
