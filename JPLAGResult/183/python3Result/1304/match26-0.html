class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        cs = [[a, 'a'], [b, 'b'], [c, 'c']]
        ans = ""
        while True:
            cs.sort(reverse=True)
            max_char_count, max_char = cs[0]
            if max_char_count == 0:
                break
            if len(ans) == 0 or ans[-1] != max_char:
                ans += max_char * min(2, max_char_count)
                cs[0][0] -= min(2, max_char_count)
                continue
            else:
                sub_max_char_count, sub_max_char = cs[1]
                if sub_max_char_count == 0:
                    break
                ans += sub_max_char * 1
                cs[1][0] -= 1
        return ans
