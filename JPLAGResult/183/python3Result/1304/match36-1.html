class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        res = ''
        last = {'a': 0, 'b': 0, 'c': 0}
        can = [[a, 'a'], [b, 'b'], [c, 'c']]
        while True:
            can.sort(reverse=True)
            if last[can[0][1]] < 2:
                if can[0][0] == 0:
                    break
                last
[can[0][1]] += 1
                last[can[1][1]] = last[can[2][1]] = 0
                res += can[0][1]
                can[0][0] -= 1
            elif last[can[1][1]] < 2:
                if can[1][0] == 0:
                    break
                last
[can[1][1]] += 1
                last[can[0][1]] = last[can[2][1]] = 0
                res += can[1][1]
                can[1][0] -= 1
            elif last[can[2][1]] < 2:
                if can[2][0] == 0:
                    break
                last[can[2][1]] += 1
                last[can[0][1]] = last[can[1][1]] = 0
                res += can[2][1]
                can[2][0] -= 1
            else:
                break
        return res
