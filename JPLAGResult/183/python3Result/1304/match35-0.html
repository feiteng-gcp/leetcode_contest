from heapq import heappop, heappush
class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        s = '' 
        h = []
        heappush(h,(-a,'a'))
        heappush(h,(-b,'b'))
        heappush(h,(-c,'c'))
        while True:
            n, c = heappop(h)
            if s.endswith(c*2):
                nn, cc = heappop(h)
                if nn < 0:
                    s += cc
                    nn += 1
                    heappush(h,(nn,cc))
                    heappush(h,(n,c))
                else:
                    return s
            elif n < 0:
                s += c
                n += 1
                heappush(h,(n,c))
            else:
                return s
