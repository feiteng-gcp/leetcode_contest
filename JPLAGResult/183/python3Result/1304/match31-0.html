class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        ans = ''
        k = [[a, 'a'], [b, 'b'], [c, 'c']]
        k.sort()
        k.reverse()
        while k[0][0] > 0:
            if k[0][0] >= 2:
                ans += k[0][1] * 2
                k[0][0] -= 2
            else:
                ans += k[0][1]
                k[0][0] -= 1
            if k[1][0] > 0:
                ans += k[1][1]
                k[1][0] -= 1
            elif k[2][0] > 0:
                ans += k[2][1]
                k[2][0] -= 1
            else:
                break
        while k[1][0] > 0:
            flag = False
            for i in range(0, len(ans)):
                if i >= 1 and ans[i - 1] == k[1][1] and ans[i] == k[1][1]:
                    continue
                elif i >= 2 and ans[i - 1] == k[1][1] and ans[i - 2] == k[1][1]:
                    continue
                elif i < len(ans) - 1 and ans[i] == k[1][1] and ans[i + 1] == k[1][1]:
                    continue
                else:
                    flag = True
                    k[1][0] -= 1
                    ans = ans[:i] + k[1][1] + ans[i:]
                    break
            if not flag:
                break
        
        while k[2][0] > 0:
            flag = False
            for i in range(0, len(ans)):
                if i >= 1 and ans[i - 1] == k[2][1] and ans[i] == k[2][1]:
                    continue
                elif i >= 2 and ans[i - 1] == k[2][1] and ans[i - 2] == k[2][1]:
                    continue
                elif i < len(ans) - 1 and ans[i] == k[2][1] and ans[i + 1] == k[2][1]:
                    continue
                else:
                    flag = True
                    k[2][0] -= 1
                    ans = ans[:i] + k[2][1] + ans[i:]
                    break
            if not flag:
                break
                
        return ans
