class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        def valid(res, l):
            if len(res) <= 1:
                return True
            if res[-1] != l or res[-2] != l:
                return True
            return False
        res = ""
        arr = [[a,'a'], [b,'b'],[c,'c']]
        while (arr[0][0] + arr[1][0] + arr[2][0]):
            arr.sort()
            if valid(res, arr[-1][1]):
                res += arr[-1][1]
                arr[-1][0] -= 1
            else:
                if arr[1][0] > 0:
                    res += arr[1][1]
                    arr[1][0] -= 1
                else:
                    return res
        return res
