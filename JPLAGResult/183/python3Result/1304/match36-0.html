class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        r = ""
        s = [0, 0, 0]
        n = [a, b, c]
        while sum(n) > 0:
            t = [(n[0], 'a', 0), (n[1], 'b', 1), (n[2], 'c', 2)]
            t.sort()
            
            if s[t[2][2]] < 2:
                if t[2][0] == 0:
                    return r
                r += t[2][1]
                n[t[2][2]] -= 1
                s[t[2][2]] += 1
                s[t[1][2]] = 0
                s[t[0][2]] = 0
            else:
                if t[1][0] == 0:
                    return r
                r += t[1][1]
                n[t[1][2]] -= 1
                s[t[1][2]] = 1
                s[t[0][2]] = 0
                s[t[2][2]] = 0
            
        return r
