class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        rr = [[a, 'a'], [b, 'b'],  [c, 'c']]
        s = []
        while sum(v[0] for v in rr) > 0:
            rr.sort(key=lambda v: -v[0])
            if len(s) > 1 and s[-1] == rr[0][1] and s[-2] == rr[0][1]:
                if rr[1][0] == 0:
                    break
                rr[1][0] -= 1
                s.append(rr[1][1])
            else:
                s.append(rr[0][1])
                rr[0][0] -= 1
        return ''.join(s)
