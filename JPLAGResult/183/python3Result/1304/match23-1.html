class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        pq 
= []
        if a > 0:
            pq.append([-a, 'a'])
        if b > 0:
            pq.append([-b, 'b'])
        if c > 0:
            pq.append([-c, 'c'])
        heapq.heapify(pq)
        
        res = []
        while pq:
            cnt, letter = heapq.heappop(pq)
            if len(res) < 2 or not (letter == res[-1] and letter == res[-2]):
                res.append(letter)
                if cnt < -1:
                    heapq.heappush(pq, [cnt + 1, letter])
            else:
                if not pq:
                    break
                cnt2, letter2 = heapq.heappop(pq)
                res.append(letter2)
                if cnt2 < -1:
                    heapq.heappush(pq, [cnt2 + 1, letter2])
                
                heapq.heappush(pq, [cnt, letter])
        
        return ''.join(res)
