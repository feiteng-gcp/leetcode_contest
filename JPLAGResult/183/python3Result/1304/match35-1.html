from heapq import heappush, heappop
class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        res = []
        h = []
        heappush(h, [-a,'a'])
        heappush(h, [-b,'b'])
        heappush(h, [-c,'c'])
        
        while len(res) == 0 or h[0][1] != res[-1]:
            count1, letter1 = heappop(h)
            count2, letter2 = heappop(h)
            if count1 == 0:
                break
            n = min(-count1,2)
            for _ in range(n):
                res.append(letter1)   
            count1 += n
            heappush(h,[count1, letter1])
            if count2 == 0:
                break
            if count2 < count1:
                heappush(h,[count2, letter2])
                continue
            count2 += 1
            res.append(letter2)
            heappush(h,[count2, letter2])
            
            
        return ''.join(res)
