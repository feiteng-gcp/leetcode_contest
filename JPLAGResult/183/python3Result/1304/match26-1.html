class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        cnt = [[a, 'a'], [b, 'b'], [c, 'c']]
        ans = ''
        while True:
            cnt.sort()
            n, ch = cnt[-1]
            if n == 0:
                break
            elif n == 1:
                ans += ch
                cnt[-1][0] -= 1
            else:
                ans += ch * 2
                cnt[-1][0] -= 2
            
            n1, ch1 = cnt[-2]
            if n1 == 0:
                break
            elif n1 >= max(2, cnt[-1][0]):
                ans += ch1 * 2
                cnt[-2][0] -= 2
            else:
                ans += ch1
                cnt[-2][0] -= 1
        return ans
