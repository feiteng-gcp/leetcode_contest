import heapq



class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        nums = [[a, 'a'], [b, 'b'], [c, 'c']]

        ans = [None]
        while True:
            nums.sort()
            m = nums[-1]

            if m[0] == 0:
                break

            if m[1] != ans[-1]:
                for _ in range(2):
                    if m[0] > 0:
                        ans.append(m[1])
                        m[0] -= 1
            else:
                m = nums[1]
                if m[0] == 0:
                    break

                ans.append(m[1])
                m[0] -= 1
        ans.pop(0)
        return ''.join(ans)
