class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        if (rooms.size() == 1) {
            return true;
        }
        ArrayList<Integer> que = new ArrayList<>();
        que.add(0);
        boolean[] visited = new boolean[rooms.size()];
        visited[0] = true;
        int op = 0;
        while (op < que.size()) {
            int k = que.get(op);
            op++;
            for (int i : rooms.get(k)) {
                if (!visited[i]) {
                    visited[i] = true;
                    que.add(i);
                    if (que.size() == rooms.size()) {
                        return true;
                    }
                }
            }
        }
        return false;
    }
}
