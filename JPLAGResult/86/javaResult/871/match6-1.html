class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
		boolean[] can = new boolean[rooms.size()];
		can[0] = true;
		LinkedList<Integer> q = new LinkedList<Integer>();
		q.add(0);
		while(!q.isEmpty()) {
			int curr = q.removeFirst();
			for(int out: rooms.get(curr)) {
				if(!can[out]) {
					can[out] = true;
					q.add(out);
				}
			}

		}
		for(boolean out: can) {
			if(!out) return false;
		}
		return true;
    }
}
