class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        int roomNum = rooms.size();
        Set<Integer> openedRoom = new HashSet<>();
        openedRoom.add(0);
        
        Queue<Integer> keys = new LinkedList<>();
        keys.add(0);
        while (!keys.isEmpty()) {
            int size = keys.size();
            for (int i =0; i < size; i++) {
                Integer room = keys.poll();
                for (Integer newKey : rooms.get(room)) {
                    
if (!openedRoom.contains(newKey)) {
                        openedRoom.add(newKey);
                        keys.add(newKey);
                    }
                }
            }
        }
        for (int i =0; i < roomNum; i++) 
{
            if (openedRoom.add(i)) {
                return false;
            }
        }
        return true;
    }
}
