class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
         int n = rooms.size();
        boolean[] visited = new boolean[n];
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(0);
        while (!queue.isEmpty()) {
            int room = queue.poll();
            visited[room] = true;
            List<Integer> keys = rooms.get(room);
            for (int key : keys) {
                if (visited[key]) continue;

                queue.offer(key);
            }
        }
        for (boolean v : visited) {
            if (!v) return false;
        }
        return true;       
    }
}
