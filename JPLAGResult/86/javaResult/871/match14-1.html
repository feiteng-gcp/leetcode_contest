class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        if(rooms.size()==0){
            return true;
        }   
        int n=rooms.size();
        boolean[] visit=new boolean[n];
        Queue<Integer> queue=new LinkedList<Integer>();
        queue.offer(0);
        //visit[0]=true;
        
while(!queue.isEmpty()){
            int index=queue.poll();
            visit[index]=true;
            for(int next:rooms.get(index)){
                if(!visit[next]){
                    queue.offer(next);
                }
            }
        }
        
        
for(boolean tag:visit){
            if(!tag){
                return false;
            }
        }
        
        return true;
    }
}
