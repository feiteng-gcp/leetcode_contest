class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        int N = rooms.size();
        if (N == 0) return true;
        
        boolean[] opened = new boolean[N];
        opened[0] = true;
        
        Queue<Integer> que = new LinkedList<>();
        que.offer(0);
        while (!que.isEmpty()) {
            int curr = que.poll();
            for (int key: rooms.get(curr)) {
                if (!opened[key]) {
                    opened[key] = true;
                    que.offer(key);
                }
            
}
        }
        
        for (int i = 0; i < N; i++) {
            if (!opened[i]) return false;
        }
        return true;
    }
}
