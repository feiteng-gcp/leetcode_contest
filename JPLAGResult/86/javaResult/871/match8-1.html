	class Solution {
	    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
	    	int n = rooms.size();
	    	Queue<Integer> q = new ArrayDeque<>();
	        boolean[] can = new boolean[n];
	        q.add(0);
	        can[0] = true;
	        while(!q.isEmpty()){
	        	int cur = q.poll();
	        	for(int x : rooms.get(cur)){
	        		if(!can[x]){
		        		can[x] = true;
		        		q.add(x);
	        		}
	        	}

	        }
	        for(int i = 0;i < n;i++){
	        	if(!can[i])return false;
	        }
	        return true;
	    }
	}
