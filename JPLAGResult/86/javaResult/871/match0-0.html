class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        if (rooms == null || rooms.size() == 0) {
            return false;
        }
        int len = rooms.size();
        boolean[] vis = new boolean[len];
        dfs(0, rooms, vis);
        for (int i = 0; i < len; i++) {
            if (!vis[i]) {
                return false;
            }
        }
        return true;
    }
    
    private void dfs(int root, List<List<Integer>> rooms, boolean[] vis) {
        vis[root] = true;
        for (int next : rooms.get(root)) {
            if (!vis[next]) {
                dfs(next, rooms, vis);
            }
        }
    }
}
