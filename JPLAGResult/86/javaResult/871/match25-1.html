class Solution {
    boolean visited[];
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        if (rooms == null) {
            return true;
        }
        int n = rooms.size();
        visited = new boolean[n];
        dfs(0, rooms);
        
for (boolean vis : visited)
            if (!vis)
                return false;
        return true;
    }
    private void dfs(int v, List<List<Integer>> g) {
        if (visited[v])
            return;
        
visited[v] = true;
        for (int next : g.get(v)) {
            if (!visited[next]) {
                dfs(next, g);
            }
        }
    }
}
