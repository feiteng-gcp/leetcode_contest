class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        int n = rooms.size();
        ArrayDeque<Integer> q = new ArrayDeque<>();
        q.add(0);
        HashSet<Integer> toVisit = new HashSet<Integer>();
        for (int i = 1; i < n; i++)
            toVisit.add(i);
        
while (!q.isEmpty()) {
            int q0 = q.removeFirst();
            for (int r : rooms.get(q0)) {
                if (toVisit.contains(r)) {
                    q.add(r);
                    toVisit.remove(r);
                }
            }
        }
        return toVisit.isEmpty();
    }
}
