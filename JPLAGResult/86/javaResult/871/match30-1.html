class Solution {
    
    class UnionFindNode {
        int size;
        UnionFindNode parent;
        
        public UnionFindNode() {
            size = 1;
            parent = this;
        }
        
        public UnionFindNode getAncestor() {
            UnionFindNode ret = this;
            while(ret != ret.parent) ret = ret.parent;
            return ret;
        }
        
        private void _setParent(UnionFindNode node) {
            node.size += this.size;
            this.parent = node;
        }
        
        public void merge(UnionFindNode node) {
            UnionFindNode p1 = this.getAncestor(), p2 = node.getAncestor();
            if (p1==p2) return;
            if (p1.size > p2.size) {
                p2._setParent(p1);
            } else {
                p1._setParent(p2);
            }
        }
    }
    
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        int count = 0, n = rooms.size();
        if (n<2) return true;
        HashSet<Integer> reach = new HashSet<Integer>();
        reach.add(0);
        HashMap<Integer, List<Integer>> roomMap = new HashMap<Integer, List<Integer>>();
        for(List<Integer> room : rooms) {
            roomMap.put(count, room);
            count += 1;
        }
        LinkedList<Integer> toVisit = new LinkedList<Integer>();
        toVisit.add(0);
        
while(toVisit.size() > 0) {
            Integer cur = toVisit.removeFirst();
            for(Integer next : roomMap.get(cur)) {
                if (reach.contains(next)) {
                    continue;
                }
                toVisit.add(next);
                reach.add(next);
            
}
        }
        return (reach.size() == n);
    }
}
