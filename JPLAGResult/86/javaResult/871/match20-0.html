class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        boolean[] accessibles = new boolean[rooms.size()];
        visit(rooms, 0, accessibles);
        boolean canVisitAllRooms = true;
        for (boolean accessible : accessibles) {
            canVisitAllRooms = canVisitAllRooms && accessible;
        }
        return canVisitAllRooms;
    }
    
    private void visit(List<List<Integer>> rooms, int i, boolean[] accessibles) {
        if (accessibles[i]) {
            return;
        }
        accessibles[i] = true;
        for (int j : rooms.get(i)) {
            visit(rooms, j, accessibles);
        }
    }
}
