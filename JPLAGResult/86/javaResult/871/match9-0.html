class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        HashSet<Integer> seen = new HashSet<>();
        Queue<Integer> q = new ArrayDeque<>();
        q.offer(0);
        seen.add(0);
        while (!q.isEmpty()) {
            int r = q.poll();
            for (int next : rooms.get(r)) {
                if (!seen.contains(next)) {
                    q.offer(next);
                    seen.add(next);
                }
            }
        }
        return seen.size() == rooms.size();
    }
}
