class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
    	if (rooms.size()<=1) return true;
    	ArrayList<Integer> keys = new ArrayList<>();
    	getKey(0, rooms, keys);
    	for (int i = 1; i < rooms.size(); i++) {
			if (!keys.contains(i)) {
				return false;
			}
		}
    	return true;
    }
    
    private void getKey(int key,List<List<Integer>> rooms,List<Integer> keys) {
    	List<Integer> list = rooms.get(key);
    	for (int i = 0; i < list.size(); i++) {
    		Integer nk = list.get(i);
    		if (!keys.contains(nk)) {
    			keys.add(nk);
    			getKey(nk, rooms,keys);
			}
    		
    		
		}
	}
}
