class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        int n = rooms.size();
        
Set<Integer> entered = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(0);
        entered.add(0);
        while (!queue.isEmpty()) {
            int curr = queue.poll();
            List<Integer> keys = rooms.get(curr);
            for (int k : keys) 
{
                if (!entered.contains(k)) {
                    entered.add(k);
                    queue.offer(k);
                }
            }
        }
        return entered.size() == n;
    }
}
