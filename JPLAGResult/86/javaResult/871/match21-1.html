class Solution {
    private int canOpen;
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        canOpen = 0;
        dfs(rooms, 0, new boolean[rooms.size()]);
        return canOpen == 
rooms.size();
    }
    
    private void dfs(List<List<Integer>> rooms, int cur, boolean[] visited) {
        if (visited[cur]) return;
        visited[cur] = true;
        canOpen++;
        
for (int room : rooms.get(cur)) {
            dfs(rooms, room, visited);
        }
    }
}
