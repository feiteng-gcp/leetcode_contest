class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        Queue<Integer> Q = new LinkedList<Integer>();
        Set<Integer> set = new HashSet<>();
        int N = rooms.size();
        
Q.add(0);
        set.add(0);
        while(Q.size() > 0){
            int cur = Q.poll();
            for(int neg : rooms.get(cur)){
                if(set.add(neg)){
                    Q.add(neg);
                }
            }
            if(set.size() == N) return true;
        }
        return set.size() == N;
    }
}
