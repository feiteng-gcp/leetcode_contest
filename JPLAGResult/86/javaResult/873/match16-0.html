/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface Master {
 *     public int guess(String word) {}
 * }
 */
class Solution {
    public void findSecretWord(String[] wordlist, Master master) {
        List<String> wl = Arrays.asList(wordlist);
        for (int i = 0; i < 10; i++) {
            wl = swh(wl, master);
        }
    }
    
    public List<String> swh(List<String> wl, Master master) {
        int n = wl.size();
        List<List<List<String>>> guesses = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            guesses.add(new ArrayList<>());
            for (int j = 0; j < 7; j++)
                guesses.get(i).add(new ArrayList<>());
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int match = 0;
                for (int k = 0; k < 6; k++) {
                    if (wl.get(i).charAt(k)==wl.get(j).charAt(k))
                        match++;
                }
                guesses.get(i).get(match).add(wl.get(j));
            }
        }
        String wtg = "";
        int itg = -1;
        int min = 100000;
        for (int i = 0; i < n; i++) {
            int max = 0;
            for (int j = 0; j < 7; j++) {
                max = Math.max(max, guesses.get(i).get(j).size());
            }
            if (max < min) {
                min = max;
                itg = i;
                wtg = wl.get(i);
            }
        }
        return guesses.get(itg).get(master.guess(wtg));
    }
}
