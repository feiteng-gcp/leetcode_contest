/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface Master {
 *     public int guess(String word) {}
 * }
 */
import java.util.concurrent.ThreadLocalRandom;

class Solution {
    public void findSecretWord(String[] wordlist, Master master) {
        Set<String> set = new HashSet<>();
        for (String str : wordlist){
            set.add(str);
        }
        while(set.size() > 0){
            List<String> tmp = new ArrayList<>(set);
            set = new HashSet<>();
            String guess = tmp.get(ThreadLocalRandom.current().nextInt(tmp.size()));
            int dist = master.guess(guess);
            // System.out.println(Arrays.toString(tmp.toArray()));
            System.out.println(guess + " " + dist);
            if (dist >= 0){
                for (int i = 0; i < tmp.size(); i++){
                    String cur = tmp.get(i);
                    int d = 0;
                    for (int k = 0; k < 6; k++){
                        if (guess.charAt(k) == cur.charAt(k)){
                            d++;
                        }
                    }
                    if (d == dist && !cur.equals(guess)){
                        set.add(cur);
                    }
                }
            }
        }
    }
}
