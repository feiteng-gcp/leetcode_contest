/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface Master {
 *     public int guess(String word) {}
 * }
 */
class Solution {
    public void findSecretWord(String[] wordlist, Master master) {
        int len = wordlist.length;
        boolean[] valid = new boolean[len];
        Arrays.fill(valid, true);
        int icur = 0;
        int imatch = master.guess(wordlist[0]);
        for (int i = 0; i < 9; i++) {
            int next = check(wordlist, valid, icur, imatch);
            imatch = master.guess(wordlist[next]);
            icur = next;
        }
    }
    private int check(String[] words, boolean[] valid, int icur, int match) {
        String w = words[icur];
        valid[icur] = false;
        int ret = 0;
        for (int i = 0; i < words.length; i++) {
            if (!valid[i]) {
                continue;
            }
            int cnt = 0;
            for (int j = 0; j < 6; j++) {
                if (words[i].charAt(j) == w.charAt(j)) {
                    cnt++;
                }
            }
            
            if (cnt != match) {
                valid[i] = false;
            } else {
                ret = i;
            }
        }
        return ret;
    }
}
