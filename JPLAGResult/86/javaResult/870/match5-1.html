class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        if (grid == null) {
            return 0;
        }
        
int nn = grid.length;
        if (nn < 3) {
            return 0;
        }
        int ret = 0;
        for (int i = 0; i < nn - 2; i++) {
            for (int j = 0; j < nn - 2; j++) {
                if (check(grid, i, j)) {
                    ret++;
                }
            }
        }
        
        return ret;
    }
    private boolean check(int[][] grid, int xs, int ys) {
        boolean[] vis = new boolean[10];
        for (int i = 0; i < 3; i++) {
            int cursum = 0;
            for (int j = 0; j < 3; j++) {
                cursum += grid[xs + i][ys + j];
                
if (grid[xs + i][ys + j] > 9 || grid[xs + i][ys + j] < 1 || vis[grid[xs + i][ys + j]] ) {
                    return false;
                }
                vis[grid[xs + i][ys + j]] = true;
                
            }
            if (cursum != 15) {
                return false;
            }
        }
        for (int j = 0; j < 3; j++) {
            int cursum = 0;
            for (int i = 0; i < 3; i++) {
                cursum += grid[xs + i][ys + j];
            }
            if (cursum != 15) {
                return false;
            }
        }
        
        int cursum2 = 0, cursum3 = 0;
        for (int i = 0; i < 3; i++) {
            cursum2 += grid[xs + i][ys + i];
            cursum3 += grid[xs + i][ys + 2 - i];
        }
        
        if (cursum2 != 15 || cursum3 != 15) {
            return false;
        }
        return true;
    }
}
