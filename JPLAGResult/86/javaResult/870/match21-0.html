class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int res = 0;
        for (int i = 0; i <= m - 3; i++) {
            for (int j = 0; j <= n - 3; j++) {
                if (isValid(grid, i, j)) {
                    res++;
                }
            }
        }
        return res;
    }
    
    private boolean isValid(int[][] grid, int x, int y) {
        int target = 0;
        boolean[] used = new boolean[10];
        int[] row = new int[3];
        int[] col = new int[3];
        int diag = 0, anti = 0;
        for (int i = 0; i < 3; i++) {
            diag += grid[x + i][y + i];
            anti += grid[x + 2 - i][y + i];
            for (int j = 0; j < 3; j++) {
                int num = grid[x + i][y + j];
                if (num == 0 || num > 9 || used[num]) {
                    return false;
                }
                used[num] = true;
                row[i] += num;
                col[j] += num;
            }
        }
        if (anti != diag) {
            return false;
        }
        for (int i = 0; i < 3; i++) {
            if (row[i] != anti || col[i] != anti) {
                return false;
            }
        }
        return true;
    }
}
