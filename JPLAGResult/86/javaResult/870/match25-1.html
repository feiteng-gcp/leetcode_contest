class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        int row = grid.length;
        int col = grid[0].length;
        if(row < 3 ||  col < 3) {
            return 0;
        }
        int res = 0;
        for(int i=0; i<row; i++) {
            for(int j=0; j<col; j++) {
                
                if(j+2>=col || i+2>=row) {
                    continue;
                }

               if( 
checkValid( i, j, grid)) {
                   res++;
               }
            }
        }
        return res;
    }
    
    private boolean checkValid(int i, int j, int[][] grid) {
        boolean[] used = new boolean[10];
        int[] row = new int[3];
        int[] col = new int[3];
        int[] dag = new int[2];
        for(int m=0; m<3; m++) {
            
for(int n=0; n<3; n++) {
                int cur = grid[i+m][j+n];
                if(cur<1 || cur>9) {
                    return false;
                }
                used[cur] = true;
                row[m] += cur;
                col[n] += cur;
                if(m==n) {
                    dag[0] += cur;
                }
                if(m+n == 2) {
                    dag[1] += cur;
                }
            }
        }
        int sum = row[0];
        for(int x: row) {
            
if(x != sum) {
                return false;
            }
        }
        for(int x: col) {
            if(x != sum) {
                return false;
            }
        }
        for(int x: dag) {
            
if(x != sum) {
                return false;
            }
        }
        return true;
    }
}
