class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        if (grid.length < 3) return 0;
        
int ret = 0;
        for (int i = 0; i + 2 < grid.length; i++) {
            for (int j = 0; j + 2 < grid.length; j++) {
                if (isMagic(grid, i, j)) ret++;
            }
        }
        return ret;
    }
    private boolean isMagic(int[][] grid, int i, int j) {
        if (grid[i+1][j+1] != 5) return false;
        boolean[] visited = new boolean[9];
        
for (int x = i; x < i+3; x++) {
            for (int y = j; y < j+3; y++) {
                if (grid[x][y] == 0 || grid[x][y] > 9) return false;
                if (visited[grid[x][y]-1]) return false;
                visited[grid[x][y]-1] = true;
            }
        }
        if (grid[i][j] + grid[i][j+1] + grid[i][j+2] != 15) return false;
        
if (grid[i+1][j] + grid[i+1][j+2] != 10) return false;
        if (grid[i+2][j] + grid[i+2][j+1] + grid[i+2][j+2] != 15) return false;
        if (grid[i][j] + grid[i+1][j] + grid[i+2][j] != 15) return false;
        if (grid[i][j+1] + grid[i+2][j+1] != 10) return false;
        if (grid[i][j+2] + grid[i+1][j+2] + grid[i+2][j+2] != 15) return false;
        if (grid[i][j+2] + grid[i+2][j] != 10) return false;
        if (grid[i][j] + grid[i+2][j+2] != 10) return false;
        return true;
    }
}
