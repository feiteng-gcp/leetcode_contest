class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        int n = grid.length, count = 0;
        for (int i = 0; i <= n - 3; i++) {
            for (int j = 0; j <= n - 3; j++) {
                if (isMS(grid, i, j)) count++;
            }
        }
        return count;
    }
    
    private boolean isMS(int[][] grid, int i, int j) {
        int[] row = new int[3], col = new int[3];
        int diag = 0, antiD = 0;
        Set<Integer> visited = new HashSet<>();
        for (int k = i; k < i + 3; k++) {
            for (int l = j; l < j + 3; l++) {
                if (!visited.add(grid[k][l]) || grid[k][l] < 1 || grid[k][l] > 9) return false;
                row[k - i] += grid[k][l];
                col[l - j] += grid[k][l];
                if (k - i == l - j) diag += grid[k][l];
                if (k - i + l - j == 2) antiD += grid[k][l];
            }
        
}
        
        for (int k = 0; k < 3; k++) {
            if (row[k] != 15 || col[k] != 15) return false;
        }
        return diag == 15 && antiD == 15;
    }
}
