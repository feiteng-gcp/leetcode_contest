class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        int n = grid.length - 3;
        int m = grid[0].length - 3;
        if (n < 0 || m < 0){
            return 0;
        }
        int ans = 0;
        for (int i = 0; i <= n; i++){
            for (int j = 0; j <= m; j++){
                if (check(grid, i, j)){
                    ans++;
                }
            }
        }
        return ans;
    }
    
    boolean check(int[][] grid, int i, int j){
        int sum = 0;
        boolean[] check = new boolean[10];
        for (int b = 0; b < 3; b++){
            sum += grid[i + b][j + b];
        }
        for (int a = 0; a < 3; a++){
            int s2 = 0;
            for (int b = 0; b < 3; b++){
                int cur = grid[i + a][j + b];
                if (cur <= 0 || cur > 9){
                    return false;
                }
                if (check[cur] == true){
                    return false;
                }
                check[cur] = true;
                s2 += grid[i + a][j + b];
            }
            if (s2 != sum){
                //System.out.printf("a:%d, s2:%d%n", a, s2);
                return false;
            }
        }
        for (int a = 0; a < 3; a++){
            int s2 = 0;
            for (int b = 0; b < 3; b++){
                s2 += grid[i + b][j + a];
            }
            if (s2 != sum){
                //System.out.printf("a:%d, s2:%d%n", a, s2);
                return false;
            }
        }
        return sum == grid[i + 2][j] + grid[i + 1][j + 1] + grid[i][j + 2];
    }
}
