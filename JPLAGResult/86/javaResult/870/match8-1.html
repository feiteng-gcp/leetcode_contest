class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        
int numberOfMagicSquares = 0;
        for (int i = 0; i < grid.length - 2; ++i) {
            for (int j = 0; j < grid[0].length - 2; ++j) {
                if (isMagicSquare(grid, i, j)) {
                    ++numberOfMagicSquares;
                }
            }
        }
        return numberOfMagicSquares;
    }
    
    private boolean isMagicSquare(int[][] grid, int i, int j) {
        for (int x = i; x < i + 3; ++x) {
            int sum = 0;
            for (int y = j; y < j + 3; ++y) {
                sum += grid[x][y];
            
}
            if (sum != 15) {
                return false;
            }
        }
        for (int y = j; y < j + 3; ++y) {
            int sum = 0;
            for (int x = i; x < i + 3; ++x) {
                sum += grid[x][y];
            }
            if (sum != 15) {
                return false;
            }
        }
        int sum = 0;
        for (int x = 0; x < 3; ++x) {
            sum += grid[i + x][j + x];
        }
        if (sum != 15) {
            return false;
        }
        sum = 0;
        for (int x = 0; x < 3; ++x) {
            sum += grid[i + 2 - x][j + x];
        }
        if (sum != 15) {
            return false;
        }
        sum = 0;
        int product = 1;
        for (int x = i; x < i + 3; ++x) {
            for (int y = j; y < j + 3; ++y) {
                sum += grid[x][y];
                product *= grid[x][y];
            }
        
}
        if ((sum != 45) || (product != 362880)) {
            return false;
        }
        return true;
    }
}
