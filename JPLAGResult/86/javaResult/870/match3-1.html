class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        int res = 0;
        
int m = grid.length, n = grid[0].length;
        for (int i = 0; i + 2 < m; i++) {
            for (int j = 0; j + 2 < n; j++) {
                if (check(grid, i, j)) {
                    res++;
                }
            }
        }
        return res;
    }
    public boolean check(int[][] grid, int x, int y) {
        if (grid[x + 1][y + 1] != 5) {
            return false;
        }
        
boolean[] taken = new boolean[9];
        for (int i = x; i < x + 3; i++) {
            int lenSum = 0;
            for (int j = y; j < y + 3; j++) {
                lenSum += grid[i][j];
                //System.out.println(grid[i][j]);
                if (grid[i][j] > 9 || grid[i][j] < 1
                    || taken[grid[i][j] - 1]) {
                    return false;
                }
                taken[grid[i][j] - 1] = true;
            }
            if (lenSum != 15) {
                return false;
            }
        }
        for (int j = y; j < y + 3; j++) {
            int lenSum = 0;
            for (int i = x; i < x + 3; i++) {
                lenSum += grid[i][j];
            }
            if (lenSum != 15) {
                return false;
            }
        }
        
if (grid[x][y] + grid[x + 1][y + 1] + grid[x + 2][y + 2] != 15 ||
           grid[x + 2][y] + grid[x + 1][y + 1] + grid[x][y + 2] != 15) {
            return false;
        }
        return true;
    }
}
