class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        int n = grid.length;
        int count = 0;
        for(int i = 0; i < n-2; i++) {
            for(int j = 0; j < n-2; j++) {
                if(check(grid, i, j)) count++;
            }
        }
        return count;
    }
    
    private boolean check(int[][] grid, int i , int j) {
        boolean[] visited = new boolean[9];
        int[] col = new int[3];
        int[] diag = new int[2];
        for(int k = 0; k < 3; k++) {
            int row = 0;
            for(int l = 0; l < 3; l++) {
                int cur = grid[i+k][j+l]; 
                if(cur >= 1 && cur <= 9 && !visited[cur-1]) {
                    row += cur;
                    col[l] += cur;
                    if(k == l) diag[0] += cur;
                    if(k+l == 2) diag[1] += cur;
                    visited[cur - 1] = true;
                } else 
                    return false;
            }
            if(row != 15) return false;
        }
        for(int k = 0; k < 3; k++) {
            if(col[k] != 15) return false;
        }
        if(diag[0] != 15 || diag[1] != 15) 
            return false;
        return true;
    }
}
