class Solution {
	public int numMagicSquaresInside(int[][] grid) {
		int h = grid.length;
		int w = grid[0].length;

		int count = 0;

		for (int i = 0; i < h; i++) {
			for (int j = 0; j < w; j++) {
				if (check(grid, i, j)) {
					count++;
				}
			}
		}
		return count;
	}

	boolean check(int[][] grid, int h, int w) {
		if (h + 2 >= grid.length || w + 2 >= grid[0].length) {
			return false;
		}

		Set<Integer> set = new HashSet<>();
		for (int i = 1; i <= 9; i++) {
			set.add(i);
		}
		for (int i = h; i < h + 3; i++) {
			for (int j = w; j < w + 3; j++) {
				set.remove(grid[i][j]);
			}
		}
		if (!set.isEmpty()) {
			return false;
		}

		int total = grid[h][w] + grid[h + 1][w + 1] + grid[h + 2][w + 2];
		for (int i = h
; i < h + 3; i++) {
			int sum = 0;
			for (int j = w; j < w + 3; j++) {
				sum += grid[i][j];
			}
			if (total != sum) {
				return false;
			}
		}
		for (int i = w; i < w + 3; i++) {
			int sum = 0;
			for (int j = h; j < h + 3; j++) {
				sum += grid[j][i];
			}
			if (total != sum) {
				return false;
			}
		}

rid[h + 2][w] + grid[h + 1][w + 1] + grid[h][w + 2]) {
			return false;
		}

		return true;
	}

}
