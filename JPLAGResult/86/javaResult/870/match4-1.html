class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        
if (grid.length < 3 || grid[0].length < 3) {
            return 0;
        }
        int count = 0;
        for (int i = 0; i < grid.length - 2; i++) {
            for (int j = 0; j < grid[0].length - 2; j++) {
                if (isValid(grid, i, j)) {
                    count++;
                }
            }
        }
        return count;
        
    }
    public boolean isValid(int[][] grid, int row, int col) {
        // for row
        Set<Integer> set = new HashSet();
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (grid[row +i][col + j] <1 || 
                    grid[row +i][col + j] >9 || 
                    !set.add(grid[row + i][col + j])) {
                    return false;
                }
            }
        }
        for (int rowOffset = 0; rowOffset < 3; rowOffset++) {
            int tmp = 0;
            for (int j = 0; j < 3; j ++) {
                tmp += grid[row + rowOffset][col + j];
            
}
            if (tmp != 15) {
                return false;
            }
        }
        for (int colOffset = 0; colOffset < 3; colOffset++) {
            int tmp = 0;
            for (int i = 0; i < 3; i++) {
                tmp += grid[row +i][col + colOffset];
            }
            if (tmp != 15) {
                return false;
            }
        }
        int tmp =0;
        for (int i = 0; i < 3; i++) {
            tmp += grid[row +i][col + i];
            
        }
        if (tmp != 15) {
            return false;
        }
        tmp  =0;
        
for (int i = 0; i < 3; i++) {
            tmp += grid[row +i][col + 2 -i];
        }
        if (tmp != 15) {
            return false;
        }
        return true;
        
    }
}
