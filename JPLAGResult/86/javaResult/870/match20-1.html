class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        int ans = 0;
        for (int i = 1; i < grid.length - 1; i++) {
            for (int j = 1; j < grid[0].length - 1; j++) {
                if (valid(grid, i, j)) {
                    ans++;
                
}
            }
        }
        return ans;
    }
    
    public boolean valid(int[][] grid, int x, int y) {
        boolean[] v = new boolean[10];
        int[][] board = new int[3][3];
        int c = grid[x][y];
        board[0][0] = grid[x - 1][y - 1];
        board[0][1] = grid[x - 1][y];
        board[0][2] = grid[x - 1][y + 1];
        board[1][0] = grid[x][y - 1];
        board[1][1] = grid[x][y];
        board[1][2] = grid[x][y + 1];
        board[2][0] = grid[x + 1][y - 1];
        board[2][1] = grid[x + 1][y];
        board[2][2] = grid[x + 1][y + 1];
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (board[i][j] > 9) {
                    return false;
                }
            }
        }
        if (board[1][1] != 5) {
            return false;
        }
        if (board[0][0] + board[2][2] != 10) {
            return false;
        }
        
if (board[0][2] + board[2][0] != 10) {
            return false;
        }
        if (board[1][0] + board[1][2] != 10) {
            return false;
        }
        if (board[0][1] + board[2][1] != 10) {
            return false;
        }
        if (board[0][0] + board[0][1] + board[0][2] != 15) {
            return false;
        }
        if (board[2][0] + board[2][1] + board[2][2] != 15) {
            return false;
        }
        if (board[0][0] + board[1][0] + board[2][0] != 15) {
            return false;
        }
        if (board[2][0] + board[2][1] + board[2][2] != 15) {
            return false;
        }
        return true;
    }
}
