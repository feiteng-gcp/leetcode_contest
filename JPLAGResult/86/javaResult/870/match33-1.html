class Solution {
    public int numMagicSquaresInside(int[][] grid) {
        
int num = 0;
        for(int or = 0; or <= grid.length - 3; ++ or) {
            for(int oc = 0; oc <= grid[0].length - 3; ++ oc) {
                if(!checkNumber(grid, or, oc)){
                    System.out.println("cn: " + or + " " + oc);
                    continue;
                }
                if(!checkSum(grid, or, oc)) {
                    System.out.println("cs: " + or + " " + oc);
                    continue;
                
}
               ++ num;
            }
        }
        return num;
    }
    
    private boolean checkNumber(int[][] grid, int or, int oc) {
        boolean[] e = new boolean[10];
        for(int i = 0; i < 3; ++ i) {
            for(int j = 0; j < 3; ++ j) {
                int n = grid[or + i][oc + j];
                if(n < 1 || n > 9) {
                    return false;
                }
                e[n] = true;
            }
        }
        for(int i = 1; i <= 9; ++ i) {
            if(!e[i]) {
                return false;
            }
        }
        return true;
    }
    
    private boolean checkSum(int[][] grid, int or, int oc) {
        for(int i = 0; i < 3; ++ i) {
            int sum = 0;
            for(int j = 0; j < 3; ++ j) {
                sum += grid[or + i][oc + j];
            
}
            if(sum != 15) {
                return false;
            }
        }
        for(int i = 0; i < 3; ++ i) {
            int sum = 0;
            for(int j = 0; j < 3; ++ j) {
                sum += grid[or + j][oc + i];
            }
            if(sum != 15) {
                return false;
            }
        }
        {
            
int sum = 0;
            for(int i = 0; i < 3; ++ i) {
                sum += grid[or + i][oc + i];
            }
            if(sum != 15) {
                return false;
            }
        }
        {
            int sum = 0;
            for(int i = 0; i < 3; ++ i) {
                sum += grid[or + 2 - i][oc + i];
            }
            if(sum != 15) {
                return false;
            }
        }
        return true;
    }
}
