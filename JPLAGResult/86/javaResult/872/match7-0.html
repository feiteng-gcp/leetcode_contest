class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        char[] A = S.toCharArray();
        int n = A.length;
        for (int len = 1; len < n; len++){
            if (!valid(A, 0, len)){
                break;
            }
            long a = Long.parseLong(S.substring(0, len));
            if (a > Integer.MAX_VALUE){
                break;
            }
            for (int len2 = 1; len + len2 < n; len2++){
                if (!valid(A, len, len2)){
                    break;
                }
                long b = Long.parseLong(S.substring(len, len + len2));
                if (b > Integer.MAX_VALUE){
                    break;
                }
                List<Integer> path = new ArrayList<>();
                path.add((int)a);
                path.add((int)b);
                if (check(A, len + len2, (int)a, (int)b, path)){
                    return path;
                }
            }
        }
        return new ArrayList<>();
    }
    
    boolean valid(char[] A, int start, int len){
        if (len > 1 && A[start] == '0'){
            return false;
        }
        return true;
    }
    
    boolean check(char[] A, int start, int a, int b, List<Integer> path){
        while (start < A.length){
            int cur = a + b;
            int val = A[start++] - '0';
            while (start < A.length && val > 0 && val < cur){
                val = val * 10 + (A[start++] - '0');
            }
            if (val == cur){
                path.add(val);
                a = b;
                b = cur;
            } else {
                return false;
            }
        }
        return true;
    }
}
