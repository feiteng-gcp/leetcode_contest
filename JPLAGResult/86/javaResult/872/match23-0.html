class Solution {
    public List<Integer> splitIntoFibonacci(String s) {
        for (int i = 1; i <= s.length() / 2 && i < 11; i++) {
            for (int j = 1; j + i + Math.max(i, j) <= s.length() && j < 11; j++) {
                Result r = help(i, j, s);
                if (r.res) {
                    return r.list;
                }
            }
        }
        return new ArrayList<>();
    }

    private Result help(int i, int j, String s) {
        Result r = new Result();
        if (i > 1 && s.charAt(0) == '0') return r;
        if (j > 1 && s.charAt(i) == '0') return r;
        long num1 = Long.parseLong(s.substring(0, i));
        if (num1 > Integer.MAX_VALUE) return r;
        r.list.add((int)num1);
        long num2 = Long.parseLong(s.substring(i, i + j));
        if (num1 > Integer.MAX_VALUE) return r;
        r.list.add((int)num2);
        long sum = 0;
        int len = (num1 + "").length() + ("" + num2).length();
        for (; len < s.length();) {
            sum = num1 + num2;
            if (sum > Integer.MAX_VALUE) return r;
            r.list.add((int)sum);
            int sumL = (sum + "").length();
            if (len + sumL > s.length() || !s.substring(len, len + sumL).equals(sum + "")) {
                return r;
            }
            len += sumL;
            num1 = num2;
            num2 = sum;
        }
        r.res = true;
        return r;
    }
}
class Result{
    boolean res = false;
    List<Integer> list = new ArrayList<>();
}
