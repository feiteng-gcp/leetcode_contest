class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> res = new ArrayList<>();
        if (S == null || S.length() == 0) return res;
        dfs(S, res);
        return res;        
    }
    
    boolean dfs(String S, List<Integer> path) {
        if (S.length() == 0 && path.size() > 2) return true;
        int i = 0;
        long x = 0, y = 0;
        while (i < S.length()) {
            x = x * 10 + (S.charAt(i) - '0');
            if (i > 0 && x < 10) break;
            if (x > Integer.MAX_VALUE) break;
            // if (path.size() > 0 && x + path.get(path.size() - 1) > Integer.MAX_VALUE) break;
            if (path.size() < 2) {
                path.add((int)x);
                if (dfs(S.substring(i + 1), path))
                    return true;
                
path.remove(path.size() - 1);
            } else {
                int f1 = path.get(path.size() - 2);
                int f2 = path.get(path.size() - 1);
                if ((int)x == f1 + f2) {
                    path.add((int)x);
                    if (dfs(S.substring(i + 1), path))
                        return true;
                    path.remove(path.size() - 1);
                }
            }
            i++;
        }
        return false;
    }
}
