class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> res = new ArrayList<>();
        if (!split(S.toCharArray(), 0, res)) return new ArrayList<>();
        return res;
    }

    private boolean split(char[] cs, int index, List<Integer> path) {
        int size = path.size();
        if (index == cs.length) {
            if (size <= 2) return false;

            // sum = path.get(size - 1) + path.get(size - 2);
            return true;
        }

        long sum = -1;
        if (size > 1) {
            sum = (long)path.get(size - 1) + path.get(size - 2);
        }
        if (cs[index] == '0') {
            if (sum > 0) {
                return false;
            }
            path.add(0);
            if (split(cs, index + 1, path)) return true;

            path.remove(path.size() - 1);
            return false;
        }
        for (int i = index; i < cs.length; i++) {
            long k = Long.parseLong(String.valueOf(cs, index, i + 1 - index));
            if (k >= Integer.MAX_VALUE) return false;
            if (sum < 0) {
                path.add((int)k);
                if (split(cs, i + 1, path)) return true;
                path.remove(path.size() - 1);
            } else if (k == sum) {
                path.add((int)k);
                if (split(cs, i + 1, path)) return true;
                path.remove(path.size() - 1);
                return false;
            } else if (k > sum) break;
        }
        // path.clear();
        // path.remove(path.size() - 1);
        return false;
    }
}
