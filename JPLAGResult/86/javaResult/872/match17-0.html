class Solution {
    private List<Integer> process(String s, int p1, int p2, LinkedList<Integer> current) {
        long sum = (long) p1 + p2;
        String sumString = String.valueOf(sum);
        if (s.equals(sumString) && s.length() <= 31) {
            current.add(Integer.parseInt(s));
            return current;
        }

        if (s.isEmpty()) {
            return null;
        }

        for (int i = 1; i < Math.min(10, s.length()); i++) {
            String newString = s.substring(0, i);
            if (newString.startsWith("0") && newString.length() > 1) {
                continue;
            }

            long newInt = Long.parseLong(newString);
            if (newInt > Integer.MAX_VALUE) {
                continue;
            }

            if (p1 + p2 != newInt) {
                continue;
            }

            current.add((int) newInt);
            List<Integer> result = process(s.substring(i), p2, (int) newInt, current);
            if (result != null) {
                return result;
            }
            current.removeLast();
        }

        return null;
    }

    public List<Integer> splitIntoFibonacci(String S) {
        int length = S.length();

        for (int i = 1; i < Math.min(10, length - 2); i++) {
            for (int j = 1; j < Math.min(10, length - i - 1); j++) {
                String s1 = S.substring(0, i);
                if (s1.startsWith("0") && s1.length() > 1) {
                    continue;
                }

                String s2 = S.substring(i, i + j);
                if (s2.startsWith("0") && s2.length() > 1) {
                    continue;
                }

                long i1 = Long.parseLong(s1);
                if (i1 > Integer.MAX_VALUE) {
                    continue;
                }

                long i2 = Long.parseLong(s2);
                if (i2 > Integer.MAX_VALUE) {
                    continue;
                }

                LinkedList<Integer> current = new LinkedList<>();
                current.add((int)i1);
                current.add((int)i2);

                List<Integer> result = process(S.substring(i + j), (int) i1, (int) i2, current);
                if (result != null) {
                    return result;
                }
            }
        }

        return new LinkedList<>();
    }
}
