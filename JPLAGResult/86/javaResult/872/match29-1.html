class Solution {
public List<Integer> splitIntoFibonacci(String S) {
    int n = S.length();

    for (int i = 1; i <= n -2; i++) {
      long a = Long.parseLong(S.substring(0, i));
      
      if (S.charAt(0) == '0' && i > 1) break;

      if (a > Integer.MAX_VALUE) break;
      

      for (int j = i+1; j <= n -1; j++) {
        long b = Long.parseLong(S.substring(i, j));
        
        if (S.charAt(i) == '0' && (j-i) > 1) break;

        if (b > Integer.MAX_VALUE) break;

        List<Long> sol = new ArrayList<>();

        long gp = a;
        long p = b;

        sol.add(gp);
        sol.add(p);

        int parPt = j;
        int cur = j + 1;

        while (cur <= n) {
          long c = Long.parseLong(S.substring(parPt, cur));
          
          if (S.charAt(parPt) == '0' && (cur - parPt) > 1) break;
          if (c > Integer.MAX_VALUE) break;

          if (c == gp + p) {
            sol.add(c);

            if (cur == n) {
              List<Integer> res = new ArrayList<>();
              for(Long l : sol) {
                res.add( l.intValue());
              }
              return res;
            }

            gp = p;
            p = c;
            parPt = cur;
          }
          cur ++;
        
}
      }
    }


    return new ArrayList<>();
  }
}
