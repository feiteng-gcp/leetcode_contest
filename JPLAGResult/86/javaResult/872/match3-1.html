class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
             List<Integer> res=new ArrayList<Integer>();
	        if(S==null||S.length()==0){
	            return res;
	        }
	        boolean res1=helper(res, S);
	        return res1?res:new ArrayList<Integer>();
    }
    
     public boolean helper(List<Integer> list, String s){
	        if(s.length()==0){
	            if(list.size()>=3){
	                return true;
	            }

	            return false;
	        }
	        
	        for(int i=0;i<s.length();i++){
	            String temp=s.substring(0, i+1);
                if(temp.length()>10||Long.parseLong(temp)>Integer.MAX_VALUE){
                    continue;
                }
	            if(temp.length()>1&&temp.charAt(0)=='0'){
	                continue;
	            }
	            if(list.size()<2){
	                list.add(Integer.parseInt(temp));
	                if(helper(list, s.substring(i+1))){
	                	return true;
	                }

	                list.remove(list.size()-1);
	            }else{
	                int num=Integer.parseInt(temp);
	                if(list.get(list.size()-2)+list.get(list.size()-1)==num){
	                    list.add(num);
	                    if(helper(list, s.substring(i+1))){
	                    	return true;
	                    }
	                    list.remove(list.size()-1);
	                }else if(list.get(list.size()-2)+list.get(list.size()-1)<num){
	                	continue;
	                }

	            }
	        }
	        return false;
	    }
}
