class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> res = new ArrayList<>();
        if(S == null || S.isEmpty()) {
            return res;
        }
        int n = S.length();
        for(int i = 0; i < n; i++) {
            if(i > 0 && S.charAt(0) == '0') {
                break;
            }
            long num1 = Long.valueOf(S.substring(0, i + 1));
            if(num1 > Integer.MAX_VALUE) {
                break;
            }
            for(int j = i + 1; j < n - 1; j++) {
                if(j > i + 1 && S.charAt(i + 1) == '0') {
                    break;
                }
                
long num2 = Long.valueOf(S.substring(i + 1, j + 1));
                if(num2 > Integer.MAX_VALUE) {
                    break;
                }
                res.add((int)num1);
                res.add((int)num2);
                if(dfs(S, j + 1, (int)num1, (int)num2, res)) {
                    return res;
                }
                res.clear();
            }
        }
        return res;
    }
    
    public boolean dfs(String s, int startIndex, long num1, long num2, List<Integer> res) {
        if(startIndex == s.length()) {
            return true;
        }
        
        for(int i = startIndex; i < s.length(); i++) {
            if(i > startIndex && s.charAt(startIndex) == '0') {
                break;
            }
            long num3 = Long.valueOf(s.substring(startIndex, i + 1));
            if(num3 > Integer.MAX_VALUE) {
                break;
            }
            if(num1 + num2 < num3) {
                return false;
            }
            if(num1 + num2 == num3) 
{
                res.add((int)num3);
                if(dfs(s, i + 1, num2, num3, res)) {
                    return true;
                }
                res.remove(res.size() - 1);
                return false;
            }
        }
        
        return false;
    }
}
