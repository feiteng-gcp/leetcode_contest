class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> res = new ArrayList<>();
        if (dfs(S, 0, res)) {
            return res;
        }
        return new ArrayList<>();
    }
    
    private boolean dfs(String S, int curr, List<Integer> res) {
        if (curr == S.length()) {
            return res.size() > 2;
        }
        if (res.size() < 2) {
            if (S.charAt(curr) == '0') {
                res.add(0);
                if (dfs(S, curr + 1, res)) {
                    return true;
                }
                res.remove(res.size() - 1);
                return false;
            }
            for (int i = curr; i < S.length(); i++) {
                long temp = Long.parseLong(S.substring(curr, i + 1));
                if (temp > Integer.MAX_VALUE) {
                    break;
                }
                res.add((int) temp);
                if (dfs(S, i + 1, res)) {
                    return true;
                }
                res.remove(res.size() - 1);
            }
            return false;
        }
        long sum = (long) res.get(res.size() - 1) + res.get(res.size() - 2);
        if (sum > Integer.MAX_VALUE) {
            return false;
        }
        String str = Integer.toString((int) sum);
        if (S.indexOf(str, curr) == curr) {
            res.add((int) sum);
            if (dfs(S, curr + str.length(), res)) {
                return true;
            }
            res.remove(res.size() - 1);
        }
        return false;
    }
}
