class Solution {
    int max = Integer.MAX_VALUE;
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> res = new ArrayList<>();
        if(S.length() < 3) return res;
        //for(int i=0;i<S.length()-2;i++){
        int i = 0;
            for(int j=i;j<S.length()-2;j++){
                if(j - i >= 1 && S.charAt(i) == '0'){
                    //System.out.println(" jj " + j);
                     break;
                }
                long num = Long.parseLong(S.substring(i,j+1));
                if(num > max) break;
                res.add((int) num);
                if(backtarck(res,j+1,S)){
                    return res;
                }
                res.remove(res.size()-1);
            }
        //}
        return new ArrayList<>();
    }
    
    public boolean backtarck(List<Integer> res,int index,String s){
        //System.out.println(res + " res index " + index);
        if(index == s.length()){
            return true;
        }
        int end = res.size() >= 2 ? s.length() : (s.length() - 1);
        int i = index;
        //for(int i=index;i<end;i++){        
            for(int j=i;j<end;j++){
                if(j - i > 1 && s.charAt(i) == '0') break;
                long num = Long.parseLong(s.substring(i,j+1));
                if(num > max) break;
                if(res.size() == 1){
                    res.add((int) num);
                    if(backtarck(res,j+1,s)){
                        return true;
                    }
                    res.remove(res.size()-1);
                }else if(num == res.get(res.size()-1) + res.get(res.size() - 2)){
                    res.add((int) num);
                    if(backtarck(res,j+1,s)){
                        return true;
                    }
                    res.remove(res.size()-1);
                }
            }
       // }
        return false;
    }
}
