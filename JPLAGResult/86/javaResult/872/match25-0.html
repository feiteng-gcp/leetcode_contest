class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> cur = new ArrayList<>();
        dfs(S, 0, res, cur);
        if (res.size() == 0) {
            return new ArrayList<>();
        }
        return res.get(0);
    }
    public void dfs(String S, int i, List<List<Integer>> res, List<Integer> cur) {
        if (i == S.length()) {
            if (cur.size() >= 3) {
                res.add(new ArrayList<>(cur));
            }
            return;
        }
        long val = 0;
        for (int idx = i; idx < S.length(); ++idx) {
            val = 10 * val + (long) (S.charAt(idx) - '0');
            if (val > Integer.MAX_VALUE) {
                return;
            }
            if (cur.size() <= 1) {
                cur.add((int) val);
                dfs(S, idx + 1, res, cur);
                cur.remove(cur.size() - 1);
            } else {
                long prev1 = (long) cur.get(cur.size() - 1);
                long prev2 = (long) cur.get(cur.size() - 2);
                if (prev1 + prev2 == val) {
                    cur.add((int) val);
                    dfs(S, idx + 1, res, cur);
                    cur.remove(cur.size() - 1);
                }
            }
            if (res.size() != 0) {
                return;
            }
            if (idx == i && val == 0) {
                return;
            }
        }
    }
}
