class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
    	for (int i = 1; i <= (S.length()+2)/3; i++) {
    		String ss = S.substring(0, i);
    		if (i>1&&ss.charAt(0)=='0') {
				break;
			}
    		Integer i1;
    		try {
    			i1 = new Integer(ss);
			} catch (Exception e) {
				break;
			}
    		for (int j = 1;j<=(S.length()-i)/2; j++) {
    			Integer i2 = null;
    			try {
    				i2 = new Integer(S.substring(i, i+j));
				} catch (Exception e) {
					break;
				}
    			List<Integer> fl = new ArrayList<>(Arrays.asList(i1,i2));
    			if (toFibonacci(fl, S.substring(i+j))) {
					return fl;
				}
			}
		}
		return new ArrayList<>();
    }
    
    private boolean toFibonacci(List<Integer> fl,String ss) {
    	Integer i1 = fl.get(fl.size()-1);
    	Integer i2 = fl.get(fl.size()-2);
    	Integer i3;
    	try {
    		i3 = i1+i2;
		} catch (Exception e) {
			return false;
		}
    	char[] charArray = i3.toString().toCharArray();
    	char[] ssc = ss.toCharArray();
    	for (int i = 0; i < charArray.length; i++) {
    		if (i>=ssc.length || ssc[i]!=charArray[i]) {
				return false;
			}
		}
    	fl.add(i3);
    	String nss = ss.substring(charArray.length);
    	if (nss.length()==0) {
			return true;
		}
    	return toFibonacci(fl, nss);
	}
}
