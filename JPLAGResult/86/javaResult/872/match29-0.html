class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        for (int i = 1; i < 10; i++) {
            if (S.length() <= i)
                break;
            if (S.charAt(0) == '0' && i>1)
                break;
            String s0 = S.substring(0,i);
            if (Long.parseLong(s0) > Integer.MAX_VALUE)
                break;
            for (int j = 1; j < 10; j++) {
                if (S.length() <= i+j)
                    break;
                if (S.charAt(i) == '0' && j>1)
                    break;
                String s1 = S.substring(i, i+j);
                if (Long.parseLong(s1) > Integer.MAX_VALUE)
                    break;
                List<Integer> fibs = new ArrayList<Integer>();
                String fibString = s0+s1;
                fibs.add(Integer.parseInt(s0));
                fibs.add(Integer.parseInt(s1));
                while (S.startsWith(fibString)) {
                    if (S.equals(fibString)) {
                        return fibs;
                    }
                    int nextFib = fibs.get(fibs.size()-2)+fibs.get(fibs.size()-1);
                    fibs.add(nextFib);
                    fibString += nextFib;
                }
            }
        }
        return new ArrayList<Integer>();
    }
}
