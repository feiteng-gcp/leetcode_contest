class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        if (S == null) {
            return new ArrayList<>();
        }
        for (int i = 0; i < S.length(); i++) {
            for (int j = i + 1; j < S.length(); j++) {
                String num1 = S.substring(0, i + 1);
                String num2 = S.substring(i + 1, j + 1);
                Double val1 = Double.valueOf(num1);
                Double val2 = Double.valueOf(num2);
                if (val1 > Integer.MAX_VALUE || val2 > Integer.MAX_VALUE) {
                    continue;
                }
                if (val1 == 0 && num1.length() > 1)
                    continue;
                if (val2 == 0 && num2.length() > 1)
                    continue;
                List<Integer> move = go(S, val1.intValue(), val2.intValue());
                if (move != null && 
move.size() >= 3) {
                    return move;
                }
            }
        }
        return new ArrayList<>();
    }
    private List<Integer> go(String s, int f1, int f2) {
        List<Integer> res = new ArrayList<>();
        
        long fl1 = f1, fl2 = f2, fl3 = 0;
        while (s.length() > 0) {
            String fs = Long.toString(fl1);
            if (fl1 <= Integer.MAX_VALUE && s.startsWith(fs)) {
                s = s.substring(fs.length());
                res.add((int) fl1);
            } else {
                return new ArrayList<>();
            }
            fl3 = fl1 + fl2;
            fl1 = fl2; fl2 = fl3;
        }
        return res;
    }
}
