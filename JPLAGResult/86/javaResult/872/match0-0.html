class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> ret = new ArrayList<>();
        if (dfs(S, 0, ret))
            return ret;
        return new ArrayList<>();
    }
    
    private boolean dfs(String S, int start, List<Integer> ret) {
        //System.out.println(ret);
        if (S.length()==start) {
            if (ret.size()>=3)
                return true;
            return false;
        }
        if (S.charAt(start)=='0') {
            if (ret.size()<2) {
                ret.add(0);
                if (dfs(S, start+1, ret))
                    return true;
                ret.remove(ret.size()-1);
            } else {
                int total = ret.size();
                if (ret.get(total-2)+ret.get(total-1)==0) {
                    ret.add(0);
                    if (dfs(S, start+1, ret))
                        return true;
                    ret.remove(ret.size()-1);
                } 
            }
            return false;
        }
        for (int end=start;end<S.length();end++) {
            String maxInt = Integer.toString(Integer.MAX_VALUE);
            if ((end-start+1)>maxInt.length()||((end-start+1)==maxInt.length() && S.substring(start,end+1).compareTo(maxInt)>0))
                break;
            if (ret.size()<2) {
                //System.out.println(start+","+end);
                ret.add(Integer.parseInt(S.substring(start,end+1)));
                if (dfs(S, end+1, ret))
                    return true;
                ret.remove(ret.size()-1);
            } else {
                int total = ret.size();
                int cur = Integer.parseInt(S.substring(start,end+1));
                if (ret.get(total-2)+ret.get(total-1)==cur) {
                    ret.add(cur);
                    if (dfs(S, end+1, ret))
                        return true;
                    ret.remove(ret.size()-1);
                } 
            }
        }
        return false;
    }
}
