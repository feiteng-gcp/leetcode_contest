class Solution {
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> result = new ArrayList();
        int len = S.length();
        if (len < 3) return result;
        
        int maxLen = 10;
        for (int j = 1; j < maxLen+1; j++) {
            for (int k = j; k < j+maxLen; k++) {
                result = split(S, j, k);
                if (result.size() > 0) return result;
            }
        }
        
        return result;
    
}
    
    private List<Integer> split(String S, int secStart, int secEnd) {
        List<Integer> result = new ArrayList();
        
        
        if (S.charAt(0) == '0' && secStart > 1) return result;
        if (S.charAt(secStart) == '0' && secEnd > secStart + 1) return result;
        
        long numLong1 = Long.valueOf(S.substring(0, secStart));
        long numLong2 = Long.valueOf(S.substring(secStart, secEnd + 1));
        if (numLong1 >= Integer.MAX_VALUE || numLong2 >= Integer.MAX_VALUE) return result;
        
        int num1 = Integer.valueOf(Long.toString(numLong1));
        int num2 = Integer.valueOf(Long.toString(numLong2));
        result.add(num1);
        result.add(num2);
        
        int start = secEnd + 1;
        int len = S.length();
        while (start < len) {
            long num3 = 0;
            String num3Str = "";
            boolean isFibonacciNext = false;
            
            if (S.charAt(start) == '0') {
                num3 = 0;
                if (num3 == num1 + num2) {
                    isFibonacciNext = true;
                    num3Str = Long.toString(num3);
                }
            } else {
                num3 = num1 + num2;
                if (num3 <= Integer.MAX_VALUE) {
                    num3Str = Long.toString(num3);
                    int num3Len = num3Str.length();
                    // System.out.println(len + " " + start + " " + num3Len);
                    if (len >= start + num3Len && S.substring(start, start+num3Len).equals(num3Str)) {
                        isFibonacciNext = true;
                    }
                }
            }
            
            // System.out.println(isFibonacciNext + " " + num1 + " " + num2 + " " + num3 + " " + result);
            
            if (isFibonacciNext) {
                int num3Int = Integer.valueOf(Long.toString(num3));
                result.add(num3Int);
                start += num3Str.length();
                num1 = num2;
                num2 = num3Int;
            } else {
                result.clear();
                return result;
            }
        }
        return result;
    }
}
