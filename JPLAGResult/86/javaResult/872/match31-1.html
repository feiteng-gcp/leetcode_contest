class Solution {
    List<Integer> cur = new ArrayList<>();
    public List<Integer> splitIntoFibonacci(String S) {
        List<Integer> res = new ArrayList<>();
        dfs(res, S.charAt(0) - '0', 1, S.toCharArray());
        return cur;
    
}
    public void dfs(List<Integer> res, int temp, int index, char[] arr) {
        if (index == arr.length && res.size() >= 2 && temp == res.get(res.size() - 1) + res.get(res.size() - 2)) {
            cur.clear();
            cur.addAll(res);
            cur.add(temp);
            return;
        }
        if (index == arr.length) {
            return;
        }
        int cur = arr[index] - '0';
        // cut 
        if (temp == 0 || res.size() >= 2 && temp == res.get(res.size() - 1) + res.get(res.size() - 2) || res.size() < 2) {
            res.add(temp);
            dfs(res, cur, index + 1, arr);
            res.remove(res.size() - 1);
        }
        // append
        if (temp != 0 && temp <= (Integer.MAX_VALUE - cur) / 10) {
            dfs(res, temp * 10 + cur, index + 1, arr);
        }
    }
}
