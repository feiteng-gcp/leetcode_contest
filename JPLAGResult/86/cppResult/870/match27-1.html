class Solution {
public:

    bool f(vector<vector<int>>& g, int r1, int c1, int r2, int c2){
        int cnt[10] = {0};
        for(int i = r1; i <= r2; ++i){
            for(int j = c1; j <= c2; ++j){
                if(g[i][j] == 0 || g[i][j] > 9) return false;
                cnt[g[i][j]] ++;
            }
        }
        
        for(int i = 1; i <= 9; ++i) if(cnt[i] != 1) return false;
        
        int s = g[r1][c1] + g[r1][c1+1] + g[r1][c1+2];
        return s == g[r1+1][c1] + g[r1+1][c1+1] + g[r1+1][c1+2] &&
            s == g[r1+2][c1] + g[r1+2][c1+1] + g[r1+2][c1+2] &&
            s == g[r1][c1] + g[r1+1][c1] + g[r1+2][c1] &&
            s == g[r1][c1+1] + g[r1+1][c1+1] + g[r1+2][c1+1] &&
            s == g[r1][c1+2] + g[r1+1][c1+2] + g[r1+2][c1+2] &&
            s == g[r1][c1] + g[r1+1][c1+1] + g[r1+2][c1+2] &&
            s == g[r1][c1+2] + g[r1+1][c1+1] + g[r1+2][c1];
    }
    
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        

        int ans = 0;
        for(int i = 0; i + 2 < m; ++i){
            for(int j = 0; j + 2 < n; ++j){
                if(f(grid, i, j, i + 2, j + 2)) ans++;
            }
        }
        
        return ans;
    }
};
