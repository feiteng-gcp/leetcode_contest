class Solution {
public:
	int numMagicSquaresInside(vector<vector<int> > & grid) {

		int s = 0;

		for (int i=0; i+3<=grid.size(); i++) {
			for (int j=0; j+3<=grid[i].size(); j++) {
				if (cb(grid, i, j)) s++;
			}
		}

		return s;
	}

	bool cb(vector<vector<int> > & a, int x, int y) {
		bool v[10] = {0};
		for (int i=x; i<x+3; i++) 
			for (int j=y; j<y+3; j++) {
				if (a[i][j] < 1 || a[i][j] > 9) return false;
				v[a[i][j]] = true;
			}
		for (int i=1; i<=9; i++)
			if (!v[i]) return false;


		for (int i=x; i<x+3; i++) {
			int s = 0;
			for (int j=y; j<y+3; j++) {
				s += a[i][j];
			}

			if (s != 15) return false;
		}

		for (int j=y; j<y+3; j++) {
			int s = 0;
			for (int i=x; i<x+3; i++) {
				s += a[i][j];
			}
			if (s != 15) return false;
		}

		return a[x][y] + a[x+1][y+1] + a[x+2][y+2] == 15 && a[x][y+2] + a[x+1][y+1] + a[x+2][y] == 15;
	}
};
