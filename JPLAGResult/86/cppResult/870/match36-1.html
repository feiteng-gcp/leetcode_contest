class Solution {
public:
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int rows = grid.size();
        if (rows < 3) return 0;

        int cols = grid[0].size();
        if (cols < 3) return 0;
        int ans = 0;
        for (int r=2; r < rows; ++r) {
            for (int c=2; c < cols; ++c) {
                vector<bool> visited(10, false);
                bool match = true;
                for (int i=0; i < 3 && match; ++i) {
                    for (int j=0; j < 3 && match; ++j) {
                        int r1 = r-i;
                        int c1 = c-j;
                        int val = grid[r1][c1];
                        if (val < 1 || val > 9) {
                            match = false;
                            break;
                        }
                        if (visited[val]) {
                            match = false;
                            break;
                        }
                        visited[val] = true;
                    }
                }
                if (!match) continue;
                for (int i=0; i < 3 && match; ++i) {
                    int tot = 0;
                    for (int j=0; j < 3; ++j) {
                        int r1 = r-i;
                        int c1 = c-j;
                        tot += grid[r1][c1];
                    }
                    if (tot != 15) {
                        match = false;
                        break;
                    }
                }
                if (!match) continue;
                for (int j=0; j < 3 && match; ++j) {
                    int tot = 0;
                    for (int i=0; i < 3; ++i) {
                        int r1 = r-i;
                        int c1 = c-j;
                        tot += grid[r1][c1];
                    }
                    if (tot != 15) {
                        match = false;
                        break;
                    }
                }

                if (!match) continue;
                if ( (grid[r][c]+grid[r-1][c-1]+grid[r-2][c-2]) != 15) continue;
                if ( (grid[r-2][c]+grid[r-1][c-1]+grid[r][c-2]) != 15) continue;
                ++ans;
            }
        }
        return ans;
    }
};
