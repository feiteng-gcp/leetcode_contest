const ll MOD=1e9+7;
class Solution {
public:
    bool good(vector<vector<int>>& g,int k,int l){
        bool z[10]{};

        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++){
                int a=g[k+i][l+j];
                if(a<1 || a>9) return 0;
                if(z[a]) return 0;
                z[a]=1;
            }
        }
        set<int> s;

        for(int i=0;i<3;i++){
            int a=0;
            for(int j=0;j<3;j++){
                a+=g[i+k][j+l];
            }
            s.insert(a);

        }
        for(int i=0;i<3;i++){
            int a=0;
            for(int j=0;j<3;j++){
                a+=g[j+k][i+l];
            }
            s.insert(a);
        }
        s.insert(g[k][l]+g[k+1][l+1]+g[k+2][l+2]);
        s.insert(g[k+2][l]+g[k+1][l+1]+g[k][l+2]);
        return s.size()==1;
    }
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int res=0;
        int N=grid.size();

        if(N==0) return 0;
        int M=grid[0].size();
        for(int k=0;k<N-2;k++){
            for(int l=0;l<M-2;l++){
                res+=good(grid,k,l);
            }
        }
        return res;
    }
};
