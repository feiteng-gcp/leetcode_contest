class Solution {
public:
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        n=grid.size();
        m=grid[0].size();
        
        int ans = 0;
        for(int i=0;i<n-2;i++){
            for(int j=0;j<m-2;j++){
                if(help(grid,i,j)) ans++;
            }
        }
        
        return ans;
    }
private:
    int n,m;
    bool help(vector<vector<int>>& grid, int x, int y){
        bool vis[20]={0};
        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++){
                if(vis[grid[x+i][y+j]]) return false;
                vis[grid[x+i][y+j]]=true;
            }
        }
        
        for(int i=1;i<10;i++) if(!vis[i]) return false;
        
        int sum=0,tmp;
        for(int i=0;i<3;i++){
            sum+=grid[x+i][y];
        }
        
        tmp=0;
        for(int i=0;i<3;i++){
            tmp+=grid[x+i][y+1];
        }
        if(tmp!=sum) return false;
        
        tmp=0;
        for(int i=0;i<3;i++){
            tmp+=grid[x+i][y+2];
        }
        if(tmp!=sum) return false;
        
        tmp=0;
        for(int i=0;i<3;i++){
            tmp+=grid[x][y+i];
        }
        if(tmp!=sum) return false;
        
        tmp=0;
        for(int i=0;i<3;i++){
            tmp+=grid[x+1][y+i];
        }
        if(tmp!=sum) return false;
        
        tmp=0;
        for(int i=0;i<3;i++){
            tmp+=grid[x+2][y+i];
        }
        if(tmp!=sum) return false;
        
        tmp=0;
        for(int i=0;i<3;i++){
            tmp+=grid[x+i][y+i];
        }
        if(tmp!=sum) return false;
        
        tmp=0;
        for(int i=0;i<3;i++){
            tmp+=grid[x+2-i][y+i];
        }
        if(tmp!=sum) return false;
        
        return true;
    }
};
