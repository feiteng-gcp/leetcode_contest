class Solution {
public:
    
    int good(vector<vector<int>>& grid, int x, int y)
    {
        int c = 0;
        for (int i = 0; i < 3; ++i) {
            c += grid[x+i][y+i];
        }
        
        int kk = 0;
        for (int i = 0; i < 3; ++i) {
            kk += grid[x+i][y+ 2 -i];
        }
        
        if (kk != c) return 0;
        
        set<int> s;

        for (int i = x; i < x + 3; ++i) {
            int k = 0;
            for (int j = y; j < y + 3; ++j) {
                k += grid[i][j];
                s.insert(grid[i][j]);
            }
           
            if (k != c) return 0;
        }
        if (*s.begin() != 1) return 0;
        if (*s.rbegin() != 9) return 0;
        if (s.size() != 9) return 0;
        

        for (int j = y; j < y + 3; ++j) {
            int k = 0;
            for (int i = x; i < x + 3; ++i) {
                k += grid[i][j];
            }
            
            if (k != c) return 0;
        }
        
       
        
        return 1;
    }
    
    int numMagicSquaresInside(vector<vector<int>>& grid) {     

        int count = 0;
        for (int i = 0; i + 2< grid.size(); ++i) {
            for (int j = 0; j + 2< grid[i].size(); ++j) {
                count += good(grid, i, j);
            }
        }
        
        return count;
    }
};
