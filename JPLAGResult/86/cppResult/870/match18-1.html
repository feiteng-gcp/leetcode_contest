class Solution {
public:
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        if (grid.size() < 3 || grid[0].size() < 3) return false;

        int rtn = 0;
        for (int i = 0; i < grid.size() - 2; ++i) {
            for (int j = 0; j < grid[0].size() - 2; ++j) {
                vector<bool> nums(10, false);
                bool ok = true;

                for (int m = i; m < i + 3; ++m) {
                    for (int n = j; n < j + 3; ++n) {
                        int tmp = grid[m][n];
                        if (tmp < 1 || tmp > 9 || nums[tmp]) {
                            //                            cout << i << " " << j << ", " << m << " " << n << endl;
                            //                            cout << tmp << endl;
                            ok = false;
                            break;
                        }
                        nums[tmp] = true;
                    }
                    if (!ok) break;
                }
                if (ok && valid(grid, i, j)) ++rtn;
            }
        }

        return rtn;
    }

    bool valid(vector<vector<int>>& grid, int i, int j) {
        int sum = grid[i][j] + grid[i][j+1] + grid[i][j+2];


        for (int m = i+1; m < i + 3; ++m) {
            int tmp = 0;
            for (int n = j; n < j + 3; ++n) {
                tmp += grid[m][n];
            }
            if (tmp != sum) return false;
        }

        for (int n = j; n < j + 3; ++n) {
            int tmp = 0;
            for (int m = i; m < i + 3; ++m) {
                tmp += grid[m][n];
            }
            if (tmp != sum) return false;
        }

        int dig;
        dig = grid[i][j] + grid[i+1][j + 1] + grid[i+2][j + 2];
        if (dig != sum) return false;

        dig = grid[i][j+2] + grid[i + 1][j + 1] + grid[i+2][j];
        if (dig != sum) return false;

        return true;
    }
};
