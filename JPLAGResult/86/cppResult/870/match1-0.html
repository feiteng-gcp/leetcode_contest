class Solution {
public:
    bool is_magic(vector<vector<int>> & grid, int row, int col) {
        vector<bool> array (10, false);
        
        // Check rows
        for (int i = 0; i < 3; i++) {
            int row_total = 0; 
            for (int j = 0; j < 3; j++) {
                int num = grid[row + i][col + j];
                if ((num < 1) || (num > 9)) return false;
                row_total += num;
                if (array[num]) return false; // duplicate num
                else array[num] = true;
            }
            if (row_total != 15) return false;
        }
        
        
        // check columns
        for (int i = 0; i < 3; i++) {
            int col_total = 0; 
            for (int j = 0; j < 3; j++) {
                col_total += grid[row + j][col + i];
            }
            if (col_total != 15) return false;
        }
        
        // Check diagonals
        int total = 0;
        for (int i = 0; i < 3; i++) {
            total += grid[row+i][col+i];  
        }
        if (total != 15) return false;

        total = 0;
        for (int i = 0; i < 3; i++) {
            total += grid[row+i][col+2-i];  
        }
        if (total != 15) return false;

        return true;
    }
    
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        
        int rows = grid.size();
        if (rows == 0) return 0;
        int cols = grid[0].size();
        int cnt = 0;
        
        for (int i = 0; i < rows - 2; i++) {
            for (int j = 0; j < cols - 2; j++) {
                if (is_magic(grid, i, j)) cnt++;
            }
        }
        
        return cnt;
        
    }
};
