class Solution {
public:
	bool isMagicSquare(vector<vector<int>>& grid, int originX, int originY) {
		vector<bool> occupied(10, false);
		for (int i = 0; i < 3; i++) {
			for (int j = 0; j < 3; j++) {
				int num = grid[originX + i][originY + j];
				if (num < 1 || num > 9 || occupied[num]) {
					return false;
				}
				occupied[num] = true;
			}
		}

		vector<int> sum(8, 0);
		for (int i = 0; i < 3; i++) {
			for (int j = 0; j < 3; j++) {
				sum[i] += grid[originX + i][originY + j];
				sum[i + 3] += grid[originX + j][originY + i];
			}
			sum[6] += grid[originX + i][originY + i];
			sum[7] += grid[originX + i][originY + 2 - i];
		}

		for (int i = 1; i < sum.size(); i++) {
			if (sum[i] != sum[i - 1]) {
				return false;
			}
		}

		return true;
	}

	int numMagicSquaresInside(vector<vector<int>>& grid) {
		if (grid.size() < 3 || grid.front().size() < 3) {
			return 0;
		}

		int magicSqureCount = 0;
		for (int i = 0; i < grid.size() - 2; i++) {
			for (int j = 0; j < grid[i].size() - 2; j++) {
 				if (isMagicSquare(grid, i, j)) {
					magicSqureCount++;
				}
			}
		}

		return magicSqureCount;
	}
};
