class Solution {
public:
    int numMagicSquaresInside(vector<vector<int>>& g) {
        int N = g.size();
        if (N < 3) return 0;
        int count = 0;
        for (int i = 0; i < N - 2; i++) {
            for (int j = 0; j < N - 2; j++) {
                bool valid = true;
                for (int a = 0; a < 3; a++) {
                    for (int b = 0; b < 3; b++) {
                        if (g[i+a][j+b]>9 || g[i+a][j+b]<1) valid = false;
                    }
                }
                if (!valid) continue;
                int sum = g[i][j] + g[i][j+1] + g[i][j+2];
                if (g[i+1][j] + g[i+1][j+1] + g[i+1][j+2]!=sum) continue;
                if (g[i+2][j] + g[i+2][j+1] + g[i+2][j+2]!=sum) continue;
                if (g[i][j] + g[i+1][j] + g[i+2][j]!=sum) continue;
                if (g[i][j+1] + g[i+1][j+1] + g[i+2][j+1]!=sum) continue;
                if (g[i][j+2] + g[i+1][j+2] + g[i+2][j+2]!=sum) continue;
                if (g[i][j] + g[i+1][j+1] + g[i+2][j+2]!=sum) continue;
                if (g[i+2][j] + g[i+1][j+1] + g[i][j+2]!=sum) continue;       
                count ++;
            }
        }
        return count;
    }
};
