class Solution {
public:
    bool ck(vector<vector<int>>& grid, int r, int c) {
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                int t = grid[r + i][c + j];
                if (t < 1 || t > 9) return false;
            }
        }
        int sum = 0;
        for (int j = 0; j < 3; j++) {
            sum += grid[r][c + j];
        }
        for (int i = 0; i < 3; i++) {
            int t = 0;
            for (int j = 0; j < 3; j++) {
                t += grid[r + i][c + j];
            }
            if (t != sum) return false;
        }
        for (int j = 0; j < 3; j++) {
            int t = 0;
            for (int i = 0; i < 3; i++) {
                t += grid[r + i][c + j];
            }
            if (t != sum) return false;
        }
        int t = 0;
        for (int j = 0; j < 3; j++) {
            t += grid[r + j][c + j];
        }
        if (t != sum) return false;
        
        t = 0;
        for (int j = 0; j < 3; j++) {
            t += grid[r + j][c + 2 - j];
        }
        if (t != sum) return false;
        return true;
    }
    
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int n = grid.size();
        if (n == 0) return 0;
        int m = grid[0].size();
        int res = 0;
        for (int i = 0; i + 2 < n; i++) {
            for (int j = 0; j + 2 < m; j++) {
                if (ck(grid, i, j)) res++;
            }
        }
        return res;
    }
};
