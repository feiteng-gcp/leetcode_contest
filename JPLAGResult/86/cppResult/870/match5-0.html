class Solution {
    int ok(vector< vector<int> > &grid, int x1, int x2, int y1, int y2) {
        set<int> st;
        for (int i = x1; i <= x2; i++) {
            for (int j = y1; j <= y2; j++) {
                if (grid[i][j] >= 1 && grid[i][j] <= 9)
                    st.insert(grid[i][j]);
                else
                    return false;
            }
        }
        if (st.size() != 9) return false;
        int reqdSum = grid[x1][y1] + grid[x1][y1 + 1] + grid[x1][y2];
        
        for (int i = x1; i <= x2; i++) {
            int sum = 0;
            for (int j = y1; j <= y2; j++) {
                sum += grid[i][j];
            }
            if (sum != reqdSum) return false;
        }
        
        for (int j = y1; j <= y2; j++) {
            int sum = 0;
            for (int i = x1; i <= x2; i++) {
                sum += grid[i][j];
            }
            if (sum != reqdSum) return false;
        }
        
        if ((grid[x1][y1] + grid[x1 + 1][y1 + 1] + grid[x1 + 2][y1 + 2]) != reqdSum) return false;
        if ((grid[x2][y1] + grid[x2 - 1][y1 + 1] + grid[x2 - 2][y1 + 2]) != reqdSum) return false;
        return true;
    }
    
public:
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int N = grid.size();
        int M = grid[0].size();
        int ans = 0;
        for (int i = 2; i < N; i++) {
            for (int j = 2; j < M; j++) {
                if (ok(grid, i - 2, i, j - 2, j)) {
                    ans += 1;
                }
            }
        }
        return ans;
    }
};
