class Solution {
public:
    int judge(int x,int y,vector<vector<int>>& grid)
    {
        //cout<<endl;
        int a[15]={0};
        for(int i=x;i<x+3;i++)
        {
            for(int j=y;j<y+3;j++)
            {
                if(grid[i][j]>=10) return -1;
                a[grid[i][j]]++;
                //cout<<grid[i][j]<<" ";
            }
            //cout<<endl;
        }
        for(int i=1;i<=9;i++)
        {
            //cout<<i<<" "<<a[i]<<endl;
            if(a[i]!=1) return -1;
        }
        int sum =-1;
        for(int i=x;i<x+3;i++)
        {
            int tmp=0;
            for(int j=y;j<y+3;j++)
            {
                tmp+=grid[i][j];
            }
            //cout<<tmp<<endl;
            if(sum==-1) sum=tmp;
            else if(sum!=tmp) return -1;
        }

        for(int j=y;j<y+3;j++)
        {
            int tmp=0;
            for(int i=x;i<x+3;i++)
            {
                tmp+=grid[i][j];
            }
            //cout<<tmp<<endl;
            if(sum!=tmp) return -1;
        }
        int tmp=0;
        tmp=grid[x][y]+grid[x+1][y+1]+grid[x+2][y+2];
        //cout<<tmp<<endl;
        if(sum!=tmp) return -1;
        tmp=grid[x][y+2]+grid[x+1][y+1]+grid[x+2][y];
        //cout<<grid[x][y+2]<<" "<<grid[x+1][y+1]<<" "<<grid[x+2][y]<<" ";
        //cout<<tmp<<endl;
        if(sum!=tmp) return -1;

        return 1;
    }
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int m = grid[0].size();
        int n=grid.size();
        //cout<<n<<" "<<m<<endl;
        int sum=0;
        for(int i=0;i+2<n;i++)
        {
            for(int j=0;j+2<m;j++)
            {
                if(judge(i,j,grid)==1)
                    sum++;
            }
        }
        return sum;
    }
};
