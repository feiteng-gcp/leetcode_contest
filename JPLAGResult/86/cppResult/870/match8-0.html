class Solution {
public:
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int res = 0;
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[0].size(); j++) {
                if (isMagicSquare(grid, i, j)) res++;
            }
        }
        return res;
    }
    
    bool isMagicSquare(vector<vector<int>>& grid, int x, int y) {
        if (x + 2 >= grid.size() || y + 2 >= grid[0].size()) return false;
        int sum = 0;
        for (int i = 0; i < 3; i++) sum += grid[x][y + i];
        for (int i = 0; i < 3; i++) {
            int temp = 0;
            for (int j = 0; j < 3; j++) {
                if (grid[x + i][y + j] == 0 || grid[x + i][y + j] > 9) return false;
                temp += grid[x + i][y + j];
            }
            if (sum != temp) return false;
        }
        for (int j = 0; j < 3; j++) {
            int temp = 0;
            for (int i = 0; i < 3; i++) temp += grid[x + i][y + j];
            if (sum != temp) return false;
        }
        int diag1 = grid[x][y] + grid[x + 1][y + 1] + grid[x + 2][y + 2];
        if (diag1 != sum) return false;
        int diag2 = grid[x][y + 2] + grid[x + 1][y + 1] + grid[x + 2][y];
        if (diag2 != sum) return false;
        return true;
    }
};
