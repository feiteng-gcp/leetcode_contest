class Solution {
private:
    bool helper(vector<vector<int>>& grid, int i, int j) {
        unordered_set<int> s;

        for (int r = i; r < i+3; r++) {
            int sum = 0;
            for (int c = j; c < j+3; c++) {
                if (grid[r][c] > 9 || grid[r][c] < 1) return false;
                s.insert(grid[r][c]);
                sum += grid[r][c];
            }
            if (sum != 15) return false;
        }
        if (s.size() != 9) return false;

        for (int c = j; c < j+3; c++) {
            int sum = 0;
            for (int r =i; r < i+3; r++) {
                sum += grid[r][c];
            }
            if (sum != 15) return false;
        }
        if (grid[i][j] + grid[i+1][j+1] + grid[i+2][j+2] != 15) return false;
        if (grid[i][j+2] + grid[i+1][j+1] + grid[i+2][j] != 15) return false;
        return true;
    }
public:
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        const int n = grid.size();
        const int m = grid[0].size();
        if (n < 3 || m < 3) return 0;

        int res = 0;
        for (int i = 0; i < n-2; i++) {
            for (int j = 0; j < m-2; j++) {
                if (helper(grid, i, j)) res++;
            }
        }
        return res;
    }
};
