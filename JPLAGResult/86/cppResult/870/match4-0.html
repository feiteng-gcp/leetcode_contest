class Solution {
public:
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        if(grid.size()<3 || grid[0].size()<3)
        {
            return 0;
        }
        
        
        int m = grid.size();
        int n = grid[0].size();
        
        int result = 0;
        for(int i=2; i<m; i++)
        {
            for(int j=2; j<n; j++)
            {
                if(isM(grid, i, j))
                {
                    result++;
                }
            }
        }
        
        return result;
    }
    
    bool isM(vector<vector<int>>& grid, int i, int j)
    {   
        int d0 = grid[i][j] + grid[i-1][j-1]+ grid[i-2][j-2];
        int d1 = grid[i][j-2] + grid[i-1][j-1]+ grid[i-2][j];
        
        if(d0!=d1)
        {
            return false;
        }
        
        vector<bool> visit(10, false);
        
        for(int x=0; x<3; x++)
        {
            int sum = 0;
            for(int y=0; y<3; y++)
            {
                if(grid[i-x][j-y]<1 || grid[i-x][j-y]>9 || visit[grid[i-x][j-y]])
                {
                    return false;
                }
                
                visit[grid[i-x][j-y]] = true;
                sum += grid[i-x][j-y];
            }
            
            if(sum != d1)
            {
                return false;
            }
        }
        
         for(int y=0; y<3; y++)
        {
            int sum = 0;
            for(int x=0; x<3; x++)
            {
                sum += grid[i-x][j-y];
            }
            
            if(sum != d1)
            {
                return false;
            }
        }
        
        return true;
    }
};
