class Solution {
public:
    bool isMagic(int posX, int posY, vector<vector<int>> &grid) {
        unordered_set<int> seen;
        int sum = -1;
        bool hasSum = false;
        
        for (int i = posX; i < posX + 3; i++) {
            int iSum = 0;
            for (int j = posY; j < posY + 3; j++) {
                if (grid[i][j] > 9 || grid[i][j] < 1) return false;
                seen.insert(grid[i][j]);
                iSum += grid[i][j];
            }
            if (!hasSum) {
                sum = iSum;
                hasSum = true;
            }
            else {
                if (iSum != sum) return false;
            }
        }
        
        for (int i = posY; i < posY + 3; i++) {
            int iSum = 0;
            for (int j = posX; j < posX + 3; j++) {
                iSum += grid[j][i];
            }
            if (!hasSum) {
                sum = iSum;
                hasSum = true;
            }
            else {
                if (iSum != sum) return false;
            }
        }
        
        int iSum = grid[posX][posY] + grid[posX + 1][posY + 1] + grid[posX + 2][posY + 2];
        if (iSum != sum) return false;
        iSum = grid[posX + 2][posY] + grid[posX + 1][posY + 1] + grid[posX][posY + 2];
        if (iSum != sum) return false;
        return seen.size() == 9;
    }
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int count = 0;
        for (int i = 0; i <= (int)grid.size() - 3; i++) {
            for (int j = 0; j <= (int)grid[0].size() - 3; j++) {
                if (isMagic(i, j, grid)) count++;
            }
        }
        return count;
    }
};
