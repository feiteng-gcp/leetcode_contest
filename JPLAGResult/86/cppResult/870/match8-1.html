class Solution {
public:
    bool check(vector<vector<int>> &v, int x, int y) {
        int m = v.size(); int n = v[0].size();
        if (x + 2 >= m || y + 2 >= n) return false;
        unordered_set<int> s;

        for(int i=x; i<x+3; ++i) {
            int rs = 0;
            for(int j=y; j<y+3; ++j) {
                s.insert(v[i][j]);
                if (v[i][j] > 9 || v[i][j] < 0) return false;
                rs += v[i][j];
            }
            if (rs != 15) return false;
        }
        for(int i=y; i<y+3; ++i) {
            int cs = 0;
            for(int j=x; j<x+3; ++j) {
                cs += v[j][i];
            }
            if (cs != 15) return false;
        }
       
        if (v[x][y] + v[x+1][y+1] + v[x+2][y+2] != 15 || v[x+1][y+1] + v[x][y+2] + v[x+2][y] != 15) return false;
        return s.size() == 9;
    }
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int res = 0;
        if (grid.empty() || grid[0].empty()) return 0;

        for(int i=0; i<grid.size(); ++i) {
            for(int j=0; j<grid[0].size(); ++j) {
                if (check(grid, i, j)) res += 1;
            }
        }
        return res;
    }
};
