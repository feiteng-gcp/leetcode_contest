class Solution {
public:

bool check(vector<vector<int>>& grid, int i, int j) {
    vector<bool> m(10, false);

    for (int ii = i; ii < i + 3; ++ii) {
        int line = 0;
        for (int jj = j; jj < j + 3; ++jj) {
            if (grid[ii][jj] < 1 || grid[ii][jj] > 9 || m[grid[ii][jj]]) {
                return false;
            }
            line += grid[ii][jj];
            m[grid[ii][jj]] = true;
        }
        if (line != 15) {
            return false;
        }

    }
    for (int col = j; col < j + 3; ++col) {
        int sum = 0;
        for (int row = i; row < i + 3; ++row) {
            sum += grid[row][col];
        }
        if (sum != 15) {
            return false;
        }
    }
    if (15 != grid[i][j] + grid[i + 1][j + 1] + grid[i + 2][j + 2] ||
        15 != grid[i][j + 2] + grid[i + 1][j + 1] + grid[i + 2][j]) {
        return false;
    }
    return true;
}

int numMagicSquaresInside(vector<vector<int>>& grid) {
    const int n = grid.size();
    if (n < 3) {
        return 0;
    }

    int res = 0;
    for (int i = 0; i < n - 2; ++i) {
        for (int j = 0; j < n - 2; ++j) {
            if (check(grid, i, j)) {
                ++res;
            }
        }
    }
    return res;
}
};
