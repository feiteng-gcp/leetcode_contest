class Solution {
    int solve(int a, int b, vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        unordered_set<int> table;
        if (a+3 > n || b+3 > m) return 0;

        for (int i = a; i < a+3; i++) {
            int cnt = 0;
            for (int j = b; j < b+3; j++) {
                cnt += grid[i][j];
                if (grid[i][j] > 9 || grid[i][j] <= 0) return 0;
                table.insert(grid[i][j]);
            }
            if (cnt != 15) return 0;
        }

        if (table.size() != 9) return 0;
        for (int j = b; j < b+3; j++) {
            int cnt = 0;
            for (int i = a; i < a+3; i++) {
                cnt += grid[i][j];
            }
            if (cnt != 15) return 0;
        }
        int cnt = 0;
        for (int k = 0; k < 3; k++) {
            cnt += grid[a+k][b+k];
        }
        if (cnt != 15) return 0;
        cnt = 0;
        for (int k = 0; k < 3; k++) {
            cnt += grid[a+k][b+2-k];
        }
        if (cnt != 15) return 0;
        return 1;
    }
public:

    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int ret = 0;
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[0].size(); j++) {
                ret += solve(i, j, grid);
            }
        }
        return ret;
        
        
    }
};
