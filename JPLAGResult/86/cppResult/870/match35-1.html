class Solution {
public:
    
    
    int a[3][3];
    int b[10];
    
    int check() {
        int flag = 1;
        
        memset(b, 0, sizeof b);
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (a[i][j] <= 0 || a[i][j] > 9) {
                    return 0;
                }
                b[a[i][j]] = 1;
            }

        }
        
        for (int i = 1; i < 10; i++) if (b[i] == 0) return 0;
        
        for (int i = 0; i < 3; i++) {
            int sum = 0;
            for (int j = 0; j < 3; j++) {
                sum += a[i][j];
            }
            if (sum != 15) return 0;
        }
        
        for (int i = 0; i < 3; i++) {
            int sum = 0;
            for (int j = 0; j < 3; j++) {
                sum += a[j][i];
            }
            if (sum != 15) return 0;
        }
        
        
        if ((a[0][0] + a[1][1] + a[2][2]) != 15) {
            return 0;
        }
        
        if ((a[0][2] + a[2][0] + a[1][1]) != 15) {
            return 0;
        }
        
        return 1;
        
    }
    
    int numMagicSquaresInside(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();

        int ans = 0;
        
        for (int i = 0; i <= n - 3; i++) {
            for (int j = 0; j <= m - 3; j++) {
                for (int x = 0; x < 3; x++) {
                    for (int y = 0; y < 3; y++) {
                        a[x][y] = grid[x+i][y+j];
                    }
                }
                ans += check();
            }
        }
        
        return ans;
        
    }
};
