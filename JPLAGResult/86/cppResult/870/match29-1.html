class Solution {
public:
    int numMagicSquaresInside(vector<vector<int>>& g) {
        int m = g.size();
        if(m < 3) return 0;
        int n = g[0].size();
        if(n < 3) return 0;
        

        int res = 0;
        for(int i = 0; i <= m - 3; ++i) {
            for(int j = 0; j <= n - 3; ++j) {
                if(check(g, i, j)) ++res;
            }
        }
        return res;
    }
    
    bool check(vector<vector<int>>& g, int x, int y) {
        unordered_set<int> s;

        for(int i = x; i <= x + 2; ++i) {
            for(int j = y; j <= y + 2; ++j) {
                if(g[i][j] < 1 || g[i][j] > 9) return false;
                s.insert(g[i][j]);
            }
        }
        
        if(s.size() < 9) return false;
        
        int a[8] = {};
        a[0] = g[x][y] + g[x + 1][y] + g[x + 2][y];
        a[1] = g[x][y + 1] + g[x + 1][y + 1] + g[x + 2][y + 1];
        a[2] = g[x][y + 2] + g[x + 1][y + 2] + g[x + 2][y + 2];
        a[3] = g[x][y] + g[x][y + 1] + g[x][y + 2];
        a[4] = g[x + 1][y] + g[x + 1][y + 1] + g[x + 1][y + 2];
        a[5] = g[x + 2][y] + g[x + 2][y + 1] + g[x + 2][y + 2];
        a[6] = g[x][y] + g[x + 1][y + 1] + g[x + 2][y + 2];
        a[7] = g[x + 2][y] + g[x + 1][y + 1] + g[x][y + 2];
        
        for(int i = 0; i < 8; ++i) if(a[i] != 15) return false;
        return true;
    }
};
