/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * class Master {
 *   public:
 *     int guess(string word);
 * };
 */
class Solution {
public:
    int ok(string &x, string &y) {
        int tot = 0;
        for (int i = 0; i < 6; ++i)
            if (x[i] == y[i])
                ++tot;
        return tot;
    }
    
    void find(vector<string> &w, Master &master) {
        if (w.size() <= 0)
            return;
        int sel = rand() % w.size();
        int x = master.guess(w[sel]);
        assert(x >= 0);
        vector<string> mk;
        for (int i = 0; i < w.size(); ++i) {
            if (sel != i && ok(w[sel], w[i]) == x)
                mk.push_back(w[i]);
        }
        find(mk, master);
    }
    
    void findSecretWord(vector<string>& wordlist, Master& master) {
        /*
        abcdef
        ghijkl
        mnopqr
        stuvwx
        yxabcd
        efghij
        klmnop
        rstuvw
        xyzabc
        defghi
        jklmno
        pqrstu
        vwxyza
        */
        find(wordlist, master);
    }
};
