/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * class Master {
 *   public:
 *     int guess(string word);
 * };
 */
class Solution {
    int getMatchCount(string &a, string &b) {
        int ret = 0;
        for (int i = 0; i < 6; i++) {
            if (a[i] == b[i]) {
                ret += 1;
            }
        }
        return ret;
    }
    
    vector<string> populate(vector<string> words, vector< pair<string, int> > guessed) {
        vector<string> ret;
        for (string ans : words) {
            bool ok = 1;
            for (auto x : guessed) {
                if (getMatchCount(x.first, ans) != x.second) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                ret.push_back(ans);
            }
        }
        return ret;
    }
    
public:
    void findSecretWord(vector<string>& wordlist, Master& master) {
        sort(wordlist.begin(), wordlist.end());
        vector< pair<string, int> > guessed;
        set<string> G;
        int seed = 42;
        srand(seed);
        
        int counter = 0;
        while(counter < 10 && wordlist.size() > 1) {
            int N = wordlist.size();
            int id = (rand() % N + N) % N;
            if (G.count(wordlist[id])) continue;
            
            counter += 1;
            int x = master.guess(wordlist[id]);
            guessed.push_back({wordlist[id], x});
            G.insert(wordlist[id]);
            
            wordlist = populate(wordlist, guessed);
        }
        master.guess(wordlist[0]);
    }
};
