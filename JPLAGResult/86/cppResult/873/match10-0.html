/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * class Master {
 *   public:
 *     int guess(string word);
 * 
 */
class Solution {
public:
    int g[111][111];
    inline int mat(string& a,string& b){
        int res=0;
        for(int i=0;i<a.size();i++) if(a[i]==b[i]) res++;
        return res;
    }
    void findSecretWord(vector<string>& wl, Master& ms) {
        int N=wl.size();
        srand(time(0));
        set<pii> s;
        for(int i=0;i<N;i++){
            s.insert({rand(),i});
            for(int j=0;j<N;j++){
                g[i][j]=mat(wl[i],wl[j]);
            }
        }
        while(!s.empty()){
            int ix=s.begin()->snd;
            s.erase(s.begin());
            int k=ms.guess(wl[ix]);
            cout<<wl[ix]<<' '<<k<<'\n';
            if(k==6) return;
            auto it=s.begin();
            while(it!=s.end()){
                if(k!=g[ix][it->snd]) s.erase(it++);
                else it++;
            }
        }

    }
};
