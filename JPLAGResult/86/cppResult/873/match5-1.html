/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * class Master {
 *   public:
 *     int guess(string word);
 * };
 */
class Solution {

    bool valid(string word,string target,int r){
        int count=0;
        for(int i=0;i<6;i++)
            if(word[i]==target[i])
                count++;
        return r==count;
    }
public:
    void findSecretWord(vector<string>& wordlist, Master& master) {
        int times=10;
        while(times--){            
            string s = wordlist[0];
            swap(wordlist[0], wordlist.back());
            wordlist.pop_back();
            int r=master.guess(s);
            if (r == 6)
                return;
            for (int i = 0; i < wordlist.size();) {
                if (!valid(wordlist[i], s, r)) {
                    swap(wordlist[i], wordlist.back());
                    wordlist.pop_back();
                }
                else
                    i++;
            }
        }
    }
};
