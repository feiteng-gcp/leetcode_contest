/**
 * // This is the Master's API interface.
 * // You should not implement it, or speculate about its implementation
 * class Master {
 *   public:
 *     int guess(string word);
 * };
 */
class Solution {

    int count(string & s1, string & s2) {
        int cnt = 0;
        for (int i = 0; i < s1.size(); i++) {
            if (s1[i] == s2[i]) cnt++;
        }
        return cnt;
    }
    void solve(int cnt, vector<string>& wordlist, Master& master) {
        if (wordlist.empty() || cnt == 0) return;
        srand( (unsigned)time( NULL ) );
        string s2 = wordlist[rand() % (int)wordlist.size()];
        int ret = master.guess(s2);
        if (ret == 6) return;
        vector<string> nxt;
        for (string s : wordlist) {
            if (ret == count(s, s2)) nxt.push_back(s);
        }
        solve(cnt-1, nxt, master);
    }
public:
    void findSecretWord(vector<string>& wordlist, Master& master) {
        solve(10, wordlist, master);
        
    }
};
