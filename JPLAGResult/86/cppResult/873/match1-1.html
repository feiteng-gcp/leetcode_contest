class Solution {
public:
    int same(const string &s, const string &t) {
        int cnt = 0;
        for (int i = 0; i < s.size(); ++i)
            if (s[i] == t[i]) ++cnt;
        return cnt;
    }
    void findSecretWord(vector<string>& words, Master& master) {
        int n = words.size(), i, cnt, round = 10;
        vector<bool> possible(n, true);
        while (round--) {
            i = rand() % n;
            while (!possible[i%n]) i = (i+1)%n;
            possible[i] = false;
            cnt = master.guess(words[i]);
            if (cnt == 6) {
                break;
            }
            for (int j = (i+1)%n; j != i; j = (j+1)%n) {
                if (possible[j] && cnt != same(words[i], words[j]))
                    possible[j] = false;
            }
        }
    }
};
