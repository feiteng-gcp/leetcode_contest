#include <queue>
using namespace std;

class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = 0;
        
        vector<int> visit(rooms.size(), 0);
        queue<int> que;
        que.push(0);
        visit[0] = 1;
        
        while(!que.empty())
        {
            int u = que.front();que.pop();
            n ++;
            for(int i = 0; i < (int)rooms[u].size(); i ++)
                if (!visit[rooms[u][i]]) {
                    visit[rooms[u][i]] = 1;
                    que.push(rooms[u][i]);
                }
        }
        
        return n == rooms.size();
    }
};
