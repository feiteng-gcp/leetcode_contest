class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        vector<bool> visit(n);
        queue<int> q;
        q.push(0);
        visit[0] = true;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (auto v : rooms[u]) {
                if (!visit[v]) {
                    visit[v] = true;
                    q.push(v);
                }
            }
        }

        for (int i = 0; i < n; ++i) {
            if (!visit[i]) return false;
        }
        return true;
    }
};
