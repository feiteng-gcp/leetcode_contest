class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        unordered_set<int> sn;
        visitRooms(rooms, 0, sn);
        return sn.size()==rooms.size();
    }
    void visitRooms(vector<vector<int>>& rooms, int start, unordered_set<int> &seen) {
        seen.insert(start);
        for(int key : rooms[start])
            if(seen.find(key)==seen.end())
                visitRooms(rooms, key, seen);
    }
};
