class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        unordered_set<int> visited;
        dfs(rooms, visited, 0);
        return visited.size() == rooms.size();
    }
    
    void dfs(vector<vector<int>>& rooms, unordered_set<int> &visited, int u) {
        visited.insert(u);
        for (int v : rooms[u]) {
            if (visited.count(v) == 0) {
                dfs(rooms, visited, v);
            }
        }
    }
};
