class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        deque<int> q;
        vector<bool> v(rooms.size(), false);
        v[0] = true;
        for (auto k : rooms[0]) {
            q.push_back(k);
            v[k] = true;
        }
        

        while (!q.empty()) {
            int k = q.front();
            q.pop_front();
            
            for (auto n : rooms[k]) {
                if (!v[n]) {
                    v[n] = true;
                    q.push_back(n);
                }
            }
        }
        
        for (auto t : v) {
            if (!t) return false;
        }
        
        return true;
    }
};
