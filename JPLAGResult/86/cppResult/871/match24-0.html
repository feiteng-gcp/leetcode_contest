class Solution {
private:
    bool opened[1010];
    vector<vector<int>> rooms;
public:
    void dfs(int curRoom) {
        if (opened[curRoom]) return;
        opened[curRoom] = true;
        for (int i=0; i<rooms[curRoom].size(); i++) {
            dfs(rooms[curRoom][i]);
        }
    }
    bool canVisitAllRooms(vector<vector<int>>& rs) {
        memset(opened, false, sizeof opened);
        rooms = rs;
        dfs(0);
        for (int i=0; i<rooms.size(); i++) if (!opened[i]) return false;
        return true;
    }
};
