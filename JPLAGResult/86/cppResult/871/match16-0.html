class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        vector<bool> vis(rooms.size(), false);
        queue<int> q;
        q.push(0);
        vis[0] = true;
        while(!q.empty()){
            int cur = q.front();
            q.pop();
            for(int k: rooms[cur]){
                if(!vis[k]){
                    q.push(k);
                    vis[k] = true;
                }
            }
        }
        for(int i = 1; i < rooms.size(); ++i)
            if(!vis[i]) return false;
        return true;
    }
};
