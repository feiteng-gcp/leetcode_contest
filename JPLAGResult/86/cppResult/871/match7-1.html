class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        vector<bool> vis = vector<bool>(n, false);
        queue<int> q;
        q.push(0); vis[0] = true;
        while (!q.empty()) {
            int cur = q.front(); q.pop();

            for (auto nxt : rooms[cur]) {
                if (vis[nxt]) continue;
                q.push(nxt); vis[nxt] = true;
            }
        }
        for (int i = 0; i < n; i++) {
            if (!vis[i]) return false;
        }
        return true;
        
    }
};
