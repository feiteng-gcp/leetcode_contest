class Solution {
public:
	bool canVisitAllRooms(vector< vector<int> > & rooms) {
		int N = rooms.size();
		vector<bool> visited(N, false);
		queue<int> qt;
		qt.push(0);
		visited[0] = true;
		while(!qt.empty()) {
			int cur = qt.front();
            qt.pop();
			vector<int> & next = rooms[cur];
			for(int i = 0; i < next.size(); i ++) {
				int & one = next[i];

				if(visited[one]) continue;
				qt.push(one);
				visited[one] = true;
			}
		}
		for(int i = 0; i < N; i ++) {
			if(!visited[i]) return false;
		}
		return true;
	}
};
