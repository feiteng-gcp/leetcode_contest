class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        const int n = rooms.size();
        queue<int> q;
        unordered_set<int> s;
        for (auto &i : rooms[0]) {
            if (i != 0) {
                q.push(i);
                s.insert(i);
            }
        }
        if (s.size() == n-1) return true;
        while (!q.empty()) {
            int f = q.front();
            q.pop();
            for (auto &i : rooms[f]) {
                if (i != 0 && s.find(i) == s.end()) {
                    q.push(i);
                    s.insert(i);
                }
            }
        }
        return s.size() == n-1;
    }
};
