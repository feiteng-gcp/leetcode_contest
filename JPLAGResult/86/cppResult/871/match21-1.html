class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        vector<int> visited(n, 0);
        queue<int> q;
        q.push(0);
        visited[0] = 1;
        while (!q.empty()) {
            int k = q.front();
            q.pop();            
            for (int r : rooms[k]) {
                if (visited[r])
                    continue;
                q.push(r);
                visited[r] = 1;
            }
        }
        bool res = true;
        for (int i : visited)
            res &= i;
        return res;
    }
};
