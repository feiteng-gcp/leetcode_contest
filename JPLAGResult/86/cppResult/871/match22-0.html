class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int N = rooms.size();
        vector<bool> visited(N, 0);
        
        visited[0] = true;
        queue<int> bfs;
        bfs.push(0);
        
        while(!bfs.empty()) {
            int u = bfs.front(); bfs.pop();
            for (int v: rooms[u]) {
                if (visited[v] == false) {
                    visited[v] = true;
                    bfs.push(v);
                }
            }
        }
        
        for (bool b: visited) {
            if (b == false) {
                return false;
            }
        }
        
        return true;
    }
};
