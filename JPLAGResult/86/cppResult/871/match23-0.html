class Solution {
public:
    vector<int> key;
    int n;
    queue<int> qu;
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        n=rooms.size();
        key.assign(n,0);
        qu.push(0);
        key[0]=1;
        while(!qu.empty())
        {
            int u=qu.front(); qu.pop();
            for(int v:rooms[u])
            {
                if(!key[v])
                {
                    key[v]=1;
                    qu.push(v);
                }
            }
        }
        int f=1;
        for(int i=0;i<n;i++)
            if(!key[i]) f=0;
        return f;
    }
};
