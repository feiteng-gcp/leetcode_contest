class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        vector<bool> mem(n, false);
        queue<int> Q;
        Q.push(0);
        mem[0] = true;

        while (!Q.empty()) {
            int room = Q.front();
            Q.pop();

            for (int k : rooms[room]) {
                if (mem[k]) continue;
                Q.push(k);
                mem[k] = true;
            }
        }

        for (int i = 0; i < n; ++i) {
            if (!mem[i]) return false;
        }
        
        return true;
    }
};
