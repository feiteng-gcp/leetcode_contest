class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        int tot = 1;
        vector<int> key(n, 0);
        key[0] = 1;
        queue<int> que;
        que.push(0);
        while (que.size()) {
            int cur = que.front();
            que.pop();
            for (auto v : rooms[cur]) {
                if (!key[v]) {
                    tot++;
                    key[v] = 1;
                    que.push(v);
                }
            }
        }
        return tot == n;
    }
};
