class Solution {
public:
    
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int V = rooms.size();
        vector<bool> seen(V + 5, 0);
        queue<int> Q;
        Q.push(0);
        seen[0] = 1;
        
        while( !Q.empty() ){
            int cur = Q.front(); Q.pop();
            for( int i = 0 ; i < rooms[cur].size() ; ++i ){
                int v = rooms[cur][i];
                if( !seen[v] ){
                    seen[v] = 1;
                    Q.push(v);
                }
            }
        }
        
        for( int i = 0 ; i < V ; ++i ) if( !seen[i]) return false;
        return true;
        
    }
};
