int vis[1005];
class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        memset(vis, 0, sizeof vis);
        
        queue<int> q;
        q.push(0);

        vis[0] = 1;
        
        while(!q.empty()) {
            int u = q.front();
            q.pop();
            for (int i = 0; i < rooms[u].size(); i++) {
                int v = rooms[u][i];
                if (vis[v] == 0) {
                    q.push(v);
                    vis[v] = 1;
                }
            }
        }
        
        for (int i = 0; i < rooms.size(); i++) {
            if (vis[i] == 0) return 0;
        }
        return 1;
    }
};
