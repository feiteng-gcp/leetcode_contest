class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        if (n<2) return true;
        int v[1500]={1}, counter = 1;
        queue<int> q;
        q.push(0);
        while (counter < n&&!q.empty()) {

            int tmp = q.front();
            q.pop();
            for (auto t:rooms[tmp]) {
                if (!v[t]) {
                    counter++;
                    q.push(t);
                    v[t]=1;
                    //cout<< t<<' '<< counter<<endl;
                }
            }
        }
        return !(n-counter);
    }
};
