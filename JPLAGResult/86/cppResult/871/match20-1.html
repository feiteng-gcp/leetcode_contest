class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        if(rooms.size()<2)
        {
            return true;
        }
        
        int n = rooms.size();
        
        vector<bool> visit(n ,false);
        queue<int> q;
        
        q.push(0);
        visit[0] = true;
        
        int left = n-1;
        
        while(q.size()>0)
        {

            auto cur = q.front(); q.pop();
            
            for(auto next : rooms[cur])
            {
                if(!visit[next])
                {
                    visit[next] = true;
                    q.push(next);
                    left--;
                }
            }
        }
        
        return left ==0;
    }
};
