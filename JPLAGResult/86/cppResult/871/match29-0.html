typedef long long ll;
#define rep(i,a,b) for (int i = (a); i <= (b); ++i)
#define rep2(i,a,b) for (int i = (a); i >= (b); --i)
class Solution {
    public:
    bool b[10100];
    void dfs(int k, vector<vector<int>>& a){
        b[k] = true;
        for (auto i: a[k]){
            if (!b[i]){
                dfs(i, a);
            }
        }
    }
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        memset(b, false, sizeof(b));
        int n = rooms.size();
        if (n==0){
            return true;
        }
        int m = rooms[0].size();
        dfs(0, rooms);
        rep (i,0,n-1){
            if (!b[i]){
                return false;
            }
        }
        return true;
    }
};
