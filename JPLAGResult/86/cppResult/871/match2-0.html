class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& a) {
        int n = a.size();
        queue<int> Q;
        Q.push(0);
        vector<bool> visit(n);
        visit[0] = true;
        while (!Q.empty()) {
            int u = Q.front();
            Q.pop();
            for (auto& v : a[u]) {
                if (!visit[v]) {
                    visit[v] = true;
                    Q.push(v);
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) ret += visit[i];
        return ret == n;
        
    }
};
