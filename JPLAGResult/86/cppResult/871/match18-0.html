class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        int visit = 1;
        vector<bool> v(n + 1, false);
        v[0] = true;
        
        queue<int> q;
        q.push(0);
        while (!q.empty()) {
            int now = q.front();
            q.pop();
            
            for (int i = 0; i < rooms[now].size(); ++i) {
                int next = rooms[now][i];
                if (v[next] == false) {
                    v[next] = true;
                    ++visit;
                    q.push(next);
                }
            }
        }
        
        return (visit == n);
    }
};
