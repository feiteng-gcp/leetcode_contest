const long long TOPMAX = (1LL<<31);

class Solution {
public:
    long long string2int(string S) {
        long long x = 0;
        for(int i = 0; i < S.length(); i ++) {
            x = x * 10 + S[i] - '0';
            if (x >= TOPMAX) return -1;
        }
        if (x != 0 && S[0] == '0') return -1;
        return x;
    }
    vector<int> splitIntoFibonacci(string S) {
        for(int i = 1; i <= S.length(); i ++) {
            for(int j = i+1; j <= S.length(); j ++) {
                long long a = string2int(S.substr(0, i));
                long long b = string2int(S.substr(i, j-i));
                if (a < 0 || b < 0) continue;
                
                vector<int> ans;
                ans.push_back(a);
                ans.push_back(b);
                int pos = j;
                while(pos < S.length()) {
                    long long c = a + b;
                    if (c >= TOPMAX) break;
                    
                    int k;
                    for(k = pos+1; k <= S.length(); k ++)
                        if (c == string2int(S.substr(pos, k-pos))) {
                            ans.push_back(c);
                            a = b;
                            b = c;
                            pos = k;
                            break;
                        }
                    if (k > S.length()) break;
                }
                if (pos == S.length() && ans.size() >= 3) return ans;
            }
        }
        return vector<int>();
    }
};
