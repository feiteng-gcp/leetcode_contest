class Solution {
public:
    int getnum(int *a, int len) {
        if (len == 0) return -1;
        if (a[0] == 0) return len == 1 ? 0 : -1;
        int ret = 0;
        for (int i = 0; i < len; i++) {
            if (10LL * ret + a[i] > 2147483647) return -1;
            ret = ret * 10 + a[i];
        }
        return ret;
    }
    
    int scan(int *a, int len, int f) {
        if (len == 0) return -1;
        if (f == 0) {
            if (a[0] != 0) return -1;
            return 1;
        }
        int x = 0;
        for (int i = 0; i < len; i++) {
            if (10LL * x + a[i] > 2147483647) return -1;
            x = x * 10 + a[i];
            if (x == f) return i + 1;

        }
        return -1;
    }
    
    vector<int> splitIntoFibonacci(string S) {
        int a[205];
        int n = S.length();
        for (int i = 0; i < n; i++) {
            a[i] = S[i] - 48;
        }
        for (int i = 1; i < n; i++) {
            int _f0 = getnum(a, i);
            if (_f0 == -1) {
                break;
            }
            for (int j = i + 1; j < n; j++) {
                int f1 = getnum(a + i, j - i);
                if (f1 == -1) {
                    break;
                }
                vector<int> ret(2);
                int f0 = _f0;
                ret[0] = f0;
                ret[1] = f1;
                int pos = j;
                while (1) {
                    int f2 = f0 + f1;
                    if (f2 < f0 || f2 < f1) break;
                    int tmp = scan(a + pos, n - pos, f2);
                    if (tmp == -1) break;
                    pos += tmp;
                    ret.push_back(f2);
                    f0 = f1;
                    f1 = f2;
                    if (pos == n) {
                        return ret;
                    }
                }
            }
        }
        return vector<int>();
    }
};
