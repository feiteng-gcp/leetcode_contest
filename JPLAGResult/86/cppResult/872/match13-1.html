typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
    vector<int> test(string& s,ll a,ll b){
        string tmp;
        ll c=a+b;
        tmp+=to_string(a);
        tmp+=to_string(b);
        tmp+=to_string(c);
        vector<int> res{a,b,c};
                // cout<<a<<' '<<b<<'\n';

        while(tmp.size()<s.size()){
            if(c>INT_MAX) return {};
            tie(a,b,c)=make_tuple(b,c,b+c);
            tmp+=to_string(c);
            res.pb(c);
        }
                // cout<<a<<' '<<b<<' '<<tmp<<'\n';

        if(c<=INT_MAX && tmp==s) return res;

        else return {};
    }

    vector<int> splitIntoFibonacci(string s) {
        int N=s.size();
        for(int i=1;i<=10;i++){
            if(i>1 && s[0]=='0') break;
            if(i>=N) break;
            for(int j=1;j<=10;j++){
                if(j>1 && s[i]=='0') break;
                if(i+j>=N) break;
                auto res=test(s,stoll(s.substr(0,i)),stoll(s.substr(i,j)));
                if(!res.empty()) return res;
            }
        }
        return {};
    }
};
