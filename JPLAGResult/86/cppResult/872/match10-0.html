class Solution {
public:
    string s;
    vector<int> fi;
    int f=0;
    int n,m=0;
    
    vector<int> splitIntoFibonacci(string S) {
        if(S.size()<3) return fi;
        s=S; n=S.length();
        long long nu1=0,nu2=0;
        for(int i=0;i<min(12,(int)s.length()-2);i++)
        {
            nu1=nu1*10+S[i]-'0';
            if(i>=1&&nu1<10) break;
            nu2=0;
            for(int j=0;j<min(12,(int)s.length()-i-1);j++)
            {
                nu2=nu2*10+S[i+1+j]-'0';
                if(j>=1&&nu2<10) break;
                if(nu1>INT_MAX||nu2>INT_MAX) break;
                fi.clear();
                fi.push_back(nu1); fi.push_back(nu2);
                int k=i+j+2;
                for(int j=2;;j++)
                {
                    if((long long)fi[j-1]+fi[j-2]>INT_MAX)
                    {
                        f=0;
                        break;
                    }
                    fi.push_back(fi[j-1]+fi[j-2]);
                    cout<<fi[i]<<endl;
                    string tmp=to_string(fi[j]);
                    //cout<<fi[j]<<endl;
                    if(k+tmp.length()>s.length()||s.substr(k,tmp.length())!=tmp)
                    {
                        f=0;
                        break;
                    }
                    k+=tmp.length();
                    if(k==n)
                    {
                        f=1;
                        break;
                    }
                }
                if(f) break;
            }
            if(f) break;
            //cout<<endl;
        }
        if(!f) fi.clear();
        return fi;
    }
};
