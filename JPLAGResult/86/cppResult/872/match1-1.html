class Solution {
public:
bool check(const string& s, int i, int j, vector<int>& res) {
    int a = atoi(s.substr(0, i).c_str());
    int b = atoi(s.substr(i, j - i).c_str());
    if (to_string(a) != s.substr(0, i) ||
        to_string(b) != s.substr(i, j - i)) {
        return false;
    }
    res.push_back(a);
    res.push_back(b);
    while (j < s.size()) {
        int c = a + b;
        string cs = to_string(c);
        if (s.substr(j, cs.size()) != cs) {
            res.clear();
            return false;
        }
        res.push_back(c);

        a = b;
        b = c;

        j += cs.size();
    }
    return true;
}

vector<int> splitIntoFibonacci(string S) {
    vector<int> res;
    for (int i = 1; i < S.size() - 1; ++i) {
        for (int j = i + 1; j < S.size(); ++j) {
            if (check(S, i, j, res)) {
                return res;
            }
        }
    }
    return res;
}
};
