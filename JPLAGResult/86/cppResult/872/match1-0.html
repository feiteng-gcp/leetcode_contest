class Solution {
public:
	bool isFibonacci(string S, vector<int>& ans, int pos) {
		while (pos < S.length()) {
			int firstNum = ans[ans.size() - 2], secondNum = ans[ans.size() - 1];
			int nextNum = firstNum + secondNum;
			string expected = to_string(nextNum);

			if (pos + expected.length() > S.length()) {
				return false;
			}
			
			string actual = S.substr(pos, expected.length());
			if (actual != expected) {
				return false;
			}

			ans.push_back(nextNum);
			pos += expected.length();
		}

		return pos == S.length() && ans.size() >= 3;
	}

	vector<int> splitIntoFibonacci(string S) {
		vector<int> ans;
		for (int i = 0; i < S.length(); i++) {
			for (int j = i + 1; j < S.length(); j++) {
				string firstStr = S.substr(0, i + 1), secondStr = S.substr(i + 1, j - i);
				int firstNum = stringToInt(firstStr), secondNum = stringToInt(secondStr);
				
				if (firstNum == -2 || secondNum == -2) {
					break;
				}
				if (firstNum == -1 || secondNum == -1) {
					continue;
				}

				ans.push_back(firstNum);
				ans.push_back(secondNum);

				if (isFibonacci(S, ans, j + 1)) {
					return ans;
				}
				ans.clear();
			}
		}

		return ans;
	}

	int stringToInt(string s) {
		if (s.length() > 1 && s.front() == '0') {
			return -1;
		}

		int num;
		try {
			num = stoi(s);
		}
		catch (...) {
			num = -2;
		}
		return num;
	}
};
