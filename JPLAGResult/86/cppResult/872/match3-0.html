class Solution {
public:
    
    long valid(string &S, int x, int y) {
        if (S[x] == '0' && x + 1 < y)
            return -1;
        return atol(S.substr(x, y - x).c_str());
    }
    
    vector<int> splitIntoFibonacci(string S) {
        for (int i = 1; i < S.size(); ++i) {
            for (int j = i + 1; j < S.size(); ++j) {
                long x = valid(S, 0, i);
                long y = valid(S, i, j);
                if (x < 0 || y < 0)
                    continue;
                string mk = S.substr(0, j);
                vector<int> ans = {x, y};
                while (mk.size() < S.size()) {
                    long z = x + y;
                    if (z >= (1L << 31))
                        break;
                    string a = to_string(z);
                    if (mk.size() + a.size() > S.size() || S.substr(mk.size(), a.size()) != a)
                        break;
                    mk += a;
                    ans.push_back(z);
                    x = y, y = z;
                }
                if (mk == S) {
                    return ans;
                }
            }
        }
        return {};
    }
};
