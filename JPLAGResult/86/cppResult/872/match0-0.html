class Solution {
    vector<int> res;
public:
    vector<int> splitIntoFibonacci(string S) {
        if (dfs(S, 0))
            return res;
        else
            return vector<int>();  
    }
    bool dfs(string &S, int s) {
        if (s==S.size())
            return res.size()>2;
        for (int l=1; l<=S.size()-s; ++l) {
            if (l>1 && S[s]=='0')
                return false;
            long n = stol(S.substr(s,l));
            if (n>=INT_MAX)
                break;
            if (res.size()>=2) {
                if (n > res.back()+res[res.size()-2])
                    break;
                else if (n < res.back()+res[res.size()-2])
                    continue;  
            }
            res.push_back((int)n);
            if (dfs(S, s+l))
                return true;
            else
                res.pop_back();
        }
        return false;
    }
};
