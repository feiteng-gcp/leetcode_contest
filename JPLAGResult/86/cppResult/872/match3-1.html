class Solution {
    typedef long long LL;
    vector<LL> solve(string & s, int i, int j) {
        vector<LL> ret;
        if (i < 1 || i > 10) return ret;
        if (j-i < 1 || j-i > 10) return ret;
        if (s[0] == '0' && i != 1) return ret;
        if (s[i] == '0' && j != i+1) return ret;
        ret.push_back(stoll(s.substr(0, i)));
        ret.push_back(stoll(s.substr(i, j-i)));
        int a = j, n = s.size();
        while (a < n) {
            LL tmpn = ret.back() + ret[ret.size()-2];
            if (tmpn > INT_MAX) return vector<LL>();
            string tmps = to_string(tmpn);
            if (tmps.size()+a > n) return vector<LL>();
            if (s.substr(a, tmps.size()) != tmps) return vector<LL>();
            a += tmps.size();
            ret.push_back(tmpn);
        }
        return ret;
        
    }
public:

    vector<int> splitIntoFibonacci(string s) {
        for (int i = 0; i < s.size(); i++) {
            for (int j = i+1; j < s.size(); j++) {
                vector<LL> tmp = solve(s, i, j);
                if (tmp.empty()) continue;
                bool flag = true;
                vector<int> ret;
                for (LL num : tmp) {
                    if (num > INT_MAX) {
                        flag = false;
                        break;
                    }
                    ret.push_back((int)num);
                }
                if (flag) return ret;
            }
        }
        return vector<int>();
        
    }
};
