class Solution {
public:
	vector<int> splitIntoFibonacci(string s) {
		int n = s.size();
		vector<int> res;

		for (int i = 0; i < 10 && i < n; ++i) {
			long num1 = stol(s.substr(0, i + 1));
			if (s[0] == '0' && i != 0) continue;
			if (num1 >= 2147483648) continue;
			for (int j = i + 1; j < i + 10 && j < n; ++j) {
				long num2 = stol(s.substr(i + 1, j - i));
				if (num2 >= 2147483648) continue;
				if (s[i + 1] == '0' && j != i + 1) continue;

				string str = s.substr(j + 1);
				res.push_back(num1);
				res.push_back(num2);
				long n1 = num1, n2 = num2;
				bool f = str.size() > 0;
				while (str.size()) {

					long num3 = n1 + n2;
					if (num3 >= 2147483648) {
						f = false;
						break;
					}
					string t = to_string(num3);
					if (str.size() < t.size() || str.substr(0, t.size()) != t) {
						f = false;
						break;
					}
					str = str.substr(t.size());
					res.push_back(num3);
					n1 = n2;
					n2 = num3;
				}
				if (f) {
					return res;
				}
				res.clear();
			}
		}
		return res;
	}
};
