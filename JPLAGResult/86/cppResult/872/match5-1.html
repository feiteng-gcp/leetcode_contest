class Solution {
public:
    vector<int> splitIntoFibonacci(string S) {        
        vector<int> ret;
        

        for (int i = 1; i < S.size() && stol(S.substr(0, i)) <= INT_MAX; ++i)
        {
            if (i > 1 && S[0] == '0') continue;
            ret.push_back(stoi(S.substr(0, i)));
                // printf("%d:\n", ret.back());
            
            for (int j = 1; j < S.size()-i && stol(S.substr(i, j)) <= INT_MAX; ++j)
            {
                if (j > 1 && S[i] == '0') continue;
                ret.push_back(stoi(S.substr(i, j)));
                
                // printf("\t%d:\n", ret.back());
                if (DFS(ret, S, i+j))
                    return ret;
                else
                    ret.resize(1);
            }
            
            ret.clear();
        }
        
        return vector<int>{};
    }
    
    bool DFS(vector<int>& v, string& S, int i)
    {
        if (i == S.size()) return true;
        if (i > S.size()) return false;
        
        int N = v.size(); 
        long next = v[N-1] + v[N-2];     
        string strNext = to_string(next);
        
        // printf("\n next %d, ind %d\n", next, i);    
        if (next > INT_MAX) return false;   
        if (S[i] == '0' && next != 0) return false;    
        if (strNext != S.substr(i, strNext.size())) return false;
        
        v.push_back(next);
        bool resNext = DFS(v, S, i+strNext.size());
        if (!resNext) 
        {
            v.resize(v.size()-1);
            return false;
        }
        else 
            return true;
    }
};
