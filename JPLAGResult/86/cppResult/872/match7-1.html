typedef long long int64;
class Solution {
public:
    int check(string& S, int k, int64 x, int64 y) {
        int ret = 2;
        while (k < S.size()) {
            int64 z = x + y;
            if (S[k] == '0' && z != 0) return 0;
            if (z > (1LL << 31) - 1) return 0;
            int64 sum = S[k] - '0';
            ++k;
            for (; k < S.size() && sum < z; ++k) sum = sum * 10 + (S[k] - '0');
            if (sum != z) return 0;
            x = y;
            y = z;
            ++ret;
        }
        return ret;
    }
    vector<int> splitIntoFibonacci(string S) {
        int len = S.length();

        for (int i = 1; i <= 10 && i < len; ++i) {
            if (S[0] == '0' && i > 1) continue;
            for (int j = 1; j <= 10 && i + j < len; ++j) {
                if (S[i] == '0' && j > 1) continue;
                int64 x = 0, y = 0;
                for (int u = 0; u < i; ++u) x = x * 10 + (S[u] - '0');
                for (int u = 0; u < j; ++u) y = y * 10 + (S[i + u] - '0');
                if (x > (1LL << 31) - 1 || y > (1LL << 31) - 1) continue;
                int m = check(S, i + j, x, y);
                if (m > 2) {
                    vector<int> ret(m);
                    ret[0] = x; ret[1] = y;
                    for (int u = 2; u < m; ++u) {
                        ret[u] = ret[u - 1] + ret[u - 2];
                    }
                    return ret;
                }
            }
        }
        return {};
    }
};
