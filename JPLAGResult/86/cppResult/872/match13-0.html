class Solution {
public:
    vector<int> splitIntoFibonacci(string S) {
        int n=S.size();
        if(n==0)
        {
            vector<int>p;
            return p;
        }
        vector<int>p;
        bool ans=cal(S,p,0,-1,-1);
        return p;
    }
    
    bool cal(string S,vector<int>&p,int m,int last1,int last2)
    {
        if(m>=S.size())
        {
            if(p.size()>=3)
                return true;
            return false;
        }
        bool ans=false;
        if(last1==-1)
        {
            int sum=0;
            for(int i=m;i<S.size();++i)
            {
             if(i>m&&S[m]=='0')
                    break;
                if(sum*10+S[i]-'0'>INT_MAX||sum*10+S[i]-'0'<0)
                    break;
                sum=sum*10+S[i]-'0';
                p.push_back(sum);
                ans=ans||cal(S,p,i+1,sum,-1);
                if(ans)
                    break;
                p.pop_back();
            }
        }else if(last2==-1){
            int sum=0;
            for(int i=m;i<S.size();++i)
            {
                  if(i>m&&S[m]=='0')
                    break;
               if(sum*10+S[i]-'0'>INT_MAX||sum*10+S[i]-'0'<0)
                    break;
            sum=sum*10+S[i]-'0';
                p.push_back(sum);
                ans=ans||cal(S,p,i+1,last1,sum);
                if(ans)
                    break;
                p.pop_back();
            }
        }else {
            int sum=0;
            for(int i=m;i<S.size();++i)
            {
                  if(i>m&&S[m]=='0')
                    break;
        
                if(sum*10+S[i]-'0'>INT_MAX||sum*10+S[i]-'0'<0)
                    break;
                sum=sum*10+S[i]-'0';
                if(sum!=last1+last2)
                    continue;
                p.push_back(sum);
                ans=ans||cal(S,p,i+1,last2,sum);
                if(ans)
                    break;
                p.pop_back();
            }
        }
        return ans;
    }
    
    
};
