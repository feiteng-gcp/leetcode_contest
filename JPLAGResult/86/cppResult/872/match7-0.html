class Solution {
public:
    vector<int> res;
    int strToInt(const string& S, int start, int length) {
        int num = 0;
        for (int i = start; i < start + length; ++i) {
            num *= 10;
            num += (S[i] - '0');
        }
        return num;
    }
    bool isFibonacci(const string& S, int first, int second, int start) {
        int third = first + second;
        int val = third, numDigits = 1;
        while (val >= 10) {
            val /= 10;
            ++numDigits;
        }
        if ((start + numDigits) > S.size()) return false;
        int thirdFromStr = strToInt(S, start, numDigits);
        if (third != thirdFromStr) return false;
        res.push_back(third);
        if ((start + numDigits) == S.size()) return true;
        return isFibonacci(S, second, third, start + numDigits);
    }
    vector<int> splitIntoFibonacci(string S) {
        for (int size1 = 1; size1 <= S.size() / 2; ++size1) {
            if (size1 > 1 && S[0] == '0') continue;
            for (int size2 = 1; min(size1, size2) <= (S.size() - size1 - size2); ++size2) {
                if (size2 > 1 && S[size1] == '0') continue;
                res.clear();
                int first = strToInt(S, 0, size1);
                int second = strToInt(S, size1, size2);
                res.push_back(first);
                res.push_back(second);
                if (isFibonacci(S, first, second, size1 + size2)) {
                    return res;
                }
            }
        }
        return {};
    }
};
