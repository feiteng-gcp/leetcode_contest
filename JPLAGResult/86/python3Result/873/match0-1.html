import random
class Solution:
    def findSecretWord(self, wordlist, master):
        """
        :type wordlist: List[Str]
        :type master: Master
        :rtype: None
        """
        def diff(a, b):
            count = 0
            for i in range(6):
                if a[i] != b[i]:
                    count += 1
            return count
        
        cur = set(wordlist)
        nxt = set()
        for _ in range(10):
            while cur:
                guess = random.sample(cur, 1)[0]
                cur.remove(guess)
                print(guess)
                match = master.guess(guess)
                if match == 6:
                    return
                for word in cur:
                    if diff(word, guess) == 6 - match:
                        nxt.add(word)
                cur = nxt
                nxt = set()
