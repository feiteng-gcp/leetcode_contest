class Solution:
    def findSecretWord(self, wordlist, master):
        """
        :type wordlist: List[Str]
        :type master: Master
        :rtype: None
        """

        n = len(wordlist)
        mat = [[set(), set(), set(), set(), set(), set(), set()] for _ in range(n)]
        for i in range(n):
            for j in range(i+1, n):
                s = 0
                for k in range(6):
                    if wordlist[i][k] == wordlist[j][k]: s+=1
                mat[i][s].add(wordlist[j])
                mat[j][s].add(wordlist[i])
        st = set(wordlist)
        mp = dict()
        for i in range(n): mp[wordlist[i]] = i
        while len(st) >= 1:
            n = len(st)
            s = st.pop()
            r = master.guess(s)
            st.intersection_update(mat[mp[s]][r])
