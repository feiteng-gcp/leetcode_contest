import random
class Solution:
    def findSecretWord(self, wordlist, master):
        """
        :type wordlist: List[Str]
        :type master: Master
        :rtype: None
        """
        m = {}
        l = len
(wordlist)
        for i in range(l - 1):
            for j in range(i + 1, l):
                r = 0
                for k in range(6):
                    if wordlist[i][k] == wordlist[j][k]:
                        r += 1
                if wordlist[i] in m:
                    if r in m[wordlist[i]]:
                        m[wordlist[i]][r].add(wordlist[j])
                    else:
                        m[wordlist[i]][r] = {wordlist[j]}
                else:
                    m[wordlist[i]] = {r : {wordlist[j]}}
                if wordlist[j] in m:
                    if r in m[wordlist[j]]:
                        m[wordlist[j]][r].add(wordlist[i])
                    else:
                        m[wordlist[j]][r] = {wordlist[i]}
                else:
                    m[wordlist[j]] = {r : {wordlist[i]}}

        s = set(wordlist)
        for i in range(10):
            t = random.choice(list(s))
            res = master.guess(t)
            if res == 6:
                return
            s = s & m[t][res]
