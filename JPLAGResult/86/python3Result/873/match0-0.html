import random

class Solution:
    def findSecretWord(self, wordlist, master):
        """
        :type wordlist: List[Str]
        :type master: Master
        :rtype: None
        """
        
        def count(a, b):
            ct = 0
            for i in range(6):
                if a[i] == b[i]: ct += 1
            return ct
        
        word = random.choice(wordlist)
        ct = master.guess(word)
        neighbor = set(wordlist)
        num = 0
        while ct != 6:
            num += 1
            new_neighbor = set()
            for nword in wordlist:
                if count(word, nword) == ct: new_neighbor.add(nword)
            neighbor = neighbor.intersection(new_neighbor)
            word = random.choice(list(neighbor))
            ct = master.guess(word)
        print(num)
