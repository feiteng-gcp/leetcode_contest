class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        count = 0
        for i in range(0, len(grid)-2, 1):
            for j in range(0, len(grid[i])-2, 1):
                isValid = True
                for k in range(i, i+3, 1):
                    for l in range(j, j+3, 1):
                        if grid[k][l] <= 0 or grid[k][l] >=10:
                            isValid = False
                            break
                    if not isValid:
                        break
                
                isValid = isValid and grid[i][j] + grid[i][j+1] + grid[i][j+2] == 15
                isValid = isValid and grid[i+1][j] + grid[i+1][j+1] + grid[i+1][j+2] == 15
                isValid = isValid and grid[i+2][j] + grid[i+2][j+1] + grid[i+2][j+2] == 15
                isValid = isValid and grid[i][j] + grid[i+1][j] + grid[i+2][j] == 15
                isValid = isValid and grid[i][j+1] + grid[i+1][j+1] + grid[i+2][j+1] == 15
                isValid = isValid and grid[i][j+2] + grid[i+1][j+2] + grid[i+2][j+2] == 15
                isValid = isValid and grid[i][j] + grid[i+1][j+1] + grid[i+2][j+2] == 15
                isValid = isValid and grid[i][j+2] + grid[i+1][j+1] + grid[i+2][j] == 15
                if isValid:
                    count += 1
        return count
