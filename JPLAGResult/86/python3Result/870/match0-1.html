class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        res = 0
        self.g = grid
        n, m = len(grid), len(grid[0])
        
for i in range(n - 2):
            for j in range(m - 2):
                if self.helper(i, j):
                    res += 1
        return res

    def helper(self, i, j):
        for ii in range(3):
            
for jj in range(3):
                if self.g[i + ii][j + jj] < 1 or self.g[i + ii][j + jj] > 9:
                    return False
        n = self.g[i][j] + self.g[i + 1][j + 1] + self.g[i + 2][j + 2]   
        if n != self.g[i][j + 2] + self.g[i + 1][j + 1] + self.g[i + 2][j]:
            return False
        for k in range(3):
            if n != self.g[i + k][j] + self.g[i + k][j + 1] + self.g[i + k][j + 2]:
                return False
            if n != self.g[i][j + k] + self.g[i + 1][j + k] + self.g[i + 2][j + k]:
                return False
        return True
