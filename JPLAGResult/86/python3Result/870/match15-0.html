class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        cnt = 0
        for i in range(0, len(grid) - 2):
            for j in range(0, len(grid[0]) - 2):
                if self.magic(self.subgrid(grid, i, j)):
                    cnt += 1
        return cnt
    
    def subgrid(self, a, i, j):
        return [a[i][j:j+3], a[i+1][j:j+3], a[i+2][j:j+3]]
        
    
    def column(self, a, i):
        return [a[0][i], a[1][i], a[2][i]]
    
    def magic(self, a):
        s = set()
        for i in range(0, 3):
            for j in range(0, 3):
                if a[i][j] > 9 or a[i][j] <= 0:
                    return False
                s.add(a[i][j])
        if len(s) != 9:
            return False
        want = sum(a[0])
        return sum(a[1]) == want and sum(a[2]) == want and sum(self.column(a, 0)) == want and sum(self.column(a, 1)) == want and sum(self.column(a, 2)) == want and a[0][0] + a[1][1] + a[2][2] == want and a[0][2] + a[1][1] + a[2][0] == want
