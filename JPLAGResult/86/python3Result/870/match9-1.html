class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        def isMagic(arr):
            if min(arr) != 1 or max(arr) != 9:
                return False
            if len(set(arr)) != 9:
                return False
            if sum(arr[0:3]) != 15 or sum(arr[3:6]) != 15 or sum(arr[6:9]) != 15:
                return False
            
if arr[0] + arr[3] + arr[6] != 15 or arr[1] + arr[4] + arr[7] != 15 or arr[2] + arr[5] + arr[8] != 15:
                return False
            if arr[0] + arr[4] + arr[8] != 15 or arr[2] + arr[4] + arr[6] != 15:
                return False
            return True
                
        
        m = len(grid)
        n = len(grid[0])
        
        if m < 3 or n < 3:
            return 0
        
        count = 0
        for i in range(m-2):
            for j in range(n-2):
                if isMagic([grid[k][l] for l in range(j, j+3) for k in range(i, i+3)]):
                    count += 1
                    
        return count
