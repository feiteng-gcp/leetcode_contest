class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m , n = len(grid),len(grid[0])
        res = 0
        if m < 3 or n < 3:
            return res
        def check(row,col):
            for add in range(3):
                if sum(grid[row+add][col:col+3]) != 15:
                    return False
                if sum([grid[row+i][col+add] for i in range(3)]) != 15:
                    return False
            if grid[row][col]+grid[row+1][col+1]+grid[row+2][col+2] != 15:
                return False
            if grid[row+2][col]+grid[row+1][col+1]+grid[row][col+2] != 15:
                return False
            nums = []
            for i in range(3):
                for j in range(3):
                    nums += grid[row+i][col+j],
            if set(nums) != set(range(1,10)):
                return False
            return True
        for i in range(m-2):
            for j in range(n-2):
                if check(i,j):
                    res += 1
        return res
