class Solution:
    def is_magic(self, grid, x, y):
        s = grid[x][y] + grid[x + 1][y] + grid[x + 2][y]
        for i in range(3):
            for j in range(3):
                if grid[x + i][y + j] > 9 or grid[x + i][y + j] < 1:
                    return False
            if s != (grid[x][y + i] + grid[x + 1]
[y + i] + grid[x + 2][y + i]) or s != (grid[x + i][y] + grid[x + i][y + 1] + grid[x + i][y + 2]):
                return False
        if s != (grid[x][y] + grid[x + 1][y + 1] + grid[x + 2][y + 2]) or s != (grid[x + 2][y] + grid[x + 1][y + 1] + grid[x][y + 2]):
            return False
        return True
    
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        if len(grid) < 3:
            return 0
        if len(grid[0]) < 3:
            return 0
        s = 0
        for i in range(len(grid) - 2):
            for j in range(len(grid[i]) - 2):
                if self.is_magic(grid, i, j):
                    s += 1
        return s
