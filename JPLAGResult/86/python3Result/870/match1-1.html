class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """

        m = len(grid)
        n = len(grid[0])
        cnt = 0
        for i in range(0, m - 2):
            for j in range(0, n - 2):
                if self.isMagic(i, j, grid): cnt += 1
        return cnt

    def isMagic(self, i, j, grid):
        cnt = [1]*9
        
for i1 in range(i, i+3):
            for j1 in range(j, j+3):
                if grid[i1][j1]<1 or grid[i1][j1]>9: return False
                cnt[grid[i1][j1]-1]-=1
        if any(cnt): return False
        for i1 in range(i, i+3):
            if sum(grid[i1][j:j+3]) != 15: return False
        for j1 in range(j, j+3):
            if grid[i][j1] + grid[i+1][j1] + grid[i+2][j1]!= 15: return False
        return grid
