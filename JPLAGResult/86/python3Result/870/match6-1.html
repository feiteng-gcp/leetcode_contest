class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        answer = 0
        for y in range(len(grid)-2):
            for x in range(len(grid[0])-2):
                if set(grid[y][x:x+3]+grid[y+1][x:x+3]+grid[y+2][x:x+3]) == set(range(1, 10)):
                    if sum(grid[y][x:x+3]) == sum(grid[y+1][x:x+3]) == sum(grid[y+2][x:x+3]) == sum([grid[y][x], grid[y+1][x+1], grid[y+2][x+2]]) == sum([
grid[y][x+2], grid[y+1][x+1], grid[y+2][x]]):
                        answer += 1
        return answer
