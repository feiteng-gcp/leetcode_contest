class Solution:
    def magicSquare(self, grid, i, j):
        for m in range(3):
            row_sum = 0
            for n in range(3):
                if not (1 <= grid[i + m][j + n] <= 9):
                    return False
                row_sum += grid[i + m][j + n]
            if row_sum != 15:
                
return False

        for m in range(3):
            col_sum = 0
            for n in range(3):
                col_sum += grid[i + n][j + m]
            if col_sum != 15:
                return False

        dia_sum = 0
        for m in range(3):
            dia_sum += grid[i + m][j + m]
        if dia_sum != 15:
            return False

        return True

    def numMagicSquaresInside(self, grid):
        
        m = len(grid)
        n = len(grid[0])
        res = 0
        
        return sum([1 for i in range(m - 2) for j in range(n - 2) if self.magicSquare(grid, i, j)])
