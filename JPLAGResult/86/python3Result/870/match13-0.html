class Solution:
    def numMagicSquaresInside(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        
        def check(grid, i, j):
            cache = set()
            if i + 2 < len(grid) and j + 2 < len(grid[0]):
                for ii in range(i, i + 3):
                    for jj in range(j, j + 3):
                        if 1 <= grid[ii][jj] <= 9 and grid[ii][jj] not in cache:
                            cache.add(grid[ii][jj])
                        else:
                            return False
                s1 = 0
                s2 = 0
                for d in range(3):
                    s1 += grid[i+d][j+d]
                    s2 += grid[i+d][j+2-d]
                if s1 != s2:
                    return False
                for ii in range(i, i + 3):
                    if s1 != sum(grid[ii][j:j+3]):
                        return False
                for jj in range(j, j + 3):
                    ss = 0
                    for ii in range(i, i + 3):
                        ss += grid[ii][jj]
                    if ss != s1:
                        return False
                return True
                    
            else:
                return False
     
        res = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if check(grid, i, j) is True:
                    res += 1
        return res
