class Solution:
    def reachingPoints(self, sx, sy, tx, ty):
        """
        :type sx: int
        :type sy: int
        :type tx: int
        :type ty: int
        :rtype: bool
        """
        if sx == tx:
            if (ty-sy)%sx==0:
                return True
            else:
                return False
        if ty == sy:
            if (tx-sx)%sy==0:
                return True
            else:
                return False
        if tx< sx or ty < sy:
            return False
        
        return self.reachingPoints(sx, sy, tx-ty, ty) or self.reachingPoints(sx, sy, tx, ty-tx)
