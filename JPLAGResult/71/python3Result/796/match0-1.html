class Solution:
    def gcd(self, a, b):
        while (b != 0):
            t = b
            b = a % b
            a = t
        return a
    
    
def reachingPoints(self, sx, sy, tx, ty):
        """
        :type sx: int
        :type sy: int
        :type tx: int
        :type ty: int
        :rtype: bool
        """
        if sx == tx and ty > sy:
            if (ty-sy) % sx == 0:
                return True
            else:
                return False
        
        if sy == ty and tx > sx:
            if (tx-sx) % sy == 0:
                return True
            else:
                return False 
        
        while tx >= sx and ty >= sy:
            if (tx == sx and ty == sy) or (sx == tx == 1) or (sy == ty == 1):
                return True
            if tx > ty:
                tx = tx - ty
            else:
                ty = ty - tx
            
        return False
