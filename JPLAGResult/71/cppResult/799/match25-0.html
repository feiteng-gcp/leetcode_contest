vector<int> a;


void doit(TreeNode* p)
{
  if (p==NULL) return;
  a.push_back(p->val);
  doit(p->left);
  doit(p->right);
}

class Solution {
public:
    int minDiffInBST(TreeNode* root) {
      int i,mi;
      a.clear();
      doit(root);
      sort(a.begin(),a.end());
      mi=2000000000;
      for (i=0;i+1<a.size();i++)
        if (a[i+1]-a[i]<mi) mi=a[i+1]-a[i];
      return mi;
    }
};
