class Solution {
public:
  int minDiffInBST(TreeNode* root) {
    vector<int> vals;
    putnodes(root, vals);
    sort(vals.begin(), vals.end());

    int minv = INT_MAX;
    for (int i = 1; i < vals.size(); ++i) {
      int tmp = vals[i] - vals[i - 1];
      if (tmp < minv) minv = tmp;
    }
    return minv;
  }
  void putnodes(TreeNode* node, vector<int>& vals) {
    if (!node) return;
    vals.push_back(node->val);
    putnodes(node->left, vals);
    putnodes(node->right, vals);
  }
};
