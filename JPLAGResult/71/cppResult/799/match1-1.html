/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    void do_in (TreeNode * r, vector <int> & in) {
        if (r == nullptr) return;
        do_in (r->left, in);
        in.push_back (r->val);
        do_in (r->right, in);
    }
    int minDiffInBST(TreeNode* root) {
        vector <int> s;
        do_in (root, s);

        int mi = 100000000;
        for (int i = 1 ; i < s.size () ; i++) {
            mi = min (abs(s[i-1] - s[i]), mi);
        }
        return mi;
    }
};
