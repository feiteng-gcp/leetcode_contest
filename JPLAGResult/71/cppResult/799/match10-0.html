/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
        vector<int> v;

public:
    void help(TreeNode* root){
        if(!root) return;
        v.push_back(root->val);
        help(root->left);
        help(root->right);
    }
    int minDiffInBST(TreeNode* root) {
        help(root);
        sort(v.begin(),v.end());
        int res=INT_MAX;
        for(int i=0;i<v.size()-1;i++){
            res=min(res,v[i+1]-v[i]);
        }
        return res;
    }
};
