/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vector<int> Q;
    void go(TreeNode* root) {
        if (root == 0) return;
        Q.push_back(root->val);
        go(root->left);
        go(root->right);
    }
    #define LL long long
public:
    int minDiffInBST(TreeNode* root) {
        go(root);
        sort(Q.begin(),Q.end());
        long long res = (LL)1e18;
        for (int i = 0; i < Q.size() - 1; i++) {
            res = min(res, abs(( long long)Q[i] - Q[i+1]));
        }
        return res;
    }
};
