/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    
    int v[105] = {0};
    int cnt = 0;
public:
    int dfs(TreeNode* tn){
        if (tn == NULL)
            return -1;
        dfs(tn->left);
        v[cnt++] = tn->val;
        dfs(tn->right);
        return 0;
    }
    
    int minDiffInBST(TreeNode* root) {
        dfs(root);
        int md = 100000000;
        for (int i = 1; i < cnt; ++i){
            int delta = v[i] - v[i-1];
            if (delta < md)
                md = delta;
        }
        return md;
    }
};
