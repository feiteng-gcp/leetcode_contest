/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
int num[105];
int cur;
void get(TreeNode* root)
{
	num[cur++]=root->val;
	if(root->left)get(root->left);
	if(root->right)get(root->right);
}
    int minDiffInBST(TreeNode* root) {
        cur=0;
	get(root);
	sort(num,num+cur);
	int ans=num[1]-num[0];
	for(int i=2;i<cur;i++)
	{
		ans=min(ans,num[i]-num[i-1]);
	}
	return ans;
    }
};
