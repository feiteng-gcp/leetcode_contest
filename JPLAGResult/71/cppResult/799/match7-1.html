/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;

    int minDiffInBST(TreeNode* root) {
        v.clear();
        dfs(root);
        int ans = INT_MAX;
        for (int i = 1; i < v.size(); i++) {
            ans = min(ans, v[i] - v[i-1]);
        }
        return ans;
    }
    void dfs(TreeNode* root) {
        if (root == nullptr) {
            return;
        }
        dfs(root->left);
        v.push_back(root->val);
        dfs(root->right);
    }
};
