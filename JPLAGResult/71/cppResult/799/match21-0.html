/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void traverse(TreeNode* cur, vector<int>& A) {
        if (cur == nullptr) return;
        A.push_back(cur->val);
        traverse(cur->left, A);
        traverse(cur->right, A);
    }
    int minDiffInBST(TreeNode* root) {
        vector<int> A;
        traverse(root, A);
        sort(A.begin(), A.end());
        int ret = 1 << 30;
        for (int i = 1; i < A.size(); ++i) {
            ret = min(ret, A[i] - A[i - 1]);
        }
        return ret;
    }
};
