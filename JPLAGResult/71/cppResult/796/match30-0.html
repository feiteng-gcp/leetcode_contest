class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if(tx==ty)return sx==sy&&tx==sx;
        if(tx<sx||ty<sy)return 0;
        if(tx<ty)
        {
            swap(tx,ty);
            swap(sx,sy);
        }
        if((tx-sx)%ty==0&&ty==sy)return 1;
        return reachingPoints(sx,sy,tx%ty,ty);
    }
};
