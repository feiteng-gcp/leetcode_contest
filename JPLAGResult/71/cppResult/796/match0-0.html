class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        // do it backwards
        while (tx >= sx && ty >= sy) {
            if (tx == sx && ty == sy)
                return true;
            if (tx > ty)
                tx -= max((tx - sx) / ty * ty, ty);
            else
                ty -= max((ty - sy) / tx * tx, tx);
        }
        return false;
    }
};
