class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while((tx >= sx) && (ty >= sy))
		{
			if(tx > ty)
			{
				if(sy == ty)
				{
					if(((tx - sx) % ty) == 0)
					{
						//cout << 1 << endl;
						return true;
					}
					else
					{
						return false;
					}
				}
				tx = tx - ty;
			}
			else
			{
				if(sx == tx)
				{
					if(((ty - sy) % tx) == 0)
					{
						//cout << 2 << endl;
						return true;
					}
					else
					{
						return false;
					}
				}
				ty = ty - tx;
			}
			
			if((sx == tx) && (sy == ty))
			{
				cout << 3 << endl;
				return true;
			}
		}
		
		return false;
    }
};
