class Solution {
public:
    

    bool dp(const int sx, const int sy, int tx, int ty) {
        if (tx<sx || ty<sy)
            return false;
        
        if (tx==sx && ty==sy)
            return true;
        
        if (tx==sx)
            return (ty-sy) % tx==0;
        if (ty==sy)
            return (tx-sx) % ty==0;
        
        return dp(sx, sy, tx, ty-tx) || dp(sx, sy, tx-ty, ty);
    }
    
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        return dp(sx, sy, tx, ty);
    }
};
