class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while (tx >= sx && ty >= sy) {
            if (sx == tx && sy == ty) return true;
            if (ty > tx) {
                int k = (ty - sy) / tx;
                if (k <= 0) break;
                ty = ty - k * tx;
            } else if (tx > ty) {
                int k = (tx - sx) / ty;
                if (k <= 0) break;
                tx = tx - k * ty;
            } else {
                return false;
            }
            
        }
        return (sx == tx && sy == ty);
        
    }
};
