class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if (tx < sx || ty < sy) return false;
        if (sx == tx) return ((ty - sy) >= 0 && (ty - sy) % sx == 0);
        if (sy == ty) return ((tx - sx) >= 0 && (tx - sx) % sy == 0);
        if (tx == ty) return reachingPoints(sx, sy, tx, 0) || reachingPoints(sx, sy, 0, ty);
        if (tx > ty) {
            int d = min((tx - ty) / ty, (tx - sx) / ty);
            return reachingPoints(sx, sy, tx - max(1, d) * ty, ty);
        }
        if (ty > tx) {
            int d = min((ty - tx) / tx, (ty - sy) / tx);
            return reachingPoints(sx, sy, tx, ty - max(1, d) * tx);
        }
        return false;
    }
};
