class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while(tx && ty){
            if(tx>=ty){
                if(ty==sy && tx>=sx && (tx-sx)%ty==0) return 1;
                tx%=ty;
            }else{
                if(tx==sx && ty>=sy && (ty-sy)%tx==0) return 1;
                ty%=tx;
            }
        }
        return 0;
    }
};
