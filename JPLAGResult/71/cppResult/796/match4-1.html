class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
      while ((tx>0)&&(ty>0))
      {
        if ((sx==tx)&&(sy==ty)) return true;
        if (tx>=ty)
        {
          if (sy==ty)
          {
            if ((tx>=sx)&&((tx-sx)%ty==0)) return true; 
          }
          tx%=ty;
        }
        else
        { 
          if (sx==tx)
          {
            if ((ty>=sy)&&((ty-sy)%tx==0)) return true;
          }
          ty%=tx;
        }
      }
      return false;
    }
};
