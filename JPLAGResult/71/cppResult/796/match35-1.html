class Solution {
public:
    
    bool reachingPoints(int sx, int sy, int tx, int ty) {
       if(tx==sx && ty==sy)
           return true;
        if(tx<sx || ty<sy)
            return false;
        if(tx>=ty){
            int k=(tx-sx)/ty;
            if(k<1)
                k=1;
            return reachingPoints(sx,sy,tx-k*ty,ty);
        }
            
        else {
              int k=(ty-sy)/tx;
            if(k<1)
                k=1;
              return reachingPoints(sx,sy,tx,ty-k*tx);
        }
          
           
    }
};
