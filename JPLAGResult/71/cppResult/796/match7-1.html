class Solution {
public:

bool reachingPoints(int sx, int sy, int tx, int ty) {
    while(1){
        if(sx==tx&&sy==ty)
            return true;
        if(sx>tx||sy>ty)
            return false;
        if(sx==tx){
            if(tx==0)
                return false;
            if((ty-sy)%tx==0)
                return true;
            return false;
        }
        if(sy==ty){
            if(ty==0)
                return false;
            if((tx-sx)%ty==0)
                return true;
            return false;
        }
        if(tx>ty){
            tx=(tx%ty);
        } else
            ty=(ty%tx);
    }
    return false;
}
};
