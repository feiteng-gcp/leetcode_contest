class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        int x = tx;
        int y = ty;
        while (x > 0 && y > 0){
            if (sx == x && sy == y) return true;
            if (x == y) return false;

            if (sx > x || sy > y) return false;
            if (x > y){
                if (x - sx < y)
                    x = x - y;
                else
                    x = x - y * ((x - sx) / y);
            }
            else{
                if (y - sy < x)
                    y = y - x;
                else
                    y = y - x * ((y - sy) / x);
            }
        }
        return false;
    }
};
