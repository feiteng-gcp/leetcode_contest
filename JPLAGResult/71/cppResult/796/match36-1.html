class Solution {
public:

    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if(tx < sx || ty < sy) return false;
        if(sx == tx && sy == ty) return true;
        if(tx > ty){
            swap(tx,ty);
            swap(sx,sy);
        }
        if(tx == sx) return (ty - sy) % sx  == 0;
        return reachingPoints(sx, sy, tx, ty - tx);
       
    }
   
};
