class Solution
{
public:
    bool reachingPoints(int sx, int sy, int tx, int ty)
    {
        while(!(tx<sx || ty<sy))
        {
        	//cout<<tx<<" "<<ty<<endl;
        	if(sx==tx && sy==ty)
        		return true;
        	if(tx==ty)
        		return (sx==tx && sy==ty);
        	else if(tx>ty)
        	{
        		if(sy==ty && tx-sx>=0 && (tx-sx)/ty*ty==(tx-sx))
        			return true;
        		tx-=tx/ty*ty;
        	}
        	else
        	{
        		if(sx==tx && ty-sy>=0 && (ty-sy)/tx*tx==(ty-sy))
        			return true;
        		ty-=ty/tx*tx;
			}
        }
        return false;
    }
};
