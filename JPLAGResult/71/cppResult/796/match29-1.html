class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while(tx >= sx && ty >= sy){
            if(ty - sy == 0){
                if((tx - sx) % sy == 0)
                    return true;
                else
                    return false;
            }
            if(tx - sx == 0){
                if((ty - sy) % sx == 0)
                    return true;
                else
                    return false;
            }
            if(ty > tx)
                ty = ty - tx;
            else
                tx = tx - ty;
            if(ty == sy && tx == sx)
                return true;
            //cout << ty << " " << tx << endl;
        }
        return (ty == sy && tx == sx) ? true : false;
        
    }
};
