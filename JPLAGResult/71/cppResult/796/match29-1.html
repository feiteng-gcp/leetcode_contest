class Solution {
public:
    bool search(int sx, int sy, int tx, int ty) {
        if (not ((tx >= sx) && (ty >= sy))) return false;
        if (sx == tx && sy == ty) return true;
        if (sx == tx && ((ty - sy) % tx) == 0) return true;
        if (sy == ty && ((tx - sx) % ty) == 0) return true;
        if (tx > ty) {
            return search(sx, sy, tx % ty, ty);
        } else {
            return search(sx, sy, tx, ty % tx);
        }
    }
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        return search(sx, sy, tx, ty);
    }
};
