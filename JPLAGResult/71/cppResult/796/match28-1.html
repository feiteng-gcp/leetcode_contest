class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if(sx == tx)
            return ((ty-sy) % tx == 0);
        if(sy == ty)
            return ((tx-sx) % ty == 0);
        while(tx > sx || ty > sy){
            if(tx < sx || ty < sy)
                return false;
            if(tx>ty)
                tx -= ty;
            else
                ty -= tx;
        }
        if(tx == sx && ty == sy)
            return true;
        return false;
    }
};
