class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if(tx > ty)
        {
            int n = tx / ty;
            tx -= n * ty;
            while(tx < sx) tx += ty;
        }
        else
        {
            int n = ty / tx;
            ty -= n * tx;
            while(ty < sy) ty += tx;
        }

        while(tx >= sx && ty >= sy)
        {
            if(tx == sx && ty == sy) return true;

            if(tx > ty)
            {
                tx = tx - ty;
            }
            else
            {
                ty = ty - tx;
            }
        }

        return false;
    }
};
