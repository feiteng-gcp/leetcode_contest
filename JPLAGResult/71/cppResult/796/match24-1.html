class Solution {
public:
  bool reachingPoints(int sx, int sy, int tx, int ty) {

    if (tx < sx || ty < sy) return false;
    if (tx == sx && ty == sy) return true;
    if (tx < ty) {
      int a = (ty - sy) / tx;
      a = max(1, a - 1);
      cout << "a " << a << endl;
      return reachingPoints(sx, sy, tx, ty - a * tx);
    } else {
      int a = (tx - sx) / ty;
      a = max(1, a - 1);
      cout << "a " << a << endl;
      return reachingPoints(sx, sy, tx - a * ty, ty);
    }
  }
};
