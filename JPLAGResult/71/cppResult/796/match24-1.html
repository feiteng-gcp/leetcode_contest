class Solution {
public:
    unordered_map<int, unordered_map<int, int>> memory;

    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if (tx < sx) return false;
        if (ty < sy) return false;
        if (ty == sy && tx == sx) return true;
        
        if (tx > ty) {
            if (tx - ty < sx) return false;
            return reachingPoints(sx, sy, (tx - sx) % ty + sx, ty);
        }
        else {
            if (ty - tx < sy) return false;
            return reachingPoints(sx, sy, tx, (ty - sy) % tx + sy);
        }
    }
};
