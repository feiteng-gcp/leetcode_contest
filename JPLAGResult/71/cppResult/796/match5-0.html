class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while(tx > sx || ty > sy) {
            if(tx == ty || tx == 0 || ty == 0) return false;
            else if(tx > ty) {
                if(ty == sy) {
                    if((tx - sx) % sy == 0) return true;       
                }
                tx = tx - (tx / ty) * ty;
            }
            else {
                if(tx == sx && (ty - sy) % sx == 0) return true;
                ty = ty - (ty / tx) * tx;
            }
        }
        return tx == sx && ty == sy;
    }
};
