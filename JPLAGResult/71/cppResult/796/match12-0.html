class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while(true){
            //cout << tx << " "<< ty <<endl;
            if((tx==sx)&&(ty == sy)) return true;
            //if(tx == ty) return false;
            if((tx<sx)||(ty<sy)) return false;
            if(tx > ty){
                if((tx-sx)/ty == 0) return false;
                tx-=((tx-sx)/ty)*ty;
                //if(tx==0) tx =ty;
            }else{
                if((ty-sy)/tx == 0) return false;
                ty-=((ty-sy)/tx)*tx;
                //if(ty==0) ty = tx;
            }
        }
    }
};
