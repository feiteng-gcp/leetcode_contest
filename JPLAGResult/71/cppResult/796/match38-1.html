class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if (sx == 0) return tx == sy && ty == sy;
        if (sy == 0) return tx == sx && ty == sx;
        while (tx >= sx && ty >= sy) {
            if (tx == sx && ty == sy) return true;
            if (ty >= tx) {
                ty = ty % tx;
            } else {
                tx = tx % ty;
            }
            if (ty < sy && tx == sx && (sy-ty) % tx == 0) return true;
            if (tx < sx && ty == sy && (sx-tx) % ty == 0) return true;
        }
        
        return false;
    }
};
