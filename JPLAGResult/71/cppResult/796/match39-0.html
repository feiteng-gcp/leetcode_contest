class Solution {
public:
    set<pair<int,int> >s;
bool isok(int sx,int sy,int tx,int ty,int f,int dep)
{
	if(sx==tx&&sy==ty)return true;
	if(sx>tx||sy>ty||dep>45)return false;
    if(sx==tx)
	{
		if((ty-sy)%tx)return false;
		else return true;
	}
	if(sy==ty)
	{
		if((tx-sx)%ty)return false;
		else return true;
	}
	pair<int,int> p;
	if(!f)
	{
		int maxm=(tx-sx)/ty;
		for(int i=maxm;i>=1;i--)
		{
			if(ty>sy&&ty-(tx-i*ty)<sy)break;
			p=make_pair(tx-i*ty,ty);
			if(s.find(p)==s.end()&&isok(sx,sy,tx-i*ty,ty,1-f,dep+1))return true;
		}
	}
	else
	{
		int maxm=(ty-sy)/tx;
		for(int i=maxm;i>=1;i--)
		{
			if(tx>sx&&tx-(ty-i*tx)<sx)break;
			p=make_pair(tx,ty-i*tx);
			if(s.find(p)==s.end()&&isok(sx,sy,tx,ty-i*tx,1-f,dep+1))return true;
		}
	}
	return false;
}
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        return isok(sx,sy,tx,ty,0,1)||isok(sx,sy,tx,ty,1,1);
    }
};
