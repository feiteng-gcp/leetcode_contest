class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        int k;
        while (tx >= sx && ty >= sy) {
            if (tx == sx && ty == sy) return true;
            if (ty > tx) {
                if (ty - sy >= tx) ty -= (ty - sy) / tx * tx;
                else ty -= tx;
            }
            else {
                if (tx - sx >= ty) tx -= (tx - sx) / ty * ty;
                else tx -= ty;
            }
        }
        
        return false;
    }
};
