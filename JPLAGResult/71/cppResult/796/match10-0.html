class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if(tx < sx || ty < sy) return false;
        if(tx == sx && ty == sy) return true;
        int d;
        if(sx == tx){
            if((ty-sy)%tx == 0)return true;
            else return false;
        }
        
        if(sy == ty) {
            if((tx-sx)%ty == 0)return true;
            else return false;
        }
        
        if(tx >= ty){
            d = tx - ty;
            return reachingPoints(sx, sy, d, ty);
        }else{
            d = ty - tx;
            return reachingPoints(sx, sy, tx, d);
        }
    }
};
