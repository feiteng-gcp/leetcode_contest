class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while(sx<=tx && sy<=ty){
            if(sx==tx && sy==ty) return true;
            if(tx<=ty) ty-=max((ty-sy)/tx,1)*tx;
            else tx-=max((tx-sx)/ty,1)*ty;
        }
        return false;
    }
};
