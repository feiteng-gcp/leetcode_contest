class Solution {
public:
  int numRabbits(vector<int>& answers) {
    vector<int> counts = getCounts(answers);
    int sum = 0;
    for (int ans: answers) {
      int count = counts.at(ans);
      if (count == -1) continue;
      cout << ans << " appears " << count << endl;
      sum += ceil((float)count / (ans + 1)) * (ans + 1);
      cout << "sum == " << sum << endl;
      counts[ans] = -1;
    }
    return sum;
  }
  vector<int> getCounts(vector<int>& answers) {
    vector<int> counts(1000);
    for (int ans: answers) {
      counts.at(ans)++;
    }
    return counts;
  }
};
