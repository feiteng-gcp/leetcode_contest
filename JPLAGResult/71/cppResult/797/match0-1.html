class Solution {
public:
    int numRabbits(vector<int>& answers) {
        map<int, int> m;
        for (int i = 0; i < answers.size(); i++){
            m[answers[i]]++;
        }
        int res = 0;
        if (m.find(0) != m.end())
            res += m[0];
        for (map<int, int>::iterator it = m.begin(); it != m.end(); it++){
            if (it->first != 0){
                res += ((it->second - 1)/(it->first + 1) + 1) * (it->first + 1);
            }
        }
        return res;
    }
};
