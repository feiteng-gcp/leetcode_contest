class Solution {
public:
    int numRabbits(vector<int>& answers) {
        map<int,int> cnt;
        int sum = 0;
        for (auto v : answers) {
            ++cnt[v];
        }
        for (auto v : cnt) {
            sum += (v.second+v.first)/(v.first+1)*(v.first+1);
        }
        return sum;
    }
};
