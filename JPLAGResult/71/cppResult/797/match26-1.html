class Solution {
public:
    int numRabbits(vector<int>& answers) {
        int res[1005];
	memset(res,0,sizeof(res));

	int n=answers.size();
	int ans=0;
	for(int i=0;i<n;i++)
	{
		if(!res[answers[i]])
		{
			ans+=answers[i]+1;
			res[answers[i]]=answers[i];
		}
		else res[answers[i]]--;
	}
	return ans;
    }
};
