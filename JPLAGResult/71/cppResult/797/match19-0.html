class Solution {
public:
    int numRabbits(vector<int>& answers) {
        unordered_map<int, int> counter;
        for (int x : answers) {
            ++counter[x + 1];
        }
        int rtn = 0;
        for (auto it : counter) {
            rtn += (1 + (it.second - 1) / it.first) * it.first;
        }
        return rtn;
    }
};
