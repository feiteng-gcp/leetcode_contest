class Solution {
public:
    int numRabbits(vector<int>& answers) {
        unordered_map<int, int>ump;
        int ans = 0;
        for(auto i:answers){
            ump[i] += 1;
        }
        for(auto i: ump){
            if(i.first == 0) ans += i.second;
            else{
                ans += (i.second / (i.first + 1)) * (i.first + 1);
                if(i.second % (i.first + 1))
                    ans += (i.first + 1);
            }
        }
        return ans;
    }
};
