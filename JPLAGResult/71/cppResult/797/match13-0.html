typedef long long LL;
typedef long double LD;
int DEBUG=0,MULTI=0,GCJ=0;

class Solution {
public:
    int numRabbits(vector<int>& answers) {
        map<int,int> res;
        for(int v:answers) res[v] += 1;
        LL ans = 0;
        for(auto el:res){
            LL val = el.first + 1;
            LL cnt = (el.second + val - 1) / val * val;
            ans += cnt;
        }
        return ans;
    }
};
