class Solution {
public:
    int numRabbits(vector<int>& answers) {
        unordered_map<int, int> counts;
        int c = 0;
        for( int i = 0; i < answers.size(); i++ ) {
            if( counts.find( answers[ i ] ) == counts.end() ) {
                counts[ answers[ i ] ] = answers[ i ];
                c += answers[ i ] + 1;
            } else {
                if( counts[ answers[ i ] ] == 0 ) {
                    counts[ answers[ i ] ] = answers[ i ];
                    c += answers[ i ] + 1;
                } else {
                    counts[ answers[ i ] ]--;
                }
            }
        }
        return c;
    }
};
