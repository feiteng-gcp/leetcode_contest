class Solution {
public:
    int numRabbits(vector<int>& a) {
        sort(a.begin(), a.end());
        int n = a.size();
        int ret = 0;
        for (int i = 0, j; i < n; i = j) {
            int cur = a[i] + 1;
            for (j = i + 1; j < n && a[j] == a[i] && j - i + 1 <= cur; ++j);
            ret += cur;
        }
        return ret;
    }
};
