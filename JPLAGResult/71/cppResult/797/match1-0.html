class Solution {
public:
    int numRabbits(vector<int>& answers) {
        int num = 0;
        map<int, int> dict;
        for (int answer : answers) {
            if (!dict.count(answer)) {
                dict[answer] = 0;
            }
            dict[answer]++;
        }
        for (map<int, int>::iterator it = dict.begin(); it != dict.end(); it++) {
            num += ceil((float)it->second / (it->first + 1)) * (it->first + 1);
        }
        return num;
    }
};
