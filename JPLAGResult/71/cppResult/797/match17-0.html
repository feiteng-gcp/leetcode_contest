class Solution {
public:
    int numRabbits(vector<int>& answers) {
        map <int, int> cmp;
        int len = answers.size();
        for (int i = 0; i < len; ++i){
            int a = answers[i];
            if (cmp.find(a) != cmp.end()){
                cmp[a] = cmp[a] + 1;
            }
            else{
                cmp[a] = 1;
            }
        }
        int num = 0;
        map<int,int>::iterator it = cmp.begin();
        while(it != cmp.end()){
            int k = it->first;
            int v = it->second;
            it++;
            num += ceil(float(v) / float((k+1))) * (k+1);
        }
        return num;
    }
};
