class Solution {
public:
    int numRabbits(vector<int>& answers) {
        unordered_map<int, int> mp;
        int cnt = 0;
        for(int i : answers){
            if(mp[i] == i+1){
                cnt += mp[i];
                mp[i] = 1;
            }else{
                ++mp[i];
            }
        }
        
        for(auto &p : mp){
            cnt += p.first + 1;
        }
        
        return cnt;
    }
};
