class Solution {
public:
    int numRabbits(vector<int>& answers) {
        unordered_map<int, int> mp;
        int res=0;
        for(int i=0;i<answers.size();++i){
            if(mp.find(answers[i]+1)==mp.end()){
                res+=answers[i]+1;
                mp[answers[i]+1]=answers[i];
            }else if(mp[answers[i]+1]==0){
                res+=answers[i]+1;
                mp[answers[i]+1]=answers[i];
            }else{
                mp[answers[i]+1]--;
            }
        }
        return res;
    }
};
