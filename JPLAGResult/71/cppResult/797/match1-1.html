class Solution {
public:
	int numRabbits(vector<int>& answers) {
		int total = 0;
		unordered_map<int, int> m;
		for (int a : answers) {
			if (a == 0) {
				total++;
			}
			else {
				m[a]++;
			}
		}
		for(pair<int, int> p : m) {
			int r = p.second;
			total += int(ceil(r * 1.0 / (p.first + 1))) * (p.first + 1);
		}
		return total;
	}
};
