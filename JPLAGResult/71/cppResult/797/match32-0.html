class Solution {
public:
    int numRabbits(vector<int>& answers) {
        if(answers.size() == 0){
            return 0;
        }
        
        sort(answers.begin(), answers.end());
        unordered_map<int, int> mp;
        for(auto n : answers){
            mp[n]++;
        }
        
        int ret = 0;
        
        
        
        for(auto m : mp){
            if(m.first == 0){
                ret += m.second;
            } else {
                if(m.second <= m.first + 1){
                    ret += 1;
                    ret += m.first;
                } else {
                    int addon = 1 + m.first;
                    int times = m.second / addon;
                    ret += times * addon;
                    if(m.second % addon != 0){
                        ret += addon;
                    }
                    
                }
            }
            
            
        }
        
        return ret;
        
    }
};
