class Solution {
public:
    int numRabbits(vector<int>& answers) {
        
        unordered_map<int,int> mp;
        for (int i = 0; i < answers.size(); ++i) {
            ++mp[answers[i]];
        }
        
        int res = 0;
        for (auto& m : mp) {
            int num_group = (m.second-1) / (m.first+1) + 1;
            res += num_group * (m.first + 1);
        }
        
        return res;
    }
};
