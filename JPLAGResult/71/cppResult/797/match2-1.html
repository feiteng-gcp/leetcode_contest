class Solution {
public:
    int numRabbits(vector<int>& answers) {
        if(answers.size() == 0)
            return 0;
        sort(answers.begin(), answers.end());
        int res = 0;
        
        int cur = -1;

        int num = 0;
        for(int i = 0; i < answers.size(); ++i){
            if(cur != answers[i]){
                cur = answers[i];
                num = 0;
            }
            if(num == 0)
                res = res + cur + 1;
            ++num;
            if(num == cur + 1)
                num = 0;              
        }
        
        return res;
    }
};
