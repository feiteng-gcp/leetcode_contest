class Solution {
public:
    int numRabbits(vector<int>& answers) {
        map<int, int> m;
        
        for (int k : answers)
            m[k]++;
        
        int res = 0;
        for (auto it : m)
        {
            int cnt = it.first + 1;
            int group = it.second/cnt;
            if (it.second % cnt > 0) group++;
            
            res += cnt*group;
            
            // printf("cnt %d group %d \n", cnt, group);
        }
        
        return res;
    }
};
