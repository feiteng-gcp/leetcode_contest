class Solution {
public:
    int numRabbits(vector<int>& answers) {
        unordered_map<int, int> hash;
        int len = answers.size();
        for(int i = 0; i < len; ++i) {
            if(hash.find(answers[i]) == hash.end()) hash[answers[i]] = 1;
            else hash[answers[i]] += 1;
        }
        int ans = 0;
        for(auto iter = hash.begin(); iter != hash.end(); iter++) {
            if(iter->first == 0)  {
                ans += iter->second;
                continue;
            }
            if(iter->first + 1 >= iter ->second) ans += iter->first + 1;
            else {
                ans += ((iter->second / (iter->first + 1) + (iter->second % (iter->first + 1) == 0 ? 0 : 1)) * (iter->first + 1));
            }
        }
        return ans;
    }
};
