class Solution {
public:
    int numRabbits(vector<int>& answers) {
        if(answers.size()==0){
            return 0;
        }
        sort(answers.begin(),answers.end());
        int ans=answers[0]+1;
        int last=answers[0];
        int count=1;
        for(int i=1;i<answers.size();i++){
            
            if(answers[i]==0){
                count=1;
                ans++;
            }
            else if(answers[i]!=last){
                ans+=answers[i]+1;
                count=1;
            }
            else if(answers[i]==last && count>=answers[i]+1){
                ans+=answers[i]+1;
                count=1;
            }
            else if(answers[i]==last){
                count++;
            }
            last=answers[i];
            //cout<<answers[i]<<" "<<ans<<endl;
            
        }
        return ans;
    }
};
