class Solution {
public:
    int numRabbits(vector<int>& anss) {
        if (anss.size() == 0) return 0;
        map<int, int> Map;

        for (int a : anss) {
            Map[a + 1]++;
        }
        int ans = 0;
        for (auto m : Map) {
            if (m.second <= m.first) {
                ans += m.first;
            } else {
                ans += (m.second + m.first - 1) / m.first * m.first;
            }
        }
        return ans;
    }
};
