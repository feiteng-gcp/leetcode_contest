class Solution {
public:
    int movesToChessboard(vector<vector<int>>& b) {
        int n = b.size();
        vector<vector<int>> d = b;
        int resb = 0, resd = 0;
        
        for(int i = 0; i < n; ++i) for(int j = 0; j < n; ++j) d[i][j] = b[j][i];
        
        for(int i = 0; i < n; ++i) {
            int nb = 0, nd = 0, cb = 0, cd = 0;
            for(int j = 0; j < n; ++j){
                nb += b[i][j];
                nd += d[i][j];
            }
            vector<int> x, y;
            for(int j = 0; j < n; ++j) {
                if(j % 2 == 0) {
                    x.push_back(0);
                    y.push_back(1);
                }
                else {
                    x.push_back(1);
                    y.push_back(0);
                }
            }
            if(n % 2 == 0) {
                if(nb != n / 2 || nd != n / 2) return -1;
                int xb = 0, xd = 0, yb = 0, yd = 0;
                for(int j = 0; j < n; ++j) {
                    if(b[i][j] == 1 && b[i][j] == x[j]) xb++;
                    if(b[i][j] == 1 && b[i][j] == y[j]) yb++;
                    if(d[i][j] == 1 && d[i][j] == x[j]) xd++;
                    if(d[i][j] == 1 && d[i][j] == y[j]) yd++;                    
                }
                cb = n / 2 - max(xb, yb);
                cd = n / 2 - max(xd, yd);
                resb = max(resb, cb);
                resd = max(resd, cd);
            }
            else {
                if(nb < n / 2 || nb > n / 2 + 1 || nd < n /2 || nd > n / 2 + 1) return -1;
                int xb = 0, xd = 0, yb = 0, yd = 0;
                for(int j = 0; j < n; ++j) {
                    if(b[i][j] == 1 && b[i][j] == x[j]) xb++;
                    if(b[i][j] == 1 && b[i][j] == y[j]) yb++;
                    if(d[i][j] == 1 && d[i][j] == x[j]) xd++;
                    if(d[i][j] == 1 && d[i][j] == y[j]) yd++;                    
                }
                if (nb == n / 2) {
                    cb = n / 2 - xb;
                }
                else {
                    cb = n / 2 + 1 - yb;
                } 
                
                if (nd == n / 2) {
                    cd = n / 2 - xd;
                }
                else {
                    cd = n / 2 + 1 - yd;
                }                
                
                resb = max(resb, cb);
                resd = max(resd, cd);
            }
        }
        return resb + resd;
    }
};
