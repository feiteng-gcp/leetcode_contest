class Solution {
public:
    int movesToChessboard(vector<vector<int>>& board) {
        int n = board.size();
        int p0;
        int p1;
        int m;
        int c0 = 0;
        int c1 = 0;
        if (n % 2 == 0) {
            m = n/2;
            p0 = -1;
            p1 = -1;
            c0 = 0;
            c1 = 0;
            for (int j=0; j<n; j++) {
                int sum = 0;
                int stamp = 0;
                for (int i=0; i<n; i++) {
                    sum += board[i][j];
                    stamp += (board[i][j] << i);
                }
                if (sum != m) {
                    return -1;
                }
                if (board[0][j] == 0) {
                    if (p0 != -1 && p0 != stamp) {
                        return -1;
                    }
                    p0 = stamp;
                    c0++;
                } else {
                    if (p1 != -1 && p1 != stamp) {
                        return -1;
                    }
                    p1 = stamp;
                    c1++;
                }                
            }
            if (c0 != c1) { return -1;}
            p0 = -1;
            p1 = -1;
            c0 = 0;
            c1 = 0;
            for (int i=0; i<n; i++) {
                int sum = 0;
                int stamp = 0;
                for (int j=0; j<n; j++) {
                    sum += board[i][j];
                    stamp += (board[i][j] << j);
                }
                if (sum != m) {
                    return -1;
                }
                if (board[i][0] == 0) {
                    if (p0 != -1 && p0 != stamp) {
                        return -1;
                    }
                    p0 = stamp;
                    c0 ++;
                } else {
                    if (p1 != -1 && p1 != stamp) {
                        return -1;
                    }
                    p1 = stamp;
                    c1 ++;
                }
            }
            if (c0 != c1) { return -1;}   
            int result = 0;
            int sum = 0;
            for (int j=0; j<n; j+=2) {
                sum += board[0][j];
            }
            result += min(sum, m-sum);
            sum = 0;
            for (int i=0; i<n; i+=2) {
                sum += board[i][0];
            }
            result += min(sum, m-sum);
            return result;
        }
        if (n % 2 != 0) {
            m = n/2;
            p0 = -1;
            p1 = -1;
            c0 = 0;
            c1 = 0;
            for (int j=0; j<n; j++) {
                int sum = 0;
                int stamp = 0;
                for (int i=0; i<n; i++) {
                    sum += board[i][j];
                    stamp += (board[i][j] << i);
                }
                if (sum != m && sum != m+1) {
                    return -1;
                }
                if (board[0][j] == 0) {
                    if (p0 != -1 && p0 != stamp) {
                        return -1;
                    }
                    p0 = stamp;
                    c0++;
                } else {
                    if (p1 != -1 && p1 != stamp) {
                        return -1;
                    }
                    p1 = stamp;
                    c1++;
                }            
            }
            int result = 0;

            int sum = 0;
            for (int j=0; j<n; j+=2) {
                sum += board[0][j];
            }
            if (c0 == c1 - 1) {
                result += m+1 - sum;
            } else if (c0 == c1 + 1) {
                result += sum;
            } else {
                return -1;
            }
            p0 = -1;
            p1 = -1;
            c0 = 0;
            c1 = 0;
            for (int i=0; i<n; i++) {
                int sum = 0;
                int stamp = 0;
                for (int j=0; j<n; j++) {
                    sum += board[i][j];
                    stamp += (board[i][j] << j);
                }
                if (sum != m && sum != m+1) {
                    return -1;
                }
                if (board[i][0] == 0) {
                    if (p0 != -1 && p0 != stamp) {
                        return -1;
                    }
                    p0 = stamp;
                    c0 ++;
                } else {
                    if (p1 != -1 && p1 != stamp) {
                        return -1;
                    }
                    p1 = stamp;
                    c1 ++;
                } 
                
            }
            sum = 0;

            for (int i=0; i<n; i+=2) {
                sum += board[i][0];
            }
            if (c0 == c1 - 1) {
                result += m+1 - sum;
            } else if (c0 == c1 + 1) {
                result += sum;
            } else {
                return -1;
            }
            return result;
        }
    }
};
