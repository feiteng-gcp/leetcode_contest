class Solution {
public:
  int movesToChessboard(vector<vector<int>>& board) {
    if (!legalrows(board)) return -1;
    if (!legalcols(board)) return -1;

    int n = board.size();
    int rowswap = minSwapVec(board[0]);
    if (rowswap == -1) return -1;

    vector<int> col0(n);
    for (int i = 0; i < n; ++i) {
      col0[i] = board[i][0];
    }
    int colswap = minSwapVec(col0);
    if (colswap == -1) return -1;
    return rowswap + colswap;
  }

  bool legalrows(vector<vector<int>>& board) {
    int n = board.size();
    for (int i = 0; i < n; ++i) {

      for (int j = i + 1; j < n; ++j) {
        int diff = abs(board[j][0] - board[i][0]);
        for (int k = 0; k < n; ++k) {
          if (abs(board[j][k] - board[i][k]) != diff) {
            return false;
          }
        }
      }
    }
    return true;
  }

  bool legalcols(vector<vector<int>>& board) {
    int n = board.size();
    for (int i = 0; i < n; ++i) {
      for (int j = i + 1; j < n; ++j) {
        int diff = abs(board[0][j] - board[0][i]);
        for (int k = 0; k < n; ++k) {
          if (abs(board[k][j] - board[k][i]) != diff) {
            return false;
          }
        }
      }
    }
    return true;
  }

  int minSwapVec(vector<int> vec) {
    int count1 = 0, count0 = 0;
    for (int v : vec) v? count1++: count0++;
    if (abs(count1 - count0) > 1) return -1;
    int count1_even = 0, count0_even = 0;
    for (int i = 0; i < vec.size(); i += 2) {
      vec[i]? count1_even++: count0_even++;
    }
    if (count1 > count0) {
      return count0_even;
    } else if (count0 > count1) {
      return count1_even;
    } else {
      return min(count0_even, count1_even);
    }
  }

};
