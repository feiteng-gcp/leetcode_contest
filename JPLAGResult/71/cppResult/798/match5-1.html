class Solution {
public:
    int solve(vector<int>& u, int n) {
        int cnt = 0;
        for (int i = 0; i < n; ++i) cnt += u[i];
        if (n % 2 == 1) {
            if (cnt != n / 2 && cnt != n / 2 + 1) return -1;
            if (cnt == n / 2) {
                int sum = 0;
                for (int i = 0; i < n; ++i) {
                    if (i % 2 != u[i]) ++sum;
                }
                return sum / 2;
            } else {
                int sum = 0;
                for (int i = 0; i < n; ++i) {
                    if (i % 2 == u[i]) ++sum;
                }
                return sum / 2;
            }
        } else {

            if (cnt != n / 2) return -1;
            int sum = 0;
            for (int i = 0; i < n; ++i) {
                if (i % 2 != u[i]) ++sum;
            }
            int cur = sum / 2;
            sum = 0;
            for (int i = 0; i < n; ++i) {
                if (i % 2 == u[i]) ++sum;
            }
            cur = min(cur, sum / 2);
            return cur;
        }
        return -1;
    }
    int movesToChessboard(vector<vector<int>>& a) {
        int n = a.size();
        set<vector<int>> row, col;
        for (int i = 0; i < n; ++i) {
            row.insert(a[i]);
        }
        for (int j = 0; j < n; ++j) {
            vector<int> b;
            for (int i = 0; i < n; ++i) {
                b.push_back(a[i][j]);
            }
            col.insert(b);
        }
        if (row.size() != 2 || col.size() != 2) return -1;
        int ret = 0;
        auto u = *row.begin();
        int cur = solve(u, n);
        if (cur < 0) return -1;
        cout << "cur=" << cur << endl;
        ret += cur;
        auto v = *col.begin();
        cur = solve(v, n);
        if (cur < 0) return -1;
        cout << "cur=" << cur << endl;
        ret += cur;
        
        return ret;
        
        
    }
};
