class Solution {
private:
    int moveRow(vector<vector<int> > &board) {
        vector<int> data;
        int first = -1, second = -1;
        for (vector<int> row : board) {
            int curr = 0;
            for (int x : row) {
                curr = curr * 2 + x;
            }
            if (first == -1) {
                first = curr;
                data.push_back(0);
            } else if (curr != first) {
                if (second == -1) {
                    second = curr;
                    data.push_back(1);
                } else if (second != curr) {
                    return -1;
                } else {
                    data.push_back(1);
                }
            } else {
                data.push_back(0);
            }
        }
        int n = board.size();
        int n01 = 0, n10 = 0;
        for (int i = 0; i < n; ++i) {
            if (data[i] == 0 && (i % 2 == 1)) ++n01;
            if (data[i] == 1 && (i % 2 == 0)) ++n10;
        }
        int tmp1 = (n01 == n10) ? n01 : -1;
        n01 = 0, n10 = 0;
        for (int i = 0; i < n; ++i) {
            if (data[i] == 0 && (i % 2 == 0)) ++n01;
            if (data[i] == 1 && (i % 2 == 1)) ++n10;
        }
        int tmp2 = (n01 == n10) ? n01 : -1;
        if (tmp1 == -1) return tmp2;
        if (tmp2 == -1) return tmp1;
        return min(tmp1, tmp2);
    }
    
    void transpose(vector<vector<int> > &board) {
        int n = board.size();
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                swap(board[i][j], board[j][i]);
            }
        }
    }

public:
    int movesToChessboard(vector<vector<int>>& board) {
        int move_row = moveRow(board);
        transpose(board);
        int move_col = moveRow(board);
        if (move_row == -1) return -1;
        if (move_col == -1) return -1;
        return move_row + move_col;
    }
};
