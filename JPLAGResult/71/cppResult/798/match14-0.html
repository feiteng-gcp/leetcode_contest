class Solution {
public:
    int movesToChessboard(vector<vector<int>>& board) {
        int n=board.size();
	int ans=0;
	int kind1=0,kind2=0;
	for(int i=0;i<n;i++)
	{
		int cur=0;
		for(int j=0;j<n;j++)
		{
			cur=cur+(board[0][j]+board[i][j])%2;
		}
		if(cur==0)kind1++;
		else if(cur==n)kind2++;
		else return -1;
	}
	if(kind1-kind2>1||kind2-kind1>1)return -1;
	int now=0;
	for(int i=2;i<n;i+=2)
	{
		if(board[i][0]!=board[0][0])now++;
	}
	if(kind1==kind2)
	{
		if(now>(n/2)/2)now=n/2-now;
	}
	else if(kind1<kind2)now=(n+1)/2-now;
	ans=ans+now;
	kind1=0;
	kind2=0;
	for(int i=0;i<n;i++)
	{
		int cur=0;
		for(int j=0;j<n;j++)
		{
			cur=cur+(board[j][0]+board[j][i])%2;
		}
		if(cur==0)kind1++;
		else if(cur==n)kind2++;
		else return -1;
	}
	if(kind1-kind2>1||kind2-kind1>1)return -1;
	now=0;
	for(int i=2;i<n;i+=2)
	{
		if(board[0][i]!=board[0][0])now++;
	}
	if(kind1==kind2)
	{
		if(now>(n/2)/2)now=n/2-now;
	}
	else if(kind1<kind2)now=(n+1)/2-now;
	ans=ans+now;
	return ans;
    }
};
