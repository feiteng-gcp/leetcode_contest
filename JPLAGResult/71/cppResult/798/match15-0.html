template<typename T1, typename T2> bool Max(T1& a, T2 b) { return a<b ? a = b, 1 : 0; }
template<typename T1, typename T2> bool Min(T1& a, T2 b) { return a>b ? a = b, 1 : 0; }
class Solution {
public:
    int movesToChessboard(vector<vector<int>>& a) {
        int n = a.size();
        int cnt = 0;
        int n1 = 0;
        rep(j,n){
            cnt += (j&1) && a[0][j];
            n1 += a[0][j];
        }
        if(abs(n1 * 2 - n) > 1) return -1;
        int ans1 = cnt;
        if(n1*2 < n) ans1 = n1 - cnt;
        else if(n1 * 2 == n) ans1 = min(cnt, n1 - cnt);
        
        rep(i,n){
            int dif = 0;
            rep(j,n) dif += a[0][j] != a[i][j];
            if(dif != 0 && dif != n) return -1;
        }
        
        cnt = 0;
        n1 = 0;
        rep(i,n){
            cnt += (i&1) && a[i][0];
            n1 += a[i][0];
        }
        if(abs(n1 * 2 - n) > 1) return -1;
        int ans2 = cnt;
        if(n1*2 < n) ans2 = n1 - cnt;
        else if(n1 * 2 == n) ans2 = min(cnt, n1 - cnt);
        
        rep(j,n){
            int dif = 0;
            rep(i,n) dif += a[i][0] != a[i][j];
            if(dif != 0 && dif != n) return -1;
        }
        return ans1 + ans2;
    }
};
