class Solution {
public:
    
int movesToChessboard(vector<vector<int>>& b) {
   int N=b.size();
   for(int i=0;i<N;i++){
     for(int j=0;j<N;j++){
       if((b[i][0]^b[i][j]^b[0][0]^b[0][j])) return -1;
     }
   }

   int cnt1=0,cnt2=0;
   for(int i=0;i<N;i++){
     if(b[i][0])cnt1++;
     if(b[0][i])cnt2++;
   }
   if(cnt1!=N/2 && cnt1!=(N+1)/2) return -1;
   if(cnt2!=N/2 && cnt2!=(N+1)/2) return -1;
	
   int res=0;
   int w[2]{};
   for(int i=0;i<N;i++){
     w[b[i][0]^(i%2==0)]++;
   }
   if(N%2==0){
     res+=min(w[0],w[1])/2;
   }else{
     if(cnt1==N/2) res+=w[0]/2;
     else res+=w[1]/2;
   }
   w[0]=0,w[1]=0;
   for(int i=0;i<N;i++){
     w[b[0][i]^(i%2==0)]++;
   }
   if(N%2==0){
     res+=min(w[0],w[1])/2;
   }else{
     if(cnt2==N/2) res+=w[0]/2;
     else res+=w[1]/2;
   }
   return res;
}

};
