class Solution {
public:
    int n;
    int solve(int x) {
        int res = 10000000;
        int t = 0;
        int one = 0;
        for (int i = 0, j = 0; i < n; i++, j ^= 1) {
            // cout << "j : "<<j << " x : " << (x >> i & 1) << endl;

            int v = (x >> i) & 1;
            if (v) one++;
            if (j) one--;
            if (v != j) t++;
        }
        if (one == 0) {
            res = min(res, t / 2);
        }
        one = 0;
        t = 0;
        for (int i = 0, j = 1; i < n; i++, j ^= 1) {
            int v = (x >> i) & 1;
            if (v) one++;
            if (j) one--;
            if (v != j) t++;
        }
        if (one == 0) {
            res = min(res, t / 2);
        }
        
        return res;
    }
    int movesToChessboard(vector<vector<int>>& b) {
        n = b.size();
        map<int, vector<int>> col;
        int row1 = -1, row2 = -1;
        for (int j = 0; j < n; j++) {
            int t = 0;
            for (int i = 0; i < n; i++) {
                if (b[i][j]) {
                    t |= (1 << i);                    
                }
            }
            if (row1 == -1) row1 = t;
            else if (t != row1) row2 = t;
            col[t].push_back(j);
        }
        if (col.size() != 2) return -1;

        
        if ((row1 & row2) !=0 || (row1 | row2) != ((1 << n) - 1)) return -1;
        // cout << row1 << " " << row2 << endl;


        int col1 = -1, col2 = -1;
        for (auto p : col) {
            int t = 0;
            for (auto v : p.second) {
                t |= (1 << v);
            }
            if (col1 == -1) col1 = t;
            else col2 = t;
        }
        if ((col1 & col2) !=0 || (col1 | col2) != ((1 << n) - 1)) return -1;
        // cout << col1 << " " << col2 << endl;
        int res = solve(col1) + solve(row1);
        if (res >= 10000000) res = -1;
        return res;
        
    }
};
