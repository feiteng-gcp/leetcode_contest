class Solution {
    public int numRabbits(int[] answers) {
        int n = answers.length;
        if (n == 0) return 0;
        if (n == 1) return answers[0] + 1;
        
        Map<Integer, Integer> map = new HashMap<>();
        for (int ans : answers) {
            map.putIfAbsent(ans, 0);
            map.put(ans, map.get(ans) + 1);
        
}
        
        int res = 0;
        for (int k : map.keySet()) {
            int count = map.get(k);
            int r = count % (k + 1);
            int d = count / (k + 1);
            res += (k + 1) * (d + (r != 0 ? 1 : 0));
        }
        return res;
    }
}
