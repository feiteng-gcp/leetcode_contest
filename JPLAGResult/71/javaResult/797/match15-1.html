class Solution {
    public int numRabbits(int[] answers) {
        int res = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for (int i : answers) {
            if (!map.containsKey(i)) {
                map.put(i, 1);
            }
            else {
                map.put(i, map.get(i) + 1);
            }
        }
        for (int key : map.keySet()) {
            int count = map.get(key);
            int group = count / (key + 1);
            if (count % (key + 1) != 0) {
                group += 1;
            }
            res += group * (key + 1);
        }
        return res;
    }
}
