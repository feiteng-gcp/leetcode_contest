class Solution {
    public int numRabbits(int[] answers) {
        int[] cnt = new int[1005];
        int ret = 0;
        for(int i = 0; i < answers.length; i++) {
            if(cnt[answers[i] + 1] == 0) 
{
                cnt[answers[i] + 1] = answers[i] + 1;
                ret += answers[i] + 1;
            }
            cnt[answers[i] + 1]--;
        }
        return ret;
    }
}
