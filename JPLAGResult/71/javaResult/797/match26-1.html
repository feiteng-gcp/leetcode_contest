class Solution {
    public int numRabbits(int[] answers) {
        int count = 0;
        if(answers == null || answers.length == 0) {
            return count;
        }
        
        Map<Integer, Integer> map = new HashMap<>();
        for(int i =0; i < answers.length; i++) {
            if(answers[i] == 0) {
                count++;
            } 
            else {
                int cur = answers[i];
                if(map.get(cur) == null) {
                    count++;
                    count+=answers[i];
                    map.put(cur, 1);
                }
                else 
{
                    int cc = map.get(cur);
                    if(cc == cur) {
                        map.remove(cur);
                    }
                    else {
                        map.put(cur, cc+1);
                    }
                }
            }
        }
        
        return count;
    }
}
