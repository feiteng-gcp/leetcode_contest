class Solution {
    public int numRabbits(int[] answers) {
        // sort then analyze
        int sum = 0;
        Arrays.sort(answers);
        int current = answers[0];
        int limit = current + 1;
        sum += limit;
        for (int i = 0; i < answers.length; i++) {
            if (answers[i] == current) {
                limit--;
                if (limit == 0 && i + 1 < answers.length) {
                    current = answers[i + 1];
                    
limit = current + 1;
                    sum += limit;
                }
            } else {
                current = answers[i];
                limit = current + 1;
                sum += limit;
                limit--;
            }
        }
        return sum;
    }
}
