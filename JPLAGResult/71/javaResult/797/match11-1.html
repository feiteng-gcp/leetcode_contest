class Solution {
    public int numRabbits(int[] answers) {
        Map<Integer, Integer> hashMap = new HashMap<Integer, Integer>();
        int result = 0;
        
for(int ans: answers) {
            if(ans == 0) {
                result += 1;
                continue;
            }
            if(hashMap.containsKey(ans)) {
                int value = hashMap.get(ans);
                if(value == ans + 1) {
                    result += (ans + 1);
                    hashMap.put(ans, 1);
                } else {
                    hashMap.put(ans, value+1);
                
}
            } else {
                hashMap.put(ans, 1);
            }
        }
        
        for(Map.Entry<Integer, Integer> entry: hashMap.entrySet()) {
            result += (entry.getKey() + 1);
        }
        
        return result;
    }
}
