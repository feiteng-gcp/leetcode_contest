class Solution {
    public int numRabbits(int[] answers) {
        int sum = 0;
        Map<Integer,Integer> map = new HashMap<>();
        
for(int i : answers){
            sum++;
            if(!map.containsKey(i)){
                map.put(i,i);
            }else{
                map.put(i,Math.max(0,map.get(i)-1));
                if(map.get(i)==0)map.remove(i);
            }
        }
        for(int i : map.values()){
            sum += i;
        }
        return sum;
    }
}
