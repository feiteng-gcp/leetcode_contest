class Solution {
    public int numRabbits(int[] a) {
        HashMap<Integer, ArrayList<Integer>> map = new HashMap<>();
        for(int i = 0; i < a.length; i++){
            if(map.containsKey(a[i] + 1)){
                int idx = 0;
                ArrayList<Integer> list = map.get(a[i] + 1);
                while(idx < list.size() && list.get(idx) <= 0){
                    idx++;
                }
                if(idx >= list.size()){
                    ArrayList<Integer> temp = new ArrayList<Integer>();
                    temp.add(a[i]);
                    map.put(a[i] + 1, temp);
                }else{
                    list.set(idx, list.get(idx) - 1);
                }
            }else{
                ArrayList<Integer> temp = new ArrayList<Integer>();
                temp.add(a[i]);
                map.put(a[i] + 1, temp);
            }
        }
        int res = 0;
        for(int key : map.keySet()){
            for(int val : map.get(key)){
                res += val;
            }
        }
        return res + a.length;
    }
}
