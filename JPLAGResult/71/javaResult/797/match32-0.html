class Solution {
    public int numRabbits(int[] answers) {
        if(answers == null || answers.length == 0) return 0;
        int len = answers.length;
        Arrays.sort(answers);
        int res = 0;
        int count = 0;
        int cur = answers[0];
        for(int i = 0; i < len; i++){
            if(answers[i] == 0) {
                res += 1;
            }else if(count == 0){
                cur = answers[i];
                count++;
            }else if(answers[i] == cur) {
                count++;
                if(count == cur + 1) {
                    count = 0;
                    res += (cur + 1);
                }
            }else {
                res += (cur+1);
                count = 1;
                cur = answers[i];
            }
        }
        if(count != 0) res += (cur + 1);
        return res;
    }
}
