class Solution {
    public int numRabbits(int[] answers) {
        // the rabit with the same color should have the same answer
        // the count of the same color is limited
        
        Map<Integer, Integer> map = new HashMap<Integer, Integer>(); // the count of each answer
        int n = answers.length;
        
        for (int i = 0 ; i < n; ++i) {
            
            if (!map.containsKey(answers[i])) map.put(answers[i], 0);
            map.put(answers[i], map.get(answers[i])+1);
            
        
}
        
        int res = 0;
        
        for (Map.Entry<Integer, Integer> e: map.entrySet()) {
            
            int ans = e.getKey();
            int count = e.getValue();
            
            while (count > 0) {
                
                res += 1 + ans;
                count -= 1 + ans;
                
            }
            
        }
        
        return res;
        
    }
}
