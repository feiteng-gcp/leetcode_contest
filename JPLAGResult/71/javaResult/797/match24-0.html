class Solution {
    public int numRabbits(int[] answers) {
        Map<Integer,Integer> map=new HashMap<>();
        for (int x:answers) map.put(x,map.getOrDefault(x,0)+1);
        int ans=0;
        for (int x:map.keySet())
        {
            int o=map.get(x);
            int p=(o-1)/(x+1)+1;
            ans+=p*(x+1);
        }
        return ans;
    }
}
