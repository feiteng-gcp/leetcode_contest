class Solution {
    public int numRabbits(int[] answers) {
        if (answers.length == 0) return 0;
        int[] f= new int[1002];
        int sum = 0;
        
        for (int i : answers) {
            i++;
            f[i]++;
            if (f[i] >= i) {
                f[i] = 0;
                sum += i;
            }
        }
        
        for (int i = 0; i < f.length; i++) {
            if (f[i] > 0) {
                sum += i;
            }
        }
        
        return sum;
    }
}
