class Solution {
    public int numRabbits(int[] answers) {
        int[] countPresent = new int[1000];
        for(int i=0; i<answers.length; i++)
            countPresent[answers[i]]++;
        
        int sum = 0;
        for(int i=0; i<countPresent.length; i++)
            sum += Math.ceil((double)countPresent[i]/(i+1))*(i+1);
        
        return sum;
    }
}
