class Solution {
    public int numRabbits(int[] answers) {
        int res = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for (int n : answers) {
            if (map.containsKey(n) && map.get(n) > 0) {
                map.put(n, map.get(n) - 1);
                continue;
            }
            res = res + n + 1;
            map.put(n, n);
        }
        return res;
    }
}
