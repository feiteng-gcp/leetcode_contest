class Solution {
    public int numRabbits(int[] answers) {
        HashMap<Integer, Integer> count = new HashMap<Integer, Integer>();
        int ret = 0;
        for (int v : answers) {
            Integer c = count.get(v+1);
            if (c==null) c = 0;
            count.put(v+1, c+1);
        }
        for (Map.Entry<Integer, Integer> entry : count.entrySet()) {
            Integer key = entry.getKey(), c = entry.getValue();
            ret += divideCeiling(c, key) * key;
        }
        return ret;
    }
    
    private int divideCeiling(int a, int b) {
        //return the minimum integer >= a/b
        if (a%b==0) return a/b;
        else return (a/b) + 1;
    }
}
