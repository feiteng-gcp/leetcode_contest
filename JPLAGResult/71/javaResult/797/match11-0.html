class Solution {
    public int numRabbits(int[] ans) {
        if(ans == null || ans.length == 0)
            return 0;
        
        int count = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for(int a : ans) {
            if(a == 0) {
                count++;
                continue;
            }
            if(map.containsKey(a)) {
                int expected = map.get(a);
                expected--;
                if(expected == 0)
                    map.remove(a);
                else
                    map.put(a, expected);
            } else {
                map.put(a, a);
            }
            count++;
        }
        
        for(Map.Entry<Integer, Integer> entry : map.entrySet())
            count += entry.getValue();
        
        return count;
        
    }
}
