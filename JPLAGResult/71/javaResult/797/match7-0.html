class Solution {
	private int[] cnt = new int[1000];

    public int numRabbits(int[] answers) {

    	Arrays.fill(cnt , 0);
    	for (int value : answers) {
    		cnt[value] ++;
    	}
    	int ans = 0;
    	for (int i = 0;i < 1000;i ++) {
    		int batch = i + 1;
    		int value = cnt[i];
    		int temp = value / batch;
    		if (value % batch != 0) {
    			temp ++;
    		}
    		ans += temp * batch;
    	}
    	return ans;
    	
    }
}
