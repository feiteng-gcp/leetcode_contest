class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if(sx > tx || sy > ty ){
            return false;
        }
        
        if(sx == tx && sy == ty){
            return true;
        }
        
        if(tx > ty){
            return reachingPoints(sy, sx, ty, tx);
        }
        
        if(tx == ty){
            return false;
        }
        
        
        int div = (ty- Math.max(sy, tx))%tx == 0?(ty- Math.max(sy, tx))/tx: (ty- Math.max(sy, tx))/tx  +1;
        
        ty = ty - div*tx;
        
        if(ty == sy){
            return (tx - sx)%ty== 0;
        }
        
        
        
        return reachingPoints(sx, sy, tx, ty);
        
        
        
        
    }
}
