class Solution {
 public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        int currentX = tx;
        int currentY = ty;

        while (currentX > sx && currentY > sy) {
            if (currentX > currentY) {
                currentX %= currentY;
            } else if (currentY > currentX) {
                currentY %= currentX;

            } else {
                if (sx == 0) {
                    currentX -= currentY;
                } else {
                    currentY -= currentX;
                }

            
}
        }

        if (currentX == sx && currentY == sy) return true;

        if (currentX == sx && (currentY % currentX == sy % sx )) {
            return true;
        }

        if (currentY == sy && (currentX % currentY == sx % sy )) {
            return true;
        }

        return false;


    }
}
