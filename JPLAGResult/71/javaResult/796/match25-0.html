class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
   
    	while (tx > 0 && ty > 0) {
    		if (tx == ty) {
    			if (sx == tx && sy == ty) {
    				return true;
    			} else {
    				return false;
    			}
    		} else {
    			if (tx > ty) {
    				if (sy == ty && sx <= tx) {
    					int temp = tx - sx;
    					if (temp % ty == 0) {
    						return true;
    					} else {
    						return false;
    					}
    				} else {
    					tx %= ty;
    				}
    			} else if (tx < ty) {
    				if (sx == tx && sy <= ty) {
    					int temp = ty - sy;
    					if (temp % tx == 0) {
    						return true;
    					} else {    						
    						return false;
    					}
    				} else {
    					ty %= tx;
    				}
    			}
    		}
    	}
    	return false;
    	
    }
}
