class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        int x = tx, y = ty;
        while(x>sx && y>sy) {
            if (x>y) x %= y;
            else y %= x;
        }
        if (x<sx || y<sy) return false;
        if (x==sx) {
            return ((y-sy)%x==0);
        } else {
            return ((x-sx)%y==0);
        }
    }
}
