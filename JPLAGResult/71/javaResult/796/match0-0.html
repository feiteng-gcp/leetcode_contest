class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if (sx == tx && sy == ty) {
            return true;
        }
        
        while (tx >= sx && ty >= sy) {
            if (tx > ty) {
                if (ty == sy) {
                    return (tx - sx) % sy == 0;
                }
                tx = tx % ty;
            } else {
                if (tx == sx) {
                    return (ty - sy) % sx == 0;
                }
                ty = ty % tx;
            }
            
            if (sx == tx && sy == ty) {
                return true;
            }
        }
        
        return false;
    }
}
