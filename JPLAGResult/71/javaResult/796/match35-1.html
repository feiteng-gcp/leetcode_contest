class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        // inverse call
        if (tx < sx || ty < sy) return false;
        
        if (tx < ty) {
            
            // ty % tx
            if (tx == sx && (ty - sy)%tx == 0) return true;
            
            ty = ty % tx;
            return reachingPoints(sx, sy, tx, ty);
            
        }
        else {
            
            if (ty == sy && (tx - sx)%ty == 0) return true;
            tx = tx % ty;
            return reachingPoints(sx,  sy,  tx,  ty);
            
        }
        
        
    }
}
