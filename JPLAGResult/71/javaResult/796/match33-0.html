class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if (tx == sx) {
            return ((ty - sy) % tx == 0);
        }
        if (ty == sy) {
            return ((tx - sx) % ty == 0);
        }
        if (tx < ty) {
            ty = ty % tx;
            if (sx == tx && sy == ty) return true;
            if (sx > tx || sy > ty) return false;
        }
        else if (ty < tx) {
            tx = tx % ty;
            if (sx == tx && sy == ty) return true;
            if (sx > tx || sy > ty) return false;
        } else {
            return false;
        }
        if (sx > tx || sy > ty) return false;
        return reachingPoints(sx, sy, tx, ty);
    }
}
