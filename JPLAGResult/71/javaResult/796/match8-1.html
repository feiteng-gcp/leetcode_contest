class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        while (tx != ty) {
            // System.out.printf ("tx:(%d), ty:(%d)\n", tx, ty);///
            if (tx > ty) {
                if (tx % ty > 0)
                    tx %= ty;
                else
                    tx -= ty;
            // System.out.printf ("tx:(%d), ty:(%d)\n\n", tx, ty);///
                if (sy == ty)
                    return (sy == 1 && tx > sx) || sx == tx || Math.abs (sx - tx) == sy || (sy > 1 && (sx - tx) % sy == 0);
            } else {
                if (ty % tx > 0)
                    ty %= tx;
                else
                    ty -= tx;
            // System.out.printf ("tx:(%d), ty:(%d)\n\n", tx, ty);///
                if (sx == tx)
                    return (sx == 1 && ty > sy) || sy == ty || Math.abs (sy - ty) == sx || (sx > 1 && (sy - ty) % sx == 0);
            }
        }
        return sx == tx && sy == ty;
    }
}
