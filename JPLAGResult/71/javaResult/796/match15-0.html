class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if(sx==tx&&sy==ty){
            return true;
        }
        if(sx>tx||sy>ty){
            return false;
        }
        if(tx==sx){
            return (ty-sy)%sx==0;
        }
        if(ty==sy){
            return (tx-sx)%sy==0;
        }
        if(tx<ty){
            return reachingPoints(sx,sy,tx,ty-tx);
        }
        return reachingPoints(sx,sy,tx-ty,ty);
    }
}
