class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        while(tx>=sx&&ty>=sy) {
            if(sx==tx&&ty==sy)
                return true;
            else if(sx==tx) {
                double n = (double)(ty-sy)/tx;
                return n==Math.floor(n);
            }
            else if(ty==sy) {
                double n = (double)(tx-sx)/ty;
                return n==Math.floor(n);
            }
            
            if(ty>tx) {
                int n = (int)Math.ceil((double)ty/tx -1);
                if(n<=0)
                    n = 1;
                ty = ty - n*tx;
            }
            else {
                
                int n = (int)Math.ceil((double)tx/ty - 1);
                if(n<=0)
                    n = 1;
                tx = tx - n*ty;
                
            }
        }
        return false;
    }
}
