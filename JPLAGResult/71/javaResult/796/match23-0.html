class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if (tx == sx && ty==sy) return true;
        if (sx * sy * tx * ty ==0) return (sx==tx && sy==ty);
        while (tx >= sx && ty >= sy) { 
            //System.out.println("start "+sx+" "+sy+" "+tx+" "+ty);
            if (tx>ty) {
                tx = tx%ty;
                if (tx == 0) tx=ty;
                //System.out.println("end "+sx+" "+sy+" "+tx+" "+ty);
                continue;
            }
            if (ty>tx) {
                ty = ty%tx;
                if (ty == 0) ty=tx;
                //System.out.println("end "+sx+" "+sy+" "+tx+" "+ty);
                continue;
            }
            if (tx == ty) {
                //sx = sx % sy;
                //if (sx == 0) sx = sy;
                //sy = sy % sx;
                //if (sy == 0) sy = sx;
                //System.out.println("end "+sx+" "+sy+" "+tx+" "+ty);
                return (tx==sx && ty==sy);
            }
            
        }
        System.out.println("end "+sx+" "+sy+" "+tx+" "+ty);
        if (tx == sx && ty==sy) return true;
        if (sx < sy) {
            sy=sy%sx;
            if (sy==0) sy=sx;
        }
        else {
            sx=sx%sy;
            if (sx ==0) sx = sy;
        }
        return (tx == sx && ty==sy);
    }
}
