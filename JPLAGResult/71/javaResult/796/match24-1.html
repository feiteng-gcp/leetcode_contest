class Solution {

    public boolean reachingPoints(int sx, int sy, int tx, int ty) 
{
        
        if(ty == sy) {
            if(tx >= sx && (tx - sx) % sy == 0 ) {
                return true;
            }
            else {
                return false;
            }
        }
        if(tx == sx) {
            
if(ty >= sy && (ty - sy) % sx == 0 ) {
                return true;
            }
            else {
                return false;
            }
        }
        while(tx>=1 && ty >=1 && tx >= sx && ty >= sy) {
            if(tx == sx && ty == sy) {
                return true;
            }
            if(tx == ty) {
                return false;
            }
            else if(tx > ty) {
                tx = tx -ty;
            }
            else { //tx < ty
                ty = ty -tx;
            }
        }
        
        return false;
    }
}
