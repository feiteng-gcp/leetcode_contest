class Solution {
    
    class Point {
        int x, y;
        public Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
        
        @Override
        public boolean equals(Object o) {
            if (this == o) return true;
            if (o == null || getClass() != o.getClass()) return false;

            Point point = (Point) o;

            if (x != point.x) return false;
            return y == point.y;

        }

        @Override
        public int hashCode() {
            int result = x;
            result = 31 * result + y;
            return result;
        }        
    }
    
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if (sx == tx) {
            return (ty - sy) % sx == 0;
        }
        if (sy == ty) {
            return (tx - sx) % sy == 0;
        }
        
        Point start = new Point(sx, sy);
        Set<Point> startSet = new HashSet<>();
        startSet.add(start);
        Point end = new Point(tx, ty);
        Set<Point> endSet = new HashSet<>();
        endSet.add(end);
        
        Set<Point> used = new HashSet<>();
        
        boolean forward = true;
        while (startSet.size() > 0 && endSet.size() > 0) {
            // System.out.println(startSet.size() + " " + endSet.size() );
            if (startSet.size() > endSet.size()) {
                Set<Point> temp = endSet;
                endSet = startSet;
                startSet = temp;
                forward = !forward;
            }
            
            Set<Point> nextStart = new HashSet<>();
            for (Point cur: startSet) {
                if (endSet.contains(cur)) {
                    return true;
                }
                used.add(cur);
                
                if (forward) {
                    if (cur.x <= end.x && cur.x + cur.y <= end.y) {
                        Point left = new Point(cur.x, cur.x + cur.y);
                        if (!used.contains(left)) {
                            nextStart.add(left);
                        }
                    }
                    if (cur.x + cur.y <= end.x && cur.y <= end.y) {
                        Point right = new Point(cur.x + cur.y, cur.y);
                        if (!used.contains(right)) {
                            nextStart.add(right);
                        }            
                    }
                } else {
                    if (cur.x >= start.x && cur.y - cur.x >= start.y) {
                        Point left = new Point(cur.x, cur.y - cur.x);
                        if (!used.contains(left)) {
                            nextStart.add(left);
                        }
                    }
                    if (cur.x - cur.y >= start.x && cur.y >= start.y) {
                        Point right = new Point(cur.x - cur.y, cur.y);
                        if (!used.contains(right)) {
                            nextStart.add(right);
                        }            
                    }
                }
            }
            startSet = nextStart;
        }
        return false;
    }
}
