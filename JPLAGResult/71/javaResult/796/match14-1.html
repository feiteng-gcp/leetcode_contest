class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if(sx==tx&&sy==ty) return true;
        if(tx==ty) return false;
        while(tx>=sx&&ty>=sy){
            if(sx==tx&&sy==ty) return true;
            if(tx>ty){
                if(tx-ty<sx) return false;
                tx=(tx-sx)%ty+sx;
            }
            else if(tx==ty) return false;
            else{
                if(ty-tx<sy) return false;
                ty=(ty-sy)%tx+sy;
            
}
        }
        return false;
    }
}
