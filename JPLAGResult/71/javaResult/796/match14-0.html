class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {   
        while ((sx != tx || sy != ty) && tx > 0 && ty > 0 && sx <= tx && sy <= ty) {
            //System.out.println(tx);
            //System.out.println(ty);
            if (tx > ty) {
                int k = (int)((tx - sx) / ty);
                if (k == 0) {
                    return false;
                }
                tx -= k * ty;
            } else if (tx < ty) {
                int k = (int)((ty - sy) / tx);
                if (k == 0) {
                    return false;
                }
                ty -= k * tx;
            } else {
                return false;
            }
        }
        return sx == tx && sy == ty;
    }
}
