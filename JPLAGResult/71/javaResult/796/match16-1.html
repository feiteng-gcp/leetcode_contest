class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        while (true){
            if (sx == tx && sy ==ty){
                return true;
            }
            if (tx == ty){
                return false;
            }
            if (sx > tx || sy > ty){
                return false;
            }
            if (tx > ty){
                int times = Math.max((tx - Math.max(sx,ty)) / ty, 1);
                tx = tx - ty * times;
            } else {
                int times = Math.max((ty - Math.max(sy,tx)) / tx, 1);
                ty = ty - tx * times;
            }
        }
        // return false;
    }
