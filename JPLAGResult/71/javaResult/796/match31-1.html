class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if (sx == tx && sy == ty) {
            return true;
        }
        while (tx > 0 && ty > 0) {
            if (tx > ty) {
                int count = Math.max(1, (tx - sx) / ty);
                tx -= count * ty;
            }
            else {
                int count = Math.max(1, (ty - sy) / tx);
                ty -= count * tx;
            }
            if (tx == sx && ty == sy) {
                return true;
            }
        }
        return false;
    }
}
