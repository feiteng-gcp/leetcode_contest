class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        //(x,y)-> (x, x+ y)
        int x = tx;
        int y = ty;
        while (tx >= sx && ty >= sy) {
            if (tx < ty) {
                x = tx;
                y = ty - tx;
            } else if (tx > ty) {
                x = tx - ty;
                y = ty;
            } 
            tx = x;
            ty = y;
            if (tx == sx && ty == sy) {
                return true;
            }
            if (tx == ty)
                return false;
            if (ty == sy) {
                if ((tx - sx)% ty == 0)
                    return true;
                else
                    return false;
            }
            if (tx == sx)
            {
                if ((ty - sy)% tx == 0)
                    return true;
                else
                    return false;
            }
              
        }
        return false;
    }
}
