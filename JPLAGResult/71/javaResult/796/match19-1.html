class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        return g(sx,sy,tx,ty);
    }
    
    HashSet<Long> c = new HashSet<>();
        
        
    public boolean g(int sx, int sy, int tx, int ty){
        if(c.contains((long)(tx*10000007+ty))){
            return false;
        
}
        if(tx<sx||ty<sy) return false;
        if(tx==sx&&ty==sy) return true;
        if(tx>ty){
            int n = tx/ty;
            if(ty==sy&&((tx-sx)%ty)==0) return true;
     //       for(int i=1;i<=n;i++){
    //            if(tx-i*ty==sx&&ty==sy) return true;
    //        }
            
            
       //     for(int i=1;i<=n;i++){

                if(g(sx,sy,tx-n*ty,ty)){
                    return true;
                }else{
                    c.add((long)((tx-n*ty)*10000007+ty));
                }
        //    }
        }else{
            int n = ty/tx;
            if(tx==sx&&((ty-sy)%tx)==0) return true;
            
      //      for(int i=1;i<=n;i++){
     //           if(tx==sx&&ty-i*tx==sy) return true;
    //        }
   //         for(int i=1;i<=n;i++){
                if(g(sx,sy,tx,ty-n*tx)){
                    return true;
                }else{
                    c.add((long)(tx*10000007+ty-n*tx));
                }
  //          }
        }
        return false;
    }
}
