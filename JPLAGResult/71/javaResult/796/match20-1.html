class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
    	
    	if (sx == tx && sy == ty) return true;
    	if (sx > tx || sy > ty) return false;
    	
    	if (sx == tx){
    		int left = ty-sy;
    		return (left/sx) * sx == left;
    	} else if (sy == ty){
    		int left = tx-sx;
    		return (left/sy) * sy == left;
    	}
    	
    	if (ty > tx){
    		int x = tx;
    		int y = ty-tx;
    		return reachingPoints(sx, sy, x, y);
    	} else if (ty < tx) {
    		int x = tx-ty;
    		int y = ty;
    		return reachingPoints(sx, sy, x, y);
    	} else {
    		if ((ty / 2) * 2 != ty) return false;
    		int x = ty/2;
    		int y = ty/2;
    		return reachingPoints(sx, sy, x, y);
    	}
    }
    
}
