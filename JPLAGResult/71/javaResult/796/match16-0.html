class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
		while (tx >= sx && ty >= sy) {
			if (tx == sx && ty == sy) {
				return true;
			}
			if (tx >= ty) {
				tx = Math.min((tx - sx) % ty + sx, tx - ty);
			} else {
				ty = Math.min((ty - sy) % tx + sy, ty - tx);
			}
		}
		return false;
    }
}
