class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if (tx == sx && ty == sy)
            return true;
        if (tx < sx+sy && ty < sx + sy)
            return false;
        
        int x = tx;
        int y = ty;
        while(x > 0 && y > 0) {
            if (sx == x && sy == y)
                return true;
            if (x == y)
            		return false;
            if (x < sx || y < sy)
            		return false;
            int nextX;
            int nextY;
            if (x > y) {
            	if (x/y == 1)
            		nextX = x - y;
            	else {
            		int k = (x - sx) / y;
            		nextX = x - k*y;
                    if (nextX == x)
            			return false;
            	}
                nextY = y;
            } else {
                nextX = x;
                if (y/x == 1)
        			nextY = y - x;
        		else {
        			int k = (y - sy) / x;
        			nextY = y - k*x;
                    if (nextY == y)
            			return false;
        		}
            }
            x = nextX;
            y = nextY;
        }
        return false;
    }
}
