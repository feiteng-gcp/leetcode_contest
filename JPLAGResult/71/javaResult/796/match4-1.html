class Solution {
public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        while(tx!=sx||ty!=sy) {
            if(tx==sx){
                return (ty-sy)%tx==0;
            
}
            if(sy==ty){
                return (tx-sx)%ty==0;
            }
            if (tx < sx || ty < sy) {
                return false;
            }
            if (tx == ty) {
                return false;
            }
            if (tx > ty) {
                tx = tx-ty;
            }
            else{
                ty = ty-tx;
            }
        }
        return true;
    }
}
