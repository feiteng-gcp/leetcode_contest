class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        int sxm= sx;
        int sym= sy;
        int tym= ty;
        int txm= tx;
            
        boolean canReach = false;
        while(sx <= tx && sy<= ty){
            if(sx==tx && sy==ty){
                canReach = true;
                break;
            }
            if(sx == tx){
                canReach=(((ty-sy)%sx)==0 && (ty > sy))?true:false;
                break;
            } else if(sy==ty){
                canReach=(((tx-sx)%sy)==0 && (tx > sx))?true:false;
                break;
            }
            if(tx == ty){
                canReach = false;
                break;
            } else if (tx < ty){
                ty=ty-tx;
            } else {
                tx=tx-ty;
            }
        }
        return canReach;
    }
}
