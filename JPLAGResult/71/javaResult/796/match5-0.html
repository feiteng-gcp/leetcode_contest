class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        if (sx == tx && sy == ty) {
            return true;
        }
        
        if (tx < sx || ty < sy || tx == ty) {
            return false;
        }
        
        if (tx < ty) {
            int a = Math.max(1, (ty - sy) / tx);
            return reachingPoints(sx, sy, tx, ty - a * tx);
        } else {
            int a = Math.max(1, (tx - sx) / ty);
            return reachingPoints(sx, sy, tx - a * ty, ty);
        }
    }
}
