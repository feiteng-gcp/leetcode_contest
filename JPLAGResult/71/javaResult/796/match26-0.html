class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        while (true)
        {
            if (tx==sx && ty==sy) return true;
            if (tx==sx)
            {
                return (ty-sy)%tx==0;
            }
            if (ty==sy)
            {
                return (tx-sx)%ty==0;
            }
            if (tx>ty) 
            {
                tx%=ty; 
            }
            else 
            {
                ty%=tx;
            }
            if (tx==sx && ty==sy) return true;
            if (tx<sx) return false;
            if (ty<sy) return false;
        }
        //return (tx==sx && ty==sy);
    }
}
