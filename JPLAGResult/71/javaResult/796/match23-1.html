class Solution {
    public boolean reachingPoints(int sx, int sy, int tx, int ty) {
        
        if(tx == sx && sy == ty) return true;
        while(!(sx > tx || sy > ty)) {
            
            if(tx == sx && sy == ty) return true;
            
            if(tx < ty) {
                
                int max = Math.max(sy, tx);
                int k = (ty - max)/tx;
                if(k == 0) k++;
                ty = ty - k*tx;
                
            } else if(ty < tx) {
                int max = Math.max(sx, ty);
                int k = (tx - max)/ty;
                if(k == 0) k++;
                tx = tx - k*ty;
                
            } else {
                return false;
            }
        }
        
        if(tx == sx && sy == ty) return true;
        return false;
    }
}
