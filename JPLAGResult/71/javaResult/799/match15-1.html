/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minDiffInBST(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int[] min = new int[] {Integer.MAX_VALUE};
        search(root, min);
        return min[0];
    }
    
    public int[] search(TreeNode node, 
int[] min) {
        if (node == null) {
            return null;
        }
        int[] left = search(node.left, min);
        int[] right = search(node.right, min);
    
        if (left != null || right != null) {
            if (left == null) {
                min[0] = Math.min(min[0], Math.abs(right[0] - node.val));
            } else if (right == null) {
                min[0] = Math.min(min[0], Math.abs(left[1] - node.val));
            } else {
                min[0] = Math.min(min[0], Math.min(Math.abs(left[1] - node.val), Math.abs(right[0] - node.val)));
            }
        }
        int[] current = new int[] {node.val, node.val};
        if (left != null) {
            current[0] = left[0];
        }
        if (right != null) {
            current[1] = right[1];
        }
        return current;
    }
}
