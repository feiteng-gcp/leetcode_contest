/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    // traverse inorder and find the min distance
    public int minDiffInBST(TreeNode root) {
        LinkedList<Integer> list = new LinkedList<Integer>();
        sortedList(root, list);
        Iterator<Integer> it = list.iterator();
        int minDiff = Integer.MAX_VALUE;
        int prev = it.next();
        while(it.hasNext()) {
            int curr = it.next();
            if(Math.abs(prev-curr) < minDiff)
                minDiff = Math.abs(prev-curr);
            prev = curr;
        }
        return minDiff;
    }
    
    public void sortedList(TreeNode root, 
LinkedList<Integer> list) {
        if(root == null)
            return;
        
        sortedList(root.left, list);
        list.push(root.val);
        sortedList(root.right, list);
    }
}
