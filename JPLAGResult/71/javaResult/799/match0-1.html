/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minDiffInBST(TreeNode root) {
        int res = Integer.MAX_VALUE;
        Stack<TreeNode> stack = new Stack<>();
        TreeNode tmp = root;
        
TreeNode pre = null;
        while(!stack.isEmpty() || tmp != null) {
            if (tmp != null) {
                stack.push(tmp);
                tmp = tmp.left;
            }
            else {
                tmp = stack.pop();
                if (pre != null && tmp.val - pre.val < res) {
                    res = tmp.val - pre.val;
                
}
                pre = tmp;
                tmp = tmp.right;
            }
        }
        return res;
    }
}
