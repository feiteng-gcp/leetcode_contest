/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int diff;
    public int minDiffInBST(TreeNode root) {
        diff = Integer.MAX_VALUE;
        if(root == null || root.left == null && root.right==null) {
            return diff;
        }
        
        
process(root);
        
        return (int)diff;
    }
    
    private int[] process(TreeNode node) {
        if(node == null) {
            return null;
        }
        
        int[] left = process(node.left);
        int[] right = process(node.right);
        
        int s = node.val ;
        int b = node.val;
        
        if(left != null) {
            diff = Math.min(node.val - left[1], diff);
            s = Math.min(node.val, left[0]);
        }
        if(right != null) {
            diff = Math.min(right[0] - node.val, diff);
            b = Math.max(node.val, right[1]);
        }
        
        return new int[]{s, b};
    }
}
