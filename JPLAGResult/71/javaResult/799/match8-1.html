/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minDiffInBST(TreeNode root) {
        Integer[] prev = new Integer[1];
        Integer[] ans = new Integer[1];
        dfs(root, prev, ans);
        return ans[0];
    }
    
    void dfs(TreeNode rt, 
Integer[] prev, Integer[] ans) {
        if (rt == null) {
            return;
        }
        dfs(rt.left, prev, ans);
        
        if (prev[0] != null) {
            if (ans[0] == null || ans[0] > rt.val - prev[0]) {
                ans[0] = rt.val - prev[0];
            }
        }
        prev[0] = rt.val;
        
        dfs(rt.right, prev, ans);
    }
}
