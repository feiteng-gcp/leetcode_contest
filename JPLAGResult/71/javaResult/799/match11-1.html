/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
	private boolean first = true;
	private long ans = - 1 , prev = - 1;
	
    public int minDiffInBST(TreeNode root) {
     
    	prev = - 1;
    	ans = - 1;
    	first = true;
    	dfs(root);
    	ret
urn (int) ans;
    	
    }
    
    
    private void dfs(TreeNode root) {
    	
    	if (root.left != null) {
    		dfs(root.left);
    	}
    	if (first) {
    		first = false;
    		prev = root.val;
    	} else {
    		long temp = (long) root.val - prev;
    		if (temp < ans || ans < 0) {
    			ans = temp;
    		}
    		prev = root.val;
    	}
    	if (root.right != null) {
    		dfs(root.right);
    	}
    	
    }
}
