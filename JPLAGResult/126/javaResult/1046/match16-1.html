class Solution {
    public int longestOnes(int[] A, int K) {
        int zeros = 0, start = 0, max = 0;
        for (int end=0; end<A.length; end++) {
            if (A[end] == 0) {
                zeros++;
            }
            while (zeros > K) {
                if (A[start] == 0) {
                    zeros--;
                }
                start++;
            }
            max = Math.max(max, end-start+1);
        }
        return max;
    }
}
