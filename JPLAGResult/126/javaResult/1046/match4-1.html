class Solution {
    public int longestOnes(int[] A, int K) {
        int cnt = K;
        int res = 0;
        int i = 0;
        for (int j = 0; j < A.length; j++) {
            if (A[j] == 0) {
                cnt -= 1;
            }
            while (cnt == -1) {                
                if (A[i] == 0) {
                    cnt += 1;
                }
                i += 1;
            }
            res = Math.max(res, j - i + 1);
        }
        return res;
    }
}
