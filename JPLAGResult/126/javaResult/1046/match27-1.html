class Solution {
    public int longestOnes(int[] A, int K) {
        int res = 0, zero = 0, l = 0;
        for (int r = 0; r < A.length; ++r) {
            if (A[r] == 0) ++zero;
            
while (zero > K) {
                if (A[l++] == 0) --zero;
            }
            res = Math.max(res, r - l + 1);
        }
        return res;
    }
}
