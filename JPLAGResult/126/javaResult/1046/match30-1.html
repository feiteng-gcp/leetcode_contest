class Solution {
    public int longestOnes(int[] A, int K) {
        int start = 0;
        int count = 0;
        int res = 0;
        for(int i=0; i<A.length; i++) {
            if(A[i] == 0) {
                count++;
            }
            while(count > K) {
                if(A[start] == 0) {
                    count--;
                
}
                start++;
            }
            res = Math.max(res, i-start+1);
        }
        return res;
    }
}
