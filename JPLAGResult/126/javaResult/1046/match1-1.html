class Solution {
    public int longestOnes(int[] A, int K) {
        int res = 0;
        int left = 0, zeros = 0;
        for (int i = 0; i < A.length; i++) {
            if (A[i] == 0) {
                zeros++;
            }
            while (zeros > K) {
                if (A[left] == 0) {
                    zeros--;
                }
                left++;
            }
            res = Math.max(res, i - left + 1);
        }
        
        return res;
    }
}
