class Solution {
    public int longestOnes(int[] A, int K) {
        
int start = 0;
        int ret = 0;
        int k = K;
        for(int i = 0; i < A.length; i++) {
            if(A[i] == 1) {
                ret = Math.max(ret, i - start + 1);
            } else {
                if(k > 0) {
                    k--;
                    ret = Math.max(ret, i - start + 1);
                } else {
                    while(start <= i && k == 0) {
                        if(A[start] == 0) k++
;
                        start++;
                    }
                    k--;
                    ret = Math.max(ret, i - start + 1);
                }
            }
        }
        return ret;
    }
}
