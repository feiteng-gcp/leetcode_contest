class Solution {
    public int longestOnes(int[] A, int K) {
        if (A == null || A.length == 0) {
            return 0;
        }
        
        int len = A.length;
        int i = 0;
        int j = 0;
        int zeroCount = 0;
        int ans = 0;
        
        for (j = 0; j < len; j++) 
{
            if (A[j] == 0) {
                zeroCount++;
            } 
            
            while (zeroCount > K) {
                if (A[i] == 0) {
                    zeroCount--;
                }
                i++;
            }
            ans = Math.max(ans, j - i + 1);

        }
        
        return ans;
    }
}
