class Solution {
   public int longestOnes(int[] A, int K) {
        int ans = 0;
        int slow = 0;
        int len = 0;
        int n = 0;
        for (int i = 0; i < A.length; i++) {
               if (A[i] == 1) {
                   len += 1;
                   ans = Math.max(ans, len);
               }else {
                   if (n < K) {
                       len++;
                       ans = Math.max(ans, len);
                       n++;
                   }else {
                       while(n == K) {
                           int s = A[slow++];
                           len--;
                           if (s == 0) {
                               n--;
                           }
                       }
                       len++;
                       n++;
                       ans = Math.max(ans, len);
                   }
               }
        }
        return ans;
    }
}
