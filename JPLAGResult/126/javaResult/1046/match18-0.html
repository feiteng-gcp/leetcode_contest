class Solution {
    public int longestOnes(int[] A, int K) {
        int max = 0, zeros = 0, left = 0;
        for (int i = 0; i < A.length; i++) {
            if (A[i] == 0) {
                zeros++;
            }
            while (zeros > K) {
                if (A[left++] == 0) {
                    zeros--;
                }
            }
            
            max = Math.max(max, i - left + 1);
        }
        
        return max;
    }
}
