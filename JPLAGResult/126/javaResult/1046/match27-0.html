class Solution {
    public int longestOnes(int[] A, int K) {
        int startIndex = 0;
        int zeroCount = 0;
        int res = 0;
        for(int i=0; i<A.length; i++){
            zeroCount+=(A[i]==0?1:0);
            while(zeroCount>K){
                if(A[startIndex]==0) zeroCount--;
                startIndex++;
            }
            res=Math.max(res,i-startIndex+1);
        }
        
        return res;
    }
}
