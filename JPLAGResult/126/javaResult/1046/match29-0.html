class Solution {
    public int longestOnes(int[] A, int K) {
        if(A == null || A.length == 0){
            return 0;
        }
        int start = 0, end = 0;
        int count = 0, max = 0;
        while(end < A.length){
            if(A[end++] == 0){
                count++;
            }
            while(count > K){
                if(A[start++] == 0){
                    count--;
                }
            }
            max = Math.max(max, end - start);
        }
        
        return max;
    }
}
