class Solution {
    public List<String> commonChars(String[] A) {
        
HashMap<Character, Integer>[] maps = new HashMap[A.length];
        for (int i = 0;i < A.length; i++) {
            maps[i] = new HashMap<>();
            for (char aa : A[i].toCharArray()) {
                maps[i].put(aa, maps[i].getOrDefault(aa, 0) + 1);
            }
        }
        List<String> res = new ArrayList<>();
        for (char ii = 'a'; ii <= 'z'; ii++) {
            int val = Integer.MAX_VALUE;
            for (int i = 0; i < A.length; i++) {
                int cur = maps[i].getOrDefault(ii, 0);
                val = 
Math.min(val, cur);
            }
            while (val-- > 0) {
                res.add(ii + "");
            }
        }
        return res;
    }
}
