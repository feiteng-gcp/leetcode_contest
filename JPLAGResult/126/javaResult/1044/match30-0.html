	class Solution {
	    public List<String> commonChars(String[] A) {
	        int[] f = new int[26];
	        Arrays.fill(f, 99999);
	        for(String x : A){
	        	int[] lf = new int[26];
	        	for(char c : x.toCharArray()){
	        		lf[c-'a']++;
	        	}
	        	for(int i = 0;i < 26;i++){
	        		f[i] = Math.min(f[i], lf[i]);
	        	}
	        }
	        List<String> ret = new ArrayList<>();
	        for(int i = 0;i < 26;i++){
	        	for(int j = 0;j < f[i];j++){
	        		ret.add("" + (char)('a'+i));
	        	}
	        }
	        return ret;
	    }
	}
