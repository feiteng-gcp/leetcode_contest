class Solution {
    public List<String> commonChars(String[] A) {
        
int[] res = new int[26];
        Arrays.fill(res, Integer.MAX_VALUE);
        for (String str : A) {
            int[] count = new int[26];
            for (int i = 0; i < str.length(); i++) {
                count[str.charAt(i) - 'a']++;
            
}
            for (int i = 0; i < 26; i++) {
                res[i] = Math.min(res[i], count[i]);
            }
        }
        List<String> result = new ArrayList<>();
        
for (int i = 0; i < 26; i++) {
            while (res[i] != Integer.MAX_VALUE && res[i] > 0) {
                result.add((char)('a' + i) + "");
                res[i]--;
            }
        }
        return result;
        
    }
}
