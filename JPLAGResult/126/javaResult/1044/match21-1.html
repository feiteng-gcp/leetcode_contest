class Solution
{
    public List<String> commonChars(String[] st) 
    {
    int[][] ar = new int[st.length][26];
    List<String> list = new ArrayList<String>();
    
for(int i = 0;i < st.length; i++)
    {
        for(int j = 0;j < st[i].length(); j++)
        {
            ar[i][st[i].charAt(j)-'a']++;
        }
    }
        
    
for(int i = 0;i < 26; i++)
    {
        int max = Integer.MAX_VALUE;
        for(int j = 0;j < st.length; j++)
        {
            max = Math.min(max, ar[j][i]);
        }
        char ch = (char)(97+i);
        
for(int k = 0; k < max; k++)
        {
            list.add(Character.toString(ch));
        }
    }
        
    return list;
    }
}
