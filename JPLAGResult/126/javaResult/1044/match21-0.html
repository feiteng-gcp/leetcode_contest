class Solution {
    public List<String> commonChars(String[] A) {
        int[][] bucket = new int[A.length][26];
        
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[i].length(); j++) {
                bucket[i][A[i].charAt(j) - 'a']++;
            }
        }
        
        List<String> ret = new ArrayList<>();
        
        for (int j = 0; j < 26; j++) {
            int min = Integer.MAX_VALUE;
            for (int i = 0; i < A.length; i++) {
                min = Math.min(bucket[i][j], min);
            }
            for (int i = 0; i < min; i++) {
                ret.add(Character.toString((char)('a' + j)));
            }
        }
        
        return ret;
    }
}
