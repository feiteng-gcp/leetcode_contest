class Solution {
    public List<String> commonChars(String[] A) {
        int[] arr = new int[26];
        for (int i = 0; i < 26; ++i) {
            arr[i] = 100;
        }
         
for (String s : A) {
            int[] tmp = new int[26];
            for (char c : s.toCharArray()) {
                tmp[c - 'a']++;
            }
            for (int i = 0; i < 26; ++i) {
                arr[i] = Math.min(arr[i], tmp[i]);
            }
        }
        List<String> result = new ArrayList<>();
        for (int i = 0; i < 26; ++i) {
            for (int j = 0; j < arr[i]; ++j) {
                String s = ((char) ('a' + i)) + "";
                
result.add(s);
            }
        }
        return result;
    }
}
