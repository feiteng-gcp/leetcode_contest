class Solution {
    public List<String> commonChars(String[] A) {
        int n = A.length;
        Map<Character, Integer>[] map = new Map[n];
        for (int i = 0; i < n; i++) {
            map[i] = new HashMap<>();
            for (char c : A[i].toCharArray()) {
                map[i].put(c, map[i].getOrDefault(c, 0) + 1);
            }
        }
        List<String> res = new ArrayList<>();
        for (char c = 'a'; c <= 'z'; c++) {
            int count = Integer.MAX_VALUE;
            for (int i = 0; i < n; i++) {
                count = Math.min(count, map[i].getOrDefault(c, 0));
            }
            while (count-- > 0) {
                res.add("" + c);
            }
        }
        return res;
    }
}
