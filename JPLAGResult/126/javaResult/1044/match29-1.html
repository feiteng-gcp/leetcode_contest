class Solution {
    public List<String> commonChars(String[] A) {
        int[] result = new int[26];
        for (int i = 0; i < 26; i++) {
            result[i] = Integer.MAX_VALUE;
        }
        for (String s : A) {
            int[] tmp = new int[26];
            for (int i = 0; i < s.length(); i++) {
                tmp[s.charAt(i) - 'a']++;
            }
            for (int i = 0; i < 26; i++) {
                if (result[i] > tmp[i]) {
                    result[i] = tmp[i];
                }
            
}
        }
        List<String> r = new ArrayList<>();
        for (int i = 0; i < 26; i++) {
            for (int j = 0; j < result[i]; j++) {
                
r.add("" + (char)(i+'a'));
            }
        }
        return r;
    }
}
