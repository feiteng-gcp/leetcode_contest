class Solution {
    public int mergeStones(int[] st, int k) {
        int n = st.length;
        
        int[][][] dp = new int[n][n][k + 1];
        
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < n; j++)
            {
                for(int l = 0; l <= k; l++)
                {
                    dp[i][j][l] = -1;
                }
            }
        }
        
        int[][] sum = new int[n][n];
        
        
for(int i = 0; i < n; i++)
        {
            for(int j = i; j < n; j++)
            {
                for(int l = i; l<= j; l++)
                {
                    sum[i][j] += st[l];
                
}
            }
        }
        
        for(int i = 0; i < n; i++)
        {
            dp[i][i][1] = 0;
        }
        
        for(int l = 2; l<= n; l++)
        {
            for(int i = 0; i < n - l + 1; i++)
            {
                int j = i + l - 1;
                
                for(int x = i; x < j; x++)
                {
                    for(int a = 1; a <= k; a++)
                    {
                        for(int b = 1; b <= k; b++)
                        {
                            int v1 = dp[i][x][a];
                            int v2 = dp[x + 1][j][b];
                            if(v1 != -1 && v2 != -1 && a + b<= k)
                            {
                                if(a + b == k)
                                {
                                    if(dp[i][j][1] == -1)
                                    dp[i][j][1] = v1+ v2 + sum[i][j];
                                    else
                                        dp[i][j][1] = Math.min(dp[i][j][1], v1+ v2 + sum[i][j]);
                                }else
                                {
                                    if(dp[i][j][a + b] == -1)
                                    dp[i][j][a + b] = v1+ v2;
                                    else
                                        dp[i][j][a + b] = Math.min(dp[i][j][a +b], v1+ v2);
                                }
                            }
                        }
                    }
                }
            }
        }
        
        return dp[0][n - 1][1];
    }
    
}
