class Solution {
    public int mergeStones(int[] stones, int K) {
        
int[] pre = new int[stones.length + 1];
        for (int i = 0; i < stones.length; i++) {
            pre[i+1] = pre[i] + stones[i];
        }
        int n = stones.length;
        int[][][]dp = new int[40][40][40];
        
for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                for(int k = 1; k <= n; k++) {
                    dp[i][j][k] = 100000000;
                }
            }
        }
        for (int i = 1; i <= n; i++) {
            dp[i][i][1] = 0;
        }
        for (int len = 1; len <= n; len++) {
            for (int i = 1; i + len - 1 <= n; i++) {
                int j = i + len - 1;
                for (int k = 2; k <= len; k++) {
                    for (int t = i; t + 1 <= j; t++) {
                        dp[i][j][k] = Math.min(dp[i][j][k], dp[i][t][k - 1] + dp[t + 1][j][1]);
                    }
                }
                dp[i][j][1] = Math.min(dp[i][j][1], dp[i][j][K] + pre[j] - pre[i - 1]);   
            
}
        }
        if (dp[1][n][1] >= 100000000) {
            return -1;
        }
        return dp[1][n][1];
    }
}
