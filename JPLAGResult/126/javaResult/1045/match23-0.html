class Solution {
    public boolean isValid(String S) {
        Stack<Integer> stack = new Stack<>();
        for(int i = 0; i < S.length(); i++){
            char sChar = S.charAt(i);
            if(sChar == 'a'){
                stack.add(i);
            }else if(sChar == 'b'){
                if(stack.isEmpty() || S.charAt(stack.peek()) != 'a')
                    return false;
                stack.add(i);
            }else{
                if(stack.isEmpty() || S.charAt(stack.pop()) != 'b')
                    return false;
                if(stack.isEmpty() || S.charAt(stack.pop()) != 'a')
                    return false;
            }
        }
        return stack.isEmpty();
    }
}
