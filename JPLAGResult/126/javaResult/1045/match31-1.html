class Solution {
    public boolean isValid(String S) {
        if (S == null || S.equals("")) return false;
        int n = S.length();
        if (n % 3 != 0 || S.charAt(0) != 'a' || S.charAt(n - 1) != 'c') return false;
        String next = S.replaceAll("abc", "");
        if (next.equals("")) return true;
        return isValid(next);
    }
}
