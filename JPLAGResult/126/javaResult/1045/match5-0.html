class Solution {
    public boolean isValid(String s) {
    
        int l = s.length();
       
        if((l%3)!=0){
            return false;
        }
        
        ArrayDeque<Integer> st = new ArrayDeque<>();
        
        char cs[] = s.toCharArray();
        
        for(char c:cs){
            if(c=='a'){
                st.push(2);
            }else if(c=='b'){
                if(st.isEmpty()||st.peek()!=2){
                    return false;
                }
                st.pop();
                st.push(1);
            }else{
                if(st.isEmpty()||st.peek()!=1){
                    return false;
                }
                st.pop();
            }
            
        }
        return st.isEmpty();
        
        
        
        
    }
    
   
    
}
