class Solution {
    public boolean isValid(String S) {
        Stack<Character> stack = new Stack<>();
        for(int i = 0; i < S.length(); i ++)
        {
            char c = S.charAt(i);
            if(c == 'c')
            {
                if(stack.size() < 2) return false;
                if(stack.pop() != 'b')
                    return false;
                if(stack.pop() != 'a')
                    return false;                  
            }
            else
                stack.push(c);
        }
        return stack.size() == 0;
    }
}
