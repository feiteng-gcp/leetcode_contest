class Solution {
    public boolean isValid(String S) {
        
Stack<Character> stack = new Stack<>();
        for (char c : S.toCharArray()) {
            if (c == 'c') {
                if (stack.size() >= 2) {
                    char c1 = stack.pop();
                    char c2 = stack.pop();
                    if (c2 == 'a' && c1 == 'b') {
                        // do nothing
                    } else {
                        stack.push(c2);
                        stack.push(c1);
                        stack.push(c);
                    }
                } else {
                    stack.push(c);
                }
            
} else {
                stack.push(c);
            }
        }
        return stack.size() == 0;
    }
}
