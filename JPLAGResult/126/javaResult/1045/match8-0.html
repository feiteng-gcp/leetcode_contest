class Solution {
    public boolean isValid(String S) {
        int a = 0;
        int b = 0;
        for (char c : S.toCharArray()) {
            if (c == 'a') {
                a++;
            } else if (c == 'b') {
                if (a == 0) return false;
                a--;
                b++;
            } else {
                if (b == 0) return false;
                b--;
            }
        }
        return a == 0 && b == 0;
    }
}
