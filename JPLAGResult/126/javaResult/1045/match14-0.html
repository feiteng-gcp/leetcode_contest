class Solution {
    public boolean isValid(String S) {
        int len = S.length();
        Deque<Character> mystack = new ArrayDeque<>();
        for (int i = 0; i < len; i++) {
            char c = S.charAt(i);
            if (c == 'c') {
                if (mystack.size() >= 2) {
                    char top1 = mystack.pollLast();
                    char top2 = mystack.pollLast();
                 //   System.out.println(top1 + " " + top2);
                    if (top1 == 'b' && top2 == 'a') {
                        continue;
                    } else {
                        return false;
                    }
                } else {
                    return false;
                }
            } else {
                mystack.offerLast(c);
             //   System.out.println("in" + c);
            }
          //  System.out.println(c);
        }
        return mystack.size() == 0;
    }
}
