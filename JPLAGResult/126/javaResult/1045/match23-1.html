class Solution {
    public boolean isValid(String S) {
        int n=S.length();
        Stack<Character> s=new Stack<>();
        for(int i=0;i<n;i++)
{
            char c=S.charAt(i);
            if(c=='a'){
                s.push(c);
            }
            else if(c=='b'){
                if(s.isEmpty() || s.pop()!='a') return false;
                s.push(c);
            }
            else{
                if(
s.isEmpty() || s.pop()!='b') return false;
            }
        }
        return s.isEmpty();
    }
}
