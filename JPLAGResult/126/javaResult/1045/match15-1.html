class Solution {
   public boolean isValid(String S) {
        int a = 0;
        int b = 0;
        int c = 0;
        for (char aa : S.toCharArray()) {
            if (aa == 'a') {
                a ++;
            } else if (aa == 'b') {
                b ++;
            } else {
                c ++;
            
}
            if (a < b || b < c) {
                return false;
            }
        }
        if (a != b || b != c) {
            return false;
        }
        return true;
    }
}
