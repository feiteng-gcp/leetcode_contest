class Solution {
    public boolean isValid(String S) {
        Stack<Character> stack = new Stack<>();
        for (char c : S.toCharArray()) {
            if (c == 'a' || c == 'b') {
                stack.push(c);
            } else 
{
                if (stack.isEmpty() || stack.peek() != 'b') {
                    return false;
                }
                stack.pop();
                
if (stack.isEmpty() || stack.peek() != 'a') {
                    return false;
                }
                stack.pop();
            }
        }
        return stack.isEmpty();
    }
}
