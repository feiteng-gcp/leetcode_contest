class Solution {
    public boolean isValid(String S) {
        Stack<Character> stack = new Stack<>();
        for(char c : S.toCharArray())
{
            if(c == 'a'){
                stack.push(c);
            }else if(c == 'b'){
                if(stack.isEmpty() || stack.peek() != 'a'){
                    return false;
                }
                stack.pop();
                stack.push(c);
            }else{
                if(stack.isEmpty() || stack.peek() != 'b'){
                    return false;
                }
                stack.pop();
            }
        }
        return stack.size() == 0;
    }
}
