class Solution {
    public boolean isValid(String S) {
        Stack<Character> stack = new Stack<>();
        for (int i = 0; i < S.length(); i++) {
            char curr = S.charAt(i);
            if (curr == 'a') {
                stack.push(curr);
            } else if (stack.isEmpty()) {
                return false;
            } else if (curr == 'b') {
                if (stack.peek() == 'a') {
                    stack.pop();
                }
                stack.push(curr);
            } else if (curr == 'c') {
                if (stack.peek() == 'b') {
                    stack.pop();
                } else {
                    stack.push(curr);
                }
            
}
        }
        
        return stack.isEmpty();
    }
}
