class Solution {
    public boolean isValid(String S) {
        
if(S.length() == 0) return true;
        int i = S.indexOf("abc", -1);
        if(i == -1) return false;
        String left = S.substring(0, i);
        String right = S.substring(i + 3, S.length());
        return isValid(left + right);
    }
}
