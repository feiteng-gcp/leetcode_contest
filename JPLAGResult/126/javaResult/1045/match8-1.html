class Solution {
    public boolean isValid(String S) {
        int counta=0,countb=0;
        int len = S.length();
        if(len % 3 != 0) return false;
        
for(char c:S.toCharArray()){
            if(c == 'a'){
                counta++;
            }
            else if(c == 'b'){
                if(counta == 0) return false;
                countb++;
                counta--;
            }
            else{
                if(countb == 0) return false;
                countb--;
            }
        }
        return  (counta==0 && countb==0);
    }
}
