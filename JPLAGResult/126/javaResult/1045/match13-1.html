class Solution {
    public boolean isValid(String S) {
        if (S == null || S.length() < 3) {
            return false;
        }
        
        Deque<Character> stack = new ArrayDeque<>();
        for (int i = 0; i < S.length(); i++) 
{
            char cur = S.charAt(i);
            if (cur == 'c') {
                if (stack.size() >= 2) {
                    char pre1 = stack.pollFirst();
                    char pre2 = stack.pollFirst();
                    if (pre1 != 'b' || pre2 != 'a') 
{
                        return false;
                    }
                } else {
                    return false;
                }
            } else {
                stack.offerFirst(cur);                
            }
        }
        
        return stack.isEmpty();
    }
}
