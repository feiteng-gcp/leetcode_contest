class Solution {
    public boolean isValid(String S) {
        if(S == null){
            return false;
        }
        
if(S.equals("abc")){
            return true;
        }
        int idx = S.indexOf("abc");
        if(idx < 0){
            return false;
        }
        return isValid(S.substring(0, idx) + S.substring(idx + 3, S.length()));
    }
}
