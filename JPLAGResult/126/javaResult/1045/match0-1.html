class Solution {
    public  boolean isValid(String S) {
        int[] count = new int[3];
        for(char c:S.toCharArray()){
            count[c-'a']++;
            if(count[1]>count[0]||count[2]>count[1])
                return false;
        }
        return count[0]==count[1]&&count[1]==count[2];

    }
}
