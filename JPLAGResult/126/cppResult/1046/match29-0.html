class Solution {
public:
    int longestOnes(vector<int>& A, int K) {
        int start = 0, end = 0;
        int len = A.size();
        int change = 0;
        int max_len = 0;
        while (end < len) {
            if (A[end] == 0) {
                ++change;
            }
            
            while (start < len && change > K) {
                if (A[start] == 0) {
                    --change;
                }
                ++start;
            }
            
            max_len = max(max_len, end - start + 1);
            ++end;
        }
        return max_len;
    }
};
