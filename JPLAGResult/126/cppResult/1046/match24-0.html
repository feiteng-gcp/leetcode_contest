class Solution {
public:
    int longestOnes(vector<int>& A, int K) {
        int pos = 0;
        int cnt = 0;
        int ans = 0;
        for (int i = 0; i < A.size(); i++)
        {
            if (A[i] == 0) cnt++;
            while (cnt > K)
            {
                if (A[pos] == 0) cnt--;
                pos++;
            }
            ans = max(ans, i - pos + 1);
        }
        return ans;
    }
};
