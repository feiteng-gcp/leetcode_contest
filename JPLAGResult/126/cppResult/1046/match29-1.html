class Solution {
public:
    int longestOnes(vector<int>& A, int K) {
        int l = 0, r = 0;
        int cnt = 0;
        int res = 0;

        for (; r < A.size(); ++r) {
            if (A[r] == 0) {
                ++cnt;
            }
            while (cnt > K) {
                if (A[l] == 0) {
                    --cnt;
                }
                ++l;
            }
            res = max(res, r - l + 1);
        }
        return res;
    }
};
