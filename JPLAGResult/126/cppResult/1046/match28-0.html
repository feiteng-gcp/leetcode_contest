class Solution {
public:
    int longestOnes(vector<int>& A, int K) {
        int len = (int)A.size(), b = 0, e = 0, cnt = 0, ans = K;
    while(e < len)
    {
        if(A[e] == 0) cnt++;
        e++;
        while(cnt > K)
        {
            if(A[b] == 0) cnt--;
            b++;
        }
        ans = max(ans, e - b);
    }
    return ans;
    }
};
