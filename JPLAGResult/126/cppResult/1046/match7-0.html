class Solution {
public:
    int longestOnes(vector<int>& A, int K) {
        int fr = 0, cnt = 0, ans = 0;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] == 0)
                cnt++;
            while (cnt > K) {
                if (A[fr] == 0)
                    cnt--;
                fr++;
            }
            ans = max(ans, i + 1 - fr);
        }
        return ans;
    }
};
