class Solution {
public:
    int longestOnes(vector<int>& a, int k) {
        vector<int> zp;
        for(int i=0;i<a.size();i++)
        {
            if(a[i]==0)
            {
                zp.emplace_back(i);
            }
        }
        if(zp.size() == 0 && zp.size()<=k)
        {
            return a.size();
        }
        int b=0;
        int e=a[0];
        int ret=e-b;
        for(int i=0;i+k<zp.size();i++)
        {
            e=zp[i+k];
            int l=e-b;
            if(l>ret)
            {
                ret=l;
            }
            b=zp[i]+1;
        }
        e=a.size();
        if(e-b>ret)
            ret=e-b;
        return ret;
    }
};
