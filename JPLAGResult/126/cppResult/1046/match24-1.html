class Solution {
public:
    int longestOnes(vector<int>& a, int k) {
        int cnt=0;
        int l=0;
        int ret=0;
        for(int i=0;i<a.size();i++){
            if(a[i]==0)cnt++;
            while(l<=i&&cnt>k){
                if(a[l]==0)cnt--;
                l++;
            }
            ret=max(ret,i-l+1);
        }
        return ret;
    }
};
