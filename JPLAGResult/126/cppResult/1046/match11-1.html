class Solution {
public:
  int longestOnes(vector<int>& A, int K) {
    int res = 0, left = 0;
    queue<int> q;
    for (int right = 0; right < A.size(); ++right) {
      if (A[right] == 0)
        q.push(right);
      if (q.size() > K) {
        left = q.front() + 1;
        q.pop();
      }
      res = max(res, right - left + 1);
    }
    return res;
  }
};
