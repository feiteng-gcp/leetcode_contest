class Solution {
public:
    int longestOnes(vector<int>& A, int K) {
        int n = A.size(), L = 0, count = 0, max_len = 0;
        for(int R = 0; R < n; R++)
        {
            if(A[R] == 0)
                count++;
            while(count > K)
            {
                if(A[L] == 0)
                    count--;
                L++;
            }
            max_len = max(max_len, R - L + 1);
        }
        return max_len;
    }
};
