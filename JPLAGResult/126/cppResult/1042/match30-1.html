class Solution {
public:

    int f[100][100];
    int k;

    inline bool valid(int n) {
        return (n - 1) % (k - 1) == 0;
    }

    inline int getTimes(int n) {
        return (n - 1) / (k - 1);
    }

    int prefix[100];

    inline int sum(int l, int r) {
        return prefix[r + 1] - prefix[l];
    }

    int INF = 0x3f3f3f3f;

    int buf[100];
    int g[100][100];
    vector<int> a;

    inline int getMinimum(int l, int r) {

        memset(g, 0x3f, sizeof g);
        g[0][l] = 0;
        for (int i = 1; i <= k; ++i) {
            for (int j = l; j < r; ++j) {
                g[i][j + 1] = INF;
                for (int p = j; p >= l; --p) {
                    if (j == p) {
                        if (g[i - 1][p] == INF) continue;
                        g[i][j + 1] = min(g[i][j + 1], g[i - 1][p] + a[j]);
                    } else {
                        if (g[i - 1][p] == INF || f[p][j + 1] == INF) continue;
                        g[i][j + 1] = min(g[i][j + 1], g[i - 1][p] + f[p][j + 1] + sum(p, j));
                    }
                }
            }
        }
//        printf("min %d %d %d\n", l, r, g[k][r - 1]);
        return g[k][r];
    }

    int mergeStones(vector<int>& stones, int K) {
        k = K;
        if (!valid(stones.size())) return -1;
        int n = stones.size();
        a = stones;
        
        if (n == 1) return 0;

        prefix[0] = 0;
        for (int i = 0; i < n; ++i)
            prefix[i + 1] = prefix[i] + stones[i];

        memset(g, 0x3f, sizeof g);
        memset(buf, 0x3f, sizeof buf);
        memset(f, 0x3f, sizeof f);


        for (int len = K; len <= n; len += K - 1) {
            for (int l = 0; l + len <= n; ++l) {
                int r = l + len;
                f[l][r] = getMinimum(l, r);
//                printf("%d %d %d\n", l, r, f[l][r]);
            }
        }
        return f[0][n];
    }
};
