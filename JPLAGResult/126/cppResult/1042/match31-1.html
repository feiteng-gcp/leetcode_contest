class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        int a[33] = {0};
        int s[33] = {0};
        for (int i = 1; i <= n; ++ i)
        {
            a[i] = stones[i-1];
            s[i] = s[i-1]+a[i];
        }
        const int INF = 1000000000;
        int F[33][33], G[33][33];
        for (int i = 0; i < 33; ++ i)
            for (int j = 0; j < 33; ++ j)
            {
                F[i][j] = G[i][j] = 0;
            }
        for (int i = 1; i < 33; ++ i)
            for (int j = i+1; j <= 32; ++ j)
            {
                F[i][j] = INF;
                G[i][j] = INF;
            }
        for (int len = 2; len <= n; ++ len)
        {
            for (int i = 1; i <= n; ++ i)
            {
                int j = i+len-1;
                if (j > n) continue;
                for (int x = i-1; x <= j; ++ x)

                    for (int k = 0; k <= K; ++ k)
                        G[x][k] = INF;
                G[i-1][0] = 0;
                for (int x = i; x <= j; ++ x)
                    for (int k = 1; k <= K; ++ k)
                    {
						G[x][k] = min(G[x][k], G[x-1][k-1]);
                        for (int y = i; y <= x; ++ y)
                            G[x][k] = min(G[x][k], G[y-1][k-1]+F[y][x]);
                    }
                F[i][j] = min(F[i][j], G[j][K]+s[j]-s[i-1]);
            }
        }
        return F[1][n] == INF ? -1 : F[1][n];
    }
};
