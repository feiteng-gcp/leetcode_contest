class Solution {
public:
    int mergeStones(vector<int>& stones, int K)
{
    int len = (int)stones.size();
    if((len - 1) % (K - 1)) return -1;
    vector<int> preS(len + 1, 0);
    for(int i = 0; i < len; i++) preS[i + 1] = preS[i] + stones[i];
    vector<vector<int> > dp(len + 1, vector<int>(len, 0));
    for(int i = 0; i + K <= len; i++)
        dp[K][i] = preS[i + K] - preS[i];
    for(int l = K + 1; l <= len; l++)
        for(int i = 0; i + l <= len; i++)
        {
            dp[l][i] = 10000;
            for(int k = 1; k < l; k += K - 1)
                dp[l][i] = min(dp[l][i], dp[k][i] + dp[l - k][i + k]);
            if((l - 1) % (K - 1) == 0) dp[l][i] += preS[i + l] - preS[i];
        }
    return dp[len][0];
}
};
