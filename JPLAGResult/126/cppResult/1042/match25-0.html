class Solution {
    void DFS(vector<vector<int>> &DP,
             vector<vector<int>> &Sum,
            int start,int end,int K,
             int i, int step, int Cost){
        if(step == K){
            if(i == end){
               if(DP[start][end] == -1 ||
                   Cost < DP[start][end]){
                    DP[start][end] = Cost;
                } 
            }
            return;
        }
        if((end-start-1) % (K-1) != 0)
            return;
        for(int j = i + 1 ; j <= end ; j += K-1){
            DFS(DP,Sum,
               start,end,K,
               j,step+1,Cost+DP[i][j]);
        }
    }
public:
    int mergeStones(vector<int>& stones, int K) {
        int N = stones.size();
        //cannot merge
        if((N-1) % (K-1) != 0)
            return -1;
        
        vector<vector<int>> Sum(N+1,vector<int>(N+1,0));
        for(int i = N - 1 ; i >= 0 ; --i){
            for(int j = i + 1 ; j <= N ; j ++){
                Sum[i][j] = stones[i] + Sum[i+1][j];
            }
        }
        
        vector<vector<int>> DP(N+1,vector<int>(N+1,-1));
        for(int i = N - 1 ; i >= 0 ; --i){
            for(int j = i + 1 ; j <= N ; j += K-1){
                //1.No merge is needed
                if(j == i + 1){
                    DP[i][j] = 0;
                    continue;
                }
                //2.one way to merge
                if(j == i + K){
                    DP[i][j] = Sum[i][j];
                    continue;
                }
                //3.merge K segments into one if possible
                DFS(DP,Sum,
                   i,j,K,
                    i,0,0);
                DP[i][j] += Sum[i][j];
            }
        }
        return DP[0][N];
    }
};
