class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        if (n == 1) return 0;
        if (n < K || (n - K) % (K - 1) != 0) return -1;
        
        vector<int> sum(n+1);
        for (int i = 0; i < stones.size(); i++) sum[i+1] = sum[i] + stones[i];
        
        vector<vector<vector<int>>> f(n + 1, vector<vector<int>>(n + 1, vector<int>(K+1, -1)));
        for (int i = 0; i < n; i++) f[i][i+1][1] = 0;
        
        for (int i = 2; i <= n; i++)

            for (int j = 0; j + i <= n; j++) {
                for (int k = 2; k <= K; k++) {
                    for (int l = j + 1; l + k - 1 <= i + j; l++) {
                        if (f[j][l][1] != -1 && f[l][j+i][k-1] != -1) {
                            if (f[j][i+j][k] == -1)
                                f[j][i+j][k] = f[j][l][1] + f[l][j+i][k-1];
                            else
                                f[j][i+j][k] = min(f[j][i+j][k], f[j][l][1] + f[l][j+i][k-1]);
                        }
                    }
                }
                if (f[j][i+j][K] != -1)
                    f[j][i+j][1] = f[j][i+j][K] + sum[i+j] - sum[j];
            }
            
        
        return f[0][n][1];
    }
};
