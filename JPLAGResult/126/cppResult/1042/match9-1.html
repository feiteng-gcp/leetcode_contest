class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        if(stones.size()==1)
        {
            return 0;
        }
       int n = stones.size();
        
        vector<int> sums(n+1, 0);
        for(int i=1; i<=n; i++)
        {
            sums[i] = sums[i-1] + stones[i-1]; 
        }
        
        vector<vector<vector<int>>> dp(n+1, vector<vector<int>>(n+1, vector<int>(n+1, INT_MAX/2)));
        
        
        for(int i=1; i<=n; i++)
        {

            dp[i][i][1] = 0;
        }
        
        for (int len = 1; len <= n; len++) 
        {
            for (int i = 1; i + len - 1 <= n; i++) 
            {
                int j = i + len - 1;
                for (int k = 2; k <= len; k++) 
                {
                    for (int t = i; t + 1 <= j; t++) 
                    {
                        dp[i][j][k] = min(dp[i][j][k], dp[i][t][k - 1] + dp[t + 1][j][1]);
                    }
                }
                
                dp[i][j][1] = min(dp[i][j][1], dp[i][j][K] + sums[j] - sums[i - 1]);   
            }
        }
        
        if (dp[1][n][1] >= INT_MAX/2) {
            return -1;
        }
        
        return dp[1][n][1];
    }
};
