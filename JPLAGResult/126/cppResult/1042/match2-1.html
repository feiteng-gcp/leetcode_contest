class Solution {
public:
    void chmin(int& a, int b){
        a = min(a, b);
    }

    int mergeStones(vector<int>& stones, int K) {
        int N = stones.size();
        if((N-1) % (K-1)) return -1;
        int M = (N-1) / (K-1);

        int dp[31][31];
        int dp2[31][31];


        for(int i=0; i<N; i++) for(int j=0; j<=M; j++) dp[i][j] = 1e9;
        for(int i=0; i<N; i++) dp[i][0] = 0;

        for(int m=1; m<=M; m++){
            int diff = m * (K-1) + 1;
            for(int l=0; l+diff<=N; l++){
                int r = l+diff;
                int sum = 0;

                for(int x=l; x<r; x++) sum += stones[x];

                for(int x=l; x<=r; x++) for(int k=0; k<=K; k++) dp2[x][k] = 1e9;
                dp2[l][0] = sum;
                for(int x=l; x<r; x++){
                    for(int k1=0; k1<K; k1++){
                        for(int k2=0; k2<m; k2++){
                            int adv = k2 * (K-1) + 1;
                            if(x+adv > r) continue;
                            chmin(dp2[x+adv][k1+1], dp2[x][k1] + dp[x][k2]);
                        }
                    }
                }
                dp[l][m] = dp2[r][K];
            }
        }

        return dp[0][M];
    }
};
