class Solution {
public:
    #define LL long long
    
    void dfs(int r, int last, int tot, int max_num, vector<int> &cur, vector<vector<int>> &split, vector<vector<int>> &f) {
        if (last > r)
            return;
        if (tot == max_num && f[last][r] != INT_MAX) {
            cur.push_back(r);
            split.push_back(cur);
            cur.pop_back();
            return;
        }
        
        for (int i = last; i <= r; i++) 
            if (f[last][i] != INT_MAX) {
                cur.push_back(i);
                dfs(r, i + 1, tot + 1, max_num, cur, split, f);
                cur.pop_back();
            }
    }
    
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        vector<vector<int>> f(n, vector<int>(n, INT_MAX));
        
        vector<int> sum(n, 0);
        sum[0] = stones[0];
        
        for (int i = 1; i < n; i++)
            sum[i] = sum[i - 1] + stones[i];
        
        for (int i = 0; i < n; i++)
            f[i][i] = 0;
        
        for (int len = K; len <= n; len++)
            for (int i = 0; i < n - len + 1; i++) {
                int j = i + len - 1;
                vector<vector<int>> split;
                vector<int> cur;
                dfs(j, i, 0, K - 1, cur, split, f);
                int mint = INT_MAX;
                for (auto &sp : split) {
                    int tot = 0;
                    for (int k = 0; k < K; k++) {
                        if (k == 0)
                            tot += f[i][sp[k]];
                        else
                            tot += f[sp[k - 1] + 1][sp[k]];
                    }
                    mint = min(mint, tot);
                }
                if (mint != INT_MAX) {
                    if (i == 0)
                        f[i][j] = mint + sum[j];
                    else
                        f[i][j] = mint + sum[j] - sum[i - 1];
                }
            }
        
        if (f[0][n - 1] == INT_MAX)
            f[0][n - 1] = -1;
        
        return int(f[0][n - 1]);
    }
};
