class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        int pre[33];
        pre[0] = 0;
        for(int i=1; i<=n; i++)
            pre[i] = pre[i-1]+stones[i-1];

        int dp[33][33][33];
        for(int i=1; i<=n; i++)
            for(int j=1; j<=n; j++)
                for(int k=1; k<=n; k++)
                    dp[i][j][k] = 1000000;
        for(int i=1; i<=n; i++)
            dp[i][i][1] = 0;            
        for(int L=1; L<=n; L++)
            for(int i=1; i<=n; i++) {
                int j = i+L-1;
                if(j>n)
                    break;
                for(int k=2; k<=L; k++)
                    for(int cut=i; cut<j; cut++)
                        dp[i][j][k] = min(dp[i][j][k], dp[i][cut][k-1] + dp[cut+1][j][1]);
                dp[i][j][1] = min(dp[i][j][1], dp[i][j][K]+pre[j]-pre[i-1]);   
            }
        if(dp[1][n][1] == 1000000)
            return -1;
        return dp[1][n][1];
    }
};
