class Solution {
public:
    const int INFTY = 0x3ffffff;
    int mergeStones(vector<int>& stones, int K) {
        int N = stones.size();
        int f[35][35][35];
        int s[35];
        s[0] = stones[0];

        for (int i = 1; i < N; ++i)
            s[i] = s[i - 1] + stones[i];
        for (int i = 0; i < N; ++i) {
            f[i][i][1] = 0;
            for (int k = 2; k <= K; ++k)
                f[i][i][k] = INFTY;
        }
        for (int len = 2; len <= N; ++len) {
            for (int i = 0, j = len - 1; j <= N - 1; ++i, ++j) {
                for (int k = 2; k <= K; ++k) {
                    f[i][j][k] = INFTY;
                    for (int t = i; t < j; ++t)
                        f[i][j][k] = min(f[i][j][k], f[i][t][k - 1] + f[t + 1][j][1]);
                }
                f[i][j][1] = f[i][j][K] + s[j] - (i > 0 ? s[i - 1] : 0);
                if (f[i][j][1] > INFTY) f[i][j][1] = INFTY;
            }
        }
        return f[0][N - 1][1] >= INFTY ? -1 : f[0][N - 1][1];
    }
};
