class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        if (K > 2 && (n % (K - 1)) != 1) return -1;
        
        sum[0] = 0;
        for (int i = 1; i <= n; i++) sum[i] = sum[i - 1] + stones[i - 1];

        for (int i = 1; i <= n; i++)
            for (int j = i; j <= n; j++)
                for (int k = 1; k <= K; k++) f[i][j][k] = inf;
        for (int i = 1; i <= n; i++) f[i][i][1] = 0;
            for (int j = 1; j < n; j++)
                for (int i = 1; i + j <= n; i++)
                {
                    for (int k = 1; k < K; k++)
                        for (int mid = i; mid < i + j; mid++)
                        {
                            if (f[i][mid][k] == inf) continue;
                            if (f[mid + 1][i + j][1] == inf) continue;
                            f[i][i + j][k + 1] = min(f[i][i + j][k + 1], f[i][mid][k] + f[mid + 1][i + j][1]);
                        }
                    if (f[i][i + j][K] == inf) continue;
                    f[i][i + j][1] = min(f[i][i + j][K] + sum[i + j] - sum[i - 1], f[i][i + j][1]);
                }
        return f[1][n][1];
    }
    
private:
    const int inf = 1000000007;
    int sum[32];
    int f[32][32][32];
};
