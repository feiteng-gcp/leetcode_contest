class Solution {
public:
    vector<vector<int>> dp;
    int mergeStones(vector<int>& s, int K) {
        vector<vector<int>> dp(s.size() + 1, vector<int>(s.size() + 1, INT_MAX / 3));
        vector<vector<int>> sum(s.size() + 1, vector<int>(s.size() + 1));
        for (int i = 0; i < s.size(); i++)
            for (int j = i + 1; j <= s.size(); j++)
                for (int k = i; k < j; k++)
                    sum[i][j] += s[k];
        for (int i = 0; i < s.size(); i++)
            dp[i][i + 1] = 0;
        for (int i = K; i <= s.size(); i += K - 1)
            for (int j = 0; j + i <= s.size(); j++) {
                vector<vector<int>>dpp(s.size() + 1, vector<int>(K + 1, INT_MAX / 3));
                dpp[j][0] = 0;
                for (int k = j + 1; k <= j + i; k++)
                    for (int l = j; l < k; l++)
                        for (int m = 0; m < K; m++)
                            dpp[k][m + 1] = min(dpp[k][m + 1], dpp[l][m] + dp[l][k]);
                dp[j][j + i] = dpp[j + i][K] + sum[j][j + i];
                // printf("%d %d  %d %d\n", j, j + i, dp[j][j + i], sum[j][j + i]);
            }
        return dp[0][s.size()] == INT_MAX / 3 ? -1 : dp[0][s.size()];
    }
};
