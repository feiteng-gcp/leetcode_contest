class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        if ((n - 1) % (K - 1) != 0)
            return -1;
        vector<vector<int>> f(n + 2, vector<int>(n + 2));
        vector<int> sum(n + 1);
        for (int i = 0; i < n; ++i)
            sum[i + 1] = sum[i] + stones[i];
        for (int i = n; i >= 1; --i) {
            f[i][i] = 0;
            for (int j = i + 1; j <= n; ++j) {
                f[i][j] = 1e9;
                for (int k = i; k < j; k += K - 1)
                    f[i][j] = min(f[i][j], f[i][k] + f[k + 1][j]);
                if ((j - i) % (K - 1) == 0)
                    f[i][j] += sum[j] - sum[i - 1];
                //printf("%d %d %d\n", i, j, f[i][j]);
            }
            //printf("\n");
        }
        return f[1][n];
    }
};
