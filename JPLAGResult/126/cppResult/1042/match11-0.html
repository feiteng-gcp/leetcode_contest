class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        if (K == 1) {
            return -1;
        }
        
        const int INF = INT_MAX / 5;
        
        
        vector<int> sums(n + 1, 0);
        for (int i = 0; i < n; ++i) {
            sums[i + 1] = sums[i];
            sums[i + 1] += stones[i];
        }
        
        // vector<vector<int>> dp(n + 1, vector<int>(n + 1, INF));
        vector<vector<vector<int>>> dp(n + 1, vector<vector<int>>(n + 1, vector<int>(K + 1, INF)));

        for (int i = 0; i < n; ++i) {
            dp[i + 1][i + 1][1] = 0;
        }
        
        for (int l = 1; l <= n; ++l) {
            for (int i = 0; i + l <= n; ++i) {
                int j = i + l - 1;
                
                for (int k = 1; k < K - 1; ++k) {
                    int& ans = dp[i + 1][j + 1][k + 1];
                    for (int a = i; a < j; ++a) {
                        ans = min(ans, dp[i + 1][a + 1][k] + dp[a + 1 + 1][j + 1][1]);
                    }
                    // cout << i << " " << j << " " << k + 1 << " " << ans << endl;
                }
                
                for (int a = i; a < j; ++a) {
                    int& ans = dp[i + 1][j + 1][1];
                    int sum = sums[j + 1] - sums[i];
                    ans = min(ans, dp[i + 1][a + 1][K - 1] + dp[a + 1 + 1][j + 1][1] + sum);
                    // cout << i << " " << j << " " << 0 << " : " << ans << endl;
                }
            }
        }
        
        int res = dp[1][n][1];
        if (res == INF) {
            return -1;
        }
        return res;
        
    }
    

    
private:
    const int INF = INT_MAX / 5;
};
