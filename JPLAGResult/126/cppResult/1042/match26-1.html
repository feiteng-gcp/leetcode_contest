class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        if (K != 2 && n % (K - 1) != 1)
            return -1;
        
        vector<vector<int>> min_cost(n+1, vector<int>(n+1, 30 * 30 * 100));
        for (int i = 0; i < n; i++)
            min_cost[i][i+1] = 0;

        vector<vector<int>> range_sums(n+1, vector<int>(n+1, 0));

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j <= n; j++) {
                for (int k = i; k < j; k++) {
                    range_sums[i][j] += stones[k];
                }
            }
        }
        

        for (int size = K; size <= n; size += K - 1) {
            int m = (size - K) / (K - 1); // number of blobs
            for (int i = 0; i + size <= n; i++) {
                vector<int> sticks(K - 1 + m, 1);
                for (int j = 0; j < K - 1; j++)
                    sticks[j] = 0;
                do {
                    int cost = 0;
                    int left = i, right = i + 1;
                    for (int j = 0; j < K - 1 + m; j++) {
                        if (sticks[j] == 0) {
                            cost += min_cost[left][right] + range_sums[left][right];
                            left = right;
                            right = left + 1;
                        } else {
                            right += K - 1;
                        }
                    }
                    cost += min_cost[left][right] + range_sums[left][right];
                    assert(right == i + size);
                    min_cost[i][i + size] = min(min_cost[i][i + size], cost);
                } while (next_permutation(sticks.begin(), sticks.end()));
                
                //cout << i << " " << i + size << " " << min_cost[i][i + size] << endl;
            }
        }
        return min_cost[0][n];
    }
};
