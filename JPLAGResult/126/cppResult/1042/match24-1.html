class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int dp[35][35][35] = {0};
        int sum[35] = {0};
        int n = stones.size();
        memset(dp, 1e9 + 7, sizeof(dp));
        for (int i = 1; i <= n; i++) {
            dp[i][i][1] = 0;
            sum[i] = sum[i - 1] + stones[i - 1];
        }
        for (int i = 1; i + K <= n; i++) {
            int j = i + K - 1;
            dp[i][j][K] = 0;
            dp[i][j][1] = sum[j] - sum[i - 1];
        }

        for (int x = 1; x < n; x++) {
            for(int i = 1; i + x <= n; i++)
            {
                int j = x + i;

                for(int k = i; k < j; k++)
                {
                    for(int c = 2; c <= x + 1; c++)
                    {
                        dp[i][j][c] = min(dp[i][j][c], dp[i][k][c-1] + dp[k+1][j][1]);
                    }
                }
                dp[i][j][1] = min(dp[i][j][1], dp[i][j][K] + sum[j]- sum[i-1]);
            }
        }
        int p = n - K;
        if (p % (K - 1) != 0) return -1;
        else return dp[1][n][1];
    }
};
