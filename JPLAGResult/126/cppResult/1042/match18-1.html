typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
  ll dp[33][33][33];
  int mergeStones(vector<int>& v, int K) {
    int N=v.size();
    if(K!=2 && N%(K-1)!=1) return -1;
    if(N==1) return 0;
    for(int i=0;i<N;i++){

      for(int j=0;j<N;j++){
        for(int k=0;k<N;k++){
          dp[i][j][k]=1e18;
        }
      }
      dp[i][i][1]=0;
    }

    for(int s=2;s<=N;s++){
      for(int i=0;i+s-1<N;i++){
        int j=i+s-1;
         
        int cur=0;
        for(int jj=i;jj<=j;jj++) cur+=v[jj];

        for(int k=2;k<=K;k++){
          for(int jj=i;jj<=j-1;jj++){
            dp[i][j][k]=min(dp[i][j][k],dp[i][jj][k-1]+dp[jj+1][j][1]);
          }
        }
        dp[i][j][1]=min(dp[i][j][1],dp[i][j][K]+cur);
      }
    }
    return dp[0][N-1][1];
  }
};
