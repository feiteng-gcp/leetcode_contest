class Solution {
public:
    int mergeStones(vector<int>& stones, int K) {
        int n = stones.size();
        if(n < 2)   return 0;
        vector<vector<vector<int>>> dp(n,vector<vector<int>>(n,vector<int>(n+1, 1000000000)));
        vector<vector<int>> sum(n,vector<int>(n,0));
        for(int i = 0; i < n; i++) {
            sum[i][i] = stones[i];
            for(int j = i + 1; j < n; j++) {
                sum[i][j] = sum[i][j-1] + stones[j];
            }
        }
        for (int i = 0; i < n; i++) {
            dp[i][i][1] = 0;
        }
        for(int length = 2; length <= n; length++) {
            for(int i = 0; i < n - length + 1; i++) {
                
                int j = i + length - 1;
                
                for(int k = 2; k <= length; k++) {
                    int temp = INT_MAX;
                    for(int mid = i; mid < i + length -1; mid++) {
                        temp = min(temp, dp[i][mid][k-1] + dp[mid+1][j][1]);
                        dp[i][j][k] = temp;
                    }
                }
                dp[i][j][1] = INT_MAX;

                dp[i][j][1] = min(dp[i][j][1], dp[i][j][K] + sum[i][j]);
            }
        }
        if(dp[0][n-1][1] >= 1000000000) return -1;
        return dp[0][n-1][1];

    }
};
