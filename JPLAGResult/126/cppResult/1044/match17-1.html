int c[111][111];

class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        memset(c,0,sizeof(c));
        int n=A.size();
        for(int i=0;i<n;i++)
        {
            string s=A[i];
            for(int j=0;j<s.length();j++)

            {
                c[i][A[i][j]-'a']++;
            }
        }
        vector<string> ans;
        ans.clear();
        for(int i=0;i<26;i++)
        {
            int mn=c[0][i];
            for(int j=0;j<n;j++)
            {
                mn=min(mn,c[j][i]);
            }
            char c='a'+i;

            string cur="";
            cur=cur+c;
            for(int j=0;j<mn;j++)
            {
                ans.push_back(cur);
            }
        }
        return ans;
    }
};
