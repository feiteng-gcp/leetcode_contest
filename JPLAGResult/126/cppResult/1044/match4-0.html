class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<int> h(128, INT_MAX);
        
        for (string & s : A) {
            vector<int> h2(128, 0);
            for (char ch : s) {
                h2[ch] += 1;
            }
            for (int i = 0; i < h.size(); ++i) {
                h[i] = std::min(h[i], h2[i]);
            }
        }
        
        vector<string> result;
        for (int i = 0; i < h.size(); ++i) {
            for (int j = 0; j < h[i]; ++j) {
                string s(1, (char)i);
                result.push_back(s);
            }
        }
        
        return result;
    }
};
