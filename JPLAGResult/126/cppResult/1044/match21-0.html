class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int n = A.size();
        if(n == 1)
            return A;
        vector<int> chars(26, 0);
        for(int i = 0; i < A[0].size(); i++)
        {
            chars[A[0][i] - 'a']++;
        }
        
        for(int i = 1; i < n; i++)
        {
            string cur = A[i];
            vector<int> cchars(26, 0);
            for(int j = 0; j < cur.size(); j++)
            {
                cchars[cur[j]-'a']++;
            }
            for(int j = 0; j < 26; j++)
            {
                chars[j] = min(chars[j], cchars[j]);
            }
        }
        vector<string> ans;
        for(int i = 0; i < 26; i++)
        {
            int k = chars[i];
            for(int j = 0; j < k; j++)
            {
                char c = 'a' + i;
                string astr;
                astr += c;
                ans.push_back(astr);
            }
        }
        return ans;
    }
};
