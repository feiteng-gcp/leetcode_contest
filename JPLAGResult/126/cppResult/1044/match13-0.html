class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int n = A.size();
        int cnt[n][26];
        memset(cnt, 0, sizeof(cnt));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < A[i].size(); j++) {
                cnt[i][A[i][j]-'a']++;
            }
        }
        vector<string> ret;
        string add = "a";
        for (int i = 0; i < 26; i++) {
            int t = 1e9;
            for (int j = 0; j < n; j++) {
                t = min(t, cnt[j][i]);
            }
            while (t--) {
                ret.push_back(add);
            }
            add[0]++;
        }
        return ret;
    }
};
