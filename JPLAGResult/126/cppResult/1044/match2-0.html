class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<string> res;
        if (A.size() == 0) return res;
        vector<int> tags(26, 0);
        for (int i = 0; i < A[0].size(); ++i) {
            tags[A[0][i] - 'a']++;
        }
        for (int i = 1; i < A.size(); ++i) {
            vector<int> tmp(26, 0);
            for (int j = 0; j < A[i].size(); ++j) {
                tmp[A[i][j] - 'a']++;
            }
            for (int k = 0; k < 26; ++k) {
                tags[k] = min(tags[k], tmp[k]);
            }
        }
        for (int i = 0; i < 26; ++i) {
            while (tags[i]--) {
                res.push_back(string(1, 'a' + i));
            }
        }
        return res;
    }
};
