class Solution {
public:
    vector<string> commonChars(vector<string>& strings) {
        const int n = strings.size();
        vector<int> m1(256, 0);
        vector<int> m2(256, INT_MAX);

        for (int i = 0; i < n; ++i) {
            vector<int> count(256, 0);            
            for (auto& ch : strings[i]) {
                //cout << "here" << endl;
                ++count[ch];
            }
            for (int j = 0; j < 256; ++j) {
                if (count[j] > 0) {
                    ++m1[j];
                    m2[j] = min(m2[j], count[j]);
                }

            }
        }
        
        vector<string> res;
        for (int i = 0; i < 256; ++i) {
            if (m1[i] == n) {
                for (int j = 0; j < m2[i]; ++j) {
                    res.emplace_back(string(1, i));
                }
            }
        }
        return res;
    }
};
