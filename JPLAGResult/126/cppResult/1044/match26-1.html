class Solution {
public:
    vector<string> commonChars(vector<string>& vs) {
        vector<int> ans(30, 9999999);
        

        for (string s: vs) {
            vector<int> cnt(30, 0);
            for (char c: s)
                cnt[c - 'a']++;
            for (int i = 0; i < 30; i++)
                ans[i] = min(ans[i], cnt[i]);
        }
        
        vector<string> s;
        for (int i = 0; i < 30; i++)
            while (ans[i]--)
                s.push_back(string(1, i + 'a'));
        return s;
    }
};
