class Solution {
public:
vector<string> commonChars(vector<string>& a) {
    int n = a.size();
    vector<vector<int>> s(n, vector<int>(26, 0));
    for (int i = 0; i < n; i++) {

    	for (char c : a[i]) {
    		s[i][c - 'a' ]++;
    	}
    }
    vector<string> res;
    for (int i = 0; i < 26; i++) {
    	int sa = n;
    	for (int j = 0; j < n; j++) {
    		sa = min(sa, s[j][i]);
    	}
        //cout << i << " " << sa << endl;
    	while(sa > 0) {
    		res.push_back(string(1, char('a' + i)));
            sa--;
    	}
    }
    return res;
}
};
