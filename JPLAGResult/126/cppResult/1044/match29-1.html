class Solution {
public:
    vector<string> commonChars(vector<string>& a) {
        if (a.empty()) return {};
        int n = a.size();

        vector<vector<int>> v(n, vector<int>(26, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < a[i].size(); j++) {
                v[i][a[i][j] - 'a']++;
            }
        }
        vector<string> ret;
        vector<int> u(26, 0);
        for (int j = 0; j < 26; j++) {
            u[j] = v[0][j];
        }
        for (int i = 1; i < n; i++) {

            for (int j = 0; j < 26; j++) {
                u[j] = min(u[j], v[i][j]);
            }
        }
        for (int i = 0; i < 26; i++) {
            if (u[i] > 0) {
                for (int j = 0; j < u[i]; j++) {
                    ret.push_back(string(1, (char)('a' + i)));
                }
            }
        }
        return ret;
    }
};
