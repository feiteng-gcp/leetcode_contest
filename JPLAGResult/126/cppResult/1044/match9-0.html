class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<string> ans;
        for (char ch='a';ch<='z';ch++)
        {
            int minn=2100000000;
            for (int i=0;i<A.size();i++)
            {
                int cnt=0;
                for (int j=0;j<A[i].length();j++) if (A[i][j]==ch) cnt++;
                minn=min(minn,cnt);
            }
            string ls=" ";ls[0]=ch;
            for (int i=1;i<=minn;i++) ans.push_back(ls);
        }
        return ans;
    }
};
