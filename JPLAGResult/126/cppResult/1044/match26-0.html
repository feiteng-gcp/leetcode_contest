class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<int> cnt(26, 0);
        
        for (char c : A[0])
            cnt[c-'a']++;
        
        for (string &word : A)
        {
            vector<int> tmp(26, 0);
            for (char c : word)
                tmp[c-'a']++;
            
            for (int i = 0; i < 26; i++)
                cnt[i] = min(cnt[i], tmp[i]);
        }
        
        vector<string> ans;
        for (int i = 0; i < 26; i++)
            while (cnt[i]-- > 0)
            {
                string s;
                s += (char)('a' + i);
                ans.push_back(s);
            }
        
        return ans;
    }
};
