class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int n = A.size();
        vector<vector<int>> cnt(n, vector<int>(26, 0));
        for (int i = 0; i < n; ++i) {
            int l = A[i].length();
            for (int j = 0; j < l; ++j)
                ++cnt[i][A[i][j] - 'a'];
        }
        vector<string> ans;

        string s = "a";
        for (int i = 0; i < 26; ++i) {
            int x = cnt[0][i];
            for (int j = 1; j < n; ++j)
                x = min(x, cnt[j][i]);
            s[0] = 'a' + i;
            while (x--)
                ans.push_back(string(s));
        }
        return ans;
    }
};
