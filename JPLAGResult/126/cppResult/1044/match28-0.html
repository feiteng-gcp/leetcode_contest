class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int cnt[26];
        for (int i = 0; i < 26; ++i) cnt[i] = 100;
        for (int i = 0; i < A.size(); ++i) {
            int w[26];
            memset(w, 0, sizeof(w));
            for (int j = 0; j < A[i].length(); ++j) {
                w[A[i][j]-'a']++;
            }
            for (int j = 0; j < 26; ++j) {
                if (cnt[j] > w[j]) cnt[j] = w[j];
            }
        }
        vector<string> res;
        for (int i = 0; i < 26; ++i) {
            for (int j = 0; j < cnt[i]; ++j) {
                res.push_back(string(1,i+'a'));
            }
        }
        return res;
    }
};
