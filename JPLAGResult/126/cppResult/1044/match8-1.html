class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int n=A.size();
        vector<string> ret;

        for(char ch='a'; ch<'z'+1; ch++) {
            int ans=101;
            for(int i=0; i<n; i++) {
                int cnt=0;
                for(auto& c : A[i]) {
                    if(c==ch)
                        cnt++;
                }

                ans=min(ans,cnt);
            }
            string s="";
            s+=ch;
            for(int i=0;i<ans;i++)
                ret.push_back(s);
        }
        return ret;
    }
};
