class Solution {
public:
    vector<string> commonChars(vector<string>& a) {
        int as=a.size();
        vector<int> dir(26,0);
        for(int i=0;i<as;i++)
        {
            vector<int> dirt(26,0);
            for(int j=0;j<a[i].size();j++)
            {
                int c=a[i][j]-'a';
                if(i == 0)
                {
                    dir[c]++;
                }else
                {
                    dirt[c]++;
                }
            }
            if(i>0)
                andD(dir,dirt);
        }
        vector<string> ret;
        for(int i=0;i<26;i++)
        {
                for(int j=0;j<dir[i];j++)
                {
                    string t;
                    t+=(i+'a');
                    ret.emplace_back(t);
                }
        }
        return ret;
    }
    void pD(vector<int>& d)
    {
        for(int i=0;i<26;i++)
        {
            cout<<i<<" "<<d[i]<<endl;
        }
    }
    void andD(vector<int>& d, vector<int>& t)
    {
        for(int i=0;i<26;i++)
        {
            d[i]=min(d[i],t[i]);
           // cout<<i<<" "<<d[i]<<endl;
        }
    }
};
