class Solution {
public:
#define min(a, b) a < b ? a : b

    vector<string> commonChars(vector<string>& A) {
        vector<string> res;
        int m = A.size();
        vector<vector<int>> tmp(m, vector<int> (26, 0));

        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < A[i].size(); ++j) {
                tmp[i][A[i][j] - 'a']++;
            }
        }

        for (int i = 0; i < 26; ++i) {
            int min1 = INT_MAX;
            for (int j = 0; j < m; ++j) {
                min1 = min(min1, tmp[j][i]);
            }
            for (int k = 0; k < min1; ++k) {
                string s;
                s = i + 'a';
                res.push_back(s);
            }
        }
        return res;
    }
};
