class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int cnt[26] = {};
        for (int i = 0; i < 26; i++) {
            cnt[i] = INT_MAX;
        }
        for (auto &s : A) {
            int cnt2[26] = {};

            for (char ch : s) {
                cnt2[ch - 'a']++;
            }
            for (int i = 0; i < 26; i++) {
                cnt[i] = min(cnt[i], cnt2[i]);
            }
        }
        vector<string> ans;
        for (char i = 'a'; i <= 'z'; i++) {

            for (int j = 0; j < cnt[i - 'a']; j++) {
                ostringstream oss;
                oss << i;
                ans.emplace_back(oss.str());
            }
        }
        return ans;
    }
};
