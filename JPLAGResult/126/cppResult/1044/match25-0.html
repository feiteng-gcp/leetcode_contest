class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<string>  res;
        if(A.size() == 0)   return res;
        vector<int> m(26, INT_MAX);
        for(auto s:A) {
            vector<int> temp(26, 0);
            for(char c:s) {
                temp[c - 'a']++;
            }
            
            for(int i = 0; i < 26; i++) {
                m[i] = min(m[i], temp[i]);
            }
        }
        
        
        for(int i = 0; i < 26; i++) {
            for(int j = 0; j < m[i]; j++) {
                string s = "";
                s += ('a' + i);
                res.push_back(s);
            }
        }
        return res;
    }
};
