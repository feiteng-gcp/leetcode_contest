class Solution {
public:
    vector<int> L;
    vector<vector<int>> A;
    
    int ind(char a){
        return(a - 'a');
    }
    
    char ch(int i){
        return(i +'a');
    }
    
    void count_string(string s){
        int l = s.length();
        A.push_back(L);
        int f = A.size() -1;
        for(int i=0; i<l; i++){
            A[f][ind(s[i])]++;
        }
        // cout << A.size() << " " << A[f].size() << endl;
        // for(int i=0; i<A[f].size(); i++){
        //     cout << A[f][i] << " ";
        // }
        // cout <<endl;
    }
    
    vector<string> commonChars(vector<string>& B) {
        int n = B.size();
        
        for(int i=0; i<26; i++){
            L.push_back(0);
        }
        
        for(int i=0; i<n; i++){
            count_string(B[i]);
        }
        vector<string> ans;
        int m = 10000;
        vector<int> all;
        for(int i=0; i<26; i++){

            m=100000;
            for(int j=0; j<A.size(); j++){
                m = min(m, A[j][i]);
            }
            all.push_back(m);
        }
        
        // cout << A.size() << " " << A[0].size();
//         for(int i=0; i<A.size(); i++){
//             for(int j=0; j<A[i].size(); j++){
//                 cout << A[i][j] << " ";
//             }
//             cout << endl;
        
//         }
        
        for(int i=0; i<26; i++){
            for(int j=0; j<all[i]; j++){
                ans.push_back(" ");
                ans[ans.size()-1][0] = ch(i);
            }
        }
        return(ans);
        
    }
};
