class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int N = A.size();
        map<char, int> mp[N];
        for(int i=0; i<N; i++){
            for(char c : A[i]) mp[i][c]++;

        }

        vector<string> ret;
        for(char c='a'; c<='z'; c++){
            int num = 1e5;
            for(int i=0; i<N; i++) num = min(num, mp[i][c]);
            string s = {c};
            while(num--) ret.push_back(s);
        }

        return ret;
    }
};
