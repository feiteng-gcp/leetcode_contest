class Solution {
public:
    int a[26];
    int b[26];
    void merge() {
        for (int i = 0;i < 26;i++) {
            a[i] = a[i] < b[i] ? a[i] : b[i];
        }
    }
    vector<string> commonChars(vector<string>& A) {
        memset(a, 0, sizeof(a));
        for (int i = 0;i < A.size();i++) {
            memset(b, 0, sizeof(b));
            for (int j = 0;j < A[i].size();j++) {
                b[A[i][j] - 'a']++;
            }
            if (i == 0) {
                for (int j = 0;j < 26;j++) {
                    a[j] = b[j];
                }
            } else {
                merge();
            }
        }
        vector<string> ret;
        for (int i = 0;i < 26;i++) {
            for (int j = 0;j < a[i];j++) {
                string s;
                s.push_back(char(i + 'a'));
                ret.push_back(s);
            }
        }
        return ret;
    }
};
