class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<int> counts(26, 0);
        int n = A.size();
        for(auto c : A[0])
        {
            counts[c-'a'] += 1;
        }
        for(int i = 1; i < n; i++)
        {
            vector<int> temp(26, 0);
            for(auto c : A[i])
            {
                temp[c-'a'] += 1;
            }
            for(int j = 0; j < 26; j++)
            {
                counts[j] = min(counts[j], temp[j]);
            }
        }
        vector<string> res;
        for(int i = 0; i < 26; i++)
        {
            if(counts[i] > 0)
            {
                for(int j = 0; j < counts[i]; j++)
                {
                    res.push_back(string(1, 'a'+i));
                }
            }
        }
        return res;
    }
};
