class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int n = A.size();
        vector<vector<int>> count(n,vector<int> (26,0));
        for(int i = 0;i < n;i++){
            string x = A[i];

            for(char c:x){
                count[i][c - 'a']++;
            }
        }
        vector<string> res;
        for(int i = 0;i < 26;i++){
            int maxx = INT_MAX;
            for(int j = 0;j < n;j++)maxx = min(maxx,count[j][i]);
            for(int j = 0;j < maxx;j++){
                char t = 'a' + i;
                string tt = "";
                tt += t;
                res.push_back(tt);
            }
        }
        return res;
        //20190302 CONTEST;
        
    }
};
