class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<int> cnts(26, INT_MAX);
        for (const string &s : A) {
            vector<int> local_cnts(26, 0);
            for (int i = 0; i < s.size(); ++i) {
                local_cnts[s[i] - 'a']++;
            }
            for (int i = 0; i < 26; ++i) {
                cnts[i] = min(cnts[i], local_cnts[i]);
            }
        }
        vector<string> result;
        for (int i = 0; i < 26; ++i) {
            for (int j = 0; j < cnts[i]; ++j) {
                result.emplace_back();
                result.back().push_back('a' + i);
            }
        }
        return result;
    }
};
