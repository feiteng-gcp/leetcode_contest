class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<vector<int> > cnt(200, vector<int> (200, 0));
        for (int i = 0; i < A.size(); ++i) {
          for (int j = 0; j < A[i].size(); ++j) {
            ++cnt[i][A[i][j]];
          }
        }

        vector<int> ans(200, 200);
        for (int i = 0; i < A.size(); ++i) {
          for (int j = 0; j < 200; ++j) {
            ans[j] = min(ans[j], cnt[i][j]);
          }
        }
        
        vector<string> str;
        for (int i = 0; i < 200; ++i) {
          while (ans[i] > 0) {
            --ans[i];
            string tmp = "a";
            tmp[0] = (char)i;
            str.push_back(tmp);
          }
        }

        return str;
    }
};
