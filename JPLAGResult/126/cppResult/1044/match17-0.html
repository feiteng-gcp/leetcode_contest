class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int len = A.size();
        vector<vector<int>> m(len, vector<int>(26, 0));
        for(int i=0;i<len;i++)
        {
            string str = A[i];
            for(auto c : str)
            {
                m[i][c-'a'] ++;
            }
        }
        vector<string> res;
        
        for(int i=0;i<26; i++)
        {
            int mk = 9999;
            for(int j=0;j<len;j++)
            {
                mk = min(mk, m[j][i]);
            }
            string t = "";
            t += ('a'+i);
            for(int k=0;k<mk;k++)
            {
                res.push_back(t);
            }
        }
        return res;
    }
};
