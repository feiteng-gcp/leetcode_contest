class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<string> ans;
        for(char i = 'a';i<='z';i++)
        {
            int temp = INT_MAX;
            for(int j=0;j<A.size();j++)
            {
                int count = 0;
                for(int k=0;k<A[j].size();k++)
                {
                    if(A[j][k]==i)
                        count++;
                }
                temp = min(temp,count);
                if(temp==0)
                    break;
            }
            while(temp--)
            {
                ans.push_back(string(1, i));
            }
        }
        return ans;
    }
};
