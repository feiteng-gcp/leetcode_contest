class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        int n = A.size();
        vector<map<char,int>> cnt(n);
        
        for (int i = 0; i < n; ++i)
        {
            for (char ch : A[i])
            {
                cnt[i][ch]++;
            }
        }
        
        vector<string> ans;
        for (char ch = 'a'; ch <= 'z'; ++ch)
        {
            int cur = INT_MAX;
            for (int i = 0; i < n; ++i)
            {
                cur = min(cur, cnt[i][ch]);
            }
            
            string s;
            s += ch;
            for (int i = 0; i < cur; ++i)
            {
                ans.push_back(s);
            }
        }
        
        return ans;
    }
};
