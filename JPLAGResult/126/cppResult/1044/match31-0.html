class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<int>cnt;
        vector<int>ans(26, 1000);
        
        for (auto& s : A) {
            cnt = vector<int>(26, 0);
            for (auto& c : s) {
                cnt[c - 'a']++;
            }
            for (int i = 0; i < 26; i++) {
                ans[i] = min(ans[i], cnt[i]);
            }
        }
        
        vector<string> ret;
        for (int i = 0; i < 26; i++) {
            while (ans[i]--) {
                ret.push_back(string(1, 'a' + i));
            }
        }
        return ret;
    }
};
