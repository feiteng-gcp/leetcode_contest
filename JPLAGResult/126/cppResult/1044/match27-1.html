class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<int> f(26, 0);
        int flag = 0;
        for(auto &s: A){
            vector<int> g(26, 0);

            for(char c: s){
                g[c-'a']++;
            }
            if(flag){
                for(int i =0; i <26; i++){
                    f[i] = min(f[i], g[i]);
                }
            } else {
                f = g;
                flag=1;

            }
        }
        vector<string> result;
        for(int i =0 ; i < 26;i++){
            for(int j = 0; j < f[i];j++){
                result.push_back(string() + char('a' + i));
            }
        }
        return result;
    }
};
