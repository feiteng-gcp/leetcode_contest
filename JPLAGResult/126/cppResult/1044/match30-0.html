class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<string> res;
        vector<int> total(26, INT_MAX);
        for (string& s : A) {
            vector<int> cnt(26, 0);
            for (char c : s) {
                ++cnt[c-'a'];
            }
            for (int i=0; i<cnt.size(); ++i) {
                total[i] = min(total[i], cnt[i]);
            }
        }
        for (int i=0; i<26; ++i) {
            for (int t=total[i]; t>0; --t) {
                string a; a.push_back('a'+i);
                res.push_back(a);
            }
        }
        
        return res;
    }
};
