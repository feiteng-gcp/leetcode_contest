class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<string> ret;
        int s[26] ={0};
        for(int i = 0;i<26;i++)
            s[i] = 105;
        for(int i =0; i < A.size();i++){
            int temp[26] = {0};

            for(int j = 0 ;j < A[i].size();j++){
                temp[A[i][j] - 'a']++;
            }
            for(int c = 0 ;c<26;c++){
                s[c] = min(s[c],temp[c]);
            }
        }
        for(int i = 0 ;i<26;i++){
            for(int j =0;j<s[i];j++){
                string s; 
                s.push_back('a'+i);
                ret.push_back(s);
            }
        }
        return ret;
    }
};
