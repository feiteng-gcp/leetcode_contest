class Solution {
public:
    bool isValid(string S) {
        vector<char> stack;
        for (int i = 0; i < S.size(); i++) {
            char ch = S[i];


            if (ch == 'a') {
                stack.push_back(ch);
                continue;
            }

            if (stack.empty()) {
                return false;
            }

            if (ch == 'b' && stack.back() == 'a') {
                stack.back() = 'b';
                continue;
            }
            
            if (ch == 'c' && stack.back() == 'b') {
                stack.pop_back();
                continue;
            }
            
            return false;
        }
        
        return stack.empty();
    }
};
