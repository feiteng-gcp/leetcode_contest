class Solution {
public:
    bool isValid(string S) {
        vector<int> stack;
        for (char ch : S) {
            if (ch == 'a') {
                stack.push_back(0);
            } else if (ch == 'b') {
                if (!stack.empty() && stack.back() == 0) {
                    stack.back() += 1;
                } else {
                    return false;
                }
            } else if (ch == 'c') {
                if (!stack.empty() && stack.back() == 1) {
                    stack.pop_back();
                } else {
                    return false;
                }
            }
        }
        return stack.empty();
    }
};
