class Solution {
public:
    bool isValid(string S) {
        int n=S.size();
        if (S[0]=='b' || S[n-1]=='b') return false;
        for(int i=0;i+1<n;i++) if (S[i]=='b' && S[i+1]=='b') return false;

        int ca=0,cb=0,cc=0;
        for(int i=0;i<n;i++)
        {
            if (S[i]=='a') ca++;
            if (S[i]=='b') cb++;
            if (S[i]=='c') cc++;
        }
        if (ca!=cb || ca!=cc || cb!=cc) return false;
        
        int p=0;
        for(int i=0;i<n;i++)
        {
            if (S[i]=='b')
            {
                if (p==0) return false;
            }
            if (S[i]=='a') p++;
            if (S[i]=='c') p--;
            if (p<0) return false;
        }
        if (p!=0) return false;
        return true;
    }
};
