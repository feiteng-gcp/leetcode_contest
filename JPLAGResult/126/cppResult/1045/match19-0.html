class Solution {
public:
    bool isValid(string S) {
        int ptr = 0;
        for (int i = 0; i < S.length(); i++)
        {
            ptr = i;
            if (ptr + 2 < S.length() && S[ptr] == 'a' && S[ptr + 1] == 'b' && S[ptr + 2] == 'c')
            {
                S.erase(S.begin() + ptr, S.begin() + ptr + 3);
                ptr--;
                ptr--;
                ptr--;
                i = ptr;
                if (ptr < 0)
                    i = -1;
            }
            //cout << S << endl;
        }
        if (S.length() > 0)
            return false;
        else
            return true;
    }
};
