#define ALL(x) (x).begin(), (x).end()
#define FOR(i, a, b) for (remove_cv<remove_reference<decltype(b)>::type>::type i = (a); i < (b); i++)
#define REP(i, n) FOR(i, 0, n)

class Solution {
public:
  bool isValid(string S) {
    if (S.empty() || S == "abc") return true;
    if (S[0]!='a' || S.back()!='c') return false;
    int a = 0, b = 0, c = 0, n = S.size();
    REP(i, n-1) {
      if (S[i] == 'a') a++;
      else if (S[i] == 'b') b++;
      else c++;
      if (a==b&&b==c)
        return isValid(S.substr(0,i+1)) && isValid(S.substr(i+1));
    }
    a=b=c=0;
    FOR(i, 1, n-1) {
      if (S[i]=='b' && a==b&&b==c) return isValid(S.substr(1,i-1)) && isValid(S.substr(i+1,n-i-2));
      if (S[i] == 'a') a++;
      else if (S[i] == 'b') b++;
      else c++;
    }
    return false;
  }
};
