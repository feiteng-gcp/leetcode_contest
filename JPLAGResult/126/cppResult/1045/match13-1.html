class Solution {
public:
    bool isValid(string S) {
        int cnt[3], len = S.length();
        cnt[0] = cnt[1] = cnt[2] = 0;

        for (int i = 0; i < len; i++)
        {
            cnt[S[i] - 'a']++;
            if (cnt[0] < cnt[1] || cnt[1] < cnt[2]) return 0;
        }
            
        return (cnt[0] == cnt[1]) && (cnt[1] == cnt[2]);
    }
};
