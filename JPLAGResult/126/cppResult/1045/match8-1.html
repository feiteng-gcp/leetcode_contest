class Solution {
public:
    bool isValid(string S) {
        stack<char> s;
        if (s.size() % 3 != 0) {
            return false;
        }

        for (int i = 0; i < S.size(); ++i) {
            if (S[i] == 'a') {
                s.push('a');
            } else if (S[i] == 'b') {
                if (s.size() > 0 && s.top() == 'a') {
                    s.top() = 'b';
                } else {
                    return false;
                }
            } else if (S[i] == 'c') {
                if (s.size() > 0 && s.top() == 'b') {
                    s.pop();
                } else {
                    return false;
                }
            } else {
                return false;
            }
        }
        return s.size() == 0;
    }
};
