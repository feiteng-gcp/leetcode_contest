class Solution {
public:
    bool isValid(string S) {
        stack<char> buffer;
        
        for (char c : S)
        {
            if (buffer.empty())
            {
                if (c == 'a')
                    buffer.push(c);
                else
                    return false;
            }
            else if (buffer.top() == 'a')
            {
                if (c == 'a')
                {
                    buffer.push(c);
                }
                else if (c == 'b')
                {
                    buffer.pop();
                    buffer.push(c);

                }
                else
                {
                    return false;
                }   
            }
            else if (buffer.top() == 'b')
            {
                if (c == 'a')
                {
                    buffer.push(c);
                }
                else if (c == 'b')
                {
                    return false;
                }
                else
                    buffer.pop();
            }
            else
            {
                return false;
            }
        }
        
        return buffer.empty();
    }
};
