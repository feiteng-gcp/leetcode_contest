class Solution {
public:
    bool isValid(string S) {
        stack<char> st;
        
        for (char c : S) {
            if (c == 'a' || c == 'b') {
                st.push(c);
            }
            else {

                if (!st.empty() && st.top() == 'b') {
                    st.pop();
                    if (!st.empty() && st.top() == 'a') {
                        st.pop();
                    }
                    else {
                        return false;
                    }
                }
                else {
                    return false;
                }
            }
        }
        
        return st.empty();
    }
};
