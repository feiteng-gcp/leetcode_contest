class Solution {
public:
    bool isValid(string S) {
        stack<string> abcStack;
        
        for(auto& c: S){
            if(c == 'a'){
                //make a new stack frame;
                abcStack.push("a");
            } else if(c == 'b') {
                //check if the existing top of the stack frame already contains "a"
                if(abcStack.empty() || abcStack.top() != "a") return false;
                abcStack.top() += c;

            } else {
                if(abcStack.empty() || abcStack.top() != "ab") return false;
                abcStack.pop();
            }
        }
        
        if(!abcStack.empty()) return false;
        
        return true;
    }
};
