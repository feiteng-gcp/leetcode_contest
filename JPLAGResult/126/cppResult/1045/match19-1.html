class Solution {
public:
    bool isValid(string S) {
        int cur = -1;
        for (int i = 0; i < S.size(); i++) {

            cur++;
            S[cur] = S[i];
            
            if (cur >= 2 && S[cur - 2] == 'a' && S[cur - 1] == 'b' && S[cur] == 'c') {
                cur -= 3;
            }
        }
        
        return cur == -1;
    }
};
