class Solution {
public:
    char stack[20010];
    int n;
    bool isValid(string S) {
        n = 0;
        for (int i = 0;i < S.size();i++) {
            stack[n++] = S[i];
            while (n >= 3 && stack[n - 1] == 'c' && stack[n - 2] == 'b' && stack[n - 3] == 'a') {
                n -= 3;
            }
        }
        if (n == 0) {
            return true;
        } else {
            return false;
        }
    }
};
