class Solution {
public:
    
    bool isValid(string S)
{
    int len = (int)S.size();
    vector<int> cnt(3, 0);
    for(int i = 0; i < len; i++)
    {
        cnt[S[i] - 'a']++;
        if(cnt[0] < cnt[1] || cnt[1] < cnt[2] || cnt[0] < cnt[2]) return false;
    }
    return cnt[0] == cnt[1] && cnt[1] == cnt[2];
}
};
