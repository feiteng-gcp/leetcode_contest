class Solution {
private:
    const string valid = "abc";
public:
    bool isValid(string S) {

        while (true) {
            if (S.empty()) {
                return false;
            } else if (S == valid) {
                return true;
            } else {
                auto idx = S.find(valid);
                if (idx == S.npos) {
                    return false;
                } else {
                    string str1 = S.substr(0, idx);
                    string str2 = S.substr(idx + valid.size());
                    S = str1 + str2;
                }
            }
        }
    }
};
