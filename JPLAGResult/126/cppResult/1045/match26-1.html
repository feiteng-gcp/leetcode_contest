class Solution {
public:
    bool isValid(string S) {
        stack<char> st;
        for(auto i: S){
            if( i == 'c' and st.empty() == false ){
                // Check prev two.
                if( st.size() < 2 ){ return false; }
                char f = st.top() ; st.pop();
                if( f != 'b' ){ return false; }
                char ff = st.top(); st.pop();
                if( ff != 'a' ){ return false; }
            }else{
                st.push(i);

            }
        }
        if( st.empty() ){ return true; }
        else{ return false; }
    }
};
