class Solution {
public:
    bool isValid(string S) {
        vector<char> dat;
        for (int i = 0; i < S.size(); ++i) {
            if (S[i] == 'c' && dat.size() >= 2 && dat[dat.size() - 1] == 'b' && dat[dat.size() - 2] == 'a') {
                dat.pop_back();
                dat.pop_back();
            } else
                dat.push_back(S[i]);
        }
        return dat.size() == 0;
    }
};
