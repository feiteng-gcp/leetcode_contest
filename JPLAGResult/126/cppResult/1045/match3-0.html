class Solution {
public:
    bool isValid(string S) {
        string s = S;
        int n = s.length();
        
        stack<char> stk;
        for (int i = 0; i < n; ++i) {
            // cout << i << endl;
            if (s[i] == 'a') {
                stk.push(s[i]);
            } else if (s[i] == 'b') {
                if (stk.empty()) {
                    return false;
                }
                auto t = stk.top();
                stk.pop();
                // cout << "t " << t << endl;
                if (t != 'a') {
                    return false;
                }
                stk.push(s[i]);
            } else if (s[i] == 'c') {
                if (stk.empty()) {
                    return false;
                }
                auto t = stk.top();
                stk.pop();
                if (t != 'b') {
                    return false;
                }
                // stk.push(s[i]);
            } else {
                // cout << "error " << endl;
                return false;
            }
        }
        // cout << "size " << stk.size() << endl;
        return stk.empty();
     }
};
