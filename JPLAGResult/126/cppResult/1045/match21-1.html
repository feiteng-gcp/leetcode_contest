class Solution {
public:
    bool isValid(string S) {
        int n = S.size();
        if(n == 0 || n%3 != 0) return false;
        stack<char> st;
        for(auto c : S)
        {
            if(c == 'c')
            {
                if(st.size() < 2) return false;
                char cur1= st.top();
                st.pop();
                if(cur1 != 'b') return false;

                char cur2 = st.top();
                st.pop();
                if(cur2 != 'a') return false;
            }
            else
            {
                st.push(c);
            }
        }
        return st.empty();
    }
};
