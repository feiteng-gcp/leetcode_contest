class Solution {
public:
    bool isValid(string s) {
        stack<char> ab;
        stack<char> bc;
        int ss=s.size();
        for(int i=0;i<ss;i++)
        {
            if(s[i] == 'a')
            {
                ab.push('a');
            }else if(s[i] == 'b')
            {
                if(ab.empty())
                    return false;
                ab.pop();
                bc.push('b');
            }else
            {
                if(bc.empty())
                    return false;
                bc.pop();
            }
        }
        if(ab.empty() && bc.empty())
            return true;
        return false;
    }
};
