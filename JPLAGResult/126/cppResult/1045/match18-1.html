class Solution {
public:
    bool isValid(string S) {
        int n=S.size();

        stack<char> st;
        for(int i=0; i<n; i++) {
            //
            if(S[i]=='c') {
                if(st.size()<2) {
                    return false;
                }
                char b,a;
                b=st.top();
                st.pop();
                a=st.top();
                st.pop();
                if(a!='a'||b!='b') {
                    return false;
                }
            } else {
                st.push(S[i]);
            }
        }
        return st.empty();
    }
};
