#include <bits/stdc++.h>
using namespace std;

deque<char> D;

class Solution {
public:
    bool isValid(string S) {
        D.clear();
        for (auto c : S) {
            D.push_back(c);
            while (D.size() >= 3) {
                int N = D.size();
                if (D[N-3] == 'a' && D[N-2] == 'b' && D[N-1] == 'c') {
                    D.pop_back();
                    D.pop_back();
                    D.pop_back();
                }
                else break;
            }
        }
        return D.size() == 0;
    }
};
