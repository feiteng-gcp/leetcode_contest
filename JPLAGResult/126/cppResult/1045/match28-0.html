class Solution {
public:
    bool isValid(string S) {
        stack<char> st;
        
        for (char c : S)
        {
            if (c == 'c')
            {
                if (st.empty())
                {
                    // std::cout << "here1" << c << std::endl;
                    return false;
                }
                else if (st.top() != 'b')
                {
                    // std::cout << "here2" << c << std::endl;
                    return false;
                }
                else 
                {
                    st.pop();
                    if (st.empty())
                    {
                        // std::cout << "here1" << c << std::endl;
                        return false;
                    }
                        
                    else if (st.top() != 'a')
                    {
                        // std::cout << "here1" << c << std::endl;
                         return false;
                    }
                       
                    
                    st.pop();
                }
                continue;
            }
            
            // std::cout << "pushing" << c << std::endl;
            st.push(c);
        }
        // std::cout << "here" << std::endl;
        return st.empty();
    }
};
