class Solution {
public:
    bool isValid(string S) {
        //用栈
        stack<char> strStack;
        for (int index = 0; index < S.length(); index ++) {
            //则需要判断
            if(S[index] == 'c') {
                if(strStack.size() < 2) {
                    return false;
                }
                char last = strStack.top();
                strStack.pop();
                if(last != 'b') {
                    return false;
                }
                char lastlast = strStack.top();
                strStack.pop();
                if(lastlast != 'a') {
                    return false;
                }
                continue;
            }
            strStack.push(S[index]);
        }
        return strStack.empty();
    }
};
