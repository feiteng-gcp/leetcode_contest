class Solution {
public:
    bool isValid(string S) {
        stack<string> stk;
        for (int i = 0; i < S.size(); ++i) {
            if (S[i] == 'a') {
                if (i + 1 >= S.size() || S[i + 1] == 'c') {
                    return false;
                }
                if (S[i + 1] == 'a') {
                    stk.push("a");
                    continue;
                }
                if (i + 2 >= S.size()) {
                    return false;
                }
                if (S[i + 2] == 'c') {
                    i += 2;
                    continue;
                }
                stk.push("ab");
                ++i;
            } else if (S[i] == 'b') {
                if (stk.size() < 1 || stk.top() != "a") {
                    return false;
                }
                if (i + 1 >= S.size() || S[i + 1] == 'b') {
                    return false;
                }
                if (S[i + 1] == 'c') {
                    stk.pop();
                    ++i;
                } else {
                    stk.pop();
                    stk.push("ab");
                }
            } else {
                if (stk.size() < 1 || stk.top() != "ab") {
                    return false;
                }
                stk.pop();
            }
        }
        return stk.size() == 0;
    }
};
