from itertools import accumulate
class Solution:
    def mergeStones(self, stones: List[int], K: int) -> int:
        length = len(stones)
        if K > 2 and length % (K-1) != 1:
            return -1
        prefix = [0]+list(accumulate(stones))
        INF = 10000
        dp = [[[INF]*35 for i in range(35)] for j in range(35)]
        for i in range(1,length+1):
            dp[i][i][1] = 0
        for l in range(1,length+1):
            for i in range(1,length-l+2):
                j = i + l - 1
                for k in range(2,l+1):
                    for m in range(i,j):
                        dp[i][j][k] = min(dp[i][j][k],dp[i][m][k-1]+dp[m+1][j][1])
                dp[i][j][1] = min(dp[i][j][1],dp[i][j][K] + prefix[j]-prefix[i-1])
        return dp[1][length][1]
