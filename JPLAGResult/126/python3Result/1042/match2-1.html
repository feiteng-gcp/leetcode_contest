class Solution:
    def mergeStones(self, stones: List[int], K: int) -> int:
        n = len(stones)
        if (n - 1) % (K - 1) != 0:
            return -1
        
        prefixSum = [0 for i in range(n + 1)]
        for i in range(1, n + 1):
            prefixSum[i] = prefixSum[i - 1] + stones[i - 1]
        
        dp = [[{} for _ in range(n)] for _ in range(n)]
        
        for i in range(n):
            dp[i][i][1] = 0
            for k in range(2, K + 1):
                dp[i][i][k] = float('inf')
        
        for j in range(1, n):
            for i in range(j - 1, -1, -1):
                for k in range(2, K + 1):
                    dp[i][j][k] = float
('inf')
                    for cut in range(i, j):
                        dp[i][j][k] = min(dp[i][j][k], dp[i][cut][k - 1] + dp[cut + 1][j][1])
                dp[i][j][1] = dp[i][j][K]
                dp[i][j][1] += prefixSum[j + 1] - prefixSum[i]
                
        if dp[0][n - 1][1] == float('inf'):
            return 0
        
        return dp[0][n - 1][1]
