class Solution:
    def longestOnes(self, A: List[int], K: int) -> int:
        count0 = 0
        left = 0
        max_len = 0
        n = len(A)
        for i in range(n):
            if A[i] == 0:
                count0 += 1
            while (count0 > K):
                if A[left] == 0:
                    count0 -= 1
                left += 1
            max_len = max(max_len, i-left + 1)
        return max_len
