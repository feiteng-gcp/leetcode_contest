class Solution:
    def longestOnes(self, A: List[int], K: int) -> int:
        if not A:
            return 0
        if K < 0:
            K = 0
            
        ans = 0
        zeroes = 0
        i = 0
        
        for j in range(len(A)):
            if A[j] == 0:
                zeroes += 1
            
            while i < j and zeroes > K:
                if A[i] == 0:
                    zeroes -= 1
                i += 1
            
            if zeroes <= K:
                ans = max(ans, j - i + 1)
        
        return ans
