class Solution:
    def longestOnes(self, A: List[int], K: int) -> int:
        left = 0
        res = 0
        queue = []
        for right in range(len(A)):
            if A[right] == 0:
                queue.append(right)
            if len(queue) > K:
                left = queue[0] + 1
                queue.pop(0)
            res = max(res, right - left + 1)
        return res
