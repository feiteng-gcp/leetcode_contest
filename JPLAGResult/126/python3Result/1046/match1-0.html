class Solution:
    def longestOnes(self, A: List[int], K: int) -> int:
        n = len(A)
        z, l, res = 0, 0, 0
        for r in range(n):
            if A[r] == 0:
                z += 1
            while z > K:
                if A[l] == 0:
                    z -= 1
                l += 1
            res = max(res, r - l + 1)
        return res
