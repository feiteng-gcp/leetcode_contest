class Solution:
    def longestOnes(self, A: List[int], K: int) -> int:
        r = 0
        s = 0
        zn = 0
        for i in range(len(A)):
            if A[i] == 0:
                zn += 1
            while zn > K:
                if A[s] == 0:
                    zn -= 1
                s += 1
            r = max(r, i - s + 1)
        return r
