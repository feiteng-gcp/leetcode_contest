class Solution:
    def longestOnes(self, A: List[int], K: int) -> int:
        left = 0
        cnt = 0
        rtn = 0
        for right in range(len(A)):
            if A[right] == 0:
                cnt += 1
            while cnt > K:
                if A[left] == 0:
                    cnt -= 1
                left += 1
            rtn = max(rtn, right - left + 1)
        
        return rtn
