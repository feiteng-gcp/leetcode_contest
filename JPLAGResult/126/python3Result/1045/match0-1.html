class Solution:
    def isValid(self, S: str) -> bool:
        stack = []
        for c in S:
            if c == 'a':
                stack.append('a')
            elif c == 'b':
                if len(stack) == 0 or stack[-1][0] != 'a':
                    return False
                stack[-1] += 'b'
            elif c == 'c':
                if len(stack) == 0 or stack[-1] != 'ab':
                    return False
                stack.pop()
            else:
                return False
            #print (stack)
        return len(stack) == 0
