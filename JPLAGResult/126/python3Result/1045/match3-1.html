class Solution:
    def isValid(self, S: str) -> bool:
        stack = []
        
        for i in range(len(S)):
            if S[i] == 'a':
                stack.append('a')
            elif S[i] == 'b':
                if len(stack)<1:
                    return False
                if stack[-1]!='a':
                    return False
                stack.append('b')
            elif S[i] == 'c':
                
if len(stack)<2:
                    return False
                if stack[-1]!='b' or stack[-2]!='a':
                    return False
                stack.pop()
                stack.pop()
        return True if len(stack)==0 else False
