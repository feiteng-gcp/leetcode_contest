class Solution:
    def isValid(self, S: str) -> bool:
        if len(S) % 3: return False
        
        st = []
        for ch in S:
            if ch == 'c':
                if len(st) < 2: return False
                if st[-1] != 'b' or st[-2] != 'a':
                    return False
                st.pop()
                st.pop()
            else:
                st.append(ch)
        return len(st) == 0
