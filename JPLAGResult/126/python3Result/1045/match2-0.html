class Solution:
    def isValid(self, S: str) -> bool:
        dictionary={'a':0,'b':0,'c':0}
        for c in S:
            dictionary[c]=dictionary[c]+1
            if dictionary['a']<dictionary['b'] or dictionary['b']<dictionary['c']:
                return False
        if not dictionary['a']==dictionary['b'] or not dictionary['b']==dictionary['c']:
            return False
        return True
