class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        cnts = []
        for a in A:
            cnts.append(collections.Counter(a))
        ans = []
        for x in range(26):
            ch = chr(ord('a') + x)
            m = min([cnt.get(ch, 0) for cnt in cnts])
            if m > 0:
                ans.extend([ch] * m)
        return ans
