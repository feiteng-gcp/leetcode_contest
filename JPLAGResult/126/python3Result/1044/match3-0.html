from collections import defaultdict
class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        n = len(A)
        word_compo = defaultdict(int)
        for i in A[0]:
            word_compo[i] += 1
        for j in range(1,n):
            tmp_word = defaultdict(int)
            for k in A[j]:
                tmp_word[k] += 1
            for i in word_compo:
                word_compo[i] = min(word_compo[i], tmp_word[i])
        ans = []
        for i in word_compo:
            for j in range(word_compo[i]):
                ans.append(i)
        return ans
