class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        if not A:
            return []
        count = collections.Counter(A[0])
        for w in A[1:]:
            ct 
= collections.Counter(w)
            for c in 'abcdefghijklmnopqrstuvwxyz':
                count[c] = min(count[c], ct[c])
        res = []
        for c in 'abcdefghijklmnopqrstuvwxyz':
            res.extend([c] * count[c])
        return res
