from collections import defaultdict
class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        def get_count(s):
            d = defaultdict(int)
            for c in s:
                d[c] += 1
            return d
        if not A:
            return []
        d = get_count(A[0])
        for i in range(1, len(A)):
            new_d = get_count(A[i])
            
for c in d:
                d[c] = min(d[c], new_d[c])
        res = []
        for c in d:
            for _ in range(d[c]):
                res.append(c)
        return res
