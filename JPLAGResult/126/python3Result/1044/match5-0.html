class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        counters = []
        for a in A:
            counters.append(collections.Counter(a))
        min_count_dict = {}
        for c in 'abcdefghijklmnopqrstuvwxyz':
            min_count = min([x[c] for x in counters])
            if min_count:
                min_count_dict[c] = min_count 
        
        ans = []
        for c, count in min_count_dict.items():
            ans.extend([c] * count)
        
        return ans
