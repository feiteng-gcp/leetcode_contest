class Solution:
    def getD(self, s):
        d = {}
        for c in s:
            if c not in d:
                d[c] = 1
            else:
                d[c] += 1
        return d
    
    def findCommon(self, d1, d2):
        out = {}
        for k, v in d1.items():
            if k in d2:
                out[k] = min(v, d2[k])
        return out
    
    def convertD(self, d):
        out 
= []
        for k, v in d.items():
            for i in range(v):
                out.append(k)
        return out
    
    def commonChars(self, A: List[str]) -> List[str]:
        dics = [self.getD(s) for s in A]
        for d in dics[1:]:
            dics[0] = self.findCommon(dics[0], d)
        return self.convertD(dics[0])
