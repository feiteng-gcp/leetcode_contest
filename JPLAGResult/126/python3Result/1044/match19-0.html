from collections import Counter
class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        global_counter = Counter(A[0])
        res = []
        for string in A:
            counter = Counter(string)
            for key, val in global_counter.items():
                if key not in counter or global_counter[key] == -1:
                    global_counter[key] = -1
                    continue
                global_counter[key] = min(global_counter[key], counter[key])
        for key, val in global_counter.items():
            for i in range(val):
                res.append(key)      
        return res
