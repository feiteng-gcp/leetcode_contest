class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        
        def findDict(word):
            char_dict = {}
            for char in word:
                if char not in char_dict.keys():
                    char_dict[char] = 1
                else:
                    char_dict[char] += 1
            return char_dict
        
        min_dict = findDict(A[0])
        for word in A:
            c_dict = findDict(word)
            for key in min_dict.keys():
                if key not in c_dict.keys():
                    min_dict[key] = 0
                else:
                    min_dict[key] = min(min_dict[key], c_dict[key])
        
        output = []
        for key in min_dict.keys():
            output += [key]*min_dict[key]
        return output
