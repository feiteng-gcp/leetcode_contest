class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        sta = {}
        for i in A[0]:
            if i in sta:
                sta[i] += 1
            else:
                sta[i] = 1
        
        for i in range(1, len(A)):
            dic = {}
            for k in A[i]:
                if k in dic:
                    dic[k] += 1
                else:
                    dic[k] = 1
            temp_dic = {}
            for k in sta:
                if k in dic:
                    temp_dic[k] = min(sta[k], dic[k])
            sta =
 temp_dic
        
        rtn = []
        for i in sta:
            for k in range(sta[i]):
                rtn.append(i)
        return rtn
