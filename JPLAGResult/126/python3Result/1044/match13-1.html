class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        c = collections.Counter(A[0])
        for i in range(1, len(A)):
            c2 = collections.Counter(A[i])
            cc = collections.Counter()
            for a, b in c.items():
                n = min(b, c2[a])
                if n > 0:
                    cc[a] = 
n
            c = cc
        r = []
        for a, b in c.items():
            for i in range(b):
                r.append(a)
        return r
