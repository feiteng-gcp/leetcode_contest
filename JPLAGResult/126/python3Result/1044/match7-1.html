class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        counts = {}
        for c in A[0]:
            counts[c] = counts.get(c, 0) + 1
        for a in A[1:]:
            for k, v in counts.items():
                counts[k] = min(v, a.count(k))
        result = []
        for k, v in counts.items():
            result += [k] * v
        return result
