class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        if not A: return []
        b = dict(collections.Counter(A[0]))
        for w in A[1:]:
            wc = collections.Counter(w)
            for k in b.keys():
                if k not in wc:
                    b[k] = 0
                else:
                    b[k] = min(b[k], wc[k])
        ret = []
        # print(b)
        for k,v in b.items():
            if v > 0:
                ret.extend([k] * v)
        return ret
