class Solution:
    def commonChars(self, A: List[str]) -> List[str]:
        dic = {}
        if not A: return []
        
for c in A[0]:
            dic[c] = dic.get(c, 0) + 1
        for w in A[1:]:
            tmp = {}
            for c in w:
                if c in dic:
                    tmp[c] = tmp.get(c, 0) + 1
            for k, v in tmp.items():
                tmp[k] = min(v, dic[k])
            dic = tmp
        res = []
        for k, v in tmp.items():
            res += [k] * v
        return res
