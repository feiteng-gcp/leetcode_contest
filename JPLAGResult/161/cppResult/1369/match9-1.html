class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int n=s1.length();
        int a=0,b=0;
        for(int i=0;i<n;i++){
            if(s1[i]==s2[i])continue;
            if(s1[i]=='x')a++;
            else b++;
        }
        int ans=a/2+b/2;
        a=a&1;
        b=b&1;
        if(a+b==0)return ans;
        if(a+b==1)return -1;
        return ans+2;
    }
};
