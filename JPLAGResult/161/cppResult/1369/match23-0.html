class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int xn = 0;
        int yn = 0;
        for (int i = 0; i < s1.size(); i++) {
            if (s1[i] != s2[i]) {
                if (s1[i] == 'x') {
                    xn += 1;
                } else {
                    yn += 1;
                }
            }
        }
        if ((xn + yn) % 2 != 0) {
            return -1;
        }
        int ret = 0;
        if (xn % 2 == 0) {
            ret += (xn / 2);
        } else {
            ret += (xn / 2 + 1);
        }
        if (yn % 2 == 0) {
            ret += (yn / 2);
        } else {
            ret += (yn / 2 + 1);
        }
        return ret;
    }
};
