class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int n=s1.length();
        int c0=0,c1=0;
        for(int i=0;i<n;i++){
            if(s1[i]==s2[i])continue;
            if(s1[i]=='x')c0++;
            else c1++;
        }
        int ans=c0/2+c1/2;
        c0&=1;
        c1&=1;
        if(c0+c1==0)return ans;
        if(c0+c1==1)return -1;
        return ans+2;
    }
};
