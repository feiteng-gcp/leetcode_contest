class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int xy = 0, yx = 0;
        for(int i=0;i<s1.length();i++){
            if(s1[i] == 'x' && s2[i]=='y') xy++;
            else if(s1[i] == 'y' && s2[i]=='x') yx++;
        }
        if((xy+yx)&1)
            return -1;
        int ans = 0;
        ans = xy/2 + yx/2;
        if(xy&1 && yx&1){
            ans+=2;
        }
        return ans;
    }
};
