class Solution {
public:
  int minimumSwap(string s1, string s2) {
    int diff_s1_x = 0;
    int diff_s1_y = 0;
    for (int i = 0 ; i < s1.size(); ++i) {
      if (s1[i] == 'x' && s2[i] == 'y') diff_s1_x++;
      if (s1[i] == 'y' && s2[i] == 'x') diff_s1_y++;
    }
    if ((diff_s1_x + diff_s1_y) % 2) return -1;
    return diff_s1_x / 2 + diff_s1_y / 2 + (diff_s1_x % 2) * 2;
  }
};
