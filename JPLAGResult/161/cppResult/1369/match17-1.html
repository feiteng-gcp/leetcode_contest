class Solution {
public:
    int minimumSwap(string s, string t) {
        int n = s.size();
        
        int x = 0, y = 0;
        for (int i = 0; i < n; ++ i)
        {
            if (s[i] == t[i]) continue;

            if (s[i] == 'x') ++ x; else ++ y;
        }
        if ((x+y)%2 == 1) return -1;
        
        int ret = x/2+y/2;
        x %= 2;
        y %= 2;
        if (x || y) ret += 2;
        return ret;
    }
};
