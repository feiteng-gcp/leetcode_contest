class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int cnt1=0,cnt2=0;
        
        for(int i = 0 ; i < s1.size() ; i++) {
            if(s1[i]==s2[i]) continue;
            if(s1[i] == 'x') cnt1++;
            else cnt2++;
        }
        
        // cout << cnt1 << endl << cnt2 << endl;
        
        int ans = 0;
        ans += cnt1/2;
        cnt1 &= 1;
        ans += cnt2/2;
        cnt2 &= 1;
        
        if(cnt1==0 && cnt2==0) return ans;
        else if(cnt1==1 && cnt2==1) return ans+2;
        else return -1;
    }
};
