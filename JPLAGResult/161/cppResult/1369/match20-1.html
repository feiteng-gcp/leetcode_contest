class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int cntx=0, cnty=0;
        //cout << "hi1 " << endl;
        for(int i=0; i<s1.size(); i++){
            if(s1[i]!=s2[i] && s1[i]=='x') cntx++;
            else if(s1[i]!=s2[i] && s1[i]=='y') cnty++;
        }
        //cout << "hi " << endl;
        if((cntx+cnty)%2) return -1;
        else return cntx/2+cnty/2+(2*(cntx%2));
    }
};
