class Solution {
public:
    int minimumSwap(string s1, string s2) {

        int n = s1.size(), oneX = 0, oneY = 0;
        for (int i=0; i<n; ++i){
            if (s1[i]!=s2[i]){
                if (s1[i]=='x'){
                    ++oneX;
                }else{
                    ++oneY;
                }
            }
        }
        if ((oneX+oneY)%2!=0){
            return -1;
        }
        
        int ans = 0;
        if (oneX%2==1){
            ans += oneX/2+1;
        }else{
            ans += oneX/2;
        }
        
        if (oneY%2==1){
            ans += oneY/2+1;
        }else{
            ans += oneY/2;
        }
        
        return ans;
    }
};
