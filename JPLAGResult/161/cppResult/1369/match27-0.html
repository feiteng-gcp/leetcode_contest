class Solution {
public:
    int minimumSwap(string s1, string s2) {
        if (s1.size() != s2.size()) return -1;
        int cnt_xy = 0;
        int cnt_yx = 0;
        for (int i = 0; i < s1.size(); ++i) {
            if (s1[i] == 'x' && s2[i] == 'y') ++cnt_xy;
            if (s1[i] == 'y' && s2[i] == 'x') ++cnt_yx;
        }
        int ans = cnt_xy / 2 + cnt_yx / 2;
        cnt_xy %= 2;
        cnt_yx %= 2;
        if (cnt_xy + cnt_yx == 2) ans += 2;
        if (cnt_xy + cnt_yx == 1) return -1;
        return ans;
    }
};
