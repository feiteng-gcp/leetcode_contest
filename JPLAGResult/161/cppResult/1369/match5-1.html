class Solution {
public:
    int minimumSwap(string s1, string s2) 
    {
        int count1=0, count2=0;
        for (int i=0; i<s1.size(); i++)
        {
            if (s1[i]==s2[i]) continue;
            if (s1[i]=='x') count1++;
            else count2++;
        }
        
        int count=0;
        count+=count1/2;
        count1=count1%2;
        count+=count2/2;
        count2=count2%2;
        
        if (count1!=count2)
            return -1;
        else if (count1==1)
            return count+2;
        else
            return count;
    }
};
