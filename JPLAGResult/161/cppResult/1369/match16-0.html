class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int nxy = 0;
        int nyx = 0;
        for (int i = 0; i < s1.size(); ++i) {
            if (s1[i] == s2[i]) continue;
            if (s1[i] == 'x' && s2[i] == 'y') 
                nxy++;
            else 
                nyx++;
        }
        if ((nxy+nyx)%2 != 0)
            return -1;
        int ans = 0;
        ans += nxy/2;
        ans += nyx/2;
        if (nxy%2 != 0)
            ans += 2;
        return ans;
    }
};
