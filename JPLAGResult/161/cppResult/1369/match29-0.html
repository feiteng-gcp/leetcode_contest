class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int len = s1.length();
        int x=0;
        int y=0;
        for(int i=0;i<len;i++)
        {
            if(s1[i]!=s2[i])
            {
                if(s1[i]=='x') x++;
                else y++;
            }
        }
        if((x+y)%2) return -1;
        if(x+y ==0) return 0;
        int res = x/2+y/2;
        return res+(x+y-2*res);
    }
};
