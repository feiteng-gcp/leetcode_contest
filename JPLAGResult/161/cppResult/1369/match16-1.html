class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int cnt1 = 0;
        int cnt2 = 0;
        for (int i = 0; i < s1.size(); i++)
        {
            if (s1[i] == 'x' && s2[i] == 'y')
                cnt1++;
            if (s1[i] == 'y' && s2[i] == 'x')

                cnt2++;
        }
        if ((cnt1 + cnt2) % 2)
            return -1;
        int ans = 0;
        ans += cnt1 / 2;
        ans += cnt2 / 2;
        if (cnt1 % 2)
            ans += 2;
        return ans;
    }
};
