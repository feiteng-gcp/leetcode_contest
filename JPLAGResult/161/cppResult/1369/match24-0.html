class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int cxy = 0, cyx = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1[i] == 'x' && s2[i] == 'y') {
                cxy++;
            } else if (s1[i] == 'y' && s2[i] == 'x') {
                cyx++;
            }
        }
        if (cxy % 2 == 0 && cyx % 2 == 0) {
            return (cxy + cyx) / 2;
        } else if (cxy % 2 == 1 && cyx % 2 == 1) {
            return (cxy + cyx) / 2 + 1;
        }
        return -1;
    }
};
