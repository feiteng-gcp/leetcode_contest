class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int a=0, b=0, i=0;
        while( i<s1.size() ){
            if ( s1[i]==s2[i] ){ ++i; continue;}
            if ( s1[i]=='x' ) ++a;
            if ( s2[i]=='x' ) ++b;
            ++i;
        }
        if ((a+b)%2==1 ) return -1;
        return (a/2)+(b/2)+( a%2 )*2;
    }
};
