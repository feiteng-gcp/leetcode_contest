class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int n_xy = 0, n_yx = 0;
        for(int i=0;i<s1.size();++i) {
            if(s1[i] == 'x' && s2[i] == 'y')
                n_xy++;
            if(s1[i] == 'y' && s2[i] == 'x')
                n_yx++;
        }
        
        if((n_xy + n_yx) % 2 == 1) 
            return -1;
        
        return (n_xy + 1) / 2 + (n_yx + 1) / 2;
    }
};
