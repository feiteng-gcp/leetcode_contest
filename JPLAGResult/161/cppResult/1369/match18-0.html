class Solution {
public:
    int minimumSwap(string s1, string s2) {

        int n = s1.size();

        int cnt1 = 0, cnt2 = 0;
        for (int i = 0; i < n; i++)
        {
            if (s1[i] == 'x' && s2[i] == 'y')
                cnt1 += 1;
            if (s2[i] == 'x' && s1[i] == 'y')
                cnt2 += 1;
        }

        if (cnt1 % 2 == 1 && cnt2 % 2 == 0)
            return -1;
        if (cnt1 % 2 == 0 && cnt2 % 2 == 1)
            return -1;

        if (cnt1 % 2 == 1 && cnt2 % 2 == 1)
            return (cnt1 / 2 + cnt2 / 2 + 2);

        return cnt1 / 2 +  cnt2 / 2;
        

    }
};
