class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int n = s1.size();
        int c1 = 0, c2 = 0;
        for(int i=0;i<n;i++){
            if(s1[i]=='x'&&s2[i]=='y')c1++;
            else if(s1[i]=='y'&&s2[i]=='x')c2++;
        }
        if((c1+c2)&1)return -1;
        int ret = c1/2 + c2/2;
        c1%=2;
        c2%=2;
        if(c1)ret+=2;
        return ret;
    }
};
