class Solution {
public:
    bool vis[100005];
    string minRemoveToMakeValid(string s) {
        stack< int > sta;
        for(int i = 0; i < s.length(); ++i) {
            if(s[i] == '(') {
                sta.push(i);
            } else if(s[i] == ')') {
                if(sta.empty())
                    vis[i] = true;
                else {
                    sta.pop();
                }
            }
                
        }
        while(!sta.empty()) {
            vis[sta.top()] = true;
            sta.pop();
        }
        string ans;
        for(int i = 0; i < s.length(); ++i)
            if(!vis[i])
                ans += s[i];
        return ans;
    }
};
