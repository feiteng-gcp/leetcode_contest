class Solution {
public:
    string minRemoveToMakeValid(string s) {
        string ans = "", s2 = "";
        int open = 0;
        vector<char> v;
        
        for (char ch:s) {
            if (ch == '(') {
                open++;
                s2 += ch;
            } else if (ch == ')') {
                if (open > 0) {
                    open--;
                    s2 += ch;
                }
            } else {
                s2 += ch;
            }
        }
        
        reverse(s2.begin(), s2.end());
        s = s2;
        
        open = 0;
        s2 = "";
        for (char ch:s) {
            if (ch == ')') {
                open++;
                s2 += ch;
            } else if (ch == '(') {
                if (open > 0) {
                    open--;
                    s2 += ch;
                }
            } else {
                s2 += ch;
            }
        }
        
        reverse(s2.begin(), s2.end());
        return s2;
    }
};
