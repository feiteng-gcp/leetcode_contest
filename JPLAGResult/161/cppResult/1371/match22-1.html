class Solution {
public:
    string minRemoveToMakeValid(string s) {
        vector<bool> del(s.size(), false);

        stack<int> st;
        
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '(') {
                st.push(i);
            }
            if (s[i] == ')') {
                if (st.empty()) {
                    del[i] = true;
                } else {
                    st.pop();
                }
            }
        }
        
        while (!st.empty()) {
            del[st.top()] = true;
            st.pop();
        }
        
        string answer;
        

        for (int i = 0; i < s.size(); i++) {
            if (!del[i]) {
                answer.push_back(s[i]);
            }
        }
        
        return answer;
    }
};
