class Solution {
public:
    string minRemoveToMakeValid(string s) {
        stack<int> st;
        string ans="";

        int n=s.size();
        vector<int> rem(n,0);
        for(int i=0;i<n;i++){
            if(s[i]=='(')st.push(i);
            else if(s[i]==')'){
                if(st.empty())rem[i]=1;
                else st.pop();
            }
        }
        while(!st.empty()){
            rem[st.top()]=1;
            st.pop();
        }
        for(int i=0;i<n;i++){
            if(!rem[i])ans+=s[i];
        }
        return ans;
    }
};
