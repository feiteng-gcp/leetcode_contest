class Solution {
public:
	string minRemoveToMakeValid(string s) {
		vector<bool> rm(s.length(), false);

		stack<int> l;
		for (int i = 0; i < s.length(); i++)
		{
			if (s[i] == '(')
				l.push(i);
			else if (s[i] == ')')
			{
				if (l.empty())
					rm[i] = true;
				else
					l.pop();
			}
		}
		while (!l.empty())
		{
			rm[l.top()] = true;
			l.pop();
		}
		string ans;
		for (int i = 0; i < s.length(); i++)
			if (!rm[i])
				ans.push_back(s[i]);
		return ans;
	}
};
