class Solution {
public:
    string minRemoveToMakeValid(string s) {
        string ret;
        int cnt = 0;
        for (auto c : s)

        {
            if (c == '(')
            {
                ret += c;
                cnt ++;
            }
            else if (c == ')')
            {
                if (cnt >= 1)
                {
                    ret += c;
                    cnt --;
                }
            }
            else
                ret += c;
        }
        
        
        cnt = 0;
        s = ret;
        reverse(s.begin(), s.end());
        ret = "";
        for (auto c : s)

        {
            if (c == ')')
            {
                ret += c;
                cnt ++;
            }
            else if (c == '(')
            {
                if (cnt >= 1)
                {
                    ret += c;
                    cnt --;
                }
            }
            else
                ret += c;
        }
        
        reverse(ret.begin(), ret.end());
        
        return ret;
    }
};
