class Solution {
public:
    string minRemoveToMakeValid(string s) {
        stack<int> leftp, rightp;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '(') {
                leftp.push(i);
            } else if (s[i] == ')') {
                if (!leftp.empty()) {
                    leftp.pop();
                } else {
                    rightp.push(i);
                }
            } else {
                continue;
            }
        }
        unordered_set<int> indices;
        while (!leftp.empty()) {
            indices.insert(leftp.top());
            leftp.pop();
        }
        while (!rightp.empty()) {
            indices.insert(rightp.top());
            rightp.pop();
        }
        string valids;
        for (int i = 0; i < s.size(); i++) {
            if (indices.find(i) == indices.end()) {
                valids.append(string(1, s[i]));
            }
        }
        return valids;
    }
};
