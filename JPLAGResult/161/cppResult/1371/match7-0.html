class Solution {
public:
    string minRemoveToMakeValid(string s) 
    {
        stack<int> st;
        int n=s.size();
        vector<bool> v(n,0);
        //return s;
        for(int i=0;i<n;i++)
        {
            if(s[i]=='(')
            {
                st.push(i);
            }
            else if(s[i]==')')
            {
                if(st.empty())
                {
                    v[i]=1;
                }
                else
                {
                    st.pop();
                }
            }
        }
        while(!st.empty())
        {
            v[st.top()]=1;
            st.pop();
        }
        string ans="";
        //return ans;
        for(int i=0;i<n;i++)
        {
            if(!v[i])
            {
                ans+=s[i];
            }
        }
        return ans;
    }
};
