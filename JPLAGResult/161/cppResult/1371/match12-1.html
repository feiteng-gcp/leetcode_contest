class Solution {
public:
    string minRemoveToMakeValid(string s) {
        int n = s.length();

        int balance = 0;
        for (int i = 0; i < n; i++) {
            if (s[i] == '(') {
                balance++;
            } else if (s[i] == ')') {
                if (balance > 0) {
                    balance--;
                } else {
                    s[i] = ' ';
                }
            }
        }
        

        if (balance) {
            int t = balance;
            balance = 0;
            for (int i = n-1; i >= 0; i--) {
                if (s[i] == ')') {
                    balance--;
                } else if (s[i] == '(') {
                    if (balance < 0) {
                        balance++;
                    } else {
                        s[i] = ' ';
                    }
                }
            }
        }
        
        string ans;

        for (int i = 0; i < n; i++) {
            if (s[i] != ' ') {
                ans += s[i];
            }
        }
        
        return ans;        
    }
};
