class Solution {
public:
    string minRemoveToMakeValid(string s) {
        vector<bool> del(s.size());
        stack<int> sl;
        stack<int> sr;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '(') {
                sl.push(i);
            } else if (s[i] == ')') {
                if (!sl.empty()) {
                    sl.pop();
                } else {
                    sr.push(i);
                }
            }
        }
        while (!sl.empty()) {
            del[sl.top()] = true;
            sl.pop();
        }
        while (!sr.empty()) {
            del[sr.top()] = true;
            sr.pop();
        }
        string result;
        for (int i = 0; i < s.size(); ++i) {
            if (!del[i]) {
                result.push_back(s[i]);
            }
        }
        return result;
    }
};
