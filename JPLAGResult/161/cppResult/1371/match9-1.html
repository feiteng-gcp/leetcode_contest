class Solution {
public:
    string minRemoveToMakeValid(string s) {
        unordered_set<int> to_remove;
        

        stack<int> ss;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '(') {
                ss.emplace(i);
            } else if (s[i] == ')') {
                if (!ss.empty()) {
                    ss.pop();
                } else {
                    to_remove.emplace(i);
                }
            }
        }
        
        while (!ss.empty()) {
            to_remove.emplace(ss.top());
            ss.pop();
        }
        
        string res;

        for (int i = 0; i < s.size(); ++i) {
            if (to_remove.count(i) == 0) {
                res.push_back(s[i]);
            }
        }
        return res;
    }
};
