class Solution {
public:
    string minRemoveToMakeValid(string s) 
    {
        string t;
        int count = 0;
        for (auto ch:s)
        {
            if (isalpha(ch))
                t.push_back(ch);
            else if (ch=='(')
            {
                count++;
                t.push_back(ch);
            }
            else
            {
                if (count>0)
                {
                    t.push_back(ch);
                    count--;
                }
            }
        }
        
        //cout<<t<<endl;
        if (count==0) return t;
        
        string r;
        count = 0;
        for (int i=t.size()-1; i>=0; i--)
        {
            char ch = t[i];
            if (isalpha(ch))
                r.push_back(ch);
            else if (ch==')')
            {
                count++;
                r.push_back(ch);
            }
            else
            {
                if (count>0)
                {
                    r.push_back(ch);
                    count--;
                }
            }
        }
        
        reverse(r.begin(),r.end());
        
        return r;
    }
};
