class Solution {
public:
    string minRemoveToMakeValid(string s) {
        stack<int> stk;
        int n = s.size();
        vector<int> f(n);
        for(int i=0;i<n;++i){
            if(s[i] == '(') stk.push(i);
            else if(s[i] == ')'){
                if(stk.size() == 0) f[i] = 1;
                else stk.pop();
            }
        }
        while(stk.size()){
            f[stk.top()] = 1;
            stk.pop();
        }
        string ans;
        for(int i=0;i<n;++i){
            if(!f[i]) ans.push_back(s[i]);
        }
        return ans;
    }
};
