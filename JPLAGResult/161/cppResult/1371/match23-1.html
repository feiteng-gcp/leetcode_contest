class Solution {
public:
    string minRemoveToMakeValid(string s) {
        string ans="";
        int cnt = 0;

        for(int i=0;i<s.size();i++){
            if(s[i]=='('){ans+=s[i]; cnt++;}
            else if(s[i]==')'){
                if(cnt>0) {ans+=s[i]; cnt--;}
            }else{
                ans+=s[i];
            }
        }
        cout<<ans<<endl;
        s = ans;
        string t="";
        cnt=0;
        for(int i=s.size()-1;i>=0;i--){

            if(s[i]==')') {t+=s[i];cnt++;}
            else if(s[i]=='('){
                if(cnt>0) {t+=s[i];cnt--;}
            }else{
                t+=s[i];
            }
        }
        reverse(t.begin(),t.end());
        return t;
    }
};
