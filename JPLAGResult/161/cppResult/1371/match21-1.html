class Solution {
public:
    string minRemoveToMakeValid(string s) {
        int i, x, l;
        l = s.size();
        vector<int> v(l, 0);

        stack<int> idx;
        for (i = 0; i < l; i++) {
            if (s[i] == '(') {
                idx.push(i);
            } else if (s[i] == ')') {
                if (idx.size() > 0) {
                    v[i] = 1;
                    v[idx.top()] = 1;
                    idx.pop();
                }
            } else {
                v[i] = 1;
            }
        }
        string ans;
        for (i = 0; i < l; i++) {
            if (v[i]) {
                ans += s[i];
            }
        }
        return ans;
    }
};
