bool vis[100005];

class Solution {
public:
    string minRemoveToMakeValid(string s) {
        stack<int> le;
        memset(vis, 0, sizeof vis);
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '(') {
                le.push(i);
            } else if (s[i] == ')') {
                if (!le.empty()) {
                    le.pop();
                } else {
                    vis[i] = 1;
                }
            }
        }
        while (!le.empty()) {
            vis[le.top()] = 1;
            le.pop();
        }
        string ans = "";
        for (int i = 0; i < s.size(); i++) {
            if (!vis[i]) {
                ans += s[i];
            }
        }
        return ans;
    }
};
