class Solution {
public:
    string minRemoveToMakeValid(string s) {
        stack<int> left;
        unordered_set<int> del;
        for (int i=0; i<s.size(); ++i) {
            if (s[i] == '(') {
                left.push(i);
            } else if (s[i] == ')') {
                if (left.empty()) del.insert(i);
                else left.pop();
            }
        }
        while (!left.empty()) {
            del.insert(left.top());
            left.pop();
        }
        
        string res = "";
        for (int i=0; i<s.size(); ++i) {
            if (del.count(i) > 0) continue;
            else res += s[i];
        }
        return res;
    }
};
