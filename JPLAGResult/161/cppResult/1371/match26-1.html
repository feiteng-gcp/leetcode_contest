class Solution {
public:
    string minRemoveToMakeValid(string s) {
        set<int> v;
        stack<int> stk;
        for(int i=0;i<s.size();i++) {
            if(s[i]=='(') {
                stk.push(i);
            } else if(s[i]==')') {
                if(stk.size()>0) {
                    stk.pop();
                } else {
                    v.insert(i);
                }
            }
        }
        while(!stk.empty()) {
            int p = stk.top();
            stk.pop();
            v.insert(p);
        }
        string out;

        for(int i=0;i<s.size();i++) {
            if(v.find(i)!=v.end()) continue;
            else out+=s[i];
        }
        return out;
    }
};
