class Solution {
public:
    string minRemoveToMakeValid(string s) {
        int cnt=0;
        string res ="";
        for ( auto c:s ){
            if ( c=='(' ){
                res += '(';
                cnt++;
            }else if ( c==')' ){
                --cnt;
                if ( cnt<0 ){
                    cnt=0;
                }else{
                    res += ')';
                }
            }else res+=c;
        }
        string ret ="";
        cnt=0;
        for ( int i=(int)res.size()-1; i>=0; --i ){
            if ( res[i]==')' ){
                ret += ')';
                cnt++;
            }else if ( res[i]=='(' ){
                --cnt;
                if ( cnt<0 ){
                    cnt=0;
                }else{
                    ret += '(';
                }
            }else ret += res[i];
        }
        reverse(ret.begin(), ret.end());
        return ret;
    }
};
