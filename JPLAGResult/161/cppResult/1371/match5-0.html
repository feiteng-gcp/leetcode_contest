bool vis[100005];
class Solution {
public:
    string minRemoveToMakeValid(string s) {
        stack<int> st;
        memset(vis, 0, sizeof vis);
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '(') {
                st.push(i);
            } else if (s[i] == ')') {
                if (!st.empty()) {
                    st.pop();
                } else {
                    vis[i] = 1;
                }
            }
        }
        while (!st.empty()) {
            vis[st.top()] = 1;
           st.pop();
        }
        string ans = "";
        for (int i = 0; i < s.size(); i++) {
            if (!vis[i]) {
                ans += s[i];
            }
        }
        return ans;
    }
};
