class Solution {
public:
    int A[100100],B[100010];
    string minRemoveToMakeValid(string s) {
        stack<int>sta;
        memset(A,-1,sizeof(A));
        memset(B,-1,sizeof(B));
        for(int i=0;i<s.size();i++){
            if(s[i]=='(')sta.push(i);
            else if(s[i]==')'){
                if(!sta.empty()){
                    A[sta.top()]=i;
                    B[i]=sta.top();
                    sta.pop();
                }
            }
        }
        string ans;
        for(int i=0;i<s.size();i++){
            if(islower(s[i]))ans.push_back(s[i]);
            else if(s[i]=='('&&A[i]!=-1)ans.push_back(s[i]);
            else if(s[i]==')'&&B[i]!=-1)ans.push_back(s[i]);
        }
        return ans;
    }
};
