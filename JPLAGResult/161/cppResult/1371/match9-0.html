class Solution {
public:
    string minRemoveToMakeValid(string s) {
        vector<int> st;
        for(int i = 0; i < s.size(); ++i){
            if(s[i] == '('){
                st.push_back(i);
            }
            else if(s[i] == ')'){
                if(st.size() > 0 && s[st.back()] == '('){
                    st.pop_back();
                }
                else{
                    st.push_back(i);
                }
            }
        }
        unordered_set<int> set(begin(st), end(st));
        
        string result;
        for(int i = 0; i < s.size(); ++i){
            if(set.find(i) == end(set)){
                result.push_back(s[i]);
            }
        }
        return result;
    }
};
