class Solution {
 public:
  string minRemoveToMakeValid(string s) {
    stack<int> lefts;
    vector<bool> dels(s.size(), false);
    for (int i = 0; i < s.size(); ++i) {
      if (s[i] == '(') {
        lefts.push(i);
      } else if (s[i] == ')') {
        if (lefts.empty()) {
          dels[i] = true;
        } else {
          lefts.pop();
        }
      }
    }
    while (!lefts.empty()) {
      dels[lefts.top()] = true;
      lefts.pop();
    }
    string result;
    for (int i = 0; i < s.size(); ++i) {
      if (!dels[i]) {
        result += s[i];
      }
    }
    return result;
  }
};
