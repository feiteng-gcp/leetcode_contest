class Solution {
public:
    string minRemoveToMakeValid(string s) {
        int L = s.length();
        for (int i = 0, balance = 0; i < L; ++i) {
            if (s[i] == '(') {

                ++balance;
            } else if (s[i] == ')') {
                if (balance == 0) {
                    s[i] = ' ';   
                } else {
                    --balance;
                }
            }
        }

        for (int i = L - 1, balance = 0; i >= 0; --i) {
            if (s[i] == ')') {
                ++balance;
            } else if (s[i] == '(') {
                if (balance == 0) {
                    s[i] = ' ';   
                } else {
                    --balance;
                }
            }
        }
        string ans = "";
        for (int i = 0; i < L; ++i) {
            if (s[i] != ' ') {
                ans.push_back(s[i]);
            }
        }
        return ans;
    }
};
