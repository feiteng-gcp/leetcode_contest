class Solution {
public:
    string minRemoveToMakeValid(string s) {
        int n = s.size();
        stack<int>st;
        unordered_map<int,bool>in_valid;

        for(int i = 0; i < n; ++i){
            if(s[i] == '('){
                st.push(i);
            }
            else
            if(s[i] == ')'){
                if(!st.empty()){
                    st.pop();
                }
                else{
                    in_valid[i] = true;

                }
            }
        }
        while(!st.empty()){
            in_valid[st.top()] = true;
            st.pop();
        }
        
        string answer = "";
        for(int i = 0; i < n; ++i){
            if(in_valid.find(i) == in_valid.end()){
                answer += s[i];
            }
        }
        return answer;
        
    }
};
