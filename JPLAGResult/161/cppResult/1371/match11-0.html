class Solution {
public:
    string minRemoveToMakeValid(string s) {
        stack<int> l;
        vector<int> re;
        for(int i = 0; i < s.length(); ++i) {
            if(s[i] == '(') {
                l.push(i);
            } else if(s[i] == ')') {
                if(l.size() == 0) {
                    re.push_back(i);
                } else {
                    l.pop();
                }
            }
        }
        while(!l.empty()) {
            re.push_back(l.top());
            l.pop();
        }
        sort(re.begin(), re.end());
        string ree;
        for(int i = 0, j = 0; i < s.length(); ++i) {
            if(j < re.size() && i == re[j]) {
                j++;
            } else {
                ree += s[i];
            }
        }
        return ree;
    }
};
