class Solution {
    public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> odds;
        odds.push_back(-1);
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] % 2 == 1) {
                odds.push_back(i);
            }
        }
        odds.push_back(nums.size());
        if (odds.size() == 2) return 0;
        // -1, 0, 2, 7, 9
        int res = 0;
        for (int s = 1; s < (int) odds.size() - k; s++) {
            res += (odds[s] - odds[s-1]) * (odds[s+k] - odds[s+k-1]);
        }
        return res;
    }
};
