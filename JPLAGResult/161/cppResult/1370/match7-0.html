class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> intervals;
        int start = 0;
        for(int i = 0; i < nums.size(); ++i)
        {
            if(nums[i] & 1)
            {
                intervals.push_back(start);
                start = 0;
            }
            else
            {
                start++;
            }
        }
        intervals.push_back(start);
        int res = 0;
        for(int i = k; i < intervals.size(); ++i)
        {
            res += (intervals[i - k] + 1) * (intervals[i] + 1);
        }
        return res;
    }
};
