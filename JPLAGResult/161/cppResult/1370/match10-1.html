class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> v;
        v.push_back(-1);
        for(int i = 0; i < nums.size(); ++ i)
        {
            if(nums[i] % 2)
            {
                v.push_back(i);
            }
        }
        v.push_back(nums.size());
        

        int ans = 0;
        for(int i = 1; i < (int)v.size() - k; ++ i)
        {
            ans += (v[i] - v[i - 1]) * (v[i + k] - v[i + k - 1]);
        }
        
        return ans;
    }
};
