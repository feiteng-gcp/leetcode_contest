class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> odd;
        odd.push_back( -1 );
        for( int i = 0 ; i < nums.size() ; ++i ) {
            if( nums[i] % 2 ) {
                odd.push_back( i );
            }
        }
        odd.push_back( nums.size() );
        #if 0
        for( auto& i: odd ) {
            cout << " " << i;
        }
        #endif
        cout << endl;
        int ans = 0;
        // (i+1) ~ (i+k+1)
        for( int i = 0 ; i+k+1 < odd.size() ; ++i ) {
            //cout << "i " << i << endl;
            ans += (odd[i+1]-odd[i]) * (odd[i+k+1]-odd[i+k]);
        }
        return ans;
    }
};
