class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> odds(1,-1);
        for(int i =0; i<nums.size(); i++){
            if(nums[i]%2) odds.push_back(i);
        }
        odds.push_back(nums.size());
        if(odds.size()<k+2) return 0;

        int total = 0;
        for(int left = 1; left + k < odds.size(); left++){
            total += (odds[left] - odds[left-1]) * (odds[left+k] - odds[left+k-1]);
        }
        return total;
    }
};
