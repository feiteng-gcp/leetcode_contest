class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int N=nums.size();
        vector<int> vIndex;
        vIndex.push_back(-1);
        for(int i=0;i<N;i++){
            if((nums[i]&1)>0){
                vIndex.push_back(i);
            }

        }
        vIndex.push_back(N);
        int ans=0;
        for(int b=1;b+k<vIndex.size();b++){
            int e=b+k-1;
            //cout<<b<<" "<<e<<" "<<vIndex.size()<<endl;
            int x=vIndex[b]-vIndex[b-1];
            int y=vIndex[e+1]-vIndex[e];
            ans+=x*y;
        }
        return ans;
    }
};
