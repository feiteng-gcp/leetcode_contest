class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> odd_index_vec;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] % 2 == 1) odd_index_vec.push_back(i);
        }
        if (odd_index_vec.size() < k) return 0;
        int res = 0;
        for (int i = 0; i + k <= odd_index_vec.size(); ++i) {
            int l = i == 0 ? odd_index_vec[i] : odd_index_vec[i] - odd_index_vec[i - 1] - 1,
                r = i + k == odd_index_vec.size() ? nums.size() - odd_index_vec[i + k - 1] - 1 : odd_index_vec[i + k] - odd_index_vec[i + k - 1] - 1;
            res += l * r + l + r + 1;
        }
        return res;
    }
};
