class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> A;
        int n = 0;
        for (int x : nums) {
            if (x % 2) {
                A.push_back(n);
                n = 0;
            }
            else
                n += 1;
        }
        A.push_back(n);
        
        int result = 0;
        for (int i = 0; i + k < A.size(); ++i) {
            result += (A[i] + 1) * (A[i + k] + 1);
        }
        return result;
    }
};
