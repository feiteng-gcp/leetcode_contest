class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> v;
        int c = 0;
        for (auto n : nums) {
            if (n % 2 == 1) {
                v.push_back(c);
                c = 0;
            }
            else {
                c++;
            }
        }
        v.push_back(c);
        int res = 0;
        for (int i = 0; i + k < v.size(); i++) {
            res += (v[i] + 1) * (v[i + k] + 1);
        }
        return res;
    }
};
