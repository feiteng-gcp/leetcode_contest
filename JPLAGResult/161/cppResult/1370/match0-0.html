class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int res = 0;
        
        vector<int> ind;
        
        ind.emplace_back(0);
        
        for(int i = 0; i < nums.size(); ++i){
            if(nums[i]%2!=0)
                ind.emplace_back(i+1);
        }
        
        ind.emplace_back(nums.size()+1);
        
        int L = ind.size()-1;
        
        for(int i = 1; i+k-1 < L; ++i){
            //cout<<ind[i]<<" "<<ind[i-1]<<" "<<ind[i+k]<<" "<<ind[i+k-1]<<endl;
            res += (ind[i] - ind[i-1])*(ind[i+k]-ind[i+k-1]);
        }
        
        return res;
    }
};
