class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int N = nums.size();
        vector<int> odd(N, 0);
        for (int i = 0; i < N; ++i) {
            odd[i] = nums[i]%2;
            if (i != 0)
                odd[i] += odd[i-1]; 
        }

        unordered_map<int, int> s;
        s[0] = 1;
        int ans = 0;
        for (int i = 0; i < N; ++i) {
            ans += s[odd[i]-k];
            s[odd[i]]++;
        }
        return ans;
    }
};
