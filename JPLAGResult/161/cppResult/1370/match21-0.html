class Solution {
public:
    int numberOfSubarrays(vector<int>& a, int m) {
        int n = a.size();
        int count = 0; 
        int prefix[n] = { 0 }; 
        int odd = 0; 

        for (int i = 0; i < n; i++)  { 
            prefix[odd]++; 
            if (a[i] & 1) 
                odd++; 
            if (odd >= m) 
                count += prefix[odd - m]; 
        } 
        return count; 
    }
};
