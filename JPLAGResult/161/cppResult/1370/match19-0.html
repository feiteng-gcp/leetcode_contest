class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> v(n+1);
        for(int i=0;i<n;++i) v[i+1] = v[i] + nums[i]%2;
        map<int, int> m;
        m[0] = 1;
        int ans = 0;
        for(int i=1;i<=n;++i){
            int x = v[i] - k;
            ans += m[x];
            ++m[v[i]];
        }
        return ans;
    }
};
