class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> a(n+1);
        for(int i=0;i<n;i++)
        {
            if(nums[i]%2 == 1)
            {
                a[i+1] = 1;
            }
        }
        for(int i=1;i<=n;i++)
        {
            a[i] += a[i-1];
            //printf("%d ",a[i]);
        }
        //cout<<endl;

        int ans = 0;
        for(int i=0;i<n;i++)
        {
            int x = a[i];
            int y = x+k;
            int pos = lower_bound(a.begin(),a.end(),y)-a.begin();
            int pos1 = upper_bound(a.begin(),a.end(),y)-a.begin();
            //printf("%d %d\n",pos,pos1);
            ans += max(pos1-pos,0);
        }
        return ans;
    }
};
