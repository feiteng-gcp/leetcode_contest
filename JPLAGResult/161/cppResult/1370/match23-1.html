class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> odd;
        for(int i = 0; i < nums.size(); ++i)
        {
            if(nums[i] % 2 == 1)
            {
                odd.push_back(i);
            }
        }
        if(odd.size() < k) return 0;
        int result = 0;
        int a = 0, b = 0;
        for(int i = 0; i < odd.size() - k + 1; ++i)
        {

            if(i == 0)
            {
                a = odd[i] - 0 + 1;
            }
            else
            {
                a = odd[i] - odd[i - 1];
            }
            if(i == odd.size() - k)
            {
                b = nums.size() - odd[i + k - 1];
            }
            else
            {
                b = odd[i + k] - odd[i + k - 1];
            }
            result += a * b;
        }
        return result;
    }
};
