class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> interval;
        int cnt = 0;
        for(auto i : nums) {
            if(i%2 == 1) {
                interval.push_back(cnt);
                cnt = 0;
            } else {
                cnt++;
            }
        }
        interval.push_back(cnt);
        int rst=0;
        for(int i = 0; i + k < interval.size(); ++i) {
            // cout << interval[i] << interval[i+k] << endl;
            rst += ((interval[i]+1)*(interval[i+k]+1));
        }
        return rst;
    }
};
