class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> a;
        a.push_back(-1);
        for(int i=0;i<nums.size();i++){
            if(nums[i]%2==1)
                a.push_back(i);
        }
        a.push_back(nums.size());
        int r=0;
        for(int i=1;i+k<a.size();i++){
            r+=(a[i]-a[i-1])*(a[i+k]-a[i+k-1]);
        }
        return r;
    }
};
