class Solution {
public:
    int numberOfSubarrays(vector<int>& v, int k) {
        vector<int> oddIndex;
        for (int i = 0; i < v.size(); i++) {
            if (v[i] % 2) oddIndex.push_back(i);
        }
        int res = 0;
        for (int i = k - 1; i < oddIndex.size(); i++) {
            int leftIndex = i - k + 1;
            int a = 0;
            if (leftIndex > 0) {
                a = oddIndex[leftIndex] - oddIndex[leftIndex - 1];
            } else {
                a = oddIndex[leftIndex] + 1;
            }
            int b = 0;
            int rightIndex = i + 1;
            if (rightIndex < oddIndex.size()) {
                b = oddIndex[rightIndex] - oddIndex[i];
            } else {
                b = v.size() - oddIndex[i];
            }
            res += a * b;
        }
        return res;
    }
};
