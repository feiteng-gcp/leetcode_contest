class Solution {
public:
    int A[50050],L[50050],R[50050],_1[50050],c;
    int numberOfSubarrays(vector<int>& nums, int k) {
        int n=nums.size();
        for(int i=1;i<=n;i++){
            A[i]=nums[i-1]%2;
            if(A[i])_1[++c]=i;
        }
        for(int i=1;i<=n;i++){
            L[i]=L[i-1];
            if(A[i]==1)L[i]=i;
        }
        R[n+1]=n+1;
        for(int i=n;i>=1;i--){
            R[i]=R[i+1];
            if(A[i]==1)R[i]=i;
        }
        long long ans=0;
        for(int i=k;i<=c;i++){
            int p1=_1[i-k+1],p2=_1[i];
            ans+=1ll*(p1-L[p1-1])*(R[p2+1]-p2);
        }
        return ans;
    }
};
