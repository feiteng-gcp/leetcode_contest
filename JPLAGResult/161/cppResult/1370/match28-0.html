class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int ret = 0;
        vector<int> pos;
        pos.push_back(-1);
        for (int i = 0; i < nums.size(); ++i)
            if (nums[i]&1) pos.push_back(i);
        pos.push_back(nums.size());
        int n = pos.size();
        for (int i = 1; i+k-1 <= n-2; ++i) {
            int L = pos[i];
            int R = pos[i+k-1];
            int pre = pos[i-1];
            int nxt = pos[i+k];
            
            ret += (L-pre) * (nxt - R);
        }
        return ret;
    }
};
