class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        // 滑动窗口
        deque<int> q;
        int l=0,r=0, cur=0;
        int res = 0;
        while( r<nums.size() ){
            while( r<nums.size() && cur<k ){
                if ( nums[r]%2==1 ){
                    ++cur;
                    q.push_back(r);
                }
                ++r;
            }
            if ( r>=nums.size() && cur<k ) break;
            int left=q.front()-1;
            int right = q.back()+1;
            int lc=1, rc=1;
            while( left>=0 && nums[left]%2==0 ){
                --left; ++lc;
            }
            while( right<nums.size() && nums[right]%2==0 ){
                ++right; ++rc;
            }
            res += ( lc*rc );
            r = right;
            l = q.front()+1; q.pop_front();
            cur--;
        }
        return res;
    }
};
