class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) 
    {
        int n = nums.size();
        nums.insert(nums.begin(),0);
        vector<int>p;
        int count = 0;
        p.push_back(0);
        
        int ret = 0;
        
        for (int i=1; i<=n; i++)
        {
            if (nums[i]%2==1)
                count++;
            
            if (count>=k)
            {
                auto iter1 = lower_bound(p.begin(),p.end(),count-k);
                auto iter2 = lower_bound(p.begin(),p.end(),count-k+1);
                ret += (iter2-iter1);
                
                //cout<<i<<":"<<count<<":"<<iter1-p.begin()<<","<<iter2-p.begin()<<endl;
            }
                        
            p.push_back(count);
            
        }
        
        return ret;
    }
};
