class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> l(1, 0);
        for (int i = 0; i < nums.size(); i++)
        {
            if (nums[i] & 1)
            {
                l.push_back(0);
            }
            else
            {
                l.back()++;
            }

        }
        int ans = 0;
        for (int i = k; i < l.size(); i++)
        {
            ans += (l[i - k] + 1) * (l[i] + 1);
        }
        return ans;
    }
};
