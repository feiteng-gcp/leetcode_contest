class Solution {
public:
    int numberOfSubarrays(vector<int>& A, int k) {
      int n = A.size();
      vector<int> prev(A.size() + 1);
      for (int i = 0; i < n; ++i) prev[i + 1] = prev[i] + A[i] % 2;
      map<int, int> cnt;
      cnt[0] = 1;
      int ans = 0;
      for (int r = 1; r <= n; ++r) {
        ans += cnt[prev[r] - k];
        ++cnt[prev[r]];
      }
      return ans;
    }
};
