class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int res = 0, cnt = 0;
        vector<int> v;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i]%2 == 0) ++cnt;
            if (nums[i]%2 == 1) {
                v.push_back(cnt+1);
                cnt = 0;
            }
        }
        v.push_back(cnt+1);
        // for (int i : v) cout << i << " ";
        for (int i = 0; i+k < v.size(); ++i) {
            res += v[i]*v[i+k];
        }
        return res;
    }
};
