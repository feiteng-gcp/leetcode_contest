class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        
        vector<int> oddIndex;
        for (int i = 0; i < nums.size(); i++)
        {
            if (nums[i] % 2 == 1)
            {
                oddIndex.push_back(i);
            }
        }
        
        int result = 0;
        
        for (int start = 0; start + k - 1 < oddIndex.size(); start++)
        {

            int sn = 1;
            if (start > 0)
            {
                sn = oddIndex[start] - oddIndex[start-1];
            }
            else
            {
                sn = oddIndex[start] + 1;
            }
            int end = start + k - 1;
            int en = 1;
            if (end + 1 < oddIndex.size())
            {
                en = oddIndex[end + 1] - oddIndex[end];
            }
            else
            {
                en = nums.size() - oddIndex[end];
            }
            result = result + sn * en;
        }
        
        return result;
    }
};
