class Solution {
public:

    int numberOfSubarrays(const vector<int>& nums, const int k) {
        vector<int> evens;
        int tmp = 0;
        for (int num: nums) {
            if (num % 2 == 1) {
                evens.push_back(tmp);
                tmp = 0;
            } else {
                tmp++;
            }
        }
        evens.push_back(tmp);
        int res = 0;
        for (int i = 0; i + k < evens.size(); ++i) {
            res += (evens.at(i) + 1) * (evens.at(i + k) + 1);
        }
        return res;
    }
};
