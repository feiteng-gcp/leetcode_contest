class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        int r = 0, rr = 0;
        vector<int> a, b;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] % 2) {
                a.push_back(i);
                b.push_back(r);
                r = 0;

            } else {
                r++;
            }
        }
        b.push_back(r);
        for (int i = 0; i + k <= a.size(); ++i) {
            rr += (b[i] + 1) * (b[i + k] + 1);
        }
        return rr;
    }
};
