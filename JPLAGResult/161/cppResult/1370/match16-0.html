class Solution {
public:
    int numberOfSubarrays(vector<int>& nums, int k) {
        vector<int> sp;
        int cnt = 0;
        for (int n : nums) {
            if (n&1) {
                sp.push_back(cnt + 1);
                cnt = 0;
            } else {
                ++ cnt;
            }
        }
        sp.push_back(cnt + 1);
        int sum = 0;
        for (int i=k;i<sp.size();++i) {
            sum += sp[i] * sp[i-k];
        }
        return sum;
    }
};
