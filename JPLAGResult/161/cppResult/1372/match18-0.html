class Solution {
public:
    
    //x = 4, y = 3
    //5, 7, 12, 23
    
    //5x+y=1
    //y=1+5x
    //7, 14, 21
    
    //5x+y=1
    //y= 1+5x
    //
    
    //6, 20, 29
    //6x = 1
    //20x = 1+6y = 7, 13, 19, 25, 31, 37, 43: 7, 13, 19, 5, 11, 
    
    //3x = 1
    //6x = 1 + 3y, 1, 4, 7, 10, 13, 16, 19: 1, 4, 1, 4, 1, 4
    
    //5, 7
    //(1,-2), -9
    //(1,-1), -2
    //(1, 0), 5
    //(1, 1), 12
    //(1, 2), 19
    
    bool isGoodArray(vector<int>& nums) {
        int n = nums.size();
        int g = nums[0];
        for(int i = 1; i < n; ++i) {
            g = __gcd(g, nums[i]);
        };
        return g == 1;
    }
};
