class Solution {
public:
    bool isGoodArray(vector<int>& a) {
        if(a.size()==1 && a[0]==1)
            return true;
        int g = a[0];
        for(int i=1;i<a.size();i++){
            g = __gcd(g, a[i]);
            if(g==1) return true;
        }
        return false;
    }
};
