class Solution {
    public String minRemoveToMakeValid(String s) {
        StringBuilder sb = new StringBuilder();
        char[] schars = s.toCharArray();
        int cnt = 0;
        for (char c : schars) {
            if (c=='(') {
                cnt++;
                sb.append(c);
            } else if (c==')') {
                if (cnt>0) {
                    cnt--;
                    sb.append(c);
                }
            } else {
                sb.append(c);
            }
        }
        String s2 = sb.reverse().toString();
        sb.setLength(0);
        char[] schars2 = s2.toCharArray();
        cnt = 0;
        for (char c : schars2) {
            if (c==')') {
                cnt++;
                sb.append(c);
            } else if (c=='(') {
                if (cnt>0) {
                    cnt--;
                    sb.append(c);
                }
            } else {
                sb.append(c);
            }
        }
        return sb.reverse().toString();
    }
}
