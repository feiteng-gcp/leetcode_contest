class Solution {
    public String minRemoveToMakeValid(String s) {
        List<Integer> list = new ArrayList<>();
        Stack<Integer> sta = new Stack<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == ')') 
{
                if (!sta.isEmpty() && s.charAt(sta.peek()) == '(') {
                    sta.pop();
                } else {
                    sta.push(i);
                }
            } else if (c == '(') {
                sta.push(i);
            }
        }
        int len = s.length();
        boolean[] ck = new boolean[len];
        
while (!sta.isEmpty()) {
            ck[sta.pop()] = true;
        }
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < len; i++) 
{
            if (ck[i] == false) {
                ans.append(s.charAt(i));
            }
        }
        return ans.toString();
    }
}
