class Solution {
  public String minRemoveToMakeValid(String s) {
    StringBuilder sb = new StringBuilder();
    
    int index = 0;
    for (char c : s.toCharArray()) {
      if (c == '(') {
        index++;
      } else if (c == ')') {
        index--;
      
}
      
      if (index < 0) {
        index = 0;
      } else {
        sb.append(c);
      }
    }
    
    while (index > 0) {
      sb.deleteCharAt(sb.lastIndexOf("("));
      index--;
    }
    
    return sb.toString();
  }
}
