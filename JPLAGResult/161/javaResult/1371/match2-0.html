class Solution {
    public String minRemoveToMakeValid(String s) {
		StringBuilder sb = new StringBuilder();
		int pCount = 0;
		for (char c : s.toCharArray()) {
			if (c != '(' && c != ')') {
				sb.append(c);
			} else if (c == ')') {
				if (pCount > 0) {
					pCount--;
					sb.append(c);
				}
			} else {
				pCount++;
				sb.append(c);
			}
		}
		s = sb.toString();
		sb = new StringBuilder();
		pCount = 0;
		for (int i = s.length() - 1; i >= 0; i--) {
			char c = s.charAt(i);
			if (c != '(' && c != ')') {
				sb.append(c);
			} else if (c == '(') {
				if (pCount > 0) {
					pCount--;
					sb.append(c);
				}
			} else {
				pCount++;
				sb.append(c);
			}
		}
		return sb.reverse().toString();
    }
}
