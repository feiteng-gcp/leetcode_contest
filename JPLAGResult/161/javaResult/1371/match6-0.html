class Solution {
    public String minRemoveToMakeValid(String s) {
		StringBuilder stringBuilder = new StringBuilder("");
		char[] arr = s.toCharArray();
		int left = 0;
		for(char x : arr) {
			if(x=='(') {
				left++;
				stringBuilder.append(x);
			}
			else if(x==')') {
				if(left>0) {				
					left--;
					stringBuilder.append(x);
				}
			}
			else {stringBuilder.append(x);}
		}
		String temp = stringBuilder.toString();
		
		StringBuilder stringBuilder2 = new StringBuilder("");
		char[] arr2 = temp.toCharArray();
		int right = 0;
		for(int index=arr2.length-1; index>=0; index--) {
			if(arr2[index]==')') {
				right++;
				stringBuilder2.append(arr2[index]);
			}
			else if(arr2[index]=='(') {
				if(right>0) {				
					right--;
					stringBuilder2.append(arr2[index]);
				}
			}
			else {stringBuilder2.append(arr2[index]);}
		}
		String ans = stringBuilder2.toString();
		
		
		
		StringBuilder stringBuilder3 = new StringBuilder("");
		char[] arr3 = ans.toCharArray();
		for(int index=arr3.length-1; index>=0; index--) {
			stringBuilder3.append(arr3[index]);
		}
		String ans2 = stringBuilder3.toString();

		
		return ans2;
				  
     }

}
