class Solution {
    public String minRemoveToMakeValid(String s) {
        Stack<Integer> loc = new Stack<>();
        boolean[] shouldRemove = new boolean[s.length()];
        int i = 0;
        for(char c : s.toCharArray()){
            if(c == '('){
                loc.push(i);
            }
            else if(c == ')'){
                if(loc.isEmpty()){
                    shouldRemove[i] = true;
                }
                else{
                    loc.pop();
                }
            
}
            i++;
        }
        
        while(!loc.isEmpty()){
            shouldRemove[loc.pop()] = true;
        }
        
        StringBuilder res = new StringBuilder();
        for(i = 0; i < 
s.length(); i++){
            if(!shouldRemove[i]){
                res.append(s.charAt(i));
            }
        }
        
        return res.toString();
    }
}
