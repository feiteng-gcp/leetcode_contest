class Solution {
    public String minRemoveToMakeValid(String s) {
        StringBuilder ans = new StringBuilder();
        int num = 0;
        for (char c : s.toCharArray()) {
            
if (c=='(') {
                ++num;
                ans.append(c);
            } else if (c==')') {
                if (num > 0) {
                    --num;
                    ans.append(c);
                }
            } else {
                ans.append(c);
            }
        }
        s = ans.reverse().toString();
        ans = new StringBuilder();
        num = 0;
        for (char c : s.toCharArray()) {
            
if (c==')') {
                ++num;
                ans.append(c);
            } else if (c=='(') {
                if (num > 0) {
                    --num;
                    ans.append(c);
                }
            } else {
                ans.append(c);
            }
        }
        return ans.reverse().toString();
    }
}
