class Solution {
    public String minRemoveToMakeValid(String s) {
        int n = s.length();
        boolean[] flags = new boolean[n];
        Stack<Integer> stk = new Stack<Integer>();
        char str[] = s.toCharArray();
        for (int i = 0; i < n; i++) {
            
if (str[i] == ')') {
                if (stk.isEmpty()) {
                    flags[i] = true;
                } else {
                    stk.pop();
                }
            } else if (str[i] == '(') {
                stk.push(i);
            
}
        }
        
        while(!stk.isEmpty()) {
            flags[stk.pop()] = true;
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (!flags[i]) {
                
sb.append(str[i]);
            }
        }
        return sb.toString();
    }
}
