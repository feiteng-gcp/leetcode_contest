class Solution {
    public String minRemoveToMakeValid(String s) {
        StringBuilder sb = new StringBuilder();
        int delta = 0;
        for (char c : s.toCharArray()) 
{
            if (c == '(') {
                delta++;
            } else if (c == ')') {
                if (delta == 0) {
                    continue;
                }
                delta--;
            }
            sb.append(c);
        }
        String t = sb.toString();
        sb = new StringBuilder();
        delta=0;
        
for (int i = t.length() - 1; i >= 0; i--) {
            char c = t.charAt(i);
            if (c == ')') {
                delta++;
            } else if (c == '(') {
                if (delta == 0) {
                    continue;
                }
                delta--;
            }
            sb.append(c);
        }
        String ret = sb.reverse().toString();
        return ret;
    }
}
