class Solution {
    public String minRemoveToMakeValid(String s) {
        char[] chars = s.toCharArray();

        int n = 0;
        for (int i = 0; i < chars.length; i++) {
            if (chars[i] == '(' || chars[i] == ')') {
                n++;
            }
        }

        if (n == 0) return s;

        int[] pos = new int[n];


        for (int i = 0, j = 0; i < chars.length; i++) {
            if (chars[i] == '(' || chars[i] == ')') {
                pos[j++] = i;
            }
        }

        Stack<Integer> stack = new Stack<>();


        boolean[] vst = new boolean[chars.length];
        
for (int i = 0; i < n; i++) {
            if (chars[pos[i]] == ')') {
                if (stack.isEmpty()) {
                    vst[pos[i]] = true;
                } else {
                    stack.pop();
                }
            } else {
                stack.push(pos[i]);
            
}
        }

        while (!stack.isEmpty()) {
            vst[stack.pop()] = true;
        }

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < chars.length; i++) {
            if (!vst[i]) {
                sb.append(chars[i]);
            }
        }
        return sb.toString();
    }
}
