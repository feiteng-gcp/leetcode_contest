class Solution {
    public String minRemoveToMakeValid(String s) {
        List<Integer> invalidRight = new ArrayList<>();
        Stack<Integer> invalidLeft = new Stack<>();
        char[] arr = s.toCharArray();
        for (int i = 0; i < arr.length; ++i) {
            if (arr[i] == '(') {
                invalidLeft.push(i);
            } else if (arr[i] == ')') {
                if (invalidLeft.isEmpty()) {
                    invalidRight.add(i);
                } else {
                    invalidLeft.pop();
                }
            }
        }
        while (!invalidLeft.isEmpty()) {
            invalidRight.add(invalidLeft.pop());
        }
        Collections.sort(invalidRight);
        StringBuilder stringBuilder = new StringBuilder(s);
        for (int i = invalidRight.size() - 1; i >= 0; --i) {
            stringBuilder.deleteCharAt(invalidRight.get(i));
        }
        return stringBuilder.toString();
    }
}
