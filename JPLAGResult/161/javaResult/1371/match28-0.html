class Solution {
    public String minRemoveToMakeValid(String s) {
        StringBuilder b = new StringBuilder();
        int lCnt = 0, rCnt = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '(') lCnt++;
            else if (s.charAt(i) == ')') rCnt++;
            if (lCnt >= rCnt) b.append(s.charAt(i));
            else {
                rCnt--;
            }
        }
        s = b.toString();
        //System.out.println(s);
        b = new StringBuilder();
        lCnt = 0; rCnt = 0;
        for (int i = s.length()-1; i >= 0; i--) {
            if (s.charAt(i) == '(') lCnt++;
            else if (s.charAt(i) == ')') rCnt++;
            if (rCnt >= lCnt) b.append(s.charAt(i));
            else {
                lCnt--;
            }
        }
        
        return b.reverse().toString();
    }
}
