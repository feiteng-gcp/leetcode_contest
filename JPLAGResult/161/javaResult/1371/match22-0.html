class Solution {
    public String minRemoveToMakeValid(String s) {
        if(s == null || s.length() == 0) {
            return "";
        }
        StringBuilder sb = new StringBuilder();
        int left = 0;
        char[] arr = s.toCharArray();
        for(char c: arr) {
            if(c == '(') {
                left++;
            } else if(c == ')') {
                if(left == 0) {
                    continue;
                } 
                left--;
            }
            sb.append(c);
        }
        s = sb.toString();
        sb = new StringBuilder();
        int right = 0;
        for(int i = s.length() - 1; i >= 0; i--) {
            char c = s.charAt(i);
            if(c == ')') {
                right++;
            } else if(c == '(') {
                if(right == 0) {
                    continue;
                } 
                right--;
            }
            sb.append(c);
        }
        return sb.reverse().toString();
    }
}
