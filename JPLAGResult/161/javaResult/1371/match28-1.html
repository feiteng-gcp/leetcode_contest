class Solution {
	public String minRemoveToMakeValid(String s) {
		StringBuffer sb = new StringBuffer();
		int cnt = 0;

		for (int i = 0; i < s.length(); i++) {
			if (s.charAt(i) == '(')
				cnt++;
			else if (s.charAt(i) == ')')
				cnt--;
			if (cnt >= 0)
				sb.append(s.charAt(i));
			else
				cnt++;
		}

		sb.reverse();
		s = sb.toString();
//		Helper.tr(s);
		sb.setLength(0);
		cnt = 0;

		for (int i = 0; i < s.length(); i++) {
			if (s.charAt(i) == ')')
				cnt++;
			else if (s.charAt(i) == '(')
				cnt--;
			if (cnt >= 0)
				sb.append(s.charAt(i));
			else
				cnt++;
		}
		sb.reverse();
		return sb.toString();
	}
}
