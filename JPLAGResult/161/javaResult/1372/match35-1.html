    public boolean isGoodArray(int[] nums) {
        if(nums.length == 1) {
            return nums[0] == 1 || nums[0] == -1;
        }
        Arrays.sort(nums);
        if(nums[0] == 1) {
            return true;
        }
        return findGCD(nums) == 1;
    }
    
private int findGCD(int nums[]) {
        int gcd = nums[0]; 
        for (int i = 1; i < nums.length; i++) {
            gcd = gcd(nums[i], gcd);
            if(gcd == 1) {
                return 1;
            }
        
}
        return gcd; 
    } 
    private int gcd(int a, int b) { 
        if (a == 0) {
            return b; 
        }
        return gcd(b % a, a); 
    }
