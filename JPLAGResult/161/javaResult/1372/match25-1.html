class Solution {
    public boolean isGoodArray(int[] nums) {
        int t = nums[0];
        for(int i = 1; i < nums.length; ++i) {
            t = gcd(t, nums[i]);
            if(t == 1) return true;
        
}
        return t == 1;
    }
    int gcd(int x, int y) {
        if(x < y) return gcd(y, x)
;
        while(x % y != 0) {
            int z = x % y;
            x = y;
            y = z;
        }
        return y;
    }
}
