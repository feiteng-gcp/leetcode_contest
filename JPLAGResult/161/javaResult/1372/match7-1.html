class Solution {
    public boolean isGoodArray(int[] nums) {
        int label = nums[0];
        if(label==1)return true;
        for(int i = 1; i < nums.length; i++){
            label = helper(label,nums[i]);
            if(label==1)return true;
        }
        return false;
    }
    
    public int helper(int a, int b){
        while(0!=b){
            int temp = a;
            a = b;
            b = temp%b;
        }
        return a;
    }
}
