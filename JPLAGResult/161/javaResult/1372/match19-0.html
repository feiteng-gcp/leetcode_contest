class Solution {
    public boolean isGoodArray(int[] nums) {
        int last = nums[0];
        for (int i = 1; i < nums.length; i++) {
            last = gcd(last, nums[i]);
            if (last == 1) {
                return true;
            }
        }
        return last == 1;
    }

    private int gcd(int a, int b) {
        if (a < b) {
            return gcd(b, a);
        }

        if (b == 0) {
            return a;
        }

        return gcd(b, a % b);
    }
}
