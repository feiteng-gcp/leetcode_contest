class Solution {
    public boolean isGoodArray(int[] nums) {
        int res = findGCD(nums, nums.length);
        return res == 1;
    }
    private int findGCD(int arr[], int n) { 
        int result = arr[0]; 
        for (int i = 1; i < n; i++) 
            result = gcd(arr[i], result); 
  
        return result; 
    } 
    private int gcd(int a, int b) 
    { 
        if (a == 0) 
            return b; 
        return gcd(b % a, a); 
    } 
}
