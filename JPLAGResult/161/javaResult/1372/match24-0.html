class Solution {
    public boolean isGoodArray(int[] nums) {
        int curGCD = nums[0];
        if(curGCD == 1)
            return true;
        int n = nums.length;
        for(int i = 1; i < n; i++){
            curGCD = GCD(curGCD, nums[i]);
            if(curGCD == 1)
                return true;
        }
        return false;
    }
    
    public int GCD(int a, int b) {
        if (b==0) return a;
        return GCD(b,a%b);
    }
}
