class Solution {
    public boolean isGoodArray(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        
        
int gcd = nums[0];
        for (int i = 1; i < n; i++) {
            gcd = helper(gcd, nums[i]);
        }
        
        return gcd == 1;
    }
    
    public int helper(int n1, int n2) {
        if (n1 < n2) return helper(n2, n1);
        
        if (n1 % n2 == 0) return n2;
        
        return helper(n2, n1 % n2);
    }
}
