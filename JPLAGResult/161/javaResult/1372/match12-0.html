class Solution {
    public boolean isGoodArray(int[] nums) {
        int num=nums[0];
        for(int i=1;i<nums.length;i++){
            num=gcd(num,nums[i]);
        }
        return num==1;
    }
    
    public int gcd(int x, int y){
        if(y%x==0) return x;
        return gcd(y,x%y);
    }
}
