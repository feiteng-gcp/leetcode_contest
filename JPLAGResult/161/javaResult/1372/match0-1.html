class Solution {
    private int gcd(int x, int y) {
        if (y == 0)
            return x;
        return gcd(y, x%y);
    }
    public boolean isGoodArray(int[] nums) {
        int x = nums[0];
        for (int i=1; i<nums.length; i++)
            x = gcd(x, nums[i]);
        return x == 1;
    }
}
