class Solution {
    
public int gcd(int a, int b) {
        if (b == 0) return a;
        return gcd(b, a % b);
    }
    
public boolean isGoodArray(int[] nums) {
        int d = nums[0];
        for (int i = 1; i < nums.length; i++) d = gcd(d, nums[i]);
        return d == 1;
    }
}
