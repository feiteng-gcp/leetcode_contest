class Solution {
    public boolean isGoodArray(int[] nums) {
        int gcd = nums[0];
        for (int n : nums) {
            gcd = gcd(gcd, n);
            if (gcd == 1) return true;
        }
        return false;
    
}
    private int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
}
