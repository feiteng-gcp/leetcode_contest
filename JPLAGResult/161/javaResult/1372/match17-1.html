class Solution {
    public boolean isGoodArray(int[] nums) {
        if(nums.length == 1) return nums[0] == 1 || nums[0] == -1;
        int g = gcd(nums[0], nums[1]);
        
if(g == 1) return true;
        for(int i = 2; i < nums.length; i++){
            g = gcd(g, nums[i]);
            if(g == 1)
                return true;
        }
        return false;
    }
    int gcd(int m, int n){
        while(m % n != 0){
            int p = m % n;
            m = n;
            n = p;
        }
        return n;
    }
}
