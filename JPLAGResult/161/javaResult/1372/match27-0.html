class Solution {
    public boolean isGoodArray(int[] nums) {
        int g = nums[0];
        for (int x : nums) {
            g = gcd(g, x);
        }
        return g == 1;
    }
    
    public int gcd(int a, int b) {
        if (b == 0) {
            return a;
        }
        return gcd(b, a%b);
    }
}
