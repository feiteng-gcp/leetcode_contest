class Solution {
    public boolean isGoodArray(int[] nums) {
        int g = nums[0];
        for (int i = 1; i < nums.length && g != 1; i++) {
            g = gcd (g, nums[i]);
        }
        return g == 1;
    }
    int gcd(int a, int b) {
        if (a < b) return gcd(b,a);
        if (a % b == 0) return b;
        return gcd (b, a % b);
    }
}
