class Solution {
    public boolean isGoodArray(int[] nums) {
        int res = nums[0];
        for (int i: nums) {
            res = gcd(res, i);
        }
        return res == 1;
    }
    
    private int gcd(int a, int b) {
        if (a < b) {
            int t = a;
            a = b;
            b = t;
        }
        while (a % b != 0) {
            int t = b;
            b = a % b;
            a = t;
        }
        return b;
    }
}
