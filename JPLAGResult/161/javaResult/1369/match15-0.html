class Solution {
    public int minimumSwap(String s1, String s2) {
        char[] arr1 = s1.toCharArray();
        char[] arr2 = s2.toCharArray();
        int numX = 0;
        int numY = 0;
        for (int i = 0; i < arr1.length; ++i) {
            if (arr1[i] != arr2[i]) {
                if (arr1[i] == 'x') numX++;
                else numY++;
            }
        }
        if ((numX +numY) % 2 == 0) {
            return (numX / 2) + (numY / 2) + (numX%2 +numY%2);
        } else {
            return -1;
        }
    }
}
