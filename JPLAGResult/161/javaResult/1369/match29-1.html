class Solution {
    public int minimumSwap(String s1, String s2) {
        if (s1.length() != s2.length()) return -1;
        int xCnt = 0, yCnt = 0;
        int xxCnt = 0, yyCnt = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                if (s1.charAt(i) == 'x') xCnt++;
                else yCnt++;
            }else {
                if (s1.charAt(i) == 'x') xxCnt++;
                else yyCnt++;
            }
        
}
        if (xCnt%2==0 && yCnt%2 == 0) return xCnt/2+yCnt/2;
        if (xCnt%2==0 || yCnt%2==0) return -1;
        return (xCnt+yCnt)/2+1;
    }
}
