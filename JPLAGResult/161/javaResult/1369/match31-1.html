class Solution {

    public int minimumSwap(String s1, String s2) {
        int xCount = 0;
        int yCount = 0;

        int n = s1.length();

        for (int i = 0; i < n; i++) {
            if (s1.charAt(i) == 'x') xCount++;
            else yCount++;

            if (s2.charAt(i) == 'x') xCount++;
            else yCount++;
        }

        if (xCount %2 != 0 || yCount %2 != 0) return -1;

        int type1 = 0;
        int type2 = 0;

        for (int i = 0; i < n; i++) {
            char a = s1.charAt(i);
            char b = s2.charAt(i);

            
if (a == b) continue;

            if (a == 'x') type1++;
            else type2++;
        }

        return (type1 / 2) + (type1 % 2) + (type2 / 2) + (type2 % 2);
    }
}
