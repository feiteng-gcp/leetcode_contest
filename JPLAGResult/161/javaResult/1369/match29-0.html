class Solution {
    
    public int minimumSwap(String s1, String s2) {
        
        if(s1.length()!=s2.length())
            return -1;
        
        int freq[]=new int[2];
        
        for(int i=0; i<s1.length(); i++)
            if(s1.charAt(i)!=s2.charAt(i))
                freq[s1.charAt(i)-'x']+=1;
        
        int cnt=0;
        for(int i=0; i<2; i++) {
            cnt+=freq[i]/2;
            freq[i]%=2;
        }
        
        if(freq[0]==0 && freq[1]==0)
            return cnt;
        if(freq[0]==1 && freq[1]==1)
            return cnt+2;
        return -1;
    }
}
