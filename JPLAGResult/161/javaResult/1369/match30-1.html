class Solution {
  public int minimumSwap(String s1, String s2) {
    if (s1.length() != s2.length()) {
      return -1;
    }

    
char[] a1 = s1.toCharArray();
    char[] a2 = s2.toCharArray();
    
    int xCount = 0;
    int yCount = 0;
    for (int i = 0; i < a1.length; i++) {
      if (a1[i] == 'x') {
        xCount++;
      } else {
        yCount++;
      
}
      if (a2[i] == 'x') {
        xCount++;
      } else {
        yCount++;
      }
    }
    
    if (xCount % 2 == 1) {
      return -1;
    }
    
    int count = 0;
    for (int i = 0; i < a1.length; i++) {
      if (a1[i] != a2[i]) {
        int swapPos = i;
        for (int j = i + 1; j < a1.length; j++) {
          if (a1[j] == a1[i] && a2[i] == a2[j]) {
            swapPos = j;
            break;
          }
        }
        char temp = a2[i];
        a2[i] = a1[swapPos];
        a1[swapPos] = temp;
        i--;
        count++;
      }
    }
    return count;
  }
}
