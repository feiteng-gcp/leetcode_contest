class Solution {
    public int minimumSwap(String s1, String s2) {
        if(s1.length() != s2.length())
            return -1;
        int n = s1.length();
        int xy = 0;
        int yx = 0;
        for(int i = 0; i < n; i++)
{
            if(s1.charAt(i) == 'x' && s2.charAt(i) == 'y')
                xy++;
            else if(s1.charAt(i) == 'y' && s2.charAt(i) == 'x')
                yx++;
        }
        int c = xy/2 + yx/2;
        int remxy = xy % 2;
        int remyx = yx % 2;
        
if(remxy == 1 && remyx == 1)
            return c + 2;
        else if(remxy == 1 || remyx == 1)
            return -1;
        else
            return c;
    }
}
