class Solution {
    public int minimumSwap(String s1, String s2) {
        int x1 = 0, x2 = 0, y1 = 0, y2 = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) == 'x' && s2.charAt(i) == 'y') {
                x1++;
                y2++;
            } else if (s2.charAt(i) == 'x' && s1.charAt(i) == 'y') {
                x2++;
                y1++;
            }
        }
        if ((x1 + x2) % 2 == 1) {
            return -1;
        }
        return x1 / 2 + x2 / 2 + (x1 % 2 * 2);
    }
}
