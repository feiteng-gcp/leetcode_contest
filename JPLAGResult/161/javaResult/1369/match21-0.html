class Solution {
    public int minimumSwap(String s1, String s2) {
        int x=0, y=0;
        for(int i=0;i<s1.length();i++){
            char c1=s1.charAt(i),c2=s2.charAt(i);
            if(c1!=c2){
                if(c1=='x'){
                    x++;
                }else{
                    y++;
                }
            }
        }
        
        int ans=0;
        ans+=x/2;
        ans+=y/2;
        x%=2;
        y%=2;
        if(x+y==1) return -1;
        return ans+x+y;
    }
}
