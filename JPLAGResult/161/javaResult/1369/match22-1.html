class Solution {
    public int minimumSwap(String s1, String s2) {
        int n = s1.length();
        if (
s2.length() != n) {
            return 0;
        }
        
        int xCnt = 0;
        int yCnt = 0;
        for (int i = 0; i < n; i++) 
{
            char ch1 = s1.charAt(i);
            char ch2 = s2.charAt(i);
            if (ch1 != ch2) {
                if (ch1 == 'x') {
                    xCnt++;
                } else {
                    yCnt++;
                }
            }
        }
        int allCnt = xCnt+yCnt;
        if (allCnt%2 != 0) {
            return -1;
        } else {
            int ans = xCnt/2 + yCnt/2;
            if (xCnt%2 == 1 && yCnt%2 == 1) {
                ans += 2;
            }
            return ans;
        }
    }
}
