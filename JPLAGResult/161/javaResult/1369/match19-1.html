class Solution {
    public int minimumSwap(String s1, String s2) {
        if (s1.length() != s2.length()) {
            return -1;
        }
        int cnt1 = 0, cnt2 = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                if (s1.charAt(i) == 'x') 
{
                    cnt1++;
                } else {
                    cnt2++;
                }
            }
        }
        if (((cnt1 + cnt2) & 1) == 1) {
            return -1;
        } else {
            if ((cnt1 & 1) == 0) {
                return (cnt1 + cnt2) >> 1;
            } else {
                return ((cnt1 + cnt2) >> 1) + 1;
            }
        }
    }
}
