class Solution {
    public int minimumSwap(String s1, String s2) {
        if(s1.length() != s2.length()){
            return -1;
        }
        
        int xy = 0;
        int yx = 0;
        for(int i = 0; i < s1.length(); i++){
            char c1 = s1.charAt(i);
            char c2 = s2.charAt(i);
            if(c1 != c2){
                if(c1 == 'x'){
                    xy++;
                }
                else{
                    yx++;
                }
            
}
        }
        
        if(((xy + yx) & 1) == 1){
            return -1;
        }
        
        int count = 0;
        if((xy & 1) == 1 && (yx & 1) == 1){
            count += 2;
        }
        
        count += (xy >> 1);
        count += (yx >> 1);
        return count;
    }
}
