class Solution {
    public int minimumSwap(String s1, String s2) {
        int xy = 0;
        int yx = 0;
        if(s1.length() != s2.length()) {
            return -1;
        }
        for (int i = 0; i < s1.length(); i++) {
            if(s1.charAt(i) == s2.charAt(i)) {
                continue;
            }
            if(s1.charAt(i) == 'x' && s2.charAt(i) == 'y') {
                xy++;
            } else if(s1.charAt(i) == 'y' && s2.charAt(i) == 'x') {
                yx++;
            }
        }
        if((xy + yx) % 2 != 0) {
            return -1;
        }
        int total = 0;
        total += xy / 2 + yx / 2;
        total += xy % 2 + yx % 2;
        return total;
    }
}
