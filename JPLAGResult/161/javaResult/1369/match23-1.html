class Solution {
    public int minimumSwap(String s1, String s2) {
        int[] record = new int[]{0, 0};
        for(int i = 0; i < s1.length(); i++) {
            if(s1.charAt(i) != s2.charAt(i)) {
                if(s1.charAt(i) == 'x') 
{
                    record[0]++;
                } else {
                    record[1]++;
                }
            }
        }
        if((record[0] + record[1]) % 2 == 1) {
            return -1;
        }
        int res = 0;
        res += record[0] / 2;
        res += record[1] / 2;
        if(record[0] % 2 == 1) {
            res += 2;
        }
        return res;
    }
}
