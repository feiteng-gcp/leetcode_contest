class Solution {
    public int minimumSwap(String s1, String s2) {
        
int m = 0, n = 0;
        for (int i = 0; i < s2.length(); i++) {
            if (s1.charAt(i) == 'x' && s2.charAt(i) == 'y') m++;
            else if (s1.charAt(i) == 'y' && s2.charAt(i) == 'x') n++;
        }
        
        if ((m + n) % 2 == 1) return -1;
        return (m + 1) / 2 + (n + 1) / 2;
    }
}
