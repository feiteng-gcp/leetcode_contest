class Solution {
    public int minimumSwap(String s1, String s2) {
        char[] arr1 = s1.toCharArray();
        char[] arr2 = s2.toCharArray();
        
        int xy = 0, yx = 0;
        for(int i = 0; i < arr1.length; i++) {
            if(arr1[i] != arr2[i]) {
                if(arr1[i] == 'x') xy++; else yx++;
            }
        }
        
        int ret = xy / 2 + yx / 2;
        if((xy + yx) % 2 == 1) return -1;
        if(xy % 2 == 1) ret += 2;
        return ret;
    }
}
