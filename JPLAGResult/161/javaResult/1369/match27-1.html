class Solution {
    public int minimumSwap(String s1, String s2) {
        int N = s1.length();
        int c1 = 0;
        int c2 = 0;
        for (int i = 0; i < N; i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                if (s1.charAt(i) == 'x') {
                    c1++;
                }
                if (s2.charAt(i) == 'x') {
                    c2++;
                }
            }
        }

        int cnt = c1 + c2;
        if (cnt % 2 == 1) {
            return -1;
        } else 
if(c1%2==0){
            return cnt / 2;
        } else {
            return cnt/2+1;
        }
    }

    public static void main(String[] args) {
        System.out.println(new Solution().minimumSwap("xxyyxyxyxx", "xyyxyxxxyx"));
    }
}
