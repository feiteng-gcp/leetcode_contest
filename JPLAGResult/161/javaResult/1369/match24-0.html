class Solution {
    public int minimumSwap(String s1, String s2) {
        if (s1.length() != s2.length()) return -1;
        int res = 0;
        int c_xy = 0, c_yx = 0;
        for (int i=0; i<s1.length(); i++) {
            char x = s1.charAt(i);
            char y = s2.charAt(i);
            if (x == y) continue;
            if (x == 'x') c_xy++;
            else c_yx++;
        }
        res += c_xy / 2 + c_yx / 2;
        c_xy %= 2;
        c_yx %= 2;
        if (c_xy != c_yx) return -1;
        res += c_xy + c_yx;
        return res;
    }
}
