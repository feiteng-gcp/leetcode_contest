class Solution {
    public int minimumSwap(String s1, String s2) {
		int xy = 0;
		int yx = 0;
		for (int i = 0; i < s1.length(); i++) {
			if (s1.charAt(i) == 'x' && s2.charAt(i) == 'y') {
				xy++;
			} else if (s1.charAt(i) == 'y' && s2.charAt(i) == 'x') {
				yx++;
			}
		}
		int total = 0;
		total += xy / 2;
		total += yx / 2;
		xy = xy % 2;
		yx = yx % 2;
		if (xy == 1 && yx == 1) {
			return total + 2;
		} else if (xy == 1 || yx == 1) {
			return -1;
		} else {
			return total;
		}
    }
}
