class Solution {
    public int minimumSwap(String s1, String s2) {
        if (s1.length() != s2.length()) return -1;
        int cnt1 = 0, cnt2 = 0;
        for (char ch :
                s1.toCharArray()) {
            if (ch == 'x') cnt1++;
            else cnt2++;
        }
        for (char ch :
                s2.toCharArray()) {
            if (ch == 'x') cnt1++;
            else cnt2++;
        }

        if (cnt1 % 2 == 1 || cnt2 % 2 == 1) return -1;

        int sum1 = 0, sum2 = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) == 'x' && s2.charAt(i) == 'y')
                sum1++;
            else if (s1.charAt(i) == 'y' && s2.charAt(i) == 'x')
                sum2++;
        }

        return sum1 / 2 + sum2 / 2 + (sum1 % 2 == 1 ? 1 : 0) + (sum2 % 2 == 1 ? 1 : 0);

    }

}
