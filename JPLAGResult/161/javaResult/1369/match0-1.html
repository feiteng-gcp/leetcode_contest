class Solution {
    public int minimumSwap(String s1, String s2) {
		int x1 = 0;
		int y1 = 0;
		for (int i = 0; i < s1.length(); i++) {
			char c1 = s1.charAt(i);
			char c2 = s2.charAt(i);
			if (c1 == c2) {
				continue;
			}
			if (c1 == 'x') {
				x1++;
			} else {
				y1++;
			}
		}
		if ((x1 + y1) % 2 == 1) {
			return -1;
		}
		return x1 / 2 + y1 / 2 + (x1 % 2) * 2;
	}
}
