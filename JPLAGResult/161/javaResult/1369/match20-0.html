class Solution {
    public int minimumSwap(String s1, String s2) {
        if (s1.length() != s2.length()) {
            return -1;
        }
        final int N = s1.length();
        int xy = 0;
        int yx = 0;
        for (int i = 0; i < N; ++i) {
            char c1 = s1.charAt(i);
            char c2 = s2.charAt(i);
            if (c1 == 'x' && c2 == 'y') {
                ++xy;
            } else if (c1 == 'y' && c2 == 'x') {
                ++yx;
            }
        }
        if ((xy + yx) % 2 != 0) {
            return -1;
        }
        int count = 0;
        if (xy % 2 == 1 && yx % 2 == 1) {
            count += 2;
            --xy;
            --yx;
        }
        count += xy / 2;
        count += yx / 2;
        return count;
    }
}
