class Solution {

    public int minimumSwap(String s1, String s2) {
        int cnt = 0, cnt2 = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) == 'x' && s2.charAt(i) == 'y') {
                cnt++;
            } else if (s1.charAt(i) == 'y' && s2.charAt(i) == 'x') {
                cnt2++;
            }
        }
        
        int ans = cnt / 2 + cnt2 / 2;
        cnt %= 2;
        cnt2 %= 2;
        
        if (cnt == 0 && cnt2 == 0) {
            return ans;
        } else if (cnt == 1 && cnt2 == 1) {
            return ans + 2;
        } else {
            return -1;
        }
    }
}
