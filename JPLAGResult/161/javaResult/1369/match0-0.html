class Solution {
    public int minimumSwap(String s1, String s2) {
        int xy = 0, yx = 0;
        for (int i = 0; i < s1.length(); i++) {
            char x = s1.charAt(i), y = s2.charAt(i);
            if (x == y) {
                continue;
            }
            if (x == 'x') {
                xy ++;
            } else {
                yx ++;
            }
        }
        if ((xy + yx) % 2 != 0) {
            return -1;
        }
        return (xy + 1) / 2 + (yx + 1) / 2;
    }
}
