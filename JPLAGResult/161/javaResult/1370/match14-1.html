class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
    	int num = 0;
    	int[] pre = new int[nums.length+1];
    	int
 len = 0;
    	for(int index=0; index<nums.length; index++) {
    		if(nums[index]%2==1) {
    			pre[len]=num;
    			len++;
    			num = 0;    			
    		}
    		else {
    			num++;    			
    		}
    	}
    	pre[len] = num;
    	len++;
    	
    	
    	int ans = 0;
    	for(int i=0; i+k-1<len-1; i++) {
    		ans += (pre[i]+1)*(pre[i+k]+1);
    	}
    	return ans;
    }

}
