class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int length = nums.length;
        List<Integer> oddIndices = new ArrayList<Integer>();
        oddIndices.add(-1);
        for (int i = 0; i < length; i++) {
            if (nums[i] % 2 != 0)
                oddIndices.add(i);
        }
        oddIndices.add(length);
        int size = oddIndices.size() - 2;
        if (size < k)
            return 0;
        int count = 0;
        int low = 1, high = size - k + 1;
        for (int i = low; i <= high; i++) {
            int begin = oddIndices.get(i);
            int end = oddIndices.get(i + k - 1);
            int prev = oddIndices.get(i - 1);
            int next = oddIndices.get(i + k);
            count += (begin - prev) * (next - end);
        }
        return count;
    }
}
