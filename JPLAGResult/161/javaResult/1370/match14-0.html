class Solution {
   public int numberOfSubarrays(int[] nums, int k) {
		int len = nums.length;
		int[] ans = new int[len + 1];
		int h = 1;
		int key = 0;
		for (int i = 0; i < nums.length; i++) {
			if (nums[i] % 2 == 1) {
				ans[key++] = h;
				h = 1;
			} else {
				h++;
			}
		}
		ans[key++] = h;
		int a = 0;
		for (int i = k; i < key; i++) {
			a += ans[i] * ans[i - k];
		}
		return a;
	}
}
