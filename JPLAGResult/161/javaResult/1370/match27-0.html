class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int N = nums.length;
        int[] sum = new int[N + 1];
        for (int i = 0; i < N; i++) {
            if (nums[i] % 2 == 1) {
                sum[i + 1] = sum[i] + 1;
            } else {
                sum[i + 1] = sum[i];
            }
        }
        int s1 = 0;
        int s2 = 0;
        int ret = 0;
        for (int e = 1; e <= N; e++) {
            if (sum[e] < k) {
                continue;
            }
            while (sum[s1] < sum[e] - k) {
                s1++;
            }
            while (sum[s2] < sum[e] - k + 1) {
                s2++;
            }
            ret += s2 - s1;
        }
        return ret;
    }
}
