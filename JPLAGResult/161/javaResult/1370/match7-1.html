class Solution {

	public int numberOfSubarrays(int[] nums, int k) {
		List<Integer> loc = new ArrayList<>();
		loc.add(-1);
		for (int i = 0; i < nums.length; i++) {
			if (nums[i] % 2 == 1) {
				loc.add(i);
			}
		}
		loc.add(nums.length);
		int res = 0;
		for (int i = k; i < loc.size() - 1; i++) {
			int l = loc.get(i-k+1) - loc.get(i-k);
			int r = loc.get(i+1) - loc.get(i);
			res += l * r;
		}
		return res;
	}
}
