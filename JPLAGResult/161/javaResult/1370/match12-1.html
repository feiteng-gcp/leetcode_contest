class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int n = nums.length;
        
int res = 0;
        int j = 0;
        int cnt = 0;
        int pre = 0;
        for (int i = 0; i < n;i++) {
            if (nums[i] % 2 == 1) {
                cnt++;
                if (cnt >= k) {
                    pre = 1;
                    while (j < i && nums[j] % 2 == 0) {
                        pre++;
                        j++;
                    }
                    j++;
                    cnt--;
                
}
            }
            res += pre;
     //       System.out.println(i+","+pre+":"+res);
        }
        return res;
    }
}
