class Solution {
 int numberOfSubarrays(int[] nums, int k) {

        int n = nums.length;

        int[] sum = new int[n + 1];

        for (int i = 1; i <= n; i++) {
            if (nums[i - 1] % 2 == 1) {
                sum[i] = sum[i - 1] + 1;
            } else {
                sum[i] = sum[i - 1];
            
}
        }

        Map<Integer, Integer> count = new HashMap<>();

        count.put(0, 1);

        int ans = 0;
        for (int i = 1; i <= n; i++) {
            int t = sum[i] - k;

            ans += count.getOrDefault(t, 0);

            count.put(sum[i], count.getOrDefault(sum[i], 0) + 1);
        }

        return ans;

    }
}
