class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int[] prev = new int[nums.length];
        int[] next = new int[nums.length];
        
        int p = -1;
        for(int i = 0; i < nums.length; i++) {
            if(nums[i] % 2 == 1) {
                prev[i] = p;
                p = i;
            }
        }
        
        p = nums.length;
        for(int i = nums.length - 1; i >= 0; i--) {
            if(nums[i] % 2 == 1) {
                next[i] = p;
                p = i;
            }
        }
        
        
int start = -1;
        int cnt = 0;
        int ret = 0;
        
        for(int j = 0; j < nums.length; j++) {
            if(nums[j] % 2 == 1) {
                if(start == -1) start = j;
                cnt++;
                if(cnt == k) {
                    int w1 = start - prev[start];
                    int w2 = next[j] - j;
                    ret += w1 * w2;
                    cnt--;
                    start = next[start];
                
}
            }
        }
        
        return ret;
    }
}
