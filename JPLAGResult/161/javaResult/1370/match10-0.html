class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n + 1];
        for(int i = 1; i <= n; i++) {
            if(nums[i - 1] % 2 == 1) dp[i] = dp[i - 1] + 1;
            else dp[i] = dp[i - 1];
        }
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        
        int res = 0;
        
        for(int i = 1; i <= n; i++) {
            if(map.containsKey(dp[i] - k)) {
                res += map.get(dp[i] - k);
            }
            map.put(dp[i], map.getOrDefault(dp[i], 0) + 1);
        }
        return res;
        
    }
}
