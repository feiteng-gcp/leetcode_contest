class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int count = 0; 
        int prefix[] = new int[nums.length]; 
        int odd = 0; 
      
        // traverse in the array 
        for (int i = 0; i < nums.length; i++) 
        { 
            prefix[odd]++; 
      
            // if array element is odd 
            if ((nums[i] & 1) == 1) 
                odd++; 
      
            // when number of odd  
            // elements >= k
            if (odd >= k) 
                count += prefix[odd - k]; 
        } 
      
        return count; 
    }
}
