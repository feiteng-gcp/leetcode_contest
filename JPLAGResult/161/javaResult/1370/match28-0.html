class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        Map<Integer , Integer> map = new HashMap<>();
        map.put(0 , 1);
        int odd = 0;
        int ans = 0;
        for(int x : nums){
            if(x % 2 != 0){
                odd++;
            }
            map.put(odd , map.getOrDefault(odd , 0) + 1);
            ans += map.getOrDefault(odd - k , 0);
        }
        return ans;
    }
}
