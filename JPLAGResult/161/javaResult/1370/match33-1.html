class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int n = nums.length;
        int[] last = new int[n];
        int lastOdd = -1;
        for (int i = 0; i < n; i++) {
            last[i] = lastOdd;
            if (nums[i] % 2 == 1) lastOdd = i;
        }
        
        int left = 0;
        int right = -1;
        int countOdd = 0;        
        while (left < n && nums[left] % 2 == 0) {
            left++;
            right++;
        }
        int ans = 0;
        while (right < n) {
            if (countOdd < k) {
                if (++right < n && nums[right] % 2 == 1)
                    countOdd++;
            } else if (countOdd > k) {
                left++;
                countOdd--;
                while (left <= right && nums[left] % 2 == 0) left++;
            } else {
                ans += (left - last[left]);
                if (++right < n && nums[right] % 2 == 1)
                    countOdd++;
            }
        }
        return ans;
    }
}
