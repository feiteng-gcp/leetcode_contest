class Solution {
    
    public int numberOfSubarrays(int[] nums, int k) {
        
        HashMap<Integer, Integer> map=new HashMap<>();
        map.put(0, 1);
        
        int cnt=0, ans=0;
        
        for(int item: nums) {
            
            if(item%2==1)
                cnt+=1;
            map.put(cnt, map.getOrDefault(cnt, 0)+1);
            
            ans+=map.getOrDefault(cnt-k, 0);
        }
        
        return ans;
    }
}
