import java.util.ArrayDeque;

class Solution {
	public int numberOfSubarrays(int[] nums, int k) {
		ArrayDeque<Integer> s = new ArrayDeque<Integer>();
		int l = 0;
		int ans = 0;
		for (int i = 0; i < nums.length; i++) {
			if (nums[i] % 2 == 1) {
				s.add(i);
			}

			if (s.size() > k) {
				l = s.pollFirst();
				l++;
			}
			if (s.size() == k)
				ans += s.peekFirst() - l + 1;
		}
		return ans;
	}
}
