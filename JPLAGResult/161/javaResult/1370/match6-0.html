class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        final int N = nums.length;
        ArrayList<Integer> lst = new ArrayList<>();
        lst.add(-1);
        for (int i = 0; i < N; ++i) {
            if (nums[i] % 2 == 1) {
                lst.add(i);
            }
        }
        lst.add(N);
        int count = 0;
        for (int i = 1; i + k < lst.size(); ++i) {
            int left = lst.get(i) - lst.get(i - 1);
            int right = lst.get(i + k) - lst.get(i + k - 1);
            count += left * right;
        }
        return count;
    }
}
