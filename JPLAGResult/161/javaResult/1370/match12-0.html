class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int count = 0;
        int result = 0;
        int left = 0;
        int leftCount = 0;
        for (int i = 0; i < nums.length; i++) {
            if ((nums[i] & 1) == 1) {
                count++;
                if (count == k) {
                    leftCount = 1;
                    for (; (nums[left] & 1) == 0; left++) {
                        leftCount++;
                    }
                    result += leftCount;
                } else if (count > k) {
                    count--;
                    leftCount = 1;
                    left++;
                    for (; (nums[left] & 1) == 0; left++) {
                        leftCount++;
                    }
                    result += leftCount;
                }
            } else {
                if (count == k) {
                    result += leftCount;
                }
            }
        }
        return result;
    }
}
