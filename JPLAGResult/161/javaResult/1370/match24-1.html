class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        
        int n = nums.length;
        for(int i = 0; i < n; i++)
        {
            if(nums[i] % 2 == 0)
                nums[i] = 0;
            else
                nums[i] = 1;
        
}
        
        Map<Integer, Integer> cnt = new HashMap<Integer, Integer>();
        cnt.put(0, 1);
        int res = 0;
        int sum = 0;
        for(int i = 0; i < n; i++)
        {
            sum += nums[i];
            if(cnt.containsKey(sum - k))
            {
                res += cnt.get(sum - k);
            }
            
            cnt.put(sum, cnt.getOrDefault(sum, 0) + 1);
        }
        
        return res;
    }
}
