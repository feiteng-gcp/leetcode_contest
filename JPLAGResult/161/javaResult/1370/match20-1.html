class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        List<Integer> oddNumberIndexes = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 == 1) {
                oddNumberIndexes.add(i);
            }
        }
        TreeSet<Integer> indexes = new TreeSet<>();
        indexes.addAll(oddNumberIndexes);
        indexes.add(-1);
        indexes.add(nums.length);
        int output = 0;
        for (int i = 0; i + k - 1 < oddNumberIndexes.size(); i++) {
            output += (oddNumberIndexes.get(i) - indexes.lower(
oddNumberIndexes.get(i))) * (indexes.higher(oddNumberIndexes.get(i + k - 1)) - oddNumberIndexes.get(i + k - 1));
        }
        return output;
    }
}
