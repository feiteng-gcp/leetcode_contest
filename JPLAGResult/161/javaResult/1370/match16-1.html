class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int res = 0;
        int len = nums.length;
        
        
int count = 1;
        List<Integer> list = new ArrayList<>();
        
        for(int i=0 ; i<len ; i++){
            if(nums[i]%2==1){
                list.add(count);
                count=1;
            } else{
                ++count;
            }
        }
        list.add(count);
        
        if(list.size()<=k) return 0;
        
        for(int i=0 ; i<list.size()-k ; i++)
{
            res += list.get(i)*list.get(i+k);
        }
        
        return res;
    }
}
