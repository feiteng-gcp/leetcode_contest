class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int[] odd = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 == 1) odd[i] = 1;
        }
        int sol = 0;
        int ct = 0;
        int end1 = -1;
        int end2 = -1;
        for (int i = 0; i < nums.length; i++) {
            ct += odd[i];
            if (ct == k) {
                end1 = i;
                int j = end1 + 1;
                while (j < nums.length && odd[j] == 0) j++;
                end2 = j - 1;
                break;
            }
        }
        if (end1 == -1) return 0;
        sol = end2 - end1 + 1;
        //System.out.println(sol);
        for (int i = 0; i < nums.length; i++) {
            if (odd[i] == 0) sol += end2 - end1 + 1;
            else {
                if (end2 + 1 < nums.length) {
                    end1 = end2 + 1;
                    for (end2 = end1 + 1; end2 < nums.length; end2++) {
                        if (odd[end2] == 1) {
                            end2--;
                            break;
                        }
                    }
                    if (end2 == nums.length) end2--;
                    sol += end2 - end1 + 1;
                } else {
                    break;
                }
            }
        }
        return sol;
    }
}
