class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int ans = 0;
        int prev = 0;
        int oddCnt = 0;
        int evenCnt = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i]%2 == 1) {
                oddCnt++;
            }
            
            if (oddCnt > k) {
                prev++;
                oddCnt--;
                evenCnt = 0;
            }
            
            if (oddCnt == k) {
                if (nums[prev]%2 == 0) {
                    evenCnt = 0;
                    while(nums[prev]%2 == 0 && prev < i) {
                        prev++;
                        evenCnt++;
                        // System.out.println(i);
                    }
                }
                ans += evenCnt+1;
                // System.out.println(i + " " + evenCnt);
            } 
        }
        return ans;
    }
}
