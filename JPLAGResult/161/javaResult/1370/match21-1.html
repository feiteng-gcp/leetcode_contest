class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 == 1) {
                list.add(i);
            }
        }
        if (list.size() < k) {
            return 0;
        }
        int ans = 0;
        int pre = -1;
        for (int i = 0; i <= list.size() - k; i++) {
            int start = list.get(i), end = list.get(i + k - 1);
            int max = i + k == list.size() ? nums.length : list.get(i + k);
            int left = start - pre;
            int right = max - end;
            ans += left * right;
            pre = start;
        }
        return ans;
    }
}
