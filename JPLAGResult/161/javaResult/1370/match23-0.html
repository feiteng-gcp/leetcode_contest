class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
		List<Integer> oddList = new ArrayList<>();
		for (int i = 0; i < nums.length; i++) {
			if (nums[i] % 2 == 1) {
				oddList.add(i);
			}
		}
		if (oddList.size() < k) {
			return 0;
		}
		int count = 0;
		for (int i = k - 1; i < oddList.size(); i++) {
			int size1 = (i - k + 1) == 0 ? oddList.get(0) + 1 : oddList.get(i - k + 1) - oddList.get(i - k);
			int size2 = (i == oddList.size() - 1) ? nums.length - oddList.get(i) : oddList.get(i + 1) - oddList.get(i);
			count += size1 * size2;
		}
		return count;
    }
}
