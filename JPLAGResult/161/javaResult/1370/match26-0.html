class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int res = 0;
        Queue<Integer> q = new LinkedList<>();
        for (int i = -1, j = 0; j < nums.length; ++j) {
            if (nums[j] % 2 == 1) q.add(j);
            if (q.size() > k) {
                i = q.poll();
            }
            if (q.size() == k) res += q.peek() - i;
        }
        return res;
    }
}
