class Solution {
    public int numberOfSubarrays(int[] a, int m) {
        int n=a.length;
        int count = 0; 
        int prefix[] = new int[n]; 
        int odd = 0; 
      
        // traverse in the array 
        for (int i = 0; i < n; i++) 
        { 
            prefix[odd]++; 
      
            // if array element is odd 
            if ((a[i] & 1) == 1) 
                odd++; 
      
            // when number of odd  
            // elements >= M 
            if (odd >= m) 
                count += prefix[odd - m]; 
        } 
      
        return count; 
        
    }
}
