class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        ArrayList <Integer> list = new ArrayList();
        int n = nums.length, ans = 0;
        list.add(-1);
        for(int i = 0 ; i < n; i ++)
            if(nums[i] % 2 == 1)
                list.add(i);
        list.add(n);
        if(list.size() < k + 2)
            return 0;
        for(int i = 0; i + k + 1 < list.size() ; i ++)
        {
            ans += (list.get(i + k + 1) - 
list.get(i + k)) * (list.get(i + 1) - list.get(i));
        }
        return ans;
    }
}
