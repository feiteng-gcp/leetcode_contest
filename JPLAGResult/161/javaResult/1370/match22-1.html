import java.util.ArrayList;
import java.util.List;

class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        int res = 0;
        
List<Integer> list = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 == 1) {
                list.add(i);
            }
        }
        for (int i = 0; i < list.size(); i++) {
            int tail = i + k - 1;
            if (tail >= list.size()) {
                break;
            }
            
int r1 = i == 0 ? list.get(i) + 1 : list.get(i) - list.get(i - 1);
            int r2 = tail == list.size() - 1 ? nums.length - list.get(tail) : list.get(tail + 1) - list.get(tail);
            res += r1 * r2;


        }
        return res;
    }

    public static void main(String[] args) {
        System.out.println(new Solution().numberOfSubarrays(new int[]{1, 1, 2, 1, 1}, 3));
        System.out.println(new Solution().numberOfSubarrays(new int[]{2, 4, 6}, 1));
        System.out.println(new Solution().numberOfSubarrays(new int[]{2, 2, 2, 1, 2, 2, 1, 2, 2, 2}, 2));
    }
}
