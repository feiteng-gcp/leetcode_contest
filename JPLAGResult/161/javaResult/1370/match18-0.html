class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        if(nums == null || nums.length == 0 || nums.length < k) {
            return 0;
        }
        List<Integer> odds = new ArrayList<>();
        for(int i = 0; i < nums.length; i++) {
            if(nums[i] % 2 == 1) {
                odds.add(i);
            }
        }
        // System.out.println(odds);
        if(odds.size() < k) {
            return 0;
        }
        int i = 0;
        int j = k - 1;
        int res = 0;
        for(;j < odds.size(); i++, j++) {
            int ll =0;
            if(i != 0) {
                ll = odds.get(i - 1) + 1;
            }
            int rr = nums.length - 1;
            if(j != odds.size() - 1) {
                rr = odds.get(j + 1) - 1;
            }
            res += (odds.get(i) - ll + 1) * (rr - odds.get(j) + 1);
        }
        return res;
    }
}
