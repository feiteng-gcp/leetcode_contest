class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        L=[-1]
        for i in range(len(nums)):
            if nums[i]%2==1:
                L.append(i)
        L.append(len(nums))
        res=0
        for j in range(1,len(L)-k):
            res+=(L[j]-L[j-1])*(L[j+k]-L[j+k-1])
        return res
