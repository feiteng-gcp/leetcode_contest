class Solution:
    def numberOfSubarrays(self, nums, k) -> int:
        odd_indices 
= []
        for i in range(len(nums)):
            if nums[i] % 2 != 0:
                odd_indices.append(i)
        ans = 0
        for i in range(len(odd_indices) - k + 1):
            head = odd_indices[i]
            tail = odd_indices[i + k - 1]
            ans += (head - (odd_indices[i - 1] if i > 0 else -1)) * (-tail + (odd_indices[i + k] if i + k < len(odd_indices)  else len(nums)))
        return ans
