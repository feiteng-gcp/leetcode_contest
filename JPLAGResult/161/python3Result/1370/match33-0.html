class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        odds_ind = [i for i, num in enumerate(nums) if num % 2]
        odds_ind = [-1] + odds_ind + [len(nums)]
        # print(odds_ind)
        
        res = 0
        for i in range(1, len(odds_ind)-k):
            res += (odds_ind[i]-odds_ind[i-1])*(odds_ind[i+k]-odds_ind[i+k-1])
        
        return res
