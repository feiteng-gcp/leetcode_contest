class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        oddId = [-1]
        for i in range(len(nums)):
            if nums[i] % 2:
                oddId.append(i)
        oddId.append(len(nums))
        
        count = 0
        
        firstOdd = 1
        lastOdd = len(oddId) - k
        for i in range(firstOdd, lastOdd):
            count += (oddId[i] - oddId[i - 1]) * (oddId[i + k] - oddId[i + k - 1])
            
        return count
