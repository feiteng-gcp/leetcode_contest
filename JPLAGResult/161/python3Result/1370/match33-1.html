class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        if not nums: return 0
        memo = [-1]
        for i, num in enumerate(nums):
            if num%2: memo.append(i)
        if len(memo)<k+1: return 0
        memo.append(len
(nums))
        res = 0
        for i in range(1, len(memo)-k):
            res += (memo[i]-memo[i-1])*(memo[i+k] - memo[i+k-1]) 
        return res
