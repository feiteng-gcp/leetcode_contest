class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        idx = []
        for i,num in enumerate(nums):
            if num%2==1:
                idx
.append(i)
        res = 0
        for i in range(len(idx)-k+1):
            start = idx[i]
            end = idx[i+k-1]
            if i==0:
                left = start+1
            else:
                left = start-idx[i-1]
            if i+k==len(idx):
                right = len(nums)-end
            else:
                right = idx[i+k]-end
            res += (left*right)
            
        return res
