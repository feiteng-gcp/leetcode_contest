class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        cnt = []
        c = 0
        for i in nums:
            if i % 2:
                cnt.append(c)
                c = 0
            else:
                c += 1
        cnt.append(c)
        ret = 0
        for i in range(len(cnt) - k):
            ret += (cnt[i] + 1) * (cnt[i + k] + 1)
        return ret
