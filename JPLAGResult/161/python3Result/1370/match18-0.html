class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        ans = 0
        n = len(nums)
        ind = [-1]
        for i in range(n):
            if nums[i]%2:
                ind.append(i)
        ind.append(n)
        m = len(ind)
        if m-2<k:
            return 0
        for i in range(k,m-1):
            ans += (ind[i+1]-ind[i])*(ind[i-k+1]-ind[i-k])
        return ans
