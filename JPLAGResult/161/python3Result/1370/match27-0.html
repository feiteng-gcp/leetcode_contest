class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        ind = -1
        evenlen = []
        n = len(nums)
        for i in range(n):
            if nums[i] % 2 == 1:
                evenlen.append(i-ind)
                ind = i
        evenlen.append(n-ind)
        s = 0
        for i in range(len(evenlen)-k):
            s += evenlen[i] * evenlen[i+k]
        #print(evenlen)
        return s
