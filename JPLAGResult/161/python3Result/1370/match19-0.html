class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        pos = [-1]
        for i in range(len(nums)):
            if nums[i]%2==1:
                pos.append(i)
        pos.append(len(nums))
        res = 0
        for i in range(k, len(pos)-1):
            far_right = pos[i+1]-1
            far_left = pos[i-k]+1
            res += (far_right-pos[i]+1)*(pos[i-k+1]-far_left+1)
        return res
