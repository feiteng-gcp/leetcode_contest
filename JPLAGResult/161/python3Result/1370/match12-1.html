class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        ids = [-1]
        for i in range(len(nums)):
            if nums[i] % 2 != 0:
                ids.append(i)
        ids.append(len
(nums))
        if len(ids) < k + 2:
            return 0
        
        # print(ids)
        p = 1
        res = 0
        while p + k < len(ids):
            res += (ids[p] - ids[p - 1]) * (ids[p + k] - ids[p + k - 1])
            p += 1
        return res
