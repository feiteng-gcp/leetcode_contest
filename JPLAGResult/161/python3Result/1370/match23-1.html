class Solution:
    def check(self, nums, k):
        n 
= len(nums)
        l, odd_cnt, ans = 0, 0, 0
        for r in range(n):
            if nums[r] % 2 == 1:
                odd_cnt += 1
            while odd_cnt > k:
                if nums[l] % 2 == 1:
                    odd_cnt -= 1
                l += 1
            ans += r - l + 1
        return ans

    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        return self.check(nums, k) - self.check(nums, k - 1)
