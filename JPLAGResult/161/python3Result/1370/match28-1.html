class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        count = collections.defaultdict(int)
        res = 0
        cur = 0
        count[0] 
= 1
        for val in nums:
            if val & 1 != 0:
                cur += 1
            res += count[cur - k]
            count[cur] += 1
        return res
