class Solution:
    def numberOfSubarrays(self, arr: List[int], k: int) -> int:
        loc = [-1]
        for i,val in enumerate(arr):
            if val%2:
                loc += [i]
        loc += [len(arr)]
        if len(loc) < k - 2:
            return 0
        res = 0
        s = 1
        while s+ k-1 < len(loc) -1:
            res += (loc[s] - loc[s-1])*(loc[s+k]-loc[s+k-1])
            s += 1
        return res
