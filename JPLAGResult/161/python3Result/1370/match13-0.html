class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        lst = [-1]
        for i in range(len(nums)):
            if nums[i] % 2:
                lst.append(i)
        lst.append(len(nums))
        res = 0
        for i in range(1, len(lst) - k):
            res += (lst[i] - lst[i-1]) * (lst[i+k] - lst[i+k-1])
        return res
