class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        # convert to 0/1
        # for i in range(len(nums)):
        #    nums[i] %= 2
        oddpos = []
        for i in range(len(nums)):
            if nums[i] % 2 == 1:
                oddpos.append(i)
        #print(oddpos)
        res = 0
        for i in range(len(oddpos)-k+1):
            #print(i, i+k-1)
            left = oddpos[i]
            right = oddpos[i+k-1]
            if i == 0:
                ll = 0
            else:
                ll = oddpos[i-1]+1
            if i+k-1 == len(oddpos)-1:
                rr = len(nums)-1
            else:
                rr = oddpos[i+k]-1
            #print(ll,left,right,rr)
            left_var = left-ll+1
            right_var = rr-right+1
            res += left_var * right_var
        return res
