class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        arr = []
        count = 0
        for i in nums:
            if i%2 != 0:
                arr.append(count)
                count = 0
            else:
                count += 1
        
        arr.append(count)
        #print(arr)
        ans = len(arr)-k
        if ans < 0:
            return 0
        
        for i in range(1,len(arr)-1):
            if i+k-1 < len(arr):
                ans += arr[i-1]
                ans += arr[i+k-1]
                ans += arr[i-1]*arr[i+k-1]
        
        
        return ans
