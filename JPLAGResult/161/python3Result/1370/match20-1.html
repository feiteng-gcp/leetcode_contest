class Solution:
    def numberOfSubarrays(self, A: List[int], k: int) -> int:
        loc = [-1]
        for i in range(len(A)):
            if A[i] % 2 == 1:
                loc.append(i)
        loc.append(len(A))
        ans = 0
        for i in range(len(loc)-k-1):
            ans += (loc[i+1] - loc[i]) * (loc[i+k+1] - loc[i+k])
        return ans
