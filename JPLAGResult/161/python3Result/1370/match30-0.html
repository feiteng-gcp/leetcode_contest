class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        gaps = []
        gap = 1
        for num in nums:
            if num % 2 == 1:
                gaps.append(gap)
                gap = 1
            else:
                gap += 1
        gaps.append(gap)
        res = 0
        for i in range(len(gaps) - k):
            res += gaps[i]*gaps[i+k]
        return res
