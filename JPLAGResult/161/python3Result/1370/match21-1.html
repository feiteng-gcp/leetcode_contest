class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        oddIndexs = []
        ret 
= 0
        for i in range(len(nums)):
            if nums[i]%2 == 1:
                oddIndexs.append(i)
        
        if k > len(oddIndexs):
            return 0
        
        else:
            for i in range(len(oddIndexs)-k+1):
                left = 0
                right = 0
                
if i == 0:
                    left = oddIndexs[i] + 1
                else:
                    left = oddIndexs[i] - oddIndexs[i-1]
                
                if i+k == len(oddIndexs):
                    right = len(nums) - oddIndexs[i+k-1]
                else:
                    right = oddIndexs[i+k] - oddIndexs[i+k-1]
                
                ret += left*right
            
        
        return ret
