class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        x2y = 0
        y2x = 0
        n = len(s1)
        for i in range(n):
            if s1[i] == 'x' and s2[i] == 'y':
                x2y += 1
            elif s1[i] == 'y' and s2[i] == 'x':
                y2x += 1
        m = x2y // 2
        n = y2x // 2
        l = x2y % 2
        k = y2x % 2
        if l != k:
            return -1
        return m + n + l * 2
