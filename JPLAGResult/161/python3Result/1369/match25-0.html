class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        
        move = 0
        countXY = 0
        countYX = 0
        
        for i in range(len(s1)):
            if s1[i] == 'x' and s2[i] == 'y':
                countXY += 1
            if s1[i] == 'y' and s2[i] == 'x':
                countYX += 1
                
        move += countXY // 2 + countYX // 2
        
        if countXY % 2 != countYX % 2:
            return -1
        elif countXY % 2:
            return move + 2
        else:
            return move
