class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        n1 = 0
        n2 = 0
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                if s1[i] == 'x':
                    n1 += 1
                else:
                    n2 += 1
        res = (n1 // 2) + (n2 // 2)
        rem = (n1 % 2) + (n2 % 2)
        
        if rem == 2:
            res += 2
        elif rem == 1:
            res = -1
            
        return res
