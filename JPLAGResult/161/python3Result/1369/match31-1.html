class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        leng = len(s1)
        total_x = 0
        diffx = 0
        diffy = 0
        for i in range(len(s1)):
            if s1[i] == 'x':
                total_x += 1
            if s2[i] == 'x':
                total_x += 1
            
if s1[i] != s2[i]:
                if s1[i] == 'x':
                    diffx += 1
                else:
                    diffy += 1
        if total_x % 2 == 1:
            return -1
        return (diffx//2)+(diffx%2)+(diffy//2)+(diffy%2)
