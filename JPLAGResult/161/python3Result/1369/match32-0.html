class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        x = 0
        y = 0
        if len(s1) != len(s2):
            return -1
        
        for i in range(len(s1)):
            if s1[i] == 'x' and s2[i] == 'y':
                x += 1
            elif s1[i] == 'y' and s2[i] == 'x':
                y += 1
        
        if (x + y) % 2 != 0:
            return -1
        if x % 2 == 0:
            return (int) ((x + y) / 2)
        else:
            return (int) ((x + y) / 2 + 1)
