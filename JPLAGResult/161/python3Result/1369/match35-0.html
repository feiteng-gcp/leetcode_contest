class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        size = len(s1)
        x1 = x2 = y1 = y2 = 0
        for l in s1:
            if l == 'x':
                x1 += 1
            else:
                y1 += 1
        for l in s2:
            if l == 'x':
                x2 += 1
            else:
                y2 += 1
        if (x1+x2)%2 == 1:
            return -1
        
        
        l1 = l2 = 0
        for i in range(size):
            if s1[i] == 'x' and s2[i] == 'y':
                l1 += 1
            elif s1[i] == 'y' and s2[i] == 'x':
                l2 += 1
        if l1 % 2 == 0 and l2 % 2 == 0:
            return int((l1+l2)/2)
        else:
            return int((l1+l2)/2)+1
