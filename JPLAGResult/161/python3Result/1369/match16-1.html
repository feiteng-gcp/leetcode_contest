class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        
        if len(s1) != len(s2):
            return -1
        
        
        cnt_x = 0
        cnt_y = 0
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                if s1[i] == 'x':
                    cnt_x += 1
                else:
                    cnt_y += 1
        
        ans = (cnt_x // 2) + (cnt_y // 2)
        cnt_x = (cnt_x % 2)
        cnt_y = (cnt_y % 2)
        
        if (cnt_x == 1 and cnt_y == 1):
            ans += 2
        elif (cnt_x == 0 and cnt_y == 0):
            pass
        else:
            return -1
        
        return ans
