class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        cnt_xy, cnt_yx 
= 0, 0
        for p, q in zip(s1, s2):
            if (p, q) == ('x', 'y'):
                cnt_xy += 1
            elif (p, q) == ('y', 'x'):
                cnt_yx += 1
        if (cnt_xy + cnt_yx) % 2 != 0:
            return -1
        return cnt_xy // 2 + cnt_yx // 2 + (cnt_xy % 2) * 2
