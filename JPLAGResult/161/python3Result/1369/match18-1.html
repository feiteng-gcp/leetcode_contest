class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        xy = 0
        yx = 0
        for i in range(0, len(s1)):
            if s1[i] == 'x' and s2[i] == 'y':
                xy = xy + 1
            elif s1[i] == 'y' and s2[i] == 'x':
                yx = yx + 1
        if (xy + yx) % 2 == 1:
            return -1
        else:
            return (xy // 2) + (yx // 2) + (xy % 2) + (yx % 2)
