class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        n1 = 0
        n2 = 0
        for i in range(len(s1)):
            if s1[i]=='x' and s2[i]=='y':
                n1 += 1
            elif s1[i]=='y' and s2[i]=='x':
                n2 += 1
        sol = 0
        sol += n1//2
        sol += n2//2
        n1 = n1%2
        n2 = n2%2
        if n1==n2:
            sol += n1+n2
            return sol
        else:
            return -1
