"""
xx : yy -> 1
xy : yx -> 2
x : x -> 0
y : y -> 0
xy
yx
"""
class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        n = len(s1)
        cxy, cyx 
= 0, 0
        for a, b in zip(s1, s2):
            if a == b: continue
            if a == 'x':
                cxy += 1
            else:
                cyx += 1
        if (cxy + cyx) % 2 == 1:
            return -1
        ans = (cxy) // 2 + (cyx) // 2 + (cxy) % 2 * 2
        return ans
