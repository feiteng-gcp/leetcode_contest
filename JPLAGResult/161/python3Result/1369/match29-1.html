class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        re = 0
        xy, yx = 0, 0
        for ch1, ch2 in zip(s1, s2):
            if ch1 != ch2:
                if ch1 == "x":
                    xy += 1
                else:
                    yx += 1
        #print(xy, yx)
        
        if (xy%2)!=(yx%2):
            return -1
        else:
            re += (xy // 2)
            re += (yx // 2)
            if xy % 2 != 0:
                re += 2
        return re
