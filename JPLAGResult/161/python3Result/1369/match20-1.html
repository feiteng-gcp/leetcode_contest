class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        from collections import Counter
        
if len(s1) != len(s2): return -1
        xy = yx = 0
        for i in range(len(s1)):
            if s1[i]!=s2[i]:
                if s1[i] == 'x':
                    xy += 1
                else:
                    yx += 1
        res = xy//2 + yx//2
        xy %= 2
        yx %= 2
        if xy + yx == 1: return -1
        return res + xy+yx
