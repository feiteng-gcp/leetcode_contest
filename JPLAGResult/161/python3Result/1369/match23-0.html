class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        x, y = 0, 0
        for a, b in zip(s1, s2):
            if a == b:
                continue
            if a == 'x':
                x += 1
            else:
                y += 1
        if (x&1) ^ (y&1):
            return -1
        return (x>>1) + (y>>1) + (x&1) + (y&1)
