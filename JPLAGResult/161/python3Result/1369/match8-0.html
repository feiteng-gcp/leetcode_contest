class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        nx, ny = 0, 0
        L = len(s1)
        for i in range(L):
            if s1[i] != s2[i]:
                if s1[i] == "x":
                    nx += 1
                else:
                    ny += 1
        #print(nx, ny)
        if (nx + ny) % 2 == 1:
            return -1
        return int(nx/2) + int(ny/2) + (nx%2)*2
