class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        x_mis = 0
        y_mis = 0
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                if s1[i] == 'x':
                    x_mis += 1
                else:
                    y_mis += 1
        
        if (x_mis + y_mis) % 2:
            return -1
        
        return x_mis // 2 + x_mis % 2  + y_mis // 2 + y_mis % 2
