class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        if len(s1)!= len(s2):
            return -1
        ct1
=0
        ct2=0
        for i in range(len(s1)):
            if s1[i]!=s2[i]:
                if s1[i]=='x':
                    ct1+=1
                else:
                    ct2+=1
        if (ct1+ct2)%2!=0:
            return -1
        return ct1//2+ct2//2+2*(ct1%2)
