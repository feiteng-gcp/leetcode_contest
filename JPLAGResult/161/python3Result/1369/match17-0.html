class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        ret = 0
        xy = yx = 0
        
        for i in range(len(s1)):
            if s1[i] == 'x' and s2[i] == 'y':
                xy += 1
            elif s1[i] == 'y' and s2[i] == 'x':
                yx += 1
            
        if xy & 1 == 0 and yx & 1 == 0:
            return xy // 2 + yx // 2
        elif xy & 1 == 1 and yx & 1 == 1:
            return xy // 2 + yx // 2 + 2
        else:
            return -1
