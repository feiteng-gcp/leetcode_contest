class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        c1 = collections.Counter([i for i in s1+s2])
        if c1['x']% 2 + c1['y']%2:return -1
        cnt = {'x':0,'y':0}
        for a,b in zip([i for i in s1],[i for i in s2]):
            if a!= b: cnt[a] += 1
        res = list(cnt.values())
        res.sort()
        return res[0]//2+res[1]//2 + res[0]%2 + res[1]%2
