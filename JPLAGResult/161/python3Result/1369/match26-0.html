class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        one = 0
        two = 0
        n = len(s1)
        for i in range(n):
            if s1[i] == 'x' and s2[i] == 'y':
                one += 1
            elif s1[i] == 'y' and s2[i] == 'x':
                two += 1
        
        ans = (one//2)+(two//2)
        
        if one%2 != two%2:
            return -1
        
        if one%2 == 1:
            ans += 2
        
        return ans
