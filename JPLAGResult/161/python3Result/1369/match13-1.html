class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        if len(s1) != len(s2):
            return -1
        count1, count2 
= 0, 0
        for i in range(len(s1)):
            if s1[i] == 'x' and s2[i] == 'y':
                count1 += 1
            elif s1[i] == 'y' and s2[i] == 'x':
                count2 += 1
        if (count1 + count2) % 2 == 1:
            return -1
        if count1 % 2 == 0:
            return count1 // 2 + count2 // 2
        else:
            return count1 // 2 + count2 // 2 + 2
