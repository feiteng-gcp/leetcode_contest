class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        xy, yx = 0, 0
        for c1, c2 in zip(s1, s2):
            if c1 != c2:
                if c1 == 'x':
                    xy += 1
                else:
                    yx += 1
        if xy % 2 == 1 and yx % 2 == 1:
            return 2 + xy // 2 + yx // 2
        if xy % 2 == 1 or yx % 2 == 1:
            return -1
        return xy // 2 + yx // 2
