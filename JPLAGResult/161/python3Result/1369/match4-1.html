class Solution:
    def minimumSwap(self, s1: str, s2: str) -> int:
        if len(s1)!= len(s2):
            return -1
        x1, y1 
= 0, 0
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                if s1[i] == 'x':
                    x1 += 1
                else:
                    y1 += 1
        if (x1 + y1) % 2 != 0:
            return -1
        return x1 // 2 + y1 // 2 + x1 % 2 + y1 % 2
