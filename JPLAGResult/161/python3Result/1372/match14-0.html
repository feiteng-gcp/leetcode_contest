class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        l = nums[0]
        for i in range(1, len(nums)):
            l = self.lcf(l, nums[i])
        if l == 1:
            return True
        return False
    
    def lcf(self, num1, num2):
        if num2 == 0:
            return num1
        return self.lcf(num2, num1%num2)
