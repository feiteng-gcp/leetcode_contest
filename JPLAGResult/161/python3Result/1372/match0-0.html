class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        if len(nums) == 1: return nums[0] == 1 or nums[0] == -1
        a = math.gcd(nums[0],nums[1])
        if a ==1:return True
        for i in range(2,len(nums)):
            a = math.gcd(a,nums[i])
            if a == 1:return True
        return False
