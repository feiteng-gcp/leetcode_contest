class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        if len(nums) == 1 and nums[0] == 1:
            return True
        from math import gcd
        cur = nums[0]
        for i in range(1, len(nums)):
            cur = gcd(cur, nums[i])
            if cur == 1:
                return True
        
        return False
