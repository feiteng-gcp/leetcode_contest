class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        def gcd(a, b):
            if a%b == 0:
                return b
            return gcd(b, a%b)
        L = len(nums)
        if L == 1:
            return True if nums[0] == 1 else False
        t = nums[0]
        for i in range(1, L):
            t = gcd(t, nums[i])
            if t == 1:
                return True
        return False
