class Solution:
    def gcd(self, a: int, b: int) -> int:
        if a < b:
            a, b = b, a
        if a % b == 0:
            return b
        else:
            return self.gcd(b, a % b)
    
    def isGoodArray(self, nums: List[int]) -> bool:
        gcdOfNums = nums[0]
        for i in range(1, len(nums)):
            gcdOfNums = self.gcd(gcdOfNums, nums[i])
        return gcdOfNums == 1
