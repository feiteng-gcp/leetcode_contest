from math import gcd
class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        if 1 in nums: return True
        if len(nums)==1: return False
        cur 
= nums[0]
        for i in range(1,len(nums)):
            cur = gcd(cur,nums[i])
            if cur==1: return True
        return False
