class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        def gcd(x, y): 
            while(y): 
                x, y = y, x % y 
            return x
        x = nums[0]
        for n in nums[1:]:
            x = gcd(x,n)
            if x == 1:
                return True
        else:
            return nums == [1]
