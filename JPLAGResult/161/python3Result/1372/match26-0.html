from math import gcd
class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        if len(nums) == 1:
            return nums[0] == 1
        g = nums[0]
        for i in range(1, len(nums)):
            g = gcd(g, nums[i])
            if g == 1:
                return True
        return False
