class Solution:
    def isGoodArray(self, nums: List[int]) -> bool:
        nums = sorted(nums)
        if 1 == nums[0]:
            return True
        gcd 
= nums[0]
        for i in range(1, len(nums)):
            gcd = self._gcd(gcd, nums[i])
            if gcd == 1:
                return True
        return False
    
    def _gcd(self, a, b):
        if a == 0:
            return b
        return self._gcd(b % a, a)
