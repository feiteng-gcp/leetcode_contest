class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        ans = []
        left = 0
        for i in s:
            if i == '(':
                left += 1
                ans.append(i)
            elif i == ')':
                if left > 0:
                    left -= 1
                    ans.append(i)
            else:
                ans.append(i)
        
        right = 0
        ans.reverse()
        ans1 = []
        for i in ans:
            if i == ')':
                right += 1
                ans1.append(i)
            elif i == '(':
                if right > 0:
                    right -= 1
                    ans1.append(i)
            else:
                ans1.append(i)
        
        ans1.reverse()
        s_ans = ''
        for i in ans1:
            s_ans += i
        
        return s_ans
