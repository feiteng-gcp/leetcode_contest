class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        # stack
        stack = []
        inv = []
        for i,c in enumerate(s):
            if c == '(':
                stack.append(i)
            elif c == ')':
                if not stack:
                    # mark invalid
                    inv.append(i)
                else:
                    stack.pop()
        inv.extend(stack)
        inv = set(inv)
        res = ''
        for i,c in enumerate(s):
            if i not in inv:
                res+=c
        return res
