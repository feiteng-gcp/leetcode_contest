class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        left = []
        right = []
        for i in range(len(s)):
            if s[i] == '(':
                left += [i]
            if s[i] == ')':
                if len(left) > len(right):
                    right += [i]
        while len(left) > len(right):
            left.pop()
        left = set(left)
        right = set(right)
        res = ''
        for i in range(len(s)):
            if s[i] not in '()':
                res += s[i]
            if i in left or i in right:
                res += s[i]
        return res
