class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        L = 0
        out = []
        for c in s:
            if c == '(':
                L += 1
                out.append(c)
            elif c == ')':
                if L > 0:
                    L -= 1
                    out.append(c)
            else:
                out.append(c)
                
        R = 0
        out2 = []
        for c in out[::-1]:
            if c == ')':
                R += 1
                out2.append(c)
            elif c == '(':
                if R > 0:
                    R -= 1
                    out2.append(c)
            else:
                out2.append(c)
        return ''.join(out2[::-1])
