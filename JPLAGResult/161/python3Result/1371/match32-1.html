class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        popidx = set()
        L 
= []
        for i,c in enumerate(s):
            if c == '(':
                L.append((0,i))
            elif c==')':
                if L and L[-1][0]==0:
                    L.pop()
                else:
                    L.append((1,i))
        popidx = set(i for t,i in L)
                
                
        
        return ''.join(s[i] for i in range(len(s)) if i not in popidx)
