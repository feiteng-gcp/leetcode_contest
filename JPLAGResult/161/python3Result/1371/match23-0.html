class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        opens = 0
        ans = ''
        for c in s:
            if c == '(':
                opens += 1
                ans += c
            elif c == ')':
                if opens:
                    opens -= 1
                    ans += c
            else:
                ans += c
        
        opens = 0
        ans2 = ''
        for c in ans[::-1]:
            if c == ')':
                opens += 1
                ans2 += c
            elif c == '(':
                if opens:
                    opens -= 1
                    ans2 += c
            else:
                ans2 += c
        return ans2[::-1]
