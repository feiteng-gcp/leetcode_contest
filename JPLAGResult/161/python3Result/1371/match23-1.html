class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        l2r = ''
        left = 0
        for c in s:
            if c == '(':
                l2r += c
                left += 1
            elif c == ')':
                if left > 0:
                    l2r += c
                    left -= 1
            else:
                l2r += c
        if left > 0:
            temp = ''
            for c in l2r[::-1]:
                if left > 0 and c == '(':
                    left -= 1
                    continue
                temp = c + temp
            l2r = temp

        r2l 
= ''
        right = 0
        for c in s[::-1]:
            if c == ')':
                r2l = c + r2l
                right += 1
            elif c == '(':
                if right > 0:
                    r2l = c + r2l
                    right -= 1
            else:
                r2l = c + r2l
        if right > 0:
            temp = ''
            for c in r2l:
                if right > 0 and c == ')':
                    right -= 1
                    continue
                temp += c
            r2l = temp   
        return max([r2l, l2r], key = len)
