class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        keep = [1]*len(s)
        stack = []
        for i,c in enumerate(s):
            if c=='(':
                stack.append(i)
            if c==')':
                if len(stack)>0:
                    stack.pop(len
(stack)-1)
                else:
                    keep[i] = 0
        for i in stack:
            keep[i] = 0
        #print(keep)
        #print([c for c,valid in zip(s,keep) if valid==1])
        return "".join([c for c,valid in zip(s,keep) if valid==1])
