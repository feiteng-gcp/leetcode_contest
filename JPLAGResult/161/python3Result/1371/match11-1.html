class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        s = list(s)
        
        forward 
= []
        count = 0
        for c in s:
            if c == '(':
                count += 1
                forward.append(c)
            elif c == ')':
                if count != 0:
                    count -= 1
                    forward.append(c)
            else:
                forward.append(c)
        
        count = 0
        backward = []
        for c in forward[::-1]:
            if c == ')':
                count += 1
                backward.append(c)
            elif c == '(':
                if count != 0:
                    backward.append(c)
                    count -= 1
            else:
                backward.append(c)
        
        return ''.join(backward[::-1])
