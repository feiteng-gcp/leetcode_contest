class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        stack = []
        to_remove = set()
        for i,c in enumerate(s):
            if c=='(':
                stack.append((c,i))
            elif c==')':
                if stack and stack[-1][0]=='(':
                    stack.pop()
                else:
                    to_remove.add(i)
        while stack:
            to_remove.add(stack.pop()[1])
        res = []
        for i,c in enumerate(s):
            if i in to_remove:
                continue
            else:
                res.append(c)
        return ''.join(res)
