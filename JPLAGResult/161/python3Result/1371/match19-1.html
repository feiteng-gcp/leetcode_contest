class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        res = ''
        l = []
        r = []
        for i in range(len(s)):
            if s[i] == '(':
                l.append(i)
            elif s[i] == ')':
                
if l:
                    l.pop()
                else:
                    r.append(i)
        ss = set(l + r)
        res = ''.join([s[i] for i in range(len(s)) if i not in ss])
        return res
