class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        if not s: return s
        left 
= []
        right = []
        for i, c in enumerate(s):
            if c == '(':
                left.append(i)
            elif c == ')':
                if not left:
                    right.append(i)
                else:
                    left.pop()
        res = list(s)
        for idx in (left+right):
            res[idx] = ""
        return "".join(res)
