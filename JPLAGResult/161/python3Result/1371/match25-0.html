class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        stack = []
        for i,c in enumerate(s):
            if c == "(":
                stack.append((c, i))
            elif c== ")":
                if stack and stack[-1][0]=="(":
                    stack.pop()
                else:
                    stack.append((c, i))
        #print(s)
        stack_set = set(stack)
        out = ""
        for i,c in enumerate(s):
            if (c, i) in stack_set:
                continue
            out += c
        return out
