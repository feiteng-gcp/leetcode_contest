class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        count, lst 
= 0, []
        for c in s:
            if c == "(":
                count += 1
                lst.append(c)
            elif c == ")":
                if count > 0:
                    count -= 1
                    lst.append(c)
            else:
                lst.append(c)
        s = "".join(lst)
        count, lst 
= 0, []
        for c in s[::-1]:
            if c == ")":
                count += 1
                lst.append(c)
            elif c == "(":
                if count > 0:
                    count -= 1
                    lst.append(c)
            else:
                lst.append(c)
        return "".join(lst[::-1])
