class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        opens = []
        removes = set()
        
for idx, ch in enumerate(s):
            if ch == '(':
                opens.append(idx)
            elif ch == ')':
                if opens:
                    opens.pop()
                else:
                    removes.add(idx)
        removes 
|= set(opens)
        ret = ''
        for idx, ch in enumerate(s):
            if idx not in removes:
                ret += ch
        return ret
