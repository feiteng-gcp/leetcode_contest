class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        stack = []
        vis = set()
        
        for i, ch in enumerate(s):
            if ch == '(':
                stack.append((i, ch))
            elif ch == ')':
                if not stack:
                    vis
.add(i)
                else:
                    stack.pop()
        
        while stack:
            vis.add(stack.pop()[0])
        
        ret = [ch for i, ch in enumerate(s) if i not in vis]
        return "".join(ret)
