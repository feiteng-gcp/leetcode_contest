class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        left = []
        
        removed = set()
        for idx, ch in enumerate(s):
            if ch == '(':
                left.append(idx)
            elif ch == ')':
                if not left:
                    removed.add(idx)
                else:
                    left.pop()
        
        removed |= set(left)
        
        s1 = ''
        for idx in range(len(s)):
            if idx not in removed:
                s1 += s[idx]
        return s1
