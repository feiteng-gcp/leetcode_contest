class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        remove = [False] * len(s)
        stack = []
        for i, c in enumerate(s):
            if c == '(':
                stack.append(i)
            if c == ')':
                if stack:
                    stack.pop()
                else:
                    remove[i] = True
        for i in stack:
            remove[i] = True
        return ''.join([c for i, c in enumerate(s) if not remove[i]])
