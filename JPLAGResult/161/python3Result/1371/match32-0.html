class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        stack = []
        for i, c in enumerate(s):
            if c == '(':
                stack.append((c, i))
            elif c == ')':
                if stack and stack[-1][0] == '(':
                    stack.pop()
                else:
                    stack.append((c, i))
        indices = set()
        if stack:
            _, indices = zip(*stack)
            indices = set(indices)
        l = []
        for i, c in enumerate(s):
            if i not in indices:
                l.append(c)
        return ''.join(l)
