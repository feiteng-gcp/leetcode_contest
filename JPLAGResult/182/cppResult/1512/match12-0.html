class UndergroundSystem {
public:
    map<pair<string, string>, int> cost;
    map<pair<string, string>, int> cnt;
    map<int ,pair<int, string>> sta;
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        sta[id] = {t, stationName};
    }
    
    void checkOut(int id, string stationName, int t) {
        string stas = sta[id].second;
        int beg = sta[id].first;
        cnt[{stas, stationName}]++;
        cost[{stas, stationName}] += t-beg;
    }
    
    double getAverageTime(string startStation, string endStation) {
        double ans = 0.0;
        return 1.0*cost[{startStation, endStation}]/cnt[{startStation, endStation}];
    }
};
