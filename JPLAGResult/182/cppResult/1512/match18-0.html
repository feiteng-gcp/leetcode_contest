#include <bits/stdc++.h>
using namespace std;

map<pair<string, string>, int> total;
map<pair<string, string>, int> M;
map<int, pair<int, string>> q;
class UndergroundSystem {
public:

    UndergroundSystem() {
        M.clear();
    }
    
    void checkIn(int id, string stationName, int t) {
        // cout << "came id = " << id << endl;
        q[id] = {t, stationName};
      // q[id].push({t, stationName});
    }
    
    void checkOut(int id, string stationName, int t) {
      //   assert(q[id].size() == 1);
      // while(!q[id].empty()){
         auto it = q[id];//.front();
         M[{it.second, stationName}] += t - it.first;
         total[{it.second, stationName}] += 1;
      // }
    }
    
    double getAverageTime(string startStation, string endStation) {
      return M[{startStation, endStation}] / (1.0 * total[{startStation, endStation}]);
    }
};
