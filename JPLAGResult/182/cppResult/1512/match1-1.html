class UndergroundSystem {
public:
  map<int, string> start_station;
  map<int, int> start_time;
  map<pair<string, string>, int> cnt;
  map<pair<string, string>, int> total_time;
  
  UndergroundSystem() {
  }

  void checkIn(int id, string stationName, int t) {
    start_station[id] = stationName;
    start_time[id] = t;
  }

  void checkOut(int id, string stationName, int t) {
    pair<string, string> p = {start_station[id], stationName};
    ++cnt[p];
    total_time[p] += t - start_time[id];
  }

  double getAverageTime(string startStation, string endStation) {
    pair<string, string> p = {startStation, endStation};
    return (double) total_time[p] / (double) cnt[p];
  }
};

 * UndergroundSystem* obj = new UndergroundSystem();
