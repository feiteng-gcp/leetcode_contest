class UndergroundSystem {
public:
    map<int,int>tim;
    map<int,string>station;
    map< pair<string,string>,pair<int,int> >mp;
    UndergroundSystem() {
        
    }
    
    void checkIn(int id, string stationName, int t) {
        tim[id]=t;
        station[id] = stationName;
    }
    
    void checkOut(int id, string stationName, int t) {
        pair<string,string> cur = {station[id],stationName};

        mp[cur].second++;
        mp[cur].first += (t - tim[id]);
    }
    
    double getAverageTime(string startStation, string endStation) {
        pair<string,string> cur = {startStation, endStation};
        return mp[cur].first*1.0 / mp[cur].second;
    }
};

 * UndergroundSystem* obj = new UndergroundSystem();
 * obj->checkIn(id,stationName,t);
 * obj->checkOut(id,stationName,t);
 * double param_3 = obj->getAverageTime(startStation,endStation);
