class UndergroundSystem {
public:
    map<pair<string, string>, pair<double, double>> cnt;

    map<int, pair< string, int>> time;
    UndergroundSystem() {

    }

    void checkIn(int id, string stationName, int t) {
        time[id] = { stationName,t };
    }

    void checkOut(int id, string stationName, int t) {
        string begin = time[id].first;

        cnt[{begin, stationName}].first += t - time[id].second;
        cnt[{begin, stationName}].second += 1;
    }

    double getAverageTime(string startStation, string endStation) {
        return cnt[{startStation, endStation}].first / cnt[{startStation, endStation}].second;
    }
};
