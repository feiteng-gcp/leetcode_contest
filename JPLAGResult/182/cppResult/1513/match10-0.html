#include <bits/stdc++.h>

int nodenum;
int tree[1001][26];
bool isexist[1001];
int fail[10001];

const int mod = (int )1e9 + 7;
int f[601][601][2][2];
bool vis[601][601][2][2];
string tstr;
int curt;
class Solution {
public:
    void add(string s)
    {
        int now = 1;
        for (int i = 0; i < (int )s.size(); i++)
        {
            int k = s[i] - 'a';
            if (!tree[now][k])
            {
                now = tree[now][k] = nodenum++;
                memset(tree[now], 0, sizeof(tree[now]));
                isexist[now] = 0;
            }
            else now = tree[now][k];
        }
        isexist[now] = true;
    }
    void buildfail()
    {
        fail[1] = 0;
        queue<int> q;
        for (int i = 0; i < 26; i++)
        {
            if (tree[1][i])
            {
                q.push(tree[1][i]);
                fail[tree[1][i]] = 1;
            }
            else tree[1][i] = 1;
        }
        while (!q.empty())
        {
            int now = q.front();
            q.pop();
            isexist[now] |= isexist[fail[now]];
            for (int i = 0; i < 26; i++)
            {
                if (!tree[now][i])
                    tree[now][i] = tree[fail[now]][i];
                else {
                    fail[tree[now][i]] = tree[fail[now]][i];
                    q.push(tree[now][i]);
                }
            }
        }
    }
    int gao(int x, int state, int term, bool flag, bool ok) {
        if (x == term) {
            if (flag && curt == 1) return 0;
            return ok == false;
        }
        if (vis[x][state][flag][ok]) return f[x][state][flag][ok];
        vis[x][state][flag][ok] = true;
        int ed = flag == true ? tstr[x] - 'a' : 25;
        int ans = 0;
        for (int i = 0; i <= ed; i ++) {
            ans += gao(x + 1, tree[state][i], term, flag & (i == ed), ok | isexist[tree[state][i]]);
            ans %= mod;
        }
        return f[x][state][flag][ok] = ans;
    }
    int findGoodStrings(int n, string s1, string s2, string evil) {
        nodenum = 2;
        add(evil);
        buildfail();
        tstr = s2; curt = 0;
        int tmpx = gao(0, 1, n, true, false);
        for (int i = 0; i < n; i ++)
            for (int j = 1; j <= nodenum; j ++)
                for (int k = 0; k < 2; k ++)
                    for (int l = 0; l < 2; l ++) vis[i][j][k][l] = false;
        tstr = s1; curt = 1;
        int tmpy = gao(0, 1, n, true, false);
        for (int i = 0; i < n; i ++)
            for (int j = 1; j <= nodenum; j ++)
                for (int k = 0; k < 2; k ++)
                    for (int l = 0; l < 2; l ++) vis[i][j][k][l] = false;
        for (int j = 1; j <= nodenum; j ++)
            for (int k = 0; k < 26; k ++) tree[j][k] = 0;
        return (tmpx - tmpy + mod) % mod;
    }
};
