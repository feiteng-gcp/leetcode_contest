class UndergroundSystem {

    private Map<Integer, Item> map;
    private Map<String, Double> average;
    private Map<String, Integer> count;
    public UndergroundSystem() {
        map = new HashMap<>();
        average = new HashMap<>();
        count = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        map.put(id, new Item(stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        Item checkin = map.get(id);
        String key = checkin.station + ":" + stationName;
        double prevAvg = average.getOrDefault(key, 0.0);
        int prevCount = count.getOrDefault(key, 0);
        count.put(key, prevCount+1);
        average.put(key, (prevAvg*prevCount + (t-checkin.ts)) / (prevCount+1));
    }
    
    public double getAverageTime(String startStation, String endStation) {
        return average.get(startStation+":"+endStation);
    }
}

class Item {
    public String station;
    public int ts;
    public Item(String s, int t) {
        station = s;
        ts = t;
    }
}
