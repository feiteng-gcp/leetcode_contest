class UndergroundSystem {
    Map<Integer, String> loc;
    Map<Integer, Integer> in;
    Map<String, Integer> avg;
    Map<String, Integer> total;
    String mark = "yuxi";
    public UndergroundSystem() {
        loc = new HashMap<>();
        in = new HashMap<>();
        avg = new HashMap<>();
        total = new HashMap<>();

    }
    
    public void checkIn(int id, String stationName, int t) {
        loc.put(id, stationName);
        in.put(id, t);
    }
    
    public void checkOut(int id, String stationName, int t) {
        int intime = in.get(id);
        String key = loc.get(id) + mark + stationName;
        total.put(key, total.getOrDefault(key, 0) + 1);
        avg.put(key, avg.getOrDefault(key, 0) + (t - intime));
        loc.remove(id);
        in.remove(id);
    }
    //
    public double getAverageTime(String startStation, String endStation) {
        String key = startStation + mark + endStation;
        return ((double) avg.get(key)) / total.get(key); 
    }
}
