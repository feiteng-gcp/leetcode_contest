import java.util.*;
import java.io.*;


class UndergroundSystem {
    Map<String, Time> avgTime;
    Map<Integer, Status> inStation;

    public UndergroundSystem() {
        avgTime = new HashMap<>();
        inStation = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        inStation.put(id, new Status(t, stationName));
    }
    
    public void checkOut(int id, String sn, int t) {
        Status st = inStation.get(id);
        inStation.remove(id);

        String str = st.s + " " + sn;

        if (!avgTime.containsKey(str)) {avgTime.put(str, new Time());}

        Time time = avgTime.get(str);
        time.total += (t - st.t); 
        time.n++;

    }
    
    public double getAverageTime(String startStation, String endStation) {
        String s = startStation + " " + endStation;
        Time t = avgTime.get(s);
        return (double) t.total / t.n;
    }

    class Status {
      int t;
      String s;
      Status(int tt, String ss) {t = tt; s = ss;}
    }

    class Time {
      int n;
      int total;
    }
}
