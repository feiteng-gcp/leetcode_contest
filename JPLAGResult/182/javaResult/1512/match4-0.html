class UndergroundSystem {
    Map<Integer, Q> mp2;

    Map<String, Integer> mp4;
    Map<String, Integer> mp5;
    public UndergroundSystem() {
        mp2 = new HashMap<>();
        mp4 = new HashMap<>();
        mp5 = new HashMap<>();
    }

    public void checkIn(int id, String stationName, int t) {
        mp2.put(id, new Q(stationName, t));
    }

    public void checkOut(int id, String stationName, int t) {
        Q q = mp2.get(id);
        new P(q.pos, stationName, q.time, t);
        String str = q.pos + "#" + stationName;
        mp4.put(str, mp4.getOrDefault(str, 0) + t - q.time);
        mp5.put(str, mp5.getOrDefault(str, 0) + 1);
    }

    public double getAverageTime(String startStation, String endStation) {
        String str = startStation + "#" + endStation;
        int time = mp4.get(str);
        int size = mp5.getOrDefault(str, 0);
        if (size == 0) return 0;
        return 1.0 * time / size;
    }

    class P {
        String start, end;
        int tstart, tend;

        public P(String start, String end, int tstart, int tend) {
            this.start = start;
            this.end = end;
            this.tstart = tstart;
            this.tend = tend;
        }
    }

    class Q {
        String pos;
        int time;

        public Q(String pos, int time) {
            this.pos = pos;
            this.time = time;
        }
    }
}
