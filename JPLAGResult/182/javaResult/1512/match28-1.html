import java.util.AbstractMap;

class UndergroundSystem {

  private Map<AbstractMap.SimpleEntry<String, String>, Integer> xingchengSum;
  private Map<AbstractMap.SimpleEntry<String, String>, Integer> xingchengCnt;
  private Map<Integer, AbstractMap.SimpleEntry<String, Integer>> start;
  public UndergroundSystem() {
    xingchengSum = new HashMap<>();
    xingchengCnt = new HashMap<>();
    start = new HashMap<>();
  }

  public void checkIn(int id, String stationName, int t) {
    start.put(id, new AbstractMap.SimpleEntry<>(stationName, t));
  
}

  public void checkOut(int id, String stationName, int t) {
    AbstractMap.SimpleEntry<String, Integer> e = start.get(id);
    int cost = t - e.getValue();
    AbstractMap.SimpleEntry<String, String> key = new AbstractMap.SimpleEntry<>(e.getKey(), stationName);
    
xingchengSum.put(key, xingchengSum.getOrDefault(key, 0) + cost);
    xingchengCnt.put(key, xingchengCnt.getOrDefault(key, 0) + 1);
  }

  public double getAverageTime(String startStation, String endStation) {
    AbstractMap.SimpleEntry<String, String> key = new AbstractMap.SimpleEntry<>(startStation, endStation);
    return 1d * xingchengSum.get(key) / xingchengCnt.get(key);
  }
}
