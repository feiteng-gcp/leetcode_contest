class UndergroundSystem {

    class Data{
        
        int people;
        double avg;
        
        public Data(){
            this.people = 0;
            this.avg = 0.0;
        }
        
        public void update(int time){
            this.avg = (double)((this.avg * this.people) + ((double) time))/(this.people + 1);
            this.people++;
        }
        
    }
    
    class Commuter{
        
        String station;
        int checkIn;
        
        public Commuter(String station, int checkIn){
            this.station = station;
            this.checkIn = checkIn;
        }
        
    }
    
    private Map<String, Map<String, Data>> map;
    private Map<Integer, Commuter> commuters;
    
    public UndergroundSystem() {
        map = new HashMap<>();
        commuters = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        commuters.put(id, new Commuter(stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        Commuter c = commuters.get(id);
        String st1 = c.station;
        if(!map.containsKey(st1)) map.put(st1, new HashMap<>());
        if(!map.get(st1).containsKey(stationName)) map.get(st1).put(stationName, new Data());
        map.get(st1).get(stationName).update(t - c.checkIn);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        return map.get(startStation).get(endStation).avg;
    }
}
