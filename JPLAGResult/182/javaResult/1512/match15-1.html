class UndergroundSystem {
    
class Pair {
        String station;
        int time;
        Pair (String station, int time) {
            this.station = station;
            this.time = time;
        }
    }
    
    
Map<Integer, Pair> user_map;
    Map<String, Long> station_time_map;
    Map<String, Long> station_count_map;
    public UndergroundSystem() {
        user_map = new HashMap<>();
        station_time_map = new HashMap<>();
        station_count_map = new HashMap<>();
    }
    
    public void checkIn(int id, String stationName, int t) {
        user_map.put(id, new Pair(stationName, t));
    }
    
    public void checkOut(int id, String stationName, int t) {
        Pair prev = user_map.get(id);
        String key = prev.station + "->" + stationName;
        long duration = t - prev.time;
        
        station_time_map.put(key, 
station_time_map.getOrDefault(key, 0l) + duration);
        station_count_map.put(key, station_count_map.getOrDefault(key, 0l) + 1);
    }
    
    public double getAverageTime(String startStation, String endStation) {
        String key = startStation + "->" + endStation;
        if (station_time_map.get(key) == null) {
            return 0;
        }
        
        double average = 1.0 * station_time_map.get(key) / (station_count_map.get(key));
        return average;
    }
}
