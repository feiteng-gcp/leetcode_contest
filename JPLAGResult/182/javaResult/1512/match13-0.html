class UndergroundSystem {

    Map<String, Double> totalTime;
    Map<String, Integer> count;
    Map<Integer, String> inUserStation;
    Map<Integer, Integer> inUserTime;

    public UndergroundSystem() {
        totalTime = new HashMap<>();
        count = new HashMap<>();
        inUserStation = new HashMap<>();
        inUserTime = new HashMap<>();
    }

    public void checkIn(int id, String stationName, int t) {
        inUserStation.put(id, stationName);
        inUserTime.put(id, t);
    }

    public void checkOut(int id, String stationName, int t) {
        String from = inUserStation.get(id);
        String key = from + "-" + stationName;
        int time = t - inUserTime.get(id);
        totalTime.put(key, totalTime.getOrDefault(key, 0.0) + time);
        count.put(key, count.getOrDefault(key, 0) + 1);
    }

    public double getAverageTime(String startStation, String endStation) {
        String key = startStation + "-" + endStation;
        double total = totalTime.get(key);
        return total / count.get(key);
    }
}
