class UndergroundSystem {

    Map<Integer, String> checkInStatation = new HashMap<>();
    Map<Integer, Integer> checkInTime = new HashMap<>();

    Map<String, Long> totalTime = new HashMap<>();
    Map<String, Integer> totalCount = new HashMap<>();
    public UndergroundSystem() {

    }

    public void checkIn(int id, String stationName, int t) {
        checkInStatation.put(id, stationName);
        checkInTime.put(id, t);
    }

    public void checkOut(int id, String stationName, int t) {
        String startStation = checkInStatation.get(id);
        int startT = checkInTime.get(id);

        
long range = t - startT;
        String pair = startStation + "," + stationName;
        totalTime.put(pair, totalTime.getOrDefault(pair, 0L) + range);
        totalCount.put(pair, totalCount.getOrDefault(pair, 0) + 1);
    }

    public double getAverageTime(String startStation, String endStation) {

        String pair = startStation + "," + endStation;
        if (!totalTime.containsKey(pair)) {
            return 0.0;
        }
        return (double) totalTime.get(pair) / (double) totalCount.get(pair);
    }
}
