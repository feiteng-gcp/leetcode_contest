class UndergroundSystem:

    def __init__(self):
        self.avg = {}
        self.check = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.check[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        ent = self.check[id]
        dt = t - ent[1]
        link = ( ent[0], stationName )
        if link in self.avg:
            self.avg[link] = ( self.avg[link][0] + dt, self.avg[link][1] + 1 )
        else:
            self.avg[link] = ( dt, 1 )

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        conn = self.avg[ ( startStation, endStation ) ]
        return conn[0] / conn[1]
