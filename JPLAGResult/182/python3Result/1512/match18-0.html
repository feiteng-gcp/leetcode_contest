class UndergroundSystem:

    def __init__(self):
        self.avgt={}
        self.inline={}

    def checkIn(self, idd: int, stationName: str, t: int) -> None:
        self.inline[idd]=(stationName,t)

    def checkOut(self, idd: int, stationName: str, t: int) -> None:
        (start_sta,start_t)=self.inline[idd]
        if (start_sta, stationName) in self.avgt:
            (acc_t,people)= self.avgt[(start_sta, stationName)]
            new_time=(acc_t*people+t-start_t)/(people+1)
            self.avgt[(start_sta, stationName)]=(new_time,people+1)
        else:
            self.avgt[(start_sta, stationName)]=(t-start_t,1)

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.avgt[(startStation,endStation)][0]
