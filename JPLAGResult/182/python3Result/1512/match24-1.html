class UndergroundSystem:

    def __init__(self):
        self.travels = dict()
        self.sum_cnt = dict()
        

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.travels[id] = (stationName, t)
        

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        start_stn, start_t = self.travels[id]
        duration = t-start_t
        s, cnt = self.sum_cnt.get((start_stn, stationName), (0,0))
        s, cnt 
= s+duration, cnt+1
        self.sum_cnt[(start_stn, stationName)] = (s,cnt)
    
        

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        s, cnt = self.sum_cnt[(startStation, endStation)]
        return s/cnt
