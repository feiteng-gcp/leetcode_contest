class UndergroundSystem:

    def __init__(self):
        self.records = {}
        self.timeSummations = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.records[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        startStation, startTime = self.records[id]
        endStation, endTime = stationName, t
        path = (startStation, endStation)
        duration = endTime - startTime

        
if path in self.timeSummations:
            summationTime, summationCount = self.timeSummations[path]
            self.timeSummations[path] = [summationTime + duration, summationCount + 1]
        else:
            self.timeSummations[path] = [duration, 1]

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        path = (startStation, endStation)
        time, count = self.timeSummations[path]
        return time / count
