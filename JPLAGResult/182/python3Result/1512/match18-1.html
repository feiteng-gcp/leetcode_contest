class UndergroundSystem:

    def __init__(self):
        self.it = {}
        self.times = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.it[id] = (stationName, t)

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        (name, t0) = self.it[id]
        if (name, stationName) in self.times:
            (a, b) = self.times[(name, stationName)]
            self
.times[(name, stationName)] = (a+t-t0, b+1)
        else:
            self.times[(name, stationName)] = (t-t0, 1)
        

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        (a, b) = self.times[(startStation, endStation)]
        return a/b
