from collections import Counter
class UndergroundSystem:

    def __init__(self):
        self.users = {}
        self.stations_total = Counter()
        self.stations_count = Counter()

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.users[id] = (stationName, t)
        

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        in_station, t_0 = self.users.pop(id)
        self.stations_total[(in_station, stationName)] += t - t_0
        self.stations_count[(in_station, stationName)] += 1
        

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return self.stations_total[(startStation, endStation)] / self.stations_count[(startStation, endStation)]
