class UndergroundSystem:

    def __init__(self):
        self.route = {}
        self.id = {}

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        self.id[id] = [stationName,t]

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        if self.id[id][0] not in self.route:
            self.route[self.id[id][0]] = {}
        if stationName not in self.route[self.id[id][0]]:
            self.route[self.id[id][0]][stationName] = [0,0]  #第一个表示次数，第二个表示总时间
        self.route[self.id[id][0]][stationName][0] += 1
        self.route[self.id[id][0]][stationName][1] += (t - self.id[id][1])

    def getAverageTime(self, startStation: str, endStation: str) -> float:
        return  self.route[startStation][endStation][1]/self.route[startStation][endStation][0]
