class UndergroundSystem:

    def __init__(self):
        self.a={}
        self.b={}

    def checkIn(self, id: int, s: str, t: int) -> None:
        self.a[id]=(s,t)


    def checkOut(self, id: int, e: str, t: int) -> None:
        x,y=self.b.get((
self.a[id][0],e)) or (0,0)
        self.b[(self.a[id][0],e)]=(x+1,y+(t-self
.a[id][1]))

    def getAverageTime(self, s: str, e: str) -> float:
        return self.b[(s,e)][1]/self.b[(s,e)][0]
