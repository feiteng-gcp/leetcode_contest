from collections import defaultdict
class UndergroundSystem:

    def __init__(self):
        self.g = defaultdict(dict)
        self.p = dict()


    def checkIn(self, id_: int, sn: str, t: int) -> None:
        self.p[id_] = (sn, t)

    def checkOut(self, id_: int, sn: str, t: int) -> None:
        ssn, st = self.p[id_]
        del self.p[id_]
        tot, cnt = self.g[ssn].get(sn, (0, 0))
        tot += t - st
        cnt += 1
        self.g[ssn][sn] = (tot, cnt)

    def getAverageTime(self, st: str, en: str) -> float:
        tot, cnt = self.g[st][en]
        return tot / cnt
