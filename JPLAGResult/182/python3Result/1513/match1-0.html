class Solution:
    def findGoodStrings(self, n: int, s0: str, s1: str, s2: str) -> int:            
        f = [[[[0, 0], [0, 0]] for _ in range(55) ] for _ in range(505)]
        mod = 10**9 + 7
        kmp = [0] * 505
        m = len(s2)
        s2 = ' ' + s2
        i, j = 2, 0
                    
        while i <= m:
            while j > 0 and s2[i] !=s2[j+1]: j=kmp[j]
            if s2[i] == s2[j+1]: j += 1
            kmp[i] = j
            i += 1
        f[0][0] = [[1, 0], [0, 0]]
        for i in range(n):
            for j in range(m):
                for l in [0, 1]:
                    for s in [0, 1]:
                        for kk in range(26):
                            k = chr(ord('a') + kk)
                            if not l and k < s0[i]: continue
                            if not s and k > s1[i]: continue
                            # print(k)
                            u = j
                            while u > 0 and k != s2[u + 1]: 
                                u = kmp[u]
                            if k == s2[u + 1]:
                                u += 1
                            if u >= m: continue
                            ll = int(l or (k > s0[i]))
                            ss = int(s or (k < s1[i]))
                            f[i + 1][u][ll][ss] = (f[i][j][l][s] + f[i + 1][u][ll][ss]) % mod
        return sum(g[0][0] + g[0][1] + g[1][0] + g[1][1] for g in f[n]) % mod
