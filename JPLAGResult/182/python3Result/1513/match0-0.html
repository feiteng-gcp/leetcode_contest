class Solution:
    def findGoodStrings(self, n: int, s1: str, s2: str, evil: str) -> int:
        from collections import defaultdict
        mod=1000000007
        trans=[[0 for i in range(26)] for j in range(len(evil))]
        trans[0][ord(evil[0])-ord('a')]=1
        second=0
        for i in range(1,len(evil)):
            for j in range(ord('a'),ord('z')+1):
                if j==ord(evil[i]):
                    trans[i][j-ord('a')]=i+1
                else:
                    trans[i][j-ord('a')]=trans[second][j-ord('a')]
            second=trans[second][ord(evil[i])-ord('a')]


        dp={(True,True,0):1} #s1,s2,evil
        for i in range(n):
            dpnext=defaultdict(int)
            for k,v in dp.items():
                check1,check2,checkE=k
                start=ord(s1[i]) if check1 else ord('a')
                end=ord(s2[i]) if check2 else ord('z')
                for j in range(start,end+1):
                    c=chr(j)
                    next1=(c==s1[i] and check1)
                    next2=(c==s2[i] and check2)
                    nextE=trans[checkE][j-ord('a')]
                    if nextE<len(evil):
                        dpnext[(next1,next2,nextE)]+=v
                        dpnext[(next1,next2,nextE)]%=mod
            dp=dpnext.copy()
        return sum(dp.values())%mod
