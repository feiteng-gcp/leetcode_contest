class Solution:
    def numTeams(self, rating: List[int]) -> int:
        size = len(rating)
        res = 0
        for i in range(size-2):
            for j in range(i+1,size-1):
                for k in range(j+1,size):
                    if rating[i]<rating[j]<rating[k] or rating[i]>rating[j]>rating[k]:
                        res += 1
        return res
