class Solution {
public:
    struct Task{
        int difficulty, profit;
        Task(int d, int p):difficulty(d), profit(p){}
    };
    

    int maxProfitAssignment(vector<int>& difficulty, vector<int>& profit, vector<int>& worker) {
        int n=difficulty.size();
        vector<Task> tasks;
        for(int i=0;i<n;i++){
            tasks.push_back(Task(difficulty[i], profit[i]));
        }
        sort(tasks.begin(), tasks.end(), [&](const Task &t1, const Task &t2){
            if(t1.profit==t2.profit){
                return t1.difficulty < t2.difficulty;
            }
            return t1.profit > t2.profit;
        });
        
        //按照profit倒序,难度升序排列
        int sum=0;
        for(int k=0;k<worker.size();k++){
            //每一个员工,选取最大化利益的1个工作
            for(int i=0;i<tasks.size();i++){
                if(worker[k]<tasks[i].difficulty) continue;
                sum+=tasks[i].profit;break;//break when find one
            }
        }
        return sum;
    }
};
