struct jobs{
    int difficulty;
    int profit;
};
class CompGreater  
     {  
     public:  
         bool operator ()(const jobs& stItem1, const jobs& stItem2)  

         {  
             return stItem1.profit > stItem2.profit;  
         }  
     }; 

class Solution {
public:
    
    int maxProfitAssignment(vector<int>& difficulty, vector<int>& profit, vector<int>& worker) {
        int dif = 0;
        int ans = 0;
        
        vector<jobs> jobs_v;
        for(int i = 0; i < difficulty.size(); ++i){
            jobs tmp;
            tmp.difficulty = difficulty[i]; tmp.profit = profit[i];
            jobs_v.push_back(tmp);
        }
        sort(jobs_v.begin(), jobs_v.end(), CompGreater());
        sort(worker.begin(), worker.end(), greater<int>());
        
//         for(int i = 0; i < jobs_v.size(); ++i){
//             cout << jobs_v[i].profit << endl;
//         }
//         for(int i : worker)
//             cout << i << endl;
        
        for(int i = 0; i < worker.size(); ++i){
            while(jobs_v[dif].difficulty > worker[i]){
                ++dif;
            }
            if(dif >= jobs_v.size())
                return ans;
            ans += jobs_v[dif].profit;
        }
        
        return ans;
    }
};
