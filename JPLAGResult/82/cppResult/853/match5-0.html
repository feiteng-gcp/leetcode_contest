#include <algorithm>
#include <utility>
bool myfun(pair<int, int>& a, pair<int, int>& b) {
    if (a.second < b.second) {
        return true;
    }
    else if (a.second > b.second) {
        return false;
    }
    else {
        return a.first < b.first;
    }
}

class Solution {
public:
    int maxProfitAssignment(vector<int>& difficulty, vector<int>& profit, vector<int>& worker) {
        vector<pair<int, int> > pd;
        int pr = 0;
        for (int i = 0; i < difficulty.size(); i++) {
            pd.push_back(make_pair(difficulty[i], profit[i]));
        }
        sort(pd.begin(), pd.end(), myfun);
        sort(worker.begin(), worker.end());
        /*for (auto p : pd) {
            cout << p.first << " " << p.second << endl;
        }
        for (auto w : worker) {
            cout << w << endl;
        }*/
        int j = difficulty.size()-1;
        for (int i = worker.size()-1; i>=0; i--) {
            //find a job for worker i
            while ((worker[i] < pd[j].first) && j >= 0) {
            //    cout << worker[i] << " " << pd[j].first << endl;
                j--;
            }
            //cout << "j " << j << endl;
            if (j >= 0) {
                pr += pd[j].second;
            }
            else {
                break;
            }
        }
        return pr;
    }
};
