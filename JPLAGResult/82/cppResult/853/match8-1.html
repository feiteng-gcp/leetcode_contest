struct node{
    int difficulty, profit;
    node(int difficulty, int profit):difficulty(difficulty), profit(profit){};
    bool friend operator < (const node n1, const node n2){
        return n1.difficulty < n2.difficulty;
    }
};
vector<node> vec;
const int maxn = 10000+10;
int pre[maxn];

class Solution {
public:
    int maxProfitAssignment(vector<int>& difficulty, vector<int>& profit, vector<int>& worker) {
        vec.clear();
        int len = difficulty.size();
        for(int i=0;i<len;i++){
            vec.push_back(node(difficulty[i], profit[i]));
        }
        sort(vec.begin(), vec.end());
        pre[0] = vec[0].profit;
        
        for(int i=1;i<len;i++){
            pre[i] = max(vec[i].profit, pre[i-1]);
        }
        int ans=0;
        for(int i=0;i<worker.size();i++){
            int l=0, r=len-1, num=worker[i];
            while(l<r){
                int mid = (l+r)/2;
                if(vec[mid].difficulty <= num){
                    l = mid+1;
                }else{
                    r = mid;
                }
            }
            //cout << i << " " << vec[l].difficulty << " ";
            if(vec[l].difficulty > num)
                l--;
            if(l>=0){
                //cout << pre[l];
                ans += pre[l];
            }
            //cout << endl;
        }
        return ans;
    }
};
