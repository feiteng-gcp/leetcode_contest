struct zz
{
    int d;
    int p;
    zz (int dd, int pp)
    {
        d=dd;
        p=pp;
    }
};
bool cmp(zz a, zz b)
{
    return a.d<b.d;
}

class Solution {
public:
    
    int maxProfitAssignment(vector<int>& d, vector<int>& p, vector<int>& w) 
    {
        int n=d.size();
        vector<zz>v;
        for(int i=0;i<n;i++)
        {
            zz t(d[i],p[i]);
            v.push_back(t);
        }
        sort(v.begin(),v.end(),cmp);
        
        for(int i=0;i<n-1;i++)
        {
           if(v[i+1].p<v[i].p)
           {
               v[i+1].p=v[i].p;
           }
        }
        
        int ret=0;
        for(int i=0;i<w.size();i++)
        {
            int x=w[i];
            
            if(v[0].d>x)
                continue;
            int l=0,r=n;
            while(l<r-1)
            {
                int mid=(l+r)>>1;
                if(v[mid].d<=x)
                {
                    l=mid;
                }
                else
                    r=mid;
            }
            
            ret+=v[l].p;
        }
        return ret;
    }
};
