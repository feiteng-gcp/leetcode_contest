class Solution {
public:
   int maxProfitAssignment(vector<int>& difficulty, vector<int>& profit, vector<int>& worker) {
	map<int, int> mp, idx;
	int n = profit.size();
	if (!n) return 0;
	int res = 0;
	for (int i = 0; i < n; ++i) {
		if (idx[difficulty[i]] && profit[i] < profit[idx[difficulty[i]]]) continue;
		else idx[difficulty[i]] = i;
	}
	sort(difficulty.begin(), difficulty.end());
	mp[difficulty[0]] = profit[idx[difficulty[0]]];
	for (int i = 1; i < n; ++i) {
		mp[difficulty[i]] = max(mp[difficulty[i - 1]], profit[idx[difficulty[i]]]);
	}
	int m = worker.size();
	for (int i = 0; i < m; ++i) {
		auto pos = upper_bound(difficulty.begin(),difficulty.end(),worker[i]);
		int t = 0;
		if (pos != difficulty.begin()) pos--;
		else continue;
		res  += mp[*pos];
	}
	return res;
}
};
