class Solution {
public:
    int maxProfitAssignment(vector<int>& d, vector<int>& p, vector<int>& worker) {
        map<int, int> dp; // <diffi, profit>
        int res = 0;
        
        if (d.empty()) return 0;
        
        for (int i=0; i<d.size(); i++) {
            if (dp.find(d[i])!= dp.end())
                dp[d[i]] = max(dp[d[i]], p[i]);
            else 
                dp.insert({d[i], p[i]});
        }

        for (auto it = dp.begin(); it!= dp.end(); ) {
            auto it_p = it++; // it_p, it
            if (it->second < (it_p)->second)
                it->second = (it_p)->second;
        }
        for (auto w : worker) {
            auto it = dp.upper_bound(w); 
            if (it == dp.begin()) continue; // can not complete any
            it--;
            res += it->second;
        }
        return res;
        
    }
};
