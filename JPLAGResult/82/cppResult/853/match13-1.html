class Solution {
public:
    int maxProfitAssignment(vector<int>& difficulty, vector<int>& profit, vector<int>& worker) {
        //1.for same diff, choose the max profit
        unordered_map<int, int> m; // <min_diff, profit>
        for (int i = 0; i < difficulty.size(); i++) {
            if (m.find(difficulty[i]) == m.end()) {
                m[difficulty[i]] = profit[i];
            } else {
                m[difficulty[i]] = max (m[difficulty[i]], profit[i]);
            }
        }
        vector<pair<int,int>> v;
        for (auto it : m) {
            v.push_back(pair<int,int>{it.first, it.second});
        }
        sort(v.begin(), v.end(), [](pair<int,int>& p1, pair<int,int>&p2){
            return p1.first < p2.first;
        });
        vector<int> vp ((int)v.size(),0); // max profit when diff <= v[i]
        vector<int> vd ((int)v.size(),-1);
        int mp = -1;
        for (int i = 0; i < v.size(); i++) {
            mp = max(mp, v[i].second); 
            vp[i] = mp;
            vd[i] = v[i].first;
            //cout << vp[i] << endl;
        }
        int res = 0;
        for (int i = 0 ; i < worker.size(); i++) {
            auto up = std::upper_bound(vd.begin(), vd.end(), worker[i]);
            int pid = -1;
            if (up == vd.end()) {
                pid = vd.size()-1;
                //cout << "workder too strong" << endl;
            } else {
                pid = (up-vd.begin())-1;
                //cout << (up-vd.begin())-1 << endl;
            }
            if (pid >=0 ) {
                res += vp[pid];
            }
        }
        return res;
    }
    
};
