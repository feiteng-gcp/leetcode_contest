#include <algorithm>

#define MAXN 200
#define xx first 
#define yy second 
#define mp make_pair
#define pb push_back
#define MOD 1000000007
#define INF (1<<30)
#define prime 37
#define eps 0.0000001
using namespace std;
class Solution {
public:
	int n , m;
	vector<pair<int,int> > pr;
	int maxProfitAssignment(vector<int>& diff, vector<int>& p, vector<int>& w) {
        n = diff.size();
        m = w.size();
        for (int i = 0; i < n; i ++ )
        	pr.pb( mp(diff[i] , p[i]) );
		sort(pr.begin() , pr.end() );
		sort(w.begin() , w.end());
		for (int i = 1; i < n; i ++ ) {
			pr[i].yy = max(pr[i].yy , pr[i - 1].yy);
		}
		int idx = 0;
		int res = 0;
		for (int i = 0; i < m; i ++ ) {
			while ( idx + 1 < n && w[i] >= pr[idx + 1].xx ) {
				idx ++;
			}
			if (pr[idx].xx <= w[i] ) {
				res = res + pr[idx].yy;
			}
		}
		return res;
    }
};
