class Solution {
    
public:
    struct S{
        int d;
        int p;
    };
    static bool f(S s1, S s2){
        return s1.d < s2.d;
    }
    int maxProfitAssignment(vector<int>& difficulty, vector<int>& profit, vector<int>& worker) {
        
        vector<int> m;
        vector<S> v;
        int k = profit.size();
        for(int i = 0; i < k; i++){
            S s;
            s.d = difficulty[i];
            s.p = profit[i];
            v.push_back(s);
        }
        sort(v.begin(),v.end(),f);
        
        int ma = -1;
        vector<int> d;
       // vector<int> p;
        for(int i = 0; i < k; i++){
            if(v[i].p > ma)
                ma = v[i].p;
            m.push_back(ma);
            d.push_back(v[i].d);
          //  p.push_back(v[i].p);
        }
        

        
        int n = worker.size();
        int sum = 0;
        for(int i = 0; i < n; i++){
            std::vector<int>::iterator low;
            low=std::lower_bound (d.begin(), d.end(), worker[i]+1);
            int j = low-d.begin();
            sum += m[j-1];
        }
        
        return sum;
    }
};
