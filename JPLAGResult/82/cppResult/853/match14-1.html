class Solution {
public:
    int maxProfitAssignment(vector<int>& d, vector<int>& p, vector<int>& w) {
        vector<pair<int, int>> vec;
        for(int i = 0; i < d.size(); i ++) {
            vec.push_back(make_pair(d[i], p[i]));
        }
        
        sort(vec.begin(), vec.end());

        int nmax = 0;
        for(int i = 0; i < vec.size(); i ++) {
            nmax = max(nmax, vec[i].second);
            vec[i].second = nmax;
        }
        int ans = 0;
        for(int i = 0; i < w.size(); i ++) {
            ans += binary(w[i], vec);
        }
        
        return ans;
    }
    
    int binary(int d, vector<pair<int, int>>& vec) {
        int lf = 0, rt = vec.size();

        while(lf < rt) {
            int m = lf + (rt - lf) / 2;
            if(vec[m].first > d) {
                rt = m;
            } else {
                lf = m + 1;
            }
        }
        //cout << d << " " << vec[m-1].first << " " << vec[m-1].second << endl;
        return vec[rt-1].second;
    }
};
