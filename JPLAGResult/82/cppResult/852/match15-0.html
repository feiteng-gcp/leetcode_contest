class Solution {
public:
    int numFriendRequests(vector<int>& ages) {
        sort(ages.begin(),ages.end());
        int n = ages.size();
        int res = 0;
        for(int i = n-1; i > 0; --i) {
            double t = 0.5 * ages[i] + 7;
            for(int j = i - 1; j >= 0; j--) {
                if((double)ages[j] <= t) break;
                if(ages[i] == ages[j]) res += 2;
                else res++;
            }
        }
        return res;
    }
};
