class Solution {
    int agea[150];
    
public:
    int numFriendRequests(vector<int>& ages) {
        memset(agea,0,sizeof(agea));
        
        for(int &age:ages)
        {
            agea[age]++;
        }
        
        int result=0;
        for(int A=1;A<=120;A++)
        {   
            for(int B=1;B<=120;B++)
            if(true){
                bool cond1 = 2 * (B - 7) <= A;
                bool cond2 = B > A;
                bool cond3 = B > 100 && A < 100;
                
                if(cond1 || cond2 || cond3) ;
                else{
                    if(B!=A)
                        result += agea[A]*agea[B];
                    else
                        result += (agea[A]*(agea[A]-1));
                }
            }
        }
        
        return result;
    }
};
