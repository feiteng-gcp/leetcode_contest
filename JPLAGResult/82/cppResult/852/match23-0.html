class Solution {
public:
   // int f(vector<int>& ages,int from,int to, float n){
        
    //}
    int numFriendRequests(vector<int>& ages) {
        sort(ages.begin(),ages.end());
        int s = ages.size();
        int sum = 0;
        for(int i = 0; i < s; i++){
            if(ages[i] <= 14)
                continue;
            for(int j = i+1; j<s; j++){
                if(j == s-1 && ages[i] == ages[j]){
                    int k = j+1-i;
                    sum +=k*(k-1)/2;
                    i = j;
                }
                else if(ages[i] == ages[j]){
                    continue;
                }
                else{
                    int k = j-i;
                    sum += k*(k-1)/2;
                    i = j;
                }
            }
                
        }
        for(int i = 0; i < s; i++){
            if(ages[i] <= 14)
                continue;
            float num = 0.5*(float)ages[i]+7;
            //int k = f(ages,i,num);
            std::vector<int>::iterator low;
            low=std::lower_bound (ages.begin(), ages.begin()+i, floor(num)+1);
            //cout<<low-ages.begin()<<endl<<i<<endl;
            sum += i-(low-ages.begin());
        }
        return sum;
    }
};
