class Solution {
public:
    int numFriendRequests(vector<int>& ages) 
    {
        int count[121]{};
        for(auto age : ages)
        {
            count[age]++;
        }
        int res = 0;
        for(int A = 1;A<=120;A++)
        {
            for(int B = 1;B<A;B++)
            {
                if(B > A*0.5+7)
                {
                    res += count[A]*count[B];
                }
            }
        }
        for(int i =15;i<=120;i++)
        {
            if(count[i] > 1 )
                res += (count[i]*(count[i]-1));
        }
        return res;
    }
};
