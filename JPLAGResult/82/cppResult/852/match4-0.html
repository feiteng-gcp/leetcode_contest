using namespace std;
class Solution {
	int h[MAXN];
	bool used[MAXN];
	public:int numFriendRequests(vector<int>& ages) {
    	int n = ages.size();
		int AGE = 120;
		for (int i = 0; i <= AGE; i ++ ){	
			h[i] = 0;
			used[i] = 0;
		}
		for (int i = 0; i < n; i ++ )
			h[ages[i]] ++;
		long long res = 0;
		/*
		age[B] <= 0.5 * age[A] + 7
		age[B] > age[A]
		age[B] > 100 && age[A] < 100
		*/
		for (int a = 0; a <= AGE; a ++ ){
			for (int b = 0; b <= AGE; b ++) {
				if ( b > 0.5 * a + 7 && a >= b && !( b > 100 && a < 100 ) ) {
					if (a != b)
						res = res + h[a] * h[b];
					else {
						if ( !used[a] ) {
							res = res + (1LL * h[a] * (h[a] - 1) );
							used[a] = 1; 
						}
					}
				}
			}		
		}
		return res;	
    }
};
