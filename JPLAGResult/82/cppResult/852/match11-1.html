class Solution {
public:
    int numFriendRequests(vector<int>& ages) {
        vector<int> v(121, 0); // counts of age[1-120], [0] ignored

        for (auto a : ages) {
            v[a]++;
        }
        int res = 0;
        for (int A = 1; A <= 120; A++) {
            for (int B = 0.5 * A + 7; B <= A; B++) {                
                if (B <= 0.5 * A + 7) continue;
                if (B > 100 && A < 100) continue;
                if (B != A) {
                    res += v[A] * v[B];
                } else {
                    res += v[A] * (v[A] - 1);
                }
            }
        }
        return res;
    }
};
