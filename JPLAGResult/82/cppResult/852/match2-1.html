class Solution {
public:
    int numFriendRequests(vector<int>& ages) {
        sort(ages.begin(), ages.end());
        int rst = 0;
        int count = 0; // same ages appeared how many times
        for (int i = 0; i < ages.size(); ++i)
        {
            count = 1;
            while(i+1 < ages.size() && ages[i+1] == ages[i]) {i++; count++;}
            int lower = 0.5 * ages[i] + 7;
            for (int j = 0; j < i; j++)
            {
                if (ages[j] > lower)
                {
                    rst += (i - j) * count;
                    break;
                }
            }
        }
        return rst;
    }
};
