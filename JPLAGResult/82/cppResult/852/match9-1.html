class Solution {
public:
    int numFriendRequests(vector<int>& ages) {
        sort(ages.begin(),ages.end());
        int ans = 0;
        int len = ages.size();
        for(int i=0;i<len;i++){
            int ind = lower_bound(ages.begin()+i+1, ages.end(), 2*(ages[i]-7)) - (ages.begin()+i+1);
            ans += ind;
            if(ages[i]>100){
                ind = lower_bound(ages.begin()+i+1, ages.end(), 100) - (ages.begin()+i+1);
                ans -= ind;
            }
            if(ages[i]>14){
                ind = upper_bound(ages.begin()+i+1, ages.end(), ages[i]) - (ages.begin()+i+1);
                if(ages[ind]>ages[i])
                    ind--;
                ans += ind;
            }
        }
        return ans;
    }
};
