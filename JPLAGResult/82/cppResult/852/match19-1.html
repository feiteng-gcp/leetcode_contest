class Solution {
public:
    int numFriendRequests(vector<int>& ages) {
        sort(ages.begin(), ages.end());
        int pos = 0;
        int ans = 0;
        for(int i = 1; i < ages.size(); ++i){
            // for(int j = pos; j < i; ++j){
            //     if(ages[j] <= 0.5 * (float)ages[i] + 7){
            //         pos += 1;
            //         continue;
            //     }           
            //     if(ages[j] == ages[i])
            //         ++ans;
            //     ++ans;
            // }
            
            while(pos < i && ages[pos] <= 0.5 * (float)ages[i] + 7)
                ++pos;
            if(ages[pos] == ages[i])
                ans += 2 * (i - pos);
            else{
                ans += i - pos;
                for(int j = pos + 1; j < i; ++j){
                    if(ages[j] == ages[i]){
                        ans += i - j;
                        break;
                    }
                }
            }
        }
        return ans;
    }
};
