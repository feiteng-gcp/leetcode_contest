class Solution {
public:
    string toGoatLatin(string S) {
        stringstream ss(S);
        string token;
        string ans = "";
        string as = "a";
        while(ss >> token) {
            if(ans != "") ans  = ans + " ";
            char c = token[0];

            if (c >='A' and c<='Z') c = c-'A' + 'a';
            if (c == 'a' or c=='e' or c=='i' or c=='o' or c=='u') {
                ans = ans + token + "ma" + as;
            } else {
                ans = ans + token.substr(1) + token[0] + "ma" + as;
            }
            as = as + "a";
        }
        return ans;
    }
};
