class Solution {
private: 
    bool isAlphabet (char ch) {
        return (ch >= 'A' && ch <= 'Z') || (ch >= 'a' && ch <= 'z'); 
    }
public:
    string toGoatLatin(string S) {
        vector<string> wordVec; 
        int lPtr = 0, rPtr = 0, sLen = S.length(); 
        
        while (lPtr < sLen && rPtr < sLen) {
            while (lPtr < sLen && S[lPtr] == ' ') {
                lPtr ++; 
            }
            rPtr = lPtr; 
            while (rPtr + 1 < sLen && isAlphabet(S[rPtr + 1])) {
                rPtr ++; 
            }
            wordVec.push_back(S.substr(lPtr, rPtr - lPtr + 1)); 
            rPtr ++; 
            lPtr = rPtr; 
        }
        int wordCnt = wordVec.size(); 
        for (int i = 0; i < wordCnt; ++ i) {

            if (wordVec[i][0] == 'a' || wordVec[i][0] == 'A' || wordVec[i][0] == 'e' || wordVec[i][0] == 'E' || wordVec[i][0] == 'i' || wordVec[i][0] == 'I' || wordVec[i][0] == 'o' || wordVec[i][0] == 'O' || wordVec[i][0] == 'u' || wordVec[i][0] == 'U' ) {
                wordVec[i] += "ma"; 
            }
            else {
                string tmpStr = wordVec[i].substr(1); 
                tmpStr.push_back(wordVec[i][0]); 
                wordVec[i] = tmpStr + "ma"; 
            }
            
            for (int j = 0; j <= i; ++ j) {
                wordVec[i].push_back('a'); 
            }
        }
        
        string ansStr; 
        for (string& word : wordVec) {
            ansStr += word; 
            ansStr.push_back(' '); 
        }
        if (ansStr.length()) ansStr.pop_back(); 
        
        return ansStr; 
    }
};
