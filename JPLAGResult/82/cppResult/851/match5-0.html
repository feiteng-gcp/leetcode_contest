vector<string> split(const string&s, char c=' ') {
    int p=0;
    vector<string>ret;
    for (int i=0; i<int(s.size()); i++) {
	if (s[i]==c) {
	    ret.push_back(s.substr(p, i-p));
	    p=i+1;
	}
    }
    ret.push_back(s.substr(p));
    return ret;
}

bool isVowel(char c) {
    c = tolower(c);
    return c == 'a'
	|| c == 'e'
	|| c == 'i'
	|| c == 'o'
	|| c == 'u';
}

class Solution {
public:
    string toGoatLatin(string S) {
	vector<string> W = split(S);
	string ans;
	REP (i, W.size()) {
	    if (i) ans += " ";

	    char c = W[i][0];
	    string tmp;
	    if (isVowel(c)) {
		ans += W[i] + "ma";
	    } else {
		ans += W[i].substr(1) + string(1, c) + "ma";
	    }

	    ans += string(i+1, 'a');
	}
        
	return ans;
    }
};
