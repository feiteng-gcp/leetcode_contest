class Solution {
public:
    bool vow(char ch){
        ch=tolower(ch);
        if(ch=='a'  || ch=='e' || ch=='i' || ch=='o' || ch=='u')return 1;
        return 0;
    }
    
    string f(string s,int cont){
        string x(cont,'a');

        string dev="";
        
        if( vow(s[0])){
            dev=s;
            dev+="ma";
        }else{
            dev=s.substr(1);
            dev+=s[0];
            dev+="ma";
        } 
        
        
        dev+=x;
        return dev;
    }
    
    string toGoatLatin(string S) {
        string dev="";
        istringstream is(S);
        string aux;
        bool ok=0;
        int cont=1;
        
        while(is>>aux){
            if(ok)dev+=" ";
            if(ok==0)ok=1;
            dev+=f(aux,cont);
            cont++;
        }
        
        return dev;
    }
};
