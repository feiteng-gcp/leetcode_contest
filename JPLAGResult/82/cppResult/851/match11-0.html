class Solution {
public:
    bool ck(char ch) {
        string v = "aeiouAEIOU";
        return (v.find(ch) != string::npos);
    }
    string toGoatLatin(string S) {
        istringstream is(S);
        vector<string> res;
        string s;
        string A = "";
        while (is >> s) {
            A += "a";
            string t = "";
            if (ck(s[0])) {
                t = s + "ma";
            } else {
                t = s.substr(1) + s[0] + "ma";
            }
            t += A;
            res.push_back(t);
        }
        string ans = res[0];
        for (int i = 1; i < (int)res.size(); i++) {
            ans += " " + res[i];
        }
        return ans;
    }
};
