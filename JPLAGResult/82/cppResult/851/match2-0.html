bool isvowel(char c){
	//a, e, i, o, or u
	if ( c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' ) return 1;
	if ( c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U' ) return 1;
	return 0;
}
class Solution {
	public:string toGoatLatin(string s) {
    	int n = s.size();
    	stringstream inn(s);
    	int idx = 0;
    	string w;
    	string res = "";
		string end = "a";
		while (inn>>w) {
			if ( isvowel(w[0]) ) {
				res = res + w + "ma" + end;
			}else {
				res = res + w.substr(1 , w.size() - 1) + w[0] + "ma" + end;
			}				
			res = res + " ";
			end = end + "a";
		}
		res.resize(res.size() - 1);
		return res;
   		
    }
};
