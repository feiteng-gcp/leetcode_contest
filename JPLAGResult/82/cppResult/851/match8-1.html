class Solution {
public:

    string helper(string t, int start,int end,int count){
        string r = "";
        if(t[start] == 'a' ||t[start] == 'e' ||t[start] == 'i' ||t[start] == 'o' ||t[start] == 'u' ||t[start] == 'A' ||t[start] == 'E' ||t[start] == 'I' ||t[start] == 'O' ||t[start] == 'U'){
            r += t.substr(start,end-start + 1);
            r += "ma";
        } else{
            r += t.substr(start + 1, end-start);
            r += t[start];
            r += "ma";
        }
        r += string(count,'a');
        return r;
    }
    string toGoatLatin(string S) {
        int start = 0;
        int end = 0;
        string r="";
        int count = 1;
        for(int i = 0; i < S.size(); i++){
            if(S[i] == ' '){
                end = i-1;
                r = r + helper(S,start,end, count) + " ";
                count ++;
                start = i + 1;
            } else if(i == S.size()-1){
                end = i;
                r = r + helper(S, start, end, count);
            }
        }
        
        return r;
    }
};
