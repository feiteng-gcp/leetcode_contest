class Solution {
public:
    string toGoatLatin(string S) {
        string vocals = "aeiouAEIOU";
        string as = "a", ans;
        bool first = false;
        for (int i = 0; i < S.size(); i++) {
            if (first) ans.push_back(' ');
            string tmp;
            while (i < S.size() && S[i] != ' ') {
                tmp.push_back(S[i++]);
            }

            if (vocals.find(tmp[0]) != string::npos) {
                tmp = tmp + "ma" + as;
            } else {
                tmp = tmp.substr(1, tmp.size()) + tmp[0] + "ma" + as;
            }
            ans = ans + tmp;
            as.push_back('a');
            first = true;
            
        } 
        return ans;
    }
};
