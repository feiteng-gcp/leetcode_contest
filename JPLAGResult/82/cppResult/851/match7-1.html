class Solution {
public:
    
    bool isv(char c){
        if( c=='a' || c=='e' || c=='i' || c=='o' || c=='u' ||
            c=='A' || c=='E' || c=='I' || c=='O' || c=='U') {
            return true;
        }
        return false;
    }
    
    string toGoatLatin(string S) {
        
        string r;
        string a("a");
        
        char *str = (char*) S.c_str();
        char *pch = strtok (str," ");
        while (pch != NULL)
        {
            string s(pch);

            if(isv(s[0])){
                s = s + "ma";
            } else {
                s = s.substr(1) + s[0] + "ma";
            }
            s += a;
            a += "a";
            if(a.size()==2)
                r = s;
            else
                r += " " + s;
            pch = strtok (NULL, " ");
        }
        
        return r;
    }
};
