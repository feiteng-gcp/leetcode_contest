class Solution {
public:
    string toGoatLatin(string S) {
        S = S+' ';
        int sz = S.size();
        string buffer = "";
        string res = "";
        int counter = 1;
        char first;
        for(int i = 0; i < sz; i++){
            if(buffer.size() == 0) { first = S[i];}
            if(S[i] == ' ' && buffer.size() > 0){
                if(first=='a' || first=='e' || first =='i' || first=='o' || first=='u' ||
                  first=='A' || first=='E' || first =='I' || first=='O' || first=='U')
                  {
                    buffer += "ma";
                }
                else{                    
                    string tmp = buffer.substr(1);                                        
                    buffer = tmp + first+"ma";                    
                }
                for(int j = 1; j <= counter; j++) buffer+="a";
                if(res.size() > 0)
                    res = res + " " +buffer;
                else res = buffer;
                buffer = "";
                counter++;                
            }            
            if(S[i] != ' ')
                buffer += S[i];
        } 
        return res;
    }
};
