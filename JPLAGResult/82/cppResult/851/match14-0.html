class Solution {
public:
    bool vowel(char x){
        return x=='a'||x=='e'||x=='i'||x=='o'||x=='u'||x=='A'||x=='E'||x=='I'||x=='O'||x=='U';
    }
    string gao(string x,int idx){
        char hd = x[0];
        string ret = "";
        if(vowel(hd)){
            ret = x + "ma";
        }
        else{
            ret = x.substr(1,x.size());
            ret.push_back(hd);
            ret = ret + "ma";
        }
        for(int i=0;i<idx;i++)
            ret.push_back('a');
        return ret;
    }
    string toGoatLatin(string S) {
        string delimiter = " ";
        size_t pos = 0;
        string token;
        string ans = "";
        int idx = 1;
        while ((pos = S.find(delimiter)) != string::npos) {
            token = S.substr(0, pos);
            token = gao(token,idx++);
            //std::cout << token << std::endl;
            ans = ans+token+" ";
            S.erase(0, pos + delimiter.length());
        }
        ans = ans + gao(S,idx);
        //std::cout << S << std::endl;
        return ans;
    }
};
