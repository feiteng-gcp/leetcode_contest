class Solution {
public:
    string toGoatLatin(string S) {
        char word[110];
        char str[10000] = {};
        int len = S.size();
        int k = 0;
        int cnt = 1;
        int ind = 0;
        for(int i = 0; i < len ; i++){
            if(S[i] != ' ' && i != len-1)
                continue;
            if(i == len-1)
                i++;
            int from = 0;
            if(S[k] != 'a' && S[k] != 'e' && S[k] != 'i' && S[k] != 'o' && S[k] != 'u'
              && S[k] != 'A' && S[k] != 'E' && S[k] != 'I' && S[k] != 'O' && S[k] != 'U')
                from++;
            for(int j = 0; j < i-k; j++)
                str[ind++] = S[k+((from+j)%(i-k))];
            str[ind++] = 'm';
            str[ind++] = 'a';
            for(int j = 0; j < cnt; j++)
                str[ind++] = 'a';
            if(i != len)
                str[ind++] = ' ';
            k = i+1;
            cnt++;
        }
        string ret = str;
        return ret;
    }
};
