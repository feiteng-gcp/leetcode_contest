class Solution {
public:
    string toGoatLatin(string S) {
        string substr;
        string ans="";
        int i,j;
        string suf="";
        i=0;
        while (i<S.size()){
            suf +='a'; 
            j=i;
            while (j<S.size() && S[j]!=' ') j++;
            //[i,j-1] 
            
            if (S[i]=='a' || S[i]=='A' || S[i]=='e' ||S[i]=='E'||S[i]=='o' || S[i]=='O' || S[i]=='i' || S[i]=='I' ||S[i]=='u' ||S[i]=='U'){
               substr = S.substr(i, j-i); 
            }else{
                substr = S.substr(i+1,j-i-1)+ S[i];
            }
            substr += "ma"+suf;
            i = j+1; 
            if (suf.size()==1)
                ans += substr;
            else
                ans = ans + " " +substr;
        }
        return ans;
    }
};
