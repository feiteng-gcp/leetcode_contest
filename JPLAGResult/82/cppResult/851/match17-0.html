class Solution {
public:
    string toGoatLatin(string S) 
    {
        string res;
        stringstream ss(S);
        string tmp;
        int index = 0;
        while(ss >> tmp)
        {
            index++;
            
            if(tolower(tmp[0]) =='a' ||
                tolower(tmp[0]) =='e' || 
                tolower(tmp[0]) =='i' ||
                tolower(tmp[0])=='o'||
                tolower(tmp[0])=='u')
            {
                tmp += "ma";
            }
            else
            {
                tmp.push_back(tmp[0]);
                tmp += "ma";
                tmp = std::string(tmp.begin()+1,tmp.end());
            }
            for(int i = 0;i<index;i++)
            {
                tmp += "a";
            }
            if(index == 1)
            {
                res += tmp;
            }
            else
            {
                res += " "+tmp;
            }
            tmp = "";
        }
        return res;
    }
};
