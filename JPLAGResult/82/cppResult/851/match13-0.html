class Solution {
public:
    string toGoatLatin(string s) 
    {
        s+=' ';
        string ret;
        string t;
        int cnt=0;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]==' ')
            {
                cnt++;
                if(t[0]=='i'||t[0]=='e'||t[0]=='o'||t[0]=='u'||t[0]=='a'||t[0]=='I'||t[0]=='E'||t[0]=='O'||t[0]=='U'||t[0]=='A')
                    {
                       if(cnt>1)
                        ret+=' '+t+"ma";
                      else
                         ret+=t+"ma"; 
                    }
                else
                {
                    if(cnt>1)
                    ret+=' '+t.substr(1,t.size()-1)+t[0]+"ma";
                    else
                       ret+=t.substr(1,t.size()-1)+t[0]+"ma";  
                }
                
                 for(int j=0;j<cnt;j++)
                            ret+='a';
                t="";
                while(i<s.size()&&s[i]==' ')
                    i++;
                i--;
            }
            else
            {
                t+=s[i];
            }
        }
        return ret;
    }
};
