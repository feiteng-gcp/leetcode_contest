class Solution {
public:
    string toGoatLatin(string S) {
        vector<string> vs;
        stringstream ss(S);
        string tmp;
        unordered_set<char> us = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        while (getline(ss, tmp, ' ')) {
            vs.push_back(tmp);
        }
        for (int i = 0; i < vs.size(); ++i) {

            if (us.count(vs[i][0])) {
                vs[i] = vs[i] + "ma";
            }
            else {
                vs[i] = vs[i].substr(1) + vs[i][0] + "ma";
            }
            vs[i] += string(i + 1, 'a');
        }
        string res = "";
        for (int i = 0; i < vs.size(); ++i) {
            if (i == vs.size() - 1) res += vs[i];
            else res += vs[i] + " ";
        }
        return res;
    }
};
