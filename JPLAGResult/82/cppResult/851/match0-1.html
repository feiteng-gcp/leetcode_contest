class Solution {
public:
    string toGoatLatin(string S) {
        istringstream iss(S);
        int i = 0;
        string result = "";
        string word;
        while (iss >> word){
            i++;
            if(word[0]=='a' ||word[0]=='e' ||word[0]=='i' ||word[0]=='o' ||word[0]=='u' ||
               word[0]=='A' ||word[0]=='E' ||word[0]=='I' ||word[0]=='O' ||word[0]=='U'){
               result += word + "ma"; 
            }else{
                

                result+= word.substr(1)  + word[0]+"ma";
            }
            for(int j = 0; j < i; j++){
                result += "a";
            }
            result+= " ";
        }
        result = result.substr(0,result.length()-1);
        return result;
    }
};
