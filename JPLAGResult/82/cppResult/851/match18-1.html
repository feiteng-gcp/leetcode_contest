class Solution {
public:
    bool vowel(char c) {
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' ||
            c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U';
    }
    vector<string> split(const string& s) {
        vector<string> ret;
        int n = s.size();
        for (int i = 0, j; i < n; i = j) {
            for (; i < n && s[i] == ' '; ++i);
            if (i == n) break;
            for (j = i + 1; j < n && s[j] != ' '; ++j);
            ret.push_back(s.substr(i, j - i));
        }
        return ret;
    }
    string toGoatLatin(string S) {
        vector<string> words = split(S);
        string ret;
        int k = 1;
        for (auto& s : words) {
            //cout << "s=" << s << endl;
            string cur;
            if (vowel(s[0])) {
                cur = s + "ma";
            } else {
                cur = s.substr(1) + s[0] + "ma";
            }

            for (int i = 0; i < k; ++i) cur += 'a';
            if (k == 1) {
                ret += cur;
            } else {
                ret += " " + cur;
            }
            ++k;
        }
        return ret;
    }
};
