class Solution {
public:
    void dfs(vector<vector<int>>& grid, int x, int y, int & space, int target){
        if(x<0 || x>= grid.size() || y<0 || y >= grid[x].size() || grid[x][y] !=  target) return;
        space += 1;
        //cout<<x<<" "<<y<<" "<<space<<endl;
        grid[x][y] = -target;
        //visited[x][y] = true;
        dfs(grid, x+1, y, space, target);
        dfs(grid, x-1, y, space, target);
        dfs(grid, x, y+1, space, target);
        dfs(grid, x, y-1, space, target);
    }
    
    void dfsdfs(vector<vector<int>>& grid, int x, int y, int & space, int target, unordered_set<int> & visited){
        if(x<0 || x>= grid.size() || y<0 || y >= grid[x].size() || grid[x][y] !=  target || visited.find(x*100+y) != visited.end()) return;
        space += 1;        
        visited.insert(x*100+y);
        dfsdfs(grid, x+1, y, space, target,  visited);
        dfsdfs(grid, x-1, y, space, target,  visited);
        dfsdfs(grid, x, y+1, space, target,  visited);
        dfsdfs(grid, x, y-1, space, target,  visited);
    }
    
    
    int largestIsland(vector<vector<int>>& grid) {  

        int space = 0;
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid[i].size(); j++){                
                if(grid[i][j] != 1) continue;

                int ts = 0;
                dfs(grid, i, j, ts, 1);    
                space = max(space, ts);
            }      
        }
        //cout<<space<<endl;
        
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid[i].size(); j++){                
                if(grid[i][j] != 0) continue;
                unordered_set<int> visited;
                grid[i][j] = -1;
                int ts = 0;
                dfsdfs(grid, i, j, ts, -1, visited);    
                space = max(space, ts);
                grid[i][j] = 0;
            }      
        }
        
        return space;
    }
};
