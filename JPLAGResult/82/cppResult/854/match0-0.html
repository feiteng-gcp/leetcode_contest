class Solution {
public:
    int c[3000];
    int f[3000];
    int n, m;
    int find(int x) {
        if (x == f[x]) return x;
        return f[x] = find(f[x]);
    }
    int dir[8] = {0, 1, 0, -1, 1, 0, -1, 0};
    void uni(int x, int y) {
        x = find(x);
        y = find(y);
        if (x != y) {
            f[x] = y;
            c[y] += c[x];
        }
    }
    int largestIsland(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int id = i * m + j;
                f[id] = id;
                c[id] = grid[i][j];
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0) continue;
                for (int k = 0; k < 8; ) {
                    int dx = i + dir[k++];
                    int dy = j + dir[k++];
                    if (dx < 0 || dx >= n || dy < 0 || dy >= m || grid[dx][dy] == 0) continue;
                    uni(i * m + j, dx * m + dy);
                }
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    res = max(res, c[find(i * m + j)]);
                } else {
                    int tmp = 1;
                    vector<int> nxt;
                    for (int k = 0; k < 8; ) {
                        int dx = i + dir[k++];
                        int dy = j + dir[k++];
                        if (dx < 0 || dx >= n || dy < 0 || dy >= m || grid[dx][dy] == 0) continue;
                        int id = find(dx * m + dy);
                        bool ok = true;
                        for (auto v : nxt) {
                            if (id == v) {
                                ok = false;
                                break;
                            }
                        }
                        if (ok) {
                            tmp += c[id];
                            nxt.push_back(id);
                        }
                    }
                    res = max(res, tmp);
                }
            }
        }
        return res;
    }
};
