#define FOR(i, a, b) for (remove_cv<remove_reference<decltype(b)>::type>::type i = (a); i < (b); i++)
#define REP(i, n) FOR(i, 0, n)

class Solution {
  vector<int> uf;
  int f(int x) {
    while (uf[x] > 0)
      uf[x] = uf[uf[x]] > 0 ? uf[uf[x]] : uf[x], x = uf[x];
    return x;
  }
  void merge(int x, int y) {
    x = f(x);
    y = f(y);
    if (x != y) {
      uf[x] += uf[y];
      uf[y] = x;
    }
  }
public:
  int largestIsland(vector<vector<int>>& g) {
    int m = g.size(), n = g[0].size();
    uf.assign(m*n+1, 0);
    REP(i, m)
      REP(j, n)
        if (g[i][j] == 1) {
          uf[i*n+j+1] = -1;
          if (i && g[i-1][j]) merge((i-1)*n+j+1, i*n+j+1);
          if (j && g[i][j-1]) merge(i*n+j, i*n+j+1);
        }
    int r = 0;
    REP(i, m)
      REP(j, n) {
        int t = uf[i*n+j+1];
        if (t < 0) r = max(r, -t);
        if (g[i][j] == 0) {
          int cs[4] = {};
          if (i) cs[0] = f((i-1)*n+j+1);
          if (i+1 < m) cs[1] = f((i+1)*n+j+1);
          if (j) cs[2] = f(i*n+j);
          if (j+1 < n) cs[3] = f(i*n+j+2);
          sort(cs, cs+4);
          int k = unique(cs, cs+4) - cs;
          t = 1;
          while (k--) t -= uf[cs[k]];
          r = max(r, t);
        }
      }
    return r;
  }
};
