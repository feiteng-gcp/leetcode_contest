class Solution:
    def toGoatLatin(self, S):
        """
        :type S: str
        :rtype: str
        """
        words = S.split(' ')
        res = []
        for i, word in enumerate(words) :
            if word[0] not in ['a','e','i','o','u','A','E','I','O','U'] :
                word = word[1:] + word[0]
            word = word + 'ma' + 'a' * (i+1)
            res.append(word)
        return ' '.join(res)
