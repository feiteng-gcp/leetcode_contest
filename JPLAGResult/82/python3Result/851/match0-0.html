class Solution:
    def toGoatLatin(self, S):
        sol = []
        wovels = 'aeiouAEIOU'
        for i, word in enumerate(S.split()):
            wt = word[:]
            if wt[0] in wovels:
                wt = wt + 'ma' + 'a'*(i+1)
            else:
                wt = wt[1:] + wt[0] + 'ma' + 'a'*(i+1)
            sol.append(wt)
        return ' '.join(sol)
