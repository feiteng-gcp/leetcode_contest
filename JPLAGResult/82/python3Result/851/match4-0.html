class Solution:
    def toGoatLatin(self, S):
        words = S.split()
        ans = list()

        for i, word in enumerate(words):
            ch = word[0].lower()
            if ch not in ['a', 'e', 'i', 'o', 'u']:
                word = word[1:] + word[0]
            word += 'm' + 'a' * (i + 2)
            ans.append(word)

        return ' '.join(ans)
