class Solution:
    def toGoatLatin(self, S):
        """
        :type S: str
        :rtype: str
        """
        cs = {'a','e','i','o','u','A','E','I','O','U'}
        l = S.split()
        res = []
        for i in range(len
(l)):
            e = l[i]
            if e[0] in cs:
                t = e
            else:
                t = e[1:]+e[0]
            res.append(t+'ma'+'a'*(i+1))
        return ' '.join(res)
