class Solution {
    public String toGoatLatin(String S) {
        Set<Character> vowel = new HashSet<>();
        vowel.add('a');
        vowel.add('e');
        vowel.add('i');
        vowel.add('o');
        vowel.add('u');
        
        String[] strs = S.split(" ");
        StringBuilder res = new StringBuilder();
        for (int i = 0; i < strs.length; i++) {
            char start = strs[i].charAt(0);
            if (vowel.contains(Character.toLowerCase(start))) res.append(strs[i]);
            else {
                res.append(strs[i].substring(1));
                res.append(start);
            }
            res.append("ma");
            int count = i + 1;
            while (count > 0) {
                res.append("a");
                count--;
            }
            res.append(" ");
        }
        res.deleteCharAt(res.length() - 1);
        return res.toString();
    }
}
