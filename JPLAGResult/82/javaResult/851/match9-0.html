class Solution {
    public String toGoatLatin(String S) {
        String[] str = S.split(" ");
        //case1: start with vovel
        //case2: start with non vovel
        
        //append (i + 1) * a
        Set<Character> vovels = new HashSet<>();
        vovels.add('a');
        vovels.add('e');
        vovels.add('i');
        vovels.add('o');
        vovels.add('u');
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < str.length; i++) {
            if (vovels.contains(str[i].toLowerCase().charAt(0))) {
                sb.append(str[i]).append("ma");
            } else {
                sb.append(str[i].substring(1)).append(str[i].charAt(0)).append("ma");
            }
            for (int j = 0; j < i + 1; j++) {
                sb.append("a");
            }
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
