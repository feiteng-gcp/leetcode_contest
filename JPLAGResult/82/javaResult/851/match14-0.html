class Solution {
    public String toGoatLatin(String S) {
        StringBuilder sb = new StringBuilder();
        Set<Character> vowels = new HashSet<>();
        vowels.add('a');
        vowels.add('e');
        vowels.add('i');
        vowels.add('o');
        vowels.add('u');
        vowels.add('A');
        vowels.add('E');
        vowels.add('I');
        vowels.add('O');
        vowels.add('U');
        int i = 0;
        for(String str : S.split(" ")){
            StringBuilder tmp = new StringBuilder();
            if(vowels.contains(str.charAt(0))){
                tmp.append(str).append("ma");
            }else{
                tmp.append(str.substring(1)).append(str.charAt(0)).append("ma");
            }
            for(int j = 0; j <= i; j++){
                    tmp.append("a");
            }
            i++;
            sb.append(tmp.toString()).append(" ");
        }
        sb.setLength(sb.length() - 1);
        return sb.toString();
    }
}
