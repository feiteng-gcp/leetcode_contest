class Solution {
    public String toGoatLatin(String S) {
    	if(S==null) {
    		return "";
    	}
    	HashSet<String> set =new HashSet<String>();
    	set.add("a");
    	set.add("A");
    	set.add("e");
    	set.add("E");
    	set.add("i");
    	set.add("I");
    	set.add("o");
    	set.add("O");
    	set.add("u");
    	set.add("U");
    	
        String[] strArr =S.split(" ");
        StringBuilder sb =new StringBuilder();
        int aNum=1;
        for(int i =0 ; i<strArr.length ;i++) {
        	String tmp=strArr[i];
        	if( set.contains(    tmp.substring(0, 1) ) ) {
        		sb.append(tmp);
        		sb.append("ma");
        	}else {
        		sb.append(tmp.substring(1,tmp.length()));
        		sb.append(tmp.substring(0,1));
        		sb.append("ma");
        	}
        	for(int j=0;j<aNum;j++) {
        		sb.append("a");
        	}
        	aNum++;
        	sb.append(" ");
        }
        return sb.substring(0, sb.length()-1).toString();
    }
}
