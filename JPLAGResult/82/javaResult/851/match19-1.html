class Solution {
    public String toGoatLatin(String S) {
        StringBuilder stringBuilder = new StringBuilder();

        String[] ws = S.split(" ");
        
HashSet<Character> vs = new HashSet<>();
        vs.add('a');
        vs.add('e');
        vs.add('i');
        vs.add('o');
        vs.add('u');
        vs.add('A');
        vs.add('E');
        vs.add('I');
        vs.add('O');
        vs.add('U');
        
for(int i = 0;i<ws.length;i++){
            StringBuilder w = new StringBuilder(ws[i]);
            if(!vs.contains(w.charAt(0))){
                char c = w.charAt(0);
                w.deleteCharAt(0);
                w.append(c);
            }
            w.append("ma");
            for(int j = 0;j<=i;j++){
                w.append('a');
            }
            if(stringBuilder.length() > 0) stringBuilder.append(' ');
            stringBuilder.append(w);

        }
        return stringBuilder.toString();
    }
}
