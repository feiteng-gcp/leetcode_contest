class Solution {
    public String toGoatLatin(String S) {
        StringBuilder result = new StringBuilder();
        String ma = "ma";
        int index = 1;
        for(String each: S.split(" ")) {
            char first = each.charAt(0);
            if("AEIOUaeiou".indexOf(first) != -1){
                result.append(each).append(ma);
            }
            else {
                result.append(each.substring(1)).append(first).append(ma);
            }
            for(int i = 1; i <= index; i++) result.append("a");
            result.append(" ");
            index++;
        }
        return result.toString().trim();
    }
}
