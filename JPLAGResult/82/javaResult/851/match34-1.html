class Solution {
    public String toGoatLatin(String S) {
        Set<Character> vowel = new HashSet<>(Arrays.asList('a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'));
        String[] words = S.split("\\s+");
        for(int i = 0; i < words.length; i++) {
            String w = words[i];
            if(!vowel.contains(w.charAt(0))) 
{
                if(w.length() > 1) {
                    w = w.substring(1) + w.charAt(0);
                }
            }
            w += "ma";
            for(int j = 0; j <= i; j++) {
                w += "a";
            }
            words[i] = w;
        }
        String res = "";
        for(int i = 0; i < words.length; i++) {
            String w = words[i];
            if(i == words.length - 1) res += w;
            else res = res + w + " ";
        }
        return res;
    }
}
