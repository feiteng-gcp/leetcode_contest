class Solution {
    public String toGoatLatin(String S) {
        String[] parts = 
S.split("\\s");
        for(int i = 0; i < parts.length; i++) {
            StringBuilder sb = new StringBuilder(parts[i]);
            if(sb.charAt(0) == 'a' || sb.charAt(0) == 'e' || sb.charAt(0) == 'i' || sb.charAt(0) == 'o' || sb.charAt(0) == 'u' || sb.charAt(0) == 'A' || sb.charAt(0) == 'E' || sb.charAt(0) == 'I' || sb.charAt(0) == 'O' || sb.charAt(0) == 'U') {
                sb.append("ma");
            }else {
                char c = sb.charAt(0);
                sb.deleteCharAt(0);
                sb.append(c);
                sb.append("ma");
            }
            int cnt = i + 1;
            while(cnt > 0) {
                sb.append('a');
                cnt--;
            }
            
parts[i] = sb.toString();
        }
        StringBuilder res = new StringBuilder(parts[0]);
        for(int i = 1; i < parts.length; i++) {
            res.append(" ");
            res.append(parts[i]);
        }
        return res.toString();
    }
}
