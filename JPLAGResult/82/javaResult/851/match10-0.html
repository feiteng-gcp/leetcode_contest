class Solution {
    public String toGoatLatin(String S) {
        String[] token = S.split(" ");
        
        String vowel = "AaEeIiOoUu";

        
        StringBuilder sb = new StringBuilder();
        
        for (int i=0; i<token.length; i++) {
        	if (vowel.contains(token[i].charAt(0) + "")) {
        		sb.append(token[i]);
        	}
        	else {
        		sb.append(token[i].substring(1, token[i].length())).append(token[i].charAt(0));
        	}
        	sb.append("ma");
        	
        	for (int j=0; j<i+1; j++) {
        		sb.append("a");
        	}
        	
        	sb.append(' ');
        		
        }
        
        sb.delete(sb.length()-1, sb.length());
        
        return sb.toString();
    }

}
