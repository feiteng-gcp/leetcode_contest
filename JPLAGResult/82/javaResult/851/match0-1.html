class Solution {
    public String toGoatLatin(String S) {
            StringBuilder sb = new StringBuilder();
            String[] sts = S.split(" ");
            for (int i = 0; i < sts.length; i++) {
                String ss = sts[i];
                char c = ss.charAt(0);
                if (isv(c) || 
isv((char)(c + 'A' - 'a')) || isv((char)(c + 'a' - 'A'))) {
                    sb.append(ss);

                } else {
                    sb.append(ss.substring(1));
                    sb.append(c);
                }
                sb.append("ma");
                for (int j = 0; j < i + 1; j++) {
                    sb.append("a");
                }
                if (i < sts.length - 1) {
                    sb.append(" ");
                }
            
}
            return sb.toString();
        }
        
        public boolean isv(char c){
            return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
        }
}
