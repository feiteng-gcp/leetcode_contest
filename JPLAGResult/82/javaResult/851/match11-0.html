	class Solution {
	    public String toGoatLatin(String S) {
	        String[] s = S.split(" ");
	        StringBuilder sb = new StringBuilder();
	        for(int i = 0;i < s.length;i++){
	        	if("AEIOUaeiou".indexOf(s[i].charAt(0)) >= 0){
	        		sb.append(" "  + s[i] + "ma");
	        	}else{
	        		sb.append(" " + s[i].substring(1) + s[i].charAt(0) + "ma");
	        	}
	        	for(int j = 0;j < i+1;j++){
	        		sb.append("a");
	        	}
	        }
	        return sb.substring(1);
	    }
	}
