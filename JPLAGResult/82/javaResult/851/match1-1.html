class Solution {
    public String toGoatLatin(String S) {
        String vowel = "aeiouAEIOU";
        StringBuilder res = new StringBuilder();
        
        String[] arr = S.split(" ");
        String suffix = "a"; 
        
        for(int i = 0; i< arr.length; i++){
            String start = arr[i].substring(0,1);
            
            if(
vowel.indexOf(start)!=-1){
                res.append(arr[i]);
            }
            else{
                res.append(arr[i].substring(1));
                res.append(start);
            }
            
            res.append("ma");
            res.append(suffix);
            
if(i< arr.length-1){
                res.append(" ");
            }
            suffix += "a";
        }
        
        return res.toString();
        
    }
}
