import java.util.Arrays;
import java.util.Comparator;
import java.util.HashMap;
import java.util.Map;

class Solution {
    private int[][] dp;
    public int numFriendRequests(int[] ages) {
        Arrays.sort(ages);
        Map<Integer, Integer> map = new HashMap<>();
        for (int k : ages) map.put(k, map.getOrDefault(k, 0) + 1);
        Node[] nodes = new Node[map.size()];
        int idx = 0;
        int sum = 0;
        for (Map.Entry<Integer, Integer> e : map.entrySet()) {
            nodes[idx++] = new Node(e.getKey(), e.getValue());
            if (e.getKey() > 14) {
                sum += e.getValue() * (e.getValue() - 1);
            }
        }
        Arrays.sort(nodes, new Comparator<Node>() {
            @Override
            public int compare(Node o1, Node o2) {
                return o1.val - o2.val;
            }
        });
        for (int i = 0; i < nodes.length; i++) {
            if (nodes[i].val <= 14) continue;
            int index = search(nodes, 0, i, Math.floor(0.5*nodes[i].val + 8));
            if (index < i) {
                int temp = 0;
                for (int j = index; j < i; j++) {
                    temp += nodes[j].count;
                }
                sum += temp * nodes[i].count;
            
}
        }
        return sum;
    }

    private int search(Node[] nodes, int start, int end, double target) {
        
int l = start, r = end - 1;
        while (l <= r) {
            int mid = (r - l)/2 + l;
            if (1.0*nodes[mid].val == target) {
                r = mid - 1;
            } else if (nodes[mid].val < target) {
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return l;
    }

    private static class Node {
        int val;
        int count;
        public Node(int a, int b) {
            this.val = a;
            this.count = b;
        }
    }
}
