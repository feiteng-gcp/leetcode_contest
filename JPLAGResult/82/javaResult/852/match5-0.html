class Solution {
    public int numFriendRequests(int[] ages) {
        Arrays.sort(ages);
        int ret = 0;
        int n = ages.length;
        
        //count (A,B) so that ages[A]*0.5 + 7 < ages[B] <= ages[A]
        for(int i=0;i<n;i++) {
            //i is A, count how many B's are there
            int leftbar = (int)(ages[i]*0.5) + 8, rightbar = ages[i] + 1;
            int count;
            if (leftbar>=rightbar) {
                count = 0;
            } else {
                int left = search(ages, leftbar), right = search(ages, rightbar);
                count = right - left - 1; // A iteslf
            }
            // System.out.println(i + " " + ages[i] + " " + count);
            ret += count;
        }
        return ret;
    }
    
    private int search(int[] nums, int v) {
        //return last i so that nums[i] < v; nums[-1] = -infinity
        int n = nums.length;
        int left = 0, right = n - 1;
        while(right - left > 1) {
            int middle = (right + left) / 2;
            if (nums[middle] >= v) {
                right = middle;
            } else {
                left = middle;
            }
        }
        if (nums[right] < v) return right;
        else if (nums[left] < v) return left;
        else return -1;
    }
}
