class Solution {
    public int numFriendRequests(int[] ages) {
        
        int result = 0;
        int[] frequency = new int[121];
        for(int i: ages)
        {
            frequency[i]++;
        }
        
        ArrayList<Integer>[] requests = new ArrayList[121];
        for(int i=1; i<=120; i++)
        {
            requests[i] = new ArrayList<Integer>();
            for(int j=1; j<=120; j++)
            {
                if(!IsNotMatch(i, j))
                {
                    requests[i].add(j);
                }
            }
        }
        
        for(int i: ages)
        {
            for(int match: requests[i])
            {
                result += frequency[match];
                if(match == i)
                {
                    result = result - 1;
                
}
            }
        }
        
        return result;
    }
    
    public Boolean IsNotMatch(int A, int B)
    {
        return (B <= 0.5 * A + 7) || (B > A) || (B > 100 && A < 100);
    }
}
