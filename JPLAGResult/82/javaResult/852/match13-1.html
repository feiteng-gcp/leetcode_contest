class Solution {
      public int numFriendRequests(int[] ages) {
       int []count = new int[121];
       for (int age: ages) {
           count[age]++;
       }
       int ans = 0;
       for (int i = 0; i < ages.length; i++) {
           for (int j = 0; j < 121; j++) {
               if (isNotValid(ages[i], j)) {
                   continue;
               }
               ans += count[j];
               if (j == ages[i]) {
                   ans -=1;
               
}
           }
       }
       return ans;
    }
    
    public boolean isNotValid(double  a, double b) {
        
        return b <= 0.5 * a + 7 || b > a || (b > 100 && a < 100);
    }
}
