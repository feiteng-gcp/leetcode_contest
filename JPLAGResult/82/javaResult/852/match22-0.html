class Solution {
    public int numFriendRequests(int[] ages) {
        // Arrays.sort(ages);
        int cnt = 0;
        Map<Integer, Integer> cntMap = new HashMap<>();
        for (Integer age : ages) {
            int tmp = cntMap.getOrDefault(age, 0);
            tmp++;
            cntMap.put(age, tmp);
        }
        for (Integer age1 : cntMap.keySet()) {
            for (Integer age2 : cntMap.keySet()) {
                if (age1 != age2 && isFriend(age1, age2)) {
                    cnt += cntMap.get(age1) * cntMap.get(age2);
                } else if (age1 == age2 && isFriend(age1, age2)) {
                    cnt += cntMap.get(age1) * (cntMap.get(age2) - 1);
                }
            }
        }
        
        return cnt;
        // int cnt = 0;
        // for (int i = 0; i < ages.length; i++) {
        //     for (int j = i + 1; j < ages.length; j++) {
        //         if (isFriend(ages[i], ages[j])) {
        //             cnt++;
        //         }
        //         if (isFriend(ages[j], ages[i])) {
        //             cnt++;
        //         }
        //     }
        // }
        // return cnt;
        
    }
    
    public boolean isFriend(int a, int b) {
        if (b <= 0.5 * a + 7 || b > a || (b > 100 && a < 100)) {
            return false;
        }
        return true;
        
    }
}
