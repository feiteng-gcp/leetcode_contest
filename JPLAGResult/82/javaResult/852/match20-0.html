class Solution {
    public int numFriendRequests(int[] ages) {
        Arrays.sort(ages);
        int le = ages.length;
        int c = 0;
        for(int i=0; i<le; i++){
            int l = 0, h = i-1;
            while(l<=h){
                int m = (l+h)/2;
                if(ages[m]<=(ages[i]*0.5 + 7)){
                    l = m+1;
                }
                else{
                    h = m-1;
                }
            }
            c += i-l;
           //System.out.print(i+" "+ages[i]+" "+l);
            l = i+1; h = le-1;
            while(l<=h){
                int m = (l+h)/2;
                if(ages[m]==ages[i]){
                    l = m+1;
                }
                else{
                    h = m-1;
                }
            }
            //if(l<le && ages[l]==ages[i])
            if(ages[i]>(ages[i]*0.5 + 7))
                c+= l-i-1;
            //System.out.println(" "+l);
        }
        return c;
    }
}
