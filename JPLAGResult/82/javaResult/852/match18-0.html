class Solution {
   public int numFriendRequests(int[] ages) {    	
    	Arrays.sort(ages);
    	
    	int accu = 0;
    	
    	for (int i=ages.length-1; i>0; i--) {
    		for (int j = i -1; j >=0; j--) {
    			if (ages[i] >= (2*ages[j] - 14)) {
    				break;
    			}
    			else {
    				accu++;
    				if (ages[i] == ages[j]) {
    					accu++;
    				}
    			}
    		}
    	}

    	return accu;
    	
        
    }
}
