class Solution {
    public int numFriendRequests(int[] ages) {
        int count = 0;
        Arrays.sort(ages);

        outer:
        for (int i = ages.length - 1; i > 0; i--) {
            int limit = (int) (ages[i] * 0.5 + 7);
            for (int j = i - 1; j >= 0; j--) {
                if (ages[j] <= limit) {
                    continue outer;
                }
                if (ages[i] == ages[j]) {
                    count += 2;
                } else {
                    count++;
                
}
            }
        }

        return count;
    }
}
