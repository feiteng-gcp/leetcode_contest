class Solution {
    class Node {
        int profit;
        int difficulty;
        Node (int profit, int difficulty) {
            this.profit = profit;
            this.difficulty = difficulty;
        }
        // @Override
        // public int compareTo(Node n) {
        //     return profit - n.profit;
        // }
    }
    public int maxProfitAssignment(int[] difficulty, int[] profit, int[] worker) {
        //find the biggest difficulity smaller or equal to worker 
        Node[] ns = new Node[difficulty.length];
        for (int i = 0; i < difficulty.length; i++) {
            Node node = new Node(profit[i], difficulty[i]);
            ns[i] = node;
        }
        // Arrays.sort(ns);
        Arrays.sort(ns, new Comparator<Node>() {
            public int compare(Node n1, Node n2){
                return n2.profit - n1.profit;
            }
        });
        int res = 0;
        for (Integer work : worker) {
            int index = 0;
            while (index < ns.length) {
                if (ns[index].difficulty <= work) {
                    res += ns[index].profit;
                    break;
                }
                index++;
            }
            // int start = 0;
            // int end = difficulty.length - 1;
            // while (start + 1 < end) {
            //     int mid = start + (end - start) / 2;
            //     if (ns[mid].difficulty == work) {
            //         res += ns[mid].profit;
            //     }
            //     if (ns[mid].difficulty > work ) {
            //         end = mid;
            //     } else {
            //         start = mid;
            //     }
            // }
            // if (ns[end].difficulty < work) {
            //     res += ns[end].profit;
            // } else if (ns[start].difficulty < work) {
            //     res += ns[start].profit;
            // }
        }
        return res;
    }
}
