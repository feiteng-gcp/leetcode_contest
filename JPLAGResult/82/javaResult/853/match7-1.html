class Solution {
 public int maxProfitAssignment(int[] difficulty, int[] profit, int[] worker) {
        int maxProfit = 0;

        TreeMap<Integer, Integer> profits = getMaxProfits(difficulty, profit);

        for (int w : worker) {
            Integer key = profits.floorKey(w);
            if (key != null) 
{
                maxProfit += profits.get(key);
            }
        }

        return maxProfit;
    }

    private TreeMap<Integer, Integer> getMaxProfits(int[] difficulty, int[] profit) {
        // get profits
        TreeMap<Integer, Integer> profits = new TreeMap<>();
        for (int i = 0; i < difficulty.length; i++) {
            final int currentProfit = profit[i];
            profits.compute(difficulty[i], (k,v) -> v == null ? currentProfit : Math.max(v, currentProfit));
        }
        // get max profits
        TreeMap<Integer, Integer> maxProfits = new TreeMap<>();
        Integer maxProfit = Integer.MIN_VALUE;
        for (Map.Entry<Integer, Integer> entry : profits.entrySet()) {
            maxProfit = Math.max(maxProfit, entry.getValue());
            maxProfits.put(entry.getKey(), maxProfit);
        }

        return maxProfits;
    }
}
