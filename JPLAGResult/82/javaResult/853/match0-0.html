class Solution {
    class work implements Comparable<work>{
        int dif,prof;
        work(int d, int p){
            dif = d;
            prof = p;
        }
        public int compareTo(work a){
            return dif -a.dif;
        }
    }
    public int maxProfitAssignment(int[] difficulty, int[] profit, int[] worker) {
        int l = profit.length;
        work arr[] = new work[l];
        for(int i=0; i<l; i++){
            arr[i] = new work(difficulty[i], profit[i]);
        }
        Arrays.sort(arr);
        int dp[] = new int[l];
        dp[0] = arr[0].prof;
        for(int i=1; i<l; i++){
            dp[i] = Math.max(dp[i-1],arr[i].prof);
        }
        Arrays.sort(worker);
        int ans=0;
        int s = worker.length;
        for(int i=0; i<s; i++){
            int lo = 0, hi = l-1;
            while(lo<=hi){
                int m = lo +(hi-lo)/2;
                if(arr[m].dif>worker[i]){
                    hi = m-1;
                }
                else{
                    lo = m+1;
                }
            }
            if(lo>0){
                ans+=dp[lo-1];
                //System.out.println(i+" "+worker[i]+" "+arr[lo-1].prof);
            }
        }
        return ans;
    }
}
