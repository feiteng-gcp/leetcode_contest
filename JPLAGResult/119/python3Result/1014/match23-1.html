class Solution:
    def kClosest(self, points, K):
        """
        :type points: List[List[int]]
        :type K: int
        :rtype: List[List[int]]
        """
        array = [(points[i][0]**2 + points[i][1]**2, i) for i in range(len(points))]
        array
.sort(key = lambda x:x[0])
        ans = []
        for i in range(K):
            ans.append(points[array[i][1]])
        return ans
