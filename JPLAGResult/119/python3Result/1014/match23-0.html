class Solution:
    def kClosest(self, points, K):
        """
        :type points: List[List[int]]
        :type K: int
        :rtype: List[List[int]]
        """
        result = [(0, 0)] * len(points)
        for i,p in enumerate(points):
            result[i] = (p[0]*p[0] + p[1]*p[1], i)
        result.sort(key = lambda x:x[0])
        ret = []
        for i in range(K):
            ret.append(points[result[i][1]])
        return ret
