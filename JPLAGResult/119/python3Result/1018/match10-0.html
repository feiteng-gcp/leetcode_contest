class Solution:
    def largestPerimeter(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        A.sort()
        for i in range(len(A) - 1 , 1, -1):
            x, y, z = A[i], A[i - 1], A[i - 2]
            if (x + y > z) and (y + z > x) and (z + x > y):
                return x + y + z

        return 0
