class Solution:
    def largestPerimeter(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        # 排序 + 滑动窗口
        A.sort(reverse = True)
        
for i in range(len(A)-2):
            a,b,c = A[i], A[i+1], A[i+2]
            if b + c > a and b - c < a:
                return a+b+c
        return 0
