class Solution:
    def largestPerimeter(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        maxi = 0
        n = len(A) 
        A.sort(reverse = True) 
  
        
for i in range(0, n - 2): 
            if A[i] < (A[i + 1] + A[i + 2]): 
                maxi = max(maxi, A[i] + A[i + 1] + A[i + 2]) 
                break

        if(maxi == 0): 
            return 0
        else: 
            return maxi
