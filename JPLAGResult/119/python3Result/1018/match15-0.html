class Solution:
    def largestPerimeter(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        largestP = 0
        A.sort(reverse = True)
        N = len(A)
        A = tuple(A)
        for i in range(N - 2):
            if A[i] < A[i + 1] + A[i + 2]:
                largestP = max(largestP, A[i] + A[i + 1] + A[i + 2])
                break
        return largestP
