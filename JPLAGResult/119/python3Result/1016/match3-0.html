class Solution:
    def subarraysDivByK(self, A, K):
        """
        :type A: List[int]
        :type K: int
        :rtype: int
        """
        mod = []
        for i in range(K + 1):
            mod.append(0)
        
        cumSum = 0
        N = len(A)
        for i in range(N):
            cumSum += A[i]
            mod[((cumSum % K) + K) % K] = mod[((cumSum % K) + K) % K] + 1
            
        result = 0
        for i in range(K):
            if (mod[i] > 1):
                result += (mod[i] * (mod[i] - 1)) // 2
                
        result += mod[0]
        return result
