class Solution:
    def subarraysDivByK(self, A, K):
        """
        :type A: List[int]
        :type K: int
        :rtype: int
        """
        
        n = len(A)
        mod = []
        for i in range(K+1):
            mod.append(0)
            
        
        cumSum = 0
        for i in range(n):
            cumSum += A[i]
            mod[(cumSum%K+K)%K] +
= 1
        ans = 0
        for i in range(K):
            if mod[i]>1:
                ans+=(mod[i]*(mod[i]-1))//2
        ans += mod[0]
        return ans
