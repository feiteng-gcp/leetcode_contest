class Solution:
    def subarraysDivByK(self, A, K):
        """
        :type A: List[int]
        :type K: int
        :rtype: int
        """
        SumA=[0]
        S=0
        for i in A:
            S+=i
            S%=K
            SumA.append(S)
        SumSet=collections.Counter(SumA)
        count=0
        for i,val in SumSet.items():
            count+=(val*(val-1))//2
        return count
