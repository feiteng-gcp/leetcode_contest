class Solution:
    def subarraysDivByK(self, A, K):
        """
        :type A: List[int]
        :type K: int
        :rtype: int
        """
        hashmap = {0:1}
        cur = 0
        ans = 0
        for i in A:
            cur += i
            cur %= K
            if cur in hashmap:
                ans += hashmap[cur]
                hashmap[cur] += 1
            else:
                hashmap[cur] = 1
        return ans
