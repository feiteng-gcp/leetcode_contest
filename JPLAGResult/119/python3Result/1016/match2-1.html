class Solution:
    def subarraysDivByK(self, A, K):
        """
        :type A: List[int]
        :type K: int
        :rtype: int
        """
        
        d = {}
        d[0] = 1
        sum = 0
        ans = 0
        for a in A:
            sum 
= (sum + a) % K
            if sum in d:
                ans += d[sum]
                d[sum] += 1
            else:
                d[sum] = 1

        return ans
