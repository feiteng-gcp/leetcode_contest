class Solution:
    def subarraysDivByK(self, A, K):
        """
        :type A: List[int]
        :type K: int
        :rtype: int
        """
        su = 0
        a = []
        for i in A:
            su+=i
            su%=K
            a.append(su)
        
        d={0:[-1]}
        for i,v in enumerate(a):
            if v not in d: d[v]=[]
            d[v].append(i)
            
        res=0
        for i in d:
            l = len(d[i])
            res+=l*(l-1)//2
        return res
