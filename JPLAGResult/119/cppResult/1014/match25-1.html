class Solution {
public:
    vector<vector<int>> kClosest(vector<vector<int>>& p, int k) {
        sort(p.begin(), p.end(), [&](vector<int>& l, vector<int>& r) {
            return l[0] * l[0] + l[1] * l[1] < r[0] * r[0] + r[1] * r[1];
        });
        vector<vector<int>> res;
        for(int i = 0; i < k; ++i) res.emplace_back(p[i]);
        return res;
    }
};
