#include <algorithm>

using namespace std;

inline bool cmp (vector <int> left, vector <int> right)
{
    int a1 = left[0] * left[0] + left[1] * left[1];
    int a2 = right[0] * right[0] + right[1] * right[1];

    return a1 < a2;
}

class Solution {
public:
    vector<vector<int>> kClosest(vector<vector<int>>& points, int K) {
        sort (points.begin(), points.end(), cmp);
        vector <vector <int> > res;
        for (int i = 0; i < K; i++)
            res.push_back(points[i]);
        return res;
    }
};
