struct Node {
  int x, y;
  int dis;
};

bool cmp(const Node &a, const Node &b) {
    return a.dis < b.dis;
}

class Solution {
public:
    
    int getDis(int x, int y) {
        return x*x + y*y;
    }
    
    vector<vector<int>> kClosest(vector<vector<int>>& points, int K) {
        vector<Node> v;
        for (int i = 0; i < points.size(); i++) {
            int dis = getDis(points[i][0], points[i][1]);
            Node node;
            node.x = points[i][0];
            node.y = points[i][1];
            node.dis = dis;
            v.push_back(node);
        }
        sort(v.begin(), v.end(), cmp);
        vector< vector<int> > res;
        for (int i = 0; i < K; i++) {
            vector<int> vv;
            vv.push_back(v[i].x);
            vv.push_back(v[i].y);
            res.push_back(vv);
        }
        return res;
    }
};
