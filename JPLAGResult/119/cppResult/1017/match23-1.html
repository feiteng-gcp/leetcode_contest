class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        // (i, 0): 2 * i
        // (i, 1): 2 * i + 1
        map<int, int> nums;
        int N = A.size();
        nums[A[N - 1]] = N - 1;
        unordered_map<int, vector<int> > graph;
        for (int i = N - 2; i >= 0; --i) {
            // find the (i, 0) -> (j, 1) link
            auto it = nums.lower_bound(A[i]);
            if (it != nums.end()) {
                // found it
                int j = it->second;
                graph[j * 2 + 1].push_back(i * 2);
            }
            // find the (i, 1) -> (j, 0) link
            it = nums.upper_bound(A[i]);
            if (it != nums.begin()) {
                // found it
                --it;
                int j = it->second;
                graph[j * 2].push_back(i * 2 + 1);
            }
            nums[A[i]] = i;
        }
        // BFS from the starting node
        unordered_set<int> found;
        vector<int> level;
        if (graph.find(N * 2 - 1) != graph.end()) {
            level.push_back(N * 2 - 1);
            found.insert(N * 2 - 1);
        }
        if (graph.find(N * 2 - 2) != graph.end()) {
            level.push_back(N * 2 - 2);
            found.insert(N * 2 - 2);
        }
        while (!level.empty()) {
            vector<int> new_level;
            for (int node : level) {
                if (graph.find(node) != graph.end()) {
                    for (int nbr : graph[node]) {
                        if (found.find(nbr) == found.end()) {
                            new_level.push_back(nbr);
                            found.insert(nbr);
                        }
                    }
                }
            }
            level = new_level;

        }
        int rtn = 1;
        for (int i = 0; i < N - 1; ++i) {
            if (found.find(i * 2) != found.end()) {
                ++rtn;
            }
        }
        return rtn;
    }
};
