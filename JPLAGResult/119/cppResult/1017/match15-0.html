class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int n = A.size();
        vector<int> odd_jump(n, -1), even_jump(n, -1);
        vector<vector<int>> odd_left(n, vector<int>()), even_left(n, vector<int>());
        set<pair<int, int>> odd_set, even_set;

        for (int i = n - 1; i >= 0; i--) {
            auto it = odd_set.lower_bound(make_pair(A[i], i));

            if (it != odd_set.end())
                odd_jump[i] = it->second;

            auto even_it = even_set.lower_bound(make_pair(-A[i], i));

            if (even_it != even_set.end())
                even_jump[i] = even_it->second;

            // cerr << i << ": " << odd_jump[i] << ' ' << even_jump[i] << endl;

            if (odd_jump[i] != -1)
                odd_left[odd_jump[i]].push_back(i);

            if (even_jump[i] != -1)
                even_left[even_jump[i]].push_back(i);

            odd_set.insert(make_pair(A[i], i));
            even_set.insert(make_pair(-A[i], i));
        }

        vector<vector<bool>> can_reach(n, vector<bool>(2, false));
        can_reach[n - 1][0] = can_reach[n - 1][1] = true;

        for (int i = n - 1; i >= 0; i--) {
            if (can_reach[i][0]) {
                for (int j : odd_left[i])
                    can_reach[j][1] = true;
            }

            if (can_reach[i][1]) {
                for (int j : even_left[i])
                    can_reach[j][0] = true;
            }
        }

        int answer = 0;

        for (int i = 0; i < n; i++)
            if (can_reach[i][1])
                answer++;

        return answer;
    }
};
