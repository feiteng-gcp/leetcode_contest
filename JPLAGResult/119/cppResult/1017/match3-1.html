class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        map<int,int> pos;
        int n = A.size();
        bool first[n];
        bool second[n];
        int ans = 1;
        first[n - 1] = second[n - 1] = true;
        pos[A[n - 1]] = n - 1;
        for (int i = n - 2; i >= 0; i--)
        {
            first[i] = second[i] = false;
            auto it = pos.lower_bound(A[i]);
            if (it != pos.end())
            {
                if (second[it->second]) first[i] = true;
            }

            it = pos.upper_bound(A[i]);
            if (it != pos.begin())
            {
                it--;
                if (first[it->second]) second[i] = true;
            }
            if (first[i]) ans++;
            pos[A[i]] = i;
        }
        return ans;
    }
};
