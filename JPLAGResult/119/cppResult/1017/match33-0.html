class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
	int N = A.size();
	VI good_odd(N);
	VI good_even(N);
	map<int, int> mp;
	good_odd[N-1] = true;
	good_even[N-1] = true;
	mp[A[N-1]] = N-1;
	for (int i=N-2; i>=0; i--) {

	    { // even jump
		auto it = mp.upper_bound(A[i]);
		if (it == mp.begin() || !good_odd[prev(it)->second]) {
		    good_even[i] = false;
		} else {
		    good_even[i] = true;
		}
	    }
	    { // odd jump
		auto it = mp.lower_bound(A[i]);
		if (it == mp.end() || !good_even[it->second]) {
		    good_odd[i] = false;
		} else {
		    good_odd[i] = true;
		}
	    }

	    mp[A[i]] = i;
	}
        
	int ans = 0;
	REP (i, N) ans += good_odd[i];
	return ans;
    }
};
