#include <bits/stdc++.h>
#define rep(i,n)for(int i=0;i<(n);i++)
#define MOD 1000000007
#define INF 0x3f3f3f3f
#define INFL 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
typedef pair<int,int>P;

#ifdef DEBUG
struct TreeNode{
	int val;
	TreeNode*left;
	TreeNode*right;
	TreeNode(int x):val(x),left(NULL),right(NULL){}
};
#endif

class Solution{
public:
	bool dp[30000][2];

	int oddEvenJumps(vector<int>A){
		rep(i,A.size())rep(j,2)dp[i][j]=false;
		set<P>se,se2;
		dp[A.size()-1][0]=dp[A.size()-1][1]=true;
		se.insert(P(A.back(),A.size()-1));

		se2.insert(P(A.back(),-(int)(A.size())+1));
		for(int i=A.size()-2;i>=0;i--){
			auto it=se.lower_bound(P(A[i],0));
			if(it!=se.end()&&dp[it->second][1])dp[i][0]=true;
			auto it2=se2.lower_bound(P(A[i],INF));
			if(it2!=se2.begin()){
				it2--;
				if(dp[-(it2->second)][0])dp[i][1]=true;
			}
			se.insert(P(A[i],i));
			se2.insert(P(A[i],-i));
		}
		int ans=0;
		rep(i,A.size()){
			if(dp[i][0])ans++;
		}
		return ans;
	}
};

#ifdef DEBUG
int main(){
	Solution sol;
	cout<<sol.oddEvenJumps({5,1,3,4,2})<<endl;
}
#endif
