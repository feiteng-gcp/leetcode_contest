class Solution {
public:
	int oddEvenJumps(vector<int>& A) {
		int n = A.size(), i, j, k, ct = 0;
		vector<int> odd(n, -1), even(n, -1);
		map<int, int> v;		// value -> index
		map<int, int>::iterator it;
		v[A[n - 1]] = n - 1;
		for (i = n - 2; i >= 0; i--) {
			// odd jump
			it = v.lower_bound(A[i]);
			if (it != v.end()) odd[i] = it->second;

			// even jump
			it = v.upper_bound(A[i]);
			if (it != v.begin()) {
				it--;
				even[i] = it->second;
			}

			v[A[i]] = i;
		}

		ct = 1;	// last one
		vector<int> dpodd(n, false), dpeven(n, false);
		dpodd[n - 1] = true;
		dpeven[n - 1] = true;
		for (i = n - 2; i >= 0; i--) {
			if ((odd[i] != -1) && (dpeven[odd[i]])) dpodd[i] = true, ct++;
			if ((even[i] != -1) && (dpodd[even[i]])) dpeven[i] = true;
		}

		return ct;
	}
};
