class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int n = A.size();
        map<int, int> idx;
        vector< vector<bool> > f(n, vector<bool>(2, 0));
        f[n - 1][0] = f[n - 1][1] = true;
        idx[A[n - 1]] = n - 1;
        int ans = 1;
        for (int i = n - 2; i >= 0; i--) {
            f[i][0] = f[i][1] = false;
           // printf("%d %d %d\n", i, (int)f[i][0], (int)f[i][1]);
            auto it = idx.lower_bound(A[i]);
            if (it != idx.end()) f[i][1] = f[it->second][0];
            it = idx.upper_bound(A[i]);
            if (it != idx.begin()) {
                it--;
                if (it->first <= A[i]) f[i][0] = f[it->second][1];
            }
            //printf("%d %d %d\n", i, (int)f[i][0], (int)f[i][1]);
            if (f[i][1]) ans++;
            idx[A[i]] = i;
        }
        return ans;
        
    }
};
