class Solution {
public:
    int oddEvenJumps(vector<int>& a) {
        int n = a.size();
        vector<vector<bool> > dp(n, vector<bool>(2));
        dp[n - 1][0] = dp[n - 1][1] = true;
        map<int, int> mp, rmp;
        mp[a[n - 1]] = n - 1;
        rmp[-a[n - 1]] = n - 1;
        for (int i = n - 2; i >= 0; i--) {
            int idx = mp.lower_bound(a[i])->second;
            int ridx = rmp.lower_bound(-a[i])->second;
            if (dp[idx][1]) {
                dp[i][0] = true;
            }
            if (dp[ridx][0]) {
                dp[i][1] = true;
            }
            mp[a[i]] = i;
            rmp[-a[i]] = i;
            // cout << i << ", " << dp[i][0] << ", " << dp[i][1] << endl;

        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (dp[i][0]) {
                res++;
            }
        }
        return res;
    }
};
