class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        
        const int n = A.size();
        unordered_set<int> st;
        st.insert(n-1);
        
        vector<int> odd_next(n, -1), even_next(n, -1);
        // build next steps
        odd_next[n-1]=n-1; even_next[n-1]=n-1; // trapped here
        
        for (int i = n-2; i >= 0; --i) {
            int minval = INT_MAX, maxval = INT_MIN;
            int idx1 = -1, idx2 = -1;
            for (int j = i+1; j < n; ++j) {
                if (A[j] >= A[i] && A[j]<minval) {
                    minval = A[j];
                    idx1 = j;
                }
                if (A[j] <= A[i] && A[j]>maxval) {
                    maxval = A[j];
                    idx2 = j;
                }
                
            }
            odd_next[i] = idx1;
            even_next[i] = idx2;            
        }
        
        for (int i = n-2; i >= 0; --i) {
            if (odd_next[i]>=0 && st.count(even_next[odd_next[i]]))
                st.insert(i);
        }
        return st.size();
    }
};
