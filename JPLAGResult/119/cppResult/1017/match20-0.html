class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int n = A.size();
        vector<int> odd(n, -1), even(n, -1);
        map<int, int> mp1, mp2;
        for (int i = n - 1; i >= 0; --i) {
            auto p1 = mp1.lower_bound(A[i]);
            auto p2 = mp2.lower_bound(-A[i]);
            if (p1 != mp1.end())
                odd[i] = p1->second;
            if (p2 != mp2.end())
                even[i] = p2->second;
            mp1[A[i]] = i;
            mp2[-A[i]] = i;
        }
        
//         for (int i = 0; i < n; ++i)
//             cout << odd[i] << " ";
//         cout << endl;
        
//         for (int i = 0; i < n; ++i)
//             cout << even[i] << " ";
//         cout << endl;
        
        int ans = 1;
        for (int i = 0; i < n - 1; ++i) {
            int k = i, c = 1;
            while (k < n - 1 && k != -1) {
                if (c % 2) {
                    k = odd[k];
                } else {
                    k = even[k];
                }
                ++c;
            }
            if (k != -1) {
                ++ans;
            }
        }
        return ans;
    }
};
