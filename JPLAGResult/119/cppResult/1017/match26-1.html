class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int n = A.size();
        map<int, int> m;
        map<int, int>::const_iterator it;
        map<int, vector<int>> rev;
        for (int i = n-1; i >= 0; --i) {
            int v = A[i];
            it = m.lower_bound(v);
            if (it != m.end()) {
                rev[2*(it->second)].push_back(2*i+1);
            }
            it = m.lower_bound(v+1);
            if (it != m.begin()) {
                --it;
                rev[2*(it->second)+1].push_back(2*i);
            }
            m[v] = i;
        }
        vector<bool> poss(2*A.size(), false);
        poss[2*n-1] = poss[2*n-2] = true;
        int r = 0;
        for (int i = 2*n-1; i >= 0; --i) {
            if (!poss[i]) continue;
            r += i%2;
            for (int k : rev[i]) poss[k] = true;
        }
        return r;
    }
};
