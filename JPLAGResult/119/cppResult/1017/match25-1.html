class Solution {
public:
  int oddEvenJumps(vector<int>& A) {
    int n=A.size();
    vector<int> a(A);
    map<int, int> pos;
    vector<int> dp1(n,0),dp2(n,0);
    int ans=0;
    dp1[n-1]=1;
    dp2[n-1]=1;
    for(int i=n-1;i>=0;i--){
      if(i+1<n){
        {
          auto it=pos.lower_bound(a[i]);
          if(it!=pos.end()){
            int k=it->second;
            dp1[i]=dp2[k];
          }
        }
        {
          auto it=pos.upper_bound(a[i]);         
          if(it!=pos.begin()){
            --it;
            int k=it->second;
            dp2[i]=dp1[k];
          }
        }
      }
      // cout<<i<<" "<<dp1[i]<<" "<<dp2[i]<<endl;
      ans+=dp1[i];
      pos[a[i]]=i;
    }
    return ans;
  }
};
