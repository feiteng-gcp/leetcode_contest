class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int ans = 1, N = A.size();
        multimap<int, int> mt;
        multimap<int, int> mtNegative;
        vector< vector<bool> > flag(N, vector<bool>(2, false));
        flag[N-1][0] = flag[N-1][1] = true;
        mt.insert(pair<int, int>(A[N-1], N - 1));
        mtNegative.insert(pair<int, int>(-A[N-1], N -1));
        for(int i = N - 2; i >= 0; i --) {
            multimap<int, int>::const_iterator citer = mt.lower_bound(A[i]);
            if(citer != mt.end()) {
                std::pair< multimap<int, int>::const_iterator, multimap<int, int>::const_iterator > pr = mt.equal_range(citer->first);
                multimap<int, int>::const_iterator cx = pr.first;
                int index = cx->second;
                while(cx != pr.second) {
                    if(cx->second < index) index = cx->second;
                    cx ++;
                }
                if(flag[index][1]) {
                    ans ++;
                    flag[i][0] = true;
                }
            }
            citer = mtNegative.lower_bound(-A[i]);
            if(citer != mtNegative.end()) {
                std::pair< multimap<int, int>::const_iterator, multimap<int, int>::const_iterator > pr = mtNegative.equal_range(citer->first);
                multimap<int, int>::const_iterator cx = pr.first;
                int index = cx->second;
                while(cx != pr.second) {
                    if(cx->second < index) index = cx->second;
                    cx ++;
                }
                if(flag[index][0]) {
                    flag[i][1] = true;
                }
            }
            mt.insert(pair<int, int>(A[i], i));
            mtNegative.insert(pair<int, int>(-A[i], i));
        }
        return ans;
    }
};
