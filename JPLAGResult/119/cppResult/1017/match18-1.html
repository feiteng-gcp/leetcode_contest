class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int n = A.size();
        vector<vector<bool>> dp(n, vector<bool>(2, false));
        
        dp[n - 1][0] = dp[n - 1][1] = true;
        
        
        for (int i = n - 2; i >= 0; --i)
        {
            int odd = -1;
            int even = -1;
            for (int j = i + 1; j < n; ++j)
            {
                if (A[i] <= A[j])
                {
                    if (odd == -1 || A[j] < A[odd]) odd = j;
                }
                
                if (A[i] >= A[j])
                {
                    if (even == -1 || A[j] > A[even]) even = j;
                }          
                
            }
            
            if (odd != -1)
            {
                dp[i][1] = dp[odd][0];
            }

            if (even != -1)
            {
                dp[i][0] = dp[even][1];

            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; ++i)
        {
            if (dp[i][1])
                ++ans;
        }
        
        return ans;
    }
};
