class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int N = A.size();
        /*
        vector<pair<int, int> > o1, o2;
        for (int i = 0; i < N; i++) {
            o1.push_back(make_pair(A[i], i));//o1: incr
            o2.push_back(make_pair(-A[i], i));//o2: dec
        }
        sort(o1.begin(), o1.end());
        sort(o2.begin(), o2.end());
        */
        /*
        for (int i = 0; i < N; i++) {
            cout << o1[i].first << " ";
        }
            cout  << endl;
        for (int i = 0; i < N; i++) {
            cout << o2[i].first << " ";
        }
            cout  << endl;
        
        */
        vector<int> jmpodd(N, -1);
        vector<int> jmpevn(N, -1);
        vector<vector<int> > Eodd(N);
        vector<vector<int> > Eevn(N);
        
        set<pair<int, int> > st;
        for (int i = N - 1; i>= 0; i--) {
            if (st.lower_bound(make_pair(A[i], i)) != st.end()) {
                int j = st.lower_bound(make_pair(A[i], i))->second;
                jmpodd[i] = j;
                Eodd[j].push_back(i);
            }
            st.insert(make_pair(A[i], i));
        }
        
        set<pair<int, int> > st2;
        for (int i = N - 1; i>= 0; i--) {
            if (st2.lower_bound(make_pair(-A[i], i)) != st2.end()) {
                int j = st2.lower_bound(make_pair(-A[i], i))->second;
                jmpevn[i] = j;
                Eevn[j].push_back(i);
                
            }
            st2.insert(make_pair(-A[i], i));
        }
        /*
        for (int i = 0; i < N ; ) {
            int v = o1[i].first;
            vector<int> ids;
            for (; v == o1[i].first i < N; i++) {
                ids.push_back(o1[i].second);
            } 
            
        }
        for (int i = 0; i < N ; i++) {
            bool b = true;
            for (int j = i - 1; j >= 0 && && b; j++) {
                if (o1[j].second > o1[i].second && A[o1[j].second] >= A[o1[i].second]) {
                    jmpodd[o1[i].second] = o1[j].second;
                    Eodd[o1[j].second].push_back(o1[i].second);
                    b = false;
                }
            }
            for (int j = i + 1; j < N && b; j++) {
                if (o1[j].second > o1[i].second && A[o1[j].second] >= A[o1[i].second]) {
                    jmpodd[o1[i].second] = o1[j].second;
                    Eodd[o1[j].second].push_back(o1[i].second);
                    b = false;
                }
            }
            b = true;
            for (int j = 0; j < N && b; j++) {
                
                if (o2[j].second > o2[i].second && A[o2[j].second] <= A[o2[i].second]) {
                    jmpevn[o2[i].second] = o2[j].second;
                    Eevn[o2[j].second].push_back(o2[i].second);
                    b = false;
                }
            }
        }
        */
        /*
        for (int i = 0; i < N; i++) {
            cout << jmpodd[i] << " ";
        }
        cout << endl;
        for (int i = 0; i < N; i++) {
            cout << jmpevn[i] << " ";
        }
        cout << endl;
        */
        int ret = 0;
        vector<bool> bodd(N, false);//Eodd[u] = v: odd jump at v can jump to u  
        vector<bool> bevn(N, false);
        queue<pair<int, int>> Q;
        Q.push(make_pair(N - 1, 0));
        Q.push(make_pair(N - 1, 1));
        while (!Q.empty()) {
            pair<int, int> p = Q.front(); 
            Q.pop(); 
            int a = p.first;
            if (p.second == 0) {
                if (bevn[a] == false) {
                    ret++;
                    bevn[a] = true;
                    for (auto &u: Eevn[a])
                        Q.push(make_pair(u, 1));
                }
            } else {
                if (bodd[a] == false) {
                    bodd[a] = true;
                    for (auto &u: Eodd[a])
                        Q.push(make_pair(u, 0));
                }
            }
        }
            
        return ret;
    }
};
/*
From starting index i = 0, we can jump to i = 2 (since A[2] is the smallest among A[1], A[2], A[3], A[4] that is greater or equal to A[0]), then we can't jump any more.
From starting index i = 1 and i = 2, we can jump to i = 3, then we can't jump any more.
From starting index i = 3, we can jump to i = 4, so we've reached the end.
From starting index i = 4, we've reached the end already.

*/
