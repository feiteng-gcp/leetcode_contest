class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        vector<bool> a_odd(A.size());
        vector<bool> a_even(A.size());
        int result = 1;
        a_odd[A.size() - 1] = true;
        a_even[A.size() - 1] = true;
        for(int i = A.size() - 2; i >= 0; i--)
        {
            int k_odd = i;
            int k_even = i;
            int n = A[i];
            int n_odd = 100001;

            int n_even = -1;
            for(int j = i + 1; j < A.size(); j++)
            {
                if(A[j] >= n && A[j] < n_odd)
                {
                    n_odd = A[j];
                    k_odd = j;
                }
                if(A[j] <= n && A[j] > n_even)
                {
                    n_even = A[j];
                    k_even = j;
                }
            }
            bool state_odd = a_even[k_odd];
            bool state_even = a_odd[k_even];
            a_odd[i] = state_odd;
            a_even[i] = state_even;
            if(state_odd)
                result++;
        }
        return result;
    }
};
