class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int n = A.size();
        vector<vector<int>> even_prev(n);
        vector<vector<int>> odd_prev(n);
        
        set<pair<int,int>> odd_remaining, even_remaining;
        for (int i = 0; i < n; i++) {
            odd_remaining.insert(pair<int,int>(A[i], i));
            even_remaining.insert(pair<int,int>(-A[i], i));
        }
        
        for (int i = 0; i < n; i++) {
            pair<int,int> oval(A[i], i);
            pair<int,int> eval(-A[i], i);
            
            odd_remaining.erase(oval);
            even_remaining.erase(eval);
            
            auto it = odd_remaining.lower_bound(oval);
            if (it != odd_remaining.end()) {
                odd_prev[it->second].push_back(i);
                //cout << "odd " << i << " to " << it->second << endl;
            }

            auto it2 = even_remaining.lower_bound(eval);
            if (it2 != even_remaining.end()) {
                even_prev[it2->second].push_back(i);
                //cout << "even " << i << " to " << it2->second << endl;
            }
        }

        vector<bool> good(n, false);
        vector<bool> odd_seen(n, false);
        vector<bool> even_seen(n, false);
    
        function<void(int, int)> search = [&](int x, int parity) {
            if (parity == 0) {
                if (even_seen[x])
                    return;
                even_seen[x] = true;
                
                for (int y : odd_prev[x])
                    search(y, 1);
            } else {
                if (odd_seen[x])
                    return;
                odd_seen[x] = true;
                good[x] = true;
                  
                for (int y : even_prev[x])
                    search(y, 0);
            }
        };
        search(n - 1, 0);
        search(n - 1, 1);
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            //cout << i << ": " << good[i] << endl;
            if (good[i])
                ans++;
        }
        return ans;
    }
};
