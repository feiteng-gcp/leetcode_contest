static int endp[20010][2];

class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int N = A.size();
        
        map<int, int> mp;
        int res = 0;

        for (int i = N-1; i >= 0; i--) {
            for (int up = 0; up < 2; up++) {
                endp[i][up] = i;

                if (up) {
                    auto it = mp.lower_bound(A[i]);
                    if (it != mp.end()) {

                        endp[i][up] = endp[it->second][1-up];
                    }
                } else {
                    auto it = mp.upper_bound(A[i]);
                    if (it != mp.begin()) {
                        it--;
                        endp[i][up] = endp[it->second][1-up];
                    }
                }
                
                //printf("endp[%d][%d] = %d\n", i, up, endp[i][up]);
            }
            
            mp[A[i]] = i;
            
            res += endp[i][1] == N-1;
        }
  
        return res;
    }
};
