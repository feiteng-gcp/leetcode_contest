class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int sz = A.size();
        if(sz == 0) return 0;
        int count = 1;
        vector<int> flags(sz);

        flags[sz-1] = 3;
        for(int i = sz - 2; i >= 0; i --){
            int maxofSmall = i;
            int minofLarge = i;
            for(int j = i + 1; j < sz; j ++){
                if(A[j] >= A[i]){
                    if(minofLarge == i){
                        minofLarge = j;
                    } else if (A[j] < A[minofLarge]){
                        minofLarge = j;
                    }
                }
                if(A[j] <= A[i]){
                    if(maxofSmall == i){
                        maxofSmall = j;
                    } else if (A[j] > A[maxofSmall]){
                        maxofSmall = j;
                    }
                }
                // if(A[j] >= A[i] && !hasOdd){
                //     if(flags[j] & 1){
                //         count ++;
                //         flags[i] |= 2;
                //     }
                //     hasOdd = true;
                // }
                // if(A[j] <= A[i] && !hasEven){
                //     if(flags[j] & 2){
                //         flags[i] |= 1;
                //     }
                // }
            }
            if(minofLarge != i){
                if(flags[minofLarge] & 1){
                    count ++;
                    flags[i] |= 2;
                }
            }
            if(maxofSmall != i ){
                if(flags[maxofSmall] & 2){
                    flags[i] |= 1;
                }
            }
        }
        return count;
    }
};
