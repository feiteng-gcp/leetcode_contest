class Solution {
public:
    int oddEvenJumps(vector<int>& a) {
        int n = a.size();
        vector<int> next(n, n);
        vector<int> next2(n, n);
        map<int, int> mp, mp2;
        mp[1000000007] = n;
        mp2[1000000007] = n;
        for(int i = n-1; i >= 0; --i) {
            {
                int pos = mp.lower_bound(a[i])->second;
                next[i] = pos;
            }
            { // >=
                int pos = mp2.lower_bound(-a[i])->second;
                next2[i] = pos;
            }
            mp[a[i]] = i;
            mp2[-a[i]] = i;
        }
        
        vector<vector<int>> dp(n + 1, vector<int>(2, 0));
        dp[n-1][0] = dp[n-1][1] = 1;
        for(int i = n-2; i >= 0; --i) {
            dp[i][0] = dp[next[i]][1];
            dp[i][1] = dp[next2[i]][0];
        }
        
        int ans = 0;
        for(int i = 0; i < n; ++i) {
            ans += dp[i][0];
        }
        return ans;
    }
};
