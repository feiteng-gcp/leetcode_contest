class Solution {
public:
    
    int n;
    vector<int> e[20005][2];
    set<int> gao;
    map<int, int> bxsc;
    queue<pair<int, int> > zyd;
    int vis[20005][2];
    
    void addedge(int a, int b, int s) {
      //  cout << a<< "  " <<" "<<b<<" "<<s<< endl;
        if (b == n) return;
        e[b][s].push_back(a);
    }
    
    int oddEvenJumps(vector<int>& A) {
        n = A.size();
        /*
        for (int i = 0; i < n; ++i)
            gao.push_back(make_pair(A[i], i));
        sort(gao.begin(), gao.end());
        int index = n;
        for (int i = 0; i < n; ++i) {
            addedge(gao[i].second, index, 0);
            index = min(index, gao[i].second);
        }
        for (int i = 0; i < n; ++i) {
            gao[i].second *= -1;
            gao[i].first *= -1;
        }
        sort(gao.begin(), gao.end());
        index = n;
        for (int i = 0; i < n; ++i) {
            addedge(-gao[i].second, index, 1);
            index = min(index, -gao[i].second);
        }*/
        
        gao.clear();
        for (int i = n - 1; i >= 0; i--) {
            if (gao.upper_bound(A[i] - 1) != gao.end()) {
                int t = *(gao.upper_bound(A[i] - 1));
                addedge(i, bxsc[t], 1);
            }
            gao.insert(A[i]);
            bxsc[A[i]] = i;
        }
        
        gao.clear();
      //  cout << gao.size()<<endl;
        for (int i = n - 1; i >= 0; i--) {
            if (gao.upper_bound(A[i]) != gao.begin()) {
                int t = *(--gao.upper_bound(A[i]));
                //cout << i <<" "<<A[i]<<" "<<bxsc[t]<<" "<<t << endl;
                addedge(i, bxsc[t], 0);
            }           
                gao.insert(A[i]);
                bxsc[A[i]] = i;
        }
        
        memset(vis, 0, sizeof(vis));
        vis[n - 1][0] = 1;
        vis[n - 1][1] = 1;
        while(!zyd.empty()) zyd.pop();      
        zyd.push(make_pair(n - 1, 0));
        zyd.push(make_pair(n - 1, 1));
        
        while(!zyd.empty()) {
            int v = zyd.front().first;
            int s = zyd.front().second;
            zyd.pop();
            for (int i = 0; i < (int)e[v][1 - s].size(); ++i) {
                int tv = e[v][1 - s][i];
                int ts = 1 - s;
                if (vis[tv][ts] == 0) {
                    vis[tv][ts] = 1;
                    zyd.push(make_pair(tv, ts));
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) { 
            if (vis[i][1] == 1) {
                ans ++;
             //   cout << i << endl;
            }
        }
        return ans;
        
    }
};
