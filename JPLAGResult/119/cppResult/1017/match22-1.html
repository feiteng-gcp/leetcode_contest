class Solution {
public:
    map<int,int> inds;
    
    // find largest number that is still less than us
    int findEvenHop(vector<int>& A, int i) {
        auto iter = inds.lower_bound(A[i]);
        if (iter == inds.end()) --iter;
        while (iter != inds.begin() && iter->first > A[i]) --iter;
        if (iter == inds.begin() && iter->first > A[i]) return -1;
        return iter->second;
    }
    
    int findOddHop(vector<int>& A, int i) {
        auto iter = inds.lower_bound(A[i]);
        if (iter == inds.end()) return -1;
        return iter->second;
    }
    int oddEvenJumps(vector<int>& A) {
        int n = A.size();

        vector<vector<int>> dp(n, vector<int>(2,0));
        dp[n-1][0]=dp[n-1][1]=n-1;
        inds[A[n-1]] = n-1;
        
        for (int i=n-2; i>=0; i--) {
            // even hop
            int j = findEvenHop(A, i);
            if (j!=-1) dp[i][0] = dp[j][1];
            
            // odd hop A[i]<=A[j]
            j = findOddHop(A, i);
            if (j!=-1) dp[i][1] = dp[j][0];

            inds[A[i]]=i;

        }
        
        int count = 0;
        
        for (int i=n-1; i>=0; i--) {
            if (dp[i][1]==n-1) {
                count++;
            }
        }
        ///
        // for (int i=0; i<n; i++) {
        //     cout<<dp[i][0]<<" ";
        // }
        // cout<<endl;
        // for (int i=0; i<n; i++) {
        //     cout<<dp[i][1]<<" ";
        // }
        // cout<<endl;
        ///
        return count;
    }
};
