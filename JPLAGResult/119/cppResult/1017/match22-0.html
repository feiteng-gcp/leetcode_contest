typedef pair<int, int> ii;

class Solution {
public:
    int oddEvenJumps(vector<int>& a) {
        int n = a.size();
        set<ii> A, B;
        vector<vector<bool>> dp(n, vector<bool>(2));
        dp[n - 1][0] = dp[n - 1][1] = true;
        A.insert({a[n - 1], n - 1});
        B.insert({-a[n - 1], n - 1});
        for (int i = n - 2; i >= 0; --i) {
            {
                auto it = A.lower_bound(ii{a[i], 0});
                if (it != A.end()) {
                    int y = it->second;
                    dp[i][0] = dp[y][1];
                }
            }
            {
                auto it = B.lower_bound(ii{-a[i], 0});
                if (it != B.end()) {
                    int y = it->second;
                    dp[i][1] = dp[y][0];
                }
            }
            A.insert(ii{a[i], i});
            B.insert(ii{-a[i], i});
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            if (dp[i][0]) {
                ++ret;
            }
        }
        return ret;
    }
};
