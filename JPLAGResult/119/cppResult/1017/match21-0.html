class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        int n=A.size();
        vector<int>odd(n,-1);
        vector<int>even(n,-1);
        map<int,int>m;
        for(int i=n-1;i>=0;i--){
            auto it=m.lower_bound(A[i]);
            if(it!=m.end()){
                odd[i]=it->second;
            }
            it=m.upper_bound(A[i]);
            if(it!=m.begin()){
                it--;
                even[i]=it->second;
            }
            m[A[i]]=i;
        }
        vector<pair<int,int>>dp(n);
        int ans=0;
        for(int i=n-1;i>=0;i--){
            if(i+1==n){
                dp[i]={1,1};
            }
            else{
                if(odd[i]!=-1 and dp[odd[i]].second){
                    dp[i].first=1;
                }
                if(even[i]!=-1 and dp[even[i]].first){
                    dp[i].second=1;
                }
            }
            ans+=dp[i].first;
        }
        return ans;
    }
};
