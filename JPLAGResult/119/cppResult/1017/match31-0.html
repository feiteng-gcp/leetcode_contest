class Solution {
public:
    int oddEvenJumps(vector<int>& a) {
        int n = a.size();
        vector<vector<int>> o(n), e(n);
        map<int, int> v;
        for (int i = n; i-- > 0; ) {
            {
                auto it = v.lower_bound(a[i]);
                if (it != v.end()) {
                    o[it->second].push_back(i);
                }
            }
            {
                auto it = v.upper_bound(a[i]);
                if (it != v.begin()) {
                    --it;
                    e[it->second].push_back(i);
                }
            }
            v[a[i]] = i;
        }
        vector<vector<bool>> f(n, vector<bool>(2));
        f[n - 1][0] = f[n - 1][1] = true;
        deque<int> que;
        que.push_back((n - 1) * 2);
        que.push_back((n - 1) * 2 + 1);
        while (!que.empty()) {
            int a = que.front(), b = a % 2;
            a /= 2;
            que.pop_front();
            if (b) {
                for (int x : o[a]) {
                    if (!f[x][1 - b]) {
                        f[x][1 - b] = true;
                        que.push_back(x * 2 + (1 - b));
                    }
                }
            } else {
                for (int x : e[a]) {
                    if (!f[x][1 - b]) {
                        f[x][1 - b] = true;
                        que.push_back(x * 2 + (1 - b));
                    }
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) if (f[i][0]) ++ret;
        // for (int i = 0; i < n; ++i) if (f[i][0]) cout << i << ' '; cout << endl;
        return ret;
    }
};
