#include <set>
#include <vector>
#include <algorithm>

using namespace std;
const int MAXN = 20100;
int N;
int nval[MAXN][2];
bool works[MAXN][2];

class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        N = A.size();
        set <pair <int, int> > s;
        for (int i = N - 1; i >= 0; i--)
        {
            pair <int, int> p;
            p.first = A[i];
            p.second = i;
            if (s.upper_bound(p) != s.end())
                nval[i][1] = s.upper_bound(p) -> second;
            else
                nval[i][1] = N;
            s.insert(p);
        }
        s.clear();
        for (int i = N - 1; i >= 0; i--)
        {
            pair <int, int> p;
            p.first = -A[i];
            p.second = i;
            if (s.upper_bound(p) != s.end())
                nval[i][0] = s.upper_bound(p) -> second;
            else
                nval[i][0] = N;
            s.insert(p);
        }
        s.clear();
        
        works[N-1][0] = works[N-1][1] = true;
        for (int i = N - 2; i >= 0; i--)
        {
            if (nval[i][0] < N && works[nval[i][0]][1])
                works[i][0] = true;
            else
                works[i][0] = false;
            
            if (nval[i][1] < N && works[nval[i][1]][0])
                works[i][1] = true;
            else
                works[i][1] = false;
        }
        
        int ans = 0;
        for (int i = 0; i < N; i++)
            if (works[i][1])
                ans++;
        return ans;
    }
};
