class Solution {
public:
    int oddEvenJumps(vector<int>& A) {
        
        vector<int> odd(A.size());
        vector<int> even(A.size());
        vector<int> oddi(A.size());
        vector<int> eveni(A.size());
        
        int maxi = 100001;
        int mini = -1;
        for(int i=0;i<A.size();i++)
        {
            odd[i] = maxi;
            even[i] = mini;

            oddi[i] = -1;
            eveni[i] = -1;
            
            for(int j=i+1;j<A.size();j++)
            {
                
                if(A[j]>=A[i] && A[j]<odd[i])
                {odd[i] = A[j];
                 oddi[i] =j;
                }
                if(A[j]<=A[i] && A[j]>even[i])
                { 
                    even[i] = A[j];
                 eveni[i] = j;
                }
            }
            
            
        }
        
        int ans=0;
        
        for(int j=0;j<A.size();j++)
        {
        
            int flag=0;
         
            int i=j;
            
        while(1)
        {   
            if(i==A.size()-1)
            {
                ans++;
               // cout<<j<<" ye j "<<endl;
                break;
            }
            
            else if(flag==0)
            {
                
                //cout<<"pehle me"<<endl;
            if(odd[i]!=maxi)
            { i= oddi[i];
             flag=1;
            }
             else
                 break;
            }
            
             else if(flag==1)
            {
                 
                // cout<<"dusre me"<<endl;
            if(even[i]!=mini)
            { i= eveni[i];
             flag=0;
            }
             else
                 break;
            }
            
        }
        }
        
        return ans;
        
    }
};
