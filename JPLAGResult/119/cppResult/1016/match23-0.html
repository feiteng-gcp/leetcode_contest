class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        vector<int> hist(K, 0);
        int sum = 0;
        hist[0]++;
        for (int i = 0; i < A.size(); i++) {
            sum += A[i];
            sum %= K;
            sum += K;
            sum %= K;
            //cout << sum << endl;
            hist[sum]++;
        }
        int total = 0;
        for (int i : hist) {
            if (i > 1) {
                total += i*(i - 1)/2;
            }
        }
        return total;
    }
};
