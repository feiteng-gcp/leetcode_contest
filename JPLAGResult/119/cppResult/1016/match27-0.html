class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        int sum = 0, ans = 0;
        int mod[10003];
        memset(mod, 0, sizeof(mod));
        mod[0] += 1;
        for (int i = 0; i < A.size(); i++){
            sum = (sum +A[i]) % K;
            if (sum < 0)
                sum += K;
            ans += mod[sum];
            mod[sum] += 1;
        }
        return ans;
    }
};
