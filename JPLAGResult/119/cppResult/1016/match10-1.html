class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        unordered_map<int, int> h;
        h[0] += 1;
        

        int sum = 0, result = 0;
        for (int i = 0; i < A.size(); ++i) {
            sum = ((sum + A[i]) % K + K) % K;
            result += h[sum];
            h[sum] += 1;
        }
        
        return result;
    }
};
