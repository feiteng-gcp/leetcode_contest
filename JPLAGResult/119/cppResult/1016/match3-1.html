class Solution {
public:
    int subarraysDivByK(vector<int>& a, int k) {
        int n = a.size();
        vector<int> s(n + 1);
        for (int i = 0; i < n; ++i) {
            s[i + 1] = ((s[i] + a[i]) % k + k) % k;
        }
        unordered_map<int, int> c;

        int ret = 0;
        for (int i = 0; i <= n; ++i) {
            ret += c[s[i]];
            ++c[s[i]];
        }
        return ret;
    }
};
