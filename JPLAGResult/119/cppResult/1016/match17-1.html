class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        map<int,int> cnt;
        cnt[0] = 1;
        int sum = 0;
        int ans = 0;
        for (int i = 0; i < A.size(); i++)
        {
            sum += A[i];
            int rem = sum >= 0 ? sum % K : (K + sum % K) % K;
            ans += cnt[rem];
            cnt[rem]++;
        }
        return ans;
    }
};
