const int maxk = 10000 + 10;
int cnt[maxk];
class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        memset(cnt, 0, sizeof(cnt));
        int n = A.size();
        int sum = 0, ans = 0;
        cnt[0] = 1; // at beginning, empty, sum is 0
        for (int i = 0; i < n; i++) {
            sum = ((sum + A[i]) % K + K) % K;
            ans += cnt[sum];
            cnt[sum]++;
        }
        return ans;
    }
};
