class Solution {
    unordered_map<int, int> cnt;
public:
    int subarraysDivByK(vector<int>& a, int k) {
        cnt[0]++;
        int sum = 0;

        int res = 0;
        for (int i = 0; i < a.size(); ++i) {
            sum = (sum + a[i]) % k;
            if (sum < 0) sum += k;
            res += cnt[sum];
            cnt[sum]++;
        }
        return res;
    }
};
