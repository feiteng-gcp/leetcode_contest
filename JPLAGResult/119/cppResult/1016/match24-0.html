class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        static int cs[10000 + 5];
        memset(cs, 0, sizeof(cs));
        int s = 0, ans = 0;
        cs[0] = 1;
        for (int i = 0; i < A.size(); i ++) {
            s = ((s + A[i]) % K + K) % K;
            ans += cs[s];
            cs[s] += 1;
        }
        return ans;
    }
};
