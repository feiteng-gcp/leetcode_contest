class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        for (int i = 0; i < A.size(); ++i) {
            A[i] = (A[i] % K + K) % K;
        }
        map<int, int> count;
        count[0] = 1;
        int sum = 0;
        int result = 0;
        for (int i = 0; i < A.size(); ++i) {
            sum += A[i];
            sum %= K;
            result += count[sum];
            count[sum]++;
        }
        return result;
    }
};
