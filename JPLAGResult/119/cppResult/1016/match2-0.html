class Solution {
public:
  int subarraysDivByK(vector<int>& A, int K) {
    int n=A.size();
    vector<int> a(A);
    for(int i=0;i<n;i++) a[i]=((a[i]%K)+K)%K;

    int ans=0;
    map<int, int> dp;
    int sum=0;
    dp[sum]++;
    for(int i=0;i<n;i++){            
      sum+=a[i];
      sum%=K;
      ans+=dp[sum];
      dp[sum]++;
    }
    
    return ans;
  }
};
