class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        vector<int> se(K,0);
        se[0]=1;

        int sum=0,n=A.size();
        int ans=0;
        for(int i=0;i<n;i++)
        {
            sum+=A[i];
            sum=(sum%K+K)%K;
            ans+=se[sum];
            se[sum]++;
        }
        return ans;
    }
};
