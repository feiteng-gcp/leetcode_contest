class Solution {
public:
    int buc[20000];
    int subarraysDivByK(vector<int>& A, int K) {
        for(int &el: A){
            if(el < 0) el = -abs(el) % K;
            el = (K + el) % K;
        }
        

        int rela = 0, ans = 0;
        
        for(int i = A.size() - 1; i >= 0; i--){
            buc[rela]++;
            
            rela = (rela + K - A[i]) % K;
            
            ans += buc[rela];
            
        }
        return ans;
        
    }
};
