class Solution {
public:
    int zyd[55000];
    int subarraysDivByK(vector<int>& A, int K) {
        int n = A.size();
        memset(zyd, 0, sizeof(zyd));
        int ans = 0;
        int cur = 0;
        zyd[0] = 1;
        for (int i = 0; i < n; ++i) {
            cur = ((cur + A[i]) % K + K) % K;
            ans = ans + zyd[cur];
            zyd[cur] += 1;
        }
        return ans;
    }
};
