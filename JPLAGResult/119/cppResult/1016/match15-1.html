class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        int n = A.size();
        for(int i=0;i<n;i++) A[i]=(A[i]+10000*K)%K;

        vector<int> cnt(K,0);
        cnt[0]=1;
        int ans=0;
        int sum = 0;
        for(int i=0;i<n;i++){
            sum=(sum+A[i])%K;
            ans+= cnt[sum];
            cnt[sum]+=1;
        }
        
        return ans;
    }
};
