class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        unordered_map<int, int> cnt;
        cnt[0] = 1;
        int ans = 0;
        int tot = 0;
        for (auto &num : A) {
            tot = ((tot + num) % K + K) % K;
           // printf("%d %d\n", tot, cnt[tot]);
            ans += cnt[tot];
            cnt[tot]++;
        }
        return ans;
    }
};
