class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        vector<int>count(K,0);
        int sum=0;
        count[0]=1;
        for(int i=0;i<A.size();i++){
            sum+=A[i];
            sum%=K;
            sum=(sum+K)%K;
            count[sum]+=1;
        }
        int res=0;
        for(auto i: count){
            res+=(i-1)*i/2;
        }
        return res;
    }
};
