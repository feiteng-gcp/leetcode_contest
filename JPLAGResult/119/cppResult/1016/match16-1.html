class Solution {
public:

    int subarraysDivByK(vector<int>& A, int K) {
        int n = A.size();
        for (int i=0; i<n; i++) {
            A[i]%=K;
            while (A[i]<0) A[i] += K;
            A[i]%=K;
        }

        vector<int> viz(K, 0);
        viz[0] = 1;
        int prefix = 0;
        int count = 0;
        for (int i=0; i<n; i++) {
            prefix += A[i];
            prefix %= K;
            count += viz[prefix];
            viz[prefix]++;
        }
        return count;
    }
};
