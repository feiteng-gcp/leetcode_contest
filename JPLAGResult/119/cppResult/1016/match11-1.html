class Solution {
public:
    int subarraysDivByK(vector<int>& a, int k) {
        int n = a.size();
        map<int, int> mp;
        a[0] = (a[0] + 10000*k) % k;
        mp[0]++;
        mp[a[0]]++;
        for (int i = 0; i < a.size() - 1; i++) {

            a[i + 1] = (a[i] + a[i + 1] + 10000*k) % k;
            mp[a[i + 1]]++;
        }
        int res = 0;
        for (auto item : mp) {
            res += item.second * (item.second - 1) / 2;
        }
        return res;
    }
};
