class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        unordered_map<int, int> mp1;
        mp1[0] = 1;
        int sum = 0, res = 0;
        for (int i = 0; i < (int)A.size(); i++) {
            sum += A[i];
            if (mp1.count((sum % K + K) % K)) res += mp1[(sum % K + K) % K];
            mp1[(sum % K + K) % K]++;
        }
        return res;
    }
};
