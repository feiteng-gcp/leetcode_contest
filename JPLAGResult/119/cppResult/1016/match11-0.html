class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        int i = 0;
        map<int, int> m;
        m[0]++;
        for(int j : A){
            i+=j;
            i%=K;
            i+=K;
            i%=K;
            m[i]++;
        }
        int tot = 0;
        for(auto p : m){
            tot += (p.second * (p.second-1))/2;
        }
        return tot;
    }
};
