class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        vector<int> cnt(K,0);
        int N=A.size();
        vector<int> sums(N+1,0);
        cnt[0]++;
        int res=0;
        for(int i=1;i<=N;i++){
            sums[i]=sums[i-1]+A[i-1];
            int r=sums[i]%K;
            if(r<0) r+=K;
            res+=cnt[r];
            cnt[r]++;
        }
        return res;
    }
};
