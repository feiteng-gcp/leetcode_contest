#define ll long long int
class Solution {
public:
    int subarraysDivByK(vector<int>& A, int K) {
        vector<int> kcount(K,0);
        
        int cursum = 0;
        
        for(auto it:A)   
        {
            cursum+=it;
            kcount[((cursum%K)+K)%K]++;
        }
        
        kcount[0]++;
        
        ll ans = 0;
        for(int i=0;i<K;i++)
        {
            if(kcount[i]>1)
            {
                ans = ans+((kcount[i]*(kcount[i]-1))/2);
            }
        }
        
        return ans;
        
    }
};
