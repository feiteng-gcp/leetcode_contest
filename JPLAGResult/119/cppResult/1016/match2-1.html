class Solution {
public:
    int subarraysDivByK(vector<int>& a, int K) {
        int n = a.size();
        vector<int> sum(n + 1);
        for (int i = 0; i < n; ++i) sum[i + 1] = ((sum[i] + a[i]) % K + K) % K;
        vector<int> cnt(K);
        cnt[0] = 1;
        int ret = 0;
        for (int i = 1; i <= n; ++i) {
            ret += cnt[sum[i]];
            cnt[sum[i]]++;
        }
        return ret;
    }
};
