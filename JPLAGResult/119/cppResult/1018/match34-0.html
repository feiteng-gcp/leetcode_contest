class Solution{
public:
	int largestPerimeter(vector<int> A) {
		sort(A.begin(),A.end());
		int Max=0;
		for(int i=2;i<A.size();i++){
			if(A[i-1]+A[i-2]<=A[i])continue;
			Max=max(Max,A[i]+A[i-1]+A[i-2]);
		}
		return Max;
    }
};
