#include <bits/stdc++.h>
using namespace std;
using i64 = int64_t;
using vi = vector<int>;
using vvi = vector<vi>;

class Solution {
public:
    int largestPerimeter(vector<int>& A) {
        sort(A.begin(), A.end());
        int n = A.size();
        for (int b = n - 1; b >= 2; b--) {
            if (A[b] < A[b - 1] + A[b - 2]) {
                return A[b] + A[b - 1] + A[b - 2];
            }
        }
        return 0;
    }
};
