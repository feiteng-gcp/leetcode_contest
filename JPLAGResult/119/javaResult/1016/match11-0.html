class Solution {
    public int subarraysDivByK(int[] A, int K) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int n = A.length;
        int res = 0;
        for (int i = 0, s = 0; i < n; i++) {
            s += A[i];
            s %= K;
            s = (s + K) % K;
            res += map.getOrDefault(s, 0);
            map.put(s, map.getOrDefault(s, 0) + 1);
        }
        // System.out.println(map);
        return res;
    }
}
