class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int n = A.length;
        int[] B = new int[n];
        B[0] = A[0];
        int res = A[0] % K == 0 ? 1 : 0;
        for (int i = 1; i < n; i++) {
            B[i] = B[i - 1] + A[i];
            if (B[i] % K == 0) {
                res++;
            }
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {

                if ((B[i] - B[j]) % K == 0) {
                    res++;
                }
            }
        }
        return res;
    }
}
