class Solution {
    public int subarraysDivByK(int[] A, int K) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int sum = 0, res = 0;
        for (int i = 0; i < A.length; i++) {
            sum += A[i];
            int n = sum % K;
            if (n < 0) n += K;
            int v = map.getOrDefault(n, 0);
            res += v;
            
//             System.out.println(map);
//             System.out.printf("%d %d\n", sum, res);
            map.put(n, v + 1);
        }
        return res;
    }
}
