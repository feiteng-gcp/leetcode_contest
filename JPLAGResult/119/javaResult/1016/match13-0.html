class Solution {
    public int subarraysDivByK(int[] A, int K) {
        if (A == null || A.length == 0) {
            return 0;
        }
        int n = A.length;
        int res = 0;
        int mod[] = new int[K];
        Arrays.fill(mod, 0);
        int cumSum = 0;
        for (int i = 0; i < n; i++) {
            cumSum += A[i];
            mod[((cumSum % K) + K) % K]++;
        }
        for (int i = 0; i < K; i++) {
            if (mod[i] > 1) {
                res += (mod[i] * (mod[i] - 1)) / 2;
            }
        }
        res += mod[0];
        return res;
    }
}
