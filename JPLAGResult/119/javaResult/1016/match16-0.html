class Solution {
    public int subarraysDivByK(int[] arr, int K) {
        int mod[] = new int[K]; 
        Arrays.fill(mod, 0); 

        int cumSum = 0; 
        for (int i = 0; i < arr.length; i++) { 
            cumSum += arr[i]; 
            mod[((cumSum % K) + K) % K]++; 
        } 
  
        int result = 0; 
  
        for (int i = 0; i < K; i++) 
            if (mod[i] > 1) 
                result += (mod[i] * (mod[i] - 1)) / 2; 
        result += mod[0]; 
  
        return result; 
    }
}
