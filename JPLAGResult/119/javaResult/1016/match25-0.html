class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int mod[] = new int[K];
        Arrays.fill(mod, 0);
        int cumSum = 0;
        int n = A.length;
        for (int i = 0; i < n; i++) {
            cumSum += A[i];
            mod[((cumSum % K) + K) % K]++;
        }
        int result = 0;
        for (int i = 0; i < K; i++)
            if (mod[i] > 1)
                result += (mod[i] * (mod[i] - 1)) / 2;
        result += mod[0];
        return result;
    }
}
