class Solution {
    public int subarraysDivByK(int[] nums, int k) {
        int dp[] = new int[k]; 
        int n = nums.length;
        int cumSum = 0; 
        for (int num : nums) { 
            cumSum += num; 
            dp[(cumSum % k + k) % k]++; 
        } 
        int result = 0; 
        for (int i : dp){
            result += i * (i - 1) / 2;      
        } 
        return result + dp[0]; 
    }
}
