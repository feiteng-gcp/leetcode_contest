class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int[] map = new int[K];
        map[0]++;
        
int sum = 0, res = 0;
        for (int a : A) {
            sum += a;
            res += map[(sum % K + K) % K];
            map[(sum % K + K) % K]++;
        }
        return res;
    }
}
