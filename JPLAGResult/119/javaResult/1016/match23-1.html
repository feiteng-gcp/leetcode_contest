class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int[] a = new int[K];
        
int n = A.length;
        int[] s = new int[n + 1];
        for (int i = 0; i < A.length; i++) {
            s[i + 1] = s[i] + A[i]; 
        }
        
for(int x : s) {
            x %= K;
            if (x < 0) {
                x += K;
            }
            a[x]++;
        }
        int res = 0;
        for (int x : a) {
            if (x > 1) {
                res += x * (x - 1) / 2;
            }
        }
        return res;
    }
}
