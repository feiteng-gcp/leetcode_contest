class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int[] count = new int[K];
        count[0] = 1;
        int sum = 0;
        for (int x : A) {
            sum += x;
            ++count[((sum % K) + K) % K];
        }
        int ans = 0;
        for (int i = 0; i < K; i++) {
            ans += count[i] * (count[i] - 1) / 2;
        }
        return ans;
    }
}
