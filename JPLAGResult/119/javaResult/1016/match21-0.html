class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int n = A.length;
        int[] pref = new int[1 + n];
        for (int i = 1; i <= n; i++) {
            pref[i] = pref[i - 1] + A[i - 1];
        }
        for (int i = 0; i <= n; i++) {
            pref[i] %= K;
            if (pref[i] < 0) {
                pref[i] += K;
            }
        }

        Map<Integer, Integer> hm = new HashMap<>();
        hm.put(0, 1);
        int res = 0;
        for (int i = 1; i <= n; i++) {
            int p = pref[i];
            res += hm.getOrDefault(p, 0);
            hm.put(p, hm.getOrDefault(p, 0) + 1);
        }

        return res;
    }
}
