class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int[] bucket = new int[K];
        
        
int sum = 0;
        for (int i = 0; i < A.length; i++) {
            sum += A[i];
            bucket[((sum % K) + K) % K]++;
        }
        
        int res = 0;
        for (int i = 0; i < K; i++) {
            if (bucket[i] > 1) {
                res += (bucket[i] * (bucket[i] - 1)) / 2;
            }
        }
        res += bucket[0];
        
        return res;
    }
}
