class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int len = A.length;
        int[] sumLeft = new int[len + 1];
        for (int i = 0; i < len; i++) {
            sumLeft[i + 1] = sumLeft[i] + A[i];
        }

        int count = 0;
        for (int i = 0; i < len; i++) {
            
for (int j = i + 1; j <= len; j++) {
                if ((sumLeft[j] - sumLeft[i]) % K == 0) {
                    count++;
                }
            }
        }
        return count;
    }
}
