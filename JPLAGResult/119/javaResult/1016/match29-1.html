class Solution {
    public int subarraysDivByK(int[] A, int K) {
        int n = A.length;
        int[] dp = new int[n+1]; //sum of A[0:n], prefix sum, dp[0] = 0, dp[1]=A[0]
        dp[0]=0;
        for(int i=1; i<=n; i++){
            dp[i] = (dp[i-1] + A[i-1])%K;
        }
        int cnt=0;
        for(int i=0; i<n; i++){
            for(int j=i+1; j<=n; j++){
                int sum = dp[j] - dp[i];
                if(sum % K ==0)
                    cnt++
;    
            }
        }
        return cnt;
    }
}
