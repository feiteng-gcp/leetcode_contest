class Solution {
    public int subarraysDivByK(int[] A, int K) {
        if (A == null || K == 0) {
            return -1;
        }
        if (A.length == 0) {
            return 0;
        }
        // int[] preSum = new int[A.length];
        int sum = 0;
        int[] count = new int[K];
        
int res = 0;
        for (int i = 0; i < A.length; i++) {
            sum += A[i];
            count[((sum % K) + K) % K]++; 
        }
        for (int i = 0; i < K; i++) {
            if (count[i] > 1) {
                res += (count[i] * (count[i] - 1)) / 2; 
            }
        }
        res += count[0]; 
        return res;
    }
}
