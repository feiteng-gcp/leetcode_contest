class Solution {
    public int[][] kClosest(int[][] points, int K) {
        List<int[]> list = new ArrayList<>();
        for (int[] point : points) {
            list.add(new int[]{point[0] * point[0] + point[1] * point[1], point[0], point[1]});
        }
        Collections.sort(list, (a, b) -> a[0] - b[0]);
        int[][] res = new int[K][2];
        for (int i = 0; i < K; i++) {
            res[i][0] = list.get(i)[1];
            res[i][1] = list.get(i)[2];
        }
        return res;
    }
}
