class Solution {
    public int[][] kClosest(int[][] points, int K) {
        int[][] ans = new int[K][2];
        Queue<int[]> queue = new PriorityQueue<>((a, b) -> {
            int A = a[0] * a[0] + a[1] * a[1];
            int B = b[0] * b[0] + b[1] * b[1];
            return B - A;
        }); 
        for (int[] point : points) {
            queue.offer(point);
            if (queue.size() > K) {
                queue.poll();
            }
        }
        int idx = 0;
        while (!queue.isEmpty()) {
            ans[idx ++] = queue.poll();
        }
        return ans;
    }
}
