class Solution {
    public int[][] kClosest(int[][] points, int K) {
        PriorityQueue<int[]> que = new PriorityQueue<>((a, b)-> a[0]*a[0]+a[1]*a[1]-b[0]*b[0]-b[1]*b[1]);
        for(int i=0; i<points.length; i++) {
            que.add(points[i]);
        }
        int[][] rv = new int[K][2];
        for(int i=0; i<K; i++) {
            int[] t = que.poll();
            rv[i][0] = t[0];
            rv[i][1] = t[1];
        }
        return rv;
    }
}
