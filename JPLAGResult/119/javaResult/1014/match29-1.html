class Solution {
    public int[][] kClosest(int[][] points, int K) {
        int[][] arr = new int[K][2];
        Queue<int[]> queue = new PriorityQueue<int[]>((x, y) -> y[0] * y[0] + y[1] * y[1] - x[0] * x[0] - x[1] * x[1]);
        for (int[] p : points) {
            queue.offer(new int[]{p[0], p[1]});
            if (queue.size() > K)
                queue.poll()
;
        }
        int index = 0;
        while (!queue.isEmpty()) {
            arr[index++] = queue.poll();
        }
        return arr;
    }
}
