class Solution {
    public int[][] kClosest(int[][] points, int K) {
        PriorityQueue<Point> q = new PriorityQueue<>((a, b) -> b.d - a.d);
        for (int[] p : points) {
            q.add(new Point(p));
            if (q.size() > K) {
                q.remove();
            }
        }
        int[][] ans = new int[K][];
        for (int i = 0; i < K; i++) {
            ans[i] = q.remove().point;
        }
        return ans;
    }
}

class Point {
    int[] point;
    int d;
    
    public Point(int[] point) {
        this.point = point;
        this.d = point[0] * point[0] + point[1] * point[1];
    }   
}
