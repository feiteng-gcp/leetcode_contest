class Solution {
    public int[][] kClosest(int[][] points, int K) {
        Queue<int[]> queue = new PriorityQueue<>(new Comparator<int[]>() {
			@Override
			public int compare(int[] o1, int[] o2) {
				Long t1 = (long) o1[0] * o1[0] + (long) o1[1] * o1[1];
				Long t2 = (long) o2[0] * o2[0] + (long) o2[1] * o2[1];
				return t2.compareTo(t1);
			}
        	
        });
        for (int[] point : points) {
        	queue.add(point);
        	if (queue.size() > K) {
        		queue.poll();
        	}
        }
        int[][] res = new int[K][2];
        int i = 0;
        while(!queue.isEmpty()) {
        	res[i++] = queue.poll();
        }
        return res;
    }
}
