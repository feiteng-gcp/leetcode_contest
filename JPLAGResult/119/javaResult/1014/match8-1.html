class Solution {
    public int[][] kClosest(int[][] points, int K) {
        PriorityQueue<int[]> q = new PriorityQueue<>(new Comparator<int[]>(){
            @Override
            public int compare(int[] p2, int[] p1) {
                return p1[0]*p1[0] + p1[1]*p1[1] - p2[0]*p2[0] - p2[1]*p2[1];
            
}
        });
        
        for (int[] p : points) {
            q.add(p);
            if (q.size() > K) {
                q.poll();
            }
        }
        int[][] res = new int[K][2];
        int idx = 0;
        while (!q.isEmpty()) {
            res[idx++] = q.poll();
        }
        return res;
    }
}
