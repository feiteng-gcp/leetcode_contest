class Solution {
    public int[][] kClosest(int[][] points, int K) {
        Queue<int[]> q = new PriorityQueue<>(new Comparator<int[]>() {
            
            public int compare(
int[] a, int[] b) {
                return a[0] * a[0] + a[1] * a[1] - b[0] * b[0] - b[1] * b[1];
            }
        });
        
        for (int[] x : points) {
            q.add(x);
        }
        int[][] ret = new int[K][2];
        int cnt = 0;
        while (cnt < K) 
{
            ret[cnt++] = q.poll(); 
        } 
        return ret; 
    }
}
