class Solution {
    public int[][] kClosest(int[][] points, int K) {
        int[][] res = new int[K][2];
        PriorityQueue<int[]> pq = new PriorityQueue<>(K + 1, (b, a)->(a[0]*a[0] + a[1]*a[1] - b[0] * b[0] - b[1]* b[1]));
        for (int[] p : points) {
            pq.offer(p);
            if (pq.size() > K) {
                pq.poll();
            }
        }
        for (int i = res.length - 1; i >= 0; i--) {
            res[i] = pq.poll();
        }
        return res;
    }
}
