import java.util.*;
import java.util.stream.*;
import java.math.*;

public class Solution {

    public int[][] kClosest(int[][] points, int K) {
        return Stream.of(points).sorted(Comparator.<int[]>comparingLong(arr -> (long) arr[0] * arr[0] + (long) arr[1] * arr[1])).limit(K).toArray(int[][]::new);
    }


    public static void main(String[] args) {
        Solution sol = new Solution();

    }
}
