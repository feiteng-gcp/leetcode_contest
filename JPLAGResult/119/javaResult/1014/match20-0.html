class Solution {
    public int[][] kClosest(int[][] points, int k) {
        Queue<int[]> q = new PriorityQueue<>((o1, o2) -> -compareTo(o1, o2));
        for(int[] p : points){
            q.add(p);
            if(q.size() > k){
                q.remove();
            }
        }
        int[][] res = new int[k][2];
        for(int i = k - 1; i >= 0; i--){
            res[i] = q.remove();
        }
        return res;
    }
    public int compareTo(int[] p1, int[] p2){
        int dx1 = p1[0];
        int dy1 = p1[1];
        int dx2 = p2[0];
        int dy2 = p2[1];
        int d = dx1 * dx1 + dy1 * dy1 - dx2 * dx2 - dy2 * dy2;
        return d;
    }
}
