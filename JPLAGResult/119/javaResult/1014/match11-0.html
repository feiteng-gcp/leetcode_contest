class Solution {
    public int[][] kClosest(int[][] points, int K) {
        int[][] result = new int[K][2];
        PriorityQueue<int[]> pq = new PriorityQueue<>(new Comparator<int[]>() {
            public int compare(int[] a, int[] b) {
                int sqrA = (int)Math.pow(a[0], 2) + (int)Math.pow(a[1], 2);
                int sqrB = (int)Math.pow(b[0], 2) + (int)Math.pow(b[1], 2);
                return sqrA - sqrB;
            }
        });
        for (int[] point : points) {
            pq.offer(point);
        }
        for (int i = 0; i < K; i++) {
            result[i] = pq.poll();
        }
        return result;
    }
}
