class Solution {
    
public int[][] kClosest(int[][] points, int K) {
        int len = points.length;
        point[] allp = new point[len];
        for(int i = 0; i < len; i++){
            allp[i] = new point(points[i][0], points[i][1]);
        }
        Arrays.sort(allp);
        int[][] result = new int[K][2];
        
for(int i = 0; i < K; i++){
            result[i][0] = allp[i].x;
            result[i][1] = allp[i].y;
        }
        return result;
    }
}

class point implements Comparable<point>{
    int x;
    int y;
    double dis;
    public point(int x, int y){
        this.x = x;
        this.y = y;
        this.dis = Math.sqrt(x * x + y * y);
    }
    public int compareTo(point a){
        if(dis < a.dis)
            return -1;
        else
            return 1;
    }
}
