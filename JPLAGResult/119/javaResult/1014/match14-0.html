class Solution {
    public int[][] kClosest(int[][] points, int K) {
        PriorityQueue<int[] > pq = new PriorityQueue<>((a, b) ->{
            int disa = (int) (Math.pow((a[0] - 0),2) + Math.pow((a[1] - 0), 2));
                int disb = (int) (Math.pow((b[0] - 0),2) + Math.pow((b[1] - 0), 2));
                int sub = disa - disb;
                return sub;
        });
        
        for(int [] p : points){
            pq.offer(p);
        }
        
        int[][] res = new int [K][2];
        for(int i = 0 ; i < K ; i++){
            res[i] = pq.poll();
        }
        
        return res;
    }
}
