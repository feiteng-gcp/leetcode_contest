class Solution {
    public int[][] kClosest(int[][] points, int K) {
        if(points ==  null || points.length == 0) return points;
        
PriorityQueue<int[]> pq = new PriorityQueue<int[]>((a, b) -> (a[0] * a[0] + a[1] * a[1] - b[0] * b[0] - b[1] * b[1]));
        for(int i = 0; i < points.length; i++) {
            pq.offer(points[i]);
        }
        int[][] out = new int[K][2];
        for(int i = 0; i < K; i++) {
            int[] temp = pq.poll();
            out[i][0] = temp[0];
            out[i][1] = temp[1];
        }
        return out;
    }
}
