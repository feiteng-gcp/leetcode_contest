class Solution {
    public int[][] kClosest(int[][] points, int K) {
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>(1, new Comparator<int[]>(){
            public int compare(int[] a, int[] b) {
                long ax = (long)a[0], ay = (long)a[1], bx = (long)b[0], by = (long)b[1];
                return Long.compare(bx*bx + by*by, ax*ax + ay*ay);
            }
        });
        for (int[] point : points) {
            pq.add(point);
            if (pq.size() > K) {
                pq.poll();
            }
        }
        int[][] kClosest = new int[K][2];
        for (int i = 0; i < K; i++) {
            kClosest[i] = pq.poll();
        }
        return kClosest;
    }
}
