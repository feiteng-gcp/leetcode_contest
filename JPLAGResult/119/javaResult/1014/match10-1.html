class Solution {
    public int[][] kClosest(int[][] points, int K) {
        Arrays.sort(points, (int[] a, int[] b) -> {
            Long dist1 = new Long(a[0]*a[0]);
            dist1 += a[1]*a[1];
            Long dist2 = new Long(b[0]*b[0]);
            dist2 += b[1]*b[1];
            
return dist1.compareTo(dist2);
        });
        
        int[][] res= new int[K][2];
        for (int i=0; i<K; i++){
            res[i][0] = points[i][0];
            res[i][1] = points[i][1];
        }
        
        return res;
    }  
}
