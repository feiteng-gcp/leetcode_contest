class Solution {
    public int largestPerimeter(int[] A) {
        Arrays.sort(A);
        int n = A.length;
        for(int i=n-1;i>=2;i--) {
            int j = i-1, k = i-2;
            if (A[j] + A[k] > A[i]) {
                return A[j] + A[k] + A[i];
            }
        }
        return 0;
    }
}
