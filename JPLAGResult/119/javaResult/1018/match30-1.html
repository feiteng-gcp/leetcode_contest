class Solution {
  public int largestPerimeter(int[] A) {
    if (A == null || A.length < 3) {
      return 0;
    }
    
    Arrays.sort(A);
    
int res = 0;
    for (int i = A.length - 1; i >= 2; i--) {
      if (A[i - 2] + A[i - 1] > A[i]) {
        res = Math.max(res, A[i - 2] + A[i - 1] + A[i]);
      }
    }
    
    return res;
  }
}
