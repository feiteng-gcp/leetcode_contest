class Solution {
    public int largestPerimeter(int[] A) {
        int res = 0;
        Arrays.sort(A);
        
for (int i = A.length - 1; i > 1; i--) {
            if (A[i] < A[i - 1] + A[i - 2]) {
                res = Math.max(res, A[i] + A[i - 1] + A[i - 2]);
            }
        }
        return res;
    }
}
