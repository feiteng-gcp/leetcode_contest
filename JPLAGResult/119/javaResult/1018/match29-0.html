class Solution {
public int largestPerimeter(int[] A) {
        Arrays.sort(A);
        int n = A.length;
        int ans = 0;
        for (int i = n - 1; i >= 0; i--) {
            if (i - 2 >= 0 && A[i] < A[i - 1] + A[i - 2]) {
                ans = Math.max(ans, A[i] + A[i - 1] + A[i - 2]);
            }
        }
        return ans;
    }
}
