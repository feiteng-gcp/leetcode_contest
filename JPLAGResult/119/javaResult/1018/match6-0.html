class Solution {
    public int largestPerimeter(int[] A) {
        Arrays.sort(A);
        int res = 0;
        int n = A.length;
        for (int i = 2; i < n; i++) {
            if (A[i - 2] + A[i - 1] > A[i]) {
                res = Math.max(res, A[i - 2] + A[i - 1] + A[i]);
            }
        }

        return res;
    }
}
