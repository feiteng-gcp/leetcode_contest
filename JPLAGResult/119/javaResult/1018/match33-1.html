import java.util.*;
import java.util.stream.*;
import java.math.*;


public class Solution {

    public int largestPerimeter(int[] A) {
        Arrays.sort(A);
        int ans = 0;
        for (int i = 0; i + 2 < A.length; i++) {
            int a = A[i], b = A[i + 1], c = A[i + 2];
            
if (a + b > c) {
                ans = Math.max(ans, a + b + c);
            }
        }
        return ans;
    }


    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.largestPerimeter(new int[]{2, 1, 2}));
        System.out.println(sol.largestPerimeter(new int[]{1, 2, 1}));
        System.out.println(sol.largestPerimeter(new int[]{3, 2, 3, 4}));
        System.out.println(sol.largestPerimeter(new int[]{3, 6, 2, 3}));
    }
}
