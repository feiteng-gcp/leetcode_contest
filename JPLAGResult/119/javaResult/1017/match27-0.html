class Solution {
    public int oddEvenJumps(int[] A) {
        int ans = 0;
        for (int i = 0; i < A.length; i++) {
            if (valid(A, i, true)) {
                System.out.println(i);
                ans ++;
            }
        }
        return ans;
    }
    
    private boolean valid(int[] A, int i, boolean odd) {
        if (i == A.length - 1) {
            return true;
        }
        int idx = helper(A, i, odd);
        if (idx == -1) {
            return false;
        }
        return valid(A, idx, !odd);
    }
    
    private int helper(int[] A, int i, boolean odd) {
        int min = Integer.MAX_VALUE, max = Integer.MIN_VALUE;
        int oddIdx = -1, evenIdx = -1;
        for (int j = i + 1; j < A.length; j++) {
            if (odd) {
                if (A[i] <= A[j] && A[j] < min) {
                    min = A[j];
                    oddIdx = j;
                }
            } else {
                if (A[i] >= A[j] && A[j] > max) {
                    max = A[j];
                    evenIdx = j;
                }
            }
        }
        return odd ? oddIdx : evenIdx;
    }
}
