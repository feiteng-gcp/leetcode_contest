class Solution {
  public int oddEvenJumps(int[] A) {
    if (A == null || A.length == 0) {
      return 0;
    }
    int len = A.length;
    boolean[] even = new boolean[A.length];
    boolean[] odd = new boolean[A.length];
    TreeMap<Integer, Integer> map = new TreeMap<>();
    even[len - 1] = true;
    odd[len- 1] = true;
    
map.put(A[len - 1], len - 1);
    for (int i = A.length - 2; i >= 0; i--) {
      Map.Entry<Integer, Integer> e_even = map.floorEntry(A[i]);
      if (e_even != null) {
        even[i] = odd[e_even.getValue()];
      }
      
      Map.Entry<Integer, Integer> e_odd = map.ceilingEntry(A[i]);
      if (e_odd != null) 
{
        odd[i] = even[e_odd.getValue()];
      }
      
      map.put(A[i], i);
    }
    
    int res = 0;
    for (int i = 0; i < len; i++) {
      
if (odd[i]) {
        res++;
      }
    }
    return res;
  }
}
