class Solution {
    public int oddEvenJumps(int[] A) {
        boolean[] odd = new boolean[A.length];
        boolean[] even = new boolean[A.length];
        Map<Integer, Integer> pm = new HashMap<>();
        TreeSet<Integer> ts = new TreeSet<>();
        odd[A.length - 1] = true;
        even[A.length - 1] = true;
        ts.add(A[A.length - 1]);
        pm.put(A[A.length - 1], A.length - 1);
        
        for(int i = A.length - 2; i >= 0; i --) {
            // cnt
            Integer nextV = ts.ceiling(A[i]);
            Integer next = nextV == null ? null : pm.get(nextV);
            odd[i] = next != null && even[next];
            nextV = ts.floor(A[i]);
            next = nextV == null ? null : pm.get(nextV);
            even[i] = next != null && odd[next];
            
            // update
            if(!pm.containsKey(A[i])) {
                ts.add(A[i]);
            }
            pm.put(A[i], i);
        }
        
        int ans = 0;
        for(int i = 0; i < odd.length; i ++) {
            if(odd[i]) {
                ans ++;
            }
        }
        return ans;
    }
}
