class Solution {
    public int oddEvenJumps(int[] A) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        
int[] greater = new int[A.length];
        int[] less = new int[A.length];
        Arrays.fill(greater, -1);
        Arrays.fill(less, -1);
        for (int i = A.length - 1; i >= 0; i--)
        {
            Integer larger = map.ceilingKey(A[i]);
            if (larger != null)
                greater[i] = map.get(larger);
            Integer smaller = map.floorKey(A[i]);
            if (smaller != null)
                less[i] = map.get(smaller); 
            map.put(A[i], i);
        }
        int count = 0;
        
for (int i = 0; i < A.length; i++)
        {
            if (isReachable(A, greater, less, i))
                count++;
        }
        return count;
    }
    
    private boolean isReachable(int[] A, int[] greater, int[] less, int index)
    {
        if (index == A.length - 1) return true;
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(index);
        int step = 1;
        while (!queue.isEmpty())
        {
            int cur = queue.poll();
            if (cur == A.length - 1)
                return true;
            else if (cur == -1)
                return false;
            if (step % 2 == 0)
                queue.offer(less[cur]);
            else
                queue.offer(greater[cur]);
            step++;
            //System.out.printf("cur = %d, greater = %d, less = %d\n", cur, greater[cur], less[cur]);
        }
        return false;
    }
}
