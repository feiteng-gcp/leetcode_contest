class Solution {
    public int oddEvenJumps(int[] A) {
        // for each i, store the next odd jump and even jump
        int[] nextOddJump = new int[A.length];
        int[] nextEvenJump = new int[A.length];
        TreeMap<Integer, Integer> map = new TreeMap<>(); // value -> index
        map.put(A[A.length-1], A.length-1);
        for(int i = A.length-2; i>= 0; i--) {
            Integer ceilingKey = map.ceilingKey(A[i]);
            Integer floorKey = map.floorKey(A[i]);
            
            nextOddJump[i] = ceilingKey == null ? -1 : map.get(ceilingKey);
            nextEvenJump[i] = floorKey == null ? -1 : map.get(floorKey);
            
            map.put(A[i], i);
        }
//         for(int i = 0; i < A.length; i++) {
//             System.out.println("nextOddJump " + i + " - " + nextOddJump[i]);
//             System.out.println("nextEvenJump " + i + " - " + nextEvenJump[i]);

//         }
        boolean[][] dp = new boolean[2][A.length];
        dp[0][A.length - 1] = true; // can reach end by doing odd jump
        dp[1][A.length - 1] = true; // can reach end by doing even jump

        for(int i = A.length-2; i>= 0; i--) {
            int nextOdd = nextOddJump[i];
            int nextEven = nextEvenJump[i];
            
if(nextOdd != -1 && dp[1][nextOdd]) dp[0][i] = true;
            if(nextEven != -1 && dp[0][nextEven]) dp[1][i] = true;
        }
        int count = 0;
        for(int i = 0; i < dp[0].length; i++) {
            if(dp[0][i]) {
                // System.out.println("ODD: can reach from index " + i);
                count++;
            }
            // if(dp[1][i]) System.out.println("EVEN: can reach from index " + i);
        }
        return count;
    }
}
