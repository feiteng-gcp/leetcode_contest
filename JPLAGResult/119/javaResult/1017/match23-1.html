class Solution {
	int[] odd, even;
	Boolean[][] mem;
	int len;
	
    public int oddEvenJumps(int[] A) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        len = A.length;
        odd = new int[len]; even = new int[len];
        for (int i = len - 1; i >= 0; --i) {
        	Integer
 tmp = map.ceilingKey(A[i]);
        	if (tmp == null) {
        		odd[i] = -1;
        	}else {
        		odd[i] = map.get(tmp);
        	}
        	map.put(A[i], i);
        }
        map = new TreeMap<>();
        for (int i = len - 1; i >= 0; --i) {
        	Integer
 tmp = map.floorKey(A[i]);
        	if (tmp == null) {
        		even[i] = -1;
        	}else {
        		even[i] = map.get(tmp);
        	}
        	map.put(A[i], i);
        }
        int res = 0;
        mem = new Boolean[len][2];
        mem[len-1][0] = true; mem[len-1][1] = true;
        for (int i = 0; i < len; ++i) {
        	if (helper(i, true)) {

        		++res;
        	}
        }
        return res;
    }
    
    private boolean helper(int p, boolean isOdd) {
    	int p2 = 0;
    	if (isOdd) {
    		p2 = 1;
    	}
    	if (mem[p][p2] != null) {
    		return mem[p][p2];
    	}
    	int next = -1;
    	if (isOdd) {
    		next = odd[p];
    	}else {
    		next = even[p];
    	}
    	if (next == -1) {
    		mem[p][p2] = false;
    	}else {
    		mem[p][p2] = helper(next, !isOdd);
    	}
    	return mem[p][p2];
    }
}
