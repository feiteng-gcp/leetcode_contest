class Solution {
    public int oddEvenJumps(int[] A) {
		int n = A.length;
		int[][] dp = new int[n][2];
		dp[n-1][0] = dp[n-1][1] = 1;
		TreeMap<Integer, Integer> map = new TreeMap<>();
		map.put(A[n-1], n-1);
		for (int i = n- 2; i >= 0; i--) {
			Map.Entry<Integer, In
teger> entry = map.floorEntry(A[i]);
			if (entry != null) {
				dp[i][0] = dp[entry.getValue()][1];
			} else {
				dp[i][0] = 0;
			}
			entry = map.ceilingEntry(A[i]
);
			if (entry != null) {
				dp[i][1] = dp[entry.getValue()][0];
			} else {
				dp[i][1] = 0;
			}
			map.put(A[i], i);
		}
		int ans = 0;
		for (int i = 0; i < n; i++) {
			ans += dp[i][1];
		}
		return ans;
	}
}
