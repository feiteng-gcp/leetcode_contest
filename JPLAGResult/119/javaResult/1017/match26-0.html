class Solution {
    public int oddEvenJumps(int[] A) {
        int n = A.length;
        if (n <= 1) {
            return n;
        }
        
        int[] incNext = new int[n];
        int[] decNext = new int[n];
        Arrays.fill(incNext, -1);
        Arrays.fill(decNext, -1);
        
        TreeMap<Integer, Integer> treeMap = new TreeMap<>();
        for (int i = n - 1; i >= 0; i--) {
            Integer floor = treeMap.floorKey(A[i]);
            if (floor != null) {
                decNext[i] = treeMap.get(floor);
            } 
            Integer ceiling = treeMap.ceilingKey(A[i]);
            if (ceiling != null) {
                incNext[i] = treeMap.get(ceiling);
            }
            treeMap.put(A[i], i);
        }
        
        boolean[] inc = new boolean[n], dec = new boolean[n];
        int ans = 1;
        inc[n - 1] = dec[n - 1] = true;
        for (int i = n - 2; i >= 0; i--) {
            if (incNext[i] != -1 && dec[incNext[i]]) {
                inc[i] = true;
                ans++;
            }
            if (decNext[i] != -1 && inc[decNext[i]]) {
                dec[i] = true;
            }
        }
        return ans;
    }
}
