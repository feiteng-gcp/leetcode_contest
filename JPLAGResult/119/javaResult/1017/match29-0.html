class Solution {
    public int oddEvenJumps(int[] A) {
        int res = 1;
        int[] odd = new int[A.length], even = new int[A.length];
        Arrays.fill(odd, -1);
        Arrays.fill(even, -1);
        for (int i = 0; i < A.length-1; i++){
            for (int j = i + 1; j < A.length; j++){
                if (A[j] == A[i]){
                    odd[i] = j;
                    even[i] = j;
                    break;
                } else if (A[j] > A[i]){
                    if (odd[i] == -1){
                        odd[i] = j;
                    } else {
                        odd[i] = A[odd[i]] <= A[j] ? odd[i] : j;
                    }
                } else {
                    if (even[i] == -1){
                        even[i] = j;
                    } else {
                        even[i] = A[even[i]] >= A[j] ? even[i] : j;
                    }
                }
            }
        }
        for (int i = 0; i < A.length-1; i++){
            if (dfs(A, i, 1, odd, even)) res++;
        }
        return res;
    }
    
    public boolean dfs(int[] a, int idx, int cnt, int[] odd, int[] even){
        if (idx == a.length-1) return true;
        if (cnt % 2 == 0){
            if (even[idx] == -1) return false;
            return dfs(a, even[idx], cnt+1, odd, even);
        } else {
            if (odd[idx] == -1) return false;
            return dfs(a, odd[idx], cnt+1, odd, even);
        }
    }
}
