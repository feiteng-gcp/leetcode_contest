class Solution {
    int[][] next;
    int[][] dp;
    int n;
    
    public int oddEvenJumps(int[] A) {
        n = A.length;
        TreeMap<Integer, Integer> m = new TreeMap<>();
        next = new int[n][2];
        dp = new int[n][2];
        
        for(int i = A.length - 1; i >= 0; i--){
            int cur = A[i];
            Integer oddKey = m.ceilingKey(cur);
            if(oddKey != null){
                next[i][0] = m.get(oddKey);
            }
            else{
                next[i][0] = -1;
            }
            
            Integer evenKey = m.floorKey(cur);
            if(evenKey != null){
                next[i][1] = m.get(evenKey);
            }
            else{
                next[i][1] = -1;
            }
            
            m.put(cur, i);
        }
        
        //System.out.println(Arrays.toString(next[3]));
        
        int result = 0;
        for(int i = 0; i < n; i++){
            if(canReach(i, 0)){
                result++;
                //System.out.println(i);
            }
        }
        
        return result;
    }
    
    private boolean canReach(int cur, int isEven){
        if(cur == -1){
            return false;
        }
        if(cur == n - 1){
            return true;
        }
        
        if(dp[cur][isEven] == 0){
            if(canReach(next[cur][isEven], (isEven + 1) % 2)){
                dp[cur][isEven] = 1;
            }
            else{
                dp[cur][isEven] = -1;
            }
        }
        
        return dp[cur][isEven] == 1;
    }
}
