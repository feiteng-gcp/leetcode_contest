class Solution {
    Map<Integer, Boolean> odd = new HashMap<>();
    Map<Integer, Boolean> even = new HashMap<>();
    int[] A;
    
    public int oddEvenJumps(int[] A) {
        int ans = 0;
        this.A = A;

        for (int i = 0; i < A.length; i++) {
            if (jumpToEnd(i, true))
                ans++
;
        }
        return ans;
    }
    
    private boolean jumpToEnd(int i, boolean isOdd) {
        if (i == A.length - 1) return true;
        if (isOdd) {
            if (odd.containsKey(i))
                return odd.get(i);
            
int minVal = Integer.MAX_VALUE, minIdx = -1;
            for (int j = i + 1; j < A.length; j++) {
                if (A[i] <= A[j]) {
                    if (A[j] < minVal) {
                        minVal = A[j];
                        minIdx = j;
                    }
                }
            }
            if (minIdx == -1) {
                odd.put(i, false);
                return false;
            } else {
                boolean bool = jumpToEnd(minIdx, false);
                odd.put(i, bool);
                return bool;
            }
        } else {
            if (even.containsKey(i))
                return even.get(i);
            int maxVal = -1, maxIdx = -1;
            for (int j = i + 1; j < A.length; j++) {
                if (A[i] >= A[j]) 
{
                    if (A[j] > maxVal) {
                        maxVal = A[j];
                        maxIdx = j;
                    }
                }
            }
            if (maxIdx == -1) {
                even.put(i, false);
                return false;
            } else {
                boolean bool = jumpToEnd(maxIdx, true);
                even.put(i, bool);
                return bool;
            }
        }
        
    }
}
