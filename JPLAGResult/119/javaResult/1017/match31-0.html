class Solution {
    public int oddEvenJumps(int[] A) {
        int[] even = new int[A.length], odd = new int[A.length];
        TreeMap<Integer, Integer> map = new TreeMap<>();
        map.put(A[A.length-1], A.length-1);
        for (int i = A.length-2; i >= 0; i--) {
            Integer key = map.ceilingKey(A[i]);
            if (key == null) {
                odd[i] = -1;
            } else {
                odd[i] = map.get(key);
            }
            map.put(A[i], i);
        }
        map.clear();
        map.put(A[A.length-1], A.length-1);
        for (int i = A.length-2; i >= 0; i--) {
            Integer key = map.floorKey(A[i]);
            if (key == null) {
                even[i] = -1;
            } else {
                even[i] = map.get(key);
            }
            map.put(A[i], i);
        }
        boolean[] canEven = new boolean[A.length], canOdd = new boolean[A.length];
        canEven[A.length-1] = canOdd[A.length-1] = true;
        for (int i = A.length-2; i >= 0; i--) {
            if (even[i] != -1 && canOdd[even[i]])
                canEven[i] = true;
            if (odd[i] != -1 && canEven[odd[i]])
                canOdd[i] = true;
        }
        int res = 0;
        for (int i = 0; i < canOdd.length; i++)
            if (canOdd[i])
                res++;
        return res;
    }
}
