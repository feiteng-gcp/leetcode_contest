class Solution {
    public int oddEvenJumps(int[] A) {
        int n = A.length;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        boolean[][] flags = new boolean[n][2];
        flags[n - 1][0] = true;
        flags[n - 1][1] = true;
        map.put(A[n - 1], n - 1);
        int res = 1;

        for (int i = n - 2; i >= 0; i--) {
            // Odd step
            Integer nextGreater = map.ceilingKey(A[i]);
            if (nextGreater != null) {
                flags[i][0] = flags[map.get(nextGreater)][1];
            }
            // Even step
            Integer nextSmaller = map.floorKey(A[i]);
            if (nextSmaller != null) {
                flags[i][1] = flags[map.get(nextSmaller)][0];
            }
            map.put(A[i], i);

            res += flags[i][0] ? 1 : 0;
        }

        return res;
    }
}
