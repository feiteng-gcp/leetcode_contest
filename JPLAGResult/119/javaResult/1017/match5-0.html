class Solution {
    public int oddEvenJumps(int[] A) {
        if (A.length == 0) {
            return 1;
        }
        int n = A.length;
        int[] nextGreatIndex = new int[n];
        int[] nextSmallerIndex = new int[n];
        TreeMap<Integer, Integer> map = new TreeMap<>();
        map.put(A[n - 1], n - 1);
        for (int i = n - 2; i >= 0; i--) {
            Map.Entry<Integer, Integer> nextGreat = map.ceilingEntry(A[i]);
            if (nextGreat == null) {
                nextGreatIndex[i] = -1;
            } else {
                nextGreatIndex[i] = nextGreat.getValue();
            }
            
            Map.Entry<Integer, Integer> nextSmaller = map.floorEntry(A[i]);
            if (nextSmaller == null) {
                nextSmallerIndex[i] = -1;
            } else {
                nextSmallerIndex[i] = nextSmaller.getValue();
            }
            
            map.put(A[i], i);
        }
        
        int res = 1;
        for (int start = 0; start < n - 1; start++) {
            int cnt = 1;
            int i = start;
            while (i < n - 1 && i >= 0) {
                if (cnt % 2 == 1) {
                    i = nextGreatIndex[i];
                } else {
                    i = nextSmallerIndex[i];
                }
                cnt++;
            }
            if (i == n - 1) {
                res++;
            }
        }
        
        return res;
    }
}
