class Solution {
    
    Map<Integer, Boolean> oddCache = new HashMap<>();
    Map<Integer, Boolean> evenCache = new HashMap<>();
    public int oddEvenJumps(int[] A) {
        int result = 0;
        for(int i = 0 ; i < A.length; i ++) {
            if(odd(A,i)){
                result++;
            }
        }
        return result;
    }
    
    public boolean odd(int[] A, int startIndex) {
        
        if(startIndex == A.length - 1) {
            return true;
        }
        
        if(oddCache.containsKey(startIndex)) {
            return oddCache.get(startIndex);
        }
        
        int minNumGreater = Integer.MAX_VALUE;
        int nextIndexFound = -1;
        for(int i = startIndex + 1; i < A.length ; i++) {
            if(A[i] >= A[startIndex]) {
                if(minNumGreater > A[i]) {
                    minNumGreater = A[i];
                    nextIndexFound = i;
                }
            }
        }
        
        if(nextIndexFound == -1) {
            oddCache.put(startIndex, false);
            return false;
        }
        else {
            boolean result = even(A,nextIndexFound);
            oddCache.put(startIndex,result);
            return result;
        }
    }
    
    public boolean even(int[] A, int startIndex) {
        if(startIndex == A.length - 1) {
            return true;
        }
        
        if(evenCache.containsKey(startIndex)) {
            return evenCache.get(startIndex);
        }
        
        int maxNumSmaller = Integer.MIN_VALUE;
        int nextIndexFound = -1;
        for(int i = startIndex + 1; i < A.length ; i++) {
            if(A[i] <=  A[startIndex]) {
                if(maxNumSmaller < A[i]) {
                    maxNumSmaller = A[i];
                    nextIndexFound = i;
                }
            }
        }
        
        if(nextIndexFound == -1) {
            evenCache.put(startIndex, false);
            return false;
        }
        else {
            boolean result = odd(A,nextIndexFound);
            evenCache.put(startIndex,result);
            return result;
        }
    }
}
