class Solution {
    public int oddEvenJumps(int[] A) {
        int[] nextOddJumpIndex=new int[A.length];
        int[] nextEvenJumpIndex=new int[A.length];
        makeOddArray(nextOddJumpIndex,A);
        makeEvenArray(nextEvenJumpIndex,A);
        boolean[][] res=new boolean[A.length][2];
        res[A.length-1][0]=true;
        res[A.length-1][1]=true;
        for(int i=A.length-2;i>=0;i--){
            int oddNext=nextOddJumpIndex[i];
            int evenNext=nextEvenJumpIndex[i];
            if(oddNext!=-1&&res[oddNext][1]){
                res[i][0]=true;
            }
            if(evenNext!=-1&&res[evenNext][0]){
                res[i][1]=true;
            }
        }
        
        int count=0;
        for(int i=0;i<res.length;i++){
            if(res[i][0]){
                count++;
            }
        }
        return count;
        
    }
    public void makeOddArray(int[] nextOddJumpIndex,int[] A){
        TreeMap<Integer,Integer> ts=new TreeMap<Integer,Integer>();
        for(int i=A.length-1;i>=0;i--){
            int curr=A[i];
            Integer ceil=ts.ceilingKey(curr);
            if(ceil==null){
                nextOddJumpIndex[i]=-1;
            }
            else{
                nextOddJumpIndex[i]=ts.get(ceil);
            }
            ts.put(curr,i);
        }
    }
    public void makeEvenArray(int[] nextEvenJumpIndex,int[] A){
        TreeMap<Integer,Integer> ts=new TreeMap<Integer,Integer>();
        for(int i=A.length-1;i>=0;i--){
            int curr=A[i];
            Integer floor=ts.floorKey(curr);
            if(floor==null){
                nextEvenJumpIndex[i]=-1;
            }
            else{
                nextEvenJumpIndex[i]=ts.get(floor);
            }
            ts.put(curr,i);
        }
    }
}
