class Solution {
    public int oddEvenJumps(int[] A) {
        TreeSet<int[]> set = new TreeSet<>((a, b) -> (a[0] - b[0]));

        int n = A.length;

        int[] odd = new int[n];
        int[] even = new int[n];
        for (int i = n - 1; i >= 0; i--) {
            odd[i] = even[i] = -1;
            int[] node = new int[] {A[i], i};
            // odd
            int[] ceiling = set.ceiling(node);
            if (ceiling != null) {
                odd[i] = ceiling[1];
            }

            // even
            int[] floor = set.floor(node);
            if (floor != null) {
                even[i] = floor[1];
            }

            if (set.contains(node)) {
                
set.remove(node);
                
            }
            set.add(node);
        }

        boolean[] oddPass = new boolean[n];
        boolean[] evenPass = new boolean[n];

        
oddPass[n - 1] = evenPass[n - 1] = true;
        for (int i = n - 2; i >= 0; i--) {
            if (odd[i] != -1 && evenPass[odd[i]]) {
                oddPass[i] = true;
            }

            if (even[i] != -1 && oddPass[even[i]]) {
                evenPass[i] = true;
            }
        }

        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (oddPass[i]) {
                ans++;
            }
        }
        return ans;
    }

}
