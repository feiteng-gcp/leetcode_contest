class Solution {
    public List<String> braceExpansionII(String expression) {
        Map<Integer, Integer> pairs = new HashMap<>();
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < expression.length(); i++) 
{
            char c = expression.charAt(i);
            if (c == '{') {
                st.push(i);
            } else if (c == '}') {
                int top = st.pop();
                pairs.put(top, i);
            }
        }
        List<String> l = evaluate(expression, 0, expression.length() - 1, pairs);
        Set<String> s = new HashSet<>(l);
        l = new LinkedList<>();
        for (String x : s) {
            l.add(x);
        }
        Collections.sort(l);
        return l;
    }

    public List<String> evaluate(String expression, int l, int r, Map<Integer, Integer> pairs) {
        ArrayList<List<String>> list = new ArrayList<>();
        int i = l;
        while (i <= r) {
            List<String> cur = new LinkedList<>();
            
char c = expression.charAt(i);
            if (c == ',') {
                cur.add(",");
            } else if (c == '{') {
                cur = evaluate(expression, i + 1, pairs.get(i) - 1, pairs);
                // System.out.println("/");
                // cur.forEach(x -> System.out.println("cur -> " + x));
                i = pairs.get(i);
            } else {
                String s = "";
                while (i <= r) {
                    if (Character.isLowerCase(expression.charAt(i))) {
                        s += expression.charAt(i);
                    } else {
                        i--;
                        break;
                    }
                    i++;
                }
                cur.add(s);
            }
            list.add(new LinkedList<>(cur));
            i++;
        }

        return union(list);
    }

    public List<String> union(ArrayList<List<String>> list) {
        List<String> res = new LinkedList<>();
        int i = 0;
        // for (int j = 0; j < list.size(); j++) {
        // System.out.println(j);
        // list.get(j).forEach(x -> System.out.println(x));
        // }
        while (i < list.size()) {
            List<String> temp = new LinkedList<>(list.get(i));
            i++;
            while (i < list.size()) {
                List<String> cur = list.get(i);
                if (cur.get(0).equals(",")) {
                    i++;
                    break;
                } else {
                    List<String> newList = new LinkedList<>();
                    for (String a : temp) {
                        for (String b : cur) {
                            // System.out.println("a " + a + " , " + b);
                            newList.add(a + b);
                        }
                    }
                    // System.out.println("newList");
                    // newList.forEach(x -> System.out.println(x));
                    // System.out.println("newList->");
                    temp = newList;
                }
                i++;
            }
            res.addAll(temp);
        }
        return res;
    }
}
