class Solution {
    public List<String> braceExpansionII(String expression) {
        List<String> ans = unsortedProcess(expression);
        Collections.sort(ans);
        return ans;
    }

    public List<String> unsortedProcess(String expression) {
        if (expression.length() == 0){
            return new ArrayList<>();
        }
        Stack<List<String>> tWords = new Stack<>();
        int i = 0;
        while (i < expression.length()) {
            char c = expression.charAt(i);
            if (c == ',') {
                i++;
                continue;
            }
            int right;
            List<String> words;
            if (c == '{') {
                right = findMatchClose(expression, i);
                words = unsortedProcess(expression.substring(i+1, right));
            } else if (c >= 'a' && c <= 'z'){
                right = findLetterEnds(expression, i);
                words = new ArrayList<>();
                words.add(expression.substring(i, right+1));
            } else {
                throw new RuntimeException(String.format("Error parsing: %s at index: %d", expression, i));
            }
            if (i == 0 || (expression.charAt(i-1) == ',')){
                tWords.push(words);
            } else {
                List<String> s1 = tWords.pop();
                tWords.push(prod(s1, words));
            }
            i = right+1;
        }
         while (tWords.size() > 1){
                List<String> s1 = tWords.pop();
                List<String> s2 = tWords.pop();
                tWords.push(union(s1, s2));
        }
        //System.out.println("tWords:" + tWords);
        return tWords.pop();
    }
    
    int findLetterEnds(String str, int index){
        while (index < str.length() && str.charAt(index) >= 'a' && str.charAt(index) <= 'z') {
            index++;
        }
        index--;
        return index;
    }
    
    
    int findMatchClose(String str, int index){
        int left = 0;
        while (index < str.length()) {
            if (str.charAt(index) == '{') {
                left ++;
            } else if (str.charAt(index) == '}') {
                left --;
                if (left == 0){
                    return index;
                }
            }
            // System.out.println("left:" + left);
            index++;
        }
        return str.length()-1;
    }
    
    List<String> prod(List<String> l1, List<String> l2){
        Set<String> set = new HashSet<>();
        for (String str1 : l1)
            for (String str2 : l2){
                set.add(str1+str2);
            }
        return new ArrayList<>(set);
    }
    
    
    List<String> union(List<String> l1, List<String> l2){
        Set<String> set = new HashSet<>();
        set.addAll(l1);
        set.addAll(l2);
        return new ArrayList<>(set);
    }
}
