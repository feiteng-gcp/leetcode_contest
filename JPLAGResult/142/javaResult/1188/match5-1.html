class Solution {
    
    int index = 0;
    public List<String> braceExpansionII(String expression) {
        List<String> list = new ArrayList(help(expression));
        Collections.sort(list);
        return list;
    }

    Set<String> help(String expression) {
        Set<String> allSet = new HashSet<>();
        Set<String> curSet = new HashSet<>();
        StringBuilder sb = new StringBuilder();
        while(index < expression.length()) {
            if(expression.charAt(index) == '{'){
                index++;
                Set<String> set = help(expression);
                if(set.isEmpty()) continue;
                if(!curSet.isEmpty()){
                    Set<String> tmp = new HashSet<>();
                    String mid = sb.toString();
                    sb = 
new StringBuilder();
                    for(String pre : curSet) {
                        for(String suf : set) {
                            tmp.add(pre + mid + suf);
                        }
                    }
                    curSet = tmp;
                }else {
                    String pre = sb.toString();
                    sb = new StringBuilder();
                    for(String suf : set) curSet.add(pre + suf);
                }
            }else if(expression.charAt(index) >='a' && expression.charAt(index) <='z') {
                sb.append(expression.charAt(index));
                index++;
            }else if(expression.charAt(index) == ',' || expression.charAt(index) == '}') {
                if(curSet.isEmpty()) {
                    allSet.add(sb.toString());
                    sb = new StringBuilder();
                }else {
                    if(sb.length() == 0) {
                        allSet.addAll(curSet);
                        curSet.clear();
                    }else {
                        String str = sb.toString();
                        
sb = new StringBuilder();
                        for(String s : curSet) {
                            allSet.add(s+str);
                        }
                        curSet.clear();
                    }
                }
                if(expression.charAt(index) == '}') {
                    index++;
                    break;
                }
                index++;
            }
        }

        allSet.addAll(curSet);
        if(sb.length() > 0) {
            Set<String> result = new HashSet<>();
            if(allSet.isEmpty()) result.add(sb.toString());
            else {
                String suf = sb.toString();
                for(String pre: allSet) result.add(pre + suf);
            }
            return result;
        }
        return allSet;
    }
}
