class Solution {
    public List<String> braceExpansionII(String expression) {
        List<String> ans =braceExpansionIIDfs(expression);
        Collections.sort(ans);
        return ans;
    }


    public List<String> braceExpansionIIDfs(String expression) {
        //System.out.println("exp = " + expression);
        int n = expression.length();
        List<String> list = new ArrayList<>();
        if (n == 1) {
            list.add(expression);
            return list;
        }



        int[] match = new int[n];
        Arrays.fill(match, -1);
        Stack<Integer> stack =new Stack<>();
        for (int i = 0; i < n; i++) {
            char ch = expression.charAt(i);
            if (ch == '{') {
                stack.push(i);
            } else if (ch == '}'){
                match[i] = stack.pop();
            }
        }


        if (match[n - 1] == 0) {
            //System.out.println("match exp == " + expression);
            Set<String> set = new HashSet<>();

            int last = 0;
            int lCount = 0;
           // stack = new Stack<>();
            for (int i = 1; i < n - 1; i++) {
                char ch = expression.charAt(i);
                if (ch == '{') {
                    lCount++;
                } else if (ch == '}'){
                    lCount--;
                } else if (ch == ',') {
                    if (lCount == 0) {
                        List<String> t = braceExpansionIIDfs(expression.substring(last + 1, i));

                        set.addAll(t);

                        last = i;
                    }
                }
            }

            set.addAll(braceExpansionIIDfs(expression.substring(last + 1, n - 1)));
            list.addAll(set);
        } else {
            List<int[]> pair = new ArrayList<>();
            stack =new Stack<>();
            for (int i = 0; i < n; i++) {
                char ch = expression.charAt(i);
                if (ch == '{') {
                    stack.push(i);
                } else if (ch == '}'){
                    int l = stack.pop();
                    if (stack.isEmpty()) {
                        // [l, r]
                        pair.add(new int[] {l, i});
                    }
                }
            }

            //list.add("");
            for (int i = 0; i < pair.size(); i++) {
                int[] p = pair.get(i);
                int l = p[0];
                int r = p[1];

                if (i == 0) {
                    if (l > 0) {
                        List<String> t = braceExpansionIIDfs(expression.substring(0, l));
                        list = combine(list, t);
                    }
                } else {
                    if (pair.get(i - 1)[1] + 1 < l) {
                        List<String> t = braceExpansionIIDfs(expression.substring(pair.get(i - 1)[1] + 1, l));
                        list = combine(list, t);
                    }


                }

                List<String> t2 = braceExpansionIIDfs(expression.substring(l, r + 1));
                list = combine(list, t2);
            }

            if (pair.size() == 0) {
                list.add(expression);
            } else if (pair.size() > 0 && pair.get(pair.size() - 1)[1] + 1 < n) {
                List<String> t = braceExpansionIIDfs(expression.substring(pair.get(pair.size() - 1)[1] + 1));
                list = combine(list, t);
            }

        }
        //System.out.println("ret == exp = " + expression + " list = " + list);
        return list;


    }


    List<String> combine(List<String> a, List<String> b) {
        List<String> ret = new ArrayList<>();
        if (a.isEmpty()) {
            ret.addAll(b);
        } else if (b.isEmpty()) {
            ret.addAll(a);
        } else {
            for (int i = 0; i < a.size(); i++) {
                for (int j = 0; j < b.size(); j++) {
                    ret.add(a.get(i) + b.get(j));
                }
            }
            Set<String> set = new HashSet<>(ret);
            ret = new ArrayList<>(set);
        }

        return ret;
    }
}
