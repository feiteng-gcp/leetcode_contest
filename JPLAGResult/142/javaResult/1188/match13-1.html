class Solution {

	private void build(String expression) {
		LinkedList<Integer> stack = new LinkedList<>();
		int i , length = expression.length();
		for (i = 0;i < length;i ++) {
			if (expression.charAt
(i) == '{') {
				stack.push(i);				
			} else if (expression.charAt(i) == '}') {				
				backetRight[stack.pop()] = i;				
			}
		}
	}
	
	private int[] backetRight = new int[100];	
	
    public List<String> braceExpansionII(String expression) {
    	build(expression);    	
    	Set<String> set = new HashSet<>();
    	backtracking(expression , 0 , 0 , expression.length() - 1 , new StringBuilder() , set);
    	List<String> list = new ArrayList<>(set);
    	Collections.sort(list);
    	return list;    	
    }
    
    private List<int[]> split(String expression , int from , int to) {    	
    	// split {}
    	int prev = from + 1 , cnt = 0;
    	List<int[]> list = new ArrayList<>();
    	for (int i = from;i <= to;i ++) {
    		if (expression.charAt(i) == '{') {
    			cnt ++;
    		} else if (expression.charAt(i) == '}') {
    			cnt --;    			
    		}
    		if (expression.charAt(i) == ',' && cnt == 1) {
    			list.add(new int[] {prev , i - 1});
    			prev = i + 1;    		
    		}
    	}
    	list.add(new int[] {prev , to - 1});    	
    	return list;
    }
    
    private void backtracking(String expression , int current , int from , int to , StringBuilder builder , Set<String> set) {
    	if (current == to + 1) {
    		set.add(builder.toString());    		
    	} else {
    		// left bracket    		
    		if (expression.charAt(current) == '{') {
    			int right = backetRight[current];
    			// current -> right {}
    			List<int[]> list = split(expression , current , right);    			    		
    			Set<String> s = new HashSet<>();    			
    			for (int[] interval : list) {
    				Set<String> temp = new HashSet<>();
    				backtracking(expression , interval[0] , interval[0] , interval[1] , new StringBuilder() , temp);	
    				s.addAll(temp);
    			}
    			for (String string : s) {
    				StringBuilder newBuilder = new StringBuilder(builder.toString());
    				newBuilder.append(string);
    				backtracking(expression , right + 1 , from , to , newBuilder , set);    				
    			}
    		} else {    		
    			builder.append(expression.charAt(current));
    			backtracking(expression , current + 1 , from , to , builder , set);
    			builder.deleteCharAt(builder.length() - 1);    			
    		}
    	}    	
    }
	
}
