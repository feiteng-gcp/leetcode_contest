class Solution {
    Stack<Integer> st = new Stack<>();
    int d[];
    public List<String> braceExpansionII(String s) {
        char cs[] = s.toCharArray();
        int len  = cs.length;
        d = new int[len];
        
        for(int i=0;i<len;++i){
            if(cs[i]=='{'){
                st.push(i);
            }else if(cs[i]=='}'){
                d[st.pop()] = i;
            }
        }
        
        
        List<String> res = p(cs,0,len-1);
        HashSet<String> f = new HashSet<>(res);
        
        res = new ArrayList<>();
        
        for(String y:f){
            res.add(y);
        }
        
Collections.sort(res);
        return res;
        
    }
    
    public List<String> p(char[] cs,int b , int e){
        List<String> all = new ArrayList<>();
        List<String> tp = null;
        for(int i=b;i<=e;){
            if(cs[i]=='{'){
                
               
                
               List<String> intp = p(cs, i+1, d[i]-1);
               if(tp!=null){
                   List<String> ntp = new ArrayList<>();
                   for(String cur:tp){
                       for(String next:intp){
                           ntp.add(cur+next);
                       }
                   }
                   tp = ntp;
               }else{
                   tp = intp;
               } 
                
               i = d[i]+1;
            }else if(cs[i]>='a'&&cs[i]<='z'){
                
                List<String> ntp = new ArrayList<>();
                if(tp!=null){
                   for(String cur:tp){
                       ntp.add(cur+cs[i]);
                   }
                }else{
                    ntp.add(""+cs[i]);
                }
                tp = ntp;
                ++i;
            }else if(cs[i]==','){
                
                all.addAll(tp);
                tp = null;
                ++i;
            }else{
                ++i;
            }
            
        }
        if(tp!=null){
            
            all.addAll(tp);
        }
        return all;
        
        
    }
    
}
