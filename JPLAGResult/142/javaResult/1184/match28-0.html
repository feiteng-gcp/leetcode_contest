class Solution {
    public boolean carPooling(int[][] trips, int capacity) {
        int len = 1001;
        int[] count = new int[len];
        for (int i = 0; i < trips.length; i++) {
            int num = trips[i][0];
            int l = trips[i][1]; 
            int r = trips[i][2]; 
            count[l] += num;
            count[r] -= num;
        }
        
        int curr = 0;
        for (int i = 0; i < len; i++) {
            curr += count[i];
            if (curr > capacity) {
                return false;
            }
        }
        return true;
    }
}
