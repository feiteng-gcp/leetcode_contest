class Solution {
    public boolean carPooling(int[][] trips, int capacity) {
        int[] s = new int[1001];
        for (int[] t : trips){
            s[t[1]] += t[0];
            s[t[2]] -= t[0];
        
}
        int sum = 0;
        for (int i = 0; i<=1000; i++){
            sum += s[i];
            if (sum>capacity) return false;
        }
        return true;
    }
}
