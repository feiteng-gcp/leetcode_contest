	class S
olution {
	    public boolean carPooling(int[][] trips, int capacity) {
	    	int[] imos = new int[1005];
	        for(int[] t : trips){
	        	imos[t[1]] += t[0];
	        	imos[t[2]] -= t[0];
	        }
	        for(int i = 0;i < 1001;i++){
	        	imos[i+1] += imos[i];
	        }

	        for(int i = 0;i < 1001;i++){
	        	if(imos[i] > capacity){
	        		return false;
	        	}
	        }
	        return true;
	    }
	}
