class Solution {
    public boolean carPooling(int[][] trips, int capacity) {
        int[] stuff = new int[1001];
        for(int[] arr : trips)
        {
            for(int i = arr[1]; i<arr[2]; i++)
            {
                stuff[i] += arr[0];
                if(stuff[i] > capacity)
                {
                    return false;
                
}
            }
        }
        return true;
    }
}
