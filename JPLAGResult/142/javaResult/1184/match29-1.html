class Solution {
    static class Node {
        boolean isStart;
        int x;
        int n;
        public Node(boolean isStart, int x, int n) {
            this.isStart = isStart;
            this.x = x;
            this.n = n;
        }
    }
    
    static class MyComparator implements Comparator<Node> {
        public int compare(Node a, Node b) {
            if (a.x != b.x) {
                return a.x - b.x;
            }
            return a.isStart ? 1 : -1;
        }
    }
    public boolean carPooling(int[][] trips, int cap) {
        if (trips == null || trips.length == 0) {
            return true;
        }
        
List<Node> nodes = new ArrayList();
        for (int[] trip : trips) {
            nodes.add(new Node(true, trip[1], trip[0]));
            nodes.add(new Node(false, trip[2], trip[0]));
        }

        nodes.sort(new MyComparator());
        
        
long cur = 0;
        for (Node curnode : nodes) {
            if (curnode.isStart) {
                cur += curnode.n;
                if (cur > cap) {
                    return false;
                }
            } else {
                cur -= curnode.n;
            }
        }
        return true;
    }
}
