class Solution {
    static class Event {
        int pos;
        int people;
        boolean up;

        Event(int pos, int people, boolean up) {
            this.pos = pos;
            this.people = people;
            this.up = up;
        }
    }

    public boolean carPooling(int[][] trips, int capacity) {
        List<Event> events = new ArrayList<>();

        for(int[] trip : trips) {
            events.add(new Event(trip[1], trip[0], true));
            events.add(new Event(trip[2], trip[0], false));
        }
        
        events.sort((a, b) -> a.pos == b.pos ? (a.up ? 1: -1) : a.pos - b.pos);
        
        int count = 0;
        
        for(Event event : events) {
            if(event.up) {
                count += event.people;
                
                if(count > capacity) {
                    return false;
                }
            } else {
                count -= event.people;
            }
        }
        
        return true;
    }
}
