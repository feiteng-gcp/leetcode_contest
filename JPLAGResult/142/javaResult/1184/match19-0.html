class Solution {
    public boolean carPooling(int[][] trips, int capacity) {
        int[] num = new int[1005];

        for (int[] trip : trips) {
            for (int i = trip[1]; i< trip[2]; i++) {
                num[i] += trip[0];
            }
        }
        
        for (int n : num) {
            if (n > capacity) {
                return false;
            }
        }
        return true;
    }

}
