class Solution {
    public boolean carPooling(int[][] trips, int capacity) {
        int[] cnt = new int[1001];
        for (int[] t: trips) {
            cnt[t[1]] += t[0];
            cnt[t[2]] -= t[0];
        }
        
        long sum = 0l;
        for (int i = 0; i <= 1000; ++i) {
            sum += cnt[i];
            if (sum > capacity) {
                return false;
            }
        }
        return true;
    }
}
