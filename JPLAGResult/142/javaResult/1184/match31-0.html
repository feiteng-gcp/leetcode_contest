class Solution {
    public boolean carPooling(int[][] trips, int capacity) {
        int[] num = new int[1001];
        
        for (int i = 0; i <= 1000; i++) {
            for (int j = 0; j < trips.length; j++) {
                if (trips[j][1] <= i && trips[j][2] > i) {
                    num[i] += trips[j][0];
                }
            }
        }
        
        for (int cb : num) {
            if (cb > capacity) {
                return false;
            }
        }
        
        return true;
    }
}
