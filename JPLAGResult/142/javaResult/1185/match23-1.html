/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 

class Solution {
    public int findInMountainArray(int target, MountainArray mountainArr) {
        int len = mountainArr.length();
        int left = 0, right = len - 2;
        while (left < right) {
            int mid = (left + right) / 2;
            if (mountainArr.get(mid) > mountainArr.get(mid + 1))
                right = mid;
            else
                left = mid + 1;
        }
        int middle = left;
        // int midVal = mountainArr.get(middle);
        // if (target == midVal)
            
        
left = 0;
        right = middle;
        while (left <= right) {
            int mid = (left + right) / 2;
            int v = mountainArr.get(mid);
            if (v == target)
                return mid;
            else if (v > target)
                right = mid - 1;
            else
                left = mid + 1;
        }
        
left = middle;
        right = len-1;
        while (left <= right) {
            int mid = (left + right) / 2;
            int v = mountainArr.get(mid);
            if (v == target)
                return mid;
            else if (v < target)
                right = mid - 1;
            else
                left = mid + 1;
        }
        return -1;
    }
}
