    private int binarySearchLeft(MountainArray ma, int s, int e, int target) {
    	while(s < e) {
    		int mid = (e + s) / 2;
    		int g = get(ma, mid);
    		if (g == target) {
    			return mid;
    		}else if (g > target) {
    			e = mid - 1;
    		}else {
    			s = mid + 1;
    		}
    	}
    	return s;
    }
    
    private int binarySearchRight(MountainArray ma, int s, int e, int target) {
    	while(s < e) {
    		int mid = (e + s) / 2;
    		int g = get(ma, mid);
    		if (g == target) {
    			return mid;
    		}else if (g > target) {
    			s = mid + 1;
    		}else {
    			e = mid - 1;
    		}
    	}
    	return s;
    }
    
    private int get(MountainArray ma, int pos) {
    	if (mem.containsKey(pos)) {
    		return mem.get(pos);
    	}else {
    		mem.put(pos, ma.get(pos));
    		return mem.get(pos);
    	}
    }
