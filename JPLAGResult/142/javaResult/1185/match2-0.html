/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
        public int findInMountainArray(int target, MountainArray mountainArr) {
        int len = mountainArr.length();
        int lo=0,hi = len-1;
        while(lo<hi){
            int mi = lo+(hi-lo)/2;
            if(mountainArr.get(mi)<mountainArr.get(mi+1))
                lo=mi+1;
            else
                hi=mi;
        }
        int pv = lo;
        lo=0;
        hi=pv;
        while(lo<hi){
            int mi = lo+(hi-lo)/2;
            int vmi = mountainArr.get(mi);
            if(vmi==target)
                return mi;
            if(vmi<target)
                lo=mi+1;
            else
                hi=mi;
        }
        if(mountainArr.get(lo)==target)
            return lo;
        lo=pv;
        hi=len-1;
        while(lo<hi){
            int mi = lo+(hi-lo)/2;
            int vmi = mountainArr.get(mi);
            if(vmi==target)
                return mi;
            if(vmi>target)
                lo=mi+1;
            else
                hi=mi;
        }
        if(mountainArr.get(lo)==target)
            return lo;
        return -1;
    }
}
