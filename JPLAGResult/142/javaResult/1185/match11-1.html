/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 

class Solution {
    public int findInMountainArray(int t, MountainArray m) {
        int peek = peakIndexInMountainArray(m);
        int p1 = search1(m, 0, peek, t);
        if (p1 != -1){
            return p1;
        }
        return search2(m, peek, m.length() - 1, t);
    
}
    private int search1(MountainArray m, int start, int end, int t){
        while (start <= end){
            int mid = start + (end - start) / 2;
            int v = m.get(mid);
            if (v == t){
                return mid;
            } else 
if (v < t){
                start = mid + 1;
            } else {
                end = mid - 1;
            
}
        }
        return -1;
    }
    private int search2(MountainArray m, int start, int end, int t){
        while (start <= end){
            int mid = start + (end - start) / 2;
            int v = m.get(mid);
            if (v == t){
                return mid;
            } else 
if (v < t){
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return -1;
    
}
    private int peakIndexInMountainArray(MountainArray m) {
        int lo = 0, hi = m.length() - 1;
        while (lo < hi) {
            int mi = lo + (hi - lo) / 2;
            if (m.get(mi) < m.get(mi + 1))
                lo = mi + 1;
            else
                hi = mi;
        }
        return lo;
    }
}
