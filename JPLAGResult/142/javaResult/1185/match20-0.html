/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray mr) {
        int peak = findpeak(mr);
        int find = search(target, mr, 0, peak, true);
        if(find > -1) return find;
        return search(target, mr, peak, mr.length(), false);
    }
    
    private int search(int target, MountainArray mr, int low, int high, boolean left){
        while(low < high){
            int mid = low + (high - low) / 2;
            int n = mr.get(mid);
            if(n == target) return mid;
            if(n > target){
                if(left){
                    high = mid;
                }else{
                    low = mid+1;
                }
            }else{
                if(left){
                    low = mid+1;
                }else{
                    high = mid;
                }
            }
        }
        return -1;
    }
    
    private int findpeak(MountainArray mr){
        int low = 0;
        int high = mr.length()-1;
        while(low < high){
            int mid = low + (high - low) / 2;
            if (mr.get(mid) < mr.get(mid + 1)){
                low = mid + 1;
            }else{
                high = mid;
            }
        }
        return low;
    }
}
