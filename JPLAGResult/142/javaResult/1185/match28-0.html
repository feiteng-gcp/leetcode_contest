/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray mountainArr) {
        MountainArray mt = mountainArr;
        int N = mt.length();
        //binary search to find peak
        int l = 0, r = N - 1;
        while (l < r) {
            int m = l + (r - l) / 2;
            int n1 = mt.get(m), n2 = mt.get(m + 1);
            if (n1 < n2) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        int p = l;
        
        // binary search to find if left exist
        l = 0; r = p;
        while (l <= r) {
            int m = l + (r - l) / 2;
            int n = mt.get(m);
            if (n == target) return m;
            if (n < target) {
                l = m + 1;
            } else {
                r = m - 1;
            }
        }
        
        
        // if not binary search to find if right exist
        l = p; 
        r = N - 1;
        while (l <= r) {
            int m = l + (r - l) / 2;
            int n = mt.get(m);
            if (n == target) return m;
            if (n > target) {
                l = m + 1;
            } else {
                r = m - 1;
            }
        }
        return -1;
    }
}
