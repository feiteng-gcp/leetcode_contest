/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 

class Solution {
    public int findInMountainArray(int target, MountainArray m) {
        int len = m.length();
        
        int lo = 1;
        int hi = len-1;
        
        while(lo<hi){
            int mi = (lo+hi+1)/2;
            
            int v2 = m.get(mi);
            int v1 = m.get(mi-1);
            
            
if(v1<v2){
                lo = mi;
            }else{
                hi = mi-1;
            }
            
            
        }
        
        int peek = lo;
        System.out.print(peek);
        
        int f = 0 ;
        int t = peek;
        
        
while(f<=t){
            int mi = (f+t)/2;
            
            int v = m.get(mi);
        //    System.out.println("try"+mi);
            if(v==target){
                return mi;
            }else if(v<target){
                f = mi+1;
            }else{
                t = mi-1;
            }
            
        }
        
        f = peek ;
        t = len-1;
        
        while(f<=t){
            int mi = (f+t+1)/2;
            
            int v = m.get(mi);
            
            if(v==target){
                return mi;
            }else if(v<target){
                t = mi-1;
            }else{
                f = mi+1;
            }
            
        }
        return -1;
        
        
        
        
        
        
        
    }
}
