/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {

   
    public int findInMountainArray(int target, MountainArray mountainArr) {
        int n = mountainArr.length();

        int first = mountainArr.get(0);
        if (first == target) return 0;

        
int l = 1;
        int r = n-1;

        while (l + 1 < r) {
            int m = (l + r) / 2;
            int v1 = mountainArr.get(m-1);
            int v2 = mountainArr.get(m);

            if (v1 > v2) {
                r = m;
            } else {
                l = m;
            }
        }
        int mid = l;
        int midv = mountainArr.get(mid);
        if (midv == target) return mid;

        l = 1;
        r = mid;

        int v = solve(1, mid, target, mountainArr);
        if (v != -1) return  v;
        v = solve2(mid+1, n, target, mountainArr);
        if (v != -1) return v;
        return -1;
    }

    int solve(int l, int r, int target, MountainArray mountainArr) {
        
while (l + 1 < r) {
            int m = (l + r) / 2;
            int v = mountainArr.get(m);
            if (v == target) return m;
            if (v < target) {
                l = m+1;
            } else {
                r = m;
            }
        }
        int v = mountainArr.get(l);
        if (v == target) return l;
        return -1;
    }

    int solve2(int l, int r, int target, MountainArray mountainArr) {
        
while (l + 1 < r) {
            int m = (l + r) / 2;
            int v = mountainArr.get(m);
            if (v == target) return m;
            if (v > target) {
                l = m+1;
            } else {
                r = m;
            }
        }
        int v = mountainArr.get(l);
        if (v == target) return l;
        return -1;
    }
}
