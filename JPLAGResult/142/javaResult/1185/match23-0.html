/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray mt) {
        int n = mt.length();
        int l = 0; 
        int r = n - 1;
        int mid = 0;
        while(l < r) {
            mid = l + (r - l) / 2;
            int next = mid + 1;
            int v1 = mt.get(mid);
            int v2 = mt.get(next);
            if(v2 > v1)
                l = mid + 1;
            else
                r = mid;
        }
        mid = l;
        l = 0;
        r = mid;
        while(l < r) {
            int m = l + (r - l) / 2;
            int v = mt.get(m);
            if(v == target)
                return m;
            else if(v > target)
                r = m - 1;
            else
                l = m + 1;
        }
        if(mt.get(l) == target)
            return l;
        l = mid + 1;
        r = n - 1;
        while(l < r) {
            int m = l + (r - l) / 2;
            int v = mt.get(m);
            if(v == target)
                return m;
            else if(v < target)
                r = m - 1;
            else
                l = m + 1;
        }
        if(mt.get(l) == target)
            return l;
        return -1;
    }
}
