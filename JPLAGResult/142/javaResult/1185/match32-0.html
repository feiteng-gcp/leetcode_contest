/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray mountainArr) {
        int length = mountainArr.length();
        int mid = this.findMid(mountainArr);
        int mv = mountainArr.get(mid);
        if (mv == target)
            return mid;
        int l = findLeft(mountainArr, mid, target);
        if (l != -1)
            return l;
        return findRight(mountainArr, mid, target);
    }

    int findLeft(MountainArray mountainArray, int mid, int target) {
        int l = 0, r = mid;
        while (l < r) {
            int m = (l + r) / 2;
            int mv = mountainArray.get(m);
            if (mv == target) {
                return m;
            } else if (mv > target) {
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        if (l <= mid && mountainArray.get(l) == target)
            return l;
        if (r >= 0 && mountainArray.get(r) == target)
            return r;
        return -1;
    }

    int findRight(MountainArray mountainArray, int mid, int target) {
        int length = mountainArray.length();
        int l = mid, r = length - 1;
        while (l < r) {
            int m = (l + r) / 2;
            int mv = mountainArray.get(m);
            if (mv == target) {
                return m;
            } else if (mv > target) {
                l = m + 1;
            } else {
                r = m - 1;
            }
        }
        if (l < length && mountainArray.get(l) == target)
            return l;
        if (r >= mid && mountainArray.get(r) == target)
            return r;
        return -1;
    }

    int findMid(MountainArray mountainArr) {
        int length = mountainArr.length();
        int l = 0;
        int r = length - 1;
        while (true) {
            int mid = (r + l) / 2;
            int lv = mountainArr.get(mid - 1);
            int mv = mountainArr.get(mid);
            int rv = mountainArr.get(mid + 1);
            if (lv < mv && mv < rv) {
                l = mid + 1;
            } else if (lv < mv && mv > rv) {
                return mid;
            } else if (lv > mv && mv > rv) {
                r = mid;
            }
        }
    }
}
