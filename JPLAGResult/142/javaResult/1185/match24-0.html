/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray mountainArr) {
        int length = mountainArr.length();
        int peakIndex = peakIndexInMountainArray(mountainArr, length);
        int ascendingIndex = binarySearchAscending(mountainArr, target, 0, peakIndex);
        if (ascendingIndex >= 0)
            return ascendingIndex;
        int descendingIndex = binarySearchDescending(mountainArr, target, peakIndex, length - 1);
        return descendingIndex;
    }

    private int peakIndexInMountainArray(MountainArray mountainArr, int length) {
        int low = 0, high = length - 1;
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (mountainArr.get(mid) < mountainArr.get(mid + 1))
                low = mid + 1;
            else
                high = mid;
        }
        return low;
    }

    private int binarySearchAscending(MountainArray mountainArr, int target, int beginIndex, int endIndex) {
        int low = beginIndex, high = endIndex;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            int midElement = mountainArr.get(mid);
            if (midElement == target)
                return mid;
            else if (midElement < target)
                low = mid + 1;
            else
                high = mid - 1;
        }
        return -1;
    }

    private int binarySearchDescending(MountainArray mountainArr, int target, int beginIndex, int endIndex) {
        int low = beginIndex, high = endIndex;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            int midElement = mountainArr.get(mid);
            if (midElement == target)
                return mid;
            else if (midElement > target)
                low = mid + 1;
            else
                high = mid - 1;
        }
        return -1;
    }
}
