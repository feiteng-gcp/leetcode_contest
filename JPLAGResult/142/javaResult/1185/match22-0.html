/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray A) {
        int lo = 0, hi = A.length() - 1;
        while (lo < hi) {
            int mi = lo + (hi - lo) / 2;
            if (A.get(mi) < A.get(mi + 1))
                lo = mi + 1;
            else
                hi = mi;
        }
        
        int start = 0;
        int end = lo;
        while(start + 1 < end){
            int mid = start + (end - start) / 2;
            int cand = A.get(mid);
            if (cand == target) {
                return mid;
            } else if (cand < target) {
                start = mid;
            } else {
                end = mid;
            }
        }
        
        if (A.get(start) == target) {
            return start;
        }
        if (A.get(end) == target) {
            return end;
        }
        
        start = lo;
        end = A.length() - 1;
        while(start + 1 < end){
            int mid = start + (end - start) / 2;
            int cand = A.get(mid);
            if (cand == target) {
                return mid;
            } else if (cand < target) {
                end = mid;
            } else {
                start = mid;
            }
        }
        
        if (A.get(start) == target) {
            return start;
        }
        if (A.get(end) == target) {
            return end;
        }
        return -1;
    }
}
