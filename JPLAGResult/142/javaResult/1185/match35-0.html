/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray mountainArr) {
        int pivot = findPivot(mountainArr);
        int idx = binarySearch(mountainArr, 0, pivot, target);
        if(idx != -1) {
            return idx;
        }
        
        return reverseBinarySearch(mountainArr, pivot, mountainArr.length() - 1, target);
    }
    
    private int findPivot(MountainArray mountainArr) {
        int left = 1;
        int right = mountainArr.length() - 2;
        while(left <= right) {
            int mid = left + (right - left) / 2;
            int midVal = mountainArr.get(mid);
            int midLeftVal = mountainArr.get(mid - 1);
            int midRightVal = mountainArr.get(mid + 1);
            if(midVal > midLeftVal && midVal > midRightVal) {
                return mid;
            } else if(midVal > midLeftVal){
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return left;
    }
    
    private int binarySearch(MountainArray ma, int left, int right, int target) {
        while(left <= right) {
            int mid = left + (right - left) / 2;
            int midVal = ma.get(mid);
            if(midVal == target) {
                return mid;
            } else if(midVal < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return -1;
    }
    
    private int reverseBinarySearch(MountainArray ma, int left, int right, int target) {
        while(left <= right) {
            int mid = left + (right - left) / 2;
            int midVal = ma.get(mid);
            if(midVal == target) {
                return mid;
            } else if(midVal < target) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        return -1;
    }
}
