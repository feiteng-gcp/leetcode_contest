/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray arr) {
        int len = arr.length();
        int start = 1;
        int end = len - 2;
        while (start + 1  < end) {
            int mid = start + (end - start) / 2;
            int v0 = arr.get(mid);
            int v1 = arr.get(mid + 1);
            if (v0 > v1) {
                end = mid;
            } else {
                start = mid;
            }
        }
        int pivot = end;
        int vpivot = -1;
        int v0 = arr.get(start);
        int v1 = arr.get(start + 1);
        if (arr.get(start) > arr.get(start + 1)) {
            pivot = start;
            vpivot = v0;
        } else {
            vpivot = v1;
        }
        if (vpivot == target) {
            return pivot;
        }
        
        int ret = find(target, arr, 0, pivot - 1);
        if (ret != -1) {
            return ret;
        }
        ret = findrev(target, arr, pivot + 1, len - 1);
        if (ret != -1) {
            return ret;
        }
        return -1;
    }
    
    private int find(int target, MountainArray arr, int start, int end) {
        int ret = -1;
        while (start + 1 < end) {
            int mid = start + (end - start)  / 2;
            int v = arr.get(mid);
            if (v == target) {
                return mid;
            } else if (v < target) {
                start = mid;
            } else {
                end = mid;
            }
        }
        if (arr.get(start) == target) {
            return start;
        } else if (arr.get(end) == target) {
            return end;
        }
        return -1;
    }
    
    private int findrev(int target, MountainArray arr, int start, int end) {
        int ret = -1;
        while (start + 1 < end) {
            int mid = start + (end - start)  / 2;
            int v = arr.get(mid);
            if (v == target) {
                return mid;
            } else if (v < target) {
                end = mid;
            } else {
                start = mid;
            }
        }
        if (arr.get(start) == target) {
            return start;
        } else if (arr.get(end) == target) {
            return end;
        }
        return -1;
    }
}
