/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 

class Solution {
    public int findInMountainArray(int target, MountainArray A) {
        int peak = findPeak(A);
        int res = findAsc(target, 0, peak, A);
        if (res != -1) 
            return res;
        return findDesc(target, peak, A.length() - 1, A);
    
}
    
    private int findPeak(MountainArray A) {
        int lo = 0, hi = A.length() - 1;
        while (lo < hi) {
            int mi = lo + (hi - lo) / 2;
            if (A.get(mi) < A.get(mi + 1)) {
                lo = mi + 1;
            } else {
                hi = mi;
            }
        }
        return lo;    
    }

    private int findAsc(
int target, int lo, int hi, MountainArray A) {
        while (lo < hi) {
            int mi = lo + (hi - lo) / 2;
            int el = A.get(mi);
            if (el == target) return mi;
            if (el < target) {
                lo = mi + 1;
            } else {
                hi = mi;
            }
        }
        if (A.get(lo) == target) return lo;
        return -1;
    }

    private int findDesc(int target, int lo, int hi, MountainArray A) {
        while (lo < hi) {
            int mi = lo + (hi - lo) / 2;
            int el = A.get(mi);
            if (el == target) return mi;
            
if (el > target) {
                lo = mi + 1;
            } else {
                hi = mi;
            }
        }
        if (A.get(lo) == target) return lo;
        return -1;
    }
}
