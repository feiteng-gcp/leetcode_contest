/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    public int findInMountainArray(int target, MountainArray mountainArr) {
        int len = mountainArr.length();
        int left = 0;
        int right = len - 1;
        int leftval = -1;
        int rightval = -1;
        int midIndex = -1;
        while (left < right - 1) {
            int mid = (left + right) /2;
            int midval = mountainArr.get(mid);
            if (midval > mountainArr.get(mid + 1)) {
                if (midval > mountainArr.get(mid - 1)) {
                    midIndex = mid;
                    break;
                }
                else {
                    right = mid;
                }
            }
            else {
                left = mid;
            }
        }

        // if (midIndex == -1) {

        // }

        left = 0;
        right = midIndex;
        while (left < right - 1) {
            int mid = (left + right) / 2;
            int midval = mountainArr.get(mid);
            if (midval == target) {
                return mid;
            }
            else if (midval < target) {
                left = mid;
            }
            else {
                right = mid;
            }
        }

        if (mountainArr.get(left) == target) {
            return left;
        }
        if (mountainArr.get(right) == target) {
            return right;
        }

        left = midIndex + 1;
        right = len - 1;
        while (left < right - 1) {
            int mid = (left + right) / 2;
            int midval = mountainArr.get(mid);
            if (midval == target) {
                return mid;
            }
            else if (midval < target) {
                right = mid;
            }
            else {
                left = mid;
            }
        }

        if (mountainArr.get(left) == target) {
            return left;
        }
        if (mountainArr.get(right) == target) {
            return right;
        }
        return -1;
     }
}
