class Solution {
    public double[] sampleStats(int[] count) {
        double mean = 0, median = 0, mode = 0, min = -1, max = 0, sum = 0;
        int modeCnt = 0, nonZeroElements = 0, nonzCnt = 0;
        for (int i = 0; i < count.length; ++i) {
            sum += i * count[i];
            if (count[i] > 0) {
                nonZeroElements += count[i];
                max = i;
                if (min == -1) min = i;
                if (count[i] > modeCnt) {
                    modeCnt = count[i];
                    mode = i;
                }
            }
        }
        for (int i = 0; i < count.length; ++i) {
            if (count[i] > 0) {
                nonzCnt += count[i];
                if (nonzCnt >= nonZeroElements / 2) {
                    if (nonZeroElements % 2 == 1) { 
                        median = i;
                    } else {
                        if (nonzCnt > nonZeroElements / 2) {
                            median = i;
                        } else {
                            for (int j = i + 1; j < count.length; ++j) {
                                if (count[j] > 0) {
                                    median = (double)(i + j) / 2.0;
                                    break;
                                }
                            }
                        }
                    }
                    break;
                }
            }
        }
        
        return new double[] { min, max, sum / (double)nonZeroElements, median, mode };
    }
}
