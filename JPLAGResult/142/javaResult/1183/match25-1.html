class Solution {
    public double[] sampleStats(int[] count) {
        int min = -1;
        int max = -1;
        long sum = 0;
        int samples = 0;
        int mode = 0;
        int times = 0;

        for (int i = 0; i < count.length ;i++) {
            if(count[i] == 0) {
                continue;
            }
            if (min == -1) {
                min = i;
            }
            max = i;
            sum += (long)i * (long)count[i];
            samples += count[i];
            
if ( count[i] > times) {
                times = count[i];
                mode = i;
            }
        }

        int start = 0;
        int target1 = (samples+1)/2;
        double median1 = -1;

        int target2 = (samples+2)/2;
        double median2 = -1;

        // 1
        // 1 2 1
        // 1-1 2-3 4-4
        for (int i = 0; i < count.length ;i++) {
            if(count[i] == 0) {
                continue;
            }

            if (target1 >= start + 1 && target1 <= start + count[i]) {
                median1 = i;
            }

            if (target2 >= start + 1 && target2 <= start + count[i]) {
                median2 = i;
            
}
            start += count[i];
        }
            // minimum, maximum, 
    // mean, median, and mode of the sample respectively
        return new double[]{min,max,(double)sum/samples,(median1 + median2)/2,mode};
    }
}
