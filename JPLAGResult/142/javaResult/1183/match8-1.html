class Solution {
    public double[] sampleStats(int[] counts) {
        double[] ans = new double[5];
        
double sum = 0;
        int count = 0;
        int min = -1;
        int max = -1;
        int mode = 0;
        int maxCount = 0;
        for(int i = 0; i < counts.length; i++) {
            if(min == -1 && counts[i] != 0)
                min = i;
            if(counts[i] != 0)
                max = i;
            if(counts[i] > maxCount) {
                maxCount = counts[i];
                mode = i;
            }
            sum += i * counts[i];
            count += counts[i];
        }
        ans[0] = min;
        ans[1] = max;
        ans[2] = sum / count;
        ans[4] = mode;
        
int temp = 0;
        int half = count / 2;
        int left = 0;
        int right = 0;
        for(int i = 0; i < counts.length; i++) {
            if((count & 1) == 0) {
                if(temp < half && temp + counts[i] == half) {
                    left = i;
                } if(temp < half && temp + counts[i] > half) {
                    ans[3] = i;
                    break;
                } if(temp == half && temp + counts[i] > half) {
                    ans[3] = (double)(left + i) / 2;
                    
break;
                }
                
            } else {
                if(temp <= half && temp + counts[i] > half) {
                    ans[3] = i;
                    break;
                }
            }
            temp += counts[i];
        }
        return ans;
    }
}
