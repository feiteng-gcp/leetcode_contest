class Solution {
    public double[] sampleStats(int[] count) {
        double min = 256, max = 0, mean, median, mode = 0, modeNum = 0, sum = 0;
        int num = 0;
        for (int i = 0; i< count.length; i++) {
            if (count[i] != 0) {
                min = Math.min(i, min);
                max = Math.max(i, max);
                if (count[i] > modeNum) {
                    mode = i;
                    modeNum = count[i];
                }
            }
            sum += count[i] * i;
            num += count[i];
        }

        mean = sum / num;

        int i = 0, j = 0;

        if (num % 2 == 0) {
            i = num / 2; j = num/2 + 1;
        } else {
            i = num / 2;
            j = num / 2;
        }


        int n = 0, m1 = -1, m2 = -1;
        for (int x = 0; x< count.length; x++) {
            n += count[x];
            if (n >= i && m1 == -1) {
                m1 = x;
            }
            if (n >= j && m2 == -1) {
                m2 = x;
            }
        }

        median = (m1 + m2) / 2.0;

        return new double[] {min, max, mean, median, mode};
    }
}
