class Solution {
    public double[] sampleStats(int[] count) {
        double[] ans = new double[5];
        
        
for (int i = 0; i < 256; i ++) {
            if (count[i] > 0) {
                ans[0] = (i) * 1.0;
                break;
            }  
        }
        
for (int i = 255; i >= 0; i --) {
            if (count[i] > 0) {
                ans[1] = (i) * 1.0;
                break;
            }
            
        }
        long sum = 0;
        long total_count = 0;
        int max_count = 0;
        int max_index = -1;
        
        
        for (int i = 0; i < 256; i ++) {
            if (count[i] > 0) {
                sum += (i) * count[i];
                total_count += count[i];
                if (max_count < count[i]) {
                    max_count = count[i];
                    max_index = i;
                }
            }
            
        }
        // System.out.println(total_count);
        
ans[2] = 1.0 * sum / total_count;
        ans[4] = (max_index) * 1.0;
        //now find median
        long current_count = 0;
        double val1 = 0.0, val2 = 0.0;
        
        for (int i = 0; i < 256; i ++) {
            current_count += count[i];
            if (current_count > total_count / 2) {
            
                if (val1 > 0.0) {
                    ans[3] = (val1 + i) / 2.0;
                }
                else ans[3] = i + 0.0;
                break;
            
}
            if (current_count == total_count / 2 && total_count % 2 == 0) {
                // System.out.println(current_count);
                val1 = i + 0.0;
            }  
        }
        return ans;
        
        
        
    }
}
