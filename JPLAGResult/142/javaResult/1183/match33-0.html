class Solution {
    public double[] sampleStats(int[] count) {
        double min, max, mean, median, mode;
        int[] lSum = new int[count.length];
        
        min = -1;
        max = -1;
        mode = -1;
        int modeNum = -1;
        int total = 0;
        double sum = 0;
        for (int i = 0 ; i <= 255 ; i++)
        {
        	if (count[i] > modeNum)
        	{
        		modeNum = count[i];
        		mode = i;
        	}
        	if (count[i] != 0)
        	{
        		if (min == -1)
        			min = i;
        		max = i;
        	}
        	
        	total = total + count[i];
        	sum = sum + count[i] * i; 
        	
        	if (i == 0)
        		lSum[i] = count[i];
        	else
        		lSum[i] = lSum[i - 1] + count[i];
        }
        mean = sum / total;
        
        if (total % 2 == 0)
        {
            int midA = total / 2 - 1;
            int midB = total / 2;
            
            double numA = 0, numB = 0;
            
            for (int i = 1 ; i <= 255 ; i++)
            {
            	if ((lSum[i - 1] <= midA) && (midA < lSum[i]))
            		numA = i;
            	if ((lSum[i - 1] <= midB) && (midB < lSum[i]))
            		numB = i;
            }
            
            median = (numA + numB) / 2;
        }
        else
        {
            int midA = total / 2;
            
            double numA = 0;
            
            for (int i = 1 ; i <= 255 ; i++)
            {
            	if ((lSum[i - 1] < midA) && (midA <= lSum[i]))
            		numA = i;
            }
            
            median = numA;
        	
        }
        
        double[] ans = new double[5];
        ans[0] = min;
        ans[1] = max;
        ans[2] = mean;
        ans[3] = median;
        ans[4] = mode;
        
        return ans;
    }
}
