class Solution {
    public double[] sampleStats(int[] count) {
        
double min = 0, max= 0, mean = 0, median=0, mode = 0;
        int cnt = 0;
        double sum = 0;
        int maxOcc = 0;
        
        for(int i = 0; i < 256; i++) {
            if(count[i] != 0) {
                min = i;
                break;
            
}
        }
        
        for(int i = 255; i >= 0; i--) {
            if(count[i] != 0) {
                max = i;
                break;
            }
        }
        
        for(int i = 0; i < 256; i++) {
            cnt += count[i];
            sum += i * count[i];
            
            if(count[i] > maxOcc) {
                maxOcc = count[i];
                mode = i;
            }
        }
        mean = sum / (double)cnt;
        
        int c = 0;
        int pre = -1;
        for(int i = 0; i < 256; i++) {
            c += count[i];
            
            if(2 * c == cnt) 
{
                for(int j = i + 1; j < 256; j++) {
                    if(count[j] != 0) {
                        median = (double)(i + j) / 2;
                        break;
                    }
                }
                
                break;
                
            } else if(2 * c > cnt) {
                if(count[i] > 1) {
                    median = i;
                } else {
                    median = (double)(pre + i) / 2;    
                }
                
                break;
            }
            
            if(count[i] != 0) {
                pre = i;
            
}
        }
        
        return new double[]{min, max, mean, median, mode};
        
    }
}
