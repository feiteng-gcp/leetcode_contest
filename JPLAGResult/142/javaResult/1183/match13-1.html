class Solution {
    public double[] sampleStats(int[] count) {
        int total = 0;
        for (int i = 0; i < 256; i++) {
            total += count[i];
        }

        
double min = 1e100;
        double max = -1;
        double mean = 0.0;
        double median = 0.0;
        double mode = 0;

        int maxCount = 0;

        for (int i = 0; i < 256; i++) {
            if (count[i] > 0) {
                min = Math.min(min, (double) i);
                max = Math.max(max, (double) i);

                mean += (double) count[i] * i;

                if (maxCount < count[i]) {
                    maxCount = count[i];
                    mode = i;
                }
            }
        }

        mean = mean/ total;
        if (total % 2 == 1) {
            median = (double) getKth(count, total / 2);
        } else {
            int a = getKth(count, total / 2);
            int b = getKth(count, total / 2 + 1);
            median = ((double)(a) + b) / 2.0;
        
}

        return new double[] {min, max, mean, median, mode};
    }

    // K = 1 .. sum()
    int getKth(int[] count, int K) {
        int sum = 0;
        for (int i = 0; i < 256; i++) {
            sum += count[i];
            if (sum >= K) return i;
        }
        return 0;
    }
}
