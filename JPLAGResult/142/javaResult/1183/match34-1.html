class Solution {
    public double[] sampleStats(int[] count) {
         double[] ret = new double[5];
        
double sum = 0,cnt = 0,curMax = 0;
        boolean metMin = false;
        for (int i = 0; i < count.length; i++) {
            if(count[i] > 0){
                if(!metMin){
                    metMin = true;
                    ret[0] = i;
                }
                ret[1] = i;
            
}
            if(count[i] > curMax){
                ret[4] = i;
                curMax = count[i];
            }
            cnt += count[i];
            sum += count[i] * i;
        }

        ret[2] = sum / cnt;
        
int mf = (int) ((cnt + 1) / 2),ms = (int) ((cnt + 2) / 2);
        double first = -1,second = 0;
        int now = 0;
        for (int i = 0; i < count.length; i++) {
            now += count[i];

            if(now >= mf && first < 0) first = i;
            if(now >= ms){
                second = i;
                break;
            }
        }
        ret[3] = (first + second) / 2;
        return ret;
    }
}
