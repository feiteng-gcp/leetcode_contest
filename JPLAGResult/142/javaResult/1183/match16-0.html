	class Solution {
	    public double[] sampleStats(int[] count) {
	    	int n = count.length;
	    	double min = Double.POSITIVE_INFINITY;
	    	double max = -1;
	    	double s = 0;
	    	int all = 0;
	    	int mode = -1;
	    	int best = -1;
	        for(int i = 0;i < n;i++){
	        	if(count[i] > 0){
	        		min = Math.min(min, i);
	        		max = Math.max(max, i);
	        		s += (double)count[i] * i;
	        		all += count[i];
	        		if(count[i] > best){
	        			best = count[i];
	        			mode = i;
	        		}
	        	}
	        }
	        double med = 0;
	        if(all % 2 != 0){
		        int u = 0;
		        for(int i = 0;i < n;i++){
		        	u += count[i];
		        	if(u >= (all+1)/2){
		        		med = i;
		        		break;
		        	}
		        }
	        }else{
		        int u = 0;
		        for(int i = 0;i < n;i++){
		        	u += count[i];
		        	if(u >= all/2){
		        		med += i;
		        		break;
		        	}
		        }
		        u = 0;
		        for(int i = 0;i < n;i++){
		        	u += count[i];
		        	if(u >= all/2+1){
		        		med += i;
		        		break;
		        	}
		        }
		        med /= 2;
	        }
	        return new double[]{min, max, s/all, med, mode};
	    }
	}
