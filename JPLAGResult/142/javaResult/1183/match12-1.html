class Solution {
    public double[] sampleStats(int[] count) {
        int min = Integer.MAX_VALUE;
        int max = Integer.MIN_VALUE;
        long sum = 0l;
        int quan = 0;
        int maxCount = 0;
        int maxIndex = -1;
        for (int i = 0; i < count.length; i++) {
            if (count[i] != 0) {
                min = Math.min(min, i);
                max = Math.max(max, i);
                sum += 1l * i * count[i];
                quan += count[i];
                if (count[i] > maxCount) {
                    maxCount = count[i];
                    maxIndex = i;
                }
            }
        }
        
        return new double[]{min, max, 1.0 * sum / quan, getMedian(count, quan), maxIndex};
    }
    
    public double getMedian(int[] count, int quan) {
        if (quan % 2 == 1) {
            return findKth(count, quan / 2 + 1);
        } else {
            return (findKth(count, quan / 2) + findKth(count, quan / 2 + 1)) / 2;
        }
    
}
    
    public double findKth(int[] count, int k) {
        int quan = 0;
        for (int i = 0; i < count.length; i++) {
            if (count[i] != 0) {
                quan += count[i];
                if (quan >= k) {
                    return i;
                
}
            }
        }
        return -1;
    }
}
