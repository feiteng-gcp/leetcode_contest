class Solution {
    public double[] sampleStats(int[] count) {
        double[] res = new double[5];
        
double min = -1, max = 0, sum = 0, mode = 0, num = 0;
        int size = 0;
        for (int i = 0; i < 256; i++){
            if (count[i] != 0){
                if (min == -1){
                    min = (double) i;
                }
                max = (double) i;
                sum += (double) i * count[i];
                size += count[i];
                if (count[i] > num){
                    num = count[i];
                    mode = (double) i;
                }
            }
        }
        double mid = size % 2 == 0 ? (kth(count, size / 2) + kth(count, size / 2 + 1)) / 2.0 : kth(count, size / 2 + 1) * 1.0;
        res[0] = min;
        res[1] = max;
        res[2] = sum / size;
        res[3] = mid;
        res[4] = mode;
        return res;
    
}
    private int kth(int[] count, int k){
        for (int i = 0; i < 256; i++){
            if (count[i] > 0){
                if (count[i] >= k){
                    return i;
                } else {
                    k = k - count[i];
                
}
            }
        }
        return -1;
    }
}
