class Solution {
    public double[] sampleStats(int[] count) {
        double[] result = new double[5];
        result[0] = -1.0;
        long total_num = 0;
        long total_sum = 0;
        int mod = 0;
        int mod_result = 0;
        // List<Integer> total_list = new ArrayList();
        for (int i = 0; i < count.length; i++) {
            if (result[0] == -1.0 && count[i] != 0) {
                result[0] = (double)(i);
            }
            
            if (count[i] != 0) {
                result[1] = (double)(i);
            }
            
            total_num += count[i];
            total_sum += i*count[i];
            // for (int j = 0; j < count[i]; j++)  total_list.add(i);
            
            if (count[i] > mod) {
                mod = count[i];
                mod_result = i;
            }
        }
        
        result[2] = total_sum/(double)(total_num);
        if (total_num % 2 == 0) {
            // result[3] = (total_list.get((int)total_num/2)+total_list.get((int)total_num/2-1))/2.0;
            long target1 = total_num/2, target2 = total_num/2-1;
            // System.out.printf("%d %d\n", target1, target2);
            int result1 = -1, result2 = -1;
            int cur_sum = 0;
            for (int i = 0; i < count.length; i++) {
                if (cur_sum+count[i]-1 >= target2 && result2 == -1)    result2 = i;
                if (cur_sum+count[i]-1 >= target1 && result1 == -1)    result1 = i;
                cur_sum += count[i];
            }
            result[3] = (result1+result2)/2.0;
            
        } else {
            // result[3] = (double)(total_list.get((int)total_num/2));
            long target2 = total_num/2-1;
            int result2 = -1;
            int cur_sum = 0;
            for (int i = 0; i < count.length; i++) {
                if (cur_sum+count[i]-1 >= target2 && result2 == -1)    result2 = i;
                cur_sum += count[i];
            }
            result[3] = (double)result2;
        }
        result[4] = (double)(mod_result);
        return result;
    }
}
