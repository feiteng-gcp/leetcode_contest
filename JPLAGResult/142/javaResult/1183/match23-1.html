class Solution {
    public double[] sampleStats(int[] count) {
        double min = 100000, max = -1, sum = 0, mode = 0, n1 = 0, n2 = 0;
        int acc = 0, maxN = 0, total = 0;
        for (int c : count)
            total += c;
        for (int i = 0; i < count.length; i++) {
            if (count[i] != 0) {
                if (i < min) 
                    min = i;
                
if (i > max)
                    max = i;
                if (count[i] > maxN) {
                    maxN = count[i];
                    mode = i;
                }
                sum += (long)i * count[i];
                if (acc < (total+1)/2 && acc + count[i] >= (total+1)/2 )
                    n1 = i;
                if (acc < (total+2)/2 && acc + count[i] >= (total+2)/2 )
                    n2 = i
;
                acc += count[i];
            }
        }
        return new double[]{min, max, sum / total, (n1 + n2) / 2, mode};
    }
}
