class Solution {
    public double[] sampleStats(int[] count) {
        double[] res = new double[5];
        double min = 0;
        double max = 0;
        double mean = 0;
        double median = 0;
        double mode = 0;
        int total = 0;
        for(int i = 0; i < count.length; i++){
            if(count[i] != 0){
                min = i;
                break;
            }
        }
        for(int i = count.length - 1; i >= 0; i--){
            if(count[i] != 0){
                max = i;
                break;
            }
        }
        for(int i = 0; i < count.length; i++){
            total += count[i];
        }
        for(int i = 0; i < count.length; i++){
            mean = mean + (count[i] * i);
        }
        mean = mean / total;
        int maxOcc = 0;
         for(int i = 0; i < count.length; i++){
             if(count[i] > maxOcc){
                 mode = i;
                 maxOcc = count[i];
             }
         }
        
        int curSum = 0;
        int target = 0;
        if(total % 2 == 1){
            target = (total+1) / 2;
            for(int i = 0; i < count.length; i++){
                curSum += count[i];
                if(curSum >= target){
                    median = i;
                    break;
                }
            }
        }else{
            double v1 = 0;
            double v2 = 0;
            int target1 = total / 2;
            int target2 = total / 2 + 1;
            for(int i = 0; i < count.length; i++){
                curSum += count[i];
                if(curSum >= target1){
                    v1 = i;
                    break;
                }
            }
            curSum = 0;
            for(int i = 0; i < count.length; i++){
                curSum += count[i];
                if(curSum >= target2){
                    v2 = i;
                    break;
                }
            }
            median = (v1 + v2) / 2;
        }
        
        res[0] = min;
        res[1] = max;
        res[2] = mean;
        res[3] = median;
        res[4] = mode;
        return res;
        
    }
}
