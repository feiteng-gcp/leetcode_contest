class Solution {
    public double[] sampleStats(int[] count) {
        double[] ans=new double[5];
        int min=255,max=0,maxCount=0,x=0;
        long sum=0,c=0;
        for(int i=0;i<count.length;i++)
        {
            sum+=count[i]*i;
            c+=count[i];
            if(count[i]>0)
            {    
                min=Math.min(min,i);
                max=Math.max(max,i);
            }
            if(maxCount<count[i]){
                maxCount=count[i];
                x=i;
            }
        }
        ans[0]=min;
        ans[1]=max;
        ans[2]=sum*1.0/c;
        ans[4]=x;
        if(c%2==0)
        {
            long index1=(long)(c/2);
            long index2=(long)(c/2)+1;
            long countN=0;
            long res=0;
            for(int i=0;i<count.length;i++){
                countN+=count[i];
                if(countN>=index1&&index1!=-1)
                {    
                   
                    res+=i;
                    index1=-1;
                }
                if(countN>=index2&&index2!=-1)
                {    
                 
                    res+=i;
                    index2=-1;
                }
            }
            ans[3]=res*1.0/2;
        }
        else{
            long index=(long)(c/2)+1;
            long countN=0;
            int res=0;
            for(int i=0;i<count.length;i++){
                countN+=count[i];
                if(countN>=index&&index!=-1)
                {    
                    res+=i;
                    index=-1;
                }
            }
            ans[3]=res;
        }
        return ans;
    }
}
