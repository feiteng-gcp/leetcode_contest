class Solution {
    public double[] sampleStats(int[] count) {
        int min = -1;
        int max = -1;
        double sum = 0;
        int maxnum = 0;
        int mode = 0;
        int totalnum = 0;
        for (int i = 0; i < count.length; i++) {
            if (count[i] > 0) {
                if (min == -1) {
                    min = i;
                }
                
                max = i;
                
                
if (maxnum < count[i]) {
                    mode = i;
                    maxnum = count[i];
                }
                
                totalnum += count[i];
                sum += (double) count[i] * i;
            }
        }
        
        double[] ans = new double[5];
        ans[0] = min;
        ans[1] = max;
        ans[3] = 0;
        ans[2] = (double) sum / (double) totalnum;
        ans[4] = mode;
        
        
int num = 0;
        int cb = -1;
        int wb = -1;
        for (int i = 0; i < count.length; i++) {
             if (count[i] > 0) {
                 if (num < totalnum / 2 && (num + count[i]) >= totalnum / 2) {
                     cb = i;
                 }
                 
                 if (num < totalnum / 2 + 1 && (num + count[i]) >= totalnum / 2 + 1) {
                    wb = i;
                 }
                 
                 num += count[i];
             }
            
            if (cb >= 0 && wb >= 0) {
                break;
            }
        }
        
        if (totalnum % 2 == 0) {
            ans[3] = (double) cb / 2 + (double) wb / 2;
        } else {
            ans[3] = cb;
        }
        
        return ans;
    }
}
