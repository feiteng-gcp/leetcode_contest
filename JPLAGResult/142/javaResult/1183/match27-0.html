class Solution {
    public double[] sampleStats(int[] count) {
        double max = 0;
        double min = 255;
        double mode = 0;
        int cnt = 0;
        int totalCount = 0;
        double median;
        double mean = 0;
        for(int i = 0; i < count.length; ++i) {
            if(count[i] > 0) {
                if(count[i] > cnt) {
                    mode = i;
                    cnt = count[i];    
                }
                max = i;
                min = Math.min(min, i);
                mean += (i * count[i]);
                totalCount += count[i];
            }
        }
        
        if(totalCount % 2 == 0) {
            median = (findKth(count, totalCount / 2) + findKth(count, totalCount / 2 + 1)) / 2;
        } else {
            int i = totalCount / 2 + 1;
            median = findKth(count, i);
        }
        
        mean /= totalCount;
        
        
        return new double[]{min, max, mean, median, mode};
    }
    
    private double findKth(int[] count, int c) {
        for(int i = 0; i < count.length; ++i) {
            if(count[i] > 0) {
                c -= count[i];
                if(c <= 0) {
                    return (double) i;
                }
            }
        }
        
        return 0.0;
    }
}
