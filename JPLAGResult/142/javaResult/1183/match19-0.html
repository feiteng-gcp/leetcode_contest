class Solution {
    public double[] sampleStats(int[] cnt) {
        long min = 300;
        long max = -1;
        long sum = 0;
        long count = 0;
        
        long mode = 0;
        long modeCnt = 0;
        
        for (int i = 0; i < 256; i++) {
            int c = cnt[i];
            if (c > 0) {
                min = Math.min(min, i);
                max = Math.max(max, i);
                count += c;
                sum += i * c;
                
                if (c > modeCnt) {
                    modeCnt = c;
                    mode = i;
                }
            }
        }
        
        long a = count / 2 + 1;
        long b = a;
        if (count % 2 == 0) {
            a = count / 2 - 1;
            b = a + 1;
        }
        
        double median = 0.0;
        int i = -1;
        for (int j = 0; j < 256; j++) {
             int c = cnt[j];
             if (c > 0) {
                i += c;
                 if (i >= a) {
                     median += j;
                     a = Integer.MAX_VALUE;
                 }
                 if (i >= b) {
                     median += j;
                     a = Integer.MAX_VALUE;
                     break;
                 }
             }
         }
        
        return new double[] {min, max, 1.0 * sum / count, median / 2.0, mode};
                
    }
}
