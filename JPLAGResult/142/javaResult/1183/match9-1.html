class Solution {
    public double[] sampleStats(int[] count) {
        double sum = 0;
        double min = 200;
        double max = -1;
        double totalCount = 0;
        double modeCount = 0;
        double mode = 0;
        
        for (int i = 0; i < count.length;i++){
            int currCount = count[i];
            sum += i * currCount;
            totalCount += currCount;
            
if (currCount > 0){
                min = Math.min(min, i);
                max = Math.max(max, i);
                if (currCount > modeCount){
                    modeCount = currCount;
                    mode = i;
                }
            }
            
        }
        double curr = 0;
        double median = 0;
        double left = 0;
        double right = 0;
        for (int i = 0; i < count.length;i++){
            curr += count[i];
            if (curr >= totalCount/2){
                left = i;
                break;
            }
        }
        curr = 0;
        for (int i = count.length-1;i >= 0;i--){
            curr += count[i];
            if (curr >= totalCount/2){
                right = i;
                break;
            }
        }
        median = (left+right)/2;
        double[] ans = new double[]{min, max, sum/totalCount,median, mode};
        return ans;
    }
}
