class Solution {
    public double[] sampleStats(int[] count) {
        double[] ans = new double[5];
        ans[0] = -1;
        ans[1] = -1;
        ans[3] = -1;
        int totalC=0;
        double sum = 0;
        int highF = 0;
        for(int i = 0; i< count.length; i++){
            totalC += count[i];
            sum += count[i]*i;
            if(count[i]>0){
                if(ans[0]<0) {ans[0] = i;}
                else{ans[0] = Math.min(i,ans[0]);}
                
                if(ans[1]<0) {ans[1] = i;}
                else{ans[1] = Math.max(i,ans[1]);} 
            }
            if(count[i]>highF){
                ans[4] = i;
                highF = count[i];
            }
        }
        int x = 0;
        if(totalC%2==0){
            for(int i = 0; i < count.length; i++){
                  if(ans[3] < 0 && x<totalC/2 && x+count[i]>=totalC/2+1){
                      ans[3] = i; break;
                  }
                  if(ans[3]< 0 && x<totalC/2 && x+count[i]==totalC/2){
                      ans[3] = i;
                  }
                  if(ans[3] >=0 && count[i]>0){
                      ans[3] = (double)(ans[3]+i)/2;
                  }
                  x += count[i];
            }            
        }else{
            for(int i = 0; i < count.length; i++){
                  if(x<totalC/2 && x+count[i]>=totalC/2) {ans[3] = i;break;}
                  x += count[i];
            }  
        }

        
        ans[2] = (double)sum/totalC;
        return ans;
    }
}
