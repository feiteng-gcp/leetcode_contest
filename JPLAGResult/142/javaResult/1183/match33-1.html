class Solution {
   public double[] sampleStats(int[] count) {
        int maxN = 0, minN = 255, many = 0, cnt = 0;
        double sum = 0;
        for (int i = 0; i < 256; i++) {
            sum +=  i * count[i];
            if (count[i] > 0) {
                maxN = Math.max(maxN, i);
                minN = Math.min(minN, i);
            }
            cnt += count[i];
            if (count[i] > count[many]) many = i;
        }
        double avg = sum / (cnt * 1.0);
        
double mid = 0.0;
        int tmp = 0;
        int first = -1;
        int second = -1;
        for (int i = 0; i < 256; i++) {
            tmp += count[i];
            if (tmp >= cnt / 2 && first == -1) first = i;
            if (tmp >= cnt / 2 + 1 && second == -1) second = i;
        }
        if (cnt % 2 == 0) {
            mid = (first + second) * 1.0 / 2;
        } else {
            
mid = second;
        }
        double[] res = new double[5];
        res[0] = minN;
        res[1] = maxN;
        res[2] = avg;
        res[3] = mid;
        res[4] = many;
       
        return res;
    }
}
