class Solution {
    public double[] sampleStats(int[] count) {
        int totalNum = 0;
        double totalSum = 0.0;
        int min = -1, max = 257;
        int modeCount = 0; double mode = -1.0;
        
        for (int i=0; i<count.length; ++i) {
            int c = count[i];
            totalNum += c;
            
            //min
            if (c != 0 && min == -1) min = i;
            //max
            if (c != 0) max = i;
            
            totalSum += c * i;
            //mode
            if (c > modeCount) {
                modeCount = c;
                mode = (double)i;
            }
        }
        
        double mean = totalSum / totalNum;
        int countSoFar = 0;
        boolean flag = false;
        int target = (totalNum + 1) / 2;
        double median = 0.0;
        for (int i=0; i<count.length; ++i) {
            int c = count[i];
            countSoFar += c;
            if (countSoFar == target) {
                if (totalNum % 2 == 1) {
                    median = (double) i;
                    break;
                } else {
                    flag = true;
                    median += i;
                }
            } else if (countSoFar > target) {
                if (totalNum % 2 == 1) {
                    median = (double) i;
                    break;
                } else {
                    median = flag ? (median + i) / 2.0 : i;
                    break;
                }
            }
        }
        
        return new double[]{(double)min, (double)max, mean, median, mode};
    }
}
