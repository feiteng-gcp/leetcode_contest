class Solution {
    public double[] sampleStats(int[] count) {
        double min = 300, max = -1, sum = 0, mode = 0;
        int cnt = 0, max_cnt = 0;
        for(int i = 0; i < count.length; i++) {
            if(count[i] != 0) {
                min = Math.min(min, i);
                max = Math.max(max, i);
                cnt += count[i];
                sum += (i * count[i]);
                if(count[i] > max_cnt) {
                    max_cnt = count[i];
                    mode = i;
                }
            }
        }
        double median;
        if(cnt % 2 == 1) {
            median = findKth((cnt+1)/2, count);
        } else {
            median = (findKth(cnt/2, count) + findKth(cnt/2 + 1, count)) / 2.0;
        }
        
        return new double[] {min, max, sum/cnt, median, mode};
    }
    
    private double findKth(int k, int[] count) {
        int idx = 0;
        for(int i = 0; i < count.length; i++) {
            if(count[i] != 0) {
                idx += count[i];
                if(idx >= k)
                    return i;
            }
        }
        return -1;
    }
}
