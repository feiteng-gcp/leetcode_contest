class Solution {
    public double[] sampleStats(int[] count) {
        
int min = 255;
        int max = 0;
        double sum = 0;
        int total = 0;
        int maxCount = 0;
        int mod = 0;
        
        for(int i = 0; i < 256; i++) {
            if(count[i] > 0) {
                min = Math.min(min, i);
                max= Math.max(max, i);
                
                if(count[i] > maxCount) {
                    maxCount = count[i];
                    mod = i;
                }
                
                sum += i * count[i];
                total += count[i];
            }
        }
        
        int half = 0;
        
double mid = 0;
        int pre = 0;
        int preHalf = 0;
        
        for(int i = 0; i < 256; i++) {
            if(count[i] > 0) {
                preHalf = half;
                half += count[i];
            }
            
            if(half * 2 > total) {
                if(total % 2 == 0) {
                    if(preHalf == total / 2) {
                        mid = (pre + i) / 2.0;
                    } else {
                        mid = i;
                    }
                } else {
                    mid = i;
                }
                
                break;
            }
            
            pre = i;
        
}
        
        return new double[] {
            (double)min, (double)max, sum / (double)total, (double) mid, (double)mod
        };
    }
}
