class Solution {
    public double[] sampleStats(int[] count) {
        double[] ans = new double[5];
        ans[0] = (double) 256;
        int N = count.length;
        for (int i = 0; i < N; i++) {
            if (count[i] > 0) {
                ans[0] = (double) i;
                break;
            }
        }
        for (int i = N - 1; i >= 0; i--) {
            if (count[i] > 0) {
                ans[1] = (double) i;
                break;
            }
        }
        long total = 0;
        for (int i = 0; i < N; i++) {
            total += (long) i * count[i];
        }
        
        int n = 0;
        for (int i = 0; i < N; i++) n += count[i];
        ans[2] = (double) total / n;
        int m1 = n / 2;
        int m2 = n / 2;
        if (n % 2 == 0) {
            m1 = m2 - 1;
        }
        m2++;
        m1++;
        int n1 = 0, n2 = 0;
        int sum = 0;
        for (int i = 0; i < N; i++) {
            int cnt = count[i];
            if (sum  < m1 && sum + cnt  >= m1) {
                n1 = i;
                break;
            }
            sum += cnt;
        }
        sum = 0;
        for (int i = 0; i < N; i++) {
            int cnt = count[i];
            if (sum < m2 && sum + cnt >= m2) {
                n2 = i;
                break;
            }
            sum += cnt;
        }
        ans[3] = 1.0 * (n1 + n2) / 2.0;
        int max = 0;
        int mode = 0;
        for (int i = 0; i < N; i++) {
            if (count[i] > max) {
                max = count[i];
                mode = i;
            }
        }
        ans[4] = (double) mode;
        //System.out.println(m1 + " " + m2 + " " + n1 + " " + n2);
        return ans;
    }
}
