class Solution {
    public double[] sampleStats(int[] count) {
        double[] ret = new double[5];
        
long total = 0;
        for (int i = 0; i < 256; i++) {
            if (count[i] != 0) {
                ret[0] = i;
                break;
            }
        }
        for (int i = 255; i >= 0; i--) {
            if (count[i] != 0) {
                ret[1] = i;
                break;
            }
        }
        
        int fmax = 0;
        
int vfmax = -1;
        int sum = 0;
        for (int i = 0; i < 256; i++) {
            if (count[i] > fmax) {
                fmax = count[i];
                vfmax = i;
            }
            sum += count[i];
            
            total += i * count[i];
        }
        ret[4] = vfmax; 
        ret[2] = total / 1.0 / sum;
        if (sum % 2 == 1) {
            int tmpsum = 0;
            for (int i = 0; i < 256; i++) {
                if (count[i] == 0) {
                    continue;
                }
                if (tmpsum <= sum / 2 && (tmpsum + count[i]) > sum / 2) {
                    ret[3] = i;
                    return ret;
                }
                tmpsum += count[i];
            }
        }
        
        int tmpsum = 0;
        int left = -1;
        int right = -1;
        for (int i = 0; i < 256; i++) {
            if (count[i] == 0) {
                continue;
            }
            tmpsum += count[i];
            
if (tmpsum >= sum / 2) {
                left = i;
                break;
            }
        }
        tmpsum = 0;
        for (int i = 255; i >= 0; i--) {
            if (count[i] == 0) {
                continue;
            }
            tmpsum += count[i];
            
if (tmpsum >= sum / 2) {
                right = i;
                break;
            }
        }
       // System.out.println(left + ","  + right) ;
        ret[3] = (left + right) / 2.0;
        return ret;
    }
}
