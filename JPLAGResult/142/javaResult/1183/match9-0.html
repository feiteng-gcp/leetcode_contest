class Solution {
    public double[] sampleStats(int[] count) {
        int min = Integer.MAX_VALUE, max = Integer.MIN_VALUE;
        int mod_idx = -1, mod_val = 0, cnt = 0;
        long sum = 0;
        for (int i = 0; i < count.length; ++i) {
            if (count[i] > 0) {
                min = Math.min(min, i);
                max = Math.max(max, i);
                if (mod_val < count[i]) {
                    mod_val = count[i];
                    mod_idx = i;
                }
                sum += 1L * i * count[i];
                cnt += count[i];
            }
        }

        double mean = (double) sum / cnt;

        double med = 0;
        int curr_cnt = 0;
        int med_cnt = cnt / 2;
        for (int i = 0; i < count.length; ++i) {
            if (cnt % 2 == 0) {
                if (med == 0 && curr_cnt + count[i] >= med_cnt) {
                    med = i;
                }
                if (curr_cnt + count[i] >= med_cnt  +1) {
                    med += i;
                    med /= 2;
                    break;
                }
            } else {
                if (curr_cnt + count[i] >= med_cnt) {
                    med = i;
                    break;
                }
            }
            curr_cnt += count[i];
        }

        return new double[] {min, max, mean, med, mod_idx};
    }
}
