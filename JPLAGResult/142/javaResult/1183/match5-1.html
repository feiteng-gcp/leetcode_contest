class Solution {
    public double[] sampleStats(int[] count) {
        double[] result = new double[5];
        
long sum = 0;
        int cnt = 0;
        int min = Integer.MAX_VALUE;
        int max = Integer.MIN_VALUE;
        int maxCnt = 0;
        int mode = -1;
        for (int i = 0; i < count.length; i++) {
            cnt += count[i];
            sum += count[i] * i;
            if (count[i] > 0) {
                min = Math.min(i, min);
                max = Math.max(i, max);
            }
            
            if (maxCnt < count[i]) {
                maxCnt = count[i];
                mode = i;
            }
        }
        
        
long acc = 0;
        long mid1 = (cnt - 1) / 2;
        long mid2 = cnt / 2;
        for (int i = 0; i < count.length; i++) {
            if (acc <= mid1 && acc + count[i] > mid1) {
                mid1 = i;
            }
            
            if (acc <= mid2 && acc + count[i] > mid2) {
                mid2 = i;
            }
            
            acc += count[i];
        }
        
        return new double[] { min, max, (double) sum / cnt, (double) (mid1 + mid2) / 2.0, mode };
    }
}
