class Solution {
    public double[] sampleStats(int[] count) {
        long sum=0,mostF=-1,mostV=-1,minV=-1,maxV=-1,cnt=0;
        int[] arr=new int[256];
        for(int i=0;i<256;i++){
            sum+=i*count[i];
            cnt+=count[i];
            if(minV==-1&&count[i]>0){
                minV=i;
            }
            if(count[i]>0){
                maxV=i;
            }
            if(mostF<count[i]){
                mostV=i;
                mostF=count[i];
            }
        }
        double[] ans=new double[5];
        ans[0]=minV;
        ans[1]=maxV;
        ans[2]=((double)sum)/cnt;
        int tmp=0;
        for(int i=0;i<256;i++){
            tmp+=count[i];
            if(tmp>cnt/2){
                if(tmp>cnt/2){
                if(cnt%2==0){
                    if(tmp-count[i]<cnt/2) ans[3]=i;
                    else ans[3]=((double)(i+i-1))/2;
                }else{
                    ans[3]=i;
                }
                break;
            }
            }
        }
        ans[4]=mostV;
        return ans;
    }
}
