class Solution {
public:
    using pii = pair<int, int>;
    set<string> H;
    void dfs(int k, string now, vector<vector<string> > &tmp)
    {
        if (k == tmp.size())
        {
            H.insert(now);
            return;
        }
        for (int i = 0; i < tmp[k].size(); ++ i)
        {
            dfs(k+1, now+tmp[k][i], tmp);
        }
    }
    vector<string> braceExpansionII(string s) {
        int n = s.size();
        vector<pii> v;
        for (int i = 0, cur = 0, last = 0; i < n; ++ i)
        {
            if (s[i] == '{')
            {
                if (cur == 0) last = i;
                //printf("%d\n", i);
                cur ++;
            }
            else if (s[i] == '}')
            {
                cur --;
                if (cur == 0)
                {
                    v.push_back(pii(last, i));
                    //printf("%d %d\n", last, i);
                }
            }
            else if (cur == 0 && isalpha(s[i]))
            {
                v.push_back(pii(i, i));
            }
        }
        if (v.size() == 1)
        {
            if (v[0].first == v[0].second) return {s.substr(v[0].first, v[0].second-v[0].first+1)};
            vector<pii> w;
            s[n-1] = ',';

            for (int i = 1, cur = 0, last = 1; i <= n-1; ++ i)
            {
                if (s[i] == '{') cur ++;
                if (s[i] == '}') cur --;
                if (s[i] == ',' && cur == 0)
                {
                    w.push_back(pii(last, i-1));
                    last = i+1;
                }
            }
            //for (int i = 0; i < w.size(); ++ i)
            //    printf("%s ", s.substr(w[i].first, w[i].second-w[i].first+1).c_str());
            set<string> H;
            for (int i = 0; i < w.size(); ++ i)
            {
                auto cur = braceExpansionII(s.substr(w[i].first, w[i].second-w[i].first+1));
                for (auto s : cur) H.insert(s);
            }
            vector<string> ret;
            for (auto x : H) ret.push_back(x);
            
            //printf("%s: ", s.c_str());
            //for (auto x : ret) printf("%s ", x.c_str()); puts("");
            
            return ret;
        }
        else
        {
            //for (int i = 0; i < v.size(); ++ i)
            //    printf("%s ", s.substr(v[i].first, v[i].second-v[i].first+1).c_str());
            vector<vector<string> > tmp(v.size());
            for (int i = 0; i < v.size(); ++ i)
            {
                tmp[i] = braceExpansionII(s.substr(v[i].first, v[i].second-v[i].first+1));
            }
            H.clear();
            dfs(0, "", tmp);
            vector<string> ret;
            for (auto x : H) ret.push_back(x);
            
            //printf("%s: ", s.c_str());
            //for (auto x : ret) printf("%s ", x.c_str()); puts("");
            
            return ret;
        }
    }
};
