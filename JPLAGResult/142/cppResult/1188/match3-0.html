class Solution {
public:
    vector<string> braceExpansionII(string exp) {
        set<string> v;
        int len = exp.size();
        vector<string> pre({""});
        for (int idx = 0; idx < len;) {
            if (exp[idx] == ',') {
                for (auto & str : pre)
                    v.insert(str);
                pre.clear();
                idx++;
            } else if (exp[idx] == '{') {
                int start = ++idx;
                int cnt = 1;
                while (cnt) {
                    if (exp[idx] == '{')
                        cnt++;
                    else if (exp[idx] == '}')
                        cnt--;
                    idx++;
                }
                auto tmp = braceExpansionII(exp.substr(start, idx - 1 - start));
                vector<string> vv;
                if (pre.empty())
                    pre.push_back("");
                for (auto & ss : pre)
                    for (auto & str : tmp)
                        vv.push_back(ss + str);
                swap(vv, pre);
            } else if (islower(exp[idx])) {
                if (pre.empty())
                    pre.push_back("");
                for (auto & str : pre)
                    str.push_back(exp[idx]);
                idx++;
            } else {
                for (auto & str : pre)
                    v.insert(str);
                idx++;
            }
        }
        for (auto & str : pre)
            v.insert(str);
        
        return vector<string>(v.begin(), v.end());
    }
};
