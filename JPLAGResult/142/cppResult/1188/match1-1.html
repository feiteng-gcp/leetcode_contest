class Solution {
 public:
  string s;

  vector<string> merge(const vector<string>& L, const vector<string>& R) {
    vector<string> ret;
    for (auto& x : L) {
      for (auto& y : R) {
        ret.emplace_back(x + y);
      }
    }
    sort(ret.begin(), ret.end());
    return ret;
  }
  vector<string> solve(int& i) {
    //trace(s, i);
    if (s[i] != '{') {
      string cur;
      cur += s[i];
      vector<string> L = {cur};
      if (s[i + 1] == '}' || s[i + 1] == ',') return L;
      ++i;
      auto R = solve(i);
      auto ret = merge(L, R);
      // trace("111");
      // for (auto& it : ret) cout << it << " ";
      // cout << endl;
      return ret;
    }
    set<string> all;
    while (true) {
      if (s[i] == '}') break;
      ++i;
      auto L = solve(i);
      ++i;
      for (auto& it : L) all.insert(it);
    }
    vector<string> L(all.begin(), all.end());
    // trace("222");
    // for (auto& it : L) cout << it << " ";
    // cout << endl;
    if (s[i + 1] == '}' || s[i + 1] == ',') return L;
    ++i;
    auto R = solve(i);
    auto ret = merge(L, R);
    // trace("333");
    // for (auto& it : ret) cout << it << " ";
    // cout << endl;
    return ret;
  }
  vector<string> braceExpansionII(string expression) {
    this->s = expression + '}';
    int i = 0;
    auto ret = solve(i);
    return ret;
  }
};
