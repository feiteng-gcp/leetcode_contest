class Solution {
public:
    vector<string> dfs(string& input, int& pos){
    if(pos >= input.size())  return {};
    vector<string> interResult, result;
    interResult.push_back("");
    while(pos < input.size()){
        if(input[pos] == '{'){
            pos++;
            vector<string> tmp = dfs(input, pos);
            vector<string> newResult;
            for(auto prefix : interResult){
                for(auto suffix : tmp){
                    newResult.push_back(prefix + suffix);
                }
            }
            interResult = newResult;
        }else if(input[pos] == '}'){
            pos++;
            result.insert(result.end(), interResult.begin(), interResult.end());
            return result;
        }else if(input[pos] == ','){
            result.insert(result.end(), interResult.begin(), interResult.end());
            interResult.clear(), interResult.push_back(""), pos++;
        }else{
            for(int i  = 0; i < interResult.size(); ++i) interResult[i] += input[pos];
            pos++;
        }
    }
    result.insert(result.end(), interResult.begin(), interResult.end());
    return result;
    }
    vector<string> braceExpansionII(string expression) {
        int start = 0;
        vector<string> ret = dfs(expression, start);
        set<string> ss(ret.begin(), ret.end());
        vector<string> tmp(ss.begin(), ss.end());
        return tmp;
    }
};
