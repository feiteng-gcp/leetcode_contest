#include<bits/stdc++.h>


class Solution {
public:
    set<string> plus(set<string> a, set<string> b) {
        set<string> res;
        for (auto i: a) res.insert(i);
        for (auto i: b) res.insert(i);
        return res;
    }
    set<string> mult(set<string> a, set<string> b) {
        if (b.size()==0) return a;
        set<string> res;
        for (auto i: a) for (auto j: b) res.insert(i+j);
        return res;
    }
    
    set<string> dfs(string str) {
        set<string> res;
        int n = (int)str.length();
        if (str=="") return res;
        
        if (str[0]=='{') {
            int dep = 0;
            int back = 0;
            for (int i=0; i<n; i++) {
                if (str[i]=='{') dep++;
                else if (str[i]=='}') dep--;
                if (dep==0) {
                    back = i;
                    break;
                }
            }
            
            vector<pair<int,int>> segs;
            dep = 0;
            int fr = 1;
            for (int i=1; i<back; i++) {
                if (str[i]=='{') dep++;
                if (str[i]=='}') dep--;
                if (dep==0 && str[i]==',') {
                    segs.push_back( make_pair(fr, i-1) );
                    fr = i+1;
                }
            }
            segs.push_back( make_pair(fr, back-1) );
            for (auto i: segs) {
                res = plus(res, dfs(str.substr(i.first, i.second-i.first+1)) );
            }
            
            if (back+1 < n) {
                res = mult(res, dfs(str.substr(back+1)));
            }
        } else {
            res.insert( str.substr(0,1) );
            res = mult(res, dfs(str.substr(1)));
        }
        
        //printf("calc %s\n",str.c_str());
        //for (auto i: res) printf("%s ", i.c_str()); puts("");
        return res;
    }
    
    vector<string> braceExpansionII(string expression) {
        set<string> ans = dfs(expression);
        vector<string> res;
        for (auto i:ans) res.push_back(i);
        return res;
    }
};
