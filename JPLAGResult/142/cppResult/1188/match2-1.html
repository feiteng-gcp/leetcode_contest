class Solution {

    vector<string> Times(vector<string> a, vector<string> b) {
        vector<string> res;
        for (auto& sa : a) {
            for (auto& sb : b) {
                res.push_back(sa + sb);
            }
        }
        return res;
    }
    vector<string> Plus(vector<string> a, vector<string> b) {
        unordered_set<string> res;
        res.insert(a.begin(), a.end());
        res.insert(b.begin(), b.end());
        return {res.begin(), res.end()};
    }
public:
    vector<string> braceExpansionII(string s) {
        s += ',';
        int cnt = 0;
        set<string> res;
        vector<string> cur, last = {""};
        string word = "";
        for (int i=0; i<s.size(); i++) {
            if (s[i]>='a' && s[i]<='z') {
                word += s[i];
            } else {
                cur.push_back(word);
                word = "";
                last = Times(last, cur);
                cur = {};
                
                if (s[i] == '{') {
                    cnt += 1;
                    int j;
                    for (j=i+1; j<s.size(); j++) {
                        cnt += s[j]=='{' ? 1 : s[j]=='}' ? -1 : 0;
                        if (s[j]=='}' && cnt==0) break;
                    }
                    cur = braceExpansionII(s.substr(i+1, j-i-1));
                    last = Times(last, cur);
                    cur = {};
                    i = j;
                } else if (s[i] == ',') {
                    res.insert(last.begin(), last.end());
                    last = {""};
                }
            }
        }
        return {res.begin(), res.end()};
    }
};
