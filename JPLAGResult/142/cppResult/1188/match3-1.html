class Solution {
public:
    void generate(const vector<vector<string>> &sections, int index, string current, vector<string> &result) {
        if (index >= (int) sections.size()) {
            result.push_back(current);
            return;
        }

        for (string s : sections[index])
            generate(sections, index + 1, current + s, result);
    }

    vector<string> braceExpansionII(string expression) {
        // cout << expression.size() << ": " << expression << endl;

        if (expression.size() == 1) {
            assert(islower(expression[0]));
            return vector<string>(1, expression);
        }

        int n = expression.size();
        vector<vector<string>> sections;
        bool single = false;

        for (int i = 0, j = 0; i < n; i = j) {
            if (islower(expression[i])) {
                j = i + 1;
                sections.push_back(braceExpansionII(expression.substr(i, j - i)));
                continue;
            }


            j = i + 1;
            int count = 1;

            while (j < n && count > 0) {
                if (expression[j] == '{')
                    count++;
                else if (expression[j] == '}')
                    count--;

                j++;
            }

            if (i == 0 && j == n) {
                single = true;
                break;
            }

            sections.push_back(braceExpansionII(expression.substr(i, j - i)));
        }

        if (!single) {
            vector<string> result;
            generate(sections, 0, "", result);
            sort(result.begin(), result.end());
            result.resize(unique(result.begin(), result.end()) - result.begin());
            return result;
        }

        expression = expression.substr(1, n - 2);
        n -= 2;
        vector<string> result;

        vector<int> level(n, 0);

        for (int i = 0; i < n; i++) {
            level[i] = i == 0 ? 0 : level[i - 1];

            if (expression[i] == '{')
                level[i]++;
            else if (expression[i] == '}')
                level[i]--;
        }

        single = false;

        for (int i = 0, j = 0; i < n; i = j + 1) {
            j = i;

            while (j < n && (level[j] != 0 || expression[j] != ','))
                j++;

            if (i == 0 && j == n) {
                single = true;
                break;
            }

            vector<string> section = braceExpansionII(expression.substr(i, j - i));
            result.insert(result.end(), section.begin(), section.end());
        }

        if (single)
            return braceExpansionII(expression);

        sort(result.begin(), result.end());
        result.resize(unique(result.begin(), result.end()) - result.begin());
        return result;
    }
};
