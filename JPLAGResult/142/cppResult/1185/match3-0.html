/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int findInMountainArray(int target, MountainArray &mountainArr) {
	int N = mountainArr.length();
	VI H(N, -1);
	auto get = [&](int i) {
	    if (H[i] == -1) {
		H[i] = mountainArr.get(i);
	    }
	    return H[i];
	};

	int lo = 0, hi = N-1;
	while (hi - lo > 2) {
	    int mid = (lo + hi) / 2;
	    if (get(mid) < get(mid+1)) lo = mid;
	    else hi = mid;
	}

	int peak = lo;
	REP (t, 2) {
	    if (peak+1 < N && get(peak) < get(peak+1)) peak++;
	}
        
	int left, right;
	left = 0; right = peak;
	while (right - left > 1) {
	    int mid = (left + right) / 2;
	    if (get(mid) <= target) left = mid;
	    else right = mid;
	}
	if (get(left) == target) return left;


	left = peak; right = N;
	while (right - left > 1) {
	    int mid = (left + right) / 2;
	    if (get(mid) >= target) left = mid;
	    else right = mid;
	}
	if (get(left) == target) return left;

	return -1;
    }
};
