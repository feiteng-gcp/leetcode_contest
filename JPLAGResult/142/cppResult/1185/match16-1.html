/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int findpeak(MountainArray &aim)
    {
        int be = 0, en = aim.length()-1;
        while (be<en)
        {
            int mid = be + ((en-be)>>1);
            if (mid<aim.length()-1 && aim.get(mid)<aim.get(mid+1))
            {
                be = mid+1;
            }
            else
                en = mid;
        }
        return be;
    }

    int search(int l, int r, MountainArray &aim, int target)
    {
        int be = l, en = r;
        while (be<=en)
        {
            int mid = be+((en-be)>>1);
            int t = aim.get(mid);
            if (t == target)
                return mid;
            else
                if (t<target)
                    be = mid + 1;
                else 
                    en = mid - 1;
        }
        return -1;
    }
    
    int search1(int l, int r, MountainArray &aim, int target)
    {
        int be = l, en = r;
        while (be<=en)
        {
            int mid = be+((en-be)>>1);
            int t = aim.get(mid);
            if (t == target)
                return mid;
            else
                if (t<target)
                    en = mid - 1;
                else 
                    be = mid + 1;
        }
        return -1;
    }
    
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int n = mountainArr.length();
        int peak =  findpeak(mountainArr);
        cout<<"peak:"<<peak<<endl;
        int res = search(0,peak,mountainArr,target);
        cout<<"res1:"<<res<<endl;
        if (res == -1)
            res = search1(peak+1,n-1,mountainArr,target);
        return res;
    }
};
