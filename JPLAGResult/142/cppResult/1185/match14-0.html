/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int n = mountainArr.length();
        int peak = find_peak(mountainArr);
        cout << peak << endl;
        int left = find_left(mountainArr, target, 0, peak);
        if(left != -1) return left;
        int right = find_right(mountainArr, target, peak, n);
        return right;
    }
    
private:
    int find_peak(MountainArray &mountainArr) {
        int n = mountainArr.length();
        int low = 0, high = n;
        while(low < high) {
            int mid = low + (high - low) / 2;
            if(mid + 1 < n && mountainArr.get(mid + 1) > mountainArr.get(mid)) low = mid + 1;
            else high = mid;
        }
        return low;
    }
    
    int find_left(MountainArray &mountainArr, int target, int low, int high) {
        while(low < high) {
            int mid = low + (high - low) / 2;
            int val = mountainArr.get(mid);
            if(val == target) return mid;
            else if(val < target) low = mid + 1;
            else high = mid;
        }
        return -1;
    }
    
    int find_right(MountainArray &mountainArr, int target, int low, int high) {
        while(low < high) {
            int mid = low + (high - low) / 2;
            int val = mountainArr.get(mid);
            if(val == target) return mid;
            else if(val > target) low = mid + 1;
            else high = mid;
        }
        return -1;
    }
};
