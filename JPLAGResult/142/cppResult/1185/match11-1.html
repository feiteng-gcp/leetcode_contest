/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
    int findMid(MountainArray &mountainArr) {

        int l = 1, r = mountainArr.length() - 2;
        while (l != r) {
            int m = (l + r) >> 1;
            if (mountainArr.get(m) < mountainArr.get(m + 1)) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l;
    }

    int findl(MountainArray &mountainArr, int l, int r, int target) {

        while (l != r) {
            int m = (l + r) >> 1;
            if (mountainArr.get(m) < target) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return mountainArr.get(l) == target ? l : -1;
    }

    int findr(MountainArray &mountainArr, int l, int r, int target) {

        while (l != r) {
            int m = (l + r) >> 1;
            if (mountainArr.get(m) > target) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return mountainArr.get(l) == target ? l : -1;
    }
public:
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int mid = findMid(mountainArr);
        int x;
        if (~(x = findl(mountainArr, 0, mid, target))) return x;
        return findr(mountainArr, mid + 1, mountainArr.length() - 1, target);
    }
};
