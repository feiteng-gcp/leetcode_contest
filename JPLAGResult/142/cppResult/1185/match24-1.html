/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */

class Solution {
public:
    int findInMountainArray(int tgt, MountainArray &arr) {
        int n = arr.length();
        int lo = 0, hi = n-1, mid = 0;
        while ((hi - lo) >= 2) {
            mid = lo + (hi - lo)/2;
            int slp1 = arr.get(mid) - arr.get(mid-1);
            int slp2 = arr.get(mid+1) - arr.get(mid);    // slp stands for slope
            if (slp1 > 0 && slp2 < 0) 
                break;
            else if (slp1 > 0 && slp2 > 0) 
                lo = mid;
            else 
                hi = mid;
        }
        
        int idx = mid;
        
        // binary search (0, idx), then search (idx, n-1)
        lo = 0, hi = idx;
        while (lo <= hi) {

            mid = lo + (hi - lo)/2;
            int v = arr.get(mid);
            if (v == tgt) 
                return mid;
            else if (v < tgt) 
                lo = mid+1;
            else
                hi = mid-1;
        }
        
        // binary search (idx, n-1)
        lo = idx, hi = n-1;
        while (lo <= hi) {

            mid = lo + (hi - lo)/2;
            int v = arr.get(mid);
            if (v == tgt) 
                return mid;
            else if (v < tgt)
                hi = mid-1;
            else
                lo = mid+1;
        }
        return -1;       
    }
    
    
};
