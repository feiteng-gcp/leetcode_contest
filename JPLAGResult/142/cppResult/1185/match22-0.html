/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int findPos(MountainArray &mountainArr,int &maxE){
        int n = mountainArr.length();
        int left = 0, right = n - 1;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (mountainArr.get(mid) < mountainArr.get(mid + 1))
                left = mid + 1;
            else 
                right = mid;
        }
        maxE = mountainArr.get(right);
        return right;
        
    }
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int maxE = INT_MIN;
        int p = findPos(mountainArr,maxE);
        if(target > maxE)
            return -1;
        if(target == maxE)
            return p;
        int start = 0,end = p;
        while(start <= end){
            int mid = (start + end)/2;
            int t = mountainArr.get(mid);
            if(t == target)
                return mid;
            
            if(t > target){
                end = mid-1;
                
            }
            else{
                start = mid+1;
            }
        }
        
        
        start = p,end=mountainArr.length() -1;
        while(start <= end){
            int mid = (start + end)/2;
            int t = mountainArr.get(mid);
            
            if(t == target)
                return mid;
            
            
            if(t < target){
                end = mid-1;
            }
            else{
                start = mid+1;
            }
        }
        
        return -1;
    }
};
