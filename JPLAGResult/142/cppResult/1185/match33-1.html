class Solution {
public:
	int findInMountainArray(int target, MountainArray& mountainArr) {
		int n = mountainArr.length();

		int lo = 0, hi = n;
		while (lo + 1 != hi) {
			int mi = (lo + hi) / 2;
			if (mountainArr.get(mi) - mountainArr.get(mi - 1) > 0) lo = mi;
			else hi = mi;
		}
		int cent = lo;

		int A = mountainArr.get(0);
		int B = mountainArr.get(cent);
		int C = mountainArr.get(n-1);

		if (A == target) return 0;
		if (A < target and target < B) {

			lo = 0, hi = cent;
			while (lo + 1 != hi) {
				int mi = (lo + hi) / 2;
				if (target <= mountainArr.get(mi)) hi = mi;
				else lo = mi;
			}
			if (mountainArr.get(hi) == target) return hi;
		}
		if (target == B) return cent;
		if (C < target and target < B) {

			hi = cent, lo = n-1;
			while (hi + 1 != lo) {
				int mi = (lo + hi) / 2;
				if (target <= mountainArr.get(mi)) hi = mi;
				else lo = mi;
			}
			if (mountainArr.get(hi) == target) return hi;
		}
		if (target == C) return n-1;
		return -1;
	}
};
