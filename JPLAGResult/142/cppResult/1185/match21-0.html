/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int i;
		int N = mountainArr.length();
		
		int S = 1;
		int E = N - 2;
		
		while(S < E)
		{
			int C = ((S + E + 1) / 2);
			
			if(mountainArr.get(C) < mountainArr.get(C - 1))
			{
				E = C - 1;
			}
			else
			{
				S = C;
			}
		}
		
		int Temp = E;
		S = 0;
		
		while(S < E)
		{
			int C = ((S + E) / 2);
			
			if(mountainArr.get(C) >= target)
			{
				E = C;
			}
			else
			{
				S = C + 1;
			}
		}
		
		if(mountainArr.get(S) == target)
		{
			return S;
		}
		else
		{
			S = Temp;
			E = N - 1;
			
			while(S < E)
			{
				int C = ((S + E) / 2);
			
				if(mountainArr.get(C) > target)
				{
					S = C + 1;
				}
				else
				{
					E = C;
				}
			}
			
			if(mountainArr.get(S) == target)
			{
				return S;
			}
			else
			{
				return -1;
			}
		}
    }
};
