/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */

class Solution {
public:
    int findInMountainArray(int t, MountainArray &a) {
        int n = a.length();
        int l = 0, r = n - 2;
        while(l <= r) {
            int m = (l + r) / 2;
            if(a.get(m) < a.get(m+1)) {
                l = m + 1;
            } else {
                r = m - 1;
            }
        }
        int i = l;
        l = 0, r = i;
        while(l <= r) {
            int m = (l + r) / 2;
            int x = a.get(m);
            if(x == t) {
                return m;
            }

            if(x < t) {
                l = m + 1;
            } else {
                r = m - 1;
            }
        }
        l = i, r = n-1;
        while(l <= r) {
            int m = (l + r) / 2;
            int x = a.get(m);
            if(x == t) {
                return m;
            }

            if(x > t) {
                l = m + 1;
            } else {
                r = m - 1;
            }
        }
        return -1;
    }
};
