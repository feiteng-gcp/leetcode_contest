/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
private:
    vector<bool> vis = vector<bool>(10005, false);
    vector<int> arr = vector<int>(10005, 0);
    int getarr(int pos, MountainArray &mountainArr) {
        if (!vis[pos])
            arr[pos] = mountainArr.get(pos);
        return arr[pos];
    }
public:
    
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int len = mountainArr.length();
        int l = 0, r = len - 1, m;
        while (l < r) {
            m = (l + r) >> 1;
            if (getarr(m, mountainArr) > getarr(m + 1, mountainArr)) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        int peak = l;
        // cout << peak << endl;
        l = 0, r = peak;
        while (l <= r) {
            m = (l + r) >> 1;
            int cur = getarr(m, mountainArr);
            // cout << m << " " << cur << endl;
            // cout << l << " " << r << endl;
            if (cur == target) {
                // cout << "1: " << m << endl;
                return m;
            } else if (cur > target) {
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        l = peak + 1;
        r = len - 1;
        while (l <= r) {
            m = (l + r) >> 1;
            int cur = getarr(m, mountainArr);
            if (cur == target) {
                // cout << "2: " << m << endl;
                return m;
            } else if (cur > target) {
                l = m + 1;
            } else {
                r = m - 1;
            }
        }
        return -1;
    }
};
