/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int findleft(MountainArray &mountainArr, int l, int r, int target){
        int L = l, R = r;
        while(L <= R){
            int mid = (L + R) / 2;
            int midnum = mountainArr.get(mid);
            if(midnum == target) return mid;
            else if (midnum < target) L = mid + 1;
            else R = mid - 1;
        }
        return -1;
    }
    int findright(MountainArray &mountainArr, int l, int r, int target){
        int L = l, R = r;
        while(L <= R){
            int mid = (L + R) / 2;
            int midnum = mountainArr.get(mid);
            if(midnum == target) return mid;
            else if (midnum < target) R = mid - 1;
            else L = mid + 1;
        }
        return -1;
    }
    int findInMountainArray(int target, MountainArray &mountainArr) {
        //find the peak num index
        int len = mountainArr.length();
        int l = 0, r = len - 1;
        while(l <= r){
            int mid = (l + r) / 2;
            int num = mountainArr.get(mid);
            if(mid + 1 < len){
                if(mountainArr.get(mid + 1) > num)l = mid + 1;
                else r = mid - 1;
            }
        }
        
        int middle = l;
        cout<<middle<<endl;
        
        int ret = findleft(mountainArr, 0, l, target);
        if(ret != -1) return ret;
        else return  findright(mountainArr, l + 1, len - 1, target);
        
    }
};
