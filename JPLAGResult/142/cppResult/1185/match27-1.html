/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */

class Solution {
public:
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int L=0, R=mountainArr.length();
        while(L+1<R) {
            int M=(L+R)/2;
            int K=(M+R)/2;
            if (mountainArr.get(M)>mountainArr.get(K)) {
                R=K;
            }
            else L=M;
        }
        int top=L;
        for(int i=L-1; i<=L+1; i++) {
            if (mountainArr.get(i)>mountainArr.get(top)) top=i;
        }
        
        cout<<top<<endl;
        
        if (mountainArr.get(top)==target) return top;
        L=0, R=top;

        while(L<R) {
            int M=(L+R)/2;
            int v=mountainArr.get(M);
            if (v==target) return M;
            if (v<target) L=M+1;
            else R=M-1;
        }
        if (mountainArr.get(L)==target) return L; 
        
        L=top+1, R=mountainArr.length()-1;

        while(L<R) {
            int M=(L+R)/2;
            int v=mountainArr.get(M);
            if (v==target) return M;
            if (v<target) R=M-1;
            else L=M+1;
        }
        if (mountainArr.get(L)==target) return L; 
        
        return -1;
    }
};
