/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int get(int i,MountainArray &mountainArr,unordered_map<int,int> &cache)
    {
        if(cache.find(i)!=cache.end()){
            return cache[i];
        }
        cache[i]=mountainArr.get(i);
        return cache[i];
    }
    int findInMountainArray(int target, MountainArray &mountainArr) {
        
        unordered_map<int,int> cache;
        
        int n=mountainArr.length();
        
        int l=1,u=n-2;
        int mi=1;
        while(u>=l){
            int mid=(u+l)/2;
            int m1=get(mid-1,mountainArr,cache);
            int m2=get(mid,mountainArr,cache);
            int m3=get(mid+1,mountainArr,cache);
            
            if(m2>m1 && m2>m3){
                mi=mid;
                break;
            }else if(m3>m2){
                l=mid+1;
            }else{
                u=mid-1;
            }
        }
        
        if(get(mi,mountainArr,cache)==target) return mi;
        
        l=0,u=mi;
        while(u>=l){
            int mid=(u+l)/2;
            int m2=get(mid,mountainArr,cache);
            if(m2==target){
                return mid;
            }else if(m2>target){
                u=mid-1;
            }else{
                l=mid+1;
            }
        }
        
        l=mi,u=n-1;
        while(u>=l){
            int mid=(u+l)/2;
            int m2=get(mid,mountainArr,cache);
            if(m2==target){
                return mid;
            }else if(m2<target){
                u=mid-1;
            }else{
                l=mid+1;
            }
        }
        
        return -1;
    }
};
