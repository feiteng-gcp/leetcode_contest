/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:

    int findInMountainArray(int target, MountainArray &inp) {
        int n = inp.length();
        int lo = 0, hi = n - 2;
        while(lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if(inp.get(mid) < inp.get(mid + 1)) lo = mid + 1;
            else hi = mid - 1;
        }
        int pt = lo;

        lo = 0, hi = pt;
        while(lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            int cur = inp.get(mid);
            if(cur == target) return mid;
            else if(cur > target) hi = mid - 1;
            else lo = mid + 1;
        }
        hi++;
        if(hi <= pt && inp.get(hi) == target) return hi;

        lo = pt + 1, hi = n - 1;
        while(lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            int cur = inp.get(mid);
            if(cur == target) return mid;
            else if(cur > target) lo = mid + 1;
            else hi = mid - 1;
        }
        return -1;
    }
};
