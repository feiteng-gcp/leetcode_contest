/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */
class Solution {
public:
    int findPeak(MountainArray& mountainArr, int left, int right) {
        while(left < right) {
            int mid = (left + right) >> 1;
            int l = mountainArr.get(mid - 1), m = mountainArr.get(mid), r = mountainArr.get(mid + 1);
            if(m > l && m > r) return mid;
            else if(m > l && m < r) {
                left = mid + 1;
            }
            else {
                right = mid - 1;
            }
        }
        return left;
    }
    
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int n = mountainArr.length();

        int pid = findPeak(mountainArr, 1, n - 2);
        // cout << pid << endl;
        int peak = mountainArr.get(pid);
        if(target > peak) return -1;
        if(target == peak) return pid;
        int left = 0, right = pid - 1;
        while(left <= right) {
            int mid = (left + right) >> 1;
            int m = mountainArr.get(mid);
            if(m == target) return mid;

            else if(m > target) {
                right = mid - 1;
            }
            else {
                left = mid + 1;
            }
        }
        left = pid + 1, right = n - 1;
        while(left <= right) {
            int mid = (left + right) >> 1;
            int m = mountainArr.get(mid);
            if(m == target) return mid;

            else if(m < target) {
                right = mid - 1;
            }
            else {
                left = mid + 1;
            }
        }
        return -1;
    }
};
