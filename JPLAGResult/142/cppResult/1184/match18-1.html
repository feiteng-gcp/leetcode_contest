class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector < pair <int, pair <int, int> > > t;
        // for (int x = 0; x < trips.size(); x++) {
        //     t.push_back (make_pair (trips[x][1], make_pair (trips[x][0], trips[x][2])));
        // }
//         sort (t.begin(), t.end());

        int psum[1005]={0};

        for (int x = 0; x < trips.size(); x++) {
            psum[trips[x][1]] += trips[x][0];
            psum[trips[x][2]] -= trips[x][0];
        } 
        for (int x = 1; x <= 1001; x++) {
            psum[x] += psum[x-1];
            if (psum[x] > capacity) {
                return false;
            }
        } 
        return true;
    }
};
