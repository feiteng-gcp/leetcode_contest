class Solution {
public:
    bool carPooling(vector<vector<int>>& t, int c) {
        vector<int> cc(1010, 0);
        for (auto v : t) {
            cc[v[1]] += v[0];
            cc[v[2]] -= v[0];
        }
        for (int i = 0; i < 1010; ++i) {
            if (i) cc[i] += cc[i - 1];
            if (cc[i] > c) return false;
        }
        return true;
    }
};
