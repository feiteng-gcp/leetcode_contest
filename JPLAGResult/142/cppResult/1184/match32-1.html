class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<int> cnt(1001,0);
        for(auto v:trips){
            cnt[v[1]]+=v[0];
            cnt[v[2]]-=v[0];
        }
        
        int c=capacity;
        for(int i=0;i<=1000;i++){
            c-=cnt[i];
            if(c<0) return false;
        }
        return true;
    }
};
