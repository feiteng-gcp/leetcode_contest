class Solution {
public:
    bool carPooling(vector<vector<int>>& t, int capacity) {
        int n = t.size();
        vector<pair<int, int> > a;
        for (int i = 0; i < n; ++i) {
            a.push_back(make_pair(t[i][1], -t[i][0]));
            a.push_back(make_pair(t[i][2], t[i][0]));
        }
        sort(a.begin(), a.end(), [](pair<int, int> p1, pair<int, int> p2) {
           if (p1.first != p2.first) return p1.first < p2.first;
            return p1.second > p2.second;
        });
        int c = capacity;
        //cout << "start" << endl;
        for (int i = 0; i < n * 2; ++i) {
            c += a[i].second;
            //cout << c << endl;
            if (c < 0) return false;
        }
        return true;
    }
};
