class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<int> m(1001, 0);
        for (auto trip : trips) {
            for (int i = trip[1]; i < trip[2]; i++) {
                m[i] += trip[0];
                if (m[i] > capacity)
                    return false;
            }
        }
        return true;
    }
};
