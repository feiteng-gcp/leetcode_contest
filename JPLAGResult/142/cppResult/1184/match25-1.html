class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        map<int, int> m;
        for (const auto& v: trips) {
            m[v[1]] += v[0];
            m[v[2]] -= v[0];

        }
        int cnt = 0;
        for (const auto& p: m) {
            cnt += p.second;
            if (cnt > capacity) {
                return false;
            }
        }
        return true;
    }
};
