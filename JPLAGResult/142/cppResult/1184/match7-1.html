class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        map<int, int> need;
        
        for (auto &t : trips) {
            need[t[1]] += t[0];
            need[t[2]] -= t[0];
        }
        
        int nn = 0;
        for (auto &n : need) {
            nn += n.second;
            if (nn > capacity) return false;
        }
        
        return true;
    }
};
