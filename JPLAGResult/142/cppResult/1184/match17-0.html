class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<int> pts(1001);
        for (int i=0; i<trips.size(); i++) {
            pts[trips[i][1]] -= trips[i][0];
            pts[trips[i][2]] += trips[i][0];
        }
        
        for (int i=0; i<1001; i++) {
            capacity += pts[i];
            if (capacity < 0) {
                return false;
            }
        }
        
        return true;
    }
};
