class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        using PII = pair<int, int>;
        vector< PII> zz;
        for (auto g: trips) {
            zz.push_back( { g[1], g[0]});
            zz.push_back( { g[2], -g[0]});
        }
        sort(zz.begin(), zz.end());
        int now = 0;
        for (auto g: zz) {
            now += g.second;
            if (now>capacity) return false;
        }
        return true;
    }
};
