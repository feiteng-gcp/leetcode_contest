class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<int> num(1000);
        
        for (auto trip : trips) {
            for (int i = trip[1]; i < trip[2]; i++) {
                num[i] += trip[0];
                if (num[i] > capacity) {
                    return false;
                }
            }
        }
        return true;
    }
};
