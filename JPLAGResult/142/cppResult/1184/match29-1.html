class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<vector<int>> distance(1001, vector<int>(2, 0));
        for (auto trip : trips) {
            distance[trip[1]][0] += trip[0];
            distance[trip[2]][1] += trip[0];
        }

        int passenger = 0;
        for (int i = 0; i < distance.size(); i++) {
            const vector<int>& tmp = distance[i];
            if (passenger + tmp[0] - tmp[1] > capacity)
                return false;
            passenger += (tmp[0] - tmp[1]);
        }
        return true;
    }
};
