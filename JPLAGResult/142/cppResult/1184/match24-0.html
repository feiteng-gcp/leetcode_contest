class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        int L = 1000 + 5;
        vector<int> dp;
        dp.resize(L);
        for (int i = 0; i < trips.size(); i++) {
            for(int j = trips[i][1]; j < trips[i][2]; j++){
                dp[j] += trips[i][0];
            }
        }
        for (int i = 0; i < L; i++){
            if(dp[i] > capacity)
                return false;
        }
        return true;
        
    }
};
