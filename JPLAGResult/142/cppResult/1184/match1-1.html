class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int cap) {
        map<int, int> mp;
        for (auto & v : trips) {
            mp[v[1]] += v[0];
            mp[v[2]] -= v[0];
        }
        int cur = 0;
        for (auto & e : mp) {
            cur += e.second;
            if (cur > cap)
                return false;
        }
        return true;
    }
};
