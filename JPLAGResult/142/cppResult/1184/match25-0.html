class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<pair<int, int>> d_p;
        for (const auto& trip : trips) {
            d_p.emplace_back(trip[0], trip[1]);
            d_p.emplace_back(-trip[0], trip[2]);
        }
        sort(d_p.begin(), d_p.end(), [](const pair<int, int>& p1, const pair<int, int>& p2) {
            if (p1.second != p2.second) {
                return p1.second < p2.second;
            }
            return p1.first < p2.first;
        });
        int sum = 0;
        for (const auto& p : d_p) {
            sum += p.first;
            if (sum > capacity) {
                return false;
            }
        }
        return true;
    }
};
