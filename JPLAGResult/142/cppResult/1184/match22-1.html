class Solution {
public:
  bool carPooling(vector<vector<int>>& trips, int capacity) {
    int n = trips.size();
    vector<pair<int, int>> p(n * 2);

    for (int i = 0; i < n; ++i) {
      p[i * 2] = make_pair(trips[i][1], trips[i][0]);
      p[i * 2 + 1] = make_pair(trips[i][2], -trips[i][0]);
    }
    sort(p.begin(), p.end());
    int cnt = 0;
    for (int i = 0; i < n * 2; ++i) {
      cnt += p[i].second;
      //cout << p[i].first << " " << cnt << endl;
      if (cnt > capacity) return false;
    }
    return true;
  }
};
