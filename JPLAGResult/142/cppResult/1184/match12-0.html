class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        map<int, int> dict;
        for(auto trip:trips){
            dict[trip[1]] += trip[0];
            dict[trip[2]] -= trip[0];
        }
        int cur = 0;
        for(auto item:dict){
            cur += item.second;
            if(cur > capacity) return false;
        }
        return true;
    }
};
