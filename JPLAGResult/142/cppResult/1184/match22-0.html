class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        int a[1005];
        memset(a,0,sizeof(a));
        for(int i=0;i<trips.size();i++)
        {
            a[trips[i][1]]+=trips[i][0];
            a[trips[i][2]]-=trips[i][0];
        }
        int sum=0;
        for(int i=0;i<1005;i++)
        {
            sum+=a[i];
            if(sum>capacity)
                return false;
        }
        return true;
    }
};
