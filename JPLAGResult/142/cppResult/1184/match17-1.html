class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        
        vector<int> balance(1001,0);
        
        for(int i=0;i<trips.size();i++) {
            balance[trips[i][1]]+=trips[i][0];
            balance[trips[i][2]]-=trips[i][0];
        }
        
        int net = 0;

        for(int i=0;i<=1000;i++) {
            net+=balance[i];
            if(net > capacity) {
                return false;
            }
        }
        
        return true;
    }
};
