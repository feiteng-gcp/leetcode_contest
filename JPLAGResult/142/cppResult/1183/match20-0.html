class Solution {
public:
    int find_k(vector<int>& count, int k) {
        int n = 0;
        for (int i = 0; i < count.size(); i++) {
            n += count[i];
        }
        int cur = 0;
        for (int i = 0; i < count.size(); i++) {
            cur += count[i];
            if (cur >= k) {
                return i;
            }
        }
        return -1;
    }
    
    vector<double> sampleStats(vector<int>& count) {
        double minimum = 0, maximum = 0, mean = 0, median = 0, mode = 0;
        for (int i = 0; i < count.size(); i++) {
            if (count[i]) {
                minimum = i;
                break;
            }
        }
        for (int i = count.size() - 1; i >= 0; i--) {
            if (count[i]) {
                maximum = i;
                break;
            }
        }
        double sum = 0;
        int n = 0;
        for (int i = 0; i < count.size(); i++) {
            sum += count[i] * i;
            n += count[i];
        }
        mean = sum / n;
        if (n % 2 == 0) {
            median = (find_k(count, n / 2) + find_k(count, n / 2 + 1)) / 2.0;
        }
        else {
            median = find_k(count, n / 2);
        }
        int mode_count = 0;
        for (int i = 0; i < count.size(); i++) {
            if (count[i] > mode_count) {
                mode = i;
                mode_count = count[i];
            }
        }
        vector<double> ans = {minimum, maximum, mean, median, mode};
        return ans;
    }
};
