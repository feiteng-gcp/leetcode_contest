class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ans = {1000, -1, 0, 0, -1};

        int n = 256, mx = 0;
        for(int i = 0; i < n; i++) {
            if(count[i] > 0 && ans[0] > i)
                ans[0] = i;
            if(count[i] > 0 && ans[1] < i)
                ans[1] = i;
            ans[2] += (i * count[i]);
            ans[3] += count[i];
            if(count[i] > mx) {
                mx = count[i];
                ans[4] = i;
            }
        }
        ans[2] = ans[2] / ans[3];
        if((int)ans[2] % 2) {

            int lsum = 0;
            for(int i = 0; i < n; i++) {
                lsum = lsum + count[i];
                if(lsum >= (ans[3] / 2)) {
                    ans[3] = i;
                    break;
                }
            }
        } else {
            int m1 = -1, m2 = -1, lsum = 0;
            for(int i = 0; i < n; i++) {
                lsum = lsum + count[i];
                if(m1 == -1 && lsum >= (ans[3] / 2))
                    m1 = i;
                if(m2 == -1 && lsum >= (ans[3] / 2 + 1))
                    m2 = i;
            }
            ans[3] = double(m1 + m2) / 2.0; 
        }
        return ans;
    }
};
