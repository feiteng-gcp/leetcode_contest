class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        double min_ = 256, max_ = -1, sum = 0, median = 0, mode = 0, modeOccur = 0;
        int cnt = 0;
        for (int i = 0; i < 256; ++i) {
            min_ = min(min_, count[i] ? i : min_);
            max_ = max(max_, count[i] ? i : max_);
            cnt += count[i];
            sum += i * count[i];
            if (count[i] >= modeOccur) {
                modeOccur = count[i];
                mode = i;
            }
        }
        double currCount = 0;
        int prevNumber = 0;
        int midCount = (int)(cnt / 2);

        for (int i = 0; i < 256; ++i) {
            currCount += count[i];
            // cout << currCount << endl;
            // cout << "asdasd " << (cnt & 1) << endl;
            if ((cnt & 1) == 0) {
                // cout << "currCount " << currCount << endl;
                // even
                if (currCount - count[i] == midCount && currCount > midCount) {
                    // cout << prevNumber << " " << i << endl;
                    median = (prevNumber + i) / 2.0;
                } else if (currCount - count[i] < midCount && currCount > midCount) {
                    median = i;
                }
            } else {
                // odd
                if (currCount - count[i] < midCount + 1 && currCount >= midCount + 1) {
                    median = i;
                }
            }
            if (count[i]) prevNumber = i;
        }
        return {min_, max_, sum / cnt, median, mode};
    }
};
