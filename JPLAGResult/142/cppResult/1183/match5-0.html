class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ans(5);
        
        for (int i = 0; i < count.size(); i++) {
            if (count[i] != 0) {
                ans[0] = i;
                break;
            }
        }
        
        for (int i = count.size() - 1; i >= 0; i--) {
            if (count[i] != 0) {
                ans[1] = i;
                break;
            }
        }
        
        long long sum = 0;
        long long num = 0;
        for (int i = 0; i < count.size(); i++) {
            sum += i * count[i];
            num += count[i];
        }
        ans[2] = (double) sum / (double) num;
        
        long long cur = 0;
        for (int i = 0; i < count.size(); i++) {
            if ((cur + count[i]) * 2 > num) {
                ans[3] = i;
                break;
            } else if ((cur + count[i]) * 2 == num) {
                for (int j = i + 1; j < count.size(); j++) {
                    if (count[j] == 0) {
                        continue;
                    }
                    ans[3] = (i + j) / 2.0;
                    break;
                }
                break;
            }
            cur += count[i];
        }
        
        int maxp = -1, maxn = -1;
        for (int i = 0; i < count.size(); i++) {
            if (maxn < count[i]) {
                maxn = count[i];
                maxp = i;
            }
        }
        ans[4] = maxp;
        
        return ans;
    }
};
