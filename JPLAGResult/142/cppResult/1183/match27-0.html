class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        long long min_val = 0, max_val = 255, avg = 0, most = 0;
        double mid = 0;
        long long most_num = 0;
        long long num = 0;
        
        for(int i = 0; i < 256; i++){
            num += count[i];
            if(count[i] != 0){
                max_val = i;
                if(count[i] > most_num){
                    most_num = count[i];
                    most = i;
                }
            }
            avg += count[i]*i;
        }
        
        for(int i = 255; i >= 0; i--){
            if(count[i] != 0){
                min_val = i;
            }
        }
        if(num%2 == 0){
            // even numbers
            int pre = 0;
            for(int i = 0; i < 256; i++){
                pre += count[i];
                if(pre >= num/2){
                    mid += i;
                    break;
                }
            }
            pre = 0;
            for(int i = 0; i < 256; i++){
                pre += count[i];
                if(pre >= num/2 + 1){
                    mid += i;
                    break;
                }
            }
            mid /= 2.0;
        }else{
            int pre = 0;
            for(int i = 0; i < 256; i++){
                pre += count[i];
                if(2*pre + 1 >= num){
                    mid = i;
                    break;
                }
            }
        }
        
        return {min_val, max_val, 1.0*avg/num, mid, most};
    }
};
