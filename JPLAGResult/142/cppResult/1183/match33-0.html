class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        vector<double> res(5, 0);
        int n = count.size();
        for (int i = 0; i < n; ++i) {
            if (count[i]) {
                res[0] = i;
                break;
            }
        }
        
        for (int i = n - 1; i; --i) {
            if (count[i]) {
                res[1] = i;
                break;
            }
        }
        int cnt = 0;
        int mode = 0;
        for (int i = 0; i < n; ++i) {
            if (count[i] >= mode) {
                mode = count[i];
                res[4] = i;
            }
            cnt += count[i];
            res[2] += (i * count[i]);
        }
        res[2] = res[2] * 1.0 / cnt;
        int target = 0;
        bool isOdd = (cnt % 2);
        for (int i = 0; i < n; ++i) {
            if (target + count[i] < cnt >> 1) {
                target += count[i];
            } else if (target + count[i] == cnt >> 1 && !isOdd) {
                int j = i + 1;
                while (count[j] == 0) {
                    ++j;
                }
                res[3] = (i + j) / 2.0;
                break;
            } else {
                res[3] = i;
                break;
            }
        }
        return res;
    }
};
