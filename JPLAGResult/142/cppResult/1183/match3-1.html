class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        double minD,maxD,avrD,midD,mostD;
        for(int i =0;i<count.size();i++){
            if(count[i]>0){
                minD = i;
                break;
            }
        }
        
        for(int i =count.size() -1;i>=0;i--){
            if(count[i]>0){
                maxD = i;
                break;
            }
        }
        
        int num=0;

        double sumD = 0;
        for(int i =0;i<count.size();i++){
            if(count[i]>0){
                num += count[i];
                sumD += i * count[i];
            }
        }
        avrD = sumD / num;
        
        if(num % 2){
            int mid = num/2 + 1;
            int now = 0;
            for(int i =0;i<count.size();i++){
                if(now + count[i]>=mid){
                    midD = i;
                    break;
                }
                now += count[i];
            }
        }
        else{
            int mid1 = num/2;
            int mid2 = num/2+1;
            int now = 0;
            for(int i =0;i<count.size();i++){
                if(now + count[i]>=mid1){
                    mid1 = i;
                    break;
                }
                now += count[i];
            }
            now = 0;
            for(int i =0;i<count.size();i++){
                if(now + count[i]>=mid2){
                    mid2 = i;
                    break;
                }
                now += count[i];
            }
            midD = double(mid1 + mid2)/2;
        }
        
        int most = 0;
        for(int i =0;i<count.size();i++){
            if(count[i]>most){
                most = count[i];
                mostD = i;
            }
        }
        
        vector<double> ret;
        ret.push_back(minD);
        ret.push_back(maxD);
        ret.push_back(avrD);
        ret.push_back(midD);
        ret.push_back(mostD);
        
        return ret;
    }
};
