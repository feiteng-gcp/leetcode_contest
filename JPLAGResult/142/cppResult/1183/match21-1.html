class Solution {
public:
    
    int getIth(vector<int> &count, int k){
        int n=256;
        for (int i=0; i<n; i++){
            k-=count[i];
            if (k<=0) return i;
        }
        return 0;
    }
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ans; ans.clear();
        int n=256;
        
        long long sum=0;
        int mx=0, mi=256, siz=0;
        for (int i=0; i<n; i++){
            sum+=(long long)count[i]*(long long)i;
            siz+=count[i];
            if (count[i]>0) mx=max(mx, i);
            if (count[i]>0) mi=min(mi, i);
        }
        
        int showmx=0;
        for (int i=0; i<n; i++){
            if (count[showmx]<count[i]) showmx=i;
        }
        
        ans.push_back(mi);
        ans.push_back(mx);
        ans.push_back((double)sum/siz);
        if (siz&1) ans.push_back(getIth(count, siz/2+1)); else ans.push_back((double)(getIth(count, siz/2) + getIth(count, siz/2+1))/2.0);
        ans.push_back(showmx);
        
        return ans;
    }
};
