class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ans(5);

        for (int i = 0; i < count.size(); i++) {
            if (count[i]) {
                ans[0] = i;
                break;
            }
        }

        for (int i = count.size() - 1; i >= 0; i--) {
            if (count[i]) {
                ans[1] = i;
                break;
            }
        }

        unsigned long sum = 0, n = 0;
        for (int i = 0; i < count.size(); i++) {
            sum += i * count[i];
            n += count[i];
        }
        ans[2] = (double)sum / (double)n;


        int mode, modeN = INT_MIN;
        for (int i = 0; i < count.size(); i++) {
            if (count[i] > modeN) {
                modeN = count[i];
                mode = i;
            }
        }
        ans[4] = mode;


        if (n % 2 == 1) {
            unsigned long target = n / 2 + 1;
            unsigned long t = 0;
            for (int i = 0; i < count.size(); i++) {
                if (t < target && t + count[i] >= target) {
                    ans[3] = i;
                    break;
                }
                t += count[i];
            }
        }
        else {
            unsigned long target1 = n / 2, target2 = target1 + 1;
            unsigned long t = 0;
            double a1 = -1, a2 = -1;
            for (int i = 0; i < count.size(); i++) {
                if (t < target1 && t + count[i] >= target1) {
                    a1 = i;
                    break;
                }
                t += count[i];
            }
            t = 0;
            for (int i = 0; i < count.size(); i++) {
                if (t < target2 && t + count[i] >= target2) {
                    a2 = i;
                    break;
                }
                t += count[i];
            }
            ans[3] = (a1 + a2) / 2;
        }
        return ans;
    }
};
