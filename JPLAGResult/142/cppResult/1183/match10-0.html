class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        int len=count.size();
        long long sum=0;
        long long suma=0;
        int mi=-1;
        int ma=-1;
        int z=0;
        int c=0;
        for(int i=0;i<len;i++)
        {
            if(mi==-1&&count[i]!=0)
                mi=i;
            if(ma<i&&count[i]!=0)
                ma=i;             
            sum+=count[i];
            suma+=count[i]*i;
            if(count[i]>c)
            {
                c=count[i];
                z=i;
            }
        }
        vector<double>ans;
        ans.push_back(mi*1.0);
        ans.push_back(ma*1.0);
        ans.push_back((suma*1.0)/sum);
        if(sum%2==0)
        {
            int t=0;
            int tt=0;
            int f1=-1;
            int f2=-1;
            for(int i=0;i<len;i++)
            {
                t+=count[i];
                if(t>=sum/2&&f1==-1)
                {
                    f1=0;
                    tt+=i;
                }
                if(t>=sum/2+1&&f2==-1)
                {
                    f2=0;
                    tt+=i;
                }
            }
            ans.push_back(tt*1.0/2);
        }
        else
        {
            int t=0;
            int tt=0;
            for(int i=0;i<len;i++)
            {
                t+=count[i];
                if(t>=sum/2+1)
                {
                    tt+=i;
                    break;
                }
            }
            ans.push_back(tt*1.0);
        }
        ans.push_back(z*1.0);
        return ans;
        
        
    }
};
