class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        double mx = -1, mn = -1, sum = 0, modCnt = 0, mod = -1;
        
        int allCnt = 0;
        double med = 0;
        

        for(int i = 0; i < count.size(); i ++) allCnt += count[i];
        
        int targetCnt = (allCnt+1)/2;
        int curCnt = 0;
        
        for(int i = 0; i < count.size(); i ++)
        {
            sum += i*count[i];
            
            if(count[i] != 0)
            {
                if(mn == -1) mn = i;
                mx = i;
            }
            
            if(count[i] > modCnt)
            {
                mod = i;
                modCnt = count[i];
            }
            
            
            if(count[i] != 0)
            {
                if(curCnt == targetCnt) med += i;

                if(curCnt < targetCnt)
                {
                    if(curCnt+count[i] == targetCnt)
                    {
                        med += i;
                    }
                    else if(curCnt+count[i] > targetCnt)
                    {
                        med += 2*i;
                    }
                }
                curCnt += count[i];
            }
        }
        
        return {mn, mx, sum/allCnt, med/2, mod};
    }
};
