class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ans;
        ans.push_back(findMin(count));
        ans.push_back(findMax(count));
        ans.push_back(calMean(count));
        ans.push_back(calMedian(count));
        ans.push_back(findMode(count));
        return ans;
    }
    
    double findMax(vector<int>& count) {
        int n=count.size()-1;
        while(n>=0 && count[n]==0) --n;
        return n;
    }
    
    double findMin(vector<int>& count) {
        int n=0;
        while(n<count.size() && count[n]==0) ++n;
        return n;
    }
    
    double calMean(vector<int>& count) {
        long long sum=0;
        int total=0;
        for(int i=0;i<count.size();++i) {
            total+=count[i];
            sum+=(long long)count[i]*i;

        }
        return 1.0*sum/total;
    }
    
    double calMedian(vector<int>& count) {
        int total=0;
        for(int i=0;i<count.size();++i) {
            total+=count[i];
        }
        if(total&1) {
            int k=(total>>1)+1;
            for(int i=0;i<count.size();++i) {
                k-=count[i];
                if(k<=0) return i;
            }
        } else {
            int ans=0;
            int k=(total>>1);
            for(int i=0;i<count.size();++i) {
                k-=count[i];
                if(k<=0) {ans+=i;break;}
            }
            k=(total>>1)+1;
            for(int i=0;i<count.size();++i) {
                k-=count[i];
                if(k<=0) {ans+=i;break;}
            }
            return ans/2.0;
        }
        return 0.0;
    }
    
    double findMode(vector<int>& count) {

        int maxn=0, maxv=0;
        for(int i=0;i<count.size();++i) {
            if(count[i]>maxn) {
                maxn=count[i];
                maxv=i;
            }
        }
        return maxv;
    }

};
