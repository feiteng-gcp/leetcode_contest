class Solution {
public:
	vector<double> sampleStats(vector<int>& count) {
		vector<double> res;
		res.push_back(getMin(count));
		res.push_back(getMax(count));
		res.push_back(getAv(count));
		res.push_back(getMid(count));
		res.push_back(getMost(count));
		return res;
	}
	double getMin(vector<int>& count) {
		for (int i = 0; i <= 255; i++) {
			if (count[i] != 0) return double(i);
		}
        return 0;
	}
	double getMax(vector<int>& count) {
		for (int i = 255; i >= 0; i--) {
			if (count[i] != 0) return double(i);
		}
        return 0;
	}
	double getAv(vector<int>& count) {
		int n = 0;
		double sum = 0;
		for (int i = 255; i >= 0; i--) {
			sum += count[i] * i;
			n += count[i];
		}
		return sum / n;
	}
	double kth(vector<int>& count,int n) {
		int num = 0;
		for (int i = 0; i <= 255; i++) {
			num += count[i];
			if (num >= n) {
				return double(i);
			}
		 }
        return 0;
	}
	double getMid(vector<int>& count) {
		int sum = 0;
		for (int i = 0; i <= 255; i++) {
			sum += count[i];
		}
		if (sum % 2 == 1) {
			return kth(count, sum / 2 + 1);
		}
		else {
			return (kth(count, sum / 2) + kth(count, sum / 2 + 1))*0.5;
		}
	}
	double getMost(vector<int>& count) {
		int max = 0;
		double res = 0;
		for (int i = 0; i <= 255; i++) {
			if (count[i] >= max) {
				res = i;
				max = count[i];
			}
		}
		return res;
	}
};
