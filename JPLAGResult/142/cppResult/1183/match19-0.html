class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        int len = 256;
        vector<double> ans(5);
        double sum = 0;
        int max_num = count[0];
        ans[4] = 0.0;
        for (int i = 0; i < len; i++) {
            if (count[i]) {
                ans[0] = i;
                break;
            }
        }
        for (int i = len - 1; i >= 0; i--) {
            if (count[i]) {
                ans[1] = i;
                break;
            }
        }
        for (int i = 1; i < len; i++) {
            if (count[i] > max_num) {
                max_num = count[i];
                ans[4] = i;
            }
            sum += count[i] * i;
            count[i] += count[i - 1];
        }
        ans[2] = sum / count.back();
        int total = count.back();
        if (total % 2) {
            int k = total / 2 + 1;
            auto it = upper_bound(count.begin(), count.end(), k);
            it--;
            int idx = distance(count.begin(), it);
            if (*it == k)
                ans[3] = idx;
            else
                ans[3] = idx + 1;
        } else {
            int k = total / 2;
            double l, r;
            auto it = upper_bound(count.begin(), count.end(), k);
            it--;
            int idx = distance(count.begin(), it);
            if (*it == k)
                l = idx;
            else
                l = idx + 1;
            it = upper_bound(count.begin(), count.end(), ++k);
            it--;
            idx = distance(count.begin(), it);
            if (*it == k)
                r = idx;
            else
                r = idx + 1;
            ans[3] = (l + r) / 2;
        }
        return ans;
    }
};
