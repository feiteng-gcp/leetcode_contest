class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        double mi=-1, ma=0, mean=0, med=0, mode=0, sum=0;
        int num = 0;
        for (int i=0; i<256; i++) {
            if (count[i]) {
                if (mi==-1) mi = i;
                ma = i;
                num += count[i];
                sum += i * count[i];
                if (count[i]>count[mode]) mode = i;
            }
        }
        mean = double(sum) / double(num);
        int m = num/2;
        if (num&2==1) {
            int cnt = 0;
            for (int i=0; i<256; i++) {
                if (cnt<m && cnt+count[i]>=m) {
                    med = i;
                    break;
                }
                cnt += count[i];
            }
        }else {
            int cnt = 0, med1;
            for (int i=0; i<256; i++) {
                if (cnt<m && cnt+count[i]>=m) {
                    med = i;
                    break;
                }
                cnt += count[i];
            }
            cnt = 0;
            for (int i=0; i<256; i++) {
                if (cnt<m+1 && cnt+count[i]>=m+1) {
                    med1 = i;
                    break;
                }
                cnt += count[i];
            }
            cout << m << med << med1 ;
            med = (double)(med + med1) / 2.0;
        }
        return {mi, ma, mean, med, mode};
    }
};
