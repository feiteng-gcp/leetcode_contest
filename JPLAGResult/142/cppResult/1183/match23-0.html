class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        double mode = 0, maximum=INT_MIN, minimum=INT_MAX, sum = 0;
        int num = 0;
        for(int i = 0; i < count.size(); i++) {
            if(count[i]==0) continue;
            if(i>maximum) maximum = i;
            if(i<minimum) minimum = i;
            if(count[i]>count[mode]) mode = i;
            num+=count[i];
            sum+=count[i]*i;
        }
        double mean = sum/(double)num;
        double median = 0;
        if(num%2==1)  {
            int med = (num+1)/2;
            int cur = 0;
            for(int i = 0; i < count.size(); i++) {
                cur+=count[i];
                if(cur>=med) {
                    median = i;
                    break;
                }
            }
        } else {
            bool first = false, second = false;
            int l = num/2,r = l+1, cur = 0;
            for(int i = 0; i < count.size(); i++) {
                cur+=count[i];
                if(!first&&cur>=l) {
                    median += i;
                    first = true;
                }
                if(!second&&cur>=r) {
                    second = true;
                    median += i;
                }
                if(first&&second) break;
            }
            median = median/(double)2;
        }
        return {minimum, maximum, mean, median, mode};
    }
};
