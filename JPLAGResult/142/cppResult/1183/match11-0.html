class Solution {
public:
    double max(vector<int>& count) {
        for (int i = 255;i >= 0;i--) {
            if (count[i] != 0) {
                return i;
            }
        }
        return -1;
    }
    double min(vector<int>& count) {
        for (int i = 0;i < 256;i++) {
            if (count[i] != 0) {
                return i;
            }
        }
        return 0;
    } 
    double avg(vector<int>& count) {
        double s = 0;
        int c = 0;
        for (int i = 0;i < count.size();i++) {
            s += count[i] * i;
            c += count[i];
        }
        return s / c;
    }
    double find(vector<int>& count, int num) {
        for (int i = 0;i < count.size();i++) {
            if (count[i] >= num) {
                return i;
            } else {
                num -= count[i];
            }
        }
        return 0;
    }
    double zhong1(vector<int>& count) {
        int c = 0;
        for (int i = 0;i < count.size();i++) {
            c += count[i];
        }
        if (c % 2 == 0) {
            return (find(count, c / 2) + find(count, c / 2 + 1)) / 2;
        } else {
            return find(count, c / 2 + 1);
        }
    } 
    double zhong2(vector<int>& count) {
        int m = -1;
        for (int i = 0;i < 256;i++) {
            if (count[i] != 0) {
                if (m == -1 || count[i] > count[m]) {
                    m = i;
                }
            }
        }
        return m;
    }
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ret;
        ret.push_back(min(count));
        ret.push_back(max(count));
        ret.push_back(avg(count));
        ret.push_back(zhong1(count));
        ret.push_back(zhong2(count));
        return ret;
    }
};
