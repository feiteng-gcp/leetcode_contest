class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        double minimum = -1, maximum = -1, mean = -1, median = -1, mode = -1;
        double sum = 0;
        int n = 0;
        int maxn = 0;
        for (int i = 0; i < count.size(); i++) {
            if (minimum < 0 && count[i] > 0) minimum = i;
            if (count[i] > 0) {
                maximum = i;
                sum += i * count[i];
                n += count[i];
                if (count[i] > maxn) {
                    mode = i;
                    maxn = count[i];
                }
            }
        }
        mean = sum / n;
        int cnt = 0;
        if (n & 1) {
            n++;
            for (int i = 0; i < count.size(); i++) {
                cnt += count[i];
                if (cnt >= n / 2) {
                    median = i;
                    break;
                }
            }
        } else {
            //cout << n << endl;
            for (int i = 0; i < count.size(); i++) {
                cnt += count[i];
                if (cnt == n / 2) {
                    for (int j = i + 1; j < count.size(); j++) {
                        if (count[j] > 0) {
                            cout << i << " " << j << endl;
                            median = (i + j) / 2.0;
                            break;
                        }
                    }
                    if (median > -1) break;
                }
                else if (cnt > n / 2) {
                    median = i;
                    break;
                }
            }
        }
        // for (int i = 0; i < count.size(); i++) {
        //     cnt += count[i];
        //     if (cnt >= n / 2) {
        //         if (cnt * 2 == n) {
        //             median = i;
        //         } else {
        //             for (int j = i + 1; j < count.size(); j++) {
        //                 if (count[j] > 0) {
        //                     median = (i + j) / 2;
        //                     break;
        //                 }
        //             }
        //         }
        //         break;
        //     }
        // }
        return vector<double>{minimum, maximum, mean, median, mode};
        
    }
};
