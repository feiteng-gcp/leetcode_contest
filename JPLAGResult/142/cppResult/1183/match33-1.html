class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        double minv, maxv, mean, median, mode;
        int modecnt = 0;
        int sz = 0;

        double sum = 0;
        for (int i = 0; i < count.size(); ++i) {
            if (count[i]) {
                minv = i;
                break;
            }
        }
        for (int i = 0; i < count.size(); ++i) {
            if (count[i]) {
                maxv = i;
            }
            sz += count[i];
            sum += count[i] * i;
            if (count[i] > modecnt) {
                modecnt = count[i];
                mode = i;
            }
        }
        mean = sum / sz;
        
        if (sz % 2 == 1) {

            int medianidx = sz / 2;
            int idx = 0;
            for (int i = 0; i < count.size(); ++i) {
                if (count[i] && idx <= medianidx && idx + count[i] > medianidx) {
                    median = i;
                    break;
                }
                idx += count[i];
            }
        } else {
            int medianidx = sz / 2 - 1;
            int idx = 0;
            for (int i = 0; i < count.size(); ++i) {
                if (count[i] && idx <= medianidx && idx + count[i] > medianidx) {
                    if (idx + count[i] > medianidx + 1) {
                        median = i;
                    } else {
                        int j = i + 1;
                        for (; j < count.size(); ++j) {
                            if (count[j]) {
                                 break;
                            }
                        }
                        median = (i + j) / 2.0;
                    }
                    break;
                }
                idx += count[i];
            }
        }
        
        return {minv, maxv, mean, median, mode};
        
    }
};
