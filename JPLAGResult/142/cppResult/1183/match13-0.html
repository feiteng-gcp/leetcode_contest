class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        // min, max, mean, median, mode
        vector<double> ans={256, -1, 0, 0, 0};
        int maxCount=0;
        int sum_count = 0;
        
        for (int i=0; i<count.size(); i++) {
            if (ans[0]==256&&count[i]!=0) ans[0] = i;
            if (count[i]!=0) ans[1] = i;
            sum_count += count[i];
            ans[2] += i*count[i];
            if (count[i]>maxCount) {
                maxCount = count[i];
                ans[4] = i;
            }
        }
        ans[2] /= sum_count;
        
        int curr_count = 0;
        for (int i=0; i<count.size(); i++) {
            if ((curr_count+count[i])*2>=sum_count) {
                if ((curr_count+count[i])*2>sum_count) {
                    ans[3] = i;
                }
                else {
                    int j;
                    for (j=i+1; j<count.size(); j++) {
                        if (count[j]>0) break;
                    }
                    ans[3] = (i+j)/2.0;
                }
                break;
            }
            curr_count += count[i];
        }
        
        return ans;
    }
};
