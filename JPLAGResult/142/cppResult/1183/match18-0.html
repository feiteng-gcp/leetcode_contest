class Solution {
public:
    vector<double> sampleStats(vector<int>& cnt) {
        int L, R;
        for (L = 0; L < 256 && cnt[L] == 0; ++L);
        for (R = 255; L >= 0 && cnt[R] == 0; --R);
        int n = accumulate(cnt.begin(), cnt.end(), 0);
        double sum = 0;
        for (int i = 0; i < 256; ++i) {
            sum += (double)i * cnt[i];
        }
        double ave = sum / n;
        int cur = 0;
        double mid = 0;
        if (n % 2) {
            cur = 0;
            for (int i = 0; i < 256; ++i) {
                cur += cnt[i];
                if (cur >= (n + 1) / 2) {
                    mid = i;
                    break;
                }
            }
        } else {
            int x = n / 2, y = x + 1;
            cur = 0;
            for (int i = 0; i < 256; ++i) {
                cur += cnt[i];
                if (cur >= x) {
                    mid += i;
                    break;
                }
            }
            cur = 0;
            for (int i = 0; i < 256; ++i) {
                cur += cnt[i];
                if (cur >= y) {
                    mid += i;
                    break;
                }
            }
            mid /= 2;
        }
        int max_freq = *max_element(cnt.begin(), cnt.end());
        double mode = -1;
        for (int i = 0; i < 256; ++i) {
            if (cnt[i] == max_freq) mode = i;
        }
        return {L, R, ave, mid, mode};
    }
};
