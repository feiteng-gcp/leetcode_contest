class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ans;
        for(int i=0;i<5;i++)
        {
            ans.push_back(0.0);
        }

        for(int i=0;i<256;i++)
        {
            if(count[i])
            {
                ans[0]=i;
                break;
            }
        }
        for(int i=255;i>=0;i--)
        {
            if(count[i])
            {
                ans[1]=i;
                break;
            }
        }
        double tot=0;
        int cnt=0;
        for(int i=0;i<256;i++)
        {
            tot=tot+i*count[i];
            cnt=cnt+count[i];
        }
        ans[2]=tot/cnt;
        int now=0;
        for(int i=0;i<256;i++)
        {
            now=now+count[i];
            if(now>cnt/2)
            {
                if(cnt%2)ans[3]=i;
                else
                {
                    if(now-count[i]<cnt/2)ans[3]=i;
                    else
                    {
                        for(int j=i-1;j>=0;j--)
                        {
                            if(count[j])
                            {
                                ans[3]=(j+i)/2.0;
                                break;
                            }
                        }
                    }
                }
                break;
            }
        }
        for(int i=0;i<256;i++)
        {
            if(count[i]>count[(int)ans[4]])ans[4]=i;
        }
        return ans;
    }
};
