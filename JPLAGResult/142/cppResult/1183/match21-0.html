class Solution {
public:
    int getk(vector<int>& count,int k) {
        int cnt=0;
        for (int i=0; i<256; i++) {
            cnt += count[i];
            if (cnt>=k) return i;
        }
        return -1;
    }
    
    vector<double> sampleStats(vector<int>& count) {
        vector<double> ans;
        
        int mn=-1, mx=-1;
        long long sum=0, tot=0;
        int mo=0;
        for (int i=0; i<256; i++) {
            if (count[i]) {
                if (mn==-1) mn=i;
                mx = i;
            }
            if (count[i] > count[mo]) {
                mo = i;
            }
            sum += (long long)i*count[i];
            tot += count[i];
        }
        ans.push_back( mn );
        ans.push_back( mx );
        ans.push_back( (double)sum/tot );
        if (tot%2==1) {
            ans.push_back( getk(count, tot/2+1) );
        } else {
            ans.push_back( (getk(count, tot/2) + getk(count, tot/2+1))/2.0 );
        }
        ans.push_back(mo);
        
        return ans;
    }
};
