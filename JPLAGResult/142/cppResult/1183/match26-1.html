class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        vector<double> r(5);
        int maxcidx = -1;
        double minv = -1, maxv = -1;

        double sum = 0;
        int cnt = 0;
        for (int i = 0; i < count.size(); i++) {
            if (count[i] > 0) {
                if (minv < 0) minv = i;
                maxv = i;
                sum += i * count[i];
                cnt += count[i];
                if (maxcidx == -1 || count[i] > count[maxcidx]) maxcidx = i;
            }
        }

        double avg = sum / cnt;
        int t = 0;
        double med = 0;
        if (cnt % 2) {
            for (int i = 0; i < count.size(); i++) {
                t += count[i];
                if (t >= cnt / 2 + 1) {
                    med = i;
                    break;
                }
            }
        } else {
            int ct = 0;
            for  (int i = 0; i < count.size(); i++) {
                t += count[i];
                if (ct == 0 && t >= cnt / 2) {
                    med += i * 1.0 / 2;
                    ct++;
                }
                if (ct == 1 && t >= cnt / 2 + 1) {
                    med += i * 1.0 / 2;
                    break;
                }
            }
        }
        r[0] = minv;
        r[1] = maxv;
        r[2] = avg;
        r[3] = med;
        r[4] = maxcidx;
        return r;
    }
};
