typedef long long lld;

class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        int mn = -1;
        int mx = -1;
        lld sum = 0LL;
        int mode = 0;
        int N = count.size();
        int M = 0;
        for (int i = 0; i < N; ++i) {
            if (count[i]) {
                if (mn == -1) mn = i;
                mx = i;
            }
            if (count[i] > count[mode]) {
                mode = i;
            }
            sum += i * count[i];
            M += count[i];
        }
        int sum2 = 0;
        int half = (M + 1) / 2;
        double med = 0;
        if (M & 1) {
            for (int i = 0; i < N; ++i) {
                sum2 += count[i];
                if (sum2 >= half) {
                    med = i;
                    break;
                }
            }
        } else {
            for (int i = 0; i < N; ++i) {
                sum2 += count[i];
                if (sum2 > half) {
                    med = i;
                    break;
                } else if (sum2 == half) {
                    for (int j = i+1; j < N; ++j) {
                        if (count[j]) {
                            med = (i + j) / 2.0;
                            break;
                        }
                    }
                    break;
                }
            }
        }
        return { mn, mx, 1.0 * sum / M, med, mode };
    }
};
