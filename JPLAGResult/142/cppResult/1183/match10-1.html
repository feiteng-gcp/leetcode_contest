class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        // minimum, maximum, mean, median, and mode
        double minimum = 300, maximum = -1, mean = -1, median = -1, mode = -1;
        double sum = 0;
        int maxCnt = 0, totalCnt = 0;
        for(int k = 0; k < count.size(); k ++) {
            if(count[k] > 0) {
                minimum = min(minimum, k * 1.0);
                maximum = max(maximum, k * 1.0);
                sum += 1.0 * k * count[k];
                totalCnt += count[k];
                if(count[k] > maxCnt) {
                    maxCnt = count[k];
                    mode = k;
                }
            }
        }
        if(totalCnt & 1) {
            // odd
            median = 0;

            int index = totalCnt / 2;
            int cur = -1;
            for(int k = 0; k < count.size(); k ++) {
                cur += count[k];
                if(cur >= index) {
                    median = k;
                    break;
                }
            }
        }
        else {
            // even

            median = 0;
            int index = totalCnt / 2 - 1;
            int cur = -1;
            for(int k = 0; k < count.size(); k ++) {
                cur += count[k];
                // cout << "cur: " << cur << "index: " << index << endl;
                if(cur >= index && 0 == median) {
                    median += k;
                }
                if(cur >= index + 1) {
                    median += k;
                    break;
                }
            }
            // cout << "median: " << median << endl;
            median = median / 2.0;
        }
        mean = sum / totalCnt;
        vector<double> result;
        result.push_back(minimum);
        result.push_back(maximum);
        result.push_back(mean);
        result.push_back(median);
        result.push_back(mode);
        return result;
    }
};
