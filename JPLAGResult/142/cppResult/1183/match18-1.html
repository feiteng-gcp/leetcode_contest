class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        long sum = accumulate(count.begin(), count.end(), 0);
        int mini;
        int maxi;
        vector<double>ans;
        for(int i=0;i<count.size();i++)
        {
            if(count[i]>0)
            {
                mini = i;
                break;
            }
        }
        for(int i=0;i<count.size();i++)
        {
            if(count[i]>0)
            {
                maxi = i;
            }
        }
        int mode;
        int curr_max = INT_MIN;
        for(int i=0;i<count.size();i++)
        {
            if(count[i]>curr_max)
            {
                mode = i;
                curr_max = count[i];
            }
        }
        double mean;
        long total_sum = 0;
        for(int i=0; i<count.size();i++)
        {
            total_sum += count[i]*i;
        }
        mean = (total_sum*1.0)/sum;
        
        double median;
        if(sum%2 == 1)
        {
            long required = (sum+1)/2;

            long cur = 0;
            for(int i=0; i<count.size();i++)
            {
                cur = cur + count[i];
                if(cur >= required)
                {
                    median = i;
                    break;
                }
            }
        }
        else
        {
            long required1 = sum/2;
            long required2 = required1+1;
            long cur = 0;
            long first;
            long second;

            for(int i=0; i<count.size();i++)
            {
                cur = cur + count[i];
                if(cur >= required1)
                {
                    first = i;
                    break;
                }
            }
            cur = 0;
            for(int i=0; i<count.size();i++)
            {
                cur = cur + count[i];
                if(cur >= required2)
                {
                    second = i;
                    break;
                }
            }
            median = (first + second)/2.0;
            
        }
        ans.push_back((double)mini);
        ans.push_back((double)maxi);
        ans.push_back(mean);
        ans.push_back(median);
        ans.push_back((double)mode);
        return ans;
    }
};
