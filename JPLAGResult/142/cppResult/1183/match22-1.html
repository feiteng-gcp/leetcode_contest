class Solution {
public:
    
    int get(vector<int>& count, int pos) {
        int sum=0;
        for(int i=0; i<count.size(); i++) {
            sum+=count[i];
            if (sum>=pos) return i;
        }
        return -1;
    }
    
    vector<double> sampleStats(vector<int>& count) {
        int inf=0x3f3f3f3f;
        double mn=inf, mx=-inf, sum=0, mid, fre;
        int sz=0;
        int ans=0;
        for(auto v:count) sz+=v;
        for(int i=0; i<count.size(); i++) {
            int v=count[i];
            if (!v) continue;
            sum+=v*i;
            mn=min(mn, (double)i);
            mx=max(mx, (double)i);
            if (v>ans) {
                ans=v;
                fre=i;
            }
        }
        
        if (sz%2==1) mid=get(count, sz/2+1);
        else {
            mid=get(count, sz/2)+get(count, sz/2+1);
            mid/=2;
        }
        
        vector<double> V;
        V.push_back(mn);
        V.push_back(mx);
        V.push_back(sum/sz);
        V.push_back(mid);
        V.push_back(fre);
        return V;
    }
};
