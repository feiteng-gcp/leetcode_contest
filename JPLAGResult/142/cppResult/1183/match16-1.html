class Solution {
public:
    vector<double> sampleStats(vector<int>& count) {
        int len = count.size();
        
        double minval = DBL_MAX;
        for(int i = 0; i < len; ++i) {
            if(count[i] > 0) minval = min(minval, i * 1.0);
        }
        double maxval = DBL_MIN;
        for(int i = 0; i < len; ++i) {
            if(count[i] > 0) maxval = max(maxval, i * 1.0);
        }
        
        int n = 0;
        double sum = 0.0;
        for(int i = 0; i < len; ++i) {
            n += count[i];
            sum += i * count[i] * 1.0;
        }
        
        //cout << n << " " << sum << endl;
        
        double mean = sum / n * 1.0;
        
        int midleft = (n - 1) / 2;
        int midright = (n) / 2;
        midleft++;
        midright++;
        double midleftval = 0.0;
        double midrightval = 255.0;

        int cnt = 0;
        for(int i = 0; i < len; ++i) {
            cnt += count[i];
            if(cnt >= midleft) {
                midleftval = i * 1.0;
                break;
            }
        }
        cnt = 0;
        for(int i = 0; i < len; ++i) {
            cnt += count[i];
            if(cnt >= midright) {
                midrightval = i * 1.0;
                break;
            }
        }
        double median = (midleftval + midrightval) / 2.0;
        
        int maxcnt = 0;

        for(int x: count) maxcnt = max(maxcnt, x);
        double mode = 0.0;
        for(int i = 0; i < len; ++i) {
            if(count[i] == maxcnt) mode = i * 1.0;
        }
        
        return {minval, maxval, mean, median, mode};
    }
};
