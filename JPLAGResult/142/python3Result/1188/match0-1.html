class Solution:
    def braceExpansionII(self, expression: str):
        ans = set()
        idx = 0

        tmp = set()
        while idx < len(expression):
            if expression[idx].isalpha():
                t = ""
                while idx < len(expression) and expression[idx].isalpha():
                    t += expression[idx]
                    idx += 1
                if len(tmp) > 0:
                    new_tmp = set()
                    for candi in tmp:
                        new_tmp.add(candi+t)
                    tmp = new_tmp
                else:
                    tmp.add(t)
            elif expression[idx] == ',':
                idx += 1
                ans = ans | tmp
                tmp.clear()
            elif expression[idx] == '{':
                idx += 1
                stack = 0
                candidate = ""
                while idx < len(expression):
                    if expression[idx] == '}':
                        if stack == 0:
                            back = self.braceExpansionII(candidate)
                            if len(tmp) > 0:
                                new_tmp 
= set()
                                for t in tmp:
                                    for b in back:
                                        new_tmp.add(t+b)
                                tmp = new_tmp
                            else:
                                tmp = set(back)
                            idx += 1
                            break
                        else:
                            stack -= 1
                    elif expression[idx] == '{':
                        stack += 1
                    candidate += expression[idx]
                    idx += 1
        ans = ans | tmp
        return sorted(list(ans))
