class Solution:
    def braceExpansionII(self, expression: str) -> List[str]:
        i = 0
        res = set()
        temp = set()
        while i < len(expression):
            
            if expression[i] not in '{},': # combine strings
                j = i
                while j < len(expression) and expression[j] not in '{},':
                    j += 1
                if temp:
                    buffer = set()
                    for elem in temp:
                        buffer.add(elem + expression[i:j])
                    temp = buffer
                else:
                    temp.add(expression[i:j])
                i = j
            
            elif expression[i] == ',':
                for elem in temp:
                    res.add(elem)
                temp = set()
                i += 1
            
            elif expression[i] == '{':
                stack = 1
                j = i + 1
                while j < len(expression) and stack != 0:
                    if expression[j] == '{':
                        stack += 1
                    elif expression[j] == '}':
                        stack -= 1
                    j += 1
                
                partial_res = self.braceExpansionII(expression[i+1:j-1])
                if temp: # combine two sets
                    buffer = set()
                    for exp1 in temp:
                        for exp2 in partial_res:
                            buffer.add(exp1 + exp2)
                    temp = buffer
                else:
                    temp = set(partial_res)
                i = j
        
        for elem in temp:
            res.add(elem)
        
        return sorted(res)
