#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        le = mountain_arr.length()
        l = 0
        r = le-1
        cont = 0
        while r>l:

            mid1 = l + (r - l)//3
            mid2 = r - (r - l)//3
            x1 = mountain_arr.get(mid1)
            x2 = mountain_arr.get(mid2)
            if x1 > x2:
                r = mid2-1
            elif x1 < x2:
                l = mid1+1
            else :
                l = mid1+1
                r = mid2-1
        
        mi = r

        l = 0 
        r = mi
        while r >= l:
            mid = (l+r)//2
            x = mountain_arr.get(mid)
            if x > target:
                r = mid - 1
            elif x < target:
                l = mid + 1
            else:
                return mid
        
        l = mi
        r = le-1
        
        while r >= l:
            mid = (l+r)//2
            x = mountain_arr.get(mid)
            if x < target:
                r = mid - 1
            elif x > target:
                l = mid + 1
            else:
                return mid
        
        return -1
