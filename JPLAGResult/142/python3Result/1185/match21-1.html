#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        length = mountain_arr.length()
        def top_of(mountain_arr):
            if mountain_arr.get(length - 2) < mountain_arr.get(length - 1):
                return length - 1
            elif mountain_arr.get(0) > mountain_arr.get(1):
                return 0
            # the top exists in between
            # searching space: [lo, hi)
            lo, hi 
= 0, length
            while hi - lo > 1:
                mid = (lo + hi) // 2
                if mountain_arr.get(mid - 1) < mountain_arr.get(mid):
                    lo = mid
                else:
                    hi = mid
            return lo

        top_idx = top_of(mountain_arr)
        # print(top_idx)
        # search for the first interval [0, top_idx)
        # searching space: (lo, hi]
        lo, hi = -1, top_idx - 1
        while hi - lo > 1:
            mid = (lo + hi) // 2
            if mountain_arr.get(mid) < target:
                lo = mid
            else:
                hi = mid
        if hi >= 0 and mountain_arr.get(hi) == target:
            return hi

        # search for the second interval [top_idx, length)
        # searching space: (lo, hi]
        lo, hi = top_idx - 1, length - 1
        while hi - lo > 1:
            mid = (lo + hi) // 2
            if mountain_arr.get(mid) > target:
                lo = mid
            else:
                hi = mid
        if mountain_arr.get(hi) == target:
            return hi
        else:
            return -1
