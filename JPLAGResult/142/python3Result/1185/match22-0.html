#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target, A):
        
        l, r = 0, A.length()-1
        top = None
        while l <= r:
            m = (l+r) >> 1
            pre, cur, succ = A.get(m-1), A.get(m), A.get(m+1)
            if pre < cur and cur > succ:
                top = m
                break
            elif pre < cur and cur < succ:
                l = m
            else:
                r = m

        ans = self.bs_search( A, target, 0, top, True )
        if ans != -1: return ans
        ans = self.bs_search( A, target, top+1, A.length()-1, False )
        return ans
    
    def bs_search(self, A, target, l, r, inc):
        
        while l <= r:
            m = (l+r) >> 1
            cur = A.get(m)
            if cur == target: 
                return m
            elif cur < target:
                if inc: l = m+1
                else: r = m-1
            else:
                if inc: r = m-1
                else: l = m+1
        return -1
