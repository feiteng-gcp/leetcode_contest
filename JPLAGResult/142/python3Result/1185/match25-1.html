# """

#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        a = mountain_arr
        size = a.length()
        memo = {}
        memo[-1] = memo[size] = -1
        
        def get(idx):
            if idx not in memo:
                memo[idx] = a.get(idx)
            return memo[idx]
        
        l, r = 0, size - 1
        while l < r:
            mid = l + (r - l) // 2
            if get(mid) < get(mid + 1):
                l = mid + 1
            elif get(mid) > get(mid + 1):
                r = mid
        peak = l
        
        l, r = 0, peak
        while l <= r:
            mid = l + (r - l) // 2
            if get(mid) == target:
                return mid
            elif get(mid) < target:
                l = mid + 1
            else:
                r = mid - 1
        
        l, r 
= peak, size - 1
        while l <= r:
            mid = l + (r - l) // 2
            if get(mid) == target:
                return mid
            elif get(mid) < target:
                r = mid - 1
            else:
                l = mid + 1
        
        return -1
