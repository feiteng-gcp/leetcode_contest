class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        peek=self.findPeek
(mountain_arr)
        l,r=0,peek-1
        while(l<r):
            mid=l+(r-l)//2
            if mountain_arr.get(mid)<target:
                l=mid+1
            else:
                r=mid
        if mountain_arr.get(l)==target:
            return l
        else:
            l,r=peek,mountain_arr.length()-1
            
while(l<r):
                mid=l+(r-l)//2
                if mountain_arr.get(mid)>target:
                    l=mid+1
                else:
                    r=mid
            if mountain_arr.get(l)==target:
                return l
        return -1
    def findPeek(self,mountain_arr):
        left = 1
        right = mountain_arr.length() - 2
        while left < right:
            mid = (right + left) // 2
            if mountain_arr.get(mid) < mountain_arr.get(mid+1):
                left = mid + 1
            if mountain_arr.get(mid) > mountain_arr.get(mid+1):                
                right = mid
        return left
