# """
# This is MountainArray's API interface.
# You should not implement it, or speculate about its implementation
# """
#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, arr: 'MountainArray') -> int:
        length = arr.length()
        low, up = 0, arr.length()
        
        while up - low >= 3:
            d = up - low
            mid1 = low + d//3
            mid2 = up - d//3
            
            val1 = arr.get(mid1)
            val2 = arr.get(mid2)
            
            if val2 > val1:
                low = mid1
            else:
                up = mid2
        
        max_ind = low
        max_val = arr.get(low)
        
        for i in range(low+1, up + 1):
            cur_val = arr.get(i)
            if cur_val < max_val:
                break
            else:
                max_val = cur_val
                max_ind = i
                
        
        if target == max_val:
            return max_ind
        
        if target > max_val:
            return -1
        
        arr_0 = arr.get(0)
        if target >= arr_0:
            l, u = 0, max_ind
            
            while l <= u:
                mid = (l + u) // 2
                
                val = arr.get(mid)
                if target == val:
                    return mid
                
                if target > val:
                    l = mid + 1
                else:
                    u = mid - 1
        
        arr_last = arr.get(length-1)
        
        if target < arr_last:
            return -1
        
        #print(arr_last)
        
        l, u = max_ind, length-1
            
        while l <= u:
            mid = (l + u) // 2
            

            val = arr.get(mid)
            if target == val:
                return mid
            
            #print(mid, l, u, "val ", val)
            
            if target > val:
                u = mid - 1
            else:
                l = mid + 1
                
        return -1
