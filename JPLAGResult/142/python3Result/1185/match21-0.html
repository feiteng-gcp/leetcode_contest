#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def find_peak(self, mountain_arr: 'MountainArray', length):
        left = 0
        right = length
        while left < right - 1:
            mid = left + (right - left) // 2
            if mountain_arr.get(mid) > mountain_arr.get(mid - 1):
                left = mid 
            else:
                right = mid
        return left

    def binSearch(self, first, last, value, cmp):
        while first < last:
            mid = first + ((last - first) >> 1)
            if cmp(mid, value):
                first = mid + 1
            else:
                last = mid
        return first

    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        mr_length = mountain_arr.length()
        peak = self.find_peak(mountain_arr, mr_length)
        i1 = self.binSearch(0, peak, target, lambda m, v:mountain_arr.get(m) < v)
        if i1 < mr_length and mountain_arr.get(i1) == target:
            return i1
        i2 = self.binSearch(peak, mr_length, target, lambda m, v:mountain_arr.get(m) > v)
        if i2 < mr_length and mountain_arr.get(i2) == target:
            return i2
        return -1
