def findPeakUtil(arr,low,high,n): 
    mid=(low+high)//2    
    amid=arr.get(mid)
    if ((mid==0 or arr.get(mid-1) <= amid) and \
        (mid==n-1 or arr.get(mid+1) <= amid)): 
        return mid 
    elif (mid > 0 and arr.get(mid-1) > amid): 
        return findPeakUtil(arr,low,mid-1,n)
    else: 
        return findPeakUtil(arr,mid+1,high,n) 

def binarySearch1(arr,l,r,x): 
    while l<=r:   
        mid=(l+r)//2; 
        #print(l,mid,r)
        armid=arr.get(mid)
        #print(armid)
        if armid==x: return mid 
        elif armid<x: l=mid+1
        else: r=mid-1
    return 1000000

def binarySearch2(arr,l,r,x): 
    while l<=r:   
        mid=(l+r)//2; 
        #print(l,mid,r)
        armid=arr.get(mid)
        #print(armid)
        if armid==x: return mid 
        elif armid>x: l=mid+1
        else: r=mid-1
    return 1000000

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        n=mountain_arr.length()
        pi=findPeakUtil(mountain_arr,0,n-1,n) 
        print(pi)
        cv=mountain_arr.get(pi)
        ind1,ind2=1000000,1000000
        if cv>=target:
            ind1=binarySearch1(mountain_arr,0,pi,target)
            ind2=binarySearch2(mountain_arr,pi+1,n-1,target)
            print(ind1,ind2)
        ret=min(ind1,ind2)
        if ret==1000000:
            return -1
        return ret
