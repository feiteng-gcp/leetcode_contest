# """
# This is MountainArray's API interface.
# You should not implement it, or speculate about its implementation
# """
#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        cache = {}
        def get(i):
            if i not in cache:
                cache[i] = mountain_arr.get(i)
            return cache[i]
        
        m = mountain_arr.length()//2
        l = 0
        r = mountain_arr.length()
        
        while not get(m-1) < get(m) > get(m+1):
            if get(m-1) < get(m) < get(m+1):
                l = m
            else:
                r = m
            m = (l + r) // 2
            
        l = 0
        r = m+1
        
        l2 = m
        r2 = mountain_arr.length()
        
        while l < r:
            m = (l + r) // 2
            if get(m) == target:
                return m
            if get(m) > target:
                r = m
            else:
                l = m+1
                
        while l2 < r2:
            m = (l2 + r2) // 2
            if get(m) == target:
                return m
            if get(m) > target:
                l2 = m+1
            else:
                r2 = m
        return -1
