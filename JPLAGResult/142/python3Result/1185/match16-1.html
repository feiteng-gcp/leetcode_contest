class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        peak_idx, peak = self.findPeak(mountain_arr)
        
        if target > peak:
            return -1
        
        if target == peak:
            return peak_idx
        
        idx = self.findPeakUpward(target, mountain_arr, peak_idx - 1)
        if idx == -1:
            return self.findPeakDownward(target, mountain_arr, peak_idx + 1)
        
        return idx

    def findPeak(self, mountain_arr):
        left, right = 0, mountain_arr.length() - 1
        
        while left + 1 < right:
            mid = (left + right) // 2
            if mountain_arr.get(mid) < mountain_arr.get(mid + 1):
                left = mid
            else:
                right = mid
        lp = mountain_arr.get(left)
        rp =  mountain_arr.get(right)
        if lp < rp:
            return right, rp
        return left, lp
