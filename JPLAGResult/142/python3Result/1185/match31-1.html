#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        def getMaxIdx(l, r):
            while l + pow(1, -10) < r:
                mid 
= (l + r) // 2
                midd = (mid + r) // 2
                mid_v = mountain_arr.get(mid)
                midd_v = mountain_arr.get(midd)
                if mid_v > midd_v:
                    r = midd
                else:
                    l = mid
            return l

        idx = getMaxIdx(0, mountain_arr.length())
        l, r = 0, idx
        while l <= r:
            mid = (l + r) // 2
            if mountain_arr.get(mid) == target:
                return mid
            elif mountain_arr.get(mid) > target:
                r = mid - 1
            else:
                l = mid + 1
                
        l, r = idx, mountain_arr.length() - 1
        while l <= r:
            mid = (l + r) // 2
            if mountain_arr.get(mid) == target:
                return mid
            elif mountain_arr.get(mid) > target:
                l = mid + 1
            else:
                r = mid - 1
        return -1
