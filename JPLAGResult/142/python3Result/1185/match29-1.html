# """
# This is MountainArray's API interface.
# You should not implement it, or speculate about its implementation
# """
#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
      mr = mountain_arr
      n = mr.length()
      l, h = 0, n - 1
      p = -1
      
while l <= h:
        mid = (l + h) >> 1
        v1 = -1 if mid == 0 else mr.get(mid - 1)
        v2 = mr.get(mid)
        v3 = -1 if mid == n - 1 else mr.get(mid + 1)
        if v2 > v1 and v2 > v3:
          p = mid
          break
        elif v1 < v2 and v2 < v3:
          l = mid + 1
        elif v1 > v2 and v2 > v3:
          h = mid - 1
      l, h = 0, p
      while l <= h:
        mid = (l + h) >> 1
        v = mr.get(mid)
        if v == target:
          return mid
        elif v > target:
          h = mid - 1
        else:
          l = mid + 1
      l, h = p, n - 1
      while l <= h:
        mid = (l + h) >> 1
        v = mr.get(mid)
        if v == target:
          return mid
        elif v > target:
          l = mid + 1
        else:
          h = mid - 1
      return -1
