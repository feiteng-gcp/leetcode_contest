#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        length = mountain_arr.length()

        left = 1
        right = length-1
        high = -1
        while left <= right:
            mid = (left+right)//2
            # print (high, mid, left, right)
            if mountain_arr.get(mid-1) < mountain_arr.get(mid):
                high = max(high, mid)
                left = mid+1
            else:
                right = mid-1
        # print (high)
        left = 0
        right = high
        
while left <= right:
            mid = (left+right)//2
            this = mountain_arr.get(mid)
            if this == target:
                return mid
            if this < target:
                left = mid+1
            else:
                right = mid-1
        
        left = high
        right = length-1
        
while left <= right:
            mid = (left+right)//2
            this = mountain_arr.get(mid)
            if this == target:
                return mid
            if this < target:
                right = mid-1
            else:
                left = mid+1
        return -1
