#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        n =  mountain_arr.length()
        low = 0
        high = n-1
        dic = {}
        while(low<high-1):
            mid = (low+high)//2
            if mid not in dic: dic[mid] = mountain_arr.get(mid)
            if mid+1 not in dic: dic[mid+1] = mountain_arr.get(mid+1)
            if dic[mid]<dic[mid+1]:
                low = mid
            else:
                high = mid
        top = high
        if dic[top]==target:
            return top
        if target > dic[top]:
            return -1
        if 0 not in dic: dic[0] = mountain_arr.get(0)
        if target==dic[0]:
            return 0
        if target>dic[0]:
            low = 0
            high = top
            while(high-low>1):
                mid = (high + low)//2
                if mid not in dic: dic[mid] = mountain_arr.get(mid)
                if dic[mid] > target:
                    high = mid
                elif dic[mid] < target:
                    low = mid
                else:
                    return mid
        if n-1 not in dic: dic[n-1] = mountain_arr.get(n-1)
        if dic[n-1]==target:
            return n-1
        if dic[n-1]<target:
            low = top
            high = n-1
            while(high-low>1):
                mid = (low+high)//2
                if mid not in dic: dic[mid] = mountain_arr.get(mid)
                if dic[mid] > target:
                    low = mid
                elif dic[mid] < target:
                    high = mid
                else:
                    return mid
        return -1
