class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        arr_l = mountain_arr.length()
        peak = self.get_peak(arr_l, mountain_arr)
        
        res = self.find_target_asc(0, peak, target, mountain_arr)
        if res != -1:
            return res
        return self.find_target_des(peak, arr_l - 1, target, mountain_arr)
    
    def get_peak(self, arr_l, mountain_arr):
        s = 0
        e = arr_l
        
        while s + 1 < e:
            mid = (s + e) // 2
            if mid == 0:
                return 1
            if mid == arr_l - 1:
                return mid - 1
            prev = mountain_arr.get(mid - 1)
            cur = mountain_arr.get(mid)
            next = mountain_arr.get(mid + 1)
            if prev < cur > next:
                return mid
            if prev < cur < next:
                s = mid
            elif prev > cur > next:
                e = mid
        prev = (mountain_arr.get(mid - 1), mid - 1)
        cur = (mountain_arr.get(mid), mid)
        next = (mountain_arr.get(mid + 1), mid + 1)
        
        return max([prev, cur, next])[1]
    
    def find_target_asc(self, s, e, t, mountain_arr):
        while s + 1 < e:
            mid = (s + e) // 2
            cur = mountain_arr.get(mid)
            if cur == t:
                return mid
            if cur < t:
                s 
= mid
            else:
                e = mid
        if mountain_arr.get(s) == t:
            return s
        if mountain_arr.get(e) == t:
            return e
        return -1
    
    def find_target_des(self, s, e, t, mountain_arr):
        while s + 1 < e:
            mid = (s + e) // 2
            cur = mountain_arr.get(mid)
            if cur == t:
                return mid
            if cur > t:
                s 
= mid
            else:
                e = mid
        if mountain_arr.get(s) == t:
            return s
        if mountain_arr.get(e) == t:
            return e
        return -1
