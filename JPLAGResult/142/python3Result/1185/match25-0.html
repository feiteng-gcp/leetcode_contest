#class MountainArray:

#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        L = mountain_arr.length()
        
        elem = {}
        
        def iget(item):
            if item in elem:
                return elem[item]
            else:
                res = mountain_arr.get(item)
                elem[item] = res
                return res
        
        # find the peak
        low, high = 0, L - 1
        while low < high:
            mid = (low + high + 1) // 2
            if iget(mid) > iget(mid - 1): # peak in the second half
                low = mid
            else:
                high = mid - 1
        
        midpoint = low
        #print(iget(midpoint))
        
        # find the element
        if target >= iget(0): # search in the first half
            low, high = 0, midpoint
            while low <= high:
                mid = (low + high) // 2
                if iget(mid) == target:
                    return mid
                elif iget(mid) > target:
                    high = mid - 1
                else:
                    low = mid + 1
        
        low, high = midpoint, L - 1
        while low <= high:
            #print(low, high)
            mid = (low + high) // 2
            if iget(mid) == target:
                return mid
            elif iget(mid) < target:
                high = mid - 1
            else:
                low = mid + 1
        
        return -1
