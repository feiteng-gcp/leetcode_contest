class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        L = mountain_arr
        N = mountain_arr.length()

        lo = 0
        hi = N - 2
        while lo < hi:
            mid = (lo + hi) // 2
            if L.get(mid) < L.get(mid + 1):
                lo 
= mid + 1
            else:
                hi = mid

        P = lo

        lo = 0
        hi = P
        while lo < hi:
            mid = (lo + hi) // 2
            x = L.get(mid)
            if x == target:
                return mid
            elif x < target:
                lo = mid + 1
            else:
                hi = mid

        if L.get(lo) == target: return lo

        lo = P
        hi = N - 1
        while lo < hi:
            mid = (lo + hi) // 2
            x = L.get(mid)
            if x == target:
                return mid
            elif x > target:
                lo = mid + 1
            else:
                hi = mid

        if L.get(lo) == target: return lo

        return -1
