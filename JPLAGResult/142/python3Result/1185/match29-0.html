#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, arr: 'MountainArray') -> int:
        l = 0
        h = arr.length()-1
        while l < h:
            m = (l+h)//2
            a1 = arr.get(m-1)
            a2 = arr.get(m)
            a3 = arr.get(m+1)
            if a1<a2<a3:
                l = m + 1
            if a1<a2 and a2>a3:
                if a2 == target:
                    return m
                break
            if a1>a2>a3:
                h = m
        # print(m)
        ll = 0
        rr = m
        while ll < rr:
            m1 = (ll+rr)//2
            if arr.get(m1) < target:
                ll = m1 + 1
            else:
                rr = m1
        m1 = ll
        
        ll = m
        rr = arr.length()-1
        while ll < rr:
            m2 = (ll+rr)//2
            if arr.get(m2) > target:
                ll = m2 + 1
            else:
                rr = m2
        m2 = ll
        
        if arr.get(m1) == target:
            res1 = m1
        else:
            res1 = float('inf')
        if arr.get(m2) == target:
            res2 = m2
        else:
            res2 = float('inf')
        res = min(res1,res2)
        return res if res != float('inf') else -1
