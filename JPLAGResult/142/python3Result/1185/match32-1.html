# """

#class MountainArray:

#    def length(self) -> int:


class Solution:
    def findInMountainArray(self, target: int, circumvent : 'MountainArray') -> int:
        N = circumvent.length()
        memo = {}
        left = 0
        right = N-1
        while left < right:
            mid = (left+ right)//2
            if mid not in memo:
                memo[mid] = circumvent.get(mid)
            mid_val = memo[mid]
            if mid+1 not in memo:
                memo[mid+1] = circumvent.get(mid+1)
            mid_right = memo[mid+1]
            if mid_val < mid_right:
                left = mid +1
            else:
                right = mid
        peak = left        
        left = 0 
        right = peak
        while left <= right:
            mid = (left+ right)//2
            if mid not in memo:
                memo[mid] = circumvent.get(mid)
            mid_val = memo[mid]
            if mid_val ==target:
                return mid
            elif mid_val < target:
                left = mid+1
            else:
                right = mid-1
                
        left = peak+1
        right = N-1
        while left <= right:
            mid = (left+ right)//2
            if mid not in memo:
                memo[mid] = circumvent.get(mid)
            mid_val = memo[mid]
            if mid_val ==target:
                return mid
            elif mid_val < target:
                right = mid-1
            else:
                left = mid+1
        return -1
