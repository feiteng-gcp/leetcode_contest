class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        min_ = 255
        for i in range(256):
            if count[i] != 0:
                min_ = i
                break
        
        max_ = 0
        for i in range(255, -1, -1):
            if count[i] != 0:
                max_ = i
                break
                
        c = 0
        sum_ = 0  
        max_num = 0
        mode = 0
        for i in range(256):
            if count[i] > max_num:
                max_num = count[i]
                mode = i
            c += count[i]
            sum_ += count[i]*i 
        avg = sum_/c
        
        if c % 2 == 1:
            mean_idx = c // 2 + 1
            s = 0
            for i in range(256):
                s += count[i]
                if s >= mean_idx:
                    mean = i
                    break
        else:
            mean1, mean2 = -1, -1
            mean_idx1, mean_idx2 = c // 2, c // 2 + 1
            s = 0
            for i in range(256):
                s += count[i]
                if s >= mean_idx1 and mean1 == -1:
                    mean1 = i
                if s >= mean_idx2:
                    mean2 = i
                if mean1 != -1 and mean2 != -1:
                    mean = (mean1+mean2) / 2
                    break
        
        return [float(min_), float(max_), float(avg), float(mean), float(mode)]
