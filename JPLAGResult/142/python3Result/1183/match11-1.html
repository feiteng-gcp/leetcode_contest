class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        N = 256
        res = []
        
for i in range(N):
            if count[i] != 0:
                break
        minimal = i
        
        for i in range(N - 1, -1, -1):
            if count[i] != 0:
                break
        maximum = i
        
        s = 0
        total = 0
        for i in range(minimal, maximum + 1):
            s += i * count[i]
            total += count[i]
        mean = float(s/total)
        
        
        left = right = -1
        curr_count = 0
        for i in range(minimal, maximum + 1):
            curr_count += count[i]
            if curr_count >= total // 2:
                if left == -1:
                    left = i
            if curr_count >= total // 2 + 1:
                if left == -1:
                    left = i
                if right == -1:
                    right = i
            if left > -1 and right > -1:
                break
                
        if total % 2 == 0:
            median = (left + right) / 2
        else:
            median = float(right)
                
        
        mode = 0
        mode_count = 0
        for i in range(minimal, maximum + 1):
            if count[i] > mode_count:
                mode_count = count[i]
                mode = i
        
        
        res = [float(minimal), float(maximum), mean, median, float(mode)]
        
        return res
