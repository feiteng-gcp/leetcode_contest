class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        min_item = float(next(i for i, c in enumerate(count) if c > 0))
        max_item = float(next(i for i in reversed(range(256)) if count[i] > 0))

        total_count = sum(count)
        total_sum = 0
        for i, c in enumerate(count):
            total_sum += i * c
        mean = total_sum / total_count

        cur_count = 0
        median_indices = [total_count // 2, total_count // 2 + 1] if total_count % 2 == 0 \
            else [total_count // 2 + 1]
        cur_idx = 0
        while cur_idx < len(count) and cur_count + count[cur_idx] < median_indices[0]:
            cur_count += count[cur_idx]
            cur_idx += 1
        median_values = [cur_idx]
        if len(median_indices) == 2 and cur_count + count[cur_idx] < median_indices[1]:
            median_values.append(next(i for i, c in enumerate(count[cur_idx + 1:], start=cur_idx + 1) if c > 0))
        # print(median_values)
        median = sum(median_values) / len(median_values)

        mode = -1
        max_count = 0
        for i, c in enumerate(count):
            if c > max_count:
                mode = float(i)
                max_count = c
        return [min_item, max_item, mean, median, mode]
