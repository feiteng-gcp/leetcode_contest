class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        for i in range(256):
            if count[i]>0:
                minval = float(i)
                break
        for i in range(255, -1, -1):
            if count[i]>0:
                maxval = float(i)
                
break
        cnt = 0
        maxc = 0
        common = 0
        s = 0
        for i in range(256):
            if count[i]>0:
                s += count[i]*i
                cnt += count[i]
            if count[i]>maxc:
                maxc = count[i]
                common = float(i)
        mean = float(s)/cnt
        tmp = 0
        if cnt%2==1:
            for i in range(256):
                tmp += count[i]
                if tmp>cnt//2:
                    median = float(i)
                    break
        else:
            first = -1
            second = -1
            for i in range(256):
                tmp += count[i]
                if tmp==cnt//2:
                    first = i
                if tmp>cnt/2:
                    second = i
                    if first == -1:
                        first = i
                    break
            median = float(first+second)/2
               
        return [minval, maxval, mean, median, common]
