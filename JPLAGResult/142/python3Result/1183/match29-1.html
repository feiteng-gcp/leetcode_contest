class Solution:
    def sampleStats(self, count):
        l = sum(count)
        ansMin = -1
        ansMax = -1
        for k in range(256):
            if count[k]:
                if ansMin == -1:
                    ansMin = k
                ansMax = k
                
        ansMean = sum(k*count[k] for k in range(256) if count[k]) / l 
        if l%2 == 0:
            curCount = 0
            left = -1
            right = -1
            for k in range(256):
                if count[k]:
                    newCurCount = curCount + count[k]
                    if curCount < l//2 <= newCurCount:
                        left = k
                    if curCount < (l//2)+1 <= newCurCount:
                        right = k
                    curCount = newCurCount
            ansMedian = (left + right)/2
        else:
            curCount = 0
            for k in range(256):
                if count[k]:
                    newCurCount = curCount + count[k]
                    if curCount < (l//2)+1 <= newCurCount:
                        ansMedian = k
                    curCount = newCurCount

        ansMode = -1
        ansModeCount = -1
        for k in range(256):
            if count[k]:
                if count[k] > ansModeCount:
                    ansModeCount = count[k]
                    ansMode =
 k 
        return [float(ansMin), float(ansMax), float(ansMean), float(ansMedian), float(ansMode)]
