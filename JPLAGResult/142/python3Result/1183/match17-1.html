class Solution:
    def sampleStats(self, count):
        """
        minimum, maximum, mean, median, and mode
        """
        minimum = -1
        maximum = -1
        median = -1
        mode = -1

        n = sum(count)
        total_sum = 0
        curr_max = 0
        curr_count 
= 0
        for number, freq in enumerate(count):
            if freq != 0:
                maximum = number
                if minimum == -1:
                    minimum = number
            total_sum += number * freq
            curr_count += freq
            if n // 2 <= curr_count and n % 2 == 0 and median == -1:
                median = number
            if n // 2 < curr_count and n % 2 != 0 and median == -1:
                median = number
            if n % 2 == 0 and n // 2 < curr_count <= freq + n // 2:
                median += number
                median /= 2
            if freq > curr_max:
                curr_max = freq
                mode = number
        mean = total_sum / n

        return [float(minimum), float(maximum), mean, float(median), float(mode)]
