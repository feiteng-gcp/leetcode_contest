class Solution:
  def sampleStats(self, count: List[int]) -> List[float]:
    sum_cnt = sum(count)
    mid_cnt = sum_cnt // 2
    is_odd = sum(count) % 2 == 1
    mid = -1 if is_odd else [-1, -1]
    min_ = -1
    max_ = -1
    lot_id = -1
    lot_cnt = -1
    total = 0
    cnt_sofar = 0
    for i in range(256):
      if(min_ == -1 and count[i] != 0):
        min_ = i
      if(count[i] != 0):
        max_ = i
      total += i * count[i]
      cnt_sofar += count[i]
      if(is_odd):
        if(mid == -1 and cnt_sofar >= mid_cnt):
          mid = i
      else:
        if(mid[0] == -1 and cnt_sofar >= mid_cnt):
          mid[0] = i
        if(mid[1] == -1 and cnt_sofar >= mid_cnt + 1):
          mid[1] = i
      if(lot_cnt < count[i]):
        lot_cnt = count[i]
        lot_id = i
    return [float(min_), float(max_), float(total / sum_cnt), float((mid if is_odd else sum(mid)/2)), float(lot_id)]
