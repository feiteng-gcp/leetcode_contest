class Solution:
    def sampleStats(self, count):
        cnt = 0
        total = 0
        mini = 256
        maxi = 0
        mid = 0
        mode = (0,0)

        for i in range(256):
            n = count[i]
            if n > mode[0]:
                mode = (n, i)
            cnt += n
            total += n*i
            if n > 0:
                mini = min(mini, i)
                maxi = max(maxi, i)

        mean = total/cnt
        if cnt % 2 == 1:
            need = cnt // 2
            for i in range(256):
                need -= count[i]
                if need < 0:
                    mid 
= i
                    break
        else:
            need = cnt // 2
            for i in range(256):
                need -= count[i]
                if need < 0:
                    mid = i
                    break
                elif need == 0:
                    f1 = i
                    for j in range(i+1, 256):
                        if count[j] > 0:
                            mid = (f1 + j)/2
                            break
                    break
        return [float(mini), float(maxi), float(mean), float(mid), float(mode[1])]
