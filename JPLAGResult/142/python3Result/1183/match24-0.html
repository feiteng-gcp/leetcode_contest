class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        
        max_freq = 0
        mode = -1
        minimum = float('inf')
        maximum = float('-inf')
        N = 0
        total = 0
        for i, c in enumerate(count):
            if c > 0:
                minimum = min(minimum, i)
                maximum = max(maximum, i)
                if c > max_freq:
                    mode = i
                    max_freq = c
                N += c
                total += i * c
        
        mean = total / N
        
        # find median
        lo = (N + 1) // 2
        hi = (N + 2) // 2
        cum_count = 0
        lo_median = -1
        hi_median = -1
        for i, c in enumerate(count):
            if c > 0:
                cum_count += c
                if lo_median == -1 and cum_count >= lo:
                    lo_median = i
                if hi_median == -1 and cum_count >= hi:
                    hi_median = i
            if lo_median > -1 and hi_median > -1:
                break
        median = (lo_median + hi_median) / 2
        
        return [float(minimum), float(maximum), float(mean), float(median), float(mode)]
