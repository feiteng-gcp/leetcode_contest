class Solution:
    def sampleStats(self, A: List[int]) -> List[float]:
        total_sum = 0
        small = float('inf')
        large = 0
        total_num = 0
        mod = 0
        for i in range(256):
            total_sum += A[i] * i
            total_num += A[i]
            mod = i if A[mod] < A[i] else mod
            small = min(small, i) if A[i] != 0 else small
            large = max(large, i) if A[i] != 0 else large
        mid = total_num // 2
        count = 0
        median1 = None
        median2 = None
        if total_num % 2 == 0:
            for i in range(256):
                count += A[i]
                if count == mid and not median1:
                    median1 = i
                if count > mid and not median2:
                    if not median1:
                        median2 = i
                        break
                    else:
                        median2 = (i + median1) / 2
        else:
            
for i in range(256):
                count += A[i]
                if count >= mid and not median2:
                    median2 = i
                    break
        return [float(small), float(large), float(total_sum/total_num), float(median2), float(mod)]
