class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        mode 
= 0
        mc = 0
        total = 0
        total_num = 0
        miin = 256
        maax = -1
        for k, ck in enumerate(count):
            if ck > mc:
                mode = k
                mc = ck
            total += k*ck
            total_num += ck
            if ck > 0:
                miin = min(miin, k)
                maax = max(maax, k)
        idx1, idx2 = (total_num-1)//2+1, total_num//2+1
        def find(seen, ni, idx):
            while seen + count[ni] < idx:
                seen += count[ni]
                ni += 1
            return ni, seen
        v1, seen = find(0, 0, idx1)
        v2, seen = find(seen, v1, idx2)
        #print(v1, v2, idx1, idx2)
        return [float(miin), float(maax), total/total_num, (v1+v2)/2, float(mode)]
