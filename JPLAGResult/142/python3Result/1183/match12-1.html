# Solution
class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        amin = -1
        for i in range(256):
            if count[i]:
                amin = i
                break
        amax = 256
        
for i in range(255, -1, -1):
            if count[i]:
                amax = i
                break
        total = 0
        tot_count = 0
        for i in range(256):
            total += i * count[i]
            tot_count += count[i]
        mean = total / tot_count
        odd = tot_count % 2 == 1
        mediani = tot_count // 2
        prefc = 0
        median = -1

        for i in range(256):
            if prefc + count[i] < mediani:
                prefc += count[i]
            else:
                if odd or prefc + count[i] > mediani:
                    median = i
                else:
                    for j in range(i + 1, 256):
                        if count[j]:
                            median = (i + j) / 2
                            break
                break

        mode = max(range(256), key=lambda i: count[i])
        return [float(amin), float(amax), float(mean), float(median), float(mode)]
