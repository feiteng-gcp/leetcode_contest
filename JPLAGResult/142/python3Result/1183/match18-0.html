class Solution:
    def sampleStats(self, count: int) -> float:
        _min = -1
        _max = -1
        _sum = 0
        _count = 0
        _mode = -1
        max_c = 0
        for idx, c in enumerate(count):
            if c > max_c:
                max_c = c
                _mode = idx
            _count += c
            if c != 0:
                if _min == -1:
                    _min = idx
                _max = idx
            _sum += c * idx
        _mean = float(_sum) / _count
        mid_cnt = _count / 2
        
        odd = _count % 2 != 0
        _cur = 0
        
        if odd:
            _medium = -1
            for idx, c in enumerate(count):
                if _cur + c >= mid_cnt:
                    _medium = idx
                    break
                _cur += c
        else:
            mid_cnt = int(mid_cnt)
            first = -1
            second = -1
            for idx, c in enumerate(count):
                # print("_cur", _cur, "c", c, "idx", idx, "mid_cnt", mid_cnt)
                if _cur + c >= mid_cnt:
                    if first < 0:
                        first = idx
                    else:
                        second = idx
                        break
                    mid_cnt += 1
                if _cur + c >= mid_cnt:
                    second = idx
                    break
                _cur += c
            _medium = (first + second)/2
                
                
        return [float(_min), float(_max), float(_mean), float(_medium), float(_mode)]
