from collections import Counter
class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        total = sum(count)
        minimum, maximum = None, None
        s, cnt = 0, 0
        median = 0
        mode = 0
        for i, x in enumerate(count):
            if minimum is None and x > 0: minimum = i
            if x > 0: maximum = i
            if x > count[mode]:
                mode = i
            s += i*x
            if total%2:
                if cnt < (total+1)//2 and cnt+x >= (total+1)//2:
                    median = i
            else:
                if cnt < total//2 and cnt+x >= total//2:
                    median += i
                if cnt < total//2+1 and cnt+x >= total//2+1:
                    median += i
                    median /= 2
            cnt += x
        mean = s/total
        return [float(minimum), float(maximum), mean, float(median), float(mode)]
