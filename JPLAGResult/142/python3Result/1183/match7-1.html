class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        total = sum(count)
        min_ = -1
        max_ = -1 
        median_ = -1 
        cumulate = 0
        mode_ = -1 
        maxC = 0
        summation = 0
        for i,c in enumerate(count):
            if c and min_==-1:
                min_ = i 
            if c:
                max_ = i
            cumulate += c 
            summation+= c*i
            if median_==-1:
                if total%2:
                    if cumulate > total//2 :
                        median_ = i 
                else:
                    if cumulate == total//2 :
                        median_ = (i+i+1)/2.0
                    elif cumulate > total//2:
                        median_ = i
            if c > maxC:
                maxC = c
                mode_ = i
        mean_ = summation/total
        return list(map(float,[min_,max_,mean_,median_,mode_]))
