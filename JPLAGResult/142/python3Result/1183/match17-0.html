class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        total = sum(count)
        max_v = max(count)
        mode = count.index(max_v)
        median_idx = None
        min_v,max_v,median,mean,cnt_v = -1,0,0,0,0
        cnt_idx = 0
        if total % 2 == 1:
            median_idx = [total // 2 + 1]
#             1,2,3,4,5,6
        else:
            median_idx = [total // 2, total // 2 + 1]
        left_m,right_m = None,None
        for i,v in enumerate(count):
            if v > 0:
                max_v = i
                if min_v == -1:
                    min_v = i
            cnt_v += (i * v)
            cnt_idx += v
            if len(median_idx) == 1 and cnt_idx - v < median_idx[0] <= cnt_idx:
                median = i
            elif len(median_idx) == 2:
                if cnt_idx >= median_idx[0] and not left_m:
                    left_m = i
                if cnt_idx >= median_idx[1] and not right_m:
                    right_m = i
        if left_m and right_m:
            median = (left_m + right_m) / 2 
        mean = cnt_v / total
            
        
        return [float(min_v),float(max_v),float(mean),float(median),float(mode)]
