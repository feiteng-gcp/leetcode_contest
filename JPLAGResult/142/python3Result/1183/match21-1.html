class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        mini=-1
        n=0
        s=0
        for i in range(256):
            
            s+=i*count[i]
            n+=count[i]
            if count[i]>0:
                maxi=i
                if mini==-1:
                    mini=i
        
        
        i=0
        cnt=0
        while 2*cnt<n:
            cnt+=count[i]
            i+=1
            
        if 2*cnt>n:
            median=i-1
            
        else:
            hold=i-1
            
            while True:
                if count[i]>0:
                    break
                i+=1
            #print(hold,i)
            median=(hold+i)/2
            
        mean=s/n        
        
        mode=0
        modetally=0
       
        for i in range(0,len(count)):
            
if count[i]>modetally:
                modetally=count[i]
                mode=i
                
                    
        return [float(mini), float(maxi), float(mean), float(median), float(mode)]
