class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        average = sum(i * cnt for i, cnt in enumerate(count)) / sum(count)
        n = len(count)
        
for i in range(n):
            if count[i] != 0:
                min_val = i
                break
        for i in range(n - 1, -1, -1):
            if count[i] != 0:
                max_val = i
                break
        mode = max((cnt, i) for i, cnt in enumerate(count))[-1]
        
        pre = [0 for _ in range(n)]
        pre[0] = count[0]
        for i in range(1, n):
            pre[i] = pre[i - 1] + count[i]
        
        total = sum(count)
        
        if total % 2 == 1:
            median = self.getKth(pre, (total + 1) // 2)
        else:
            small = self.getKth(pre, total // 2)
            large = self.getKth(pre, total // 2 + 1)
            median = (small + large) / 2
        
        return min_val * 1.0, max_val * 1.0, average * 1.0, median * 1.0, mode * 1.0
    
    def getKth(self, pre, k):
        print(k, pre)
        n = len(pre)
        i = 0
        while i < n and pre[i] < k:
            i += 1
        return i
