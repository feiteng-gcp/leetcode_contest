class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        min_v, max_v, mean_v, median_v, mode_v = -1, -1, 0, -1, -1
        s, cnt = sum(count), 0
        most_freq = 0
        for i, c in enumerate(count):
            if min_v == -1 and c:
                min_v = i
            if c:
                max_v = i
                if c > most_freq:
                    most_freq = c
                    mode_v = i
            mean_v += c * i
            cnt += c
            if median_v == -1:
                if s % 2:
                    if cnt > s//2:
                        median_v = i
                else:
                    if cnt > s//2:
                        median_v = i
                    elif cnt == s//2:
                        median_v = i
                        j = i + 1
                        while j < 256 and not count[j]: j+= 1
                        median_v += j
                        median_v /= 
2.0
        mean_v /= s
        return [float(min_v), float(max_v), float(mean_v), float(median_v), float(mode_v)]
