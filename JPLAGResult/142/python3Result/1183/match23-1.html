class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
      minn = maxx = mean = median = 0.0
      for i in range(256):
        if count[i] > 0:
          minn += i
          break
      for i in reversed(range(256)):
        if count[i] > 0:
          maxx += i
          break
      sums = cnt 
= 0
      mcnt = mode = 0
      for i in range(256):
        sums += count[i] * i
        cnt += count[i]
        if count[i] > mcnt:
          mcnt = count[i]
          mode = i
      mean = sums / cnt
      cnt1 = 0
      for i in range(256):
        cnt1 += count[i]
        if cnt % 2 == 1:
          if cnt1 >= cnt // 2 + 1:
            return [minn, maxx, mean, i + 0.0, mode + 0.0]
        if cnt % 2 == 0:
          if cnt1 == cnt // 2:
            for j in range(i + 1, 256):
              if count[j] > 0:
                return [minn, maxx, mean, (i + j) / 2, mode + 0.0]
          elif cnt1 > cnt // 2:
            return [minn, maxx, mean, i + 0.0, mode + 0.0]
