class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        for i in range(256):
            if count[i]>0:
                minn=i
                break
        for i in range(255,-1,-1):
            if count[i]>0:
                maxx=i
                break
        s=0
        total=0
        for i in range(256):
            total+=count[i]
            s+=i*count[i]
        mean=s/total
        Max=0
        imax=0
        for i in range(256):
            if count[i]>Max:
                Max=count[i]
                imax=i
        mode=imax
        now=0
        if total%2==1:
            v=(total+1)//2
            for i in range(256):
                now+=count[i]
                if now>=v:
                    median=i
                    break
        else:
            median1=None
            v=total//2
            for i in range(256):
                now+=count[i]
                if now>=v and median1==None:
                    median1=i
                if now>=v+1:
                    median2=i
                    break
            median=(median1+median2)/2
        return [float(minn), float(maxx), float(mean), float(median), float(mode)]
