class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        valid = [i for i in range(256) if count[i] > 0]
        minimum = min(valid)
        maximum = max(valid)
        mean = sum([count[i]*i for i in valid])/sum(count)
        mode = max(range(256), key=lambda x: count[x])
        total = sum(count)
        halfcount = total//2
        if halfcount * 2 == total:
            for index_i, i in enumerate(valid):
                halfcount -= count[i]
                if halfcount == 0:
                    # print(i, valid[index_i+1])
                    median = (i + valid[index_i+1])/2.0
                    break
                elif halfcount < 0:
                    median = i
                    break
        else:
            halfcount += 1
            for index_i, i in enumerate(valid):
                halfcount -= count[i]
                if halfcount <= 0:
                    median = i
                    break
        return [float(x) for x in [minimum, maximum, mean, median, mode]]
