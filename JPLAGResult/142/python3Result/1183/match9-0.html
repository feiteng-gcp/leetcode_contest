class Solution:
    def sampleStats(self, A: List[int]) -> List[float]:
        n = len(A)
        a = -1
        b = c = d = e = f = g = 0
        s = sum(A)
        m = max(A)
        t = s // 2
        for i in range(n):
            if A[i] and a < 0:
                a = i
            if A[i]:
                b = i
                c += A[i]*i
                f += A[i]
                if s % 2 == 1:
                    if f > t and e == 0:
                        e = i
                else:
                    if f > t-1 and g == 0:
                        g = i
                    if f > t and e == 0:
                        e = i
                        e = (e + g) / 2

            if A[i] == m: d = i

        return float(a),float(b),c/f,float(e),float(d)
