class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        res=[]
        ma=0
        cma=0
        cmi=0
        su=0
        times=0
        for i in range(256):
            if count[i]:
                if not res:
                    res.append(float(i))
                ma=i
            if cma<count[i]:
                cma=count[i]
                cmi=i
            times+=count[i]
            su+=(count[i]*i)
        res.append(float(ma))
        if times:
            res.append(su/times)
        else:
            for i in range(4):
                res.append(float(0))
            return res
        temp=times
        if times%2==1:
            for i in range(256):
                temp-=count[i]
                if temp<=((times-1)/2):
                    res.append(float(i))
                    break
        else:
            for i in range(256):
                temp-=count[i]
                if temp==(times/2):
                    res.append((i+i+1)/2)
                    break
                elif temp<(times/2):
                    res.append(float(i))
                    break
        res.append(float(cmi))
        return res
