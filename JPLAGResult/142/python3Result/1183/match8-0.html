class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        smp_cnt = sum(count)
        minn = 255
        maxx = 0
        moden = 0
        modev = 0
        avgg = 0.0
        m1 = 0
        m2 = 0
        m1i = smp_cnt // 2 + 1
        m2i = (smp_cnt - 1) // 2 + 1
        curcnt = 0
        for i in range(256):
            if not count[i]:
                continue
            maxx = max(i, maxx)
            minn = min(i, minn)
            if moden < count[i]:
                modev = i
                moden = count[i]
            avgg += i * count[i] / smp_cnt
            if curcnt < m1i and curcnt + count[i] >= m1i:
                m1 = i
            if curcnt < m2i and curcnt + count[i] >= m2i:
                m2 = i
            curcnt += count[i]
        return [minn * 1.0, maxx * 1.0, avgg * 1.0, (m1 + m2) / 2.0, modev * 1.0]
