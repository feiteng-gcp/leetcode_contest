class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        s = 0
        c = 0
        maximum = 0
        minimum = -1
        mode = [0, 0]
        for n in range(len
(count)):
            s += n * count[n]
            c += count[n]
            if count[n] != 0:
                maximum = n
                if minimum == -1:
                    minimum = n
            if mode[0] < count[n]:
                mode[0] = count[n]
                mode[1] = n
        
        
        median = 0
        cur = 0
        l = c // 2
        r = l if c % 2 == 1 else l + 1
        a = -1
        b = -1
        mean = s / c
        for n in range(len(count)):
            cur += count[n]
            if cur >= l and a == -1:
                a = n
            if cur >= r and b == -1:
                b = n
            if a != -1 and b != -1:
                median = (a + b) / 2
                break
        
        
        return [float(minimum), float(maximum), float(mean), float(median), float(mode[1])]
