from collections import Counter

class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        s = 0
        m, M = 255, 0
        mode, freq = 0, 0
        median1 = -1
        median2 = -1
        
        total_count = sum(count)
        cur_count = 0
        
        for i, c in enumerate(count):
            if c != 0:
                s += (i * c)
                m = min(m, i)
                M = max(M, i)
                
                if c > freq:
                    freq = c
                    mode = i
                    
                cur_count += c
                
                if total_count % 2 == 1:
                    if cur_count > total_count/2 and median1 == -1:
                        median1 = i
                else:
                    if cur_count >= total_count//2 and median1 == -1:
                        median1 = i
                    if cur_count >= total_count//2 + 1 and median2 == -1:
                        median2 = i
                        median1 = (median1 + median2)/2.0
                    
                
        
        return [float(m), float(M), float(s/total_count), float(median1), float(mode)]
