class Solution:
    def sampleStats(self, count: List[int]) -> List[float]:
        minimum = float(min(i for i, x in enumerate(count) if x > 0))
        maximum = float(max(i for i, x in enumerate(count) if x > 0))
        mean = sum(float(x) * i for i, x in enumerate(count)) / sum(count)
        mode = float(max((x, i) for i, x in enumerate(count))[1])
        
        c = sum(count)
        j = 0
        if c % 2 == 1:
            for i, x in enumerate(count):
                if j + x >= (c // 2) + 1:
                    median = float(i)
                    break
                j += x
        else:
            a, b = None, None
            for i, x in enumerate(count):
                if a is not None and b is not None:
                    break
                if a is None and j + x >= c // 2:
                    a = i
                if b is None and j + x >= (c // 2) + 1:
                    b = i
                j += x
            median = float((a + b) / 2)
        
        
        return [minimum, maximum, mean, median, mode]
