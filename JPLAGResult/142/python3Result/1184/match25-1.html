class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        points = []
        for n,start,end in trips:
            points
.append((start,n))
            points.append((end,-n))
        points.sort()
        cap = 0
        for time,delta in points:
            cap += delta
            if cap > capacity:
                return False
        return True
