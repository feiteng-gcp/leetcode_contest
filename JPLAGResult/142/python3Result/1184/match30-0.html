class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        events = []
        for count, start, end in trips:
            events.append([start, count])
            events.append([end, -count])
        
        events.sort()
        count = 0
        for event in events:
            count += event[1]
            if count > capacity:
                return False
        return True
