class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        N = 1001
        pick = [0] * N
        drop = [0] * N
        
        for num, start, end in trips:
            pick[start] += num
            drop[end] += num
            
        cur = 0
        
        for i in range(N):
            p, d = pick[i], drop[i]
            cur += p - d
            
            #if p != 0 or d != 0:
                #print(p, d, cur)
            if cur > capacity:
                return False
            
        return True
