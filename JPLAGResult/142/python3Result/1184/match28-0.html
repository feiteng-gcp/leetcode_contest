class Solution:
    def carPooling(self, trips, capacity: int) -> bool:
        pick = [0 for _ in range(1001)]
        drop = [0 for _ in range(1001)]

        for a, b, c in trips:
            pick[b] += a
            drop[c] += a

        for i in range(1001):
            capacity += drop[i]
            capacity -= pick[i]
            if capacity < 0 :
                return False

        return True
