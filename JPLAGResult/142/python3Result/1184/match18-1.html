from itertools import chain
class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        ops = list(chain.from_iterable(map(lambda trip: [[trip[1], trip[0]], [trip[2], -trip[0]]], trips)))
        ops
.sort()
        
        now = 0
        for op in ops:
            now += op[1]
            if now > capacity:
                return False
        return True
