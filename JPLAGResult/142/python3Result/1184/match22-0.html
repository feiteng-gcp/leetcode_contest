class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        pe=[0 for _ in range(1001)]
        for trip in trips:
            for i in range(trip[1],trip[2]):
                pe[i]+=trip[0]
        #print(pe[:20])
        for i in range(len(pe)):
            if pe[i]>capacity:
                return False
        return True
