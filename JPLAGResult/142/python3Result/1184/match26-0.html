class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        intervals = []
        for n,s,e in trips:
            intervals.append((s,1,n))
            intervals.append((e,0,n))
            # (x[1], x[2])
        intervals = sorted(intervals, key = lambda x: (x[0],x[1]))
        
        cnt = 0
        max_cnt = 0
        for p,v,num in intervals:
            if v == 0:
                cnt -= num
            else:
                cnt += num
            max_cnt = max(max_cnt,cnt)
        return max_cnt <= capacity
