class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        hm = collections.Counter()
        for passengers, start, finish in trips:
            hm[start] += passengers
            hm[finish] -= passengers
        a = [e for e in hm.items() if e[1] != 0]
        a.sort()
        for pos, pas in a:
            capacity -= pas
            if capacity < 0:
                return False
        return True
