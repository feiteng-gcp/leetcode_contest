class Solution:
    def carPooling(self, trips, capacity: int) -> bool:
        p 
= [0] * 1001
        for n,l,r in trips:
            p[l] += n
            p[r] -= n
        s = 0
        for x in p:
            s += x
            if s > capacity:
                return False
        return True
