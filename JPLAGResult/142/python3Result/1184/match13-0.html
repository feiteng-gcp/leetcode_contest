class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        points = []
        for num, start, end in trips:
            points.append((start, 1, num))
            points.append((end, 0, num))
        points.sort()
        total = 0
        for loc, ptype, num in points:
            if ptype:
                total += num
            else:
                total -= num
            if total > capacity: return False
        return True
