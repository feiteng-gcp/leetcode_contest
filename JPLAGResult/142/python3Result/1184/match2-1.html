class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        diff_passengers_at = [0] * 1001
        for n, s, e in trips:
            diff_passengers_at[s] += n
            diff_passengers_at[e] -= n

        cur_num_passengers = 0
        for d in diff_passengers_at:
            cur_num_passengers += d
            if cur_num_passengers > capacity:
                return False
        return True
