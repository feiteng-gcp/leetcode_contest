class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        events = []
        for [num, start, end] in trips:
            events.append ((start, 'pick', num))
            events.append ((end, 'drop', num))
        events.sort ()
        size = 0
        for (loc, kind, num) in events:
            if kind == 'pick':
                size += num
                if size > capacity:
                    return False
            else:
                size -= num
        return True
