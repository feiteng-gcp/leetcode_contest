class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        dic = {}
        for trip in trips:
            if trip[1] not in dic:
                dic[trip[1]] = trip[0]
            else:
                dic[trip[1]] += trip[0]
            if trip[2] not in dic:
                dic[trip[2]] = -trip[0]
            else:
                dic[trip[2]] -= trip[0]
        pos = list(dic.keys())
        pos.sort()
        count = 0
        for p in pos:
            count += dic[p]
            if count>capacity:
                return False
        return True
