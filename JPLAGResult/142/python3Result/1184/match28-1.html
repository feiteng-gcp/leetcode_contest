class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
      a = [0] * 1001
      b = [0] * 1001
      for t in trips:
        a[t[1]] += t[0]
        b[t[2]] += t[0]
      c = 0
      
for i in range(1001):
        c -= b[i]
        c += a[i]
        if c > capacity:
          return False
      return True
