class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        events = []
        for (num, start, end) in trips:
            # dropping off is event type 0
            events.append((start, 1, num))
            events.append((end, 0, num))
            
        events.sort()
            
        max_people = 0
        
        for time, evt_type, num in events:
            if evt_type == 1:
                max_people += num
            else:
                max_people -= num
                
            if max_people > capacity:
                return False
        
        return True
