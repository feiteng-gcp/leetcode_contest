class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        state 
= [0] * 1001
        for x, i, j in trips:
            state[i] += x
            state[j] -= x
        cnt = 0
        for x in state:
            cnt += x
            if cnt > capacity:
                return False
        return True
