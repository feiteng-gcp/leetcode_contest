class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        add = [0]*(1010)
        for c, start, end in trips:
            add[start] += c
            add[end] -= c
        cur = 0
        for i in range(1010):
            cur += add[i]
            if cur > capacity:
                return False
        return True
