class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        stream = [(s, p) for p, s, _ in trips] + [(e, -p) for p, _, e in trips]
        stream.sort()
        
        cnt = 0
        for t, p in stream:
            cnt += p
            if cnt > capacity:
                return False
        return True
