class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        L = []
        for trip in trips:
            n, p1, p2 = trip
            L.append((p1, n))
            L.append((p2, -n))

        L
.sort()
        n = 0
        for e in L:
            n += e[1]
            if n > capacity:
                return False
        return True
