class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        ts = []
        for t in trips:
            ts.append((t[1], 1, t[0]))
            ts.append((t[2], 0, -t[0]))
        ts
.sort()
            
        for t in ts:
            capacity -= t[2]
            if capacity < 0:
                return False
        return True
