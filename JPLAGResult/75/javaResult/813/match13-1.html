class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        List<List<Integer>> answer = new ArrayList<>();
        if (graph == null || graph.length == 0 || graph[0].length == 0) {
            return answer;
        
}
        helper(0, graph, new ArrayList<>(), answer);
        return answer;
    }
    
    private void helper(int cur, int[][] graph, List<Integer> path, List<List<Integer>> answer) {
        if (cur == graph.length - 1) {
            // reached last node
            path.add(cur);
            answer.add(new ArrayList<>(path));
            path.remove(path.size() - 1);
            return;
        }
        
        path.add(cur);
        int[] out = graph[cur];
        for (int o : out) {
            helper(o, graph, path, answer);
        }
        path.remove(path.size() - 1);
    }
}
