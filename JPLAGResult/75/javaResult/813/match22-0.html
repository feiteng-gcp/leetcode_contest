import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    Boolean[] f;
    int n;

    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        n = graph.length;
        f = new Boolean[n];

        List<List<Integer>> ans = new ArrayList<>();
        dfs(0, graph, new ArrayList<>(), ans);


        return ans;
    }

    private boolean f(int u, int[][] g) {
        if (u == n - 1) return true;
        if (f[u] != null) return f[u];

        boolean ans = false;
        for (int v : g[u]) {
            if (f(v, g)) {
                ans = true;
                break;
            }
        }
        f[u] = ans;
        return ans;
    }

    private void dfs(int u, int[][] g, List<Integer> cur, List<List<Integer>> ans) {
        cur.add(u);

        if (u == n - 1) {
            ans.add(new ArrayList<>(cur));
        } else {
            for (int v : g[u]) {
                if (f(v, g)) {
                    dfs(v, g, cur, ans);
                }
            }
        }

        cur.remove(cur.size() - 1);
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        int[][] g = {
                {},
                {3},
                {3},
                {},
        };
        System.out.println(sol.allPathsSourceTarget(g));
    }
}
