class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        /*int[][] paths=new int[graph.length][graph.length];
        for (int y=0;y<graph.length;y++)
        {
            for (int x=0;x<graph[y].length;x++)
            {
                paths[y][graph[y][x]]=1;
            }
        }*/
        int target=graph.length-1;
        Queue<Path> q=new LinkedList<Path>();
        LinkedList<Integer> ini=new LinkedList<Integer>();
        List<List<Integer>> result=new LinkedList<List<Integer>>();
        ini.add(0);
        Path h=new Path(ini);
        q.offer(h);
        while (q.size()>0)
        {
            Path cur=q.poll();
            int now=
cur.ePath.getLast();
            //System.out.println(now);
            if (now==target)
            {
                result.add(cur.ePath);
            }
            else
            {
                for (int x=0;x<graph[now].length;x++)
                {
                    LinkedList<Integer> next=(LinkedList<Integer>)cur.ePath.clone();
                    next.add(graph[now][x]);
                    Path nr=new Path(next);
                    
q.offer(nr);
                }
            }
        }
        return result;
    }
    
    class Path{
        public LinkedList<Integer> ePath;
        public Path(LinkedList<Integer> p){
            ePath=p;
        }
    }
}
