class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        List<List<Integer>> res = new LinkedList();
        dfs(graph,0,new LinkedList(),res);
        return res;
    }
    
    public void dfs(
int[][] graph, int nodeId, LinkedList<Integer> path, List<List<Integer>> res){
        if (nodeId == graph.length-1){
            path.add(nodeId);
            res.add(new LinkedList(path));
            path.removeLast();
            
return;
        }
        path.add(nodeId);
        for (int next = 0; next < graph[nodeId].length; next++){
            dfs(graph, graph[nodeId][next], path, res);
        }
        path.removeLast();
    }
    
}
