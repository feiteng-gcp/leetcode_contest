class Solution {
    List<List<Integer>> ans = new ArrayList<>();
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        
fill(graph, 0, new ArrayList<>());
        return ans;
    }
    void fill(int [][] graph, int pp, ArrayList<Integer> n) {
        if(pp == graph.length - 1) ans.add(n);
        
        n.add(pp);
        for(int v : graph[pp]) {
            fill(graph, v, new ArrayList<>(n))
;        }
    }
}
