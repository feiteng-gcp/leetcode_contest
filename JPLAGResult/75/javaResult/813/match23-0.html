	class Solution {
		List<List<Integer>> ret;
		
	    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
	    	ret = new ArrayList<>();
	    	List<Integer> path = new ArrayList<>();
	        dfs(0, graph, path);
	        return ret;
	    }
	    
	    void dfs(int cur, int[][] g, List<Integer> path)
	    {
	    	path.add(cur);
	    	if(cur == g.length-1){
	    		ret.add(new ArrayList<>(path));
	    	}else{
		    	for(int e : g[cur]){
		    		dfs(e, g, path);
		    	}
	    	}
	    	path.remove(path.size()-1);
	    }
	}
