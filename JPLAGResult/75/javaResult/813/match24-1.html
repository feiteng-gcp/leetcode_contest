class Solution {
    
    public void pathRec(int[][]G ,List<List<Integer>> allPaths, LinkedList<Integer> path, int src, int dest, boolean [] visited) {
        visited[src] = true;
        path.add(src);
        if (src==dest) {
            LinkedList<Integer> pathCopy = new LinkedList<>(path);
            allPaths.add(pathCopy);
        }else {
            for (int x: G[src]) {
                if (!visited[x]) 
{
                    pathRec(G,allPaths,path,x,dest,visited);
                }
            }
        }
        path.removeLast();
        visited[src] = false;
    }
    
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        List<List<Integer>> allPaths = new LinkedList<>();
        pathRec(graph, allPaths, new LinkedList<Integer>(), 0, graph.length-1,new boolean [graph.length] );
        return allPaths;
    }
}
