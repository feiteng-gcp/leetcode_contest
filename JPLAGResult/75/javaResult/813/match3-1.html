class Solution {
    List<List<Integer>> list = new ArrayList<>();
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        
        List<Integer> cur_list = new ArrayList<Integer>();
        cur_list.add(0);
        dfs(graph, list, cur_list, 0);
        
return list;
    }
    public void dfs(int[][] graph, List<List<Integer>> list, List<Integer> cur_list, int cur){
        // System.out.println(cur_list);
        if (cur == graph.length - 1){
            // System.out.println(cur_list);
            list.add(new ArrayList<Integer>());
            for (int i = 0; i < cur_list.size(); i++){
                list.get(list.size()-1).add(cur_list.get(i));
            }
            return;
        }
        for (int next: graph[cur]){
            cur_list.add(next);
            dfs(graph, list, cur_list, next);
            cur_list.remove(cur_list.size() - 1);
        }
    }
}
