class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        return apHelper(graph, 0);
    }
    
    public List<List<Integer>> apHelper(int[][] graph, int u) {
        List<List<Integer>> result = new ArrayList<>();
        if (u==graph.length-1) {
            result.add(new ArrayList<>());
            result.get(0).add(u);
            return result;
        }
        for (int v : graph[u]) {
            List<List<Integer>> partial = apHelper(graph, v);
            for (List<Integer> list : partial) {
                List<Integer> nl = new ArrayList<>();
                nl.add(u);
                nl.addAll(list);
                result.add(nl);
            }
        }
        return result;
    }
}
