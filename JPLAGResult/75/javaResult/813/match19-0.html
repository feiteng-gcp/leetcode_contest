class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        List<List<Integer>> result = new ArrayList<>();
        LinkedList<Integer> tmp = new LinkedList<>();
        tmp.add(0);
        add(0, graph.length - 1, tmp, result, graph);
        return result;
    }
    
    private static void add(int start, int end, LinkedList<Integer> tmp, List<List<Integer>> result, int[][] graph) {
        if (start == end) {
            result.add(new ArrayList<>(tmp));
            return;
        }
        for (int next : graph[start]) {
            tmp.add(next);
            add(next, end, tmp, result, graph);
            tmp.removeLast();
        }
    }
}
