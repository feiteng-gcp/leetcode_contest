class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        List<List<Integer>> result = new ArrayList<>();
        
        if(graph==null || graph.length==0) return result;
        
        
dfs(0, graph.length-1, new ArrayList<Integer>(), result, graph);
        
        return result;
    }
    
    void dfs(int start, int target, List<Integer> curr, List<List<Integer>> result, int[][] graph) {
        if(start == target) {
            curr.add(target);
            result.add(new ArrayList<Integer>(curr));
            curr.remove(new Integer(target));
            return;
        }
        
        curr.add(start);
        
        for(int neighbor: graph[start]) {
            dfs(neighbor, target, curr, result, graph);
        }
        
        curr.remove(new Integer(start));
    }
}
