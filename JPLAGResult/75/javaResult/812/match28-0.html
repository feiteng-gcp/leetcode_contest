class Solution {
    public boolean rotateString(String A, String B) {
        if (A == null || B == null || A.length() != B.length()) return false;
        for (int i = 0; i < A.length(); i++) {
            boolean flag = true;
            for (int j = 0; j < A.length() && flag; j++) {
                if (A.charAt((i+j)%A.length()) != B.charAt(j))
                    flag = false;
            }
            if (flag) return true;
        }
        return false;
    }
}
