class Solution {
    public boolean rotateString(String A, String B) {
        if (A.length() != B.length())
            return false;
        char[] a = A.toCharArray();
        char[] b = B.toCharArray();
        int len = a.length;
        for (int i = 0; i < len; i++) {
            if (a[i] == b[0]) {
                boolean match = true;
                for (int j = 1; j < len; j++) {
                    if (a[(j + i) % len] != b[j]) {
                        match = false;
                        break;
                    }
                }
                if (match)
                    return true
;
            }
        }
        return false;
    }
}
