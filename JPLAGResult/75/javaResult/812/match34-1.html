class Solution {
    String my_shift(String A) {
        StringBuilder sb = new StringBuilder();
        for (int i = 1; i < A.length(); ++i) sb.append(A.charAt(i));
        sb.append(A.charAt(0));
        return sb.toString();
    }
    
public boolean rotateString(String A, String B) {
        if (A.length() != B.length()) return false;
        int len = A.length();
        String AA = A;
        while (len > 0) {
            len -- ;
            AA= my_shift(AA);
            if (AA.equals(B)) return true;
        }
        return false;
        
    }
}
