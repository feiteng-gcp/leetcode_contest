class Solution {
    public boolean rotateString(String A, String B) {
        if (A.length() != B.length())
            return false;
        if (A.equals(B))
            return true;
        for (int i = 1; i < A.length(); i++){
            if (B.equals(A.substring(i, A.length()) + 
A.substring(0, i)))
                return true;
        }
        return false;
    }
}
