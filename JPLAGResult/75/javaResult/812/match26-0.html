class Solution {
    public boolean rotateString(String A, String B) {
        char[] s1 = A.toCharArray();
        char[] s2 = B.toCharArray();
        int len = s1.length;
        if(len!=s2.length)
            return false;
        for(int i=0;i<len;i++){
            int j=0;
            for(;j<len;j++){
                if(s1[j]!=s2[(i+j)%len])
                    break;
            }
            if(j>=len)
                return true;
        }
        return false;
    }
}
