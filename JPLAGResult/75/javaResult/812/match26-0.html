class Solution {
    public boolean rotateString(String A, String B) {
     
    	if (A.length() == B.length()) {
    		if (A.equals(B)) {
    			return true;
    		} else {
    			for (int i = 0;i < A.length();i ++) {
    				String temp = A.substring(i + 1);
    				temp += A.substring(0 , i + 1);
    				if (temp.equals(B)) {
    					return true;
    				}
    			}
    			return false;
    		}
    	} else {
    		return false;
    	}
    	
    }
}
