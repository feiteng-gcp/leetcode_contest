class Solution {
    public boolean rotateString(String A, String B) {
        if (A.equals(B)) return true;
        int len = A.length();
        if (len != B.length()) return false;
        for (int i = 1; i < len; i++) {
            String temp = B.substring(i) + B.substring(0, i);
            if (temp.equals(A)) return true;
        }
        return false;
    }
}
