class Solution {
    public boolean rotateString(String A, String B) {
        int l1 = A.length(), l2 = B.length();
        if (l1 != l2) return false;
        if (l1 == 0) 
return true;
        
        for (int sh = 0; sh < l1; ++sh) {
            boolean match = true;
            for (int i = 0; i < l1; ++i) {
                if (A.charAt(((sh+i) % l1)) != B.charAt(i)) {
                    match = false;
                    break;
                }
            }
            if (match) {
                return true;
            }
        }
        
        return false;
    }
}
