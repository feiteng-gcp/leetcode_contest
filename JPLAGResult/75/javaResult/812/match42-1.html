class Solution {
    public boolean rotateString(String A, String B) {
        // edge case
        if (A.length() != B.length()) {
            return false;
        }
        
        int[] arr = new int[256];
        for (char c : A.toCharArray()) {
            arr[c]++;
        }
        for (char c : B.toCharArray()) {
            arr[c]--;
        }
        for (int i = 0; i < 256; ++i) {
            if (arr[i] != 0) {
                return false;
            }
        }
        
        // main idea
        for (int i = 0; i < A.length(); ++i) {
            if (A.charAt(i) == B.charAt(0)) {
                int left = i;
                int right = A.length() - i;
                if (A.substring(i).equals(B.substring(0, right)) && A.substring(0, left).equals(B.substring(right))) {
                    return true;
                }
            }
        }
        
        return false;
    }
}
