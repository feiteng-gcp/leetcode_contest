class Solution {
    public boolean rotateString(String A, String B) {
        if (A.length() != B.length()) return false;
        for (int i = 0; i < A.length(); ++i) {
            String str = A.substring(i, 
A.length()) + A.substring(0, i);
            if (B.equals(str)) return true;
        }
        return false;
    }
}
