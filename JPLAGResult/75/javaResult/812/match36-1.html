class Solution {
    public boolean rotateString(String A, String B) {
        return rotateString(A.toCharArray(), B.toCharArray());
    }
    
    private boolean rotateString(char[] A, char[] B) {
        int n = A.length;
        
if (B.length != n) {
            return false;
        }
        if (n==0) {
            return true;
        }
        for(int i=0;i<n;i++) {
            if(isRotate(A,B,i)) {
                return true;
            }
        }
        return false;
    }
    
    private boolean isRotate(char[] A, char[] B, int k) {
        int n = A.length;
        for(int i=0;i<n;i++) {
            if(A[i] != B[(i+k)%n]) {
                return false;
            }
        }
        return true;
    }
}
