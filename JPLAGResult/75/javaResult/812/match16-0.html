class Solution {
    public boolean rotateString(String A, String B) {
        char chs[] = A.toCharArray();
        //System.out.println(rot(chs));
         if(A.equals(B)){
                return true;
            }
        for(int i=0;i<A.length()-1;i++){
            A = rot(chs);
            chs = A.toCharArray();
            if(A.equals(B)){
                return true;
            }
        }
        return false;
    }
    private static String rot(char[] a){
        char temp = a[0];
        for(int i=0;i<a.length-1;i++){
            a[i] = a[i+1];
        }
        a[a.length - 1] = temp;
        return (new String(a));
    }
    
}
