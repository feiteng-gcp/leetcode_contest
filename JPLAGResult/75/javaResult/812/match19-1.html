class Solution {
	public boolean rotateString(String A, String B) {
		if (A.length() != B.length())
			return false;
		for (int i = 0; i < A.length(); i++) {
			int j = 0;
			for (; j < B.length(); j++)
				if (A.charAt((i + j) % A.length()) != B.charAt(j))
					break;
			if (j == A.length())

				return true;
		}
		return false;
	}
}
