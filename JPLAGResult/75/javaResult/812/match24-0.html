class Solution {
    public boolean rotateString(String A, String B) {
        
        if(A.length()!=B.length())
            return false;
        
        
        StringBuilder sb = new StringBuilder();
        String C = null;
        
        for(int i=0;i<A.length();i++)
        {
            if(A.charAt(i)!=B.charAt(0))
                sb.append(A.charAt(i));
            else
            {   
                C = A.substring(i) + sb.toString();
                if(C.equals(B))
                    return true;
                else
                    sb.append(A.charAt(i));
            }
        }
        
       
        return false;
        
    }
}
