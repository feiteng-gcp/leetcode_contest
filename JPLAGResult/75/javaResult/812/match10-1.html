class Solution {
    public boolean rotateString(String A, String B) {
        if (A.length()!=B.length())
            return false;
        else if (A.equals(B))
            return true;
        else
        {
            String C=A;
            for (int i=0;i<A.length();i++)
            {
                C=
C.substring(1,C.length())+C.charAt(0);
                //System.out.println(C);
                if (C.equals(B))
                    return true;
            }
            return false;
        }
    }
}
