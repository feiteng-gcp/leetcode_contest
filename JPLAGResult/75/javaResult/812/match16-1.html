class Solution {
    public boolean rotateString(String A, String B) {
        if (A == null && B == null)
            return true;
        if (A.length() != B.length())
            return false;
        
for (int i = 0; i < A.length(); i++){
            String temp = A.substring(i)+ A.substring(0,i);
            if (temp.equals(B))
                return true;
        }
        return false;
    }
}
