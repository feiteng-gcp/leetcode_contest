class Solution {
    public boolean rotateString(String A, String B) {
        if (A.length() != B.length()) {
            return false;
        }
        if (A.equals(B)) {
            return true;
        }
        StringBuilder temp = new StringBuilder(A);
        temp.append(temp.charAt(0)).deleteCharAt(0);
        while (!temp.toString().equals(A)) {
            if (temp.toString().equals(B)) {
                return true;
            }
            temp.append(temp.charAt(0)).deleteCharAt(0);
        }
        return false;
    }
}
