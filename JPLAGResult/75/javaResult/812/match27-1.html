class Solution {
    public boolean rotateString(String A, String B) {
        int lA = A.length();
        int lB = B.length();
        if (lA != lB) return false;
        char[] arr = new char[lA*2];
        for (int i = 0; i < lA * 2; ++i){
            int idx = i % lA;
            arr[i] = A.charAt(idx);
            // System.out.print(arr[i] + " ");
        }
        
        for (int i = 0; i < lA; ++i){
            int cur = 0;
            boolean ok = true;
            for (int j = i; j < (i+lA); ++j){
                if (arr[j] != 
B.charAt(cur)){
                    ok = false;
                    break;
                }
                cur++;
            }
            if (ok) return true;
        }
        return false;
    }
}
