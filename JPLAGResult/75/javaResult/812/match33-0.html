class Solution {
    public boolean rotateString(String A, String B) {
        if (A.length() != B.length()) return false;
        for (int rotation = 0; rotation < A.length(); rotation++) if (same(A, B, rotation)) return true;
        return false;
    }
    
    private static boolean same(String A, String B, int rotation) {
        for (int i = 0; i < A.length(); i++) if (B.charAt(i) != A.charAt((i + rotation) % A.length())) return false;
        return true;
    }
}
