class Solution {
    public boolean rotateString(String A, String B) {
        if (A.length()!=B.length())
            return false;
        for (int i = 0; i < A.length(); i++) {
            boolean can = true;
            for (int j = 0; j < A.length(); j++) {
                if (A.charAt(j) != B.charAt((j+i)%A.length())) {
                    can = false;
                    break;
                }
            }
            if (can)
                return true;
        }
        return false;
    }
}
