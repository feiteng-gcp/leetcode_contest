class Solution {
    public double champagneTower(int poured, int query_row, int query_glass) {
        double[] state = new double[query_row+1];
        
        state[0] = poured;
        for(int i = 0; i < query_row; i++){
            for(int j=i; j>=0; j--){
                if(state[j] > 1){
                    double half = (state[j] - 1.0)/2;
                    state[j+1] += half;
                    state[j] = half;
                }else state[j] = 0;
            }
        }
        
        
        return state[query_glass] > 1 ? 1 : state[query_glass];
    }
}
