class Solution {
      public double champagneTower(int poured, int query_row, int query_glass) {
        double[][] res = new double[100][100];
        res[0][0] = poured;
        for (int i = 0; i < 99; i++) {


            for (int r = 0; r <= i; r++) {
                if (res[i][r] > 1) {
                    double ex = res[i][r] - 1;
                    res[i][r] = 1;
                    res[i + 1][r] += ex / 2;
                    res[i + 1][r + 1] += ex / 2;
                }

            }
//            System.out.println(Arrays.toString(res[i]));
        }

        return res[query_row][query_glass];
    }
}
