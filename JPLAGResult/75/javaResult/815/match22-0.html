class Solution {
    public double champagneTower(int poured, int query_row, int query_glass) {
        double[] cache = new double[1];
        cache[0] = poured;
        for (int i=0; i<query_row; i++) {
            double[] newcache = new double[i+2];
            for (int j=0; j<=i; j++) {
                if (cache[j] > 1)
                {
                    newcache[j] += (cache[j] - 1) / 2;
                    newcache[j+1] += (cache[j] - 1) / 2;
                }
            }
            cache = newcache;
        }
        
        return Math.min(cache[query_glass], 1);
    }
}
