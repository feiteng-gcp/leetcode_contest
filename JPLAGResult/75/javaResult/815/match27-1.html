import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    public double champagneTower(int poured, int query_row, int query_glass) {
        int n = 100;
        
double[][] f = new double[n + 1][n + 1];
        f[0][0] = poured;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                if (f[i][j] > 1) {
                    double more = f[i][j] - 1;
                    
f[i][j] = 1;
                    f[i + 1][j] += 0.5 * more;
                    f[i + 1][j + 1] += 0.5 * more;
                }
            }
        }
        return f[query_row][query_glass];
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(sol.champagneTower(1, 1, 1));
        System.out.println(sol.champagneTower(2, 1, 1));
        System.out.println(sol.champagneTower(10000000, 99, 99));
        System.out.println(sol.champagneTower(10000000, 99, 45));
        System.out.println(sol.champagneTower(10000000, 99, 65));
        System.out.println(sol.champagneTower(10000000, 99, 85));
        System.out.println(sol.champagneTower(10000000, 99, 75));
        System.out.println(sol.champagneTower(10000000, 99, 70));
        for (int i = 0; i < 100; i++) {
            System.out.println(sol.champagneTower(10000000, 99, i));
        }
    }
}
