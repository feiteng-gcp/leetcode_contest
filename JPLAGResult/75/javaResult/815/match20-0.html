class Solution {
    public double champagneTower(int poured, int query_row, int query_glass) {
        double[][] remaining = new double[101][101];
        remaining[0][0] = poured;
        for (int row = 0; row < 100; row++) {
            for (int col = 0; col <= row; col++) {
                if (remaining[row][col] >= 1) {
                    remaining[row+1][col] += (remaining[row][col] - 1) / 2.0;
                    remaining[row+1][col+1] += (remaining[row][col] - 1) / 2.0;
                    remaining[row][col] = 1;
                }
            }
        }
        return remaining[query_row][query_glass];
    }
}
