class Solution {
    public double champagneTower(int poured, int query_row, int query_glass) {
        double liquid[][] = new double[100][100];
        liquid[0][0] = poured;
        
        for (int i = 0; i < 99; i++) {
            for (int j = 0; j <= i; j++) {
                double remaining = Math.max(liquid[i][j] - 1, 0);
                liquid[i][j] = Math.min(liquid[i][j], 1);
                
                liquid[i + 1][j] += (remaining/2.0);
                liquid[i + 1][j + 1] += (remaining/2.0);
            }
        }
        
        return liquid[query_row][query_glass];
    }
}
