class Solution {
    public double champagneTower(int poured, int query_row, int query_glass) {
        double[] volume = new double[1];
        volume[0] = poured;
        for (int row = 1; row <= query_row; row++){
            double[] next = new double[row+1];
            for (int i = 0; i <= row; i++){
                if (i >= 1){
                    next[i] = Math.max(volume[i-1]-1,0)/2;
                }
                if (i < row){
                    next[i] += Math.max(volume[i]-1,0)/2;
                
}
            }
            volume = next;
        }
        
        return Math.min(1,volume[query_glass]);
    }
}
