class Solution {
    public int bestRotation(int[] A) {
        if (A.length <= 1) {
            return 0;
        }
        int[] scores = new int[A.length];
        
        for (int i=0; i<A.length; i++) {
            if (A[i] <= i) {
                for (int j=0; j<= i-A[i] && j<=A.length-1; j++) {
                    scores[j]++;
                }
                for (int j=i+1; j<= A.length-1; j++) {
                    scores[j]++;
                }
            } else {
                for (int j=i+1; j<= A.length+i-A[i]; j++) {
                    scores[j]++;
                }
            }
        }
        
        int max = scores[0];
        int k = 0;
        for (int i=1; i<A.length; i++) {
            
if (scores[i] > max) {
                max = scores[i];
                k = i;
            }
        }
        
        return k;
    }
}
