import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    final int offset = 10;
    final int max = 50000;
    int[] bit;


    void add(int index, int delta) {
        index += offset;
        for (; index < bit.length; index += Integer.lowestOneBit(index))
            bit[index] += delta;
    }

    int sum(int index) {
        index += offset;
        int ans = 0;
        while (index > 0) {
            ans += bit[index];
            index -= Integer.lowestOneBit(index);
        }
        return ans;
    }

    int sum(int l, int r) {
        return sum(r) - sum(l - 1);
    }


    public int bestRotation(int[] A) {
        int n = A.length;
        bit = new int[max];
        int K = 0, max = 0;
        int line = 0;

        for (int i = 0; i < A.length; i++) {
            int diff = i - A[i];
            if (diff >= 0) {
                add(diff, 1);
            }
            max = sum(line, line + n - 1);
        }
//        System.out.println(max);

        for (int i = 1; i < A.length; i++) {
            int val = A[i - 1];

            line++;
            int diff = n - 1 - val;
            if (diff >= 0) {
                add(line + diff, 1);
            }

            int tmp = sum(line, line + n - 1);
//            System.out.println(tmp);
            if (tmp > max) {
                max = tmp;
                K = i;
            }
        }
        return K;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(sol.bestRotation(new int[]{2, 3, 1, 4, 0}));
        System.out.println(sol.bestRotation(new int[]{1, 3, 0, 2, 4}));
    }
}
