class Solution {
    public int bestRotation(int[] A) {
        int len = A.length;
        int[] score = new int[len];
        
        for(int i=0; i<len; i++){
            for(int p=0; A[i] <= i-p && p <= i; p++)
                score[p]++;
            for(int p=i+1; p < len && A[i] <= len + i - p; p++)
                score[p]++;
            // for(int s: score)
            //     System.out.print(s+"\t");
            // System.out.println();
        }
        
        int max = 0, pos = 0;
        for(int i=0; i<len; i++){
            if(max < score[i]){
                max = score[i];
                pos = i;
            }
            // System.out.println(i + "\t" + score[i]);
        }
        
        return pos;
    }
}
