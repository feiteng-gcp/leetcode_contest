class Solution {
  public int bestRotation(int[] a) {
        int[] res = new int[a.length];
        for (int f = 0; f < a.length; f++) {
            int pp = a[f];
            int start = pp;
            int end = a.length;
            if (pp >= end) continue;

            start -= f;
            end -= f;
            if (start >= 0) {
                res[start]++;
                if (end < a.length)
                    res[end]--;
            } else {
                res[0]++;
                res[start + a.length]++;
                if (end < a.length)
                    res[end]--;
            }


        }
        for (int k = 1; k < a.length; k++) {
            res[k] += res[k - 1];
        }
//        System.out.println(Arrays.toString(res));
        int min = 0;
        for (int t = a.length - 1; t >= 1; t--) {
            if (res[t] > res[min]) min = t;
        }
        return (a.length - min) % a.length;
    }
}
