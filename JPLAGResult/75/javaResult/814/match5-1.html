class Solution {
    public int bestRotation(int[] A) {
        int[] scoreList=new int[A.length];
        for (int i=0;i<A.length;i++)
        {
            int lowS=rotate(i,A[i],A.length);
            int highS=rotate(i,A.length-1,A.length);
            //System.out.println(lowS+" "+highS);
            scoreList[highS]++;
            while (highS!=lowS)
            {
                highS++;
                if (highS==A.length)
                    highS=0;
                scoreList[highS]++;
            }
        }
        int maxv=scoreList[0];
        int maxi=0;
        for (int i=0;i<scoreList.length;i++)
        {
            //System.out.println(scoreList[i]);
            if (maxv<scoreList[i])
            {
                maxv=scoreList[i];
                maxi=i;
            }
        }
        return maxi;
    }
    public int rotate(int org, int tar, int len)
    {
        int t=org-tar;
        if (t<0)
            return t+len;
        else
            return t;
    }
}
