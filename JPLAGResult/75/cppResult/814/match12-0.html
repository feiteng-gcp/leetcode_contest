typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 100005

int s[MAXN];
void add(int k, int v){
  k++;
  for(;k<MAXN;k+=-k&k) s[k]+=v;
}
int get(int k){
  if (k < 0) return 0;
  k++;
  int ans = 0;
  for(;k;k-=-k&k) ans += s[k];
  return ans;
}
class Solution {
public:
  int bestRotation(vector<int>& A) {
    FILL(s, 0);
    int n = A.size();
    REP(i,0,n) {
      int x = A[i];
      int l = 0, r = i - x;
      if(l<=r) {
        add(l, 1);
        add(r+1,-1);
      }
      l = i+1; r = i+n-x;
      if(l<=r) {
        add(l, 1);
        add(r+1, -1);
      }
    }
    int best = -1, ans;
    REP(i,0,n) {
      int score = get(i);
      if (score > best) {
        best = score;
        ans = i;
      }
    }
    return ans;
  }
};
