class Solution {
public:
    int bestRotation(vector<int>& A) {
        vector< int > sc = vector< int >(A.size() * 3, 0);
        for (int i = 0; i < A.size(); i++) {
            sc[i - (A.size() - 1) + A.size()]++;
            sc[i - A[i] + A.size() + 1]--;
           // printf("%d %d\n", i - (A.size() - 1) + A.size(), i - A[i] + A.size() + 1);
        }
        for (int i = 1; i < sc.size(); i++)
            sc[i] += sc[i - 1];
        //for (int i = 0; i < sc.size(); i++) printf("%d%c", sc[i], i == sc.size() - 1 ? '\n' :' ');
        for (int i = A.size(); i < sc.size(); i++)
            sc[i % A.size()] += sc[i];
        int ans = -1, re;
        //for (int i = 0; i < A.size(); i++) printf("%d%c", sc[i], i == A.size() - 1 ? '\n' :' ');
        for (int i = 0; i < A.size(); i++)if (sc[i] > ans)ans = sc[i], re = i;
        return re;
    }
};
