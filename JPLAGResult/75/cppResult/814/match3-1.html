class Solution {
public:
  int bestRotation(vector<int>& A) {
    int n = A.size();
    vector<int> scores(n + 1, 0);
    for (int i = 0; i != n; ++i) {
      int a = A[i];
      if (a >= n) continue;
      int start1 = i + 1;
      int end1 = start1 + (n - a);
      int start2 = -1;
      int end2 = -1;
      if (end1 > n) {
        start2 = 0;
        end2 = end1 - n;
        end1 = n;
      }
      ++scores[start1];
      if (start2 > 0)
        ++scores[start2];
      --scores[end1];
      if (end2 > 0)
        --scores[end2];
    }
    int acc = 0, max = 0, ret = 0;
    for (int i = 0; i != n; ++i) {
      acc += scores[i];
      if (acc > max) {
        max = acc;
        ret = i;
      }
    }
    return ret;
  }
};
