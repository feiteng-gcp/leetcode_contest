class Solution {
public:
		int bestRotation(vector<int>& a) {
		int n = a.size(), i, j, k;
		vector<int> aa = a;
		for (i = 0; i < n; i++) aa.push_back(a[i]);
		for (i = 0; i < n + n; i++) aa[i] -= i; 

		map<int, int> ct;
		vector<int> cc;
		int c = 0;

		for (i = 0; i < n; i++) {
			if (aa[i] <= 0) {
				ct[aa[i]]++;
				c++;
			}
		}
		cc.push_back(c);

		int best = c, bestp = 0; 

		for (i = 1; i < n; i++) {
			if (aa[i - 1] <= -(i - 1)) {
				ct[aa[i - 1]]--;
				c--;
			}
			c -= ct[-(i - 1)];


			if (aa[i + n - 1] <= -i) {
				ct[aa[i + n - 1]]++;
				c++;
			}
			if (c > best) {
				best = c;
				bestp = i;
			}
			cc.push_back(c);
		}

		return bestp;
	}
};
