const int N = 1e5 + 10;

int c[N];

void add(int x, int y) {
    for (; x < N; x += x & -x) {
        c[x] += y;
    }
}

int query(int x) {
    int ret = 0;
    for (; x; x -= x & -x) {
        ret += c[x];
    }
    return ret;
}

class Solution {
public:
    int bestRotation(vector<int>& a) {
        int n = a.size();
        int best = 0, best_k = 0;
        for (int i = 0; i < n; ++i) {
            a[i] = i - a[i];
        }
        memset(c, 0, sizeof(c));
        int offset = 2 * n;
        for (int i = 0; i < n; ++i) {
            add(offset + a[i], 1);
        }
        best = query(N - 1) - query(offset - 1);
        for (int k = 1; k < n; ++k) {
            add(offset + a[k - 1], -1);
            add(offset + a[k - 1] + n, 1);
            int cur = query(N - 1) - query(offset - 1 + k);
            if (cur > best) {
                best = cur;
                best_k = k;
            }
        }
        return best_k;
    }
};
