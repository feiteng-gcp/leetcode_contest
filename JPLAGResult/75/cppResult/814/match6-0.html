class Solution {
public:
    int cnt[20005];
    int bestRotation(vector<int>& A) {
        int n = A.size();
        for(int i = 0;i < n;i++){
            int m = A[i];
            if(m <= i){
                cnt[0]++;
                cnt[i - m + 1]--;
                cnt[i + 1]++;
                cnt[n]--;
            }else{
                cnt[i + 1]++;
                cnt[n - m + i + 1]--;
            }
        }
        for(int i = 1;i <= n;i++){
            cnt[i] = cnt[i - 1] + cnt[i];
        }
        int mx = -1;
        int mxid = -1;
        for(int i = 0;i < n;i++){
            if(cnt[i] > mx){
                mx = cnt[i];
                mxid = i;
            }
        }
        return mxid;
    }
};
