class Solution {
public:
    struct fff
{
	int p;
	bool operator < (const fff &a) const {return p>a.p;}
};
    int bestRotation(vector<int>& A) {
        int n=A.size();

	int ans=0,sco=0;
	priority_queue<fff> q;
	for(int i=0;i<n;i++)
	{
		if(A[i]<=i)
		{
			sco++;
			fff f;
			f.p=i-A[i];
			q.push(f);
		}
	}
	//printf("sco:%d\n",sco);
	int nowsco=sco;
	for(int i=1;i<n;i++)
	{
		fff f;
		f.p=n-1-A[i-1]+i;
		nowsco++;
		q.push(f);
		while(!q.empty()&&q.top().p<i)
		{
			nowsco--;
			q.pop();
		}
		//printf("i:%d nowsco:%d\n",i,nowsco);
		if(nowsco>sco)
		{
			sco=nowsco;
			ans=i;
		}
	}
	return ans;
    }
};
