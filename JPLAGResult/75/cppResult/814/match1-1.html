class Solution {
public:
    int bestRotation(vector<int>& A) {
        vector<pair<int, int>> rotate;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] > i) {
                if (A[i] < A.size()) {
                    rotate.emplace_back(make_pair(i + 1, 1));
                    rotate.emplace_back(make_pair(i + 1 + A.size() - A[i], -1));
                }
            } else {
                rotate.emplace_back(make_pair(0, 1));
                rotate.emplace_back(make_pair(i - A[i] + 1, -1));
                rotate.emplace_back(make_pair(i + 1, 1));
                rotate.emplace_back(make_pair(A.size(), -1));
            }
        }
        sort(rotate.begin(), rotate.end(), [](const pair<int, int>& a, const pair<int, int>& b){
            return a.first < b.first;
        });
        int ans =-1, ans_value = -1;
        for (int i = 0, value = 0, cur = 0; i < A.size(); i++) {
            while (cur < rotate.size() && rotate[cur].first <= i) {
                value += rotate[cur++].second;
            }
            if (value > ans_value) {
                ans_value = value;
                ans = i;
            }
        }
        return ans;
    }
};
