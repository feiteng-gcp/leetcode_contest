class Solution {
public:
    int bestRotation(vector<int>& A) {
        vector<int> counter((2 * A.size()) + 1, 0);
        
        for (int i = 0; i < A.size(); ++i){
            if (A[i] <= i)
            {
                counter[0]++;
                counter[i-A[i]+1]--;
                
                 // cout << A[i] << " --> " << 0 << ", " << (i-A[i]) << endl;
            }
            
            counter[i+1]++;
            counter[i+1+(A.size()-A[i]-1)+1]--;
            
              // cout << A[i] << " --> " << (i+1) << ", " << (i+1+(A.size()-A[i]-1)) << endl;
        }
        
        int ans = 0;
        int ansCounter = INT_MIN;
        int temp = 0;
        for (int i = 0; i < A.size(); ++i){
            temp += counter[i];
            
            if (temp > ansCounter){
                ansCounter = temp;
                ans = i;
            }
            
             // cout << "temp = " << temp << endl;
        }
        
        return ans;
    }
};
