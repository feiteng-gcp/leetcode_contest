class Solution {
public:
    int bestRotation(vector<int>& A) {
        map<int, int> index;
        for(int i = 0; i < A.size(); ++i) {
            int start = 0;
            int end = i - A[i];
            
            if(end >= start) {
                ++index[start];
                --index[end + 1];
            }
            
            start = i + 1;
            end = i - A[i] + A.size();
            if(end >= start) {
                ++index[start];
                --index[end + 1];
            }
        }
        
        int ans = 0, ans_k = -1, acc = 0;
        for(auto& it: index) {
            acc += it.second;
            if(acc > ans) {
                ans = acc;
                ans_k = it.first;
            }
        }
        return ans_k;
    }
};
