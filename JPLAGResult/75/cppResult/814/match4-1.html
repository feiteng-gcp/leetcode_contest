class Solution {
public:
    vector<int> st;
    void add(int x,int del){
        while (x<st.size()){
            st[x]+=del;
            x+=x&(-x);
        }
    }
    int sum(int x){
        int ans=0;
        while (x>0){
            ans+=st[x];
            x-=x&(-x);
        }
        return ans;
    }
    int bestRotation(vector<int>& A) {
        st=vector<int>(A.size()*3+1,0);
        for (int i=0;i<A.size();i++){
            add(A[i]-i+A.size()*2,1);
        }
        int now=sum(A.size()*2);

        int ans=0;
        for (int i=1;i<A.size();i++){
            add(A.size()*2+A[i-1]-(i-1),-1);
            add(A.size()*2+A[i-1]-(i-1)-(A.size()),1);
            int t=sum(A.size()*2-i);
            if (t>now){
                ans=i;
                now=t;
            }
        }
        return ans;
    }
};
