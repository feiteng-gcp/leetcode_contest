class Solution {
    int getMoves(int from, int to, int n) {
        return (from - to + n) % n;
    }
    
public:
    int bestRotation(vector<int>& A) {
        vector< pair<int, int> > goodRanges;
        int n = A.size();
        
        for (int i = 0; i < n; i++) {
            int v1 = getMoves(i, n - 1, n);
            int v2 = getMoves(i, A[i], n);
            
            if (v1 <= v2) {
                goodRanges.push_back({v1, 1});
                goodRanges.push_back({v2 + 1, -1});
            } else {
                goodRanges.push_back({0, 1});
                goodRanges.push_back({v2 + 1, -1});
                
                goodRanges.push_back({v1, 1});
                goodRanges.push_back({n, -1});

            }
        }
        
        sort(goodRanges.begin(), goodRanges.end());
        
        int ans = 0, currSum = 0, ans_val = 0;
        for (auto it: goodRanges) {
            //cout << it.first << " " << it.second << "\n";
            currSum += it.second;
            if (currSum > ans) {
                ans = currSum;
                ans_val = it.first;
            }
        }
        //cout << ans << "\n";
        return ans_val;
    }
};
