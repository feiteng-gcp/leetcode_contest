class Solution {
public:
    bool rotateString(string A, string B) {
        if(A==B)
            return true;
        for(int i=1; i<A.size();i++){
            string C = A.substr(i) + A.substr(0,i);
            if(C==B)
                return true;
        }
        return false;
    }
};
