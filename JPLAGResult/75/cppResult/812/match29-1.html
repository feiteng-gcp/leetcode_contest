class Solution {
public:
    bool rotateString(string A, string B) {
        int m = A.length();
        int n = B.length();
        if(m != n) return false;
        if(0 == m) return true;
        int i = 0;
        while(i < m && B[0] != A[i]){
            i++;
        }
        if(i>=m) return false;
        
        while(i<m){
            if(B[0] == A[i]){
                if(i == 0) {
                    if(A == B)
                        return true;
                }else if((A.substr(i) == B.substr(0,m-i))&&
                    (A.substr(0,i) == B.substr(m-i)))
                    return true;
            }
            i++;
        }
        return false;
    }
};
