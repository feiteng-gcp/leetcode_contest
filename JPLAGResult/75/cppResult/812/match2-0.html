class Solution {
public:
    bool rotateString(string A, string B) {
        if (A == B) return true;
        for (int i = 0; i < A.size() - 1; i++) {
            string a = A.substr(i + 1) + A.substr(0, i + 1);
            if (a == B) return true;
        }
        return false;
    }
};
