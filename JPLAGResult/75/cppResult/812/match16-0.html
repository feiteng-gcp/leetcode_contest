class Solution {
public:
    bool rotateString(string A, string B) {
        if (A.size()!=B.size()) return false;
        for (int i = 0;i<A.size();i++){
            string A1 = A.substr(0,i+1);
            string A2 = A.substr(i+1,A.size()-i-1);
            if (A2+A1 == B) return true;
        }
        return false;
    }
};
