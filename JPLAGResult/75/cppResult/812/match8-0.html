class Solution {
public:
    bool rotateString(string A, string B) {
        if(A.size() != B.size()) {
            return false;
        }
        for(int i = 0; i < A.size(); ++i) {
            bool succ = true;
            for(int j = 0; j < A.size(); ++j) {
                if(A[(i + j) % A.size()] != B[j]) {
                    succ = false;
                    break;
                }
            }
            if(succ) {
                return true;
            }
        }
        return false;
    }
};
