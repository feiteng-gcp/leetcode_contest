class Solution {
public:
    bool rotateString(string A, string B) {
        for(int i = 1;i <= 1000;i++){
            string s = A.substr(1,A.length() - 1);
            s = s + A[0];
            if(s == B) return true;
            A = s;
        }
        return false;
    }
};
