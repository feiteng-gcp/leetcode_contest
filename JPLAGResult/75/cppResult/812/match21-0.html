class Solution {
public:
    bool rotateString(string A, string B) {
        if (A.size() != B.size()) return false;
        for (int i = 0; i < A.size(); ++i) {
            bool ok = true;
            for (int j = 0, k = i; j < A.size(); ++j, k = (k + 1) % A.size()) {
                if (B[j] != A[k]) {
                    ok = false;
                    break;
                }
            }
            if (ok) return true;
        }
        return false;
    }
};
