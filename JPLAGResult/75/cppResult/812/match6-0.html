class Solution {
public:
    bool rotateString(string A, string B) {
        if (A.size() != B.size()) return false;
        for (int i = 1; i < A.size(); ++i) {
            string str = A.substr(i) + A.substr(0, i);
            if (str == B) return true;
        }
        return false;
    }
};
