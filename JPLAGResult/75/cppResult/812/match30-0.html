class Solution {
public:
    bool rotateString(string A, string B) 
    {
        int la=A.size();
        int lb=B.size();
        if(la!=lb) return false;
        for(int i=0;i<la;i++)
        {
            if(check(A,B,i)) return true;
        }
        return false;
    }
    
    bool check(string A, string B,int start)
    {
        int la=A.size();
        int p1=0;
        int p2=start;
        while(p1<la)
        {
            if(A[p1]!=B[p2%la]) return false;
            p1++;
            p2++;
        }
        return true;
    }
};
