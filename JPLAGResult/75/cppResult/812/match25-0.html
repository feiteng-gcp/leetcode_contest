class Solution {
public:
    string rotate( string A , int n ){
        string s= A.substr( 0 , n ) ;
        string t =A.substr( n,A.size()-n);
    
        return t+s;
    }
    bool rotateString(string A, string B) {
        for( int i=0 ; i<A.size() ; i++){
            string tmp = rotate( A , i ) ;
            //cout << tmp <<endl;
            if( tmp == B)
                return true ;
        }
        return false ;
    }
};
