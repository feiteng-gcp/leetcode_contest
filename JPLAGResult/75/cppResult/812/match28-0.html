class Solution {
public:
    bool rotateString(string A, string B) {
        if(A.empty()) return B.empty();
        int n = A.size();
        if(B.size()!=n) return false;
        for(int offset = 0;offset<n;offset++){
            int i = 0;
            for(i=0;i<n;i++){ // in B
                if(A[(i+offset)%n]!=B[i]) break;
            }
            if(i==n) return true;
        }
        return false;
    }
};
