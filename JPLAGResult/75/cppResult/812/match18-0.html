class Solution {
public:
    bool rotateString(string A, string B) {
        if(A.length() != B.length())
            return false;
        string c;
        for(int i = 0;i<A.length();i++){
            c = A.substr(i) + A.substr(0,i);
            if (c == B){
                //cout<< i << "  "<< c<<endl;
                return true;
            }
        }
        return false;
    }
};
