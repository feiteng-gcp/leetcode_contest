class Solution {
public:
    bool rotateString(string A, string B) {
        int la = A.size();
        int lb = B.size();
        if (la != lb) return false;
        if (la == 0) return true;
        
        string C = A + A;
        for (int i = 0; i < la; ++i) {
            if (C.substr(i, la) == B)
                return true;
        }
        
        return false;
    }
};
