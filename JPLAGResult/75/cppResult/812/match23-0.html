class Solution {
public:
    bool rotateString(string A, string B) {
        // A = "gcmbf";
        // B = "fgcmb";
        if(A == "" && B == "") return true;
        int l1 = A.length(), l2 = B.length();
        if(l1 != l2) return false;
        for(int i = 1; i < l1+1; ++i) {
            if(A.substr(0, i) == B.substr(l1-i) && A.substr(i) == B.substr(0, l1-i)) return true;
        }
        return false;
    }
};
