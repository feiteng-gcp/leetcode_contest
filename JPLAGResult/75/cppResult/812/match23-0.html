class Solution {
public:
    bool rotateString(string A, string B) {
        if (A.length() != B.length())
            return false;
        if (A.length() == 0)
            return true;
        for (int i = 0; i < A.length(); i++) {
            bool can = true;
            for (int j = 0; j < A.length() && can; j++)
                if (A[j] != B[(j + A.length() - i) % A.length()])
                    can = false;
            if (can)
                return true;
        }
        return false;
    }
};
