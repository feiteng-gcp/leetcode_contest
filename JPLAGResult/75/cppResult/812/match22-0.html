class Solution {
public:
    bool rotateString(string A, string B) {
        if(A.length() != B.length()) return false;
        A += A;
        for(int i = 0; i < A.length() / 2; i++){
            bool flag = true;
            for(int j = i, k = 0; k < B.length(); j++, k++){
                if(A[j] != B[k]){
                    flag = false;
                    break;
                }
            }
            if(flag){
                    return true;
                }
        }
        return false;
    }
};
