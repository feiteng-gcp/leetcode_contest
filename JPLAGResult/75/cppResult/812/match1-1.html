class Solution {
public:
    bool rotateString(string A, string B) {
        int n = A.length();

        if (n == 0) return (B.length() == 0);
        for (int i = 0; i < n; i ++) {
            A.push_back(A[0]);
            A = A.substr(1);
            if (A==B) return true;            
        }
        return false;
    }
};
