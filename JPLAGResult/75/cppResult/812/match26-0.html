class Solution {
public:
    bool rotateString(string A, string B) {
        int nA=A.length(), nB=B.length();
        if (nA!=nB) return false;
        if (A==B) return true;
        for (int shift=0; shift<nA; shift++) {
            A = A.substr(1,nA-1)+A[0];
            if (A==B) return true;
        }
        return false;
    }
};
