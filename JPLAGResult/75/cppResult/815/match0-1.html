class Solution {
public:
    double champagneTower(int poured, int query_row, int query_glass) {
        if (poured == 0) { return 0; }

        vector<vector<double>> cup(100, vector<double>(100, 0));
        cup[0][0] = poured;
        for (int i = 0; i < query_row; i++) {
            for (int j = 0; j <= i; j++) {
                if (cup[i][j] > 1.0) {
                    double temp = (cup[i][j] - 1.0) / 2.0;
                    cup[i + 1][j] += temp;
                    cup[i + 1][j + 1] += temp;
                }
            }
        }
        return min(cup[query_row][query_glass], 1.0);
    }
};
