class Solution {
public:
    double champagneTower(int poured, int target_row, int target_col) {

        vector<vector<double>> data(101, vector<double>(101, 0));
        data[0][0] = poured;
        for (int i = 0; i <= target_row; i++) {
            for (int j = 0; j <= i; j++) {
                data[i + 1][j] += data[i][j] <= 1 ? 0 : (data[i][j] - 1) / 2;
                data[i + 1][j + 1] += data[i][j] <= 1 ? 0 : (data[i][j] - 1) / 2;
            }
        }
        return data[target_row][target_col] > 1 ? 1 : data[target_row][target_col];
    }
};
