class Solution {
public:
    double champagneTower(int poured, int query_row, int query_glass) {
        // the cup number of the middle glass 
        // = left top cup glass size > 1 ? 0.5 * (left top cup glass size - 1) : 0
        // + right top cup glass size > 1 ? 0.5 * (right top cup glass size - 1) : 0
        
        // the cup number of the side glass = 
        // top cup glass number > 1 : 0.5 * (top cup glass number - 1) : 0
        
        if(poured == 0) return 0;
        
        if(query_row == 0) return poured >= 1 ? 1 : 0;
        
        vector<vector<double>> numberOfCup(query_row + 1,vector<double>(query_row + 1,0));
        
        numberOfCup[0][0] = poured;
        
        for(int i = 1 ; i <= query_row ; ++i) {
            
            for(int j = 0 ; j <= i ; ++j) {
                
                numberOfCup[i][j] = (j > 0 ? ( numberOfCup[i-1][j-1] > 1 ? 0.5 *(numberOfCup[i-1][j-1] - 1) :0 ) : 0)  +
                                    (j < i ? ( numberOfCup[i-1][j] > 1 ? 0.5 * (numberOfCup[i-1][j] - 1) : 0 ) : 0);
                
                if(i == query_row && j == query_glass) 
                    return numberOfCup[i][j] >= 1 ? 1 : numberOfCup[i][j];
                       
            }
            
        }
        
    }
};
