class Solution {
public:
    double champagneTower(int poured, int query_row, int query_glass) {
        vector<double> row[105];
        for (int i = 0; i < 105; ++i) row[i].resize(105, 0.);
        
        query_row++;
        query_glass++;
        
        row[1][1] = poured;
        for (int i = 2; i <= query_row; ++i)
            for (int j = 1; j <= i; ++j)
                row[i][j] = max(0., max(0., (row[i-1][j-1]-1)) * 0.5 + max(0., (row[i-1][j]-1)) * 0.5);
        
        // for (int i = 1; i <= query_row; ++i){
        //     for (int j = 1; j <= i; ++j){
        //         cout << row[i][j] << " ";       
        //     }
        //     cout << endl;
        // }
        return min(1., row[query_row][query_glass]);
    }
};
