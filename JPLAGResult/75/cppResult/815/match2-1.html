class Solution {
public:
  double champagneTower(int p, int r, int g) {
    vector<double> wines;
    wines.push_back(p);

    for (int i = 1; i <= r; ++i) {
      vector<double> next_wines(wines.size() + 1, 0.0);
      for (int j = 0; j != wines.size(); ++j) {
        double w = wines[j] - 1.0;
        if (w <= 0.0) continue;
        next_wines[j] += w / 2.0;
        next_wines[j + 1] += w / 2.0;
      }
      wines = std::move(next_wines);
    }
    return (wines[g] > 1.0)? 1.0 : wines[g];
  }
};
