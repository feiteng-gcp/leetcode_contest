class Solution {
public:
    double champagneTower(int poured, int query_row, int query_glass) {
        
        vector<double> vec0(query_row + 1, 0);                       
        
        vec0[0] = poured;
        for (int i = 0; i <= query_row; ++i)
        {
            vector<double> vec1(query_row + 2, 0);
            for (int t = 0; t <= i; ++t)
            {                
                double amount = vec0[t];                 
                if (amount > 1.0) 
                {
                    vec0[t]      = 1.0;                
                    vec1[t]     += (amount - vec0[t]) / 2.0;
                    vec1[t + 1] += (amount - vec0[t]) / 2.0;                                                            
                }                
                
            }                        
            
            if (i == query_row) break;
            
            vec0 = vec1;
        }
        
        return vec0[ query_glass ];
    }
};
