class Solution {
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        vector<vector<int>> res;
        vector<int> cur{0};
        vector<bool> visited(graph.size(), false);
        visited[0] = true;
        dfs(graph, cur, visited, res, graph.size() - 1);

        return res;
    }
    
    void dfs(vector<vector<int>>& graph, vector<int>& cur, vector<bool>& visited, vector<vector<int>>& res, int target){
        if(cur.back() == target){
            res.push_back(cur);
            return;
        }
        
        for(int next : graph[cur.back()]){
            
            if(!visited[next]){
                visited[next] = true;
                cur.push_back(next);
                dfs(graph, cur, visited, res, target);
                cur.pop_back();
                visited[next] = false;
            }
        }   
    }
    
};
