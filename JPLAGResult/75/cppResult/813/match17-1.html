class Solution {
public:
    vector<vector<int>> ans;
    int end;
    vector<vector<int>> a;

    void dfs(vector<int> path,int now){
        if (now==end){
            ans.push_back(path);
            return ;
        }
        for (int i=0;i<a[now].size();i++){
            int v=a[now][i];
            path.push_back(v);
            dfs(path,v);
            path.pop_back();
        }
    }
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        end=graph.size()-1;
        a=graph;
        vector<int> path;
        path.push_back(0);
        dfs(path,0);
        return ans;
    }
};
