class Solution {
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        
        if(!graph.size()) return {};
        
        vector<int> current(1,0);
        vector<vector<int>> result;
        vector<bool> used(graph.size(), false);
        used[0] = true;
        dfs(current, result, used, graph);
        return result;
    }
    

    void dfs(vector<int>& current, vector<vector<int>>& result, vector<bool>& used, vector<vector<int>>& graph) {
        
        // if the last element is the last element ...
        if(*(current.end()-1) == graph.size() - 1) {
            result.push_back(current);
            return;
        }
        
        // if not, go to next adjacent node
        for(auto each_adj:graph[*(current.end()-1)]) {
            
            if(used[each_adj]) continue;
            
            used[each_adj] = true;
            current.push_back(each_adj);
            dfs(current,result,used,graph);
            current.pop_back();
            used[each_adj] = false;
            
        }
        
    }
};
