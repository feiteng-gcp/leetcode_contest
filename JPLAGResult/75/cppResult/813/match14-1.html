class Solution {

    void generate(int node, vector< vector<int> > &edges, vector< vector<int> > &ans, vector<int> v) {
        int n = edges.size();
        v.push_back(node);
        if (node == n - 1) {
            ans.push_back(v);
            return;
        }
        
        for (int x: edges[node]) {
            generate(x, edges, ans, v);
        }
    }
    
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        vector< vector<int> > ans;
        vector<int> curr;
        generate(0, graph, ans, curr);
        return ans;
    }
};
