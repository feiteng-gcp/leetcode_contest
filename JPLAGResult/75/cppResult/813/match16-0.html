class Solution {
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        vector<vector<int>> res;
        if (graph.empty()) return res;
        vector<int> temp;
        temp.push_back(0);
        helper(res,graph,temp,0);
        return res;
    }
    void helper(vector<vector<int>>&res,vector<vector<int>>& graph,vector<int> &temp, int Node){
        if (Node == (graph.size()-1)){
            res.push_back(temp);
            return;
        }
        for (int i = 0;i<graph[Node].size();i++){
            temp.push_back(graph[Node][i]);
            helper(res,graph,temp,graph[Node][i]);
            temp.pop_back();
        }
        return;
    }
};
