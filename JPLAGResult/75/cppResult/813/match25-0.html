class Solution {
public:
    vector<vector<int>> g;
    vector<int> path;
    void search(int u, int target, vector<vector<int>>& ret) {
        if (u == target) {
            ret.push_back(path);
            return;
        }
        for (auto& v : g[u]) {
            path.push_back(v);
            search(v, target, ret);
            path.pop_back();
        }
    }
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& g) {
        int n = g.size();
        this->g = g;
        path = {0};
        vector<vector<int>> ret;
        search(0, n - 1, ret);
        return ret;
    }
};
