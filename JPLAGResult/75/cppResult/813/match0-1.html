class Solution {
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        int n = graph.size();
        vector<int> cur_result = {0};

        vector<vector<int>> result;
        get_result(graph, 0, n, cur_result, result);
        return result;
    }
    
    void get_result(vector<vector<int>>& graph, int cur, int n, vector<int> &cur_result, vector<vector<int>> &result) {
        if (cur == n - 1) {
            result.push_back(cur_result);
        } else {
            for (auto next : graph[cur]) {
                cur_result.push_back(next);
                get_result(graph, next, n, cur_result, result);
                cur_result.pop_back();
            }
        }
    }
};
