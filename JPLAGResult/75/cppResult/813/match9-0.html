class Solution {
public:
    vector<vector<int>> path;
    int nodemax;
    void dfs(vector<int> &p, int cur, vector<vector<int>>& g)
    {
        if (cur == nodemax) 
        {
            p.push_back(cur);
            path.push_back(p);
            p.pop_back();
            return;
        }
        for (int i = 0; i < g[cur].size(); ++i)
        {
            p.push_back(cur);
            dfs(p, g[cur][i], g);
            p.pop_back();
        }
    }
    
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& g) 
    {
        int len = g.size();
        nodemax = len - 1;
        vector<int> p;
        for (int i = 0; i < g[0].size(); ++i)
        {
            p.push_back(0);
            dfs(p, g[0][i], g);
            p.pop_back();
        }
        return path;
    }
};
