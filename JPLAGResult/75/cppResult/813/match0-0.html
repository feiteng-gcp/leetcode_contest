class Solution {
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        int n = graph.size();
        vector<vector<int>> res;
        vector<int> r;
        dfs(res, r, graph, 0, n-1);
        return res;
    }

    void dfs(vector<vector<int>>&res, vector<int>& r, vector<vector<int>>& graph, int u, int N) {
        if (u == N) {
            r.push_back(u);
            res.push_back(r);
            r.pop_back();
            return;
        }

        r.push_back(u);

        for (auto g: graph[u])
            dfs(res, r, graph, g, N);

        r.pop_back();
    }
};
