class Solution {
public:
    void print(vector<int> table)
    {
        for(auto x:table)
            cout<<x<<" ";
        cout<<endl;
    }
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) 
    {
        vector<vector<int> > ans;
        vector<int> res;
        res.push_back(0);
        recurse(res,0,graph,ans);
        return ans;
    }
    void recurse(vector<int>& res,int src,vector<vector<int> >& graph, vector<vector<int> >& ans)
    {
        int n = graph.size();
        if(src==n-1)
        {
            // res.push_back(src);
            ans.push_back(res);
            // res.pop_back();
        }
        for(auto v:graph[src])
        {
            res.push_back(v);
            recurse(res,v,graph,ans);
            res.pop_back();
        }
    }
};
