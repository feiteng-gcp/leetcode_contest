class Solution {
private:
    vector<vector<int> > ans;
    int last_node;

    void dfs(int pos, vector<int>& path, vector<vector<int>>& graph) {
        if(pos == last_node) {
            ans.push_back(path);
            return;
        }
        for(auto next: graph[pos]) {
            path.push_back(next);
            dfs(next, path, graph);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        last_node = graph.size() - 1;
        vector<int> tmp({0});
        dfs(0, tmp, graph);
        return ans;
    }
};
