class Solution {
public:
    bool G[20][20];
    int n;
    bool vis[20];
    bool dfs(int u){
        if(u == n - 1){
            vis[u] = true;
            return 1;
        }
        if(vis[u]) return 1;
        for(int j = 0;j < n;j++){
            if(G[u][j]){
                if(dfs(j)){
                    vis[u] = true;
                    return 1;
                }
            }
        }
        return 0;
    }
    vector<vector<int> > ans;
    void DFS(int u,vector<int> now){
        now.push_back(u);
        if(u == n - 1){
            ans.push_back(now);
            return;
        }
        for(int j = 0;j < n;j++){
            if(vis[j] && G[u][j]){
                DFS(j,now);
            }
        }
    }
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        n = graph.size();
        memset(G,0,sizeof(G));
        memset(vis,0,sizeof(vis));
        for(int i = 0;i < graph.size();i++){
            for(int j = 0;j < graph[i].size();j++){
                G[i][graph[i][j]] = 1;
            }
        }
        ans.clear();
        for(int i = 0;i < n;i++)
            dfs(i);
        for(int i = 0;i < n;i++){
            printf("%d\n",vis[i]);
        }
        DFS(0,vector<int>(0));
        return ans;
    }
};
