class Solution {
public:

    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        vector<vector<int>> ans;
        vector<int> cur;
        dfs(0, graph, cur, ans);
        return ans;
    }
    
    void dfs(int cur, vector<vector<int>>& graph, vector<int>& path, vector<vector<int>>& ans) {
        int n = graph.size();
        path.push_back(cur);
        if (cur == n - 1) {
            ans.push_back(path);
            path.pop_back();
            return;
        }
        for (const auto& iter : graph[cur]) {
            dfs(iter, graph, path, ans);
        }
        path.pop_back();
    }
};
