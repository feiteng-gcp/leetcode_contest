class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> ans;

        int c = 0;
        int i=A.size() - 1;
        while (K && i>=0) {
            int s = K%10 + A[i] + c;
            ans.push_back(s%10);
            c=s/10;
            K/=10;
            i--;
        }
        
        while (K) {
            int s = K%10 + c;
            ans.push_back(s%10);
            c=s/10;
            K/=10;
        }
        
        while (i>=0) {
            int s = A[i] + c;
            ans.push_back(s%10);
            c=s/10;
            i--;
        }
        
        while (c) {
            ans.push_back(c%10);
            c/=10;
        }
        
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
