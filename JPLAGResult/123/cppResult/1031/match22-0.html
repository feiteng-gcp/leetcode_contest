class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        int carry = 0;
        int n = A.size() - 1;
        while(K || carry){
        	//cout << K << " " <<  carry << endl;
    		int a = K % 10;
        	if(n >= 0){
        		int new_carry = (K % 10 + carry + A[n]) / 10;
        		A[n] = (A[n] + carry + K % 10) % 10;
        		carry = new_carry;
        		n--;
        	}
        	else{
        		A.insert(A.begin(), (K % 10 + carry) % 10);
        		carry = (K % 10 + carry) / 10;
        	}
        	K = K / 10;
        }
        return A;
    }
};
