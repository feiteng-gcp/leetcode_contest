class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        reverse(A.begin(), A.end());
        int carry = 0;
        for (int i = 0; (carry > 0 || K > 0); i++) {
            if (i == A.size()) A.push_back(0);
            A[i] += K % 10 + carry;
            carry = A[i] / 10;
            A[i] %= 10;
            K /= 10;
        }
        reverse(A.begin(), A.end());
        return A;
    }
};
