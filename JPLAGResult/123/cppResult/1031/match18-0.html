class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        int t = K;
        for(int i = A.size() - 1;i >= 0;i--){
            A[i] += t;
            t = A[i] / 10;
            A[i] = A[i] % 10;
        }
        while(t != 0){
            A.insert(A.begin(),t);
            t = A[0] / 10;
            A[0] = A[0] % 10;
        }
        return A;
    }
};
