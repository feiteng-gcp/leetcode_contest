class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        reverse(A.begin(), A.end());
        int i = 0;
        int carry = 0;
        while(K > 0 || i < A.size()){
            int c = K % 10;
            if( i >= A.size()) A.push_back(0);
            
            A[i] += (c + carry);
            if(A[i] >= 10){
                A[i] -= 10;
                carry = 1;
            }else{
                carry = 0;
            }
            //cout<<carry<<endl;
            K = K / 10;
            i++;
        }
        
        
        
        if(carry > 0) A.push_back(1);
        reverse(A.begin(), A.end());
        return A;
    }
};
