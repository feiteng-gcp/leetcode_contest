class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> vK;
        while(K > 0) {
            vK.push_back(K%10);
            K /= 10;
        }
        int i = 0, j = vK.size() - 1;
        while(i < j) {
            swap(vK[i++], vK[j--]);
        }
        vector<int> ans;
        i = A.size() - 1, j = vK.size() - 1;
        int c = 0;
        while(i >= 0 && j >= 0) {
            int t = A[i] + vK[j] + c;
            ans.push_back(t%10);
            // cout << t % 10 << " c: " << c << endl;

            c = t / 10;
            i --;
            j --;
        }
        while(i >= 0) {
            int t = A[i] + c;
            ans.push_back(t%10);
            // cout << t % 10 << endl;
            c = t / 10;
            i --;
        }
        while(j >= 0) {
            int t = vK[j] + c;
            ans.push_back(t%10);
            // cout << t % 10 << endl;
            c = t / 10;
            j --;
        }
        if(c > 0) {
            ans.push_back(c);
        }
        i = 0;
        j = ans.size() - 1;
        while(i < j) {
            swap(ans[i++], ans[j--]);
        }
        return ans;
    }
};
