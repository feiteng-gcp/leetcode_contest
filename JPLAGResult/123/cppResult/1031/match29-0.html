class Solution {
public:
	vector<int> addToArrayForm(vector<int>& A, int K) {
		int n = A.size(), i, j, k;
		A[n - 1] += K;
		for (i = n - 1; i >= 1; i--) {
			A[i - 1] += A[i] / 10;
			A[i] %= 10;
		}
		while (A[0] >= 10) {
			A.insert(A.begin(), A[0] / 10);
			A[1] %= 10;
		}
		return A;
	}
};
