class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        int jin = 0;
        for (int i = A.size() - 1; i >= 0; i --) {
            A[i] = (A[i] + K % 10 + jin);
            K /= 10;
            jin = A[i] / 10;
            A[i] %= 10;
        }
        vector<int> ans;
        while (K != 0 || jin != 0) {
            ans.push_back((jin + K % 10) % 10);
            jin = (jin + K % 10) / 10;
            K /= 10;
        }
        std::reverse(ans.begin(), ans.end());
        ans.insert(ans.end(), A.begin(), A.end());
        return ans;
    }
};
