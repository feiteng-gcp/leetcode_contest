class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        A[A.size() - 1] += K;

        int i = A.size() - 1;
        while (i > 0 && A[i] >= 10) {
            A[i - 1] += A[i] / 10;
            A[i] %= 10;
            i--;
        }
        while (A[0] >= 10) {
            A.insert(A.begin(), A[0] / 10);
            A[1] %= 10;
        }
        return A;
    }
};
