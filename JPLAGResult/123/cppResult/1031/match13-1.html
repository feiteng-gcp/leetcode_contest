class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        int L = A.size();

        A[L-1] += K;
        for (int i=L-1;i>=1;i--) {
            if (A[i]>=10) {
                A[i-1] += (A[i]/10);
                A[i] %= 10;
            }
        }
        while (A[0]>9) {
            A.insert(A.begin(),A[0]/10);
            A[1] %= 10;
        }
        return A;
    }
};
