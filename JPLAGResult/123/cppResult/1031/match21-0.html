class Solution {
public:
	vector<int> addToArrayForm(vector<int>& A, int K) {
		vector<int> ret;
		int car = 0;
		for (int i = A.size() - 1; i >= 0; i--) {
			int b = A[i] + K % 10 + car;
			K /= 10;
			ret.push_back(b % 10);
			car = b / 10;
		}
		K += car;
		while (K){
			ret.push_back(K % 10);
			K /= 10;
		}
		reverse(ret.begin(), ret.end());

		return ret;
	}
};
