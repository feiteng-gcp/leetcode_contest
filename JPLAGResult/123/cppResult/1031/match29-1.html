class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        reverse(A.begin(),A.end());
        int n=A.size(),i;
        A[0]+=K;
        for(i=0;i+1<n;i++)if(A[i]>9)
        {
            A[i+1]+=A[i]/10;
            A[i]%=10;
        }
        for(;A[n-1]>9;)
        {
            A.push_back(A[n-1]/10);
            A[n-1]%=10;
            n++;
        }
        reverse(A.begin(),A.end());
        return A;
    }
};
