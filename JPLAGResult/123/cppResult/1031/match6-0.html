class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> result;
        int sz = A.size(), i = sz - 1, carry = K;
        while (i >= 0 or carry > 0)
        {
            int val = (i >= 0 ? A[i] : 0) + carry % 10;
            carry = carry / 10 + val / 10;
            result.push_back(val % 10);
            --i;
        }
        reverse(result.begin(), result.end());
        return result;
    }
};
