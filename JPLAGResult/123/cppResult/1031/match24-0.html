class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        reverse(A.begin(), A.end());
        int idx = 0;
        int carry = 0;
        while (K > 0 || carry > 0) {
            int d = K % 10;
            if (idx >= A.size()) {
                A.push_back(0);
            }
            int s = A[idx] + d + carry;
            carry = s / 10;
            A[idx] = s % 10;
            idx++;
            K /= 10;
        }
        reverse(A.begin(), A.end());
        return A;
    }
};
