class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        int n=A.size();
        int s=K;
        for(int i=n-1;i>=0;--i)
        {
        	A[i]+=s;
        	s=A[i]/10;
        	A[i]%=10;
        }
        while(s>0)
        {
        	A.insert(A.begin(),s%10);
        	s/=10;
        }
    return A;
    }
};
