class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> ans;
        int i=A.size()-1,c=0;
        while(K){
            int n=i>=0?A[i--]:0;
            int s=K%10+n+c;
            ans.push_back(s%10);
            c=s/10;
            K/=10;
        }
        while(i>=0){
            int s=A[i--]+c;
            ans.push_back(s%10);
            c=s/10;
        }
        if(c) ans.push_back(1);
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
