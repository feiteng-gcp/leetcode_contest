class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        int len=A.size(), num=K, carry=0;
        vector<int> ret;
        for (int i=len-1; i>=0; i--) {
            int temp=A[i]+num%10+carry;
            carry=temp/10;
            num=num/10;
            ret.push_back(temp%10);
        }
        while (num!=0 || carry!=0) {
            int temp=num%10+carry;
            num=num/10;
            carry=temp/10;
            ret.push_back(temp%10);
        }
        reverse(ret.begin(), ret.end());
        return ret;
    }
};
