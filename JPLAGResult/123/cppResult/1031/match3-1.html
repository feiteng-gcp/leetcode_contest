#include <algorithm>
#include <vector>

using namespace std;


class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> b = A;
        reverse(b.begin(), b.end());
        //lower first
        int c = 0;
        for(int i  = 0; i < b.size(); i++){
            b[i] += c + K % 10;
            c = b[i] / 10;
            b[i] %= 10;
            K /= 10;
            if(c == 0 && K == 0){
                break;
            }
        }
        while(c > 0 || K > 0){
            c += K % 10;
            b.push_back(c % 10);
            c /= 10;
            K /= 10;
        }
        reverse(b.begin(), b.end());
        return b;
    }
};
