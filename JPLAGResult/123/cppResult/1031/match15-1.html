class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> B(A);
        reverse(B.begin(), B.end());
        
        B[0] += K;
        int carry = 0;
        for (int i=0; i<B.size(); ++i) {
            int sum = B[i] + carry;
            B[i] = sum % 10;
            carry = sum / 10;
        }
        
        while (carry > 0) {
            B.push_back(carry % 10);
            carry /= 10;
        }
        
        reverse(B.begin(), B.end());
        return B;
    }
};
