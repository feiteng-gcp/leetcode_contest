class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> result;
        int cin = 0;
        for (int i = A.size() - 1; i >= 0 || K > 0; --i) {
            int val = cin;
            if (K > 0) {
                val += K % 10;
            }
            if (i >= 0) {
                val += A[i];
            }
            result.push_back(val % 10);
            cin = val / 10;
            K /= 10;
        }
        if (cin > 0) {
            result.push_back(cin);
        }
        return vector<int>(result.rbegin(), result.rend());
    }
};
