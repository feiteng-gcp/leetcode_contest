class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        int n=A.size();
        int c=0;
        
        for(int i=n-1;i>=0;i--){
            int d=K%10;
            K/=10;
            A[i]+=d+c;
            c=A[i]/10;
            A[i]%=10;
        }
        while(K){
            int d=K%10;
            K/=10;
            int s=d+c;
            c=s/10;
            s=s%10;
            A.insert(A.begin(),s);
        }
        if(c) A.insert(A.begin(),c);
        return A;
    }
};
