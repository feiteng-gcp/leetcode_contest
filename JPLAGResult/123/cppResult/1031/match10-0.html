class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> ans;
        int left = 0;
        for(int i = A.size() - 1; i >= 0; i--) {
            int cur = A[i] + left + K % 10;
            ans.push_back(cur % 10);
            left = cur / 10;
            K /= 10;
        }
        while(K != 0 || left != 0) {
            int cur = left + K % 10;
            ans.push_back(cur % 10);
            left = cur / 10;
            K /= 10;
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
