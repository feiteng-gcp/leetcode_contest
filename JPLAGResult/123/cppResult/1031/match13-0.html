class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> res;
        for(int i = A.size() - 1; i > -1; i--) {
            res.push_back(A[i]);
        }
        
        res[0] += K;
        for(int i = 0; i < res.size() - 1; i++) {
            if(res[i] > 9) {
                res[i + 1] += res[i] / 10;
                res[i] %= 10;
            }
        }
        
        while(res[res.size() - 1] > 9) {
            res.push_back(res[res.size() - 1] / 10);
            res[res.size() - 2] %= 10;
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
