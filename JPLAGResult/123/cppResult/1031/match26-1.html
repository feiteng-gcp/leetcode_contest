class Solution {
public:
	vector<int> addToArrayForm(vector<int>& A, int K) {
		int n = A.size();
		int pos = n-1;
        
		while (K && (pos >= 0)) {
            // cout << A[pos] << endl;
			A[pos] += K % 10;
            // cout << A[pos] << endl;
			K = (K / 10) + (A[pos] / 10);
			A[pos] = A[pos] % 10;
            // cout << A[pos] << endl;
			--pos;
		}

		while (K) {
			A.insert(A.begin(), K % 10);
			K /= 10;
		}

		return A;
	}
};
