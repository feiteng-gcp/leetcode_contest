class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> B = A;
    reverse(B.begin(), B.end());
    int id = 0;
    while(K)
    {
        if(id >= B.size()) B.push_back(0);
        B[id] += K;
        K = B[id] / 10;
        B[id] %= 10;
        id++;
    }
    reverse(B.begin(), B.end());
    return B;
    }
};
