class Solution 
{
public:
    vector<int> addToArrayForm(vector<int>& A, int K) 
    {
        reverse(A.begin(), A.end());
        int d = 0;
        int size = A.size();
        while(K > 0)
        {
            while(d >= size-2)
            {
                A.push_back(0);
                size++;
            }
            int num = K%10;
            K /= 10;
            A[d] += num;
            A[d+1] += A[d]/10;
            A[d] %= 10;
            d++;
        }
        size++;
        A.push_back(0);
        while(d < size-1)
        {
            A[d+1] += A[d]/10;
            A[d] %= 10;
            d++;
        }
        while(size > 1 && A[size-1] == 0)
        {
            A.pop_back();
            size--;
        }
        reverse(A.begin(), A.end());
        return A;
    }
};
