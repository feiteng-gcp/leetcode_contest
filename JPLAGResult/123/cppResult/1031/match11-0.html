class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> B;
        while(K>0){
            B.push_back(K%10);
            K/=10;
        }
        reverse(A.begin(),A.end());
        int add=0;
        vector<int> C;
        int p=0;
        while(p<A.size()&&p<B.size()){
            int d=add+A[p]+B[p];
            C.push_back(d%10);
            add=d/10;
            p++;
        }
        
         while(p<A.size()){
            int d=add+A[p];
            C.push_back(d%10);
            add=d/10;
             p++;
        }
        
        while(p<B.size()){
            int d=add+B[p];
            C.push_back(d%10);
            add=d/10;
            p++;
        }
        
        if(add>0){
            C.push_back(add);
        }
        reverse(C.begin(),C.end());
        return C;
    }
};
