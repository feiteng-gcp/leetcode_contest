class Solution {
public:
    vector<int> addToArrayForm(vector<int>& A, int K) {
        vector<int> ans;
        int cur = A.size() - 1;
        int carry = 0;
        while (cur >=0 || K > 0 || carry > 0) {
            int digit = K % 10;
            K /= 10;
            int a = cur >= 0 ? A[cur] : 0;
            if (digit + a + carry >= 10) {
                ans.push_back(digit + a + carry - 10);
                carry = 1;
            }
            else {
                ans.push_back(digit + a + carry);
                carry = 0;
            }
            --cur;
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
