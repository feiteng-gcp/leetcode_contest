class Solution {
public:
    int p[26];
    void init() {
        for (int i = 0;i < 26;i++) {
            p[i] = i;
        }
    }
    int find_set(int x) {
        if (p[x] == x) {
            return x;
        } else {
            return p[x] = find_set(p[x]);
        }
    }
    void union_set(int x, int y) {
        x = find_set(x);
        y = find_set(y);
        if (x == y) {
            return;
        }
        p[x] = y;
    }
    bool equationsPossible(vector<string>& equations) {
        init();

        for (int i = 0;i < equations.size();i++) {
            string s = equations[i];
            if (s[1] == '=') {
                union_set(s[0] - 'a', s[3] - 'a');
            }
        }
        for (int i = 0;i < equations.size();i++) {
            string s = equations[i];
            if (s[1] == '!') {
                if (find_set(s[0] - 'a') == find_set(s[3] - 'a')) {
                    return false;
                }
            }
        }
        return true;
    }
};
