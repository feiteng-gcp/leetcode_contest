class Solution {
public:
    int f[30];
    void ini()
    {
        for(int i=0;i<30;++i)
        	f[i]=i;
    }
    int father(int i)
    {
    	if(f[i]==i)
    		return i;
    	else 
    		f[i]=father(f[i]);
    	return f[i];
    } 
    bool equationsPossible(vector<string>& equations) {
        int n=equations.size();
        ini();
        for(int i=0;i<n;++i)
        	if(equations[i][1]=='=')
        	{
        		int x,y;
        		x=father(equations[i][0]-'a');
        		y=father(equations[i][3]-'a');
        		f[x]=y;
        	}
        for(int i=0;i<n;++i)
        	if(equations[i][1]=='!')

        	{
        		int x,y;
        		x=father(equations[i][0]-'a');
        		y=father(equations[i][3]-'a');
        		if(x==y)
        		{
        			ini();
        			return false;
        		}
        	}
        return true;
    }
};
