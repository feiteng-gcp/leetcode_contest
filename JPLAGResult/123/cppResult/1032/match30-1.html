class Solution {
public:
    int p[26];
    bool equationsPossible(vector<string>& equations) {
        for (int i = 0; i < 26; i++) {
            p[i] = i;
        }
        for (auto &e : equations)
            if (e[1] == '=') merge(e[0] - 'a', e[3] - 'a');
        for (auto &e : equations)
            if (e[1] == '!' && find(e[0] - 'a') == find(e[3] - 'a')) return false;
        return true;                

    }
    
    int find(int x) {
        return p[x] == x ? x : (p[x] = find(p[x]));
    }
    
    void merge(int x, int y) {
        x = find(x);
        y = find(y);
        if (x != y) p[y] = x;
    }
};
