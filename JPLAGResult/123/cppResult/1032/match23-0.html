class Solution {
public:
    int Father[26];
    int Rank[26];
    int find(int x) {
        while(Father[x] != x) {
            x = Father[x];
        }
        return x;
    }
    bool equationsPossible(vector<string>& equations) {
        int r = 0;
        int m = equations.size();
        for (int i = 0; i<26; ++i) {
            Father[i] = i;
            Rank[i] = 0;
        }
        int cnt = 0;
        for (int i = 0; i<m; ++i) {
            if(equations[i][1]=='=') {
                swap(equations[i],equations[cnt]);
                ++cnt;
            }
        }
        
        for(int i=0; i<m; ++i) {
            int ra = find(equations[i][0]-'a'), rb = find(equations[i][3]-'a');
            if(equations[i][1]=='!') {
                if(ra == rb) return false;
            }
            else {
                if(ra!=rb) {
                    if(Rank[ra]>Rank[rb])
                        swap(ra,rb);
                    Father[ra] = rb;
                    if(Rank[ra] == Rank[rb]) Rank[rb]++;
                }
            }
        }
        return true;
    }
};
