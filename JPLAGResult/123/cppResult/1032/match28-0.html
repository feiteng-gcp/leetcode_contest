class Solution {
public:
int fa[26];

int Find(int x)
{
    return x==fa[x]?x:fa[x]=Find(fa[x]);
}

void Union(int x,int y)
{
    int a=Find(x),b=Find(y);
    fa[a]=b;
}

bool equationsPossible(vector<string>& equations)
{
    for(int i=0;i<26;i++)
        fa[i]=i;
     vector<pair<int,int> > res;
     int n=equations.size();
     for(int i=0;i<n;i++)
     {
         if(equations[i][0]==equations[i][3])
         {
             if(equations[i][1]=='!') return false;
             continue;
         }
         int a=equations[i][0]-'a';
         int b=equations[i][3]-'a';
         if(equations[i][1]=='=')
            Union(a,b);
         else
            res.push_back(make_pair(a,b));
     }
     for(auto it:res)
     {
         if(Find(it.first)==Find(it.second)) return false;
     }
     return true;
}
};
