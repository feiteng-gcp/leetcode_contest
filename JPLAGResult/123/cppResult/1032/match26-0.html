class Solution {
public:
    int f[26];
    int get(int x)
    {
        if(f[x]==x)return x;
        return f[x]=get(f[x]);
    }
    bool equationsPossible(vector<string>& equations) {
        int m=equations.size(),n=26,i;
        for(i=0;i<n;i++)f[i]=i;
        for(i=0;i<m;i++)if(equations[i][1]=='=')f[get(equations[i][0]-'a')]=get(equations[i][3]-'a');
        for(i=0;i<m;i++)if(equations[i][1]=='!'&&get(equations[i][0]-'a')==get(equations[i][3]-'a'))return false;
        return true;
    }
};
