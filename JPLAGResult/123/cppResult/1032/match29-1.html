class Solution {
public:
    int find(vector<int>& v, int cur) {
        while (v[cur] != -1) {
            cur = v[cur];
        }
        
        return cur;
    }
    
    bool equationsPossible(vector<string>& equations) {
        vector<int> v(26, -1);
        for (int i = 0; i < equations.size(); ++i) {
            int v1 = equations[i][0] - 'a';
            int v2 = equations[i][3] - 'a';
            if (equations[i][1] == '=') {
                int a1 = find(v, v1);
                int a2 = find(v, v2);
                if (a1 != a2) {
                    v[a1] = a2;
                }
            }
        }
        for (int i = 0; i < equations.size(); ++i) {
            int v1 = equations[i][0] - 'a';
            int v2 = equations[i][3] - 'a';
            if (equations[i][1] != '=') {
                if (v1 == v2) {
                    return false;
                }
                int a1 = find(v, v1);
                int a2 = find(v, v2);
                if (a1 == a2) {
                    return false;
                }
            }
        }
        return true;
    }
};
