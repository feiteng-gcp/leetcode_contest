class Solution {
public:
    int fa[30];
    int find(int x) {
        while (x != fa[x]) {
            fa[x] = fa[fa[x]];
            x = fa[x];
        }
        return x;
    }
    void merge(int x, int y) {
        x = find(x);
        y = find(y);
        fa[x] = y;
    }
    bool equationsPossible(vector<string>& equations) {
        for (int i = 0; i < 30; i ++) {
            fa[i] = i;
        }
        vector<int> a, b;
        for (int i = 0; i < equations.size(); i ++) {
            int x = equations[i][0] - 'a', y = equations[i][3] - 'a';
            if (equations[i][1] == '=') {
                merge(x, y);
            } else {
                a.push_back(x);
                b.push_back(y);
            }
        }
        for (int i = 0; i < a.size(); i ++) {
            if (find(a[i]) == find(b[i])) {
                return false;
            }
        }
        return true;
    }
};
