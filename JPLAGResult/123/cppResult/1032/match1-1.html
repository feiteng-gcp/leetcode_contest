class Solution {
public:
    int root(int a,vector<int>&id){
        while(id[a] != a){
            id[a] = id[id[a]];
            a = id[a];
        }
        return a;
    }
    void union1(int a,int b,vector<int>&id){
        int x = root(a,id);
        int y = root(b,id);
        id[x] = id[y];
    }
    
    bool equationsPossible(vector<string>& equations) {
        //unordered_map<char,char>eq,noteq;

        vector<int>id(26,0);
        for(int i=0;i<26;++i)id[i]=i;
        
        for(auto i:equations){
            if(i[1]=='='){
                union1(i[0]-'a',i[3]-'a',id);
            }
        }
        for(auto i:equations){
            if(i[1]=='!'){
                if(root(i[0]-'a',id) == root(i[3]-'a',id))return 0;
            }
        }
        return 1;
    }
};
