#include <vector>
#include <numeric>
using namespace std;

vector<int> id;
vector<int> sz;
int groups;


inline void init(int n) {
	groups = n;
	id.resize(n);
	sz.resize(n, 1);
	for (int i = 0; i < n; ++i) {
		id[i] = i;
	}
}

inline int findRoot(int p) {
	while (p != id[p]) {
		id[p] = id[id[p]];
		p = id[p];
	}
	return p;
}

inline void comb(int p, int q) {
	int i = findRoot(p);
	int j = findRoot(q);
	if (i == j) {
		return;
	}
	if (sz[i] < sz[j]) {
		id[i] = j;
		sz[j] += sz[i];
	} else {
		id[j] = i;
		sz[i] += sz[j];
	}
	--groups;
}
class Solution {
public:
    bool equationsPossible(vector<string>& e) {
        vector<string> eq1;
        init(26);
        for(string &p:e){
            if(p[1]=='!'){
                eq1.push_back(p);
            }else{
                comb(p[0]-'a',p[3]-'a');
            }
        }
        for(string &p:eq1){
            if(findRoot(p[0]-'a')==findRoot(p[3]-'a')){
                return false;
            }   
            
        }
        return true;
        
    }
};
