#define maxn 1000000
#define _for(i,a,b) for(int i = (a);i < (b);i ++)
#define pb push_back


class Solution
{
	public:
		vector<int> root;
		int _find(int x)
		{
			if(root[x]==x)    return x;
			return root[x] = _find(root[x]);
		}
		void merge(int x,int y)
		{
			int rx = _find(x);
			int ry = _find(y);
			if(rx==ry) return ;
			root[rx] = ry;
		}
		bool equationsPossible(vector<string>& equations)
		{
			for(int i = 0;i < 27;i ++)
				root.pb(i);

			for(int i = 0; i < equations.size(); i ++)
			{
				if(equations[i][1]=='=')
				{
					merge(equations[i][0]-'a',equations[i][3]-'a');
				}
			}
			for(int i = 0; i < equations.size(); i ++)
			{
				if(equations[i][1]=='!')
				{
					if(_find(equations[i][0]-'a')==_find(equations[i][3]-'a'))
						return false;
				}
			}
			return true;
		}
};
