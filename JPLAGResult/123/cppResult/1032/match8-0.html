class Solution {
    vector<int> ss;
    int find(int s){
        if(ss[s]!=s) ss[s]=find(ss[s]);
        return ss[s];
    }
    
    void merge(int a,int b){
        ss[find(a)]=find(b);
    }
public:
    bool equationsPossible(vector<string>& equations) {
        ss=vector<int>(26);
        for(int i=0;i<26;i++) ss[i]=i;
        for(auto s:equations){
            if(s[1]=='='){
                merge(s[0]-'a',s[3]-'a');
            }
        }
        
        for(auto s:equations){
            if(s[1]=='!'){
                if(find(s[0]-'a')==find(s[3]-'a')) return false;
            }
        }
        return true;
    }
};
