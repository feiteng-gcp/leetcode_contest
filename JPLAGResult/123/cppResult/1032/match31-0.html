class Solution {
public:
    bool equationsPossible(vector<string>& e) {
        int a[130];
        for(int i = 'a';i<='z';i++)
            a[i] = i;
        for(int i = 0;i<e.size();i++){
            if(e[i][1]=='!')continue;
            int temp = min(a[e[i][0]],a[e[i][3]]);
            int temp0 = max(a[e[i][0]],a[e[i][3]]);
            for(int j = 'a';j<='z';j++)
                if(a[j]==temp0)a[j] = temp;
        }
        

        for(int i = 0;i<e.size();i++){
            if(e[i][1]=='!'&&a[e[i][0]]==a[e[i][3]])return false;
        
        }
        return true;
    }
};
