#define pb push_back
#define ll long long
#define mp make_pair
#define f first
#define s second
#define pii pair < int, int >
#define pll pair < ll, ll >
#define all(s) s.begin(), s.end()
#define sz(s) (int) s.size()
#define forit(it, s) for(__typeof(s.begin()) it = s.begin(); it != s.end(); it++)
#define vi vector < int >

const int inf = (int)1e9;
const int mod = (int) 1e9 + 7;

class Solution {
public:
    vector < int > g[33];
    vector < int > gr[33];
    int group[33];
    int col[33];
    bool result;

    
    void dfs(int v, int c) {
        group[v] = c;
        for(int to : gr[v]) {
            if (group[to] == 0) {
                dfs(to, c);
            } else {
                if (group[to] != c) {
                    result = false;
                }
            }
        }
    }
    
    bool equationsPossible(vector<string>& e) {
        for (int i=0;i<e.size();i++) {
            string s = e[i];
            int ff = s[0] - 'a';
            int ss = s[3] - 'a';
            if (s[1] == '!') {
                g[ff].pb(ss);
                g[ss].pb(ff);
            } else {
                gr[ff].pb(ss);
                gr[ss].pb(ff);
                continue;
            }
        }
        int cc = 1;
        result = true;
        for (int i=0;i<26;i++) {
            if (group[i] == 0) {
                dfs(i, cc);            
                cc++;
            }
        }
        for (int i=0;i<26;i++) {
            for (int to : g[i]){
                if (group[i] == group[to]) {
                    result = false;
                }
            }
        }
                
        return result;
    }
};
