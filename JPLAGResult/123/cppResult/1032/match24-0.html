class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        vector<int> parent(26, -1);
        for (int i = 0; i < 26; i++)
            parent[i] = i;
        
        function<int(int)> root = [&](int x) {
            return (parent[x] == x ? x : parent[x] = root(parent[x]));
        };
        
        for (const string &s : equations) {
            int a = s[0] - 'a', b = s[3] - 'a';
            if (s[1] == '=')
                parent[root(a)] = root(b);
        }
        
        for (const string &s : equations) {
            int a = s[0] - 'a', b = s[3] - 'a';
            if (s[1] == '!' && root(a) == root(b)) {
                return false;
            }
        }
        return true;
    }
};
