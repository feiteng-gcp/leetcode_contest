class Solution {
public:
    int root(vector<int> &p, int i)
    {
        while (p[i] != i) {
            p[i] = p[p[i]];
            i = p[i];
        }
        return i;
    }
    
    void uni(vector<int> &p, int i, int j)
    {
        int x = root(p, i);
        int y = root(p, j);
        p[x] = y;
    }
    
    bool equationsPossible(vector<string>& equations) {
        if (equations.empty()) return true;
        vector<int> v(26, -1);
        for (int i = 0; i < 26; i++) v[i] = i;
        int n = equations.size();
        for (int i = 0; i < n; i++) {
            int a = equations[i][0] - 'a';
            int b = equations[i][3] - 'a';
            char sign = equations[i][1];
            if (a == b) continue;
            if (sign == '=') {
                uni(v, a, b);
            }
        }
        for (int i = 0; i < n; i++) {
            char sign = equations[i][1];
            int a = equations[i][0] - 'a';
            int b = equations[i][3] - 'a';
            int x = root(v, a);
            int y = root(v, b);
            if (sign == '!' && (x == y)) {
                return false;
            }
        }
        return true;
    }
};
