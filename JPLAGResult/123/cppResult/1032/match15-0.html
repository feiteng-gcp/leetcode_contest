class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        vector<unordered_set<char>> e(26),ne(26);
        for(int i=0;i<26;i++) e[i].insert('a'+i);
        int len=equations.size();
        for(auto str:equations){
            if(str[1]=='='){
                for(int i=0;i<26;i++){
                    if(e[i].find(str[0])!=e[i].end()){
                        e[i].insert(str[3]);
                    }
                }
                for(int i=0;i<26;i++){
                    if(e[i].find(str[3])!=e[i].end()){
                        e[i].insert(str[0]);
                    }
                }
            }else{
                for(int i=0;i<26;i++){
                    if(e[i].find(str[0])!=e[i].end()){
                        ne[i].insert(str[3]);
                    }
                }
                for(int i=0;i<26;i++){
                    if(e[i].find(str[3])!=e[i].end()){
                        ne[i].insert(str[0]);
                    }
                }
            }
        }
        for(int i=0;i<26;i++){
            for(char c:e[i]){
                if(ne[i].find(c)!=ne[i].end()) return false;
            }
        }
        return true;
    }
};
