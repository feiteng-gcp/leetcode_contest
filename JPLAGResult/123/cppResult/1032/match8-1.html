class Solution {
    vector<int> fus;
    int find(int x) {

        return x == fus[x] ? x : fus[x] = find(fus[x]);
    }
public:
    bool equationsPossible(vector<string>& equations) {
        fus = vector<int>(26);
        for (int i = 0; i < 26; ++i)
            fus[i] = i;
        for (string equ : equations) {
            int a = equ[0] - 'a';
            int b = equ[3] - 'a';
            if (equ[1] == '=') {
                fus[find(a)] = find(b);
            }
        }
        for (string equ : equations) {
            int a = equ[0] - 'a';
            int b = equ[3] - 'a';
            if (equ[1] == '!')
                if (find(a) == find(b))
                    return false;
        }
        return true;
    }
};
