class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        map<int,set<int> > degraph;
        int N=equations.size();
        for(int i=0;i<N;i++){
            if(equations[i][1]=='!'){
                degraph[equations[i][0]-'a'].insert(equations[i][3]-'a');
                degraph[equations[i][3]-'a'].insert(equations[i][0]-'a');
                if(equations[i][0]-'a'==equations[i][3]-'a') return false;
            }
        }
        vector<int> vec(26);
        for(int i=0;i<26;i++)
            vec[i]=i;
        for(int i=0;i<N;i++){
            if(equations[i][1]=='='){
                int from=equations[i][0]-'a';
                int to=equations[i][3]-'a';
                int x=findpar(from,vec);
                int y=findpar(to,vec);
                vec[x]=y;
            }
        }
        for(int i=0;i<26;i++)
            for(int j=i+1;j<26;j++)
                if(findpar(i,vec)==findpar(j,vec) &&degraph[i].find(j)!=degraph[i].end()) return false;
        return true;
    }
    int findpar(int i,vector<int> vec){
        while(i!=vec[i]) i=vec[i];
        return i;
    }
};
