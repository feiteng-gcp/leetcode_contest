class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        vector< vector<int> > vv(26, vector<int>(26, 0));
        vector<bool> ve(26, false);
        for(int i = 0; i < equations.size(); i ++) {
            string & one = equations[i];
            char p1 = one[0], p2 = one[3];
            if('=' == one[1]) {
                vv[p1-'a'][p2-'a'] = 1;
                vv[p2-'a'][p1-'a'] = 1;
                ve[p1-'a'] = ve[p2-'a'] = true;
            }
        }
        int val = 0;
        unordered_map<char, int> um;
        for(int i = 0; i < 26; i ++) {
            if(!ve[i]) continue;
            val ++;
            queue<char> qc;
            qc.push(i+'a');
            um[i+'a'] = val;
            ve[i] = false;
            while(!qc.empty()) {
                char c = qc.front();
                qc.pop();
                vector<int> & one = vv[c-'a'];

                for(int i = 0; i < one.size(); i ++) {
                    if(one[i] > 0 && ve[i]) {
                        qc.push(i+'a');
                        um[i+'a'] = val;
                        ve[i] = false;
                    }
                }
            }
        }
        /*
        unordered_map<char, int>::const_iterator citer = um.begin();
        while(citer != um.end()) {
            cout << citer->first << " -> " << citer->second << endl;
            citer ++;
        }
        */
        for(int i = 0; i < equations.size(); i ++) {
            string & one = equations[i];
            char p1 = one[0], p2 = one[3];
            if('!' == one[1]) {
                if(p1 == p2) return false;
                if(um.find(p1) != um.end() && um.find(p2) != um.end()) {
                    if(um[p1] == um[p2]) return false;
                }
            }
        }
        return true;
    }
};
