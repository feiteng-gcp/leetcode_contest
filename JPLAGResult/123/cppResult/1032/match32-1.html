class Solution {
public:
    int m[30][30], v[30], dfn = 0;
    
    void color(int cur)
    {
        v[cur] = dfn;
        for(int i = 0; i < 26; ++i)
            if(m[cur][i] == 1 && !v[i])
                color(i);
    }
    
    bool equationsPossible(vector<string>& equations) {
        memset(m, 0, sizeof m);
        memset(v, 0, sizeof v);
        for(string s : equations)
        {
            if(s[1] == '!')
            {
                if(m[s[0]-'a'][s[3]-'a'] == 1 || s[0] == s[3]) return false;
                m[s[0]-'a'][s[3]-'a'] = m[s[3]-'a'][s[0]-'a'] = -1;
            }
            else
            {
                if(m[s[0]-'a'][s[3]-'a'] == -1) return false;
                m[s[0]-'a'][s[3]-'a'] = m[s[3]-'a'][s[0]-'a'] = 1;
            }
        }

        for(int i = 0; i < 26; ++i)
        {
            if(!v[i])
            {
                ++dfn;
                color(i);
            }
        }
        for(int i = 0; i < 26; ++i)
        {
            for(int j = 0; j < 26; ++j)
            {
                if(m[i][j] == -1 && v[i] == v[j]) return false;
            }
        }
        return true;
    }
};
