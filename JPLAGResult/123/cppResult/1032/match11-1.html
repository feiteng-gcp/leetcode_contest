class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        vector<int> setNum(26);
        for (char i = 'a'; i<='z'; ++i) {
            setNum[i-'a'] = i - 'a';
        }
        
        // Process all equals

        for (auto& eq : equations) {
            if (eq[1] == '=') {
                union_set(setNum, eq[0], eq[3]);
            }
        }
            
        // Check all non-equal
        for (auto& eq : equations) {
            if (eq[1] == '!') {
                int id1 = get_root(setNum, eq[0]);
                int id2 = get_root(setNum, eq[3]);
                if (id1 == id2) return false;
            }
        }
        
        return true;
    }

private:
    int get_root(vector<int>& setNum, char a) {
        int root = a - 'a';
        while (setNum[root] != root) root = setNum[root];
        
        return root;
    }
    
    void union_set(vector<int>& setNum, char a, char b) {
        int rootA = get_root(setNum, a);    
        int rootB = get_root(setNum, b);
        setNum[rootA] = rootB;
    }
};
