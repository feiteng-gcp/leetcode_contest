class Solution {
public:
    vector<int> equal;
    bool equationsPossible(vector<string>& equations) {
        equal.resize(26);

        for(int i=0;i<26;i++) equal[i]=i;
        for(auto str:equations){
            if(str[1]=='='){
                connect(str[0]-'a',str[3]-'a');
            }
        }
        for(auto str:equations){
            if(str[1]=='!'){
                if(find(str[0]-'a')==find(str[3]-'a')) return false;
            }
        }
        return true;
    }
    int find(int i){
        if(equal[i]!=i) equal[i]=find(equal[i]);
        return equal[i];
    }
    
    void connect(int i,int j){
        int rooti=find(i);
        int rootj=find(j);
        if(rooti!=rootj) equal[rooti]=equal[rootj];
    }
};
