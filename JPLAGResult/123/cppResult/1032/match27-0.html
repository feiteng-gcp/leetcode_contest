class Solution {
public:
    class UF
    {
    public:
        unordered_map<char, char> parent;
        unordered_map<char, int> rank;
        
        char find(char c) {
            if (parent.count(c)==0) {
                parent[c]=c;
                rank[c]=0;
                return c;
            }
            while (c!=parent[c]) {
                parent[c]=parent[parent[c]];
                c=parent[c];
            }
            return c;           
        }
        
        void merge(char c1, char c2) {
            char p1=find(c1), p2=find(c2);
            if (p1==p2)
                return;
            if (rank[p1]<rank[p2]) {
                parent[p1]=p2;
                rank[p2]++;
            }
            else {
                parent[p2]=p1;
                rank[p1]++;
            }
            return;
        }
    };
    bool equationsPossible(vector<string>& equations) {
        int len=equations.size();
        UF uf;
        for (int i=0; i<len; i++) {
            string temp=equations[i];
            if (temp[1]=='=') {
                uf.merge(temp[0], temp[3]);
            }
        }
        for (int i=0; i<len; i++) {
            string temp=equations[i];
            if (temp[1]=='!') {
                char p1=uf.find(temp[0]), p2=uf.find(temp[3]);
                if (p1==p2)
                  return false;
            }
        }
        return true;
    }
};
