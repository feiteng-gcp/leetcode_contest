class Solution {
public:
    int find(vector<int>&father,int x){
        while(x != father[x]) x = father[x];
        return x;
    }
    bool equationsPossible(vector<string>& equations) {
        vector<int>father(30);
        int ans = 1;

        for(int i = 0;i < 26;i++) father[i] = i;
        for(int i = 0;i < equations.size();i++){
            if(equations[i][1] == '='){
                int l = equations[i][0] - 'a';
                int r = equations[i][3] - 'a';
                if(find(father,l) != find(father,r)) father[find(father,l)] = find(father,r);
            }
        }
        for(int i = 0;i < equations.size();i++){
            if(equations[i][1] == '!'){
                int l = equations[i][0] - 'a';
                int r = equations[i][3] - 'a';
                if(find(father,l) == find(father,r)){
                    ans = 0;
                    break;
                }
            }
        }
        if(ans == 0) return false;
        return true;
    }
};
