class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        vector<vector<int>> m(26, vector<int>(26, INT_MAX));
        vector<vector<bool>> b(26, vector<bool>(26, false));
        
        for(int i = 0; i < equations.size(); i++) {
            int x = equations[i][0] - 'a';
            int y = equations[i][3] - 'a';
            if(x == y and equations[i][1] == '!') return false;
            if(equations[i][1] == '=') {
                if(b[x][y]) return false;
                m[x][y] = 1;
                m[y][x] = 1;
            } else {
                b[x][y] = true;
                b[y][x] = true;
            }
        }
        
        for(int k = 0; k < m.size(); k++) {
            for(int i = 0; i < m.size(); i++) {
                for(int j = 0; j < m.size(); j++) {
                    if(m[i][k] < INT_MAX and m[k][j] < INT_MAX)
                       if(m[i][j] == INT_MAX)
                           m[i][j] = m[i][k] + m[k][j];
                        else
                            m[i][j] = min(m[i][j], m[i][k] + m[k][j]);
                }
            }
        }
        
        for(int i = 0; i < m.size(); i++) {
            for(int j = 0; j < m.size(); j++) {
                if(b[i][j] and m[i][j] < INT_MAX) return false;
            }
        }
        
        return true;
    }
};
