class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        vector<int> item(26);
        vector<bool> mark(26, 0);
    
        for(int i = 0; i < 26; i++){
            item[i]=i;
        }
        
        for(int i = 0; i < equations.size(); i++){
            if(equations[i][1]=='='){
                int a=equations[i][0]-'a';
                int b=equations[i][3]-'a';
                
                if(a != b){
                    if(mark[b]==0){
                        mark[b]=1;
                        if(mark[a]==1)
                            item[b]=item[a];
                        else{
                            item[b]=a;
                            mark[a]=1;
                        }
                    }
                    else{
                        if(mark[a]==0){
                            mark[a]=1;
                            item[a]=item[b];
                        }
                        else{
                            item[b]=item[a];
                            for(int i = 0; i < 26; i++){
                                if(item[i]==b){
                                    item[i]=item[a];
                                }
                            }
                        }
                    }
                }
            }
        }
        for(int i = 0; i < equations.size(); i++){
            if(equations[i][1]=='!'){
                int a=equations[i][0]-'a';
                int b=equations[i][3]-'a';
                
                if(item[a]==item[b])
                    return false;
            }
        }
        return true;
    }
};
