class Solution {
public:
    
    
    int& getParent(int parents[], int from){
        while(parents[from] != from){
            from = parents[from];
        }
        return parents[from];
    }
    
    bool equationsPossible(vector<string>& equations) {
        // create 26 groups
        // for every equal, merge the group and assign the master be the lowest one
        // one we have non-equal, check conflict
        int parents[26];
        for(int i = 0;i<26;i++){
            parents[i] = i; // parent itself
        }
        // process all equals
        for(auto& eq : equations){
            int first = eq[0] - 'a';
            int last = eq[3] - 'a';
            if(first > last) swap(first, last);
            // first always <= last
            if(eq[1] == '='){
                // equal
                getParent(parents, last) = getParent(parents,first);
            }
            else{
                // non-equal
                // skip now
            }
        }
        // process all non-equals
        for(auto& eq : equations){
            int first = eq[0] - 'a';
            int last = eq[3] - 'a';
            if(first > last) swap(first, last);
            // first always <= last
            if(eq[1] == '!'){
                // two parent must be different
                if(getParent(parents, last) == getParent(parents,first)){
                    return false;
                }
            }
        }
        // no conflict during check
        return true;
    }
};
