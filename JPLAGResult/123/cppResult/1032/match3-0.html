class Solution {
public:
	bool equationsPossible(vector<string>& equations) {
		for (auto equation :equations)
		{
			insertall(equation[0], equation[0]);
			if (equation[1] == '=')
			{
				insertall(equation[0],equation[3]);
				insertall(equation[3], equation[0]);
			}
		}
		for (auto equation : equations)
		{
			if (equation[1] == '!')
			{
				if (equationchar[equation[0]].end() != equationchar[equation[0]].find(equation[3])|| equationchar[equation[3]].end() != equationchar[equation[3]].find(equation[0]))
					return false;
			}
		}
		return true;
	}
private:
	map<char, set<char>> equationchar;
	void insertall(char a,char b) {
		equationchar[a].insert(a);
		equationchar[a].insert(b);
		for (auto temp:equationchar[b])
		{
			equationchar[a].insert(temp);
		}
	}
};
