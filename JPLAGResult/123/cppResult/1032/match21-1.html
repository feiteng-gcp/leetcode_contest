class Solution {
public:
    int root(int mySet[],int index)  {
        while (mySet[index]!=-1) {
            index = mySet[index];
        }
        return index;
    }
    
    bool equationsPossible(vector<string>& equations) {
        int mySet[26]={0};
        for (int i=0;i<26;i++) mySet[i] = -1;
        int L = equations.size();
        bool ret = true;
        for (int i=0;i<L;i++) {
            if (equations[i][1]=='='&&equations[i][0]!=equations[i][3]) {
                char x=equations[i][0],y=equations[i][3];
                if (root(mySet,x-'a')!=root(mySet,y-'a'))
                    mySet[root(mySet,x-'a')] = root(mySet,y-'a');
            }
        }
        
        for (int i=0;i<L;i++) {
            if (equations[i][1]=='!') {
                char x=equations[i][0],y=equations[i][3];
                if (root(mySet,x-'a')==root(mySet,y-'a')&&root(mySet,x-'a')!=-1) {
                    ret = false ;
                    break;
                }
            }
        }
        
        return ret;
    }
};
