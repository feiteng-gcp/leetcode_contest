class Solution {
public:
    int uf[26] ; 
    int find( int u ){
        if(  uf[u] == u ) 
            return u ; 
        return uf[u]  = find( uf[u] ) ; 
    }
    void Union( int u , int v ){

        int fu = find( u ) ; 
        int fv = find( v ) ; 
        if( fu == fv ) 
            return  ;
        uf[fv] = fu ; 
    }
    bool equationsPossible(vector<string>& equations) {
        memset( uf, 0 , sizeof( uf ) ) ; 
        for( int i=0 ; i< 26 ; i++)
            uf[i] = i ; 
        for( auto &x : equations ){
            int u = (x[0]-'a' ); 
            int v = (x[3]-'a' ); 
            if( x[1] == '=' ) {
                Union( u , v ) ; 
            }
        }
        for( auto &x : equations ){
            int u = x[0]-'a' ; 
            int v = x[3]-'a' ; 
            if( x[1] == '!' ) {
                int fu = find( u )  ;
                int fv = find( v ) ; 
                if( fu == fv ) 
                    return false ; 
            }
        }
        return true ;
    }
};
