class Solution {
public:
    bool equationsPossible(vector<string>& equations) {
        vector<vector<int>> vv(26,vector<int>(26,0));
         for(int i = 0;i <vv.size();i++){
            vv[i][i]=1;
        }
        for(int i = 0 ;i < equations.size();i++){
            if(equations[i][1]=='='){
                vv[equations[i][0]-'a'][equations[i][3]-'a']=1;
                vv[equations[i][3]-'a'][equations[i][0]-'a']=1;
            }
        }
        for(int i = 0;i < vv.size();i++){
            for(int j = 0;j<vv.size();j++){
                for(int k = 0;k<vv.size();k++){
                    if(vv[i][j]==1 && vv[k][i]==1){
                        vv[k][j]=1;
                        vv[j][k]=1;
                    }
                }
            }
        }
        for(int i = 0 ;i < equations.size();i++){
            if(equations[i][1]=='!'){
                if(vv[equations[i][0]-'a'][equations[i][3]-'a']==1||vv[equations[i][3]-'a'][equations[i][0]-'a']==1){
                    return false;
                }
            }
        }
        return true;
                   
    }
};
