class Solution {
public:
    int subarraysWithKDistinct(vector<int>& A, int K) {
        int i = 0, j = -1, k = 0, ans = 0;
        int num[20007];
        memset(num, 0, sizeof num);
        while(j < int(A.size())) // [i, j]
        {
            while(k < K && j+1 < A.size())
            {
                if(num[A[++j]]++ == 0) k++;
            }
            if(k < K) break;
            int jj = j+1;
            while(jj < A.size() && num[A[jj]]) jj++;
            ans += jj-j;
            //cout << "i:" << i << " j:" << j <<" k:" << k << " ans:"  << ans << endl;///
            int ii = i;
            while(k >= K && ii <= j)
            {
                if(--num[A[ii++]] == 0) k--;
            }
            ans += (ii-i-1)*(jj-j);
            i = ii;
            //cout << "i:" << i << " j:" << j <<" k:" << k << " ans:"  << ans << endl;///
        }
        return ans;
    }
};
