int numl[20001],numr[20001];
class Solution {
public:
    int subarraysWithKDistinct(vector<int>& A, int K) {
        int n=A.size();
	int cntl=0,cntr=0;
	int l=0,r=0,p=0;
	memset(numl,0,sizeof(numl));
	memset(numr,0,sizeof(numr));
	for(int i=0;i<n;i++,p++)
	{
		numl[A[i]]++;
		if(numl[A[i]]==1)
		{
			cntl++;
			if(cntl==K)
			{
				cntl--;
				numl[A[i]]--;
				break;
			}
		}
		numr[A[i]]++;
		if(numr[A[i]]==1)cntr++;
	}
	if(p==n)return 0;
	int ans=0;
	while(p<n)
	{
		numl[A[p]]++;
		if(numl[A[p]]==1)cntl++;
		numr[A[p]]++;
		if(numr[A[p]]==1)cntr++;
		p++;
		while(cntl>K)
		{
			numl[A[l]]--;
			if(numl[A[l]]==0)cntl--;
			l++;
		}
		while(cntr==K)
		{
			numr[A[r]]--;
			if(numr[A[r]]==0)cntr--;
			r++;
		}
		ans=ans+r-l;
	}
	return ans;
    }
};
