class Solution {
public:
    int subarraysWithKDistinct(vector<int>& A, int K) {
        map<int,int> m;
        int cnt = 0;
        int start = 0;
        for(int i = 0;i < A.size();i++){
            if(m.size()<K){
                m[A[i]]++;
                if(m.size()==K){
                    cnt++;
                    int k = start;
                    map<int, int> mcp(m);
                    while (mcp[A[k]]>1 && k<i) {
                        mcp[A[k]]--;
                        k++;
                        cnt++;
                    }
                }
            }else if(m.size()==K && m.find(A[i])!=m.end()){
                m[A[i]]++;
                cnt++;
                int k = start;
                map<int, int> mcp(m);
                while (mcp[A[k]]>1 && k<i) {
                    mcp[A[k]]--;
                    k++;
                    cnt++;
                }
            }else if(m.size()==K && m.find(A[i])==m.end()){
                while (m[A[start]]>=1  && m.size()>=K) {
                    m[A[start]]--;
                    if(m[A[start]]==0){
                        m.erase(A[start]);
                    }
                    start++;
                }
                m[A[i]]++;
                cnt++;
                int k = start;
                map<int, int> mcp(m);
                while (mcp[A[k]]>1 && k<i) {
                    mcp[A[k]]--;
                    k++;
                    cnt++;
                }
            }
        }
        return cnt;
    }
};
