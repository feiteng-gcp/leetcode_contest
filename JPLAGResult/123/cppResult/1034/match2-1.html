class Solution {
public:
    int subarraysWithKDistinct(vector<int>& A, int K) {
        int n = A.size();
        int i_lo = 0, i_hi = 0, sum_lo = 0, sum_hi = 0, ans = 0;
        unordered_map<int, int> freq_lo;
        unordered_map<int, int> freq_hi;
        for (int j = 0; j < A.size(); j++) {
            freq_lo[A[j]]++;
            sum_lo = freq_lo[A[j]] == 1 ? sum_lo + 1 : sum_lo;
            while (i_lo < j && sum_lo > K) {
                freq_lo[A[i_lo]]--;
                sum_lo = freq_lo[A[i_lo]] == 0 ? sum_lo - 1 : sum_lo;
                i_lo++;
            }
            
            freq_hi[A[j]]++;
            sum_hi = freq_hi[A[j]] == 1 ? sum_hi + 1 : sum_hi;
            while (i_hi < j && (sum_hi > K || (sum_hi == K && freq_hi[A[i_hi]] > 1))) {
                freq_hi[A[i_hi]]--;
                sum_hi = freq_hi[A[i_hi]] == 0 ? sum_hi - 1 : sum_hi;
                i_hi++;
            }
            
            if (sum_lo == K) {
                ans += (i_hi - i_lo + 1);
            }
        }
        return ans;
    }
};
