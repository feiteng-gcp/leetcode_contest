class Solution {
        map<int, int> mp;
        set<int> st;
    void add(int a) {
        if (st.find(a) == st.end()) {
            st.insert(a);
            mp[a] = 1;
        }
        else {
            mp[a]++;
        }
    }
    void rem(int a) {
        if(--mp[a] == 0) st.erase(a);
    }
    bool ex(int a) {
        return (st.find(a) != st.end());
    }
public:
    int subarraysWithKDistinct(vector<int>& A, int K) {
        int N = A.size();
        int l = 0, r = 0;
        int ret = 0;
        int tmpr;
        bool last = false;
        for (; r < N ||( l < r && st.size() == K) ;) {
            for (;st.size() < K && r < N; r++) {
                add(A[r]);
            }
            //cout << l << " " << r << " ";
            if (st.size() == K) {
                
                last = true;
                ret++;
                tmpr = r;
                for (;tmpr < N && ex(A[tmpr]); tmpr++) {
                    ret++;
                }
            } else {
                last = false;
            }
            //cout << ret << " " << endl;
            rem(A[l]); l++;
        }
        return ret;
        
    }
};
