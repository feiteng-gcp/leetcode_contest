class Solution {
public:
    int countOne(vector<int>& A) {
        int ans = 0;
        int i = 0;
        while (i < A.size()) {
            int j = i;
            while (j < A.size() -1 && A[j] == A[j + 1]) {
                j++;
            }
            if (j == i) ans++;
                else ans += (j - i + 2) * (j - i + 1) / 2;
            i = j + 1;
        }
        return ans;
    }
    
    int subarraysWithKDistinct(vector<int>& A, int K) {
        if (K == 1) return countOne(A);
        
        int next[20001];
        int last[20001] = {0};
        for (int i = A.size() - 1; i>=0; i--) {
            next[i] = last[A[i]];
            last[A[i]] = i;
        }

        

        int ans = 0;
        int count[20001] = {0};
        int d = 1;
        int l = 0;
        int r = 0;
        count[A[0]]++;
        while (true) {
            if (d < K) {
                r++;
                if (r == A.size()) break;
                if (count[A[r]]++ == 0) d++;
            } else if (d == K) {
                ans++;
                int nr = r + 1;
                while (nr < A.size() && count[A[nr]] > 0) {
                    count[A[nr]]++;
                    nr++;
                    ans++;
                }              
                int nl = l;
                int max = 0;
                while (nl < nr && count[A[nl]] > 1) {
                    count[A[nl]]--;
                    int n = next[nl];
                    if (n != 0) {
                        if (n < r) n = r;
                        if (n > max) max = n;
                        if (max < nr) {
                            ans += nr - max;
                           // printf("%d %d %d\n", nl, n, nr);
                        }
                    }
                    nl++;
                }
                
                if (nr < A.size()) {
                    count[A[nr]]++;
                    if (count[A[nr]] == 1) d++;
                } else break;
                r = nr;
                l = nl;
            } else {
                if (count[A[l]]-- == 1) d--;
                l++;
            }
        }
        return ans;
    }
};
