class Solution {
public:
    int subarraysWithKDistinct(vector<int>& A, int K) {
        map<int,int> mm;
        int N=A.size();
        vector<int> s(N,N);
        vector<int> e(N,-1);
        int p=0,q=0;
        while(p<A.size()){
           while(mm.size()<K&&q<N){
                mm[A[q]]++;
               q++;
           }
            
            if(mm.size()==K) s[p]=q-1;
            mm[A[p]]--;
            if(mm[A[p]]==0) mm.erase(A[p]);
            p++;
        }
        
        mm.clear();
        p=0;q=0;
        while(p<A.size()){
           while(mm.size()<=K&&q<N){
                mm[A[q]]++;
               q++;
           }
            
            if(mm.size()==K+1) e[p]=q-2;
            else if (q==N&&mm.size()==K) e[p]=q-1;
            mm[A[p]]--;
            if(mm[A[p]]==0) mm.erase(A[p]);
            p++;
        }
        
        int ans=0;
        for(int i=0;i<N;i++){
            //cout<<s[i]<<" "<<e[i]<<endl;
           ans+=max(0,e[i]-s[i]+1); 
        }
        return ans;
    }
};
