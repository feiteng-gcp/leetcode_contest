class Solution {
public:
    int cnt[20001];
    int subarraysWithKDistinct(vector<int>& A, int K) {
        for(int i=1;i<=20000;i++)
            cnt[i]=0;
        int r=0,tmp=0,ans=0;
        int n=(int) A.size();
        for(int i=0;i<n;i++){
            while(tmp<K&&r<n){
                if(cnt[A[r]]==0)
                    tmp++;   
                cnt[A[r]]++;
                r++;
            }
            if(tmp<K)    break;
            int now=r;
            while(tmp==K&&now<n){
                if(cnt[A[now]]==0){
                    break;   
                }
                now++;
            }
            ans+=now-r+1;
            cnt[A[i]]--;
            if(!cnt[A[i]])
                tmp--;
        }
        return ans;
    }
};
