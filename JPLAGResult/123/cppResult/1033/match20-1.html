class Solution {
public:
    int brokenCalc(int X, int Y) {
        if(X>=Y) return X-Y;
        int ope=0;
        while(Y>X){
            if(Y%2==1){
                Y=Y+1;
                ope++;
            }
            else{
                ope++;
                Y=Y/2;
            }
        }
        return X-Y+ope;
    }
};
