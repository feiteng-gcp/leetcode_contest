class Solution {
public:
  int brokenCalc(int X, int Y) {
    if (X == 0) return 1000000000;
    if (X >= Y) return X - Y;
    if (Y % 2) return brokenCalc(X, Y + 1) + 1;
    if (X < Y && Y <= X*2) {
      int res1 = 2*X - Y + 1;
      int res2 = (Y % 2) ? brokenCalc(X, (Y + 1) / 2) + 2 : brokenCalc(X, Y / 2) + 1;
      return min(res1, res2);
    }
    return brokenCalc(X, Y / 2) + 1;
  }
};
