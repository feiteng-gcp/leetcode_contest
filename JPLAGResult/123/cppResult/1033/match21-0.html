class Solution {
public:
    int brokenCalc(int X, int Y) {
        
        int res = 0;
        
        while (X != Y) {
            if ((Y % 2 == 1) || (Y < X)) {
                Y++;
                res++;
            }
            else {
                Y /= 2;
                res++;
            }
        }
        
        return res;
        
    }
};
