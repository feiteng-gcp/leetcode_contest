class Solution {
public:
    
   
    
    int brokenCalc(int x, int y) {
     if(x>=y)return x-y;
        int r = 0;
      while(x<y){
        if(y%2==1){
            y++;
            r++;
        }else{
            y = y/2;
            r++;
        }
      }
        r += (x-y);
        
       return r; 
       
    }
};
