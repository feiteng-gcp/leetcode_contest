class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> res = new ArrayList<>();
        int id = A.length - 1;
        int c = 0;
        while(id >= 0 || K != 0) {
            int a = 0;
            int b = 0;
            b = K % 10;
            K /= 10;
            if(id >= 0) a = A[id--];
            a += b + c;
            res.add(a % 10);
            c = a / 10;
        }
        if(c != 0) res.add(c);
        Collections.reverse(res);
        return res;
    }
}
