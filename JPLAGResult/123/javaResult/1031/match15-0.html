class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        LinkedList<Integer> list = new LinkedList<>();
        String B = K + "";
        int i = A.length-1; int j = B.length()-1;
        int carry = 0;
        while (i>=0 || j>=0 || carry==1) {
            int a = (i >= 0) ? A[i] : 0;
            int b = (j >= 0) ? (int)(B.charAt(j)-'0') : 0;
            int sum = a + b + carry;
            if (sum >= 10) {
                sum -= 10;
                carry = 1;
            } else {
                carry = 0;
            }
            list.addFirst(sum);
            i--; j--;
        }
        return list;
    }
}
