class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> k = new ArrayList<>();
        while (K > 0) {
            k.add(K % 10);
            K /= 10;
        }
        
        
        List<Integer> res = new ArrayList<>();
        int i = A.length-1;
        int j = 0;
        int carry = 0;
        while (i >= 0 && j < k.size()) {
            int val = carry + A[i] + k.get(j);
            res.add(val % 10);
            carry = val / 10;
            i--;
            j++;
        }
        
        while (j < k.size()) 
{
            int val = carry + k.get(j);
            res.add(val % 10);
            carry = val / 10;
            j++;
        }
        while (i >= 0) {
            int val = carry + A[i];
            res.add(val % 10);
            carry = val / 10;
            i--;
        }
        if (carry > 0) res.add(carry);
        
        List<Integer> ans = new ArrayList<>();
        for (int x = res.size()-1; x >= 0; x--) {
            ans.add(res.get(x));
        }
        return ans;
    }
}
