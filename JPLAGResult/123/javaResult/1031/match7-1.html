class Solution {
    public List<Integer> addToArrayForm(int[] A, int k) {
        List<Integer> ans = new LinkedList<>();
        
        int carry = 0;
        for (int i=A.length-1; i>=0; i--) {
            int sum = carry + A[i] + k%10;
            ans.add(0, sum%10);
            carry = sum/10;
            k = k/10;
        }
        
        while(k > 0) {
            int sum = carry + k%10;
            ans.add(0, sum%10);
            carry = sum/10;
            k = k/10;
        }
        
        
while(carry > 0) {
            ans.add(0, carry%10);
            carry = carry/10;
        }
        return ans;
    }
}
