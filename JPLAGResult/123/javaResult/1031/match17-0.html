class Solution {
    public List<Integer> addToArrayForm(int[] a, int k) {
        int n = a.length;
        int index = n - 1;
        int next = 0;
        List<Integer> res = new ArrayList<>();
        while (index >= 0 || k > 0) {
            if (index >= 0) {
                next += a[index];
                index--;
            }

            if (k > 0) {
                next += k % 10;
                k /= 10;
            }

            res.add(next % 10);
            next /= 10;
        }
        
        if (next > 0) res.add(next);

        Collections.reverse(res);
        return res;
    }
}
