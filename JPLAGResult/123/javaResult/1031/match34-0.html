class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> list =  new ArrayList<>();
        int n = A.length - 1;
        int carry = 0;
        while(K > 0 && n>= 0){
           int sum = K % 10 + A[n] + carry;
           n--;
           K /= 10;
           carry = 0;
           if(sum >= 10){
             carry += 1;
             sum -= 10;
           }
           list.add(sum);
        }
      while(n >= 0){
        if(carry != 0){
           int sum = A[n] + carry;
           carry = 0;
          if(sum >= 10){
            carry = 1;
            sum -= 10;
          }
          list.add(sum);
        }
        else list.add(A[n]);
        n--;
      }
      while(K > 0){
        if(carry != 0){
          int sum = K % 10 + carry;
          carry = 0;
          if(sum >= 10){
            carry = 1;
            sum -= 10;
          }
          list.add(sum);
        }
        else list.add(K % 10);
        K /= 10;
      }
      if(carry != 0){
        list.add(1);
      }
      Collections.reverse(list);
      return list;
    }
}
