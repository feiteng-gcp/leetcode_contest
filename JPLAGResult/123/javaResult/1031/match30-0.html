class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        int[] kArray = convert(K);
        int carrier = 0;
        List<Integer> result = new ArrayList<>();
        for (int i = 0; i < Math.min(A.length, kArray.length); i++) {
            int aElement = A[A.length - 1 - i];
            int kElement = kArray[kArray.length - 1 - i];
            int intermediateResult = aElement + kElement + carrier;
            if (intermediateResult > 9) {
                carrier = 1;
                intermediateResult = intermediateResult % 10;
            } else {
                carrier = 0;
            }
            result.add(0, intermediateResult);
        }
        if (kArray.length > A.length) {
            for (int i = Math.min(A.length, kArray.length); i < kArray.length; i++) {
                int intermediateResult = kArray[kArray.length - 1 - i] + carrier;
                if (intermediateResult > 9) {
                carrier = 1;
                intermediateResult = intermediateResult % 10;
            } else {
                carrier = 0;
            }
                result.add(0, intermediateResult);
            }
        } else {
            for (int i = Math.min(A.length, kArray.length); i < A.length; i++) {
                int intermediateResult = A[A.length - 1 - i] + carrier;
                if (intermediateResult > 9) {
                carrier = 1;
                intermediateResult = intermediateResult % 10;
            } else {
                carrier = 0;
            }
                result.add(0, intermediateResult);
            }
        }
        if (carrier != 0) {
            result.add(0, carrier);
        }
        if (result.get(0) == 0 && result.size() != 1) {
            result.remove(0);
        }
        return result;
    }
    
    // convert K into array
    public int[] convert(int K) {
        List<Integer> list = new ArrayList<>();
        while (K != 0) {
            list.add(0, K%10);
            K = K / 10;
        }
        list.add(0, 0); // just add 0 at the front
        int[] result = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            result[i] = list.get(i);
        }
        return result;
    }
}
