class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> l = new ArrayList<>(); 
        while (K > 0) {
            l.add(K % 10);
            K /= 10; 
        }
        
        Collections.reverse(l);
        List<Integer> sum = new ArrayList<>(); 
        int i = A.length - 1;
        int j = l.size() - 1;
        int carry = 0; 
        while (i >= 0 && j >= 0) {
            int val = A[i--] + l.get(j--) + carry;
            sum.add(val % 10);
            carry = val / 10; 
            
        }
        while (i >= 0) {
            int val = A[i--] + carry; 
            sum.add(val % 10);
            carry = val / 10; 
            
        }
        while (j >= 0) {
            int val = l.get(j--) + carry; 
            sum.add(val % 10);
            carry = val / 10; 
        }
        if (carry == 1) {
            sum.add(1);
        }
        Collections.reverse(sum);
        return sum; 
    }
}
