class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        int c = 0;
        List<Integer> res = new ArrayList<>();
        int end = A.length - 1;
        if (A.length == 0) {
            return res;
        }
        while (end >= 0 && K != 0) {
            int sum = c + A[end] + K % 10;
            res.add(0, sum % 10);
            c = sum / 10;
            K /= 10;
            end--;
        }
        while (end >= 0) {
            int sum = c + A[end];
            res.add(0, sum % 10);
            c = sum / 10;
            end--;
        }
        while (K != 0) {
            int sum = c + K % 10;
            res.add(0, sum % 10);
            c = sum / 10;
            K /= 10;
        }
        if (c != 0) {
            res.add(0, c);
        }
        return res;
    }
}
