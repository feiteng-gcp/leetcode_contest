class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        LinkedList<Integer> ans = new LinkedList<>();
        int plus = 0;

        for(int i=A.length-1; i>=0; i--) {
            int num = A[i] + plus + K % 10;
            K /= 10;
            if(num >= 10) {
                plus = 1;
                
num = num % 10;
            } else {
                plus = 0;
            }
            ans.addFirst(num);
        }
        while(K > 0) {
            int num = plus + K % 10;
            K /= 10;
            if(num >= 10) {
                plus = 1;
                num = num % 10;
            
} else {
                plus = 0;
            }
            ans.addFirst(num);
        }
        if(plus == 1) {
            ans.addFirst(1);
        }
        return ans;
    }
}
