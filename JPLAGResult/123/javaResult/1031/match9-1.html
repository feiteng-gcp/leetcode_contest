class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> list = new ArrayList<>();
        
        int ia = A.length - 1;
        int sum = 0;
        while (ia >= 0 || K > 0 || sum > 0) {
            if (ia >= 0) {
                sum += A[ia];
                --ia;
            }
            if (K > 0) {
                sum += K % 10;
                K /= 10;
            }
            
            list.add(sum % 10);
            sum /= 10;
        }
        
        Collections.reverse(list);
        return list;
    }
}
