class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        // A = new int[]{9,9,9,9,9,9,9,9,9,9};
        // K = 1;
        LinkedList<Integer> list = new LinkedList<>();
        char[] arr = Integer.toString(K).toCharArray();
        int one = 0, i = A.length - 1, j = arr.length - 1;
        
while (i > -1 && j > -1) {
            int n = A[i--] + arr[j--] - '0' + one;
            list.addFirst(n % 10);
            one = n / 10;
        }
        while (i > -1) {
            int n = A[i--] + one;
            list.addFirst(n % 10);
            one = n / 10;
        }
        while (j > -1) {
            int n = arr[j--] - '0' + one;
            list.addFirst(n % 10);
            one = n / 10;
        }
        if (one == 1) list.addFirst(1);
        return list;
    }
}
