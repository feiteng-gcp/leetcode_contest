class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> ret = new LinkedList<>();
        int carryOver = 0;
        int sum = 0;
        for(int i = A.length -1; i >= 0 || K != 0; i--){
            sum = ((i>= 0)? A[i]:0) + carryOver + K % 10;
            carryOver = sum / 10;
            sum %= 10;
            ret.add(0,sum);
            K /= 10;
        }
        if(carryOver != 0)ret.add(0,carryOver);
        return ret;
    }
}
