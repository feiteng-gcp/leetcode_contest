class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> ans = new LinkedList<>();
        int aptr = A.length -1 ;
        String k = String.valueOf(K);
        int kptr = k.length() - 1;
        int carry = 0;
        
        while (aptr >= 0 && kptr >= 0){
            int aa = A[aptr];
            int kk = (int)(k.charAt(kptr) - '0');
            int sum = aa + kk + carry;
            
            ans.add(0,sum % 10);
            carry = sum / 10;
        
            aptr--;
            kptr--;
        }
        
        while (kptr >= 0){
            
            int kk = (int)(k.charAt(kptr) - '0');
            int sum = kk + carry;

            ans.add(0,sum % 10);
            carry = sum / 10;
        
            kptr--;
        }
        
        while (aptr >= 0 ){
            int aa = A[aptr];
            
            int sum = aa + carry;
            
            ans.add(0,sum % 10);
            carry = sum / 10;
        
            aptr--;
        
}
        if (carry > 0){
            ans.add(0,carry);
        }
        
        
        return ans;
    }
}
