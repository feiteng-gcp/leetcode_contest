class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        int i=A.length-1;
        int remain=0;
        List<Integer> res = new ArrayList<>();
        while(i>=0 && K >0) {
            int cur = remain + A[i] + K%10;
            res.add(0,cur%10);
            remain = cur/10;
            K /= 10;
            i--;
        }
        
        while(i>=0) {
            int cur = remain + A[i];
            res.add(0,cur%10);
            remain = cur/10;
            i--;
        }
        
        while(K>0) {
            int cur = remain + K%10;
            res.add(0,cur%10);
            remain = cur/10;
            K /= 10;
        }
        
        if(remain !=0) res.add(0, remain);
        return res;
    }
}
