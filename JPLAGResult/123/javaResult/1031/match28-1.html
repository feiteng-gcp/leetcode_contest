class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> result = new ArrayList<>();
        int sum = 0;
        for (int i = A.length - 1; i >= 0; i--) {
            sum += A[i] + K % 10;
            K /= 10;
            result.add(sum % 10);
            sum /= 10;
        }
        while (K > 0 || sum > 0) {
            sum += K % 10;
            K /= 10;
            result.add(sum % 10);
            sum /= 10;
        }
        Collections.reverse(result);
        return result;
    }
}
