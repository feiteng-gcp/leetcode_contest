class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        
        List<Integer> res = new ArrayList<>();
        int carry = 0;
        int idx = A.length - 1;
        while (K > 0 || idx >= 0) {
            int curr = (idx >= 0 ? A[idx] : 0 ) + (K%10) + carry;
            carry = curr/10;
            curr = curr % 10;
            
            K = K/10;
            idx--;
            res.add(curr);
        }
        if (carry != 0) res.add(carry);
        
        List<Integer> ret = new ArrayList<>();
        for (int i = res.size() - 1;i >= 0;i--) ret.add(res.get(i));
        return ret;
    }
}
