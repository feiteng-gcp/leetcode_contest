class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> res = new ArrayList<>();
        int len = A.length;
        
        int carry = 0;
        int idx = len - 1;
        while((K != 0 || carry != 0) && idx >= 0) {
            
int cur = K % 10;
            K /= 10;
            
            A[idx] += cur + carry;
            if(A[idx] >= 10) {
                A[idx] -= 10;
                carry = 1;
            } else {
                carry = 0;
            }
            
            idx--;
        }
        
        for(int i = 0; i < len; i++) {
            res.add(A[i]);
        }
        
        while(K != 0) {
            int num = K % 10 + carry;
            K /= 10;
            
            
            if(num >= 10) {
                res.add(0, num - 10);
                carry = 1;
            } else {
                res.add(0, num);
                carry = 0;
            }
        }
        if(carry != 0) {
            res.add(0, carry);
        }
        
        return res;
    }
}
