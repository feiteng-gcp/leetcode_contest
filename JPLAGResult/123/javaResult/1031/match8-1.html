class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> res = new ArrayList<>();
        int len = A.length, c = 0, i = len - 1;
        while(i >= 0 || K > 0) {
            int a = i >= 0 ? A[i] : 0;
            int b = K > 0 ? K%10 : 0;
            int sum = a + b + c;
            res.add(0,sum%10);
            c = sum/10;
            K /= 10;
            i--;
        
}
        if (c == 1) res.add(0,1);
        return res;
    }
}
