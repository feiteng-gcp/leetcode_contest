class Solution {
    public List<Integer> addToArrayForm(int[] A, int k) {
        List<Integer> list = new ArrayList<Integer>();
        int index1 = 0, carry = 0;
        while ((index1 < A.length && k != 0) || carry != 0) {
            int l = index1 < A.length ? A[A.length - 1 - index1] : 0;
            index1++;
            int r = k != 0 ? k % 10 : 0;
            k /= 10;
            int value = l + r + carry;
            list.add(value % 10);
            carry = value / 10;
        }
        while (index1 < A.length) {
            list.add(A[A.length - 1 - (index1++)]);
        }
        while (k != 0) {
            list.add(k % 10);
            k /= 10;
        }
        Collections.reverse(list);
        return list;
    }
}
