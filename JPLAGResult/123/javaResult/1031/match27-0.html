class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> res = new ArrayList<>();
        int N = A.length;
        int c = 0;
        for (int i = N - 1; i >= 0; i--) {
            c = c + A[i] + (K % 10);
            res.add(0, c % 10);
            if (c >= 10) {
                c = 1;
            } else {
                c = 0;
            }
            K /= 10;
        }
        while (c > 0 || K > 0) {
            c = c + (K % 10);
            res.add(0, c % 10);
            if (c >= 10) {
                c = 1;
            } else {
                c = 0;
            }
            K /= 10;
        }
        return res;        
    }
}
