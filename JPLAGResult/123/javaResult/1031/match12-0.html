class Solution {
    public List<Integer> addToArrayForm(int[] A, int k) {
        int carry = 0;
        int n = A.length;
        int i = n-1;
        List<Integer> list = new ArrayList<>();
        while (k > 0 || i >= 0) {
            int t = carry;
            if (k > 0) {
                t += k%10;
                k /= 10;
            }
            if (i >= 0) {
                t += A[i];
                i--;
            }
            if (t >= 10) {
                carry = 1;
                t -= 10;
            } else {
                carry = 0;
            }
            list.add(t);
        }
        if (carry > 0) {
            list.add(1);
        }
        Collections.reverse(list);
        return list;
    }
}
