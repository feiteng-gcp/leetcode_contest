class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> result = new ArrayList<>();
        int i = 0, len = A.length;
        int flag = 10;
        int m = 0;
        
while (i < len || K > 0 || m != 0) {
            if (i < len) {
                m += A[len - 1 - i];
                i++;
            }
            if (K > 0) {
                m += K % flag;
                K /= 10;
            }
            result.add(0, m % 10);
            m /= 10;
        }
        return result;
    }
}
