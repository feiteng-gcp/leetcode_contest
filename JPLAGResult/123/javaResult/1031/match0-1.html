class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        int lengthA = A.length;
        int lengthK = String.valueOf(K).length();
        int sumLength = Math.max(lengthA, lengthK) + 1;
        int[] sumArray = new int[sumLength];
        int indexA = A.length - 1;
        int remainK = K;
        int indexSum = sumLength - 1;
        sumArray[indexSum] = A[indexA] + K;
        indexA--;
        indexSum--;
        while (indexA >= 0) {
            sumArray[indexSum] = A[indexA];
            indexA--;
            indexSum--;
        }
        for (int i = sumLength - 1; i >= 0; i--) {
            if (sumArray[i] >= 10) {
                sumArray[i - 1] += sumArray[i] / 10;
                sumArray[i] %= 10;
            }
        }
        List<Integer> ret = new ArrayList<Integer>();
        int startIndex = 0;
        for (int i = 0; i < sumLength; i++) {
            if (sumArray[i] != 0) {
                startIndex = i;
                break;
            }
        }
        for (int i = 0; i < sumLength; i++)
            ret.add(sumArray[i]);
        int size = ret.size();
        for (int i = 0; i < size - 1; i++) {
            if (ret.get(i) == 0) {
                ret.remove(i);
                i--;
                size--;
            } else
                break;
        }
        return ret;
    }
}
