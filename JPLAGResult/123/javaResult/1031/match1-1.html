class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> result = new LinkedList<>();
        
        int memo = 0;
        for (int i = A.length - 1; i >= 0; i--) {
            A[i] += K % 10 + memo;
            K /= 10;
            memo = A[i] / 10;
            A[i] = A[i] % 10;
            result.add(A[i]);
        
}
        
        while (K != 0) {
            int x = (K % 10 + memo);
            result.add(x % 10);
            memo = x / 10;
            K /= 10;
        }
        
        if (memo != 0) {
            result.add(memo);
        }
        
        Collections.reverse(result);        
        return result;
    }
}
