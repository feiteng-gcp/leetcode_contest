import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    public List<Integer> addToArrayForm(int[] A, int K) {
        StringBuilder builder = new StringBuilder();
        for (int i : A) {
            builder.append(i);
        }
        BigInteger sum = new BigInteger(builder.toString()).add(BigInteger.valueOf(K));
        String str = sum.toString();
        List<Integer> res = new ArrayList<>(str.length());
        for (int i=0; i<str.length(); i++) {
            res.add(str.charAt(i) - '0');
        }
        return res;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(sol.addToArrayForm(new int[]{1, 2, 0, 0}, 34));
        System.out.println(sol.addToArrayForm(new int[]{2, 7, 4}, 181));


    }
}
