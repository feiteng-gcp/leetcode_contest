class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        int len = A.length;
        List<Integer> list = new ArrayList<Integer>();
        int curr = len - 1;
        int carry = 0;
        while(curr >= 0 && K > 0){
            int rem = K%10;
            K /= 10;
            
int temp = A[curr] + rem + carry;
            list.add(temp%10);
            carry = temp/10;
            curr--;
        }
        
        while(curr >= 0){
            int temp = A[curr] + carry;
            list.add(temp%10);
            carry = temp/10;
            curr--;
        }
        
        while(K > 0){
            int rem = K%10;
            K /= 10;
            int temp = rem + carry;
            
list.add(temp%10);
            carry = temp/10;
        }
        
        if(carry > 0){
            list.add(carry);
        }
        
        Collections.reverse(list);
        return list;
    }
}
