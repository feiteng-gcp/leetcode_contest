class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> res = new ArrayList<>();
        int carry = 0;
        int i = A.length - 1;
        while (i >= 0 || K > 0) {
            int n = i >= 0 ? A[i--] : 0;
            int k = K % 10;
            int num = (n + k + carry) % 10;
            carry = (n + k + carry) / 10;
            res.add(0, num);
            K /= 10;
        }
        if (carry > 0) {
            res.add(0, carry);
        }
        return res;
    }
}
