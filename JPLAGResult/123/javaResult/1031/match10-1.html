class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        
        int add = 0;
        List<Integer> res = new ArrayList();
        
        for (int i = A.length - 1; i >= 0; i--) {
            int val = A[i] + (K % 10) + add;
            K /= 10;
            
            add = val / 10;
            res.add(val % 10);    
        }
        
        while (K > 0) {
            int val = (K % 10) + add;
            K /= 10;
            
            add = val / 10;
            
res.add(val % 10);
        }
        
        if (add > 0) {
            res.add(add);
        }
        
        Collections.reverse(res);
        return res;
    }
}
