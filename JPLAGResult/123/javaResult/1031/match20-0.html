class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> result = new ArrayList<>();
        // if(K == 0) {
        //     return Arrays.stream(A).boxed().collect(Collectors.toList());;
        // }
        List<Integer> B = new ArrayList<>();

        while(K > 0) {
            B.add(0, K % 10);
            K = K / 10;
        }
        if(A == null || A.length == 0) {
            return B;
        }
        int l1 = A.length;
        int l2 = B.size();
        int i = l1 - 1;
        int j = l2 - 1;
        int carry = 0;
        while(i >= 0 && j >= 0) {
            int cur = A[i--] + B.get(j--) + carry;
            if(cur >= 10) {
                carry = 1;
                cur = cur % 10;
            } else {
                carry = 0;
            }
            result.add(0, cur);
        }
        while(i >= 0) {
            int cur = A[i--] + carry;
            if(cur >= 10) {
                carry = 1;
                cur = cur % 10;
            } else {
                carry = 0;
            }
            result.add(0, cur);
        }
        while(j >= 0) {
            int cur = B.get(j--) + carry;
            if(cur >= 10) {
                carry = 1;
                cur = cur % 10;
            } else {
                carry = 0;
            }
            result.add(0, cur);
        }
        if(carry > 0) {
            result.add(0, carry);
        }
        return result;
    }
}
