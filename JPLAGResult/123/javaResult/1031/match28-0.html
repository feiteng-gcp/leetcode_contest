class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        List<Integer> result = new ArrayList<>();
        int res = K;
        for (int i=A.length - 1;i>=0;i--) {
            res = A[i] + res;
            result.add(0, res % 10);
            res = res/10;
        }
        
        while (res > 0) {
            result.add(0, res%10);
            res /= 10;
        }
        
        return result;
    }
}
