class Solution {
public List<Integer> addToArrayForm(int[] A, int K) {
		int add = 0;
		List<Integer> res = new ArrayList<>();
		for (int i = A.length - 1; i >= 0; i--) {
			int sum = add + A[i] + (K % 10);
			K /= 10;
			if (sum >= 10) {
				sum -= 10;
				add = 1;
			} else {
				add = 0;
			}
			res.add(sum);
		}
		while (K > 0) {
			int sum = add + (K % 10);
			K /= 10;
			if (sum >= 10) {

				sum -= 10;
				add = 1;
			} else {
				add = 0;
			}
			res.add(sum);
		}
		if (add == 1) {
			res.add(1);
		}
		Collections.reverse(res);
		return res;
    }
}
