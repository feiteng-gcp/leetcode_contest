class Solution {
    public List<Integer> addToArrayForm(int[] A, int K) {
        LinkedList<Integer> res = new LinkedList();
        int i = A.length - 1, c = 0;
        while (i >= 0 || K > 0) {
            if (i >= 0) c += A[i--];
            if (K > 0) {
                c += K % 10;
                K /= 10;
            }
            res.addFirst(c % 10);
            c /= 10;
        }
        if (c > 0) res.addFirst(c);
        return res;
    }
}
