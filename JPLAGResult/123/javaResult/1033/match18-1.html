import java.util.ArrayDeque;
import java.util.HashMap;
import java.util.Map;
import java.util.Queue;


public class Solution {

    public int brokenCalc(int X, int Y) {
        if (X >= Y) {
            return X - Y;
        }
        return dfs(X, Y);
    }

    int dfs(int x, int y) {
        
if (x >= y) {
            return x - y;
        }

        if (y % 2 == 1) {
            return dfs(x, y + 1) + 1;
        } else {
            return dfs(x, y / 2) + 1;
        }
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(sol.brokenCalc(2, 3));
        System.out.println(sol.brokenCalc(5, 8));
        System.out.println(sol.brokenCalc(3, 10));
        System.out.println(sol.brokenCalc(1024, 1));
        System.out.println(sol.brokenCalc(1, 123434345));
    }
}
