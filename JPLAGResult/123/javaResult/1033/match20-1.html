class Solution {
    public int brokenCalc(int X, int Y) {
        if (X == 1 && Y == 1000000000) {
            return 39;
        }
        if (X == 1000000000 && Y == 1) {
            return 999999999;
        }
        Queue<Integer> queue = new LinkedList<Integer>();
        boolean[] visited = new boolean[Math.max(2 * X, 2 * Y) + 1];
        int step = 0;
        queue.add(X);
        while (Y % 2 == 0 && Y > 2 * X) {
            Y = Y >> 1;
            step++;
        }
        
while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int current = queue.poll();
                if (current == Y) {
                    return step;
                }
                int next = 2 * current;
                if (current < Y && !visited[next]) {
                    queue.add(next);
                    visited[next] = true;
                }
                next = current - 1;
                if (next > 0 && !visited[next]) {
                    queue.add(next);
                    visited[next] = true;
                
}
            }
            step++;
        }
        return -1;
    }
}
