class Solution {
    public int brokenCalc(int X, int Y) {
        if (X >= Y)
            return X - Y;
        int count = 0;
        while (Y >= X * 2) {
        	if (Y % 2 == 1) {
        		Y++;
        		count++;
        	}
        	Y /= 2;
        	count++;
        }
        int[] dp = new int[Y + 2];
        for (int i = 0; i < Y + 2; i++)
        	dp[i] = -1;
        dp[X] = 0;
        for (int i = 1; i < X; i++)
            dp[i] = X - i;
        int lowerBound = Math.max(1, X / 2 + 1);
        int upperBound = (Y + 1) / 2;
        for (int i = lowerBound; i <= upperBound; i++) {
            int doubleIndex = i * 2;
            if (dp[doubleIndex] == -1)
            	dp[doubleIndex] = dp[i] + 1;
            if (dp[doubleIndex - 1] == -1)
            	dp[doubleIndex - 1] = dp[doubleIndex] + 1;
        }
        return dp[Y] + count;
    }
}
