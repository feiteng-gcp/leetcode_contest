class Solution {
	public int brokenCalc(int X, int Y) {
		Set<Integer> seen = new HashSet<>();
		int steps = 0;

		Queue<Integer> queue = new LinkedList<>();

//		if (Y == X) {
//			return 0;
//		}

		queue.offer(Y);
		while (!queue.isEmpty()) {
			int size = queue.size();
			for (int i = 0; i < size; i++) {
				int num = queue.poll();
				if (num <= X) {
					return steps + (X - num);
				}
				if (num % 2 != 0 ) {
						seen.add(num + 1);
						queue.offer(num + 1);
				}
				else {
						seen.add(num / 2);
						queue.offer(num / 2);

				}
			}
            			steps++;

		}

		return -1;

	}
}
