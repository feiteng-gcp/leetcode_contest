class Solution {
    public int brokenCalc(int x, int y) {
        if (x > y) {
            return x - y;
        }
        if (x == y) {
            return 0;
        }
        int level = 0;
        Queue<Integer> q = new LinkedList<>();
        q.add(y);
        while (!q.isEmpty()) {
            int size = q.size();
            level += 1;
            //System.out.println("level = " + level);
            for (int i = 0; i < size; i++) {
                int t = q.poll();
                if (x > t) {
                    return level -1 + x - t;
                }
                //System.out.println("t = " + t);
                if (t == x) {
                    return level-1;
                }
                if (t % 2 == 0) {
                    q.add(t >> 1);
                }
                if (x * 10 > t || (t%2 == 1)) {
                    q.add(t+1);
                }
            }
        }
        return -1;
    }
}
