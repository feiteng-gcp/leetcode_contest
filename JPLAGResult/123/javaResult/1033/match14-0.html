class Solution {
    public int brokenCalc(int a, int b) {
        if (a >= b) return a - b;
        int count = 0;
        while (a < b) {
            if (b % 2 == 0) {
                b /= 2;
                count++;
            }else {
                b += 1;
                b/=2;
                count += 2;
            }
        }
        count += a - b;
        return count;
    }
}
