class Solution {
    public int brokenCalc(int X, int Y) {
        return get(X, Y);
    }
    
    private int get(int x, int y) {
        if (x >= y) {
            return x - y;
        }
        if (y % 2 == 0) {
            return get(x, y / 2) + 1;
        }
        return get(x, y + 1) + 1;
    }
}
