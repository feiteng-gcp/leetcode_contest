class Solution {
    public int brokenCalc(int X, int Y) {
        int mul=0,pro=1;
        while (X<Y)
        {
            mul++;
            X<<=1;
            pro<<=1;
        }
        int minus=X-Y,ans=mul;
        while (pro>0)
        {
            ans+=(minus/pro);
            minus%=pro;
            pro>>=1;
        }
        return ans;
    }
}
