class Solution {
    
    public int brokenCalc(int X, int Y) {
        if(X >= Y){
            return (X-Y);
        }
        int x =  X;
        int y =  Y;
        
        int mid = (Y+1)/2;
        int midAns = (mid*2 == Y ? 1 : 2);
        if(x == mid){
            return midAns; 
        }else if(x > mid){
            return ((x-mid) + midAns);
        }else{
            return (midAns + brokenCalc(X, mid));
        }
    }
}
