class Solution {
    public int brokenCalc(int X, int Y) {
        int step = 0;
        int flag = 1;
        while (X < Y) {
            X <<= 1;
            flag <<= 1;
            step += 1;
        }
        int d = X - Y;
        while (d > 0) {
            step += d / flag;
            d %= flag;
            flag >>= 1;
        }
        return step;
    }
}
