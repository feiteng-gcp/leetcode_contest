class Solution {

	private int findSet(int x) {
		if (x == set[x]) {
			return x;
		} else {
			return set[x] = findSet(set[x]);
		}
	}
	
	private int[] set = new int[100];
	
    public boolean equationsPossible(String[] equations) {
     
    	int i;
    	for (i = 0;i < 100;i ++) {

    		set[i] = i;
    	}
    	for (String string : equations) {    		
    		if (string.contains("==")) {
    			int index1 = string.charAt(0) - 'a';
    			int index2 = string.charAt(3) - 'a';
    			int f1 = findSet(index1) , f2 = findSet(index2);
    			if (f1 != f2) {
    				set[f1] = f2;    				
    			}
    		}
    	}

    	for (String string : equations) {
    		if (string.contains("!=")) {
    			int index1 = string.charAt(0) - 'a';
    			int index2 = string.charAt(3) - 'a';
    			int f1 = findSet(index1) , f2 = findSet(index2);
    			if (f1 == f2) {
    				return false;    				
    			}
    		}
    	}
    	return true;
    	
    }
}
