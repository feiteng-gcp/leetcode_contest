class Solution {
    static class UF {
        Map<Character, Character> fathers = new HashMap<>();
        char findFather(char c) {
            fathers.putIfAbsent(c, c);
            Character cf = fathers.get(c);
            if (cf == c) {
                return c;
            }
            cf = findFather(cf);
            fathers.put(c, cf);
            return cf;
        }
        boolean connect(char a, char b) {
            char af = findFather(a);
            char bf = findFather(b);
            if (af != bf) {
                fathers.put(bf, af);
                return true;
            }
            return false;
        }
        boolean isconnect(char a, char b) {
            char af = findFather(a);
            char bf = findFather(b);
            return af == bf;
        }
    }
    public boolean equationsPossible(String[] equations) {
        UF myuf = new UF();
        for (String s : equations) {
            char left = s.charAt(0);
            char right = s.charAt(3);
            if (s.charAt(1) == '=') {
                myuf.connect(left, right);
            } 
        }
        
        for (String s : equations) {
            char left = s.charAt(0);
            char right = s.charAt(3);
            if (s.charAt(1) == '!') {
                if (myuf.isconnect(left, right)) {
                    return false;
                }
            } 
        }
        
        return true;
    }
}
