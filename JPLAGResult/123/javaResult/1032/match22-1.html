class Solution {
	static int[] union_find = new int[26];

	public static boolean equationsPossible(String[] equations) {
		Arrays.fill(union_find, -1);
		for (String eq
uation : equations) {
			if (equation.charAt(1) == '=') {
				int a = equation.charAt(0) - 'a', b = equation.charAt(3) - 'a';
				
				int parentA = find_parent(a), parentB = find_parent(b);
				if (parentA != parentB)
					union_find[find_parent(b)] = find_parent(a);
			}

		}

		for (String equation : equations) {
			if (equation.charAt(1) == '!') {
				int a = equation.charAt(0) - 'a', b = equation.charAt(3) - 'a';
				if (find_parent(a) == find_parent(b))
					return false;
			}
		}

		return true;
	}

	private static int find_parent(int index) {
		int parent = index;
		while (parent != -1) {
			index = parent;
			parent = union_find[index];
		}
		return index;
	}
}
