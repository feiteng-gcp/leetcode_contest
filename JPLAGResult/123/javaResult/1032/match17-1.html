class Solution {
    int[] id;
    public boolean equationsPossible(String[] equations) {
        id = new int[26];
        for (int i = 0; i < 26; i++) {
            id[i] = i;
        }
        
for (String str : equations) {
            int first = (int) (str.charAt(0) - 'a');
            int second = (int) (str.charAt(3) - 'a');
            if (str.charAt(1) == '=') {
                union(first, second);
            }
        }
        
        for (String str : equations) {
            int first = (int) (str.charAt(0) - 'a');
            int second = (int) (str.charAt(3) - 'a');
            if (str.charAt(1) == '!') {
                if (root(first) == root(second)) return false;
            }
        }
        
        return true;
    }
    
    private int root(int i) {
        while (id[i] != i) {
            id[i] = id[id[i]];
            i = id[i];
        }
        
return i;
    }
    
    private void union(int p, int q) {
        int rootP = root(p);
        int rootQ = root(q);
        if (rootP == rootQ) return;
        id[rootP] = rootQ;
    }
}
