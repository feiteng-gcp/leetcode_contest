class Solution {
    int[] f = new int[26]; // a->a
    public boolean equationsPossible(String[] eq) {
        int n = eq.length;
        for (int i = 0; i < 26; i++) f[i] = i;
        for (String s : eq) {
            if (s.charAt(1) == '=') {
                int x = find(s.charAt(0) - 'a');
                int y = find(s.charAt(3) - 'a');
                if (f[x] != y) {
                    f[x] = y;
                }
            }
        }
        for (String s: eq) {
            if (s.charAt(1) == '!') {
                int x = find(s.charAt(0) - 'a');
                int y = find(s.charAt(3) - 'a');
                if (x == y) {
                    return false;
                }
            }
        }
        return true;
    }
    
    public int find(int x) {
        if (f[x] != f[f[x]]) {
            f[x] = find(f[x]);
        }
        return f[x];
    }
}
