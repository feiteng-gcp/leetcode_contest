class Solution {
    public boolean equationsPossible(String[] E) {
        int[] vars = new int[26];
        for (int i = 0; i < vars.length; i++) {
            vars[i] = i;
        }
        for (String e : E) {
            int a = e.charAt(0) - 'a';
            int b = e.charAt(3) - 'a';
            if (e.charAt(1) == '=' && a != b) {
                int p = vars[a];
                for (int i = 0; i < vars.length; i++)
                    if (vars[i] == p)
                        vars[i] = vars[b];
            }
        }
        System.out.println(Arrays.toString(vars));
        for (String e : E) {
            int a = e.charAt(0) - 'a';
            int b = e.charAt(3) - 'a';
            if (e.charAt(1) == '!' && vars[a] == vars[b])
                return false;
        }
        return true;
    }
}
