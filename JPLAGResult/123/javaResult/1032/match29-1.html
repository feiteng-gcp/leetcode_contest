class Solution {
	int[] fa = new int[26];
	
	public boolean equationsPossible(String[] equations) {
		
		for (int i = 0; i < 26; i++) fa[i] = i;
		
		boolean res = true;
		for (int i = 0; i < equations.length; i ++) {
			String s = equations[i];
			int a = s.charAt(0) -
 'a';
			int b = s.charAt(3) - 'a';
			if (s.charAt(1) == '=') {
				int af = find(a);
				int bf = find(b);
				fa[af] = bf;
			}
		}

		for (int i = 0; i < equations.length; i ++) {
			String s = equations[i];
			int a = s.charAt(0) -
 'a';
			int b = s.charAt(3) - 'a';
			if (s.charAt(1) == '!') {
				int af = find(a);
				int bf = find(b);
				if (af == bf) res = false;T color="#77bfc7">

			}
		}
		return res;
	}
	
	private int find(int a) {
		if (fa[a] == a) return a;
		return fa[a] = find(fa[a]);
	}
}
