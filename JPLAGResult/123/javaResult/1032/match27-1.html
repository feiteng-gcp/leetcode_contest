class Solution {
    public boolean equationsPossible(String[] equations) {
        int[] uf = new int[26]; 
        for (int i = 0; i < 26; i++) {
            uf[i] = i; 
        }
        for (int i = 0; i < equations.length; i++) {
            char a = equations[i].charAt(0);
            char b = equations[i].charAt(3); 
            if (equations[i].charAt(1) == '=') {
                int fi = find(uf, a - 'a');
                int fj = find(uf, b - 'a');
                if (fi != fj) {
                    uf[fi] = fj; 
                }
            }
        }
        for (int i = 0; i < equations.length; i++) {
            char a = equations[i].charAt(0);
            char b = equations[i].charAt(3); 
            if (equations[i].charAt(1) == '!') {
                
int fi = find(uf, a - 'a');
                int fj = find(uf, b - 'a');
                if (fi == fj) {
                    return false;  
                }
            }
        }
        return true; 
    }
    private int find(int[] uf, int i) {
        if (uf[i] == i) {
            return i;
        }
        uf[i] = find(uf, uf[i]);
        return uf[i]; 
    }
}
