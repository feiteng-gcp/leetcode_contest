class Solution {
	private static int[] parent, size;
	private static int N = 26;

	public boolean equationsPossible(String[] equations) {
		int curr = 1;
		parent = new int[N];
		size = new int[N];

		for (int i = 0; i < N; i++) {
			parent[i] = i;
		}
		for (int i = 0; i < N; i++) {
			size[i] = 1;
		}
		for (int i = 0; i < equations.length; i++) {
			String str = equations[i];
			int first = str.charA
t(0) - 'a';
			int second = str.charAt(3) - 'a';
			if (str.charAt(1) == '=') {
				join(first, second);
			}
		}
		for (int i = 0; i < equations.length; i++) {
			String str = equations[i];
			int first = str.charA
t(0) - 'a';
			int second = str.charAt(3) - 'a';
			if (str.charAt(1) == '!') {
				if (find(first) == find(second)) {
					return false;
				}
			}
		}
		return true;
	}

	public static int sizeOf(int target) {
		return size[find(target)];
	}

	public static int find(int target) {
		int parentIndex = target;
		while (parent[parentIndex] != parentIndex) {
			parentIndex = parent[parentIndex];
		}
		int i = target;
		while (i != parentIndex) {
			i = parent[i];
			parent[i] = parentIndex;
		}
		return parentIndex;
	}

	public static void join(int tar
get1, int target2) {
		int root1 = find(target1);
		int root2 = find(target2);
		if (root1 != root2) {
			parent[root1] = root2;
			size[root2] += size[root1];
		}
	}
}
