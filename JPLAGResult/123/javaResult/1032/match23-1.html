class Solution {
    public boolean equationsPossible(String[] equations) {
        int[] record = 
new int[26];
        for (int i=0; i<26; i++) {
            record[i] = i;
        }
        for (String equation: equations){
            if (equation.charAt(1) == '=') {
                int left = equation.charAt(0)-'a';
                int right = 
equation.charAt(3)-'a';
                merge(left,right,record);
            }
        }
        /*for (int i=0; i<26; i++) {
            record[i] = root(i,record);
            System.out.print(record[i]+"  ");
        }*/
        for (String equation: equations){
            if (equation.charAt(1) == '!') {
                int left = equation.charAt(0)-'a';
                int right = equation.charAt(3)-'a';
                if (root(left,record) == root(right,record)) return false
;
            }
        }
        return true;
    }
    int root(int num,int[] record) {
        while (record[num] != num){
            int head = record[num];
            record[num] = record[head];
            num = head;
        }
        return num;
    }
    void merge(int left, int right, int[] record){
        int left_root = root(left,record);
        int right_root = root(right,record);
        record[left_root] = right_root;
    }
}
