class Solution {
    int[] root;
    public boolean equationsPossible(String[] equations) {
        int id = 0;
        HashMap<Character, Integer> map = new HashMap<>();
        for(String x : equations) {
            char a = x.charAt(0);
            char b = x.charAt(3);
            if(!map.containsKey(a)) map.put(a, id++);
            if(!map.containsKey(b)) map.put(b, id++);
        }
        root = new int[id];
        
for(int i = 0; i < id; ++i) root[i] = i;
        for(String x : equations) {
            char a = x.charAt(0);
            char b = x.charAt(3);
            if(x.charAt(1) == '=') {
                union(map.get(a), map.get(b));
            
}
        }
        for(String x : equations) {
            char a = x.charAt(0);
            char b = x.charAt(3);
            if(x.charAt(1) == '!') {
                if(find(map.get(a)) == find(map.get(b))) return false;
            }
        }
        return true;
    }
    void union(int a, int b) {
        int ra = find(a);
        int rb = find(b);
        if(ra != rb) root[ra] = rb;
    
}
    int find(int x) {
        while(x != root[x]) {
            root[x] = root[root[x]];
            x = root[x];
        }
        return x;
    }
}
