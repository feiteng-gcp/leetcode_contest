class Solution {
    public boolean equationsPossible(String[] equations) {
        int[] map = new int[128];
        for (int i = 0; i < map.length; i++) {
            map[i] = i;
        }
        
        for (String x : equations) {
            if (x.charAt(1) == '!') {
                continue;
            }
            char a = x.charAt(0);
            char b = x.charAt(3);
            char fatherA = find(map, a);
            char fatherB = find(map, b);
            map[fatherA] = (int) fatherB; 
        }
        
        for (String x : equations) {
            if (x.charAt(1) == '!') {
                char a = x.charAt(0);
                char b = x.charAt(3);
                char fatherA = find(map, a);
                char fatherB = find(map, b);
                if (fatherA == fatherB) {
                    return false;
                }
            }
        }
        
        return true;
    }
    
    private char find(int[] map, char x) {
        char cur = x;
        while (map[cur] != cur) {
            cur = (char) map[cur];
        }
        
        return cur;
    }
}
