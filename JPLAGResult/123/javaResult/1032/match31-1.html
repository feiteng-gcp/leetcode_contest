class Solution {
    public boolean equationsPossible(String[] equations) {
        List[] l = new ArrayList[26];
        for (int i = 0; i < 26; i++) l[i] = new ArrayList()
;
        for (String eq : equations) {
            int c1 = eq.charAt(0) - 'a', c2 = eq.charAt(3) - 'a';
            if (eq.charAt(1) == '=') {
                l[c1].add(c2);
                l[c2].add(c1);
            }
        }
        boolean[] seen = new boolean[26];
        int[] g = new int[26];
        int j = 0;
        for (int i = 0; i < 26; i++) {
            dfs(i, seen, l, i, g);
        }
        //System.out.println(Arrays.toString(g));
        
for (String eq : equations) {
            int c1 = eq.charAt(0) - 'a', c2 = eq.charAt(3) - 'a';
            if (eq.charAt(1) == '!' && g[c1] == g[c2]) return false;
        }
        return true;
    }
    
    private void dfs(int i, boolean[] seen, List<Integer>[] l, int j, int[] g) {
        if (!seen[i]) {
            seen[i] = true;
            g[i] = j;
            for (Integer k : l[i]) {
                dfs(k, seen, l, j, g);
            }
        }
    }
}
