class Solution {
    public boolean equationsPossible(String[] equations) {
        int[] parent = new int[26];
        for (int i=0; i<26; i++) {
            parent[i] = -1;
        }
        
        // first union by ""==""
        for (String str : equations) {
            int x = str.charAt(0)-'a'; int y = str.charAt(3)-'a';
            if (str.charAt(1) == '=' && x!=y) {
                union(parent, x, y);
            }
        }
        
        // second find confliction
        
for (String str : equations) {
            int x = str.charAt(0)-'a'; int y = str.charAt(3)-'a';
            if (str.charAt(1) == '!') {
                if (x == y) {
                    return false;
                }
                int xroot = find(parent, x); int yroot = find(parent, y);
                if (xroot == yroot) {
                    return false;
                }
            }
        }
        return true;
    }
    
    private int find(int[] parent, int i) {
        if (parent[i] == -1) {
            return i;
        }
        return find(parent, parent[i]);
    }
    
    private void union(int[] parent, int x, int y) {
        int xroot = find(parent, x);
        int yroot = find(parent, y);
        if (xroot == yroot) {return;}  // important
        parent[xroot] = yroot;
    }
}
