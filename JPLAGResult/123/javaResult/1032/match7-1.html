class Solution {
    public boolean equationsPossible(String[] equations) {
        int[] map = new int[26];
        for(int i = 0 ; i < 26; i++){
            map[i] = i;
        }
        for(String e : equations){
            int a = e.charAt(0) - 'a';
            int b = e.charAt(3) - 'a';
            if(e.charAt(1) == '='){
                union(a,b,map);
            }
        }
        
for(String e : equations){
            int a = e.charAt(0) - 'a';
            int b = e.charAt(3) - 'a';
            if(e.charAt(1) == '!'){
                if(find(a,map) == find(b,map)){
                    return false;
                }
            }
        }
        return true;
    }
    public void union(int a, int b, int[] map){
        int finda = find(a,map);
        int findb = find(b,map);
        if(finda == findb){
            return;
        }
        map[finda] = findb;
    }
    public int find(int a, int[] map){
        while(map[a] != map[map[a]]){
            map[a] = map[map[a]];
        }
        return map[a];
    }
    
}
