class Solution {
    char[] parent;
    public boolean equationsPossible(String[] equations) {
        parent = new char[26];
        for(int i = 0; i < 26; i++) {
            parent[i] = (char)('a' + i);
        }

        for(String e : equations) {
            if(e.charAt(1) == '=') {
                parent[find(e.charAt(0)) - 'a'] = find(e.charAt(3));
            }
        }
                            
        for(String e : equations) {
            if(e.charAt(1) == '!') {
                if(find(e.charAt(0)) == find(e.charAt(3))) {
                    return false;   
                }
            }
        }
        return true;
    }
    public Character find(Character child) {
        int i = child - 'a';
        if(child != parent[i]) {
            parent[i] = parent[parent[i] - 'a'];
        }
        return parent[i];
    }
}
