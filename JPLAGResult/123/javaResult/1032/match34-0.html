class Solution {
    int[] ids = new int[26];
    public boolean equationsPossible(String[] equations) {
        for(int j = 0; j < 26; j++)
            ids[j] = j;
        for(String equation : equations) {
            int f = equation.charAt(0) - 'a'; int s = equation.charAt(3) - 'a';
            boolean eq = equation.charAt(1) == '=';
            if (!eq) continue;
            int fRoot = FindRoot(f);int sRoot = FindRoot(s);
            if (fRoot != sRoot)
                ids[fRoot] = sRoot;
        }
        
        for(String equation : equations) {
            int f = equation.charAt(0) - 'a'; int s = equation.charAt(3) - 'a';
            boolean eq = equation.charAt(1) == '=';
            if (eq) continue;
            int fRoot = FindRoot(f);int sRoot = FindRoot(s);
            if (fRoot == sRoot) return false;
        }
        return true;
    }
    
    int FindRoot(int id) {
        while(ids[id] != id) {
            ids[id] = ids[ids[id]];
            id = ids[id];
        }
        return id;
    }
}
