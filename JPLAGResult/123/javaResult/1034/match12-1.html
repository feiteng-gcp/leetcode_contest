	class Solution {
	    public 
int subarraysWithKDistinct(int[] A, int K) {
	    	return count(A, K) - count(A, K-1);
	    }
	    
	    int count(int[] a, int K)
	    {
	    	int n = a.length;
	        int[] f = new int[n+1];
	        int dis = 0;
	        int p = 0;
	        int ret = 0;
	        for(int i = 0;i < n;i++){
	        	if(++f[a[i]] == 1){
	        		dis++;
	        	}
	        	while(dis > K){
	        		if(--f[a[p++]] == 0){
	        			dis--;
	        		}
	        	}
	        	ret += i-p+1;
	        }
	        return ret;
	    }
	}
