class Solution {
    public int subarraysWithKDistinct(int[] A, int K) {
        Map<Integer, Integer> counts_j = new HashMap<>();
        int result = 0, i = 0, j = 0, len = A.length;
        while (counts_j.size() < K && j < len) {
            counts_j.compute(A[j++], (k, v) -> v == null ? 1 : v + 1);
        }
        if (counts_j.size() != K) return 0;

        int r = 1, a, c;
        Map<Integer, Integer> counts_i = new HashMap<>();
        while (true) {
            a = A[i];
            if (counts_i.getOrDefault(a, 0) == counts_j.get(a) - 1) break;
            r++;
            counts_i.compute(a, (k, v) -> v == null ? 1 : v + 1);
            i++;
        }
        result += r;

        while (j < len) {
            c = counts_j.compute(A[j++], (k, v) -> v == null ? 1 : v + 1);
            if (c == 1) {
                r = 1;
                a = A[i++];
                counts_i.compute(a, (k, v) -> v == null ? 1 : v + 1);
                counts_i.forEach((key, value) -> counts_j.computeIfPresent(key, (k, v) -> v.equals(value) ? null : v - value));
                counts_i.clear();
            }
            while (true) {
                a = A[i];
                if (counts_i.getOrDefault(a, 0) == counts_j.get(a) - 1) break;
                r++;
                counts_i.compute(a, (k, v) -> v == null ? 1 : v + 1);
                i++;
            }
            result += r;
        }

        return result;
    }
}
