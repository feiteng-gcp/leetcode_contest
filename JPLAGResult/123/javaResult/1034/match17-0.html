class Solution {
    public int subarraysWithKDistinct(int[] A, int K) {
        Map<Integer, Integer> map1 = new HashMap<Integer, Integer>();
        Map<Integer, Integer> map2 = new HashMap<Integer, Integer>();
        int s = 0;
        int e = 0;
        int n = A.length;
        int curr = 0;
        while(map1.size() < K && curr < n){
                map1.put(A[curr], map1.getOrDefault(A[curr], 0) + 1);
                map2.put(A[curr], map2.getOrDefault(A[curr], 0) + 1);
                curr++;
        }
        int ans = 0;
        //System.out.println("curr: " + curr);
        for(int i=curr;i<=n;i++){
            // System.out.println("map1: " + map1);
            // System.out.println("size1: " + map1.size());
            while(map1.size() > K && s < n){
                int temp = map1.get(A[s]);
                if(temp == 1){
                    map1.remove(A[s]);
                }else{
                    map1.put(A[s], temp-1);
                }
                s++;
            }
            // System.out.println("map2: " + map2);
            // System.out.println("size2: " + map2.size());
            while(map2.size() == K && e < n){
                int temp = map2.get(A[e]);
                if(temp == 1){
                    map2.remove(A[e]);
                }else{
                    map2.put(A[e], temp-1);
                }
                e++;
            }
            ans += (e-s);
            //System.out.println("s: " + s + " e: " + e);
            if(i != n)
            {
                map1.put(A[i], map1.getOrDefault(A[i], 0) + 1);
                map2.put(A[i], map2.getOrDefault(A[i], 0) + 1);
            }
        }
        return ans;
        
    }
}
