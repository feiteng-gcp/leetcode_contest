class Solution {
    public int subarraysWithKDistinct(int[] A, int K) {
        if (K == 0 || A.length == 0) {
            return 0;
        }
        Map<Integer, Integer> map = new HashMap<>();
        int left = 0, right = 0;
        int res = 0;
        while (right < A.length) {
            map.put(A[right], map.getOrDefault(A[right], 0) + 1);
            if (map.size() == K) {
                int tmp = right;
                int tmpNum = 0;
                while (tmp < A.length && map.containsKey(A[tmp])) {
                    tmp++;
                    tmpNum++;
                }
                while (map.size() == K) {
                    res += tmpNum;
                    if (map.get(A[left]) == 1) {
                        map.remove(A[left]);
                    } else {
                        map.put(A[left], map.get(A[left]) - 1);
                    }
                    left++;
                }
            } 
            right++;
        }
        return res;
    }
}
