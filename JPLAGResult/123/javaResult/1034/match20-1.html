class Solution {
    public int subarraysWithKDistinct(int[] A, int K) {
        Map<Integer, Integer> map = new HashMap();
        int start = 0;
        int count = 0;
        
        for (int i = 0; i < A.length; i++) {
            if (map.containsKey(A[i])) {
                map.put(A[i], map.get(A[i]) + 1);
            } else {
                map.put(A[i], 1);
            }
            
            if (map.size() >= K) {
                while (map.size() > K) 
{
                    map.put(A[start], map.get(A[start]) - 1);
                    
                    if (map.get(A[start]) == 0) {
                        map.remove(A[start]);
                    }
                    
                    start++;
                }
                
                int localStart = start;
                Map<Integer, Integer> localMap = new HashMap(map);
                while (localMap.size() == K) {
                    count++;
                    localMap.put(A[localStart], localMap.get(A[localStart]) - 1);
                    
                    if (localMap.get(A[localStart]) == 0) {
                        break;
                    }
                    
                    localStart++;
                }
            }
        }
        
        return count;
    }
}
