class Solution {
    public int subarraysWithKDistinct(int[] A, int K) {
        Map<Integer, Integer> lastIndex = new HashMap<>();
        TreeMap<Integer, Integer> treeMap = new TreeMap<>();
        Map<Integer, Integer> map = new HashMap<>();
        int result = 0;
        int left = -1;
        for (int right = 0; right < A.length; right++) {
            int prevLastIndex = lastIndex.getOrDefault(A[right], -1);
            if (prevLastIndex != -1) {
                treeMap.remove(prevLastIndex);
            }
            treeMap.put(right, A[right]);
            lastIndex.put(A[right], right);
            int count = map.getOrDefault(A[right], 0);
            if (count == 0) {
                K--;
            }
            map.put(A[right], count + 1);
            while (K < 0) {
                left++;
                count = map.getOrDefault(A[left], 0);
                if (count == 1) { // the last one
                    K++;
                    lastIndex.remove(A[left]);
                    treeMap.remove(left);
                }
                map.put(A[left], count - 1);
            }
            
            if (K == 0) {
                // System.out.println(left + " " + right + "  " + treeMap.firstKey());
                int temp = 
treeMap.firstKey();
                result += temp - left;
            }
        }
        return result;
    }
}
