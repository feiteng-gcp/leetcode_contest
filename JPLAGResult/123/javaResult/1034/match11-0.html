class Solution {
    public int subarraysWithKDistinct(int[] a, int k) {
        Map<Integer, Integer> num2Idx = new HashMap<>();
        Map<Integer, Integer> nextIdx = new HashMap<>();
        Map<Integer, Integer> cnts = new HashMap<>();
        int len = a.length;
        for (int i = len - 1; i >= 0; i--) {
            Integer nextId = num2Idx.get(a[i]);
            nextIdx.put(i, nextId);
            num2Idx.put(a[i], i);
        }
        int ileft = 0;
        int iright0 = -1, iright1 = -1;
        for (int i = 0; i < len; i++) {
            if (cnts.size() == k && cnts.get(a[i]) == null) {
                break;
            }
            cnts.put(a[i], cnts.getOrDefault(a[i], 0) + 1);
            if (iright0 == -1  && cnts.size() == k ) {
                iright0 = i;
            }
            if (cnts.size() == k) {
                iright1 = i;
            }
        }
        
           // for (Map.Entry<Integer, Integer> e : cnts.entrySet()) {
               // System.out.print(e.getKey() + ":"+ e.getValue() + " ");
            //}        
        
        if (iright0 == -1) {
            return 0;
        }
        int ret = iright1 - iright0 + 1;
        for (ileft = 0; ileft < len; ileft++) {
            int cur = a[ileft];
            Integer nextId = nextIdx.get(ileft);
            if (nextId == null || nextId > iright1) {
                cnts.remove(cur);
                boolean start = false;

                for (int i = iright1 + 1; i < len; i++) {
                    if (cnts.size() == k && cnts.get(a[i]) == null) {
                        break;
                    }
                    cnts.put(a[i], cnts.getOrDefault(a[i], 0) + 1);
                    if (start == false && cnts.size() == k) {
                        iright0 = i;
                        start = true;
                    }
                    if (cnts.size() == k) {
                        iright1 = i;
                    } 
                }
                if (cnts.size() < k) {
                    break;
                }
                ret += iright1 - iright0 + 1;
            } else if (nextId > iright0) {
                iright0 = nextId;
                ret += iright1 - iright0 + 1;
                cnts.put(cur, cnts.get(cur) - 1);
            } else {
                ret += iright1 - iright0 + 1;
                cnts.put(cur, cnts.get(cur) - 1);
            }
          //  System.out.println(iright0 + " " + iright1);
        //    for (Map.Entry<Integer, Integer> e : cnts.entrySet()) {
           //     System.out.print(e.getKey() + ":"+ e.getValue() + " ");
            //}
            //System.out.println(" ");
            if (cnts.size() < k) {
                return ret;
            }
        }
        
        return ret;
    }
}
