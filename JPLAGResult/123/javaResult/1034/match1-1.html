class Solution {
    public int subarraysWithKDistinct(int[] A, int K) {
        Map<Integer,Integer> map=new HashMap<>();
        TreeSet<Integer> set=new TreeSet<>();
        int head=0,ans=0;
        for (int i=0;i<A.length;i++)
        {
            int x=A[i];
            if (map.size()==K && !map.containsKey(x))
            {
                int first=set.first();
                head=first+1;
                set.remove(first);
                map.remove(A[first]);
            }
            /*
            if (map.size()==K && !map.containsKey(x))
            {
                int l=i-head;
                ans+=l*(l+1)/2;
                System.out.println(ans+" "+l+" "+i+" "+head);
                int first=set.first();
                head=first+1;
                set.remove(first);
                map.remove(A[first]);
            }
            */
            if (map.containsKey(x)) set.remove(map.get(x));
            map.put(x,i);
            set.add(i);
            if (map.size()==K)
            {
                ans+=
set.first()-head+1;
            }
        }
        //int l=A.length-head;
        //ans+=l*(l+1)/2;
        return ans;
    }
}
