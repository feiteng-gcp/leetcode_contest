class Solution {
    public int subarraysWithKDistinct(int[] A, int K) {
		int total = 0;
		Map<Integer, Integer> cntMap = new HashMap<>();
		
		int j = 0;
		for (int i = 0; i < A.length; i++) {
			for (; j < A.length; j++) {
				cntMap.put(A[j], cntMap.getOrDefault(A[j], 0) + 1);
				if (cntMap.size() == K) {
					total++;
				}
				if (cntMap.size() > K) {
					cntMap.remove(A[j]);
					break;
				}
			}
			if (j == A.length && cntMap.size() < K) {
				return total;
			}
			cntMap.put(A[i], cntMap.get(A[i]) - 1);
			if (cntMap.get(A[i]) == 0) {
				cntMap.remove(A[i]);
			} else {
				j = i + 1;
				cntMap.clear();
			}
		}
		return total;
    }
}
