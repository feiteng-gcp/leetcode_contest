import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {


    public int subarraysWithKDistinct(int[] A, int K) {
        return dfs(A, K, 0, A.length - 1);

    }

    int dfs(int[] A, int K, int l, int r) {
        if (l == r) {
            return K == 1 ? 1 : 0;
        }

        int ans = 0;
        int mid = (l + r) / 2;
        Map<Integer, Integer> cnt = new HashMap<>();
        Map<Integer, Integer> cnt1 = new HashMap<>();
        for (int i = l; i <= mid; i++) {
            add(cnt, A[i]);
            add(cnt1, A[i]);
        }
        add(cnt, A[mid + 1]);
        add(cnt1, A[mid + 1]);
        for (int i = l, j = mid + 1, jj = mid + 1; i <= mid && j <= r; i++) {
            while (cnt.size() < K && j + 1 <= r) {
                add(cnt, A[j + 1]);
                j++;
            }
            while (cnt1.size() <= K && jj + 1 <= r) {
                add(cnt1, A[jj + 1]);
                jj++;
            }
            if (cnt.size() == K) {
                if (cnt1.size() == K) {
                    ans += r - j + 1;
                } else if (cnt1.size() == K + 1) {
                    ans += jj - j;
                }
            }

            dec(cnt, A[i]);
            dec(cnt1, A[i]);
        }

        ans += dfs(A, K, l, mid);
        ans += dfs(A, K, mid + 1, r);
        return ans;
    }

    void add(Map<Integer, Integer> map, int i) {
        map.put(i, map.getOrDefault(i, 0) + 1);
    }

    void dec(Map<Integer, Integer> map, int i) {
        map.put(i, map.get(i) - 1);
        if (map.get(i) == 0) {
            map.remove(i);
        }
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(sol.subarraysWithKDistinct(new int[]{1, 2, 1, 2, 3}, 2));
        System.out.println(sol.subarraysWithKDistinct(new int[]{1, 2, 1, 3, 4}, 3));
        System.out.println(sol.subarraysWithKDistinct(new int[]{1, 2, 3, 4, 5}, 1));
        System.out.println(sol.subarraysWithKDistinct(new int[]{0, 0, 0, 0, 0}, 1));
        System.out.println(sol.subarraysWithKDistinct(new int[]{0, 0, 0, 0, 0, 0}, 1));
        System.out.println(sol.subarraysWithKDistinct(new int[]{0}, 1));
    }
}
