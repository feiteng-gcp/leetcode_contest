class Solution:
    def addToArrayForm(self, A: 'List[int]', K: 'int') -> 'List[int]':
        num = 0 
        for i in A:
            num = num * 10 + i
        num = num + K
        if num ==0: return [0]
        res = []
        while num:
            res.append(num % 10)
            num //= 10
        return res[::-1]
