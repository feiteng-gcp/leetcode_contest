class Solution:
    def addToArrayForm(self, A: 'List[int]', K: 'int') -> 'List[int]':
        x = 0
        for a in A:
            x = x * 10 + a
        x += K
        if x == 0:
            return [0]
        y = []
        while x > 0:
            y.append(x % 10)
            x //= 10
        return list(reversed(y))
