class Solution:
    def find(self, a):
        while self.father[a] != a:
            a = self.father[a]
        return a

    def union(self, a, b):
        f_a = self.find(a)
        f_b = self.find(b)
        self.father[f_b] = f_a

    def equationsPossible(self, equations: 'List[str]') -> 'bool':
        self.father = [i for i in range(26)]
        for eq in equations:
            a, b, q = ord(eq[0]) - 97, ord(eq[3]) - 97, eq[1]

            if q == '=':
                self.union(a, b)

        for eq in equations:
            a, b, q = ord(eq[0]) - 97, ord(eq[3]) - 97, eq[1]

            if q == '!':
                if self.find(a) == self.find(b):
                    return False

        return True
