class UnionFind:
    def __init__(self, N):
        self.pr = [i for i in range(N)]
        self.sz = [1] * N
    
    def root(self, i):
        while self.pr[i] != i:
            self.pr[i], i = self.pr[self.pr[i]], self.pr[i]
        return i
    
    def union(self, i, j):
        i, j = self.root(i), self.root(j)
        if i == j:
            return i
        elif self.sz[i] < self.sz[j]:
            self.pr[i] = j
            self.sz[j] += self.sz[i]
            return j
        else:
            self.pr[j] = i
            self.sz[i] += self.sz[j]
            return i
    
    def connected(self, i, j):
        i, j = self.root(i), self.root(j)
        return i == j
    
    def size(self, i):
        i = self.root(i)
        return self.sz[i]

class Solution:
    def equationsPossible(self, equations: 'List[str]') -> 'bool':
        uf = UnionFind(26)
        for eq in equations:
            x, e, _, y = list(eq)
            x, y = ord(x) - ord('a'), ord(y) - ord('a')
            if e == '=':
                uf.union(x, y)
        for eq in equations:
            x, e, _, y = list(eq)
            x, y = ord(x) - ord('a'), ord(y) - ord('a')
            if e != '=':
                if uf.connected(x, y):
                    return False
        return True
