class Solution:
    def equationsPossible(self, equations: 'List[str]') -> 'bool':
        d = dict()
        count = 1
        for equation in equations:
            if equation[1] == '=':
                if equation[0] in d and equation[3] in d:
                    if d[equation[0]] != d[equation[3]]:
                        new = d[equation[0]]
                        old = d[equation[3]]
                        for i in d:
                            if d[i] == old:
                                d[i] = new
                elif equation[0] in d and equation[3] not in d:
                    d[equation[3]] = d[equation[0]]
                elif equation[0] not in d and equation[3] in d:
                    d[equation[0]] = d[equation[3]]
                else:
                    d[equation[0]] = count
                    d[equation[3]] = count
                    count += 1
                #print(d)
        for equation in equations:
            if equation[1] == '!':
                if equation[0] == equation[3]:
                    return False
                if equation[0] in d and equation[3] in d:
                    if d[equation[3]] == d[equation[0]]:
                        return False
        return True
