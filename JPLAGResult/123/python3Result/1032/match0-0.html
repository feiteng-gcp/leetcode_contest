class Solution:
    def equationsPossible(self, equations: 'List[str]') -> 'bool':
        eqs, neqs, p = [], [], list(range(26))
        for a,b,_,c in equations:
            a, c = ord(a)-97, ord(c)-97
            if b == "!": neqs.append((a, c))
            else: eqs.append((a, c))
        def find(x):
            if x != p[x]: p[x] = find(p[x])
            return p[x]
        def union(x, y):
            px, py = find(x), find(y)
            if px != py: p[px] = py
        for a, b in eqs: union(a, b)
        for a, b in neqs:
            if find(a) == find(b): return False
        return True
