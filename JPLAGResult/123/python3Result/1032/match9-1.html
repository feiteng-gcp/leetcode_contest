from collections import defaultdict
# union found revision
class UnionFind():
    def __init__(self):
        self.uf, self.rank, self.size = {}, {}, {}
        
    def add(self, x):
        if x not in self.uf:
            self.uf[x], self.rank[x], self.size[x] = x, 0, 1
        
    def find(self, x):
        self.add(x)
        if x != self.uf[x]:
            self.uf[x] = self.find(self.uf[x])
        return self.uf[
x]

    def union(self, x, y):  
        xr, yr = self.find(x), self.find(y)
        if xr == yr: return
        if self.rank[xr] <= self.rank[yr]:
            self.uf[xr] = yr
            self.size[yr] += self.size[xr]
            self.rank[yr] += (self.rank[xr] == self.rank[yr])
        else:
            self.uf[yr] = xr
            self.size[xr] += self.size[yr]
class Solution:
    def equationsPossible(self, equations: 'List[str]') -> 'bool':
        u = UnionFind()
        for s in equations:
            if s[1] == '=':
                u.add(s[0])
                u.add(s[3])
                u.union(s[0],s[3])
            if s[1] == '!':
                u.add(s[0])
                u.add(s[3])
        for s in equations:
            if s[1] == '=':                    
                if u.find(s[0]) != u.find(s[3]):
                        return False
            else:
                if u.find(s[0]) == u.find(s[3]):
                        return False
        return True
