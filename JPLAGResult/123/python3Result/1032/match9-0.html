class UF:
    def __init__(self, val):
        self.val = val
        self.rank = 1
        self.parent = None
    def __repr__(self):
        return "{}->{}".format(self.val, self.parent.val if self.parent else None)
    def find(self):
        if self.parent is None:
            return self
        else:
            self.parent = self.parent.find()
            return self.parent

    def union(self, rhs):
        pl, pr = self.find(), rhs.find()
        if pl == pr:
            return
        if pl.rank < pr.rank:
            pl.parent = pr
        elif pl.rank < pr.rank:
            pr.parent = pl
        else:
            pl.parent = pr
            pr.rank += 1

        nodes = {}
        for e in equations:
            l, r = e[0], e[-1]

                nodes[l] = UF(l)
            if r not in nodes:
                nodes[r] = UF(r)
