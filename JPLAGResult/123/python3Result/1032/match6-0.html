class Union:
    def __init__(self, n):
        self.parent = [i for i in range(n)]

    def search(self, i):
        if self.parent[i] != i:
            self.parent[i] = self.search(self.parent[i])
        return self.parent[i]

    def merge(self, i, j):
        # print('merge', i, j)
        p = self.search(i)
        q = self.search(j)
        self.parent[p] = q

class Solution:
    def equationsPossible(self, equations: 'List[str]') -> 'bool':
        chars = {}
        k = 0
        for x in equations:
            if '!' in x:
                i, j = x.split("!=")[0], x.split("!=")[1]
            else:
                i, j = x.split("==")[0], x.split("==")[1]
            if i not in chars:
                chars[i] = k
                k += 1
            if j not in chars:
                chars[j] = k
                k += 1
        # print(chars)
        u = Union(len(chars))
        for x in equations:
            if not '!' in x:
                # print("==: " + x.split("==")[0] + x.split("==")[1])
                i, j = x.split("==")[0], x.split("==")[1]
                u.merge(chars[i], chars[j])

        for x in equations:
            if '!' in x:
                # print("!: " + x.split("!=")[0] + x.split("!=")[1])
                i, j = x.split("!=")[0], x.split("!=")[1]
                if u.search(chars[i]) == u.search(chars[j]):
                    return False

        return True
