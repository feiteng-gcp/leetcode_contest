class Solution:
    
class UF:
        def __init__(self,m):
            self.fa=[i for i in range(m)]
            self.wei=[1]*m
        def find(self,x):
            while self.fa[x]!=x:
                self.fa[x]=self.fa[self.fa[x]]
                x=self.fa[x]
            return x
        def join(self,x,y):
            fx=self.find(x)
            fy=self.find(y)
            if fx==fy:
                return
            if self
.wei[fx]<self.wei[fy]:               
                self.fa[fx]=fy
                self.wei[fy]+=self.wei[fx]
            else:
                self.fa[fy]=fx
                self.wei[fx]+=self.wei[fy]
    def equationsPossible(self, equations: 'List[str]') -> 'bool':
        eq={}
        nq={}
        n=len(equations)
        uf=Solution.UF(26)
        oa=ord('a')
        for e in equations:
            if e[1]=='=':
                uf.join(ord(e[0])-oa,ord(e[3])-oa)
        
        for e in equations:
            if e[1]=='!':
                if uf.find(ord(e[0])-oa)==uf.find(ord(e[3])-oa):
                    return False
        return True
