class Solution:
    def brokenCalc(self, X: 'int', Y: 'int') -> 'int':
        if X>=Y:
            return X-Y
        ans=0
        y=Y
        x=X
        while True:
            
if x>=y:
                return ans+x-y
            if y%2==1:
                y+=1
                ans+=1
            else:
                ans+=1
                y//=2
