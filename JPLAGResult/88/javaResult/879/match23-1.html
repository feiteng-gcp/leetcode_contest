class Solution {
    public int maxDistToClosest(int[] seats) {
        int n = seats.length;
        if (n==0) return 0;
        int[] left = new int[n], right = new int[n];
        
        left[0] = Integer.MAX_VALUE;
        if(seats[0]==1) left[0] = 0;
        
for(int i=1;i<n;i++) {
            if(seats[i]==1) {
                left[i] = 0;
            } else if(left[i-1]==Integer.MAX_VALUE) {
                left[i] = Integer.MAX_VALUE;
            } else {
                left[i] = left[i-1] + 1;
            }
        }
        
        right[n-1] = Integer.MAX_VALUE;
        if(seats[n-1]==1) right[n-1] = 0;
        
for(int i=n-2;i>=0;i--) {
            if(seats[i]==1) {
                right[i] = 0;
            } else if(right[i+1]==Integer.MAX_VALUE) {
                right[i] = Integer.MAX_VALUE;
            } else {
                right[i] = right[i+1] + 1;
            }
        }
        
        int ret = 0;
        for(int i=0;i<n;i++) {
            int tmp = Math.min(left[i], right[i]);
            if (tmp > ret) ret = tmp;
        }
        return ret;
        
    }
}
