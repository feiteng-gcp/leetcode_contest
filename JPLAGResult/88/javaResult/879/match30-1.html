class Solution {
   public int maxDistToClosest(int[] seats) {
        if (seats == null || seats.length == 0) {
            return -1;
        } else {
            
int size = seats.length;
            int[] left = new int[size];

            int prevMax = -1;
            for (int i = 0; i < size; i++) {
                if (seats[i] == 1) {
                    prevMax = i;
                } else if (seats[i] == 0) {
                    if (prevMax == -1) {
                        left[i] = Integer.MAX_VALUE;
                    } else {
                        left[i] = i - prevMax;
                    }
                }
            }

            int[] right = new int[size];
            prevMax = -1;
            
for(int i = size-1; i>=0; i--) {
                if (seats[i] == 1) {
                    prevMax = i;
                } else if (seats[i] == 0) {
                    if (prevMax == -1) {
                        right[i] = Integer.MAX_VALUE;
                    } else {
                        right[i] = prevMax - i;
                    }
                }
            }
            
            int max = -1;
            int runner = -1;
            for(int i = 0; i < size; i ++) {
                
                if (seats[i] != 1 && max < Math.min(left[i], right[i])) {
                    max = Math.min(left[i], right[i]);   
                    runner = i;
                }
            }

            return max;
        }
    }
}
