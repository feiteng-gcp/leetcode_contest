class Solution {
    public int maxDistToClosest(int[] seats) {
        int length= 0;
        int idx_lo=0;
        while(idx_lo< seats.length){
            if(seats[idx_lo]==1){
                length=Math.max(length, idx_lo-0);
                break;
            }
            else{
                idx_lo++;
            }
        }
        
        int idx_hi=seats.length-1;
        while(idx_hi>=0){
            if(seats[idx_hi]==1){
                length=Math.max(length, seats.length-1-idx_hi);
                break;
            }
            else{
                idx_hi--;
            }
        }
        int pre=idx_lo;
        
for(int i=idx_lo+1;i<=idx_hi;i++){
            if(seats[i]==1){
                length=Math.max(length, (i-pre)/2);
                pre=i;
            }
        }
        return length;
    }
}
