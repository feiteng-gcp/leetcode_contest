class Solution {
    public int maxDistToClosest(int[] seats) {
        int res = 0, l = -1;
        
        for (int i = 0; i < seats.length; ++i) {
            if (seats[i] == 1) {
                
if (l == -1) {
                    res = Math.max(res, i);
                } else {
                    res = Math.max(res, (int) ((i - l) / 2));
                }
                
                l = i;
            } else if (i == seats.length - 1) {
                res = 
Math.max(res, seats.length - l - 1);
            }
        }
        
        return res;
    }
}
