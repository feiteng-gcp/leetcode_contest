class Solution {
    public int maxDistToClosest(int[] seats) {
        int left=0,right=0,mid=0;
        int i=0;
        for(i=0;i<seats.length;i++){
            if(seats[i]!=1) left++;
            else break;
        }
        i++;
        int count=0;
        for(;i<seats.length;i++){
            if(seats[i]==0){
                count++;
            }else{
                mid=Math.max(mid, count);
                count=0;
            }
        }
        right=count;
        return Math.max(left, Math.max(right, (mid+1)/2));
    }
}
