class Solution {
    public int maxDistToClosest(int[] seats) {
        int n = seats.length;
        List<Integer> nums = new ArrayList<>();
        for(int i = 0; i < n; ++ i){
            if(seats[i] == 1) nums.add(i);
        }
        int res = 0;
        
for(int i = 0; i < nums.size()-1; ++ i){ 
            res = Math.max(res, (nums.get(i+1)-nums.get(i))/2);
        }
         res = Math.max(res, nums.get(0)-0);
         res = Math.max(res, n-1- nums.get(nums.size()-1));
        return res;
    }
}
