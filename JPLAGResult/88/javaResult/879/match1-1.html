class Solution {
    public int maxDistToClosest(int[] seats) {
        int N = seats.length;
        
int last = -1;
        int o = 0;
        for (int i = 0; i < N; i++) {
            if (seats[i] == 1) {
                if (last == -1) {
                    o = i;
                
} else {
                    o = Math.max(o, (i - last) / 2);
                }
                last = i;
            }
        }
        o = Math.max(o, N - 1 - last);
        return o;
    }
}
