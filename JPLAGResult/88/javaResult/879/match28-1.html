class Solution {
    public int maxDistToClosest(int[] S) {
        int n = S.length;
        int max = 0;
        int d = -1;
        for(int i=0;i<n;i++){
            if(S[i] == 1){
                if(d == -1)
                    max = Math.max(max,i);
                else
                    max = Math.max(max,(i-d)/2);
                
d = i;
            }    
        }
        return Math.max(max,n-d-1);
    }
}
