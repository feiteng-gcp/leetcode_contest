class Solution {
    public int maxDistToClosest(int[] seats) {
        int idx=-1, res=0;
        for (int i=0; i<seats.length; i++){
            if (seats[i]==1){
                if (idx==-1) res=Math.max(res, i-idx-1);
                else res=Math.max(res, (i-idx)/2)
;
                idx=i;
            }
        }
        res=Math.max(res, (seats.length-idx-1));
        return res;
    }
}
