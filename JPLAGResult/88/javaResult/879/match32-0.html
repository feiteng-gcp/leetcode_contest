class Solution {
    public int maxDistToClosest(int[] seats) {
        int start = 0;
        int idx = 0;
        while (seats[idx]==0)
            idx++;
        
        int max = idx;
        start = idx;
        for (int i=idx+1; i<seats.length; i++) {
            if (seats[i] == 1) {
                max = Math.max((i-start)/2, max);
                start = i;
            }
        }
        
        if (seats[seats.length-1] == 0)
            max = Math.max(seats.length-start-1, max);
        
        return max;
    }
}
