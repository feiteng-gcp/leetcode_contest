class Solution {
    public int maxDistToClosest(int[] seats) {
        if (seats == null || seats.length == 0) return 0;
        int last = -1;
        int n = seats.length;
        if (n == 1) return 0;
        
int best = 0;
        for (int i = 0; i < n; ++i){
            if (seats[i] == 1){
                if (last == -1){
                    best = Math.max(best, i);
                }else{
                    int len = i - last - 1;
                    if (len > 0){
                        int k = len/2;
                        if (len % 2 != 0) k++;
                        best = Math.max(best, k);
                    }    
                
}
                
                last = i;
            }
        }
        if (last != n - 1){
            int k = n - last - 1;
            best = Math.max(best, k);
        }
        return best;        
    }
}
