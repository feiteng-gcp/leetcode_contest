class Solution {
    public int maxDistToClosest(int[] seats) {
        if(seats.length==0){
            return 0;
        }
        
int lastPos=-1;
        int max=0;
        for(int i=0;i<seats.length;i++){
            if(seats[i]==1){
                if(lastPos==-1){
                    max=Math.max(i,max);
                }
                else{
                    max=Math.max(max,(i-lastPos)/2);
                }
                lastPos=i;
            }
        }
        max=Math.max(seats.length-1-lastPos,max);
        return max;
    }
}
