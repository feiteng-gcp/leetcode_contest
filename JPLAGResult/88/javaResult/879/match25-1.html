class Solution {
    public int maxDistToClosest(int[] seats) {
        int zeros = 0;
        // Find first 1
        int i=0;
        while(seats[i++] == 0) {
            zeros++;
        }
        int max = zeros;
        zeros = 0;
        
for(; i<seats.length; i++) {
            if(seats[i] == 0) {
                zeros++;
            } else {
                max = Math.max(max,(zeros+1)/2);
                zeros = 0;
            }
        }
        max = Math.max(max,zeros);
        return max;
    }
}
