class Solution {
    public int maxDistToClosest(int[] seats) {
        if(seats == null || seats.length<2)
            return 0;
        int n = seats.length;
        int[] distFromLeft = new int[n];
        int prevPerson = -1;
        for(int i=0;i<n;i++){
            if(seats[i] == 1){
                prevPerson = i;
            } else if(prevPerson == -1){
                distFromLeft[i] = Integer.MAX_VALUE;
            } else{
                distFromLeft[i] = i - prevPerson;
            }
        }
        prevPerson = n;
        int distance = Integer.MIN_VALUE;
        for(int i=n-1;i>=0;i--){
            if(seats[i] == 1){
                prevPerson = i;
            } else if(prevPerson == n){
                distance = Math.max(distance, distFromLeft[i]);
            } else{
                distance = Math.max(distance, Math.min(distFromLeft[i], prevPerson - i));
            }
        }
        return distance;
    }
}
