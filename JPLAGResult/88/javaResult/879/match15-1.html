class Solution {
    public int maxDistToClosest(int[] seats) {
        int len = seats.length;
        List<Integer> oc = new ArrayList<>();
        for(int i = 0; i<len; i++){
            if(seats[i] == 0){
                continue;
            }
            if(oc.isEmpty()){
                oc.add(-i);
            }
            oc.add(i);
        }
        int gap = len - oc.get(oc.size() - 1) - 1;
        
for(int i = 1; i<oc.size(); i++){
            gap = Math.max(gap, (oc.get(i) - oc.get(i - 1)) / 2);
        }
        return gap;
    }
}
