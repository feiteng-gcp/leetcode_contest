class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        int n = quiet.length;
		List<Integer>[] edges = new ArrayList[n];
		int[] incoming = new int[n];
		for (int i = 0; i < n; i++) edges[i] = new ArrayList<Integer>();
		
		for (int[] arr : richer) {
			edges[arr[0]].add(arr[1]);
			incoming[arr[1]]++;
		}
		
		Queue<Integer> q = new LinkedList<>();
		int[] res = new int[n];
		for (int i = 0; i < n; i++) {
			res[i] = i;
			if (incoming[i] == 0) {
				q.offer(i);

			}
		}
		
		while (!q.isEmpty()) {
			int cur = q.poll();
			for (int e : edges[cur]) {
				if (quiet[res[cur]] < quiet[res[e]]) {
					res[e] = res[cur];
				}
				if (--incoming[e] == 0) q.offer(e);

			}
		}
		return res;
    }
}
