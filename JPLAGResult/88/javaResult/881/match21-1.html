class Solution {
    
public int[] loudAndRich(int[][] richer, int[] quiet) {
        Map<Integer, Set<Integer>> graph = new HashMap<>();
        for (int i = 0; i < richer.length; i++) {
            Set<Integer> neis = graph.getOrDefault(richer[i][1], new HashSet<>());
            neis.add(richer[i][0]);
            graph.put(richer[i][1], neis);
        }
        
        Map<Integer, Integer> quietmap = new HashMap<>();
        for (int i = 0; i < quiet.length; i++) {
            quietmap.put(quiet[i], i);
        }

        Integer[] dp = new Integer[quiet.length];
        for (int i = 0; i < quiet.length; i++) {
            if (dp[i] == null) {
                //dp[i] is the smallest under the current node
                dp[i] = dfs(dp, i, graph, quiet);
            }
        
}

        int[] res = new int[quiet.length];
        for (int i =0; i < res.length; i++) {
            res[i] = quietmap.get(dp[i]);
        }
        return res;
    }

    private int dfs(Integer[] dp, int node, Map<Integer, Set<Integer>> graph, int[] quiet) {
        if (dp[node] != null) {
            return dp[node];
        }

        dp[node] = quiet[node];
        Set<Integer> neis = graph.getOrDefault(node,new HashSet<>());
        for (Integer nei : neis) {
            dp[node] = Math.min(dp[node], dfs(dp, nei, graph, quiet));
        }
        return dp[node];
    }
}
