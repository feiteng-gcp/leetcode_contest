class Solution {
    // find all people richer than current
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        // map curr people to all that richer than him
        Map<Integer,Set<Integer>> map=new HashMap<Integer,Set<Integer>>();
        Map<Integer,Set<Integer>> richMap=new HashMap<Integer,Set<Integer>>();
        
for(int i=0;i<quiet.length;i++){
            map.put(i,new HashSet<Integer>());
        }
        for(int i=0;i<richer.length;i++){
            map.get(richer[i][1]).add(richer[i][0]);
        }
        
for(int i=0;i<quiet.length;i++){
            populateRicherMap(i,richMap,map);
        }
        
        int[] res=new int[quiet.length];
        for(int i=0;i<res.length;i++){
            Set<Integer> set=richMap.get(i);
            res[i]=findLoud(set,quiet)[0];
        }
        return res;
    }
    public int[] findLoud(Set<Integer> set,int[] quiet){
        int[] res={-1,Integer.MAX_VALUE};
        for(int p:set){
            int[] curr={p,quiet[p]};
            if(curr[1]<res[1]){
                res=curr;
            }
        
}
        return res;
    }
    public void populateRicherMap(int curr,Map<Integer,Set<Integer>> richMap,Map<Integer,Set<Integer>> map){
        if(richMap.containsKey(curr)){
            return;
        }
        Set<Integer> toadd=new HashSet<Integer>();
        toadd.add(curr);
        Set<Integer> next=map.get(curr);
        for(int n:next){
            populateRicherMap(n,richMap,map);
            toadd.addAll(richMap.get(n));
        }
        richMap.put(curr,toadd);
    }
}
