class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        if (richer == null || quiet == null || quiet.length == 0) return new int[0];
        int n = quiet.length;
        List<List<Integer>> con = new ArrayList<List<Integer>>();
        
for (int i = 0; i < n; ++i){
            con.add(new ArrayList<Integer>());
        }
        for (int[] k : richer){
            con.get(k[1]).add(k[0]);
        }
        int[] res = new int[n];
        for (int i = 0; i < n; ++i){
            int[] vis = new int[n];
            // System.out.println("i: " + i);
            
res[i] = DFS(i,n,con,quiet,i,vis);
        }
        return res;
    }
    
    private int DFS(int cur, int n, List<List<Integer>> con, int[] quiet, int curq, int[] vis){
        // System.out.println(" cur: " + cur + " curq: " + curq + " quiet: " + quiet[curq]);
        vis[cur] = 1;
        int nxtq = curq;
        if (quiet[cur] < quiet[curq]){
            nxtq = cur;
        }
        for (int u : con.get(cur)){
            if (vis[u] == 1) continue;
            int idx = DFS(u, n, con, quiet, nxtq, vis);
            if (quiet[nxtq] > quiet[idx]){
                nxtq = idx;
            }
        }
        return nxtq;
    }
}
