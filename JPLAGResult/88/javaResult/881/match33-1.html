class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        int n=quiet.length;
        Set<Integer>[] up= new HashSet[n], down= new HashSet[n];
        for (int i=0; i<n; i++) up[i]=new HashSet<>();
        for (int i=0; i<n; i++) down[i]=new HashSet<>();
        for (int[] r: richer){
            up[r[1]].add(r[0]);
            down[r[0]].add(r[1]);
        }
        
        int[] res= new int[n];
        Set<Integer> valid= new HashSet<>();
        for (int i=0; i<n; i++){
            res[i]=i;
            if (up[i].isEmpty()) valid.add(i);
        } 
        
        while(!valid.isEmpty()){
            Set<Integer> next= new HashSet<>();
            for (int cur: valid){
                for (int sub: down[cur]){
                    if (quiet[sub]>=quiet[cur]){
                        quiet[sub]=quiet[cur];
                        res[sub]=res[cur];
                    } 
                    up[sub].remove(cur);
                    if (up[sub].isEmpty()) next.add(sub);
                }
            }
            valid=next;
        }
        return res;
    }
}
