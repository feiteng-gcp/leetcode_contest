class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        
        for(int i=0; i<richer.length; i++) {
            if (map.containsKey(richer[i][1])) {
                map.get(richer[i][1]).add(richer[i][0]);
            } else {
                Set<Integer> set = new HashSet<Integer>();
                set.add(richer[i][0]);
                map.put(richer[i][1], set);
            }
        }
        
        int[] answer = new int[quiet.length];
        Map<Integer, Integer> quietSet = new HashMap<Integer, Integer>();
        for (int i=0; i<quiet.length; i++) {
            answer[i] = getQueit(map, quietSet, quiet, i);
        }
        
        return answer;
    }
    
    private int getQueit(Map<Integer, Set<Integer>> map, Map<Integer, Integer> quietSet, int[] quiet, int cur) {
        if (quietSet.containsKey(cur))
            return quietSet.get(cur);
        if (!map.containsKey(cur))
            return cur;
        int minIdx = cur;
        for (Integer i : map.get(cur)) {
            int minI = getQueit(map, quietSet, quiet, i);
            if (quiet[minI] < quiet[minIdx])
                minIdx = minI;
        }
        quietSet.put(cur, minIdx);
        
        return minIdx;
    }
}
