class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        int n = quiet.length;
        List<List<Integer>> graph = new ArrayList();
        for (int i = 0; i < n; i++)
            graph.add(new ArrayList<Integer>());
        for (int[] r : richer)
            graph.get(r[1]).add(r[0]);
        int[] res = new int[n];
        for (int i = 0; i < n; i++) {
            int min = quiet[i];
            res[i] = i;
            boolean[] used = new boolean[n];
            used[i] = true;
            Queue<Integer> q = new LinkedList();
            q.offer(i);
            while (!q.isEmpty())
                for (int j = q.size(); j > 0; j--) {
                    int cur = q.poll();
                    for (int next : graph.get(cur)) 
                        if (!used[next]) {
                            used[next] = true;
                            if (quiet[next] < min) {
                                min = quiet[next];
                                res[i] = next;
                            }
                            q.offer(next);
                        }
                }
        }
        return res;
    }
}
