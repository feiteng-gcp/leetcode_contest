class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        //1. Find who has >= money than x
        Map<Integer,List<Integer>> m = new HashMap<Integer,List<Integer>>(); // People who definitely have more money than me
        
for(int i=0; i<quiet.length; i++) {
            m.put(i, new ArrayList<Integer>());
        }
        for(int[] rich : richer) {
            m.get(rich[1]).add(rich[0]);
        }
        // Compute quietest for each one
        int[] output = new int[quiet.length];
        Map<Integer,Integer> quietest = new HashMap<>();
        for(int i=0; i<quiet.length; i++) {
            output[i] = getQuietest(quietest,m,quiet,i);
        
}
        return output;
    }
    
    int getQuietest(Map<Integer,Integer> quietest, Map<Integer,List<Integer>> adj, int[] quiet, int i) {
        if(!quietest.containsKey(i)) {
            PriorityQueue<Integer> que = new PriorityQueue<>((a,b)->quiet[a]-quiet[b]);
            
for(int a : adj.get(i)) {
                que.add(getQuietest(quietest, adj, quiet, a));
            }
            que.add(i);
            quietest.put(i, que.poll());
        }
        return quietest.get(i);
    }
}
