class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
		Map<Integer, Set<Integer>> map = new HashMap<>();
		int leng = quiet.length;
		for (int i = 0; i < leng; i++) {
			map.put(i, new HashSet<>());
		}
		for (int[] arr: richer) {
			int rich = arr[0];
			int poor = arr[1];
			map.get(poor).add(rich);
		}
		boolean[] done = new boolean[leng];
		for (int i = 0; i < leng; i++) {
			reduce(done, map, i);
		}
		int[] ans = new int[leng];
		for (int i = 0; i < leng; i++) {
			int q = quiet[i];
			int qIdx = i;
			for (int rich : map.get(i)) {
				if (quiet[rich] < q) {
					q = quiet[rich];
					qIdx = rich;
				}
			}
			ans[i] = qIdx;
		}
		return ans;
    }

	private void reduce(boolean[] done, Map<Integer, Set<Integer>> map, int i) {
		if (done[i]) {
			return;
		}
		for (int rich : map.get(i)) {
			reduce(done, map, rich);
		}
        Set<Integer> newRich = new HashSet<>();
		for (int rich : map.get(i)) {
			newRich.addAll(map.get(rich));
		}
        map.get(i).addAll(newRich);
		done[i] = true;
	}
}
