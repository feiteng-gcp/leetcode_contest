class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        int n=quiet.length;
        
        int[] dp=new int[n];     //记录每个节点的结果。
        
        Arrays.fill(dp,-1);
        
        ArrayList<Integer>[] ps=new ArrayList[n];
        for(int i=0;i<n;i++) ps[i]=new ArrayList<>();
        for(int i=0;i<richer.length;i++){
            ps[richer[i][1]].add(richer[i][0]);
        }
        for(int i=0;i<n;i++){
            DP(ps, dp, i, quiet);
        }
        return dp;
    }
    
    public int DP(ArrayList<Integer>[] ps, int[] dp, int p, int[] quiet){
        if(dp[p]!=-1)return dp[p];
        
        if(ps[p].size()==0){          //没有比他更有钱的
            dp[p]=p;
            return p;
        }
        int index=p;
        int queit=quiet[p];
        for(int i=0;i<ps[p].size();i++){
            int temp = DP(ps, dp, ps[p].get(i), quiet);     //找子节点的结果
            if(quiet[temp] < queit){                //比较每个子节点的结果，找最值
                index = temp;
                queit = quiet[index];
            }
        }
        dp[p]=index;
        return index;
    }
}
