class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        // naive: n * n
        // topological sort
        int N = quiet.length;
        int[] res = new int[N];
        // int[][] quietarr = new int[N][2];
        boolean[][] map = new boolean[N][N];
        boolean[] visited = new boolean[N];
        
        for (int i = 0; i < N; ++i) {
            res[i] = -1;
            // quietarr[i][0] = i;
            // quietarr[i][1] = quiet[i];
        }
        
        // Arrays.sort(quietarr, (a, b) -> (a[1] - b[1]));
        
        for (int[] pair : richer) {
            map[pair[0]][pair[1]] = true;
            visited[pair[1]] = true;
        }
        
        Queue<Integer> q = new LinkedList<>();
        
        for (int i = 0; i < N; ++i) {
            if (!visited[i]) {
                res[i] = i;
                q.offer(i);
            }
        }
        
        while (!q.isEmpty()) {
            int cur = q.poll();
            
            for (int i = 0; i < N; ++i) {
                if (map[cur][i] && (res[i] == -1 || quiet[res[cur]] < quiet[res[i]])) {
                    res[i] = quiet[i] < quiet[res[cur]] ? i : res[cur];
                    q.offer(i);
                }
            }
        }
        
        return res;
        // [5,5,2,5,4,5,6,7]
    }
}
