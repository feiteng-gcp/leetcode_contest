class Solution {    
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int[] r: richer) {
            if (!map.containsKey(r[1])) {
                map.put(r[1], new HashSet<>());
            }
            map.get(r[1]).add(r[0]);
        }
        
        int[] res = new int[quiet.length];
        boolean[] visited = new boolean[quiet.length];
        for (int i = 0; i < res.length; i++) {
            res[i] = leastQuietPerson(allRicher(map, i, visited), i, quiet);
        }
        return res;
    }
    
    Set<Integer> allRicher(Map<Integer, Set<Integer>> map, int p, boolean[] visited) {
        if (!map.containsKey(p)) return new HashSet<>();
        if (visited[p]) return map.get(p);
        
        visited[p] = true;
        Set<Integer> set = new HashSet<>();
        for (int next: map.get(p)) {
            Set<Integer> s = allRicher(map, next, visited);
            if (s.size() > 0) set.addAll(s);
        }
        if (set.size() > 0) map.get(p).addAll(set);
        return map.get(p);
    }
    
    int leastQuietPerson(Set<Integer> persons, int i, int[] quiet) {
        int res = i;
        int least = quiet[i];
        for (int p: persons) {
            if (quiet[p] < least) {
                least = quiet[p];
                res = p;
            }
        }
        return res;
    }
}
