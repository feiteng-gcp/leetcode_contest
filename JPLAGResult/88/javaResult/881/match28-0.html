class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        int N= quiet.length;
        Map<Integer, Set<Integer>> map=new HashMap<>();
        for(int[] r: richer){
            Set<Integer> children= map.getOrDefault(r[1], new HashSet<Integer>());
            children.add(r[0]);
            if(!map.containsKey(r[1])) map.put(r[1],children);
        }
        
        //boolean[] visited= new boolean[quiet.length];
        int[] result= new int[quiet.length];
        Map<Integer, Integer> memory= new HashMap<>();
        for(int i =0;i< N;i++){
            dfs(map, i,quiet, result,memory);
        }
        return result;
    }
    
    private int dfs(Map<Integer, Set<Integer>> map, int cur, int[] quiet, int[] result, Map<Integer, Integer> memo){
        if(memo.containsKey(cur)) return memo.get(cur);
        if(!map.containsKey(cur)){
            result[cur]=cur;
            memo.put(cur, cur);
            return cur;
        }
        else{
            Set<Integer> set=map.get(cur);
            int quietest= cur;
            for(int num: set){
                int q=dfs(map, num, quiet, result,memo);
                if(quiet[q]< quiet[quietest]) quietest=q;
            }
            result[cur]=quietest;
            memo.put(cur, quietest);
            return quietest;
        }
    }
}
