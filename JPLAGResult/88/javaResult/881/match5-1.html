class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        Map<Integer, List<Integer>> edges = new HashMap<>();
        int n = quiet.length;
        
for (int i = 0; i < n; i++){
            edges.put(i, new ArrayList<>());
        }
        for (int[] rich : richer){
            int x = rich[0];
            int y = rich[1];
            edges.get(y).add(x);
        }
        Map<Integer, int[]> res = new HashMap<>();
        int[] ans = new int[n];
        
for (int i = 0; i < n; i++){
            ans[i] = dfs(edges, i, res, quiet)[1];
        }
        return ans;
    }
    
    int[] dfs(Map<Integer, List<Integer>> edges, int start, Map<Integer, int[]> res, int[] quiet){
        if (res.containsKey(start)){
            return res.get(start);
        }
        int min = quiet[start];
        int p = start;
        for (int nxt : edges.get(start)){
            int[] nA = dfs(edges, nxt, res, quiet);
            if (nA[0] < min){
                min = nA[0];
                p = nA[1];
            }
        }
        int[] ans = new int[]{min, p};
        res.put(start, ans);
        return ans;
    }
}
