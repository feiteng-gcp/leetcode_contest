class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        if (shifts == null || shifts.length == 0) return S;
        
char[] arr = S.toCharArray();
        long sum = 0L;
        for (int i = arr.length-1; i >= 0; --i){
            sum += (long)(shifts[i]);
            long idx = (long)(arr[i] - 'a');
            idx += sum;
            idx %= 26;
            arr[i] = (char)(idx + 'a');
        }
        return new String(arr);
    }
}
