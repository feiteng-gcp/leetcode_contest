class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        
        
        int N = shifts.length;
        
        for (int i = N - 2; i >= 0; --i) {
            shifts[i] = (shifts[i] + shifts[i + 1]) % 26;
        }
        
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < N; ++i) 
{
            char c = S.charAt(i);
            int shift = (shifts[i] + (c - 'a')) % 26;
            sb.append((char)('a' + shift));
        }
        
        return sb.toString();
    }
}
