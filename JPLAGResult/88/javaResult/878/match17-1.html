class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        if(S==null||S.length()==0||shifts.length==0) return S;
        int[] total= new int[shifts.length];
        total[total.length-1]=shifts[total.length-1]%=26;
        
for(int i =total.length-2; i>=0;i--){
            total[i]= total[i+1]+ shifts[i];
            total[i]%=26;
        }
        
        StringBuilder sb= new StringBuilder();
        for(int i =0;i< S.length() ;i++)
{
            char ch= (char)('a'+(S.charAt(i)-'a'+total[i])%26);
            sb.append(ch);
        }
        return sb.toString();
    }
}
