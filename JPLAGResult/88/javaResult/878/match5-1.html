class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        int n = S.length();

        long sum = 0;
        
for (int i = n - 1; i >= 0; i--) {
            sum += shifts[i];
            shifts[i] = (int) (sum % 26);
        }

        StringBuilder sb = new StringBuilder();

        for (int i = 0; i < n; i++) {
            char c = (char) ((
S.charAt(i) - 'a' + shifts[i]) % 26 + 'a');
            sb.append(c);
        }
        return sb.toString();
    }
}
