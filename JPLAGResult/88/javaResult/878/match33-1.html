class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        char[] ca= S.toCharArray();
        int sum=0;
        for (int i =shifts.length-1; i>-1; i--){
            sum= (sum+shifts[i])%26;
            ca[i]= (char)('a'+(ca[i]-'a'+sum)%26);
        }
        return new String(ca);
    }
}
