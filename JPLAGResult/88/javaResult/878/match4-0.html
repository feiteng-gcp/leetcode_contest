class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        int[] totalshift = new int[shifts.length];
        totalshift[shifts.length - 1] = shifts[shifts.length - 1];
        for(int i = shifts.length - 2; i >=0; i--){
            totalshift[i] = (totalshift[i+1] + shifts[i])%26;
        }
        StringBuilder res =  new StringBuilder();

        for(int i = 0; i < shifts.length; i++){
            
            res.append((char)((S.charAt(i) - 'a' + totalshift[i]) % 26 +'a'));
            /*totalshift[i] = totalshift[i]%26;
            if(S.charAt(i) - 'a' + totalshift[i] > 25)
                res += (char)(S.charAt(i) + totalshift[i] - 26);
            else
                res += (char)(S.charAt(i) + totalshift[i]);*/
        }
        return res.toString();
    }
}
