class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        if(shifts == null || shifts.length == 0 || S == null || S.isEmpty()){
            return S;
        }
        int n = shifts.length;
        long[] total = new long[n+1];
        total[0] = 0;
        for(int i=1;i<=n;i++){
            total[i] = shifts[i-1]%26 + total[i-1];
        }
        char[] letters = S.toCharArray();
        for(int i=0;i<letters.length;i++){
            letters[i] = shiftLetter(letters[i], total[total.length-1] - total[i]);
        }
        return new String(letters);
    }
    private char shiftLetter(char c, long times){
        long diff = c-'a' + times;
        return (char) ('a' + diff%26);
    }
}
