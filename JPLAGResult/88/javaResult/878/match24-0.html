class Solution {
	public String shiftingLetters(String S, int[] shifts) {
    	char[] a = S.toCharArray();
    	int sh = 0;
    	for (int i=shifts.length-1; i>=0; i--) {
    		sh = (sh+shifts[i])%26;
    		a[i]+=sh;
    		if (a[i]>'z')
    			a[i] = (char) (a[i]-'z'+'a'-1);
    	}
		return new String(a);
    	
        
    }
}
