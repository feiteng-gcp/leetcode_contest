class Solution {
    public String shiftingLetters(String S, int[] A) {
        int n = A.length;
        for (int i = 0; i < n; i++){
            A[i] = A[i] % 26;
        }
        for (int i = n - 2; i >= 0; i--){
            A[i] = (A[i] + A[i+1]) % 26 ;
        }
        char[] B = S.toCharArray();
        int m = B.length;
        for (int i = 0; i < m; i++){
            int key = B[i] - 'a';
            B[i] = (char)((key + A[i]) % 26 + 'a');
        }
        return new String(B);
    }
}
