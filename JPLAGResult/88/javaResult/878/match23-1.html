class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        char[] chs = S.toCharArray();
        int tmp = 0;
        for (int i=chs.length-1; i>=0; i--) {
            tmp = (tmp + shifts[i])%26;
            chs[i] += tmp;
            if (chs[i]>'z') {
                chs[i]-=26;
            }
        }
        return new String(chs);
    }
}
