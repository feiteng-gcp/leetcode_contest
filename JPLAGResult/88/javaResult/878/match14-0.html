class Solution {
    public String shiftingLetters(String s, int[] sfs) {
        int sum = 0;
        for (int i = sfs.length - 1; i >= 0; i--) {
            sum = (sum + sfs[i]) % 26;
            sfs[i] = sum;
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            int ci = s.charAt(i) - 'a';
            int resultCi = (ci + sfs[i]) % 26;
            sb.append((char)('a' + resultCi));
        }
        return sb.toString();
    }
}
