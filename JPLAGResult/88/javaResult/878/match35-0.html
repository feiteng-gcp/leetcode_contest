class Solution {
    public String shiftingLetters(String S, int[] sh) {
        int n = sh.length;
        char[] A = S.toCharArray();
        for(int i = n-2;i>=0;i--){
            sh[i] = (sh[i+1]+sh[i])%26;
        }
        for(int i =0;i<n;i++){
            char c = A[i];
            int v = c - 'a';
            v = v + sh[i];
            v = v %26;
            A[i] = (char)('a' + v);
        }
        return new String(A);
    }
}
