class Solution {
   public String shiftingLetters(String s, int[] sums) {
        long[] cs = new long[sums.length];
        cs[sums.length - 1] = sums[sums.length - 1];
        for (int i = sums.length - 1; i > 0; i--)
            cs[i - 1] = cs[i] + sums[i - 1];
        StringBuilder r = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            int shift = (int) (cs[i] % 26);
            int p = (s.charAt(i) - 'a' + shift) % 26;
            r.append((char) (p + 'a'));
        }
        return r.toString();
    }
}
