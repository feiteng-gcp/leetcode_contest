class Solution {
    public String shiftingLetters(String S, int[] shifts) {
        for(int i=shifts.length-1;i>0;i--){
            shifts[i-1]+=shifts[i];
            shifts[i-1]%=26;
        }
        shifts[shifts.length-1]%=26;
        StringBuilder sb=new StringBuilder();
        for(int i=0;i<S.length();i++){
            int move=shifts[i];
            //if(shifts[i]>=26)move=move%26;
            if((char) (S.charAt(i)+move)>'z') {
                int tmp='z'-S.charAt(i);
                move-=tmp;
                move--;
                sb.append((char)('a'+move));
            }
            else sb.append((char)(S.charAt(i)+move));
        }
        return sb.toString();
    }
}
