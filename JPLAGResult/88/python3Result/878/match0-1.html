class Solution:
    def shiftingLetters(self, S, shifts):
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """
        
for i in range(len(shifts)-2,-1,-1):
            shifts[i]=(shifts[i]+shifts[i+1])%26
        S0=""
        for i in range(len(shifts)):
            S0=S0+chr((ord(S[i])-97+shifts[i])%26+97)
        return S0
