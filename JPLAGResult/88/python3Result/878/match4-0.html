class Solution:
    def shiftingLetters(self, S, shifts):
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """
        N = len(S)
        for i in range(1,N):
            shifts[N-1-i] += shifts[N-i]
        rList = []
        for i, ch in enumerate(S):
            j = ord(ch)-ord('a')
            t = ord('a') + (j + shifts[i]) % 26
            rList.append(chr(t))
        return ''.join(rList)
