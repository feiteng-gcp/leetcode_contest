class Solution:
    def shiftingLetters(self, S, shifts):
        length = len(shifts)
        preSum = [0] * (length + 1)
        for i in reversed(range(length)):
            preSum[i] = preSum[i + 1] + shifts[i]
        tmp = [0] * length

        for idx, c in enumerate(S):
            n = ord(c) + preSum[idx]
            if n <= 122:
                tmp[idx] = n
            else:
                n = (n - 123) % 26 + 97
                tmp[idx] = n
        return ''.join([chr(n) for n in tmp])
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """
