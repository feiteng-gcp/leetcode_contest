class Solution:
    def shiftingLetters(self, S, shifts):
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """

        def shift(letter, time):
            return chr((ord(letter) - ord('a') + time) % 26 + ord('a'))
        N = len(shifts)
        Sum = [0 for _ in range(N)]
        Sum[-1] = shifts[-1]
        for i in range(N - 2, -1, -1):
            Sum[i] = Sum[i + 1] + shifts[i]
        ans = ''
        for i in range(N):
            ans += shift(S[i], Sum[i])
        return ans
