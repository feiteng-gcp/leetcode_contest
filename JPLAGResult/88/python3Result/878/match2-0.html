class Solution:
    def shiftingLetters(self, s, shifts):
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """
        n = len(shifts)
        for i in range(n-2,-1,-1):
            shifts[i] = shifts[i+1] + shifts[i]
        
        res =""
        for i,shift in enumerate(shifts):
            # print(shift)
            c = chr(ord('a') + ( ord(s[i]) - ord('a') + shift ) % 26)
            res += c
        return res
