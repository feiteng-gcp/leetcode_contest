class Solution:
    def shiftingLetters(self, S, shifts):
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """
        shifts[-1]%=26
        n
=len(shifts)
        for i in range(n-2,-1,-1):
            shifts[i]=(shifts[i+1]+shifts[i])%26
        ans=''
        for i in range(n):
            ans+=chr((ord(S[i])-ord('a')+shifts[i])%26+ord('a'))
        return ans
