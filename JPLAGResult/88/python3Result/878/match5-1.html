class Solution:
    def shiftingLetters(self, S, shifts):
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """
        for i in range(len(shifts)-2, -1, -1):
            shifts[i] += shifts[i+1]
        
        s_arr = list(S)
        
        for i in range(len(s_arr)):
            c = s_arr[i]
            x = (ord(c) - ord('a') + shifts[i]) % 26
            x += ord('a')
            s_arr[i] = chr(x)
        
        return ''.join(s_arr)
