class Solution:
    def shiftingLetters(self, S, shifts):
        N = len(shifts)
        for i in range(N-2, -1, -1):
            shifts[i] += shifts[i+1]
        
        ret = []
        for i in range(N):
            ret.append(chr((ord(S[i])-ord('a')+shifts[i])%26+ord('a')))
        return ''.join(ret)
