class Solution:
    def shiftingLetters(self, S, shifts):
        """
        :type S: str
        :type shifts: List[int]
        :rtype: str
        """
        
        for i in range(len(shifts)-2, -1, -1):
            shifts[i] += shifts[i+1]
        a = list(S)
        for i, s in enumerate(a):
            o = (ord(s) + shifts[i] - ord('a')) % 26
            a[i] = chr(ord('a') + o)
            
        return ''.join(a)
