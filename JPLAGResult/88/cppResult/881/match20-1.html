vector<int>v[501];
int memo[501];

vector<int>q;
map<int,int>m;
int dp(int pos){
    if(memo[pos]!=-1)return memo[pos];
    int dev=q[pos];
    
    for(int i=0;i<v[pos].size();i++){
        dev=min(dev,dp(v[pos][i]));    
    }
        
    memo[pos]=dev;
    return dev;
}

class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        q=quiet;
        m.clear();
        for(int i=0;i<501;i++)v[i].clear();
        
        for(int i=0;i<quiet.size();i++){
            m[quiet[i]]=i;
        }
        
        for(int i=0;i<richer.size();i++){
            int a=richer[i][0];
            int b=richer[i][1];
            v[b].push_back(a);
        }
        
        memset(memo,-1,sizeof(memo));
        vector<int>dev((int)quiet.size(),0);
        for(int i=0;i<quiet.size();i++){
            dev[i]=m[dp(i)];
        }
        
        return dev;
    }
};
