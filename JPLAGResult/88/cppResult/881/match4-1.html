class Solution {
public:
    int dfs(vector<int> &answer, vector<vector<int>> &g, vector<int> &quiet, int x) {
        if (answer[x] >= 0) {
            return answer[x];
        }
        //cout << "x = " << x << endl;
        answer[x] = x;
        for (int i = 0; i < g[x].size(); ++i) {
            int y = g[x][i];
            answer[y] = dfs(answer, g, quiet, y);
            if (quiet[answer[y]] < quiet[answer[x]]) {
                answer[x] = answer[y];
            }
        }

        return answer[x];
    }
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n = quiet.size();
        vector<int> answer(n, -1);
        vector<vector<int> > g;
        g.resize(n);
        for (int i = 0; i < richer.size(); ++i) {
            g[richer[i][1]].push_back(richer[i][0]);
        }
        //cout << "ooo" << endl;
        for (int i = 0; i < n; ++i) {
            answer[i] = dfs(answer, g, quiet, i);
        }
        return answer;
    }
};
