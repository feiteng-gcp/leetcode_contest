class Solution {
private:
    int return_minimum(vector<vector<int> >& Graph, vector<int>& quiet, vector<int>& ANS, int input){
      if(ANS[input] >= 0) return ANS[input];
      if(Graph[input].size() == 0){
        ANS[input] = input;
        return input;
      }

      int ans = input;
      for(int i=0; i<Graph[input].size(); ++i){
        int tmp = return_minimum(Graph, quiet, ANS, Graph[input][i]);
        if(quiet[tmp] < quiet[ans])
          ans = tmp;
      }
      ANS[input] = ans;
      return ans;
    }
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
      int V = quiet.size();
      vector<vector<int> > Graph;
      //Graph[x][y] shows connection from x to y
      Graph.resize(V);
      
      for(int i=0; i<richer.size(); ++i){
        int from = richer[i][1];
        int to = richer[i][0];
        Graph[from].push_back(to);
      }
      
      vector<int> ANS(V, -1);
      
      for(int i=0; i<V; ++i){
        return_minimum(Graph, quiet, ANS, i);
      }
      return ANS;
    }
};
