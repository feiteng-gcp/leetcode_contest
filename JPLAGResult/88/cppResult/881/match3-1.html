class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int size = quiet.size();
        map<int, int> answerMap;

        vector<vector<int>> sons(size);
        vector<int> result(size, 0);
        for (int i = 0; i < richer.size(); i++) {
            int x = richer[i][0];
            int y = richer[i][1];
            sons[y].push_back(x);
        }
        for (int i = 0; i < size; i++) {
            if (result[i] == 0) result[i] = dfs(quiet, sons, result, i);
        }
        return result;
    }
    
    int dfs(vector<int>& quiet, vector<vector<int>>& sons, vector<int>& result, int cur) {

        vector<int> son = sons[cur];
        
        int min = cur;
        for (int i = 0; i < son.size(); i++) {
            int next = son[i];
            if (result[next] == 0) {
                result[next] = dfs(quiet, sons, result, son[i]);
            }
            int person = result[next];
            if (quiet[person] < quiet[min]) {
                min = person;
            }
        }
        
        return min;
    } 
};
