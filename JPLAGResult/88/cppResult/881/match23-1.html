class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n = quiet.size();
        vector<vector<int>> rLinks (n);
        for (auto & link : richer)
        {
            rLinks[link[1]].push_back(link[0]);
        }
        
        vector<int> result;
        for (int i = 0; i < n; ++i)
        {
            bool visited[n];
            memset(visited, 0, sizeof(visited));
            tuple<int, int> res = traverse(rLinks, quiet, visited, i, quiet[i]);
            result.push_back(get<1>(res));            
        }
        
        return result;
    }
    
private:
    tuple<int, int> traverse(vector<vector<int>>& rLinks, vector<int>& quiet, bool* visited, int curNode, int curMin)
    {
        int min = curMin;
        int minNode = curNode;
        if (quiet[curNode] < min)
        {
            min = quiet[curNode];
        }
        visited[curNode] = true;
                
        for (int i = 0; i < rLinks[curNode].size(); i++)
        {
            if (!visited[rLinks[curNode][i]])
            {
                int res;
                tuple<int, int> val = traverse(rLinks, quiet, visited, rLinks[curNode][i], min);
                if (get<0>(val) < min)
                {
                    min = get<0>(val);
                    minNode = get<1>(val);
                }
            }
        }
        
        return make_tuple(min, minNode);
    }
};
