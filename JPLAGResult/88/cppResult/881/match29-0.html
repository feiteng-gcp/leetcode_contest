class Solution {
public:
    int dfs(int x,vector<vector<int>> &table,vector<int> & quiet,vector<int> &v){
        int ans=x;
        v[x]=1;
        for(int i=0;i<table[x].size();i++)
        if(!v[table[x][i]]){
            int q = dfs(table[x][i],table,quiet,v);
            if(quiet[q]<quiet[ans])
                ans=q;
        }
        return ans;
    }
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        vector<vector<int>> table;
        for(int i=0;i<quiet.size();i++){
            vector<int> x;
            table.push_back(x);
        }
        for(int i=0;i<richer.size();i++){
            int x=richer[i][0],y=richer[i][1];
            table[y].push_back(x);
        }
        vector<int> answer;
        vector<int> v;
        for(int i=0;i<quiet.size();i++){
            v.clear();
            for(int j=0;j<quiet.size();j++)
                v.push_back(0);
            answer.push_back(dfs(i,table,quiet,v));
        }
        return answer;
    }
};
