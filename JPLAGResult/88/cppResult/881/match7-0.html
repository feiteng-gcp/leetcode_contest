class Solution {
public:
    int dfs(vector<vector<int>>& g,vector<int>& f,vector<int>& q,int u){
        if (f[u]!=-1)
            return f[u];
        f[u]=u;
        for (int i=0;i<g[u].size();i++){
            int v=g[u][i];
            int minfa=dfs(g,f,q,v);
            if (q[minfa]<q[f[u]])
                f[u]=minfa;
        }
        return f[u];
    }
    
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n=quiet.size();
        vector<int> f(n,-1);
        vector<vector<int>> g(n,vector<int>());
        for (auto &v:richer){
            g[v[1]].push_back(v[0]);
        }
        for (int i=0;i<n;i++)
            f[i]=dfs(g,f,quiet,i);
        return f;
    }
};
