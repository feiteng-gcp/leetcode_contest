class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int N = quiet.size();
        vector<unordered_set<int>> d(N);
        vector<int> out(N,-1);
        for(vector<int> x:richer){
            d[x[1]].insert(x[0]);
            
        }
        for(int i = 0;i<N;i++){
            int ans = i;
            stack<int> s;
            for(int x:d[i]){
                s.push(x); 
            }
            while(!s.empty()){
                int cur = s.top();
                s.pop();
                if(quiet[cur]<quiet[ans])
                    ans = cur;
                if(out[cur] != -1){
                    if(quiet[out[cur]] <= quiet[ans]){
                        ans = out[cur];
                    }
                }
                else
                for(int y:d[cur])
                    s.push(y);
            }
            out[i] = ans;
        }
        return out;
    }
};
