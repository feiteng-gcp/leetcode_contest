class Solution {
public:
    int dfs(int x,const vector<int> &quiet, const vector<vector<int>> &es, vector<int> &ans){
        if(ans[x]>=0)
            return ans[x];
        vector<int> e = es[x];
        int s = x;
        for(int i=0;i<e.size();++i){
            int t = dfs(e[i], quiet, es, ans);
            if(quiet[t]<quiet[s])
                s=t;
        }
        ans[x] = s;
        return s;
    }
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n=quiet.size();
        vector<vector<int>> es(n);
        vector<int> ans(n);
        for(int i=0;i<richer.size();++i){
            int x=richer[i][0];
            int y=richer[i][1];
            es[y].push_back(x);
        }
        
        for(int i=0;i<n;++i)
            ans[i]=-1;
        for(int i=0;i<n;++i){
            dfs(i,quiet, es,ans);
        }
        return ans;
        
    }
};
