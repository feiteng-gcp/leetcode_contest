class Solution {
public:
    vector<int> loudAndRich(vector<vector<int> >& richer, vector<int>& quiet)
{
    int N = (int)quiet.size(), len = (int)richer.size();
    vector<vector<bool> > lnk(N, vector<bool>(N, false));
    vector<int> ans(N);
    for(int i = 0; i < N; i++)
    {
        lnk[i][i] = true;
        ans[i] = i;
    }
    for(int i = 0; i < len; i++) lnk[richer[i][1]][richer[i][0]] = true;
    
    for(int k = 0; k < N; k++)
        for(int i = 0; i < N; i++) if(i != k && lnk[i][k])
            for(int j = 0; j < N; j++) if(j != k && lnk[k][j])
                lnk[i][j] = true;
    
    
    for(int i = 0; i < N; i++)
        for(int j = 0; j < N; j++) if(lnk[i][j])
            if(quiet[j] < quiet[ans[i]]) ans[i] = j;
    return ans;
}
};
