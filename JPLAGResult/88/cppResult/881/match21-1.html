class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int len=quiet.size();
        vector<vector<int> > richLevel(len, vector<int>(len, 0));
        for(int i=0;i<richer.size();i++)
            richLevel[richer[i][0]][richer[i][1]]=1;
        for(int i=0;i<len;i++)
        {
            for(int j=0;j<len;j++)
            {
                if(richLevel[i][j]==1)
                {
                    for(int m=0;m<len;m++)
                    {
                        if(richLevel[j][m]==1)
                            richLevel[i][m]=1;
                    }
                }
            }
        }
        vector<int> res;
        for(int i=0;i<len;i++)
        {
            int louder=quiet[i],pos=i;
            for(int j=0;j<len;j++)
            {
                if(richLevel[j][i]==1)
                {
                    if(quiet[j]<louder)
                    {
                        louder=quiet[j];
                        pos=j;
                    }
                }
            }
            res.push_back(pos);
        }
        return res;
    }
};
