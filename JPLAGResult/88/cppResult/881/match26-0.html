const int MAXN = 550;
vector<int> edges[MAXN];
bool vis[MAXN];
class Solution {
    int solve(int n, vector<int>& quiet) {
        memset(vis, false, sizeof(vis));
        vis[n] = true;
        int ret = n;
        queue<int> q; q.push(n);
        while (!q.empty()) {
            int cur = q.front(); q.pop();
            for (int nxt : edges[cur]) {
                if (vis[nxt]) continue;
                vis[nxt] = true; 
                if (quiet[nxt] < quiet[ret]) {
                    ret = nxt;
                }
                q.push(nxt);
            }
        }
        return ret;
    }
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n = quiet.size();
        for (int i = 0; i < n; i++) edges[i].clear();
        for (vector<int> & item : richer) {
            edges[item[1]].push_back(item[0]);
        }
        vector<int> ret;
        for (int i = 0; i < n; i++) {
            ret.push_back(solve(i, quiet));
        }
        return ret;
        
        
        
        
    }
};
