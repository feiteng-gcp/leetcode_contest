class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n = quiet.size();
        vector<int> sol(n, 0);
        vector<vector<int>> c(n);
        for (auto r : richer) {
            int a = r[0];
            int b = r[1];
            c[b].push_back(a);
        }
        
        for (int i = 0; i < n; ++i) {
            // t = all that > i
            queue<int> q;
            q.push(i);
            vector<int> v(n, 0);
            v[i] = 1;
            int min_q = INT_MAX;
            int min_s = i;
            while (!q.empty()) {
                int now = q.front();
                q.pop();
                if (min_q > quiet[now]) {
                    min_q = quiet[now];
                    min_s = now;
                }
                for (int j = 0; j < c[now].size(); ++j) {
                    int next = c[now][j];
                    if (v[next]) continue;
                    v[next] = 1;
                    q.push(next);
                }
            }
            sol[i] = min_s;
            
        }
        
        
        return sol;
    }
};
