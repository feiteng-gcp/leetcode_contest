void dfs(vector<int> &color, vector<vector<int> > &G, int x, int c) {
    if (color[x]==c) return;
    color[x]=c;
    for (int y : G[x]) {
        dfs(color,G,y,c);
    }
}

class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        vector<vector<int> > G(quiet.size());
        for (auto e : richer) {
            G[e[0]].push_back(e[1]);
        }
        vector<pair<int,int> > Q;

        for (int i = 0; i < quiet.size(); ++i) {
            Q.push_back(make_pair(quiet[i],i));
        }
        sort(Q.begin(), Q.end());
        vector<int> res(Q.size(), -1);
        for (int i = Q.size()-1; i>=0; --i) {
            dfs(res,G,Q[i].second,Q[i].second);
        }
        return res;
    }
};
