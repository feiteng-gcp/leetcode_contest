class Solution {
public:
    pair<int,int> findMin(int i, vector<vector<int>>& f, vector<int>& quiet, vector<int>& res, vector<int>& minIndex){
        if (res[i] != INT_MIN) return {res[i], minIndex[i]};
        int minV = quiet[i], index = i;
        for (int j=0; j<quiet.size(); j++){
            if (f[i][j] == 1){
                pair<int, int> t = findMin(j, f, quiet, res, minIndex);
                if (t.first < minV){
                    minV = t.first;
                    index = t.second;
                }
            }
        }
        cout<<"i="<<i<<"index="<<index<<"minV="<<minV<<endl;
        res[i] = minV;
        minIndex[i] = index;
        return {minV, index};
    }
    
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n = quiet.size();
        vector<int> temp;
        if (n == 0) return temp;
        
        vector<int> rows(n, INT_MIN);

        vector<vector<int>> f(n, rows);
        vector<int> res(n, INT_MIN), dp(n, INT_MIN);
        
        for(int i=0; i<richer.size(); i++){
            f[richer[i][1]][richer[i][0]] = 1;
            f[richer[i][0]][richer[i][1]] = 0;
        }
        
        for (int i=0; i<n; i++)
            if (dp[i] == INT_MIN){
                pair<int, int> t = findMin(i, f, quiet, res, dp);
                dp[i] = t.second;
            }
        return dp;
    }
};
