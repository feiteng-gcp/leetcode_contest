class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int N = quiet.size();
        vector<int> res(N, -1);
        vector<vector<int>> mem(N);
        for(auto& vec : richer)
            mem[vec[1]].push_back(vec[0]);
        
        for(int i = 0; i < N; i++)
            dfs(mem, quiet, res, i);
        return res;
    }
    
    int dfs(vector<vector<int>>& mem, vector<int>& quiet, vector<int>& res, int i) {
        if(res[i] != -1)
            return res[i];
        int tmp = i;
        for(int j : mem[i]) {
            int t = dfs(mem, quiet, res, j);
            if(quiet[tmp] > quiet[t])
                tmp = t;
        }
        res[i] = tmp;
        return res[i];
    }
};
