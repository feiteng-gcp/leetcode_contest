class Solution {
private:
    int search(unordered_map<int, vector<int>> &graph, vector<int>& quiet, vector<int> &result, int cur) {
        if (result[cur] != -1)
            return result[cur];
        int least = cur;
        for (auto nei : graph[cur]) {
            int idx = search(graph, quiet, result, nei);
            if (quiet[idx] < quiet[least]) {
                least = idx;
            }
        }
        return result[cur] = least;
    }
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int N = quiet.size();
        vector<int> result(N, -1);

        unordered_map<int, vector<int>> graph;
        for (auto r : richer) {
            graph[r[1]].push_back(r[0]);
        }
        for (int i = 0; i < N; i++) {
            search(graph, quiet, result, i);
        }
        return result;
    }
};
