class Solution {
public:
  vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
    int n = quiet.size();
    vector<int> ans(n);
    for (int i = 0; i < n; ++i) {
      ans[i] = i;
    }
    vector<vector<int>> e(n, vector<int>());
    vector<int> d(n);
    for (auto& p: richer) {
      e[p[0]].push_back(p[1]);
      ++d[p[1]];
    }
    queue<int> q;
    for (int i = 0; i < n; ++i) {
      if (d[i] == 0) {
        q.push(i);
      }
    }
    while (!q.empty()) {
      int head = q.front();
      q.pop();
      for (auto& t: e[head]) {
        if (quiet[ans[head]] < quiet[ans[t]]) {
          ans[t] = ans[head];
        }
        --d[t];
        if (d[t] == 0) {
          q.push(t);
        }
      }
    }
    return ans;
  }
};
