class Solution {
public:
    #define MAX 505
    vector<int> adj[MAX];
    bool seen[ MAX ];
    
    int solve(int source, vector<int> &quiet ){
        memset(seen , 0 , sizeof(seen) );

        queue<int> Q;
        seen[source] = 1;
        Q.push( source );
        int minimum = quiet[source];
        int index = source;
        while( !Q.empty() ){
            int x = Q.front(); Q.pop();
            if( minimum > quiet[x] ){
                minimum = quiet[x];
                index = x;
            }
            
            for( int i = 0 ; i < (int)adj[x].size() ; ++i ){
                if( !seen[ adj[x][i] ]  ){
                    Q.push( adj[x][i] );
                    seen[ adj[x][i] ] = 1;
                }
            }
        }
        return index;
    }
    
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int V = quiet.size();
        
        for( int i = 0 ; i < MAX;  ++i ) adj[i].clear();        
        for( int i = 0 ; i < richer.size() ; ++i ){
            adj[ richer[i][1] ].push_back( richer[i][0] );
        }
        
        vector<int> result(V);
        for( int i = 0 ; i < V ; ++i ){
            result[i] = solve(i,quiet);
        }
        return result;
    }
};
