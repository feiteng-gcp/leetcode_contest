class Solution {
public:
	void visit(int i,vector<bool> &r, vector<vector<int>>& map) {
		if (r[i]) {
			return;
		}
		r[i] = true;
		for (int j = 0;j < map[i].size();j++) {
			visit(map[i][j], r, map);
		}
	}
	vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
		vector<int> re(quiet.size(),0);
		
		vector<vector<int>> map(quiet.size(),vector<int>());
		vector<vector<bool>> r;
		for (int i = 0;i < quiet.size();i++) {
			r.push_back(vector<bool>(quiet.size(),false));
			//r[i][i] = true;
		}
		for (int i = 0;i < richer.size();i++) {			
			map[richer[i][1]].push_back(richer[i][0]);
		}
		for (int i = 0;i < quiet.size();i++) {
			visit(i,r[i], map);
		}
		for (int i = 0;i < quiet.size();i++) {
			int lq=i;
			for (int j = 0;j < quiet.size();j++) {
				if (r[i][j] && quiet[j]<quiet[lq]) {
					lq = j;
				}
			}
			re[i] = lq;
		}
		return re;
	}
};
