class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>& quiet) {
        int n = quiet.size();
        vector<int> res(n);
        vector<unordered_set<int> > neighbors(n);
        for (auto& vec : richer) {
            neighbors[vec[1]].insert(vec[0]);
        }
        vector<pair<int, int> > dp(n, {INT_MAX, INT_MAX});
        for (int i = 0; i < res.size(); i++) {
            res[i] = dfs(neighbors, quiet, dp, i).first;
        }
        return res;
    }
    pair<int, int> dfs (vector<unordered_set<int> >& neighbors, vector<int>& quiet, vector<pair<int, int> >& dp, int p) {
        if (dp[p].second != INT_MAX)
            return dp[p];
        int min_quiet = quiet[p];
        int min_person = p;
        for (auto& neighbor : neighbors[p]) {
            pair<int, int> temp_res = dfs (neighbors, quiet, dp, neighbor);
            if (temp_res.second < min_quiet) {
                min_quiet = temp_res.second;
                min_person = temp_res.first;
            }
        }
        return dp[p] = {min_person, min_quiet};
    }
};
