class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int z=0;
        int mx=0;
        for(int i=0;i<seats.size();++i){
            if(seats[i]==0){
                ++z;
                cout<<i<<"!\n";
            }
            else{
                if(z>mx)
                    mx = z;
                z=0;
            }
        }
        cout<<mx<<endl;
        mx = (mx+1)/2;
        z=0;
        for(int i=0;i<seats.size();++i){
            if(seats[i]==0){
                ++z;
            }
            else{
                break;
            }
        }
        mx=max(z,mx);
        z=0;
        for(int i=seats.size()-1;i>=0;--i){
            if(seats[i]==0)
                ++z;
            else
                break;
        }
        mx=max(z,mx);
        return mx;
    }
};
