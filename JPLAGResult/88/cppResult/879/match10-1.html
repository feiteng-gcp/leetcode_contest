class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int n = seats.size();
        vector<int> dist(n, n);
        int prev = -n;
        for (int i = 0; i < n; i++) {
            if (seats[i] == 1) {
                prev = i;

                dist[i] = -1;
            } else {
                dist[i] = min(dist[i], i - prev);
            }
        }
        prev = n * 2;
        for (int i = n - 1; i >= 0; i--) {
            if (seats[i] == 1) {
                prev = i;
                dist[i] = -1;

            } else {
                dist[i] = min(dist[i], prev - i);
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (dist[i] > res) {
                res = dist[i];
            }
        }
        return res;
    }
};
