class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        
        int maxx = 0;
        int maxx1 = 0;
        int maxx2 = 0;
        for (int i=0;i<seats.size();i++)
            if (seats[i]==1)
                break;
            else
                maxx1++;
        for (int i=seats.size()-1;i>=0;i--)
            if (seats[i]==1)
                break;
            else
                maxx2++;
        
        int loc = 0;
        for (int i=0;i<seats.size();i++) {
            if (seats[i]==0) {
                loc++;    
            } else {
                maxx = max(maxx, loc);
                loc=0;
            }
        }
        maxx = max(maxx,loc);
        maxx = (maxx + 1)/2;
        return max(maxx1,max(maxx2,maxx));
    }
};
