class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int n=seats.size();
        vector<int> V;
        for(int i=0; i<n; i++) {
            if (seats[i]==1) V.push_back(i+1);
        }
        int ans=0;
        for(int i=0; i<V.size()-1; i++) {
            cout<<V[i+1]-V[i]<<endl;
            ans=max(ans, (V[i+1]-V[i])/2);
        }
        if (V.size()>0) ans=max(ans, V[0]-1);
        if (V.size()>0) ans=max(ans, n-V.back());
        return ans;
    }
};
