class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int len = (int)seats.size(), last = -1, ans = 1, i = 0;
    for(i = 0; i < len; i++)
        if(seats[i] == 1)
        {
            if(last == -1) ans = max(ans, i);
            else ans = max((i - last) / 2, ans);
            last = i;
        }
    ans = max(ans, len - 1 - last);
    return ans;
    }
};
