class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        vector<int> p;
        int n = seats.size();
        for (int i = 0; i < n; ++i) {
            if (seats[i]) p.push_back(i);
        }
        int res = 0;
        if (p.size() == 1) { res = max(p[0], n - 1 - p[0]); }
        else {
            int pre = p[0];
            for (int i = 1; i < p.size(); ++i) {
                res = max(res, (p[i] - pre) / 2);
                pre = p[i];
            }
            res = max(res, max(p[0], n - 1 - p[p.size() - 1]));
        }
        return res;
    }
};
