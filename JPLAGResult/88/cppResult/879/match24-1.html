const int maxn = 20000+10;
int pre[maxn];
int after[maxn];
class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        memset(pre,-1,sizeof(pre));
        memset(after,-1,sizeof(after));

        int len = seats.size();
        int pos=-1;
        for(int i=0;i<len;i++){
            if(seats[i]==1){
                pos=i;
            }else{
                pre[i]=pos;
            }
        }

        pos = -1;
        for(int i=len-1;i>=0;i--){
            if(seats[i]==1){
                pos=i;
            }else{
                after[i]=pos;
            }
        }
        int ans = 0;
        for(int i=0;i<len;i++){
            if(seats[i]==0){
                int tmp = 30000;
                if(pre[i]!=-1){
                    tmp = min(tmp,i-pre[i]);
                }
                if(after[i]!=-1){
                    tmp = min(tmp,after[i]-i);
                }
                ans = max(ans,tmp);
            }
        }
        return ans;
    }
};
