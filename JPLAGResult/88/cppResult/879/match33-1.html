class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int n = seats.size();
        int maxi = 0;
        int index = 0;
        for( int i = 0 ; i < n ; ++i ){
            if( seats[i] == 0 ){
            	int j = i + 1;
                int cnt = 1;
            	while( j < n && seats[j] == 0 ){
                    j++; cnt++;
            	}
                
                if( i == 0 ){
                    maxi = cnt;
                    index = 0;
                }else if( j == n ){
                    if( cnt > maxi ){
                        maxi = cnt;
                        index = j - 1;
                    }
                }else{
                    if( cnt & 1 ){
                        int dist = (cnt + 1)/2;
                        if( dist > maxi ){
                            maxi = dist;
                            index = i + cnt/2;
                        }
                    }else{
                        int dist = cnt/2;
                        if( dist > maxi ){
                            maxi = dist;
                            index = i + cnt/2;
                        }
                    }
                }
                
                i = j - 1;
            }
        }
        return maxi;
    }
};
