class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int n = seats.size();
        int ret = 0;
        for (int i = 0, j; i < n; ) {
            if (seats[i]) ++i;
            else {
                for (j = i; j < n && !seats[j]; ++j) ;
                if (i < 1) ret = max(ret, j - 0);
                else if (j >= n) ret = max(ret, n - i);
                else {
                    int d = j - i;
                    ret = max(ret, (d + 1) / 2);
                }
                i = j;
            }
        }
        return ret;
    }
};
