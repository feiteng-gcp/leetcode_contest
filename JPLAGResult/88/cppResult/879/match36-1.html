class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int minpos=0,minlength=0;
        int len=seats.size();
        for(int i=0;i<len;i++)
        {
            if(seats[i]==1)
                continue;
            int left=0,right=0,pos=i;
            while(seats[pos]!=1 && pos>=0)
            {
                pos--;
                left++;
            }
            pos=i;
            while(seats[pos]!=1 && pos<len)
            {
                pos++;
                right++;
            }
            if(i!=len-1&&i!=0)
            {
                if(right>=left)
                {
                    if(left>minlength)
                        minlength=left;
                }
                if(left>=right)
                {
                    if(right>minlength)
                        minlength=right;
                }
            }
            else if(i==len-1)
            {
                if(left>minlength)
                    minlength=left;
            }
            else if(i==0)
            {
                if(right>minlength)
                    minlength=right;
            }
        }
        return minlength;
    }
};
