int DR[22222];

int DL[22222];

class Solution {
public:
    int maxDistToClosest(vector<int>& s) {
        int last = -(1<<30);
        for (int i = 0; i < s.size(); ++i) {
            if (s[i]==1) last = i;
            else DL[i]=i-last;
        }
        last = 1<<30;
        for (int i = s.size()-1; i>=0; --i) {
            if (s[i]==1) last = i;
            else DR[i]=last-i;
        }
        int best = 0;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i]==0) {
                best = max(best, min(DL[i],DR[i]));
            }
        }
        return best;
    }
};
