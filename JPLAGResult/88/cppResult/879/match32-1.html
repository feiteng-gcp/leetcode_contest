class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int left_most_one = -1;
        vector<int> left_distance(seats.size(), INT_MAX);
        for (int i = 0; i < seats.size(); i++) {
            if (seats[i] == 1)
                left_most_one = i;
            else if (seats[i] == 0) {
                if (left_most_one != -1)
                    left_distance[i] = i - left_most_one;
            }
        }
        int right_most_one = -1;
        vector<int> right_distance(seats.size(), INT_MAX);
        for (int i = seats.size() - 1; i >= 0; i--) {
            if (seats[i] == 1)
                right_most_one = i;
            else if (seats[i] == 0) {
                if (right_most_one != -1)
                    right_distance[i] = right_most_one - i;
            }
        }
        int res = 0;
        for (int i = 0; i < seats.size(); i++) {
            if (seats[i] == 0) {
                res = max (res, min (left_distance[i], right_distance[i]));
            }
        }
        return res;
    }
};
