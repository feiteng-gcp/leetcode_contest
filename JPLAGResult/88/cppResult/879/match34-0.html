class Solution {
public:
	int maxDistToClosest(vector<int>& seats) {
		int last = -1;
		int max = -1;
		int n = seats.size();
		for (int i = 0; i < n; ++i)
		{
			if (seats[i] == 1)
			{
				if (last == -1)
				{
					max = i ;
				}
				else
				{
					int cur = (i - last) / 2;
					if (cur > max)
					{
						max = cur;
					}
				}

				last = i;
			}
		}

		if (n - last - 1 > max)
		{
			max = n - last - 1;
		}

		return max;
	}
};
