class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int n = (int)seats.size();
        int d[n];
        fill(d, d + n, 0);
        int cur = 0;
        int x = 0;
        while (seats[x] == 0) {
            cur++;
            x++;
        }
        for (int i = 0; i < x; i++) {
            d[i] = cur;
            cur--;
        }
        for (int i = x; i < n; i++) {
            if (seats[i] == 0) {
                cur++;
            } else {
                cur = 0;
            }
            d[i] = cur;
        }
        cur = 0;
        for (int i = n - 1 - d[n - 1]; i > -1; i--) {
            if (seats[i] == 0) {
                cur++;
            } else {
                cur = 0;
            }
            d[i] = min(d[i], cur);
        }
        int ans = 0;
        for (int x : d) {
            ans = max(ans, x);
        }
        return ans;
    }
};
