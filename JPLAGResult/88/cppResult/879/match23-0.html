class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int sum=0;
        int has=0;
        int now=0;
        for(int i=0;i<seats.size();i++){
            now++;
            if(seats[i]){
                if(has){
                    sum=max(sum,now/2);
                    now=0;
                }else{
                    sum=max(sum,now-1);
                    has=1;
                    now=0;
                }
            }
        }
        sum=max(sum,now);
        return sum;
    }
};
