class Solution {
public:
  int maxDistToClosest(vector<int>& seats) {
    int maxZero = 0;
    bool endEmpty = false;
    int sum = 0;
    for (int i = 0; i < seats.size(); i++) {
      if (seats[i] == 0) {
        if ((sum == 0 && i == 0) || (i == seats.size() - 1)) {
          endEmpty = true;
        }
        sum++;
        if (endEmpty) {
          maxZero = max(sum, maxZero);
        } else {
          maxZero = max((sum + 1) / 2, maxZero);
        }
      } else {
        sum = 0;
                  endEmpty = false;
      }
    }
    return maxZero;    
  }
};
