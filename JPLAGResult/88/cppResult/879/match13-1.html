class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int size = seats.size();
        if (size == 0) return 0;

        vector<int> pos;
        for (int i = 0; i < size; i++) {
            if (seats[i] == 1) pos.push_back(i);
        }
        
        int result = 0;
        for (int i = 0; i < (pos.size()-1); i++) {
            result = max(result, (pos[i+1] - pos[i])/2);
        }
        
        // 考虑起始和结束
        result = max(result, pos[0]);
        result = max(result, size - pos[pos.size()-1] - 1);
        
        return result;
    }
};
