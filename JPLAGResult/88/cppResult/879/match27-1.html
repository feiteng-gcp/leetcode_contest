class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int n = seats.size();
        int last = -n;
        vector<int> a(n, -n);

        for (int i = 0; i < n; ++i) {
            if (seats[i] == 1) {
                last = i;
            } else {
                a[i] = last;
            }
        }
        last = n * 2;

        int answer = 0;
        for (int i = n - 1; i >= 0; --i) {
            if (seats[i] == 1) {
                last = i;
            } else {
                int temp = min(i - a[i], last - i);
                if (answer < temp) {
                    answer = temp;
                }
                
            }
            
        }
        return answer;
        
            
        
    }
};
