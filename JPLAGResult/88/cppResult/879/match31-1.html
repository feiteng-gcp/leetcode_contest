class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        vector<int> l(seats.size(), INT_MAX), r(seats.size(), INT_MAX);
        int count =100000;
        for(int i=0; i<seats.size(); i++)
        {
            count++;
            if(seats[i])
                count=0;
            else
               l[i]=count;
        }
        count =100000;
        for(int i=seats.size()-1; i>=0; i--)
        {
            count++;
            if(seats[i])
                count=0;
            else
                r[i]=count;
        }
        int mx=-1;
        for(int i=0; i<seats.size(); i++)
        {
            int mi=min(l[i],r[i]);
            if(mi<30000)
                mx=max(mi,mx);
        }
        return mx;
    }
};
