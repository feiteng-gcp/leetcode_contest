class Solution {
public:
	int maxDistToClosest(vector<int>& seats) {
		vector<int>x;
		int n = seats.size();
		for (int i = 0; i<seats.size(); i++) {
			if (seats[i] == 1)x.push_back(i);
		}
		int maxx = 0;
		if (x[0] != 0)maxx = x[0] - 0;
		for (int i = 0; i<x.size() - 1; i++) {
			maxx = max(maxx, (x[i + 1] - x[i]) / 2);
		}
		if (x[x.size() - 1] != n - 1)maxx = max(maxx, n-1 - x[x.size() - 1]);
		return maxx;
	}
};
