class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        int len=seats.size();
        vector<int> notEmpty;
        for(int i=0;i<len;++i)
            if(seats[i]==1)
                notEmpty.push_back(i);
        int n=notEmpty.size(), maxdist=0;
        for(int i=0, j=0;i<len && j<=n;++i){
            if(seats[i]==1){
                ++j;
                continue;
            }
            if(j==n)
                maxdist=max(maxdist, abs(i-notEmpty[n-1]));
            else if(j==0)
                maxdist=max(maxdist, abs(i-notEmpty[0]));
            else
                maxdist=max(maxdist, min(abs(i-notEmpty[j-1]), abs(i-notEmpty[j])));
        }
        return maxdist;
    }
};
