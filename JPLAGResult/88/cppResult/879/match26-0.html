class Solution {
    typedef long long LL;
public:
    int maxDistToClosest(vector<int>& seats) {
        vector<LL> pos;
        pos.push_back(INT_MIN);
        for (int i = 0; i < seats.size(); i++) {
            if (seats[i] == 1) pos.push_back(i);
        }
        pos.push_back(INT_MAX);
        int ret = 0;
        for (int i = 0; i < seats.size(); i++) {
            if (seats[i] == 1) continue;
            auto p = lower_bound(pos.begin(), pos.end(), i);
            LL cur = *p - i;
            p--;
            cur = min(cur, i- *p);
            ret = max(ret, (int)cur);
        }
        return ret;
        
    }
};
