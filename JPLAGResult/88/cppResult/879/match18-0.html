class Solution {
public:
    int maxDistToClosest(vector<int>& seats) {
        stack<int> s;
        int res=0;
        int len=seats.size();
        for (int i=0;i<len;i++){
            if (seats[i]){
                if (s.empty())
                    res=max(res,i);
                else
                    res=max(res,(i-s.top())/2);
                s.push(i);
            }
        } 
        res=max(res,len-1-s.top());
        return res;        
    }
};
