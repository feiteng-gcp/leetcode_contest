typedef pair<int,int> per;
#define mp(x,y) make_pair(x,y)
typedef long long ll;
const int N = 100005;
const int M = 2001;
class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
		string s;
        int n=S.size();
		int pre=0;
		for(int i=n-1;i>=0;i--)
		{
			pre=(pre+shifts[i])%26;
			int x=(S[i]-'a'+pre)%26;
			string str;
			str+=('a'+x);
			s=str+s;
		}
		return s;
    }
};
