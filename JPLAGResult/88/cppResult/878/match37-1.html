class Solution {
public:
    char shift(char ch, int x) {
        return 'a' + (ch - 'a' + x) % 26;
    }
    
    string shiftingLetters(string S, vector<int>& shifts) {
        int n = shifts.size();
        vector<int> sum(n + 1);
        sum[n] = 0;
        for (int i = n; i > 0; i--) {

            sum[i - 1] = (sum[i] + shifts[i - 1]) % 26;
        }
        for (int i = 0; i < n; i++) {
            S[i] = shift(S[i], sum[i]);
        }
        return S;
    }
};
