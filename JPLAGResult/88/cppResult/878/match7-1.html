class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
        vector<int> sumy(S.size(),0),out;
        string res;
        for (int i=S.size()-1;i>=0;i--){
            if (i==S.size()-1)
                sumy[i]=shifts[i]%26;
            else
                sumy[i]=(sumy[i+1]+shifts[i])%26;
        }
        for (int i=0;i<S.size();i++){
            res+='a'+((S[i]-'a')+sumy[i])%26;
        }
        
        return res;
    }
};
