class Solution {
public:
    char shiftOne(char s, int x) {
        //cout<<"s="<<s<<", x="<<x<<", "<<(s - 'a' + x) % 26<<endl;
        return 'a' + (s - 'a' + x) % 26;
    }
    string shiftingLetters(string S, vector<int>& shifts) {
        int n1 = shifts.size(), n2 = S.size(), s = 0;
        int n = min(n1, n2);
        vector<int> num(n, 0);
        for (int i = 0; i < n; i++) {
            s += shifts[n - 1 - i] % 26;
            //cout<<"i="<<i<<", s="<<s<<endl;
            num[i] = s;
        }
        for (int i = 0; i < n; i++) {
            S[i] = shiftOne(S[i], num[n - 1 - i]);
        }
        return S;
    }
};
