class Solution {
public:
  string shiftingLetters(string S, vector<int>& shifts) {
    int n = S.length();
    for (int i = n - 2; i >= 0; --i) {
      shifts[i] += shifts[i + 1];
      shifts[i] %= 26;
    }
    for (int i = 0; i < n; ++i) {
      int c = S[i];
      c += shifts[i] % 26;
      if (c > 'z') {
        c -= 26;
      }
      S[i] = c;
    }
    return S;
  }
};
