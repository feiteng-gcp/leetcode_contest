class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
        int n=shifts.size();
        int total_s=0;
        for (int i=n-1; i>=0; i--) {
                  total_s = (total_s + shifts[i]) % 26;
                  S[i]= ((S[i]-'a'+total_s) % 26) + 'a';
        }
        return S;
        
    }
};
