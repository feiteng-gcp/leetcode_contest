class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
        int s = 0;
        for (int i = shifts.size() - 1; i >= S.size(); i--) {
            s = (s + shifts[i]) % 26;
        }
        for (int i = S.size() - 1; i >= 0; i--) {
            s = (s + shifts[i]) % 26;
            S[i] = 'a' + (((S[i] - 'a') + s) % 26);
        }
        return S;
    }
};
