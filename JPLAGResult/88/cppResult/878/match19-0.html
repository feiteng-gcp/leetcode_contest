class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
        for (int i = shifts.size() - 1; i > 0 ; --i) {
            shifts[i-1] = (shifts[i-1] + shifts[i]) % 26;
        }
        for (int i = 0; i < S.size(); i++) {
            S[i] = 'a' + (((S[i] - 'a') + shifts[i]) % 26);
        }
        return S;
    }
};
