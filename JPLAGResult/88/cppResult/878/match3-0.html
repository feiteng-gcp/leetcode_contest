class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
      int n = S.size();
      int shift = 0;
      for (int i = n-1; i >= 0; --i) {
        shift = (shift + shifts[i]) % 26;
        int pos = S[i] - 'a';
        pos = (pos + shift) % 26;
        S[i] = 'a' + pos;
      }
      return S;
    }
};
