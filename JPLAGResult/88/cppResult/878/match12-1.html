class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
        int n = S.length();
        int cnt = 0;
        for (int i = n - 1; i >= 0; i--) {
            cnt = (cnt + shifts[i]) % 26;
            S[i] = (char) ((S[i] - 'a' + cnt) % 26 + 'a');
        }
        return S;
    }
};
