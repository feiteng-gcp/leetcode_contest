class Solution {
public:
    string shiftingLetters(string S, vector<int>& shifts) {
        int index = 0;
        int c = 0;
        for (int i = shifts.size() - 1; i >= 0; --i) {
            c += (shifts[i] % 26);
            shifts[i] = (c % 26);
        }
        
        for (char& c: S) {
            c = (c - 'a' + shifts[index++]) % 26 + 'a';
        }
        return S;
    }
};
