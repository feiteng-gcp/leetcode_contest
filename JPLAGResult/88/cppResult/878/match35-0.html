class Solution {
public:
    
    char shift(char c, long long i) {
        long long x = c - 'a';
        x = (x + i)%26;
        if (x<0)
            x+=26;
        //cout << x << endl;
        return x + 'a';
    }
    
    string shiftingLetters(string S, vector<int>& shifts) {
        vector<long long> sh(S.size(),0);
        
        long long sum = 0;
        for (int i=0;i<S.size();i++)
            sum += shifts[i];
        
        for (int i=0;i<S.size();i++) {
            sh[i] = sum;
            sum-= shifts[i];
        }

        string ans;
        for (int i=0;i<S.size();i++)
            ans += shift(S[i], sh[i]);
        return ans;
    }
};
