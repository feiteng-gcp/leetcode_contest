class Solution {
public:
    typedef long long LL;
int rectangleArea(vector<vector<int> >& rectangles)
{
    const int mod = 1e9 + 7;
    vector<vector<int> >& r = rectangles;
    int len = (int)r.size();
    map<int, int> tabUD, tabLR;
    for(int i = 0; i < len; i++)
    {
        tabUD[r[i][1]] = 0; tabUD[r[i][3]] = 0;
        tabLR[r[i][0]] = 0; tabLR[r[i][2]] = 0;
    }
    vector<int> ud, lr;
    for(auto p = tabUD.begin(); p != tabUD.end(); p++)
    {
        p->second = (int)ud.size();
        ud.push_back(p->first);
    }
    for(auto p = tabLR.begin(); p != tabLR.end(); p++)
    {
        p->second = (int)lr.size();
        lr.push_back(p->first);
    }
    int udL = (int)ud.size(), lrL = (int)lr.size();
    vector<vector<int> > cov(udL - 1, vector<int>(lrL));
    for(int i = 0; i < len; i++)
    {
        int did = tabUD[r[i][1]], uid = tabUD[r[i][3]];
        int lid = tabLR[r[i][0]], rid = tabLR[r[i][2]];
        for(int j = did; j < uid; j++)
        {
            cov[j][lid]++;
            cov[j][rid]--;
        }
    }
    int ans = 0;
    for(int i = 0; i < udL - 1; i++)
    {
        LL h = ud[i + 1] - ud[i];
        for(int j = 0; j < lrL - 1; j++)
        {
            if(j) cov[i][j] += cov[i][j - 1];
            if(cov[i][j]) ans = (ans + h * (lr[j + 1] - lr[j])) % mod;
        }
    }
    return ans;
}
};
