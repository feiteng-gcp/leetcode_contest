int D[500][500];
int mod = 1000*1000*1000+7;

class Solution {
public:
    int rectangleArea(vector<vector<int>>& R) {
        vector<int> X;
        vector<int> Y;
        for (int i = 0; i < R.size(); ++i) {
            X.push_back(R[i][0]);
            X.push_back(R[i][2]);
            Y.push_back(R[i][1]);
            Y.push_back(R[i][3]);
        }
        sort(X.begin(), X.end()); X.erase(unique(X.begin(),X.end()), X.end());
        sort(Y.begin(), Y.end()); Y.erase(unique(Y.begin(),Y.end()), Y.end());
        memset(D,0,sizeof(D));
        for (int i = 0; i < R.size(); ++i) {
            int x1 = lower_bound(X.begin(),X.end(),R[i][0])-X.begin();
            int x2 = lower_bound(X.begin(),X.end(),R[i][2])-X.begin();
            int y1 = lower_bound(Y.begin(),Y.end(),R[i][1])-Y.begin();
            int y2 = lower_bound(Y.begin(),Y.end(),R[i][3])-Y.begin();
            
            for (int x = x1; x<x2; ++x) {
                for (int y = y1; y < y2; ++y) {
                    D[x][y]=1;
                }
            }
        }
        long long res = 0;
        for (int i = 0; i < X.size()-1; ++i) {
            for (int j = 0; j < Y.size()-1; ++j) {
                if (D[i][j]) {
                    res += (long long)(X[i+1]-X[i]) * (long long) (Y[j+1]-Y[j]);
                    res %=mod;
                }
            }
        }
        return res%mod;
    }
};
