class Solution {
public:
    int M = 1000000007;
    long long mul(long long x, long long y) {
        return x * y % M;
    }
    void make(vector<vector<bool> > &s, int x1, int y1, int x2, int y2) {
        if (x1  > x2) {
            int t = x1;
            x1 = x2;
            x2 = t;
        }
        if (y1 > y2) {
            int tt = y1;
            y1 = y2;
            y2 = tt;

        }
        for (int i = x1; i < x2; ++i) {
            for (int j = y1; j < y2; ++j) {
                s[i][j] = true;
            }
        }
    }
    int rectangleArea(vector<vector<int>>& a) {
        map<int, int> have;
        int n = a.size();
        for (int i = 0; i < n; ++i) {
            ++have[a[i][0]];
            ++have[a[i][1]];
            ++have[a[i][2]];
            ++have[a[i][3]];
        }
        vector<int> p;
        for (map<int, int>::iterator t = have.begin(); t != have.end(); ++t) {
            t->second = p.size();
            p.push_back(t->first);
            
        }
        int m = p.size();
        vector<vector<bool> > s;
        s.resize(m + 1);
        for (int i = 0; i <= m; ++i) {
            s[i].resize(m + 1, false);
        }
        for (int i = 0; i < n; ++i) {
            make(s, have[a[i][0]], have[a[i][1]], have[a[i][2]], have[a[i][3]]);
        }

        long long answer = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < m; ++j) {
                if (s[i][j]) {
                    answer += mul(p[i + 1] - p[i], p[j + 1] - p[j]);
                }
            }
        }
        return answer % M;
        
        
    }
};
