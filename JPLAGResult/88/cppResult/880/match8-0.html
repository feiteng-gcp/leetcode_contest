class Solution {
public:
    int rectangleArea(vector<vector<int>>& rectangles) {
        int mod = 1e9+7;
        int n = rectangles.size();
        int res = 0;
        vector<int> xs, ys;
        for(auto& vec: rectangles) {
            xs.push_back(vec[0]);
            xs.push_back(vec[2]);
            ys.push_back(vec[1]);
            ys.push_back(vec[3]);
        }
        sort(xs.begin(), xs.end());
        sort(ys.begin(), ys.end());
        auto it = unique(xs.begin(), xs.end());
        int num1 = it-xs.begin();
        it = unique(ys.begin(), ys.end());
        int num2 = it-ys.begin();

        vector<vector<bool>> mem(num1, vector<bool>(num2, false));
        for(auto& vec : rectangles) {
            int Lx = lower_bound(xs.begin(), xs.begin()+num1, vec[0])-xs.begin();
            int Rx = lower_bound(xs.begin(), xs.begin()+num1, vec[2])-xs.begin();
            int Ly = lower_bound(ys.begin(), ys.begin()+num2, vec[1])-ys.begin();
            int Ry = lower_bound(ys.begin(), ys.begin()+num2, vec[3])-ys.begin();
            for(int i = Lx; i < Rx; i++)
                for(int j = Ly; j < Ry; j++)
                    mem[i][j] = true;
        }
        for(int i = 0; i < num1; i++)
            for(int j = 0; j < num2; j++)
                if(mem[i][j])
                    res = (res + ((long long)xs[i+1]-xs[i])*(ys[j+1]-ys[j]))%mod;
        return res;
    }
};
