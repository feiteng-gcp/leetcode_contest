class Solution {
public:
    int rectangleArea(vector<vector<int>>& rects) {
        int n = rects.size();
        vector<int> xx, yy;
        for (const auto& r : rects) {
            xx.push_back(r[0]);
            yy.push_back(r[1]);
            xx.push_back(r[2]);
            yy.push_back(r[3]);
        }
        sort(xx.begin(), xx.end());
        sort(yy.begin(), yy.end());
        int nx = unique(xx.begin(), xx.end()) - xx.begin();
        int ny = unique(yy.begin(), yy.end()) - yy.begin();
        // cout << nx << " " << ny << endl;
        vector<vector<bool>> g(nx + 1, vector<bool>(ny + 1));

        for (const auto& r : rects) {
            // cout << r[0] << " " << r[1] << " " << r[2] << " " << r[3] << endl;
            int x1 = lower_bound(xx.begin(), xx.begin() + nx, r[0]) - xx.begin();
            int y1 = lower_bound(yy.begin(), yy.begin() + ny, r[1]) - yy.begin();
            int x2 = lower_bound(xx.begin(), xx.begin() + nx, r[2]) - xx.begin();
            int y2 = lower_bound(yy.begin(), yy.begin() + ny, r[3]) - yy.begin();
            // cout << x1 << " " << y1 << " " << x2 << " " << y2 << endl;
            for (int x = x1; x < x2; ++x) for (int y = y1; y < y2; ++y) g[x][y] = true;
        }
        long long ret = 0;
        for (int x = 0; x < nx; ++x) for (int y = 0; y < ny; ++y) if (g[x][y]) {
            ret += ((long long)xx[x + 1] - xx[x]) * ((long long)yy[y + 1] - yy[y]) % 1000000007;
        }
        return ret % 1000000007;
    }
};
