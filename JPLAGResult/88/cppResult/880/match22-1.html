class Solution {
public:
#define pb push_back
#define mp make_pair
#define CLR(a) memset(a, 0, sizeof(a))
#define DBG(x) cout<<(#x)<<"="<<x<<endl
#define FOR(i, a, b)  for(int i=(a); i<(b); i++)
#define REP(i, a, b)  for(int i=(a); i<=(b); i++)
#define DOWN(i, a, b) for(int i=(a); i>=(b); i--)

typedef long long ll;
typedef unsigned long long ull;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;

const double eps = 1e-8;
const int INF = 0x3f3f3f3f;
const ll LL_INF = 0x3f3f3f3f3f3f3f3f;
const ll mod = 1e9+7;


int n, cnt, amt;
ll X1, X2, Y1, Y2;
ll x[100000];
ll qL, qR, num;
ll cover[100000];
ll len[100000];
ll ans;

struct node{
	ll l, r;
	ll y;
	int p;

	bool operator < (const node& rhs) const {
		return y<rhs.y;
	}
}Line[100000];

void Up(int o, int L, int R) {
	if (cover[o]>0) len[o]=x[R]-x[L];
	else if (L+1==R) len[o]=0; 
	else len[o]=len[o*2]+len[o*2+1];
}

void Add(int o, int L, int R) {
	if (qL<=L&&R<=qR) {
		cover[o]+=num;
		Up(o, L, R);
		return;
	}
	int M=L+(R-L)/2;
	if (qL<M) Add(o*2, L, M);
	if (qR>M)  Add(o*2+1, M, R);
	Up(o, L, R);
}


int rectangleArea(vector<vector<int>>& rectangles) {
        n=rectangles.size();
		CLR(len);
		CLR(cover);
		cnt=amt=0;
		ans=0;
		FOR(i, 0, n) {
			X1=rectangles[i][0];
            Y1=rectangles[i][1];
            X2=rectangles[i][2];
            Y2=rectangles[i][3];
			Line[cnt++]={X1, X2, Y1, 1};
			Line[cnt++]={X1, X2, Y2, -1};
			x[amt++]=X1;
			x[amt++]=X2;
		}
		//cnt is the number of line
		sort(Line, Line+cnt);
		//amt is the number of the point
		sort(x, x+amt);
		amt=unique(x, x+amt) - x;
		FOR(i, 0, n*2-1) {
			qL=lower_bound(x, x+amt, Line[i].l)-x;
			qR=lower_bound(x, x+amt, Line[i].r)-x;
			num=Line[i].p;	
			Add(1, 0, amt-1);
			ll tmp= ((Line[i+1].y-Line[i].y)%mod * len[1]%mod) % mod;
            ans=(ans+tmp) % mod;
		}
        return ans;
}
};
