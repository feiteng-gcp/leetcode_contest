class Solution {
public:
  int rectangleArea(vector<vector<int>>& rectangles) {
    unordered_map<int, int> mx, my;
    vector<int> vx, vy;
    for (auto& rec: rectangles) {
      vx.push_back(rec[0]);
      vy.push_back(rec[1]);
      vx.push_back(rec[2]);
      vy.push_back(rec[3]);
    }
    sort(vx.begin(), vx.end());
    sort(vy.begin(), vy.end());
    vx.resize(distance(vx.begin(), unique(vx.begin(), vx.end())));
    vy.resize(distance(vy.begin(), unique(vy.begin(), vy.end())));
    for (int i = 0; i < vx.size(); ++i) { mx[vx[i]] = i; }
    for (int i = 0; i < vy.size(); ++i) { my[vy[i]] = i; }
    vector<vector<bool>> fill(vx.size(), vector<bool>(vy.size(), false));
    for (auto& rec: rectangles) {
      int x1 = mx[rec[0]];
      int y1 = my[rec[1]];
      int x2 = mx[rec[2]];
      int y2 = my[rec[3]];
      for (int i = x1; i < x2; ++i) {
        for (int j = y1; j < y2; ++j) {
          fill[i][j] = true;
        }
      }
    }
    long long ans = 0;
    for (int i = 0; i < vx.size() - 1; ++i) {
      for (int j = 0; j < vy.size() - 1; ++j) {
        if (fill[i][j]) {
          ans = (ans + ((long long)(vx[i + 1] - vx[i]) * (long long)(vy[j + 1] - vy[j]))) % 1000000007;
        }
      }
    }
    return ans;
  }
};
