class Solution {
public:
    int rectangleArea(vector<vector<int>>& v) {
        vector<int>x;
        vector<int>y;
        
        set<int>Sx;
        set<int>Sy;
        
        for(int i=0;i<v.size();i++){
            Sx.insert(v[i][0]);
            Sx.insert(v[i][2]);
            Sy.insert(v[i][1]);
            Sy.insert(v[i][3]);
        }
        
        x=vector<int>(Sx.begin(),Sx.end());
        y=vector<int>(Sy.begin(),Sy.end());
        
        bool visited[x.size()][y.size()];
        memset(visited,0,sizeof(visited));
        
        map<int,int>idx;
        map<int,int>idy;
        for(int i=0;i<x.size();i++)idx[x[i]]=i;
        for(int i=0;i<y.size();i++)idy[y[i]]=i;
        
        

        for(int i=0;i<v.size();i++){
            int x1=idx[v[i][0]];
            int y1=idy[v[i][1]];
            int x2=idx[v[i][2]];
            int y2=idy[v[i][3]];
            
            for(int j=x1;j<x2;j++)
                for(int k=y1;k<y2;k++)
                    visited[j][k]=1;
        }
        
        long long dev=0;
        
        for(int i=0;i+1<x.size();i++)
            for(int j=0;j+1<y.size();j++){
                if(visited[i][j]){
                    long long p=(x[i+1]-x[i]);
                    long long q=(y[j+1]-y[j]);
                    long long h=(p*q)%1000000007;
                    
                    dev=(dev+ h);
                    dev=dev%1000000007;
                }
                
            }
        
        
        return (int)dev;
        
    }
};
