class Solution {
public:
   struct P{
        int x;
        int yd;
        int yu;
        bool f;
        P() {}
        P(int _x, int _yd, int _yu, bool _f) : x(_x), yd(_yd), yu(_yu), f(_f) {}
        bool operator < (const P& p) const {
            return x < p.x;
        }
    };
    long long int calyaxis(vector<pair<int, int>> &y) {
        long long int l = 0;
        int s = 0;
        int prey = y[0].first;
        for (int i = 0; i < y.size(); ++i) {
//                cout << i << " y:" << y[i].first << " f:" << y[i].second << endl;
            if (y[i].second) {
                // add
                if (s) l += y[i].first - prey;
                ++s;

                prey = y[i].first;
            } else {
                // del
                --s;
                l += y[i].first - prey;

                prey = y[i].first;
            }
        }
        return l;
    }
    int rectangleArea(vector<vector<int>>& rectangles) {
        vector<P> v;
        for (auto r : rectangles) {
            v.push_back(P(r[0], r[1], r[3], true));
            v.push_back(P(r[2], r[1], r[3], false));
        }

        sort(v.begin(), v.end());
        vector<pair<int, int>> y;
        // at least 1 rec
        int prex = v[0].x;
        long long int area = 0;
        long long int l = 0;
        for (int i = 0; i < v.size(); ++i) {
            P &p = v[i];
//            cout << endl<<p.f << " " << p.x << " [" << p.yd << " " << p.yu << endl;
            if (p.f) {
                // add
                area += l * (p.x - prex);
                area %= 1000000007;

                y.push_back(make_pair(p.yd, 1));
                y.push_back(make_pair(p.yu, 0));
                sort(y.begin(), y.end());

            } else {
                // del
                area += l * (p.x - prex);
                area %= 1000000007;
                int a = 1;

                int b = 1;
                for (int i = 0; i < y.size(); ++i) {
                    if (a && y[i].first == p.yd && y[i].second == 1) {
                        y.erase(y.begin() + i);
                        --i;
                        a = 0;
                    }
                    else if (b && y[i].first == p.yu && y[i].second == 0) {
                        y.erase(y.begin() + i);
                        --i;
                        b = 0;
                    }
                }

            }

            l = calyaxis(y);
            prex = p.x;
//cout << l << " p.x - prex = " << p.x - prex << " area = " << area << endl;
        }
        return area;

    }
};
