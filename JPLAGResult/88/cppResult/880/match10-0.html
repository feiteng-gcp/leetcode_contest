class Solution {
public:
    int rectangleArea(vector<vector<int>>& rectangles) {
        vector<long long int> xv;
        vector<long long int> yv;
        for(int i=0;i<rectangles.size();++i){
            const vector<int> &t = rectangles[i];
            xv.push_back(t[0]);
            xv.push_back(t[2]);
            yv.push_back(t[1]);
            yv.push_back(t[3]);
        }
        sort(xv.begin(),xv.end());
        sort(yv.begin(),yv.end());
        int m=xv.size();
        int n=rectangles.size();
        int mp[500][500] = {0};
        long long ret = 0;
        const long long int MOD = 1e9+7;
        for(int k=0;k<n;++k){
            const vector<int> &t = rectangles[k];
            int x1 = min(t[0],t[2]);
            int x2 = max(t[0],t[2]);
            int y1 = min(t[1],t[3]);
            int y2 = max(t[1],t[3]);
            
            for(int i=1;i<m;++i){
                for(int j=1;j<m;++j){
                    if(xv[i-1]>=x1 && xv[i]<=x2 && yv[j-1]>=y1 &&yv[j]<=y2){
                        if(mp[i][j]==0){
                            mp[i][j]=1;
                            ret = (ret + (xv[i]-xv[i-1])*(yv[j]-yv[j-1])%MOD)%MOD;
                        }
                    }
                }
            }
        }
        //for(int i=1;i<m;++i){
        //    for(int j=1;j<m;++j)
        //        cout<<mp[i][j];
        //    cout<<endl;
        //}
        return ret;
        
    }
};
