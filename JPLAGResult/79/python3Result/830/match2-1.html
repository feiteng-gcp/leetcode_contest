class Solution:
    def largestTriangleArea(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        
        def area_square(a, b, c):
            d1 = ((a[0] - b[0]) ** 2 + (a[1] - b[1]) ** 2) ** 0.5
            d2 = ((a[0] - c[0]) ** 2 + (a[1] - c[1]) ** 2) ** 0.5
            d3 = ((c[0] - b[0]) ** 2 + (c[1] - b[1]) ** 2) ** 0.5
            p  = 0.5 * (d1 + d2 + d3)
            
return p * (p-d1) * (p-d2) * (p-d3)
        
        largest = 0
        for i in range(len(points)):
            for j in range(i + 1, len(points)):
                for k in range(j + 1, len(points)):
                    area2 = area_square(points[i], points[j], points[k])
                    if area2 > largest:
                        largest = area2
        return largest ** 0.5
