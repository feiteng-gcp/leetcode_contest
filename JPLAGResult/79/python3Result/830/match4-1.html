class Solution:
    def largestTriangleArea(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        areas = []
        for i, p1 in enumerate(points):
            for j, p2 in enumerate(points[i:]):
                for k, p3 in enumerate(points[j:]):
                    a, b, c = ((p[0] ** 2 + p[1] ** 2)**0.5 for p in [
[p1[0] - p2[0], p1[1] - p2[1]],
                        [p3[0] - p1[0], p3[1] - p1[1]], [p2[0] - p3[0], p2[1] - p3[1]]])
                    s = (a + b + c) / 2
                    val = s * (s - a) * (s - b) * (s - c)
                    if (val < 0):
                        val = 0
                    a = (val)**0.5
                    areas.append(a)
        return max(areas)
