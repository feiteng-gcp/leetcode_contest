import math

class Solution:
	def largestTriangleArea(self, points):
		"""
		:type points: List[List[int]]
		:rtype: float
		"""
		def dist(x, y):
			return math.sqrt((x[0] - y[0]) ** 2 + (x[1] - y[1]) ** 2)

		def area(x, y, z):
			a = dist(x, y)
			b = dist(x, z)
			c = dist(y, z)
			p = (a + b + c) / 2

			# print(x, y, z, a, b, c, p)
			# print(x, y, z, math.sqrt(p * (p - a) * (p - b) * (p - c)))
			try:
				return math.sqrt(p * (p - a) * (p - b) * (p - c))
			except:
				return 0

		n = len(points)
		ans = None
		for i in range(n):
			for j in range(i + 1, n):
				for k in range(j + 1, n):
					cur = area(points[i], points[j], points[k])
					# print(ans, cur)
					if ans is None or ans < cur:
						ans = cur
		return ans
