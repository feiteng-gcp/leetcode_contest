class Solution:
    def largestTriangleArea(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        import math
        def get_area(points, i, j, k):
            a = math.sqrt((points[i][1] - points[j][1])*(points[i][1] - points[j][1]) + (points[i][0] - points[j][0])*(points[i][0] - points[j][0]))
            if a == 0:
                return 0
            h = ((points[i][1] - points[j][1])*points[k][0] - (points[i][0] - points[j][0])*points[k][1] + points[i][0]*points[j][1] - points[i][1]*points[j][0])/a
            return 0.5 * a * h if h > 0 else -0.5 * a * h
        
        largest = 0
        for i in range(len(points)-2):
            for j in range(i+1, len(points)-1):
                for k in range(i+2, len(points)):
                    largest = max(largest, get_area(points, i, j, k))
        return largest
