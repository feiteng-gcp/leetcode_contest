class Solution {
public:
    double average(vector<int> &A, int start, int end){
        double ans = 0;
        for(int i = start; i<=end; i++){
            ans += A[i];
        }
        return ans/(end-start+1);
    }
    
    double largestSumOfAverages(vector<int>& A, int K) {
        vector<vector<double>> dp(A.size(), vector<double> (K+1, 0.0));
        for(int i = 0; i<A.size(); i++){

            dp[i][1] = average(A, 0, i);
        }
        for(int i = 0; i<A.size(); i++){
            for(int j = 2; j<=K; j++){
                if(j-i>1) continue;
                for(int k = 0; k<i; k++){
                    dp[i][j] = max(dp[i][j], dp[k][j-1]+average(A, k+1, i));
                }
            }    
        }
        double ans = dp[A.size()-1][1];
        for(int j = 2; j<=K; j++){
            ans = max(ans, dp[A.size()-1][j]);
        }
        return ans;
    }
};
