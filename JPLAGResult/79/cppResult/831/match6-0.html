class Solution {
public:
    double sum[105];
    double dp[105][105];
    double largestSumOfAverages(vector<int>& A, int K) {
        for(int i = 0; i < A.size(); i++)
        {
            sum[i + 1] = sum[i] + A[i];
            dp[i][1] = sum[i + 1] / (i + 1);
            //cout<<dp[i][1]<<endl;
        }
        for(int j = 2; j <= K; j++)
        {
            for(int i = j - 1; i < A.size(); i++)
            {
                for(int k = i; k < A.size(); k++)
                {
                    dp[k][j] = max(dp[k][j], dp[i - 1][j - 1] + (sum[k + 1] - sum[i]) / (k - i + 1));
                }
            }
        }
        double res;
        return dp[A.size() - 1][K];
    }
};
