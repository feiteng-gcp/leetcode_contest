const int MAXN = 1000005; // 1e6;
double dp[105][105];
class Solution {
public:
    double largestSumOfAverages(vector<int>& a, int k) {
        memset(dp,0,sizeof dp);
        int n = a.size();
        double tot=0;
        for (int i=0;i<n;i++) 
        {
            tot+=a[i],dp[i][1]=tot/(i+1);
            cout<<dp[i][1]<<endl;
        }
        for (int i=0;i<n;i++)
        {
        	double p = 0;

        	for (int t=i;t>=0;t--)
        	{
        		p+=a[t];
        		for (int j=2;j<=k;j++)
        		{
        			dp[i][j]=max(dp[i][j], (t>0?dp[t-1][j-1]:0)+p/(i-t+1));
        		}
        	}
        }
        return dp[n-1][k];
    }
};
