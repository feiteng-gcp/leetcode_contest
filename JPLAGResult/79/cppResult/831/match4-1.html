class Solution {
public:
    double largestSumOfAverages(vector<int>& A, int K) {
        if(A.empty())
            return 0;
        
        int n = A.size();
        vector<vector<double>> dp(K+1,vector<double>(n+1,0));
        
        double s = 0;

        for(int i=n-1; i>=0; i--){
            s += A[i];
            dp[1][i] = s/(n-i);
        }
        
        for(int p = 2; p<=K; p++){
            for(int i=0; i<n; i++){
                            
                double maxavg = 0;
                double sum = 0;
                
                //int split;
                for(int j=i; j<n; j++){
                    sum += A[j];
                    
                    // if(sum/(j-i+1) + dp[p-1][j+1] > maxavg)
                    //     split = j;
                    
                    maxavg = max(maxavg, sum/(j-i+1) + dp[p-1][j+1]);
                }
                
                dp[p][i] = max(maxavg,dp[p-1][i]);
                
                //cout << "dp[" << p << "][" << i << "] : " << dp[p][i] << "  split:" << split << endl;
            }
        }
        
        return dp[K][0];
    }
};
