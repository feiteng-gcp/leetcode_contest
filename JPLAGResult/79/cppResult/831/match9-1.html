double dp[105][10005];
class Solution {
    
public:
    
    double f(vector<int>& A,int i ,  int k){
        // cout<<"blah"<<endl;
        int n = A.size();
        if(i>=n){
            if(k==0){
                return 0;
            }
            return INT_MIN;
        }
        // cout<<"blah1"<<endl;
        if(k<=0){
            return INT_MIN;
        }
        // cout<<"blah2"<<endl;
        if(dp[i][k]!=(double)-1){
            return dp[i][k];
        }
        // cout<<"blah3"<<endl;
        double ans = -1;
        double av=0;
        for(int j =i;j<n;j++){
            // cout<<"blah";
            av+=A[j];
            double temp = (av)/(j-i+1);
            double tempAns = f(A,j+1,k-1);
            ans = max(ans, temp+tempAns);
        }
        return dp[i][k]=ans;
    }
    double largestSumOfAverages(vector<int>& A, int K) {

       double aa=-1;
        for(int i =0;i<105;i++){
           for(int j =0;j<10005;j++){
               dp[i][j]=aa;
           }
        }
        return f(A,0,K);
    }
};
