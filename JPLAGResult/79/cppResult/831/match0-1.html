class Solution {
public:
    double largestSumOfAverages(vector<int>& A, int K) {
        int len = (int)A.size();
    vector<int> preS(len + 1, 0);

    for(int i = 0; i < len; i++) preS[i + 1] = preS[i] + A[i];
    vector<vector<double> > dp(len, vector<double>(K, 0));
    for(int i = 0; i < len; i++) dp[i][0] = preS[i + 1] * 1.0 / (i + 1);
    for(int i = 1; i < len; i++)
        for(int j = 1; j < min(K, i + 1); j++)
            for(int k = j - 1; k < i; k++)
                dp[i][j] = max(dp[i][j], dp[k][j - 1] +
                               (preS[i + 1] - preS[k + 1]) * 1.0 / (i - k));
    return dp[len - 1][K - 1];
    }
};
