class Solution {
public:
    #define maxn 105
    int sum[maxn];
    double dp[maxn][maxn];
    double largestSumOfAverages(vector<int>& A, int K) {
        sum[0] = 0;
        for (int i = 0; i < A.size(); i ++)
            sum[i + 1] = sum[i] + A[i];
        memset(dp, 0, sizeof(dp));
        for (int i = 1; i <= A.size(); i ++) {
            dp[i][1] = 1.0 * sum[i] / i;

            for (int j = 2; j <= i; j ++) {
                for (int k = j - 1; k < i; k ++) {
                    dp[i][j] = max(dp[i][j], dp[k][j - 1] + 1.0 * (sum[i] - sum[k]) / (i - k));
                }
            }
        }
        return dp[A.size()][K];
    }
};
