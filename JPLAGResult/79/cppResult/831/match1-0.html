class Solution {
public:
    double largestSumOfAverages(vector<int>& A, int K) {
        vector<vector<double>> s(A.size() + 1, vector<double>(A.size() + 1, 0));
        for (int i = 0; i < A.size(); ++i) {
            s[i][i] = A[i];
            for (int j = i + 1; j < A.size(); ++j) {
                s[i][j] = s[i][j - 1] + A[j];
                //cout << i << ' ' << j << " " << s[i][j] << endl;
            }
        }
        vector<vector<double>> f(A.size() + 1, vector<double>(K + 1, -1));
        for (int i = 0; i < A.size(); ++i) {
            f[i][1] = s[0][i] / (i + 1);
            for (int k = 2; k <= K; ++k) {
                for (int j = 0; j < i; ++j) {
                    if (f[j][k - 1] >= 0) {
                        double tmp = f[j][k - 1] + s[j + 1][i] / (i - j);
                        f[i][k] = max(tmp, f[i][k]);
                    }
                }
                //cout << "IK: " << i << " " << k << " " << f[i][k] << endl;
            }
        }
        double ret = 0;
        for (int k = 1; k <= K; ++k) {
            //cout << k << " " << f[A.size() - 1][k] << endl;
            ret = max(ret, f[A.size() - 1][k]);
        }
        return ret;
    }
};
