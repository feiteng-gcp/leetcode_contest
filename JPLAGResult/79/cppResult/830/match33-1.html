using vi = vector<int>;
class Solution {
public:
    double get(vi a, vi b, vi c) {
        int x = b[0] - a[0], y = b[1] - a[1];

        int xx = c[0] - a[0], yy = c[1] - a[1];
        return fabs((1.0 * x * yy - 1.0 * y * xx) / 2);
    }
    double largestTriangleArea(vector<vector<int>>& points) {
        double res = 0;
        int n = points.size();
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    res = max(res, get(points[i], points[j], points[k]));
                }
            }
        }
        return res;
    }
};
