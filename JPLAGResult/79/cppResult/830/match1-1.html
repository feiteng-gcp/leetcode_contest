class Solution {
public:
    double largestTriangleArea(vector<vector<int>>& points) {
        double maxArea = 0;
        for (int i = 0; i < points.size() - 2; i++) {
            for (int j = i + 1; j < points.size() - 1; j++) {
                for (int k = j + 1; k < points.size(); k++) {
                    double a = pow(pow(points[i][0] - points[j][0], 2) + pow(points[i][1] - points[j][1], 2), 0.5);
                    double b = pow(pow(points[j][0] - points[k][0], 2) + pow(points[j][1] - points[k][1], 2), 0.5);
                    double c = pow(pow(points[k][0] - points[i][0], 2) + pow(points[k][1] - points[i][1], 2), 0.5);
                    double s = (a + b + c) / 2;
                    double area = pow(s*(s - a)*(s-b)*(s-c), 0.5);
                    if (area > maxArea) {
                        maxArea = area;
                    }
                }
            }
        }
        return maxArea;
    }
};
