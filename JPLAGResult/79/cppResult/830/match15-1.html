class Solution {
public:
    double largestTriangleArea(vector<vector<int>>& points) {

        double ret=0;
        for(int i=0;i<points.size()-2;i++){
            for(int j=i+1;j<points.size()-1;j++){
                for(int k=j+1;k<points.size();k++){
                    auto p1=points[i];auto p2=points[j];auto p3=points[k];
                    double a=sqrt(pow(p1[0]-p2[0],2)+pow(p1[1]-p2[1],2));
                    double b=sqrt(pow(p1[0]-p3[0],2)+pow(p1[1]-p3[1],2));
                    double c=sqrt(pow(p3[0]-p2[0],2)+pow(p3[1]-p2[1],2));
                    double s=(a+b+c)/2;
                    double temp=sqrt(s*(s-a)*(s-b)*(s-c));
                    ret=max(ret,temp);
                }
            }
        }
        return ret;
    }
};
