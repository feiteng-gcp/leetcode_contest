class Solution {
    
    double sqr(int x) {
        return x * x;
    }
    
    double len(vector<int> & p1, vector<int> & p2) {    
        return sqrt(sqr(p1[0] - p2[0]) + sqr(p1[1] - p2[1]));
    }
    
public:
    double largestTriangleArea(vector<vector<int>>& points) {
        double ret = 0;
        for (int i = 0; i < points.size(); ++i) {
            for (int j = i + 1; j < points.size(); ++j) {
                for (int k = j + 1; k < points.size(); ++k) {
                    double l1 = len(points[i], points[j]), l2 = len(points[j], points[k]), l3 = len(points[k], points[i]);
                    double p = (l1 + l2 + l3) / 2;
                    double s = sqrt(p * (p - l1) * (p - l2) * (p - l3));
                    ret = max(ret, s);
                }
            }
        }
        return ret;
    }
};
