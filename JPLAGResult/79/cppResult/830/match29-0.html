class Solution {
public:
    
    int dot(vector<int> a, vector<int> b) {
        return abs(a[0] * b[1] - a[1] * b[0]);
    }
    
    double largestTriangleArea(vector<vector<int>>& points) {
        double ans = 0;
        for (int i = 0; i < points.size(); i ++) {
            for (int j = i + 1; j < points.size(); j ++) {
                for (int k = j + 1; k < points.size(); k ++) {
                    vector<int> a, b;
                    a.push_back(points[j][0] - points[i][0]);
                    a.push_back(points[j][1] - points[i][1]);
                    b.push_back(points[k][0] - points[i][0]);
                    b.push_back(points[k][1] - points[i][1]);
                    int tmp = dot(a, b);
                    ans = max(ans, 0.5 * tmp);
                }
            }
        }
        return ans;
    }
};
