class Solution {
public:
    vector<vector<int>> points;
    
    double l(int x,int y){
        return sqrt(pow((points[x][0]-points[y][0]),2)+pow((points[x][1]-points[y][1]),2));
    }
    
    double calc(int i,int j,int k){
        double p,a,b,c;
        a=l(i,j); b=l(i,k); c=l(j,k);
        p=(a+b+c)/2;
        return sqrt(p*(p-a)*(p-b)*(p-c));
    }
    
    double largestTriangleArea(vector<vector<int>>& points) {
        this->points=points;
        int s=points.size();
        double m=0.0;
        for (int i=0;i<s;i++){
            for (int j=i+1;j<s;j++){
                for (int k=j+1;k<s;k++){
                    if (calc(i,j,k)>m)
                        m=calc(i,j,k);
                }
            }
        }
        return m;
    }
};
