#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

class Solution {
public:
    double largestTriangleArea(vector<vector<int>>& points) {

        double res = 0;
        for(int i = 0 ; i < points.size() ; i ++)
            for(int j = i + 1 ; j < points.size() ; j ++)
                for(int k = j + 1 ; k < points.size() ; k ++){
                    double a = dis(points[i], points[j]);
                    double b = dis(points[i], points[k]);
                    double c = dis(points[j], points[k]);

                    double p = (a + b + c) / 2;
                    res = max(res, sqrt(p * (p - a) * (p - b) * (p - c)));
                }
        return res;
    }

private:
    double dis(const vector<int>& v1, const vector<int>& v2){
        int res = (v1[0] - v2[0]) * (v1[0] - v2[0]) + (v1[1] - v2[1]) * (v1[1] - v2[1]);
        return sqrt(res);
    }
};
