class Solution {
    public double largestSumOfAverages(int[] A, int K) {
        
		int rows = K + 1;
		int cols = A.length;
		double[][] dp = new double[rows][cols];
		for (int i = 1; i < cols; i++)
			A[i] += A[i - 1];
		for (int i = 0; i < cols; i++) {
			dp[1][i] = A[i] * 1.0 / (i + 1);
		}

		for (int row = 2; row < rows; row++)
			for (int col = row - 1; col < cols; col++) {
				dp[row][col] = Integer.MIN_VALUE;
				for (int cut = row - 2; cut < col; cut++) {
					dp[row][col] = Math.max(dp[row][col
],
							(dp[row - 1][cut] + (A[col] - A[cut]) * 1.0 / (col - cut)) );
				}
			}

		return dp[K][cols - 1];
	
    }
}
