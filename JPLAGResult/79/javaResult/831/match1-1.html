class Solution {
    private double[] cumSum;
    private double[][] memo;
    
    public double largestSumOfAverages(int[] A, int K) {
        cumSum = new double[A.length+1];
        memo = new double[K][];
        for (int i=0; i<K; i++) {
            memo[i] = new double[A.length];
        }
        int sum = 0;
        for (int i=0 ;i<A.length; i++) {
            cumSum[i+1] = cumSum[i] + A[i];
        }
        
return helper(A, 0, K);
    }
    
    private double helper(int[] A, int start, int K) {
        if (memo[K-1][start] != 0) {
            return memo[K-1][start];
        }
        double result = 0;
        if (K == 1) {
            result = (cumSum[A.length] - cumSum[start]) / (A.length - start);
        }
        else {
            for (int i=start+1; i<=A.length-K+1; i++) {
                double tmp = (cumSum[i]-cumSum[start])/(i-start) + helper(A, i, K-1);
                result = Math.max(result, tmp);
            }
        }
        memo[K-1][start] = result;
        return result;
    }
}
