class Solution {
      private Double [][]MEMO;
    public double largestSumOfAverages(int[] A, int K) {
        MEMO = new Double[A.length + 1][K + 1];
        return dfs(A,K, 0);
    }

    private double dfs(int []A, int k, int start) {
        if (start == A.length && k >= 0) {
            return 0.0;
        }
        if (k <= 0) {
            return -1.0;
        }
        if (MEMO[start][k] != null) {
            return MEMO[start][k];
        }
        double sum = 0.0;
        int count = 0;
        double ans = 0.0;
        for (int i = start; i < A.length; i++) {
            sum += A[i];
            count++;
            double me = dfs(A, k - 1, i + 1);
            if (me != -1.0 ) {
                ans = Math.max(ans, sum / count + me);
            }

        }
        return MEMO[start][k] = ans;
    }
}
