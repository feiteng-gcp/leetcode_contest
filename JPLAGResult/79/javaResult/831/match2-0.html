class Solution {
    public double largestSumOfAverages(int[] A, int K) {
        int len = A.length;
        int[][] sum = new int[len + 1][len + 1];
        for (int i = 0; i < len; ++i)
            for (int j = i + 1; j <= len; ++j) {
                for (int m = i; m < j; ++m) {
                    sum[i][j] += A[m];
                }
            }
        mx = new double[len + 1][K + 1];
        for (int i = 1; i <= K; ++i)
            for (int j = i; j <= len; ++j) {
                for (int m = 0; m < j; ++m) {
                  if (m > 0 && i - 1 == 0) continue;
                  double aver = sum[m][j] * 1.0 / (j - m);
                  mx[j][i] = Math.max(mx[m][i - 1] + aver, mx[j][i]);
                }
            }
        return mx[len][K];
    }
    
    double mx[][];
}
