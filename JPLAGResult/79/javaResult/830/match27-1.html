class Solution {
    public double largestTriangleArea(int[][] points) {
        double max = 0;
        for (int i = 0; i < points.length; i++) {
            for (int j = i+1; j < points.length; j++) {
                for (int z = j+1; z < points.length; z++) {
                    int[] a = points[i];
                    int[] b = points[j];
                    int[] c = points[z];
                    double x1 = bianchang(a,b);
                    double x2 = bianchang(a,c);
                    double x3 = bianchang(b,c);
                    if (x1+ x2 > x3 && x1+x3 > x2 && x2+x3 > x1) {
                        double p = (x1+x2+x3)/2;
                        double temp = Math.sqrt(p*(p-x1)*(p-x2)*(p-x3));
                        max = Math.max(max, temp);
                    
}
                    
                }
            }
        }
        return max;
    }
    public double bianchang(int[] a, int[] b) {
        return Math.sqrt(Math.pow(a[0]-b[0],2)+Math.pow(a[1]-b[1],2));
    }
}
