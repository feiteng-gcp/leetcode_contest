class Solution {
    private double calculateArea(int[][] points) {
        return Math.abs(0.5*(points[0][0]*(points[1][1]-points[2][1])+points[1][0]*(points[2][1]-points[0][1])+points[2][0]*(points[0][1]-points[1][1])));
    }
    public double largestTriangleArea(int[][] points) {
        double result = Double.MIN_VALUE;
        for (int i = 0; i < points.length; i++) {
            for (int j = i + 1; j < points.length; j++) {
                for (int k = j + 1; k < points.length; k++) {
                    result = Math.max(result, calculateArea(new int[][]{{points[i][0], points[i][1]}, {points[j][0], points[j][1]}, {points[k][0], points[k][1]}}));
                }
            }
        }
        return result;
    }
}
