class Solution {
     public double largestTriangleArea(int[][] points) {
    int n = points.length;
    
double res = 0;
    for(int i=0; i<n; i++) {
      for(int j=i+1; j<n; j++) {
        for(int k=j+1; k<n; k++) {
          res = Math.max(res, area(points[i], points[j], points[k]));
        }
      }
    }
    return res;
  }

  double area(int[] a, int[] b, int[] c) {
    double ac0 = a[0]-c[0];
    double ac1 = a[1] - c[1];
    double ab0 = a[0]-b[0];
    double ab1 = a[1]-b[1];
    return Math.abs((ac0*ab1-ac1*ab0)/2);
  }
}
