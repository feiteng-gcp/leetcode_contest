class Solution {
    
    double dist(int a , int b , int c , int d) {
        return Math.sqrt(Math.pow(a-c,2)+
Math.pow(b-d,2));
    }
    
    public double largestTriangleArea(int[][] points) {
        
        double max = 0;
        for(int i = 0 ; i<points.length ; i++) {
            for(int j = i+1 ; j<points.length ; j++) {
                for(int k = j+1 ; k<points.length ; k++) {
                    double a = dist(points[i][0] , points[i][1] , points[j][0] , points[j][1]);
                    double b = dist(points[i][0] , points[i][1] , points[k][0] , points[k][1]);
                    double c = dist(points[k][0] , points[k][1] , points[j][0] , points[j][1]);
                    
                    if(a+b>c && b+c>a && a+c>b) {
                        double s = (a+b+c)/2;
                        double area = Math.sqrt(s*(s-a)*(s-b)*(s-c));
                        max = Math.max(max , area);
                    
}
                }
            }
        }
        
        return max;
    }
}
