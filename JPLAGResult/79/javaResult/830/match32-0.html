class Solution {
  public double largestTriangleArea(int[][] points) {
        double max = 0.0;
        for (int i = 0; i < points.length; i++) {
            for (int j = i + 1; j < points.length;j++) {
                for (int k = j + 1; k < points.length; k++) {
                    if (canFormTriangle(points[i],points[j], points[k])){
                        max = Math.max(getArea(points[i],points[j],points[k]),max);
                    }
                }
            }
        }
        return max;
    }

    private boolean canFormTriangle(int[] A, int [] B, int []C) {
        int a = A[0]*(B[1] - C[1]) + B[0] * (C[1] - A[1]) + C[0] * (A[1] - B[1]);

        return a != 0;
    }

    private double getArea(int[]A, int B[], int C[]) {
        double ans = 1.0 * A[0]*(B[1] - C[1]) + 1.0 * B[0] * (C[1] - A[1]) + 1.0 * C[0] * (A[1] - B[1]);
         return Math.abs(ans) / 2.0;
    }
}
