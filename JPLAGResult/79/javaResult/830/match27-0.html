class Solution {
    public double largestTriangleArea(int[][] points) {
        double largestArea = 0;
        for (int i = 0; i < points.length - 2; i++) {
            for (int j = i + 1; j < points.length - 1; j++) {
                for (int k = j + 1; k < points.length; k++) {
                    int[] iPoints = points[i];
                    int[] jPoints = points[j];
                    int[] kPoints = points[k];
                    double length1 = getLength(iPoints, jPoints);
                    double length2 = getLength(iPoints, kPoints);
                    double length3 = getLength(jPoints, kPoints);
                    double area = findArea(length1, length2, length3);
                    if (area > largestArea) {
                        largestArea = area;
                    }
                }
            }
        }
        return largestArea;
    }

    public double getLength(int[] first, int[] second) {
        return Math.sqrt(Math.abs(Math.pow(first[0] - second[0], 2) + Math.pow(first[1] - second[1], 2)));
    }

    public double findArea(double sideA, double sideB, double sideC) {
        double area;
        double s;
        s = 0.5 * (sideA + sideB + sideC);
        area = Math.sqrt(s * (s - sideA) * (s - sideB) * (s - sideC));
        return area;
    }

}
