class Solution {
    public double largestTriangleArea(int[][] points) {
        int l = points.length;

        double max = 0.0;
        for (int i = 0; i < l-2; i++) {
            for (int j = i+1; j < l-1; j++) {
                for (int k = j+1; k < l; k++) {
                    double area = computeArea(points[i], points[j], points[k]);
                    if (area > max) {
                        max = area;
                    }
                }
            }
        }

        return max;
    }

    public double computeArea(int[] p1, int[] p2, int[] p3) {
        int a2 = (p1[0] - p2[0]) * (p1[0] - p2[0]) + (p1[1] - p2[1]) * (p1[1] - p2[1]);
        int b2 = (p3[0] - p2[0]) * (p3[0] - p2[0]) + (p3[1] - p2[1]) * (p3[1] - p2[1]);
        int c2 = (p1[0] - p3[0]) * (p1[0] - p3[0]) + (p1[1] - p3[1]) * (p1[1] - p3[1]);

        double a = Math.sqrt(a2);
        double b = Math.sqrt(b2);
        double c = Math.sqrt(c2);

        double s = (a + b + c)/2;

        return Math.sqrt(s * (s-a) * (s-b) * (s-c));
    }
}
