class Solution {
    public double largestTriangleArea(int[][] points) {
        
        double maxArea = 0;
        
        for (int i = 0; i + 2< points.length; i++)
        {
            for (int j = i + 1; j + 1< points.length; j++)
            {
                for (int k = j + 1; k < points.length; k++)
                {
                    double a = getDist(points[i], points[j]);
                    double b = getDist(points[i], points[k]);
                    double c = getDist(points[j], points[k]);
                    
                    double p = (a + b + c) / 2;
                    double area = Math.sqrt(p * (p - a) * (p - b) * (p - c));
                    
                    if (area > maxArea)
                    {
                        maxArea = area;
                    }
                }
                
            }
        }
        
        return maxArea;
    }
    
    private double getDist(int[] a, int[] b)
    {
        double dist = Math.sqrt((a[0] - b[0])* (a[0] - b[0]) + (a[1] - b[1]) * (a[1] - b[1]));
        return dist;
    }
}
