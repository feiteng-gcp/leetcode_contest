class Solution {
    public double largestTriangleArea(int[][] points) {
        double result = 0; 
        for (int i=0; i<points.length-2; i++){
            for (int j=i+1; j<points.length-1; j++){
                for (int k=j+1; k<points.length; k++){
                    int[] pa = points[i];
                    int[] pb = points[j];
                    int[] pc = points[k];
                    double a = distance(pa[0], pa[1], pb[0], pb[1]);
                    double b = distance(pa[0], pa[1], pc[0], pc[1]);
                    double c = distance(pc[0], pc[1], pb[0], pb[1]);
                    
                    double area = area(a,b,c);
                    result = Math.max(result, area);
                }
            }    
        }
        
        return result;
    }
    
    private double distance(int x, int y, int x2, int y2){
        return Math.sqrt( (x-x2)*(x-x2) + (y-y2)*(y-y2));
    }
    
    private double area(double a, double b, double c){
        double p = (a+b+c)/2;
        return Math.sqrt(Math.abs(p*(p-a)*(p-b)*(p-c)));
    }
}
