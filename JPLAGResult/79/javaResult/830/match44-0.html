class Solution {

    // 允许n^3, 50^3=2^16
    /**
     * 错1
     * Input:
     * [[-35,19],[40,19],[27,-20],[35,-3],[44,20],[22,-21],[35,33],[-19,42],[11,47],[11,37]]
     * Output:
     * �
     * Expected:
     * 1799.0
     * maybe: 3 point in 1 line
     * Input:
     * [[0,0],[0,1],[1,0],[0,2],[2,0]]
     * Output:
     * 1.0
     * Expected:
     * 2.0
     * 
     */
    public double largestTriangleArea(int[][] p) {

        double result = 0;
        int n = p.length;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    // try {
                    int x1 = p[i][0], y1 = p[i][1], x2 = p[j][0], y2 = p[j][1], x3 = p[k][0],
                            y3 = p[k][1];
                    if ((x1 - x2) * (y1 - y3) == (x1 - x3) * (y1 - y2)) {
                        continue;
                    }
                    // if (p[i][1] - p[j][1] == 0 || p[i][1] - p[j][1])
                    // if ((p[i][0] - p[j][0]) / (p[i][1] - p[j][1]) == (p[i][0] - p[k][0])
                    // / (p[i][1] - p[k][1])) {
                    // continue;
                    // }
                    // } catch (ArithmeticException e) {
                    // continue;
                    // }
                    // Heron's formula
                    double a = dist(p[i], p[j]), b = dist(p[j], p[k]), c = dist(p[k], p[i]);
                    double area = helen(a, b, c);
                    result = Math.max(result, area);
                }
            }
        }
        return result;

    }

    private double helen(double a, double b, double c) {
        double p = (a + b + c) / 2;
        return Math.sqrt(p * (p - a) * (p - b) * (p - c));
    }

    private double dist(int[] p1, int[] p2) {
        return Math.sqrt(Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2));
    }

}
