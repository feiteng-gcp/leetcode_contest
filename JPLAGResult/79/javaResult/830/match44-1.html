class Solution {
    public double largestTriangleArea(int[][] points) {
        
double ret = 0.0;
        for(int i = 0; i < points.length; i++) {
            for(int j = i + 1; j < points.length; j++) {
                for(int k = j + 1; k < points.length; k++) {
                    double a = (double) points[i][0];
                    double b = (double) points[i][1];
                    double c = (double) points[j][0];
                    double d = (double) points[j][1];
                    double e = (double) points[k][0];
                    double f = (double) points[k][1];
                    
                    
double area = 0.5 * Math.abs(a*d + c*f + e*b - a*f - e*d - c*b);
                    ret = Math.max(ret, area);
                }
            }
        }
        return ret;
    }
}
