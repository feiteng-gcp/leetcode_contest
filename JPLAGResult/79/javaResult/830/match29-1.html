class Solution {
    
    
    
private double getArea(int x1, int y1, int x2, int y2) {
        return 0.5*Math.abs(x1*y2 - x2*y1);
    }
    
    private double getArea(int x1, int y1, int x2, int y2, int x3, int y3) {
        return getArea(x2 - x1, y2 - y1, x3 - x1, y3 - y1);
    }
    
    
public double largestTriangleArea(int[][] points) {
        double maxArea = -1;
        for (int i = 0; i < points.length; i++) {
            for (int j = i+1; j < points.length; j++) {
                for (int k = j+1; k < points.length; k++) {
                    double area = getArea(points[i][0], points[i][1], points[j][0], points[j][1], points[k][0], points[k][1]);
                    if (area > maxArea) maxArea = area;
                
}
            }
        }
        return maxArea;
    }
}
