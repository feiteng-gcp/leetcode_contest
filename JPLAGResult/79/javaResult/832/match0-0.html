/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    class Wrap {
        public Wrap(TreeNode root, boolean allzeros) {
            this.root = root;
            Allzeros = allzeros;
        }

        TreeNode root;
        boolean Allzeros;
    }
    
    public TreeNode pruneTree(TreeNode root) {
        pruneTreeHelper(root);
        return root;
    }
    
    private Wrap pruneTreeHelper(TreeNode root) {
        if (root == null) {
            return new Wrap(null, true);
        }
        Wrap left = pruneTreeHelper(root.left);
        Wrap right = pruneTreeHelper(root.right);
        if (left.Allzeros) {
            root.left = null;
        }
        if (right.Allzeros) {
            root.right = null;
        }
        if (root.left == null && root.right == null && root.val == 0) {
            return new Wrap(null, true);
        }
        return new Wrap(root, false);
    }
}
