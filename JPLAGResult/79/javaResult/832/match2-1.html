/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
public TreeNode pruneTree(TreeNode root) {
        prun(root, 0);
        return root;
    }
    public int prun(TreeNode root, int cone) {
        
        if (root == null) return 0;
        int a = prun(root.left, cone);
        int b = prun(root.right, cone);
        if (a == 0) root.left = null;
        if (b == 0) root.right = null;
        cone = cone+a+b;
        if (root.val == 1) cone++;
        return cone;
    }
}
