class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        
        def count_divisors(num):
            upper = int(num**0.5)
            res = 0
            count = 0
            for i in range(1, upper + 1):
                if num % i == 0:
                    res += i
                    j = num // i
                    count += 1
                    if j > i:
                        res += j
                        count += 1
            if count == 4:
                return res
            else:
                return 0
            
        ans = 0
        for num in nums:
            ans += count_divisors(num)
        
        return ans
