class Solution:
    def sumFourDivisors(self, a: List[int]) -> int:
        
        ret = 0
        
        for x in a:
            divs = set()
            
            for i in range(1, isqrt(x)+1):
                if x % i == 0:
                    divs.add(i)
                    divs.add(x//i)
            
            #divs.append(x)
            #/print(divs)
            
            if len(divs) == 4:
                ret += sum(divs)
            
        return ret
