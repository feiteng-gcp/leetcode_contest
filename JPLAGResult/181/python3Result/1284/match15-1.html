class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def divisorGenerator(n):
            d = []
            for i in range(1,round
(n**(1/2)+1)):
                if n % i == 0:
                    d.append(i)
                    if n/i!=i:
                        d.append(n / i)
            return d
        res = 0
        for n in nums:
            divisors = divisorGenerator(n)
            if len(divisors) == 4:
                res += sum(divisors)
        return int(res)
