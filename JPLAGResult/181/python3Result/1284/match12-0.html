class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        ret = 0
        for n in nums:
            ans = set()
            for i in range(1, int(math.sqrt(n))+1):
                if n % i == 0:
                    ans.add(i)
                    ans.add(n // i)
            if len(ans) == 4:
                ret += sum(list(ans))
        return ret
