import math
class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def good(x):
            res = 0
            sume = 0
            for i in range(1, int(math.sqrt(x))+1):
                if res > 4:
                    return 0
                if x % i == 0:
                    sume 
+= i
                    res += 1
                    if x // i != i:
                        sume += x//i
                        res += 1
            if res == 4:
                return sume
            else:
                return 0
        ans = 0
        for i in nums:
            ans += good(i)
        return ans
