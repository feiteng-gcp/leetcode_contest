class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def get(x):
            sq = int(math.sqrt(x))
            if sq ** 2 == x:
                return 0
            cnt, ans = 0, 0
            for i in range(1, sq+1):
                if x % i == 0:
                    cnt += 2
                    ans += (i + x//i)
                if cnt > 4:
                    return 0
            return ans if cnt == 4 else 0
        return sum(get(x) for x in nums)
