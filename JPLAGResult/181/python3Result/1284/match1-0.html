class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        
        ans = 0
        for n in nums:
            ans += self.find_divisor(n)
        return ans
    
    def find_divisor(self, num):
        ret = set([1, num])
        s = 0
        target = math.sqrt(num)
        for i in range(2, int(target) + 1):
            temp = num % i
            if temp == 0:
                ret.add(i)
                ret.add(num // i)
            
            if len(ret) > 4:
                break
        
        if len(ret) == 4:
            for r in ret:
                s += r
        return s
