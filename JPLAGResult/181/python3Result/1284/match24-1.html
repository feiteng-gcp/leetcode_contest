class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        def for_divisor(num):
            ans = []
            for i in range(1, int(num**0.5) + 1):
                if num % i == 0:
                    if num // i == i:
                        ans.append(i)
                    else:
                        ans 
+= [i, num//i]
            return ans
        ans = 0
        for num in nums:
            divisors = for_divisor(num)
            if len(divisors) == 4:
                ans += sum(divisors)
        return ans
