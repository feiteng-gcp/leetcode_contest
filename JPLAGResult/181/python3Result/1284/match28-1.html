class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        
        def numDivisors(n):
            # Note that this loop runs till square root 
            i = 1
            count = 0
            s = 0
            while i <= math.sqrt(n): 
                if (n % i == 0) : 
                    # If divisors are equal, print only one 
                    if (n / i == i) : 
                        count += 1 
                        s += i
                    else : 
                        # Otherwise print both 
                        count += 2
                        s 
+= i
                        s += n // i
                i = i + 1
            return count, s
        
        res = 0
        for n in nums:
            k, s = numDivisors(n)
            if k == 4:
                res += s
        return res
