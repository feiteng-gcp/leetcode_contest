class Solution:
    def sumFourDivisors(self, nums: List[int]) -> int:
        N = max(nums) + 1
        prime = [True] * N
        a = 2
        while a * a < N:
            if prime[a] == True:
                i = a * 2
                while i < N:
                    prime[i] = False
                    i += a
            a += 1
        p = []
        link = collections.defaultdict(set)
        for i in range(2, N):
            if prime[i]:
                p.append(i)
        for i in range(len(p)):
            num = p[i]
            if 1 * num * num * num <= N:
                link[1 * num * num * num] = {1, num, num * num, num * num * num}
            for j in range(i + 1, len(p)): 
                q = p[j]; 
                if 1 * num * q > N: 
                    break; 
                link[num * q] = {1, num, q, num * q}
        res = 0
        for num in nums:
            if num in link:
                res += sum(link[num])
        return res
