class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        dirs = {0: [-1, 0], 1: [0, 1], 2: [1, 0], 3: [0, -1]}
        m = {
                1: {1, 3, 5},
                3: {1, 4, 6},
                0: {2, 3, 4}, 
                2: {2, 5, 6}
            }
        
        m2 = {
            1: {1, 3},
            2: {0, 2},
            3: {2, 3},
            4: {1, 2},
            5: {0, 3},
            6: {0, 1}
        }
        
        queue = [(0, 0)]
        visited = set(queue)
        
        while len(queue) != 0:
            x, y = queue.pop(0)
            if x == len(grid) - 1 and y == len(grid[0]) -1:
                return True
            
            for a, d in dirs.items():
                if a not in m2[grid[x][y]]:
                    continue
                ix, iy = d
                xx, yy = x+ix ,y+iy
                
                if 0 <= xx < len(grid) and 0 <= yy < len(grid[0]) and (xx, yy) not in visited:
                    if grid[xx][yy] in m[a]:
                        queue.append((xx, yy))
                        visited.add((xx, yy))
        return False
