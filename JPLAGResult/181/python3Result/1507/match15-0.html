class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        n, m = len(grid), len(grid[0])
        h = set()
        h.add((0, 0))
        q = [(0, 0)]
        t = 0
        while t < len(q):
            x, y = q[t]
            t += 1
            if x == n - 1 and y == m - 1:
                return True
            if x > 0 and (x - 1, y) not in h and grid[x][y] in [2, 5, 6] and grid[x - 1][y] in [2, 3, 4]:
                q.append((x - 1, y))
                h.add((x - 1, y))
            if y > 0 and (x, y - 1) not in h and grid[x][y] in [1, 3, 5] and grid[x][y - 1] in [1, 4, 6]:
                q.append((x, y - 1))
                h.add((x, y - 1))
            if x < n - 1 and (x + 1, y) not in h and grid[x][y] in [2, 3, 4] and grid[x + 1][y] in [2, 5, 6]:
                q.append((x + 1, y))
                h.add((x + 1, y))
            if y < m - 1 and (x, y + 1) not in h and grid[x][y] in [1, 4, 6] and grid[x][y + 1] in [1, 3, 5]:
                q.append((x, y + 1))
                h.add((x, y + 1))
        return False
