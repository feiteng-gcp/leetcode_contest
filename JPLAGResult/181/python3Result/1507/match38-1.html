class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        
        m = len(grid)
        n = len(grid[0])
        
        q = [(0, 0)]
        seen = set()
        
        g = {
            
1: [(0, -1), (0, 1)],
            2: [(-1, 0), (1, 0)],
            3: [(0, -1), (1, 0)],
            4: [(1, 0), (0, 1)],
            5: [(-1, 0), (0, -1)],
            6: [(-1, 0), (0, 1)],
        }
        
        
        
        while q:
            
            i, j = q.pop(0)
            # print(i, j)
            if (i, j) == (m-1, n-1):
                return True
            
            for di, dj in g[grid[i][j]]:
                ni = i + di
                nj = j + dj
                if (ni, nj) not in seen and 0 <= ni < m and 0 <= nj < n:
                    if (
di == 1 and grid[ni][nj] in (2, 5, 6)) or \
                       (di == -1 and grid[ni][nj] in (2, 3, 4)) or \
                       (dj == 1 and grid[ni][nj] in (1, 3, 5)) or \
                       (dj == -1 and grid[ni][nj] in (1, 4, 6)):
                        
                        seen.add((ni, nj))
                        q.append((ni, nj))
        
        return False
