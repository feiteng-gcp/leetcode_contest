class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        if m == 1 and n == 1:
            return True
        dirs = {
            1: [(0, 1), (0, -1)],
            2: [(1, 0), (-1, 0)],
            3: [(0, -1), (1, 0)],
            4: [(1, 0), (0, 1)],
            5: [(0, -1), (-1, 0)],
            6: [(0, 1), (-1, 0)]
        }
        mark = set()
        s = []
        for dx, dy in dirs[grid[0][0]]:
            xx, yy = dx, dy
            if 0 <= xx < m and 0 <= yy < n:
                s.append([(0, 0), (xx, yy)])
        
        mark.add((0, 0))
        
        while s:
            last, (x, y) = s.pop()
            if any((x + dx, y + dy) == last for dx, dy in dirs[grid[x][y]]):
                if x == m - 1 and y == n - 1:
                    return True
                mark.add((x, y))
                for dx, dy in dirs[grid[x][y]]:
                    xx, yy = x + dx, y + dy
                    if 0 <= xx < m and 0 <= yy < n and (xx, yy) not in mark:
                        s.append([(x, y), (xx, yy)])
        return False
