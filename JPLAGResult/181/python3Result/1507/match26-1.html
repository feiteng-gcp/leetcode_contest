class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        go = {1:{(0,1):[1,3,5],(0,-1):[1,4,6]}, 2:{(-1,0):[2,3,4],(1,0):[2,5,6]}, 3:{(
0,-1):[1,4,6],(1,0):[2,5,6]},4:{(0,1):[1,3,5],(1,0):[2,5,6]},5:{(-1,0):[2,3,4],(0,-1):[1,4,6]},6:{(-1,0):[2,3,4],(0,1):[1,3,5]}}
        dirs = ((-1,0),(1,0),(0,1),(0,-1))
        bfs = [(0,0)]
        m,n= len(grid),len(grid[0])
        v = [[False]*n for _ in range(m)]
        v[0][0] = True
        while bfs:
            x,y = bfs.pop(0)
            if (x,y) == (m-1,n-1): return True
            for dx, dy in dirs:
                nx, ny = x+dx, y+dy
                if 0 <= nx< m and 0<=ny<n and not v[nx][ny] and (dx, dy) in go[grid[x][y]] and grid[nx][ny] in go[grid[x][y]][(dx,dy)]:
                    if (nx,ny) ==(m-1,n-1): return True
                    bfs
.append((nx,ny))
                    v[nx][ny] = True
        return False
