class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        deq = collections.deque([(0, 0)])
        seen = {(0, 0)}
        while deq:
            r, c = deq.popleft()
            if r == m - 1 and c == n - 1:
                return True
            if grid[r][c] == 1:
                if c + 1 < n and grid[r][c + 1] in (1, 3, 5) and (r, c + 1) not in seen:
                    seen.add((r, c + 1))
                    deq.append((r, c + 1))
                if c - 1 >= 0 and grid[r][c - 1] in (1, 4, 6) and (r, c - 1) not in seen:
                    seen.add((r, c - 1))
                    deq.append((r, c - 1))
            elif grid[r][c] == 2:
                if r + 1 < m and grid[r + 1][c] in (2, 5, 6) and (r + 1, c) not in seen:
                    seen.add((r + 1, c))
                    deq.append((r + 1, c))
                if r - 1 >= 0 and grid[r - 1][c] in (2, 3, 4) and (r - 1, c) not in seen:
                    seen.add((r - 1, c))
                    deq.append((r - 1, c))
            elif grid[r][c] == 3:
                if c - 1 >= 0 and grid[r][c - 1] in (1, 4, 6) and (r, c - 1) not in seen:
                    seen.add((r, c - 1))
                    deq.append((r, c - 1))
                if r + 1 < m and grid[r + 1][c] in (2, 5, 6) and (r + 1, c) not in seen:
                    seen.add((r + 1, c))
                    deq.append((r + 1, c))
            elif grid[r][c] == 4:
                if c + 1 < n and grid[r][c + 1] in (1, 3, 5) and (r, c + 1) not in seen:
                    seen.add((r, c + 1))
                    deq.append((r, c + 1))
                if r + 1 < m and grid[r + 1][c] in (2, 5, 6) and (r + 1, c) not in seen:
                    seen.add((r + 1, c))
                    deq.append((r + 1, c))
            elif grid[r][c] == 5:
                if c - 1 >= 0 and grid[r][c - 1] in (1, 4, 6) and (r, c - 1) not in seen:
                    seen.add((r, c - 1))
                    deq.append((r, c - 1))
                if r - 1 >= 0 and grid[r - 1][c] in (2, 3, 4) and (r - 1, c) not in seen:
                    seen.add((r - 1, c))
                    deq.append((r - 1, c))
            elif grid[r][c] == 6:
                if r - 1 >= 0 and grid[r - 1][c] in (2, 3, 4) and (r - 1, c) not in seen:
                    seen.add((r - 1, c))
                    deq.append((r - 1, c))
                if c + 1 < n and grid[r][c + 1] in (1, 3, 5) and (r, c + 1) not in seen:
                    seen.add((r, c + 1))
                    deq.append((r, c + 1))
        return False
