class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        nrow, ncol = len(grid), len(grid
[0])
        direction = {1: [(0, -1), (0, 1)], 2: [(1, 0), (-1, 0)], 3: [(0, -1), (1, 0)], 4: [(0, 1), (1, 0)], 5: [(-1, 0), (0, -1)], 6: [(-1, 0), (0, 1)]}

        def dfs(i, j, si, sj, visited):
            if (si - i, sj - j) not in direction[grid[i][j]]:
                return False
            
            if (i, j) == (nrow - 1, ncol - 1):
                return True
            
            if (i, j) in visited:
                
return False
            
            visited.add((i, j))
            
            for di, dj in direction[grid[i][j]]:
                ni, nj = i + di, j + dj
                if 0 <= ni < nrow and 0 <= nj < ncol and (ni, nj) not in visited:
                    if dfs(ni, nj, i, j, visited):
                        return True
            
            return False
        
        return dfs(0, 0, -1, 0, set()) or dfs(0, 0, 0, -1, set()) or dfs(0, 0, 1, 0, set()) or dfs(0, 0, 0, 1, set())
