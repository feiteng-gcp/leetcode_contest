class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        dirs = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        nMap = {1: [2, 3], 2: [0, 1], 3: [2, 1], 4: [1, 3], 5: [0, 2], 6: [0, 3]}
        q = [
(0, 0)]
        visited = set([(0, 0)])
        while q:
            i, j = q.pop(0)
            #print(i, j)
            if i == m-1 and j == n-1:
                return True
            for dCode in nMap[grid[i][j]]:
                di, dj = dirs[dCode]
                ni, nj = i + di, j + dj
                if 0 <= ni < m and 0 <= nj < n and (ni, nj) not in visited:
                    if dCode < 2: 
                        match = 1 - dCode
                    else:
                        match = 5 - dCode
                    if match in nMap[grid[ni][nj]]:
                        visited.add((ni, nj))
                        q.append((ni, nj))
        
        return False
