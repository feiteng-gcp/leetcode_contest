class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        queue = [[0,0]]
        visited = set()
        visited.add((0,0))
        while len(queue) > 0:
            x,y = queue.pop(0)
            if x == len(grid)-1 and y == len(grid[0]) - 1:
                
return True
            if grid[x][y] == 1:
                if y-1 >= 0 and (x,y-1) not in visited and grid[x][y-1] in (1,4,6):
                    visited.add((x,y-1))
                    queue.append([x,y-1])
                if y+1 < len(grid[0]) and (
x,y+1) not in visited and grid[x][y+1] in (1,3,5):
                    visited.add((x,y+1))
                    queue.append([x,y+1])
            elif grid[x][y] == 2:
                if x-1 >= 0 and (x-1,y) not in visited and grid[x-1][y] in (2,3,4):
                    visited.add((x-1,y))
                    queue.append([x-1,y])
                if x+1 < len(grid) and (
x+1,y) not in visited and grid[x+1][y] in (2,5,6):
                    visited.add((x+1,y))
                    queue.append([x+1,y])
            elif grid[x][y] == 3:
                if y-1 >= 0 and (x,y-1) not in visited and (grid[x][y-1] ==4 or grid[x][y-1] == 1 or grid[x][y-1] == 6):
                    visited.add((x,y-1))
                    queue.append([x,y-1])
                if x+1 < len(grid) and (x+1,y) not in visited and (grid[x+1][y] ==2 or grid
[x+1][y] == 5 or grid[x+1][y] == 6):
                    visited.add((x+1,y))
                    queue.append([x+1,y])
            elif grid[x][y] == 4:
                if y+1 < len(grid[0]) and (x,y+1) not in visited and (grid[x][y+1] ==1 or grid
[x][y+1] == 3 or grid[x][y+1] == 5):
                    visited.add((x,y+1))
                    queue.append([x,y+1])
                if x+1 < len(grid) and (x+1,y) not in visited and (grid[x+1][y] ==2 or grid
[x+1][y] == 5 or grid[x+1][y] == 6):
                    visited.add((x+1,y))
                    queue.append([x+1,y])
            elif grid[x][y] == 5:
                if x-1 >= 0 and (x-1,y) not in visited and (grid[x-1][y] ==2 or grid[x-1][y] == 3 or grid[x-1][y] == 4):
                    visited.add((x-1,y))
                    queue.append([x-1,y])
                if y-1 >= 0 and (x,y-1) not in visited and (grid[x][y-1] ==4 or grid
[x][y-1] == 6 or grid[x][y-1] == 1):
                    visited.add((x,y-1))
                    queue.append([x,y-1])
            else:
                if x-1 >= 0 and (x-1,y) not in visited and (grid[x-1][y] ==2 or grid[x-1][y] == 3 or grid[x-1][y] == 4):
                    visited.add((x-1,y))
                    queue.append([x-1,y])
                if y+1 < len(grid[0]) and (x,y+1) not in visited and (grid[x][y+1] ==1 or grid
[x][y+1] == 3 or grid[x][y+1] == 5):
                    visited.add((x,y+1))
                    queue.append([x,y+1])
        return False
