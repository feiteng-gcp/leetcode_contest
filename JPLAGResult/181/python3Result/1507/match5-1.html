class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        if not grid or not grid[0]:
            return False
        row, col 
= len(grid), len(grid[0])
        helper ={1:[(0, -1), (0, 1)],
                2:[(-1, 0), (1, 0)],
                3:[(0, -1), (1, 0)],
                4:[(1, 0), (0, 1)],
                5:[(0, -1), (-1, 0)],
                6:[(-1, 0), (0, 1)]}
        visited = {(0, 0)}
        stack = [(0, 0), ]    
        while stack:
            r, c = stack.pop()
            if (r, c) == (row - 1, col - 1):
                return True
            
for dr, dc in helper[grid[r][c]]:
                nr, nc = r + dr, c + dc
                if -1 < nr < row and -1 < nc < col and (nr, nc) not in visited and (-dr, -dc) in helper[grid[nr][nc]]:
                    stack.append((nr, nc))
                    visited.add((nr, nc))
        return False
