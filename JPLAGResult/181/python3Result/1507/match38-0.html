class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        dirs = [
            lambda x, y: (x, y + 1, 4),
            lambda x, y: (x, y - 1, 3),
            lambda x, y: (x + 1, y, 2),
            lambda x, y: (x - 1, y, 1)
        ]
        ways = [(0, 0)]
        record = {(0, 0)}
        while ways:
            temp = []
            for x, y in ways:
                if x == len(grid) - 1 and y == len(grid[0]) - 1:
                                return True
                for d in dirs:
                    m, n, p = d(x, y)
                    if 0 <= m < len(grid) and 0 <= n < len(grid[0]) and (m, n) not in record:
                        if any([
                            grid[x][y] == 1 and p == 4 and grid[m][n] in (1, 3, 5),
                            grid[x][y] == 1 and p == 3 and grid[m][n] in (1, 4, 6),
                            grid[x][y] == 2 and p == 1 and grid[m][n] in (2, 3, 4),
                            grid[x][y] == 2 and p == 2 and grid[m][n] in (2, 5, 6),
                            grid[x][y] == 3 and p == 3 and grid[m][n] in (1, 4, 6),
                            grid[x][y] == 3 and p == 2 and grid[m][n] in (2, 5, 6),
                            grid[x][y] == 4 and p == 4 and grid[m][n] in (1, 3, 6),
                            grid[x][y] == 4 and p == 2 and grid[m][n] in (2, 5, 6),
                            grid[x][y] == 5 and p == 1 and grid[m][n] in (2, 3, 4),
                            grid[x][y] == 5 and p == 3 and grid[m][n] in (1, 4, 6),
                            grid[x][y] == 6 and p == 1 and grid[m][n] in (2, 3, 4),
                            grid[x][y] == 6 and p == 4 and grid[m][n] in (1, 3, 5)
                        ]):
                            temp.append((m, n))
                            record.add((m, n))
            ways = temp
                            
        return False
