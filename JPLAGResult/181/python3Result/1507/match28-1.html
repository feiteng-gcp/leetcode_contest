class Solution:
    def hasValidPath(self, grid: List[List[int]]) -> bool:
        d = {
            1: [
                (-1,0,[1,4,6]),
                (1,0,[1,3,5])
            ],
            2: [
                (0,-1,[2,3,4]),
                (0,1,[2,5,6])
            ],
            3: [
                (-1,0,[1,4,6]),
                (0,1,[2,5,6])
            ],
            4: [
                (1,0,[1,3,5]),
                (0,1,[2,5,6])
            ],
            5: [
                (0,-1,[2,3,4]),
                (-1,0,[1,4,6])
            ],
            6: [
                (0,-1,[2,3,4]),
                (1,0,[1,3,5])
            ]
        }

        m, n = len(grid[0]), len(grid)
        used = [[False]*n for _ in range(m)]
        stack = [
(0,0)]
        while stack:
            i,j = stack.pop()
            if i==m-1 and j==n-1: return True
            for di,dj,c in d[grid[j][i]]:
                ni,nj = i+di,j+dj
                if 0<=ni<m and 0<=nj<n:
                    if used[ni][nj] == True: continue
                    if grid[nj][ni] in c:
                        used[ni][nj] = True
                        stack.append((ni,nj))
        return False
