class Solution {
public:
    bool vis[400][400];
    int dis[400][400];
    int n,m,maze[400][400];
    bool check(int x,int y){
        return x>=0&&x<n&&y>=0&&y<m;
    }
    void bfs(int sx,int sy){
        queue<pair<int,int> > q;
        memset(vis,0,sizeof(vis));
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                dis[i][j]=1e9;
            }
        }
        q.push({sx,sy});
        dis[sx][sy]=0;
        while(!q.empty()){
            pair<int,int> rt=q.front();
            q.pop();
            int nx=rt.first;
            int ny=rt.second;
            if(vis[nx][ny]) continue;
            vis[nx][ny]=1;
            if(maze[nx][ny]==1){
                int dy=ny-1;

                int dx=nx;//左
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==4||maze[dx][dy]==6||maze[dx][dy]==1)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
                dy=ny+1;
                dx=nx;//右
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==3||maze[dx][dy]==5||maze[dx][dy]==1)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
            }
            if(maze[nx][ny]==2){
                int dy=ny;

                int dx=nx-1;//上
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==2||maze[dx][dy]==3||maze[dx][dy]==4)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
                dy=ny;
                dx=nx+1;//下
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==2||maze[dx][dy]==5||maze[dx][dy]==6)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
            }  
            if(maze[nx][ny]==3){
                int dy=ny-1;

                int dx=nx;//左
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==1||maze[dx][dy]==4||maze[dx][dy]==6)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
                dy=ny;
                dx=nx+1;//下
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==2||maze[dx][dy]==5||maze[dx][dy]==6)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
            }
            if(maze[nx][ny]==4){
                int dy=ny+1;

                int dx=nx;//右
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==1||maze[dx][dy]==3||maze[dx][dy]==5)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
                dy=ny;
                dx=nx+1;//下
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==2||maze[dx][dy]==5||maze[dx][dy]==6)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
            } 
            if(maze[nx][ny]==5){
                int dy=ny;

                int dx=nx-1;//上
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==2||maze[dx][dy]==3||maze[dx][dy]==4)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
                dy=ny-1;
                dx=nx;//左
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==1||maze[dx][dy]==4||maze[dx][dy]==6)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
            }  
            if(maze[nx][ny]==6){
                int dy=ny;

                int dx=nx-1;//上
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==2||maze[dx][dy]==3||maze[dx][dy]==4)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
                dy=ny+1;
                dx=nx;//右
                if(check(dx,dy)&&!vis[dx][dy]&&(maze[dx][dy]==3||maze[dx][dy]==5||maze[dx][dy]==1)){
                    if(dis[dx][dy]>dis[nx][ny]+1){
                        dis[dx][dy]=dis[nx][ny]+1;
                        q.push({dx,dy});
                    }
                }
            }  
        }
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        n=(int)grid.size();
        m=grid[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                maze[i][j]=grid[i][j];
            }
        }
        bfs(0,0);
        if(dis[n-1][m-1]!=1e9) return true;
        return false;
    }
};
