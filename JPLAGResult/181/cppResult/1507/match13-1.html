class Solution {
public:
    int n,m;
    bool res = false;
    unordered_map<int,unordered_map<int,int>> visited;
    bool valid(int i,int j){
        if(i < 0 || i >= n || j < 0 || j >= m)return false;
        return true;
    }
    void dfs(vector<vector<int>>& grid,int i,int j){
        if(res)return;
        if(visited[i][j]++ > 0)return;
        if(i == n - 1 && j == m - 1){
            res = true;
            return;
        }

        if(grid[i][j] == 1){
            //left
            if(valid(i,j - 1) && (grid[i][j - 1] == 4 || grid[i][j - 1] == 6 || grid[i][j - 1] == 1)){
                dfs(grid,i,j - 1);
            }
            //right
            if(valid(i,j + 1) && (grid[i][j + 1] == 5 || grid[i][j + 1] == 3 || grid[i][j + 1] == 1)){
                dfs(grid,i,j + 1);
            }
        }
        if(grid[i][j] == 2){
            //up
            if(valid(i - 1, j) && (grid[i - 1][j] == 4 || grid[i - 1][j] == 2 || grid[i - 1][j] == 3)){
                dfs(grid,i - 1, j);
            }
            //down
            if(valid(i + 1, j) && (grid[i + 1][j] == 5 || grid[i + 1][j] == 2 || grid[i + 1][j] == 6)){
                dfs(grid,i + 1, j);
            }
        }
        if(grid[i][j] == 3){
            //left
            if(valid(i,j - 1) && (grid[i][j - 1] == 4 || grid[i][j - 1] == 6 || grid[i][j - 1] == 1)){
                dfs(grid,i,j - 1);
            }
            //down
            if(valid(i + 1, j) && (grid[i + 1][j] == 5 || grid[i + 1][j] == 2 || grid[i + 1][j] == 6)){
                dfs(grid,i + 1, j);
            }
        }
        if(grid[i][j] == 4){
            //right
            if(valid(i, j + 1) && (grid[i][j + 1] == 5 || grid[i][j + 1] == 3 || grid[i][j + 1] == 1)){
                dfs(grid,i,j + 1);
            }
            //down
            if(valid(i + 1, j) && (grid[i + 1][j] == 5 || grid[i + 1][j] == 2 || grid[i + 1][j] == 6)){
                dfs(grid,i + 1, j);
            }
        }
        if(grid[i][j] == 5){
            //up
            if(valid(i - 1, j) && (grid[i - 1][j] == 4 || grid[i - 1][j] == 2 || grid[i - 1][j] == 3)){
                dfs(grid,i - 1,j);
            }
            //left
            if(valid(i,j - 1) && (grid[i][j - 1] == 4 || grid[i][j - 1] == 6 || grid[i][j - 1] == 1)){
                dfs(grid,i,j - 1);
            }
        }
        if(grid[i][j] == 6){
            //right
            if(valid(i,j + 1) && (grid[i][j + 1] == 5 || grid[i][j + 1] == 3 || grid[i][j + 1] == 1)){
                dfs(grid,i,j + 1);
            }
            //up
            if(valid(i - 1, j) && (grid[i - 1][j] == 4 || grid[i - 1][j] == 2 || grid[i - 1][j] == 3)){
                dfs(grid,i - 1,j);
            }
        }
        
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        dfs(grid,0,0);
        return res;
    }
};
