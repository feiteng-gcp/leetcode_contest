class Solution {

    int st[6][2] = {{0,1}, {2,3}, {0,2}, {1,2}, {0,3}, {1,3}};
    int d[4][2] = {{0, -1}, {0,1}, {1,0}, {-1,0}};
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        queue<pair<int, int>> bfsQ;
        vector<vector<bool>> vis(m, vector<bool>(n, false));
        bfsQ.push({0, 0});
        vis[0][0] = true;
        while (!bfsQ.empty()) {

            auto curr = bfsQ.front(); bfsQ.pop();
            if (curr.first == m-1 && curr.second == n-1)
                return true;
            int street = grid[curr.first][curr.second]-1;
            for (int i = 0; i < 2; i++) {
                int dir = st[street][i];
                int nx = d[dir][0] + curr.first, ny = d[dir][1] + curr.second;
                if (0 <= nx && nx < m && 0 <= ny && ny < n && !vis[nx][ny]) {
                    int street2 = grid[nx][ny]-1;
                    if ((dir == 0 && (st[street2][0] == 1 || st[street2][1] == 1)) || 
                        (dir == 1 && (st[street2][0] == 0 || st[street2][1] == 0)) ||
                        (dir == 2 && (st[street2][0] == 3 || st[street2][1] == 3)) ||
                        (dir == 3 && (st[street2][0] == 2 || st[street2][1] == 2))) {
                        bfsQ.push({nx, ny});
                        vis[nx][ny] = true;
                    }
                    
                }
            }
        }
        return false;
    }
};
