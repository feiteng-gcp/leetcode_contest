class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int x = grid.size();
        int y = grid[0].size();
        bool find[x][y];
        memset(find, false, sizeof(find));
        find[0][0] = true;
        queue<pair<int,int>> q;
        q.push({0,0});
        while(!q.empty()){
            int nowx = q.front().first;
            int nowy = q.front().second;
            q.pop();
            
            if(grid[nowx][nowy] == 1){
                if(nowy-1 >= 0 && !find[nowx][nowy-1] && (grid[nowx][nowy-1] == 1 || grid[nowx][nowy-1] == 4 ||grid[nowx][nowy-1] == 6)){
                    find[nowx][nowy-1] = true;
                    q.push({nowx, nowy-1});
                }
                if(nowy+1 < y && !find[nowx][nowy+1] && (grid[nowx][nowy+1] == 1 || grid[nowx][nowy+1] == 3 ||grid[nowx][nowy+1] == 5)){
                    find[nowx][nowy+1] = true;
                    q.push({nowx, nowy+1});
                }
            }
            if(grid[nowx][nowy] == 2){
                if(nowx-1 >= 0 && !find[nowx-1][nowy] && (grid[nowx-1][nowy] == 2 || grid[nowx-1][nowy] == 3 ||grid[nowx-1][nowy] == 4)){
                    find[nowx-1][nowy] = true;
                    q.push({nowx-1, nowy});
                }
                if(nowx+1 < x && !find[nowx+1][nowy] && (grid[nowx+1][nowy] == 2 || grid[nowx+1][nowy] == 5 ||grid[nowx+1][nowy] == 6)){
                    find[nowx+1][nowy] = true;
                    q.push({nowx+1, nowy});
                }
            }
            if(grid[nowx][nowy] == 3){
                if(nowy-1 >= 0 && !find[nowx][nowy-1] && (grid[nowx][nowy-1] == 1 || grid[nowx][nowy-1] == 4 ||grid[nowx][nowy-1] == 6)){
                    find[nowx][nowy-1] = true;
                    q.push({nowx, nowy-1});
                }
                if(nowx+1 < x && !find[nowx+1][nowy] && (grid[nowx+1][nowy] == 2 || grid[nowx+1][nowy] == 5 ||grid[nowx+1][nowy] == 6)){
                    find[nowx+1][nowy] = true;
                    q.push({nowx+1, nowy});
                }
            }
            if(grid[nowx][nowy] == 4){
                if(nowx+1 < x && !find[nowx+1][nowy] && (grid[nowx+1][nowy] == 2 || grid[nowx+1][nowy] == 5 ||grid[nowx+1][nowy] == 6)){
                    find[nowx+1][nowy] = true;
                    q.push({nowx+1, nowy});
                }
                if(nowy+1 < y && !find[nowx][nowy+1] && (grid[nowx][nowy+1] == 1 || grid[nowx][nowy+1] == 3 ||grid[nowx][nowy+1] == 5)){
                    find[nowx][nowy+1] = true;
                    q.push({nowx, nowy+1});
                }
            }
            if(grid[nowx][nowy] == 5){
                if(nowy-1 >= 0 && !find[nowx][nowy-1] && (grid[nowx][nowy-1] == 1 || grid[nowx][nowy-1] == 4 ||grid[nowx][nowy-1] == 6)){
                    find[nowx][nowy-1] = true;
                    q.push({nowx, nowy-1});
                }
                if(nowx-1 >= 0 && !find[nowx-1][nowy] && (grid[nowx-1][nowy] == 2 || grid[nowx-1][nowy] == 3 ||grid[nowx-1][nowy] == 4)){
                    find[nowx-1][nowy] = true;
                    q.push({nowx-1, nowy});
                }
            }
            if(grid[nowx][nowy] == 6){
                if(nowx-1 >= 0 && !find[nowx-1][nowy] && (grid[nowx-1][nowy] == 2 || grid[nowx-1][nowy] == 3 ||grid[nowx-1][nowy] == 4)){
                    find[nowx-1][nowy] = true;
                    q.push({nowx-1, nowy});
                }
                if(nowy+1 < y && !find[nowx][nowy+1] && (grid[nowx][nowy+1] == 1 || grid[nowx][nowy+1] == 3 ||grid[nowx][nowy+1] == 5)){
                    find[nowx][nowy+1] = true;
                    q.push({nowx, nowy+1});
                }
            }
        }
        return find[x-1][y-1];
    }
};
