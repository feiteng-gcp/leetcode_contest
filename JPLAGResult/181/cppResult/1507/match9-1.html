class Solution {
    struct node
    {
        int x,y;
    };
    int n,m;
    bool valid(int x,int y)
    {
        return x>=0&&x<n&&y>=0&&y<m;
    }
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        n=grid.size(),m=grid[0].size();
        int vis[n][m];
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                vis[i][j]=0;
        queue<node> q;
        q.push({0,0});
        vis[0][0]=1;
        while(!q.empty())
        {
            node u=q.front();
            q.pop();
           // cout<<u.x<<","<<u.y<<" "<<q.size()<<endl;
            if(u.x==n-1&&u.y==m-1)
                return true;
            int tx,ty;
            switch(grid[u.x][u.y])
            {

                case 1: tx=u.x,ty=u.y+1; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]&1))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }
                        tx=u.x,ty=u.y-1; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]==1||grid[tx][ty]==4||grid[tx][ty]==6))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }
                        break;
                    

                case 2: tx=u.x+1,ty=u.y; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]>=5||grid[tx][ty]==2))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }  
                        tx=u.x-1,ty=u.y; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]>=2&&grid[tx][ty]<=4))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        } 
                        break;

                case 3: tx=u.x+1,ty=u.y; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]>=5||grid[tx][ty]==2))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }  
                        tx=u.x,ty=u.y-1; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]==1||grid[tx][ty]==4||grid[tx][ty]==6))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        } 
                        break;

                case 4: tx=u.x+1,ty=u.y;
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]>=5||grid[tx][ty]==2))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        } 
                        tx=u.x,ty=u.y+1;
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]&1))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }
                        break;

                case 5: tx=u.x-1,ty=u.y; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]>=2&&grid[tx][ty]<=4))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        } 
                        tx=u.x,ty=u.y-1; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]==1||grid[tx][ty]==4||grid[tx][ty]==6))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }
                        break;

                case 6: tx=u.x-1,ty=u.y; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]>=2&&grid[tx][ty]<=4))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }
                        tx=u.x,ty=u.y+1; 
                        if(valid(tx,ty)&&!vis[tx][ty]&&(grid[tx][ty]&1))
                        {
                            vis[tx][ty]=1;
                            q.push({tx,ty});
                        }
                        break;       
            } 
        }
        return false;
        
    }
};
