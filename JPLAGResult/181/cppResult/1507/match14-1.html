class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        if (grid.empty()) return true;
        queue<array<int, 2>> bfs;
        int n = grid.size(), m = grid[0].size();
        array<int, 2> End = {n-1, m-1};
        bfs.push({0, 0});
        vector<vector<int>> vis(n, vector<int>(m, 0));
        vis[0][0] = 1;
        while(!bfs.empty()) {
            auto now = bfs.front(); bfs.pop();
            // cout << now[0] << " " << now[1] << " here\n";
            if (now == End) return true;

            int x = now[0], y = now[1];
            if (x > 0 && !vis[x-1][y]) {
                if (grid[x][y] == 2 || grid[x][y] == 5 || grid[x][y] == 6) {
                    if (grid[x-1][y] == 2 || grid[x-1][y] == 3 || grid[x-1][y] == 4) {
                        bfs.push({x-1, y});
                        vis[x-1][y] = 1;
                    }
                }
            }
            
            if (y > 0 && !vis[x][y-1]) {
                if (grid[x][y] == 1 || grid[x][y] == 3 || grid[x][y] == 5) {
                    if (grid[x][y-1] == 1 || grid[x][y-1] == 4 || grid[x][y-1] == 6) {
                        bfs.push({x, y-1});
                        vis[x][y-1] = 1;
                    }
                }
            }
            
            
            if (x < n-1 && !vis[x+1][y]) {
                if (grid[x][y] == 2 || grid[x][y] == 3 || grid[x][y] == 4) {
                    if (grid[x+1][y] == 2 || grid[x+1][y] == 5 || grid[x+1][y] == 6) {
                        bfs.push({x+1, y});
                        vis[x+1][y] = 1;
                    }
                }
            }
            
            if (y < m-1 && !vis[x][y+1]) {
                if (grid[x][y] == 1 || grid[x][y] == 4 || grid[x][y] == 6) {
                    if (grid[x][y+1] == 1 || grid[x][y+1] == 3 || grid[x][y+1] == 5) {
                        bfs.push({x, y+1});
                        vis[x][y+1] = 1;
                    }
                }
            }
        }
        return false;
    }
};
