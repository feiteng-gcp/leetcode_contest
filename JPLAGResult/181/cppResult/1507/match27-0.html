class Solution {
public:
 int n, m;
    int s;
    int t;
    int tot;
    vector<vector<int>> edge;
    bool flag;
    
    int dx[4] = {0, 0, 1, -1};
    int dy[4] = {-1, 1, 0, 0};
    
    void link(int i, int j, int k) {
        int tx = i + dx[k];
        int ty = j + dy[k];
        if (tx >= 0 && tx < n && ty >= 0 && ty < m) {
            edge[i * m + j].push_back(tx * m + ty);
        }
    }
    
    vector<int> vis;
    
    void go(int x) {
        if (x == t) {
            flag = true;
            return;
        }
        
        vis[x] = true;
        for (auto& to: edge[x]) {
            if (!vis[to]) {
                bool flag = false;
                for (auto& v: edge[to]) if (v == x) {
                  flag = true;
                  break;
                }
                if (flag) {
                  go(to);
                }
            }
        }
        vis[x] = false;
    }
    
    bool hasValidPath(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        s = 0;
        t = n * m - 1;
        tot = n * m;
        flag = false;
        
        edge.resize(tot);
        vis.resize(tot);
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    link(i, j, 0);
                    link(i, j, 1);
                } else if (grid[i][j] == 2) {
                    link(i, j, 2);
                    link(i, j, 3);
                } else if (grid[i][j] == 3) {
                    link(i, j, 0);
                    link(i, j, 2);
                } else if (grid[i][j] == 4) {
                    link(i, j, 1);
                    link(i, j, 2);
                } else if (grid[i][j] == 5) {
                    link(i, j, 0);
                    link(i, j, 3);
                } else if (grid[i][j] == 6) {
                    link(i, j, 1);
                    link(i, j, 3);
                }
            }
        }
        
        go(s);
        
        return flag;        
    }
};
