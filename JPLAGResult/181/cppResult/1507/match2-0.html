class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        
        int h = grid.size(), w = grid[0].size();
        vector<vector<int>> dist(h, vector<int>(w, 1000000));
        
        queue<pair<int, int>> q;
        q.push({0, 0});
        dist[0][0] = 0;
        
        while(q.size()){
            
            auto p = q.front(); q.pop();
            
            int y = p.first, x = p.second;
            //cout << y << " " << x << endl;
            if(grid[y][x] == 1){
                int ny = y, nx = x-1;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 4 || grid[ny][nx] == 6 || grid[ny][nx] == 1){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
                ny = y, nx = x+1;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 3 || grid[ny][nx] == 5 || grid[ny][nx] == 1){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
            }
            
            if(grid[y][x] == 2){
                int ny = y-1, nx = x;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] ==3 || grid[ny][nx] == 4 || grid[ny][nx] == 2){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
                ny = y+1, nx = x;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){

                    if(grid[ny][nx] == 5 || grid[ny][nx] == 6 || grid[ny][nx] == 2){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
            }
            
            if(grid[y][x] == 3){
                int ny = y, nx = x-1;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 1 || grid[ny][nx] == 4 || grid[ny][nx] == 6){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
                ny = y+1, nx = x;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 2 || grid[ny][nx] == 5 || grid[ny][nx] == 6){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
            }
            if(grid[y][x] == 4){
                int ny = y+1, nx = x;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 2 || grid[ny][nx] == 5 || grid[ny][nx] == 6){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
                ny = y, nx = x+1;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 1 || grid[ny][nx] == 3 || grid[ny][nx] == 5){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
            }
            if(grid[y][x] == 5){
                int ny = y, nx = x-1;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 1 || grid[ny][nx] == 4 || grid[ny][nx] == 6){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
                ny = y-1, nx = x;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 2 || grid[ny][nx] == 3|| grid[ny][nx] == 4){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
            }
            if(grid[y][x] == 6){
                int ny = y-1, nx = x;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 2 || grid[ny][nx] == 3 || grid[ny][nx] == 4){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
                ny = y, nx = x+1;
                if(ny >= 0 && ny < h && nx >= 0 && nx < w && dist[ny][nx] == 1000000){
                    if(grid[ny][nx] == 1|| grid[ny][nx] == 3 || grid[ny][nx] == 5){
                        dist[ny][nx] = dist[y][x] + 1;
                        q.push({ny,nx});
                    }
                }
            }
        }
        if(dist[h-1][w-1] != 1000000) return true;
        return false;
    }
};
