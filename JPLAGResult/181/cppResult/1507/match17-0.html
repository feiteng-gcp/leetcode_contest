class Solution {
    int n, m;
    
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        queue<pair<int, int>> q;
        q.push(make_pair(0, 0));
        while (!q.empty()) {
            int row, col;
            tie(row, col) = q.front();
            q.pop();
            if (row == n - 1 && col == m - 1) return true;
            if (grid[row][col] == 1) {
                if (canGo(row, col - 1) && (grid[row][col-1] == 1 || grid[row][col-1] == 4 || grid[row][col-1] == 6)) {
                    q.push(make_pair(row, col-1));
                }
                if (canGo(row, col + 1) && (grid[row][col+1] == 1 || grid[row][col+1] == 3 || grid[row][col+1] == 5)) {
                    q.push(make_pair(row, col + 1));
                }
            }
            else if (grid[row][col] == 2) {
                if (canGo(row - 1, col) && (grid[row-1][col] == 2 || grid[row-1][col] == 3 || grid[row-1][col] == 4)) {
                    q.push(make_pair(row - 1, col));
                }
                if (canGo(row + 1, col) && (grid[row + 1][col] == 2 || grid[row + 1][col] == 5 || grid[row + 1][col] == 6)) {
                    q.push(make_pair(row + 1, col));
                }
            }
            else if (grid[row][col] == 3) {
                if (canGo(row, col - 1) && (grid[row][col-1] == 1 || grid[row][col-1] == 4 || grid[row][col-1] == 6)) {
                    q.push(make_pair(row, col - 1));
                }
                if (canGo(row + 1, col) && (grid[row + 1][col] == 2 || grid[row + 1][col] == 5 || grid[row + 1][col] == 6)) {
                    q.push(make_pair(row + 1, col));
                }
            }
            else if (grid[row][col] == 4) {
                if (canGo(row, col + 1) && (grid[row][col+1] == 1 || grid[row][col+1] == 3 || grid[row][col+1] == 5)) {
                    q.push(make_pair(row, col + 1));
                }
                if (canGo(row + 1, col) && (grid[row + 1][col] == 2 || grid[row + 1][col] == 5 || grid[row + 1][col] == 6)) {
                    q.push(make_pair(row + 1, col));
                }
            }
            else if (grid[row][col] == 5) {
                if (canGo(row - 1, col) && (grid[row-1][col] == 2 || grid[row-1][col] == 3 || grid[row-1][col] == 4)) {
                    q.push(make_pair(row - 1, col));
                }
                if (canGo(row, col - 1) && (grid[row][col-1] == 1 || grid[row][col-1] == 4 || grid[row][col-1] == 6)) {
                    q.push(make_pair(row, col - 1));
                }
            }
            else if (grid[row][col] == 6) {
                if (canGo(row - 1, col) && (grid[row-1][col] == 2 || grid[row-1][col] == 3 || grid[row-1][col] == 4)) {
                    q.push(make_pair(row - 1, col));
                }
                if (canGo(row, col + 1) && (grid[row][col+1] == 1 || grid[row][col+1] == 3 || grid[row][col+1] == 5)) {
                    q.push(make_pair(row, col + 1));
                }
            }
            grid[row][col] = -1;
        }
        return false;
    }
    
    bool canGo(int i, int j) {
        return (i >= 0 && i < n && j >= 0 && j < m);
    }
};
