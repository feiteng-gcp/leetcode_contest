class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        return dfs(0, 0, grid.size(), grid[0].size(), grid);
    }
private:
    const vector<int> dy{1, 0, 0, -1};
    const vector<int> dx{0, -1, 1, 0};
    vector<vector<int>> dir{{}, {0, 3},{1, 2},{2, 3}, {0, 2}, {1, 3}, {0, 1}};

    bool dfs(int i, int j, int n, int m, vector<vector<int>>& grid) {
        //cout << i << " " << j << "\n";
        if (i == n-1 && j == m-1) return true;
        auto &temp = dir[grid[i][j]];
        grid[i][j] = 0;
        for (auto k : temp) {
            int x = i + dx[k], y = j + dy[k];
            if (x < 0 || y < 0 || x >= n || y >= m || !grid[x][y]) continue;
            bool ok = false;
            for (auto a : dir[grid[x][y]]) {
                if (3-a == k) ok = true;
            }
            if (ok && dfs(x, y, n, m, grid)) return true;
        }
        return false;
    }
};
