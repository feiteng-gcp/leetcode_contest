#define P pair<int,int>
class Solution {
public:
    int n, m;
    vector<vector<bool>> vis;
    bool ok(int x, int y) {
        if(x < 0 || x >= n || y < 0 || y >= m || vis[x][y]) {
            return false;
        }
        return true;
    }
    vector<vector<int>> grid;
    bool bfs() {

        
        queue<P> q;
        vis[0][0] = true;
        
        q.push({0, 0});
        
        while(!q.empty()) {
            P top = q.front();
            q.pop();
            int x = top.first, y = top.second;
            if(x == n - 1 && y == m - 1) {
                return true;
            }
            
            int now = grid[x][y];
            if(now == 1) {
                int nx = x, ny = y + 1;
                //由
                if(ok(nx, ny) && (grid[nx][ny] == 3 || grid[nx][ny] == 5 || grid[nx][ny] == 1)) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
                //左
                nx = x, ny = y - 1;
                if(ok(nx, ny) && (grid[nx][ny] == 6 || grid[nx][ny] == 4 || grid[nx][ny] == 1) ) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
            }
            else if(now == 2) {
                int nx = x + 1, ny = y;
                //向下走
                if(ok(nx, ny) && (grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6)) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
                //上
                nx = x - 1, ny = y;
                if(ok(nx, ny) && (grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) ) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
                
            }else if(now == 3) {
                //左
                int nx = x, ny = y - 1;

                if(ok(nx, ny) && (grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6)) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
                //下
                nx = x + 1, ny = y;
                if(ok(nx, ny) && (grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) ) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }

            }else if(now == 4) {
                int nx = x, ny = y + 1;
                //右边
                if(ok(nx, ny) && (grid[nx][ny] == 3 || grid[nx][ny] == 5 || grid[nx][ny] == 1)) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
                //下边
                nx = x + 1, ny = y;
                if(ok(nx, ny) && (grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) ) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }

            }else if(now == 5) {
                int nx = x - 1, ny = y;
                //上
                if(ok(nx, ny) && (grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4)) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
                //左
                nx = x, ny = y - 1;
                if(ok(nx, ny) && (grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6) ) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }

                
            }else {
                int nx = x, ny = y + 1;
                //右
                if(ok(nx, ny) && (grid[nx][ny] == 3 || grid[nx][ny] == 5 || grid[nx][ny] == 1)) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }
                //上
                nx = x - 1, ny = y;
                if(ok(nx, ny) && (grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) ) {
                    vis[nx][ny] = true;
                    q.push({nx, ny});
                }

            }            
        }
        return false;

    }
    bool hasValidPath(vector<vector<int>>& grid) {
        this->grid = grid;
        n = grid.size(), m = grid[0].size();
        vis.resize(n + 1, vector<bool>(m + 1));
        return bfs();
    }
};
