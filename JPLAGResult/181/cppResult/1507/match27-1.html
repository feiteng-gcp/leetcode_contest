#define U(i) (2*(i))
#define C(i) (2*(i)+1)
#define D(i) (2*(i)+2)
#define L(j) (2*(j))
#define R(j) (2*(j)+2)

struct DSU {
    vector<int> par;
    int find(int x) {
        if (par[x] == x) return x;
        return par[x] = find(par[x]);
    }
    void uni(int x, int y) {
        int px = find(x), py = find(y);
        par[px] = py;
    }
    DSU(int n): par(n, 0) {
        for (int i = 0; i < n; i++)
            par[i] = i;
    }
};

class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();

        int N = 2*n+1, M = 2*m+1;
        DSU dsu((2*n+1)*(2*m+1));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    dsu.uni(C(i)*M+L(j), C(i)*M+C(j));
                    dsu.uni(C(i)*M+C(j), C(i)*M+R(j));
                } else if (grid[i][j] == 2) {
                    dsu.uni(U(i)*M+C(j), C(i)*M+C(j));
                    dsu.uni(C(i)*M+C(j), D(i)*M+C(j));
                } else if (grid[i][j] == 3) {
                    dsu.uni(C(i)*M+L(j), C(i)*M+C(j));
                    dsu.uni(C(i)*M+C(j), D(i)*M+C(j));
                } else if (grid[i][j] == 4) {
                    dsu.uni(C(i)*M+R(j), C(i)*M+C(j));
                    dsu.uni(C(i)*M+C(j), D(i)*M+C(j));
                } else if (grid[i][j] == 5) {
                    dsu.uni(U(i)*M+C(j), C(i)*M+C(j));
                    dsu.uni(C(i)*M+C(j), C(i)*M+L(j));
                } else if (grid[i][j] == 6) {
                    dsu.uni(U(i)*M+C(j), C(i)*M+C(j));
                    dsu.uni(C(i)*M+C(j), C(i)*M+R(j));
                }
            }
        }

        //return dsu.find(C(0)*M+C(0)) == dsu.find(C(1)*M+C(2));
        return dsu.find(C(0)*M+C(0)) == dsu.find(C(n-1)*M+C(m-1));
    }
};
