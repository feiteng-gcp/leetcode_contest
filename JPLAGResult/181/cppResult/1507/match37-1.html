class Solution {
 public:
  bool hasValidPath(vector<vector<int>>& grid) {
    
    return hasPath(grid, 0, 0);
  }
 private:
  vector<vector<int>> dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

  vector<vector<int>> path = {{}, {2, 3}, {0, 1}, {2, 1}, {3, 1}, {0, 2}, {0, 3}};
  bool hasPath(vector<vector<int>> &grid, int x, int y) {
    int m = grid.size(), n = grid[0].size();
    if (x == m - 1 && y == n - 1) {
      return true;
    } else {
      int p = grid[x][y];

      grid[x][y] = -1;
      for (int i = 0; i < 2; i++) {
        int nx = x + dirs[path[p][i]][0];
        int ny = y + dirs[path[p][i]][1];
        if (inRange(grid, nx, ny) && grid[nx][ny] != -1) {
          int pp = grid[nx][ny];
          for (int j = 0; j < 2; j++) {
            int px = nx + dirs[path[pp][j]][0], py = ny + dirs[path[pp][j]][1];
            if (px == x && py == y) {
              if (hasPath(grid, nx, ny)) 
                return true;
            }
          }
        }
      }
      return false;
    }
  }
  bool inRange(vector<vector<int>> &grid, int x, int y) {
    int m = grid.size(), n = grid[0].size();
    return !(x < 0 || y < 0 || x >= m || y >= n);
  }
};
