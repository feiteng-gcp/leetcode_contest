class Solution {
public:
    bool hasValidPath(vector<vector<int>>& grid) {
        int  m = grid.size(), n = grid[0].size();
        vector<bool> seen(m*n);
        vector<int> dir{0,-1,0,1,0};
        queue<int> q;
        q.push(0);
     //   seen[0] = 1;
        while(!q.empty()) {
            int nd = q.front();
            q.pop();
            int x = nd/n;
            int y = nd%n;
            if(x == m-1 && y == n-1)
                return true;
            
            int value = grid[x][y];
            for(int i = 0; i < 4; ++i) {
                int r = x + dir[i];
                int c = y + dir[i+1];
                
                if(r < 0 || r >= m || c < 0 || c >= n)
                    continue;
                if(seen[r*n+c])
                    continue;
                int next = r*n + c;
                int nv = grid[r][c];
                
                if(value == 1 && r == x) {
                    if(c == y+1 && (nv==1 || nv == 3 || nv == 5)) {
                        q.push(next);
                    }
                    if(c == y - 1 && (nv == 1 || nv == 4 || nv == 6)) {
                        q.push(next);
                    }
                }
                if(value == 2 && c == y) {
                    if(r == x-1 && (nv == 2 || nv == 3 || nv == 4)) {
                        q.push(next);
                    }
                    if(r == x + 1 && (nv == 2 || nv == 5 || nv == 6)) {
                        q.push(next);
                    }
                }
                if(value == 3 ) {
                    if(r == x && c == y-1 && (nv == 4 || nv == 6 || nv == 1)) {
                        q.push(next);
                    }
                    if(c == y && r == x+1 && (nv == 2 || nv == 5 || nv == 6)) {
                        q.push(next);
                    }
                }
                if(value == 4) {
                    if(r == x && c == y + 1 && (nv == 3 || nv == 5 || nv == 1)) {
                        q.push(next);
                    }
                    if(c == y && r == x+1 && (nv == 2 || nv == 5 || nv == 6)) {
                        q.push(next);
                    }
                }
                if(value == 5 ) {
                    if(r == x && c == y-1 && (nv == 4 || nv == 6 || nv == 1)) {
                        q.push(next);
                    }
                    if(c == y && r == x-1 && (nv == 2 || nv == 3 || nv == 4)) {
                        q.push(next);
                    }
                }
                if(value == 6) {
                    if(r == x && c == y + 1 && (nv == 3 || nv == 5 || nv == 1)) {
                        q.push(next);
                    }
                    if(c == y && r == x-1 && (nv == 2 || nv == 3 || nv == 4)) {
                        q.push(next);
                    }
                }
            }
            seen[x*n + y]=1;
        }
        return false;
    }
};
