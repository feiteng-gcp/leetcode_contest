class Solution {
public:
    bool safe(int i, int j, int n, int m){
        return (i>=0 && i<n && j>=0 && j<m);
    }
    vector<pair<int,int>> give_next(int i, int j, int n, int m, vector<vector<int>>& grid, vector<vector<bool>>& vis){
        vector<pair<int,int>> next;
        int curr = grid[i][j];

        if(curr == 1){
            // left
            if(safe(i,j-1,n,m) && !vis[i][j-1] && (grid[i][j-1] == 4 || grid[i][j-1] == 6 || grid[i][j-1] == 1))
                next.push_back({i,j-1});
            // right
            if(safe(i,j+1,n,m) && !vis[i][j+1] && (grid[i][j+1] == 3 || grid[i][j+1] == 5 || grid[i][j+1] == 1))
                next.push_back({i,j+1});
        }else if(curr == 2){
            // up
            if(safe(i-1,j,n,m) && !vis[i-1][j] && (grid[i-1][j] == 3 || grid[i-1][j] == 4 || grid[i-1][j] == 2))
                next.push_back({i-1,j});
            // down
            if(safe(i+1,j,n,m) && !vis[i+1][j] && (grid[i+1][j] == 5 || grid[i+1][j] == 6 || grid[i+1][j] == 2))
                next.push_back({i+1,j});
        }else if(curr == 3){
            // left
            if(safe(i,j-1,n,m) && !vis[i][j-1] && (grid[i][j-1] == 1 || grid[i][j-1] == 4 || grid[i][j-1] == 6))
                next.push_back({i,j-1});
            // down
            if(safe(i+1,j,n,m) && !vis[i+1][j] && (grid[i+1][j] == 2 || grid[i+1][j] == 5 || grid[i+1][j] == 6))
                next.push_back({i+1,j});
        }else if(curr == 4){
            // right
            if(safe(i,j+1,n,m) && !vis[i][j+1] && (grid[i][j+1] == 1 || grid[i][j+1] == 3 || grid[i][j+1] == 5))
                next.push_back({i,j+1});
            // down
            if(safe(i+1,j,n,m) && !vis[i+1][j] && (grid[i+1][j] == 2 || grid[i+1][j] == 5 || grid[i+1][j] == 6))
                next.push_back({i+1,j});
        }else if(curr == 5){
            // left
            if(safe(i,j-1,n,m) && !vis[i][j-1] && (grid[i][j-1] == 1 || grid[i][j-1] == 4 || grid[i][j-1] == 6))
                next.push_back({i,j-1});
            // up
            if(safe(i-1,j,n,m) && !vis[i-1][j] && (grid[i-1][j] == 2 || grid[i-1][j] == 3 || grid[i-1][j] == 4))
                next.push_back({i-1,j});
        }else{
            // right
            if(safe(i,j+1,n,m) && !vis[i][j+1] && (grid[i][j+1] == 1 || grid[i][j+1] == 3 || grid[i][j+1] == 5))
                next.push_back({i,j+1});
            // up
            if(safe(i-1,j,n,m) && !vis[i-1][j] && (grid[i-1][j] == 2 || grid[i-1][j] == 3 || grid[i-1][j] == 4))
                next.push_back({i-1,j});
        }
        return next;
    }
    bool dfs(int i, int j, int n, int m, vector<vector<int>>& grid, vector<vector<bool>>& vis){
        if(i == n-1 && j == m-1)
            return true;
        auto next = give_next(i,j,n,m,grid,vis);
        if(next.size() == 0)
            return false;
        for(auto p:next){
            vis[p.first][p.second] = true;
            if(dfs(p.first, p.second, n, m, grid, vis))
                return true;
            vis[p.first][p.second] = false;
        }
        return false;
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<bool>> vis(n, vector<bool> (m,false));
        vis[0][0] = true;
        bool ans = dfs(0,0,n,m,grid,vis);
        return ans;
    }
};
