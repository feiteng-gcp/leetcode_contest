class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int sum = 0;
        for(int num : nums) {
            int cur_sum = num+1;
            int count = 0;
            for(int n = 2; n<=floor(sqrt(num)); ++n) {
                if(num % n == 0) {
                    if(num/n == num || num/n == 1) continue;
                    if(num/n == n) {
                        count+=1;
                        cur_sum+=n;
                    } else {
                        count+=2;
                        cur_sum+=n;
                        cur_sum+=num/n;
                    }
                }
            }
            //cout<<"for "<<num<<" count is "<<count<<endl;
            if(count == 2) {
                sum+=cur_sum;
            }
        }
        
        return sum;
    }
};
