using namespace std;

class Solution {
public:
    int check(int num) {
        int count = 0;
        int res = 0;
        for (int i = 1; i * i <= num; i++) {
            if (num % i == 0) {
                count++;
                res += i;
                if (i * i != num) count++, res += num / i;
                if (count > 4) break;
            }

        }
        return count == 4 ? res : 0;
    }
    int sumFourDivisors(vector<int>& nums) {
        int res = 0;
        for (auto num : nums) res += check(num);
        return res;
    }
};
