class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for (int i : nums) {
            int sum = 0, cnt = 0;
            for (int j = 1; j * j <= i; j++) {
                if (i % j) continue;
                cnt += 2;
                sum += j + i / j;

                if (j * j == i) {
                    cnt--;
                    sum -= j;
                }
            }
            if (cnt == 4) ans += sum;
        }
        return ans;
    }
};
