class Solution {
public:
    int getP(int num){
        int cnt = 0;
        int sum = 0;
        for(int i = 1;i <= pow(num, 0.5);i ++){
            if(num%i == 0){
                cnt ++;
                sum += i;
                if(i*i != num){
                    cnt ++;
                    sum += num/i;
                }
            }
        }
        if(cnt == 4) return sum;
        return 0;
    }
    
    int sumFourDivisors(vector<int>& nums) {
        int sum = 0;
        for(auto& e: nums){
            sum += getP(e);
        }
        return sum;
    }
};
