class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int res = 0;
        for (auto i : nums) {
            res += f(i);
        }
        return res;
    }
private:
    int f(int n) {
        int Sqrt = sqrt(n), cnt = 2, res = 1+n;
        if (Sqrt*Sqrt == n) return 0;
        for (int i = 2; i <= Sqrt; ++i) {
            if (n % i == 0) {
                res += i + n/i;
                if ((cnt += 2) > 4) return 0;
            }
        }
        return cnt == 4 ? res : 0;
    }
};
