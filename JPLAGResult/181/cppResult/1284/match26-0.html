class Solution {
public:
    int check(int n) {
        int res = 1 + n, c = 2;
        for(int i = 2; i <= n / 2; i++) {
            if(n % i == 0) {
                c++;
                res += i;
            }
            if(c > 4) {
                return 0;
            }
        }
        return c == 4? res: 0;
    }
    int sumFourDivisors(vector<int>& nums) {
        int res = 0;
        for(auto &i : nums) {
            res += check(i);
        }
        return res;
    }
};
