class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int res = 0;
        for (int i = 0; i < nums.size(); ++i) {
            int cnt = 0;
            int tmp = 0;
            for (int j = 1; j <= sqrt(nums[i]); ++j) {
                if (nums[i] % j == 0) {
                    if (j * j == nums[i]) {
                        ++cnt;
                        tmp += j;
                    } else {
                        cnt += 2;
                        tmp += j + nums[i] / j;
                    }
                }
                if (cnt > 4) {
                    break;
                }
            }
            if (cnt == 4) {
                res += tmp;
            }
            //cout << nums[i] << " " << cnt << endl;
        }
        return res;
    }
};
