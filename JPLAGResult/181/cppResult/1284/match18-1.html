class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int result = 0;
        for(auto num : nums) {
            int total = 0, sum = 0;
            for(int i = 2; i <= sqrt(num); i++) {
                if(num % i) continue;
                sum += i;
                total += 1;

                if(num / i != i) {
                    sum += num / i;
                    total += 1;
                }
            }
            if(total == 2) result += sum + 1 + num;
        }
        return result;
    }
};
