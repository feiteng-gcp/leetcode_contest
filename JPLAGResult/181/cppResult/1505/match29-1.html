class Solution {
public:
    vector<int> createTargetArray(vector<int>& nums, vector<int>& index) {
        int n = nums.size();
        vector<int> ans(n, -1);
        for(int i = 0; i < n; i++){
            int x = nums[i], idx = index[i];
            for(int j = n-1; j > idx; j--) ans[j] = ans[j-1];
            ans[idx] = x;
        }
        return ans;
    }
};
