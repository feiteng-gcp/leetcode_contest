class Solution {
public:
    int longestPrefixSuffix(string s) 
    { 
        int n = s.length(); 

        int lps[n]; 
        lps[0] = 0;


        int len = 0;
        int i = 1; 
        while (i < n) 
        { 
            if (s[i] == s[len]) 
            { 
                len++; 
                lps[i] = len; 
                i++; 
            } 
            else 
            { 
                if (len != 0) 
                { 
                    len = lps[len-1]; 
                } 
                else
                { 
                    lps[i] = 0; 
                    i++; 
                } 
            } 
        } 

        int res = lps[n-1]; 

        // Since we are looking for 
        // non overlapping parts. 
        return (res == s.size()) ? res -1 : res; 
    } 
    string longestPrefix(string s) {
        return s.substr(0, longestPrefixSuffix(s));
    }
};
