class Solution {
public:
    int LengthlongestPrefixSuffix(string s) 
{ 
    int n = s.length(); 
  
    vector<int> lps(n,0);
    lps[0] = 0; 
    int len = 0; 
  
    int i = 1; 
    while (i < n) { 
        if (s[i] == s[len]) { 
            len++; 
            lps[i] = len; 
            i++; 
        } 
        else { 
            if (len != 0) { 
                len = lps[len - 1]; 
            } 
  
            // If len = 0 
            else { 
                lps[i] = 0; 
                i++; 
            } 
        } 
    } 
  
    int res = lps[n - 1]; 
  
    return res; 
} 
    string longestPrefix(string s) {
        
        int l = LengthlongestPrefixSuffix(s);
        
        return l?s.substr(0,l):"";
        
    }
};
