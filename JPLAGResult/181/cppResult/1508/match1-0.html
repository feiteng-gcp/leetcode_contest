class Solution {
public:
    string longestPrefix(string s) 
    {
        vector<int> next(s.size());
        next[0] = 0;// 部分匹配表格 也就是上面的 value那一行，第一个字符最大前后缀长度是0
        int j = 0;
        for (int i = 1; i < s.size(); i++)//从第2个字符开始计算
        {
            while (j >0 && s[i]!= s[j])//循环 很关键，很难理解 下面解释
                j = next[j - 1];
            if (s[i] == s[j])
                ++j;
            next[i] = j;
        }
        
        // for (auto x:next)
        //     cout<<x<<" ";
        // cout<<endl;
                
        return s.substr(0,next.back());
    }
};
