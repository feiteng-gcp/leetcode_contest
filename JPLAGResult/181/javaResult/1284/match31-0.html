class Solution {
    public int sumFourDivisors(int[] nums) {
        int sum = 0;
        for (int n : nums) {
            int[] cur = cntD(n);
            if (cur[0] == 4) {
                sum += cur[1];
            }
        }
        
        return sum;
    }
    
    private int[] cntD(int n) {
        int bound = (int)Math.sqrt(n);
        int cnt = 0, sum = 0;
        for (int i = 1; i <= bound; i++) {
            if (n % i == 0) {
                if (i == n / i) {
                    cnt++;
                    sum += i;
                } else {
                    cnt += 2;
                    sum += i + n / i;
                }
            }
            if (cnt > 4) return new int[]{-1, sum};
        }
        
        return new int[]{cnt, sum};
    }
}
