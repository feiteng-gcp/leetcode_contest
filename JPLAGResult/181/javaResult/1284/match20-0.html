class Solution {
    
    public int sum(int n) {
        int sum = 0;
        int divisor = 0;
        for(int i=1; i<=Math.sqrt(n); i++) {
            if(n%i == 0) {
                if(n/i != i) divisor += 2;
                else divisor += 1;
                sum += n/i;
                sum += i;
            }
        }
        return divisor == 4 ? sum : 0;
    }
    
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for(int n: nums) {
            res += sum(n);
        }
        return res;
    }
}
