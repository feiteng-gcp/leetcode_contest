class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for(int num: nums) {
            res += getSum(num);
        }
        return res;
    }
    
    private int getSum(int n) {
        int sum = 0;
        int count = 0;
        for(int i=1;i<=Math.sqrt(n);i++) {
            if (n%i == 0) {
                if (n/i == i) {
                    count++;
                    sum +=i;
                } else {
                    count+=2;
                    sum = sum + i + n/i;
                
}
            }
        }
        
        return count == 4? sum: 0;
    }
}
