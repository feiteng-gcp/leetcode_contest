class Solution {
    public int sumFourDivisors(int[] nums) {
        int n = nums.length, res = 0;
        for(int i = 0; i < n ; i ++)
        {
            int cnt = 0, div = 0, j = 2;
            for(; j * j < nums[i]; j++)
            {
                if(nums[i] % j == 0)
                {
                    cnt++;
                    div = j;
                }
                if(cnt > 2) break;
            }
            if(cnt == 1 && nums[i] != j * j)            
                res += 1 + nums[i] + nums[i] / div + div;            
        }
        return res;
    }
}
