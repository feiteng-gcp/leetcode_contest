class Solution {
    public int sumFourDivisors(int[] nums) {
        int result = 0;
        for (int num : nums) {
            Set<Integer> set = getDivisors(num);
            if (set.size() == 4) {
                for (int x : set) {
                    result += x;
                }
            }
        }
        return result;
    }
    
    private Set<Integer> getDivisors(int num) {
        Set<Integer> set = new HashSet<>();
        int limit = (int)Math.sqrt(num);
        for (int i = 1; i <= limit; i++) {
            if (num % i == 0) {
                set.add(i);
                set.add(num / i);
            }
            if (set.size() > 4) {
                break;
            }
        }
        return set;
    }
}
