class Solution {
    
public int sumFourDivisors(int[] nums) {
        
        int c = 0;
        for (int n : nums) {
            c += fn(n);
        }
        return c;
    }
    
    
private int fn(int n) {
        int count = 0;
        int sum = 0;
        for (int i = 1; i <= n; i++) {
            if (n % i == 0) {
                count++;
                sum += i;
                if (count > 4) {
                    break;
                
}
            }
        }
        return count == 4 ? sum : 0; 
    }
}
