class Solution {
    public int sumFourDivisors(int[] nums) {
        int sum = 0;
        for(int i = 0; i < nums.length; i++){
            Set<Integer> set = div(nums[i]);
            if(set.size() == 4){
                for(int vv: set){
                    sum += vv;
                
}
            }
        }
        return sum;
    }
    private Set<Integer> div(int v){
        Set<Integer> set = new HashSet<>();
        for(int i = 1; i*i <= v; i++)
{
            if(v % i == 0){
                set.add(i);
                set.add(v/i);
            }
        }
        return set;
    }
}
