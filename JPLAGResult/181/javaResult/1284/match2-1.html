class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for (int num : nums) {
            int cnt = 0, sum = 0;
            for (int i = 2; i <= (int) Math.sqrt(num); ++i) {
                if (0 == num % i) {
                    if (num != i * i) {
                        cnt += 2;
                        sum += i + num / i;
                    } else cnt++;
                    if (2 < cnt) break
;
                }
            }
            if (2 == cnt) res += sum + num + 1;
        }
        return res;
    }
}
