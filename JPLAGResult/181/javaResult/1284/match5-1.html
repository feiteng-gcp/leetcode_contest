class Solution {
    
    int res = 0;
    public int sumFourDivisors(int[] nums) {
        
for (int num : nums) {
            List<Integer> curr = helper(num);
            if (curr.size() == 4) {
                for (int k : curr) res += k
;
            }
        }
        return res;
    }
    
    private List<Integer> helper(int num) {
        List<Integer> curr = new ArrayList<>();
        int k = (int)Math.sqrt(num);
        if (k * k == num) return curr;
        
for (int i = 1; i <= k; i++) {
            if (num % i == 0) {
                curr.add(i);
                curr.add(num / i);
            }
        }
        return curr;
    }
}
