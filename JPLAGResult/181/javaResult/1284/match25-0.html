class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for (int num : nums) {
            Set<Integer> divisors = findDivisors(num);
            if (divisors.size() == 4) {
                for (int div : divisors) res += div;
            }
        }
        return res;
    }
    
    private Set<Integer> findDivisors(int num) {
        Set<Integer> res = new HashSet<>();
        for (int i = 1; i <= Math.sqrt(num); i++) {
            if (num % i == 0) {
                res.add(i);
                res.add(num / i);
            }
        }
        return res;
    }
}
