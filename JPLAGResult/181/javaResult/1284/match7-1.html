class Solution {
    public int sumFourDivisors(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            int[] res = check(num);
            if (res[0] == 4) {
                sum += res[1];
            
}
        }
        return sum;
    }
    
    private int[] check(int x) {
        Set<Integer> set = new HashSet<>();
        for (int i = 1; i <= (int)Math.sqrt(x); i++) 
{
            if (x % i == 0) {
                set.add(i);
                set.add(x/i);
            }
        }
        int sum = 0;
        for (int s : set) {
            sum += s;
        }
        return new int[]{set.size(), sum};
    }
}
