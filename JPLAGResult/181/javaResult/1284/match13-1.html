class Solution {
    public int sumFourDivisors(int[] nums) {
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            int count = 0;
            int sum = 0;
            int num = nums[i];
            for (int j = 1; j * j <= num; j++) {
                if (num % j == 0) {
                    count += j * j < num ? 2 : 1;
                    
if (count > 4) {
                        break;
                    }
                    sum += j + num / j;
                }
            }
            if (count == 4) {
                res += sum;
            }
        }
        return res;
    }
}
