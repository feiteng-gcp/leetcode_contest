class Solution {
    public boolean hasValidPath(int[][] grid) {
        int [][] dir = 
new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}}; // 0 right, 1 down, 2 left, 3 up
        
int [][] conn = new int[][] {{}, {0, -1, 2, -1}, {-1, 1, -1, 3}, {1, -1, -1, 3}, {-1, -1, 1, 0}, {3, 2, -1, -1}, {-1, 0, 3, -1}};
        int n = grid.length, m = grid[0].length;
        int x = 0, y = 0, road = grid[x][y];
        int curdir = Math.max(conn[road][0], conn[road][1]);
        if(x == n - 1 && y == m - 1) return true;
        //System.out.println(" ");
        //System.out.println(x + " " + y + " " + road + " " + curdir);
        for(int i = 0; i < 4; i++)
        {
            if(conn[road][i] != -1)
            {
                curdir = conn[road][i];
                while(true)
                {
                    if(x == n - 1 && y == m - 1) return true;
                    x = x + dir[curdir][0]; y = y + dir[curdir][1];            
                    if(x < 0 || x >= n || y < 0 || y >= m) return false;
                    road = grid[x][y];
                    curdir = conn[road][curdir];
                    if(curdir == -1) return false;
                    //System.out.println(x + " " + y + " " + road + " " + curdir);
                }
            }
        }
        
        
        return false;
    }
}
