class Solution {
    public boolean hasValidPath(int[][] grid) {
        Queue<int[]> q = new LinkedList<>();
        int[][] vis = new int[grid.length][grid[0].length];
        
q.offer(new int[]{0, 0});
        vis[0][0] = 1;
        while(!q.isEmpty()) {
            int[] p = q.poll();
            int x = p[0], y = p[1];
            if(x == grid.length - 1 && y == grid[0].length - 1) return true;
            switch(grid[x][y]) {
                case 1:
                    
if(y - 1 >= 0 && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6) && vis[x][y - 1] == 0) {
                        vis[x][y - 1] = 1;
                        q.offer(new int[]{x, y - 1});
                    }
                    if(y + 1 < grid[0].length && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5) && vis[x][y + 1] == 0) {
                        vis[x][y + 1] = 1;
                        q.offer(new int[]{x, y + 1});
                    }
                    break;
                case 2:
                    
if(x - 1 >= 0 && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4) && vis[x - 1][y] == 0) {
                        vis[x - 1][y] = 1;
                        q.offer(new int[]{x - 1, y});
                    }
                    if(x + 1 < grid.length && (grid[x+1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6) && vis[x + 1][y] == 0) {
                        vis[x + 1][y] = 1;
                        q.offer(new int[]{x + 1, y});
                    }
                    break;
                case 3:
                    
if(y - 1 >= 0 && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6) && vis[x][y - 1] == 0) {
                        vis[x][y - 1] = 1;
                        q.offer(new int[]{x, y - 1});
                    }
                    if(x + 1 < grid.length && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6) && vis[x + 1][y] == 0) {
                        vis[x + 1][y] = 1;
                        q.offer(new int[]{x + 1, y});
                    }
                    break;
                case 4:
                    
if(x + 1 < grid.length && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6) && vis[x + 1][y] == 0) {
                        vis[x + 1][y] = 1;
                        q.offer(new int[]{x + 1, y});
                    }
                    if(y + 1 < grid[0].length && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5) && vis[x][y + 1] == 0) {
                        vis[x][y + 1] = 1;
                        q.offer(new int[]{x, y + 1});
                    }
                    break;
                case 5:
                     
if(y - 1 >= 0 && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6) && vis[x][y - 1] == 0) {
                        vis[x][y - 1] = 1;
                        q.offer(new int[]{x, y - 1});
                    }
                    if(x - 1 >= 0 && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4) && vis[x - 1][y] == 0) {
                        vis[x - 1][y] = 1;
                        q.offer(new int[]{x - 1, y});
                    }
                    break;
                case 6:
                    
if(x - 1 >= 0 && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4) && vis[x - 1][y] == 0) {
                        vis[x - 1][y] = 1;
                        q.offer(new int[]{x - 1, y});
                    }
                    if(y + 1 < grid[0].length && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5) && vis[x][y + 1] == 0) {
                        vis[x][y + 1] = 1;
                        q.offer(new int[]{x, y + 1});
                    }
                    break;
            }
        }
        return false;
    }
}
