class Point{
    final int i;
    final int j;
    public Point(int i, int j){
        this.i = i;
        this.j = j;
    }
}
class Solution {
    private static final int LEFT = 0;
    private static final int RIGHT = 1;
    private static final int UP = 2;
    private static final int DOWN = 3;

    
    private static final boolean[][][] validMove = new boolean[7][7][4];
    static{
        validMove[1][1][RIGHT] = true;
        validMove[1][3][RIGHT] = true;
        validMove[1][5][RIGHT] = true;
        validMove[1][1][LEFT] = true;
        validMove[1][4][LEFT] = true;
        validMove[1][6][LEFT] = true;
        
        validMove[2][2][DOWN] = true;
        validMove[2][5][DOWN] = true;
        validMove[2][6][DOWN] = true;
        validMove[2][2][UP] = true;
        validMove[2][3][UP] = true;
        validMove[2][4][UP] = true;
      
        validMove[3][2][DOWN] = true;
        validMove[3][5][DOWN] = true;
        validMove[3][6][DOWN] = true;

        validMove[3][1][LEFT] = true;
        validMove[3][4][LEFT] = true;
        validMove[3][6][LEFT] = true;
        
        validMove[4][2][DOWN] = true;
        validMove[4][5][DOWN] = true;
        validMove[4][6][DOWN] = true;
        validMove[4][1][RIGHT] = true;
        validMove[4][3][RIGHT] = true;
        validMove[4][5][RIGHT] = true;
        
        validMove[5][2][UP] = true;
        validMove[5][3][UP] = true;
        validMove[5][4][UP] = true;
        validMove[5][1][LEFT] = true;
        validMove[5][4][LEFT] = true;
        validMove[5][6][LEFT] = true;
        
          
        validMove[6][2][UP] = true;
        validMove[6][3][UP] = true;
        validMove[6][4][UP] = true;
        validMove[6][1][RIGHT] = true;
        validMove[6][3][RIGHT] = true;
        validMove[6][5][RIGHT] = true;
    }
    
    
    public boolean hasValidPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        Queue<Point> queue = new ArrayDeque<>();
        boolean[][] visited = new boolean[m][n];
        visited[0][0] = true;
        queue.add(new Point(0,0));
        while(!queue.isEmpty()){
            Point p = queue.poll();
            // check four direction
            checkValid(grid, visited, p.i, p.j, p.i-1, p.j, UP, queue);
            checkValid(grid, visited, p.i, p.j, p.i+1, p.j, DOWN, queue);
            checkValid(grid, visited, p.i, p.j, p.i, p.j-1, LEFT, queue);
            checkValid(grid, visited, p.i, p.j, p.i, p.j+1, RIGHT, queue);
            if(visited[m-1][n-1]){
                return true;
            }
        }
        return false;
    }
    
    private void checkValid(int[][] grid, boolean[][] visited, int si, int sj, int ti, int tj, int direct, Queue<Point> queue){
        if(ti<0||tj<0||ti>=grid.length || tj>=grid[ti].length || visited[ti][tj]){
            return;
        }
        if(validMove[grid[si][sj]][grid[ti][tj]][direct]){
            visited[ti][tj] = true;
            queue.add(new Point(ti, tj));
        }
    }
  
    
    
}
