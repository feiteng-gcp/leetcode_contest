class Solution {
   int[][] dir(int d) {
            if (d == 1) {
                return new int[][]{{0, -1}, {0, 1}};
            } else if (d == 2) {
                return new int[][]{{-1, 0}, {1, 0}};
            } else if (d == 3) {
                return new int[][]{{0, -1}, {1, 0}};
            } else if (d == 4) {
                return new int[][]{{1, 0}, {0, 1}};
            } else if (d == 5) {
                return new int[][]{{0, -1}, {-1, 0}};
            } else {
                return new int[][]{{-1, 0}, {0, 1}};
            }
        }

        boolean dfs(int n, int m, int x, int y, boolean[][] visit, int[][] grid, int fx, int fy) {
            int[][] d = dir(grid[x][y]);
            if (fx != -1) {
                boolean f = false;
                for (int[] dd: d) {
                    int dx = x + dd[0];
                    int dy = y + dd[1];
                    if (dx == fx && dy == fy) {
                        f = true;
                        break;
                    }
                }
                if (!f) return false;
            }
            if (x == n - 1 && y == m - 1) return true;
            visit[x][y] = true;
            for (int[] dd: d) {
                int dx = x + dd[0];
                int dy = y + dd[1];
                if (dx >= 0 && dx < n && dy >= 0 && dy < m && ! visit[dx][dy]) {
                    if (dfs(n, m, dx, dy, visit, grid, x, y)) return true;
                }
            }
            return false;
        }

        public boolean hasValidPath(int[][] grid) {
            int n = grid.length;
            int m = grid[0].length;
            boolean[][] visit = new boolean[n][m];
            return dfs(n, m, 0, 0, visit, grid, -1, -1);
        }
}
