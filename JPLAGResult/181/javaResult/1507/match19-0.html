class Solution {
    public boolean hasValidPath(int[][] grid) {
        int n = grid.length, m = grid[0].length;
        boolean[][][] seen = new boolean[n][m][4];
        Deque<int[]> queue = new ArrayDeque<>();
        queue.add(new int[]{0, 0, 0});
        queue.add(new int[]{0, 0, 1});
        queue.add(new int[]{0, 0, 2});
        queue.add(new int[]{0, 0, 3});
        seen[0][0][0] =seen[0][0][1] =seen[0][0][2] =seen[0][0][3] = true;
        while(!queue.isEmpty()){
            int[] cur = queue.poll();
            int r = cur[0], c = cur[1];
            int nr = 0, nc = 0;
            if(grid[r][c] == 1){
                if(cur[2] == 0 || cur[2] == 2){
                    continue;
                }
                if(r == n - 1 && c == m - 1){
                    return true;
                }
                if(cur[2] == 1){
                    nr = r;
                    nc = c - 1;
                    if(isValid(seen, nr, nc, 1)){
                        seen[nr][nc][1] = true;
                        queue.add(new int[]{nr, nc, 1});
                    }
                } else if(cur[2] == 3){
                    nr = r;
                    nc = c + 1;
                    if(isValid(seen, nr, nc, 3)){
                        seen[nr][nc][3] = true;
                        queue.add(new int[]{nr, nc, 3});
                    }
                }
            } else if(grid[r][c] == 2){
                if(cur[2] == 1 || cur[2] == 3){
                    continue;
                }
                if(r == n - 1 && c == m - 1){
                    return true;
                }
                if(cur[2] == 0){
                    nr = r + 1;
                    nc = c;
                    if(isValid(seen, nr, nc, 0)){
                        seen[nr][nc][0] = true;
                        queue.add(new int[]{nr, nc, 0});
                    }
                } else if(cur[2] == 2){
                    nr = r - 1;
                    nc = c;
                    if(isValid(seen, nr, nc, 2)){
                        seen[nr][nc][2] = true;
                        queue.add(new int[]{nr, nc, 2});
                    }
                }
            } else if(grid[r][c] == 3){
                if(cur[2] == 0 || cur[2] == 1){
                    continue;
                }
                if(r == n - 1 && c == m - 1){
                    return true;
                }
                if(cur[2] == 2){
                    nr = r;
                    nc = c - 1;
                    if(isValid(seen, nr, nc, 1)){
                        seen[nr][nc][1] = true;
                        queue.add(new int[]{nr, nc, 1});
                    }
                } else if(cur[2] == 3){
                    nr = r + 1;
                    nc = c;
                    if(isValid(seen, nr, nc, 0)){
                        seen[nr][nc][0] = true;
                        queue.add(new int[]{nr, nc, 0});
                    }
                }
            } else if(grid[r][c] == 4){
                if(cur[2] == 0 || cur[2] == 3){
                    continue;
                }
                if(r == n - 1 && c == m - 1){
                    return true;
                }
                if(cur[2] == 1){
                    nr = r + 1;
                    nc = c;
                    if(isValid(seen, nr, nc, 0)){
                        seen[nr][nc][0] = true;
                        queue.add(new int[]{nr, nc, 0});
                    }
                } else if(cur[2] == 2){
                    nr = r;
                    nc = c + 1;
                    if(isValid(seen, nr, nc, 3)){
                        seen[nr][nc][3] = true;
                        queue.add(new int[]{nr, nc, 3});
                    }
                }
            } else if(grid[r][c] == 5){
                if(cur[2] == 1 || cur[2] == 2){
                    continue;
                }
                if(r == n - 1 && c == m - 1){
                    return true;
                }
                if(cur[2] == 0){
                    nr = r;
                    nc = c - 1;
                    if(isValid(seen, nr, nc, 1)){
                        seen[nr][nc][1] = true;
                        queue.add(new int[]{nr, nc, 1});
                    }
                } else if(cur[2] == 3){
                    nr = r - 1;
                    nc = c;
                    if(isValid(seen, nr, nc, 2)){
                        seen[nr][nc][2] = true;
                        queue.add(new int[]{nr, nc, 2});
                    }
                }
            } else if(grid[r][c] == 6){
                if(cur[2] == 2 || cur[2] == 3){
                    continue;
                }
                if(r == n - 1 && c == m - 1){
                    return true;
                }
                if(cur[2] == 0){
                    nr = r;
                    nc = c + 1;
                    if(isValid(seen, nr, nc, 3)){
                        seen[nr][nc][3] = true;
                        queue.add(new int[]{nr, nc, 3});
                    }
                } else if(cur[2] == 1){
                    nr = r - 1;
                    nc = c;
                    if(isValid(seen, nr, nc, 2)){
                        seen[nr][nc][2] = true;
                        queue.add(new int[]{nr, nc, 2});
                    }
                }
            }
        }
        return false;
    }
    
    private boolean isValid(boolean[][][] seen, int r, int c, int dir){
        int n = seen.length, m = seen[0].length;
        return r >= 0 && r < n && c >= 0 && c < m && !seen[r][c][dir];
    }
}
