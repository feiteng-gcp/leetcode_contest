class Solution {
    int UP = 0;
    int RIGHT = 1;
    int DOWN = 2;
    int LEFT = 3;
    
    int[][][] path = 
{{{LEFT, RIGHT}, {RIGHT, LEFT}},
                      {{UP, DOWN}, {DOWN, UP}},
                      {{LEFT, DOWN}, {DOWN, LEFT}},
                      {{RIGHT, DOWN}, {DOWN, RIGHT}},
                      {{LEFT, UP}, {UP, LEFT}},
                      {{RIGHT, UP}, {UP, RIGHT}}};
    int x = 0;
    int y = 0;
    int direct = 0;
    
    public boolean hasValidPath(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        
        boolean res = false;
        
        for(int i = 0; i < 4; i++){
            boolean[][] record = new boolean[n][m];
            x = 0;
            y = 0;
            direct = i;
            
            while(true){
                if(x < 0 || y < 0 || x >= n || y >= m || record[x][y]){
                    break;
                }
                
                
                record[x][y] = true;
                int cur = grid[x][y] - 1;
                if(direct == path[cur][0][0]){
                    if(x == n - 1 && y == m - 1){
                        res = true;
                        break;
                    }
                    move(cur, 0);
                }
                else if(direct == path[cur][1][0]){
                    if(x == n - 1 && y == m - 1){
                        res = true;
                        break;
                    }
                    move(cur, 1);
                }
                else{
                    break;
                }
                
            }
            
            if(res){
                return true;
            }
        }
        
        
        return false;
    }
    
    private void move(int cur, int loc){
        int end = path[cur][loc][1];
        switch (end){
            case 0:
                direct = DOWN;
                x--;
                break;
            case 2:
                direct = UP;
                x++;
                break;
            case 3:
                direct = RIGHT;
                y--;
                break;
            case 1:
                direct = LEFT;
                y++;
                break;
        }
    }
}
