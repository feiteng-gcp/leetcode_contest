import java.util.LinkedList;
import java.util.Queue;

class Solution {
    
class Node {
        public int x;
        public int y;

        public Node(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

    public boolean hasValidPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[][][] chs = getChange();
        
Queue<Node> que = new LinkedList<>();
        boolean[][] vis = new boolean[m][n];
        vis[0][0] = true;
        que.add(new Node(0, 0));
        while (!que.isEmpty()) {
            Node cur = que.remove();
            if (cur.x == m - 1 && cur.y == n - 1) {
                return true;
            }
            int[][] ch = chs[grid[cur.x][cur.y]];
            for (int i = 0; i <= 1; i++) {
                int nx = cur.x + ch[i][0];
                int ny = cur.y + ch[i][1];
                if (nx >= 0 && nx <= m - 1 && ny >= 0 && ny <= n - 1 && !vis[nx][ny]) {
                    Node nn = new Node(nx, ny);
                    if (isConnected(cur, nn, chs[grid[nx][ny]])) {
                        vis[nx][ny] = true;
                        que.add(new Node(nx, ny));
                    }
                }
            }
        
}
        return false;
    }

    private boolean isConnected(Node n1, Node n2, int[][] ch) {
        for (int i = 0; i <= 1; i++) {
            int nx = n2.x + ch[i][0];
            int ny = n2.y + ch[i][1];
            if (nx == n1.x && ny == n1.y) {
                return true;
            }
        }
        return false;
    }

    private int[][][] getChange() {
        int[][][] ch = new int[7][2][2];
        
ch[1][0][0] = 0;
        ch[1][0][1] = -1;
        ch[1][1][0] = 0;
        ch[1][1][1] = 1;
        ch[2][0][0] = -1;
        ch[2][0][1] = 0;
        ch[2][1][0] = 1;
        ch[2][1][1] = 0;
        ch[3][0][0] = 0;
        ch[3][0][1] = -1;
        ch[3][1][0] = 1;
        ch[3][1][1] = 0;
        ch[4][0][0] = 0;
        ch[4][0][1] = 1;
        ch[4][1][0] = 1;
        ch[4][1][1] = 0;
        ch[5][0][0] = 0;
        ch[5][0][1] = -1;
        ch[5][1][0] = -1;
        ch[5][1][1] = 0;
        ch[6][0][0] = -1;
        ch[6][0][1] = 0;
        ch[6][1][0] = 0;
        ch[6][1][1] = 1;
        return ch;
    }
}
