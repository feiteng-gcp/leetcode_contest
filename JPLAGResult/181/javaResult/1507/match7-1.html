class Solution {
    List<int[][]> dirs;
        
    public boolean hasValidPath(int[][] grid) {
        // Do a graph traversal from (0,0)
        // Stop when we reach last cell, or no more cells left we can reach.
        int M = grid.length;
        int N = grid[0].length;
        
        // We're at the last cell already.
        if (M == 1 && N == 1) return true;
        
        dirs = new ArrayList<>();
        dirs.add(new int[][]{});        
        dirs.add(
new int[][] {{0, -1}, {0, 1}});
        dirs.add(
new int[][] {{-1, 0}, {1, 0}});
        dirs.add(
new int[][] {{0, -1}, {1, 0}});
        dirs.add(
new int[][] {{0, 1}, {1, 0}});
        dirs.add(
new int[][] {{-1, 0}, {0, -1}});
        dirs.add(
new int[][] {{-1, 0}, {0, 1}});
        
        LinkedList<int[]> queue = new LinkedList<>();
        queue.add(new int[] {0, 0});
        boolean[][] visited = new boolean[M][N];
        visited[0][0] = true;
        
        while (!queue.isEmpty()) {
            int[] curr = queue.removeFirst();
            int r = curr[0];
            int c = curr[1];
            
            if (r == M-1 && c == N-1) return true;
            
            // Determine what tile we have currently.
            // Then check if we can connect to adjacent tiles.
            int[][] d = dirs.get(grid[r][c]);
            
            for (int[] dir : d) {
                int nextR = r + dir[0];
                int nextC = c + dir[1];
                
                if (nextR >= 0 && nextR < M && nextC >= 0 && nextC < N && !visited[nextR][nextC]) {
                    if (canMove(grid[nextR][nextC], dir)) {
                        queue.addLast(new int[] {nextR, nextC});
                        visited[nextR][nextC] = true;
                    }
                }
            
}
        }
        
        return false;
    }
    
    // Check if it's possible to move to tile of type from a direction.
    public boolean canMove(int type, int[] direction) {
        if (type == 1) {
            return (direction[0] == 0 && direction[1] == 1) || (direction[0] == 0 && direction[1] == -1);
        }
        else 
if (type == 2) {
            return (direction[0] == 1 && direction[1] == 0) || (direction[0] == -1 && direction[1] == 0);
        }
        else if (type == 3) {
            return (direction[0] == 0 && direction[1] == 1) || (direction[0] == -1 && direction[1] == 0);
        }
        else if (type == 4) {
            return (direction[0] == 0 && direction[1] == -1) || (direction[0] == -1 && direction[1] == 0);
        
}
        else if (type == 5) {
            return (direction[0] == 0 && direction[1] == 1) || (direction[0] == 1 && direction[1] == 0);
        }
        else if (type == 6) {
            return (direction[0] == 1 && direction[1] == 0) || (direction[0] == 0 && direction[1] == -1);
        }
        
        return false;
    }
}
