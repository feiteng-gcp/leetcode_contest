class Solution {
    public boolean hasValidPath(int[][] grid) {
		int n = grid.length;
		int m = grid[0].length;
		boolean[][] visited = new boolean[n][m];
		Queue<int[]> queue = new LinkedList<>();
		visited[0][0] = true;
		queue.add(new int[] {0, 0});
		while (!queue.isEmpty()) {
			int[] cur = queue.poll();
			int x = cur[0];
			int y = cur[1];
			if (grid[x][y] == 1) {
				int nx = x;
				int ny = y - 1;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
				ny = y + 1;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 1 || grid[nx][ny] == 3 || grid[nx][ny] == 5) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
			} else if (grid[x][y] == 2) {
				int nx = x-1;
				int ny = y;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
				nx = x + 1;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
			} else if (grid[x][y] == 3) {
				int nx = x;
				int ny = y-1;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
				nx = x + 1;
				ny = y;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
			} else if (grid[x][y] == 4) {
				int nx = x;
				int ny = y+1;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 1 || grid[nx][ny] == 3 || grid[nx][ny] == 5) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
				nx = x + 1;
				ny = y;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 2 || grid[nx][ny] == 5 || grid[nx][ny] == 6) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
			} else if (grid[x][y] == 5) {
				int nx = x;
				int ny = y-1;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 1 || grid[nx][ny] == 4 || grid[nx][ny] == 6) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
				nx = x - 1;
				ny = y;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
			} else if (grid[x][y] == 6) {
				int nx = x;
				int ny = y+1;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 1 || grid[nx][ny] == 3 || grid[nx][ny] == 5) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
				nx = x - 1;
				ny = y;
				if (nx >= 0 && nx < n && ny >= 0 && ny < m) if (!visited[nx][ny]) {
					if (grid[nx][ny] == 2 || grid[nx][ny] == 3 || grid[nx][ny] == 4) {
						visited[nx][ny] = true;
						queue.add(new int[] {nx, ny});
					}
				}
			}
		}
		return visited[n-1][m-1];
	}
}
