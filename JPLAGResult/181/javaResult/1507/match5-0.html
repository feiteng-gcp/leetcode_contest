class Solution {
    public boolean hasValidPath(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        boolean[][] visited = new boolean[m][n];
        Queue<int[]> q = new LinkedList<>();
        q.add(new int[]{0, 0});
        
        visited[0][0] = true;
        while (!q.isEmpty()) {
            int[] c = q.poll();
            if (c[0] == m - 1 && c[1] == n - 1) {
                return true;
            }
            int x = c[0];
            int y = c[1];
            int t = grid[c[0]][c[1]];
            if (t == 1) {
                if (y > 0 && !visited[x][y - 1] && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6)) {
                    visited[x][y - 1] = true;
                    q.add(new int[] {x, y - 1});
                }
                if (y + 1 < n && !visited[x][y + 1]  && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5)) {
                    visited[x][y + 1] = true;
                    q.add(new int[] {x, y + 1});
                }
            }
            if (t == 2) {
                if (x > 0 && !visited[x - 1][y] && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4)) {
                    visited[x - 1][y] = true;
                    q.add(new int[] {x - 1, y});
                }
                if (x + 1 < m && !visited[x + 1][y] && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6)) {
                    visited[x + 1][y] = true;
                    q.add(new int[] {x + 1, y});
                }
            }
            if (t == 3) {
                if (y > 0 && !visited[x][y - 1] && (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 6)) {
                    visited[x][y - 1] = true;
                    q.add(new int[] {x, y - 1});
                }
                if (x + 1 < m && !visited[x + 1][y] && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6)) {
                    visited[x + 1][y] = true;
                    q.add(new int[] {x + 1, y});
                }
            }
            if (t == 4) {
                if (y + 1 < n && !visited[x][y + 1] && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5)) {
                    visited[x][y + 1] = true;
                    q.add(new int[] {x, y + 1});
                }
                if (x + 1 < m && !visited[x + 1][y] && (grid[x + 1][y] == 2 || grid[x + 1][y] == 5 || grid[x + 1][y] == 6)) {
                    visited[x + 1][y] = true;
                    q.add(new int[] {x + 1, y});
                }
            }
            if (t == 5) {
                if (y > 0 && !visited[x][y - 1]&& (grid[x][y - 1] == 1 || grid[x][y - 1] == 4 || grid[x][y - 1] == 5)) {
                    visited[x][y - 1] = true;
                    q.add(new int[] {x, y - 1});
                }
                if (x > 0 && !visited[x - 1][y] && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4)) {
                    visited[x - 1][y] = true;
                    q.add(new int[] {x - 1, y});
                }
            }
            if (t == 6) {
                if (y + 1 < n && !visited[x][y + 1] && (grid[x][y + 1] == 1 || grid[x][y + 1] == 3 || grid[x][y + 1] == 5)) {
                    visited[x][y + 1] = true;
                    q.add(new int[] {x, y + 1});
                }
                if (x > 0 && !visited[x - 1][y] && (grid[x - 1][y] == 2 || grid[x - 1][y] == 3 || grid[x - 1][y] == 4)) {
                    visited[x - 1][y] = true;
                    q.add(new int[] {x - 1, y});
                }
            }
        }
        return false;
    }
}
