import java.util.Deque;
import java.util.LinkedList;

class Solution {
    int m, n;
    
int move[][] = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    int check[][][] = 
new int[][][]{
            {{1, 3, 5}, {}, {}, {1, 3, 5}, {}, {1, 3, 5}},
            {{1, 4, 6}, {}, {1, 4, 6}, {}, {1, 4, 6}, {}},
            {{}, {2,5,6}, {2,5,6}, {2,5,6}, {}, {}},
            {{}, {2,3,4}, {}, {}, {2,3,4}, {2,3,4}}
    };

    public boolean hasValidPath(int[][] grid) {
        m = grid.length;
        n = grid[0].length;

        Deque<int[]> queue = new LinkedList<>();

        queue.add(new int[]{0, 0});
        int vis[][] = new int[m][n];

        while (!queue.isEmpty()) {
            int[] record = queue.poll();
            int x = record[0];
            int y = record[1];

            if (x == m-1 && y == n-1) {
                return true;
            }

            vis[x][y] = 1;

            int type = grid[x][y];

            for (int i = 0; i < 4; i++) {
                int newX = x + move[i][0];
                int newY = y + move[i][1];

                if (newX >= 0 && newX < m && newY >= 0 && newY < n && vis[newX][newY] == 0) {
                    if (check[i][type-1].length > 0) {
                        int c[] = check[i][type-1];

                        for (int num : c) {
                            if (num == grid[newX][newY]) {
                                if (newX == m-1 && newY == n-1) {
                                    return true;
                                }
                                queue.add(new int[]{newX, newY});
                                vis[newX][newY] = 1;
                                break;
                            }
                        }
                    }
                }
            }
        }

        return false;
    }
}
