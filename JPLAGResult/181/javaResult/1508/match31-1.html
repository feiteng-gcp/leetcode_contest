class Solution {
    public String longestPrefix(String s) {
        if (s.length() <= 1) {
            return "";
        }
        
int n = s.length();
        char[] array = s.toCharArray();
        int[] lps = new int[n];
        int len = 0;
        int i = 1;
        while (i < n) {
            if (array[i] == array[len]) {
                len++;
                lps[i] = len;
                i++;
            } else {
                if (len != 0) {
                    len = lps[len - 1];
                
} else {
                    lps[i] = 0;
                    i++;
                }
            }
        }
        if (lps[n - 1] == 0) {
            return "";
        }
        return s.substring(n - lps[n - 1], n);
    }
}
