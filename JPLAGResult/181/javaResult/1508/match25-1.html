class Solution {
    
public String longestPrefix(String s) {
        int[] lens = new int[s.length()];
        int len = 0;
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(len)) {
                len++;
                lens[i] = len;
            } else {
                if (len == 0) lens[i] = 0;
                else 
{
                    len = lens[len - 1];
                    i--;
                }
            }
        }
        
        return len > 0 ? s.substring(0, len) : "";
    }
}
