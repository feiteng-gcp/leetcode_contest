class Solution {
    public String longestPrefix(String s) {
        int[] kmp = new int[s.length()];
        int len = 0;
        int i = 1;
        int n = s.length();
        while(i<n){
         if(s.charAt(i)== s.charAt(len)){
             len++;
             kmp[i]= len;
             i++;
         }else{
             if(len!=0){
                len=kmp[len-1];
             }
             else{
                kmp[len] = 0;
                i++;
             }
         }   
        }
        int res = kmp[n-1];
        return s.substring(0,res);
    }
}
