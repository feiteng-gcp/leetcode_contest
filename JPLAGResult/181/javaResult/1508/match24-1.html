class Solution {
    public String longestPrefix(String s) {
        if (s.length() == 1) {
            return "";
        }
        
        int length = 0; 
        
        int state[] = new int[s.length()];
       
        
int index = 1;
        
        while (index < s.length()) { 
            if (s.charAt(index) == s.charAt(length)) { 
                length++; 
                state[index] = length; 
                index++;
            } else {
                if (length == 0) {
                    state[index] = 0;
                    index++;
                } else {
                    length = state[length - 1];
                }
            }
        } 
        return s.substring(0, state[s.length() - 1]); 
        
    }
}
