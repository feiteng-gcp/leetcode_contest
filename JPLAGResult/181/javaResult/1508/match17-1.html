class Solution {
    
public String longestPrefix(String s) {
        int[] backFunc = new int[s.length() + 1];
        backFunc[0] = backFunc[1] = 0;
        for (int i = 1; i < s.length(); ++i) {
            int testing = i - 1;
            while (backFunc[testing] != testing) {
                if (s.charAt(backFunc[testing]) == s.charAt(i - 1)) {
                    backFunc[i] = backFunc[testing] + 1;
                    break;
                } else {
                    testing = backFunc[testing];
                }
            }
        }
        String tmp = s.substring(1);
        s = s + "$";
        
int curpos = 0;
        for (int i = 0; i < tmp.length(); ++i) {
            while (curpos != backFunc[curpos]) {
                if (tmp.charAt(i) == s.charAt(curpos)) {
                    ++curpos;
                    break;
                } else {
                    curpos = backFunc[curpos];
                }
            }
            if (curpos == 0 && s.charAt(0) == tmp.charAt(i)) {
                ++curpos;
            }
        }
        return s.substring(0, curpos);
    }
}
