class Solution {
    public String longestPrefix(String s){
        int len = longestPrefixLen(s);
        if(len == 0) return "";
        return s.substring(0, len);
    }
    public int longestPrefixLen(String s) { 
        int n = s.length(); 
        int pp[] = new int[n]; 
        pp[0] = 0;
        int len = 0; 
        int i = 1; 
        while (i < n) 
        { 
            if (s.charAt(i) == s.charAt(len)) 
            { 
                len++; 
                pp[i] = len; 
                i++; 
            } 
            else
            {
                if (len != 0) 
                { 
                    len = pp[len-1]; 
                }
                else 
                { 
                    pp[i] = 0; 
                    i++; 
                } 
            } 
        } 
      
        int res = pp[n-1]; 
        return res;
    } 
}
