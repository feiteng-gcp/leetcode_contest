class Solution {
    // Create the prefix suffix array.
    public int[] getLPS(String s) {
        
int N = s.length();
        
        int[] arr = new int[N];
        
        int index = 1;
        int length = 0;
        
        while (index < N) {
            if (s.charAt(index) == s.charAt(length)) 
{
                length++;
                
                arr[index] = length;
                
                index++;                
            }
            else {
                if (length == 0) {
                    index++;
                }
                else {
                    length = arr[length-1];
                }
            }
        }
        
        return arr;    
    }
    
    public String longestPrefix(String s) {
        if (s == null || s.length() < 2) return "";
        
        int[] arr = getLPS(s);
        
        return s.substring(0, arr[s.length() - 1]);
    }
}
