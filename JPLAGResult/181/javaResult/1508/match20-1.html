class Solution {
    public String longestPrefix(String s) {
        
int end=helper(s);
        return s.substring(0,end);
    }
    
    public int helper(String s){   //KMP算法
    int[] table = new int[s.length()+1];
    int j=0,k=-1;table[0]=-1;
    while(j<s.length())
    {
        if(k==-1||s.charAt(k)==s.charAt(j))table[++j]=++k;
        else
        k=table[k];
    }

    return table[s.length()];
}
}
