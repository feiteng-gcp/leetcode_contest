import java.util.*;

class Solution {
    public String longestPrefix(String s) {
        int[] lps = computeLPSArray(s);
        int res = lps[s.length() - 1];
        return s.substring(0, res);
    }

    int[] computeLPSArray(String pat) {
        int n = pat.length();
        int len = 0;
        int i = 1;
        int[] res = new int[n];
        res[0] = 0;

        while (i < n) {
            if (pat.charAt(i) == pat.charAt(len)) {
                len++;
                res[i] = len;
                i++;
            } else {
                if (len != 0) {
                    len = res[len - 1];
                } else {
                    res[i] = len;
                    i++;
                }
            }
        }
        return res;
    }


}
