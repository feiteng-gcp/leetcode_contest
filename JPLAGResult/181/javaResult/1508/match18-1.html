class Solution {
    public String longestPrefix(String s) {
        int N = s.length();
        if (N<=1) return "";
        
        int[] dict = new int[N];
        
int len = 0;
        int i = 1;
        while(i < N) {
            if (s.charAt(i) == s.charAt(len)){
                len++;
                dict[i++] = len;
            } else {
                if (len != 0) {
                    len = dict[len-1];
                } else {
                    dict[i] = 0;
                    i++;
                }
            }
        }
        
        int max = dict[N-1];
        return s.substring(0, max);
    }
}
