class Solution {
    public int[] createTargetArray(int[] nums, int[] index) {
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            int n = nums[i];
            int in = index[i];
            list.add(in, n);
        }
        
        int[] re = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            re[i] = list.get(i);
        } 
        return re;
    }
}
