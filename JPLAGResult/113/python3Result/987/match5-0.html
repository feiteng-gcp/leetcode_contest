class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        if len(deck)==1:
            return deck
        deck.sort(key=lambda x:-x)
        from collections import deque
        d=deque()
        d.append(deck[0])
        for x in deck:
            if x!=deck[0]:
                last=d.pop()
                d.appendleft(last)
                d.appendleft(x)
        return list(d)
