class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        n=len(deck)
        if n<3:
            return deck
        ans=[0]*n
        deck.sort()
        if n%2==0:
            for i in range(n//2):
                ans[2*i]=deck[i]
            t=self.deckRevealedIncreasing
(deck[n//2:])
            for i in range(n//2):
                ans[2*i+1]=t[i]
        else:
            for i in range(n//2+1):
                ans[2*i]=deck[i]
            t=self.deckRevealedIncreasing(deck[n//2+1:])
            ans[1]=t[-1]
            for i in range(len(t)-1):
                ans[2*i+3]=t[i]
        return ans
