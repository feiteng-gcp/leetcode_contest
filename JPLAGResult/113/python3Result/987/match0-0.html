class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        deck.sort(reverse=True)
        r = collections.deque()
        r.append(deck[0])
        for x in deck[1:]:
            r.appendleft(r.pop())
            r.appendleft(x)
        return list(r)
