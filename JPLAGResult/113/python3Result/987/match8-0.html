class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        if not deck:
            return []
        deck = sorted(deck)
        n = len(deck)
        m = n
        cur_idx = list(range(n))
        ret_idx = []
        while len(ret_idx) < n:
            ret_idx += [cur_idx[i] for i in range(m) if i % 2 == 0]
            cur_idx = [cur_idx[i] for i in range(m) if i % 2 == 1]
            if m % 2 == 1 and m > 1:
                cur_idx = cur_idx[1:] + [cur_idx[0]]
            m = len(cur_idx)
        # print(ret_idx)
        new_deck = [0] * n
        for j, i in enumerate(ret_idx):
            new_deck[i] = deck[j]
        return new_deck
