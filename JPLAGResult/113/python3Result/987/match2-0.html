from collections import deque

class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        
        if len(deck) == 1:
            return deck
        
        if len(deck) == 2:
            return list(sorted(deck))
        
        sorted_deck = list(sorted(deck))[::-1]
        
        ans = deque()
        ans.append(sorted_deck[0])
        
        for i in sorted_deck[1:]:
            k = ans.pop()
            ans.appendleft(k)
            ans.appendleft(i)
        
        return list(ans)
