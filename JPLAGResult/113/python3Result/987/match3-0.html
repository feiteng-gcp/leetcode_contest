class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """

        deck = sorted(deck, reverse=True)
        res = []
        for e in deck:
            if len(res) == 0:
                res.append(e)
            else:
                tmp = res.pop()
                res.insert(0,tmp)
                res.insert(0,e)
        return res
