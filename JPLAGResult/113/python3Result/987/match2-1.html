from collections import deque
class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        deck = sorted(deck, reverse=True)
        q = deque
([deck[0]])
        for card in deck[1:]:
            tail = q.pop()
            q.appendleft(tail)
            q.appendleft(card)
        return list(q)
