class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        if len(deck)==0:
            return []
        deck.sort
()
        N=len(deck)
        ans=[0]*N
        a=list(range(N))
        i=0
        while len(a)>0:
            b=a.pop(0)
            ans[b]=deck[i]
            i=i+1
            if len(a)>1:
                a=a[1:]+[a[0]]
        return ans
