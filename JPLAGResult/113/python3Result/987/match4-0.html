class Solution:
    def deckRevealedIncreasing(self, deck):
        """
        :type deck: List[int]
        :rtype: List[int]
        """
        deck.sort(reverse= True)
        res = collections.deque()
        for i in deck:
            if not res:
                res.append(i)
            else:
                p = res[-1]
                res.pop()
                res.appendleft(p)
                res.appendleft(i)
        return list(res)
