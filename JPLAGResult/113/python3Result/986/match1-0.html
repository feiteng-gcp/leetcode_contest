class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        import itertools
        maxh, maxm = 0, 0
        maxt = ''
        for p in itertools.permutations(A):
            # print(p)
            h = p[0] * 10 + p[1]
            m = p[2] * 10 + p[3]
            if 0 <= h <= 23 and 0 <= m <= 59:
                if (h, m) >= (maxh, maxm):
                    maxh, maxm = h, m
                    maxt = '{0}{1}:{2}{3}'.format(*p)
        return maxt
