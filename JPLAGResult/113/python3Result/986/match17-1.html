class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        min_time=-1
        for i in range(4):
            for j in range(4):
                if i==j:
                    continue
                if A[i]*10+A[j]>23:
                    
continue
                for k in range(4):
                    if (i==k) or (j==k):
                        continue
                    for m in range(4):
                        if (i==m) or (j==m) or (k==m):
                            continue
                        if A[k]*10+A[m]>59:
                            continue
                        if 60*(A[i]*10+A[j])+A[k]*10+A[m]>min_time:
                            min_time=60*(A[i]*10+A[j])+A[k]*10+A[m]
                            s1=str(A[i]*10+A[j])
                            if len(s1)<2:
                                s1='0'+s1
                            s2=str(A[k]*10+A[m])
                            if len(s2)<2:
                                s2='0'+s2
                            ans=s1+":"+s2
        if min_time>-1:
            return ans
        else:
            return ""
