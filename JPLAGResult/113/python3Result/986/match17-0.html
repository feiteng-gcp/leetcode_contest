class Solution:
    def is_valid_time(self, s):
        return 0 <= int(s[:2]) <= 23 and 0 <= int(s[2:]) <= 59
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        A = [str(a) for a in sorted(A, reverse=True)]
        for i0 in range(4):
            for i1 in range(4):
                if i0 == i1: continue
                for i2 in range(4):
                    if i2 == i0 or i2 == i1: continue
                    for i3 in range(4):
                        if i3 == i0 or i3 == i1 or i3 == i2: continue
                        a, b, c, d = A[i0], A[i1], A[i2], A[i3]
                        if self.is_valid_time(a+b+c+d):
                            return a+b+':'+c+d
        return ''
