import itertools

class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        
        perm = itertools.permutations(A)
        res = "0000"
        find = False
        for cur in perm:
            cur = [str(x) for x in cur]
            hour = int(''.join(cur[:2]))
            minute = int(''.join(cur[2:]))
            #print(cur, hour, minute)
            if hour <= 23 and minute <= 59:
                if ''.join(cur) >= res:
                    find = True
                    res = ''.join(cur)
        return res[0] + res[1] + ":" + res[2] + res[3] if find else ""

s = Solution()
