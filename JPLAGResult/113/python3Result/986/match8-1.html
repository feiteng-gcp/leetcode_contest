import itertools
class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        def totime(list):
            return "%d%d:%d%d"%(list)
        allp = list(itertools.permutations(A))
        largest = ""
        for p in allp:
            t = totime
(p)
            h = int(t.split(":")[0])
            m = int(t.split(":")[1])
            if h >= 24 or m >= 60:
                continue
            else:
                if largest == "":
                    largest = t
                else:
                    lh = int(largest.split(":")[0])
                    lm = int(largest.split(":")[1])
                    if h > lh or (h==lh) and (m>lm):
                        largest = t
        return largest
