class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        largest, ret = None, None
        for p in itertools.permutations(A):
            hour, second = p[0] * 10 + p[1], p[2] * 10 + p
[3]
            if 0 <= hour < 24 and 0 <= second < 60:
                time = hour * 60 + second
                if largest is None or time > largest:
                    largest = time
                    ret = str(p[0]) + str(p[1]) + ':' + str(p[2]) + str(p[3])
        if largest is None:
            return ''
        return ret
