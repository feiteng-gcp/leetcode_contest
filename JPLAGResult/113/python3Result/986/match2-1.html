import itertools
class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        mt 
= -1
        for d in itertools.permutations(A):
            h = d[0]*10+d[1]
            m = d[2]*10+d[3]
            if h<=23 and m <= 59:
                if h*60+m > mt:
                    mt = h*60+m
        if mt == -1:
            return ""
        else:
            return "{:02}:{:02}".format(mt // 60, mt % 60)
