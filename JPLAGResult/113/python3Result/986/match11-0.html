class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        
        from itertools import permutations
        
        max_hour = -float('inf') 
        max_minute = -float('inf')
        
        for order in permutations(A, 4):
            hour = order[0]*10+order[1]
            minute = order[2]*10+order[3]
            if 0 <= hour <= 23 and 0 <= minute < 59:
                if hour > max_hour or (hour == max_hour and minute > max_minute):
                    max_hour = hour
                    max_minute = minute
        if max_hour == -float('inf'):                   
            return ''
        return "{:02d}:{:02d}".format(max_hour, max_minute)
