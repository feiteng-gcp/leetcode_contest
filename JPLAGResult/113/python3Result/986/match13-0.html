class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        maxs=-1
        maxi=""
        for i in range(4):
            for j in range(4):
                for k in range(4):
                    for t in range(4):
                        if i!=j and i!=k and i!=t and j!=k and j!=t and k!=t:
                            if A[i]==2 and A[j]<=3 and A[k]<=5 and A[t]<=9:
                                if (A[i]*10+A[j])*60+A[k]*10+A[t]>maxs:
                                    maxs=(A[i]*10+A[j])*60+A[k]*10+A[t]
                                    maxi=""+chr(ord('0')+A[i])+chr(ord('0')+A[j])+":"+chr(ord('0')+A[k])+chr(ord('0')+A[t])
                            elif A[i]<=1 and A[j]<=9 and A[k]<=5 and A[t]<=9:
                                
                                if (A[i]*10+A[j])*60+A[k]*10+A[t]>maxs:
                                    maxs=(A[i]*10+A[j])*60+A[k]*10+A[t]
                                    maxi=""+chr(ord('0')+A[i])+chr(ord('0')+A[j])+":"+chr(ord('0')+A[k])+chr(ord('0')+A[t])
        return maxi
