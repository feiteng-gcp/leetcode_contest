class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        hh = -1
        mm = -1
        for a in range(len(A)):
            for b in range(len(A)):
                if a != b:
                    for c in range(len(A)):
                        if a!=c and b!= c:
                            for d in range(len(A)):
                                if a!=d and b!=d and c!= d:
                                    h 
= A[a]*10+A[b]
                                    m = A[c]*10+A[d]
                                    if h < 24 and m < 60:
                                        if h > hh or (h == hh and m > mm):
                                            hh = h
                                            mm = m
        if hh == -1:
            return ""
        else:
            hhh = str(hh)
            mmm = str(mm)
            if len(hhh) < 2:
                hhh = '0'+hhh
            if len(mmm) < 2:
                mmm = '0'+mmm
            return hhh+':'+mmm
