import itertools

class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        r = ""
        rn = -1
        for a in itertools.permutations(A):
            if a[0] > 2:
                continue
            if a[0] == 2 and a[1] >= 4:
                continue
            if a[2] > 5:
                continue
            c = a[0] * 1000 + a[1] * 100 + a[2] * 10 + a[3]
            if c > rn:
                rn = c
                r = str(a[0]) + str(a[1]) + ":" + str(a[2]) + str(a[3])
        return r
