class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        maxs = -1
        ans = ''
        for i in range(len(A)):
            for j in range(len(A)):
                if i == j: continue
                for k in range(len(A)):
                    if k == i or k == j: continue
                    for p in range(len(A)):
                        if p == i or p == k or p == j: continue
                        
if A[i]*10+A[j]<=23 and A[k]*10+A[p]<=59:
                            if 60*(A[i]*10+A[j])+ A[k]*10+A[p]>maxs:
                                maxs = 60*(A[i]*10+A[j])+ A[k]*10+A[p]
                                ans = str(A[i])+str(A[j])+':'+str(A[k])+str(A[p])
        return ans
