class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        import itertools

        ans = None
        best = None
        for a in itertools.permutations(A):
            h = a[0] * 10 + a[1]
            m = a[2] * 10 + a[3]
            if not 0 <= h < 24 or not 0 <= m < 60:
                continue

            if best is None or best < (h, m):
                best = (h, m)
                ans = '{:02d}:{:02d}'.format(h, m)
        return ans if ans is not None else ''
