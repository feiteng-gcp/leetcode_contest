import itertools

class Solution:
    def largestTimeFromDigits(self, A):
        ans = [-1, -1]
        for p in itertools.permutations(A):
            h = int(p[0]) * 10 + int(p[1])
            m = int(p[2]) * 10 + int(p[3])
            if h > 23 or m > 59:
                continue
            if h > ans[0]:
                ans = [h, m]
            elif h == ans[0] and m > ans[1]:
                ans = [h, m]
        if ans[0] < 0:
            return ''
        ret = ''
        if ans[0] < 10:
            ret += '0' + str(ans[0])
        else:
            ret += str(ans[0])
        ret += ':'
        if ans[1] < 10:
            ret += '0' + str(ans[1])
        else:
            ret += str(ans[1])
        return ret
