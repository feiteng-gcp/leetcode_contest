class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        
        from itertools import permutations
        
        # O(10^4)
        
        ans = ""
        max_val = -float("inf")
        for c in permutations(A):
            
            hour = int(str(c[0]) + str(c[1]))
            minute = int(str(c[2]) + str(c[3]))
            if 0 <= hour <= 23 and 0 <= minute <= 59:
                val = hour * 60 + minute
                
                if val > max_val:
                    max_val = val
                    ans = str(c[0]) + str(c[1]) + ':' + str(c[2]) + str(c[3])
        return ans
