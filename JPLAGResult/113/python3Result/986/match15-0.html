import itertools

class Solution:
    def largestTimeFromDigits(self, A):        
        res = itertools.permutations(A)        
        r = []
        for i in res:
            if (i[0] == 2 and i[1] <= 3 and i[2] <= 5) or (i[0] <= 1 and i[2] <= 5):
                r.append(str(i[0]) + str(i[1]) + ':' + str(i[2]) + str(i[3]))
                
                
                
        return max(r) if r else ''
