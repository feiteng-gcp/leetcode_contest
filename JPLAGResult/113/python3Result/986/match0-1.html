class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        import itertools
        res = []
        A = sorted(A)
        for i in itertools.permutations(A, 4):
            res.append(i)

        for i in range(len(res)-1,-1,-1):
            e = res[i]
            if not e[2] <= 5:
                continue
            if e[0] <= 2:
                if e[0] == 2:
                    if e[1] <= 3:
                        return "{}{}:{}{}".format(e[0], e[1], e[2], e[3])
                else:
                    return "{}{}:{}{}".format(e[0], e[1], e[2], e[3])
        return ""
