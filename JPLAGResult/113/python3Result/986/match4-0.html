import itertools
class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        ans = (0,0)
        flag = False
        for t in list(itertools.permutations(A,4)):
            h = int(str(t[0]) + str(t[1]))
            m = int(str(t[2]) + str(t[3]))
            if 0 <= h <= 23 and 0<= m <= 59:
                flag = True
                if (h,m) > ans:
                    ans = (h,m)
        return str(ans[0]).zfill(2)+':'+str(ans[1]).zfill(2) if flag else ""
