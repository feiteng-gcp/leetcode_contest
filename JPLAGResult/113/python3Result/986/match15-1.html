class Solution:
    def largestTimeFromDigits(self, A):
        """
        :type A: List[int]
        :rtype: str
        """
        ans = []
        for i in range(4):
            A[0], A[i] = A[i], A[0]
            for j in range(1, 4):
                A[1], A[j] = A[j], A[1]
                for k in range(2, 4):
                    A[2], A[k] = A
[k], A[2]
                    ans.append(str(A[0]) + str(A[1]) + str(A[2]) + str(A[3]))
                    A[2], A[k] = A[k], A[2]
                A[1], A[j] = A[j], A[1]
            A[0], A[i] = A[i], A[0]
        ans.sort(reverse = True)
        
        for i in ans:
            if int(i[:2]) < 24 and int(i[2:]) < 60:
                return i[:2] + ':' + i[2:]
        return ''
