# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipEquiv(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        # 100 nodes, for each node, we make 4 calls recursively
        # O(4^100)???
        
        if root1 is None and root2 is None:
            return True
        elif root1 is None:
            return False
        elif root2 is None:
            return False
        elif root1
.val != root2.val:
            return False
        
        # if this node's children are flipped, check the flip equivalence recursively
        return (self.flipEquiv(root1.left, root2.left) and self.flipEquiv(root1.right, root2.right)) or\
               (self.flipEquiv(root1.left, root2.right) and self.flipEquiv(root1.right, root2.left))
