class Solution {
public:
    int largestComponentSize(vector<int>& A) {
        unordered_map<int, int> indexes;
        int max_v = 0;
        for (int i = 0; i < A.size(); ++i) {
            indexes[A[i]] = i;
            max_v = max(max_v, A[i]);
        }
        vector<int> roots(A.size(), -1);
        bitset<100001> vis;
        for (int p = 2; 2 * p <= max_v; ++p) {
            if (!vis[p]) {
                int idx = -1;
                for (int i = 1; i * p <= max_v; ++i) {
                    vis[i * p] = true;
                    if (indexes.count(i * p) == 0) {
                        continue;
                    } else if (idx < 0) {
                        idx = indexes[i * p];
                    } else {
                        merge(roots, idx, indexes[i * p]);
                    }
                }
            }
        }
        int result = 0;
        for (int i = 0; i < roots.size(); ++i) {
            result = max(result, -roots[i]);
        }
        return result;
    }
private:
    void merge(vector<int> &roots, int i, int j) const {
        i = get_root(roots, i);
        j = get_root(roots, j);
        if (i == j) {
            return;
        } else if (roots[i] < roots[j]) {
            roots[i] += roots[j];
            roots[j] = i;
        } else {
            roots[j] += roots[i];
            roots[i] = j;
        }
    }
    int get_root(vector<int> &roots, int i) const {
        if (roots[i] < 0) {
            return i;
        } else {
            return roots[i] = get_root(roots, roots[i]);
        }
    }
};
