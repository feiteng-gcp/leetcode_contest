#define rep(i, n) for (int i = 0; i < (n); i++)
#define itrep(i, a) for (auto i = (a).begin(); i != (a).end(); i++)
#define REP(i, a, b) for (int i = a; i <= (b); i++)
#define all(a) (a).begin(), (a).end()

const int MAX_PRIME = 100010;
bool isPrime[MAX_PRIME];

void calcPrimes() {
	for (int i = 2; i < MAX_PRIME; i++) isPrime[i] = true;
	for (int i = 2; i < MAX_PRIME; i++) {
		if (!isPrime[i]) continue;
		for (int j = 2 * i; j < MAX_PRIME; j += i) isPrime[j] = false;
	}
}

class UnionFind {
private:
	vector<int> pastNode;

	class Node {
	public:
		int parent;
		int count;

		Node(int parent) {
			this->parent = parent;
			this->count = 1;
		}
	};

	vector<Node> node;
public:
	UnionFind(int size) {
		for (int i = 0; i < size; i++) {
			node.push_back(Node(i));
		}
	}

	int root(int x) {
		pastNode.clear();

		while (x != node[x].parent) {
			pastNode.push_back(x);
			x = node[x].parent;
		}
		rep(i, pastNode.size()) {
			node[pastNode[i]].parent = x;
		}
		return x;
	}

	void unite(int x, int y) {
		if (root(x) == root(y)) return;

		node[root(y)].count += node[root(x)].count;
		node[root(x)].parent = root(y);
	}

	bool same(int x, int y) {
		return root(x) == root(y);
	}

	int countSames(int x) {
		return node[root(x)].count;
	}
};

class Solution {
	vector<vector<int>> nums;

public:
    int largestComponentSize(vector<int>& A) {
		calcPrimes();
		vector<int> primes;
		REP(i, 2, 100000) if (isPrime[i]) primes.push_back(i);

		unordered_map<int, int> m;
		rep(i, primes.size()) {
			m[primes[i]] = i;
		}

		nums.resize(primes.size());
		int rt = sqrt(100000);

		rep(i, A.size()) {
			rep(j, primes.size()) {
				if (primes[j] > rt) break;
				bool pushed = false;
				while (A[i] % primes[j] == 0) {
					if (!pushed) nums[j].push_back(i);
					A[i] /= primes[j];
					pushed = true;
				}
			}
			if (A[i] > 1) nums[m[A[i]]].push_back(i);
		}

		UnionFind uf(A.size());
		rep(i, nums.size()) {
			if (nums[i].size() < 2) continue;
			REP(j, 1, nums[i].size() - 1) {
				uf.unite(nums[i][0], nums[i][j]);
			}
		}

		int ret = 0;
		rep(i, A.size()) ret = max(ret, uf.countSames(i));

		return ret;
    }
};
