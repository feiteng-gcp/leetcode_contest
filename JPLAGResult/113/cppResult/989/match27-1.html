class Solution {
public:
    int N = 100000;
    int s[100000 + 100000 + 5];
    int f[100000 + 100000 + 5];
    
    int find(int i) {
        return f[i] == i ? i : f[i] = find(f[i]);
    }
    
    void merge(int a, int b) {
        //printf("%d %d\n", a, b);
        a = find(a), b = find(b);
        if (a != b) {
            f[a] = b;
            s[b] += s[a];
        }
    }
    
    int largestComponentSize(vector<int>& A) {
        for (int i = 0; i <= 100000; i++) {
            s[i] = 0;
            s[i + N + 1] = 1;
            f[i] = i;
            f[i + N + 1] = i + N + 1;
        }
        
        int now = 0;

        for (int vv : A) {
            int v = vv;
            for (int i = 2; i * i <= v; i++) {
                if (v % i == 0) {
                    merge(now + N + 1, i);
                    while (v % i == 0) {
                        v /= i;
                    }
                }
            }
            
            if (v > 1) {
                merge(now + N + 1, v);
            }
            
            now++;
        }
        
        int ans = 0;
        for (int i = 0; i <= N + N; i++) {
            if (f[i] == i) {
                ans = max(ans, s[i]);
            }
        }
        
        return ans;
    }
};
