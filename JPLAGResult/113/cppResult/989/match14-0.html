class Solution {
    int SIZE = 100000;
    vector<bool> isPrime;
    vector<int> primes;
    vector<vector<int> > Hash;
    vector<int> ufds,ufdsRank;
    vector<int> rootHash;
public:
    
    int MAX(int x,int y){
        return x>y?x:y;
    }
    Solution(){
        isPrime.assign(SIZE+1,true);
        isPrime[0] = isPrime[1] = false;
        for(int i=2;i<=SIZE;i++)
            if(isPrime[i]){
                primes.push_back(i);
                for(int j=2*i;j<=SIZE;j+=i)
                    isPrime[j] = false;
            }
    }
    
    void initialise(int N){
        ufds.resize(N);
        ufdsRank.assign(N,1);
        for(int i=0;i<N;i++)
            ufds[i] = i;
    }
    
    int rootOf(int x){
        while(ufds[x] != x){
            ufds[x] = ufds[ufds[x]];
            x = ufds[x];
        }
        return x;
    }
    
    void connect(int x,int y){
        int rootx = rootOf(x);
        int rooty = rootOf(y);
        if(ufdsRank[rootx] > ufdsRank[rooty]){
            ufds[rooty] = ufds[rootx];
            ufdsRank[rootx] += ufdsRank[rooty];
        }
        else{
            ufds[rootx] = ufds[rooty];
            ufdsRank[rooty] += ufdsRank[rootx];
        }
    }
    
    bool connected(int x,int y){
        return rootOf(x) == rootOf(y);
    }
    
    
    void make_edge(int pf,int idx){
        int u,v;
        v = idx;
        for(int i=0;i<Hash[pf].size();i++){
            u = Hash[pf][i];
            if(!connected(u,v))
                connect(u,v);
            
        }
    }
    
    void process(int x,int idx){
        int id = 0;
        long int pf = primes[id];
        while(pf * pf <= x){
            if(x%pf == 0){
                make_edge(pf,idx);
                Hash[pf].push_back(idx);
            }
            while(x%pf == 0)
                x /= pf;
            pf = primes[++id];
        }
        if(x != 1){
            make_edge(x,idx);
            Hash[x].push_back(idx);
        }
    }
    
    
    int largestComponentSize(vector<int>& A) {
        int N;
        N = A.size();
        initialise(N);
        Hash.resize(SIZE+1);
        for(int i=0;i<N;i++)
            process(A[i],i);
        
        
        int ans = 0;
        rootHash.assign(N,0);
        int x;
        for(int i=0;i<N;i++)
            rootHash[rootOf(i)]++;
        
        for(int i=0;i<N;i++)
            ans = MAX(ans,rootHash[i]);
        
        return ans;
    }
};
