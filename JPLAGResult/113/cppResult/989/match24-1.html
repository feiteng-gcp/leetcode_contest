class Solution {
public:
    vector<int> facs;

    unordered_map<int,int> fa_fac;
    int find_fa(int x){
        return fa_fac[x]==x?x:fa_fac[x]=find_fa(fa_fac[x]);
    }
    void union_set(int x,int y){
        fa_fac[find_fa(y)]=find_fa(x);
    }
    vector<int> get_fac(int x){
        vector<int> res(1,x);
        for(int i=2;i*i<=x;i++){
            if(x%i==0)
            { res.push_back(i);res.push_back(x/i);}
        }
        return res;
    }
    int largestComponentSize(vector<int>& A) {
        
        int n=A.size();
        vector<int> fa_A(n);
        int res=0;
        for(int i=0;i<n;i++){
            if(A[i]==1){
                res=1;
                continue;
            }
            vector<int> cur=get_fac(A[i]);
            for(int i=0;i<cur.size();i++){
                if(fa_fac.count(cur[i])==0)
                    fa_fac[cur[i]]=cur[i];
                union_set(cur[0],cur[i]);
            }
            fa_A[i]=cur[0];
        }
        unordered_map<int,int> cnt;
        for(int i=0;i<n;i++){
            if(A[i]==1){
                continue;
            }
            cnt[find_fa(fa_A[i])]++;
        }
        for(auto&i:cnt){
            res=max(res,i.second);
        }
        return res;
    }
};
