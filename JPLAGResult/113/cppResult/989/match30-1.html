class Solution {
    vector<int> ss;
    int find(int i){
        if(ss[i]!=i) ss[i]=find(ss[i]);
        return ss[i];
    }
    void merge(int i,int j){
        ss[find(i)]=find(j);
    }
public:
    int largestComponentSize(vector<int>& A) {
        ss=vector<int>(A.size());

        for(int i=0;i<A.size();i++) ss[i]=i;
        vector<int> last(100001,-1);
        for(int j=0;j<A.size();j++){
            int n=A[j];
            for(int i=2;i*i<=n;i++){
                if(n%i!=0) continue;
                if(last[i]!=-1) merge(j,last[i]);
                if(last[n/i]!=-1) merge(j,last[n/i]);
                last[i]=j;
                last[n/i]=j;
            }
            if(n>1){
            if(last[n]!=-1) merge(j,last[n]);
            last[n]=j;
            }
        }
        vector<int> cnt(A.size(),0);
        for(int i=0;i<A.size();i++) cnt[find(i)]++;
        return *max_element(cnt.begin(),cnt.end());
    }
};
