typedef vector<int> vi;
#define ll long long
class Solution {
public:
    vi id;
vi sz;
int groups;

    ll gcd(ll a, ll b) {
	unsigned c = 0;
	while (1)
	{
		if (a == 0)
			return b << c;
		else if (b == 0)
			return a << c;
		if (!(a & 1) && !(b & 1))
		{
			a >>= 1; b >>= 1; ++c;
		}
		else if (!(a & 1) && (b & 1))
		{
			a >>= 1;
		}
		else if ((a & 1) && !(b & 1))
		{
			b >>= 1;
		}
		else if ((a & 1) && (b & 1))
		{
			unsigned tmp = a > b ? b : a;
			a = a > b ? a - b : (b - a);
			b = tmp;
		}
	}
}
    
void init(int n)
{
	groups = n;
	id = vector<int>(n, 0);
	for (int i = 0; i < n; i++)
	{
		id[i] = i;
	}
	sz = vi(n, 1);
}

int findRoot(int p)
{
	while (p != id[p])
	{
		id[p] = id[id[p]];
		p = id[p];
	}
	return p;
}

void comb(int p, int q) {
	int i = findRoot(p);
	int j = findRoot(q);
	if (i == j)
	{
		return;
	}
	if (sz[i] < sz[j])
	{
		id[i] = j;
		sz[j] += sz[i];
	}
	else
	{
		id[j] = i;
		sz[i] += sz[j];
	}
	groups--;
}

    
    
    int largestComponentSize(vector<int>& A) {
        int l  =A.size();
        vector<int> AA= A;
        init(100007);
        vector<int> ss(100007,0);
        for(int i=0;i<l;i++){
            vector<int> r;
            r.push_back(A[i]);
            for(int j=2;j*j<=A[i];j++){
                if((A[i]%j)==0){
                    
                    while((A[i]%j)==0){
                        A[i]/=j;
                    }
                    
                    r.push_back(j); 
                    if(A[i]>1){
                        r.push_back(A[i]);
                    }
                }
                
            }
            
       
            for(int u=1;u<r.size();u++){
                comb(r[0],r[u]);
            }
            
            
        }
         int ma= -1;
        for(int i=0;i<l;i++){
           ss[findRoot(AA[i])]++;
            ma  =max(ma,ss[findRoot(AA[i])]);
            
        }
        
       
       

        
        return ma;
    }
};
