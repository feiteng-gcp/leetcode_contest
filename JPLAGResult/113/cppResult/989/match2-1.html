const int N = 100000 + 5;
int f[N];  
int c[N];
int find(int x) {
        if (f[x] != f[f[x]]) {
                f[x] = find(f[x]);
        }
        return f[x];
}


class Solution {        
public:
    int largestComponentSize(vector<int>& a) {
         int n = a.size();
         int m = 100000;
         vector<vector<int>> A(m);   

         for (auto x : a) {
               int y = x;
               for (int i = 2; i*i <= y; i++) {
                       if (y % i == 0) {
                               while(y % i == 0) {
                                       y = y / i;
                               }
                               A[i].push_back(x);
                       }
               }  
               if (y > 1) {
                       A[y].push_back(x);
               }  
         }   
         for (int i = 0; i < N; i++) {
                 f[i] = i;
                 c[i] = 1;
         }
         for (int i = 2; i < m; i++) {
                 if (A[i].size() < 2) {
                         continue;
                 }
                 for (int j = 1; j < A[i].size(); j++) {
                         int x = A[i][j-1];
                         int y = A[i][j];
                         int rx = find(x);
                         int ry = find(y);
                         if (rx != ry) {
                                 f[ry] = rx;
                                 c[rx] += c[ry];
                         } 
                 }

         }   
         int ans = 1;
         for (int i = 1; i < N; i++) {
                 if (find(i) == i) {
                         ans = max(ans, c[i]);
                 }
         }   
         return ans;    
    }
};
