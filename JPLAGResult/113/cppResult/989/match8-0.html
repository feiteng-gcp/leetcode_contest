class Solution {
    array<int,20000> ufs;
    int find(int x){
        if(ufs[x]<0)return x;
        int ret=find(ufs[x]);
        ufs[x]=ret;
        return ret;
    }
    int size(int x){
        return -ufs[find(x)];
    }
    bool merge(int x,int y){
        x=find(x);
        y=find(y);
        if(x==y)return false;
        if(ufs[x]>ufs[y])swap(x,y);
        int tmp=ufs[x];
        ufs[x]=y;
        ufs[y]+=tmp;
        return true;
    }
public:
    int largestComponentSize(vector<int>& A) {
        unordered_map<int,vector<int>> factor;
        int n=A.size();
        fill_n(ufs.begin(),n,-1);
        for(int i=0;i<A.size();i++){
            int a=A[i];
            for(int j=2;j*j<=a;j++){
                if(a%j==0)factor[j].push_back(i);
                while(a%j==0)a/=j;
            }
            if(a>1)factor[a].push_back(i);
        }
        for(auto &p:factor){
            auto &v=p.second;
            for(int i=1;i<v.size();i++)
                merge(v[0],v[i]);
        }
        int ans=0;
        for(int i=0;i<n;i++)ans=max(ans,size(i));
        return ans;
    }
};
