class Solution {
public:
    int N = 100001;
    map<int, int> count;
        map<int, int> fa;
    
    void uniq(int x, int y) {
        if(faa(x) != faa(y)) {
            fa[faa(x)] = faa(y);
        }
    }
    int faa(int x) {
        return x == fa[x] ? x : (fa[x] = faa(fa[x]));
    }
    
    int largestComponentSize(vector<int>& A) {
        int vi[N];
        memset(vi, 0, sizeof(vi));
        vector<int> v;
        for(int i = 2; i < 340; ++i) {
            if(vi[i] == 0) {
                for(int j = i*i; j < N; j+=i) {
                    vi[j] = 1;
                }
            }
        }
        for(int i = 2; i < N; ++i) {
            if(vi[i] == 0) v.push_back(i);
        }
        count.clear();
        fa.clear();
        for(int i = 0; i < v.size(); ++i) {
            fa[ v[i] ] = v[i];
            count[ v[i] ] = 0;
        }
        for(int i = 0; i < A.size(); ++i) {
            int first = -1;
            while(A[i] != 1) {
                if(vi[ A[i] ] == 0) {
                    if(first != -1) {
                        uniq(first, A[i]);
                    } else {
                        ++count[ A[i] ];
                    }
                    break;
                }
                for(int j = 0; j < v.size(); ++j) {
                    if(v[j] > A[i]/2) {
                        break;
                    }
                    if(A[i] % v[j] == 0) {
                        if(first == -1) {
                            first = v[j];
                        } else {
                            uniq(first, v[j]);
                        }
                        while(A[i] % v[j] == 0) {
                            A[i] /= v[j];
                        }
                    }
                }
                if(first != -1) {
                    ++count[ first ];
                }
            }
        }
        
        
        int re = 0;
        for(int i = 0; i < v.size(); ++i) {
            if(faa(v[i]) != v[i]) {
                count[ faa(v[i]) ] += count[v[i]];
                count[v[i]] = 0;
            }
            if(count[faa(v[i])] > re) {
                re = count[ faa(v[i]) ];
            }
        }
        return re;
    }
};
