class Solution {
public:
    const int N = 100002;
    int lp[100002],parent[100002];
    vector<int> g[100002];
    
    int find(int a){return a == parent[a] ? a : parent[a] = find(parent[a]);}
    void merge(int a, int b){parent[find(a)] = find(b);}
    
    int largestComponentSize(vector<int>& A) {
        for(int i = 2; i < N; ++i) if(!lp[i]){
            lp[i] = i;            
            for(long long j = 1ll*i*i; j < N; j += i)
                lp[j] = i;
        }
        int n = A.size();
        for(int i = 0; i < n; ++i){
            int x = A[i];
            while(lp[x] > 1) g[lp[x]].push_back(i), x /= lp[x];
            parent[i] = i;
        }
        for(int i = 2; i < N; ++i) 
            for(int j = 1; j < g[i].size(); ++j)
                merge(g[i][j], g[i][j-1]);
        map<int,int> f;
        for(int i = 0; i < n; ++i) ++f[find(i)];
        int ans = 0;
        for(auto p : f) ans = max(ans, p.second);
        return ans;
    }
};
