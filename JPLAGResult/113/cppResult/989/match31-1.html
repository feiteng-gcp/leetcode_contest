class Solution {
public:
    int largestComponentSize(vector<int>& A) {
        unordered_map<int, vector<int>> v_f, f_i;

        int n = A.size();
        vector<int> group(n, -1);
        
        for(int i=0;i<n;i++){
            int v = A[i];
            for(int f=2;f * f <= v; f++){
                if(v%f == 0){
                    v_f[A[i]].push_back(f);
                    f_i[f].push_back(i);
                    while(v%f == 0){
                        v /= f;
                    }
                }
            }
            
            if(v != 1){
                v_f[A[i]].push_back(v);
                f_i[v].push_back(i);
            }
        }
        
        int last_group = 0;
        
        function<void(int)> dfs = [&](int cur) {
          //cout<<"dfs("<<cur<<")\n";
            
          for(int f: v_f[A[cur]]){
               //cout<<"f:"<<f<<endl;
               for(int ni: f_i[f]){
                   //cout<<"ni:"<<ni<<endl;
                   if(group[ni] == -1){
                       group[ni] = last_group;
                       dfs(ni);
                   }
               }
            }
            
        };
        
        for(int i=0;i<n;i++){
            if(group[i] == -1) {
                group[i] = ++last_group;
                dfs(i);
            }
        }
        
        unordered_map<int, int> cnt;
        int ans = 0;
        
        for(int i=0;i<n;i++){
            //cout<<i<<" "<<group[i]<<endl;
            if( ++cnt[group[i]] > ans){
                ans = cnt[group[i]];
            }
            
        }
        
        return ans;
    }
    
private:
    int gcd(int a, int b){
        if(b == 1){
            return 1;
        }
        
        return gcd(b, a%b);
    }
};
