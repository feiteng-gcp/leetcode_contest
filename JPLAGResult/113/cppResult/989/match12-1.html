class Solution {
public:
    int pre[20000],num[20000];
    int Find( int x )
    {
        if ( x!=pre[x] )
            pre[x] = Find( pre[x] );

        return pre[x];
    }
    void Merge( int x , int y )
    {
        int fx = Find(x);
        int fy = Find(y);
        if ( fx!=fy )
        {
            pre[fy] = fx;
            num[fx] += num[fy];
        }
    }
    int largestComponentSize(vector<int>& A) {
        vector<int> a[100001];
        for( int i=0 ; i<A.size() ; i++ )
            a[A[i]].push_back(i);
        for( int i=0 ; i<A.size() ; i++ )
        {
            pre[i] = i;
            num[i] = 1;
        }

        for( int i=2 ; i<=100000 ; i++ )
        {
            int last = -1;
            for( int j=i ; j<=100000 ; j+=i )
            {
                for( int k=0 ; k<a[j].size() ; k++ )
                {
                    if ( last!=-1 ) Merge( last , a[j][k] );
                    last = a[j][k];
                }
            }
        }
        int ans = 1;
        for( int i=0 ; i<A.size() ; i++ )
            if ( i==Find(i) ) ans = max( ans , num[i] );
        return ans;
    }
};
