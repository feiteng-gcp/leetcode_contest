class Solution {
public:
    vector<int>G[100004];
    bool vis[100004];
    int cnt=0,t[100004];
    void dfs(int x){
         vis[x]=1;
         if(t[x])++cnt;
         for(int i=0;i<G[x].size();i++)
             if(!vis[G[x][i]])dfs(G[x][i]);
    }
    int largestComponentSize(vector<int>& A) {
        int ans=0;
        memset(vis,0,sizeof(vis));
        memset(t,0,sizeof(t));
        for(int i=0;i<=100000;i++)G[i].clear();
        for(int i=0;i<A.size();i++)t[A[i]]=1;
        for(int i=2;i<=100000;i++)
            for(int j=i;j<=100000;j+=i)
                if(t[j]){
                    G[i].push_back(j);
                    G[j].push_back(i);
                }
        for(int i=0;i<A.size();i++){
            if(!vis[A[i]]){
                cnt=0;
                dfs(A[i]);
                ans=max(ans,cnt);
            }
        }
        return ans;
    }
};
