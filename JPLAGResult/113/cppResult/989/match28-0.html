class Solution {
public:
    int f[100005];
    int find(int x){
        if(f[x]==x)return x;
        return f[x]=find(f[x]);
    }
    void uni(int x,int y){
        int fx=find(x);
        int fy=find(y);
        f[fy]=fx;
    }
    int largestComponentSize(vector<int>& a) {
        int b[100005]={0};
        for(int i=0;i<a.size();i++)
            b[a[i]]++;
        for(int i=1;i<=100000;i++)
            f[i]=i;
        vector<int>v;
        for(int i=2;i<=100000;i++){
            for(int j=1;j*i<=100000;j++){
                if(b[j*i]){
                    v.push_back(i);
                    break;
                }
            }
        }
        for(int k=0;k<v.size();k++){
            int i=v[k];
            int s=100005;
            for(int j=1;j*i<=100000;j++){
                if(b[j*i]&&j*i<s)
                    s=j*i;
                if(b[j*i]&&s<100005)
                    uni(s,j*i);
            }
        }
        map<int,int>m;
        for(int i=1;i<=100000;i++){
            if(b[i])
                m[find(i)]+=b[i];
        }
        int ret=0;
        for(auto it=m.begin();it!=m.end();it++){
            ret=max(ret,it->second);
        }
        return ret;
    }
};
