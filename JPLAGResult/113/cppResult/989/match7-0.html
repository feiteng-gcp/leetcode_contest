class Solution {
public:
    map<int, vector<int> > fm;
    int p[20010];

    int find(int x) {
      if (x == p[x]) return x;
      p[x] = find(p[x]);
      return p[x];
    }

    int largestComponentSize(vector<int>& A) {
      int n = A.size();
      for (int i = 0; i < n; ++i) {
        int x = A[i];
        for (int k = 2; x > 1 && k <= 1000; ++k) {
          if (x % k == 0) {
            fm[k].push_back(i);
            while (x > 1 && x % k == 0) x /= k;
          }
        }
        if (x > 1) fm[x].push_back(i);
      }
        //printf("????\n");
      for (int i = 0; i < n; ++i) p[i] = i;
      for (auto& e : fm) {
        int f = e.first;
        auto& v = e.second;
          //printf("%d: ", f);
          //for (auto& id:v) printf("%d ", A[id]); printf("\n");
        for (auto& id : v) {
          int rt = find(id);
          p[id] = v.front();
          p[rt] = p[id];
            //printf("==== %d: %d\n", id, rt);
        }
      }

      std::map<int, int> ans;
      for (int i = 0; i < n; ++i) {
        int rt = find(i);
        ans[rt] += 1;
      }
      int maxv = 0;
      for (auto& e : ans) maxv = max(maxv, e.second);
      return maxv;
    }
};
