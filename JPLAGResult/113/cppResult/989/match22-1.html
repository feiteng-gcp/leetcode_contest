const int N = 20010;
int pre[N];
int c[N];

int find(int x) {
	if (pre[x] == -1) return x;
	return pre[x] = find(pre[x]);
}

class Solution {
public:
    int largestComponentSize(vector<int>& A) {
		vector<int> fac[100010];
		memset(pre, -1, sizeof(pre));
		fill(c, c + N, 1);

		for (int i = 0; i < A.size(); i++) {
			int x = A[i];
			for (int j = 1; j * j <= x; j++) {
				if (x % j) continue;
				if (j > 1) fac[j].push_back(i);
				if (j * j != x) fac[x / j].push_back(i);
			}
		}

		int ans = 1;
		for (int i = 2; i < 100010; i++) {
			for (int j = 1; j < fac[i].size(); j++) {
				int x = find(fac[i][0]), y = find(fac[i][j]);
				if (x == y) continue;
				pre[x] = y;
				c[y] += c[x];
				ans = max(ans, c[y]);
			}
		}
		return ans;
    }
};
