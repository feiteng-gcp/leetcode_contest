typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))


const int INF = (int)1E9;
#define MAXN 100005

int rep[MAXN], cnt[MAXN];
int find(int x) {
  return rep[x] == x ? x : rep[x] = find(rep[x]);
}
void unionset(int x, int y) {
  int rx = find(x), ry = find(y);
  if (rx == ry) return;
  rep[rx] = ry;
  cnt[ry] += cnt[rx];
}

class Solution {
public:
  int largestComponentSize(vector<int>& A) {
    map<int, VI> fs;
    int n = A.size();
    REP(i,0,n) {
      rep[i] = i;
      cnt[i] = 1;
    }
    REP(i,0,n) {
      int x = A[i];
      for (int d = 2; d*d <= x; d++) {
        if (x % d == 0) {
          while (x % d == 0) x /= d;
          fs[d].push_back(i);
        }
      }
      if (x > 1) fs[x].push_back(i);
    }
    for (auto it = fs.begin(); it != fs.end(); it++) {
      VI &v = it->second;
      int k = v.size();
      REP(i,0,k-1) {
        unionset(v[i], v[i+1]);
      }
    }
    int ans = 0;
    REP(i,0,n) ans = max(ans, cnt[find(i)]);
    return ans;
  }
};
