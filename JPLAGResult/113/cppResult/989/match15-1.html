typedef long long llong;
typedef vector<int> VI;
typedef vector<VI> VVI;


class DisjointSet {
    int N;
    vector<int> par;
    vector<int> rank;

public:
    vector<int> comp_size;
    DisjointSet(size_t _N)
        : N(_N), par(_N, -1), rank(_N, 0), comp_size(_N, 1) {}
    int find_rep(int u) {
        return par[u] < 0 ? u : par[u] = find_rep(par[u]);
    }
    bool union_rep(int u, int v) {
    //  cerr << "UNION " << u << ' ' << v << endl;
        int u_root = find_rep(u);
        int v_root = find_rep(v);
        if (u_root == v_root)
            return false;
        if (rank[u_root] > rank[v_root]) {
            par[v_root] = u_root;
            comp_size[u_root] += comp_size[v_root];
        }
        else {
            par[u_root] = v_root;
            if (rank[u_root] == rank[v_root])
                rank[v_root] = rank[u_root] + 1;
            comp_size[v_root] += comp_size[u_root];
        }
        return true;
    }
};

class Solution {
public:
    int largestComponentSize(vector<int>& A) {
        if (A.size() == 1)
            return 1;

        vector<bool> vertex_exists(100001);
        for (int x : A)
            vertex_exists[x] = true;

        DisjointSet dset(100001);


        for (int d = 2; d <= 100000; ++d) {
            int first_vertex = -1;
            for (int m = d; m <= 100000; m += d) {
                if (vertex_exists[m]) {
                    if (first_vertex < 0) {
                        first_vertex = m;
                    }

                    else {
                        dset.union_rep(first_vertex, m);
                    }
                }
            }
        }

        int res = 0;
        for (int u = 1; u <= 100000; ++u) {
            res = max(res, dset.comp_size[u]);
        }
        return res;
    }
};
