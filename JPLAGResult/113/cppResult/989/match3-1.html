class Solution {
public:
    int dfs(int node, vector<int>& A, vector<vector<int>>& fac, vector<vector<int>>& v, vector<int>& visit){
        //cout << node << endl;
        int ret = 1;
        visit[node] = 1;
        
        for(int i = 0;i < fac[node].size();i++){
            int n = fac[node][i];
            for(int j = 0;j < v[n].size();j++){
                if(!visit[v[n][j]]){
                    ret += dfs(v[n][j], A, fac, v, visit);
                }
            }
        }
        return ret;
    }
    
    int largestComponentSize(vector<int>& A){
        vector<vector<int>> fac(A.size());
        vector<vector<int>> v(100001);
        vector<int> visit(A.size(), 0);
        

        for(int i = 0;i < A.size();i++){
            if(A[i] == 1){
                continue;
            }
            for(int j = 2;j <= sqrt(A[i])+1 && j <= A[i];j++){
                if(A[i]%j == 0){
                    fac[i].push_back(j);
                    v[j].push_back(i);
                    fac[i].push_back(A[i]/j);
                    v[A[i]/j].push_back(i);

                }
            }
            fac[i].push_back(A[i]);
            v[A[i]].push_back(i);
        }
        int ret = 1;
        for(int i = 0;i < A.size();i++){
            if(!visit[i]){
                //cout << i << endl;
                ret = max(ret, dfs(i, A, fac, v, visit));
            }
            if(ret >= A.size()/2)
                break;
        }
        /*
        for(int i = 0;i < A.size();i++){
            for(int j = 0;j < fac[i].size();j++){
                cout << fac[i][j] << " ";
            }
            cout << endl;
        }
        */
        return ret;
    }
};
