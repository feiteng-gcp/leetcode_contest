int rt[20200], sz[20200], ht[20200];

int rof(int x) {
    return x == rt[x] ? x : (rt[x] = rof(rt[x]));
}

class Solution {
public:
    int largestComponentSize(vector<int>& a) {
        unordered_map<int, vector<int>> f;
        int n = a.size();

        for (int i = 0; i < n; ++i) {
            int x = a[i];
            for (int d = 2; d * d <= x; ++d) {
                if (0 == x % d) {
                    f[d].push_back(i);
                    while (0 == x % d) x /= d;
                }
            }
            if (x > 1) f[x].push_back(i);
        }
        for (int i = 0; i < n; ++i) rt[i] = i, sz[i] = 1;
        for (const auto& p : f) {
            for (int i : p.second) {
                int a = rof(p.second[0]), b = rof(i);
                if (a != b) {
                    if (ht[a] < ht[b]) swap(a, b);
                    rt[b] = a;
                    sz[a] += sz[b];
                    ht[a] = max(ht[a], ht[b] + 1);
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) if (i == rof(i)) ret = max(ret, sz[i]);
        return ret;
    }
};
