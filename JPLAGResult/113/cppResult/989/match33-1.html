class Solution {
public:
    int f[120005],s[120005];
    int get(int x)
    {
        if(f[x]==x)return x;
        return f[x]=get(f[x]);

    }
    void merge(int x,int y)
    {
        if(get(x)==get(y))return;
        s[get(x)]+=s[get(y)];
        f[get(y)]=get(x);
    }
    int largestComponentSize(vector<int>& A) {
        int n=A.size(),m=100000,i,j,ans=0;
        for(i=1;i<=n+m;i++)f[i]=i;
        for(i=1;i<=n;i++)if(A[i-1]!=1)
        {
            s[i+m]=1;
            for(j=1;j*j<=A[i-1];j++)if(A[i-1]%j==0)
            {
                if(j!=1)merge(i+m,j);
                merge(i+m,A[i-1]/j);
            }
            ans=max(ans,s[get(i+m)]);
        }
        return ans;
    }
};
