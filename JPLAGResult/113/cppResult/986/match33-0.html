class Solution {
public:
    string largestTimeFromDigits(vector<int>& a) {
        sort(a.begin(), a.end());
        string hh, mm;
        do {
            string h, m;
            h.push_back((char)('0' + a[0]));
            h.push_back((char)('0' + a[1]));

            m.push_back((char)('0' + a[2]));
            m.push_back((char)('0' + a[3]));

            if (h < "24" && m < "60") {
                if (hh.empty()) {
                    hh = h;
                    mm = m;
                } else if ((hh < h) || (hh == h && mm < m)) {
                    hh = h;
                    mm = m;
                }
            }
        } while (next_permutation(a.begin(), a.end()));

        if (!hh.empty()) {
            return hh + ":" + mm;
        } else {
            return "";
        }
    }
};
