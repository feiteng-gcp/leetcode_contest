class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        string res="";

        int r=-1;
        for(int i=0;i<4;i++){
            for(int j=0;j<4;j++){
                if(j==i) continue;
                for(int k=0;k<4;k++){
                    if(k==j || k==i) continue;
                    for(int l=0;l<4;l++){
                        if(l==i|| l==j || l==k) continue;
                        if(A[i]>=3) continue;
                        if(A[i]==2 && A[j]>3) continue;
                        if(A[k]>=6) continue;
                        int temp = A[k]*10 + A[l] + (A[i]*10+A[j])*60;
                        if(temp>r) {
                            r=temp;
                            res=to_string(A[i])+to_string(A[j])+":"+to_string(A[k])+to_string(A[l]);
                        }
                    }
                }
            }
        }
        if(r==-1) return "";
        return res;
    }
};
