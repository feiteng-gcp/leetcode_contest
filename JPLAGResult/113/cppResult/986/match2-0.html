class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        int max_ans = -1;
        for(int i = 0; i < 4; i++)
        {
            for(int j = 0; j < 4; j++)
            {
                if(j != i && (A[i] * 10 + A[j] < 24))
                {
                    for(int k = 0; k < 4; k++)
                    {
                        if(k != i && k != j)
                        {
                            for(int l = 0; l < 4; l++)
                            {
                                if(l != i && l != j && l != k && A[k] * 10 + A[l] < 59)
                                {
                                    int ans = A[i] * 1000 + A[j] * 100 + A[k] * 10 + A[l];
                                    if(ans > max_ans)
                                        max_ans = ans;
                                }
                            }
                        }
                    }
                }
            }
        }
        if(max_ans == -1)
            return "";
        else
        {
            string str = "";
            int a = max_ans / 1000;
            int b = (max_ans % 1000) / 100;
            int c = (max_ans % 100) / 10;
            int d = max_ans % 10;
            return std::to_string(a) + std::to_string(b) + ":" + std::to_string(c) + std::to_string(d);
        }
    }
};
