class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        int n=A.size(),mx=-1;
        for(int i=0;i<4;i++){
            if(A[i]>2) continue;
            for(int j=0;j<4;j++){
                if((j==i) || (A[i]==2 && A[j]>3)) continue;                
                for(int v=0;v<4;v++){
                    if(v==i || v==j) continue;
                    if(A[v]>5) continue;
                    for(int u=0;u<4;u++){
                        if(u==i || u==j || u==v) continue;
                        mx=max(mx,1000*A[i]+100*A[j]+10*A[v]+A[u]);
                    }
                }
            }
        }
        if(mx==-1) return "";
        string res=to_string(mx);
        while(res.size()<4) res.insert(res.begin(),'0');
        res.insert(res.begin()+2,':');
        return res;
    }
};
