class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {

        int hour = -1, minute = 0;
        sort(A.begin(), A.end());
        do{
            int h = A[0] * 10 + A[1];
            int m = A[2] * 10 + A[3];
            if(h <= 23 && m <= 59){
                if(h > hour){
                    hour = h;
                    minute = m;
                }else if(h == hour && m > minute){
                    hour = h;
                    minute = m;
                }
            }
        }while(next_permutation(A.begin(), A.end()));
        if(hour == -1){
            return "";
        }else{
            string res;
            if(hour < 10){
                res += "0" + to_string(hour);
            }else{
                res += to_string(hour);
            }
            res += ':';
            if(minute < 10){
                res += "0" + to_string(minute);
            }else{
                res += to_string(minute);
            }
            return res;
        }
    }
};
