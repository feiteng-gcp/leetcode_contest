class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        sort(A.begin(), A.end());
        string res = "";
        do {
            int a = A[0] * 10 + A[1];
            int b = A[2] * 10 + A[3];
            if(a >= 0 && a <= 23 && b >= 0 && b <= 59) {
                res = "";
                res += A[0] + '0';
                res += A[1] + '0';
                res += ':';
                res += A[2] + '0';
                res += A[3] + '0';
            }
        } while(next_permutation(A.begin(), A.end()));
        return res;
    }
};
