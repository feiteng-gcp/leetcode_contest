class Solution {
public:
    bool ck(int a, int b, int c, int d) {
        if (a > 2) return false;
        if (a == 2 && b >= 4) return false;
        if (c >= 6) return false;
        return true;
    }
    string largestTimeFromDigits(vector<int>& A) {
        string ret = "", tmp = "";
        for (int i = 0; i < 4; ++i) {
            for (int j = 0; j < 4; ++j) {
                if (i == j) continue;
                for (int k = 0; k < 4; ++k) {
                    if (k == i || k == j) continue;
                    for (int l = 0; l < 4; ++l) {
                        if (l == i || l == j || l == k)
                            continue;
                        if (ck(A[i], A[j], A[k], A[l])) {
                            tmp = "";
                            tmp += '0' + A[i];
                            tmp += '0' + A[j];
                            tmp += ':';
                            tmp += '0' + A[k];
                            tmp += '0' + A[l];
                            ret = max(ret, tmp);
                        }
                    }
                }
            }
        }
        return ret;
    }
};
