class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        int hh = -1;

        int mm = -1;
        for (int a = 0; a < 4; a++) {
            for (int b = 0; b < 4; b++) {
                if (a == b) {
                    continue;
                }
                for (int c = 0; c < 4; c++) {
                    if (a == c || b == c) {
                        continue;
                    }
                    for (int d = 0; d < 4; d++) {
                        if (a == d || b == d || c == d) {
                            continue;
                        }
                        int h = A[a] * 10 + A[b];
                        if (h > 23) {
                            continue;
                        }
                        int m = A[c] * 10 + A[d];
                        if (m > 59) {
                            continue;
                        }
                        if (h * 60 + m > hh * 60 + mm) {
                            hh = h;
                            mm = m;
                        }
                    }
                }
            }
        }
        if (hh == -1) {
            return "";
        }
        return to_string(hh / 10) + to_string(hh % 10) + ":" + to_string(mm / 10) + to_string(mm % 10);
    }
};
