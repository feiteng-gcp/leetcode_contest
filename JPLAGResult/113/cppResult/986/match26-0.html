class Solution {
    bool check(int a, int b){
        if(a<=23 && b<=59)
            return true;
        return false;
    }
public:
    string largestTimeFromDigits(vector<int>& A) {
        string res="";
        int vv = -1;
        for(int i = 0; i < 4; i++){
            for(int j = 0; j<4; j++){
                if(j == i)continue;
                for(int k = 0; k < 4; k++){
                    if(k==j||k==i)continue;
                    int l = 6-i-j-k;
                    int t1 = A[i]*10+A[j];
                    int t2 = A[k]*10+A[l];
                    if(check(t1,t2) && (t1*100+t2)>vv){
                        res.clear();
                        vv = t1*100+t2;
                        if(t1<10)
                            res += "0";
                        res += to_string(t1)+":";
                        if(t2<10)
                            res += "0";
                        res += to_string(t2);
                    }
                }
            }
        }
        return res;
    }
};
