class Solution {
public:
    bool valid(int h, int m) {
        return h <= 23 && m <= 59;
    }
    bool greater(int h, int m, int a, int b) {
        if(h != a) return h > a;
        return m > b;
    }
    string largestTimeFromDigits(vector<int>& A) {
        int x = -1, y = -1;
        sort(A.begin(), A.end());
        do {
            int a = A[0] * 10 + A[1];
            int b = A[2] * 10 + A[3];
            if(valid(a, b) && greater(a, b, x, y)) {
                x = a, y = b;
            }
        } while(next_permutation(A.begin(), A.end()));
        if(x == -1) return "";
        return (x < 10 ? "0" : "") + to_string(x) + ":" + (y < 10 ? "0" : "") + to_string(y);
    }
};
