class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        string largest = "";
        int min = 0, sec = 0;
        for (int i = 0; i < 4; ++i) {
            for (int j = 0; j < 4; ++j) {
                if (i == j) continue;
                for (int k = 0; k < 4; ++k) {
                    if (k == i || k == j) continue;
                    for (int l = 0; l < 4; ++l) {
                        if (l == i || l == j || l == k) continue;
                        if (A[i]*10+A[j] > 23 || A[k]*10+A[l] > 59) continue;
                        if (A[i]*10+A[j] > min ||  (A[i]*10+A[j] == min && A[k]*10+A[l] >= sec)) {
                            min = A[i]*10+A[j];
                            sec = A[k]*10+A[l];
                            largest.assign(5, '0');
                            largest[0] = A[i]+'0';
                            largest[1] = A[j]+'0';
                            largest[2] = ':';
                            largest[3] = A[k]+'0';
                            largest[4] = A[l]+'0';
                        }
                    }
                }
            }
        }
        return largest;
    }
};
