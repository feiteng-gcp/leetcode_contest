class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        string ans = "";
        int ansHour = -1; int ansMin = -1;
        sort(A.begin(), A.end());
        do{
            int t1 = A[0]*10 + A[1];
            int t2 = A[2]*10 + A[3];
            // printf("%d %d\n", t1, t2);
            if(t1 <= 23 && t2 <= 59) {
                // cout << "hhh" << endl;
                if(t1 > ansHour) {
                    ansHour = t1; ansMin = t2;
                } else if(t1 == ansHour && t2 > ansMin){
                    ansHour = t1; ansMin = t2;
                }
            } 
        }while(next_permutation(A.begin(), A.end()));
        
        if(ansHour != -1) {
            
            ans = (ansHour < 10? "0" : "") + to_string(ansHour) + ":" + (ansMin < 10? "0" : "") + to_string(ansMin);
        }
        return ans;
    }
};
