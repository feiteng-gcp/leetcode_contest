class Solution {
public:
    string largestTimeFromDigits(vector<int>& a) {
        string ret="0000";
        sort(a.begin(),a.end());
        string ans="";
        do{
            for(int i=0;i<4;i++)
                ret[i]=(char)(a[i]+'0');
            if(ret>="0000"&&ret<="2359"&&ret[2]<='5'){
                ans="";
                ans+=ret[0];
                ans+=ret[1];
                ans+=":";
                ans+=ret[2];
                ans+=ret[3];
            }
        }while(next_permutation(a.begin(),a.end()));
        return ans;
    }
};
