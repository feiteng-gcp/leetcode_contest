class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        vector<int> v = A;
        sort(v.begin(),v.end());
        int now = -1;
        string ans;
        do
        {
            int hh = v[0] * 10 + v[1];

            int mm = v[2] * 10 + v[3];
            int tot = hh * 60 + mm;
            if(tot < 1440 && tot > now && hh < 24 && mm < 60)
            {
                now = tot;
                ans.clear();
                ans += v[0] + '0';
                ans += v[1] + '0';
                ans += ":";
                ans += v[2] + '0';
                ans += v[3] + '0';
            }
        }
        while( next_permutation(v.begin(),v.end()));
        
        return ans;
        
    }
};
