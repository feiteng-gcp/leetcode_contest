class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        int maxH = 0;
        int maxM = 0;
        string result;
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                if (j == i) continue;
                for (int k = 0; k < 4; k++) {
                    if (k == i || k == j) continue;
                    for (int l = 0; l < 4; l++) {
                        if (l == i || l == j || l == k) continue;
                        int h = A[i] * 10 + A[j];
                        int m = A[k] * 10 + A[l];
                        if (h > maxH && h < 24 && m < 60) {
                            maxH = h;
                            maxM = m;
                            result = to_string(A[i]) + to_string(A[j]) + ":" + to_string(A[k]) + to_string(A[l]);
                        } else if (h == maxH && m >= maxM && m < 60) {
                            maxH = h;
                            maxM = m;
                            result = to_string(A[i]) + to_string(A[j]) + ":" + to_string(A[k]) + to_string(A[l]);
                        }
                    }
                }
            }
        }
        return result;
    }
};
