class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        int val = -1;
        string ret = "";
        for(int i=0;i<A.size();i++){
            for(int j=0;j<A.size();j++){
                if(i != j){
                    for(int k=0;k<A.size();k++){
                        if(i != k && j != k){
                            for(int l=0;l<A.size();l++){
                                if(l != i && l != j && l != k){
                                    int h = A[i]*10+A[j];
                                    int m = A[k]*10+A[l];
                                    if(h >= 0 && h < 24 && m >= 0 && m < 60){
                                        if(h*60 + m > val){
                                            val = h*60 + m;
                                            ret = to_string(A[i])+to_string(A[j]) + ':' + to_string(A[k])+to_string(A[l]);
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        return ret;
    }
};
