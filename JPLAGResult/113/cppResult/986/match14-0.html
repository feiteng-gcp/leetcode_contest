class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        string ret,s;
        for(int a: A)s +='0'+a;
        sort(s.begin(),s.end());
        do{
           if(s.substr(0,2) <= "23" && s.substr(2) <= "59") {
               ret=s;
               ret.insert(2, ":");
           }
        }while(next_permutation(s.begin(),s.end()));
        
        return ret;
    }
};
