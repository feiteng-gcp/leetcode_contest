class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        
        string ret;
        int max = -1;

        for(int i=0;i<4;++i) {
            if (A[i] > 2) continue;
            
            for(int j=0;j<4;++j) {
                if(i==j) continue;
                if (A[i]*10+A[j] > 23) continue;
                
                for(int k=0;k<4;++k) {
                    if (i==k || j==k) continue;
                    if (A[k] > 5) continue;
                    
                    for(int l=0;l<4;++l) {

                        if (i==l || j==l || k==l) continue;
                        int curTime = A[i]*1000+A[j]*100+A[k]*10+A[l];
                        if (curTime > max) {
                            max = curTime;
                            ret = to_string(A[i])+to_string(A[j])+":"+to_string(A[k])+to_string(A[l]);
                        }
                    }
                }
            }
        }
        
        return ret;
    }
};
