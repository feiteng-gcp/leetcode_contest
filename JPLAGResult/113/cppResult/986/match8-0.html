class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        sort(A.begin(), A.end());
        
        int ans_h = -1;
        int ans_m = -1;
               
        do{
            int h = A[0] * 10 + A[1];
            int m = A[2] * 10 + A[3];
            
            if(h < 24 && m < 60){
                
                if(h > ans_h || (h == ans_h && m > ans_m)){
                    ans_h = h;
                    ans_m = m;
                }
                
            }
        }while(next_permutation(A.begin(), A.end()));
        
        if(ans_h == -1){
            return "";
        }
        
        return to_string(ans_h/10) + to_string(ans_h%10) + ":" + to_string(ans_m/10) + to_string(ans_m%10);
    }
};
