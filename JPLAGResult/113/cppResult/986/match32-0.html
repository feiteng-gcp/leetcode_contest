class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        sort(A.begin(), A.end());
        int rh, rm;
        rh = rm = -1;
        do {
            int hr = A[0] * 10 + A[1];
            int mn = A[2] * 10 + A[3];
            
            if (hr < 24 && mn < 60) {
                rh = hr;
                rm = mn;
            }
        }while (next_permutation(A.begin(), A.end()));
        if (rh == -1) return "";
        char str[10];
        sprintf(str, "%02d:%02d", rh, rm);
        string s = str;
        return s;
    }
    
};
