class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        sort(A.begin(), A.end());
        
        int mx = -1;
        string s;
        do {
            int h = A[0] * 10 + A[1], m = A[2] * 10 + A[3];
            if (h >= 0 && h < 24 && m >= 0 && m < 60) {
                if (h * 60 + m > mx) {
                    mx = h * 60 + m;
                    s = to_string(A[0]) + to_string(A[1]) + ':' + to_string(A[2]) + to_string(A[3]);
                }
            }
        } while (next_permutation(A.begin(), A.end()));

        return s;
    }
};
