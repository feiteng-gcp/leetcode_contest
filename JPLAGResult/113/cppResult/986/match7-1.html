class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {

        int n = A.size();
        string ret = "";
        bool mark[4];
        for (int h0 = 0; h0 < n; h0 ++) {
            for (int h1 = 0; h1 < n; h1 ++) {
                if (h1 == h0) continue;
                int h = A[h0] * 10 + A[h1];

                if (h < 0 || h > 23) continue;
                for (int m0 = 0; m0 < n; m0 ++) {
                    if (m0 == h0 || m0 == h1) continue;
                    for (int m1 = 0; m1 < n; m1 ++) {
                        if (m1 == h0 || m1 == h1 || m1 == m0) continue;
                        int m = A[m0] * 10 + A[m1];
                        if (m < 0 || m > 59) continue;
                        string cur = to_string(A[h0]) + to_string(A[h1]) + ":" + to_string(A[m0]) + to_string(A[m1]);
                        if (ret == "" || cur > ret) {
                            ret = cur;
                        }
                    }
                }
            }
        }
        return ret;
    }
};
