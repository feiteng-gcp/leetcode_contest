class Solution {
    bool isValid(int hour, int minute)
    {
        return (hour >= 0) && (hour < 24) && (minute >= 0) && (minute < 60);
    }
    vector<int> composeHourAndMinute(int a, int b, int c, int d)
    {
        vector<int> res;
        res.push_back(a*10 + b);
        res.push_back(c*10 + d);
        return res;
    }
    bool isBetter(int hour, int minute, int h, int m)
    {
        return (hour * 60 + minute > h * 60 + m);
    }
public:
    string largestTimeFromDigits(vector<int>& A) {

        int bestHour = -1, bestMinute = -1;
        for (int i = 0; i < 4; i++)
        {
            for (int j = 0; j < 4; j++)
            {
                if (j == i) continue;
                for (int k = 0; k < 4; ++k)
                {
                    if (k == i || k==j) continue;
                    for (int m = 0; m < 4; m++)
                    {
                        if (m == k || m== j || m == i) continue;
                        vector<int> res = composeHourAndMinute(A[i], A[j], A[k], A[m]);
                        if (!isValid(res[0], res[1])) continue;
                        if (isBetter(res[0], res[1], bestHour, bestMinute))
                        {
                            bestHour = res[0];
                            bestMinute = res[1];
                        }
                    }
                }
                
            }
        }
        if (bestHour >= 0)
        {
            string hour = to_string(bestHour);
            hour = (hour.length() == 2) ? (hour) : ("0" + hour);
            string minute = to_string(bestMinute);
            minute = (minute.length() == 2) ? (minute) : ("0" + minute);
            string ret = hour + ":" + minute;
            return ret;
        }
        return "";
    }
};
