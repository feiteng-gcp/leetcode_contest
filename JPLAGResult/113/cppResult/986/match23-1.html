class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {

        int a=-1,b=-1;
        string ans;
        sort(A.begin(),A.end());
        do
        {
            int ta=A[0]*10+A[1],tb=A[2]*10+A[3];
            if(ta<=23&&tb<=59)
            {
                if(ta>a||ta==a&&tb>b)
                {
                    a=ta; b=tb;
                }
            }
        }
        while(next_permutation(A.begin(),A.end()));
        if(a==-1) return "";
        if(a<10) ans="0"; 
        ans+=to_string(a);
        ans+=":";
        if(b<10) ans+="0";
        ans+=to_string(b);
        return ans;
    }
};
