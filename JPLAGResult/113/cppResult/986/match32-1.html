class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        sort(A.begin(), A.end());
        int mx1 = -1, mx2 = 0;
        string ret;
        do {
            int v1 = A[0]*10 + A[1];
            int v2 = A[2]*10 + A[3];
            if (v1 <= 23 && v2 < 60) {
                if (v1 > mx1 || (v1 == mx1 && v2 > mx2)) {
                    mx1 = v1;
                    mx2 = v2;
                    ret = "";
                    for (int i = 0; i < A.size(); ++ i) {
                        if (i == 2)
                            ret += ":";
                        ret += ('0' + A[i]);
                    }
                }
            }
        }while(next_permutation(A.begin(), A.end()));
        return ret;
    }
};
