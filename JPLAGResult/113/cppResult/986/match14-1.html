class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        
        if(A.size()!=4)
            return ""; 
        
        sort(A.begin(), A.end()); 
        

        string res = ""; 
        
        
        do{
            
            if(A[0]*10+A[1] < 24 && A[2]*10+A[3]<60) {
                
                res = string(1, '0'+A[0]) + string(1, '0'+A[1]) +":"+ string(1, '0'+A[2])+ string(1, '0'+A[3]); 
            }
            
        }while(next_permutation(A.begin(),A.end())); 
        return res; 
        
    }
};
