#include <algorithm>
class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        int maxx = -1;
        for (int i1 = 0; i1 < 4; i1 ++){
            for (int i2 = 0; i2 < 4; i2 ++){
                for (int i3 = 0; i3 < 4; i3 ++){
                    for (int i4 = 0; i4 < 4; i4 ++){
                        if (i1 != i2 && i1 != i3 && i1 != i4 && i2 != i3 && i2 != i4 && i3 != i4){
                            int h = A[i1]*10+A[i2];
                            int m = A[i3]*10+A[i4];
                            if (h*60+m < 24*60 && m < 60) maxx = max(maxx, h*60+m);
                        }
                    }
                }
            }
        }
        if (maxx == -1) return "";
        string hh = to_string(maxx/60);
        if (hh.length()==1) hh = "0"+hh;
        string mm = to_string(maxx%60);
        if (mm.length()==1) mm = "0"+mm;
        string s = hh+":"+mm;
        return s;
    }
};
