class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        vector<int> count(10, 0);
        for (auto&& a : A) {
            ++count[a];
        }
        for (int i = 23;i >= 0;--i) {
            for (int j = 59;j >= 0;--j) {
                vector<int> temp(10, 0);
                ++temp[i%10];
                ++temp[i/10];
                ++temp[j%10];
                ++temp[j/10];
                bool valid = true;
                for (int k = 0;k < 10;++k) {
                    if (temp[k] != count[k]) {
                        valid = false;
                        break;
                    }
                }
                if (valid) {
                    stringstream result;
                    if (i < 10) {
                        result << "0";
                    }
                    result << i << ":";
                    if (j < 10) {
                        result << "0";
                    }
                    result << j;
                    return result.str();
                }
            }
        }
        return "";
    }
};
