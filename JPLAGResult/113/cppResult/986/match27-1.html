class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        string res;
        int max_hr = -1;
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 4; j++){
                for(int k = 0; k < 4; k++){
                    for(int l = 0; l < 4; l++){
                        if(i != j && j != k && k != l && i != l && i != k && j != l){
                            //cout << i<<j<<k<<l;
                            if(A[i] * 10 + A[j] <= 23){
                                if(max(A[k],A[l]) <= 5){
                                    int hr = A[i] * 10 + A[j];
                                    if(hr > max_hr){
                                        max_hr = hr;
                                        res = to_string(A[i]) + to_string(A[j]) + ":" + to_string(max(A[k],A[l])) + to_string(min(A[k],A[l]));
                                    }
                                }
                                else if(min(A[k],A[l]) <= 5){
                                    int hr = A[i] * 10 + A[j];
                                    if(hr > max_hr){
                                        max_hr = hr;
                                        res = to_string(A[i]) + to_string(A[j]) + ":" + to_string(min(A[k],A[l])) + to_string(max(A[k],A[l]));
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        return res;
    }
};
