class Solution {
public:
    int calc(int a, int b, int c, int d){
        int h = a * 10 + b, m = c * 10 + d;
        if(h >= 24) return -1;
        if(m >= 60) return -1;
        return h * 60 + m;
    }
    
    string largestTimeFromDigits(vector<int>& A) {
        string ans;
        int tmp = -1;
        for(int i = 0; i < 4; ++i){
            for(int j = 0; j < 4; ++j){
                if(i == j) continue;
                for(int p = 0; p < 4; ++p){
                    if(i == p || j == p) continue;
                    for(int q = 0; q < 4; ++q){
                        if(i == q || j == q || p == q) continue;
                        int t = calc(A[i], A[j], A[p], A[q]);
                        string s = to_string(A[i]) + to_string(A[j]) + ":" + to_string(A[p]) + to_string(A[q]);
                        //cout<<s<<" "<<t<<endl;
                        if(t == -1) continue;
                        if(tmp < t){
                            tmp = t;
                            ans = s;
                        }
                    }
                }
            }
        }  
        
        return ans;
    }
};
