class Solution {
public:
    
    string largestTimeFromDigits(vector<int>& A) {
        sort(A.begin(), A.end());
        int tmp = 0;
        string ans = "";
        do{
            int h = (A[0]*10 + A[1]);
            int m = (A[2]*10 + A[3]);
            if (h >= 24) continue;
            if (m >= 60) continue;
            
            // cout << h  << " " << m << endl;
            if(h * 60 + m >= tmp) {
                tmp = h*60+m;
                ans = ( h < 10 ? "0"+to_string(h) : to_string(h)) + ":" + ( m < 10 ? "0"+to_string(m) : to_string(m));
            }
            
        }while(next_permutation(A.begin(), A.end()));
        return ans;
    }
};
