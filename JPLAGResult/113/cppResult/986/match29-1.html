class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        vector<int> cnt(10,0);
        for(auto v:A){
            cnt[v]++;
        }
        for(int i =23;i>=0;i--){
            for(int j =59;j>=0;j--){
                vector<int> needs(10,0);
                needs[i/10]++;
                needs[i%10]++;
                needs[j/10]++;
                needs[j%10]++;
                int a= 0;
                for( a= 0;a<10;a++){
                    if(needs[a]>cnt[a])
                        break;
                }
                if(a==10){
                    string ret = "00:00";
                    ret[0]='0'+(i/10);
                    ret[1]='0'+(i%10);
                    ret[3]='0'+(j/10);
                    ret[4]='0'+(j%10);
                    return ret;
                }
                
            }
        }
        return "";
    }
};
