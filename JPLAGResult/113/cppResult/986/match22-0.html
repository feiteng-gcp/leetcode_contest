class Solution {
public:
	bool is_valid( int h, int m ) {
		return( h < 24 && m < 60 );
	}
    string largestTimeFromDigits(vector<int>& A) {
		int max_min = -1;
		string res("");
        for ( int i1 = 0; i1 < 4; ++i1 ) {
			for ( int i2 = 0; i2 < 4; ++i2 ) {
				if ( i2 == i1 ) {
					continue;
				}
				for ( int i3 = 0; i3 < 4; ++i3 ) {
					if ( i3 == i2 || i3 == i1 ) {
						continue;
					}
					for ( int i4 = 0; i4 < 4; ++i4 ) {
						if ( i4 == i3 || i4 == i2 || i4 == i1 ) {
							continue;
						}
						int h = A[i1]*10 + A[i2];
						int m = A[i3]*10 + A[i4];
                        //cout << h << " " << m << endl;
						if ( is_valid( h, m ) ) {
							if ( h * 60 + m > max_min ) {
								max_min = h * 60 + m;
								res = string(1,A[i1]+'0')+string(1,A[i2]+'0')+
									string(1,':')+
									string(1,A[i3]+'0')+string(1,A[i4]+'0');
							}
						}
					}
				}
			}
		}
		return res;
    }
};
