class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        string ans = "";
        for (int i = 0; i < 4; i++)

            for (int j = 0; j < 4; j++) {
                if (i == j)
                    continue;
                for (int k = 0; k < 4; k++) {
                    if (k == i || k == j)
                        continue;
                    int m = 0 + 1 + 2 + 3 - i - j - k;
                    int hour = A[i] * 10 + A[j];
                    int mini = A[k] * 10 + A[m];
                    if (hour >= 0 && hour < 24 && mini >= 0 && mini < 60) {
                        string str = "";
                        str.push_back('0' + A[i]);
                        str.push_back('0' + A[j]);
                        str.push_back(':');
                        str.push_back('0' + A[k]);
                        str.push_back('0' + A[m]);
                        if (ans.empty() || ans.compare(str) < 0)
                            ans = str;
                    }
                }
            }
        return ans;
    }
};
