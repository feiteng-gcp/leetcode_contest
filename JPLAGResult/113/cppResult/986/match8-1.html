class Solution {
public:
    string largestTimeFromDigits(vector<int>& A) {
        int hour1 = -1;
        int minute1 = -1;
        sort(A.begin(), A.end());
        do{
            int hour = A[0]*10+A[1];
            int minute = A[2]*10+A[3];
            if(hour>=0&&hour<=23&&minute>=0&&minute<60){
                if(hour>hour1||(hour==hour1&&minute>minute1)){
                    hour1 = hour;
                    minute1 = minute;
                }
            }
        }while(next_permutation(A.begin(), A.end()));
        if(hour1!=-1){
            string hs = to_string(hour1);
            if(hs.length()<2){
                hs = "0"+hs;
            }
            string ms = to_string(minute1);
            if(ms.length()<2){
                ms = "0"+ms;
            }
            return hs+":"+ms;
        }else{
            return "";
        }
    }
};
