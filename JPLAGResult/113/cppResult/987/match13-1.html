class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        deque<int> r;
        for (auto iter = deck.rbegin(); iter != deck.rend(); ++iter) {
            if (!r.empty()) {
                int x = r.back();
                r.pop_back();
                r.push_front(x);
            }
            r.push_front(*iter);
        }
        vector<int> res(r.begin(), r.end());
        return res;
    }
};
