class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        vector<int> res;
        sort(deck.begin(),deck.end());
        deque<int> dq;
        for(int i=deck.size()-1;i>=0;i--){
            if(dq.size()){
                int f = dq.back();
                dq.pop_back();
                dq.push_front(f);
            }
            dq.push_front(deck[i]);
        }
        while(dq.size()){
            res.push_back(dq.front());
            dq.pop_front();
        }
        return res;
    }
};
