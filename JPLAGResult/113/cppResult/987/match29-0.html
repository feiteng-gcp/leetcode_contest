class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        deque<int>q;
        int n=deck.size();
        
        sort(deck.begin(),deck.end());
        reverse(deck.begin(),deck.end());
        
        for(int i=0;i<n;i++){
            if(i==0)q.push_back(deck[i]);
            else{
                int val=q.back();
                q.pop_back();
                q.push_front(val);
                q.push_front(deck[i]);
            }
                
        }
        
        vector<int>ans;
        while(q.size()>0){
            ans.push_back(q[0]);
            q.pop_front();
        }
        
        return ans;
    }
};
