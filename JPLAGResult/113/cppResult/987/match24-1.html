class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        int len = deck.size();
        queue<int> Q;
        for(int i = 0; i < len; i++) Q.push(i);
        vector<int> I(len);
        int cnt = 0;
        while(Q.size()){
            int t = Q.front();
            Q.pop();
            I[t] = cnt++;
			if (Q.size()) {

				int n = Q.front();
				Q.pop();
				Q.push(n);
			}
        }
        vector<int> ret(len);
        sort(deck.begin(), deck.end());
        for(int i = 0; i < len; i++){
            ret[i] = deck[I[i]];
        }
        return ret;
    }
};
