class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        deque<int> dq;
        
        for(int i = deck.size()-1; i>= 0; i--)
        {
            if(!dq.empty())
            {
                int e = dq.back();
                dq.push_front(e);
                dq.pop_back();
            }
            dq.push_front(deck[i]);
        }
        
        vector<int> res(dq.begin(), dq.end());
        
        return res;
    }
};
