class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        
        sort(deck.begin(),deck.end());
        int n = deck.size();
        vector<int> r(n,-1);
        vector<int> next(n,-1);
        
        for(int i=0;i<n;i++){
            next[i]= i+1;
        }
        next[n-1] = 0;
        
       int g = 0;
        int prev = n-1;
        int i = 0;
        
        while(g<n){
            i = next[prev];
            r[i] = deck[g++];
            int go = next[i];
            next[prev] = go;
            prev = go;
        }
        return r;
        
    }
};
