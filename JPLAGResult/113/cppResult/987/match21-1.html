class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        int n = deck.size();
        
        vector<int> v;
        for (int i = 0; i < n; i++) {
            v.push_back(i);
        }
        
        vector<int> res;
        while (v.size() > 0) {
            res.push_back(v[0]);
            v.erase(v.begin());
            if (v.size() > 0) {
                v.push_back(v[0]);
                v.erase(v.begin());

            }
        }
        
        vector<int> ret(n);
        for (int i = 0; i < n; i++) {
            ret[res[i]] = deck[i];
        }
        return ret;
        
    }
};
