class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        int n = deck.size();
        sort(deck.begin(),deck.end());
        vector<int> result;
        result.push_back(deck.back());
        for(int i = n - 2; i >=0; --i){
            int tail = result.back();
            result.pop_back();
            result.insert(result.begin(), tail);
            result.insert(result.begin(), deck[i]);
        }
        return result;
    }
};
