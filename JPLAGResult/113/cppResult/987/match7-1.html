class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        queue<int> q;
        for(int i=0;i<deck.size();i++)q.push(i);
        vector<int> inter;
        while(!q.empty())
        {
            inter.push_back(q.front());
            q.pop();
            if(!q.empty())
            {
                int tmp=q.front();q.pop();
                q.push(tmp);
            }
        }
        //for(auto &i:inter)cout<<i<<' ';cout<<endl;
        sort(deck.begin(),deck.end());
        vector<int> ans(deck.size(),0);
        for(int i=0;i<deck.size();i++)ans[inter[i]]=deck[i];
        return ans;
    }
};
