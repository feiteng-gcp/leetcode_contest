class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        vector<int> ans(deck.size());
        queue<int> ind;
        for (int i = 0; i < deck.size(); i++)
            ind.push(i);
        int cnt = 0;
        while (!ind.empty()) {
            ans[ind.front()] = deck[cnt++];
            ind.pop();
            if (!ind.empty()) {
                ind.push(ind.front());
                ind.pop();
            }
        }
        return ans;
    }
};
