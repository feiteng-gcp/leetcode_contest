class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort (deck.begin(), deck.end());
        int n = deck.size();
        queue <int> que;
        que.push(deck[n-1]);
        for (int i = n-2; i >= 0; --i) {

            int now = que.front();
            que.pop();
            que.push(now);
            que.push(deck[i]);
        }
        for (int i = n - 1; i >= 0; --i) {
            deck[i] = que.front();
            que.pop();
        }
        return deck;
    }
};
