class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(),deck.end());
        int N=deck.size();
        deque<int> Q;
        for(int i=0;i<N;++i){
            Q.push_back(i);
        }
        vector<int> A(N);
        for(int i=0;i<N;++i){
            A[i]=Q.front(); Q.pop_front();
            if(!Q.size())break;
            int u=Q.front();
            Q.pop_front(); Q.push_back(u);
        }
        vector<int> R(N);
        for(int i=0;i<N;++i){
            R[A[i]]=deck[i];
        }
        return R;
    }
};
