class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        vector<int> indexes;
        deque<int> q;
        for (int i = 0; i < deck.size(); ++i) {
            q.push_back(i);
        }
        for (int state = 0; !q.empty(); state = 1 - state) {
            if (state == 0) {
                indexes.push_back(q.front());
                q.pop_front();
            } else {
                q.push_back(q.front());
                q.pop_front();
            }
        }
        sort(deck.begin(), deck.end());
        vector<int> results(deck.size());
        for (int i = 0; i < indexes.size(); ++i) {
            results[indexes[i]] = deck[i];
        }
        return results;
    }
};
