class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        vector<int>ans;
        queue <int>q;
        for( int i=0 ; i<deck.size() ; i++ )
            q.push(i);
        while( !q.empty() )
        {
            int a =q.front(); q.pop();
            ans.push_back(a);
            if ( !q.empty() )
            {
                int b = q.front(); q.pop();
                q.push(b);
            }
        }
        sort( deck.begin() ,  deck.end() ); int a[1010];
        for ( int i=0 ; i<deck.size() ; i++ )
            a[ans[i]] = deck[i];
        vector<int> res;
        for ( int i=0 ; i<deck.size() ; i++ )
            res.push_back(a[i]);
        return res;
    }
};
