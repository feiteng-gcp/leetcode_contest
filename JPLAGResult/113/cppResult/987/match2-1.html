class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        deque<int> card;
        for(int i = deck.size() - 1; i >=0; i--)
        {
            if(card.size())
            {
                auto t = card.back();
                card.pop_back();
                card.push_front(t);
            }
            card.push_front(deck[i]);
        }
        vector<int> ans(deck.size(), 0);
        for(int i = 0; i < card.size(); i++)
        {
            ans[i] = card[i];
        }
        return ans;
    }
};
