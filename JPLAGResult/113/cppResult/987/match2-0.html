class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        deque<int> q;
        for (int i = deck.size() - 1; i >= 0; i--) {
            if (!q.empty()) {
                auto v = q.back();
                q.pop_back();
                q.push_front(v);
            }
            q.push_front(deck[i]);
        }
        vector<int> res;
        while (!q.empty()) {
            res.push_back(q.front());
            q.pop_front();
        }
        return res;
    }
};
