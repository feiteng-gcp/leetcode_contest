class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        vector<int> sol(deck.size());
        deque<int> q;
        for (int i = deck.size() - 1; i >= 0; i --) {
            if (q.size()) {
                int v = q.back();
                q.pop_back();
                q.push_front(v);
            }
            q.push_front(deck[i]);
        }
        for (int i = 0; i < deck.size(); ++i) {
            sol[i] = q[i];
        }
        return sol;
    }
};
