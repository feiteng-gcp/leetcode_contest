class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        vector<int> re;
        list<int> li;
        for(int i = deck.size()-1; i >= 0; --i) {
            if(li.size() > 1) {
                int back = li.back();
                li.pop_back();
                li.push_front(back);
            }
            li.push_front(deck[i]);
        }
        for(auto it = li.begin(); it != li.end(); ++it) {
            re.push_back(*it);
        }
        return re;
    }
};
