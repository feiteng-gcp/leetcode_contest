class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& a) {
        int n = a.size();
        deque<int> p(n);
        for (int i = 0; i < n; ++i) p[i] = i;
        vector<int> q;
        while (p.size()) {
            q.push_back(p.front());
            p.pop_front();
            if (p.size()) {
                auto x = p.front();
                p.push_back(x);
                p.pop_front();
            }
        }
        sort(a.begin(), a.end());
        vector<int> ret(n);
        for (int i = 0; i < n; ++i) {
            ret[q[i]] = a[i];
        }
        return ret;
    }
};
