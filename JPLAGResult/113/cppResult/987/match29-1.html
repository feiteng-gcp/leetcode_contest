class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        reverse(deck.begin(), deck.end());
        deque < int > ans;

        for (int i = 0; i < deck.size(); i++) {
            if (ans.size() == 0) ans.push_back(deck[i]);
            else {
                int t = ans.back();
                ans.pop_back();
                ans.push_front(t);
                ans.push_front(deck[i]);
            }
        }
        vector < int > ret;
        while (!ans.empty()) {
            ret.push_back(ans.front());
            ans.pop_front();
        }
        return ret; 
    }
};
