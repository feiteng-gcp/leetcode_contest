class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        sort(deck.begin(), deck.end());
        deque<int> dq;
        for (int i = deck.size()-1; i >= 0; i --){
            if (dq.size()>0){
                int b = dq.back(); dq.pop_back();
                dq.push_front(b);
            }
            dq.push_front(deck[i]);
        }
        vector<int> res;
        while (dq.size()>0){
            int b = dq.front();
            res.push_back(b);
            dq.pop_front();
        }
        return res;
    }
};
