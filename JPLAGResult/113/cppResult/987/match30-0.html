class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& a) {
		sort(a.begin(), a.end());
		int n = a.size();
		list<int> tmp;
		tmp.push_back(a[n - 1]);
		for (int l = 2; l <= n; ++l)
		{
			int nl = l - 1;
			auto it = tmp.back();
			tmp.pop_back();
			tmp.push_front(it);
			tmp.push_front(a[n - l]);
		}
		vector<int> ans(n);
		// for (int e : tmp)
		// 	cout << e << " ";
		// cout << endl;
		auto it = tmp.begin();
		for (int i = 0; i < n; ++i)
		{
			ans[i] = *it;
			it++;
		}
		return ans;   
    }
};
