class Solution {
public:
    vector<int> deckRevealedIncreasing(vector<int>& deck) {
        queue<int> qu;
        for(int i = 0; i < deck.size(); i++)
            qu.push(i);
        vector<int> vec;
        while(!qu.empty())
        {
            vec.push_back(qu.front());
            qu.pop();
            if(!qu.empty())
            {
                int v = qu.front();
                qu.pop();
                qu.push(v);
            }
        }
        sort(deck.begin(), deck.end());
        vector<int> res(deck.size(), 0);
        for(int i = 0; i < res.size(); i++)
            res[vec[i]] = deck[i];
        return res;
    }
};
