class Solution {
    public int largestComponentSize(int[] A) {
        int n = A.length;
        Map<Integer, Integer> map = new HashMap();
        for (int num : A) {
            if (num == 1) {map.put(1, 1); continue;}
            int idx = 2;
            
            while ((num % idx) != 0 && idx * idx <= num) idx++;
            if (idx * idx > num) {map.put(num, find(num, map)); continue;}
            int pp = find(idx, map);
            map.put(idx, pp);
            map.put(num, pp);
            num /= idx;
            while (num > 1) {
                while ((num % idx) != 0 && idx * idx <= num) idx++;
                if (idx * idx > num) {map.put(find(num, map), pp); break;}
                map.put(find(idx, map), pp);
                num /= idx;
            }
        }
        Map<Integer, Integer> countMap = new HashMap();
        int res = 0;
        for (int num : A) {
            int pp = find(map.get(num), map);
            countMap.put(pp, countMap.getOrDefault(pp, 0) + 1);
            res = Math.max(res, countMap.get(pp));
        }
        return res;
    }
    int find(int idx, Map<Integer, Integer> map) {
        if (!map.containsKey(idx)) return idx;
        if (map.get(idx) != idx) {
            map.put(idx, find(map.get(idx), map));
        }
        return map.get(idx);
    }
}
