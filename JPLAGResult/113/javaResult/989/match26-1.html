import java.util.*;

class Solution {

    class Node {
        List<Node> neighbors = new ArrayList<Node>();
        boolean visited = false;
        boolean isNumberNode = false;
    }

    public int largestComponentSize(int[] A) {

        Map<Integer,Node> factorToNode = new HashMap<>();
        Node[] numberNodes = new Node[A.length];

        for (int i = 0; i < A.length; i++) {
            int a = A[i];
            numberNodes[i] = new Node();
            numberNodes[i].isNumberNode = true;

            // Discover the factors of a and connect.
            for(int factor = 2; factor <= (int)Math.sqrt(a); factor++) {

                if (a % factor == 0) {

                    int factor2 = a / factor;

                    factorToNode.putIfAbsent(factor, new Node());
                    factorToNode.putIfAbsent(factor2, new Node());

                    Node factorNode = factorToNode.get(factor);
                    Node factor2Node = factorToNode.get(factor2);

                    factorNode.neighbors.add(numberNodes[i]);
                    factor2Node.neighbors.add(numberNodes[i]);
                    numberNodes[i].neighbors.add(factorNode);
                    numberNodes[i].neighbors.add(factor2Node);
                }
            }

            factorToNode.putIfAbsent(a, new Node());
            factorToNode.get(a).neighbors.add(numberNodes[i]);
            numberNodes[i].neighbors.add(factorToNode.get(a));
        }

        int maxGroupSize = 1;
        int thisGroupSize = 0;
        for (int i = 0; i < A.length; i++) {

            if (numberNodes[i].visited) { continue; }

            thisGroupSize = 0;
            
Queue<Node> que = new ArrayDeque<>();
            que.add(numberNodes[i]);
            numberNodes[i].visited = true;
            while (!que.isEmpty()) {

                Node on = que.poll();
                if (on.isNumberNode) { thisGroupSize++; }

                for(Node neighbor : on.neighbors) {
                    if (!neighbor.visited) {
                        neighbor.visited = true;
                        que.add(neighbor);
                    }
                }
            
}
            maxGroupSize = Math.max(thisGroupSize, maxGroupSize);
        }
        return maxGroupSize;
    }
}
