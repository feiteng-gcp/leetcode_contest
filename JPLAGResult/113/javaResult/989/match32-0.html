class Solution {

	private int findSet(int x) {
		if (x == set[x]) {
			return x;
		} else {
			return set[x] = findSet(set[x]);
		}
	}
	
	private static int[] cnt = new int[100010];
	private static int[] set = new int[100010];
	private static boolean[] flag = new boolean[100010];
	
    public int largestComponentSize(int[] A) {
    
    	Arrays.fill(cnt , 0);
    	Arrays.fill(flag , false);
    	for (int i = 1;i <= 100000;i ++) {
    		set[i] = i;
    	}
    	
    	for (int value : A) {
    		cnt[value] ++;
    	}
    	
    	for (int i = 2;i <= 100000;i ++) {
    		// prime
    		if (!flag[i]) {
    			int prevF = - 1;
    			if (cnt[i] > 0) {
    				prevF = findSet(i);
    			}
    			for (int j = 2;j * i <= 100000;j ++) {    				
    				flag[j * i] = true;
    				if (cnt[j * i] > 0 && prevF >= 0) {
    					int f = findSet(j * i);
    					if (f != prevF) {
	    					set[f] = prevF;
	    					cnt[prevF] += cnt[f];
    					}
    				}
    				if (cnt[j * i] > 0 && prevF < 0) {
    					prevF = findSet(j * i);
    				}
    			}
    		}
    	}    	
    	
    	int ans = 0;
    	for (int i = 1;i <= 100000;i ++) {
    		int f = findSet(i);
    		if (cnt[f] > ans) {
    			ans = cnt[f];
    		}
    	}
    	
    	return ans;
    	
    }
}
