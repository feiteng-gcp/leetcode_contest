public class Solution {
	public int largestComponentSize(int[] A) {
		Arrays.sort(A);
		int n = A.length;
		Map<Integer, Integer> idx = new HashMap<>();
		for (int i = 0; i < n; ++i) idx.put(A[i], i);
		int[] parent = new int[n];
		for (int i = 0; i < n; ++i) parent[i] = i;
		boolean[] isPrime = new boolean[100001];
		Arrays.fill(isPrime, true);
//		Set<Integer> primes = new HashSet<>();
		for (int i = 2; i < isPrime.length; ++i) {
			if (!isPrime[i]) continue;
//			primes.add(i);
			Set<Integer> multiples = new HashSet<>();
			if (idx.containsKey(i))
				multiples.add(idx.get(i));
			for (int j = 2 * i; j < isPrime.length; j += i) {
				isPrime[j] = false;
				if (idx.containsKey(j))
					multiples.add(idx.get(j));
			}
			if (!multiples.isEmpty()) {
				int base = multiples.iterator().next();
				for (int num : multiples) {
					merge(parent, base, num);
				}
			}
		}
		int[] cnt = new int[n];
		for (int i = 0; i < n; ++i)
			cnt[find(parent, i)]++;

		int ans =0;
		for (int c : cnt) {
			ans = Math.max(ans, c);
		}
		return ans;
	}

	private void merge(int[] parent, int i1, int i2) {
		int p1 = find(parent, i1), p2 = find(parent, i2);
		parent[p2] = p1;
	}

	private int find(int[] parent, int i) {
		if (parent[i] == i) return i;
		return parent[i] = find(parent, parent[i]);
	}
}
