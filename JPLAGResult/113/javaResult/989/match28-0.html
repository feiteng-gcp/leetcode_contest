class Solution {
    public int largestComponentSize(int[] A) {
        ArrayList<Integer> primes = new ArrayList<Integer>();
        boolean[] sieve = new boolean[100001];
        boolean[] nums = new boolean[100001];
        for (int i : A)
            nums[i] = true;
        for (int i = 2; i <= 100000; i++) {
            if (!sieve[i]) {
                primes.add(i);
                if (i > 1000)
                    continue;
                for (int j = i*i; j <= 100000; j += i)
                    sieve[j] = true;
            }
        }
        ArrayList<ArrayList<Integer>> g = new ArrayList<ArrayList<Integer>>();
        for (int i = 0; i < primes.size(); i++)
            g.add(new ArrayList<>());
        for (int i = 0; i < primes.size(); i++) {
            int p1 = primes.get(i);
            if (p1>1000)
                break;
            for (int j = i+1; j < primes.size(); j++) {
                int p2 = primes.get(j);
                if (p1*p2>100000)
                    break;
                int mul = p1*p2;
                for (int k = mul; k <= 100000; k += mul) {
                    if (nums[k]) {
                        g.get(i).add(j);
                        g.get(j).add(i);
                        break;
                    }
                }
            }
        }
        int[] att = new int[primes.size()];
        for (int i = 0; i < primes.size(); i++) {
            att[i] = -1;
        }
        for (int i = 0; i < primes.size(); i++) {
            if (att[i] == -1) {
                att[i] = i;
                ArrayDeque<Integer> q = new ArrayDeque<Integer>();
                q.add(i);
                while (!q.isEmpty()) {
                    int u = q.removeFirst();
                    for (int v : g.get(u)) {
                        if (att[v]==-1) {
                            att[v] = i;
                            q.add(v);
                        }
                    }
                }
            }
        }
        ArrayList<HashSet<Integer>> ng = new ArrayList();
        for (int i = 0; i < primes.size(); i++) {
            ng.add(new HashSet<Integer>());
        }
        int ans = 0;
        for (int i = 0; i < primes.size(); i++) {
            int a = att[i];
            int p = primes.get(i);
            for (int j = p; j <= 100000; j += p) {
                if (nums[j]) {
                    ng.get(a).add(j);
                }
            }
            if (ng.get(a).size() > ans)
                ans = ng.get(a).size();
        }
        return ans;
    }
}
