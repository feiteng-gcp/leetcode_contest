class Solution {
    
    
private Set<Integer> primeFactors(int n) { 
        Set<Integer> arr = new HashSet<>();
        while (n % 2 == 0) 
        { 
            arr.add(2);
            n = n/2; 
        } 

        for (int i = 3; i <= Math.sqrt(n); i = i+2) 
        { 
            // While i divides n, print i and divide n 
            while (n%i == 0) 
            { 
                arr.add(i); 
                n = n/i; 
            } 
        } 

        if (n > 2) 
            arr.add(n);
        return arr;
    } 
    
    HashMap<Integer,List<Integer>> hs = new HashMap<>();
    
    int[] parent;
    int[] rank;
    
    private int find(int i) {
        if (parent[i] == i) return i;
        else {
            int res = find(parent[i]);
            parent[i] = res;
            return res;
        }
    
}
    
    private void union(int i, int j) {
        
        int iparent = find(i);
        int jparent = find(j);
        int irank = rank[iparent];
        int jrank = rank[jparent];
        
        if (irank > jrank) {
            parent[jparent] = iparent;
        }
        else if (irank < jrank) {
            parent[iparent] = jparent;
        }
        else {
            parent[iparent] = jparent;
            rank[jparent]++;
        }
        
    }
    
    public int largestComponentSize(int[] A) {
        parent = new int[A.length];
        rank = new int[A.length];
        for (int i=0; i < A.length; i++) { 
            parent[i] = i;
            rank[i] = 1;
        }
        for (int i=0; i < A.length; i++) {
            for (Integer e : primeFactors(A[i])) {
                if (!hs.containsKey(e)) hs.put(e, new ArrayList<>());
                
                for (Integer p : hs.get(e)) {
                    union(p, i);
                    break;
                }
                hs.get(e).add(i);
                
            }
        }
        
        HashMap<Integer,Integer> count = new HashMap<>();
        int max = 0;
        for (int i=0; i < A.length; i++) {
            int p = find(i);
            count.put(p, count.getOrDefault(p,0)+1);
            max = Math.max(max,count.get(p));
        }
        
        return max;
    }
}
