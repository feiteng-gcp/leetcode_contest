class Solution {
    int[] parent = new int[100001];
    int[] size = new int[100001];
    public int largestComponentSize(int[] A) {
        for(int i = 0; i < parent.length; i++){
            parent[i] = i;
        }
        for(int a: A) {
            size[find(a)]++;
            int tmp = a;
            for(int i = 2; i*i <= a; i++) {
                if(a % i == 0) {
                    union(a, i);
                }
                while(tmp % i == 0) tmp/= i;
            }
            if(tmp > 1) union(a, tmp);
        
}
        
        int res = 0;
        for(int i = 1; i < size.length; i++){
            res = Math.max(res, size[i]);
        }
        return res;
    
}
    
    private void union(int a, int b) {
        int pa = find(a);
        int pb = find(b);
        if(pa != pb) {
            size[pa] += size[pb];
            parent[pb] = pa;
        }
    }
    
    private int find(int x) {
        if(parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        
        return parent[x];
    }
}
