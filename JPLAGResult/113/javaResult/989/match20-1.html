	class Solution {
	    public int largestComponentSize(int[] A) {
	    	int[] lpf = enumLowestPrimeFactors(100001);
	        DJSet ds = new DJSet(100001);
	        for(int
 v : A){
	        	ds.w[v] = 1;
	        }
	        for(int v : A){
	        	int vv = v;
	        	while(vv > 1){
	        		ds.union(v, lpf[vv]);
	        		vv /= lpf[vv];
	        	}
	        }
	        int ret = 0;
	        for(int i = 1;i <= 100000;i++){
	        	if(ds.upper[i] < 0){
	        		ret = Math.max(ret, ds.w[i]);
	        	}
	        }

	        return ret;
	    }
	    
	    public int[] enumLowestPrimeFactors(int n) {
			int tot = 0;
			int[] lpf = new int[n + 1];
			int u = n + 32;
			double lu = Math.log(u);
			int[] primes = new int[(int) (u / lu + u / lu / lu * 1.5)];
			for (int i = 2; i <= n; i++)
				lpf[i] = i;
			for (int p = 2; p <= n; p++) {
				if (lpf[p] == p)
					primes[tot++] = p;
				int tmp;
				for (int i = 0; i < tot && primes[i] <= lpf[p] && (tmp = primes[i] * p) <= n; i++) {
					lpf[tmp] = primes[i];
				}
			}
			return lpf;
		}

	    
	    public class DJSet {
			public int[] upper;
			public int[] w;

			public DJSet(int n) {
				upper = new int[n];
				Arrays.fill(upper, -1);
				w = new int[n];
			}

			public int root(int x) {
				return upper[x] < 0 ? x : (upper[x] = root(upper[x]));
			}

			public boolean equiv(int x, int y) {
				return root(x) == root(y);
			}

			public boolean union(int x, int y) {
				x = root(x);
				y = root(y);
				if (x != y) {
					if (upper[y] < upper[x]) {
						int d = x;
						x = y;
						y = d;
					}
					upper[x] += upper[y];
					upper[y] = x;
					w[x] += w[y];
				}
				return x == y;
			}

			public int count() {
				int ct = 0;
				for (int u : upper)
					if (u < 0)
						ct++;
				return ct;
			}
		}

	}
