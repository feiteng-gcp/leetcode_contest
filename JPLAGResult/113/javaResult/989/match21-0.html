class Solution {

    private int[] par = new int[100001];
    private int[] rank = new int[100001];
    private int[] ct = new int[100001];

    private void makeSet(int x) {
        par[x] = x;
        rank[x] = 0;
        ct[x] = 0;
    }

    private void union(int x, int y) {
        int xroot = find(x);
        int yroot = find(y);
        if (rank[xroot] > rank[yroot]) {
            par[yroot] = xroot;
            ct[xroot] += ct[yroot];
        } else {
            if (rank[xroot] < rank[yroot]) {
                par[xroot] = yroot;
                ct[yroot] += ct[xroot];
            } else if (xroot != yroot) {
                par[yroot] = xroot;
                ct[xroot] += ct[yroot];
                rank[xroot] += 1;
            }
        }
    }

    private int find(int x) {
        if (par[x] == x)
            return x;
        par[x] = find(par[x]);
        return par[x];
    }

    public int largestComponentSize(int[] A) {
        for (int i = 0; i < A.length; i++) {
            int v = A[i];
            if (v == 1)
                continue;
            ArrayList<Integer> divs = new ArrayList<>();
            int j = 2;
            while (j * j < v) {
                if (v % j == 0) {
                    divs.add(j);
                    divs.add(v / j);
                }
                j += 1;
            }
            if (j * j == v) {
                divs.add(j);
            }
            divs.add(v);
            int firstDiv = divs.get(0);
            if (par[firstDiv] == 0) {
                makeSet(firstDiv);
            }
            for (int ki = 1; ki < divs.size(); ki++) {
                int kdiv = divs.get(ki);
                if (par[kdiv] == 0) {
                    makeSet(kdiv);
                }
                union(firstDiv, kdiv);
            }
            ct[find(firstDiv)] += 1;
        }
        int res = 0;
        for (int i = 0; i <= 100000; i++) {
            res = (ct[i] > res) ? ct[i] : res;
        }
        return res;
    }
}
