class Solution {
    private class UF {
        private int[] root;
        private int[] size;
        private int N;
        public UF(int n) {
            N = n;
            root = new int[N];
            size = new int[N];
            for (int i = 0; i < N; i++) {
                root[i] = i;
                size[i] = 1;
            }
        }
        public int find(int p) {
            while (p != root[p]) p = root[p];
            return p;
        }
        public void union(int p, int q) {
            int rp = find(p);
            int rq = find(q);
            if (rp == rq) return;
            if (size[rp] < size[rq]) {
                size[rq] += size[rp];
                root[rp] = rq;
            }
            else {
                size[rp] += size[rq];
                root[rq] = rp;
            }
        }
        public int getMax() {
            int max = 0;
            for (int i = 0; i < N; i++) {
                max = Math.max(max, size[i]);
            }
            return max;
        }
    }
    public int largestComponentSize(int[] A) {
        // A = new int[]{2,7,522,526,535,26,944,35,519,45,48,567,266,68,74,591,81,86,602,93,610,621,111,114,629,641,131,651,142,659,669,161,674,163,180,187,190,194,195,206,207,218,737,229,240,757,770,260,778,270,272,785,274,290,291,292,296,810,816,314,829,833,841,349,880,369,147,897,387,390,905,405,406,407,414,416,417,425,938,429,432,926,959,960,449,963,966,929,457,463,981,985,79,487,1000,494,508};
        int m = 100000;
        boolean[] arr = new boolean[m + 1];
        Map<Integer, Integer> map = new HashMap<>();
        UF uf = new UF(A.length);
        for (int i = 0; i < A.length; i++) {
            if (map.containsKey(A[i])) uf.union(map.get(A[i]), i);
            else map.put(A[i], i);
        }
        for (int i = 2; i <= m; i++) {
            if (!arr[i]) {
                int j = i;
                int k = 0;
                for (;j <= 100000; j += i) {
                    if (j <= m) arr[j] = true;
                    if (map.containsKey(j)) {
                        k = map.get(j);
                        j += i;
                        break;
                    }
                }
                for (;j <= 100000; j += i) {
                    if (j <= m) arr[j] = true;
                    if (map.containsKey(j)) {
                        uf.union(k, map.get(j));
                    }
                }
            }
        }
        return uf.getMax();
    }
}
