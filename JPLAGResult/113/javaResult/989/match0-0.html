class Solution {
    int[] root = new int[100001];
    int[] size = new int[100001];
    private int find(int a) {
        if (a == root[a]) {
            return a;
        }
        return root[a] = find(root[a]);
    }

    private void union(int a, int b) {
        int rootA = find(a);
        int rootB = find(b);
        if (rootA != rootB) {
            root[rootA] = rootB;
            size[rootB] += size[rootA];
        }
    }

    public int largestComponentSize(int[] A) {
        Set<Integer> set = new HashSet<>();
        int maxx = 0;
    
        for (int a : A) {
            //root[a] = a;
            set.add(a);
            size[a]++;
            maxx = Math.max(a, maxx);
        }
        
        for (int i = 0; i < 100001; i++) {
            root[i] = i;
        }
        // for (int i = 0; i < 100001; i++) {
        //     if (size[i] != 0)
        //         System.out.println(i + "," + size[i]);
        //     //max = Math.max(max, size[i]);
        // }
        
        int[] primes = prime(maxx);
        int nn = primes.length;

        
        for (int p : primes) {
            for (int j = 2; p * j <= maxx; j++) {
                if (set.contains(p * j)) {
                    union(p * j, p);
                }
            }
        }
        
        
        // for (int a : A) {
        //     for (int i = 0; i < nn && primes[i] * 2 <= a; i++) {
        //         //List<Integer> list = new ArrayList<>();
        //         if (a % primes[i] == 0) {
        //             union(a, primes[i]);
        //             //list.add(i);
        //             //System.out.println(a + "," + i);
        //         }
        //         // for (int c = 0; c < list.size(); c++) {
        //         //     for (int d = c + 1 ; d < list.size(); d++) {
        //         //         union(list.get(c), list.get(d));
        //         //     }
        //         // }
        //     }
        // }
        
        int max = 0;
        for (int i = 0; i <= maxx; i++) {
            
            max = Math.max(max, size[i]);
        }
        return max;
    }
    
    
    
    private int[] prime(int max) {
        boolean[] primes = new boolean[max + 1];
        for (int i = 2; i <= max; i++) {
            for (int j = 2; j * i <= max; j++) {
                primes[i * j] = true;
            }
        }
        List<Integer> ps = new ArrayList<>();
        for (int i = 2; i <= max; i++) {
            if (!primes[i]) {
                ps.add(i);
            }
        }
        int[] pa = new int[ps.size()];
        for (int i = 0; i < ps.size(); i++) {
            pa[i] = ps.get(i);
        }
        return pa;
    }
}
