class Solution {
    void getDiv(Set<Integer> set, int n) {
        int sqrt = (int) Math.sqrt(n);
        for (int i = sqrt; i >= 2; i--) {
            if (n % i == 0) {
                getDiv(set, i);
                getDiv(set, n / i);
                return;
            }
        }
        set.add(n);
    }

    public static boolean[] generatePrime(int n) {
        boolean p[] = new boolean[n + 1];
        p[2] = true;

        for (int i = 3; i <= n; i += 2) {
            p[i] = true;
        }

        for (int i = 3; i <= Math.sqrt(n); i += 2) {
            if (!p[i]) {
                continue;
            }
            for (int j = i * i; j <= n; j += i << 1) {
                p[j] = false;
            }
        }
        return p;
    }

    public int largestComponentSize(int[] A) {
        boolean primeB[] = generatePrime(100000);
        List<Integer> prime = new ArrayList<>();
        Map<Integer, Integer> idx = new HashMap<>();
        for (int i = 2; i <= 100000; i++) {
            if (primeB[i]) {
                idx.put(i, prime.size());
                prime.add(i);
            }
        }
        int pn = prime.size();
        List<Integer> edges[] = new List[pn];
        for (int i = 0; i < pn; i++) {
            edges[i] = new ArrayList<>();
        }
        int cnt[] = new int[pn];
        int n = A.length;
        boolean has[] = new boolean[pn];
        for (int i = 0; i < n; i++) {
            if (A[i] == 1) {
                continue;
            }
            Set<Integer> set = new HashSet<>();
            getDiv(set, A[i]);
            List<Integer> list = new ArrayList<>(set);
            cnt[idx.get(list.get(0))]++;
            for (int j = 0; j < list.size(); j++) {
                int from = idx.get(list.get(j));
                has[from] = true;
                for (int k = j + 1; k < list.size(); k++) {
                    int to = idx.get(list.get(k));
                    edges[from].add(to);
                    edges[to].add(from);
                }
            }
        }
        int max = 0;
        boolean vis[] = new boolean[pn];
        for (int i = 0; i < pn; i++) {
            if (vis[i] || !has[i]) {
                continue;
            }
            int cur = 0;
            Queue<Integer> queue = new LinkedList<>();
            queue.offer(i);
            vis[i] = true;
            while (!queue.isEmpty()) {
                int x = queue.size();
                while (x-- > 0) {
                    int q = queue.poll();
                    cur += cnt[q];
                    List<Integer> es = edges[q];
                    for (int j = 0; j < es.size(); j++) {
                        int nxt = es.get(j);
                        if (vis[nxt]) {
                            continue;
                        }
                        vis[nxt] = true;
                        queue.offer(nxt);
                    }
                }
            }
            max = Math.max(max, cur);
        }

        return max;
    }
}
