class Solution {
    int[] fa=new int[100000];
    List<Integer> getPrime(int n) 
    { 
        boolean[] flag=new boolean[n+1];   
        List<Integer> ans=new ArrayList<>();;
        int cnt=0;    
        for (int i=2;i<=n;i++) 
        {
            if (!flag[i]) 
            {
                ans.add(i); 
                cnt++;
            }
            for (int j=0; j<cnt; ++j) 
            { 
                int now=ans.get(j);
                if (i*now>n) break;
                flag[i*now]=true;
                if (i%now==0) break;
            }
        }
        return ans;
    
}
    int find(int x)
    {
        if (fa[x]==x) return x;
        fa[x]=find(fa[x]);
        return fa[x];
    }
    public int largestComponentSize(int[] A) {
        List<Integer> pr=getPrime(100000);
        int n=pr.size();
        for (int i=0;i<100000;i++) fa[i]=i;
        int[] s=new int[A.length];
        for (int k=0;k<A.length;k++)
        {
            int x=A[k];
            int y=x;
            List<Integer> fac=new ArrayList<>();
            for (int i=0;i<n;i++)
            {
                int now=pr.get(i);
                if (now*now>x) break;
                if (y%now!=0) continue;
                fac.add(now);
                while (y%now==0) y/=now;
            }
            if (y>1) fac.add(y);
            int m=fac.size();
            if (m==0) continue;
            for (int i=0;i<m-1;i++)
            {
                int p=fac.get(i),q=fac.get(i+1),fap=find(p),faq=find(q);
                fa[faq]=fap;
            }
            s[k]=find(fac.get(0));
        }
        int[] count=new int[100000];
        for (int i=0;i<A.length;i++) count[find(s[i])]++;
        int ans=1;
        for (int i=0;i<100000;i++) ans=Math.max(ans,count[i]);
        return ans;
    }
}
