class Solution {
    int[] sz;
    int[] par;
    int n;
    final int M = 100001;
    int[] fac = new int[M];
    int maxSz = 0;
    
    public int largestComponentSize(int[] A) {
        for (int i = 1; i < M; i++) fac[i] = i;
        for (int i = 2; i < M / i; i++) {
            if (fac[i] == i) {
                for (int j = i * i; j < M; j += i)
                    fac[j] = i;
            }
        }
        
        n = A.length;
        sz = new int[M];
        par = new int[M];
        for (int i = 2; i < M; i++) {
            par[i] = i;
        }
        for (int v : A) {
            sz[v]++;
        }
        
        for (int i = 2; i < M; i++) {
            if (sz[i] == 0) continue;
            int q = i;
            int ii = i;
            while (ii > 1) {
                //System.out.println("i = " + ii + ", p = " + fac[ii]);
                int p = fac[ii];
                add(p, q);
                ii/= p;
            }
        
}
        return maxSz;
    }
    
    private int find(int p) {
        int r = p;
        while (par[r] != r) r = par[r];
        
        while (p != r) {
            int t = par[p];
            par[p] = r;
            p = t;
        }
        return r;
    
}
    
    private void add(int p, int q) {
        int rp = find(p);
        int rq = find(q);
        if (rp == rq) return;
        
        int st = sz[rp] + sz[rq];
        if (sz[rp] < sz[rq]) {
            par[rp] = rq;
            sz[rq] = st;
        }else {
            par[rq] = rp;
            sz[rp] = st;
        }
        
        if (maxSz < st) maxSz = st;
    }
    
}
