class Solution {
    public String largestTimeFromDigits(int[] A) {
        int mt = -1;
        String st = "";
        for (int a = 0; a < 4; a++) {
            for (int b = 0; b < 4; b++) {
                if (a==b)
                    continue;
                for (int c = 0; c < 4; c++) {
                    if (a==c||b==c)
                        continue;
                    for (int d = 0; d < 4; d++) {
                        if (a==d||b==d||c==d)
                            continue;
                        String time = ""+A[a]+A[b]+":"+A[c]+A[d];
                        int h = A[a]*10+A[b];
                        int m = A[c]*10+A[d];
                        if (h<24&&m<60) {
                            if (h*60+m>mt) {
                                st = time;
                                mt = h*60+m;
                            }
                        }
                    }
                }
            }
        }
        return st;
    }
}
