class Solution {
    public String largestTimeFromDigits(int[] A) {
        int max = -1;
        int[] nums = new int[4];
        String result = "";
        
for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                if (j == i) {
                    continue;
                }
                for (int k = 0; k < 4; k++) {
                    if (k == i || k == j) {
                        continue;
                    }
                    for (int l = 0; l < 4; l++) {
                        if (l == i || l == j || l == k) {
                            continue;
                        }
                        nums[0] = A[i];
                        nums[1] = A[j];
                        nums[2] = A[k];
                        nums[3] = A[l];
                        if (isValid(nums)) {
                            int val = (nums[0] * 10 + nums[1]) * 60 + (nums[2] * 10 + nums[3]);
                            if (val > max) {
                                max = val;
                                result = "" + nums[0] + nums[1] + ":" + nums[2] + nums[3];
                            }
                        
}
                    }
                }
            }
        }
        return result;
    }
    
    private boolean isValid(int[] nums) {
        int h = nums[0] * 10 + nums[1];
        int m = nums[2] * 10 + nums[3];
        return (h >= 0 && h <= 23) && (m >= 0 && m <= 59);
    }
}
