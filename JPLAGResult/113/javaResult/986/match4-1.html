class Solution {
    public String largestTimeFromDigits(int[] A) {
        Arrays.sort(A);
        if (A[0] > 2) {
            return "";
        }
        
int maxTime = -1;
        String res = "";        
        // try all combination
        for (int i = 0; i < 4; i++) {            
            for (int j = 0; j < 4; j++) {
                if (j == i) continue;
                for (int k = 0; k < 4; k++) {
                    if (k == i || k == j) continue;
                    int l = 6 - i - j - k;
                    if (A[i] > 2) continue;
                    if (A[i] == 2 && A[j] > 3) continue;
                    if (A[k] > 5) continue;
                    int cur = (A[i] * 10 + A[j]) * 60 + A[k] * 10 + A[l];
                    if (cur > maxTime) {
                        cur = maxTime;
                        res = "" + A[i] + A[j] + ":" + A[k] + A[l];                            
                    }
                
}
            }
        }
        return res;
    }
}
