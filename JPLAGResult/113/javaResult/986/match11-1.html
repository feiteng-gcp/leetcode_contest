class Solution {
	private String best = "";
	public String largestTimeFromDigits(int[] A) {
		DFS(A, 0);
		return best;

	}

	private void DFS(int[] a, int i) {
		if (i == 4) {
			int hrs = a[0] * 10 + a[1], mins = a[2] * 10 + a[3];
			if (hrs <=23 && mins <= 59) {
				String cand = "" + a[0] + a[1] + ":" + a[2] + a[3];
				if (best.equals("") || (best.compareTo(cand) < 0))
					best = cand;
			}
		} else {
			for (int j = i; j < 4; ++j) {
				swap(a, i, j);
				DFS(a, i + 1);
				swap(a, i, j);
			}
		}
	}


	private void swap(int[] a, int i, int j) {
		int tmp = a[i];
		a[i] = a[j];
		a[j] = tmp;
	}
}
