class Solution {
    public String largestTimeFromDigits(int[] A) {
        int[] map = new int[10];
        for (int i= 0; i < A.length; i++){
            map[A[i]]++;
        }
        for (int i = 1439; i >= 0; i--){
            int hour = i / 60;
            int min = i % 60;
            int[] count = new int[10];
            count[hour % 10]++;
            count[hour / 10]++;
            count[min % 10]++;
            count[min / 10]++;
            boolean flag = true;
            
for (int j = 0; j < 10; j++){
                if (count[j] != map[j]){
                    flag = false;
                    break;
                }
            }
            
            if (flag) {
                String h = (hour >= 10) ? Integer.toString(hour) : "0"+ Integer.toString(hour);
                String m = (min >= 10) ? Integer.toString(min) : "0" + Integer.toString(min);
                return h + ":" + m;
            }
                
        }
        return "";
    }
}
