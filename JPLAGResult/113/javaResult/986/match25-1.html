class Solution {
     String best="";
    void swap(int[] a, 
int x, int y) {
        int t=a[x];
        a[x]=a[y];
        a[y]=t;
    }
    void dfs(int []b, int n) {
        if (n ==4) {
            if (b[0]*10+b[1]<=23&&b[2]*10+b[3]<=59) 
{
                String temp=String.format("%d%d:%d%d", b[0],b[1],b[2], b[3]);
                if (best.isEmpty()||best.compareTo(temp)<0) {
                    best=temp;
                }
            }
            return;
        
}
        for (int j=n;j<4;++j) {
            swap(b, n, j);
            dfs(b, n+1);
            swap(b, n, j);
        }
    }
    public String largestTimeFromDigits(int[] A) {

        best="";
         dfs(A, 0);
         return best;
    }

}
