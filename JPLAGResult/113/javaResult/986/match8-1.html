class Solution {
    public String largestTimeFromDigits(int[] A) {
        int[] hash = new int[10];
        for(int i=0;i<4;i++) {
            hash[A[i]]++;
        }
        
        for(int i=0;i<10;i++) {
            // System.out.print(hash[i] + " ");
        }
        
        for(int i=23;i>=0;i--) {
            for(int j=59;j>=0;j--) {
                int h1 = i/10;
                int h2 = i%10;
                int m1 = j/10;
                int m2 =  j%10;
                hash[h1]--;hash[h2]--; hash[m1]--;hash[m2]--;
                if(sumZero(hash))
{
                    return String.valueOf(h1)+String.valueOf(h2)+":"+String.valueOf(m1)+String.valueOf(m2);
                }
                hash[h1]++;hash[h2]++; hash[m1]++;hash[m2]++;
            }
        }
        return "";
    }
    public boolean sumZero(int[] arr) {
        int sum = 0;
        for(int i=0;i<10;i++) {
            if(arr[i]<0) return false;
            sum+=arr[i];
        }
        
        return sum==0;
    }
}
