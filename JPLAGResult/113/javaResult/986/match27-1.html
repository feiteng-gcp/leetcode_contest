class Solution {
    public String largestTimeFromDigits(int[] A) {

		String ans = "
";

		for (int i = 0; i < 4; i++) {
			for (int j = 0; j < 4; j++) {
				for (int k = 0; k < 4; k++) {
					for (int l = 0; l < 4; l++) {
						if (i != j && i != k && i != l && j != k && j != l && k != l) {
							if (isValid(A, i, j, k, l)) {
								if ("".equals(ans)) {
									ans = A[i] + "" + A[j] + ":" + A[k] + "" + A[l];
								} else if (compare(ans, A, i, j, k, l) < 0) {
									ans = A[i] + "" + A[j] + ":" + A[k] + "" + A[l];
								}

							}
						}
					}
				}
			}
		}
		return ans;

	}

	private boolean isValid(int []A, int i, int j, int k, int l) {

		if (A[i] > 2) {
			return false;
		}

		if (A[i] == 2 && A[j] > 3) {
			return false;
		}

		if (A[k] >= 6) {
			return false;
		}

		return true;

	}

	private int compare(String time, int A[], int i, int j, int k, int l) {
		String x = A[i] + "" + A[j] + ":" + A[k] + "" + A[l];
		return time.compareTo(x);
	}
}
