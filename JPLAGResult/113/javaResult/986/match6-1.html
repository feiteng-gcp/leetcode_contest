class Solution {
    public String largestTimeFromDigits(int[] A) {
        int time = -1;
        String res = "";
        
for (int first = 0; first < A.length; first++) {
            if (A[first] > 2) continue;
            for (int second = 0; second < A.length; second++) {
                if (A[first] == 2 && A[second] > 3) continue;
                for (int third = 0; third < A.length; third++) {
                    if (A[third] > 5) continue; 
                    for (int fourth = 0; fourth < A.length; fourth++) {
                        
                        if (first == second || first == third || first == fourth || second == third || second == fourth || third == fourth) {
                            continue;
                        } else {
                            int hour = A[first] * 10 + A[second];
                            if (hour > 23) continue;
                            int min = A[third] * 10 + A[fourth];
                            if (min > 59) continue;
                            if (60 * hour + min > time) {
                                time = hour * 60 + min;
                                String hourStr = "";
                                if (hour < 10) {
                                    hourStr = "0" + String.valueOf(hour);
                                } else {
                                    hourStr = String.valueOf(hour);
                                }
                                String minStr = "";
                                if (min < 10) {
                                    minStr = "0" + String.valueOf(min);
                                } else {
                                    minStr = String.valueOf(min);
                                }
                                
res = hourStr + ":" + minStr;
                            }
                        }
                    }
                }
            }
        }
        if (time < 0) return "";
        else return res;
    }
}
