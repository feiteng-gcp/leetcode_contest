class Solution {
    
    String res = "";
    int max = -1;
    
    public String largestTimeFromDigits(int[] A) {
        // corner cases
        for (int num: A) {
            if (num < 0) {
                return "";
            }
        }
        dfs(A, 0);
        return res;
    }
    
    private void dfs(int[] A, int index) {
        if (index == A.length) {
            int mins = getMins(A);
            // System.out.println(Arrays.toString(A));
            if (mins > max) {
                max = mins;
                res = "" + A[0] + A[1] + ":" + A[2] + A[3];
            }
        } else {
            for (int i = index; i < A.length; i++) {
                swap(A, i, index);
                dfs(A, index + 1);
                swap(A, i, index);
            }
        }
        
    }
    
    private void swap(int[] A, int i, int j) {
        int tmp = A[i];
        A[i] = A[j];
        A[j] = tmp;
    }
    
    private int getMins(int[] A) {
        int hour = A[0] * 10 + A[1];
        int min = A[2]*10 + A[3];
        if (hour >= 24 || min >= 60) {
            return -1;
        }
        return hour*60 + min;
    }
    
    
}
