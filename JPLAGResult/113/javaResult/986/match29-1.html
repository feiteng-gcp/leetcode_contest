class Solution {
    public String largestTimeFromDigits(int[] A) {
        int max = -1;
        String h = "";
        String m = "";
        for (int i=0; i<4; i++) {
            for (int j=0; j<4; j++) {
                if (i != j) {
                    int hour = A[i] * 10 + A[j];
                    String shour = A[i] + "" + A[j];
                    if (hour>=0 && hour<24) {
                        for (int a=0; a<4; a++) {
                            for (int b=0; b<4; b++) {
                                if (a != b && a!=i && a!=j && b!=i && b!= j) {
                                    int minute = A[a] * 10 + A[b];
                                    String smin = A[a] + "" + A[b];
                                    if (minute>=0 && minute<60) {
                                        int time = hour * 60 + minute;
                                        if (time > max) {
                                            max = time;
                                            h = shour;
                                            m = smin;
                                        
}
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        
        
if (max==-1) {
            return "";
        }
        return h + ":" + m;
    }
}
