class Solution {
	    public String largestTimeFromDigits(int[] A) {
	    	int max = -1; int o=-1;int oo=-1;
	        for(int i=0;i<4;i++) {
	        	for(int j=0;j<4;j++) {
	        		if(i==j) continue;
	        		for(int p=0;p<4;p++) 
{
	        			if(p==j||p==i) continue;	        			
	        			for(int q=0;q<4;q++) {
	        				if(q==i||q==j||q==p) continue;
	        				int h = A[i]*10+A[j];
	        				int m = A[p]*10+A[q];
	        				if(h<24&&m<60) {
	        					if(h*60+m>max) {
	        						max = h*60+m;
	        						o=h;
	        						oo=m;
	        					}
	        				}

	        			}
	        		}
	        	}
	        }
	        if(max==-1) return "";
	        else {
		        if(o<10) {
		        	if(oo<10) return "0"+o+":"+"0"+oo;
		        	else return "0"+o+":"+oo;
		        }else {
		        	if(oo<10) return o+":"+"0"+oo;
		        	else return o+":"+oo;
		        }
	        }
	    }
}
