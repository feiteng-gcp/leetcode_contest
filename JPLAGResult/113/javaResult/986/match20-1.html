class Solution {
    public String largestTimeFromDigits(int[] A) {
        int maxTime = -1;
        String maxString = "";
        for (int i = 0; i < 4; i++) {
            if (A[i] > 2) continue;
            for (int j = 0; j < 4; j++) {
                if (j == i || A[i]*10 + A[j] > 23) continue;
                for (int k = 0; k < 4; k++) {
                    if (k == i || k == j) continue;
                    if (A[k] > 5) continue;
                    for (int l = 0; l < 4; l++) {
                        if (l == k || l == j || l == i) continue;
                        if (A[k]*10 + A[l] > 59) continue;
                        int myTime = (A[i]*10 + A[j])*60 + (A[k]*10 + A[l]);
                        if (myTime > maxTime) {
                            maxTime = myTime;
                            maxString = A[i] + "" + A[j] + ":" + A[k] + "" + A[l];
                        
}
                    }
                }
            }       
        }
        return maxString;
    }
}
