class Solution {
    int hour = -1, min = -1;
    
    public String largestTimeFromDigits(int[] A) {
        //hour: 00 01 -- 23
        //try all
        int[] buf = new int[4];
        bt(A, A.length, 0, buf, new boolean[4]);
        if (hour < 0) return "";
        
        return format(hour) + ":" + format(min);
    }
    
    private String format(int v) {
        if (v < 10) return "0" + v;
        return "" + v;
    }
    private void bt(int[] A, int n, int lo, int[] cand, boolean[] used) {
        if (lo == n) {
            int h = cand[0] * 10 + cand[1];
            int m = cand[2] * 10 + cand[3];
            if (h < 24 && m < 60) {
                if (h > hour || (h == hour && m > min)) {
                    hour = h;
                    min = m;
                }
            }
            //System.out.println(h + ":" + m);
            return;
        }
        
        for (int i = 0; i < n; i++) {
            if (!used[i]) {
                cand[lo] = A[i];
                used[i] = true;
                bt(A, n, lo + 1, cand, used);
                used[i] = false;
            }
        }
    }
}
