class Solution
{
    private String ans;

    private void dfs(int idx, int[] comp, int[] a, boolean[] used)
    {
        if (idx == 4)
        {
            int hr = comp[0] * 10 + comp[1];
            int min = comp[2] * 10 + comp[3];
            if (hr <= 23 && min <= 59)
            {
                String cur = String.format("%02d:%02d", hr, min);
                if (ans.compareTo(cur) < 0) ans = cur;
            }
            return;
        }
        for (int i = 0; i < 4; i++)
        {
            if (!used[i])
            {
                used[i] = true;
                comp[idx] = a[i];
                dfs(idx + 1, comp, a, used);
                used[i] = false;
            }
        }
    }

    public String largestTimeFromDigits(int[] a)
    {
        ans = "";
        dfs(0, new int[4], a, new boolean[4]);
        return ans;
    }
}
