class Solution {
    public String largestTimeFromDigits(int[] A) {
        PriorityQueue<String> pq = new PriorityQueue<>((o1, o2) -> (o2.compareTo(o1)));
        dfs(A, 0, pq);
        return pq.isEmpty() ? "" : pq.peek();
    
}
    
    private void dfs(int[] arr, int cur, PriorityQueue<String> pq) {
        if (cur == arr.length) {
            if (arr[0] > 2 || arr[2] > 5 || (arr[0] == 2 && arr[1] > 3)) return;
            pq.offer(arr[0] + "" + arr[1] + ":" + arr[2] + arr[3]);
            return;
        
}
        for (int i = cur; i < arr.length; i++) {
            swap(arr, cur, i);
            dfs(arr, cur + 1, pq);
            swap(arr, cur, i);
        }
    }
    
    private void swap(int[] arr, int i, int j) {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
}
