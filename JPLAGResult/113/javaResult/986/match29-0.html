class Solution {
    public String largestTimeFromDigits(int[] A) {
        
        int hour = -1;
        int minute = -1;
        String res = "";
        
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A.length; j++) {
                if (i != j) {
                    int h = A[i] * 10 + A[j];
                    if (0 <= h && h < 24 && h >= hour) {
                        int n = 0;
                        for (int k = 0; k < A.length; k++) {
                            if (k != i && k != j) {
                                n = n * 10 + A[k];
                            }
                        }
                        
                        int m = larger(hour, minute, h, n);
                        if (m != -1) {
                            hour = h;
                            minute = m;
                            if (0 <= hour && hour < 10) {
                                res = "0" + hour + ":";
                            } else {
                                res = hour + ":";
                            }
                            if (0 <= minute && minute < 10) {
                                res += "0" + minute;
                            } else {
                                res += minute;
                            }
                        }
                    }
                }
            }
        }
        
        return res;
    }
    
    private int larger(int hour, int minute, int h, int n) {
        int n1 = n;
        int n2 = n / 10 + n % 10 * 10;
        
        n = Math.max(n1, n2);
        
        if (n1 > 59) {
            n = n2;
        }
        
        if (n2 > 59) {
            n = n1;
        }
        
        if (n > 59) {
            return -1;
        }
        
        return n;
    }
}
