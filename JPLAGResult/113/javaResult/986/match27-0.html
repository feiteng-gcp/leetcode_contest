class Solution {
    public String largestTimeFromDigits(int[] A) {
        int largest = -1;
        String answer = "";
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                for (int k = 0; k < 4; k++) {
                    for (int m = 0; m < 4; m++) {
                        if (i != j && i != k && i != m && j != k && j != m && k != m) {
                            if (((A[i] * 10 + A[j]) <= 23) && ((A[k] * 10 + A[m]) <= 59)) {
                                int num = (A[i] * 10 + A[j]) * 60 + (A[k] * 10 + A[m]);
                                if (num > largest) {
                                    answer = "" + A[i] + A[j] + ":" + A[k] + A[m];
                                    largest = num;
                                }                          
                            }
                        }
                    }
                }
            }
        }
        return answer;
    }
}
