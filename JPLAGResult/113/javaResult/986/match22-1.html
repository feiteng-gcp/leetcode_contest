class Solution {
    int hour = -1;
    int minute = -1;
	public String largestTimeFromDigits(int[] A) {
		boolean[] used = new boolean[A.length];
		List<Integer> res = new ArrayList<>();
		dfs(A, used, 0, res);
		if (hour == -1 || minute == -1) return "";
		String h = Integer.toString(hour);
		if (h.length() == 1) h = "0" + h;
		String m = Integer.toString(minute);
		if (m.length() == 1) m = "0" + m;

		return h + ":" + m;
	}

	private void dfs(int[] a, boolean[] used, int index, List<Integer> res) {
		if (index == a.length) {
			int h = 0;
			int m = 0;
			for (int j = 0; j < 2; j++) h = h * 10 + res.get(j);
			for (int j = 2; j < 4; j++) m = m * 10 + res.get(j);
			if (h >= 0 && h < 24 
&& m >= 0 && m < 60) {
				if (h > hour || (h == hour && m > minute)) {
					hour = h;
					minute = m;
				}
			}
			return;
		}
		for (int i = 0; i < a.length; i++) {
			if (!used[i]) {
				res.add(a[i]);
				used[i] = true;
				dfs(a, used, index+1, res);
				res.remove(res.size() - 1);

				used[i] = false;
			}
		}
	}
}
