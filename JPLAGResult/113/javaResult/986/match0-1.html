class Solution {
    public String largestTimeFromDigits(int[] A) {
        String res = "";
        
int rh=0, rm=0;
        for(int i=0; i<A.length; i++){
            for(int j=0; j<A.length; j++){
                if(j==i)continue;
                int ih = A[i]*10+A[j];
                if(ih>23) continue;
                String hour = A[i]+""+A[j];   
                
for(int m=0; m<A.length; m++){
                    if(j==m || i==m) continue;
                    for(int n=0; n<A.length; n++){
                        if(i==n || j==n || m==n) continue;
                        int im=A[m]*10+A[n];
                        if(im>59) continue;
                        String minue = A[m]+""+A[n];
                        if(ih>rh||ih==rh&&im>=rm){
                            res = hour+":"+minue;
                            rh = ih;
                            rm = im;
                        
}   
                    }
                }
            }
        }
        return res;
    }
}
