class Solution {
    public String largestTimeFromDigits(int[] A) {
        int max = -1;
        String ans = "";
        boolean[] used = new boolean[4];
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                if (i == j) {
                    continue;
                }
                for (int k = 0; k < 4; k++) {
                    if (i == k || j == k) {
                        continue;
                    }
                    for (int l = 0; l < 4; l++) {
                        if (i == l || j == l || k == l ) {
                            continue;
                        }
                        int a = A[i];
                        int b = A[j];
                        int c = A[k];
                        int d = A[l];
                        if (a * 10 + b >= 24) {
                            continue;
                        }
                        if (c * 10 + d >= 60) {
                            continue;
                        }
                        if ((a * 10 + b) * 60 + c * 10 + d > max) {
                            ans = "" + a + b + ":" + c + d;
                            max = (a * 10 + b) * 60 + c * 10 + d;
                        }
                    }
                }
            }
        }
        return ans;
    }
}
