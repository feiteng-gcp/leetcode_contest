class Solution {
    public String largestTimeFromDigits(int[] A) {
        int maxHour = -1;
        int maxMin = -1;

        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                if (j == i) {
                    continue;
                }
                for (int m = 0; m < 4; m++) {
                    if (m == i || m == j) {
                        continue;
                    }
                    for (int n = 0; n < 4; n++) {
                        if (n == i || n == j || n == m) {
                            continue;
                        }
                        int x = A[i] * 10 + A[j];
                        int y = A[m] * 10 + A[n];
                        if (valid(x, y)) {
                            if (x > maxHour) {
                                maxHour = x;
                                maxMin = y;
                            } else if (x == maxHour && y > maxMin) {
                                maxMin = y;
                            
}
                        }
                    }
                }
            }
        }

        if (maxHour == -1) {
            return "";
        }

        String ans = "";
        if (maxHour > 9) {
            ans += maxHour + ":";
        } else {
            ans += "0" + maxHour + ":";
        }

        if (maxMin > 9) {
            ans += maxMin + "";
        } else {
            ans += "0" + maxMin;
        }

        return ans;
    }
    
    private boolean valid(int x, int y) {
        if (x > 23) {
            return false;
        }
        if (y > 59) {
            return false;
        }
        return true;
    }
}
