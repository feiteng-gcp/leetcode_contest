class Solution {
    public String largestTimeFromDigits(int[] A) {
        String[] res = new String[] { "" };
        helper(A, 0, res);
        return res[0];
    }
    
    private void helper(int[] A, int index, String[] res) {
        if (index == 4) {
            if (valid(A)) {
                String cur = toString(A);
                if (res[0].equals("") || cur.compareTo(res[0]) > 0) {
                    res[0] = cur;
                }
            }
        }
        for (int i = index; i < 4; i++) {
            swap(A, index, i);
            helper(A, index + 1, res);
            swap(A, index, i);
        }
    }
    
    private void swap(int[] A, int a, int b) {
        int tmp = A[a];
        A[a] = A[b];
        A[b] = tmp;
    }
    
    private boolean valid(int[] A) {
        if (A[0] > 2) {
            return false;
        }
        if (A[0] == 2 && A[1] >= 4) {
            return false;
        }
        if (A[2] >= 6) {
            return false;
        }
        return true;
    }
    
    private String toString(int[] A) {
        return "" + A[0] + A[1] + ":" + A[2] + A[3];
    }
}
