class Solution {
    public String largestTimeFromDigits(int[] A) {
        String res = null;
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                if (i == j) continue;
                for (int k = 0; k < 4; k++) {
                    if (i == k || j == k) continue;
                    for (int l = 0; l < 4; l++) {
                        if (i == l || j == l || k == l) continue;
                        String time = A[i] + "" + A[j] + "" + A[k] + "" + A[l] + "";
                        if (isValid(time)) {
                            if (res == null) res = time;
                            else {
                                if (getDiff(time, res) > 0) res = time;
                            }
                        }
                    }
                }
            }
        }
        return res == null ? "" : res.substring(0, 2) + ":" + res.substring(2);
    }
    
    public boolean isValid(String s) {
        String hour = s.substring(0, 2);
        String min = s.substring(2);
        if (Integer.valueOf(hour) > 23) return false;
        if (Integer.valueOf(min) > 59) return false;
        return true;
    }
    
    public int getDiff(String a, String b) {
        int h1 = Integer.valueOf(a.substring(0, 2));
        int h2 = Integer.valueOf(b.substring(0, 2));
        int m1 = Integer.valueOf(a.substring(2));
        int m2 = Integer.valueOf(b.substring(2));
        return (h1 - h2) * 60 + m1 - m2;
    }
}
