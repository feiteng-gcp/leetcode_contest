class Solution {
    public String largestTimeFromDigits(int[] A) {
        int[] count = new int[10];
        for (int num : A) {
            count[num]++;
        }
        for (int i = 60 * 24 - 1; i >= 0; i--) {
            int hour = i / 60;
            int min = i % 60;
            int[] map = new int[10];
            int d1 = hour / 10;
            int d2 = hour % 10;
            int d3 = min / 10;
            int d4 = min % 10;
            map[d1]++;
            map[d2]++;
            map[d3]++;
            map[d4]++;
            boolean flag = true;
            for (int j = 0; j < 10; j++) {
                if (map[j] != count[j]) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                return String.valueOf(d1) + String.valueOf(d2) + ":" + String.valueOf(d3) + String.valueOf(d4);
            }
        }
        return "";
    }
}
