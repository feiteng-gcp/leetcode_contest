class Solution {
    int maxHour = Integer.MIN_VALUE;
    int maxMin = Integer.MIN_VALUE;
    public String largestTimeFromDigits(int[] A) {
        dfs(A, 0);
        if (maxHour == Integer.MIN_VALUE || maxMin == Integer.MIN_VALUE) {
            return "";
        }
        String result = ((maxHour < 10) ? ("0" + maxHour) : maxHour) + ":" + (maxMin < 10 ? ("0" + maxMin) : maxMin);
        return result;
    }
    //
    private void dfs(int[] A, int index) {
        if (index == A.length) {
            // System.out.println(A[0] + " " + A[1] + " " + A[2] + " " + A[3]);
            int hour = A[0] * 10 + A[1];
            int min = A[2] * 10 + A[3];
            if (hour < 24 && min < 60 && (hour > maxHour || hour == maxHour && min > maxMin)) {
                maxHour = hour;
                maxMin = min;
            }
        }
        for (int i = index; i < A.length; i++) {
            swap(i, index, A);
            dfs(A, index + 1);
            swap(i, index, A);
        }
    }
    
    private void swap(int i, int j, int[] A) {
        int temp = A[i];
        A[i] = A[j];
        A[j] = temp;
    }
}
