class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        List<Integer> list = new LinkedList();
        for(int i = deck.length - 1; i >= 0; i--) {
            int card = deck[i];
            if(list.size() < 2) {
                list.add(0, card);
            } else {
                int last = list.remove(list.size() - 1);
                list.add(0, last);
                list.add(0, card);
            }
        }
        int[] res = new int[deck.length];
        for(int i = 0; i < res.length; i++) {
            res[i] = list.get(i);
        }
        return res;
    }
}
