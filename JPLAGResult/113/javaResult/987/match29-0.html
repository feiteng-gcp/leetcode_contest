class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        int len = deck.length;
        int[] res = new int[len];
        int[] order = new int[len];
        int index = 0;
        Deque<Integer> deque = new ArrayDeque<>();
        for (int i = 0; i < len; i++) {
            deque.offer(i);
        }
        while (!deque.isEmpty()) {
            order[index++] = deque.pollFirst();
            if (!deque.isEmpty()) {
                deque.offerLast(deque.pollFirst());
            }  
        }
        
        for (int i = 0; i < len; i++) {
            res[order[i]] = deck[i];
        }
        return res;
    }
}
