class Solution
{
    public int[] deckRevealedIncreasing(int[] deck)
    {
        Queue<Integer> ll = new LinkedList<>();
        for (int i = 0; i < deck.length; i++) ll.add(i);
        List<Integer> takeOrder = new ArrayList<>();
        while (!ll.isEmpty())
        {
            takeOrder.add(ll.poll());
            if (!ll.isEmpty())
            {
                ll.add(ll.poll());
            }
        }
        Arrays.sort(deck);
        int[] ans = new int[deck.length];
        for (int i = 0; i < ans.length; i++)
        {
            int to = takeOrder.get(i);
            ans[to] = deck[i];
        }
        return ans;
    }
}
