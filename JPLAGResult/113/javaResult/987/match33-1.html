class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        int n = deck.length;
        
int[] res = new int[n];
        Queue<Integer> que = new ArrayDeque();
        for (int i = 0; i < n; i++) que.offer(i);
        int idx = 0;
        while (!que.isEmpty()) {
            int cur = que.poll();
            res[cur] = deck[idx++];
            if (!que.isEmpty()) que.offer(que.poll());
        }
        return res;
    }
}
