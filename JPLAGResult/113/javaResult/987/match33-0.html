import java.util.*;
import java.util.stream.*;
import java.math.*;

public class Solution {

    public int[] deckRevealedIncreasing(int[] deck) {
        int n = deck.length;
        Arrays.sort(deck);
        int[] ans = new int[n];
        Queue<Integer> queue = new ArrayDeque<>();
        for (int i = 0; i < n; i++) queue.add(i);

        int ptr = 0;
        while (!queue.isEmpty()) {
            int head = queue.poll();
            ans[head] = deck[ptr++];
            if (!queue.isEmpty()) {
                queue.add(queue.poll());
            }
        }
        return ans;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(Arrays.toString(sol.deckRevealedIncreasing(new int[]{17, 13, 11, 2, 3, 5, 7})));
        System.out.println(Arrays.toString(sol.deckRevealedIncreasing(new int[]{1})));
        System.out.println(Arrays.toString(sol.deckRevealedIncreasing(new int[]{1, 2})));
    }
}
