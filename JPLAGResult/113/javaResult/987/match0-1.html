class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        int n = deck.length;
        LinkedList<Integer> ll = new LinkedList<>();
        ll.addFirst(deck[n-1]);
        for (int i=n-2; i>=0; i--) {
            int tail = 
ll.removeLast();
            ll.addFirst(tail);
            ll.addFirst(deck[i]);
        }
        int[] res = new int[n];
        for (int i=0; i<n; i++) {
            res[i] = ll.removeFirst();
        }
        return res;
    }
}
