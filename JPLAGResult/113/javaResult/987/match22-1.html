class Solution {
	public int[] deckRevealedIncreasing(int[] deck) {
		int n = deck.length;
		Arrays.sort(deck);
		Deque<Integer> deque = new ArrayDeque<>();
		deque.addFirst
(deck[n - 1]);
		for (int i = n - 2; i >= 0; --i) {
			deque.addFirst(deque.pollLast());
			deque.addFirst(deck[i]);
		}
		int[] ans = new int[n];
		for (int i = 0
; i < n; ++i) {
			ans[i] = deque.pollFirst();
		}
		return ans;
	}
}
