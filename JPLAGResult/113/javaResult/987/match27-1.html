class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(deck[deck.length - 1]);
        for (int i = deck.length - 2; i >= 0; i--) {
            int temp = 
queue.poll();
            queue.offer(temp);
            queue.offer(deck[i]);
        }
        for (int i = deck.length - 1; i >= 0; i--) {
            deck[i] = queue.poll();
        }
        return deck;
    }
}
