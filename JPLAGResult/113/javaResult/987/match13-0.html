class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        int n = deck.length;
        Deque<Integer> deque = new LinkedList<>();
        for (int i = 0; i < n; i++) {
            deque.addLast(i);
        }
        List<Integer> order = new ArrayList<>();
        while (!deque.isEmpty()) {
            order.add(deque.pollFirst());
            if (!deque.isEmpty()) {
                deque.addLast(deque.pollFirst());
            }
        }
        Arrays.sort(deck);
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            int idx = order.get(i);
            ans[idx] = deck[i];
        }
        return ans;
    }
}
