class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        int n = deck.length;
        Deque<Integer> newDeck = new ArrayDeque<>();
        newDeck.offerFirst(deck[n-1]);
        for (int i = n - 2; i >= 0; --i) {
            newDeck.offerFirst(newDeck.pollLast());
            newDeck.offerFirst(deck[i]);
        }
        int[] ret = new int[n];
        for (int i = 0; i < n; ++i) {
            ret[i] = newDeck.pollFirst();
        }
        return ret;
    }
}
