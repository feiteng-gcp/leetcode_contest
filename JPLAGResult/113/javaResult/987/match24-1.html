class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        Queue<Integer> queue = new LinkedList();
        for (int i = 0; i < deck.length; i++)
            queue.add(i);

        int res[] = new int[deck.length];
        int j = 0;
        while(!queue.isEmpty()) {
            int cur = queue.remove();
            res[cur] = deck[j++];

            
if (!queue.isEmpty()) {
                queue.add(queue.remove());
            }
        }

        return res;
    }
}
