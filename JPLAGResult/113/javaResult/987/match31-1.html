class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        int n = deck.length;
        
        List<Integer> list = new ArrayList<>();
        
        for(int i = n - 1; i >= 0; i--) {
            int cur = deck[i];
            
move(list);
            
            list.add(0, cur);
        }
        
        int[] res = new int[n];
        for(int i = 0; i < n; i++) {
            res[i] = list.get(i);
        }
        return res;
    }
    
    private void move(List<Integer> list) {
        if(list.size() == 0) {
            return;
        }
        
        int last = list.remove(list.size() - 1);
        list.add(0, last);
    }
}
