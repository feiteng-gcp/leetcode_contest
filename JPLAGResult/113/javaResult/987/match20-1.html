class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        if(deck == null || deck.length == 0) {
            return new int[0];
        }
        int n = deck.length;
        int[] res = new int[n];
        Deque<Integer> deq = new ArrayDeque<>();
        Arrays.sort(deck);
        
        
for(int i = n - 1; i >= 0; i--) {
            if(!deq.isEmpty()) {
                deq.offerFirst(deq.pollLast());
            }
            deq.offerFirst(deck[i]);
        }
        
        
for(int i = 0; i < n; i++) {
            res[i] = deq.pollFirst();
        }
        return res;
        
    }
}
