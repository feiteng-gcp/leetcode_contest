class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);
        int len = deck.length;
        if (len <= 2) {
            return deck;
        }        
        List<Integer> list = new ArrayList<>();
        list.add(deck[len - 2]);
        list.add(deck[len - 1]);
        for (int i = len - 3; i >= 0; i--) {
            list.add(0, list.remove(list.size() - 1));
            list.add(0, deck[i]);
        }
        
        int[] answer = new int[len];
        for (int i = 0; i < len; i++) {
            answer[i] = list.remove(0);
        }
        return answer;
    }
}
