class Solution {
    public int[] deckRevealedIncreasing(int[] d) {
        int n = d.length;   
        Arrays.sort(d);
        if (n < 3) return d;
        
        Deque<Integer> dq = new ArrayDeque<>();
        dq.addLast(d[n-2]);
        dq.addLast(d[n-1]);
        for (int i = n - 3; i >= 0; i--) {
            int v = dq.removeLast();
            dq.addFirst(v);
            dq.addFirst(d[i]);
        }
        int[] r = new int[n];
        for (int i = 0; i < n; i++) r[i] = dq.removeFirst();
        return r;
    }
}
