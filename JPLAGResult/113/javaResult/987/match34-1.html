class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        int[] ans = new int[deck.length];
        
LinkedList<Integer> seq = new LinkedList<>();
        LinkedList<Integer> curr = new LinkedList<>();
        for (int i = 0; i < deck.length; ++i){
            curr.add(i);
        }
        while (!curr.isEmpty()){
            seq.add(
curr.pollFirst());
            if (!curr.isEmpty()){
                int i = curr.pollFirst();
                curr.add(i);
            }
        }
        int pos = 0;
        Arrays.sort(deck);
        for (Integer num : seq){
            ans[num] = deck[pos];
            ++pos;
        }
        return ans;
    }
}
