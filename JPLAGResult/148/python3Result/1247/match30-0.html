class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        ans = sum(nums)
        for b in range(2):
            res = 0
            for i in range(b, len(nums), 2):
                k = 0
                if i > 0:
                    k = max(k, nums[i] - nums[i-1] + 1)
                if i < len(nums)-1:
                    k = max(k, nums[i] - nums[i+1] + 1)
                res += k
            ans = min(ans, res)
        return ans
