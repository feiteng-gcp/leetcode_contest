class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        ret=0
        numsc=nums[:]
        for i in range(len(nums)-1):
            if i%2==0:
                if nums[i]<=nums[i+1]:
                    ret+=(nums[i+1]-nums[i]+1)
                    nums[i+1]=nums[i]-1
            else:
                if nums[i]>=nums[i+1]:
                    ret+=(nums[i]-nums[i+1]+1)
                    nums[i]=nums[i+1]-1
        nums=numsc
        retc=ret
        ret=0
        for i in range(len(nums)-1):
            if i%2==1:
                if nums[i]<=nums[i+1]:
                    ret+=(nums[i+1]-nums[i]+1)
                    nums[i+1]=nums[i]-1
            else:
                if nums[i]>=nums[i+1]:
                    ret+=(nums[i]-nums[i+1]+1)
                    nums[i]=nums[i+1]-1
        return min(ret,retc)
