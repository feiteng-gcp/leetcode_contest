class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        
        if len(nums) <= 1:
            return 0
            
        ans1 = 0
        for i in range(0, len(nums), 2):
            value = None
            if i > 0:
                value = nums[i-1]
            if i < len(nums)-1:
                if value:
                    value = min(value, nums[i+1])
                else:
                    value = nums[i
+1]
            if value <= nums[i]:
                ans1 += (nums[i] - value + 1)
                
        ans2 = 0
        for i in range(1, len(nums), 2):
            value = nums[i-1]
            
if i < len(nums)-1:
                value = min(value, nums[i+1])
            if value <= nums[i]:
                ans2 += (nums[i] - value + 1)
        #print(ans1, ans2)
        return min(ans1, ans2)
