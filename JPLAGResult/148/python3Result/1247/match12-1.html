class Solution(object):
    def movesToMakeZigzag(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        ori = nums.copy()
        ans = 0
        for i in range(1, len(nums)):
            if i % 2 == 1:
                
if nums[i] <= nums[i-1]:
                    ans += (nums[i-1] - nums[i] + 1)
                    nums[i-1] = nums[i] - 1
            else:
                if nums[i] >= nums[i-1]:
                    ans += (nums[i] - nums[i-1] + 1)
                    nums[i] = nums[i-1] - 1
        tmp = 0
        nums = ori
        for i in range(1, len(nums)):
            if i % 2 == 0:
                
if nums[i] <= nums[i-1]:
                    tmp += (nums[i-1] - nums[i] + 1)
                    nums[i-1] = nums[i] + 1
            else:
                if nums[i] >= nums[i-1]:
                    tmp += (nums[i] - nums[i-1] + 1)
                    nums[i] = nums[i-1] - 1

        return min(tmp, ans)
