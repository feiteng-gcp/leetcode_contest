class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        nums_copy = nums[:]
        size = len(nums)
        
        moves_a = 0
        for idx in range(size-1):
            if idx % 2 == 0:
                
if nums[idx] <= nums[idx+1]:
                    moves_a += nums[idx+1] - nums[idx] + 1
                    nums[idx+1] = nums[idx] - 1
            else:
                if nums[idx] >= nums[idx+1]:
                    moves_a += nums[idx] - nums[idx+1] + 1
                    nums[idx] = nums[idx+1] - 1
                    
        nums = nums_copy[:]
        moves_b = 0
        for idx in range(size-1):
            if idx % 2 == 0:
                
if nums[idx] >= nums[idx+1]:
                    moves_b += nums[idx] - nums[idx+1] + 1
                    nums[idx] = nums[idx+1] - 1
            else:
                if nums[idx] <= nums[idx+1]:
                    moves_b += nums[idx+1] - nums[idx] + 1
                    nums[idx+1] = nums[idx] - 1
                    
        return min(moves_a, moves_b)
