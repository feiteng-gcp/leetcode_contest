class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        if len(nums) <= 2:
            return 0
        tmp1 = 0
        tmp2 = 0
        i = 1
        while i < len(nums):
            if i+1 < len(nums) and nums[i] >= min(nums[i-1],nums[i+1]):
                tmp1 += nums[i]-min(nums[i-1],nums[i+1])+1
            if i+1 >= len(nums) and nums[i] >= nums[i-1]:
                tmp1 += nums[i] - nums[i-1] + 1
            i += 2
        i = 0
        while i < len(nums):
            if i+1 < len(nums) and i >= 1 and nums[i] >= min(nums[i-1],nums[i+1]):
                tmp2 += nums[i] - min(nums[i-1],nums[i+1]) +1
            elif i+1 >= len(nums) and nums[i] >= nums[i-1]:
                tmp2 += nums[i] - nums[i-1] + 1
            elif i == 0 and nums[1] <= nums[0]:
                tmp2 += nums[0]-nums[1]+1
            i += 2
        return min(tmp1,tmp2)
