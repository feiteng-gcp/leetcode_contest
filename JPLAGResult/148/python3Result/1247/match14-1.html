class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        cnt 
= 0
        ns = nums.copy()
        for i in range(0, len(ns), 2):
            if i > 0 and ns[i-1] >= nums[i]:
                cnt += ns[i-1] - nums[i] + 1
                ns[i-1] = nums[i] - 1
            if i < len(ns) - 1 and ns[i+1] >= nums[i]:
                cnt += ns[i+1] - nums[i] + 1
                ns[i+1] = nums[i] - 1
        ans = cnt
        
        cnt = 0
        ns = nums.copy()
        for i in range(1, len(ns), 2):
            if i > 0 and ns[i-1] >= nums[i]:
                cnt += ns[i-1] - nums[i] + 1
                ns[i-1] = nums[i] - 1
            if i < len(ns) - 1 and ns[i+1] >= nums[i]:
                cnt += ns[i+1] - nums[i] + 1
                ns[i+1] = nums[i] - 1
        ans = min(ans, cnt)
        return ans
