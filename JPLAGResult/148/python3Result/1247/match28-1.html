class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        orig_num = list(nums)
        # odd 
        n = len(nums)
        odd_move = 0
        for i in range(n):
            if not i%2:
                if i+1 < n and nums[i+1] >= nums[i]:
                    odd_move += nums[i+1] - nums
[i] + 1 
                    nums[i+1] -= nums[i+1] - nums[i] + 1
                if i-1 >=0 and nums[i-1] >= nums[i] :
                    odd_move += nums[i-1] - nums[i] + 1 
                    nums[i-1] -= nums[i-1] - nums[i] + 1
        # print("nums = ", nums , " odd_move = ", odd_move)
        # even
        nums = orig_num
        even_move = 0
        for i in range(n):
            if i%2:
                if i+1 < n and nums[i+1] >= nums[i]:
                    even_move += nums[i+1] - nums
[i] + 1 
                    nums[i+1] -= nums[i+1] - nums[i] + 1
                if i-1 >=0 and nums[i-1] >= nums[i] :
                    even_move += nums[i-1] - nums[i] + 1 
                    nums[i-1] -= nums[i-1] - nums[i] + 1
        # print("nums = ", nums , " even_move = ", even_move)

        return even_move if even_move < odd_move else odd_move
