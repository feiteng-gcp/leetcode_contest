class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        def zigzag(start = 0):
            total = 0
            for i in range(start, len(nums), 2):
                n1 = nums[i - 1] if i > 0 else float('inf')
                n2 = nums[i + 1] if i < len(nums) - 1 else float('inf')
                total += max(nums[i] - min(n1, n2) + 1, 0)
            return total
        return min(zigzag(0), zigzag(1))
