class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        # even
        even = 0
        for i in range(len(nums)):
            m = 0
            if i % 2 == 1:
                if i - 1 >= 0:
                    m = max(m, nums[i] - nums[i-1] + 1)
                if i + 1 < len(nums):
                    m = max(m, nums[i] - nums[i+1] + 1)
            even += m
        # odd
        odd = 0
        for i in range(len(nums)):
            m = 0
            if i % 2 == 0:
                if i - 1 >= 0:
                    m = max(m, nums[i] - nums[i-1] + 1)
                if i + 1 < len(nums):
                    m = max(m, nums[i] - nums[i+1] + 1)
            odd += m
        return min(even, odd)
