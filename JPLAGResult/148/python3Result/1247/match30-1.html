from typing import *
import json


class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        ans = 1 << 29

        a = nums.copy
()

        cnt = 0
        for i in range(0, len(a), 2):
            tmp = 0
            if i - 1 >= 0:
                tmp = max(tmp, a[i] - a[i - 1] + 1)
            if i + 1 < len(a):
                tmp = max(tmp, a[i] - a[i + 1] + 1)
            cnt += tmp
        ans = min(ans, cnt)

        cnt = 0
        for i in range(1, len(a), 2):
            tmp = 0
            if i - 1 >= 0:
                tmp = max(tmp, a[i] - a[i - 1] + 1)
            if i + 1 < len(a):
                tmp = max(tmp, a[i] - a[i + 1] + 1)
            cnt += tmp
        ans = min(ans, cnt)

        return ans
