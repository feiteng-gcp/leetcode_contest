class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        if len(nums) <= 2: return 0
        ans = 0
        n = len(nums)
        cnt = 0
        nums_ = nums.copy()
        for i in range(1, n, 2):
            if nums[i] <= nums[i-1]:
                cnt += (nums[i-1] - nums[i] + 1)
            if i + 1 < n and nums[i] <= nums[i+1]:
                cnt += (nums[i+1] - nums[i] + 1)
                nums[i+1] = nums[i] - 1
        ans = cnt
        cnt = 0
        nums = nums_
        for i in range(1, n, 2):
            if nums[i] >= nums[i-1]:
                cnt += (nums[i] - nums[i-1] + 1)
                nums[i] = nums[i-1] - 1
            if i + 1 < n and nums[i] >= nums[i+1]:
                cnt += (nums[i] - nums[i+1] + 1)
                nums[i] = nums[i+1] - 1
        ans = min(ans, cnt)
        return ans
