class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        def f1(nums):
            A = [x for x in nums]
            ans, n = 0, len(A)
            
            for i in range(1, n, 2):
                
if A[i] <= A[i-1]:
                    ans += A[i-1] - A[i] + 1 
                    A[i-1] = A[i] - 1
                    
                if i+1 < n and A[i] <= A[i+1]:
                    ans += A[i+1] - A[i] + 1
                    A[i+1] = A[i] - 1
                    
            return ans
        
        def f2(nums):
            A = [x for x in nums]
            ans, n = 0, len(A)
            
            for i in range(0, n, 2):
                
if i != 0 and A[i] <= A[i-1]:
                    ans += A[i-1] - A[i] + 1 
                    A[i-1] = A[i] - 1
                    
                if i+1 < n and A[i] <= A[i+1]:
                    ans += A[i+1] - A[i] + 1
                    A[i+1] = A[i] - 1
                    
            return ans
                    
        #print(f1(nums), f2(nums))
        return min(f1(nums), f2(nums))
