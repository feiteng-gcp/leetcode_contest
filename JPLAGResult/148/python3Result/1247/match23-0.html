class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        nums_original = list(nums)
        even_moves = 0
        i = 0
        while i < len(nums):
            if i > 0:
                if nums[i] >= nums[i-1]:
                    even_moves += nums[i]-nums[i-1]+1
                    nums[i] = nums[i-1]-1
            if i < len(nums)-1:
                if nums[i] >= nums[i+1]:
                    even_moves += nums[i]-nums[i+1]+1
                    nums[i] = nums[i+1]-1
            i += 2
            
        nums = nums_original
        odd_moves = 0
        i = 1
        while i < len(nums):
            if i > 0:
                if nums[i] >= nums[i-1]:
                    odd_moves += nums[i]-nums[i-1]+1
                    nums[i] = nums[i-1]-1
            if i < len(nums)-1:
                if nums[i] >= nums[i+1]:
                    odd_moves += nums[i]-nums[i+1]+1
                    nums[i] = nums[i+1]-1
            i += 2
            
        return min(even_moves, odd_moves)
