class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        a = 0
        b = 0
        nums1 = nums.copy()
        for i in range(0,len(nums),2):
            if i-1 >= 0 and nums1[i-1] >= nums1[i]:
                a += nums1[i-1] - nums1[i] + 1
                nums1[i-1] = nums1[i]-1
            if i+1 < len(nums) and nums1[i+1] >= nums1[i]:
                a += nums1[i+1] - nums1[i] + 1
                nums1[i+1] = nums1[i] - 1
        b = a
        a = 0
        
        nums1 = nums.copy()
        for i in range(1,len(nums),2):
            if i-1 >= 0 and nums1[i-1] >= nums1[i]:
                a += nums1[i-1] - nums1[i] + 1
                nums1[i-1] = nums1[i]-1
            if i+1 < len(nums) and nums1[i+1] >= nums1[i]:
                a += nums1[i+1] - nums1[i] + 1
                nums1[i+1] = nums1[i] - 1
        return min(a,b)
