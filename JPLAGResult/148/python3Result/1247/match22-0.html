class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        a = 0
        numsb = nums[::]
        b = 0
        for i in range(len(nums)-1):
            if i % 2 == 0:
                if nums[i] <= nums[i+1]:
                    a += (nums[i+1] - nums[i] + 1)
                    nums[i+1] = nums[i] - 1
            else:
                if nums[i] >= nums[i+1]:
                    a += (nums[i] - nums[i+1] + 1)
                    nums[i] = nums[i+1] - 1
        for i in range(len(nums)-1):
            if i % 2 == 1:
                if numsb[i] <= numsb[i+1]:
                    b += (numsb[i+1] - numsb[i] + 1)
                    numsb[i+1] = numsb[i] - 1
            else:
                if numsb[i] >= numsb[i+1]:
                    b += (numsb[i] - numsb[i+1] + 1)
                    numsb[i] = numsb[i+1] - 1
        return min([a,b])
