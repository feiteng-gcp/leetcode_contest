class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        res = float('inf')
        N = len(nums)
        if N < 2:
            return 0
        
        count = 0
        for i in range(0, N, 2):
            cap = float('inf')
            if i - 1 >= 0:
                cap = min(cap, nums[i - 1])
            if i + 1 < N:
                cap = min(cap, nums[i + 1])
            count += nums[i] - (cap - 1) if cap <= nums[i] else 0
        
        res = min(res, count)
        
        count = 0
        for i in range(1, N, 2):
            cap = float('inf')
            if i - 1 >= 0:
                cap = min(cap, nums[i - 1])
            if i + 1 < N:
                cap = min(cap, nums[i + 1])
            count += nums[i] - (cap - 1) if cap <= nums[i] else 0
        
        res = min(res, count)
        
        return res
