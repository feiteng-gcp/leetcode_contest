class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        a, b = nums[:], nums[:]
        n = len(nums)
        aa = 0
        for i in range(1, n):
            if i % 2:
                if a[i] >= a[i - 1]:
                    aa += a[i] - a[i - 1] + 1
                    a[i] = a[i - 1] - 1
            else:
                if a[i] <= a[i - 1]:
                    aa += a[i - 1] + 1 - a[i]
                    a[i - 1] = a[i] - 1 
                    
        bb = 0
        for i in range(1, n):
            if i % 2 == 0:
                if b[i] >= b[i - 1]:
                    bb += b[i] - b[i - 1] + 1
                    b[i] = b[i - 1] - 1
            else:
                if b[i] <= b[i - 1]:
                    bb += b[i - 1] + 1 - b[i]
                    b[i - 1] = b[i] - 1 
                    
        return min(aa, bb)
