class Solution:
    def movesToMakeZigzag(self, nums) -> int:
        res = 9999999
        tr = 0
        di = 1
        nums2 =  nums[:]
        for i in range(1,len
(nums)):
            if di>0:
                if nums[i]<=nums[i-1]:
                    tr+=nums[i-1]-nums[i]+1
                    nums[i-1] = nums[i]-1
            if di <0:
                
if nums[i]>=nums[i-1]:
                    tr+=nums[i]-nums[i-1]+1
                    nums[i] = nums[i-1]-1
            di = di*-1
        res = tr
        #print("r1",res)
        tr = 0
        di = -1
        nums = nums2[:]
        for i in range(1,len
(nums)):
            if di>0:
                if nums[i]<=nums[i-1]:
                    tr+=nums[i-1]-nums[i]+1
                    nums[i-1] = nums[i]-1
            if di <0:
                
if nums[i]>=nums[i-1]:
                    tr+=nums[i]-nums[i-1]+1
                    nums[i] = nums[i-1]-1
            di = di*-1
        #print(nums,nums2)
        res = min(res,tr)
        return res
