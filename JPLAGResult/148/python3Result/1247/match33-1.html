class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        if len(nums) == 1: return 0
        
        even = 0
        for i in range(1, len(nums), 2):
            prev = nums[i-1]
            succ = nums[i+1] if i+1 < len(nums) else float('inf')
            even += max(nums[i] - min(prev, succ) + 1, 0)
            
        odd 
= 0
        for i in range(0, len(nums), 2):
            prev = nums[i-1] if i-1 >= 0 else float('inf')
            succ = nums[i+1] if i+1 < len(nums) else float('inf')
            odd += max(nums[i] - min(prev, succ) + 1, 0)
            
        return min(even, odd)
