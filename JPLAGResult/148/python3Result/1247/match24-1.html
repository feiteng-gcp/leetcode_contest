class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        
        n 
= len(nums)
        if n <= 2:
            return 0
        
        odd, even = 0, 0
        
        for i in range(0, n, 2):
            min_ = float('inf')
            if i > 0:
                min_ = min(min_, nums[i-1])
            if i < n-1:
                min_ = min(min_, nums[i+1])
            odd += max(nums[i] - min_ + 1, 0)
            
        
for i in range(1, n, 2):
            min_ = float('inf')
            if i > 0:
                min_ = min(min_, nums[i-1])
            if i < n-1:
                min_ = min(min_, nums[i+1])
            even += max(nums[i] - min_ + 1, 0)
            
        return min(odd, even)
