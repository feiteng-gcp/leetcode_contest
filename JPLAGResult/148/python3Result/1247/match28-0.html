class Solution:
    def movesToMakeZigzag(self, nums: List[int]) -> int:
        n = len(nums)
        even = 0
        temp = nums.copy()
        for idx in range(n//2 + 1):
            i = idx*2
            if i >= n:
                continue
            if i - 1 >= 0 and nums[i-1] >= nums[i]:
                even += nums[i-1] - nums[i] + 1
            if i + 1 < n and nums[i+1] >= nums[i]:
                even += nums[i+1] - nums[i] + 1
                nums[i+1] -= nums[i+1] - nums[i] + 1
        # print(nums)
        nums = temp
        odd = 0
        for idx in range(n//2 + 1):
            i = idx*2 - 1
            if i < 0: 
                continue
            if i - 1 >= 0 and nums[i-1] >= nums[i]:
                odd += nums[i-1] - nums[i]+ 1
            if i + 1 < n and nums[i+1] >= nums[i]:
                odd += nums[i+1] - nums[i]+ 1
                nums[i+1] -= nums[i+1] - nums[i]+ 1
        # print(nums)
        # print(even, odd)
        return min(even, odd)
