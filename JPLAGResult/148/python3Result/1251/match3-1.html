class Solution:
    def longestDecomposition(self, text: str) -> int:
        
        N = len(text)
        
        
        def dfs(l: int, r: int) -> int:
            if l == r + 1:
                return 1
            if l >= r:
                return 0
            i = l + 1
            j 
= r - 1
            while i <= j:
                if text[l:i] == text[j:r]:
                    return 2 + dfs(i, j)
                i += 1
                j -= 1
            return 1
        
        return dfs(0, N)
