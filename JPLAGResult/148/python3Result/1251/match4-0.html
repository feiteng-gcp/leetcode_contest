class Solution:
    def longestDecomposition(self, text: str) -> int:
        n = len(text)
        l, r = 0, n - 1
        tl ,tr = '', ''
        ans = 0
        while l < r:
            tl += text[l]
            tr = text[r] + tr
            #print(tl ,tr)
            if tl == tr:
                tl, tr = '', ''
                ans += 2
            l += 1
            r -= 1
        #print(tl ,tr)
        if tl or n % 2 == 1:
            ans += 1
        return ans
