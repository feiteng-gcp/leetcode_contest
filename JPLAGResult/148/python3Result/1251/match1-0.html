class Solution:
    def longestDecomposition(self, text: str) -> int:
        def greedy(i, j):
            # print(i, j)
            if i == j:
                return 0
            for k in range(1, (j - i) // 2 + 1):
                if text[i:i + k] == text[j - k:j]:
                    return 2 + greedy(i + k, j - k)
            return 1
        return greedy(0, len(text))
