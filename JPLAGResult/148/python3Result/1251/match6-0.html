class Solution:
    def longestDecomposition(self, text: str) -> int:
        l_start, l_end = 0, 0
        n = len(text)
        r_start, r_end = n-1, n-1
        res = 0
        while l_end < r_start:
            if text[l_start:l_end+1] == text[r_start:r_end+1]:
                print(text[l_start:l_end+1])
                res += 2
                l_end = l_end + 1
                l_start = l_end
                r_start = r_start -1
                r_end = r_start
            else:
                l_end = l_end + 1
                r_start = r_start -1
        if l_start <= r_end:
            res += 1
        return res
