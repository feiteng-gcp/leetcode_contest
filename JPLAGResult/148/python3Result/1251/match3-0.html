class Solution:
    def longestDecomposition(self, text: str) -> int:
        def helper(start, end):
            if start > end:
                return 0
            i, j = start, end
            while i < j:
                if text[start:i + 1] == text[j: end + 1]:
                    return 2 + helper(i + 1, j - 1)
                i += 1
                j -= 1
            return 1
        
        return helper(0, len(text) - 1)
