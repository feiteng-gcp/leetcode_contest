class Solution:
    def longestDecomposition(self, text: str) -> int:
        n = len(text)

        if n == 0: 
            return 0
        if n == 1: 
            return 1

        for i in range
(1, n//2 + 1): 
            #print(text[:i], text[-i:])

            if text[:i] == text[-i:]: 
                return 2 + self.longestDecomposition(text[i: -i])

        return 1
