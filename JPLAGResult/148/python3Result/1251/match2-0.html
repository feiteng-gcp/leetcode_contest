class Solution:
    def longestDecomposition(self, text: str) -> int:
        n = len(text)
        def helper(left, right, cnt):
            if left == right:
                return cnt
            if left > n//2:
                return cnt
            for i in range(1, n//2 - left + 1):
                if text[left: left + i] == text[right - i: right]:
                    #print(text[left: left + i], text[right - i: right])
                    return helper(left + i, right - i, cnt + 2)
            return cnt + 1
            
        return helper(0, n, 0)
