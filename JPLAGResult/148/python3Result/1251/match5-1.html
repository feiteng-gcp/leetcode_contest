class Solution:
    def longestDecomposition(self, text: str) -> int:
        i,j = 0, len(text)-1
        L,R = 0, len(text)-1
        ans = 0
        while(i<j):
            if text[L:i+1] == text[j:R+1]:
                ans +=2
                # print(text[L:i+1])
                L 
= i+1
                R = j-1
            i +=1
            j -=1
        if L<=R:
            ans +=1
            # print(text[L:R+1])
        return ans
