class Solution:
    def longestDecomposition(self, text: str) -> int:
        N = len(text)
        cache = {}
        
        def dfs(i):
            if i * 2 == N:
                return 0
            elif i * 2 + 1 == N:
                return 1
            if i not in cache:
                t 
= 1
                for j in range(i + 1, N // 2 + 1):
                    if text[i:j] == text[N-j:N-i]:
                        t = max(t, dfs(j) + 2)
                cache[i] = t
            return cache[i]
        return dfs(0)
