class Solution:
    def longestDecomposition(self, text: str) -> int:
        i = 0
        j = len(text) - 1
        count = 0
        l = 0
        while i + l < j - l:
            if text[i + l] == text[j] and (l == 0 or text[i:i+l+1] == text[j-l:j+1]):
                    # print(text[i:i+l+1], text[j-l:j+1])
                    i 
+= l + 1
                    j -= l + 1
                    l = 0
                    count += 2
            else:
                l += 1
        if i <= j:
            count += 1
        return count
