class Solution:
    def longestDecomposition(self, text: str) -> int:
        front = 0
        end 
= len(text) - 1
        res = 0
        f_s = ''
        e_s = ''
        while end > front:
            f_s += text[front]
            e_s = text[end] + e_s
            if f_s == e_s:
                res += 2
                f_s = ''
                e_s = ''
            
            front += 1
            end -= 1
        if front == end or len(f_s):
            res += 1
            
        return res
