class Solution:
    def longestDecomposition(self, text: str) -> int:
        dp = {}
        def solve(left, right):
            if (left, right) in dp:
                return dp[(left, right)]
            if left == right:
                return 1
            if left > right:
                return 0
            half = (right - left + 1) // 2
            maxSp = 1
            for x in range(1, half + 1):
                if text[left:left + x] == text[right - x + 1:right + 1]:
                    maxSp = max(maxSp, 2 + solve(left + x, right - x))
            dp[(left, right)] = maxSp
            # print(maxSp)
            return maxSp
        r = solve(0, len(text) - 1)
        # print(dp)
        return r
