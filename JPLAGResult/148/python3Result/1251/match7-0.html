class Solution:
    import functools
    @functools.lru_cache(None)
    def longestDecomposition(self, text: str) -> int:
        if len(text) < 2:
            return len(text)
        if text[0] == text[-1]:
            return 2 + self.longestDecomposition(text[1:-1])
        for length in range(1, len(text)//2+1):
            if text[:length] == text[-length:]:
                #print(text[:length])
                return 2 + self.longestDecomposition(text[length:-length])
        return 1
