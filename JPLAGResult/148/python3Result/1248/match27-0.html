# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        def find_total_children(root):
            if not root:
                return 0
            cnt = 0
            q = [root]
            while q:
                root = q.pop()
                cnt += 1
                if root.left: q.append(root.left)
                if root.right: q.append(root.right)
                
            return cnt
        
        def determine_winner(x):
            x_cnt = find_total_children(x)
            oppo_max = min(x_cnt, n - find_total_children(x.left), n - find_total_children(x.right))
            if oppo_max < n / 2:
                return True
            return False
        
        q = [root]
        while q:
            root = q.pop()
            if root.val == x:
                return determine_winner(root)
            if root.left: q.append(root.left)
            if root.right: q.append(root.right)
