# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        
        def tree_size(node):
            if node is None:
                return 0
            return tree_size(node.left) + tree_size(node.right) + 1
        
        def search(node, x):
            
            if node is None:
                return None
            if node.val == x:
                return node
            nd = search(node.left, x)
            if nd:
                return nd
            return search(node.right, x)
        
        nx = search(root, x)
        assert nx is not None
        ls = tree_size(nx.left)
        rs = tree_size(nx.right)
        rest = n - ls - rs - 1
        return max(ls, rs, rest) >= (n+1) // 2
