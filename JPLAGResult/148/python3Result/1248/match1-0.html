# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        self.loc = None
        def dfs(node):
            if not node:
                return
            if node.val == x:
                self.loc = node
                return
            else:
                dfs(node.left)
                dfs(node.right)
        dfs(root)
        self.l = 0
        self.r = 0
        def cnt_l(node):
            if node:
                self.l += 1
                cnt_l(node.left)
                cnt_l(node.right)
        def cnt_r(node):
            if node:
                self.r += 1
                cnt_r(node.left)
                cnt_r(node.right)
        cnt_l(self.loc.left)
        cnt_r(self.loc.right)
        t = n - 1 -self.l - self.r
        return max(t,max(self.l,self.r)) > n/2
