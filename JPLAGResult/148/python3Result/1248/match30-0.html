# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        
        def C(node):
            if not node: return 0
            return 1+C(node.left)+C(node.right)
        
        def F(node):
            if not node: return None
            if node.val == x: return node
            return F(node.left) or F(node.right)
            
        node = F(root)
        l, r = C(node.left), C(node.right)
        print(n-l-r-1, l, r)
            
        return max(n-l-r-1, l, r) > n // 2
