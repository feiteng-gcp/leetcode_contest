# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        
        def count_nodes(node):
            if not node:
                return 0
            
            return 1 + count_nodes(node.left) + count_nodes(node.right)
        
        def find_node(node, value):
            if not node:
                return
            
            if node.val == value:
                return node
            
            return find_node(node.left, value) or find_node(node.right, value)
        
        x_node = find_node(root, x)
        left_subtree = count_nodes(x_node.left)
        right_subtree = count_nodes(x_node.right)
        parent_subtree = n - left_subtree - right_subtree - 1
        
        return max(left_subtree, right_subtree, parent_subtree) > n // 2
