class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        def su(node):
            if not node:
                return 0
            return su(node.left)+su(node.right)+1
        def dfs(node):
            if not node:
                return None
            #print(node.val)
            if node.val == x:
                return node
            nl = dfs(node.left)
            nr = dfs
(node.right)
            #print(nl,nr)
            return nl if nl else nr
        target = dfs(root)
        l = su(target.left)
        r = su(target.right)
        t = n
        p = t-l-r-1
        #print(l,r,t,p)
        return l*2>t or (r*2>t) or(p*2>t)
