# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        node = self.findNode(root, x)
        l = self.countNode(node.left)
        r = self.countNode(node.right)
        return max(l, r, n-l-r-1) > n//2
        
        
    def findNode(self, root, x):
        if root.val == x:
            return root
        if root.left:
            find = self.findNode(root.left, x)
            if find:
                return find
        if root.right:
            return self.findNode(root.right, x)
        
return None
    
    def countNode(self, root):
        if root is None:
            return 0
        return self.countNode(root.left) + self.countNode(root.right) + 1
