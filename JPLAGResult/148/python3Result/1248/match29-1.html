#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        
        def find(root, x):
            if root.val == x:
                return root
            l = find(root.left, x) if root.left else None
            if l: return l
            r = find(root.right, x) if root.right else None

        def count(root):
            if not root:
                return 0
            return 1 + count(root.left) + count(root.right)

        cl, cr = count(nodex.left),  count(nodex.right)
        cu = n - cl - cr - 1
        return max(cl, cr, cu) > n // 2
