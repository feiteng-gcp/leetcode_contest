# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        def num_nodes(root):
            if root is None:
                return 0
            return 1 + num_nodes(root.left) + num_nodes(root.right)
        
        def dfs(root, x):
            if root is None:
                return None
            if root.val == x:
                return root
            left_ret = dfs(root.left, x)
            return left_ret if left_ret is not None else dfs(root.right, x)
        
        node = dfs(root, x)
        num_left = num_nodes(node.left)
        num_right = num_nodes(node.right)
        num_parent = n - num_left - num_right - 1
        
        maxi = max(num_left, num_right, num_parent)
        return maxi > sum([num_left, num_right, num_parent]) - maxi
