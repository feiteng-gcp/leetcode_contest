# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        def size(root):
            if root is None:
                return 0
            return size(root.left) + size(root.right) + 1
        def find(root, x):
            if root is None:
                return None
            if root.val == x:
                return root
            l, r = find(root.left, x), find(root.right, x)
            return l if l else r
        
        nd = find(root, x)
        A = size(nd.left)
        B = size(nd.right)
        return max(A, B, n - A - B - 1) * 2 > n
