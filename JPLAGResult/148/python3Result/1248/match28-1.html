# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def count(self, node):
        if node == None:
            return 0
        return self.count(node.left) + self.count(node.right) + 1
    
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        stack = [root]

        
while True:
            node = stack.pop(0)
            if node.val == x:
                break
            if node.left:
                stack.append(node.left)
            if node.right:
                stack.append(node.right)
        
        left = self.count(node.left)
        right = self.count(node.right)
        up = n - left - right - 1

        ans = sorted([left, right, up])

        return ans[0] + ans[1] + 1 < ans[2]
