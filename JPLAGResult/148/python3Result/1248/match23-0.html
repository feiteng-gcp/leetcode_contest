# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        node = self.val2node(root, x)
        lc, rc = self.countchild(node.left), self.countchild(node.right)
        father = n - 1 - lc - rc
        c, b, a = sorted([lc, rc, father])
        if a > b + c + 1:
            return True
        return False

    def countchild(self, root):
        if not root:
            return 0
        return self.countchild(root.left) + self.countchild(root.right) + 1
    
    def val2node(self, root, x):
        if root.val == x:
            return root
        if root.left:
            node = self.val2node(root.left, x)
            if node:
                return node
        if root.right:
            node = self.val2node(root.right, x)
            if node:
                return node
