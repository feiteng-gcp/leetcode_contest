# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        node_x = self.find_x(root, x)
        left_x = self.count_subtree(node_x.left)
        right_x = self.count_subtree(node_x.right)
        rest_x = n - left_x - right_x - 1
        if left_x > n // 2 or right_x > n // 2 or rest_x > n // 2:
            return True
        else:
            return False
    
    
    def count_subtree(self, root):
        if not root:
            return 0
        left = self.count_subtree(root.left)
        right = self.count_subtree(root.right)
        return left + right + 1
    
    def find_x(self, root, x):
        if not root:
            return None
        if root.val == x:
            return root
        left = self.find_x(root.left, x)
        if left:
            return left
        right = self.find_x(root.right, x)
        if right:
            return right
        return None
