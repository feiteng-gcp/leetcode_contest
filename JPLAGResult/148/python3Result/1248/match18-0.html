# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        start = self.find(root, x)
        left = self.count(start.left)
        right = self.count(start.right)
        total = self.count(start)
        
        #print(left, right, total)
        pleft = left > n - left
        pright = right > n - right
        pp = n - total > total
        
        return pleft or pright or pp
        
    def count(self, node):
        if not node: return 0
        return 1 + self.count(node.left) + self.count(node.right)
    
    def find(self, node, val):
        if not node: return None
        if node.val == val: return node
        left, right = self.find(node.left, val), self.find(node.right, val)
        if left: return left
        if right: return right
        return None
