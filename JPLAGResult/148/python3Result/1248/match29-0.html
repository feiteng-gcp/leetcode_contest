# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        def helper(r):
            if not r:
                return None
            if r.val == x:
                return r
            else:
                return helper(r.left) or helper(r.right)
            
        def count(r):
            if not r:
                return 0
            return count(r.left) + count(r.right) + 1
            
        xn = helper(root)
        left, right = count(xn.left), count(xn.right)
        up = n - left - right - 1
        
        return left > 1 + up + right or right > 1 + up + left or up > 1 + left + right
