# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        def count(root):
            if not root: return 0
            return count(root.left) + count(root.right) + 1
        
        
        
        def findx(cur, x):
            if not cur: return None
            if cur.val == x: return cur
            l = findx(cur.left, x)
            r = findx(cur.right, x)
            if l: return l
            return r
        
        node = findx(root, x)
        
        xleft = count(node.left)
        xright = count(node.right)
        xpar = n - xleft - xright - 1
        
        # print(xleft, xright, xpar, n, xleft > (n - xleft), xright > (n - xright))
        if xleft > (n - xleft) or xright > (n - xright) or xpar > (n - xpar):
            return True
        return False
