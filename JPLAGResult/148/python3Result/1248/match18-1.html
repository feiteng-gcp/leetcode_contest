# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    
def size(self, u):
        if u is None:
            return 0
        return 1 + self.size(u.left) + self.size(u.right)

    def find(self, u, x):
        if u is None:
            return None
        if u.val == x:
            return u
        l = self.find(u.left, x)
        if l is not None:
            return l
        return self.find(u.right, x)
    
    
def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        u = self.find(root, x)
        l = self.size(u.left)
        r = self.size(u.right)
        top = n - 1 - l - r
        return 2*max(l, r, top) >= n
