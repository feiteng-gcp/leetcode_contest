# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def btreeGameWinningMove(self, root: TreeNode, n: int, x: int) -> bool:
        def f(root):
            if root == None:
                return 0
            res = 1
            res += f(root.left)
            res += f(root.right)
            return res
        def findnode(root):
            if root == None:
                return None
            if root.val == x:
                return root
            a = findnode(root.left)
            if a != None:
                return a
            b = findnode(root.right)
            if b != None:
                return b
            return None
        node = findnode(root)
        a = f(node.left)
        b = f(node.right)
        c= n-a-b-1
        if c > a+b+1 or a > b+c+1 or b > a+c+1:
            return True
        else:
            return False
