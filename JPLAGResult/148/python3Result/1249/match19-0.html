from collections import defaultdict
import bisect
class SnapshotArray:
    def __init__(self, length: int):
        self.snap_i = 0
        self.d = defaultdict(list)

    def set(self, index: int, val: int) -> None:
        self.d[index].append((self.snap_i, val))

    def snap(self) -> int:
        self.snap_i +=1
        return self.snap_i - 1

    def get(self, index: int, snap_id: int) -> int:
        i = bisect.bisect_left(self.d[index], (snap_id + 1, -1))
        return 0 if i == 0 else self.d[index][i-1][1]
