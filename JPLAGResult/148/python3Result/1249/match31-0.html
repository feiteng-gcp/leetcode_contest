class SnapshotArray:

    def __init__(self, length: int):
        self.current_snap_id = 0
        self.nums = {}
        self.length = length
        

    def set(self, index: int, val: int) -> None:
        if index not in self.nums:
            self.nums[index] = [(self.current_snap_id, val)]
        elif self.nums[index][-1][0] == self.current_snap_id:
            self.nums[index][-1] = (self.current_snap_id, val)
        else:
            self.nums[index].append((self.current_snap_id, val))
        

    def snap(self) -> int:
        self.current_snap_id += 1
        return self.current_snap_id - 1
        

    def get(self, index: int, snap_id: int) -> int:
        if index >= self.length or snap_id >= self.current_snap_id:
            return None
        if index not in self.nums:
            return 0
        else:
            if self.nums[index][0][0] > snap_id:
                return 0
            last_snap_id_pos = self.find_last_change(index, snap_id)
            return self.nums[index][last_snap_id_pos][1]

    
    
    def find_last_change(self, index, snap_id):
        left = 0
        right = len(self.nums[index]) - 1
        if self.nums[index][right][0] <= snap_id:
            return right
        while left + 1 < right:
            mid = (left + right) // 2
            if self.nums[index][mid][0] <= snap_id:
                left = mid
            else:
                right = mid
        return left
