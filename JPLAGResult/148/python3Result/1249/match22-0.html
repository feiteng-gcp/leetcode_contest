class SnapshotArray:

    def __init__(self, length: int):
        self.ar=[{-1:0} for _ in range(length)]
        self.sn=0
        return

    def set(self, index: int, val: int) -> None:
        if self.sn in self.ar[index]:
            self.ar[index][self.sn]=val
        else:
            self.ar[index][self.sn]=val
        return

    def snap(self) -> int:
        self.sn+=1
        return self.sn-1

    def get(self, index: int, snap_id: int) -> int:
        cv=snap_id
        while cv not in self.ar[index]: cv-=1
        return self.ar[index][cv]
