class SnapshotArray:

    def __init__(self, length: int):
        self.id_map = {i:[[-1],[0]] for i in range(length)}
        self.snap_id = -1

    def set(self, index: int, val: int) -> None:
        self.id_map[index][0] += [self.snap_id]
        self.id_map[index][1] += [val]

    def snap(self) -> int:
        self.snap_id += 1
        return self.snap_id

    def get(self, index: int, snap_id: int) -> int:
        loc = bisect.bisect_left(self.id_map[index][0],snap_id,0,len(self.id_map[index][0]))
        return self.id_map[index][1][loc-1]
