class SnapshotArray:

    def __init__(self, length: int):
        self.L=[[] for i in range
(length)]
        self.current=0

    def set(self, index: int, val: int) -> None:
        self.L[index].append((self.current,val))

    def snap(self) -> int:
        self.current+=1
        return self.current-1

    def get(self, index: int, snap_id: int) -> int:
        A=self.L[index]
        ind= bisect.bisect(A,(snap_id,1e10))
        if ind>0:
            return A[ind-1][1]
        return 0
