class SnapshotArray:

    def __init__(self, length: int):
        # the dictionary with snap_id:{}
        # in the dictionary, save index:value
        self._d = {}
        # counter of calling snap
        self._snapID = 0
        self._d
[self._snapID] = {}
        
    def set(self, index: int, val: int) -> None:            
        self._d[self._snapID][index] = val

    def snap(self) -> int:
        self._snapID += 1
        self._d[self._snapID] = {}
        return self._snapID - 1
        
    def get(self, index: int, snap_id: int) -> int:
        while index not in self._d[snap_id]:
            snap_id -= 1
            if snap_id == -1:
                return 0
        return self._d[snap_id][index]
