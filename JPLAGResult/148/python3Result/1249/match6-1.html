class SnapshotArray:

    def __init__(self, length: int):
        self._next_snap_id = 0
        self
._snaps = {}
        self._values = {}

    def set(self, index: int, val: int) -> None:
        self._values[index] = val

    def snap(self) -> int:
        self._snaps[self._next_snap_id] = self._values.copy()
        self._next_snap_id += 1
        return self._next_snap_id - 1

    def get(self, index: int, snap_id: int) -> int:
        return self._snaps[snap_id].get(index, 0)
