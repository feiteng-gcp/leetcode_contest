import bisect
import math

class SnapshotArray:

    def __init__(self, length: int):
        self.snap_c = 0
        self.data = {i: [[0, 0]] for i in range(length)}

    def set(self, index: int, val: int) -> None:
        snap, _ = self.data[index][-1]
        if snap == self.snap_c:
            self.data[index][-1][1] = val
        else:
            self.data[index].append([self.snap_c, val])

    def snap(self) -> int:
        self.snap_c += 1
        return self.snap_c - 1
    
    def get(self, index: int, snap_id: int) -> int:
        i = bisect.bisect_right(self.data[index], [snap_id, math.inf]) - 1
        return self.data[index][i][1]
