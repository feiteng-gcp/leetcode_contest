class SnapshotArray:

    def __init__(self, length: int):
        self.strc = [[(0,0)] for _ in range(length)]
        self.curr_time = 0

    def set(self, index: int, val: int) -> None:
        if self
.strc[index][-1][0] == self.curr_time:
            self.strc[index][-1] = (self.curr_time, val)
        else:
            self.strc[index].append((self.curr_time, val))

    def snap(self) -> int:
        self.curr_time += 1
        return self.curr_time - 1

    def get(self, index: int, snap_id: int) -> int:
        from bisect import bisect_right
        cidx = bisect_right(self.strc[index], (snap_id + 0.1, ))
        return self.strc[index][cidx - 1][1]
