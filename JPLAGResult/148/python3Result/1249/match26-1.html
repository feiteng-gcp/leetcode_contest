import collections
class SnapshotArray:

    def __init__(self, length: int):
        self.dic = collections.defaultdict(dict) # key = index, value = {k = snap_id, v=value}
        self.cur_snap =
 0

    def set(self, index: int, val: int) -> None:
        self.dic[index][self.cur_snap] = val

    def snap(self) -> int:
        self.cur_snap += 1
        return self.cur_snap - 1
        

    def get(self, index: int, snap_id: int) -> int:
        lst = self.dic[index]
        while snap_id not in lst:
            if snap_id == 0:
                return 0
            snap_id -= 1
        return self.dic[index][snap_id]
