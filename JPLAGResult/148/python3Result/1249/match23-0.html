class SnapshotArray:

    def __init__(self, length: int):
        self.nums = [0 for i in range(length)]
        self.cur_snap = {}
        self.snaps = []
        

    def set(self, index: int, val: int) -> None:
        self.cur_snap[index] = val
        

    def snap(self) -> int:
        self.snaps.append(self.cur_snap)
        self.cur_snap = {}
        return len(self.snaps) - 1
        

    def get(self, index: int, snap_id: int) -> int:
        for snap in self.snaps[snap_id::-1]:
            if index in snap:
                return snap[index]
        return self.nums[index]
