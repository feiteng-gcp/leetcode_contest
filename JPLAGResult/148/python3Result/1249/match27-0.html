class SnapshotArray:

    def __init__(self, length: int):
        self.sid = 0
        self.lst = {i:{0:0} for i in range(length)}

    def set(self, index: int, val: int) -> None:
        self.lst[index][self.sid] = val

    def snap(self) -> int:
        self.sid += 1
        return self.sid-1

    def get(self, index: int, snap_id: int) -> int:
        d = self.lst[index]
        #print(d)
        if snap_id in d: return d[snap_id]
        for i in list(d.keys())[::-1]:
            if i <= snap_id:
                return d[i]
