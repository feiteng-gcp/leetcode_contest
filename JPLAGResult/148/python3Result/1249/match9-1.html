class SnapshotArray:

    def __init__(self, length: int):
        self.snap_id = 0
        self.array = [[[0, 0]] for _ in range(length)]
        

    def set(self, index: int, val: int) -> None:
        temp = self.array[index][-1]
        if temp[0] == self.snap_id:
            temp[1] 
= val
        else:
            self.array[index].append([self.snap_id, val])

    def snap(self) -> int:
        self.snap_id += 1
        return self.snap_id - 1

    def get(self, index: int, snap_id: int) -> int:
        def upper_bound(array, l, r, target):
            while l < r:
                mid = l + (r - l) // 2
                if array[mid][0] > target:
                    r = mid
                else:
                    l = mid + 1
            return l
        temp = upper_bound(self.array[index], 0, len(self.array[index]), snap_id)
        # print(self.array, temp)
        return self.array[index][temp - 1][1]
