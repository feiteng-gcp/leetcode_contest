class SnapshotArray:

    def __init__(self, length: int):
        self.arr = [0] * length
        self.snaps = 0
        self.change_set = {}
        self.snap_history = {}

    def set(self, index: int, val: int) -> None:
        self.change_set[index] = val

    def snap(self) -> int:
        self.snap_history[self.snaps] = self.change_set.copy()
        self.snaps += 1
        return self.snaps - 1

    def get(self, index: int, snap_id: int) -> int:
        res = self.snap_history[snap_id]
        if index in res:
            return res[index]
        return 0
