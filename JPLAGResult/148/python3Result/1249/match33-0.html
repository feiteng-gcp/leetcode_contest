class SnapshotArray:

    def __init__(self, length: int):
        self.MAP = dict([(i,[(0,0)]) for i in range(length)])
        self.cnt = 0
        

    def set(self, index: int, val: int) -> None:
        self.MAP[index].append((self.cnt,val))

    def snap(self) -> int:
        self.cnt+=1
        return self.cnt-1

    def get(self, index: int, snap_id: int) -> int:
        snap = self.MAP[index]
        l,r = 0,len(snap)-1
        while l+1<r:
            m = (l+r)//2
            idx,val = snap[m]
            if idx<=snap_id:
                l = m
            else:
                r = m
        return snap[r][1] if snap[r][0]<=snap_id else snap[l][1]
