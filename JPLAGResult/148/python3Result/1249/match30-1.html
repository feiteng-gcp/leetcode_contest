class SnapshotArray:

    def __init__(self, length: int):
        self.list = [[[0, 0]] for i in range(length)]
        self.seq = 0

    def set(self, index: int, val: int) -> None:
        if self.list[index]:
            if self.list[index][-1][0] != self.seq:
                self.list[index].append([self.seq, val])
            else:
                self.list[index][-1][1] = val

    def snap(self) -> int:
        self.seq += 1
        return self.seq - 1
        

    def get(self, index: int, snap_id: int) -> int:
        # print(self.list)
        l, r = 0, len(self.list[index]) - 1
        while l + 1 < r:
            m = (l + r) // 2
            if self.list[index][m][0] <= snap_id:
                l = m
            else:
                r = m - 1
        if 0 <= r < len(self.list[index]) and self.list[index][r][0] <= snap_id:
            return self.list[index][r][1]
        if 0 <= l < len(self.list[index]) and self.list[index][l][0] <= snap_id:
            return self.list[index][l][1]
