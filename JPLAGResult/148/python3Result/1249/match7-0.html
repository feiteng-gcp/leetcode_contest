class SnapshotArray:

    def __init__(self, length: int):
        self.t = {}
        self.tt = {}
        self.num = -1

    def set(self, index: int, val: int) -> None:
        self.t[index] = val
        

    def snap(self) -> int:
        self.num += 1
        self.tt[self.num] = {i: self.t[i] for i in self.t}
        return self.num

    def get(self, index: int, snap_id: int) -> int:
        if index not in self.tt[snap_id]:
            return 0
        return self.tt[snap_id][index]
