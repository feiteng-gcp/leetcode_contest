        self.id = 0
        
    def set(self, index: int, val: int) -> None:
        self.m[index].append((self.id, val))

    def snap(self) -> int:
        self.id += 1
        return self.id - 1

    def get(self, index: int, snap_id: int) -> int:
        arr = self.m[index]
        #print(arr)
        if not arr: return 0
        loc = bisect.bisect_left(arr, (snap_id, float("inf")))
        if loc == 0: return 0
        return arr[loc - 1][1]
