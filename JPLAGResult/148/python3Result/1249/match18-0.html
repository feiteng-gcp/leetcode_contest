class SnapshotArray:

    def __init__(self, length: int):
        self.snap_id = 0
        self.vals = [[(0, 0)] for _ in range(length)] # (snap_id, val)

    def set(self, index: int, val: int) -> None:
        if self.vals[index][-1][0] == self.snap_id:
            self.vals[index].pop()
        self.vals[index].append((self.snap_id, val))

    def snap(self) -> int:
        self.snap_id += 1
        return self.snap_id - 1

        # print(index, snap_id, idx, self.vals[index])
        # if idx == len(self.vals[index]):
        #     idx -= 1
        return self.vals[index][idx-1][1]
