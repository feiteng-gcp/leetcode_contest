class SnapshotArray:

    def __init__(self, length: int):
        self.curr_id = 0
        self.mp = {}
        for i in range(length):
            self.mp[(i,self.curr_id)] = 0

    def set(self, index: int, val: int) -> None:
        self.mp[(index,self.curr_id)] =
 val

    def snap(self) -> int:
        self.curr_id += 1
        return self.curr_id - 1

    def get(self, index: int, snap_id: int) -> int:
        id = snap_id
        while (index, id) not in self.mp:
            id -= 1
        return self.mp[(index,id)]
