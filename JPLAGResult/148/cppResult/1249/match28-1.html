class SnapshotArray {

    unordered_map<int, int> cur;
    vector<unordered_map<int, int>> snaps;
    int s;
public:
    SnapshotArray(int length) {
        s = 0;
    }
    
    void set(int index, int val) {
        cur[index] = val;
    }
    
    int snap() {
        snaps.push_back(cur);
        return snaps.size() - 1;
    }
    
    int get(int index, int snap_id) {
        return snaps[snap_id][index];
    }
};
