class SnapshotArray {
public:

    SnapshotArray(int length):m_changes(length, map<int, int>()), m_id(0) {
        
    }
    
    void set(int index, int val) {
        m_changes[index][m_id - 1] = val;
    }
    
    int snap() {
        return m_id++;
    }
    
    int get(int index, int snap_id) {
        auto &e = m_changes[index];
        auto it = e.lower_bound(snap_id);
        if (it == e.begin()) 
            return 0;
        else 
            return prev(it)->second;
    }
    
private:
    vector<map<int, int>> m_changes;  
    int m_id;
};

 * Your SnapshotArray object will be instantiated and called as such:
 * SnapshotArray* obj = new SnapshotArray(length);
 * obj->set(index,val);
 * int param_2 = obj->snap();
 * int param_3 = obj->get(index,snap_id);
 */
