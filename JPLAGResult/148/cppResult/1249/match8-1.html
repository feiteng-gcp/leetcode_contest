map<int,int> M[50505];
int ID;
class SnapshotArray {
public:

    SnapshotArray(int length) {
        int i;
        ID=0;
        FOR(i,length) {
			M[i].clear();
			M[i][0]=0;
		}

    }
    
    void set(int index, int val) {
        M[index][ID]=val;
    }
    
    int snap() {
        return ID++;
    }
    
    int get(int index, int snap_id) {
        auto it=M[index].lower_bound(snap_id+1);
        it--;
        return it->second;
    }


};
