class SnapshotArray {
public:
  
    map<pair<int,int>,int> allShots;
    int currSnap;
    SnapshotArray(int length) {
        ios::sync_with_stdio(false);
      cin.tie(0);
      cout.tie(0);

        currSnap = 0;
    }
    
    void set(int index, int val) {
        allShots[make_pair(currSnap,index)] = val;   
    }
    
    int snap() {
        currSnap++; 
        return currSnap-1;
    }
    
    int get(int index, int snap_id) {
        
      for(int i = snap_id ; i>=0 ; i--){
        if(allShots.find(make_pair(i,index)) != allShots.end())
        {
          return allShots[make_pair(i,index)];
        }
      }
      return 0;
    }
};
