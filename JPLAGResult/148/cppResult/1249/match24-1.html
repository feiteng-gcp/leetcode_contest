class SnapshotArray {
private:
    int id = 0;
    unordered_map<int, map<int, int>> valByIdx;
public:
    SnapshotArray(int length) {
        valByIdx = unordered_map<int, map<int, int>>();

    }
    
    void set(int index, int val) {
        valByIdx[index][id] = val;
    }
    
    int snap() {
        id++;
        return id-1;
    }
    
    int get(int index, int snap_id) {
        if(valByIdx.find(index) == valByIdx.end()){
            return 0;
        }

        map<int, int>::iterator next = valByIdx[index].upper_bound(snap_id);
        if(next == valByIdx[index].begin()){
            return 0;
        }
        next--;
        return next->second;
    }
};
