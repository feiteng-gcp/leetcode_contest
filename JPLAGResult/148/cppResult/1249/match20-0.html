class SnapshotArray {
public:
    SnapshotArray(int length): values(length) {
        
    }
    
    void set(int index, int val) {
        values[index][-t] = val;
    }
    
    int snap() {
        return t++;
    }
    
    int get(int index, int snap_id) {
        auto it = values[index].lower_bound(-snap_id);
        if (it == end(values[index])) {
            return 0;
        }
        return it->second;
    }
private:
    int t = 0;
    vector<map<int, int>> values;
};

 * int param_3 = obj->get(index,snap_id);
