class SnapshotArray {
public:
    SnapshotArray(int length) {
        map<int,int> mp;
        mp[0]=0;
        arr.resize(length, mp);
    }
    
    void set(int index, int val) {
        arr[index][sn] = val;
    }
    
    int snap() {
        return sn++;
    }
    
    int get(int index, int snap_id) {
        auto it = arr[index].upper_bound(snap_id);
        --it;
        return it->second;
    }
private:
    int sn = 0;
    vector<map<int,int>> arr;
};
