class SnapshotArray {
private:
    vector<map<int, int>> v;
    int id;
public:
    SnapshotArray(int length) {
        for (int i = 0; i < length; ++i) {
            v.push_back(map<int, int>());
            v.back()[-1] = 0;
        }

        id = 0;
    }
    
    void set(int index, int val) {
        v[index][id] = val;
    }
    
    int snap() {
        return id++;
    }
    
    int get(int index, int snap_id) {
        auto iter = v[index].upper_bound(snap_id);
        iter--;
        return iter->second;
    }
};
