class SnapshotArray {
public:
    vector<map<int, int>> v;
    map<int, int> ma;
    SnapshotArray(int length) {
        v.clear();
        ma.clear();
    }
    
    void set(int index, int val) {
        ma[index] = val;
    }
    
    int snap() {
        v.push_back(ma);
        return v.size() - 1;
    }
    
    int get(int index, int snap_id) {
        auto &a = v[snap_id];
        auto it = a.find(index);
        if(it != a.end())
            return it->second;
        return 0;
    }
};
