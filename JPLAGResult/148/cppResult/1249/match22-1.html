class SnapshotArray {
public:
    vector<map<int, int>> sa;
    unordered_map<int, int> modified;
    int sid;
    

    SnapshotArray(int length) {
        sa.resize(length);    
        sid = 0;
    }
    
    void set(int index, int val) {
        modified[index] = val;
    }
    
    int snap() {
        for (auto &p : modified) {

            sa[p.first][sid] = p.second;
            // cout << p.first << " " << sid << " " << p.second << endl;
        }
        modified.clear();
        return sid ++;
    }
    
    int get(int index, int snap_id) {
        auto &a = sa[index];
        auto it = a.upper_bound(snap_id);
        if (it == a.begin()) return 0;
        it--;
        return it->second;
    }
};

 * SnapshotArray* obj = new SnapshotArray(length);
 * obj->set(index,val);
 * int param_2 = obj->snap();
 * int param_3 = obj->get(index,snap_id);
 */
