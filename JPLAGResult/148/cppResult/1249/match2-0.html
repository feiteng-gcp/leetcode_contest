class SnapshotArray {
public:
    vector<int> nums;
    map<int, int> change;
    vector<map<int, int>> snapshot;
    SnapshotArray(int length) {
        nums.resize(length, 0);
    }
    
    void set(int index, int val) {
        change[index] = val;
    }
    
    int snap() {
        snapshot.push_back(change);
        change.clear();
        return snapshot.size() - 1;
    }
    
    int get(int index, int snap_id) {
        int i;
        for (i = snap_id; i >= 0; i--) {
            if (snapshot[i].find(index) != snapshot[i].end()) {
                return snapshot[i][index];
            }
        }
        return 0;
    }
};
