class SnapshotArray {
    int snap_times;
    vector<map<int,int>> changes;
    map<int,int> m;
    vector<int> a;
    //vectorint> lastchange;
public:
    SnapshotArray(int length) {
        snap_times=0;
        for(int i=0;i<length;i++){
            a.push_back(0);
            //lastchange.push_back(0);
        }

    }
    
    void set(int index, int val) {
        m[index]=val;
        //lastchange()
    }
    
    int snap() {
        changes.push_back(m);
        m.clear();
        snap_times+=1;
        return snap_times-1;
    }
    
    int get(int index, int snap_id) {
        for(int i=snap_id;i>=0;i--){
            if(changes[i].find(index)!=changes[i].end())
                return changes[i][index];
        }
        return a[index];
    }
};

 * SnapshotArray* obj = new SnapshotArray(length);
 * obj->set(index,val);
 * int param_2 = obj->snap();
 * int param_3 = obj->get(index,snap_id);
 */
