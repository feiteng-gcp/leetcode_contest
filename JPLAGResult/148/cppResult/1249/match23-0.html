class SnapshotArray {
    int snapCnt;
    std::vector<std::map<int, int>> map;
public:
    SnapshotArray(int length) {
        snapCnt = 0;
        map.resize(length);
        for(int i = 0; i < length; ++i) {
            map[i][0] = 0;
        }
    }
    
    void set(int index, int val) {
        map[index][snapCnt] = val;
    }
    
    int snap() {
        return snapCnt++;
    }
    
    int get(int index, int snap_id) {
        auto it = map[index].upper_bound(snap_id);
        --it;
        return it->second;
    }
};
