class SnapshotArray {
public:
    int N;
    vector<map<int,int>> snaparr;
    map<int,int> mp;
    int snapid = -1;
    SnapshotArray(int length) {
        N = length;
        snapid = -1;
    }
    
    void set(int index, int val) {
        mp[index] = val;
    }
    
    int snap() {
        snaparr.push_back(mp);
        mp.clear();
        snapid++;
        return snapid;
    }
    
    int get(int index, int snap_id) {
        for(int inx = snap_id; inx >= 0; inx--){
            if(snaparr[inx].find(index)!=snaparr[inx].end()){
                return snaparr[inx][index];
            }
        }
        return 0;
    }
};

 * SnapshotArray* obj = new SnapshotArray(length);
 * obj->set(index,val);
 * int param_2 = obj->snap();
 * int param_3 = obj->get(index,snap_id);
 */
