class SnapshotArray {
public:
    SnapshotArray(int length) {
        for (int i = 0; i < length; i++)

            m[i][-1] = 0;
    }
    
    void set(int index, int val) {
        m[index][id] = val;
    }
    
    int snap() {
        return id++;
    }
    
    int get(int index, int snap_id) {
        auto tmp = m[index].upper_bound(snap_id);
        tmp--;
        return tmp->second;
    }
private:
    int id = 0;
    map<int, map<int, int>> m;
};
