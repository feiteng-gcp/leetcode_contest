class SnapshotArray {

    unordered_map<int, map<int, int>> m;
    int snapid = 0;
public:
    SnapshotArray(int length) {        
    }
    
    void set(int index, int val) {
        m[index][snapid] = val;
    }
    
    int snap() {
        return snapid++;
    }
    
    int get(int index, int snap_id) {
        if (!m.count(index)) return 0;
        if (m[index].empty()) return 0;
        auto lb = m[index].lower_bound(snap_id);
        if (lb != m[index].end() && lb->first == snap_id)
            return lb->second;
        if (lb == m[index].begin())
            return 0;
        return prev(lb)->second;
    }
};
