class SnapshotArray {
public:
    map<int, int> value[50010];
    int now;
    
    SnapshotArray(int length) {
        for (int i = 0;i < length;i++) {
            value[i][0] = 0;
        }
        now = 0;
    }
    
    void set(int index, int val) {
        value[index][now] = val;
    }
    
    int snap() {
        now++;
        return now - 1;
    }
    
    int get(int index, int snap_id) {
        map<int, int>::iterator it = value[index].lower_bound(snap_id);
        if (it != value[index].end() && it->first == snap_id) {
            return it->second;
        }
        it--;
        return it->second;
    }
};

 * SnapshotArray* obj = new SnapshotArray(length);
 * obj->set(index,val);
 * int param_2 = obj->snap();
 * int param_3 = obj->get(index,snap_id);
