class SnapshotArray {
public:
    vector<map<int, int>> m;
    SnapshotArray(int length) {
        snap();
    }
    
    void set(int index, int val) {
        m[m.size()-1][index] = val;
    }
    
    int snap() {
        map<int,int> x;
        m.push_back(x);
        return m.size() - 2;
    }
    
    int get(int index, int snap_id) {
        for (int i = snap_id; i >= 0; i--) {
            if (m[i].find(index) != m[i].end()) {
                return m[i][index];
            }
        }
        return 0;
    }
};
