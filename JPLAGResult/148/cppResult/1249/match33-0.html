typedef pair<ll,ll> P;

map<ll,ll> m;
int k=0,c=0;
vector<P> a[55555];
class SnapshotArray {
public:
  SnapshotArray(int n) {
    k=0,c=0;
    rep(i,n) a[i].clear();m.clear();
    rep(i,n) a[i].pb(P(0,0));
  }
    
  void set(int index, int val) {
    a[index].pb(P(c,val));
    c++;
  }
    
  int snap() {
    m[k]=c;
    k++;
    return k-1;
  }
    
  int get(int index, int snap_id) {
    ll i=upper_bound(all(a[index]),P(m[snap_id],0))-a[index].begin();
    return a[index][i-1].S;
  }
};
