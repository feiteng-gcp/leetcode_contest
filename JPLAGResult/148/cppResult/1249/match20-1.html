class SnapshotArray {
private:
    int nextid;
    unordered_map<int, map<int, int>> m;
public:
    SnapshotArray(int length) {
        nextid = 0;

    }
    
    void set(int index, int val) {
        m[index][nextid] = val;
    }
    
    int snap() {
        return nextid++;
    }
    
    int get(int index, int snap_id) {
        auto s = m[index].upper_bound(snap_id);
        if (s == m[index].begin()) {
            return 0;
        }
        return prev(s)->second;
    }
};
