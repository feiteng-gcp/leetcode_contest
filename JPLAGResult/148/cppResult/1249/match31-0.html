class SnapshotArray {
	vector<vector<pair<int,int>>> v;
	std::set<int> changed;
	vector<int> que;
	int cu;
public:
	SnapshotArray(int length) {
		cu = 0;
		v.resize(0); v.clear();
		v.resize(length + 1);
		que.resize(0); que.clear();
		que.resize(length + 1, -1);
		changed.clear();

		rep(i, 0, length) {
			changed.insert(i);
			que[i] = 0;
		}
	}

	void set(int index, int val) {
		changed.insert(index);
		que[index] = val;
	}

	int snap() {
		fore(i, changed) {
			v[i].push_back({ cu, que[i] });
			que[i] = -1;
		}
		changed.clear();
		cu++;
		return cu - 1;
	}

	int get(int index, int snap_id) {
		auto ite = lower_bound(all(v[index]), make_pair(snap_id, inf));
		ite--;
		return ite->second;
	}
};
