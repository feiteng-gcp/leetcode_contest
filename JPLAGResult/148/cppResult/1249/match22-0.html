class SnapshotArray {
public:
    SnapshotArray(int length) {
        data.resize(length + 2);
        id = 0;
        chgs.clear();
    }
    
    void set(int index, int val) {
        chgs[index] = val;
    }
    
    int snap() {
        for (auto& p : chgs) {
            int x = p.first;
            int v = p.second;
            data[x].emplace_back(id, v);
        }
        chgs.clear();
        
        return id++;
    }
    
    int get(int index, int snap_id) {
        auto& V = data[index];
        
        auto it = upper_bound(V.begin(), V.end(), make_pair(snap_id, INT_MAX));
        if (it == V.begin()) {
            return 0;
        }
        --it;
        return it->second;
    }
    
private:
    vector<vector<pair<int,int>>> data;
    int id;
    map<int, int> chgs;
};

 * int param_3 = obj->get(index,snap_id);
