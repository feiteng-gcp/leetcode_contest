class SnapshotArray {
	vector<map<int, int>> plist;
	int t;
public:
	SnapshotArray(int length) {
		plist.clear();
		plist.resize(length);
		t = 0;
	}

	void set(int index, int val) {
		plist[index][t] = val;
	}

	int snap() {
		return t++;
	}

	int get(int index, int snap_id) {
		if (plist[index].empty()) return 0;
		if (plist[index].find(snap_id) != plist[index].end()) return plist[index][snap_id];
		map<int, int>::iterator it = plist[index].upper_bound(snap_id);
		if (it == plist[index].begin()) return 0;
		it--;
		return it->second;
	}
};
