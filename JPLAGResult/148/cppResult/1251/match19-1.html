class Solution {
    int solve(const string &text, int lo, int hi) {
        if(lo == hi) return 1;
        if(lo > hi) return 0;

        for(int len=1;len<=hi-lo&&lo+len-1<hi-len+1;len++) {
            if(text.substr(lo, len) == text.substr(hi-len+1, len)) return solve(text, lo + len, hi - len) + 2;
        }
        return 1;
    }
public:
    int longestDecomposition(string text) {        
        int n = text.size();
        return solve(text, 0, n - 1);
    }
};
