class Solution {
public:
    int longestDecomposition(string text) {
        string s = text;
        if(s.size() < 2) return s.size();
        int n = s.size();
        int res = 0;
        int l = 0;
        int r = n - 1;
        int l_start = 0;
        int r_end = n - 1;
        string left = "";
        string right = "";
        while(l < r){
             left = s.substr(l_start, l - l_start + 1);
             right = s.substr(r , r_end - r + 1);
            if(left == right){
                res += 2;
                l_start = l + 1;
                r_end = r - 1;
            }
            l++;
            r--;
        }
        if(l_start <= r){
            res++;
        }
        return res;
    }
};
