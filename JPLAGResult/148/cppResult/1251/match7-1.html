class Solution {
    vector<vector<int>> cache;
    string text;
    int dp(int s,int e){
        if(s>e) return 0;
        if(s==e) return 1;
        if(cache[s][e]!=-1) return cache[s][e];

        int ans=1;
        for(int i=1;s+i-1<e-i+1;i++){
            bool same=true;
            for(int j=0;j<i;j++){
                if(text[s+j]!=text[e-i+1+j]){
                    same=false;
                    break;
                }
            }
            if(same){
                ans=max(dp(s+i,e-i)+2,ans);
            }
        }
        cache[s][e]=ans;
        return ans;
    }
public:
    int longestDecomposition(string text) {
        int N=text.size();
        cache=vector<vector<int>>(N,vector<int>(N,-1));
        this->text=text;
        return dp(0,N-1);
    }
};
