class Solution {
public:
    unordered_map<string, int> mem;
    
    int longestDecomposition(string text) {
        if (text.empty()) {
            return 0;
        }
        
        auto [it, inserted] = mem.try_emplace(text, 1);
        if (!inserted) return it->second;
        

        int& res = it->second;
        for (int i = 1; 2 * i <= text.size(); ++i) {
            string a = text.substr(0, i);
            string b = text.substr(text.size() - i);
            if (a == b) {
                res = max(res, 2 + longestDecomposition(text.substr(i, text.size() - 2 * i)));
            }
        }
        return res;
    }
};
