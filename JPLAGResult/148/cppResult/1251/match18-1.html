class Solution {
public:
    int longestDecomposition(string text) {
        return inner_decomposition(text);
    }
    
    int inner_decomposition(const string& text) {
        cout << text << '\n';
        if (text.empty()) {
            return 0;
        }

        const int len = text.size();
        for (int i = 1; i <= len / 2; ++i) {
            if (equal(text.begin(), text.begin() + i, text.end() - i, text.end())) {
                string new_text = text.substr(i, len - i - i);
                return 2 + inner_decomposition(new_text);
            }
        }
        return 1;
    }
};
