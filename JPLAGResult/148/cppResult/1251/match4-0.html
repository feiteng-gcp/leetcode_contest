class Solution {
public:
    int f(string& s, int b, int n){
        if(b == n) return 0;
        int max_l = 0;
        for(int l=1;b+l<=n-l;l++){
            bool is = true;
            for(int i=0;i<l;i++){
                if(s[b+i] != s[n-l+i]){
                    is = false;
                    break;
                }
            }
            if(is) return 2 + f(s, b+l, n-l);
        }
        return 1;
    }
    int longestDecomposition(string text) {
        return f(text, 0, text.size());        
    }
};
