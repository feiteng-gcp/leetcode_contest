class Solution {
public:
    int longestDecomposition(string text) {
        if (text.size() == 0)
            return 0;
        
        if (m_cache.count(text))
            return m_cache[text];
        
        int best = 1;
        for (int l = 1; l * 2 <= text.size(); ++l) {
            if (strncmp(text.c_str(), text.c_str() + text.size() - l, l) == 0) {
                best = std::max(best, longestDecomposition(text.substr(l, text.size() - l * 2)) + 2);
            }
        }
        return best;
    }
    map<string, int> m_cache;
};
