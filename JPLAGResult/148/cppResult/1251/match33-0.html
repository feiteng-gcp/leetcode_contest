class Solution {
    map<pair<int, int>, int> cache;
public:
    int helper(string& t, int i, int j) {
        if (i > j)
            return 0;
        if (i == j)
            return 1;
        auto p = make_pair(i, j);
        if (cache.find(p) != cache.end())
            return cache[p];
        int mid = t.size() / 2;
        char c = t[i];
        int res = 1;
        for (int k = j; k >= mid; --k) {
            if (t[k] == c) {
                int flag = true;
                for (int l = 0; l < j - k + 1; ++l) {
                    if (t[i + l] != t[k + l]) {
                        flag = false;
                        break;
                    }
                }
                if (flag) {
                    res = max(res, 2 + helper(t, i + j - k + 1, k - 1));
                }
            }
        }
        cache[p] = res;
        return res;
    }
    int longestDecomposition(string text) {
        return helper(text, 0, text.size() - 1);
    }
};
