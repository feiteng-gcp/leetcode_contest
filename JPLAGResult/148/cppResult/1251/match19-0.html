class Solution {
private:
    bool same(const deque<char>& a, const deque<char>& b) {
        for (int i=0; i<a.size(); i++) {
            if (a[i] != b[i]) return false;
        }
        return true;
    }
public:
    int longestDecomposition(string text) {
        int chunks = 0;
        int N = text.size();
        bool full = 0;
        deque<char> str1, str2;
        for (int i=0; i<N/2; i++) {
            str1.push_back(text[i]);
            str2.push_front(text[N-i-1]);
            if (same(str1, str2)) {
                chunks += 2;
                str1.clear();
                str2.clear();
                full = i+1 == N-i-1;
            }
        }
        
        return chunks + (full ? 0 : 1);
    }
};
