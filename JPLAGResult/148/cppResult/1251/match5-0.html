class Solution {
public:
	int longestDecomposition(string text) {
		int n = text.size();
		int dp[201];
		memset(dp, -1, sizeof(dp));
		dp[0] = 0;

		for (int i = 0; i < n; i++) {
			if (dp[i] == -1)continue;
			int j = n - i;

			for (int k = 1; i + k <= n && j - k >= 0; k++) {
				bool ok = true;
				for (int x = 0; x < k; x++) {
					if (text[i + x] != text[j - k + x]) {
						ok = false;
						break;
					}
				}
				if (ok) {
					dp[i + k] = max(dp[i + k], dp[i] + 1);
				}
			}
		}

		return dp[n];
	}
};
