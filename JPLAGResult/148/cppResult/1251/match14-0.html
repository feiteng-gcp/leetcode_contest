class Solution {
public:
    vector<vector<int>> dp;
    
    int LD(string &s, int start, int end) {
        if (start > end) return 0;
        if (start == end) return 1;
        if (dp[start][end] > 0) return dp[start][end];
        dp[start][end] = 1;
        string s1, s2;
        for (int i = start; i <= (start + end)/2; ++i) {
            s1 += s[i];
            s2 = s[end - (i - start)] + s2;
            if (s1 == s2) {
                dp[start][end] = max(dp[start][end], 2 + LD(s, i + 1, (end - (i - start) - 1)));
            }
        }
        return dp[start][end];
    }
    
    int longestDecomposition(string text) {
   //     text = "antaprezatepzapreanta";
   //     text = "merchant";
    //    text = "aaa";
        
        dp.resize(text.size(), vector<int>(text.size(), 0));
        
        return LD(text, 0, text.size() - 1);
    }
};
