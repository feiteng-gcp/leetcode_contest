class Solution {
public:
    int longestDecomposition(string text) {

        int l = 0;
        int r = text.length() - 1;
        int ans = 0;
        string ls = "";
        string rs = "";
        while(l <= r) {
            ls += text[l];
            rs += text[r];
            string rev = rs;
            reverse(rev.begin(), rev.end());
            if(ls == rev) {
                if(l == r) {
                    ans++;
                }
                else {
                    ans += 2;
                }

                ls = "";
                rs = "";
            }
            l++;
            r--;
        }
        if(ls != "" || rs != "") {
            ans++;
        }
        return ans;
    }
};
