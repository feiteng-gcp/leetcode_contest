class Solution {
public:
    int longestDecomposition(string text) {
      int n = text.size();
      map<int, int> memo;
      function<int(int)> solve = [&](int ind) -> int {
        if (ind > (n-1)/2) return 0;
        for (int i = 0, j = n - ind - 1; ind + i < j; ++i, --j) {
          if (text.substr(ind, i+1) == text.substr(j, i+1)) {
            return 2 + solve(ind + i + 1);
          }
        }
        return 1;
      };
      return solve(0);
    }
};
