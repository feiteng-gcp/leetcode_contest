class Solution {
public:

	int solve(string &text, int b, int e){
		
		
		
		//cout<<"b = "<<b<<" e = "<<e<<endl;
		
		if(b >= e) return 0;
		
		//for(int i = b; i < e; i++){
		//	cout<<text[i];
		//}
		//cout<<endl;
		
		for(int i = b; i < e; i++){
			if(e - b < (i - b + 1) * 2) break;
			//cout<<"i = "<<i<<" "<<e - b <<" < "<< (i - b + 1) * 2<<endl;

			bool same = true;
			
			for(int j = 0; j <= i-b; j++){
				if(text[j+b] != text[e-(i-b)-1+j]){
					same = false;
					break;
				}
			}
			
			if(same){
				//cout<<"SAME "<<endl;
				return solve(text, i+1, e-(i-b)-1) + 2;
			}
		}
		
		return 1;
	}
    int longestDecomposition(string text) {
        return solve(text, 0, text.size());
    }
};
