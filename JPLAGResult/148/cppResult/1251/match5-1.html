class Solution {
public:
    int longestDecomposition(string text) {
        if(text.size()==0) return 0;
        int ll = 1;

        int state = 1;
        for(int ll = 1; ll <= text.size()/2; ll++){
            state = 1;
            for(int i = 0; i < ll; i++){
                if(text[i] != text[text.size()-ll+i]){
                    state = 0; break;
                }
            }
            if(state){
                return longestDecomposition(text.substr(ll,text.size()-ll*2))+2;
            }
        }
        return 1;
    }
};
