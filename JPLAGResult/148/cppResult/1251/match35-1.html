class Solution {
public:
    int longestDecomposition(string text) {
        int ret = 0;
        int p = 0;
        int len = text.size();
        
        while (p * 2 < len) {   
            int ch = text[p], i;
            for (i = len-1-p; i >= (len+1)/2; --i) {
                int F = 1;
                for (int j = i; j <= len-1-p; ++j) {
                    if (text[p+j-i] != text[j]) {
                        printf("%d:%c %d:%c\n", p+j-i, text[p+j-i], j, text[j]);
                        F = 0;
                        break;
                    }
                }
                if (F) {
                    ret += 2;
                    p += len-p-i;
                    printf("%d %d\n", ret, p);
                    break;
                }
            }
            if (i < (len+1)/2) return ret+1;
        }
        return ret;
    }
};
