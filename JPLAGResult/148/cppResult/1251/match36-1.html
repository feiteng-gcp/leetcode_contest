class Solution {
private:
    bool eq(string& text, int s1, int e1, int s2, int e2){

        for(int i=s1, j=s2; i<=e1; i++, j++){
            if(text[i]!=text[j]){
                return false;
            }
        }
        return true;
    }
public:
    int longestDecomposition(string text) {
        int s=0, e=text.size()-1;
        int k = 0;
        while(s<=e){
            bool f = false;
            for(int len=1; len<=(e-s+1)/2; len++){
                if(eq(text, s, s+len-1, e-len+1, e)){
                    f=true;
                    k+=2;
                    s=s+len;
                    e=e-len;
                    break;
                }
            }
            if(!f){
                return k+1;
            }
        }
        return k;
    }
};
