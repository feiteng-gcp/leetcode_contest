int memo[201][201];

class Solution {
public:
    int dfs(int i, int j, string& s)
    {
        if (i == j) return 1;
        if (i > j) return 0;

        if (memo[i][j] != -1) return memo[i][j];
        int ans = 1;
        for (int l = 1; i + l - 1 < j - l + 1; ++l)
        {
            bool same = true;
            for (int k = 0; same && k < l; ++k)
            {
                if (s[i + k] != s[j - l + 1 + k])
                    same = false;
            }
            if (same)
            {
                ans = max(ans, dfs(i + l, j - l, s) + 2);
            }
        }
        return memo[i][j] = ans;
    }

    int longestDecomposition(string s) {
        memset(memo, -1, sizeof(memo));
        return dfs(0, s.size() - 1, s);
    }
};
