class Solution {
public:
    int longestDecomposition(string text) {
        memset(dp_, 0, sizeof(int) * 200 * 200);
        return dp(text, 0, text.size() - 1);
    }
    
    int dp(const string& text, int l, int r) {
        if (l == r) {
            return 1;
        }

        if (l > r) {
            return 0;
        }
        if (dp_[l][r]) {
            return dp_[l][r];
        }
        
        int ans = 1;
        for (int len = 1; l + len - 1 < r - len + 1; len++) {
            if (text.substr(l, len) == text.substr(r - len + 1, len)) {
                ans = max(ans, 2 + dp(text, l + len, r - len));
            }
        }
        
        return dp_[l][r] = ans;
    }
    
private:
    int dp_[200][200];
};
