class Solution {
public:
    int longestDecomposition(string text) {
        if (text == "") {
            return 0;
        }
        // cout<<text<<endl;
        for (int i = 1; i*2 <= text.length(); i++) {
            if (text.substr(0, i) == text.substr(text.length() - i, text.length())) {
                // cout<<"found: "<<text.substr(0, i)<<endl;
                return 2 + longestDecomposition(text.substr(i, text.length() - 2*i));
            }
        }
        return 1;
    }
};
