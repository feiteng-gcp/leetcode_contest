class Solution {
public:
    int longestDecomposition(string text) {
        if (text.length() == 0)
            return 0;
        int count = 0;
        int left = 0, right = (int)text.length()-1;

        int prevl = left, prevr = right;
        while (left < right)
        {
            string strleft = text.substr(prevl,left-prevl+1);
            string strright = text.substr(right,prevr-right+1);
            if (strleft == strright)
            {
                //cout << strleft << endl;
                prevl = left+1;
                prevr = right-1;
                count += 2;
            }
            left++;
            right--;
        }
        //cout << prevl << " " << prevr << endl;
        if (prevl <= prevr)
            count += 1;
        return count;
    }
};
