class Solution {
public:
    int calcu(string &s, int l, int r){
        //cout<<l<<" : "<<r<<endl;
        if(l == r)
            return 1;
        if(l > r)
            return 0;
        int back = r;
        int half = l + (r - l) / 2;
        //cout<<"half: "<<half<<back<<endl;
        while(back > half){
            if(s[back] == s[l]){
                int cap = l + r - back;
                bool suc = true;
                for(int j = l; j <= cap; ++j){
                    if(s[j] != s[back + j - l]){
                        suc = false;
                        break;
                    }
                }
                if(suc)
                    return 2 + calcu(s, r - back + 1 + l, back - 1);
            }
            --back;
        }
        
        return 1;
    }
    
    
    int longestDecomposition(string text) {
        int res = calcu(text, 0, text.size() - 1);
        return res;
        
        
        
        
    }
};
