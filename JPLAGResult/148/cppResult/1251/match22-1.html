class Solution {
public:
    int longestDecomposition(string t) {
        int k = 0, i, lenx;
        for (i = 0, lenx = t.size()-1; i*2 < t.size(); ) {
            bool flag = false;
            for (int j = 0; i + j < lenx - i - j; j++) {
                // cout << j << "," << lenx - i - j << endl;
                int cnt = 0;
                while(j >= cnt && t[i+cnt] == t[lenx-i-j+cnt]) cnt++;
                if (j == cnt-1) {

                    i += cnt, k+=2;
                    // cout << i << cnt << endl;
                    flag = true;
                    break;
                }
            }
            if (!flag) break;
        }
        if (i*2 < t.size()) k++;
        return k;
    }
};
