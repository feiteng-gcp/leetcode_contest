class Solution {
public:
    int longestDecomposition(string text) {
        int L = text.length();
        if (L == 1) {
            return 1;
        }
        int max_num[L / 2], ans = 1;
        for (int i = 0; i < L / 2; i++) {

            max_num[i] = -1E3;
            for (int j = 0; j <= i; j++) {
                bool same = true;
                for (int k = j; k <= i && same; k++) {
                    if (text[k] != text[L - 1 - i - j + k]) {
                        same = false;
                    }
                }
                if (same) {
                    int cur_num = (j == 0) ? 1 : max_num[j - 1] + 1;
                    max_num[i] = max(max_num[i], cur_num);
                }
            }
            int total = max_num[i] * 2;
            if ((i + 1) * 2 != L) {
                total++;
            }
            ans = max(ans, total);
        }
        return ans;
    }
};
