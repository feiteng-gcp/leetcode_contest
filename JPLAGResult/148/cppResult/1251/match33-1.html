class Solution {
public:
    int longestDecomposition(string text) {
        if (text.size() == 0)
            return 0;
        if (text.size() == 1)
            return 1;

        for (int i = text.size() - 1; i >= 0; i--) {
            int flag = 1;
            for (int j = 0; j < text.size() - i; j++) {
                if (text[i + j] != text[j]) {
                    flag = 0;
                    break;
                }
            }
            if (flag == 1) {
                if (2 * (text.size() - i) > text.size())
                    return 1;
                return longestDecomposition(string(text.begin() + text.size() - i, text.end() - text.size() + i)) + 2;
            }
        }
        return 1;
    }
};
