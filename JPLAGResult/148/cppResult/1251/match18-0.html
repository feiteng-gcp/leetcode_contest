class Solution {
public:
    int solve(string s, string cur, int cnt, int len) {
        int a = s.size(), b = cur.size();
        if(b == 0) return cnt;
        if(b <= 1) return (cnt > 0 && a - len <= 1) ? cnt + 1 : 1;
        
        bool yes = false;
        for(int i = 0; i < b / 2; ++i) {
            //cout << cur.substr(0, i + 1) << " " << cur.substr(b - i - 1, i + 1) << endl;
            if(cur.substr(0, i + 1) == cur.substr(b - i - 1, i + 1)) {
                yes = true;
                return solve(s, cur.substr(i + 1, b - 2 * (i + 1)), cnt + 2, len + (i + 1) * 2);
            }
        }
        
        return cnt + 1;
    }
    int longestDecomposition(string text) {
        return solve(text, text, 0, 0);
    }
};
