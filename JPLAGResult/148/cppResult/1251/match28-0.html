class Solution {
private:
    vector<int>f;
    int n;
    int dfs(int l,int r, string& s){
        if(l==r)return 1;
        if(l>r)return 0;
        if(f[l]!=-1)return f[l];
        int ret = 1;
        for(int i=l;r-(i-l)>i;i++){
            bool flag = true;
            int st = r - (i-l);
            for(int j = l;j<=i;j++,st++){
                if(s[j]!=s[st]){
                    flag =false;
                    break;
                }
            }
            if(flag)ret=max(ret,dfs(i+1,r-(i-l)-1,s)+2);
        }
        return f[l]=ret;
    }
public:
    int longestDecomposition(string s) {
        n = s.size();
        f = vector<int>(n,-1);
        return dfs(0,n-1,s);
    }
};
