class Solution {
public:
    int longestDecomposition(string text) {
        vector<vector<int>> code(text.length(), vector<int>(text.length(), -1));
        {
            unordered_map<string, int> strings;
            for (int i = 0; i < text.length(); i++) {
                for (int l = 1; i + l <= text.length(); l++) {
                    const auto& str = text.substr(i, l);
                    auto iter = strings.find(str);
                    if (iter == strings.end()) {
                        strings.insert({str, strings.size()});
                        code[i][i + l - 1] = strings.size() - 1;
                    } else {
                        code[i][i + l - 1] = iter->second;
                    }
                }
            }
        }
        return func(code, 0, text.length());
    }
private:
    int func(const vector<vector<int>>& code, int s, int e) {
        if (e - s < 1) {
            return 0;
        }
        if (e - s == 1) {
            return 1;
        }
        int result = 1;
        for (int l = 1; s + l <= e - l; l++) {
            if (code[s][s + l - 1] == code[e - l][e - 1]) {
                result = max(result, 2 + func(code, s + l, e - l));
            }
        }
        return result;
    }
};
