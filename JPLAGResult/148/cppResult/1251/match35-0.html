class Solution {

public:
    //int helper(std::string& text, int left, int right)
    
    int longestDecomposition(string text) {
        int left=0;
        int right=text.length()-1;
        
        int res=0;
        
        while(left<=right){
            ++res;
            int len=1;
            
            while(text.substr(left,len)!=text.substr(right-len+1,len)){
                ++len;
            }
            
            if(len!=right-left+1){
                ++res;
            }
            
            left+=len;
            right-=len;
        }
        
        return res;
    }
};
