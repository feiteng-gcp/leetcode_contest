class Solution {
public:
    int longestDecomposition(string t) {
        int n = t.size();
        if (n == 0) return 0;
        int l = 0, r = n - 1;
        int cnt = 0;
        while (r > l) {
            int len = r - l + 1;

            bool suc = false;
            for (int k = 1; k <= len / 2; ++k) {
                if (t.substr(l, k) == t.substr(r - k + 1, k)) {
                    cnt += 2;
                    l += k;
                    r -= k;
                    suc = true;
                    break;
                }
            }
            if (!suc) {
                ++cnt;
                break;
            }
        }
        if (l == r)
            ++cnt;
        return cnt;
    }
};
