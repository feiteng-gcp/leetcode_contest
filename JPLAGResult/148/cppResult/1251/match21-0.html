class Solution {
public:
  int longestDecomposition(string text) {
    int n = text.size();
    int tot = 0;
    vector<int> dp(n);
    int l = 0, r = text.size();
    while (l < r) {
      bool found = false;
      for (int i = 1; l + i <= r - i; i++) {
        if (text.substr(l, i) == text.substr(r - i, i)) {
          found = true;
          l+=i, r-=i;
          tot+=2;
          break;
        }
      }
      if (!found) break;
    }
    if (l < r)tot++;
    return tot;
  }
};
