class Solution {
public:
    unordered_map<string, int> memo;
    
    int helper(string s, int l, int r) {
        if (l > r) return 0;
        

        int maxx = 1;
        if (memo.find(s.substr(l, r - l + 1)) != memo.end())
            return memo[s.substr(l, r - l + 1)];
        

        for (int len = 1; len <= (r - l + 1) / 2; len++) {
            if (s.substr(l, len) == s.substr(r - len + 1, len)) {
                maxx = max(maxx, 2 + helper(s, l + len, r - len));
            }
        }
        memo[s.substr(l, r-l+1)] = maxx;
        return maxx;
    }
    
    int longestDecomposition(string text) {
        return helper(text, 0, text.length() - 1);
    }
};
