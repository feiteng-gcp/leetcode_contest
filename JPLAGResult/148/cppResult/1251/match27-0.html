class Solution {
public:
    int longestDecomposition(string text) {
        int res = 0;
        int n = text.size();
        int lf = 0, rt = n;
        bool flag = true;
        while (flag) {
            int len = rt - lf;
            flag = false;
            for (int i = 1; i + i <= len; ++i) {
                if (text.substr(lf, i) == text.substr(rt-i, i)) {
                    flag = true;
                    lf = lf+i;
                    rt = rt-i;
                    break;
                }
            }
            if (flag) res += 2;
            else if (len > 0) res += 1;
        }
        return res;
    }
};
