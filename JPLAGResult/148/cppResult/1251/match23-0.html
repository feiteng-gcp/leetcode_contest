class Solution {
public:
    int go(string s,int x,int y) {
        if(x>=y) return 0;
        for(int i = x; i < x + (y-x)/2; i++) {
            if(s[i] == s[y-1]) {
                int a,b;
                a = i;
                b = y-1;
                for(;a>x && s[a]==s[b];) {
                    a--;
                    b--;
                }
                if(a==x && s[a]==s[b]) {
                    // cout << s.substr(x,i+1-x) << endl;
                    return go(s,i+1,y-(i+1-x)) + 2;
                }
            }
        }
        return 1;
    }
    int longestDecomposition(string text) {
        return go(text,0,text.length());
    }
};
