class Solution {
public:
    int longestDecomposition(string t) {
        vector<vector<int> > dp(t.size(), vector<int>(t.size()));
        return split(0, t.size() - 1, t, dp);
    }
    int split(int i, int j, string& t, vector<vector<int> >& dp) {
        if (dp[i][j]) return dp[i][j];
        if (i > j) return 0;
        int n = (j - i + 1) / 2, res = 1;
        for (int k = 1; k <= n; k++) {
            if (t.substr(i, k) == t.substr(j - k + 1, k)) {
                res = max(res, split(i + k, j - k, t, dp) + 2);
            }
        }
        dp[i][j] = res;
        return res;
    }
};
