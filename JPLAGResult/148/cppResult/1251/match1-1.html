  class Solution {
  public:
	  string S;
	  int N;
	  int solve(string curr, int ct, int k) {
		  if (curr.size() == 0)return ct;
		  if (curr.size() == 1) {
			  if (ct > 0 && S.size() - k <= 1)return ct + 1;

			  return 1;
		  }
		  for (int i = 0; i < curr.size() / 2; i++) {
			  if (curr.substr(0, i + 1) == curr.substr(curr.length() - 1 - i)) {
				  return solve(curr.substr(i + 1,curr.length()-2*i-2), ct + 2, (i + 1) * 2 + k);
			  }
		  }
		  return ct+1;
	  }
	  int longestDecomposition(string text) {
		  S = text;
		  N = S.length();
		  return solve(text, 0, 0);
	  }
  };
