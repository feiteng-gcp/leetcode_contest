class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int sum1 = 0;
        auto temp = nums;
        for (int i = 0; i < nums.size(); i += 2) {
            int d = 0;
            if (i - 1 >= 0 && nums[i - 1] <= nums[i]) {
                d += nums[i] - nums[i - 1] + 1;
                nums[i] = nums[i - 1] - 1;
            }
            if (i + 1 < nums.size() && nums[i + 1] <= nums[i]) {
                d += nums[i] - nums[i + 1] + 1;
                nums[i] = nums[i + 1] + 1;
            }
            sum1 += d;
        }
        int sum2 = 0;
        nums = temp;
        for (int i = 1; i < nums.size(); i += 2) {
            int d = 0;
            if (i - 1 >= 0 && nums[i - 1] <= nums[i]) {
                d += nums[i] - nums[i - 1] + 1;
                nums[i] = nums[i - 1] - 1;
            }
            if (i + 1 < nums.size() && nums[i + 1] <= nums[i]) {
                d += nums[i] - nums[i + 1] + 1;
                nums[i] = nums[i + 1] + 1;
            }
            sum2 += d;
        }
        return min(sum1, sum2);
    }
};
