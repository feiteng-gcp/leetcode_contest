class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {

        int N = nums.size(), even_move = 0, odd_move = 0;
        for (int i = 1; i < N; i += 2) {
            int min_adjacent = nums[i - 1];
            if (i < (N - 1)) {
                min_adjacent = min(min_adjacent, nums[i + 1]);
            }
            if (nums[i] > (min_adjacent - 1)) {
                even_move += nums[i] - (min_adjacent - 1);
            }
        }

        for (int i = 0; i < N; i += 2) {
            int min_adjacent = 1000;
            if (i > 0) {
                min_adjacent = min(min_adjacent, nums[i - 1]);
            }
            if (i < (N - 1)) {
                min_adjacent = min(min_adjacent, nums[i + 1]);
            }
            if (nums[i] > (min_adjacent - 1)) {
                odd_move += nums[i] - (min_adjacent - 1);
            }
        }
        return min(even_move, odd_move);
    }
};
