class Solution {
public:
    int movesToMakeZigzag(vector<int>& a) {
        if(a.size()<3)return 0;
        int ret=1e9;
        int num=0;
        vector<int>b;

        for(int i=0;i<a.size();i++)
            b.push_back(a[i]);
        for(int i=1;i<a.size();i+=2){
            if(i&&a[i]<=a[i-1]){num+=a[i-1]-a[i]+1;a[i-1]=a[i]-1;}
            if(i+1<a.size()&&a[i]<=a[i+1]){num+=a[i+1]-a[i]+1;a[i+1]=a[i]-1;}
        }

        ret=num;
        num=0;
        for(int i=0;i<b.size();i+=2){
            if(i&&b[i]<=b[i-1]){num+=b[i-1]-b[i]+1;b[i-1]=b[i]-1;}
            if(i+1<b.size()&&b[i]<=b[i+1]){num+=b[i+1]-b[i]+1;b[i+1]=b[i]-1;}
        }
        ret=min(ret,num);
        return ret;
    }
};
