class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int even=0,odd=0;
        int n=nums.size();
        

        std::vector<int> orig=nums;
        
        for(int i=0;i<n;i+=2){
            int b=0;
            
            if(i-1>=0 && nums[i]<=nums[i-1]){
                b+=(nums[i-1]-nums[i]+1);
                nums[i-1]=nums[i]-1;
            }
            
            if(i+1<n && nums[i]<=nums[i+1]){
                b+=(nums[i+1]-nums[i]+1);
                nums[i+1]=nums[i]-1;
            }
            
            even+=b;
        }
        

        nums=orig;
        
        for(int i=1;i<n;i+=2){
            int b=0;
            
            if(i-1>=0 && nums[i]<=nums[i-1]){
                b+=(nums[i-1]-nums[i]+1);
                nums[i-1]=nums[i]-1;
            }
            
            if(i+1<n && nums[i]<=nums[i+1]){
                b+=(nums[i+1]-nums[i]+1);
                nums[i+1]=nums[i]-1;
            }
            
            odd+=b;
        }
        
        return std::min(even,odd);
    }
};
