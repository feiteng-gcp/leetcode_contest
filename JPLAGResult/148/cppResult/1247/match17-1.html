class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {

        int res = 0, ans = 0x3F3F3F3F;
        for (int i = 1; i < nums.size(); i += 2) {
            int cur = nums[i - 1];
            if (i + 1 < nums.size()) {
                cur = min(cur, nums[i + 1]);
            }
            if (nums[i] >= cur) {
                res += nums[i] - cur + 1;
            }
        }
        ans = min(ans, res);
        res = 0;

        for (int i = 0; i < nums.size(); i += 2) {
            int cur = 0x3F3F3F3F;
            if (i - 1 >= 0) {
                cur = min(cur, nums[i - 1]);
            }
            if (i + 1 < nums.size()) {
                cur = min(cur, nums[i + 1]);
            }
            if (nums[i] >= cur) {
                res += nums[i] - cur + 1;
            }
        }
        ans = min(ans, res);
        return ans;
    }
};
