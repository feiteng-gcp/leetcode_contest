class Solution {
public:
    int movesToMakeZigzag(vector<int>& a) {
        int ans = 1000000009;
        for(int k = 0; k <= 1; ++k) {
            int cnt = 0;
            for(int i = k; i < a.size(); i += 2) {
                int mi = 1000000009;
                if(i - 1 >= 0) mi = min(mi, a[i-1]);
                if(i + 1 < a.size()) mi = min(mi, a[i+1]);
                cnt += max(0, a[i] - (mi - 1));
            }
            ans = min(ans, cnt);
        }
        return ans;
    }
};
