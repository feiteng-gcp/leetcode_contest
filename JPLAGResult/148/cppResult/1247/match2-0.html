class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
      int over = 0;
      int below = 0;
      for (int i = 0; i < nums.size(); i += 2) {
        int dec = 0;
        if (i > 0)  {
          dec = max(dec, nums[i] - nums[i-1] + 1);
        }
        if (i < nums.size() - 1) {
          dec = max(dec, nums[i] - nums[i+1] + 1);
        }
        below += dec;
      }
      for (int i = 1; i < nums.size(); i += 2) {
        int dec = 0;
        if (i > 0)  {
          dec = max(dec, nums[i] - nums[i-1] + 1);
        }
        if (i < nums.size() - 1) {
          dec = max(dec, nums[i] - nums[i+1] + 1);
        }
        over += dec;
      }
      
      return min(over, below);
    }
};
