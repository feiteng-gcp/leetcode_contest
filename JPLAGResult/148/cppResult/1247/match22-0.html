class Solution {
public:
    int movesToMakeZigzag(vector<int>& a) {
        int n = a.size();
        int ans0 = 0;
        for (int i = 1; i < n; i += 2) {
            int target = min(a[i - 1], i + 1 >= n ? 10001 : a[i + 1]) - 1;
            if (a[i] > target)
                ans0 += a[i] - target;
        }
        
        int ans1 = 0;
        for (int i = 0; i < n; i += 2) {
            int target = min(i - 1 < 0 ? 10001 : a[i - 1], i + 1 >= n ? 10001 : a[i + 1]) - 1;
            if (a[i] > target)
                ans1 += a[i] - target;
        }
        
        return min(ans0, ans1);
    }
};
