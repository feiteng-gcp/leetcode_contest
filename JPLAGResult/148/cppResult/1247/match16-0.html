class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int n = nums.size();
        int ans1 = 0, ans2 = 0;
            for (int i = 1; i < n; i += 2) {
                int temp = nums[i - 1];
                if (i < n - 1) {
                    temp = min(nums[i + 1], temp);
                }
                if (nums[i] >= temp) {
                    ans1 += nums[i] - temp + 1;
                }
            }
            for (int i = 0; i < n; i += 2) {
                int temp = ~0u >> 1;
                if (i != 0) {
                    temp = min(temp, nums[i - 1]);
                }
                if (i != n - 1) {
                    temp = min(temp, nums[i + 1]);
                }
                if (nums[i] >= temp) {
                    ans2 += nums[i] - temp + 1;
                }
            }
        return min(ans1, ans2);
    }
};
