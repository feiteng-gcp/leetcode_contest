class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int ans1 = 0;
        for (int i = 0; i < nums.size(); i += 2) {
            int target = nums[i] + 1;
            if (i - 1 >= 0 && nums[i - 1] < target) {
                target = nums[i - 1];
            }
            if (i + 1 < nums.size() && nums[i + 1] < target) {
                target = nums[i + 1];
            }
            ans1 += nums[i] - (target - 1);
        }
        
        int ans2 = 0;
        for (int i = 1; i < nums.size(); i += 2) {
            int target = nums[i] + 1;
            if (i - 1 >= 0 && nums[i - 1] < target) {
                target = nums[i - 1];
            }
            if (i + 1 < nums.size() && nums[i + 1] < target) {
                target = nums[i + 1];
            }
            ans2 += nums[i] - (target - 1);
        }
        
        return min(ans1, ans2);
    }
};
