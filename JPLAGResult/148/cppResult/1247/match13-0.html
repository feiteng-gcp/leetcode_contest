class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int result1 = 0, result2 = 0;
        for(int i = 0; i < nums.size(); i += 2){
            int temp = 1001;
            if(i != 0) temp = min(temp, nums[i - 1]);
            if(i < nums.size() - 1) temp = min(temp, nums[i + 1]);
            result1 += nums[i] - min(nums[i], temp - 1);
        }
        for(int i = 1; i < nums.size(); i += 2){
            int temp = 1001;
            if(i != 0) temp = min(temp, nums[i - 1]);
            if(i < nums.size() - 1) temp = min(temp, nums[i + 1]);
            result2 += nums[i] - min(nums[i], temp - 1);
        }
        return min(result1, result2);
        
    }
};
