class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {

        int res1 = 0;
        for(int i = 1; i < nums.size(); i += 2){
            int x = nums[i - 1];
            if(i + 1 < nums.size()) x = min(x, nums[i + 1]);
            x --;
            res1 += max(nums[i] - x, 0);
        }
//        cout << res1 << endl;

        int res2 = 0;

        for(int i = 0; i < nums.size(); i += 2){
            int x = INT_MAX;
            if(i - 1 >= 0) x = min(x, nums[i - 1]);
            if(i + 1 < nums.size()) x = min(x, nums[i + 1]);
            x --;
            res2 += max(nums[i] - x, 0);
        }
//        cout << res2 << endl;

        return min(res1, res2);
    }
};
