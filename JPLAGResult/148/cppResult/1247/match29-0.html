class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int result;
        int ra=0;
        
        vector<int> cnums = nums;
        
        for(int i=0;i<nums.size();i+=2)
        {
            if(i-1>=0 && nums[i-1]>=nums[i]){
                ra+=(nums[i-1]-nums[i]+1);
                nums[i-1] = nums[i] - 1;
            }
            if(i+1<nums.size() && nums[i+1]>=nums[i]){
                ra+=(nums[i+1]-nums[i]+1);
                nums[i+1] = nums[i] - 1;
            }
        }
        
        result = ra;
        int rb = 0;
        
        nums = cnums;
        
        for(int i=1;i<nums.size();i+=2)
        {
            if(i-1>=0 && nums[i-1]>=nums[i]){
                rb+=(nums[i-1]-nums[i]+1);
                nums[i-1] = nums[i] - 1;
            }
            if(i+1<nums.size() && nums[i+1]>=nums[i]){
                rb+=(nums[i+1]-nums[i]+1);
                nums[i+1] = nums[i] - 1;
            }
        }
        
        result = min(result, rb);
        return result;
    }
};
