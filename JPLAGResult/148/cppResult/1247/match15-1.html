class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int res = INT_MAX;
        int ttl = 0;
        for (int i = 1; i < nums.size(); i += 2) {
            int tmp = 0;
            tmp = max(nums[i] - nums[i - 1] + 1, tmp);
            if (i < nums.size() - 1) {
                tmp = max(nums[i] - nums[i + 1] + 1, tmp);
            }
            ttl += tmp;
        }
        res = min(ttl, res);
        ttl = 0;

        for (int i = 0; i < nums.size(); i += 2) {
            int tmp = 0;
            if (i > 0) {
                tmp = max(nums[i] - nums[i - 1] + 1, tmp);
            }
            if (i < nums.size() - 1) {
                tmp = max(nums[i] - nums[i + 1] + 1, tmp);
            }
            ttl += tmp;
        }
        return min(ttl, res);
    }
};
