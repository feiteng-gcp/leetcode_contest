class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        // fix odd, and even < odd

        int res1 = 0;
        for (int i = 0; i < nums.size(); i+=2) {
            int minVal = nums[i];
            if (i-1 >= 0) {
                minVal = min(minVal, nums[i-1]-1);
            } 
            if (i+1 < nums.size()) {
                minVal = min(minVal, nums[i+1]-1);
            }
            res1 += nums[i] - minVal;
        }
        // fix odd, and even > odd
        bool res2 = true;
        for (int i = 0; i < nums.size(); i+=2) {
            if (i-1 >= 0 && nums[i] <= nums[i-1]) {
                res2 = false;
                break;
            }
            if (i+1 < nums.size() && nums[i] <= nums[i+1]) {
                res2 = false;
                break;
            }
        }
        if (res2)
            return 0;
        // fix even, and even < odd
        bool res3 = true;
        for (int i = 1; i < nums.size(); i+=2) {
            if (i-1 >= 0 && nums[i] <= nums[i-1]) {
                res3 = false;
                break;
            }
            if (i+1 < nums.size() && nums[i] <= nums[i+1]) {
                res3 = false;
                break;
            }
        }
        if (res3)
            return 0;
        // fix even, and even > odd
        int res4 = 0;

        for (int i = 1; i < nums.size(); i+=2) {
            int minVal = nums[i];
            if (i-1 >= 0) {
                minVal = min(minVal, nums[i-1]-1);
            } 
            if (i+1 < nums.size()) {
                minVal = min(minVal, nums[i+1]-1);
            }
            res4 += nums[i] - minVal;
        }
        return min(res1, res4);
    }
};
