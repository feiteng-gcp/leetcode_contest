class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        auto tmp = nums;
        int a = 0;
        for (int i = 1; i < nums.size(); i+= 2) {
            int add = 0;
            if (i + 1 < nums.size())
                add = max(add, nums[i] - nums[i + 1] + 1);
            if (i)
                add = max(add, nums[i] - nums[i - 1] + 1);
            a += add;
        }
        int b = 0;
        for (int i = 0; i < nums.size(); i+= 2) {
            int add = 0;
            if (i + 1 < nums.size())
                add = max(add, nums[i] - nums[i + 1] + 1);
            if (i)
                add = max(add, nums[i] - nums[i - 1] + 1);
            b += add;
        }
        return min(a, b);
    }
};
