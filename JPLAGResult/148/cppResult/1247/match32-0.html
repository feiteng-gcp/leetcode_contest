class Solution {
public:


    int movesToMakeZigzag(vector<int>& nums) {
		int ret = 0;	
		for (int i = 1; i < sz(nums); i += 2) {
			int v = nums[i - 1];		
			if (i + 1 < sz(nums)) v = min(v, nums[i + 1]);
			if (nums[i] >= v) {
				ret += nums[i] - (v - 1);
			}
		}
		int ans = 0;
		for (int i = 0; i < sz(nums); i += 2) {
			int v = INT_MAX;
			if (i + 1 < sz(nums)) v = min(v, nums[i + 1]);            
			if (0 <= i - 1) v = min(v, nums[i - 1]);
			if (nums[i] >= v) {
				ans += nums[i] - (v - 1);
			}
		}
		return min(ret, ans);
    }
};
