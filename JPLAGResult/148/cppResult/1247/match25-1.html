class Solution {

public:
    int movesToMakeZigzag(vector<int>& nums) {
        int ans = 0;
        for(int i = 0; i<nums.size(); i+=2){
            int minus = 0;
            if(i>0 && nums[i]>=nums[i-1]) {
                minus = max(minus, nums[i]-nums[i-1]+1);
            } 
            if(i+1<nums.size() && nums[i]>=nums[i+1]) {
                minus = max(minus, nums[i]-nums[i+1]+1);
            }
            ans += minus;

        }
        int ans1 = 0;
        for(int i = 1; i<nums.size(); i+=2){
            int minus = 0;
            if(i>0 && nums[i]>=nums[i-1]) {
                minus = max(minus, nums[i]-nums[i-1]+1);
            } 
            if(i+1<nums.size() && nums[i]>=nums[i+1]) {
                minus = max(minus, nums[i]-nums[i+1]+1);
            }
            ans1 += minus;
        }
        return min(ans, ans1);
    }
};
