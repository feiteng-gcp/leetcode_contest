class Solution {
public:
    // int f[1002][2][2];
    int movesToMakeZigzag(vector<int>& a) {
        int n = a.size(), ans = 0, sum = 0;
        for (int i = 1; i < n; i += 2) {
            int tmp = 0;
            if (a[i] >= a[i-1])
                tmp = a[i] - a[i-1] + 1;
            if (i < n-1 && a[i] >= a[i+1])
                tmp = max(tmp, a[i] - a[i+1] + 1);
            ans += tmp;
        }
        
        for (int i = 0; i < n; i += 2) {
            int tmp = 0;
            if (i > 0 && a[i] >= a[i-1])
                tmp = a[i] - a[i-1] + 1;
            if (i < n-1 && a[i] >= a[i+1])
                tmp = max(tmp, a[i] - a[i+1] + 1);
            sum += tmp;
        }
        
        return min(sum, ans);
        // return min(f[n-1][0][0], f[n-1][0][1], );
    }
};
