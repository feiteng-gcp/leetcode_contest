class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int m1 = 0, m2 = 0;
        for (int i = 1; i < nums.size(); i += 2) {
            int mn = min(nums[i-1], i+1 == (int)nums.size() ? INT_MAX : nums[i+1]);

            if (mn <= nums[i]) m1 += nums[i] - mn + 1;
        }
        
        for (int i = 0; i < nums.size(); i += 2) {
            int mn = min(i > 1 ? nums[i-1] : INT_MAX, i+1 == (int)nums.size() ? INT_MAX : nums[i+1]);
            if (mn <= nums[i]) m2 += nums[i] - mn + 1;
        }
        
        return min(m1, m2);
    }
};
