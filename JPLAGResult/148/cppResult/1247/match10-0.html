class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int ans1 = 0, ans2 = 0;
        for (int i = 1; i < nums.size(); i += 2) {
            int cur = max(0, nums[i] - nums[i-1] + 1);
            if (i + 1 < nums.size()) cur = max(cur, nums[i] - nums[i+1] + 1);
            ans1 += cur;
        }
        for (int i = 0; i < nums.size(); i += 2) {
            int cur = 0;
            if (i > 0) cur = max(cur, nums[i] - nums[i-1] + 1);
            if (i + 1 < nums.size()) cur = max(cur, nums[i] - nums[i+1] + 1);
            ans2 += cur;
        }
        return min(ans1, ans2);
    }
};
