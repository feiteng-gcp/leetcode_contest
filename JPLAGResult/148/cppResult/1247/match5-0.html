class Solution {
public:
	int movesToMakeZigzag(vector<int>& nums) {
		int n = nums.size(), i, j, k, ans1 = 0, ans2 = 0, x;

		for (i = 0; i < n; i += 2) {
			x = 0;
			if ((i > 0) && (nums[i] >= nums[i - 1])) x = max(x, nums[i] - (nums[i - 1] - 1));
			if ((i < n - 1) && (nums[i] >= nums[i + 1])) x = max(x, nums[i] - (nums[i + 1] - 1));
			ans1 += x;
		}

		for (i = 1; i < n; i += 2) {
			x = 0;
			if ((i > 0) && (nums[i] >= nums[i - 1])) x = max(x, nums[i] - (nums[i - 1] - 1));
			if ((i < n - 1) && (nums[i] >= nums[i + 1])) x = max(x, nums[i] - (nums[i + 1] - 1));
			ans2 += x;
		}

		return min(ans1, ans2);
	}
};
