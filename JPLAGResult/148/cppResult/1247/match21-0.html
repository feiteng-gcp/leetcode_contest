class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
    //    nums = {9,6,1,6,2};
        int m1 = 0, m2 = 0;
        for (int i = 0; i < nums.size(); i += 2) {
            int adj = min((i == 0)?INT_MAX:nums[i - 1], (i == nums.size() - 1)?INT_MAX:nums[i + 1]);
            if (adj <= nums[i])
                m1 += nums[i] - adj + 1;
        }
        for (int i = 1; i < nums.size(); i += 2) {
            int adj = min((i == 0)?INT_MAX:nums[i - 1], (i == nums.size() - 1)?INT_MAX:nums[i + 1]);
            if (adj <= nums[i])
                m2 += nums[i] - adj + 1;
        }
        return min(m1, m2);
    }
};
