class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int ans1 = 0, ans2 = 0, n = nums.size();
        
        for (int i = 1; i < n; i += 2) {
            int t = nums[i - 1];

            if (i + 1 < n) t = min(t, nums[i + 1]);
            ans1 += max(0, nums[i] - (t - 1));
        }
        
        for (int i = 0; i < n; i += 2) {
            int t = INT_MAX;
            if (i > 0) t = min(t, nums[i - 1]);
            if (i + 1 < n) t = min(t, nums[i + 1]);
            ans2 += max(0, nums[i] - (t - 1));
        }
        
        return min(ans1, ans2);
    }
};
