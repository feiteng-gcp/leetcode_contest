const int kMaxVal = 1000;
class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int n = nums.size();

        int res1 = 0;
        for (int i = 0; i < n; i+=2) {
            int minv = kMaxVal;
            if (i - 1 >= 0) minv = min(minv, nums[i-1]);
            if (i + 1 < n) minv = min(minv, nums[i+1]);
            res1 += max(0, nums[i] - minv + 1);
        }
        int res2 = 0;

        for (int i = 1; i < n; i+=2) {
            int minv = kMaxVal;
            if (i - 1 >= 0) minv = min(minv, nums[i-1]);
            if (i + 1 < n) minv = min(minv, nums[i+1]);
            res2 += max(0, nums[i] - minv + 1);
        }
        return min(res1, res2);
    }
};
