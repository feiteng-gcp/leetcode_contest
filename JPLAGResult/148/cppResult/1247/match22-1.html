class Solution {
public:
    int movesToMakeZigzag(vector<int>& nums) {
        int n = nums.size();
        int ans1 = 0;
        for (int i = 1; i < n; i += 2) {
            int l = nums[i-1], r = i+1 < n ? nums[i+1] : 1001;

            ans1 += max(0, nums[i] - min(l,r) + 1);
        }
        int ans2 = 0;
        for (int i = 0; i < n; i += 2) {
            int l = i == 0 ? 1001 : nums[i-1], r = i+1 < n ? nums[i+1] : 1001;
            ans2 += max(0, nums[i] - min(l,r) + 1);
        }
        return min(ans1, ans2);
    }
};
