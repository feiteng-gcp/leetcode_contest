/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
      vector<int> d(n+1, 0);
      TreeNode* xNode;
      function<int(TreeNode*)> dfs =
      [&](TreeNode* cur) -> int {
        if (!cur) return 0;
        if (cur->val == x) {
          xNode = cur;
        }
        d[cur->val] = dfs(cur->right) + dfs(cur->left) + 1;
        return d[cur->val];
      };
      dfs(root);
      bool ret = n - d[x] > d[x];
      if (xNode->right) {
        ret |= (n - d[xNode->right->val]) < d[xNode->right->val];
      }
      if (xNode->left) {
        ret |= (n - d[xNode->left->val]) < d[xNode->left->val];
      }
      return ret;
    }
};
