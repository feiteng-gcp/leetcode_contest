/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode * node = findNode(root, x);
        int nleft = count(node->left);
        int nright = count(node->right);
        int nparent = n - nleft - nright - 1;
        vector<int> A = {nleft, nright, nparent};
        std::sort(A.begin(), A.end());
        return A[2] > A[1] + A[0] + 1;
    }
    TreeNode * findNode(TreeNode * root, int x) {
        if (!root)
            return nullptr;
        if (root->val == x)
            return root;
        TreeNode * tmp = findNode(root->left, x);
        if (tmp)
            return tmp;
        return findNode(root->right, x);
    }
    int count(TreeNode * root) {
        if (!root)
            return 0;
        return count(root->left) + count(root->right) + 1;
    }
};
