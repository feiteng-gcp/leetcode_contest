/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* find(TreeNode* root, int x)
    {
        if(root == NULL) return NULL;
        if(root -> val == x) return root;
        auto m = find(root -> left, x);
        if(m != NULL) return m;
        return find(root -> right, x);
    }
    
    int countnum (TreeNode* root){
        if(root == NULL) return 0;
        return countnum(root -> left) + countnum(root -> right) + 1;
    }
    
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode* t = find(root, x);
        int num1 = countnum(t -> left);
        int num2 = countnum(t -> right);
        cout << num1 << num2 << n;
        if(num1 > n / 2 || num2 > n / 2 || (n - num1 - num2 - 1) > n / 2) return true;
        return false;
    }
};
