/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode* nodeX = findNode(root, x);
        int treeSize = getSize(root);
        int leftSize = getSize(nodeX->left);
        int rightSize = getSize(nodeX->right);
        int upSize = treeSize - leftSize - rightSize - 1;
        return max({upSize, leftSize, rightSize}) > treeSize / 2;
    }
    
    int getSize(TreeNode* node) {
        if (!node)
            return 0;
        return 1 + getSize(node->left) + getSize(node->right);
    }
    
    TreeNode* findNode(TreeNode* node, int x) {
        if (!node)
            return nullptr;
        if (node->val == x)
            return node;
        TreeNode* left = findNode(node->left, x);
        if (left)
            return left;
        return findNode(node->right, x);
    }
};
