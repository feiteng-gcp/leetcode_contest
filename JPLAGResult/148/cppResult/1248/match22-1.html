/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* findNode(TreeNode* root, int x) {
        if (root == nullptr)
            return nullptr;
        if (root -> val == x)
            return root;
        TreeNode* leftNode, *rightNode;
        leftNode = findNode(root -> left, x);
        rightNode = findNode(root -> right, x);
        if (leftNode != nullptr)
            return leftNode;
        if (rightNode != nullptr)
            return rightNode;
        return nullptr;
    }
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode* node = findNode(root, x);
        TreeNode* node_temp = node;
        int c1 = 0, c2 = 0;
        vector<TreeNode*> v;
        if (node -> left != nullptr)
            v.push_back(node -> left);
        
        while (v.size() > 0) {
            node = *v.rbegin();
            v.pop_back();
            c1++;
            if (node -> left != nullptr)
                v.push_back(node -> left);
            if (node -> right != nullptr)
                v.push_back(node -> right);
        }
        v.clear();
        
        if (node_temp -> right != nullptr)
            v.push_back(node_temp -> right);
        while (v.size() > 0) {
            node = *v.rbegin();
            v.pop_back();
            c2++;
            if (node -> left != nullptr)
                v.push_back(node -> left);
            if (node -> right != nullptr)
                v.push_back(node -> right);
        }
        int c = max(c1, c2);
        c = max(c, n - c1 - c2 - 1);
        // cout << c << endl;
        if (2 * c >= n)
            return true;
        else
            return false;
    }
};
