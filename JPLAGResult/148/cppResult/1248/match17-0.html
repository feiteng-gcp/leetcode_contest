/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    int cnt_tree(TreeNode *cur)
    {
        if(!cur) return 0;
        return cnt_tree(cur->left)+cnt_tree(cur->right)+1;
    }
    
    void dfs(TreeNode *cur, int x, int &nx, int &ns)
    {
        if(!cur) return;
        if(nx) return;
        if(cur->val == x)
        {
            nx = cnt_tree(cur);
            ns = max(cnt_tree(cur->left), cnt_tree(cur->right));
            return;
        }
        dfs(cur->left, x, nx, ns);
        if(nx) return;
        dfs(cur->right, x, nx, ns);
        if(nx) return;
    }
    
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        if(root->val == x)
        {
            return cnt_tree(root->left) != cnt_tree(root->right);
        }
        else
        {
            int nx = 0, ns = 0;
            dfs(root, x, nx, ns);
            cout << nx << " " << ns << endl;
            if(ns > n/2 || n-nx > n/2) return true;
            return false;
        }
    }
};
