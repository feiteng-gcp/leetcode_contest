/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    TreeNode* findX(TreeNode* root, int x) {
        if (root == NULL) return NULL;
        if (root->val == x) return root;
        TreeNode* l = findX(root->left, x);
        if (l != NULL) return l;
        return findX(root->right, x);
    }
    int countX(TreeNode* root) {
        if (root == NULL) return 0;
        return countX(root->left) + countX(root->right) + 1;
    }
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode* xn = findX(root, x);
        int l = countX(xn->left), r = countX(xn->right);
        int c = max(l, r);
        c = max(c, n - l - r - 1);
        return c > n - c;
    }
};
