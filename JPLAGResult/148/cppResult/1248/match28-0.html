/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vector<pair<int, int>> cnt;
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        cnt = vector<pair<int, int>>(n+1, {0, 0});
        tra(root);
        
        int l = cnt[x].first;
        int r = cnt[x].second;
        int leftover = n-1-l-r;
        
        int mx = max(leftover, max(l, r));
        int other = n-mx;
        
        return mx > other;
    }
    
    int tra(TreeNode* root)
    {
        if(!root) return 0;
        
        int l = tra(root->left);
        int r = tra(root->right);
        
        cnt[root->val].first = l;
        cnt[root->val].second = r;
        
        return l+r+1;
    }
};
