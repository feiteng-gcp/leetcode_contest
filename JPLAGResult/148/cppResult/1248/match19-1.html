/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    TreeNode* findNode(TreeNode* root, int x){
        if(root == nullptr)
            return nullptr;
        if(root->val == x)
            return root;
        auto l = findNode(root->left, x);
        auto r = findNode(root->right, x);

        if(l == nullptr)
            return r;
        return l;
    }
    int getNodeCnt(TreeNode* root){
        if(root == nullptr)
            return 0;
        return 1 + getNodeCnt(root->left) + getNodeCnt(root->right);
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        auto node = findNode(root, x);
        int half = n / 2 + 1;
        int l = getNodeCnt(node->left);
        int r = getNodeCnt(node->right);
        int p = n - l - r - 1;
        return l >= half || r >= half || p >= half;
    }
};
