/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int cnt(TreeNode* root) {
        if (!root) {return 0;}
        int left = cnt(root->left);
        int right = cnt(root->right);
        return left + right + 1;
    }
    TreeNode* find(TreeNode* root,int x) {
        if (!root) {return NULL;}
        if (root->val == x) {return root;}
        TreeNode* left = find(root->left,x);
        if (left) {return left;}
        return find(root->right,x);
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        if (n == 1) {return false;}
        TreeNode* p1 = find(root,x);
        if (!p1) {return true;}
        int left = cnt(p1->left);
        int right = cnt(p1->right);
        //cout<<"left = "<<left<<endl;
        //cout<<"right = "<<right<<endl;
        int res = max(left,right);
        res = max(res,n-1-left-right);
        return res*2>n;
    }
};
