/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
    
    int r, l;
    
    int travel(TreeNode *node, int x) {
        if (node == NULL) return 0;
        
        int right = travel(node->left, x);
        int left = travel(node->right, x);
        
        if (node->val == x) {
            r = right;
            l = left;
        }
        
        return right+left+1;
    }
    
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        travel(root, x);
        
        int w = (n-1)/2+1;
        if (r>=w || l >= w || (n-1-r-l) >= w) return true;
        return false;
    }
};
