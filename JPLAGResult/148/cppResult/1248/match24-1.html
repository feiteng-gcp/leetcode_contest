/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int xn, nn; bool ans = false;
    int DFS(TreeNode* now) {
        int ln = now->left ? DFS(now->left) : 0, rn = now->right ? DFS(now->right) : 0;
        if (now->val == xn) {

            int npart = max(nn - ln - rn - 1, max(ln, rn));
            ans = npart > nn - npart;
        }
        return ln + rn + 1; 
    }
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        xn = x, nn = n;
        DFS(root);
        return ans;
    }
};
