/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    TreeNode *findNode(TreeNode *p, int target){
        if(!p)
            return nullptr;
        if(p -> val == target)
            return p;
        TreeNode *pl = findNode(p -> left, target);
        TreeNode *pr = findNode(p -> right, target);
        if(pl)
            return pl;
        if(pr)
            return pr;
        return nullptr;
    }
    int count(TreeNode *p){
        if(!p)
            return 0;
        return 1 + count(p -> left) + count(p -> right);
    }
    
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode *p = findNode(root, x);
        int nleft = count(p -> left);
        int nright = count(p -> right);
        int maxi = n - nleft - nright - 1;
        //cout<<maxi<<nleft<<nright<<endl;
        maxi = max(maxi, max(nleft, nright));
        return maxi > n - maxi;
        
    }
};
