/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int solve(TreeNode *root, int x, vector<int> &tmp) {
        if(root == NULL) return 0;
        int l = solve(root->left, x, tmp);
        int r = solve(root->right, x, tmp);
        if(root->val == x) {
            tmp.push_back(l);
            tmp.push_back(r);
            return 0;
        }
        return l + r + 1;
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        vector<int> tmp;
        tmp.push_back(solve(root, x, tmp));
        assert(tmp.size() == 3);
        sort(tmp.begin(), tmp.end());
        return tmp[2] > tmp[1] + tmp[0] + 1;
    }
};
