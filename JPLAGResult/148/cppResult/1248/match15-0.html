/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int mx=0;
    int siz[150];
    int dfs(TreeNode* root, int x){
        int v=root->val;
        siz[v]=1;
        
        if (root->left) {
            int c=dfs(root->left, x);
            siz[v]+=c;
            if (v==x) mx=max(mx, c);
        }
        if (root->right) {
            int c=dfs(root->right, x);
            siz[v]+=c;
            if (v==x) mx=max(mx, c);
        }
        
        return siz[v];
    }
    
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        mx=0;
        dfs(root, x);
        mx=max(mx, n-siz[x]);
        return mx>n/2;
    }
};
