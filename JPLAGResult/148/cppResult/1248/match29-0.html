/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<int, int> parent;
    map<int, TreeNode*> idx;
    void solve(TreeNode* p, vector<int>& sz) {
        if (!p) return;
        idx[p->val] = p;
        sz[p->val] = 1;
        if (p->left) {
            parent[p->left->val] = p->val;
            solve(p->left, sz);
            sz[p->val] += sz[p->left->val];
        }
        if (p->right) {
            parent[p->right->val] = p->val;
            solve(p->right, sz);
            sz[p->val] += sz[p->right->val];
        }
    }
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        vector<int> sz(n + 1);
        parent.clear();
        idx.clear();
        solve(root, sz);
        int best = 0;
        if (parent.count(x)) {
            int y = parent[x];
            best = max(best, n - sz[x]);
        }
        TreeNode* p = idx[x];
        if (p->left) best = max(best, sz[p->left->val]);
        if (p->right) best = max(best, sz[p->right->val]);
        cout << best << endl;
        return best > n - best;
    }
};
