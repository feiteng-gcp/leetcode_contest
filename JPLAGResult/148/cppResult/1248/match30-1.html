/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    typedef std::unordered_map<int, std::vector<int>> graph;
    

    void build_graph(graph &g, TreeNode* root) {
        if (!root) return;
        if (root->left) {
            g[root->val].push_back(root->left->val);
            g[root->left->val].push_back(root->val);
            build_graph(g, root->left);
        }
        if (root->right) {
            g[root->val].push_back(root->right->val);
            g[root->right->val].push_back(root->val);
            build_graph(g, root->right);
        }
    }
    
    // Count the number of nodes reachable from start without crossing forbidden.
    // Assumes that forbidden is adjacent to start.
    int countRegion(const graph &g, int start, int forbidden) {
        auto it = g.find(start);
        if (it == g.end()) return 0;

        int region = 1;
        for (int neighbor : it->second) {
            if (neighbor == forbidden)
                continue;
            region += countRegion(g, neighbor, start);
        }
        return region;
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        // Thesis: an optimal strategy for player 2 is to pick the "best" neghbor of x.
        // Optimal play involves fighting for the geodesic between
        // our two starting points. Anything reachable and off the geodesic
        // can be taken later, for free.
        // So when should we leave a nontrivial geodesic? Never.
        // Compare the two games: in G1 we pick a non-neighboring node y1;
        // in G2, we pick the y2 := (neighbor of x that sits closest to y1).
        // Assuming P1 plays optimally, in G1 we will capture some of the
        // nodes along the geodesic from y1 to x (precisely the closer half
        // of the nodes, rounded down), plus all of their descendents away
        // from the geodesic. But we also get those in G2!
        
        // So, the algorithm: compare each of the (up to) three neighbors y of x.
        // Compare the size of our region (think: flood fill starting at y, without)
        // crossing x against their region (flood fill from x without crossing y).
        // If for any neighbor y we have more than them, return true.
        // Else, false.
        
        (void) n;
        graph g;
        build_graph(g, root);
        for (int y : g[x]) {
            int myScore = countRegion(g, y, x);
            int theirScore = countRegion(g, x, y);
            if (myScore > theirScore)
                return true;
        }
        return false;
    }
};
