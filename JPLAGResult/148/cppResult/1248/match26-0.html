/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        //convert to graph: greedy: we can choose its parent or its left or right 
        //to see if we can get 
        //vector<vector<int>> adj(n);
        //build_graph(root,0,adj);
        //count its left and right subtree nodes
        TreeNode* node=find(root,x);
        int nl=count(node->left);
        int nr=count(node->right);
        if(nl+nr+1<=n/2) return 1;
        if(nl>n/2||nr>n/2) return 1;
        return 0;
                
    }
    TreeNode* find(TreeNode* root,int x){
        if(!root) return 0;
        if(root->val==x) return root;
        auto l=find(root->left,x);
        if(l) return l;
        return find(root->right,x);
    }
    int count(TreeNode* root){
        if(!root) return 0;
        return 1+count(root->left)+count(root->right);
    }
    void build_graph(TreeNode* root,TreeNode* parent,vector<vector<int>>& adj){
        if(!root) return;
        if(parent) {
            adj[root->val-1].push_back(parent->val-1);
            adj[parent->val-1].push_back(root->val-1);
        }
        build_graph(root->left,root,adj);
        build_graph(root->right,root,adj);
    }
};
