/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        int ans=0;
        TreeNode* red=findX(root,x);
        //cout<<countExcept(root,x)<<" "<<countExcept(red->left,-1)<<" "<<countExcept(red->right,-1)<<" "<<(n+1)/2<<endl;
        ans=max(ans,countExcept(root,x));
        ans=max(ans,countExcept(red->left,-1));
        ans=max(ans,countExcept(red->right,-1));
        return ans>=(n+1)/2;
    }
    
    int countExcept(TreeNode* root,int val){
        if(!root)return 0;
        if(root->val==val)return 0;
        return 1+countExcept(root->left,val)+countExcept(root->right,val);
    }
    

    TreeNode* findX(TreeNode* root,int x){
        if(!root)return nullptr;
        if(root->val==x)return root;
        TreeNode* left=findX(root->left,x);
        TreeNode* right=findX(root->right,x);
        if(left!=nullptr)return left;
        if(right!=nullptr)return right;
        return nullptr;
    }
};
