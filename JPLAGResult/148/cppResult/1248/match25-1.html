/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        auto sizes = findChildrenSize(root, x);
        vector<int> s = {sizes.first, sizes.second};
        s.push_back(n - 1 - s[0] - s[1]);
        sort(s.begin(), s.end());

        return s.back() > n - s.back();
    }
private:
    pair<int, int> findChildrenSize(TreeNode* root, int x) {
        if (root == nullptr) {
            return {-1, -1};
        }
        if (root->val == x) {
            return {findSize(root->left), findSize(root->right)};
        }
        auto left = findChildrenSize(root->left, x);
        if (left.first > -1) {
            return left;
        }
        return findChildrenSize(root->right, x);
    }
    int findSize(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }
        int left = root->left == nullptr ? 0 : findSize(root->left);
        int right = root->right == nullptr ? 0 : findSize(root->right);
        return 1 + left + right;
    }
};
