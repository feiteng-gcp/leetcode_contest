/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    map<TreeNode*,int>m;
    map<TreeNode*,TreeNode*>p;
    TreeNode* t;
    void dfs(TreeNode* r,int x){
        if(r==NULL)return;
        if(r->val==x)
            t=r;
        m[r]=1;
        dfs(r->left,x);
        dfs(r->right,x);
        if(r->left){
            p[r->left]=r;
            m[r]+=m[r->left];
        }
        if(r->right){
            p[r->right]=r;
            m[r]+=m[r->right];
        }
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        m.clear();
        dfs(root,x);

        bool ret=false;
        if(t->left){
            int numy=m[t->left];
            int numx=n-numy;
            if(numy>numx)return true;
        }
        if(t->right){
            int numy=m[t->right];
            int numx=n-numy;
            if(numy>numx)return true;
        }
        if(p.find(t)!=p.end()){
            int numx=m[t];
            int numy=n-numx;
            if(numy>numx)return true;
        }
        return ret;
    }
};
