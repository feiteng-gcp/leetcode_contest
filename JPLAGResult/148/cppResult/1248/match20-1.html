/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int lc, rc, total, target;
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
      lc = 0, rc = 0, total = 0;
      target = x;
      total = dfs(root);

      if((total - 1 - lc - rc) > (1 + lc + rc))
        return true;
      if(lc > (total - lc))
        return true;
      if(rc > (total - rc))
        return true;
      return false;
    }
    
    int dfs(TreeNode* root) {
      if(!root)
        return 0;
      int ln = dfs(root->left);
      int rn = dfs(root->right);
      if(root->val == target) {
        lc = ln;
        rc = rn;
      }
      return 1 + ln + rn;
    }
};
