/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int ans = 0;
    int dfs(TreeNode* root, int n, int x) {
        int ma = 0, re = 1;
        if (root->left) {
            int ret = dfs(root->left, n, x);
            re += ret;
            ma = max(ma, ret);
        }
        if (root->right) {
            int ret = dfs(root->right, n, x);
            re += ret;
            ma = max(ma, ret);
        }
        if (x == root->val) {
            ans = max(n - re, ma);
        }
        return re;
    }
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        dfs(root, n, x);
        //printf("%d\n", ans);
        return ans * 2 > n;
    }
};
