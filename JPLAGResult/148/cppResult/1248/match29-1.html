/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int f[110], l[110], r[110];

    void dfs(TreeNode* root) {

        f[root->val] = 1;
        if (root->left) {
            dfs(root->left);
            l[root->val] = root->left->val;
            f[root->val] += f[root->left->val];
        }
        if (root->right) {
            dfs(root->right);
            r[root->val] = root->right->val;
            f[root->val] += f[root->right->val];
        }
    }
    
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        for (int i = 1; i <= 100; i++) {
            f[i] = l[i] = r[i] = 0;
        }
        dfs(root);
        if (f[root->val] - f[x] > f[x]) return true;
        if (l[x]) {
            if (f[l[x]] > f[root->val] - f[l[x]]) return true;
        }
        if (r[x]) {
            if (f[r[x]] > f[root->val] - f[r[x]]) return true;
        }
        return false;
    }
};
