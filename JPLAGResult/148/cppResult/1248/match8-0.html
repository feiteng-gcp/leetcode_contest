/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* getX(TreeNode* r, int x) {
        if(r == NULL) return NULL;
        if(r->val == x) return r;
        TreeNode* tl = getX(r->left, x);
        if(tl != NULL) return tl;
        TreeNode* tr = getX(r->right, x);
        if(tr != NULL) return tr;
        return NULL;
    }
    int getSize(TreeNode* r) {
        if(r == NULL) return 0;
        int res = 1;
        res += getSize(r->left);
        res += getSize(r->right);
        return res;
    }
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode* xnode = getX(root, x);
        int lsz = getSize(xnode->left);
        int rsz = getSize(xnode->right);
        int rest = n - 1 - lsz - rsz;
        if(2 * lsz > n) return true;
        if(2 * rsz > n) return true;
        if(2 * rest > n) return true;
        return false;
    }
};
