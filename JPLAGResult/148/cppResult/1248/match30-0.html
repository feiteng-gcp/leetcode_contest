/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vvi nxt;
    void parse(TreeNode* root){
        if (!root) return;
        if (root->left){
            nxt[root->val -1].pb(root->left->val -1);
            nxt[root->left->val -1].pb(root->val -1);
            parse(root->left);
        }
        if (root->right){
            nxt[root->val -1].pb(root->right->val -1);
            nxt[root->right->val -1].pb(root->val -1);
            parse(root->right);
        }
    }
    int count(int u, int x){
        int sum = 1;
        for (int v: nxt[u]){
            if (v == x) continue;
            sum += count(v, u);
        }
        return sum;
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        nxt.resize(n);
        parse(root);
        int best = 1;
        for (int u: nxt[x-1]){
            best = max(best, count(u, x-1));
        }
        return (best > n-best);
    }
};
