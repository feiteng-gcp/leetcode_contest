/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode* xnode = findx(root, x);
        int l = count(xnode->left, x), r = count(xnode->right, x), all = count(root, x);
        vector<int> v({l, r, all});
        sort(v.begin(), v.end());
        if (v[0] + v[1] < v[2]) return true;
        return false;
    }
    TreeNode* findx(TreeNode* root, int x) {
        if (!root) return root;
        if (root->val == x) return root;
        TreeNode* l = findx(root->left, x);
        if (l) return l;
        TreeNode* r = findx(root->right, x);
        return r;
    }
    int count(TreeNode* root, int x) {
        if (!root) return 0;
        if (root->val == x) return 0;
        return count(root->left, x) + count(root->right, x) + 1;
    }
};
