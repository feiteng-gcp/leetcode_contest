/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        pair<int, int> sizes = getSubTreeSize(root, x);
        int ans = max(n - sizes.first - sizes.second - 1, max(sizes.first, sizes.second));
        return ans > n / 2;
    }
private:
    pair<int, int> getSubTreeSize(TreeNode* root, int x) {
        if (!root) {
            return {-1, -1};
        }
        if (root->val == x) {
            return {countNodes(root->left), countNodes(root->right)};
        }
        auto left = getSubTreeSize(root->left, x);
        auto right = getSubTreeSize(root->right, x);
        if (left.first == -1 && right.first == -1) {
            return {-1, -1};
        } else if (left.first != -1) {
            return left;
        } else {
            return right;
        }
    }
    int countNodes(TreeNode* root) {
        if (!root) {
            return 0;
        }
        return countNodes(root->left) + countNodes(root->right) + 1;
    }
};
