/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    int count(TreeNode* root) {
        if (!root) return 0;

        return count(root->left) + count(root->right) + 1;
    }
    TreeNode* find(TreeNode* root, int x) {
        if (!root) {
            return NULL;
        }
        if (root->val == x) {
            return root;
        }

        auto res = find(root->left, x);
        if (res) {
            return res;
        }
        return find(root->right, x);
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        if (x == root->val) {
            int lsub = count(root->left);
            int rsub = n - 1 - lsub;
            if (lsub > rsub) {
                return lsub > rsub + 1;
            }
            return rsub > lsub + 1;
        }
        auto xroot = find(root, x);
        auto xleft = count(xroot->left);
        auto xright = count(xroot->right);
        if (n > 2 * (xleft + xright + 1)) {
            return true;
        }
        if (2 * max(xleft, xright) > n) {
            return true;
        }
        return false;
        
    }
};
