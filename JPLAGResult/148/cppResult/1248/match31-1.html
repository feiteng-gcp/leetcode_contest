/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<int, TreeNode*> res;
    vector<int> sz;
    
    void dfs(TreeNode* x) {
        res[x->val] = x;
        sz[x->val] = 1;
        if (x->left) {
            dfs(x->left);
            sz[x->val] += sz[x->left->val];
        }
        if (x->right) {
            dfs(x->right);
            sz[x->val] += sz[x->right->val];
        }
    }
    
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        res.clear();
        sz.resize(n + 1);
        dfs(root);
        
        TreeNode* node = res[x];

        bool ans = (n - sz[x] > sz[x]);
        if (node->left) {
            ans |= sz[node->left->val] > n - sz[node->left->val];
        }
        if (node->right) {
            ans |= sz[node->right->val] > n - sz[node->right->val];
        }
        return ans;
    }
};
