/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    unordered_map<TreeNode*,TreeNode*> parent;
    unordered_map<TreeNode*, int> counts;
    TreeNode* xNode = nullptr;
    TreeNode* root = nullptr;
    
    int traverse(TreeNode* node, TreeNode* p, int x) {
        if (!node) return 0;
        parent[node] = p; 
        if (node->val == x) xNode = node;
        int l = traverse(node->left, node, x);
        int r = traverse(node->right, node, x);
        return counts[node] = l+r+1;
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {

        this->root = root;
        traverse(root,nullptr, x);
        assert(xNode);
        //printf("%d %d %d\n", counts[root], counts[root->left], counts[root->right]);
        
        if (parent[xNode]) {
            int xs = counts[xNode];
            int ys = counts[root] - xs;
            if (ys > xs) return true;
        }
        if (xNode->left) {
            int ys = counts[xNode->left];
            int xs = counts[root] - ys;
            if (ys > xs) return true;
        }
        if (xNode->right) {
            int ys = counts[xNode->right];
            int xs = counts[root] - ys;
            if (ys > xs) return true;
        }
        return false;
    }
};
