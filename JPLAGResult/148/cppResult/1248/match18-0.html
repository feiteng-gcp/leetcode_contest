/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        int p = 0, l = 0, r = 0;
        count(root, x, p, l, r);
        //cout << p << " " << l << " " << r << "\n";
        return p > l + r + 1 || l > p + r + 1 || r > p + l + 1;
    }
    void count(TreeNode* root, int x, int& p, int& l, int& r) {
        if (!root) return;
        if (root->val == x) {
            l = count(root->left), r = count(root->right);
        } else {
            p++;
            count(root->left, x, p, l, r);
            count(root->right, x, p, l, r);
        }
    }
    int count(TreeNode* root) {
        if (!root) return 0;
        return count(root->left) + count(root->right) + 1;
    }
};
