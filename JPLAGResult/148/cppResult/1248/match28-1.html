/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {

        TreeNode* node_x = find(root, x);
        int l_count = count(node_x->left);
        int r_count = count(node_x->right);
        int remain = n - (l_count + r_count +1);
        //printf("%d %d %d\n", l_count, r_count, remain);
        int blue = max(max(l_count, r_count), remain);
        return blue * 2 > n;
    }
    TreeNode* find(TreeNode* root, int x) {
        if (root == nullptr)
            return nullptr;
        else if (root->val == x)
            return root;
        else {
            TreeNode* lc = find(root->left, x);
            TreeNode* rc = find(root->right, x);
            if (lc != nullptr)  return lc;
            if (rc != nullptr)  return rc;
            
            return nullptr;
        }
    }
    int count(TreeNode* root) {
        if (root == nullptr)
            return 0;
        else {
            int lc = count(root->left);
            int rc = count(root->right);
            return lc + rc +1;
        }
    }
};
