/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        int a = helper(root, x);
        if (a > n-a)
            return true;
        
        TreeNode *cur = findNode(root, x);
        
        int b = helper(cur->left, -1), c = helper(cur->right, -1);
        a = max(a, max(b, c));
        return a > n-a;
    }
    
    int helper(TreeNode *root, int x) {
        if (!root || root->val == x)
            return 0;
        
        return 1 + helper(root->left, x) + helper(root->right, x);
    }
    
    TreeNode* findNode(TreeNode *root, int x) {
        if (!root)
            return NULL;
        if (root->val == x)
            return root;
        TreeNode *tmp = findNode(root->left, x);
        if (!tmp)
            return findNode(root->right, x);
        else
            return tmp;
    }
    
};
