/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:

    int nofSubtree(TreeNode* root)
    {
        if (root == nullptr)
            return 0;
        return 1 + nofSubtree(root->left) + nofSubtree(root->right);
    }
    
    TreeNode* findNode(TreeNode* root, int x)
    {
        if (!root)
            return nullptr;
        if (root->val == x)
            return root;
        TreeNode *p = findNode(root->left, x);
        if (p)
            return p;
        p = findNode(root->right, x);
        if (p)
            return p;
        return nullptr;
    }
    
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        if (x == root->val)
        {
            if (nofSubtree(root->left) != nofSubtree(root->right))
                return true;
            else
                return false;
        }
        TreeNode *p = findNode(root, x);
        if (!p)
            return true;
        int nleft = nofSubtree(p->left);
        int nright = nofSubtree(p->right);
        int nparent = n - nleft - nright - 1;
        if (nleft > n/2 || nright > n/2 || nparent > n/2)
            return true;
        else
            return false;
    }
};
