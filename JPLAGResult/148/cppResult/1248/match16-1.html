/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    int cnt(TreeNode* root, int x) {
        if(!root || root->val == x) return 0;
        return 1 + cnt(root->left, x) + cnt(root->right, x);
    }
    TreeNode* find(TreeNode* root, int x) {
        if(!root) return NULL;
        if(root->val == x) return root;
        TreeNode* l = find(root->left, x);
        if(l) return l;
        TreeNode* r = find(root->right, x);
        if(r) return r;
        return NULL;
    }
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        int up = cnt(root, x);
        TreeNode* tx = find(root, x);
        int l = cnt(tx->left, x);
        int r = cnt(tx->right, x);
        
        //cout << up << " " << l << " " << r << endl;
        
        if(up > l + r + 1 || l > up + r + 1 || r > up + l + 1) return true;
        return false;
    }
};
