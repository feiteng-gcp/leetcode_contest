/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        TreeNode* p=find(root,x);
        int l1=count(p->left);
        int l2=count(p->right);
        int c=l1+l2+1;
       // cout<<c<<" "<<l1<<" "<<l2;
        if(c<(n-c)||l1>(n-l1)||l2>(n-l2))
            return true;
        else return false;     
    }
    TreeNode* find(TreeNode* r,int x){
        if(r==NULL) return NULL;
        if(r->val==x) return r;
        TreeNode* p=find(r->left,x);
        if(p!=NULL)
            return p;
        return find(r->right,x);
    }
    int count(TreeNode* p){
        if(p!=NULL) 
            return(count(p->left)+count(p->right)+1);
        else return 0;
    }
    
};
