/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
	int N, X, ysize;
	int dfs(TreeNode *t) {
		if (t == NULL) return 0;
		int size = 1;
		int lsize = dfs(t->left);
		int rsize = dfs(t->right);	
		if (t->val == X) {
			ysize = max(lsize, rsize);	
			ysize = max(ysize, N - (1 + lsize + rsize));
		}
		return 1 + lsize + rsize;
	}
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
		N = n, X = x;	 
		dfs(root);
		return ysize > N - ysize;
    }
};
