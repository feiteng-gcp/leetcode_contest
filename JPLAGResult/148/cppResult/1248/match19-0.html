/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vector<TreeNode*> path;
    bool dfs(TreeNode *root, int x) {
        if (!root) return false;
        path.push_back(root);
        if (root->val == x)
            return true;
        if (dfs(root->left, x) || dfs(root->right, x))
            return true;
        path.pop_back();
        return false;
    }
    int count(TreeNode *root) {
        if (!root) return 0;
        return count(root->left) + count(root->right) + 1;
    }
public:
    bool btreeGameWinningMove(TreeNode* root, int n, int x) {
        path.clear();
        dfs(root, x);
        int l = count(path.back()->left), r = count(path.back()->right);
        cout << l << ',' << r << ',' << path.size() << ',' << path.back()->val << endl;
        return l > n / 2 || r > n / 2 || n - (l + r + 1) > n / 2;
    }
};
