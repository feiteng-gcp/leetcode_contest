class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int move1 = 0;
        for (int i = 0; i < nums.length; i+=2) {
            int left = i > 0 ? nums[i - 1] : 10000;
            int right = i < nums.length - 1 ? nums[i + 1] : 10000;
            int min = Math.min(left, right);
            if (nums[i] >= min) {
            	move1 += nums[i] - min + 1;
            
}
        }
        int move2 = 0;
        for (int i = 1; i < nums.length; i+=2) {
        	int left = i > 0 ? nums[i - 1] : 10000;
            int right = i < nums.length - 1 ? nums[i + 1] : 10000;
            int min = Math.min(left, right);
            if (nums[i] >= min) {
            	move2 += nums[i] - min + 1;
            }
        }
        return Math.min(move1, move2);
    }
}
