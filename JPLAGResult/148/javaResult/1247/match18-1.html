class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int n = nums.length;
        int val1=0, val2=0;
        for(int i=0; i<nums.length; i++) {
            if(i % 2 == 0) {
                int min = Math.min(i>0?nums[i-1]:Integer.MAX_VALUE, i<n-1?nums[i+1]:Integer.MAX_VALUE);
                val1 += Math.max(0, nums[i]-(min-1)); 
            } else {
                int min = Math.min(i>0?nums[i-1]:Integer.MAX_VALUE, i<n-1?nums[i+1]:Integer.MAX_VALUE);
                val2 += Math.max(0, nums[i]-(min-1)); 
            
}
        }
        
        return Math.min(val1, val2);
    }
}
