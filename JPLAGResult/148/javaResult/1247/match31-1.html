class Solution {
    public int movesToMakeZigzag(int[] nums) {
        if (nums.length==1)
            return 0;
        int d1 = 0;
        int d2 = 0;
        for (int i = 0; i < nums.length; i++) {
            int num = 0;
            if (i > 0)
                num = Math.max(num, nums[i]-nums[i-1]+1);
            if (i < nums.length-1)
                num = Math.max(num, nums[i]-nums[i+1]+1);
            if (i%2==0) {
                d1 += num;
            } else {
                d2 += num;
            
}
        }
        return Math.min(d1, d2);
    }
}
