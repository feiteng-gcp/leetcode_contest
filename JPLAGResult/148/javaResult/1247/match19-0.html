class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int n = nums.length;
        int decrease1 = 0;
        int decrease2 = 0;
        for (int i = 0; i < n; i++) {
            if (i % 2 == 0) {
                int min = i-1 >= 0 && i+1 < n ? Math.min(nums[i-1], nums[i+1]) : i-1 >= 0 ? nums[i-1] : nums[i+1];
                decrease1 += nums[i] >= min ? (nums[i] - min) + 1 : 0;
            } else {
                int min = i+1 < n ? Math.min(nums[i-1], nums[i+1]) : nums[i-1];
                decrease2 += nums[i] >= min ? (nums[i] - min) + 1 : 0;
            }
        }
        return Math.min(decrease1, decrease2);
    }
}
