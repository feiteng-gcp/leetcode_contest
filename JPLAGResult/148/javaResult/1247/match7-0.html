class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int c1 = 0;
        for (int i = 0; i < nums.length; i += 2) {
            int min = Integer.MAX_VALUE;
            if (i == 0) min = nums[i + 1];
            else if (i == nums.length - 1) min = nums[i - 1];
            else min = Math.min(nums[i + 1], nums[i - 1]);
            if (nums[i] >= min) c1 += nums[i] - min + 1;
        }
        
        int c2 = 0;
        for (int i = 1; i < nums.length; i += 2) {
            int min = Integer.MAX_VALUE;
            if (i == nums.length - 1) min = nums[i - 1];
            else min = Math.min(nums[i + 1], nums[i - 1]);
            if (nums[i] >= min) c2 += nums[i] - min + 1;
        }
        
        return Math.min(c1, c2);
    }
}
