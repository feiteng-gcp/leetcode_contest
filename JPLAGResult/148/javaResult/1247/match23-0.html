class Solution {
    public int movesToMakeZigzag(int[] nums) {
        return Math.min(
            moves(Arrays.copyOf(nums, nums.length), true), 
            moves(Arrays.copyOf(nums, nums.length), false));
    }
    
    private int moves(int[] nums, boolean up) {
        int n = nums.length;
        int res = 0;
        for(int i = 1; i < n; i ++) {
            int cur = nums[i];
            int prev = nums[i - 1];
            if(up) {
                if(cur <= prev) {
                    // dec prev
                    res += prev - cur + 1;
                }
            } else {
                if(cur >= prev) {
                     // dec
                    nums[i] = prev - 1;
                    res += cur - (prev - 1);
                }
            }
            up = !up;
        }
        return res;
    }
}
