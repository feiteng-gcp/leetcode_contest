class Solution {
    public int movesToMakeZigzag(int[] nums) {
        //9, 6, 1, 6, 2
        int even = 0;
        int l = nums.length;
        
        for (int i = 1; i < l; i += 2) {
            int add = 0;
            add = Math.max(add, nums[i] - nums[i - 1] + 1);
            if (i + 1 < l) {
                add = Math.max(add, nums[i] - nums[i + 1] + 1);
            
}
            even += add;
        }
        
        int odd = 0;
        for (int i = 0; i < l; i += 2) {
            int add = 0;
            if (i - 1 >= 0) {
                add = Math.max(add, nums[i] - nums[i - 1] + 1);
            
}
            if (i + 1 < l) {
                add = Math.max(add, nums[i] - nums[i + 1] + 1);
            }
            odd += add;
        }
        
        return Math.min(even, odd);
    }
}
