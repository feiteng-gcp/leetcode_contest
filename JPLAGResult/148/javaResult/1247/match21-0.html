class Solution {

	private int check1(int[] nums) {
		int ans = 0;
		for (int i = 0;i < nums.length;i += 2) {
			int value = Integer.MAX_VALUE;
			if (i + 1 < nums.length) {
				value = Math.min(nums[i + 1] , value);
			}
			if (i - 1 >= 0) {
				value = Math.min(nums[i - 1] , value);
			}
			value --;			
			if (nums[i] > value) {
				ans += (nums[i] - value);				
			}
		}		
		return ans;
	}
	
	private int check2(int[] nums) {
		int ans = 0;
		for (int i = 1;i < nums.length;i += 2) {			
			int value = Integer.MAX_VALUE;
			if (i + 1 < nums.length) {
				value = Math.min(nums[i + 1] , value);
			}
			if (i - 1 >= 0) {
				value = Math.min(nums[i - 1] , value);
			}
			value --;			
			if (nums[i] > value) {
				ans += (nums[i] - value);				
			}
		}		
		return ans;
	}
	
    public int movesToMakeZigzag(int[] nums) {
    	return Math.min(check1(nums) , check2(nums));    	
    }
}
