class Solution {
    
    private static final int MAX = 1001;
    private static final int MIN = -1;
    
    public int movesToMakeZigzag(int[] nums) {
        return 
Math.min(movesToMakeZigzag(nums, true), movesToMakeZigzag(nums, false));
    }
    
    private int movesToMakeZigzag(int[] nums, boolean isIncreasingInTheBeginning) {
        int n = nums.length;
        int[] zigzag = new int[n];
        boolean shouldIncrease = isIncreasingInTheBeginning;
        zigzag[0] = nums[0];
        int modifiedCount = 0;
        for(int i=1;i<n;i++) {
            zigzag[i] = nums[i];
            if (shouldIncrease) {
                if (zigzag[i] <= zigzag[i-1]) {
                    modifiedCount += (zigzag[i-1] - zigzag[i] + 1);
                    zigzag[i-1] = zigzag[i] - 1;
                
}
            } else {
                if (zigzag[i] >= zigzag[i-1]) {
                    modifiedCount += (zigzag[i] - zigzag[i-1] + 1);;
                    zigzag[i] = zigzag[i-1] - 1; 
                }
            }
            shouldIncrease = !shouldIncrease;
        }
        return modifiedCount;
    }
}
