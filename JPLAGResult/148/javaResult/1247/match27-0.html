class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int a = 0, b = 0;
        for (int i = 0;i < nums.length;i++) {
            if (i % 2 == 0) {
                int t1 = ((i == 0) || (nums[i] < nums[i - 1])) ? 0 : (nums[i] - nums[i - 1] + 1);
                int t2 = ((i == (nums.length - 1)) || (nums[i] < nums[i + 1])) ? 0 : (nums[i] - nums[i + 1] + 1);
                a += Math.max(t1, t2);
            } else {
                int t1 = ((i == 0) || (nums[i] < nums[i - 1])) ? 0 : (nums[i] - nums[i - 1] + 1);
                int t2 = ((i == (nums.length - 1)) || (nums[i] < nums[i + 1])) ? 0 : (nums[i] - nums[i + 1] + 1);
                b += Math.max(t1, t2);
            }
        }
        return Math.min(a, b);
    }
}
