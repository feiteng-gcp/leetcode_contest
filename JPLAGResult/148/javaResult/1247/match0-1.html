class Solution {
    
public int movesToMakeZigzag(int[] nums) {
        return Math.min(upDown(nums), downUp(nums));
    }
    
    
private static int upDown(int[] a) {
        int moves = 0;
        for (int i = 0; i < a.length; i += 2) {
            int target = Integer.MAX_VALUE;
            if (i > 0) {
                target = Math.min(target, a[i - 1]);
            }
            if (i + 1 < a.length) {
                target = Math.min(target, a[i + 1]);
            }
            if (a[i] >= target) {
                moves += a[i] - target + 1;
            }
        }
        return moves;
    }
    
    private static int downUp(int[] a) {
        int moves = 0;
        for (int i = 1; i < a.length; i += 2) {
            int target = Integer.MAX_VALUE;
            if (i > 0) {
                target = Math.min(target, a[i - 1]);
            }
            if (i + 1 < a.length) {
                target = Math.min(target, a[i + 1]);
            }
            if (a[i] >= target) {
                moves += a[i] - target + 1;
            }
        }
        return moves;
    }
}
