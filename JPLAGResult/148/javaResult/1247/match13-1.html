class Solution {
    public int movesToMakeZigzag(int[] nums) {
        return Math.min(oddGreaterMove(nums), evenGreaterMove(nums));
    }
    
    public int oddGreaterMove(int[] nums) {
        int count = 0;
        for (int i = 0; i < nums.length; i += 2) {
            int oddNeighorMinVal = Math.min(i > 0 ? nums[i - 1] : Integer.MAX_VALUE, 
                               i < nums.length - 1 ? nums[i + 1] : Integer.MAX_VALUE);
            if (oddNeighorMinVal <= nums[i]) 
{
                count += nums[i] - oddNeighorMinVal + 1;
            }
        }
        return count;
    }
    
    public int evenGreaterMove(int[] nums) {
        int count = 0;
        for (int i = 1; i < nums.length; i += 2) {
            int evenNeighorMinVal = Math.min(i > 0 ? nums[i - 1] : Integer.MAX_VALUE, 
                               i < nums.length - 1 ? nums[i + 1] : Integer.MAX_VALUE);
            if (evenNeighorMinVal <= nums[i]) 
{
                count += nums[i] - evenNeighorMinVal + 1;
            }
        }
        return count;
    }
}
