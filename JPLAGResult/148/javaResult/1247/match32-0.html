class Solution {
    public int movesToMakeZigzag(int[] A) {
        int n  = A.length;
        if(n<3)return 0;
        int count1 = 0, count2 = 0;
        for(int i = 1; i < n; i+=2){
            int l = A[i-1]>A[i] ? 0 : A[i]-A[i-1]+1;
            int r = (i == n - 1 ||  A[i + 1] > A[i]) ? 0 : A[i]- A[i+1] + 1;
            count1 += Math.max(l ,r);
        }
        for(int i = 0; i < n; i+=2){
            int l = ( i == 0 ||A[i-1]>A[i]) ? 0 : A[i]-A[i-1]+1;
            int r = (i == n - 1 ||  A[i + 1] > A[i]) ? 0 : A[i]- A[i+1] + 1;
            count2 += Math.max(l ,r);
        }
        return Math.min(count1 , count2);
    }
}
