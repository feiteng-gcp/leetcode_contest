class Solution {
    public int movesToMakeZigzag(int[] nums) {

        
int[] numscopy = new int[nums.length];
        
        for(int i=0; i<nums.length; i++){
            numscopy[i] = nums[i];
        }
        
        int tmp1 = 0;
        for(int i=0; i<nums.length-1; i++){
            if(i % 2 == 0){
                if(nums[i] > nums[i+1]){
                    continue;
                }else{
                    tmp1 += nums[i+1] - nums[i] + 1;
                    nums[i+1] = nums[i] - 1;
                }
            }else {
                if(nums[i] >= nums[i+1]){
                    tmp1 += nums[i] - nums[i+1] + 1;
                }
            
}
        }
        
        int tmp2 = 0;
        for(int i=0; i<numscopy.length-1; i++){
            if(i % 2 == 0){
                if(numscopy[i] >= numscopy[i+1]){
                    tmp2 += numscopy[i] - numscopy[i+1] + 1;
                    numscopy[i] = numscopy[i+1] - 1;
                }
            }else {
                if(numscopy[i] > numscopy[i+1]){
                    continue;
                }else{
                    tmp2 += numscopy[i+1] - numscopy[i] + 1;
                    numscopy[i+1] = numscopy[i] - 1;
                }
            
}
        }
        
        return Math.min(tmp1, tmp2);
        
    }
}
