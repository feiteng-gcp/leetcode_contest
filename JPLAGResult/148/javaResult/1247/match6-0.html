class Solution {
    public int movesToMakeZigzag(int[] nums) {
        //s1 start big
        int sum = 0;
        for(int i=1; i<nums.length; i+=2){
            int decTo;
            if(i+1<nums.length) decTo = Math.min(nums[i-1], nums[i+1])-1;
            else decTo = nums[i-1] - 1;
            
            if(nums[i]>decTo) sum += nums[i]-decTo;
        }
        //s2 start small
        int sum2 = 0;
        for(int i=0; i<nums.length; i+=2){
            int left = i-1>0 ? nums[i-1] : Integer.MAX_VALUE;
            int right = i+1<nums.length ? nums[i+1] : Integer.MAX_VALUE;
            int decTo = Math.min(left, right)-1;
            
            if(nums[i]>decTo) sum2 += nums[i]-decTo;
        }
        
        return Math.min(sum, sum2);
    }
}
