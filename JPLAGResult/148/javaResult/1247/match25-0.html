class Solution {
    public int movesToMakeZigzag(int[] n) {
        int even = 0;
        int odd = 0;
        int[] n2 = Arrays.copyOf(n, n.length);
        
        for (int i=0;i<n.length-1;i++) {
            if (i%2==0) {
                if (n[i] <= n[i+1]) {
                    even += n[i+1] - n[i] + 1;
                    n[i+1] = n[i] -1;
                }
                if (n2[i] >= n2[i+1]) {
                    odd += n2[i] - n2[i+1] + 1;
                    n2[i] = n2[i+1] -1;
                }
            }else {
                if (n2[i] <= n2[i+1]) {
                    odd += n2[i+1] - n2[i] + 1;
                    n2[i+1] = n2[i] -1;
                }
                if (n[i] >= n[i+1]) {
                    even += n[i] - n[i+1] + 1;
                    n[i] = n[i+1] -1;
                }
            }
        }
        return Math.min(even, odd);
    }
}
