class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int n = nums.length;
        int[] a = new int[n];
        for (int i = 0; i < n; ++i) {
            a[i] = nums[i];
        }
        int res1 = 0;
       for (int i = 0; i < n; i += 2) {
            if (i - 1 >= 0 && nums[i - 1] >= nums[i]) {
                int val = nums[i] - 1;
                res1 += Math.abs(nums[i - 1] - val);
                nums[i - 1] = val;
            }
           if (i + 1 < n && nums[i + 1] >= nums[i]) {
               int val = nums[i] - 1;
               res1 += Math.abs(nums[i + 1] - val);
               nums[i + 1] = val;
           }
       }
        
        int res2 = 0;
        for (int i = 1; i < n; i += 2) {
            if (i - 1 >= 0 && a[i - 1]>= a[i]) {
                int val = a[i] - 1;
                res2 += a[i - 1] - val;
                a[i - 1] = val;
            }
           if (i + 1 < n && a[i + 1] >=a[i]) {
               int val = a[i] - 1;
               res2 += a[i + 1] - val;
               a[i + 1] = val;
           }
        }
        
        return Math.min(res1, res2);
    }
}
