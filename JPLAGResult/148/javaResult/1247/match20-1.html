class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int can1 = 0;
        int can2 = 0;
        for(int i = 0; i < nums.length; i++) {
            if(i % 2 == 0) {
                //update can1
                
int to = Integer.MAX_VALUE;
                if(i != 0) {
                    to = Math.min(to, nums[i - 1]);
                }
                if(i != nums.length - 1) {
                    to = Math.min(to, nums[i + 1]);
                }
                can1 += Math.max(nums[i] - to + 1, 0);
            } else {
                //update can2
                
int to = Integer.MAX_VALUE;
                if(i != 0) {
                    to = Math.min(to, nums[i - 1]);
                }
                if(i != nums.length - 1) {
                    to = Math.min(to, nums[i + 1]);
                }
                can2 += Math.max(nums[i] - to + 1, 0);
            
}
        }
        return Math.min(can1, can2);
    }
}
