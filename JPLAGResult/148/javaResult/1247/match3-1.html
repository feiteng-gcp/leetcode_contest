class Solution {
	public int movesToMakeZigzag(int[] nums) {
		if (nums.length <= 2) return 0;
		int res1 = 0;
		for (int i = 0; i < nums.length; i+=2) {
			int num = nums[i];
			int min = Integer.MAX
_VALUE;
			if (i-1 >= 0) {
				min = Math.min(nums[i-1], min);
			}
			if (i+1 < nums.length) {
				min = Math.min(nums[i+1], min);
			}
			if (num >= min) {
				res1 += (num - min) + 1;
			}
		}
		int res2 = 0;
		for (int i = 1; i < nums.length; i+=2) {
			int num = nums[i];
			int min = Integer.MAX
_VALUE;
			if (i-1 >= 0) {
				min = Math.min(nums[i-1], min);
			}
			if (i+1 < nums.length) {
				min = Math.min(nums[i+1], min);
			}
			if (num >= min) {
				res2 += (num - min) + 1;
			}
		}
		return Math.min(res1, res2);
	}
}
