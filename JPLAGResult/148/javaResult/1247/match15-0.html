class Solution {
    public int movesToMakeZigzag(int[] nums) {
        return Math.min(solve(nums, false), solve(nums, true));
    }
    private int solve(int[] nums, boolean odd) {
        int res = 0, start = odd ? 1 : 2;
        for (int i = start; i < nums.length; i += 2) {
            int g = 2 <= i ? Math.min(nums[i], nums[i - 2]) : nums[i];
            if (g <= nums[i - 1]) res += nums[i - 1] - g + 1;
            if (i + 2 == nums.length && nums[i] <= nums[i + 1]) {                
                res += nums[i + 1] - nums[i] + 1;
            }
        }        
        return res;
    }
}
