class Solution {
    public int movesToMakeZigzag(int[] nums) {
        return Math.min(even(nums), odd(nums));
    }
    
    static int even(int[] nums) {
        int result = 0;
        for(int i = 1, n = nums.length; i < n; i += 2) {
            int min = Math.min(nums[i-1], i + 1 >= n ? Integer.MAX_VALUE : nums[i + 1]);
            if (nums[i] < min) {
                continue;
            } else {
                result += nums[i] - min + 1;
            }
        }
        return result;
    }
    
    static int odd(int[] nums) {
        int result = 0;
        for(int i = 0, n = nums.length; i < n; i += 2) {
            int min = Math.min(i + 1 >= n ? Integer.MAX_VALUE : nums[i+1], i - 1 < 0 ? Integer.MAX_VALUE : nums[i - 1]);
            if (nums[i] < min) {
                continue;
            } else {
                result += nums[i] - min + 1;
            }
        }
        return result;
    }
}
