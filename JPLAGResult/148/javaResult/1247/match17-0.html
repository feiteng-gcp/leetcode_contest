class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int res = 0;
        int n = nums.length;
        for(int i = 0;i<nums.length;i++){
            int diff = 0;
            if(i % 2== 0){
                diff = 0;
                if(i>0 && nums[i] >= nums[i-1] ){
                     diff = nums[i] - nums[i-1]+1 ;
                }
                if(i<n-1 && nums[i] >= nums[i+1]){
                    diff = Math.max(diff,  nums[i] - nums[i+1] +1);
                }
            }
            res += diff;
        }
        int odd_res = 0;
        for(int i = 0;i<nums.length;i++){
            int diff = 0;
            if(i % 2== 1){
                diff = 0;
                if(i>0 && nums[i] >= nums[i-1] ){
                     diff = nums[i] - nums[i-1]+1;
                }
                if(i<n-1 && nums[i] >= nums[i+1]){
                    diff = Math.max(diff,  nums[i] - nums[i+1] +1);
                }
            }
            odd_res += diff;
        }
        
        return Math.min(res, odd_res);
        
        
    }
}
