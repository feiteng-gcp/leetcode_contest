class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int count = 0;
        for(int i = 0; i < nums.length; i += 2) { // try
            int min = Integer.MAX_VALUE;
            if (i > 0 && nums[i-1] < min) min = nums[i-1];
            if (i < nums.length-1 && nums[i+1] < min) min = nums[i+1]
;
            if (nums[i] >= min) count += (nums[i] - min + 1);
        }
        int count1 = 0;
        for(int i = 1; i < nums.length; i += 2) {
            int min = nums[i-1];
            if (i < nums.length-1 && nums[i+1] < min) min = nums[i+1]
;
            if (nums[i] >= min) count1 += (nums[i] - min + 1);
        }
        return Math.min(count, count1);
    }
}
