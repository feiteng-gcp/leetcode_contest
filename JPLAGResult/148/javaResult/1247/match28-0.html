class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int n = nums.length;
        int temp1 = 0, temp2 = 0;
        for (int i = 1; i < n; i += 2) {
            int adj = Math.min(nums[i-1], ((i < n - 1) ? nums[i+1] : Integer.MAX_VALUE));
            if (adj <= nums[i]) temp1 += nums[i] - adj + 1;
        }
        for (int i = 0; i < n; i += 2) {
            int adj = Math.min((i < n - 1) ? nums[i+1] : Integer.MAX_VALUE, ((i > 0) ? nums[i-1] : Integer.MAX_VALUE));
            if (adj <= nums[i]) temp2 += nums[i] - adj + 1;
        }
        return Math.min(temp1, temp2);
    }
}
