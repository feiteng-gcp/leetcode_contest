class Solution {
    public int movesToMakeZigzag(int[] nums) {
        int res1 = 0, res2 = 0, n = nums.length;
        int[] a = new int[n + 2];
        for(int i = 0 ; i < n; i ++)        
            a[i + 1] = nums[i];
        a[0] = 2000;
        a[n + 1] = 2000;
        for(int i = 1; i <= n; i += 2)
        {
            int tmp = Math.min(a[i - 1], a[i + 1]);
            
if(tmp <= a[i])
                res1 += (a[i] - tmp + 1);
        }
        for(int i = 2; i <= n; i += 2)
        {
            int tmp = Math.min(a[i - 1], a[i + 1]);
            
if(tmp <= a[i])
                res2 += (a[i] - tmp + 1);
        }
        return Math.min(res1, res2);
        
    }
}
