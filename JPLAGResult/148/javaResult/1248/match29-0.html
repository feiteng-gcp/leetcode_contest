/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int total = 0;
    int leftTotal = 0;
    int rightTotal = 0;
    int bothTotal = 0;
    int x = 0;

    public boolean btreeGameWinningMove(TreeNode root, int n, int x) {
        this.total = n;
        this.x = x;
        dfs(root);
        return (leftTotal > (total - leftTotal) || rightTotal > (total - rightTotal) || bothTotal < (total - bothTotal));
    }
    
    private int dfs(TreeNode root){
        if(root == null){
            return 0;
        }
        if (root.val == x){
            leftTotal = dfs(root.left);
            rightTotal = dfs(root.right);
            bothTotal = leftTotal + rightTotal + 1;
            return bothTotal;
        }else{
            return dfs(root.left) + dfs(root.right) + 1;
        }
    }
}
