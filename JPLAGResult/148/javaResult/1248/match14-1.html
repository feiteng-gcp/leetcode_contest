/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    TreeNode cx = null;
    public boolean btreeGameWinningMove(TreeNode root, int n, int x) {
        dfs(root, x);
        int a = cnt(cx.left);
        int b = cnt(cx.right);
        int c = n - a - b - 1;
        
return a > n / 2 ||b > n / 2 || c > n / 2;
    }
    void dfs(TreeNode r, int x) {
        if(r == null) return;
        if(r.val == x) {
            cx = r;
            return;
        }
        dfs(r.left, x);
        if(cx == null) dfs(r.right, x);
    
}
    int cnt(TreeNode r) {
        if(r == null) return 0;
        return 1 + cnt(r.left) + cnt(r.right);
    }
}
