/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
private int leftnumber;
    private int rightnumber;
    private int number;
    
    private int f;

    public boolean btreeGameWinningMove(TreeNode root, int n, int x) {
        f = x;
        number = fd(root);
        
        int maxfen = Math.max(Math.max(leftnumber, rightnumber), number - 1 - leftnumber - rightnumber);
        
return maxfen > number - maxfen;
    }
    
    private int fd(TreeNode root){
        if(root == null){
            return 0;
        }else{
            if(root.val == f){
                leftnumber = fd(root.left);
                rightnumber = fd(root.right);
                
return leftnumber + rightnumber + 1;
            }else{
                return fd(root.left) + fd(root.right) + 1;
            }
        }
        
    }
}
