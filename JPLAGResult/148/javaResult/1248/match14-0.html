/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    TreeNode target = null;
    // find node x , count # of left , right subtree , and # others
    public boolean btreeGameWinningMove(TreeNode root, int n, int x) {
        dfs(root, x);
        int countL = number(target.left);
        int countR = number(target.right);
        
        int first = countL + countR + 1;
        int others = n - first;
        if (others > first) return true;
        if (countL > countR + 1 + others) return true;
        if (countR > countL + 1 + others) return true;
        
        return false;
    }
    
    public void dfs(TreeNode root, int x) {
        if (root == null) return;
        if (root.val == x) {
            target = root;
            return;
        }
        dfs(root.left, x);
        dfs(root.right, x);
    }
    
    public int number(TreeNode root) {
        if (root == null) return 0;
        return 1 + number(root.left) + number(root.right);
    }
}
