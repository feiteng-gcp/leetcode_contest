class Solution {
    public int longestDecomposition(String text) {
        return longestDecomposition(text, 0, text.length()-1);
    }

    public int longestDecomposition(String text, int start, int end) {
        if (start > end) return 0;
        for (int i = start; i < end ; i++) {
            String prefix = text.substring(start, i + 1);
            String suffix = text.substring(end - i + start, end + 1);
            if (prefix.equals(suffix)) {
                return 2 + longestDecomposition(text, i + 1, end - i + start - 1);
            }
        }
        return 1;
    }
}
