class Solution {
    
    int f[][] = new int[1001][1001];
    
    
public int longestDecomposition(String text) {
        return dp(0,text.length()-1,text);
    }
    
    public int dp(int l,int r,String t){
        
if(l==r){
            return 1;
        }else if(l>r){
            return 0;
        }
        if(f[l][r]>0) return f[l][r];
        
        int res = 1;
        for(int i=l,j=r;i<j;++i,--j){
            String s1 = t.substring(l,i+1);
            String s2 = t.substring(j,r+1);
            
            if(s1.equals(s2)){
                
                int num = dp(i+1,j-1,t);
                res = Math.max(res,num+2);
            }
        }
        return f[l][r] = res;
        
    }
}
