class Solution {
    public int longestDecomposition(String text) {
        int n = text.length();
        if (n == 0) return 0;
        
        int max = 1;
        for (int l = 1; l <= n / 2; l++) {
            String a = text.substring(0, l);
            String b = text.substring(n - l, n);
            if (a.equals(b)) {
                max = Math.max(max, 2 + longestDecomposition(text.substring(l, n - l)));
            }
        }        
        return max;
    }
}
