class Solution {
    public int longestDecomposition(String text) {
        int n = text.length();
        if (n <= 1) {
            return n;
        }
        for (int i = 1; i <= n / 2; i++) {
            if (text.substring(0, i).equals(text.substring(n - i))) {
                return 2 + longestDecomposition(text.substring(i, n - i));
            }
        }
        return 1;
    }

        System.out.println(s.longestDecomposition("merchant"));

        System.out.println(s.longestDecomposition("aaa"));
