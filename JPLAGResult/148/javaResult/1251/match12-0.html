class Solution {
	public int longestDecomposition(String text) {
		if (text == null || text.length() == 0)
			return 0;
		int sum = 0;
		int l = 0, r = text.length() - 1;
		int preL = l, preR = r;
		while (l < r) {
			String left = text.substring(preL, l + 1);
			String right = text.substring(r, preR + 1);
			if (left.equals(right)) {
				preL = l + 1;
				preR = r - 1;
				sum += 2;
			}
			l++;
			r--;
		}
		if (preL <= preR)
			sum += 1;
		return sum;

	}
}
