class Solution {
    public int longestDecomposition(String text) {
        int left = 0, right = text.length();
        int l = left + 1, r = right - 1;
        int ans = 0;
        while (l <= r) {
            if (text.substring(left, l).equals(
text.substring(r, right))) {
                ans += 2;
                left = l;
                right = r;
            }
            l++;
            r--;
        }
        if (left < right) ans++;
        return ans;
    }
}
