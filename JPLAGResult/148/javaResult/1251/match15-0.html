class Solution {
    
    private int[][] dp;
    
    public int longestDecomposition(String text) {
        dp = new int[text.length()][text.length()];
        return helper(text, 0, text.length() - 1);
    }
    
    private int helper(String text, int left, int right) {
        if (left > right)
            return 0;
        if (dp[left][right] != 0)
            return dp[left][right];
        
        int i = 0;
        int res = 1;
        while (left + i < right - i) {
            if (text.substring(left, left + i + 1).equals(text.substring(right - i, right + 1))) {
                res = Math.max(res, helper(text, left + i + 1, right - i - 1) + 2);
            }
            ++i;
        }
        dp[left][right] = res;
        return res;
    }
}
