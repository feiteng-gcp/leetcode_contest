class Solution {
    public int longestDecomposition(String text) {
        char[] cs = text.toCharArray();
        int i = 0, j = cs.length - 1, pow = 0, left = 0, right = 0, res = 0;
        StringBuilder sbl = new StringBuilder(), sbr = new StringBuilder();
        while (i < j) {
            sbl.append(cs[i++]);
            sbr.insert(0, cs[j--]);
            if (sbl.toString().equals(sbr.toString())) {
                res += 2;
                sbl.setLength(0);
                sbr.setLength(0);
            }
        }
        if (i == j || 0 != sbl.length()) res++;
        return res;
    }
}
