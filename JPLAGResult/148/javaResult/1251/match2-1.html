class Solution {
    public int longestDecomposition(String text) {
        return helper(
text.toCharArray(), 0, text.length() - 1);
    }
    
    private int helper(char[] sc, int left, int right) {
        if(left > right) return 0;
        if(left == right) return 1;
        if(sc[left] == sc[right]) {
            return 2 + helper(sc, left + 1, right - 1);
        }
        
        for(int diff = 1; left + diff < right - diff; diff++) {
            
if(compare(sc, left, right, diff)) {
                return 2 + helper(sc, left + diff + 1, right - diff - 1);
            }
        }
        return 1;
    }
    
    private boolean compare(char[] sc, int l, int r, int len) {
        int ls = l;
        int rs = r - len;
        for(int i = 0; i <= len; i++) {
            if(sc[ls + i] != sc[rs + i]) {
                return false;
            }
        }
        return true;
    }
}
