class Solution {
    public int longestDecomposition(String text) {
        if(text == null || text.length() == 0) return 0;
        int l = text.length();
        int[] dp = new int[l];
        Arrays.fill(dp, -1);
        
return util(text, 0, dp);
    }
    
    private int util(String text, int i, int[] dp) {
        if(dp[i] >= 0) return dp[i];
        String sub = text.substring(i, text.length() - i);
        //base case
        if(sub.length() == 0) return 0;
        if(sub.length() == 1) return 1;
        //regular case
        int res = 1;
        for(int k = 1; k <= sub.length() / 2; k++) {
            
String prefix = sub.substring(0, k);
            if(sub.endsWith(prefix)) {
                res = Math.max(res, 2 + util(text, i + k, dp));
            }
        }
        dp[i] = res;
        return res;
    }
}
