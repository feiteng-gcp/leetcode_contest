class Solution {
    HashMap<String, Integer> map = new HashMap<>();
    
    public int longestDecomposition(String text) {
        // dp(i, j) = max(dp(i + k, j - k))) + 2; where s(i, k - 1) == s(j - k + 1, j)
        
        return dp(text, 0, text.length() - 1);
    }
    
    int dp(String text, int s, int e) {
        String key = s + ";" + e;
        int max = 1;
        
        if (map.containsKey(key)) {
            return map.get(key);
        } else if (s == e) {
            return 1;
        } else if (e < s) {
            return 0;
        }
        
        for (int i = 1; s + i <= e - i + 1; i++) {
            if (text.substring(s, s + i).equals(text.substring(e - i + 1, e + 1))) {
                max = Math.max(max, dp(text, s + i, e - i) + 2);    
            }
        }
        
        map.put(key, max);
        return max;
    }
}
