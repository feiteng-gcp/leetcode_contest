class Solution {
    Map<String, Integer> memo = new HashMap<>();
    public int longestDecomposition(String s) {
        if (s == null || s.length() <= 0)   return 0;
        int n = s.length();
        
if (n == 1)    return 1;
        if (memo.containsKey(s))    return memo.get(s);
        int cnt = 1;
        for (int l = 1; l <= n / 2; l++) {
            boolean palin = true;
            for (int i = 0, j = n - l; i < l; i++, j++){
                if (s.charAt(i) != s.charAt(j)) {
                    palin = false;
                    break;
                }
            }
            if (palin)  cnt = Math.max(cnt, 2 + 
longestDecomposition(s.substring(l, n - l)));
        }
        memo.put(s, cnt);
        return cnt;
    }
    
    public int cntMax(String s, int ll, int rr) {
        return 0;
    }
}
