class Solution {
    public int longestDecomposition(String text) {
        int n = text.length();
        int[] memo = new int[n];
        char[] s = text.toCharArray();

        dp(s, 0, memo);
        return memo[0];
    }

    private int dp (char[] s, int start, int[] memo) {
        if (start * 2 >= s.length) 
{
            return 0;
        }
        if (memo[start] != 0) {
            return memo[start];
        }
        int top = s.length - 2 * start;
        if ((s.length & 1) == 0) {
            top++;
        }
        for (int len=1; 2 * len < top; len++) {
            if (equal(s, start, len)) {
                int next = dp(s, start+len, memo);
                if (next >= 0) {
                    memo[start] = next + 2;
                    return memo[start];
                }
            }
        }
        memo[start] = 1;
        return memo[start];
    }

    private boolean equal(char[] s, int start, int len) {
        int left = start, right = s.length - start - len;
        while (len-- > 0) {
            if (s[left] != s[right]) 
{
                return false;
            }
            left++;
            right++;
        }
        return true;
    }
}
