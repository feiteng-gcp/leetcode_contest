class Solution {
    public int longestDecomposition(String text) {
        int res = 0, len = text.length();
        for(int i = 0, j = 0; i < (len + 1) / 2; ){
            char c = text.charAt(i);
            StringBuilder sb = new StringBuilder();
            for(j = len - i - 1; j >= (len + 1) / 2; j --){
                char m = text.charAt(j);
                sb.insert(0, m);
                
if(text.substring(i, len - 1 - j + 1).equals(sb.toString())){
                    i = len - 1 - j + 1;
                    res += 2;
                    break;
                }
            }
            if(j < (len + 1) / 2){
                res ++;
                break;
            }
        }
        return res;
    }
}
