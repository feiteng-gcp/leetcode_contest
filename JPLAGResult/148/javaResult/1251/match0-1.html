public class Solution {
    public int longestDecomposition(String text) {
        if(text.length()==0) return 0;
        int start=0, end=text.length();
        
for(int len=1;start+len<=end-len;len++){
            if(text.substring(start, start+len).equals(text.substring(end-len, end))){
                return 2+longestDecomposition(text.substring(len, end-len));
            }
        }
        return 1;
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        System.out.println(solution.longestDecomposition("aaa"));
    }
}
