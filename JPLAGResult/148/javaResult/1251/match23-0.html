class Solution {
    
    Map<Integer, Integer> m = new HashMap<>();
    boolean[][] pre = null;
    int len = 0;
    
    int dp(int l) {
        if (l >= len / 2 && len % 2 == 0 || l > len / 2) return 0;
        if (m.containsKey(l)) return m.get(l);
        
        int ret = 1;
        
        for (int i = l; i < len / 2; ++i) {
            int next = i + 1;
            if (pre[l][i]) ret = Math.max(ret, 2 + dp(next));
        }
        
        m.put(l, ret);
        return ret;
    }
    
    public int longestDecomposition(String text) {
        len = text.length();
        pre = new boolean[len + 5][len + 5];
        
        for (int l = 0; l < len / 2; ++l) {
            for (int r = l; r < len / 2; ++r) {
                if (text.substring(l, r + 1).equals(text.substring(len - r - 1, len - l))) pre[l][r] = true;
            }
        }
        
        int ret = dp(0);
        System.out.println(m);
        
        return ret;
    }
}
