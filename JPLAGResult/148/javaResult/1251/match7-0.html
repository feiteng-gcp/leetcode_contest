class Solution {

  Map<Integer, Integer> mem;

  private int helper(String s, int id) {
//    System.out.println("helper : " + id);
    int st = id, en = s.length()-id-1;
    if(st > en) return 0;
    if(st == en) return 1;
    if(mem.containsKey(st)) return mem.get(st);
    int mp = (en-st-1)/2;
    int res = 1;
    for(int i=st; i<=st+mp; i++) {
//      System.out.println("sub : " + i);
//      System.out.println(en);
//      System.out.println(s.substring(st, i+1));
//      System.out.println(s.substring(en-i+st, en+1));
      if(s.substring(st, i+1).equals(s.substring(en-i+st, en+1))) res = Math.max(res, 2+helper(s, i+1));
    }
    mem.put(id, res);
    return res;
  }

  public int longestDecomposition(String s) {
    mem = new HashMap<>();
    return helper(s, 0);
  }

}
