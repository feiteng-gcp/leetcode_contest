class Solution {
    int res;

    public int longestDecomposition(String text) {
        res = 0;
        dfs(text, 0, text.length() - 1,0);
        return res;
    }

    public void dfs(String text, int l, int r, int cur) {
        int j = r;
        for (int i = l; i < j; i++, j--) {
            if (text.substring(l, i + 1).equals(text.substring(j, r + 1))) {
                dfs(text, i + 1, j - 1,cur+2);
            }
        }
        if (l > r)
            res = Math.max(cur, res);
        else
            res = Math.max(cur + 1, res);
    }
}
