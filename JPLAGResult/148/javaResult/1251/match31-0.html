class Solution {
    public int longestDecomposition(String text) {
        return dfs(text);
    }
    Map<String, Integer> map = new HashMap();
    
    public int dfs(String text) {
        if(map.containsKey(text)) return map.get(text);
        if(text.length()< 2) return text.length();
        int[] next = get(text);
        int size = next[next.length - 1];
        int n = text.length();
        if(size * 2 > n){
            size = n /2 ;
        }
        
        
        
        
        
       int tmp=size==0?1: dfs(text.substring(0, size)) *2+ dfs(text.substring(size,n-size));
        
        
        
        
        
        
        map.put(text,tmp);
        // System.out.println(text+"  "+next[next.length - 1]);
        // System.out.println(text+"  "+tmp);
        return tmp;
    }
    public int[] get(String s){
        int[] next = new int[s.length() + 1];
        next[0] = -1;
        int k = -1;
        int i = 0;
        while(i<s.length()){
            if(k== -1 || s.charAt(i) ==  s.charAt(k)){
                i++;
                k++;
                next[i]=k
                    ;
            }else{
                k = next[k];
            }
        }
        return next;
    }
}
