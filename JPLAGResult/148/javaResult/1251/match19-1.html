class Solution {
    public int longestDecomposition(String s) {
        int cnt = 0;
        int l = 0, r = s.length() - 1;
        boolean ok = true;
        while (l <= r){
            ok = false;
            for (int len = 1; len * 2 <= r - l + 1; len++) {
                if (
s.substring(l, l + len).equals(s.substring(r-len+1, r+1))) {
                    cnt += 2;
                    l += len;
                    r -= len;   
                    ok = true;
                    break;
                }
            }
            if (!ok) {
                cnt++;
                break;
            }
        }
        return cnt;
    }
}
