class Solution {
	
	private int[][] dp = new int[210][210];	
	
	public int longestDecomposition(String text) {
		int i , j;
		for (i = 0;i < 210;i ++) {
			for (j = 0;j < 210;j ++) {
				dp[i][j] = - 1;				
			}
		}
		return solve(text , 0 , text.length() - 1);		
    }
	
	private boolean check(String text , int from , int to , int len) {
		for (int i = 0;i < len;i ++) {
			if (text.charAt(from + i) != text.charAt(to - (len - 1) + i)) {
				return false;
			}
		}		
		return true;
	}
	
	private int solve(String text , int from , int to) {
		if (from > to) {
			return 0;
		} else if (from == to) {
			return 1;
		} else {
			if (dp[from][to] >= 0) {
				return dp[from][to];				
			}
			int total = to - from + 1 , ans = 1;			
			for (int len = 1;len <= total / 2;len ++) {				
				if (check(text , from , to , len)) {
					int result = solve(text , from + len , to - len) + 2;
					ans = Math.max(ans , result);					
				}
			}				
			return dp[from][to] = ans;
		}	
	}
	
}
