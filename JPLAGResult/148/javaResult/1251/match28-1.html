class Solution {
    int maxL = 1;
    public int longestDecomposition(String text) {
        maxL = helper(text, 0, text.length()-1);
        return maxL;
    }
    public int helper(String text, int head, int tail) {
        if (head == tail) return 1;
        else if (head == tail+1) return 0;
        else if (head > tail) return Integer.MIN_VALUE;
        int ind = (tail-head+1+1)/2;
        int maxF = 1;
        for (int i = tail; i >= ind; i--) {
            int next_head = head + tail - (i-1);
            int next_tail = i-1;
            int pos1 = head;
            int pos2 = i;
            while (pos1 <= tail && pos2 <= tail && text.charAt(pos1) == text.charAt(pos2)) {
                pos1++;
                pos2++;
            }
            if (pos2 == tail+1) maxF = Math.max(maxF, 2+helper(text, next_head, next_tail));
        }
        return maxF;
    }
}
