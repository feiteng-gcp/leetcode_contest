class Solution {
    HashMap<Integer, Integer> map = new HashMap<>();
    public boolean check(int s, int e, int len, char[] cs) {
        int es = e - len + 1; 
        // System.out.println(cs.length + " " + s + " " + es + " " + len);
        for (int i = 0;i < len;i++) {
            if (cs[s + i] != cs[es + i]) return false;
        }
        return true;
    }
    
    public int dfs(int s, int e, char[] cs) {
        
if (s == e) return 1;
        if (s > e) return 0;
        if (map.containsKey(s)) return map.get(s);
        
        int mid = (s + e + 1)/2;
        int max = 1;
        for (int i = 0;s + i < mid;i++) {
            if (check(s, e, i + 1, cs)) {
                int curr = 2 + dfs(s + i + 1, e - (i + 1), cs);
                max = Math.max(max, curr);
            
}
        }
        map.put(s, max);
        return max;
    }
    
    public int longestDecomposition(String text) {
        return dfs(0, text.length() - 1, text.toCharArray());
    }
}
