class Solution {
    public int longestDecomposition(String text) {
        int n = text.length();
        int[][] dp = new int[n][n];
        for(int j = 0; j < n; ++j) {
            for(int i = j; i >= 0; --i) {
                int result = 1;
                for(int x = i, y = j; x < y; ++x, --y) {
                    boolean equal = true;
                    for(int p = i, q = y; p <= x; ++p, ++q) {
                        if (text.charAt(p) != text.charAt(q)) {
                            equal = false;
                            break;
                        }
                    }
                    if (equal) {
                        result += 1 + dp[x+1][y-1];
                        break;
                    }
                }
                dp[i][j] = result;
            }
        }
        return dp[0][n-1];
    }
}
