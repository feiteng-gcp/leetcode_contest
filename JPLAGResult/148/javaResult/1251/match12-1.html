class Solution {
    public int longestDecomposition(String text) {
        
int start = 0;
        int end = text.length() - 1;
        int length = 1;
        int total = 0;
        while(start <= end){
            String first = text.substring(start, start + length);
            String last = text.substring(end, end + length);
            if(first.equals(last)){
                if(start < end){
                    total++;
                }
                total++;
                start = start + length;
                end = end - 1;
                length = 1;
            }else{
                length++;
                end = end - 1;
            }
        }
        return total;
    }
}
