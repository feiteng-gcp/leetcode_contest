class Solution {
    StringBuilder sb;

    public int longestDecomposition(String text) {
        sb = new StringBuilder(text);
        return helper(0, text.length()-1);
    }

    private int helper(int i, int j) {
        if (i == j)
            return 1;
        if (i > j) return 0;
        char end = sb.charAt(j);
        char start = sb.charAt(i);
        if (start == end) {
            return 2 + helper(i + 1, j - 1);
        }
        for (int k = i + 1; k <= (j + i) / 2; k++) {
            if (sb.charAt(k) == end && check(i, j, k)) {
                return 2 + helper(k+1, j-k+i-1);
            }
        }
        return 1;
    }

    private boolean check(int i, int j, int k) {
        if (k - i >= j - k) {
            return false;
        }
        if (sb.substring(i, k + 1).equals(sb.substring(j-k+i, j + 1)))
            return true;
        return false;
    }
}
