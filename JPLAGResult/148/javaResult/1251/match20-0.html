class Solution {
    public int longestDecomposition(String text) {
        int length = text.length();
        int[][] dp = new int[length][length];
        boolean[][] vis = new boolean[length][length];
        boolean[][] mirror = new boolean[length][length];
        for (int i = 0; i < length / 2; i++) {
            for (int j = i; j < length / 2; j++) {
                String front = text.substring(i, j + 1);
                String back = text.substring(length - j - 1, length - i);
                // System.out.println(front + " " + back);
                if (front.equals(back)) {
                    mirror[i][j] = true;
                }
            }
        }
        return dfs(text, 0, length - 1, dp, vis, mirror);
    }

    public int dfs (String text, int x, int y, int[][] dp, boolean[][] vis, boolean[][] mirror) {
        if (x > y) {
            return 0;
        }
        if (vis[x][y]) {
            return dp[x][y];
        }
        vis[x][y] = true;
        dp[x][y] = 1;
        for (int i = x; i < (x + y + 1) / 2; i++) {
            if (mirror[x][i]) {
                dp[x][y] = Math.max(
                    dp[x][y],
                    dfs(text, i + 1, y - (i - x + 1), dp, vis, mirror) + 2
                );
            }
        }
        return dp[x][y];
    }
}
