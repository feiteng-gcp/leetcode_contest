class Solution {
	public int longestDecomposition(String text) {
		char[] chars = text.toCharArray();
		int start = 0;
		int end = chars.length - 1;
		StringBuilder startBuilder = new StringBuilder();
		StringBuilder endBuilder = new StringBuilder();

		int numChunks = 0;

		while (start < end) {
			startBuilder.append(chars[start++]);
			endBuilder.append(chars[end--]);
			if (startBuilder.toString().equals(endBuilder.reverse().toString())) {
				numChunks += 2;
				startBuilder.setLength(0);
				endBuilder.setLength(0);
			}
			endBuilder.reverse();
		}

		if (start == end || startBuilder.length() > 0) numChunks++;
		return numChunks;
	}
}
