class Solution {
    public int longestDecomposition(String text) {
        /* 
            left = 0, 3
            right = len - 3, len
            
            left = 2, 2 + 3
            right = len - 2 - 3, len - 2
        
        */
        int res = 0;
        int start = 0;
        for (;;){
            int nextStart = decompose(start, text);
            if (text.length() % 2 == 0 && nextStart == text.length() / 2){
                return res + 2;
            }
            if (nextStart == -1){
                return res + 1;
            }
            start = nextStart;
            res += 2;
        }
    }
    
    private int decompose(int start, String text){
        int len = 1;
        while (start + len <= text.length()/2){
            String a = text.substring(start, start + len);
            String b = text.substring(text.length() - len - start, text.length() - start);
            if (a.equals(b)){
                return start + len;
            }
            len++;
        }
        return -1;
    }
}
