class Solution {
	private int[][] mem;
    public int longestDecomposition(String text) {
    	int N = text.length();
        mem = new int[N][N];
       	for (int i = 0; i < N; i++)
       		for (int j = 0; j < N; j++) 
       			mem[i][j] = -1;
       	return 
helper(text, 0, N-1);
    }
    private int helper(String text, int left, int right){

    	if (left == right) return 1;
    	if (left > right) return 0;
    	if (mem[left][right] != -1) return mem[left][right];
    	int ans = 0;

    	int i = left, j = right;
    	while (i < j) {
    		String s1 = text.substring(left, i+1);
    		String s2 = text.substring(j, right+1);
    		if (s1.compare
To(s2) == 0) {
    			ans = Math.max(ans, 2 + helper(text,i+1, j-1));
    		}
    		i++; j--;
    	}
    	if (ans == 0) ans++;
    	mem[left][right] = ans;
    	return ans; 	
    }
}
