class SnapshotArray {
    Map<Integer, Map<Integer, Integer>> map;
    Map<Integer, Integer> m;
    int[] nums;
    int t = 0;
    public SnapshotArray(int length) {
        map = new HashMap<>();
        m = new HashMap<>();
        nums = new int[length];
    
}
    
    public void set(int index, int val) {
        nums[index] = val;
        m.put(index, val);
    }
    
    public int snap() {
        map.put(t, 
new HashMap<>(m));
        return t++;
    }
    
    public int get(int index, int snap_id) {
        Map<Integer, Integer> k = map.get(snap_id);
        if (!k.containsKey(index)) return 0;
        return k.get(index);
    }
}
