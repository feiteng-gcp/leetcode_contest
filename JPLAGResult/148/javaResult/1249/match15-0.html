class SnapshotArray {

    // id -> value
    List<TreeMap<Integer, Integer>> snap;
    int id;
    
    public SnapshotArray(int length) {
        snap = new ArrayList<>();
        for(int i = 0; i < length; i++) {
            snap.add(new TreeMap<Integer, Integer>());
        }
        id = 0;
    }
    
    public void set(int index, int val) {
        snap.get(index).put(id, val);
    }
    
    public int snap() {
        id++;
        return id - 1;
    }
    
    public int get(int index, int snap_id) {
        TreeMap<Integer, Integer> map = snap.get(index);
        Integer floorKey = map.floorKey(snap_id);
        if(floorKey == null) return 0;
        return map.get(floorKey);
    }
}
