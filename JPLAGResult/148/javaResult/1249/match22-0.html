    class  SnapshotArray {
        ArrayList<TreeMap<Integer,Integer>> mp;
        int snap;
        public SnapshotArray(int length) {
            mp=new ArrayList<>();
            for(int i=0;i<length;i++){
                TreeMap<Integer,Integer> temp=new TreeMap<>();
                temp.put(-1,0);
                mp.add(temp);
            }
            snap=-1;
        }

        public void set(int index, int val) {
            mp.get(index).put(snap,val);
        }

        public int snap() {
            return ++snap;
        }

        public int get(int index, int snap_id) {
            int i=mp.get(index).floorKey(snap_id-1);
            return mp.get(index).get(i);
        }
    }
