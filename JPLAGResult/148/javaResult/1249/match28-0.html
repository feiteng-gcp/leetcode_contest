import java.util.HashMap;
import java.util.Map;

class SnapshotArray {

	private Map<Integer, Map<Integer, Integer>> snapshotMap = new HashMap<>(); // index -> (snapshotId -> value)
	private int currentSnapshot = 0;

	public SnapshotArray(int length) {
		for (int i = 0; i < length; i++) {
			snapshotMap.put(i, new HashMap<>());
			snapshotMap.get(i).put(currentSnapshot, 0);
		}
	}

	public void set(int index, int val) {
		if (!snapshotMap.containsKey(index)) {
			snapshotMap.put(index, new HashMap<>());
		}
		snapshotMap.get(index).put(currentSnapshot, val);
	}

	public int snap() {
		return currentSnapshot++;
	}

	public int get(int index, int snap_id) {
		int val;
		do {
			val = snapshotMap.get(index).getOrDefault(snap_id--, -1);
		} while (val == -1);
		return val;
	}
}
