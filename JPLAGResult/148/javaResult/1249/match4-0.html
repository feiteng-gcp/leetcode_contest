class SnapshotArray {

    TreeMap<Integer, Integer>[] d;
    int sid;
    
    public SnapshotArray(int length) {
        d = new TreeMap[length];
        for(int i=0; i<d.length; i++) {
            d[i] = new TreeMap<>();
        }
        sid = 0;
    }
    
    public void set(int index, int val) {
        d[index].put(sid, val);
    }
    
    public int snap() {
        sid++;
        return sid-1;
    }
    
    public int get(int index, int snap_id) {
        Map.Entry<Integer, Integer> e = d[index].floorEntry(snap_id);
        if(e == null) return 0;
        return e.getValue();
    }
}
