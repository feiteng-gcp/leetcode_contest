class SnapshotArray {
	private TreeMap<Integer, Integer>[] trees;
	private int snapId;
    public SnapshotArray(int length) {
        this.trees = new TreeMap[length];
        for (int i = 0; i < length; i++) {
            trees[i] = 
new TreeMap<Integer, Integer>();
        }
        this.snapId = 0;
    }
    
    public void set(int index, int val) {
        TreeMap<Integer, Integer> tree = trees[index];
        
tree.put(snapId, val);
    }
    
    public int snap() {
        snapId++;
        return snapId-1;
        
    }
    
    public int get(int index, int snap_id) {
        TreeMap<Integer, Integer> tree = trees[index];
        
Integer prev = tree.floorKey(snap_id);
        if (prev == null) return 0;
        
        return tree.get(prev);
    }
}
