class SnapshotArray {

     private HashMap<Integer, Integer> cloneMap(HashMap<Integer, Integer> array) {
            HashMap<Integer, Integer> map = new HashMap<>();
            for (Integer key : array.keySet()) {
                map.put(key, array.get(key));
            }
            return map;
        }

        private HashMap<Integer, Integer> array;
        private HashMap<Integer, HashMap<Integer, Integer>> snapshotMap = new HashMap<>();
        private int snapshotId = 0;
        public SnapshotArray(int length) {
            
array = new HashMap<>();
        }

        public void set(int index, int val) {
            array.put(index, val);
        }

        public int snap() {
            snapshotMap.put(snapshotId, cloneMap(array));
            snapshotId++;
            
return snapshotId-1;
        }

        public int get(int index, int snap_id) {
            return snapshotMap.get(snap_id).getOrDefault(index, 0);
        }
}
