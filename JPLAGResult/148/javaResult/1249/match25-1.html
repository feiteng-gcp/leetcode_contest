class SnapshotArray {

    
public SnapshotArray(int length) {
        maps = (TreeMap<Integer, Integer>[])new TreeMap<?,?>[length];
        for (int i = 0; i < length; ++i) {
            maps[i] = new TreeMap<Integer, Integer>();
            maps[i].put(0,0);
        
}
        snap_id_cur = 0;
    }
    
    public void set(int index, int val) {
        
maps[index].put(snap_id_cur, val);
    }
    
    public int snap() {
        ++snap_id_cur;
        return snap_id_cur - 1;
    }
    
    public int get(int index, int snap_id) {
        TreeMap<Integer, Integer> map = maps[index];
        return map.floorEntry(snap_id).getValue();
    }
    
    private int snap_id_cur = 0;
    private TreeMap<Integer, Integer>[] maps = null;
    
}
