class SnapshotArray
{

    private List<TreeMap<Integer, Integer>> store;
    private int snapId;

    public SnapshotArray(int length)
    {
        snapId = 0;
        store = new ArrayList<>();
        for (int i = 0; i < length; i++)
        {
            TreeMap<Integer, Integer> ts = 
new TreeMap<>();
            ts.put(snapId, 0);
            store.add(ts);
        }
    }

    public void set(int index, int val)
    {
        store.get(index).put(snapId, val);
    }

    public int snap()
    {
        return snapId++;
    }

    public int get(int index, int snap_id)
    {
        return store.get(index).floorEntry(snap_id).getValue();
    }
}
