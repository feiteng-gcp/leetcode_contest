class SnapshotArray {
    ArrayList<TreeMap<Integer, Integer>> altm = new ArrayList<>();
    int sid = 0;

    public SnapshotArray(int length) {
        for (int i = 0; i < length; i++) {
            altm.add(new TreeMap<Integer, Integer>());
            altm.get(i).put(0, 0);
        }
    }
    
    public void set(int index, int val) {
        
altm.get(index).put(sid, val);
    }
    
    public int snap() {
        return sid++;
    }
    
    public int get(int index, int snap_id) {
        return altm.get(index).floorEntry(snap_id).getValue();
    }
}
