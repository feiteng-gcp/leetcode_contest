class SnapshotArray {
    
    int currentSnapId = 0;
    Map<Integer, Map<Integer, Integer>> m;

    public SnapshotArray(int length) {
        m = new HashMap<>();
        m.put(currentSnapId, new HashMap<>());
    }
    
    public void set(int index, int val) {
        m.get(currentSnapId).put(index, val);
    }
    
    public int snap() {
        currentSnapId++;
        m.put(currentSnapId, new HashMap<>());
        for (Map.Entry<Integer, Integer> entry : m.get(currentSnapId - 1).entrySet()){
            m.get(currentSnapId).put(entry.getKey(), entry.getValue());
        }
        return currentSnapId - 1;
    }
    
    public int get(int index, int snap_id) {
        return m.get(snap_id).getOrDefault(index, 0);
    }
}
