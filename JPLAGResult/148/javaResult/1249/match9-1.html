class SnapshotArray {
    int sid = 0;
    TreeMap<Integer, Integer>[] data;
    public SnapshotArray(int length) {
        data = new TreeMap[length];
        for (int i = 0; i < length; ++i) data[i] = new TreeMap();
    
}
    
    public void set(int index, int val) {
        data[index].put(sid, val);
    }
    
    public int snap() {
        return sid++;
    }
    
    public int get(int index, int snap_id) {
        TreeMap<Integer, Integer> map = data[index];
        Integer id = 
map.floorKey(snap_id);
        if (null == id) return 0;
        return map.get(id);
    }
}
