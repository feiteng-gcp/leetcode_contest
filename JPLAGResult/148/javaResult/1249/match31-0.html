class SnapshotArray {

    int[] arr;
    Map<Integer,Map<Integer,Integer>> snapshot;
    int snapid;
    Map<Integer,Integer> changes;
    
    public SnapshotArray(int length) {
        arr = new int[length];
        snapshot = new HashMap<>();
        changes = new HashMap<>();
        snapid = 0;
    }
    
    public void set(int index, int val) {
        arr[index] = val;
        changes.put(index,val);
    }
    
    public int snap() {
        snapshot.put(snapid,changes);
        changes = new HashMap<>();
        return snapid++;
    }
    
    public int get(int index, int snap_id) {
        Map<Integer,Integer> map = snapshot.get(snap_id);
        if(map!=null) {
            Integer val = map.get(index);
            while(val==null&&snap_id>0) {
                map = snapshot.get(--snap_id);
                val = map.get(index);
            }
            return val==null?0:val;
        }
        return 0;
    }
}
