class SnapshotArray {
    
    Map<Integer, Map<Integer, Integer>> snaps;
    int id = 0;
    public SnapshotArray(int length) {
        snaps = new HashMap<>();
        snaps.put(id, new HashMap<>());
    }
    
    public void set(int index, int val) {
        snaps.get(id).put(index, val);
    }
    
    public int snap() {
        Map<Integer, Integer> clone = 
new HashMap();
        for(int key: snaps.get(id).keySet()){
            clone.put(key, snaps.get(id).get(key));
        }
        int returnval = id;
        id++;
        snaps.put(id, clone);
        
return returnval;
    }
    
    public int get(int index, int snap_id) {        
        return snaps.get(snap_id).getOrDefault(index, 0);
    }
}
