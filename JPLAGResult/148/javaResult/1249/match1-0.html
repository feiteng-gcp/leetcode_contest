class SnapshotArray {

    TreeMap<Integer, Integer>[] aa;
    int v = 0;
    
    public SnapshotArray(int length) {
        aa = new TreeMap[length];
        for (int i = 0; i < length; ++i) {
            aa[i] = new TreeMap<Integer, Integer>();
            aa[i].put(v, 0);
        }
    }
    
    public void set(int index, int val) {
        aa[index].put(v, val);
    }
    
    public int snap() {
        return v++;
    }
    
    public int get(int index, int snap_id) {
        return aa[index].floorEntry(snap_id).getValue();
    }
}
