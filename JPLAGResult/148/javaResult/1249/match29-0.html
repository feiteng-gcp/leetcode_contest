class SnapshotArray {

    Map<Integer, Map<Integer,Integer> > map;
    Map<Integer, Integer> cur;
    int id;
    public SnapshotArray(int length) {
        id=0;
        cur = new HashMap();
        map = new HashMap();
    }
    
    public void set(int index, int val) {
        cur.put(index, val);
        if(val==0) cur.remove(index);
    }
    
    public int snap() {
        map.put(id, new HashMap(cur));
        return id++;
    }
    
    public int get(int index, int snap_id) {
        return map.get(snap_id).getOrDefault(index,0);
    }
}
