class SnapshotArray {
    
    Map<Integer, Map<Integer, Integer>> snapid2array = new HashMap<>();
    Map<Integer, Integer> snapNow;
    int snapId = 0;
    int l;

    public SnapshotArray(int length) {
        l = length;
        snapNow = new HashMap<>();
    }
    
    public void set(int index, int val) {
        snapNow.put(index, val);
    }
    
    public int snap() {
        snapid2array.put(snapId, new HashMap<>(snapNow));
        snapId++;
        return snapId-1;
    }
    
    public int get(int index, int snap_id) {
        return snapid2array.get(snap_id).getOrDefault(index, 0);
    }
}
