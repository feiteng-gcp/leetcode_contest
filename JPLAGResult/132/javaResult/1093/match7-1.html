/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        
return helper(S, 1);
    }

    private TreeNode helper(String s, int depth) {
        if(s == null || s.isEmpty()) {
            return null;
        }

        int num = 0;
        int i = 0;
        
while(i < s.length() && Character.isDigit(s.charAt(i))) {
            num = num * 10 + (s.charAt(i) - '0');
            
            i++;
        }

        final TreeNode root = new TreeNode(num);

        if(i + depth >= s.length()) {
            return root;
        }

        int leftStart = i + depth;

        int rightStart = leftStart + 1;
        int count = 0;

        while(rightStart < s.length()) {
            if(s.charAt(rightStart) == '-') {
                count++;
            } else {
                if(count == depth) {
                    break;
                }
                count = 0;
            }
            
            rightStart++;
        }

        if(rightStart == s.length()) {
            root.left = helper(s.substring(leftStart), depth + 1);

            return root;
        }

        root.left = helper(s.substring(leftStart, rightStart - depth), depth + 1);
        root.right = helper(s.substring(rightStart), depth + 1);

        return root;
    }
}
