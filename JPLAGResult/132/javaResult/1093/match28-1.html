/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class NP {
    int depth;
    int val;
}
class Solution {
    ArrayList<NP> np = new ArrayList<>();

    public TreeNode recoverFromPreorder(String S) {
        if(S == null || S.length() == 0) return null;
        parse(S);
        TreeNode root = 
build(0, 0, np.size() - 1);
        return root;
    }

    TreeNode build(int d, int from, int to) {
        TreeNode root = new TreeNode(np.get(from).val);
        if (from == to) return root;
        int leftStart = from + 1;
        int rightStart = to;
        while (np.get(rightStart).depth != d + 1) {
            rightStart --;
        }
        if (rightStart == leftStart) {
            root.left = build(d + 1, leftStart, to);
        } else {
            root.left = build(d + 1, leftStart, rightStart - 1);
            
root.right = build(d + 1, rightStart, to);
        }
        return root;
    }

    void parse(String s) {
        int count = 0;
        boolean inCount = true;
        int num = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == '-') {
                if (inCount) {
                    count ++;
                } else {
                    NP nn = new NP();
                    nn.val = num;
                    nn.depth = count;
                    np.add(nn);
                    count = 1;
                    inCount = true;
                }
            } else {
                int n = c - '0';
                if (inCount) {
                    inCount = false;
                    num = n;
                } else {
                    num = num * 10 + n;
                }
            }
        }
        
if (num > 0) {
            NP nn = new NP();
            nn.val = num;
            nn.depth = count;
            np.add(nn);
        }
    }
}
