/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        StringBuilder sb = new StringBuilder(S);
        return helper(sb, 0);
    }
    
    private TreeNode helper(StringBuilder sb, int depth) {
        if (sb == null || sb.length() == 0) {
            return null;
        }
        int index = 0;
        while (sb.charAt(index) == '-') {
            index++;
        }
        if (index != depth) {
            return null;
        }
        int numberIndex = index;
        while (numberIndex < sb.length() && sb.charAt(numberIndex) != '-') {
            numberIndex++;
        }
        int val = Integer.valueOf(sb.substring(index, numberIndex).toString());
        sb.delete(0, numberIndex);
        TreeNode root = new TreeNode(val);
        root.left = helper(sb, depth + 1);
        root.right = helper(sb, depth + 1);
        return root;
    }
}
