/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String s) {
        if(s==null || s.length()==0){
            return null;
        }
        
        return construct( s, 1);
    }
    
    public TreeNode construct(String s, int level){
        //System.out.println(" s "+s);
        if(s==null || s.length()==0){
            return null;
        }

        int leftIndex = getIndex(s,0, level);
        if(leftIndex ==-1){
            return new TreeNode(Integer.parseInt(s));
        }
        int currentNode = Integer.parseInt(s.substring(0,leftIndex));
        //System.out.println("Currne Node "+currentNode );
        TreeNode root = new TreeNode(currentNode);
        int rightIndex = getIndex(s,leftIndex+level,level);
        if(rightIndex !=-1){
            root.left = construct(s.substring(leftIndex+level,rightIndex),level+1);
            root.right = construct(s.substring(rightIndex+level),level+1);
            
        }else{
            root.left = construct(s.substring(leftIndex+level),level+1);
        }
        
        return root;
        
    }
    
    public int getIndex(String s, int index, int freq){
        //int index = -1;
        int count=0;
        for(int i =index;i< s.length();i++){
            if(s.charAt(i)=='-'){
                count++;
            }else{
                if(count ==freq){
                    return i-freq;
                }
                count=0;
            }
        }
        
        return -1;
    }
}
