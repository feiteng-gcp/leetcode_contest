/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Integer> depthMap = new HashMap<>();
    int idx = 0;
    public TreeNode recoverFromPreorder(String S) {
        List<TreeNode> nodes = new ArrayList<>();
        int nm = 0;
        int i = 0;
        while (i < S.length() && 
Character.isDigit(S.charAt(i))) {
            nm = nm * 10 + S.charAt(i) - '0';
            i++;
        }
        TreeNode c = new TreeNode(nm);
        nodes.add(c);
        depthMap.put(c, 0);
        while (i < S.length()) {
            int d = 0, num = 0;
            while (i < S.length() && S.charAt(i) == '-') {
                d++;
                i++;
            }
            while (i < S.length() && Character.isDigit(S.charAt(i))) {
                num = num * 10 + S.charAt(i) - '0';
                i++;
            }
            TreeNode cur = new TreeNode(num);
            nodes.add(cur);
            depthMap.put(cur, d);
        }
        
return dfs(nodes, 0);
    }
    
    private TreeNode dfs(List<TreeNode> nodes, int depth) {
        if (idx == nodes.size() || depthMap.get(nodes.get(idx)) != depth) return null;
        //if (depthMap.get(nodes.get(idx)) == depth) 
        TreeNode cur = nodes.get(idx++);
        cur.left = dfs(nodes, depth + 1);
        cur.right = dfs(nodes, depth + 1);
        return cur;
    }
}
