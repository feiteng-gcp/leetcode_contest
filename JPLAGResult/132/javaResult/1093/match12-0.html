/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private int idx = 0;
    private int nodeLevel = 0;
    private int nodeVal = 0;
    public TreeNode recoverFromPreorder(String S) {
        int rootVal = 0;
        while (idx < S.length() && S.charAt(idx) != '-') {
            rootVal = rootVal * 10 + S.charAt(idx) - '0';
            idx++;
        }
        TreeNode root = new TreeNode(rootVal);
        if (idx == S.length())
            return root;
        return recover(S, root, 0);
    }
    private TreeNode recover(String S, TreeNode cur, int level) {
        int dCnt = 0;
        if (idx == S.length())
            return cur;
        while (S.charAt(idx) == '-') {
            idx++;
            dCnt++;
        }
        nodeLevel = dCnt;
        while (idx < S.length() && S.charAt(idx) != '-') {
            nodeVal = nodeVal * 10 + S.charAt(idx) - '0';
            idx++;
        }
        //System.out.println("nodeLevel:" + nodeLevel + " nodeVal:" + nodeVal + " level:" + level);
        if (nodeLevel > level) {
            cur.left = new TreeNode(nodeVal);
            nodeVal = 0;
            recover(S, cur.left, level+1);
            if (nodeLevel > level) {
                if (nodeVal != 0) {
                    cur.right = new TreeNode(nodeVal);
                    nodeVal = 0;
                    recover(S, cur.right, level+1);
                }

            }
        }
        return cur;
    }
}
