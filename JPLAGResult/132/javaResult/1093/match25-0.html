/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        int len = S.length();
        List<Integer> num = new ArrayList<Integer>();
        List<Integer> depth = new ArrayList<Integer>();
        int currNum = 0, dash = 0, i = 0;
        
        while(i < len) {
            char ch = S.charAt(i);
            if(ch == '-') {
                while(i < len && S.charAt(i) == '-') {
                    dash++; i++;
                }
                num.add(currNum);
                currNum = 0;
            } 
            else {
                while(i < len && S.charAt(i) != '-') {
                    currNum = 10 * currNum + (S.charAt(i) - '0');
                    i++;
                }
                depth.add(dash);
                dash = 0;
            }
        }
        num.add(currNum);
        // System.out.println(num.toString());
        // System.out.println(depth.toString());
        
        return helper(num, depth, 0, num.size()-1);
    }
    
    
    public TreeNode helper(List<Integer> num, List<Integer> depth, int start, int end) {
        if(start > end) return null;
        // System.out.println(start+" "+end);
        TreeNode node = new TreeNode(num.get(start));
        if(start == end) return node;
        int currDepth = depth.get(start+1);
        int i = start+2;
        for(; i <= end; i++) {
            if(depth.get(i) == currDepth)
                break;
        }
        if(i > end) {
            node.left = helper(num, depth, start+1, end);
        } else {
            node.left = helper(num, depth, start+1, i - 1);
            node.right = helper(num, depth, i, end);
        }
        return node;
    }
}
