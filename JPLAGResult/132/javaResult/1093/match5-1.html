/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        LinkedList<int[]> nodes = new LinkedList<>();
        int i = 0;
        while (i < S.length()) {
            int currDepth = 0;
            while (i < S.length() && S.charAt(i) == '-') {
                currDepth++;
                i++;
            }
            int currVal = 0;
            while (i < S.length() && S.charAt(i) >= '0' && S.charAt(i) <= '9') {
                currVal *= 10;
                currVal += S.charAt(i) - '0';
                i++;
            }
            nodes.add(new int[]{currDepth, currVal});
        }
        
return recover(-1, nodes);
    }
    
    private TreeNode recover(int prevDepth, LinkedList<int[]> nodes) {
        if (nodes.isEmpty()) {
            return null;
        }
        if (nodes.peekFirst()[0] != prevDepth + 1) {
            return null;
        }
        int[] currNodeInfo = nodes.removeFirst();
        TreeNode currNode = new TreeNode(currNodeInfo[1]);
        currNode.left = recover(currNodeInfo[0], nodes);
        currNode.right = recover(currNodeInfo[0], nodes);
        return currNode;
    }
}
