/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        Deque<int[]> q = new LinkedList<>();
        int depth = 0;
        int num = 0;
        boolean numHandled = false;
        for (char it : S.toCharArray()) {
            if (it == '-' && numHandled) {
                q.offer(new int[]{num, depth});
                depth = 1;
                num = 0;
                numHandled = false;
            } else if (it == '-') {
                depth++;
            } else {
                num = num * 10 + it - '0';
                numHandled = true;
            }
        }
        if (numHandled) {
            q.offer(new int[]{num, depth});
        }
        
        return helper(q, 0);
    }
    
    private TreeNode helper(Deque<int[]> q, int depth) {
        if (q.isEmpty()) {return null;}
        if (depth > q.getFirst()[1]) {
            return null;
        }
        assert(depth == q.getFirst()[1]);
        int[] curt = q.poll();
        TreeNode root = new TreeNode(curt[0]);
        root.left = helper(q, depth + 1);
        root.right = helper(q, depth + 1);
        return root;
    }
}
