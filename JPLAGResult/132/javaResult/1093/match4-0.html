/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        return helper(S, 0);
    }
    
    private TreeNode helper(String S, int d) {
        if (S.length() == 0) return null;
        int fd = S.indexOf('-', d + 1);
        fd = fd == -1 ? S.length() : fd;
        TreeNode root = new TreeNode(Integer.valueOf(S.substring(d, fd)));
        if (fd == S.length()) {
            return root;
        }
        int i = fd + d + 1; 
        int st = fd;
        while (i < S.length()) {
            int len = 0;
            while (i < S.length() && S.charAt(i) == '-') {
                len++;
                i++;
            }
            if (len == d + 1) {
                break;
            }
            i++;
        }
        if (i == S.length()) {
            root.left = helper(S.substring(st), d + 1);
            root.right = null;
        } else {
            root.left = helper(S.substring(st, i - d - 1), d + 1);
            root.right = helper(S.substring(i - d - 1), d + 1);
        }
        return root;
    }
}
