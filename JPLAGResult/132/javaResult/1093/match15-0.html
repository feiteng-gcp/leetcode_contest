/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int index;
    public TreeNode recoverFromPreorder(String S) {
        index = 0;
        TreeNode root = new TreeNode(getNumber(S));
        dfs(S, root, 1);
        return root;
    }
    public void dfs(String S, TreeNode node, int l) {
        if(index == S.length()) return;
        int cnt = 0;
        while(index < S.length() && S.charAt(index) == '-') {
            index ++;
            cnt ++;
        }
        if(cnt == l) {
            node.left = new TreeNode(getNumber(S));
            dfs(S, node.left, l+1);
        } else {
            index -= cnt;
            return;
        }
        cnt = 0;
        while(index < S.length() && S.charAt(index) == '-') {
            index ++;
            cnt ++;
        }
        if(cnt == l) {
            node.right = new TreeNode(getNumber(S));
            dfs(S, node.right, l+1);
        } else {
            index -= cnt;
            return;
        }
    }
    public int getNumber(String S) {
        int ret = 0;
        while(index < S.length() && S.charAt(index) != '-') {
            ret = ret*10 + S.charAt(index) - '0';
            index ++;
        }
        return ret;
    }
}
