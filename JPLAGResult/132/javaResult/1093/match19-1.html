/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public static int index = 0;
    public static int curDepth = 0;
    public TreeNode recoverFromPreorder(String S) {
        index = 0;
        curDepth = 0;
        
if(S == null || S.length() == 0) return null;
        return helper(S, 0);
    }
    
    private TreeNode helper(String s, int dep) {
        if(dep != curDepth || index >= s.length()) return null;
        int val = getVal(s);
        TreeNode root = new TreeNode(val);
        curDepth = getDepth(s);
        root.left = helper(s, dep + 1);
        root.right = helper(s, dep + 1);
        return root;
    }
    
    private int getVal(String s) {
        int j = index + 1;
        while(j < s.length() && s.charAt(j) != '-') {
            j++;
        }
        int ret = Integer.parseInt(s.substring(index, j));
        index = j;
        return ret;
    }
    
    private int getDepth(String s) {
        int j = index + 1;
        while(j < s.length() && s.charAt(j) == '-') {
            j++;
        }
        int ret = j - index;
        index = j;
        return ret;
    }
}
