class Solution {
    public TreeNode recoverFromPreorder(String S) {

        Tokenizer tokenizer = new Tokenizer(S);
        return dfs(tokenizer, 0);
    }

    private TreeNode dfs(Tokenizer tokenizer, int expectedDepth) {

        if (!tokenizer.hasNext()) { return null; }
        if (tokenizer.nextDepth() != expectedDepth) { return null; }
        TreeNode node = new TreeNode(tokenizer.nextVal());
        tokenizer.advance();

        node.left = dfs(tokenizer, expectedDepth+1);
        node.right = dfs(tokenizer, expectedDepth+1);

        return node;
    }

    class Tokenizer { // consumer needs to advance() himself after consuming once
        String S;
        int slen;
        boolean hasNext;
        int nextDepth, nextVal;
        int pos; // where we would advance on next.
        Tokenizer(String S) {
            this.S=S;
            slen=S.length();
            pos = 0;
            advance();
        }
        void advance() {
            if (pos >= slen) { hasNext = false; return; } // implicitly nextDepth,nextVal are garbage
            hasNext = true;
            // read the depth.
            nextDepth = 0;
            while(S.charAt(pos) == '-') { pos++; nextDepth++; }
            // read the val.
            nextVal = 0;
            while(pos < slen && Character.isDigit(S.charAt(pos))) {
                char c = S.charAt(pos);
                int placeval = c - '0';
                nextVal *= 10;
                nextVal += placeval;
                pos++;
            }
        }
        boolean hasNext() { return hasNext; }
        int nextDepth() { return nextDepth; }
        int nextVal() { return nextVal; }
    }
}
