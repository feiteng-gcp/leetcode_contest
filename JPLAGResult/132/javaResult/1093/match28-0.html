/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        List<Token> tokens = tokenize(S);
        return parse(tokens);
    }
    
    private TreeNode parse(List<Token> tokens) {
        TreeNode dummy = new TreeNode(0);
        parse(tokens, 1, dummy, tokens.get(0));
        return dummy.left;
    }
    
    private int parse(List<Token> tokens, int index, TreeNode parent, Token pt) {
        if (index >= tokens.size()) return index;
        Token cur = tokens.get(index);
        if (cur.level == pt.level + 1) { //left
            TreeNode left = new TreeNode(cur.val);
            parent.left = left;
            index = parse(tokens, index+1, left, cur);
        }
        
        if (index >= tokens.size()) return index;
        cur = tokens.get(index);
        if (cur.level == pt.level + 1) { //right
            TreeNode right = new TreeNode(cur.val);
            parent.right = right;
            index = parse(tokens, index+1, right, cur);
        }
        return index;
    }
    
    private List<Token> tokenize(String s) {
        List<Token> tokens = new ArrayList<>();
        tokens.add(new Token(-1, 0));
        
        int index = 0;
        while(index<s.length()) {
            StringBuilder sb = new StringBuilder();
            int level = 0;
            while(index<s.length() && s.charAt(index)=='-') {
                level++; index++;
            }
            while(index<s.length() && s.charAt(index)!='-') {
                sb.append(s.charAt(index));
                index++;
            }
            tokens.add(new Token(level, Integer.parseInt(sb.toString())));
        }
        return tokens;
    }
    
    public static class Token {
        int level;
        int val;
        
        public Token(int level, int val) { this.level = level; this.val = val; }
    }
}
