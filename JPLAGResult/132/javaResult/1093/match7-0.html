/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        Reader r = new Reader();
        r.s = S;
        r.next();
        return recoverFromPreorder(r, 0);
    }
    
    public TreeNode recoverFromPreorder(Reader reader, int depth) {
        if(reader.depth() != depth)
        {
            return null;
        }
        TreeNode node = new TreeNode(reader.val());
        reader.next();
        node.left = recoverFromPreorder(reader, depth + 1);
        node.right = recoverFromPreorder(reader, depth + 1);
        return node;
    }
    
    public static class Reader{
        String s;
        int rpos;
        int begin;
        int val;
        
        
        public void next(){
            if(rpos >= s.length())
            {
                begin = -1;
                val = -1;
                return;
            }
            
            begin = 0;
            while(rpos < s.length() && s.charAt(rpos) == '-')
            {
                rpos++;
                begin++;
            }
            val = 0;
            while(rpos < s.length() && s.charAt(rpos) >= '0' && s.charAt(rpos) <= '9')
            {
                val = val * 10 + s.charAt(rpos) - '0';
                rpos++;
            }
        }
        
        public int depth(){
            return begin;
        }
        
        public int val(){
            return val;
        }
    }
}
