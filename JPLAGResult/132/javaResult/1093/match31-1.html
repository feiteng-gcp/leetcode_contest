/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    public TreeNode helper(String S, int start, int end){
        if(start == -1 || end == -1){
            return null;
        }
        if(end < start){
            return null;
        }else{
            int temp = 0;
            int curr = start;
            while(curr <= end &&  Character.isDigit(S.charAt(curr))){
                temp = temp*10 + (int) (S.charAt(curr) - '0');
                curr++;
            }
            TreeNode ret = new TreeNode(temp);
            if(curr > end){
                return ret;
            }
            int lev = 0;
            while(curr <= end && S.charAt(curr) == '-'){
                curr++;
                lev++;
            }
            int leftStart = curr;
            int leftEnd = end;
            int ss = curr;
            int ss2 = curr;
            int rightStart = -1;
            int rightEnd = -1;
            
            while(ss <= end)
            {
                ss2 = ss;
                while(ss2 <= end && Character.isDigit(S.charAt(ss2)) ){
                    ss2++;
                }
                int currLevel = 0;
                ss = ss2;
                while(ss <= end && S.charAt(ss) == '-'){
                    ss++;
                    currLevel++;
                }
                if(currLevel == lev){
                    leftEnd = ss2-1;
                    rightStart = ss;
                    rightEnd = end;
                    break;
                }
            }
            TreeNode left = helper(S, leftStart, leftEnd);
            TreeNode right = helper(S, rightStart, rightEnd);
            ret.left = left;
            ret.right = right;
            return ret;
        }
    
}
    
    public TreeNode recoverFromPreorder(String S) {
        int n = S.length();
        return helper(S, 0, n-1);
    }
}
