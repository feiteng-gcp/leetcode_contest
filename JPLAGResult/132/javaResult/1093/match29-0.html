/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        char[] arr = S.toCharArray();
        int[] idx = new int[1];
        return dfs(arr, idx);
    }
    
    private TreeNode dfs(char[] arr, int[] idx) {
        System.out.print(idx[0]);
        if (idx[0] == arr.length) {
            return null;
        }

        
        int curdepth = readcnt(arr, idx[0]);
        idx[0] += curdepth;
        
        int val = read(arr, idx);
        
        TreeNode root = new TreeNode(val);
        
        int count = readcnt(arr, idx[0]);
        if (count == 0) {
            return root;
        } else if (curdepth >= count) {
            return root;
        }
        
        root.left = dfs(arr, idx);
        count = readcnt(arr, idx[0]);
        if (count == 0 || curdepth >= count) {
            return root;
        }
        root.right = dfs(arr, idx);
        return root;
    }
    
    private int readcnt(char[] arr, int idx) {
        int count = 0;
        while (idx < arr.length && arr[idx] == '-') {
            idx++;
            count++;
        }       
        return count;
    }
    
    private int read(char[] arr, int[] idx) {
        int cur = 0;
        while (idx[0] < arr.length && arr[idx[0]] >= '0' && arr[idx[0]] <= '9') {
            cur = cur * 10 + (arr[idx[0]] - '0');
            idx[0]++;
        }
        return cur;
    }
}
