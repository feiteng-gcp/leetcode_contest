/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        int curPos = 0;
        char[] chars = S.toCharArray();
        
        List<ValWithDepth> list = new ArrayList<ValWithDepth>();
        while (curPos < S.length())
        {
            int depth = 0;
            while (S.charAt(curPos) == '-')
            {
                depth++;
                curPos++;
            }
            
            int nextDashPos = S.indexOf('-', curPos);
            
            if (nextDashPos == -1)
            {
                nextDashPos = S.length();
            }
            
            int val = Integer.parseInt(S.substring(curPos, nextDashPos));
            
            list.add(new ValWithDepth(val, depth));
            
            curPos = nextDashPos;
        }
        
        
return constructTree(list, 0, list.size() - 1);
    }
    
    private TreeNode constructTree(List<ValWithDepth> list, int startIndex, int endIndex)
    {
        if (startIndex > endIndex)
        {
            return null;
        }
        else if (startIndex == endIndex)
        {
            return new TreeNode(list.get(startIndex).val);
        }
        else
        {
            TreeNode root = new TreeNode(list.get(startIndex).val);
            int rootDepth = list.get(startIndex).depth;
            
            int rightSubTreeStartIndex = -1;
            for (int i = startIndex + 1; i <= endIndex; i++)
            {
                if (list.get(i).depth == rootDepth + 1 && i != startIndex + 1)
                {
                    rightSubTreeStartIndex = i;
                    break;
                
}
            }
            
            if (rightSubTreeStartIndex == -1)
            {
                root.left = constructTree(list, startIndex + 1, endIndex);
            }
            else
            {
                root.left = constructTree(list, startIndex + 1, rightSubTreeStartIndex - 1);
                root.right = constructTree(list, rightSubTreeStartIndex, endIndex);
            }
            
            return root;
        }
    }
    
    class ValWithDepth
    {
        public int val;
        public int depth;
        
        public ValWithDepth(int val, int depth)
        {
            this.val = val;
            this.depth = depth;
        }
    }
}
