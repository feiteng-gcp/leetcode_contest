/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode recoverFromPreorder(String S) {
        
        if(S == null || S.isEmpty()) return null;
        
        int rootVal = 0;
        
        
int i = 0;
        while(i < S.length() && S.charAt(i) != '-') {
            rootVal = rootVal*10+S.charAt(i)-'0';
            ++i;
        }
        
        TreeNode root = new TreeNode(rootVal);
        
        
        
        int dashCount = 0;
        
        while(i < S.length() && S.charAt(i) == '-')  {
            ++dashCount;
            ++i;
        }
        
        
        int start = i;
        
        int curDashCount = 0;
        
        while( i < S.length()) {
            if(S.charAt(i) != '-') {
                if(curDashCount == dashCount) {
                    root.left = recoverFromPreorder(S.substring(start, i));
                    root.right = recoverFromPreorder(S.substring(i, S.length()));

                    return root;
                
}
                curDashCount = 0;
            } else {
                ++curDashCount;
            }
            
            ++i;
        }
        
        root.left = recoverFromPreorder(S.substring(start, S.length()));
        
        return root;
        
    }
}
