/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private int maxDiff = 0;
    public int maxAncestorDiff(TreeNode root) {
    	maxAncestorDiff(root, root.val, root.val);
    	return maxDiff;
    }
    
	private void maxAncestorDiff(TreeNode root, int max, int min) {
		if (root == null) {
			return;
		}
		maxDiff = Math.max(Math.abs(max - root.val), maxDiff);
		maxDiff = Math.max(Math.abs(min - root.val), maxDiff);
		max = Math.max
(root.val, max);
		min = Math.min(root.val, min);
		maxAncestorDiff(root.left, max, min);
		maxAncestorDiff(root.right, max, min);
	}
}
