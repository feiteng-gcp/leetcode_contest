/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
	int res = 0;
	public int maxAncestorDiff(TreeNode root) {
		g(root, root.val, root.val);
		return res;
	}

	private void g(TreeNode root, int max, int min) {
		if (root == null) {
			res = Math.max(res, max - min);
			return;
		}
		if (max < root.val) {
			max = root.val;
		}
		if (min > root.val) {
			min = root.val;
		}
		g(root.left, max, min);
		g(root.right, max, min);
	}

}
