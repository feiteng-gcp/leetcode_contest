/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int maxAncestorDiff(TreeNode root) {
        int[] ret = new int[1];
        dfs(root.left, root.val, root.val, ret);
        dfs(root.right, root.val, root.val, ret);
        return ret[0];
    }
    
    private void dfs(TreeNode root, 
int vmax, int vmin, int[] ret) {
        if (root == null) {
            return;
        }
        ret[0] = Math.max(ret[0], Math.abs(vmax - root.val));
        ret[0] = Math.max(ret[0], Math.abs(vmin - root.val));
        dfs(root.left, Math.max(vmax, root.val), Math.min(vmin, root.val), ret);
        dfs(root.right, Math.max(vmax, root.val), Math.min(vmin, root.val), ret);
    }
}
