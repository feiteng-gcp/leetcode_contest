/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int maxAbs;
    
    public int maxAncestorDiff(TreeNode root) {
        maxAbs = 0;
        dfs(root, root.val, root.val);
        return maxAbs;
    }
    
    public void dfs(TreeNode node, int max, int min) {
        if(node == null) return;
        maxAbs = Math.max(maxAbs, Math.max(Math.abs(node.val - max),
                                           Math.abs(node.val - min)));
        dfs(node.left, max>node.val?max:node.val, min<node.val?min:node.val);
        dfs(node.right, max>node.val?max:node.val, min<node.val?min:node.val);
        return;
    }
}
