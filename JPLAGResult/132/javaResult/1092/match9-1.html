/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public static int res = 0;
    public int maxAncestorDiff(TreeNode root) {
        if (root == null) {
            return -1;
        }
        
res = 0;
        helper(root);
        return res;
    }
    private int[] helper(TreeNode root) {
        int[] maxMin = new int[]{root.val, root.val};
        if(root.left != null) {
            int[] leftRes = helper(root.left);
            res = Math.max(res, Math.max(
Math.abs(root.val - leftRes[0]), Math.abs(root.val - leftRes[1])));
            maxMin[0] = Math.min(maxMin[0], leftRes[0]);
            maxMin[1] = Math.max(maxMin[1], leftRes[1]);
        }
        if(root.right != null) {
            int[] rightRes = helper(root.right);
            res = Math.max(res, Math.max(
Math.abs(root.val - rightRes[0]), Math.abs(root.val - rightRes[1])));
            maxMin[0] = Math.min(maxMin[0], rightRes[0]);
            maxMin[1] = Math.max(maxMin[1], rightRes[1]);
        }
        return maxMin;     
    }
}
