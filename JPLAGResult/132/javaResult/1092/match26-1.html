/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private int maxDiff;
    public int maxAncestorDiff(TreeNode root) {
        if(root == null) {
            return 0;
        }
        
        maxDiff = 0;
        
        
helper(root, root.val, root.val);
        
        return maxDiff;
    }
    
    private void helper(TreeNode root, int preMax, int preMin) {
        if(root == null) {
            return;
        }
        
        maxDiff = Math.max(maxDiff, Math.abs(preMax - root.val));
        maxDiff = Math.max(maxDiff, Math.abs(preMin - root.val));
        
        int newMax = Math.max(preMax, root.val);
        int newMin = Math.min(preMin, root.val);
        
        helper(root.left, newMax, newMin);
        helper(root.right, newMax, newMin);
    }
}
