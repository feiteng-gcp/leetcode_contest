/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int max = 0;
    public int maxAncestorDiff(TreeNode root) {
        maxAncestorDiffRec(root, root.val, root.val);
        return max;
    }
    
    public void maxAncestorDiffRec(TreeNode root, int min, int max){
        if(root == null)
            return;
        if(Math.abs(root.val -max)>this.max ){
            this.max = Math.abs(root.val -max);
            
        }
        
        if(Math.abs(root.val-min)>this.max){
            this.max = Math.abs(root.val -min);
        }
        max = Math.max(max, root.val);
        min = Math.min(min, root.val);
        //System.out.println(min+ " "+max+ " "+root.val);
        
        maxAncestorDiffRec(root.left, min, max);
        maxAncestorDiffRec(root.right, min, max);
    }
}
