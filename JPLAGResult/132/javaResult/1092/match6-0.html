/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int maxAncestorDiff(TreeNode root) {
        int[] res = new int[1];
        helper(root, res);
        return res[0];
    }
    
    public int[] helper(TreeNode node, int[] res) {
        if (node.left == null && node.right == null) {
            return new int[]{node.val, node.val};
        }
        int[] r = new int[]{node.val, node.val};
        if (node.left != null) {
            int[] left = helper(node.left, res);
            res[0] = Math.max(res[0], Math.max(Math.abs(node.val - left[0]), Math.abs(node.val - left[1])));
            r[0] = Math.min(r[0], left[0]);
            r[1] = Math.max(r[1], left[1]);
        }
        
        if (node.right != null) {
            int[] right = helper(node.right, res);
            res[0] = Math.max(res[0], Math.max(Math.abs(node.val - right[0]), Math.abs(node.val - right[1])));
            r[0] = Math.min(r[0], right[0]);
            r[1] = Math.max(r[1], right[1]);
        }
        return r;
    }
}
