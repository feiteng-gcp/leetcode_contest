import java.util.*;

class Solution {

    private int globalMaxDiff = 0;

    public int maxAncestorDiff(TreeNode root) {
        search(root, root.val, root.val);
        return globalMaxDiff;
    }

    private void search(TreeNode node, int minAbove, int maxAbove) {

        if (node == null) { return; }

        int maxCandidate = Math.max(Math.abs(node.val-minAbove), Math.abs(node.val-maxAbove));
        globalMaxDiff = Math.max(globalMaxDiff, maxCandidate);

        int minAboveForChildren = Math.min(minAbove, node.val);
        int maxAboveForChildren = Math.max(maxAbove, node.val);
        search(node.left, minAboveForChildren, maxAboveForChildren);
        search(node.right, minAboveForChildren, maxAboveForChildren);
    }
}
