/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int maxAncestorDiff(TreeNode root) {
        int[] result = new int[] {0};
        dfs(root, result);
        return result[0];
    }
    
    private int[] dfs(TreeNode root, int[] result) {
        if (root.left == null && root.right == null) {
            return new int[] {root.val, root.val}; 
        }
        int[] cur = new int[] {root.val, root.val};
        if (root.left != null) {
            int[] left = dfs(root.left, result);
            result[0] = Math.max(result[0], Math.abs(root.val - left[0]));
            result[0] = Math.max(result[0], Math.abs(root.val - left[1]));
            cur[0] = Math.max(cur[0], left[0]);
            cur[1] = Math.min(cur[1], left[1]);
        }
        if (root.right != null) {
            int[] left = dfs(root.right, result);
            result[0] = Math.max(result[0], Math.abs(root.val - left[0]));
            result[0] = Math.max(result[0], Math.abs(root.val - left[1]));
            cur[0] = Math.max(cur[0], left[0]);
            cur[1] = Math.min(cur[1], left[1]);
        }
        return cur;
    }
}
