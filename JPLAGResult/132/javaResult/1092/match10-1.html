/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int res;
    public int maxAncestorDiff(TreeNode root) {
        this.res = Integer.MIN_VALUE;
        getMinMax(root);
        return res;
    }
    int MAX = 1000000;
    int [] getMinMax(TreeNode root) {
        
        if(root == null) return new int[]{MAX, -MAX};
        
        if(root.left == null && root.right == null) return new int[]{root.val, root.val};
        
        
int [] minMax = new int[]{root.val, root.val};
        
        if(root.left != null) {
            int [] l= getMinMax(root.left);
            res = Math.max(res, Math.max(
Math.abs(root.val - l[0]), Math.abs(l[1] - root.val)));
            minMax[0] = Math.min(minMax[0], l[0]);
            minMax[1] = Math.max(minMax[1], l[1]);
        }
        
        if(root.right != null) {
            // int [] r = getMinMax(root.right);
            int [] l= getMinMax(root.right);
            res = Math.max(res, Math.max(
Math.abs(root.val - l[0]), Math.abs(l[1] - root.val)));
            minMax[0] = Math.min(minMax[0], l[0]);
            minMax[1] = Math.max(minMax[1], l[1]);
        }
        
        return minMax;
        
    }
}
