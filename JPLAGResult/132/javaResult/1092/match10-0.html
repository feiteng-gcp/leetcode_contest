/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int max = 0;
    public int maxAncestorDiff(TreeNode root) {
        max = 0;
        traverse(root);
        return max;
    }
    
    public int[] traverse(TreeNode root) {
        int[] extremes = new int[]{root.val, root.val};
        if (root.left != null) {
            int[] temp = traverse(root.left);
            max = Math.max(Math.abs(root.val - temp[0]), max);
            max = Math.max(Math.abs(root.val - temp[1]), max);
            extremes[0] = Math.min(extremes[0], temp[0]);
            extremes[1] = Math.max(extremes[1], temp[1]);
        }
        if (root.right != null) {
            int[] temp = traverse(root.right);
            max = Math.max(Math.abs(root.val - temp[0]), max);
            max = Math.max(Math.abs(root.val - temp[1]), max);
            extremes[0] = Math.min(extremes[0], temp[0]);
            extremes[1] = Math.max(extremes[1], temp[1]);
        }
        return extremes;
    }
}
