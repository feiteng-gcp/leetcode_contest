/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int res = Integer.MIN_VALUE;
    public int maxAncestorDiff(TreeNode root) {
        helper(root.left, root.val, root.val); 
        helper(root.right, root.val, root.val);
        return res;
    
}
    private void helper(TreeNode node, int max, int min) {
        if (node == null) {
            return;
        }
        res = Math.max(res, Math.abs(max - node.val));
        res = Math.max(res, 
Math.abs(min - node.val));
        helper(node.left, Math.max(node.val, max), Math.min(node.val, min));
        helper(node.right, Math.max(node.val, max), Math.min(node.val, min));
    }
}
