/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
int res = 0;

			ans[0] = Math.min(ans[0], l[0]);
			ans[1] = Math.max(ans[1], l[1]);
		}

		if (cur.right != null) {
			int[] l = solve(cur.right);
			res = Math.max(res, Math.max(Math.abs(cur.val - l[0]), Math.abs(cur.val - l[1])));
			ans[0] = Math.min(ans[0], l[0]);
			ans[1] = Math.max(ans[1], l[1]);
		}
		return ans;
	}
	public int maxAncestorDiff(TreeNode root) {
		solve(root);
		return res;
	}
}
