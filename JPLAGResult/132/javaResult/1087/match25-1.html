class Solution {
    public int longestArithSeqLength(int[] A) {
        int max = 0;
        
        Map<Integer, Integer>[] dp = new Map[A.length];
        
        for(int i = 0; i < A.length; i++) {
            dp[i] = new HashMap<>();
        }
        
        for(int i = 0; i < A.length; i++) {
            for(int j = i + 1; j < A.length; j++) {
                int diff = A[i] - A[j];
                
                
dp[j].put(diff, dp[i].getOrDefault(diff, 0) + 1);
                max = Math.max(max, dp[j].get(diff));
            }
        }
        
        return max + 1;
    }
}
