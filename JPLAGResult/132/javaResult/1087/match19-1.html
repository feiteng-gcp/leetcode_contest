class Solution {
    public int longestArithSeqLength(int[] A) {
        
int ans = 2;
        Map<Integer, Integer>[] dp = new Map[A.length];
        for (int i=0; i<A.length; ++i) {
            dp[i] = new HashMap<>();
            for (int j=0; j<i; ++j) {
                int tmp = A[i]-A[j];
                dp[i].put(tmp, Math.max(dp[i].getOrDefault(tmp, 0), dp[j].getOrDefault(tmp, 1)+1));
                ans = Math.max(ans, dp[i].get(tmp));
            }
        }
        return ans;
    }
}
