class Solution {
    public int longestArithSeqLength(int[] A) {
       	int n=A.length;
		if(n==0||n==1)return n;
		int rs=Integer.MIN_VALUE;
		for(int i=0;i<n;i++) {
			for(int j=i+1;j<n;j++) {
				int cha=A[j]-A[i];
				int last=A[j];
				int count=2;
				for(int k=j+1;k<n;k++) {
					if(A[k]-last==cha) {
						count++;
						last=A[k];
					}
				}
				rs=Math.max(count, rs);
			}
		}
		return rs;
    }
}
