        //use array of hashmaps to save
        int n = A.length;
        if (n == 0)
            return 0;
        Map<Integer, Integer>[] count = new Map[n];
        int ans = 1;
        
for (int i = 0; i < n; i++){
            count[i] = new HashMap<>();
            for (int j = 0; j < i; j++){
                int diff = A[i] - A[j];
                int curr = count[i].getOrDefault(diff, 1);
                int prev = count[j].getOrDefault(diff, 1);
                if (prev+1 > curr){
                    count[i].put(diff, prev+1);
                }
                
ans = Math.max(ans, count[i].getOrDefault(diff, 1));
            }
        }
        return ans;
        
        
    }
