class Solution {
    public int longestArithSeqLength(int[] A) {
        int len = A.length, res = 0;
        List<HashMap<Integer, Integer>> dp = new ArrayList<>();
        for (int i = 0; i < A.length; i++) {
            dp.add(new HashMap<>());
        }
        
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < i; j++) {
                int diff = A[i] - A[j];
                dp.get(i).put(diff, dp.get(j).getOrDefault(diff, 1) + 1);
                res = Math.max(res, dp.get(i).get(diff));
            }
        }
        
        return res;
    }
}
