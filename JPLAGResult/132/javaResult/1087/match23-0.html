class Solution {
    public int longestArithSeqLength(int[] A) {
        List<Map<Integer, Integer>> maps = new ArrayList<>();
        int res = 2;
        for (int i = 1; i < A.length; i++) {
            int num = A[i];
            int dis = num - A[0];
            Map<Integer, Integer> map = new HashMap<>();
            map.put(dis, 2);
            for (int j = 1; j < i; j++) {
                dis = num - A[j];
                map.put(dis, maps.get(j - 1).getOrDefault(dis, 1) + 1);
                res = Math.max(res, map.get(dis));
            }
            maps.add(map);
        }
        return res;
    }
}
