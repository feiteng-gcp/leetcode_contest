class Solution {
    public int longestArithSeqLength(int[] A) {
        int ALen = A.length;
        if (ALen < 3)
            return ALen;
        
int maxLen = 2;
        for (int i = 0; i < ALen-1; i++) {
            for (int j = i+1; j < ALen; j++) {
                int diff = A[j] - A[i];
                int prev = A[j];
                int count = 2;
                for (int k = j+1; k < ALen; k++) {
                    if (A[k] == prev + diff){
                        count++;
                        prev += diff;
                    
}
                }
                if (count > maxLen){
                    maxLen = count;
                }
            }
        }
        return maxLen;
    }
}
