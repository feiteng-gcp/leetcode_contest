class Solution {
    public int longestArithSeqLength(int[] A) {

		int res = 0;
		
		for (int i = 0; i + 1 < A.length; i++) {
			for (int j = i + 1; j < A.length; j++) {
				
				int d = A[j] - A[i];
				
				int next = A[j] + d;
				int ans = 2;
				for (int k = j + 1; k < A.length; k++) {
					if (A[k] == next) {
						ans++;
						next += d;
					}
				}
				res = Math.max(res, ans);
				
			}
		}
		return res;

	}
}
