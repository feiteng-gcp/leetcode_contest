class Solution {
    public int longestArithSeqLength(int[] A) {
        int n = A.length;
        HashMap<Integer,Integer>[] rec = new HashMap[n];
        for(int i=0;i<n;i++) rec[i]=new HashMap<>()
;
        int ans = 1;
        for(int i=1;i<n;i++){
            for(int j=0;j<i;j++){
                int dif = A[i]-A[j];
                int cur = 1+rec[j].getOrDefault(dif,1);
                ans = Math.max(ans,cur);
                rec[i].put(dif,Math.max(cur,
rec[i].getOrDefault(dif,0)));
            }
        }
        return ans;
    }
}
