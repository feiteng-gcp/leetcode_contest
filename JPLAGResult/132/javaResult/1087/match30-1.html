class Solution {
    public int longestArithSeqLength(int[] A) {
        int ans = 0;
        Map<Integer, Map<Integer, Integer>> map = new HashMap<>();
        for(int i=1;i<A.length;i++) {
            map.put(i, new HashMap<>());
            
            for(int j=i-1;j>=0;j--) {
                if(map.containsKey(j) && map.get(j).containsKey(A[i]-A[j])) {
                    if(!map.get(i).containsKey(A[i]-A[j])) {
                        map.get(i).put(A[i]-A[j], 1);
                    }
                    map.get(i).put(A[i]-A[j], Math.max(map.get(i).get(A[i]-A[j]), map.get(j).get(A[i]-A[j])+1));
                } else {
                    if(!map.get(i).containsKey(A[i]-A[j])) {
                        map.get(i).put(A[i]-A[j], 1);
                    }
                }
                
ans = Math.max(ans, map.get(i).get(A[i]-A[j]));
            }
        }
        return ans+1;
    }
}
