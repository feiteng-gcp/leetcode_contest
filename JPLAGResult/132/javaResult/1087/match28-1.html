class Solution {
    public int longestArithSeqLength(int[] A) {
        int res = 0;
        int N = A.length;
        Map<Integer,TreeSet<Integer>> m = new HashMap();
        for (int i = N - 1; i >= 0; i--) {
            m.computeIfAbsent(A[i], (key) -> 
new TreeSet()).add(i);
        }
        for (int i = 0; i < N; i++) {
            for (int j = i + 1; j < N; j++) {
                int diff = A[j] - A[i];
                if (diff == 0)
                    continue;
                int len = 2;
                int next = A[j] + diff;
                int idx = j;
                while (m.containsKey(next) && m.get(next).higher(idx) != null) {
                    idx = m.get(next).higher(idx);
                    next += diff;
                    len++;
                }
                res = 
Math.max(len, res);
            }
        }
        return res;
    }
}
