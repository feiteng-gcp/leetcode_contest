class Solution {
    public int longestArithSeqLength(int[] nums) {
        int n = nums.length;
        if (n <= 1)
            return n;
        int ans = 0;
        HashMap<Integer, Integer>[] maps = new HashMap[n];
        for (int i=0; i<n; i++) {
            maps[i] = new HashMap<>();
            for (int j=0; j<i; j++) {
                int diff = nums[i] - nums[j];
                maps[i].put(diff, Math.max(maps[i].getOrDefault(diff, 0), maps[j].getOrDefault(diff, 0)+1));
                ans = Math.max(ans, maps[i].get(diff));
            }
        }
        return ans+1;
    }
}
