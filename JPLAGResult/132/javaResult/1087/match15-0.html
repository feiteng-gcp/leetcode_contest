class Solution {
public int longestArithSeqLength(int[] a) {
    int result = 1;
    for(int i=0; i<a.length;i++){
      for(int j=i+1; j< a.length;j++){
        result = Math.max(result, findLen(a, i, j));
      }
    }
    return result;
  }

  private int findLen(int[] a, int i, int j) {
    int result =2;
    int sub = a[j]-a[i];
    int current = a[j];
    for(int k = j+1; k< a.length;k++){
      if(a[k] == current + sub){
        result++;
        current = a[k];
      }
    }
    return result;
  }
}
