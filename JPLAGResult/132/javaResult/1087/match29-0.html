class Solution {
   public int longestArithSeqLength(int[] A) {
        int n = A.length;
        Map<Integer, Integer>[] maps = new Map[n];

        for (int i = 0; i < n; i++) {
            maps[i] = new HashMap<>();
        }

        int ans = 1;
        for (int i = 1; i < n; i++) {
            for (int j = i - 1; j >= 0; j--) {
                int d = A[i] - A[j];
                int t = maps[i].getOrDefault(d, 0);
                if (maps[j].containsKey(d)) {
                    t = Math.max(t, maps[j].get(d) + 1);
                } else {
                    t = Math.max(t, 2);
                }
                maps[i].put(d, t);
                ans = Math.max(ans, t);
            }
        }
        return ans;
    }
}
