class Solution {
    public int longestArithSeqLength(int[] A) {
        int n = A.length;
        Map<Integer, Integer>[] map = (Map<Integer, Integer>[]) new HashMap[n];
        int res = 2;
        
for (int i = 0; i < n; i++) {
            map[i] = new HashMap<Integer, Integer>();
            map[i].put(0, 1);
            for (int j = 0; j < i; j++) {
                int diff = A[i] - A[j];
                int tmp = 1 + map[j].getOrDefault(diff, 1);
                res = Math.max(tmp, res);
                map[i].put(diff, 
Math.max(tmp, map[i].getOrDefault(diff,0)));
            }
        }
        return res;
    }
}
