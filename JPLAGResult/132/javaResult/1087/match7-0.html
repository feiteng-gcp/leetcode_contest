class Solution {
    public int longestArithSeqLength(int[] A) {
        if(A == null) return 0;
        int ans = 0;
        for(int i = A.length - 1; i >=0; i--){
            for(int j = i - 1; j>=0; j--){
                int tmp = 2;
                int diff = A[j] - A[i];
                int curr = A[j];
                for(int k = j - 1; k >= 0; k--){
                    if(A[k] - curr == diff){
                        tmp++;
                        curr = A[k];
                    }
                }
                ans = Math.max(ans, tmp);
            }
        }
        return ans;
    }
}
