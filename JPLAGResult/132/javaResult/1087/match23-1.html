class Solution {
    
    public int longestArithSeqLength(int[] A) {
        ArrayList<HashMap<Integer,Integer>> al = new ArrayList<>();
        // ;
        int ans = 0;        
        for(int i=0;i<A.length;i++){
            HashMap<Integer,Integer> hm = new HashMap<>();
            for(int j = 0;j<i;j++){
                int x = A[i]-A[j];
                int y = 
al.get(j).getOrDefault(x,1)+1;
                ans = Math.max(ans,y);
                hm.put(x,y);
            }
            al.add(hm);
        }
        return ans;
    }
}
