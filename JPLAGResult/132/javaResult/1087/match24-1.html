class Solution {
    public int longestArithSeqLength(int[] A) {
    Map<Integer, Map<Integer, Integer>> outer = new HashMap<>();  //外层是下标，内层的key是差值
    int ans = 0;
    for (int i = 0; i < A.length; i++) {
      HashMap<Integer, Integer> map = new HashMap<>();
      for (int j = 0; j < i; j++) {
        Map<Integer, Integer> inner = outer.get(j);
        int diff = A[i] - A[j];
        int val = 
inner.getOrDefault(diff, 0) + 1;
        ans = Math.max(ans, val);
        map.put(diff, val);
      }
      outer.put(i, map);
    }
    for (int temp : outer.get(A.length - 1).values()) {
      ans = Math.max(ans, temp);
    }
    return ans + 1;
  }
}
