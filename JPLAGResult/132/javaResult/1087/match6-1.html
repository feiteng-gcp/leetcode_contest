class Solution {
    public int longestArithSeqLength(int[] A) {
        
int max = 2;
        for (int i = 0; i < A.length - 1; i++) {
            for (int j = i + 1; j < A.length; j++) {
                int d = A[i] - A[j];
                int cur = A[j];
                int len = 2;
                for (int k = j + 1; k < A.length; k++) {
                    if (cur - A[k] == d) {
                        len++;
                        cur = A[k];
                    }
                }
                max = Math.max(max, len);
            }
        }
        return max;
    }
}
