class Solution {
    public int longestArithSeqLength(int[] A) {
        if(A.length == 0) return 0;
        int max = 1;
        Map<Integer,Integer>[] maps = new HashMap[A.length];
        for(int i = 0 ; i < A.length;i++) {
            maps[i] = new HashMap<>();
            maps[i].put(0,1);
            for(int j = 0;j<i;j++) {
                int diff = A[i] - A[j];
                int cnt = maps[j].getOrDefault(diff,1)+1;
                cnt = Math.max(maps[i].getOrDefault(diff,1),cnt);
                max = Math.max(max,cnt);
                maps[i].put(diff,cnt);
            }
        }
        return max;
    }
}
