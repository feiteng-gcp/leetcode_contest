class Solution {
    public int longestArithSeqLength(int[] A) {
        int n = A.length;
        boolean[][] dp = new boolean[n][n];
        int ans = 0;
        for (int i = 0;i < n;i++) {
            for (int j = i + 1;j < n;j++) {
                if (dp[i][j]) continue;
                int diff = A[j] - A[i];
                int prev = j;
                int count = 2;
                for (int k = j + 1;k < n;k++) {
                    if (A[k] == A[prev] + diff) {
                        count++;
                        dp[prev][k] = true;
                        prev = k;
                    }
                }
                
ans = Math.max(ans, count);
            }
        }
        return ans;
    }
}
