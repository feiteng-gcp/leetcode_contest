class Solution {
    public int longestArithSeqLength(int[] A) {
        int n = A.length;
        int[][] dp = new int[n][n];
        int res = 2;
        for (int i = 0; i < n - 1; ++i) {
            for (int j = i + 1; j < n; ++j) {
                dp[i][j] = 2;
                for (int k = 0; k < i; ++k) {
                    if (A[j] - A[i] == A[i] - A[k]) {
                        dp[i][j] = dp[k][i] + 1;
                        res = Math.max(dp[i][j], res);
                    }
                }
            }
        }
        return res;
    }
    
    // dp[i][j]
}
