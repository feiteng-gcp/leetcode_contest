class Solution {
    public int longestArithSeqLength(int[] A) {
        int n = A.length;
        Map<Integer, Integer>[] map = new HashMap[n];
        map[0] = new HashMap();
        int longest = 1;
        for(int i = 1; i < n; i++)
        {
            map[i] = new HashMap();
            for(int j = 0; j < i; j++)
            {
                int d = A[i] - A[j];
                int l = map[j].getOrDefault(d, 1) + 1;
                map[i].put(d, Math.max(l, map[i].getOrDefault(d, 1)));
                longest = Math.max(longest, l);
            }
        }
        
        return longest;
    }
}
