class Solution {
public boolean divisorGame(int n) {
    boolean[] results = new boolean[n+1];
    results[0] = false;
    results[1] = false;
    //int turn = 1;
    for(int i=2;i<n+1;i++){
      results[i] = false;
      for(int j=1;j<i;j++){
        if(!results[i-j] && i%j==0){
          results[i] = true;
        }
      }
    }
    return results[n];
  }
}
