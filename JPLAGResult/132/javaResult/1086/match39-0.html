class Solution {
    public boolean divisorGame(int N) {
        boolean[][] dp = new boolean[1010][2];
        dp[1][0] = false;
        dp[1][1] = true;
        for (int i = 2; i <= N; i++) {
            dp[i][1] = true;
            for (int j = 1; j < i; j++) {
                if (!dp[j][0] && i % (i - j) == 0) {
                    dp[i][1] = false;
                    break;
                }
            }
            dp[i][0] = false;
            for (int j = 1; j < i; j++) {
                if (dp[j][1] && i % (i - j) == 0) {
                    dp[i][0] = true;
                    break;
                }
            }
            // System.out.println(dp[2][0] + " " + dp[2][1]);
        }
        return dp[N][0];
    }
}
