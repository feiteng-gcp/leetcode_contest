class Solution {
    public boolean divisorGame(int N) {
    boolean[] dp = new boolean[N + 1];
    dp[1] = false;
    for (int i = 2; i < N + 1; i++) {
      for (int j = 1; j < i / 2 + 1; j++) {
        if (i % j == 0) {
          if (!dp[i - j]) {
            dp[i] = true;
            break;
          }
        }
      }
    }
    return dp[N];
  }
}
