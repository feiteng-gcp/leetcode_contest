class Solution {
    public boolean divisorGame(int N) {
        boolean[] results = new boolean[N + 1];
        results[1] = false;

        for (int i = 2; i <= N; i++) {
            int to = (int)Math.sqrt(i);
            for (int j = 1; j <= to; j++) {
                if (i % j == 0 && !results[i-j]) {
                    results[i] = true;
                    break;
                }
            }
        }
//
//        for (int i = 1; i <= N; i++) {
//            System.out.println(String.format("i=%d, r=%b", i, results[i]));
//        }
        return results[N];
    }
}
