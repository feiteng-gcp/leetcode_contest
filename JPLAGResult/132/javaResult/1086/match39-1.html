class Solution {
    public boolean divisorGame(int N) {
        boolean[] win = new boolean[1001];
        win[2] = true;
        win[3] = false;
        
        for(int i=4;i<N+1;i++) {
            
for(int j=1;j<=i/2;j++) {
                if(i%j==0 && !win[i-j]) {
                    win[i] = true;
                    break;
                }
            }
        }
        
        return win[N];
    }
}
