class Solution {
    public boolean divisorGame(int N) {
        if(N==1) return false;
        boolean[] dp = new boolean[N+1];
        dp[1] = false;
        dp[2] = true;
        for(int i=3; i<=N; i++) {
            for(int j=1; j<=i/2; j++) {
                if(!dp[i-j] && i%j == 0) {
                    dp[i] = true;
                    break;
                }
            }
        }
        
        return dp[N];
    }
}
