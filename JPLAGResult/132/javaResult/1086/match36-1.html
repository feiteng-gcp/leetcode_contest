class Solution {
    public boolean divisorGame(int N) {
        boolean[] table = new boolean[N + 1];
        table[1] = false;
        for (int i = 2; i <= N; i++) {
            
for (int j = 1; j < i; j++) {
                if (i % j == 0) {
                    if (!table[i - j]) {
                        table[i] = true;
                        break;
                    }
                }
            }
        }
        return table[N];
    }
}
