class Solution {
    public boolean divisorGame(int N) {
        boolean[] memo = new boolean[N+1];
        memo[1] = false;
        // memo[2] = true;
        // memo[3] = false;
        for (int i=2; i<=N; i++) {
            for (int j=1; j<i; j++) {
                if (i % j == 0) {
                    if (!memo[i-j]) {
                        memo[i] = true;
                        break;
                    }
                }
            }
        }
        return memo[N];
    }
}
