# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        stack=[]
        distances=[]
        directions=[]
        num=0
        dis=0
        head=None
        for i in range(len(S)):
            if S[i] != '-':
                num=num*10+ord(S[i])-ord('0')
                if i==len(S)-1 or S[i+1]=='-':
                    while distances and dis<=distances[-1]:
                        stack.pop()
                        distances.pop()
                        directions.pop()
                    node=TreeNode(num)
                    if head==None:
                        head=node
                    else:
                        if directions[-1]==0:
                            stack[-1].left=node
                            directions[-1]=1
                        else:
                            stack[-1].right=node
                            stack.pop()
                            directions.pop()
                            distances.pop()
                    stack.append(node)
                    directions.append(0)
                    distances.append(dis)                      
                    dis=0                 
            else:
                num=0
                dis+=1
                
        return head
