# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def dfs(self,v,d,n,prev_level):
        
        if self.i>=n or d[self.i]!=prev_level+1:
            return None
        
        root = TreeNode(v[self.i])
        self.i+=1
        root
.left = self.dfs(v,d,n,prev_level+1)
        root.right = self.dfs(v,d,n,prev_level+1)
        return root
        
        
        
    def recoverFromPreorder(self, S: str) -> TreeNode:
        v=[]
        d=[]
        i = 0
        n = len(S)
        while(i<n):
            j = i
            while(j<n and S[j]=='-'):
                j+=1
            d.append(j-i)
            i = j
            while j<n and S[j]!='-':
                j+=1
            v.append(int(S[i:j]))
            i = j
        #print(v,d) 
        if len(v)==0: #wont happen
            return None
        self.i = 0
        return self.dfs(v,d,len(v),-1)
