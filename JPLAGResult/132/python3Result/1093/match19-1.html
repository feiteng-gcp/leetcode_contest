# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        stack = []
        root = None
        
        i = 0
        while i < len(S):
            j = i
            while S[j] == '-': j += 1
            dep = j - i
            while j < len(S) and S[j].isdigit(): j += 1
            val = int(S[i+dep:j])
            
            tmp = TreeNode(val)
            
            if not stack:
                root = tmp
                stack.append((0, tmp))
            else:
                if stack
[-1][0] == dep - 1:
                    stack[-1][1].left = tmp
                    stack.append((dep, tmp))
                else:
                    while stack and stack[-1][0] != dep - 1:
                        stack.pop()
                    stack[-1][1].right = tmp
                    stack.append((dep, tmp))
            i = j
        return root
