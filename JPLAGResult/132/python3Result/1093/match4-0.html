# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        if "-" not in S:
            return TreeNode(int(S))
        rootVal = ""
        for i in range(len(S)):
            if S[i] == '-':
                break
            rootVal += S[i]
        root = TreeNode(int(rootVal))
        stack = [[root, 0, 0]] # node, level, how many child
        def helper(S):
            deep, val = 0, ""
            for c in S:
                if c == "-" and val == "":
                    deep += 1
                if c == "-" and val != "":
                    yield deep, int(val)
                    deep = 1
                    val = ""
                elif c != "-":
                    val += c
            yield deep, int(val)
            
        for deep, val in helper(S[i:]):
            new_node = TreeNode(val)
            while deep != stack[-1][1] + 1:
                stack.pop(-1)
            
            if stack[-1][2] == 0:
                stack[-1][0].left = new_node
                stack[-1][2] += 1
            elif stack[-1][2] == 1:
                stack[-1][0].right = new_node
                stack.pop(-1)
            stack.append([new_node, deep, 0])
        return root
