# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        if not S:
            return None
        i = 0
        n = len(S)
        while i < n:
            if S[i] == '-':
                break
            i += 1
        root = TreeNode(int(S[:i]))
        left_part = ""
        j = i + 1
        while j < n:
            if S[j] == '-':
                cnt = 0
                while j + 1 < n and S[j + 1] == '-':
                    cnt += 1
                    j += 1
                if cnt == 0:
                    break
                left_part += '-' * cnt
            else:
                left_part += S[j]
            j += 1
        right_part = ""
        j += 1
        while j < n:
            if S[j] == '-':
                cnt = 0
                while j + 1 < n and S[j + 1] == '-':
                    cnt += 1
                    j += 1
                if cnt == 0:
                    #print("should not happen", S[j], j)
                    break
                right_part += '-' * cnt
            else:
                right_part += S[j]
            j 
+= 1
        root.left = self.recoverFromPreorder(left_part)
        root.right = self.recoverFromPreorder(right_part)
        return root
