# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def _recoverFromPreorder(self, nodes: List[List[int]]) -> TreeNode:
        if len(nodes) == 0:
            return None
        pd = nodes[0][1]
        rt = TreeNode(nodes[0][0])
        i = 2
        while i < len(nodes) and nodes[i][1] != pd + 1:
            i += 1
        rt.left = self._recoverFromPreorder(nodes[1:i])
        rt.right = self._recoverFromPreorder(nodes[i:len(nodes)])
        return rt
    
    def recoverFromPreorder(self, S: str) -> TreeNode:
        d 
= 0
        v = 0
        nodes = []
        for i in range(len(S)):
            if S[i] in "0123456789":
                v = v * 10 + ord(S[i]) - ord("0")
            else:
                if v > 0:
                    nodes.append((v, d))
                    d = 0
                    v = 0
                d += 1
        if v > 0:
            nodes.append((v, d))
        return self._recoverFromPreorder(nodes)
