# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        
        def gen(S):
            ret = []
            
            i = 0
            while i < len(S):
                lv = 0
                while i < len(S) and S[i] == '-':
                    lv += 1
                    i += 1
                val = 0
                while i < len(S) and '0' <= S[i] <= '9':
                    val = val * 10 + int(S[i])
                    i += 1
                ret.append([val, lv])
            
            return ret
        
        pres = gen(S)
        st = [TreeNode(pres[0][0])]
        for val, lv in pres[1:]:
            while len(st) > lv:
                st.pop()
            node = TreeNode(val)
            if st[-1].left: st[-1].right = node
            else: st[-1].left = node
            st.append(node)
        
        return st[0]
