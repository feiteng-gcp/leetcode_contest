# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def parse(self, s):
        ret = []
        idx = 0
        while idx < len(s):
            a = idx
            while s[idx] == '-':
                idx += 1
            b = idx
            while idx < len(s) and s[idx] != '-':
                idx += 1
            ret.append((b - a, int(s[b:idx])))
        return ret
    
    def helper(self, traversal, starting_depth):
        if len(traversal) == 0:
            return None
        depth, val = traversal[0]
        if depth < starting_depth:
            return None
        ret = TreeNode(val)
        traversal.pop(0)
        ret.left = self.helper(traversal, starting_depth + 1)
        ret.right = self.helper(traversal, starting_depth + 1)
        return ret
        
    def recoverFromPreorder(self, S: str) -> TreeNode:
        traversal = self.parse(S)
        return self.helper(traversal, 0)
