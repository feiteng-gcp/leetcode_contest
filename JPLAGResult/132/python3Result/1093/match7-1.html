class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        x = S.find('-')
        if x == -1:
            return TreeNode(int(S))
        a = [[0, int(S[:x])]]
        while True:
            for i in range(x,len(S)):
                if S[i] in "0123456789":
                    y = i
                    break
            num1 = y - x
            z = S.find('-', y)
            if z == -1:
                a.append([num1, int(S[y:])])
                break
            else:
                a.append([num1, int(S[y:z])])
            x = z
        print(a)
        stack = [[TreeNode(a[0][1]), 0]]
        for i in range(1, len(a)):
            n = TreeNode(a[i][1])
            while stack[-1][1] != a
[i][0] - 1:
                stack.pop()
            if stack[-1][0].left == None:
                stack[-1][0].left = n
            else:
                stack[-1][0].right = n
            stack.append([n, a[i][0]])
        return stack[0][0]
