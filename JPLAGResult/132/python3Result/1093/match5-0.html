# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        st = []
        i = 0
        n = len(S)
        while i < n:
            l = 0
            while S[i] == '-':
                l += 1
                i += 1
            j = i + 1
            while j < n and S[j].isdigit():
                j += 1
            node = TreeNode(int(S[i:j]))
            if l == 0:
                st.append(node)
            else:
                t = st[l - 1]
                if t.left is None:
                    t.left = node
                else:
                    t.right = node
                if l >= len(st):
                    st.append(node)
                else:
                    st[l] = node
            i = j
        return st[0]
