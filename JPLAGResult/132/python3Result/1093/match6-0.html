class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        stack = []
        i,n = 0, len(S)
        root = TreeNode(0)
        stack.append((root, -1, False))
        while i < n:
            count = 0
            while i<n and S[i]=='-':
                i += 1
                count += 1
                
            num = 0
            while i<n and S[i].isdigit():
                num = 10*num + (ord(S[i])-ord('0'))
                i += 1
                
            node = TreeNode(num)
            
            while count != stack[-1][1]+1:
                stack.pop()
            
            if not stack[-1][2]:
                parent = stack.pop()
                stack.append((parent[0], parent[1], True))
                stack[-1][0].left = node
                stack.append((node, count, False))
            else:
                stack[-1][0].right = node
                stack.append((node, count, False))
            
        return root.left
