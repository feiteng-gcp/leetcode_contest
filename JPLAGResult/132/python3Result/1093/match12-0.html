# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        if not S:
            return None
        return self.helper(S, 0, 0, 0)[0]
        
    def helper(self, S, depth, dash, idx):
        if dash < depth:
            return None,idx,dash
        j = idx
        while j < len(S) and S[j].isdigit():
            j += 1
        root = TreeNode(int(S[idx:j]))
        dash = 0
        while j < len(S) and S[j] == '-':
            dash += 1
            j += 1
        root.left,idx,dash = self.helper(S, depth+1, dash, j)
        root.right,idx,dash = self.helper(S, depth+1, dash, idx)
        return root,idx,dash
