import re
from collections import deque
class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        def getToken(S: str):
            ptn = re.compile(r'(-*)(\d+)')
            for match in ptn.finditer(S):
                d = len(match.group(1))
                val = int(match.group(2))
                yield d, val
        roof = TreeNode(0)
        dq = deque()
        dq.append((roof, -1))
        for d, val in getToken(S):
            r 
= TreeNode(val)
            while d<= dq[-1][1]:
                dq.pop()
            r0 = dq[-1][0]
            if r0.left is None:
                r0.left = r
            elif r0.right is None:
                r0.right = r
            dq.append((r, d))
        return roof.left
