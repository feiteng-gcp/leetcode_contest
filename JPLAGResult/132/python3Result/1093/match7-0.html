# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
from collections import deque
class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        def fetch(s):
            if not s: return None
            curd = 0
            while s[curd] == "-": curd += 1
            n = 0
            for i, c in enumerate(s[curd:]):
                if c.isdigit():
                    n = n * 10 + int(c)
                else:
                    s = s[curd + i:]
                    break
            else:
                s=""
            cur = TreeNode(n)
            return (cur, curd, s)

        l = []
        s = S
        while s:
            cur, curd, s = fetch(s)
            if not l: l.append((cur, curd))
            else:
                while l[-1][1]+1 != curd:
                    l.pop()
                if l[-1][0].left:
                    l[-1][0].right = cur
                else:
                    l[-1][0].left = cur
                l.append((cur,curd))
        return l[0][0]
