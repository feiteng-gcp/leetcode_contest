# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        split_arr = S.split("-")
        def deal(arr):
            res = [arr[0]]
            cnt = 1
            for i in range(1,len(arr)):
                if arr[i] == "":
                    cnt += 1
                else:
                    res.append(cnt)
                    res.append(arr[i])
                    cnt = 1
            return res
        
        def dfs_build(arr):
            if len(arr) == 0:
                return None
            if len(arr) == 1:
                return TreeNode(int(arr[0]))
            root = TreeNode(int(arr[0]))
            next_depth = arr[1]
            indices = [i for i, x in enumerate(arr) if x == next_depth]
            # print(indices)
            if len(indices) > 0:
                second_indice = indices[1] if len(indices) > 1 else len(arr)
                # print(arr[indices[0] + 1:second_indice])
                root.left = dfs_build(arr[indices[0] + 1:second_indice])
            if len(indices) > 1:
                # print(arr[indices[1] + 1:])
                root.right = dfs_build(arr[indices[1] + 1:])
            
            return root
        print(deal(split_arr))
        return dfs_build(deal(split_arr))
