# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        if not S:
            return None
        dash = 0
        nodes = []
        num = ''
        for char in S:
            if char != '-':
                num += char
            else:    
                if num != '':
                    nodes.append((num, dash))
                    dash = 0
                    num = ''
                dash += 1
        nodes.append((num, dash))        
        root = TreeNode(nodes[0][0])
        d = {}
        d[nodes[0]] = root
        stack = [nodes[0]]
        i = 1
        while i < len(nodes):
            node = TreeNode(nodes[i][0])
            d[nodes[i]] = node
            if nodes[i][1] > stack[-1][-1]:
                
if d[stack[-1]].left:
                    d[stack[-1]].right = node
                else:    
                    d[stack[-1]].left = node
                stack.append(nodes[i])
                i += 1    
            else:        
                stack.pop()
        return root
