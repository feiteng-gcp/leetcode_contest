# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    # number of dashes to expect
    def recoverFromPreorder(self, S: str, expect=0) -> TreeNode:
        ans = TreeNode(0)
        dashes = 0
        ParentForDashes = {}
        i = 0
        while i < len(S):
            if S[i] == '-':
                dashes = 0
                while S
[i] == '-':
                    dashes += 1
                    i += 1
            else:
                val = 0
                while i<len(S) and S[i] != '-':
                    val = 10*val + int(S[i])
                    i += 1
                tmp = TreeNode(val)
                if dashes == 0:
                    ans = tmp
                else:
                    par = ParentForDashes[dashes]
                    if par.left == None:
                        par.left = tmp
                    else:
                        par.right = tmp
                ParentForDashes[dashes+1] = tmp
        return ans
