# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        
        stack = []
        
        i = 0
        root = None
        while i<len(S):
   
            j = i
            d = 0
            while j<len(S) and S[j]=='-':
                j += 1
            d = j-i
            
            i = j
            j = i
            num = 0
            while j<len(S) and S[j].isdigit():
                num = num*10 + int(S[j])
                j += 1
            i = j
            node = TreeNode(num)
            if not root: 
                root = node
            if not stack:
                stack.append([node,d])
            elif  d>stack[-1][1]:
                stack[-1][0].left = node
                stack
.append([node,d])
            else:
                while stack and stack[-1][1]>=d:
                    stack.pop()
                if stack:
                    stack[-1][0].right = node
                stack.append([node,d])
                
        return root
