# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        return self.helper(S, 1)
    
    def helper(self, s, d):
        if not s or s == '':
            return None
        if '-' not in s:
            return TreeNode(int(s))
        idx = s.find('-')
        root = TreeNode(s[:idx])
        
        n = len(s)
        cnt = 0
        idx2 = n + 1
        for i in range(idx + 1, n):
            if s[i] == '-':
                cnt += 1
            else:
                if cnt == d:
                    idx2 = i - d
                    break
                cnt = 0
        
        if idx2 < n + 1:
            root.left = self.helper(s[idx + d : idx2], d + 1)
            root.right = self.helper(s[idx2 + d :], d + 1)
        else:
            root.left = self.helper(s[idx + d:], d + 1)
        return root
