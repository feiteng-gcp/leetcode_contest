# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        nodes = []
        curdepth = 0
        curnum = ""
        for c in S:
            if c != "-":
                curnum += c
            elif len(curnum) == 0:
                curdepth += 1
            else:
                nodes.append((int(curnum), curdepth))
                curnum = ""
                curdepth = 1
        if curnum:
            nodes.append((int(curnum), curdepth))
        if len(nodes) == 0:
            return None
        stack = []
        for val, depth in nodes:
            node = TreeNode(val)
            while len(stack) > depth:
                stack.pop()
            if len(stack) > 0:
                parent 
= stack[-1]
                if parent.left is None:
                    parent.left = node
                else:
                    parent.right = node
            stack.append(node)
        return stack[0]
