# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        # convert string s to a [depth, val] list
        splitString = S.split('-')
        l = []
        l.append((0, int(splitString[0])))
        depth = 1
        for item in splitString[1:]:
            if item != '':
                l.append((depth, int(item)))
                depth = 1
            else:
                depth += 1
        
        # use a stack to store (depth, val)
        root = TreeNode(l[0][1])
        stack = [(0, root)]
        
        for depth, val in l[1:]:
            currentNode = TreeNode(val)
            while depth < stack[-1][0] + 1:
                stack.pop()   
            # now depth == stack[-1][0] + 1:
            parentNode = stack[-1][1]
            if not parentNode.left:
                parentNode.left = currentNode
            else:
                parentNode
.right = currentNode
                stack.pop()
            stack.append((depth, currentNode))
            
        return root
