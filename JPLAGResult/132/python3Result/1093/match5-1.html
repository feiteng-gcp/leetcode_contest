# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        index = 0
        length = len(S)
        
        layers = []
        root = None
        
        
while index < length:
            counts = 0
            while S[index] == "-":
                counts += 1
                index += 1
                
            num = 0
            while index < length and S[index].isdigit():
                num = num * 10 + int(S[index])
                index += 1
                
            node = TreeNode(num)
            
            if counts > 0:
                pnt = layers[counts - 1]
                if not pnt.left:
                    pnt.left = node
                else:
                    pnt.right = node
            else:
                root = node
            
            if len(layers) == counts:
                layers.append(node)
            else:
                layers[counts] = node
                
        return root
