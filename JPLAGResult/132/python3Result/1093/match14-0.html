# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        stack = [[] for i in range(1000)]
        stack_top = -1
        cur_depth = 0
        
        i = 0
        
        while i < len(S):
            if S[i] == '-':
                cur_depth += 1
                i += 1
            else:
                j = i
                
                while j < len(S) and S[j].isdigit():
                    j += 1
                
                cur_node = TreeNode(int(S[i:j]))
                i = j
                
                if cur_depth == 0:
                    stack_top += 1
                    stack[stack_top] = [cur_node, cur_depth]
                else:
                    while stack[stack_top][1] != cur_depth - 1:
                        stack_top -= 1
                    
                    if not stack[stack_top][0].left:
                        stack[stack_top][0].left = cur_node
                    else:
                        stack[stack_top][0].right = cur_node
                    
                    stack_top += 1
                    stack[stack_top] = [cur_node, cur_depth]
                
                cur_depth = 0
        
        return stack[0][0]
