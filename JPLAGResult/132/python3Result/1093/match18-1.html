# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        parsed_s = re.findall('[-]+|\d+', S)
        arr = []
        for i in range(1, len(parsed_s), 2):
            arr.append((int(parsed_s[i+1]), len(parsed_s[i])))
        
        root = TreeNode(int(parsed_s[0]))
        nodes = [root]
        depth = 0
        for val, d in arr:
            new_node = TreeNode(val)
            while depth + 1 > d:
                nodes.pop()
                depth -= 1
                
            
if nodes[-1].left:
                nodes[-1].right = new_node
            else:
                nodes[-1].left = new_node
            nodes.append(new_node)
            depth += 1
        return root
