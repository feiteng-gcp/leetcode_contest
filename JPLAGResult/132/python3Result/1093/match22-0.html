# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def recoverFromPreorder(self, S: str) -> TreeNode:
        if not S:
            return None

        stk = []
        isWord = True
        l = 0
        word = []
        for c in S+'-':
            if c=='-':
                if isWord:
                    node = TreeNode(int(''.join(word)))
                    while l != len(stk):
                        stk.pop()
                    if stk:
                        pnode = stk[-1]
                        if not pnode.left:
                            pnode.left = node
                        else:
                            pnode.right = node
                    stk.append(node)
                    isWord = False
                    l = 0
                    word = []
                l+=1
                continue
            else:
                isWord = True
                word.append(c)
        return stk[0]
