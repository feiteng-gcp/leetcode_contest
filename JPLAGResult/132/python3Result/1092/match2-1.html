# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        if root is None:
            return 0
        
        max_diff = 0
        def dfs(root, max_val, min_val):
            if root is None:
                nonlocal max_diff
                max_diff = max(max_val-min_val, max_diff)
                return
            max_val 
= max(root.val, max_val)
            min_val = min(root.val, min_val)
            dfs(root.left, max_val, min_val)
            dfs(root.right, max_val, min_val)
        dfs(root, -float('inf'), float('inf'))
        return max_diff
