# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        self.result = 0
        self.dfs(root, 100001, -1)
        
        return self.result
    
    def dfs(self, node, cur_min, cur_max):
        if not node.left and not node.right:
            self.result = max(self.result, abs(max(cur_max, node.val) - min(cur_min, node.val)))
            return
        
        if node.left:
            self.dfs(node.left, min(cur_min, node.val), max(cur_max, node.val))
        
        if node.right:
            self.dfs(node.right, min(cur_min, node.val), max(cur_max, node.val))
