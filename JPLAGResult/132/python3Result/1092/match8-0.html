# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        self.ans = 0
        def helper(root):
            if root.left is None and root.right is None:
                return [root.val, root.val]
            elif root.left is None:
                l,h = helper(root.right)
                
            elif root.right is None:
                l, h = helper(root.left)
            else:
                l1, h1 = helper(root.left)
                l2,h2 = helper(root.right)
                l = min(l1, l2)
                h = max(h1, h2)
            self.ans = max(self.ans, abs(root.val-l))
            self.ans = max(self.ans, abs(root.val-h))
            l = min(l, root.val)
            h = max(h, root.val)
            return [l,h]
        helper(root)
        return self.ans
