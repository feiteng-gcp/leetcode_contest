# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        self.ans = 0
        if not root:
            return self.ans
        vmax = vmin = root.val
        self.dfs(root.left, vmax, vmin)
        self.dfs(root.right, vmax, vmin)
        return self.ans
