# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def _maxAncestorDiff(self, root: TreeNode) -> List[int]: # (maxdiff, maxval, minval)
        if root is None:
            return (-1, -1, 100001)
        lmd, lpv, lnv = self._maxAncestorDiff(root.left)
        rmd, rpv, rnv = self._maxAncestorDiff(root.right)
        pv = max(lpv, rpv, root.val)
        nv = min(lnv, rnv, root.val)
        md = max(lmd, rmd)
        cpv = max(lpv, rpv)
        cnv = min(lnv, rnv)
        if cpv >= 0:
            md = max(md, abs(root.val - cpv))
        if cnv <= 100000:
            md = max(md, abs(root.val - cnv))
        return (md, pv, nv)
        
    
    def maxAncestorDiff(self, root: TreeNode) -> int:
        md, pv, nv = self._maxAncestorDiff(root)
        return md
