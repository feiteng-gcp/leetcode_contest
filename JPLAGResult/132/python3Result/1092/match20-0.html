# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        def helper(root):
            if not root.left and not root.right:
                return 0, root.val, root.val
            min_anc = max_anc = root.val
            V = 0
            if root.left:
                left_V, left_min, left_max = helper(root.left)
                V = left_V
                min_anc = left_min
                max_anc = left_max
            if root.right:
                right_V, right_min, right_max = helper(root.right)
                V = max(V, right_V)
                min_anc = min(min_anc, right_min)
                max_anc = max(max_anc, right_max)
            
            return max(V, abs(root.val - min_anc), abs(root.val - max_anc)), \
                            min(root.val, min_anc), max(root.val, max_anc)
        return helper(root)[0]
