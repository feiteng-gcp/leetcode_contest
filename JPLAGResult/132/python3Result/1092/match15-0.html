# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        def solve(node):
            maxD, minD, v = node.val, node.val, 0
            if node.left:
                maxL, minL, vl = solve(node.left)
                v = max(vl, v, node.val - minL, maxL - node.val)
                maxD = max(maxD, maxL)
                minD = min(minD, minL)
            if node.right:
                maxL, minL, vl = solve(node.right)
                v = max(vl, v, node.val - minL, maxL - node.val)
                maxD = max(maxD, maxL)
                minD = min(minD, minL)
            return maxD, minD, v
        _, _, v = solve(root)
        return v
