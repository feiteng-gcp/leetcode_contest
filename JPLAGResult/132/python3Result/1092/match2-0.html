# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        ans=[0]
        def dfs(x, mn, mx):
            if not x: return
            a=max(abs(mx-x.val), abs(mn-x.val))
            ans[0]=max(a,ans[0])
            mn=min(mn,x.val)
            mx=max(mx,x.val)
            dfs(x.left,mn,mx)
            dfs(x.right,mn,mx)
        dfs(root, root.val,root.val)
        return ans[0]
