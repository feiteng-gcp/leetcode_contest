# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        
        
        self
.ans = 0
        
        def traverse(node):
            if not node:
                return -1,-1
            
            lmin,lmax = traverse(node.left)
            rmin,rmax = traverse(node.right)
            cmin = node.val
            cmax = node.val
            for v in [lmin,lmax,rmin,rmax]:
                if v!=-1:
                    self.ans=max(self.ans, abs(node.val-v))
                    cmin=min(cmin,v)
                    cmax=max(cmax,v)
            return cmin,cmax
                        
        traverse(root)
        return self.ans
