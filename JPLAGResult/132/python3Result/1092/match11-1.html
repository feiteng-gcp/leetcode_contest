# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        if not root: return None
        self.ans=0
        def search(root,minv,maxv):
            self
.ans=max(self.ans,abs(root.val-minv))
            self.ans=max(self.ans,abs(root.val-maxv))
            newmin=min(root.val,minv)
            newmax=max(root.val,maxv)
            if root.left:
                search(root.left,newmin,newmax)
            if root.right:
                search(root.right,newmin,newmax)
        if root.left:
            search(root.left,root.val,root.val)
        if root.right:
            search(root.right,root.val,root.val)
        return self.ans
