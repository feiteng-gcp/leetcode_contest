# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def dfs(self,root,amin,amax):
        if root==None:
            return 0  # 0 for null
        if amin==None:
            amin = root.val
            amax = root.val
        res = max(abs(amin-root.val),abs(amax-root.val))
        amin = min(root.val,amin)
        amax = max(root.val,amax)
        return max(self.dfs(root.left,amin,amax),self.dfs(root.right,amin,amax),res)
        
    def maxAncestorDiff(self, root: TreeNode) -> int:
        return self.dfs(root,None,None)
