# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        self.ans=0
        def dfs(root):
            if root == None:
                return 1000000,-1
            if root.left==None and root.right == None:
                return root.val,root.val
            l_min,l_max=dfs(root.left)
            r_min,r_max=dfs(root.right)
            mi = min(l_min,r_min)
            ma = max(l_max,r_max)
            self.ans=max(self.ans,abs(root.val-mi),abs(ma-root.val))
            return min(mi,root.val),max(ma,root.val)
        dfs(root)
        return self.ans
