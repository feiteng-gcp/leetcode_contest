# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        self.ans = -float('inf')
        def helper(node):
            if node.left is None and node.right is None:
                return node.val, node.val
            minl = minr = float('inf')
            maxl = maxr = -float('inf')
            
if node.left:
                minl, maxl = helper(node.left)
            if node.right:
                minr, maxr = helper(node.right)
            _min = min(minl, minr)
            _max = max(maxl, maxr)
            self.ans = max(self.ans, abs(node.val - _min), abs(node.val - _max))
            return min(_min, node.val), max(_max, node.val)
        helper(root)
        return self.ans
