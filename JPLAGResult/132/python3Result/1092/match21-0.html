# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        if not root:
            return 0
        maxDiff = [0]
        self.helper(root, maxDiff)
        return maxDiff[0]
    
    def helper(self, root, maxDiff):
        if not root:
            return -1, -1
        leftMin, leftMax = self.helper(root.left, maxDiff)
        rightMin, rightMax = self.helper(root.right, maxDiff)
        minVal, maxVal = root.val, root.val
        if leftMin != -1:
            maxDiff[0] = max(maxDiff[0], abs(root.val - leftMin), abs(root.val - leftMax))
            minVal = min(minVal, leftMin)
            maxVal = max(maxVal, leftMax)
        if rightMin != -1:
            maxDiff[0] = max(maxDiff[0], abs(root.val - rightMin), abs(root.val - rightMax))
            minVal = min(minVal, rightMin)
            maxVal = max(maxVal, rightMax)
        return minVal, maxVal
