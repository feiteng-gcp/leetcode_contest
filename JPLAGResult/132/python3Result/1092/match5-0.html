# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        def walk(node, ans):
            if not node.left and not node.right:
                return [node.val]
            a = []
            if node.left:
                a += walk(node.left, ans)
            if node.right:
                a += walk(node.right, ans)
            for i in a:
                ans[0] = max(abs(i - node.val), ans[0])
            return [max(a + [node.val]), min(a + [node.val])]
        ans = [0]
        walk(root, ans)
        return ans[0]
