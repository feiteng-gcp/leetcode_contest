# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        if not root:
            return 0
        self.max_diff = 0
        self.dfs(root)
        return self.max_diff
    
    def dfs(self, node):
        m, M = node.val, node.val
        if node.left:
            lm, lM = self.dfs(node.left)
            m 
= min(lm, m)
            M = max(lM, M)
        if node.right:
            rm, rM = self.dfs(node.right)
            m = min(rm, m)
            M = max(rM, M)
        self.max_diff = max(abs(node.val - m), self.max_diff)
        self.max_diff = max(abs(node.val - M), self.max_diff)
        return m, M
