# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        self.ans = -1
        self.findMinMax(root)
        return self.ans
    
    def findMinMax(self, root):
        if not root.left and not root.right:
            return root.val, root.val
        lmin = rmin = sys.maxsize
        rmax = lmax = -sys.maxsize
        if root.left:
            lmin, lmax = self.findMinMax(root.left)
        if root.right:
            rmin, rmax = self.findMinMax(root.right)
        min_, max_ = min(lmin, rmin), max(lmax, rmax)
        # print(min_, max_, root.val)
        self.ans = max(abs(root.val - min_), abs(root.val - max_), self.ans)
        # print('ans:', self.ans)
        return min(min_, root.val), max(max_, root.val)
