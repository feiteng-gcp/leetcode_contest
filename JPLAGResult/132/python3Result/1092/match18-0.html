# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def helper(self, root):
        low, high, v = root.val, root.val, 0
        if root.left is not None:
            llow, lhigh, lv = self.helper(root.left)
            low = min(low, llow)
            high = max(high, lhigh)
            v = max(v, lv)
        if root.right is not None:
            rlow, rhigh, rv = self.helper(root.right)
            low = min(low, rlow)
            high = max(high, rhigh)
            v = max(v, rv)
        v = max(v, high - root.val, root.val - low)
        return low, high, v
    
    def maxAncestorDiff(self, root: TreeNode) -> int:
        l, h, v = self.helper(root)
        return v
