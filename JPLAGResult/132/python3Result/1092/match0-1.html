# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        maxDiff = 0
        def dfs(node, maxAnc, minAnc):
            nonlocal maxDiff
            
            if not node:
                return
            maxDiff = max(maxDiff, abs(maxAnc-node.val), abs
(minAnc-node.val))
            maxAnc = max(maxAnc, node.val)
            minAnc = min(minAnc, node.val)
            dfs(node.left, maxAnc, minAnc)
            dfs(node.right, maxAnc, minAnc)
        dfs(root, root.val, root.val)
        return maxDiff
