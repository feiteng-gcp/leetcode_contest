# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        if not root:
            return 0
        self.maxdiff = 0
        def dfs(node,p_min,p_max):
            if not node:
                return
            p_min = min(node.val,p_min)
            p_max = max(node.val,p_max)
            self.maxdiff = max(self.maxdiff,abs(p_min - node.val),abs
(p_max - node.val))
            dfs(node.left,p_min,p_max)
            dfs(node.right,p_min,p_max)
        dfs(root,root.val,root.val)
        
        return self.maxdiff
