class Solution:
    def __init__(self):
        self.ans = 0

    def maxAncestorDiff(self, root: TreeNode) -> int:
        def dfs(root):
            if not root:
                return None
            s1 = dfs(root.left)
            s2 = dfs(root.right)
            maxium = -1
            minium = 9999999
            if s1:
                maxium = s1[0]
                minium = s1[1]
            if s2:
                maxium = max(maxium, s2[0])
                minium = min(minium, s2[1])
            if maxium >= 0: 
                self.ans = max(self.ans, abs(root.val - maxium), abs(root.val - minium))
            if maxium == -1:
                return [root.val, root.val]
            else:
                return [max(maxium, root.val), min(root.val, minium)]

        dfs(root)
        return self.ans
