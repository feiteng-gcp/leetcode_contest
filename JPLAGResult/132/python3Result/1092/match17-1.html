class Solution:
    def maxAncestorDiff(self, root: TreeNode) -> int:
        maxDiff = 0
        def post(root: TreeNode):
            nonlocal maxDiff
            i0, j0 = root.val, root.val
            if root.left is not None:
                i1, j1 = post
(root.left)
                i0 = min(i0, i1)
                j0 = max(j0, j1)
            if root.right is not None:
                i1, j1 = post(root.right)
                i0 = min(i0, i1)
                j0 = max(j0, j1)
            maxDiff = max(maxDiff, root.val - i0, j0 - root.val)
            return i0, j0
        if root is None:
            return 0
        else:
            post(root)
        return maxDiff
