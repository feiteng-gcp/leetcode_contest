class Solution:
    def divisorGame(self, N: int) -> bool:
        n = N
        dp = [False]* (1+n)
        dp[0]=True
        dp[1]=False
        for j in range(2,n+1):
            for k in range(1,j):
                if (j%k==0):
                    if (dp[j-k]==False):
                        dp[j] = True
                        break
        return dp[n]
