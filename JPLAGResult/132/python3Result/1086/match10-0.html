class Solution:
    def divisorGame(self, N: int) -> bool:
        arr=[False]*(1001)
        arr[2]=True
        for i in range(3,N+1):
            badToOpponent=False
            for j in range(1,int(i**0.5)+1):
                if i%j==0 and arr[i-j]==False:
                    badToOpponent=True
                    break
            arr[i]=badToOpponent
        return arr[N]
