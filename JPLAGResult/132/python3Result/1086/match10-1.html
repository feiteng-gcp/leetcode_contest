class Solution:
        
    def divisorGame(self, N: int) -> bool:
        p1wins = [None] * (N + 1)
        p1wins[1] = False
        for i in range(2, N + 1):
            canwin = False
            for x in range
(1, i):
                if i % x == 0 and p1wins[i - x] == False:
                    canwin = True
                    break
            p1wins[i] = canwin
        return p1wins[N]
