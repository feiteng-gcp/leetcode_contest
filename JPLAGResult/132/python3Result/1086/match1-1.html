class Solution:
    def divisorGame(self, N: int) -> bool:
        # 10 - A5, 5 - B1, 4 - A1, 3 - B1, 2 - A1, 1, win
        if N < 2:
            return False
        a = [False] * (N + 1)
        a[2] 
= True
        
        for i in range(3, N + 1):
            for j in range(1, i):
                if i % j == 0 and a[i - j] == False:
                    a[i] = True
                    break
        return a[N]
