class Solution:
    def divisorGame(self, N: int) -> bool:
        win = [None] * (N + 1)
        win[1] = False
        for i in range(2, N + 1):
            for j in range(1, N // 2 + 1):
                if i % j == 0 and (win[i - j] == False):
                    win[i] = True
                    break
            else:
                win[i] = False
        return win[N]
