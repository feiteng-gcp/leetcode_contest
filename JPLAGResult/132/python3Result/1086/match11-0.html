class Solution:
    def divisorGame(self, N: int) -> bool:
        dp = [False] * (N + 1)
        dp[1] = False
        for i in range(2, N + 1):
            for j in range(1, int(math.sqrt(i)) + 1):
                if i % j: continue
                if not dp[i - j]:
                    dp[i] = True
                    break
            else:
                dp[i] = False
        
        return dp[N]
