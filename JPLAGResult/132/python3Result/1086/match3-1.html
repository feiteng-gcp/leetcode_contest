class Solution:
    def divisorGame(self, N: int) -> bool:
        dp 
= [False] * (N+1)
        for i in range(2, N+1):
            flag = False
            for j in range(1, i):
                if i % j == 0 and not dp[i-j]:
                    flag = True
                    break
            dp[i] = flag
        return dp[N]
