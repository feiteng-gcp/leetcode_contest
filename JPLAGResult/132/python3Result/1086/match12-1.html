class Solution:
    def divisorGame(self, N: int) -> bool:
        opt = [False] * (N+1)
        for i in range(2, N+1):
            for j in range(1,i):
                if j * j > i:
                    break
                if i % j == 0:
                    if not opt[i-j]:
                        opt[i] = True
                        break
                    
if i // j < i and not opt[i-i // j]:
                        opt[i] = True
                        break
        return opt[N]
