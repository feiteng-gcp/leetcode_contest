class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        n=len(A)
        dp={}
        best=2
        for i in range(1,n):
            tmp={}
            for j in range(i-1,-1,-1):
                nxt=A[j]*2-A[i]
                if (A[j], nxt) in dp:
                    tmp[A[i],A[j]]=dp[A[j], nxt]+1
                    best=max(tmp[A[i],A[j]], best)
                else:
                    dp[A[i],A[j]]=2
            for k, v in tmp.items():
                dp[k]=v
        return best
