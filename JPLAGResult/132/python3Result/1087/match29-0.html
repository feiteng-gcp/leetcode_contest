class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        if len(A)==2:
            return 2
        dp=[{}]
        temp={}
        temp[A[1]-A[0]]=2
        dp.append(temp)
        ans=2
        for i in range(2,len(A)):
            temp={}
            seen=set()
            for j in range(i-1,-1,-1):
                dif = A[i]-A[j]
                if dif in seen:
                    continue
                seen.add(dif)
                if dif in dp[j]:
                    temp[dif] = dp[j][dif]+1
                else:
                    temp[dif]=2
                
                ans=max(ans,temp[dif])
            dp.append(temp)
        return ans
