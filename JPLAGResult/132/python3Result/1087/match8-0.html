class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        dp = [dict() for _ in range(len(A))]
        for j in range(1, len(A)):
            for i in range(j - 1, -1, -1):
                d = A[j] - A[i]
                if not d in dp[i]:
                    dp[j][d] = dp[j].get(d, 2)
                else:
                    dp[j][d] = max(dp[j].get(d, 2), dp[i][d] + 1)
        return max([max(_dp.values()) for _dp in dp[1:]])
