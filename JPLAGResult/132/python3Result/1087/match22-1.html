class Solution:
    def longestArithSeqLength(self, a: List[int]) -> int:
        # [20, 1, 15, 3, 10, 5, 8]
        #  0   1  2   3  4   5  6
        #    -19, 1
        #       -5, 2
        #       14, 2

        n = len(a)
        if n < 3:
            return n
        
        dic, m = [{} for _ in range(n)], 0
        for i in range(1, n):
            for j in range(i):
                d = a[i] - a[j]
                dic[i][d] = dic
[j][d] + 1 if d in dic[j] else 2
                m = max(m, dic[i][d])
        return m
