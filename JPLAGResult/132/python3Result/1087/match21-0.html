class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        longest,n = 0,len(A)
        length = [collections.Counter() for _ in range(n)]
            
        for i in range(1,n):
            for j in range(i):
                diff = A[i] - A[j]
                length[i][diff] = length[j][diff] + 1
                longest = max(longest, length[i][diff])
                
        return longest+1
