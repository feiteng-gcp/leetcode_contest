class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        n = len(A)
        memo = {}
        maxLen = 2
        for r in range(1, n):
            for l in range(0, r):
                d = A[r] - A[l]
                if (l, d) in memo:
                    memo[(r, d)] = memo[(l, d)] + 1
                    maxLen = max(maxLen, memo[(r, d)])
                else:
                    memo[(r, d)] = 2
        return maxLen
