class Solution:
    def longestArithSeqLength(self, A) -> int:
        memo = {}
        res = 0
        for i in range(1,len(A)):
            for j in range(i):
                dif = A[i]-A[j]
                v = 2
                if (j,dif) in memo:
                    v = memo[(j,dif)] +1
                memo[(i,dif)] = v
                res = max(v,res)
        return res
