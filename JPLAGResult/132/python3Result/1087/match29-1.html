class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
      dp = []
      dp.append({})
      ans = 0
      for i in range(1,len(A)):
        curr = A[i]
        curr_dp = {}
        for j in range(i):
          diff = A[i] - A[j]
          if diff in dp[j]:
            curr_dp[diff] = max(dp[j][diff]+1, curr_dp.get(diff,1))
          else:
            curr_dp
[diff] = 2
          ans = max(ans, curr_dp[diff])
        dp.append(curr_dp)
      # print(dp)
      return ans
