class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        def find(memo, A, i2, i3):
            target = 2 * A[i2] - A[i3]
            if 0 <= target <= 10000:
                for i1 in memo[target][::-1]:
                    if i1 < i2:
                        return i1
            return -1
        
        
        memo = [[] for i in range(10001)]
        N = len(A)
        ans = 2
        for i in range(N):
            memo[A[i]].append(i)
        
        dp = [[2] * N for i in range(N)]
        for i2 in range(1, N - 1):
            for i3 in range(i2 + 1, N):
                i1 = find(memo, A, i2, i3)
                if i1 != -1:
                    dp[i2][i3] = dp[i1][i2] + 1
                    ans = max(ans, dp[i2][i3])
        return ans
