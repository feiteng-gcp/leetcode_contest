class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        res = 1
        n = len(A)
        dp = [{} for _ in range(n)]
        
        for i in range
(1, n):
            for j in range(i):
                diff = A[j] - A[i]
                if diff in dp[j]:
                    dp[i][diff] = dp[j][diff] + 1
                else:
                    dp[i][diff] = 2
                res = max(res, dp[i][diff])
        
        return res
