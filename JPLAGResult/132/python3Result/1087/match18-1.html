class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        n = len(A)
        opt = [{} for _ in range(n)]
        ans 
= 2
        for i in range(1, n):
            for j in range(i):
                d = A[i] - A[j]
                if d not in opt[i]:
                    opt[i][d] = 2
                if d in opt[j] and opt[j][d] + 1 > opt[i][d]:
                    opt[i][d] = opt[j][d] + 1
                    ans = max(ans, opt[i][d])
        return ans
