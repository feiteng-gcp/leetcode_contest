class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        n = len(A)
        if n <= 2:
            return n
        from collections import defaultdict
        d = defaultdict(dict)
        res = 1
        for i in range
(1, n):
            for prev in range(i):
                diff = A[i] - A[prev]
                length = 2
                if prev in d and diff in d[prev]:
                    length = d[prev][diff] + 1
                di = d[i]    
                di[diff] = length
                res = max(length, res)
        return res
