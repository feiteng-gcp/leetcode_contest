class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        
        diff = [{} for _ in range(len(A))]
        
        ans = 2
        
for i in range(len(A)):
            for j in range(i):
                d = A[i]-A[j]
                if d in diff[j]:
                    diff[i][d] = diff[j][d] + 1
                else:
                    diff[i][d] = 2
                    
                ans = max(ans,diff[i][d])
                
        return ans
