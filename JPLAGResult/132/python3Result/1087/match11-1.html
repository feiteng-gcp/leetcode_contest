class Solution:
    def longestArithSeqLength(self, A) -> int:
        n 
= len(A)
        reco = [ collections.defaultdict(int) for _ in range(n) ]
        res = 0
        for i in range(1,n):
            for j in range(i):
                reco[i][A[i]-A[j]] =  reco[j][A[i]-A[j]]+1
            res = max(res, max(reco[i].values()) + 1 )
        return res
