class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        dic = {}
        res = 1
        for i in range(len(A)):
            visited = {}
            for j in range(i):
                diff = A[i] - A[j]
                if diff not in visited:
                    dic[(A[i], diff)] = dic.get((
A[j], diff), 1) + 1
                # if dic[(A[i], diff)] == 7:
                #     print(i, A[i], diff)
                    res = max(res, dic[(A[i], diff)])
                    visited[diff] = 1
            # print(A[i], res)
        return res
