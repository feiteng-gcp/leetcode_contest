class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
#         dp[i] = 
#         dis[n]
        dp = [1 for _ in range(len(A))]
        dis = [collections.defaultdict(int) for _ in range(len(A))]
        max_length = 1
        for i in range(len(A)):
            for j in range(i):
                diff = A[j] - A[i]
                if diff not in dis[j]:
                    dis[j][diff] = 1
                    dis[i][diff] = dis[j][diff] + 1
                else:
                    dis[i][diff] = dis[j][diff] + 1
                
                max_length = max(dis[i][diff],max_length)
                dp[i] = max_length
        return max_length
