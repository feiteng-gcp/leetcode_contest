class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        n = len(A)
        d = [collections.defaultdict(lambda : 1) for i in range(n)]
        ans = 2
        for i in range(1, n):
            for j in range(i):
                df = A[i] - A[j]
                d[i][df] = max(d[i][df], d[j][df] + 1)
                ans = max(ans, d[i][df])
        return ans
