class Solution:
    def longestArithSeqLength(self, A: List[int]) -> int:
        if len(A) <= 2:
            return len(A)
        max_l = 0
        mapping = [{} for _ in range(len(A))]
        for i in range(len
(A)):
            for j in range(i):
                d = A[j] - A[i]
                if d in mapping[j]:
                    mapping[i][d] = mapping[j][d] + 1
                else:
                    mapping[i][d] = 2
                max_l = max(mapping[i][d], max_l)
        return max_l
