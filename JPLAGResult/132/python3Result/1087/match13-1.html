import collections
class Solution:
    def longestArithSeqLength(self,A):
        s=set(A);l=len(A);ans=2
        dic=collections.defaultdict()
        dic[(1,A[1]-A[0])]=2
        for i in range(2,l):
            for j in range(0,i):
                if (j,A[i]-A[j]) in dic:
                    dic[(i,A[i]-A[j])]=dic
[(j,A[i]-A[j])]+1
                    ans=max(ans,dic[(i,A[i]-A[j])])
                else:
                    dic[(i,A[i]-A[j])]=2
        return ans
