class Solution {

    bool divisorGame(int N, vector<int>& dp) {
        if (dp[N] != -1) return dp[N];
        for (int i = 1; i < N; i++) {
            if (N % i == 0) {
                if (divisorGame(N-i, dp) == 0) {
                    dp[N] = 1;
                    return true;
                }
            }
        }
        dp[N] = 0;
        return false;
    }
public:
    bool divisorGame(int N) {
        if (N == 1) return false;
        if (N == 2) return true;
        vector<int> dp(N+1, -1);
        dp[2] = 1;
        dp[1] = 0;
        return divisorGame(N, dp);
    }
};
