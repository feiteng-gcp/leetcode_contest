class Solution {
public:
    bool divisorGame(int N) {
        vector<bool> dp(N + 1, false);
        dp[1] = false;
        dp[2] = true;
        
        for (int i = 3; i <= N; i++) {
            for (int div = 1; div < i; div++) {
                if ((i % div) == 0) {
                    if (dp[i - div] == false) {
                        dp[i] = true;
                    }
                }
            }
        }
        return dp[N];
    }
};
