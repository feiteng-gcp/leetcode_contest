class Solution {
public:
    bool divisorGame(int N) {
        vector<int> dp(N+10, 0);
        dp[1] = 0;
        dp[2] = 1;
        dp[3] = 0;
        
        for(int i = 4; i <= N; i ++)
        {
            for(int j = 1; j < i; j ++ )
            {
                if(i%j == 0)
                {
                    if(dp[i-j] == 0)
                    {
                        dp[i] = 1;
                        break;
                    }
                }
            }
        }
        
        return dp[N] == 1;
    }
};
