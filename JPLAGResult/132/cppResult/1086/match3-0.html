class Solution {
public:
    bool divisorGame(int N) {
        if (N == 1) return false;
        vector<bool> res(N + 1, false);
        for(int i = 2; i <= N; ++i) {
            for (int j = 1; j < i; ++j) {
                if (i % j == 0 && !res[i - j]) {
                    res[i] = true;
                    break;
                }
            }
        }
        return res[N];
    }
};
