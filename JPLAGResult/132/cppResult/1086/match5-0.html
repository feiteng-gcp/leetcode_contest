class Solution {
public:
    bool divisorGame(int N) {
        vector<bool> canwin(N+1, false);
        canwin[1] = false;
        canwin[2] = true;
        for (int i=3; i<=N; ++i) {
            for(int j=1; j<i; ++j) {
                if (i%j==0 && canwin[i-j]==false) {
                    canwin[i] = true;
                    break;
                }
            }
        }
        return canwin[N];
    }
};
