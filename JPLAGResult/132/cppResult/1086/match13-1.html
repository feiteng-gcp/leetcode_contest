class Solution {
    vector<int> have;

    bool dfs(int n){
        if(have[n] != -1){
            return have[n];
        }
        int ans = 0;
        for(int i=1;i<n;i++){
            if(n%i==0 && dfs(n-i) == 0){
                ans = 1;
                break;
            }
        }
        return have[n] = ans;
    }

public:
    bool divisorGame(int N) {
        have.resize(N+1, -1);
        have[1] = 0;
        return dfs(N);
    }
};
