class Solution {
public:
    bool divisorGame(int N) {
        vector<int> dp(N+1, false);
        

        for (int i = 1; i <= N; ++i) {
            for (int d = 1; d < i; ++d) {
                if (i % d == 0) {
                    if (dp[i-d] == false) {
                        dp[i] = true;
                        break;
                    }
                }
            }
        }
        return dp[N];
    }
};
