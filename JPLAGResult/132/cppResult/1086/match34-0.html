class Solution {
public:
    unordered_map<int, bool> mp;
    bool divisorGame(int N) {
        if(mp.count(N)) return mp[N];
        for(int i = 1; i < N; ++i){
            if(N%i == 0){
                if(!divisorGame(N - i)){
                    mp[N] = true;
                    return true;
                }
            }
        }
        mp[N] = false;
        return false;
    }
};
