class Solution {
public:
    bool divisorGame(int N) {
        static bool ans[1100];

        ans[1] = false;
        for (int i = 2; i <= N; i += 1) {
            ans[i] = false;
            for (int j = 1; j < i; j += 1) {
                if (i % j == 0) {
                    if (ans[i - j] == false) {
                        ans[i] = true;
                        break;
                    }
                }
            }
        }
        return ans[N];
    }
};
