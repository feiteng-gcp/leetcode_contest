class Solution {
public:
    bool v[1001];
    bool divisorGame(int n) {
        v[1] = false;
        for (int i = 2; i <= 1000; i++) {
            v[i] = false;
            for (int j = 1; j < i; j++)
                if (i % j == 0 && !v[i - j])
                    v[i] = true;
        }
        return v[n];
    }
};
