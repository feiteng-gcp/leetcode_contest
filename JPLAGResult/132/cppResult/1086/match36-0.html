class Solution {
public:
    bool divisorGame(int N) {
        //N = 9;
        if (N == 1)
        {
            return false;
        }
        
        bool *res = new bool[N+1];
        res[1] = false;
        
        for (int i=2; i<=N; i++)
        {
            res[i] = false;
            for (int x=1; x<i; x++)
            {
                if (i % x == 0)
                {
                    
                    if (!res[i - x])
                    {
                        res[i] = true;
                        break;
                    }
                }
            }
            //cout << i << " " << res[i] << endl;
        }
        return res[N];
    }
};
