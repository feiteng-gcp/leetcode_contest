class Solution {
public:
    bool divisorGame(int N) {
        vector<int> dp(N);
        dp[0] = 0;
        for (int i = 1; i < N; ++i) {
            for (int j = 0; (j + 1) * 2 <= i + 1; ++j) {
                if ((i + 1) % (j + 1) == 0 && dp[i - j - 1] == 0) {
                    dp[i] = 1;
                    break;
                }
            }
        }
        return dp.back() == 1;
    }
};
