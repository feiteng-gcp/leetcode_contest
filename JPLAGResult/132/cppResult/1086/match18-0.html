class Solution {
public:
    bool divisorGame(int N) {
       // N=4;
        vector<bool> fw(N + 1, false), sw(N + 1, false);
        fw[1] = false;
        sw[1] = true;
        
        for (int i = 2; i <= N; ++i) {
            for (int j = 1; j < i; ++j) {
                if (N % j == 0) {
                    fw[i] = fw[i] || sw[i - j];
                    sw[i] = sw[i] || fw[i - j];
                }
            //    printf("%d %d %d\n", i fw[])
            }
        }
            
        return fw[N];
    }
};
