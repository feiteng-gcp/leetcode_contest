class Solution {
public:
    bool divisorGame(int N) {
        vector<bool> canWin(N + 1, false);

        for (int i = 2; i <= N; i++) {
            for (int x = 1; x < i; x++) {
                if (i % (i - x) == 0 && !canWin[x]) {
                    canWin[i] = true;
                    break;
                }
            }
        }
        return canWin.back();
    }
};
