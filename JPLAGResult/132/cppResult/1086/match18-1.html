class Solution {
public:
    bool divisorGame(int n) {
        vector<bool> v(1001, false);
        v[0] = true;
        v[2] = true;
        
        for (int i = 4; i <= n; i++) {
            for (int j = 1; j * j <= i; j++) {
                if (i % j == 0) {
                    v[i] = v[i] || (!v[i - j]);
                    v[i] = v[i] || (!v[i - (i / j)]);
                }
            }
        }
        return v[n];
    }
};
