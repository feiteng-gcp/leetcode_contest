/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int dfs(TreeNode* root,int par_max,int par_min){
        if(root==NULL) return 0;
        int ans = 0;
        ans = max(abs(root->val-par_max),(root->val-par_min));
        ans = max(ans,dfs(root->left,max(par_max,root->val),min(par_min,root->val)));
        ans = max(ans,dfs(root->right,max(par_max,root->val),min(par_min,root->val)));
        return ans;
    }
    int maxAncestorDiff(TreeNode* root) {
        return dfs(root,root->val,root->val);
    }
};
