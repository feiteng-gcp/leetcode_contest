/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        return help(root)[0];
    }
    
    vector<int> help(TreeNode* root) {
        if (root == nullptr) {
            return vector<int>{0, -1, -1};
        }
        
        vector<int> cur{0, -1, -1};
        cur[1] = root->val;
        cur[2] = root->val;

        if (root->left) {
            auto l = help(root->left);
            cur[0] = max(cur[0], l[0]);
            cur[0] = max(cur[0], abs(root->val - l[1]));
            cur[0] = max(cur[0], abs(root->val - l[2]));
            cur[1] = min(cur[1], l[1]);
            cur[2] = max(cur[2], l[2]);
        }
        
        if (root->right) {
            auto r = help(root->right);
            cur[0] = max(cur[0], r[0]);
            cur[0] = max(cur[0], abs(root->val - r[1]));
            cur[0] = max(cur[0], abs(root->val - r[2]));
            cur[1] = min(cur[1], r[1]);
            cur[2] = max(cur[2], r[2]);
        }
        return cur;
    }
    
};
