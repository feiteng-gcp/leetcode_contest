/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

tuple<int,int, int> dfs(TreeNode* u){
    tuple<int,int, int> ret(u->val, u->val, 0);
    if(u->left){
        auto t = dfs(u->left);
        get<0>(ret) = min(get<0>(ret), get<0>(t));
        get<1>(ret) = max(get<1>(ret), get<1>(t));
        get<2>(ret) = max(get<2>(ret), get<2>(t));
        
        get<2>(ret) = max(get<2>(ret) , abs(u->val - get<1>(t)));
        get<2>(ret) = max(get<2>(ret) , abs(u->val - get<0>(t)));
    }
    if(u->right){
        auto t = dfs(u->right);
        get<0>(ret) = min(get<0>(ret), get<0>(t));
        get<1>(ret) = max(get<1>(ret), get<1>(t));
        get<2>(ret) = max(get<2>(ret), get<2>(t));
        
        get<2>(ret) = max(get<2>(ret) , abs(u->val - get<1>(t)));
        get<2>(ret) = max(get<2>(ret) , abs(u->val - get<0>(t)));
    }
    
    //cout<<get<1>(ret)<<' '<<get<0>(ret)<<endl;
    return ret;
}

class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        if(root==nullptr)return 0;
        
        return get<2>(dfs(root));
    }
};
