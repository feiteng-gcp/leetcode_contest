/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* root,int& minm,int& maxm)
    {
        if(root==NULL)return 0;
        int minm1=999999,minm2=999999,maxm1=-999999,maxm2=-999999;
        int ans1=dfs(root->left,minm1,maxm1),ans2=dfs(root->right,minm2,maxm2);
        int ans=max(ans1,ans2);
        if(minm1!=999999)
        {
            minm=min(minm,minm1);
            maxm=max(maxm,maxm1);
        }
        if(minm2!=999999)
        {
            minm=min(minm,minm2);
            maxm=max(maxm,maxm2);
        }
        ans=max(ans,max(root->val-minm,maxm-root->val));
        minm=min(minm,root->val);
        maxm=max(maxm,root->val);
        return ans;
    }
    int maxAncestorDiff(TreeNode* root) {
        int maxm=-999999,minm=999999;
        return dfs(root,minm,maxm);
    }
};
