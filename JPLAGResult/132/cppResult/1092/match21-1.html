/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
const int INF = 1e9;

class Solution {
public:
    
    
    int go(TreeNode* r, int mn = INF, int mx = -INF) {
        if (!r) return 0;
        int best = 0;
        if (mn != INF) {
            best = max(best, abs(mn - r->val));
        }
        if (mx != -INF) {
            best = max(best, abs(mx - r->val));
        }
        best = max(best,go(r->left, min(mn,r->val), max(mx,r->val)));
        best = max(best,go(r->right, min(mn,r->val), max(mx,r->val)));
        return best;
    }
    
    int maxAncestorDiff(TreeNode* root) {
        return go(root);
    }
};
