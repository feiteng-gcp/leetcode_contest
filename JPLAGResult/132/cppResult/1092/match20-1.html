/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution 
{
public:
	int maxAncestorDiff(TreeNode* root) 
	{
		return dfs(root, root->val, root->val);
	}
	int dfs(TreeNode* root, int max_v, int min_v)
	{
		int ret;
		if(root->val > max_v)
			max_v = root->val;
		if(root->val < min_v)
			min_v = root->val;
		ret = abs(max_v - min_v);
		if(root->left)
			ret = max(ret, dfs(root->left, max_v, min_v));
		if(root->right)
			ret = max(ret, dfs(root->right, max_v, min_v));
		return ret;
	}
};
