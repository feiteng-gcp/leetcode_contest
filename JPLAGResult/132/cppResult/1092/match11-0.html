/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
	int ans=0;
	int mxx(TreeNode* root)
	{
		int tmp=root->val;
		if(root->left!=NULL)
			tmp=max(tmp,mxx(root->left));
		if(root->right!=NULL)
			tmp=max(tmp,mxx(root->right));
		ans=max(ans,tmp-root->val);
		return tmp;
	}
	int mii(TreeNode* root)
	{
		int tmp=root->val;
		if(root->left!=NULL)
			tmp=min(tmp,mii(root->left));
		if(root->right!=NULL)
			tmp=min(tmp,mii(root->right));
		ans=max(ans,root->val-tmp);
		return tmp;
	}
    int maxAncestorDiff(TreeNode* root) {
    	mxx(root);
    	mii(root);
        return ans;
    }
};
