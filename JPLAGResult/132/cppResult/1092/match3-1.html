/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int sol = 0;
    void dfs(TreeNode* r, int b, int s) {
        if (!r) return;
        sol = max(sol, abs(r->val - b));
        sol = max(sol, abs(r->val - s));
        b = max(b, r->val);
        s = min(s, r->val);
        dfs(r->left, b, s);
        dfs(r->right, b, s);
    }
    int maxAncestorDiff(TreeNode* root) {
        if (!root) return 0;
        
        dfs(root, root->val, root->val);
        return sol;
    }
};
