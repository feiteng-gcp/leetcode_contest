/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int f(TreeNode * root, int maxv, int minv) {
        if (root == nullptr)
            return 0;
        int res = 0;
        res = max(abs(maxv - root -> val), abs(minv - root -> val));
        maxv = max(maxv, root -> val);
        minv = min(minv, root -> val);
        res = max(res, f(root -> left, maxv, minv));
        res = max(res, f(root -> right, maxv, minv));
        return res;
    }
    int maxAncestorDiff(TreeNode* root) {
        return max(f(root -> left, root -> val, root -> val), f(root -> right, root -> val, root -> val));
    }
};
