/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    const int INF = 1e9+7;

    int ans = 0;
    int dfs(TreeNode* root){
        int minn = INF;
        if(root->left!=NULL)minn = min(minn,dfs(root->left));
        if(root->right!=NULL)minn = min(minn,dfs(root->right));

        if(minn!=INF)ans = max(ans,abs(root->val - minn));
        return min(root->val,minn);
    }
    int dfs1(TreeNode* root){
        int maxx = 0;
        if(root->left!=NULL)maxx = max(maxx,dfs1(root->left));
        if(root->right!=NULL)maxx = max(maxx,dfs1(root->right));
        if(maxx!=0)ans = max(ans,abs(root->val - maxx));
        return max(root->val,maxx);
    }
    int maxAncestorDiff(TreeNode* root) {
        dfs(root);
        dfs1(root);
        return ans;
    }
};
