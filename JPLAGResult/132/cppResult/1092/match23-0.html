/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int result;
    int maxAncestorDiff(TreeNode* root) {
        result = 0;
        helper(root->left, root->val, root->val);
        helper(root->right, root->val, root->val);
        return result;
    }
    void helper(TreeNode* root, int min_so_far, int max_so_far) {
        if (root == nullptr) {
            return;
        } else {
            result = max(abs(root->val - min_so_far), result);
            result = max(abs(root->val - max_so_far), result);
            min_so_far = min(min_so_far, root->val);
            max_so_far = max(max_so_far, root->val);
            helper(root->left, min_so_far, max_so_far);
            helper(root->right, min_so_far, max_so_far);
        }
    }
};
