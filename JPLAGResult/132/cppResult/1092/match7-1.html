/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int ret = 0;
    
    void helper(TreeNode *root, int vmin, int vmax) {
        // corner
        if (root == NULL)
            return;
        //
        ret = max(abs(vmin - root->val), ret);
        ret = max(abs(vmax - root->val), ret);
        vmin = min(vmin, root->val);
        vmax = max(vmax, root->val);
        helper(root->left, vmin, vmax);
        helper(root->right, vmin, vmax);
        return;
    }
    
    int maxAncestorDiff(TreeNode* root) {
        helper(root, root->val, root->val);
        return ret;
    }
};
