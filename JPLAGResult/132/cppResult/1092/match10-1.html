/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        stack<int> maxSt, minSt;
        
        int res = 0;
        
        dfs(res, root, maxSt, minSt);
        
        return res;
    }
    
    void dfs(int &res, TreeNode *root, stack<int> &maxSt, stack<int> &minSt) {
        if(!root)   return ;
        
        if(maxSt.empty() || root->val >= maxSt.top()) {
            maxSt.push(root->val);
        }
        if(minSt.empty() || root->val <= minSt.top()) {
            minSt.push(root->val);
        }
        
        if(!maxSt.empty())
            res = max(res, abs(maxSt.top() - root->val));
        if(!minSt.empty()) 
            res = max(res, abs(minSt.top() - root->val));
        
        dfs(res, root->left, maxSt, minSt);
        dfs(res, root->right, maxSt, minSt);
        
        if(!maxSt.empty() && maxSt.top() == root->val) {
            maxSt.pop();
        }
        if(!minSt.empty() && minSt.top() == root->val) {
            minSt.pop();
        }
    }
};
