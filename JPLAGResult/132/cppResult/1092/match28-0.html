/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    int ret;
    pair<int,int> dfs(TreeNode* root) {
        pair<int,int> cur = make_pair(root->val,root->val);
        if(root->left!=NULL) {
            pair<int,int> l = dfs(root->left);
            cur.first = min(cur.first, l.first);
            cur.second = max(cur.second, l.second);
        }
        if(root->right!=NULL) {
            pair<int,int> r = dfs(root->right);
            cur.first = min(cur.first, r.first);
            cur.second = max(cur.second, r.second);
        }
        ret = max(ret, abs(root->val-cur.first));
        ret = max(ret, abs(root->val-cur.second));
        return cur;
    }
public:
    int maxAncestorDiff(TreeNode* root) {
        ret = 0;
        dfs(root);
        return ret;
    }
};
