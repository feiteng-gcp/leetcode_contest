/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        return helper(root, root->val, root->val);
    }
    
    int helper(TreeNode* n, int Min, int Max)
    {
        int res = max(abs(Min - n->val), abs(Max - n->val));
        Min = min(Min, n->val);
        Max = max(Max, n->val);
        if (n->left) res = max(res, helper(n->left, Min, Max));
        if (n->right) res = max(res, helper(n->right, Min, Max));
        return res;
    }
};
