/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root, int low=99999999, int hi=-99999999) {
        if (root == 0) return -9999999;
        int z = -9999999;
        if (low != 99999999) z = max(root->val - low, hi - root->val);
        z = max(
                z,
                max(
                    maxAncestorDiff(root->left, min(root->val, low), max(root->val, hi)),
                    maxAncestorDiff(root->right, min(root->val, low), max(root->val, hi))
                )
            );
        return z;
    }
};
