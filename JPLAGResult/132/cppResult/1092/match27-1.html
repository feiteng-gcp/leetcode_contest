/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

pair<int, pair<int, int>> dfs(TreeNode* root) {

    pair<int, pair<int, int>> re = {0, {root->val, root->val}};
    if (root->left) {
        auto ret = dfs(root->left);
        re.second.first = min(re.second.first, ret.second.first);
        re.second.second = max(re.second.second, ret.second.second);
        re.first = max(re.first, ret.first);
    }
    if (root->right) {
        auto ret = dfs(root->right);
        re.second.first = min(re.second.first, ret.second.first);
        re.second.second = max(re.second.second, ret.second.second);
        re.first = max(re.first, ret.first);
    }
    re.first = max(re.first, root->val - re.second.first);
    re.first = max(re.first, re.second.second - root->val);
    return re;
}

class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        return dfs(root).first; 
    }
};
