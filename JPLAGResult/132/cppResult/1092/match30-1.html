/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {

        int maxdiff = 0, maxa = root->val, mina = maxa;
        findmaxd(root, maxdiff, maxa, mina);
        return maxdiff;
    }
    void findmaxd(TreeNode* root, int & maxdiff, int maxa, int mina){
        if (!root->left && !root->right) return;
        maxa = max(maxa, root->val), mina = min(mina, root->val);
        int diff = 0;
        if (root->left){
            diff = max(diff, abs(maxa - root->left->val));
            diff = max(diff, abs(mina - root->left->val));
            findmaxd(root->left, maxdiff, maxa, mina);
        }
        if (root->right){
            diff = max(diff, abs(maxa - root->right->val));
            diff = max(diff, abs(mina - root->right->val));
            findmaxd(root->right, maxdiff, maxa, mina);
            
        }
        maxdiff = max(maxdiff, diff);
    }
};
