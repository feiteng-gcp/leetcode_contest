/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void dfs(TreeNode* root, int mx, int mn,int &ans){
        if( root == NULL ) return;
        ans = max(ans, max(mx-root->val, root->val-mn));
        dfs(root->left, max(mx, root->val), min(mn, root->val), ans);
        dfs(root->right, max(mx, root->val), min(mn, root->val), ans);
    }
    
    int maxAncestorDiff(TreeNode* root) {
        int ans = 0;
        dfs(root, -1e9, 1e9, ans);
        return ans;
        
    }
};
