/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        return solve(root).at(2);
    }
private:
    vector<int> solve(const TreeNode *node) const {
        vector<int> results = {node->val, node->val, INT_MIN};
        if (nullptr != node->left) {
            auto sub = solve(node->left);
            results[0] = min(results[0], sub[0]);
            results[1] = max(results[1], sub[1]);
            results[2] = max(sub[2], max(node->val - results[0], results[1] - node->val));
        }
        if (nullptr != node->right) {
            auto sub = solve(node->right);
            results[0] = min(results[0], sub[0]);
            results[1] = max(results[1], sub[1]);
            results[2] = max(max(results[2], sub[2]), max(node->val - results[0], results[1] - node->val));
        }
        return results;
    }
};
