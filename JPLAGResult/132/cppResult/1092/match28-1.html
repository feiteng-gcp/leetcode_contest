/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans;
    void update(int &a, int b) {
        if (a < b) a = b;
    }
    
    pair<int, int> dfs(TreeNode *ptr) {
        int val = ptr -> val;
        int xiao = val;

        int da = val;
        if (ptr -> left != NULL) {
            pair<int, int> pii = dfs(ptr -> left);
            update(ans, abs(pii.second - val));
            update(ans, abs(val - pii.first));
            xiao = min(xiao, pii.first);
            da = max(da, pii.second);
        }
        if (ptr -> right != NULL) {
            pair<int, int> pii = dfs(ptr -> right);
            update(ans, abs(pii.second - val));
            update(ans, abs(val - pii.first));
            xiao = min(xiao, pii.first);
            da = max(da, pii.second);
        }
        return make_pair(xiao, da);
    }
    
    int maxAncestorDiff(TreeNode* root) {
        ans = 0;
        dfs(root);
        return ans;
    }
};
