/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        int lo = root->val, hi = root->val, ans = 0;
        dfs(root, lo, hi, ans);
        return ans;
    }
    
    void dfs(TreeNode *cur, int lo, int hi, int &ans) {
        if (!cur)
            return;
        
        int v = cur->val;
        ans = max(ans, max(abs(v - lo), abs(v - hi)));
        
        dfs(cur->left, min(lo, v), max(hi, v), ans);
        dfs(cur->right, min(lo, v), max(hi, v), ans);
    }
};
