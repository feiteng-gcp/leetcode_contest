/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
	int maxAncestorDiff(TreeNode* root) {
		set<int> pool;
		int res = 0;
		helper(root, pool, res);
		return res;
	}

	void helper(TreeNode* node, set<int> &pool, int &res) {
		if (!node) return;
		if (!pool.empty()) {
			int diff1 = abs(node->val - *pool.begin());
			int diff2 = abs(node->val - *pool.rbegin());
			res = max(res, max(diff1, diff2));
		}
		pool.insert(node->val);
		helper(node->left, pool, res);
		helper(node->right, pool, res);
		auto it = pool.find(node->val);
		pool.erase(it);
	}
};
