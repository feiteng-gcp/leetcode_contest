/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        if(root == nullptr) return 0;
        int res = 0;
        dfs(root->left, root->val, root->val, res);
        dfs(root->right, root->val, root->val, res);
        return res;
    }
    
    void dfs(TreeNode* root, int maxVal, int minVal, int& res){
        if(root == nullptr) return;
        res = max(res, abs(root->val - maxVal));
        res = max(res, abs(root->val - minVal));
        dfs(root->left, max(maxVal, root->val), min(root->val, minVal), res);
        dfs(root->right, max(maxVal, root->val), min(root->val, minVal), res);
        return;
    }
};
