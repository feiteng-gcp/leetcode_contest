/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
 public:
  int maxAncestorDiff(TreeNode* root) {
    dfs(root);
    return ans;
  }

 private:
  int ans = 0;

  pair<int, int> dfs(TreeNode* node) {
    int mn = node->val, mx = node->val;

    if (node->left) {
      auto p = dfs(node->left);
      mn = min(mn, p.first);
      mx = max(mx, p.second);
      ans = max(ans, max(abs(node->val - p.first), abs(node->val - p.second)));
    }
    if (node->right) {
      auto p = dfs(node->right);
      mn = min(mn, p.first);
      mx = max(mx, p.second);
      ans = max(ans, max(abs(node->val - p.first), abs(node->val - p.second)));
    }
    return {mn, mx};
  }
};
