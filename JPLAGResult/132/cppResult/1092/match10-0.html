/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        multiset<int> S;
        int rtn = 0;
        help(root, S, rtn);
        return rtn;
    }
    
    void help(TreeNode *node, multiset<int> &S, int &rtn) {
        if (!node) return;
        
        S.insert(node->val);
        int v = abs(*S.rbegin() - *S.begin());
        //cout << *S.rbegin() << " " <<  *S.begin() << " " << v << endl;
        rtn = max(rtn, v);
        help(node->left, S, rtn);
        help(node->right, S, rtn);
        
        S.erase(S.find(node->val));
    }
};
