/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    void solve(TreeNode* node, int maxAncestor, int minAncestor, int& ans) {
        if (node == NULL) return;
        ans = max(ans, max(abs(maxAncestor - node->val), abs(minAncestor - node->val)));
        maxAncestor = max(maxAncestor, node->val);
        minAncestor = min(minAncestor, node->val);
        solve(node->left, maxAncestor, minAncestor, ans);
        solve(node->right, maxAncestor, minAncestor, ans);
    }
    
    int maxAncestorDiff(TreeNode* root) {
        int ans = INT_MIN;
        solve(root, root->val, root->val, ans);
        return ans;
    }
};
