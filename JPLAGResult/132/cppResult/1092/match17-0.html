/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxAncestorDiff(TreeNode* root) {
        int ret = 0;
        solve(root, -1, 100005, ret);
        return ret;
    }
    
    void solve(TreeNode* root, int maxv, int minv, int& res) {
        if (root == nullptr) return;
        
        if (maxv >= 0) {
            res = max(res, abs(root->val - minv));
            res = max(res, abs(root->val - maxv));
        }
        
        maxv = max(root->val, maxv);
        minv = min(root->val, minv);
        
        solve(root->left, maxv, minv, res);
        solve(root->right, maxv, minv, res);
    }
};
