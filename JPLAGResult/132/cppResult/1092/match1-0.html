/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void maxAncestorDiff(TreeNode* cur, vector<int>& tmp) {
        if (cur == nullptr) return;
        for (auto t : tmp) ans = max(ans, abs(cur -> val - t));
        tmp.push_back(cur -> val);
        maxAncestorDiff(cur -> left, tmp);
        maxAncestorDiff(cur -> right, tmp);
        tmp.pop_back();
    }
    int maxAncestorDiff(TreeNode* root) {
        vector<int> tmp;
        tmp.push_back(root -> val);
        maxAncestorDiff(root -> left, tmp);
        maxAncestorDiff(root -> right, tmp);
        return ans;
    }
private:
    int ans = -1;
};
