template <class T> bool chmax(T &a, const T &b) { if (a < b) { a = b; return true; }return false; }


class Solution {
public:
	int maxAncestorDiff(TreeNode* root) {
		int ans = 0;
		dfs(root, 1e6, -1e6, ans);
		return ans;
	}
	void dfs(TreeNode* root, int m, int M, int &ans) {
		chmax(ans, M - m);
		if (root == NULL)return;
		dfs(root->left, min(m, root->val), max(M, root->val), ans);
		dfs(root->right, min(m, root->val), max(M, root->val), ans);
	}
};
