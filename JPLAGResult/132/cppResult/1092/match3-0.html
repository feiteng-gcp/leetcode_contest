/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans = 0;
    void dfs(TreeNode *u, int pre, int pre2)
    {
        if (!u) return ;
        ans = max(ans, abs(pre - u->val));
        ans = max(ans, abs(pre2 - u->val));
        pre = max(pre, u->val);
        pre2 = min(pre2, u->val);
        dfs(u->left, pre, pre2);
        dfs(u->right, pre, pre2);
    }
    int maxAncestorDiff(TreeNode* root) {
        if (!root) return 0;
        ans = 0;
        dfs(root, root->val, root->val);
        return ans;
    }
};
