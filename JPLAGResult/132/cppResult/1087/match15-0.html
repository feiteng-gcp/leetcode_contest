class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        int ans = 0;
        map<int,int> dp[2200];
        for(int i=1;i<A.size();i++){
            for(int j=0;j<i;j++){
                int diff = A[i]-A[j];
                dp[i][diff] = max(dp[i][diff], dp[j][diff]+1);
                ans = max(ans, dp[i][diff]);
            }
        }
        return ans+1;
        
        
    }
};
