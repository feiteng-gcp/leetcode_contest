class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        int n = A.size();
        vector<unordered_map<int, int>> dp(n);
        int m = 0;
        
        for (int i = n -1; i >= 0; --i) {
            for (int j = n - 1; j > i; --j) {
                int diff = A[j] - A[i];
                dp[i][diff] = max(dp[j][diff] + 1, dp[i][diff]);
                m = max(dp[i][diff], m);
            }
        }
        
        return m + 1;
    }
};
