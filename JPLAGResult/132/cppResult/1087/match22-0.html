class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        int result = 0;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = i + 1; j < A.size(); ++j) {
                int target = A[j] + A[j] - A[i];
                int diff = A[j] - A[i];
                int temp = 2;
                for (int k = j + 1; k < A.size(); ++k) {
                    if (A[k] == target) {
                        temp++;
                        target += diff;
                    }
                }
                result = max(result, temp);
            }
        }
        return result;
    }
};
