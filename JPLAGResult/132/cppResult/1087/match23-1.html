class Solution {
public:
    int longestArithSeqLength(vector<int>& A)
    {
        int n = A.size();
        vector<unordered_map<int, int>> dp(n, unordered_map<int, int>());
        int ret = 0;
        for (int i = 1; i < n; ++i)
        {
            for (int j = 0; j < i; ++j)
            {

                if (dp[j].count(A[i] - A[j]) != 0)
                {
                    dp[i][A[i] - A[j]] = max(dp[i][A[i] - A[j]], dp[j][A[i] - A[j]] + 1);
                }
                else
                {
                    dp[i][A[i] - A[j]] = 2;
                }
                ret = max(ret, dp[i][A[i] - A[j]]);
            }
        }
        return ret;
    }
};
