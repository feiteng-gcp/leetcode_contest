class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        unordered_map<int, int> m[2001];
        int sol = 0;

        for (int i = 0; i < A.size(); ++i) {
            for (int j = i + 1; j < A.size(); ++j) {
                int dif = A[i] - A[j];
                
                if (m[i].find(dif) != m[i].end()) {
                    m[j][dif] = m[i][dif] + 1;
                } else {
                    m[j][dif] = 1;
                }
                sol = max(sol, m[j][dif]);
            }
        }
        return sol + 1;
        
        
    }
};
