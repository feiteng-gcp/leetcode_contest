class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        vector<map<int, int> > maxtr;
        maxtr.resize(A.size());
        int ans = 2;
        for(int i=0;i<A.size();i++){
            set<int> have;
            auto& m_now = maxtr[i];
            for(int j=i-1;j>=0;j--){
                int val = A[i] - A[j];
                auto& m_pre = maxtr[j];

                if(have.find(val) != have.end())continue;
                have.insert(val);

                if(m_pre.find(val) == m_pre.end()){
                    m_now[val] = 2;
                }else{
                    m_now[val] = m_pre[val] + 1;
                    ans = max(ans, m_now[val]);
                }
            }
        }
        return ans;
    }
};
