class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        int size= A.size();
        vector<unordered_map<int, int>> vc(size);
        for(int i = 1; i < size; ++i) {
            for(int k = 0; k < i; ++k) {
                int delta = A[i] - A[k];
                vc[i][delta] = vc[k][delta] + 1;
            }
        }
        
        int ret = 0;
        for(int i = 1; i < size; ++i) {
            for(auto it : vc[i])
                ret = max(ret, it.second);
        }
        return ret + 1;
    }
};
