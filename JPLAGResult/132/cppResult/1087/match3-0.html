class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        
        vector<unordered_map<int, int>> ms(A.size());
        
        int ans = 0;
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < i; j++) {
                int d = A[i] - A[j];
                if (ms[j].count(d)) {
                    ms[i][d] = ms[j][d] + 1;
                } else {
                    ms[i][d] = 2;
                }
                ans = max(ans, ms[i][d]);
            }
        }
        
        return ans;
        
    }
};
