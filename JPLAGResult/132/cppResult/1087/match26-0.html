class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        int n = A.size();
        map <int, int> dp[n];
        
        // int dp[n+1][20005];
        // fill(dp[0], dp[n+1], 0);
        // i番目までみたときに、差がj-10000の列の最長の長さ
        
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int diff = A[j] - A[i] + 10000;
                dp[j][diff] = max(dp[j][diff], dp[i][diff] + 1);
            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (auto itr = dp[i].begin(); itr != dp[i].end(); itr++) {
                ans = max(ans, itr->second);
            }
        }
        
        return ans + 1;
    }
};
