class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        int ans=2;
        for(int i=0;i<A.size();i++)
        {
            for(int j=i+1;j<A.size();j++)
            {
                int n=A.size();
                if(n-j+1<=ans)continue;
                int cur=A[j];
                int diff=A[j]-A[i];
                int count=2;
                for(int k=j+1;k<A.size();k++)
                {
                    if(A[k]-cur==diff)
                    {
                        cur=A[k];
                        count++;
                    }
                }
                ans=max(ans,count);
                
                
            }
        }
        return ans;
    }
};
