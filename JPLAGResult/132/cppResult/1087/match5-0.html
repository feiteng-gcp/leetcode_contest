class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        vector<unordered_map<int, int>> dp(A.size());
        int result = 0;
        for (int i = 1; i < A.size(); ++i) {
            for (int j = 0; j < i; ++j) {
                int key = A[i] - A[j];
                if (dp[j].count(key) == 0) {
                    dp[i][key] = max(dp[i][key], 2);
                } else {
                    dp[i][key] = max(dp[i][key], dp[j][key] + 1);
                }
                result = max(result, dp[i][key]);
            }
        }
        return result;
    }
};
