class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {

        vector<unordered_map<int, int>> lengths(A.size());
        int res = 0;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < i; ++j) {
                int diff = A[i] - A[j];
                if (lengths[j].find(diff) == lengths[j].end()) {
                    lengths[i][diff] = max(lengths[i][diff], 2);
                }
                else {
                    lengths[i][diff] = max(lengths[j][diff] + 1, lengths[i][diff]);
                }
                res = max(lengths[i][diff], res);
            }
        }
        return res;
    }
};
