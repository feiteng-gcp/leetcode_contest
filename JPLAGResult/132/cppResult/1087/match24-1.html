class Solution {
public:
    unordered_map<int,int>mp[2020];
    int longestArithSeqLength(vector<int>& A) {
        int ans = 0;
        for(int i=1;i<A.size();i++){

            for(int j=0;j<i;j++){
                int dif = A[i] - A[j];
                mp[i][dif] = max(mp[i][dif],mp[j][dif]+1);
                ans = max(ans,mp[i][dif]+1);
            }
        }
        return ans;
    }
};
