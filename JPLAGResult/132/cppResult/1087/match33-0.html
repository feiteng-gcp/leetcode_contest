class Solution {
public:
    
    int recur(vector<vector<int>>& dp, vector<int>& A, int size, int i, int j) {
        if (dp[i][j] != 0)   return dp[i][j];
        else {
            
        }
        return -1;
    }
    
    int longestArithSeqLength(vector<int>& A) {
        
        int size = A.size();
        /*
        sort(A.begin(), A.end());
        vector<vector<int>> dp(size, vector<int>(size, 0));
        recur(dp, A, size, 0, size - 1);
        */
        int ret = 1;
        for (int i = 0; i < size - 1; i++) {
            for (int j = i + 1; j < size; j++) {
                int diff = A[j] - A[i];
                int prev = A[j];
                int count = 2;
                for (int k = j + 1; k < size; k++) {
                    if (A[k] == prev + diff) {
                        prev = A[k];
                        count++;
                    }
                }
                ret = max(count, ret);
            }
        }
        return ret;
            
    }
};
