class Solution {
public:
    vector<unordered_map<int, int>> dp;
    int longestArithSeqLength(vector<int>& A) {
        int ret=2;
        for (int i=0; i<A.size(); i++) {
            dp.push_back(unordered_map<int, int>());
            for(int j=0; j<i; j++) {
                int diff=A[i]-A[j];
                dp.back()[diff]=max(dp.back()[diff], dp[j][diff]+1);
                ret=max(ret,  1+dp.back()[diff]);
            }
        }
        return ret;
    }
};
