class Solution {
public:
    static int findSolution(vector< pair<int,int> > &vec){
        int ans = 0;

        int size = vec.size();
        
        for (int i = 0; i < size; i++){
            for (int j = i + 1; j < size; j++){
                if (vec[i].second > vec[j].second) 
                    continue;

                int diff = vec[j].first - vec[i].first;
                // cout << "diff " << diff << " " << vec[i].second << " " << vec[j].second << endl;
                int tempans = 2;
                int previdx = j;
                for (int k = j + 1; k < size; k++){
                    if (vec[k].first - vec[previdx].first == diff && vec[k].second >  vec[previdx].second){
                        previdx = k;
                        tempans++;
                    }
                }
                ans = max(ans, tempans);
            }
        }
        return ans;
    }
    
    static bool cf(const pair<int,int> &a, const pair<int,int> &b){
        if (a.first == b.first)
            return a.second < b.second;
        return a.first > b.first;
    }
    
    
    int longestArithSeqLength(vector<int>& A) {
        vector<pair<int,int>> vec;
        int size = A.size();
        
        for (int i = 0; i < size; i++){
            vec.push_back(make_pair(A[i], i));
        }
        
        int ans = 0;
        
        sort(vec.begin(), vec.end());
        
        ans = max(ans, findSolution(vec));
        
        sort(vec.begin(), vec.end(), cf);
        
        ans = max(ans, findSolution(vec));
        return ans;
    }
};
