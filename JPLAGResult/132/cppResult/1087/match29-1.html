class Solution {
public:
    int longestArithSeqLength(vector<int>& a) {
        
        int n = a.size();
        unordered_map<int, int> mp[n];
        
        int ans = 1;
        for(int i = 1; i < n; ++i) {
            for(int j = i-1; j >= 0; --j) {
                int r = a[i] - a[j];
                mp[i][r] = max(mp[i][r], 1 + mp[j][r]);
                ans = max(ans, mp[i][r] + 1);
            }
        }
        return ans;
    }
};
