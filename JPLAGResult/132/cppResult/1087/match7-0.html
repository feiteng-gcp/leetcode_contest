class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        int n = A.size();
        vector<unordered_map<int,int>> vmp(n);
        int res = 2;
        
        for(int i = 1; i < n; i++) {
            for(int j = i-1; j>=0; j--) {
                int gap = A[i]-A[j];
                vmp[i][gap] = max(vmp[i][gap],vmp[j][gap]+1);
                res = max(vmp[i][gap]+1, res);
            }
        }
        return res;
    }
};
