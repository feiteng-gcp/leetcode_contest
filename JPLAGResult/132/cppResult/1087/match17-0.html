int dp[2001];
class Solution {
public:
    int longestArithSeqLength(vector<int>& A) {
        return llap(A.size(), A);
    }
    int llap(int n, vector<int> &a)
{
    if(n<=2)
        return n;
 
    int i,j,k,d;
    int mxl=2;
    int current;
    int last;
 
    //i will be the index of first element of the ap
    for(i=0;i<(n-mxl);i++)
    {
        //j will be the index of second element of the ap
        for(j=i+1;j<(n-mxl+1);j++)
        {
            //common difference
            d=a[j]-a[i];
            last=a[j];
            current=2;
 
            for(k=j+1;k<n;k++)
            {
                if(a[k]-last==d)
                {
                    //here is our element
                    current++;
                    last=a[k];
 
                }
            }
 
            mxl=max(mxl,current);
 
        }
 
    }
 
    return mxl;
}
};
