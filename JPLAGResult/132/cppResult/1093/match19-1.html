/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        stack < pair < TreeNode*, int > > stk;
        TreeNode* root = NULL;
        
        int p = 0;
        while (p < S.size())
        {
            if (S[p] == '-')

            {
                int d = 0;
                while (p < S.size() && S[p] == '-')
                {
                    d += 1;
                    p += 1;
                }
                
                int x = 0;
                while (p < S.size() && S[p] >= 48 && S[p] <= 57)
                {
                    x = x * 10 + (S[p] - 48);
                    p += 1;
                }
                
                // cout << "if " << x << " " << d << endl;
                while ((stk.top().second + 1) != d)
                    stk.pop();
                
                if (stk.top().first->left == NULL)
                {
                    TreeNode* l = new TreeNode(x);
                    stk.top().first->left = l;
                    stk.push(make_pair(l, d));
                }
                else
                {
                    TreeNode* r = new TreeNode(x);
                    stk.top().first->right = r;
                    stk.push(make_pair(r, d));
                }
            }
            else
            {
                int x = 0;
                while (p < S.size() && S[p] >= 48 && S[p] <= 57)
                {
                    x = x * 10 + (S[p] - 48);
                    p += 1;
                }
                
                // cout << "else " << x << endl;
                root = new TreeNode(x);
                stk.push(make_pair(root, 0));
            }
        }
        
        return root;
    }
};
