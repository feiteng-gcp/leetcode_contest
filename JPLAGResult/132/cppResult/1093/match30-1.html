/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
		int i;
		int N = S.size();
		unordered_map<int, TreeNode*> Map;
		
		int C = 0;
		int N1 = 0;
		for(i = 0; i < N; i++)
		{
			if(S[i] == '-')
			{
				if(N1)
				{
					cout << C << " " << N1 << endl;

					Map[C] = new TreeNode(N1);
					
					if(C)
					{
						if(Map[(C - 1)]->left)
						{
							Map[(C - 1)]->right = Map[C];
						}
						else
						{
							Map[(C - 1)]->left = Map[C];
						}
					}
					
					N1 = 0;
					C = 1;
				}
				else
				{
					C++;
				}
			}
			else
			{
				N1 = (10 * N1) + (S[i] - '0');
			}
		}
		
		Map[C] = new TreeNode(N1);
		
		if(C)
		{
			if(Map[(C - 1)]->left)
			{
				Map[(C - 1)]->right = Map[C];
			}
			else
			{
				Map[(C - 1)]->left = Map[C];
			}
		}
		
		return Map[0];
    }
};
