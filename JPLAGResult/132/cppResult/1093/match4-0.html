/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    int get_depth(int &idx, string &s) {
        int cnt = 0;
        while (idx < s.size() && s[idx] == '-') {
            ++idx;
            ++cnt;
        }
        return cnt;
    }
    
    int get_value(int &idx, string &s) {
        int val = 0;
        while (idx < s.size() && isdigit(s[idx])) {
            val = val * 10 + (s[idx] - '0');
            ++idx;
        }
        return val;
    }
    
    TreeNode* recoverFromPreorder(string s) {
        int idx = 0;
        vector<pair<int,int>> nodes;
        while (idx < s.length()) {
            int depth = get_depth(idx, s);
            int val = get_value(idx, s);
            nodes.push_back(make_pair(depth, val));
        }
        
        return build(nodes, 0);
    }
    
    TreeNode* build(vector<pair<int,int>> &nodes, int depth) {
        if (nodes.empty()) return nullptr;
        TreeNode* root = new TreeNode(nodes[0].second);
        if (nodes.size() == 1) return root;
        
        vector<pair<int,int>> l, r;
        int rindex = nodes.size();
        for (int i = 2; i < nodes.size(); i++) {
            if (nodes[i].first == depth + 1) {
                rindex = i;
                break;
            }
        }
        
        for (int i = 1; i < rindex; i++) l.push_back(nodes[i]);
        for (int i = rindex; i < nodes.size(); i++) r.push_back(nodes[i]);
        root->left = build(l, depth + 1);
        root->right = build(r, depth + 1);
        return root;
    }
};
