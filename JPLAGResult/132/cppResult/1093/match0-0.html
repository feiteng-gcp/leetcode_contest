/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        if (S.size() == 0) {
            return NULL;
        }
        //cout << S << endl;
        int x = 0, i = 0, j, c;
        while (i < S.size() && S[i] >= '0' && S[i] <= '9') {
            x = x * 10 + S[i] - '0';
            i++;
        }
        TreeNode *root = new TreeNode(x);
        string a, b;
        c = 0;
        while (i < S.size()) {
            j = i + 1;
            while (j < S.size() && S[j] == '-') {
                a += S[j];
                j++;
            }
            //cout << i << " " << j << " " << c << endl;
            if (j - i == 1) {
                if (c == 0) {
                    c = 1;
                } else {
                    break;
                }
            }
            while (j < S.size() && S[j] != '-') {
                a += S[j];
                j++;
            }
            //cout << i << " " << j << endl;
            i = j;
        }
        while (i < S.size()) {
            j = i + 1;
            while (j < S.size() && S[j] == '-') {
                b += S[j];
                j++;
            }
            while (j < S.size() && S[j] != '-') {
                b += S[j];
                j++;
            }
            i = j;
        }
        root->left = recoverFromPreorder(a);
        root->right = recoverFromPreorder(b);
        return root;
    }
};
