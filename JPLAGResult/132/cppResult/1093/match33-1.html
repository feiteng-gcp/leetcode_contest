/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        next = 0;
        while (next != S.size()) {
            auto [node, depth] = recoverOneNode(S);
            if (lastNode.size() == depth) {
                lastNode.push_back(node);
            } else {
                lastNode[depth] = node;
            }
            if (depth != 0) {
                if (lastNode[depth - 1]->left == nullptr) {
                    lastNode[depth - 1]->left = node;
                } else {
                    lastNode[depth - 1]->right = node;
                }
            }
        }
        return lastNode[0];
    }
private:

    pair<TreeNode*, int> recoverOneNode(const string& S) {
        int depth = 0;
        while (next != S.size() && S[next] == '-') {
            ++depth;
            ++next;
        }
        int value = 0;
        while (next != S.size() && isdigit(S[next])) {
            value *= 10;
            value += S[next] - '0';
            ++next;
        }
        return {new TreeNode(value), depth};
    }
    int next;
    vector<TreeNode*> lastNode;
};
