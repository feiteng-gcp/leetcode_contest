/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int len;
    TreeNode* recoverFromPreorder(string s) {
        int idx = 0;
        int depth = 0;
        vector<TreeNode*> m(1005, nullptr);
        TreeNode* dummy = new TreeNode(0);
        m[0] = dummy;
        len = s.length();
        while (idx < len) {
            int cnt = cnt_dash(s, idx);
            if (m[cnt + 1] != nullptr) {
                for (int i = depth; i >= cnt; i--) {
                    m[i + 1] = nullptr;
                }
                depth = cnt;
                TreeNode* cur = new TreeNode(get_num(s, idx));
                m[cnt]->right = cur;
                m[cnt + 1] = cur;
            }
            else {
                TreeNode* cur = new TreeNode(get_num(s, idx));
                depth = cnt;
                m[cnt]->left = cur;
                m[cnt + 1] = cur;
            }
        }
        return m[1];
    }
    // "1-401--349---90--88"
    int cnt_dash(string &s, int &stat) {
        int cnt = 0;
        while (stat < s.length() && s[stat] == '-') {
            cnt++;
            stat++;
        }
        return cnt;
    }
    int get_num(string &s, int &stat) {
        int num = 0;
        while (stat < s.length() && s[stat] != '-') {
            num *= 10;
            num += s[stat] - '0';
            stat++;
        }
        return num;
    }
};
