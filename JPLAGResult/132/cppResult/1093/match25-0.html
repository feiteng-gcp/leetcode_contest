/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<pair<int,int> >xl;
    int now = 0;
    TreeNode* dfs(int level){
        TreeNode *root = NULL;
        if(level == xl[now].first){
            root = new TreeNode(xl[now++].second);
            if(now<xl.size())root -> left = dfs(level+1);
            if(now<xl.size())root -> right = dfs(level+1);
        }
        return root;
    }
    TreeNode* recoverFromPreorder(string S) {
        int cnt = 0,val = 0;
        for(int i=0;i<S.size();i++){
            if(S[i]=='-'){
                if(val!=0){
                    xl.push_back({cnt,val});
                    cnt = 1;
                    val = 0; 
                }else cnt++;
            }else{
                val = val*10+S[i]-'0';
            }
        }
        xl.push_back({cnt,val});
        return dfs(0);
    }
};
