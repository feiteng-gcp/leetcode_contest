/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        deque<pair<int, TreeNode*>> q;
        int n = S.size();
        for (int i = 0; i < n; ++i) {
            int depth = 0;
            while (S[i] == '-') {
                ++depth;
                ++i;
            }
            int val = S[i] - '0';
            while (i + 1 < n && isdigit(S[i + 1])) {
                val = val * 10 + S[++i] - '0';
            }
            TreeNode* cur = new TreeNode(val);
            while (!q.empty() && depth <= q.back().first) q.pop_back();
            if (!q.empty()) {
                auto parent = q.back().second;
                if (parent->left == NULL) parent->left = cur;
                else parent->right = cur;
            }
            q.push_back({depth, cur});
        }
        return q.front().second;
    }
};
