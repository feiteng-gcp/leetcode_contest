/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        vector<vector<TreeNode*>> vt(S.size());
        for(int i = 0; i < S.size();) {
            int x = 0;
            while(S[i] == '-') {
                x++;
                i++;
            }
            int j = i;
            while(i < S.size() && S[i] != '-') i++;
            string sv = S.substr(j,i-j);
            int val = stoi(sv);
            TreeNode* cur = new TreeNode(val);
            vt[x].push_back(cur);
            if(x > 0)
                if(vt[x-1].back()->left)
                    vt[x-1].back()->right = cur;
                else
                    vt[x-1].back()->left = cur;
        }
        return vt[0][0];
    }
};
