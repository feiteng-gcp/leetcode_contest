/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        int n = S.size();
        
        int i = 0;
        TreeNode *t = new TreeNode(1);
        vector <TreeNode*> vec(1000);
        

        while (i < n) {
            int depth = 0;
            while (S[i] == '-') { depth++; i++; }
            
            int num = 0;
            while (i < n && S[i] - '0' >= 0 && S[i] - '0' <= 9) num = num * 10 + S[i++] - '0';
            
            if (depth == 0) {
                t->val = num;
                vec[0] = t;
            } else {
                TreeNode *par = vec[depth - 1];
                TreeNode *cur = new TreeNode(num);
                vec[depth] = cur;
                if (par->left == NULL) {
                    par->left = cur;
                } else {
                    par->right = cur;
                }
            }
        }
        
        return t;
    }
};
