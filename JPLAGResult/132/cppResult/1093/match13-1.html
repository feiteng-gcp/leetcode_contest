#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> pii;

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        vector<pii> V;
        int t = 0, N = S.length();
        while (t < N) {
            int d = 0, x = 0;
            while (S[t] == '-') {
                d++;
                t++;
            }
            while (t < N && S[t] >= '0' && S[t] <= '9') {
                x = x * 10 + S[t++] - '0';
            }
            V.push_back(pii(d, x));
        }
        //for (auto v : V) printf("(%d, %d)\n", v.first, v.second);
        t = 1;
        TreeNode* root = new TreeNode(V[0].second);
        recur(root, 0, V, t);
        return root;
    }
    

    void recur(TreeNode* root, int d, vector<pii> &V, int &t) {
        if (t >= V.size()) return;
        if (V[t].first == d+1) {
            root->left = new TreeNode(V[t].second);
            recur(root->left, d+1, V, ++t);
        }
        if (t >= V.size()) return;
        if (V[t].first == d+1) {
            root->right = new TreeNode(V[t].second);
            recur(root->right, d+1, V, ++t);
        }
    }
};
