/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        //cout << S << endl;
        TreeNode *root = NULL;
        if (S.size() == 0)
        {
            return root;
        }
        
        int depth = 0;
        for (int i=0; i<S.size(); i++)
        {
            if (S[i] != '-')
            {
                break;
            }
            depth ++;
        }
        S = S.substr(depth);
        //cout << "remove : " << S <<  endl;
        //treeNode value
        
        int value = 0;
        int vn = 0;
        for (int i=0; i<S.size(); i++)
        {
            if (S[i] >= '0' && S[i] <= '9')
            {
                value = value * 10 + S[i] - '0';
                vn++;
            }
            else
            {
                break;
            }
        }
        S = S.substr(vn);

        root = new TreeNode(value);
        //cout << "value : " << S <<  endl;
        
        //left start index
        int childDepth = 0;
        int right_index = -1;
        for (int i=depth+1; i<S.size(); i++)
        {
            if (S[i] == '-')
            {
                childDepth++;
            }
            else
            {
                if (childDepth == depth + 1)
                {
                    right_index = i - depth - 1;
                    break;
                }
                else
                {
                    childDepth = 0;
                }
            }
        }
        
        if (right_index < 0)
        {
            root->left = recoverFromPreorder(S);
            root->right = NULL;
        }
        else
        {
            root->left = recoverFromPreorder(S.substr(0, right_index));
            root->right = recoverFromPreorder(S.substr(right_index));
        }
        return root;
    }
};
