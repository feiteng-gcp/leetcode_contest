/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    // (depth, val)
    pair<int, int> readNext(const string& S, size_t& pos) {
        if (pos >= S.size()) {
            return make_pair(-1, -1);
        }
        if (pos == 0) {
            pos = S.find('-');
            if (pos == string::npos) {
                pos = S.size();
            }
            return make_pair(0, atoi(S.substr(0, pos).c_str()));
        }

        size_t depthPos = S.find_first_not_of("-", pos);
        if (depthPos == string::npos) {
            return make_pair(-1, -1);
        }
        int depth = depthPos - pos;
        size_t numPos = S.find('-', depthPos);
        if (numPos == string::npos) {
            numPos = S.size();
        }
        int val = atoi(S.substr(depthPos, numPos-depthPos).c_str());
        pos = numPos;
        return make_pair(depth, val);
    }
public:
    TreeNode* recoverFromPreorder(string S) {
        vector<TreeNode*> path;
        size_t pos = 0;
        while (true) {
            pair<int, int> pr = readNext(S, pos);
            if (pr.first < 0) {
                break;
            }
            while (int(path.size()) > pr.first) {
                path.pop_back();
            }
            TreeNode* node = new TreeNode(pr.second);

            if (path.empty()) {
                path.push_back(node);
            }
            else {
                if (!path.back()->left) {
                    path.back()->left = node;
                }
                else {
                    path.back()->right = node;
                }
                path.push_back(node);
            }
        }
        return path.front();
    }
};
