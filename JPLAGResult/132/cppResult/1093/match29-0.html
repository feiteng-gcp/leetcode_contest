/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        vector<pair<TreeNode*, int>> stk;
        for (int i = 0; i < S.size(); ) {
            int depth = 0;
            for (; i < S.size() && S[i] == '-'; ++i) {
                ++depth;
            }
            int val = 0;
            for (; i < S.size() && S[i] != '-'; ++i) {
                val = 10 * val + (S[i] - '0');
            }
            //cout <<"depth = "<<depth<<", val = "<<val<<endl;
            while (!stk.empty() && stk.back().second >= depth) {
                stk.pop_back();
            }
            TreeNode *node = new TreeNode(val);
            node->left = node->right = nullptr;
            if (depth == 0) {
                stk.emplace_back(node, 0);
            } else {
                //cout <<"parent = "<<stk.back().first->val<<endl;
                if (nullptr == stk.back().first->left) {
                    stk.back().first->left = node;
                } else {
                    stk.back().first->right = node;
                }
                stk.emplace_back(node, depth);
            }
        }
        return stk.front().first;
    }
};
