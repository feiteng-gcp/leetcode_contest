/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int n;
    vector<int> zyd;
    vector<int> bxsc;
    int index;
    void gao(TreeNode *ptr, int dep) {
        if (index < n && dep + 1 == zyd[index]) {
            //if (ptr -> left == NULL) {
            TreeNode *tptr = new TreeNode(bxsc[index ++]);
            ptr -> left = tptr;
            gao(tptr, dep + 1);
            //}   
        }
        if (index < n && dep + 1 == zyd[index]) {
            TreeNode *tptr = new TreeNode(bxsc[index ++]);
            ptr -> right = tptr;
            gao(tptr, dep + 1);
        }
    }
    
    TreeNode* recoverFromPreorder(string S) {
        for (int i = 0; i < S.size();) {
            int cntD = 0;
            int val = 0;
            while (S[i] == '-') {
                i++;
                cntD++;
            }
            while (i < S.size() && S[i] != '-') {
                val = val * 10 + S[i] - '0';
                i ++;
            }
            zyd.push_back(cntD);
            bxsc.push_back(val);
        }
        n = zyd.size();
        TreeNode *root = new TreeNode(bxsc[0]);
        index = 1;
        gao(root, 0);
        return root;
    }
};
