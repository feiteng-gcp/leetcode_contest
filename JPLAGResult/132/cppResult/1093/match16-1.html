/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    
    TreeNode* recoverFromPreorder(string S) {
        vector<pair<int,int>> X;

        int p = 0;
        while(p < S.size()) {
            int d = 0;
            while (S[p]=='-') {++d; p++;}
            int x = 0;
            while(p<S.size() && S[p]!='-') {
                x = x*10 + (S[p++]-'0');
            }
            X.push_back({d,x});
        }
        vector<TreeNode*> curr(2000);
        TreeNode* res = new TreeNode(X[0].second);
        curr[0] = res;
        
        for (int i = 1; i < X.size(); ++i) {
            int d = X[i].first;
            
            TreeNode* p = curr[d-1];
            TreeNode* c = new TreeNode(X[i].second);
            if (!p->left) {
                p->left = c;    
            } else {
                p->right = c;
            }
            curr[d] = c;
        }
        return res;
    }
};
