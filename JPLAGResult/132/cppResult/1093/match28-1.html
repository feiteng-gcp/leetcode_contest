/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        if (S.empty())
            return nullptr;
        
        const char * p = S.c_str();
        auto take = [&](){
            if (!*p)
                return make_pair(-1, -1);

            int depth = 0;
            while (*p == '-') {
                ++depth;
                ++p;
            }
            int val = 0;
            while (*p >= '0' && *p <= '9') {
                val = val * 10 + *p - '0';
                ++p;
            }
            return make_pair(val, depth);
        };
        
        vector<TreeNode *> stack;
        TreeNode * root = nullptr;
        while (*p) {
            auto x = take();
            while (stack.size() > x.second)
                stack.pop_back();
            TreeNode * node = new TreeNode(x.first);
            if (!stack.empty()) {
                if (!stack.back()->left)
                    stack.back()->left = node;
                else
                    stack.back()->right = node;
            } else {
                root = node;
            }
            stack.push_back(node);
        }
        return root;
    }
};
