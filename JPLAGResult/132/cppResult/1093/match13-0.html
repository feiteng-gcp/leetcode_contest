/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    vector<pair<int, int>> dv_pairs;
    vector<pair<int, int>>::iterator to_construct;
    
    void construct(TreeNode* node, int index) {
        if (to_construct == dv_pairs.end())
            return;
        
        if (to_construct->first == dv_pairs[index].first + 1) {
            node->left = new TreeNode(to_construct->second);
            to_construct++;
            construct(node->left, distance(dv_pairs.begin(), to_construct) - 1);
        }
        
        if (to_construct == dv_pairs.end())
            return;
        
        if (to_construct->first == dv_pairs[index].first + 1) {
            node->right = new TreeNode(to_construct->second);
            to_construct++;
            construct(node->right, distance(dv_pairs.begin(), to_construct) - 1);
        }
    }
public:
    TreeNode* recoverFromPreorder(string S) {
        int n = S.length();
        if (!n)
            return nullptr;
        int last_num_end = 0, last_num_start = 0;
        for (int i = 1; i < n; i++)
        {
            if (S[i] == '-' && S[i - 1] != '-') {
                dv_pairs.emplace_back(last_num_start - last_num_end, stoi(S.substr(last_num_start, i - last_num_start)));
                last_num_end = i;
            }
            else if (S[i] >= '0' && S[i] <= '9' && S[i - 1] == '-') {
                last_num_start = i;
            }
        }
        dv_pairs.emplace_back(last_num_start - last_num_end, stoi(S.substr(last_num_start)));
        
        // for (auto const& p : dv_pairs)
        //     printf("%d %d\n", p.first, p.second);
        
        TreeNode* root = new TreeNode(dv_pairs.begin()->second);
        to_construct = next(dv_pairs.begin());
        construct(root, 0);
        return root;
    }
};
