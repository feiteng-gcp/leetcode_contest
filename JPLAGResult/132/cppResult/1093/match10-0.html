/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
TreeNode *store[1001];
class Solution {
public:
    int now;
    int getnum(string S,int len)
    {
        int num=0;
        while(now<len)
        {
            if(S[now]=='-')break;
            num=num*10+(S[now]-'0');
            now++;
        }
        return num;
    }
    int getdepth(string S,int len)
    {
        int dep=0;
        while(now<len)
        {
            if(S[now]!='-')break;
            dep++;
            now++;
        }
        return dep;
    }
    TreeNode* recoverFromPreorder(string S) {
        int len=S.size();
        now=0;
        int top=0;
        int rootval=getnum(S,len);
        TreeNode *root=new TreeNode(rootval);
        store[top++]=root;
        while(now<len)
        {
            int dep=getdepth(S,len),val=getnum(S,len);
            while(top>dep)top--;
            TreeNode *p=new TreeNode(val);
            if(store[top-1]->left==NULL)store[top-1]->left=p;
            else store[top-1]->right=p;
            store[top++]=p;
        }
        return root;
    }
};
