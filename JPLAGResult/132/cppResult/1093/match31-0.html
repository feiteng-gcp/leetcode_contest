/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

pair<int,int> val(const string &S , int idx){
    int val=0;
    for(;idx<S.size();idx++){
        if(S[idx] == '-')break;
        val = val*10+S[idx]-'0';
    }
    return make_pair(val , idx);
}
pair<int,int> dash(const string &S , int idx){
    int ret=0;
    for(;idx<S.size();idx++){
        if(S[idx] != '-')break;
        ret++;
    }
    return make_pair(ret , idx);
}

class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        map<int, TreeNode*> ds;
        auto p = val(S, 0);
        ds[0] = new TreeNode(p.first);
        int i=p.second;
        for(;i<S.size();){
            p = dash(S, i);
            int depth = p.first;
            i = p.second;
            
            p = val(S, i);
            int v = p.first;
            i = p.second;
            
            TreeNode *u = ds[depth - 1];
            TreeNode *c = new TreeNode(v);
            if(!u->left){
                u->left = c;
            }else{
                u->right = c;
            }
            ds[depth] = c;
        }
        return ds[0];
    }
};
