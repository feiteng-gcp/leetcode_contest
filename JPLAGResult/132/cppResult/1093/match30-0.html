/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    unordered_map<int, TreeNode*> data;
    TreeNode* recoverFromPreorder(string S) {
        size_t pos;
        int d=0;
        int n;
        while(S.size()>0) {
            // cout<<pos<<endl;
            if(S[0]=='-') {
                d++;
                S=S.substr(1);
            } else {
                n=stoi(S, &pos);
                S=S.substr(pos);
                // cout<< n<< " " << d <<endl;
                data[d]=new TreeNode(n);
                if (d>0) {
                    if (!data[d-1]->left) {
                        data[d-1]->left=data[d];
                    } else {
                        data[d-1]->right=data[d];
                    }
                }
                d=0;
            }
        }
        return data[0];
    }
};
