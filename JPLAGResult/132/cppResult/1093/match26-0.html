/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    //"1-2--3--4-5--6--7"
    // 1, 10
    // 3, 8
    // 6, -1
    // 9, -1
    
    // 0, 17
    // 2, 9
    // 5, 6
    // 9, 6
    // 8, 9
    // 12, 9
    // 10, 17
    // 13, 14
    // 17, 14
    // 16, 17
    TreeNode *build(string S, int start, int end, int dep) {
        // cout << start << ", " << end << endl;
        if (start >= end) { return NULL; }
        int num = 0;
        int p = start;
        while(p < end && '-' != S[p]) { num = num * 10 + S[p] - '0'; ++p; }
        TreeNode *root = new TreeNode(num);
        
        int cnt = 0;
        int idx = -1;
        for (int i = p + dep; i < end; ++i) {
            if ('-' == S[i]) { ++cnt; }
            else {
                if (cnt == dep) { idx = i; break; }
                cnt = 0;
            }
        }
        // cout << p << ", " << idx << endl;
        if (-1 == idx) { 
            root -> left = build(S, p + dep, end, dep + 1);
        }
        else {
            root -> left = build(S, p + dep, idx - dep, dep + 1);
            root -> right = build(S, idx, end, dep + 1);
        }
        return root; 
    }
    
public:
    TreeNode* recoverFromPreorder(string S) {
        return build(S, 0, S.length(), 1);
    }
};
