/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        stack<pair<TreeNode*, int>> s;
        int i = 0;
        TreeNode* root = nullptr;
        while (i < S.size()) {
            // count '-'
            int depth = 0;
            while (S[i] == '-') {depth++; i++;}
            // get number
            int num = 0;
            while (i < S.size() && isdigit(S[i])) {
                num = num * 10 + (S[i] - '0');
                i++;
            }
            // pop until depth
            while (!s.empty() && s.top().second != depth - 1) {
                s.pop();
            }
            if (s.empty()) {
                root = new TreeNode(num);
                s.push({root, 0});
            } else if (!s.top().first->left) {
                s.top().first->left = new TreeNode(num);
                s.push({s.top().first->left, depth});
            } else {
                s.top().first->right = new TreeNode(num);
                s.push({s.top().first->right, depth});
            }
        }
        return root;
    }
};
