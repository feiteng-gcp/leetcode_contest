/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
struct p {
    int val;
    int depth;
};
class Solution {
private:
    TreeNode* traverse(vector<p>& v, int& pos, int depth){
        if(pos >= v.size()){
            return nullptr;
        }
        
        if(v[pos].depth != depth){
            pos --;
            return nullptr;
        }
        
        TreeNode* node = new TreeNode(v[pos].val);
        pos ++;
        node->left = traverse(v, pos, depth + 1);
        pos ++;
        node->right = traverse(v, pos, depth + 1);
        return node;
    }
public:

    TreeNode* recoverFromPreorder(string S) {
        vector<p> v;
        int depth = 0;
        string tmp = "";
        for(int i = 0; i < S.size(); i ++){
            if(S[i] == '-'){
                if(i && S[i - 1] != '-'){
                    v.push_back({stoi(tmp), depth});
                    depth = 1;
                    tmp = "";
                }
                else{
                    depth ++;
                }
            }
            else{
                tmp += S[i];
            }
        }
        v.push_back({stoi(tmp), depth});
        for(int i = 0; i < v.size(); i ++){
            cout << v[i].val << " " << v[i].depth << endl;
        }
        
        int pos = 0;
        return traverse(v, pos, 0);
    }
};
