/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    using pii = pair<int, int>;
    TreeNode* solve(pii *v, int L, int R)
    {
        if (L > R) return 0;
        TreeNode *it = new TreeNode(v[L].first);
        int dis = v[L].second;
        vector<int> p;
        for (int i = L+1; i <= R; ++ i)
            if (v[i].second == dis+1)
            {
                p.push_back(i);
            }
        if (p.empty())
        {
            
        }
        else if (p.size() == 1)
        {
            it->left = solve(v, L+1, R);
        }
        else
        {
            it->left = solve(v, L+1, p[1]-1);
            it->right = solve(v, p[1], R);
        }
        return it;
    }
    TreeNode* recoverFromPreorder(string s) {
        int n = s.size();
        vector<pii> v;
        int i = 0;
        while (i < n)
        {
            int cnt = 0;
            while (i < n && s[i] == '-')
            {
                ++ cnt;
                ++ i;
            }
            int x = 0;
            while (i < n && isdigit(s[i]))
            {
                x = x*10+s[i]-'0';
                ++ i;
            }
            v.emplace_back(x, cnt);
        }
        return solve(v.data(), 0, v.size()-1);
    }
};
