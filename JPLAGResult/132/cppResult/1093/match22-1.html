/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    typedef pair<int, TreeNode *> pr;
public:
    TreeNode* recoverFromPreorder(string S) {
        int len = S.size();
        size_t idx = 0, i2;
        int num = stoi(S, &idx);
        TreeNode * root = new TreeNode(num);
        stack<pr> stk;
        stk.push(pr(0, root));
        int level;
        while (idx < len) {
            level = 0;
            while (idx < len && S[idx] == '-') ++idx, ++level;
            if (idx == len) return root;
            i2 = idx;
            num = stoi(S.substr(idx), &idx);// << " ";
            idx += i2;
            while (stk.top().first >= level) stk.pop();

            if (!stk.top().second->left) {
                stk.top().second->left = new TreeNode(num);
                stk.push(pr(level, stk.top().second->left));
                
            }
            else {
                stk.top().second->right = new TreeNode(num);
                stk.push(pr(level, stk.top().second->right));
            }
            
        }
        return root;
       // TreeNode * root
    }
};
