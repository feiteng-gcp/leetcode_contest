/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
 public:
  TreeNode* recoverFromPreorder(string S) {
    int i = 0, n = S.size();
    int val = 0;
    while (i < n && S[i] != '-') {
      val = val * 10 + (S[i++] - '0');
    }
    TreeNode* root = new TreeNode(val);
    stack<pair<TreeNode*, int>> stk;
    stk.emplace(root, 0);
    while (i < n) {
      int depth = 0;
      while (i < n && S[i] == '-') {
        ++depth;
        ++i;
      }
      val = 0;
      while (i < n && S[i] != '-') {
        val = val * 10 + (S[i++] - '0');
      }
      while (!stk.empty() && depth != stk.top().second + 1) {
        stk.pop();
      }
      assert(!stk.empty());
      if (stk.top().first->left == nullptr) {
        stk.top().first->left = new TreeNode(val);
        stk.emplace(stk.top().first->left, depth);
      } else {
        stk.top().first->right = new TreeNode(val);
        stk.emplace(stk.top().first->right, depth);
      }
    }
    return root;
  }
};
