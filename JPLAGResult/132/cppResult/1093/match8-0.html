/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int getNum(string &S, int &idx) {
        int r = 0;
        while (idx < S.length() && S[idx] >= '0' && S[idx] <= '9') {
            r = r * 10 + S[idx] - '0';
            idx++;
        }
        return r;
    }
    int getLevel(string &S, int &idx) {
        int r = 0;
        while (idx < S.length() && S[idx] == '-') {
            r++;
            idx++;
        }
        return r;        
    }
public:
    TreeNode* recoverFromPreorder(string S) {
        unordered_map<int, TreeNode*> level;
        
        int idx = 0;
        TreeNode* root = new TreeNode(getNum(S, idx));
        level[0] = root;
        
        while (idx < S.length()) {
            int l = getLevel(S, idx);
            
            TreeNode* node = new TreeNode(getNum(S, idx));
            
            if (!level[l - 1]->left) {
                level[l - 1]->left = node;
            } else {
                level[l - 1]->right = node;
            }
            
            level[l] = node;
        }
        
        return root;
    }
};
