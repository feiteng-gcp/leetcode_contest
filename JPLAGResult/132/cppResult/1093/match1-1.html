/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* recoverFromPreorder(string S) {
        if(S.empty()) return nullptr;
    int tmp=0,i,mark=S.size(),index;
    for(i=0;i<S.size();i++){
        if(S[i]>='0' and S[i]<='9') tmp=tmp*10+(S[i]-'0');
        else break;
    }
    i++;
    index=i;
    TreeNode* node=new TreeNode(tmp);
    bool flag=true,swift=false;
    string s1="",s2="";
    for(i;i<S.size();i++){
        if(S[i]=='-'){
            if(flag and i+1<S.size() and S[i+1]!='-'){  swift=true;    }
            else{
                if(!flag){
                    if(swift) s2+=S[i];
                    else s1+=S[i];
                }
                flag=false;
            }
        }
        else{
            flag=true;
            if(swift) s2+=S[i];
            else s1+=S[i];   
        }
    }
    // cout<<s1<<endl;
    // cout<<s2<<endl;
    node->left=recoverFromPreorder(s1);
    node->right=recoverFromPreorder(s2);
    return node;
    }
};
