class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        int v[3]={a,b,c};
        std::sort (v, v+3);
        cout<<v[0]<<", "<<v[1]<<", "<<v[2]<<endl;

        vector<int> ans(2,0);
        if ( (v[2]-v[1]==1) && (v[1]-v[0]==1))
            ans[0] =0;
        else if ((v[2]-v[1]<=2) || (v[1]-v[0]<=2))
            ans[0] = 1;
        else 
            ans[0] = 2;
        ans[1] = v[2]-v[0]-2;
        return ans;
    }
};
