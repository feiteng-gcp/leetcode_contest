class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if (a>b) swap(a,b);
        if (a>c) swap(a,c);
        if (b>c) swap(b,c);
        int ans1=0;
        if (a+1!=b) ans1++;
        if (b+1!=c) ans1++;
        if (a+2==b || b+2==c) ans1=1;
        int ans2=0;
        ans2+=b-a-1;
        ans2+=c-b-1;
        vector<int> ret;
        ret.clear();
        ret.push_back(ans1);
        ret.push_back(ans2);
        return ret;
    }
};
