class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if (a > b) swap(a, b);
        if (b > c) swap(b, c);
        if (a > b) swap(a, b);
        int most = c - a - 2;
        int least = 2;

        if (a + 1 == b || b + 1 == c)
            least = 1;

        if (a + 2 == b || b + 2 == c)
            least = 1;

        if (a + 1 == b && b + 1 == c)
            least = 0;

        return {least, most};
    }
};
