class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        int minp = min(min(a, b), c);
        int maxp = max(max(a, b), c);
        int res_max = maxp - minp - 2;
        if(res_max==0) return vector<int>{0, 0}; 
        int res_min = 2;
        if(abs(a-b)<=2||abs(b-c)<=2||abs(a-c)<=2)
        	res_min=1;
        return vector<int>{res_min, res_max}; 
    }
};
