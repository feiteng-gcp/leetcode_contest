class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> v = { a, b, c };
        sort( v.begin(), v.end() );
        int minv = 0;
        int maxv = 0;
        if( v[1] != v[0] + 1 ) {
            minv += 1;
            maxv += v[1] - v[0] - 1;
        }
        if( v[2] != v[1] + 1 ) {
            minv += 1;
            maxv += v[2] - v[1] - 1;
        }
        if( v[1]-v[0] == 2 || v[2]-v[1] == 2 ) minv = 1;
        return {minv, maxv};
    }
};
