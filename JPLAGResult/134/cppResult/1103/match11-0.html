class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if (a > b)
            swap(a, b);
        if (a > c)
            swap(a, c);
        if (b > c)
            swap(b, c);
        int minimum_moves = 2;
        int maximum_moves = 0;
        if (b-a == 1 && c-b == 1)
            minimum_moves = 0;
        else if (b-a <= 2 || c-b <= 2)
            minimum_moves = 1;
        maximum_moves = c-a-2;
        return {minimum_moves, maximum_moves};
    }
};
