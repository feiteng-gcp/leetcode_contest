class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> x = {a,b,c};
        sort(x.begin(),x.end());
        vector<int> ret(2);
        ret[0]=(x[1]-x[0]-1>0)+(x[2]-x[1]-1>0);
        if(x[0]+2==x[1])
            ret[0]=min(ret[0],1);
        if(x[1]+2==x[2])
            ret[0]=min(ret[0],1);
        ret[1]=(x[1]-x[0]-1)+(x[2]-x[1]-1);
        return ret;
    }
};
