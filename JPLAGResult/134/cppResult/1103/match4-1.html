class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> list({a,b,c});
        sort(list.begin(), list.end());
        int m = 2;
        if(list[0]+1==list[1] and list[1]+1==list[2])
            m = 0;
        else if(list[0]+2==list[1] or list[1]+2==list[2] or list[0]+1==list[1] or list[1]+1==list[2])
            m=1;
        int M = list[1]-list[0]-1+list[2]-list[1]-1;
        return {m, M};
    }
};
