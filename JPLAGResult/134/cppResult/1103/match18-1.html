class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if (a > b)
            swap(a, b);
        if (b > c)
            swap(b, c);
        if (a > b)
            swap(a, b);
        
        int x = (a + 1 < b ? 1 : 0) + (b + 1 < c ? 1 : 0);
        if (a + 2 == b || b + 2 == c)
            x = 1;
        int y = (b - a - 1) + (c - b - 1);
        return {x, y};
    }
};
