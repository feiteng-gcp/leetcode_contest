class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c)
{
    vector<int> tab = {a, b, c};
    sort(tab.begin(), tab.end());
    int a1 = (tab[1] - tab[0] > 1) + (tab[2] - tab[1] > 1);
    if(tab[1] - tab[0] == 2 || tab[2] - tab[1] == 2) a1 = 1;
    return {a1, tab[2] - tab[0] - 2};
}
};
