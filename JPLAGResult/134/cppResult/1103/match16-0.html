class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        int mini = min(a, min(b,c));
        int maxi = max(a, max(b,c));
        int midi = a+b+c-mini-maxi;
        if(maxi-midi == 1 && midi-mini == 1)
            return vector<int> {0, 0};
        
        int res1 = 2;
        if(maxi-midi <= 2 || midi-mini <= 2)
            res1 = 1;
        int res2;
        if(maxi-midi == 1)
            res2 = midi-1-mini;
        else if(midi-mini == 1)
            res2 = maxi-midi-1;
        else
            res2 = maxi-mini-2;
        vector<int> ans{res1, res2};
        return ans;
    }
};
