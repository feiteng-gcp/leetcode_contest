class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> stones = {a, b, c};
        sort(stones.begin(), stones.end());
        
        a = stones[0];
        b = stones[1];
        c = stones[2];
        
        int Min = 2;
        if(b == a+1)    Min--;
        if(c == b+1)    Min--;
        
        if(a + 2 == b || b + 2 == c)
            Min = 1;
        
        int Max = (b - a - 1) + (c - b - 1);
        
        return {Min, Max};
    }
};
