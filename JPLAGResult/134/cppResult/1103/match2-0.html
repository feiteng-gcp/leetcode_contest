class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if (b > c) {
            int t = c;
            c = b;
            b = t;
        }
        if (a > b) {
            int t = a;
            a = b;
            b = t;
        }
        if (b > c) {
            int t = c;
            c = b;
            b = t;
        }
        
        vector<int> res;
        if (a == c - 2) {
            res.push_back(0);
        }
        else if (a == b - 1 || b == c - 1 || a == b - 2 || b == c - 2) res.push_back(1);
        else res.push_back(2);
        res.push_back(c - a - 2);
        return res;
    }
};
