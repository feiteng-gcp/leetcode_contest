class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> t = {a,b,c};
        sort(t.begin(), t.end());
        if(t[1] - t[0] == 1 && t[2] - t[1] == 1) return {0,0};
        vector<int> res(2,0);
        if(t[1] - t[0] == 1 || t[2] - t[1] == 1) res[0] = 1;
        else if(t[1] - t[0] == 2 || t[2] - t[1] == 2) res[0] = 1;
        else res[0] = 2;
        res[1] = (t[1] - t[0] - 1) + (t[2] - t[1] -1);
        //if(t[1] - t[0] != 1){
        //    res[0] += 1;
        //    res[1] += (t[1] - t[0] - 1);
        //}
        //if(t[2] - t[1] != 1){
        //    res[0] += 1;
        //    res[1] += (t[2] - t[1] -1);
        //}
        return res;
    }
};
