class Solution {
public:
	vector<int> numMovesStones(int a, int b, int c) {
		vector<int>v = { a,b,c };
		sort(all(v));
		int M = v[2] - v[0] - 2;
		int m = 0;
		if (v[1] - v[0] == 2 or v[2] - v[1] == 2) {
			m = 1;
		}
		else {
			if (v[0] != v[1] - 1)m++;
			if (v[2] != v[1] + 1)m++;
		}
		vector<int>ans = { m,M };
		return ans;
	}
};
