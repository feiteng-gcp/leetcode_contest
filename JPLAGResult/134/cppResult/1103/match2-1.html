class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if(a>b)
        {
            int tmp=a;
            a=b;
            b=tmp;
        }
        if(b>c)
        {
            int tmp=b;
            b=c;
            c=tmp;
        }
        if(a>b)
        {
            int tmp=a;
            a=b;
            b=tmp;
        }
        vector<int> ans;
        if(b==a+1&&c==b+1)
        {
            ans.push_back(0);
            ans.push_back(0);
        }
        else
        {
            if(b-a<=2||c-b<=2)ans.push_back(1);
            else ans.push_back(2);
            ans.push_back(c-a-2);
        }
        return ans;
    }
};
