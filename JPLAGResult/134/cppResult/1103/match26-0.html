class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> input = {a, b, c};
        sort(input.begin(), input.end());
        vector<int> res(2, 0);
        res.back() = input.back() - input.front() - 2;
        if (input[0] == input[1] - 1 && input[2] == input[1] + 1) return res;
        if (input[1] - input[0] < 3 || input[2] - input[1] < 3) res.front() = 1;
        else res.front() = 2;
        return res;
    }
};
