class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> vals = { a, b, c};
        sort(vals.begin(), vals.end());
        a = vals[0];
        b = vals[1];
        c = vals[2];
        int minVal = 0;
        if (a + 2 == c) {
            minVal = 0;
        }
        else if (a + 1 == b || b + 1 == c) {
            minVal = 1;
        }
        else if (a + 2 == b || b + 2 == c) {
            minVal = 1;
        }
        else {
            minVal = 2;
        }
        int maxVal = 0;
        maxVal += b - a - 1;
        maxVal += c - b - 1;
        return { minVal, maxVal };
    }
};
