class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if(a>b)swap(a,b);
        if(a>c)swap(a,c);
        if(b>c)swap(b,c);
        int m1=(c==b+1?0:1)+(a==b-1?0:1);
        if(c==b+2) m1=1;
        if(b==a+2) m1=1;
        return vector<int>{m1,c-b+b-a-2};
    }
};
