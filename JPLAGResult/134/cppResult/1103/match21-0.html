class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        int q[3] = {a, b, c};
        sort(q, q+3);
        int minn = 0;
        if(q[1] == q[0] + 1 and q[1] == q[2] - 1) minn = 0;
        else if(q[0] == q[1] - 2 or q[0] == q[1] - 1) minn = 1;
        else if(q[1] == q[2] - 2 or q[1] == q[2] - 1) minn = 1;
        else minn = 2;
        
        int maxx = q[1] - q[0] - 1 + q[2] - q[1] - 1;
        vector <int> res;
        res.push_back(minn);
        res.push_back(maxx);
        return res;
    }
};
