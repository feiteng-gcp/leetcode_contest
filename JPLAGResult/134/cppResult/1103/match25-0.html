class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> x = {a, b, c};
        sort(x.begin(), x.end());
        int rmin;
        if (x[2] - x[0] <= 2) {
            rmin = 0;
        } else if (x[1] - x[0] <= 2 || x[2] - x[1] <= 2) {
            rmin = 1;
        } else {
            rmin = 2;
        }
        return {rmin, x[2] - x[0] - 2};
    }
};
