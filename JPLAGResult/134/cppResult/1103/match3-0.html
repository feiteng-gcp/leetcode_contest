class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> v{a, b, c};
        sort(v.begin(), v.end());
        vector<int> ret(2);
        ret[0] = (v[1] > v[0] + 1) + (v[2] > v[1] + 1);
        if (v[2] - v[1] == 2 or v[1] - v[0] == 2) ret[0] = 1;
        
        ret[1] = v[2] - v[0] - 2;
        return ret;
    }
};
