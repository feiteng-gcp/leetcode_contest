class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> vec = {a, b, c};
        sort(vec.begin(), vec.end());
        a = vec[0], b = vec[1], c = vec[2];
        int x = c - a - 2;
        int y = 2;
        if (b - a <= 2 || c - b <= 2) y = 1;
        if (c - a == 2) y = 0;
        return {y, x};
    }
};
