class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        int z = max(a,max(b,c));
        int x = min(a,min(b,c));
        int y = a+b+c-x-z;
        int maxv = y-x-1 + (z-y-1);
        int minv = 2;
        if(z-y==1 && y-x==1)minv=0;
        else if(z-y==1 || y-x==1)minv=1;
        else if(z-y==2 || y-x==2)minv=1;
        
        return {minv,maxv};
    }
};
