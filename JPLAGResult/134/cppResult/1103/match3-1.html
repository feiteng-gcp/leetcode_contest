class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> v{a,b,c};
        sort(v.begin(), v.end());
        int minMove = 100000;
        if(v[1]-v[0]<=2 || v[2]-v[1]<=2)
            minMove=1;
        minMove = min(minMove, (v[0] != v[1] - 1) + (v[1] != v[2] - 1));
        return {minMove, v[2] - v[0] - 2};
    }
};
