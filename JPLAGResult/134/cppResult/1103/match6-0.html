class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> s = {a,b,c};
        sort(s.begin(),s.end());
        vector<int> res(2,0);
        if(s[1]-s[0]==2||s[2]-s[1]==2){
            res[0]=1;
        }else{
            if(s[1]-s[0]!=1)res[0]++;
            if(s[2]-s[1]!=1)res[0]++;
        }
        
        res[1] += s[1]-s[0]-1;
        res[1] += s[2]-s[1]-1;
        return res;
    }
};
