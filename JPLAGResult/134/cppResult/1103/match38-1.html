class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> A = {a, b, c};
        sort(A.begin(), A.end());
        a = A[0]; b = A[1]; c = A[2];
        vector<int> ret(2);
        ret[0] = (b - a - 1) + (c - b - 1);        
        ret[1] = (a < b - 1) + (c > b + 1);
        if (c == b + 2 || a == b - 2) ret[1] = 1;
        sort(ret.begin(), ret.end());
        return ret;
    }
};
