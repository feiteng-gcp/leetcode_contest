class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        if (a > b)
            swap(a, b);
        if (a > c)
            swap(a, c);
        if (b > c)
            swap(b, c);
        int res1 = 0, res2;
        if (a != b - 1)
            ++res1;
        if (b != c - 1)
            ++res1;
        if (a == b - 2 || b == c - 2)
            res1 = min(res1, 1);
        res2 = b - a - 1 + c - b - 1;
        vector<int>res = vector<int>({res1, res2});
        return res;
    }
};
