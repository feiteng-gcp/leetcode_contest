class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> stones({a, b, c});
        sort(stones.begin(), stones.end());
        
        int maximum_moves = stones[2] - stones[0] - 2;
        if (maximum_moves == 0) return vector<int>({0, 0});
        
        int minimum_moves = 2;
        if (stones[2] - stones[1] == 2 || stones[1] - stones[0] == 2 || 
            stones[2] - stones[1] == 1 || stones[1] - stones[0] == 1) minimum_moves = 1;
        return vector<int>({minimum_moves, maximum_moves});
    }
};
