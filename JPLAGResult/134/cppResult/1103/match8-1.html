class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {

        vector<int> A = {a, b, c};
        sort(A.begin(), A.end());

        int min_res = 2;
        if(A[1] - A[0] == 1 && A[2] - A[1] == 1)
            min_res = 0;
        else if(A[1] - A[0] <= 2 || A[2] - A[1] <= 2)
            min_res = 1;

        int max_res = (A[1] - A[0] - 1) + (A[2] - A[1] - 1);
        return {min_res, max_res};
    }
};
