class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector <int> position = {a, b, c};
        sort(position.begin(), position.end());
        a = position[0];
        b = position[1];
        c = position[2];
        int min_move = 2, max_move = 0;
        if(b == a + 1 && c == b + 1)
            min_move = 0;
        else if(b == a + 1 || b == a + 2 || b == c - 1 || b == c - 2)
            min_move = 1;
        max_move = c - a - 2;
        return {min_move, max_move};
    }
};
