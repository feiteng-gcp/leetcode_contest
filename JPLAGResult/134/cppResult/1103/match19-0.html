class Solution {
public:
	vector<int> numMovesStones(int a, int b, int c) {
		int aa = min(min(a, b), c);
		int cc = max(max(a, b), c);
		int bb = a + b + c - aa - cc;

		int M = cc - aa - 2;
		int m = 2;
		if(aa == bb - 1 && bb == cc - 1) m = 0;
		else if(aa < bb - 1 && bb == cc - 1) m = 1;
		else if(aa == bb - 1 && bb < cc - 1) m = 1;
		else if(aa == bb - 2 || bb == cc - 2) m = 1;
		return {m, M};
	}
};
