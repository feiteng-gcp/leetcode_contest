class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector < int > q = {a, b, c};
        sort(q.begin(), q.end());
        a = q[0];
        b = q[1];
        c = q[2];
        
        int ans1 = b - 1 - a;
        int ans2 = c - (b + 1);
        vector < int > ans;
        if (ans1 == 0 && ans2 == 0) {
            return {0, 0};
        }
        
        for (int i=1;i<=100;i++) {
            if (i == a || i == b || i == c) continue;
            vector < int > qq = {i, b, c};
            sort(qq.begin(), qq.end());
            if (qq[0] + 1 == qq[1] && qq[1] + 1 == qq[2]) {
                return {1, ans1 + ans2};
            }
        }
        
        for (int i=1;i<=100;i++) {
            if (i == a || i == b || i == c) continue;
            vector < int > qq = {i, a, b};
            sort(qq.begin(), qq.end());
            if (qq[0] + 1 == qq[1] && qq[1] + 1 == qq[2]) {
                return {1, ans1 + ans2};
            }
        }

        
        return {2, ans1 + ans2};
    }
};
