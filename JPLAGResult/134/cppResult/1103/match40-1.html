class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        vector<int> v = {a, b, c};
        sort(v.begin(), v.end());
        int a1 = bool(v[2]-v[1]-1)+bool(v[1]-v[0]-1);
        int a2 = v[2]-v[1]-1+v[1]-v[0]-1;
        if (v[1]-v[0] == 2 || v[2]-v[1] == 2) a1 = min(a1, 1);
        return {a1, a2};
    }
};
