class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        int t[3] = {a, b, c};
        sort(t, t + 3);
        
        a = t[0], b = t[1], c = t[2];
        
        int max_step = c - a - 2;
        int min_step = 2;
        
        if ((b == a + 1) && (c == b + 1)) {
            min_step = 0;
        }
        else if ((b == a + 2) || (c == b + 2) || (b == a + 1) || (c == b + 1)) {
            min_step = 1;
        }
        
        return {min_step, max_step};
    }
};
