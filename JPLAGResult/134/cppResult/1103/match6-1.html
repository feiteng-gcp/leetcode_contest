class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) {
        int A[3] = {a, b, c};
        sort(A, A+3);
        vector<int> ans(2, 0);
        if (A[0] + 2 == A[1] || A[1] + 2 == A[2]) {
            ans[0] = 1;
        }
        else {
            if (A[0] + 1 != A[1]) {
                ans[0]++;
            }
            if (A[1] + 1 != A[2]) {
                ans[0]++;
            }
        }
        
        
        
        ans[1] = A[1] - A[0] - 1 + A[2] - A[1] - 1;
        
        return ans;
    }
};
