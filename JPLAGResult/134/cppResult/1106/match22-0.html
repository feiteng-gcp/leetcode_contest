class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        vector<vector<int>> used=vector<vector<int>>(401,vector<int>(401,false));
        vector<vector<int>> block=vector<vector<int>>(401,vector<int>(401,false));
        for(int i=0;i<blocked.size();i++){
            int x=blocked[i][0]-source[0];
            int y=blocked[i][1]-source[1];
            if(x>200||x<-200||y>200||y<-200) continue;
            block[x+200][y+200]=true;
        }
        queue<pair<int,int>> qq;
        used[200][200]=true;
        qq.push(make_pair(0,0));
        int dd[][2]={{1,0},{-1,0},{0,1},{0,-1}};
        int dx=target[0]-source[0];
        int dy=target[1]-source[1];
        while(!qq.empty()){
            auto p=qq.front();
            qq.pop();
            for(int i=0;i<4;i++){
                int x=dd[i][0]+p.first;
                int y=dd[i][1]+p.second;
                if(x==dx&&y==dy) return true;
                if(x+source[0]<0||x+source[0]>=1000000||y+source[1]<0||y+source[1]>=1000000) continue;
                if(x>200||x<-200||y>200||y<-200) return true;
                if(block[x+200][y+200]) continue;
                if(used[x+200][y+200]) continue;
                used[x+200][y+200]=true;
                qq.push(make_pair(x,y));
            } 
        }
        return false;
    }
};
