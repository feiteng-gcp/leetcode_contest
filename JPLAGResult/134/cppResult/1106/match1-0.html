int dir[][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
class Solution {
public:
    long long int getidx(int x, int y) {
        long long int key = x;
        key = key * 1000000 + y;
        return key;
    }
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        unordered_set<long long int> m;
        for (auto it: blocked) {
            m.insert(getidx(it[0], it[1]));
        }
        
        int R = 999999;
        int C = 999999;
        queue<int> qx;
        queue<int> qy;
        qx.push(source[0]);
        qy.push(source[1]);
        m.insert(getidx(source[0], source[1]));
        int area = 0;
        while (!qx.empty()) {
            int x = qx.front(); qx.pop();
            int y = qy.front(); qy.pop();
            ++area;
            if (area > 10000) return true;
            for (int i = 0; i < 4; ++i) {
                int nx = x + dir[i][0];
                int ny = y + dir[i][1];
                if (nx < 0 || ny < 0 || nx >= R || ny >= C) continue;
                if (m.count(getidx(nx, ny))) continue;
                m.insert(getidx(nx, ny));
                qx.push(nx);
                qy.push(ny);
            }
        }
        return false;
    }
};
