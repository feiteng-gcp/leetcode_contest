class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        int bs = blocked.size();
        int max_blocked = (bs/2+1)*(bs/2+1);
        map<int, unordered_set<int>> block_adj;
        for(auto b:blocked){
        	block_adj[b[0]].insert(b[1]);
        }
        if(bfs(block_adj, source, max_blocked)&&bfs(block_adj, target, max_blocked)) return true;
        return false;

    }

private:
	int hm[4] = {0, 1, 0, -1};
	int vm[4] = {1, 0, -1, 0};
	bool bfs(map<int, unordered_set<int>>& blocked, vector<int>& start, int target){
		map<int, unordered_set<int>> visited;
		queue<vector<int>> to_visit;
		to_visit.push(start);
		int count = 0;
		while(!to_visit.empty()){
			vector<int> cur = to_visit.front();
			visited[cur[0]].insert(cur[1]);
			to_visit.pop();
			for(int i=0;i<4;i++){
				if(cur[0]+hm[i]<0||cur[0]+hm[i]>999999||cur[1]+vm[i]<0||cur[1]+vm[i]>999999) continue;
				if(visited[cur[0]+hm[i]].find(cur[1]+vm[i])==visited[cur[0]+hm[i]].end()&&blocked[cur[0]+hm[i]].find(cur[1]+vm[i])==blocked[cur[0]+hm[i]].end()){
					to_visit.push(vector<int>{cur[0]+hm[i], cur[1]+vm[i]});
					visited[cur[0]+hm[i]].insert(cur[1]+vm[i]);
					count+=1;
				}
			}
			if(count>target) return true;
		}
		return false;

	}
};
