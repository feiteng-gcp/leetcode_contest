class Solution {
public:
	bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
		set<pair<int, int>> block, v1, v2;
		for (auto p : blocked) block.insert({ p[0], p[1] });
		int x = source[0], y = source[1], xx = target[0], yy = target[1], a, b, aa, bb;
		int v[4][2] = { 1, 0, 0,1, -1, 0, 0, -1 };
		int i, j, k;

		queue<pair<int, int>> q;
		q.push({ x, y });
		v1.insert({ x, y });
		while (!q.empty() && (v1.size() < 15000)) {
			a = q.front().first;
			b = q.front().second;
			q.pop();

			for (i = 0; i < 4; i++) {
				aa = a + v[i][0]; 
				bb = b + v[i][1];

				if ((block.find({ aa, bb }) == block.end()) && (v1.find({ aa, bb }) == v1.end())
					&& (aa >= 0) && (bb >= 0) && (aa < 1000000) && (bb < 1000000)) {
					q.push({ aa, bb });
					v1.insert({ aa, bb });
				}
			}
		}

		queue<pair<int, int>> q2;
		q2.push({ xx, yy });
		v2.insert({ xx, yy });
		while (!q.empty() && (v2.size() < 15000)) {
			a = q.front().first;
			y = q.front().second;
			q.pop();

			for (i = 0; i < 4; i++) {
				aa = a + v[i][0];
				bb = b + v[i][1];

				if ((block.find({ aa, bb }) == block.end()) && (v2.find({ aa, bb }) == v2.end())
					&& (aa >= 0) && (bb >= 0) && (aa < 1000000) && (bb < 1000000)) {
					if (v1.find({ aa, bb }) != v1.end()) return true;

					q2.push({ aa, bb });
					v2.insert({ aa, bb });
				}
			}
		}

		if ((v1.size() >= 15000) && (v2.size() >= 15000)) return true;
		return false;
	}
};
