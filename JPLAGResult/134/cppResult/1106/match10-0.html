class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        const int MAX_STEP = 200;
        const int MAX_GRID = 1000000;
        set <pair<int, int> > blocked_set;
        for(auto& vs: blocked){
            blocked_set.insert(make_pair(vs[0], vs[1]));
        }
        map <pair<int, int>, int > visit;
        queue <pair<int, int> > Q;
        Q.push(make_pair(source[0], source[1]));
        visit[make_pair(source[0], source[1])] = 0;
        
        const int OFF_R[4] = {1, 0, -1, 0};
        const int OFF_C[4] = {0, 1, 0, -1};
        while(!Q.empty()){
            pair<int, int> u = Q.front(); Q.pop();
            int now = visit[u];
            if(u.first == target[0] and u.second == target[1]) return true;
            if(now >= MAX_STEP) return true;
            for(int k = 0; k < 4; k++){
                int nr = u.first + OFF_R[k];
                int nc = u.second + OFF_C[k];
                if(nr >= 0 and nr < MAX_GRID and nc >= 0 and nc < MAX_GRID){
                    pair<int, int> v = make_pair(nr, nc);
                    if(blocked_set.find(v) != blocked_set.end()) continue;
                    if(visit.find(v) == visit.end()){
                        visit[v] = now + 1;
                        Q.push(v);
                    }
                }
            }
        }
        return false;
    }
};
