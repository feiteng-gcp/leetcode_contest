int R, C, N;
vector<int> rows, cols;
vector<bool> is_blocked;
vector<bool> visited;
vector<vector<int>> adj;

int compress_row(int r) {
    return lower_bound(rows.begin(), rows.end(), r) - rows.begin();
}

int compress_col(int c) {
    return lower_bound(cols.begin(), cols.end(), c) - cols.begin();
}

int convert(int r, int c) {
    return r * C + c;
}

void add_edge(int a, int b) {
    adj[a].push_back(b);
    adj[b].push_back(a);
}

void bfs(int src) {
    vector<int> queue;
    visited[src] = true;
    queue.push_back(src);
    int front = 0;

    while (front < (int) queue.size()) {
        int current = queue[front++];

        for (int neighbor : adj[current])
            if (!visited[neighbor] && !is_blocked[neighbor]) {
                visited[neighbor] = true;
                queue.push_back(neighbor);
            }
    }
}

class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        rows = cols = {};

        for (vector<int> x : blocked) {
            rows.push_back(x[0]);
            cols.push_back(x[1]);
        }

        rows.push_back(source[0]);
        cols.push_back(source[1]);

        rows.push_back(target[0]);
        cols.push_back(target[1]);

        rows.push_back(0);
        rows.push_back(999999);

        cols.push_back(0);
        cols.push_back(999999);

        sort(rows.begin(), rows.end());
        rows.resize(unique(rows.begin(), rows.end()) - rows.begin());

        sort(cols.begin(), cols.end());
        cols.resize(unique(cols.begin(), cols.end()) - cols.begin());

        R = rows.size();
        C = cols.size();
        N = R * C;
        is_blocked.assign(N, false);
        adj.assign(N, {});
        visited.assign(N, false);

        for (vector<int> x : blocked) {
            int r = compress_row(x[0]);
            int c = compress_col(x[1]);
            is_blocked[convert(r, c)] = true;
        }

        for (int r = 0; r < R; r++)
            for (int c = 0; c < C; c++) {
                int index = convert(r, c);

                if (r + 1 < R)
                    add_edge(index, convert(r + 1, c));

                if (c + 1 < C)
                    add_edge(index, convert(r, c + 1));
            }

        int src = convert(compress_row(source[0]), compress_col(source[1]));
        int tgt = convert(compress_row(target[0]), compress_col(target[1]));

        if (is_blocked[src] || is_blocked[tgt])
            return false;

        bfs(src);
        return visited[tgt];
    }
};
