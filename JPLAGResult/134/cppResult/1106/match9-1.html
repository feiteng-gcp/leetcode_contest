#define rep(i,l,r) for(int i=l;i<r;++i)
typedef pair<int,int>pii;

class Solution {
public:
	bool isEscapePossible(vector<vector<int>>& blocked,vector<int>& source,vector<int>& target) {
		unordered_map<int,unordered_set<int>>mp;
		for(auto v:blocked)
			mp[v[0]].insert(v[1]);
		bool res=isblock(source[0],source[1],blocked,mp)||isblock(target[0],target[1],blocked,mp);
		return 1-res;
	}

	bool isblock(int x0,int y0,vector<vector<int>>&blocked,unordered_map<int,unordered_set<int>>&mp){
		int nb=blocked.size();
		int tar=(nb-1)*nb/2;
		queue<pii>que;unordered_map<int,unordered_set<int>>vis;
		que.push(pii(x0,y0));vis[x0].insert(y0);
		int cnt=1;
		int dx[4]={-1,0,0,1},dy[4]={0,-1,1,0};
		while(!que.empty()){
			if(cnt>tar)return false;
			int x=que.front().first,y=que.front().second;que.pop();
			rep(i,0,4){
				int tx=dx[i]+x,ty=dy[i]+y;
				if(tx>=0&&tx<1e6&&ty>=0&&ty<1e6&&(mp.count(tx)==0||mp[tx].count(ty)==0)&&(vis.count(tx)==0||vis[tx].count(ty)==0)){
					que.push(pii(tx,ty));
					vis[tx].insert(ty);
					cnt++;
				}
			}
		}
		return true;
	}
};
