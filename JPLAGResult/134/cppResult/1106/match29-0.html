#define ALL(x) (x).begin(), (x).end()

class Solution {
public:
  bool isEscapePossible(vector<vector<int>>& bs, vector<int>& src, vector<int>& dst) {
    vector<int> xs;
    vector<int> ys;
    xs.push_back(0); xs.push_back(999999);
    ys.push_back(0); ys.push_back(999999);
    for (auto &b: bs) {
      xs.push_back(b[0]); ys.push_back(b[1]);
      if (b[0]) xs.push_back(b[0]-1);
      xs.push_back(b[0]+1);
      if (b[1]) xs.push_back(b[1]-1);
      ys.push_back(b[1]+1);
    }
    sort(ALL(xs)); xs.erase(unique(ALL(xs)), xs.end());
    sort(ALL(ys)); ys.erase(unique(ALL(ys)), ys.end());

    int n = xs.size(), m = ys.size();
    vector<vector<char>> v(n, vector<char>(m));
    for (auto &b: bs) {
      int x = lower_bound(ALL(xs), b[0])-xs.begin();
      int y = lower_bound(ALL(ys), b[1])-ys.begin();
      v[x][y] = 1;
    }

    int sx = lower_bound(ALL(xs), src[0])-xs.begin();
    int sy = lower_bound(ALL(ys), src[1])-ys.begin();
    int tx = lower_bound(ALL(xs), dst[0])-xs.begin();
    int ty = lower_bound(ALL(ys), dst[1])-ys.begin();
    vector<pair<int,int>> q;
    q.emplace_back(sx,sy);
    v[sx][sy] = 1;
    for(int i=0;i<q.size();i++) {
      int r = q[i].first, c = q[i].second;
      if(r==tx && c == ty) return true;
      if(r&&!v[r-1][c]++) q.emplace_back(r-1,c);
      if(r+1<n&&!v[r+1][c]++) q.emplace_back(r+1,c);
      if(c && !v[r][c-1]++) q.emplace_back(r,c-1);
      if(c+1<m&& !v[r][c+1]++) q.emplace_back(r,c+1);
    }
    return false;
  }
};
