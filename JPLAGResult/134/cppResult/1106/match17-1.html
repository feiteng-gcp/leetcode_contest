class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        vector<int> xs, ys;
        for (auto p: blocked) {
            xs.push_back(p[0]);
            ys.push_back(p[1]);
        }
        xs.push_back(source[0]);
        ys.push_back(source[1]);
        xs.push_back(target[0]);
        ys.push_back(target[1]);
        xs.push_back(0);
        ys.push_back(0);
        xs.push_back(1000000 - 1);
        ys.push_back(1000000 - 1);
        sort(xs.begin(), xs.end());
        sort(ys.begin(), ys.end());
        xs.erase(unique(xs.begin(), xs.end()), xs.end());
        ys.erase(unique(ys.begin(), ys.end()), ys.end());
        map<int, int> mapx, mapy;
        for (int i = 0; i < xs.size(); i++) mapx[xs[i]] = i;
        for (int i = 0; i < ys.size(); i++) mapy[ys[i]] = i;
        int h = ys.size(), w = xs.size();
        vector<vector<int>> fld(h, vector<int>(w, 0));
        for (auto p: blocked) fld[mapy[p[1]]][mapx[p[0]]] = 1;
        vector<vector<int>> seen(h, vector<int>(w, false));
        seen[mapy[source[1]]][mapx[source[0]]] = true;
        queue<pair<int, int>> q;
        q.push({mapy[source[1]], mapx[source[0]]});
        vector<int> dy = {0, 0, -1, 1}, dx = {1, -1, 0, 0};
        while (q.size() > 0) {
            auto cur = q.front();
            q.pop();
            for (int k = 0; k < 4; k++) {

                auto ny = cur.first + dy[k], nx = cur.second + dx[k];
                if (ny >= 0 and nx >= 0 and ny < h and nx < w) {
                    if ((not seen[ny][nx]) and (fld[ny][nx] == 0)) {
                        seen[ny][nx] = true;
                        q.push({ny, nx});
                    }
                }
            }
        }
        return seen[mapy[target[1]]][mapx[target[0]]];
    }
};
