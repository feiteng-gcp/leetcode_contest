typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 666
#define B 999999
int xs[MAXN], X, ys[MAXN], Y;
bool v[MAXN][MAXN];
int dir[4][2] = {-1,0,1,0,0,1,0,-1};
class Solution {
public:
  void add(int x, int y) {
    xs[X++] = x;
    if (x + 1 <= B) xs[X++] = x+1;
    if (x - 1 >= 0) xs[X++] = x-1;
    ys[Y++] = y;
    if (y + 1 <= B) ys[Y++] = y+1;
    if (y - 1 >= 0) ys[Y++] = y-1;
  }
  int getX(int x) {
    return lower_bound(xs, xs + X, x) - xs;
  }
  int getY(int y) {
    return lower_bound(ys, ys + Y, y) - ys;
  }
  bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
    X = Y = 0;
    REP(i,0,blocked.size()) {
      int x = blocked[i][0];
      int y = blocked[i][1];
      add(x, y);
    }
    add(source[0], source[1]);
    add(target[0], target[1]);
    xs[X++] = 0;
    xs[X++] = 999999;
    ys[Y++] = 0;
    ys[Y++] = 999999;

    sort(xs, xs + X);
    sort(ys, ys + Y);
    X = unique(xs, xs + X) - xs;
    Y = unique(ys, ys + Y) - ys;

    FILL(v, false);

    REP(i,0,blocked.size()) {
      int x = getX(blocked[i][0]);
      int y = getY(blocked[i][1]);
      v[x][y] = true;
    }
      
    int sx = getX(source[0]);
    int sy = getY(source[1]);
    int tx = getX(target[0]);
    int ty = getY(target[1]);
    if (v[tx][ty]) return false;
      
    queue<PII> q;
    q.push(PII(sx, sy));
    v[sx][sy] = true;
    while (q.size()) {
      PII now = q.front(); q.pop();
      int x = now.first, y = now.second;
      REP(d,0,4) {
        int nx = x + dir[d][0], ny = y + dir[d][1];
        if (nx < 0 || ny < 0 || nx >= X || ny >= Y || v[nx][ny]) continue;
        q.push(PII(nx, ny));
        v[nx][ny] = true;
      }
    }
    return v[tx][ty];
  }
};
