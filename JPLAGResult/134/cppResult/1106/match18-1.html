class Solution {
public:
    int dx[4] = {0, 0, 1, -1};
    int dy[4] = {1, -1, 0, 0};
    const int maxn = 1e4+10;
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        typedef pair<int, int> pii;
        set<pii> st, vis;
        for(auto v: blocked){
            st.insert(make_pair(v[0], v[1]));
        }
        queue<pii> q;
        q.push(make_pair(source[0], source[1]));
        int cur = 0;
        vis.insert(make_pair(source[0], source[1]));
        while(!q.empty() && cur < maxn){

            cur++;
            pii p = q.front(); q.pop();
            for(int i = 0; i < 4; i++){
                int nx = p.first+dx[i], ny = p.second+dy[i];
                if(nx >= 0 && nx < 1e6 && ny >= 0 && ny < 1e6 && !st.count(make_pair(nx, ny)) && !vis.count(make_pair(nx, ny))){
                    vis.insert(make_pair(nx, ny));
                    q.push(make_pair(nx, ny));
                }
            }
        }
        if(cur != maxn)return 0;
        while(!q.empty())q.pop();
        q.push(make_pair(target[0], target[1]));
        cur = 0;
        while(!q.empty() && cur < maxn){
            cur++;
            pii p = q.front(); q.pop();
            for(int i = 0; i < 4; i++){
                int nx = p.first+dx[i], ny = p.second+dy[i];
                if(nx >= 0 && nx < 1e6 && ny >= 0 && ny < 1e6 && !st.count(make_pair(nx, ny)) && !vis.count(make_pair(nx, ny))){
                    vis.insert(make_pair(nx, ny));
                    q.push(make_pair(nx, ny));
                }
            }
        }
        if(cur != maxn)return 0;
        return 1;
    }
};
