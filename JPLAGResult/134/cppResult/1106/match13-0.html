class Solution {
public:
    vector<int> allX, allY;
    
    void add(vector<int> blocked) {
        if (blocked[0] > 0)
            allX.push_back(blocked[0] - 1);
        allX.push_back(blocked[0]);
        if (blocked[0] < 999999)
            allX.push_back(blocked[0] + 1);

        if (blocked[1] > 0)
            allY.push_back(blocked[1] - 1);
        allY.push_back(blocked[1]);
        if (blocked[1] < 999999)
            allY.push_back(blocked[1] + 1);
    }
    
    int getX(int x) {
        return lower_bound(allX.begin(), allX.end(), x) - allX.begin();
    }
    
    int getY(int y) {
        return lower_bound(allY.begin(), allY.end(), y) - allY.begin();
    }
    
    vector<vector<int>> grid;
    
    int targetX, targetY;
    
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        add(source);
        add(target);
        for (int i = 0; i < blocked.size(); ++i) {
            add(blocked[i]);
        }
        
        sort(allX.begin(), allX.end());
        allX.resize(unique(allX.begin(), allX.end()) - allX.begin());
        
        sort(allY.begin(), allY.end());
        allY.resize(unique(allY.begin(), allY.end()) - allY.begin());
        
        grid.resize(allX.size(), vector<int>(allY.size(), 0));
        
        for (int i = 0; i < blocked.size(); ++i) {
            grid[getX(blocked[i][0])][getY(blocked[i][1])] = 2;
        }
        
        targetX = getX(target[0]);
        targetY = getY(target[1]);
        
        //cout << targetX << "    " << targetY << endl;
        
        return go(getX(source[0]), getY(source[1]));
    }
    
    int dx[4] = {-1, 0, 0, 1};
    int dy[4] = {0, -1, 1, 0};
    
    bool go(int x, int y) {
        vector<pair<int, int>> q;
        q.push_back({x, y});
        grid[x][y] = 1;
        for (int i = 0; i < q.size(); ++i) {
            x = q[i].first, y = q[i].second;
            if (x == targetX && y == targetY) return true;
            
            for (int t = 0; t < 4; ++t) {
                int nx = x + dx[t], ny = y + dy[t];
                if (nx >= 0 && nx < grid.size() && ny >= 0 && ny < grid[0].size() && grid[nx][ny] == 0) {
                    q.push_back({nx, ny});
                    grid[nx][ny] = 1;
                }
            }
        }
        
        return false;
    }
};
