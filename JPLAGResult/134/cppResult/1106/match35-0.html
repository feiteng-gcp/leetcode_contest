class Solution {
public:
    const int N = 20000;
    int bfs(set<pair<int, int>>& tt, int sx, int sy, set<pair<int, int>>& se) {
      queue<pair<int, int>> q;
      q.push({sx, sy});
      set<pair<int, int>> vis;
      vis.insert({sx, sy});
      int dx[] = {-1, 0, 0, 1};
      int dy[] = {0, -1, 1, 0};
      int cnt = 1;
      while (!q.empty()) {
        int x, y;
        tie(x, y) = q.front();
        q.pop();
        for (int d = 0; d < 4; d++) {
          int tx = x + dx[d], ty = y + dy[d];
          if (tx >= 0 && tx < 1000000 && ty >= 0 && ty < 1000000 && !se.count({tx, ty})) {
            if (tt.count({tx, ty})) return -1;
            if (!vis.count({tx, ty})) {
              vis.insert({tx, ty});
              q.push({tx, ty});
              cnt++;
            }
          }
        }
        if (cnt >= N) break;
      }
      for (auto e : vis) {
        tt.insert(e);
      }
      return cnt;
    }  
  
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
      int sx = source[0], sy = source[1], tx = target[0], ty = target[1];
      set<pair<int, int>> se;
      for (auto& v : blocked) {
        se.insert({v[0], v[1]});
      }
      set<pair<int, int>> vis;
      int k1 = bfs(vis, sx, sy, se);
      if (k1 == -1) return true;
      int k2 = bfs(vis, tx, ty, se);
      if (k2 == -1) return true;
      return k1 >= N && k2 >= N;
    }
};
