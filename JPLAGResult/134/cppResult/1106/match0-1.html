typedef pair<int, int> ii;

const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        int sx = source[0], sy = source[1];
        int tx = target[0], ty = target[1];
        set<ii> forbid;
        for (auto& it : blocked) {
            forbid.insert(ii{it[0], it[1]});
        }
        set<ii> visit;
        queue<ii> Q;
        Q.push({sx, sy});
        visit.insert({sx, sy});
        int limit = 5e4;
        int n = 1e6;
        while (!Q.empty() && limit--) {
            //if (limit % 1000 == 0) cout << limit << endl;
            ii cur = Q.front();
            Q.pop();
            int x = cur.first, y = cur.second;

            if (ii{x, y} == ii{tx, ty}) return true;
            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0];
                int ny = y + d[k][1];
                if (nx < 0 || nx >= n || ny < 0 || ny >= n) continue;
                if (forbid.count({nx, ny})) continue;
                if (visit.count({nx, ny})) continue;
                visit.insert({nx, ny});
                Q.push({nx, ny});
            }
        }
        //cout << limit << endl;
        return limit <= 0;
    }
};
