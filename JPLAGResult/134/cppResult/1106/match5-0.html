class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
       std::queue<pair<int, int>> q;
      int limit = blocked.size() * blocked.size() * 3 + 5;
      pair<int, int> S{source[0], source[1]};
      pair<int, int> T{target[0], target[1]};
      return bfs(blocked, S, T, limit) >= limit && bfs(blocked, T, S, limit) >= limit;
    }
  int kN = 1000000;
  int bfs(const vector<vector<int>> &blocked, pair<int, int> S, pair<int, int> T, int limit) {
    std::set<pair<int, int>> blocked_set;
    for (const auto &b : blocked) {
      blocked_set.emplace(b[0], b[1]);
    }
    queue<pair<int, int>> q; 
    q.push(S);
    std::vector<std::pair<int, int>> movements = {
      {0, 1}, {0, -1}, {1, 0}, {-1, 0}
    };
    std::set<pair<int, int>> visited;
    visited.insert(S);
    while (q.size() && visited.size() < limit) {
      auto s = q.front(); 
      if (s == T) {
        return limit;
      }
      q.pop();
      for (const auto &movement : movements) {
        pair<int, int> v = s;
        v.first += movement.first;
        v.second += movement.second;
        if (0 <= v.first && v.first < kN && 0 <= v.second && 0 < v.second && !blocked_set.count(v) && !visited.count(v)) {
          visited.insert(v);
          q.push(v);
        }
      }
    }
    return visited.size();
  }
};
