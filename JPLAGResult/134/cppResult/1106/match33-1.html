class Solution {
public:
    bool isEscapePossible(vector<vector<int>>& blocked, vector<int>& source, vector<int>& target) {
        	int bn = blocked.size();
	int N = 1000000;
	int md = min(bn, abs(source[0] - target[0]) + abs(source[1] - target[1]) - 1);
	set<vector<int>> s;
	for (vector<int> v : blocked)
		s.insert(v);
	for (int i = 1; i <= md; i++) {
		bool f = true;
		for (int x = source[0], y = source[1] + i; y > source[1]; x++, y--) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (!f)
			continue;

		for (int x = source[0] + i, y = source[1]; x < source[0]; x--, y--) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (!f)
			continue;

		for (int x = source[0], y = source[1] - i; y < source[1]; x--, y++) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (!f)
			continue;

		for (int x = source[0] - i, y = source[1]; x < source[1]; x++, y++) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (f)
			return false;
	}

	for (int i = 1; i <= md; i++) {
		bool f = true;
		for (int x = target[0], y = target[1] + i; y > target[1]; x++, y--) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (!f)
			continue;

		for (int x = target[0] + i, y = target[1]; x < target[0]; x--, y--) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (!f)
			continue;

		for (int x = target[0], y = target[1] - i; y < target[1]; x--, y++) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (!f)
			continue;

		for (int x = target[0] - i, y = target[1]; x < target[1]; x++, y++) {
			if (x >= N || x < 0 || y >= N || y < 0)
				continue;
			if (s.count({ x,y }) == 0) {
				f = false;
				break;
			}
		}
		if (f)
			return false;
	}

	return true;
    }
};
