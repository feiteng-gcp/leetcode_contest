class Solution {
public:
    int maxUncrossedLines(vector<int>& A, vector<int>& B) {
      int n = A.size(), m = B.size();
      vector<vector<int>> f(n + 1, vector<int>(m + 1));
      for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
          if (A[i - 1] == B[j - 1]) f[i][j] = f[i - 1][j - 1] + 1;
          else f[i][j] = max(f[i][j - 1], f[i - 1][j]);
        }
      }
      return f[n][m];
    }
};
