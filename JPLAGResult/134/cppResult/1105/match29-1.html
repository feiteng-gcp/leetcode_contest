class Solution {
public:
    int maxUncrossedLines(vector<int>& A, vector<int>& B) {
        int N = A.size(), M = B.size();
        vector<vector<int>> f(N + 1, vector<int>(M + 1, 0));
        for (int i = 0; i <= N; ++i)
            f[i][0] = 0;
        for (int i = 0; i <= M; ++i)
            f[0][i] = 0;

        for (int i = 1; i <= N; ++i) {
            for (int j = 1; j <= M; ++j) {
                f[i][j] = max(f[i - 1][j], f[i][j - 1]);
                if (A[i - 1] == B[j - 1])
                    f[i][j] = max(f[i][j], f[i - 1][j - 1] + 1);
            }
        }
        return f[N][M];
    }
};
