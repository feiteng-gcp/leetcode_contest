class Solution {
public:
    int maxUncrossedLines(vector<int>& A, vector<int>& B) {
        int n = A.size();
        int m = B.size();
        vector<vector<int>> cnt(n + 1, vector<int>(m + 1, 0));
        for(int i = 0; i < n; i++){
            for(int j = 0; j < B.size(); j++){
                if(A[i] == B[j]){
                    cnt[i + 1][j + 1] = cnt[i][j] + 1;
                }else{
                    cnt[i + 1][j + 1] = max(cnt[i][j + 1], cnt[i + 1][j]);
                }
            }
        }
        return cnt[n][m];
    }
};
