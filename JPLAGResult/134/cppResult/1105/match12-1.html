class Solution {
public:
    int maxUncrossedLines(vector<int>& A, vector<int>& B) {
        int m = A.size();
        int n = B.size();
        vector<vector<int>> C(m+1, vector<int>(n+1,0));
        for (int i = 1; i<=m;i++)
        {
            for (int j=1;j<=n;j++)
            {
                if (A[i-1]==B[j-1])
                {
                    C[i][j] = C[i-1][j-1]+1;
                }
                else
                {
                    C[i][j] = max(C[i-1][j],C[i][j-1]);
                }
            }
        }
        return C[m][n];
    }
};
