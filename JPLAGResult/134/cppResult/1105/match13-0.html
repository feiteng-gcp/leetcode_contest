class Solution {
public:
    int maxUncrossedLines(vector<int>& A, vector<int>& B) {
        int n = A.size(),m = B.size();
        // map<pair<int,int>,int> lines;
        vector<vector<int>> dp(n + 1,vector<int> (m + 1,0));
        // dp[0][0] = 1;
        // for(int i = 0;i < n;i++){
        //     for(int j = 0;j < m;j++){
        //         if(A[i] == B[j])lines[{i,j}] = 1;
        //     }
        // }
        for(int i = 1;i <= n;i++){
            for(int j = 1;j <= m;j++){
                if(A[i - 1] == B[j - 1]){
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                }else{
                    dp[i][j] = max(dp[i - 1][j],dp[i][j - 1]);
                }
                // cout<<dp[i][j]<<" ";
            }
            // cout<<endl;
        }
        
        return dp[n][m];
        
    }
};
