class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<bool>> flag(grid.size(), vector<bool>(grid[0].size(), false));
        vector<vector<bool>> flag1(grid.size(), vector<bool>(grid[0].size(), false));
        int tx[4] = {0, 0, 1, -1};
        int ty[4] = {1, -1, 0, 0};
        queue<vector<int>> q;
        q.push({r0, c0});
        flag[r0][c0] = true;
        int color1 = grid[r0][c0];
        while(!q.empty()){
            vector<int> tmp = q.front();
            int r = tmp[0], c = tmp[1];
            if(r == 0 || c == 0 || r == grid.size()-1 || c == grid[0].size()-1){
                flag1[r][c] = true;
            }
            else{
                for(int i = 0; i < 4; ++i){
                    if(color1 != grid[r+tx[i]][c+ty[i]]){
                        flag1[r][c] = true;
                        break;
                    }
                }
            }
            q.pop();
            for(int i = 0; i < 4; ++i){
                int r1 = r + tx[i], c1= c + ty[i];
                if(r1 >= 0 && c1 >= 0 && r1 < grid.size() && c1 < grid[0]. size() && flag[r1][c1] == false && color1 == grid[r1][c1]){
                    flag[r1][c1] = true;
                    q.push({r1, c1});
                }
            }
        }
        
        vector<vector<int>> res = grid;

        for(int i = 0; i < grid.size(); ++i){
            for(int j = 0; j < grid[i].size(); ++j){
                if(flag1[i][j] == true){
                    res[i][j] = color;
                }
            }
        }
        return res;
    }
};
