class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        if(grid[r0][c0]==color)
            return grid;
        

        int origin = grid[r0][c0];
        DFS(grid, r0, c0, origin);
        
        for(int i=0; i<grid.size(); i++)
            for(int j=0; j<grid[0].size(); j++){
                if(grid[i][j]==0)
                    grid[i][j] = color;

                else if(grid[i][j]<0)
                    grid[i][j] = -grid[i][j];
                }
        return grid;
    }
    
    void DFS(vector<vector<int>>& grid, int x, int y, int& origin) {
// cout<<x<<y<<endl;
        if(x==0 or x==grid.size()-1 or y==0 or y==grid[0].size()-1)
            grid[x][y] = 0;
        else if(good(grid[x-1][y], origin) and good(grid[x+1][y], origin) and good(grid[x][y-1], origin) and good(grid[x][y+1], origin)){
            grid[x][y] = -origin;
        }
        else
            grid[x][y] = 0;
        
// cout<<"new color : "<<grid[x][y]<<endl;
        
// cout<<(y<grid[0].size()-1 and grid[x][y+1]==origin)<<endl;

        if(x and grid[x-1][y]==origin)
            DFS(grid, x-1, y, origin);
        if(x<grid.size()-1 and grid[x+1][y]==origin)
            DFS(grid, x+1, y, origin);
        if(y and grid[x][y-1]==origin)
            DFS(grid, x, y-1, origin);
        if(y<grid[0].size()-1 and grid[x][y+1]==origin)
            DFS(grid, x, y+1, origin);
    }
    
    bool good(int& c1, int& c2){
        return c1==0 or c1==c2 or c1+c2==0;
    }
};
