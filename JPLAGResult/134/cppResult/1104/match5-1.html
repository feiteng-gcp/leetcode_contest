class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> todo = {{r0, c0}};
        int color0 = grid[r0][c0];
        vector<vector<int>> rv = grid;
        int dx[] = {-1,1,0,0};
        int dy[] = {0,0,-1,1};
        while( !todo.empty() ) {
            vector<vector<int>> ntodo;
            for( auto v: todo ) {
                int r = v[0], c = v[1];
                grid[r][c] = -1;
                if( r == 0 || c == 0 ) rv[r][c] = color;
                else if( r + 1 ==  grid.size() || c + 1 == grid[0].size() ) rv[r][c] = color;
                else {

                    bool bo = false;
                    for( int i = 0; i < 4; ++i ) {
                        int nr = r + dx[i];
                        int nc = c + dy[i];
                        if( grid[nr][nc] != -1 && grid[nr][nc] != color0 ) {
                            bo = true;
                            break;
                        }
                    }
                    if( bo ) rv[r][c] = color;

                }
                for( int i = 0; i < 4; ++i ) {
                    int nr = r + dx[i];
                    int nc = c + dy[i];
                    if( nr < 0 || nc < 0 ) continue;
                    if( nr >= grid.size() || nc >= grid[0].size() ) continue;
                    if( grid[nr][nc] != color0 ) continue;
                    ntodo.push_back( {nr, nc} );
                }
            }
            swap( todo, ntodo );
        }
        return rv;
    }
};
