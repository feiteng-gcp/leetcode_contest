typedef pair<int, int> ii;

const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& a, int x, int y, int c) {
        int n = a.size(), m = a[0].size();
        vector<ii> Q;
        Q.push_back({x, y});
        vector<vector<bool>> visit(n, vector<bool>(m));
        visit[x][y] = true;
        for (int k = 0; k < Q.size(); ++k) {

            int x = Q[k].first, y = Q[k].second;
            for (int i = 0; i < 4; ++i) {
                int nx = x + d[i][0];
                int ny = y + d[i][1];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m || a[nx][ny] != a[x][y]) continue;
                if (visit[nx][ny]) continue;
                Q.push_back({nx, ny});
                visit[nx][ny] = true;
            }
        }
        vector<vector<int>> ret = a;
        for (auto& it : Q) {
            int x = it.first, y = it.second;
            bool found = false;
            for (int i = 0; i < 4; ++i) {
                int nx = x + d[i][0];
                int ny = y + d[i][1];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m || a[nx][ny] != a[x][y]) found = true;
            }
            if (found) ret[x][y] = c;
        }
        return ret;
    }
};
