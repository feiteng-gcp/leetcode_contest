class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int tmp = grid[r0][c0];
        helper(grid, r0, c0, grid[r0][c0]);
        for (int i = 0; i < grid.size(); ++i)
            for (int j = 0; j < grid[0].size(); ++j){
                if (grid[i][j] == -2) grid[i][j] = color;
                if (grid[i][j] == -1) grid[i][j] = tmp;
            }
        return grid;
    }
    void helper(vector<vector<int>>& grid, int r, int c, int color) {
        if (r < 0 || r >= grid.size() || c < 0 || c >= grid[0].size() || grid[r][c] < 0 || (grid[r][c] > 0 && grid[r][c]!= color))return;
        if ((r > 0 && ((grid[r - 1][c] == color) || (grid[r - 1][c] == -1) || (grid[r - 1][c] == -2))) && (r < grid.size() - 1 && ((grid[r + 1][c] == color) || (grid[r + 1][c] == -2) || (grid[r + 1][c] == -1))) && (c > 0 && ((grid[r][c - 1] == color) || (grid[r][c - 1] == -2) || (grid[r][c - 1] == -1)))  && (c < grid[0].size() - 1 && ((grid[r][c + 1] == color || grid[r][c + 1] == -2 || grid[r][c + 1] == -1)))) grid[r][c] = -1;
        else grid[r][c] = -2; 
        helper(grid, r, c-1, color);
        helper(grid, r, c+1, color);
        helper(grid, r-1, c, color);
        helper(grid, r+1, c, color);
    }
};
