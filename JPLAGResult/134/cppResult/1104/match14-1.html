class Solution {
public:
    struct Node{
        int x, y;
    };
    int dx[4]={ 0, 0,-1, 1};
    int dy[4]={-1, 1, 0, 0};
    int vis[107][107];
    
    queue<Node> Q;  
    
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        memset(vis, 0, sizeof(vis));
        int n=grid.size();
        int m=grid[0].size();
        
         
        vis[r0][c0]=1;
        
        Node q;
        q.x=r0;q.y=c0;
        
        Q.push(q);
        
        while(!Q.empty()) {

            Node u=Q.front();Q.pop();
            // cout<<u.x<<u.y<<endl;
            for(int i=0; i<4; i++) {
                int Nx=u.x+dx[i];
                int Ny=u.y+dy[i];
                //cout<<Nx<<' '<<Ny<<endl;
                if (Nx<0||Nx>=n||Ny<0||Ny>=m) continue;
                if (grid[Nx][Ny]!=grid[u.x][u.y]) continue;
                if(vis[Nx][Ny]) continue;
                vis[Nx][Ny]=1;

                q.x=Nx;q.y=Ny;
                Q.push(q);
            }
        }
        for(int i=0; i<n; i++) {
            for(int j=0; j<m; j++) {
                // cout<<vis[i][j]<<' ';
                // if (j==m-1) cout<<endl;
                if (!vis[i][j]) continue;

                int ok=0;
                for(int k=0; k<4; k++) {
                    int Nx=i+dx[k];
                    int Ny=j+dy[k];
                    if (Nx<0||Nx>=n||Ny<0||Ny>=m) {ok=1;break;}
                    if (!vis[Nx][Ny]) {ok=1;break;}
                }
                if (ok) grid[i][j]=color;
            }
        }
        return grid;
    }
};
