class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int m = grid.size(), n = grid[0].size(), cc = grid[r0][c0];
        queue<pair<int, int>> q;
        q.push({r0, c0});
        grid[r0][c0] = -1;
        vector<vector<int>> dirs{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        while(!q.empty()){
            auto p = q.front();
            q.pop();
            int border = 0;
            for(int i = 0; i < 4; i++){
                int x = dirs[i][0] + p.first, y = dirs[i][1] + p.second;
                if(x >= 0 && x < m && y >= 0 && y < n){
                    if(grid[x][y] == cc){
                        q.push({x, y});
                        grid[x][y] = -1;
                        border++;
                    }
                    else if(grid[x][y] == -1 || grid[x][y] == -2){
                        border++;
                    }
                }
            }
            if(border == 4) grid[p.first][p.second] = -2;
        }
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(grid[i][j] == -1){
                    grid[i][j] = color;
                }
                else if(grid[i][j] == -2){
                    grid[i][j] = cc;
                }
            }
        }
        return grid;
    }
};
