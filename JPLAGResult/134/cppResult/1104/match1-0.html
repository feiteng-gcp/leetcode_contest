class Solution {
    vector<vector<int>> dir = {{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
    void dfs(vector<vector<int>>& grid, int x, int y, int color, int orig, vector<vector<int>>& visited, vector<vector<int>>& res) {
        visited[x][y] = 1;
        res[x][y] = grid[x][y];
        bool boarder = false;
        for (int i = 0; i < 4; i++) {
            int nx = x + dir[i][0];
            int ny = y + dir[i][1];
            if (nx >= 0 && nx < grid.size() && ny >= 0 && ny < grid[0].size()) {
                if (grid[nx][ny] != orig) boarder = true;
                if (grid[nx][ny] == orig && visited[nx][ny] == 0) {
                    dfs(grid, nx, ny, color, orig, visited, res);
                }
            }
        }
        if (x == 0 || x == grid.size()-1 || y == 0 || y == grid[0].size()-1 || boarder) {
            //cout << "coloring " << x << " " << y << endl;
            res[x][y] = color;
        }
    }
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> visited(grid.size(), vector<int>(grid[0].size(), 0));
        vector<vector<int>> res(grid.size(), vector<int>(grid[0].size(), 0));
        dfs(grid, r0, c0, color, grid[r0][c0], visited, res);
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[0].size(); j++) {
                if (res[i][j] == 0) {
                    res[i][j] = grid[i][j];
                }
            }
        }
        return res;
    }
};
