class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        m = grid.size(), n = grid[0].size();
        vector<vector<bool>> visited(m, vector<bool>(n));
        vector<vector<bool>> border(m, vector<bool>(n));
        c = grid[r0][c0];
        dfs(grid, visited, border, r0, c0);
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++)
                if (border[i][j]) grid[i][j] = color;

        return grid;
    }
    void dfs(vector<vector<int>>& grid, vector<vector<bool>>& visited, vector<vector<bool>>& border, int si, int sj) {
        if (si < 0 || si >= m || sj < 0 || sj >= n || visited[si][sj]) return;
        visited[si][sj] = true;
        for (int i = 0; i < 4; i++) {
            int ti = si + dir[i], tj = sj + dir[i + 1];
            if (ti < 0 || ti >= m || tj < 0 || tj >= n || grid[ti][tj] != c) {
                border[si][sj] = true;
            }
            else if (!visited[ti][tj]) {
                dfs(grid, visited, border, ti, tj);
            }
        }
    }
    int m, n;
    int c;
    int dir[5] = {0, -1, 0, 1, 0};
};
