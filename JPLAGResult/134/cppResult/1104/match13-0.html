int dir[][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int R = grid.size();
        int C = grid[0].size();
        int origin = grid[r0][c0];
        
        queue<int> qx;
        queue<int> qy;
        qx.push(r0);
        qy.push(c0);
        grid[r0][c0] = -1;
        while (!qx.empty()) {
            int x = qx.front(); qx.pop();
            int y = qy.front(); qy.pop();
            
            for (int i = 0; i < 4; ++i) {
                int nx = x + dir[i][0];
                int ny = y + dir[i][1];
                if (nx < 0 || ny < 0 || nx >= R || ny >= C) continue;
                if (grid[nx][ny] != origin) continue;
                grid[nx][ny] = -1;
                qx.push(nx);
                qy.push(ny);
            }
        }


        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {
                // check 4 dir
                if (grid[i][j] == -1) {
                    bool bord = false;
                    for (int k = 0; k < 4; ++k) {
                        int nx = dir[k][0] + i;
                        int ny = dir[k][1] + j;
                        if (nx < 0 || ny < 0 || nx >= R || ny >= C) {
                            bord = true;
                            break;
                        }
                        if (grid[nx][ny] > 0) bord = true;
                    }
                    // cout << i << " " << j << " " << bord << endl;
                    if (bord) grid[i][j] = -2;
                }
            }
        }

        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {
                if (grid[i][j] == -1) {
                    grid[i][j] = origin;
                }
                if (grid[i][j] == -2) {
                    grid[i][j] = color;
                }
            }
        }
        return grid;
    }
};
