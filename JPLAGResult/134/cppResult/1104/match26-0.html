class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {        
        int init_color = grid[r0][c0];
        if(init_color == color)
            return grid;
        
        int row = grid.size();
        int col = grid[0].size();
        
        queue<pair<int, int>> bfs;
        unordered_set<int> visited;
        bfs.push(pair<int, int>(r0,c0));
        
        pair<int, int> loc, tmp;
        vector<int> direction = {1,0,-1,0,1};
        while(!bfs.empty()){
            loc = bfs.front();
            bfs.pop();
            
            if(isBorder(grid, loc.first, loc.second)){
                grid[loc.first][loc.second] = -1*color;
            }
            
            visited.insert(loc.first * col + loc.second);
            
            for(int i=0; i<=3; i++){
                tmp = loc;
                tmp.first = loc.first + direction[i];
                tmp.second = loc.second + direction[i+1];
                if(tmp.first < row && tmp.second < col && tmp.first>=0 && tmp.second>=0
                   && grid[tmp.first][tmp.second] == init_color
                   && visited.count(tmp.first * col + tmp.second)==0){
                    bfs.push(tmp);
                }                
            }            
        }
        
        for(int i=0; i<row; i++)
            for(int j=0; j<col; j++)
                if(grid[i][j] == -1*color) grid[i][j] = color;
        
        return grid;
    }
    
    bool isBorder(vector<vector<int>>& grid, int i, int j){
        if(i == grid.size()-1 || j == grid[0].size()-1 || i==0 || j==0)
            return true;
        
        if(grid[i+1][j] != grid[i][j] && grid[i+1][j] > 0)
            return true;
        
        if(grid[i-1][j] != grid[i][j] && grid[i-1][j] > 0)
            return true;
        
        if(grid[i][j+1] != grid[i][j] && grid[i][j+1] > 0)
            return true;
        
        if(grid[i][j-1] != grid[i][j] && grid[i][j-1] > 0)
            return true;
        
        return false;
        
    }
};
