const int dir[][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int theColor = grid[r0][c0];
        vector<pair<int, int>> border;
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<bool>> visited(m, vector<bool>(n));
        stack<pair<int, int>> st;
        st.push({r0, c0});
        visited[r0][c0] = true;
        while (!st.empty()) {

            pair<int, int> cur = st.top();
            st.pop();
            bool isborder = false;
            for (int i = 0; i < 4; ++i) {
                int x = cur.first + dir[i][0];
                int y = cur.second + dir[i][1];
                if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] != theColor) {
                    isborder = true;
                    continue;
                }
                if (visited[x][y])
                    continue;
                visited[x][y] = true;

                st.push({x, y});
            }
            if (isborder)
                border.push_back(cur);
        }
        for (auto p : border) {
            grid[p.first][p.second] = color;
        }
        return grid;
    }
};
