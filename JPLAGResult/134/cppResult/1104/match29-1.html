class Solution {
public:
    struct A{
        int x;
        int y;
    };
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        if(grid.size()==0){
            return grid;
        }
        queue<A> q;
        vector<vector<int>> newgrid=grid;

        vector<vector<int>> dis={{1,0},{-1,0},{0,1},{0,-1}};
        q.push({r0,c0});
        int colorx = grid[r0][c0];
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<bool>> visited(n,vector<bool>(m,false));
        visited[r0][c0] = true;
        while (!q.empty()) {
            A a = q.front();
            q.pop();
            for(int i = 0;i<dis.size();i++){
                int newx = a.x+dis[i][0];
                int newy = a.y+dis[i][1];
                if(newx<0||newy<0||newx>=n || newy>=m || grid[newx][newy]!=colorx){
                    newgrid[a.x][a.y] = color;
                    continue;
                }
                if(visited[newx][newy]) continue;
                visited[newx][newy] = true;
                q.push({newx,newy});
            }
        }
        return newgrid;
    }
};
