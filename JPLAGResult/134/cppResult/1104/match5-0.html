class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int dir[] = {1,0,-1,0,1};
        queue<pair<int, int>> bfs;
        bfs.push(make_pair(r0,c0));
        int ori = grid[r0][c0];
        set<pair<int, int>> visited;
        visited.insert(make_pair(r0,c0));
        vector<vector<int>> g = grid;
        while(bfs.size() > 0) {
            auto p = bfs.front();
            bfs.pop();
            bool isb = false;
            for(int i = 0; i < 4; i++) {
                int xx = p.first + dir[i];
                int yy = p.second + dir[i+1];
                if(xx < 0 || xx >= grid.size() || yy < 0 || yy >= grid[0].size() || grid[xx][yy] != ori) {
                    isb = true;
                    break;
                }
            }
            if(isb) {
                g[p.first][p.second] = color;
            }
             for(int i = 0; i < 4; i++) {
                int xx = p.first + dir[i];
                int yy = p.second + dir[i+1];
                if(xx >= 0 && xx < grid.size() && yy >= 0 && yy < grid[0].size() && grid[xx][yy] == ori) {
                    if(visited.count(make_pair(xx,yy)) == 0) {
                        visited.insert(make_pair(xx,yy));
                        bfs.push(make_pair(xx,yy));
                    }
                }
            }
        }
        return g;
    }
};
