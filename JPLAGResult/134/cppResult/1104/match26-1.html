class Solution {
public:
    void f(vector<vector<int>>& g, int r, int c, int color, int old){
        if(r < 0 || c < 0 || r >= g.size() || c >= g[0].size())
            return;
        if(g[r][c] != old)
            return;
        g[r][c] = color;
        f(g, r-1, c, color, old);
        f(g, r+1, c, color, old);
        f(g, r, c-1, color, old);
        f(g, r, c+1, color, old);
    }

    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> vv = grid;
        if(color == vv[r0][c0])
            return vv;
        int old = vv[r0][c0];
        f(vv, r0, c0, color, vv[r0][c0]);

        for(int i = 1; i < vv.size()-1; i++)
            for(int j = 1; j < vv[0].size()-1; j++)
                if(vv[i][j] != grid[i][j] && grid[i-1][j] == old && grid[i+1][j] == old
                                             && grid[i][j-1] == old && grid[i][j+1] == old)
                    vv[i][j] = old;
        return vv;
    }
};
