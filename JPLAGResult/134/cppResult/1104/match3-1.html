class Solution {
    const int dx[4] = {0, 1, 0, -1};
    const int dy[4] = {1, 0, -1, 0};
    
    int n, m;
    
    void dfs(vector<vector<int>>& grid, vector<vector<bool>>& visited, int i, int j, int color, int org) {

        visited[i][j] = true;
        for (int dir = 0; dir < 4; dir++) {
            int ni = i + dx[dir];
            int nj = j + dy[dir];
            
            if (ni < 0 || ni >= n || nj < 0 || nj >= m) {
                grid[i][j] = -org;
            }
            else if (grid[ni][nj] > 0 && grid[ni][nj] != org) {
                grid[i][j] = -org;
            }
            else if (grid[ni][nj] == org) {
                if (!visited[ni][nj]) dfs(grid, visited, ni, nj, color, org);
            }
            
        }    
    }
    
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        n = grid.size();
        m = grid[0].size();
        
        vector<vector<bool>> visited(n, vector<bool>(m));
        dfs(grid, visited, r0, c0, color, grid[r0][c0]);
        

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] < 0) grid[i][j] = color;
            }
        }
        return grid;
    }
};
