class Solution {
public:
    bool isBorder(int x, int y, vector<vector<int>>& grid, vector<vector<int>>& visited, int n, int m) {
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];

            if (nx < 0 || nx >= n || ny < 0 || ny >= m || grid[nx][ny] != grid[x][y]) {
                return true;
            }
        }
        return false;
    }
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int n = grid.size();
        int m = grid[0].size();
        queue<pair<int, int>> q;
        vector<vector<int>> visited(n, vector<int>(m, 0));
        vector<vector<int>> ans(grid);
        q.emplace(r0, c0);
        visited[r0][c0] = 1;
        originc = grid[r0][c0];
        while (!q.empty()) {
            auto p = q.front(); q.pop();

            int x = p.first, y = p.second;
            if (isBorder(x, y, grid, visited, n, m)) {
                ans[x][y] = color;
            }
            for (int i = 0; i < 4; ++i) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m || 
                    visited[nx][ny] || grid[nx][ny] != originc) {
                    continue;
                }
                visited[nx][ny] = 1;
                q.emplace(nx, ny);
            }
        }
        return ans;
    }
private:
    vector<int> dx = {0, 0, -1, 1};
    vector<int> dy = {-1, 1, 0, 0};
    int originc;
};
