class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int R = grid.size(), C = grid[0].size();
        int c = grid[r0][c0];
        typedef pair<int, int> P;
        int dx[] = {1, -1, 0, 0};
        int dy[] = {0, 0, 1, -1};
        queue<P> que;
        que.push({r0, c0});
        bitset<50> visited[50], border[50];
        visited[r0][c0] = 1;

        while(que.size()){
            auto p = que.front(); que.pop();
            int i = p.first, j = p.second;
            for(int k=0; k<4; k++){
                int x = i+dx[k], y = j+dy[k];
                if(0<=x && x<R && 0<=y && y<C && grid[x][y] == c){
                    if(!visited[x][y]){
                        visited[x][y] = 1;
                        que.push({x, y});
                    }
                }else{
                    border[i][j] = 1;
                }
            }
        }

        for(int i=0; i<R; i++) for(int j=0; j<C; j++) if(border[i][j]) grid[i][j] = color;
        return grid;
    }
};
