class Solution {
public:
  
  struct UnionFind{
    int n;
    vector<int> r,p;
    UnionFind(){}
    UnionFind(int sz):n(sz),r(sz,1),p(sz,0){iota(p.begin(),p.end(),0);}
    int find(int x){
      return (x==p[x]?x:p[x]=find(p[x]));
    }
    bool same(int x,int y){
      return find(x)==find(y);
    }
    void unite(int x,int y){
      x=find(x);y=find(y);
      if(x==y) return;
      if(r[x]<r[y]) swap(x,y);
      r[x]+=r[y];
      p[y]=x;
    }
    int size(int x){
      return r[find(x)];

    }
  };

  //INSERT ABOVE HERE
  vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
    int h=grid.size();
    int w=grid[0].size();
    auto idx=[&](int y,int x){return y*w+x;};

    vector<vector<int>> res(grid);
    UnionFind uf(h*w);
    for(int i=0;i<h;i++){
      for(int j=0;j<w;j++){
        if(i+1<h&&grid[i][j]==grid[i+1][j])
          uf.unite(idx(i,j),idx(i+1,j));
        if(j+1<w&&grid[i][j]==grid[i][j+1])
          uf.unite(idx(i,j),idx(i,j+1));
      }
    }
    auto in=[&](int y,int x){return 0<=y&&y<h&&0<=x&&x<w;};
    int dy[]={0,0,1,-1};
    int dx[]={1,-1,0,0};
    for(int i=0;i<h;i++){
      for(int j=0;j<w;j++){
        if(!uf.same(idx(r0,c0),idx(i,j))) continue;
        int flg=0;
        for(int k=0;k<4;k++){
          int ny=i+dy[k],nx=j+dx[k];
          if(!in(ny,nx)){
            flg=1;
            continue;
          }
          if(grid[i][j]!=grid[ny][nx]) flg=1;
        }
        if(flg) res[i][j]=color;
      }
    }
    return res;
  }
};
