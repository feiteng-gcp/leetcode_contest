const int D = 4;
static int dx[D] = {0, 0, 1, -1};
static int dy[D] = {1, -1, 0, 0};
inline int get_color(vector<vector<int>>& grid, int h, int w, int y, int x) {
    if (y < 0 || y >= h || x < 0 || x >= w) {
        return -123;
    }
    return grid[y][x];
}

void search1(vector<vector<int>>& grid, int h, int w, int y, int x, int old_color) {
    if (get_color(grid, h, w, y, x) != old_color) {
        return ;
    }
    grid[y][x] = -1;
    for (int i = 0; i < D; i++) {
        search1(grid, h, w, y + dy[i], x + dx[i], old_color);
    }
}
void search2(vector<vector<int>>& grid, int h, int w, int y, int x) {
    if (get_color(grid, h, w, y, x) != -1) {
        return ;
    }
    bool is_boundary = false;
    for (int i = 0; i < D; i++) {
        int c = get_color(grid, h, w, y + dy[i], x + dx[i]);
        if (c != -1 && c != -2 && c != -3) {
            is_boundary = true;
            break;
        }
    }
    if (is_boundary) {
        grid[y][x] = -2;
    } else {
        grid[y][x] = -3;
    }
    for (int i = 0; i < D; i++) {
        search2(grid, h, w, y + dy[i], x + dx[i]);
    }
}
class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int h = grid.size();
        int w = grid[0].size();
        int old_color = grid[r0][c0];
        search1(grid, h, w, r0, c0, old_color);
        search2(grid, h, w, r0, c0);

        for (int r = 0; r < h; r++) {
            for (int c = 0; c < w; c++) {
                if (grid[r][c] == -3) {
                    grid[r][c] = old_color;
                }
                if (grid[r][c] == -2) {
                    grid[r][c] = color;
                }
            }
        }
        return grid;
    }
};
