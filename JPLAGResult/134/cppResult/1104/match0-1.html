class Solution {
public:
void dfs(vector<vector<int>>& grid, vector<vector<bool>>& vis, vector<vector<bool>>& b, int x, int y) {
	int r = grid.size(), c = grid[0].size();
	if (x < 0 || x == r || y < 0 || y == c)
		return;

	if (vis[x][y])
		return;
	vis[x][y] = 1;

	int t = 0;
	if (x > 0 && grid[x][y] == grid[x - 1][y]) {
		++t;
		dfs(grid, vis, b, x - 1, y);
	}

	if (y > 0 && grid[x][y] == grid[x][y - 1]) {
		++t;
		dfs(grid, vis, b, x, y - 1);
	}

	if (x < r - 1 && grid[x][y] == grid[x + 1][y]) {
		++t;
		dfs(grid, vis, b, x + 1, y);
	}

	if (y < c - 1 && grid[x][y] == grid[x][y + 1]) {
		++t;
		dfs(grid, vis, b, x, y + 1);
	}

	if (t < 4)
		b[x][y] = 1;
}

vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
	int r = grid.size(), c = grid[0].size();
	vector<vector<bool>> vis(r, vector<bool>(c, 0));
	vector<vector<bool>> b(r, vector<bool>(c, 0));
	dfs(grid, vis, b, r0, c0);

	for(int i=0;i<r;i++)
		for (int j = 0; j < c; j++) {
			if (b[i][j])
				grid[i][j] = color;
		}

	return grid;
}
};
