class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int dx[4] = {0, 1, 0, -1};
        int dy[4] = {1, 0, -1, 0};
        int m = grid.size();
        int n = grid[0].size();
        int origin = grid[r0][c0];
        vector<vector<bool>> visit(m, vector<bool>(n, false));
        queue<pair<int, int>> q;
        q.push(make_pair(r0, c0));
        visit[r0][c0] = true;
        grid[r0][c0] = color;
        
        while(!q.empty()) {
            auto cur = q.front();
            q.pop();
            
            for(int k=0; k<4; k++) {
                int x = cur.first + dx[k];
                int y = cur.second + dy[k];
        // cout <<"sssss " << x <<" "<<y<<endl;
                if(x>=0 && x<m && y>=0 && y<n && !visit[x][y] && grid[x][y]==origin) {
                    grid[x][y] = color;
                    visit[x][y] = true;
                    q.push(make_pair(x, y));
                }
            }
        }
        // cout <<" head"<<endl;
        for(int i=1; i<m-1; i++) {
            for(int j=1; j<n-1; j++) {
                if(visit[i][j] && visit[i-1][j] && visit[i+1][j] && visit[i][j-1] && visit[i][j+1]) {
                    grid[i][j] = origin;
                }
            }
        }
        
        return grid;
    }
