class Solution {
public:
    void dfs(int i, int j, int m, int n, int c, vector<vector<int>>& grid, vector<vector<int>>& visited) {
        if (i < 0 || i >= m || j < 0 || j >= n || grid[i][j] != c || visited[i][j] != 0) {
            return;
        }
        visited[i][j] = 1;
        if (i - 1 < 0 || i + 1 >= m || j - 1 < 0 || j + 1 >= n || grid[i + 1][j] != c || grid[i][j + 1] != c || grid[i - 1][j] != c || grid[i][j - 1] != c) {
            visited[i][j] = 2;
        }
        
        dfs(i + 1, j, m, n, c, grid, visited);
        dfs(i, j + 1, m, n, c, grid, visited);
        dfs(i - 1, j, m, n, c, grid, visited);
        dfs(i, j - 1, m, n, c, grid, visited);

    }
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<int>> visited(m, vector<int>(n, 0));
        dfs(r0, c0, m, n, grid[r0][c0], grid, visited);

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (visited[i][j] == 2) {
                    grid[i][j] = color;
                }
            }
        }
        return grid;
    }
};
