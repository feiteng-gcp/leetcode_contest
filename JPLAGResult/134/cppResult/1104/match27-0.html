class Solution {
public:
    vector<int> dirx = {-1,0,1,0};
    vector<int> diry = {0,1,0,-1};
    
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int n = grid.size();
        int m = grid[0].size();
        queue<pair<int,int>> qu;
        vector<vector<int>> res(n,vector<int> (m,0));
        vector<vector<int>> visited(n,vector<int> (m,0));
        qu.push({r0,c0});
        int maincolor = grid[r0][c0];
        while(!qu.empty()){
            int x = qu.front().first;
            int y = qu.front().second;
            qu.pop();
            visited[x][y] = 1;
            for(int k = 0;k < 4;k++){
                int xx = x + dirx[k];
                int yy = y + diry[k];
                if(xx < 0 || xx >= n || yy < 0 || yy >= m || grid[xx][yy] != maincolor || visited[xx][yy] == 1)continue;
                qu.push({xx,yy});
            }
        }
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                res[i][j] = grid[i][j];
            }
        }
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                if(visited[i][j]){
                    if(i == 0 || i == n - 1 || j == 0 || j == m - 1)res[i][j] = color;
                    else {
                        int f = 0;
                        for(int k = 0;k < 4;k++){
                            int xx = i + dirx[k];
                            int yy = j + diry[k];
                            if(xx < 0 || xx >= n || yy < 0 || yy >= m)continue;
                            if(grid[xx][yy] != grid[i][j])f = 1;
                        }
                        if(f)res[i][j] = color;
                    }
                }
            }
        }
        return res;
        
    }
};
