class Solution {
    bool colo(vector<vector<int>>& grid, int x, int y, int color) {
        if (x < 0 || y < 0 || x >= grid.size() || y >= grid[0].size()) return false;
        return grid[x][y] == color;
    }
    void dfs(vector<vector<int>>& res, vector<vector<int>>& grid, int x, int y, int color, vector<vector<bool>> visited, int c) {
        if (x < 0 || y < 0 || x >= grid.size() || y >= grid[0].size() || visited[x][y]) return;
        if (grid[x][y] != color) return;
        visited[x][y] = true;
        if (!(colo(grid, x+1, y, color) && colo(grid, x, y+1, color) && colo(grid, x-1, y, color) && colo(grid, x, y-1, color))) {
            res[x][y] = c;
        }
        dfs(res, grid, x+1, y, color, visited, c);
        dfs(res, grid, x-1, y, color, visited, c);
        dfs(res, grid, x, y+1, color, visited, c);
        dfs(res, grid, x, y-1, color, visited, c);
    }
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> res(grid);
        vector<vector<bool>> visited(grid.size(), vector<bool>(grid[0].size(), false));
        dfs(res, grid, r0, c0, grid[r0][c0], visited, color);
        return res;
    }
};
