const int DIRS = 4;
const int DR[] = {-1, 0, +1, 0};
const int DC[] = {0, +1, 0, -1};

const int RC_MAX = 55;

int R, C;
vector<vector<int>> grid;
bool visited[RC_MAX][RC_MAX];
bool is_border[RC_MAX][RC_MAX];

bool valid(int r, int c) {
    return r >= 0 && r < R && c >= 0 && c < C;
}

void dfs(int r, int c, int color) {
    if (visited[r][c])
        return;

    visited[r][c] = true;

    for (int d = 0; d < DIRS; d++) {
        int nr = r + DR[d];
        int nc = c + DC[d];

        if (!valid(nr, nc) || grid[nr][nc] != color) {
            is_border[r][c] = true;
            continue;
        }

        dfs(nr, nc, color);
    }
}

class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& _grid, int r0, int c0, int color) {
        grid = _grid;
        R = grid.size();
        C = grid[0].size();
        memset(visited, false, sizeof(visited));
        memset(is_border, false, sizeof(is_border));
        dfs(r0, c0, grid[r0][c0]);

        for (int r = 0; r < R; r++)
            for (int c = 0; c < C; c++)
                if (is_border[r][c])
                    grid[r][c] = color;

        return grid;
    }
};
