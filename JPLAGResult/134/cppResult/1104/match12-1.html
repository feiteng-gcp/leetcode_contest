class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int m = grid.size();
        int n = grid[0].size();
        queue<pair<int, int>> q;
        int c = grid[r0][c0];
        grid[r0][c0]=-c;

        static int offsets[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        if(r0==0||r0==m-1||c0==0||c0==n-1)grid[r0][c0]=-color;
        else{

            for(int j = 0; j<4; j++){
                            int nei_r = r0 +offsets[j][0];
                            int nei_c = c0 + offsets[j][1];
                            if(nei_r>=0&&nei_r<m&&nei_c>=0&&nei_c<n&&abs(grid[nei_r][nei_c])!=c&&grid[nei_r][nei_c]!=-color){
                                grid[r0][c0]=-color;
                                break;
                            }
                        }
        }
        q.push(make_pair(r0, c0));
        
        while(!q.empty()){
            auto p = q.front();
            q.pop();
            for(int i = 0; i<4; i++){
                int new_r = p.first+offsets[i][0];
                int new_c = p.second+offsets[i][1];
                if(new_r>=0&&new_r<m&&new_c>=0&&new_c<n&&grid[new_r][new_c]==c){
                    grid[new_r][new_c]=-c;
                    //check if boarder
                    if(new_r==0||new_r==m-1||new_c==0||new_c==n-1){
                        grid[new_r][new_c]=-color;
                    }
                    else{
                        for(int j = 0; j<4; j++){
                            int nei_r = new_r +offsets[j][0];
                            int nei_c = new_c + offsets[j][1];
                            if(nei_r>=0&&nei_r<m&&nei_c>=0&&nei_c<n&&abs(grid[nei_r][nei_c])!=c&&grid[nei_r][nei_c]!=-color){
                                grid[new_r][new_c]=-color;
                                break;
                            }
                        }
                    }
                    q.push(make_pair(new_r, new_c));

                }
            }
        }
        for(int i = 0; i<m; i++){
            for(int j = 0; j<n; j++){
                grid[i][j]=abs(grid[i][j]);
            }
        }
        return grid;
    }
};
