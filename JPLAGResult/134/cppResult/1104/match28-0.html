class Solution {
public:
    vector<int> dx={1,-1,0,0};
    vector<int> dy={0,0,1,-1};
    bool isBorder(int r, int c, vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        for(int k=0; k<4; k++) {
            int rr=r+dx[k];
            int cc=c+dy[k];
             if(0<=rr&&rr<n&&0<=cc&&cc<m) {
                 if(grid[r][c]!=grid[rr][cc])
                     return true;
             } else {
                 return true;
             }
        }
        return false;
    }
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int n=grid.size();
        int m=grid[0].size();
        vector<vector<int>> ret = vector<vector<int>>(n,vector<int>(m,0));
        for(int i=0; i<n; i++)for(int j=0;j<m;j++)ret[i][j]=grid[i][j];
        vector<vector<bool>> visited=vector<vector<bool>>(n,vector<bool>(m,false));
        visited[r0][c0]=true;
        queue<pair<int,int>> q;
        q.push(make_pair(r0,c0));
        while(!q.empty()) {
            int r,c;
            tie(r,c)=q.front();
	    visited[r][c]=true;
            q.pop();        
            if(isBorder(r,c,grid)) {
                ret[r][c]=color;
            }
            for(int k=0; k<4; k++) {
                int rr=r+dx[k];
                int cc=c+dy[k];
                if(0<=rr&&rr<n&&0<=cc&&cc<m&&grid[r][c]==grid[rr][cc]&&!visited[rr][cc]) { // 
                    q.push(make_pair(rr,cc));
                }
            }
        }
        return ret;
    }
};
