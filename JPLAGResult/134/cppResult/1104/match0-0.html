class Solution {
public:
    
    int row;
    int col;
    
    void dfs(vector<vector<int>>& grid, vector<vector<int>>& ans, vector<vector<int>>& seen, int r0, int c0, int color) {
        int orig = grid[r0][c0];
        
        if (seen[r0][c0])
            return;
        
        seen[r0][c0] = 1;
        
        int count = 0;
        if (r0 > 0 && grid[r0-1][c0] == orig) {
            count++;
            dfs(grid, ans, seen, r0-1, c0, color);
        }
        
        if (r0 + 1 < row && grid[r0+1][c0] == orig) {
            count++;
            dfs(grid, ans, seen, r0+1, c0, color);
        }
        
        if (c0 > 0 && grid[r0][c0-1] == orig) {
            count++;
            dfs(grid, ans, seen, r0, c0-1, color);
        }
        
        if (c0 + 1 < col && grid[r0][c0+1] == orig) {
            count++;
            dfs(grid, ans, seen, r0, c0+1, color);
        }
        
        if (count != 4)
            ans[r0][c0] = color;
    }
    
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> ans = grid;
      
        row = grid.size();
        if (!row)
            return ans;
        
        col = grid[0].size();
        
        vector<vector<int>> seen(row, vector<int>(col, 0));
 #if 1         
        dfs(grid, ans, seen, r0, c0, color);
 #endif       
        return ans;
    }
};
