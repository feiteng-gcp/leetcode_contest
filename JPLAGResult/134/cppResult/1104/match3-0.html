class Solution {
    vector<vector<int>> nb = {{0, -1}, {0, 1}, {1, 0}, {-1, 0}};
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> used(grid.size(), vector<int>(grid[0].size(), 0));
        helper(grid, used, r0, c0);
        for(int i = 0; i < grid.size(); i++) {
            for(int j = 0; j < grid[0].size(); j++) {
                if(used[i][j] == 2)
                    grid[i][j] = color;
            }
        }
        
        return grid;
    }
    // 1 used, 2 need color
    void helper(vector<vector<int>>& grid, vector<vector<int>> &used, int x, int y) {
        if(used[x][y] > 0)  return;
        
        int cur = grid[x][y];
        used[x][y] = 1;
        for(int i = 0; i < nb.size(); i++) {
            int nextX = x + nb[i][0];
            int nextY = y + nb[i][1];
            
            if(nextX < 0 || nextX >= grid.size() || nextY < 0 || nextY >= grid[0].size() || grid[nextX][nextY] != cur) {
                used[x][y] = 2;
                continue;
            }
            
            helper(grid, used, nextX, nextY);
        }
    }
};
