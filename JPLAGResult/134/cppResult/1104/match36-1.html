class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int m = grid.size(), n = grid[0].size(), c = grid[r0][c0];
        vector<vector<bool>> visited(m, vector<bool>(n));
        vector<vector<int>> result = grid;
        queue<pair<int, int>> q;
        q.push({r0, c0});

        visited[r0][c0] = true;
        vector<vector<int>> dir = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        while (!q.empty()) {

            int curx = q.front().first, cury = q.front().second, count = 0;
            q.pop();
            for (int i = 0; i < 4; i++) {
                int nextx = curx + dir[i][0], nexty = cury + dir[i][1];
                if (nextx < 0 || nextx >= m || nexty < 0 || nexty >= n || grid[nextx][nexty] != c) continue;
                count++;
                if (visited[nextx][nexty]) continue;
                visited[nextx][nexty] = true;
                q.push({nextx, nexty});
            }
            if (count != 4) result[curx][cury] = color;
        }
        return result;
    }
};
