class Solution {
public:
    void seed(vector<vector<int>>& grid, int r0, int c0, int color) {
        int t = grid[r0][c0];
        grid[r0][c0] = color;
        if(c0+1 < grid[r0].size() && grid[r0][c0+1] == t)
            seed(grid, r0, c0+1, color);
        if(c0-1 >=0 && grid[r0][c0-1] == t)
            seed(grid, r0, c0-1, color);
        if(r0+1 < grid.size() && grid[r0+1][c0] == t)
            seed(grid, r0+1, c0, color);
        if(r0-1 >= 0 && grid[r0-1][c0] == t)
            seed(grid, r0-1, c0, color);
    }

    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> ret = grid;
        seed(grid, r0, c0, color + 10000);
        for(int i=0;i<grid.size();i++)
            for(int j=0;j<grid[i].size();j++) {
                if(grid[i][j] == color + 10000 &&
                   (j==0 || i==0 || j==grid[i].size()-1 || i==grid.size()-1 ||
                    grid[i-1][j] != grid[i][j] || grid[i+1][j] != grid[i][j] || grid[i][j+1] != grid[i][j] || grid[i][j-1] != grid[i][j]))
                    ret[i][j]=grid[i][j] - 10000;
            }
        
        return ret;
    }
};
