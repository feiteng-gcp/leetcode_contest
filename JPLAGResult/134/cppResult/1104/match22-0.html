class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        const int di[4] = {-1, 1, 0, 0};
        const int dj[4] = {0, 0, -1, 1};
        
        int org_color = grid[r0][c0];
        
        if (org_color == color) return grid;
        
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<bool>> visited(m, vector<bool>(n, false));
        queue<vector<int>> q;
        vector<vector<int>> borders;
        
        q.push({r0, c0});
        visited[r0][c0] = true;
        
        while (!q.empty())
        {
            vector<int> pos = q.front();
            q.pop();
            bool is_border = false;
            
            for (int d = 0; d < 4; d++)
            {
                int ni = pos[0] + di[d];
                int nj = pos[1] + dj[d];
                
                if (ni < 0 || ni >= m || nj < 0 || nj >= n || grid[ni][nj] != org_color)
                {
                    is_border = true;
                    continue;
                }
                
                if (!visited[ni][nj])
                {
                    visited[ni][nj] = true;
                    q.push({ni, nj});
                }
            }
            
            if (is_border)
                borders.push_back(pos);
        }
        
        for (auto border : borders)
        {
            grid[border[0]][border[1]] = color;
        }
        
        return grid;
    }
};
