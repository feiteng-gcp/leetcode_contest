class Solution {
public:
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        int m=grid.size(),n=grid[0].size();
        vector<vector<int>> used(m,vector<int>(n,false));
        int tc=grid[r0][c0];
        queue<pair<int,int>> qq;

        used[r0][c0]=true;
        qq.push(make_pair(r0,c0));
        int dd[][2]={{1,0},{-1,0},{0,1},{0,-1}};
        while(!qq.empty()){
            auto p=qq.front();
            qq.pop();
            for(int i=0;i<4;i++){
                int x=p.first+dd[i][0];
                int y=p.second+dd[i][1];
                if(x<0||x>=m||y<0||y>=n) continue;
                if(used[x][y]||grid[x][y]!=tc) continue;
                used[x][y]=true;
                qq.push(make_pair(x,y));
            }
        }
        
        auto r=grid;
        for(int i=0;i<m;i++)
            for(int j=0;j<n;j++){
                if(!used[i][j]) continue;
                bool flag=false;
                for(int k=0;k<4;k++){
                    int x=i+dd[k][0];
                    int y=j+dd[k][1];
                    if(x<0||x>=m||y<0||y>=n){
                        flag=true;
                        break;
                    }
                    if(grid[x][y]!=tc)
                    {
                        flag=true;
                        break;
                    }
                }
                if(flag){
                    r[i][j]=color;
                }
            }
        return r;    
    }
};
