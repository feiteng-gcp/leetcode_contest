class Solution {
public:
    struct node{
        int x;
        int y;
    };
    
    vector<vector<int>> colorBorder(vector<vector<int>>& grid, int r0, int c0, int color) {
        vector<vector<int>> vvt=grid;
        int r=grid.size(),c=grid[0].size(), clr=grid[r0][c0];
        node nd;
        nd.x=r0;nd.y=c0;
        queue<node> qe;
        qe.push(nd);
        set<int> st;
        st.insert(r0*c+c0+1);
        
        int dir[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
        
        while(!qe.empty()){
            node nd=qe.front();
            int x=nd.x,y=nd.y,x2,y2,num=0;
            for(int i=0;i<4;i++){
                x2=x+dir[i][0];
                y2=y+dir[i][1];
                if(x2>=0 && x2<r && y2>=0 && y2<c && grid[x2][y2]==clr){
                    num++;
                    if(!st.count(x2*c+y2+1)){
                        node nd2;
                        nd2.x=x2;nd2.y=y2;
                        qe.push(nd2);
                        st.insert(x2*c+y2+1);
                    }
                    
                }
            }
            if(num!=4)
                vvt[x][y]=color;
            qe.pop();
        }
        return vvt;
    }
};
