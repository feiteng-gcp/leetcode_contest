class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        if (grid.length == 0 || grid[0].length == 0 || r0 < 0 || r0 >= grid.length || c0 < 0 || c0 >= grid[0].length) {
            return grid;
        }
        boolean[][] marked = new boolean[grid.length][grid[0].length];
        colorBorder(grid, r0, c0, grid[r0][c0], marked, new boolean[grid.length][grid[0].length]);
        for (int r = 0; r < grid.length; r++) {
            for (int c = 0; c < grid[0].length; c++) {
                if (marked[r][c]) {
                    grid[r][c] = color;
                }
            }
        }
        return grid;
    }
    
    private static int[][] DELTAS = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    
    private void colorBorder(int[][] grid, int r, int c, int color, boolean[][] marked, boolean[][] visited) {
        for (int[] delta : DELTAS) {
            int nr = r + delta[0];
            int nc = c + delta[1];
            if (nr < 0 || nr >= grid.length || nc < 0 || nc >= grid[0].length || grid[nr][nc] != color) {
                marked[r][c] = true;
                continue;
            }
            if (!visited[nr][nc]) {
                visited[nr][nc] = true;
                colorBorder(grid, nr, nc, color, marked, visited);
            }
        }
    }
}
