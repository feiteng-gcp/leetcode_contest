class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int n = grid.length;
        int m = grid[0].length;
        boolean[][] cover = new boolean[n][m];
        
boolean[][] visited = new boolean[n][m];
        dfs(cover, visited, grid, r0, c0, grid[r0][c0], color);
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < m; j++)
            {
                if(cover[i][j])
                {
                    grid[i][j] = color;
                }
            }
        }
        return grid;
    }
    
    public boolean sameColor(int[][] grid, int i, int j, int c)
    {
        if(i < 0 || j < 0 || i >= grid.length || j >= grid[0].length)
        {
            return false;
        }
        return grid[i][j] == c;
    }
    
    public boolean inBoard(int[][] grid, int i, int j, int c)
    {
        boolean b = sameColor(grid, i, j + 1, c) && sameColor(grid, i, j - 1, c) && sameColor(grid, i + 1, j, c)
            && sameColor(grid, i - 1, j, c);
        b = !b;
        b = b || i == 0 || j == 0 || i == grid.length - 1 || j == grid[0].length - 1;
        return b;
    }
    
    public void dfs(boolean[][] cover, boolean[][] visited, int[][] grid, int i, int j, int c, int color)
    {
          if(i < 0 || j < 0 || i >= grid.length || j >= grid[0].length)
        {
            return;
        }
        if(visited[i][j])
        {
            return;
        }
        visited[i][j] = true;
        if(grid[i][j] != c)
        {
            return;
        }
        cover[i][j] = inBoard(grid, i, j, c);
        dfs(cover, visited, grid, i + 1, j, c, color);
        dfs(cover, visited, grid, i - 1, j, c, color);
        dfs(cover, visited, grid, i, j + 1, c, color);
        dfs(cover, visited, grid, i, j - 1, c, color);
    }
}
