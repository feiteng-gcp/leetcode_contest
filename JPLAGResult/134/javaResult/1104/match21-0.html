	class Solution {
	    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
	    	int n = grid.length;
	    	int m = grid[0].length;
	    	Queue<int[]> q = new ArrayDeque<>();
	    	int[] dr = { 1, 0, -1, 0 };
			int[] dc = { 0, 1, 0, -1 };
			q.add(new int[]{r0, c0});
			boolean[][] that = new boolean[n][m];
			that[r0][c0] = true;
			while(!q.isEmpty()){
				int[] cur = q.poll();
				int r = cur[0], c = cur[1];
				for(int k = 0;k < 4;k++){
					int nr = r + dr[k], nc = c + dc[k];
					if(nr >= 0 && nr < n && nc >= 0 && nc < m && grid[nr][nc] == grid[r][c]){
						if(!that[nr][nc]){
							that[nr][nc] = true;
							q.add(new int[]{nr, nc});
						}
					}
				}
			}
			int[][] ret = new int[n][m];
			for(int i = 0;i < n;i++){
				for(int j = 0;j < m;j++){
					
					if(!that[i][j]){
						ret[i][j] = grid[i][j];
					}else{
						int r = i, c = j;
						int s = 0;
						for(int k = 0;k < 4;k++){
							int nr = r + dr[k], nc = c + dc[k];
							if(nr >= 0 && nr < n && nc >= 0 && nc < m && grid[nr][nc] == grid[r][c]){
								s++;
							}
						}
						if(s != 4){
							ret[i][j] = color;
						}else{
							ret[i][j] = grid[i][j];
						}
					}
				}
			}
			return ret;
	    }
	}
