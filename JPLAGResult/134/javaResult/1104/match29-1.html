class Solution {
    private static final int[][] neighbors = new int[][] {{0, +1}, {+1, 0}, {0, -1}, {-1, 0}};

    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        
int R = grid.length;
        int C = grid[0].length;
        boolean[][] visited = new boolean[R][C];
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[] {r0, c0});
        List<int[]> cc = new ArrayList<>();
        while (!queue.isEmpty()) {
            int[] position = queue.remove();
            int r = position[0];
            int c = position[1];
            if (r < 0 || r >= R || c < 0 || c >= C || grid[r][c] != grid[r0][c0] || visited[r][c]) {
                continue;
            }
            visited[r][c] = true;
            cc.add(position);
            for (int[] neighbor : neighbors) {
                queue.add(new int[] {r + neighbor[0], c + neighbor[1]});
            }
        }
        List<int[]> border = cc.stream()
                .filter(position -> isBorder(grid, position[0], position[1]))
                .collect(Collectors.toList());
        border.forEach(position -> grid[position[0]][position[1]] = color);
        return grid;
    }

    private boolean isBorder(int[][] grid, int r, int c) {
        return r == 0 || r == grid.length - 1 || c == 0 || c == grid[0].length - 1
                || Arrays.stream(neighbors).anyMatch(neighbor -> grid[r][c] != grid[r + neighbor[0]][c + neighbor[1]]);
    }
}
