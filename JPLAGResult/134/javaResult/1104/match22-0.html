class Solution {
    int[][] nei = {{0,1},{0,-1},{1,0},{-1,0}};
    Set<int[]> paint = new HashSet<>();
    
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int r = grid.length;
        int c = grid[0].length;
        colorIt(grid, r0, c0, grid[r0][c0], new boolean[r][c]);
        
        for (int[] p : paint) {
            grid[p[0]][p[1]] = color;
        }
        return grid;
    }
    
    private void colorIt(int[][] grid, int r, int c, int oldC, boolean[][] visited) {
        if (visited[r][c]) { return; }
        visited[r][c] = true;
        int count = 0;
        for (int[] n : nei) {
            int nr = n[0] + r;
            int nc = n[1] + c;
            if (nr >= 0 && nr < grid.length && nc >= 0 && nc < grid[0].length) {
                if (grid[nr][nc] == oldC) {
                    count++;
                    colorIt(grid, nr, nc, oldC, visited);
                }
            }
        }
        if (count < 4) {
            paint.add(new int[]{r, c});
        }
    }
}
