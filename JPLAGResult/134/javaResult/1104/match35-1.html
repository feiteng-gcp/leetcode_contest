class Solution {
    int newcolor;
    int[] dir1 = {-1, 1, 0, 0};
    int[] dir2 = 
{0, 0, -1, 1};
    int[][] visited;
    int oldcolor;
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        visited = new int[grid.length][grid[0].length];
        newcolor = color;
        oldcolor = grid[r0][c0];
        dfs(grid, r0, c0);
        return grid;
    }
    
    private void dfs(
int[][] grid, int i, int j){
        if (visited[i][j] == 1)
            return;
        visited[i][j] = 1;
        /*
        //check real boundary
        if (i == 0 || j == 0 || i == grid.length-1 || j == grid.length-1){
            grid[i][j] = newcolor;
            return;
        }*/
        //check color boundary
        for (int k = 0; k < dir1.length; k++){
            int newi = i + dir1[k];
            int newj = j + dir2[k];
            if (newi < 0 || newj < 0 || newi >= grid.length || newj >= grid[0].length || (visited[newi][newj] == 0 && grid[newi][newj] != oldcolor)){
                grid[i][j] = newcolor;
            }
            else{
                dfs(grid, newi, newj);
            }
        }
    }
    
}
