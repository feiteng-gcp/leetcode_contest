class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        final int n = grid.length, m = grid[0].length;
        if (color == grid[r0][c0])
            return grid;
        boolean[][] visited = new boolean[n][m];
        dfs(grid, r0, c0, grid[r0][c0], color, visited);
        return grid;
    }
    public static final int[][] DIRS = {{0,1}, {0,-1}, {1, 0}, {-1, 0}};
    private void dfs(int[][] grid, int i, int j, int orgColor, int color, boolean[][] visited) {
        visited[i][j] = true;
        final int n = grid.length, m = grid[0].length;
        if (i == n-1 || j == m-1 || i == 0 || j == 0)
            grid[i][j] = color;
        for (int[] dir : DIRS) {
            int ni = i + dir[0], nj = j + dir[1];
            if (ni < 0 || ni == n || nj < 0 || nj == m || visited[ni][nj])
                continue;
            if (grid[ni][nj] != orgColor) {
                grid[i][j] = color;
                continue;
            }
            dfs(grid, ni, nj, orgColor, color, visited);
        }
    }
}
