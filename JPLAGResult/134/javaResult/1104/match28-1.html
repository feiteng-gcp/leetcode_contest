class Solution {
    static int[][] DIRS = new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        if (grid == null || grid.length == 0 || grid[0].length == 0) return grid;
        if (grid[r0][c0] == color) return grid;
        int m = grid.length;
        int n = grid[0].length;

        int oldColor = grid[r0][c0];
        boolean[][] visited = new boolean[m][n];
        Queue<int[]> queue = new LinkedList<>();
        
queue.offer(new int[]{r0, c0});
        visited[r0][c0] = true;
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            for (int[] dir : DIRS) {
                int x = cur[0] + dir[0];
                int y = cur[1] + dir[1];
                if (x >= 0 && x < m && y >= 0 && y < n && (grid[x][y] == oldColor || visited[x][y])) {
                    if (visited[x][y]) continue;
                    visited[x][y] = true;
                    queue.offer(new int[]{x, y});
                } else {
                    grid[cur[0]][cur[1]] = color;
                }
            }
        }
        return grid;
    }
}
