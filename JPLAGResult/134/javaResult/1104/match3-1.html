class Solution {
    public int[][] dirs = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
    
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int[][] tmp = new int[grid.length][grid[0].length];
        for(int i = 0; i < tmp.length; ++i) tmp[i] = Arrays.copyOf(grid[i], grid[i].length);
        
helper(r0, c0, grid, new boolean[grid.length][grid[0].length], color, grid[r0][c0], tmp);
        return tmp;
    }
    
    public void helper(int i, int j, int[][] gg, boolean[][] vis, int color, int cc, int[][] tmp) {
        if(vis[i][j]) return;
        vis[i][j] = true;
        // boolean flag = true;
        if(i == 0 || j == 0 || i == gg.length - 1 || j == gg[0].length - 1) tmp[i][j] = color;
        for(int[] dir : dirs) {
            int r = i + dir[0], c = j + dir[1];
            if(r < 0 || c < 0 || r >= gg.length || c >= gg[0].length || vis[r][c]) continue;
            if(gg[r][c] != cc) {tmp[i][j] = color; continue;}
            // flag = false;
            helper(r, c, gg, vis, color, cc, tmp);
        }
        // if(!flag) gg[i][j] = color;
    }
}
