class Solution {
    
    
    List<int[]> li = new ArrayList<int[]>();
    
    boolean vis[][] = new boolean[51][51];
    
    void dfs(int[][] g,int r,int c,int ck){
        int l = g.length;
        int ll = g[0].length;
        if(r<0||c<0||r>=l||c>=ll) return;
        if(vis[r][c]) return;
        vis[r][c] = true;
        if(g[r][c]!=ck) return;
        
        boolean ok = false;
        if((r==0||c==0||r==l-1||c==ll-1)){
            ok = true;
    
        }
        
        if((g[r][c]==ck)&&((r+1<l&&g[r+1][c]!=ck)||(r-1>=0&&g[r-1][c]!=ck)||(c-1>=0&&g[r][c-1]!=ck)||(c+1<ll&&g[r][c+1]!=ck))){
            ok = true;
            
        }
        if(ok){
            li.add(new int[]{r,c});
        }
        
        
        dfs(g,r+1,c,ck);dfs(g,r-1,c,ck);dfs(g,r,c+1,ck);dfs(g,r,c-1,ck);
        
        
    }
    
    
    
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        dfs(grid,r0,c0,grid[r0][c0]);
        
        for(int[] p:li){
            grid[p[0]][p[1]] = color;
        }
        return grid;
        
    }
}
