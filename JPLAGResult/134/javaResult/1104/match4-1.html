class Solution {
    private int[][] ds = new int[][]{{-1, 0},{1, 0},{0, -1},{0, 1}};
    private int r, c;
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        // grid = new int[][]{{1,1,1},{1,1,1}, {1,1,1}}; r0 = 1; c0 = 1; color = 3;
        r = grid.length; c = grid[0].length;
        
dfs(new boolean[r][c], grid, r0, c0, grid[r0][c0], color);
        return grid;
    }
    private boolean dfs(boolean[][] visited, int[][] grid, int i, int j, int oc, int nc) {
        if (i < 0 || i == r || j < 0 || j == c) return true;
        if (visited[i][j]) return false;
        if (grid[i][j] != oc) return true;
        visited[i][j] = true;
        boolean isborder = false;
        for (int[] d : ds) {
            isborder |= dfs(visited, grid, i + d[0], j + d[1], oc, nc);
        }
        if (isborder) grid[i][j] = nc;
        return false;
    }
}
