class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        if(grid==null || grid.length == 0|| grid[0].length == 0){
            return grid;
        }
        boolean [][]visited = new boolean[grid.length][grid[0].length];
        dfs(grid,r0,c0,-1,grid[r0][c0],visited);
        
for(int i=0;i<grid.length;i++){
            for(int j=0;j<grid[0].length;j++){
                if(grid[i][j] == -1){
                    grid[i][j] = color;
                }
            }
        }
        return grid;
    }
    boolean dfs(int [][]grid,int i, int j, int color, int precolor,boolean [][]visited){
          if(i<0||i>=grid.length||j<0||j>=grid[0].length||(grid[i][j]!=color && grid[i][j]!=precolor)){
              return true;
          } 
          if(visited[i][j]){
              return false;
          } 
          visited[i][j] = true;
          
int [][]next = {{-1,0},{1,0},{0,-1},{0,1}};
        for(int k=0;k<4;k++){
            boolean ret = dfs(grid,i+next[k][0],j+next[k][1],color,precolor,visited);
            if(ret){
                grid[i][j] = color;
            }
        }
        
        return false;
              
    }
}
