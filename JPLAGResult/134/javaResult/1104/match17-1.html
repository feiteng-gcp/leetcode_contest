import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    
int[] dx = {0, 0, 1, -1};
    int[] dy = {1, -1, 0, 0};

    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        
boolean[][] visited = new boolean[grid.length][grid[0].length];
        dfs(grid, r0, c0, grid[r0][c0], visited);
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[i].length; j++) {
                if (visited[i][j]) {
                    boolean isBoundary = i == 0 || i == grid.length - 1 || j == 0 || j == grid[i].length - 1;

                    for (int k = 0; k < 4; k++) {
                        int x = i + dx[k];
                        int y = j + dy[k];
                        if (x >= 0 && x < grid.length && y >= 0 && y < grid[x].length && !visited[x][y]) {
                            isBoundary = true;
                        }
                    }
                    if (isBoundary) {
                        grid[i][j] = color;
                    }
                }
            }
        
}
        return grid;
    }

    void dfs(int[][] g, int i, int j, int c, boolean[][] visited) {
        visited[i][j] = true;
        
for (int k = 0; k < 4; k++) {
            int x = i + dx[k];
            int y = j + dy[k];
            if (x >= 0 && x < g.length && y >= 0 && y < g[x].length && g[x][y] == c && !visited[x][y]) {
                dfs(g, x, y, c, visited);
            }
        }
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

    }
}
