class Solution {
    int[] dx = {-1, 0, 0, 1};
    int[] dy = {0, -1, 1, 0};
    int R;
    int C;
    
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        if (grid[r0][c0] == color) {
            return grid;
        }
        
        R = grid.length;
        C = grid[0].length;
        // 0 unvisited
        // 1 visited
        // 2 border
        int[][] visited = new int[R][C];
        
        helper(grid, visited, r0, c0, grid[r0][c0]);
        
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (visited[i][j] == 2) {
                    grid[i][j] = color;
                }
            }
        }
        
        return grid;
    }
    
    private void helper(int[][] grid, int[][] visited, int r, int c, int oldColor) {
        // check if border
        if (isBorder(r, c, grid)) {
            visited[r][c] = 2;
        } else {
            visited[r][c] = 1;
        }
        // check neighbor 
        for (int i = 0; i < 4; i++) {
            int nr = r + dx[i];
            int nc = c + dy[i];
            if (nr >= 0 && nr < R && nc >= 0 && nc < C && grid[nr][nc] == oldColor && visited[nr][nc] == 0) {
                helper(grid, visited, nr, nc, oldColor);
            }
        }
    }
    
    private boolean isBorder(int r, int c, int[][] grid) {
        if (r == 0 || r == R - 1 || c == 0 || c == C - 1) {
            return true;
        }
        
        for (int i = 0; i < 4; i++) {
            int nr = r + dx[i];
            int nc = c + dy[i];
            if (grid[nr][nc] != grid[r][c]) {
                return true;
            }
        }
        
        return false;
    }
}
