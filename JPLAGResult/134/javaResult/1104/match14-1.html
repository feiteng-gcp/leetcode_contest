class Solution {
    static final int[][] dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int m = grid.length;
        int n = grid[0].length;
        boolean[][] vst = new boolean[m][n];

        List<int[]> list = new ArrayList<>();
        dfs(grid, r0, c0, grid[r0][c0], vst, list);

        for (int i = 0; i < list.size(); i++) {
            grid[list.get(i)[0]][list.get(i)[1]] = color;
        
}

        return grid;
    }

    private void dfs(int[][] grid, int x, int y, int color, boolean[][] vst, List<int[]> list) {
        int m = grid.length;
        int n = grid[0].length;

        if (!(x >= 0 && x < m && y >= 0 && y < n)) return;
        if (vst[x][y]) return;
        if (grid[x][y] != color) return
;

        vst[x][y] = true;

        int count = 0;
        for (int i = 0; i < dir.length; i++) {
            int r = x + dir[i][0];
            int c = y + dir[i][1];

            if (!(r >= 0 && r < m && c >= 0 && c < n)) {
                count++;
                continue;
            }
            if (grid[r][c] != color) {
                count++;
                continue;
            }
            dfs(grid, r, c, color, vst, list);
        
}

        if (count > 0) {
            list.add(new int[]{x, y});
        }
    }
}
