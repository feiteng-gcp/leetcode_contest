class Solution {
    public int[][] colorBorder(int[][] g, int r0, int c0, int color) {
        int n = g.length, m = g[0].length;
        Deque<Integer> q = new ArrayDeque<>();
        boolean[][] visited = new boolean[n][m];
        q.add(r0*m+c0);
        visited[r0][c0] = true;
        
int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
        List<Integer> b = new ArrayList<>();
        int g0 = g[r0][c0];
        
        while (!q.isEmpty()) {
            int cur = q.remove();
            int cx = cur / m;
            int cy = cur % m;
            
boolean isBorder = false;
            for (int[] dir : dirs) {
                int x = cx + dir[0];
                int y = cy + dir[1];
                if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == g0) {
                    if (!visited[x][y]) {
                        visited[x][y]= true;
                        q.add(x * m + y);                        
                    }                    
                }else {
                    isBorder = true;
                }
            }
            if (isBorder) b.add(cur);
        }
        
        
for (int v : b) {
            g[v / m][v % m] = color;
        }
        return g;
    }
}
