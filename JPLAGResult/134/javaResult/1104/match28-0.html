class Solution {
    
    int[][] ds = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    int[] add(int[] v1, int[] v2) {
        return new int[]{v1[0] + v2[0], v1[1] + v2[1]};
    }
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        Set<int[]> border = new HashSet<>();
        int rows = grid.length;
        int cols = grid[0].length;
        Deque<int[]> dq = new LinkedList<>();
        boolean[][] visited = new boolean[rows][cols];
        dq.offerLast(new int[]{r0, c0});
        visited[r0][c0] = true;
        while (!dq.isEmpty()) {
            int[] cp = dq.pollFirst();
            for (int[] d : ds) {
                int[] np = add(d, cp);
                if (np[0] < 0 || np[0] == rows || np[1] < 0 || np[1] == cols || grid[np[0]][np[1]] != grid[r0][c0]) {
                    border.add(cp);
                    continue;
                }
                if (!visited[np[0]][np[1]]) {
                    dq.offerLast(np);
                    visited[np[0]][np[1]] = true;
                }
            }
        }
        for (int[] p : border) {
            grid[p[0]][p[1]] = color;
        }
        return grid;
    }
}
