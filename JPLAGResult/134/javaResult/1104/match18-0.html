class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        if (grid[r0][c0] == color) return grid;
        
        boolean[][] seen = new boolean[grid.length][grid[0].length];
        
        color(grid, r0, c0, grid[r0][c0], seen);
        
        for (int i = 0; i < grid.length; ++i) {
            for (int j = 0; j < grid[i].length; ++j) {
                if (grid[i][j] == -1) {
                    grid[i][j] = color;
                }
            }
        }
        
        return grid;
    }
    
    public void color(int[][] grid, int r, int c, int o, boolean[][] seen) {
        if (0 > r || r >= grid.length || 0 > c || c >= grid[r].length || seen[r][c] || grid[r][c] != o) return;
        
        if (0 == r || r == grid.length - 1 || 0 == c || c == grid[r].length - 1 ||
            (grid[r - 1][c] != o && grid[r - 1][c] != -1) ||
            (grid[r + 1][c] != o && grid[r + 1][c] != -1) ||
            (grid[r][c - 1] != o && grid[r][c - 1] != -1) ||
            (grid[r][c + 1] != o && grid[r][c + 1] != -1))
        {
            // System.out.print("hi ");
            grid[r][c] = -1;
        }
        
        seen[r][c] = true;
        
        color(grid, r - 1, c, o, seen);
        color(grid, r + 1, c, o, seen);
        color(grid, r, c - 1, o, seen);
        color(grid, r, c + 1, o, seen);
    }
}
