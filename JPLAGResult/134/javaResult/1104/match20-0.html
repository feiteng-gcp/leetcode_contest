class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int m = grid.length, n = grid[0].length;
        boolean[][] visited = new boolean[m][n];
        boolean[][] boarder = new boolean[m][n];
        
        dfs(grid, boarder, visited, r0, c0, grid[r0][c0]);
        
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (boarder[i][j]) {
                    grid[i][j] = color;
                }
            }
        }
        return grid;
    }
    
    int[] dx = new int[]{0, -1, 0, 1};
    int[] dy = new int[]{1, 0, -1, 0};
    void dfs(int[][] grid, boolean[][] boarder, boolean[][] visited, int x, int y, int color) {
        if (visited[x][y]) return;
        
        visited[x][y] = true;
        boarder[x][y] = isBoarder(grid, x, y, color);
        // System.out.println(boarder[x][y] + " " + x + " "+ y);
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i], ny = y + dy[i];
            if (nx >= 0 && ny >=0 && nx < grid.length && ny < grid[0].length && !visited[nx][ny] && grid[nx][ny] == color) {
                dfs(grid, boarder, visited, nx, ny, color);
            }
        }
    }
    
    boolean isBoarder(int[][] grid, int x, int y, int color) {
        boolean boarder = false;
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i], ny = y + dy[i];
            // System.out.println(isBorderHelper(grid, nx, ny, color));
            boarder = boarder || isBorderHelper(grid, nx, ny, color);
        }
        return boarder;
    }
    
    boolean isBorderHelper(int[][] grid, int x, int y, int color) {
        if (x < 0 || y < 0 || x == grid.length || y == grid[0].length ||grid[x][y] != color) return true;
        return false;
    }
}
