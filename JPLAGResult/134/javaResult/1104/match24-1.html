class Solution {
    private static final int[][] dirs = new int[][] {{0,1},{1,0},{0,-1},{-1,0}};
    public int[][] colorBorder(int[][] grid, int r, int c, int color) {
        if (grid[r][c] == color) return grid;
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[] {r, c});
        int co = grid[r][c];
        
while (!q.isEmpty()) {
            int[] p = q.poll();
            boolean shouldColor = false;
            for (int[] d : dirs) {
                int x = p[0] + d[0];
                int y = p[1] + d[1];
                if (x >= 0 && x < grid.length && y >= 0 && y < grid[0].length) {
                    if (grid[x][y] == co) q.offer(new int[] {x, y});
                    else if (grid[x][y] == 0 || grid[x][y] == -1) continue;
                    else shouldColor = true;
                } else shouldColor = true;
            }
            if (shouldColor) grid[p[0]][p[1]] = 0;
            else grid[p[0]][p[1]] = -1;
        }
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0) grid[i][j] = color;
                if (grid[i][j] == -1) grid[i][j] = co;
            }
        }
        return grid;
    }
}
