class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int m = grid.length;
        int n = grid[0].length;
        boolean[][] mark = new boolean[m][n];
        boolean[][] visited = new boolean[m][n];
        int c = grid[r0][c0];
        dfs(grid, mark, visited, r0, c0, c);
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (mark[i][j]) {
                    grid[i][j] = color;
                }
            }
        }
        return grid;
    }
    
    private void dfs(int[][] grid, boolean[][] mark, boolean[][] visited, int x, int y, int c) {
        if (visited[x][y]) return;
        visited[x][y] = true;
        if (x == 0 || x == grid.length - 1 || y == 0 || y == grid[0].length - 1) {
            mark[x][y] = true;
        }
        
        if (x > 0) {
            if (grid[x - 1][y] != c)
                mark[x][y] = true;
            else {
                dfs(grid, mark, visited, x - 1, y, c);
            }
        }
        if (y > 0) {
            if (grid[x][y - 1] != c)
                mark[x][y] = true;
            else {
                dfs(grid, mark, visited, x, y - 1, c);
            }
        }
        if (x < grid.length - 1) {
            if (grid[x + 1][y] != c)
                mark[x][y] = true;
            else {
                dfs(grid, mark, visited, x + 1, y, c);
            }
        }
        if (y < grid[0].length - 1) {
            if (grid[x][y + 1] != c)
                mark[x][y] = true;
            else {
                dfs(grid, mark, visited, x, y + 1, c);
            }
        }
    }
}
