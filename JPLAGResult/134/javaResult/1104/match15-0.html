class Solution {
    int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        dfs(r0, c0, grid, grid[r0][c0], color, new boolean[grid.length][grid[0].length]);
        return grid;
    }
    
    
    public boolean dfs(int r, int c, int[][] g, int num, int color, boolean[][] visited){
        if(r<0||c<0||r>g.length-1||c>g[0].length-1) return true;
        if(visited[r][c]) return false;
        if(g[r][c]!=num) return true;
        visited[r][c] = true;
        for(int[] dir:dirs){
            if(dfs(r+dir[0], c+dir[1], g, num, color, visited)){
                g[r][c] = color;
            }
        }
        if(r==0||c==0||r==g.length-1||c==g[0].length-1){
            g[r][c] = color;
        }
        return false;
        
    }
    
}
