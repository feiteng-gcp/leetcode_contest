class Solution {
    int[][] directions = new int[][]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        Set<String> visited = new HashSet<>();
        Queue<int[]> path = new LinkedList<>();
        path.add(new int[]{r0, c0});
        int r = grid.length, c = grid[0].length, oC = grid[r0][c0];
        visited.add(r0 + "-" + c0);
        while (!path.isEmpty()){
            int[] one = path.poll();
            int board = 0;
            for(int[] dir : directions) {
                int nextR = one[0] + dir[0], nextC = one[1] + dir[1];
                if(nextR >= 0 && nextR < r && nextC >= 0 && nextC < c) {
                    if(visited.contains(nextR + "-" + nextC)) {
                        board++;
                    } else if (grid[nextR][nextC] == oC) {
                        board++;
                        path.add(new int[]{nextR, nextC});
                        visited.add(nextR + "-" + nextC);
                    }
                }
            }
            if(board < 4) 
                grid[one[0]][one[1]] = color;
        }
        return grid;
    }
}
