class Solution {
    int[][] moves = new int[][]{{0,1},{0,-1},{1,0},{-1,0}};
    int m;
    int n;
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int origin = grid[r0][c0];
        m=grid.length;
        n=grid[0].length;
        boolean[][] visited = new boolean[m][n];
        helper(grid,r0,c0,origin,color,visited);
        return grid;
        
    }
    public void helper(int[][] grid, int i, int j, int origin, int color,boolean[][] visited){
        if(i<0||i>=m||j<0||j>=n) return;
        visited[i][j] = true;
        for(int[] move : moves){
            int x = i+move[0];
            int y = j+move[1];
            if(x>=0&&x<m&&y>=0&&y<n&&visited[x][y]) continue;
            if(x<0||x>=m||y<0||y>=n||grid[x][y]!=origin){
                grid[i][j] = color;
                break;
            }
        }
        for(int[] move : moves){
            int x = i+move[0];
            int y = j+move[1];
            if(x<0||x>=m||y<0||y>=n||grid[x][y]!=origin||visited[x][y]) continue;
            helper(grid,x,y,origin,color,visited);
        }
    }
}
