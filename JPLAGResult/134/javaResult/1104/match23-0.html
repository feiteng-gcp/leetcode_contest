class Solution {
    private int[][] grid;

    private int[][] result;

    private int R;

    private int C;

    private boolean[][] visited;

    private int finalColor;

    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        this.grid = grid;
        this.R = grid.length;
        this.C = grid[0].length;

        visited = new boolean[R][C];
        finalColor = color;

        result = new int[R][C];
        for (int r=0; r<R;r++) {
            for (int c=0;c<C;c++) {
                result[r][c] = grid[r][c];
            }
        }
        dfs(r0,c0, grid[r0][c0]);
        return result;
    }

    private void dfs(int r, int c, int targetColor) {
        if (r >= R || r < 0 || c >= C || c < 0)
            return;
        if (grid[r][c] != targetColor)
            return;
        if (visited[r][c])
            return;

        visited[r][c] = true;
        dfs(r, c-1, targetColor);
        dfs(r, c+1, targetColor);
        dfs(r-1, c, targetColor);
        dfs(r+1, c, targetColor);

        if (isBorder(r,c, targetColor))
            result[r][c] = finalColor;
    }

    private boolean isBorder(int r, int c, int targetColor) {
        if (r == R - 1 || c == C - 1 || r == 0 || c == 0)
            return true;
        int left = grid[r][c-1];
        int right = grid[r][c+1];
        int top = grid[r-1][c];
        int bottom = grid[r+1][c];

        if (left != targetColor || right != targetColor || top != targetColor || bottom != targetColor)
            return true;
        return false;
    }
}
