class Solution {
    
    
int[][] isVisited;
    int[][] gridCopy;
    
    int x;
    int y;
    
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        
         x = grid.length;
         y = grid[0].length;
        
        gridCopy = new int[x][y];
        
        
isVisited = new int[grid.length][grid[0].length];
        
        for(int i =0 ; i < x;i++)
        {
            for (int j = 0; j < y; j++)
            {
                isVisited[i][j] = 0;
                gridCopy[i][j] = grid[i][j];
            }
        }
        
        
        helper(grid, r0,c0,color,grid[r0][c0]);
        return gridCopy;
    }
    
    void helper(int[][] grid, int r0, int c0, int color, int colorInit)
    {
        
        if (isVisited[r0][c0] == 1 || grid[r0][c0] != colorInit)
            return;
        
        isVisited[r0][c0]= 1;
        if (isBorder(grid,r0,c0))
            gridCopy[r0][c0] = color;
        if(r0 > 0)
            helper(grid,r0-1,c0,color,colorInit);
        if (r0 < x-1)
            helper(grid,r0+1,c0,color,colorInit);
        if (c0 > 0)
            helper(grid,r0,c0-1,color,colorInit);
        if (c0 < y-1)
            helper(grid,r0,c0+1,color,colorInit);
        
    
}
    
    boolean isBorder(int[][] grid, int r0, int c0)
    {
        if(r0 == 0 || r0 == x-1 || c0 == 0 || c0 == y-1)
            return true;
        if (grid[r0][c0] != grid[r0-1][c0])
            return true;
        if (grid[r0][c0] != grid[r0+1][c0])
            return true;
        if (grid[r0][c0] != grid[r0][c0-1])
            return true;
        if (grid[r0][c0] != grid[r0][c0+1])
            return true;
        return false;
        
    }
}
