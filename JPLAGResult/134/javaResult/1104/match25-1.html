class Solution {
    public int[][] colorBorder(int[][] grid, int r0, int c0, int color) {
        int oldColor = grid[r0][c0];
        int newColor = 99999;
        Queue<Integer> queue = new LinkedList<>();
        queue.offer((r0 << 16) + c0);
        grid[r0][c0] = newColor;
        while (!queue.isEmpty()) {
            int temp = queue.poll();
            int x = temp >>> 16;
            int y = temp & 0xffff;
            
if (x > 0 && grid[x - 1][y] == oldColor) {
                grid[x - 1][y] = newColor;
                queue.offer(((x - 1) << 16) + y);
            }
            if (x < grid.length - 1 && grid[x + 1][y] == oldColor) {
                grid[x + 1][y] = newColor;
                queue.offer(((x + 1) << 16) + y);
            }
            if (y > 0 && grid[x][y - 1] == oldColor) {
                grid[x][y - 1] = newColor;
                queue.offer((x << 16) + y - 1);
            }
            if (y < grid[0].length - 1 && grid[x][y + 1] == oldColor) {
                grid[x][y + 1] = newColor;
                queue.offer((x << 16) + y + 1);
            }
        }
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == newColor) {
                    if (i == 0 || j == 0 || i == grid.length - 1 || j == grid[0].length - 1 ||
                        grid[i-1][j] < newColor || grid[i+1][j] < newColor ||
                        grid[i][j-1] < newColor || grid[i][j+1] < newColor) {
                        grid[i][j]++;
                    }
                }
            }
        }
        for (int i = 0; i < grid.length; i++) {
            for (
int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == newColor) {
                    grid[i][j] = oldColor;
                } else if (grid[i][j] > newColor) {
                    grid[i][j] = color;
                }
            }
        }
        return grid;
    }
}
