class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] arr = {a, b, c};
        Arrays.sort(arr);
        int diff = arr[2] - arr[0];
        int min = 0;
        if (diff == 2) {
            min = 0;
        } else if (arr[1] - arr[0] == 1 || arr[2] - arr[1] == 1 || arr[1] - arr[0] == 2 || arr[2] - arr[1] == 2) {
            min = 1;
        } else {
            min = 2;
        }
        int max = 0;
        if (diff == 2) {
            max = 0;
        } else {
            max = diff - 2;
        }
        return new int[]{min, max};
    }
}
