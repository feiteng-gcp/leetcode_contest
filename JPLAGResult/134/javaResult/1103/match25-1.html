class Solution {
    // 1, _, 3, _, 5
    
    public int[] numMovesStones(int a, int b, int c) {
        int[] arr = new int[]{a, b, c};
        Arrays.sort(arr);
        
        int max = arr[2] - arr[0] - 2;
        int min = 0;
        
        if (arr[1] == arr[0] + 2 || arr[2] == arr[1] + 2) {
            min = 1;
        } else {
            if (arr[1] > arr[0] + 2) min++;
            if (arr[2] > arr[1] + 2) min++;
        
}
        
        return new int[]{min, max};
    }
}
