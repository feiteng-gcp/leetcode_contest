class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] q = new int[]{a, b, c};
        Arrays.sort(q);
        a = q[0];
        b = q[1];
        c = q[2];
        
        if (a + 1 == b && b + 1 == c) { 
            return new int[]{0, 0};
        } else if (b - a <= 2 || c - b <= 2) {
            return new int[]{1, c - a - 2};
        } else {
            return new int[]{2, c - a - 2};
        }
    }
}
