class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] array = {a, b, c};
        Arrays.sort(array);
        a = array[0]; b = array[1]; c = array[2];
        int max = (c - b - 1) + (b - a - 1);
        int min = 0;
        if (c - b == 2 || b - a == 2)
            min = 1;
        else
            min = ((c - b - 1) == 0 ? 0 : 1) + ((b - a - 1) == 0 ? 0 : 1);
        return new int[]{min, max};

    }
}
