class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] tmp = new int[]{a, b, c},res = new int[2];
        Arrays.sort(tmp);
        res[1] = tmp[2]-tmp[0]-2;
        
if(tmp[1]-tmp[0]==1 && tmp[2]-tmp[1]==1){
            res[0] = 0;
        }else if(tmp[1]-tmp[0]<=2 || tmp[2]-tmp[1]<=2 ){
            res[0] = 1;
        }else{
            res[0] = 2;
        }
        return res;
    }
}
