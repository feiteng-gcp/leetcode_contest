class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] nums = new int[]{a,b,c};
        int[] ans = new int[2];
        int min = 0, max = 0;
        Arrays.sort(nums);
        if(nums[2] - nums[0] == 2) return ans;
        max = nums[2] - nums[0] - 2;
        if(nums[2] - nums[1] <= 2 || nums[1] - nums[0] <= 2)
            min = 1;
        else
            min = 2;
        ans[0] = min;
        ans[1] = max;
        return ans;
    }
}
