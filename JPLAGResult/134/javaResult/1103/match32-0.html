class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] path = new int[]{a, b, c};
        Arrays.sort(path);
        int min = 0;
        int max = 0;
        for(int i = 1; i < path.length; i++) {
            if(path[i] != path[i - 1] + 1) {
                min += 1;
                max += path[i] - path[i - 1] - 1;
            }
        }
        if(path[1] == path[0] + 2 || path[2] == path[1] + 2) {
            min = 1;
        }
        return new int[]{min, max};
    }
}
