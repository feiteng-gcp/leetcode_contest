class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] nums = {a,b,c};
        Arrays.sort(nums);
        int max = 0, min=0;
        max = nums[2]-nums[1]+nums[1]-nums[0]-2;
        int gap1 = nums[1]-nums[0], gap2 = nums[2]-nums[1];
        if(gap1==1&&gap2==1) min = 0;
        else if(gap1==2 || gap2==2) min = 1;
        else if(gap1==1 || gap2==1) min = 1;
        else min = 2;
        return new int[]{min, max};
    }
}
