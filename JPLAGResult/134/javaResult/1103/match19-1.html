class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] x = new int[3];
        x[0] = a;
        x[1] = b;
        x[2] = c;
        Arrays.sort(x);
        
if (x[1] - x[0] == 1 && x[2] - x[1] == 1) {
            return new int[] {0,0};
        } else if (x[1] - x[0] == 1 || x[2] - x[1] == 1 || x[1] - x[0] == 2 || x[2] - x[1] == 2) {
            return new int[] {1, x[1] - x[0] + x[2] - x[1] - 2};
        } else {
            return new int[] {2, x[1] - x[0] + x[2] - x[1] - 2};
        }
    }
}
