class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] idx = new int[]{a,b,c};
        Arrays.sort(idx);
        int[] res = new int[2];
        if(idx[0]+1==idx[1]&&idx[1]+1==idx[2]) return res;
        res[1] = idx[1]-idx[0]-1+idx[2]-idx[1]-1;
        if(idx[0]+1==idx[1]||idx[1]+1==idx[2]||idx[0]+2==idx[1]||idx[1]+2==idx[2]) res[0] = 1;
   
        else res[0] = 2;
        return res;
    }
}
