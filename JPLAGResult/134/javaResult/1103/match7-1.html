	class S
olution {
	    public int[] numMovesStones(int a, int b, int c) {
	        int[] d = {a, b, c};
	        Arrays.sort(d);
	        int max = d[2]-d[0]-2;
	        int min = 2;
	        if(d[2] - d[0] == 2){
	        	min = 0;
	        }else{
	        	if(d[1] - d[0] <= 2 || d[2] - d[1] <= 2){
	        		min = 1;
	        	}
	        }

	        return new int[]{min, max};
	    }
	}
