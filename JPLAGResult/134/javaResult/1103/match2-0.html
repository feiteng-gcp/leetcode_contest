class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int x[] = {a,b,c};
        Arrays.sort(x);
        
        int ma = x[2]-x[1]-1+x[1]-x[0]-1;
        
        int mi = Math.min(x[2]-x[1]-1,1)+Math.min(x[1]-x[0]-1,1);
        
        if(ma==0){
             return new int[]{mi,ma};
        }
        
        if(mi>=2){
            if(x[1]-x[0]==2||x[2]-x[1]==2){
                mi = 1;
            }
        }
        
        
        
        
        
        
        return new int[]{mi,ma};
        
        
    }
}
