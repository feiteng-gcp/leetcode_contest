class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] A = new int[] {a, b, c};
        Arrays.sort(A);
        int min = 0;
        
if (A[2] - A[1] == 1 && A[1] - A[0] == 1) min = 0;
        else if (A[2] - A[1] > 2 && A[1] - A[0] > 2) min = 2;
        else min = 1;
        int max = A[2] - A[0] - 2;
        return new int[] {min, max};
    }
}
