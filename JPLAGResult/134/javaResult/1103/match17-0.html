class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] arr = {a, b, c};
        int[] res = new int[2];
        Arrays.sort(arr);
        a = arr[0];
        b = arr[1];
        c = arr[2];
        if ((b - a == 1 && c - b == 1)) {
            res[0] = 0;
        } else if (b - a == 1 || c - b == 1 || b - a == 2 || c - b == 2) {
            res[0] = 1;
        } else {
            res[0] = 2;
        }
        res[1] = (c - b - 1) + (b - a - 1);
        return res;
    }
}
