class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int x = Math.min(a, Math.min(b, c));
        int z = Math.max(a, Math.max(b, c));
        int y = a + b + c - x - z;
        int[] ans = {2, z - x - 2};
        
if (y - x <= 2 || z - y <= 2) {
            ans[0] = 1;
        }
        if (z - x <= 2) {
            ans[0] = 0;
        }
        return ans;
    }
}
