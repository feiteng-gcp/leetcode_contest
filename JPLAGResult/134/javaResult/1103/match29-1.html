class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int min1= Math.min(a,Math.min(b,c));
        int min3 =  Math.max(a,Math.max(b,c));
        int min2 = a+b+c-min1-min3;
        int max = Math.max(0,min3-min1-2); 
        int diff1 = min2-min1-1;
        int diff2 = min3-min2-1;
        int min = 2;
        
if(diff1 == diff2 && diff1 ==0){
            min = 0;
        }else if(diff1 <=1||diff2 <=1){
            min = 1;
        }else{
            min = 2;
        }
        return new int[]{min,max};
    }
}
