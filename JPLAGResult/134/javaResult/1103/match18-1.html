class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int [] A= {a,b,c};
        Arrays.sort(A);
        if (A[0]+1==A[1] && A[1]+1==A[2]) {
            return new int[]{0,0};
        }

        
if (A[0]+1==A[1]) {
            return new int[]{1,A[2]-A[1]-1};
        } else if (A[1]+1 == A[2]) {
            return new int[]{1,A[1]-A[0]-1};
        } else {
            int minMove = 2;
            if (A[0] + 2 == A[1] || A[1]+2 == A[2])
                minMove = 1;
            
return new int[]{minMove, A[2]-A[1]-1 + A[1]-A[0]-1};
        }
    }
}
