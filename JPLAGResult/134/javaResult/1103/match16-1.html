class Solution {
    public int[] numMovesStones(int a, int b, int c) {
        int[] arr = new int[]{a, b, c};
        Arrays.sort(arr);
        a = arr[0];
        b = arr[1];
        c = arr[2];
        if (a < b && b < c) {
            
int max = b - a - 1 + c - b - 1;
            int min = 0;
            if (b - a == 2 || c - b == 2) min = 1;
            else min = ((b == a + 1) ? 0 : 1) + ((c == b + 1) ? 0: 1);
            
            return new int[]{min, max};
        }
        return new int[2];
    }
}
