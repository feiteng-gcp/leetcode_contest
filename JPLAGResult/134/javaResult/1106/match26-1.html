class Solution {
private static class Position {
		public int x , y;
		public Position(int x , int y) {
			this.x = x;
			this.y = y;			
		}
		@Override
		public int hashCode() {
			return this.x * 31 + this.y;
		}
		@Override
		public boolean equals(Object object) {
			Position position = (Position) object;
			return this.x == position.x && this.y == position.y;
		}
	}
	
	private
 int[] dx = new int[] {- 1 , 0 , 1 , 0};
	private int[] dy = new int[] {0 , 1 , 0 , - 1};	
	
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
    	Queue<Position> queue = new LinkedList<>();    
    	queue.add(new Position(source[0] , source[1]));
    	Set<Position> blockSet = new HashSet<>();
    	Set<Position> vis = new HashSet<>();
    	for (int[] pos : blocked) {    		
    		blockSet.add(new Position(pos[0] , pos[1]));    		
    	}
    	// try 50 layers    	
    	int layer = 0;
    	vis.add(new Position(source[0] , source[1]));    	
    	while (!queue.isEmpty() && layer < 50) {
    		List<Position> nextList = new ArrayList<>();
    		while (!qu
eue.isEmpty()) {
    			Position current = queue.poll();
    			for (int i = 0;i < 4;i ++) {
    				int tx = current.x + dx[i];
    				int ty = current.y + dy[i];
    				if (tx >= 0 && tx < 1000000 && ty >= 0 && ty < 1000000 && !blockSet.contains(new Position(tx , ty))) {
    					Position next = new Position(tx , ty);
    					if (next.x == target[0] && next.y == target[1]) {
    						return true;    						
    					}
    					if (!vis.contains(next)) {
    						vis.add(next);
    						nextList.add(next);    						
    					}
    				}
    			}
    		}    		
    		layer ++;
    		queue.addAll(nextList);    		
    	}   
    	if (layer == 50) {
    		return true;
    	} else {
    		return false;    		
    	}
    }
}
