class Solution {
    
public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        HashSet<Long> bhs = new HashSet<Long>();
        for (int i = 0; i < blocked.length; i++) {
            bhs.add(((long)blocked[i][0])*1000000+blocked[i][1]);
        }
        int count = 0;
        ArrayDeque<Long> q = new ArrayDeque<Long>();
        q.add(((long)source[0])*1000000+source[1]);
        HashSet<Long> vis = new HashSet<Long>();
        vis.add(((long)source[0])*1000000+source[1]);
        long dest = ((long)target[0])*1000000+target[1];
        while (!q.isEmpty() && count <= 40000) {
            long q0 = q.removeFirst();
            int x = (int)(q0/1000000);
            int y = (int)(q0%1000000);
            for (int i = -1; i <= 1; i++) {
                for (int j = -1; j <= 1; j++) {
                    if (i==0&&j==0)
                        continue;
                    if (i!=0&&j!=0)
                        continue;
                    
int a = x+i;
                    int b = y+j;
                    if (a<0||b<0||a>=1000000||b>=1000000)
                        continue;
                    long v = ((long)a)*1000000+b;
                    if (!bhs.contains(v)&&!vis.contains(v)) {
                        if (v==dest)
                            return true;
                        vis.add(v);
                        q.add(v);
                        count++;
                    }
                }
            }
        }
        if (count < 40000)
            return false;
        q = new ArrayDeque<Long>();
        vis = 
new HashSet<Long>();
        count = 0;
        vis.add(((long)target[0])*1000000+target[1]);
        q.add(((long)target[0])*1000000+target[1]);
        while (!q.isEmpty() && count <= 40000) {
            long q0 = q.removeFirst();
            int x = (int)(q0/1000000);
            int y = (int)(q0%1000000);
            for (int i = -1; i <= 1; i++) {
                for (int j = -1; j <= 1; j++) {
                    if (i==0&&j==0)
                        continue;
                    if (i!=0&&j!=0)
                        continue;
                    int a = x+i;
                    int b = y+j;
                    if (a<0||b<0||a>=1000000||b>=1000000)
                        continue;
                    long v = ((long)a)*1000000+b;
                    if (!bhs.contains(v)&&!vis.contains(v)) {
                        vis.add(v);
                        q.add(v);
                        count++;
                    }
                }
            }
        }
        if (count < 40000)
            return false;
        return true;
    }
}
