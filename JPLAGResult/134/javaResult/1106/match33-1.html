class Solution {
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        Deque<int[]> queueS = new LinkedList<>();
        Deque<int[]> queueT = new LinkedList<>();
        queueS.add(source);
        queueT.add(target);
        Map<Integer,Set<Integer>> blocks = new HashMap<>();
        Map<Integer,Set<Integer>> visited = new HashMap<>();
        visited.put(source[0],new HashSet<Integer>());
        if (source[0]!=target[0]) {
           visited.put(target[0],new HashSet<Integer>()); 
        }
        visited.get(source[0]).add(source[1]);
        visited.get(target[0]).add(target[1]);
        
for (int[] points:blocked) {
            if (!blocks.containsKey(points[0])) {
                blocks.put(points[0],new HashSet<Integer>());
            }
            blocks.get(points[0]).add(points[1]);
        }
        int step= 0;
        Set<Integer> emptySet = new HashSet<>();
        while (step <120) {
            Deque<int[]> currQueue = queueS;
            int size = currQueue.size();
            if (size == 0) return false;
            for (int i=0; i<size; i++) {
                int[] point = currQueue.poll();
                int x = point[0];
                int y = point[1];
                if (x>=1 && !visited.getOrDefault(x-1,emptySet).contains(y) && !blocks.getOrDefault(x-1,emptySet).contains(y)) {
                    
currQueue.add(new int[]{x-1,y});
                    if (!visited.containsKey(x-1)) {
                        visited.put(x-1,new HashSet<Integer>());
                    }
                    visited.get(x-1).add(y);
                }
                if (x+1<1000000 && !visited.getOrDefault(x+1,emptySet).contains(y) && !blocks.getOrDefault(x+1,emptySet).contains(y)) {
                    currQueue.add(new int[]{x+1,y});
                    
if (!visited.containsKey(x+1)) {
                        visited.put(x+1,new HashSet<Integer>());
                    }
                    visited.get(x+1).add(y);
                }        
                if (y>=1 && !visited.getOrDefault(x,emptySet).contains(y-1) && !blocks.getOrDefault(x,emptySet).contains(y-1)) {
                    currQueue.add(new int[]{x,y-1});
                    if (!visited.containsKey(x)) {
                        visited.put(x,new HashSet<Integer>());
                    }
                    visited.get(x).add(y-1);
                }
                if (y+1<1000000 && !visited.getOrDefault(x,emptySet).contains(y+1) && !blocks.getOrDefault(x,emptySet).contains(y+1)) {
                    currQueue.add(new int[]{x,y+1});
                    if (!visited.containsKey(x)) {
                        visited.put(x,new HashSet<Integer>());
                    }
                    visited.get(x).add(y+1);
                }                   
            } 
            currQueue = queueT;
            size = currQueue.size();
            if (size == 0) return false;
            for (int i=0; i<size; i++) {
                int[] point = currQueue.poll();
                int x = point[0];
                int y = point[1];
                if (x>=1 && !visited.getOrDefault(x-1,emptySet).contains(y) && !blocks.getOrDefault(x-1,emptySet).contains(y)) {
                    currQueue.add(new int[]{x-1,y});
                    if (!visited.containsKey(x-1)) {
                        visited.put(x-1,new HashSet<Integer>());
                    }
                    visited.get(x-1).add(y);
                }
                if (x+1<1000000 && !visited.getOrDefault(x+1,emptySet).contains(y) && !blocks.getOrDefault(x+1,emptySet).contains(y)) {
                    currQueue.add(new int[]{x+1,y});
                    if (!visited.containsKey(x+1)) {
                        visited.put(x+1,new HashSet<Integer>());
                    }
                    visited.get(x+1).add(y);
                }        
                if (y>=1 && !visited.getOrDefault(x,emptySet).contains(y-1) && !blocks.getOrDefault(x,emptySet).contains(y-1)) {
                    currQueue.add(new int[]{x,y-1});
                    if (!visited.containsKey(x)) {
                        visited.put(x,new HashSet<Integer>());
                    }
                    visited.get(x).add(y-1);
                }
                if (y+1<1000000 && !visited.getOrDefault(x,emptySet).contains(y+1) && !blocks.getOrDefault(x,emptySet).contains(y+1)) {
                    currQueue.add(new int[]{x,y+1});
                    if (!visited.containsKey(x)) {
                        visited.put(x,new HashSet<Integer>());
                    }
                    visited.get(x).add(y+1);
                }                   
            }
            step++;
        }
        
        return true;
    }
    
}
