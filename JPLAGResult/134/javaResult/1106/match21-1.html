class Solution {
    Set<String> set = new HashSet<>();
    
static final int[][] DIRS = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        for (int[] b : blocked) set.add(b[0] + "," + b[1]);
        BFSImpl bfsA = new BFSImpl(source);
        BFSImpl bfsB = new BFSImpl(target);
        while (bfsA.hasNext() && bfsB.hasNext()) {
            if (bfsA.next(bfsB)) return true;
            if (bfsB.next(bfsA)) return true;
        }
        return false;
    }
    
    class BFSImpl {
        private Queue<int[]> q;
        private Set<String> visited; 

        public BFSImpl(int[] init) {
            q = new ArrayDeque<>();
            visited = new HashSet<>();
            q.offer(init);
            String s = init[0] + "," + init[1];
            visited.add(s);
        }

        public boolean hasNext() {
            return !q.isEmpty();
        }

        public boolean next(BFSImpl other) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int[] cur = q.poll();
                for (int[] dir : DIRS) { 
                    int dx = cur[0] + dir[0];
                    int dy = cur[1] + dir[1];
                    String nei = dx + "," + dy;
                    if (dx < 0 || dy < 0 || dx >= 1000000 || dy >= 1000000 || set.contains(nei)) continue;
                    if (other.visited.contains(nei)) return true;
                    if (!visited.contains(nei)) {
                        visited.add(nei);
                        q.offer(new int[] {dx, dy});
                    }
                }
            }
            if (q.size() >= set.size() && other.q.size() >= set.size()) return true;
            return false;
        }
    }
}
