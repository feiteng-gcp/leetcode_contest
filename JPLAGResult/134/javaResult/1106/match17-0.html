class Solution {
    // 1, 1, 1
    // 2, 2, 1
    // 1, 1, 1
    
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        long h = 1_000_000;
        long w = 1_000_000;
        
        Set<Long> blockedSet = new HashSet<>();
        for (int i = 0; i < blocked.length; i++) {
            blockedSet.add(w * blocked[i][0] + blocked[i][1]);
        }
        int blockedCount = blocked.length;
        
        Deque<Long> queue = new ArrayDeque<>();
        Set<Long> visited = new HashSet<>();
        
        long sourcePos = w * source[0] + source[1];
        if (blockedSet.contains(sourcePos)) return false;
    
        visited.add(sourcePos);
        queue.addFirst(sourcePos);
        
        int[] nextX = new int[]{0, 0, 1, -1};
        int[] nextY = new int[]{-1, 1, 0, 0};
        
        while (!queue.isEmpty()) {
            long u = queue.removeFirst();
            long x = u / w;
            long y = u % w;
            
            for (int i = 0; i < 4; i++) {
                long x1 = x + nextX[i];
                long y1 = y + nextY[i];
                
                if (x1 < 0 || y1 < 0 || x1 >= h || y1 >= w) continue;
                long v = w * x1 + y1;
                if (!blockedSet.contains(v) && !visited.contains(v)) {
                    visited.add(v);
                    queue.addLast(v);
                }
            }
            
            if (queue.size() > blockedCount) {
                return true;
            }
        }
        
        return false;
    }
}
