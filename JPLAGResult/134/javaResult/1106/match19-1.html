class Solution {
    private static final int[][] dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    private static final int MIN = 0;
    private static final int MAX = 999999;
    private static final int THRESHOLD = 50000;

    boolean foundTarget = false;
    int total = 0;

    
public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {


        Set<String> bk = new HashSet<>();
        for (int i = 0; i < blocked.length; i++) {
            bk.add(blocked[i][0] + "," + blocked[i][1]);
        }

        foundTarget = false;
        total = 0;

        bfs(source, target, bk);
        if (foundTarget) return true;
        if (total <= THRESHOLD) return false;


        foundTarget = false;
        total = 0;
        bfs(source, target, bk);

        if (foundTarget) return true;
        if (total <= THRESHOLD) return false;

        return true;
    }

    void bfs(int[] source, int[] target, Set<String> bk) {
        Deque<String> q = new ArrayDeque<>();

        Set<String> vst = new HashSet<>();
        total = 1;
        q.add(source[0] + "," + source[1]);
        vst.add(source[0] + "," + source[1]);

        while (!q.isEmpty()) {
            String key = q.poll();
            String[] s = key.split(",");
            int x = Integer.parseInt(s[0]);
            int y = Integer.parseInt(s[1]);

            if (x == target[0] && y == target[1]) {
                foundTarget = true;
                return;
            }

            for (int i = 0; i < dir.length; i++) {
                int r = x + dir[i][0];
                int c = y + dir[i][1];
                if (!(r >= MIN && r <= MAX && c >= MIN && c <= MAX)) continue;
                String k = r + "," + c;
                if (bk.contains(k) || vst.contains(k)) {
                    continue;
                }

                vst.add(k);
                total++;
                q.add(k);

                if (total > THRESHOLD) return;
            }

        }

    }

}
