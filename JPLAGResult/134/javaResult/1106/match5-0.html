import java.util.ArrayDeque;
import java.util.HashSet;
import java.util.Set;

class Solution {
	public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
		Set<Long> b = new HashSet<Long>();
		for (int p[] : blocked)
			b.add(getLong(p));
		return isEscapePossible2(b, source, target) && isEscapePossible2(b, target, source);
	}

	private boolean isEscapePossible2(Set<Long> b, int[] target, int[] source) {
		ArrayDeque<int[]> q = new ArrayDeque<int[]>();
		q.add(source);
		Set<Long> visited = new HashSet<Long>();
		while (!q.isEmpty()) {
			int[] p = q.poll();
			if (p[0] == target[0] && p[1] == target[1])
				return true;
			if (p[0] >= 0 && p[1] >= 0)
				if (visited.add(getLong(p)) && !b.contains(getLong(p))) {
					q.add(new int[] { p[0] + 1, p[1] });
					q.add(new int[] { p[0] - 1, p[1] });
					q.add(new int[] { p[0], p[1] + 1 });
					q.add(new int[] { p[0], p[1] - 1 });
				}

			if (visited.size() > 40000)
				return true;
		}
		return false;
	}

	private Long getLong(int[] p) {
		return 1000000L * p[0] + p[1];
	}

}
