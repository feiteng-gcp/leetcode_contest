class Solution {
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        HashSet<String> set = new HashSet<>();
        for(int[] b: blocked){
            set.add(b[0]+":"+b[1]);
        }
        return !locked(source, new HashSet<>(set))&&!locked(target, new HashSet<>(set));
    }
    
    private boolean locked(int[] init, Set<String> set){
        int size = set.size(), end = (int)1e6;
        
int[][] dirs = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
        Queue<int[]> q = new LinkedList<>();
        q.offer(init);
        set.add(init[0]+":"+init[1]);
        while(!q.isEmpty()){
            for(int i=q.size(); i>0; i--){
                int[] cur = q.poll();
                int x=cur[0],  y=cur[1];
                for(int[] dir : dirs){
                    int nx=dir[0]+x, ny=dir[1]+y;
                    if(nx<0||nx>=end||ny<0||ny>=end||set.contains(nx+":"+ny)) continue;
                    q.offer(new int[]{nx, ny});
                    set.add(nx+":"+ny);
                }
            }
            
if(q.size()>size) return false;
        }
        return true;
    }
}
