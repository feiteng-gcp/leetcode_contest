public class Solution {
    private static final int[][] dirs = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

    private int R = 1_000_000, C = 1_000_000;
    Set<String> set = new HashSet<>();
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        Queue<int[]> frontier = new ArrayDeque<>();
        frontier.add(source);
        for (int[] b : blocked) {
            set.add(b[0] + "," + b[1]);
        }
        set.add(source[0] + "," + source[1]);
        for (int iters = 0; iters < 200; ++iters) {
            if (frontier.isEmpty()) return false;
            int size = frontier.size();
            for (int i = 0; i < size; ++i) {
                int[] cur = frontier.poll();
                for (int[] dir : dirs) {
                    int nR = cur[0] + dir[0], nC = cur[1] + dir[1];
                    String s = nR + "," + nC;
                    if (nR >= 0 && nR < R && nC >= 0 && nC < C && !set.contains(s)) {
                        set.add(s);
                        frontier.offer(new int[]{nR, nC});
                    }
                }
            }
        }
        return true;

    }
}
