class Solution {
    int[][] b;
    int nb;
    long n = 1000000L;
    Set<Long> set;
    
int[][] dirs = {{0,1}, {0,-1}, {1,0}, {-1,0}};
    
    
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        if (blocked == null || blocked.length < 2) return true;
        set = new HashSet<>();
        
        b = blocked;
        nb = b.length;
        for (int[] v : b) {
            set.add(v[0] * n + v[1]);
        }
        
        
if (find(source, target)) return true;
        
        if (enclosed(source) || enclosed(target))
            return false;
        return true;
    }
    private boolean find(int[] point, int[] p2) {
        Deque<Long> q = new ArrayDeque<>();
        Set<Long> visited = new HashSet<>();
        long s = point[0] * n + point[1];
        long t = p2[0] * n + p2[1];
        q.add(s);
        visited.add(s);
        int d = 0;
        while (!q.isEmpty()) {
            d++;
            if (d > nb) return false;
            int sz = q.size();
            for (int i = 0; i < sz; i++) {
            long cur = q.remove();
            long x = (cur / n);
            long y = cur % n;
            
            for (int[] dir : dirs) {
                long xx = x + dir[0];
                long yy = y + dir[1];
                long xy = xx * n + yy;
                
                    if (xy == t)
                        return true;
                
                if (xx >= 0 && xx < n && yy >= 0 && yy < n 
                    && !visited.contains(xy) && !set.contains(xy)) {
                    visited.add(xy);
                    q.add(xy);
                }
            }
            }
        }
        return false;
    }
    
    
    private boolean enclosed(int[] point) {
        Deque<Long> q = new ArrayDeque<>();
        Set<Long> visited = new HashSet<>();
        long s = point[0] * n + point[1];
       
        q.add(s);
        visited.add(s);
        int d = 0;
        while (!q.isEmpty()) {
            d++;
            if (d > nb) return false;
            int sz = q.size();
            for (int i = 0; i < sz; i++) {
            long cur = q.remove();
            long x = (cur / n);
            long y = cur % n;
            
            for (int[] dir : dirs) {
                long xx = x + dir[0];
                long yy = y + dir[1];
                long xy = xx * n + yy;
                
                if (xx >= 0 && xx < n && yy >= 0 && yy < n 
                    && !visited.contains(xy) && !set.contains(xy)) {
                    visited.add(xy);
                    q.add(xy);
                }
            }
            }
        }
        return true;
    }
}
