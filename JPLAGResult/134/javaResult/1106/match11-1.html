class Solution {
    private static final int[][] neighbors = new int[][] {{0, +1}, {+1, 0}, {0, -1}, {-1, 0}};
    private static final int bfs_limit = 5000;
    private static final int size_limit = 1000000;

    
public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        Set2D<Integer, Integer> blockers = new Set2D<>();
        for (int[] blocker : blocked) {
            blockers.put(blocker[0], blocker[1]);
        
}
        return isUnlimitedOrReachableToTarget(blockers, source, target) && isUnlimitedOrReachableToTarget(blockers, target, source);
    }

    private boolean isUnlimitedOrReachableToTarget(Set2D<Integer, Integer> blockers, int[] source, int[] target) {
        Set2D<Integer, Integer> visited = new Set2D<>();
        Queue<int[]> queue = new LinkedList<>();
        queue.add(source);
        while (!queue.isEmpty()) {
            int[] position = queue.remove();
            int r = position[0];
            int c = position[1];
            if (r < 0 || r >= size_limit || c < 0 || c >= size_limit || blockers.contains(r, c) || visited.contains(r, c)) {
                continue;
            }
            visited.put(r, c);
            if (visited.size() > bfs_limit
                    || (r == target[0] && c == target[1])) {
                return true;
            }
            Arrays.stream(neighbors).forEach(neighbor -> queue.add(new int[] {r + neighbor[0], c + neighbor[1]}));
        }
        return false;
    }

    class Set2D<K1, K2> {
        private Map<K1, Set<K2>> map = new HashMap<>();
        private int size = 0;

        public boolean contains(K1 k1, K2 k2) {
            return map.getOrDefault(k1, Collections.emptySet()).contains(k2);
        }

        public void put(K1 k1, K2 k2) {
            if (map.computeIfAbsent(k1, k -> new HashSet<>()).add(k2)) {
                size++;
            }
        }

        public int size() {
            return this.size;
        }
    }
}
