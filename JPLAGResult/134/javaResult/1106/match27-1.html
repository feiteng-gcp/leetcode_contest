class Solution {
        int[] dx = {1, -1, 0, 0};
    int[] dy = {0, 0, 1, -1};
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        Queue<int[]> queue = new LinkedList<>();
        Set<String> set = new HashSet<>();
        set.add(source[0] + "," + source[1]);

        Set<String> bl = new HashSet<>();
        for (int[] b : blocked) {
            bl.add(b[0] + "," + b[1]);
        }
        queue.offer(source);
        int d = 0;
        while (!queue.isEmpty() && d <= 201) {
            d++;
            int size = queue.size();
            for (int j = 0; j < size; j++) {
                int[] cur = queue.poll();
                //System.out.println(Arrays.toString(cur));
                if (cur[0] == target[0] && cur[1] == target[1]) {
                    return true;
                }
                for (int i = 0; i < 4; i++) {
                    int x = cur[0] + dx[i];
                    int y = cur[1] + dy[i];
                    if (x >= 0 && x < 1000000 && y >= 0 && y < 1000000) {
                        if (set.contains(x + "," + y)) {
                            continue;
                        }
                        if (bl.contains(x + "," + y)) {
                            continue;
                        }
                        queue.offer(new int[]{x, y});
                        set.add(x + "," + y);
                    
}
                }
            }
        }
        return !queue.isEmpty();
    }
}
