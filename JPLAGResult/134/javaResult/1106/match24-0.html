class Solution {
    
    int[][] ds = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    int[] add(int[] v1, int[] v2) {
        return new int[]{v1[0] + v2[0], v1[1] + v2[1]};
    }

    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        return isEscapeInternal(blocked, source, target) && isEscapeInternal(blocked, target, source);
    }

    public boolean isEscapeInternal(int[][] blocked, int[] source, int[] target) {
        int MAX = 1_000_000;
        Set<Long> bSet = new HashSet<>();
        for (int[] b : blocked) {
            bSet.add(toLong(b));
        }
        Deque<int[]> dq = new LinkedList<>();
        Set<Long> seen = new HashSet<>();
        dq.offerLast(source);
        seen.add(toLong(source));
        int step = 0;
        while (!dq.isEmpty()) {
            if (step > 200) {
                return true;
            }
            step++;
            int N = dq.size();
            for (int i = 0; i < N; i++) {
                int[] cp = dq.pollFirst();
                for (int[] d : ds) {
                    int[] np = add(d, cp);
                    if (np[0] == -1 || np[0] == MAX || np[1] == -1 || np[1] == MAX || bSet.contains(toLong(np)) || seen.contains(toLong(np))) {
                        continue;
                    }
                    if (np[0] == target[0] && np[1] == target[1]) {
                        return true;
                    }
                    dq.offerLast(np);
                    seen.add(toLong(np));
                }
            }
        }
        return false;
    }

    long toLong(int[] a) {
        return (long) a[0] * 1000_000 + a[1];
    }
}
