class Solution {
    private int[][] ds = {{0,1},{0,-1},{1,0},{-1,0}};
    public boolean isEscapePossible(int[][] blocked, int[] source, int[] target) {
        // blocked = new int[0][0];
        // source = new int[]{0,0};
        // target = new int[]{999999,999999};
        Set<String> blocks = new HashSet<>();
        for (int[] block : blocked) {
            blocks.add(Integer.toString(block[0]) + "," + Integer.toString(block[1]));
        }
        return dfs(blocks, new HashSet<>(), source[0], source[1], source) &&
               dfs(blocks, new HashSet<>(), target[0], target[1], target);
    }
    private boolean dfs(Set<String> blocks, Set<String> visited, int i, int j, int[] arr) {
        if (i < 0 || i == 1000000 || j < 0 || j == 1000000) return false;
        String pos = Integer.toString(i) + "," + Integer.toString(j);
        if (blocks.contains(pos)) return false;
        if (Math.abs(i - arr[0]) + Math.abs(j - arr[1]) == 201) return true;
        if (!visited.add(pos)) return false;
        for (int[] d : ds) {
            if (dfs(blocks, visited, i + d[0], j + d[1], arr)) return true;
        }
        return false;
    }
}
