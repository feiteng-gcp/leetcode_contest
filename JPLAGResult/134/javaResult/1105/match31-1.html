class Solution {
    
public int maxUncrossedLines(int[] A, int[] B) {
            int m = A.length;
    int n = B.length;

    // C[i][j] are answer answer for A[0,i) vs B[0,j), we want C[m][n]
    int[][] C = new int[m+1][n+1];
    for (int i = 1; i <= m; i++) {
      for (int j = 1; j <= n; j++) {
        if (A[i-1] == B[j-1]) {
          C[i][j] = C[i-1][j-1] + 1;
        } else {
          C[i][j] = Math.max(C[i-1][j], C[i][j-1]);
        
}
      }
    }
    return C[m][n];
    }
}
