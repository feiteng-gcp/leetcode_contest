class Solution {
    public int maxUncrossedLines(int[] A, int[] B) {
 int LA = A.length;
        int LB = B.length;
        int [][] dp = new int [LA+1][LB+1];
        
        for (int j = 0; j < LA; j++) {
        	for (int k = 0; k < LB; k++) {
        		if (A[j] == B[k]) {
        			dp[j+1][k+1] = dp[j][k] + 1;
        		} else {

        			dp[j+1][k+1] = Math.max(dp[j][k+1], dp[j+1][k]);
        		}
        	}
        }
        return dp[LA][LB];    }
}
