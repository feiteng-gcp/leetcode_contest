class Solution {
    public int maxUncrossedLines(int[] A, int[] B) {
        final int n = A.length, m = B.length;
        int[][] dp = new int[n+1][m+1];
        for (int i = n-1; i >= 0; i--) {
            for (int j = m-1; j >= 0; j--) {
                if (A[i] == B[j]) {
                    dp[i][j] = Math.max(dp[i][j], dp[i+1][j+1] + 1);
                } else {
                    dp[i][j] = Math.max(dp[i+1][j], dp[i][j+1]);
                }
            }
        }
        return dp[0][0];
    }
}
