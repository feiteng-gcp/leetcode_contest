class Solution {
		int max(int a,int b)
		{
			return a>b?a:b;
		}
	    public int maxUncrossedLines(int[] A, int[] B) {
	        int alen = A.length;
	        int blen = B.length;
	        int dp[][] = new int [alen+1][blen+1];
	        for(int i=1;i<=alen;i++)
	        {
	        	for(int j=1;j<=blen;j++)
	        	{
	        		if(A[i-1]==B[j-1])
	        		{
	        			dp[i][j] = dp[i-1][j-1] + 1;
	        		}else{
	        			dp[i][j] = max(dp[i-1][j-1],dp[i][j-1]);
	        			dp[i][j] = max(dp[i][j],dp[i-1][j]);
	        		}
	        	}
	        }
	        return dp[alen][blen];
	    }
	}
