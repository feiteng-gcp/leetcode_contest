class Solution {
    public int maxUncrossedLines(int[] A, int[] B) {
        int N1 = A.length;
        int N2 = B.length;
        int[][] dp = new int[N1 + 1][N2 + 1];
        for (int i = 0; i < N1; i++) {
            for (int j = 0; j < N2; j++) {
                if (A[i] == B[j]) {
                    dp[i + 1][j + 1] = dp[i][j] + 1;
                } else {
                    dp[i + 1][j + 1] =  Integer.max(dp[i][j + 1], dp[i + 1][j]);
                }
            }
        }
        return dp[N1][N2];
    }
}
