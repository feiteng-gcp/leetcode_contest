public class Solution {
	public int maxUncrossedLines(int[] A, int[] B) {
		int la = A.length, lb = B.length;
		int[][] dp = new int[la + 1][lb + 1];
		for (int r = 1; r <= la; ++r) {
			for (int c = 1; c <= lb; ++c) {
				if (A[r - 1] == B[c - 1]) {
					dp[r][c] = dp[r - 1][c - 1] + 1;
				} else {

					dp[r][c] = Math.max(dp[r - 1][c], dp[r][c - 1]);
				}
			}
		}
		return dp[la][lb];
	}
}
