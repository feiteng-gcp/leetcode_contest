class Solution:
    def numMovesStones(self, a, b, c):
        x = min([a,b,c])
        z = max([a,b,c])
        y = a+b+c-x-z
        x2 = min([a,b,c])
        z2 = max([a,b,c])
        y2 = a+b+c-x-z
        min_ = 0
        max_ = 0
        while (y-x)>1 or (z-y)>1:
            if (y-x)>(z-y):
                x = z-1
                if x == y:
                    x-=1                
            else:
                z = x+1
                if z == y:
                    z+=1
            min_ += 1
            a = x
            b = y
            c = z
            x = min([a,b,c])
            z = max([a,b,c])
            y = a+b+c-x-z
        while (y2-x2)>1 or (z2-y2)>1:
            if (y2-x2)>(z2-y2):
                x2 = x2+1          
            else:
                z2 = z2-1
            max_ += 1
        return([min_,max_])
