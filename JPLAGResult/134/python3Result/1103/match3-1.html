class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        a, b, c = sorted([a, b, c])
        _max = b - a - 1 + c - b - 1
        if a == b - 1 and c == b + 1:
            _min = 0
        elif a == b - 1 or a == b - 2 or c == b + 1 or c == b + 2:
            _min = 1
        else:
            _min = 2
        return [_min, _max]
