class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        
        nums = [a, b, c]
        nums.sort()
        a, b, c = nums[0], nums[1], nums[2]
        
        min_moves, max_moves = 2, (b-a-1) + (c-b-1)
        if a + 1 == b and b + 1 == c:
            min_moves = 0
        elif a + 1 == b or b + 1 == c:
            min_moves = 1
        elif a + 2 == b or b + 2 == c:
            min_moves = 1

        return [min_moves, max_moves]
