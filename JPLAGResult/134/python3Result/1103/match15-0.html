class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        l=sorted([a,b,c])
        mx=0
        if l[2]-l[0]>2:
            mx=l[2]-l[0]-2
        mi=2
        if l[2]-l[1]==1 and l[1]-l[0]==1:
            mi=0
        elif l[2]-l[1]==2 or l[1]-l[0]==2 or l[2]-l[1]==1 or l[1]-l[0]==1:
            mi=1
        return [mi,mx]
