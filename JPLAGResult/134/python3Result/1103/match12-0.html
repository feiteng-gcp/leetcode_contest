class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        a, b, c = sorted([a, b, c])
        maxMove = c - a - 2
        if a + 2 == b + 1 == c:
            return [0, 0]
        if b == c - 1 or b == a + 1 or b == a + 2 or b == c - 2:
            return [1, maxMove]
        return [2, maxMove]
