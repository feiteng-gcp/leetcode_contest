class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        x,y,z = tuple(sorted([
a,b,c]))
        max_move = z-x-2
        if z-y == 1 and y-x == 1:
            min_move = 0
        elif z-y == 1 or y-x == 1:
            min_move = 1
        elif z-y == 2 or y-x == 2:
            min_move = 1
        else:
            min_move = 2
        return [min_move, max_move]
