class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        a, b, c = sorted([a, b, c])
        max_move = (b - a - 1) + (c - b - 1)
        if b - a == 2 or c - b == 2:
            min_move = 1
        else:
            min_move = 0
            if b > a + 1:
                min_move += 1
            if c > b + 1:
                min_move += 1
        return [min_move, max_move]
