class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        array = [a,b,c]
        array.sort()
        if array[-1] - array[0] == 2:
            return [0,0] # cannot make any move
        
        minn, maxx = 0, 0
        if array[1] - array[0] <= 2 or array[2] - array[1] <= 2:
            minn = 1
        else: minn = 2
        
        maxx = array[-1] - array[0] - 2
        return [minn, maxx]
