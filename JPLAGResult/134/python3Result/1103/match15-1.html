class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        pos = [a,b,c]
        pos.sort()
        maxMove = pos[2]-pos[0]-2
        minMove = 2
        if pos[0]+1==pos[1] and pos[1]+1==pos[2]:
            maxMove = 0
            minMove 
= 0
        elif pos[0]+1==pos[1] or pos[0]+2==pos[1] or pos[1]+1==pos[2] or pos[1]+2==pos[2]:
            minMove = 1
        return [minMove, maxMove]
