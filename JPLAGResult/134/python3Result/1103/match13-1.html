class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        t = [a,b,c]
        t.sort()
        a,b,c = t[0],t[1],t[2]
        a1 = 0
        a2 = 0
        if b - a > 1:
            a1 += 1
            a2 += b - a - 1
        if c - b > 1:
            a1 += 1
            a2 += c - b - 1
        if (b - a == 2 and c - b >= 2) or (b - a >= 2 and c - b == 2):
            a1 -= 1
        return [a1,a2]
