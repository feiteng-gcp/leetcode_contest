class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        a, b, c = sorted([a, b, c])
        mi = 0
        if b - a > 1 and c - b == 2:
            mi = 1
        elif b - a == 2 and c - b > 1:
            mi = 1
        else:
            if c - b > 1 and b - a > 1:
                mi = 2
            elif c - b > 1:
                mi = 1
            elif b - a > 1:
                mi = 1
            
                
        ans = [mi, max(0, b - a - 1) + max(0, c - b - 1)]
        return ans
