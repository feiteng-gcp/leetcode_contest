class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        arr = [a, b, c]
        arr.sort()
        d1 = arr[1] - arr[0]
        d2 = arr[2] - arr[1]
        myMax = d1 - 1 + d2 - 1
        myMin = 0
        if d1 == 1 and d2 == 1:
            myMin = 0
        elif d1 == 1 or d2 == 1 or d1 == 2 or d2 == 2:
            myMin = 1
        else:
            myMin = 2
        return [myMin, myMax]
