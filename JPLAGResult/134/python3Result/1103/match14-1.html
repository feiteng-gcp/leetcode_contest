class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        s=[a,b,c]
        s.sort()
        if s[0]+2==s[2]: return [0,0]
        mx = s[2]-s[0]-2 
        if s[1]-s[0]==2 or s[2]-s[1]==2: return [1,mx]
        return  [(0 if s[1]-s[0]==1 else 1)+(0 if s[2]-s[1]==1 else 1),mx]
