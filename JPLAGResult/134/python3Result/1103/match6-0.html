class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        nums = [a, b, c]
        nums.sort()
        a, b, c = nums[0], nums[1], nums[2]
        maxStep = c - a - 2
        if c - b == 1 and b - a == 1:
            minStep = 0
        elif c - b == 1 or b - a == 1:
            minStep = 1
        elif c - b == 2 or b - a == 2:
            minStep = 1
        else:
            minStep = 2
        return [minStep, maxStep]
