class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        A=[a,b,c]
        A.sort()
        a,b,c=A
        res1=0
        res2=b-a-1+c-b-1
        x1=b-a-1
        x2=c-b-1
        if x1==0 and x2==0:
            res1=0
        elif x1<=1 or x2<=1:
            res1=1
        else:
            res1=2
        return [res1,res2]
