class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        s = [a, b, c]
        s.sort()
        resmin = 0
        resmax = 0
        a, b, c = s[0], s[1], s[2]
        if a + 1 == b == c - 1:
            return [0,0]
        
        resmax 
= (b-a-1) + (c-b-1)
        if a + 1 == b or b + 1 == c:
            return [1, resmax]
        if a + 2 == b or b + 2 == c:
            return [1, resmax]
        return [2, resmax]
