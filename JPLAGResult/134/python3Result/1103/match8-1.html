class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        mn = min(a,b,c)
        mx = max(a,b,c)
        y 
= (a+b+c)-(mn+mx)
        A = ((y-1)-mn)+((mx)-(y+1))
        B= 2
        if(mx-mn==2):
            B = 0
        elif(((y-mn)==2) or ((mx-y)==2)):
            B = 1  
        elif((y-1==mn) or (y+1==mx)):
            B = 1
        return [B,A]
