class Solution:
    def numMovesStones(self, a: int, b: int, c: int) -> List[int]:
        a, b, c = sorted([a, b, c])
        max_step = c - b - 1 + b - a - 1
        if b == a + 1 and c == b + 1:
            min_step = 0
        elif b == a + 1 or c == b + 1 or b == a + 2 or c == b + 2:
            min_step = 1
        else:
            min_step = 2
        return [min_step, max_step]
