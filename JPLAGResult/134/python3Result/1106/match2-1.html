class Solution:
    def isEscapePossible(self, blocked: List[List[int]], source: List[int], target: List[int]) -> bool:
        block = set()
        for x, y in blocked:
            block.add((x, y))

        return self.isOpen(source, block, target) and self.isOpen(target, block, source)    
    
    def isOpen(self, position, block, target):
        x0, y0 = position
        x1, y1 = target
        queue = [(x0, y0)]
        print(queue)
        visited = set(queue)
        while queue:
            x, y = queue.pop()
            if (x == x1 and y == y1) or abs(x - x0) > 100 or abs(y - y0) > 100:
                return True
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                x_, y_ = x + dx, y + dy
                if 0 <= x_ < 1e6 and 0 <= y_ < 1e6 and (x_, y_) not in visited and (x_, y_) not in block:
                    queue.append((x_, y_))
                    print('appended', x_, y_)
                    visited.add((x_, y_))
        return False
