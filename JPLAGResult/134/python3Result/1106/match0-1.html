class Solution:
    def isEscapePossible(self, blocked: List[List[int]], source: List[int], target: List[int]) -> bool:
        #r*2*pi == 200
        R = 32
        def bfs(s, t):
            q = collections.deque()
            q.append(tuple(s))
            visited = set()
            visited.add(tuple(s))
            m,n = 10**6, 10**6
            count = 1
            while q:
                l = len(q)
                
for _ in range(l):
                    r, c = q.popleft()
                    if [r,c] == t:
                        return True
                    for d in [[0,1], [0,-1],[1,0], [-1,0]]:
                        rn, cn = r+d[0], c+d[1]
                        if 0 <= rn < m and 0 <= cn < n and [rn,cn] not in blocked and (rn,cn) not in visited:
                            q.append((rn,cn))
                            visited.add((rn,cn))
                            count += 1
                    if count >= 3400:
                        return True
            return False
        return bfs(source, target) and bfs(target, source)
