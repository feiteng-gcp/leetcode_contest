class Solution:
    def isEscapePossible(self, blocked: List[List[int]], source: List[int], target: List[int]) -> bool:
        b = set()
        for i in blocked:
            b.add((i[0],i[1]))
        z1={(source[0],source[1])}
        z1f={(source[0],source[1])}
        while len(z1f)>0 and len(z1)<15000: #10k should be enough for 200 blocked
            (x,y) = z1f.pop()
            if x>1 and (x-1,y) not in b and (x-1,y) not in z1:
                if x-1==target[0] and y==target[1]: return True
                z1.add((x-1,y))
                z1f.add((x-1,y))
            if x<999999 and (x+1,y) not in b and (x+1,y) not in z1:
                if x+1==target[0] and y==target[1]: return True
                z1.add((x+1,y))
                z1f.add((x+1,y))                
            if y>1 and (x,y-1) not in b and (x,y-1) not in z1:
                if x==target[0] and y-1==target[1]: return True
                z1.add((x,y-1))
                z1f.add((x,y-1))
            if y<999999 and (x,y+1) not in b and (x,y+1) not in z1:
                if x==target[0] and y+1==target[1]: return True
                z1.add((x,y+1))
                z1f.add((x,y+1))
        if len(z1f)==0: return False
        #overflow on this zone its trillion-ish size so check the other one
        z1={(target[0],target[1])}
        z1f={(target[0],target[1])}
        while len(z1f)>0 and len(z1)<15000: #10k should be enough for 200 blocked
            (x,y) = z1f.pop()
            if x>1 and (x-1,y) not in b and (x-1,y) not in z1:
                if x-1==source[0] and y==source[1]: return True
                z1.add((x-1,y))
                z1f.add((x-1,y))
            if x<999999 and (x+1,y) not in b and (x+1,y) not in z1:
                if x+1==source[0] and y==source[1]: return True
                z1.add((x+1,y))
                z1f.add((x+1,y))                
            if y>1 and (x,y-1) not in b and (x,y-1) not in z1:
                if x==source[0] and y-1==source[1]: return True
                z1.add((x,y-1))
                z1f.add((x,y-1))
            if y<999999 and (x,y+1) not in b and (x,y+1) not in z1:
                if x==source[0] and y+1==source[1]: return True
                z1.add((x,y+1))
                z1f.add((x,y+1))
        if len(z1f)==0: return False
        return True # both are in huge connected component
