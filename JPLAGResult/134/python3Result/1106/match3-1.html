class Solution:
    def isEscapePossible(self, blocked: List[List[int]], source: List[int], target: List[int]) -> bool:
        blocked_set = set()
        for r, c in blocked:
            blocked_set.add((r, c))
        # perform search at source
        frontier = [source]
        visited = set()
        deltas = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        while frontier and len(visited) <= 10000:
            r, c = frontier.pop()
            if (r, c) in visited:
                continue
            if r == target[0] and c == target[1]:
                return True
            visited.add((r, c))
            neighbors = [[r + delta[0], c + delta[1]] for delta in deltas
                         if (r + delta[0] >= 0 and c + delta[1] >= 0
                             and (r + delta[0], c + delta[1]) not in blocked_set)]
            for neighbor in neighbors:
                frontier.append(neighbor)
        if len(visited) <= 10000: # we got closed off!
            return False
        
        # begin search at target
        frontier = [target]
        visited = set()
        deltas = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        while frontier and len(visited) <= 10000:
            r, c = frontier.pop()
            if (r, c) in visited:
                continue
            if r == source[0] and c == source[1]:
                return True
            visited.add((r, c))
            neighbors = [[r + delta[0], c + delta[1]] for delta in deltas
                         if (r + delta[0] >= 0 and c + delta[1] >= 0
                             and (r + delta[0], c + delta[1]) not in blocked_set)]
            for neighbor in neighbors:
                frontier.append(neighbor)
        if len(visited) <= 10000: # we got closed off!
            return False
        
        return True # not possible to close us off!
