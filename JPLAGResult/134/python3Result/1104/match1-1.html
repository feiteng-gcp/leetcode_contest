class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        m, n = len(grid), len(grid[0])
        visited = set()
        to_color = set()
        direct = ((0, 1), (
0, -1), (1, 0), (-1, 0))
        def dfs_color(r, c):
            if (r, c) in visited:
                return
            visited.add((r, c))
            for dr, dc in direct:
                r1, c1 = r + dr, c + dc
                if r1 < 0 or r1 >= m or c1 < 0 or c1 >= n or grid[r][c] != grid
[r1][c1]:
                    to_color.add((r, c))
                else:
                    dfs_color(r1, c1)
        dfs_color(r0, c0)
        # print(to_color)
        for r, c in to_color:
            grid[r][c] = color
        return grid
