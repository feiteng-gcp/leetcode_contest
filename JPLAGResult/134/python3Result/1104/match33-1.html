class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        b = collections.deque([(r0, c0)])
        dx, dy = [-1, 0, 1, 0], [0, 1, 0, -1]
        n, m = len(grid), len(grid[0])
        c = grid[r0][c0]
        d = set()
        while b:
            x, y = b.popleft()
            for r in range(4):
                mx, my = x + dx[r], y + dy[r]
                if not (0 <= mx < n and 0 <= my < m and abs(grid[mx][my]) == c):
                    d.add((x, y))
                    break
            for k in range(4):
                nx, ny = x + dx[k], y + dy[k]
                if 0 <= nx < n and 0 <= ny < m and grid[nx][ny] == c:
                    b.append((nx, ny))
                    grid[nx][ny] *= -1
        for i, j in d:
            grid[i][j] = color
        
for i in range(n):
            for j in range(m):
                if grid[i][j] < 0:
                    grid[i][j] *= -1
        return grid
