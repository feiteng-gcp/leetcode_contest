class Solution:
    def dfs(self, i, j, grid, m, n, color, q):
        q.add((i, j))
        
        
for di, dj in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
            x, y = i + di, j + dj
            if 0 <= x < m and 0 <= y < n and (x, y) not in q and grid[x][y] == color:
                self.dfs(x, y, grid, m, n, color, q)
    
    def select(self, i, j, grid, m, n, old_color):
        for di, dj in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
            x, y = i + di, j + dj
            if x < 0 or x >= m or y < 0 or y >= n or grid[x][y] != old_color:
                return True
        return False
    
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        if grid[r0][c0] == color:
            return grid
        q = set()
        m, n = len(grid), len(grid[0]) if grid else 0
        self.dfs(r0, c0, grid, m, n, grid[r0][c0], q)
        
        res = [(i, j) for i, j in q if self.select(i, j, grid, m, n, grid[r0][c0])]
        
        for i, j in res:
            grid[i][j] = color
        
        return grid
