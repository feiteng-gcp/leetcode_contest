import copy


class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        new_grid = copy.deepcopy(grid)
        m = len(grid)
        n = len(grid[0])
        seen = set()
        
        def dfs(i, j, old, new):
            seen.add((i, j))
            if not (0 <= i < m and 0 <= j < n):
                return True
            if grid[i][j] == old:
                
if i == 0 or i == m - 1 or j == 0 or j == n - 1:
                    new_grid[i][j] = new
                res = []
                for r, c in [(i-1, j), (i+1, j), (i, j-1), (i, j+1)]:
                    if 0 <= r < m and 0 <= c < n:
                        if grid[r][c] != old:
                            new_grid[i][j] = new
                    if (r, c) not in seen:
                        res.append(dfs(r, c, old, new))
                return False
            else:
                return True
            
        dfs(r0, c0, grid[r0][c0], color)
        return new_grid
