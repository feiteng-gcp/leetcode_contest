from collections import deque


class Solution:
    def getComponent(self, grid, r0, c0, dx, dy, m, n):
        component = [
            [False for _ in range(n)]
            for _ in range(m)
        ]
        seen = [
            [False for _ in range(n)]
            for _ in range(m)
        ]
        
        que = deque()
        que.append((r0, c0))
        seen[r0][c0] = True
        original = grid[r0][c0]
        
        while que:
            ox, oy = que.popleft()
            component[ox][oy] = True
            for i in range(4):
                nx = ox + dx[i]
                ny = oy + dy[i]
                if 0 <= nx < m and 0 <= ny < n and grid[nx][ny] == original and not seen[nx][ny]:
                    que.append((nx, ny))
                    seen[nx][ny] = True
                    
        return component
                    
    def getBoarder(self, component, r0, c0, dx, dy, m, n):
        boarder = [
            [False for _ in range(n)]
            for _ in range(m)
        ]
        seen = [
            [False for _ in range(n)]
            for _ in range(m)
        ]
        que = deque()
        que.append((r0, c0))
        seen[r0][c0] = True
        
        while que:
            ox, oy = que.popleft()
            is_boarder = False
            if ox in (0, m - 1) or oy in (0, n - 1):
                is_boarder = True
            
            for i in range(4):
                nx = ox + dx[i]
                ny = oy + dy[i]
                if 0 <= nx < m and 0 <= ny < n:
                    if not component[nx][ny]:
                        is_boarder = True
                    elif component[nx][ny] and not seen[nx][ny]:
                        seen[nx][ny] = True
                        que.append((nx, ny))
            
            if is_boarder:
                boarder[ox][oy] = True
                
        return boarder
                
                    
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        m = len(grid)
        n = len(grid[0])
        
        dx = [0, 0, 1, -1]
        dy = [1, -1, 0, 0]
        
        component = self.getComponent(grid, r0, c0, dx, dy, m, n)
        boarder = self.getBoarder(component, r0, c0, dx, dy, m, n)
        
        for i in range(m):
            for j in range(n):
                if boarder[i][j]:
                    grid[i][j] = color
        
        return grid
