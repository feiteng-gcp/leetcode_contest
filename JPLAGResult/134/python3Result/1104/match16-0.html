class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        dirs = [
            (0, 1),
            (1, 0),
            (0, -1),
            (-1, 0),
        ]
        m, n = len(grid), len(grid[0])
        q = [(r0, c0)]
        visited = set()
        visited.add((r0, c0))
        cc_color = grid[r0][c0]
        borders = []
        while q:
            r, c = q.pop()
            if r in [0, m - 1] or c in [0, n - 1]:
                borders.append((r, c))
            else:
                count = 0
                for dr in dirs:
                    rr = r + dr[0]
                    cc = c + dr[1]
                    if grid[rr][cc] == cc_color:
                        count += 1
                if count != 4:
                    borders.append((r, c))
            for dr in dirs:
                rr = r + dr[0]
                cc = c + dr[1]
                if 0 <= rr < m and 0 <= cc < n and (rr, cc) not in visited and grid[rr][cc] == cc_color:
                    visited.add((rr, cc))
                    q.append((rr, cc))
        
        for r, c in borders:
            grid[r][c] = color
        
        return grid
