class Solution:
    def getBorderCoords(self, grid: List[List[int]], r0: int, c0: int) -> List[List[int]]:
        comp_color = grid[r0][c0]
        border = []
        frontier = [[r0, c0]]
        visited = set()
        while frontier:
            r, c = frontier.pop()
            if (r, c) in visited:
                continue
            visited.add((r, c))
            # neighbors that are the same component color
            deltas = [[1, 0], [-1, 0], [0, 1], [0, -1]]
            sim_neighbors = [[r + delta[0], c + delta[1]] for delta in deltas
                             if (r + delta[0] >= 0 and r + delta[0] < len(grid) and
                                 c + delta[1] >= 0 and c + delta[1] < len(grid[0]) and
                                 grid[r + delta[0]][c + delta[1]] == comp_color)]
            diff_neighbors = [[r + delta[0], c + delta[1]] for delta in deltas
                             if (r + delta[0] >= 0 and r + delta[0] < len(grid) and
                                 c + delta[1] >= 0 and c + delta[1] < len(grid[0]) and
                                 grid[r + delta[0]][c + delta[1]] != comp_color)]
            if (r == 0 or r == len(grid) - 1 or
                c == 0 or c == len(grid[0]) - 1
                or len(diff_neighbors) > 0):
                border.append([r, c])
            for neighbor in sim_neighbors:
                frontier.append(neighbor)
        return border
            

    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        coords = self.getBorderCoords(grid, r0, c0)
        for r, c in coords:
            grid[r][c] = color
        return grid
