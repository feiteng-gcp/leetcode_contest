class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        R = len(grid)
        C = len(grid[0])
        que = [(r0, c0)]
        c = grid[r0][c0]
        ans = [[-1] * C for i in range(R)]
        used = [[-1] * C for i in range(R)]
        while que:
            x, y = que.pop(0)
            flag = False
            for dx, dy in [(0, 1), (1, 0), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if not (0 <= nx < R and 0 <= ny < C):
                    ans[x][y] = color
                    flag = True
                else:
                    if grid[nx][ny] == c:
                        if used[nx][ny] == -1:                         
                            que.append((nx, ny))
                            used[nx][ny] = 1
                    else:
                        flag = True
                    
            if flag:
                ans[x][y] = color
                
                        
        for i in range(R):
            for j in range(C):
                if ans[i][j] == -1:
                    ans[i][j] = grid[i][j]
        
        return ans
