class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        n = len(grid)
        m = len(grid[0])
        a = grid[r0][c0]
        vis = set()
        def dfs(x, y):
            if 0 <= x < n and 0 <= y < m:
                
                nonlocal a
                if grid[x][y] != a:
                    return
                if (x, y) in vis:
                    return
                vis.add((x, y))
                # print(x, y)
                # grid[x][y] = color
                dfs(x + 1, y)
                dfs(x, y + 1)
                dfs(x - 1, y)
                dfs(x, y - 1)
        dfs(r0, c0)
        for (x, y) in vis:
            if any((a, b) not in vis for (a, b) in [(x + 1, y), (x - 1, y), (x, y - 1), (x, y + 1)]):
                grid[x][y] = color
        return grid
