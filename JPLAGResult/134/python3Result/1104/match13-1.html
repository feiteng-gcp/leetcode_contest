class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        
        if not grid or not grid[0]:
            return grid
        
        m, n = len(grid), len(grid[0])
        
        com_color = grid[r0][c0]
        
        if self.is_border_cell(grid, com_color, (r0,c0)):
            grid[r0][c0] = -1
        else:
            grid[r0][c0] = 0
            
        queue = collections.deque([(r0,c0)])
        while queue:
            i, j = queue.popleft()
            deltas = [(0,1),(0,-1),(1,0),(-1,0)]
            for dx, dy in deltas:
                x, y = i+dx, j+dy
                if 0<=x<m and 0<=y<n and grid[x][y] == com_color:
                    if self.is_border_cell(grid, com_color, (x,y)):
                        grid[x][y] = -1
                    else:
                        grid[x][y] = 0
                    queue.append((x,y))
        
        
for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    grid[i][j] = com_color
                elif grid[i][j] == -1:
                    grid[i][j] = color
                    
        return grid
        
    def is_border_cell(self, grid, com_color, coord):
        
        m, n = len(grid), len(grid[0])
        
        i, j = coord
        
        if i == 0 or i == m-1 or j == 0 or j == n-1:
            return True
        
        deltas = [(0,1),(0,-1),(1,0),(-1,0)]
        for dx, dy in deltas:
            x, y = i+dx, j+dy
            if grid[x][y] not in (com_color, -1, 0):
                return True
        
        return False
