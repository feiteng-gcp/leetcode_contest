class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        if grid[r0][c0]==color: return grid
        borderset = set()
        seen = set()
        stack = [(r0,c0)]
        seen.add((r0,c0))
        m = len(grid)
        n = len(grid[0])
        while len(stack)!=0:
            cpi,cpj = stack.pop()
            for di,dj in [(0,1),(1,0),(0,-1),(-1,0)]:
                npi,npj = cpi+di,cpj+dj
                if not (0<=npi<m and 0<=npj<n): 
                    borderset.add((cpi,cpj))
                elif (0<=npi<m and 0<=npj<n) and grid[npi][npj]!=grid[r0][c0]: 
                    borderset.add((cpi,cpj))
                elif  (0<=npi<m and 0<=npj<n) and grid[npi][npj]==grid[r0][c0] and (npi,npj) not in seen:
                    stack.append((npi,npj))
                    seen.add((npi,npj))
        for i,j in borderset:
            grid[i][j]=color
        return grid
