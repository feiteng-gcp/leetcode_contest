class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        R = len(grid)
        C = len(grid[0])
        original = grid[r0][c0]
        
        def neighbor(i,j):
            for r, c in [[1,0], [0,1], [-1,0], [0,-1]]:
                if 0 <= i+r < R and 0 <= j+c < C:
                    yield i+r, j+c
        
        def is_border(i, j):
            for r, c in [[1,0], [0,1], [-1,0], [0,-1]]:
                if 0 > i+r or i+r >= R or 0 > j+c or j+c >= C or (grid[i+r][j+c] != original and grid[i+r][j+c] != -1 and grid[i+r][j+c] != 0):
                    return True
            return False
        
        def dfs(i, j):
            if is_border(i,j):
                grid[i][j] = -1
            else:
                grid[i][j] = 0
            for row, col in neighbor(i,j):
                if grid[row][col] == original:
                    dfs(row, col)
            
            
        dfs(r0, c0)
        for i in range(R):
            for j in range(C):
                if grid[i][j] == -1:
                    grid[i][j] = color
                if grid[i][j] == 0:
                    grid[i][j] = original
        
        return grid
