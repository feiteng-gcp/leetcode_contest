class Solution:
    
    def helper(self, grid, r, c, ori, color, visited):
        if r < 0 or r >= len(grid) or c < 0 or c >= len(grid[0]) or grid[r][c] != ori:
            return True
        if visited[r][c]:
            return False
        flag = False
        visited[r][c] = True
        if self.helper(grid, r-1, c, ori, color, visited):
            flag = True
        if self.helper(grid, r+1, c, ori, color, visited):
            flag = True
        if self.helper(grid, r, c-1, ori, color, visited):
            flag = True
        if self.helper(grid, r, c+1, ori, color, visited):
            flag = True
        if flag:
            self.cache.add((r,c))
        return False
        
        
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        visited = [[False for _ in range(len(grid[0]))] for _ in range(len(grid))]
        self.cache = set
()
        self.helper(grid, r0, c0, grid[r0][c0], color, visited)
        for (i,j) in self.cache:
            grid[i][j] = color
        return grid
