class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        x_max,y_max = len(grid),len(grid[0])
        origin_color = grid[r0][c0]
        # find connected
        visited = collections.Counter()
        stack = [(r0,c0)]
        while len(stack)>0:
            x,y = stack.pop(0)
            if (x,y) in visited: continue
                
            visited[(x,y)]=0
            for a,b in [(x-1,y),(x+1,y),(x,y-1),(x,y+1)]:
                if 0<=a<x_max and 0<=b<y_max and (a,b) not in visited and grid[a][b]==origin_color:
                    stack.append((a,b))
        # label border, assign value = 1 if is border
        for x,y in visited.keys():
            for a,b in [(x-1,y),(x+1,y),(x,y-1),(x,y+1)]:
                if (a,b) not in visited:
                    visited[(x,y)]=1
                    break
        # color border
        for pair, is_border in visited.items():
            if is_border:
                grid[pair[0]][pair[1]]=color
                
        return grid
