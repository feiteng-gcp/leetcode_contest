class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        ori = grid[r0][c0]
        R, C = len(grid), len(grid[0])
        ans = [[False for i in range(C)] for j in range(R)]
        for i in range(R):
            for j in range(C):
                ans[i][j] = grid[i][j]
        visited = [[False for i in range(C)] for j in range(R)]
        def bfs(r, c):
            if r < 0 or r >= R or c < 0 or c >= C: return 
            if visited[r][c]: return 
            visited[r][c] = True
            isBorder = (r == 0 or r == R-1 or c == 0 or c == C-1)
            if grid[r][c] == ori:
                if r+1 < R and grid[r+1][c] != ori: isBorder = True
                if r-1 >= 0 and grid[r-1][c] != ori: isBorder = True
                if c+1 < C and grid[r][c+1] != ori: isBorder = True
                if c-1 >= 0 and grid[r][c-1] != ori: isBorder = True
                if isBorder:
                    ans[r][c] = color
                bfs(r-1, c)
                bfs(r+1, c)
                bfs(r, c-1)
                bfs(r, c+1)
        bfs(r0, c0)
        return ans
