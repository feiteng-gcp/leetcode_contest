class Solution:
    def colorBorder(self, grid, r0: int, c0: int, color: int):
        seen, border = set(), set()
        m, n, c = len(grid), len(grid[0]), grid[r0][c0]
        def dfs(i, j):
            if (i, j) in seen:
                return
            seen.add((i, j))
            for a, b in [(i-1, j), (i+1, j), (i, j-1), (i, j+1)]:
                if a< 0 or b < 0 or a>=m or b >= n or grid[a][b] != c:
                    border.add((i, j))
                else:
                    dfs(a, b)
                    
        dfs(r0, c0)
        for i, j in border:
            grid[i][j] = color
        return grid
