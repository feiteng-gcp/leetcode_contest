class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        col = grid[r0][c0]
        n,m = len(grid),len(grid[0])
        bfs = [(r0,c0)]
        borders = []
        mark = [[False]*m for _ in range(n)]
        while(len(bfs) != 0 ):
            x,y =  bfs[0][0],bfs[0][1]
            bfs.pop(0)
            if(mark[x][y]==False):
                mark[x][y]=True
                if(x+1<n and grid[x+1][y]==col):
                    bfs.append((x+1,y))
                if(x-1>=0 and grid[x-1][y]==col):
                    bfs.append((x-1,y))
                if(y+1<m and grid[x][y+1]==col):
                    bfs.append((x,y+1))
                if(y-1>=0 and grid[x][y-1]==col):
                    bfs.append((x,y-1))
                border = False
        
                if(x==n-1 or y==m-1 or x*y==0 ):
                    border = True
                if(x+1<n and grid[x+1][y]!=col):
                    border = True
                if(x-1>=0 and grid[x-1][y]!=col):
                    border = True
                if(y+1<m and grid[x][y+1]!=col):
                    border = True
                if(y-1>=0 and grid[x][y-1]!=col):
                    border = True
                if(border==True):
                    borders.append((x,y))
        for i,j in borders:
            grid[i][j]=color
        return grid
