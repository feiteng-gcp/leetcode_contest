class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        queue = [(r0, c0)]
        tmp = []
        visited = {(r0, c0)}
        dx = [1, -1, 0, 0]
        dy = [0, 0, 1, -1]
        m, n = len(grid), len(grid[0])
        while queue:
            x, y = queue.pop(0)
            cnt = 0
            for i in range(4):
                tx, ty = x + dx[i], y + dy[i]
                if 0 <= tx <= m - 1 and 0 <= ty <= n - 1 and grid[tx][ty] == grid[x][y]:
                    cnt += 1
                    if (tx, ty) not in visited:
                        visited.add((tx, ty))
                        queue.append((tx, ty))
            if cnt != 4:
                tmp.append((x, y))
        for x, y in tmp:
            grid[x][y] = color
        return grid
