class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        if not grid:
            return grid
        
        m = len(grid)
        n = len(grid[0])
        flag 
= [[False for j in range(n)] for i in range(m)]
        def dfs(i,j,c1,c2):
            if i < 0 or i > m-1 or j < 0 or j > n-1:
                return True
            
            if flag[i][j]:
                return False
            
            if grid[i][j] != c1:
                return True
            
            flag[i][j] = True
            if dfs(i-1,j,c1,c2) | dfs(i,j-1,c1,c2) | dfs(i+1,j,c1,c2) | dfs(i,j+1,c1,c2):
                grid[i][j] = c2
            
            return False
        
        dfs(r0,c0,grid[r0][c0],color)
        return grid
