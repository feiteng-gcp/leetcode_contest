class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        origColor = grid[r0][c0]
        self.dfs(grid, r0, c0, origColor)
        m, n = len(grid), len(grid[0])
        for i in range(m):
            for j in range(n):
                if grid[i][j] == -1:
                    grid[i][j] = color
                elif grid[i][j] == -2:
                    grid[i][j] = origColor
        return grid
    
    def dfs(self, grid, r, c, origColor):
        m, n = len(grid), len(grid[0])
        if self.isEdge(grid, r, c, origColor):
            grid[r][c] = -1
            # print(r, c)
        else:
            grid[r][c] = -2
        for dr, dc in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
            x, y = r + dr, c + dc
            if 0 <= x < m and 0 <= y < n and grid[x][y] == origColor:
                self.dfs(grid, x, y, origColor)
                
    def isEdge(self, grid, r, c, origColor):
        m, n = len(grid), len(grid[0])
        if r == 0 or r == m - 1 or c == 0 or c == n - 1:
            return True
        elif (grid[r - 1][c] == origColor or grid[r - 1][c] < 0) and (grid[r + 1][c] == origColor or grid[r + 1][c] < 0) and (grid[r][c - 1] == origColor or grid[r][c - 1] < 0) and (grid[r][c + 1] == origColor or grid[r][c + 1] < 0):
            return False
        else:
            return True
