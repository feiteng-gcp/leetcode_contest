class Solution:
    
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        n = len(grid)
        m = len(grid[0])
        dp = [[0]*m for i in range(n)]
        visited= [[0]*m for i in range(n)]
        for i in range(n):
            for j in range(m):
                dp[i][j]=grid[i][j]
        q=[]
        q.append((r0,c0))
        while q:
            x,y = q.pop()
            if visited[x][y]:
                continue
            visited[x][y]=1
            #check if border
            if x==0 or y==0 or x==n-1 or y==m-1:
                dp[x][y] = color
            #check neibour
            cnt=0
            for dx,dy in [[1,0],[-1,0],[0,-1],[0,1]]:
                i = x+dx
                j = y+dy
                if i>=0 and j>=0 and i<n and j<m and grid[i][j]==grid[r0][c0] :
                    if not visited[i][j]:
                        q.append((i,j))
                    cnt+=1
            if cnt<4:
                dp[x][y] = color
        return dp
