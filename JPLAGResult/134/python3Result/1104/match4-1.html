from typing import List


class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        if len(grid) == 0 or len(grid[0]) == 0:
            return []

        m, n = len(grid), len(grid[0])

        component_color = grid[r0][c0]
        if component_color == color:
            return grid

        # find components
        q = [(r0, c0)]
        visited = {(
r0, c0)}
        while len(q) > 0:
            r, c = q.pop()
            for nr, nc in [(r - 1, c), (r + 1, c), (r, c - 1), (r, c + 1)]:
                if 0 <= nr < m and 0 <= nc < n and grid
[nr][nc] == component_color and (nr, nc) not in visited:
                    q.append((nr, nc))
                    visited.add((nr, nc))

        # find borders
        borders = set()
        for r, c in visited:
            if r == 0 or c == 0 or r == m - 1 or c == n - 1:
                borders.add((r, c))
            else:
                is_border = False
                for nr, nc in [(r - 1, c), (r + 1, c), (r, c - 1), (r, c + 1)]:
                    if 0 <= nr < m and 0 <= nc < n and grid[nr][nc] != component_color:
                        is_border = True
                        break
                if is_border:
                    borders.add((r, c))

        ret_grid = [[grid[i][j] if (i, j) not in borders else color for j in range(n)] for i in range(m)]
        return ret_grid
