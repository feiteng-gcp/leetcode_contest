class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        border = []
        m = len(grid)
        n = len(grid[0])
        visited = [[0 for _ in range(n)] for _ in range(m)]
        self.findBorder(grid, visited, r0, c0, border, m, n, grid[r0][c0])
        for c in border:
            grid[c[0]][c[1]] = color
        return grid
        
    
    def findBorder(self, grid, visited, i, j, res, m, n, target):
        if i < 0 or i >= m or j < 0 or j >= n or visited[i][j] == 1 or grid[i][j] != target:
            return
        if i == 0 or i == m - 1 or j == 0 or j == n - 1 \
        or (grid[i - 1][j] != target or grid[i][j - 1] != target or grid[i + 1][j] != target or grid[i][j + 1] != target):
            visited[i][j] = 1
            res.append((i, j))
            self.findBorder(grid, visited, i - 1, j, res, m, n, target)
            self.findBorder(grid, visited, i, j - 1, res, m, n, target)
            self.findBorder(grid, visited, i + 1, j, res, m, n, target)
            self.findBorder(grid, visited, i, j + 1, res, m, n, target)
            return
        else:
            visited[i][j] = 1
            self.findBorder(grid, visited, i - 1, j, res, m, n, target)
            self.findBorder(grid, visited, i, j - 1, res, m, n, target)
            self.findBorder(grid, visited, i + 1, j, res, m, n, target)
            self.findBorder(grid, visited, i, j + 1, res, m, n, target)
            return
