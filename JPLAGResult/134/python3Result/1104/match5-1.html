class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        toColor = []
        queue = [(r0, c0)]
        visited = set(queue)
        while queue:
            r, c = queue.pop()

            if self.isBorder(grid, r, c):
                toColor.append((r,c))
            for dx, dy in [(0, 1), (0, -1), (-1, 0), (1, 0)]:
                r_ = r + dx
                c_ = c + dy
                if self.isValid(grid, r_, c_) and (
r_, c_) not in visited and grid[r_][c_] == grid[r][c]:
                    queue.append((r_, c_))
                    visited.add((r_, c_))
               
        for r, c in toColor:
            grid[r][c] = color
            
        return grid
    
    
    def isBorder(self, grid, r, c):
        m, n = len(grid), len(grid[0])
        if r == 0 or r == m-1 or c == 0 or c == n-1:
            return True
        
        for dr, dc in [(0, 1), (0, -1), (-1, 0), (1, 0)]:
            r_, c_ = r + dr, c + dc
            if grid[r_][c_] != grid[r][c]:
                return True
            
        return False
    
    
    def isValid(self, grid, r, c):
        m, n = len(grid), len(grid[0])
        return 0 <= r < m and 0 <= c < n
