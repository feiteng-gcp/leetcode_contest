def connect_component(A, i, j):
    x = A[i][j]
    index = [(m, n) for m in range(len(A)) for n in range(len(A[0])) if A[m][n] == x]
    S1 = {(i, j)}
    S0 = set()
    index.remove((i, j))
    while len(S1) > 0:
        a = S1.pop()
        S0.add(a)
        if (a[0] + 1, a[1]) in index:
            S1.add((a[0] + 1, a[1]))
            index.remove((a[0] + 1, a[1]))
        if (a[0] - 1, a[1]) in index:
            S1.add((a[0] - 1, a[1]))
            index.remove((a[0] - 1, a[1]))
        if (a[0], a[1] + 1) in index:
            S1.add((a[0], a[1] + 1))
            index.remove((a[0], a[1] + 1))
        if (a[0], a[1] - 1) in index:
            S1.add((a[0], a[1] - 1))
            index.remove((a[0], a[1] - 1))
    return S0


def border(S):
    S1 = set()
    for s0 in S:
        if (s0[0] + 1, s0[1]) in S and (s0[0] - 1, s0[1]) in S and (s0[0], s0[1] + 1) in S and (s0[0], s0[1] - 1) in S:
            S1.add(s0)
    return S - S1


class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        S = connect_component(grid, r0, c0)
        index = border(S)
        for s0 in index:
            grid[s0[0]][s0[1]] = color
        return grid
