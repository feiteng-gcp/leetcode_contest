class Solution:
    def colorBorder(self, grid: List
[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        m, n = len(grid), len(grid[0])
        stack = [(r0, c0)]
        visited = set()
        ocolor = grid[r0][c0]
        while stack:
            x, y = stack.pop()
            visited.add((x, y))
            found = False
            for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                nx, ny = x + dx, y + dy
                if nx < 0 or nx >= m or ny < 0 or ny >= n or (ocolor != grid[nx][ny] and (nx, ny) not in visited):
                    found = True
                    continue
                if (nx, ny) in visited:
                    continue
                stack.append((nx, ny))
            if x == 0 or y == 0 or x == m-1 or y == n-1 or found:   
                grid[x][y] = color
        return grid
