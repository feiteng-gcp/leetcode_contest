class Solution:
    def colorBorder(self, grid: List[List[int]], r0: int, c0: int, color: int) -> List[List[int]]:
        q = collections.deque()
        q.append((r0,c0))
        m,n = len(grid), len(grid[0])
        visited = [[0 for _ in range(n)] for _ in range(m)]
        visited[r0][c0] = True
        ori_color = grid[r0][c0] 
        def is_border(r,c):
            if r==0 or r==m-1:
                return True
            if c == 0 or c == n-1:
                return True
            
for d in [[0,1], [0,-1],[1,0], [-1,0]]:
                if 0 <= r+d[0] < m and 0 <= c+d[1] < n and grid[r+d[0]][c+d[1]] != ori_color:
                    return True
            return False
        out = [[None for _ in range(n)] for _ in range(m)]
        for r in range(m):
            for c in range(n):
                out[r][c] = grid[r][c]
                
        while q:
            r, c = q.popleft()
            if is_border(r,c):
                out[r][c] = color
            for d in [[0,1], [0,-1],[1,0], [-1,0]]:
                if 0 <= r+d[0] < m and 0 <= c+d[1] < n and grid[r+d[0]][c+d[1]] == ori_color and not visited[r+d[0]][c+d[1]]:
                    q.append((r+d[0], c+d[1]))
                    visited[r+d[0]][c+d[1]] = True
        return out
