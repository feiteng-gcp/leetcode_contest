class Solution:
    def maxUncrossedLines(self, A: List[int], B: List[int]) -> int:
        n,m = len(A),len(B)
        DP=[[0]*(m+1)  for _ in range(n+1)]
        for i in range(1,n+1):
            for j in range(1,m+1):
                DP[i][j] = max(DP[i-1][j],DP[i][j-1])
                if(A[i-1]==B[j-1]):
                    DP[i][j] = max(DP[i][j],DP[i-1][j-1]+1)
        return DP[-1][-1]
