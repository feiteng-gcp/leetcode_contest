class Solution:
    def maxUncrossedLines(self, A: List[int], B: List[int]) -> int:
        n = len(A)
        m = len(B)
        best = [[0] * m for _ in range(n)]
        matched = False
        for i in reversed(range(m)):
            if A[n - 1] == B[i] or matched:
                best[n - 1][i] = 1
                matched = True
        matched = False
        for i in reversed(range(n)):
            if A[i] == B[m - 1] or matched:
                best[i][m - 1] = 1
                matched = True
        
for i in range(n - 2, -1, -1):
            for j in range(m - 2, -1, -1):
                if A[i] == B[j]:
                    best[i][j] = max(best[i+1][j+1] + 1, best[i][j+1], best[i+1][j])
                else:
                    best[i][j] = max(best[i+1][j+1], best[i][j+1], best[i+1][j])
        return best[0][0]
