class Solution(object):
    def maxUncrossedLines(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: int
        """
        s1 = set(A)
        s2 = set(B)
        s = s1 & s2
        A = [_ for _ in A if _ in s]
        B = [_ for _ in B if _ in s]
        l1, l2 = len(A), len(B)
        if min(l1,l2) <= 1:
            return 1 if s else 0

        ans = [[0]*l2 for i in range(l1)]
        ans[0] = [int(A[0] in B[:k]) for k in range(1,l2+1)]
        for k in range(1, l1 + 1):
            if B[0] in A[:k]:
                ans[k-1][0] = 1
        for i in range(1,l1):
            for j in range(1,l2):
                i_tmp, j_tmp = i,j
                while i_tmp >= 0:
                    if A[i_tmp] != B[j]:
                        i_tmp -= 1
                    else:
                        break
                while j_tmp >= 0:
                    if B[j_tmp] != A[i]:
                        j_tmp -= 1
                    else:
                        break
                compare_tmp = [ans[i-1][j], ans[i][j-1]]
                if i_tmp >0:
                    compare_tmp.append(1 + ans[i_tmp-1][j-1])
                if j_tmp >0:
                    compare_tmp.append(1 + ans[i-1][j_tmp-1])
                ans[i][j] = max(compare_tmp)
        return ans[-1][-1]
