class Solution:
    def maxUncrossedLines(self, A: List[int], B: List[int]) -> int:
        na,nb=len(A),len(B)
        dp=[[0] * (na+1) for _ in range(nb+1)]
        for ib in range(1, nb+1):
            for ia in range(1, na+1):
                if A[ia-1]==B[ib-1]: dp[ib][ia]=1+dp[ib-1][ia-1]
                dp[ib][ia]=max(dp[ib][ia], dp[ib-1][ia], dp[ib][ia-1])
        return dp[nb][na]
