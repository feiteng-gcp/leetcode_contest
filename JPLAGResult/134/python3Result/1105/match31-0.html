class Solution:
    def maxUncrossedLines(self, A: List[int], B: List[int]) -> int:
        la,lb=len(A),len(B)
        dp=[[0]*(lb+1) for _ in range(la+1)]
        for i in range(la):
            for j in range(lb):
                if A[i]==B[j]:
                    dp[i+1][j+1]=max(dp[i][j]+1,dp[i][j+1],dp[i+1][j])
                else:
                    dp[i+1][j+1]=max(dp[i][j],dp[i][j+1],dp[i+1][j])
        return dp[-1][-1]
