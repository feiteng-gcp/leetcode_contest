class StockSpanner:

    def __init__(self):
        self.li = []
        self.le = 0

    def next(self, price):
        """
        :type price: int
        :rtype: int
        """
        if self.le == 0 or price<self.li[-1][0]:
            self.li.append((price, self.le))
        else:
            num = self.li[-1][1]
            if num>0:
                num -= 1
            while num>=0 and self.li[num][0]<=price:
                num = self.li[num][1]
                num -= 1
            self.li.append((price, num+1))
        self.le += 1
        return self.le-self.li[-1][1]
