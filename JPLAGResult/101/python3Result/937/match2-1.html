class StockSpanner:

    def __init__(self):
        self.A = []
        self.result = []

    def next(self, price):
        """
        :type price: int
        :rtype: int
        """
        if self.A == [] or price < self.A[-1]:
            self.A.append(price)
            self.result.append(1)
            return 1
        else:
            idx = len(self.A)-1
            current = self.A[idx]
            count = self.result[idx]
            while(idx>=0 and current <= price):
                idx -= count
                current = self.A[idx]
                count = self.result[idx]
            if idx < 0:
                self.A.append(price)
                self.result.append(len(self.A))
                return len(self.A)
            else:
                self.A.append(price)
                self.result.append(len(self.A)-idx-1)
                return len(self.A)-idx-1
