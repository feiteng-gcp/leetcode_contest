class StockSpanner:

    def __init__(self):
        self.spans = []

    def next(self, price):
        """
        :type price: int
        :rtype: int
        """
        if not self.spans or price < self.spans[-1][0]:
            self.spans.append((price, 1))
            return 1
        i = len(self.spans) - 1
        total = 1
        while i >= 0 and price >= self.spans[i][0]:
            total += self.spans[i][1]
            i -= self.spans[i][1]
        self.spans.append((price, total))
        return total
