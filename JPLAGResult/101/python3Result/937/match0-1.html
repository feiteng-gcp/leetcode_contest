class StockSpanner:

    def __init__(self):

        self.list = []

    def next(self, price):
        """
        :type price: int
        :rtype: int
        """

        i 
= len(self.list) - 1
        result = 1
        while i >= 0 and self.list[i][0] <= price:
            result += self.list[i][1]
            i -= self.list[i][1]

        self.list.append((price, result))

        return result
