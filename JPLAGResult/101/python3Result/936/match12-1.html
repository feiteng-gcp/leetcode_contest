class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self
.A = A

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        if self.A == []:
            return -1
        if self.A[0] == 0:
            self.A = self.A[2:]
            return self.next(n)
        if n == 1:
            self.A[0] -= 1
            return self.A[1]
        else:
            if n > self.A[0]:
                newn = n-self.A[0]
                self.A = self.A[2:]
                return self.next(newn)
            else:
                self.A[0] -= n-1
                return self.next(1)
