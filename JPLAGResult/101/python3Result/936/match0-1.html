class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.A = A
        self.ptr = 0
        

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        
while(self.ptr < len(self.A) and self.A[self.ptr] < n):
            n -= self.A[self.ptr]
            self.ptr += 2
        if(self.ptr >= len(self.A)):
            return -1
        elif(self.A[self.ptr] == n):
            self.ptr += 2
            return self.A[self.ptr-1]
        else:
            self.A[self.ptr] -= n
            return self.A[self.ptr+1]
