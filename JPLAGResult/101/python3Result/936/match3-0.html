class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.q = A
        
        
    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        res = -1
        while len(self.q) > 0 and n > 0:
            if self.q[0] == 0:
                self.q.pop(0)
                self.q.pop(0)
            elif self.q[0] > n:
                self.q[0] -= n
                res = self.q[1]
                n = 0
            elif self.q[0] == n:
                n -= self.q.pop(0)
                res = self.q.pop(0)
            else:
                n -= self.q.pop(0)
                self.q.pop(0)
        
        return res
