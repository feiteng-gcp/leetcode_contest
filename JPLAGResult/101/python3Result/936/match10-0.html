class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self._Ap = []
        self._i = 0
        self._n = 0
        
        accum = 0
        for i in range(len(A) // 2):
            accum += A[i * 2]
            self._Ap += [(accum, A[(i * 2) + 1])]

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        self._n += n
        while self._i < len(self._Ap) and self._n > self._Ap[self._i][0]:
            self._i += 1
        return self._Ap[self._i][1] if self._i < len(self._Ap) else -1
