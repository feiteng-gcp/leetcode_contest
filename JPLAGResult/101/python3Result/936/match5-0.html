class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.li = []
        for i in range(len(A)//2-1, -1, -1):
            self.li.append([A[2*i+1], A[2*i]])

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        if len(self.li)==0 and n>0:
            return -1
        if self.li[-1][1]>=n:
            self.li[-1][1]-=n
            return self.li[-1][0]
        else:
            n -= self.li[-1][1]
            self.li.pop()
            return self.next(n)
