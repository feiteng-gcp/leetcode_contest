class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.A = A
        self.rem = 0
        self.idx = 0

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        self.rem -= n
        while self.rem < 0:
            if self.idx >= len(self.A):
                return -1
            self.rem += self.A[self.idx]
            self.idx += 2
        # print(n, self.idx, self.rem)

        return self.A[self.idx-1]
