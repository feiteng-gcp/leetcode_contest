class RLEIterator:
    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.A = A
        self.L = len(A)
        self.prev = 0
        self.remain = self.A[0] if A != [] else 0
        self.flag = 0
        
    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        if self.flag == 1:
            return -1
        if self.L == 0 or self.prev == self.L or (self.prev == self.L-2 and n > self.remain):
            self.flag = 1
            return -1
        if n < self.remain:
            self.remain = self.remain - n
            return self.A[self.prev + 1]
        elif n == self.remain:
            self.remain = self.A[self.prev + 2] if self.prev + 2 < self.L else 0
            self.prev = self.prev + 2
            return self.A[self.prev - 1]
        else:
            cum_cnt = self.remain
            for i in range(self.prev + 2, self.L, 2):
                if self.A[i] == 0:
                    continue
                cum_cnt += self.A[i]
                if cum_cnt == n:
                    self.prev = i + 2
                    self.remain = self.A[i+2] if i+2 < self.L else 0
                    return self.A[i+1]
                elif cum_cnt > n:
                    self.remain = cum_cnt - n
                    self.prev = i
                    return self.A[i+1]
            self.flag = 1
            return -1
