class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.nums = A[::2]
        self.digits = A[1::2]


    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n <= 0:
            if len(self.nums) != 0:
                return self.digits[0]
            else:
                return -1
        ans = -1
        while n > 0:
            if len(self.nums) == 0:
                return -1
            if self
.nums[0] < n:
                n -= self.nums[0]
                self.nums.pop(0)
                self.digits.pop(0)
            elif self.nums[0] == n:
                n = 0
                ans = self.digits[0]
                self.nums.pop(0)
                self.digits.pop(0)
            else:
                self.nums[0] -= n
                ans = self.digits[0]
                n = 0
        
        return ans
