class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.x = []
        for i in range(len(A)//2):
            self.x.append((A[2*i], A[2*i + 1]))
        self.cur_ind = 0
        self.cur_rep = 0
    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        new_n = n + self.cur_rep
        while self.cur_ind < len(self.x) and new_n > self.x[self.cur_ind][0]:
            new_n -= self.x[self.cur_ind][0]
            self.cur_ind += 1
        if self.cur_ind == len(self.x):
            return -1
        else:
            self.cur_rep = new_n
            return self.x[self.cur_ind][1]
