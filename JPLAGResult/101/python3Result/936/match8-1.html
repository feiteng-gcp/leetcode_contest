class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """

        n = len(A) // 2
        self.indexes = []
        self.values = []

        prefix = 0
        for i in range(n):
            i1 = 2 * i
            i2 = i1 + 1
            if A[i1] == 0: continue

            prefix += A[i1]
            self.indexes.append(prefix)
            self.values.append(A[i2])

        self.count = 0


    def next(self, n):
        """
        :type n: int
        :rtype: int
        """

        index = self.count + n
        self.count = index

        N = len(self.indexes)

        if index > self.indexes[-1]: return -1

        lo = 0
        hi = N-1
        while lo<hi:
            mid = lo + (hi-lo) // 2
            if self.indexes[mid] < index:
                lo = mid + 1
            else:
                hi = mid

        return self.values[lo]
