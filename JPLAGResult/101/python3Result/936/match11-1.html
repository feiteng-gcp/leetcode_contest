class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.arr = A
        self.curr_count_ind = 0
        

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        count = n
        curr_val = -1
        while count > 0:
            if self.curr_count_ind < len(self.arr):
                curr_val = self.arr[self.curr_count_ind + 1]
                if count > self
.arr[self.curr_count_ind]:
                    count -= self.arr[self.curr_count_ind]
                    self.arr[self.curr_count_ind] = 0
                    self.curr_count_ind += 2
                else:
                    self.arr[self.curr_count_ind] -= count
                    count = 0
            else:
                return -1
        return curr_val
