class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.seq = []
        self.count = [0]
        self.pos = -1
        self.ind = 0
        for i in range(len(A)//2):
            if A[2*i]>0:
                self.seq.append(A[2*i+1])
                self.count.append(self.count[-1]+A[2*i])
        self.count.pop(0)

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        if self.pos+n < self.count[-1]:
            self.pos += n
            while self
.count[self.ind]-1 < self.pos:
                self.ind += 1
            return self.seq[self.ind]
        else:
            self.pos = self.count[-1]
            return -1
