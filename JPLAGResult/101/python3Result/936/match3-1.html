class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.A=A[::-1]
        

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        res=-1
        while self
.A:
            if self.A[-1]==0:
                self.A.pop()
                self.A.pop()
            elif self.A[-1]<n:
                n-=self.A[-1]
                self.A.pop()
                self.A.pop()
            else:
                self.A[-1]-=n
                res=self.A[-2]
                break
        return res
