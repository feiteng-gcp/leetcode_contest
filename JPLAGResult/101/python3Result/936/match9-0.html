class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.A = []
        for i in range(len(A)//2):
            if A[i*2] > 0:
                self.A.extend([A[i*2], A[i*2+1]])

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        while n > 0:
            if not self.A:
                return -1
            if n < self.A[0]:
                self.A[0] -= n
                return self.A[1]
            elif n == self.A[0]:
                ret = self.A[1]
                self.A = self.A[2:]
                return ret
            else:
                n -= self.A[0]
                self.A = self.A[2:]
