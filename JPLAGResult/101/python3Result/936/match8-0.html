from bisect import bisect_left

class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        n = len(A)
        self.cnt = 0
        self.B = []
        cnt = 0
        for i in range(0, n, 2):
            cnt += A[i]
            self.B.append((cnt, A[i+1]))

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        self.cnt += n
        idx = bisect_left(self.B, (self.cnt, -1))
        if idx == len(self.B): return -1
        return self.B[idx][1]
