class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.seq = A
        self.i = 0
            
    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        if self.i >= len(self.seq):
            return -1
        else:
            while self.i < len(self.seq) and self.seq[self.i] < n:
                n -= self.seq[self.i]
                self.i += 2
            
            if self.i >= len(self.seq):
                return -1
            else:
                # here n is remaining amount to exhaust
                self.seq[self.i] -= n
                return self.seq[self.i + 1]
