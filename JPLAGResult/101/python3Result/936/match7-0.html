class RLEIterator:

    def __init__(self, A):
        """
        :type A: List[int]
        """
        self.data = A
        

    def next(self, n):
        """
        :type n: int
        :rtype: int
        """
        res = -1
        while n > 0 and len(self.data) > 0:
            if self.data[0] <= n:
                n -= self.data[0]
                if self.data[0] > 0:
                    res = self.data[1]
                else:
                    res = -1
                del self.data[0]
                del self.data[0]
            else:
                self.data[0] -= n
                res = self.data[1]
                n = 0
        if n > 0:
            return -1
        return res
