class Solution {
public:
    vector<int> a;

    int p[10];
    int solve(int m, int len) {
        //cout << "solve: " << m << endl;
        if (m == 0) return len == 0;
        int sum = 1;
        for (int k = 1; k < len; ++k) sum *= 10;
        int t = m / sum;
        int n = a.size(), ret = 0;
        for (int i = 0; i < n; ++i) {
            if (a[i] < t) {
                ret += p[len - 1];
            } else if (a[i] == t) {
                ret += solve(m - t * sum, len - 1);
                break;
            } else {
                break;
            }
        }
        return ret;
    }
    int atMostNGivenDigitSet(vector<string>& s, int m) {
        int n = s.size();
        a.resize(n);
        for (int i = 0; i < n; ++i) {
            stringstream ss(s[i]);
            ss >> a[i];
        }
        sort(a.begin(), a.end());
        p[0] = 1;
        for (int i = 1; i < 10; ++i) p[i] = p[i - 1] * n;
        int len = 0;
        for (int t = m; t; t /= 10) ++len;
        int ret = 0;
        for (int i = 1; i < len; ++i) ret += p[i];
        ret += solve(m, len);
        return ret;
    }
};
