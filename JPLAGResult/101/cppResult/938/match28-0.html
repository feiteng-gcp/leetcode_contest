const ll MOD=1e9+7;
class Solution {
public:
  int atMostNGivenDigitSet(vector<string>& D, int N) {
    int K=D.size();
    if(K==0) return 0;
    string s=to_string(N);
    int res=0;
    N=s.size();
    bool good=1;
    for(int i=0;i<N;i++){
      char d=s[i];
      int cnt=0;
      good=0;
      for(auto dd:D){
        if(dd[0]<d) cnt++;
        else if(dd[0]==d){
          good=1;
        }
      }
      int tmp=cnt;
      for(int j=0;j<N-1-i;j++) tmp*=K;
      res+=tmp;
      if(good==0){
        break;
      }
    }
    if(good) res++;
    int tmp=K;
    for(int i=0;i<N-1;i++){
      res+=tmp;
      tmp*=K;
    }
    return res;
  }
};
