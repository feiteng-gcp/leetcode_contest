class Solution {
public:
    vector<string>d;
    vector<int>v;
    
    int digits(int n){
        int cont=0;
        v.clear();
        
        while(n!=0){
            cont++;
            v.push_back(n%10);
            n/=10;
        }
        
        reverse(v.begin(),v.end());
        
        return cont;
    }
    
    int memo[35][2];
    
    int dp(int pos,bool menor){
        if(pos==v.size()){
            return 1;    
        }
        
        if(memo[pos][menor]!=-1)return memo[pos][menor];
        int dev=0;
        
        if(menor){
            dev+=d.size()*dp(pos+1,menor);
        }else{
            
            int value=v[pos];
            for(int i=0;i<d.size();i++){
                int vald= d[i][0]-'0';
                
                if(value>vald){
                    dev+=dp(pos+1,1);
                }
                
                if(value==vald){
                    dev+=dp(pos+1,0);
                }
                
            }
            
        }
        
        memo[pos][menor]=dev;
        return dev;
    }
    
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        set<string>S(D.begin(),D.end());
        d=vector<string>(S.begin(),S.end());
        sort(d.begin(),d.end());

        int t= digits(N);
        
        int power=d.size();
        int dev=0;
        for(int i=0;i<t-1;i++){
            dev+=power;
            power*=(d.size());
        }
        
        memset(memo,-1,sizeof(memo));
        return dev+dp(0,0);
    }
};
