class Solution
{
public:
    int a[20];
    int dig[10];
    int atMostNGivenDigitSet(vector<string>& D, int N)
    {
        int cnt = 0;
        while (N)
        {
            a[cnt++] = N % 10;
            N /= 10;
        }
        memset(dig, 0, sizeof(dig));
        for (int i = 0; i < D.size(); ++i)
        {
            dig[D[i][0] - 48] = 1;
        }
        for (int i = 1; i < 10; ++i)
        {
            dig[i] += dig[i - 1];
        }
        int ans = 0;
        for (int i = 0; i < cnt - 1; ++i)
        {
            int anss = 1;
            for (int j = i + 1; j < cnt; ++j)
            {
                anss *= dig[9];
            }
            ans += anss;
        }

        bool flag = true;
        for (int i = cnt - 1; i >= 0; --i)
        {
            int anss = dig[a[i] - 1];
            for (int j = 0; j < i; ++j)
            {
                anss *= dig[9];
            }
            ans += anss;
            if (dig[a[i]] == dig[a[i] - 1])
            {
                flag = false;
                break;
            }
        }
        if (flag) ++ans;
        return ans;
    }
};
