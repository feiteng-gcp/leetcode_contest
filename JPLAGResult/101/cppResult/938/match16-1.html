class Solution {
public:
    vector<int> dig;
    int cal(int c, int len) {
        //if (len == 0) return 0;
      int ret = 1;
      for (int i = 0; i < len; i++) ret *= c;
      return ret;
    }

    void dfs(int cur, int N, vector<string>& D, int& ret) {
      if (cur >= N) {ret++; return;}
      for (int i = 0; i < D.size(); i++) {
        if (D[i][0]-'0' < dig[cur]) ret += cal(D.size(), N-cur-1);
        else if (D[i][0]-'0' == dig[cur]) dfs(cur+1, N, D, ret);
        else break;
      }
    }

    int atMostNGivenDigitSet(vector<string>& D, int N) {
      int n = N;

      while (n) {
        dig.push_back(n % 10);
        n /= 10;
      }
      reverse(dig.begin(), dig.end());

      int ret = 0;
      for (int i = 1; i < dig.size(); i++) ret += cal(D.size(), i);
      dfs(0, dig.size(), D, ret);
      return ret;
    }
};
