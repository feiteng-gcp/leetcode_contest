class Solution {
public:
    int F[10];
    int G[20];
    int ans = 0;
    int atMostNGivenDigitSet(vector<string>& D, int N) {

        int len = 0;
        int tmpn = N;
        while (tmpn > 0) {
            len = len + 1;
            tmpn = tmpn /10;
        }
        G[0] = 1;
        for (int i = 1; i <= 10; i++) G[i] = G[i-1] * D.size();
        int prev = 1;
        tmpn = N;
        int ans = 0;
        for (int i = 0; i < len; i++) {
            int sum = 0;
            int num = tmpn%10;
            tmpn = tmpn/10;
            for (int j = 0; j < D.size(); j++) {
                if (D[j][0] - '0' < num) sum = sum + G[i];
                if (D[j][0] - '0' == num) sum = sum + prev;
                if (D[j][0] - '0' > num) break;
            }
            prev = sum;
            if (i != len-1) ans = ans + G[i+1];
        }
        ans = ans + prev;
        return ans;        
    }
};
