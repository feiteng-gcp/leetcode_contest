class Solution {
public:
    int d[15],n[15],pwr[15];
    int count(int i){
        if(i==0)return 1;
        int x=lower_bound(d+1,d+d[0]+1,n[i])-d;
        int ans=(x-1)*pwr[i-1];
        if(d[x]==n[i])ans+=count(i-1);
        return ans;
    }
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        d[0]=D.size();
        for(int i=0;i<d[0];++i)
            d[i+1]=stoi(D[i]);
        sort(d+1,d+d[0]+1);
        
        for(n[0]=0;N;N/=10)n[++n[0]]=N%10;
        

        pwr[0]=1;
        for(int i=1;i<=n[0];++i)pwr[i]=pwr[i-1]*d[0];
        
        // printf("d=");
        // for(int i=1;i<=d[0];++i)printf("%d%c",d[i]," \n"[i==d[0]]);
        // printf("n=");
        // for(int i=1;i<=n[0];++i)printf("%d%c",n[i]," \n"[i==n[0]]);
        // printf("pwr=");
        // for(int i=1;i<=n[0];++i)printf("%d%c",pwr[i]," \n"[i==n[0]]);
        
        int ans=0;
        for(int i=1;i<n[0];++i)ans+=pwr[i];
        return count(n[0])+ans;
    }
};
