class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        vector<int> digits;
        int number = N;
        while(number != 0){
            digits.push_back(number % 10);
            number /= 10;
        }
        reverse(digits.begin(),digits.end());
        int len = digits.size();
        int total = 0;
        int equl;
        int small;
        for(int i = 1; i <= len - 1; i++){
            int local = 1;
            for(int j = 0; j < i; j++){
                local = local*D.size();
            }
            total += local;
        }
        for(int i = 0; i < len ; i++){
            int ls = 0;
            int le = 0;
            for(int j = 0; j < D.size(); j++){
                if(D[j][0] - '0' < digits[i]){
                   ls ++;
                }
                else if(D[j][0] - '0' == digits[i]){
                   le ++;
                }
            }
            if(i == 0){
               small = ls;
               equl = le;
            }
            else{
               small = small*D.size() + equl*ls;
               equl = equl*le;
            }
        }
        return total + small + equl;
    }
};
