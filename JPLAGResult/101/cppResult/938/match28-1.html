#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
      string NS = to_string(N);
      int ans = 0, i = 0;
      for (; i < NS.size(); ++i) {
        int p = 0;
        char c = NS[i];
        for (; p < D.size(); ++p) {
          if (D[p][0] >= c) {
            break;
          }
        }
        int v = p;
        for (int j = i + 1; j < NS.size(); ++j) {
          v *= D.size();
        }
        ans += v;
        if (p == D.size() || D[p][0] != c) {
          break;
        }
      }
      if (i == NS.size()) {
        ++ans;
      }

      int v = D.size();
      for (int i = 1; i < NS.size(); ++i) {
        ans += v;
        v *= D.size();
      }
      return ans;
    }
};
