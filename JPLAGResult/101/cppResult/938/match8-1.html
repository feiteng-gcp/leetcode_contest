class Solution {
public:
    int get_num(vector<string>& D, char c) {
        int id = -1;
        for(int i = 0; i < D.size(); i++) {
            if(D[i] >= string(1, c)) break;
            id = i;
        }
        return id + 1;
    }
    
    
    int atMostNGivenDigitSet(vector<string>& D, int N) {

        string str = to_string(N);
        int len = str.length();
        int ans = 0;
        int tmp = D.size();
        for(int i = 1; i < len; i++) {
            ans += tmp;
            tmp *= D.size();
        }
        for(int i = 0; i < len; i++) {
            int no = get_num(D, str[i]);
            ans += no * pow(D.size(), len - i - 1);
            if(D[no] != string(1, str[i])) break;
            if(i == len - 1) ans++;
        }
        
        return ans;
    }
};
