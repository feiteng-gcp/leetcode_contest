class Solution {
    int f(vector<char> & D, string num) {
        if(num.size() == 0)
            return 1;
        int res = 0;
        int count_less = 0, count_equal = 0;
        for(auto c : D) {
            if(c < num[0])
                count_less++;
            else if(c == num[0])
                count_equal++;
        }
        res += f(D, string(num.size() - 1, '9')) * count_less;
        res += f(D, string(num.begin() + 1, num.end())) * count_equal;
        return res;
    }
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        vector<char> DD;
        for(auto & s : D)
            DD.push_back(s[0]);
        string num;
        while(N) {
            num.push_back('0' + N % 10);
            N /= 10;
        }
        reverse(num.begin(), num.end());
        int res = f(DD, num);
        for(int i = num.size() - 1; i >= 1; i--)
            res += f(DD, string(i, '9'));
        return res;
    }
};
