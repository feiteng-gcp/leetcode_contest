class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        string s = to_string(N);
        
        // less digit

        int c = 0;
        int d = 1;
        for (int i = 1; i < s.size(); ++i) {
            d *= D.size();
            c += d;
        }
       // printf("%d\n",c);
        for (int i = 0; i < s.size(); ++i) {
            string cur = s.substr(i, 1);
            
            auto it = lower_bound(D.begin(), D.end(), cur);
            
            int k = distance(D.begin(), it);
            int b = k != D.size() && D[k] == cur;
          //  printf(" %s %d %d\n", cur.c_str(), k, b);
            c += k * d;
         //   printf("%d\n",c);
            d /= D.size();
            if (!b) {
                break;
            }
            
            if (b && i + 1 == s.size()) {
                ++c;
            }
        }
        
        return c;
        
    }
};
