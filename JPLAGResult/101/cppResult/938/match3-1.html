class Solution {
public:
    vector<string> v;
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        v = D;
        int order = 0;
        //order is the number of digits in max number ex. order(999)=3
        long tens = 1;
        //tens is the first multiple of 10 greater than N;
        int result = 0;
        while (N/tens>0){
            order++;
            tens*=10;
        }
        tens/=10;
        for (int i = order - 1; i >0; --i){
            result+=pow(v.size(),i);
        }
        int temp =recurse(N,tens,order);
        return result+temp;
    }
    int recurse(int N, long t, int o){
        if (o==0){
            return 0;
        }
        int dig = (N%(t*10))/t;
        if (dig==0){
            return 0;
        }
        int result = 0;
        for (int i = 0; i < v.size(); ++i){
            int c = v[i][0]-'0';

            if (c<dig){
                result+=pow(v.size(),o-1);
            }
            else if(c==dig){
                if (o==1){
                    result+=1;
                }
                else{
                    result+=recurse(N,t/10,o-1);                    
                }
            }
        }
        return result;
    }
};
