class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        int n = D.size();
        auto s = to_string(N);
        int m = s.size();
        int res = 0;
        int factor = 1;
        
        // Any length less than m is ok
        for (int i=1; i<m; ++i) {
            factor *= n;
            res += factor;
        }
        
//        cout << m << ";" << n << ";" << res << endl;
        
        for (int i=0; i<m; ++i) {
            auto ch = s.substr(i, 1);
            
            int cnt = 0;
            bool equal = false;
            for (int j=0; j<n; ++j) {
                if (D[j] < ch) {cnt++; continue;}
                if (D[j] == ch) equal = true;
                break;
            }
            
            res += cnt * pow(n, m - i - 1);
            if (equal && i == m-1) res++;
            if (equal) continue;
            return res;
        }
        return res;
        
    }
};
