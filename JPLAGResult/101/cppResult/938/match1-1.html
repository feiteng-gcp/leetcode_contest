class Solution {
public:
    
    vector <string> A;
    vector <string> num;
    int dp[15][5];
    /// state == 1      larger
    /// state == 0      equal
    /// state == 2      smaller 
    int rec(int idx, int state) {
        if(state == 1 and idx == num.size() - 1) {
            return 1;
        }
        
        if(idx == num.size()) {
            return state != 1;
        }
        
        int &ret = dp[idx][state];
        if(ret != -1) {
            return ret;
        }
        

        ret = 1;
        for(int i = 0;i < A.size();++i) {
            if(A[i] == num[idx]) {
                
                ret += rec(idx + 1, state);
                
            } else if(A[i] > num[idx]) {
                
                if(state == 2) {
                    ret += rec(idx + 1, 2);
                } else {
                    ret += rec(idx + 1, 1);
                }
                
            } else {
                
                if(state == 1) {
                    ret += rec(idx + 1, 1);
                } else {
                    ret += rec(idx + 1, 2);
                }
                
            }
        }
        return ret;
    }
    
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        A = D;
        string x = to_string(N);
        for(int i = 0;i < x.size();++i) {
            string y;
            y += x[i];
            num.push_back(y);
        }
        memset(dp, -1, sizeof(dp));
        
        int ans = rec(0, 0) - 1;
        return ans;
    }
};
