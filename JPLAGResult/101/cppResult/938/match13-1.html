class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        vector<int> D1;
        for (string &s: D) {
            D1.push_back(s[0] - '0');
        }
        vector<int> N1;
        while (N > 0) {
            N1.push_back(N % 10);
            N /= 10;
        }
        return doit(D1, N1);
    }
    int doit(vector<int> &D1, vector<int> &N1) {

        int s = D1.size();
        int ss = 1;
        int result = 0;
        for (int i = 1; i <= N1.size() - 1; i++) {
            ss *= s;
            result += ss;
        }
        
        while (N1.size()) {
            bool eq = false;
            for (int i = 0; i < D1.size(); i++) {
                if (D1[i] < N1.back()) result += ss;
                else {
                    if (D1[i] == N1.back()) eq = true;
                    break;
                }
            }
            if (!eq) break;
            N1.pop_back();
            if (N1.size() == 0) result++;
            ss /= D1.size();
        }
        return result;
    }
};
