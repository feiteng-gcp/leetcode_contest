class Solution {
public:
    int dig(vector<int> &t, vector<int> &rem, vector<string> &D, int depth){
        if (depth < 0) return 1;
        int p = 1;
        for (int i=0;i<depth;i++) p = p*D.size();
        int c = 0;
        int i;
        for (i=0;i<D.size();i++){
            if (D[i][0]-'0' < t[depth]) 
                c++;
            else 
                break;
        }
        
        int res = c*p;
        if ((i < D.size()) && (D[i][0]-'0' == t[depth])) res += dig(t, rem, D, depth-1);
        
        return res;
    }
    
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        vector<int> t;
        vector<int> rem;
        vector<char> avail(10,0);
        int p = N;
        int x = 1;
        while (p > 0){
            t.push_back(p%10);
            if (!rem.empty()){
                rem.push_back((p%10)*x+rem.back());
            } else {
                rem.push_back((p%10)*x);
            }
            p = p/10;
            x = x*10;
        }
        
        int ans = 0;
        int c = D.size();
        for (int i=0;i<t.size()-1;i++){
            ans += c;
            c = c*D.size();
        }
        
        ans += dig(t, rem, D, t.size()-1);
        
        return ans;
    }
};
