class Solution {
public:
    int search(vector<int> &digits, int pos, vector<int> &iD, int cur_acc) {
        int sum = 0;
        for (int i = 0; i < iD.size(); i++) {
            if (iD[i] < digits[pos]) {
                sum += cur_acc;
            }
            else if(iD[i] == digits[pos]) {
                if (pos == 0) {
                    sum++;
                }
                else {
                    sum += search(digits, pos - 1, iD, cur_acc / iD.size());
                }
            }
            else break;
        }
        
        return sum;
    }
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        vector<int> digits;
        vector<int> iD;
        
        for (string &d : D) {
            iD.push_back(d[0] - '0');
        }
        
        int len = 0;
        while (N > 0) {
            len++;
            digits.push_back(N % 10);
            N /= 10;
        }
        
        int sum = 0;
        int cur = D.size();
        for (int i = 1; i < len; i++) {
            sum += cur;
            cur *= D.size();
        }
        
        sum += search(digits, digits.size() - 1, iD, cur / D.size());
        
        return sum;
    }
};
