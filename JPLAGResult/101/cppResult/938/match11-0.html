class Solution {
public:
    int countWays(vector<string>&D, string ns, int nd, vector<int> & ndpow){
        int nsl = ns.length();
        if(nsl == 0 )
            return 0;
        if(ns[0] > D[nd-1][0]){            
            return ndpow[nsl];
        }
        int res = 0;
        for(int j = 0; j < nd; ++j){
            if(D[j][0]>= ns[0]){
                res = j * ndpow[nsl-1];
                if(D[j][0] == ns[0]){
                    res += countWays(D, ns.substr(1), nd, ndpow);
                    if(nsl == 1){
                        res+=1;
                    }
                    break;
                }
                break;
            }

        } 
        return res;
    }
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        string sn = to_string(N);
        int n = sn.length();
        int nd = D.size();
        int res = 0 ;
        vector<int> ndpow;
        ndpow.push_back(1);
        for(int i = 1; i < n; ++i )
        {
            ndpow.push_back(ndpow.back()*nd);
            res += ndpow.back();
        }
        ndpow.push_back(ndpow.back()*nd);
        res += countWays(D,sn,nd,ndpow);
        return res; 
        
    }
};
