class Solution {
private:
    unordered_map<int, int> hash;
    int solve(int id, vector<int>& digits, vector<int>& dn) {
        if (id >= dn.size()) return 1;
        if (hash.find(id) != hash.end()) {
            return hash[id];
        }

        if (dn[id] < digits[0]) return 0;
        int tag = -1;
        for (int i = 0; i < digits.size(); i++) {
            if (digits[i] >= dn[id]) {
                tag = i;
                break;
            }
        }
        if (-1 == tag) {
            hash[id] = (int)pow(digits.size(), dn.size()-id);
            return hash[id];
        } else {
            int res = 0;
            if (tag - 1 >= 0) {
                res += tag * (int)pow(digits.size(), dn.size()-id-1);
            }
            if (digits[tag] == dn[id]) {
                res += solve(id+1, digits, dn);
            }
            return hash[id] = res;
        }
    }
    
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        vector<int> dn;
        while (N > 0) {
            dn.push_back(N % 10);
            N /= 10;
        }
        reverse(dn.begin(), dn.end());
        vector<int> digits;
        for (auto s : D) {
            digits.push_back(s[0]-'0');
        }
        sort(digits.begin(), digits.end());
        int res = 0;
        for (int i = dn.size() - 1; i > 0; i--) {
            res += (int)pow(digits.size(), i);
        }
        res += solve(0, digits, dn);
        return res;
    }
};
