class Solution {
public:
    char str[100];
    int res;
    vector<int> v;
    
    int dfs(int u, int len, string &s, bool isLess) {
        if (u == len) return 1;
        if (isLess) {
            int rem = len - u;
            int p = 1;
            for (int i = 0; i < rem; i++) {
                p *= v.size();
            }
            return p;
        } else {
            int res = 0;
            for (int i = 0; i < v.size(); i++) {
                if (v[i] < (s[u] - '0') ) {
                    res += dfs(u + 1, len, s, true);
                } else if (v[i] == (s[u] - '0')) {
                    res += dfs(u + 1, len, s, isLess);
                }
            }
            return res;
        }        
    }
    
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        
        for (int i = 0; i < D.size(); i++) {
            v.push_back(D[i][0] - '0');
        }
        sort(v.begin(), v.end());
        sprintf(str, "%d", N);
        string s = str;
        int len = s.size();
        int allAllowed = len - 1;
        

        res = 0;
        int cnt = 1;
        for (int i = 1; i <= allAllowed; i++) {
            cnt *= v.size();
            res += cnt;
        }
        
        res += dfs(0, len, s, false);
        return res;
    }
};
