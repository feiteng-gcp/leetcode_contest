#define FORI(s,n) for(int i = s; i < n; i++)
#define FORJ(s,n) for(int j = s; j < n; j++)

using ull = unsigned long long;
using vi = vector<int>;
using vs = vector<string>;
using vvi = vector<vi>;
using pii = pair<ull, ull>;
using vp = vector<pii>;
using vvp = vector<vp>;

int calc (vector<char> v, string s) {
   int c = 0;
   int n = v.size ();
   int d = s.size ();
   if (d == 0) return 1;
   FORI (0, n) {
      if (v[i] > s[0]) break;
      else if (v[i] < s[0]) c += pow (n, d - 1);
      else if (v[i] == s[0]) {
         //v.erase (v.begin () + i);
         s.erase (0, 1);
         c += calc (v, s);
         break;
      }
   }
   return c;
}

class Solution {
public:
   int atMostNGivenDigitSet (vector<string>& D, int N) {
      int n = to_string (N).size ();
      int d = D.size ();
      int ans = 0;
      FORI (1, n) ans += pow (d, i);
      vector<char> v (d);
      FORI (0, d) v[i] = D[i][0];
      ans += calc (v, to_string (N));
      return ans;
   }
};
