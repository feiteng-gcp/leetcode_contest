int _pow(int a, int b)
{
    int res = 1;
    while (b--) {
        res *= a;
    }
    return res;
}

int count_full_len(int digmap[], int N, int L)
{
    int count = 0;
    int msd;
    int setsz = 0;
    
    for (int i = 1; i <= 9; i++) {
            setsz += digmap[i];
        }

    int t = N;
    int len = 1;
    while (t >= 10) {
        t /= 10;
        len++;
    }
    if (len < L)
        return 0;
    msd = t;
    for (int i = 1; i < msd; i++) {
        if (digmap[i]) {
            count += _pow(setsz, len - 1);
        }
    }
    if (digmap[msd]) {
        if (len == 1) {
            return count + 1;
        } else {
            return count + count_full_len(digmap, N - msd * _pow(10, len-1), L-1);

        }
    }
    return count;
}

class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        int count = 0;
        int setsz = 0;
        int digmap[10] = {0};
        
        if (N <= 0)
            return 0;

        for (string s: D) {
            digmap[s[0] - '0'] = 1;
        }
        for (int i = 1; i <= 9; i++) {
            setsz += digmap[i];
        }

        int t = N;
        int len = 1;
        while (t >= 10) {
            t /= 10;
            len++;
        }
        
        count = count_full_len(digmap, N, len);
        for (int i = 1; i < len; i++) {
            count += _pow(setsz, i);
        }
        return count;
    }
};
