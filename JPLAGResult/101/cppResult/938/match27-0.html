class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        int size = D.size();
        string target = to_string(N);
        if (target.length() == 0) return 0;

        int result = 0;
        for (int i = 1; i < target.length(); i++) {
            result += pow(size, i);
        }

        int cnt = 0;
        bool duplicate = false;
        for (int i = 0; i < size; i++) {
            if (D[i][0] < target[0]) cnt++;
            else if (D[i][0] == target[0]) duplicate = true;
        }

        result += cnt * pow(size, target.length()-1);
        if (duplicate) result += calculate(D, target.substr(1, target.length()-1));

        return result;

    }

    int calculate(vector<string>& D, string target) {
        if (target.length() == 0) return 1;

        int cnt = 0;
        bool duplicate = false;
        for (int i = 0; i < D.size(); i++) {
            if (D[i][0] < target[0]) cnt++;
            else if (D[i][0] == target[0]) duplicate = true;
        }

        int result = 0;
        result += cnt * pow(D.size(), target.length()-1);
        if (duplicate) result += calculate(D, target.substr(1, target.length()-1));

        return result;
    }
};
