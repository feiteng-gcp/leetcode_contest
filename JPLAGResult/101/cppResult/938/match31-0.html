class Solution {
public:
	int NumDigits(int x) {
		int ret = 1;
		while (x > 9) {
			x = x / 10;
			ret ++;
		}
		return ret;
	}

	string toString(int x) {
		char buf[100];
		sprintf(buf, "%d", x);
		return string(buf);
	}

	int mypow(int b, int e) {
		int ret = 1;
		for (int i=0; i<e; i++) ret *= b;
		return ret;
	}

	// Only for same digits
	void do_dfs(vector<string>& D, string n_str, string curr, long* ret) {
//		printf("dfs %s ret=%ld\n", curr.c_str(), *ret);
		int currlen = int(curr.size());
		if (curr.size() > 0) {
			if (curr.size() < n_str.size()) {
				if (curr.back() < n_str[currlen-1]) {
					int len_rem = n_str.size() - currlen;
					*ret += mypow(int(D.size()), len_rem);
					return;
				}
				else if (curr.back() > n_str[currlen-1]) {
					return;
				}
			} else if (curr.size() == n_str.size()) {
				if (curr <= n_str) (*ret)++;
			} else return;
		}
		for (string delta : D) {
			do_dfs(D, n_str, curr+delta, ret);
		}
	}

    int atMostNGivenDigitSet(vector<string>& D, int N) {
		long ret = 0;
		int dsize = int(D.size());
        int ND = NumDigits(N);
//		printf("numdigits=%d\n", N);

		for (int i=1; i<=ND-1; i++) {
//			printf("pow(%d,%d)\n", dsize, i);
			ret += mypow(dsize, i);
		}

//		printf("before dfs ret=%ld\n", ret);

		do_dfs(D, toString(N), "", &ret);
		return int(ret);
    }
};
