class Solution {
public:
    int atMostNGivenDigitSet(vector<string>& D, int N) {
        int M = D.size();
        if (M == 0) return 0;
        vector<int> pow(10, 1);
        for (int i=1; i<10; i++) pow[i] = pow[i-1] * M;
        int sum = 0, idx;
        long r;
        for (idx=0, r=1; r<=N; r*=10, idx++);
        for (int i=1; i<idx; i++) sum += pow[i];
        r /= 10;
        for (int i=idx; i>0; i--) {
            int n1 = N / r;
            int c = 0, has = 0;
            for (string ds : D) {
                int d = ds[0] - '0';
                if (i>1) c += d<n1 ? 1 : 0;
                else c+= d<=n1 ? 1:0;
                if (d == n1) has = 1;
            }
            sum += c * pow[i-1];
            if (has == 0) break;
            N %= r;
            r /= 10;
        }
        return sum;
    }
};
