class Solution {
    vector<vector<long long>> a;
public:
    int numPermsDISequence(string S) {
        int n = S.size();
        a.resize(n + 1);
        for (auto &it: a) it.resize(n + 1);
        a[0][0] = 1;
        for (int i = 0; i < n; i++) {

            for (int j = 0; j <= i; j++) {
                if (S[i] == 'D') {
                    for (int k = 0; k <= j; k++) {
                        a[i+1][k] = (a[i+1][k] + a[i][j]) % (1000000007);
                    }

                } else {
                    for (int k = j + 1; k <= i + 1; k++) {
                        a[i+1][k] = (a[i+1][k] + a[i][j]) % (1000000007);
                    }
                }
            }
        }
        long long result = 0;
        for (int i = 0; i <= n; i++) {
            result = (result + a[n][i]) % 1000000007;
        }
        return (int) result;
    }
};
