class Solution {
public:
	int numPermsDISequence(string S) {
		int n = S.size() + 1;
		const int mo = 1000000007;
//		cout << n << endl;
		vector<vector<int> > f(n, vector<int>(n, 0));

		for (int i = 0; i < n; ++i) f[0][i] = 1;
		for (int i = 0; i < n-1; ++i) {
			for (int j = 0; j < n-i; ++j) {
				if (S[i] == 'D') {
					for (int k = 0; k < j; ++k) {
						f[i+1][k] = (f[i+1][k] + f[i][j]) % mo;
					}
				} else {
					for (int k = j+1; k < n-i; ++k) {
						f[i+1][k-1] = (f[i+1][k-1] + f[i][j]) % mo;
					}
				}
			}
		}
		int ans = 0;
		for (int i = 0; i < n; ++i) {
			ans = (ans + f[n-1][i]) % mo;
		}
		return ans;
	}
};
