class Solution {
public:
    string S;
    int memo[203][203];
    
    int dp(int left,int right,int pos){
        //dir -> 0  D  ,  1 I
        if(pos==S.size())return 1;
        if(memo[left][right]!=-1)return memo[left][right];
        int dev=0;
        if(S[pos]=='D'){
            
            if(left>0)
            for(int i=0;i<left;i++){
                dev=(dev+dp(i,left+right-i-1,pos+1))%1000000007;
            }
        }else{
            
            if(right>0)
            for(int i=0;i<right;i++){
                dev=(dev+dp(left+right-i-1,i,pos+1))%1000000007;
            }
            
        }
        
        memo[left][right]=dev;
        return dev;
    }
    
    int numPermsDISequence(string _S) {
        S=_S;
        int n=S.size()+1;
        int dev=0;
        memset(memo,-1,sizeof(memo));
        for(int i=0;i<n;i++)
            dev=(dev+dp(i,n-1-i,0))%1000000007;
        
        return dev;
    }
};
