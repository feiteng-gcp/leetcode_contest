#define M(x) ((x) % 1'000'000'007)

class Solution {
public:
    int numPermsDISequence(string S) {
        int n = S.size() + 1;
        vector<vector<int>> dp(n, vector<int>(n, 0));
        
        dp[0][0] = 1;
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                int c = dp[i - 1][j];
                if (S[i-1] == 'D') {
                    for (int k = 0; k <= j; k++) {
                        dp[i][k] = M(dp[i][k] + c);
                    }
                } else {
                    for (int k = j+1; k < n; k++) {
                        dp[i][k] = M(dp[i][k] + c);
                    }
                }
            }
        }
        
        int c = 0;
        for (int i = 0; i < n; ++i) {
            c = M(c + dp[n-1][i]);
        }
        
        return c;
        
    }
};
