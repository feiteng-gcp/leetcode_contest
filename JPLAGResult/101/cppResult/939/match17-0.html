class Solution {
public:
    int dp[205][205];
    int mod=1e9+7;
    int numPermsDISequence(string S) {
        dp[0][0]=1;
        for(int i=0;i<S.size();i++){
            int sum=0;
            if(S[i]=='D'){
                for(int k=i;k>=0;k--){
                    sum+=dp[i][k];
                    sum%=mod;
                    dp[i+1][k]=sum;
                }
            }
            else{
                for(int k=0;k<=i+1;k++){
                    dp[i+1][k]=sum;
                    sum+=dp[i][k];
                    sum%=mod;
                }
            }
        }
        int ans=0;
        for(int i=0;i<=S.size();i++)
        {
            ans+=dp[S.size()][i];
            ans%=mod;
        }
        return ans;
    }
};
