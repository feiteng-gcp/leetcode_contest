class Solution {
public:
    string A;
    int dp[205][205];
    const int Mod = 1000000007;
    
    void take_mod(int &x) {
        while(x >= Mod) {
            x -= Mod;
        }
        
        while(x < 0) {
            x += Mod;
        }
    }
    
    int rec(int rem, int smaller) {
        if(rem == 0) {
            return 1;
        }
        
        int larger = rem - smaller;
        
        int &ret = dp[rem][smaller];
        if(ret != -1) {
            return ret;
        }
        ret = 0;
        if(A[rem - 1] == 'I') {
            
            for(int i = 0;i < larger;++i) {
                ret += rec(rem - 1, smaller + i);
                take_mod(ret);
            }
            
        } else {
            
            for(int i = 0;i < smaller;++i) {
                ret += rec(rem - 1, i);
                take_mod(ret);
            }
            
        }
        return ret;
    }
    
    int numPermsDISequence(string S) {
        A = S;
        // cout << A.size() << "\n";
        
        int ans = 0;
        memset(dp, 0xff, sizeof(dp));
        for(int i = 0;i <= A.size();++i) {
            ans += rec(A.size(), i);
            take_mod(ans);
        }
        return ans;
    }
};
