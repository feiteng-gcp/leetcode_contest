#include <string>
#include <vector>
#include <numeric>
#include <iostream>
#include <cassert>

using namespace std;

class Solution {
  static const int MOD = 1000000007;
public:
  int numPermsDISequence(string str) {
    int s = str.size();
    std::vector<int> dp(s + 1), dp1(s + 1);
    dp1[0] = 1;
    for (int i = s - 1; i >= 0; --i) {
      if (str[i] == 'D') {
        dp[0] = 0;
        int sum = dp1[0];

        for (int j = 1; j <= s - i; ++j) {
          dp[j] = sum;
          sum += dp1[j];
          sum %= MOD;
        }
      } else {
        dp[s - i] = 0;
        int sum = dp1[s - i];

        for (int j = s - i - 1; j >= 0; --j) {
          sum += dp1[j];
          sum %= MOD;
          dp[j] = sum;
        }
      }
      swap(dp, dp1);
    }
    int ret = 0;
    for (auto v : dp1) {
      ret += v;
      ret %= MOD;
    }
    return ret;
  }
};
