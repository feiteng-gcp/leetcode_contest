class Solution {
public:
    int numPermsDISequence(string S) {
        long long MO = 1000000007;
        int n = S.length();
        vector<vector<long long>> dp(205, vector<long long>(205,0));
        dp[0][0] = 1;
        for (int i=1;i<=n;i++){
            if (S[i-1] == 'D'){
                for (int j=0;j<=i;j++){
                    for (int k=j;k<=i-1;k++){
                        dp[i][j] = (dp[i][j] + dp[i-1][k])%MO;
                    }
                }
            } else {
                for (int j=0;j<=i;j++){
                    for (int k=0;k<=j-1;k++){
                        dp[i][j] = (dp[i][j] + dp[i-1][k])%MO;
                    }
                }
            }
        }
        
        long long ans = 0;
        for (int i=0;i<=n;i++){
            ans = (ans + dp[n][i])%MO;
        }
        
        return (int)ans;
    }
};
