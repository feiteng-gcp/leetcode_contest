class Solution {
public:
    int dp[205][205];
    int n;
    const int mod = 1000000000 + 7;
    string s;

    int memo(int lower, int higher, int u) {
        if (u == n) return 1;
        if (dp[lower][higher] != -1) return dp[lower][higher];
        int res = 0;
        if (s[u] == 'D') {
            for (int i = 0; i < lower; i++) {
                res += memo(i, lower - i - 1 + higher, u + 1);
                res %= mod;
            }
        } else  {
            for (int i = 0; i < higher; i++) {
                res += memo(i + lower, higher - i - 1, u + 1);
                res %= mod;
            }
        }
        return dp[lower][higher] = res;
    }
    
    int numPermsDISequence(string _s) {
        s = _s;
        n = s.size();
        
        for (int i = 0; i <= n; i++)
            for (int j = 0; j <= n; j++)
                dp[i][j] = -1;
        
        int res = 0;
        for (int i = 0; i <= n; i++) {
            res += memo(i, n - i, 0);
            res %= mod;
        }
        return res;
    }
};
