class Solution {
public:
    int M = 1E9 + 7;
    int find(const string& S, int pos, int numOfLess, vector<vector<int>>& F) {
        int len = S.length();
        if (pos > len) {
            return 1;
        }
        int& t = F[pos][numOfLess];
        if (t < 0) {
            t = 0;
            if (pos > 0 && S[pos - 1] == 'D') {
                for (int i = 0; i < numOfLess; ++i) {
                    t += find(S, pos + 1, i, F);
                    t %= M;
                }
            }
            if (pos == 0 || (pos > 0 && S[pos - 1] == 'I')) {
                for (int i = numOfLess; i <= len - pos; ++i) {
                    t += find(S, pos + 1, i, F);
                    t %= M;
                }
            }
        }
        return t;
    }
    
    int numPermsDISequence(string S) {
        vector<vector<int>> F;
        int len = S.length();
        F.resize(len + 1);
        for (int i = 0; i <= len; ++i) {
            F[i].resize(len + 1);
            for (int j = 0; j <= len; ++j) {
                F[i][j] = -1;
            }
        }
        find(S, 0, 0, F);
        return F[0][0];
    }
};
