const int N = 200 + 10;
const int MOD = 1e9 + 7;
bool visit[N][N];
int dp[N][N];

class Solution {
public:
    string S;
    int n;
    int solve(int x, int pos) {
        if (pos == n - 1) return 1;
        int& ret = dp[x][pos];
        if (visit[x][pos]) return ret;
        visit[x][pos] = true;

        ret = 0;
        if (S[pos] == 'I') {
            for (int y = x + 1; y < n - pos; ++y) {
                ret = (ret + solve(y - 1, pos + 1)) % MOD;
            }
        } else {
            for (int y = 0; y < x; ++y) {
                ret = (ret + solve(y, pos + 1)) % MOD;
            }
        }
        return ret;
    }
    int numPermsDISequence(string S) {
        this->n = S.size() + 1;
        this->S = S;
        memset(visit, 0, sizeof(visit));

        int ret = 0;
        for (int i = 0; i < n; ++i) {
            ret = (ret + solve(i, 0)) % MOD;
        }
        return ret;
    }
};
