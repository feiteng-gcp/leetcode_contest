typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 205

const int mod = 1000000007;
ll C[MAXN][MAXN];
// pairwise binomial up to N^2
void binomial(ll mod) {
  FILL(C, 0);
  for (int i = 0; i < MAXN; i++) C[i][0] = 1;
  for (int i = 1; i < MAXN; i++){
    for (int j = 1; j <= i; j++){
      C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % mod;
    }
  }
}
string S;
int n, dp[MAXN][MAXN];
int solve(int l, int r) {
  if (l > r) return 1;
  if (dp[l][r] != -1) return dp[l][r];
  int ans = 0;
  REP(i,l,r+1) {
    string t = "I";
    if (i > l) t = S[i - 1];
    if (i == r) t += "D";
    else t += S[i];
    if (t != "ID") continue;
    int cho = C[r - l][i - l];
    ans += (ll)cho * ((ll)solve(l, i - 1) * solve(i + 1, r) % mod) % mod;
    ans %= mod;
  }
  return dp[l][r] = ans;
}
class Solution {
public:
  int numPermsDISequence(string _S) {
    S = _S;
    binomial(mod);
    n = S.size() + 1;
    FILL(dp, -1);
    return solve(0, n-1);
  }
};
