class StockSpanner {
public:
    vector<int> prices;
    vector<int> nums;
    

    StockSpanner() {
        
    }
    
    int next(int price) {
        if (prices.empty()) {
            prices.push_back(price);
            nums.push_back(1);
            return 1;
        }
        int index = prices.size() - 1;
        while (index >= 0 && price >= prices[index]) {
            index -= nums[index];
        }
        prices.push_back(price);
        nums.push_back(nums.size() - index);
        return nums.size() - index - 1;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
