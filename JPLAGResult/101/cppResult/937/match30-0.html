#define pb push_back
#define fst first
#define snd second

typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class StockSpanner {
public:
  vector<pii> st;
  int N;
  
  StockSpanner() {
      N=0;
  }
  
  int next(int pr) {
    N++;
    while(!st.empty() && st.back().fst<=pr){
      st.pop_back();
    }
    int k=st.empty()?0:st.back().snd;
    st.pb({pr,N});
    return N-k;
  }
};
