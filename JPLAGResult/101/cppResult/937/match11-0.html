class StockSpanner {
public:
    vector<int> prices;
    vector<int> spans;
    StockSpanner() {}
    
    int next(int price) {
        int ans = 1;
        int idx = (int)prices.size() - 1;
        while (idx >= 0 && price >= prices[idx]) {
            ans += spans[idx];
            idx -= spans[idx];
        }
        prices.push_back(price);
        spans.push_back(ans);
        return ans;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
