class StockSpanner {
    vector<int> dp;
    vector<int> stock;
public:
    StockSpanner() {
    }
    
    int next(int price) {
        int k = stock.size()-1;
        int ans = 1;
        while (k >= 0 && stock[k] <= price) {
            ans += dp[k];
            k -= dp[k];
        }
        dp.push_back(ans);
        stock.push_back(price);
        return ans;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
