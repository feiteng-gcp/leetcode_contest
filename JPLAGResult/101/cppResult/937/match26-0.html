class StockSpanner {
public:
    deque<pair<int, int>> q; // pair<index, value>
    int total;
    StockSpanner() {
        total = 0;
        q.push_back(make_pair(-1, 1e6));
    }
    
    int next(int price) {
        auto now = make_pair(total++, price);
        while (!q.empty() && q.back().second <= price) {
            q.pop_back();
        }
        int res = total - q.back().first - 1;
        q.push_back(now);
        return res;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
