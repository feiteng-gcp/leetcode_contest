class StockSpanner {
public:
    vector<int> v;
    vector<int> res;
    int n;
    StockSpanner() {
       v.clear();
       res.clear();
       n = 0;
    }
    
    int next(int price) {
        v.push_back(price);
        n++;
        int cnt = 1;
        for (int i = n - 2; i >= 0; i--) {
            if (v[n-1] >= v[i]) {
                cnt += res[i];
                i -= res[i];
                i++;
            }
            else break;
        }
        res.push_back(cnt);
        return cnt;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
