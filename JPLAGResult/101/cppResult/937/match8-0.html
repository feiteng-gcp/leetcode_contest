class StockSpanner {
public:
    vector<int> spans;
    vector<int> prices;
    StockSpanner() : spans(1, 1), prices(1, 1e6) {
        
    }
    
    int next(int price) {
        int cur = spans.size()-1;
        int span = 1;
        while (cur >= 0)
        {
            if (price >= prices[cur])
            {
                span += spans[cur];
                cur -= spans[cur];
            }
            else
                break;
        }
        spans.push_back(span);
        prices.push_back(price);
        return span;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
 */
