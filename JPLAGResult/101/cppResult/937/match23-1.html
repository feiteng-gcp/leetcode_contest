class StockSpanner {
private:
    vector<int> a;
    vector<int> s;
public:
    StockSpanner() {
        
    }
    
    int next(int price) {
        if (a.empty()) {
            a.push_back(price);
            s.push_back(1);
            return 1;
        }
        int res = 1;
        int i = (int)a.size() - 1;
        a.push_back(price);
        while (i >= 0) {
            if (price >= a[i]) {
                res += s[i];
                i -= s[i];
            } else {
                break;
            }
        }
        s.push_back(res);
        return res;
    }
};


/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
