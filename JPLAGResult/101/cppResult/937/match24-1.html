class StockSpanner {
    vector<int> prices;
    vector<int> span;
public:
    StockSpanner() {
        
    }
    
    int next(int price) {
        int _span = 1;
        prices.push_back(price);
        ssize_t pos = prices.size() - 2;
        while (pos >= 0 && prices[pos] <= price) {
            _span += span[pos];
            pos -= span[pos];
        }
        span.push_back(_span);
        return _span;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
 */
