class StockSpanner {
public:
    vector<int> stack;
    vector<int> pos;
    int pos_count;
    StockSpanner() {
        pos_count = 0;
    }
    
    int next(int price) {
        while (!stack.empty() && stack.back() <= price) {
            stack.pop_back();
            pos.pop_back();
        }
        stack.push_back(price);
        pos.push_back(pos_count++);
        int last_pos = stack.size() > 1 ? pos[stack.size() - 2] : -1;
        return pos_count - last_pos - 1;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
 */
