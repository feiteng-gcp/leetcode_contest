class StockSpanner {
public:
    stack<pair<int, int> > st;
    StockSpanner() {
        
    }
    
    int next(int price) {
        int cnt = 1;
        while (!st.empty()) {
            pair<int, int>& tmp = st.top();
            
            if (tmp.first > price) {
                break;
            } else {
                int tp = tmp.first;
                cnt += tmp.second;
                st.pop();
                if (tp == price) {
                    break;
                }
            }
        }
        st.push(make_pair(price, cnt));
        return cnt;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
