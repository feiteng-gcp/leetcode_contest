class StockSpanner {
    vector<int> prices;
    vector<int> result;
    
public:
    StockSpanner() {
        
    }
    
    int next(int price) {
        prices.push_back(price);
        
        if(result.empty()) {
            result.push_back(1);
        }else{
            int index = result.size() - 1;
            int this_result = 1;
            while(index>=0 && prices[index]<=price) {
                this_result += result[index];
                index-=result[index];
            }
            result.push_back(this_result);
        }
        return result.back();
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
 */
