class StockSpanner {
private:
    deque<pair<int, int>> d_num_stack;
public:
    StockSpanner() {
        
    }
    
    int next(int price) {
        int rst = 1;
        while (!d_num_stack.empty() && d_num_stack.back().first <= price) {
            rst += d_num_stack.back().second;
            d_num_stack.pop_back();
        }
        d_num_stack.push_back(make_pair(price, rst));
        return rst;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
