class StockSpanner {
    stack<pair<int, int>> dst; // decreasing stack

    int count;
public:
    StockSpanner() {
        count = 0;
    }
    
    int next(int price) {
        while (!dst.empty() && dst.top().first <= price) {
            dst.pop();
        }
        int topIdx = 0;
        if (dst.empty()) {
            topIdx = -1;
        } else {
            topIdx = dst.top().second;
        }
        int res = count - topIdx;
        dst.push(make_pair(price, count));
        count ++;
        return res;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
