class StockSpanner {
    std::vector<pair<int, int>> stack;
    int day;
public:
    StockSpanner() : day(0) {
        stack.emplace_back(day, numeric_limits<int>::max());

    }
    
    int next(int price) {
        while (stack.back().second <= price) {
            stack.pop_back();
        }
        int ret = ++day - stack.back().first;
        stack.emplace_back(day, price);
        return ret;
    }
};

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
 */
