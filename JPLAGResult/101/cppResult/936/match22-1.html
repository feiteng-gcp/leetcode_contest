class RLEIterator {
public:
	vector<int> rec;
	RLEIterator(vector<int> A) {
		rec = A;
		rec.insert(rec.begin(), 2, 1);
	}

	int next(int n) {
//		cout << rec[0] << n << endl;

		for (;rec.size() && rec[0] <= n;) {
			n -= rec[0];
			rec.erase(rec.begin());
			rec.erase(rec.begin());
//			cout << n << endl;
		}
		if (rec.size()) {
			rec[0] -= n;
			return rec[1];
		} else {
			return -1;
		}
	}
};
