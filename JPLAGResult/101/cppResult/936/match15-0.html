class RLEIterator {
public:
    int curValIdx;
    vector<int> A;
    RLEIterator(vector<int> A) : A(A), curValIdx(1) {
        assert(A.size() % 2 == 0);
    }
    
    int next(int n) {
        while (n && curValIdx < A.size()) {
            if (A[curValIdx - 1] >= n) {
                A[curValIdx - 1] -= n;
                n = 0;
            } else {
                n -= A[curValIdx - 1];
                curValIdx += 2;
            }
        }
        if (curValIdx >= A.size()) {
            return -1;
        } else {
            return A[curValIdx];
        }
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
