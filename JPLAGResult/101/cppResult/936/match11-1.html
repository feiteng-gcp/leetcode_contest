class RLEIterator {
public:
    vector<int> a;
    int now, sz;
    RLEIterator(vector<int> A) {
        a = A;
        now = 0;
        sz = A.size();
    }
    
    int next(int n) {
        int ret = 0;
        while (n) {
            if (now == sz) return -1;
            if (a[now] >= n) {
                a[now] -= n;
                ret = a[now+1];
                if (a[now] == 0) now += 2;
                return ret;
            } else {
                n -= a[now];
                now += 2;
            }
        }
    }
};

 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
