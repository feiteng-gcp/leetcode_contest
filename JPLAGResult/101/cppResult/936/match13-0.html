class RLEIterator {
    vector<int> A;
    long long count = 0, pos = 0;
public:
    RLEIterator(vector<int> _A): A(_A) {
        
    }
    
    int next(int n) {
        count += n;
        while(pos < A.size() && count > A[pos]) {
            count -= A[pos];
            pos += 2;
        }
        if(pos == A.size())
            return -1;
        return A[pos + 1];
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
