class RLEIterator {
public:
    vector<int> v;
    int i;
    RLEIterator(vector<int> A) {
        v=A;
        i=0;
    }
    
    int next(int n) {
        if (i>=v.size()){
            return -1;
        }
        if (n<=v[i]){
            v[i]-=n;
            return v[i+1];
        }
        while (n>0){
            if (n>v[i]){
                n=n-v[i];
                v[i]=0;
                i=i+2;
                if (i>=v.size()){
                    return -1;
                }
            }
            else{
                v[i]-=n;
                n = 0;
            }
        }
        return v[i+1];        
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
