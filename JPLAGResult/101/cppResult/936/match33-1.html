class RLEIterator {
public:
    RLEIterator(vector<int> A) {
        for (int i = 0; i < A.size(); i += 2) {
            times.push(make_pair(A[i], A[i + 1]));
        }
    }
    
    int next(int n) {
        while (!times.empty()) {
            auto &cur = times.front();
            int consumed = min(cur.first, n);
            cur.first -= consumed;
            n -= consumed;
            if (cur.first == 0) {
                times.pop();
            }
            if (n < 0) {
                return -1;
            } else if (n == 0) {
                return cur.second;
            }
        }
        return -1;
    }
private:
    queue<pair<int, int>> times;
};

 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
