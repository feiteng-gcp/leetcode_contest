class RLEIterator {

public:
    RLEIterator(vector<int> A) {
        idx = 0;
        len = A.size() / 2;
        a.assign(A.begin(), A.end());
    }
    
    int next(int n) {
        while (idx < len && a[idx * 2] < n) {
            n -= a[idx * 2];
            idx++;
        }
        if (idx >= len) {
            return -1;
        } else {
            a[idx * 2] -= n;
            return a[idx * 2 + 1];
        }
    }
private:
    int idx, len;
    vector<int> a;
};

 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
