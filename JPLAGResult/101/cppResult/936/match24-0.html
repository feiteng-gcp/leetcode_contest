class RLEIterator {
    int index;
    int count;
    vector<int> seq;
    
public:
    RLEIterator(vector<int> A) 
        : index(0), count(0), seq(A.begin(), A.end()) {
    }
    
    int next(int n) {
        if (index >= seq.size()) return -1;
        if (n <= seq[index]) {
            seq[index] -= n;
            return seq[index+1];
        }
        else {
            n -= seq[index];
            seq[index] = 0;
            index += 2;
            return next(n);
        }
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
