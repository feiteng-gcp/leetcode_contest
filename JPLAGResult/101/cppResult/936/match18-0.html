class RLEIterator {
    vector<int> code;
    int i;
public:
    RLEIterator(vector<int> A) {
        code=A;
        i=0;
    }
    
    int next(int n) {
        while(code[i]<n&&i<code.size()){
            n-=code[i];
            i+=2;
        }
        if(i>=code.size())return -1;
        code[i]-=n;
        return code[i+1];
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
 */
