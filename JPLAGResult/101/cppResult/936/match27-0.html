class RLEIterator {
public:
    RLEIterator(vector<int> a) {
        idx = 0;
        A = a;
    }
    
    int next(int n) {
        int res = 0;
        while (idx < A.size() && res < n) {
            res += A[idx];
            idx += 2;
        }
        
        if (res < n) {
            return -1;
        }
        
        idx -= 2;
        A[idx] = res - n;
        return A[idx + 1];
    }
    
    int idx;
    vector<int> A;
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
