class RLEIterator {
public:
    RLEIterator(vector<int> A) 
    {
        v.clear();

        idx=0;
        v=A;
    }
    
    int next(int n) 
    {
        while(idx<v.size()&&n)
        {
            if(v[idx]>=n)
            {
                v[idx]-=n;
                n=0;
            }
            else
            {
                n-=v[idx];
                idx+=2;
            }
        }
        if(idx>=v.size()) return -1;
        else return v[idx+1];
    }
private:
    vector<int> v;
    int idx;
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
