class RLEIterator {
    vector<int> numbers;
    int pos;
public:
    RLEIterator(vector<int> A) {
        numbers = A;
        pos = 0;
    }
    
    int next(int n) {
        if (pos+1 >= numbers.size()) return -1;
        while (n > 0 && pos + 1 < numbers.size()) {
            if (numbers[pos] >= n) {
                numbers[pos] -= n;
                return numbers[pos+1];
            } else {
                n -= numbers[pos];
                numbers[pos] = 0;
                pos += 2;
            }
        }
        return -1;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
