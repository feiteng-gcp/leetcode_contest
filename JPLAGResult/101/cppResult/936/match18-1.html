#include <vector>
using namespace std;


class RLEIterator {
    vector<int> a;
    int pos;
public:
    RLEIterator(vector<int> A) {
        a = A;
        pos = 0;
    }
    
    int next(int n) { //n > 0
        while(n > 0 && pos < a.size()){
            if(a[pos] > n){
                a[pos] -= n;
                return a[pos + 1];
            }else{
                n -= a[pos];
                pos += 2;
                if(n == 0){
                    return  a[pos - 1];
                }
            }
        }
        return -1;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
 */
