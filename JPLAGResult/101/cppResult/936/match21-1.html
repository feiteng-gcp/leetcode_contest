class RLEIterator {
public:
    RLEIterator(vector<int> A) {
        a = A;
        current = a.begin();
        offset = 0;

    }
    
    int next(int n) {
        while(current != a.end()){
            if(current[0] - offset < n){
                // Move to next
                n -= current[0] - offset;
                current += 2;
                offset = 0;
            }else{
                offset += n;
                return current[1];
            }
        }
        return -1;
    }
private:
    vector<int> a;
    vector<int>::iterator current;
    int offset;
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
