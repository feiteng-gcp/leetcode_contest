class RLEIterator {
public:
    vector < int > B;
    RLEIterator(vector<int> A) {
        B = A;
        reverse(B.begin(), B.end());
    }
    
    int next(int n) {
        
        while(B.size() and n > B[B.size() - 1]) {
            n -= B[B.size() - 1];
            B.pop_back();
            B.pop_back();
        }
        
        if(B.size()) {
            B[B.size() - 1] -= n;
            return B[B.size() - 2];
        } else {
            return -1;
        }
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
