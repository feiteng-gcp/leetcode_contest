class RLEIterator {
public:
    vector<int> arr;
    int p;
    RLEIterator(vector<int> A) {
        arr=A;
        p=0;
    }

    int next(int n) {
        if(p>=arr.size())
            return -1;
        if(arr[p]>=n){
            arr[p]-=n;
            return arr[p+1];
        }
        int nxtn=n-arr[p];
        p+=2;
        return next(nxtn);
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
 */
