class RLEIterator {
public:
    vector<int> nums;
    int index;
    RLEIterator(vector<int> A) {
        nums = A;
        index = 0;
    }
    
    int next(int n) {
        while (n > 0 && index < nums.size()) {
            if (nums[index] >= n) {
                nums[index] -= n;
                return nums[index+1];
            }
            else {
                n -= nums[index];
                index += 2;
            }
        }
        return -1;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
