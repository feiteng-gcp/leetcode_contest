class RLEIterator {
public:
    int p;
    int len;
    vector<int> AA;
    RLEIterator(vector<int> A) {
        len=A.size();
        for(int i=0;i<len;i++)
        {
            AA.push_back(A[i]);
        }
        p=0;

    }
    
    int next(int n) {
        while(n&&p<len)
        {
            if(AA[p]<n)
            {
                n=n-AA[p];
                p=p+2;
            }
            else
            {
                AA[p]=AA[p]-n;
                n=0;
            }
        }
        if(p>=len)return -1;
        return AA[p+1];
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
