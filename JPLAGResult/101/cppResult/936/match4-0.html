class RLEIterator {
public:
    RLEIterator(vector<int> A) {
        m_data = A;
        m_index = 0;
        m_dataIndex = 0;
    }
    
    int next(int n) {
        m_index += n;
        
        while (m_dataIndex < m_data.size() && m_index > m_data[m_dataIndex]) {
            m_index -= m_data[m_dataIndex];
            m_dataIndex += 2;
        }
        
        if (m_dataIndex < m_data.size()) {
            return m_data[m_dataIndex + 1];
        } else {
            return -1;
        }
    }
    
private:
    vector<int> m_data;
    int m_index;
    int m_dataIndex;
};

 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
