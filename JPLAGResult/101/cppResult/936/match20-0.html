class RLEIterator {
public:
    RLEIterator(vector<int> A) {
        data = A;
        size = A.size();
        pos = 0;
    }
    
    int next(int n) {
        while (pos < size) {
            if (data[pos] < n) {
                n -= data[pos];
                pos += 2;
            } else {
                data[pos] -= n;
                break;
            }
        }
        return (pos < size ? data[pos + 1] : -1);
    }
private:
    vector<int> data;
    int pos, size;
};

 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
