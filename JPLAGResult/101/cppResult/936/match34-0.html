#define PII pair<int, int>
#define mp make_pair

class RLEIterator
{
public:
    queue<PII> Q;
    
    RLEIterator(vector<int> A)
    {
        for (int i = 0; i < A.size(); i += 2)
        {
            Q.push(mp(A[i], A[i + 1]));
        }
    }
    
    int next(int n)
    {
        if (Q.empty()) return -1;
        while (!Q.empty())
        {
            n -= Q.front().first;
            if (n <= 0)
            {
                Q.front().first = -n;
                return Q.front().second;
            }
            Q.pop();
        }
        return -1;
    }
};
