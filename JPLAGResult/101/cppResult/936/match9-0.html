class RLEIterator {
    vector<int> vals;
    int pos = 0, count = 0;
public:
    RLEIterator(vector<int> A) {
        vals = A;
    }
    
    int next(int n) {
        count += n;
        while (pos < vals.size() && count > vals[pos]) {
            count -= vals[pos];
            pos += 2;
        }
        if (pos >= vals.size()) return -1;
        return vals[pos + 1];
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
