class RLEIterator {
public:
    vector<int> _A;
    int now;
    RLEIterator(vector<int> A) {
        _A=A;
        now=0;
    }
    
    int next(int n) {
        while(n){
            if(now>=_A.size())
                return -1;
            _A[now]-=n;
            n=0;
            if(_A[now]<0){
                n=-_A[now];
                now+=2;
            }
        }
        return _A[now+1];
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
