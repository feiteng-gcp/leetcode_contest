class RLEIterator {
    int p;
    vector<int> A;
public:
    RLEIterator(vector<int> A) : A(A), p(0) {
        
    }
    
    int next(int n) {
        while (p < A.size() && n > 0) {
            if (n > A[p]) {
                n -= A[p];
                A[p] = 0;
                p += 2;
            } else {
                A[p] -= n;
                break;
            }
        }
        return p < A.size() ? A[p + 1] : -1;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
