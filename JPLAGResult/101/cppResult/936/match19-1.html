class RLEIterator {
private:
    vector<int> B;
public:

    RLEIterator(vector<int> A) {
        reverse(A.begin(), A.end());
        B = A;
    }
    
    int next(int n) {
        int sz = B.size();
        for (int i = sz-1; i > 0; i -= 2) {
            if (n < B[i]) {
                B[i] -= n;
                return B[i-1];
            } else if (n == B[i]) {
                int x = B[i-1];
                B.pop_back();
                B.pop_back();
                return x;
            } else {
                n -= B[i];
                B.pop_back();
                B.pop_back();
            }
        }
        return -1;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
