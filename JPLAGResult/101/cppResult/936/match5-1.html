class RLEIterator {
public:
    RLEIterator(vector<int> code) {
        A = code;

        index = 0;
        curused = 0; 
    }
    
    int next(int n) {
        if (index >= A.size())
            return -1;
        while ( n > 0 && index < A.size()) {
            if (curused + n < A[index]) {
                curused += n;
                return A[index + 1];
            } else {
                n -= (A[index] - curused);
                index += 2;
                curused = 0;
            }
        }
        if (index < A.size()) {// n ==0;
            return A[index-1];
        }else {
            return -1;
        }
        
    }
    private:
    int index;
    int curused; 
    vector<int> A;
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
 */
