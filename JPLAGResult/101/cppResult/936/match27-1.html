class RLEIterator {
public:
    RLEIterator(vector<int> A) {
        ptr = 0;
        a = A;
    }
    
    int next(int n) {
        
        int ret = -1;
        
        while ( n > 0 ){
            while ( ptr + 1 < a.size() && a[ptr] == 0 )
                ptr += 2;
            if ( ptr + 1 >= a.size() )
                break;
            if ( a[ptr] >= n ){
                a[ptr] -= n;
                ret = a[ptr+1];
                break;
            }
            n -= a[ptr];
            ptr += 2;
            
        }
        return ret;
        
        
    }
    int ptr;
    vector<int> a;
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
