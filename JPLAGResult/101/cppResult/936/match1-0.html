class RLEIterator {
    vector<int> A;
    int current = 0;
    
public:
    RLEIterator(vector<int> A) {
        this->A = A;
    }
    
    int next(int n) {
        while(current < A.size() && n>A[current]){
            n-=A[current];
            current+=2;
        }
        
        if(current<A.size()) {
            A[current]-=n;
            return A[current+1];
        }
        
        return -1;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
 */
