class RLEIterator {
private:
    vector<int> A;
    int curPos;
    int curCnt;
public:
    RLEIterator(vector<int> A) {
        this->A = A;
        curPos = 0;

        curCnt = -1;
    }
    
    int next(int n) {
        curCnt += n;
        while (curPos < A.size() && A[curPos] <= curCnt) {
            curCnt -= A[curPos];
            curPos += 2;
        }
        if (curPos >= A.size()) return -1;
        return A[curPos+1];
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
