class RLEIterator {
public:
    vector<int> inp;
    int iter, n;
    RLEIterator(vector<int> A) {
        inp = A;
        iter = 0;
        n = inp.size();
    }
    
    int next(int n2) {
        int cnt = 0;
        while(cnt < n2) {
            if(iter >= n) return -1;
            if(inp[iter] == 0){
                iter += 2;
                continue;
            }
            long long times = inp[iter];
            long long rem = n2 - cnt;
            long long advance = min(times, rem);
            inp[iter] -= advance;
            cnt += advance;
        }
        // for(int i : inp) cout << i << ' '; cout << endl;
        if(iter >= n) return -1;
        return inp[iter + 1];
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
