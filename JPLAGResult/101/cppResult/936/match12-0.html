class RLEIterator {
public:
	vector<int> num;
	int pos, len;
    RLEIterator(vector<int> A) {
        num = A;
		len = A.size();
		pos = 0;
    }
    
    int next(int n) {
        int i = pos, re;
		while (i<len && n > num[i]) {
			n -= num[i];
			i += 2;
		}
		if (i>=len) {
			re = -1;
			pos = len;
		} else if (n == num[i]) {
			re = num[i+1];
			pos = i+2;
		} else {
			re = num[i+1];
			num[i] -= n;
			pos = i;
		}
		return re;
    }
};
