vector<int>freq;
vector<int>ele;

class RLEIterator {
public:
    RLEIterator(vector<int> A) {
        freq.clear();
        ele.clear();
        for( int i = 0 ; i < A.size() ; i+=2 ){
            freq.push_back(A[i]);
            ele.push_back(A[i+1]);
        }
    }
    
    int next(int n) {
        int ret = -1;
        while(n){
            if(freq.empty())
                break;
            int val = freq[0];
            if( val < n ){
                n-=val;
                freq.erase(freq.begin());
                ele.erase(ele.begin());
            }
            else{
                val-=n;
                ret = ele[0];
                n=0;
                if(!val){
                    freq.erase(freq.begin());
                    ele.erase(ele.begin());
                }
                else{
                    freq[0] = val;
                }
            }
        }
        return ret;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
 */
