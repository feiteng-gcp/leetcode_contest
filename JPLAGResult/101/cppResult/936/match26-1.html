class RLEIterator {
public:
    vector<pair<int, int>> v;
    RLEIterator(vector<int> A) {
      for (int i = 0; i < A.size(); i+=2) {
        if (A[i]) v.push_back(make_pair(A[i], A[i+1]));
      }
      reverse(v.begin(), v.end());
    }

    int next(int n) {
      int ret = -1;
      int idx = v.size() - 1;
      while (idx >= 0 && n && ret == -1) {
        //printf("idx=%d, c=%d, val=%d\n", idx, v[idx].first, v[idx].second);
        if (v[idx].first >= n) v[idx].first -= n, n = 0, ret = v[idx].second;
        else n -= v[idx].first, v[idx].first = 0;
        if (v[idx].first == 0) v.pop_back(), idx--;
      }
        return ret;
    }
};

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
