class Solution {
    long MOD = 1000000007L;
    public int numPermsDISequence(String S) {
        int n = S.length();
        if (n == 1) return 1;
        long[][] dp = new long[n + 1][n];
        if (S.charAt(n - 1) == 'D') {
            dp[0][n - 1] = 0;
            dp[1][n - 1] = 1;
        }
        else {
            dp[0][n - 1] = 1;
            dp[1][n - 1]  = 0;
        }
        long ans = 0;
        for (int j = n - 2; j >= 0; j--) {
            for (int i = 0; i <= n - j; i++) {
                dp[i][j] = 0;
                int start = 0, end = n - j - 1;
                if (S.charAt(j) == 'D') {
                    end = i - 1;
                }
                else {
                    start = i;
                }
                for (int k = start; k <= end; k++)
                    dp[i][j] = (dp[i][j] + dp[k][j + 1]) % MOD;
                if (j == 0) ans = (ans + dp[i][j]) % MOD
;
            }
        }
        return (int) ans;
    }
}
