class StockSpanner {
    Stack<Integer> prices; // decreasing values
    Stack<Integer> indexes;
    int index;
    public StockSpanner() {
        prices = new Stack<>();
        indexes = new Stack<>();
        index = 0;
        prices.push(Integer.MAX_VALUE);
        
indexes.push(index++);
    }
    
    public int next(int price) {
        while(!prices.isEmpty() && prices.peek() <= price) {
                prices.pop();
                indexes.pop();
        }
        if(prices.isEmpty()) {
            prices.push(price);
            indexes.push(index++);
            return 1;
        } else {
            int last = indexes.peek();
            prices.push(price);
            indexes.push(index++);
            return index - last - 1;
        }
    }
}
