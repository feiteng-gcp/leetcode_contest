class StockSpanner {
	int[] stack;
	int[] pos;
    int n;
	int index = 0;
    int max = 10005;
    public StockSpanner() {
        stack = new int[max];
        pos = new int[max];
        n = 0;
        index = 0;
    }
    
    public int next(int price) {
        while (n > 0 && stack[n - 1] <= price) {
        	--n;
        }
        int lastPos = n == 0 ? 0 : pos[n - 1] + 1;
        stack[n] = price;
        pos[n] = index++;
        ++n;
        return index - lastPos;
    }
}
