class StockSpanner {
    List<Integer> history;
    List<Integer> result;
    public StockSpanner() {
        history = new ArrayList<Integer>();
        result = new ArrayList<Integer>();
    }
    
    public int next(int price) {
        int cur = history.size();
        int count = 1;
        int p = cur - 1;
        while(p >= 0 && history.get(p) <= price){
            count += result.get(p);
            p -= 
result.get(p);
        }
        result.add(count);
        history.add(price);
        return count;
    }
}
