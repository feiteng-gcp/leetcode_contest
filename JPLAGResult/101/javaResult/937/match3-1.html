	class StockSpanner {
		int[] stack;

		int[] value;
		int sp;
		int gen = 0;

	    public StockSpanner() {
	        stack = new int[11000];
	        value = new int[11000];
	        sp = 0;
	        gen = 0;
	    }
	    
	    public int next(int price) {
	        while(sp > 0 && value[sp-1] <= price){
	        	sp--;
	        }
	        int ret = gen - (sp == 0 ? -1 : stack[sp-1]);
	        stack[sp] = gen++;
	        value[sp] = price;
	        sp++;
	        return ret;
	    }
	}
