class StockSpanner {

        int index = 0;
        Deque<int[]> stk;
        public StockSpanner() {
            index = 0;
            stk = new ArrayDeque<>();
        }

        public int next(int price) {
            int[] pair = new int[] {index++, price};
            while (!stk.isEmpty() && (stk.peek())[1] <= price) {
                pair[0] = (stk.pop())[0];
            }
            stk.push(pair);
            return index - pair[0];
        }
    }

 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
