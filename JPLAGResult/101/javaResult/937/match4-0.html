class StockSpanner {
    
    int[] stack;
    int[] list;
    int index1, index2;
    public StockSpanner() {
        stack = new int[10005];
        list = new int[10005];
        index1 = 0;
        index2 = 0;
        stack[index2++] = -1;
    }
    
    public int next(int price) {
        while (index2 != 1 && list[stack[index2-1]] <= price)
            index2--;
        int res = index1 - stack[index2-1];
        stack[index2++] = index1;
        list[index1++] = price;
        // index++;
        return res;
    }
}

 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
