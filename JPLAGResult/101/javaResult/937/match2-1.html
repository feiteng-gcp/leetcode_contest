class StockSpanner {

	TreeMap<Integer, Integer> priceMap = new TreeMap<Integer, Integer>();
	int cur
rent = 0;
	
    public StockSpanner() {
    	priceMap = new TreeMap<Integer, Integer>();
    }
    
    public int next(int price) {
    	priceMap.put(price, current);
    	current++;
    	Integer k = 0;
    
    	while((k = priceMap.lowerKey(price)) != null) {
    		priceMap.remove(k);
    	}
    	Integer retKey = priceMap.higherKey(price);//.getValue();
    	if(retKey == null) return current;
    	//System.out.println(priceMap.ceilingEntry(price).getKey() + ", " + priceMap.ceilingEntry(price).getValue());
    	//System.out.println(priceMap);
    	return current - 1 - priceMap.get(retKey);
    	
    }
}
