class StockSpanner {
    public List<Integer> history;
    public List<Integer> count;
    
    public StockSpanner() {
       this.history = new ArrayList<Integer>();
        this.count = new ArrayList<Integer>();
    }
    
    public int next(int price) {
        history.add(price);
        
        if(count.size() == 0) {
            count.add(1);
            return 1;
        } else {
            int newIndex = count.size();
            
            if(history.get(newIndex - 1) > price) {
                count.add(1);
                return 1;
            } else {
                int currentIndex = newIndex - 1;
                int currentCount = 1;
                
                while(currentIndex >= 0 && history.get(currentIndex) <= price) {
                    currentCount += count.get(currentIndex);
                    currentIndex -= count.get(currentIndex);
                } 
                
                count.add(currentCount);
                
                return currentCount;
            }
        }
    }
}
