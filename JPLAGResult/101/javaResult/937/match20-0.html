class StockSpanner {
    Deque<int[]> ans;
    public StockSpanner() {
        ans = new LinkedList<>();
    }
    
    public int next(int price) {
        int howMany = 1;
        while(!ans.isEmpty() && ans.peekLast()[1]<=price)
        {
            int[] next = ans.removeLast();
            howMany+=next[0];
        }
        ans.addLast(new int[]{howMany,price});
        return howMany;
    }
}
