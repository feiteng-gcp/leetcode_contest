class StockSpanner {
    List<Integer> dp, stock;
    int index = 0;
    public StockSpanner() {
        dp = new ArrayList<>();
        stock  = new ArrayList<>();
        stock.add(0);
        dp.add(0);
    }
    
    public int next(int price) {
        int i = index;
        int ans = 1;
        while (i > 0 && price >= stock.get(i)){
            ans += dp.get(i);
            i = i - dp.get(i);
        }
        stock.add(price);
        dp.add(ans);
        index++;
        return ans;
    }
}
