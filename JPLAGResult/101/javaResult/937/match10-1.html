class StockSpanner {
    HashMap<Integer, Integer> hashMap;
    Deque<Integer> stack;
    int day;
    public StockSpanner() {
        hashMap = new HashMap();
        stack = new ArrayDeque<Integer>();
        day = 0;
        
stack.push(-1);
    }
    
    public int next(int price) {
        while (stack.peek() != -1 && hashMap.get(stack.peek()) <= price) {
            int index = stack.pop();
            hashMap.remove(index);
        
}
        int span = day - stack.peek();
        hashMap.put(day, price);
        stack.push(day);
        day++;
        return span;
    }
}

 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
