class StockSpanner {
    
List<Integer> nums = new ArrayList<>(), cnt = new ArrayList<>();

    public StockSpanner() {        
    }
    
    public int next(int price) {

        int idx = nums.size()-1;
        int res = 1;
        while(idx>=0&&nums.get(idx)<=price){
            res += cnt.get(idx);
            idx -= cnt.get(idx);
        }
        nums.add(price);
        cnt.add(res);
        return res;
    }
}

/**
 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
