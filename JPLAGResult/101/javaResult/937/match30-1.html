class StockSpanner {
    
    
class Val{
        int val;
        int count;
        Val(int val, int count) {
            this.val = val;
            this.count = count;
        }
    }

    Val[] list = new Val[150007];
    int len;

    public StockSpanner() {
        len = 0;
    }
    
    public int findIndex(int p) {
        int l = 0, r = len-1;
        while (l < r) {
            int mid = (l+r) / 2;
            if (list[mid].val <= p) {
                r = mid;
            } else {
                l = mid+1;
            }
        }
        return l;
    }

    public int next(int price) {
        int count = 0, old = 0;
        int index = findIndex(price);
        if (index >= 0 && index < len) {
            if (list[index].val <= price) index--;
        }
        if (len == 0) {
            count = 0;
        }else if (index < 0) {
            count = list[len-1].count;
            len = 0;
        } else {
            count = list[len-1].count - list[index].count;
            len = index + 1;
            old = list[index].count;
        }
        list[len++] = new Val(price, count + 1 + old);
        return count + 1;
    }
}
