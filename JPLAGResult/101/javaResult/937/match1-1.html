class StockSpanner {
    class point{
        int val;
        int day;
        public point(int val){
            this.val = val;
            this.day = 1;
        }
    }
    
private Stack<point> decreasing;
    public StockSpanner() {
        decreasing = new Stack<point>();
    }
    
    public int next(int price) {
        point cur = new point(price);
        while(!decreasing.isEmpty() && decreasing.peek().val <= cur.val ){
            cur.day+=decreasing.pop().day;
        }
        decreasing.push(cur);
        return cur.day;
    }
}

 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
