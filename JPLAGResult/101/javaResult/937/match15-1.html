class StockSpanner {
    
    private List<Integer> nums;
    
    private List<Integer> indexes;

    public StockSpanner() {
        nums = new ArrayList<Integer>();
        indexes = new ArrayList<Integer>();
    }
    
    public int next(int price) {
        //System.out.println("Add: " + price);
        if (nums.isEmpty()) {
            nums.add(price);
            indexes.add(0);
            return 1;
        }
        
        int checker = nums.get(nums.size() - 1);
        if (checker > price) {
            nums.add(price);
            indexes.add(indexes.size());
            return 1;
        }
        
        int index = indexes.get(indexes.size() - 1);
        //System.out.println(checker + "=>" + index);
        while (index >= 0 && price >= checker) {
            int newIndex = indexes.get(index);
            if (newIndex < index) {
                index = newIndex;
            } else {
                index--;
                if (index < 0) {
                    break;
                }
            }
            checker = nums.get(index);
        }
        nums.add(price);
        indexes.add(index + 1);
        //System.out.println("Reach: " + index);
        return indexes.size() - 1 - index;
    }
}

 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
