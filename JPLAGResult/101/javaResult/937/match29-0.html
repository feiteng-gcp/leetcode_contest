class StockSpanner {

    Stack<Pair> spanner = new Stack<>();
    
    public StockSpanner() {
        
    }
    
    public int next(int price) {
        int span = 1;
        if(spanner.size() == 0){
            spanner.push(new Pair(price, 1));
            return span;
        }else{
            while(spanner.size() > 0){
                if(spanner.peek().price > price){
                    spanner.push(new Pair(price, span));
                    return span;
                }else{
                    Pair cur = spanner.pop();
                    span += cur.span;
                }
            }
            spanner.push(new Pair(price, span));
            return span;
        }
    }
    
    class Pair{
        int price;
        int span;
        public Pair(int p, int s){
            price = p;
            span = s;
        }
    }
}

 * int param_1 = obj.next(price);
