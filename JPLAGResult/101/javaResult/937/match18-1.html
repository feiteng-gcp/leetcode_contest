class StockSpanner {
    int[] cnt = new int[150000 + 1];
    int[] data = new int[150000 + 1];
    int p = 0;

    public StockSpanner() {
    }

    public int next(int price) {
      data[p] = price;
      if (p > 0 && data[p] >= data[p - 1]) {
        int k = p - 1;
        cnt[p] = 1;
        while (k >= 0 && data[p] >= data[k]) {
          cnt[p] += cnt[k];
          k -= cnt[k];
        }
      } else {
        cnt[p] = 1;
      }
      return cnt[p++];
    }
  }

 * Your StockSpanner object will be instantiated and called as such:
 * StockSpanner obj = new StockSpanner();
 * int param_1 = obj.next(price);
