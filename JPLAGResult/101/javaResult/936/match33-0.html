class RLEIterator {
    private int[] sequence;
    private int index;
    private int count;
    public RLEIterator(int[] A) {
        sequence = A;
        index = 0;
        if (A == null || A.length == 0) {
            count = 0;
        }
        else {
            count = sequence[index];
        }
    }
    
    public int next(int n) {
        if (sequence == null || index >= sequence.length - 1) {
            return -1;
        }
        while (n > count) {
            n -= count;
            index += 2;
            if (index >= sequence.length - 1) {
                return -1;
            }
            count = sequence[index];
        }
        if (index + 1 > sequence.length - 1) {
            return -1;
        }
        int value = sequence[index + 1];
        count -= n;
        if (count == 0) {
            index += 2;
            count = sequence[index];
        }
        return value;
    }
}
