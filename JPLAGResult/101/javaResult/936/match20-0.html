class RLEIterator {
    int[] RLE;
    int i;
    int count;
    boolean flag;
    public RLEIterator(int[] A) {
        RLE = A;
        i = 0;
        count = 0;
        flag = false;
    }
    
    public int next(int n) {
        if(flag) {
            return -1;
        }
        count += n;
        while(count > RLE[i]) {
            if(i >= RLE.length - 2) {
                flag = true;
                return -1;
            }
            count -= RLE[i];
            i += 2;
        }
        return RLE[i + 1];
    }
}
