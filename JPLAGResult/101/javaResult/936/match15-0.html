class RLEIterator {
    private int index, count;
    private int[] A;

    public RLEIterator(int[] A) {
        index = 0;
        count = 0;
        this.A = A;
        if (index < A.length) {
            count = A[0];
        }
    }
    
    public int next(int n) {
        while (n > count) {
            n -= count;
            index += 2;
            if (index < A.length) {
                count = A[index];
            } else {
                count = 0;
                return -1;
            }
        }
        count -= n;
        return A[index + 1];
    }
}
