class RLEIterator {
    
int[] tmp;
    int cur = 0;
    public RLEIterator(int[] A) {
        tmp = A;
        
    }
    
    public int next(int n) {
        while (cur < tmp.length && tmp[cur] < n) {
            n -= tmp[cur];
            cur += 2;
        }
        if (cur < tmp.length && tmp[cur] >= n) {
            tmp[cur] -= n;
            return tmp[cur + 1];
        }
        return -1;
    }
}
