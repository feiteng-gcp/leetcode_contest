class RLEIterator {
    private int[] ma = null;
    private int ima = 0;
    private int remaining = 0;
    
    public RLEIterator(int[] A) {
        ma = A;
        ima = 0;
        remaining = ma[ima];
        
    }
    
    public int next(int n) {
        
        while(n > remaining) {
            if (remaining == -1) return -1;
            n = n - remaining;
            if (ima+2<ma.length) {
                ima += 2;
                remaining = ma[ima];
            } else {
                remaining = -1;
            }
        }
        
        remaining = remaining - n;
        return ma[ima+1];
        
    }
}
