class RLEIterator {
        
        private static class Pair {
            final int number;
            int count;
            
            Pair(int number, int count) {
                this.number = number;
                this.count = count;
            }
            
        }
        
        private final Deque<Pair> deque = new LinkedList<>();
        
        public RLEIterator(int[] A) {
            for (int i = 0; i < A.length; i += 2) {
                int count = A[i];
                int number = A[i + 1];
                Pair pair = new Pair(number, count);
                deque.addLast(pair);
            }
        }
        
        public int next(int n) {
            int popNum = n;
            while (!deque.isEmpty()) {
                Pair p = deque.pollFirst();
                assert p != null;
                int count = p.count, number = p.number;
                if (count > popNum) {
                    p.count -= popNum;
                    deque.addFirst(p);
                    return number;
                } else if (count < popNum) {
                    popNum -= count;
                    // many left to pop
                } else {
                    // count == popNum
                    return number;
                }
            }
            return -1;
        }
        
    }

/**
 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
