class RLEIterator {
    int[] nums;
    int iter = 0;
    public RLEIterator(int[] A) {
        nums = new int[A.length];
        for(int i = 0 ; i < A.length ; i++) {
            nums[i] = A[i];
        }
    
}
    
    public int next(int n) {
        if(iter >= nums.length) {
            return -1;
        }
        nums[iter] -= n;
        int rem = nums[iter];
        
while(rem < 0) {
            iter += 2;
            if(iter >= nums.length) {
                return -1;
            }
            nums[iter] += rem;
            rem = nums[iter];
        }
        return nums[iter + 1];
    }
}
