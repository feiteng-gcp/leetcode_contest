class RLEIterator {
    class pair{
        int num;
        int cnt;
        pair(int n,int c){
            num = n;
            cnt = c;
        }
    }
    Queue<pair> q;
    

    public RLEIterator(int[] A) {
        q = new LinkedList<>();
        for(int i=0; i<A.length; i+=2){
                q.offer(new pair(A[i+1],A[i]));
        }
    }
    
    public int next(int n) {
        int c = n;
        while(c>0){
            if(q.isEmpty())
                return -1;
            if(q.peek().cnt<c){
                c -= q.poll().cnt;
            }
            else if(q.peek().cnt==c){
                return q.poll().num;
            }
            else{
                q.peek().cnt -= c;
                return q.peek().num;
            }
        }
        return -1;
    }
}

 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
