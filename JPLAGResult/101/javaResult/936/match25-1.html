class RLEIterator {
    int[] data;
    int currPos;
    int currCnt;

    public RLEIterator(int[] A) {
      this.data = A;
      this.currCnt = 0;
      this.currPos = 0;
    }

    public int next(int n) {
      while (currPos < data.length && n > data[currPos] - currCnt) {
        n -= (data[currPos] - currCnt);
        currPos += 2;
      }
      if (currPos >= data.length) {
        return -1;
      }
      data[currPos] -= n;
      return data[currPos + 1];
    }
  }

 * Your RLEIterator object will be instantiated and called as such:
 * RLEIterator obj = new RLEIterator(A);
 * int param_1 = obj.next(n);
