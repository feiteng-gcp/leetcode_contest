class RLEIterator {
    int[] A;
    int cur, count, N;
    public RLEIterator(int[] A) {
        this.A = A;
        N = A.length;
        cur = 0;
        count = 0;
    }
    
    public int next(int n) {
        if (cur >= N) return -1;
        int res = -1;
        while (n > 0 && cur < N) {
            int rem = A[cur] - count;
            int d = Math.min(rem, n);
            n -= d;
            count += d;
            res = A[cur + 1];
            if (count == A[cur]) {
                cur += 2;
                count = 0;
            }
        }
        if (n == 0) return res; else return -1;
    }
}
