class RLEIterator {
    Long i;
    Long l;
    Long count; 
    TreeMap<Long, Integer> map;
    public RLEIterator(int[] A) {
        i = new Long(0);
        l = A != null ? A.length : new Long(0);
        map = new TreeMap<>();
        count = new Long(0);
        for(int j = 0; j < l-1 ; j +=2) {
            if(A[j] != 0){
                count+= A[j];
                map.put(count, A[j+1]);
            }
        }
    
}
    
    public int next(int n) {
        i += n;
        if( i > count){
            return -1;
        }
        Long key = map.ceilingKey(i);
        
        return key != null ? map.get(key) : -1;
    }
}
