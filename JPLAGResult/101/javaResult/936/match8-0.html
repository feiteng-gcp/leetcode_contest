class RLEIterator {

    int[] A;
    int index = 0;

    public RLEIterator(int[] A) {
        this.A = A;
    }

    public int next(int n) {
        if (index>=A.length) return -1;
        if (n<=A[index]){
            A[index]-=n;
            return A[index+1];
        }else {
            int oi = index;
            index+=2;
            return next(n-A[oi]);
        }
    }
}
