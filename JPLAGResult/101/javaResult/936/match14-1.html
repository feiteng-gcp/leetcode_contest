import java.util.Arrays;

public class RLEIterator {

    private int i = 0;
    private int[] arr;

    public RLEIterator(int[] A) {
        arr = Arrays.copyOf(A, A.length);
    
}

    public int next(int n) {
        while (n > 0) {
            if (i == arr.length) {
                return -1;
            }
            if (arr[i] >= n) {
                arr[i] -= n;
                
return arr[i + 1];
            } else {
                n -= arr[i];
                i += 2;
            }
        }

        return -1;
    }
}
