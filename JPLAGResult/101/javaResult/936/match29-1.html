    public RLEIterator(int[] A) {
    	seqMap = new TreeMap<Long, Integer>();
    	long cur = 0;
    	for(int i = 0; i < A.length; i++) {
    		if(A[i] == 0) {
    			i++;
    			continue;
    		}
    		seqMap.put(cur, A[i+1]);
            //System.out.println("put: " + cur + ", " + A[i+1]);
            cur += (long)A[i];
            i++;
    	}
    	seqMap.put(cur, -1);
    	current = -1;
    }
    
    public int next(int n) {
        current += (long)n;
        return(seqMap.get(seqMap.floorKey(current)));
    }
}
