class RLEIterator {

    int start;
    
int index;
    int len;
    int[] array;
    public RLEIterator(int[] A) {
        start = 0;
        index = 0;
        len = A.length;
        array = A;
    
}
    
    public int next(int n) {
        while (index < len && n > 0) {
            if (n > array[index] - start) {
                n -= array[index] - start;
                index += 2;
                start = 0;
            }
            else if (n == array[index] - start) {
                int oldIndex = index;
                start = 0;
                index += 2;
                return array[oldIndex + 1];                
            }
            else {
                start += n;        
                return array[index + 1];
            }
        }
        return -1;
    }
}

 * RLEIterator obj = new RLEIterator(A);
