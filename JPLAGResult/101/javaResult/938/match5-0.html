class Solution {
    public int atMostNGivenDigitSet(String[] D, int N) {
        long n = (long)N;
        String s = ""+n;
        int[] full = new int[10];
        full[0] = 1;
        for (int i = 1; i < 10; i++) {
            full[i] = full[i-1]*D.length;
        }
        int ans = 0;
        for (int i = 1; i < s.length(); i++)
            ans += full[i];
        boolean sgi = true;
        int dl = s.length();
        for (char c : s.toCharArray()) {
            dl--;
            if (!sgi)
                return ans;
            sgi = false;
            for (String da : D) {
                char d = da.charAt(0);
                if (d==c&&dl==0)
                    ans++;
                if (d==c)
                    sgi = true;
                if (d>=c)
                    break;
                if (d<c)
                    ans += full[dl];
            }
        }
        return ans;
    }
}
