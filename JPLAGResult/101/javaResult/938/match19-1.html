class Solution {
    public int atMostNGivenDigitSet(
String[] D, int N) {
        if (D == null || D.length == 0 || N <= 0) {
            return 0;
        }
        
        int result = 0;
        String num = N + "";
        int len = num.length();
        int multiple = 1;
        for (int i = 0; i < len - 1; i++) {
            multiple *= D.length;
            result += multiple;
        }
        
        //System.out.println(result);
        
        // [(index, has smaller -> 1 / 0)]
        int[][] dp = new int[len][2];
        for (int i = 0; i < len; i++) {
            int val = Integer.parseInt(num.charAt(i) + "");
            int smallerCount = 0;
            int hasNumber = 0;
            for (int j = 0; j < D.length; j++) {
                int dVal = Integer.parseInt(D[j]);
                if (dVal < val) {
                    smallerCount++;
                } else if (dVal == val) {
                    hasNumber = 1;
                } else {
                    break;
                }
            }
            
            //System.out.println("smaller: " + smallerCount + " hasTarget: " + hasNumber);
            
            if (i == 0) {
                dp[i][0] = smallerCount;
                dp[i][1] = hasNumber == 1 ? 1 : 0;
            } else {
                dp[i][0] = dp[i - 1][0] * D.length + dp[i - 1][1] * smallerCount;
                if (hasNumber == 1 && dp[i - 1][1] == 1) {
                    dp[i][1] = 1;
                }
            }
            
            //System.out.println(dp[i][0] + " - " + dp[i][1]);
        }
        result += (dp[len - 1][0] + dp[len - 1][1]);
        return result;
    }
}
