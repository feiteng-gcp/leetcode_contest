class Solution {
	char[] D;
    public int atMostNGivenDigitSet(String[] D, int N) {
        this.D = new char[D.length];
        for (int i = 0; i < D.length; ++i) {
        	this.D[i] = D[i].charAt(0);	
        }
        return solve(String.valueOf(N));
    }
    
    int rec(String str, int power) {
    	if (str.length() == 0) return 1;
    	int res = 0;
    	power /= D.length;
    	for (int i = 0; i < D.length; ++i) {
    		if (D[i] < str.charAt(0)) {
    			res += power;
    		} else if (D[i] == str.charAt(0)) {
    			res += rec(str.substring(1), power);
    		}
    	}
    	return res;
    }
    
    int solve(String str) {
    	int n = str.length();
    	int res = 0, power = 1;
    	for (int i = 1; i + 1 <= n; ++i) {
    		power *= D.length;
    		res += power;
    	}
    	for (int i = 0; i < D.length; ++i) {
    		if (D[i] < str.charAt(0)) {
    			res += power;
    		} else if (D[i] == str.charAt(0)) {
    			res += rec(str.substring(1), power);
    		}
    	}
    	return res;
    }
    
    public static void main(String[] args) {
    	Solution solution = new Solution();
    	String[] D = {"1","2","3","6","7","8"};
    	int N = 211;
    	int res = solution.atMostNGivenDigitSet(D, N);
    	System.out.println(res);
    }
}
