class Solution {
  public int atMostNGivenDigitSet(String[] D, int N) {
    int[] full = new int[10];
    full[0] = 1;
    for (int i = 1; i < full.length; ++i) {
      full[i] = full[i - 1] * D.length;
    }
    //
    int sum = 0;
    char[] arr=Integer.toString(N).toCharArray();
    
for (int n = 1; n <= arr.length; n++) {
      sum += count(D, n, 0, arr, full);
    }
    return sum;
  }

  private int count(String[] d, int n, int i, char[] charArray, int[] full) {
    //System.out.println(n+" "+i);
    if (i >= n) {
      return 1;
    }
    if (n < charArray.length) {
      return full[n];
    }
    int sum = 0;
    int cnt = 0;
    int k;
    for (k = 0; k < d.length && d[k].charAt(0) < charArray[i]; ++k) {
      cnt++;
    }
    if (cnt > 0) {
      sum += cnt * full[n - i - 1];
    }
    if (k < d.length && d[k].charAt(0) == charArray[i]) {
      sum += count(d, n, i + 1, charArray, full);
    }
    return sum;
  }
  
}
