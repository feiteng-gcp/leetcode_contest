class Solution {
    public int atMostNGivenDigitSet(String[] D, int N) {
        int result = 0,ln = 1,len = D.length;
        String ns = String.valueOf(N);
        int nsl = ns.length();
        for (int i = 1;; i++)
            if (i < nsl) {
                ln *= len;
                result += ln;
            } else {
                for (int a = 0, j = 0; a < len && j< nsl; a++) {
                    String aD = D[a];
                    if (aD.charAt(0) < ns.charAt(j)) result += ln;
                    else if (aD.charAt(0) == ns.charAt(j)) {
                        if (j==nsl-1) return ++result;
                        j++;
                        ln /= len;
                        a = -1;
                    } else return result;
                }
                return result;
            }
    }
}
