class Solution {
    int all = 0;

    public int atMostNGivenDigitSet(String[] D, int N) {
        String nn = ""+N;
        int zs = D.length;
        int l = nn.length();
        int c = zs;
        
for(int i=1;i<l;i++){
            all += c;
            c = c*zs;
        }
        
        boolean go = true;
        for(int x=0;x<nn.length()&&go;x++){
        go = false;
        int t = 0;
        for(String num:D){
            int n = (num.charAt(0)-'0');
            if(n<(nn.charAt(x)-'0')){
                t++;
            }
            if(n==(nn.charAt(x)-'0')){
                go = true;
            }
        }
        c = 1;
        for(int i=1;i<l-x;i++){
            c = c*zs;
        }
        all += t*c;
        
        }
        
      if(go){
          all++;
      }
        return all;
        
    }
}
