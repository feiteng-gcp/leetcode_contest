class Solution {
    public int atMostNGivenDigitSet(String[] D, int N) {
        int digit = 0, copy = N;
        String number = "";
        while(copy > 0){
            number = (copy % 10) + number;
            copy = copy/10;
            digit++;
        }
        int result = 0;
        for(int i = 0; i < digit - 1; i++) result += Math.pow(D.length, i+1);
        char[] choice = new char[D.length];
        for(int i = 0; i < D.length; i++) choice[i] = D[i].toCharArray()[0];
        
        //System.out.println("at this point, result is " + result);
        
        for (int i = 0; i < choice.length && choice[i] <= number.charAt(0); i++){
            if (choice[i] == number.charAt(0)){
                result += helper(choice, number, N, i, "", digit);                    
            }else{
                result += Math.pow(D.length, number.length() - 1);
            }
        }
        return result;
    }
    
    private int helper(char[] choice, String number, int N, int index, String cur, int digit){
        cur += choice[index];
        //System.out.println("current is " + cur);
        if (cur.length() == digit){
            //System.out.println(cur);
            return 1;
        }
        int result = 0;
        for (int i = 0 ; i < choice.length && choice[i] <= number.charAt(cur.length()); i++){
            
            if (choice[i] == number.charAt(cur.length())){
                result += helper(choice, number, N, i, new String(cur), digit);
            }else{
                result += Math.pow(choice.length, number.length() - cur.length() - 1);
            }            
        }
        return result;
    }
}
