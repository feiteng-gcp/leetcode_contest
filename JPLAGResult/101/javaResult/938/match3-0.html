class Solution {
    long sum[] = new long[15];
    long mul[] = new long[15];
    public int atMostNGivenDigitSet(String[] D, int N) {
        String n = Integer.valueOf(N).toString();
        mul[0] = 1;
        for (int i = 1; i <= 11; i++) mul[i] = mul[i-1] * D.length;
        sum[0] = 0;
        for (int i = 1; i <= 11; i++) sum[i] = sum[i-1] + mul[i];

        int ans = (int)sum[n.length() - 1];
        Integer[][] memo = new Integer[15][15];
        for (int i = 0; i < D.length; i++) {
            ans += dfs(0, i, n, D, memo);
        }
        return ans;
    }
    int dfs(int nidx, int didx, String n, String[] D, Integer[][] memo) {
        if (memo[nidx][didx] != null) {
            return memo[nidx][didx];
        }
        int ans = 0;
        int diff = D[didx].compareTo(n.substring(nidx, nidx + 1));
        if (diff > 0) {
            ans = 0;
        }
        else if (diff == 0) {
            if (nidx == n.length() - 1) ans = 1;
            else {
                for (int i = 0; i < D.length; i++) {
                    ans += dfs(nidx + 1, i, n, D, memo);
                }
            }
        }
        else {// diff < 0
            if (nidx == n.length() - 1) ans = 1;
            else ans = (int)mul[n.length() - 1 - nidx];
        }
        return memo[nidx][didx] = ans;
    }
}
