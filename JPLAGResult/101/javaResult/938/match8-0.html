class Solution {
    
    private static int[] getDigits(String[] D) {
        int n = D.length;
        int[] digits = new int[n];
        for (int i = 0; i < n; i++) {
            digits[i] = Integer.parseInt(D[i]);
        }
        return digits;
    }
    
    private static int pow(int base, int exp) {
        int product = 1;
        for (int j = 0; j < exp; j++) {
            product *= base;
        }
        return product;
    }
    
    public int atMostNGivenDigitSet(String[] D, int N) {
        int[] digitSet = getDigits(D); // sorted
        int[] digits = getDigits(String.valueOf(N).split(""));
        // System.out.println(Arrays.toString(digits));
        int sum = 0;
        // First find all (digits.length - 1) digit numbers or less
        for (int i = 1; i < digits.length; i++) {
            int t = pow(digitSet.length, i);
            // System.out.println(t);
            sum += t;
        }
        // Then analyze the equal digit case one digit by one
        // Highest digit first
        for (int i = 0; i < digits.length; i++) {
            int digit = digits[i];
            int numLess = 0;
            boolean hasEqual = false;
            for (int v : digitSet) {
                if (digit > v) {
                    numLess++;
                } else if (digit == v) {
                    if (i == digits.length - 1) {
                        numLess++;
                    }
                    hasEqual = true;
                    break;
                } else {
                    hasEqual = false;
                    break;
                }
            }
            // Less than part
            int a = numLess * pow(digitSet.length, digits.length - i - 1);
            // System.out.println(a + " " + numLess);
            sum += a;
            if (!hasEqual) {
                break;
            }
        }
        return sum;
    }
}
