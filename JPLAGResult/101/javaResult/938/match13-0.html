class Solution {
    public int find(int dp[], Stack<Integer> st, String[] d){
        int ans=0;
        if(st.isEmpty())
            return 1;
            int top = st.pop();
            int s = st.size();
            for(int i=0; i<d.length; i++){
                if(Integer.parseInt(d[i])<top){
                    ans+=dp[s];
                }
                else if(Integer.parseInt(d[i])==top){
                    ans+=find(dp,st,d);
                    return ans;
                }
                else
                    return ans;
            }
        return ans;
    }
    public int atMostNGivenDigitSet(String[] D, int N) {
        int l = D.length;
        int n = N,cnt=0;
        Stack<Integer> st= new Stack<>();
        while(n>0){
            int r = n%10;
            st.push(r);
            n = n/10;
        }
        int dp[] = new int[st.size()+1];
        dp[0] = 1;
        int total=0;
        for(int i=1; i<st.size(); i++){
            dp[i] = l*dp[i-1];
            total+=dp[i];
        }
        total+=find(dp,st,D);
        return total;
    }
}
