class Solution {
    public int atMostNGivenDigitSet(String[] ds, int num) {
        int n = ds.length;
        int sz = 0;
        int cnum = num;
        
        while(cnum > 0)
        {
            cnum /= 10;
            sz++;
        }
        
        int[] numa = new int[sz];
        
        cnum = num;
        int ci = sz - 1;    
        
        while(cnum > 0)
        {
            numa[ci] = cnum % 10;
            cnum /= 10;
            ci--;
        }
        
        int[] d = new int[n];
        
        for(int i = 0; i < n; i++)
        {
            d[i] = ds[i].charAt(0) - '0';
        }
        
        long[][] dp = new long[sz][2];
        
        dp[sz - 1][1] = n;
        
        for(int i = 0; i < n; i++)
        {
            if(d[i] <= numa[sz - 1])
                dp[sz - 1][0]++;
        }
        
        long ans = 0;
        
        ans += dp[sz - 1][0];
        
        for(int i = sz - 2; i >= 0; i--)
        {
            // if(numa[i] > 0)
            //     dp[i][0] = dp[i + 1][1];
            // else
            //     dp[i][0] = dp[i + 1][0];
            
            for(int j = 0; j < n; j++)
            {
                if(d[j] < numa[i])
                {
                    dp[i][0] += dp[i + 1][1];
                }
                else if(d[j] == numa[i])
                {
                    dp[i][0] += dp[i + 1][0];
                }
            }
            
            dp[i][1] = n * dp[i + 1][1];
            
            long cans = dp[i][0];
            
            cans += dp[i + 1][1] - dp[i + 1][0];
            
            // for(int j = 0; j < n; j++)
            // {
            //     if(d[j] < numa[i])
            //     {
            //         cans += dp[i + 1][1] - dp[i + 1][0];
            //     }
            // }
            
            ans += cans;
        }
        
        return (int)(ans);
        
    }
}
