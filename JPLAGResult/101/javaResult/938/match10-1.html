class Solution {

    int res = 0;
    public  int atMostNGivenDigitSet(String[] D, int N) {

        String val = String.valueOf(N);
        int cur = D.length;
        int[] dp = new int[val.length()];
        dp[0] = 1;
        
for (int i = 1; i <= val.length() - 1; i++) {
            dp[i] = cur;
            res += cur;
            cur *= D.length;
        }
        dfs(D, val, 0, dp);
        return res;
    }
    public void dfs(String[] D, String val, int index, int[] dp) {
        if (index == val.length()) {
            res ++;
            return;
        }
        for (int i = 0; i < D.length; i++) {
            if (D[i].charAt(0) < val.charAt(index)) {
                res += dp[val.length() - 1 - index];
            } else if (D[i].charAt(0) == val.charAt(index)) {
                dfs(D, val, index + 1, dp);
            }
        }
    }
}
