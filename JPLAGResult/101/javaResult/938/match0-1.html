class Solution {
    public int atMostNGivenDigitSet(String[] D, int N) {
        String bound = Integer.toString(N);
        
        
int[] numbers = new int[D.length];
        for(int i = 0; i < D.length; i ++) {
            numbers[i] = Integer.parseInt(D[i]); 
        }
        
        int numCount = numbers.length;
        int boundCount = bound.length();
        int result = 0;
        
        for(int i = 1;  i < boundCount; i ++) {
            result += Math.pow(numCount, i);
        }
        
        for(int i = 0; i < boundCount; i ++) {
            int num = bound.charAt(i) - '0';
            int head = 0;
            
            while(head < numbers.length && numbers[head] < num) {
                head ++;
            }
            
            result += head * Math.pow(numCount, boundCount - 1 - i);
            
            if(head >= numbers.length || numbers[head] != num) {
                break;
            
} else if(i == boundCount - 1) {
                result += 1;
            }
        }
        
        return result;
    }
}
