class Solution {
    public int atMostNGivenDigitSet(String[] D, int N) {
        String str = String.valueOf(N);
        int len = str.length();
        
int cnt = D.length, res = 0, seed = 1;
        int[] dp = new int[len+1];
        dp[0] = 1;
        for(int i=0;i<len-1;i++){
            seed *= cnt;
            dp[i+1] = seed;
            res += seed;
        }
        for(int i=0;i<len;i++){
            char cur = str.charAt(i);
            int idx = Arrays.binarySearch(D, ""+cur);
           // if(idx>=0&&i==len-1) res++; // for the last equal case
            if(idx<0){
                idx = -(idx+1);
                res += idx*dp[len-i-1];
                break;
            }
            else{
                res += idx*dp[len-i-1];        
                if(i==len-1) res++;
            }

        }
        return res;
        
    }
}
