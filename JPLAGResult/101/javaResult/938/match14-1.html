class Solution {
    public int atMostNGivenDigitSet(String[] D, int N) {
        int res = 0, size = (N + "").length(), cur = 1, base = 1;
        if (N < 10) {
            for (String d : D) {
                if (d.charAt(0) - '0' <= N)
                    res++;
            }   
            return res;
        }
        for (int i = 1; i < size; i++) {
            cur *= D.length;
            res += cur;
            base *= 10;
        }
        char[] ch = (N + "").toCharArray();
        for (String s : D) {
            char c = s.charAt(0);
            if (c < ch[0]) {
                res += cur;
            } else if (c == ch[0]) {
                res += helper(D, N - (c - '0') * base, ch.length-1);
            } else {
                break;
            
}
        }
        return res;
    }
    
    public int helper(String[] D, int N, int digit) {
        int res = 0, size = (N + "").length(), cur = 1, base = 1;
        if (N < 10) {
            if (digit > 1)
                return 0;
            for (String d : D) {
                if (d.charAt(0) - '0' <= N)
                    res++;
            }   
            return res;
        }
        char[] ch = (N + "").toCharArray();
        if (digit < ch.length) {
            res = 1;
            for (int i = 1; i <= digit; i++)
                res *= D.length;
            return res;
        } else if (digit == ch.length) {
            for (int i = 1; i < ch.length; i++) {
                cur *= D.length;
                base *= 10;
            }
            for (String s : D) {
                char c = s.charAt(0);
                if (c < ch[0]) {
                    res += cur;
                } else if (c == ch[0]) {
                    res += helper(D, N - (c - '0') * base, digit - 1);
                } else {
                    break;
                }
            }
            return res;
        } else {
            return 0;
        }
    }
}
