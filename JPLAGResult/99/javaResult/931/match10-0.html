class FreqStack {
   static class Node {
      public int val;
      public int seq;
      public int freq;
    }

    int globalSeq = 0;
    Map<Integer, Integer> freqMap = new HashMap<>();
    PriorityQueue<Node> stack;

    public FreqStack() {
      stack = new PriorityQueue<>(16, new Comparator<Node>() {
        @Override
        public int compare(Node o1, Node o2) {
          if (o1.freq == o2.freq) {
            return o2.seq - o1.seq;
          } else {
            return o2.freq - o1.freq;
          }
        }
      });
    }

    public void push(int x) {
      Integer freq = freqMap.get(x);
      freq = freq == null ? 0 : freq;
      Node n = new Node();
      n.val = x;
      n.seq = globalSeq++;
      n.freq = ++freq;
      freqMap.put(x, freq);
      stack.add(n);
    }

    public int pop() {
      return stack.poll().val;
    }
}
