class FreqStack
{
    class Node
    {
        public int val;
        public Stack<Integer> stack;

        public Node(int val, int height)
        {
            this.val = val;
            this.stack = new Stack<>();
            this.stack.push(height);
        }

        public void push(int height)
        {
            this.stack.push(height);
        }

        public void pop()
        {
            this.stack.pop();
        }
    }

    TreeSet<Node> ts;
    HashMap<Integer, Node> map;
    int top;

    public FreqStack()
    {
        ts = new TreeSet<>((a, b) ->
        {
            if (a.stack.size() != b.stack.size()) return b.stack.size() - a.stack.size();
            return b.stack.peek() - a.stack.peek();
        });
        map = new HashMap<>();
        top = 0;
    }

    public void push(int x)
    {
        if (map.containsKey(x))
        {
            Node node = map.get(x);
            ts.remove(node);
            node.push(top++);
            ts.add(node);
        }
        else
        {
            Node node = new Node(x, top++);
            map.put(x, node);
            ts.add(node);
        }
    }

    public int pop()
    {
        Node top = ts.pollFirst();
        top.pop();
        if (top.stack.size() == 0)
        {
            map.remove(top.val);
        }
        else
        {
            ts.add(top);
        }
        return top.val;
    }
}
