class FreqStack {
    
    Map<Integer, Node> map;
    TreeSet<Node> set;
    int gTime;
    
    public FreqStack() {
        set = new TreeSet<>();
        map = new HashMap<>();
        gTime = 0;
    
}
    
    public void push(int x) {
        if (!map.containsKey(x)) {
            Node n = new Node(x);
            n.timer.addLast(gTime++);
            map.put(x, n);
            set.add(n);
        } else {
            Node n = map.get(x);
            set.remove(n);
            n.freq++;
            n.timer.addLast(gTime++);
            set.add(n);
        }
    }
    
    public int pop() {
        Node top = set.first();
        set.remove(top);
        top.freq--;
        top.timer.removeLast();
        if (top.freq == 0) {
            map.remove(top.val);
        } else {
            set.add(top);
        }
        return top.val;
    }
}

class Node implements Comparable<Node> {
    int val;
    int freq;
    Deque<Integer> timer;
    public Node(
int v) {
        val = v;
        freq = 1;
        timer = new ArrayDeque<>();
    }
    public int hashCode() {
        return val;
    }
    public boolean equals(Object o) {
        if (!(o instanceof Node)) return false;
        Node n = (Node) o;
        return val == n.val;
    }
    public int compareTo(
Node n) {
        if (freq == n.freq) return Integer.compare(n.timer.peekLast(), timer.peekLast());
        return Integer.compare(n.freq, freq);
    }
}
