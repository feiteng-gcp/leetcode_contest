class FreqStack {

    TreeMap<Integer, Stack<Integer>> stacks = new TreeMap<>();
    TreeMap<Integer, Integer> cnts = new TreeMap<>();

    public FreqStack() {
    }

    public void push(int x) {
        int freq = cnts.getOrDefault(x, 0) + 1;
        cnts.put(x, freq);
        stacks.computeIfAbsent(freq, k -> new Stack<>()).push(x);
    }

    public int pop() {
        int max = stacks.lastKey();
        Stack<Integer> vals = stacks.get(max);
        int val = vals.pop();
        if (vals.isEmpty())
            stacks.remove(max);
        cnts.compute(val, (v, c) -> c > 1 ? c-1 : null);
        return val;
    }
}
