class FreqStack {
    class Node{
        int freq;
        int timestamp;
        int val;
        //int prev;
        public Node(int val,int timestamp,int freq){
            this.val = val;
            this.timestamp = timestamp;
            this.freq = freq;
        
}
    }
    HashMap<Integer,Integer> map;
    PriorityQueue<Node> pq;
    int count;
    public FreqStack() {
        count = 0;
        map = new HashMap<Integer,Integer>();
        pq = new PriorityQueue<Node>(new Comparator<Node>(){
            public int compare(Node a,Node b){
                if(a.freq == b.freq){
                    return b.timestamp - a.timestamp;
                }
                else{
                    return b.freq-a.freq;
                }
            }
        });
    }
    
    public void push(int x) {
        pq.offer(new Node(x,count++,map.getOrDefault(x,0)+1));
        map.put(x,map.getOrDefault(x,0)+1);
    }
    
    public int pop() {
        Node res = pq.poll();
        if(map.get(res.val) == 1)
            map.remove(res.val);
        else
            map.put(res.val,map.get(res.val)-1);
        return res.val;
    }
}
