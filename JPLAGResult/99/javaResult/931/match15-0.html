public class FreqStack {
    TreeMap<Integer, Set<Integer>> treeMap;
    Map<Integer, Integer> map;
    Stack<Integer> st;
    public FreqStack() {
        treeMap = new TreeMap<>(); //map freq to listofnumber
        map = new HashMap<>(); //map integer to freq
        st = new Stack<>();
    }
    
    public void push(int x) {
        st.push(x);
        if (map.containsKey(x)){
            int preF = map.get(x);
            treeMap.get(preF).remove(x);
            if (!treeMap.containsKey(preF + 1)){
                treeMap.put(preF + 1, new HashSet<>());
            }
            treeMap.get(preF + 1).add(x);
            map.put(x, map.get(x) + 1);
        } else {
            map.put(x, 1);
            if (!treeMap.containsKey(1)){
                treeMap.put(1, new HashSet<>());
            }
            treeMap.get(1).add(x);
            map.put(x, 1);
        }
    }
    
    public int pop() {
        int max = treeMap.lastKey();
        int ans = -1;
        Set<Integer> set = treeMap.get(max);
        
        Stack<Integer> temp = new Stack<>();
        while (!set.contains(st.peek())){
            temp.push(st.pop());
        }
        //System.out.print(st.size());
        ans = st.pop();
        //reset freq
        int pref = map.get(ans);
        treeMap.get(max).remove(ans);
        if (treeMap.get(max).size() == 0){
            treeMap.remove(max);
        }
        if (pref - 1 > 0) {
            map.put(ans, pref - 1);
            if (!treeMap.containsKey(pref - 1)){
                treeMap.put(pref - 1, new HashSet<>());
            }
            treeMap.get(pref - 1).add(ans);
        } else {
            map.remove(ans);
        }

        while (!temp.isEmpty()){
            st.push(temp.pop());
        }
        return ans;
    }
}
