class FreqStack {
    class Tuple {
        int val;
        int count;
        int time;
        public Tuple(int val, int count, int time) {
            this.val = val;
            this.count = count;
            this.time = time;
        }
    }
    
    int timestamp = 0;
    Stack<Integer> stack = new Stack<>();
    Map<Integer, LinkedList<Tuple>> map = new HashMap<>();
    PriorityQueue<Tuple> pq = new PriorityQueue<Tuple>((a, b) -> a.count==b.count ? b.time - a.time : b.count - a.count);

    public FreqStack() {
        
    }
    
    public void push(int x) {
        timestamp++;
        stack.push(x);
        if (!map.containsKey(x)) {
            map.put(x, new LinkedList<Tuple>());
        }
        LinkedList<Tuple> value = map.get(x);
        Tuple tuple = new Tuple(x, 1, timestamp);
        if (value.isEmpty()) {
            tuple = new Tuple(x, 1, timestamp);
            value.add(tuple);
        }
        else {
            tuple = new Tuple(x, value.get(value.size() - 1).count + 1, timestamp);
            value.add(tuple);
        }
        map.put(x, value);
        pq.offer(tuple);
    }
    
    public int pop() {
        
        Tuple pop = pq.poll();
        int x = pop.val;
        LinkedList<Tuple> value = map.get(x);
        value.removeLast();
        map.put(x, value);
        
        return x;
    }
}
