class FreqStack {

    PriorityQueue<int[]> pq;
    Map<Integer, Integer> map;
    int time;
    
    public FreqStack() {
        pq = new PriorityQueue<>((a,b) -> (a[1] == b[1]) ? b[2] - a[2] : b[1] - a[1]);    
        map = new HashMap<>();
        time = 0;
    }
    
    public void push(int x) {
        time++;
        int cnt = map.getOrDefault(x, 0) + 1;
        map.put(x, cnt);
        pq.add(new int[]{x, cnt, time});
    }
    
    public int pop() {
        int[] res = pq.poll();
        return res[0];
    }
}
