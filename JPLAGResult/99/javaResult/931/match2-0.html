class FreqStack {
    Map<Integer, Set<Integer>> freObj = null;
    Map<Integer, ArrayList<Integer>> frePos = null;
    Map<Integer, List<Integer>> bucket = null;
    Map<Integer, Integer> freTable = null;
    int n = 0;
    int max = 0;
    public FreqStack() {
        freObj = new HashMap<>();
        frePos = new HashMap<>();
        bucket = new HashMap<>();
        freTable = new HashMap<>();
    }
    
    public void push(int x) {
        if (!freTable.containsKey(x)) {
            freTable.put(x, 1);
        } else {
            freTable.put(x, freTable.get(x) + 1);
        }
        if (frePos.get(freTable.get(x)) == null) {
            frePos.put(freTable.get(x), new ArrayList<Integer>());
        }
        if (freObj.get(freTable.get(x)) == null) {
            freObj.put(freTable.get(x), new HashSet<Integer>());
        }
        if (freObj.get(freTable.get(x) - 1) != null) {
            freObj.get(freTable.get(x) - 1).remove(x);
        }
        freObj.get(freTable.get(x)).add(x);
        frePos.get(freTable.get(x)).add(x);
        // if (frePos.get(freTable.get(x) - 1) != null) {
        //     frePos.get(freTable.get(x) - 1).remove(frePos.get(freTable.get(x) - 1).size() - 1);
        // }
        max = Math.max(max, freTable.get(x));
        if (bucket.get(x) == null) {
            bucket.put(x, new ArrayList<>());
        }
        bucket.get(x).add(n++);
    }
    
    public int pop() {
        Set<Integer> v = freObj.get(max);
        int last = -1;
        int rm = -1;
        rm = frePos.get(max).get(frePos.get(max).size() - 1);
        // for (int i : v) {
        //     if (bucket.get(i).get(bucket.get(i).size() - 1) > last) {
        //         rm = i;
        //         last = bucket.get(i).get(bucket.get(i).size() - 1);
        //     }
        // }
        bucket.get(rm).remove(bucket.get(rm).size() - 1);
        freTable.put(rm, freTable.get(rm) - 1);
        freObj.get(freTable.get(rm) + 1).remove(rm);
        frePos.get(max).remove(frePos.get(max).size() - 1);
        // if (max - 1 != 0) {
        //     frePos.get(max - 1).add(rm);
        // }
        if (freTable.get(rm) != 0) {
            freObj.get(freTable.get(rm)).add(rm);
        }
        if (freObj.get(max).size() == 0) {
            max--;
        }
        return rm;
    }
}
