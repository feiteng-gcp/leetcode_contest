class FreqStack {
    HashMap<Integer, ArrayList<Integer>> hm = new HashMap<Integer, ArrayList<Integer>>();
    ArrayList<TreeMap<Integer, Integer>> al = new ArrayList<TreeMap<Integer, Integer>>();
    int max = 0;
    int calls = 0;

    public FreqStack() {
        al.add(new TreeMap<Integer, Integer>());
    }
    
    public void push(int x) {
        int c = 0;
        if (hm.containsKey(x)) {
            c = hm.get(x).size();
        } else {
            hm.put(x, new ArrayList<Integer>());
        }
        if (c+1 > max) {
            max = c+1;
        }
        
if (c+1 > al.size()-1) {
            al.add(new TreeMap<Integer, Integer>());
        }
        hm.get(x).add(calls);
        al.get(c+1).put(calls, x);
        calls++;
    }
    
    public int pop() {
        if (max==0) {
            System.out.println(al.get(2).size());
        }
        int mx = al.get(max).pollLastEntry().getValue();
        hm.get(mx).remove(max-1);
        if (al.get(max).isEmpty())
            max--;
        return mx;
    }
}
