class FreqStack {
    
private final PriorityQueue<Element> pq;
    private final Map<Integer, Element> map;
    private int count = 0;

    public FreqStack() {
        pq = new PriorityQueue<>((x, y) -> {
            if (x.freq == y.freq) return Integer.compare(y.getLastPos(), x.getLastPos());
            return Integer.compare(y.freq, x.freq);
        });
        map = new HashMap<>();
    
}

    public void push(int x) {
        count++;

        if (!map.containsKey(x)) {
            Element element = new Element(x, 1);
            element.pos.add(count);
            map.put(x, element);
            pq.offer(element);
            return;
        }

        Element element = map.get(x);
        pq.remove(element);
        element.freq++;
        element.pos.add(count);
        pq.offer(element);
    }

    public int pop() {
        Element element = pq.poll();
        int ans = element.val;
        element.freq--;
        element.pos.remove(element.pos.size() - 1);
        if (element.freq == 0) {
            map.remove(ans);
        } else pq.offer(element); // put back.
        return ans;
    }

    private static class Element {
        int val;
        int freq;
        List<Integer> pos;

        public Element(int val, int freq) {
            this.val = val;
            this.freq = freq;
            this.pos = new ArrayList<>();
        }

        private int getLastPos() {
            return pos.get(pos.size() - 1);
        }
    }
}

 * FreqStack obj = new FreqStack();
 * obj.push(x);
