/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> allPossibleFBT(int N) {
        List<TreeNode> r = new LinkedList<>();
        if (N % 2 == 0) {
            return r;
        }
        return helper(N);
    }
    
    public List<TreeNode> helper(int N) {
        List<TreeNode> res = new LinkedList<>();
        if (N == 1) {
            res.add(new TreeNode(0));
            return res;
        }
        
        for (int i = 2; i < N; i++) {
            
            int left = i - 1;
            int right = N - i;
            if (left % 2 == 0 || right % 2 == 0) {
                continue;
            }
            List<TreeNode> lf = helper(left);
            List<TreeNode> rt = helper(right);
            for (TreeNode l : lf) {
                for (TreeNode r : rt) {
                    TreeNode r1 = new TreeNode(0);
                    r1.left = l;
                    r1.right =r;
                    res.add(r1);
                }
            }
        }
        return res;
    }
}
