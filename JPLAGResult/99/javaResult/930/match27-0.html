/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> allPossibleFBT(int res) {
        
        
       
        List<TreeNode> result = new ArrayList<>();
        if (res == 1) {
            result.add(new TreeNode(0));
            return result;
        }
        int left = 1;
        while (left < res - 1) {
            List<TreeNode> les = allPossibleFBT(left);
            List<TreeNode> rights = allPossibleFBT(res - 1 - left);
            for (TreeNode node1 : les) {
                for (TreeNode node2 : rights) {
                    TreeNode returnNode = new TreeNode(0);
                    returnNode.left = node1;
                    returnNode.right = node2;
                    result.add(returnNode);
                }
            }
            left += 2;
        }
        return result;
    }
