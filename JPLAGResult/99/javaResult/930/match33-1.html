/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    HashMap<Integer,List<TreeNode>> map;
    public List<TreeNode> allPossibleFBT(int N) {
        if(N%2 == 0)
            return new LinkedList<TreeNode>();
        map = new HashMap<Integer,List<TreeNode>>();
        for(int i = 1;i<=N;i+=2){
            aux(i);
        }
        return map.get(N);
    }
    
    
public void aux(int N){
        List<TreeNode> list = new LinkedList<TreeNode>();
        if(N == 1){
            
            list.add(new TreeNode(0));
            map.put(1,list);
            return;
        }
        for(int i = 1;i<N;i+=2)
{
            List<TreeNode> llist = map.get(i);
            List<TreeNode> rlist =  map.get(N-1-i);
            for(TreeNode lnode:llist){
                for(TreeNode rnode:rlist){
                    TreeNode can = new TreeNode(0);
                    can.left = lnode;
                    can.right = rnode;
                    list.add(can);
                }
            }
        }
        map.put(N,list);
    }
}
