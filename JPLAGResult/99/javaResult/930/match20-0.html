/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    public List<TreeNode> allPossibleFBT(int n) {
        List<TreeNode> res = new ArrayList<>();
        if(n%2 == 0) return res;
        Map<Integer, List<TreeNode>> topologies = new HashMap<>();
 
		topologies.put(1, new ArrayList<TreeNode>());
		topologies.get(1).add(new TreeNode(0));
 
		for(int i=3;i<=n;i+=2){
 
			List<TreeNode> list = new ArrayList<>();
 
			for(int j=1;j<i;j+=2){
				List<TreeNode> left = topologies.get(j);
				List<TreeNode> right = topologies.get(i-j-1);
				list.addAll(generateAllCombinations(left,right));
			}
			topologies.put(i, list);
		}
 
		List<TreeNode> result = topologies.get(n);
        
        
        return result;
    }
    private List<TreeNode> generateAllCombinations(List<TreeNode> left, List<TreeNode> right) {
 
		List<TreeNode> list = new ArrayList<TreeNode>();
		for(int i=0;i<left.size();i++){
			for(int j=0;j<right.size();j++){
				TreeNode nNode = new TreeNode(0);
				nNode.left = cloneTree(left.get(i));
				nNode.right = cloneTree(right.get(j));
				list.add(nNode);
			}
		}
		return list;
	}
    TreeNode cloneTree(TreeNode t) {
            TreeNode nNode = new TreeNode(t.val);
			if(t.left != null) nNode.left = cloneTree(t.left);
			if(t.right != null) nNode.right = cloneTree(t.right);
 
			return nNode;
    }
}
