/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private TreeNode clone(TreeNode root) {
        if (root == null) return null;
        TreeNode n = new TreeNode(root.val);
        n.left = clone(root.left);
        n.right = 
clone(root.right);
        return n;
    }
    private List<TreeNode> combo(List<TreeNode> left, List<TreeNode> right) {
        List<TreeNode> l = new ArrayList<TreeNode>();
        for(int i = 0; i < left.size(); i++){
            for(int j = 0; j < right.size(); j++){
                TreeNode root = new TreeNode(0);
                root.left = clone(left.get(i));
                root.right = clone(right.get(j));
                l.add(root);
            }
        }
        return l;
    }
    public List<TreeNode> allPossibleFBT(int N) {
        List<TreeNode> l = new ArrayList<TreeNode>();
        if(N % 2 == 0) {
            return l;
        }
        if(N == 1) {
            l.add(new TreeNode(0));
            return l;
        }
        else {
            
for(int i = 1; i < N; i += 2){
                List<TreeNode> left = allPossibleFBT(i);
                List<TreeNode> right = allPossibleFBT(N - i - 1);
                l.addAll(combo(left, right));
            }
            return l;
        }
    }
}
