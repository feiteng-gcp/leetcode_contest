/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    Map<Integer, List<TreeNode>> map;
    public List<TreeNode> allPossibleFBT(int N) {
        if (N % 2 == 0) return new ArrayList<>();
        map = new HashMap<>();
        return dfs(N);        
    }
    
    private List<TreeNode> dfs(int rem){
        if (map.containsKey(rem)) return map.get(rem);
        
List<TreeNode> lst = new ArrayList<>();
        if (rem == 1){
            lst.add(new TreeNode(0));
            map.put(1, lst);
            return lst;
        }
        for (int i = 1; i < rem; ++i){
            if (i % 2 == 0) continue;
            
List<TreeNode> left = dfs(i);
            List<TreeNode> right = dfs(rem-1-i);
            for (TreeNode leftNode : left){
                for (TreeNode rightNode: right){
                    TreeNode root = new TreeNode(0);
                    root.left = leftNode;
                    root.right = rightNode;
                    lst.add(root);
                }
            }            
        }
        map.put(rem, lst);
        return lst;
    }
}
