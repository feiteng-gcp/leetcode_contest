/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> allPossibleFBT(int N) {

	    if(N%2 == 0) return new ArrayList<TreeNode>();;


	    Map<Integer, List<TreeNode>> res = new HashMap<Integer, List<TreeNode>>();

	    res.put(1, new ArrayList<TreeNode>());
	    res.get(1).add(new TreeNode(0));

	    for(int i=3;i<=N;i+=2){

	        List<TreeNode> list = new ArrayList<TreeNode>();

	        for(int j=1;j<i;j+=2){
	            List<TreeNode> left = res.get(j);
	            List<TreeNode> right = res.get(i-j-1);
	            list.addAll(generateAllCombinations(left,right));
	        }
	        res.put(i, list);
	    }

	    List<TreeNode> result = res.get(N);
	    return result;
	}

	private List<TreeNode> generateAllCombinations(List<TreeNode> left, List<TreeNode> right) {

	    for(int i=0;i<left.size();i++){
	        for(int j=0;j<right.size();j++){

	            list.add(nNode);
	        }
	    }
	    return list;
	}
}
