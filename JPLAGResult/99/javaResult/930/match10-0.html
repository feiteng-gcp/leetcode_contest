/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> allPossibleFBT(int N) {
        List<TreeNode> ret = new ArrayList<TreeNode>();
		if (N == 1) {
			TreeNode n = new TreeNode(0);
			ret.add(n);
			return ret;
		}
		
		for (int k = 1; k < N-1; k+=2) {
			List<TreeNode> retL = allPossibleFBT(k);
			List<TreeNode> retR = allPossibleFBT(N-(k+1));
			for (TreeNode l:retL) {
				for (TreeNode r:retR) {
                    TreeNode n = new TreeNode(0);

					n.left = l;
					n.right = r;
					ret.add(n);
				}
			}
		}
		
		return ret;
    }
}
