/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    HashMap<Integer, ArrayList<TreeNode>> map;
    
    public List<TreeNode> allPossibleFBT(int N) {
        List<TreeNode> res = new ArrayList<TreeNode>();
        if(N % 2 == 0) return res;
        map = new HashMap<>();
        return build(N);    
    }
    
    public ArrayList<TreeNode> build(int k) {
        ArrayList<TreeNode> res = new ArrayList<>();
        if(map.containsKey(k)) return map.get(k);
        
        if(k == 1) {
            TreeNode root = new TreeNode(0);
            res.add(root);
            map.put(1, res);
            return res;
        }
        
        k -= 1;
        for(int i = 1; i < k; i += 2) {
            ArrayList<TreeNode> leftSet = build(i);
            ArrayList<TreeNode> rightSet = build(k - i);
            for(TreeNode left : leftSet) {
                for(TreeNode right : rightSet) {
                    TreeNode root = new TreeNode(0);
                    root.left = left;
                    root.right = right;
                    res.add(root);
                }
            }
        }
        map.put(k, res);
        return res;
        
    }
    
    
    
    
    
    
}
