class Solution {
    public int numSpecialEquivGroups(String[] A) {
    	int res = 0;
    	boolean[] group = new boolean[A.length];
        for (int i = 0; i < A.length; ++i) {
        	if (!group[i]) {
        		res += 1;
        		for (int j = i + 1; j < A.length; ++j) {
        			if (!group[j] && specialEqual(A[i], A[j])) {
        				group[j] = true;
        			}
        		}
        	}
        }
        return res;
    }
    
    private boolean specialEqual(String A, String B) {
    	if (A.length() != B.length()) {
    		return false;
    	}
    	int
[] odd = new int[26], even = new int[26];
    	for (int i = 0; i < A.length(); ++i) {
    		if (i % 2 == 0) {
    			even[A.charAt(i) - 'a'] = even[A.charAt(i) - 'a'] + 1;
    			even[B.charAt(i) - 'a'] = even[B.charAt(i) - 'a'] - 1;
    		}else {
    			odd[A.charAt(i) - 'a'] = odd[A.charAt(i) - 'a'] + 1;
    			odd[B.charAt(i) - 'a'
] = odd[B.charAt(i) - 'a'] - 1;
    		}
    	}
    	for (int i = 0; i < 26; ++i) {
    		if (even[i] != 0 || odd[i] != 0) {
    			return false;
    		}
    	}
    	return true;
    }
}
