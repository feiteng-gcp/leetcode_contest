class Solution {
   	static String orig(String s) {
		char[] cs = s.toCharArray();
		char [] cs1 = new char[(s.length() + 1)/2];
		char [] cs2 = new char[s.length()/2];
		for (int k = 0,j = 0; k < s.length(); k+=2, j++) {
			cs1[j] = cs[k];
		}
		for (int k = 1,j = 0; k < s.length(); k+=2, j++) {
			cs2[j] = cs[k];
		}
		Arrays.sort(cs1);
		Arrays.sort(cs2);
		return new String(cs1) + new String(cs2);
	}
	public int numSpecialEquivGroups(String[] A) {
        int ret = 0;
        
        HashSet<String> s = new HashSet<String>();
        
        for (String str:A) {
        	String o = orig(str);
        	if (!s.contains(o)) {
        		s.add(o);
        	}
        }
        
        return s.size();
    }
}
