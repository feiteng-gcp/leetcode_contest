class Solution {
    public int numSpecialEquivGroups(String[] A) {
        HashMap<String, Set<String>> map = new HashMap<>();
        for(int i = 0; i < A.length; i ++) {
            char[] even = new char[A[i].length() - A[i].length() / 2];
            char[] odd = new char[A[i].length() / 2];
            
            for(int j = 0; j < A[i].length(); j ++) {
                if(j % 2 == 0) even[j / 2] = A[i].charAt(j);
                else odd[j / 2] = A[i].charAt(j);
            }
            Arrays.sort(even);
            Arrays.sort(odd);
            String key = new String(even) + new String(odd);
            
            if(map.containsKey(key)) {
                map.get(key).add(A[i]);
            }
            else{
                Set<String> tmp = new HashSet<String>();
                tmp.add(A[i]);
                map.put(key, tmp);
            }
        }
        
        return map.size();
        
    }
}
