class Solution {
    private boolean isSpecialEquiv(String A, 
String B){
        Map<Character,Integer> even = new HashMap<>();
        Map<Character,Integer> odd = new HashMap<>();
        for(int i=0; i<A.length();++i){
            char c = A.charAt(i);
            
if(i%2==0){
                even.put(c,even.getOrDefault(c,0)+1);
            }
            else{
                odd.put(c,odd.getOrDefault(c,0)+1);
            }
        }
        for(int i=0; i<B.length();++i){
            char c = B.charAt(i);
            if(i%2==0){
                Integer amount = even.get(c);
                if(amount==null)return false;
                amount-=1;
                if(amount==0){
                    even.remove(c);
                }
                else{
                     even.put(c,amount);   
                }
            }
            else{
               Integer amount = odd.get(c);
                if(amount==null)return false;
                amount-=1;
                if(amount==0){
                    odd.remove(c);
                }
                else{
                     odd.put(c,amount);   
                }
            }
        }
        return (odd.isEmpty() && even.isEmpty());
    }
    
public int numSpecialEquivGroups(String[] A) {
        int count = 0;
        List<String> groups = new ArrayList<>();
        for(int i =0; i<A.length;++i){
            boolean matchedGroup=false;
            for(int j=0; j<groups.size();++j){
                if(isSpecialEquiv(A[i],groups.get(j))){
                    matchedGroup=true;
                    break;
                }   
            }
            if(!matchedGroup){
                groups.add(A[i]);
            }
            
        }
        return groups.size();
    }
}
