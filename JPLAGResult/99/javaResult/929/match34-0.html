class Solution {
    public int numSpecialEquivGroups(String[] a) {
            Map<List<String>, Integer> m = new HashMap<>();
            for (String s: a) {
                List<String> group = canonize(s);
                m.put(group, m.getOrDefault(group, 0)+1);
            }
            return m.size();
        }
        List<String> canonize(String s) {
            List<Character> s1 = new ArrayList<>();
            List<Character> s2 = new ArrayList<>();
            for (int i = 0; i < s.length(); i++) {
                if (i%2 == 0) {
                    s1.add(s.charAt(i));
                } else {
                    s2.add(s.charAt(i));
                }
            }
            Collections.sort(s1);
            Collections.sort(s2);
            return Arrays.asList(
                    s1.stream().map(c -> ""+c).collect(Collectors.joining("")),
                    s2.stream().map(c -> ""+c).collect(Collectors.joining("")));
        }
}
