class Solution {
    public int numSpecialEquivGroups(String[] A) {
        if(A == null || A.length == 0) {
            return 0;
        }
        
int count = 0;
        Set<String> set = new HashSet<>();
        
        for(String str : A) {
            String pair = getOddEven(str);
            if(!set.contains(pair)){
                count++;
                set.add(pair);
            }
        }
        
        return count;
    }
    
    private String getOddEven(String str){
        char[] cc = str.toCharArray();
        int n = cc.length;
        char[] odd =new  char[(n+1)/2];
        char[] even =new  char[n/2];
        for(int i= 0; i < n; i++){
            if(i%2 ==0) {
                odd[i/2] = cc[i];
            }
            else{
                even[i/2] = cc[i];
            }
        
}
        
        Arrays.sort(odd);
        Arrays.sort(even);
        return new String(odd) + "+" + new String(even);
    }
}
