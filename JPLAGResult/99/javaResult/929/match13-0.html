import java.util.HashMap;
import java.util.Map;
import java.util.Arrays;
class Solution {
    public int numSpecialEquivGroups(String[] A) {
      
          HashMap<String, Integer> map = new HashMap<>();
         
      
        
        int n = A[0].length();
        
        for (String s : A){
            int[] v = new int[52];
            for (int i = 0; i < n; i += 2 )
                v[s.charAt(i)-'a'] ++;
             for (int i = 1; i < n; i += 2 )
                v[s.charAt(i)-'a'+26] ++;
            map.put(Arrays.toString(v),0);
            
        }
            
        return map.size();
        
        
    }
}
