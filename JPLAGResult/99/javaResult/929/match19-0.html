class Solution {
    public int numSpecialEquivGroups(String[] A) {
        Set<String> set = new HashSet<>();
        int count = 0;
        for (String s : A){
            StringBuilder sb1 = new StringBuilder();
            StringBuilder sb2 = new StringBuilder();
            for (int i = 0; i < s.length(); i++){
                if (i % 2 == 0) sb1.append(s.charAt(i));
                else sb2.append(s.charAt(i));
            }
            char[] letter1 = sb1.toString().toCharArray();
            char[] letter2 = sb2.toString().toCharArray();
            Arrays.sort(letter1);
            Arrays.sort(letter2);
            String s1 = new String(letter1);
            String s2 = new String(letter2);
            if (set.contains(s1 + " " + s2)){
                continue;
            } else {
                set.add(s1 + " " + s2);
                //System.out.println(s1 + " " + s2);
                count++;
            }
        }
        return count;
    }
}
