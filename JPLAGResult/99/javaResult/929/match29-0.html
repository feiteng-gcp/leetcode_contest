class Solution {
    public int numSpecialEquivGroups(String[] A) {
        
        int count = 0;
        Set<String> set = new HashSet<>();
        for (String str : A) {
            if (set.add(getSign(str))) count++;
        }
        return count;
    }
    
    private String getSign(String input) {

        List<Character> single = new ArrayList<>();
        List<Character> doub = new ArrayList<>();
        for (int i = 0; i < input.length(); i++) {
            if (i % 2 == 0) {
                single.add(input.charAt(i));
            } else {
                doub.add(input.charAt(i));
            }
        }
        Collections.sort(single);
        Collections.sort(doub);
        return single.toString() + "_" + doub.toString();
    }
    
}
