class Solution {
    public int numSpecialEquivGroups(String[] A) {
        int len = A.length;
        if(len == 0){
            return 0;
        }
        Map<Integer,Map<Integer,List<Character>>> map = new HashMap<>();
        for(int i = 0 ; i < len ; ++i){
            String s= A[i];
            List<Character> tmp1 = new ArrayList<>();
            List<Character> tmp2 = new ArrayList<>();
            for(int j = 0 ; j < s.toCharArray().length ; ++j){
                if(j % 2 == 0){
                    tmp1.add(s.toCharArray()[j]);
                }else{
                    tmp2.add(s.toCharArray()[j]);
                }
            }
            Collections.sort(tmp1);
            Collections.sort(tmp2);
            Map<Integer,List<Character>> tmp = new HashMap<>();
            tmp.put(0,tmp1);
            tmp.put(1,tmp2);
            map.put(i,tmp);
        }
        Map<Integer,Boolean> hasGroup = new HashMap<>();
        hasGroup.put(0,false);
        for(int i = 1 ; i < len ; ++i){
            boolean tag = false;
            for(int j = 0 ; j < i ; ++j){
                Map<Integer,List<Character>> tmp1 = map.get(j);
                Map<Integer,List<Character>> tmp2 = map.get(i);
                if(tmp1.get(0).size() != tmp2.get(0).size()
                        || tmp1.get(1).size() != tmp2.get(1).size()){
                    continue;
                }
                int cnt = 0 ;
                for(int k = 0 ; k <= 1 ; ++k){
                    for(int l = 0 ; l < tmp1.get(k).size() ; ++l){
                        if(tmp1.get(k).get(l) != tmp2.get(k).get(l)){
                            ++cnt;
                            break;
                        }
                    }
                }
                if(cnt == 0){
                    tag = true;
                    break;
                }
            }
            if(tag){
                hasGroup.put(i,true);
            }else{
                hasGroup.put(i,false);
            }
        }
        int res = 0;
        for(int i = 0; i < len ; ++i){
            if(!hasGroup.get(i)){
                ++res;
            }
        }
        return res;
    }
}
