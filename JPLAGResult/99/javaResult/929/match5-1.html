class Solution {
    public int numSpecialEquivGroups(String[] A) {
        List<List<String>> groups = new ArrayList<>();
        for (String s : A) {
            char[] even = new char[s.length() / 2 + 1];
            char[] odd = new char[s.length() / 2];
            for (int i = 0; i < s.length(); i++) {
                if (i % 2 == 0) {
                    even[i / 2] = s.charAt(i);
                } else 
{
                    odd[i / 2] = s.charAt(i);
                }
            }
            Arrays.sort(even);
            Arrays.sort(odd);
            String e = new String(even);
            String o = new String(odd);
            List<String> l = new ArrayList<>(2);
            l.add(e);
            l.add(o);

            if (!groups.contains(l)) {
                groups.add(l);
            }            
        }
        return groups.size();
    }
}
