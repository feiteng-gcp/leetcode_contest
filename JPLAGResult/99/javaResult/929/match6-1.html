class Solution {
    public int numSpecialEquivGroups(String[] A) {
        Set<String> result = new HashSet<>();
        for (String str: A) {
            result.add(move(str));
        }
        return result.size();
    }
    
    private String move(String str) {
        char[] chars = new char[str.length()];
        int evenNum = str.length() % 2 == 0 ? str.length() / 2: (
str.length() / 2 + 1);
        for (int i = 0; i < str.length(); i++) {
            if (i % 2 == 0) {
                chars[i / 2] = str.charAt(i);
            } else {
                chars[str.length() - i / 2 - 1] = str.charAt(i);
            }
        }
        Arrays.sort(chars, 0, evenNum);
        Arrays.sort(chars, evenNum, str.length());
        return new String(chars);
    }
}
