class Solution {
    public int numSpecialEquivGroups(String[] A) {
        Map<Struct, Integer> freq = new HashMap<>();
        for (String s : A) {
            Struct str = new Struct(s);
            
freq.put(str, freq.getOrDefault(str, 0) + 1);
        }
        return freq.size();
    }
}

class Struct {
    int[] freqEven;
    int[] freqOdd;
    int e;
    int o;
    public Struct(String s) {
        freqEven = new int[26];
        freqOdd = new int[26];
        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 0) {
                e += s.charAt(i);
                freqEven[s.charAt(i) - 'a']++;
            } else {
                o += s.charAt(i);
                freqOdd[s.charAt(i) - 'a']++;
            }
        }
    }
    public int hashCode() {
        return e + o * 2;
    }
    public boolean equals(Object o) {
        if (!(o instanceof Struct)) return false;
        Struct s = (Struct) o;
        return Arrays.equals(freqEven, s.freqEven) && Arrays.equals(freqOdd, s.freqOdd);
    }
}
