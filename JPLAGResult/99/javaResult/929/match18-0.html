class Solution {
    public int numSpecialEquivGroups(String[] A) {
        Set<String> r = new HashSet<String>();
        for (String a: A) {
            r.add(change(a));
        }
        return r.size();
    }
    
    public String change(String a) {
        char[] ch = a.toCharArray();
        int count = 0;
        List<Character> e = new LinkedList<>();
        List<Character> o = new LinkedList<>();
        for (char c: ch) {
            if (count % 2 == 0) {
                e.add(c);
            } else {
                o.add(c);
            }
            count++;
        }
        StringBuilder st = new StringBuilder();
        Collections.sort(e);
        Collections.sort(o);
        for (int i = 0; i < o.size(); i++) {
            st.append(e.get(i));
            st.append(o.get(i));
        }
        if (e.size() > o.size()) {
            st.append(e.get(e.size() -1));
        }
        //System.out.println(st.toString());
        return st.toString();
    }
}
