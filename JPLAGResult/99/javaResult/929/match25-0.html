class Solution {
    //calculate the odd and even character count for each string in A. And store as string in a hashset.
    public int numSpecialEquivGroups(String[] A) {
        HashSet<String> odd = new HashSet<String>();
        //HashSet<String> even = new HashSet<String>();
        for(String word: A) {
            int[] oddc = new int[26];
            int[] evenc = new int[26];
            for(int i=0;i<word.length();i++) {
                if(i%2 == 0) evenc[word.charAt(i)-'a']++;
                else oddc[word.charAt(i)-'a']++;
            }
            StringBuilder sb = new StringBuilder("");
            for(int i=0;i<26;i++) {
                sb.append((char)('a'+i));
                sb.append(oddc[i]);
            }
            for(int i=0;i<26;i++) {
                sb.append((char)('a'+i));
                sb.append(evenc[i]);
            }
            odd.add(sb.toString());
        }
        return odd.size();
    }
}
