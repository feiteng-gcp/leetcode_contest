class Solution {
    public int numSpecialEquivGroups(String[] A) {
        Set<String> groups = new HashSet<>();
        
for (String a : A) {
            char[] even = new char[(a.length() + 1) >> 1];
            char[] odd = new char[a.length() >> 1];
            for (int i = 0, j = 0; i < a.length(); i += 2, j++) {
                even[j] = a.charAt(i);
                if (i + 1 < a.length()) 
{
                    odd[j] = a.charAt(i + 1);
                }
            }
            Arrays.sort(even);
            Arrays.sort(odd);
            groups.add(String.valueOf(even) + "|" + String.valueOf(odd));
        }
        return groups.size();
    }
}
