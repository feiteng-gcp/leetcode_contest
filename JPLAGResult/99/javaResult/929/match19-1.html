class Solution
{
    public int numSpecialEquivGroups(String[] A)
    {
        HashSet<String> group = new HashSet<>();
        int ans = 0;
        for (String a : A)
        {
            List<Character> odd = new ArrayList<>(), even = new ArrayList<>();
            for (int i = 0; i < a.length(); i += 2) even.add(a.charAt(i));
            for (int i = 1; i < a.length(); i += 2) odd.add(a.charAt(i));
            Collections.sort(odd);
            Collections.sort(even);
            StringBuilder sb = new StringBuilder();
            for (char c : odd) sb.append(c);
            sb.append('_');
            for (char c : even) sb.append(c);
            if (!group.contains(sb.toString()))
            {
                ans++;
                group.add(
sb.toString());
            }
        }
        return ans;
    }
}
