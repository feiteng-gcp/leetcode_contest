class Solution {
    public int numSpecialEquivGroups(String[] A) {
        Set<String> set = new HashSet<>();
        for (String s : A) {
            set.add(s);
            for (String target : set) {
                if (isEqual(target, s) && !s.equals(target)) {
                    set.remove(s);
                    break;
                }
            }
        }
        return set.size();
    }
    
    private boolean isEqual(String a, String b) {
        int[] odd = new int[26];
        int[] even = new int[26];
        
        for (int i = 0; i < a.length(); i ++) {
            if (i % 2 == 0) {
                even[a.charAt(i) - 'a'] ++;
                even[b.charAt(i) - 'a'] --;
            } else {
                odd[a.charAt(i) - 'a'] ++;
                odd[b.charAt(i) - 'a'] --;
            }
        }
        for (int i = 0; i < 26; i ++) {
            if (odd[i] != 0 || even[i] != 0) {
                return false;
            }
        }
        return true;
    }
}
