class Solution {
    public int surfaceArea(int[][] grid) {
        int result = 0;
        for(int i=0;i<grid.length;i++) {
            for(int j=0;j<grid[0].length;j++) {
                if(grid[i][j] > 0) result += 2;
                if(j>0 && grid[i][j] > grid[i][j-1]) result += grid[i][j] - grid[i][j-1];
                else if(j == 0) result += grid[i][j];
                if(j<grid[0].length-1 && grid[i][j] > grid[i][j+1]) result += grid[i][j] - grid[i][j+1];
                else if(j == grid[0].length-1) result += grid[i][j];
                if(i>0 && grid[i][j] > grid[i-1][j]) result += grid[i][j] - grid[i-1][j];
                else if(i == 0) result += grid[i][j];
                if(i<grid.length-1 && grid[i][j] > grid[i+1][j]) result += grid[i][j] - grid[i+1][j];
                else if(i == grid.length-1) result += grid[i][j];
            }
        }
        return result;
    }
}
