class Solution {
    public int surfaceArea(int[][] grid) {
        int sum = 0;
        int rows = grid.length;
        int cols = grid[0].length;
        sum += 2*(rows*cols);
        for (int i = 0; i < rows; ++i) {
            for (int j = 0; j < cols; ++j) {
                if (grid[i][j] == 0) {
                    sum -= 2;
                    continue;
                }
                sum += 4*grid[i][j];
                if (i < rows-1) {
                    int min = Math.min(grid[i][j],grid[i+1][j]);
                    sum -= 2*min;
                }
                if (j < cols-1) {
                    int min = Math.min(grid[i][j],grid[i][j+1]);
                    sum -= 2*min;
                
}
            }
        }
        return sum;
    }
}
