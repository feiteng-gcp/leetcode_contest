class Solution {
    public int surfaceArea(int[][] grid) {
        if(grid.length==0 || grid[0].length==0)return 0;
        int n=grid.length;
        
int m=grid[0].length;
        int topBottom=0;
        int around=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]!=0){
                    topBottom += 2;
                    around += 4*grid[i][j];
                }
            }
        }
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){ 
                if(i+1<n){
                    around-=Math.min(grid[i+1][j], grid[i][j]);
                }
                if(i-1>=0){
                    around-=Math.min(grid[i-1][j], grid[i][j]);
                
}
                if(j+1<m){
                    around-=Math.min(grid[i][j+1], grid[i][j]);
                }
                if(j-1>=0){
                    around-=Math.min(grid[i][j-1], grid[i][j]);
                }
            }
        }
        return topBottom+around;
    }
}
