class Solution {
    public int surfaceArea(int[][] grid) {
        int m=grid.length, n=grid[0].length;
        int[] row= new int[m], col= new int[n];
        int res=0;
        for (int i=0; i<m; i++){
            for (int j=0; j<n; j++){
                if (grid[i][j]>0) res++;
                row[i]+=Math.abs(grid[i][j]-(i>0?grid[i-1][j]:0));
                col[j]+=Math.abs(grid[i][j]-(j>0?grid[i][j-1]:0));
                if (i==m-1) row[i]+=grid[i][j];
                if (j==n-1) col[j]+=grid[i][j];
            }
        }
        res*=2;
        for (int r: row) res+=r;
        for (int c: col) res+=c;
        return res;
    }
}
