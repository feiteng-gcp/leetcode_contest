class Solution {
    public int surfaceArea(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int res = 0;
        
        int[][] dirs = {{1,0},{-1, 0}, {0,1}, {0,-1}};
        
        for (int i = 0; i < m; i ++) {
            for (int j = 0; j < n; j ++) {
                if (grid[i][j] != 0) {
                    res += 2;
                    for (int[] dir : dirs) {
                        int x = i + dir[0];
                        int y = j + dir[1];
                        if (x < 0 || y < 0 || x >= m || y >= n) {
                            res += grid[i][j];
                        }
                        else if (grid[x][y] < grid[i][j]) {
                            res += (grid[i][j] - grid[x][y]);
                        }
                    }
                }
            }
        }
        
        return res;
    }
}
