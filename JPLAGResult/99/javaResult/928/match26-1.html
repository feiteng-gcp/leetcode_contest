class Solution {
    public int surfaceArea(int[][] grid) {
        int ret = 0;
        if(grid.length == 0) return 0;
        for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                int val = grid[i][j];
                if(val == 0) continue;
                else{
                    ret += 2;
                    int left = (j == 0) ? val : Math.max(val - grid[i][j - 1], 0);
                    int right = (j == grid[0].length - 1) ? val : Math.max(val - grid[i][j + 1], 0);
                    int up = (i == 0) ? val : Math.max(val - grid[i - 1][j], 0); 
                    int down = (i == grid.length - 1) ? val : Math.max(val - grid[i + 1][j], 0);
                    ret = ret + left + right + up + down;
                
}
            }
        }
        return ret;
    }
}
