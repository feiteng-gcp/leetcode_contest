class Solution {
    
    public int surfaceArea(int[][] grid) {
        int rows = grid.length;
        if(rows == 0){
            return 0;
        }
        
int cols = grid[0].length;
        int cnt = 0;
        int sum = 0;
        for(int i = 0 ; i < rows ; ++i){
            for(int j = 0 ; j < cols ; ++j){
                if(grid[i][j] == 0){
                    continue;
                }
                ++cnt;
                int tmp = grid[i][j]*4;
                
if(j - 1 >= 0 && grid[i][j-1] != 0){
                    tmp -= getDup(grid[i][j-1],grid[i][j]);
                }
                if(i - 1 >= 0 && grid[i-1][j] != 0){
                    tmp -= getDup(grid[i-1][j],grid[i][j]);
                }
                sum += tmp;
            }
        }
        sum += (cnt + cnt);
        return sum;
    }

    private int getDup(int hight1,int hight2){
        return hight1 < hight2 ? hight1 + hight1 : hight2 + hight2;
    }
}
