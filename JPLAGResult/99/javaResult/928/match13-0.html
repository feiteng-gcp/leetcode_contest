class Solution {
    public int surfaceArea(int[][] grid) {
                int M = grid.length;
        int N = grid[0].length;
        int a = 0;
        for (int m = 0; m < M; m++) {
        	for (int n = 0; n < N; n++) {
        		if (grid[m][n] == 0)
        			continue;
        		a += grid[m][n] * 4 + 2;
        		
        		
        		if (m-1 >= 0) {
        			a -= Math.min(grid[m-1][n], grid[m][n])*2;
        		}
        		if (n - 1 >=0) {
        			a -= Math.min(grid[m][n-1], grid[m][n])*2;
        		}
        	}
        }
        return a;
    }
}
