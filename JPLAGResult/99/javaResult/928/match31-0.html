class Solution {
    public int surfaceArea(int[][] g) {
        int l = g.length;
        int ll = g[0].length;
        
        int s = 0 ;
        for(int i=0;i<l;i++){
            
           for(int j=0;j<ll;j++){
             if(g[i][j]>0){
                 s+= 2; 
             }
               if(j==0){
                   s += g[i][j];
               }
               if(j==ll-1){
                   s += g[i][j];
               }
               if(j>=1){
                   s += Math.abs(g[i][j]-g[i][j-1]);
               }
               
           } 
            
        }
        
        for(int j=0;j<ll;j++){
            
           for(int i=0;i<l;i++){
             if(i==0){
                   s += g[i][j];
               }
               if(i==l-1){
                   s += g[i][j];
               }
               if(i>=1){
                   s += Math.abs(g[i][j]-g[i-1][j]);
               }
               
               
               
           } 
            
        }
        
        return s;
        
    }
}
