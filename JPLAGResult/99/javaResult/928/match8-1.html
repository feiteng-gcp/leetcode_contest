class Solution {
    public int surfaceArea(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int res = 0;
        int [][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        for (int i = 0 ;i < m; ++i) {
            for(int j = 0 ;j < n ; ++j) {
                
for (int [] dir: dirs) {
                    int nx = i + dir[0];
                    int ny = j + dir[1];
                    if (nx <0 || ny < 0 || nx >= m || ny >=n) {
                        res += grid[i][j];
                    } else {
                        res += Math.max(0, grid[i][j] - grid[nx][ny]);
                    }
                    
                }
                if (grid[i][j] > 0) {
                     res +=2;
                
}
            }
        }
        return res;
    }
}
