class Solution {
   public int surfaceArea(int[][] grid) {
        int n = grid.length;

        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == 0) {
                    res += grid[i][j];
                } else {
                    res += Math.abs(grid[i - 1][j] - grid[i][j]);
                }

                if (i == n - 1) {
                    res += grid[i][j];
                }

                if (j == 0) {
                    res += grid[i][j];
                } else {
                    res += Math.abs(grid[i][j - 1] - grid[i][j]);
                }

                if (j == n - 1) {
                    res += grid[i][j];
                }

                if (grid[i][j] > 0) {
                    res += 2;
                }
            }
        }
        return res;
    }

}
