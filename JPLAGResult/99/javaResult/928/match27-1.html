class Solution {
    public int surfaceArea(int[][] grid) {
        int counter = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[i].length; j++) {
                if (grid[i][j] != 0)
                    counter += (2 + 4 * grid[i][j]);
                if (i - 1 >= 0)
                    counter -= Math.min(grid[i-1][j], grid[i][j]);
                if (i + 1 < grid.length)
                    counter -= Math.min(grid[i+1][j], grid[i][j]);
                if (j - 1 >= 0)
                    counter -= Math.min(grid[i][j-1], grid[i][j]);
                if (j + 1 < grid[i].length)
                    counter -= Math.min(grid[i][j+1], grid[i][j])
;
            }
        }
        return counter;
    }
}
