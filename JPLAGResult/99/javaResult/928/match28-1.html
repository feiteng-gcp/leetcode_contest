class Solution {
    public int surfaceArea(int[][] grid) {
        if (grid.length == 0) return 0;
        
        
int result = 0;
        for (int i=0; i<grid.length; i++) {
            for (int j=0; j<grid[0].length; j++) {
                if (grid[i][j] > 0) {
                    result += 2;
                    
                    if (i==0) {
                        result += grid[i][j];
                    } else {
                        result += Math.max(0, grid[i][j] - grid[i-1][j]);
                    
}
                    
                    if (i==grid.length-1) {
                        result+= grid[i][j];
                    } else {
                        result += Math.max(0, grid[i][j] - grid[i+1][j]);
                    }
                    
                    if (j==0) {
                        result += grid[i][j];
                    } else {
                        result += Math.max(0, grid[i][j] - grid[i][j-1]);
                    }
                    
                    if (j==grid[0].length-1) {
                        result+= grid[i][j];
                    } else {
                        result += Math.max(0, grid[i][j] - grid[i][j+1]);
                    }
                
}
            }
        }
        
        return result;
    }
}
