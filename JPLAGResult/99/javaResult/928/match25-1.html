class Solution {
    public int surfaceArea(int[][] grid) {
        int verticalCount = 0, horizonCount = 0, count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                count += grid[i][j];
                verticalCount += Math.max(grid[i][j] - 1, 0);
                if (i > 0) 
                    horizonCount += Math.min(grid[i][j], grid[i - 1][j]);
                if (j > 0)
                    horizonCount += Math.min(grid[i][j], grid[i][j - 1])
;
            }
        }
        
        return count * 6 - horizonCount * 2 - verticalCount * 2;
    }
}
