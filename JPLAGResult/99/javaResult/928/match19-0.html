class Solution {
    public int surfaceArea(int[][] grid) {
                if (grid.length == 0 || grid[0].length == 0) return 0;
        int res = 0;
        int common_sur = 0;
        int common_sur2 = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                res += grid[i][j];
                if (grid[i][j] - 1 > 0)
                common_sur2 += (grid[i][j] - 1);
                if (i < grid.length - 1) {
                    common_sur += Math.min(grid[i][j], grid[i + 1][j]);
                }

                if (j < grid[0].length - 1) {
                    common_sur += Math.min(grid[i][j], grid[i][j + 1]);
                }
            }
        }
        
        return res * 6 - common_sur * 2 - common_sur2 * 2;
    }
}
