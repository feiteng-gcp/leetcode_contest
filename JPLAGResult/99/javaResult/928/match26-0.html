class Solution {
    public int surfaceArea(int[][] grid) {
        int sum = 0;
        if(grid == null || grid.length == 0 || grid[0].length == 0) return sum;
        for(int i = 0; i < grid.length; i ++) {
            for(int j = 0; j < grid[0].length; j ++) {
                int num = grid[i][j];
                if(num > 0) {
                    sum += num * 6 - (num - 1) * 2;
                    if(j - 1 >= 0) {
                        int num1 = Math.min(grid[i][j - 1], num);
                        sum -= num1 * 2;
                    }
                    if(i - 1 >= 0) {
                        int num2 = Math.min(grid[i - 1][j], num);
                        sum -= num2 * 2;
                    }
                }
            }
        }
        return sum;
    }
}
