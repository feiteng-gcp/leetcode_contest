class Solution {
    public int surfaceArea(int[][] grid) {
        int N = grid.length;
        int ret= 0;
        
        int[][] dir = {{-1,0},{1,0},{0,-1},{0,1}};
        int nonZero=0;
        for(int r=0;r<N;r++){
            for(int c=0;c<N;c++){
                if(grid[r][c]!=0) nonZero++;
                for(int i=0;i<4;i++){
                    int nr = r + dir[i][0];
                    int nc = c + dir[i][1];
                    if(nr<0||nc<0||nr>=N||nc>=N){
                        ret+=grid[r][c];
                        continue;
                    }
                    
                    if(grid[r][c]>grid[nr][nc]){
                        ret+=grid[r][c]-grid[nr][nc];
                    }                    
                }
            }
        }
        
        return ret+nonZero*2;
    }
}
