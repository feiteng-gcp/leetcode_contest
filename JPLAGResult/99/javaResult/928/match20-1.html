class Solution {
	private final int[][] moves = {{0, 1}, {0, -1}, {-1, 0}, {1, 0}};
    public int surfaceArea(int[][] grid) {
    	int res = 0;
        for (int i = 0; i < grid.length; ++i) {
        	for (int j = 0; j < grid[0].length; ++j) {
        		if (grid[i][j] == 0) {
        			continue;
        		}
        		res += 2;
        		for (int[] move : moves) {
        			int di = i + move[0], dj = j + move[1];
        			if (di >= 0 && di < grid.length && dj >= 0 && dj < grid[0].length) {
        				int diff = grid[i][j] - grid[di][dj];
        				if (diff > 0) {
        					res = res + diff;
        				}
        			}else {
        				res += grid[i][j];
        			}

        		}
        	}
        }
        return res;
    }
}
