class Solution {
    public int surfaceArea(int[][] grid) {
        
        int m = grid.length, n = grid[0].length;
        int sum = 0;
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                sum += 6 * grid[i][j] - (grid[i][j] > 1 ? (grid[i][j] - 1) * 2 : 0);
                if (j > 0) 
                    sum -= Math.min(grid[i][j], grid[i][j - 1]) * 2;
                if (i > 0)
                    sum -= Math.min(grid[i][j], grid[i - 1][j]) * 2
;
            }
        }
        
        return sum;
    }
}
