class Solution
{
    private static int[][] vectors = new int[][]
        {
            new int[] { 0, 1 },
            new int[] { 0, -1 },
            new int[] { 1, 0 },
            new int[] { -1, 0 }
        };
    public int surfaceArea(int[][] grid)
    {
        int area = 0;
        for (int i = 0; i < grid.length; i++)
        {
            for (int j = 0; j < grid[0].length; j++)
            {
                if (grid[i][j] == 0) continue;
                area += 2 + 4 * grid[i][j];
                for (int[] v : vectors)
                {
                    int ii = i + v[0], jj = j + v[1];
                    if (0 <= ii && ii < grid.length &&
                        0 <= jj && jj < grid[0].length)
                    {
                        area -= Math.min(grid[ii][jj], grid[i][j]);
                    }
                }
            }
        }
        return area;
    }
}
