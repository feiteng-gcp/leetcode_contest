class Solution:
    def surfaceArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n = len(grid)
        res = 0
        for i in range(n):
            for j in range(n):
                if grid[i][j] == 0:
                    continue
                res += 2
                if j > 0:
                    res += max(0, grid[i][j] - grid[i][j - 1])
                else:
                    res += grid[i][j]
                if j < n - 1:
                    res += max(0, grid[i][j] - grid[i][j + 1])
                else:
                    res += grid[i][j]
                if i > 0:
                    res += max(0, grid[i][j] - grid[i - 1][j])
                else:
                    res += grid[i][j]
                if i < n - 1:
                    res += max(0, grid[i][j] - grid[i + 1][j])
                else:
                    res += grid[i][j]
        return res
