class Solution:
    def surfaceArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        x = y = z = 0
        for i in range(len(grid)):
            sumx, sumy = grid[i][0] + grid[i][-1], grid[0][i] + grid[-1][i] 
            for j in range(len(grid)):
                if grid[i][j] != 0: z += 2
                if j == 0:
                    continue
                sumx += abs(grid[i][j] - grid[i][j - 1])
                sumy += abs(grid[j][i] - grid[j - 1][i])
            x += sumx
            y += sumy
        return x + y + z
