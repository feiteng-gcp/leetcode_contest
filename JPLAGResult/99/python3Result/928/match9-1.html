class Solution:
    def surfaceArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        
        len_row = len(grid)

        result = 6 * sum(sum(row) for row in grid)

        for i in range(len_row):
            for j in range(len_row):
                if grid[i][j] > 0:
                    result -= 2 * (grid[i][j] - 1)
                if i + 1 < len_row:
                    result -= 2 * min(grid[i][j], grid[i + 1][j])
                if j + 1 < len_row:
                    result 
-= 2 * min(grid[i][j], grid[i][j + 1])

        return result
