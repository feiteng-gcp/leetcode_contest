class Solution:
    def surfaceArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        full_grid = [[0] + row + [0] for row in grid]
        empty_row = [0 for _ in range(len(grid) + 2)]
        full_grid = [empty_row] + full_grid + [empty_row]
        area = 0
        for i in range(len(grid) + 1):
            for j in range(len(grid) + 1):
                if full_grid[i][j]:
                    area += 2    
                area 
+= abs(full_grid[i][j] - full_grid[i + 1][j])
                area += abs(full_grid[i][j] - full_grid[i][j + 1])
        return area
