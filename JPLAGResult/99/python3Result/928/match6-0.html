class Solution:
    def surfaceArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        ans = 0
        for line in grid: ans+=sum(line)*4
        for line in grid:
            for x in line:
                if x>0: ans+=2
        N = len(grid)
        for i in range(N):
            for j in range(N):
                if j<N-1: ans-=2*min(grid[i][j], grid[i][j+1])
                if i<N-1: ans-=2*min(grid[i][j], grid[i+1][j])
        return ans
