class Solution:
    def surfaceArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        ans = 2*sum(1 for r in grid for i in r if i != 0)
        m, n = len(grid), len(grid[0])
        for i in range(m):
            for j in range(n):
                if i == 0:
                    ans += grid[i][j]
                else:
                    ans += max(0, grid[i][j] - grid[i-1][j])
                if j == 0:
                    ans += grid[i][j]
                else:
                    ans += max(0, grid[i][j] - grid[i][j-1])
                if i == m - 1:
                    ans += grid[i][j]
                else:
                    ans += max(0, grid[i][j] - grid[i+1][j])
                if j == n - 1:
                    ans += grid[i][j]
                else:
                    ans += max(0, grid[i][j] - grid[i][j+1])
        return ans
