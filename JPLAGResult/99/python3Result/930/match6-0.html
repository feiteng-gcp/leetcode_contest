# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def allPossibleFBT(self, N):
        """
        :type N: int
        :rtype: List[TreeNode]
        
        """
        if N ==1:           
            return [TreeNode(0)]
        self._FBT_list = []
        self.target=N
        self.helper(N)
        return self._FBT_list
    
    def helper(self, N):
        
        if N ==1:           
            return [TreeNode(0)]
        root_list = []
        for i in range(1,N,2):
        
            root_list_left=self.helper(i)
            root_list_right=self.helper(N-i-1)
            for left in root_list_left:
                for right in root_list_right:
                    root = TreeNode(0)
                    root.left=left
                    root.right=right
                    root_list.append(root)
        if N == self.target:
            self._FBT_list += root_list
        return root_list
