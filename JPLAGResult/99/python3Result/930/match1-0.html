# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def allPossibleFBT(self, N):
        """
        :type N: int
        :rtype: List[TreeNode]
        """
        if N == 0:
            return [None]
        elif N % 2 == 0:
            return []
        elif N == 1:
            return [TreeNode(0)]
        tree_result = []
        for i in range(N-2):
            for left in self.allPossibleFBT(i+1):
                for right in self.allPossibleFBT(N-2-i):
                    root = TreeNode(0)
                    root.left = left
                    root.right = right
                    tree_result.append(root)
        return tree_result
