# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def allPossibleFBT(self, N):
        """
        :type N: int
        :rtype: List[TreeNode]
        """
        if N % 2 == 0 or N < 0:
            return []
        if N == 1:
            
return [TreeNode(0)]
        res = []
        for ii in range(1, N - 1, 2):
            for left in self.allPossibleFBT(ii):
                for right in self.allPossibleFBT(N - 1 - ii):
                    node = TreeNode(0)
                    node.left = left
                    node.right = right
                    res.append(node)
        return res
