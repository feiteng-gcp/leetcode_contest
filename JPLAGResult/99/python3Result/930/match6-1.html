# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def allPossibleFBT(self, N):
        """
        :type N: int
        :rtype: List[TreeNode]
        """
        if N % 2 == 0: 
            return []
        if N == 1:
            return [TreeNode(0)]
        ret = []
        i = 1
        while i < N:
            leftlist 
= self.allPossibleFBT(i)
            rightlist = self.allPossibleFBT(N - 1 - i)
            for left in leftlist:
                for right in rightlist:
                    root = TreeNode(0)
                    root.left = left
                    root.right = right
                    ret.append(root)
            i += 2
        return ret
