# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def allPossibleFBT(self, N):
        """
        :type N: int
        :rtype: List[TreeNode]
        """
        def make_tree(num, dic={1: [TreeNode(0)]}): # trees share some nodes!!
            if num & 1:
                if num not in dic:
                    res = []
                    for i in range(1, num, 2):
                        left = make_tree(i)
                        right = make_tree
(num - i - 1)
                        for i in left:
                            for j in right:
                                node = TreeNode(0)
                                node.left = i
                                node.right = j
                                res.append(node)
                    dic[num] = res
                return dic[num]
            return []

        return make_tree(N)
