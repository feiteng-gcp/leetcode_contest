# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def copy(self, root):
        if root is None:
            return None
        else:
            c = TreeNode(0)
            left = self.copy(root.left)
            right = self.copy(root.right)
            c.left = left
            c.right = right
            return c
    
    def allPossibleFBT(self, N):
        """
        :type N: int
        :rtype: List[TreeNode]
        """
        x = N + 1
        if x % 2 != 0:
            return []
        res = []
        
        if N == 1:
            return [TreeNode(0)]
        else:
            for i in range(1, (N + 1) // 2, 2):
                v = i + 1
                if i > N - 1 - i:
                    break
                else:
                    left 
= self.allPossibleFBT(i)
                    right = self.allPossibleFBT(N - 1 - i)
                    for ll in left:
                        for rr in right:
                            root = TreeNode(0)
                            root.left = self.copy(ll)
                            root.right = self.copy(rr)
                            res.append(root)
                            
                            if i != N - 1 - i:
                                root = TreeNode(0)
                                root.right = self.copy(ll)
                                root.left = self.copy(rr)
                                res.append(root)
            return res
