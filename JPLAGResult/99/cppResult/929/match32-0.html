class Solution {
public:
    bool isEq(string a, string b){
        if(a.length() != b.length())
            return false;
        int cc[26] = {0};
        int l = a.length();
        if(a == b)
            return true;
        
        for(int i = 0 ; i < l; i+=2){
            cc[a[i]-'a']++;
        }
        for(int i = 0 ; i < l; i+=2){
            cc[b[i]-'a']--;
        }
        for(auto k : cc)
            if(k!=0)
                return false;
        for(int i = 1 ; i < l; i+=2){
            cc[a[i]-'a']++;
        }
        for(int i = 1 ; i < l; i+=2){
            cc[b[i]-'a']--;
        }
          for(auto k : cc)
            if(k!=0)
                return false;
        
        return true;
        
    }
    int numSpecialEquivGroups(vector<string>& A) {
        int n = A.size();
        vector<int> uf(n);
        for(int i = 0 ; i < n; ++i)
            uf[i] = i;
        
        for(int i = 1 ; i < n ; ++i){
            for(int j = 0 ; j < i; ++j){
                if(isEq(A[i], A[j])){
                  //  cout<<A[i]<<","<<A[j]<<endl;
                    uf[i] = uf[j];
                    break;
                }
            }
        }
       /* for(auto k : uf)
            cout<<k<<endl;*/
        unordered_set<int> se(uf.begin(),uf.end());
        return se.size();
        
        
    }
};
