const int SIZE = 1e6+10;
class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        set<pair<string,string>>AA;
        for(string s: A){
            string s1,s2;
            REP(i,SZ(s)){
                if(i&1)s1+=s[i];
                else s2+=s[i];
            }
            sort(ALL(s1));
            sort(ALL(s2));
            AA.insert({s1,s2});
        }
        return SZ(AA);
    }
};
