class Solution {
public:
	int numSpecialEquivGroups(vector<string>& A) {
		set<pair<string, string>> ans;

		for (auto s : A) {
			string s1 = "", s2 = "";

			for (int i = 0; i < s.length(); i++) {
				if (i % 2 == 0) s1 = s1 + s[i];
				else s2 = s2 + s[i];
			}
			sort(s1.begin(), s1.end());
			sort(s2.begin(), s2.end());
			ans.insert({ s1, s2 });
		}
		return ans.size();
	}
};
