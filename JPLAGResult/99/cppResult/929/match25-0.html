class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        unordered_set<string> normalize_strings;
        for(auto a: A){
            string norm = normalize(a);
            normalize_strings.insert(norm);
        }
        return normalize_strings.size();
    }
private:
    string normalize(string s){
        stringstream odd, even;
        for(int i = 0; i < s.size(); i ++){
            if ( i % 2 == 0){
                even << s[i];
            }else {
                odd << s[i];
            }
        }
        string odd_str = odd.str(), even_str = even.str();
        sort(odd_str.begin(), odd_str.end());
        sort(even_str.begin(), even_str.end());
        return odd_str + even_str;
    }
};
