class Solution {
public:
    string findKey(string A) {
        string odd = "", even = "";
        
        for (int i = 0; i < A.size(); ++i) {
            if (i % 2) {
                odd += A[i];
            } else {
                even += A[i];
            }
        }
        
        sort(odd.begin(), odd.end());
        sort(even.begin(), even.end());
        
        return odd + "*" + even;
    }
    
    int numSpecialEquivGroups(vector<string>& A) {
        unordered_set<string> dict;
        
        for (auto a : A) {
            string key = findKey(a);
            dict.insert(key);
        }
        
        return dict.size();
    }
};
