class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        set<string> tab;
    int len = (int)A.size(), wlen = (int)A[0].size();
    for(int i = 0; i < len; i++)
    {
        string s, t, r;
        for(int j = 0; j < wlen; j += 2) s.push_back(A[i][j]);
        for(int j = 1; j < wlen; j += 2) t.push_back(A[i][j]);
        sort(s.begin(), s.end()); sort(t.begin(), t.end());
        for(int j = 0; j < wlen; j++)
        {
            if(j & 1) r.push_back(t[j / 2]);
            else r.push_back(s[j / 2]);
        }
        tab.insert(r);
    }
    return (int)tab.size();
    }
};
