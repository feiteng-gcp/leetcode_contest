class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        int sz = A.size();
        int ans = 0;
        unordered_set<string> mp;
        for (int i=0; i < sz; ++i) {
            int len = A[i].length();
            vector<char> first;
            vector<char> second;

            for (int j=0; j < len; ++j) {
                if ( (j%2) == 0) {
                    first.push_back(A[i][j]);
                } else {
                    second.push_back(A[i][j]);
                }
            }
            sort(first.begin(), first.end());
            sort(second.begin(), second.end());
            string s;
            for (int j=0; j < first.size(); ++j)
                s += first[j];
            for (int j=0; j < second.size(); ++j)
                s += second[j];
            if (mp.find(s) == mp.end()) {
                ++ans;
                mp.insert(s);
            }
        }
        return ans;
    }
};
