class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        map<pair<vector<int>,vector<int>>, int> mmp;
        for(auto str: A)
        {
            vector<int> vec1, vec2;
            vec1.resize(26, 0);
            vec2.resize(26, 0);
            for(int i = 0; i < str.size(); i++)
            {
                if(i%2==0)
                    vec1[str[i]-'a']++;
                else
                    vec2[str[i]-'a']++;
            }
            mmp[make_pair(vec1, vec2)] += 1;
        }
        return mmp.size();
    }
};
