class Solution {
 public:
  int numSpecialEquivGroups(vector<string> A) {
    set<string> ans;
    for (auto& a : A) {
      string odd, even;
      for (int i = 0; i < a.size(); ++i) {
        if (i % 2) odd.push_back(a[i]);
        else even.push_back(a[i]);
      }
      sort(odd.begin(), odd.end());
      sort(even.begin(), even.end());
      ans.insert(odd+even);
    }

    return ans.size();
  }
};
