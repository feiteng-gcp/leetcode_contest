class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        int ans=0;
        int n=A.size();
        vector<int>vis(n,0);
        for(int i=0;i<n;++i){
            if(vis[i]) continue;
            int m=A[i].length();
            vector<int>a1(26,0),a2(26,0);
            for(int j=0;j<m;j+=2){
                int t=A[i][j]-'a';
                ++a1[t];
            }
            for(int j=1;j<m;j+=2){
                int t=A[i][j]-'a';
                ++a2[t];
            }
            for(int j=i+1;j<n;++j){
                if(vis[j]) continue;
                int m2=A[i].length();
                if(m2!=m) continue;
                vector<int>b1(26,0),b2(26,0);
                for(int k=0;k<m;k+=2){
                    int t=A[j][k]-'a';
                    ++b1[t];
                }
                for(int k=1;k<m;k+=2){
                    int t=A[j][k]-'a';
                    ++b2[t];
                }
                bool flag=1;
                for(int k=0;k<26;++k){
                    if(a1[k]!=b1[k]||a2[k]!=b2[k]){
                        flag=0;
                        break;
                    }
                }
                if(flag){
                    vis[j]=1;
                }
            }
            vis[i]=1;
            ++ans;
        }
        return ans;
    }
};
