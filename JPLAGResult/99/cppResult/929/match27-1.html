class Solution {
public:
    string hash(string & s) {
        string a = "";
        string b = "";

        for(int i = 0; i < s.size(); i += 2) {
            a += s[i];
        }
        for(int i = 1; i < s.size(); i += 2) {
            b += s[i];
        }

        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        return a + b;
    }
    int numSpecialEquivGroups(vector<string>& A) {
        set<string> s;
        for(auto a: A) {
            s.insert(hash(a));
        }

        return s.size();
    }
};
