class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        int n = A.size();
        set<pair<string,string>> st;
        for(int i=0; i<n; i++) {
            string a = "", b = "";
            for(int j=0; j<A[i].size(); j++) {
                if(j%2) {
                    a += A[i][j];
                } else {
                    b += A[i][j];
                }
            }
            sort(a.begin(), a.end());
            sort(b.begin(), b.end());
            auto tmp = make_pair(a,b);
            if(!st.count(tmp)) 
                st.insert(tmp);
        }
        return st.size();
    }
};
