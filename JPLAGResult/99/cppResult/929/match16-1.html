class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        vector<pair<string, string>> conv;
        conv.reserve(A.size());
        for (string s: A) {
          string odd;
          odd.reserve(s.size());
          string even;
          even.reserve(s.size());

          for (int i =0; i< s.size(); i++) {
            if (i%2) {
              even.push_back(s[i]);
            } else {
              odd.push_back(s[i]);              
            }
          }
          sort(odd.begin(), odd.end());
          sort(even.begin(), even.end());
          conv.emplace_back(move(odd), move(even));
        }
              set<pair<string, string>> m(conv.begin(), conv.end());
      return m.size();

    }
};
