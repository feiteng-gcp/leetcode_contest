class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        set<string> d;
        for (auto s : A) {
            string s0, s1;
            for (int i = 0; i < s.size(); i += 2) {
                s0 += s[i];
            }
            for (int i = 1; i < s.size(); i += 2) {
                s1 += s[i];
            }
            sort(s0.begin(), s0.end());
            sort(s1.begin(), s1.end());
            d.insert(s0 + "_" + s1);
        }
        return d.size();
    }
};
