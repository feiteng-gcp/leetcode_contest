class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        set<string> se;
        for(auto &e:A){
            string a[2];
            string b;
            for(int i=0;i<e.size();++i){
                a[i&1]+=e[i];
            }
            sort(a[0].begin(),a[0].end());
            sort(a[1].begin(),a[1].end());
            for(int i=0;i<e.size();++i){
                b+=a[i&1][i>>1];
            }
            se.insert(b);
        }
        return se.size();
    }
};
