class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
    
        map<string,int> m_map;
        for(int i=0;i<A.size();i++){
            string str=A[i];
            string one="";
            string two="";
            for(int j=0;j<str.size();j++){
                if(j&1){
                    one.append(1,str[j]);
                }else{
                    two.append(1,str[j]);
                }
            }
            sort(one.begin(),one.end());
            sort(two.begin(),two.end());
            string tmp = one+two;
            if(m_map.count(tmp)==0){
                m_map[tmp]=1;
            }
        }
        return m_map.size();
    }
};
