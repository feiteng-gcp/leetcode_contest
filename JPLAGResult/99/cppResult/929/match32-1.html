class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        int n = A.size();
        int sz = A[0].size();
        
        int res = 0;
        unordered_set<string> set;
        for (int i = 0; i < n; ++i) {
            string t1, t2;

            for (int j = 0; j < sz; j+=2) {
                t1 += A[i][j];
            }
            for (int j = 1; j < sz; j+=2) {
                t2 += A[i][j];
            }
            sort(t1.begin(), t1.end());
            sort(t2.begin(), t2.end());
            string t3 = t1+t2;
            if (set.find(t3) == set.end()) {
                ++res;
                set.insert(t3);
            }
        }
        return res;
    }
};
