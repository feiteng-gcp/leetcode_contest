class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        set<pair<string,string>> dp;
        for(auto it:A){
            string ta,tb;

            for(int i=0;i<it.length();i++){
                if(i%2) ta+=it[i];
                else tb+=it[i];
            }
            sort(ta.begin(),ta.end());
            sort(tb.begin(),tb.end());
            dp.insert({ta,tb});
        }
        return dp.size();
    }
};
