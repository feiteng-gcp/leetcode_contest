class Solution {
public:
    
    string getString(string str)
    {
        string A, B;
        for (int i = 0; i < str.size(); i++)
        {
            if (i%2 == 0)
                A += string(1, str[i]);
            else
                B += string(1, str[i]);
        }
        
        sort(A.begin(), A.end());
        sort(B.begin(), B.end());
       
        string result;
        int j = 0, k = 0;
        for (int i = 0; i < str.size(); i++)
        {
            if (i%2 == 0)
                result += string(1, A[j++]);
            else
                result += string(1, B[k++]);
        }
        return result;
    }
    
    int numSpecialEquivGroups(vector<string>& A) {
        
        unordered_set<string> hashset;
        for (int i = 0; i < A.size(); i++)
        {
            string str = getString(A[i]);
            
            hashset.insert(str);
        }
        
        return hashset.size();
    }
};
