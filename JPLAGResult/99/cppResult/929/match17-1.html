class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        unordered_set<string> c;
        for (string &s: A) {
            string a, b;

            for (int i = 0; i < s.length(); ++i) {
                if (i&1)
                    a.push_back(s[i]);
                else
                    b.push_back(s[i]);
            }
            sort(a.begin(), a.end());
            sort(b.begin(), b.end());
            c.insert(a+b);
        }
        return c.size();
    }
};
