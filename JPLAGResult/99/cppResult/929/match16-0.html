class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        
        set<pair<set<char>,set<char> > > ans;
        
        FOR(i, 0, SZ(A)){
            string a = A[i];
           
            set<char> even, odd;
            FOR(j, 0, SL(a)){
                if(j%2==0){
                    even.insert(a[j]);
                } else{
                    odd.insert(a[j]);
                }
            }
            ans.insert(make_pair(even, odd));
        } 
        return ans.size();
    }
};
