class vector_hash {
public:
  std::size_t operator()(std::vector<int> const& vec) const {
    std::size_t ret = 0;
    for(int i=0; i<vec.size(); i++) {
      ret ^= vec[i]-2*i+(i-137)*i;
    }
    return ret;
  }
};
class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        unordered_set<vector<int>, vector_hash> st;
        int ans=0;
        for(string& str:A)
        {
            vector<int> count(52,0);
            for(int i=0; i<str.size(); i++)
            {
                char ch=str[i];

                if(i%2)
                {
                    count[ch-'a']++;
                }
                else{
                    count[ch-'a'+26]++;
                }
            }
            auto iter=st.find(count);
            if(iter==st.end())
            {
                ++ans;
                st.insert(count);
            }
        }
        return ans;
    }
};
