class Solution {
public:
    int numSpecialEquivGroups(vector<string>& A) {
        set<pair<multiset<char>,multiset<char>> > s;
        for(int i=0;i<A.size();i++){
            string st = A[i];
            multiset<char> m1;
            multiset<char> m2;
            for(int i=0;i<st.length();i++){
                if(i%2==0)
                    m1.insert(st[i]);
                else
                    m2.insert(st[i]);
            }
            s.insert(make_pair(m1,m2));
        }
        return s.size();
        
    }
};
