class Solution {
public:
    int surfaceArea(vector<vector<int>>& A) {
      int r = A.size(),ans = 0;
      int c = A[0].size();
      for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
          if(A[i][j]==0) continue;
          int n=A[i][j],s=A[i][j],e=A[i][j],w=A[i][j];
          if(i>0) w = max(0,A[i][j]-A[i-1][j]);
          if(j>0) n = max(0,A[i][j]-A[i][j-1]);
          if(i<r-1) e = max(0,A[i][j]-A[i+1][j]);
          if(j<c-1) s = max(0,A[i][j]-A[i][j+1]);
          ans+=(2+n+s+e+w);
        }
      }
      return ans;
    }
};
