class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int count = 0, toge = 0;
        
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[0].size(); ++j) {
                if (grid[i][j] > 1) {
                    toge += (grid[i][j] - 1);
                }
                count += grid[i][j];
                
                if (j < grid[0].size() - 1) {
                    toge += min(grid[i][j], grid[i][j + 1]);
                }
            }
        }
        
        for (int i = 0; i < grid[0].size(); ++i) {
            for (int j = 0; j < grid.size() - 1; ++j) {
                toge += min(grid[j][i], grid[j + 1][i]);
            }
        }
        
        return count * 6 - toge * 2;
    }
};
