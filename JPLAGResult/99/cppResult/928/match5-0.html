class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int ans = 0;
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[i].size(); j++) {
                for (int k = 1; k <= grid[i][j]; k++) {
                    if (k == 1) ans += 1;
                    if (k == grid[i][j]) ans += 1;
                    if (i == 0 || k > grid[i - 1][j]) ans += 1;
                    if (i == grid.size() - 1 || k > grid[i + 1][j]) ans += 1;
                    if (j == 0 || k > grid[i][j - 1]) ans += 1;
                    if (j == grid[i].size() - 1 || k > grid[i][j + 1]) ans += 1;
                }
            }
        }
        return ans;
    }
};
