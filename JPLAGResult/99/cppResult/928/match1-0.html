class Solution {
public:
    int surfaceArea(vector<vector<int>>& a) {
		int ans = 0;
		int n = a.size();
		int dr[] = { 0,1,0,-1 };
		int dc[] = { 1,0,-1,0 };

		for (int i = 0; i < n; ++i)
		{
			for (int j = 0; j < n; ++j)
			{
				if (a[i][j])
				{
					ans += a[i][j] * 4 + 2;
					for (int d = 0; d < 4; ++d)
					{
						int nx = i + dr[d];
						int ny = j + dc[d];
						if (0 <= nx && nx < n && 0 <= ny && ny < n)
						{
							ans -= min(a[nx][ny], a[i][j]);
						}
					}
				}
			}
		}
		return ans;
    }
};
