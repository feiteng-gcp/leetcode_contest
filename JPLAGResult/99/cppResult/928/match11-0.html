const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
class Solution {
public:
    int surfaceArea(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (a[i][j]) ret += 2;
            }
        }
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int k = 0; k < 4; ++k) {
                    int ni = i + d[k][0];
                    int nj = j + d[k][1];
                    if (ni < 0 || ni >= n || nj < 0 || nj >= m) {
                        ret += a[i][j];
                    } else {
                        if (a[i][j] > a[ni][nj]) {
                            ret += a[i][j] - a[ni][nj];
                        }
                    }
                }
            }
        }
        return ret;
        
    }
};
