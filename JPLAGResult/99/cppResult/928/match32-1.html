class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int area = 0;
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid[0].size(); j++){
                if(grid[i][j] != 0){
                    // add z top and bottom contribution
                    area += 2;
                    
                    // from remaining 4 sides
                    area += max(0, grid[i][j] - ((i == 0) ? 0 : grid[i - 1][j]));
                    area += max(0, grid[i][j] - ((i == grid.size() - 1) ? 0 : grid[i + 1][j]));
                    area += max(0, grid[i][j] - ((j == 0) ? 0 : grid[i][j - 1]));
                    area += max(0, grid[i][j] - ((j == grid[0].size() - 1) ? 0 : grid[i][j + 1]));
                }
            }
        }
        return area;
    }
};
