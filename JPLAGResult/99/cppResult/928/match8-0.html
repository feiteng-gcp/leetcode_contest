class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int ans=0;
        int n=grid.size();
        int m=grid[0].size();
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++){
                if(i==0){
                    ans+=grid[i][j];
                }
                else {
                    ans+=max(grid[i][j]-grid[i-1][j],0);
                }
                if(i==n-1){
                    ans+=grid[i][j];
                }
                else{
                    ans+=max(grid[i][j]-grid[i+1][j],0);
                }
                if(j==0){
                    ans+=grid[i][j];
                }
                else{
                    ans+=max(grid[i][j]-grid[i][j-1],0);
                }
                if(j==m-1){
                    ans+=grid[i][j];
                }
                else{
                    ans+=max(grid[i][j]-grid[i][j+1],0);
                }
                if(grid[i][j])
                ans+=2;
            }
        return ans;
    }
};
