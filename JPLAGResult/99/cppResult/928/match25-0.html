class Solution {
public:
    
    
    int surfaceArea(vector<vector<int>>& grid) {
        int ans = 0;
        
        int mr[] = { 1, -1, 0,  0};
        int mc[] = { 0,  0, 1, -1};
        
        for (int i = 0; i < grid.size(); ++i)
            for (int j = 0; j < grid[i].size(); ++j){
                for (int k = 0; k < 4; ++k){
                    int it = i+mr[k];
                    int jt = j+mc[k];
                    
                    int neighbor = (it < 0 || it >= grid.size() || jt < 0 || jt >= grid[it].size())? 0 : grid[it][jt];
                    ans += max(0, grid[i][j] - neighbor);
                }
                
                if (grid[i][j] > 0) ans += 2;
            }
        
        return ans;
    }
};
