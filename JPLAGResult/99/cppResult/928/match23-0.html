class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int ret=0;
        if(grid.size()==0) return 0;
        int row=grid.size();
        int col=grid[0].size();
        for(int i=0;i<grid.size();i++){
            for(int j=0;j<grid[0].size();j++){
                if(grid[i][j]>0) ret+=2;
            }
        }
        for(int i=0;i<grid.size();i++){
            for(int j=0;j<col-1;j++){
                if(grid[i][j]>grid[i][j+1]){
                    ret+=(grid[i][j]-grid[i][j+1]);
                }
                if(grid[i][col-1-j]>grid[i][col-2-j]){
                    ret+=(grid[i][col-1-j]-grid[i][col-2-j]);
                }
            }
            ret+=grid[i][0];
            ret+=grid[i][col-1];
        }
        for(int j=0;j<col;j++){
            for(int i=0;i<row-1;i++){
                if(grid[i][j]>grid[i+1][j]){
                    ret+=(grid[i][j]-grid[i+1][j]);
                }
                if(grid[row-1-i][j]>grid[row-2-i][j]){
                    ret+=(grid[row-1-i][j]-grid[row-2-i][j]);
                }
            }
            ret+=grid[0][j];
            ret+=grid[row-1][j];
        }
        return ret;
    }
};
