class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int n = grid.size();
        int result = 0;
        vector<int> dx = {-1, 1, 0, 0};
        vector<int> dy = {0, 0, -1, 1};
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (grid[i][j]>0) result+=2;
                for (int k = 0; k < 4; k++)
                {
                    int x = dx[k]+i;
                    int y = dy[k]+j;
                    for (int h = 0; h < grid[i][j]; h++)
                    {
                       if (x<0 || x>=n || y<0 || y>=n || grid[x][y]<grid[i][j]-h)
                            result++; 
                    }
                }
                
            }
        }
        return result;
    }
};
