class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int n = grid.size();
        if (grid.empty())
            return 0;
        
        int result = 2 * n * n;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0) {
                    result -= 2;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            result += grid[i][0] + grid[i][n - 1];
        }
        for (int j = 0; j < n; j++) {

            result += grid[0][j] + grid[n - 1][j];
        }
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n; j++) {
                result += abs(grid[i][j] - grid[i + 1][j]);
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - 1; j++) {
                result += abs(grid[i][j] - grid[i][j + 1]);
            }
        }
        return result;
    }
};
