class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int n = grid.size();

        vector<vector<int>> dirs = {{0,1},{0,-1},{1,0},{-1,0}};

        int sa = 0;
        for(int i = 0 ; i < n ; ++i){
            for(int j = 0; j < n; ++j){
                if(grid[i][j] !=0){
                    sa += (4* grid[i][j] + 2);
                    for(auto dir : dirs){
                        int ii = i+ dir[0];
                        int jj = j + dir[1];
                        if(ii >=0 && ii <n && jj >=0 && jj < n){
                            sa -= min(grid[i][j], grid[ii][jj]);
                        }
                    }
                }
            }
        }
        return sa;
        
    }
};
