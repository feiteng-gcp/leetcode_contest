class Solution {
public:
    int surfaceArea(vector<vector<int>>& a) {
        if (a.empty())
            return 0;
        int ans = 0;

        int n = a.size();
        int m = a[0].size();
        int sx[] = {-1, 0, 1, 0};
        int sy[] = {0, 1, 0, -1};
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) if (a[i][j]) {

                ans += 2;
                for (int k = 0; k < 4; ++k) {
                    int nx = i + sx[k];
                    int ny = j + sy[k];
                    if (nx < 0 || nx >= n || ny < 0 || ny >= m) {
                        ans += a[i][j];
                        continue;
                    }
                    int diff = a[i][j] - a[nx][ny];
                    if (diff >= 0)
                        ans += diff;
                }
            }
        }
        return ans;
    }
};
