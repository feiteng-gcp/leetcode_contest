int get_val(int i,int j,int n,int m,vector<vector<int>>&a){
    int x=i-1,y=j,val=a[i][j],ans=0;
    if(x<0){
        ans+=a[i][j];
    }
    else{
        ans+=(val>a[x][y])?val-a[x][y]:0;
    }
    x=i+1;
    if(x>=n){
        ans+=a[i][j];
    }
    else{
        ans+=(val>a[x][y])?val-a[x][y]:0;
    }
    x=i; y=j-1;
    if(y<0){
        ans+=a[i][j];
    }
    else{
        ans+=(val>a[x][y])?val-a[x][y]:0;
    }
    y=j+1;
    if(y>=m){
        ans+=a[i][j];
    }
    else{
        ans+=(val>a[x][y])?val-a[x][y]:0;
    }
    return ans;
}
class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int ans=0;
        int n=grid.size();
        int m=grid[0].size();
        int cnt=0;
        for(int i=0;i<n;++i){
            for(int j=0;j<grid[i].size();++j){
                if(grid[i][j]){
                    ++cnt;
                    ans+=get_val(i,j,n,m,grid);
                }
            }
        }
        ans+=cnt*2;
        return ans;
    }
};
