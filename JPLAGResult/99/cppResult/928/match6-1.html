class Solution {
public:
	int surfaceArea(vector<vector<int>> grid) {
		int n = grid.size();
		if (n == 1) {
			if (grid[0][0] == 0)return 0;
			return grid[0][0] * 4 + 2;
		}

		int ret = 0;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (grid[i][j] > 0)ret++;
			}
		}
		ret *= 2;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (!grid[i][j])continue;
				if (!i) ret += grid[i][j];
				if (i == n - 1) ret += grid[i][j];

				if (i < n - 1) ret += max(grid[i][j] - grid[i + 1][j],0);
				if (i) ret += max(grid[i][j] - grid[i - 1][j],0);


				if (!j) ret += grid[i][j];
				if (j) ret += max(grid[i][j] - grid[i][j - 1],0);
				if (j == n - 1) ret += grid[i][j];
				if (j < n - 1) ret += max(grid[i][j] - grid[i][j + 1],0);
			}
			
		}
		return ret;
	}
};
