class Solution {
public:

    int dx[4]={0,-1,0,1},dy[4]={-1,0,1,0};
    int surfaceArea(vector<vector<int>>& grid) {
        int res=0;
        int n=grid.size();
        for(int i=0;i<n;++i){
            for(int j=0;j<n;++j){
                if(grid[i][j]){
                    res+=2;
                    for(int d=0;d<4;++d){
                        int x=i+dx[d],y=j+dy[d];
                        if(x<0 || x>=n || y<0 || y>=n){
                            res+=grid[i][j];
                        }
                        else{
                            if(grid[i][j]>grid[x][y]){
                                res+=grid[i][j]-grid[x][y];
                            }
                        }
                    }
                }
            }
        }
        return res;
    }
};
