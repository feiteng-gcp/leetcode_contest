class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int n=grid.size();
        int offs[4][2]={{-1,0},{1,0},{0,-1},{0,1}};
        int res=0;
        for(int i=0;i<n;++i)
            for(int j=0;j<n;++j){
                if(grid[i][j]){
                    res+=2;
                    for(int k=0;k<4;++k){
                        int x=i+offs[k][0];
                        int y=j+offs[k][1];
                        if(x>=0 && x<n && y>=0 && y<n){
                            if(grid[i][j]>grid[x][y])
                                res+=grid[i][j]-grid[x][y];
                        }
                        else
                            res+=grid[i][j];
                    }
                }
            }
        return res;
    }
};
