class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
      int area = 0;
      int nrow = grid.size();
      int ncol = grid[0].size();
      for (int r = 0; r < nrow; r++) {
        for (int c = 0; c < ncol; ++c) {
          int ncube = grid[r][c];
          if (ncube == 0) continue;
          area += 4 * ncube + 2;
          if (r - 1 >= 0 && grid[r-1][c] > 0) {
            int min_comm = min(grid[r-1][c], ncube);
            area -= 2 * min_comm;
          }
          if (c - 1 >= 0 && grid[r][c-1] > 0) {
            int min_comm = min(grid[r][c-1], ncube);
            area -= 2 * min_comm;
          }
        }
      }
      return area;
    }
};
