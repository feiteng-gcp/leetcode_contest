class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        if(grid.empty() || grid[0].empty()) return 0;
        
        int m = grid.size(), n = grid[0].size();
        int dirs[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        int total = 0, neig = 0, ud = 0;
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                total += grid[i][j];
                if(grid[i][j] > 1)
                    ud += grid[i][j] - 1;
                for(int d = 0; d < 4; d++){
                    int ii = i + dirs[d][0];
                    int jj = j + dirs[d][1];
                    if(ii >= 0 && ii < m && jj >= 0 && jj < n){
                        neig += min(grid[i][j], grid[ii][jj]);
                    }
                }
            }
        }
        return total * 6 - neig - ud * 2;
    }
};
