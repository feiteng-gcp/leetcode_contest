class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int sz = grid.size();
        int empty = 0;
        int tot = 0;
        int entire = 0;

        for (int r=0; r < sz; ++r) {
            for (int c=0; c < sz; ++c) {
                if (grid[r][c] == 0) {
                    ++empty;
                    continue;
                }
                entire += (grid[r][c]*4) + 2;
                //int tot = 0;
                if (r > 0) {
                    tot += min(grid[r][c], grid[r-1][c]);
                }
                if (r < (sz-1)) {
                    tot += min(grid[r][c], grid[r+1][c]);
                }
                if (c > 0) {
                    tot += min(grid[r][c], grid[r][c-1]);
                }
                if (c < (sz-1)) {
                    tot += min(grid[r][c], grid[r][c+1]);
                }
            }
        }
        return (entire - tot);
    }
};
