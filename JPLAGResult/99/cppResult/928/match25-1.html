class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
      N = grid.size();

      int answer = 0;
      int dx[] = {1,0,-1,0};
      int dy[] = {0,1,0,-1};
      
      for (int i =0; i< N; i++)
      for (int j =0; j< N; j++) {
      for (int k = 0; k<4; k++)
      {
        if (get(i,j,grid) > get(i+dx[k], j+dy[k],grid)) {
          answer += get(i,j,grid) - get(i+dx[k], j+dy[k],grid);
        }
      }
        if (get(i,j,grid)) answer+=2;
      }
      return answer;
    }
  int get(int i, int j, vector<vector<int>>& grid) {
    if (i < 0 || i>=N || j<0 || j >= N) 
      return 0;
    return grid[i][j];
  }
  int N;
};
