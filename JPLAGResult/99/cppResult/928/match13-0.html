class Solution {
public:
    int help[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
    int surfaceArea(vector<vector<int>>& grid) {
        int res = 0;
        int m = grid.size();
        
        for(int i=0;i<m;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]){
                    res+=2;
                    for(int t=0;t<4;t++){
                        int ni = i+help[t][0];
                        int nj = j+help[t][1];
                        if(ni<0||ni>=m||nj<0||nj>=m||grid[ni][nj]==0){
                            res+=grid[i][j];
                        }
                        else if(grid[i][j]>grid[ni][nj]){
                            res+=grid[i][j]-grid[ni][nj];
                        }
                    }
                }
            }
        }
        return res;
    }
};
