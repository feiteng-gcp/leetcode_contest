const int dr[4]={-1,0,1,0};
const int dc[4]={0,1,0,-1};
class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid)
    {
        if(grid.empty())return 0;
        

        int area=0;
        int n=grid.size();
        for(int r=0;r<n;++r)
        {
            for(int c=0;c<n;++c)
            {
                int h=grid[r][c];

                if(0<h)
                {
                    area+=2;
                    for(int i=0;i<4;++i)
                    {
                        int r1=r+dr[i];
                        int c1=c+dc[i];
                        if(r1<0 || r1>=n || c1<0 || c1>=n)
                        {
                            area+=h;
                        }
                        else if(grid[r1][c1]<h)
                        {
                            area+=h-grid[r1][c1];
                        }
                    }
                }
            }
        }
        return area;
    }
};
