class Solution {
public:
    int surfaceArea(vector<vector<int>>& g) {
        if (g.empty() || g[0].empty()) return 0;
        int res = 0;
        for(int i=0; i<g.size(); ++i) {
            for(int j=0; j<g[0].size(); ++j) {
                int h = g[i][j];
                for(int k=0; k<h; ++k) {
                    if (k == 0) res += 1;
                    if (k+1 == h) res += 1;
                    // i, j
                    if (i == 0 || k+1 > g[i-1][j]) res += 1;
                    if (i + 1 == g.size() || k+1 > g[i+1][j]) res += 1;
                    if (j == 0 || k+1 > g[i][j-1]) res += 1;
                    if (j+1 == g[0].size() || k+1 > g[i][j+1]) res += 1;
                    
                }
                
            }
        }
        return res;
    }
};
