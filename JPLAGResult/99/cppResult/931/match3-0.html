class FreqStack {
public:
    
    list<int> l;
    unordered_map<int, int> hashmap; 
    map<int, unordered_set<int> > hashmap2; 
    
    FreqStack() {
        
    }
    
    void push(int x) {
        
        l.push_front(x);
        
        if (hashmap.count(x) == 0)
        {
            hashmap[x] = 1;
            hashmap2[1].insert(x);
        }
        else
        {
            hashmap[x] = hashmap[x] + 1;
            
            int freq = hashmap[x];
            hashmap2[freq].insert(x);
            hashmap2[freq-1].erase(x);
            
            if (hashmap2[freq-1].size() == 0)
                hashmap2.erase(freq-1);
        }
    }
    
    int pop() {
        
        int freq = hashmap2.rbegin()->first;
      //  cout << "popong freq" << freq << endl;
        for (auto it = l.begin(); it != l.end(); it++)
        {
            if (hashmap2[freq].count(*it))
            {
                int result = *it;
               // cout << "result " <<result <<endl;
                //find it 
                l.erase(it);
                hashmap[result] = hashmap[result] - 1;
                if (hashmap[result] == 0)
                {
                    hashmap.erase(result);
                    hashmap2[1].erase(result);
                    if (hashmap2[1].size() == 0)
                        hashmap2.erase(1);
                }
                else
                {
                    int oldFreq = freq;
                    hashmap2[oldFreq].erase(result);
                    if (hashmap2[oldFreq].size() == 0)
                        hashmap2.erase(oldFreq);
                    hashmap2[oldFreq-1].insert(result);
                }
                
                return result;
            }
        }
        
        return 0;
    }
};
