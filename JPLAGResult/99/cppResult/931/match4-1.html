class FreqStack {
public:
    FreqStack() {
        
    }
    
    void push(int x) {
        mp[x]++;
        data d;
        d.num=x;
        d.fre=mp[x];
        d.idx=idx++;
        pq.push(d);
    }
    
    int pop() {
        data d=pq.top();
        pq.pop();
        return d.num;
    }
private:
    int idx=0;
    unordered_map<int,int> mp;

    struct data{
        int num=0;
        int fre=0;
        int idx=0;
    };
    struct cmp{
        bool operator()(const data d1,const data d2){
            return d1.fre==d2.fre?d1.idx<d2.idx:d1.fre<d2.fre;
        }
    };
    priority_queue<data,vector<data>,cmp> pq;
};
