class FreqStack {
private:
    int tot;
    map<int,int> ma;
    map<pair<int,int> , int> mm;
public:
    FreqStack() {
        tot=0;
        ma.clear();
        mm.clear();
    }

    void push(int x) {
        mm[pair<int,int>(++ma[x],++tot)] = x;
    }

    int pop() {
        auto it=mm.end();
        --it;
        int ans=it->second;
        --ma[ans];
        mm.erase(it);
        return ans;
    }
};
