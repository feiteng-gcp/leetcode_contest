class FreqStack {
    unordered_map<int, int> cc;
    unordered_map<int, set<int> > pos;
    set< pair<int, pair<int, int> > > order;
    int id;
public:
    FreqStack() {
        id = 0;
    }
    
    void push(int x) {
        id += 1;
        if (cc[x] == 0) {
            cc[x] = 1;
            pos[x].insert(id);
            order.insert(make_pair(1, make_pair(id, x)));
        } else {
            int old_freq = cc[x];
            int old_pos = *pos[x].rbegin();
            order.erase(order.find(make_pair(old_freq, make_pair(old_pos, x))));
            old_freq += 1;
            pos[x].insert(id);
            cc[x] += 1;
            order.insert(make_pair(old_freq, make_pair(id, x)));
        }
    }
    
    int pop() {
        auto item = *order.rbegin();
        int x = item.second.second;
        int freq = item.first;
        int loc = item.second.first;
        if (freq == 1) {
            pos.erase(x);cc.erase(x);
            order.erase(std::prev(order.end()));
        } else {
            pos[x].erase(pos[x].find(loc));
            cc[x] -= 1;
            loc = *pos[x].rbegin();
            order.erase(std::prev(order.end()));
            order.insert(make_pair(freq-1, make_pair(loc, x)));
        }
        return x;
    }
};
