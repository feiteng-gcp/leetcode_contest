class FreqStack {
    
private:
    int id;
    map<int, int> freq;
    priority_queue<pair<int,pair<int,int> > > freqPQ;
    
public:
    
    
    FreqStack() {
        id = 0;
    }
    
    void push(int x) {
        freq[x]++;
        freqPQ.push(make_pair(freq[x], make_pair(id++,x)));
    }
    
    int pop() {
        while (freqPQ.top().first != freq[freqPQ.top().second.second])
            freqPQ.pop();
        
        int ans = freqPQ.top().second.second;
        freq[ans]--;
        freqPQ.pop();
        //id--;
        return ans;
    }
};

 * FreqStack obj = new FreqStack();
 * obj.push(x);
 * int param_2 = obj.pop();
