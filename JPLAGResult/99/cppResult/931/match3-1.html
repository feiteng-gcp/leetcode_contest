class FreqStack
{
public:
	using Iter = list<int>::iterator;
	FreqStack()
	{

	}

	void push(int x)
	{
		auto& cnt = num_cnt_[x];
		++cnt;
		cnt_map_[cnt].insert(x);
		if (cnt>1)
		{
			cnt_map_[cnt - 1].erase(x);
		}
		nums_.push_front(x);
	}

	int pop()
	{
		auto pos = cnt_map_.begin();
		int cnt = pos->first;

		auto& tmp_set = pos->second;
		for (auto it = nums_.begin(); it != nums_.end(); ++it)
		{
			if (tmp_set.find(*it) != tmp_set.end())
			{
				int val = *it;
				nums_.erase(it);
				tmp_set.erase(val);
				--num_cnt_[val];
				if (tmp_set.empty())cnt_map_.erase(pos);
				if (cnt>1) cnt_map_[cnt - 1].insert(val);
				return val;
			}
		}
		return -1;
	}
private:
	unordered_map<int, int> num_cnt_;
	map<int, unordered_set<int>, greater<int>> cnt_map_;
	list<int> nums_;
};
