class Solution {
public:
	int n;
	vector<TreeNode*> solve(int x) {
		vector<TreeNode*>ret;
		if (!x || x== 2) return vector<TreeNode*>();
		if (x == 1) {
			ret.push_back(new TreeNode(0));
			return ret;
		}
		for (int i = 1; i < x - 1; i++) {

			auto a = allPossibleFBT(i),b = allPossibleFBT(x-i-1);
			for (auto it:a) {
				for (auto jt : b) {
					TreeNode* p = new TreeNode(0); p->left = it, p->right = jt;
					ret.push_back(p);
				}
			}
		}
		return ret;
	}
	vector<TreeNode*> allPossibleFBT(int N) {
		n = N;
		//ret.clear();
		
		return solve(n);
	}
};
