/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> allPossibleFBT(int N) {
        vector<TreeNode*>  ret;
        
        
        if(N==1){
            TreeNode * root=new TreeNode(0);
            ret.push_back(root);
            return ret;
        }
        int left=1;
        while(left<N-1){
            if(left&1==0 ||(N-1-left)&1==0){
                continue;
            }

            vector<TreeNode*> ll = allPossibleFBT(left);
             vector<TreeNode*> rr=allPossibleFBT(N-1-left);
             for(int i=0;i<ll.size();i++){
                 for(int j=0;j<rr.size();j++){
                      TreeNode * root=new TreeNode(0);
                      root->left=ll[i];
                      root->right=rr[j];
                      ret.push_back(root);
                 }
             }
            left++;
        }
        return ret;
        
    }
};
