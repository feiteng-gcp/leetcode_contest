/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> allPossibleFBT(int N) {
        vector<TreeNode*> result;
        if(N % 2 == 0 )
            return result;
        if(N==1){

            result.push_back(new TreeNode(0));
            return result;
        } else {
            for(int i = 1 ; i < N; i+=2){
                vector<TreeNode*> lt = allPossibleFBT(i);
                vector<TreeNode*> rt = allPossibleFBT(N-1-i);
                for(auto k : lt){

                        for(auto p : rt){
                            TreeNode * root = new TreeNode(0);
                            root->left = k;
                            root->right = p;
                            result.push_back(root);
                        }
                    }
            }
            return result;
        }
        
    }
};
