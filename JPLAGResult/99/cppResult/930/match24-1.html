/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> allPossibleFBT(int N) {
        vector<vector<TreeNode*>> FBTs(N/2+1);
        TreeNode* root=new TreeNode(0);
        if (N&1) {
            FBTs[0].push_back(root);
            for (int i=1; i*2<N; i++)

                for (int j=0; j<i; j++)
                    for (int k=0; k<FBTs[j].size(); k++)
                        for (int p=0; p<FBTs[i-j-1].size(); p++) {
                            TreeNode* root=new TreeNode(0);
                            root->left=FBTs[j][k];
                            root->right=FBTs[i-j-1][p];
                            FBTs[i].push_back(root);
                        }
        }
        return FBTs[N/2];
    }
};
