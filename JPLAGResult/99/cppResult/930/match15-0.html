/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> allPossibleFBT(int N) {
        if(N % 2 == 0) return {};
        
        unordered_map<int, vector<TreeNode*>> dict;
        return allPossibleFBT(N, dict);
    }
    
    vector<TreeNode*> allPossibleFBT(int N, unordered_map<int, vector<TreeNode*>>& dict){
        if(N < 0) return {};
        if(N == 1) return {new TreeNode(0)};
        if(dict.find(N) != dict.end()) return dict[N];
        
        vector<TreeNode*> res;
        for(int mid = 2; mid < N; mid += 2){
            vector<TreeNode*> leftRes = allPossibleFBT(mid - 1, dict);
            vector<TreeNode*> rightRes = allPossibleFBT(N - mid, dict);
            for(auto l: leftRes){
                for(auto r: rightRes){
                    TreeNode* root = new TreeNode(0);
                    root->left = l;
                    root->right = r;
                    res.push_back(root);
                }
            }
        }
        dict[N] = res;
        return res;
    }
};
