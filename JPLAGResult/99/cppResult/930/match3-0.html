/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    TreeNode* clone(TreeNode* root)
    {
        if (root == NULL)
            return root;
        
        TreeNode* newRoot = new TreeNode(0);
        newRoot->left = clone(root->left);
        newRoot->right = clone(root->right);
        
        return newRoot;
    }
    
    vector<TreeNode*> allPossibleFBT(int N) {
        vector<vector<TreeNode*>> result;
        result.push_back(vector<TreeNode*>());
        result.push_back(vector<TreeNode*>({new TreeNode(0)}));
        result.push_back(vector<TreeNode*>());
        
        
        for (int i = 3; i <= N; i++)
        {
            result.push_back(vector<TreeNode*>());
            for (int j = 1; j <= i-1; j++)
            {
                auto lT = result[j];
                auto rT = result[i-j-1];
                
                if (lT.size() >0 && rT.size() >0)
                {
                    
                    for (int k = 0; k < lT.size(); k++)
                    {
                        for (int l = 0; l < rT.size(); l++)
                        {
                            TreeNode* root = new TreeNode(0);
                            root->left = clone(lT[k]);
                            root->right = clone(rT[l]);
                            result[i].push_back(root);
                        }
                    }
                }
            }
        }
                         
        return result[N];
    }
};
