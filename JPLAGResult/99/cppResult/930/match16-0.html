/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> allPossibleFBT(int N) {
        unordered_map<string, vector<TreeNode*>> memo;
        return buildTree(0, N-1, memo);
    }
private:
    vector<TreeNode*> buildTree(int start, int end, unordered_map<string, vector<TreeNode*>>& memo){
        if ( (end - start + 1) % 2  == 0) return {};
        if (start > end) return {nullptr};
        if (start == end) return {new TreeNode(0)};
        string key = to_string(start) + "#" + to_string(end);
        // cout << "key:" << key << endl;
        if (memo.count(key)) return memo[key];
        
        vector<TreeNode*> res;
        
        for(int i = start + 1; i <= end; i +=2){
            vector<TreeNode*> left = buildTree(start, i-1, memo);
            vector<TreeNode*> right = buildTree(i+1, end, memo);
            for(auto l: left){
                for(auto r: right){
                    TreeNode * root = new TreeNode(0);
                    root->left = l;
                    root->right = r;
                    res.push_back(root);
                }
            }
            
        }
        memo[key] = res;
        return res;
    }
};
