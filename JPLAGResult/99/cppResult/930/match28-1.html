/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> allPossibleFBT(int N) {
        if(!(N%2))
        {
            return {};
        }
        if(N==1){
            TreeNode* ptr = new TreeNode(0);
            return {ptr};
        }
        vector<TreeNode*> ans;

        for(int left=1, right= N-1-left; right>=1; left+=2, right-=2){
            auto left_cand = allPossibleFBT(left);
            auto right_cand = allPossibleFBT(right);
            for(auto lptr:left_cand)
            {
                for(auto rptr:right_cand)
                {
                    TreeNode* ptr = new TreeNode(0);
                    ptr->left = lptr;
                    ptr->right = rptr;
                    ans.push_back(ptr);
                }
            }
        }
        return ans;
    }
};
