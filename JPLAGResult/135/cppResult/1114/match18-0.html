/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int cur = 0;
    TreeNode* bstToGst(TreeNode* root) {
        go(root);
        return root;
    }
    
    void go(TreeNode* root) {
        if (!root)
            return ;
        go(root->right);
        cur += root->val;
        root->val = cur;
        go(root->left);
    }
};
