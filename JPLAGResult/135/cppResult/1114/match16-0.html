/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    int curr;
    void rightFirst(TreeNode* root){
        if(!root) return;
        rightFirst(root->right);
        root->val += curr;
        curr = root->val;
        rightFirst(root->left);
    }
public:
    TreeNode* bstToGst(TreeNode* root) {
        curr = 0;
        rightFirst(root);
        return root;
    }
};
