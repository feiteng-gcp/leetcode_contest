/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
    void dfs(TreeNode* root, int &s) {
        if (!root)
            return;
        dfs(root->right, s);
        s += root->val;
        root->val = s;
        dfs(root->left, s);
    }
public:
    TreeNode* bstToGst(TreeNode* root) {
        int s = 0;
        dfs(root, s);
        return root;
    }
};
