/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void ReverseInOrder(TreeNode* root, int& curValue)
	{
		if (root == nullptr)
			return;
		ReverseInOrder(root->right, curValue);
		root->val += curValue;
        curValue = root->val;
		ReverseInOrder(root->left, curValue);
	}

	TreeNode* bstToGst(TreeNode* root) {
		int curValue = 0;
		ReverseInOrder(root, curValue);
		return root;
	}
};
