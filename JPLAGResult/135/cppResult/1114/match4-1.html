/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* bstToGst(TreeNode* root) {
        int sum = 0;
        dfs(root, sum);
        return root;
    }
    void dfs(TreeNode* curr, int& sum)
    {
        if(!curr)
            return;
        dfs(curr->right, sum);
        sum+=curr->val;
        curr->val = sum;
        dfs(curr->left, sum);
    }
};
