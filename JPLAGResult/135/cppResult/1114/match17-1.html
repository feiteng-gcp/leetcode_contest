/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int ans = 0;
    void dfs(TreeNode *root) {
        if (root == nullptr) return;
        dfs(root -> right);
        ans += root -> val;
        root -> val = ans;
        dfs(root -> left);
    }
    TreeNode* bstToGst(TreeNode* root) {
        dfs(root);
        return root;
    }
};
