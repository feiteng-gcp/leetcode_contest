/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstToGst(TreeNode* root) {
        int val = 0;
        solve(root, val);
        return root;
    }
    
    void solve(TreeNode* root, int& val) {
        if (root == NULL) return;
        solve(root->right, val);
        val += root->val;
        root->val = val;
        solve(root->left, val); 
    }
};
