/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    void dfs(TreeNode *cur, int &acc)
    {
        if (cur==nullptr)
            return ;
        dfs(cur->right, acc);
        acc += cur->val;
        cur->val = acc;
        dfs(cur->left, acc);
    }
    TreeNode* bstToGst(TreeNode* root) {
        int acc = 0;
        dfs(root, acc);
        return root;
    }
};
