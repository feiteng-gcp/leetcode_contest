/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    void F(TreeNode* root, int &Sum){
        if(!root)
            return;
        F(root->right, Sum);
        root->val = Sum += root->val;
        F(root->left, Sum);
    }
public:
    TreeNode* bstToGst(TreeNode* root) {
        int Sum = 0;
        F(root, Sum);
        return root;
    }
};
