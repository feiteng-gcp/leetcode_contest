class Solution {
public:

    int minScoreTriangulation(vector<int>& A) {
        int n = A.size();
        int DP[n][n];
        for (int gap = 0; gap < n; gap++) {
            for (int i = 0, j = gap; j < n; i++, j++) {
                if (j < i + 2) DP[i][j] = 0;
                else {
                    DP[i][j] = 1 << 30;
                    for (int k = i + 1; k < j; k++) {
                        int val = DP[i][k] + DP[k][j] + A[i] * A[j] * A[k];
                        if (DP[i][j] > val) DP[i][j] = val;
                    }
                }
            }
        }  
        return  DP[0][n-1];
    }
};
