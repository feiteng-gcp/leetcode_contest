class Solution {
public:
    int INF=0x3f3f3f;
    int minScoreTriangulation(vector<int>& A) {
        int n=A.size();
        vector<vector<int> > dp(n,vector<int>(n,0));
        for(int len=3;len<=n;len++)
        {
            for(int l=0,r;(r=l+len-1)<n;l++)
            {
                dp[l][r]=INF;
                for(int k=l+1;k<r;k++)
                {
                    dp[l][r]=min(dp[l][r],dp[l][k]+dp[k][r]+A[l]*A[r]*A[k]);
                }
                    
            }
        }
        return dp[0][n-1];
    }
};
