class Solution {
public:
    int minScoreTriangulation(vector<int>& A) {
     if (A.size() < 3) 
          return 0; 
       int table[A.size()][A.size()]; 
       for (int gap = 0; gap < A.size(); gap++) 
       { 
          for (int i = 0, j = gap; j < A.size(); i++, j++) 
          { 
              if (j < i+2) 
                 table[i][j] = 0; 
              else
              { 
                  table[i][j] = INT_MAX; 
                  for (int k = i+1; k < j; k++) 
                  { 
                    int val = table[i][k] + table[k][j] + A[i]*A[j]*A[k]; 
                    if (table[i][j] > val) 
                         table[i][j] = val; 
                  } 
              } 
          } 
       } 
       return  table[0][A.size()-1]; 
    }
};
