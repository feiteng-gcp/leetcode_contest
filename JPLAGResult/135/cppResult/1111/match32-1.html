class Solution {
    int cost(vector<int> &p, int i, int j, int k) {
        return p[i] * p[j] * p[k];
    }
    int mTCDP(vector<int> &A, int n) {

       if (n < 3) 
          return 0; 
       int dp[n][n]; 
       for (int gap = 0; gap < n; gap++) {
          for (int i = 0, j = gap; j < n; i++, j++) {
              if (j < i+2) 
                 dp[i][j] = 0; 
              else{
                  dp[i][j] = INT_MAX; 
                  for (int k = i+1; k < j; k++) { 
                    int val = dp[i][k] + dp[k][j] + cost(A,i,j,k); 
                    if (dp[i][j] > val) 
                         dp[i][j] = val; 
                  } 
              } 
          } 
       } 
       return  dp[0][n-1]; 
    } 
public:
    int minScoreTriangulation(vector<int>& A) {
        return mTCDP(A, A.size());
    }
};
