class Solution {
private:
    int mtc(vector<int>& A, int i, int j){
        if(j < i + 2) return 0;
        int res = INT_MAX;
        for(int k = i + 1; k < j; ++k){
            res = min(res, (mtc(A, i, k) + mtc(A, k,j) + A[i] + A[k] + A[j]));
        }
        return res;
    }
public:
    int minScoreTriangulation(vector<int>& A) {
        int n = A.size();
   int table[n][n]; 
  
   // Fill table using above recursive formula. Note that the table 
   // is filled in diagonal fashion i.e., from diagonal elements to 
   // table[0][n-1] which is the result. 
   for (int gap = 0; gap < n; gap++) 
   { 
      for (int i = 0, j = gap; j < n; i++, j++) 
      { 
          if (j < i+2) 
             table[i][j] = 0; 
          else
          { 
              table[i][j] = INT_MAX; 
              for (int k = i+1; k < j; k++) 
              { 
                int val = table[i][k] + table[k][j] + A[i]*A[k]*A[j]; 
                if (table[i][j] > val) 
                     table[i][j] = val; 
              } 
          } 
      } 
   } 
   return  table[0][n-1]; 
    }
};
