class Solution {
public:
    int n,dp[55][55];
    int cal(int i,int j,vector<int>&a){
        if(dp[i][j]!=-1)return dp[i][j];
        if(j<i+2)
            return dp[i][j]=0;

        int ret=1e9;
        for(int k=i+1;k<j;k++)
            ret=min(ret,cal(i,k,a)+cal(k,j,a)+a[i]*a[k]*a[j]);
        return dp[i][j]=ret;
    }
    int minScoreTriangulation(vector<int>& a) {
        n=(int)a.size();
        for(int i=0;i<=n;i++)
            for(int j=0;j<=n;j++)
                dp[i][j]=-1;
        return cal(0,n-1,a);
    }
};
