class Solution {
public:
    int minScoreTriangulation(vector<int>& A) {
        int N=A.size();
        if (N==3)
            return(A[0]*A[1]*A[2]);
        int table[N][N];
        for(int k=0;k<N;k++){
            for(int i=0, j=k;j<N;i++, j++){
                if (j<i+2)
                    table[i][j]=0;
                else {
                    table[i][j]=INT_MAX;
                    for(int l=i+1; l<j;l++){
                        int val = table[i][l]+table[l][j]+(A[i]*A[j]*A[l]);
                        if (table[i][j]>val)
                            table[i][j]=val;
                    }
                }
            }
        }
        return table[0][N-1];
    }
    
    
};
