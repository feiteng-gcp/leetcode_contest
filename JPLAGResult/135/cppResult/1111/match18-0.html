const int INF = 1e9 + 7;

class Solution {
public:
    int minScoreTriangulation(vector<int>& A) {
        int n = A.size();
        vector<vector<int>> f(n, vector<int>(n, INF));
        for (int i = n - 2; i >= 0; i--) {
            f[i][i + 1] = 0;
            for (int j = i + 2; j < n; j++) {
                for (int k = i + 1; k < j; k++) {
                    f[i][j] = min(f[i][j], f[i][k] + f[k][j] + A[i] * A[j] * A[k]);
                }
            }
        }
        return f[0][n - 1];
    }
};
