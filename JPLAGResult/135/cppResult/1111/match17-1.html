class Solution {
public:

    int minScoreTriangulation(vector<int>& A) {
        int n=A.size();
        vector<vector<int>> dp(n, vector<int>(n,0));
        for(int l=2;l<n;++l){
            for(int i=0, j=l;j<n;++i,++j){

                dp[i][j]=INT_MAX;
                for(int k=i+1;k<j;++k){
                    dp[i][j]=min(dp[i][j], dp[i][k]+dp[k][j]+A[i]*A[j]*A[k]);
                }
            }
        }
        
        // for(int i=0;i<n-2;++i){
        //     for(int j=i+2;j<n;++j){
        //         dp[i][j]=INT_MAX;
        //         for(int k=i+1;k<j;++k){
        //             dp[i][j]=min(dp[i][j], dp[i][k]+dp[k][j]+A[i]*A[j]*A[k]);
        //         }
        //     }
        // }
        return dp[0][n-1];
    }
};
