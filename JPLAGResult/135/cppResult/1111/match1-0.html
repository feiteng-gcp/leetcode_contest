class Solution {
public:
    int minScoreTriangulation(vector<int>& A) {
       int n = A.size();
       int table[n][n];

       for (int gap = 0; gap < n; gap++) {
          for (int i = 0, j = gap; j < n; i++, j++) {
             if (j < i+2)
                table[i][j] = 0.0;
             else {
                table[i][j] = INT_MAX;
                for (int k = i+1; k < j; k++) {
                   int val = table[i][k] + table[k][j] + A[i]*A[j]*A[k];
                   if (table[i][j] > val)
                      table[i][j] = val;
                }
             }
          }
       }  
       return  table[0][n-1];
    }    
    
};
