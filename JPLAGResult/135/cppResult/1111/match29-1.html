class Solution {
public:
    int minScoreTriangulation(vector<int>& A) {
        vector<int> v;
        for(int a:A)v.push_back(a);
        vector<vector<int>> dp(v.size(),vector<int>(v.size(),0));

        int n=v.size();
        for(int interval=2;interval<n;interval++)
        {
            for(int i=0;i<n-interval;i++)
            {
                int j=i+interval;
                dp[i][j]=INT_MAX;
//                cout<<i<<" "<<j<<endl;
                for(int k=i+1;k<j;k++)
                {
                    dp[i][j]=min(dp[i][j],dp[i][k]+dp[k][j]+v[i]*v[j]*v[k]);
                }
            }

        }
        return dp[0][n-1];        
    }
};
