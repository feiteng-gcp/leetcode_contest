class Solution {
public:
    int minScoreTriangulation(vector<int>& A) {
        int dp[100][100];
        memset(dp,0,sizeof(dp));

        int n = A.size();
        for(int len=1;len<n-1;len++)
        {
            for(int i=1;i+len<n;i++)
            {
                int l = i, r = i + len;
                dp[l][r] = 0x3f3f3f3f;

                for(int k=l;k<r;k++)
                {
                    dp[l][r] = min(dp[l][r],dp[l][k]+dp[k+1][r]+A[l-1]*A[k]*A[r]);
                }
            }
        }
        return dp[1][n-1];
    }
};
