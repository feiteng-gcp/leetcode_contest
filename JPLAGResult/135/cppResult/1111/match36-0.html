class Solution {
public:
    int minScore(vector<int> &A, int st, int ed,vector< vector<int> > &dp)
    {
        if(ed-st<2)return 0;
        if(dp[st][ed]!=-1)return dp[st][ed];
        int ans = INT_MAX;
        for(int k=st+1;k<ed;k++)
            ans = min(ans,minScore(A,st,k,dp)+minScore(A,k,ed,dp)+A[k]*A[st]*A[ed]);
        return dp[st][ed]=ans;
    }
    int minScoreTriangulation(vector<int>& A) {
        vector<vector<int > > dp(A.size(),vector<int>(A.size(),-1));
        return minScore(A,0,A.size()-1,dp);
    }
};
