class Solution {
public:
    int minScoreTriangulation(vector<int>& A) {
        int N = (int)A.size();

        vector<vector<int>> dp(N, vector<int>(N));
        for(int g = 0; g < N; g++) {
            for(int i = 0, j = g; j < N; i++, j++) {
                if(j < i+2) {
                    dp[i][j] = 0;
                }
                else {
                    dp[i][j] = INT_MAX;
                    for(int k = i+1; k < j; k++) {
                        int c = dp[i][k] + dp[k][j] + A[i]*A[j]*A[k];
                        dp[i][j] = min(dp[i][j], c);
                    }
                }
            }
        }
        return dp[0][N-1];
    }
};
