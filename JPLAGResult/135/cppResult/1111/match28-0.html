class Solution {
public:
    int f[55][55];
    int minScoreTriangulation(vector<int>& A) {
        int n = A.size(); A.emplace_back(A[0]);
        for (int r = 2 ; r <= n - 1 ; ++ r) {
            for (int i = 1 ; i <= n - r + 1 ; ++ i) {
                int j = i + r - 1;
                //cout << i << ' ' << j << endl;
                f[i][j] = f[i + 1][j] + A[i - 1] * A[i] * A[j];
                for (int k = i + 1 ; k < j ; ++ k) {
                    f[i][j] = min(f[i][j], f[i][k] + f[k + 1][j] + A[i - 1] * A[k] * A[j]);
                }
            }
        }
        return f[1][n - 1];    
    }
};
