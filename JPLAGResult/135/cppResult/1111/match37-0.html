class Solution {
public:
    int minScoreTriangulation(vector<int>& a) {
        int n = a.size();
        vector<vector<int>> dp(n + 1, vector<int>(n + 1, 0));
        for (int i = 0; i < n; ++i) {
            dp[i][i + 1] = 0;
            for (int j = i + 2; j < n; ++j) dp[i][j] = 0x3f3f3f3f;
        }
        for (int i = n - 3; i >= 0; --i) {
            for (int j = i + 2; j < n; ++j) {
                for (int k = i + 1; k < j; ++k) {
                    int c = dp[i][k] + dp[k][j] + a[i] * a[j] * a[k];
                    dp[i][j] = min(dp[i][j], c);
                }
            }
        }
        return dp[0][n - 1];
    }
};
