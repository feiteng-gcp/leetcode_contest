class Solution {
public:
    
int cost(vector<int>& A,int i, int j, int k) 
{ 
    int p1 = A[i], p2 = A[j], p3 = A[k]; 
    return p1*p2*p3;
} 
    
    
    
    int minScoreTriangulation(vector<int>& A) {
        
      int n=A.size();
        
        
    if (n < 3) 
      return 0; 
  
   int table[n][n]; 

   for (int gap = 0; gap < n; gap++) 
   { 
      for (int i = 0, j = gap; j < n; i++, j++) 
      { 
          if (j < i+2) 
             table[i][j] = 0; 
          else
          { 
              table[i][j] = 10000000000; 
              for (int k = i+1; k < j; k++) 
              { 
                int val = table[i][k] + table[k][j] + cost(A,i,j,k); 
                if (table[i][j] > val) 
                     table[i][j] = val; 
              } 
          } 
      } 
   } 
   return  table[0][n-1]; 
        
    }
};
