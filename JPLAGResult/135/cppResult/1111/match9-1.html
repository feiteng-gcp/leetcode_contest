class Solution {
public:

    int calc(vector<int> &a, int n) 
    { 
       if (n < 3) 
          return 0; 
        
       int table[n][n]; 

       // Fill table using above recursive formula. Note that the table 
       // is filled in diagonal fashion i.e., from diagonal elements to 
       // table[0][n-1] which is the result. 
       for (int gap = 0; gap < n; gap++) 
       { 
          for (int i = 0, j = gap; j < n; i++, j++) 
          { 
              if (j < i+2) 
                 table[i][j] = 0; 
              else
              { 
                  table[i][j] = INT_MAX; 
                  for (int k = i+1; k < j; k++) 
                  { 
                    int val = table[i][k] + table[k][j] + a[i]*a[j]*a[k]; 
                    if (table[i][j] > val) 
                         table[i][j] = val; 
                  } 
              } 
          } 
       } 
       return  table[0][n-1]; 
    } 
    int minScoreTriangulation(vector<int>& A) {
        return calc(A, A.size());
    }
};
