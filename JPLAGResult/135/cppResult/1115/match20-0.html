class Solution {
public:
    bool isBoomerang(vector<vector<int>>& points) {
        auto v1 = vec(points[0], points[1]);
        auto v2 = vec(points[1], points[2]);
        int l1 = len2(v1);
        int l2 = len2(v2);
        int d = dot(v1, v2);
        if (l1 == 0 || l2 == 0) {
            return false;
        }
        else {
            return d*d != l1 * l2;
        }
    }
private:
    vector<int> vec(vector<int>& p1, vector<int> p2) {
        return {p1[0] - p2[0], p1[1] - p2[1]};
    }
    int len2(vector<int>& v) {
        return v[0] * v[0] + v[1] * v[1];
    }
    int dot(vector<int>& v1, vector<int>& v2) {
        return v1[0] * v2[0] + v1[1] * v2[1];
    }
};
