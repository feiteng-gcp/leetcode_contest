class Solution {
public:
    bool isBoomerang(vector<vector<int>>& points) {
        set<vector<int>> s;
        s.insert(points[0]);
        s.insert(points[1]);
        s.insert(points[2]);
        if(s.size() < 3)
            return false;
        // cout << "aksdj";
        if((points[0][1] - points[1][1]) * (points[1][0] - points[2][0]) == (points[0][0] - points[1][0]) * (points[1][1] - points[2][1]))
            return false;
        return true;
    }
};
