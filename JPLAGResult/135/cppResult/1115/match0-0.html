class Solution {
public:
    bool isBoomerang(vector<vector<int>>& points) {
        int r1[2], r2[2];
        
        r1[0]= points[0][0] - points[2][0];
        r1[1]= points[0][1] - points[2][1];
        
        r2[0]= points[1][0] - points[2][0];
        r2[1]= points[1][1] - points[2][1];        
        
        
        return  r2[1]*r1[0] - r2[0]*r1[1] != 0; 
        
    }
};
