class Solution {
public:
    bool isBoomerang(vector<vector<int>>& points) {
		if(points.size() != 3) return false;
		//if(points[0] == points[1] || points[1] == points[2] || points[0] == points[2]) return false;
		int x = points[2][0] - points[0][0];
		int y = points[2][1] - points[0][1];
		int x1 = points[1][0] - points[0][0];
		int y1 = points[1][1] - points[0][1];
		return x*y1 - x1*y != 0; 
    }
};
