class Solution {
public:
    bool isBoomerang(vector<vector<int>>& P) {
        int k1_up = P[1][1] - P[0][1];
        int k1_down = P[1][0] - P[0][0];
        int k2_up = P[2][1] - P[1][1];
        int k2_down = P[2][0] - P[1][0];
        if(!k1_up && !k1_down)
            return false;
        if(!k2_up && !k2_down)
            return false;
        return k1_up * k2_down != k1_down * k2_up;
    }
};
