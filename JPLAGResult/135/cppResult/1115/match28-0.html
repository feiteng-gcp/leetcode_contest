class Solution {
public:
    bool isBoomerang(vector<vector<int>>& p) {
        for(int i=0;i<3;i++){
            for(int j=i+1;j<3;j++){
                if(p[i][0]==p[j][0] && p[i][1]==p[j][1])return 0;
            }
        }
        int a=p[1][0] - p[0][0], b = p[1][1] - p[0][1];
        int x=p[2][0] - p[0][0], y = p[2][1] - p[0][1];
        return b*x!=a*y;
    }
};
