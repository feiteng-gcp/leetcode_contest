class Solution {
public:
    bool isBoomerang(vector<vector<int>>& points) {
        set<vector<int>> p(points.begin(), points.end());
        if(p.size() < 3) return false;
        if((points[1][0] - points[0][0])*(points[2][1] - points[0][1]) == (points[1][1] - points[0][1])*(points[2][0] - points[0][0]))
            return false;
        return true;
    }
};
