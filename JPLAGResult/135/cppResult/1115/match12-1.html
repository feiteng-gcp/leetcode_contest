class Solution {
public:
    bool isBoomerang(vector<vector<int>>& points) {
        sort(points.begin(), points.end());
        int vy01 = points[1][1] - points[0][1];
        int vx01 = points[1][0] - points[0][0];
        int vy12 = points[2][1] - points[1][1];
        int vx12 = points[2][0] - points[1][0];
        if(points[0] == points[1] or points[1] == points[2] or points[0] == points[2]) return false;
        if(vy01 * vx12 == vy12 * vx01) return false;
        return true;
    }
};
