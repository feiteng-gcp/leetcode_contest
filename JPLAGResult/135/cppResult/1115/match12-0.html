class Solution {
public:
    bool isBoomerang(vector<vector<int>>& points) {
        int a = points[1][0] - points[0][0];
        int b = points[1][1] - points[0][1];
        int c = points[2][0] - points[0][0];
        int d = points[2][1] - points[0][1];
        if (d * a - b * c == 0) return false;
        else return true;
    }
};
