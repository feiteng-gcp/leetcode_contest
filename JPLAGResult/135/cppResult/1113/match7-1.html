class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
        sort(stones.begin(), stones.end());
        int len = stones.size();
        int maximum = stones[len-1]-stones[0]+1-len-min(stones[1]-stones[0], stones[len-1]-stones[len-2])+1;
        int minimum = maximum, occupied = 0, last = 0, start = 0;
        while(last < len){
            while(last < len && stones[last]-stones[start]+1 <= len){
                occupied++;
                last++;
            }
            if(!(len-occupied == 1) || !(stones[last-1]-stones[start]+1 != len))
                minimum = min(minimum, len-occupied);
            start++;
            occupied--;
        }
        return vector<int>({minimum, maximum});
    }
};
