class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
    	sort(all(stones));
    	int n = stones.size();
    	int minimum_moves = INT_MAX;
		int maximum_moves = stones.back() - stones.front() + 2 - n 
			- min(stones[1] - stones[0], stones.back() - stones[n - 2]);
		for (int l = 0, r = 0; l < n; ++l) {
			while (r < n && stones[r] <= stones[l] + n - 1)
				++r;
			if (stones[r - 1] == stones[l] + n - 1
				|| n - r >= 2
				|| (l > 0 && n - r > 0)) {
				minimum_moves = min(minimum_moves, n - (r - l));
			} else minimum_moves = min(minimum_moves, n - (r - l) + 1); 
		}
		return vector<int>({minimum_moves, maximum_moves});
	}
};
