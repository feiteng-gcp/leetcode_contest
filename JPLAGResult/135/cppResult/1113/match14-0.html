class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
        sort(stones.begin(), stones.end());
        int n = stones.size();
        int m = stones.back() - stones.front() + 1;
        int max_moves = m - n;
        max_moves -= min(stones[1] - stones[0] - 1, stones[n - 1] - stones[n - 2] - 1);
        
        int max_coverage = 0;
        int right = 0;
        for (int left = 0; left < n - 1; left++) {
            while (right < n && stones[right] < stones[left] + n)
                right++;
            if (left == 0 && stones[right - 1] < stones[left] + n - 1)
                continue;
            
            if (right == n) {
                if (stones[right - 1] == stones[left] + n - 1 || left >= 2) {
                    max_coverage = max(max_coverage, right - left);
                }
                break;
            }
            max_coverage = max(max_coverage, right - left);
        }
        int min_moves = n - max_coverage;
        return { min_moves, max_moves };
    }
};
