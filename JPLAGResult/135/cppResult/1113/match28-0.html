class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& A) {
        int n = A.size();
        sort(A.begin(), A.end());
        
        int best = 0;
        int j = 0;
        for (int i = 0; i < n; i++) {
            while (A[i] - A[j] >= n) {
                j++;
            }
            int sz = i - j + 1;
            if (sz > 1 && sz < n && A[i] - A[j] == i - j) {
                sz--;
            }
            best = max(best, sz);
        }
        
        vector<int> B;
        for (int i = 1; i < n; i++) {
            B.push_back(A[i] - A[i - 1] - 1);
        }
        
        int sum = 0;
        for (auto &x : B) {
            sum += x;
        }
        
        return {n - best, sum - min(B.front(), B.back())};
    }
};
