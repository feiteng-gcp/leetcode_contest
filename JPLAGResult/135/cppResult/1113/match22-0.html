class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
        sort(stones.begin(),stones.end());
        int n=stones.size();
        int f=1;
        for(int i=1;i<n;i++)
            if(stones[i]!=stones[i-1]+1)
            {
                f=0;
                break;
            }
        if(f) return {0,0};
        int ma=stones[n-1]-stones[0]+1-n;
        int fi=min(stones[n-1]-stones[n-2]-1,stones[1]-stones[0]-1);
        ma-=fi;
        //cout<<ma<<" "<<mi<<endl;
        queue<int> qu;
        int inq=0;
        int ans=0;
        for(int i=0;i<n;i++)
        {
            while(!qu.empty()&&stones[i]-qu.front()>=n)
                qu.pop(),inq--;
            qu.push(stones[i]); inq++;
            if(inq!=n-1) ans=max(ans,inq);
            else
            {
                if(stones[i]-qu.front()+1==n) ans=max(ans,inq);
                else ans=max(ans,inq-1);
            }
        }
        int mi=n-ans;
        return {mi,ma};
    }
};
