class Solution {
public:

    vector<int> numMovesStonesII(vector<int>& a) {
        int n = a.size();
        sort(a.begin(), a.end());
        int hi = a[n-1] - a[0] - n + 2 - min(a[1] - a[0], a[n-1] - a[n-2]), lo = n;
        for (int i = 0, j = 0; i < n; ++i) {
            while (a[i] - a[j] >= n) ++j;
            lo = min(lo, n - (i - j + 1));
        }
        if (a[n-2] == a[0]+n-2 || a[1]+n-2 == a[n-1])
            lo++;
        if (lo > hi)
            lo--;
        return {lo, hi};
    }
};
