class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& a) {
        sort(a.begin(), a.end());
        int n = a.size();
        int x1 = 1e9;
        for (int i = 0, r = 0; a[n - 1] - a[i] + 1 >= n; i++) {
            for (; r < a.size() && a[r] - a[i] + 1 <= n; r++);
            bool ok = (i > 0 && r < n);
            ok |= (i == 0 && a[r - 1] - a[i] + 1 == n);
            ok |= (r == n && a[r - 1] - a[i] + 1 == n);
            if (ok) {
                x1 = min(x1, n - (r - i));
            } 
        }
        int x2 = max(a[n - 2] - a[0], a[n - 1] - a[1]) - n + 2;
        return {x1, x2};
    }
};
