class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
        sort(stones.begin(),stones.end());
        long long sum=0;
        int count=0;
        int n=stones.size();
        for(int i=0;i<stones.size()-1;i++){
            sum+=stones[i+1]-stones[i]-1;
        }
        //cout<<sum<<endl;
        int ma = sum - min(stones[n-1]-stones[n-2]-1, stones[1]-stones[0]-1);
        //cout<<ma<<endl;
        int mi=10000;
        int st=0 , end=0;
        set<int> senu;
        for(st=0;st<n;st++){
            senu.insert(stones[st]);
            while(end<n && stones[end]-stones[st]<n){senu.insert(stones[end]);end++;}
            if(senu.find(stones[st]+n-1) == senu.end() && (end>=n-1)){mi=min(mi,n-end+1+st);}
            else mi = min(mi, n-end+st);
            
            senu.erase(stones[st]);
        }
        return vector<int>{mi, ma};
    }
};
