class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
        int n = stones.size();
        sort(stones.begin(), stones.end());
        vector<int> a = stones;
        int spaces = (a[n-1]-a[0]) - (n-1);
        if(spaces == 0){
            return vector<int>(2,0);
        }
        int mx = spaces - min(a[1]-a[0]-1, a[n-1]-a[n-2]-1);
        int mn = 0;
        if(spaces == 1){
            mn = 1;
        }else if(a[1]-a[0]-1 == spaces || a[n-1]-a[n-2]-1 == spaces){
            mn = 2;
        }else{
            mn = n;
            int it = 0;
            for(int i=0; i<n; i++){
                while(it<n && a[it]<=a[i]+(n-1))it++;
                mn = min(mn, n-(it-i));
            }
        }
        return vector<int>({mn,mx});
    }
};
