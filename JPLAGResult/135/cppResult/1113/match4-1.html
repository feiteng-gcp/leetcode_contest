class Solution {
public:
	vector<int> g;
	deque<int> s;
	vector<int> numMovesStonesII(vector<int>& vv) {
		g = vv;
		sort(g.begin(), g.end());
		if (g.back() - g[0] == g.size() - 1) {
			vector<int> ret;
			ret.push_back(0);
			ret.push_back(0);
			return ret;
		}
		int rng = -1;
		if (g.back() - g[1] != g.size() - 2) {
			rng = g.back() - g[1];
		}
		if (g[g.size() - 2] - g[0] != g.size() - 2) {
			rng = max(rng, g[g.size() - 2] - g[0]);
		}
		//max(g.back() - g[1], g[g.size() - 2] - g[0]);
		rng--;
		rng -= ((int)vv.size() - 3);

		int z = 0;
		int n = vv.size();
		int mini = n;
		for (int i = 0; i < g.size(); i++) {
			int rig = g[i] + n - 1;
			while (z < g.size() && g[z] <= rig) {
				z++;
			}
			int res = n - (z - i);
			if (z - i == n - 1 && g[z - 1] != rig) {
				continue;
			}
			mini = min(mini, n - (z - i));
		}
		z = n - 1;
		for (int i = g.size() - 1; i >= 0; i--) {
			int lef = g[i] - n + 1;
			while (z >= 0 && g[z] >= lef) {
				z--;
			}
			if (i - z == n - 1 && g[z + 1] != lef) {
				continue;
			}
			mini = min(mini, n - (i - z));
		}
		vector<int> r;
		r.push_back(mini);
		r.push_back(rng);
		return r;
	}
} ;
