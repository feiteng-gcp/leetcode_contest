typedef pair<int, int> ii;


class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& a) {
        int n = a.size();
        sort(a.begin(), a.end());
        vector<int> ret(2);
        ret[0] = 1 << 30;
        //cout << "---" << endl;
        for (int i = 0; i < n; ++i) {
            int high = a[i];
            int low = high - n + 1;
            int k = lower_bound(a.begin(), a.end(), low) - a.begin();
            if (a[k] != low && k == 0) continue;
            if (a[k] != low && k == 1 && i == n - 1) continue;
            //cout << low << " " << high << " " << i << " " << k << " " << n - (i - k + 1) << endl;
            ret[0] = min(ret[0], n - (i - k + 1));
        }
        for (int i = 0; i < n; ++i) {
            int low = a[i];
            int high = low + n - 1;
            int k = upper_bound(a.begin(), a.end(), high) - a.begin() - 1;
            if (a[k] != high && k == n - 1) continue;
            if (a[k] != high && k + 1 == n - 1 && i == 0) continue;
            //cout << low << " " << high << " " << i << " " << k << " " << n - (k - i + 1) << endl;
            ret[0] = min(ret[0], n - (k - i + 1));
        }
        vector<ii> A;
        for (int i = 0, j; i < n; i = j) {
            for (j = i + 1; j < n && a[j] - a[i] == 1; ++j);
            A.push_back({a[i], a[j - 1]});
        }
        n = A.size();
        cout << n << endl;
        int sum = 0;
        for (int i = 1; i < n; ++i) {
            sum += A[i].first - A[i - 1].second - 1;
        }
        if (A[0].second - A[0].first > 0 || A[n - 1].second - A[n - 1].first > 0) {   
            ret[1] = max(ret[1], sum);
        } else {
            sum -= min(A[1].first - A[0].second - 1, A[n - 1].first - A[n - 2].second - 1);
            ret[1] = max(ret[1], sum);
        }
        return ret;
    }
};
