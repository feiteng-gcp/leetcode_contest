class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
        vector<int> s = stones;
        int n=s.size();
        sort(s.begin(), s.end());
        int mn = 1e9+1, mx = 0;
        for(int i=0;i<n;i++){
            int k = upper_bound(s.begin()+i, s.end(), s[i]+s.size()-1) - (s.begin() + i);
            mn = min(mn, n-k);
            if(mn==0)break;
            if(mn==1){
                int flag=1;
                for(int k=0;k<n-2;k++){
                    if(s[i+k] == s[i+k+1]-1);
                    else flag=0;
                }
                if(flag){
                    mn = 2;
                }
                break;
            }
        }
        mx = s[s.size()-1]-s[0]+1-s.size()-min(s[1]-s[0]-1, s[s.size()-1]-s[s.size()-2]-1);
        mn = min(mn, mx);
        vector<int> ans;
        ans.push_back(mn);
        ans.push_back(mx);
        return ans;
    }
};
