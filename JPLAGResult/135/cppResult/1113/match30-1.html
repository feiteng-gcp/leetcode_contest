int getMin(VI x) {
    int n = x.size();
    int next = 0;
    int best = 0;
    bool spe = false;
    for (int i = 0; i < n; ++i) {
        while (next < n && x[next] <= x[i] + n - 1) ++next;
        best = max(best, next - i);
        if (next - i == n - 1) {
            if (x[next - 1] - x[i] == n - 1) {
                spe = true;
            }
        }
    }
    if (best == n) return 0;
    if (best == n - 1) {
        return spe ? 1 : 2;
    }
    return n - best;
}

int go(VI x) {
    SORT(x);
    int n = x.size();
    return x[n - 1] - x[1] + 1 - (n - 1);
}

int getMax(VI x) {
    int ret = go(x);
    TR(it, x) *it = -*it;
    ret = max(ret, go(x));
    return ret;
}

class Solution {
public:
    vector<int> numMovesStonesII(vector<int>& stones) {
        SORT(stones);
        VI ret;
        ret.PB(getMin(stones));
        ret.PB(getMax(stones));
        return ret;
    }
};
