class Solution(object):
    def minScoreTriangulation(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        from functools import lru_cache
        map_={i:val for i,val in enumerate(A)}
        
        def cost(i,j,k):
            return map_[i]*map_[j]*map_[k]
        
@lru_cache(None)
        def helper(i,j):
            if j<i+2:
                return 0
            res = float('inf')
            for k in range(i+1,j):
                res=min(res, helper(i,k)+ helper(k,j)+ cost(i,k,j))
            return res
        
        return helper(0, len(A)-1)
