class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        MAX = 10 ** 9
        n = len(A)
        table = [[0] * n for _ in range(n)]
        for gap in range(n): 
            for i, j in enumerate(range(gap, n)):
                if j < i + 2: table[i][j] = 0
                else: 
                    table[i][j] = MAX
                    for k in range(i + 1, j): 
                        table[i][j] = min(table[i][j], table[i][k] + table[k][j] + A[i] * A[j] * A[k])
                        
        return table[0][n - 1]
