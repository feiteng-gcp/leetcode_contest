class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        def dfs(i,j,cache):
            if j-i < 2: return 0
            if (i,j) in cache: return cache[(i,j)]
            ans = math.inf
            for k in range(i+1,j):
                ans = min(ans,dfs(i,k,cache)+ dfs(k,j,cache)+ A[i]*A[j]*A[k])
            cache[(i,j)] = ans
            return ans
        ans = dfs(0,len(A)-1,{})
        return ans if ans < math.inf else 0
