class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        n = len(A)
        a = A + A
        f = [[0] * (n + n) for i in range(n + n)]
        for i in range(n + n, -1, -1):
            for j in range(i + 2, n + n):
                f[i][j] = math.inf
                for k in range(i + 1, j):
                    f[i][j] = min(f[i][j], f[i][k] + f[k][j] + a[i] * a[j] * a[k])
        return f[0][n - 1]
