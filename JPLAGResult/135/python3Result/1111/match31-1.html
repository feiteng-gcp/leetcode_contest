class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        n = len(A)
        dp = [[None for _ in range(n)] for _ in range(n)]
        
        def cost(i, j, n, dp):
            if j <= i+1:
                return 0
            elif j == i+2:
                if dp[i][j] != None:
                    return dp[i][j]
                else:
                    dp[i][j] = A[i] * A[i+1] * A[i+2]
                    return dp[i][j]
            else:
                if dp[i][j] != None:
                    return dp[i][j]
                else:
                    c = float('inf')
                    
for k in range(i+1, j):
                        c = min(c, cost(i, k, n, dp) + cost(k, j, n, dp) + A[i] * A[k] * A[j])
                    dp[i][j] = c
                    return c
            
        return cost(0, n-1, n, dp)
