class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        if len(A) == 3:
            return A[0]*A[1]*A[2]
        from functools import lru_cache
        @lru_cache(None)
        def f(i, j):
            if j < i+2:
                return 0
            res 
= float('inf')
            for k in range(i+1, j):
                res = min(res, f(i, k)+f(k, j)+A[i]*A[j]*A[k])
            return res    
        return f(0, len(A)-1)
