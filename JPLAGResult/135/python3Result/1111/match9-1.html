class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        n = len(A)
        dp = [[0]*(n+1) for i in range(n+1)]
        for i in range(3, n+1):
            for j in range(0, n - i + 1):
                if i == 3:
                    dp[j][j+i]=A[j]*A[j+1]*A
[j+2]
                else:
                    for d in range(j+1,j+i-1):
                        #print(A[j]*A[j+i-1]*A[d] + dp[j][d+1] + dp[d][j+i])
                        dp[j][j+i] = min(100000000 if dp[j][j+i] == 0 else dp[j][j+i],A[j]*A[j+i-1]*A[d] + dp[j][d+1] + dp[d][j+i])
        #print(dp)
        return dp[0][n]
