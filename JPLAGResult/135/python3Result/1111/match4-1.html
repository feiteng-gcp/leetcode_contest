class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        n = len(A)
        dp = [[float('inf')] * (n + 2) for _ in range(n + 1)]
        for l in range(n):
            for j in range(l, n):
                i = j - l
                if l < 2:
                    dp[i][j] = 0
                else:
                    m = dp[i][j]
                    
for k in range(i + 1, j):
                        m = min(m, dp[i][k] + dp[k][j] + A[i] * A[j] * A[k])
                    dp[i][j] = m
        return dp[0][n - 1]
