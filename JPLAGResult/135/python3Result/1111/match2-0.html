from functools import lru_cache
class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        self.A = A
        return self.helper(0, len(A) - 1)
    @lru_cache(None)
    def helper(self, i, j):
        res = float("inf")
        if j < i + 2:
            return 0
        k = i + 1
        while k < j:
            res = min(res, self.helper(i, k) + self.helper(k, j) + self.A[i] * self.A[j] * self.A[k])
            k += 1
        return res
