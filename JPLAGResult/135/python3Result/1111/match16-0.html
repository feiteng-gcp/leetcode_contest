class Solution:
    def minScoreTriangulation(self, nums: List[int]) -> int:
        dp = [[float('inf') for _ in range(len(nums))] for _ in range(len(nums))]
        for gap in range(len(nums)):
            for i in range(len(nums)):
                for j in range(gap, len(nums)):
                    if j < i + 2:
                        dp[i][j] = 0
                    else:
                        for k in range(i + 1, j):
                            val = dp[i][k] + dp[k][j] + nums[i]*nums[j]*nums[k]
                            if val < dp[i][j]:
                                dp[i][j] = val
        return dp[0][-1]
