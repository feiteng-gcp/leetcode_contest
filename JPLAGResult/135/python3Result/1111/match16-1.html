class Solution(object):
    def minScoreTriangulation(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        return self.mST(A, len(A))
        
    def mST(self, A, n) :
        if n < 3: 
            return 0
        dp = [[0 for _ in range(n+1)] for _ in range(n+1)]

        for gap in range(n):
            for i in range(n-gap):
                j = i+gap
                if (j-i)<2:
                    dp[i][j] = 0
                else:
                    dp
[i][j] = 999999999
                    for k in range(i+1, j):
                        val = dp[i][k] + dp[k][j] + A[i] * A[j] * A[k]
                        if dp[i][j] > val: 
                            dp[i][j] = val
        return dp[0][n-1]
