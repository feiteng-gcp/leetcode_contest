class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        n = len(A)
        if n == 3:
            return A[0]*A[1]*A[2]
        dp 
= [[0]*n for _ in range(n)]
        for x in range(n-3, -1, -1):
            for y in range(x+2, n):
                if y-x == 2:
                    dp[x][y] = A[x]*A[x+1]*A[y]
                else:
                    temp = 1<<31
                    for k in range(x+1, y):
                        cur = A[x]*A[y]*A[k] + dp[x][k] + dp[k][y]
                        temp = min(
temp ,cur)
                    dp[x][y] = temp
        return dp[0][n-1]
