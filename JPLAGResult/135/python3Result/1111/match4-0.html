class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        n = len(A)
        opt = [[0] * n for _ in range(n)]
        for j in range(2,n):
            for i in range(n-j):
                if j == 2:
                    opt[i][i+j] = A[i] * A[i+1] * A[i+2]
                else:
                    mins = 1000000000
                    for k in range(i+1, i+j):
                        mins = min(mins, opt[i][k] + opt[k][i+j] + A[i]*A[k]*A[i+j])
                    opt[i][i+j] = mins
        return opt[0][n-1]
