class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        self.memo = {}
        def search(i,j):
            if j < i+2:
                self.memo[(i,j)] = 0
            if (i,j) not in self.memo:
                tmp = float('inf')
                for k in range(i+1,j):
                    tmp = min(tmp,search(i,k)+search(k,j)+A[i]*A[j]*A[k])
                self.memo[(i,j)] = tmp
            return self.memo[(i,j)]
        return search(0,len(A)-1)
