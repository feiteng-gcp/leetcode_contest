class Solution:
    def cost(self, A, i, j, k):
        return A[i]*A[j]*A[k]
    def subScore(self, A, i, j):
        if j < i + 2 :return 0
        res = float('inf')
        for k in range(i+1, j):
            res = min(res, self.subScore(A, i, k) + self.subScore(A, k, j) + self.cost(A, i, j, k))
        return res
        
    def minScoreTriangulation(self, A: List[int]) -> int:
        if not A or len(A) < 3: return 0
        n = len(A)
        dp = [[0 for _ in range(n)] for i in range(n)]
        for gap in range(n):
            i, j = 0, gap
            while j < n:
                if j < i + 2 : 
                    dp[i][j] = 0
                else:
                    dp[i][j] = float('inf')
                    for k in range(i+1, j):
                        val = dp[i][k] + dp[k][j] + self.cost(A, i, j, k)
                        dp[i][j] = min(dp[i][j], val)
                i += 1
                j += 1
        return dp[0][n-1]
