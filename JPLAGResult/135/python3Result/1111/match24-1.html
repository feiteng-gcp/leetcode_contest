class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        import math
        class point:
            def __init__(self, x, y):
                self.x = x
                self.y = y
            
        def dist(p1, p2):
            return math.sqrt((p1.x - p2.x)*(p1.x - p2.x) + 
                (p1.y - p2.y)*(p1.y - p2.y))

        def cost(points, i, j, k):
            p1,p2,p3 = points[i],points[j],points[k]
            return (dist(p1, p2)*dist(p2, p3)*dist(p3, p1))
        
        def sol(points, n):
            if n < 3:return 0
            table 
=  [ [ 0 for i in range(n) ] for j in range(n)]
            for gap in range(0, n):
                i, j = 0, gap
                while(j < n):
                    if j < i+2:
                        table[i][j] = 0.0
                    else:
                        table[i][j] = float('inf')
                        for k in range(i+1, j):
                            val = table[i][k] + table[k][j] + A[i]*A[j]*A[k]
                            table
[i][j] = min(table[i][j], val)
                    i += 1
                    j += 1
            return int(table[0][n-1])
        return (sol(A, len(A)))
