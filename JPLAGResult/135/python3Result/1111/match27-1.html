class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        ans = 0
        a = A[-1:] + A + A[:1]
        f = [[0] * len(a) for i in range(len(a))]
        print(a)
        for i in range(2, len(A)):
            for j in range(len(a) - i):
                for k in range(j+1, j+i):
                    value 
= f[j][k] + f[k][i+j] + a[j] * a[k] * a[i+j]
                    if f[j][i+j] == 0 or f[j][i+j] > value:
                        f[j][i+j] = value
                        
        
                print(j, i+j, f[j][i+j])
                if i == len(A) - 1:
                    if ans == 0 or f[j][i+j] < ans:
                        ans = f[j][i+j]
                        
        return ans
