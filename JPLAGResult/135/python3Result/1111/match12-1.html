from functools import lru_cache, reduce
from operator import mul
class Solution:
    def minScoreTriangulation(self, A: List[int]) -> int:
        #product = lambda *args: reduce(mul, tuple(args), 1)
        def cost(*args):
            return reduce(mul, map(lambda arg: A[arg], args), 1)
        @lru_cache(None)
        def min_cost(i=0, j=len(A)-1):
            
if j - i < 2:
                return 0
            ans = float('inf')
            for k in range(i+1, j):
                ans = min(ans, min_cost(i, k) + min_cost(k, j) + cost(i, j, k))
            return ans
        return min_cost()
