# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        
        def getval(node): 
            if node.left is not None: return getval(node.left)
            return node.val
        
        def update(node, rightsum):
            if node.right is not None: 
                update(node.right, rightsum)
                rightsum = getval
(node.right)
            node.val += rightsum
            rightsum = node.val
            if node.left is not None: 
                update(node.left, rightsum)
        
        update(root, 0)
        return root
