# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        def h(curr, greaterSum):
            if curr is None:
                return 0
            tmp = curr.val
            r = h(curr.right, greaterSum)
            curr.val += r + greaterSum
            l = h(curr.left, curr.val)
            
            return r + l + tmp
        h(root, 0)
        return root
