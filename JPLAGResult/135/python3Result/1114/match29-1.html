# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        if root is None:
            return root
        
        curr_sum = 0
        
        q = []
        p = root
        
        while p or q:
            if p:
                q.append(p)
                p = p.right
            else:
                p = q.pop()
                curr_sum += p.val
                p.val = curr_sum
                p = p.left
        return root
