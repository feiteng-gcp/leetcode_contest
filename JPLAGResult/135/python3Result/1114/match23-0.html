# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def transformUtil(self, root, cur_sum):
        if not root: return
        self.transformUtil(root.right, cur_sum)
        cur_sum[0] += root.val
        root.val = cur_sum[0]
        self.transformUtil(root.left, cur_sum)
            
    def bstToGst(self, root: TreeNode) -> TreeNode:
        self.transformUtil(root, [0])
        return root
