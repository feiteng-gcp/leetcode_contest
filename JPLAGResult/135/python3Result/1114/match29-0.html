# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        # 二叉树的中序遍历
        if not root:
            return []
        temp = root
        stack = []
        res = []
        while stack or root:
            if root is not None:
                stack.append(root)
                root = root.left
            else:
                this_node = stack.pop()
                res.append(this_node)
                root = this_node.right
        sum_v = 0
        for each_node in res[::-1]:
            sum_v += each_node.val
            each_node.val = sum_v
        return temp
