# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        
        
def w(p):
            nonlocal carry
            if not p:
                return
            w(p.right)
            carry += p.val
            p.val = carry
            w(p.left)
        
        carry = 0
        w(root)
        return root
