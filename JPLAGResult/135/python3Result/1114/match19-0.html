# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        self.sum = 0
        def rlm(node):
            if not node:
                return
            rlm(node.right)
            self.sum += node.val
            node.val = self.sum
            rlm(node.left)
            return node
        if not root:
            return []
        return rlm(root)
