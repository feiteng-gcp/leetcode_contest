# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        self.now = 0
        def dfs(root):
            if not root:
                return
            dfs(root.right)
            temp = self.now
            self
.now += root.val
            root.val += temp
            dfs(root.left)
        dfs(root)
        return root
