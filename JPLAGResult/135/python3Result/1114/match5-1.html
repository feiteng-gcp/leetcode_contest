# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        def helper(root, parent):
            if root is None:
                return 0
            res = root.val
            rightsum = helper(root.right, parent)
            root.val += parent+rightsum
            leftsum = helper(root.left, root.val)
            return leftsum + rightsum + res
            
        helper(root,0)
        return root
