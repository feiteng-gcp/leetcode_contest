# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        s = [ 0 ]

        def helper(root, s):
            if not root: return
            helper(root.right, s)
            s[0] += root.val
            root.val = s[0]
            helper(root.left, s)

        helper(root, s)
        return root
