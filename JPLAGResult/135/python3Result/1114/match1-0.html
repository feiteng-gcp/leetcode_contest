# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        s = 0
        def dfs(u):
            nonlocal s
            if not u:
                return
            dfs(u.right)
            s += u.val
            u.val = s
            dfs(u.left)
        dfs(root)
        return root
