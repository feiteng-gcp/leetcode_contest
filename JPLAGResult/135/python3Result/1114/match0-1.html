# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def bstToGst(self, root: TreeNode) -> TreeNode:
        cur = 0
        
        def dfs(node):
            nonlocal cur
            
            if not node: return
            dfs(node.right)
            cur += node.val
            node.val = cur
            dfs(node.left)
        
        dfs(root) 
        return root
