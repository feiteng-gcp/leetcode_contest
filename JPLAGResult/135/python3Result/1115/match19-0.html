from typing import List, Tuple, Dict, TextIO

class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        st = set()
        p = points
        for pt in p:
            st.add((pt[0], pt[1]))
        if len(st) < 3: return False
        return (p[0][1] - p[1][1]) * (p[1][0] - p[2][0]) != (p[0][0] - p[1][0]) * (p[1][1] - p[2][1])
