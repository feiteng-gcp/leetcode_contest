class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        points.sort(key = lambda x: (x[0], x[1]))
        if points[0][0] == points[1][0] and points[1][0] == points[2][0]:
            return False
        if points[0][1] == points[1][1] and points[1][1] == points[2][1]:
            return False
        difx1 
= points[1][0] - points[0][0]
        difx2 = points[2][0] - points[1][0]
        
        dify1 = points[1][1] - points[0][1]
        dify2 = points[2][1] - points[1][1]
        
        return difx1 * dify2 != difx2 * dify1
