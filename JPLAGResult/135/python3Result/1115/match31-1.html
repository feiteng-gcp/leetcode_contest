class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        if points[0] == points[1] or points[1] == points[2] or points[2] == points[0]:
            return False
        points.sort()
        rise1 
= points[1][1] - points[0][1]
        run1 = points[1][0] - points[0][0]
        
        rise2 = points[2][1] - points[0][1]
        run2 = points[2][0] - points[0][0]
        
        if run2 == 0 and run1 == 0:
            return False
        if run2 == 0 or run1 == 0:
            return True
        
        m1 = rise1/run1
        m2 = rise2/run2
        if -1e-4 <= m1 - m2 <= 1e-4:
            return False
        return True
