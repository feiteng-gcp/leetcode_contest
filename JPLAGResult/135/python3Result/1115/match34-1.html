class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        ft = set()
        for i in range(len(points)):
            ft.add((points[i][0], points[i][1]))
        if len(ft)!=len(points):
            return False
        if points[0][1]-points[1][1]==0:
            return points[2][1]-points[1][1]!=0
        if points[0][0]-points[1][0]==0:
            return points[2][0]-points[1][0]!=0
        if points[2][1]-points[1][1]==0:
            return points[0][1]-points[1][1]!=0
        if points[2][0]-points[1][0]==0: 
            return points[1][0]-points[0][0]!=0
        
        
if (points[0][1]-points[1][1])/(points[0][0]-points[1][0])!=(points[2][1]-points[1][1])/(points[2][0]-points[1][0]):
            return True
        else:
            return False
