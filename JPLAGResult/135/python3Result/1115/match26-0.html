class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        points.sort()
        if (points[0][0]==points[1][0] and points[0][1]==points[1][1]  ) or  (points[2][0]==points[1][0] and points[2][1]==points[1][1]  ):
            return False
        a = points[1][0]-points[0][0]
        b = points[1][1]-points[0][1]
        c = points[2][0]-points[0][0]
        d = points[2][1]-points[0][1]
        if a*d==b*c:
            return False
        else:
            return True
