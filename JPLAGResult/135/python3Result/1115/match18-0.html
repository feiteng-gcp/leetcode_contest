class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        points = list(map(tuple, points))
        if len(set(points)) < 3: return False
        a, b, c = points
        return not (a[0] == b[0] == c[0] or (a[0] != b[0] and b[0] != c[0] and (a[1]-b[1])/(a[0]-b[0]) == (b[1]-c[1])/(b[0]-c[0])))
