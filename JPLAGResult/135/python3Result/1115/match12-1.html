class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        dx12 
= points[0][0] - points[1][0]
        dy12 = points[0][1] - points[1][1]
        dx23 = points[1][0] - points[2][0]
        dy23 = points[1][1] - points[2][1]
        return dx12 * dy23 != dy12 * dx23
