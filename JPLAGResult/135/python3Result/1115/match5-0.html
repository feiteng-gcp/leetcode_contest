class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        p1, p2, p3 = points
        if p1==p2 or p2==p3 or p1==p3:
            return False
        if p1[0] == p2[0] and p2[0] == p3[0]:
            return False
        if (p1[0]-p2[0])*(p1[0]-p3[0]) == 0:
            return True
        if (p1[1]-p2[1])*(p2[0]-p3[0]) == (p2[1]-p3[1])*(p1[0]-p2[0]):
            return False
        return True
