class Solution {
    
    private static final double ERR = 1e-6;
    public boolean isBoomerang(int[][] points) {
        if (same(points[0], points[1]) || same(points[1], points[2]) || same(points[0], points[2])) {
            return false;
        }
        
        double s1 = slope(points[0], points[1]);
        double s2 = slope(points[1], points[2]);
        
        return Math.abs(s1 - s2) > ERR;
    }
    
    private double slope(int[] p1, int[] p2) {
        return ((double) (p2[1] - p1[1])) / (p2[0] - p1[0]);
    }
    
    private boolean same(int[] p1, int[] p2) {
        return p1[0] == p2[0] && p1[1] == p2[1];
    }
}
