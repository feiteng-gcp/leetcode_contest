class Solution {
	boolean same(int[] p1, int[] p2) {
		return p1[0] == p2[0] &&  p1[1] == p2[1];
	}
	public boolean isBoomerang(int[][] points) {
		if (same(points[0], points[1]) || same(points[0], points[2]) ||
				same(points[1], points[2])) return false;
		int x1 = points[1][0] - points[0][0];
		int y1 = points[1][1] - points[0][1];
		int x2 = points[2][0] - points[0][0];
		int y2 = points[2][1] - points[0][1];
		
		return y1 * x2 != y2 * x1;
	}
}
