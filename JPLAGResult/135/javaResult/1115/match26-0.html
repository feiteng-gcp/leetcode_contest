class Solution {
    public boolean isBoomerang(int[][] points) {
        if (points[0][0]==points[1][0]&&points[1][0]==points[2][0])
            return false;
        if (points[0][1]==points[1][1]&&points[1][1]==points[2][1])
            return false;
        int d1 = points[1][0]-points[0][0];
        d1 *= points[2][1]-points[1][1];
        int d2 = points[1][1]-points[0][1];
        d2 *= points[2][0]-points[1][0];
        return d1!=d2;
    }
}
