class Solution 
{
    public boolean isBoomerang(int[][] ar) 
    {
        if(ifEqual(ar[0], ar[1]) || ifEqual(ar[1],ar[2]) || ifEqual(ar[0], ar[2])) return false;
        if(getSlop(ar[0], ar[1]) == getSlop(ar[1], ar[2])) return false;
        return true;
    }
    
    public boolean ifEqual(int[] ar, int[] br)
    {
       if(ar[0]==br[0] && ar[1]==br[1]) return true;
        return false;
    }
    
    public double getSlop(int[] ar, int [] br)
    {
        if(ar[0]==br[0]) return Integer.MAX_VALUE;
        if(ar[1]==br[1]) return 0;
        return (double)(br[1]-ar[1])/(br[0]-ar[0]);
    }
}
