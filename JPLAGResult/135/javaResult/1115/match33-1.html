class Solution {
    public boolean isBoomerang(int[][] points) {
        if (equals(points[0], points[1]) || equals(points[0], points[2]) || equals(points[1], points[2])) {
            return false;
        }
        if (points[0][0] == points[1][0] && points[0][0] == points[2][0]) {
            return false;
        }
        
        if (points[0][0] == points[1][0]) {
            swap(points, 1, 2);
        }
        
        double x1 = points[0][0], y1 = points[0][1];
        double x2 = points[1][0], y2 = points[1][1];
        double x3 = points[2][0], y3 = points[2][1];
        double k = (y1 - y2) / (x1 - x2);
        double a = y1 - x1 * k;
        System.out.println(k);
        System.out.println(a);
        System.out.println(k * x3 + a);
        if (k * x3 + a == y3) {
            return false;
        } else {
            return true;
        }
    
}
    
    private boolean equals(int[] p1, int[] p2) {
        return p1[0] == p2[0] && p1[1] == p2[1];
    }
    
    private void swap(int[][] p, int i, int j) {
        int[] tmp = p[i];
        p[i] = p[j];
        p[j] = tmp;
    }
}
