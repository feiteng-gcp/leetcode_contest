class Solution {
    public boolean isBoomerang(int[][] points) {
        int[] a = points[0], b = points[1], c = points[2];
        
        
if (a[0] == b[0] && a[1] == b[1]) return false;
        if (a[0] == c[0] && a[1] == c[1]) return false;
        if (c[0] == b[0] && c[1] == b[1]) return false;
        
        int l1 = a[0] - b[0], l2 = a[0] - c[0];
        int h1 = a[1] - b[1], h2 = a[1] - c[1];
        
if (l1 == 0 && l2 == 0) return false;
        if (h1 == 0 && h2 == 0) return false;
        if ((l1 + 0.0) / (h1 + 0.0) == (l2 + 0.0) / (h2 + 0.0)) return false;
        return true;
    }
}
