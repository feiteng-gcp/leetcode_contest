class Solution {
    public boolean isBoomerang(int[][] points) {
        return !colLine(points[0][0], points[0][1], points[1][0], points[1][1], points[2][0], points[2][1]);
    }
    boolean colLine(int x1, int y1, int x2, int y2, int x3, int y3) {
        if ((y3 - y2) * (x2 - x1) ==
                (y2 - y1) * (x3 - x2))
            return true;
        else
            return false;
    }
}
