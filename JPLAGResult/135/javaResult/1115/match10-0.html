class Solution {
    public boolean isBoomerang(int[][] p) {
        if (isSame(p[0], p[1]) || isSame(p[1], p[2])  || isSame(p[0], p[2]) ) {
            return false;
        }
        
        int[] d1 = new int[2];
        int[] d2 = new int[2];
        d1[0] = p[1][0] - p[0][0];
        d1[1] = p[1][1] - p[0][1];
        
        d2[0] = p[2][0] - p[0][0];
        d2[1] = p[2][1] - p[0][1];   
        
        int d = d1[0] * d2[0] + d1[1] * d2[1];
        int len1 = d1[0] * d1[0] + d1[1] * d1[1];
        int len2 = d2[0] * d2[0] + d2[1] * d2[1];
        return len1 * len2 != d * d;
    }
    private boolean isSame(int[] a, int[] b) {
        return a[0] == b[0] && a[1] == b[1];
    }
}
