class Solution {
    public boolean isBoomerang(int[][] points) {
        int[] a = points[0], b = points[1], c = points[2];

        int xa = a[0],ya = a[1];
        int xb = b[0],yb = b[1];
        int xc = c[0],yc = c[1];

        // not distinct
        if(xa == xb && ya == yb){
            return false;
        }
        if(xa == xc && ya == yc){
            return false;
        }
        
if(xb == xc && yb == yc){
            return false;
        }
        
        // parallel to y axis
        if(xa == xb && xa == xc){ return false; }
        
        // in a straight line
        
        double k = (ya - yb) / (xa - xb + 0.0);
        double m = ya - k * xa;
        
        return (double) yc != k * xc + m;
    }
}
