class Solution {
    private boolean samePoint(int[] p1, int[] p2) {
        return p1[0] == p2[0] && p1[1] == p2[1];
    }

    private double slope(int[] p1, int[] p2) {
        if (p2[0] == p1[0]) {
            return -1000;
        }
        return 1.0 * (p2[1] - p1[1]) / (p2[0] - p1[0]);
    }

    public boolean isBoomerang(int[][] points) {
        int[] p1 = points[0];
        int[] p2 = points[1];
        int[] p3 = points[2];

        if (samePoint(p1, p2) || samePoint(p2, p3) || samePoint(p1, p3)) {
            return false;
        }
        
        if ((p1[0] == p2[0] && p2[0] == p3[0]) || (p1[1] == p2[1] && p2[1] == p3[1])) {
            return false;
        }

        Set<Double> set = new HashSet<>();   
        set.add(slope(p1, p2));
        set.add(slope(p2, p3));


        return set.size() > 1;
    }
}
