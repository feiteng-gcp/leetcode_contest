class Solution {
    public boolean isBoomerang(int[][] points) {
        if (isSame(points[0], points[1]) || isSame(points[0], points[2]) || isSame(points[1], points[2])) return false;
        if (getSlope(points[0], points[1]) == getSlope(points[1], points[2])) return false;
        return true;
    }
    
    public boolean isSame(int[] a, int[] b) {
        if (a[0] == b[0] && a[1] == b[1]) return true;
        else return false;
    
}
    
    public double getSlope(int[] a, int[] b) {
        if (b[0] == a[0]) return Integer.MAX_VALUE;
        return (b[1] - a[1]) * 1.0 / (b[0] - a[0]);
    }
}
