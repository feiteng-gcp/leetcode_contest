class Solution {
    public boolean isBoomerang(int[][] points) {
        //check same 
        if (same(points[0], points[1]) || same(points[1], points[2]) || same(points[0], points[2]))
            return false;
        
        
        //first check y axis
        if (points[0][0] == points[1][0] && points[1][0] == points[2][0])
            return false;
        else if (points[0][0] == points[1][0] || points[0][0] == points[2][0] || points[1][0] == points[2][0])
            return true;
        
        
        float k1 = (points[0][1] - points[1][1])/(float)(points[0][0] - points[1][0]);
        float k2 =  (points[0][1] - points[2][1])/(float)(points[0][0] - points[2][0]);
        return k1 != k2;
    }
    
    private boolean same(int[] p1, int[] p2){
        if (p1[0] == p2[0] && p1[1] == p2[1])
            return true;
        else
            return false;
    }
}
