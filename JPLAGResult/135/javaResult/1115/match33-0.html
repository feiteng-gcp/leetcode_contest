class Solution {
    public boolean isBoomerang(int[][] points) {
        if (same(points[0], points[1]) || same(points[0], points[2]) || same(points[1], points[2])) {
            return false;
        }
        double d1 = points[1][1] != points[0][1] ?
                ((double)points[1][0] - points[0][0]) / ((double)points[1][1] - points[0][1]) :
                Double.MAX_VALUE;
        double d2 = points[2][1] != points[0][1] ?
                ((double)points[2][0] - points[0][0]) / ((double)points[2][1] - points[0][1]) :
                Double.MAX_VALUE;
        return d1 != d2;
    }
    
    private static boolean same(int[] p1, int[] p2) {
        return p1[0] == p2[0] && p1[1] == p2[1];
    }
}
