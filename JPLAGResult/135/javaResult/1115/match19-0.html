class Solution {
    public boolean isBoomerang(int[][] points) {
        int vx1 = points[0][0] - points[1][0];
        int vy1 = points[0][1] - points[1][1];
        
        int vx2 = points[0][0] - points[2][0];
        int vy2 = points[0][1] - points[2][1];
        
        if(vx1 == 0 && vy1 == 0) {
            return false;
        }
        if(vx2 == 0 && vy2 == 0) {
            return false;
        }
        if(vx1 == 0) {
            return vx2 != 0;
        }
        if(vx2 == 0) {
            return vx1 != 0;
        }
        return vy1 * vx2 != vy2 * vx1;
    }
}
