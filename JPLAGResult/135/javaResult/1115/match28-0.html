class Solution {
    public boolean isBoomerang(int[][] points) {
		int[] a = points[0];
		int[] b = points[1];
		int[] c = points[2];
		if (a[0] == b[0] && a[1] == b[1]) {
			return false;
		}
		if (a[0] == c[0] && a[1] == c[1]) {
			return false;
		}
		if (b[0] == c[0] && b[1] == c[1]) {
			return false;
		}
		if (a[1] == b[1] && a[1] == c[1]) {
			return false;
		}
		if ((a[0] - b[0]) * (a[1] - c[1]) == (a[0] - c[0]) * (a[1] - b[1])) {
			return false;
		}
		return true;
	}
}
