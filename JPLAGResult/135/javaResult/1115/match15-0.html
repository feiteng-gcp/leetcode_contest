class Solution {
    public boolean isBoomerang(int[][] points) {
        double ax = points[0][0] - points[1][0];
        double ay = points[0][1] - points[1][1];
        double bx = points[0][0] - points[2][0];
        double by = points[0][1] - points[2][1];
        if(ax == ay && ax == 0) return false;
        if(bx == by && bx == 0) return false;
        if(ax == 0 && bx == 0) return false;
        if(ax == 0 || bx == 0) return true;
        if(ay / ax == by / bx) return false;
        return true;
    }
}
