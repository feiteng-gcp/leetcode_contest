class Solution {
boolean equal(int[] a, int[] b) {
        return a[0] == b[0] && a[1] == b[1];
    }


    public boolean isBoomerang(int[][] points) {
        if (equal(points[0], points[1]) || equal(points[1], points[2]) ||
                equal(points[0], points[2])) return false;

        if ((points[1][1] - points[0][1]) * (points[2][0] - points[0][0]) ==
                (points[2][1] - points[0][1]) * (points[1][0] - points[0][0])) return false;
        return true;

    }
}
