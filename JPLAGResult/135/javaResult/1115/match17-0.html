class Solution {
    public boolean isBoomerang(int[][] points) {
        return det(points[0][0],points[0][1],points[1][0],points[1][1],points[2][0],points[2][1]);
    }
    
    private boolean det(int px,int py,int qx,int qy,int rx,int ry){
        int flag=(qx-px)*(ry-py)-(qy-py)*(rx-px);
        if(flag==0){
            return false;
        }
        return true;
    }
}
