class Solution {
    public boolean isBoomerang(int[][] points) {
        return cross(points[1][0] - points[0][0], points[1][1] - points[0][1], points[2][0] - points[0][0], points[2][1] - points[0][1]) != 0;
    }
    
    public static int cross(int x1, int y1, int x2, int y2)
    {
        //System.out.println(x1 + ":" + y1 +":" + x2 +":" + y2);
        return x1 * y2 - y1 * x2;
    }
}
