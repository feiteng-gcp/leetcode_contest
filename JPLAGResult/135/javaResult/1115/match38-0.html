class Solution {
    public boolean isBoomerang(int[][] points) {
        if (points[0][0] == points[1][0] && points[0][0] == points[2][0]) {
            return false;
        }
        if (points[0][1] == points[1][1] && points[0][1] == points[2][1]) {
            return false;
        }
        long verify1 = (points[1][1] - points[0][1]) * (points[2][0] - points[0][0]);
        long verify2 = (points[2][1] - points[0][1]) * (points[1][0] - points[0][0]);
        return verify1 != verify2;
    }
}
