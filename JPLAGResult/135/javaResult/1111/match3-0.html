class Solution {
    public int minScoreTriangulation(int[] A) {
        int n = A.length;
        int[][] dp = new int[n][n];
        for (int i = 0; i < n-2; i++) {
            dp[i][i+2] = A[i]*A[i+1]*A[i+2];
        }
        for (int d = 3; d < n; d++) {
            for (int j = d; j < n; j++) {
                int i = j-d;
                dp[i][j] = 1000000000;
                for (int k = i+1; k < j; k++) {
                    dp[i][j] = Math.min(dp[i][j], A[i]*A[j]*A[k]+dp[i][k]+dp[k][j]);
                }
            }
        }
        return dp[0][n-1];
    }
}
