class Solution {
    public int minScoreTriangulation(int[] A) {
        int n = A.length;
        
if (n < 3)
            return 0;

        int[][] table = new int[n][n];

        for (int gap = 0; gap < n; gap++)
        {
            for (int i = 0, j = gap; j < n; i++, j++)
            {
                if (j < i+2)
                    table[i][j] = 0;
                else
                {
                    table[i][j] = Integer.MAX_VALUE;
                    for (int k = i+1; k < j; k++)
                    {
                        int val = (A[i]*A[j]*A[k])+table[i][k]+table[k][j];
                        if (table[i][j] > val)
                            table[i][j] = val;

                    }
                }
            }
        }
        return  table[0][n-1];
    }
}
