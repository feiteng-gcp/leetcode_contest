class Solution {
    public int minScoreTriangulation(int[] A) {
        int N = A.length;
        int[][] dp = new int[N][N];
        
        for (int i = 0; i < A.length; i++) {
            dp[i][(i + 2) % N] = A[i] * A[(i + 1) % N] * A[(i + 2) % N];
        }
        
        for (int n = 3; n < N; n++) {
            for (int i = 0; i < A.length; i++) {
                int j = i + n;
                dp[i][j % N] = Integer.MAX_VALUE;
                for (int k = i + 1; k < j; k++) {
                    dp[i][j % N] = Math.min(dp[i][j % N], dp[i][k % N] + dp[k % N][j % N] + A[i] * A[j % N] * A[k % N]);
                }
            }
        }
        
        for (int[] d :dp) {
            System.out.println(Arrays.toString(d));
        }
        return dp[0][N - 1];
    }
}
