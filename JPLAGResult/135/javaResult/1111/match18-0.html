class Solution {
    double cost(int[] A,int i,int j,int k){
        return A[i]*A[j]*A[k];
    }
    public int minScoreTriangulation(int[] A) {
       int n = A.length;
       double[][] dp = new double[n][n]; 
       for(int diff=0;diff<n;diff++){
           for(int i=0,j=diff;j<n;i++,j++){
                if(j<i+2) 
                    dp[i][j] = 0.0; 
                else{ 
                    dp[i][j] = Double.MAX_VALUE; 
                    for(int k=i+1;k<j;k++){
                    double val=dp[i][k]+dp[k][j]+cost(A,i,j,k); 
                    if (dp[i][j]>val) 
                        dp[i][j] = val; 
                    } 
                } 
            } 
        } 
        return  (int)dp[0][n-1]; 
    }
}
