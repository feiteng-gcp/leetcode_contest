class Solution {
    public int minScoreTriangulation(int[] A) {
        int n = A.length;
        return helper(A, n);
    }

    int helper(int[] A, 
int n) {
        if (n < 3) return 0;
        int[][] B = new int[n][n];
        for (int gap = 0; gap < n; gap++) {
            for (int i = 0, j = gap; j < n; i++, j++) {
                if (j < i + 2)
                    B[i][j] = 0;
                else {
                    B[i][j] = Integer.MAX_VALUE;
                    for (int k = i + 1; k < j; k++) {
                        int val = B[i][k] + B[k][j] + A[i] * A[j] * A[k];
                        if (B[i][j] > val)
                            B[i][j] = val;
                    }
                }
            }
        }
        return B[0][n - 1];
    }
}
