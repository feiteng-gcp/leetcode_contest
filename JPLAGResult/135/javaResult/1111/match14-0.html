class Solution {
    int helper(int []A){
        int n = A.length; // Sorry C++ boy here :(
        if (n < 3) return 0; 
        int[][] dp = new int[n][n]; 
        for (int skip = 0; skip < n; skip++){ 
            for (int i = 0, j = skip; j < n; i++, j++){ 
                if (j < i+2) dp[i][j] = 0; 
                else{ 
                    dp[i][j] = Integer.MAX_VALUE; 
                    for (int k = i+1; k < j; k++) { 
                        int val = dp[i][k] + dp[k][j] + A[i]*A[j]*A[k]; 
                        if (dp[i][j] > val) dp[i][j] = val; 
                    } 
                } 
            } 
        }
        return dp[0][n-1];
    }
    public int minScoreTriangulation(int[] A) {
        return helper(A);
    }
}
