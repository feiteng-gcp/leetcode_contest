class Solution {
    public int minScoreTriangulation(int[] A) {
        int n = A.length;
                
        return minScore(A, n);
    }
    
    public int minScore(int A[], int n){
        if(n < 3)
            return 0;
        
        int cache[][] = new int[n][n];
        
        for(int x = 0; x < n; x++) { 
            for(int i = 0, j = x; j < n; i++, j++) { 
                if (j < i+2) 
                    cache[i][j] = 0; 
                else{ 
                    cache[i][j] = Integer.MAX_VALUE; 
                    for(int k = i + 1; k < j; k++) { 
                        int val = cache[i][k] + cache[k][j] + A[i] * A[j] * A[k]; 
                        if (cache[i][j] > val) 
                            cache[i][j] = val; 
                    } 
                } 
            } 
        } 
        
        return  cache[0][n-1];
    }
}
