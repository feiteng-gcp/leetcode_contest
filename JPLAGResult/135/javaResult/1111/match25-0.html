class Solution {
    public int minScoreTriangulation(int[] A) {
        int[][] f = new int[A.length][A.length];
        int n = A.length;
        for (int gap = 0; gap < n; gap++) { 
            for (int i = 0, j = gap; j < n; i++, j++) { 
              if (j < i+2)  {
                 f[i][j] = 0; 
              }
              else { 
                f[i][j] = Integer.MAX_VALUE; 
                for (int k = i+1; k < j; k++) { 
                    int val = f[i][k] + f[k][j] + A[i] * A[j] * A[k]; 
                    if (f[i][j] > val) 
                         f[i][j] = val; 
                } 
                } 
            } 
            
        }
        return f[0][A.length - 1];
    }
}
