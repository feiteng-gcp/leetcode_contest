class Solution {
    public int minScoreTriangulation(int[] A) {
        int len = A.length;
        if (len == 3) return cost(A, 0, 1, 2);
        
int[][] dp = new int[len][len];
        for (int gap = 0; gap < len; gap++) {
            for (int i = 0, j = gap; j < len; i++, j++) {
                if (j < i+2)
                    dp[i][j] = 0;
                else {
                    dp[i][j] = Integer.MAX_VALUE; 
                    for (int k = i + 1; k < j; k++) {
                        int val = dp[i][k] + dp[k][j] + cost(A, i, j, k);
                        if (dp[i][j] > val)
                            dp[i][j] = val;
                    }
                }
            }
        } 
        return dp[0][len - 1];
    }
    private int cost(int[] A, int i, int j, int k) {
        return A[i] * A[j] * A[k];
    }
}
