class Solution {
    int len;
    public int minScoreTriangulation(int[] A) {
        len = A.length;
        long[][] dp = new long[len * 2][len * 2];
        for (int i = 0; i < len * 2; i++) {
            for (int j = 0; j < len * 2; j++) {
                dp[i][j] = -1;
            }
        }
        return (int)dfs(0, len - 1, dp, A);
    }
    
    private long dfs(int start, int end, long[][] dp, int[] A) {
        if (end -start < 2) {
            return 0;
        }
        if (dp[start][end] != -1) {
            return dp[start][end];
        }
        if (end - start == 2)  {
            return (long)A[start % len] * A[(start + 1) % len] * A[(start + 2) % len];
        }
        long ret = Integer.MAX_VALUE;
        for (int mid = start + 1; mid < end; mid++) {
            ret = Math.min(ret, (long)A[start % len] * A[end % len] * A[mid % len] + dfs(start, mid, dp, A) + dfs(mid, end, dp, A) );
        }
        
        return dp[start][end] = ret;
    }
}
