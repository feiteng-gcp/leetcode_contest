class Solution {
    public int minScoreTriangulation(int[] A) {
        if (A == null || A.length == 0) {
            return 0;
        }
        
int n = A.length;
        int[][] table = new int[n][n]; 

      
        for (int gap = 0; gap < n; gap++) { 
          for (int i = 0, j = gap; j < n; i++, j++) { 
              if (j < i+2)  {
                table[i][j] = 0; 
              } else { 
                table[i][j] = Integer.MAX_VALUE; 
                  for (int k = i+1; k < j; k++) { 
                    int val = table[i][k] + table[k][j] + A[i]*A[j]*A[k]; 
                    if (table[i][j] > val) 
                         table[i][j] = val; 
                  } 
              } 
          } 
       } 
       return  table[0][n-1]; 
        
    }
}
