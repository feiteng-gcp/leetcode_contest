class Solution {
    public int minScoreTriangulation(int[] A) {
        return dp(A,A.length);
    } 
 
    private int dp(int[] A,
int n){
        int[][] score = new int[n][n];
        for(int i=0;i<n;i++){
            int start=0;
            int end=i;
            while(end<n){
                if(start>end-2){
                    score[start][end]=0;
                }
                else{
                    score[start][end]=Integer.MAX_VALUE;
                    for(int j=start+1;j<end;j++){
                        int currentScore=score[start][j]+score[j][end]+A[start]*A[j]*A[end];
                        if(currentScore<score[start][end]){
                            score[start][end]=currentScore;
                        }
                    
}
                }
                start++;
                end++;
            }
        }
        return score[0][n-1];
    }
}
