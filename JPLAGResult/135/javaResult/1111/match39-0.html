class Solution {
    public int minScoreTriangulation(int[] A) {
        int[][] dp = new int[A.length][A.length];
        for (int g = 0; g < A.length; ++g) {
            int i = 0, j = g;
            while (j < A.length) {
                if(j < i+2) {
                    dp[i][j] = 0;
                } else {
                    dp[i][j] = Integer.MAX_VALUE;
                    for (int x = i+1; x < j; ++x) {
                        int y = dp[i][x] + dp[x][j] + A[i]*A[j]*A[x];
                        dp[i][j] = Math.min(dp[i][j], y);
                    }
                }
                i++; j++;
            }
        }
        return dp[0][A.length-1];
    }
}
