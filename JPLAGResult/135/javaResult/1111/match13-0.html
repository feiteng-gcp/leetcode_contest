class Solution {
    public int minScoreTriangulation(int[] A) {
        return helper(0, A.length - 1, A, new Integer[A.length][A.length]);
    }
    
    private int helper(int i, int j, int[] A, Integer[][] memo) {
        if (j < i + 2) return 0;
        if(memo[i][j] != null) return memo[i][j];
        
        int min = Integer.MAX_VALUE;
        for (int k = i + 1; k < j; k++) {
            min = Math.min(min, helper(i, k, A, memo) + helper(k, j, A, memo) + A[i] * A[k] * A[j]);
        }
        
        memo[i][j] = min;
        return min;
    }
}
