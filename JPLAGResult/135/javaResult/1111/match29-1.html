class Solution {
    public int minScoreTriangulation(int[] A) {
        if(A.length<3) return 0;
        int len=A.length;
        
int[][] dp=new int[len][len];
        
        for(int gap=0; gap<len; gap++){
            for(int i=0,j=gap;j<len;i++,j++){
                 if(j<i+2)
                     dp[i][j]=0;
                 else{
                     dp[i][j]=Integer.MAX_VALUE;
                     for(int k=i+1;k<j;k++){
                         int val = dp[i][k]+dp[k][j]+A[i]*A[j]*A[k];
                         if(dp[i][j]>val){
                             dp[i][j]=val;
                         
}
                     }
                 }
            }
        }
        
        return dp[0][len-1];
    }
}
