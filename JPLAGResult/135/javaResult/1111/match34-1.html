class Solution {
    public int minScoreTriangulation(int[] A) {
        if (A == null || A.length == 0) {
            return 0;
        }
        return helper(A, A.length);
    }
    
    public int helper (int[] A, int n) {
        if (n < 3) {
            return 0;
        }
    
        
int[][] table = new int[n][n];

        for (int gap = 0; gap < n; gap++) {
            for (int i = 0, j = gap; j < n; i++, j++) {
                if (j < i+2) {
                    table[i][j] = 0;
                } else {
                    table[i][j] = Integer.MAX_VALUE;

                    for (int k = i+1; k < j; k++) {
                        int val = table[i][k] + table[k][j] + cost(A,i,j,k);
                        if (table[i][j] > val) {
                            table[i][j] = val;    //update table data to minimum value
                        
}
                    }
                }
            }
        }  
        return  table[0][n-1];
    }
    
    public int cost(int[] A, int i, int j, int k) {
        int p1 = A[i], p2 = A[j], p3 = A[k];
        return p1 * p2 * p3;
    }
