class Solution {
    
    
 
    int rec(int[] poly, int i, int j) 
    { 
        if (j<i+2) 
            return 0; 
  
        int ans = Integer.MAX_VALUE; 
        for (int k=i+1; k<j; k++) {
            ans = Math.min(ans, (rec(poly, i, k) + rec(poly, k, j) + (poly[i]*poly[j]*poly[k]) )); 
        }
        
        return  ans; 
} 
    
    public int minScoreTriangulation(int[] A) {
        // int l = A.length;
        // return rec(A,0,l-1);
        
        int n = A.length;
        if(n<3) 
            return 0; 
  
        int[][] tab= new int[n][n]; 
  
   
   for(int x=0; x<n; x++) 
   { for (int i=0, j=x; j<n; i++, j++) 
      { if (j<i+2) 
             tab[i][j] = 0; 
          else
          { tab[i][j] = Integer.MAX_VALUE; 
              for (int k = i+1; k < j; k++) 
              { int val = tab[i][k] + tab[k][j] + (A[i]*A[j]*A[k]); 
                if (tab[i][j] > val) 
                     tab[i][j] = val; 
              } } }
   } 
   return  tab[0][n-1];    
    }
}
