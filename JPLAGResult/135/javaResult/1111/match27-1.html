class Solution {
    public int minScoreTriangulation(int[] A) {
        final int n = A.length;
        
int[][] dp = new int[n][n];
        for (int side = 3; side <= n; side++) {
            for (int i = 0, j = side-1; j < n; i++, j++) {
                if (j - i + 1 == 3) {
                    dp[i][j] = A[i] * A[i+1] * A[i+2];
                } else {
                    int min = Integer.MAX_VALUE;
                    for (int k = i+1; k < j; k++) {
                        
min = Math.min(min, dp[i][k] + dp[k][j] + A[i] * A[j] * A[k]);
                    }
                    dp[i][j] = min;
                }
            }
        }
        return dp[0][n-1];
    }
}
