class Solution {
    public int minScoreTriangulation(int[] A) {
        int n = A.length;
        
        // dp[i][j] = minScoreTriangulation(A[i..j])
        int[][] dp = new int[n][n];
        
        //dp[0][n-1] = min {A[0] * A[k] * A[n-1] + dp[0][k] + dp[k][n-1]}
        
        
        for(int i=0;i<n-1;i++) {
            dp[i][i+1] = 0;
        }
        
        for(int l=2;l<n;l++) {
            for(int i=0;i<n-l;i++) {
                int j = i + l;
                //compute dp[i][j]
                //dp[i][j] = min {A[i] * A[k] * A[j] + dp[i][k] + dp[k][j] for k, i<k<j}
                dp[i][j] = Integer.MAX_VALUE;
                for(int k=i+1;k<j;k++) {
                    dp[i][j] = Math.min(dp[i][j], A[i] * A[k] * A[j] + dp[i][k] + dp[k][j]);
                }
            }
        }
        
        return dp[0][n-1];
    }
}
