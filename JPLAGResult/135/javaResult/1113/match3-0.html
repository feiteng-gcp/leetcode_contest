class Solution {
    public int[] numMovesStonesII(int[] pos) {
        int N = pos.length;

        Arrays.sort(pos);

        int mins = N;
        if ((pos[N - 1] - pos[1] + 1 == N - 1 && pos[1] - pos[0] > 2) || (pos[N - 2] - pos[0] + 1 == N - 1 && pos[N - 1] - pos[N - 2] > 2)) {
            mins = 2;
        } else {
            int end = 0;
            int cnt = 0;
            for (int i = 0; i < N; ++i) {
                while (end < N && pos[end] - pos[i] < N) {
                    ++cnt;
                    ++end;
                }
                mins = Math.min(mins, N - cnt);
                --cnt;
            }
        }

        int maxs = pos[N - 1] - pos[0] + 1 - N - Math.min(pos[1] - pos[0] - 1, pos[N - 1] - pos[N - 2] - 1);

        return new int[]{mins, maxs};
    }
}
