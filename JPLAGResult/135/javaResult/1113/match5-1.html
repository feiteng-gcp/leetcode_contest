	class Solution {
	    public int[] numMovesStonesII(int[] a) {
	        Arrays.sort(a);
	        int n = a.length;
	        int maxlen = 0;
	        for(int i = 0;i < n;i++){
	        	int ind = Arrays.binarySearch(a, a[i]+n-1);
	        	if(ind < 0)ind = -ind-2;
	        	maxlen = Math.max(maxlen, ind-i+1);
	        }
	        
	        if(maxlen == n-1){
		        for(int i = 0;i < n;i++){
		        	int ind = Arrays.binarySearch(a, a[i]+n-1);
		        	int hit = ind;
		        	if(ind < 0)ind = -ind-2;
		        	if(ind-i+1 == maxlen){
		        		if(hit >= 0){
		        			break;
		        		}
			        	if((a[n-1] - a[n-2] > 2 || a[1] - a[0] > 2)){
			        		maxlen--;
			        		break;
			        	}
		        	}
		        }
	        }
	        int retmin = n-maxlen;
	        
	        int max
 = Math.max(a[n-1] - a[1] - (n-2), a[n-2] - a[0] - (n-2));
	        return new int[]{retmin, max};
	    }
	}
