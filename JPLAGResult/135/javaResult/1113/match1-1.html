class Solution {
    public int[] numMovesStonesII(int[] stones) {
        Arrays.sort(stones);
        int n = stones.length;
        int a = stones[0];
        int b = stones[n - 1];
        int[] res = new int[2];
        if(b - a == n - 1) return res;
        int t1 = stones[n - 2] - stones[0];
        int t2 = stones[n - 1] - stones[1];
        int t = Math.max(t1, t2);
        res[1] = t + 2 - n;
        int l = 0;
        int r = 0;
        int max = 0;
        int acc = 0;
        while(r < n) {
            if(stones[r] - stones[l] < n) {
                ++acc;
            
} else {
                while(stones[r] - stones[l] >= n) {
                    ++l;
                    --acc;
                }
                ++acc;
            }
            if(r == l || stones[r] - stones[l] != acc - 1) {
                if(acc > max) max = acc;
            } else {
                if(acc - 1 > max) max = acc - 1;
            }
            ++r;
        }
        res[0] = n - max;
        return res;
    }
}
