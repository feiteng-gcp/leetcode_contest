class Solution {
    public int[] numMovesStonesII(int[] a) {
        Arrays.sort(a);
        int n = a.length;
        // max is the number of holes if some end have >= 2 stones
        int min = 0, max = 0;
        
        int sumh = 0;
        List<Integer> holes = new ArrayList<>();
        for (int i = 0, pre = a[0] - 1; i < n; i++) {
            if (pre + 1 == a[i]) {
                // do nothing
            }else {
                //current hole size
                int ch = a[i] - pre - 1;
                sumh += ch;
                holes.add(ch);
            }
            pre = a[i];
        }
        int nhole = holes.size();
        int[] ans = new int[2];
        if (nhole == 0) return ans;
        
        if (a[0] + 1 == a[1] || a[n-2] + 1 == a[n-1]) {
            max = sumh;
        }else {
            max = sumh - Math.min(holes.get(0), holes.get(nhole - 1));
        }
        ans[1] = max;
        // min ? 
        boolean first = true, last = true;
        if (nhole == 1 && (a[0] + 1 < a[1] || a[n-2] + 1 < a[n-1])) {
            if (a[0] + 1 < a[1])
                last = false;
            else
                first = false;
        }
        
        int[] dp = new int[n];
        for (int i = 0, j = 0; i < n; i++) {
            while (j < n && a[i] + n - 1 >= a[j]) j++;
            dp[i] = j - i;
        }
        min = n;
        for (int s = 0; s < n - 1; s++) {
            if (!first && s==0) continue;
            if (a[s] + n - 1 > a[n-1]) break;
            if (a[s] + n - 1 == a[n-1] && !last) break;
            
            min = Math.min(min, n - dp[s]);
        }
        
        ans[0] = min;
        return ans;
    }
}
