class Solution {
    	public int[] numMovesStonesII(int[] stones) {
		Arrays.sort(stones);
		int count = stones.length;
		if (stones[count - 1] - stones[0] == count - 1) {
			return new int[]{0, 0};
		}
		// find min move
		int left = 0;
		int range = stones[left] + count;
		int maxInRange = 1;
		int currInRange = 1;
		boolean conti = false;
		for (int right = 1; right < count; right++) {
			if (stones[right] < range) {
				currInRange++;
				if (currInRange > maxInRange) {
					maxInRange = currInRange;
					conti = (stones[right] - stones[left] == right - left);
				}
			} else {
				while (stones[right] >= range) {
					left++;
					range = stones[left] + count;
				}
				currInRange = right - left + 1;
				if (currInRange == maxInRange) {
					if (stones[right] - stones[left] > right - left) {
						conti = false;
					}
				}
			}
		}
		int minMove = count - maxInRange;
		if (conti) {
			minMove++;
		}
		int maxMove = Math.max(stones[count - 1] - stones[1], stones[count - 2] - stones[0]) + 2 - count; 
		return new int[]{minMove, maxMove};
    }
}
