class Solution {
    public int[] numMovesStonesII(int[] stones) {
        Arrays.sort(stones);
        int[] result = new int[2];
        // min
        int n = stones.length;
        int maxK = 1;
        int k = 1;
        for(int i = 1; i < n; i++) {
            if(stones[i] - stones[i-k] < n) {
                ++k;
                maxK = Math.max(k, maxK);
            } else {
                while(stones[i] - stones[i - k + 1] >= n) {
                    --k;
                }
            }
        }
        if(maxK == n-1) {
            if(stones[n - 2] - stones[0] == n - 2 || stones[n-1] - stones[1] == n - 2) {
                if(stones[n-1] - stones[0] == n) { 
                    result[0] = 1;
                } else {
                    result[0] = 2;
                }
            } else {
                result[0] = n - maxK;
            }
        } else {
            result[0] = n - maxK;
        }

        //max
        result[1] = stones[n-1] - stones[0] - 1 - (Math.min(stones[1] - stones[0], stones[n-1] - stones[n-2]) - 1) - (n - 2);
        return result;
    }
    
}
