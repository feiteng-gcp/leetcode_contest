class Solution {
    public int[] numMovesStonesII(int[] stones) {
        // stones = new int[]{6,5,4,3,8};
        Arrays.sort(stones);
        int len = stones.length, n = 0;
        for (int i = 0, j = 1; j < stones.length; ) {
            if (stones[j] - stones[i] < len) {
                n = Math.max(n, j - i + 1);
                j++;
            }
            else i++;
        }
        int min = len - n;
        if (min == 1) {
            boolean con1 = true, con2 = true;
            for (int i = 1; i < len - 1; i++) {
                if (stones[i] != stones[i - 1] + 1) {
                    con1 = false;
                }
                if (stones[i + 1] != stones[i] + 1) {
                    con2 = false;
                }
            }
            if (con1 || con2) min = 2;
        }
        int max = 0;
        for (int j = 2; j < stones.length - 1; j++) {
            max += stones[j] - stones[j - 1] - 1;
        }
        max = max + Math.max(stones[1] - stones[0], stones[len - 1] - stones[len - 2]) - 1;
        return new int[]{Math.min(min, max), max};
    }
}
