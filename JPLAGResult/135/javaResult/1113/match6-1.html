import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    public int[] numMovesStonesII(int[] stones) {
        if (stones.length < 3) {
            return new int[]{0, 0};
        }
        
Arrays.sort(stones);

        int n = stones.length;
        if (stones[n - 1] - stones[0] == n - 1) {
            return new int[]{0, 0};
        }

        int maxMoves = 0;
        for (int i = 0; i < n - 1; i++) {
            maxMoves += stones[i + 1] - stones[i] - 1;
        }
        maxMoves -= Math.min(stones[1] - stones[0] - 1, stones[n - 1] - stones[n - 2] - 1);

        int minMoves = minMoves(stones);


        return new int[]{minMoves, maxMoves};
    }

    int minMoves(int[] stones) {
        int maxWindowSize = 0;
        int a = stones[1];
        int b = stones[stones.length - 2];

        for (int i = 0, j = i; i < stones.length; i++) {
            while (j + 1 < stones.length && stones[j + 1] - stones[i] + 1 <= stones.length) {
                j++;
            }
            int left = stones[i];
            int right = stones[j];
            int tmp = j - i + 1;
            if (right - left == j - i) {
                if (j == stones.length - 1 || i == 0)  {
                    tmp--;
                }
            }


            maxWindowSize = Math.max(maxWindowSize, tmp);
//            System.out.println(stones[i] + " " + stones[j] + " " + maxWindowSize);
        }
        return stones.length - maxWindowSize;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(Arrays.toString(sol.numMovesStonesII(new int[]{1, 500000000, 1000000000})));
        System.out.println(Arrays.toString(sol.numMovesStonesII(new int[]{2, 5, 6, 7, 8})));
        System.out.println(Arrays.toString(sol.numMovesStonesII(new int[]{7, 4, 9})));
        System.out.println(Arrays.toString(sol.numMovesStonesII(new int[]{1, 2, 5})));
        System.out.println(Arrays.toString(sol.numMovesStonesII(new int[]{-1, -2, -5})));
        System.out.println(Arrays.toString(sol.numMovesStonesII(new int[]{6, 5, 4, 3, 10})));
        System.out.println(Arrays.toString(sol.numMovesStonesII(new int[]{100, 101, 104, 102, 103})));

    }
}
