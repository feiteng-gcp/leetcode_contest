class Solution {

	private int getCnt(int[] stones) {
		int ans = 1;	
		for (int i = 0;i < stones.length - 1;i ++) {
			if (stones[i + 1] != stones[i] + 1) {
				ans ++;				
			}
		}

		return ans;
	}
	
	private int getMin(int[] stones) {
		int ans = - 1 , i , j = 0 , n = stones.length , cnt = getCnt(stones);
		if (cnt == 1) {
			return 0;
		} else if (cnt == 2) {
			int diff = - 1;
			for (i = 0;i < n - 1;i ++) {
				if (stones[i + 1] != stones[i] + 1) {
					diff = stones[i + 1] - stones[i];
					break;					
				}
			}	
			if (diff == 2) {
				return 1;				
			} else {				
				return 2;				
			}
		} else {
			for (i = 0;i < n;i ++) {
				while (j <= i && stones[i] - stones[j] >= n) {
					j ++;				
				}
				int remain = n - (i - j + 1);
				if (remain < ans || ans < 0) {
					ans = remain;
				}
			}
			j = n - 1;		
			for (i = n - 1;i >= 0;i --) {			
				while (j >= i && stones[j] - stones[i] >= n) {
					j --;				
				}
				int remain = n - (j - i + 1);
				if (remain < ans || ans < 0) {				
					ans = remain;				
				}
			}
			return ans;
		}
	}
	
	private int getResult(int[] stones , int from , int to) {
		int ans = 0;		
		for (int i = from;i < to;i ++) {				
			ans += (stones[i + 1] - stones[i] - 1);				
		}
		return ans;
	}
	
	private int getMax(int[] stones , int from , int to) {		
		int ans = 0;		
		if (stones[from] + 1 == stones[from + 1] || stones[to - 1] + 1 == stones[to]) {
			ans = getResult(stones , from , to);			
		} else {
			int result1 = getResult(stones , from + 1 , to);
			int result2 = getResult(stones , from , to - 1);
			ans = Math.max(result1 , result2);			
		}
		return ans;
	}
	
    public int[] numMovesStonesII(int[] stones) {
    	Arrays.sort(stones);
    	return new int[] {getMin(stones) , getMax(stones , 0 , stones.length - 1)};    	
    }
}
