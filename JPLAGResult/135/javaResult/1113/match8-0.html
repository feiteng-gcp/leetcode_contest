class Solution {

	public int[] numMovesStonesII(int[] stones) {
		Arrays.sort(stones);
		int n = stones.length;
		int mx = 0;
		for (int i = 0; i < n - 1; i++) mx += stones[i+1] - stones[i] - 1;
		mx -= Math.min(stones[1] - stones[0] - 1, stones[n-1] - stones[n-2] - 1);

		int mi = stones[n-1] - stones[0] + 1;
		int l = 0;
		int r = 0;
		for (; l < n; l++) {
			while (r < n && stones[r] - stones[l] + 1 <= n) r++;
			if (r - l == n-1 && stones[r-1] - stones[l] + 1 == n-1) {
				if (r < n) {
					mi = Math.min(mi, Math.min(stones[r] - stones[r - 1] - 1, 2));
				} else {
					mi = Math.min(mi, Math.min(stones[l] - stones[l - 1] - 1, 2));
				}
			} else {
				mi = Math.min(mi, n - (r - l));
			}
		}
		return new int[] {mi, mx};
	}
}
