class Solution {
    public int[] numMovesStonesII(int[] stones) {
        Arrays.sort(stones);
        int[] res = new int[2];
        int n = stones.length;
        int unoccupiedCount = stones[n - 1] -stones[0] - n + 1;
        int toLose = Math.min(stones[1] - stones[0] - 1, stones[n - 1] - stones[n - 2] - 1);
        res[1] = unoccupiedCount - toLose;
        res[0] = res[1];
        if (res[1] == 0) {
            return res;
        }
        
        if (stones[n - 1] - stones[1] == n - 2 || stones[n - 1] - stones[0] == n - 2) {
            res[0] = 2;
            return res;
        }
        
        //stones[1] + n - 1;
        int idx = 2;
        for (int i = 1; i < n - 1; ++i) {
            int endNum = stones[i] + n - 1;
            while (idx < n && stones[idx] < endNum) {
                idx++;
            }
            if (idx >= n || stones[idx] > endNum) {
                int actualNumOfElem = idx - i;
                res[0] = Math.min(res[0], n - actualNumOfElem);
            } else {
           
                int actualNumOfElem = idx - i + 1;
                res[0] = Math.min(res[0], n - actualNumOfElem);
            }
        }
        return res;
    }
}
