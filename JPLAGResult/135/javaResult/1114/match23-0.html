/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int sum = 0;
    public TreeNode bstToGst(TreeNode root) {
        Helper(root);
        return root;
        
    }
    
    public void Helper(TreeNode node) {
        if (node == null) {
            return;
        }
        
        Helper(node.right);
        
        //do something
        sum += node.val;
        node.val = sum;
        
        Helper(node.left);
    }
}
