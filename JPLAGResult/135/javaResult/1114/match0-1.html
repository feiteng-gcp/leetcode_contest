/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private int sum;
    public TreeNode bstToGst(TreeNode root) {
        if(root==null)return null;
        sum=0;
        dfs(root);
        return root;
    }

    private void dfs(TreeNode root) {
        if(root==null)return;
        dfs(root.right);
        sum+=root.val;
        root.val=sum;
        dfs(root.left);
    }
}
