class Solution {
    int dfs(TreeNode node, int v){
        if(node==null) return v;
        int vv = dfs(node.right, v);
        node.val += vv;
        vv = dfs(node.left, node.val);
        return vv;
    }

    public TreeNode bstToGst(TreeNode root) {
        dfs(root, 0);
        return root;
    }
}
