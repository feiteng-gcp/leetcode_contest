/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int sum = 0;
    public TreeNode bstToGst(TreeNode root) {
        dfs(root);
        return root;
    }
    
    public void dfs(TreeNode node){
        if(node==null) return;
        dfs(node.right);
        node.val += sum;
        sum = node.val;
        dfs(node.left);
    }
}
