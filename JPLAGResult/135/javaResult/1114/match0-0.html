/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int sum;
    public TreeNode bstToGst(TreeNode root) {
        if (root == null) return root;
        
        sum = 0;
        revOrder(root);
        return root;
    }
    
    private void revOrder(TreeNode root) {
        if (root == null) return;
        revOrder(root.right);
        sum += root.val;
        root.val = sum;
        revOrder(root.left);
    }
}
