/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode bstToGst(TreeNode root) {
        List<TreeNode> inorderList = inorder(root);
        int size = inorderList.size();
        int rootIndex = 0;
        boolean flag = false;
        for (int i = size - 2; i >= 0; i--) {
            if (!flag) {
                int currentValue = inorderList.get(i).val;
                if (currentValue == root.val) {
                    rootIndex = i;
                    flag = true;
                }
            }
            int previousSum = inorderList.get(i + 1).val;
            inorderList.get(i).val += previousSum;
        }
        return inorderList.get(rootIndex);
    }

    public List<TreeNode> inorder(TreeNode root) {
        List<TreeNode> list = new ArrayList<TreeNode>();
        if (root == null)
            return list;
        else if (root.left == null && root.right == null) {
            list.add(root);
            return list;
        } else {
            TreeNode left = root.left;
            TreeNode right = root.right;
            List<TreeNode> leftList = inorder(left);
            List<TreeNode> rightList = inorder(right);
            list.addAll(leftList);
            list.add(root);
            list.addAll(rightList);
            return list;
        }
    }
}
