/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode bstToGst(TreeNode root) {
        solve(root);
        return root;
    }
    int sum = 0;
    private void solve(TreeNode node) {
        if (node == null) return;
        solve(node.right);
        node.val += sum;
        sum = node.val;
        solve(node.left);
    }
}
