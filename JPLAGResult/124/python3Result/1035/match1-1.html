# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        dir = dict()
        def dfs(node, d, p):
            nonlocal dir
            
if node:
                dir[node.val] = (d, p)
                dfs(node.left, d + 1, node.val)
                dfs(node.right, d + 1, node.val)
        dfs(root, 0, None)
        
        dx, px = dir.get(x, (-1, None))
        dy, py = dir.get(y, (-1, None))
        
        if dx == -1 or dx != dy or px == py:
            return False
        return True
