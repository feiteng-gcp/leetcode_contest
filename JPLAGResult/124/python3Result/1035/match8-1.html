# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        d={}
        def dfs(node, level):
            if not node: return
            if node.left:
                d[node.left.val]=(node.val, level)
                dfs(node.left,level+1)
            if node.right:
                d[node.right.val]=(node.val,level)
                dfs(node.right, level+1)
        dfs(root, 0)
        return x in d and y in d and d[x][0] != d[y][0] and d[x][1] == d[y][1]
