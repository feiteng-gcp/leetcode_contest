# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        record = [None] * 101
        def bfs(root, level, par):
            record[root.val] = (level, par)
            
            if root.left:
                bfs(root.left, level + 1, root.val)
            if root.right:
                bfs(root.right, level + 1, root.val)
        
        bfs(root, 0, 0)
        return record[x][0] == record[y][0] and record[x][1] != record[y][1]
