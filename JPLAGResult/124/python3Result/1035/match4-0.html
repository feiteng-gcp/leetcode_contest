# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

def getDepth(node, value, depth, parent):
    if node is None:
        return None
    if node.val == value:
        return [depth, parent]
    r = getDepth(node.left, value, depth + 1, node)
    if r is not None:
        return r
    r = getDepth(node.right, value, depth + 1, node)
    return r

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        ld, lp = getDepth(root, x, 0, None)
        rd, rp = getDepth(root, y, 0, None)
        return ld == rd and lp != rp
