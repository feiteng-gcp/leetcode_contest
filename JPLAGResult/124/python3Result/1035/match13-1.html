# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        def search(root, parent, depth, target):
            if root is None:
                return None
            if root.val == target:
                return depth, parent
            lr = search(root.left, root, depth + 1, target)
            if lr is not None:
                return lr
            return search(root.right, root, depth + 1, target)
        
        xd, xp = search(root, None, 0, x)
        yd, yp = search(root, None, 0, y)
        return xd == yd and xp != yp
