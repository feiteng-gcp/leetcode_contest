# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        
        def dfs1(root, d, x):
            if root is None:
                return 0
            
            if root.val == x:
                return d
            
            t = dfs1(root.left, d + 1, x)
            if t != 0:
                return t
            t = dfs1(root.right, d + 1, x)
            if t != 0:
                return t
            return 0
        
        def dfs2(root, x, y):
            if root is None:
                return True
            
            if (root.left is not None and root.right is not None) and ((root.left.val == x and root.right.val == y) or (root.left.val == y and root.right.val == x)):
                return False
            
            return (dfs2(root.left, x, y) and dfs2(root.right, x, y))
        
        if dfs1(root, 0, x) == dfs1(root, 0, y) and (dfs2(root, x, y) == True):
            return True
        else:
            return False
