from collections import deque

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':

        
        QUE = deque()
        QUE.append((
root,0))
        LIST=[None]*300
        LIST[1]=(0,0)

        while QUE:
            n,h=QUE.pop()

            if n.left!=None:
                LIST[n.left.val]=(n,h+1)
                QUE.append((n.left,h+1))
                
            
            if n.right!=None:
                LIST[n.right.val]=(n,h+1)
                QUE.append((n.right,h+1))

        if LIST[x][1]==LIST[y][1] and LIST[x][0]!=LIST[y][0]:
            return True
        else:
            return False
