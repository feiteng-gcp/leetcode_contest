# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        def helper(node, depth, val):
            if node is None:
                return None
            if node.left is not None and node.left.val == val:
                return [node, depth + 1]
            if node.right is not None and node.right.val == val:
                return [node, depth + 1]
            left = helper(node.left, depth+1,val)
            if left:
                return left
            right = helper(node.right, depth+1,val)
            if right:
                return right
        l = helper(root,0,x)
        r = helper(root,0,y)
        if l and r and len(l) == 2 and len(r) == 2 and l[0] != r[0] and l[1] == r[1]:
            return True
        return False
