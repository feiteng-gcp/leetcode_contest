# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        BFS=[(root,None)]
        tgt=[]
        while(BFS):
            BFSNext=[]
            for n,p in BFS:
                
if n.left:BFSNext.append((n.left,n))
                if n.right:BFSNext.append((n.right,n))
                if (n.val==x)|(n.val==y):
                    tgt.append((n,p))
            BFS=BFSNext
            if len(tgt)==1:return False
            if len(tgt)==2:
                (n1,p1),(n2,p2)=tuple(tgt)
                if p1==p2:
                    return False
                else:
                    return True
