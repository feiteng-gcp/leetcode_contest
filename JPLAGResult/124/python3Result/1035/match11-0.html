# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        if root.val in (x, y):
            return False
        
        def calc(par, node, z):
            if not node:
                return None
            if node.val == z:
                return par.val, 1
            l = calc(node, node.left, z)
            if l:
                return l[0], l[1] + 1
            r = calc(node, node.right, z)
            if r:
                return r[0], r[1] + 1
        
        res_x = calc(None, root, x)
        res_y = calc(None, root, y)
        # print(res_x, res_y)
        return res_x[0] != res_y[0] and res_x[1] == res_y[1]
