# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCousins(self, root: 'TreeNode', x: 'int', y: 'int') -> 'bool':
        d = []
        def dfs(u, dep):
            nonlocal d
            if not u:
                return
            if u.left and u.right and {u.left.val, u.right.val} == {x, y}:
                d += [None]
            if u.val in {x, y}:
                d += [dep]
            dfs(u.left, dep + 1)
            dfs(u.right, dep + 1)
        dfs(root, 0)
        if len(d) == 2 and d[0] == d[1]:
            return True
        return False
