class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        def rottenCount(g):
            return len([cell for row in g for cell in row if cell == 2])
        def orangeCount(g):
            return len([cell for row in g for cell in row if cell != 0])
        tc = orangeCount(grid)
        rc = rottenCount(grid)
        t = 0
        h=len(grid)
        w=len(grid[0])
        while tc != rc:
            t+=1
            prev=rc
            newGrid=[row[:] for row in grid]
            for y, row in enumerate(grid):
                for x, cell in enumerate(row):
                    if cell==2:
                        
for ny, nx in [(y+1,x),(y-1,x), (y,x+1),(y,x-1)]:
                            if ny >=0 and ny<h and nx>=0 and nx <w and newGrid[ny][nx]==1:
                                newGrid[ny][nx]=2
                                rc+=1
            if rc == prev:
                return -1
            grid=newGrid
        return t
