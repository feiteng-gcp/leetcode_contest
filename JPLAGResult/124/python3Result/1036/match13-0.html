def hasRottenNeighbor(g, r, c):
    if r > 0 and g[r - 1][c] == 2:
        return True
    if r < len(g) - 1 and g[r + 1][c] == 2:
        return True
    if c > 0 and g[r][c - 1] == 2:
        return True
    if c < len(g[0]) - 1 and g[r][c + 1] == 2:
        return True
    return False
    

class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        minutes = 0
        while True:
            willRott = []
            for r in range(len(grid)):
                for c in range(len(grid[0])):
                    if grid[r][c] == 1 and hasRottenNeighbor(grid, r, c):
                        willRott.append((r, c))
            if not willRott:
                break
            for r, c in willRott:
                grid[r][c] = 2
            minutes += 1
        for r in range(len(grid)):
            for c in range(len(grid[0])):
                if grid[r][c] == 1:
                    return -1
        return minutes
