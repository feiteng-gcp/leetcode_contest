class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        fresh = set()
        if not grid:
            return -1
        dx, dy = len(grid), len(grid[0])
        
        for i in range(dx):
            for j in range(dy):
                if grid[i][j] == 1:
                    fresh.add((i, j))
                    
        minute = 0
        while fresh:
            minute += 1
            change = set()
            for i, j in fresh:
                for x, y in ((i + 1, j), (i - 1, j), (i, j + 1), (i, j - 1)):
                    if x >= 0 and x < dx and y >= 0 and y < dy:
                        if grid[x][y] == 2:
                            change.add((i, j))
                            break
            if not change:
                return -1
            fresh = fresh - change
            for x, y in change:
                grid[x][y] = 2
       
        return minute
