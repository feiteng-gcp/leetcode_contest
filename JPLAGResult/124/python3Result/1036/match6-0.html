import collections

class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        fresh = 0
        maxMin = 0
        n, m = len(grid), len(grid[0])
        q = collections.deque([])
        for row in range(n):
            for col in range(m):
                if grid[row][col] == 1:
                    fresh += 1
                if grid[row][col] == 2:
                    q.append((row, col, 0))
        while q:
            x, y, minute = q.popleft()
            if minute > maxMin:
                maxMin = minute
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                if 0 <= x + dx < n and 0 <= y + dy < m and grid[x+dx][y+dy] == 1:
                    grid[x+dx][y+dy] = 2
                    q.append( (x+dx, y+dy, minute+1) )
                    fresh -= 1
        if fresh == 0:
            return maxMin
        else:
            return -1
