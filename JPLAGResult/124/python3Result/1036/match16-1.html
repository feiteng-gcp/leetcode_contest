class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        m, n = len(grid), len(grid[0])
        fresh = 0
        q = collections.deque()
        for i in range(m):
            for j in range(n):
                if grid[i][j]==1:
                    fresh += 1
                elif grid[i][j]==2:
                    q.append((i,j))
        if len(q)==0:
            if fresh>0:
                return -1
            return 0
        dirs = [[-1,0],[1,0],[0,-1],[0,1]]
        minute = 0
        while len(q)>0:
            size = len(q)
            for _ in range(size):
                i,j = q.popleft()
                for each in dirs:
                    ni, nj = i+each[0], j+each[1]
                    
if ni>=0 and ni<m and nj>=0 and nj<n and grid[ni][nj]==1:
                        grid[ni][nj] = 2
                        fresh -= 1
                        q.append((ni,nj))
            minute += 1
        if fresh>0:
            return -1
        return minute-1
