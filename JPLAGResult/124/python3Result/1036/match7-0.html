class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        n = len(grid)
        m = len(grid[0])
        que = collections.deque()
        dis = [[0] * m for i in range(n)]
        for i in range(n):
            for j in range(m):
                if grid[i][j] == 2:
                    que += [(i,j)]
        ans = 0
        def go(x, y, d):
            nonlocal n, m, ans, que
            if 0 <= x < n and 0 <= y < m:
                if grid[x][y] == 0:
                    return
                if grid[x][y] == 2:
                    return
                grid[x][y] = 2
                que += [(x, y)]
                dis[x][y] = d
                ans = d
        while que:
            x, y = que.popleft()
            d = dis[x][y]
            go(x + 1, y, d + 1)
            go(x, y + 1, d + 1)
            go(x - 1, y, d + 1)
            go(x, y - 1, d + 1)
        
        for i in range(n):
            for j in range(m):
                if grid[i][j] == 1:
                    return -1
        return ans
