class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        """

        """
        self.grid = grid
        cnt_good = 0
        m, n = len(grid), len(grid[0])
        rotten = collections.deque()

        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    cnt_good += 1
                elif grid[i][j] == 2:
                    rotten.append( (i,j, 0) )
        
        cnt_rotten = len(rotten)
        ts = 0
        while rotten:
            ci, cj, cdist = rotten.popleft()
            ts = max(ts, cdist)
            grid[ci][cj] = 2
            
            for ni,nj in self.neighbors(ci, cj):
                if grid[ni][nj] == 1:
                    cnt_good -= 1
                    grid[ni][nj] = 2
                    rotten.append( (ni, nj, cdist + 1) )
            
        if cnt_good > 0:
            return -1
        return ts
            
    
    def neighbors(self, i, j):
        m, n = len(self.grid), len(self.grid[0])
        ns = [(i-1,j), (i+1, j), (i, j-1), (i, j+1)]
        return ((ni,nj) for (ni,nj) in ns if 0 <= ni < m and 0 <= nj < n)
