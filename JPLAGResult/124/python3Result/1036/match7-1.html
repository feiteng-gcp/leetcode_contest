from collections import deque

class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        m, n = len(grid), len(grid[0])
        q, d, fresh 
= deque(), 0, 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    fresh += 1
                elif grid[i][j] == 2:
                    q.append((i, j))
        
        if fresh == 0:
            return 0
        
        while q:
            d += 1
            
            for _ in range(len(q)):
                i, j = q.popleft()
                
                for x, y in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
                    if 0 <= x < m and 0 <= y < n and grid[x][y] == 1:
                        grid[x][y] = 0
                        fresh -= 1
                        q.append((x, y))
                        
            if fresh == 0:
                return d
            
        return -1
