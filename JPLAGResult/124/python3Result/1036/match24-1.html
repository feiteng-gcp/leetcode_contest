class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        
        res = 0
        m = len(grid)
        n = len(grid[0])
        dx = [0, 0, 1, -1]
        dy = [-1, 1, 0, 0]
        tempS = ""
        
        for i in range(0, m):
            for j in range(0, n):
                tempS = tempS + str(grid[i][j])
        while True:
            res += 1
            for i in range(0, m):
                for j in range(0, n):
                    if (grid[i][j] == 2):
                        for k in range(0, 4):
                            if (i + dx[k] >= 0 and i + dx[k] < m and j + dy[k] >= 0 and j + dy[k] < n
                                and grid[i + dx[k]][j + dy[k]] == 1):
                                grid[i + dx[k]][j + dy[k]] = 3
            
            s = ""
            for i in range(0, m):
                for j in range(0, n):
                    if grid[i][j] == 3:
                        grid[i][j] = 2
                    s = s + str(grid[i][j])    
            
            
            if s == tempS:
                break
            else:
                tempS = 
s
                
        for i in range(0, m):
            for j in range(0, n):
                if grid[i][j] == 1:
                    return -1
        return res - 1
