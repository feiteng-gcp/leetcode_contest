class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        que 
= []
        cmp = 0
        for i in range( len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] in [1, 2]:
                    cmp += 1
                if grid[i][j] == 2:
                    que.append((i, j))
                    grid[i][j] = 0
        if not que:
            return 0 if not cmp else -1
        count = 0      
        rnd = 0
        while que:
            rnd += 1
            count += len(que)
            new_que = []
            for i, j in que:
                for x, y in [[i + 1, j],[i - 1, j],[i, j + 1], [i, j - 1]]:
                    if x in range(len(grid)) and y in range(len(grid[0])) and grid[x][y]:
                        new_que.append((x, y))
                        grid[x][y] = 0
            que = new_que
        return rnd - 1 if count == cmp else -1
