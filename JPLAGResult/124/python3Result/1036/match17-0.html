class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        m = len(grid)
        n = len(grid[0])
        queue = []
        vis = [[False] * n for i in range(m)]
        need = 0
        c = 0
        res = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    need += 1
                elif grid[i][j] == 2:
                    queue.append((i,j))
                    vis[i][j] = True
        while len(queue) > 0 and c != need:
            res += 1
            size = len(queue)
            for i in range(size):
                element = queue.pop(0)
                row = element[0]
                col = element[1]
                if row - 1 >= 0 and not vis[row-1][col] and grid[row-1][col] == 1:
                    vis[row-1][col] = True
                    c += 1
                    queue.append((row-1,col))
                if col - 1 >= 0 and not vis[row][col-1] and grid[row][col-1] == 1:
                    vis[row][col-1] = True
                    c += 1
                    queue.append((row,col-1))
                if row + 1 < m and not vis[row+1][col] and grid[row+1][col] == 1:
                    vis[row+1][col] = True
                    c += 1
                    queue.append((row+1,col))
                if col + 1 < n and not vis[row][col+1] and grid[row][col+1] == 1:
                    vis[row][col+1] = True
                    c += 1
                    queue.append((row,col+1))
        return res if c == need else -1
