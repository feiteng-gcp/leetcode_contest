class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        a = grid
        r = len(a)
        c = len(a[0])

        cnt = 0
        for i in range(r):
            for j in range(c):
                if a[i][j] == 1:
                    cnt += 1
        if cnt == 0:
            return 0

        ans = 0
        while True:
            diff = 0
            for i in range(r):
                for j in range(c):
                    if a[i][j] != 1:
                        continue

                    if any([
                        i > 0 and a[i - 1][j] == 2,
                        j > 0 and a[i][j - 1] == 2,
                        i < r - 1 and a[i + 1][j] == 2,
                        j < c - 1 and a[i][j + 1] == 2
                        ]):
                        a[i][j] = 3

            for i in range(r):
                for j in range(c):
                    if a[i][j] == 3:
                        a[i][j] = 2
                        cnt -= 1
                        diff += 1

            ans += 1
            if cnt == 0:
                return ans
            if diff == 0:
                return -1
