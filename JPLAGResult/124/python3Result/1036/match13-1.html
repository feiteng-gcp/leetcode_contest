class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
#         flag = True
#         for i in range(len(grid)):
#             for j in range(len(grid[0])):
#                 if grid[i][j] == 1:
#                     flag = False
#                     break
        
#         # no fresh orange
#         if flag:
#             return 0
        
        changed = True
        minute = 0
        
        
        while changed:
            changed = False
            new_grid = [[grid[i][j] for j in range(len(grid[0]))] for i in range(len(grid))]
            
            for i in range(len(grid)):
                for j in range(len(grid[0])):
                    if grid[i][j] == 2:
                        for x, y in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
                            if 0 <= x < len(grid) and 0 <= y < len(grid[0]) and new_grid[x][y] == 1:
                                changed = True
                                new_grid[x][y] = 2
            
            minute += 1
            grid 
= new_grid
            
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    return -1
            
        return minute - 1
