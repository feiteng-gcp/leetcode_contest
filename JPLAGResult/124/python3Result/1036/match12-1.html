from collections import deque

class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        N=len(grid)
        M=len(grid[0])

        QUE=deque()

        for i in range(N):
            for j in range(M):
                if grid[i][j]==2:
                    QUE.append((i,j))

        NOW=0
        for NOW in range(100):
            flag=0
            for i in range(N):
                for j in range(M):
                    if grid[i][j]==1:
                        flag=1
                        break
                if flag:
                    break
            else:
                return NOW

            NQUE=deque()

            while QUE:
                x,y=QUE.pop()
                for i,j in [(x+1,y),(x-1,y),(x,y+1),(
x,y-1)]:
                    if 0<=i<N and 0<=j<M and grid[i][j]==1:
                        NQUE.append((i,j))
                        grid[i][j]=2

            QUE=NQUE

        

        else:
            return -1
