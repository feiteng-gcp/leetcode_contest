class Solution:
    def orangesRotting(self, grid: 'List[List[int]]') -> 'int':
        row = len(grid)
        col = len(grid[0])
        k = True
        def f(x,y,n):
            if x < 0 or x >= row or y < 0 or y >= col:
                return
            if grid[x][y] == 1:
                grid[x][y] = n
        for i in range(2,70):
            k 
= True
            for x in range(row):
                for y in range(col):
                    if grid[x][y] == 1:
                        k = False
            if k:
                return i-2
            for x in range(row):
                for y in range(col):
                    if grid[x][y] == i:
                        f(x+1,y,i+1)
                        f(x,y+1,i+1)
                        f(x-1,y,i+1)
                        f(x,y-1,i+1)
        return -1
