class Solution {
    int res = 0;

    public int numSquarefulPerms(int[] A) {
        int N = A.length;
        Arrays.sort(A);
        boolean[][] table = new boolean[N][N];
        for (int i = 0; i < N; i++) {
            for (int j = i + 1; j < N; j++) {
                if (isSquare(A[i] + A[j])) {
                    table[i][j] = true;
                    table[j][i] = true;
                }
            }
        }
        res = 0;
        boolean[] visited = new boolean[N];
        for (int i = 0; i < N; i++) {
            if (i > 0 && A[i] == A[i - 1]) {
                continue;
            }
            visited[i] = true;
            dfs(A, visited, table, 1, i);
            visited[i] = false;
        }
        return res;
    }

    void dfs(int[] A, boolean[] visited, boolean[][] table, int i, int pi) {
        if (i == A.length) {
            res++;
        }
        int N = A.length;
        int p = -1;
        for (int j = 0; j < N; j++) {
            if (visited[j]) {
                continue;
            }
            if (p >= 0 && A[j] == p) {
                continue;
            }
            if (table[pi][j]) {
                visited[j] = true;
                p = A[j];
                dfs(A, visited, table, i + 1, j);
                visited[j] = false;
            }
        }
    }

    boolean isSquare(int x) {
        int r = (int) Math.sqrt(x);
        for (int i = r - 1; i <= r + 1; i++) {
            if (i * i == x) {
                return true;
            }
        }
        return false;
    }
}
