class Solution {
    Set<List<Integer>> set = new HashSet<>();
    
    public int numSquarefulPerms(int[] A) {
        boolean[] visited = new boolean[A.length];
        Arrays.sort(A);
        dfs(0, A, new ArrayList<>(), visited);
        return set.size();
    }
    
    private void dfs(int index, 
                     int[] A,
                     List<Integer> buffer,
                     boolean[] visited) {
        if (index == A.length) {
            set.add(new ArrayList<>(buffer));
        }
        
        for (int i = 0; i < A.length; i++) {
            if (visited[i] == true) continue;
            if (i > 0 && A[i] == A[i - 1] && visited[i - 1] == false) continue;
            if (buffer.size() == 0) {
                buffer.add(A[i]);
                visited[i] = true;
                dfs(index + 1, A, buffer, visited);
                visited[i] = false;
                buffer.remove(buffer.size() - 1);
            } else {
                if (visited[i] == true) continue;
                if (valid(A[i] + buffer.get(buffer.size() - 1))) {
                    buffer.add(A[i]);
                    visited[i] = true;
                    dfs(index + 1, A, buffer, visited);
                    visited[i] = false;
                    buffer.remove(buffer.size() - 1);
                }
            }
        }
    }
    
    private boolean valid(int A) {
        double sr = Math.sqrt(A); 
        return ((sr - Math.floor(sr)) == 0); 
    }
}
