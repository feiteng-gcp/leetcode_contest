class Solution {
    int count = 0;
    public int numSquarefulPerms(int[] A) {
        int n = A.length;
        if(n < 2) return 0;
        Arrays.sort(A);
        dfs(A, 0, -1, new boolean[n], n);
        return count;
    }
    private void dfs(int[] A, 
int nums,int last, boolean[] visited, int n){
        if(nums == n){
            count++;
            return;
        }
        for(int i = 0; i < n; i++){
            if(visited[i]) continue;
            if(i > 0 && A[i] == A[i-1] && !visited[i-1]) continue; 
            if(last == -1){
                visited[i] = true;
                dfs(A, nums + 1, A[i], visited, n);
                visited[i] = false;
            }else{
                int sum = last + A[i];
                int temp = (int)Math.sqrt(sum);
                if(temp * temp != sum) continue;
                //System.out.println(temp);
                visited[i] = true;
                dfs(A, nums + 1, A[i], visited, n);
                visited[i] = false;
            }
            
        }
    }
}
