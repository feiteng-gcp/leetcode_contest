class Solution {
    private int ans=0;
    public int numSquarefulPerms(int[] A) {
        
HashMap<Integer,List<Integer>> hmap = new HashMap<>();
        for(int i=0;i<A.length;++i){
            if(hmap.containsKey(A[i])==false){
                hmap.put(A[i],new ArrayList<>());
            }
            for(int j=0;j<A.length;++j){
                if(j==i)
                    continue;
                if(isSq(A[i]+A[j])&&!hmap.get(A[i]).contains(A[j]))
{
                    hmap.get(A[i]).add(A[j]);
                }
            }
        }
        boolean[] used = new boolean[A.length];
        for(int a:hmap.keySet()){
            for(int i=0;i<A.length;++i){
                if(A[i]==a){
                    used[i]=true;
                    combin(A,used,hmap,1,a);
                    used[i]=false;
                    break;
                }
            }
        
}
        return ans;
    }
    private void combin(int[] A,boolean[] used,HashMap<Integer,List<Integer>> hmap,int p,int prv){
        if(p==A.length){
            ++ans;
            return;
        }
        List<Integer> ls = hmap.get(prv);
        for(int i=0;i<ls.size();++i){
            for(int j=0;j<A.length;++j){
                if(A[j]==ls.get(i)&&used[j]==false){
                    used[j]=true;
                    combin(A,used,hmap,p+1,ls.get(i));
                    used[j]=false;
                    break;
                }
            }
        }
        return;
    }
    private boolean isSq(int n){
        int sqn = (int)Math.sqrt(n);
        if(sqn*sqn==n)
            return true;
        return false;
    }
}
