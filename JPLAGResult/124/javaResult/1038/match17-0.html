class Solution {
    int res = 0;
    List<Integer> curP = new ArrayList();
    Set<String> set = new HashSet();

    void bdfs(int[] A,List<Integer>[] graph,List<Integer> curP,int i, boolean[] hasVisited,int curVisitedNum ){
        if(hasVisited[i] == true || (i>0 && A[i-1] == A[i] && hasVisited[i-1] == false))
            return;
        
        hasVisited[i] = true;
        curP.add(i);
        curVisitedNum++;
        if(curVisitedNum == hasVisited.length){
            hasVisited[i] = false;
            curP.remove(curP.size()-1);
            StringBuilder sb = new StringBuilder();
            for(int c : curP){
                sb.append('#');
                sb.append(A[c]);
            }
            String ssb = sb.toString();
            if(!set.contains(ssb)){
                res++;
                set.add(ssb);
            }
            return;
        }
        for(int ni : graph[i])
            bdfs(A,graph,curP,ni,hasVisited,curVisitedNum);
        hasVisited[i] = false;
        curP.remove(curP.size()-1);

    }
    public int numSquarefulPerms(int[] A) {
        List<Integer>[] graph = new List[A.length];
        Arrays.sort(A);
        for(int i = 0 ; i < graph.length ; i++)
            graph[i] = new LinkedList();
        for(int i = 0 ;i < A.length; i++)
            for(int j  = i + 1; j < A.length ; j++){
                double cSqrt = Math.sqrt(A[i] + A[j]);
                if(cSqrt == ((int)cSqrt)){
                    graph[i].add(j);
                    graph[j].add(i);
                }
            }
        boolean[] hasVisited = new boolean[A.length];
        for(int i = 0; i < A.length ; i++){
            if(graph[i].size()!=0){
                bdfs(A,graph,curP,i,hasVisited,0);
            }
        }
        return res;
    }
}
