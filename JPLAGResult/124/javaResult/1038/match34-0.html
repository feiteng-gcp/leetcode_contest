class Solution {
    public int numSquarefulPerms(int[] A) {
        // length < 12 -> brute force
        Arrays.sort(A);
        dfs(A, new int[A.length], 0, 0);
        return re;
    }
    
    void dfs(int[] a, int[] b, int visited, int pos)
    {
        if(visited == (1 << a.length) - 1)
        {
            // check for condition
            // if(check(b)) 
            re++;
            return;
        }
        // guarantee every step is square
        for(int i = 0; i < a.length; i++)
        {
            if((visited & (1 << i)) != 0) continue;
            if(i > 0 && a[i] == a[i - 1] && (visited & (1 << (i - 1))) == 0) continue;
            if(pos > 0 && !isSquare(a[i] + b[pos - 1])) continue;
            visited |= 1 << i;
            b[pos] = a[i];
            dfs(a, b, visited, pos + 1);
            visited &= ~(1 << i);
        }
    }
    
    boolean check(int[] a)
    {
        int n = a.length;
        for(int i = 0; i < n - 1; i++)
        {
            int s = a[i] + a[i + 1];
            if(!isSquare(s)) return false;
        }
        return true;
    }
    
    boolean isSquare(int k)
    {
        int sqrt = (int) Math.sqrt(k);
        return sqrt * sqrt == k;
    }
    
    int re = 0;
}
