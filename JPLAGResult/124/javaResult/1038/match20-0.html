class Solution {
    int count = 0;
    public int numSquarefulPerms(int[] A) {
        if (A.length == 0) {
            return 0;
        }
        Arrays.sort(A);
        boolean[] used = new boolean[A.length];
        backtracking(A, used, new ArrayList<>());
        return count;
    }
    private void backtracking(int[] A, boolean[] used, List<Integer> list) {
        if (list.size() == A.length) {
            count ++;
            return;
        }
        for (int i = 0; i < A.length; i ++) {
            if (used[i] || (i - 1 >= 0 && A[i] == A[i - 1] && !used[i - 1])) continue;
            if (list.size() == 0 || isSquare(list.get(list.size() - 1) + A[i])) {
                list.add(A[i]);
                used[i] = true;
                backtracking(A, used, list);
                list.remove(list.size() - 1);
                used[i] = false;
            }
        }
    }
    private boolean isSquare(int x) {
        double r = Math.sqrt(x);
        if ((r - Math.floor(r)) == 0) {
            return true;
        }
        return false;
    }
}
