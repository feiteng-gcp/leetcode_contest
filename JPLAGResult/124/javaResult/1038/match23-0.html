class Solution {
    
    private int ans;
    
    public int numSquarefulPerms(int[] A) {
        Arrays.sort(A);
        for (int i = 0; i < A.length; i++) {
            if (i > 0 && A[i] == A[i - 1]) continue;
            List<Integer> list = new ArrayList<>();
            Set<Integer> set = new HashSet<>();
            list.add(A[i]);
            set.add(i);
            dfs(A, list, set);
        }
        return ans;
    }
    
    private void dfs(int[] A, List<Integer> list, Set<Integer> set) {
        if (set.size() == A.length) {
            ans++;
            return;
        }
        
        Set<Integer> v = new HashSet<>();
        for (int i = 0; i < A.length; i++) {
            if (set.contains(i)) continue;
            if (v.contains(A[i])) continue;
            if (helper(A[i], list.get(list.size() - 1))) {
                v.add(A[i]);
                set.add(i);
                list.add(A[i]);
                dfs(A, list, set);
                set.remove(i);
                list.remove(list.size() - 1);
            }
        }
    }
    
    private boolean helper(int a, int b) {
        long sum = (long) a + (long) b;
        return (long) Math.sqrt(sum) * (long) Math.sqrt(sum) == sum;
    }
}
