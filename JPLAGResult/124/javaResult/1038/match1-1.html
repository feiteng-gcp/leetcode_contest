class Solution {
    public int numSquarefulPerms(int[] A) {
        Arrays.sort(A);
        size = A.length;
      //  System.out.println(isPerfect(1, 8));
        
DFS(new ArrayList<>(), A, new HashSet<Integer>());
        return count;
    }
    
    int count;
    HashSet<String> checked = new HashSet<>();
    int size = 0;
    
    private void DFS(
List<Integer> result, int[] A, HashSet<Integer> used ){
        if(result.size() == size){
            String key = getCode(result);
            if( checked.add(key) ){
                 count++;
            }
            return;
        }     
        
        if(used.size() == A.length) return;
        
        
for(int i = 0; i < A.length; i++){
            if(used.contains(i)) continue;
            if(result.isEmpty() || isPerfect( result.get(result.size() - 1), A[i]  ) ){
                used.add(i);
                result.add(A[i]);
                DFS(result, A, used);
                result.remove(result.size() - 1);
                used.remove( i ) ;
            }
            while(i + 1 < A.length && A[i] == A[i+1]) i++;
            
        
}
        
    }
    
    private String getCode(List<Integer> result){
        StringBuilder sb = new StringBuilder();
        for(int i : result){
            sb.append(i).append("#");
        
}
        return sb.toString();
    }
    private boolean isPerfect(int a, int b){
        for(long j = 0; j * j <= 0L + a + b; j++){
            if(j*j == 0L + a + b) return true;
        }
        return false;
    }
}
