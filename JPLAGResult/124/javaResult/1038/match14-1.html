class Solution {
    public int numSquarefulPerms(int[] A) {
        
Map<Integer, List<Integer>> map = new HashMap<>();
        Map<Integer, Integer> count = new HashMap<>();
        for (int a : A) {
            if (!count.containsKey(a)) {
                count.put(a, 1);
                map.put(a, new ArrayList<>());
            } else {
                count.put(a, count.get(a) + 1);
            }
        }
        for (int a : count.keySet()) {
            for (int b : count.keySet()) {
                long sum = a + b, sqrt = (long)Math.sqrt(sum);
                if (sqrt * sqrt == sum) {
                    
map.get(a).add(b);
                }
            }
        }
        int sum = 0;
        for (int a : count.keySet()) {
            int val = 
count.get(a);
            count.put(a, val - 1);
            sum += dfs(map, count, 1, a, A.length);
            count.put(a, val);
        }
        return sum;
    }
    
    private int dfs(Map<Integer, List<Integer>> map, Map<Integer, Integer> count, int curIndex, int prevVal, int len) {
        if (curIndex == len) return 1;
        int sum = 0;
        for (int valid : map.get(prevVal)) {
            int remain = count.get(valid);
            if (remain > 0) {
                
count.put(valid, remain - 1);
                sum += dfs(map, count, curIndex + 1, valid, len);
                count.put(valid, remain);
            }
        }
        return sum;
    }
}
