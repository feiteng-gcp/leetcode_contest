class Solution {
    int result;
    public int numSquarefulPerms(int[] nums) {
        if (nums == null || nums.length == 0) {
            return 0;
        }
        
Map<Integer, Integer> timesMap = new HashMap<>();
        Map<Integer, Set<Integer>> neighborsMap = new HashMap();
        
        for (int i : nums) {
            if (!timesMap.containsKey(i)) {
                timesMap.put(i, 1);
            } else {
                timesMap.put(i, timesMap.get(i) + 1);
            }
            neighborsMap.put(i, new HashSet<Integer>());
        }
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (j == i) {
                    continue;
                }
                if (isPair(nums[i], nums[j])) 
{
                    neighborsMap.get(nums[i]).add(nums[j]);
                }
            }
        }
        for (Map.Entry<Integer, Integer> entry : timesMap.entrySet()) {
            int key = entry.getKey();
            timesMap.put(key, 
timesMap.get(key) - 1);
            helper(nums, key, timesMap, neighborsMap, nums.length - 1);
            timesMap.put(key, timesMap.get(key) + 1);
        }
        return result;
    }
    public void helper(int[] nums, int lastNum, Map<Integer, Integer> timesMap, 
                       Map<Integer, Set<Integer>> neighborsMap, int leftNum) {
        if (leftNum == 0) {
            result++;
            return;
        }
        for (Map.Entry<Integer, Integer> entry : timesMap.entrySet()) {
            int nextNum = entry.getKey();
            if (timesMap.get(nextNum) != 0 && isPair(lastNum, nextNum)) {
                timesMap.put(nextNum, timesMap.get(nextNum) - 1);
                helper(nums, nextNum, timesMap, neighborsMap, leftNum - 1);
                timesMap.put(nextNum, timesMap.get(nextNum) + 1);
            }
        }
    }
    public boolean isPair(int i1, int i2) {
        int sum = i1 + i2;
        double tmp = Math.sqrt((double)sum);
        double floor = Math.floor(tmp);
        if (Math.pow(floor, 2) == Math.pow(tmp, 2)) {
            return true;
        }
        return false;
    }
}
