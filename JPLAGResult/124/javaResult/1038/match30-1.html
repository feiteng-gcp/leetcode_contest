class Solution {
    int count;
    public int numSquarefulPerms(int[] A) {
        List<Integer>[] pairs = new List[A.length];
        for (int i = 0; i < A.length; i++) {
            List<Integer> tmp = new ArrayList<>();
            pairs[i] = tmp;
            for (int j = 0; j < A.length; j++) {
                if (j == i) continue;
                int sum = A[i] + A[j];
                int sq = (int)Math.sqrt(sum);
                if (sq * sq == sum) {
                    tmp.add(j);
                }
            }
        }
        
count = 0;
        Set<Integer> used = new HashSet<>();
        Set<Integer> seen = new HashSet<>();
        for (int i = 0; i < A.length; i++) {
            if (seen.add(A[i])) {
                used.add(i);
                dfs(A, i, 1, pairs, used);
                used.remove(i);
            }
        
}
        return count;
    }
    
    private void dfs(int[] A, int prev, int idx, List<Integer>[] pairs, Set<Integer> used) {
        //System.out.println(prev + " at " + idx);
        if (idx == A.length) {
            count++;
        }
        Set<Integer> seen = new HashSet<>();
        for (int p : pairs[prev]) {
            if (!used.contains(p) && !seen.contains(A[p])) {
                
used.add(p);
                seen.add(A[p]);
                dfs(A, p, idx + 1, pairs, used);
                used.remove(p);
            }
        }
    }
}
