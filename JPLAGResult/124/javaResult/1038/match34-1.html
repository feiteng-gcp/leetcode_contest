class Solution {
    List<List<Integer>> res = new ArrayList<List<Integer>> ();
    public int numSquarefulPerms(int[] A) {
        int d = 0;
         for (int i = 0; i < A.length - 1; i++) {
             for (int j = i; j < A.length - 1; j++) {
                if (isSquare((A[j] + A[j + 1]))) d ++;
             }
         }
        if (d < A.length - 1) return 0;
        Arrays.sort(A);
        helper(A, new ArrayList<Integer> (), new boolean[A.length]);
        int count = 0;
        for (List<Integer> li : res) {
            int num = 0;
            for (int i = 0; i < li.size() - 1; i++) {
                if (isSquare((li.get(i) + li.get(i + 1)))) num ++;
            }
            if (num == li.size() - 1) count++
;
        }
        return count;
    }
    public boolean isSquare(int m) {
        Double d = Double.valueOf(m);
        double i = Math.sqrt(d);
        return i % 1 == 0.0;
    }
    public void helper(int[] nums, List<Integer> list, boolean[] used) {
        if (list.size() == nums.length) {
            res.add(new ArrayList<Integer> (list));
            
return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (used[i]) continue;
            if (i > 0 && nums[i - 1] == nums[i] && !used[i - 1]) continue;
            list.add(nums[i]);
            used[i] = true;
            helper(nums, list, used);
            used[i] = false;
            list.remove(list.size() - 1);
        }
    }
}
