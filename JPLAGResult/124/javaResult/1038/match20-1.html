class Solution {
    int count = 0;
    public int numSquarefulPerms(int[] A) {
        int n = A.length;
        if (n == 1) {
            return isSquare(A[0]) ? 1 : 0;
        
}
        Arrays.sort(A);
        boolean[] taken = new boolean[n];
        dfs(A, taken, new ArrayList<Integer>());
        return count;
    }
    
    private void dfs(int[] A, boolean[] taken, List<Integer> list) {
        if (list.size() == A.length) {
            count++;
        }
        for (int i=0; i<A.length; i++) {
            if (i!= 0 && A[i] == A[i-1] && !taken[i-1]) {
                continue;
            }
            if (!taken[i]) {
                if (list.size() == 0) {
                    list.add(A[i]);
                    taken[i] = true;
                    dfs(A, taken, list);
                    taken[i] = false;
                    list.remove(list.size()-1);
                } else {
                    int sum = list.get(list.size()-1) + A[i];
                    if (isSquare(sum)) {
                        list.add(A[i]);
                        taken[i] = true;
                        dfs(A, taken, list);
                        taken[i] = false;
                        list.remove(list.size()-1);
                    }
                }
            
}
        }
    }
    
    private boolean isSquare(int num) {
        int sqrt = (int)(Math.sqrt(num));
        return (sqrt*sqrt==num);
    }
}
