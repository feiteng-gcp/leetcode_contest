class Solution {
    Map<Integer, Set<Integer>> per = new HashMap<>();
    Map<Integer, Integer> counts = new HashMap<>();
    Set<Long> perfects = new HashSet<>();
    Set<String> ans = new HashSet<>();
    Set<Integer> used = new HashSet<>();
    public int numSquarefulPerms(int[] A) {
        int max1 = -1;
        int max2 = -1;
        int n = A.length;
        for(int i = 0; i < n; i++){
            counts.put(A[i], counts.getOrDefault(A[i], 0) + 1);
            if(A[i] >= max1){
                max2 = max1;
                max1 = A[i];
            }else if(A[i] >= max2){
                max2 = A[i];
            }
        }
        for(long i = 0; i * i <= (long)(max1 + max2 + 1); i++){
            perfects.add(i * i);
        }
        
        // System.out.println(perfects);
        
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(perfects.contains((long)(A[i] + A[j]))){
                    per.computeIfAbsent(A[i], x-> new HashSet<>()).add(A[j]);
                }
            }
        }
        
        for(int i = 0; i < n; i++){
            if(used.add(A[i])){
                counts.put(A[i], counts.get(A[i]) - 1);
                dfs(A[i], A[i] + "", 1, A);
                counts.put(A[i], counts.get(A[i]) + 1);
            }
        }
        
        return ans.size();
    }
    
    public void dfs(int cur, String res, int count, int [] A){
        if(count == A.length){
            ans.add(res);
            return;
        }
        // try each possible perfect
        if(per.containsKey(cur) == false){
            // System.out.println(cur);
            // System.out.println(res);
            // System.out.println("---");
            return;
        }
        for(int n: per.get(cur)){
            if(perfects.contains((long)(n) + (long)(cur)) && counts.get(n) > 0){
                counts.put(n, counts.get(n) - 1);
                dfs(n, res + "" + n + "", count + 1, A);
                counts.put(n, counts.get(n) + 1);
            }
        }
    }
}
