class Solution {
    int res = 0; 
    //Set<String> resStr = new HashSet<>(); 
    int targetLen;
    
    public int numSquarefulPerms(int[] A) {
        targetLen = A.length; 
        Map<Integer, Integer> choiceMap = new HashMap<>();
        for (int x : A) {
            choiceMap.put(x, choiceMap.getOrDefault(x, 0) + 1);
        }
        dfs(0, new ArrayList<>(), choiceMap);
        return res; 
    }
    
    private void dfs(int len, List<Integer> cur, Map<Integer, Integer> choiceMap) {
        if (len == targetLen) {
            res++;
            return;
        }
        Set<Integer> keys = new HashSet<>();
        keys.addAll(choiceMap.keySet());
        for (int key : keys) {
            if (cur.size() == 0 || checkSquare(cur.get(cur.size() - 1) + key)) {
                cur.add(key);
                choiceMap.put(key, choiceMap.get(key) - 1);
                if (choiceMap.get(key) == 0) {
                    choiceMap.remove(key);
                }
                dfs(len + 1, cur, choiceMap); 
                cur.remove(cur.size() - 1); 
                choiceMap.put(key, choiceMap.getOrDefault(key, 0) + 1);
            }
        }
    }
    
    private boolean checkSquare(int input) {
        int sqt = (int)Math.sqrt(input);
        ///System.out.println(input + " : input");
        return sqt * sqt == input; 
    }
}
