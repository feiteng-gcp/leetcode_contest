class Solution {
    boolean sq(int x) {
        int i = (int)(Math.sqrt(x));
        return i * i == x;
    }
    int l;
    public int numSquarefulPerms(int[] A) {
        l = A.length;
        Map<Integer, Set<Integer>> map = new HashMap<>();
        Map<Integer, Integer> ct = new HashMap<>();
        for(int i = 0; i < A.length; i++) {
            if(!ct.containsKey(A[i])) ct.put(A[i], 0);
            ct.put(A[i], ct.get(A[i]) + 1);
            if(!map.containsKey(A[i])) map.put(A[i], new HashSet<>());
            for(int j = i + 1; j < A.length; j++) {
                if(sq(A[i] + A[j])) {
                    if(!map.containsKey(A[i])) map.put(A[i], new HashSet<>());
                    if(!map.containsKey(A[j])) map.put(A[j], new HashSet<>());
                    map.get(A[i]).add(A[j]);
                    map.get(A[j]).add(A[i]);
                }
            }
        }
        int single = 0;
        for(int i: map.keySet()) {
            if(map.get(i).size() == 0) return 0;
            if(map.get(i).size() == 1) {
                for(int j: map.get(i)) {
                    if(ct.get(j) == 1) single++;
                }
            }
            if(single > 2) return 0;
        }
        int re = 0;
        for(int i: map.keySet()) {
            ct.put(i, ct.get(i) - 1);
            re += dfs(i, 1, map, ct);
            ct.put(i, ct.get(i) + 1);
        }
        return re;
    }
    int dfs(int prev, int start, Map<Integer, Set<Integer>> map, Map<Integer, Integer> ct) {
        if(start == l) {
            return 1;   
        }
        int re = 0;
        for(int i: map.keySet()) {
            if(map.get(prev).contains(i) && ct.get(i) > 0) {
                ct.put(i, ct.get(i) - 1);
                re += dfs(i, start+1, map, ct);
                ct.put(i, ct.get(i) + 1);
            }
        }  
        return re; 
    }
}
