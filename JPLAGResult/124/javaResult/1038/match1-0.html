class Solution {
    public int numSquarefulPerms(int[] A) {
        backtrack(A, 0, new HashSet<>(), new ArrayList<>());
        return ans;
    }
    
    int ans = 0;
    Set<String> seen = new HashSet<>();
    private void backtrack(int[] A, int index, Set<Integer> usedIndex, List<Integer> curr) {
        if (usedIndex.size() == A.length) {
            String key = key(curr);
            if (seen.contains(key) == false) {
                ans++;
                seen.add(key);
                
            }
            return;
        }
        
        String key = key(curr);
        if (seen.contains(key)) return;
        seen.add(key);
        
        // System.out.println(curr);
        for(int i = 0; i < A.length; i++) {
            if (usedIndex.contains(i)) continue;
            
            if (curr.size() == 0 || issquare(curr.get(curr.size() - 1), A[i])) {
                curr.add(A[i]);
                usedIndex.add(i);
                backtrack(A, 0, usedIndex, curr);
                curr.remove(curr.size() - 1);
                usedIndex.remove(i);
            }
        }
    }
    
    private String key(List<Integer> cur) {
        StringBuilder sb = new StringBuilder();
        for(int i : cur) {
            sb.append(i + ",");
        }
        return sb.toString();
    }
    
    private boolean issquare(int a, int b) {
        int sqrt = (int) Math.sqrt(a + b);
        if (a + b == sqrt * sqrt) {
            // System.out.println("is square " + a + "  " + b);
            return true;
        }
        return false;
    }
}
