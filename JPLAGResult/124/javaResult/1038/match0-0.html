class Solution {
    ArrayList<Integer> g[] = null;
int[] AA;
    int l = 0;
    
    Set<List<Integer>> r = new HashSet<>();
    
    
    boolean ok(int a,int b){
        int s = a+b;
        int n = (int)Math.sqrt(s);
        return n*n==s;
    }
    
    void dfs(int c,int[] vis,int n,List<Integer> m){
        if(n==l){
            if(!r.contains(m)){
                
                List<Integer> mm = new ArrayList<>(m);
                r.add(mm);
            }
            return;
        }
        
        Set<Integer> sm = new HashSet<>();
        
        for(Object go1:g[c]){
            Integer go  =(Integer)go1;
            
            
            if(vis[go]==0){
                if(!sm.contains(AA[go])){
                sm.add(AA[go]);
                vis[go] = 1;
                m.add(AA[go]);
                dfs(go,vis,n+1,m);
                vis[go] = 0;
                m.remove(m.size()-1);
            }
            }
        }
        
        
    }
    
    public int numSquarefulPerms(int[] A) {
        AA = A;
        l = A.length;
        g = new ArrayList[l];
        
        for(int i=0;i<l;++i){
            
         g[i] = new ArrayList();   
            
        }
        
        for(int i=0;i<l;++i){
            for(int j=i+1;j<l;++j){
                if(ok(A[i],A[j])){
                    g[i].add(j);
                    g[j].add(i);
                }   
            }
        }
        
        Set<Integer> sm = new HashSet<>();
        for(int i=0;i<l;++i){
            
            if(!sm.contains(A[i])){
            sm.add(A[i]);
            int vis[] = new int[l];
            List<Integer> m = new ArrayList<>();
            m.add(A[i]);
            vis[i] = 1;
            dfs(i,vis,1,m);
            }
            
        }
        return r.size();
        
        
        
        
        
        
        
        
        
    }
}
