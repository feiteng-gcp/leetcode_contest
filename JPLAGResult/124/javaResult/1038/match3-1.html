class Solution {
    int count = 0;
    public int numSquarefulPerms(int[] A) {
        Arrays.sort(A);
        generate(A, new boolean[A.length], new ArrayList<>());
        
return count;
    }

    private void generate(int[] nums, boolean[] visited, List<Integer> list) {
        if (list.size() == nums.length) {
            count++;
            
return;
        }

        for (int i = 0; i < nums.length; i++) {
            if (visited[i]) continue;
            if (i > 0 && nums[i] == nums[i - 1] && !visited[i - 1]) continue;

            if (list.size() > 0 && !isSquare(nums[i] + list.get(list.size() - 1))) continue;

            
list.add(nums[i]);
            visited[i] = true;
            generate(nums, visited, list);
            visited[i] = false;
            list.remove(list.size() - 1);
        }
    }

    private boolean isSquare(int num) {
        int s = (int) Math.sqrt(num);
        return s * s == num;
    }
}
