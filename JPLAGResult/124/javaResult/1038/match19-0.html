public class Solution {
    private int res;
    private Set<Integer> square;

    public int numSquarefulPerms(int[] A) {
        res = 0;
        Map<Integer, Integer> avail = new HashMap<>();
        Set<Integer> set = new HashSet<>();
        square = new HashSet<>();

        for (int i = 0; i < Math.sqrt(Math.pow(10, 9) * 2 + 2) + 1; i++) {
            square.add(i * i);
        }

        for (int i : A) {
            avail.put(i, avail.getOrDefault(i, 0) + 1);
            set.add(i);
        }

        int res = 0;
        for (int s : set) {
            avail.put(s, avail.get(s) - 1);
            res += helper(set, s, avail, 1, A.length);
            avail.put(s, avail.get(s) + 1);

        }
        return res;
    }

    private int helper(Set<Integer> set, int start,Map<Integer, Integer> avail, int cnt, int n) {
        if (n == cnt) {
            return 1;
        }
        int res = 0;
        for (int next : set) {
//            System.out.println(avail[1] +"---" + avail[8] + "---" + avail[17]);
            if (avail.get(next) > 0) {
                int sum = start + next;
                if (square.contains(sum)) {
                    avail.put(next, avail.get(next) - 1);
//                    System.out.println(start + "\t" + next + "\t" + cnt);
                    res += helper(set, next, avail, cnt + 1, n);
                    avail.put(next, avail.get(next) + 1);
                }
            }

        }
        return res;
    }
}
