class Solution {
    int res = 0;
    public int numSquarefulPerms(int[] A) {
        Arrays.sort(A);
        boolean []visit = new boolean[A.length];
        dfsHelper(visit, null, 0, A);
        
        return res;
    }

    private void dfsHelper(boolean[] visit, Integer lastInteger, int count, int[]A) {
        if (A.length == count) {
            res++;
        }
        else {
            for (int i = 0; i < A.length; i++) {
                if (i != 0 && A[i] == A[i-1] && !visit[i-1]) {
                    continue;
                }
                
                if (!visit[i])
                {
                    if (isPerfectSquare(lastInteger, A[i])) {
                        visit[i] = true;
                        dfsHelper(visit, A[i], count + 1, A);
                        visit[i] = false;
                    }
                }
            }
        }
    }

    private boolean isPerfectSquare(Integer lastInteger, int i) {
        if (lastInteger == null) return true;
        int sum = lastInteger + i;
        int root = (int)Math.sqrt(sum);
        return root * root == sum;
    }
}
