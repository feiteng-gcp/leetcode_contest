class Solution {
    public int numSquarefulPerms(int[] A) {
        int n = A.length;
        boolean[][] connected = new boolean[n][n];
        Map<Integer, Set<Integer>> edge = new HashMap<>();
        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 1; j < n; j++) {
                int result = A[i] + A[j];
                int sqrt = (int) Math.sqrt(result);
                if (sqrt * sqrt == result) {
                    connected[i][j] = true;
                    connected[j][i] = true;
                    if (!edge.containsKey(i)) {
                        edge.put(i, new HashSet<>());
                    }
                    if (!edge.containsKey(j)) {
                        edge.put(j, new HashSet<>());
                    }
                    edge.get(i).add(j);
                    edge.get(j).add(i);
                }
            }
        }
        
Set<Integer> used = new HashSet<>();
        Set<Integer> tried = new HashSet<>();
        Set<String> perm = new HashSet<>();
        StringBuilder sb = new StringBuilder();
        int[] count = new int[1];
        for (int i = 0; i < A.length; i++) {
            if (tried.contains(A[i])) {
                continue;
            }
            used.add(i);
            tried.add(A[i]);
            sb.append(A[i]);
            search(A, edge, count, used, i, perm, sb);
            sb = new StringBuilder();
            used.remove(i);
        
}
        return perm.size();
    }
    
    public void search(int[] A, Map<Integer, Set<Integer>> edge, int[] count, Set<Integer> used, int last, Set<String> perm, StringBuilder sb) {
        if (used.size() == A.length) {
            //System.out.println(sb.toString());
            perm.add(
sb.toString());
            return;
        }
        if (!edge.containsKey(last)) {
            return;
        }
        Set<Integer> tried = new HashSet<>();
        for (int next : edge.get(last)) {
            if (!used.contains(next) && !tried.contains(A[next])) {
                used.add(next);
                tried.add(A[next]);
                
                int len = sb.length();
                
sb.append(',');
                sb.append(A[next]);
                search(A, edge, count, used, next, perm, sb);
                sb.setLength(len);
                used.remove(next);
            }
        }
    }
}
