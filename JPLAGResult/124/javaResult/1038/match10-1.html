class Solution {
    public int numSquarefulPerms(int[] A) {
        Set<List<Integer>> res = new HashSet<>();
        List<Integer> path = new ArrayList<>();
        Arrays.sort(A);
        boolean[] used = new boolean[A.length];
        helper(res, path, A, used);
        
return res.size();
    }
    
    private void helper(Set<List<Integer>> res, List<Integer> path, int[] A, boolean[] used) {
        if(path.size() == A.length) {
            res.add(new ArrayList<Integer>(path));
            return;
        }
        for(int i=0; i<A.length; i++) {
            if(i-1 >=0 && A[i-1] == A[i] && used[i-1]) {
                continue;
                    // let say there are 3 aaa, we only allow 321, not 123 or 213, this way we avoid duplicates
            }
            if(!used[i]) {
                if(path.size() == 0 || 
isSquare(A[i]+path.get(path.size()-1))) {
                    path.add(A[i]);
                    used[i] = true;
                    helper(res, path, A, used);
                    used[i] = false;
                    path.remove(path.size()-1);
                }
            }
        }
    }
    
    private boolean isSquare(int number) {
     int sqrt = (int) Math.sqrt(number);
     if(sqrt*sqrt == number) {
        return true;
     }
     return false;
    }


}
