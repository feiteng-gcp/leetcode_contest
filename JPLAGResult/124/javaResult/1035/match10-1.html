/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private TreeNode px, py;
    private int l1, l2;
    
    public boolean isCousins(TreeNode root, int x, int y) {
        searchHelper(root, null, 0, x, y);
        if (px == null || py == null) {
            return false;
        
}
        return l1 == l2 && px != py;
    }
    
    private void searchHelper(TreeNode current, TreeNode parent, int level, int x, int y) {
        if (current == null) {
            return;
        }
        if (px != null && py != null) {
            return;
        
}
        
        if (current.val == x) {
            px = parent;
            l1 = level;
        }
        if (current.val == y) {
            py = parent;
            l2 = level;
        }
        
        searchHelper(current.left, current, level + 1, x, y);
        searchHelper(current.right, current, level + 1, x, y);
    }
}
