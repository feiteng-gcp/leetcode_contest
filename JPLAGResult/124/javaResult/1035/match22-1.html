/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    int xx = 0;
    int yy = 0;
    
int dx = -1;
    int dy = 0;
    int fx = -1;
    int fy = 0;
    
    public boolean isCousins(TreeNode root, int x, int y) {
        xx = x;
        yy = y;
        g(root,-1,0);
        return dx==dy&&fx!=fy;
    }
    
    public void g(
TreeNode root , int fa , int d){
        if(root==null) return;
        if(root.val == xx){
            dx = d;
            fx = fa;
        }else if(root.val == yy){
            dy = d;
            fy = fa;
        }
        
        g(root.left,root.val,d+1);
        g(root.right,root.val,d+1);
        
    }
}
