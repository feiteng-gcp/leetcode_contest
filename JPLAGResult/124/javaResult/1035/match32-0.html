/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int p[] = new int[101];
    int L[] = new int[101];
    void dfs(TreeNode root, int up, int lvl) {
        if (root == null) {
            return;
        }
        p[root.val] = up;
        L[root.val] = lvl;
        dfs(root.left, root.val, lvl + 1);
        dfs(root.right, root.val, lvl + 1);
    }

    public boolean isCousins(TreeNode root, int x, int y) {
        dfs(root, -1, 0);
        if (L[x] == L[y] && p[x] != p[y]) {
            return true;
        }
        return false;
    }
}
