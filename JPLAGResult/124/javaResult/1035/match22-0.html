/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int xd = -1, yd = -1;
    TreeNode prex = null, prey = null;
    public boolean isCousins(TreeNode root, int x, int y) {
        dfs(root, x, y, 0, null);
        if (xd != -1 && xd == yd && prex != null && prey != null && prex != prey)
            return true;
        return false;
    }
    public void dfs(TreeNode cur, int x, int y, int depth, TreeNode pre) {
        if (cur == null)
            return;
        if (cur.val == x) {
            xd = depth;
            prex = pre;
        }else if (cur.val == y) {
            yd = depth;
            prey = pre;
        }
        dfs(cur.left, x, y, depth + 1, cur);
        dfs(cur.right, x, y, depth + 1, cur);
    }
}
