/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int p1 = 0, l1 = 0;
    int p2 = 0, l2 = 0;
    
    public boolean isCousins(TreeNode root, int x, int y) {
        help(root, 0, x, y, 0);
        if (l1 == l2 && p1 != p2) 
            return true;
        return false;
    }
    
    private void help(TreeNode node, int parent, int x, int y, int level) {
        if (node == null) return;
        
        if (node.val == x) {
            p1 = parent;
            l1 = level;
        }
        if (node.val == y) {
            p2 = parent;
            l2 = level;
        }
        
        help(node.left, node.val, x, y, level + 1);
        help(node.right, node.val, x, y, level + 1);
    }
}
