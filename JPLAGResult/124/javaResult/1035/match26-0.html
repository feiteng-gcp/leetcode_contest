/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCousins(TreeNode root, int x, int y) {
        if (root == null)
            return false;
        if (root.val == x || root.val == y)
            return false;
        TreeNode p1 = getParent(root, x), p2 = getParent(root, y);
        if (p1 == null || p2 == null || p1.val == p2.val)
            return false;
        int d1 = getDepth(root, 0, x), d2 = getDepth(root, 0, y);
        if (d1 != 101 && d2 != 101 && d1 == d2)
            return true;
        else
            return false;
    }
    
    public int getDepth(TreeNode root, int level, int x) {
        if (root == null)
            return 101;
        if (root.val == x)
            return level;
        int res = getDepth(root.left, level + 1, x);
        if (res != 101)
            return res;
        return getDepth(root.right, level + 1, x);
    }
    
    public TreeNode getParent(TreeNode root, int x) {
        if (root == null)
            return null;
        if (root.left != null && root.left.val == x)
            return root;
        if (root.right != null && root.right.val == x)
            return root;
        TreeNode res = getParent(root.left, x);
        if (res != null)
            return res;
        return getParent(root.right, x);
    }
}
