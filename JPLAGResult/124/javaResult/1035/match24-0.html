/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCousins(TreeNode root, int x, int y) {
        int d1 = getDepth(root, x, 0);
        int d2 = getDepth(root, y, 0);
        if(d1 != d2){
            return false;
        }
        TreeNode left = find(root, x);
        TreeNode right = find(root, y);
        if(left != right){
            return true;
        }
        return false;
    }
    
    public int getDepth(TreeNode node, int x, int depth){
        if(node == null){
            return -1;
        }
        if(node.val == x){
            return depth;
        }
        int left = getDepth(node.left, x, depth + 1);
        int right = getDepth(node.right, x, depth + 1);
        return Math.max(left, right);
    }
    
    public TreeNode find(TreeNode node, int x){
        if(node == null){
            return null;
        }
        if(node.left != null && node.left.val == x){
            return node;
        }
        if(node.right != null && node.right.val == x){
            return node;
        }
        TreeNode left = find(node.left, x);
        TreeNode right = find(node.right, x);
        if(left != null){
            return left;
        }
        return right;
    }
}
