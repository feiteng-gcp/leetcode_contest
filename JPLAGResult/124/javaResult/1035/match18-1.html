/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private TreeNode px=null,py=null;
    private int levelx=0,levely=0;
    public boolean isCousins(TreeNode root, int x, int y) {
        if(root==null||root.val==x||root.val==y)
            return false;
        getLevelXandPX(root,null,0,x);
        getLevelYandPY(root,null,0,y);
        
if(levelx==levely&&px!=py)
            return true;
        return false;
    }
    private void getLevelXandPX(TreeNode node,TreeNode p,int level,int x){
        if(node==null)
            return;
        if(node.val==x){
            px=p;
            levelx=level;
            return;
        
}
        getLevelXandPX(node.left,node,level+1,x);
        getLevelXandPX(node.right,node,level+1,x);
    }
    private void getLevelYandPY(TreeNode node,TreeNode p,int level,int y){
        if(node==null)
            return;
        if(node.val==y){
            py=p;
            levely=level;
            return;
        }
        getLevelYandPY(node.left,node,level+1,y);
        getLevelYandPY(node.right,node,level+1,y);
    }
}
