/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution 
{
    TreeNode ans1 = null;
    int count1 = 0;
    TreeNode ans2 = null;
    int count2 = 0;
    public boolean isCousins(TreeNode root, int x, int y) 
    {
        dfs1(null, root, x,0);
        dfs2(null, root, y,0);
        
        //System.out.println(count1+" "+count2);
        //System.out.println(ans1+" "+ans2);
        if(ans1==null || ans2==null) return false;
        if(count1==count2 && ans1!=ans2) return true;
        
        return false;
    }
    
    public void dfs1(TreeNode par, TreeNode curr, int x, int y)
    {
        if(curr==null) return;
        
        //System.out.println("val:"+curr.val+" x:"+x);
        if(curr.val==x)
        {
            //System.out.println("com:"+curr.val);
            ans1 = par;
            count1 = y;
        }
        
        dfs1(curr, curr.left, x,y+1);
        dfs1(curr, curr.right, x,y+1);
    }
    
    public void dfs2(TreeNode par, TreeNode curr, int x, int y)
    {
        if(curr==null) return;
        
        if(curr.val==x)
        {
            ans2 = par;
            count2 = y;
        }
        
        dfs2(curr, curr.left,x, y+1);
        dfs2(curr, curr.right,x, y+1);
    }
}
