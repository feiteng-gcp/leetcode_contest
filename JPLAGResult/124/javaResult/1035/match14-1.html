/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int dx = 0;
    int dy = 0;
    TreeNode px = null;
    TreeNode py = null;
    public boolean isCousins(TreeNode root, int x, int y) {
        traverse(root, x, y, 0, null);
        
return (dx == dy && px != py);
    }
    
    private void traverse(TreeNode root, int x, int y, int depth, TreeNode par) {
        if (root == null) return
;
        if (root.val == x) {
            dx = depth;
            px = par;
        }
        if (root.val == y) {
            dy = depth;
            py = par;
        }
        traverse(root.left, x, y, depth + 1, root);
        traverse(root.right, x, y, depth + 1, root);
    }
}
