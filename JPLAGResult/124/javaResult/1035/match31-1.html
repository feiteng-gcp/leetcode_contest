/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCousins(TreeNode root, int x, int y) {
        int a = findDepth(root, x, 0);
        int b = findDepth(root, y, 0);
        if(a != b) {
            return false;
        };
        int indexA = findIndex(root, x, 0);
        int indexB = 
findIndex(root, y, 0);
        // System.out.println(indexA);
        // System.out.println(indexB);
        if(indexA/2 == indexB/2) {
            return false;
        }
        return true;
    }
    
    
private int findIndex(TreeNode root, int x, int index) {
        if(root == null) {
            return 100001;
        }
        if(root.val == x) {
            return index;
        }
        // copy下面的函数，却忘了修改 findDepth
        return Math.min(
findIndex(root.left, x, index*2), findIndex(root.right, x, index*2+1));
    }
    
    private int findDepth(TreeNode root, int x, int level) {
        if(root == null) {
            return 101;
        }
        if(root.val == x) {
            return level;
        }
        return Math.min(findDepth(root.left, x, level+1), findDepth(root.right, x, level+1));
    }
}
