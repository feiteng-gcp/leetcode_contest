/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCousins(TreeNode root, int x, int y) {
        if (sameParent(root, x, y)) {
            return false;
        }
        if (getDepth(root, 0, x) == getDepth(root, 0, y)) {
            return true;
        }
        return false;
    
}
    public boolean sameParent(TreeNode root, int x, int y) {
        if (root == null) {
            return false;
        }
        if ((root.left != null && root.left.val == x) || (root.right != null && root.right.val == x)) {
            if ((root.left != null && root.left.val == y) || (root.right != null && root.right.val == y)) {
                return true;
            } else 
{
                return false;
            }
        }
        return sameParent(root.left, x, y) || sameParent(root.right, x, y);
    }
    
    public int getDepth(TreeNode root, int depth, int x) {
        if (root == null) {
            return -1;
        }
        if (root.val == x) {
            return depth;
        }
        return Math.max(getDepth(root.left, depth + 1, x), getDepth(root.right, depth + 1, x));
    }
}
