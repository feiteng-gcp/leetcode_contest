/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int levelx = -1;
    int levely = -1;
    int px = 0;
    int py = 0;
    public boolean isCousins(TreeNode root, int x, int y) {
        findx(root, x, 0);
        
findy(root, y, 0);
        if (levelx != levely || levelx == -1) {
            return false;
        }
        return px != py;
        
    }
    
public void findx(TreeNode root, int tar, int level) {
        if (root == null) {
            return;
        }
        if ((root.left != null && root.left.val == tar) || (root.right != null && root.right.val == tar)) {
            levelx = level;
            px = root.val;
            return;
        }
        
findx(root.left, tar, level + 1);
        findx(root.right, tar, level + 1);
        
    }
    public void findy(TreeNode root, int tar, int level) {
        if (root == null) {
            return;
        }
        if ((root.left != null && root.left.val == tar) || (root.right != null && root.right.val == tar)) {
            levely = level;
            py = root.val;
            return;
        }
        findy(root.left, tar, level + 1);
        findy(root.right, tar, level + 1);
        
    }
}
