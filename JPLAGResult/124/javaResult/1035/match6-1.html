/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
int dx,dy,px,py;
    public boolean isCousins(TreeNode root, int x, int y) {
        search(root, x, y, -1,0);
        return dx==dy && px != py;
    }
    
    private void search(TreeNode root, int x, 
int y, int p, int d){
        if(root==null) return;
        if(root.val==x) {
            dx=d;
            px=p;
        }
        if(root.val==y){
            py=p;
            dy=d;
        }
        search(root.left,x,y,root.val,d+1);
        search(root.right,x,y,root.val,d+1);
    }
    
    
    
    
}
