/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isCousins(TreeNode root, int x, int y) {
        if (root.val == x || root.val == y || x == y) {
            return false;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        int k = 0;
        int xl = -1;
        TreeNode xp = null;
        int yl = -1;
        TreeNode yp = null;
        while (!queue.isEmpty() && xl == -1 && yl == -1) {
            
int size = queue.size();
            for (int i = 0; i < size; i ++) {
                TreeNode node = queue.poll();
                if (node.left != null) {
                    if (node.left.val == x) {
                        xl = k + 1;
                        xp = node;
                    } else if (node.left.val == y) {
                        yl = k + 1;
                        
yp = node;
                    }
                    queue.offer(node.left);
                }
                if (node.right != null) {
                    if (node.right.val == x) {
                        xl = k + 1;
                        xp = node;
                    } else if (node.right.val == y) {
                        yl = k + 1;
                        
yp = node;
                    }
                    queue.offer(node.right);
                }
            }
            k ++;
        }
        if (xl == -1 || yl == -1 || xl != yl) {
            return false;
        }
        return xp != yp;
    }
}
