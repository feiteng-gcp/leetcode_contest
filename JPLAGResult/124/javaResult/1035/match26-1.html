/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isCousins(TreeNode root, int x, int y) {
        if(root == null) return false;
        if(root.val == x || root.val == y) return false;
        
        int xdepth = depth(root, x);
        int ydepth = depth(root, y);
        if(xdepth != ydepth) return false;
        
        return getparent(root, x) != getparent(root, y);
    }
    
    public int depth(
TreeNode root, int x){
        if(root == null) return -1;
        if(root.val == x) return 0;
        int l = depth(root.left, x);
        if(l >= 0) return l+1;
        
        int r = depth(root.right, x);
        if(r >= 0) return r+1;
        
        return -1;
    
}
    
    public TreeNode getparent(TreeNode root, int x){
        if(root == null) return null;
        if(root.left != null && root.left.val == x) return root;
        if(root.right != null && root.right.val == x) return root;
        if(getparent(root.left, x) != null) return getparent(root.left, x);
        return getparent(root.right, x);
    }
}
