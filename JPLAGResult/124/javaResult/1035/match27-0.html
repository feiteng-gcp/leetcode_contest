/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCousins(TreeNode root, int x, int y) {
        return validate(root, x, y) && getDepth(root, x, 0) == getDepth(root, y, 0);
    }
    
    public boolean validate(TreeNode node, int x, int y) {
        if (node == null) {
            return true;
        }
        if (node.left != null && node.right != null) {
            if (node.left.val == x && node.right.val == y || node.left.val == y && node.right.val == x) {
                return false;
            }
        }
        return validate(node.left, x, y) && validate(node.right, x, y);
    }
    
    public int getDepth(TreeNode node, int val, int depth) {
        if (node == null) {
            return -1;
        }
        if (node.val == val) {
            return depth;
        }
        int left = getDepth(node.left, val, depth + 1);
        return left != -1 ? left : getDepth(node.right, val, depth + 1);
    }
}
