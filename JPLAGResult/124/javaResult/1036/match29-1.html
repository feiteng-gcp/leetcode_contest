class Solution {
    public int orangesRotting(int[][] grid) {
        
        // No corner casese need to check
        
        int result = 0;
        int rows = grid.length;
        int cols = grid[0].length;
        //boolean[][] visited = new boolean[rows][cols];
        Queue<int[]> q = new LinkedList<>();
        
int[][] dirs = new int[][]{{-1,0}, {1,0}, {0,-1}, {0,1}};
        
        
for(int i = 0; i < rows; i++){
            for(int j = 0; j < cols; j++){
                if(grid[i][j] == 2){
                    //visited[i][j] = true;
                    q.offer(new int[]{i, j});
                }
            }
        }
        if(q.isEmpty()){
            for(int i = 0; i < rows; i++){
                for(int j = 0; j < cols; j++){
                    if(grid[i][j] == 1){
                        return -1;
                    }
                }
            } 
            return 0;
        }
        // from minute 0;
        
while(!q.isEmpty()){
            int size = q.size();
            for(int i = 0; i < size; i++){
                int[] cur = q.poll();
                int r = cur[0];
                int c = cur[1];
                for(int[] dir : dirs){
                    int x = r + dir[0];
                    int y = c + dir[1];
                    if(isFresh(grid, x, y)){
                        grid[x][y] = 2;
                        q.offer(new int[]{x, y});
                        //visited[x][y] = true;
                    
}
                }
            }
            result++;
        }
        
        for(int i = 0; i < rows; i++){
            for(int j = 0; j < cols; j++){
                if(grid[i][j] == 1){
                    return -1;
                }
            }
        } 
        return result - 1;
    }
    
    private boolean isFresh(int[][] grid, int x, int y){
        return x >= 0 && x < grid.length && y >= 0 &&  y < grid[0].length && grid[x][y] == 1;
    }
}
