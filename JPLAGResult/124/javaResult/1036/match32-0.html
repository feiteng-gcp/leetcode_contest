class Solution {

    public int orangesRotting(int[][] grid) {
    	int time = bfs(grid);
    	int i , j , n = grid.length , m = grid[0].length;
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < m;j ++) {
    			if (grid[i][j] == 1) {
    				return - 1;    				
    			}
    		}
    	}
    	return time;    	
    }
    
    private int[] dx = new int[] {- 1 , 0 , 1 , 0};
    private int[] dy = new int[] {0 , 1 , 0 , - 1};
    
    private int bfs(int[][] grid) {
    	
    	Queue<int[]> queue = new LinkedList<>();
    	int i , j , n = grid.length , m = grid[0].length , ans = 0;
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < m;j ++) {
    			if (grid[i][j] == 2) {
    				queue.add(new int[] {i , j});
    			}    			
    		}
    	}
    	while (!queue.isEmpty()) {
    		List<int[]> nextList = new ArrayList<>();
    		while (!queue.isEmpty()) {
	    		int[] current = queue.poll();
	    		for (i = 0;i < 4;i ++) {
	    			int tx = current[0] + dx[i];
	    			int ty = current[1] + dy[i];
	    			if (tx >= 0 && tx < n && ty >= 0 && ty < m && grid[tx][ty] == 1) {
	    				grid[tx][ty] = 2;
	    				nextList.add(new int[] {tx , ty});    			
	    			}
	    		}
    		}
    		if (!nextList.isEmpty()) {
    			ans ++;
    		}
    		queue.addAll(nextList);
    	}
    	return ans;
    	
    }
    
}
