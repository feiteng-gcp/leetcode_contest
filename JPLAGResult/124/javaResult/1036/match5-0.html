class Solution {
    
    int[][] directions = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int orangesRotting(int[][] grid) {
        int freshCnt = 0, row = grid.length, col = grid[0].length;
        Queue<int[]> rottens = new LinkedList<>();
        
        for(int i = 0; i < row; i++){
            for(int j = 0; j < col; j++){
                if(grid[i][j] == 1)
                    freshCnt++;
                else if (grid[i][j] == 2){
                    rottens.add(new int[]{i, j});
                }
                    
            }
        }
        
        int minutes = 0;
        while(freshCnt > 0 && !rottens.isEmpty()){
            minutes++;
            int num = rottens.size();
            for(int i = 0; i < num; i++){
                int[] one = rottens.poll();
                for(int[] dir : directions){
                    int nextRow = one[0] + dir[0], nextCol = one[1] + dir[1];
                    if(nextRow >= 0 && nextRow < row && nextCol >= 0 && nextCol < col && grid[nextRow][nextCol] == 1){
                        grid[nextRow][nextCol] = 2;
                        if(freshCnt == 1)
                            return minutes;
                        freshCnt--;
                        rottens.add(new int[]{nextRow, nextCol});
                    }
                }
            }
        }
        return freshCnt > 0 ? -1 : minutes;
    }
}
