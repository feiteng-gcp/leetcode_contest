class Solution {
    int[][] dirs = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int orangesRotting(int[][] grid) {
        int row = grid.length;
        // BUG BUG, array 没有length()
        if(row == 0 || grid[0].length == 0) {
            return 0;
        }
        int col = grid[0].length;
        List<int[]> Q = new LinkedList<>();
        int total = 0;
        for(int i=0; i<row; i++) {
            for(int j=0; j<col; j++) {
                if(grid[i][j] == 2) {
                    Q.add(new int[]{i, j});
                }
                if(grid[i][j] == 1) {
                    total++;
                }
            }
        }
        int steps = 0;
        while(total > 0 && Q.size() > 0) {
            steps++;
            int k = Q.size();
            for(int i=0; i<k; i++) {
                int[] cur = Q.remove(0);
                for(int[] dir: dirs) {
                    int x = cur[0] + dir[0];
                    int y = cur[1] + dir[1];
                    if(x>=0 &&x<row && y>=0 && y<col && grid[x][y] == 1) {
                        total--;
                        grid[x][y] = 2;
                        Q.add(new int[]{x, y});
                    }
                }
            }
        }
        if(total > 0) {
            return -1;
        }
        return steps;
    }
}
