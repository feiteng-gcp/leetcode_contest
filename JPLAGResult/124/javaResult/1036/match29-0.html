class Solution {
    int[][] ds = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    public int orangesRotting(int[][] grid) {
        int rows = grid.length;
        int cols = grid[0].length;
        Deque<int[]> deque = new LinkedList<>();
        for (int x = 0; x < rows; x++) {
            for (int y = 0; y < cols; y++) {
                if (grid[x][y] == 2) {
                    deque.offerLast(new int[]{x, y});
                }
            }
        }
        int t = -1;
        while (!deque.isEmpty()) {
            t++;
            int N = deque.size();
            for (int i = 0; i < N; i++) {
                int[] cur = deque.pollFirst();
                for (int j = 0; j < ds.length; j++) {
                    int[] next = add(cur, ds[j]);
                    if (next[0] >= 0 && next[0] < rows && next[1] >= 0 && next[1] < cols && grid[next[0]][next[1]] == 1) {
                        grid[next[0]][next[1]] = 2;
                        deque.offerLast(next);
                    }
                }
            }
        }
        for (int x = 0; x < rows; x++) {
            for (int y = 0; y < cols; y++) {
                if (grid[x][y] == 1) {
                    return -1;
                }
            }
        }
        if(t<0) {
            t=0;
        }
        return t;
    }

    int[] add(int[] v1, int[] v2) {
        return new int[]{v1[0] + v2[0], v1[1] + v2[1]};
    }
}
