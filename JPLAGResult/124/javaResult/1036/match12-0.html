class Solution {
    public int orangesRotting(int[][] grid) {
        boolean[][] visited = new boolean[grid.length][grid[0].length];
        int count = grid.length * grid[0].length;
        int time = 0;
        Queue<Node> queue = new LinkedList<>();
        for(int i = 0; i < grid.length; i++) {
            for(int j = 0; j < grid[0].length; j++) {
                if(grid[i][j] == 2) {
                    count--;
                    queue.offer(new Node(i, j));
                    visited[i][j] = true;
                } else if(grid[i][j] == 0) {
                    count--;
                }
            }
        }
        if(count == 0) {
            return 0;
        }
        int[] helpx = new int[] {0, 0, 1, -1};
        int[] helpy = new int[] {1, -1, 0, 0};
        while(!queue.isEmpty()) {
            time++;
            int size = queue.size();
            for(int i = 0; i < size; i++) {
                Node tmp = queue.poll();
                for(int j = 0; j < 4; j++) {
                    int tx = tmp.x + helpx[j];
                    int ty = tmp.y + helpy[j];
                    if(tx >= 0 && tx < grid.length && ty >= 0 && ty < grid[0].length && !visited[tx][ty] && grid[tx][ty] == 1) {
                        count--;
                        queue.offer(new Node(tx, ty));
                        visited[tx][ty] = true;
                    }
                }
            }
        }
        
        
        if(count != 0) {
            return -1;
        }
        return time - 1;
    }
    
    class Node {
        int x;
        int y;
        
        Node(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
}
