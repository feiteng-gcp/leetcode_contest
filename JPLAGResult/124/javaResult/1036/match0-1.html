class Solution {
    public int orangesRotting(int[][] grid) {
        int l = grid.length;
        int ll = grid[0].length;
        int all = l*ll;
        
        int f = 0;
        int r = 0;
        int z = 0;
        
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                int v = grid[i][j];
                if(v==1){
                    ++f;
                }else if(v==2){
                    ++r;
                }
            }
        }
        
        if(f==0){
            return 0;
        }
        if(r==0){
            return -1;
        }
        
        Queue<int[]> que = new LinkedList<>();
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                int v = grid[i][j];
                if(v==2){
                    que.offer(new int[]{i,j});
                }
            }
        }
        
        
        
        
                    int stp = 0;
                    while(que.size()>0){
                        Queue<int[]> nque = new LinkedList<>();
                        while(!que.isEmpty()){
                            int[] pos = que.poll();
                            int x = pos[0];
                            int y = pos[1];
                            
                            int dx[] = new int[]{0,0,1,-1};
                            int dy[] = new int[]{1,-1,0,0};
                            for(int i=0;i<4;++i){
                                int nx = dx[i]+x;
                                int ny = dy[i]+y;
                                if(nx>=0&&nx<l&&ny>=0&&ny<ll&&grid[nx][ny]==1){
                                    grid[nx][ny] = 2;
                                    --f;
                                    if(f==0){
                                        return stp+1;
                                    }
                                    nque.offer(new int[]{nx,ny});
                                    
                                }
                                
                                
                            }
                        }
                        ++stp;
                        
                        que = nque;
                        
                    }
        
        return -1;
        
        
        
        
        
    }
}
