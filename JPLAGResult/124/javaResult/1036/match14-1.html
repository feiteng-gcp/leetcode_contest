class Solution {
    public int orangesRotting(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int total = 0;
        Queue<int[]> q = new LinkedList<>();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 2) {
                    q.offer(new int[]{i, j});
                } else if (grid[i][j] == 1) {
                    total++; 
                
}
            }
        }
        if (total == 0) {
            return 0;
        }
        int res = -1;
        
int[][] dirs = new int[][]{{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
        // Set<Integer> set = new HashSet<>();
        
while (!q.isEmpty()) {
            int size = q.size();
            // System.out.println(size);
            for (int i = 0; i < size; i++) {
                int[] p = q.poll();
                for (int[] dir : dirs) {
                    int x = dir[0] + p[0];
                    int y = dir[1] + p[1];
                    if (x >= 0 && y >= 0 && x < m && y < n && grid[x][y] == 1) {
                        total--;
                        q.offer(new int[]{x, y});
                        grid[x][y] = 2;
                        // set.add(x * n + y);
                    }
                }
            }
            res++;
        }
        if (total != 0) {
            return -1;
        }
        return res;
    }
}
