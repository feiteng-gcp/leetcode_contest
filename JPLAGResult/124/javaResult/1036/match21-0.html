class Solution {
    int[] dx = {-1, 1, 0, 0}, dy = {0, 0, 1, -1};
    
    public int orangesRotting(int[][] grid) {
        int ans = -1, m = grid.length, n = grid[0].length, totalOrange = 0;
        Queue<int[]> q = new ArrayDeque<>();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] != 0) {
                    totalOrange++;
                }
                if (grid[i][j] == 2) {
                    q.offer(new int[]{i, j});
                }
            }
        }
        if (totalOrange == 0) {
            return 0;
        }
        while (!q.isEmpty()) {
            ans++;
            int size = q.size();
            while (size-- > 0) {
                int[] cur = q.poll();
                int x = cur[0], y = cur[1];
                for (int i = 0; i < 4; i++) {
                    int xx = x + dx[i], yy = y + dy[i];
                    if (xx >= 0 && xx < m && yy >= 0 && yy < n && grid[xx][yy] == 1) {
                        grid[xx][yy] = 2;
                        q.offer(new int[]{xx, yy});
                    }
                }
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    return -1;
                }
            }
        }
        return ans;
    }
}
