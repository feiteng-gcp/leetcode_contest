class Solution {
    public int orangesRotting(int[][] grid) {
        Queue<int[]> q = new LinkedList<>();
        int m = grid.length, n = grid[0].length;
        boolean[][] rotten = new boolean[m][n];
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) if(grid[i][j] == 2) q.offer(new int[]{i, j});
        }
        int l = 0;
        int[][] dir = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        while(!q.isEmpty()) {
            for(int i = q.size(); i>0; i--) {
                int[] temp = q.poll();
                for(int j = 0;  j < 4; j++) {
                    int x = temp[0] + dir[j][0], y = temp[1] + dir[j][1];
                    if(x >= 0 && x < m && y >= 0 && y < n && grid[x][y] == 1 && !rotten[x][y]) {
                        rotten[x][y] = true;
                        q.offer(new int[]{x,y});
                    }
                }
            }
            if(q.isEmpty()) {
                for(int i = 0; i < m; i++) {
                    for(int j = 0; j < n; j++) {
                        if(grid[i][j] == 1 && !rotten[i][j]) return -1;
                    }
                }
                return l;
            }
            l++;
        }
                for(int i = 0; i < m; i++) {
                    for(int j = 0; j < n; j++) {
                        if(grid[i][j] == 1 && !rotten[i][j]) return -1;
                    }
                }
        return 0;
    }
}
