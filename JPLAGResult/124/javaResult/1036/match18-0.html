class Solution {
    private static final int[][] dir = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
    public int orangesRotting(int[][] grid) {
        int cntOfR = 0, cntOfF = 0; 
        int minute = 0;
        Queue<Integer> rotQ = new ArrayDeque<>();
        Set<Integer> vis = new HashSet<>();
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1){
                    cntOfF++;
                }
                if (grid[i][j] == 2){
                    cntOfR++;
                    rotQ.offer(i * grid[0].length + j);
                    vis.add(i * grid[0].length + j);
                }
            }
        }
       // System.out.println("Step1:" + rotQ.size());
        while (cntOfF != 0 && rotQ.size() != 0) {
            int len = rotQ.size(); 
            for (int i = 0; i < len; i++) {
                int cur = rotQ.poll();
                int curX = cur / grid[0].length;
                int curY = cur % grid[0].length;
                for (int j = 0; j < 4; j++) {
                    int nextX = curX + dir[j][0];
                    int nextY = curY + dir[j][1];
                    if (nextX < 0 || nextX >= grid.length || nextY < 0 || nextY >= grid[0].length 
                       || grid[nextX][nextY] != 1 || vis.contains(nextX * grid[0].length + nextY)) {
                        continue;
                    }
                    rotQ.add(nextX * grid[0].length + nextY);
                    vis.add(nextX * grid[0].length + nextY);
                    cntOfF--;
                }
                
            }
            minute++;
        }
        return cntOfF == 0 ? minute : -1; 
    }
}
