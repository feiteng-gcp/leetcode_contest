class Solution {
    public int orangesRotting(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int fresh = 0;
        Queue<int[]> queue = new LinkedList<>();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    fresh++;
                } else if (grid[i][j] == 2) {
                    queue.offer(new int[]{i, j});
                }
            }
        }
        if (fresh == 0) {
            return 0;
        }
        boolean[][] visited = new boolean[m][n];
        int[][] deltas = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        int minutes = 0;
        while (!queue.isEmpty()) {
            int rotten = queue.size();
            for (int i = 0; i < rotten; i++) {
                int[] curr = queue.poll();
                for (int[] d : deltas) {
                    int dr = curr[0] + d[0];
                    int dc = curr[1] + d[1];
                    if (dr < 0 || dr >= m || dc < 0 || dc >= n || grid[dr][dc] != 1 || visited[dr][dc]) {
                        continue;
                    }
                    fresh--;
                    queue.offer(new int[]{dr, dc});
                    visited[dr][dc] = true;
                }
            }
            minutes++;
            if (fresh == 0) {
                return minutes;
            }
        }
        return -1;
    }
}
