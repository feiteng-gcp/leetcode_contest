class Solution {
    
class Pair {
        int row;
        int col;
        Pair (int row, int col) {
            this.row = row;
            this.col = col;
        }
    }
    int[] rowTrans = {0, 0, -1, 1};
    int[] colTrans = {1, -1, 0, 0};
    public boolean inBound(int[][] matrix, int row, int col) {
        if (row >= 0 && row < matrix.length && col >= 0 && col < matrix[0].length) {
            return true;
        }
        return false;
    }
    public int orangesRotting(int[][] matrix) {
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) {
            return 0;
        }
        
int number = 0;
        int result = 0;
        Queue<Pair> queue = new LinkedList<>();
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                if (matrix[i][j] == 1) {
                    number++;
                }
                if (matrix[i][j] == 2) {
                    queue.offer(new Pair(i, j));
                
}
            }
        }
        if (number == 0) {
            return 0;
        }
        
while (!queue.isEmpty()) {
            result++;
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                Pair pair = queue.poll();
                for (int j = 0; j < 4; j++) {
                    int nextRow = pair.row + rowTrans[j];
                    int nextCol = pair.col + colTrans[j];
                    if (inBound(matrix, nextRow, nextCol) && matrix[nextRow][nextCol] == 1) {
                        matrix[nextRow][nextCol] = 2;
                        number--;
                        queue.offer(new Pair(nextRow, nextCol));
                    
}
                }
            }
        }
        if (number != 0) {
            return -1;
        }
        return result - 1;
    }
}
