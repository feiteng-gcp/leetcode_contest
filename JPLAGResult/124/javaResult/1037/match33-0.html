class Solution {
    public int minKBitFlips(int[] A, int K) {
        if (K == 1) {
            return (int) Arrays.stream(A).filter(k -> k == 0).count();
        }
        int p = 0;
        int n = A.length;
        int res = 0;
        while (p <= n - K) {
            if (A[p] == 0) {
                res++;
                for (int i = 0; i < K; i++) {
                    A[p + i] = 1 - A[p + i];
                }
            }
            p++;
//            System.out.println(Arrays.toString(A));
        }
        for (int i = n - K + 1; i < n; i++) {
            if (A[i] == 0) {
                return -1;
            }
        }
        return res;
    }
}
