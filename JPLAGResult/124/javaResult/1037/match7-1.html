class Solution {
    public int minKBitFlips(int[] A, int K) {
        
int result = 0;
        for (int i = 0; i <= A.length - K; i++) {
            if (A[i] == 0) {
                result++;
                for (int j = 0; j < K; j++) {
                    A[i + j] = A[i + j] ^ 1;
                
}
            }
        }
        for (int i = 0; i < A.length; i++) {
            if (A[i] == 0) {
                return -1;
            }
        }
        return result;
    }
}
