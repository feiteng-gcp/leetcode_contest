class Solution {
    public int minKBitFlips(int[] A, int K) {
        int count = 0;
        for (int i = 0; i < A.length; i++) {
            if (A[i] == 0) {
                count++;
                if (i + K > A.length) return -1;
                helper(i, K, A);
            }
        }
        return count;
    }
    
    public void helper (int index, int K, int[] A) {
        for (int i = 0; i < K; i++) {
            A[i + index] = 1 - A[i + index];
        }
    }
}
