class Solution {
    public int minKBitFlips(int[] A, int K) {
        
int i = 0, flip = 0;
        while (i < A.length - K + 1) {
            if (A[i] == 0) {
                flip++;
                for (int j = 0; j < K; j++) {
                    A[i + j] = 1 - A[i + j];
                }
            }
            i++;
        }
        for (int a : A) {
            if (a == 0) {
                return -1;
            }
        }
        return flip;
    }
}
