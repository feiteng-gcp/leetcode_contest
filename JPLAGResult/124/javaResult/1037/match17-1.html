class Solution {
    public int minKBitFlips(int[] A, int K) {
        int re = 0;
        for(int i = 0; i < A.length; i++) {
            if(A[i] == 0) {
                if(i + K > A.length) return -1;
                for(int j = i; j < i + K; j++) A[j] = 1 - A[j]
;
                re++;
            }
        }
        return re;
    }
}
