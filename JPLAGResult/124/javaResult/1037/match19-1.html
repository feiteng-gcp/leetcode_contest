class Solution {
    public int minKBitFlips(int[] A, int K) {
        int count = 0;
        for (int i = 0; i + K - 1 < A.length; i++) {
            if (A[i] == 0) {
                flip(A, K, i);
                count++;
            
}
        }
        for (int i = A.length - K; i< A.length; i++) {
            if (A[i] == 0) {
                return -1;
            }
        }
        return count;
    }
    public void flip(int[]A, int K, int pos) 
{
        for (int i = pos; i - pos < K; i++) {
            if (A[i] == 0) {
                A[i] = 1;
            } else {
                A[i] = 0;
            }
        }
    }
}
