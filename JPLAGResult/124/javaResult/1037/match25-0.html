class Solution {
    public int minKBitFlips(int[] a, int k) {
        int n = a.length;
        int ans= 0;;
        for(int  i = 0 ; i < n-k+1;i++){
            
            if(a[i] == 0 ){
                ans++;
                for(int j =i;j <i+k;j++){                    
                    a[j] = 1-a[j];
                }
            }
        }
        
        for(int i = 0 ;i< n; i++){
            if(a[i] ==0 )
                return -1;
        }
        return ans;
    }
}
