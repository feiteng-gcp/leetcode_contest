class Solution {
    public int minKBitFlips(int[] A, int K) {
         int N = A.length;
        int res = 0;
        for (int i = 0; i <= N - K; i++) {
            if (A[i] == 0) {
                res++;
                for (int j = 0; j < K; j++) {
                    A[i + j] = 1 - A[i + j];
                
}
            }
        }
        for (int i = N - K + 1; i < N; i++) {
            if (A[i] == 0) {
                return -1;
            }
        }
        return res;
    }
}
