class Solution {
    public int minKBitFlips(int[] A, int k) {
        
        int r = 0;
        int l = A.length;
        
        for(int i=0;i<l-k+1;++i){
            if(A[i]==0){
                for(int j= 0;j<k;++j){
                    A[i+j] ^=1;
                }
                ++r;
            }
        }
        
        for(int i=l-k+1;i<l;++i){
            if(A[i]==0){
                return -1;
            }
        }
        return r;
        
    }
}
