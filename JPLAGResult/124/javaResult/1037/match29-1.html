class Solution {
    public int minKBitFlips(int[] A, int K) {
        int idx = 0;
        int res = 0;
        while (idx <= A.length - K) {
            if (A[idx] == 1) {
                idx++;
            } else {
                res++;
                for (int i = idx; i < idx + K; i++) {
                    if (A[i] == 0) {
                        A[i] = 1;
                    } else {
                        A[i] = 0;
                    }
                
}
            }
        }
        for (int i = 0; i < A.length; i++) {
            if (A[i] == 0) {
                return -1;
            }
        }
        return res;
    }
}
