class Solution {
    public int minKBitFlips(int[] A, int K) {
        int n = A.length;
        int ans = 0;
        for (int i = 0; i <= n - K; i++) {
            if (A[i] == 1) {
                continue;
            }
            for (int j = i; j < i + K; j++) {
                A[j] = 1 - A[j];
            }
            ans++;
        
}
        for (int i = 0; i < n; i++) {
            if (A[i] == 0) {
                return -1;
            }
        }
        return ans;
    }
}
