class Solution {
    public int minKBitFlips(int[] A, int K) {
        
int t = 0;
        for (int i = 0;i <= A.length - K;i++) {
            if (A[i] == 0) {
                t++;
                for (int j = 0;j < K;j++) {
                    A[i + j] = 1 - A[i + j];
                }
            }
        }
        for (int i = 0;i < A.length;i++) {
            if (A[i] == 0) return -1;
        }
        return t;
    }
}
