#define FLIP(i) (-i+1)
class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        // greedy correct
        int n = A.size();
        int maxI = n - K;
        int count = 0;
        for(int i = 0;i <= maxI;i++){
            if(!A[i]){
                count++;
                // flip next K values
                for(int j = 0;j<K;j++){
                    A[i+j] = FLIP(A[i+j]);
                }
            }
        }
        // check tail
        for(int i = maxI + 1;i<n;i++){
            if(!A[i]) return -1;
        }
        
        return count;
    }
};
