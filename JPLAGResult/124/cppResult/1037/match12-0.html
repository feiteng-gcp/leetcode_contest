class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int res = 0;
        int as = A.size();
        for (int i = 0; i <= as-K; i++) {
            if (A[i] == 0) {
                res++;
                for (int j = i+1; j < i+K; j++) {
                    A[j] = 1 - A[j];
                }
            }
        }
        for (int i = as-K+1; i < as; i++) {
            if (A[i] == 0) return -1;
        }
        return res;
    }
};
