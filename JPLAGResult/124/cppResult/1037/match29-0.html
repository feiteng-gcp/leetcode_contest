class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        vector<int> flips;
        int iflip = 0;
        
        for (int i = 0; i < A.size(); ++i) {
            if (iflip < flips.size() && flips[iflip] + K <= i) {
                ++iflip;
            }
            bool fliped = (flips.size() - iflip) % 2;
            if ((!fliped && A[i] == 0) || (fliped && A[i] == 1)) {
                if (i + K <= A.size())
                    flips.push_back(i);
                else
                    return -1;
            }
        }
        
        return flips.size();
    }
};
