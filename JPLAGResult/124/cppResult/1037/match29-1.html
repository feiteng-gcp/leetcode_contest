class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        queue<int> q;
        int ans = 0;
        for (int i = 0; i < A.size(); i++) {
            if ((A[i] + q.size()) % 2 == 0) {
                q.push(i + K - 1);
                ans++;
            }
            if (!q.empty() && q.front() == i)
                q.pop();
        }
        if (!q.empty())
            return -1;
        return ans;
    }
};
