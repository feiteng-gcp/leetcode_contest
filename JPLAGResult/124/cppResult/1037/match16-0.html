class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int cnt=0;
        for (int i=0;i<A.size()-K+1;++i){
            if (A[i]==0){
                for (int j=i;j<i+K;++j)
                    A[j]=1-A[j];
                cnt++;
            }
        }
        for (int i=A.size()-K+1;i<A.size();++i){
            if (A[i]==0)
                return -1;
        }
        return cnt;
    }
};
