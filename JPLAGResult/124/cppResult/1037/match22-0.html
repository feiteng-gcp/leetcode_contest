class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int sz = A.size();
        int ret = 0;
        for (int i = 0; i <= sz - K; ++ i) {
            if (A[i] == 1) continue;
            for (int j = 0; j < K; ++ j) {
                A[i + j] = 1 - A[i + j];
            }
            ret ++;
        }
        for (int i = 0; i < sz; ++ i) {
            if (A[i] == 0) return -1;
        }
        return ret;
    }
};
