class Solution {
public:
	int sum, cnt, change[30010];
    int minKBitFlips(vector<int>& A, int K) {
        fill(change, change + 30010, 0);
        sum = cnt = 0;

        int l = A.size();
        for (int i = 0; i + K <= l; ++i) {
        	sum ^= change[i];
        	if ((A[i] ^ sum) == 0) {
        		++cnt;
        		sum ^= 1;
        		change[i + K] = 1;
			}
		}
		for (int i = l - K + 1; i < l; ++i) {
			sum ^= change[i];
			if ((A[i] ^ sum) == 0)
				return -1;
		}
		return cnt;
    }
};
