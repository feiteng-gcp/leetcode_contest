class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int m = 0;
        for(int i=0;i<A.size();i++) { 
            if(A[i] == 0 && i+K <= A.size() ) { 
                for(int k=0;k<K;k++)
                    A[i+k] ^= 1;
                m++;
            } else if(A[i] ==0 && i+K > A.size() ) {
               // cout << "HERE " ;
               // for(int j=0;j<A.size();j++) cout << A[j] << " ";
               // cout << endl;
                return -1;
            }
        }
        return m;
    }
};
