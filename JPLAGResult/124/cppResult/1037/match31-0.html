class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int n = A.size();
        vector<int> flips(n, 0);
        
        for (int i = 0; i < n - K; i++) {
            if (A[i] == 0) {
                flips[i] = 1 - flips[i];
                flips[i+1] = 1 - flips[i+1];
                A[i] = 1;
                A[i+K] = 1 - A[i+K];
            }
        }
        
        if (A[n - K] == 0) {
            flips[n - K] = 1 - flips[n - K];
        }
        
        for (int i = n - K; i < n; i++) {
            if (A[i] != A[n - K])
                return -1;
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++)
            ans += flips[i];
        return ans;
    }
};
