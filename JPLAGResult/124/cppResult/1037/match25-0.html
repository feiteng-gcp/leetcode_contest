class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int flips = 0;
        for (int i = 0; i < A.size() - K + 1; ++i) {
            if (A[i] == 1) continue;
            
            ++flips;
            for (int j = i, z = 0; z < K; ++z, ++j) {
                A[j] = 1-A[j];
            }
        }
        
        for (int i = 0; i < K; ++i) {
            if (A[A.size()-1-i] == 0) return -1;
        }
        return flips;
    }
};
