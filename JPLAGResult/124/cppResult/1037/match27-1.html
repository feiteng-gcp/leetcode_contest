class Solution {
public:
    int b[30030];

    int minKBitFlips(vector<int>& A, int K) {
        // if (K == 1) {
        //     int ans = 0;
        //     for (int i=0;i<A.size();i++) {
        //         ans += (A[i] == 0);
        //     }
        //     return ans;
        // }
        int n = A.size();
        int flip = 0;
        int ans = 0;
        for (int i=0;i<n;i++) {
            flip ^= b[i];
            A[i] ^= flip;
            if (A[i] == 0 && i + K - 1 < n) {
                ans++;
                A[i] = 1;
                b[i+K] = 1;
                flip ^= 1;

            }
        }
        
        for (int i=0;i<n;i++) {
            if (A[i] == 0) return -1;
        }
        return ans;
        
    }
};
