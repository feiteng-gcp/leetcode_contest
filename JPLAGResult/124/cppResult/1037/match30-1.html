class FenwickTree {
private:
	std::vector<int> fenwick;
	int size;

public:

	FenwickTree();

	FenwickTree(int sz);

	FenwickTree(std::vector<int> vals);

	FenwickTree(int sz, std::vector<int> vals);

	int subarray_sum(int i, int j) const;

	int prefix_sum(int i) const;

	void increment(int index, int val);

	void update(int index, int val);

	int getSize() const;
	
};


FenwickTree::FenwickTree() {

}

FenwickTree::FenwickTree(int sz) {
	fenwick = std::vector<int>(sz+1, 0);
	size = sz; 
}

FenwickTree::FenwickTree(std::vector<int> vals) {
	fenwick = std::vector<int>(vals.size() + 1, 0);
	size = vals.size();
	for (int i = 0; i < vals.size(); ++i) {
		increment(i + 1, vals[i]);
	}
	
}

FenwickTree::FenwickTree(int sz, std::vector<int> vals) {
	if (vals.size() != sz) {
		throw std::invalid_argument("Vals should be of valid size");
	}

	fenwick = std::vector<int>(sz + 1, 0);
	size = sz;

	for (int i = 0; i < vals.size(); ++i) {
		increment(i + 1, vals[i]);
	}
	
}

int FenwickTree::subarray_sum(int i, int j) const {
	if (i > j) {
		throw std::invalid_argument("i should be less than or equal to j");
	}
	return prefix_sum(j) - prefix_sum(i-1);
}

int FenwickTree::prefix_sum(int index) const {
	int sum = 0;
	for (int i = index; i > 0; i -= i & (-i)) {
		sum += fenwick[i];
	}
	return sum;
}

void FenwickTree::increment(int index, int val) {
	for (int i = index; i <= size; i += i & (-i)) {
		fenwick[i] += val;
	}
}

void FenwickTree::update(int index, int val) {
	int diff = val - subarray_sum(index, index);
	increment(index, diff);
}


int FenwickTree::getSize() const {
	return size;
}



class Solution {
public:
    
    int minKBitFlips(vector<int>& A, int K) {
        //FenwickTree fw(A);
        vector<int> init(A.size(), 0);
        FenwickTree fw(init);
        
        int count = 0;
        
        for (int i = 0; i < A.size() - K + 1; ++i) {
            if ((A[i] + fw.prefix_sum(i + 1)) % 2 == 0) {
                count++;
                fw.increment(i + 1, 1);
                fw.increment(i + K + 1, -1);
            }
        }
        
        
        for (int i = A.size() - K + 1; i < A.size(); ++i) {
            if ((A[i] + fw.prefix_sum(i + 1)) % 2 == 0) {
                return -1;
            }
        }
        
        
        return count;
    }
};
