class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int flips = 0;
        for (int i = 0; i <= A.size() - K; i++) {
            if (!A[i]) {
                flips++;
                for (int j = 0; j < K; j++) {
                    A[i + j] ^= 1;
                }
            }
        }
        return accumulate(A.begin(), A.end(), 0) < A.size() ? -1 : flips;
    }
};
