class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        /*A.clear();
        A.push_back(0);
        A.push_back(0);
        A.push_back(0);
        A.push_back(1);
        A.push_back(0);
        A.push_back(1);
        A.push_back(1);
        A.push_back(0);
        K = 3;*/
        int start_index = 0;
        int result = 0;
        while (start_index < A.size())
        {
            while (start_index < A.size() && A[start_index] == 1)
            {
                start_index ++;
            }
            
            if (start_index == A.size())
            {
                return result;
            }
            

            if (start_index + K - 1 >= A.size())
            {
                return -1;
            }
            
            for (int i=0; i<K; i++)
            {
                A[start_index + i] = 1 - A[start_index + i];
            }
            result ++;
        }
        
        return result;
    }
};
