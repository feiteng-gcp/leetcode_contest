class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        deque<int> flip_idx;
        int target = 1;
        int cnt = 0;
        for (int i = 0; i < A.size(); i++) {
            if (!flip_idx.empty() && flip_idx.front() == i) {
                target = 1 - target;
                flip_idx.pop_front();
                //cout << i;
            }
            if (A[i] != target) {
                if (i > A.size() - K) return -1;
                cnt++;
                target = 1 - target;
                flip_idx.push_back(i+K);
            }
        }
        return cnt;
    }
};
