class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {

        int n = A.size(), ans = 0;
        for (int i = 0; i <= n - K; ++i) {
            if (A[i] == 0) {
                ans++;
                for (int j = i; j < i + K; ++j) {
                    A[j] = 1 - A[j];
                }

            }
        }
        for (int i = 0; i < n; ++i) {
            if (A[i] == 0) return -1;
        }
        return ans;
    }
};
