class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {

        int ans = 0, n = A.size() - K;
        for (int i = 0; i <= n; ++i) {
            if (A[i] == 0) {
                for (int j = 0; j < K; ++j) {
                    A[i + j] = !A[i + j];
                }

                ++ans;
            }
        }
        for (int i = 0; i < A.size(); ++i) {
            if (A[i] == 0) return -1;
        }
        return ans;
    }
};
