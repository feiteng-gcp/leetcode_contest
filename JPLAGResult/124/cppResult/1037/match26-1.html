class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        queue<int> q;
        int n = A.size();

        int current = 1;
        int res = 0;
        for(int i = 0; i < n; i++){
            if(!q.empty() && i == q.front()){
                q.pop();
                current = current ^ 1;
            }
            if(A[i] != current){
                if(i+K>n)return -1;
                current = current ^ 1;
                q.push(i+K);
                res++;
            }
        }
        return res;
    }
};
