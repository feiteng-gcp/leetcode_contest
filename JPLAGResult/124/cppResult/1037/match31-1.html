class Solution {
public:
    int minKBitFlips(vector<int>& A, int K) {
        int cnt = 0;
        
        queue<int> flips;
        for (int i = 0; i < A.size(); i++) {
            if ((A[i] + flips.size()) % 2 == 0) {
                cnt++;
                flips.push(i + K - 1);
            }
            if (i == flips.front()) flips.pop();
        }
        return flips.empty() ? cnt : -1;
    }
};
