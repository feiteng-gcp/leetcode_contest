class Solution {
public:
  int minKBitFlips(vector<int>& A, int K) {
    int ans = 0;

    int n = A.size();
    for (int i = 0; i + K - 1 < n; ++i) {
      if (A[i] == 0) {
        ++ans;
        for (int j = 0; j < K; ++j) {
          A[i + j] = 1 - A[i + j];
        }
      }
    }
    for (int i = 0; i < n; ++i) {
      if (A[i] == 0) {
        return -1;
      }
    }
    return ans;
  }
};
