/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void traverse(TreeNode* root, int depth, unordered_map<int, pair<TreeNode*, int>>& data) {
        if (root->left != NULL) {
            data[root->left->val] = {root, depth};
            traverse(root->left, depth+1, data);
        }
        if (root->right != NULL) {
            data[root->right->val] = {root, depth};
            traverse(root->right, depth+1, data);
        }
        
    }
    bool isCousins(TreeNode* root, int x, int y) {
        if (x == y) return false;
        if (root->val == x) return false;
        if (root->val == y) return false;
        auto last = root;
        
        unordered_map<int, pair<TreeNode*, int>> tree_data;
        tree_data[root->val] = {NULL, -1};
        traverse(root, 0, tree_data);
        
        auto xdata = tree_data[x], ydata = tree_data[y];
        if (xdata.first != ydata.first && xdata.second == ydata.second) return true;
        else return false;
    }
};
