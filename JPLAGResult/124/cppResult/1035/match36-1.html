/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    

    void getDepth(TreeNode *root, int x, int h, int &depth, int p, int &par) {
        if(root == NULL) return;
        if(root->val == x) {
            depth = h;
            par = p;
            return;
        }
        getDepth(root->left, x, h+1, depth, root->val, par);
        getDepth(root->right, x, h+1, depth, root->val, par);
    }
    
    bool isCousins(TreeNode* root, int x, int y) {
        int depthX, depthY, parentX, parentY;
        depthX = depthY = -1;
        getDepth(root, x, 0, depthX, -1, parentX);
        getDepth(root, y, 0, depthY, -1, parentY);
        if(x == y || depthX == -1 || depthY == -1 || parentX == parentY) return false;
        return depthX == depthY;
    }
};
