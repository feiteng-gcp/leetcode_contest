/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<int,int> depth;
    map<int,int> parent;
    void dfs(TreeNode* root,int dep,int par){

        if(!root)return;
        depth[root->val] = dep + 1;
        parent[root->val] = par;
        dfs(root->left,dep + 1,root->val);
        dfs(root->right,dep + 1,root->val);
    }
    bool isCousins(TreeNode* root, int x, int y) {
        dfs(root,0,0);
        if(depth[x] != depth[y] || parent[x] == parent[y])return false;
        return true;
        
    }
};
