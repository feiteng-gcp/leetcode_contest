/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {    
public:
    unordered_map<int,vector<int>>Map;
    bool isCousins(TreeNode* root, int x, int y) 
    {        
        DFS(root,0);
        // for (auto a:Map)
        //     cout<<a.first<<" "<<(a.second)[0]<<" "<<(a.second)[1]<<endl;
        
        return Map.find(x)!=Map.end() && Map.find(y)!=Map.end() && Map[x][1]==Map[y][1] && Map[x][0]!=Map[y][0];
    }
    
    void DFS(TreeNode* node, int depth)
    {
        if (node->left!=NULL)
        {
            Map[node->left->val] = {node->val,depth+1};
            DFS(node->left,depth+1);
        }
        if (node->right!=NULL)
        {
            Map[node->right->val] = {node->val,depth+1};
            DFS(node->right,depth+1);
        }
        
    }
};
