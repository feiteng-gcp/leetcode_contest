/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        queue<TreeNode*> nodes;
        TreeNode* xParent;
        TreeNode* yParent;
        int xLevel = -1, yLevel = -1;
        int level = 0;
        int size;
        TreeNode* cur;
        nodes.push(root);
        while(!nodes.empty()) {
            size = nodes.size();
            while (size-- != 0) {
                cur = nodes.front();
                nodes.pop();

                if (cur->left != NULL) {
                    if (cur->left->val == x) {
                        xParent = cur;
                        xLevel = level + 1;
                    }
                    if (cur->left->val == y) {
                        yParent = cur;
                        yLevel = level + 1;
                    }
                    nodes.push(cur->left);
                }
                if (cur->right != NULL) {

                    if (cur->right->val == x) {
                        xParent = cur;
                        xLevel = level + 1;
                    }
                    if (cur->right->val == y) {
                        yParent = cur;
                        yLevel = level + 1;
                    }
                    nodes.push(cur->right);
                }
            }
            level++;
        }
// cout << xLevel << " " << yLevel << " " << (xParent != yParent) << endl;
        return xLevel == yLevel && xParent != yParent;
    }
};
