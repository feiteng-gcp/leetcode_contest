/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        if(root->val==x ||root->val==y)
            return false;
        queue<TreeNode* > que;
        TreeNode* p=NULL;
        TreeNode* q=NULL;
        que.push(root);
        while(que.size()){
            int len=que.size();
            bool flag1=false,flag2=false;
            while(len){
                len--;
                TreeNode* r=que.front();
                que.pop();

                if(r->left){
                    que.push(r->left);
                    if(r->left->val==x){
                        flag1=true;
                        p=r;
                    }
                    if(r->left->val==y){
                        flag2=true;
                        q=r;
                    }
                }
                if(r->right){
                    que.push(r->right);
                    if(r->right->val==x){
                        flag1=true;
                        p=r;
                    }
                    if(r->right->val==y){
                        flag2=true;
                        q=r;
                    }
                }
            }
            if(flag1 &&!flag2) return false;
            if(flag2 &&!flag1) return false;
            if(flag2 &&flag1){
                return p!=q;
            }
        }
        return false;
    }
};
