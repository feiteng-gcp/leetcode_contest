/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        map<int, pair<TreeNode*, int>> ret;
        travesal(root, 0, ret);
        return ret[x].second == ret[y].second && ret[x].first != ret[y].first;
    }
    
    void travesal(TreeNode *root, int depth, map<int, pair<TreeNode*, int>> &ret) {
        if (root == nullptr) {
            return;
        }
        
        if (root->left != nullptr) {
            ret[root->left->val] = make_pair(root, depth + 1);
        }
        
        if (root->right != nullptr) {
            ret[root->right->val] = make_pair(root, depth + 1);
        }
        
        travesal(root->left, depth + 1, ret);
        travesal(root->right, depth + 1, ret);
    }
};
