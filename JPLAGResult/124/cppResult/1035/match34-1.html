/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    int depth[105];
    int par[105];
    

    void dfs(TreeNode* node, int d, int parent) {
        if (node == NULL) return;
        depth[node->val] = d;
        par[node->val] = parent;
        dfs(node->left, d + 1, node->val);
        dfs(node->right, d + 1, node->val);
    }
    
    bool isCousins(TreeNode* root, int x, int y) {
        for (int i = 0; i <= 100; i++) depth[i] = par[i] = -1;
        dfs(root, 0, -1);
        
        if (depth[x] == depth[y] && depth[x] != -1) {
            return par[x] != par[y];
        } else {
            return false;
        }
    }
};
