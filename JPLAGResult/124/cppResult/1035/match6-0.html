/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
	int deep[2], father[2]; 
	void dfs(TreeNode* pos, int index, int value, int _deep, int _father) {
		if (pos == NULL)
			return;
		if (pos->val == value) {
			deep[index] = _deep;
			father[index] = _father;
		}
		dfs(pos->left, index, value, _deep + 1, pos->val);
		dfs(pos->right, index, value, _deep + 1, pos->val);
	}
    bool isCousins(TreeNode* root, int x, int y) {
        dfs(root, 0, x, 0, 0);
        dfs(root, 1, y, 0, 0);
        return deep[0] == deep[1] && father[0] != father[1];
    }
};
