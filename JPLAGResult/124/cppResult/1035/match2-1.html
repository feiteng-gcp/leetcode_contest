/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    vector<int> depth;
    vector<int> parent;
    void travel(TreeNode* root, int d, int p) {
        if (!root)
            return;
        depth[root->val] = d;
        parent[root->val] = p;
        travel(root->left, d + 1, root->val);
        travel(root->right, d + 1, root->val);
    }
    bool isCousins(TreeNode* root, int x, int y) {
        depth = vector<int>(101, 0);
        parent = vector<int>(101, 0);
        travel(root, 0, NULL);
        return depth[x] == depth[y] && parent[x] != parent[y];
    }
};
