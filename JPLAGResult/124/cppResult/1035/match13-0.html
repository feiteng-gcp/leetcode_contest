/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    int x_level, y_level;
    TreeNode* x_p;
    TreeNode* y_p;
    
    void dfs(TreeNode* root, int lvl, int x, int y) {
        if (root->left) {
            if (root->left->val == x) {
                x_p = root;
                x_level = lvl + 1;
            }
            if (root->left->val == y) {
                y_p = root;
                y_level = lvl + 1;
            }
            dfs(root->left, lvl + 1, x, y);
        }
        if (root->right) {
            if (root->right->val == x) {
                x_p = root;
                x_level = lvl + 1;
            }
            if (root->right->val == y) {
                y_p = root;
                y_level = lvl + 1;
            }
            dfs(root->right, lvl + 1, x, y);
        }
    }
    
    bool isCousins(TreeNode* root, int x, int y) {
        
        dfs(root, 0, x, y);
        
        if (x_level == y_level && x_p != y_p)
            return true;
        
        return false;
        
        
    }
};
