/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        vector<int> InfoX;
        vector<int> InfoY;
        helper(InfoX, root, x, 0, -1);
        helper(InfoY, root, y, 0, -1);
        return InfoX[0]==InfoY[0] && InfoX[1]!=InfoY[1];
    }
    

    void helper(vector<int>& Info, TreeNode* root, int val, int depth, int father){
        if(root==NULL) return;
        if(root->val==val){
            Info.push_back(depth);
            Info.push_back(father);
        }
        helper(Info, root->left, val, depth+1, root->val);
        helper(Info, root->right, val, depth+1, root->val);
    }
    
    
    
};
