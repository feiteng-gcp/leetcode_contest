/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int parent[101];
    int depth[101];
    
    bool isCousins(TreeNode* root, int x, int y) {
        dfs(root, -1, 0);
        return depth[x] == depth[y] && parent[x] != parent[y];
    }
    
    void dfs(TreeNode* at, int from, int d) {
        if (!at) return;
        parent[at->val] = from;
        depth[at->val] = d;
        dfs(at->left, at->val, d+1);
        dfs(at->right, at->val, d+1);
    }
};
