class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        vector<int> p1; Dfs(root, x, p1);
        vector<int> p2; Dfs(root, y, p2);
        

        if (x != y && p1.size() == p2.size()) {
            p1.pop_back();
            p2.pop_back();
            return p1.back() != p2.back();
        }
        return false;
    }
    bool Dfs(TreeNode* root, int x, vector<int> & path) {
        if (!root) {
            return false;
        }
        
        path.push_back(root->val);
        if (root->val == x) {
        
            return true;
        }
        if (Dfs(root->left, x, path) || Dfs(root->right, x, path)) {
        
            return true;
        }
        path.pop_back();
        return false;
    }
};
