/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
int l1,l2,p1,p2;


void traverse(TreeNode* root, int x, int y,int l,int p){
    if(root==NULL)
        return;
    if(root->val==x){
        l1 = l;
        p1=p;
    }
    if(root->val==y){
        l2 = l;
        p2 = p;
    }
    traverse(root->left,x,y,l+1,root->val);
    traverse(root->right,x,y,l+1,root->val);
}

class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        l1=l2=p1=p2=-1;
        traverse(root,x,y,0,-1);
        if( l1!=l2 )
            return false;
        return (!(p1==p2));
    }
};
