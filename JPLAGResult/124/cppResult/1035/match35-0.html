/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    void dp(TreeNode* root, int x,int d,int p,pair<int,int>& ans){
        if(root==NULL) return;
        if(root->val==x){
            ans.first=d;
            ans.second=p;
        }else{
            dp(root->left,x,d+1,root->val,ans);
            dp(root->right,x,d+1,root->val,ans);
        }
    }
public:
    bool isCousins(TreeNode* root, int x, int y) {
        pair<int,int> a,b;
        dp(root,x,0,-1,a);
        dp(root,y,0,-1,b);
        if(a.first!=b.first) return false;
        return a.second!=b.second;
    }
};
