/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    int px, py;
    int dx, dy;
    int vx, vy;
    

    void dfs(TreeNode* root, int d, int p){
        if(!root) return;
        if(root->val==vx){ dx = d; px = p; }
        if(root->val==vy){ dy = d; py = p; }
        dfs(root->left, d+1, root->val);
        dfs(root->right, d+1, root->val);
    }
    
    bool isCousins(TreeNode* root, int x, int y) {
        px = -1;
        py = -1;
        dx = -1;
        dy = -2;
        vx = x;
        vy = y;
        dfs(root, 0, -1);
        if(dx == dy && px!=py) return true;
        return false;
        
    }
};
