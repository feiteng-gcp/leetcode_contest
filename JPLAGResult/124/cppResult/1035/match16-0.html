/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<int,int> par;
    map<int,int> dep;
    void rec(TreeNode * node, int depth, int parent)
    {
        if (!node) return;
        par[node->val] = parent;
        dep[node->val] = depth;
        rec(node->left, depth + 1, node->val);
        rec(node->right, depth + 1, node->val);
    }
    bool isCousins(TreeNode* root, int x, int y) {
        rec(root, 0, -1);
        return dep[x] == dep[y] && par[x] != par[y];
    }
};
