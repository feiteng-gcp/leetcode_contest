/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        TreeNode *px = nullptr, *py = nullptr;
        int hx = -1, hy = -1;
        go(root, x, y, px, py, 0, hx, hy);
        return hx != -1 && hx == hy && px != py;
    }

    void go(TreeNode *root, int x, int y, TreeNode *& px, TreeNode *& py, int h, int & hx, int & hy) {

        if (root) {
            if (root->left) {
                if (root->left->val == x) {
                    px = root;
                    hx = h + 1;
                }
                if (root->left->val == y) {
                    py = root;
                    hy = h + 1;
                }
                go(root->left, x, y, px, py, h + 1, hx, hy);
            }
            if (root->right) {
                if (root->right->val == x) {
                    px = root;
                    hx = h + 1;
                }
                if (root->right->val == y) {
                    py = root;
                    hy = h + 1;
                }
                go(root->right, x, y, px, py, h + 1, hx, hy);
            }
        }
    }
};
