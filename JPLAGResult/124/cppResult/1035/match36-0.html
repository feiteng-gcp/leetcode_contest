/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int d1, p1, d2, p2;
    
    void dfs1(TreeNode* node, int tar, int d, int p) {
        if (node == NULL) return;
        if (node->val == tar) {
            d1 = d;
            p1 = p;
            return;
        }
        dfs1(node->left, tar, d+1, node->val);
        dfs1(node->right, tar, d+1, node->val);
    }
    
    void dfs2(TreeNode* node, int tar, int d, int p) {
        if (node == NULL) return;
        if (node->val == tar) {
            d2 = d;
            p2 = p;
            return;
        }
        dfs2(node->left, tar, d+1, node->val);
        dfs2(node->right, tar, d+1, node->val);
    }
    
    bool isCousins(TreeNode* root, int x, int y) {
        dfs1(root, x, 0, -1);
        dfs2(root, y, 0, -1);
        if ((d1 == d2) && (p1 != p2)) return true;
        else return false;
    }
};
