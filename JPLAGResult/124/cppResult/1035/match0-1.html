/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        pair<int, int> r1 = {-1, -1};
        pair<int, int> r2 = {-1, -1};
        dfs(root, 0, -1, r1, r2, x, y);
        return r1.first == r2.first && r1.second != r2.second;
    }
private:
    void dfs(TreeNode* node, int depth, int parent, pair<int, int>& r1, pair<int, int>& r2, int x, int y) {
        if (node) {
            if (node->val == x) {
                r1.first = depth;
                r1.second = parent;
            }
            if (node->val == y) {
                r2.first = depth;
                r2.second = parent;
            }
            dfs(node->left, depth + 1, node->val, r1, r2, x, y);
            dfs(node->right, depth + 1, node->val, r1, r2, x, y);
        }
    }
};
