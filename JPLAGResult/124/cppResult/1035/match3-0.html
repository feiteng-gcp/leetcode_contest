/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
map<int,int> d,p;

class Solution {
public:
    void build(TreeNode *root, int parent, int depth)
    {
        if(root == NULL)return;
        p[root->val] = parent;
        d[root->val] = depth;
        build(root->left, root->val, depth+1);
        build(root->right, root->val, depth+1);
    }
    bool isCousins(TreeNode* root, int x, int y) {
        d.clear(); p.clear();
        build(root,-1,0);
        if(d[x] != d[y])return false;
        return p[x] != p[y];
    }
};
