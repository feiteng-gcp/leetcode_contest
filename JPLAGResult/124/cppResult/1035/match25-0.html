#include <bits/stdc++.h>
using namespace std;

// struct TreeNode {
//     int val;
//     TreeNode *left, *right;
//     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
// };

class Solution {
    map<int, int> depth;
    map<int, int> par;
    void dfs(TreeNode* t, int p, int d) {
        par[t->val] = p;
        depth[t->val] = d;
        if (t->left) {
            dfs(t->left, t->val, d + 1);
        }
        if (t->right) {
            dfs(t->right, t->val, d + 1);
        }
    }
public:
    bool isCousins(TreeNode* root, int x, int y) {
        dfs(root, -1, 0);
        return depth[x] == depth[y] && par[x] != par[y];
    }
};
