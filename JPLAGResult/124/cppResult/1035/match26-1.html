/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCousins(TreeNode* root, int x, int y) {
        int parentX = 0;
        int depthX = 0;
        int parentY = 0;
        int depthY = 0;
        
        queue<TreeNode*> bfs;
        queue<int> depth;
        while(!bfs.empty()) bfs.pop();
        while(!depth.empty()) depth.pop();
        bfs.push(root);
        depth.push(0);
        while(!bfs.empty()) {
            TreeNode* ft = bfs.front(); bfs.pop();
            int fi = depth.front(); depth.pop();

            if(ft->left) {
                bfs.push(ft->left);
                depth.push(fi + 1);
                if(ft->left->val == x) {
                    depthX = fi + 1;
                    parentX = ft->val;
                }
                if(ft->left->val == y) {
                    depthY = fi + 1;
                    parentY = ft->val;
                }
            }
            if(ft->right) {
                bfs.push(ft->right);
                depth.push(fi + 1);
                if(ft->right->val == x) {
                    depthX = fi + 1;
                    parentX = ft->val;
                }
                if(ft->right->val == y) {
                    depthY = fi + 1;
                    parentY = ft->val;
                }
            }
        }
        
        return (depthX == depthY && parentX != parentY);
    }
};
