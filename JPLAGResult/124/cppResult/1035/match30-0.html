/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int getDepth(TreeNode *root, int x) {
        if(root == nullptr) {
            return -1;
        }
        if(root->val == x) {
            return 0;
        }
        int tmp = getDepth(root->left, x);
        if(tmp != -1) {
            return tmp + 1;
        }
        tmp = getDepth(root->right, x);
        if(tmp != -1) {
            return tmp+1;
        }
        return tmp;
    }
    int getParent(TreeNode *root, int x) {
        if(root == nullptr) {
            return -1;
        }
        if(root->left && root->left->val == x) {
            return root->val;
        }
        if(root->right && root->right->val == x) {
            return root->val;
        }
        int tmp = getParent(root->left, x);
        if(tmp != -1) {
            return tmp;
        }
        return getParent(root->right, x);
    }
public:
    bool isCousins(TreeNode* root, int x, int y) {
        if(getDepth(root, x) == getDepth(root, y) && getParent(root, x) != getParent(root, y)) {
            return true;
        }
        return false;
    }
};
