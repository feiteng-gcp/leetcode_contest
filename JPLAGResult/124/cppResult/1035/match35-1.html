/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    void helper(vector<pair<int, int>> &lookup, TreeNode *root, int depth, int parent) {
        if (!root) return;

        lookup[root->val] = {depth, parent};
        helper(lookup, root->left, depth+1, root->val);
        helper(lookup, root->right, depth+1, root->val);
    }
public:
    bool isCousins(TreeNode* root, int x, int y) {
        vector<pair<int, int>> lookup(101);
        helper(lookup, root, 0, -1);
        if (x >= 1 && x <= 100 && y >= 1 && y <= 100)
            return lookup[x].first == lookup[y].first && lookup[x].second != lookup[y].second;
        return false;
    }
};
