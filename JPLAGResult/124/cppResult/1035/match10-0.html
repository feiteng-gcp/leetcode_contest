/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* px = nullptr;
    TreeNode* py = nullptr;
    int dx = -1;
    int dy = -1;
    
    void traverse(TreeNode*parent, TreeNode* root, int x, int y, int d = 0) {
        if (!root) return;
        if (root->val == x) { px = parent; dx = d; };
        if (root->val == y) { py = parent; dy = d; };
        traverse(root,root->left,x,y,d+1);
        traverse(root,root->right,x,y,d+1);
    }
    
    bool isCousins(TreeNode* root, int x, int y) {
        traverse(nullptr, root,x,y);
        return (dx==dy && px != py);
    }
};
