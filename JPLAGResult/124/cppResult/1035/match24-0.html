/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    vector<int>dep;
    vector<int>fa;

public:
    void dfs(TreeNode* cur, int fav, int depth, int x, int y) {
        if (cur == NULL)
            return;
        if (x == cur -> val || y == cur -> val) {
            dep.push_back(depth);
            fa.push_back(fav);
        }
        dfs(cur -> left, cur -> val, depth + 1, x, y);
        dfs(cur -> right, cur -> val, depth + 1, x, y);
    }
    bool isCousins(TreeNode* root, int x, int y) {
        dep.clear();
        fa.clear();
        dfs(root, -1, 1, x, y);
        if (dep.size() == 2 && dep[0] == dep[1] && fa[0] != fa[1])
            return true;
        return false;
    }
};
