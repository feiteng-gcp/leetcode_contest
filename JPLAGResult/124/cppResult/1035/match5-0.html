/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode *px, *py;
    int dx; int dy;
    bool isCousins(TreeNode* root, int x, int y) {
        px = py = NULL;
        dx = dy = 0;
        DFS(root, x, y, 0, NULL);
        return dx == dy && px != py;        
    }
    
    void DFS(TreeNode* root, int x, int y, int d, TreeNode *pre) {
        if (root == NULL) return;
        if (root->val == x) {
            dx = d;
            px = pre;
        }
        if (root->val == y) {
            dy = d;
            py = pre;
        }
        DFS(root->left, x, y, d + 1, root);
        DFS(root->right, x, y, d + 1, root);
    }
};
