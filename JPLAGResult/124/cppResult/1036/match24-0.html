class Solution {
public:
    int dx[4] = {-1,1,0,0};
    int orangesRotting(vector<vector<int>>& grid) {
        int N = grid.size();
        int M = grid[0].size();
        int res = 0;
        while(1) {
            vector<vector<int>> ng = grid;
            int did = 0;
            for (int i = 0; i < N; ++i) {
                for (int j = 0; j < M; ++j) {
                    int c = 0;
                    for (int d = 0; d < 4; ++d) {
                        int x = i+dx[d];
                        int y = j+dx[3-d];
                        if (x<0 || y<0 || x>=N || y>=M) continue;
                        if (grid[x][y]==2) ++c;
                    }
                    if (c>0 && ng[i][j]==1) {
                        did++;
                        ng[i][j]=2;
                    }
                }
            }
            if (!did) break;
            grid = ng;
            ++res;
        }
        for (int i = 0; i < N; ++i) 
                for (int j = 0; j < M; ++j) 
                    if (grid[i][j]==1) return -1;
        return res;
    }
};
