class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        
        int count = 0;
        queue<pair<int, int>>   q;

        for(int i = 0; i < grid.size(); i++) {
            for(int j = 0; j < grid[0].size(); j++) {
                if(grid[i][j] == 2) {
                    q.push({i, j});
                } else if(grid[i][j] == 1)
                    count++;
            }
        }
        if(count == 0)  return 0;

        vector<vector<int>> dir = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        int res = 0;
        while(!q.empty()) {
            int size = q.size();
            //cout<<size<<endl;
            while(size-- > 0) {
                int x = q.front().first;

                int y = q.front().second;
                q.pop();
                for(auto d:dir) {
                    int xx = x + d[0];
                    int yy = y + d[1];
                    if(xx < 0 || xx >= grid.size() || yy < 0 || yy >= grid[0].size() || grid[xx][yy] != 1)  continue;
                    count--;
                    grid[xx][yy] = 2;
                    q.push({xx, yy});
                    //cout<<count<<endl;
                }
            }
            res++;
        }
        if(count != 0)  return -1;
        return res - 1;
    }
};
