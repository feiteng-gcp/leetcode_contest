typedef vector<int> VI;
typedef vector<VI> VVI;

int dr[] = {-1, 0, 0, 1};
int dc[] = {0, -1, 1, 0};

class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        const int R = grid.size(), C = grid[0].size();
        for (int t = 0; ; ++t) {
            int nones = 0;
            for (int r = 0; r < R; ++r) {
                for (int c = 0; c < C; ++c) {
                    if (grid[r][c] == 1)
                        ++nones;
                }
            }
            if (nones == 0)
                return t;

            int nchanges = 0;
            VVI new_grid = grid;
            for (int r = 0; r < R; ++r) {
                for (int c = 0; c < C; ++c) {
                    if (grid[r][c] != 2) continue;
                    for (int k = 0; k < 4; ++k) {
                        int nr = r + dr[k], nc = c + dc[k];
                        if (nr < 0 || nr >= R || nc < 0 || nc >= C) continue;
                        if (new_grid[nr][nc] == 1) {
                            new_grid[nr][nc] = 2;
                            ++nchanges;
                        }
                    }
                }
            }
            if (nchanges == 0)
                break;
            grid = new_grid;
        }
        return -1;
    }
};
