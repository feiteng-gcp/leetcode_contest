int move_[][2] = {-1, 0, 1, 0, 0, 1, 0, -1};

class Solution {
public:
    
    
    int n, m;
    bool allRotten(vector< vector<int> > &grid) {

        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++)
                if (grid[i][j] == 1) return false;
        return true;
    }
    
    bool inGrid(int r, int c) {
        return r >= 0 && c >= 0 && r < n && c < m;
    }
    
    int orangesRotting(vector<vector<int>>& grid) {
        int res = 0;
        n = grid.size();
        m = grid[0].size();
        while (true) {
            if (allRotten(grid)) {
                return res;
            }
            

            bool change = false;
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                    if (grid[i][j] == 2) {
                        for (int k = 0; k < 4; k++) {
                            int ii = i + move_[k][0];
                            int jj = j + move_[k][1];
                            if (inGrid(ii, jj) && grid[ii][jj] == 1) {
                                grid[ii][jj] = 3;
                                change = true;
                            }
                        }
                    }
                }
            }
            
            if (!change) return -1;
            
            for (int i = 0; i < n; i++)
                for (int j = 0; j < m; j++)
                    if (grid[i][j] == 3) grid[i][j] = 2;
            
            res++;
        }
        return -1; // shouldn't come here
    }
};
