class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        if (grid.size() == 0 || grid[0].size() == 0) return 0;
        int count = 0;
        bool finished = true;
        while (finished) {
            vector<vector<int>> nextGrid = grid;
            finished = false;
            for (int i = 0; i < grid.size(); ++i) {
                for (int j = 0; j < grid[0].size(); ++j) {
                    if (grid[i][j] == 2) {
                        if (i > 0 && nextGrid[i - 1][j] == 1) {
                            nextGrid[i-1][j] = 2;
                            finished = true;
                        }
                        if (i < grid.size() -1&& nextGrid[i + 1][j] == 1) {
                            nextGrid[i+1][j] = 2;
                            finished = true;
                        }
                        if (j > 0 && nextGrid[i][j-1] == 1) {
                            nextGrid[i][j-1] = 2;
                            finished = true;
                        }
                        if (j < grid[0].size() -1&& nextGrid[i][j+1] == 1) {
                            nextGrid[i][j+1] = 2;
                            finished = true;
                        }
                    }
                }
            }
            if (finished) count++;
            grid = nextGrid;
        }
        
        for (auto i : grid) {
            for (auto j : i) {
                if (j == 1) return -1;
            }
        }
        return count;
    }
};
