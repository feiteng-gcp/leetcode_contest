class Pos {
public:
    int r, c, l;
    Pos(int r_, int c_, int l_): r(r_), c(c_), l(l_) {}
};

class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int tot = 0;
        queue<Pos> bfs;
        for (int r = 0; r < grid.size(); ++r) {
            for (int c = 0; c < grid[0].size(); ++c) {
                if (grid[r][c] != 0) {
                    ++tot;
                    if (grid[r][c] == 2) {
                        bfs.push(Pos(r, c, 0));
                    }
                }
            }
        }
        
        int count = 0;
        int steps = 0;
        while (!bfs.empty()) {
            Pos p = bfs.front();
            bfs.pop();
            ++count;
            steps = max(steps, p.l);
            if (p.r + 1 < grid.size() && grid[p.r + 1][p.c] == 1) {
                bfs.push(Pos(p.r + 1, p.c, p.l + 1));
                grid[p.r + 1][p.c] = 2;
            }
            
            if (p.r - 1 >= 0 && grid[p.r - 1][p.c] == 1) {
                bfs.push(Pos(p.r - 1, p.c, p.l + 1));
                grid[p.r - 1][p.c] = 2;
            }
            
            if (p.c + 1 < grid[0].size() && grid[p.r][p.c + 1] == 1) {
                bfs.push(Pos(p.r, p.c + 1, p.l + 1));
                grid[p.r][p.c + 1] = 2;
            }
            
            if (p.c - 1 >= 0 && grid[p.r][p.c - 1] == 1) {
                bfs.push(Pos(p.r, p.c - 1, p.l + 1));
                grid[p.r][p.c - 1] = 2;
            }
        }
        
        if(count != tot) {
            return -1;
        }
        else {
            return steps;
        }
    }
};
