class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        const int m = grid.size(), n = grid[0].size();
        vector<bool> discovered(m * n, false);

		queue<int> q;
        for (int row = 0; row < m; ++row) {
            for (int col = 0; col < n; ++col) {
                if (grid[row][col] == 2) {
                    q.push(row * n + col), discovered[row * n + col] = true;
                }
            }
        }

        int ans = 0;

        vector<vector<int>> ds = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
		for (int depth = 1; !q.empty(); ++depth) {
			for (int i = q.size(); i > 0; --i) {
				const int u = q.front(), row = u / n, col = u % n;
				q.pop();
				for (const vector<int> &d : ds) {
                    const int y = row + d[0], x = col + d[1];
                    if (y >= 0 && y < m && x >= 0 && x < n) {
                        if (grid[y][x] == 1 && !discovered[y * n + x]) {
                            q.push(y * n + x), grid[y][x] = 2, discovered[y * n + x] = true;
                            //cout << "Add " << x << "," << y << endl;
                        }
                    }
				}
			}
            if (!q.empty()) {
                ++ans;
                //cout << "ans = " << ans << endl;

            }
		}
        for (int row = 0; row < m; ++row) {
            for (int col = 0; col < n; ++col) {
                if (grid[row][col] == 1) {
                    return -1;
                }
            }
        }

        return ans;
    }
};
