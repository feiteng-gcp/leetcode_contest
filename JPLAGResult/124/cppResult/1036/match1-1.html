class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {

        queue<pair<int, int>> q;
        vector<vector<bool>> visited(grid.size(),vector<bool>(grid[0].size(),false));
        for(int i = 0;i<grid.size();i++){
            for(int j = 0;j<grid[i].size();j++){
                if(grid[i][j] == 2){
                    q.emplace(i,j);
                    visited[i][j] = true;
                }
            }
        }
        int res = 0;
        while (!q.empty()) {
            int sz = q.size();
            while (sz>0) {
                sz--;
                pair<int, int> p = q.front();
                q.pop();
                int x = p.first;

                int y = p.second;
                if(isValid(x-1, y, grid) &&!visited[x-1][y] && grid[x-1][y]==1){
                    visited[x-1][y]=true;
                    grid[x-1][y] = 2;
                    q.emplace(x-1,y);
                }
                if(isValid(x+1, y, grid) &&!visited[x+1][y] && grid[x+1][y]==1){
                    visited[x+1][y]=true;
                    grid[x+1][y] = 2;
                    q.emplace(x+1,y);
                }
                if(isValid(x, y-1, grid) &&!visited[x][y-1] && grid[x][y-1]==1){
                    visited[x][y-1]=true;
                    grid[x][y-1] = 2;
                    q.emplace(x,y-1);
                }
                if(isValid(x, y+1, grid) &&!visited[x][y+1] && grid[x][y+1]==1){
                    visited[x][y+1]=true;
                    grid[x][y+1] = 2;
                    q.emplace(x,y+1);
                }
            }
            res++;
        }

        for(int i = 0;i<grid.size();i++){
            for(int j = 0;j<grid[i].size();j++){
                if(grid[i][j] == 1){
                    return -1;
                }
            }
        }
        return res-1>=0?res-1:0;
    }
    
    bool isValid(int x,int y,vector<vector<int>>& grid){
        if(x>=0 && x<grid.size() && y>=0&&y<grid[0].size()){
            return true;
        }else{
            return false;
        }
    }
                
};
