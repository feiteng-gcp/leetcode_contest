class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int minutes = 0;
        int rot_count = 0;
        int total_count = 0;
        for (auto r : grid) {
            for (auto c : r) {
                if (c == 2) {
                    rot_count++;
                }
                if (c != 0) {
                    total_count++;
                }
            }
        }
        if (rot_count == total_count) return 0;
        
        int new_rot_count = -1;
        while (true) {

            new_rot_count = 0;
            for (int r = 0; r < grid.size(); r++) {
                for (int c = 0; c < grid[0].size(); c++) {
                    if (grid[r][c] == 2) {
                        if (r > 0 && grid[r - 1][c] == 1) grid[r - 1][c] = 3;
                        if (r < grid.size() - 1 && grid[r + 1][c] == 1) grid[r + 1][c] = 3;
                        if (c > 0 && grid[r][c - 1] == 1) grid[r][c - 1] = 3;
                        if (c < grid[0].size() - 1 && grid[r][c + 1] == 1) grid[r][c + 1] = 3;
                    }
                }
            }

            for (int r = 0; r < grid.size(); r++) {
                for (int c = 0; c < grid[0].size(); c++) {
                    if (grid[r][c] == 2) {
                        new_rot_count++;
                    }
                    if (grid[r][c] == 3) {
                        grid[r][c] = 2;
                        new_rot_count++;
                    }
                }
            }
            
            if (new_rot_count == rot_count) break;
            minutes++;
            rot_count = new_rot_count;
        }
        if (rot_count == total_count) {
            return minutes;
        } else {
            return -1;
        }
    }
};
