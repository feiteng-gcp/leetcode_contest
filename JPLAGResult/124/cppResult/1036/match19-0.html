class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        vector<vector<bool>> visited(grid.size()+5, vector<bool>(grid[0].size()+5));
        vector<int> deltaRow = {0, 0, 1, -1};
        vector<int> deltaCols = {1, -1, 0, 0};
        queue<pair<int,int>> que;
        int target = 0;
        int total = 0;
        
        for(int i = 0; i < grid.size(); ++i)
        {
            for(int j = 0; j < grid[i].size(); ++j)
            {
                if(grid[i][j] == 2)
                {
                    que.push(pair<int, int>(i, j));
                    visited[i][j] = true;
                    total++;
                }
                
                target += (grid[i][j] >  0);
            }
        }
        
        int ans = 0;
        while(!que.empty())
        {
            int sz = que.size();
            bool has = false;
            for(int i = 0; i < sz; ++i)
            {
                auto curr = que.front();
                int currRow = curr.first;
                int currCols = curr.second;
                que.pop();
                
                for(int j = 0; j < 4; ++j)
                {
                    int nextRow = currRow + deltaRow[j];
                    int nextCols = currCols + deltaCols[j];
                    
                    if(nextRow >= 0 && nextRow < grid.size() && nextCols >= 0 && nextCols < grid[0].size())
                    {
                        if(!visited[nextRow][nextCols] && grid[nextRow][nextCols] == 1)
                        {
                            grid[nextRow][nextCols] = 2;
                            que.push(pair<int, int>(nextRow, nextCols));
                            visited[nextRow][nextCols] = true;
                            total++;
                            has = true;
                        }
                    }
                }
            }
            ans += has;
        }
        
        if(total == target)
        {
            return ans;
        }
        else
        {
            return -1;
        }
    }
};
