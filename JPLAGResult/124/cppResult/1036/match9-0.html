class Solution {
public:
    int orangesRotting(vector<vector<int>>& G) {
        int N = G.size(); if (N == 0) return 0;
        int M = G[0].size(); if (M == 0) return 0;
        vector<vector<int>>T = G;
        queue<int> Q;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (G[i][j] == 2) {
                    Q.push(i * M + j);
                    T[i][j] = 0;
                }
            }
        }
        int dx[4] = {0, 0, 1, -1};
        int dy[4] = {1, -1, 0, 0};
        int t = 0;
        while (!Q.empty()) {
            int id = Q.front();Q.pop();
            int x = id/M, y = id%M;
            for (int d = 0; d < 4; d++) {
                int nx = dx[d] + x, ny = dy[d] + y;
                if (nx >= 0 && nx < N && ny >= 0 && ny < M) {
                    if (G[nx][ny] == 1) {
                        G[nx][ny] = 2;
                        T[nx][ny] = T[x][y] + 1;
                        t = max(t, T[nx][ny]);
                        Q.push(nx * M + ny);
                    }
                }
            }
        }
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (G[i][j] == 1) {
                    return -1;
                }
            }
        }
        return t;
    }
};
