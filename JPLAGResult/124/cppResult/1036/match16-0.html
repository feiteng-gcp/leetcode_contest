class Solution {
public:
    int n,m;
    vector<int> dirx{-1,0,1,0};
    vector<int> diry{0,1,0,-1};
    void bfs(vector<vector<int>>& grid,int i,int j,vector<vector<int>>& dist){
         vector<vector<int>> visited(n,vector<int> (m,0));
        queue<pair<pair<int,int>,int>> qu;
        qu.push(make_pair(make_pair(i,j),0));
        while(!qu.empty()){
            int x = qu.front().first.first;
            int y = qu.front().first.second;
            int d = qu.front().second;
            qu.pop();
            visited[x][y] = 1;
            for(int k = 0;k < 4;k++){
                int newx = x + dirx[k];
                int newy = y + diry[k];
                if(newx < 0 || newx >= n || newy < 0 || newy >= m || grid[newx][newy] != 1 || visited[newx][newy] == 1)continue;
                dist[newx][newy] = min(dist[newx][newy],d + 1);
                qu.push(make_pair(make_pair(newx,newy),d + 1));
            }
        }
    }
    int orangesRotting(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        vector<vector<int>> dist(n,vector<int> (m,INT_MAX));
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                if(grid[i][j] == 2){
                    bfs(grid,i,j,dist);
                }
            }
        }
        int res = 0;
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                // cout<<dist[i][j] <<" ";
                if(grid[i][j] == 1){
                    if(dist[i][j] == INT_MAX)return -1;
                    res = max(res,dist[i][j]);
                }
            }
            // cout<<endl;
        }
        return res;
    }
};
