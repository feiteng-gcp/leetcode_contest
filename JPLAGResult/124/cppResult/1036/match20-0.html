#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> pii;
int D[4][2] = {{1,0}, {-1,0}, {0,1}, {0,-1}};
int dp[15][15], N, M;
queue<pii> Q;
int FAIL = -2;

bool can(int a, int b) {return a >= 0 && a < N && b >= 0 && b < M;}

class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        while(!Q.empty()) Q.pop();
        memset(dp,-1,sizeof(dp));
        N = grid.size(), M = grid[0].size();
        for (int i=0;i<N;++i)
            for (int j=0;j<M;++j)
                if (grid[i][j] == 0) dp[i][j] = FAIL;
                else if (grid[i][j] == 2) {
                    dp[i][j] = 0;
                    Q.push(pii(i,j));
                }
        int best = 0;
        while (!Q.empty()) {
            auto t = Q.front();
            int tmp = dp[t.first][t.second];
            best = tmp;
            Q.pop();
            for (int k=0;k<4;++k) {
                int a = t.first + D[k][0];
                int b = t.second + D[k][1];
                if (can(a,b) && grid[a][b] == 1 && dp[a][b] == -1) {
                    dp[a][b] = tmp + 1;
                    Q.push(pii(a,b));
                }
            }
        }
        for (int i=0;i<N;++i)
            for (int j=0;j<M;++j)
                if (grid[i][j] == 1 && dp[i][j] == -1) return -1;
        return best;
    }
};
