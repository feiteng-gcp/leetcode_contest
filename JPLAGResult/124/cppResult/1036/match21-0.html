class Solution {
public:
    int dist[12][12];
    int dx[4] = {-1,0,1,0};
    int dy[4] = {0,-1,0,1};
    
    int orangesRotting(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                dist[i][j] = -1;
            }
        }
        int qj = 0;
        vector<pair<int, int> > q;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 2) {
                    dist[i][j] = 0;
                    q.push_back({i,j});
                }
            }
        }
        
        while(qj < q.size()) {
            pair<int, int> cp = q[qj];
            qj++;
            int cx = cp.first;
            int cy = cp.second;
            for (int dir = 0; dir < 4; dir++) {
                int nx = cx + dx[dir];
                int ny = cy + dy[dir];
                if (nx >= 0 && nx < n && ny >= 0 && ny < m) {
                    if (grid[nx][ny] == 1 && dist[nx][ny] == -1) {
                        dist[nx][ny] = dist[cx][cy] + 1;
                        q.push_back({nx,ny});
                    }
                }
            }
        }
        
        int mx = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    if (dist[i][j] == -1) {
                        return -1;
                    }
                    mx = max(mx, dist[i][j]);
                }
            }
        }
        return mx;
    }
};
