class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        vector<vector<int> > time(grid.size(),vector<int>(grid[0].size(),INT_MAX));
        queue<pair<int, int> > que;
        for (int i=0;i<grid.size();++i){
            for (int j=0;j<grid[0].size();++j){
                if (grid[i][j]==2){
                    que.push(make_pair(i,j));
                    time[i][j]=0;
                }
            }
        }
        int last=0;
        while (!que.empty()){
            int i=que.front().first;
            int j=que.front().second;
            last=time[i][j];
            printf("i=%d,j=%d,time=%d ",i,j,time[i][j]);
            que.pop();
            if (i>0&&grid[i-1][j]==1){
                grid[i-1][j]=2;
                time[i-1][j]=time[i][j]+1;
                que.push(make_pair(i-1,j));
            }
            if (i<grid.size()-1&&grid[i+1][j]==1){
                grid[i+1][j]=2;
                time[i+1][j]=time[i][j]+1;
                que.push(make_pair(i+1,j));
            }
            if (j>0&&grid[i][j-1]==1){
                grid[i][j-1]=2;
                time[i][j-1]=time[i][j]+1;
                que.push(make_pair(i,j-1));
            }
            if (j<grid[0].size()-1&&grid[i][j+1]==1){
                grid[i][j+1]=2;
                time[i][j+1]=time[i][j]+1;
                que.push(make_pair(i,j+1));
            }
        }
        bool ok=true;
        for (int i=0;i<grid.size();++i){
            for (int j=0;j<grid[0].size();++j){
                if (grid[i][j]==1){
                    ok=false;
                }
            }
        }
        if (!ok) return -1;
        return last;
    }
};
