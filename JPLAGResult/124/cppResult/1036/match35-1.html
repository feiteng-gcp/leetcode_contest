const int inf = (int)1e9;

    int dx[] = {0, 1, 0, -1};
    int dy[] = {1, 0, -1, 0};


class Solution {
public:
    
    int dp[11][11];
    int D[11][11];
    vector <vector < int > > g;
    bool used[11][11];
    int n, m;
    
    
    int orangesRotting(vector<vector<int>>& g) {
        this->g = g;
        n = g.size();
        m = g[0].size();
       
        int ans = 0;
        int cnt = 0;
        while (cnt < 100) {

            bool ok = true;
            for (int i=0;i<n;i++) {
                for (int j=0;j<m;j++) {
                    if (g[i][j] == 1) {
                        ok = false;
                        break;
                    }
                }
            }
            if (ok) break;
            vector < pair < int , int >> v;

            for (int i=0;i<n;i++){
                for (int j=0;j<m;j++) {
                    if (g[i][j] != 2) continue;
                    for (int ii=0;ii<4;ii++) {
                        int tox = dx[ii] + i;
                        int toy = dy[ii] + j;
                        if (tox >= 0 && tox < n && toy >= 0 && toy < m) {
                            if (g[tox][toy] == 1)  {
                                v.push_back(make_pair(tox, toy));
                            }
                        }
                    }
                }
            }
            
            for (pair < int , int > q: v) {
                int x = q.first;
                int y = q.second;
                g[x][y] = 2;
            }
            ans++;
            cnt++;
        }
        
        if (cnt > 99) return -1;
        return ans;
    }
};
