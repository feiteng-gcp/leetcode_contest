class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int n = static_cast<int>(grid.size());
        int m = static_cast<int>(grid[0].size());
        queue<pair<int, int>> q;

        vector<vector<int>> time(n, vector<int>(m));
        for (int r = 0; r < n; ++r) {
            for (int c = 0; c < m; ++c) {
                if (grid[r][c] == 2) {
                    q.push({r, c});
                }
            }
        }
        const int DIR_R[] = {0, 1, 0, -1};
        const int DIR_C[] = {1, 0, -1, 0};
        while (!q.empty()) {

            auto u = q.front();
            q.pop();
            int r = u.first, c = u.second;
            for (int k = 0; k < 4; ++k) {
                int tr = r + DIR_R[k];
                int tc = c + DIR_C[k];
                if (0 <= tr && tr < n && 0 <= tc && tc < m) {
                    if (grid[tr][tc] == 1) {
                        grid[tr][tc] = 2;
                        time[tr][tc] = time[r][c] + 1;
                        q.push({tr, tc});
                    }

                }
            }
        }
        int result = 0;
        for (int r = 0; r < n; ++r) {
            for (int c = 0; c < m; ++c) {
                if (grid[r][c] == 1) {
                    return -1;
                }
                if (grid[r][c] != 0) {
                    result = max(result, time[r][c]);
                }
            }
        }
        return result;
    }
};
