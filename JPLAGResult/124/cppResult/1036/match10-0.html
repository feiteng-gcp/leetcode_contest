class Solution {
private:
    struct Item {
        Item(int _i, int _j, int _level) {
            i = _i;
            j = _j;
            level = _level;
        }    
        int i;
        int j;
        int level;
    };
    
public:
    int orangesRotting(vector<vector<int>>& grid) {
        if(!grid.size()) return 0;
        
        int max_level = 0;
        
        queue<Item> q;
        for(int i = 0; i<grid.size(); ++i) {
            for(int j = 0; j<grid[0].size(); ++j) {
                if(grid[i][j] == 2) {
                    q.push(Item(i, j, 0));
                }
            }
        }
        
        while(q.size()) {
            Item item = q.front();
            q.pop();
            int i = item.i;
            int j = item.j;
            int level = item.level;
            
            if(i>0 && grid[i-1][j] == 1) {
                grid[i-1][j] = 2;
                q.push(Item(i-1, j, level+1));
                max_level = max(max_level, level+1);
            }
            if(i<grid.size()-1 && grid[i+1][j] == 1) {
                grid[i+1][j] = 2;
                q.push(Item(i+1, j, level+1));
                max_level = max(max_level, level+1);
            }
            if(j>0 && grid[i][j-1] == 1) {
                grid[i][j-1] = 2;
                q.push(Item(i, j-1, level+1));
                max_level = max(max_level, level+1);
            }
            if(j<grid[0].size()-1 && grid[i][j+1] == 1) {
                grid[i][j+1] = 2;
                q.push(Item(i, j+1, level+1));
                max_level = max(max_level, level+1);
            }
        }
        
        for(int i = 0; i<grid.size(); ++i) {
            for(int j = 0; j<grid[0].size(); ++j) {
                if(grid[i][j] == 1) {
                    return -1;
                }
            }
        }
        
        return max_level;
    }
};
