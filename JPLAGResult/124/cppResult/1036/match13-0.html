class Solution {
public:
    int orangesRotting(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        
        int tx[4] = {-1, 0, 1, 0};
        int ty[4] = {0, 1, 0, -1};
        queue<pair<int, int>> q;
        int dp[20][20];
        int inf = 1000;
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) {
                if (a[i][j] == 2) {
                    q.push({i, j});
                    dp[i][j] = 0;
                } else if (a[i][j] == 1) {
                    dp[i][j] = inf;
                }
            }

        
        int ans = 0;
        while (!q.empty()) {
            auto p = q.front();
            q.pop();
            int i = p.first;
            int j = p.second;
            //printf("%d %d, %d\n", i, j, dp[i][j]);
            a[i][j] = 2;
            int d = dp[i][j] + 1;
            for (int k = 0; k < 4; ++k) {
                int nx = i + tx[k];
                int ny = j + ty[k];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m || a[nx][ny] != 1)
                    continue;
                if (dp[nx][ny] <= d)
                    continue;
                dp[nx][ny] = d;
                q.push({nx, ny});
                ans = max(ans, d);
            }
        }

        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) if (a[i][j] == 1) {
                return -1;
                }
        return ans;
    }
};
