class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int cont=0;
        int di[4]={-1,1,0,0};
        int dj[4]={0,0,1,-1};
        
        while(true){
            vector<vector<int> > v=grid;
            
            for(int i=0;i<v.size();i++){
                for(int j=0;j<v[i].size();j++){
                    if(grid[i][j]==1){
                        for(int k=0;k<4;k++){
                            int X =i+di[k];
                            int Y =j+dj[k];
                            
                            if(X>=0 && X<v.size() && Y>=0 && Y<v[i].size()){
                                if(grid[X][Y]==2){
                                    v[i][j]=2;
                                }
                            }
                            
                        }
                        
                    }
                    
                }
            }
            
            
            if(v==grid)break;
            cont++;
            grid=v;
        }
        
        
        for(int i=0;i<grid.size();i++)
            for(int j=0;j<grid[i].size();j++)
                if(grid[i][j]==1)return -1;
                
        return cont;
    }
};
