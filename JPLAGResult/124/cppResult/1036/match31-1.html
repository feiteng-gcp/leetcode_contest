class Solution {
public:
    int dx[4] = {0, 0, 1, -1};
    int dy[4] = {1, -1, 0, 0};
    
    int orangesRotting(vector<vector<int>>& g) {
        int n = g.size(), m = g[0].size();
        int ans = 0;
        while (1) {

            bool ok = false;
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                    //printf("%d ", g[i][j]);
                    if (g[i][j] == 1) {
                        ok = true;
                    }
                }
                //printf("\n");
            }
            
            if (!ok) {
                return ans;
            }
            
            vector <pair <int, int> > ss;

            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                    if (g[i][j] == 1) {
                        bool f = false;
                        for (int d = 0; d < 4; d++) {
                            int x = i + dx[d], y = j + dy[d];
                            if (x >= 0 && x < n && y >= 0 && y < m) {
                                if (g[x][y] == 2) {
                                    f = true;
                                    break;
                                }
                            }
                        }
                        
                        if (f) {
                                ss.push_back(make_pair(i, j));
                            }
                    }
                }
            }
            
            if (ss.size() == 0) {
                return -1;
            }
            
            for (int i = 0; i < ss.size(); i++) {
                g[ss[i].first][ss[i].second] = 2;
            }
            ans++;
        }
        
        return ans;
    }
};
