struct loc { 
int x,y;
};
class Solution {
public:

    int orangesRotting(vector<vector<int>>& grid) {
    int c = 0;
    while(true) { 
        int count = 0;
      for(int i=0;i<grid.size();i++) { 
          for(int j=0;j<grid[i].size();j++) {
            if(grid[i][j] == 2)  { 
                if(i-1>=0 && grid[i-1][j] == 1) {grid[i-1][j] = 5; count++;}
                if(i+1<grid.size() && grid[i+1][j] == 1) { grid[i+1][j] = 5;count++; }
                if(j-1>=0 && grid[i][j-1] == 1) {grid[i][j-1] = 2; count++; }
                if(j+1<grid[i].size() && grid[i][j+1] == 1) {grid[i][j+1] = 5; count++; }
            }
          }
      }

       for(int i=0;i<grid.size();i++) { 
          for(int j=0;j<grid[i].size();j++) {
              if(grid[i][j] == 5) grid[i][j] =2;
          }
       }
      //  cout << "count == " << count << " c= " << c << endl;
        if(count == 0 ) break; 
        c++;
    }
        if( check(grid) ) return c;
        return -1;
    }
private:
    bool check(vector<vector<int>>& grid) { 
        for(int i=0;i<grid.size();i++) { 
          for(int j=0;j<grid[i].size();j++) {
              if(grid[i][j] == 1) return false;
          }
        }
        return true;
    }
