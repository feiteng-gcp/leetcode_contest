class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        // for each start(rotten orange), find the longest path
        // use BFS
        int m = grid.size();
        int n = grid[0].size();
        queue<pair<int, int>> q;
        for(int i = 0;i<m;i++){
            for(int j = 0;j < n;j++){
                if(grid[i][j] == 2){
                    q.push(make_pair(i, j));
                }
            }
        }
        // sentinel
        q.push(make_pair(-1, -1));
        int len = 0;
        while(!q.empty()){
            auto top = q.front();
            q.pop();
            if(top.first == -1){
                // sentinel
                // check do we have more orange to be rotten?
                if(q.empty()){
                    break;
                }
                ++len;
                q.push(make_pair(-1, -1));
                continue;
            }
            // let's rotten!
            int i = top.first, j = top.second;
            // up
            if(i > 0 && grid[i - 1][j] == 1){
                q.push(make_pair(i - 1, j));
                grid[i - 1][j] = 2;
            }
            // down 
            if(i < m - 1 && grid[i + 1][j] == 1){
                q.push(make_pair(i + 1, j));
                grid[i + 1][j] = 2;
            }
            // left
            if(j > 0 && grid[i][j - 1] == 1){
                q.push(make_pair(i, j - 1));
                grid[i][j - 1] = 2;
            }
            // right
            if(j < n - 1 && grid[i][j + 1] == 1){
                q.push(make_pair(i, j + 1));
                grid[i][j + 1] = 2;
            }
        }
        // our rotten process is finished
        // check remaining fresh?
        for(int i = 0;i<m;i++){
            for(int j = 0;j < n;j++){
                if(grid[i][j] == 1){
                    return -1;
                }
            }
        }
        return len;
    }
};
