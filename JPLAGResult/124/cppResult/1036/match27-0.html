class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int N = grid.size();
        int M = grid[0].size();

        vector<int> q;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (grid[i][j] == 2) {
                    q.push_back(i*100 + j);
                }
            }
        }

        int steps = 0;
        while (!q.empty()) {
            vector<int> qq = q;
            q.clear();
            steps++;
            
            for (int pos : qq) {
                int i = pos / 100;
                int j = pos % 100;
                
                int dx[] = { 1, -1, 0, 0 };
                int dy[] = { 0, 0, 1, -1 };

                for (int d = 0; d < 4; d++) {
                    int ii = i + dx[d];
                    int jj = j + dy[d];
                    if (ii < 0 || ii >= N || jj < 0 || jj >= M) continue;
                    if (grid[ii][jj] != 1) continue;
                    grid[ii][jj] = 2;
                    q.push_back(ii*100 + jj);
                }
            }
        }
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (grid[i][j] == 1) return -1;
            }
        }
        
        return max(0, steps - 1);
    }
};
