class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int ans = 0;
        int change, fresh;
        int m = grid.size();
        int n = grid[0].size();
        
        do
        {
            for (int i = 0; i < m; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if(grid[i][j] == 2)
                    {
                        if(grid[max(0, i-1)][j] == 1) grid[max(0, i-1)][j] = 21;
                        if(grid[min(m-1, i+1)][j] == 1) grid[min(m-1, i+1)][j] = 21;
                        if(grid[i][max(0, j-1)] == 1) grid[i][max(0, j-1)] = 21;
                        if(grid[i][min(n-1, j+1)] == 1) grid[i][min(n-1, j+1)] = 21;
                    }
                }
            }

            change = 0;
            fresh = 0;
            for (int i = 0; i < m; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if(grid[i][j] == 21)
                    {
                        grid[i][j] = 2;
                        change++;
                    }
                    if(grid[i][j] == 1)
                    {
                        fresh++;
                    }
                }
            }
            if(change > 0) ans++;
        } while(change > 0);
        
        if(fresh > 0) return -1;
        return ans;
    }
};
