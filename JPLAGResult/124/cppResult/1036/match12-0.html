class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        int cnt = 0;
        while (1) {
            vector<vector<int>> tmp(m, vector<int>(n, 0));
            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    if (grid[i][j] == 0 || grid[i][j] == 2) tmp[i][j] = grid[i][j];
                    else {
                        tmp[i][j] = 1;
                        if (i - 1 >= 0 && grid[i - 1][j] == 2) tmp[i][j] = 2;
                        if (i + 1 < m && grid[i + 1][j] == 2) tmp[i][j] = 2;
                        if (j - 1 >= 0 && grid[i][j - 1] == 2) tmp[i][j] = 2;
                        if (j + 1 < n && grid[i][j + 1] == 2) tmp[i][j] = 2;
                    }
                }
            }
            if (grid == tmp) {
                for (int i = 0; i < m; i++) {
                    for (int j = 0; j < n; j++) {
                        if (grid[i][j] == 1) return -1;
                        // cout << tmp[i][j] << ' ';
                    }
                    // cout << endl;
                }
                return cnt;
            } else {
                grid = tmp;
                ++cnt;
            }
        }
        return 0;
    }
};
