int vis[11][11];
int dir[4][2]={{-1,0},{0,-1},{1,0},{0,1}};
struct fff
{
	int x;
	int y;
};
class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        memset(vis,-1,sizeof(vis));
	int n=grid.size(),m=grid[0].size();
	queue<fff> q;
	fff now;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(grid[i][j]==2)
			{
				now.x=i;
				now.y=j;
				vis[i][j]=0;
				q.push(now);
			}
		}
	}
	while(!q.empty())
	{
		now=q.front();
		q.pop();
		for(int i=0;i<4;i++)
		{
			int nowx=now.x+dir[i][0],nowy=now.y+dir[i][1];
			if(nowx>=0&&nowx<n&&nowy>=0&&nowy<m&&grid[nowx][nowy]==1&&vis[nowx][nowy]==-1)
			{
				vis[nowx][nowy]=vis[now.x][now.y]+1;
				fff news;
				news.x=nowx;
				news.y=nowy;
				q.push(news);
			}
		}
	}
	int ans=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(vis[i][j]==-1&&grid[i][j]==1)return -1;
			ans=max(ans,vis[i][j]);
		}
	}
	return ans;
    }
};
