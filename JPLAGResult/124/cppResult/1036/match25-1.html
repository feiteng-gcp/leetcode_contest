class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int ans = 0;
        int h = grid.size(), w = grid[0].size();
        bool b = true;
        while (b)
        {
            vector<vector<int>> v(grid);
            b = false;
            for (int i = 0; i < h; i++)

                for (int j = 0; j < w; j++)
                {
                    if (grid[i][j] == 2)
                    {
                        if ((i > 0) && (grid[i - 1][j] == 1))
                        {
                            v[i - 1][j] = 2;
                            b = true;
                        }
                        if ((i < h - 1) && (grid[i + 1][j] == 1))
                        {
                            v[i + 1][j] = 2;
                            b = true;
                        }
                        if ((j > 0) && (grid[i][j - 1] == 1))
                        {
                            v[i][j - 1] = 2;
                            b = true;
                        }
                        if ((j < w - 1) && (grid[i][j + 1] == 1))
                        {
                            v[i][j + 1] = 2;
                            b = true;
                        }
                    }
                }
            if (b == true)
                ans++;
            grid = v;
        }
        for (auto row : grid)
            for (auto it : row)
                if (it == 1) return -1;
        return ans;
    }
};
