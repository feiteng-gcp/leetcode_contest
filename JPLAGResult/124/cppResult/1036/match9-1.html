class Solution {
public:
    const int INF = 0x3f3f3f3f;
    const int dir[4][2] = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
    
    bool inrange(int x, int y, int n, int m) {
        return 0 <= x && x < n && 0 <= y && y < m;
    }
    
    int orangesRotting(vector<vector<int>>& grid) {
        queue<int> que;
        int n = grid.size(), m = grid[0].size();
        vector<int> dist(n * m, INF);
        vector<bool> visit(n * m, false);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 2) {
                    visit[i * m + j] = true;
                    que.push(i * m + j);
                    dist[i * m + j] = 0;
                }
            }
        }
        
        while (!que.empty()) {

            int x = que.front() / m, y = que.front() % m;
            que.pop();
            for (int i = 0; i < 4; i++) {
                int nx = x + dir[i][0], ny = y + dir[i][1];
                if (inrange(nx, ny, n, m) && !visit[nx * m + ny] && grid[nx][ny] == 1) {
                    visit[nx * m + ny] = true;
                    que.push(nx * m + ny);
                    dist[nx * m + ny] = dist[x * m + y] + 1;

                }
            }
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1 && dist[i * m + j] == INF) {
                    return -1;
                }
            }
        }
        
        int ret = 0;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    ret = max(ret, dist[i * m + j]);
                }
            }
        }
        
        return ret == INF ? 0 : ret;
    }
};
