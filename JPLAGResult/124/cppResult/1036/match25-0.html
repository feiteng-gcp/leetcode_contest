#define _for(i,a,b) for(int i = (a);i < (b);i ++)
class Solution
{
	public:
		int orangesRotting(vector<vector<int>>& grid)
		{
			int left = 0;
			int sz1 = grid.size();
			int sz2 = grid[0].size();
			_for(i,0,sz1)
				_for(j,0,sz2)
					if(grid[i][j]==1)
						left ++;
			
			int prev = left;
			int rnt = 0;
			while(left)
			{
				prev = left;
				_for(i,0,sz1)
					_for(j,0,sz2)
					{
						if(grid[i][j]==2)
						{
							if(i+1<sz1&&grid[i+1][j]==1)
							{
								left--;
								grid[i+1][j]=-1;
							}
							if(i-1>=0&&grid[i-1][j]==1)
							{
								left--;
								grid[i-1][j]=-1;
							}
							if(j-1>=0&&grid[i][j-1]==1)
							{
								left--;
								grid[i][j-1]=-1;
							}
							if(j+1<sz2&&grid[i][j+1]==1)
							{
								left--;
								grid[i][j+1]=-1;
							}
						}
					}
					
				_for(i,0,sz1)
					_for(j,0,sz2)
						if(grid[i][j]==-1)
							grid[i][j] = 2;
				rnt ++;
				if(prev==left)
					return -1;
			}
			return rnt;
		}
};
