class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        if(grid.empty()) return -1;
        int m = grid.size();
        int n = grid[0].size();
        queue<pair<int,int>> q;
        vector<vector<int>> dir = {{1,0},{0,-1},{-1,0},{0,1}};
        int count = 0;
        int total = 0;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(grid[i][j] == 2){
                    count++;
                    q.push({i,j});
                }
                if(grid[i][j] != 0) total++;
            }
        }
        if(count == total) return 0;
        int res = 0;
        while(!q.empty()){
            int len = q.size();
            bool newAddedRotten = false;
            while(len){
                auto p = q.front(); q.pop(); len--;
                for(auto d : dir){
                    int i = d[0] + p.first;
                    int j = d[1] + p.second;
                    if(i < 0 || i >= m || j < 0 || j >= n || grid[i][j] != 1) continue;
                    count++;
                    grid[i][j] = 2;
                    q.push({i,j});
                    newAddedRotten = true;
                }
            }
            if(!newAddedRotten) break;
            res++;
        }
        return total == count ? res : -1;
    }
};
