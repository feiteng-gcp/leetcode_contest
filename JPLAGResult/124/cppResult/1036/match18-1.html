class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int R=grid.size(), C=grid[0].size(), count2=0, count1=0;
        queue<pair<int , int>> Q;

        for (int i=0; i<R; i++) {
            for (int j=0; j<C; j++){
                if (grid[i][j]==2)
                    Q.push({i, j}), count2++;
                if (grid[i][j]==1)
                    count1++;
            }                
        }
        if (Q.empty()) {
            if (count1>0)
              return -1;
            return 0;
        }

        vector<vector<int>> d={{0,1},{0,-1},{1,0},{-1,0}};

        int time=0;
        while (!Q.empty()) {
            time++;
            int len=Q.size();
            for (int i=0; i<len; i++) {
                auto p=Q.front();
                Q.pop();
                for (int j=0; j<4; j++) {
                    int r=p.first+d[j][0], c=p.second+d[j][1];
                    if (r>=0 && c>=0 && r<R && c<C && grid[r][c]==1) {
                        grid[r][c]=2;
                        Q.push({r,c});
                    }
                }
            }
        }
        for (int i=0; i<R; i++) {
            for (int j=0; j<C; j++){
                if (grid[i][j]==1)
                    return -1;
            }                
        }
        return time-1;
    }
};
