class Solution {
    vector<vector<int>> d = {{0, 1},{0, -1},{1, 0},{-1,0}};
public:
    int orangesRotting(vector<vector<int>>& grid) {
        queue<pair<int, int>> rotten;
        int m = grid.size();
        int n = grid[0].size();
        int tmp = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 2) {
                    rotten.push(make_pair(i,j));
                } else if (grid[i][j] == 1) {
                    tmp ++;
                }
            }
        }
        if (tmp == 0) return 0;
        if (rotten.empty()) return -1;
        int res = 0;
        while (!rotten.empty() && tmp>0) {
            res ++;
            int size = rotten.size();
            for (int i = 0; i < size; ++i) {
                auto p = rotten.front();
                rotten.pop();
                for (int j = 0; j < 4; ++j) {
                    int newI = p.first + d[j][0];
                    int newJ = p.second + d[j][1];
                    if (newI < 0 || newI >= m || newJ < 0 || newJ >= n) continue;
                    if (grid[newI][newJ] == 1) {
                        grid[newI][newJ] = 2;
                        tmp--;
                        rotten.push(make_pair(newI, newJ));
                    }
                }
            }
        }
        if (tmp > 0) return -1;
        return res;
     }
};
