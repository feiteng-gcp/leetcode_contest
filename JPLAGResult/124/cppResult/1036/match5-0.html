class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int N = grid.size(), M = grid[0].size();
        for (int t = 0; t <= 100; t++) {
            int all_rotten = true;
            for (int i = 0; i < N; i++) {
                for (int j = 0; j < M; j++) {
                    if (grid[i][j] == 1) {
                        all_rotten = false;
                    }
                }
            }
            if (all_rotten) {
                return t;
            }
            vector<vector<int>> next = grid;
            for (int i = 0; i < N; i++) {
                for (int j = 0; j < M; j++) {
                    if (grid[i][j] == 1) {
                        if (i > 0 && grid[i - 1][j] == 2) next[i][j] = 2;
                        if (i < N-1 && grid[i + 1][j] == 2) next[i][j] = 2;
                        if (j > 0 && grid[i][j - 1] == 2) next[i][j] = 2;
                        if (j < M-1 && grid[i][j + 1] == 2) next[i][j] = 2;
                    }
                }
            }
            grid = next;
        }
        return -1;
    }
};
