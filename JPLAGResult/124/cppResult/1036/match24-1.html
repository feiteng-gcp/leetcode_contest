class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) 
    {
        int M = grid.size();
        int N = grid[0].size();
        
        queue<pair<int,int>>q;
        for (int i=0; i<M; i++)
            for (int j=0; j<N; j++)
                if (grid[i][j]==2)
                {
                    q.push({i,j});
                    grid[i][j]=-1;
                }

        auto dir = vector<pair<int,int>>({{1,0},{-1,0},{0,1},{0,-1}});

        int count = 0;
        while (q.size()>0)
        {
            int step = q.size();
            int flag = 0;
            for (int i=0; i<step; i++)
            {                
                int x = q.front().first;

                int y = q.front().second;
                q.pop();
                for (int k=0; k<4; k++)
                {
                    int xx = x+dir[k].first;
                    int yy = y+dir[k].second;
                    if (xx<0||xx>=M||yy<0||yy>=N) continue;
                    if (grid[xx][yy]==-1||grid[xx][yy]==2) continue;
                    if (grid[xx][yy]==1)
                    {       
                        flag = 1;
                        q.push({xx,yy});
                        grid[xx][yy]=-1;
                        // cout<<xx<<" "<<yy<<endl;
                    }
                }
            }            

            if (flag) count+=1;            
            // cout<<count<<endl;
        }
        
        for (int i=0; i<M; i++)
            for (int j=0; j<N; j++)
                if (grid[i][j]==1) return -1;
        
        return count;
                    
    }
};
