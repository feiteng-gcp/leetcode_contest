using namespace std;


class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int dx[] = {-1, 0, 1, 0};
        int dy[] = { 0, 1, 0,-1};
        
        map<pair<int,int>, int> mark;
        
        queue<tuple<int, int, int>> q;
        

        int ans = 0;
        
        for(int i = 0; i < grid.size(); ++i) {
            for(int j = 0; j < grid[i].size(); ++j) {
                if(grid[i][j] == 0) {
                    continue;
                }
                ans++;
                if(grid[i][j] == 1) {
                    continue;
                }
                q.push(make_tuple(i, j, 0));
                mark.insert(make_pair(make_pair(i, j), 0));
            }
        }
        
        int anw = 0;
        while(q.empty() == false) {
            auto f = q.front();
            q.pop();
            

            int x = get<0>(f);
            int y = get<1>(f);
            int s = get<2>(f);
            
           // cout << x << " " << y << " " << s << endl;
            
            for(int i = 0; i < 4; ++i) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                
                if(nx < 0 || nx >= grid.size()) {continue;}
                if(ny < 0 || ny >= grid[nx].size()) {continue;}
                if(grid[nx][ny] != 1) {continue;}
                
                grid[nx][ny] = 2;
                q.push(make_tuple(nx, ny, s+1));
                mark.insert(make_pair(make_pair(nx, ny), s + 1));
                anw = s+1;
            }
        }
        cout << ans << endl;
        cout << mark.size() << endl;
        if(mark.size() == ans) {
            return anw;
        } 
        return -1;
    }
};
