class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int r = grid.size();
        int c = grid[0].size();
        queue<pair<int,int>> q;
        int ans = 0;
        int dy[4] = {1, 0, -1, 0};
        int dx[4] = {0, 1, 0, -1};

        vector<vector<int>> d(r, vector<int>(c, 1e9));
        for(int i = 0; i < r; i++) {
            for(int j = 0; j < c; j++) {
                if(grid[i][j] == 2) {
                    q.push({i, j});
                    d[i][j] = 0;
                }
            }
        }
        while(!q.empty()) {
            int y = q.front().first;

            int x = q.front().second;
            q.pop();
            for(int i = 0; i < 4; i++) {
                int ny = y + dy[i];
                int nx = x + dx[i];
                if(ny < 0 || nx < 0 || ny >= r || nx >= c) continue;
                if(grid[ny][nx] != 1 || d[y][x] + 1 >= d[ny][nx]) continue;
                d[ny][nx] = d[y][x] + 1;
                q.push({ny, nx});
            }
        }

        for(int i = 0; i < r; i++) {
            for(int j = 0; j < c; j++) {
                if(grid[i][j] == 1) {
                    if(d[i][j] == 1e9) return -1;
                    ans = max(ans, d[i][j]);
                }
            }
        }
        return ans;
    }
};
