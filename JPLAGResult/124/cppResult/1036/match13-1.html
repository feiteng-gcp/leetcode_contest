const vector<vector<int>> walks = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

struct RottenOrange {
    int x;
    int y;
    int step;
    
    RottenOrange(int x, int y, int step) : x(x), y(y), step(step){}
};


class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int num_rotten = 0;

        int sum = 0;
        queue<RottenOrange> rottens;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 2) {
                    num_rotten++;
                    rottens.emplace(i, j, 0);
                }
                else if (grid[i][j] == 1)
                    sum++;
            }
        int cur_step = 0;
        while (!rottens.empty()) {

            RottenOrange& cur_rotten = rottens.front();
            //printf("%d %d %d\n", cur_rotten.x, cur_rotten.y, cur_rotten.step);
            for (int i = 0; i < 4; i++) {
                int next_x = cur_rotten.x + walks[i][0];
                int next_y = cur_rotten.y + walks[i][1];
                if (next_x < 0 || next_x >= n || next_y < 0 || next_y >= m || grid[next_x][next_y] != 1)
                    continue;
                grid[next_x][next_y] = 2;

                cur_step = cur_rotten.step + 1;
                rottens.emplace(next_x, next_y, cur_step);
            }
            rottens.pop();
        }
        
        //printf("%d\n", cur_step);
        
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++)
                if (grid[i][j] == 1)
                    return -1;
        return cur_step;
    }
};
