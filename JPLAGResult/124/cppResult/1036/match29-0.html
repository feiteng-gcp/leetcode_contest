class Solution {
public:
 
int orangesRotting(vector<vector<int>>& a) {
    vector<pair<int, int>> ds = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    int n = a.size(), m = a[0].size();
    vector<vector<int>> dst(n, vector<int>(m, -1));
    vector<pair<int, int>> q;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (a[i][j] == 2) {
                q.push_back({i, j});
                dst[i][j] = 0;
            }
        }
    }

    int ans = 0;
    for (int qi = 0; qi < q.size(); qi++) {
        int i = q[qi].first, j = q[qi].second;
        int d = dst[i][j];
        ans = max(ans, d);
        for (auto pr: ds) {
            int nxtI = i + pr.first, nxtJ = j + pr.second;
            if (nxtI < 0 || nxtI == n || nxtJ < 0 || nxtJ == m) {
                continue;
            }
            if (a[nxtI][nxtJ] == 1) {
                a[nxtI][nxtJ] = 2;
                q.push_back({nxtI, nxtJ});
                dst[nxtI][nxtJ] = d + 1;
            }
        }
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (a[i][j] == 1) {
                return -1;
            }
        }
    }
    return ans;
}
};
