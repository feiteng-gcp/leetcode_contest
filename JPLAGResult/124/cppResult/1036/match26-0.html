class Solution {
private:
  const int dx[4] = {0, 0, -1, 1};
  const int dy[4] = {-1, 1, 0, 0};
public:
  int orangesRotting(vector<vector<int>>& grid) {
    int n = grid.size();
    int m = grid[0].size();
    int ans = 0;
    queue<pair<int, int>> q;
    map<pair<int, int>, int> dist;
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < m; ++j) {
        if (grid[i][j] == 2) {
          q.push(make_pair(i, j));
          dist[make_pair(i, j)] = 0;
        }
      }
    }
    while (!q.empty()) {
      auto now = q.front();
      for (int d = 0; d < 4; ++d) {
        int nx = now.first + dx[d];
        int ny = now.second + dy[d];
        if (nx < 0 || nx >= n || ny < 0 || ny >= m) {
          continue;
        }
        if (grid[nx][ny] == 1) {
          grid[nx][ny] = 2;
          dist[make_pair(nx, ny)] = dist[now] + 1;
          ans = max(ans, dist[make_pair(nx, ny)]);
          q.push(make_pair(nx, ny));
        }
      }
      q.pop();
    }
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < m; ++j) {
        if (grid[i][j] == 1) {
          return -1;
        }
      }
    }
    return ans;
  }
};
