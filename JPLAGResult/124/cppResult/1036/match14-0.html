vector<pair<int, int>> DIRS = {
    {0, 1},
    {0, -1},
    {1, 0},
    {-1, 0},
};

class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        vector<pair<int, int>> rotten;
        int fresh = 0;
        
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[i].size(); ++j) {
                if (grid[i][j] == 1) {
                    fresh += 1;
                } else if (grid[i][j] == 2) {
                    rotten.emplace_back(i, j);
                }
            }
        }
        
        int steps = 0;
        while (!rotten.empty()) {
            vector<pair<int, int>> rotting;
            for (auto & x : rotten) {
                for (auto & dir : DIRS) {
                    int ii = x.first + dir.first;
                    int jj = x.second + dir.second;
                    if (ii >= 0 && ii < grid.size() && jj >= 0 && jj < grid[ii].size()) {
                        if (grid[ii][jj] == 1) {
                            grid[ii][jj] = 2;
                            fresh -= 1;
                            rotting.emplace_back(ii, jj);
                        }
                    }
                }
            }
            std::swap(rotting, rotten);
            steps += 1;
        }
        
        if (fresh == 0) {
            return std::max(0, steps - 1);
        }
        return -1;
    }
};
