class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        /*grid[1][0] = 0;
        grid[1][1] = 1;
        grid[1][2] = 1;
        
        grid[2][0] = 1;
        grid[2][1] = 0;
        grid[2][2] = 1;*/
        
        vector<int> cur_x;
        vector<int> cur_y;
        
        vector<int> next_x;
        vector<int> next_y;
        
        
        int m = grid.size();
        
        if (m == 0)
        {
            return 0;
        }
        
        int n = grid[0].size();
        

        for (int i=0; i<m; i++)
        {
            for (int j=0; j<n; j++)
            {
                if (grid[i][j] == 2)
                {
                    cur_x.push_back(i);
                    cur_y.push_back(j);
                }
            }
        }
        
        int result = 0;
        while (cur_x.size() > 0)
        {
            while (cur_x.size() > 0)
            {
                int cx = cur_x[0];

                int cy = cur_y[0];
                
                cur_x.erase(cur_x.begin());
                cur_y.erase(cur_y.begin());
                
                //up
                if (cx > 0 && grid[cx-1][cy] == 1)
                {
                    next_x.push_back(cx-1);
                    next_y.push_back(cy);
                    grid[cx-1][cy] = 2;
                }
                
                //down
                if (cx+1 < m && grid[cx+1][cy] == 1)
                {
                    next_x.push_back(cx+1);
                    next_y.push_back(cy);
                    grid[cx+1][cy] = 2;
                }
                
                //left
                if (cy > 0 && grid[cx][cy-1] == 1)
                {
                    next_x.push_back(cx);
                    next_y.push_back(cy-1);
                    grid[cx][cy-1] = 2;
                }
                
                //right
                if (cy+1 < n && grid[cx][cy+1] == 1)
                {
                    next_x.push_back(cx);
                    next_y.push_back(cy+1);
                    grid[cx][cy+1] = 2;
                }
            }
            
            if (next_x.size() > 0)
            {
                result ++;
                cur_x = next_x;
                cur_y = next_y;
                
                next_x.clear();
                next_y.clear();
            }
            else
            {
                break;

            }
        }
        
        for (int i=0; i<m; i++)
        {
            for (int j=0; j<n; j++)
            {
                if (grid[i][j] == 1)
                {
                    return -1;
                }
            }
        }
        
        return result;
    }
};
