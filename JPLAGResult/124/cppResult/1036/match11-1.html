class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int r=grid.size(),c=grid[0].size();
        int minute=0;
        while(1){
            bool finish=true;

            bool fresh=true;
            for(int i=0;i<r;i++){
                for(int j=0;j<c;j++){
                    if(grid[i][j]==1 ||grid[i][j]==3) finish=false;
                    else if(grid[i][j]==0) continue;
                    else{

                        if(i+1<r &&grid[i+1][j]==1) grid[i+1][j]=3;
                        if(i-1>=0 &&grid[i-1][j]==1) grid[i-1][j]=3;
                        if(j+1<c &&grid[i][j+1]==1) grid[i][j+1]=3;
                        if(j-1>=0 &&grid[i][j-1]==1) grid[i][j-1]=3;
                    }
                }
            }
            if(finish) return minute;

            minute++;
            for(int i=0;i<r;i++){
                for(int j=0;j<c;j++){
                    if(grid[i][j]==3){
                        grid[i][j]=2;
                        fresh=false;
                    }
                }
            }
            if(fresh) return -1;
        }
        return -1;
    }
};
