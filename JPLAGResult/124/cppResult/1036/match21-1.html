using namespace std;

int dx[4] = {0 , 0 , -1, 1};
int dy[4] = {1 , -1 , 0 , 0 };

class Solution {
	int board[20][20];
	
	vector<vector<int> > g;
	int n , m;
	bool avail (int x , int y) {
		if (x < 0 || x >= n || y < 0 || y >= m ) return false;
		if (g[x][y] != 1 ) return false;
		return true;
	}
	
	public:int orangesRotting(vector<vector<int> >& grid) {
       	g = grid;
       	n = g.size();

       	m = g[0].size();
    	queue< pair<int,int> > qe;
    	
    	for (int i = 0; i < n; i ++ ) {
			for (int j = 0; j < m; j ++ )
				board[i][j] = -1;
		}

		int c = 0;
		for (int i = 0; i < n; i ++ ) {
			for (int j = 0; j < m; j ++ ) {
				if ( g[i][j] == 1 )
					c ++;
				if (g[i][j] == 2 ) {
					board[i][j] = 0;
					qe.push(mp(i , j));
				}
			}
		}
		if (c == 0 ) return 0;
		while ( !qe.empty() ) {

			pair<int, int > cur = qe.front();
			qe.pop();
			cout << cur.xx << " " << cur.yy << endl;
			for (int i = 0; i < 4; i ++ ) {
				int nx = cur.xx + dx[i];
				int ny = cur.yy + dy[i];
				
				if (  avail(nx ,ny ) && (board[nx][ny] == -1 || board[nx][ny] > board[cur.xx][cur.yy] + 1 ) ) {
					board[nx][ny] = board[cur.xx][cur.yy] + 1 ;
					qe.push( mp (nx , ny) );

				}
			}
		}
		
		int res = -1;
		
		for (int i = 0; i < n; i ++  ){
			for (int j = 0; j < m; j ++ ) {
				if ( g[i][j] == 1 ) {
					if (board[i][j] == -1 )return -1;
					res = max (res , board[i][j]);
				}
			}
		}
		
		return res;
    }
};
