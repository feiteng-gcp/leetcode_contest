class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        int n = A.size();
        vector<vector<bool>> adjMat(A.size()+5, vector<bool>(A.size()+5));
        unordered_map<long long, long long> cnt;
            
        for(int i = 0; i < A.size(); ++i)
        {
            cnt[A[i]]++;
            for(int j = i+1; j < A.size(); ++j)
            {
                if(isPerfect(A[i]+A[j]))
                {
                    adjMat[i][j] = adjMat[j][i] =true;
                }
            }
        }
        
        vector<vector<long long>> dp((1<<14), vector<long long>(14));

        for(int i = 0; i < n; ++i) dp[0][i] = 1;
        
        for(int i = 0; i < (1 << n); ++i)
        {
            for(int j = 0; j < n; ++j)
            {
                if(i & (1 << j))
                {
                    int prev = i ^ (1 << j);
                    if(prev == 0)
                    {
                        dp[i][j]++;
                        continue;
                    }
                    for(int k = 0; k < n; ++k)
                    {
                        if(adjMat[j][k])
                        {
                            dp[i][j] += dp[prev][k];
                        }
                    }
                }
            }
        }
        
        
        long long ans = 0;
        for(int i = 0; i < n; ++i)
        {
            ans += dp[(1<<n)-1][i];
        }
        
        vector<long long> fact(14);
        fact[0] = 1;

        for(long long i = 1; i <= 12; ++i)
        {
            fact[i] = i * fact[i-1];
        }
        
        for(auto kv : cnt)
        {
            ans /= fact[kv.second];
        }
        return ans;
    }
    
    bool isPerfect(int n)
    {
         if (n < 0)
            return false;
        int root(round(sqrt(n)));
        return n == root * root;
    }
};
