class Solution {
public:
    int numSquarefulPerms(vector<int>& nums) {
        int res = 0;
        vector<int> cur;
        vector<bool>    visited(nums.size(), false);
        sort(nums.begin(), nums.end());
        dfs(nums, visited, res, cur);
        return res;
    }
    

   void dfs(vector<int> &nums, vector<bool> &visited, int &res, vector<int> &cur) {
        if(cur.size() >= nums.size()) {
            res++;
            return;
        }
        for(int i = 0; i < nums.size(); i++) {
            if(visited[i])  continue;
            if(i > 0 && nums[i] == nums[i-1] && visited[i-1] == false)   continue;
            if(cur.size() > 0 && !check(cur.back(), nums[i])) continue;
            visited[i] = true;
            cur.push_back(nums[i]);
            dfs(nums, visited, res, cur);
            cur.pop_back();
             visited[i] = 0;
        }
    }
    
    
    bool check(int x, int y) {
        int sum = x + y;
        int sq = mySqrt(sum);
        if(sq*sq != sum)    return false;
        return true;
    }
    
    
    int mySqrt(int x) {
        if(x == 0)  return 0;
        int l = 1, r = x;
        while(l+1<r) {
            int mid = l + (r - l)/2;
            if(mid <= x/mid) {
                l = mid;
            } else
                r = mid;
        }
        if(r <= x/r) return r;
        return l;
    }
};
