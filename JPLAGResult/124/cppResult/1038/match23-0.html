  class Solution {
  public:
	  vector<bool>vis;
	  int ct = 0;
	  vector<vector<int>>adj;
	  vector<int>arr;
	  void solve(int curr, int t,int n) {
		  if (curr == n)ct++;
		  for (int i = 0; i < n; i++) {
			  if (vis[i])continue;
			  if (i > 0 && !vis[i - 1] && arr[i] == arr[i - 1])continue;
			  if (curr == 0 || adj[t][i] == 1) {
				  vis[i] = true;
				  solve(curr + 1, i, n);
				  vis[i] = false;
			  }
		  }
	  }
	  int numSquarefulPerms(vector<int>& A) {
		  arr = A;
		  int n = A.size();
		  if (n == 1)return 0;
		  adj.resize(n, vector<int>(n, 0));
		  vis.assign(n, false);
		  for (int i = 0; i < n; i++) {
			  for (int j = i + 1; j < n; j++) {
				  int x = sqrt(A[i] + A[j]);
				  if (x*x == A[i] + A[j])adj[i][j] = 1, adj[j][i] = 1;
			  }
		  }
		  solve(0, -1, n);
		  return ct;
	  }
  };
