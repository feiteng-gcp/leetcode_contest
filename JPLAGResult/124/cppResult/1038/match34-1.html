class Solution {
public:
    
    int count;
    unordered_map<int, unordered_set<int>> squarePair;
    vector<int> builder;
    
    bool checkUsable(int index, int val){
        if(index > 0 && squarePair[builder[index-1]].count(val) == 0){
            return false;
        }
        // 
        return true;
        
    }
    
    void dfs(vector<int>& A, vector<bool>& used, int index){
        if(index == A.size()) count++;
        else{
            // choose a number for this index
            for(int i = 0;i < A.size();i++){
                if(i > 0 && A[i] == A[i-1] && !used[i-1]) continue; // skip duplicate
                if(!used[i] && checkUsable(index, A[i])){
                    // use this
                    used[i] = true;

                    builder[index] = A[i];
                    dfs(A, used, index + 1);
                    used[i] = false;
                }
            }
        }
    }
    
    int numSquarefulPerms(vector<int>& A) {
        sort(A.begin(), A.end());
        count = 0;
        builder.resize(A.size());
        // generate possible pair
        squarePair.clear();
        for(int i = 0;i < A.size() - 1;i++){
            for(int j = i + 1;j < A.size(); j++){
                int val = A[i] + A[j];
                double sqrtRoot = floor(sqrt(val));
                if(sqrtRoot * sqrtRoot == val){
                    // true
                    squarePair[A[i]].insert(A[j]);
                    squarePair[A[j]].insert(A[i]);
                }
            }
        }
        // enumerate
        vector<bool> used(A.size(), false);
        dfs(A, used, 0);
        return count;
    }
};
