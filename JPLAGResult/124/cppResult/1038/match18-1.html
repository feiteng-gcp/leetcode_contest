class Solution {
public:
int n;
bool mp[15][15];
int dp[1<<12][12];

bool judge(int x,int y)
{
	x+=y;
	int rt=(int)sqrt(x);
	return rt*rt==x;
}

int numSquarefulPerms(vector<int>& A) 
{
	n=A.size();
	if(n==1) return 0;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
			mp[i][j]=mp[j][i]=judge(A[i],A[j]);
	int mask=1<<n;
	for(int i=0;i<n;i++)

		dp[1<<i][i]=1;
	for(int i=0;i<mask;i++)
	{
		for(int j=0;j<n;j++)
		{
			if((i>>j)&1);

			else continue;
			for(int k=0;k<n;k++)
			{
				if(!mp[j][k]) continue;
				if((i>>k)&1) continue;
				dp[i|1<<k][k]+=dp[i][j];
			}
		}
	}
	int ans=0;
	for(int i=0;i<n;i++)
		ans+=dp[mask-1][i];
	map<int,int> s;
	int fac[13];
	fac[0]=1;
	for(int i=1;i<13;i++)
		fac[i]=i*fac[i-1];
	for(int i=0;i<n;i++)
		s[A[i]]++;
	for(auto it:s)
	{
		assert(ans%fac[it.second]==0);
		ans/=fac[it.second];
	}
	return ans;
}
};
