bool comp(int a, int b){
    return a>b;
}

class Solution {
public:
    
    int result;
    
    bool square(int x){
        int y = sqrt(x);
        return (y*y)==x;
    }

    void dfs(vector<int>& A, int start){
        if(start == A.size()){
            result++;
            return;
        }
        for(int i=start; i<A.size(); i++){
            if(i>start && (A[i]==A[start] || A[i]==A[i-1]))
                continue;
            if(start > 0 && !square(A[i]+A[start-1]))
                continue;
            swap(A[i], A[start]);
            dfs(A, start+1);
            swap(A[i], A[start]);
        }
    }
    
    int numSquarefulPerms(vector<int>& A) {
        sort(A.begin(), A.end());
        result=0;
        dfs(A, 0);
        return result;
    }
};
