class Solution {
private:
    int count;
public:
    int numSquarefulPerms(vector<int>& A) {
        vector<bool> visited(A.size(),false);
        dfs(A, vector<int>(),visited);
        return count;
    }
    
    void dfs(vector<int>& A,vector<int> v,vector<bool>&visited){
        if(v.size()==A.size()){
            count++;
            return;
        }
        map<int, int> m;
        for(int i = 0;i<A.size();i++){
            if(m.find(A[i])!=m.end()||visited[i]){

                continue;
            }
            if(v.empty()){
                v.push_back(A[i]);
                visited[i] = true;
                dfs(A, v,visited);
                visited[i] = false;
                v.pop_back();
            }else{
                int x = sqrt(A[i]+v.back());
                if(x*x == A[i]+v.back()){
                    v.push_back(A[i]);
                    visited[i]=true;
                    dfs(A, v,visited);
                    visited[i]=false;
                    v.pop_back();
                }
            }
            m[A[i]]++;
        }
    }
};
