class Solution {
public:
    bool isPerfectSquare(double x) {   
      // Find floating point value of  
      // square root of x. 
      double sr = sqrt(x); 
      
      // If square root is an integer 
      return ((sr - floor(sr)) == 0); 
    } 

    int numSquarefulPerms(vector<int>& A){
        int n = A.size();
        vector<vector<int>> graph(n, vector<int>(n, 0));
        sort(A.begin(), A.end());
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(i != j && isPerfectSquare(A[i] + A[j]))graph[i][j] = 1;
            }
        }
        int prev = -1;
        set<int> vis;
        int res = 0;
        for(int i = 0; i < n; i++){
            if(A[i] != prev){
                vis.insert(i);
                dfs(A, graph,vis, i, res);
                vis.erase(i);
                prev = A[i];
            }
        }
        return res;
    }
    void dfs(vector<int>& A, vector<vector<int>>& graph, set<int>& vis, int node, int& res){
        //cout << vis.size() << endl;
        if(vis.size() == A.size()){
            res++;
            return;
        }
        int prev = -1;

        for(int i = 0; i < A.size(); i++){
            if(vis.find(i) == vis.end() && A[i] != prev && graph[node][i]){
                prev = A[i];
                vis.insert(i);
                dfs(A,graph,vis,i, res);
                vis.erase(i);
            }
        }
    }
};
