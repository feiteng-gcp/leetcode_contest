class Solution {
public:
    vector<vector<bool>> table;
    int res = 0;
    vector<int> curr;
    vector<bool> valid;

    void travel(vector<int>& curr, vector<int>& A) {
        if (curr.size() == A.size()) {
            res++;
            return;
        }
        for (int i = 0; i < A.size(); i++) {
            if (valid[i]) {
                curr.push_back(i);
                valid[i] = false;

                if (curr.size() == 1 || table[curr[curr.size() - 1]][curr[curr.size() - 2]]) {

                    travel(curr, A);
                }
                curr.pop_back();
                valid[i] = true;
                while (i+1 < A.size() && A[i + 1] == A[i]) {
                    i++;
                }
            }

        }
    }
    int numSquarefulPerms(vector<int>& A) {
        sort(A.begin(), A.end());
        valid = vector<bool>(A.size(), true);
        table = vector<vector<bool>>(A.size(), vector<bool>(A.size(), false));
        for (int i = 0; i < A.size(); i++) {
            for (int j = i + 1; j < A.size(); j++) {
                long double prod = A[i] + A[j];
                if (sqrt(prod) - floor(sqrt(prod)) == 0) {
                    table[i][j] = table[j][i] = true;
                }
            }
        }
        travel(curr, A);
        return res;
    }
};
