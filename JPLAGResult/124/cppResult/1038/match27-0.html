class Solution {
public:
        struct node {
            int val;
            int repeat;
            vector<node*> adj;
        };

    int goal_length;
    bool issquare(int a) {
        if(a < 0) return false;
        for(int i = 0; ; ++i) {
            if(i * i == a) return true;
            if(i * i > a) return false;
        }
    }
    int dfs(node* n, int length) {
        length++;
        //cout << "search " << n->val << endl;
        if(length == goal_length) {
            //cout << "a valid path, return" << endl;
            return 1;
        }
        n->repeat--;
        int num = 0;
        for(int i = 0; i < n->adj.size(); ++i) {
            //cout << n->adj.size() << " nodes adjacent\n";
            if(n->adj[i]->repeat > 0) {
                num += dfs(n->adj[i], length);
            }
        }
        n->repeat++;
        return num;
    }
    int numSquarefulPerms(vector<int>& A) {
        goal_length = A.size();
        vector<node> graph;
        sort(A.begin(), A.end());
        int prev_val = -1;
        for(int i = 0; i < A.size(); ++i) {
            if(A[i] != prev_val) {
                node n;
                n.val = A[i];
                n.repeat = 1;
                vector<node*> adj;
                n.adj = adj;
                graph.push_back(n);
            }else {
                graph[graph.size()-1].repeat++;
            }
            prev_val = A[i];
        }
        for(int i = 0; i < graph.size(); ++i) {
            for(int j = 0; j < graph.size(); ++j) {
                if(i == j && graph[i].repeat == 1) {
                    continue;
                }
                if(issquare(graph[i].val + graph[j].val)) {
                    //cout << graph[i].val << " " << graph[j].val << endl;
                    graph[i].adj.push_back(&(graph[j]));
                    
                }
            }
        }
        int result = 0;
        for(int i = 0; i < graph.size(); ++i) {
            result += dfs(&(graph[i]), 0);
        }
        return result;
    }
};
