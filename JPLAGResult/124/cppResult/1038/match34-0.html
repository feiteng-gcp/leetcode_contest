class Solution {
private:
    int ans;
    vector<vector<int>>mat;
    vector<bool>vis;
public:
    bool sqrt_ok(int x) {
        int t = sqrt(x);
        return t * t == x;
    }
    void dfs(int i, int n, vector<int>& A, int last) {
        if (i > n) {
            ans++;
            return;
        }
        int j;
        if (i == 1) {
            for (j = 0; j < n; j++) {
                if (j > 0 && A[j] == A[j - 1])
                    continue;
                vis[j] = true;
                dfs(i + 1, n, A, j);
                vis[j] = false;
            }
        }
        else {
            set<int>t;
            for (j = 0; j < mat[last].size(); j++) {
                if (vis[mat[last][j]] == false && t.find(A[mat[last][j]]) == t.end()) {
                    vis[mat[last][j]] = true;
                    t.insert(A[mat[last][j]]);
                    dfs(i + 1, n, A, mat[last][j]);
                    vis[mat[last][j]] = false;
                }
            }
        }
    }
    int numSquarefulPerms(vector<int>& A) {
        sort(A.begin(), A.end());
        ans = 0;
        mat.clear();
        mat.resize(A.size());
        vis.resize(A.size(), false);
        for (int i = 0; i < A.size(); i++) {
            for (int j = i + 1; j < A.size(); j++) {
                if (sqrt_ok(A[i] + A[j])) {
                    mat[i].push_back(j);
                    mat[j].push_back(i);
                }
            }
        }
        int cnt = 0;
        for(int i = 0; i < A.size(); i++) {
            if (mat[i].empty())
                return 0;
            if (mat[i].size() == 1) {
                cnt++;
            }
        }
        if (cnt > 2)
            return 0;
        vector<int>b;
        dfs(1, A.size(), A, -1);
        return ans;
    }
};
