class Solution {
public:
    
    vector<int> v[20], val;
    bool vis[20];
    int cnt = 0;
    void dfs(int x, int c, int t) {
        if(c==t) {
            cnt++;
            return;
        }
        vis[x] = true;
        int tmp = -1;
        for(auto e: v[x]) {
            if(!vis[e] && tmp != val[e]) {
                tmp = val[e];
                dfs(e, c+1, t);
            }
        }
        vis[x] = false;
    }
    int numSquarefulPerms(vector<int>& A) {
        int n = (int)A.size();
        sort(A.begin(), A.end());
        val = A;

        for(int i = 0 ; i < n ; i++) {
            for(int j = i+1 ; j < n ; j++) {
                if(i!=j) {
                    int tmp = A[i] + A[j];
                    int ss = (int)sqrt(tmp);
                    if(ss*ss == tmp) {
                        v[i].push_back(j);
                        v[j].push_back(i);
                    }
                }
            }
        }
        cnt = 0;
        memset(vis, 0, sizeof(vis));

        int tmp = -1;
        for(int i = 0 ; i < n; i++) {
            if(tmp != A[i]) {
                dfs(i, 1, n);
                tmp = A[i];
            }
        }
        return cnt;
    }
};
