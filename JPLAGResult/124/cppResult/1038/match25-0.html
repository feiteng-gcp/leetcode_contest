class Solution {
    int ans;
    bool isOk(int last,int a){
        if(last==-1) return true;
        int c=last+a;
        int d=sqrt(c);
        return d*d==c;
    }
    void dfs(vector<int>& A,int last,int used){
        unordered_set<int> mm;
        if(used==(1<<A.size())-1){
            ans++;
            return;
        }
        for(int i=0;i<A.size();i++){
            if(used&(1<<i)) continue;
            if(mm.find(A[i])!=mm.end()) continue;
            if(!isOk(last,A[i])) continue;
            dfs(A,A[i],used+(1<<i));
            mm.insert(A[i]);
        }
    }
public:
    int numSquarefulPerms(vector<int>& A) {
        ans=0;
        dfs(A,-1,0);
        return ans;
    }
};
