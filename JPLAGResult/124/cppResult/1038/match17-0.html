class Solution {
public:
    int numSquarefulPerms(vector<int>& a) {
        int n = a.size();
        vector<vector<int>> f(1 << n, vector<int>(n, 0));
        
        for (int i = 0; i < n; i++) {
            f[1 << i][i] = 1;
        }
        
        for (int i = 0; i < 1 << n; i++) {
            for (int j = 0; j < n; j++) {
                if (i & (1 << j)) {
                    for (int k = 0; k < n; k++) {
                        if (!(i & (1 << k)) && isSquare(a[j] + a[k])) {
                            f[i | (1 << k)][k] += f[i][j];
                        } 
                    }
                }
            }
        }
        
        map<int, int> cnt;
        for (int x : a) {
            cnt[x]++;
        }
        
        long long div = 1;
        for (auto &it : cnt) {
            div *= fact(it.second);
        }
        long long ret = 0;
        for (int i = 0; i < n; i++) {
            ret += f[(1 << n) - 1][i];
        }
        
        return ret / div;
    }
    
    long long fact(int n) {
        long long ret = 1;
        for (int i = 1; i <= n; i++) {
            ret *= i;
        }
        return ret;
    }
    
    bool isSquare(int n) {
        int s = sqrt(n) + 0.5;
        return s * s == n;
    }
};
