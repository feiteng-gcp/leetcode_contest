class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        sort(A.begin(), A.end());
        int n = A.size(); 
        vector< vector<int> > f(1 << n, vector<int>(n, 0));

        for (int i = 0; i < n; i++) if (i == 0 || A[i] != A[i - 1]) f[1 << i][i] = 1;
        for (int st = 0; st < (1 << n); st++)
            for (int i = 0; i < n; i++) if (f[st][i] > 0)
                for (int j = 0; j < n; j++) if (((st >> j) & 1) == 0 && isSquare(A[i] + A[j])) {
                    if (j > 0 && A[j] == A[j - 1] && ((st >> j - 1) & 1) == 0) continue;
                    if (A[i] == A[j] && j < i) continue;
                    f[st | (1 << j)][j] += f[st][i];                    
                }
        return accumulate(f[(1 << n) - 1].begin(), f[(1 << n) - 1].end(), 0);
    }
    bool isSquare(int x) {
        if (x == 0) return true;
        int r = sqrt(x);
        return r * r == x;
    }
};
