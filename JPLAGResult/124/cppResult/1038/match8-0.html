class Solution {
public:
    int res = 0;
    int n;
    set<vector<int> > mem;
    int goal;
    void go(vector<int> & a, int used, vector<int> & sofar) {
        if (mem.find(sofar) != mem.end()) {
            return;
        }
        mem.insert(sofar);
        if (used == goal) {
            res++;
            return;
        }
        for (int i = 0; i < n; i++) {
            if (!(used & (1<<i))) {
                int q = sofar.back() + a[i];
                int s = (int)(sqrt(q) + 0.5);
                if (s*s == q) {
                    sofar.push_back(a[i]);
                    used ^= (1<<i);
                    go(a, used, sofar);
                    used ^= (1<<i);
                    sofar.pop_back();
                }
            }
        }
    }
    
    int numSquarefulPerms(vector<int>& a) {
        n = a.size();
        goal = (1<<n) - 1;
        for (int i = 0; i < n; i++) {
            vector<int> v(1, a[i]);
            go(a, (1<<i), v);
        }
        return res;
    }
};
