class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        sort(A.begin(),A.end());
        int cnt=0;
        vector<bool> B(A.size(),false);
        
        search(A,B,0,cnt,0);
        return cnt;
    }
    void search(const vector<int>& A,vector<bool> B,int last,int& cnt,int n){
        // printf("last=%d,n=%d",last,n);

        if (n==A.size()){
            cnt++;
            return;
        }
        for (int i=0;i<A.size();++i){
            if(i>0&&A[i]==A[i-1]&&!B[i-1]) continue;
            if (!B[i]&&(isValid(last+A[i])||n==0)){
                B[i]=true;
                search(A,B,A[i],cnt,n+1);
                B[i]=false;
            }
        }
    }
    bool isValid(int p){
        int x=sqrt(1.0*p);
        return (p==x*x);
    }
};
