class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        unordered_map<int, unordered_set<int>> graph;
        unordered_map<int, int> count;
        
        // construct the graph
        for (int i = 0; i < A.size(); ++i) {
            ++count[A[i]];
            for (int j = i + 1; j < A.size(); ++j) {
                if (is_square(A[i] + A[j])) {
                    graph[A[i]].insert(A[j]);
                    graph[A[j]].insert(A[i]);
                }
            }
        }
        
        // dfs
        int res = 0;
        int n = A.size();
        for(auto& p : count) {
            int u = p.first;
            dfs(u, graph, count, 1, n, res);
        }
        
        return res;
    }
private:
    bool is_square(int x) {
        int n = (int)sqrt(x);
        return (n-1)*(n-1) == x || n*n == x || (n+1)*(n+1) == x;
    }
    
    void dfs(int u, unordered_map<int, unordered_set<int>>& graph, unordered_map<int, int>& count, int len, const int& n, int& res) {
        if (len == n && len > 1) {
            ++res;
            return;
        }
        --count[u];
        for (auto& v : graph[u]) {
            if (count[v] > 0) {
                dfs(v, graph, count, len + 1, n, res);
            }
        }
        ++count[u];        
    }
};
