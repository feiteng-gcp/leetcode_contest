class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        unordered_map<int, int> perfect;
        for (int i = 0; i * i <= 1e9; ++i)
            perfect[i*i] = 1;
        int cnt = 0;
        unordered_map<int,int> Amap;
        for (auto a:A)
            Amap[a]++;
        /*unordered_map<int, unordered_set<int>> nbrs;
        for (int i = 0; i < A.size(); ++i)
            for (int j = i+1; j < A.size(); ++j)
                if (perfect.count(A[i]+A[j]) > 0) {
                    nbrs[A[i]].insert(A[j]);
                    nbrs[A[j]].insert(A[i]);
                }*/
        vector<int> curr;
        sort(A.begin(), A.end());
        dfs(A, Amap, perfect, curr, cnt);
        return cnt;
    }
    void dfs(vector<int> & A, unordered_map<int,int>& Amap, unordered_map<int, int>& perfect, vector<int> & curr, int &cnt) {

        if (curr.size() == A.size()) {
            cnt++;
            return;
        }
        for (int i = 0; i < A.size(); ++i) {
            if (i > 0 && A[i] == A[i-1])
                continue;
            if (Amap[A[i]] > 0 && (curr.empty() || perfect.count(A[i] + curr.back()) > 0)) {
                curr.push_back(A[i]);
                Amap[A[i]]--;
                dfs(A, Amap, perfect, curr, cnt);
                Amap[A[i]]++;
                curr.pop_back();
            }
        }
    }
};
