class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        std::set<std::pair<int, int>> mm;
        std::sort(A.begin(), A.end());
        std::map<int, int> c1, c2;
        for (auto && n: A) {
            for (auto && m: A) {
                if (square(n, m)) {
                    mm.emplace(n, m);
                }
            }
            c2[n] += 1;
        }
        int r = 0;
        std::vector<int> B(A.size());
        go(A, B, mm, c1, c2, r, 0, A.size());
        return r;
    }

    void go(std::vector<int> & A, std::vector<int> & B, std::set<std::pair<int, int>> & mm,
            std::map<int, int> & c1, std::map<int, int> & c2,
            int & r, int p, int n) {
        if (p == n) {
            r += 1;
            return;
        }
        for (int i = 0; i < n; ++i) {
            if (p != 0 && mm.find(std::make_pair(B[p-1], A[i])) == mm.end()) {
                continue;
            }
            if (i == 0 || A[i] != A[i-1]) {
                if (c1[A[i]] < c2[A[i]]) {
                    B[p] = A[i];
                    c1[A[i]] += 1;
                    go(A, B, mm, c1, c2, r, p + 1, n);
                    c1[A[i]] -= 1;
                }
            }
        }
    }

    inline bool square(int a, int b) {
        return (int)std::sqrt(a + b) * std::sqrt(a + b) == a + b;
    }
};
