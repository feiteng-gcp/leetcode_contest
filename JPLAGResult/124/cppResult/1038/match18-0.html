#define LL long long
LL dp[15][1<<12];
LL check(LL v){
    LL sv=sqrt(v);
    if (1LL*(sv-1LL)*(sv-1LL)==v) return true;
    if (sv*sv==v) return true;
    if (1LL*(sv+1LL)*(sv+1LL)==v) return true;
    return false;
}

LL fac(int c){
    LL ret=1;
    for (int i=1; i<=c; i++) ret=ret*i;
    return ret;
}

int solve(int n, vector<int>& A){
    memset(dp, 0, sizeof(dp));
    for (int i=0; i<n; i++) dp[i][1<<i]=1;
    
    int lim=1<<n;
    for (int s=0; s<lim; s++){
        for (int i=0; i<n; i++){
            if (dp[i][s]==0) continue;
            for (int j=0; j<n; j++){
                if ((s>>j)&1) continue;
                if (!check(A[i]+A[j])) continue;
                dp[j][s|(1<<j)]+=dp[i][s];
            }
        }
    }
    
    LL ans=0;
    for (int i=0; i<n; i++) ans+=dp[i][lim-1];
    
    sort(A.begin(), A.end());
    
    int rate=0;
    for (int i=0; i<n; i++){
        ++rate;
        if (i==n-1 || A[i]!=A[i+1]){
            ans/=fac(rate);
            rate=0;
        }
    }
    
    return ans;
}
class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        return solve(A.size(), A);
    }
};
