class Solution {
public:
    int numSquarefulPerms(vector<int>& A) {
        //backtracking
        if(A.empty()) return 0;
        int n = A.size();
        int res = 0;
        vector<int> visited(n, 0);
        vector<int> path;
        sort(A.begin(), A.end());
        dfs(A, visited, res, n, path);
        return res;
    }
    
    void dfs(vector<int>& A, vector<int>& visited, int& res, int n, vector<int>& path){
        if(path.size() == n){
            res++;
            return;
        }
        for(int j = 0; j < n;){
            if(visited[j]){j++;continue;}
            if(path.empty()){
                path.push_back(A[j]);
                visited[j] = 1;
                dfs(A, visited, res, n, path);
                path.pop_back();
                visited[j] = 0;
            }else{
                int val = path.back();
                if(isPerfectSqr(val + A[j])){
                    path.push_back(A[j]);
                    visited[j] = 1;
                    dfs(A, visited, res, n, path);
                    path.pop_back();
                    visited[j] = 0;
                }
            }
            j++;
            while(j < n && A[j] == A[j-1]) j++;
        }
        return;
    }
    
    bool isPerfectSqr(int num){
        if(num == 0 || num == 1) return true;
        if(num < 0) return false;
        for(int i = 2; i*i <= num; ++i){
            if(i*i == num) return true;
        }
        return false;
    }
};
