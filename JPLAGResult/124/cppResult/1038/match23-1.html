class Solution {
public:
    set<int>s;
    map<int,int>m;
    int ret,n;
    void dfs(int d,int last){
        if(d==n){
            ret++;
            return;
        }
        for(auto it=m.begin();it!=m.end();it++){

            if(it->second==0)continue;
            if(d==0||s.find(last+it->first)!=s.end()){
                m[it->first]--;
                dfs(d+1,it->first);
                m[it->first]++;
            }
        }
    }
    int numSquarefulPerms(vector<int>& a) {
        n=(int)a.size();
        ret=0;
        for(int i=0;i*i<=1e9;i++)
            s.insert(i*i);
        for(int i=0;i<n;i++)
            m[a[i]]++;
        dfs(0,0);
        return ret;
    }
};
