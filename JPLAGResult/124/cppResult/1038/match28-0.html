typedef long long int64;
const int N = 12;
bool g[N][N];
int dp[1 << N][N];
int F[N + 1];

class Solution {
public:
    bool is_square(int64 n) {
        for (int64 k = 0; k * k <= n; ++k) {
            if (k * k == n) return true;
        }
        return false;
    }
    int n;
    int solve(int mask, int x) {
        if (mask == (1 << n) - 1) return 1;
        int& ret = dp[mask][x];
        if (ret >= 0) return ret;
        ret = 0;
        for (int y = 0; y < n; ++y) {
            if (mask & (1 << y)) continue;
            if (!g[x][y]) continue;
            ret += solve(mask | (1 << y), y);
        }
        return ret;
    }
    int numSquarefulPerms(vector<int>& a) {
        n = a.size();
        F[0] = 1;
        for (int i = 1; i <= n; ++i) F[i] = F[i - 1] * i;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                g[i][j] = is_square(a[i] + a[j]);
            }
        }
        memset(dp, 255, sizeof(dp));
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            ret += solve(1 << i, i);
        }
        map<int, int> cnt;
        for (int i = 0; i < n; ++i) {
            cnt[a[i]]++;
        }
        for (auto& it : cnt) {
            ret /= F[it.second];
        }
        return ret;
    }
};
