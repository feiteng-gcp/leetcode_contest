class Solution {
public:
    int numSquarefulPerms(vector<int>& a) {
        int n = a.size(), ans = 0;
        
        sort(a.begin(), a.end());
        
        vector<vector<bool>> g(n, vector<bool>(n));
        for (int i = 0; i < n; ++i)
            for (int j = i+1; j < n; ++j)
                if (check(a[i] + a[j])) {
                    g[i][j] = g[j][i] = true;
                }
        
        vector<bool> vis(n);
        vector<int> plan(n);
        
        dfs(0, n, a, vis, plan, g, ans);
        
        return ans;
    }
    

    void dfs(int idx, int n, vector<int> &a, vector<bool> &vis, vector<int> &plan, vector<vector<bool>> &g, int &ans) {
        if (idx == n) {
            ++ans;
            return;
        }
        
        for (int i = 0; i < n; ++i)
            if (!vis[i]) {
                if (i > 0 && a[i] == a[i-1] && !vis[i-1])
                    continue;
                
                if (idx > 0 && !g[plan[idx-1]][i])
                    continue;
                
                plan[idx] = i;
                vis[i] = true;
                dfs(idx+1, n, a, vis, plan, g, ans);
                
                vis[i] = false;
            }
    }
    
    bool check(int d) {
        int tmp = sqrt(d);
        return (tmp * tmp == d);
    }
};
