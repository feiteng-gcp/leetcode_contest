class Solution {
public:
    
    vector<int> v, g;
    int res;
    int vis[15];
    int n;
    map<int, int> square;
    
    void init() {
        square.clear();
        for (int i = 0; i <= 44722; i++) {
            square[i*i] = 1;
        }
    }
    
    bool perfect(int n) {
        return square.find(n) != square.end();    
    }
    

    void dfs(int u) {
        if (u == n) {
            res++;
            return;
        }
        
        map<int, int> mp;
        
        for (int i = 0; i < n; i++) {
            if (vis[i] == 0 && mp[v[i]] == 0) {
                mp[v[i]] = 1;
                if (u == 0) {
                    g.push_back(v[i]);
                    vis[i] = 1;
                    dfs(u + 1);
                    vis[i] = 0;
                    g.pop_back();
                } else {
                    if (perfect(v[i] + g.back())) {
                        g.push_back(v[i]);
                        vis[i] = 1;
                        dfs(u + 1);
                        vis[i] = 0;
                        g.pop_back();
                    }
                }
            }
        }
    }
    
    int numSquarefulPerms(vector<int>& A) {
        init();
        v = A;
        n = v.size();
        sort(v.begin(), v.end());
        for (int i = 0; i < 15; i++) vis[i] = 0;
        res = 0;
        g.clear();
        dfs(0);
        return res;
    }
};
