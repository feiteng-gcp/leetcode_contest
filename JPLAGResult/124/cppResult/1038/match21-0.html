class Solution {
    unordered_map<int, int> m;
    unordered_map<int, unordered_set<int>> ms;
public:
    int numSquarefulPerms(vector<int>& A) {
        int n = A.size(), ans = 0;
        if (n < 2) return ans;
        for (int i = 0; i < n; ++i) {
            ++m[A[i]];
            for (int j = i + 1; j < n; ++j) {
                int temp = sqrt(A[i]+A[j]);
                if (temp*temp == A[i]+A[j]) {
                    ms[A[i]].insert(A[j]);
                    ms[A[j]].insert(A[i]);
                }
            }
        } 
        for (auto& t : m) {
            if (t.second > 0) {
                --t.second;
                dfs(ans, n, 0, t.first);
                ++t.second;
            }
        }
        return ans;
    }
    void dfs(int& ans, int n, int ind, int x) {
        // cout << x << endl;
        if (ind == n - 1) {
            ++ans;
            return;
        }
        for (auto y : ms[x]) {
            if (m[y] > 0) {
                --m[y];
                dfs(ans, n, ind+1, y);
                ++m[y];
            }
        }
    }
};
