class Solution {
public:
    int count = 0;
    int numSquarefulPerms(vector<int>& A) {
        map<int, set<int>> links;
        for (int i = 0; i < A.size() - 1; i++) {
            for (int j = i + 1; j < A.size(); j++) {
                if(check(A[i], A[j])) {
                    links[A[i]].insert(A[j]);
                    links[A[j]].insert(A[i]);
                }
            }
        }
        
        set<int> visited;
        
        map<int, int> stats;
        for (auto i : A) {
            stats[i]++;
        }
        
        for (auto p : stats) {
            stats[p.first]--;
            dfs(p.first, 1, A.size(), links, stats);
            stats[p.first]++;
        }
        
        // for (auto p : stats) {
        //     count /= f(p.second);
        // }
        

        return count;
    }
    
    void dfs(int start, int depth, int total, map<int, set<int>>& links, map<int, int>& stats) {
        
        if (depth == total) {
            count++;
            return;
        }
        // cout << start << "@" << depth << endl;
        for (auto next : links[start]) {
            if (stats[next] > 0) {
                stats[next]--;
                dfs(next, depth + 1, total, links, stats);
                stats[next]++;
            }
        }
    }
    
    bool check(int a, int b) {
        int r = sqrt(a + b);
        return r * r == a + b;
    }
};
