class Solution {
    public boolean isRobotBounded(String instructions) {
        
int[][] dirs = new int[][] {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        
int dir = 0, x = 0, y = 0;
        for (int cnt = 0; cnt < 8; cnt++) {
            for (char c : instructions.toCharArray()) {
                if (c == 'G') {
                    x += dirs[dir][0];
                    y += dirs[dir][1];
                } else if (c == 'L') {
                    if (dir == 0) { dir = 3; }
                    else { dir--; }
                } else {
                    if (dir == 3) { dir = 0; }
                    else { dir++; }
                
}
            }
            if (x == 0 && y == 0 && dir == 0) {
                return true;
            }
        }
        return false;
    }
}
