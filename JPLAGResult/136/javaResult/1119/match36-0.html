class Solution {
    
    int[][] dirs = {
        {-1, 0}, // left
        {0, 1}, // down
        {1, 0}, // right,
        {0, -1} // up
    };
    public boolean isRobotBounded(String instructions) {
        Set<String> set = new HashSet<>();
        int x = 0, y = 0;
        int i = 3;
        for (char c : instructions.toCharArray()) {
            if (c == 'G') {
                x += dirs[i][0];
                y += dirs[i][1];
            } else if (c == 'L') {
                i = next(i, 1);
            } else {
                i = next(i, -1);
            }
            set.add(x + " " + y);
        }
        for (int z = 0; z < 5; z++) {
            for (char c : instructions.toCharArray()) {
                if (c == 'G') {
                    x += dirs[i][0];
                    y += dirs[i][1];
                } else if (c == 'L') {
                    i = next(i, 1);
                } else {
                    i = next(i, -1);
                }
            }
            if (x == 0 && y == 0) return true;
        }
        return false;
    }
    
    int next(int i, int x) {
        i = i + x;
        if (i < 0) return 3;
        return i % 4;
    }
}
