class Solution {
    public boolean isRobotBounded(String instructions) {
	    int [] d = new int [4];
	    int dir
 = 0;
	    for(int j = 0; j < 4; j++) {
	    	for(int i = 0; i < instructions.length(); i++) {
		    	switch(instructions.charAt(i)) {
		    		case 'G':{

		    			d[dir]++;
		    			break;
		    		}
		    		case 'L':{
		    			dir = (dir + 3) % 4;
		    			break;
		    		}
		    		case 'R':{
		    			dir = (dir + 1) % 4;
		    		}

		    	}
		    }
	    }
	    return (d[0] == d[2] && d[1] == d[3]);
	}
}
