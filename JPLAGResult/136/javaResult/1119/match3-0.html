class Solution {
    public boolean isRobotBounded(String instructions) {
        int x = 0;
        int y = 0;
        int[][] dirs = new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int dirIdx = 0;
        for (int n = 0; n < 4; ++n) {
            for (char c: instructions.toCharArray()) {
                if (c == 'G') {
                    x += dirs[dirIdx][0];
                    y += dirs[dirIdx][1];
                } else if (c == 'L') {
                    dirIdx = (dirIdx + 3) % 4;
                } else {
                    dirIdx = (dirIdx + 1) % 4;
                }
            }
        }
        return x == 0 && y == 0 && dirIdx == 0;
    }
}
