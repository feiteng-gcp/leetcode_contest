class Solution {
    private int[][] delta = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    public boolean isRobotBounded(String instructions) {
        // instructions = "GL";
        int x = 0, y = 0, d = 0;
        
for (int i = 0; i < 4; i++) {
            for (char c : instructions.toCharArray()) {
                if (c == 'G') {
                    x += delta[d][0];
                    y += delta[d][1];
                }
                else if (c == 'R') {
                    d = (d + 1) % 4;
                }
                else if (c == 'L') {
                    d = (d + 3) % 4;
                
}
            }
            if (x == 0 && y == 0 && d == 0) return true;
        }
        return false;
    }
}
