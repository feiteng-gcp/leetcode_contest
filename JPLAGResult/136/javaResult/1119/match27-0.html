class Solution {
    public boolean isRobotBounded(String instructions) {
        int countL = 0;
        int countR = 0;
        int countG = 0;
        int x = 0;
        int y = 0;
        int i = 0;
        int j = 1;
        for (int k = 0; k < instructions.length(); k++) {
            if (instructions.charAt(k) == 'L') {
                if (i == 0 && j == 1) {
                    i = -1;
                    j = 0;
                } else if (i == -1 && j == 0) {
                    i = 0;
                    j = -1;
                } else if (i == 0 && j == -1) {
                    i = 1;
                    j = 0;
                } else {
                    i = 0;
                    j = 1;
                }
                countL++;
            } else if (instructions.charAt(k) == 'R') {
                if (i == 0 && j == 1) {
                    i = 1;
                    j = 0;
                } else if (i == 1 && j == 0) {
                    i = 0;
                    j = -1;
                } else if (i == 0 && j == -1) {
                    i = -1;
                    j = 0;
                } else {
                    i = 0;
                    j = 1;
                }
                countR++;
            } else {
                x += i;
                y += j;
                countG++;
            }
        }
        
        if (x == 0 && y == 0) {
            return true;
        }
        
        int diff = Math.abs(countL - countR);
        return diff % 4 != 0;
    }
}
