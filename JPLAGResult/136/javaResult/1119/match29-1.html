class Solution {
    
    public boolean isRobotBounded(String instructions) {
        
int[][] dirs = new int[][] {{0,1},{-1,0},{0,-1},{1,0}};
        int dir = 0;
        int[] pos = new int[] {0,0};
        for (int i=0; i<4; i++) {
            
for (char c : instructions.toCharArray()){
                if (c == 'G') {
                    pos[0] += dirs[dir][0];
                    pos[1] += dirs[dir][1];
                }
                if (c == 'L') {
                    dir = (dir +1) % 4;
                }
                if (c == 'R') {
                    dir = (dir +3) % 4;
                }                
            }
        }
        if (pos[0] == 0 && pos[1] == 0) return true;
        return false;
    }
}
