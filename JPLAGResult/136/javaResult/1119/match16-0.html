class Solution {
    public boolean isRobotBounded(String instructions) {
        // boolean flag = true;
        // for (char c : instructions.toCharArray())
        //     if (c != 'G')
        //         flag = false;
        // if (flag)
        //     return false;
        int dir = 0;
        int x = 0, y = 0;
        int[][] dirs = new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        while (true) {
            for (char c : instructions.toCharArray()) {
                if (c == 'G') {
                    x += dirs[dir][0];
                    y += dirs[dir][1];
                } else if (c == 'L') {
                    dir = (dir + 3) % 4;
                } else {
                    dir = (dir + 1) % 4;
                }
            }
            if (dir == 0)
                break;
        }
        return x == 0 && y == 0;
    }
}
