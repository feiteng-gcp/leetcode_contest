class Solution {
    
private final static int[][] DIR = new int[][] {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    
    public boolean isRobotBounded(String instructions) {
        int x = 0;
        int y = 0;
        
int dir = 0;
        
        for(int i = 0; i < 4; i++) {
            for(char c : instructions.toCharArray()) {
                if(c == 'L') {
                    dir += 3;
                } else if(c == 'R') {
                    dir++;
                } else {
                    x += DIR[dir % 4][0];
                    y += DIR[dir % 4][1];
                
}
            }
        }
        
        return x == 0 && y == 0;
    }
}
