class Solution {
    public boolean isRobotBounded(String instructions) {
        int i = 0;
        int j = 0;
        int[][] dir = new int[][]{{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
        int cur = 0;
        for (int k = 0; k < 4; k++) {
            for (char a : instructions.toCharArray()) {
                if (a == 'G') {
                    i += dir[cur][0];
                    j += dir[cur][1];
                } else if (a == 'L') {
                    cur = (cur + 4 + 1) % 4;
                } else {
                    cur = (cur + 4 - 1) % 4;
                }
            } 
        }

        return i == 0 && j == 0;
    }
}
