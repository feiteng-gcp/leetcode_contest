class Solution {
    public boolean isRobotBounded(String instructions) {
        
int x = 0, y = 0, d = 0;
        int[][] dir = {{0, 1}, {-1, 0}, {0, -1}, {1, 0}};
        
for(int k = 0; k < 4; k ++){
            for(char c : instructions.toCharArray())
            {
                if(c == 'G')
                {
                    x += dir[d][0];
                    y += dir[d][1];
                }
                else if(c == 'L')
                    d = (d + 3) % 4;
                else
                    d = (d + 1) % 4
;
            }
        }
        return (x == 0 && y == 0);
    }
}
