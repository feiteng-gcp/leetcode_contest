class Solution {
    
    int[] dx = new int[]{0, -1, 0, 1};
    int[] dy = new int[]{1, 0, -1, 0};
    
    public boolean isRobotBounded(String instructions) {
        
        int dir = 0;
        int x = 0, y = 0;
        instructions = instructions + instructions + instructions + instructions;
        for (char ch : instructions.toCharArray()) {
            
            if (ch == 'G') {
                x += dx[dir];
                y += dy[dir];
            } else if (ch == 'L') {
                dir = (dir + 1) % 4;
            } else {
                dir = (dir +3 ) % 4;
            }

        }
        
        return x == 0 && y ==0;
    }
}
