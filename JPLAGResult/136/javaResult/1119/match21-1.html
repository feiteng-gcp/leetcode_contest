class Solution {
    public boolean isRobotBounded(String instructions) {
        int[] step = new int[4];
        
int index = 0;
        for(int i = 0; i < 4; i ++) {
            for(char c : instructions.toCharArray()) {
                if(c == 'G') {
                    step[index] ++;
                } else if(c == 'L') {
                    index = (index + 1) % 4;
                } else {
                    index = (index + 3) % 4;
                }
            }
            if(step[0] == step[2] && step[1] == step[3]) {
                return true;
            }
        }
        return false;
    }
}
