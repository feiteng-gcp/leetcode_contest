class Solution {
    int[] dr = { -1, 0, 1, 0 };
    int[] dc = 
{ 0, 1, 0, -1 };
    public boolean isRobotBounded(String instructions) {
        int r = 0;
        int c = 0;
        int d = 0; // n, e, s, w
        
for (char ins: instructions.toCharArray()) {
            if(ins == 'G') {
                r += dr[d];
                c += dc[d];
            } else if(ins == 'L') {
                d = (d + 3) % 4;
            } else {
                d = (d + 1) % 4;
            }
        }
        if(d == 0) {
            return r == 0 && c == 0;
        }
        return true;
    }
}
