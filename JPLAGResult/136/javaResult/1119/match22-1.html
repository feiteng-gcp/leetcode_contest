class Solution {
    public boolean isRobotBounded(String instructions) {
        String ins = instructions + instructions + instructions + instructions;
        int x = 0, y = 0;
        
int[] dirs = {0, 1, 0, -1, 0};
        int d = 0;
        for (char i : ins.toCharArray()) {
            if (i == 'G') {
                x += dirs[d];
                y += dirs[d + 1];
            } else if (i == 'L') {
                d = (d + 3) % 4;
            } else {
                d = (d + 1) % 4;
            }
        }
        return x == 0 && y == 0;
    }
}
