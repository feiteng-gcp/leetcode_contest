class Solution {
    int[][] di={{1,0},{0,1},{-1,0},{0,-1}};
    public boolean isRobotBounded(String instructions) {
        int d=0,x=0,y=0;
        for(int i=0;i<100;i++){
            for(char c:instructions.toCharArray()){
                if(c=='G'){
                    y+=di[d][0];
                    x+=di[d][1];
                } else if(c=='L'){
                    d=(d+3)%4;
                }else{
                    d=(d+1)%4;
                }
            }
            if(x==0&&y==0&&d==0){
                return true;
            }
        }
        return false;
    }
}
