class Solution {

	public boolean isRobotBounded(String instructions) {
		int x = 0, y = 0;
		char direction = 'N';
		for (int i = 0; i < 1000; i++) {
			for (char c : instructions.toCharArray()) {
				if (c == 'G') {
					if (direction == 'N') {
						y++;
					} else if (direction == 'E') {
						x++;
					} else if (direction == 'W') {
						x--;
					} else if (direction == 'S') {
						y--;
					}
				} else if (c == 'L') {
					if (direction == 'N') {
						direction = 'W';
					} else if (direction == 'E') {
						direction = 'N';
					} else if (direction == 'W') {
						direction = 'S';
					} else if (direction == 'S') {
						direction = 'E';
					}
				} else if (c == 'R') {
					if (direction == 'N') {
						direction = 'E';
					} else if (direction == 'E') {
						direction = 'S';
					} else if (direction == 'W') {
						direction = 'N';
					} else if (direction == 'S') {
						direction = 'W';
					}
				}
			}
			if (x == 0 && y == 0 && direction == 'N') {
				return true;
			}
		}
		return false;
	}
}
