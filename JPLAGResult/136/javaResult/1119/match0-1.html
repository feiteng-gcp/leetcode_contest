class Solution {
    public boolean isRobotBounded(String s) {
        String ns = "";
        
        for(int u=0;u<500;++u){
            ns  +=  s;
        }
        
        
        int x = 0;
        int y = 0;
        int f = 0;
        
        int dir[][] = {{0,1},{-1,0},{0,-1},{1,0}};
        
        
        
        
        for(char a:ns.toCharArray()){
            if(a=='G') {
                x += dir[f][0];
                y += dir[f][1];
            }
            else if(a=='L') {
                f = f+1;
                f %= 4;
            }else{
                f = 3+f;
                f %= 4;
            }
            if(x>110||y>110||x<-110||y<-110){
                return false;
            }
        }
        
        return true;
        
    }
}
