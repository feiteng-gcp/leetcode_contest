class Solution {
    public boolean isRobotBounded(String instructions) {
        int x = 0; 
        int y = 0;
        int[] dx = {0,1, 0, -1};
        int[] dy = {1, 0, -1, 0};
        int dir = 0;
        for(char c : instructions.toCharArray()){
            if(c == 'G'){
                x += dx[dir];
                y += dy[dir];
                //if(x==0 && y == 0) return true;
            }else if(c == 'L'){
                dir = (dir +1)%4;
            }else{
                dir = (dir +3)%4;
            }
        }
        return dir != 0 || (x ==0 && y==0) ;
    }
}
