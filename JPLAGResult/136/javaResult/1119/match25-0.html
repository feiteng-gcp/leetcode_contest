class Solution {
    private int[][] dirs = new int[][]{{0, 1}, {-1, 0}, {0, -1}, {1, 0}};
    public boolean isRobotBounded(String instructions) {
        int dir = 0, x = 0, y = 0;
        for(int i = 0; i < 4; i++) {
            for(char c : instructions.toCharArray()) {
                if(c == 'G') {
                    dir = (dir % 4 + 4) % 4;
                    x += dirs[dir][0];
                    y += dirs[dir][1];
                } else if (c == 'L') {
                    dir++;
                } else {
                    dir--;
                }
            }
        }
        return x == 0 && y == 0;
    }
}
