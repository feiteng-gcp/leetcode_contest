class Solution {
    int[][] ds = new int[][]{{0, 1}, {-1, 0}, {0, -1}, {1, 0}};

    int[] add(int[] v1, int[] v2) {
        return new int[]{v1[0] + v2[0], v1[1] + v2[1]};
    }


    public boolean isRobotBounded(String instructions) {
        int N = instructions.length();
        int d = 0;
        int[] p = {0, 0};
        for (int i = 0; i < 40; i++) {
            for (int j = 0; j < N; j++) {
                char c = instructions.charAt(j);
                if (c == 'G') {
                    p = add(p, ds[d]);
                
} else if (c == 'L') {
                    d = (d + 1) % 4;
                } else {
                    d = (d + 3) % 4;
                }
            }
            if (p[0] == 0 && p[1] == 0) {
                return true;
            }
        }
        return false;
    }
}
