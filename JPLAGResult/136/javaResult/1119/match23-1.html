class Solution {
    public boolean isRobotBounded(String instructions) {
        String sx = instructions + instructions;
        sx += sx;
        int x = 0;
        
int y = 0;
        int d = 0;
        int[] dx = {1, 0, -1, 0};
        int[] dy = {0, 1, 0, -1};
        
for(char c : sx.toCharArray()) {
            if(c == 'G') {
                x += dx[d];
                y += dy[d];
            } else if(c == 'L') {
                d += 3;
                d %= 4;
            } else {
                d += 1;
                d %= 4;
            
}
        }
        return x == 0 && y == 0 && d == 0;
    }
}
