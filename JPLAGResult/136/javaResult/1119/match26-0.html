class Solution {
    private static final int[] dx = {0, 1, 0, -1};
    private static final int[] dy = {1, 0, -1, 0};
    
    public boolean isRobotBounded(String instructions) {
        int dir = 0;
        int x = 0;
        int y = 0;
        for (int i = 0; i < 4; ++i) {
            for (char c: instructions.toCharArray()) {
                switch (c) {
                    case 'G': {
                        x += dx[dir];
                        y += dy[dir];
                        break;
                    }
                    case 'L': {
                        dir = (dir + 3) % 4;
                        break;
                    }
                    case 'R': {
                        dir = (dir + 1) % 4;
                        break;
                    }
                }
            }
        }
        
        return (x == 0 && y == 0);
    }
}
