class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {


		int[] f = new int[A.length + 1];


		for (int i = 1; i < f.length; i++) {
			int tmp = Integer.MIN_VALUE;
			int max = Integer.MIN_VALUE;

			for (int j = 1; j <= K; j++) {

				if (i - j < 0) {
					break;
				}

				if (j == 1) {
					max = A[i - 1];
				} else {
					max = Math.max(max, A[i - j]);
				}

				if (f[i - j] + j * max > tmp) {
					tmp = f[i - j] + j * max;
				}
			}

			f[i] = tmp;
		}
		return f[f.length - 1];
	}
}
