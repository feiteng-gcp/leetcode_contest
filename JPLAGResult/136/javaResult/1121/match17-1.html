class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int[] result = new int[A.length];
        for (int i = 0; i < A.length; i++) {
            int maxValue = 0, maxSum = 0;
            for (int j = i; j >= 0 && i - j + 1 <= K; j--) {
                if (maxValue < A[j]) {
                    maxValue = A[j];
                }
                int tempSum = (j == 0 ? 0 : result[j - 1]) + (i - j + 1) * maxValue;
                
if (maxSum < tempSum) {
                    maxSum = tempSum;
                }
            }
            result[i] = maxSum;
        }
        return result[result.length - 1];
    }
}
