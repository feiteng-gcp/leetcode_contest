class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int[] dp = new int[A.length + 1];
        for (int i = 1; i <= A.length; i++){
            int maxV = 0;
            for (int j = 0; j < K && i - j >= 1; j++){
                maxV = Math.max(A[i - j - 1], maxV);
                dp[i] = Math.max(dp[i - j - 1] + (j + 1) * maxV, dp[i]);
            }
        }
        return dp[A.length];
    }
}
