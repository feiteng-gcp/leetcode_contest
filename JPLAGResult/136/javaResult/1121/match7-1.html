class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int len = A.length;
        int[] dp = new int[len];
        for (int i = 0; i < len; ++i) {
        	int max = 0;
        	for (int j = 0; j < K; ++j) {
        		if (i - j < 0) {
        			break;
        		}
        		max = Math.max
(A[i - j], max);
        		dp[i] = Math.max(dp[i], (i - j - 1 >= 0 ? dp[i - j - 1] : 0) + max * (j + 1));
        	}
        }
        return dp[len - 1];
    }
}
