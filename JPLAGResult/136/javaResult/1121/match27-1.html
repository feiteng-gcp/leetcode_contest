class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int n = A.length;
        int[] dp = new int[n];
        for(int i = 0; i < n; i++) {
            int max = 0;
            for(int j = i; j >= Math.max(0, i - K + 1); j--) {
                max = Math.max(max, A[j]);
                int len = i - j + 1;
                
dp[i] = Math.max(dp[i], len * max + (j == 0 ? 0 : dp[j - 1]));
            }
        }
        return dp[n - 1];
    }
}
