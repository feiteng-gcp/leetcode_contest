class Solution {
    Map<Integer, Integer> cache;
    
    public int maxSumAfterPartitioning(int[] a, int k) {
        cache = new HashMap<>();
        
        return dfs(a, k, 0);
    }
    
    int dfs(int[] a, int k, int idx) {
        if (idx == a.length) {return 0;}
        if (cache.containsKey(idx)) {return cache.get(idx);}
        int max = -1;
        int tmp = a[idx];
        for (int i = idx; i < idx + k && i < a.length; i++) {
            tmp = Math.max(tmp, a[i]);
            max = Math.max(tmp*(i-idx+1) + dfs(a, k, i+1), max);
        }
        
        cache.put(idx, max);
        return max;
    }

}
