class Solution {
    Map<Integer, Integer> cache = new HashMap<>();
    public int maxSumAfterPartitioning(int[] A, int K) {
        
return maxSumAfterPartitioning(A, K, 0);
    }
    
    public int maxSumAfterPartitioning(int[] A, int K, int index) {
        if(index >= A.length) {
            return 0;
        }
        if(cache.containsKey(index)) {
            return cache.get(index);
        }
        int maxNumber = 0;
        int maxRes = 0;
        for(int i = 0; i < K; i ++) {
            if(index + i >= A.length) {
                break;
            }
            
maxNumber = Math.max(maxNumber, A[index + i]);
            maxRes = Math.max(maxRes, maxNumber * (i + 1) + maxSumAfterPartitioning(A, K, index + i + 1));
        }
        cache.put(index, maxRes);
        return maxRes;
    }
}
