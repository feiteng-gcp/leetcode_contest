class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int n = A.length;
        int[] dp = new int[n + 1];
        
        for (int i = 1; i <= n; ++i) {
            int p = Math.max(0, i - K);
            int max = 0;
            for (int j = i; j > p; --j) {
                max = Math.max(max, A[j - 1]);
                dp[i] = Math.max(dp[i], dp[j - 1] + max * (i - j + 1));
            }
        }
        
        return dp[n];
    }
}
