class Solution {
    Map<Integer, Integer> m = new HashMap();
    public int maxSumAfterPartitioning(int[] A, int K) {
        return split(A, 0, K);
    }
    
    private int split(int[] A, int index, int K){
        if (index >= A.length) return 0;
        if (m.containsKey(index)) return m.get(index);
        int sum = 0;
        int currentMax = A[index];
        for(int i = 0; i < K && index + i < A.length; i++){
            currentMax = Math.max(currentMax, A[index + i]);
            sum = Math.max(currentMax * (i + 1) + split(A, index + i + 1, K), sum);
        }
        m.put(index, sum);
        return sum;
    }
}
