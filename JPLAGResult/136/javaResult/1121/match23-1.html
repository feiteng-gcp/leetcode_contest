class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int[] dp = new int[A.length];
        for (int i=0; i<A.length; i++) {
            int max_sum = 0;
            int max_value = A[i];
            for (int j=i; j>=0 && j>= i+1-K; j--) {
                if (A[j] > max_value) {
                    max_value = A[j];
                }
                int local_max_sum = 0;
                if (j > 0) {
                    local_max_sum = dp[j-1];
                }
                local_max_sum += max_value*(i-j+1);
                if (local_max_sum>max_sum) {
                    max_sum = local_max_sum;
                }
            
}
            dp[i] = max_sum;
        }
        return dp[A.length-1];
    }
}
