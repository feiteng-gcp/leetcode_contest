class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        
        
        int l = A.length;
        int dp[] = new int[l];
        
        for(int i = 0;i<l;++i){
            
            dp[i] = (i>=1?dp[i-1]:0)+A[i];
            
            int ma = A[i];
            
            for(int j=i-1;i-j+1<=K&&j>=0;--j){
                
                ma = Math.max(ma,A[j]);
                
                dp[i] = Math.max(dp[i],(j>=1?dp[j-1]:0)+ (i-j+1)*ma); 
                
            }
            
            
        }
            return dp[l-1];
        
        
    }
}
