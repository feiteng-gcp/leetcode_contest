class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int len = A.length;
        int[] dp = new int[len + 1];
        for (int i = 0; i < len; i++) {
            int tmp = 0;
            int max = 0;
            for (int w = 0; w < K; w++) {
                if (i - w < 0) break;
                max = Math.max(max, A[i - w]);
                tmp = Math.max(tmp, dp[i - w] + max * (w + 1));
            }
            dp[i + 1] = tmp;
        }
        return dp[len];
    }
}
