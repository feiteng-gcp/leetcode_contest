class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int N = A.length;
        int[] dp = new int[N + 1];
        dp[0] = 0;
        
        
for (int i = 1; i <= N; i++) {
            int curMax = A[i - 1];
            for (int j = 1; j <= K && i - j >= 0; j++) {
                //System.out.println(i - j + 1);
                curMax = Math.max(A[i - j ], curMax);
                dp[i] = Math.max(dp[i], curMax * j + dp[i - j]);
            }
        }
        return dp[N];
    }
}
