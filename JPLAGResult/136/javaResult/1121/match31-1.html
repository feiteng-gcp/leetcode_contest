class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        int N = A.length;
        int[][] max = new int[N][N + 1];
        for (int i = 0; i < N; i++) {
            for (int j = i + 1; j <= N; j++) {
                max[i][j] = Math.max(max[i][j - 1], A[j - 1]);
            }
        }

        int[] maxSum = new int[N + 1];
        for (int i = 1; i <= N; i++) {
            for (int l = 1; l <= K && l <= i; l++) {
                
maxSum[i] = Math.max(maxSum[i], maxSum[i - l] + l * max[i - l][i]);
            }
        }
        return maxSum[N];
    }
}
