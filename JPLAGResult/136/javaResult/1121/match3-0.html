class Solution {

	public int maxSumAfterPartitioning(int[] A, int K) {
		int dp[] = new int[A.length + 1];
		for (int i = 0; i < A.length; i++) {
			int max = A[i];
			for (int j = 0; j < K && i - j >= 0; j++) {
				max = Math.max(max, A[i - j]);
				dp[i + 1] = Math.max(dp[i + 1], dp[i - j] + max * (j + 1));

			}
		}
//		Helper.tr(dp);
		return dp[A.length];
	}

}
