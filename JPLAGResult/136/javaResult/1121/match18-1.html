class Solution {
    public int maxSumAfterPartitioning(int[] A, int K) {
        // A = new int[]{1,4,1,5,7,3,6,1,9,9,3}; K = 4;
        // int[][] max = new int[A.length][A.length];
        // for (int i = 0; i < A.length; i++) {
        //     max[i][i] = A[i];
        //     for (int j = i - 1; j > -1; j--) {
        //         max[j][i] = Math.max(max[j + 1][i], A[j]);
        //     }
        // }
        int[] dp = new int[A.length + 1];
        for (int i = 0; i < A.length; i++) {
            int temp = 0, max = 0;
            
            for (int j = i; j > -1 && i - j < K; j--) {
                max = Math.max(max, A[j]);
                temp = Math.max(temp, dp[j] + max * (i - j + 1));
                // if (i == A.length - 1) System.out.printf("%d %d %d, \n", j, temp, max);
                
            
}
            dp[i + 1] = temp;
            // System.out.printf("%d, %d\n", i, dp[i + 1]);
        }
        return dp[A.length];
    }
}
