class Solution {
    // max[i][j] = max[i][j-1]
    
    public int maxSumAfterPartitioning(int[] A, int K) {
        int n = A.length;
        int[][] max = new int[n+1][n+1];
        
        for (int i = 1; i <= n; i++) {
            for (int j = i; j <= n; j++) {
                max[i][j] = Math.max(A[j-1], max[i][j-1]);    
            }
        }
        
        int[] dp = new int[n+1];
        for (int i = 1; i <= n; i++) {
            for (int len = 1; len <= K; len++) {
                int j = i - len + 1;
                if (j <= 0) break;
                
                dp[i] = Math.max(dp[i], dp[j-1] + max[j][i] * len);
            }
        }
        
        return dp[n];
    }
}
