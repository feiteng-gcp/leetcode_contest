class Solution {
    Map<Integer, Integer> memo = new HashMap<>();
    
    public int maxSumAfterPartitioning(int[] A, int K) {
        return search(A, 0, K);
    }
    
    private int search(int[] A, int start, int K){
        if(start >= A.length) return 0;
        if(memo.containsKey(start)) return memo.get(start);
        
        int res = Integer.MIN_VALUE;
        for(int i = 1; i <= K; i++){
            if(start + i - 1 >= A.length) continue;
            int curr = i * 
max(A, start, start + i - 1);
            res = Math.max(res, curr + search(A, start + i, K));
        }
        memo.put(start, res);
        
        return res;
    }
    
    private int max(int[] A, int start, int end){
        int res = Integer.MIN_VALUE;
        for(int i = start; i <= end; i++) res = Math.max(res, A[i]);
        return res;
    }
}
