class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] res = new int[N];
        Arrays.fill(res, 5);
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int[] path : paths) {
            Set<Integer> nei1 = map.getOrDefault(path[0], new HashSet<Integer>());
            Set<Integer> nei2 = map.getOrDefault(path[1], new HashSet<Integer>());
            
nei1.add(path[1]);
            nei2.add(path[0]);
            map.put(path[0], nei1);
            map.put(path[1], nei2);
        }

        for (int i = 0; i < N; i++) {
            if (res[i] != 5) {
                continue;
            }
            Set<Integer> neis = map.get(i + 1);
            if (neis == null) {
                res[i] = 1;
            } else {
                boolean[] avail = new boolean[4];
                for (Integer nei : neis) {
                    if (res[nei - 1] != 5) {
                        avail[res[nei - 1] - 1] = true;
                    }
                
}
                for (int j = 0; j < 4; j++) {
                    if (!avail[j]) {
                        res[i] = j + 1;
                        break;
                    }
                }
            }
        }
        return res;
    }
}
