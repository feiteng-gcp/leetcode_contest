class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] res = new int[N];
        HashMap<Integer, HashSet<Integer>> map = new HashMap<>();
        for(int[] path : paths) {
            int x = path[0];
            int y = path[1];
            HashSet<Integer> set1 = map.getOrDefault(x, new HashSet<>());
            HashSet<Integer> set2 = map.getOrDefault(y, new HashSet<>());
            set1.add(y);
            set2.add(x);
            map.put(x, set1);
            map.put(y, set2);
        }
        for(int i = 0; i < N; i++) {
            HashSet<Integer> set = map.get(i+1);
            res[i] = 1;
            if(set == null) continue;
            else {
                int[] candidates = new int[] {1,2,3,4};
                for(int k : set) {
                    if(k-1 < i) candidates[res[k-1]-1] = 0;
                }
                for(int candidate : candidates) {
                    if(candidate != 0) {
                        res[i] = candidate;
                        break;
                    }
                }
            }
        }
        return res;
        
        
    }
