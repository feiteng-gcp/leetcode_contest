class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] res = new int[N];
        int[][] con = new int[N + 1][3];
        for (int[] p : paths) {
            for (int i = 0; i < 3; i++) {
                if (con[p[0]][i] == 0) {
                    con[p[0]][i] = p[1];
                    break;
                }
            }
            for (int i = 0; i < 3; i++) {
                if (con[p[1]][i] == 0) {
                    con[p[1]][i] = p[0];
                    break;
                }
            }
        }
        for (int i = 0; i < N; i++) {
            for (int c = 1; c <= 4; c++) {
                boolean f = true;
                for (int j = 0; j < 3; j++) {
                    int p = con[i + 1][j] - 1;
                    if (p >= 0 && p < i && res[p] == c) {
                        f = false;
                        break;
                    }
                }
                if (f) {
                    res[i] = c;
                    break;
                }
            }
        }
        return res;
    }
}
