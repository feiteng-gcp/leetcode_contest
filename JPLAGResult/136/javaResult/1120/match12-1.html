class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[][] neigh = new int[N][4];
        int[] nn = new int[N];
        int[] ans = new int[N];
        int x = 0, y = 0;
        for(int i = 0 ; i < paths.length; i ++)
        {
            x = paths[i][0] - 1;
            y = paths[i][1] - 1;
            neigh[x][nn[x]++] = y;
            neigh[y][nn[y]++] = x;
        }
        for(int i = 0 ; i < N; i ++)
        {
            
for(int c = 1; c <= 4; c ++)
           {
                boolean dup = false;
                for(int j = 0 ; j < nn[i]; j ++)
                {
                    if(c == ans[neigh[i][j]])
                    {
                        dup = true;
                        break;
                    }
                }
                if(dup) continue;
                else
                
{
                    ans[i] = c;
                    break;
                }
            }
        }
        return ans;
    }
}
