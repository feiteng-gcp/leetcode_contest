class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        Map<Integer, Set<Integer>> edges = new HashMap<>();
        for (int[] e : paths) {
            edges.computeIfAbsent(e[0] - 1, unused -> new HashSet<>()).add(e[1] - 1);
            edges.computeIfAbsent(e[1] - 1, unused -> new HashSet<>()).add(e[0] - 1);
        }
        int[] output = new int[N];
        for (int i = 0; i < N; i++) {
            Set<Integer> neighbors = edges.getOrDefault(i, Collections.emptySet());
            for (int color = 1; color <= 4; color++) {
                boolean valid = true;
                for (int neighbor : neighbors) 
{
                    if (output[neighbor] == color) {
                        valid = false;
                        break;
                    }
                }
                
if (valid) {
                    output[i] = color;
                    break;
                }
            }
        }
        return output;
    }
}
