class Solution {
    private static final List<Integer> EMPTY = new ArrayList<>();
    
    public int[] gardenNoAdj(int N, int[][] paths) {
        Map<Integer, List<Integer>> adjMap = new HashMap<>();
        
        for(int[] path : paths) {
            adjMap.computeIfAbsent(path[0] - 1, unused -> new ArrayList<>()).add(path[1] - 1);
            adjMap.computeIfAbsent(path[1] - 1, unused -> new ArrayList<>()).add(path[0] - 1);
        }
        
        int[] gardens = new int[N];
        
        for(int i = 0; i < N; i++) {
            if(gardens[i] != 0) {
                continue;
            }
            
            color(i, gardens, adjMap);
        }
        
        return gardens;
    }
    
    private void color(int i, int[] gardens, Map<Integer, List<Integer>> adjMap) {
        if(gardens[i] != 0) {
            return;
        }
        
        List<Integer> adjs = adjMap.getOrDefault(i, EMPTY);
        boolean[] adjColors = new boolean[5];
        
        for(Integer adj : adjs) {
            adjColors[gardens[adj]] = true;
        }
        
        for(int j = 1; j <= 4; j++) {
            if(!adjColors[j]) {
                gardens[i] = j;
                break;
            }
        }
        
        for(Integer adj : adjs) {
            color(adj, gardens, adjMap);
        }
    }
}
