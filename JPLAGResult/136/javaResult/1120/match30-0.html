class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] res = new int[N];
        int[][] linked = new int[N][4];
        for (int[] i : paths) {
        	i[0] = i[0] - 1;
        	i[1] = i[1] - 1;
        	linked[i[0]][linked[i[0]][0] + 1] = i[1];
        	linked[i[0]][0] += 1;
        	linked[i[1]][linked[i[1]][0] + 1] = i[0];
        	linked[i[1]][0] += 1;
        }
        for (int i = 0; i < N; ++i) {
        	boolean[] occ = new boolean[4];
        	for (int j = 1; j <= linked[i][0]; ++j) {
        		if (res[linked[i][j]] != 0) {
        			occ[res[linked[i][j]] - 1] = true;
        		}
        	}
        	for (int j = 0; j < 4; ++j) {
        		if (!occ[j]) {
        			res[i] = j + 1;
        			break;
        		}
        	}
        }
        return res;
    }
}
