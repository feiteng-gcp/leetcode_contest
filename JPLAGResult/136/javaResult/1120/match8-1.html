import java.util.ArrayList;
import java.util.Arrays;

class Solution {
	public static int N;
	public static int[] types;
	public static ArrayList<ArrayList<Integer>> edges;

	public int[] gardenNoAdj(int n, int[][] paths) {
		N = n;
		edges = new ArrayList<
ArrayList<Integer>>();
		for (int i = 0; i < N; i++) {
			edges.add(new ArrayList<Integer>());
		}
		for (int[] is : paths) {
			int a = is[0]-1;
			int b = is[1]-1;
			edges.get(a).add(b);
			edges.get(b).add(a);
		}
		types = new int[N];
		Arrays.fill(types, -1);
        for(int i=0;i<N;i++){
        dfs(i);
    }
        for(int i=0;i<N;i++){
types[i]++;}
		return types;
	}

	public static void dfs(int curr) {
		if (types[curr] != -1)
			return;
		boolean[] flow
ers = new boolean[4];
		for (int j : edges.get(curr)) {
			if (types[j] != -1) {
				flowers[types[j]] = true;
			}
		}
		for (int i = 0; i < 4; i++) {
			if (!flowers[i]) {
				types[curr] = i;
				break;
			}
		}
		for (int j : edges.get(curr)) {
			dfs(j);
		}
	}
}
