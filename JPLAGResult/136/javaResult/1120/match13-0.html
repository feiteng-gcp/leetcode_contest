class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] ans = new int[N];
        Arrays.fill(ans, -1);
        Map<Integer, List<Integer>> adjList = new HashMap<>();
        for (int i = 1; i <= N; i++) {
            adjList.put(i, new ArrayList<Integer>());
        }
        for (int[] path : paths) {
            adjList.get(path[0]).add(path[1]);
            adjList.get(path[1]).add(path[0]);
        }
        
        // for (int key : adjList.keySet()) {
        //     System.out.println(key + ": " + adjList.get(key));
        // }
        
        for (int i = 1; i <= N; i++) {
            boolean[] avail = new boolean[5];
            Arrays.fill(avail, true);
            // System.out.println("i: " + i);
            for (int nb : adjList.get(i)) {
                // System.out.println("nb: " + nb);
                if (ans[nb - 1] == -1) continue;
                avail[ans[nb - 1]] = false;
            }
            // System.out.println(Arrays.toString(avail));
            for (int j = 1; j <= 4; j++) {
                if (avail[j]) {
                    ans[i - 1] = j;
                    // System.out.println("j: " + j);
                    break;
                }
            }
            // System.out.println(Arrays.toString(ans));
        }
        
        return ans;
    }
}
