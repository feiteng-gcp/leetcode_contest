class Solution {
    public int[] gardenNoAdj(int n, int[][] paths) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for(int[] p : paths) {
            map.putIfAbsent(p[0], new HashSet<>());
            map.putIfAbsent(p[1], new HashSet<>());
            map.get(p[0]).add(p[1]);
            map.get(p[1]).add(p[0]);
        }
        int[] res = new int[n];
        for(int i = 0; i < n; i ++) {
            for(int type = 1; type <= 4; type ++) {
                boolean can = true;
                if(map.containsKey(i + 1)) {
                    for(int t : map.get(i + 1)) {
                        if(res[t-1] == type) {
                            can = false;
                            break;
                        }
                    }
                }
                if(can) {
                    res[i] = type;
                    break;
                }
            }
        }
        return res;
    }
