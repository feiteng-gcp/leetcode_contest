class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] colors = new int[N];
        int[] valid = new int[N];
        List<List<Integer>> path = new ArrayList<>();
        
for (int i=0; i<= N; i++) {
            path.add(new ArrayList<Integer>());
        }
        for (int[] p:paths){
            path.get(p[0]).add(p[1]);
            path.get(p[1]).add(p[0]);            
        }
        for (int i=0; i<N; i++){
            for (int j=0; j<4;j++) {
                if ((valid[i] & (1<<j)) == 0)
{
                    colors[i] = j+1;
                    break;
                }
            }
            for (int neib : path.get(i+1)) {
                valid[neib-1] = valid[neib-1] | (1<<(colors[i]-1));
            }
        }
        return colors;
    }
}
