class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] res = new int[N];
        Map<Integer, List<Integer>> graph = new HashMap<>();
        for (int i = 1; i <= N; i++) {
            graph.put(i, new ArrayList<Integer>());
        }
        for (int[] edge: paths) {
            List<Integer> nbs1 = graph.get(edge[0]);
            nbs1.add(edge[1]);
            List<Integer> nbs2 = graph.get(edge[1]);
            nbs2.add(edge[0]);
        }
        for (int i = 1; i <= N; i++) {
            boolean[] color = new boolean[4];
            List<Integer> nbs = graph.get(i);
            for (Integer j: nbs) {
                
if (res[j - 1] > 0) {
                    color[res[j - 1] - 1] = true;
                }
            }
            for (int k = 0; k < 4; k++) {
                if (!color[k]) {
                    res[i - 1] = k + 1;
                    break;
                }
            }
        }
        return res;
    }
}
