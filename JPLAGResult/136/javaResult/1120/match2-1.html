class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] result = new int[N];
        Map<Integer,List<Integer>> map = new HashMap<Integer,List<Integer>>();
        for(int i=1;i<=N;i++){
            map.put(i, new ArrayList<Integer>());
        }
        for(int[] p : paths){
            int x = p[0];
            int y = p[1];
            
map.get(x).add(y);
            map.get(y).add(x);
        }
        for(int i=1;i<=N;i++){
            boolean[] used = new boolean[4];
            for(Integer n : map.get(i)){
                
if(result[n-1]!=0){
                    used[result[n-1]-1] = true;
                }
            }
            for(int j=0;j<4;j++){
                if(!used[j]){
                    result[i-1] = j+1;
                    break;
                }
            }
        }
        return result;
    }
}
