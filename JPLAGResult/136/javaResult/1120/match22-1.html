class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        Map<Integer, Set<Integer>> graph = new HashMap();
        for (int[] p : paths) {
            int from  = p[0] - 1;
            int to = p[1] - 1;
            graph.putIfAbsent(from, new HashSet<Integer>());
            graph.get(from).add(to);
            graph.putIfAbsent(to, new HashSet<Integer>());
            
graph.get(to).add(from);
        }
        
        int[] color = new int[N];
        for (int i = 0; i < N; i++) {
            if (color[i] == 0) {
                dfs(i, color, graph);
            }
        
}
        return color;
    }
    
    private void dfs(int cur, int[] color, Map<Integer, Set<Integer>> graph) {
        Set<Integer> neighs = graph.get(cur);
        if (neighs == null) {
            color[cur] = 1;
            return;
        }
        for (int icolor = 1; icolor <= 4; icolor++) {
            boolean isOK = true;
            
for (int nn : neighs) {
                if (color[nn] == icolor) {
                    isOK = false;
                    break;
                }
            }
            if (isOK) {
                color[cur] = icolor;
                for (int nn : neighs) 
{
                    if (color[nn] == 0) {
                        dfs(nn, color, graph);
                    }
                }
            }
        }
    }
    
    
}
