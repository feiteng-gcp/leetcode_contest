class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] res = new int[N];
        boolean[][] conn = new boolean[N][N];
        for(int[] x : paths) {
            conn[x[0] - 1][x[1] - 1] = true;
            conn[x[1] - 1][x[0] - 1] = true;
        }
        for(int k = 1; k <= 4; ++k) {
            for(int i = 0; i < N; ++i) {
                if(res[i] == 0) {
                    boolean f = true;
                    for(int j = 0; j < N; ++j) {
                        if(conn[i][j] && res[j] == k) {
                            f = false;
                            break;
                        }
                    }
                    if(f) {
                        res[i] = k;
                    }
                }
            }
        }
        return res;
    }
}
