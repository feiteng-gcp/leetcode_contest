class Solution {
    List<List<Integer>> nei;
    int[] col;
    
    public int[] gardenNoAdj(int N, int[][] paths) {
        nei = new ArrayList<>();
        col=new int[N+1];
        for (int i = 0; i <= N; i++) {
            nei.add(new ArrayList<>());
        }
        for(int[] p:paths){
            nei.get(p[0]).add(p[1]);
            nei.get(p[1]).add(p[0]);
        }
        for(int i=1;i<=N;i++){
            if(col[i]==0){
                dfs(i);
            }
        }
        int[] sol=new int[N];
        for(int i=0;i<N;i++){
            sol[i]=col[i+1];
        }
        return sol;
    }
    
    void dfs(int i){
        Set<Integer> s=new HashSet<>();
        for(int n:nei.get(i)){
            s.add(col[n]);
        }
        for(int c=1;c<=4;c++){
            if(!s.contains(c)){
                col[i]=c;
                break;
            }
        }
        for(int n:nei.get(i)){
            if(col[n]==0){
                dfs(n);
            }
        }
    }
}
