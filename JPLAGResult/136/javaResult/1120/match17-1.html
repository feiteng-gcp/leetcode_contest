class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] ret = new int[N]; 
        Map<Integer, List<Integer>> p = new HashMap<>(); 
        for (int i = 0; i < paths.length; i++) {
            p.put(paths[i][0], new ArrayList<Integer>());
            
p.put(paths[i][1], new ArrayList<Integer>());
        }
        for (int i = 0; i < paths.length; i++) {
            p.get(paths[i][0]).add(paths[i][1]);
            p.get(paths[i][1]).add(paths[i][0]);
        }
        
        for (int i = 1; i <= N; i++) {
            if (ret[i - 1] != 0) {
                continue; 
            }
            boolean[] vis = new boolean[4];
            List<Integer> nei = p.get(i);
            for (int j = 0; nei != null && j < nei.size(); j++) {
                int t = nei.get(j);
                if (ret[t - 1] != 0) {
                    vis[ret[t - 1] - 1] = true;
                }
            
}
            for (int j = 0; j < 4; j++) {
                if (!vis[j]) {
                    ret[i - 1] = j + 1; 
                    break; 
                }
            }
        }
        return ret;
    }
}
