class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        ArrayList g[] = new ArrayList[N];
        
        for(int y=0;y<N;++y){
            g[y] = new ArrayList();
        }
        
        
        for(int[] p:paths){
            g[p[0]-1].add(p[1]-1);
            
            g[p[1]-1].add(p[0]-1);
        }
        
        int a[] = new int[N];
        
        for(int i=0;i<N;++i){
            if(a[i]==0){
               // a[i] = 1;
                dfs(i,g,a);
                
            }
        }
        return a;
    }
    
    public boolean dfs(int i,ArrayList[] g,int[] a){
        
        
        
        
        outer:for(int u=1;u<=4;++u){
            
            
            
            for(Object p:g[i]){
                
                int k = (Integer)p;
                if(a[k]==u){
                    
                    continue outer;
                }
                
                
            }
            a[i] = u;
             
            for(Object p:g[i]){
                
                int k = (Integer)p;
                if(a[k]==0){
                    
                    boolean r = dfs(k,g,a);
                    if(!r){
                        continue outer;
                    }
                }
                
                
            }
        }
        
        
        
        return a[i]>0;
        
        
    }
    
    
    
    
    
    
    
}
