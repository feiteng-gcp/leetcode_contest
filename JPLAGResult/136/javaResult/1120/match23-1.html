class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        int[] ret = new int[N];
        
HashMap<Integer, HashSet<Integer>> edges = new HashMap<>();
        for (int i = 0; i < paths.length; i++) {
            int s = paths[i][0] - 1;
            int e = paths[i][1] - 1;
            edges.putIfAbsent(s, new HashSet<Integer>());
            edges.putIfAbsent(e, new HashSet<Integer>());
            edges.get(s).add(e);
            edges.get(e).add(s);
        }        
        
        
for (int i = 0; i < N; i++) {
            if (ret[i] == 0) {
                fill(ret, i, edges);
            }
        }
        
        return ret;
    }
    
    public boolean fill(int[] colors, int index, HashMap<Integer, HashSet<Integer>> edges) {
        if (colors[index] != 0) {
            return true;
        }
        
        HashSet<Integer> neighbors = edges.getOrDefault(index, new HashSet<Integer>());
        
for (int i = 1; i <= 4; i++) {
            boolean colorUsed = false;
            for (int neighbor : neighbors) {
                if (colors[neighbor] == i) {
                    colorUsed = true;
                    break;
                }
            }
            if (colorUsed) {
                continue;
            }
            
            colors[index] = i;
            boolean isGood = true;
            for (int neighbor : neighbors) {
                if (!fill(colors, neighbor, edges)) {
                    isGood = false;
                    break;
                }
            }
            if (isGood) {
                return true;
            }
            colors[index] = 0;
        }
        
        return false;
    }
}
