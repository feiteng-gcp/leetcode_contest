class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int [] p: paths) {
            int x = p[0] - 1, y = p[1] - 1;
            if (!map.containsKey(x)) map.put(x, new HashSet<>());
            if (!map.containsKey(y)) map.put(y, new HashSet<>());
            map.get(x).add(y);
            map.get(y).add(x);
        }
        int[] colors = new int[N];
        for (int i = 0; i < N; i++) {
            dfs(i, colors, map);
        }
        return colors;
    }
    
    void dfs(int x, int[] colors, Map<Integer, Set<Integer>> m) {
        if (m.get(x) == null) {
            colors[x] = 1;
            return ;
        }
        int[] pre = new int[5];
        for (int next: m.get(x)) {
            pre[colors[next]] = 1;
        }
        for (int i = 1; i < 5; i++) {
            if (pre[i] == 0) {
                colors[x] = i;
                break;
            }
        }
        for (int next: m.get(x)) {
            if (colors[next] == 0) {
                dfs(next, colors, m);
            }
        }
    }
}
