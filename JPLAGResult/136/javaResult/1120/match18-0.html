class Solution {
    public int[] gardenNoAdj(int N, int[][] paths) {
        ArrayList<ArrayList<Integer>> links = new ArrayList<>(N + 1);
        for (int i = 0; i <= N; i++) {
            links.add(new ArrayList<>());
        }
        for (int[] path : paths) {
            int x = Math.min(path[0], path[1]);
            int y = Math.max(path[0], path[1]);
            links.get(y).add(x);
        }
        int[] result = new int[N];
        for (int i = 1; i <= N; i++) {
            boolean[] temp = new boolean[5];
            for (int x : links.get(i)) {
                temp[result[x - 1]] = true;
            }
            for (int j = 1; j <= 4; j++) {
                if (temp[j] == false) {
                    result[i - 1] = j;
                    break;
                }
            }
        }
        return result;
    }
}
