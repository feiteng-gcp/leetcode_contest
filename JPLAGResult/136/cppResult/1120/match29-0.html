class Solution {
public:
    void color(vector<int>& arr, unordered_map<int, vector<int>>& edge, int src) {
        arr[src] = 0;
        vector<int> dq = {src};
        while(!dq.empty()) {
            vector<int> new_dq;
            for(int node: dq) {
                if (arr[node] == -1) continue;
                vector<bool> used(4, false);
                for(int neibor: edge[node]) {                    
                    if (arr[neibor] != -1) {
                        used[arr[neibor]] = true;
                    }                    
                }
                for(int i = 0; i < 4; i++) {
                    if (used[i] == false) {
                        arr[node] = i;
                        break;
                    }
                }
                // cout << "color" << node << " = " << arr[node] << endl;
                for(int neibor: edge[node]) {                    
                    if (arr[neibor] == -1) {
                        new_dq.push_back(neibor);
                    }                    
                }
            }
            dq.swap(new_dq);
        }        
    }
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        unordered_map<int, vector<int>> edge;
        for(auto vec: paths) {
            edge[vec[0]-1].push_back(vec[1]-1);
            edge[vec[1]-1].push_back(vec[0]-1);
        }
        vector<int> ans(N, -1);
        for(int i = 0; i < N; i++) {
            if (ans[i] == -1) {
                color(ans, edge, i);
            }
        }
        for(int i = 0; i < ans.size(); i++)
            ans[i] += 1;
        return ans;
    }
};
