using namespace std;
class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> visited(N);
        
        vector<vector<int>> graph(N);
        for (auto p : paths)
        {
            graph[p[0] - 1].push_back(p[1] - 1);
            graph[p[1] - 1].push_back(p[0] - 1);
        }
        
        for (int start = 0; start < N; start++)
        {
            if (visited[start]) continue;

            queue<int> q;
            q.push(start);
            while(!q.empty())
            {
                auto p = q.front(); q.pop();
                if (visited[p]) { continue; }

                int x = 0;
                for (auto i : graph[p])
                {
                    x |= 1 << visited[i]; 
                    if (!visited[i])
                    {
                        q.push(i);
                    }
                }

                if ((x & 2) == 0) { visited[p] = 1; continue; }
                if ((x & 4) == 0) { visited[p] = 2; continue; }
                if ((x & 8) == 0) { visited[p] = 3; continue; }
                if ((x & 16) == 0) { visited[p] = 4; continue; }
            }
        }

        return visited;
    }
};
