class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> res(N,-1);
        vector<set<int>> nei(N);
        for(auto item:paths){
            nei[item[0]-1].insert(item[1]-1);nei[item[1]-1].insert(item[0]-1);
        }
        for(int i=0;i<N;i++){
            set<int> tmp{1,2,3,4};
            for(auto j: nei[i]){
                if(res[j]!=-1)
                    tmp.erase(res[j]);
            }
            res[i]=*tmp.begin();
        }
        return res;
    }
};
