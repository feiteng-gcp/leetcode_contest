class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> res(N,1);
        unordered_map<int,vector<int>> m;
        for(auto& p:paths){
            int x=p[0]-1, y=p[1]-1;
            m[x].push_back(y);
            m[y].push_back(x);
            if(res[x]==res[y]){

                for(int i=1;i<=4;++i){
                    bool done=true;
                    for(auto c:m[x]){
                        if(i==res[c]) done=false;
                    }
                    if(done){
                        res[x]=i;
                        break;
                    }
                }
            }
        }
        return res;
    }
};
