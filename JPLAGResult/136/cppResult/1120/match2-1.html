class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N, -1);
        vector<int> edges[10000];
        for(auto& v : paths){
            edges[v[0]-1].push_back(v[1]-1);
            edges[v[1]-1].push_back(v[0]-1);
        }

        for(int i=0; i<N; i++){
            set<int> ex;
            for(int j : edges[i]) ex.insert(ans[j]);
            for(int k=1; k<=4; k++) if(!ex.count(k)){
                ans[i] = k;
                break;
            }
        }
        return ans;
    }
};
