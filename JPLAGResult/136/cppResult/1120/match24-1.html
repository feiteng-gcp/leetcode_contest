vector<int> v[10001];
class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        for(int i=1;i<=N;i++)
        {
            v[i].clear();
        }
        int m=paths.size();
        for(int i=0;i<m;i++)
        {
            v[paths[i][0]].push_back(paths[i][1]);
            v[paths[i][1]].push_back(paths[i][0]);
        }

        vector<int> ans;
        for(int i=0;i<N;i++)
        {
            ans.push_back(0);
        }
        for(int i=1;i<=N;i++)
        {
            int ok[5];
            memset(ok,1,sizeof(ok));
            int len=v[i].size();
            for(int j=0;j<len;j++)
            {

                ok[ans[v[i][j]-1]]=0;
            }
            for(int j=1;j<=4;j++)
            {
                if(ok[j])
                {
                    ans[i-1]=j;
                    break;
                }
            }
        }
        return ans;
    }
};
