class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        unordered_map<int, unordered_set<int>> m;
        for (auto &p : paths) {
            m[p[0]].insert(p[1]);
            m[p[1]].insert(p[0]);
        }
        
        vector<int> colors(N+1, 1);
        for (int c = 1; c <= 4; c++) {
            for (int i = 1; i <= N; i++) {
                if (colors[i] == c) {
                    for (int j : m[i]) if (colors[j] == c) colors[j]++;
                }
            }
        }
        
        colors.erase(colors.begin());
        return colors;
    }
};
