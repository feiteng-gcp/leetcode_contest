class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<unordered_set<int>> neighbors(N, unordered_set<int>());
        vector<int> res(N, -1);
        
        for (auto p : paths) {
            neighbors[p[1]-1].insert(p[0]-1);
            neighbors[p[0]-1].insert(p[1]-1);
        }
        
        for (int i = 0; i < N; ++i) {
            string opt = "1111";
            for (auto n : neighbors[i]) {
                if (res[n] >= 0) opt[res[n]-1] = '0';
            }
            int idx = opt.find("1");
            res[i] = idx+1;
        }
        
        return res;
    }
};
