class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        
        unordered_map<int, vector<int>> neighbor;
        
        for (auto p : paths) {
            neighbor[p[0] - 1].push_back(p[1] - 1);
            neighbor[p[1] - 1].push_back(p[0] - 1);
        }
        
        vector<int> color;
        
        for (int i = 0; i < N; i++)
            color.push_back(0);
        
        color[0] = 1;
        
        for (int i = 1; i < N; i++) {
            unordered_set<int> no_color;
            for (auto n : neighbor[i])
                no_color.insert(color[n]);
            for (int j = 1; j <= 4; j++) {
                if (no_color.find(j) == no_color.end()) {
                    color[i] = j;
                    break;
                }
            }
        }
        
        
        return color;
        
    }
};
