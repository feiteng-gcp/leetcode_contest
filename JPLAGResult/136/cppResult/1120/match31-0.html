class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        unordered_map< int, vector< int > > m;
        for( auto& p : paths ) {
            m[p[0] - 1].push_back( p[1] - 1 );
            m[p[1] - 1].push_back( p[0] - 1 );
        }
        
        vector< int > res( N, -1 );
        for( int i = 0; i < N; i++ )
        {
            vector< int > cand( 4, -1 );
            if( res[i] == -1 ) {
                for( auto neighboor : m[ i ] ) {
                    if( res[neighboor] != -1 )
                        cand[ res[neighboor] - 1 ] = 1;
                }
                
                for( int j = 0; j < 4; j++ )
                    if( cand[j] == -1 ) {
                        res[i] = j + 1;
                        break;
                    }
            }
        }
        return res;
    }
};
