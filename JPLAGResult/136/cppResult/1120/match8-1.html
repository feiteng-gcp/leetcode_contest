class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) 
    {
        vector<int>color(N+1,0);
        unordered_map<int,vector<int>>Map;
        for (auto p:paths)
        {
            Map[p[0]].push_back(p[1]);
            Map[p[1]].push_back(p[0]);
        }
        
        
        for (int i=1; i<=N; i++)
        {
            set<int>Set({1,2,3,4});
            for (auto next: Map[i])
            {
                if (color[next]!=0)
                    Set.erase(color[next]);                    
            }
            color[i] = *Set.begin();       
            //cout<<i<<" "<<color[i]<<endl;
        }
        
        color.erase(color.begin());
        return color;
    }
};
