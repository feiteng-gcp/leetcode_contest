class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        unordered_map<int, vector<int>> prevPath;
        for (auto &path : paths) {
            if (path[0] < path[1]) {
                prevPath[path[1]].push_back(path[0]);
            }
            else {
                prevPath[path[0]].push_back(path[1]);
            }
        }
        vector<int> res(N, 0);
        vector<bool> colors(5, false);
        for (int i = 0; i < N; ++i) {
            int garden = i + 1;
            for (int j = 1; j < 5; ++j) {
                colors[j] = false;
            }
            for (auto x : prevPath[garden]) {
                colors[res[x - 1]] = true;
            }

            for (int j = 1; j < 5; ++j) {
                if (!colors[j]) {
                    res[i] = j;
                    break;
                }
            }
        }
        return res;
    }
};
