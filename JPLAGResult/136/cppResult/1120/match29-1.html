class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {

        vector<vector<int>> adj(N);
        for (auto&& path : paths) {
            adj[path[0] - 1].push_back(path[1] - 1);
            adj[path[1] - 1].push_back(path[0] - 1);
        }
        
        vector<int> result(N);
        for (int i = 0; i < N; i++) {
            if (result[i] == 0) {
                dfs(i, adj, result);
            }
        }
        return result;
    }
    
    void dfs(int i, vector<vector<int>>& adj, vector<int>& result) {
        vector<bool> used(5);
        for (int j : adj[i]) {
            // cout << i << ' ' << j << ' ' << result[j] << endl;
            used[result[j]] = true;

        }
        for (int color = 1; color <= 4; color++) {
            if (!used[color]) {
                result[i] = color;
                break;
            }
        }
        for (int j : adj[i]) {
            if (result[j] == 0) {
                dfs(j, adj, result);
            }
        }
    }
};
