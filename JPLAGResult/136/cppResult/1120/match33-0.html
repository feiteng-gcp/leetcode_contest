class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N, -1);
        vector< vector<int> > graph(N + 1, vector<int>());
        for(int i = 0; i < paths.size(); i ++) {
            int x = paths[i][0];
            int y = paths[i][1];
            graph[x].push_back(y);
            graph[y].push_back(x);
        }
        vector<bool> visited(N + 1, false);
        for(int i = 1; i <= N; i ++) {
            if(visited[i]) continue;
            ans[i-1] = 1;
            visited[i] = true;
            queue<int> qi;
            qi.push(i);
            while(!qi.empty()) {
                int cur = qi.front();
                // cout << "cur: " << cur << endl;
                qi.pop();
                vector<int> & next = graph[cur];
                for(int j = 0; j < next.size(); j ++) {
                    int nextOne = next[j];
                    // cout << "nextOne: " << nextOne << endl;
                    if(visited[nextOne]) continue;
                    unordered_set<int> us;
                    vector<int> & nei = graph[nextOne];
                    for(int k = 0; k < nei.size(); k ++) {
                        us.insert(ans[nei[k] - 1]);
                    }
                    for(int k = 1; k <= 4; k ++) {
                        if(us.find(k) == us.end()) {
                            ans[nextOne - 1] = k;
                            break;
                        }
                    }
                    visited[nextOne] = true;
                    qi.push(nextOne);
                }
            }
        }
        return ans;
    }
};
