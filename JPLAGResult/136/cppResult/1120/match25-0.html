class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        
        vector<int> colors(N, -1);
        vector<int> edges[11111];
        for(int i=0;i<paths.size();i++) {
            int x = paths[i][0];
            int y = paths[i][1];
            x--;
            y--;
            edges[x].push_back(y);
            edges[y].push_back(x);
        }
        
        colors[0]=1;
        for(int i=0;i<N;i++) {
            for(int c=1;c<=4;c++) {
                bool can=true;
                for(int j : edges[i]) {
                    if (colors[j]==c) {
                        can=false;
                        break;
                    }
                }
                if (can) {
                    colors[i]=c;
                    break;
                }
            }
        }
        return colors;
        
    }
};
