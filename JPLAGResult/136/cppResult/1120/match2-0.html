void dfs(int i,     vector<vector<int>>& adj,  vector<int>& ans) {
    set<int> s;
    for (int j : adj[i])
        s.insert(ans[j]);
    for (int j = 1; ; j++)
        if (!s.count(j)) {
            ans[i] = j;
            break;
        }
    for (int j : adj[i])if (!ans[j])dfs(j, adj, ans);
}

class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {vector<int> ans(N);
        vector<vector<int>> adj(N);
        for (auto& i : paths) {
            adj[i[0] - 1].push_back(i[1] - 1);
            adj[i[1] - 1].push_back(i[0] - 1);
        }
        for (int i = 0; i < N; i++)
        if (!ans[i])
        dfs(i, adj, ans);
        return ans;
    }
};
