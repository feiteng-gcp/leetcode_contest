class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N);
        vector<vector<int>> ne(N,vector<int>());
        for(int i=0;i<paths.size();i++){
            ne[paths[i][0]-1].push_back(paths[i][1]-1);
            ne[paths[i][1]-1].push_back(paths[i][0]-1);
        }
        for(int i=0;i<N;i++){
            vector<bool> allow(4,true);
            for(int n:ne[i]){
                if(n>=i) continue;
                allow[ans[n]-1]=false;

            }
            
            for(int j=0;j<4;j++){
                if(allow[j]){
                    ans[i]=j+1;
                    break;
                }
            }
        }
        return ans;
    }
};
