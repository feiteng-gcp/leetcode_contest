class Solution {
public:
	vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
		vector<vector<int>> connected(N+1, vector<int>());
		vector<int> ans(N+1, 0);

		for (auto p : paths) {
			connected[p[0]].push_back(p[1]);
			connected[p[1]].push_back(p[0]);
		}

		for (int i = 1; i <= N; i++) {
			set<int> colors = { 1,2,3,4 };
			if (i == 1) {
				ans[i] = 1;
				continue;
			}
			else {
				for (int p : connected[i]) {
					if (ans[p] != 0)
						colors.erase(ans[p]);
				}
				ans[i] = *(colors.begin());
			}
		}
		return vector<int>(ans.begin()+1, ans.end());
	}
};
