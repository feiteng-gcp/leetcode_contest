class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> adj[N];
        vector<int> color(N);
        for(auto path:paths){
            int u=path[0]-1, v=path[1]-1;
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
        color[0]=1;
        for(int i=1; i<N; i++){
            int opt[5]={0};
            for(auto v: adj[i]){
                if(v<i) opt[color[v]]=1;
            }

            for(int c=1; c<=4; c++){
                if(!opt[c]){
                    color[i] = c;
                    break;
                } 
            }
        }
        return color;
    }
};
