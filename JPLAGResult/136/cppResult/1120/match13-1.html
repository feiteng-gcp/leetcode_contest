class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        unordered_map<int, vector<int>> dict;
        for (auto path:paths) {
            dict[path[0]].push_back(path[1]);
            dict[path[1]].push_back(path[0]);
        }
        vector<int> res(N, 0);
        

        for (int i = 0; i < N; i++) {
            unordered_set<int> col = {1,2,3,4};
            for (auto j:dict[i+1]) {
                if (res[j - 1] != 0) col.erase(res[j - 1]);
            }
            res[i] = *col.begin();
        }
        
        return res;
    }
};
