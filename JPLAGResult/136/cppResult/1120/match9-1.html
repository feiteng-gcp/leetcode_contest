class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        unordered_map<int, vector<int>> path;
        for(auto v : paths){
            path[v[0]-1].push_back(v[1]-1);
            path[v[1]-1].push_back(v[0]-1);
        }
        vector<int> ans(N);
        for(int i=0;i<N;i++){
            if(path.find(i) == path.end()){
                ans[i] = 1;
                continue;
            }
            
            vector<bool> chk(5, true);
            for(auto j : path[i])
                chk[ans[j]] = false;

            for(int j=1;j<=4;j++){
                if(chk[j] == true){
                    ans[i] = j;
                    break;
                }
            }
        }
        
        return ans;
    }
};
