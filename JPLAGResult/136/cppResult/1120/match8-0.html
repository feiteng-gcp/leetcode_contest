class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ret(N, 1);
        vector<vector<int> >record(N + 1);
        vector<int> checked(N + 1, 0);
        for (auto path : paths) {
            record[path[0]].push_back(path[1]);
            record[path[1]].push_back(path[0]);
        }
        for (int i = 0; i < N; i++) {
            vector<int> tmp(5, 0);
            for (int g : record[i]) {
                tmp[ret[g-1]] = 1;
                if (ret[i-1] == ret[g-1]) {
                    for (int j = 1; j < 5; j++) {
                        if (tmp[j] == 0) {
                            ret[i-1] = j;
                        }
                    }
                }
            }
        }
        return ret;
    }
};
