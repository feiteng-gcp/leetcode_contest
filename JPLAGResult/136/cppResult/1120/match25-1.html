class Solution 
{
    vector<int> res;
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) 
    {
        unordered_map<int, set<int>> uMap;
        for(auto i : paths)
        {
            uMap[i[0]].insert(i[1]);
            uMap[i[1]].insert(i[0]);
        }
        unordered_map<int, int> color;

        for(int i = 1; i <= N; i++)
        {
            for(int c = 1; c <= 4; c++)
            {
                bool flag = false;
                for(auto j : uMap[i])

                {
                    if(color.count(j) > 0 && color[j] == c)
                    {
                        flag = true;
                        break;
                    }    
                }
                if(!flag)
                {
                    res.push_back(c);
                    color[i] = c;
                    break;
                }
            }
        }
        return res;
    }
};
