class Solution {
public:
    void dfs(vector<vector<int>> &g, int u, vector<int> &ans) {

        for (int i = 1; i <= 4; i++) {
            bool valid = true;
            for (auto v : g[u]) {
                if (ans[v-1] == i) {
                    valid = false;
                    break;
                } 
            }
            if (valid) {
                ans[u-1] = i;
                break;
            }
        }
        for (auto v : g[u]) {
            if (ans[v-1] == 0) {
                dfs(g, v, ans);
            }

        }
    }
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N);
        vector<vector<int>> g(N+1);
        for (auto &v : paths) {
            g[v[0]].push_back(v[1]);
            g[v[1]].push_back(v[0]);
        }
        for (int i = 1; i <= N; i++) {
            if (ans[i-1] == 0) {
                dfs(g, i, ans);
            }
        }
        return (ans);
    }
};
