class Solution {
public:
    void dfs(vector<int> & color, int u, vector<vector<int>> & g) {
        if (color[u] != -1) return;
        for (int c = 1; c <= 4; ++c) {
            bool flag = true;
            for (auto v : g[u]) {
                if (color[v] == c) {
                    flag = false;
                    break;
                }
            }
            if (flag == true) {
                color[u] = c;
                break;
            }
        }
        for (auto v : g[u]) {
            dfs(color, v, g);
        }
    }
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<vector<int>> g(N);
        vector<int> color(N, -1);
        for (auto & e : paths) {
            g[e[0] - 1].push_back(e[1] - 1);
            g[e[1] - 1].push_back(e[0] - 1);
        }
        for (int i = 0; i < N; ++i) {
            if (color[i] == -1)
                dfs(color, i, g);
        }
        
        return color;
    }
};
