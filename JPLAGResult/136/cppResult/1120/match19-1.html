class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N, 1);
        vector<vector<int>> nb(N);
        for(auto p : paths){
            int i = p[0];
            int j = p[1];
            nb[i-1].push_back(j-1);
            nb[j-1].push_back(i-1);
            if(ans[i-1]==ans[j-1]){
                vector<bool> avail(5, true);
                for(auto k : nb[j-1])
                    avail[ans[k]] = false;

                for(int k = 1; k<=4; k++)
                    if(avail[k]){
                        ans[j-1] = k;
                        break;
                    }
            }
        }
        return ans;
    }
};
