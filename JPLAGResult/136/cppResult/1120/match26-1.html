class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N, 0);
        vector<vector<int>> adj_node(N, vector<int>());
        for (const auto& path : paths) {
            adj_node[path[0] - 1].push_back(path[1] - 1);
            adj_node[path[1] - 1].push_back(path[0] - 1);
        }
        for (int i = 0; i < N; i++) {
            int status = 0;
            for (const auto& node : adj_node[i]) {
                if (node > i) { continue; }
                status |= 1 << (ans[node] - 1);
            }
            for (int j = 0; j < 4; j++) {
                if (status & (1 << j)) { continue; }
                ans[i] = j + 1;
                break;
            }
        }
        return ans;
    }
};
