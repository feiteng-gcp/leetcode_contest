class Solution {
public:
	vector<int> gardenNoAdj(int N, vector<vector<int>>& p) {
		vector<int> a(N, -1);
		vector<vector<int>> g(N);
		for (auto& pa : p) {
			g[pa[0] - 1].emplace_back(pa[1] - 1);
			g[pa[1] - 1].emplace_back(pa[0] - 1);
		}

		for (int i = 0; i < N; ++i) {
			if (a[i] != -1) continue;
			set<int> set;
			for (auto& to : g[i]) set.insert(a[to]);
			int p = 1;
			while (set.count(p)) ++p;
			a[i] = p;
		}

		return a;
	}
};
