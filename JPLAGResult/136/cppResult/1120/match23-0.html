class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N, 0);
        unordered_map<int, unordered_set<int>> m;
        for(vector<int> p: paths) {
            m[p[0]-1].insert(p[1]-1);
            m[p[1]-1].insert(p[0]-1);
        }
        ans[0] = 1;
        for(int i=1;i<N;i++) {
            if(ans[i]!=0) continue;
            vector<int> used(5, 0);
            for(int x: m[i]) used[ans[x]]=1;
            for(int j=1;j<5;j++) {
                if(used[j]==0) {
                    ans[i] = j;
                    break;
                }
            }
        }
        return ans;
    }
};
