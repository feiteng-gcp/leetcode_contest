class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> ans(N, 0);
        vector<vector<int>> graph(N);
        int i, j, k;
        for (i = 0; i < paths.size(); i++) {
            graph[paths[i][0] - 1].push_back(paths[i][1] - 1);
            graph[paths[i][1] - 1].push_back(paths[i][0] - 1);
        }
        while (1) {
            for (i = 0; i < N; i++) {
                if (ans[i] == 0) {
                    break;
                }
            }
            if (i == N) {
                return ans;
            }
            queue<int> que;
            que.push(i);
            while (!que.empty()) {
                k = que.front();
                que.pop();
                if (ans[k] > 0) {
                    continue;
                }
                vector<int> v(5, 0);
                for (i = 0; i < graph[k].size(); i++) {
                    if (ans[graph[k][i]] == 0) {
                        que.push(graph[k][i]);
                    }
                    v[ans[graph[k][i]]] = 1;
                }
                for (i = 1; i < 5; i++) {
                    if (v[i] == 0) {
                        break;
                    }
                }
                ans[k] = i;
            }
        }
        return ans;
    }
};
