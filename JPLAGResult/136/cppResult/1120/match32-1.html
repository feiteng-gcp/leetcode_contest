class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        unordered_map<int, set<int>> g;
        for (auto p: paths) {
            g[p[0]].insert(p[1]);
            g[p[1]].insert(p[0]);
        }
        vector<int> ans(N, 0);
        for (int i = 0; i < N; i++) {
            set<int> used;
            set<int> tovisit;
            if (ans[i] == 0) tovisit.insert(i + 1);
            for (auto n: g[i + 1]) {
                if (ans[n - 1] == 0) tovisit.insert(n);
                else used.insert(ans[n - 1]);
            }
            for (auto n: tovisit) {
                for (int i = 1; i <= 4; i ++) {
                    if (used.count(i) == 0) {
                        ans[n - 1] = i;
                        used.insert(i);
                    }
                }
            }
            
        }
        return ans;
    }
};
