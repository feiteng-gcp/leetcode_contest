class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector <int> ans(N);

        vector < vector<int> > map(N, vector<int>());
        for (auto i: paths) {
            map[i[0] - 1].push_back(i[1] - 1);
            map[i[1] - 1].push_back(i[0] - 1);
        }
        for (int i = 0; i < N; i++) {
            for (ans[i] = 1; ans[i] <= 4; ans[i]++) {

                int ok = 1;
                for (int j: map[i]) {
                    if (ans[j] == ans[i]) ok = 0;
                }
                if (ok) {
                    break;
                }
            }
        }
        return ans;
    }
};
