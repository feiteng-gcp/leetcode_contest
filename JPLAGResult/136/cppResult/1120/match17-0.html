class Solution {
public:
    vector<int> gardenNoAdj(int N, vector<vector<int>>& paths) {
        vector<int> result;
        vector<vector<int>> edges(N);
        for (auto & path : paths) {
            edges[path[0] - 1].push_back(path[1] - 1);
            edges[path[1] - 1].push_back(path[0] - 1);
        }
        
        for (int i = 0; i < N; ++i) {
            for (int c = 1; c <= 4; ++c) {
                bool repeated = false;
                for (int neib : edges[i]) {
                    if (neib < result.size() && result[neib] == c)
                        repeated = true;
                }
                if (!repeated) {
                    result.push_back(c);
                    break;
                }
            }
        }
        return result;
    }
};
