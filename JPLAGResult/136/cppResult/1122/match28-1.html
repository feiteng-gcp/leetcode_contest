class Solution {
public:
    typedef long long LL;
string longestDupSubstring(string S)
{
    int l = 0, r = (int)S.size(), len = (int)S.size();
    LL mod = 1e9 + 7, w = 128;
    string ans;
    while(l < r)
    {
        int m = l + (r - l) / 2;
        if(m == 0) break;

        map<LL, int> pos;
        LL cur = 0, ww = w;
        for(int i = 0; i < m - 1; i++)
        {
            cur = (cur * 128 + S[i]) % mod;
            ww = (ww * w) % mod;
        }
        bool ok = false;
        for(int i = m - 1; i < len; i++)
        {
            cur = (cur * 128 + S[i]) % mod;
            if(i >= m) cur = (cur - (ww * S[i - m]) % mod + mod) % mod;
            if(pos.count(cur))
            {
                if(S.substr(pos[cur], m) == S.substr(i - m + 1, m))
                {
                    ans = S.substr(pos[cur], m);
                    ok = true;
                    break;
                }
            }
            pos[cur] = i - m + 1;
        }
        if(ok) l = m + 1;
        else r = m;
    }
    return ans;
}
};
