typedef unsigned long long ull;

const ull P = 73;
int n;

class Solution {
public:
    int get(int m, vector<ull>& h, vector<ull>& p) {
        set<ull> st;

        for (int i = 0; i <= n - m; ++i) {
            ull v = h[i + m] - h[i];
            v *= p[n - i];
            if (st.count(v)) {
                return i;
            }
            st.insert(v);
        }
        return -1;
    }
    
    string longestDupSubstring(string s) {
        n = s.size();
        vector<ull> p;
        p.push_back(1);
        for (int i = 0; i < n; ++i) {
            p.push_back(p.back() * P);
        }
        
        vector<ull> h;
        h.push_back(0);
        for (int i = 0; i < n; ++i) {
            h.push_back(h.back() + (s[i] - 'a' + 1) * p[i]);
        }
        
        int l = 0, r = n;
        while (l < r) {
            int c = (l + r + 1) / 2;
            if (get(c, h, p) != -1) {
                l = c;
            } else {
                r = c - 1;
            }
        }
        if (l == 0) {
            return "";
        }
        int x = get(l, h, p);
        return s.substr(x, l);
    }
};
