class Solution {
public:
    string longestDupSubstring(string S) {
        int l = 0, r = S.length() - 1;
        int pos = 0;
        while (l < r) {
            int mid = (l + r) / 2 + 1;
            if (CheckString(S, mid, &pos)) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        return S.substr(pos, l);
    }
    
private:
    bool CheckString(string& S, int len, int* pos) {
        int n = S.length();
        long long hash_size = 1e9 + 7;
        unordered_map<long long, std::vector<int>> hash_map;
        long long hash_value = 0, top_offset = 1;
        for (int i = 0; i < len - 1; i++) {
            hash_value = (hash_value * 26 + (S[i] - 'a')) % hash_size;
            top_offset *= 26;
            top_offset %= hash_size;
        }
        for (int i = 0; i <= n - len; i++) {
            if (i > 0) {
                hash_value -= (top_offset * (S[i - 1] - 'a')) % hash_size;
                if (hash_value < 0) {
                    hash_value += hash_size;
                }
            }
            hash_value = (hash_value * 26 + (S[i + len - 1] - 'a')) % hash_size;
            auto iter = hash_map.find(hash_value);
            if (iter != hash_map.end()) {
                for (const auto& last_pos : iter->second) {
                    if (S.substr(i, len) == S.substr(last_pos, len)) {
                        *pos = i;
                        return true;
                    }
                }
            }
            hash_map[hash_value].push_back(i);
        }
        return false;
    }
};
