const int MAXN = 400010;

class SA
{
public:
    int n;
    char *s;
    int *x,*y,*sum,*wv,*sa,*rank,*h;
    int *pow,*lg;
    SA (int MaxN)
    {
        s=new char[MaxN]();
        x=new int[MaxN<<1]();
        y=new int[MaxN<<1]();
        sum=new int[MaxN]();
        wv=new int[MaxN]();
        sa=new int[MaxN]();
        rank=new int[MaxN]();
        h=new int[MaxN]();
        pow=new int[20]();
        lg=new int[MaxN]();
    }
    void build_sa(char *_s)
    {
        n=strlen(_s+1);
        for (int i=1;i<=n;++i)
            s[i]=_s[i];
        int m=255;
        for (int i=1;i<=m;++i) sum[i]=0;
        for (int i=1;i<=n;++i) sum[x[i]=s[i]]++;
        for (int i=1;i<=m;++i) sum[i]+=sum[i-1];
        for (int i=n;i>=1;--i) sa[sum[x[i]]--]=i;
        for (int j=1,p=1;p<n;j*=2,m=p)
        {
            p=0;
            for (int i=n-j+1;i<=n;++i) y[++p]=i;
            for (int i=1;i<=n;++i) if (sa[i]>j) y[++p]=sa[i]-j;
            for (int i=1;i<=n;++i) wv[i]=x[y[i]];
            for (int i=1;i<=m;++i) sum[i]=0;

            for (int i=1;i<=n;++i) sum[wv[i]]++;
            for (int i=1;i<=m;++i) sum[i]+=sum[i-1];
            for (int i=n;i>=1;--i) sa[sum[wv[i]]--]=y[i];
            for (int i=1;i<=n;++i) y[i]=x[i];
            p=x[sa[1]]=1;
            for (int i=2;i<=n;++i)
            {
                if (!(y[sa[i]]==y[sa[i-1]]&&y[sa[i]+j]==y[sa[i-1]+j])) ++p; 
                x[sa[i]]=p;
            }
        }
    }
    void build_height()
    {
        for (int i=1;i<=n;++i) rank[sa[i]]=i;
        int k=0;
        for (int i=1;i<=n;++i)
        {
            if (k) --k;
            while (s[i+k]==s[sa[rank[i]-1]+k]) ++k;
            h[rank[i]]=k;
        }
    }
}T(MAXN);

char s[MAXN];

class Solution {
public:
    using pii = pair<int, int>;
    string longestDupSubstring(string S) {
        int n = S.size();

        for (int i = 0; i < n; ++ i)
            s[i+1] = S[i];
        for (int i = n+1; i <= 4*n; ++ i)
            s[i] = 0;
        
        T.build_sa(s);
        T.build_height();
        
        pii ans(0, 1);
        for (int i = 1; i <= n; ++ i)
            ans = max(ans, pii(T.h[i], i));
        return S.substr(T.sa[ans.second]-1, ans.first);
    }
};
