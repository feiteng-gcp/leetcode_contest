#include <iostream>
#include <map>
#include <vector>
#include <string>
#include <set>
#include <random>

using namespace std;
typedef unsigned long long ll;
const int MAXN = 100100;
const ll BASE1 = 918357, BASE2 = 918357;
const ll M1 = 1e9+7, M2 = 1e9+9;

int N;
string s;
ll cp[MAXN];
ll bpow[MAXN];

ll get_hash (int l, int len)
{
    return cp[l+len] - (cp[l] * bpow[len]);
}

int oloc[MAXN];

int nshare (int x, int y)
{
    int lo = 0, hi = min (N - x, N - y);
    while (lo < hi)
    {
        int mid = (lo + hi + 1) / 2;
        if (get_hash (x, mid) == get_hash (y, mid))
            lo = mid;
        else
            hi = mid - 1;
    }
    return lo;
}

bool cmp (int x, int y)
{
    int v = nshare (x, y);
    if (v + x == N) return true;
    if (v + y == N) return false;
    return s[v+x] < s[v+y];
}

class Solution {
public:
    string longestDupSubstring(string S) {
        s = S;
        N = s.length();

        cp[0] = 0;
        for (int i = 0; i < N; i++)
            cp[i+1] = cp[i] * BASE1 + s[i];
        bpow[0] = 1;
        for (int i = 0; i < N; i++)
            bpow[i+1] = bpow[i] * BASE1;


        for (int i = 0; i < N; i++)
            oloc[i] = i;
        sort (oloc, oloc + N, cmp);

        int bmax = -1, bloc = -1;
        for (int i = 0; i < N - 1; i++)
        {
            int v = nshare (oloc[i], oloc[i+1]);
            if (v > bmax)
            {
                bmax = v;
                bloc = oloc[i];
            }
        }

        string res = "";
        for (int i = 0; i < bmax; i++)
            res += s[bloc+i];
        return res;
    }
};
