class Solution {
public:
    struct RollingHash{
        vector<int64_t> hash;
        vector<int64_t> base_pow;
        int64_t base, mod;
        const int64_t DEFAULT_BASE = 1009, DEFAULT_MOD = 1e9+7;

        RollingHash(string S="", int64_t base_=-1, int64_t mod_=-1){
            base = base_ > 0 ? base_ : DEFAULT_BASE;
            mod = mod_ > 0 ? mod_ : DEFAULT_MOD;
            if(S.size() > 0) initialize(S);
        } 


        void initialize(string S){
            hash.resize(S.size()+1);
            base_pow.resize(S.size()+1);
            hash[0] = 0;
            base_pow[0] = 1;
            for(int i=1; i<=S.size(); i++){
                hash[i] = (hash[i-1] * base + S[i-1]) % mod;
                base_pow[i] = base_pow[i-1] * base % mod;
            }
        }

        // 0-indexed, [a, b)
        int64_t between(int a, int b){
            return (mod + hash[b] - hash[a] * base_pow[b-a] % mod) % mod;
        }
    };

    string longestDupSubstring(string S) {
        int N = S.size();
        RollingHash hash1(S, 1009), hash2(S, 1007);
        int ok = 0, ng = S.size();

        typedef pair<int64_t, int64_t> P;
        while(ng-ok>1){
            int mid = (ok+ng)/2;
            bool dup = false;
            set<P> st;
            for(int i=0; i+mid<=N; i++){
                int64_t v1 = hash1.between(i, i+mid);
                int64_t v2 = hash2.between(i, i+mid);
                P p = {v1, v2};
                if(st.count(p)){
                    dup = true;
                    break;
                }else{
                    st.insert(p);
                }
            }
            (dup ? ok : ng) = mid;
        }

        set<P> st;
        for(int i=0; i+ok<=N; i++){
            int64_t v1 = hash1.between(i, i+ok);
            int64_t v2 = hash2.between(i, i+ok);
            P p = {v1, v2};
            if(st.count(p)){
                return S.substr(i, ok);
            }else{
                st.insert(p);
            }
        }
        return "";
    }
};
