class Solution {
public:
    string longestDupSubstring(string S) {
      if (S.empty()) {
        return "";
      }
      uint32_t seed = 100009;
      std::vector<uint32_t> sum(S.size() + 1), pw(S.size() + 1);
      pw[0] = 1;
      for (int i = 0; i < S.size(); ++i) {
        sum[i + 1] = sum[i] * seed + S[i];
        pw[i + 1] = pw[i] * seed;
      }
      
      auto get = [&](int l, int r) {
        return sum[r + 1] - sum[l] * pw[r - l + 1];
      };
      auto ck = [&] (int len) {
        std::map<uint32_t, int> f;
        for (int i = 0; i + len - 1 < S.size(); ++i) {
          auto hs = get(i, i + len - 1);
          if (f.count(hs) && S.substr(i, len) == S.substr(f[hs], len)) {
            return f[hs];
          }
          f[hs] = i; 
        }
        return -1;
      };
      int l = 0, r = S.size();
      while (l <= r) {
        int mid = l + (r - l) / 2;
        if (ck(mid) != -1) {
          l = mid + 1;
        } else {
          r = mid - 1;
        }
      }
      int len = r;
      if (len <= 0) {
        return "";
      }
      //std::cout << len << " " << ck(len) << std::endl;
      return S.substr(ck(len), len);
    }
};
