class Solution {
public:
    bool isRobotBounded(string instructions) {
        int x = 0;
        int y = 0;
        int d = 0;
        int dx[] = {0, 1, 0, -1};
        int dy[] = {1, 0, -1, 0};
        
        map<tuple<int,int,int>, int> memo[4];
        
        for(int i = 0 ; i < 10000 ; i++){
            if( memo[d].count({i % (int)instructions.size(), x,y})) return true;
            memo[d][{i % (int)instructions.size(), x,y}] = true;
            char c = instructions[i % instructions.size()];
            if(c == 'L'){
                d = (d+1) % 4;
            }else if(c == 'R'){
                d = (d+3) % 4;
            }else{
                x += dx[d];
                y += dy[d];
            }
        }
        return false;
                                                                                
    }
};
