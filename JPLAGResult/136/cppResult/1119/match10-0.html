class Solution {
public:
    bool isRobotBounded(string instructions) {
        int dirx[] = {0, 1, 0, -1};
        int diry[] = {1, 0, -1, 0};
        int dir(0);
        int posx(0);
        int posy(0);
        
        for (const auto c : instructions)
        {
            switch (c)
            {
                case 'G':
                    posx += dirx[dir];
                    posy += diry[dir];
                    break;
                case 'L':
                    dir = (dir + 3) % 4;
                    break;
                case 'R':
                    dir = (dir + 1) % 4;
                    break;
            }
        }
        
        return (posx == 0 && posy == 0) || dir;
    }
};
