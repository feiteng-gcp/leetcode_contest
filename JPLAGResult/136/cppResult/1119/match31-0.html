class Solution {
public:
    bool isRobotBounded(string s) {
        string expand = s + s + s + s;
        vector<vector<int>> dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        unordered_set<string> visited;
        
        int d = 0, x = 0, y = 0;
        for (char c : expand) {
            if (c == 'G') { x += dir[d][0]; y += dir[d][1]; }
            else if (c == 'R') d = (d + 1) % 4;
            else d = (d + 3) % 4;
        }
        
        return x == 0 && y == 0 && d == 0;
    }
};
