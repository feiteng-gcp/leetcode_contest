class Solution {
public:

    int dir[4][2] = {
        {-1, 0},
        {0, 1},
        {1, 0},
        {0, -1},
    };
    string get_key(int x, int y, int direction) {
        return to_string(x) + "_" + to_string(y) + "_" + to_string(direction);
    }
    unordered_set<string> visited;
    bool perform(string& instruction, int &x, int &y, int &direction) {

        for(char ch: instruction) {
            if(ch == 'G') {
                x += dir[direction][0];
                y += dir[direction][1];
            }
            else if(ch == 'L')
                direction = (direction + 3) % 4;
            else
                direction = (direction + 1) % 4;
        }
        string key = get_key(x, y, direction);
        if(visited.count(key))
                return true;
        visited.insert(key);
        return false;
    }
    bool isRobotBounded(string instructions) {
        int direction = 0, x = 0, y = 0;
        visited.insert(get_key(x, y, direction));
        for(int i = 0; i < 4; i++) {
            if(perform(instructions, x, y, direction))
                return true;
        }
        return false;
    }
};
