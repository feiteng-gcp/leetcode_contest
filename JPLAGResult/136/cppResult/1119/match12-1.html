class Solution {
public:
    bool isRobotBounded(string instructions) {
        int dir = 0;
        int x = 0, y = 0;

        vector<vector<int>> move = { {-1, 0}, {0, -1}, {1, 0}, {0, 1} };
        for (char& c : instructions) {
            if (c == 'L') {
                dir = (dir + 1) % 4;
            } else if (c == 'R') {
                dir = (dir + 3) % 4;
            } else {
                x += move[dir][0];
                y += move[dir][1];
            }
        }
        return dir != 0 || (x == 0 && y == 0);
    }
};
