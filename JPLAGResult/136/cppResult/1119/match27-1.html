class Solution {
public:
    bool isRobotBounded(string instructions) {
        unordered_set<int> exist;
        vector<pair<int, int>> dirs = { {-1, 0}, {0, -1}, {1, 0}, {0, 1}};
        int curDir = 0;
        pair<int, int> curPos = {0, 0};
        for (int step = 0; step < 4; ++step) {

            for (auto c : instructions) {
                switch (c) {
                    case 'G':
                        curPos.first += dirs[curDir].first;
                        curPos.second += dirs[curDir].second;
                        break;
                    case 'L':
                        curDir = (curDir + 1) % 4;
                        break;
                    case 'R':
                        curDir = (curDir + 3) % 4;
                        break;
                }
            }
            if (curDir == 0 && curPos.first == 0 && curPos.second == 0) {
                return true;
            }
        }
        return false;
    }
};
