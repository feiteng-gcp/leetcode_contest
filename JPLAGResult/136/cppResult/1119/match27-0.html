#define x first
#define y second
class Solution
{
public:
    bool isRobotBounded(string instr) 
    {
        pair<int, int> pos(0, 0);
        int next[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int dir = 0;
        do
        {
            for(char c : instr)
            {
                switch(c)
                {
                    case 'G':
                        pos.x += next[dir][0];
                        pos.y += next[dir][1];
                        break;
                    case 'R':
                        dir = (dir+1)%4;
                        break;
                    case 'L':
                        dir = (dir+3)%4;
                        break;
                }
            }
        }while(dir != 0);
        return pos.x == 0 && pos.y == 0;
    }
};
