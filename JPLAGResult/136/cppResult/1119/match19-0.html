class Solution {
public:
    bool isRobotBounded(string instructions) {
        int dx[] = {1, 0, -1, 0};
        int dy[] = {0, 1, 0, -1};
        int x = 0, y = 0, k = 0;
        for(int t=0; t<4; t++) for(char c : instructions){
            if(c == 'G'){
                x += dx[k];
                y += dy[k];
            }else if(c == 'L'){
                k = (k+3) % 4;
            }else{
                k = (k+1) % 4;
            }
        }
        return (x == 0 && y == 0);
    }
};
