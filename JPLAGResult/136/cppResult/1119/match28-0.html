class Solution {
public:
    bool isRobotBounded(string instructions) {
        int dirs[4][2] = {
            -1, 0,
             0, 1,
             1, 0,
             0, -1,
        };
        int turns[4][2] = {
            3, 1,
            0, 2,
            1, 3,
            2, 0,
        };
        int cr = 0, cc =0;
        int dir = 0;
        for(int i = 0; i < 8; ++i) {
            for(int j = 0; j < instructions.size(); ++j) {
                switch(instructions[j]) {
                    case 'G':
                        cr = cr + dirs[dir][0];
                        cc = cc + dirs[dir][1];
                        break;
                    case 'L':
                        dir = turns[dir][0];
                        break;
                    case 'R':
                        dir = turns[dir][1];
                        break;
                }    
            }
            if(cr == 0 && cc == 0)
                return true;
        }
        return false;
    }
};
