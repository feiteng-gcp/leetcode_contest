class Solution {
public:
    bool isRobotBounded(string instructions) {
        // just to check the new direction of the robot is different from the original one; or if the robot is at origin;

        int dirs[4][2] = {{0, 1}, {-1, 0}, {0, -1}, {1, 0}};
        int x = 0, y = 0, dir = 0;
        for(char c: instructions) {
            if(c == 'G') {
                x += dirs[dir][0];
                y += dirs[dir][1];
            }
            else if(c == 'L') {
                dir = (dir+1)%4;
            }
            else {
                dir--;
                if(dir < 0)
                    dir += 4;
            }
        }
        if(x == 0 && y == 0)
            return true;
        return dir != 0;
    }
};
