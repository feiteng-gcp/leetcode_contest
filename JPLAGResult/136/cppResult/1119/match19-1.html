class Solution {
public:
    bool isRobotBounded(string s) {
        int dx[] = {0, 1, 0, -1};
        int dy[] = {1, 0, -1, 0};
        int x = 0, y = 0;
        int d = 0;
        s = s+s+s+s;
        for (auto c : s)

        {
            if (c == 'G')
            {
                x += dx[d];
                y += dy[d];
            }
            else if (c == 'L')
            {
                d = (d+3)%4;
            }
            else if (c == 'R')
            {
                d = (d+1)%4;
            }
        }
        return x == 0 && y == 0;
    }
};
