class Solution {
    typedef tuple<int, int, int, int, int> tp;
public:
    bool isRobotBounded(string instructions) {
        set<tp> table;
        int x = 0, y = 0;
        int dx = 0, dy = 1;
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < instructions.size(); j++) {
                char c = instructions[j];

                if (c == 'G') {
                    x += dx;
                    y += dy;
                } else if (c == 'L') {
                    if (dy == 1) {
                        dx = -1;
                        dy = 0;
                    } else if (dx == -1) {
                        dy = -1;
                        dx = 0;
                    } else if (dy == -1) {
                        dx = 1;
                        dy = 0;
                    } else { // dx == 1
                        dy = 1;
                        dx = 0;
                    }
                } else { // c == 'R'
                    if (dy == 1) {
                        dx = 1;
                        dy = 0;
                    } else if (dx == 1) {
                        dy = -1;
                        dx = 0;
                    } else if (dy == -1) {
                        dx = -1;
                        dy = 0;
                    } else { // dx == -1
                        dy = 1;
                        dx = 0;
                    }
                }
                tp value{x, y, dx, dy, j};
                if (table.find(value) != table.end()) return true;
                table.insert(value);
            }
        }
        return false;
        
    }
};
