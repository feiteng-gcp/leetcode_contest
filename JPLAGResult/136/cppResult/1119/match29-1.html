class Solution {
public:
    
    bool isRobotBounded(string instructions) {
        int d = 0;
        int x = 0;
        int y = 0;
        
        int dx[] = {+1,0,-1,0};
        int dy[] = {0,+1,0,-1};
        
        while(1) {
            for (char c : instructions) {

                if (c=='L') {
                    d = (d + 3)%4;
                } else if (c=='R') {
                    d = (d + 1)%4;
                } else {
                    x += dx[d];
                    y += dy[d];
                }
            }
            if (d==0) break;
        }
        return (x==0 && y==0);
    }
};
