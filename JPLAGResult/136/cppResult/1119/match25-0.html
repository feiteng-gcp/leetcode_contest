class Solution {
public:
    bool isRobotBounded(string instructions) {
        int locx = 0,locy = 0,face1 = 1,face2 = 0;
        
        while (1) {
        	for (int i=0;i<instructions.length();i++) {
        		if (instructions[i]=='G') {
        			locx += face1;
        			locy += face2;
				} else if (instructions[i]=='L') {
					if (face1==-1) {
						face1 = 0;
						face2 = -1;
					} else if (face1==1) {
						face1 = 0;
						face2 = 1;
					} else if (face2==1) {
						face1 = -1;
						face2 = 0;
					} else {
						face1 = 1;
						face2 = 0;
					}
				} else {
					if (face1==-1) {
						face1 = 0;
						face2 = 1;
					} else if (face1==1) {
						face1 = 0;
						face2 = -1;
					} else if (face2==1) {
						face1 = 1;
						face2 = 0;
					} else {
						face1 = -1;
						face2 = 0;
					}
				}
			}
			if (face1==1&&face2==0) break;
		}
		if (locx==0&&locy==0) return true;
		else return false;
    }
};
