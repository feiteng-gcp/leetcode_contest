class Solution {

private:
    const int d[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

public:
    bool isRobotBounded(string instructions) {

        string s = "";
        for(int i = 0; i < 4; i ++) s += instructions;
        int curd = 0;

//        set<pair<int, int>> visited;
        pair<int, int> cur = make_pair(0, 0);
//        visited.insert(cur);

        for(char c: s){
            if(c == 'G'){
                cur.first += d[curd][0];
                cur.second += d[curd][1];
            }
            else if(c == 'L')
                curd = (curd + 3) % 4;
            else
                curd = (curd + 1) % 4;
        }
        return !cur.first && !cur.second;
    }
};
