class Solution {
public:
    bool isRobotBounded(string instructions) {
        int x = 0;
        int y = 0;
        int xx = 1;
        int yy = 0;
        for (int i = 0; i < 4; ++i) {
            for (char ins : instructions) {
                if (ins == 'G') {
                    x += xx;
                    y += yy;
                } else if (ins == 'L') {
                    if (xx == 1) {
                        xx = 0;
                        yy = 1;
                    } else if (yy == 1) {
                        xx = -1;
                        yy = 0;
                    } else if (xx == -1) {
                        xx = 0;
                        yy = -1;
                    } else {
                        xx = 1;
                        yy = 0;
                    }
                } else {
                    if (xx == 1) {
                        xx = 0;
                        yy = -1;
                    } else if (yy == -1) {
                        xx = -1;
                        yy = 0;
                    } else if (xx == -1) {
                        xx = 0;
                        yy = 1;
                    } else {
                        xx = 1;
                        yy = 0;
                    }
                }
            }
        }
        return (x == 0 && y == 0);
    }
};
