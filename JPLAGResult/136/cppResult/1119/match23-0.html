class Solution {
public:
    bool isRobotBounded(string instructions) {
        int dx[4] = {-1, 0, 1, 0};
        int dy[4] = {0, -1, 0, 1};
        int offset_x = 0, offset_y = 0, offset_d = 0;
        int x = 0, y = 0, d = 0;
        for (const auto& op : instructions) {
            if (op == 'G') {
                x += dx[d];
                y += dy[d];
            } else if (op == 'L') {
                d = (d + 1) % 4;
            } else {
                d = (d + 3) % 4;
            }
        }
        if (d == 0) {
            return (x == 0 && y == 0);
        } 
        return true;
    }
};
