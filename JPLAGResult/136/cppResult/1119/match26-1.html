class Solution {
public:
    bool isRobotBounded(string instructions) {
        vector<vector<int>> dirs = {{0,1}, {-1,0},{0,-1},{1,0}};
        int n=instructions.size(), i, ind_dirs=0;
        pair<int, int> pos=make_pair(0,0);
        vector<int> dir=dirs[0];

        for (i=0; i<n; ++i) {
            if (instructions[i]=='G') {
                pos.first += dirs[ind_dirs][0];
                pos.second += dirs[ind_dirs][1];
            }
            else if (instructions[i]=='L') {
                ind_dirs = (ind_dirs+3)%4;
            }
            else {
                ind_dirs = (ind_dirs+1)%4;
            }
        }
        if ((pos.first!=0 || pos.second!=0) && ind_dirs==0) return false;
        return true;
    }
};
