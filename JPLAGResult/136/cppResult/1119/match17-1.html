class Solution {
public:
    bool isRobotBounded(string ins) {
        int x = 0, y = 0;
        int dir_idx = 0;

        vector<pair<int, int>> dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

        for (int i = 0; i < 16; ++i) {
            for (char c : ins) {
                if (c == 'G') {
                    x += dir[dir_idx].first;
                    y += dir[dir_idx].second;
                } else if (c == 'R') {
                    dir_idx = (dir_idx + 1) % 4;
                } else {
                    dir_idx = (dir_idx + 3) % 4;
                }
            }
        }
        return x == 0 && y == 0 && dir_idx == 0;
    }
};
