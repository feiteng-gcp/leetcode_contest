class Solution {
public:
    bool isRobotBounded(string instructions) {
        int x = 0, y = 0, dir = 0;
        static int moves[4][2] = {
            { -1, 0 },
            { 0, 1 },
            { 1, 0 },
            { 0, -1 },
        };
        for (char c : instructions) {
            if (c == 'L') {
                dir = (dir + 3) & 3;
            } else if (c == 'R') {
                dir = (dir + 1) & 3;
            } else {
                x += moves[dir][0];
                y += moves[dir][1];
            }
        }
        return dir != 0 || (x == 0 && y == 0);
    }
};
