class Solution {
public:
    bool isRobotBounded(string instructions) {
        vector<int> xdirs = {0, 1, 0, -1};
        vector<int> ydirs = {1, 0, -1, 0};
        int i = 0;

        int x = 0, y = 0;

        for (int t = 0; t < 4; t++) {
            for (char c: instructions) {
                if (c == 'G') {
                    x += xdirs[i];
                    y += ydirs[i];
                }
                else if (c == 'L') {
                    i = (i + 3) % 4;
                }
                else if (c == 'R') {
                    i = (i + 1) % 4;
                }
            }
        }

        return x == 0 && y == 0;
    }
};
