vector<pair<int, int>> DIRS = {
    {0, -1}, {1, 0}, {0, 1}, {-1, 0}
};

class Solution {
public:
    bool isRobotBounded(string instructions) {
        int dir = 0;
        int xx = 0, yy = 0;
        
        for (int i = 0; i < 4; ++i) {
            for (char ch : instructions) {
                if (ch == 'G') {
                    xx += DIRS[dir].first;
                    yy += DIRS[dir].second;
                } else if (ch == 'L') {
                    dir = (dir + 3) % 4;
                } else if (ch == 'R') {
                    dir = (dir + 1) % 4;
                }
            }
        }
        
        return xx == 0 && yy == 0;
    }
};
