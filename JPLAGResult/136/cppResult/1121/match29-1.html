class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& A, int K) {
        vector<int> dp(A.size()+1, 0); // max of idx 0 ~ i
        // for(int i = 0; i < min((int)A.size(), K); i++) {
        for(int i = 0; i < A.size(); i++) {
            int M = -1;
            int beg = max(0, i-K+1);

            for(int j = i; j >= beg; j--) {
                M = max(M, A[j]);
                dp[i+1] = max(dp[i+1], dp[j] + M*(i-j+1));
            }
            /*
            for(int j = beg; j <= i; j++)
                M = max(M, A[j]);
            
            for(int j = beg; j <= i; j++) {
                dp[i+1] = max(dp[i+1], dp[j] + M*(i-))
            }
            dp[i+1] = max(dp[beg] + M * (i-beg+1), dp[i]+A[i]);*/
            // cout << "dp(" << i+1 << ") = " << dp[i+1] << endl;
        }
        return dp.back();
    }
    
};
