class Solution {
public:
  int maxSumAfterPartitioning(vector<int>& A, int K) {
    int N = A.size();
    vector<int> dp(N + 1);
    for (int i = 1; i <= N; i++) {
      int m = 0;
      for (int j = 1; j <= K; j++) { 
        if (i - 1 - j + 1 < 0) continue;
        m = max(m, A[i - 1 - j + 1]);
        dp[i] = max(dp[i], dp[i - j] + m * j);
      }
    }
    return dp.back();
  }
};
