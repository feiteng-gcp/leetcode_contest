class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& A, int K) {
        int i, j, k, n, x;
        n = A.size();
        vector<int> sum(n + 1, 0);
        for (j = 0; j < n; j++) {
            x = A[j];
            sum[j +1] = sum[j] + A[j];
            for (i = j - 1; i >= 0 && j - i < K; i--) {
                x = max(x, A[i]);
                sum[j + 1] = max(sum[j + 1], sum[i] + (j - i + 1) * x);
            }
            //cout << sum[j + 1] << endl;
        }
        return sum[n];
    }
};
