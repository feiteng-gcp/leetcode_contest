class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& A, int K) {
        int n = A.size();
        vector<int> dp(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            int maxVal = A[i - 1];
            int to = max(0, i - K);
            dp[i] = maxVal + dp[i - 1];
            for (int j = i - 1; j >= to; --j) {
                maxVal = max(maxVal, A[j]);
                dp[i] = max(dp[i], maxVal * (i - j) + dp[j]);
            }
        }
        return dp.back();
    }
};
