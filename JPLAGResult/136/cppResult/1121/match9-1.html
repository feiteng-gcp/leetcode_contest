class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& A, int K) {
        int n = A.size();
        vector<int> dp(n + 1);
        for (int i = 0; i < n; i++) {
            dp[i + 1] = dp[i] + A[i];

            int MAX = A[i];
            for (int j = i - 1; j >= 0 && i - j < K; j--) {
                MAX = max(A[j], MAX);
                dp[i + 1] = max(dp[i + 1], dp[j] + MAX * (i - j + 1));
            }
        }
        return dp[n];
    }
};
