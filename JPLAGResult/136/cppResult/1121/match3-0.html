class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& A, int K) {
        int n=A.size();
        vector<int>dp(n+1,0);
        for (int i=0;i<n;i++) {
            int mx=-1000000;
            for (int j=i;j>=max(0,i-K+1);j--) {
                mx=max(A[j],mx);
                dp[i+1]=max(dp[i+1], dp[j]+mx*(i-j+1));
            }
        }
        return dp[n];
    }
};
