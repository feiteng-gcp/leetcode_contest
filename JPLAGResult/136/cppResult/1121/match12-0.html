class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& A, int K) {
        int n=A.size();
        vector<int> dp(n+1,0);
        dp[0]=0;
        for(int i=1;i<=n;i++)
        { 
            int curmax=0;
            for(int j=1;j<=K && j<=i;j++)
            {
                curmax=max(curmax,A[i-j]);
                dp[i]=max(dp[i],dp[i-j]+curmax*j);
            }
    //        cout<<dp[i]<<endl;
        }
        return dp[n];
    }
};
