class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& A, int K) {
        int n = A.size();
        vector<int> dp(n, 0);
        for (int i = 0; i < n; ++i) {
            if (i < K) {
                dp[i] = (i + 1) * (*max_element(A.begin(), A.begin() + i + 1));
            } else {
                for (int j = 1; j <= K; ++j) {
                    dp[i] = max(dp[i], dp[i - j] + j * (*max_element(A.begin() + i - j + 1, A.begin() + i + 1)));
                }
            }
        }
        return dp[n - 1];
    }
};
