class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& a, int K) {
        int n = a.size();
        vector<int> f(n + 1);
        for (int i = 1; i <= n; i++) {
            int mx = 0;
            for (int j = i; j >= 1 && j >= i - K + 1; j--) {
                mx = max(mx, a[j - 1]);
                f[i] = max(f[i], f[j - 1] + mx * (i - j + 1));
            }
        }
        return f[n];
    }
};
