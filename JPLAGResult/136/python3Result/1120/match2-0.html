    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        adjMat = collections.defaultdict(list)
        for p in paths:
            adjMat[p[0]].append(p[1])
            adjMat[p[1]].append(p[0])
        print(adjMat)
        ret = [1] * N
        for n in range(0, N):
            freeColor = [True, True, True, True]
            for neigh in adjMat[n+1]:
                if (neigh-1) < n:
                    freeColor[ret[neigh-1]-1] = False
            for cIdx, c in enumerate(freeColor):
                if c==True:
                    ret[n] = cIdx+1
                    break
        return ret
