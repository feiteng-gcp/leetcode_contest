class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        edge = collections.defaultdict(list)
        for p in paths:
            edge[p[0]].append(p[1])
            edge[p[1]].append(p[0])
        ans = []
        for i in range(1, N+1):
            cand = [1,1,1,1]
            for n in edge[i]:
                if n > i:
                    continue
                cand[ans[n-1]-1] = 0
            fl = 4
            for i, v in enumerate(cand):
                if v == 1:
                    fl = i+1
                    break
            ans.append(fl)
        return ans
