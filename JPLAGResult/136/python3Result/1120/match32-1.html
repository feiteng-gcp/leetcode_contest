class Solution:
    def gardenNoAdj(self, N, paths):
        edges 
= collections.defaultdict(list)
        for u,v in paths:
            edges[u-1].append(v-1)
            edges[v-1].append(u-1)
        res = [-1] * N
        for i in range(N):
            s = {1,2,3,4}
            for j in edges[i]:
                s -= {res[j],}
            res[i] = s.pop()
        return res
