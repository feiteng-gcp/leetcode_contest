class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        graph=[[] for _ in range(N+1)]
        for x,y in paths:
            graph[x].append(y)
            graph[y].append(x)
        color=[0]*(N+1)#color all N gardons.
        used=[False]*5
        def dfs(x):
            candi_col=[]
            for col in range(1,5):
                if all(color[y]!=col for y in graph[x]):
                    candi_col.append(col)
            for col in candi_col:#choose used first.
                if used[col]:
                    color[x]=col
            if color[x]==0:#uncolored
                color[x]=candi_col[0]
                used[candi_col[0]]
            for y in graph[x]:
                if color[y]==0:
                    dfs(y)
        for i in range(1,N+1):
            dfs(i)
        return color[1:]
