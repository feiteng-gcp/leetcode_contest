class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        ans = [0] * N
        graph = [set() for _ in range(N)]
        for path in paths:
            x = path[0] - 1
            y = path[1] - 1
            graph
[x].add(y)
            graph[y].add(x)
        for i in range(N):
            flags = [True] * 4
            for child in graph[i]:
                if child < i:
                    flags[ans[child] - 1] = False
            j = 0
            while not flags[j]:
                j += 1
            ans[i] = j + 1
        return ans
