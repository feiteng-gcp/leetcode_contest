class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        con = [[] for _ in range(N)]
        for x in paths:
            con[x[0]-1].append(x[1]-1)
            con[x[1]-1].append(x[0]-1)
        
        ans = [0]*N
        
        for i in range(N):
            for f in range(1, 5):
                tmp = True
                for b in con[i]:
                    if f == ans[b]:
                        tmp = False
                        break
                if tmp:
                    ans[i] = f
                    break
                    
        return ans
