class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        edges = [[] for _ in range(N)]
        ans = [0] * N
        for p in paths:
            x, y = p
            edges[x-1].append(y-1)
            edges[y-1].append(x-1)
        for i in range(N):
            exist = []
            for j in edges[i]:
                if ans[j] != 0:
                    exist.append(ans[j])
            for t in range(1, 5):
                if t not in exist:
                    ans[i] = t
                    break
        return ans
