class Solution:
    def dfs(self,mark,color,over,graph,vertex):
        if(mark[vertex]==False):
            mark[vertex] = True
            if((over[vertex] & 1) == 0):
                c = 1
            elif((over[vertex] & 2)==0):
                c = 2
            elif((over[vertex] & 4)==0):
                c = 3
            else:
                c = 4
            color[vertex]=c
            for y in graph[vertex]:
                over[y]|= 1<<(c-1)
            for y in graph[vertex]:
                self.dfs(mark,color,over,graph,y)
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        color = [0]*N
        mark = [False]*N
        over = [0]*N
        graph = [[] for _ in range(N)]
        for edge in paths:
            graph[edge[0]-1].append(edge[1]-1)
            graph[edge[1]-1].append(edge[0]-1)
        for i in range(N):
            if(mark[i]==False):
                self.dfs(mark,color,over,graph,i)
        return color
