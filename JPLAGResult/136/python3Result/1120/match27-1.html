import collections

class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        neis = collections.defaultdict(list)
        for x, y in paths:
            neis[x].append(y)
            neis[y].append(x)
        res = [1] * N
        for cur in neis:
            for n in neis[cur]:
                if res[cur - 1] == res[n - 1]:
                    for i in range(1, 5):
                        res[n - 1] = i
                        if self.check(n, res, neis):
                            break
        return res
        
    def check(self, x, res, neis):
        for n in neis[x]:
            if res[n - 1] == res[x - 1]:
                return False
        return True
