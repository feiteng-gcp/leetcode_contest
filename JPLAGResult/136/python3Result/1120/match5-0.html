class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        p_dict = collections.defaultdict(list)
        for path in paths:
            x = path[0]
            y = path[1]
            p_dict[x].append(y)
            p_dict[y].append(x)
            
        color_dict = [0] * (N+1)
        
        def isSafe(x, c): 
            for y in p_dict[x]: 
                if color_dict[y] == c: 
                    return False
            return True
        
        def graphColourUtil(x):
            if x == N+1:
                return True
            for c in range(1, 5): 
                if isSafe(x, c) == True: 
                    color_dict[x] = c 
                    if graphColourUtil(x+1) == True: 
                        return True
                    color_dict[x] = 0
                    
        Result = graphColourUtil(1)
        return color_dict[1:] if Result else Result
