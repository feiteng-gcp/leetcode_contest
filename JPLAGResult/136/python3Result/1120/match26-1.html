class Solution:
    def gardenNoAdj(self, N: int, paths):
        d 
= collections.defaultdict(set)
        for a, b in paths:
            d[a].add(b)
            d[b].add(a)
        
        r = [0]*N
        for color in range(1, 5):
            for i in range(N):
                if r[i]: continue
                if any(r[j-1] == color for j in d[i+1]): continue
                r[i] = color
        return r
