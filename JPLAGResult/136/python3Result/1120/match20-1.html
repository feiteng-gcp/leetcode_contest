from collections import defaultdict

class Solution:
    def gardenNoAdj(self, N, paths):
        for i in range(len(paths)):
            paths[i][0] -= 1
            paths[i][1] -= 1
        
        neighbors = defaultdict
(list)
        for p in paths:
            neighbors[p[0]].append(p[1])
            neighbors[p[1]].append(p[0])
        
        flower = [None]*N
        for i in range(N):
            neighbor_flowers = set(flower[j] for j in neighbors[i])
            for j in range(1, 5):
                if j not in neighbor_flowers:
                    flower[i] = j
                    break
                    
        return flower
