class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        nei = collections.defaultdict(set)
        for a, b in paths:
            nei[a].add(b)
            nei[b].add(a)
        color = {}
        res = []
        def dfs(n):
            if n in color.keys():
                return
            neighbors = nei[n]
            temp = set()
            for neighbor in neighbors:
                if neighbor in color.keys():
                    temp.add(color[neighbor])
            for c in range(1, N + 1):
                if c not in temp:
                    color[n] = c
                    break
            for neighbor in neighbors:
                dfs(neighbor)
        for i in range(1, N + 1):
            dfs(i)
        for i in range(1, N + 1):
            if i in color.keys():
                res.append(color[i])
            else:
                res.append(1)
        return res
