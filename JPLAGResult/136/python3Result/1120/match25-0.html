class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        color = {i: 0 for i in range(1, N+1)}
        nb = collections.defaultdict(set)
        for i, j in paths:
            nb[i].add(j)
            nb[j].add(i)
        res = []
        for i in range(1, N+1):
            temp = {color[j] for j in nb[i]}
            for c in range(1, 5):
                if c not in temp:
                    color[i] = c
                    res.append(c)
                    break
        return res
