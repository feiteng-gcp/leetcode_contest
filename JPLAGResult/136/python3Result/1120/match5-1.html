class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        gardens=[0]*N
        d=collections.defaultdict(list)
        for p in paths:
            i,j=p
            d[i-1].append(j-1)
            d[j-1].append(i-1)
        #print(d)
        
def isValid(gardens, color, i):
            for n in d[i]:
                if gardens[n]==color:
                    return False
            return True
                
        def helper(gardens, paths, i):
            if i==N:
                return gardens
            #print(gardens, i)
            for color in range(1,5):
                if isValid(gardens, color, i):
                    gardens[i]=color
                    if helper(gardens, paths, i+1):
                        return gardens
                    gardens[i]=0
            #return False
        helper(gardens,paths,0)
        return gardens
