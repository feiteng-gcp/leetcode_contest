class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        color = [0] * (N + 1)
        all_colors = {
1, 2, 3, 4}
        path = collections.defaultdict(list)
        # print(path[0])
        for p in paths:
            path[p[0]].append(p[1])
            path[p[1]].append(p[0])

        need = set(range(1, N + 1))

        while len(need) > 0:
            queue = [need.pop()]
            while len(queue) > 0:
                node = queue.pop(0)
                if color[node]:
                    continue
                available = set(all_colors)
                for n in path[node]:
                    if color[n]:
                        available.discard(color[n])
                    else:
                        need.discard(n)
                        queue.append(n)
                color[node] = available.pop()

        return color[1:]
