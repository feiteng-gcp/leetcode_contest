class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        g = collections.defaultdict(list)
        for path in paths:
            g[path[0]].append(path[1])
            g[path[1]].append(path[0])
        res = [-1] * (N) # need to make it 1 based
        mp = dict()
        def dfs(idx):
            if idx in mp:
                return
            color_used = set([mp[child] for child in g[idx] if child in mp])
            for color in range(1,5):
                if color not in color_used:
                    mp[idx] = color
                    res[idx - 1] = color
                    break
        for i in range(1, N+1):
            dfs(i)
        return res
