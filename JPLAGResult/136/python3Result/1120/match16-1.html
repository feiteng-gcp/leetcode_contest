class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        g = [set() for _ in range
(N + 1)]
        for path in paths:
            g[path[0]].add(path[1])
            g[path[1]].add(path[0])
        ans = []
        for node in range(1, N + 1):
            p = set([1,2,3,4])
            for r in g[node]:
                if r < node:
                    p.discard(ans[r - 1])
            ans.append(min(list(p)))
        return ans
