class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        t = {i: [] for i in range(1, N+1)}
        for i in paths:
            t[i[0]].append(i[1])
            t[i[1]].append(i[0])
        ans = [0] * N
        tt = (i for i in range(1, N+1))
        flag = 1
        while tt:
            ttt = set()
            for i in tt:
                if i not in ttt:
                    ans[i-1] = flag
                    ttt |= set([j for j in t[i] if ans[j-1] == 0])
            flag += 1
            tt = ttt
        return ans
