from collections import defaultdict


class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        
        
        graph = defaultdict(set)
        for v,u in paths:
            graph[u-1].add(v-1)
            graph[v-1].add(u-1)
        color = [0]*N
        for node in range(len(color)):
            available = set(range(1,5))
            for nxt in graph[node]:
                if color[nxt] in available:
                    available.remove(color[nxt])
            color[node] = available.pop()
        return color
