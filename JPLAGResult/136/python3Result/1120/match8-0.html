class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        d = collections.defaultdict(list)

        for a, b in paths:
            d[a].append(b)
            d[b].append(a)
        
        res = []
        # print(d)
        for g in range(1, N + 1):
            if g not in d:
                res.append(1)
                continue
            
            values = set([1,2,3,4])
            for neighbor in d[g]:
                if neighbor <= len(res):
                    if res[neighbor - 1] in values:
                        values.remove(res[neighbor - 1])
            # print(g, values)
            res.append(values.pop())
        return res
