from collections import deque, defaultdict


class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        adj = [[] for i in range(N)]
        for path in paths:
            adj[path[0]-1].append(path[1]-1)
            adj[path[1]-1].append(path[0]-1)
        color = [-1 for i in range(N)]
        visit = [False for i in range(N)]
        ban = [defaultdict(bool) for i in range(N)]
        for i in range(N):
            if visit[i]:
                continue
            q = deque()
            q.append(i)
            while len(q) > 0:
                node = q.popleft()
                for c in range(1, 5):
                    if not ban[node][c]:
                        color[node] = c
                        break
                for next_node in adj[node]:
                    ban[next_node][color[node]] = True
                    if visit[next_node]:
                        continue
                    visit[next_node] = True
                    q.append(next_node)
        return color
