import collections

class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        out_degrees = collections.defaultdict(set)
        for path in paths:
            out_degrees[path[0]].add(path[1])
            out_degrees[path[1]].add(path[0])
        colors = [0] * (N+1)
        visited = set()
        self.a = []

        
        def helper(node, prev, level):

            for color in range(1, 5):
                valid = True
                for neighbor in out_degrees[node]:
                    if colors[neighbor] == color:
                        valid = False
                if not valid:
                    continue
                
                
                colors[node] = color
                if level == N:
                    return True
                for neighbor in out_degrees[node]:
                    if neighbor not in visited:
                        visited.add(neighbor)
                        r = helper(neighbor, node, level+1)
                        if r:
                            return r

            
            
        
        for i in range(1, N+1):
            if colors[i] == 0:
                helper(i, 0, 0)
        return colors[1:]
