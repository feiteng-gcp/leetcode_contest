from collections import defaultdict


class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        graph = defaultdict(list)
        for u, v in paths:
            graph[u].append(v)
            graph[v].append(u)
            
        res = [0 for _ in range(N+1)]
        
        def dfs(u):
            nonlocal res
            # print(u)
            if res[u]:
                return
            used = set()
            for neb in graph[u]:
                used.add(res[neb])
            for i in [1, 2, 3, 4]:
                if i not in used:
                    res[u] = i
            for neb in graph[u]:
                dfs(neb)
        for i in range(N+1):
            if res[i] == 0:
                dfs(i)
        return res[1:]
