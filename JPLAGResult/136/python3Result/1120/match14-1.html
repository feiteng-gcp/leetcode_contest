class Solution:
    def gardenNoAdj(self, N: int, paths: List[List[int]]) -> List[int]:
        fType 
= [0] * N
        adjList = [[] for _ in range(N)]
        for p in paths:
            adjList[p[0] - 1].append(p[1] - 1)
            adjList[p[1] - 1].append(p[0] - 1)

        for n in range(N):
            if fType[n] == 0:
                self.colorNode(adjList, n, fType)

        return fType

    def colorNode(self, adjList, n, fType):
        available = [True] * 4
        for nei in adjList[n]:
            if fType[nei] != 0:
                available[fType[nei] - 1] = False

        for t in range(4):
            if available[t]:
                fType[n] = t + 1
