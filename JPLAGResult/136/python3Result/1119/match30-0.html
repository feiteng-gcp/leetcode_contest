class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        x = 0
        y = 0
        d = 0
        for k in range(4):
            for i in instructions:
                if i=='G':
                    if d==0:
                        y+=1
                    if d==1:
                        x+=1
                    if d==2:
                        y-=1
                    if d==3:
                        x-=1
                elif i=='L':
                    d = (d-1)%4
                elif i=='R':
                    d = (d+1)%4
        if x==0 and y==0:
            return True
        return False
