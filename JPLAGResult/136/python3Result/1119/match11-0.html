class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        directions = {'north': {'L': 'west', 'R': 'east'}, 'south':{'L':'east', 'R':'west'}, 'west':{'L':'south', 'R':'north'}, 'east':{'L':'north', 'R':'south'}}
        direction = 'north'
        location = [0, 0]
        for ch in instructions:
            if ch == 'G':
                if direction == 'north':
                    location[1] += 1
                elif direction == 'south':
                    location[1] -= 1
                elif direction == 'east':
                    location[0] += 1
                else:
                    location[0] -= 1
            elif ch == 'L':
                direction = directions[direction]['L']
            else:
                direction = directions[direction]['R']
        if direction == 'north' and location != [0, 0]:
            return False
        return True
