import collections

class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        c = collections.Counter(instructions)
        if (c['L'] - c['R']) % 4 != 0:
            return True
        dire = 0
        real_dir = ((0,1), (-1,0),(0,-1),(1,0))
        curr = (0,0)
        
for x in instructions:
            if x == 'L':
                dire = (dire+1)%4
            elif x == 'R':
                dire = (dire-1)%4
            else:
                curr = (curr[0]+real_dir[dire][0], curr[1]+real_dir[dire][1])
        if curr[0] or curr[1]:
            return False
        return True
