class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        t = [0, 0]
        flag = 0
        for i in instructions:
            if i == 'L':
                flag = (flag + 1) % 4
            elif i == 'R':
                flag = (flag - 1) % 4
            else:
                if flag == 0:
                    t[1] += 1
                elif flag == 1:
                    t[0] -= 1
                elif flag == 2:
                    t[1] -= 1
                else:
                    t[0] += 1
        if flag == 0 and t != [0, 0]:
            return False
        return True
