from collections import Counter
class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        location = [0,0]
        d_count = 0
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        for i in instructions:
            if i == 'G':
                location[0] += directions[d_count][0]
                location[1] += directions[d_count][1]
            elif i == 'R':
                d_count += 1
            else:
                d_count -= 1
            d_count %= 4
        
        return location == [0,0] or d_count != 0
