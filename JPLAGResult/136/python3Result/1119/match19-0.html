class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        coord = [0,0]
        face = [[0,1], [1,0], [0,-1], [-1,0]]
        fId = 0
        for i in range(0,4):
            for s in instructions:
                if s=='G':
                    coord[0] += face[fId][0]
                    coord[1] += face[fId][1]
                elif s=='L':
                    fId = (fId - 1) % 4
                elif s=='R':
                    fId = (fId + 1) % 4
            if coord[0]==0 and coord[1]==0 and fId==0:
                return True
        return False
