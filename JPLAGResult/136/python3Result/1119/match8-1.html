class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        direction = 0
        pos = [0, 0]
        
for c in instructions:
            if c == "L":
                direction -= 1
            elif c == "R":
                direction += 1
            else:
                if direction % 4 == 0:
                    pos[1] += 1
                elif direction % 4 == 1:
                    pos[0] += 1
                elif direction % 4 == 2:
                    pos[1] -= 1
                elif direction % 4 == 3:
                    pos[0] -= 1
        
        return (direction % 4) != 0 or pos == [0, 0]
