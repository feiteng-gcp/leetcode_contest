class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        instructions = instructions * 4
        x, y = 0, 0
        d = [(0, 1), (-1, 0), (0, -1), (1, 0)]
        di = 0
        for ch in instructions:
            
if ch == 'G':
                x, y = x+d[di][0], y+d[di][1]
            elif ch == 'L':
                di = (di+1) % 4
            else:
                di = (di-1) % 4
        return x == 0 and y == 0
