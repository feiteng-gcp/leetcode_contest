class Solution:
    def isRobotBounded(self, s: str) -> bool:
        pos = (0, 0)
        start = (0, 0)
        dir = [[0, 1], [-1, 0], [0, -1], [1, 0]]
        cur = 0
        for each in s*4:
            if each == 'L':
                cur = (cur + 1) % 4
            elif each == 'R':
                cur = (cur - 1 + 4) % 4
            else:
                pos = (pos[0] + dir[cur][0], pos[1] + dir[cur][1])
        if pos == start:
            return True
        return False
