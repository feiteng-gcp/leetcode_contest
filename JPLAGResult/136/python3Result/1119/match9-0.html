class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        cur = [0, 0]
        direction = 0
        for i in range(0, 4):
            for ins in instructions:
                if ins == 'G':
                    if direction == 0:
                        cur[0] += 1
                    elif direction == 1:
                        cur[1] += 1
                    elif direction == 2:
                        cur[0] -= 1
                    elif direction == 3:
                        cur[1] -= 1
                    else:
                        raise Exception('WTF???')
                else:
                    if ins == 'L':
                        direction = (direction - 1) % 4
                    elif ins == 'R':
                        direction = (direction + 1) % 4
                    else:
                        raise Exception('WTF???')
            if cur == [0, 0]:
                return True
        return False
