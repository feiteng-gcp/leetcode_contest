class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        dire = 0
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        pos = (0,0)
        for _ in range(4):
            for inst in instructions:
                if inst == 'R':
                    dire = (dire + 1) % 4
                elif inst == 'L':
                    dire = (4 + dire - 1) % 4
                elif inst == 'G':
                    pos = (pos[0] + directions[dire][0], pos[1] + directions[dire][1])
        return pos == (0,0)
