class Solution:
    def isRobotBounded(self, ins: str) -> bool:
        dirx=0
        diry=1
        currx=curry=0
        for s in ins:
            if s=="L":
                dirx,diry=-diry,dirx
            elif s=="R":
                dirx,diry=diry,-dirx
            elif s=="G":
                currx+=dirx
                curry+=diry
        
        return not(dirx==0 and diry>0) or currx==curry==0
