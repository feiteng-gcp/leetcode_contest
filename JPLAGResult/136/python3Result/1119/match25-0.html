class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]]
        curpoint = [0, 0]
        curdir = 0
        newins = "".join([instructions] * 4)
        for c in newins:
            if c == 'G':
                curpoint[0] += dirs[curdir][0]
                curpoint[1] += dirs[curdir][1]
            elif c == 'L':
                curdir = (curdir - 1) % 4
            elif c =='R':
                curdir = (curdir + 1) % 4
        return curpoint == [0, 0]
