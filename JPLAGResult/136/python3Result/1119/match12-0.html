class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        direction = 0 # N
        pos = [0, 0]
        for ins in instructions * 4:
            if ins == "G":
                if direction == 0:
                    pos[1] += 1
                elif direction == 1: # E
                    pos[0] += 1
                elif direction == 2: # S
                    pos[1] -= 1
                elif direction == 3: # W
                    pos[0] -= 1
            elif ins == "L":
                direction = (direction - 1) % 4
            elif ins == "R":
                direction = (direction + 1) % 4 
        return pos == [0, 0]
