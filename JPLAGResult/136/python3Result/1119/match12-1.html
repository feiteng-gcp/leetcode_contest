class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        pos = [0, 0]
        d = 0
        for index, instruction in enumerate(instructions):
            
if instruction == 'G':
                if d == 0:
                    pos[1] += 1
                elif d == 2:
                    pos[1] -= 1
                elif d == 1:
                    pos[0] += 1
                elif d == 3:
                    pos[0] -= 1
            
            elif instruction == 'L':
                d -= 1
            elif instruction == 'R':
                d += 1
                
            if d < 0:
                d = 3
            elif d > 3:
                d = 0
                
        #print(pos, d)
        
        if d == 0:
            if pos[0] != 0 or pos[1] != 0:
                return False
            else:
                return True
        else:
            return True
