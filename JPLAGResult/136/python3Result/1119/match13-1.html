class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        d = 0
        move = [[0, 1], [-1, 0], [0, -1], [1, 0]]
        pos = [0, 0]
        for j in range(4):
            for i in instructions:
                if i == 'L':
                    d = (d + 1) % 4
                elif i == 'R':
                    d = (d - 1) % 4
                else:
                    pos[0] += move[d][0]
                    pos[1] += move[d][1]
        if pos == [0, 0]:
            return True
        return False
