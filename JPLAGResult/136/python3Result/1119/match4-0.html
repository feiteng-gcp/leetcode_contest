class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        cur = 0
        x = 0
        y = 0
        for i in instructions:
            if i == 'L':
                cur -= 1
            elif i == 'R':
                cur += 1
            else:
                d = cur % 4
                if d == 0:
                    y += 1
                elif d == 1:
                    x += 1
                elif d == 2:
                    y -= 1
                else:
                    x -= 1
        return cur % 4 != 0 or (x == 0 and y == 0)
