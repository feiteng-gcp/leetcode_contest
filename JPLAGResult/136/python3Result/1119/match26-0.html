class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        dirInd = 0
        dirs = [(0,1),(1,0),(0,-1),(-1,0)]
        visited = set([(0,0)])

        s = [0, 0]
        for _ in range(4):
            for ins in instructions:
                if ins == 'G':
                    s[0] += dirs[dirInd][0]
                    s[1] += dirs[dirInd][1]
                elif ins == 'L':
                    dirInd = (dirInd + 1) % 4
                elif ins == 'R':
                    dirInd = (dirInd - 1) % 4
            if tuple(s) in visited:
                return True
        return False
