class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        pos = [0, 0]
        d = 0
        for i in range(4):
            for instruction in instructions:
                if instruction == 'G':
                    if d == 0:
                        pos[1] += 1
                    elif d == 1:
                        pos[0] += 1
                    elif d == 2:
                        pos[1] -= 1
                    else:
                        pos[0] -= 1
                elif instruction == 'L':
                    d = d - 1 + 4
                    d = d % 4
                elif instruction:
                    d += 1
                    d = d % 4
            
        if pos == [0, 0]:
            return True
        return False
