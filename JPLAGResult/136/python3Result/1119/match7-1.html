class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        NOW=[0,0,0]

        for ins in instructions*4:
            if ins=="R":
                NOW[2]=(NOW[2]+1)%4
            elif ins=="L":
                NOW[2]=(NOW[2]-1)%4

            else:
                
if NOW[2]==0:
                    NOW[0]+=1
                elif NOW[2]==2:
                    NOW[0]-=1
                elif NOW[2]==1:
                    NOW[1]+=1
                else:
                    NOW[1]-=1

        if NOW[0]==0 and NOW[1]==0:
            return True
        else:
            return False
