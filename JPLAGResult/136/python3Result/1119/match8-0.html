class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        s = [0, 0]
        p = 0
        # p = [0, 1, 2, 3] #up right down left
        ans = False
        
        for t in range(4):
            for c in instructions:
                if c == "L":
                    p = p-1 if p >= 1 else 3
                elif c == "R":
                    p = p+1 if p <= 2 else 0
                else:
                    if p == 0:
                        s[1] += 1
                    elif p == 1:
                        s[0] += 1
                    elif p == 2:
                        s[1] -= 1
                    else:
                        s[0] -= 1
            if t == 3 and s[0] == 0 and s[1] == 0 and p == 0:
                ans = True
        
        return ans
