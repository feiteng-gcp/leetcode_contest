class Solution:
    def isRobotBounded(self, s: str) -> bool:
        s=s*4
        x,y=0,0
        dxys=[(1,0),(0,1),(-1, 0),(0,-1)]
        i=0
        
        for c in s:
            if c=="G":x,y=x+dxys[i][0],y+dxys[i][1]
            elif c=="L":i=(i+1)%4
            else:i=(i+3)%4
        return x==0 and y==0
