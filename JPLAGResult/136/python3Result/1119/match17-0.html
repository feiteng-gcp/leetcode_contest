class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        
        vecs = {
            0:(1,0),
            1:(0,1),
            2:(-1,0),
            3:(0,-1)
        }
        d= 0
        r,c =0,0
        visited = set()
        for i in instructions*4:
            visited.add((r,c,d))
            if i =='G':
                r,c = r+vecs[d][0],c + vecs[d][1]
            elif i=='R':
                d = (d+1)%4
            elif i=='L':
                d = (d-1)%4
        return r==0 and c==0
