class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        init = 0
        x = 0
        y = 0
        for s in instructions:
            if s == "L":
                init += 1
            elif s == "R":
                init -= 1
            else:
                init = init % 4
                if init == 0:
                    y += 1
                elif init == 1:
                    x -= 1
                elif init == 2:
                    y -= 1
                else:
                    x += 1
                    
        init = init % 4
        
        if x == 0 and y == 0:
            return True
        return True if init != 0 else False
