class Solution:
    def isRobotBounded(self, instructions: str) -> bool:
        pos = [0, 0]
        dirs = ['u', 'r', 'd', 'l']
        curr_dir = 0
        
        for c in instructions:
            if c == 'L':
                curr_dir = (curr_dir - 1) % 4
            elif c == 'R':
                curr_dir = (curr_dir + 1) % 4
            else:
                if dirs[curr_dir] == 'u':
                    pos[1] += 1
                elif dirs[curr_dir] == 'd':
                    pos[1] -= 1
                elif dirs[curr_dir] == 'l':
                    pos[0] -= 1
                else:
                    pos[0] += 1
        
        if pos[0] == 0 and pos[1] == 0:
            return True
        
        else:
            return dirs[curr_dir] != 'u'
