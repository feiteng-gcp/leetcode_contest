class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        dp = [0]*(len(A)+1)
        for i in range(len(A)):
            m = A[i]
            for k in range(K):
                if i-k<0:
                    break
                m = max(m,A[i-k])
                dp[i+1] = max(dp[i+1],dp[i-k]+m*(k+1))
        return dp[-1]
