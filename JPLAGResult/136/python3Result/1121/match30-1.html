class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        dp = [0 for i in A]
        max_so_far = A[0]
        dp[0] = max_so_far
        for i in range(1, K):
            max_so_far = max(max_so_far, A[i])
            dp[i] = max_so_far * (i + 1)
        
        for i in range(K, len(A)):
            prev_end = i - K
            max_so_far = A[i]
            dp[i] = dp[i - 1] + max_so_far
            for j in range(prev_end + 1, i)[::-1]:
                max_so_far 
= max(max_so_far, A[j])
                dp[i] = max(dp[i], dp[j - 1] + max_so_far * (i - j + 1))
        
        # print(dp)
        return dp[-1]
