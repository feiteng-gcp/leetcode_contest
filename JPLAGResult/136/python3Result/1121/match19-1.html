class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        n = len(A)
        dp = [0] * (n+1)
        for i in range(1,n+1):
            minele = A[i-1]
            for l in range(1, K+1):
                j = i - l
                
if ( j < 0): break
                minele = max(minele, A[j])
                dp[i] = max(dp[i], dp[j] + l * minele)
                # if(i==n): print(dp[i], minele)
            # if(i==n): print(dp[i])
        return dp[n]
