class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        dp = [0] * len(A)
        _max = 0
        for i in range(K):
            _max = max(A[i], _max)
            dp[i] = _max * (i + 1)
            
        for i in range(K, len(A)):
            _max = A[i] + dp[i - 1]
            t = A[i]
            for j in range
(1, K):
                t = max(t, A[i - j])
                _max = max(_max, t * (j + 1) + dp[i - j -1])
            dp[i] = _max
        # print(dp)
        return dp[-1]
