class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        a = A
        k = K
        n = len(A)
        dp = [0] * (n + 1)
        #dp[0] = A[0]
        for i in range(1, n + 1):
            dp[i] = dp[i - 1] + a[i- 1]
            cur = a[i - 1]
            for j in range(max
(i - 1, 1), max(i - k, 0), -1):
                cur = max(cur, a[j - 1])
                dp[i] = max(dp[i], dp[j - 1] + cur * (i - j + 1))
        return dp[n]
