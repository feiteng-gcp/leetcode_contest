class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        dp = {}
        def divide(start):
            if start >= len(A): return 0
            if start in dp: return dp[start]
            res = 0
            sub_max = 0
            for i in range(min(K, len(A)-start)):
                sub_max = max(sub_max, A[start+i])
                sub_len = i+1
                res = max(res, sub_max*sub_len+divide(start+i+1))
            dp[start] = res
            return res
        return divide(0)
