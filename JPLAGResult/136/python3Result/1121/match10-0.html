class Solution:
    def maxSumAfterPartitioning(self, A, K) :
        maxv = [0] * len(A)
        for i in range(len(A)):
            cur = 0
            res = 0
            for j in range(i, max(i - K, -1), -1):
                cur = max(cur, A[j])
                res = max(res, cur * (i - j + 1) + (maxv[j-1] if j >= 0 else 0))
                #print(i, j, cur, res)
            maxv[i] = res
        #print(maxv)
        return maxv[-1]
