class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        dp = collections.defaultdict(lambda: 0)
        Max = -1
        for i in range(K):
            Max = max(A[i], Max)
            dp[i] = (i+1) * Max
        for i in range(K, len(A)):
            Max = A[i]
            for j in range(1, K+1):
                Max = max(Max, A[i-j+1])
                dp[i] = max(dp[i], dp[i-j] + j * Max) 
        # for key in sorted(dp):
        #     print(key, dp[key])
        return dp[len(A)-1]
