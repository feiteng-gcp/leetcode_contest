class Solution:
    def maxSumAfterPartitioning(self, A, K):
        memo 
= {}
        def rec(n):
            if n >= len(A):
                return 0
            if n in memo:
                return memo[n]
            
            best = 0
            most = 0
            for k in range(1, K+1):
                if n+k > len(A):
                    break
                most = max(A[n+k-1], most)
                best 
= max(
                    best,
                    k*most + rec(n+k)
                )
            
            memo[n] = best
            return best
        
        return rec(0)
