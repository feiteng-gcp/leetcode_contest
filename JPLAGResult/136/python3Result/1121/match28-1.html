class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        dp = [0] * len(A)
        
        for i, n in enumerate(A):
            if i == 0:
                dp[0] = A[0]
                continue
            cur = 0
            for pre in range(min
(i + 1, K)):
                _max = max(A[i - pre:i+1])
                cur = max(cur, _max * (pre + 1) + dp[i - pre - 1])
            dp[i] = cur
        
        return dp[-1]
