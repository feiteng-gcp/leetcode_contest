class Solution:
    def maxSumAfterPartitioning(self, A: List[int], K: int) -> int:
        dp = [0] * (len(A) + 1)
        for i in range(len(A)):
            temp = 0
            for k in range(0, min(K, i+1)):
                temp 
= max(temp, A[i-k])
                dp[i+1] = max(dp[i+1], temp * (k+1) + dp[i-k])
        return dp[-1]
