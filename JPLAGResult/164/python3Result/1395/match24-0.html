class Solution:
    def minTimeToVisitAllPoints(self, points: List[List[int]]) -> int:
        ans = 0
        for i in range( len(points) - 1 ):
            a = points[i]
            b = points[i+1]
            dx = abs( b[0] - a[0] )
            dy = abs( b[1] - a[1] )
            ans += max(dx, dy)
        return ans
