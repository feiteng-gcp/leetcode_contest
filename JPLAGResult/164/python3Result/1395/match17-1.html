class Solution:
    def minTimeToVisitAllPoints(self, points: List[List[int]]) -> int:
        solution = 0
        for i in range(len(points)-1):
            rise = abs(points[i+1][1]-points[i][1])
            run = abs(points[i+1][0]-points[i][0])
            solution += min(rise, run) + abs(rise-run)
        return solution
