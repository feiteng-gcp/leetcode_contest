class Solution:
    def minTimeToVisitAllPoints(self, points: List[List[int]]) -> int:
        def cost(p1, p2):
            dx, dy = abs(p1[0] - p2[0]), abs(p1[1] - p2[1])
            return max(dx, dy)
        res = 0
        for i in range(1, len(points)):
            res += cost(points[i-1], points[i])
        return res
