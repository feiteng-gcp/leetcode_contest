class Solution:
    def minTimeToVisitAllPoints(self, points: List[List[int]]) -> int:
        res = 0
        for i in range(len(points)-1):
            p1 = points[i]
            p2 = points[i+1]
            dx = abs(p1[0]-p2[0])
            dy = abs(p1[1]-p2[1])
            res += max(dx,dy)
        return res
