class Solution:
    def minTimeToVisitAllPoints(self, points: List[List[int]]) -> int:
        ans = 0
        for i in range(1, len(points)):
            x1, y1 = points[i-1][0], points[i-1][1]
            x2, y2 = points[i][0], points[i][1]
            x, y = abs(x1-x2), abs(y1-y2)
            ans += max(x, y)
        return ans
