class Solution:
    """
    min(x差， y差) + 另一个差-min
    """
    def minTimeToVisitAllPoints(self, points: List[List[int]]) -> int:
        res = 0
        for i in range(1, len(points)):
            x0, y0 = points[i-1] 
            x1, y1 = points[i]
            res +=  max(abs(x0-x1), abs(y0-y1))
        return res
