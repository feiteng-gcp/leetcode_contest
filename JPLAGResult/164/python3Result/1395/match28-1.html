class Solution:
    def minTimeToVisitAllPoints(self, points: List[List[int]]) -> int:
        def dis(p1, p2):
            m, n = abs(p1[0] - p2[0]), abs(p1[1] - p2[1])
            return max(m, n)
        
        res = 0
        for i in range(1, len(points)):
            res += dis(points[i], points[i - 1])
        return res
