class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        modulo = 10 ** 9 + 7
        arrLen = min(arrLen, steps)
        d = [[0] * arrLen for _ in range(steps + 1)]
        d[0][0] = 1
        for t in range(1, steps + 1):
            for pos in range(arrLen):
                d[t][pos] += d[t - 1][pos]
                if pos > 0:
                    d[t][pos] += d[t - 1][pos - 1]
                if pos + 1 < arrLen:
                    d[t][pos] += d[t - 1][pos + 1]
                d[t][pos] %= modulo
        ans = d[steps][0]
        return ans
