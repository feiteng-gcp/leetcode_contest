from functools import lru_cache
class Solution(object):
    def numWays(self, steps, arrLen):
        """
        :type steps: int
        :type arrLen: int
        :rtype: int
        """
        INF = 10 ** 9 + 7
        @lru_cache(None)
        def dp(s, ar, c):
            if s == 0:
                return 1 if c == 0 else 0
            res = dp(s - 1, ar, c)
            if c > 0:
                res += dp(s - 1, ar, c - 1)
            if c + 1 < ar:
                res += dp(s - 1, ar, c + 1)
            # print(s, c, res)
            return res % INF
            
        return dp(steps, min(steps, arrLen), 0)
