class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        MOD = 10**9 + 7
        l = min(steps, arrLen)
        dp = [[0] * l for _ in range(steps + 1)]
        dp[0][0] = 1
        for i in range(steps):
            for j in range(l):
                dp[i + 1][j] = dp[i][j]
                if j > 0:
                    dp[i + 1][j] += dp[i][j - 1]
                if j < l - 1:
                    dp[i + 1][j] += dp[i][j + 1]
                dp[i + 1][j] %= MOD
        return dp[steps][0]
