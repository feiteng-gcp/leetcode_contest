class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        max_move = min(steps // 2 + 1, arrLen)
        dp = [[0] * max_move, [0] * max_move]
        dp
[0][0] = 1
        for i in range(1, steps+1):
            for j in range(max_move):
                if not j:
                    dp[i%2][j] = dp[(i-1)%2][j] + dp[(i-1)%2][j+1]
                elif j == max_move - 1:
                    dp[i%2][j] = dp[(i-1)%2][j] + dp[(i-1)%2][j-1]
                else:
                    dp[i%2][j] = dp[(i-1)%2][j-1] + dp[(i-1)%2][j] + dp
[(i-1)%2][j+1]
                dp[i%2][j] = dp[i%2][j] % (1000000007)
        return dp[steps % 2][0]
