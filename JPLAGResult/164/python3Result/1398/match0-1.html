class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        memo = collections.defaultdict(int)
        memo[0] = 1
        max_ = 1
        for i in range(steps):
            max_ += 1
            tmp = memo.copy()
            for key in range(min(arrLen,max_)):
                if key > 0:
                    tmp[key] += memo[key-1]
                if key < arrLen-1:
                    tmp[key] += memo[key+1]
            memo = tmp
        return memo[0]% (10**9 +7)
