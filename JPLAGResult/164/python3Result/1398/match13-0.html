class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        arrLen = min(arrLen, steps//2 + 1)
        arr = [[0] * arrLen for i in range(steps)]
        arr[0][0] = 1
        if arrLen > 1:
            arr[0][1] = 1
        for i in range(1, steps):
            for j in range(arrLen):
                arr[i][j] = arr[i-1][j]
                if j > 0:
                    arr[i][j] += arr[i-1][j-1]
                if j < arrLen - 1:
                    arr[i][j] += arr[i-1][j+1]
        return arr[-1][0] % (10**9 + 7)
