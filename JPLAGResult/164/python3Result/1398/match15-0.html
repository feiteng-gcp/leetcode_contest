class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        if arrLen == 1:
            return 1
        if arrLen > steps:
            arrLen = steps+1
        ans = [0] * arrLen
        ans[0] = 1
        MOD = 10**9 + 7
        for i in range(steps):
            nans = [0] * arrLen
            nans[0]=(ans[0]+ans[1])%MOD
            nans[arrLen-1]=(ans[arrLen-1]+ans[arrLen-2])%MOD
            for j in range(1, arrLen-1):
                nans[j]=(ans[j-1]+ans[j]+ans[j+1])%MOD
            ans = nans
        return ans[0]
