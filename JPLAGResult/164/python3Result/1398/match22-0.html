from functools import lru_cache
class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        @lru_cache(None)
        def helper(start, numSteps):

            if start<0 or start>=arrLen:
                return 0
            if numSteps == 0:
                if start == 0:
                    return 1
                else:
                    return 0
            if start==numSteps:
                return 1
            if numSteps<start:
                return 0
            x1=helper(start+1,numSteps-1)
            x2=helper(start-1,numSteps-1) 
            x3 = helper(start, numSteps-1)
            # print(x1, start+1,numSteps-1)
            # print(x2, start-1,numSteps-1)
            # print(x3, start,numSteps-1)
            return x1 + x2 + x3
        return helper(0,steps)%1000000007
