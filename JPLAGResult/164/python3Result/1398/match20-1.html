class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        length = min(501, arrLen)
        nums = [0]*length
        nums[0] = 1
        
        for _ in range(steps):
            nums_new = [0] * length
            for i in range(length):
                nums_new[i] = (nums[i] + (nums[i+1] if i+1<length else 0) + (nums[i-1] if i>0 else 0))% (10**9+7)
                if nums[i] == 0:
                    break
            nums = nums_new
        return nums[0]
