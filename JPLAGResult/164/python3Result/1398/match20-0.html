class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        arrLen = min(arrLen, 251)
        dp = [0]*(arrLen+2)
        dp[1] = 1
        for i in range(steps):
            tmp = [it for it in dp]
            for j in range(1,arrLen+1):
                dp[j] = tmp[j-1]+tmp[j]+tmp[j+1]
        return dp[1]%(10**9+7)
