class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        arrLen = min(arrLen, steps//2+1)
        now = [0]*arrLen
        now[0] = 1
        maxn = int(1e9 + 7)
        
        for i in range(steps) :
            now_n = [0]*len(now)
            now_n[0] = (now[0]+now[1]) % maxn
            now_n[-1] = (now[-1]+now[-2]) % maxn
            for j in range(1, len(now)-1) :
                now_n[j] = (now[j-1]+now[j]+now[j+1]) % maxn
            now = now_n
        return now[0] % maxn
