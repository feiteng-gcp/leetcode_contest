class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        arrLen = min(steps // 2 + 1,  arrLen)
        if arrLen == 1:
            return 1

        dp = [0] * arrLen
        dp[0] = 1
        
for _ in range(steps):
            newdp = [0] * arrLen
            for i in range(arrLen):
                if i == 0:
                    newdp[i] = dp[i] + dp[i + 1]
                elif i == arrLen - 1:
                    newdp[i] = dp[i] + dp[i - 1]
                else:
                    newdp[i] = dp[i] + dp[i + 1] + dp[i - 1]
            dp = newdp
            # print(dp)
        return dp[0] % 1000000007
