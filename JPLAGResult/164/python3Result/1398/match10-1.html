class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        M = 10 ** 9 + 7
        dp = [1] + [0] * (min
(arrLen, 251) - 1)
        for _ in range(steps):
            newdp = [0] * len(dp)
            newdp[0] = (dp[0] + dp[1]) % M
            newdp[-1] = (dp[-1] + dp[-2]) % M
            for i in range(1, len(dp) - 1):
                newdp[i] = (dp[i - 1] + dp[i] + dp[i + 1]) % M
            dp = newdp
        return dp[0]
