class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        a = [0] * min(arrLen, steps)
        a[0] = 1
        la = len(a)
        m = 10**9+7
        for i in range(1, steps+1):
            b = a.copy()
            for j in range(la):
                if j-1>=0:
                    b[j] += a[j-1]
                if j+1<la:
                    b[j] += a[j+1]
                b[j] %= m
            a = b.copy()
        return a[0]
