class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        arrSize = min(arrLen, steps)
        num_steps = [0] * arrSize
        num_steps[0] = 1
        for i in range(steps):
            new_num_steps = [0]*arrSize
            for j in range(arrSize):
                new_num_steps[j] = sum(num_steps[max(j-1,0):min(j+2, arrSize)])
            num_steps = new_num_steps
        return num_steps[0] % (10**9+7)
