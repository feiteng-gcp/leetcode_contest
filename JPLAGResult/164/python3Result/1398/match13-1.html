class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        arrLen = min(steps//2+1, arrLen)
        dp = [[0 for _ in range(arrLen)] for _ in range(steps+1)]
        dp[0][0] = 1
        
for s in range(1, steps+1):
            for i in range(arrLen):
                dp[s][i] = dp[s-1][i]
                if i+1<arrLen:
                    dp[s][i] += dp[s-1][i+1]
                if i - 1 >= 0:
                    dp[s][i] += dp[s - 1][i - 1]

        return dp[steps][0]% (10**9 + 7)
