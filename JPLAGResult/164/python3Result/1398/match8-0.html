class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        if arrLen == 1: return 1
        MOD = 10**9 + 7
        L = min(arrLen, steps*2)
        dp = [[0] * L for _ in range(steps)]
        dp[0][0] = dp[0][1] = 1
        for i in range(steps-1):
            for j in range(L):
                dp[i+1][j] += dp[i][j]
                if j > 0:
                    dp[i+1][j-1] += dp[i][j]
                if j < L-1:
                    dp[i+1][j+1] += dp[i][j]
        return dp[-1][0] % MOD
