class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        mod = 10 ** 9 + 7
        res = 0
        
        pos = collections.Counter()
        pos[0] = 1
        
        for i in range(steps):
            new_pos = collections.Counter()
            
            for x in pos.keys():
                if x - 1 >= 0:
                    new_pos[x-1] += pos[x]
                if x + 1 < arrLen:
                    new_pos[x+1] += pos[x]
                new_pos[x] += pos[x]
            pos = new_pos
        
        return pos[0] % mod
