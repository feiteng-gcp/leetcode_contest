class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        mod = 1000000007
        arrLen = min(steps, arrLen)
        dp_old = [0] * arrLen
        dp_old[0] = 1 
        for _ in range(steps):
            dp_new = dp_old.copy()
            for i in range
(arrLen):
                if i > 0:
                    dp_new[i] += dp_old[i - 1]
                if i < arrLen - 1:
                    dp_new[i] += dp_old[i + 1]
                dp_new[i] = dp_new[i] % mod 
            dp_old = dp_new
        return dp_new[0]
