class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        def recursive(x, n):
            mod = 10 ** 9 + 7
            if n == 0:
                return 1 if x == 0 else 0
            if (x, n) in dp:
                return dp[(x, n)]
            dp[(x, n)] = recursive(x, n - 1)
            if x - 1 >= 0:
                dp[(x, n)] = (dp[(x, n)] + recursive(x - 1, n - 1)) % mod
            if x + 1 < arrLen:
                dp[(x, n)] = (dp[(x, n)] + recursive(x + 1, n - 1)) % mod
            return dp[(x, n)]
            
        dp = dict()
        return recursive(0, steps)
