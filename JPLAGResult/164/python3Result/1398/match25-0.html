class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        maxLen = min(arrLen, steps // 2 + 2)
        dp = [[0] * maxLen
              for _ in range(steps + 1)]
        dp[0][0] = 1
        
        for step in range(1, steps + 1):
            for position in range(maxLen):
                if position > 0:
                    dp[step][position] += dp[step - 1][position - 1]
                dp[step][position] += dp[step - 1][position]
                if position < maxLen - 1:
                    dp[step][position] += dp[step - 1][position + 1]
                dp[step][position] = dp[step][position] % (10 ** 9 + 7)
        
        # print(dp)
        
        return dp[steps][0]
