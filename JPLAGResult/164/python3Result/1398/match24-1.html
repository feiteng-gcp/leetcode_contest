class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        times = [0] * min(steps,arrLen)
        times[0] = 1
        if steps == 1:return 1
        if steps == 2 and arrLen >= 2:return 2
        if steps == 2 and arrLen == 1:return 1
        if arrLen == 1:return 1
        if len
(times) >= 2:
            times[1] = 1
            for x in range(steps-1):
                newtimes = [0] * len(times)
                newtimes[0] = times[0] + times[1]
                for pos in range(1,len(times)-1):
                    newtimes[pos] = times[pos] + times[pos-1] + times[pos+1]
                newtimes[-1] = times[-1] + times[-2]
                times = newtimes
        return times[0] % (10**9+7)
