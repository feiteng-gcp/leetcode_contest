class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        
        if arrLen == 1:
            return steps
        
        limit = min(arrLen, 501)  # cannot walk further in 500 steps      
        dp_prev = [0] * limit
        
        # step 1
        dp_prev[0] = 1
        if len(dp_prev) > 1:
            dp_prev[1] = 1
        
        for s in range(1, steps):
            
            dp_curr = [0] * limit
            for i in range(limit):
                if i == 0:
                    dp_curr[i] = dp_prev[i] + dp_prev[i+1]
                elif i == limit-1:
                    dp_curr[i] = dp_prev[i] + dp_prev[i-1]
                else:
                    dp_curr[i] = dp_prev[i] + dp_prev[i-1] + dp_prev[i+1]
            dp_prev = dp_curr
        
            # print(dp_prev)
        
        return dp_prev[0] % 1000000007
