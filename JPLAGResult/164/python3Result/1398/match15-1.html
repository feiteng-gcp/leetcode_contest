class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        if arrLen==1: return 1
        mod = 10**9 + 7
        dp = [0] * arrLen
        dp[0] 
= 1
        for i in range(steps):
            tmp = [0] * arrLen
            tmp[0] = dp[0]+dp[1]
            tmp[-1] = dp[-1]+dp[-2]
            for j in range(1,min(steps+5,arrLen-1)):
                tmp
[j] = (dp[j] + dp[j-1] + dp[j+1])%mod
            dp = tmp
        return dp[0]%mod
