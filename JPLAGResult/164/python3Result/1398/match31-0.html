class Solution:
    def numWays(self, steps: int, arrLen: int) -> int:
        dp = [0] * (min(steps//2+2, arrLen) + 2)
        dp[1] = 1
        for _ in range(steps):
            newdp = [0]*len(dp)
            for i in range(1, len(dp)-1):
                newdp[i] = (dp[i-1] + dp[i] + dp[i+1]) % (10**9+7)
            dp = newdp
        return dp[1]
