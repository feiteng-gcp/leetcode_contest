class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        R, C = len(grid), len
(grid[0])
        r = [0] * R
        c = [0] * C
        for i in range(R):
            for j in range(C):
                r[i] += grid[i][j]
                c[j] += grid[i][j]
        ans = 0
        for i in range(R):
            for j in range(C):
                if grid[i][j] == 1:
                    if r[i] > 1 or c[j] > 1:
                        ans += 1
        return ans
