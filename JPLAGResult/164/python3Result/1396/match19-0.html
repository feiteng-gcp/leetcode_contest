class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        r = [0] * m
        l = [0] * n
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    r[i]+=1
                    l[j]+=1
        ans = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    if r[i]>1 or l[j]>1:
                        ans += 1
        return ans
