class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        if not grid:
            return 0
        dx = [0] * 300
        dy 
= [0] * 300
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    dx[i] += 1
                    dy[j] 
+= 1
        r = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    if dx[i] > 1 or dy[j] > 1:
                        r += 1
        return r
