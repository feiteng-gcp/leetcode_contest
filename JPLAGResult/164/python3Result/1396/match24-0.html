class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        if(len(grid)==0): return 0
        
        rows = [0]*len(grid)
        cols = [0]*len(grid[0])
        
        for y in range(len(grid)):
            for x in range(len(grid[0])):
                if(grid[y][x]==1):
                    rows[y] += 1
                    cols[x] += 1
        count = 0
        for y in range(len(grid)):
            for x in range(len(grid[0])):
                if(grid[y][x]==1 and (rows[y]>1 or cols[x]>1)):
                    count += 1

        return count
