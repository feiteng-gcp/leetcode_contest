class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        def dfs(i, j):
            visited[i][j] = True
            nodes[0] += 1
            for y in range(n):
                if grid[i][y] == 1 and not visited[i][y]:
                    dfs(i, y)
            for x in range(m):
                if grid[x][j] == 1 and not visited[x][j]:
                    dfs(x, j)
        
        m, n = len(grid), len(grid[0])
        visited = [[False for _ in range(n)] for _ in range(m)]
        servers = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1 and not visited[i][j]:
                    nodes = [0]
                    dfs(i, j)
                    if nodes[0] > 1:
                        servers += nodes[0]
        return servers
