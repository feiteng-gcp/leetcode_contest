from collections import defaultdict


class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        rows = len(grid)
        cols = len(grid[0])
        
        rowcnt = [0]*rows
        colcnt = [0]*cols
        
        for r in range(rows):
            for c in range(cols):
                if grid[r][c]==1:
                    rowcnt[r] += 1
                    colcnt[c] += 1
        result = 0                    
        for r in range(rows):
            for c in range(cols):
                if grid[r][c]==1:
                    if rowcnt[r]>1 or colcnt[c]>1:
                        result += 1
                        
        return result
