class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        row_svr_cnt = [0] * m
        col_svr_cnt = [0] * n
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    row_svr_cnt[i] += 1
                    col_svr_cnt[j] += 1
        
        result = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1 and (row_svr_cnt[i] > 1 or col_svr_cnt[j] > 1):
                    result += 1
        return result
