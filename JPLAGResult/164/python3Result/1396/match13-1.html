class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        
        r = [0 for _ in range(m)]
        for i in range(m):
            if sum(grid[i]) > 1:
                r[i] = 1
        
        c = [0 for _ in range(n)]
        for j in range(n):
            if sum([x[j] for x in grid]) > 1:
                c[j] = 1
        
        res = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1 and (r[i] == 1 or c[j] == 1):
                    res += 1
        
        return res
