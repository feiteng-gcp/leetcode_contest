class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len
(grid[0])
        r = [0] * m
        c = [0] * n
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    r[i] += 1
                    c[j] += 1
        ret = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] and (r[i] > 1 or c[j] > 1): ret += 1
        return ret
