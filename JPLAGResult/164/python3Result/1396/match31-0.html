class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        
        
        M = len(grid)
        if M == 0:
            return 0
        
        N = len(grid[0])
        
        row = {}
        col = {}
        
        for i in range(M):
            for j in range(N):
                if grid[i][j]==1:
                    row[i]=row.get(i,0)+1
                    col[j]=col.get(j,0)+1
        
        count = 0
        for i in range(M):
            for j in range(N):
                if grid[i][j]==1:
                    if row[i]>1 or col[j]>1:
                        count+=1
        
        return count
