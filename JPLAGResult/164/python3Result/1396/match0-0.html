class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        mcnt = [0] * m
        ncnt = [0] * n
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    mcnt[i] += 1
                    ncnt[j] += 1
        
        cnt = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] and (mcnt[i] > 1 or ncnt[j] > 1):
                    cnt += 1
        return cnt
