class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        
        row_s, col_s = collections.defaultdict(int), collections.defaultdict(int)
        n = 0
        for i in range(len(grid)):
            
            for j in range(len(grid[0])):
                
                if grid[i][j]:
                    row_s[i] += 1
                    col_s[j] += 1
                    n += 1

        out = 0
        for i in range(len(grid)):
            
            for j in range(len(grid[0])):
                
                if grid[i][j]:
                    if row_s[i] == 1 and col_s[j] == 1: out += 1
        
        return n - out
