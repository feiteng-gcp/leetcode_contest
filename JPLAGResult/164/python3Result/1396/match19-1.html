class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        nRow=len(grid)
        nCol=len(grid[0])
        h=[0]*nRow
        v=[0]*nCol
        for r,row in enumerate(grid):
            for c,col in enumerate(row):
                h[r]+=grid[r][c]
                v[c]+=grid[r][
c]
        
        ans = 0
        
        for r,row in enumerate(grid):
            for c,col in enumerate(row):
                if grid[r][c] == 1:
                    if h[r]>1 or v[c]>1:
                        ans+=1
        
        
        return ans
