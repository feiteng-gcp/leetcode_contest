class Solution:
    def countServers(self, grid) -> int:
        m 
= len(grid)
        n = len(grid[0])
        sx = [0] * m
        sy = [0] * n
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    sx[i] += 1
                    sy[j] += 1
        ans = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    if sx[i] > 1 or sy[j] > 1:
                        ans += 1
        return ans
