class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        d1 
= [0]*len(grid)
        d2 = [0]*len(grid[0])
        
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    d1[i] += 1
                    d2[j] += 1
                    
                    
        ans = 0;
        
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1 and (d1[i] > 1 or d2[j] > 1):
                    ans += 1
        
        return ans;
