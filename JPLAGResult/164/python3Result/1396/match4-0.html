class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        n,m=len(grid),len(grid[0])
        col=[0]*m
        row=[0]*n
        for i in range(n):
            for j in range(m):
                if grid[i][j]:
                    row[i]+=1
                    col[j]+=1
        ans=0
        for i in range(n):
            for j in range(m):
                if grid[i][j] and (col[j]>1 or row[i]>1):
                    ans+=1
        return ans
