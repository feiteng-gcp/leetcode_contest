class Solution:
    def countServers(self, grid: List[List[int]]) -> int:
        col_d = collections.defaultdict(set)
        row_d = collections.defaultdict(set)
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j]:
                    row_d[i].add((i,j))
                    col_d[j].add((i,j))
        res = set()
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j]:
                    if len(col_d[j]) > 1 or len(row_d[i]) > 1:
                        res.add((i,j))
        return len(res)
