class Node:
    def __init__(self):
        self.children = {}
        self.is_word = False
        self
.words = []
        
class Trie:
    def __init__(self):
        self.root = Node()
        
    def add(self, word):
        node = self.root
        for c in word:
            if c not in node.children:
                node.children[c] = Node()
            node = node.children[c]
            node.words.append(word)
        
    def suggest(self, s):
        node = self.root
        res = []
        for c in s:
            if c not in node.children:
                break
            
            res.append(sorted(node.children[c].words)[:3])
            node = node.children[c]

        while len(res) < len(s):
            res.append([])
        
return res
        
class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        trie = Trie()
        for word in products:
            trie.add(word)
        
        return trie.suggest(searchWord)
