class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        ans = []
        for i in range(len(searchWord)):
            subs = searchWord[:i+1]
            ans.append(list(filter(lambda x: x.startswith(subs), products))[:3])
        return ans
