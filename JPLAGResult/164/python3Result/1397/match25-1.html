class TrieNode:
    def __init__(self):
        self.children = collections.defaultdict(TrieNode)
        self.commonWords = []

class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        root = TrieNode()
        for p in products:
            node = root
            for ch in p:
                node = node.children[ch]
                node.commonWords.append(p)
        def dfs(node):
            node.commonWords.sort()
            for nextNode in node.children.values():
                dfs(nextNode)
        dfs(root)
        
        ans = []
        node = root
        for ch in searchWord:
            node = node.children[ch]
            ans.append(node.commonWords[:3])
        return ans
