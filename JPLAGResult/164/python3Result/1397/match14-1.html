class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        out = []
        for x in range(1, len(searchWord)+1):
            products = [y for y in products if y[:x]==searchWord[:x]]
            out.append(products[:3])
        return out
