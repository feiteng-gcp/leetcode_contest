class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        cpos=products
        ret=[]
        for i in range(1,len(searchWord)+1):
            ncpos=[]
            for cand in cpos:
                if cand[:i]==searchWord[:i]:
                    ncpos.append(cand)
            cpos=ncpos
            ret.append(cpos[:3])
        return ret
