class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        n = len(searchWord)
        m = len(products)
        ans = []
        for i in range(n):
            ansi = []
            num = 0
            for s in products:
                if s[:i+1]==searchWord[:i+1]:
                    ansi.append(s)
                    num+=1
                if num==3:
                    break
            ans.append(ansi)
        return ans
