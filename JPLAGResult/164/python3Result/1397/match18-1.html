class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products
.sort()
        ans = []
        for i in range(len(searchWord)):
            tmp = []
            for p in products:
                if i<len(p) and p[i]==searchWord[i]:
                    tmp.append(p)
            if len(tmp)>3:ans.append(tmp[:3].copy())
            else: ans.append(tmp)
            products = tmp
        return ans
