class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        m = len(searchWord)
        n = len(products)
        sol = []
        for i in range(1, m+1):
            s = searchWord[:i]
            ind = bisect.bisect_left(products, s)
            sol.append([products[ind+j] for j in range(3) if ind+j<n and products[ind+j][:len(s)]==s])
        return sol
