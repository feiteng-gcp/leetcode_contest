import bisect

class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        res = []
        for i in range(len(searchWord)):
            s0 = searchWord[:i + 1]
            s1 = s0 + 'z' * (2 * 10**4)
            i0 = bisect.bisect_left(products, s0)
            i1 = bisect.bisect_left(products, s1)
            res.append(products[i0:min(i0 + 3, i1)])
        return res
