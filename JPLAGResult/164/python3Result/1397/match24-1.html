class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        result = []
        
        for i in range(len(searchWord)):
            current = searchWord[:i + 1]
            words = []
            
            pos = 0
            while pos < len(products) and len(words) < 3 and len(products) > 0:
                if products[pos][:i + 1] == current:
                    words.append(products[pos])
                    pos += 1
                else:
                    products.pop(pos)
                    
            result.append(words)
            
        return result
