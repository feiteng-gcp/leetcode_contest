class Solution:
    def suggestedProducts(self, xs: List[str], y: str) -> List[List[str]]:
        xs.sort()
        ans = []
        for i in range(1, len(y) + 1):
            l = -1
            r = len(xs)
            while r - l > 1:
                m = (l + r) // 2
                if xs[m] >= y[:i]:
                    r = m
                else:
                    l = m
            ans.append(list(filter(lambda x : x.startswith(y[:i]), xs[r:r+3])))
        return ans
