class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        
        products.sort()
        res = list()
        
        n = len(searchWord)
        for i in range(1, n+1):
            curr_search = searchWord[:i]
            products = [i for i in products if i.startswith(curr_search)]
            if len(products) > 3:
                curr_res = products[:3]
            else:
                curr_res = products
            res.append(curr_res)
        
        return res
