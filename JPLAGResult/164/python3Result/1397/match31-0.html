class Trie:
    def __init__(self):
        self.d = {}
        self.words = []

class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        products.sort()
        n = len(products)
        root = Trie()
        # print(products)
        for i in range(n):
            p = products[i]
            curr = root
            for c in p:
                if c not in curr.d:
                    curr.d[c] = Trie()
                curr = curr.d[c]
                curr.words.append(i)
                    
        ans = []
        curr = root
        for c in searchWord:
            tmp = []
            if c not in curr.d:
                curr.d[c] = Trie()
            curr = curr.d[c]
            l = min(3, len(curr.words))
            for i in range(l):
                tmp.append(products[curr.words[i]])
            ans.append(tmp)
        return ans
