class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        pre_dict = collections.defaultdict(list)
        for p in products:
            for i in range(min(len(p),10)):
                pre_dict[p[:i+1]] += [p]
        for key in pre_dict:
            pre_dict[key].sort()
        res = []
        if len(searchWord) > 10:
            stop_key = searchWord[:10]
            stop_list = pre_dict[stop_key]
        for i in range(len(searchWord)):
            tmp = searchWord[:i+1]
            if len(tmp) <= 10:
                res += [pre_dict[tmp][:3]]
            else:
                potential = []
                for word in stop_list:
                    if word.startswith(tmp):
                        potential += [word]
                res += [potential[:3]]
                stop_list = potential
        return res
