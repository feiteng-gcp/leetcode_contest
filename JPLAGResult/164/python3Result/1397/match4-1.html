class TrieNode: 
    def __init__(self): 
        self.children = dict()
        self.isEndOfWord = 0

class Trie:  
    def __init__(self): 
        self.root = self.getNode() 
  
    def getNode(self): 
        return TrieNode()  
  
    def insert(self,key): 
        pCrawl = self.root 
        length = len(key) 
        for level in range(length): 
            if key[level] not in pCrawl.children: 
                pCrawl.children[key[level]] = self.getNode() 
            pCrawl = pCrawl.children[key[level]]
  
        pCrawl.isEndOfWord += 1

    def search(self, key):  
        pCrawl = self.root 
        length = len(key) 
        
        for level in range(length): 
            if key[level] not in pCrawl.children:
                return []
            pCrawl = pCrawl.children[key[level]]
        
        res = []
        
        def dfs(node, cur):
            if len(res) >= 3:
                return
            else:
                if node.isEndOfWord > 0:
                    res.extend([key + cur] * node.isEndOfWord)
                
                for x in sorted(node.children.keys()):
                    dfs(node.children[x], cur+x)
        
        dfs(pCrawl, '')
        
        return res[:3]
        
            

class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        t = Trie()
        for x in products:
            t.insert(x)
        
        res = []
        for i in range(len(searchWord)):
            res.append(t.search(searchWord[:i+1]))
        
        return res
