class Solution {
    private static final int MOD = 1000000000 + 7;
    public int numWays(int steps, int arrLen) {
        int[] dp = new int[arrLen];
        dp[0] = 1;
        for (int i = 0; i < steps; i++) {
            int[] next = new int[arrLen];
            for (int j = 0; j < arrLen; j++) {
                if (dp[j] == 0) {
                    break;
                }
                if (j > 0) {
                    next[j - 1] += dp[j];
                    next[j - 1] %= MOD;
                }
                next[j] += dp[j];
                next[j] %= MOD;
                if (j + 1 < arrLen) {
                    next[j + 1] += dp[j];
                    next[j + 1] %= MOD;
                }
            }
            dp = next;
        }
        return dp[0];
    }
}
