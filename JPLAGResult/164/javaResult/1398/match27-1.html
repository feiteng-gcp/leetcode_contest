class Solution {
    private static final int MOD = 1_000_000_000 + 7;
    
    
public int numWays(int steps, int arrLen) {
        int len = Math.min(steps * 2 + 1, arrLen);
        int[][] dp = new int[steps+1][len];
        dp[0][0] = 1;
        for (int step = 1; step <= steps; step++) {
            for (int pos = 0; pos <= Math.min(steps - step, len-1); pos++) {
                dp[step][pos] = dp[step-1][pos]; // stay
                if (pos >= 1) dp[step][pos] = (dp[step][pos] + dp[step-1][pos-1]) % MOD;
                if (pos < len-1) dp[step][pos] = (dp[step][pos] + dp[step-1][pos+1]) % MOD
;
                dp[step][pos] %= MOD;
            }
        }
        return dp[steps][0];
    }
}
