class Solution {
    public int numWays(int steps, int arrLen) {
        arrLen = Math.min(arrLen, 501);
        long[] arr = new long[arrLen];
        arr[0] = 1;
        int Mod = 1000_000_007;
        for(int step=0; step<steps; step++) {
            long[] next = new long[arrLen];
            for(int i=0; i<arr.length; i++) {
                if(i-1>=0) {
                    next[i-1] += arr[i];
                    next[i-1] %= Mod;
                }
                next[i] += arr[i];
                next[i] %= Mod;
                
                if(i+1<arrLen) {
                    next[i+1] += arr[i];
                    next[i+1] %= Mod;
                }
            }
            arr = next;
        }
        return (int)arr[0];
    }
    
}
