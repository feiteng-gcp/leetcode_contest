	class S
olution {
	    public int numWays(int steps, int arrLen) {
	        int mod = 1000000007;
	        int n = Math.min(arrLen, steps+1);
	        long[] dp = new long[n];
	        dp[0] = 1;
	        for(int i = 0;i < steps;i++){
	        	long[] ndp = new long[n];
	        	for(int j = 0;j < n;j++){
	        		for(int k = -1;k <= 1;k++){
	        			if(j+k >= 0 && j+k < n){
	        				ndp[j+k] += dp[j];
	        				if(ndp[j+k] >= mod)ndp[j+k] -= mod;
	        			}
	        		}
	        	}

	        	dp = ndp;
	        }
	        return (int)dp[0];
	    }
	}
