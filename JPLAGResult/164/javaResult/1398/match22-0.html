class Solution {
    public int numWays(int steps, int arrLen) {
        int mod = 1000000007;
        int n = Math.min(arrLen,steps+1);
        long[] dp = new long[n];
        dp[0] = 1;
        for(int step=0; step<steps; step++){
            long[] newDp = new long[n];
            for(int i=0; i<n; i++){
                newDp[i] = ((i-1>=0 ? dp[i-1] : 0) + dp[i] + (i+1<n ? dp[i+1] : 0))%mod;
            }
            dp = newDp;
        }
        return (int)(dp[0]%mod);
    }
}
