class Solution {
    public int numWays(int steps, int arrLen) {
		int len = Math.min(steps, arrLen);
        long[] dp = new long[len];
        dp[0] = 1;
        for (int i = 0; i < steps; i++) {
        	long[] dpTmp = new long[len];
        	dpTmp[0] = (dp[0] + dp[1]) % 1_000_000_007;
        	for (int j = 1; j < len - 1; j++) {
        		dpTmp[j] = (dp[j-1] + dp[j] + dp[j+1]) % 1_000_000_007;
        	}
        	dpTmp[len - 1] = (dp[len-2] + dp[len-1]) % 1_000_000_007;
        	dp = dpTmp;
        }
        return (int) dp[0];
    }
}
