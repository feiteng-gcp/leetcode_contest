class Solution {
    private static final int MOD = 1000000007;
    
    public int numWays(int steps, int arrLen) {
        arrLen = Math.min(steps + 1, arrLen);
        long[] dp = new long[arrLen];
        dp[0] = 1;
        for (int i = 0; i < steps; ++i) {
            long[] next = new long[arrLen];
            for (int j = 0; j < arrLen; ++j) {
                next[j] += dp[j];
                if (j - 1 >= 0) {
                    next[j] += dp[j - 1];
                }
                if (j + 1 < arrLen) {
                    next[j] += dp[j + 1];
                }
                next[j] %= MOD;
            }
            dp = next;
        }
        return (int)dp[0];
    }
}
