class Solution {
    public int numWays(int steps, int arrLen) {
        arrLen = 
Math.min(arrLen,steps + 1);
        long[] ways = new long[arrLen];
        ways[0] = 1;
        for (int i = 1; i <= steps ; i++) {
            long[] nways = new long[arrLen];
            for (int j = 0; j < arrLen; j++) {
                nways[j] = ways[j] + (j > 0 ? ways[j - 1] : 0) + (j < arrLen - 1 ? ways[j + 1] : 0);
                nways[j] %= 1000000007;
            
}
            ways = nways;
        }
        return (int) ways[0];
    }
}
