class Solution {
    private int mod = (int) Math.pow(10, 9) + 7;
    
    public int numWays(int steps, int arrLen) {
        if (steps <= 0) {
            return 1;
        }
        
        int reach = Math.min(steps, arrLen);
        
long[] visited = new long[reach];
        visited[0] = 1;
        for (int i = 0; i < steps; i++) {
            long[] current = new long[visited.length];
            for (int index = 0; index < current.length; index++) {
                if (index > 0) {
                    current[index] += visited[index - 1];
                }
                current[index] += visited[index];
                current[index] %= mod;
                if (index < current.length - 1) {
                    current[index] += visited[index + 1];
                }
                current[index] %= mod;
            }
            visited = current;
        }
        return (int) visited[0];
    }
}
