class Solution {
    public int numWays(int steps, int arrLen) {
    	arrLen = Math.min(arrLen, 505);
    	long[] dp = new long[505];
    	dp[0] = 1;
    	dp[
1] = 1;
    	for(int i = 1; i < steps; i++) {
    		long[] newdp = new long[505];
    		newdp[0] = (dp[0] + dp[1])  % 1000000007L;
    		for(int j = 1; j < arrLen - 1; j++) {
    			newdp[j] = (dp[j-1] + dp[j] + dp[j+1]) % 1000000007L;	
    		}
    		newdp[arrLen-1] = (dp[arrLen-1] + dp[arrLen-2]) % 1000000007L; 
    		
    		dp = newdp;
    	}
        return (int)dp[0];
    }
}
