class Solution {
    private static int MOD = 1000000007;
    public int numWays(int steps, int len) {
        len = Math.min(len, steps*2);
        int[][] dp = new int[len][steps+1];
        dp[0][0] = 1;
        for (int s = 1; s <= steps; s++) {
            for (int i = 0; i < len; i++) {
                if (i > 0) {
                    dp[i][s] = (dp[i][s] + dp[i-1][s-1]) % MOD;
                }
                if (i < len-1) {
                    dp[i][s] = (dp[i][s] + dp[i+1][s-1]) % MOD;
                }
                dp[i][s] = (dp[i][s] + dp[i][s-1]) % MOD;
            }
        }
        return dp[0][steps];
    }
}
