class Solution {
    public int numWays(int steps, int arrLen) {
        arrLen = Math.min(arrLen, steps + 1);
        long[] a = new long[arrLen];
        a[0] = 1;
        for (int i = 0; i < steps; i++) {
            long[] next = new long[arrLen];
            for (int j = 0; j < arrLen; j++) {
                next[j] += a[j];
                if (j > 0) {
                    next[j] += a[j - 1];
                }
                if (j < arrLen - 1) {
                    next[j] += a[j + 1];
                }
                next[j] %= 1000000000 + 7;
            
}
            a = next;
        }
        return (int) a[0];
    }
}
