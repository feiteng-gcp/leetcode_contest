class Solution {
    public int numWays(int steps, int arrLen) {
        int len = Math.min(steps, arrLen);
        int[] cur = new int[len];
        cur[0] = 1;
        for (int i = 0; i < steps; i++) {
            int[] next = new int[len];
            for (int j = 0; j < len; j++) {
                next[j] = cur[j];
                if (j > 0) {
                    next[j] += cur[j-1];
                    next[j] %= Math.pow(10, 9) + 7;
                }
                if (j < len-1) {
                    next[j] += cur[j+1];
                    next[j] %= Math.pow(10, 9) + 7;
                }
            }
            cur = next;
        }
        return cur[0];
    }
}
