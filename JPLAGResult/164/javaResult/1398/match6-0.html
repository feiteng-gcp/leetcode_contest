class Solution {
        public int numWays(int steps, int arrLen) {
        int md = 1_000_000_007;
        long[] dp1 = new long[steps+2];
        long[] dp2 = new long[steps+2];
        dp1[0] = 1;
        int minLen = Math.min(steps,arrLen);
        for(int i=0;i<steps;++i){
            for(int j=0;j<minLen;++j){
                if(dp1[j]>0){
                    if(j>0){
                        dp2[j-1]+=dp1[j];
                        dp2[j-1]%=md;
                    }
                    dp2[j]+=dp1[j];
                    dp2[j]%=md;
                    dp2[j+1]+=dp1[j];
                    dp2[j+1]%=md;
                }
            }
            dp1 = dp2;
            dp2 = new long[steps+2];
        }
        long ans = dp1[0];
        return (int)ans;
    }
}
