class Solution {
    public int numWays(int steps, int arrLen) {
        int n = Math.min(steps+1, arrLen);
        int[][] dp = new int[steps+1][n];
        dp[0][0] = 1;
        for (int i = 1; i <= steps; i++) {
            for (int j = 0; j < n; j++) {
                dp[i][j] += dp[i-1][j];
                dp[i][j] %= 1000000007;
                if (j > 0) {
                    dp[i][j] += dp[i-1][j-1];
                    dp[i][j] %= 1000000007;
                }
                if (j < n-1) {
                    dp[i][j] += dp[i-1][j+1];
                    dp[i][j] %= 1000000007;
                }
            }
        }
        return dp[steps][0];
    }
}
