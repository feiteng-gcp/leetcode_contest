class Solution {

    private static final int MOD = (int)1e9 + 7;
    
    public int numWays(int steps, int len) {
        len = Math.min((steps >> 1) + 1, len);
        long[][] dp = new long[steps + 1][len];
        dp[0][0] = 1;
        for(int i = 1; i <= steps; i++){
            for(int j = 0 ;j < len ; j++){
                long left = j == 0 ? 0 : dp[i - 1][j-1];
                long right = j == len - 1 ? 0 : dp[i - 1][j+1];
                dp[i][j] = left + right + dp[i-1][j];
                dp[i][j] %= MOD;
            }
        }
        return (int)dp[steps][0];
    }
}
