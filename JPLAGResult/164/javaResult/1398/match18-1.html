class Solution {
    public int numWays(int steps, int l) {
        if(l>steps){
            l = steps;
        }
        
        long p = 1000000007;
        
long dp[] = new long[l+1];
        dp[1] = 1;
        for(int used =0;used<steps;++used){
            long ndp[] = new long[l+1];
            for(int i=1;i<=l;++i){
                if(i+1<=l){
                    ndp[i+1] += dp[i]; 
                    ndp[i+1] %= p;
                }
                if(i-1>=1){
                    ndp[i-1] += dp[i]; 
                    ndp[i-1] %= p;
                }
                ndp[i] += dp[i];
                ndp[i] %= p;
            }
            dp = ndp;
        }
        return (int)dp[1];
        
        
    }
}
