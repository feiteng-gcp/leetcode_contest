class Solution {
    
    private class Node {
        Node[] childs = new Node[26];
        PriorityQueue<String> words = new PriorityQueue<>();
    }
    
    private Node root;
    
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        root = new Node();
        List<List<String>> res = new ArrayList<>();
        for(String product:products){
            addTrie(root, product, 0);
        }
        Node cur = root;
        for(int i = 0 ; i < searchWord.length(); i++){
            int index = searchWord.charAt(i) - 'a';
            if(cur == null || cur.childs[index] == null){
                cur = null;
                res.add(new ArrayList<>());
            } else {
                cur = cur.childs[index];
                List<String> item = new ArrayList<>();
                int cnt = 3;
                while(!cur.words.isEmpty() && cnt-- > 0){
                    item.add(cur.words.poll());
                }
                res.add(item);
            }
        }
        return res;
    }
    
    private void addTrie(Node node, String s, int sIndex){
        if(sIndex >= s.length()){
            return;
        }
        int index = s.charAt(sIndex) - 'a';
        if(node.childs[index] == null){
            node.childs[index] = new Node();
        }
        node = node.childs[index];
        node.words.add(s);
        addTrie(node, s, sIndex + 1);
    }
}
