class Solution {
    public List<List<String>> suggestedProducts(String[] p, String sw) {
        List<String> m = new ArrayList<>();
        Arrays.sort(p);
        for (String s : p) {
            if (s.charAt(0) == sw.charAt(0)) {
                m.add(s);
            }
        }
        
        List<List<String>> res = new ArrayList<>();
        
List<String> tmp = new ArrayList<>();
        for (int i = 0; i < Math.min(3, m.size()); i++) {
            tmp.add(m.get(i));
        }
        res.add(tmp);
        
        for (int i = 1; i < sw.length(); i++) {
            
List<String> mm = new ArrayList<>();
            for (String s : m) {
                if (s.length() > i && s.charAt(i) == sw.charAt(i)) {
                    mm.add(s);
                }
            }
            m = mm;
            
tmp = new ArrayList<>();
            for (int j= 0; j < Math.min(3, m.size()); j++) {
                tmp.add(m.get(j));
            }
            res.add(tmp);
            
        }
        return res;
    }
    
    
}
