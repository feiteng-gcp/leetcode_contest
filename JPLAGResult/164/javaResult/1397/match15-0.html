class Solution {
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        // In order, SOB
        Arrays.sort(products);
        
        ArrayList<List<String>> ret = new ArrayList<>();
        ArrayList<String> past = new ArrayList<>();
        
        for(int i = 0; i < products.length; i++) {
            if(products[i].length() > 0 && products[i].charAt(0) == searchWord.charAt(0)) {
                past.add(products[i]);
            }
        }
        
        ArrayList<String> toAdd = new ArrayList<>();
        
        for(int i = 0; i < Math.min(3, past.size()); i++) {
            toAdd.add(past.get(i));
        }
        ret.add(toAdd);
        
        ArrayList<String> current = new ArrayList<>();
        
        for(int i = 1; i < searchWord.length(); i++) {
            for(String product : past) {
                if(product.length() > i && searchWord.charAt(i) == product.charAt(i)) {
                    current.add(product);
                }
            }
            
            past = current;
            current = new ArrayList<>();
            
            toAdd = new ArrayList<>();
            
            for(int j = 0; j < Math.min(3, past.size()); j++) {
                toAdd.add(past.get(j));
            }
            ret.add(toAdd);
        }
        
        return ret;
    }
}
