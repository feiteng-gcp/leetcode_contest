class Solution {
    int start = 0;
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        Arrays.sort(products);
        ArrayList<List<String>> res = new ArrayList<List<String>>();
        for (int i = 0; i < searchWord.length(); i++) {
            res.add(helper(products,searchWord.substring(0,i+1)));
        }
        return res;
    }
    
    List<String> helper(String[] products,  String str) {
        ArrayList<String> res = new ArrayList<String>();
        int j = 0;
        for (int i = start; i < products.length && j < 3; ) {
            if (products[i].startsWith(str)) {
                res.add(products[i]);
                j++;
                i++;
            } else {
                int cmp = products[i].compareTo(str);
                if (cmp < 0) {
                    i++;
                    start = i;
                } else if (cmp > 0) {
                    break;
                }
            }
        }
        return res;
    }
}
