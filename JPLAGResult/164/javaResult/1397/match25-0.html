class Solution {
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        Arrays.sort(products);
        List<List<String>> ans = new ArrayList<>();
        int pos = 0;
        for (int i = 1; i <= searchWord.length(); ++i) {
            String want = searchWord.substring(0, i);
            while (pos < products.length && !products[pos].startsWith(want)) {
                ++pos;
            }
            // while (pos + 1 < products.length && want.compareTo(products[pos + 1]) < 0) {
            //     ++pos;
            // }
            List<String> lst = new ArrayList<>();
            for (int j = 0; j < 3 && pos + j < products.length; ++j) {
                if (!products[pos + j].startsWith(want)) {
                    break;
                }
                lst.add(products[pos + j]);
            }
            ans.add(lst);
        }
        return ans;
    }
}
