class Solution {
    //这题就是prefix搜索，返回每步骤的全部子串，注意最多3个，多余3个就取字母顺序小的
    //标准做法自然就是Trie树，找该子串是否存在，再往下找
    //但是这题的searchWord只要求处理一次（估计面试的话需要考虑多次调用的吧），所以完全可以O(N)遍历products，找出所有符合的，下一轮还可以迭代使用
    //之前先排序，复杂度O(nlogn + len * n) space: O(N)
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        List<List<String>> res = new ArrayList<>();
        List<String> list = new ArrayList<>();
        Arrays.sort(products);
        for(int i=0;i<products.length;i++) list.add(products[i]);
        
        for(int i=0;i<searchWord.length();i++){
            String prefix = searchWord.substring(0, i+1);
            List<String> tmp = new ArrayList<>();
            List<String> match = new ArrayList<>();
            for(int j=0;j<list.size();j++){
                String s = list.get(j);
                if(s.startsWith(prefix)){
                    tmp.add(s);
                    if(match.size() < 3) match.add(s);
                } 
            }
            res.add(match);
            list = tmp;
        }
        return res;
    }
}
