class Solution {
        public List<List<String>> suggestedProducts(String[] products, String searchWord) {
            int l = 0, r = products.length - 1;
            Arrays.sort(products);
            List<List<String>> ret = new ArrayList<>();
            for(int charpos = 0; charpos < searchWord.length(); charpos++){
                while (l <= r && (products[l].length() - 1 < charpos || products[l].charAt(charpos) != searchWord.charAt(charpos))){
                    l++;
                }
                while (l <= r && (products[r].length() - 1 < charpos || products[r].charAt(charpos) != searchWord.charAt(charpos))){
                    r--;
                }
                List<String> hell = new ArrayList<>();
                if (l <= r){
                    for(int i = l; (i <= l + 2 && i <= r); i++){
                        hell.add(products[i]);
                    }
                }
                ret.add(hell);
            }
            return ret;
        }
    }
