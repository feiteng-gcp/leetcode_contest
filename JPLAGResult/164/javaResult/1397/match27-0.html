class Solution {
    class TrieNode{
        TrieNode[] map = new TrieNode[26];
        String s = "#";
        int num = 0;
    }
    TrieNode root = new TrieNode();
    public void insert(String word){
        TrieNode cur = root;
        for(char c:word.toCharArray()){
            if(cur.map[c-'a']==null){
                cur.map[c-'a'] = new TrieNode();
            }
            cur = cur.map[c-'a'];
        }
        cur.s = word;
        cur.num++;
    }
    public List<String> search(String prev){
        TrieNode cur = root;
        for(char c:prev.toCharArray()){
            if(cur.map[c-'a']!=null){
                cur = cur.map[c-'a'];
            }else{
                return new ArrayList<>();
            }
        }
        List<String> res = new ArrayList<>();
        searchWords(cur, res);
        return res;
    }
    
    public void searchWords(TrieNode cur, List<String> res){
        if(res.size()>=3 || cur==null) return;
        if(!cur.s.equals("#")){
            int n = 0;
            while(res.size()<3 && n++<cur.num){
                res.add(cur.s);
            }
        } 
        for(int i=0;i<26;i++){
            searchWords(cur.map[i],res);
        }
    }
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        List<List<String>> res = new ArrayList<>();
        for(String w:products){
            insert(w);
        }
        String cur = "";
        for(char c:searchWord.toCharArray()){
            cur+=c;
            res.add(search(cur));
        }
        return res;
    }
}
