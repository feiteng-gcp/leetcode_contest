class Solution {
    class Tnode {
        Tnode[] chd = new Tnode[26];
        ArrayList<String> set = new ArrayList<>();
        void add(String x, String cs) {
            set.add(cs);
            Collections.sort(set);
            if(set.size() > 3) set.remove(set.size() - 1);
            
if(x.length() == 0) return;
            char fc = x.charAt(0);
            if(chd[fc - 'a'] == null) chd[fc - 'a'] = new Tnode();
            chd[fc - 'a'].add(x.substring(1), cs);
        
}
    }
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        Tnode root = new Tnode();
        for(String x : products) root.add(x, x);
        List<List<String>> res = new ArrayList<>();
        for(char x : searchWord.toCharArray()) {
            if(root == null) res.add(new ArrayList<>());
            else {
                root = root.chd[x - 'a'];
                if(root != null) res.add(new ArrayList<>(root.set));
                else res.add(new ArrayList<>())
;
            }
        }
        return res;
    }
}
