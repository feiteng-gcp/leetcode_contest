class Solution {
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        List<List<String>> res = new ArrayList<>();
        int len = searchWord.length();
        Arrays.sort(products, (a, b) -> a.compareTo(b));
        for (int i = 1; i <= len; i++) 
{
            String term = searchWord.substring(0, i);
            res.add(query(products, term));
        }
        return res;
    }
    
    private List<String> query(String[] arr, String term) {
        List<String> res = new ArrayList<>();
        int count = 0; 
        
for (String prod : arr) {
            if (prod.startsWith(term) && count < 3) {
                res.add(prod);
                count++;
            }
        }
        return res;
    }
}
