class Solution {
    public List<List<String>> suggestedProducts(String[] ps, String sw) {
        
        Arrays.sort(ps);
        
        List<String> as = new ArrayList<>();
        
        for(String ss:ps){
            as.add(ss);
        }
        List<List<String>> tot = new ArrayList<>();
        
        for(int i=0;i<sw.length();++i){
            char c = sw.charAt(i);
            ArrayList<String> ans = new ArrayList<>();
            for(int j=0;j<as.size();++j){
                String cur = as.get(j);
                if(i<cur.length()&&cur.charAt(i)==c){
                    ans.add(cur);
                }
            }
            
            if(ans.size()>3){
                
                tot.add((List<String>)ans.subList(0,3));
            }else{
                tot.add((List<String>)ans.clone());
            }
            
            as = ans;
        }
        
        return tot;
        
        
    }
}
