class Solution {
    public List<List<String>> suggestedProducts(String[] products,
			String searchWord) {
		List<List<String>> ans = new ArrayList<List<String>>(
				searchWord.length());
		Arrays.sort(pr
oducts);
		List<String> list = new ArrayList<>(products.length);
		for (int i = 0; i < products.length; i++) {
			list.add(products[i]);
		}
		for (int i = 1
; i <= searchWord.length(); i++) {
			String c = searchWord.substring(0, i);
			List<String> aa = new ArrayList<String>();
			for (String ss : list) {
				if(ss.length()<i){
					continue;
				}
				if (ss.substring(0, i).equal
s(c)) {
					aa.add(ss);
				}
			}
			ans.add(aa.size() > 3 ? aa.subList(0, 3) : aa);
			list = new ArrayList<String>(aa);
		}
		return ans;
	}
}
