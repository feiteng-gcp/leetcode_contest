class Solution {
    
    class Trie{
        Trie[] tries = new Trie[26];
        
        TreeMap<String,Integer> map = new TreeMap<>();
        
        void insert(String product, int index) {
            int cnt = map.getOrDefault(product,0);
            map.put(product,cnt+1);
            if(index == product.length()) return;
            char c = product.charAt(index);
            if(tries[c - 'a'] == null) tries[c- 'a'] = new Trie();
            tries[c - 'a'].insert(product,index+1);
        }
        
        List<String> getTop3() {
            List<String> res = new ArrayList<>();
            for(Map.Entry<String, Integer> entry : map.entrySet()) {
                for(int i = 0 ; i < entry.getValue() && res.size() <3;i++) {
                    res.add(entry.getKey());
                }
                if(res.size() >= 3) break;
            }
            return res;
        }
    }
    public List<List<String>> suggestedProducts(String[] products, String searchWord) {
        Trie trie = new Trie();
        for(String str : products) trie.insert(str,0);
        
        List<List<String>> result = new ArrayList<>();
        for(char c : searchWord.toCharArray()) {
            if(trie == null) result.add(new ArrayList<>());
            else{
                trie = trie.tries[c-'a'];
                if(trie == null) result.add(new ArrayList<>());
                else result.add(trie.getTop3());
            }
        }
        return result;
    }
}
