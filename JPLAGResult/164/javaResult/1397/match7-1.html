class Solution {

	public List<List<String>> suggestedProducts(String[] products, String searchWord) {
		Trie root = new Trie();
		for (String product : products) {
			Trie node = root;
			for (char c : product.toCharArray()) {
				if (node.children[c - 'a'] == null) {
					node.children[c - 'a'] = new Trie();
				}
				(node = node.children[c - 'a']).list.add(product);
			}

		}
		ArrayList<List<String>> result = new ArrayList<>();
		for (char c : searchWord.toCharArray()) {
			root = root == null ? null : root.children[c - 'a'];
			result.add(new ArrayL
ist<>());
			if (root != null) {
				Collections.sort(root.list);
				for (int i = 0; i < Math.min(3, root.list.size()); i++) {
					result.get(result.size() - 1).add(root.list.get(i));
				}
			}
		}
		return result;
	}

	private class Trie {
		private ArrayList<String> list = new ArrayList<>();
		private Trie[] children = new Trie[26];
	}
}
