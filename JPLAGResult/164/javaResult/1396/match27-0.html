class Solution {
    public int countServers(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[] rowCnt = new int[m];
        int[] colCnt = new int[n];
        for(int i = 0 ; i < m ; i++){
            for(int j = 0 ; j < n; j++){
                if(grid[i][j] == 1){
                    rowCnt[i]++;
                }
            }
        }
        for(int i = 0 ; i < n ; i++){
            for(int j = 0 ; j < m; j++){
                if(grid[j][i] == 1){
                    colCnt[i]++;
                }
            }
        }
        int res = 0;
        for(int i = 0 ; i < m ; i++){
            for(int j = 0 ; j < n ; j++){
             //  System.out.println(i + " " + j + " " +grid[i][j] + " " + rowCnt[i] + " " + rowCnt[j]);
                if(grid[i][j] == 1 && (rowCnt[i] > 1 || colCnt[j] > 1)){
                    res++;
                }
            }
        }
        return res;
    }
}
