class Solution {
    public int countServers(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        int[] rc = new int[n];
        int[] cc = new int[m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                rc[i] += grid[i][j];
                cc[j] += grid[i][j];
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j]==1 && (rc[i]>1||cc[j]>1))
                    ans++;
            }
        }
        return ans;
    }
}
