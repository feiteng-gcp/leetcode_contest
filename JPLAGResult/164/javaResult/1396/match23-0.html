class Solution {
    public int countServers(int[][] grid) {
        
		int row = grid.length;
		int col = grid[0].length;
		boolean[][] tag = new boolean[row][col];
		// 行搜索
		for (int i = 0; i < row; i++) {
			int count = 0;
			for (int j = 0; j < col; j++) {
				if (grid[i][j] == 1) {
					count++;
				}
			}
			if (count >= 2) {
				for (int j = 0; j < col; j++) {
					if (grid[i][j] == 1) {
						tag[i][j] = true;
					}
				}
			}
		}
		// 列搜索
		for (int i = 0; i < col; i++) {
			int count = 0;
			for (int j = 0; j < row; j++) {
				if (grid[j][i] == 1) {
					count++;
				}
			}
			if (count >= 2) {
				for (int j = 0; j < row; j++) {
					if (grid[j][i] == 1) {
						tag[j][i] = true;
					}
				}
			}
		}
		int result = 0;
		for(boolean[] t : tag) {
			for(boolean b:t) {
				if(b) {
					result++;
				}
			}
		}
		return result;
	
    }
}
