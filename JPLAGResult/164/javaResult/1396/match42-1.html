class Solution {
    public int countServers(int[][] grid) {
        int R = grid.length;
        int C = grid[0].length;
        int[] rowCount = new int[R];
        int[] colCount = new int[C];
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (grid[i][j] == 1) {
                    rowCount[i]++;
                    colCount[j]++;
                }
            
}
        }
        int servers = 0;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (grid[i][j] == 1 && rowCount[i] + colCount[j] - 1 > 1) {
                    servers++;
                
}
            }
        }
        return servers;
    }
}
