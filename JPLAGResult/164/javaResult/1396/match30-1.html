class Solution {
    public int countServers(int[][] grid) {
		int[] rN = new int[grid.length];
		int[] cN = new int[grid[0].length];
		for (int i = 0; i < grid.length; i++) {
			for (int j = 0; j < grid[0].length; j++) {
				rN[i] += grid[i][j];
				cN[j] += grid[i][j];
			}
		}
		int ans = 0;
		for (int i = 0; i < grid.length; i++) {
			for (int j = 0; j < grid[0].length; j++) {
				if (grid[i][j] == 1) {
					if (rN[i] > 1 || cN[j] > 1) {
						ans++;
					}
				}

			}
		}
		return ans;
	}
}
