class Solution {
    public int countServers(int[][] grid) {
        int M = grid.length, N = grid[0].length;
        int[] row = new int[M];
        int[] col = new int[N];
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                if (grid[i][j] == 1) {
                    row[i]++;
                    col[j]++;
                }
            }
        }
        int res = 0;
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                if (grid[i][j] == 1 && (row[i] > 1 || col[j] > 1)) {
                        res++;
                }
            }
        }
        return res;
    }
}
