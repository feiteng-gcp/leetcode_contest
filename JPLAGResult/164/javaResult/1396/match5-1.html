class Solution {
    public int countServers(int[][] grid) {
        int[] columns = new int[grid[0].length];
        int[] raws = new int[grid.length];

        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[i].length; j++) {
                raws[i] += grid[i][j];
                columns[j] += grid[i][j];
            }
        }

        int count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[i].length; j++) {
                if (grid[i][j] == 1 && (raws[i] >= 2 || columns[j] >= 2)) {
                    count++;
                }
            }
        }
        return count;
    }
}
