class Solution {
    public int countServers(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[] raw = new int[m];
        int[] col = new int[n];
        for(int i=0;i<grid.length;++i){
            int count = 0;
            for(int j=0;j<grid[0].length;++j){
                if(grid[i][j]==1)
                    ++count;
            }
            raw[i] = count;
        }
        for(int i=0;i<grid[0].length;++i){
            int count = 0;
            for(int j=0;j<grid.length;++j){
                if(grid[j][i]==1)
                    ++count;
            }
            col[i] = count;
        }
        int ans = 0;
        for(int i=0;i<m;++i){
            for(int j=0;j<n;++j){
                if(grid[i][j] == 1){
                    if(raw[i]>1||col[j]>1)
                        ++ans;
                }
            }
        }
        return ans;
    }
}
