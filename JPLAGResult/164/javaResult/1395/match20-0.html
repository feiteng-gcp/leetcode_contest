class Solution {
    public int minTimeToVisitAllPoints(int[][] points) {
        int ans = 0;
        int x = points[0][0];
        int y = points[0][1];
        for (int i = 1; i < points.length; i++) {
            int diffx = Math.abs(points[i][0] - x);
            int diffy = Math.abs(points[i][1] - y);
            int com = Math.min(diffx, diffy);
            ans += (diffx + diffy - com);
            
            x = points[i][0];
            y = points[i][1];
        }
        return ans;
    }
}
