class Solution {
    public int minTimeToVisitAllPoints(int[][] points) {
        int result = 0;
        for(int i = 1; i < points.length; i++){
            int[] point2 = points[i];
            int[] point1 = points[i - 1];
            result += Math.max(Math.abs(point1[0] - point2[0]), Math.abs(point1[1] - point2[1]));
        }
        return result;
    }
}
