class Solution {
    public int minTimeToVisitAllPoints(int[][] points) {
        int startX = points[0][0];
        int startY = points[0][1];
        int res = 0;
        
        for (int i = 1; i < points.length; i++) {
            int runX = Math.abs(startX - points[i][0]);
            int runY = Math.abs(startY - points[i][1]);
            
            int diag = Math.min(runX, runY);
            runX -= diag;
            runY -= diag;
            res += diag;
            res += runX;
            res += runY;
            startX = points[i][0];
            startY = points[i][1];
        }
        return res;
    }
}
