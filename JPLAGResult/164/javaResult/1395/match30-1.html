class Solution {
    public int minTimeToVisitAllPoints(int[][] points) {
        int len = points.length, res = 0;
        for (int i = 1; i < len; ++i) {
            int[] cur = points[i - 1], next = points[i];
            int xdif = Math.abs(cur[0] - next[0]), ydif = 
Math.abs(cur[1] - next[1]);
            res += Math.min(xdif, ydif) + Math.abs(xdif - ydif);
        }
        return res;
    }
}
