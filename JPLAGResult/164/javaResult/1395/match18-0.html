class Solution {
    public int minTimeToVisitAllPoints(int[][] points) {
        int sum = 0;
        for(int i = 1;i<points.length;i++){
            sum += help(points[i-1],points[i]);
        }
        return sum;
    }
    
    int help(int[] a, int[] b){
        int x = Math.abs(a[0] - b[0]);
        int y = Math.abs(a[1] - b[1]);
        return Math.min(x,y) + Math.abs(x-y);
    }
}
