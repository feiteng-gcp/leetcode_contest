class Solution {
public:
    #define MOD 1000000007;
    int numWays(int steps, int arrLen) 
    {
        vector<vector<long> > table(steps + 1, vector<long>(min(steps + 1, arrLen), 0));
        
        table[0][0] = 1;
        
        for(int s = 1; s <= steps; s += 1)
        {
            for(int i = 0; i < table[0].size(); i += 1)
            {
                table[s][i] = ((i == 0 ? 0 : table[s - 1][i - 1]) + table[s - 1][i] + (i == table[0].size() - 1 ? 0 : table[s - 1][i + 1])) % MOD;
            }
        }
        
        return table[steps][0];
    }
};
