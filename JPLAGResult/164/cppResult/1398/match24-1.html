class Solution {
public:
    int numWays(int steps, int arrLen) {
        const int MOD = 1000000007;
        int len = min(steps + 1, arrLen);
        vector<vector<int>> dp(2, vector<int>(len, 0));

        dp[0][0] = 1;
        for(int i = 1; i <= steps; i++){
            for(int j = 0; j < len; j++){
                dp[i%2][j] = (((j > 0 ? dp[(i-1)%2][j-1] : 0)+ dp[(i-1)%2][j]) % MOD + (j < len - 1 ? dp[(i-1)%2][j+1] : 0)) % MOD;
            }
        }
        return dp[steps%2][0];
    }
};
