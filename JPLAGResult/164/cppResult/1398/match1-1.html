class Solution {
public:
    int a_ = 0;
    vector<vector<int>> v;
    long M = 1E9+7;
    long search(int s, int pos) {
        if(pos >= a_ || pos < 0) return 0;
        if(v[s][pos] != -1) {
            return v[s][pos];
        }
        if(s == 1) {
            if(pos == 0 || pos == 1) {
                return v[s][pos] = 1;
            }
            return v[s][pos] = 0;
        }
        return v[s][pos] = (0L + search(s-1, pos) + search(s-1, pos+1) + search(s-1, pos-1)) % M;
    }
    int numWays(int steps, int arrLen) {a_ = min(steps+1, arrLen);v.resize(steps+1);for(int i = 0; i < v.size(); i++) {v[i].resize(a_+1);
            for(int j = 0; j < v[i].size(); j++) {
                v[i][j] = -1;
            }
        }
        return search(steps, 0);
    }
};
