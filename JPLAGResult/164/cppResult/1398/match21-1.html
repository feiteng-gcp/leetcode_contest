int dp[501][501];

int M = 1e9 + 7;
class Solution {
public:
    int numWays(int steps, int arrLen) {
        memset(dp, 0, sizeof(dp));
        dp[0][0] = 1;
        for (int i = 0; i < 500; i++) {
            for (int j = 0; j < arrLen && j < 501; j++) {
                if (j > 0) {
                    dp[i + 1][j - 1] += dp[i][j];
                    dp[i + 1][j - 1] %= M;
                }
                dp[i + 1][j] += dp[i][j];
                dp[i + 1][j] %= M;
                if (j < arrLen - 1 && j < 500) {
                    dp[i + 1][j + 1] += dp[i][j];
                    dp[i + 1][j + 1] %= M;
                }
            }
        }
        return dp[steps][0];
    }
};
