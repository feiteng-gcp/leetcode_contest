const ll mod = 1000000007;

const int maxn = 250 + 10;
ll dp[maxn][2*maxn];// index, step

class Solution {
public:
    int numWays(int T, int l) {
        mset(dp,0);
        dp[0][0] = 1;
        for(int t=1;t<=T;t++){
            for(int i=0;i<min(l,maxn);i++){
                dp[i][t]=dp[i][t-1];
                if(i-1>=0) {
                    dp[i][t]+=dp[i-1][t-1];
                    dp[i][t]%=mod;
                }
                if(i+1<min(l,maxn)){
                    dp[i][t]+=dp[i+1][t-1];
                    dp[i][t]%=mod;
                }
            }
        }
        return (int) dp[0][T];
    }
};
