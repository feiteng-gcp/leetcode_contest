class Solution {
public:
    
    int numWays(int steps, int arrLen) {
        vector<vector<int>> dp(2 , vector<int> (arrLen + 1 , 0)) ;
        dp[0][0] = 1 ;
        int t = 0 ;
        int mod = 1000000007 ;

        for (int s = 0 ; s < steps ; ++s) {
            for (int i = 0 ; i < min(arrLen , steps / 2 + 1) ; ++i) {
                dp[t ^ 1][i] = dp[t][i] ;
                if (i - 1 >= 0) {
                    dp[t ^ 1][i] = (dp[t ^ 1][i] + dp[t][i - 1]) % mod ;
                }
                if (i + 1 < arrLen) {
                    dp[t ^ 1][i] = (dp[t ^ 1][i] + dp[t][i + 1]) % mod ;
                }
            }
            t ^= 1 ;
        }
        return dp[t][0] ;
        return 0 ;
    }
};
