class Solution {
public:
    
    int numWays(int steps, int arrLen) {
        int **f;
        int mod = 1000000007;
        int n = steps;
        int m = min(steps + 1, arrLen);
        f = new int*[n + 1];
        for (int i = 0; i <= n; ++i) {
            f[i] = new int[m];
            for (int j = 0; j < m; ++j) {
                f[i][j] = 0;
            }
        }
        f[0][0] = 1;
        for (int i = 1; i <= n; ++i) {
            for (int j = 0; j < m; ++j) {
                f[i][j] = f[i - 1][j];
                if (j > 0) {
                    f[i][j] += f[i - 1][j - 1];
                    f[i][j] %= mod;
                }
                if (j < m - 1) {
                    f[i][j] += f[i - 1][j + 1];
                    f[i][j] %= mod;
                }
                //cout << i << ' ' << j << ' ' << f[i][j] << endl;
            }
        }
        return f[n][0];
    }
};
