class Solution {
public:
    int numWays(int steps, int arrLen) {
        arrLen=min(arrLen,steps+10);
        vector<vector<int>> dp(2,vector<int>(arrLen,0));
        int last=0,cur=1;

        int mod=(1e9)+7;
        dp[cur][0]=1;
        for(int i=0;i<steps;i++){
            swap(cur,last);
            for(int i=0;i<arrLen;i++){

                dp[cur][i]=dp[last][i];
                if(i-1>=0){
                    dp[cur][i]+=dp[last][i-1];
                    dp[cur][i]%=mod;
                }
                if(i+1<arrLen){
                    dp[cur][i]+=dp[last][i+1];
                    dp[cur][i]%=mod;
                }
            }
        }
        return dp[cur][0];
    }
};
