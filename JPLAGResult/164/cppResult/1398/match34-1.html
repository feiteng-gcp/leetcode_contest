class Solution {
public:
    int numWays(int steps, int arrLen) {
        int mod = 1e9+7;
        int len = min(steps, arrLen);
        vector<vector<int>> dp(steps+1, vector<int>(len));

        dp[0][0] = 1;
        for (int i = 1; i <= steps; i++) {
            for (int j = 0; j < len; j++) {
                dp[i][j] += dp[i-1][j];
                if (j > 0) {
                    dp[i][j] += dp[i-1][j-1];
                    dp[i][j] %= mod;
                }
                if (j+1 < len) {
                    dp[i][j] += dp[i-1][j+1];
                    dp[i][j] %= mod;
                }
            }
        }
        return dp[steps][0];
    }
};
