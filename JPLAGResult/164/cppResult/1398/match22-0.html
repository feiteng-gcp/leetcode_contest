class Solution {
    const int mod = 1000000007;
public:
    int numWays(int steps, int n) {
        if(n >= steps) n = steps;
        
        vector<vector<long long>> dp(steps+1, vector<long long>(n, 0));
        dp[0][0] = 1;
        for(int i = 1; i <= steps; i ++)
        {
            for(int j = 0; j < n; j ++)
            {
                dp[i][j] = dp[i-1][j];
                
                if(j > 0)
                {
                    dp[i][j] += dp[i-1][j-1];
                }
                
                if(j < n-1)
                {
                    dp[i][j] += dp[i-1][j+1];
                }
                
                dp[i][j] %= mod;
            }
        }
        
        return dp[steps][0] % mod;
    }
};
