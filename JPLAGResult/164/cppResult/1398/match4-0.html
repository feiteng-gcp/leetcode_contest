class Solution {
public:
    int f[505][505];
    int numWays(int n, int arrLen) {
        int mod = 1000000007;
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n; j++) {
                f[i][j] = 0;
            }
        }
        
        f[0][0] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= n; j++) {
                (f[i + 1][j] += f[i][j]) %= mod;
                if (j > 0) {
                    (f[i + 1][j - 1] += f[i][j]) %= mod;
                }
                
                if (j + 1 < arrLen) {
                    (f[i + 1][j + 1] += f[i][j]) %= mod;
                }
            }
        }
        
        return f[n][0];
    }
};
