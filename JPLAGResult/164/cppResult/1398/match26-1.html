long long dp[700][510];
class Solution {
public:
    int numWays(int steps, int arrLen) {
        memset(dp,0,sizeof dp);
        dp[0][0]=1;

        int mxs = min(arrLen,600);
        for(int i=1; i<=steps; i++){
            for(int pos=0; pos < mxs ; pos++){
                dp[pos][i] += dp[pos][i-1];
                if(pos) dp[pos][i] += dp[pos-1][i-1];
                if( pos < mxs-1) dp[pos][i] += dp[pos+1][i-1];
                dp[pos][i] %= 1000000007;
            }
        }
        return dp[0][steps];
    }
};
