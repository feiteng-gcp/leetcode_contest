const int mod = 1e9+7;
class Solution {
public:
    int dp[1000005][2];
    int numWays(int steps, int arrLen) {
        dp[0][0]=1;
        int now = 1;
        int next = 0;

        arrLen = min(1000,arrLen);
        for(int i=1;i<=steps;i++){
            for(int j=0;j<arrLen;j++){
                dp[j][now]=dp[j][next];
                if(j>0){
                    dp[j][now]=(dp[j][now]+dp[j-1][next])%mod;
                }
                if(j!=arrLen-1){
                    dp[j][now]=(dp[j][now]+dp[j+1][next])%mod;
                }
            }
            swap(now,next);
        }
        return dp[0][next];
    }
};
