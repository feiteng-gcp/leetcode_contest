class Solution {
public:
    long long BIG = 1e9 + 7;
    long long dp[569][569];
    
    int numWays(int steps, int arrLen) {
        memset(dp, 0, sizeof dp);
        dp[0][0] = 1;
        for(int i = 1; i <= steps; i++) {
            for(int j = 1; j < min(517, arrLen - 1); j++) {
                dp[i][j] = (dp[i-1][j-1] + dp[i-1][j] + dp[i-1][j+1]) % BIG;
            }
            dp[i][0] = (dp[i-1][0] + dp[i-1][1]) % BIG;
            
            if(arrLen < 569) {
                dp[i][arrLen - 1] = (dp[i-1][arrLen - 2] + dp[i-1][arrLen - 1]) % BIG;
            }
        }
        return (int)dp[steps][0];
    }
};
