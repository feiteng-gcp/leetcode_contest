class Solution {

private:
    int MOD = 1e9 + 7;

public:
    int numWays(int steps, int arrLen) {

        vector<vector<int>> dp(501, vector<int>(501, -1));
        return dfs(0, steps, arrLen, dp);
    }

private:
    int dfs(int pos, int step, int arrlen, vector<vector<int>>& dp){

        if(step == 0) return pos == 0;
        if(dp[pos][step] != -1) return dp[pos][step];

        int res = dfs(pos, step - 1, arrlen, dp);
        if(pos + 1 < arrlen) res = (res + dfs(pos + 1, step - 1, arrlen, dp)) % MOD;
        if(pos) res = (res + dfs(pos - 1, step - 1, arrlen, dp)) % MOD;
        return dp[pos][step] = res;
    }
};
