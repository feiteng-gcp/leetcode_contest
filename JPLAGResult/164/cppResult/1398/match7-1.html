class Solution {
public:
    int numWays(int steps, int arrLen) {
        int mod = 1e9 + 7;
        arrLen = min(arrLen, steps);
        vector<int> dp(arrLen, 0);
        dp[0] = 1;
        for (int i = 0; i < steps; ++i) {
            vector<int> next(dp);
            for (int j = 0; j < arrLen; ++j) {
                next[j] = (next[j] + (j - 1 >= 0 ? dp[j - 1] : 0)) % mod;
                next[j] = (next[j] + (j + 1 < arrLen ? dp[j + 1] : 0)) % mod;
            }
            dp.swap(next);
        }
        return dp[0];
    }
};
