using ll = long long;
const int MOD = 1000000007;

class Solution {
public:
    int numWays(int steps, int arrLen) 
    {
        vector<vector<ll>> dp(steps+10, vector<ll>(steps+10, 0));
        dp[0][0] = 1;
        for (int i=0; i<steps; ++i)
            for (int j=0; j<steps; ++j)
            {
                dp[i+1][j] += dp[i][j];
                dp[i+1][j] %= MOD;

                if (j > 0)
                {
                    dp[i+1][j-1] += dp[i][j];
                    dp[i+1][j-1] %= MOD;                
                }
                if (j < arrLen-1)
                {
                    dp[i+1][j+1] += dp[i][j];
                    dp[i+1][j+1] %= MOD;
                }
            }
        
        // for (int i=0; i<steps; ++i)
        //     for (int j=0; j<steps; ++j)
        //         cout << i << ' ' << j << ' ' << dp[i][j] << endl;

        return dp[steps][0];        
    }
};
