class Solution {
public:
#define ll long long
#define P ((ll)1e9+7)
    ll f[505][505];
    int numWays(int n, int m) {
        f[0][1] = 1;
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= min(n, m); j++){
                f[i][j] = ((j>1?f[i-1][j-1]:0)+f[i-1][j]+(j<n?f[i-1][j+1]:0))%P;
            }
        }
        return f[n][1];
    }
};
