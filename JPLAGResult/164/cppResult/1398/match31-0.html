class Solution {
public:
    int dp[2][1000001];
    const int mod = 1e9+7;
    int numWays(int steps, int arrLen) {
        memset(dp,0,sizeof(dp));
        dp[0][0]=1;
        int step = steps;
        int ulen = min(step,arrLen);
        for(int i = 1; i <= step; i++)
        {
            if(i&1)
            {
                for(int j = 0; j <ulen ; j++)
                {
                    dp[1][j] = dp[0][j];
                    if(j - 1 >= 0)
                        dp[1][j] = (dp[1][j] + dp[0][j-1])%mod;
                    if(j+1 < ulen)
                        dp[1][j] = (dp[1][j] + dp[0][j+1])%mod;
                }
            }
            else
            {
                for(int j = 0; j < ulen; j++)
                {
                    dp[0][j] = dp[1][j];
                    if(j - 1 >= 0)
                        dp[0][j] = (dp[0][j] + dp[1][j-1])%mod;
                    if(j+1 < ulen)
                        dp[0][j] = (dp[0][j] + dp[1][j+1])%mod;
                }
            }
            
            
        }
        if(step&1)
            return dp[1][0];
        else
            return dp[0][0];
    }
};
