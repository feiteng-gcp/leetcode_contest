class Solution {
public:
    int numWays(int steps, int arrLen) {
        int iMax = min(steps, arrLen), mod = 1e9+7;
        vector<int> dp(iMax, 0);
        dp[0] = 1;
        for (int i=0; i<steps; i++) {
            vector<int> temp(dp);
            for (int i=1; i<iMax; i++)
                temp[i] = (temp[i] + dp[i-1]) % mod;
            for (int i=iMax-2; i>=0; i--)
                temp[i] = (temp[i] + dp[i+1]) % mod;
            swap(dp, temp);
        }
        return dp[0];
    }
};
