long long dp[550][550];
const int mod = 1e9+7;
class Solution {
public:
    int numWays(int steps, int arrLen) {
        memset(dp, 0, sizeof( dp));

        dp[0][0] = 1;
        for(int i=1;i<=steps;i++){
            for(int j=0;j<=min(510,arrLen);j++){
                dp[i][j]+=dp[i-1][j];
                dp[i][j]%=mod;
                if(j>0){
                    dp[i][j]+=dp[i-1][j-1];
                    dp[i][j]%=mod;
                }
                if(j+1<=arrLen-1){
                    dp[i][j]+=dp[i-1][j+1];
                    dp[i][j]%=mod;
                }
            }
        }
        return dp[steps][0];
    }
};
