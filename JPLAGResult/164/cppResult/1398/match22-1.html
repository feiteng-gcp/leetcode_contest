typedef long long LL;
class Solution {
public:
    int numWays(int steps, int arrLen) {
        LL BASE=pow(10,9)+7;

        arrLen=min(steps+1,arrLen);
        vector<vector<LL>> dp(steps+1,vector<LL>(arrLen));

        dp[0][0]=1;
        
        for(int s=1;s<=steps;s++){
            
            for(int i=0;i<arrLen;i++){
                dp[s][i]=dp[s-1][i];
                if(i>0){
                    dp[s][i]+=dp[s-1][i-1];
                }
                if(i<arrLen-1){
                    dp[s][i]+=dp[s-1][i+1];
                }
                dp[s][i]%=BASE;
            }
        }
        
        return dp[steps][0];
    }
};
