class Solution {
public:
    int dp[550][550];
    const int mod=1e9+7;

    int numWays(int steps, int arrLen) {
        memset(dp,0,sizeof(dp));
        dp[0][0]=1;
        for(int i=1;i<=steps;i++)
        {
            int ml=min(arrLen-1,i);
            for(int j=0;j<=ml;j++)
            {
                dp[i][j]=dp[i-1][j];
                if(j-1>=0) dp[i][j]=(dp[i][j]+dp[i-1][j-1])%mod;
                if(j+1<=ml) dp[i][j]=(dp[i][j]+dp[i-1][j+1])%mod;
            }
        }
        return dp[steps][0];
    }
};
