const int mod=1e9+7;
long dp[2][2000010];
long* a=dp[0];
long* b=dp[1];
class Solution {
public:
    int numWays(int steps, int arrLen) {
        memset(dp, 0, sizeof(dp));
        a[0]=1;
        for(int i=0; i<steps; ++i) {
            int k=i+1;
            fill(b, b+k+1, 0);
            for(int j=0; j<=k; ++j) {
                b[j]+=a[j];
                if (j>0) b[j]+=a[j-1];
                if (j+1<min(k, arrLen)) b[j]+=a[j+1];
                for(int j=0; j<=k; ++j) b[j]%=mod;
            }
            swap(a, b);
        }
        return a[0];
    }
};
