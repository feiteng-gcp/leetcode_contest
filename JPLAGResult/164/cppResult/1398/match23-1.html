class Solution {
public:
    vector<vector<long long>> dp;
    int mod = 1e9 + 7;
    int numWays(int n, int m) {
        if (m > n) {
            m = n;
        }
        dp.resize(n + 1);

        for (int i = 0; i <= n; i++) {
            dp[i].assign(m, 0);
        }
        dp[0][0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < m; j++) {
                dp[i][j] += dp[i-1][j];
                if (j - 1 >= 0)
                    dp[i][j] += dp[i-1][j-1];
                if (j + 1 < m)
                    dp[i][j] += dp[i-1][j+1];
                dp[i][j] %= mod;
            }
        }
        return dp[n][0];
    }
};
