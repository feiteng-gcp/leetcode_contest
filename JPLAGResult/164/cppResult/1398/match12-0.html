class Solution {
public:
    long long mod = (1e9)+7;
    long long dis[505][505];
    int numWays(int steps, int arrLen) {
        memset(dis,0,sizeof dis);
        dis[0][0]=1;
        for(int i=1;i<=steps;i++)
        {
            for(int j=0;j<arrLen && j<=steps/2;j++)
            {
                if(j==0) dis[i][j]=(dis[i-1][j]+dis[i-1][j+1])%mod;
                else dis[i][j]=(dis[i-1][j-1]+dis[i-1][j]+dis[i-1][j+1])%mod;
            }
        }
        return dis[steps][0];
    }
};
