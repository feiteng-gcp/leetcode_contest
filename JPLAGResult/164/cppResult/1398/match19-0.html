int dp[2][505];
const int mod=1000000007;
class Solution {
public:
    int numWays(int steps, int arrLen) {
        memset(dp,0,sizeof(dp));
        dp[0][0]=1;
        for(int i=1;i<=steps;i++)
        {
            for(int j=0;j<min(arrLen,501);j++)
            {
                dp[i%2][j]=dp[(i-1)%2][j];
                if(j)dp[i%2][j]=(dp[i%2][j]+dp[(i-1)%2][j-1])%mod;
                if(j<min(arrLen,501)-1)dp[i%2][j]=(dp[i%2][j]+dp[(i-1)%2][j+1])%mod;
            }
        }
        return dp[steps%2][0];
    }
};
