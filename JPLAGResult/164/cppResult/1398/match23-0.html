#include<vector>
using namespace std;
#define ll long long
#define MOD 1000000007
class Solution {
public:
    int numWays(int steps, int arrLen) {
        arrLen = min(steps+1,arrLen);
        vector<vector<ll> > table;
        for(int i=0;i<2;i++){
            vector<ll> row(arrLen,0);
            table.push_back(row);
        }
        table[0][0] = 1;
        for(int s=1;s<=steps;s++){
            for(int i=0;i<arrLen;i++){
                table[s&1][i] = table[(s&1)^1][i];
                if(i>0)table[s&1][i] += table[(s&1)^1][i-1];
                if(i+1<arrLen)table[s&1][i] += table[(s&1)^1][i+1];
                table[s&1][i] %= MOD;
            }
        }
        return (int)table[steps&1][0];
    }
};
