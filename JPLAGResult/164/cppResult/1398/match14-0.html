class Solution {
public:
    int numWays(int steps, int arrLen) {
        constexpr int kPrime = 1e9 + 7;
        const int n = min(steps / 2 + 1, arrLen);
        vector<vector<int>> dp(steps + 1, vector<int>(n));
        dp[0][0] = 1;
        for (int i = 1; i <= steps; ++i) {
            for (int j = 0; j < n; ++j) {
                dp[i][j] = dp[i - 1][j];
                if (j > 0) {
                    dp[i][j] = (dp[i][j] + dp[i - 1][j - 1]) % kPrime;
                }
                if (j < n - 1) {
                    dp[i][j] = (dp[i][j] + dp[i - 1][j + 1]) % kPrime;
                }
            }
        }
        return dp[steps][0];
    }
};
