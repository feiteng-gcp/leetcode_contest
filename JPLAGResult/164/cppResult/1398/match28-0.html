class Solution {
public:
    int numWays(int steps, int arrLen) {
        int real_len = std::min(arrLen, steps);
        std::vector<std::vector<long long>> steps_num(steps + 1, std::vector<long long>(real_len));
        long long mod = 1e9 + 7;
        steps_num[0][0] = 1;
        for (int i = 0; i < steps; i++) {
            for (int j = 0; j < real_len; j++) {
                steps_num[i][j] %= mod;
                steps_num[i + 1][j] += steps_num[i][j];
                if (j > 0) {
                    steps_num[i + 1][j - 1] += steps_num[i][j];
                }
                if (j + 1 < real_len) {
                    steps_num[i + 1][j + 1] += steps_num[i][j];
                }
            }
        }
        return steps_num[steps][0] % mod;
    }
};
