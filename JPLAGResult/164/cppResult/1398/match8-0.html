#define K 1000000007

class Solution {
public:
    int numWays(int steps, int arrLen) {
        vector<vector<long>> dp(steps+5,vector<long>(steps+5,0));
        dp[0][0] = 1;
        for(int i=1;i<=steps;i++) {
            for(int j=0;j<=steps;j++) {
                if(j<arrLen)
                    dp[i][j] = ((j>0?dp[i-1][j-1]:0)+dp[i-1][j]+dp[i-1][j+1])%K;
            }
        }
        return dp[steps][0];
    }
};
