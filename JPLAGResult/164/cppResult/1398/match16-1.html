class Solution {
public:
    int numWays(int steps, int arrLen) {
        int mod = 1E9 + 7;
        int M = min(steps + 1, arrLen);
        vector<int> DP(M);
        DP[0] = 1;
        for (int i = 0; i < steps; ++i) {
            vector<int> prev(DP);
            for (int j = 0; j < M; ++j) {

                if (j > 0) {
                    DP[j] += prev[j - 1];
                    DP[j] %= mod;
                }
                if (j < (M - 1)) {
                    DP[j] += prev[j + 1];
                    DP[j] %= mod;
                }
            }
        }
        return DP[0];
    }
};
