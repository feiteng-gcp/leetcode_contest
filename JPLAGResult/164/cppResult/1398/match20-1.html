int dp[505][1005];
int mod = 1e9+7;
class Solution {
public:
    int numWays(int steps, int arrLen) {
        int l = min(1005, arrLen);
        for (int i = 0;i < l;i++) 
            dp[0][i] = 0;

        dp[0][0] = 1;
        for (int i = 1;i <= steps;i++) {
            for (int j = 0;j < l;j++) {
                dp[i][j] = dp[i-1][j];
                if (j > 0)
                    dp[i][j] = (dp[i][j] + dp[i-1][j-1]) % mod;
                if (j < l-1)
                    dp[i][j] = (dp[i][j] + dp[i-1][j+1]) % mod;
            }
        }
        return dp[steps][0];
    }
};
