class Solution {
    public:
    int countServers(vector<vector<int>>& g) {
        int n = g.size();
        int m = g[0].size();

        vector<int> a(n, 0);
        vector<int> b(m, 0);

        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(g[i][j] == 1) {
                    a[i] += 1;
                    b[j] += 1;
                }
            }
        }
        int ans = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(g[i][j] == 1) {
                    if(a[i] > 1 || b[j] > 1) {
                        ans += 1;
                    }
                }
            }
        }
        return ans;
    }
};
