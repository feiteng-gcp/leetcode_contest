class Solution {
public:
    int countServers(vector<vector<int>>& g) {
        int r = g.size();
        int c = g[0].size();
        vector<int> rc(r, 0);
        vector<int> cc(c, 0);
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                rc[i] += g[i][j];
                cc[j] += g[i][j];
            }
        }
        int res = 0;
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                if (g[i][j] == 1 && (rc[i] > 1 || cc[j] > 1)) res++;
            }
        }
        return res;
    }
};
