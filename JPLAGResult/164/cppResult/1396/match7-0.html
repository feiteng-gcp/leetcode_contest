class Solution {
public:
    int countServers(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<int> row_freq(n, 0);
        vector<int> col_freq(m, 0);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                row_freq[i] += grid[i][j];
                col_freq[j] += grid[i][j];
            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 1 and (row_freq[i] > 1 or col_freq[j] > 1)) {
                    ++ans;
                }
            }
        }
        return ans;
    }
};
