class Solution {
public:
    int countServers(vector<vector<int>>& grid) {
        int R=grid.size();
        int C=grid[0].size();
        vector<int> countsR(R,0);
        vector<int> countsC(C,0);
        for(int r=0;r<R;r++)
            for(int c=0;c<C;c++) {
                if (grid[r][c]) {
                    countsR[r]++;
                    countsC[c]++;
                }
            }
        int ans=0;
        for(int r=0;r<R;r++)
            for(int c=0;c<C;c++) {
                if (grid[r][c]) {
                    if (countsR[r]>1 || countsC[c]>1)
                        ans++;
                }
            }
        return ans;
    }
};
