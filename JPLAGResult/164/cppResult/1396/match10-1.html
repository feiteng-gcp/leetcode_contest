class Solution {
public:
    int countServers(vector<vector<int>>& g) {
        int m = g.size(); int n = g[0].size();
        vector<int> r, c; r.resize(m, 0); c.resize(n, 0);
        for(int i = 0; i < m; i ++){
            for(int j = 0; j < n; j ++){
                if(g[i][j]){
                    r[i]++; c[j]++;
                }
            }
        }
        int ret = 0;
        for(int i = 0; i < m; i ++){
            for(int j = 0; j < n; j ++){
                if(g[i][j] && ((r[i] > 1) || (c[j] > 1))){
                    ret ++;
                }
            }
        }
        return ret;
    }
};
