class Solution {
public:
    int countServers(vector<vector<int>>& grid) {
        int H = grid.size();
        int W = grid[0].size();
        vector<int> tate(W);
        vector<int> yoko(H);
        for (int h = 0; h < H; ++h) {
            for (int w = 0; w < W; ++w) {
                if (grid[h][w] != 0) {
                    tate[w]++;
                    yoko[h]++;
                }
            }            
        }
        int ans = 0;
        for (int h = 0; h < H; ++h) {
            for (int w = 0; w < W; ++w) {
                if (grid[h][w] != 0) {
                    if (tate[w] >= 2 || yoko[h] >= 2) {
                        ans++;
                    }
                }
            }            
        }
        return ans;
    }
};
