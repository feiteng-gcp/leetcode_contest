class Solution {
public:
    int countServers(vector<vector<int>>& grid) {
        map<int,int> R,C;
        for(int i=0;i<grid.size();i++) {
            for(int j=0;j<grid[0].size();j++) {
                if(grid[i][j] == 1) {
                    R[i]++; C[j]++;
                }
            }
        }
        
        int res = 0;
        
        for(int i=0;i<grid.size();i++) {
            for(int j=0;j<grid[0].size();j++) {
                if(grid[i][j] == 1) {
                    if(R[i] > 1 || C[j] > 1) {
                        res++;
                    }
                }
            }
        }
        
        return res;
    }
};
