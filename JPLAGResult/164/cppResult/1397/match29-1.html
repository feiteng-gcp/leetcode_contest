class Solution {
public:
    struct node{
        map<char, node*> ch;
        vector<string> v;
    };
    

    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        node* root = new node();
        for (int i = 0;i < products.size();i++) {
            node* cur = root;
            for (int j = 0;j < products[i].length();j++) {
                if (cur->ch.count(products[i][j])) {
                    cur = cur->ch[products[i][j]];
                }
                else {
                    node* tmp = new node();
                    cur->ch[products[i][j]] = tmp;
                    cur = tmp;
                }
                cur->v.push_back(products[i]);
                sort(cur->v.begin(), cur->v.end());
                if (cur->v.size() > 3)
                    cur->v.pop_back();

            }
        }
        vector<vector<string>> ret;
        node* cur = root;
        for (int i = 0;i < searchWord.length();i++) {
            if (cur->ch.count(searchWord[i])) {
                cur = cur->ch[searchWord[i]];
            }
            else {
                node* tmp = new node();
                cur->ch[searchWord[i]] = tmp;
                cur = tmp;
            }
            ret.push_back(cur->v);
        }
        return ret;
    }
};
