class Solution {
private:
    bool f(string& p, string& s, int l){
        if(p.size()<l+1)return false;
        for(int i=0;i<=l;i++)if(s[i]!=p[i])return false;
        return true;
    }
public:
    vector<vector<string>> suggestedProducts(vector<string>& p, string s) {
        vector<vector<string>> ret;
        sort(p.begin(),p.end());
        for(int i=0;i<s.size();i++){
            vector<string> tt;
            for(int j=0;j<p.size();j++){
                if(f(p[j],s,i)){
                    tt.push_back(p[j]);
                    if(tt.size()==3)break;
                }
            }
            ret.push_back(tt);
        }
        return ret;
    }
};
