class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        sort(products.begin(),products.end());
        vector<vector<string>> ans(searchWord.size());

        for(int i=0;i<searchWord.size();i++){
            auto s=searchWord.substr(0,i+1);
            int p=lower_bound(products.begin(),products.end(),s)-products.begin();
            while(ans[i].size()<3&&p<products.size()&&products[p].size()>=i+1&&products[p].substr(0,i+1)==s){
                ans[i].push_back(products[p]);
                p++;
            }
        }
        return ans;
    }
};
