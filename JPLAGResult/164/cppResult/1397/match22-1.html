class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& pr, string sw) {
        vector<vector<string>> ret;
        sort(pr.begin(), pr.end());

        string cur = "";
        for (int i = 0; i < sw.size(); i++) {
            cur += sw[i];
            vector<string> tmp;
            // vector<string> tmp;
            for (int j = 0; j < pr.size(); j++) {
                if (pr[j].find(cur) == 0) {
                    tmp.push_back(pr[j]);
                    if (tmp.size() == 3) break;
                }
            }
            ret.push_back(tmp);
        }
        return ret;
    }
};
