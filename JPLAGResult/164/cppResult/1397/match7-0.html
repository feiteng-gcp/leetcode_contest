class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& names, string w) {
        sort(names.begin(), names.end());
        vector<vector<string>> res;
        for (int i = 1; i <= w.size(); ++i) {
            vector<string> current;
            string s = w.substr(0, i);
            auto it = lower_bound(names.begin(), names.end(), s);
            for (int j = 0; j < 3 && (it + j) != names.end() && (it + j)->substr(0, s.size()) == s; ++j) {
                current.push_back(*(it + j));
            }
            res.push_back(move(current));
        }
        return res;
    }
};
