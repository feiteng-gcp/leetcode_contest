class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& a, string s) {
        sort(a.begin(), a.end());
        int l = 0, r = a.size()-1;
        vector<vector<string> > ans;
        vector<string> v;
        for(int i = 0; i < s.length(); i++){
            while(l < a.size() && (i >= a[l].length() || a[l][i] != s[i])) l++;
            while(r >= 0 && (i >= a[r].length() || a[r][i] != s[i])) r--;
            v.clear();
            for(int j = l; j <= min(r, l+2); j++){
                v.push_back(a[j]);
            }
            ans.push_back(v);
        }
        return ans;
    }
};
