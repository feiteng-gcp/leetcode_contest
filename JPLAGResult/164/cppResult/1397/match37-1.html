class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        int sz = searchWord.length();
        int psz = products.size();
        vector<vector<string>> result (sz, vector<string> ());
        sort(products.begin(), products.end());
        string tmpsub;
        int j=0;
        
        for(int i=1; i<=sz; i++)
        {
            tmpsub = searchWord.substr(0, i);
            for( ; j<psz; j++)
                if(myfind(products[j], tmpsub))
                    break;
            if(j==psz)
                break;
            result[i-1].push_back(products[j]);
            if(j+1<psz && myfind(products[j+1], tmpsub))
                result[i-1].push_back(products[j+1]);
            if(j+2<psz && myfind(products[j+2], tmpsub))
                result[i-1].push_back(products[j+2]);
        }

        return result;
    }
    
    bool myfind(string a, string b)
    {
        int sza = a.size();
        int szb = b.size();
        if(szb>sza)
            return false;
        for(int i=0; i<szb; i++)
            if(a[i]!=b[i])
                return false;
        return true;
    }
};
