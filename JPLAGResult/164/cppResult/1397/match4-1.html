using namespace std;


class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        sort(products.begin(), products.end());
        vector<vector<string>> ans;
        
        string cur;
        for (int i = 0; i < searchWord.length(); i++)
        {
            cur += searchWord[i];
            auto ptr = lower_bound(products.begin(), products.end(), cur);
            vector<string> curans;
            for (int j = 0; j < 3; j++)
            {
                if (ptr + j >= products.end() || ptr[j].find(cur, 0) != 0) break;
                curans.push_back(ptr[j]);
            }
            ans.push_back(curans);
        }
        return ans;
    }
};
