#include<vector>
#include<string>
using namespace std;
class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        sort(products.begin(),products.end());
        vector<int> prefixMatch;
        for(string s:products){
            int i=0;
            while(i<s.length() && i<searchWord.length() && searchWord[i]==s[i])i++;
            prefixMatch.push_back(i);
        }
        vector<vector<string> > ans;
        for(int l=1;l<=searchWord.length();l++){
            vector<string> row;
            for(int i=0;i<prefixMatch.size();i++){
                if(prefixMatch[i]>=l){
                    row.push_back(products[i]);
                    if(row.size()==3)break;
                }
            }
            ans.push_back(row);
        }
        return ans;
    }
};
