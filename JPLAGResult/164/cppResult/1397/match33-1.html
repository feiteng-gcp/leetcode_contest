class Solution {
    int same(string a, string b) {
        int la = a.size();
        int lb = b.size();
        int ans = 0;
        while (ans < min(la, lb)) {
            if (a[ans] != b[ans]) break;
            ans++;
        }
        return ans;
    }
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        int n = products.size();
        vector<int> num(n);
        for (int i = 0; i < n; ++i) {
            num[i] = same(products[i], searchWord);
        }

        int l = searchWord.size();
        vector<vector<string>> ans(l);
        for (int i = 0; i < l; ++i) {
            priority_queue<string> pq;
            for (int j = 0; j < n; ++j) {
                if (num[j] > i) pq.push(products[j]);
                if (pq.size() > 3) pq.pop();
            }
            stack<string> st;
            while (!pq.empty()) {
                st.push(pq.top());
                pq.pop();
            }
            while (!st.empty()) {
                ans[i].push_back(st.top());
                st.pop();
            }
        }
        return ans;
    }
};
