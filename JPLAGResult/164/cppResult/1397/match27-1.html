class Solution {
public:
    typedef unsigned long long ull;
    ull base = 23333,hs1[1010];

    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        int n=searchWord.size(),m=products.size();
        ull hs=0;
        vector<vector<string>> ans;
        for(int i=0;i<n;i++){
            vector<string>box,tmp;
            hs=hs*base+searchWord[i]-'a'+1;
            for(int j=0;j<m;j++){
                if(products[j].size()>=i+1){
                    hs1[j]=hs1[j]*base+products[j][i]-'a'+1;
                }
                if(hs1[j]==hs)box.push_back(products[j]);
            }
            sort(box.begin(),box.end());
            int t=0;
            if(box.size()>3)t=3;
            else t=box.size();
            for(int i=0;i<t;i++)tmp.push_back(box[i]);
            ans.push_back(tmp);
        }
        return ans;
    }
};
