class Solution {
public:

    bool starts(string &p, string& c) {
        if (c.size() > p.size()) return false;
        for(int i=0; i<c.size(); ++i) if (c[i] != p[i]) return false;
        return true;
    }
    vector<vector<string>> suggestedProducts(vector<string>& p, string w) {
        sort(p.begin(), p.end());
        vector<vector<string>> res;
        for(int i=0; i<w.size(); ++i) {
            string c = w.substr(0, i+1);
            auto it = lower_bound(p.begin(), p.end(), c);
            vector<string> cur;
            while(it != p.end() && starts(*it, c) && cur.size() < 3) {
                cur.push_back(*it);
                it++;
            }
            res.push_back(cur);
        }
        return res;
    }
};
