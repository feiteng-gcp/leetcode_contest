struct Node {
  vector<Node *> edges = vector<Node *>(255, nullptr);
  vector<int> words;
};

class Solution {
  void findSuggestions(Node* node, const vector<string>& products, vector<string>* result) {
    if (!node) return;
    
    for (int index : node->words) {
      if (result->size() < 3) {
        result->push_back(products[node->words.front()]);
      }
    }
    
    for (int i = 0; i < 255; i++) {
      if (node->edges[i] && result->size() < 3) {
        findSuggestions(node->edges[i], products, result);
      }
    }
  }
  
public:
  vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
    Node* root = new Node();
    
    for (int i = 0; i < products.size(); i++) {
      Node* node = root;
      for (int j = 0; j < products[i].size(); j++) {
        if (!node->edges[products[i][j]]) {
          node->edges[products[i][j]] = new Node();
        }
        
        node = node->edges[products[i][j]];
      }
      
      node->words.push_back(i);
    }
    
    vector<vector<string>> result;
    
    Node* node = root;
    for (int i = 0; i < searchWord.size(); i++) {
      vector<string> local_result;
      if (node) {
        node = node->edges[searchWord[i]];        
      } 
      
      findSuggestions(node, products, &local_result);
      result.push_back(local_result);
    }
    
    return result;
  }
};
