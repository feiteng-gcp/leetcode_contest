class Solution {
public:
    struct T{
        string s;
        bool operator < (const T &temp) const{
            for (int i = 0; i < s.size() && i < temp.s.size(); i++){
                if (s[i] > temp.s[i]) return false;
                else if (s[i] < temp.s[i]) return true;
            }
            return s.size() < temp.s.size();
        }
        T(string str){
            s = str;
        }
    };
    

    

    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        vector<vector<string>> ret;
        for (int i = 1; i <= searchWord.size(); i++){
            string cur = searchWord.substr(0, i);
            vector<T> vec;
            for (int j = 0; j < products.size(); j++){
                if (products[j].substr(0, i) == cur){
                    vec.push_back(T(products[j]));
                }
            }
            sort(vec.begin(), vec.end());
            while(vec.size() > 3) vec.pop_back();
            vector<string> v;
            for (int i = 0; i < vec.size(); i++){
                v.push_back(vec[i].s);
            }
            ret.push_back(v);
        }
        return ret;
    }
};
