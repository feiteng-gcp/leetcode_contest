class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        sort(products.begin(),products.end());
        vector<string> ans;

        int len=searchWord.length();
        vector<vector<string>> ret;
        for(int i=0;i<len;i++)
        {
            int sz=products.size();
            for(int j=0;j<sz;j++)
            {
                if(products[j].length()>i&&products[j][i]==searchWord[i])
                    ans.push_back(products[j]);
            }
            vector<string> rret;
            int anssz=ans.size();
            for(int j=0;j<3&&j<anssz;j++) rret.push_back(ans[j]);
            ret.push_back(rret);
            products=ans;
            ans.clear();
        }
        return ret;
    }
};
