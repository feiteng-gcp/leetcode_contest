class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        auto s = products;
        sort(s.begin(), s.end());
        vector<vector<string>> ans;
        string r;
        for (auto c : searchWord) {
            r.push_back(c);
            auto it = lower_bound(s.begin(), s.end(), r);
            ans.push_back({ });
            for (int i = 0; i < 3; i++) {
                if (it == s.end())
                    break;
                if (it->substr(0, r.size()) != r)
                    break;
                ans.back().push_back(*it);
                it++;
            }
        }
        return ans;
    }
};
