class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& p, string w) {
        vector <int> ex;
        for (int i = 0; i < p.size(); i++) {
            ex.push_back(1);
        }
        sort(p.begin(), p.end());
        
        vector<vector<string>> ret;

        for (int i = 0; i < w.size(); i++) {    
            vector <string> cur;
            for (int j = 0; j < p.size(); j++) {
                if (ex[j] == 1) {
                    if (p[j].size() > i && p[j][i] == w[i]) {
                        if (cur.size() < 3) {
                            cur.push_back(p[j]);
                        }
                    } else {
                        ex[j] = 0;
                    }
                }
            }
            
            ret.push_back(cur);
        }
        
        return ret;
    }
};
