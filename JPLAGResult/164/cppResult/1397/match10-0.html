class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& a, string w) {
        sort(a.begin(), a.end());
        int n = a.size();
        int len = w.length();
        vector<vector<string>> ans(len);
        vector<bool> valid(n, true);
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < n; j++) {
                if (i >= a[j].length() || w[i] != a[j][i])
                    valid[j] = false;
            }
            int count = 0;
            for (int j = 0; j < n; j++) {
                if (valid[j]) {
                    ans[i].push_back(a[j]);
                    count++;
                    if (count == 3)
                        break;
                }
            }
        }
        return ans;
    }
};
