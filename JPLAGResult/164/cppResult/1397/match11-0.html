class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        vector<vector<string>> res;
        sort(products.begin(),products.end());
        int m = products.size();
        int n = searchWord.length();
        for(int i = 0; i < n; i++) {
            string s = searchWord.substr(0,i+1);
            int lb = lower_bound(products.begin(),products.end(),s) - products.begin();
            vector<string> r;
            for(int j = lb; j < m && j<lb+3; j++) {
                if(products[j].substr(0,i+1)==s) r.push_back(products[j]);
            }
            res.push_back(r);
        }
        return res;
    }
};
