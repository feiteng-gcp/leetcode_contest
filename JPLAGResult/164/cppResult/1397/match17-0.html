class Solution {
public:
    #define all(x) x.begin(),x.end()
    inline int begin(string &s,string &t)
    {
        if(s.size() < t.size()) return 0;
        for(int i=0;i<t.size();++i) if(s[i]!=t[i]) return 0;
        return 1;
    }
    vector<vector<string>> suggestedProducts(vector<string>& products, string s) {
        sort(all(products));
        vector<vector<string>> ret;
        string tmp;
        for(int i=0;i<s.size();++i)
        {
            tmp += s[i];
            vector<string> vs;
            for(string &t : products) {
                if(begin(t, tmp)) vs.push_back(t);
                if(vs.size() == 3) break;
            }
            ret.push_back(vs);
        }
        return ret;
    }
};
