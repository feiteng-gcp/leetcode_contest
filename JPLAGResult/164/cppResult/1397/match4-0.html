class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        multiset<string> pset(products.begin(), products.end());
        vector<vector<string>> res;
        
        for (int len = 1; len <= searchWord.size(); len++) {
            string prefix = searchWord.substr(0, len);
            
            vector<string> res_i;
            auto it = pset.lower_bound(prefix);
            for (int j = 0; j < 3 && it != pset.end(); j++, ++it) {
                const string& s = *it;
                if (s.size() < prefix.size() || s.substr(0, len) != prefix) break;
                res_i.push_back(s);
            }
            
            res.push_back(res_i);
        }
        
        return res;
    }
};
