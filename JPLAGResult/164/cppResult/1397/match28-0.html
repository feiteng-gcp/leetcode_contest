class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        sort(products.begin(), products.end());
        int n = searchWord.length();
        vector<vector<string>> ans;
        int p = 0;
        string str = "";
        for (int i = 0; i < n; ++i) {
            vector<string> now;
            str += searchWord[i];
            while (p < products.size() && products[p] < str) {
               ++p;
            }
            for (int q = p; q < min((int)products.size(), p + 3); ++q) {
                if (products[q].substr(0, i + 1) == str) {
                    now.push_back(products[q]);
                }
            }
            ans.push_back(now);
        }
        return ans;
    }
};
