class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string> &products,
                                             string searchWord) {
        map<string, priority_queue<string, vector<string>, greater<string>>> sk;
        int len = products.size();
        for (int i = 0; i < len; i++) {
            int lens = products[i].length();
            for (int j = 1; j <= lens; j++) {
                sk[products[i].substr(0, j)].push(products[i]);
            }
        }
        len = searchWord.length();
        vector<vector<string>> ans;
        for (int i = 1; i <= len; i++) {
            auto s = searchWord.substr(0, i);
            vector<string> tmp;
            int count = 0;
            while (!sk[s].empty()) {
                tmp.push_back(sk[s].top());
                ++count;
                if (count >= 3)
                    break;
                sk[s].pop();
            }
            ans.push_back(tmp);
        }
        return ans;
    }
};
