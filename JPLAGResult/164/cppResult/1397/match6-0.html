#include <bits/stdc++.h> 
class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string sw) {
        vector<vector<string>> res;
        string r;
        for(auto i:sw)
        {
            vector<string> tmp;
            r+=i;
            for(auto pr:products)
            {
                if(pr.find(r)==0)
                    tmp.push_back(pr);
            }
            sort(tmp.begin(),tmp.end());
            vector<string> ins;
            for(int i=0;i<tmp.size();i++)
            {
                ins.push_back(tmp[i]);
                if(i==2)
                    break;
            }
            res.push_back(ins);
        }
        return res;
    }
};
