class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        sort(products.begin(),products.end());
        vector<vector<string> > ans(searchWord.length());
        for(int i=0;i<products.size();i++)
        {
            if(products[i][0]==searchWord[0])
                ans[0].push_back(products[i]);
        }
        for(int i=1;i<searchWord.length();i++)
        {
            for(int j=0;j<ans[i-1].size();j++)
            {
                if(ans[i-1][j][i]==searchWord[i])
                    ans[i].push_back(ans[i-1][j]);
            }
        }
        vector<vector<string> > res(searchWord.length());
        for(int i=0;i<searchWord.length();i++)
        {
            for(int j=0;j<3 && j<ans[i].size();j++)
            {
                res[i].push_back(ans[i][j]);
            }
        }
        return res;
    }
};
