class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& pro, string s) {
        int len= s.size();
        sort(pro.begin(), pro.end());
        int m=pro.size();
        
        vector<vector<string>> ans;

        for(int i=0; i<len; i++){
            vector<string> tmp;
            for(int j=0; j<m; j++){
                if (tmp.size()<3 && pro[j].substr(0,i+1)== s.substr(0, i+1) )
                    tmp.push_back(pro[j]);
            }
            ans.push_back(tmp);
        }
        
        return ans;
        
          
    }
};
