class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& p, string w) {
        vector<vector<string>> ans;
        sort(p.begin(),p.end());
        int idx = 0;
        int n = w.size();
        for(int i=1;i<=n;i++){
            string cur = w.substr(0,i);
            idx = lower_bound(p.begin(),p.end(),cur) - p.begin();
            vector<string> v;
            for(;idx<p.size();idx++){
                string tmp = p[idx].substr(0,i);
                if(cur==tmp){
                    v.pb(p[idx]);
                    for(int j=1;j<=2 && idx+j<p.size();j++){
                        string tmp1 = p[idx+j].substr(0,i);
                        if(cur==tmp1){
                            v.pb(p[idx+j]);
                        }
                        else{
                            break;
                        }
                    }
                    
                    break;
                }
            }
            ans.pb(v);
        }
        return ans;
    }
};
