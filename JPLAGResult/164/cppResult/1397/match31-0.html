class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        sort(products.begin(), products.end());
        int n = searchWord.size();
        int begin = 0;
        vector<vector<string>> ret(n, vector<string>());
        for (int i = 0; i < n; i++) {
            int count = 0;
            bool first = false;
            for (int j = begin; j < products.size(); j++) {
                if (count == 3)
                    break;
                if (products[j].size() < i+1 || products[j].substr(0, i+1) != searchWord.substr(0, i+1)) {
                    if (!first) {
                        begin++;
                    } else {
                        break;
                    }
                } else {
                    if (!first) {
                        first = true;
                    }
                    ret[i].push_back(products[j]);
                    count++;
                }
            }
        }
        // for (string s: products)
        //     cout << s << endl;
        return ret;
    }
};
