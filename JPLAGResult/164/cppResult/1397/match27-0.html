class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        int n = products.size(), m = searchWord.length();
        
        sort(products.begin(), products.end());
        
        vector<vector<string>> ans;
        string key = "";
        for (int i = 0; i < m; ++i) {
            key += searchWord[i];
            vector<string> temp;
            for (int j = 0; j < n; ++j) {
                if (products[j].find(key) == 0) temp.push_back(products[j]);
                if (temp.size() == 3) break;
            }
            ans.push_back(temp);
        }
        
        return ans;
    }
};
