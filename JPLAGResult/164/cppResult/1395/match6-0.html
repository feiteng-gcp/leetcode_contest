class Solution {
public:
    int minTimeToVisitAllPoints(vector<vector<int>>& points) {
        int P=points.size();
        int ans=0;
        for(int j=1;j<P;j++) {
            ans += max(abs(points[j][0]-points[j-1][0]),abs(points[j][1]-points[j-1][1]));
        }
        return ans;
        
    }
};
