class Solution {
public:
    int minTimeToVisitAllPoints(vector<vector<int>>& points) {
        int ret = 0;
        for (int i = 1; i < points.size(); i++){
            int deltax, deltay;
            deltax = abs(points[i][0] - points[i - 1][0]);
            deltay = abs(points[i][1] - points[i - 1][1]);
            ret += max(deltax, deltay);
        }
        return ret;
    }
};
