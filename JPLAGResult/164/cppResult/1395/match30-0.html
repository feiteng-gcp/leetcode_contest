int get(int x, int y, int a, int b) {
    if (x == a) return abs(y - b);
    if (y == b) return abs(x - a);
    int c = abs(a - x) + abs(y + a - x - b);
    int d = abs(b - y) + abs(x + b - y - a);
    int e = abs(a - x) + abs(y + x - a - b);
    int f = abs(b - y) + abs(x + y - b - a);
    return min(min(c, d), min(e, f));
}

class Solution {
public:
    int minTimeToVisitAllPoints(vector<vector<int>>& a) {
        int ans = 0, n = a.size();
        for (int i = 1; i < n; i++) {
            ans += get(a[i - 1][0], a[i - 1][1], a[i][0], a[i][1]);
        }
        return ans;
    }
};
