class Solution {
public:
    int minTimeToVisitAllPoints(vector<vector<int>>& ps) {
        int ans = 0;
        int len = ps.size();
        for (int i = 1; i < len; i++) {
            ans += max(abs(ps[i][0] - ps[i - 1][0]), abs(ps[i][1] - ps[i - 1][1]));
        }
        return ans;
    }
};
