class Solution {
public:
    int dist(vector<int>& a, vector<int>& b){
        int h = abs(a[0] - b[0]);
        int v = abs(a[1] - b[1]);
        
        return h + v - min(h,v);
    }
    int minTimeToVisitAllPoints(vector<vector<int>>& points) {
        int total = 0;
        for(int i = 0; i < points.size() - 1; ++i){
            total += dist(points[i], points[i + 1]);
        }
        return total;
    }
};
