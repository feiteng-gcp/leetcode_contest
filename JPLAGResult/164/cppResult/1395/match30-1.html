class Solution {
public:
    int minTimeToVisitAllPoints(vector<vector<int>>& points) {
        int N = points.size(), sum = 0;
        for (int i = 1; i < N; ++i) {
            sum += max(abs(points[i][0] - points[i - 1][0]), abs(points[i][1] - points[i - 1][1]));
        }
        return sum;
    }
};
