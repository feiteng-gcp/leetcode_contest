class Solution {
public:
    int minSwap(vector<int>& A, vector<int>& B) {
        const int INF = 1<<29;
        int n = A.size();
        vector<int> f(n,INF), g(n,INF);
        f[0]=0, g[0]=1;
        for (int i=1; i<n; i++){
            if (A[i-1]<A[i] && B[i-1]<B[i]){
                f[i] = min(f[i],f[i-1]);
                g[i]  =min(g[i],g[i-1]+1);
            }
            if (B[i-1]<A[i] && A[i-1]<B[i]){
                f[i] = min(f[i],g[i-1]);
                g[i]  =min(g[i],f[i-1]+1);
            }
        }
        return min(f[n-1],g[n-1]);
    }
};
