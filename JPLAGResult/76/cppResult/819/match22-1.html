class Solution {
public:
	int minSwap(vector<int>& a, vector<int>& b) {
		int n = a.size(), i, j, k, best = n + 1;
		vector<int> dp(n + 1, 0), dp2(n + 1, 0);
		dp2[0] = 1;


		for (i = 1; i < n; i++) {
			dp[i] = n + 1;
			if ((a[i] > a[i - 1]) && (b[i] > b[i - 1])) {
				dp[i] = min(dp[i], dp[i - 1]);
			}
			if ((a[i] > b[i - 1]) && (b[i] > a[i - 1])) {
				dp[i] = min(dp[i], dp2[i - 1]);
			}

			dp2[i] = n + 1;
			if ((b[i] > a[i - 1]) && (a[i] > b[i - 1])) {
				dp2[i] = min(dp2[i], dp[i - 1] + 1);
			}
			if ((a[i] > a[i - 1]) && (b[i] > b[i - 1])) {
				dp2[i] = min(dp2[i], dp2[i - 1] + 1);
			}

		}

		return min(dp[n - 1], dp2[n - 1]);
	}
};
