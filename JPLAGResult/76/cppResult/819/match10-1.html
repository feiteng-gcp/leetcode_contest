class Solution {
public:
    int minSwap(vector<int>& A, vector<int>& B) {
        int N = A.size();
        
        // res[0][j] is keep j same, res[1] is keep j swapped
        vector<vector<int>> res(2, vector<int>(N, N));

        res[0][0] = 0;
        res[1][0] = 1;
        for(int i = 1; i < N; ++i)
        {
            if(A[i] > A[i-1] && B[i] > B[i-1])
            {
                // keep i not swapped
                res[0][i] = min(res[0][i], res[0][i-1]);
                res[1][i] = min(res[1][i], 1 + res[1][i-1]);
            }
            if(A[i] > B[i-1] && B[i] > A[i-1])
            {
                res[0][i] = min(res[0][i], res[1][i-1]);
                res[1][i] = min(res[1][i], 1 + res[0][i-1]);
            }
        }
        
        return min(res[0][N-1], res[1][N-1]);
    }
};
