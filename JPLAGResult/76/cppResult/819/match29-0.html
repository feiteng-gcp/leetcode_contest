class Solution {
public:
    int minSwap(vector<int>& a, vector<int>& b) {
        int n = a.size();
        if (n < 1) return 0;
        vector<int> mm[2] = {vector<int>(n, 1000010000), vector<int>(n, 1000010000)};
        mm[0][0] = 0;
        mm[1][0] = 1;
        for (int i = 1; i < n; ++i) {
            if (a[i] > a[i - 1] && b[i] > b[i - 1]) {
                mm[0][i] = min(mm[0][i], mm[0][i - 1]);
                mm[1][i] = min(mm[1][i], mm[1][i - 1] + 1);
            }
            if (a[i] > b[i - 1] && b[i] > a[i - 1]) {
                mm[0][i] = min(mm[0][i], mm[1][i - 1]);
                mm[1][i] = min(mm[1][i], mm[0][i - 1] + 1);
            }
        }
        return min(mm[0][n - 1], mm[1][n - 1]);
    }
};
