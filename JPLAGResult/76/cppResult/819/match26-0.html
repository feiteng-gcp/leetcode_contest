class Solution {
public:
    int minSwap(vector<int>& A, vector<int>& B) {
        int n = A.size();
        int a0 = 0, a1 = 1;
        for (int i = 1; i < n; i++) {
            int b0 = n + 1, b1 = n + 1;
            if (A[i] > A[i - 1] && B[i] > B[i - 1]) {
                b0 = min(b0, a0);
                b1 = min(b1, a1 + 1);
            }
            if (A[i] > B[i - 1] && B[i] > A[i - 1]) {
                b0 = min(b0, a1);
                b1 = min(b1, a0 + 1);
            }
            a0 = b0;
            a1 = b1;
        }
        return min(a0, a1);
    }
};
