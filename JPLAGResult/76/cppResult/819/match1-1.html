class Solution {
public:
  int minSwap(vector<int>& A, vector<int>& B) {
    int n=A.size(), i, j;
    vector<int> last_switched (n, n+1), last_unswitched (n, n+1);
    last_switched[0] = 1, last_unswitched[0] = 0;

    for (i=1; i<n; ++i) {
      if (A[i]>A[i-1] && B[i]>B[i-1]) {
        last_unswitched[i] = last_unswitched[i-1];
        last_switched[i] = last_switched[i-1] + 1;  // switch A[i] & B[i] will make a feasible solution
      }
      if (A[i]>B[i-1] && B[i]>A[i-1]) {
        last_switched[i] = min(last_switched[i], last_unswitched[i-1]+1);
        last_unswitched[i] = min(last_unswitched[i], last_switched[i-1]);
      }
      // cout << "i=" << i << ": switched[i]=" << last_switched[i] << ", unswitched[i]=" << last_unswitched[i] << endl;
    }
    return min(last_switched[n-1], last_unswitched[n-1]);
  }
};
