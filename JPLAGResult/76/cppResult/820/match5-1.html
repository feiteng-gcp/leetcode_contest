class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int N = graph.size();
        vector<unordered_set<int>> g(N);
        vector<unordered_set<int>> rg(N);

        unordered_set<int> result;
        
        queue<int> q;
        for (int i=0; i<N; i++) {
            for (auto j : graph[i]) {
                g[i].insert(j);
                rg[j].insert(i);
            }
            if (graph[i].empty()) q.push(i);
        }
        
        while (!q.empty()) {
            int i = q.front();
            q.pop();
            result.insert(i);
            for (auto j : rg[i]) {
                g[j].erase(i);
                if (g[j].empty() && result.find(j)==result.end()) q.push(j);
            }
        }
        
        auto v = vector<int>(result.begin(), result.end());
        sort(v.begin(), v.end());
        return v;
    }
};
