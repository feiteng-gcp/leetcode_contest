class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {

        vector<int> safe(graph.size(), -1), rs;
        vector<bool> passing(graph.size(), false);
        for(int i = 0; i < graph.size(); i++) {
            walk(safe, passing, graph, i);
        }
        for(int i = 0; i < graph.size(); i++ ) {
            if (safe[i] == 1) rs.push_back(i);
        }
        return rs;
    }
    
    void walk(vector<int> &safe, vector<bool> &passing, vector<vector<int>>& graph, int pos) {
        if (safe[pos] == -1) {
            for(int i = 0; i < graph[pos].size(); i++) {
                int next = graph[pos][i];
                if (passing[next]) {
                    safe[pos] = 0;
                    return;
                } else {
                    passing[next] = true;
                    walk(safe, passing, graph, next);
                    passing[next] = false;
                    if (safe[next] == 0) {
                        safe[pos] = 0;
                        return;
                    }
                }
            }
            safe[pos] = 1;
        }
    }
};
