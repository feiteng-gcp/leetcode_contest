// 0:not-visited  1:in-progress  2:safe  3:cycle

class Solution {
public:
    int visit(const vector<vector<int>> &graph, int i, vector<int> &result) {
        switch (result[i]) {
            case 0:
                result[i] = 1;
                for (int j : graph[i]) {
                    if (visit(graph, j, result) == 3)
                        return result[i] = 3;
                }
                return result[i] = 2;
            case 1:
            case 3:
                return 3;
            case 2:
                return 2;

        }
    }

    vector<int> eventualSafeNodes(vector<vector<int>> &graph) {
        vector<int> result(graph.size()), ret;
        ret.reserve(graph.size());
        for (int i = 0; i < graph.size(); ++i) {
            if (result[i] == 0) visit(graph, i, result);
            if (result[i] == 2) ret.push_back(i);
        }
        return ret;
    }
};
