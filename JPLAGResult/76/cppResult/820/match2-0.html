class Solution {
public:
    // find circle !
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int n = graph.size();
        vector<int> isEventualNodes(n, -1);
        vector<int> res;
        for(int i = 0; i < graph.size(); i++){
            if(isEventualNodes[i] == -1){
                isEventualNode(graph, isEventualNodes, i, res);
            }
        }
        sort(res.begin(), res.end());
        return res;
    }
    
    bool isEventualNode(vector<vector<int>>& graph, vector<int> &isEventualNodes, int idx, vector<int>&res){
        if(isEventualNodes[idx] != -1){
            return isEventualNodes[idx]; // is 1, not 0
        }
        
        isEventualNodes[idx] = 0;
        for(auto nei : graph[idx]){
            if(!isEventualNode(graph,isEventualNodes, nei, res)){
                return isEventualNodes[idx];
            }
        }
        isEventualNodes[idx] = 1;
        res.push_back(idx);
        return true;
        
    }
};
