class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        // check whether is a loop
        int n = graph.size();
        vector<int> res;
        queue<int> q;
        unordered_map<int, vector<int>> m;
        vector<int> v (n, 0);
        
        for (int i=0; i<n; ++i) {
            /*vector<int> parent(n, -1);
            if (helper(graph, i, parent)) {
                res.push_back(i);
            }*/
            if (graph[i].empty()) q.push(i);
            else {
                v[i] = graph[i].size();
                for (int j=0; j<graph[i].size(); j++) {
                    m[graph[i][j]].push_back(i);
                }
            }
        }
        
        while(!q.empty()) {
            auto x = q.front();
            q.pop();
            res.push_back(x);
            for (auto y : m[x]) {
                --v[y];
                if (v[y] == 0) {
                    q.push(y);
                }
            }
        }
        
        sort(res.begin(), res.end());
        return res;
    }

            parent[j] = i;

        }

    }
    
    bool isParent(vector<int>& parent, int p, int c) {

        }
