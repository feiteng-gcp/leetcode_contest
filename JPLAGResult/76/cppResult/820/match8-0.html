class Solution {
public:
    int degree[10001];
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int l=graph.size();
        vector<int> eg[l];
        for(int i=0;i<l;i++)
        {
            for(int j=0;j<graph[i].size();j++)
            {
                degree[i]++;
                eg[graph[i][j]].push_back(i);
            }
        }
        queue<int> q;
        vector<int>ans;
        
        for(int i=0;i<l;i++)
        {
            if(degree[i]==0)
            {
                q.push(i);
                ans.push_back(i);
            }
            
        }
        
        while(!q.empty())
        {
            int x=q.front();
            q.pop();
            for(int i=0;i<eg[x].size();i++)
            {
                int y=eg[x][i];

                degree[y]--;
                if(degree[y]==0)
                {
                    q.push(y);
                    ans.push_back(y);
                }
            }
        }
        sort(ans.begin(),ans.end());
        return ans;
    }
};
