class Solution {
private :
	vector<bool> isTerm;
	vector<bool> isVisit;

	bool is_end(vector<vector<int>>& graph, int i)
	{
		if (isTerm[i])return true;
		if (isVisit[i])return false;
		isVisit[i] = true;
		vector<int>& list = graph[i];
		bool x = true;
		for (int to : list)
		{
			x &= is_end(graph,to);
		}
		return isTerm[i] = x;
	}
public:
	
	vector<int> eventualSafeNodes(vector<vector<int>>& graph) 
	{
		
		for (int i = 0 ; i < graph.size(); i++)
		{
			auto& list = graph[i];
			if (list.size() == 0) {isTerm.push_back(true);}
			else isTerm.push_back(false);
			isVisit.push_back(false);
		}
		
		vector<int> ret;
		for (int i = 0; i < graph.size(); i++)
		{
			if(is_end(graph, i))
				ret.push_back(i);
		}
		return ret;
	}
};
