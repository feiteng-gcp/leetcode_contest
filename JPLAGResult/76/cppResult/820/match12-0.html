class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int n = graph.size();
        if (n == 0) {
            return {};
        }
        
        vector<int> res;
        vector<int> terminals;
        
        for (int i = 0; i < n; ++i) {
            auto &v = graph[i];
            if (v.empty()) {
                terminals.push_back(i);
            }
        }
        
        vector<int> flag(n, 0);
        
        for (int i = 0; i < n; ++i) {
            dfs(graph, flag, i);
        }
        
        for (int i = 0; i < n; ++i) {
            if (flag[i] != -1) {
                res.push_back(i);
            }
        }
        
        return res;
    }
    
private:
    bool dfs(vector<vector<int>> &graph, vector<int> &flag, int i) {
        if (flag[i] != 0) {
            // cout << "F " << i << " " << flag[i] << endl;
            return flag[i] != -1; 
        }
        // cout << i << " " << endl;
        flag[i] = -1;
        for (auto j : graph[i]) {
            // cout << i << " " <<  j << endl;
            if (!dfs(graph, flag, j)) {
                // cout << i << " " << j << " c " << endl;
                return false;
            }
        }
        
        // cout << i << " n " << endl;
        flag[i] = 1;
        return true;
    }
};
