typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T> >;

const ll MOD=1e9+7;

class Solution {
public:
  vector<int> col;
  vector<int> good;
  bool dfs(vector<vector<int>>& g, int v){

    col[v]=1;
    bool fl=1;
    for(int w:g[v]){
      if(col[w]==1){
	fl=0;
      }else if(col[w]==2){
	fl&=good[w];
      }else{
	dfs(g,w);
	fl&=good[w];
      }
    }

    good[v]=fl;
    col[v]=2;
    return fl;
  }
    vector<int> eventualSafeNodes(vector<vector<int>>& g) {
      int N=g.size();
      col.resize(N);
      good.resize(N);
      for(int v=0;v<N;v++){
	if(col[v]==0){
	  dfs(g,v);
	}
      }
      vector<int> res;
      for(int v=0;v<N;v++){
	if(good[v]) res.pb(v);
      }
      return res;
    }
};
