class Solution {
public:
    const static int maxn = 1e4 + 5;
    int deg[maxn];
    vector<int> rG[maxn];
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int n = graph.size();
        queue<int> q;
        for(int i = 0;i < n;i++){
            for(int j = 0;j < graph[i].size();j++){
                int v = graph[i][j];
                int u = i;
                rG[v].push_back(u);
                deg[u]++;
            }
        }

        vector<int> ans;
        for(int i = 0;i < n;i++){
            if(deg[i] == 0) q.push(i);
        }
        while(!q.empty()){
            int p = q.front();q.pop();
            ans.push_back(p);
            for(int i = 0;i < rG[p].size();i++){
                int to = rG[p][i];
                deg[to]--;
                if(deg[to] == 0) q.push(to);
            }
        }
        sort(ans.begin(),ans.end());
        return ans;
    }
};
