class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int n = graph.size();
        vector<vector<int>> lj(n);
        vector<int> rd(n);
        for (int i = 0; i < n; i++) {
            for (auto j : graph[i]) {
                lj[j].push_back(i);
                rd[i]++;
            }
        }
        stack<int, vector<int>> s;
        for (int i = 0; i < n; i++) {
            if (rd[i] == 0) s.push(i);
        }
        vector<int> ans;
        while (!s.empty()) {
            auto now = s.top();
            s.pop();
            ans.push_back(now);
            for (auto i : lj[now]) {
                rd[i]--;
                if (rd[i] == 0) s.push(i);
            }
        }
        sort(ans.begin(), ans.end());
        return ans;
    }
};
