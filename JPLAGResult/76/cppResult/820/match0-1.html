class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        vector<int> res; 
        int n=(int)graph.size(); 
        vector<int> out_degree (n, 0); 
        vector<vector<int>> pre_node (n); 
        queue<int> my_q; 
        for(int i=0; i<n; i++) {
            if((int)graph[i].size()==0) {
                res.push_back(i); 
                my_q.push(i); 
            }
            else {
                out_degree[i]=graph[i].size(); 
                for(auto x:graph[i]) {
                    pre_node[x].push_back(i); 
                }
            }    
        }
        while((int)my_q.size()!=0) {

            int now=my_q.front(); 
            my_q.pop(); 
            for(auto x:pre_node[now]) {
                out_degree[x]--; 
                if(out_degree[x]==0) {
                    res.push_back(x); 
                    my_q.push(x); 
                }
            }
        }
        sort(res.begin(), res.end()); 
        return res; 
    }
};
