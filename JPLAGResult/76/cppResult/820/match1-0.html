#define MAXN 10000
int n, idx, blk, dfn[MAXN+10], low[MAXN+10], s[MAXN+10], col[MAXN+10], deg[MAXN+10], q[MAXN+10], f[MAXN+10];
bool in[MAXN+10];
vector<vector<int>> g,gg;

void dfs(int x) {
    dfn[x]=low[x]=++idx; s[++s[0]]=x; in[x]=1;
    for (int y : g[x]) {
        if (dfn[y] == 0) {
            dfs(y);
            low[x]=min(low[x], low[y]);
        } else {
            if (in[y]) low[x]=min(low[x], dfn[y]);
        }
    }
    if (low[x]==dfn[x]) {
        int y=s[s[0]--]; in[y]=0; col[y]=blk;
        while (y!=x) {
            y=s[s[0]--]; in[y]=0; col[y]=blk;
        }
        ++blk;
    }
}

class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        n=graph.size();
        g.clear();
        for (int i=0;i<n;++i) {
            g.push_back(vector<int>());
        }
        for (int x=0;x<n;++x) {
            for (int y : graph[x]) {
                g[x].push_back(y);
            }
        }
        idx=blk=s[0]=0;
        for (int i=0;i<n;++i) dfn[i]=low[i]=col[i]=0;
        for (int i=0;i<n;++i) in[i]=0;
        for (int i=0;i<n;++i) {
            if (dfn[i] == 0) dfs(i);
        }
        // for (int i=0;i<n;++i) printf("%d ", dfn[i]); putchar('\n');
        // for (int i=0;i<n;++i) printf("%d ", low[i]); putchar('\n');
        // for (int i=0;i<n;++i) printf("%d ", col[i]); putchar('\n');
        for (int i=0;i<n;++i) {
            if (g[i].size() == 0) {
                f[col[i]]=1;
            } else {
                f[col[i]]=0;
            }
        }
        gg.clear();
        for (int i=0;i<blk;++i) gg.push_back(vector<int>());
        for (int i=0;i<blk;++i) deg[i]=0;
        for (int x=0;x<n;++x) {
            for (int y : g[x]) {
                if (col[x]!=col[y]) {
                    gg[col[x]].push_back(col[y]);
                    ++deg[col[y]];
                }
            }
        }
        
        int l=0,r=0;
        for (int x=0;x<blk;++x) {
            if (deg[x] == 0) {
                q[++r]=x;
            }
        }
        while (l<r) {
            int x=q[++l];
            for (int y : gg[x]) {
                if (--deg[y] == 0) {
                    q[++r]=y;
                }
            }
        }
        
        for (int x=0;x<n;++x) {
            for (int y : g[x]) {
                if (col[x]==col[y]) {
                    gg[col[x]].push_back(col[y]);
                }
            }
        }
        
        for (int i=r;i>0;--i) {
            int x=q[i];
            if (f[x]==1) continue;
            f[x]=1;
            for (int y : gg[x]) {
                if (y == x || f[y] == 0) {
                    f[x] = 0; break;
                }
            }
        }
        vector<int> vec;
        for (int i=0;i<n;++i) {
            if (f[col[i]] == 1) {
                vec.push_back(i);
            }
        }
        return vec;
    }
};
