class Solution {
public:
    int n;
    vector<int> din;
    vector< vector<int> > rG;
    vector<int> ans;
    
    void init(vector<vector<int>>& G){
        n = G.size();
        din = vector<int>(n,0);
        rG.resize(n, vector<int>(0) );
        
        for (int u=0; u<n; u++){
            for (int v:G[u]){
                rG[v].push_back(u);
                din[u]++;
            }
        }
    }
    
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        init(graph);
        queue<int> que;
        for (int i=0; i<n; i++){
            if (din[i]==0) que.push(i);
        }
        while (que.size()){
            int u = que.front(); que.pop();
            ans.push_back(u);
            for (int v:rG[u]){
                din[v]--;
                if (din[v]==0) que.push(v);
            }
        }
        sort(ans.begin(),ans.end());
        return ans;
    }
};
