class Solution {
public:
    bool visitNode(int node, vector<int>& status, vector<vector<int>>& graph){
        if(status[node] == -1 ) return false;
        if(status[node] == 1 ){
            status[node] = -1;
            return false;
        }
        if(status[node] == 2) return true;
        status[node] = 1;
        bool is_safe = true;
        for(int next_node:graph[node]){
            is_safe = is_safe && visitNode(next_node, status, graph);
        }
        status[node] = is_safe ? 2 : -1;
        return is_safe;
    }
    
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int N = graph.size();
        // 0: univisited
        // 1: visiting in current st
        // 2: complete tree
        // -1: complete circle
        vector<int> status (N,0);
        for(int node=0;node<N;node++){
            visitNode(node, status, graph);

        }
        vector<int> saftNodes;
        for(int i=0;i<N;i++){
            if(status[i]==2) saftNodes.push_back(i);
        }
        return saftNodes;
    }
};
