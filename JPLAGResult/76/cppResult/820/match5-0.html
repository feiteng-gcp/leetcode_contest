class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int len = graph.size();
        vector<int> outdgs(len, 0);
        
        queue<int> q;
        
        vector<vector<int>> graph2(len, vector<int>());
        for (int i = 0; i < len; i++) {
            int tsize = graph[i].size();
            outdgs[i] = tsize;
            if (tsize == 0) {
                q.push(i);
            }
            for (int j = 0; j < tsize; j++) {
                int to = graph[i][j];
                graph2[to].push_back(i);
            }
        }
        
        vector<int> ret;
        while (q.empty() == false) {
            int cur = q.front();
            q.pop();
            ret.push_back(cur);
            for (auto val : graph2[cur]) {
                outdgs[val]--;
                if (outdgs[val] == 0) {
                    q.push(val);
                }
            }
        }
        
        sort(ret.begin(), ret.end());
        return ret;
    }
};
