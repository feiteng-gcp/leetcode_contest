class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        vector<int> ret;
        unordered_map<int, int> d;
        unordered_map<int, vector<int>> e;
        for (int i = 0; i < graph.size(); ++ i)
            for (auto& x : graph[i]){
                d[i] ++;
                e[x].push_back(i);
            }
        deque<int> dq;
        for (int i = 0; i < graph.size(); ++ i)
            if (d[i] == 0) {
                dq.push_back(i);
                ret.push_back(i);
            }
        while (!dq.empty()){
            int x = dq.front();
            for (auto& y : e[x]){
                d[y] --;
                if (d[y] == 0) {
                    dq.push_back(y);
                    ret.push_back(y);
                }
            }
            dq.pop_front();
        }
        sort(ret.begin(), ret.end());
        return ret;
    }
};
