const int N = 100100;
int n;
vector<vector<int>> g;
int ng, gid[N], sz[N], vst[N], lo[N], seq, stk[N], top;
bool in_stk[N];
char safe[N];
set<int> sol;

void scc(int a) {
    stk[top++] = a;
    in_stk[a] = true;
    vst[a] = seq++;
    lo[a] = vst[a];
    for (int b : g[a]) {
        if (vst[b] < 0) {
            scc(b);
            lo[a] = min(lo[a], lo[b]);
        } else if (in_stk[b]) {
            lo[a] = min(lo[a], vst[b]);
        }
    }
    if (vst[a] == lo[a]) {
        int b = a;
        do {
            b = stk[--top];
            in_stk[b] = false;
            gid[b] = ng;
            ++sz[ng];
        } while (a != b);
        ++ng;
    }
}

bool go(int a) {
    if (safe[a] < 0) {
        if (sz[gid[a]] > 1) safe[a] = false;
        else {
            safe[a] = true;
            for (int b : g[a]) if (!go(b)) safe[a] = false;
            if (safe[a]) sol.insert(a);
        }
    }
    return safe[a];
}

class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        g = graph;
        n = g.size();
        if (n < 1) return vector<int>();
        memset(vst, 0xff, sizeof(vst));
        memset(sz, 0, sizeof(sz));
        memset(in_stk, false, sizeof(in_stk));
        ng = 0;
        for (int a = 0; a < n; ++a) if (vst[a] < 0) top = 0, scc(a);
        // for (int a = 0; a < n; ++a) cout << a << " " << gid[a] << " " << sz[gid[a]] << endl;
        sol.clear();
        memset(safe, 0xff, sizeof(safe));
        for (int a = 0; a < n; ++a) if (binary_search(g[a].begin(), g[a].end(), a)) safe[a] = false;
        for (int a = 0; a < n; ++a) if (safe[a] < 0) go(a);
        return vector<int>(sol.begin(), sol.end());
    }
};
