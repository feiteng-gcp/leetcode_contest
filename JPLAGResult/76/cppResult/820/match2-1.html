class Solution {
public:
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int N = graph.size();
        // -1, not visited, 0: visting, 1: finished and valid, 2: can't stop
        vector<int> color(N, -1);
        vector<int> res;
        for(int i = 0; i < N; ++i)
        {
            if(color[i] == -1)
            {
                dfs(graph, i, color, res);
            }
        }
        
        sort(res.begin(), res.end());
        return res;
    }
    
    bool dfs(vector<vector<int>>& g, int v, vector<int>& color, vector<int>& res)
    {
        color[v] = 0;
        bool isValid = true;
        for(auto next : g[v])
        {
            if(color[next] == -1)
            {
                isValid &= dfs(g, next, color, res);
            }
            else if(color[next] == 0)
            {
                //cycle detected
                isValid = false;
            }
            else if(color[next] == 2)
            {
                // reached to invalid node
                isValid = false;
            }
            else
            {
                // hit valid node
            }
        }
        
        if(isValid)
        {
            res.push_back(v);
            color[v] = 1;
        }
        else
        {
            color[v] = 2;
        }
        return isValid;
    }
};
