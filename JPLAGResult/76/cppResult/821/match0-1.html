const int N = 256;
const int dr[] = {-1, 0, 1, 0};
const int dc[] = {0, 1, 0, -1};
int g[N][N], vst[N][N], seq;
int n, m;

bool internal(int r, int c) {
    return 0 <= r && r < n && 0 <= c && c < m;
}

bool falling(int r, int c) {
    if (r < 0) return false;
    if (!internal(r, c)) return true;
    if (!g[r][c]) return true;
    if (vst[r][c] == seq) return true;
    vst[r][c] = seq;
    for (int d = 0; d < 4; ++d) if (!falling(r + dr[d], c + dc[d])) return false;
    return true;
}

int fall(int r, int c) {
    if (!internal(r, c)) return 0;
    if (!g[r][c]) return 0;
    int ret = 1;
    g[r][c] = 0;
    for (int d = 0; d < 4; ++d) ret += fall(r + dr[d], c + dc[d]);
    return ret;
}

class Solution {
public:
    vector<int> hitBricks(vector<vector<int>>& grid, vector<vector<int>>& hits) {
        n = grid.size();
        m = grid[0].size();
        for (int i = 0; i < n; ++i) for (int j = 0; j < m; ++j) g[i][j] = grid[i][j];
        vector<int> ret;
        for (const auto& h: hits) {
            int r = h[0], c = h[1];
            if (g[r][c]) {

                int removal = 0;
                g[r][c] = 0;
                for (int d = 0; d < 4; ++d) {
                    int x = r + dr[d], y = c + dc[d];
                    if (!internal(x, y) || !g[x][y]) continue;
                    ++seq;
                    // cout << x << " " << y << " " << falling(x, y) << endl;
                    ++seq;
                    if (falling(x, y)) removal += fall(x, y);
                }
                ret.push_back(removal);
            } else {
                ret.push_back(0);
            }
        }
        return ret;
    }
};
