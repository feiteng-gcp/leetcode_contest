class Solution {
private:
    int f[40005];
    int r[40005];
    
    int getFather(int x){
        if (f[x] != x) f[x] = getFather(f[x]);
        return f[x];
    }

public:
    vector<int> hitBricks(vector<vector<int>>& grid, vector<vector<int>>& hits) {
        vector<int> ret;
        unordered_map<int, int> hash;
        int n = grid.size(), m = grid[0].size();
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < m; ++ j){
                f[i * 200 + j] = i * 200 + j;
                r[i * 200 + j] = 0;
            }
        int dx[4] = {0, 0, 1, -1}, 
            dy[4] = {1, -1, 0, 0};
        vector<pair<int, int>> arr;
        for (auto& p : hits){
            if (grid[p[0]][p[1]] == 1){
                grid[p[0]][p[1]] = 0;
                arr.push_back(make_pair(p[0], p[1]));
            } else {
                arr.push_back(make_pair(-1, -1));
            }
        }
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < m; ++ j){
                int x = i, y = j;
                if (grid[x][y] == 0) continue;
                for (int k = 0; k < 4; ++ k){
                    int tx = x + dx[k], ty = y + dy[k];
                    if (tx < 0 || ty < 0 || tx >= n || ty >= m) continue;
                    if (grid[tx][ty] == 0) continue;
                    auto p = 
                    f[getFather(x * 200 + y)] = getFather(tx * 200 + ty);
                }
            }
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < m; ++ j)
                if (grid[i][j] == 1){
                    r[getFather(i * 200 + j)] ++;
                }
        for (int j = 0; j < m; ++ j)
            if (grid[0][j] == 1) hash[getFather(0 * 200 + j)] = 1;
        for (int i = (int)arr.size() - 1; i >= 0; -- i){
            int x = arr[i].first, y = arr[i].second;
            if (x == -1 && y == -1){
                ret.push_back(0);
                continue;
            }
            vector<int> tmp;
            unordered_set<int> rec;
            for (int k = 0; k < 4; ++ k){
                int tx = x + dx[k], ty = y + dy[k];
                if (tx < 0 || ty < 0 || tx >= n || ty >= m) continue;
                if (grid[tx][ty] == 0) continue;
                if (rec.count(getFather(tx * 200 + ty)) == 0){
                    rec.insert(getFather(tx * 200 + ty));
                    tmp.push_back(getFather(tx * 200 + ty));
                }
            }
            int h = 0;
            int cnt = 0;
            for (auto& p : tmp)
                if (hash[p] == 1) h = 1;
            if (x == 0) h = 1;
            if (h == 1){
                for (int k = 0; k < tmp.size(); ++ k)
                    if (hash[tmp[k]] == 0) cnt += r[tmp[k]];
            }
            for (int k = 1; k < tmp.size(); ++ k){
                r[tmp[0]] += r[tmp[k]];
                f[tmp[k]] = f[tmp[0]];
            }
            if (!tmp.empty()) r[tmp[0]] ++;
            if (!tmp.empty()) hash[tmp[0]] = h;
            ret.push_back(cnt);
            grid[x][y] = 1;
            if (!tmp.empty()) f[x * 200 + y] = tmp[0];
                        else  {
                            f[x * 200 + y] = x * 200 + y;
                            r[x * 200 + y] = 1;
                            if (x == 0) hash[x * 200 + y] = 1;
                        }
        }
        reverse(ret.begin(), ret.end());
        return ret;
    }
};
