class Solution {
public:
    vector<int> hitBricks(vector<vector<int>>& grid, vector<vector<int>>& hits) {
        int n = grid.size(), m = grid[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                grid[i][j] *= INT_MAX;
            }
        }
        for (int i = 0; i < hits.size(); i++) {
            if (grid[hits[i][0]][hits[i][1]]) {
                grid[hits[i][0]][hits[i][1]] = i + 2;
            }
        }
        priority_queue<pair<int, pair<int, int> > > st;
        vector<vector<int> > dist(n, vector<int>(m, 0));
        vector<int> cnt(hits.size(), -1);
        for (int i = 0; i < m; i++) {
            if (grid[0][i]) {
                st.push(make_pair(grid[0][i], make_pair(0, i)));
                dist[0][i] = grid[0][i];
                if (dist[0][i] - 2 < cnt.size()) {
                    cnt[dist[0][i] - 2]++;
                }
            }
        }
        int dx[] = {0, 1, 0, -1}, dy[] = {-1, 0, 1, 0};
        while (!st.empty()) {
            int val = st.top().first;
            int x = st.top().second.first, y = st.top().second.second;
            // cout << val << "\t" << x << "\t" << y << endl;
            st.pop();
            for (int dir = 0; dir < 4; dir++) {
                int nx = x + dx[dir], ny = y + dy[dir];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m || dist[nx][ny] != 0 || grid[nx][ny] == 0) {
                    continue;
                }
                dist[nx][ny] = min(grid[nx][ny], dist[x][y]);
                if (dist[nx][ny] - 2 < cnt.size()) {
                    cnt[dist[nx][ny] - 2]++;
                }
                st.push(make_pair(dist[nx][ny], make_pair(nx, ny)));
                // cout << "push:\t" << dist[nx][ny] << "\t" << nx << "\t" << ny << endl;
            }
        }
        for (int i = 0; i < cnt.size(); i++) {
            cnt[i] = max(0, cnt[i]);
        }
        return cnt;
    }
};
