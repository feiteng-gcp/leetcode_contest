class Solution {
public:
    int n, m;
    int root;
    vector<int> par, sz;
    map< pair<int,int>, int > hitCnt;
    
    int find(int x){
        if (par[x]==x) return x;
        return par[x] = find(par[x]);
    }
    
    void merge(int u, int v){
        u = find(u), v= find(v);
        if (u==v) return;
        par[v] = u;
        sz[u] += sz[v];
    }
    
    bool ok(int x, int y){
        if (x<0 || x>=n) return 0;
        if (y<0 || y>=m) return 0;
        return 1;
    }
    
    int idx(int x, int y){
        return x*m + y;
    }
    
    void connect(vector<vector<int>>& grid, int x, int y){
        if ( grid[x][y]==0 )return;
        
        int dx[] = {1,0,-1,0};
        int dy[] = {0,1,0,-1};
        for (int dir=0; dir<4; dir++){
            int nx = x+dx[dir];
            int ny = y+dy[dir];
            if ( !ok(nx,ny) ) continue;
            if ( grid[nx][ny]==1 )merge( idx(x,y), idx(nx,ny) );
        }
        if (x==0){
            merge(root,idx(x,y));
        }
    }
    
    void init(vector<vector<int>>& grid, vector<vector<int>>& hits){
        n = grid.size();
        m = grid[0].size();
        root = n*m;
        for (int i=0; i<=n*m; i++){
            par.push_back(i);
            sz.push_back(1);
        }
        for (auto v:hits){
            if ( hitCnt[{v[0],v[1]}]++ == 0 ){
                if (grid[ v[0] ][ v[1] ]==0){
                    hitCnt[{v[0],v[1]}]++;
                }
                grid[ v[0] ][ v[1] ]=0;   
            }
        }
        
        for (int i=0; i<n; i++){
            for (int j=0; j<m; j++){
                connect(grid,i,j);
            }
        }
        
        sz[root]=0;
        for (int i=0; i<m; i++){
            if (grid[0][i]==1){
                merge( root, idx(0,i));
            }
        }
    }
    
    vector<int> hitBricks(vector<vector<int>>& grid, vector<vector<int>>& hits) {
        vector<int> ans;
        
        init(grid,hits);
        
        int Q = hits.size();
        
        for (int i=Q-1; i>=0; i--){
            auto &v = hits[i];
            int x=v[0], y=v[1];
            
            if ((--hitCnt[{x,y}])!=0){
                ans.push_back(0);
                continue;
            }
            
            grid[x][y]=1;
            int last = sz[ find(root) ];
            connect(grid,x,y);
            
            int delta = sz[find(root)]-last-1;
            
            ans.push_back( max(delta,0) );
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
