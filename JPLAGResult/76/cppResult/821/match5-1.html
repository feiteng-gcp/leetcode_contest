int neigh[4][2]={{0,1},{1,0},{0,-1},{-1,0}};

class Solution {

public:
    vector<int> hitBricks(vector<vector<int> >& grid, vector<vector<int> >& hits) {
      int i,j,k,x,y,m,n,cnt,xx,yy,b1;
      vector<int> ans;
      ans.clear();
      vector<pair<int,int> > q;
      m=grid.size();
      n=grid[0].size();
      for (i=0;i<hits.size();i++)
      {
        x=hits[i][0];
        y=hits[i][1];
        if (grid[x][y]==1)
        {
          grid[x][y]=-(i+1);
        }
      }
      q.clear();
      for (i=0;i<grid[0].size();i++)
        if (grid[0][i]==1)
        {
          q.push_back(make_pair(0,i));
          grid[0][i]=2;
        }
      for (i=0;i<q.size();i++)
      {
        x=q[i].first;
        y=q[i].second;
        for (j=0;j<4;j++)
        {
          xx=x+neigh[j][0];
          yy=y+neigh[j][1];
          if ((xx>=0)&&(xx<m)&&(yy>=0)&&(yy<n)&&(grid[xx][yy]==1))
          {
            grid[xx][yy]=2;
            q.push_back(make_pair(xx,yy));
          }
        }
      }
      for (i=hits.size()-1;i>=0;i--)
      {
        x=hits[i][0];
        y=hits[i][1];
        cnt=0;
        if (grid[x][y]==-(i+1))
        {
          grid[x][y]=1;
          b1=0;
          for (j=0;j<4;j++)
          {
            xx=x+neigh[j][0];
            yy=y+neigh[j][1];
            if ((xx>=0)&&(xx<m)&&(yy>=0)&&(yy<n)&&(grid[xx][yy]==2))
            {
              grid[x][y]=2;
              q.push_back(make_pair(x,y));
              b1=1;
              break;
            }
          }
          if ((b1==0)&&(x==0))
          {
            b1=1;
            grid[x][y]=2;
            q.push_back(make_pair(x,y));
          }
          if (b1==1)
          {
            for (k=q.size()-1;k<q.size();k++)
            {
              x=q[k].first;
              y=q[k].second;
              for (j=0;j<4;j++)
              {
                xx=x+neigh[j][0];
                yy=y+neigh[j][1];
                if ((xx>=0)&&(xx<m)&&(yy>=0)&&(yy<n)&&(grid[xx][yy]==1))
                {
                  grid[xx][yy]=2;
                  q.push_back(make_pair(xx,yy));
                  cnt++;
                }
              }
            }
          }
        }
        ans.push_back(cnt);
      }
      reverse(ans.begin(),ans.end());
      return ans;
    }
};
