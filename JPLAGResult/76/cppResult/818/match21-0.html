class Solution {
public:
    string similarRGB(string color) {
        string res = "#";
        for (int i=1; i<6; i+=2) {
            char firstc = color[i], secondc = color[i+1];
            int first = 0, second = 0;
            if (isdigit(firstc))
                first = firstc-'0';
            else
                first = firstc-'a'+10;
            if (isdigit(secondc))
                second = secondc-'0';
            else
                second = secondc-'a'+10;
            int n = first*16+second;            
            int c = (n%17 < (17-n%17)) ? n/17 : n/17+1;
            res += c>9 ? c-10+'a' : c+'0';
        }
        return res;
    }
};
