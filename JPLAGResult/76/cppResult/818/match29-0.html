class Solution {
public:
    int getVal(char c) {
        if(c >= '0' && c <= '9') return c - '0';
        return 10 + (c - 'a');
    }
    string similarRGB(string color) {
        vector<int> v;
        for(int i = 1; i < 7; i += 2) {
            int t = 16 * getVal(color[i]) + getVal(color[i+1]);
            v.push_back(t);
        }
        int mt = INT_MAX;
        string ans;
        for(int i = 0; i < 16; i ++) {
            for(int j = 0; j < 16; j ++) {
                for(int k = 0; k < 16; k ++) {
                    int d1 = -(i * 16 + i - v[0]) * (i * 16 + i - v[0]);
                    int d2 = -(j * 16 + j - v[1]) * (j * 16 + j - v[1]);
                    int d3 = -(k * 16 + k - v[2]) * (k * 16 + k - v[2]);
                    if(abs(d1 + d2 + d3) < mt) {
                        mt = abs(d1 + d2 + d3);
                        ans = "#";
                        addOne(ans, i);
                        addOne(ans, j);
                        addOne(ans, k);
                    }
                }
            }
        }
        return ans;
    }
    void addOne(string & ans, int x) {
        if(x <= 9) ans += (x + '0');
        else ans += (x - 10 + 'a');
    }
};
