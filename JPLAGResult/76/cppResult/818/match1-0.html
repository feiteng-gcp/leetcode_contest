class Solution {
public:
    string similarRGB(string color) {
        string rgb = "#";
        for (int i = 0; i < 3; ++i) {
          int ab = 0;
          if (isdigit(color[i * 2 + 1])) {
            ab += 16 * (color[i * 2 + 1] - '0');
          } else {
            ab += 16 * (color[i * 2 + 1] - 'a' + 10);
          }
          if (isdigit(color[i * 2 + 2])) {
            ab += color[i * 2 + 2] - '0';
          } else {
            ab += color[i * 2 + 2] - 'a' + 10;
          }
          int m  = INT_MAX;
          int r = 0;
          for (int j = 0; j <= 16; ++j) {
            if (abs(j * 16 + j - ab) < m) {
              m = abs(j * 16 + j - ab);
              r = j;
            }
          }
            cout << ab << endl;
          if (r < 10) {
            rgb.push_back(r + '0');
          } else {
            rgb.push_back(r + 'a' - 10);
          }
        }
        return rgb;
    }
};
