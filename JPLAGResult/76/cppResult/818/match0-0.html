class Solution {
public:
    string similarRGB(string color) {
        int a,b,c;
        int tmp[6];
        for(int i=1;i<7;i++){
            if('0'<=color[i]&&color[i]<='9')
                tmp[i-1]=color[i]-'0';
            if('a'<=color[i]&&color[i]<='f')
                tmp[i-1]=10+color[i]-'a'; 
        }
        a=tmp[0]*16+tmp[1];
        b=tmp[2]*16+tmp[3];
        c=tmp[4]*16+tmp[5];
        vector<int>all;
        for(int i=0;i<16;i++){
            all.push_back(i*16+i);
        }
        string rtn="#";
        int smallest=20000005,num=-1;
        for(int i=0;i<all.size();i++){
            if((all[i]-a)*(all[i]-a)<smallest){
                smallest=(all[i]-a)*(all[i]-a);
                num=all[i]%16;
            }
        }
        string s;
        if(num>=10)
            s='a'+(num-10);
        else 
            s='0'+num;
        rtn= rtn+s+s;
        smallest=20000005,num=-1;
        for(int i=0;i<all.size();i++){
            if((all[i]-b)*(all[i]-b)<smallest){
                smallest=(all[i]-b)*(all[i]-b);
                num=all[i]%16;
            }
        }
        if(num>=10)
            s='a'+(num-10);
        else 
            s='0'+num;
        rtn= rtn+s+s;
        smallest=20000005,num=-1;
        for(int i=0;i<all.size();i++){
            if((all[i]-c)*(all[i]-c)<smallest){
                smallest=(all[i]-c)*(all[i]-c);
                num=all[i]%16;
            }
        }
        if(num>=10)
            s='a'+(num-10);
        else 
            s='0'+num;
        rtn= rtn+s+s;
        return rtn;
    }
};
