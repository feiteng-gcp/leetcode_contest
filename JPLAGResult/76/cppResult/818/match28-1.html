class Solution {
public:

    string similarRGB(string color) {
        int l=color.length();
        string s=color;
        string ans="#";
        for(int i=1;i<l;i+=2)
        {
            int o=0;
            if(s[i]<='9'&&s[i]>='0')
            {
                o+=(s[i]-'0')*16;
            }
            if(s[i]>='a'&&s[i]<='f')
            {
                o+=(s[i]-'a'+10)*16;
            }

            if(s[i+1]<='9'&&s[i+1]>='0')
            {
                o+=(s[i+1]-'0');
            }
            if(s[i+1]>='a'&&s[i+1]<='f')
            {
                o+=(s[i+1]-'a'+10);
            }
            int mi=99999999,p=0;
            for(int j=0;j<=15;j++)
            {
                if(abs(j*17-o)<mi)
                {
                    p=j;
                    mi=abs(j*17-o);
                }
            }
            if(p<10)
                ans+=char(p+'0');
            else
                ans+=char(p+'a'-10);
        }
        return ans;
    }
};
