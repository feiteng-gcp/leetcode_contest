class Solution {
public:
    string similarRGB(string color) {
        int len = color.size();
        vector<int> retnum(3);
        int cnt = 0;
        for (int i = 1; i < len; i += 2) {
            char first = color[i];
            char sec = color[i + 1];
            int tmp = to_num(first) * 16 + to_num(sec);
            int low = tmp / 17;
            int high = low + 1 > 15 ? 15 : low + 1;
            if (abs(high * 17 - tmp) < abs(low * 17 - tmp)) {
                retnum[cnt++] = high;
            } else {
                retnum[cnt++] = low;
            }
        }
        
        string ret = "#";
        for (int i = 0; i < 3; i++) {

            ret += to_char(retnum[i]);
        } 
        
        return ret;
    }
    
    int to_num(char c) {
        if (isdigit(c)) {
            return c - '0';
        } 
        return (c - 'a') + 10;
    }
    char to_char(int i) {
        if (i < 10) {
            return i + '0';
        } else {
            return (i - 10) + 'a';
        }
    }
};
