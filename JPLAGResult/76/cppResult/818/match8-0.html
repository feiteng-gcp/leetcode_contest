class Solution {
public:
    int fromHex(char c) {
        if (c >= '0' && c <= '9') return c - '0';
        return c - 'a' + 10;
    }
    char toHex(int c) {
        if (c <= 9) return c + '0';
        return c + 'a' - 10;
    }
    int sq(int a) { return a * a; }
    string similarRGB(string color) {
        int r = fromHex(color[1]) * 16 + fromHex(color[2]);
        int g = fromHex(color[3]) * 16 + fromHex(color[4]);
        int b = fromHex(color[5]) * 16 + fromHex(color[6]);
        
        int br = 0, bg = 0, bb = 0;
        for (int i = 0; i < 16; i++) {
            if (sq(br * 16 + br - r) > sq(i * 16 + i - r)) br = i;
                        if (sq(bg * 16 + bg - g) > sq(i * 16 + i - g)) bg = i;
                        if (sq(bb * 16 + bb - b) > sq(i * 16 + i - b)) bb = i;
        }
        
        char cr = toHex(br), cg = toHex(bg), cb = toHex(bb);
        string s = "#      ";
        s[1] = s[2] = cr;
        s[3] = s[4] = cg;
        s[5] = s[6] = cb;
        return s;
    }
};
