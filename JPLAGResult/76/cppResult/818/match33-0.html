class Solution {
	int decode(char ch) {
		if ((ch >= 'a') && (ch <= 'f')) return (ch - 'a' + 10);
		return ch - '0';
	}
	char encode(int a) {
		if (a < 10) return '0' + a;
		return 'a' + (a - 10);
	}
public:
	string similarRGB(string color) {
		int a[6], b[6], i, j, k, t1, t2, t3, best, p;
		string id; 
		
		for (i = 0; i < 6; i++) {
			a[i] = decode(color[i + 1]);
		}
		
		best = 10000000;
		for (t1 = 0; t1 < 16; t1++) {
			for (t2 = 0; t2 < 16; t2++)
				for (t3 = 0; t3 < 16; t3++) {
					int p;
					p = ((a[0] * 16 + a[1]) - (t1 * 16 + t1)) * ((a[0] * 16 + a[1]) - (t1 * 16 + t1));
					p += ((a[2] * 16 + a[3]) - (t2 * 16 + t2)) * ((a[2] * 16 + a[3]) - (t2 * 16 + t2));
					p += ((a[4] * 16 + a[5]) - (t3 * 16 + t3)) * ((a[4] * 16 + a[5]) - (t3 * 16 + t3));
					if (p < best) {
						best = p;
						char c1 = encode(t1);
						char c2 = encode(t2);
						char c3 = encode(t3);
						id = "#";
						if (t1 < 10) id = id + char('0' + t1); else id = id + (char)('a' + (t1 - 10));
						if (t2 < 10) id = id + char('0' + t2); else id = id + (char)('a' + (t2 - 10));
						if (t3 < 10) id = id + char('0' + t3); else id = id + (char)('a' + (t3 - 10));
					}
				}
		}
		return id;
	}
};
