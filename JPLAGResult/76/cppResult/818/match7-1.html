class Solution {
public:
    string similarRGB(string color) {
        return "#"+sub(color.substr(1,2))+sub(color.substr(3,2))+sub(color.substr(5,2));
    }
    string f(string s){
        if(s[0]==s[1]) return s.substr(0,1);
        int x=0, y=0;
        int n=stoi(s,0,16), m=n;
        while(n/10 != n%10 || (n<10 and n!=0))
            n--, x++;
        while(m/10 != m%10)
            m++, y++;
        if(x<y) return n>10?to_string(n/10):to_string(n);
        else return m>10?to_string(m/10):to_string(m);
    }

    map<char,string> m={{'0',"0"},
                       {'1',"11"},{'2',"22"},{'3',"33"},{'4',"44"},{'5',"55"},{'6',"66"},
                       {'7',"77"},{'8',"88"},{'9',"99"},{'a',"aa"},{'b',"bb"},{'c',"cc"},
                       {'d',"dd"},{'e',"ee"},{'f',"ff"}
                      };
    string sub(string s){
        if(s[0]==s[1]) return s.substr(0,1);
        int x=INT_MAX,y=0;
        string r;
        for(auto e: m){
            int t=abs(stoi(e.second,0,16)-stoi(s,0,16));
            if(t<x) x=t, r=string(1,e.first);
        }
        return r;
    }
};
