class Solution {
public:
    int ToInt(char c1, char c2) {
        int n1 = 0, n2 = 0;
        n1 = isdigit(c1) ? (c1 - '0') : (c1 - 'a' + 10);
        n2 = isdigit(c2) ? (c2 - '0') : (c2 - 'a' + 10);
        return n1 * 16 + n2;
    }
    
    string similarRGB(string color) {
        string ans = "#";
        for (int i=1; i<color.size(); i+=2) {
            char c1 = tolower(color[i]), c2 = tolower(color[i+1]);
            if (c1 == c2) {
                ans.push_back(c1);
            } else {
                int n = ToInt(c1, c2), diff = INT_MIN, m = -1;
                //cout << n << endl;
                for (int i=0; i<16; i++) {
                    int d = -1 * (i*16+i - n) * (i*16+i - n);
                    if (d > diff) {
                        diff = d;
                        m = i;
                    }
                }
                //cout << m << endl;
                ans.push_back(m > 9? ('a' + m-10) : '0' + m);
            }
        }
        return ans;
    }
};
