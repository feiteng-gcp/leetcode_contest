class Solution {
public:
    string similarRGB(string color) {
        string ans = "#";
        const char * cc = color.c_str();
        char c1[2],c2[2],c3[2];
        c1[0]  = cc[1];
        c1[1]  = cc[2];
        c2[0]  = cc[3];
        c2[1]  = cc[4];
        c3[0]  = cc[5];
        c3[1]  = cc[6];
        int a[3];
        sscanf(c1,"%x",&a[0]);
        sscanf(c2,"%x",&a[1]);
        sscanf(c3,"%x",&a[2]);
        //cout << a[0] << a[1] << a[2];
        for (int k = 0;k < 3;k++) {
            int ta = (0-a[k])*(0-a[k]),tr = 0;
        for (int i = 1;i < 16;i++) {
            int t = i * 16 + i;
            int tp = (t-a[k])*(t-a[k]);
            if (tp < ta) {
                ta = tp;
                tr = i;
            }
        }
            sprintf(c1,"%x",tr);
            ans += char(c1[0]);
        }
        return ans;
    }
};
