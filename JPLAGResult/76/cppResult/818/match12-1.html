class Solution {
public:
    int cc[3];
    int hexmap[128];
    char hex[16];
    
    string similarRGB(string color) {
        for (int i = 0; i < 10; i ++) {
            hexmap[i + '0'] = i;
            hex[i] = i + '0';
        }
        for (int i = 10; i < 16; i ++) {
            hexmap[i - 10 + 'a'] = i;
            hex[i] = i - 10 + 'a';
        }
        // Parse color
        parseColor(color);

        int minCC[3], CC[3];
        int minSimilarity = INT_MAX;
        for (int i = 0; i < 16; i ++) {
            for (int j = 0; j < 16; j ++) {
                for (int k = 0; k < 16; k ++) {
                    CC[0] = i * 16 + i, CC[1] = j * 16 + j, CC[2] = k * 16 + k;
                    auto sim = similarity(CC, cc);
                    if (sim < minSimilarity) {
                        minSimilarity = sim;
                        minCC[0] = i, minCC[1] = j, minCC[2] = k;
                    }
                }
            }
        }
        // Print out
        string p = "#";
        for (int i = 0; i < 3; i ++) {
            p.push_back(hex[minCC[i]]);
        }
        return p;
    }
    
    void parseColor(string& color) {
        int j = 0;
        for (int i = 1; i < color.size() - 1; i += 2) {
            auto s = color.substr(i, 2);
            cc[j ++] = hexToInt(s);
        }
        // cout << cc[0] << " " << cc[1] << " " << cc[2] << endl;
    }
    
    int hexToInt(string s) {
        return hexmap[s[0]] * 16 + hexmap[s[1]];
    }
    
    int similarity(int a[3], int b[3]) {
        return (a[0] - b[0]) * (a[0] - b[0]) + \
               (a[1] - b[1]) * (a[1] - b[1]) + \
               (a[2] - b[2]) * (a[2] - b[2]);
    }
};
