class Solution {
public:
    string similarRGB(string color) {
        string r = find(color.substr(1, 2));
        string g = find(color.substr(3, 2));
        string b = find(color.substr(5, 2));
        return "#"+r+g+b;
    }
    
    string find(string s)
    {
        map<char, int> m;
        for(int i = 0; i < 10; ++i)
        {
            m.insert({'0' + i, i});
        }
        
        for(int i = 0; i < 6; ++i)
        {
            m.insert({'a' + i, 10 + i});
        }
        
        int val = m[s[0]] * 16 + m[s[1]];
        int dis = INT_MAX;

        int res = 0;
        for(int i = 0; i < 16; ++i)
        {
            if(abs(i * 16 + i - val) < dis)
            {
                dis = abs(i * 16 + i - val);
                res = i;
            }
        }
        if(res < 10)
        {
            return string(1, '0' + res);
        }
        else
        {
            return string(1, 'a' + res - 10);
        }
    }
};
