class Solution {
public:
    string pickNear(map<char, int>& h2d, string s)
{
    char ans = '0';
    int n = 16 * h2d[s[0]] + h2d[s[1]];
    int minN = n;
    for(auto p = h2d.begin(); p != h2d.end(); p++)
    {
        int nn = 17 * p->second;
        if(minN > abs(n - nn))
        {
            minN = abs(n - nn);
            ans = p->first;
        }
    }
    return string(1, ans);
}

string similarRGB(string color)
{
    map<char, int> h2d;
    for(int i = 0; i <= 9; i++) h2d['0' + i] = i;
    for(char c = 'a'; c <= 'f'; c++) h2d[c] = c - 'a' + 10;
    return "#" + pickNear(h2d, color.substr(1, 2)) +
    pickNear(h2d, color.substr(3, 2)) +
    pickNear(h2d, color.substr(5, 2));
}
};
