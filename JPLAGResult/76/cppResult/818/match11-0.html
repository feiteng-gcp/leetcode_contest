class Solution {
public:
    string similarRGB(string color) {
        istringstream isa(color.substr(1,2));
        istringstream isb(color.substr(3,2));
        istringstream isc(color.substr(5,2));
        int a,b,c;
        isa>>hex>>a;
        isb>>hex>>b;
        isc>>hex>>c;
        int aa,bb,cc,mx=-99999;
        //cout<<a<<" "<<b<<" "<<c;
        for(int i=0;i<16;++i)
        {
            for(int j=0;j<16;++j)
            {
                for(int k = 0;k<16;++k)
                {
                    int cnt = -(i*16+i-a)*(i*16+i-a)-(j*16+j-b)*(j*16+j-b)-(k*16+k-c)*(k*16+k-c);
                    if(cnt > mx)
                    {
                        aa=i;
                        bb=j;
                        cc=k;
                        mx = cnt;
                        //cout<<endl<<i<<" "<<j<<" "<<k<<endl;
                        //cout<<endl<<i*16+i<<" "<<j*16+j<<" "<<k*16+k<<" "<<mx<<endl;
                        //cout<<endl<<i*16+i-a<<" "<<j*16+j-b<<" "<<k*16+k-c<<" "<<mx<<endl;
                    }
                }
            }
        }
        ostringstream os;
        //cout<<mx<<endl;
        os<<'#'<<hex<<aa<<bb<<cc;
        return os.str();
    }
};
