class Solution {
public:
  inline int h(char c){
    return isdigit(c)?c-'0':c-'a'+10;
  }
  inline char inv(int n){
    return n<10?'0'+n:'a'+n-10;
  }
  inline char sim(string s){
    int a=h(s[0])*16+h(s[1]);
    int best=1e9;
    char res;

    for(int x=0;x<16;x++){
      int X=x*16+x;
      if((X-a)*(X-a)<best){
	best=(X-a)*(X-a);
	res=inv(x);
      }
    }
    return res;
  }
  
    string similarRGB(string cr) {
      int a,b,c;
      a=h(cr[1])*16+h(cr[2]);
      b=h(cr[3])*16+h(cr[4]);
      c=h(cr[5])*16+h(cr[6]);
      int best=1e9;
      string res="#";
      res+=sim(cr.substr(1,2));
      res+=sim(cr.substr(3,2));
      res+=sim(cr.substr(5,2));
      cout<<a<<' '<<b<<' '<<c<<'\n';
      return res;
    }
};
