class Solution {
public:
    int htod(char c) {
        if(c >= '0' && c <= '9')
            return c - 48;
        else
            return c - 'a' + 10;
    }
    
    char dtoh(int a) {
        if(a < 10)
            return a + '0';
        else
            return a - 10 + 'a';
    }
    
    int closest(int a1, int a2) {
        int m1 = a2 + 16 - a1 + 1;
        int m2 = abs(a2 - a1);
        int m3 = a1 + 1 + 16 - a2;
        if(a1 == 0) m1 = 256;
        if(a1 == 15) m3 = 256;
        if(m1 <= m2 && m2 <= m3)
            return a1 - 1;
        else if(m2 <= m1 && m2 <= m3)
            return a1;
        else
            return a1 + 1;
    }
    
    string similarRGB(string color) {
        string answer = "#";
        int num[7];
        for(int i = 1; i <= 6; i++)
            num[i] = htod(color[i]);
        char c = dtoh(closest(num[1], num[2]));
        answer += c;
        c = dtoh(closest(num[3], num[4]));
        answer += c;
        c = dtoh(closest(num[5], num[6]));
        answer += c;
        return answer;
    }
};
