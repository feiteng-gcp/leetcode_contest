class Solution {
public:
    string similarRGB(string color) { 
        vector<int> vec;
        map<int, char> m;
        for(char c='0'; c <='9'; c++){
            m[c-'0'] = c;
            //cout<<c-'0'<<c<<endl;
        }
        for(char c='a'; c<='f';c++){
            m[c-'a'+10] = c;
            //cout<<c-'a'+10<<c<<endl;
        }
        for(int i=1; i<color.size();i++){
            char c = color[i];
            if(c >= '0' && c<='9') vec.push_back(c-'0');
            else if( c >= 'a' && c <= 'f') vec.push_back(c-'a'+10);
        }
        int first = vec[0]*16+vec[1];
        int second = vec[2]*16 + vec[3];
        int third = vec[4]*16 + vec[5];
        int diff = INT_MAX;
        string s;
        for(int x = 0; x<16; x++){
            int xv = x*16 +x;
            for(int y=0; y<16;y++){
                int yv = y*16+y;
                for(int z=0; z<16;z++){
                    int zv = z*16 +z;
                    int sum = - (xv-first)*(xv-first) - (yv-second)*(yv-second) - (zv-third)*(zv-third);
                    if(abs(sum) < abs(diff)){
                        //cout<<x<<","<<m[x]<<","<<y<<","<<m[y]<<" z"<<z<<","<<m[z]<<endl;
                        s = '#'; s+= m[x]; s += m[y]; s+= m[z];
                        diff = sum;
                    }
                }
            }
        }
        return s;
    }
};
