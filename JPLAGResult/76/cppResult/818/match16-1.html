int getv(char c)
{
  if (c<='9') return c-'0';
  else return c-'a'+10;
}

char getc(int x)
{
  if (x<=9) return '0'+x;
  else return 'a'+x-10;
}

class Solution {
public:
    string similarRGB(string color) {
      int i,j,k,best,v1,v2,v3;
      char ans[100];
      v1=getv(color[1])*16+getv(color[2]);
      v2=getv(color[3])*16+getv(color[4]);
      v3=getv(color[5])*16+getv(color[6]);
      best=2000000000;
      for (i=0;i<16;i++)
        for (j=0;j<16;j++)
          for (k=0;k<16;k++)
          {
            if ((v1-i*16-i)*(v1-i*16-i)+(v2-j*16-j)*(v2-j*16-j)+(v3-k*16-k)*(v3-k*16-k)<best)
            {
              best=(v1-i*16-i)*(v1-i*16-i)+(v2-j*16-j)*(v2-j*16-j)+(v3-k*16-k)*(v3-k*16-k);
              sprintf(ans,"#%c%c%c",getc(i),getc(j),getc(k));
            }
          }
      return ans;
    }
};
