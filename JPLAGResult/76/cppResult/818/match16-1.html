class Solution {
public:
    int opp(char ch)
    {
        if(ch<='9'&&ch>='0')return ch-'0';
        else return ch-'a'+10;
    }
    char op(char c,char ch)
    {
        int a=opp(c),b=opp(ch);
        if(abs(b-a)>=9&&a>b&&a>0){
            if(c=='a')return '9';
            else return c-1;
        }
        else if(a<b&&abs(b-a)>=9&&a<15){
            if(c=='9')return 'a';
            else return c+1;
        }
        else return c;
    }
    string similarRGB(string color) {
        for(int i=1;i<=5;i+=2){
            char c=op(color[i],color[i+1]);
            color[i]=c;
            color[i+1]=c;
        }
        return color;
    }
};
