class Solution {
public:
  string similarRGB(string color) {
    string res = "#";
    for (int i = 0; i < 3; ++i) {
      string tmp = string() + color[i * 2 + 1] + color[i * 2 + 2];
      res += similarString(tmp);
    }
    return res;
  }
  int char_to_int(char c) {
    if (c <= '9' && c >= '0') {
      return c - '0';
    }
    return c - 'a' + 10;
  }
  char int_to_char(int c) {
    if (c < 10) {
      return c + '0';
    }
    return c + 'a' - 10;
  }
  string similarString(const string &s) {
    string res = "";
    int res_sum = 1 << 30;
    int tmp = char_to_int(s[0]) * 16 + char_to_int(s[1]);
    for (int i = 0; i < 16; ++i) {
      int x = i * 16 + i;
      if (abs(tmp - x) < res_sum) {
        res_sum = abs(tmp - x);
        res = string() + int_to_char(i);
      }
    }
    return res;
  }
};
