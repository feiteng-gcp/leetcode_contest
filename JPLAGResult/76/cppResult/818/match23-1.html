class Solution {
public:
    string similarRGB(string color) {
        vector<char> candidate;
        vector<int> candidateNum;
        for(int i = 0; i < 16; i++)
        {
            char c = i > 9?'a'+ (i-10):i+'0';
            string s = "";
            s = s + c; s = s + c;
            //cout<< s << endl;
            candidate.push_back(s[0]);
            candidateNum.push_back(getNum(s));
        }
        
        string result = "#";
        
        string r = color.substr(1,2);
        int r_num = getNum(r);
        result += candidate[getNearest(r_num, candidateNum)];
        
        string g = color.substr(3,2);
        int g_num = getNum(g);
        result += candidate[getNearest(g_num, candidateNum)];
        
        string b = color.substr(5,2);
        int b_num = getNum(b);
        result += candidate[getNearest(b_num, candidateNum)];
        
        return result;
    }
    int getNum(string r)
    {
        int r_num = 0; 
        r_num += r[0]>'9'? r[0]-'a'+ 10: r[0]-'0';
        r_num = r_num * 16;
        r_num += r[1]>'9'? r[1]-'a'+ 10: r[1]-'0';
        return r_num;
    }
    int getNearest(int s, vector<int> candidate)
    {

        int minNum = INT_MAX;
        int minIdx = 0;
        for(int i = 0; i < 16; i++)
        {
            int aa = (candidate[i] - s) * (candidate[i] - s);
            if(aa < minNum)
            {
                minNum = aa;
                minIdx = i;
            }
        }
        return minIdx;
    }
};
