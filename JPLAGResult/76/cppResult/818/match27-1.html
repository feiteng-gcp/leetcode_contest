class Solution {
public:
    char hex[17] = {"0123456789abcdef"};
    
    int toInt(char c) {
        if (c <= '9') return c-'0';
        else return c - 'a' + 10;
    }
    
    int toInt(string s) {
        return toInt(s[0]) * 16 + toInt(s[1]);
    }
    
    string similarRGB(string color) {
        string out = "#";
        for (int i=0;i<3;++i) {
            int n = toInt(color.substr(1+i*2, 2));
            int best = 0;
            int mindiff = (1<<30);

            for (int j=0;j<16;++j) {
                int diff = (j*16+j - n);
                diff = diff*diff;
                if (diff < mindiff) {
                    mindiff = diff;
                    best = j;
                }
            }
            out.append(1, hex[best]);
        }
        return out;
    }
};
