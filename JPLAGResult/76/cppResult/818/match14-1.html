class Solution {
public:
    int h2d(char c) {
        if((int)(c-'0')<=9)
            return (int)(c-'0'); 
        return (int)(10+(int)(c-'a')); 
    }
    
    int dist(int a, int b) {
        return (a-b)*(a-b); 
    }
    
    char n2h(int n) {
        if(n<10)
            return '0'+n; 
        return (char)('a'+(n-10)); 
    }
    
    string similarRGB(string color) {
        string res="#"; 
        for(int i=1; i<=5; i=i+2) {
            int num_1=h2d(color[i])*16+h2d(color[i+1]); 
            int record=INT_MAX; 

            int dig=0; 
            for(int j=0; j<16; j++) {
                int num_2=j*16+j; 
                int tmp=dist(num_1, num_2); 
                if(tmp<record) {
                    dig=j; 
                    record=tmp; 
                }
            }
            char c=n2h(dig); 
            res.push_back(c); 
        }
        return res; 
    }
};
