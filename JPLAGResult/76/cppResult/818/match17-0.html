class Solution {
    int getrep(int i){
        return i * 16 + i;
    }
    char getcharr(int i){
        assert(i >= 0);
        if(i < 10) return '0' + i;
        i -= 10;
        return 'a' + i;
    }
    
    int getval(char c){
        if(isdigit(c)) return c-'0';
        return 10 + (c - 'a');
    }
    
public:
    
    string similarRGB(string color) {
        string ans = "#";
        for(int j=0;j<3;j++){
            int n1 = getval(color[1 + 2*j]) * 16 + getval(color[1+ 2*j +1]);
            int minval = -1, mindiff =  INT_MAX;
            for(int i=0;i<16*16;i++){
                if(mindiff > abs(n1 - getrep(i))){
                    mindiff = abs(n1-getrep(i));
                    minval = i;
                }
            }
            
            ans += getcharr(minval);
        }
        
        return ans;
    }
};
