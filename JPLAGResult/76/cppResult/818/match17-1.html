class Solution {
    int ctoi (char a) {
        if(a >= '0' && a <= '9')
            return a - '0';
        else return tolower(a) - 'a' + 10;
    }

    char simto(char a, char b) {
        int d = ctoi(a) * 16 + ctoi(b);
        int mi = 999999999, k = 0;
        for(int i = 0; i < 16; i++) {
            //cout << i * 16 + i << ' ' << d << ' ' << mi << endl;
            if((i * 16 + i - d) * (i * 16 + i - d) < mi) {
                mi = (i * 16 + i - d) * (i * 16 + i - d);
                k = i;
            }
        }
        if(k < 10)
            return k + '0';
        else return k - 10 + 'a';
    }
public:
    string similarRGB(string color) {
        string s = "#      ";
        s[1] = s[2] = simto(color[1], color[2]);
        s[3] = s[4] = simto(color[3], color[4]);
        s[5] = s[6] = simto(color[5], color[6]);
        return s;
    }
};
