class Solution {
public:
    string similarRGB(string color) {
        string srgp = "#";

        for(int i = 0; i < 3; i ++) {
            int value = 0, num = 0;
            for(int j = 0; j < 2; j++) {
                if (color[i * 2 + j + 1] >= '0' && color[i * 2 + j + 1] <= '9') {
                    num = color[i * 2 + j + 1] - '0';
                } else {
                    num = color[i * 2 + j + 1] - 'a' + 10;
                }
                value = value * 16 + num;
            }
            cout << value << endl;
            int mini = 0;
            for(int i = 0; i < 16; i++) {
                if (abs(value - i * 17) < abs(value - mini * 17)) mini = i;
            }
            srgp.append(1, mini < 10 ? mini + '0' : mini - 10 + 'a');
        }
        return srgp;
    }
};
