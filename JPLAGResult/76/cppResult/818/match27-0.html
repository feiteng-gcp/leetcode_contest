class Solution {
public:
    int arr[16];
    
    string similarRGB(string color) {
        
        for (int i = 0; i < 16; ++i)
            arr[i] = i * 17;
        
        return string() + '#' + process(color[1], color[2]) + process(color[3], color[4]) + process(color[5], color[6]);
        
    }
    
    char process(char a, char b)
    {
        int x = 0;
        
        if (a >= 'a' && a <= 'f')
            x += (a - 'a' + 10) * 16;
        else if (a >= 'A' && a <= 'F')
            x += (a - 'A' + 10) * 16;
        else
            x += (a - '0') * 16;
        
        if (b >= 'a' && b <= 'f')
            x += (b - 'a' + 10);
        else if (b >= 'A' && b <= 'F')
            x += (b - 'A' + 10);
        else
            x += (b - '0');
        
        int diff_min = 999999999;
        int diff_t;
        
        for (int i = 0; i < 16; ++i)
        {
            int diff = x - arr[i];
            diff *= diff;
            
            //cout << i << ' ' << diff << endl;
            
            if (diff_min > diff)
            {
                diff_min = diff;
                diff_t = i;
            }
        }
        
        //cout << diff_t << endl;
        
        if (diff_t < 10)
            return diff_t + '0';
        else
            return diff_t - 10 + 'a';
    }
};
