class Solution {
public:

    int toInt(char c) {
        if (c >= 'a' && c <= 'f') return c - 'a' + 10;
        return c - '0';
    }
    char toChar(int n) {
        if (n >= 0 && n < 10) return '0' + n;
        return n - 10 + 'a';
    }
    string similarRGB(string color) {
        string rv = "#";
        int n = toInt(color[1]) * 16 + toInt(color[2]);
        int diff = 16 * 16 * 16;
        int s = -1;
        for (int i = 0; i < 16; i++) {
            if (abs(i * 16 + i - n) < diff) {
                s = i;
                diff = abs(i * 16 + i - n);
            }
        }
        rv += toChar(s);
        n = toInt(color[3]) * 16 + toInt(color[4]);
        diff = 16 * 16 * 16;
        s = -1;
        for (int i = 0; i < 16; i++) {
            if (abs(i * 16 + i - n) < diff) {
                s = i;
                diff = abs(i * 16 + i - n);
            }
        }
        rv += toChar(s);
        n = toInt(color[5]) * 16 + toInt(color[6]);
        diff = 16 * 16 * 16;
        s = -1;
        for (int i = 0; i < 16; i++) {
            if (abs(i * 16 + i - n) < diff) {
                s = i;
                diff = abs(i * 16 + i - n);
            }
        }
        rv += toChar(s);
        return rv;   
    }
};
