class Solution {
public:
    string similarRGB(string color) {
        string res = "#";
        for(int i = 1; i < 6; i += 2) {
            string tmp = color.substr(i, 2);
            int num = 0;
            if(isdigit(tmp[0]))
                num = 16*(tmp[0]-'0');
            else
                num = 16*(tmp[0]-'a'+10);
            if(isdigit(tmp[1]))
                num += tmp[1]-'0';
            else
                num += tmp[1]-'a'+10;
            int minNum = INT_MAX;

            int k = 0;
            for(int j = 0; j < 16; j++) {
                if(abs(16*j+j-num) < minNum) {
                    minNum = abs(16*j+j-num);
                    k = j;
                }
            }
            if(k < 10)
                res += to_string(k);
            else
                res.push_back('a'+k-10);
        }
        
            
        return res;
    }
};
