class Solution {
public:
    string findSimilar(string c) {
        int count = 0;
        
        if (c[0] >= 'a') count += (c[0] - 'a' + 10)*16;
        else count += (c[0] - '0')*16;
        
        if (c[1] >= 'a') count += (c[1] - 'a' + 10);
        else count += (c[1] - '0');
        

        int bestI = 0;
        int minGap = INT_MAX;
        for (int i = 0; i < 16; i++) {
            if (abs(i * 17 - count) < minGap) {
                minGap = abs(i * 17 - count);
                bestI = i;
            }
        }
        
        string ans;
        if (bestI < 10) ans += '0' + bestI;
        else ans += bestI - 10 + 'a';
        
        return ans;
    }
    
    
    string similarRGB(string color) {
        string ans = "#";
        ans += findSimilar(color.substr(1,2));
        ans += findSimilar(color.substr(3,2));
        ans += findSimilar(color.substr(5,2));
        
        return ans;
    }
};
