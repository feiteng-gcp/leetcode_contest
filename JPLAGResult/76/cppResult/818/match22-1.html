class Solution {
public:
    unordered_map<char,int> hmap;
int discc(char a1,char a2,char c){
    int p1=hmap[c]-hmap[a1];
    int p2=p1*16+hmap[c]-hmap[a2];
    return p2*p2;
}
    
    
    string similarRGB(string color) {
    vector<char> vc;
    for(int i=0;i<=9;++i){
        hmap['0'+i]=i;
        vc.push_back('0'+i);
    }
    for(int i=10;i<16;++i){
        hmap['a'+i-10]=i;
        vc.push_back('a'+i-10);
    }
    string ans="#";
    for(int i=0;i<3;++i){

        char myc='0';
        int d=INT_MAX;
        //cout<<color[i*2+1]<<" "<<color[i*2+2]<<endl;
        for(int j=0;j<vc.size();++j){
            int dis=discc(color[i*2+1],color[i*2+2],vc[j]);
            if(dis<d){
                d=dis;
                myc=vc[j];
            }
        }
        ans.push_back(myc);
        ans.push_back(myc);
    }
    return ans;
}
};
