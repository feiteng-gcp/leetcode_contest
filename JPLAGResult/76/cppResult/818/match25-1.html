class Solution {
public:
    string similarRGB(string color) {
        string res;
        string s = color;
        int r = toInt(s.substr(1,2));
        int g = toInt(s.substr(3,2));
        int b = toInt(s.substr(5,2));
        
        res = "#";
        res += findCloset(r);
        res += findCloset(g);
        res += findCloset(b);
        
        return res;
    }
    
private:
    int toInt(string s) {
        int res = 0;
        for (auto c : s) {
            c = toupper(c);
            if (c >= '0' && c <= '9') {
                res = res * 16 + (c - '0');  
            } else if (c >= 'A' && c <= 'F') {
                res = res * 16 + (c - 'A' + 10);
            }
        } 
        return res;
    }
    
    string findCloset(int n) {

        int d = INT_MAX;
        int t = 0;
        // cout << n << endl;
        for (int i = 0; i < 16; ++i) {
            int m = i * 16 + i;
            
            if (abs(m - n) < d) {
                d = abs(m - n);
                t = i;
            }
        }
        // cout << " f " << t << " " << endl;
        string res;
        if (t < 10) {
            res = '0' + t;
        } else {
            res = 'a' + t - 10;
        }
        
        return res;
    }
};
