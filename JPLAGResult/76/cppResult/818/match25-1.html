class Solution {
public:
    
int convertHex(string str) {
    int ret = 0;
    char ch1 = str[0];
    char ch2 = str[1];
    
    int digit1 = 0;
    int digit2 = 0;
    if (ch1 >= '0' && ch1 <= '9') {
        digit1 = ch1 - '0';
    } else {
        digit1 = 10 + ch1 - 'a';
    }
    
    if (ch2 >= '0' && ch2 <= '9') {
        digit2 = ch2 - '0';
    } else {
        digit2 = 10 + ch2 - 'a';
    }
    
    return digit1 * 16 + digit2;
}
    
string getMin(string str) {  // str has length of 2
    int target = convertHex(str);
    
    string ret;
    int low = INT_MAX;
    vector<string> vec = {"00", "11", "22", "33", "44", "55", "66",
                          "77", "88", "99", "aa", "bb", "cc", "dd", "ee", "ff"};

    for (int i = 0; i < vec.size(); i++) {
        int cur = convertHex(vec[i]);
        int cost = (target - cur) * (target - cur);
        if (cost < low) {
            low = cost;
            ret = vec[i]; //.substr(0, 1);
        }
    }
    return ret;
}

    
string similarRGB(string color) {
    string ret = "#";
    ret += getMin(color.substr(1, 2));
    ret += getMin(color.substr(3, 2));
    ret += getMin(color.substr(5, 2));
    return ret;   
}
};
