class Solution {
public:
    string similarRGB(string color) {
        string token1 = color.substr(1, 2);
        string token2 = color.substr(3, 2);
        string token3 = color.substr(5, 2);
        int n1 = hex_dec(token1), n2 = hex_dec(token2), n3 = hex_dec(token3);
        string res = "#";
        int sim = INT_MAX, best_x = 0;
        for (int i = 0; i <= 15; i++) {
            int tmp = i + i * 16;
            if ((tmp - n1) * (tmp - n1) < sim) {
                sim = (tmp - n1) * (tmp - n1);
                best_x = i;
            }            
        }

        sim = INT_MAX;
        int best_y = 0;
        for (int i = 0; i <= 15; i++) {
            int tmp = i + i * 16;
            if ((tmp - n2) * (tmp - n2) < sim) {
                sim = (tmp - n2) * (tmp - n2);
                best_y = i;
            }            
        }
        sim = INT_MAX;
        int best_z = 0;
        for (int i = 0; i <= 15; i++) {
            int tmp = i + i * 16;
            if ((tmp - n3) * (tmp - n3) < sim) {
                sim = (tmp - n3) * (tmp - n3);
                best_z = i;
            }            
        }
        if (best_x >= 10) res += 'a' + best_x - 10;
        else res += to_string(best_x);
        if (best_y >= 10) res += 'a' + best_y - 10;
        else res += to_string(best_y);
        if (best_z >= 10) res += 'a' + best_z - 10;
        else res += to_string(best_z);
        return res;
    }
    int hex_dec(string token) {
        stringstream ss(token);
        int y;
        ss >> std::hex >> y;
        return y;
    }
    
};
