int value(char c)
{
    if (c >= 'a') {
        return (c - 'a' + 10);
    }
    
    return c - '0';
}


void read(const string& color, int* input)
{
    for (int i = 0; i < 3; ++i) {
        int k = (value(color[1 + i*2])) * 16 + value(color[i*2 +2]);
        input[i] = k;
    }
}

int sqr(int x)
{
    return x * x;
}


class Solution {
public:
    string similarRGB(string color) {
        
        int input[3];
        read(color, input);
        
        int best = -1000000;
        int output[3];
        
        for (int i = 0; i < 16; ++i) {
            int x = sqr((i * 16 + i) - input[0]);
            for (int j = 0; j < 16; ++j) {
                int y = sqr((j * 16 + j) - input[1]);

                for (int k = 0; k < 16; ++k) {
                    int z = x + y + sqr((k * 16 + k) - input[2]);
                    if (-z > best) {
                        best = -z;
                        output[0] = i;
                        output[1] = j;
                        output[2] = k;
                    }
                }
            }
        }
        
        std::string ret = "#";
        for (int i = 0; i < 3; ++i) {
            char c = (output[i] >= 10 ? output[i] - 10 + 'a' : output[i] + '0');
            ret += c;
        }
        
        return ret;
    }
};
