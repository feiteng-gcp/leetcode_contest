class Solution {
public:
    int t(char c){
        if(c >= '0' && c <= '9') return c - '0';
        else return 10 + c - 'a';
    }
    char T(int x){
        if(x >= 0 && x <= 9) return x + '0';
        else return x - 10 + 'a';
    }
    int sqr(int x){
        return x * x;
    }
    string similarRGB(string color) {
        int dif = 1e8;
        int x = t(color[1]) * 16 + t(color[2]);
        int y = t(color[3]) * 16 + t(color[4]);
        int z = t(color[5]) * 16 + t(color[6]);
        string ans = "";
        for(int i = 0;i < 16;i++){
            for(int j = 0;j < 16;j++){
                for(int k = 0;k < 16;k++){
                    if(sqr(i * 16 + i - x) + sqr(j * 16 + j - y) + sqr(k * 16 + k - z) < dif){
                        cout << i << j << k << endl;
                        ans = "#";
                        ans += T(i);
                        ans += T(j);
                        ans += T(k);
                        cout << ans << endl;
                        dif = sqr(i * 16 + i - x) + sqr(j * 16 + j - y) + sqr(k * 16 + k - z);
                    }
                }
            }
        }
        return ans;
    }
};
