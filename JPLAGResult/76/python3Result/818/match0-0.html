class Solution:
	def similarRGB(self, color):
		"""
		:type color: str
		:rtype: str
		"""
		# print(color[1:3])
		# print(color[3:5])
		# print(color[5:])

		A = int(color[1:3], 16)
		B = int(color[3:5], 16)
		C = int(color[5:], 16)

		# print(A, B, C)

		ans = "#"
		for i in range(16):
			if abs(16*(i+1)+(i+1) - A) > abs(16*i+i - A):
				break
		ans += str(hex(i))[2:]

		for i in range(16):
			if abs(16*(i+1)+(i+1) - B) > abs(16*i+i - B):
				break
		ans += str(hex(i))[2:]

		for i in range(16):
			if abs(16*(i+1)+(i+1) - C) > abs(16*i+i - C):
				break
		ans += str(hex(i))[2:]
		
		return ans
