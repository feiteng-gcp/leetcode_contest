class Solution:
    def similarRGB(self, color):
        """
        :type color: str
        :rtype: str
        """
        color1 = int(color[1:3],16)
        color2 = int(color[3:5],16)
        color3 = int(color[5:7],16)
        print(color1, color2, color3)
        minDiff = -256**2 * 3
        currentColors = ''
        for x in range(0, 16):
            for y in range(0, 16):
                for z in range(0, 16):
                    diff = -(color1 - (x*16+x))**2 - (color2 - (y*16+y))**2 - (color3 - (z*16+z))**2 
                    if  diff >  minDiff:
                        minDiff = diff
                        currentColors = '#'+str(hex(x))[2]+str(hex(y))[2]+str(hex(z))[2]
        return currentColors
