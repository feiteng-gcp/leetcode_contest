class Solution:
    def similarRGB(self, color):
        """
        :type color: str
        :rtype: str
        
        poss color codes - 
            hex 10
            00  0
            11  17
            22  34
            33  51
            ...
            FF  255
        """
        red = int(color[1:3], 16)
        green = int(color[3:5], 16)
        blue = int(color[5:7], 16)
        
        closest_red = int(round(red / 17)) *17
        closest_green = int(round(green / 17)) * 17
        closest_blue = int(round(blue / 17)) * 17
        
        return "#" + hex(closest_red)[2] + hex(closest_green)[2] + hex(closest_blue)[2]
