class Solution:
    def similarRGB(self, color):
        """
        :type color: str
        :rtype: str
        """
        component1 = int(color[1:3], 16)
        component2 = int(color[3:5], 16)
        component3 = int(color[5:7], 16)
        
        best_shorts 
= None
        best_similarity = 2**32
        
        for short1 in range(16):
            for short2 in range(16):
                for short3 in range(16):
                    similarity = (short1 * 16 + short1 - component1) ** 2 + \
                        (short2 * 16 + short2 - component2) ** 2 + \
                        (short3 * 16 + short3 - component3) ** 2
                    if similarity < best_similarity:
                        best_similarity = similarity
                        best_shorts = (short1, short2, short3)
                        
        return '#' + ''.join([format(s, '01x') + format(s, '01x') for s in best_shorts])
