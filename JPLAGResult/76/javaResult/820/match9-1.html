class Solution {
        public List<Integer> eventualSafeNodes(int[][] graph) {
        int n = graph.length;
        int[] vis = new int[n];
        for (int i = 0; i < n; i++) if (vis[i] == 0) {
            dfs(i, vis, graph);
        }
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (vis[i] == 2) ans.add(i)
;
        }
        return ans;
    }
    void dfs(int u, int[] vis, int[][] g) {
        if (vis[u] == 1 || vis[u] == 2) {
            return;
        }
        vis[u] = 1;
        boolean cc = false;
        for (int v : g[u]) {
            dfs(v, vis, g);
            if (vis[v] == 1) cc = true;
        }
        if (!cc) vis[u] = 2;
    }
}
