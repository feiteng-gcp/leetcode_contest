import java.util.Arrays;
import java.util.LinkedList;
import java.util.List;

class Solution {
	int n;
	int[] last;
	boolean[] safe;
	boolean[] used;
	
	int edge;
	int[] prev;
	int[] to;
	
	void DFS(int x) {
		used[x] = true;
		safe[x] = false;
        boolean nsafe = true;
		for (int i = last[x]; i != -1; i = prev[i]) {
            //System.out.println(x + " -> " + to[i]);
			int v = to[i];
			if (!used[v]) {
				DFS(v);
			}
			if (!safe[v]) nsafe = false;
		}
        if (nsafe) safe[x] = true;
	}
	
    public List<Integer> eventualSafeNodes(int[][] graph) {
        n = graph.length;
        edge = 0;
        for (int i = 0; i < n; ++i) {
        	edge += graph[i].length;
        }
        
        last = new int[n];
        prev = new int[edge];
        to = new int[edge];
        Arrays.fill(last, -1);
        edge = 0;
        for (int i = 0; i < n; ++i) {
        	for (int j = 0; j < graph[i].length; ++j) {
        		int u = i;
        		int v = graph[i][j];
        		prev[edge] = last[u];
        		to[edge] = v;
        		last[u] = edge;
        		++edge;
        	}
        }
        
        safe = new boolean[n]; 
        used = new boolean[n];
        
        for (int i = 0; i < n; ++i) {
        	if (!used[i]) {
        		DFS(i);
        	}
        }
        
        
List<Integer> res = new LinkedList<Integer>();
        for (int i = 0; i < n; ++i) {
        	if (safe[i]) {
        		res.add(i);
        	}
        }
        return res;
    }
}
