class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        int[] outEdges = new int[graph.length];
        List<Set<Integer>> reverseGraph = new ArrayList<>();
        Queue<Integer> safe = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        
for (int i = 0; i < graph.length; i++) reverseGraph.add(new HashSet<>());
        for (int i = 0 ; i < graph.length; i++) {
            outEdges[i] = graph[i].length;
            if (graph[i].length == 0) safe.offer(i);
            for (int neigh : graph[i]) reverseGraph.get(neigh).add(i);
        }
        while (!safe.isEmpty()) {
            int i = safe.poll();
            if (visited.contains(i)) continue;
            
visited.add(i);
            for (int neigh : reverseGraph.get(i)) {
                outEdges[neigh]--;
                if (outEdges[neigh] == 0) safe.offer(neigh);
            }
        }
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < graph.length; i++) if (outEdges[i] == 0) ans.add(i);
        return ans;
    }
}
