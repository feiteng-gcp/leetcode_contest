class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        Set<Integer> active = new HashSet();
        List<Integer> res = new ArrayList();
        Map<Integer, Boolean> vis = new HashMap();
        
        for(int i = 0; i< graph.length; i++){
            DFS(i, graph, res, active, vis);
        }
        
        
Collections.sort(res);
        
        return res;
    }
    
    public boolean DFS(int i, int[][] graph, List<Integer> res, Set<Integer> active, Map<Integer, Boolean> vis){
        if(vis.containsKey(i)){
            return vis.get(i);
        }
        
        active.add(i);
        for(int j = 0; j< graph[i].length; j++){
            if(active.contains(graph[i][j]) || !DFS(graph[i][j], graph, res, active, vis)){
                active.remove(i);
                vis.put(i, false);
                return false;   
            }
        }
        
        active.remove(i);
        res.add(i);
        vis.put(i, true);
        
        return true;
    }
    
    
    
}
