class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        int len = graph.length;
        List<Integer> res = new ArrayList<>();
        Map<Integer,List<Integer>> toChild = new HashMap<>();
        int[] out = new int[len];
        for(int i = 0 ; i < graph.length ; i++){
            for(int parent : graph[i]){
                if(!toChild.containsKey(parent))
                    toChild.put(parent, new ArrayList<>());
                toChild.get(parent).add(i);
                out[i]++;
            }
        }
        
        Queue<Integer> que = new LinkedList<>();
        
for(int i = 0 ; i < len ; i++)
            if(out[i] == 0)
                que.add(i);
        
        while(!que.isEmpty()){
            int root = que.poll();
            res.add(root);
            List<Integer> childs = toChild.get(root);
            if(childs!=null)
                for(int child : childs)
                    if(--out[child] == 0)
                        que.add(child);
        }
        Collections.sort(res);
        return res;
    }
}
