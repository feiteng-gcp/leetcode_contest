class Solution {
    public List<Integer> eventualSafeNodes(int[][] a) {
        int n = a.length;
        Boolean[] memo = new Boolean[n];
        boolean[] onPath = new boolean[n];
        Set<Integer> res = new HashSet<>();
        for (int i = 0; i < n; i++) {
            if (memo[i] == null) {
                if (!dfs(res, a, onPath, memo, i)) res.add(i);
            }
        }
        List<Integer> resList = new ArrayList<>(res);
        Collections.sort(resList);
        return resList;
    }

    private boolean dfs(Set<Integer> res, int[][] a, boolean[] onpath, Boolean[] memo, int u) {
        if (onpath[u]) return true;
        if (memo[u] != null) return memo[u];
        onpath[u] = true;
        boolean cyclic = false;
        for (int v : a[u]) {
            if (dfs(res, a, onpath, memo, v)) cyclic = true;
            else res.add(v);
        }
        onpath[u] = false;
        memo[u] = cyclic;
        return cyclic;
    }

}
