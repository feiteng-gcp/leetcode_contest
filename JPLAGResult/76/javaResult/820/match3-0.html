class Solution {
    int color[];
    int n;
    Set<Integer> rs;
    public List<Integer> eventualSafeNodes(int[][] graph) {
        n = graph.length;
        color = new int[n + 1];
        rs = new HashSet();
        for(int i = 0; i < n; i++){
            if(color[i] == 0){
                dfs(i, graph);
            }
        }
        List<Integer> ret = new ArrayList<>();
        for(int i = 0; i < n; i++) if(!rs.contains(i)){
            ret.add(i);
        }
        return ret;
    }
    public boolean dfs(int u, int[][] g){
        color[u] = 1;
        for(int v :g[u]){
            if(color[v] == 1){
                rs.add(v);
                rs.add(u);
                return true;
            }
            if(color[v] == 0 && dfs(v, g)){
                rs.add(u);
                return true;
            }
        }
        color[u] = 2;
        return false;
    }
}
