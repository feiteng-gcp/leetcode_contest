class Solution {
    
    class Node {
        HashSet<Node> prev;
        HashSet<Node> next;
        int ind;
        public Node(int i) {
            ind = i;
            prev = new HashSet();
            next = new HashSet();
        }
        
        public void add(Node n) {
            n.prev.add(this);
            next.add(n);
        }
        
        public void remove(Node n) {
            n.prev.remove(this);
            next.remove(n);
        }
    }
    
    public List<Integer> eventualSafeNodes(int[][] graph) {
        int n = graph.length;
        Node[] nodes = buildGraph(graph);
        List<Integer> targets = getTargets(graph);
        List<Integer> ret = new LinkedList();
        LinkedList<Node> bag = new LinkedList();
        for(Integer i : targets) {
            bag.add(nodes[i]);
        }
        // HashSet<Node> dedupe = new HashSet<Node>();
        while(bag.size()!=0) {
            Node target = bag.removeFirst();
            ret.add(target.ind);
            // dedupe.add(target);
            
            HashSet<Node> prev = new HashSet();
            for(Node node : target.prev) {
                prev.add(node);
            }
            for(Node node : prev) {
                node.remove(target);
                if(node.next.size()==0) {
                    bag.add(node);
                }
            }
        }
        return sortedList(ret);
    }
    
    private List<Integer> sortedList(List<Integer> list) {
        int n = list.size();
        LinkedList<Integer> ret = new LinkedList();
        int[] a = new int[n];
        int c = 0;
        for(Integer i : list) {
            a[c++] = i;
        }
        Arrays.sort(a);
        for(int i=0;i<n;i++) {
            ret.add(a[i]);
        }
        return ret;
    }
    
    private List<Integer> getTargets(int[][] graph) {
        LinkedList<Integer> ret = new LinkedList<Integer>();
        for(int i=0;i<graph.length;i++) {
            if(graph[i].length==0) {
                ret.add(i);
            }
        }
        return ret;
    }
    
    private Node[] buildGraph(int[][] graph) {
        int n = graph.length;
        Node[] nodes = new Node[n];
        for(int i=0;i<n;i++) {
            nodes[i] = new Node(i);
        }
        for(int i=0;i<n;i++) {
            for(int j : graph[i]) {
                nodes[i].add(nodes[j]);
            }
        }
        return nodes;
    }
    
    private int[][] getBackGraph(int[][] graph) {
        int n = graph.length;
        int[] count = new int[n];
        Arrays.fill(count, 0);
        for(int[] node : graph) {
            for(int next : node) {
                count[next] += 1;
            }
        }
        int[][] backgraph = new int[n][];
        for(int i=0;i<n;i++) {
            backgraph[i] = new int[count[i]];
        }
        for(int i=0;i<n;i++) {
            for(int next : graph[i]) {
                count[next] -= 1;
                backgraph[next][count[next]] = i;
            }
        }
        return backgraph;
    }
}
