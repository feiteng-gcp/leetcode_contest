class Solution {
    
    public List<Integer> eventualSafeNodes(int[][] graph) {
        boolean[] safe = new boolean[graph.length];
        for (int i = 0; i < graph.length; i++) {
            int[] edges = graph[i];
            if (edges.length == 0) {
                safe[i] = true;
            }
        }
        while (true) {
            boolean[] newSafe = Arrays.copyOf(safe, safe.length);
            for (int i = 0; i < graph.length; i++) {
                if (safe[i]) {
                    continue;
                }
                boolean allSafe = true;
                for (int to: graph[i]) {
                    if (!newSafe[to]) {
                        allSafe = false;
                        break;
                    }
                }
                if (allSafe) {
                    newSafe[i] = true;
                }
            }
            if (Arrays.equals(safe, newSafe)) {
                safe = newSafe;
                break;
            }
            safe = newSafe;
        }
        List<Integer> list = new LinkedList<>();
        for (int i = 0; i < safe.length; i++) {
            if (safe[i]) {
                list.add(i);
            }
        }
        return list;
    }
}
