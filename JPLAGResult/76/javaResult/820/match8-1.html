class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        boolean[] unsafe = new boolean[graph.length];
        boolean[] visited = new boolean[graph.length];
        Set<Integer> path = new HashSet<>();
        for (int i = 0; i < graph.length; i++) {
            if (!visited[i]) {
                 path.add(i);
                 dfs(i, graph, unsafe, visited, path);
                 
path.remove(i);
            }
        }
       
        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < unsafe.length; i++) {
            if (!unsafe[i]) {
                res.add(i);
            }
        }
        return res;
    }
    
    public void dfs(int p, int[][] graph, boolean[] unsafe, boolean[] visited, Set<Integer> path) {
        visited[p] = true;
        for (int i : graph[p]) {
            if (path.contains(i)) {
                for (int j : path) {
                    unsafe[j] = true;
                }
            }
            if (!visited[i]) {
                path.add(i);
                dfs(i, graph, unsafe, visited, path);
                path.remove(i);
            }
            if (unsafe[i]) {
                unsafe[p] = true;
            }
        }
    }
}
