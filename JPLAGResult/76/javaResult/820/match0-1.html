class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        int N = graph.length;
        int[] safe = new int[N];
        
        for (int i = 0; i < N; i++) {
            dfs(graph, i, safe);
        }
        
        
List<Integer> res = new ArrayList<>();
        
        for (int i = 0; i < N; i++) {
            if (safe[i] > 0) {
                res.add(i);
            }
        }
        
        return res;
    }
    
    private int dfs(int[][] graph, int i, int[] safe) {
        if (safe[i] != 0) {
            return safe[i];
        }
        
        safe[i] = -1;
        
        int res = 1;
        
        for (int j : graph[i]) {
            res = Math.min(res, dfs(graph, j, safe));
            
            if (res == -1) {
                break;
            }
        }
        
        return safe[i] = res;
    }
}
