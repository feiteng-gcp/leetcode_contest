class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
            int n = graph.length;
            int[] stats = new int[n];
            for (int i = 0; i < stats.length; i++) {
                if (stats[i] == 0) {
                    dfs(i, graph,stats);
                }
            }
            
List<Integer> good = new ArrayList<>();
            for (int i = 0; i < n; i++) {
                if (stats[i] == 3) {
                    good.add(i);
                }
            }return good;
        }

        public int dfs(int x, int[][] graph, int[] stats) {
            stats[x] = 1; // begin visit;
            int mystat = 0;
            for (int i = 0; i < graph[x].length; i++) {
                int u = graph[x][i];
                int st;
                if (stats[u] == 0) {
                    st = dfs(u, graph, stats);
                } else if (stats[u] == 1) {
                    st = 2; // cycle;
                } else if (stats[u] == 2) {
                    st = 2;
                } else {
                    st = 3;
                }
                if (st == 2 || st == 1) {
                    mystat = 2;
                    break;
                }
            }
            if (mystat == 0) {
                return stats[x] = 3;
            } else {
                return stats[x] = 2;
            }
        }
}
