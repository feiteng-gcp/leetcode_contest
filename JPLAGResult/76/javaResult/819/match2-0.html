class Solution {
    public int minSwap(int[] A, int[] B) {
        int[][] f = new int[A.length][2];
        f[0][0] = 0;
        f[0][1] = 1;
        
        for (int i = 1; i < A.length; i++) {
            f[i][0] = -1;
            if (f[i - 1][0] != -1 && A[i - 1] < A[i] && B[i - 1] < B[i]) {
                f[i][0] = f[i - 1][0];
            }
            if (f[i - 1][1] != -1 && B[i - 1] < A[i] && A[i - 1] < B[i]) {
                f[i][0] = f[i][0] == -1 ? f[i - 1][1] : Math.min(f[i][0], f[i - 1][1]);
            }
            
            f[i][1] = -1;
            if (f[i - 1][0] != -1 && A[i - 1] < B[i] && B[i - 1] < A[i]) {
                f[i][1] = f[i - 1][0] + 1;
            }
            if (f[i - 1][1] != -1 && B[i - 1] < B[i] && A[i - 1] < A[i]) {
                f[i][1] = f[i][1] == -1 ? f[i - 1][1] + 1 : Math.min(f[i][1], f[i - 1][1] + 1);
            }
        }
        
        return Math.min(f[A.length - 1][0], f[A.length - 1][1]);
    }
}
