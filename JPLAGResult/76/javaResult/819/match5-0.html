class Solution {
    public int minSwap(int[] A, int[] B) {
        int[][] minSwaps = new int[A.length][2];
        minSwaps[0][0] = 0;
        minSwaps[0][1] = 1;
        for (int i = 1; i < A.length; i++) {
            minSwaps[i][0] = Integer.MAX_VALUE/2;
            if (A[i] > A[i-1] && B[i] > B[i-1]) minSwaps[i][0] = Math.min(minSwaps[i][0], minSwaps[i-1][0]);
            if (A[i] > B[i-1] && B[i] > A[i-1]) minSwaps[i][0] = Math.min(minSwaps[i][0], minSwaps[i-1][1]);
            minSwaps[i][1] = Integer.MAX_VALUE/2;
            if (A[i] > A[i-1] && B[i] > B[i-1]) minSwaps[i][1] = Math.min(minSwaps[i][1], 1 + minSwaps[i-1][1]);
            if (A[i] > B[i-1] && B[i] > A[i-1]) minSwaps[i][1] = Math.min(minSwaps[i][1], 1 + minSwaps[i-1][0]);
            // System.out.println(minSwaps[i][0] + " " + minSwaps[i][1]);
        }
        return Math.min(minSwaps[A.length-1][0], minSwaps[A.length-1][1]);
    }
}
