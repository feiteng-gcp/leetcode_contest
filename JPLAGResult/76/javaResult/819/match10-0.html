class Solution {
    public int minSwap(int[] A, int[] B) {
        int n = A.length;
        int[][] dp = new int[n][2];
        for (int i = 0; i < n; ++i) {
        	for (int j = 0; j < 2; ++j) {
        		dp[i][j] = Integer.MAX_VALUE;
        	}
        }
        dp[0][0] = A[0] <= B[0] ? 0 : 1;
        dp[0][1] = A[0] >= B[0] ? 0 : 1;
        for (int i = 0; i + 1 < n; ++i) {
        	int minL = Math.min(A[i], B[i]);
        	int maxL = Math.max(A[i], B[i]);
        	int minR = Math.min(A[i + 1], B[i + 1]);
        	int maxR = Math.max(A[i + 1], B[i + 1]);
        	//0 -> 0
        	if (minL < minR && maxL < maxR) {
        		dp[i + 1][0] = Math.min(dp[i + 1][0], dp[i][0] + (A[i + 1] <= B[i + 1] ? 0 : 1));
        	}
        	//0 -> 1
        	if (minL < maxR && maxL < minR) {
        		dp[i + 1][1] = Math.min(dp[i + 1][1], dp[i][0] + (A[i + 1] >= B[i + 1] ? 0 : 1));
        	}
        	//1 -> 0
        	if (maxL < minR && minL < maxR) {
        		dp[i + 1][0] = Math.min(dp[i + 1][0], dp[i][1] + (A[i + 1] <= B[i + 1] ? 0 : 1));
        	}
        	//1 -> 1
        	if (maxL < maxR && minL < minR) {
        		dp[i + 1][1] = Math.min(dp[i + 1][1], dp[i][1] + (A[i + 1] >= B[i + 1] ? 0 : 1));
        	}
        }
        int res = Math.min(dp[n - 1][0], dp[n - 1][1]);
        return res;
    }
}
