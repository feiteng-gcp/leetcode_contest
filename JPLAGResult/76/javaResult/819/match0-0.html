class Solution {
    public int minSwap(int[] A, int[] B) {
        int index = A.length - 1;
        int[][] look = new int[A.length][2];
        for(int i = 0; i < A.length; i++) {
            look[i][0] = -1;
            look[i][1] = -1;
        }
        int res = Math.min(dpHelper(A, B, index, 0, look), dpHelper(A, B, index, 1, look));
        return res;
    }
    private int dpHelper(int[] A, int[] B, int index, int swap, int[][] look) {
        if(index == 0) return (swap == 1) ? 1 : 0;
        if(look[index][swap] != -1) return look[index][swap];
        int res = 3000;
        if(swap == 1) {
            int temp = B[index];
            B[index] = A[index];
            A[index] = temp;
        }
        if(A[index] > A[index - 1] && B[index] > B[index - 1]) res = dpHelper(A, B, index - 1, 0, look);
        if(A[index] > B[index - 1] && B[index] > A[index - 1]) res = Math.min(res, dpHelper(A, B, index - 1, 1, look));
        if(swap == 1) {
            int temp = B[index];
            B[index] = A[index];
            A[index] = temp;
            res++;
        }
        look[index][swap] = res;
        return res;
    }
}
