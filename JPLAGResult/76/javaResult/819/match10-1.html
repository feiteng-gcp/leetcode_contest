class Solution {
   public int minSwap(int[] A, int[] B) {
            int n = A.length;
            int[][] dp = new int[n][2];
            for (int i = 0; i < n; i++) {
                dp[i][0] = dp[i][1] = Integer.MAX_VALUE / 2;
            }
            for (int i = n - 1; i >= 0; i--) {
                if (i == n - 1) {
                    dp[i][0] = 0;
                    dp[i][1] = 1;
                } else {
                    if (A[i] < A[i + 1] && B[i] < B[i + 1]){
                        dp[i][0] = dp[i + 1][0];
                        dp[i][1] = 1 + dp[i + 1][1];
                    }
                    if (A[i] < B[i + 1] && B[i] < A[i + 1]) {
                        dp[i][1] = Math.min(dp[i][1], dp[i + 1][0] + 1);
                        dp[i][0] = Math.min(dp[i][0], dp[i + 1][1]);
                    }
                }
            }
            return Math.min(dp[0][0], dp[0][1]);
        }
}
