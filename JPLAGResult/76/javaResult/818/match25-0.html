class Solution {
    public String similarRGB(String color) {
        String a = color.substring(1,3);
        String b = color.substring(3,5);
        String c = color.substring(5,7);
        System.out.println(a);
        System.out.println(b);
        System.out.println(c);
        String res = "#" + sim(a) + sim(b) + sim(c);
        return res;
    }
    
    private String sim(String c){
        int out = Integer.parseInt(c, 16);
        int min = Integer.MAX_VALUE;
        int cur = 0;
        for(int i = 0; i < 16; i++){
            int temp = i * 16 + i - out;
            temp = Math.abs(temp);
            if(temp < min){
                min = temp;
                cur = i * 16 + i;
            }
        }
        String res =  Integer.toHexString(cur);
        if(res.equals("0")) res += "0";
        return res;
    }
}
