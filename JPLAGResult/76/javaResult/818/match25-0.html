class Solution {
    public String similarRGB(String color) {
        return String.format ("#%s%s%s", getClosest (color.substring (1, 3)), getClosest (color.substring (3, 5)), getClosest (color.substring (5, 7)));
    }

    String getClosest (String s) {
        int val = Integer.parseInt (s, 16);
        int min_square = Integer.MAX_VALUE, min_num = -1;
        for (int i = 0; i < 16; i++) {
            int cur_square = Math.abs (i * 16 + i - val);
            if (cur_square < min_square) {
                min_square = cur_square;
                min_num = i;
            }
        }
        // System.out.printf ("%s returns %s\n", s, min_num + "");///
        return Integer.toString (min_num, 16);
    }
}
