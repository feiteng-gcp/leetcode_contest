class Solution {
    
public String similarRGB(String color) {
        String d = "0123456789abcdef";
        String res = "#";
        
        for(int i = 1;  i< 7; i+= 2){
            int min = Integer.MAX_VALUE;
            int j = -1;
            int c = convert(color, i);
            for(int k = 0; k< 16; k++){
                if((c- k*16-k)*(c- k*16-k)< min){
                    min = (c- k*16-k)*(c- k*16-k);
                    j = k;
                } 
            }    
            
            res += d.charAt(j);
            res += d.charAt(j);
        
}
        
        return res;
    }
    
    public int convert(String color, int i){
        String d = "0123456789abcdef";
        
        int res = d.indexOf(color.substring(i, i+1))*16 + d.indexOf(color.substring(i+1, i+2));
        
        return res;
        
    }
}
