class Solution {
    public String similarRGB(String color) {
        HashMap<String, Integer> map = new HashMap<String, Integer>();
        map.put("0",0);
        map.put("1", 17);
        map.put("2", 34);
        map.put("3", 51);
        map.put("4", 68);
        map.put("5", 85);
        map.put("6", 102);
        map.put("7", 119);
        map.put("8", 136);
        map.put("9", 153);
        map.put("a", 170);
        map.put("b", 187);
        map.put("c", 204);
        map.put("d", 221);
        map.put("e", 238);
        map.put("f", 255);
        
        StringBuilder sb = new StringBuilder();
        sb.append("#");
        int []part = new int[3];
        part[0] = Integer.valueOf(color.substring(1,3), 16);
        part[1] = Integer.valueOf(color.substring(3,5), 16);
        part[2] = Integer.valueOf(color.substring(5,7), 16);
        
        for(int i=0; i<3; i++)
        {
            String curr = "";
            int minDiff = Integer.MAX_VALUE;
            for(String key: map.keySet())
            {
                int val = map.get(key);
                int diff = part[i] - val;
                if(((int)Math.pow(diff, 2)) < minDiff)
                {
                    minDiff = (int)Math.pow(diff, 2);
                    curr = key;
                }
            }
            sb.append(curr);
        }
        return sb.toString();
    }
}
