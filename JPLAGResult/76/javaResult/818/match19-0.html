class Solution {
    public String similarRGB(String color) {
        Integer ai = Integer.parseInt(color.substring(1,3),16);
        Integer bi = Integer.parseInt(color.substring(3,5),16);
        Integer ci = Integer.parseInt(color.substring(5,7),16);
        
        
        return "#" + getSimilarity(ai)+getSimilarity(bi)+getSimilarity(ci);
    }
    
    private String getSimilarity(Integer in) {
        
        int result = Integer.MAX_VALUE;
        Integer ir = 0;
        
        for (int i=0; i<16; i++) {
            
            if (Math.abs(in-i*16-i) < result) {
                ir = i;
                result = Math.abs(in-i*16-i);
            }
        }
        
        return Integer.toHexString(ir);
    }
}
