class Solution {
    public String similarRGB(String color) {
        // similar value min (most minus)
        // R value min, G min, B min
        
        // AA -> BB or 99
        
        return "#" + 
similar(color.substring(1, 3)) + similar(color.substring(3, 5)) + similar(color.substring(5, 7));
        
    }
    
    String similar(String digits) {
        int val = calTwo(digits);
        List<String> candidates = new ArrayList<>();
        String m = digits.substring(0, 1);
        candidates.add(m);
        if (m.charAt(0) != '0') candidates.add(smaller(m));
        if (m.charAt(0) != 'f') candidates.add(larger(m));
        int minDiff = Integer.MAX_VALUE;
        String res = "";
        for (String can : candidates) {
            int diff = Math.abs(
calOne(can) - val);
            if (diff < minDiff) {
                minDiff = diff;
                res = can;
            }
        }
        return res;
    }
    
    int calTwo(String digits) {
        char a = digits.charAt(0);
        int va = (a >= '0' && a <= '9') ? a - '0' : 10 + a - 'a';
        char b = digits.charAt(1);
        int vb = (b >= '0' && b <= '9') ? b - '0' : 10 + b - 'a';
        return va * 16 + vb;
    }
    
    int calOne(String digit) {
        char a = digit.charAt(0);
        int va = (a >= '0' && a <= '9') ? a - '0' : 10 + a - 'a';
        return va * 16 + va;
    }
    
    String smaller(String m) {
        char a = m.charAt(0);
        if (a == 'a') a = '9';
        else a--;
        return "" + a;
    }
    
    String larger(String m) {
        char a = m.charAt(0);
        if (a == '9') a = 'a';
        else a++;
        return "" + a;
    }
}
