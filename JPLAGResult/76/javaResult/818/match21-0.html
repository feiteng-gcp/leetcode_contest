class Solution {
    public String similarRGB(String color) {
        StringBuilder sb = new StringBuilder();
        sb.append('#');
        sb.append(nearest(color.substring(1,3)));
        sb.append(nearest(color.substring(3,5)));
        sb.append(nearest(color.substring(5,7)));

        return sb.toString();
    }

    public String nearest(String color) {
        int num = Integer.parseInt(color,16);
        if (num % 17 <= 8) {
            String res = Integer.toHexString(num / 17);
            return res;
        }
        else {
            String res = Integer.toHexString(num / 17 + 1);
            return res;
        }
    }
}
