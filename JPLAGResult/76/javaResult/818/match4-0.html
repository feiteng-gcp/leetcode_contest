class Solution {
    public String similarRGB(String color) {
        if (color == null || color.length() == 0)
            return color;
        String[] whole = {"00", "11", "22", "33", "44", "55", "66", "77", "88",
                    "99", "aa", "bb", "cc", "dd", "ee", "ff"};
        String first = color.substring(1,3);
        String second = color.substring(3,5);
        String third = color.substring(5, 7);
        StringBuilder res = new StringBuilder("#");

        res.append(whole[findClosest(whole, first)].charAt(0));
        res.append(whole[findClosest(whole, second)].charAt(0));
        res.append(whole[findClosest(whole, third)].charAt(0));
        return res.toString();
    }
    private int findClosest(String[] whole, String num) {
        int min = Integer.MAX_VALUE, index = -1;
        for (int i = 0; i < whole.length; i++) {
            String s = whole[i];
            int dis = cal(num, s);
            if (Math.abs(dis) < min) {
                min = Math.abs(dis);
                index = i;
            }
            if (dis < 0) {
                break;
            }
        }
        return index;
    }

    private int cal(String a, String b) {
        int value1 = Integer.parseInt(a, 16);
        int value2 = Integer.parseInt(b, 16);
        return value1 - value2;

    }
}
