class Solution {
    static final String comps[] = {"00", "11", "22", "33", "44", "55", "66", "77", "88", "99", "aa",
                                "bb", "cc", "dd", "ee", "ff"};
    public String similarRGB(String color) {
        if(color.length() <= 4){
            color = convert(color);
        }
        String rs = "";
        int dis = Integer.MIN_VALUE;
        for(int i = 0; i < comps.length; i++){
            for(int j = 0; j < comps.length; j++){
                
for(int v = 0; v < comps.length; v++){
                    String co = "#" + comps[i] + comps[j] + comps[v];
                    int tmp = dis(color, co);
                    if(dis < tmp){
                        rs = co;
                        dis = tmp;
                    }
                }
            }
        
}
        return rs;
    }
    public String convert(String color){
        StringBuilder sb=  new StringBuilder("#");
        for(int i = 1; i < color.length(); i++){
            sb.append(color.charAt(i)).append(color.charAt(i));
        }
        return sb.toString();
    }
    public int dis(String s1, String s2){
        int dis = 0;
        for(int i = 1; i <= 5; i += 2){
            int v =  getVal(s1.charAt(i), s1.charAt(i + 1)) - getVal(s2.charAt(i), s2.charAt(i + 1));
            dis -= v * v;
        
}
        return dis;
    }
    public int getVal(char y, char x){
        int val = 0;
        if(x >= '0' && x <= '9') val += x - '0';
        else val += x - 'a' + 10;
        if(y >= '0' && y <= '9') val += (y - '0') * 16;
        else val += (y - 'a' + 10) * 16;
        return val;
    }
}
