class Solution {
    public String similarRGB(String color) {
        int dist = Integer.MAX_VALUE;
        int v1 = Integer.parseInt(color.substring(1,3), 16);
        int v2 = Integer.parseInt(color.substring(3,5), 16);
        int v3 = Integer.parseInt(color.substring(5,7), 16);
        String out = "";
        //System.out.format("%d %d %d\n", v1, v2, v3);
        
        for (int i = 0; i < 16; i++) {
            for (int j = 0; j < 16; j++) {
                for (int k = 0; k < 16; k++) {
                    int ii = i * 16 + i, jj = j * 16 + j, kk = k * 16 + k;
                    int temp = (v1 - ii) * (v1 - ii);
                    temp +=  (v2 - jj) * (v2 - jj);
                    temp += (v3 - kk) * (v3 - kk);
                    if (temp < dist) {
                        out = "#"+ Integer.toHexString(i) + Integer.toHexString(j) + Integer.toHexString(k);
                        dist = temp;
                    }
                }            
            }            
        }
        
        return out;    
    }
}
