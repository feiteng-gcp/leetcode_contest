import java.util.*;
class Solution {
    private Map<Integer, String> dict;
    public String similarRGB(String color) {
        prepare();
        int inputFirst = 0, inputSecond = 0, inputThird = 0;
        String inputFirstStr = "0x" + color.substring(1, 3), inputSecondStr = "0x" + color.substring(3, 5), inputThirdStr = "0x" + color.substring(5, 7);
        inputFirst = Integer.decode(inputFirstStr);
        inputSecond = Integer.decode(inputSecondStr);
        inputThird = Integer.decode(inputThirdStr);
        StringBuilder sb = new StringBuilder();
        sb.append("#");
        sb.append(processOne(inputFirst));
        sb.append(processOne(inputSecond));
        sb.append(processOne(inputThird));
        return sb.toString();
    }
    
    private String processOne(
int input) {
        int difference = Integer.MAX_VALUE;
        String toReturn = "";
        for(Integer curKey : dict.keySet()) {
            int curDiff = (int)Math.pow((curKey - input), 2);
            if (curDiff < difference) {
                difference = curDiff;
                toReturn = dict.get(curKey);
            }
        }
        return toReturn;
    }
    
    private void prepare() {
        dict = 
new HashMap<Integer, String>();
        dict.put(Integer.decode("0x00"), "0");
        dict.put(Integer.decode("0x11"), "1");
        dict.put(Integer.decode("0x22"), "2");
        dict.put(Integer.decode("0x33"), "3");
        dict.put(Integer.decode("0x44"), "4");
        dict.put(Integer.decode("0x55"), "5");
        dict.put(Integer.decode("0x66"), "6");
        dict.put(Integer.decode("0x77"), "7");
        dict.put(Integer.decode("0x88"), "8");
        dict.put(Integer.decode("0x99"), "9");
        dict.put(Integer.decode("0xaa"), "a");
        dict.put(Integer.decode("0xbb"), "b");
        dict.put(Integer.decode("0xcc"), "c");
        dict.put(Integer.decode("0xdd"), "d");
        dict.put(Integer.decode("0xee"), "e");
        dict.put(Integer.decode("0xff"), "f");
    }
}
