class Solution {
    public String similarRGB(String color) {
        String res = "#";
        
        res += helper(color.substring(1, 3));
        res += helper(color.substring(3, 5));
        res += helper(color.substring(5, 7));
        
        return res;
    }
    
    String helper(String color) {
        
        int min = 1000;
        String res = "";
        for (char c = '0'; c <= '9'; c++) {
            if (Math.abs(parse(c) * 17 - parse(color)) < min) {
                //System.out.println(color + "," + c + "," + min);
                min = Math.abs(parse(c) * 17 - parse(color));
                res = Character.toString(c) + Character.toString(c);
            } 
        }
        
        for (char c = 'a'; c <= 'f'; c++) {
            if (Math.abs(parse(c) * 17 - parse(color)) < min) {
                min = Math.abs(parse(c) * 17 - parse(color));
                res = Character.toString(c) + Character.toString(c);
            } 
        }
        
        return res;
    }
    
    int parse(String color) {
        return parse(color.charAt(0)) * 16 + parse(color.charAt(1));
    }
    
    int parse(char c) {
        if (Character.isDigit(c)) {
            return c - '0';
        } else {
            return c - 'a' + 10;
        }
    }
}
