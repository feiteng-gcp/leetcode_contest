public class Solution {
	public String similarRGB(String color) {
		StringBuilder 
sb = new StringBuilder("#");
		sb.append(getClosest(color.substring(1, 3)));
		sb.append(getClosest(color.substring(3, 5)));
		sb.append(getClosest(color.substring(5)));
		return sb.toString();
	}

	private String getClosest(String s) {
		int val = Integer.valueOf(s, 16);
		int best = 0, dist = Math.abs(val);
		for (int i = 1; i < 16; ++i) {
			int cand = i * 16 + i;
			if (Math.abs(cand - val) < dist) {
				dist = Math.abs(cand - val);
				best = i;
			}
		}
		return Integer.toHexString(best);
	}
}
