class Solution {
    private String res = "";
    Map<Character, Integer> map = new HashMap<>();
    public String similarRGB(String color) {
        
        List<Integer> list = new ArrayList<>();

        
        for (int i = 0; i <= 15; i++) {
            int tmp = 0;
            tmp += i;
            tmp = tmp*16 + i;
            list.add(tmp);
        }
        // System.out.println(list);
        
        
        map.put('0', list.get(0));
        map.put('1', list.get(1));
        map.put('2', list.get(2));
        map.put('3', list.get(3));
        map.put('4', list.get(4));
        map.put('5', list.get(5));
        map.put('6', list.get(6));
        map.put('7', list.get(7));
        map.put('8', list.get(8));
        map.put('9', list.get(9));
        map.put('a', list.get(10));
        map.put('b', list.get(11));
        map.put('c', list.get(12));
        map.put('d', list.get(13));
        map.put('e', list.get(14));
        map.put('f', list.get(15));
        
        
        
        helper(color.substring(1,3));
        helper(color.substring(3,5));
        helper(color.substring(5));
        
        return "#" + res;
    }
    
    void helper(String a) {
        int value = Integer.parseInt(a, 16);  
        char key = '0';
        int min = Integer.MAX_VALUE;
        for (Map.Entry<Character, Integer> entry : map.entrySet()) {
            if (Math.abs(value-entry.getValue()) < min) {
                min = Math.abs(value-entry.getValue());
                key = entry.getKey();
            }
        }
        res += key;
    }
}
