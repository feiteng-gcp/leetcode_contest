class Solution {
    public String similarRGB(String color) {
        String res = "#";
        for (int i = 0; i < 3; i++) {
            char c1 = color.charAt(i * 2 + 1);
            char c2 = color.charAt(i * 2 + 2);
            if (c1 == c2) {
                res += c1;
            } else {
                res += get(c1, c2);
            }
        }
        return res;
    }

    char get(char c1, char c2) {
        int orig = toInt(c1) * 16 + toInt(c2);
        int min = Integer.MAX_VALUE;
        char res = 0;
        for (int i = 0; i < 10; i++) {
            int diff = Math.abs(i * 16 + i - orig);
            if (diff < min) {
                res = (char)(i + '0');
                min = diff;
            }
        }
        for (char c = 'a'; c <= 'f'; c++) {
            int i = toInt(c);
            int diff = Math.abs(i * 16 + i - orig);
            //System.out.println(c + " " + i + " " + diff + " " + (i * 16 + i ) + " " + orig);
            if (diff < min) {
                res = c;
                min = diff;
            }
        }
        return res;
    }

    int toInt(char c) {
        if (c >= '0' && c <= '9') {
            return c - '0';
        } else {
            return c - 'a' + 10;
        }
    }

    void test() {
        System.out.println(similarRGB("#09f166"));
    }

}
