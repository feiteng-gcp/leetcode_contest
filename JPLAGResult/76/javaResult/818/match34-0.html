class Solution {
    int charToInt(char ch) {
        if (Character.isUpperCase(ch)) {
            return ch - 'A' + 10;
        } else if (Character.isLowerCase(ch)) {
            return ch - 'a' + 10;
        } else {
            return ch - '0';
        }
    }
    
    char intToChar(int n) {
        if (n >= 10) {
            return (char) (n - 10 + 'a');
        } else {
            return (char) (n + '0');
        }
    }
    
    String intToString(int a) {
        int high = a / 16;
        int low = a % 16;
        StringBuilder sb = new StringBuilder();
        if (high >= 10) {
            sb.append((char) (high - 10 + 'a'));
        } else {
            sb.append(high);
        }
        if (low >= 10) {
            sb.append((char) (low - 10 + 'a'));
        } else {
            sb.append(low);
        }
        return sb.toString();
    }
    int strToInt(String s) {
        return charToInt(s.charAt(0)) * 16 + charToInt(s.charAt(1));
    }
    
    int similarity(String a, String b) {
        int sum = 0;
        int n1 = strToInt(a.substring(0, 2));
        int n2 = strToInt(b.substring(0, 2));
        sum += -(n1 - n2) * (n1 - n2);
        
        n1 = strToInt(a.substring(2, 4));
        n2 = strToInt(b.substring(2, 4));
        sum += -(n1 - n2) * (n1 - n2);
        
        n1 = strToInt(a.substring(4, 6));
        n2 = strToInt(b.substring(4, 6));
        sum += -(n1 - n2) * (n1 - n2);
        
        return sum;
        
    }
    
    public String similarRGB(String color) {
        String ans = "";
        String s = color.substring(1);
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < 16; i++) {
            StringBuilder sb = new StringBuilder();
            char ch1 = intToChar(i);
           
            for (int j = 0; j < 16; j++) {
                char ch2 = intToChar(j);
                
                
                for (int k = 0; k < 16; k++) {
                    char ch3 = intToChar(k);
                    
                    String s2 = "" + ch1 + ch1 + ch2 + ch2 + ch3 + ch3;
                //    System.out.println("s2 = " + s2);
                    int si = similarity(s, s2);
                    if (si > max) {
                        max = si;
                        ans = "" + ch1 + ch2 + ch3;
                    }
                }
            }
        }
        return "#" + ans;
    }
}
