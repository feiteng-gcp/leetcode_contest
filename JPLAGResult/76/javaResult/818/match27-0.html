class Solution {
    public String similarRGB(String color) {
        char[] ccolor = color.toCharArray();
        if(ccolor.length==4) {
            return color;
        }
        char[] ret = new char[4];
        ret[0] = '#';
        for(int i=0;i<3;i++) {
            char[] c = new char[] {ccolor[i*2+1], ccolor[i*2+2]};
            ret[i+1] = findClosest(c);
        }
        return new String(ret);
    }
    
    private int diff(char[] a, char[] b) {
        if (a.length!=2 || b.length!=2) {
            String astr = new String(a), bstr = new String(b);
            throw new RuntimeException("Cannot find diff between " + astr + " and " + bstr);
        }
        int d = getValue(a) - getValue(b);
        return (- d * d);
    }
    
    private int getValue(char[] a) {
        int ret = 0;
        for(int i=0;i<a.length;i++) {
            ret = ret * 16 + getValue(a[i]);
        }
        return ret;
    }
    
    private int getValue(char a) {
        int ret = a - '0';
        if (ret<0 || ret>9) {
            ret = a - 'a' + 10;
        }
        return ret;
    }
    
    private int diff(char[] a, char b) {
        return diff(a, new char[] {b, b});
    }
    
    private char findClosest(char[] a) {
        int min = -10000000;
        // System.out.println("Finding closest to " + new String(a));
        char ret = '0';
        for(int i=0;i<16;i++) {
            char b;
            if(i<10) {
                b = (char) ('0' + i);
            } else {
                b = (char) ('a' + i - 10);
            }
            int dist = diff(a,b);
            // System.out.println("Score for " + b+ " is " + dist);
            if (dist>min) {
                min = dist;
                ret = b;
            }
        }
        // System.out.println("Closest to " + new String(a) + " is " + ret);
        return ret;
    }
}
