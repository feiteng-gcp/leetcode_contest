class Solution {
    
    
private static String similarVal(String k) {
        int val = Integer.parseInt(k, 16);
        String best = null;
        int bestDiff = Integer.MAX_VALUE;
        for (int i = 0; i < 16; i++) {
            int val0 = i * 16 + i;
            int diff = Math.abs(val0 - val);
            if (diff < bestDiff) {
                bestDiff = diff;
                best = Integer.toHexString(i);
            }
        }
        return best;
    }
    
    
    public String similarRGB(String color) {
        color = color.substring(1);
        String r = color.substring(0, 2),
                g = color.substring(2, 4),
                b = color.substring(4, 6);
        return "#" + similarVal(r) + similarVal(g) + similarVal(b);
    }
}
