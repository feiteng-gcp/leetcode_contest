class Solution {
    private String[] ss = new String[] {"00", "11", "22", "33", "44", "55", "66", "77", "88", "99", "aa", "bb", "cc", "dd", "ee", "ff"};
    
    
private String mostSimilar(String o) {
        String ans = "";
        int min = Integer.MAX_VALUE;
        for (String s : ss) {
            int diff = Math.abs(Integer.parseInt(s, 16) - 
Integer.parseInt(o, 16));
            if (diff < min) {
                min = diff;
                ans = s;
            }
        }
        return ans;
    }
    
    public String similarRGB(String color) {
        return "#" + mostSimilar(color.substring(1, 3)) + mostSimilar(color.substring(3, 5)) + mostSimilar(color.substring(5, 7));
    }
}
