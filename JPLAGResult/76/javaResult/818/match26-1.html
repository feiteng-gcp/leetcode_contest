class Solution {
    public String similarRGB(String color) {
        Integer r = Integer.parseInt(color.substring(1, 3), 16);
        Integer g = Integer.parseInt(color.substring(3, 5), 16);
        Integer b = Integer.parseInt(color.substring(5, 7), 16);
        
        int u = (int) Math.round((double) r/17);
        int w = (int) Math.round((double) g/17);
        int y = (int) Math.round((double) b/17);
        String concat = "#" + Integer.toString(u, 16) + Integer.toString(w, 16) + Integer.toString(y, 16);
        return concat;
        
    }
}
