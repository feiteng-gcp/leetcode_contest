class Solution {

	private int getDiff(String string1 , String string2) {
		
		int ans = 0;
		for (int i = 1;i < string1.length();i += 2) {
			int value1 = getValue(string1.charAt(i) , string1.charAt(i + 1));
			int value2 = getValue(string2.char
At(i) , string2.charAt(i + 1));
			ans += (- (value1 - value2) * (value1 - value2));
		}
		return ans;
		
	}
	
	private int getValue(char ch) {
		if (ch <= '9') {
			return (int) (ch - '0');
		} else {
			return (int) (ch - 'a' + 10);
		}
	}
	
	private int getValue(char ch1 , char ch2) {
	
		int value1 = getValue(ch1);
		int value2 = getValue(ch2);
		return value1 * 16 + value2;
		
	}
	
	private char getChar(int value) {
		if (value < 10) {
			return (char) (value + '0');
		} else {
			return (char) (value - 10 + 'a');
		}
	}
	
    public String similarRGB(String color) {
     
    	int
 ans = 1;
    	String str = "";
    	for (int i = 0;i < 16;i ++) {
    		for (int j = 0;j < 16;j ++) {    			
    			for (int k = 0;k < 16;k ++) {
    				String string = "#";
    				string += getChar(i) + "" + getChar(i);
    				string += getChar(j) + "" + getChar(j);
    				string += getChar(k) + "" + getChar(k);
    				int diff = getDiff(string , color);
    				if (diff > ans || ans > 0) {
    					ans = diff;
    					str = "#" + getChar(i) + "" + getChar(j) + "" + getChar(k);
    				}

    			}
    		}
    	}
    	
    	return str;
    	
    }
	
}
