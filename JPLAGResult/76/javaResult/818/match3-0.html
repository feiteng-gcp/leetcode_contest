class Solution {
    public String similarRGB(String color) {
        String res = "#";
        res = res + help(color.substring(1,3)) + help(color.substring(3,5))+help(color.substring(5,7));
        return res;
    }   
    public String help(String c){
        int color = Integer.parseInt(c,16);
        int min = 255;
        int digit = 0;
        String res = "";
        for(int i=0;i<16;i++){
            if(min > Math.abs(i*16+i-color)){
                min = Math.abs(i*16+i-color);
                digit = i;
            }       
        }
        if(digit == 10)
            return "a";
        if(digit==11)
            return "b";
        if(digit==12)
            return "c";
        if(digit==13)
            return "d";
        if(digit==14)
            return "e";
        if(digit==15)
            return "f";
        return Integer.toString(digit);
    }
}
