class Solution {
    public String similarRGB(String color) {
        StringBuilder sb = new StringBuilder();
        sb.append("#");
        for (int i = 1; i < 7; i += 2) {
            int n = Integer.valueOf(color.substring(i, i+2), 16);
            int m = n % 17;
            if (m <= 8) sb.append(Integer.toString(n / 17, 16));
            else sb.append(Integer.toString(n / 17 + 1, 16));
        }
        return sb.toString();
    }
}
