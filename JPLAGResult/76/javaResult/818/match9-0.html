class Solution {
    public String similarRGB(String color) {
        String res = "#";
        for(int i=0; i<3; i++){
            String num = findBest(color.substring(i*2+1, i*2+3));
            res += num;
        }
        
        return res;
    }
    
    private String findBest(String n){
        int target = Integer.parseInt(n, 16);
        int best_diff = Integer.MAX_VALUE;
        int best_i = 0;
        for(int i=0; i<16; i++){
            int res = i+16*i;
            if(Math.abs(res - target) < best_diff){
                best_diff = Math.abs(res - target);
                best_i = i;
            }
        }
        
        return Integer.toHexString(best_i);
    }
}
