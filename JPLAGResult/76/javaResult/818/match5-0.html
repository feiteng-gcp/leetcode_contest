class Solution {
    public String similarRGB(String color) {
        long num1 = Long.parseLong(color.substring(1, 3), 16);
        long num2 = Long.parseLong(color.substring(3, 5), 16);
        long num3 = Long.parseLong(color.substring(5), 16);
        StringBuilder sb = new StringBuilder();
        sb.append("#").append(closeTo(num1)).append(closeTo(num2)).append(closeTo(num3));
        return sb.toString();
    }
    
    private String closeTo(long num){
        long choice1 = num/ 0x11;
        if(num % 0x11 == 0){
            
            String res = Long.toHexString(choice1 * 0x11);
            if(res.length() < 2)
                res = "0" + res;
            return res;
        }
            
        else{
            long choice2 = num / 0x11 + 1;
            if(Math.abs(num - choice1 * 0x11) < Math.abs(num - choice2 * 0x11)){
                String res = Long.toHexString(choice1 * 0x11);
                if(res.length() < 2)
                    res = "0" + res;
                return res;
            }
            else{
                String res = Long.toHexString(choice2 * 0x11);
                if(res.length() < 2)
                    res = "0" + res;
                return res;
            }
        }
    }
}
