class Solution {
    String[] out = {"00", "11", "22", "33", "44", "55", "66", "77", "88", "99", "aa", "bb", "cc", "dd", "ee", "ff"};
    
    private int hexCharToInt(char c) {
        if (Character.isDigit(c)) {
            return c - '0';
        }
        if (Character.isLowerCase(c)) {
            return (c - 'a') + 10;
        }
        
return (c - 'A') + 10;
    }

    private String findClosest(String sub) {
        char[] sa = sub.toCharArray();
        int c = hexCharToInt(sa[0]) * 16 + hexCharToInt(sa[1]);
        int diff = c;
        int minIndex = 0;

        for (int i = 1; i < 16; i++) {
            
if (Math.abs(i * 16 + i - c) < diff) {
                diff = Math.abs(i * 16 + i - c);
                minIndex = i;
            }
        }
        return out[minIndex];
    }
    
    public String similarRGB(String color) {
        return "#" + findClosest(color.substring(1,3)) + findClosest(color.substring(3,5)) + findClosest(color.substring(5,7));  
    }
}
