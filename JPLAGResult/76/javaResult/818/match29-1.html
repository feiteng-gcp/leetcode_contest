class Solution {
    public String similarRGB(String color) {
        StringBuilder builder = new StringBuilder("#");
        for (int i = 0; i < 3; i++) {
            int r = 0, min = 10000;
            int o = Integer.parseInt(color.substring(1 + i * 2, 1 + (i + 1) * 2), 16);
            for (int j = 0; j < 16; j++) {
                if (Math.abs(j * 16 + j - o) < min) {
                    r = j;
                    min = Math.abs(j * 16 + j - o);
                
}
            }
            builder.append(Integer.toString(r, 16));
        }
        return builder.toString();
    }
}
