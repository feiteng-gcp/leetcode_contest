class Solution {
    public String similarRGB(String color) {
        StringBuilder sb = new StringBuilder();
        sb.append('#');
        for (int i = 1; i < color.length(); i += 2) {
        	int similarity = Integer.MAX_VALUE;
        	int ind
ex = -1;
        	int value = 0;
        	for (int j = 0; j < 2; ++j) {
        		char ch = color.charAt(i + j);
        		if (ch >= '0' && ch <= '9') {
        			value = value * 16 + ch - '0';
        		} else {
        			value = value * 16 + 
ch - 'a' + 10;
        		}
        	}
        	for (int j = 0; j < 16; ++j) {
        		int nvalue = j * 16 + j;
        		if ((nvalue - value) * (nvalue - value) < similarity) {
        			similarity = (nvalue - value) * (nvalue - value);
        			index = j;
        		}
        	}
        	char x = index < 10 ? (char)(index + '0') : (char)(index - 10 + 'a');
        	sb.append(x);
        }
        return sb.toString();
    }
}
