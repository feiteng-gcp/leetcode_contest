class Solution {
    public String similarRGB(String color) {
            Map<Character, Integer> m = new HashMap<>();
            for (int i = 0; i <= 9; i++) {
                m.put((char)(i + '0'), i);
            }
            m.put('a', 10);
            m.put('b', 11);
            m.put('c', 12);
            m.put('d', 13);
            m.put('e', 14);
            m.put('f', 15);
            int c = m.get(color.charAt(6)) + 16 * m.get(color.charAt(5));
            int b = m.get(color.charAt(4)) + 16 * m.get(color.charAt(3));
            int a = m.get(color.charAt(2)) + 16 * m.get(color.charAt(1));
            StringBuilder sb =new StringBuilder();
            sb.append("#").append(findnear(a)).append(findnear(b)).append(findnear(c));
            return sb.toString();
        }
        public char findnear(int a){
            int max = Integer.MIN_VALUE;
            int best = 0;
            for (int i = 0; i < 16; i++) {
                int b = i + 16 * i;
                b = calcDiff(a, b);
                if (b >= max) {
                    max = b;
                    best = i;
                }
            }
            if (best < 10) return (char) (best + '0');
            return (char) (best + 'a' - 10);
        }

        public int sq(int a){
            return a * a;
        }
        public int calcDiff(int a, int ca){
            return -(sq(a - ca));
        }

}
