class Solution {
    public String similarRGB(String color) {
        String res = "#";
        for (int i = 1; i < color.length(); i += 2) {
            String sub = color.substring(i, i + 2);
            String tmp = findMin(sub);
            res += tmp;
        }

        return res;
    }

    public String findMin(String sub) {
        int dist = Integer.MAX_VALUE;
        String res = "";
        for (int i = 0; i < 10; i++) {
            String can = "";
            can += i;
            can += i;
            int d = distance(sub, can);
            if (dist > d) {
                dist = d;
                res = "" + i;
            }
        }

        for (char c = 'a'; c <= 'f'; c++) {
            String can = "";
            can += c;
            can += c;
            int d = distance(sub, can);
            if (dist > d) {
                dist = d;
                res = "" + c;
            }
        }
        return res;
    }

    public int distance(String s1, String s2) {
        int v1 = getValue(s1);
        int v2 = getValue(s2);
        return (v1 - v2) * (v1 - v2);
    }

    public int getValue(String s1) {
        int res = hexBase(s1.charAt(0));
        res = res * 16 + hexBase(s1.charAt(1));

        return res;
    }
    public int hexBase(char c) {
        if (c >= '0' && c <= '9') {
            return c - '0';
        }
        else {
            return c - 'a' + 10;
        }
    }
}
