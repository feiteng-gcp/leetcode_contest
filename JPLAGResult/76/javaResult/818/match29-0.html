class Solution {
    public String similarRGB(String color) {
        char[] c = color.toCharArray();
        StringBuilder result = new StringBuilder();
        result.append('#');
        for ( int i=1; i<6; i+=2) {
            int temp = val(c[i]);
            int value = temp*16+val(c[i+1]);
            int diff = Integer.MAX_VALUE;
            int index =-1;
            //System.out.println("temp = "+ temp);
            for (int j= temp-1; j<=temp+1; j++) {
                if (j<0) continue;
                if (j>=16) continue;
                //System.out.println("j = "+ j);
                int cal = (j*16+j -value);
                cal *=cal;
                if (cal<diff) {
                    diff = cal;
                    index = j;
                }
            }
            result.append(toCh(index));
        }
        return result.toString();
    }
    public int val (char ch) {
        if (ch >= '0' && ch<= '9') return (ch-'0');
        return (10+ch-'a');
    }
    
    public char toCh (int val) {
        //System.out.println("val = "+ val);
        if (val <10) return ((char)(val+'0'));
        return ((char)(val-10+'a'));
    }
}
