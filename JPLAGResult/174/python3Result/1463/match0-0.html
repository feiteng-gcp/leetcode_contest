class Solution:
    def kWeakestRows(self, mat: List[List[int]], k: int) -> List[int]:
        
        
        pool = []
        for i,row in enumerate(mat):
            pool.append((sum(row), i))
        pool.sort(key = lambda x:(x[0], x[1]))
        
        return [i for s,i in pool[:k]]
