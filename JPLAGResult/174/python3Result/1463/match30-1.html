class Solution:
    def kWeakestRows(self, mat: List[List[int]], k: int) -> List[int]:
        res = []
        for i in range(len(mat)):
            res.append([mat[i].count(
1), i])
        res.sort()
        t = []
        for i in range(k):
            t.append(res[i][1])
        return t
