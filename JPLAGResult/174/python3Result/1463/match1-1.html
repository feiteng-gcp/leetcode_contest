class Solution:
    def kWeakestRows(self, mat: List[List[int]], k: int) -> List[int]:
        arr = []
        for i, row in enumerate(mat):
            arr.append((sum(row), i))
        arr.sort()
        out = []
        for j in range(k):
            out.append(arr[j][1])
        return out
