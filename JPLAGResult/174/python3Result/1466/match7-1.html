class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        dp = [1] * len(arr)
        indices = sorted(enumerate(arr), key=lambda x: -x[1])
        indices = [i for i, _ in indices]
        
        
for i in indices:
            for j in range(i - 1, max(0, i - d) - 1, -1):
                if arr[j] >= arr[i]:
                    break
                dp[j] = max(dp[j], dp[i] + 1)
            for j in range(i + 1, min(len(arr), i + d + 1)):
                if arr[j] >= arr[i]:
                    break
                dp[j] = max(dp[j], dp[i] + 1)
                
        return max(dp)
