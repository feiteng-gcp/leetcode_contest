import functools


class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        @functools.lru_cache(maxsize=None)
        def f(i):
            i0 = max(0, i - d)
            i1 = min(len(arr) - 1, i + d)
            s = 1
            for k in range(1, d + 1):
                if i - k < 0 or arr[i - k] >= arr[i]:
                    break
                s = max(s, f(i - k) + 1)
            for k in range(1, d + 1):
                if i + k >= len(arr) or arr[i + k] >= arr[i]:
                    break
                s = max(s, f(i + k) + 1)
            return s

        return max(f(i) for i in range(len(arr)))
