class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        conn = collections.defaultdict(list)
        for i in range(len(arr)):
            j = 1
            conn[i] += [i]
            while j <= d:
                if i-j >= 0 and arr[i-j] < arr[i]:
                    conn[i] += [i-j]
                else: break
                j += 1
            j = 1
            while j <= d:
                if i+j < len(arr) and arr[i+j] < arr[i]:
                    conn[i] += [i+j]
                else: break
                j += 1
        self.memo = {}
        def dp(node):
            if node not in self.memo:
                if len(conn[node]) == 1: self.memo[node] = 1
                else: self.memo[node] = max([dp(child) for child in conn[node] if child != node]) + 1
            return self.memo[node]
        for i in range(len(arr)): dp(i)
        return max(self.memo.values())
