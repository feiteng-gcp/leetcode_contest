class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        inds = []
        for i, x in enumerate(arr):
            inds.append((x, i))
        inds.sort()
        f = [0] * len(arr)
        for _, i in inds:
            for j in range(i - 1, max(0, i - d) - 1, -1):
                if arr[j] >= arr[i]:
                    break
                f[i] = max(f[i], f[j])
            for j in range(i + 1, min(len(arr), i + d + 1)):
                if arr[j] >= arr[i]:
                    break
                f[i] = max(f[i], f[j])
            f[i] += 1
        return max(f)
