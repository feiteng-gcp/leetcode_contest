class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        N = len(arr)
        dp = [0] * N
        tmp = [(i, arr[i]) for i in range(N)]
        tmp.sort(key=lambda x: x[1])
        
        for i in range(N):
            cur, x = tmp[i]
            for j in range(1, d+1):
                if 0 <= cur + j < N:
                    if arr[cur+j] < x:
                        dp[cur] = max(dp[cur], dp[cur+j] + 1)
                    else:
                        break
            for j in range(1, d+1):
                if 0 <= cur - j < N:
                    if arr[cur-j] < x:
                        dp[cur] = max(dp[cur], dp[cur-j] + 1)
                    else:
                        break 
        
        return max(dp) + 1
