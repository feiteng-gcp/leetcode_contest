class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        n = len(arr)
        mem = {}

        def dfs(s):
            if s in mem:
                return mem[s]
            
            res = 0
            for i in range(s+1, s+d+1):
                if i >= n or arr[i] >= arr[s]: break
                res = max(res, dfs(i))
            for i in range(s-1, s-d-1, -1):
                if i < 0 or arr[i] >= arr[s]: break
                res = max(res, dfs(i))

            mem[s] = res + 1
            return res + 1

        res = 0
        for i in range(n):
            res = max(res, dfs(i))
        return res
