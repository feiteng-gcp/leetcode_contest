from functools import lru_cache


class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        return solve(arr, d)


def solve(a, d):
    n = len(a)

    @lru_cache(None)
    def f(i):
        v = 1
        for k in range(1, d + 1):
            j = i - k
            if j < 0:
                break
            
if a[j] >= a[i]:
                break
            v = max(v, f(j) + 1)

        for k in range(1, d + 1):
            j = i + k
            if j >= n:
                break
            
if a[j] >= a[i]:
                break
            v = max(v, f(j) + 1)
        return v

    return max([f(i) for i in range(n)])
