class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        # print([i for i in reversed(range(1,10))])
        
        ans_dict = {}
        def max_step(p):
            if p in ans_dict:
                return ans_dict[p]
            ans = 0 
            for i in reversed(range(max(0,p-d),p)):
                if arr[i]>=arr[p]:
                    break
                ans = max(max_step(i)+1, ans)
                
            for i in range(p+1, min(p+d+1,len(arr))):
                if arr[i]>=arr[p]:
                    break
                ans = max(max_step(
i)+1, ans)
                
            ans_dict[p] = ans
            return ans
        
        
        ans = 0
        for i in range(len(arr)):
            ans = max(ans,max_step(i))
        # print(ans_dict)
        return ans+1
