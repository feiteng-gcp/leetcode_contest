from functools import lru_cache
class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        n = len(arr)
        @lru_cache(None)
        def helper(i):
            ans = 1
            for dd in range(1, d + 1):
                j = i - dd
                if j < 0 or arr[j] >= arr[i]: break
                ans = max(ans, 1 + helper(j))
            for dd in range(1, d + 1):
                j = i + dd
                if j > n - 1 or arr[j] >= arr[i]: break
                ans = max(ans, 1 + helper(j))
            return ans
        return max(helper(i) for i in range(n))
