class Solution:
    def maxJumps(self, arr: List[int], k: int) -> int:
        n = len(arr)
        dp = [0] * n
        d = [(i, arr[i]) for i in range(n)]
        d = sorted
(d, key=lambda x: x[1])
        
        for i in range(n):
            cur, x = d[i][0], d[i][1]
            for j in range(1, k+1):
                if 0 <= cur + j < n:
                    if arr[cur+j] < x:
                        dp[cur] = max(dp[cur], dp[cur+j] + 1)
                    else:
                        break
                        
            for j in range(1, k+1):
                if 0 <= cur - j < n:
                    if arr[cur-j] < x:
                        dp[cur] = max(dp[cur], dp[cur-j] + 1)
                    else:
                        break 
        
        return max(dp) + 1
