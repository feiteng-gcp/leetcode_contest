class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        dp = [1] * len(arr)
        length = len(arr)
        v_i = sorted([(v,i) for i,v in enumerate(arr)])
        for v, i in v_i:
            # right
            for l in range(1, d+1):
                
if i+l >= length or arr[i+l] >= v:
                    break
                dp[i] = max(dp[i], dp[i+l]+1)
            # left
            for l in range(1, d+1):
                
if i-l < 0 or arr[i-l] >= v:
                    break
                dp[i] = max(dp[i], dp[i-l]+1)
        return max(dp)
