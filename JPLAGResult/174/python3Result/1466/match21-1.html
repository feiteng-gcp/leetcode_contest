class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        n = len(arr)
        idxes = sorted([i for i in range(n)], key = lambda i:arr[i])
        res = 1
        dp = [-1 for i in range(n)]
        for idx in idxes:
            dp[idx] = 1
            # print(idx, (max(0, idx - d), min(n, idx + d)))
            
for nxt in range(idx - 1, max(-1, idx - d - 1), -1):
                if arr[idx] <= arr[nxt]: break
                dp[idx] = max(dp[idx], 1 + dp[nxt])
            for nxt in range
(idx + 1, min(n, idx + d + 1)):
                if arr[idx] <= arr[nxt]: break
                dp[idx] = max(dp[idx], 1 + dp[nxt])
            
            # for nxt in range(max(0, idx - d), min(n, idx + d + 1)):
            #     # print(idx, nxt)
            #     if dp[nxt] == -1: continue
            #     if nxt == idx: continue
            #     if arr[idx] <= arr[nxt]: continue
            #     print(idx, nxt)
            #     dp[idx] = max(dp[idx], 1 + dp[nxt])
            # print(idx, (max(0, idx - d), min(n, idx + d)), dp[idx])
        
        # print(idxes)
        # print(dp)
        return max(dp)
