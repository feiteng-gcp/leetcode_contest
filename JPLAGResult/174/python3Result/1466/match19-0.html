from functools import lru_cache

class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        
        @lru_cache(None)
        def dp(idx):
            
            res = 1
            for i in range(idx - 1, max(-1, idx - d - 1), -1):
                if arr[i] >= arr[idx]:
                    break
                res = max(res, 1 + dp(i))
                
            for i in range(idx + 1, min(len(arr), idx + d + 1)):
                if arr[i] >= arr[idx]:
                    break
                res = max(res, 1 + dp(i))
            return res
        
        return max(dp(i) for i in range(len(arr)))
