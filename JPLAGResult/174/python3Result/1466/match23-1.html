class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        sorted_idx = [t[0] for t in sorted(enumerate(arr), key=lambda x: x[1])]
        # print(sorted_idx)
        dp = [1 for i in range(len
(arr))]
        for curr in sorted_idx:
            for i in range(curr - 1, max(-1, curr - d - 1), -1):
                if arr[i] >= arr[curr]:
                    break
                dp[curr] = max(dp[curr], 1 + dp[i])

            for i in range(curr + 1, min(len(arr), curr + d + 1)):
                if arr[i] >= arr[curr]:
                    break
                dp[curr] = max(dp[curr], 1 + dp[i])

        # print(dp)
        return max(dp)
