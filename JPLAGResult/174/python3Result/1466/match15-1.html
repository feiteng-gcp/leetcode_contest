from functools import lru_cache
class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        
@lru_cache(None)
        def recur(i):
            stp = 0
            for j in range(i+1,min(i+d+1, len(arr))):
                if arr[j]>=arr[i]:
                    break
                stp = max(stp,recur(j))
            for j in range(i-1,max(i-d-1,-1),-1):
                
if arr[j]>=arr[i]:
                    break
                stp = max(stp,recur(j))
            return stp+1
        return max(recur(i) for i in range(len(arr)))
