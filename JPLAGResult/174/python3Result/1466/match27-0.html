class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        
        n = len(arr)
        sarr = sorted([(val, idx) for idx, val in enumerate(arr)])
        res = [1 for _ in range(n)]
        
        for val, idx in sarr:
            # print(idx, res)
            
            # left
            for step in range(1, d + 1):
                i = idx - step
                if i < 0 or i >= n or arr[i] >= val:
                    break
                
                res[idx] = max(res[idx], res[i] + 1)
            
            # right
            for step in range(1, d + 1):
                i = idx + step
                if i < 0 or i >= n or arr[i] >= val:
                    break
                
                res[idx] = max(res[idx], res[i] + 1)
                    
        return max(res)
