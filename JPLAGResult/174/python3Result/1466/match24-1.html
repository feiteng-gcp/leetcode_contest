import functools
class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        @functools.lru_cache(None)
        def dp(idx):
            if idx<0:
                return 0
            start
=max(0,idx-d)
            end=min(len(arr)-1,idx+d)
            res=0
            for i in range(idx-1,start-1,-1):
                if arr[idx]>arr[i]:
                    res=max(res,dp(i)+1)
                else:
                    break
            for i in range(idx+1,end+1):
                if arr[idx]>arr[i]:
                    res=max(res,dp(i)+1)
                else:
                    break
            return res
        return max(dp(i)+1 for i in range(len(arr)))
