class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        import functools
        @functools.lru_cache(None)
        def dfs(p):
            res = 0
            for i in reversed(range(max(0, p-d), p)):
                if arr[i]>=arr[p]:
                    break
                res = max(res, dfs(i))
            for i in range(p+1, min(p+d+1, len(arr))):
                if arr[i]>=arr[p]:
                    break
                res = max(res, dfs(i))
            return res + 1
        res = 0
        for i in range(len(arr)):
            res = max(res, dfs(i))
        return res
