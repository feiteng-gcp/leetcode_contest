class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        res=[[v,i] for i,v in enumerate(arr)]
        n=len(arr)
        dp=[1]*n
        res.sort()
        for v,i in res:
            for j in range(1,d+1):
                if i-j<0 or arr[i-j]>=arr[i]:
                    break
                #print(i,j)
                dp[i]=max(dp[i],dp[i-j]+1)
                #print(dp[i])
            for j in range(1,d+1):
                if i+j>=n or arr[i+j]>=arr[i]:
                    break
               # print(i,j)
                dp[i]=max(dp[i],dp[i+j]+1)
                #print(dp[i])
        #print(dp)
        return max(dp)
