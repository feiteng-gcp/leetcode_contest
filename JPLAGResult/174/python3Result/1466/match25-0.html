class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        mem = {}
        
        def dp(i):
            if i in mem:
                return mem[i]
            res = 1
            for k in range(i + 1, min(len(arr), i + d + 1)):
                if arr[k] < arr[i]:
                    res = max(res, dp(k) + 1)
                else:
                    break
                    
            for k in reversed(range(max(0, i - d), i)):
                if arr[k] < arr[i]:
                    res = max(res, dp(k) + 1)
                else:
                    break
            mem[i] = res
            return res
        
        res = 0
        for i in range(len(arr)):
            res = max(res, dp(i))
            
        # print(mem)
            
        return res
