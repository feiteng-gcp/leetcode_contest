class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        dp = [1 for i in range(len(arr))]
        hi = [ (arr[i], i) for i in range(len(arr))]
        hi.sort(reverse=True)
        
        for _, p in hi:
            for i in range(d):
                q = p + i + 1
                if q >= len(arr) or arr
[q] >= arr[p]:
                    break
                dp[q] = max(dp[q], dp[p] + 1)
            for i in range(d):
                q = p - i - 1
                
if q < 0 or arr[q] >= arr[p]:
                    break
                dp[q] = max(dp[q], dp[p] + 1)
        return max(dp)
