class Solution:
    def maxJumps(self, arr: List[int], d: int) -> int:
        N = len(arr)
        dp = [-1] * N
        def count(i):
            if dp[i] != -1:
                
return dp[i]
            max_ = 0
            for j in range(i + 1, i + d + 1):
                if j >= N or arr[j] >= arr[i]:
                    break
                max_ = max(max_, count(j))
            for j in range(i - 1, i - d - 1, -1):
                if j < 0 or arr[j] >= arr[i]:
                    break
                max_ = max(max_, count(j))
            max_ += 1
            dp
[i] = max_
            return max_
        max_ = 0
        for i in range(N):
            max_ = max(max_, count(i))
        return max_
