# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        sums = set()
        
        def helper(node):
            if not node:
                return 0
            
            left = helper(node.left)
            right = helper(node.right)
            
            ret = left + node.val + right
            sums.add(ret)
            
            return ret
        
        total = helper(root)
        
        answer = float('-inf')
        for number in sums:
            answer = max(answer, number * (total - number))
        
        return answer % (10 ** 9 + 7)
