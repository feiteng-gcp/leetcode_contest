class Solution:
    def maxProduct(self, root: TreeNode) -> int:

        def dfs_sum(node):
            if not node:
                return 0

            return node.val + dfs_sum(node.left) + dfs_sum(node.right)

        total = dfs_sum(root)

        most = 0

        def dfs(node):
            nonlocal most
            if not node:
                return 0

            cur = node.val + dfs(node.left) + dfs(node.right)
            most = max(most, (total - cur) * cur)
            return cur

        dfs(root)
        return most % int(1e9 + 7)
