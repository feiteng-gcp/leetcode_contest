# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        self.sum_dict = {0}
        def get_sum(node):
            if not node:
                return 0
            tmp = node.val + get_sum(node.left) + get_sum(node.right) 
            self.sum_dict.add(tmp)
            return tmp
        get_sum(root)
        m = max(self.sum_dict)
        res = [i*(m-i) for i in self.sum_dict]
        return max(res) % 1000000007
