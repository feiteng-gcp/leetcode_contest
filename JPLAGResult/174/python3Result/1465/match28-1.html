# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        
def dfs(node):
            res = node.val
            if node.left:
                res += dfs(node.left)
            if node.right:
                res += dfs(node.right)
            s.append(res)
            return res
        s = []
        total = dfs(root)
        ma = 0
        print(s)
        for num in s:
            tmp = num*(total-num)
            if tmp>ma:
                ma = tmp
        return int(ma%(1e9+7))
