# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def total(self, node):
        if node is None:
            return 0
        return node.val + self.total(node.left) + self.total(node.right)
    def maxpro(self, node):
        if node is None:
            return 0
        l = self.maxpro(node.left)
        r = self.maxpro(node.right)
        c = l + r + node.val
        self.ans = max(self.ans, c * (self.t - c))
        return c
    def maxProduct(self, root: TreeNode) -> int:
        self.t = self.total(root)
        self.ans = 0
        self.maxpro(root)
        return self.ans % int(1e9 + 7)
