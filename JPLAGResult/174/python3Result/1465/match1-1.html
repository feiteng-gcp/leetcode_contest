class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        def dfs_sum(root):
            if not root:
                return 0
            return root.val + dfs_sum(root.left) + dfs_sum(root.right)
        total = dfs_sum(root)
        self.res = 0
        def dfs(root):
            if not root:
                return 0
            left = dfs(root.left)
            self.res = max(self.res, left * (total-left))
            right = dfs(root.right)
            self.res = max(self.res, right*(total -right))
            return root.val + left + right
        dfs(root)
        return self.res % (10**9 + 7)
