# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        total=0
        def dfs(node):
            nonlocal total
            if not node:
                return
            total+=node.val
            dfs(node.left)
            dfs(node.right)
        dfs(root)
        res=0
        def findSub(node):
            nonlocal total,res
            if not node:
                return 0
            l=findSub(node.left)
            r=findSub(node.right)
            curr=l+r+node.val
            res=max(res,(total-curr)*curr)
            return curr
        findSub(root)
        return res%(10**9+7)
