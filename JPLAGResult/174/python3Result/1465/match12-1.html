# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        m = 10 ** 9 + 7
        self
.total = 0
        def get_total(node):
            if not node:
                return
            self.total += node.val
            get_total(node.left)
            get_total(node.right)
        get_total(root)
        # print(self.total)
        self.max_ = 0
        def inorder(node):
            if not node:
                return 0
            l_sum = inorder(node.left)
            r_sum = inorder(node.right)
            self.max_ = max(self.max_, l_sum * (self.total - l_sum), r_sum * (self.total - r_sum))
            return l_sum + node.val + r_sum
        inorder(root)
        return self.max_ % m
