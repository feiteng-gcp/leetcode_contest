# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        self.d = {}
        ans = -float("inf")
        total_s = self.dfs(root)
        for s in self.d.values():
            ans = max(
ans, s * (total_s - s))
        return ans % (10**9 + 7)
        
    def dfs(self, root):
        if root is None:
            return 0
        s = root.val
        s += self.dfs(root.left)
        s += self.dfs(root.right)
        self.d[root] = s
        return s
