# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        self.total 
= 0
        
        def count_total(root):
            if root == None: return
            self.total += root.val
            count_total(root.left)
            count_total(root.right)
            
        count_total(root)
        self.res = 0
        
        
def find_sum(root):
            if root == None: return 0
            sl = find_sum(root.left)
            sr = find_sum(root.right)
            
            self.res = max(self.res, sl * (self.total - sl), sr * (self.total - sr))
            
            return sl + sr + root.val
        find_sum(root)
            
        return self.res % (10**9 + 7)
