# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        ss = []
        
def sumr(rt) :
            if rt is None :
                return 0
            ls = sumr(rt.left)
            lr = sumr(rt.right)
            tot = ls + lr + rt.val
            ss.append(tot)
            return tot
        tot = sumr(root)
        res = 0
        for v in ss :
            u = tot - v
            res = max(u*v,res)
        return res % int(1e9+7)
