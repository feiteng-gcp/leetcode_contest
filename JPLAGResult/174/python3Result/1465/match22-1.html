# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        def dfs_sum(node):
            if not node:
                return 0
            return node.val + dfs_sum(node.left) + dfs_sum(node.right)
        
        tree_sum = dfs_sum(root)
        max_prod = -1
        
        def dfs(node):
            nonlocal max_prod
            if not node:
                return 0
            sum_l = dfs(node.left)
            sum_r = dfs(node.right)
            max_lr = max(sum_l * (tree_sum - sum_l), sum_r * (tree_sum - sum_r))
            max_prod = max(max_prod, max_lr)
            return node.val + sum_l + sum_r
        
        dfs(root)
        
        return max_prod % (10 ** 9 + 7)
