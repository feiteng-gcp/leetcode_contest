# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        self.d = {}
        self.dfs(root)
        #print(self.d)
        total = self.d[root]
        #print(total)
        mp = -float('inf')
        for k in self.d:
            mp = max(mp, self.d[k]*(total-self.d[k]))
        return mp%(10**9 + 7)
        
        
    
    def dfs(self, root):
        if root is None:
            return 0
        s = root.val
        s+=self.dfs(root.left)
        s+=self.dfs(root.right)
        self.d[root] = s
        return s
