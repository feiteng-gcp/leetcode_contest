# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        
        
        def dfs(node):  # 返回 node 子树的 sum
            if node is None:
                return 0
            return node.val + dfs(node.left) + dfs(node.right)
        
        whole = dfs(root)
        self.res = 0
        
        def dfs2(node):  # 返回 node 子树的 sum
            if node is None:
                return 0
            l, r = dfs2(node.left), dfs2(node.right)
            self.res = max(self.res, l*(whole-l), r*(whole-r))
            return node.val + l + r
        
        dfs2(root)
        return self.res % (10**9 + 7)
