# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        # (node_val + right_val) * left_val
        res = 0
        sum_val = 0
        MOD = 1000000007
        
        def get_val(node):
            nonlocal sum_val
            if node == None: return
            sum_val += node.val
            get_val(node.left)
            get_val(node.right)
        def get_sum(node):
            nonlocal res, sum_val, MOD
            if node == None: return 0
            left_sum = get_sum(node.left)
            right_sum = get_sum(node.right)
            val = (node.val + left_sum + right_sum)
            res = max(res, (sum_val - val) * val, (node.val + left_sum) * right_sum, (node.val + right_sum) * left_sum )
            return val
        get_val(root)
        get_sum(root)

       
        return int(res % MOD)
