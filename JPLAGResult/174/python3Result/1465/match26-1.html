# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        sums = []
        
        
def dfs(node):
            if not node:
                return 0
            result = dfs(node.left) + dfs(node.right) + node.val
            sums.append(result)
            return result
        
        s = dfs(root)
        return max(x * (s - x) for x in sums) % (10 ** 9 + 7)
