# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        def get_sum(root):
            if not root:
                return 0
            ret=root.val+get_sum(root.left)+get_sum(root.right)
            mem[root]=ret
            return ret
        mem={}
        _sum=get_sum(root)
        ret=0
        for k in mem:
            ret=max(ret,(_sum-mem[k])*mem[k])
        return ret%(10**9+7)
