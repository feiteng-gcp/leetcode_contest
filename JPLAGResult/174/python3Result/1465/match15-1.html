class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        
        def sumall(node):
            if not node:
                return 0
            
            return node.val + sumall(node.left) + sumall(node.right)
        
        
        s = sumall(root)
        
        
        def dfs(node):
            if not node:
                return float('-inf'), 0
            
            
            la, lb = dfs(node.left)
            ra, rb = dfs(node.right)
            
            c = node.val + lb + rb
            d = s - c
            
            return max(la, ra, c * d), c
        
        
        ans, _ = dfs(root)
        
        return ans % (10**9+7)
