# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        ### first get total sum
        self.all_sum = 0
        def traversal(root):
            if not root:
                return
            self.all_sum += root.val
            traversal(root.left)
            traversal(root.right)
            
        traversal(root)
        
        self.res = 0
        def get_sum(node):
            if not node:
                return 0
            l = get_sum(node.left)
            r = get_sum(node.right)
            self.res = max(self.res, l*(self.all_sum-l))
            self.res = max(self.res, r*(self.all_sum-r))
            return l+r+node.val
    
        get_sum(root)
        return self.res%(10**9+7)
