# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        sums=set()
        def dfs(node):
            if not node:
                return 0
            left=dfs(node.left)
            right=dfs(node.right)
            res=left+right+node.val
            sums.add(res)
            return res
        total=dfs(root)
        minDiff=math.inf
        minDiffSum=0
        for s in sorted(sums):
            diff=abs(s-total/2)
            if diff<minDiff:
                minDiff=diff
                minDiffSum=s
        return (minDiffSum*(total-minDiffSum))%(10**9+7)
