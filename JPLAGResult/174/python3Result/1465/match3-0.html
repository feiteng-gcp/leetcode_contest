class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        def dfs1(root):
            if not root:
                return 0
            left = dfs1(root.left)
            self.res = max(self.res, left * (total-left))
            right = dfs1(root.right)
            self.res = max(self.res, right*(total -right))
            return root.val + left + right
        
        
        def dfs(root):
            if not root:
                return 0
            return root.val + dfs(root.left) + dfs(root.right)
        total = dfs(root)
        self.res = 0
        
        
        
        dfs1(root)
        
        return self.res % (10**9 + 7)
