# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        s = 0
        
        def dfs(node):
            if node is None:
                return
            nonlocal s
            s += node.val
            dfs(node.left)
            dfs(node.right)
        dfs(root)
        best = 0
        def dfs2(node):
            if node is None:
                return 0
            nonlocal best
            
            val = node.val
            val += dfs2(node.left)
            val += dfs2(node.right)
            best = max(best, val * (s - val))
            #print(s, s-val, val *(s-val))
            return val
        dfs2(root)
        return best % (10 ** 9 + 7)
