# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        MOD = 1e9 + 7
        s = collections.defaultdict(int)
        
        def dfs(root):
            if not root: return 0
            s[root] = root.val
            if root.left:
                s[root] += dfs(root.left)
            if root.right:
                s[root] += dfs(root.right)
            return s[root]
        
        tot = dfs(root)
        self.res = 0
        print(s.values())
        
        def dfs2(root):
            if not root: return
            if root.left:
                self.res = max(self.res, s[root.left] * (tot - s[root.left]))
                dfs2(root.left)
            if root.right:
                self.res = max(self.res, s[root.right] * (tot - s[root.right]))
                dfs2(root.right)
                
        dfs2(root)
        return int(self.res % MOD)
