# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        MOD = 10**9 + 7
        
        self.total 
= 0

        def dfs0(node):
            if node is None:
                return
            self.total += node.val
            dfs0(node.left)
            dfs0(node.right)
        
        dfs0(root)
        
        self.res = 0
        
        def dfs1(node):
            if node is None:
                return 0
            s = dfs1(node.left) + dfs1(node.right) + node
.val
            self.res = max(self.res, s * (self.total - s))
            return s
        
        dfs1(root)
        return self.res % MOD
