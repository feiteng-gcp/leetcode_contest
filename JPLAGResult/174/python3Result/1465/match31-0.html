class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        def sum(root):
            if root == None:
                return 0
            rnt = root.val + sum(root.left) + sum(root.right)
            root.val = rnt
            return rnt

        Total = sum(root)
        ans = [0]
        # print(tree_to_list(root))
        def cal(root):
            if root == None:
                return
            ans[0] = max(ans[0], root.val * (Total - root.val))
            # print(ans)
            if root.val * 2 > Total:
                cal(root.left)
                cal(root.right)
        cal(root)
        return ans[0] % 1000000007
