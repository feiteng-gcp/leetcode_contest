# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        M = 10**9 + 7
        all_sum = []
        def get_sum(root):
            if root is None:
                return 0
            this_sum = root.val + get_sum(root.left) + get_sum(root.right)
            all_sum.append(this_sum)
            return this_sum
        tot = get_sum(root)
        max_prod = 0
        for i in all_sum:
            max_prod = max(max_prod, i * (tot - i))
        return max_prod % M
