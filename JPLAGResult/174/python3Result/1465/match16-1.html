# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.d = dict()
    
    def maxProduct(self, root: TreeNode) -> int:
        if root is None:
            return 0
        maxi = float('-inf')
        self.getsum(root)
        tot = self.d[root]
        
for key, val in self.d.items():
            maxi = max(val*(tot-val), maxi)
        return maxi%(10**9+7)
        
        
    
    def getsum(self, node):
        if node is None:
            return 0
        s = node.val
        s += self.getsum(node.left)
        s += self.getsum(node.right)
        self.d[node] = s
        return s
