# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        sums = set()
        
def dfs(node):
            if node is None:
                return 0
            res = node.val + dfs(node.left) + dfs(node.right)
            sums.add(res)
            return res
        
        allSum = dfs(root)
        diff = min([abs(allSum - 2 * s) for s in sums])
        return int((allSum - diff) * (allSum + diff) / 4) % (10**9+7)
