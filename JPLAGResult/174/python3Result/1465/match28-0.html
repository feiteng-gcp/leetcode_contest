# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        sums = []
        def dfs(node):
            s = node.val
            if node.left: s += dfs(node.left)
            if node.right: s += dfs(node.right)
            sums.append(s)
            return s
        
        s = dfs(root)
        M = 0
        for x in sums:
            M = max(M, x * (s - x))
        return M % (10**9+7)
