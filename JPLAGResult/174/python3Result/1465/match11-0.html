# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

MM = 10 ** 9 + 7
class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        res, sumv = [0], [0]
        def get_sum(node):
            if not node: return
            sumv[0] += node.val
            get_sum(node.left)
            get_sum(node.right)
        def helper(node):
            if not node: return 0

            l = helper(node.left)
            r = helper(node.right)
            res[0] = max(res[0], l * (sumv[0] - l), r * (sumv[0] - r))
            # print(node.val, l, r, res[0])

            return l + r + node.val
        get_sum(root)
        # print(sumv)
        helper(root)
        return res[0] % MM
