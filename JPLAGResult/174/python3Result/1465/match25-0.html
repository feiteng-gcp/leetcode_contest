# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxProduct(self, root: TreeNode) -> int:
        
        res = [0]
        def dfs(node):
            if node is None:
                return
            res[0] += node.val
            dfs(node.left)
            dfs(node.right)
            
        dfs(root)
        
        ret = [0]
        
        def sums(node):
            if node is None:
                return 0
            
            cur_sum = node.val
            cur_sum += sums(node.left)
            cur_sum += sums(node.right)
            ret[0] = max(ret[0], cur_sum * (res[0] - cur_sum))
            return cur_sum
            
        sums(root)
        return ret[0] % (10**9 + 7)
