# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.ans = 0
        self.sum_ = 0
        self.mod = 10 ** 9 + 7
        
        
    
def get_sum(self, node):
        if not node:
            return 0
        return node.val + self.get_sum(node.left) + self.get_sum(node.right)
    
    def postorder(self, node):
        if not node:
            return 0
        
        left_sum = self.postorder(node.left)
        right_sum = self.postorder(node.right)
        all_sum = left_sum + right_sum + node.val
        self.ans = max(self.ans, (self.sum - all_sum) * all_sum)
        return all_sum
    
    def maxProduct(self, root: TreeNode) -> int:
        self.sum = self.get_sum(root)
        self.postorder(root)
        return self.ans % self.mod
