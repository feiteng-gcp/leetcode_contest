class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        cnt = collections.Counter(arr)
        most = cnt.most_common(len
(cnt))
        cur = 0
        ans = 0
        for i, c in most:
            cur += c
            ans += 1
            if cur >= len(arr) // 2:
                return ans
