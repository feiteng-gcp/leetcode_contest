class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        vs = list(sorted([(c, n) for n, c in collections.Counter(arr).items()], reverse=True))
        tot = len(arr)
        hf = len(arr) // 2
        
        ans = 0
        for c, n in vs:
            tot -= c
            ans += 1
            if tot <= hf:
                return ans
