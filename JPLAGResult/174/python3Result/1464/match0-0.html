class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        cnt = collections.Counter(arr)
        cnt = list(cnt.values())
        cnt.sort()
        tar = len(arr)//2
        cur = 0
        res = 0
        while cur < tar:
            cur += cnt[-1]
            cnt.pop()
            res += 1
        return res
