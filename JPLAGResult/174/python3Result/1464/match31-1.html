class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        
        counts = collections.Counter(arr)
        
        counts = sorted([(counts[x], x) for x in counts], key=lambda x: (-x[0], -x[1]))
        
        l 
= len(arr)
        
        res = 0
        tot = 0
        while tot < l/2:
            tot += counts[res][0]
            res += 1
            
        return res
