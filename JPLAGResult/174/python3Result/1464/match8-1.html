class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        d = collections.Counter(arr)
        sd = sorted(d.items(), key=lambda x: -x[1])
        n = len(arr)
        num = 0
        ans = 0
        for k, v in sd:
            num += v
            ans += 1
            if num >= n // 2:
                return ans
