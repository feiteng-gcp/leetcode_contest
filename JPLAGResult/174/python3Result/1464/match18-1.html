class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        length = len(arr)
        c = collections.Counter(arr)
        ans = 0
        vals 
= list(c.values())
        vals.sort(reverse = True)
        sum_ = 0
        for x in vals:
            sum_ += x
            ans += 1
            if sum_ >= length // 2:
                return ans
