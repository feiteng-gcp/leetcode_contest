class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        n = len(arr)
        count = collections.Counter(arr)
        count = list(count.items())
        count.sort(key=lambda x: x[1], reverse=True)
        res = 0
        s = 0
        for k, c in count:
            s += c
            res += 1
            if s >= n // 2:
                return res
        return s
