class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        counter = dict()
        for x in arr:
            if x not in counter: counter[x] = 1
            else: counter[x] += 1
        counts = []
        for x in counter:
            counts.append(counter[x])
        counts.sort(key=lambda miao: -miao)
        to_remove = len(arr) // 2
        num_chosen = 0
        removed = 0
        while removed < to_remove:
            removed += counts[num_chosen]
            num_chosen += 1
        return num_chosen
