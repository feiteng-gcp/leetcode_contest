class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        n = len(arr)
        mid = n // 2
        ctr = collections.Counter(arr)
        res = 0
        for num in sorted(ctr.keys(), key=lambda v:-ctr[v]):
            res += 1
            n -= ctr[num]
            if n <= mid:
                return res
            
        return res
