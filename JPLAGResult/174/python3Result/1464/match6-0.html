class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        cnts = collections.Counter(arr)
        cnts = sorted(cnts.items(), key=lambda x : x[1], reverse=True)
        ans = 0
        cnt = 0
        for v, c in cnts:
            cnt += c
            ans += 1
            if cnt * 2 >= len(arr):
                break
        return ans
