class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        counter = collections.Counter(arr)
        total, size = len(counter), len(arr)
        values = list(counter.values
())
        values.sort(reverse=True)
        i = 0
        cnt = 0
        for val in values:
            i += val
            cnt += 1
            if i >= (size + 1) // 2:
                break
        return cnt
