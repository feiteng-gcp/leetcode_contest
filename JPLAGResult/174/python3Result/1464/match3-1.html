class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        n = len(arr)
        d = collections.Counter(arr)
        d = sorted(d.items
(), key=lambda x: -x[1])
        res = 0
        for i in range(len(d)):
            res += d[i][1]
            if res >= n // 2:
                return i+1
