class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        n = len(arr)
        d = collections.Counter(arr)
        # print(d)
        d = d.items()
        # print(d)
        d = sorted(d, key=lambda x: -x[1])
        # print(d)
        ans = 0
        for i in range(len(d)):
            ans += d[i][1]
            if ans >= n // 2:
                return i+1
