class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        from collections import Counter
        threshold = len(arr) // 2
        assert len(arr) % 2 == 0
        counts = Counter(arr)
        ans = 0
        total = 0
        for k, v in sorted(counts.items(), key=lambda kv: kv[1], reverse=True):
            
            ans += 1
            total += v
            if total >= threshold:
                break
        return ans
