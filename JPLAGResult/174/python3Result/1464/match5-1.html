from collections import Counter
class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        cnt = sorted(Counter
(arr).items(), key=lambda x: -x[1])
        tot = 0
        res = 0
        target = len(arr) // 2
        for _, c in cnt:
            tot += c
            res += 1
            if tot >= target:
                return res
