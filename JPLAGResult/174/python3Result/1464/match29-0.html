class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        cnts = dict()
        for i in arr:
            if i not in cnts:
                cnts[i] = 1
            else:
                cnts[i] += 1
        c = []
        for i in cnts:
            c.append(cnts[i])
        c = list(reversed(sorted(c)))
        n, s = len(arr), 0
        for i in range(len(c)):
            s += c[i]
            if s * 2 >= n:
                return i + 1
