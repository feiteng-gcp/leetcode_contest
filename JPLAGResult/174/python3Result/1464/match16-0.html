from collections import Counter
class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        counter = Counter()
        for i in arr:
            counter[i] += 1
        num = list(counter.values())
        num.sort(reverse=True)
        n = len(arr)
        a = 0
        for i in range(n):
            a += num[i]
            if a >= n // 2:
                return i + 1
