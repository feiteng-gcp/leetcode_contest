from collections import Counter
class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        cnt = Counter(arr)
        res = 0
        qual = 0
        for _,i in cnt.most_common():
            #print(i)
            res+=1
            qual+=i
            if qual >= len(arr)/2:
                return res
