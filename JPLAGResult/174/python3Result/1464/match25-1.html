class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        import collections
        c = collections.Counter(arr)
        res = 0
        s = 0
        for num in sorted(list
(c.keys()),key=lambda x: c[x],reverse=True):
            s += c[num]
            res += 1
            if s >= len(arr) // 2:
                return res
        return res
