from collections import Counter
class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        
        c = Counter(arr)

        counts = [
count for _, count in c.items()]
        counts.sort(reverse=True)
        # print(counts)
        halfSize = len(arr)//2
        remaining = len(arr)
        num = 0
        for count in counts:
            remaining -= count
            num += 1
            if remaining <= halfSize:
                return num
