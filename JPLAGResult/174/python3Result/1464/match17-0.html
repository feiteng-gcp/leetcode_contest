class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        
        counter = collections.Counter(arr)
        target = len(arr) // 2
        h = list(counter.values())
        h.sort(reverse=True)
        res = 0
        for c in h:
            target -= c
            res += 1
            if target <= 0:
                return res
        return res
