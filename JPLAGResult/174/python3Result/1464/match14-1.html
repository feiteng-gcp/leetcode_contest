class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        counts = collections.Counter(arr)
        target = math.ceil(len(arr) / 2)
        ans 
= 0
        for key, count in sorted(counts.items(), key=lambda x: x[1], reverse=True):
            #print("key: ", key, " count: ", count)
            target -= count
            ans += 1
            if target <= 0:
                break
        return ans
