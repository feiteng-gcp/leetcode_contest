class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        from collections import Counter
        if not arr: return 0
        counter = Counter(arr)
        values = list(counter.values())
        values.sort(reverse = True)
        half = (len(arr) + 1) // 2
        cur_sum = 0
        for i, v in enumerate(values):
            cur_sum += v
            if cur_sum >= half:
                return i + 1
