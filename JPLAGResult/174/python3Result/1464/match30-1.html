import collections


class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        count  = collections.Counter(arr)
        #print(count)
        newCount = count.most_common()
        #print(newCount)
        
        need = len(arr)//2
        ans = 0
        cur = 0
        
        for x in range(len(newCount)):
            if cur+newCount[x][1] >= need:
                return ans+1
            else:
                ans += 1
                cur += newCount[x][1]
