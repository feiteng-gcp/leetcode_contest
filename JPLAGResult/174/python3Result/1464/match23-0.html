import collections


class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        n = len(arr)
        counter = collections.Counter(arr)
        counts = sorted(counter.values(), reverse=True)
        s = 0
        res = 0
        for x in counts:
            s += x
            res += 1
            if s >= (n + 1) // 2:
                break
        return res
