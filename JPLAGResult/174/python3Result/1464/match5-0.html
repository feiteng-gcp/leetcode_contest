class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        c = collections.Counter(arr)
        c = sorted(c.items(), key=lambda x: -x[1])
        count = 0
        i = 0
        n = len(arr)
        for k, v in c:
            count += v
            i+=1
            if count>=n//2:
                return i
