class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        cnt = collections.Counter(arr)
        n = len(arr)
        l = cnt.most_common()
        acc = 0
        res = 0
        for k, c in l:
            acc += c
            res += 1
            if acc * 2 >= n:
                return res
        return res
