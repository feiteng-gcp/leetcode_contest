from collections import Counter

class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        
        c = Counter(arr)
        x = sorted((c, k) for k, c in c.items())[::-1]
        
        res 
= 0
        cur = 0
        for count, _ in x:
            res += 1
            cur += count
            if cur >= len(arr)/2:
                return res
