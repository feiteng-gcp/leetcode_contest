class Solution {
    public int maxJumps(int[] arr, int d) {
        int[] dp = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            dfs(i, arr, d, dp);
        }
        int max = 1;
        for (int i : dp) {
            max = Math.max(max, i);
        }
        return max;
    }
    
    private int dfs(int current, int[] arr, int d, int[] dp) {
        if (dp[current] > 0) return dp[current];
        int result = 1;
        int i = 1;
        while (i <= d && current - i >= 0) {
            if (arr[current - i] >= arr[current]) break;
            result = Math.max(result, 1 + dfs(current - i, arr, d, dp));
            i++;
        }
        i = 1;
        while (i <= d && current + i < arr.length) {
            if (arr[current + i] >= arr[current]) break;
            result = Math.max(result, 1 + dfs(current + i, arr, d, dp));
            i++;
        }
        dp[current] = result;
        return result;
    }
}
