class Solution {
    int[] dp;
    public int maxJumps(int[] arr, int d) {
        int l = arr.length;
        dp = new int[l];
        int res = 1;
        for (int i = 0; i < l; i++) {
            dfs(i, arr, d);
            res = Math.max(res, dp[i]);
        }
        return res;
    }
    private int dfs(int cur, int[] arr, int d) {
        if (dp[cur] > 0) return dp[cur];
        int l = arr.length;
        int res = 0;
        
        for (int i = cur + 1; i <= Math.min(l - 1, cur + d); i++) {
            if (arr[i] >= arr[cur]) break;
            res = Math.max(res, dfs(i, arr, d));
        }
        for (int i = cur - 1; i >= Math.max(0, cur - d); i--) {
            if (arr[i] >= arr[cur]) break;
            res = Math.max(res, dfs(i, arr, d));
        }
        return dp[cur] = res + 1;
    }
}
