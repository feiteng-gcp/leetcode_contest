class Solution {
    private int[] maxJumpsArr;
    
    public int maxJumps(int[] arr, int d) {
        int n = arr.length;
        maxJumpsArr = new int[n];
        Arrays.fill(maxJumpsArr, -1);
        
        for (int i = 0; i < n; i++) {
            helper(arr, d, i);
        }
        
        int max = 0;
        for (int val: maxJumpsArr) {
            max = Math.max(max, val);
        }
        
        return max;
    }
    
    private int helper(int[] arr, int d, int index) {
        if (maxJumpsArr[index] != -1) {
            return maxJumpsArr[index];
        }
        // Jump from index
        int height = arr[index];
        int max = 1;
        for (int i = 1; i <= d; i++) {
            if (index + i >= arr.length || arr[index + i] >= arr[index]) {
                break;
            }
            max = Math.max(max, helper(arr, d, index + i) + 1);
        }
        
        for (int i = -1; i >= -d; i--) {
            if (index + i < 0 || arr[index + i] >= arr[index]) {
                break;
            }
            max = Math.max(max, helper(arr, d, index + i) + 1);
        }
        
        maxJumpsArr[index] = max;
        return max;
    }
}
