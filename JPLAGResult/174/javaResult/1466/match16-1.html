class Solution {
    int[] dp;
    public int maxJumps(int[] arr, int d) {
        int N = arr.length;
        dp = new int[N];
        int ans = 1;
        for(int i = 0; i < N; i++)
            ans = Math.max(ans, dfs(arr, d, i))
;
        return ans;
    }
    public int dfs(int[] arr, int d, int i) {
        if(dp[i] != 0)
            return dp[i];
        
int cur = 1;
        for(int j = i - 1; j >= Math.max(0, i - d); j--) {
            if(arr[j] >= arr[i])
                break;
            cur = Math.max(cur, dfs(arr, d, j) + 1);
        }
        for(int j = i + 1; j <= Math.min(arr.length - 1, i + d); j++) {
            if(arr[j] >= arr[i])
                break;
            cur = Math.max(cur, dfs(arr, d, j) + 1);
        }
        
        dp[i] = cur;
        return cur;
    }
}
