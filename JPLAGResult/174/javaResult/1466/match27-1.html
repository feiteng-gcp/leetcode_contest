class Solution {
    int max = 0;
    int[] dp;
    boolean[] viewed;
    public int maxJumps(int[] arr, int d) {
        dp = new int[arr.length];
        viewed = new boolean[arr.length];
        int res = 0;
        for(int i=0; i<arr.length; i++) {
            
res = Math.max(res, dfs(i, arr, d));
        }
        
        return res;
    }
    
    private int dfs(int cur, int[] arr, int d) {
        if(viewed[cur]) {
            return dp[cur];
        
}
        
        int max = 1;
        for(int i=cur-1; i>=Math.max(0, cur-d); i--) {
            if(arr[i]>=arr[cur]) break;
            max = Math.max(max, 1+dfs(i, arr, d));
        }
        
        for(int i=cur+1; i<=Math.min(arr.length-1, cur+d); i++) {
            if(arr[cur]<=arr[i]) break;
            max = Math.max(max, 1+dfs(i, arr, d));
        }
        
        
        viewed[cur] = true;
        dp[cur] = max;
        return max;
    }
    
    
}
