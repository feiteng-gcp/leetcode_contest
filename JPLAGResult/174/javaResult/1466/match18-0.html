class Solution {
    int[] record;
    public int maxJumps(int[] arr, int d) {
        int len = arr.length;
        record = new int[len];
        Arrays.fill(record, -1);
        int res = 0;
        for(int i = 0; i < len; i++)
            res = Math.max(res, helper(arr, d, i));
        return res;
    }
    
    public int helper(int[] nums, int d, int pos){
        if(record[pos] != -1)
            return record[pos];
        record[pos] = 0;
        int res = 1;
        int max = 0;
        for(int i = 1; i <= d && pos + i < nums.length; i++){
            max = Math.max(max, nums[pos + i]);
            if(max >= nums[pos])
                break;
            res = Math.max(res, 1 + helper(nums, d, pos + i));
        }
        max = 0;
        for(int i = 1; i <= d && pos - i > -1; i++){
            max = Math.max(max, nums[pos - i]);
            if(max >= nums[pos])
                break;
            res = Math.max(res, 1 + helper(nums, d, pos - i));
        }
        record[pos] = res;
        return res;
    }
}
