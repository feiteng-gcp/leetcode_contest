import java.util.*;

class Pair {
    int idx;
    int val;
    Pair(int idx, int val) {
        this.idx = idx;
        this.val = val;
    }
}
class Solution {
    int[] arr;
    int[] memo;
    int d;
    public int maxJumps(int[] arr, int d) {
        int n = arr.length;
        this.arr = arr;
        this.d = d;
        this.memo = new int[n];
        Arrays.fill(memo, -1);
        Pair[] order = new Pair[n];
        for (int i = 0; i < n; i ++) {
            order[i] = new Pair(i, arr[i]);
        }
        Arrays.sort(order, (a,b) -> a.val - b.val);
        int res = 0;
        for (Pair o: order) {
            res = Math.max(res, dp(o.idx));
        }
        return res;
    }

    private int dp(int idx) {
        if (memo[idx] != -1) return memo[idx];
        int val = arr[idx];
        int res = 1;
        for (int j = idx + 1; j <= idx + d && j < arr.length; j ++) {
            if (arr[j] >= val) break;
            res = Math.max(res, dp(j) + 1);
        }

        for (int j = idx - 1; j >= idx - d && j >= 0; j --) {
            if (arr[j] >= val) break;
            res = Math.max(res, dp(j) + 1);
        }
        memo[idx] = res;
        return res;
    }

}
