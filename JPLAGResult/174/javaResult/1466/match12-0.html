class Solution {
    // rank by height
    public int maxJumps(int[] arr, int d) {
        TreeSet<Integer> set = new TreeSet<>();
        for(int k : arr) set.add(k);
        int n = arr.length;
        int[] rank = new int[n];
        Map<Integer, Integer> map = new HashMap<>();
        int idx = 0;
        for(int k : set) map.put(k, idx++);
        for(int i = 0; i < n; i++)
        {
            int h = arr[i];
            rank[i] = map.get(h);            
        }
        
        // try to dfs
        int[] f = new int[n];
        Arrays.fill(f, -1);
        int ans = 1;
        for(int i = 0; i < n; i++)
        {
            // if(f[i] >= 0) continue;
            dfs(arr, i, rank, f, d);
            ans = Math.max(ans, f[i]);
        }
        // System.out.println(Arrays.toString(f));
        return ans;
    }
    
    int dfs(int[] ary, int cur, int[] rank, int[] f, int d)
    {
        if(f[cur] >= 0) return f[cur];
        
        // try to
        int n = ary.length, h = ary[cur];
        int ans = 1;    // at least self
        for(int i = cur - 1; i >= Math.max(0, cur - d); i--)
        {
            if(ary[i] >= h) break;
            if(ary[i] < h) ans = Math.max(ans, 1 + dfs(ary, i, rank, f, d));
        }
        for(int i = cur + 1; i <= Math.min(n - 1, cur + d); i++)
        {
            if(ary[i] >= h) break;
            if(ary[i] < h) ans = Math.max(ans, 1 + dfs(ary, i, rank, f, d));
        }
        return f[cur] = ans;
    }
}
