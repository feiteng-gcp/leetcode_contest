class Solution {
    int[] map;
    //boolean[] seen;
    public int maxJumps(int[] arr, int d) {
        int n = arr.length;
        int res =0;
        map= new int[n];
        //seen = new boolean[n];
        for (int i=0;i<n;i++){
            if (map[i]!=0) continue;
            res= Math.max(help(arr,i,d),res);
        }
        return res;

    }

    private int help(int[] arr, int i, int d) {
        if (map[i]!=0) return map[i];
        int res = 0;
        for (int j=i+1;j<=Math.min(i+d,arr.length-1);j++){
            if (arr[j]>=arr[i]) break;
            res = Math.max(res,help(arr,j,d));
        }
        for (int j=i-1;j>=Math.max(i-d,0);j--){
            if (arr[j]>=arr[i]) break;
            res = Math.max(res,help(arr,j,d));
        }
        map[i] = res+1;
        return res+1;

    }
}
