class Solution {
    public int maxJumps(int[] arr, int d) {
        int[] sol = new int[arr.length];
        for (int i = 0; i < sol.length; i++) sol[i] = -1;
        int ans = 0;
        for (int i = 0; i < arr.length; i++) {
            sol[i] = jump(arr, i, d, sol);
            
ans = Math.max(sol[i], ans);
        }
        return ans + 1;
    }
    
    public int jump(int[] arr, int pos, int d, int[] sol) {
        if (sol[pos] != -1) return sol[pos];
        int i = pos - 1;
        int ans = 0;
        while (i >= Math.max(0, pos - d) && arr[i] < arr[pos]) {
            ans = Math.max(ans, 1 + jump(arr, i, d, sol));
            i--;
        }
        i = pos + 1;
        while (i <= Math.min(arr.length - 1, pos + d) && arr[i] < arr[pos]) {
            
ans = Math.max(ans, 1 + jump(arr, i, d, sol));
            i++;
        }
        sol[pos] = ans;
        return ans;
    }
}
