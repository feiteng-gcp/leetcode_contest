class Solution {
    public int maxJumps(int[] arr, int d) {
        int n = arr.length;
        
Integer[] memo = new Integer[n];
        int max = 1;
        for(int i = 0 ; i < n ; i++){
            max = Math.max(max, dfs(arr, d, memo,i));
        }
        return max;
    }
    
    private int dfs(int[] arr, int d, Integer[] memo, int cur){
        if(memo[cur] != null){
            return memo[cur];
        }
        int n = arr.length;
        int res = 1;
        
for(int len = 1; len <= d; len++){
            if(cur - len < 0 || arr[cur - len] >= arr[cur]){
                break;
            }
            res = Math.max(res, 1 + dfs(arr, d, memo, cur - len));
        }
        for(int len = 1; len <= d; len++){
            if(cur + len >= n || arr[cur + len] >= arr[cur]){
                break;
            }
            res = Math.max(res, 1 + dfs(arr, d, memo, cur + len));
        }
        memo[cur] = res;
        return res;
    }
}
