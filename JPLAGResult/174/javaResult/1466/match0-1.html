class Solution {
    public int maxJumps(int[] arr, int d) {
        int N = arr.length;
        List<Integer> indexes = new ArrayList();
        for (int i = 0; i < N; i++) {
            indexes.add(i);
        }
        Collections.sort(indexes, (a, b) -> arr[a] - arr[b]);
        int[] dp = new int[N];
        for (int i : indexes) {
            dp[i] = search(arr, i, d, dp);
        }
        int max = 0;
        for (int m : dp) {
            max = Math.max(max, m);
        }
        return max;
    }
    
    private int search(int[] arr, int idx, int d, int[] dp) {
        if (dp[idx] > 0) {
            return dp[idx];
        }
        int max = 0;
        // left
        int hi = 0;
        for (int i = idx - 1; i >= idx - d && i >= 0; i--) {
            if (arr[i] >= arr[idx]) {
                break;
            }
            if (arr[i] > hi) {
                max = Math.max(max, dp[i]);
            } else {
                hi = arr[i];
            }
        }
        // right
        hi = 0;
        for (int i = idx + 1; i <= idx + d && i < arr.length; i++) {
            if (arr[i] >= arr[idx]) {
                break;
            }
            if (arr[i] > hi) {
                max = Math.max(max, dp[i]);
            } else {
                hi = arr[i];
            }
        }
        dp[idx] = 1 + max;
        return dp[idx];
    }
}
