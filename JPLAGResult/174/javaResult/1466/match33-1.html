class Solution {
    private int[] arr, mem;
    private int d;
    private int INF = 0X7FFFFFFF;
    public int maxJumps(int[] arr, int d) {
        this.arr = arr;
        this.d = d;
        int n = arr.length;
        this.mem = new int[n];
        Arrays.fill(mem, -1);
        
        
int ans = 0;
        for (int i = 0; i < n; i++) {
            ans = Math.max(DFS(i), ans);
        }
        return ans;
    }
    private int DFS(int cur) {
        
        if (mem[cur] != -1) return mem[cur];
        int ans = 1;
        
        int h = 0, tmp = 0;
        
        for (int x = 1; x <= d && cur-x >= 0; x++) {
            int k = arr[cur-x];
            
if (k >= arr[cur]) break;
            tmp = Math.max(tmp, DFS(cur-x));
        }
        for (int x = 1; x <= d && cur+x < arr.length; x++) {
            int k = arr[cur+x];
            
if (k >= arr[cur]) break;
            tmp = Math.max(tmp, DFS(cur+x));
        }
        ans += tmp;
        mem[cur] = ans;
        return ans;
    }
}
