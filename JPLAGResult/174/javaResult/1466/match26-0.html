class Solution {
    public int maxJumps(int[] arr, int d) {
        int n = arr.length; 
        Integer[] dp = new Integer[n]; 
        int ret = 0;
        for (int i = 0; i < n; i++) {
            if (dp[i] != null) {
                continue;
            }
            dfs(arr, i, dp, d);
        }
        for (int x : dp) {
             ret = Math.max(ret, x);
        }
        return ret; 
        
    }
    
    private int dfs(int[] arr, int idx, Integer[] dp, int d) {
        
        // System.out.println(idx);
        if (dp[idx] != null) {
            return dp[idx]; 
        }
        int ret = 1; 
        for (int i = idx - 1; i >= Math.max(0, idx - d); i--) {
            if (arr[i] >= arr[idx]) {
                break;
            }
            int tmp = dfs(arr, i, dp, d);
            ret = Math.max(1 + tmp, ret);
        }
        
        for (int i = idx + 1; i <= Math.min(arr.length - 1, idx + d); i++) {
           if (arr[i] >= arr[idx]) {
                break;
            }
            int tmp = dfs(arr, i, dp, d);
            ret = Math.max(1 + tmp, ret); 
        }
        dp[idx] = ret;
        return dp[idx]; 
    
    }
}
