class Solution {
    public int maxJumps(int[] arr, int d) {
        int n = arr.length;
        int[] steps = new int[n];
        int result = 0;
        
        for (int i = 0; i < n; i++) {
            result = Math.max(result, 
maxSteps(i, arr, steps, d));
        }
        
        return result;
    }
    
    public int maxSteps(int i, int[] arr, int[] steps, int d) {
        if (steps[i] > 0) {
            return steps[i];
        }
        
        int nextSteps = 0;
        
        for (int ni = i - 1; ni >= 0 && ni >= i - d && arr[ni] < arr[i]; ni--) {
            int s = maxSteps(ni, arr, steps, d);
            nextSteps = Math.max(nextSteps, s);
        }
        
        for (int ni = i + 1; ni < arr.length && ni <= i + d && arr[ni] < arr[i]; ni++) {
            
int s = maxSteps(ni, arr, steps, d);
            nextSteps = Math.max(nextSteps, s);
        }
        
        steps[i] = nextSteps + 1;
        
        return nextSteps + 1;
    }
}
