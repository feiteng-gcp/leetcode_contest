class Solution {
    
    Integer[] M;
    
    public int maxJumps(int[] arr, int d) {
        M = new Integer[arr.length];
        int max = 0;
        for (int i = 0; i < arr.length; i++) {
            max = Math.max(max, dp(arr, i, d));
        }
        
        return max;
    }
    
    public int dp(int[] arr, int i, int d) {
        if (i < 0 || i >= arr.length) return 0;
        if (M[i] != null) return M[i];
        
        int max = 1;
        for (int j = i - 1; j >= 0 && j >= i - d && arr[j] < arr[i]; j--) {
            max = Math.max(max, 1 + dp(arr, j, d));
        }
        
        for (int j = i + 1; j < arr.length && j <= i + d && arr[j] < arr[i]; j++) {
            max = Math.max(max, 1 + dp(arr, j, d));
        }
        
        M[i] = max;
        return max;
    }
}
