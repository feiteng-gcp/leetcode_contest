class Solution {
    public int minSetSize(int[] a) {
        int n = a.length;
        Map<Integer, Integer> m = new HashMap<>();
        for (int ai : a) {
            m.put(ai, m.getOrDefault(ai, 0) + 1);
        }
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        for (int c : m.values()) {
            pq.offer(c);
        }
        int red = 0;
        int r = 0;
        while (red < n / 2) {
            red += pq.poll();
            r++;
        }
        return r;
    }
}
