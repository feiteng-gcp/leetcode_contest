class Solution {
    public int minSetSize(int[] arr) {
        int[] f = new int[100010];
        for(int k : arr) f[k]++;
        // count inversely
        Queue<Integer> q = new PriorityQueue<>(Collections.reverseOrder());
        for(int i = 1; i <= 100000; i++)
        {
            if(f[i] > 0) q.offer(f[i]);
        }
        int n = arr.length, n2 = n / 2;
        int ans = 0;
        while(n > n2)
        {
            n -= q.poll();
            ans++;
        }
        return ans;
    }
}
