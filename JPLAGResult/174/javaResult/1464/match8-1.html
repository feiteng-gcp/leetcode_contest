class Solution {
    public int minSetSize(int[] arr) {
        Map<Integer, Integer> countByElement = new HashMap<>();
        for (int i : arr) {
            countByElement.put(i, 1 + countByElement.getOrDefault(i, 0));
        }
        List<Integer> count = new ArrayList<>(countByElement.values());
        count.sort(Comparator.reverseOrder());
        int removedTotal = 0;
        int removedUnique = 0;
        for (int i : count) {
            removedTotal += i;
            removedUnique += 1;
            if (removedTotal >= arr.length / 2) {
                break;
            }
        }
        return removedUnique;
    }
}
