import java.util.*;

class Solution {
    public int minSetSize(int[] arr) {
        if (arr.length == 0) {
            return 0;
        }
        Map<Integer, Integer> counter = new HashMap<>();
        for (int i : arr) {
            counter.put(i, counter.getOrDefault(i, 0) + 1);
        }

        List<Integer> values = new ArrayList<>(counter.size());
        values.addAll(counter.keySet());

        values.sort(Comparator.comparingInt(counter::get).reversed());
        int cnt = 0;
        for (int i = 0; i < values.size(); i++) {
            cnt += counter.get(values.get(i));
            if ((cnt << 1) >= arr.length) {
                return i + 1;
            }
        }
        return -1;
    }
}
