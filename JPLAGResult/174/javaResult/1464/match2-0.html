class Solution {
    public int minSetSize(int[] arr) {
        int n = arr.length;
        int[] cnt = new int[100001];
        for(int num: arr){
            cnt[num]++;
        }
        int res = 0;
        n >>= 1;
        Arrays.sort(cnt);
        for(int i = cnt.length - 1; i >= 1; i--){
            if(cnt[i] > 0){
                res++;
                if(cnt[i] >= n){
                    break;
                } else {
                    n -= cnt[i];
                }
            }
        }
        return res;
    }
}
