class Solution {
    public int minSetSize(int[] arr) {
        Map<Integer, Integer> m = new HashMap<>();
        for (int i : arr) m.put(i, m.getOrDefault(i, 0) + 1);
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        for (int e : m.values()) {
            pq.offer(e);
        }
        int cnt = 0, res = 0;
        while (cnt < (arr.length + 1) / 2) {
            cnt += pq.poll();
            res++;
        }
        return res;
    }
}
