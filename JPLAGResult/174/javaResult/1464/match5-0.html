class Solution {
    public int minSetSize(int[] arr) {
       Map<Integer, Integer> map = new HashMap<>();
        
        for (int v : arr) {
            map.merge(v, 1, Integer::sum);
        }
        
        List<Integer> list = new ArrayList<>(map.values());
        list.sort(Collections.reverseOrder());
        
        int n = arr.length;
        int res = 0;
        for (int v : list) {
            n -= v;
            res++;
            if (n <= arr.length / 2) {
                break;
            }
        }
        
        return res;
    }
}
