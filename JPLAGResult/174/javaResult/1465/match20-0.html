/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long max = 0;
    List<Integer> sums = new ArrayList<>();
    public int maxProduct(TreeNode root) {
        int total = help(root);
        for (int i : sums) {
            max = Math.max(max, (long)i * (total - i));
        }
        return (int) (max % 1000000007);
    }
    int help(TreeNode c) {
        if (c == null) return 0;
        int left = help(c.left);
        int right = help(c.right);
        int sum = c.val + left + right;
        sums.add(sum);
        return sum;
    }
}
