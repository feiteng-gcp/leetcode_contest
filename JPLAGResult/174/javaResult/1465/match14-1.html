/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode(int x) { val = x; }
 * }
 */
class Solution {

    private long minDiff = Integer.MAX_VALUE;
    private long sum;


    public int maxProduct(TreeNode root) {
        sum = sum(root);

        findMinDiff(root);

        int mod = 1_000_000_007;

        long subSum1 = (sum - minDiff) / 2;
        long subSum2 = subSum1 + minDiff;
        return (int) (subSum1 * subSum2 % mod);

    }

    private long findMinDiff(
TreeNode node) {

        if (node == null) {
            return 0;
        }

        long leftSubSum = findMinDiff(node.left);
        long rightSubSum = findMinDiff(node.right);

        minDiff = Math.min(minDiff, Math.abs(sum - leftSubSum - leftSubSum));
        minDiff = Math.min(minDiff, 
Math.abs(sum - rightSubSum - rightSubSum));

        return node.val + leftSubSum + rightSubSum;

    }

    private long sum(TreeNode node) {
        if (node == null) {
            return 0;
        }
        return (long) node.val + sum(node.left) + sum(node.right);
    }
}
