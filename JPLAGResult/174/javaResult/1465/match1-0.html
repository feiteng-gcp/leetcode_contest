/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    private long res = 0;
    private static final int MOD = (int)1e9 + 7;
    
    public int maxProduct(TreeNode root) {
        res = 0;
        long sum = dfs(root);
        getMaxMul(sum, root);
        return (int)(res % MOD);
    }
    
    private long dfs(TreeNode node){
        if(node == null){
            return 0;
        }
        return dfs(node.left) + dfs(node.right) + node.val;
    }
    
    private long getMaxMul(long sum, TreeNode node){
        if(node == null){
            return 0;
        }
        long curSum = getMaxMul(sum, node.left) + getMaxMul(sum, node.right) + node.val;
        res = Math.max(res, curSum * (sum - curSum));
        return curSum;
    }
}
