/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private long max;
    public int maxProduct(TreeNode root) {
        max = 0;
        long sums = getSum(root);
        getMax(root, sums);
        return (int) (max % 1_000_000_007L);
    }
    
    private long getSum(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return root.val + getSum(root.left) + getSum(root.right);
    }
    
    private long getMax(TreeNode root, long sums) {
        if (root == null) {
            return 0;
        }
        long sumOfTree = root.val + getMax(root.left, sums) + getMax(root.right, sums);
        max = Math.max(max, sumOfTree * (sums - sumOfTree));
        return sumOfTree;
    }
}
