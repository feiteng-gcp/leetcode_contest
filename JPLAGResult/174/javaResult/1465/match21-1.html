/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    List<Long> list;
    public int maxProduct(TreeNode root) {
        list = new ArrayList<>();
        long sum = helper(root);
        long max = 0;
        for(long n : list)
            max = Math.max(max, n * (sum - n))
;
        return (int)(max % 1000000007);
    }
    
    public long helper(TreeNode root){
        if(root == null)
            return 0;
        long res = root.val + helper(root.left) + helper(root.right);
        list.add(res);
        return res;
    }
}
