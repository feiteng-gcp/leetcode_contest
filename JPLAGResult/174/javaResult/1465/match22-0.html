/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long sum = 0;
    long res;
    long mod = 1000000007;
    public int maxProduct(TreeNode root) {
        sum = getSum(root);
        res = -1;
        getProd(root);
        
        return (int)(res % mod);
    }
    private int getSum(TreeNode root) {
        if (root == null) return 0;
        return root.val + getSum(root.left) + getSum(root.right);
    }
    private int getProd(TreeNode root) {
        if (root == null) return 0;
        int l = getProd(root.left);
        int r = getProd(root.right);
        long pr = (long)l * (sum - l);
        if (pr > res) res = pr;
        pr = (long)r * (sum - r);
        if (pr > res) res = pr;
        return l + r + root.val;
    }
}
