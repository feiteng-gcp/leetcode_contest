/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public int maxProduct(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        int sum = sumUp(root, list);
        long prod = 0;
        for (Integer ele : list) 
{
            prod = Math.max(prod, (long)ele * (sum - ele));
        }
        return (int)(prod % 1000000007);
    }
    
    public int sumUp(TreeNode root, List<Integer> list) {
        if (root == null) return 0
;
        int sum = root.val;
        sum += sumUp(root.left, list);
        sum += sumUp(root.right, list);
        list.add(sum);
        return sum;
    }
}
