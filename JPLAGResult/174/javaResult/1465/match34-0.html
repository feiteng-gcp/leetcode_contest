class Solution {
    
    Map<TreeNode, Long> sums = new HashMap<>();
    
    long res = Long.MIN_VALUE;
    long mod = 1000000000 + 7;
    long total = 0;
    
    public int maxProduct(TreeNode root) {
        total = postOrderSum(root);
        dfsMax(root);
        return (int)(res % mod);
    }

    private long postOrderSum(TreeNode root) {
        if (root == null) return 0;
        long l = postOrderSum(root.left);
        long r = postOrderSum(root.right);
        long res = (l + r + root.val) % mod;
        sums.put(root, res);
        return res;
    }

    private void dfsMax(TreeNode root) {
        if (root == null) return;
        long l = 0;
        if (root.left != null) l = sums.get(root.left);
        res = Math.max(res, (l * (total - l)));
        
        long r = 0;
        if (root.right != null) r = sums.get(root.right);
        res = Math.max(res, (r * (total - r)));
        
        dfsMax(root.left);
        dfsMax(root.right);
    }
}
