/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int mod = (int) (1e9 + 7);
    TreeSet<Long> set;
    
    public int maxProduct(TreeNode root) {
        //want the sum of left and sum of right closer
        //get sum of left and right first.
//         leftSum = getSum(root.left);
//         rightSum = getSum(root.right);
//         if (leftSum == rightSum) {
//             return (leftSum * (rightSum + root.val)) % mod;
//         }
        
//         res = 0;
//         if (leftSum < rightSum) {
//             res = (leftSum + root.val) * rightSum;
//             helper(root.right, leftSum + root.val, rightSum);
//         }
//         else {
//             res = leftSum * (rightSum + root.val);
//             helper(root.left, leftSum, rightSum  + root.val);
//         }
//         return res;
        set = new TreeSet<>();
        long sum = getSum(root);
        Long upper = set.floor(sum / 2);
        Long lower = set.ceiling(sum / 2);
        if (upper == null) {
            //System.out.println("lower = " + lower);
            return (int) (lower * (sum - lower) % mod);
        }
        if (lower == null) {
            //System.out.println("upper = " + lower);
            return (int) (upper * (sum - upper) % mod);
        }
        return (int) (Math.max((lower * (sum - lower)), (upper * (sum - upper))) % mod);
    }
    
    private long getSum(TreeNode root) {
        if (root == null) {
            return 0;
        }
        
        long left = getSum(root.left);
        long right = getSum(root.right);
        set.add(left + right + root.val);
        return left + right + root.val;
    }
}
