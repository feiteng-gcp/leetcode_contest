/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long sum =0;
    
long res = 0;
    int mod = (int)1e9+7;
    public int maxProduct(TreeNode root) {
        helpsum(root);
        help(root);
        return (int)(res%mod);
    }

    private long help(TreeNode root) {
        if (root==null) return 0;
        long left = help(root.left);
        long right = help(root.right);
        //System.out.println(left+" "+right);
        res = Math.max(left*(sum-left),res);
        res = 
Math.max(right*(sum-right),res);
        return left+right+root.val;
    }

    private void helpsum (TreeNode root){
        if (root==null) return;
        sum+= root.val;
        helpsum(root.left);
        helpsum(root.right);
    }
}
