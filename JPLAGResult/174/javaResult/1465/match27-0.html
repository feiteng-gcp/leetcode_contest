/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Integer> map = new HashMap<>();
    int sum = 0;
    long max;
    long ans = 0;
    int diff = Integer.MAX_VALUE;
    int mod = 1000000007;
    public int maxProduct(TreeNode root) {
        sum = travel1(root);
        travel2(root);
        return (int)(ans%mod);
    }
    
    private void travel2(TreeNode node) {
        if (node == null)
            return;
        
        int a = map.get(node);
        int b = sum - a;
        // if ((Math.abs(sum/2) - a) < diff) {
        //     diff = Math.abs(sum/2) - a;
        //     ans = (a * b);
        //     ans %= mod;
        // }
        long curr = (long)a * (long)b;
        if (curr > ans) {
            ans = curr;
        }
        
        travel2(node.left);
        travel2(node.right);
    }
    
    private int travel1(TreeNode node) {
        if (node == null)
            return 0;
        
        int left = travel1(node.left);
        int right = travel1(node.right);
        
        int curr = left + right + node.val;
        map.put(node, curr);
        
        return curr;
    }
}
