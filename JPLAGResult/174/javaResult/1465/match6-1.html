/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public int maxProduct(TreeNode root) {
        Set<Integer> set = new HashSet<>();
        int total = sum(root, set);
        long max = -1;
        for(int v:set){
            int left = total - v;
            long pro = (long)v * (long)left;
            
max = Math.max(max, pro);
        }
        return (int)(max % 1000000007);
    }

    private int sum(TreeNode root, Set<Integer> set){
        if(root == null) return 0;
        int v = root.val + sum(root.left, set) + sum(root.right, set);
        set.add(v);
        return v;
    }
}
