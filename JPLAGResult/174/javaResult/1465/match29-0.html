/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long sum;
    long ret;
    long M = 1000000007L;
    
    public long dfs(TreeNode p) {
        if(p == null) return 0L;
        long l = dfs(p.left);
        long r = dfs(p.right);
        if(l * (sum - l) > ret) ret = l * (sum - l);
        if(r * (sum - r) > ret) ret = r * (sum - r);
        return l + r + (long)p.val;
    }
    
    public long getSum(TreeNode p) {
        if(p == null) return 0L;
        return (long)p.val + getSum(p.left) + getSum(p.right);
    }
    
    public int maxProduct(TreeNode root) {
        ret = 0;
        sum = getSum(root);
        dfs(root);
        return (int)(ret % M);
    }
}
