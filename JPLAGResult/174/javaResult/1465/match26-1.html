/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long ans = 0;
    public int maxProduct(TreeNode root) {
        ans = 0;
        
long sum = dfs1(root);
        dfs2(root, sum);
        return (int)(ans % 1000000007);
    }
    
    public long dfs1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return dfs1(root.left) + dfs1(root.right) + root.val;
    }
    
    public long dfs2(TreeNode root, long sum) {
        if (root == null) {
            return 0;
        }
        long left = dfs2(root.left, sum);
        ans = Math.max(ans, (left * (sum - left)));
        long right = dfs2(root.right, sum);
        ans = Math.max(ans,  (right * (sum - right)));
        return (left + right + root.val);
    }
}
