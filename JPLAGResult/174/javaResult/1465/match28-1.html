/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
long sum = 0, min = 0, mod = 1000000007;
    public int maxProduct(TreeNode root) {
        dfs(root);
        min = sum;
        dfs2(root);
        //System.out.println(sum + " " + min);
        
return (int) ((sum - min) / 2 * (sum + min) / 2 % mod);
    }
    void dfs(TreeNode node)
    {
        if(node == null) return;
        sum += node.val;
        dfs(node.left);
        dfs(node.right);
    
}
    long dfs2(TreeNode node)
    {
        if(node == null) return 0;
        long tmp = dfs2(node.left) + dfs2(node.right) + node.val, tmp2 = Math.abs(2 * tmp - sum);
        if(tmp2 < min)
            min = tmp2;
        return tmp;
    }
}
