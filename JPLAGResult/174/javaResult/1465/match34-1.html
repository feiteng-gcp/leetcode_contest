/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
private long sum = 0;
    private long ans = 0;
    public int maxProduct(TreeNode root) {
        sum = getSum(root);
        helper(root);
        return (int)(ans%1_000_000_007);
    }
    private long helper(TreeNode root) {
        if (root == null) return 0;
        long l = helper(root.left);
        long r = helper(root.right);
        long res1 = l * (sum-l);
        long res2 = r * (sum-r);
        if (res1 > ans) ans = res1;
        if (res2 > ans) ans = res2;
        
return l + r + root.val;
    }
    private long getSum(TreeNode root) {
        if (root == null) return 0;
        return getSum(root.left)+root.val+getSum(root.right);
    }
}
