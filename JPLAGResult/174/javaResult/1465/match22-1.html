class Solution {
    
long max, total;

    public int maxProduct(TreeNode root) {
        total = sum(root);
        max = Integer.MIN_VALUE;
        sumPlus(root);
        return (int) (max % 1_000_000_007);
    }

    private long sum(TreeNode root) {
        if (root == null) return 0L;
        return root.val + sum(root.left) + sum(root.right);
    }

    private long sumPlus(TreeNode root) {
        if (root == null) return 0L;
        long tmp = root.val + sumPlus(root.left) + sumPlus(root.right);
        max = Math.max(max, (total - tmp) * tmp);
        return tmp;
    }
}
