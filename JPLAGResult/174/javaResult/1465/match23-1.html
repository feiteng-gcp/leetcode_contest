/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Long> map = new HashMap<>();
    public int maxProduct(TreeNode root) {
        long MOD = 1_000_000_007;
        
long sum = dfs(root);
        long mid = 0;
        for(TreeNode node: map.keySet()) {
            long t = map.get(node);
            if(Math.abs(t - sum / 2) < Math.abs(mid - sum / 2))
                mid = t;
        
}
        return (int)((mid * (sum - mid)) % MOD);
    }
    public long dfs(TreeNode node) {
        if(node == null)
            return 0;
        long left = dfs(node.left);
        long right = dfs(node.right);
        long cur = left + right + node.val;
        map.put(node, cur);
        return cur;
    }
}
