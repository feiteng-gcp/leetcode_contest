/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {

	private long MOD = 1000_000_007;
	private long ans = 0;
	
    public int maxProduct(TreeNode root) {
    	ans = 0;
    	long sum = sumTree(root);
    	solve(root , sum);
    	return (int) (ans % MOD);
    }
    
    private long sumTree(TreeNode root) {    		
    	if (root == null) {
    		return 0;
    	} else {
    		return root.val + sumTree(root.left) + sumTree(root.right);    		
    	}
    }
    
    private long solve(TreeNode root , long sum) {
    	if (root == null) {
    		return 0;
    	} else {
    		long result = root.val;
    		if (root.left != null) {
    			long left = solve(root.left , sum);
    			long remain = sum - left;
    			ans = Math.max(ans , left * remain);
    			result += left;    			
    		}
    		if (root.right != null) {
    			long right = solve(root.right , sum);
    			long remain = sum - right;
    			ans = Math.max(ans , right * remain);
    			result += right;
    		}
    		return result;    		
    	}	    	
    }
}
