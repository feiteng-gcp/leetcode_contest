/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    long total = 0;
    
    long max = 0;
    public int maxProduct(TreeNode root) {
        total = sum(root);
        helper(root);
        return (int)(max % 1000000007);
    }
    
    private long sum(TreeNode node) {
        if (node == null) {
            return 0;
        }
        
        long cur = node.val;
        cur += sum(node.left);
        cur += sum(node.right);
        
        return cur;
    }
    
    private long helper(TreeNode node) {
        if (node == null) {
            return 0;
        }
        
        if (node.left == null && node.right == null) {
            return (long) node.val;
        }
        
        long left = helper(node.left);
        
        if (node.left != null) {
            max = Math.max(max, (left % 1000000007) * ((total - left) % 1000000007));
        }
        
        long right = 
helper(node.right);
        
        if (node.right != null) {
            max = Math.max(max, (right % 1000000007) * ((total - right) % 1000000007));
        }
        
        return left + right + node.val;
    }
    
}
