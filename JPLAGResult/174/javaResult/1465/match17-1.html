/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    
    public int maxProduct(TreeNode root) {
        int mod = (int) (1e9 + 7);
        TreeSet<Long> set = 
new TreeSet<>();
        long sum = getSum(root, set);
        Long upper = set.floor(sum / 2);
        Long lower = set.ceiling(sum / 2);
        if (upper == null) {
            return (int) (lower * (sum - lower) % mod);
        }
        if (lower == null) {
            return (int) (upper * (sum - upper) % mod);
        }
        return (int) (Math.max((lower * (sum - lower)), (upper * (sum - upper))) % mod);
    }
    
    private long getSum(TreeNode root, TreeSet set) 
{
        if (root == null) {
            return 0;
        }
        
        long left = getSum(root.left, set);
        long right = getSum(root.right, set);
        set.add(left + right + root.val);
        return left + right + root.val;
    }
}
