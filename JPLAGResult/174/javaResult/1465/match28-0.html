/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */


class Solution {
    int total = 0;
    int inf = (int)1e9;
    int min_diff = inf;
    int best = 0;
    int half = 0;
    int mod = (int)1e9+7;
    public int maxProduct(TreeNode root) {
        dfs_sum(root);

        half = total / 2;

        dfs(root);

        long a = best % mod;
        long b = (total-best) % mod;
        long t = (a*b) % mod;
        return (int)t;
    }

    public void dfs_sum(TreeNode root) {
        if (root == null) return;
        total += root.val;
        dfs_sum(root.left);
        dfs_sum(root.right);
        return;
    }

    public int dfs(TreeNode root) {
        if (root == null) return 0;

        int sum = root.val;
        sum += dfs(root.left);
        sum += dfs(root.right);


        if (sum != 0 && sum != total && Math.abs(half - sum) < min_diff) {
            best = sum;
            min_diff = Math.abs(half - sum);
        }

        return sum;
    }
}
