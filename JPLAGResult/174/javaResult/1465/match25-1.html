/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    class Data {
        long sum;
        long max;
        
        public Data(long s, long m){
            sum = s;
            max = m;
        }
    }
    
    public int maxProduct(TreeNode root) {
        long sum = sum(root);
        long ans = Math.max(
dfs(root.left, sum).max, dfs(root.right, sum).max);
        return (int)(ans % 1000000007);
    }
    
    long sum(TreeNode curr) {
        if (curr == null){
            return 0;
        }
        return sum(curr.left) + sum(curr.right) + curr.val;
    }
    
    Data dfs(TreeNode curr, long sum) {
        if (curr == null) {
            return new Data(0, 0);
        
}
        Data left = dfs(curr.left, sum);
        Data right = dfs(curr.right, sum);
        long s = left.sum + right.sum + curr.val;
        long max = 0;
        // max = Math.max(max, right.sum * (left.sum + curr.val));
        // max = Math.max(max, left.sum * (right.sum + curr.val));
        max = Math.max(max, s * (sum - s));
        max = Math.max(max, left.max);
        max = Math.max(max, right.max);
        return new Data(s, max);
    }
}
