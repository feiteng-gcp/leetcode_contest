/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Long> map = new HashMap<>();
    int mod = 1_000_000_007;
    
    public int maxProduct(TreeNode root) {
        // get all sum
        long sum = sumAll(root);
        long ans = dfs(root, sum) % mod;
        return (int) ans;
    }
    
    long dfs(TreeNode root, long sum)
    {
        if(root == null) return 0;
        long ans = 0;
        long s1 = map.getOrDefault(root.left, 0L),
            s2 = sum - s1;
        ans = Math.max(ans, s1 * s2);
        ans = Math.max(ans, dfs(root.left, sum));
        s1 = map.getOrDefault(root.right, 0L);
        s2 = sum - s1;
        ans = Math.max(ans, s1 * s2);
        ans = Math.max(ans, dfs(root.right, sum));
        return ans;
    }
    
    long sumAll(TreeNode root)
    {
        if(root == null) return 0;
        long ans = 0L + root.val + sumAll(root.left) + sumAll(root.right);
        map.put(root, ans);
        return ans;
    }
}
