	class Solution {
		List<Integer> sums = new ArrayList<>();
	    public int maxProduct(TreeNode root) {
	        int all = dfs(root);
	        long ret = 0;
	        int mod = 1000000007;
	        for(int v : sums){
	        	ret = Math.max(ret, (long)v * (all-v));
	        }
	        return (int)(ret%mod);
	    }
	    
	    int dfs(TreeNode root)
	    {
	    	if(root == null)return 0;
	    	int s = root.val + dfs(root.left) + dfs(root.right);
	    	sums.add(s);
	    	return s;
	    }
	}
