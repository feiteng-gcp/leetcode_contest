/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
long sum = 0;
    long max = 0L;
    public int maxProduct(TreeNode root) {
        sum = helper1(root);
        helper2(root);
        return (int)(max%(1_000_000_007));
    }
    
    private long helper1(TreeNode t) {
        if(t == null) return 0;
        return t.val + 
helper1(t.left)+helper1(t.right);
    }
    
    private long helper2(TreeNode t) {
        if(t == null) return 0L;
        long left = helper2(t.left);
        long right = helper2(t.right);
        max = Math.max(Math.max((sum-left)*left, (sum-right)*right), max);
        return t.val+left+right;
    }
}
