class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length; 
        
int[][] cnts = new int[n][2];
        for (int i = 0; i < mat.length; i++) {
            int cnt = 0;
            for (int j = 0; j < mat[i].length; j++) {
                if (mat[i][j] == 1) {
                    cnt++; 
                }
            }
            cnts[i] = new int[]{cnt, i}; 
        
}
        Arrays.sort(cnts, (a, b) -> {
            
            if (a[0] == b[0]) {
                return a[1] - b[1];
            }
            return a[0] - b[0];
        });
        int[] ret = new int[k];
        
for (int i = 0; i < k; i++) {
            ret[i] = cnts[i][1]; 
        }
        return ret;
    }
}
