class Solution {
    class Row {
        int index, count;
        Row(int count, int index) {
            this.count = count;
            this.index = index;
        }
    }
    
public int[] kWeakestRows(int[][] mat, int k) {
        List<Row> list = new ArrayList<>();
        for (int i = 0; i < mat.length; i++) {
            int[] m = mat[i];
            
int count = 0; 
            for (int a : m) {
                if (a == 1) count++;
                else break;
            }
            list.add(new Row(count, i));
        }
        Collections.sort(list, (a,b) -> a.count==b.count?a.index-b.index:a.count-b.count);
        
int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = list.get(i).index;
        }
        return res;
    }
}
