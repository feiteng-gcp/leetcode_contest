class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int[] ans = new int[k];
        
int m = mat.length, n = mat[0].length;
        int[][] cnt = new int[m][2];
        for(int i = 0; i < m; i++) {
            cnt[i][1] = i;
            for(int j = 0; j < n; j++) {
                cnt[i][0] += mat[i][j];
            }
        }
        Arrays.sort(cnt, (a, b) -> {
            if(a[0] != b[0]) {
                return a[0] - b[0];
            }
            return a[1] - b[1];
        });
        
for(int i = 0; i < k; i++) {
            ans[i] = cnt[i][1];
        }
        return ans;
    }
}
