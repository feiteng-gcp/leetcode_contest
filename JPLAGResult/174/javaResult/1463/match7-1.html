class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        
        int[] v = new int[m];
        for(int i = 0; i < m; i++){
            int count = 0;
            for(int j = 0; j < n; j++){
                if (mat[i][j] == 0) break;
                count++;
            }
            v[i] = count * 1000 + i;
        
}
        
        Arrays.sort(v);
        int[] res = new int[k];
        for(int i = 0; i < k; i++){
            res[i] = v[i] % 1000;
        }
        return res;
    }
}
