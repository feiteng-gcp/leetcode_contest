class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(new Comparator<Integer>() {
            public int compare(Integer i1, Integer i2) {
                int sum1 = 0, sum2 = 0;
                for (int i : mat[i1]) {
                    sum1 += i;
                }
                for (int i : mat[i2]) {
                    sum2 += i;
                }
                if (sum1 == sum2) {
                    return i1 - i2;
                }
                return sum1 - sum2;
            }
        });
        
        for (int i = 0; i < mat.length; i++) {
            pq.offer(i);
        }
        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = pq.poll();
        }
        return res;
    }
}
