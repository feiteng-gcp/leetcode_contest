class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int[] res = new int[k];
        
int m = mat.length, n = mat[0].length;
        int[][] arr = new int[m][2];
        for(int i=0;i<m;i++){
            arr[i][1] = i;
            for(int j=0;j<n;j++){
                arr[i][0]+=mat[i][j];
            }
        }
        Arrays.sort(arr, (a,b)->a[0]==b[0]?a[1]-b[1]:a[0]-b[0]);
        
for(int i=0;i<k;i++){
            res[i] = arr[i][1];
        }
        return res;
    }
}
