class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        int[] sdr = new int[m];
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++)
                sdr[i] += mat[i][j];
        PriorityQueue<Integer> queue = new PriorityQueue<>(
            (i, j) -> sdr[i] == sdr[j] ? (i - j) : (sdr[i] - sdr[j])
        );
        for (int i = 0; i < m; i++) queue.offer(i);
        int[] res = new int[k];
        for (int i = 0; i < k; i++) res[i] = queue.poll();
        return res;
    }
}
