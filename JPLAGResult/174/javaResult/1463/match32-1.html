class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        int a[][] = new int[m][2];
        for(int i=0; i < m; i++) {
            a[i][0] = i;
            a[i][1] = 0;
            for(int j=0; j < n; j++) {
                if(mat[i][j] == 1) a[i][1]++;
                else break;
            }
        }
        Arrays.sort(a, new Comparator<int[]>(){
           public int compare(int[]x, int y[]) {
               if(x[1] == y[1]) return Integer.compare(x[0], y[0]);
               return Integer.compare(x[1], y[1]);
           } 
        });
        
int ans[] = new int[k];
        for(int i=0; i < k; i++) {
            ans[i] = a[i][0];
        }
        return ans;
    }
}
