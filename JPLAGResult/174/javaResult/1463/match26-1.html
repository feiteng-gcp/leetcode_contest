class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
		List<Line> list = new ArrayList<>();

		int row = mat.length, col = mat[0].length;
		for (int i=0; i<row; i++) {
			int cur_cnt = 0;
			for (int j=0; j<col; j++) {
				if (mat[i][j] == 0) break;
				cur_cnt ++;
			}

			list.add(new Line(i, cur_cnt));
		}

		Collections.sort(list);

		int[] res = new int[k];
		for (int i=0; i<k; i++) {

			res[i] = list.get(i).r;
		}

		return res;
    }
}

class Line implements Comparable<Line>{
	public int r, cnt;
	public Line(int r, int cnt) {
		this.r = r;
		this.cnt = cnt;
	}
	public int compareTo(Line l)
 {
		if (this.cnt == l.cnt) {
			return this.r - l.r;
		} else {
			return this.cnt - l.cnt;
		}
	}
}
