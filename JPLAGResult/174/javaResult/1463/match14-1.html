class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int[][] cnt = new int[n][2];
        for(int i = 0; i < n; ++i) {
            cnt[i][0] = i;
            int[] x = mat[i];
            for(int t : x) cnt[i][1] += t
;
        }
        
        Arrays.sort(cnt, (a, b) -> {
            if(a[1] != b[1]) return a[1] - b[1];
            return a[0] - b[0];
        });
        System.out.println(Arrays.deepToString(cnt));
        
int[] res = new int[k];
        for(int i = 0; i < k; ++i) res[i] = cnt[i][0];
        return res;
    }
}
