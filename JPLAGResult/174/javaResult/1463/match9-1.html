class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int row = mat.length;
        int col = mat[0].length;
        int[][] count = new int[row][2];
        for (int i = 0; i < row; i++) {
            count[i] = new int[] {i, 0};
            for (int j = 0; j < col; j++) {
                if (mat[i][j] == 1) count[i][1]++;
            }
        }
        
        Arrays.sort(count, (a, b) -> a[1] != b[1] ? a[1] - b[1] : a[0] - b[0]);
        
        
int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = count[i][0];
        }
        return res;
    }
}
