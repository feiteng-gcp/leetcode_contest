class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        int[] point = new int[m];
        for (int i = 0; i < m; i++) {
            int j = 0;
            while (j < n && mat[i][j] == 1) {
                j++;
            }
            point[i] = j * m + i;
        }
        int[] result = new int[k];
        Arrays.sort(point);
        for (int i = 0; i < k; i++) {
            result[i] = point[i] % m;
        }
        return result;
    }
}
