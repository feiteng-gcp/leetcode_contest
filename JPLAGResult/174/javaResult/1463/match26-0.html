class Solution {
	
	private static class Element implements Comparable<Element> {
		public int index;
		public int count;
		public Element(int index , int count) {
			this.index = index;
			this.count = count;
		}
		@Override
		public int compareTo(Element e) {
			if (this.count != e.count) {
				return this.count - e.count;
			} else {
				return this.index - e.index;				
			}
		}
	}
	
    public int[] kWeakestRows(int[][] mat, int k) {
    	List<Element> list = new ArrayList<>();
    	int i , j , n = mat.length , m = mat[0].length;
    	for (i = 0;i < n;i ++) {
    		int count = 0;
    		for (j = 0;j < m;j ++) {
    			count += mat[i][j];
    		}
    		list.add(new Element(i , count));
    	}
    	Collections.sort(list);
    	int[] ans = new int[k];
    	for (i = 0;i < k;i ++) {
    		ans[i] = list.get(i).index;
    	}
    	return ans;
    }
	
}
