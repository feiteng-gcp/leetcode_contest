class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int m = mat[0].length;
        int[] count = new int[n];
        for (int i = 0; i < n; i++) {
            count[i] = 0;
            for (int j = 0; j < m; j++) {
                if (mat[i][j] == 1) {
                    count[i]++;
                }
            }
        }
        int[] index = new int[n];
        for (int i = 0; i < n; i++) {
            index[i] = i;
        }
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (count[i] > count[j] || (count[i] == count[j] && index[i] > index[j])) {
                    count[i] ^= count[j];
                    count[j] ^= count[i];
                    count[i] ^= count[j];
                    index[i] ^= index[j];
                    index[j] ^= index[i];
                    index[i] ^= index[j];
                }
            }
        }
        
int[] ans = new int[k];
        for (int i = 0; i < k; i++) {
            ans[i] = index[i];
        }
        return ans;
    }
}
