class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int[][] pair = new int[n][2];
        for(int i=0;i<n;i++){
            pair[i][0] = i;
            for(int j=0;j<mat[0].length;j++){
                if(mat[i][j] == 1) pair[i][1]++;
            }
        }
        Arrays.sort(pair, (a,b) -> {
           if(a[1]==b[1]) return a[0] - b[0];
            return a[1] - b[1];
        });
        int[] res = new int[k];
        for(int i=0;i<k;i++) res[i] = pair[i][0];
        return res;
    }
}
