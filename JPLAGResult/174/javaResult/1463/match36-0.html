class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int m = mat[0].length;
        int[] count = new int[n];
        
        for(int i = 0; i < n; i++){
            int cur = 0;
            for(int j = 0; j < m && mat[i][j] == 1; j++){
                cur++;
            }
            count[i] = cur;
        }
        
        Integer[] res = new Integer[n];
        for(int i = 0; i < n; i++){
            res[i] = i;
        }
        
        
        Arrays.sort(res, new Comparator<Integer>(){
            public int compare(Integer i1, Integer i2){
                return Integer.compare(count[i1], count[i2]);
            }
        });
        
        
        int[] out = new int[k];
        for(int i = 0; i < k; i++){
            out[i] = res[i];
        }
        
        return out;
    }
}
