class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        List<int[]> scores = new ArrayList<>();
        for (int i = 0; i < mat.length; i++) {
            int cnt = 0;
            for (int j = 0; j < mat[i].length; j++) {
                if (mat[i][j] == 1) cnt++;
                else break;
            }
            scores.add(new int[] {i, cnt});   
        }
        
        scores.sort((a, b) -> a[1] == b[1] ? Integer.compare(a[0], b[0]) : Integer.compare(a[1], b[1]));
        
        int[] ans = new int[k];
        for (int i = 0; i < k; i++) {
            ans[i] = scores.get(i)[0];
        }
        
        return ans;
     }
}
