class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int R = mat.length, C = mat[0].length;
        int[] w = new int[R];
        for (int r = 0; r < R; r++) {
            for (int m : mat[r]) {
                w[r] += m;
            }
        }
        
        List<Integer> list = new ArrayList();
        for (int i = 0; i < R; i++) {
            list.add(i);
        }
        Collections.sort(list, (a, b) -> w[a] == w[b] ? a - b : w[a] - w[b]);
        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = list.get(i);
        }
        return res;
    }
}
