class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int m = mat[0].length;
        int[][] counts = new int[n][2];
        for (int i = 0; i < n; i++) {
            counts[i][0] = i;
            for (int e = 0; e < m; e++) {
                counts[i][1] += mat[i][e];
            }
        }
        Arrays.sort(counts, (one, two) -> one[1] - two[1]);
        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = counts[i][0];
        }
        return res;
    }
}
