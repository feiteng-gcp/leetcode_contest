class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] == b[0] ? a[1] - b[1] : a[0] - b[0]);
        for (int i = 0; i < mat.length; i++) {
            int count = 0;
            while(count < mat[0].length && mat[i][count] == 1) {
                count++;
            }
            pq.offer(new int[]{count, i});
        }
        int[] res = new int[k];
        for (int i = 0; i < k; i++)
            res[i] = pq.poll()[1];
        return res;
    }
}
