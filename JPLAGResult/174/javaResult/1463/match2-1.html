class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int[] cnt = new int[n];
        for(int i = 0 ; i < n ; i++){
            int sum = 0, id = 0;
            while(id < mat[i].length && mat[i][id++] == 1){
                sum++;
            }
            cnt[i] = sum;
        }
        int[][] score = new int[n][2];
        for(int i = 0 ; i < n ; i++){
            score[i][0] = i;
            for(int j = 0 ; j < n ;j++){
                if(i == j){
                    continue;
                }
                if(cnt[i] > cnt[j] || cnt[i] == cnt[j] && i > j){
                    score[i][1]++;
                }
            }
        }
        Arrays.sort(score, new Comparator<int[]>(){
           @Override
            public int compare(int[] a, int[] b){
                return a[1] - b[1];
            }
        });
        int[] res = new int[k];
        for(int i = 0  ;i < k ; i++){
            res[i] = score[i][0];
        }
        return res;
    }
}
