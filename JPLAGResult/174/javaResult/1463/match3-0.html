class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int m = mat.length, n = mat[0].length;
        Queue<int[]> q = new PriorityQueue<>((a, b)->(a[1] != b[1] ? a[1] - b[1] : a[0] - b[0]));
        for(int i = 0; i < m; i++)
        {
            int sum = 0;
            for(int v : mat[i]) sum += v;
            q.offer(new int[]{i, sum});
        }
        int[] re = new int[k];
        for(int i = 0; i < k; i++) re[i] = q.poll()[0];
        return re;
    }
}
