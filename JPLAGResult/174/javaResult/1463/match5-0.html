class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int m = mat[0].length;
        int[] ans = new int[n];
        for (int i = 0; i < n; ++i) {
            ans[i] = i;
            for (int j = 0; j < m; ++j) {
                ans[i]+=mat[i][j]*1000;
            }
        }
        Arrays.sort(ans);
        int[] aa = new int[k];
        for (int i = 0; i < k; ++i) {
            aa[i] = ans[i]%1000;
        }
        return aa;
    }
}
