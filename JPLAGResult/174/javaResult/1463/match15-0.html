class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        int[] result = new int[k];
        
        int[][] data = new int[mat.length][2];
        
        for (int i = 0; i < mat.length; i++) {
            int count = 0;
            
            for (int j = 0; j < mat[0].length; j++) {
                if (mat[i][j] == 1) {
                    count++;
                }
            }
            
            data[i][0] = count;
            data[i][1] = i;
        }
        
        Arrays.sort(data, (a, b) -> {
           if (a[0] != b[0]) {
               return a[0] - b[0];
           } 
            
            return a[1] - b[1];
        });
        
        for (int i = 0; i < k; i++) {
            result[i] = data[i][1];
        }
        return result;
    }
}
