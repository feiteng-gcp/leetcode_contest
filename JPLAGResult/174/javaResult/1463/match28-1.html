class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        
int[][] ranks = new int[mat.length][2];
        for (int r = 0; r < mat.length; r++) {
            int count = 0;
            for (int c = 0; c < mat[0].length; c++) {
                if (mat[r][c] == 0) break;
                count++;
            }
            ranks[r][0] = r;
            ranks[r][1] = count;
        }
        Arrays.sort(ranks, (r1, r2) -> {
            if (r1[1] != r2[1]) return r1[1] - r2[1];
            return r1[0] - r2[0];
        });
        
int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = ranks[i][0];
        }
        return result;
    }
}
