class Pair {
    int ind;
    int num;
    Pair(int a, int b) {
        ind = a;
        num = b;
    }
}
class Solution {
    public int[] kWeakestRows(int[][] mat, int k) {
        List<Pair> list = new ArrayList<>();
        for (int i = 0; i < mat.length; i++) {
            int j = 0, total = 0;
            while (j < mat[0].length && mat[i][j] == 1) {
                j++;
                total++;
            }
            list.add(new Pair(i, total));
        }
        Collections.sort(list, (a, b) -> a.num == b.num ? a.ind - b.ind : a.num - b.num);
        int[] res = new int[k];
        for (int i = 0; i < k; i++) res[i] = list.get(i).ind;
        return res;
    }
}
