class Solution {
    class Pair implements Comparable<Pair>{
        int count;
        int index;
        public Pair(int count, int index){
            this.count = count;
            this.index = index;
        }
        
        public int compareTo(Pair other){
            if(this.count!=other.count){
                return this.count-other.count;
            }
            else{
                return this.index - other.index;
            }
        }
        
    }
    
    
public int[] kWeakestRows(int[][] mat, int k) {
        int n = mat.length;
        int m = mat[0].length;
        Pair[] A = new Pair[n];
        for(int r=0; r<n; r++){
            int soilders = 0;
            for(int c=0; c<m; c++){
                if(mat[r][c]==1){
                    soilders++;
                }
                else{
                    break;
                }
            }
            A[r] = new Pair(soilders, r);
        
}
        Arrays.sort(A);
        int[] res = new int[k];
        for(int i=0; i<k; i++){
            res[i] = A[i].index;
        }
        return res;
    }
}
