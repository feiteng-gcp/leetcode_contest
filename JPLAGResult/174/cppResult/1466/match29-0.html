class Solution {
public:
    int dp[1111];
    int Solve(const vector<int>& arr, int pos, int d)  {
        if (dp[pos] != -1)
            return dp[pos];
        int ans=1;
        for(int j=pos-1;j>=0;j--) {
            if (arr[j] >= arr[pos] || pos-j > d)
                break;
            ans = max(ans, 1+Solve(arr, j, d));
        }   
        
        for(int j=pos+1;j<arr.size();j++) {
            if (arr[j] >= arr[pos] || j-pos > d)
                break;
            ans = max(ans, 1+Solve(arr, j, d));
        }
        
        return dp[pos]=ans;
        
    }
    
    
    int maxJumps(vector<int>& arr, int d) {
        for(int i=0;i<1111;i++)
            dp[i]=-1;
        
        int ans=0;
        for(int i=0;i<arr.size();i++)
            ans=max(ans, Solve(arr, i, d));
        return ans;
        
        
    }
};
