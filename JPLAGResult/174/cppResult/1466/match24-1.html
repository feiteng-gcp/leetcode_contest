class Solution {
public:
    int maxJumps(vector<int>& arr, int d) {
        int dp[1005], ans = 0;
        vector<pair<int, int>> ar;
        for (int i = 0; i<arr.size(); i++){
            ar.push_back(make_pair(arr[i], i));
        }
        sort(ar.begin(), ar.end());
        memset(dp, 0, sizeof dp);
        for (pair<int, int> p: ar){

            int val = p.first, idx = p.second;
            for (int i = idx-1; i>=max(0, idx-d); i--){
                if (arr[i]>=val) break;
                dp[idx] = max(dp[idx], dp[i]+1);
            }
            for (int i = idx+1; i<=idx+d && i<ar.size(); i++){
                if (arr[i]>=val) break;
                dp[idx] = max(dp[idx], dp[i]+1);
            }
            ans = max(ans, dp[idx]);
        }
        return ans+1;
        
    }
};
