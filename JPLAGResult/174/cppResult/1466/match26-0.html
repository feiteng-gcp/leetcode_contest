class Solution {
public:
    int dfs(vector<int>& arr, int rk){
        if(dp[rk]!=-1){
            return dp[rk];
        }
        int maxn=1;
        for(int i=rk+1; i<min(N, rk+D+1); ++i){
            if(arr[rk]>arr[i]){
                maxn = max(maxn, 1+dfs(arr, i));
            }else{
                break;
            }
        }
        for(int i=rk-1; i>=max(0, rk-D); --i){
            if(arr[rk]>arr[i]){
                maxn = max(maxn, 1+dfs(arr, i));
            }else{
                break;
            }
        }
        dp[rk] = maxn;
        return maxn;
    }


    int maxJumps(vector<int>& arr, int d) {
        N = arr.size();
        memset(dp, -1, sizeof(dp));
        D = d;
        int ans=0;
        for(int i=0; i<N; ++i){
            ans = max(ans, dfs(arr, i));
        }
        return ans;
    }
    int dp[1005];
    int N, D;
};
