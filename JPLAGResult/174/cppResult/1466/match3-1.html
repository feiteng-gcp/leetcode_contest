class Solution {

    int memorized_search(vector<int>& arr, int n, int d, vector<int>& dp, int pos){
        if(dp[pos]!=-1) return dp[pos];
        int res = 1;
        for(int i = pos-1; i >=max(0, pos-d); --i){
            if(arr[i]>=arr[pos]) break;
            res = max(res, 1+memorized_search(arr, n, d, dp, i));
        }
        for(int i = pos+1; i <=min(n-1, pos+d); ++i){
            if(arr[i]>=arr[pos]) break;
            res = max(res, 1+memorized_search(arr, n, d, dp, i));
        }
        return dp[pos] = res;
    }
public:
    int maxJumps(vector<int>& arr, int d) {

        int n = arr.size();
        vector<int> dp(n, -1);
        int res = 1;
        for(int i = 0; i < n; ++i){
            res = max(res, memorized_search(arr, n, d, dp, i));
        }
        return res;
    }
};
