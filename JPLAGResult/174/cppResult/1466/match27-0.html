class Solution {
    int n;
    int d;
    vector<int> a;
    vector<int> f;
    
    int solve(int pos){
        if (f[pos] != -1) return f[pos];
        
        f[pos] = 1;
        for (int i=1;i<=d;++i){
            if (pos+i>=n || a[pos+i] >= a[pos]) break;
            f[pos] = max(f[pos], 1 + solve(pos + i));
        }
        
        for (int i=1;i<=d;++i){
            if (pos-i<0 || a[pos-i] >= a[pos]) break;
            f[pos] = max(f[pos], 1 + solve(pos - i));
        }
        
        return f[pos];
    }
    
public:
    int maxJumps(vector<int>& arr, int dist) {
        a = arr;
        n = a.size();
        d = dist;
        f = vector<int>(n,-1);
        
        for (int i=0;i<n;++i) solve(i);
        
        int result = 0;
        for (int i=0;i<n;++i)
            result = max(result, f[i]);
        return result;
    }
};
