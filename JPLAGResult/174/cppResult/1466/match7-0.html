class Solution {
public:
    int maxJumps(vector<int>& arr, int d) {
        stack<int> sl, sr;
        vector<vector<int>> g(arr.size(), vector<int>());
        vector<int> id(arr.size(), 0), od(arr.size(), 0);
        for (int i = 0; i < arr.size(); ++i) {
            while (!sl.empty() && arr[sl.top()] < arr[i] && sl.top() + d >= i) sl.pop();
            
            int left = 0;
            if (!sl.empty()) {
                left = max(i - d, sl.top() + 1);
            } 
            for (int j = left; j < i; ++j) {
                od[i]++;
                id[j]++;
                g[i].push_back(j);
            }
            sl.push(i);
        //    printf("left %d %d\n", i, left[i]);
        }
        
        for (int i = arr.size() - 1; i >= 0; --i) {
            while (!sr.empty() && arr[sr.top()] < arr[i] && sr.top() - d <= i) sr.pop();
            int right = arr.size() - 1;
            if (!sr.empty()) {          
                right = min(i + d, sr.top() - 1);
            } 
            for (int j = i + 1; j <= right; ++j) {
                od[i]++;
                id[j]++;
                g[i].push_back(j);
            }
            sr.push(i);
         //   printf("right %d %d\n", i, right[i]);
        }
        
        // topological sorting
        
        set<int> q, q2;
        int depth = 0;
        
        for (int i = 0; i < arr.size(); ++i) {
            if (id[i] == 0) q.insert(i);
        }
        
        while (!q.empty()) {
            for (auto &n: q) {
                for (auto &neigh: g[n]) {
                    id[neigh]--;
                    if (id[neigh] == 0) q2.insert(neigh);
                }
            }
            swap(q, q2);
            q2.clear();
            ++depth;
        } 
        
        return depth;
    }
    
};
