class Solution {
public:
    int ans[1000];
    int dfs(vector<int>& arr,int index,int d)
    {
        if(ans[index]!=-1) return ans[index];
        int localMax=0;
        for(int i=index-1;i>=index-d;i--)
        {
            if(i<0) break;
            if(arr[i]>=arr[index]) break;
            if(ans[i]==-1) dfs(arr,i,d);

            if(ans[i]>localMax) localMax=ans[i];
        }
        for(int i=index+1;i<=index+d;i++)
        {
            if(i>=arr.size()) break;
            if(arr[i]>=arr[index]) break;
            if(ans[i]==-1) dfs(arr,i,d);
            if(ans[i]>localMax) localMax=ans[i];
        }
        ans[index]=localMax+1;
        return ans[index];
    }
    int maxJumps(vector<int>& arr, int d) {
        memset(ans,-1,sizeof(ans));

        int globalMax=-1;
        for(int i=0;i<arr.size();i++)
        {
            dfs(arr,i,d);
            if(ans[i]>globalMax) globalMax=ans[i];
        }
        return globalMax;
        
    }
};
