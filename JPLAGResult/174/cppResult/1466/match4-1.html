class Solution {
public:
    vector<int> dp;
    int n;
    vector<int> nums;
    int d;
    
    int dfs(int p){
        if(dp[p] != -1) return dp[p];
        
        int cur = 1;
        for(int i=-1; i>=-d; i--){   
            int nx = p + i;
            if(nx>=0 && nx<n){
                if(nums[nx] < nums[p])
                    cur = max(cur, dfs(nx)+1);
                else
                    break;
            }
            
        }
        
         for(int i=1; i<=d; i++){   
            int nx = p + i;
            if(nx>=0 && nx<n){
                if(nums[nx] < nums[p])
                    cur = max(cur, dfs(nx)+1);
                else
                    break;
            }
            
        }
        return dp[p] = cur;
    }
    int maxJumps(vector<int>& arr, int d) {
        this->d = d;
        nums = arr;
        n = arr.size();
        dp = vector<int>(n, -1);
        
        
        int ret = 0;
        for(int i=0; i<n; i++){
            int cur = dfs(i);
            ret = max(ret, cur);
        }
        return ret;
    }
};
