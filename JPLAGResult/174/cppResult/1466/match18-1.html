class Solution {
public:
    int f(int i, int &d, vector<int>& dp, vector<int>& arr)
    {
        if (dp[i]) return dp[i];
        int &x = dp[i];
        x = 1;

        for (int j = i - 1; j >= max(0, i - d); --j) 
        {
            if (arr[j] >= arr[i]) break;
            x = max(x, f(j, d, dp, arr) + 1);
        }
        
        for (int j = i + 1; j <= min((int)dp.size() - 1, i + d); ++j)
        {
            if (arr[j] >= arr[i]) break;
            x = max(x, f(j, d, dp, arr) + 1);
        }
        
        return x;
    }
    
    int maxJumps(vector<int>& arr, int d) {
        int n = arr.size();
        vector<int> dp(n, 0);

        int res = 0;
        for (int i = 0; i < n; ++i)
        {
            res = max(res, f(i, d, dp, arr));
        }
        return res;
    }
};
