class Solution {
public:
    int dfs(vector<int>& vis, vector<int>& a, int x, int d) {
        if (vis[x]) return vis[x];
        vis[x] = 1;
        for (int i = x - 1; i >= 0 && x - i <= d; i --) {
            if (a[i] >= a[x]) break;
            vis[x] = max(vis[x], dfs(vis, a, i, d) + 1);
        }
        for (int i = x + 1; i < a.size() && i - x <= d; i ++) {
            if (a[i] >= a[x]) break;
            vis[x] = max(vis[x], dfs(vis, a, i, d) + 1);
        }
        return vis[x];
    }
    
    int maxJumps(vector<int>& arr, int d) {
        int n = arr.size();
        vector<int> vis(n);
        for (int i = 0; i < n; i ++) vis[i] = 0;
        int ans = 0;
        for (int i = 0; i < n; i ++) 
            ans = max(ans, dfs(vis, arr, i, d));
        return ans;
    }
};
