class Solution {
public:
    int maxJumps(vector<int>& arr, int d) {
        int N = arr.size();
        
        vector<pair<int, int>> vp;
        for (int i = 0; i < N; i++) {
            vp.push_back(std::pair<int,int>(arr[i], i));
        }
        sort(vp.begin(), vp.end());
        
        vector<int> best(N, 1);
        int res = 1;
        
        for (const auto& pp : vp) {
            int i = pp.second;

            int &b = best[i];
            for (int j = i + 1; j < N && arr[j] < arr[i] && (j - i) <= d; j++) {
                b = max(b, best[j] + 1);
            }
            for (int j = i - 1; j >= 0 && arr[j] < arr[i] && (i - j) <= d; j--) {
                b = max(b, best[j] + 1);
            }
            res = max(res, b);
        }
        return res;
    }
};
