class Solution {
public:
    int maxJumps(vector<int>& arr, int d) {
        int n=arr.size();
        vector<vector<int>> a(n,vector<int>(2));
        for(int i=0; i<n; i++){
            a[i][0]=arr[i];
            a[i][1]=i;
        }
        vector<int> dp(n,1);
        sort(a.begin(),a.end());
        int minele=a[0][0];
        for(int i=0; i<n; i++){
            int ele=a[i][0];
            int ind=a[i][1];
            // cout<<ind;
            if(ele==minele){
                dp[ind]=1;
                continue;
            }

            for(int j=1; ind+j<n&&j<=d; j++){
                if(arr[ind+j]>=ele)
                    break;
                dp[ind]=max(dp[ind],dp[ind+j]+1);
            }
            for(int j=1; ind-j>=0&&j<=d; j++){
                if(arr[ind-j]>=ele)
                    break;
                dp[ind]=max(dp[ind],dp[ind-j]+1);
            }
        }
        for(int i=0; i<n; i++)
            cout<<dp[i]<<" ";
        return *max_element(dp.begin(),dp.end());
    }
};
