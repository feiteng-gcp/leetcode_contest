class Solution {
    
private:
    int dp[1002];
    vector<int>ara;
public:
    
    int dp_func(int x,int d)
    {
        if(dp[x]!=-1)return dp[x];
        int ret=0;
        for(int i=x+1;i<=d+x&&i<ara.size();i++)
        {
            if(ara[i]>=ara[x])break;
            ret=max(ret,dp_func(i,d));
        }
         for(int i=x-1;i>=x-d&&i>=0;i--)
        {
            if(ara[i]>=ara[x])break;
            ret=max(ret,dp_func(i,d));
        }
        return dp[x]=ret+1;
        
    }
    int maxJumps(vector<int>& arr, int d) {
        
        ara=arr;
        memset(dp,-1,sizeof dp);
        int ans=0;
        for(int i=0;i<ara.size();i++)
        {
            ans=max(ans,dp_func(i,d));
            
        }
        return ans;
        
    }
};
