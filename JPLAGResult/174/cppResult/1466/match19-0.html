class Solution {
public:
    vector<int> dp;
    int canJump(vector<int>& arr, int d, int start){
        if(dp[start]!=-1) return dp[start];
        int ret = 1;
        for(int i=start-1;i>=start-d;i--){
            if(i<0) break;
            if(arr[i]>=arr[start]) break;
            if(1+canJump(arr, d, i)>ret) ret=1+canJump(arr, d, i);
        }
        for(int i=start+1;i<=start+d;i++){
            if(i>=arr.size()) break;
            if(arr[i]>=arr[start]) break;
            if(1+canJump(arr, d, i)>ret) ret=1+canJump(arr, d, i);
        }
        dp[start]=ret;
        //printf("from %d you can jump %d\n", start, ret);
        return ret;
    }
    int maxJumps(vector<int>& arr, int d) {
        dp.resize(arr.size(), -1);
        int ret = 1;
        for(int i=0;i<arr.size();i++){
            if(canJump(arr, d, i)>ret) ret=canJump(arr, d, i);
        }
        return ret;
    }
};
