class Solution {
public:
    int  hop;
    int dp[2000];

    int dfs(vector<int>& arr, int i)
    {
        if(dp[i]!=0)return dp[i];
        
        int ret = 0;
        for(int j =1;j<=hop;j++)
        {
            if(i+j<0||i+j>=arr.size()||arr[i+j]>=arr[i])break;
            ret = max(ret , dfs(arr, i+j));
        }
        for(int j =1;j<=hop;j++)
        {
            if(i-j<0||i-j>=arr.size()||arr[i-j]>=arr[i])break;
            ret = max(ret , dfs(arr, i-j));
        }
        return dp[i]=ret+1;
    }
    int maxJumps(vector<int>& arr, int d) {
        hop = d;

        int ans =0;
        for(int i=0;i<arr.size();i++)
        {
            ans = max(ans, dfs(arr, i));
            cout<<dp[i]<<endl;
        }
        return ans;
    }
};
