class Solution {
public:

    vector<int> dp;
    vector<int> ref;
    int D;


    int go(int idx) {
        int& v = dp[idx];
        if (v)return v - 1;


        for (int i = 1; i <= D; i++) {
            if (idx + i >= ref.size() || ref[idx + i] >= ref[idx])break;
            v = max(v, go(idx + i) + 1);
        }
        for (int i = 1; i <= D; i++) {
            if (idx - i < 0 || ref[idx - i] >= ref[idx])break;
            v = max(v, go(idx - i) + 1);
        }

        return v++;
    }

    int maxJumps(vector<int>& arr, int d) {
        dp = vector<int>(arr.size(), 0);

        ref = arr;
        D = d;

        int ans = 0;
        for (int i = 0; i < arr.size(); i++)
            ans = max(ans, go(i));
        return ans + 1;
        
    }
};
