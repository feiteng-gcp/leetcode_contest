class Solution {
public:
    int maxJumps(vector<int>& arr, int d) {
        int N=arr.size();
        vector<int> idx(N);
        vector<int> dp(N,1);
        for(int i=0;i<N;i++) idx[i]=i;
        sort(idx.begin(),idx.end(),[&](int a,int b){
            return arr[a]<arr[b];
        });
        for(int i=0;i<N;i++){
            int p=idx[i];
            for(int j=1;p-j>=0&&j<=d;j++){
                if(arr[p-j]>=arr[p]) break;
                dp[p]=max(dp[p],dp[p-j]+1);
            }
            
            for(int j=1;p+j<N&&j<=d;j++){
                if(arr[p+j]>=arr[p]) break;
                dp[p]=max(dp[p],dp[p+j]+1);
            }
        }
        int ans=0;
        for(int c:dp) ans=max(ans,c);
        return ans;
    }
};
