class Solution {
public:
        int D;
    vector < int > dp;
    vector < int > A;

    int call(int i) {
        if (dp[i] != -1) return dp[i];

        int maxx = 0;
        for (int j = i+1; j <= min(i+D, (int)A.size()-1); j++) {
            if (A[j] >= A[i]) break;
            maxx = max(maxx, call(j));
        }
        for (int j = i-1; j >= max(i-D, 0); j--) {
            if (A[j] >= A[i]) break;
            maxx = max(maxx, call(j));
        }
        return dp[i] = maxx + 1;
    }

    int maxJumps(vector<int>& arr, int d) {
        A = arr;
        D = d;
        dp.resize(arr.size()+1, -1);
        int maxx = 0;
        for (int i = 0; i < arr.size(); i++) {
            int tmp = call(i);
            maxx = max(maxx, tmp);
        }
        return maxx;
    }

};
