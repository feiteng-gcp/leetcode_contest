class Solution {
public:
    int help(map<int, int>& dp, int cur, int d, vector<int>& arr)
    {
        if(dp.find(cur)!=dp.end()) return dp[cur];

        int res = 1;
        for(int i= cur+1;i<=cur+d;i++)
        {
            if(i>=0&&i<arr.size()&&arr[i]<arr[cur])
            {
                res = max(1+help(dp, i, d, arr), res);
            }
            else
            {
                break;
            }
        }
        for(int i= cur-1;i>=cur-d;i--)
        {
            if(i>=0&&i<arr.size()&&arr[i]<arr[cur])
            {
                res = max(1+help(dp, i, d, arr), res);
            }
            else
            {
                break;
            }
        }
        dp[cur] = res;
        return res;
    }
    int maxJumps(vector<int>& arr, int d) {
        map<int, int> dp;

        int n = arr.size();
        int res = 1;
        for(int i=0;i<n;i++)
        {
            res = max(res, help(dp, i, d, arr));
        }
        return res;
    }
};
