class Solution {
    int n;
public:
    int maxJumps(vector<int>& arr, int d) {
        n = arr.size();
        auto MAX = vector<vector<int>>(n, vector<int>(n));
        for (int i = 0; i < n; i++) {
            int M = arr[i];
            for (int j = i; j < n; j++) {
                M = max(M, arr[j]);
                MAX[i][j] = M;
            }
        }
        
        vector<int> memo(n, 0);
        int res = 1;
        for (int i = 0; i < n; i++) {
            if (memo[i] != 0)    res = max(res, memo[i]);
            int ret = dfs(arr, MAX, d, memo, i);
            memo[i] = ret;
            // cout<<i<<" "<<ret<<endl;
            res = max(res, ret);
        }
        return res;
    }
    
private:
    int dfs(vector<int>& arr, vector<vector<int>>& MAX, int d, vector<int>& memo, int s) {
        if (memo[s] != 0)    return memo[s];
        int res = 1;
        for (int i = max(0, s - d); i < s; i++) {
            if (MAX[i][s - 1] >= arr[s])    continue;
            res = max(res, dfs(arr, MAX, d, memo, i) + 1);
        }
        for (int i = s + 1; i <= min(n - 1, s + d); i++) {
            if (MAX[s + 1][i] >= arr[s])    continue;
            res = max(res, dfs(arr, MAX, d, memo, i) + 1);
        }
        memo[s] = res;
        return res;
    }
};
