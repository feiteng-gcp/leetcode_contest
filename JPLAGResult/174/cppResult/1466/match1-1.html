class Solution {
public:
    int dp[1005];
    int maxJumps(vector<int>& arr, int d) {
        int n = arr.size();
        int res = 0;
        for (int i = 0; i < n; i++){
            res = max(res, helper(arr, d, i));
        }
        return res;
        
    }
    int helper(vector<int> &arr, int d, int pos){
        if (dp[pos] != 0) return dp[pos];
        dp[pos] = 1;
        for(int i = 1; i <= d; i++){
            int l = pos - i;
            if (l < 0 || arr[l] >= arr[pos]) break;
            dp[pos] = max(dp[pos], 1 + helper(arr, d, l));
        }
        for (int i = 1; i <= d; i++){
            int r = pos + i;
            if (r >= arr.size() || arr[r] >= arr[pos]) break;
            dp[pos] = max(dp[pos], 1 + helper(arr, d, r));
        }
        return dp[pos];
    }
    
};
