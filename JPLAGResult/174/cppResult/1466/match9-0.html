class Solution {
public:
    vector<int> f;
    
    int dfs(int x, vector<int>& arr, int d) {
        // cout<<x<<endl;
        if (f[x]!=-1)
            return f[x];
        f[x]=1;
        
        for (int i=x+1;i<=x+d && i<arr.size(); i++) {
            if (arr[i]>=arr[x])
                break;
            f[x]=max(f[x], dfs(i, arr, d)+1);
        }
        
        for (int i=x-1;i>=0 && i>=x-d; i--) {
            if (arr[i]>=arr[x])
                break;
            f[x]=max(f[x], dfs(i, arr, d)+1);
        }
        
        return f[x];
    }
    
    int maxJumps(vector<int>& arr, int d) {
        f.clear();
        f.resize(arr.size(), -1);
        
        for (int i=0;i<arr.size();i++)
            if (f[i]==-1)
                dfs(i, arr, d);
        
        int ans=0;
        for (auto& i:f) 
            ans=max(ans, i);
        
        // for (int i=0;i<f.size();i++)
        //     cout<<i<<" "<<f[i]<<endl;
        
        return ans;
    }
};
