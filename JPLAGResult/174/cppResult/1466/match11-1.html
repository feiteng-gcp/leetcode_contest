class Solution {
public:
    vector<int> dp;
    int maxJumps(int k, const vector<int>& arr, int d) {
		if (dp[k] == 0) {
			dp[k] = 1;
			for (int i = k - 1; i >= max(0, k - d) && arr[i] < arr[k]; i--) {
				dp[k] = max(dp[k], maxJumps(i, arr, d) + 1);
			}
			for (int i = k + 1; i <= min<int>(arr.size() - 1, k + d) && arr[i] < arr[k]; i++) {
				dp[k] = max(dp[k], maxJumps(i, arr, d) + 1);
			}
		}
		return dp[k];
	}
    
    int maxJumps(vector<int>& arr, int d) {
        dp.resize(arr.size());
        int ma = 0;
        for(int i = 0; i < arr.size(); i++) {
            ma = max(ma, maxJumps(i, arr, d));
        }
        return ma;
    }
};
