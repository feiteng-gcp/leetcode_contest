class Solution {
private:
    vector<int> visited;
public:
    int maxJumps(vector<int>& arr, int d) {
        visited = vector<int>(arr.size(), -1);
        int res = 0;
        //cout << "1" << endl;
        for(int i = 0; i < arr.size(); ++i)
        {
            res = max(res, helper(arr, d, i));
            cout << i << " " << helper(arr, d, i) << endl;
        }
            
        return res + 1;
    }
    
    int helper(vector<int> &arr, int d, int index)
    {
        if(visited[index] != -1) return visited[index];
        int curCount = 0;
        for(int i = 1; i <= d; ++i)
        {
            if(index - i < 0 || arr[index - i] >= arr[index]) break;
            curCount = max(curCount, 1 + helper(arr, d, index - i));
        }
        for(int i = 1; i <= d; ++i)
        {
            if(index + i >= arr.size() || arr[index + i] >= arr[index]) break;
            curCount = max(curCount, 1 + helper(arr, d, index + i));
        }
        return visited[index] = curCount;
    }
};
