class Solution {
public:
    int maxJumps(vector<int>& a, int d) {
        
        int n = a.size();
        
        vector<pair<int, int> > h;
        
        for(int i=0; i<n; i++)
            h.push_back({a[i], i});
        
        sort(h.begin(), h.end());
        
        vector<int> res(n, 1); 
        
        while(h.size()){
            auto x = h.back();
            h.pop_back();
            
            int i = x.second;
            
            for(int j=i-1; j>=0 && j>=(i-d); j--){
                if(a[j]>=a[i]) break;
                res[j] = max(res[j], res[i] + 1);
            }
            
            for(int j=i+1; j<n && j<=(i+d); j++){
                if(a[j]>=a[i]) break;
                res[j] = max(res[j], res[i] + 1);
            }
        }
        
        return *max_element(res.begin(), res.end());
    }
};
