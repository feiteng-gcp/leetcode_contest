class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
        vector<pair<int,int> > p;
        for(int i =  0;i<mat.size();i++){
            int num=0;
            for(int j=0;j<mat[0].size();j++){
                num+=mat[i][j];
            }
            p.push_back(make_pair(num,i));
        }
        sort(p.begin(),p.end());
        vector<int> res;
        for(int i =0;i<k;i++)res.push_back(p[i].second);
        return res;
    }
};
