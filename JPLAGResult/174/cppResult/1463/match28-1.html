class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
        int n=mat.size(),m=mat[0].size();
        vector<pair<int,int> >box;
        for(int i=0;i<n;i++){
            int sorce=0;
            for(int j=0;j<m;j++)if(mat[i][j])sorce++;

            box.push_back({sorce,i});
        }
        sort(box.begin(),box.end());
        vector<int> ans;
        for(int i=0;i<k;i++)ans.push_back(box[i].second);
        return ans;
    }
};
