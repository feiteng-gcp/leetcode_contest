class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& m, int k) {
        vector<pair<int, int>> d;
        for (int i = 0; i < m.size(); ++i) {
            int cnt = 0;
            for (int j = 0; j < m[i].size(); ++j) {
                if (m[i][j] == 1)
                    ++cnt;
                else
                    break;

            }
            d.push_back({cnt, i});
        }
        
        sort(d.begin(), d.end());
        
        vector<int> res;
        for (int i = 0; i < k; ++i)
            res.push_back(d[i].second);
        
        return res;
    }
};
