class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
        vector<int> m(mat.size(), 0);
        for (int i = 0; i < mat.size(); ++i) {
            int s = 0;
            for (int j = 0; j < mat[0].size(); ++j) {
                s += mat[i][j];
            }
            
            m[i] = s;
        }
        
        vector<int> ret(mat.size(), 0);
        
        for (int i = 0; i < mat.size(); ++i) {
            ret[i] = i;
        }
        
        sort(ret.begin(), ret.end(), [&](int &i1, int &i2) {
            if (m[i1] < m[i2]) return true;
            if (m[i1] > m[i2]) return false;
            return i1 < i2;
        });
        
        ret.resize(k);
        return ret;
    }
};
