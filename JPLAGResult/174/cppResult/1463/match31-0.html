class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
        vector<pair<int,int> > v;
        for(int i = 0;i < mat.size();i++){
            int r = i;
            int c = 0;
            for(int j = 0;j < mat[i].size();j++){
                if(mat[i][j] == 1)
                    c++;
            }
            v.push_back({c,r});
        }
        sort(v.begin(),v.end());
        vector<int> res;
        for(int i = 0;i < k;i++){
            res.push_back(v[i].second);
        }
        return res;
    }
};
