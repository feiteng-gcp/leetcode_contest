class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
        vector<pair<int, int>> ar;
        for (int i = 0; i<mat.size(); i++){
            int cnt = 0;
            for (int j = 0; j<mat[i].size(); j++){
                cnt+=mat[i][j];
            }
            ar.push_back(make_pair(cnt, i));
        }   
        sort(ar.begin(), ar.end());
        vector<int> ans;
        for (int i = 0; i<k; i++){
            ans.push_back(ar[i].second);
        }
        return ans;        
    }
};
