class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
        vector<int> v(mat.size());
        for(int i = 0; i < mat.size(); i++) {
            v[i] = i;
        }
        sort(v.begin(), v.end(), [&](const int a, const int b) {
            int va = mat[0].size();
            int vb = mat[0].size();
            for(int i = 0; i < mat[0].size(); i++) {
                if(mat[a][i] == 0) {
                    va = i;
                    break;
                }
            }
            for(int i = 0; i < mat[0].size(); i++) {
                if(mat[b][i] == 0) {
                    vb = i;
                    break;
                }
            }
            if(va != vb) {
                return va < vb;
            }
            return a < b;
        });
        
        return vector<int>(v.begin(), v.begin() + k);
    }
};
