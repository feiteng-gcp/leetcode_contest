class Solution {
public:
    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
        vector<pair<int,int>>vv;
        for(int i=0; i<mat.size(); i++)
        {
            int cnt=0;
            for(int j=0; j<mat[i].size(); j++)
            {
                if(mat[i][j]) cnt++;
                else break;
                
            }
            vv.push_back({cnt,i});
        }
        sort(vv.begin(),vv.end());
        vector<int>ans;
        for(int i=0; i<k; i++)
        {
            ans.push_back(vv[i].second);
        }
        return ans;
        
    }
};
