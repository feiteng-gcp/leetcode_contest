class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int total_count = arr.size();
        map<int, int> freq;
        for (int& val : arr) { 
            freq[val]++;
        }
        vector<int> freqList;
        for (auto it : freq) { 
            freqList.push_back(it.second);
        }
        sort(begin(freqList), end(freqList));
            int ans = 0;
        int total = 0;
        for (int i = freqList.size() - 1; i >= 0; i--) { 
            total += freqList[i];
            ans++;
            if (total >= total_count / 2) { 
                return ans;
            }
        }
        return ans;
        /*

                return ans;
            }
        }
        return ans;
        */
    }
};
