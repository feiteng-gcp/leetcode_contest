class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int n=arr.size();
        map<int, int> m;
        for (auto& i:arr)
            m[i]++;
        vector<pair<int, int>> a;
        for (auto t:m)
            a.push_back({t.second, t.first});
        
        sort(a.begin(), a.end(), greater<>());
        
        // for (auto t:a)
        //     cout<<t.first<<' '<<t.second<<endl;
        int ans=0;
        
        for (auto& t:a) {
            n-=t.first;
            ans++;
            if (n<=arr.size()/2)
                break;
        }
        
        
        return ans;
    }
};
