class Solution {
public:
    int minSetSize(vector<int>& arr) 
    {
        map<int, int> dataMap;
        for (int i = 0; i < arr.size(); i++)
        {
            dataMap[arr[i]] += 1;
        }
        vector<int> dataArr;
        int sum = 0;
        for (auto a = dataMap.begin(); a != dataMap.end(); ++a)
        {
            sum += a->second;
            dataArr.push_back(a->second);
        }
        sort(dataArr.begin(), dataArr.end());
        int sumT = 0;
        for (int i = 0; i < dataArr.size(); i++)
        {
            sumT += dataArr[dataArr.size()-1-i];
            if (sumT*2 >= sum)
            {
                return i+1;
            }
        }
        return 0;        
    }
};
