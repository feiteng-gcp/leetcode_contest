class Solution {
public:
    int minSetSize(vector<int>& arr) {
        unordered_map<int, int> cnt;
        for(int i : arr) {
            cnt[i]++;
        }
        vector<int> tmp;
        for(auto& p : cnt) {
            tmp.push_back(p.second);
        }
        sort(tmp.rbegin(), tmp.rend());
        int acnt = 0, n = arr.size();
        for(int i = 0; i < tmp.size(); ++i) {
            acnt += tmp[i];
            if(acnt >= n/2) {
                return i + 1;
            }
        }
        return -1;
        
    }
};
