class Solution {
public:
    int minSetSize(vector<int>& arr) {
        map<int, int> m;
        for (auto i: arr) m[i]++;
        vector<pair<int, int>> n;
        for (auto &e: m) n.push_back(e);
        sort(n.begin(), n.end(), [&](pair<int, int> &p1, pair<int, int> &p2) {
            if (p1.second < p2.second) return false;
            if (p1.second > p2.second) return true;
            return p1.first < p1.second;
        });
        
        int ret = 0, c = 0;
        
        for (auto &e: n) {
            ret++;
            c += e.second;
            if (c >= arr.size()/2) break;
        }
        return ret;
    }
};
