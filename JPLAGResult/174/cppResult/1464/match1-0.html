class Solution {
public:
    int minSetSize(vector<int>& arr) {
        map<int, int> m;
        for(int a: arr) m[a]++;
        multimap<int, int, std::greater<int>> rm;
        for(const auto& u: m) rm.insert({u.second,u.first});
        int ret = 0;
        size_t s = arr.size();
        for(const auto& u: rm) {
            s -= u.first;
            ret++;
            cout<<s<<endl;
            if(s <= arr.size()/2) return ret;
        }
        return 0;
    }
};
