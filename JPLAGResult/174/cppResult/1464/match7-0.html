class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int n = arr.size();
        unordered_map<int, int> m;
        for (auto &it : arr){
            m[it]++;
        }
        vector<int> res;
        for (auto it: m){
            res.push_back(it.second);
        }
        sort(res.begin(), res.end());
        // int n = res.size()
        int cnt = 0;
        int ans = 0;
        int k = res.size();
        for(int i = k - 1; i>= 0; i--){
            // if (res[i])
            cnt += res[i];
            ans++;
            if (cnt >= n / 2) break;
        }
        return ans;
    }
};
