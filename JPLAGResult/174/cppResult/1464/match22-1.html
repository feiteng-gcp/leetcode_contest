class Solution {
public:
    int minSetSize(vector<int>& arr) {
        unordered_map<int, int> r;
        for (int a: arr) {
            r[a] += 1;
        }
        vector<int> cs;
        for (auto it = r.begin(); it != r.end(); it++) {
            cs.push_back(it->second);
        }
        sort(cs.begin(), cs.end(), [](int a, int b) ->bool {
            return a > b;
        });
        int all = accumulate(cs.begin(), cs.end(), 0);

        int q = 0;
        for (int i = 0; i < cs.size(); i++) {
            q += cs[i];
            if (2 * q >= all) {
                return i + 1;
            }
        }
        return cs.size();
    }
};
