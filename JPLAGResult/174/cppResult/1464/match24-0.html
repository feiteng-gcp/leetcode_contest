class Solution {
public:
    int minSetSize(vector<int>& arr) {
        
        
        map<int,int> counts;
        for(int x : arr) {
            counts[x]++;
        }
        
        vector<int> cts;
        for(auto x : counts) {
            cts.push_back(x.second);
        }
        
        sort(cts.begin(),cts.end(),greater<int>());
        
      
        int num_removed=0;
        
        for(int i=0;i<cts.size();i++) {
            num_removed += cts[i];
            
            if (2 * num_removed >= arr.size())
                return i+1;
        }
        
        return -1;
        
        
        
    }
};
