class Solution {
public:
    int minSetSize(vector<int>& arr) {
        map<int,int> mp;
        for(int a:arr){
            mp[a]++;
        }
        int n=arr.size()/2;
        vector<int> v;
        for(map<int,int>::iterator it=mp.begin();it!=mp.end();it++){
            v.push_back(it->second);
            
        }
        sort(v.begin(),v.end());
        reverse(v.begin(),v.end());

        int sum=0;
        for(int i=0;i<v.size();i++){
            sum+=v[i];
            if(sum>=n){
                return i+1;
            }
        }
        return 0;
    }
};
