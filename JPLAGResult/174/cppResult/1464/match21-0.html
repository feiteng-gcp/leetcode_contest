class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int n = arr.size();
        if(n <= 2)
            return n - 1;
        unordered_map<int, int> fm;
        for(int a : arr) {
            fm[a]++;
        }
        
        vector<int> fv;
        for(auto it : fm) {
            fv.push_back(it.second);
        }
        sort(fv.begin(), fv.end(), greater<int>() );
        int cnt = 0;
        int res = 0;
        for(int i = 0; i < fv.size(); i++) {
            cnt += fv[i];
            res++;
            if(cnt >= n/2)
                return res;
        }
        return res;
    }
};
