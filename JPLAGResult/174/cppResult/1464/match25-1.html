class Solution {
public:
    int minSetSize(vector<int>& arr) {
        unordered_map<int, int> v2c;
        for (auto val: arr) {
            v2c[val]++;
        }
        
        vector<pair<int, int>> digits;
        for (auto it = v2c.begin(); it != v2c.end(); ++it) {
            digits.push_back({it->second, it->first});
        }
        
        sort(digits.begin(), digits.end());
        

        int res = 0, cSum = 0;
        for (int i = digits.size() - 1; i >= 0; --i) {
            cSum += digits[i].first; res += 1;
            if (cSum >= arr.size() / 2) {return res;}
        }
        
        return digits.size();
        
    }
};
