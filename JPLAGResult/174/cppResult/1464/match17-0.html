class Solution {
public:
    int minSetSize(vector<int>& arr) {
        unordered_map<int, int> mmap;
        for (int n : arr)
            mmap[n]++;
        vector<int> vec;
        for (auto iter = mmap.begin(); iter != mmap.end(); iter++)
            vec.push_back(iter->second);
        
        sort(vec.begin(), vec.end());
        int res = 0;
        int sum = 0;
        int size = arr.size() / 2;
        for (int i = vec.size() - 1; i >= 0; i--)
        {
            res++;
            sum += vec[i];
            if (sum >= size)
                return res;
        }
        
        return 0;
    }
};
