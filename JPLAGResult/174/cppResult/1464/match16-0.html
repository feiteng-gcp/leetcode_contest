class Solution {
public:
    int minSetSize(vector<int>& arr) {
        unordered_map<int,int> m;
        for(int a:arr) m[a]++;
        vector<int> v;
        for(auto p:m){
            v.push_back(p.second);
        }
        sort(v.begin(),v.end());
        int ans=0;
        int c=0;
        for(int i=v.size()-1;i>=0;i--){
            c+=v[i];
            ans++;
            if(c>=(arr.size()+1)/2) return ans;
        }
        return ans;
    }
};
