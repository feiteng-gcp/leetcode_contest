class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int i, n = arr.size();
        sort(arr.begin(), arr.end());
        vector<int>a;
        int cnt = 1;
        for (i = 0; i < n; i++) {
            if (i == n - 1 || arr[i] != arr[i + 1]) {
                a.emplace_back(-cnt);
                cnt = 1;
            } else {
                ++cnt;
            }
        }
        sort(a.begin(), a.end());
        int tot = 0;
        for (i = 0; i < a.size(); i++) {
            if (tot >= n / 2)
                break;
            tot -= a[i];
        }
        return i;
    }
};
