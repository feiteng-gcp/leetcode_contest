class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int cntArray[100000+5]={0};

        memset(cntArray, 0, sizeof(cntArray));
        int size_arr = arr.size();
        for(int i = 0; i < size_arr; ++i) {
            cntArray[arr[i]]++;
        }
        sort(cntArray, cntArray+100005);
        int tmp_cnt = 0, tmp_sum = 0;
        for(int t = 100000+4; t >= 0; --t) {
            tmp_sum += cntArray[t];
            tmp_cnt++;
            if(tmp_sum >= size_arr / 2) break;
        }
        return tmp_cnt;
        
    }
};
