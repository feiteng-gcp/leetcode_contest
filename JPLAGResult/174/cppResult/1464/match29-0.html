class Solution {
public:
    int minSetSize(vector<int>& arr) {
        unordered_map<int,int> valCount;
        for (int v: arr) ++ valCount[v];
        vector<pair<int,int>> countAndVal;
        for (auto iter = valCount.begin(); iter != valCount.end(); ++ iter) {
            countAndVal.push_back({iter->second, iter->first});
        }
        sort(countAndVal.begin(), countAndVal.end(), greater<pair<int,int>>());
        int total = arr.size();
        int selected = 0;
        int sum = 0;
        while (sum * 2 < total) {
            sum += countAndVal[selected].first;
            ++ selected;
        }
        return selected;
    }
};
