class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int i, n = arr.size();
        sort(arr.begin(), arr.end());
        vector<int>res;
        int cnt = 1;
        for (i = 0; i < n; i++) {
            if (i == n - 1 || arr[i] != arr[i + 1]) {
                res.emplace_back(-cnt);
                cnt = 1;
            } else {
                ++cnt;
            }
        }
        
        
        sort(res.begin(), res.end());
        int tmp = 0;
        for (i = 0; i < res.size(); i++) {
            if (tmp >= n / 2)
                break;
            tmp -= res[i];
        }
        return i;
    }
};
