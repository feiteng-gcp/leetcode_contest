class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int n = arr.size();
        map<int, int> mps;
        for(int i=0; i<n; i++){
            mps[arr[i]] ++;
        }
        
        vector<pair<int,int>> nums;
        for(auto k:mps){
            nums.push_back(make_pair(k.second, k.first));
        }
        sort(begin(nums), end(nums));
        
        int t = n/2;
        int ret = 0;
        int cur = 0;
        for(int i=nums.size()-1; i>=0; i--){
            cur += nums[i].first;
            ret += 1;
            if(cur >= t) return ret;
        }
        return 0;
    }
};
