class Solution {
public:
    int minSetSize(vector<int>& arr) {
        map<int, int> dp;
        for(auto a:arr)
        {
            dp[a]++;
        }
        vector<pair<int, int>> vp;
        for(auto x:dp)
        {
            vp.push_back({x.second, x.first});
        }
        sort(vp.begin(), vp.end());
        int n = vp.size();
        int h = arr.size()/2;
        int sum = 0;
        int res = 0;
        for(int i=n-1;i>=0;i--)
        {
            sum+=vp[i].first;
            res++;
            //cout<<vp[i].first<<", "<<vp[i].second<<"," <<res<<endl;
            if(sum>=h){
                return res;
            }
        }
        return -1;
    }
};
