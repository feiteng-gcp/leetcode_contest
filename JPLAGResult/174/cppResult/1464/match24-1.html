class Solution {
public:
    int minSetSize(vector<int>& arr) {
        int cnt0 =arr.size();
        // long long sum = accumulate(begin(arr), end(arr), 0ll);

        unordered_map<int,int> ff;
        for(int i:arr){
            ff[i]++;
        }
        vector<int> n;
        for(auto [i,j]: ff){
            n.push_back(j);
        }
        sort(begin(n), end(n), greater<int>());
        int cnt = 0;
        for(int i=0;i<n.size();i++){
            cnt += n[i];
            if(cnt>=cnt0/2) return i+1;
        }
        return -1;
    }
};
