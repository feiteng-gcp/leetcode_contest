class Solution {
public:
    long long sum = 0,cz = 1000000000000ll;
    void dfs1(TreeNode *root){
        if(root == NULL) return;
        sum += root->val;
        dfs1(root->left);
        dfs1(root->right);
    }
    long long dfs2(TreeNode *root){
        if(root == NULL) return 0;
        long long he = dfs2(root->left) + dfs2(root->right) + root->val;
        long long hs = sum - he;
        if(abs(sum - he - he) < cz) cz = abs(sum - he - he);
        return he;
    }
    int maxProduct(TreeNode* root) {
        dfs1(root);
        dfs2(root);
        long long l = (sum - cz) / 2;
        long long r = sum - l;
        long long ans = l * r % 1000000007;
        return ans;
    }
};
