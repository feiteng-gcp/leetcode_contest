/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    long long mx = 0;
    long long sum = 0;
    const int mod = 1e9 + 7;
    void dfs1(TreeNode* cur){
        if(!cur)return;
        sum += cur -> val;
        dfs1(cur -> left);
        dfs1(cur -> right);
    }
    long long dfs2(TreeNode* cur){
        if(!cur) return 0;
        long long ret = cur -> val;
        ret += dfs2(cur -> left);
        ret += dfs2(cur -> right);
        mx = max(mx, ret * (sum - ret));
        return ret;
    }
    int maxProduct(TreeNode* root) {
        dfs1(root);
        dfs2(root);
        int ret = mx % mod;
        return ret;
    }
};
