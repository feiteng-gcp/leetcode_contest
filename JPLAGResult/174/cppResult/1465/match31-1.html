/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    const long long mo = 1e9 + 7;
    unordered_map<TreeNode*, long long> sum;
    
    int maxProduct(TreeNode* root) {
        sum[NULL] = 0;
        long long tot = dfs(root);
        

        long long ans = 0;
        dfs2(root, tot, ans);
        return ans % mo;
    }
    
    int dfs(TreeNode* root) {
        if (!root)
            return 0;
        
        sum[root] = root->val + dfs(root->left) + dfs(root->right);
        return sum[root];
    }
    
    void dfs2(TreeNode* cur, long long tot, long long &ans) {
        if (!cur)
            return;
        
        if (cur->left) {
            long long tmp = sum[cur->left];
            ans = max(ans, tmp * (tot - tmp));
        }
        if (cur->right) {
            long long tmp = sum[cur->right];
            ans = max(ans, tmp * (tot - tmp));
        }
        
        dfs2(cur->left, tot, ans);
        dfs2(cur->right, tot, ans);
    }
};
