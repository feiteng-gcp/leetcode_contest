/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
const int mod = 1000000007;
class Solution {
public:
    long long res, tot;
    long long dfs(TreeNode *rt, int col) {
        if (!rt) {
            return 0;
        }

        long long t = rt -> val;
        t += dfs(rt -> left, col);
        t += dfs(rt -> right, col);
        if (col) {
            res = max(res, t * (tot - t));
        }
        return t;
    }
    int maxProduct(TreeNode* root) {
        res = -1;
        tot = dfs(root, 0);
        dfs(root, 1);
        return res % mod;
    }
};
