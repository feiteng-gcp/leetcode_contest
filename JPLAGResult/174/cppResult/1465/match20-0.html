/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    #define ll long long
    const ll mod=  1e9 + 7 ;
    vector<ll>list;
    ll dfs(TreeNode* u)
    {
        ll ans = u->val;
        if(u->left !=NULL) ans+=dfs(u->left);
        if(u->right !=NULL) ans+=dfs(u->right);
        list.push_back(ans);
        return ans;
    }
    int maxProduct(TreeNode* root) {
        ll tot = dfs(root);
        ll ans=0;
        for(auto x: list)
        {
            cout<<tot<<' '<<x<<' '<<x*(tot-x)<<endl;
            ans = max(ans, x*(tot-x));
        }
        return int(ans%mod);
    }
};
