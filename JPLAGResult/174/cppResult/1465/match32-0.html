/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    long long getS(TreeNode *root,map<TreeNode *,long long> &m){
        if(root == NULL)
            return 0;
        long long t= root->val + getS(root->left,m) + getS(root->right,m);
        m[root] = t;
        return t;
    }
    int maxProduct(TreeNode* root) {
        
        long long ans = 0;
        map<TreeNode *,long long> m;
        long long sum = getS(root,m);
        for(auto p:m){
            if(p.first != root){
                ans = max(ans,(sum - p.second) * p.second);
            }
        }
        return ans % (1000000000 + 7);
    }
};
