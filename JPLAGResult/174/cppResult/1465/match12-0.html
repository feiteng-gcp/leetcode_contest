/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

long long sum,ans;

void go(TreeNode* u)
{
    if(!u) return;
    
    sum+=u->val;
    
    go(u->left);
    go(u->right);
}


long long go_ans(TreeNode* u)
{
    if(!u) return 0LL;
    
    long long cur=u->val;
    
    cur+=go_ans(u->left);
    cur+=go_ans(u->right);
    
    ans=max(ans,cur*(sum-cur));
    
    return cur;
}




class Solution {
public:
    int maxProduct(TreeNode* root) {
        
        sum=0;
        ans=0;
        
        go(root);
        go_ans(root);
        
        return ans%1000000007;
    }
};
