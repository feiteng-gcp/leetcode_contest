typedef long long ll;

const int mod = (int)1e9 + 7;

class Solution {
public:
    vector<ll> sums;


    ll dfs(TreeNode *p) {
        ll sum = p->val;
        if (p->left != NULL) sum += dfs(p->left);
        if (p->right != NULL) sum += dfs(p->right);
        sums.push_back(sum);
        return sum;
    }

    int maxProduct(TreeNode* root) {
        sums.clear();
        ll total = dfs(root);
        ll ans = 0LL;
        for (auto x : sums)
            ans = max(ans, (ll)x * (total - x));
        return ans % mod;
    }
};
