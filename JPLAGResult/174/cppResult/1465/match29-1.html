/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

long long total, ans;


void dfs1(TreeNode* a) {
    if (!a)return;
    total += a->val;
    dfs1(a->left);
    dfs1(a->right);
}

long long dfs2(TreeNode* a) {
    if (!a)return 0;
    long long re = a->val;
    re += dfs2(a->left);
    re += dfs2(a->right);
    if (re != total) ans = max(ans, re * (total - re));
    return re;
}

class Solution {
public:
    int maxProduct(TreeNode* root) {
        total = 0;
        dfs1(root);
        ans = 0;
        dfs2(root);
        return ans % 1000000007;
    }
};
