#define MOD 1000000007

typedef long long LL;

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    set<LL> S;
    LL recur(TreeNode* root) {
        if (root == NULL) return 0;

        LL x = root->val + recur(root->left) + recur(root->right);
        S.insert(x);
        return x;
    }
    int maxProduct(TreeNode* root) {
        LL ans = 0;
        LL tot = recur(root);
        for (auto x : S) ans = max(ans, x * (tot - x));
        return ans % MOD;
    }
};
