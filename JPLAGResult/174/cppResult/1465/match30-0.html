/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
	long long sum;
	long long res;
	long long ref = 1000000007;
	void dfs(TreeNode* root) {
		if (!root) return;
		sum += root->val;
		dfs(root->left);
		dfs(root->right);
	}
	long long dfs2(TreeNode* root) {
		if (!root) return 0;
		long long left = dfs2(root->left);
		long long right = dfs2(root->right);
		long long temp = left + right + root->val;
		long long nt = temp * (sum - temp);
		res = max(res, nt);
		return temp;
	}
public:
	int maxProduct(TreeNode* root) {
		sum = 0;
		res = 0;
		dfs(root);
		dfs2(root);
		return res % ref;
	}
};
