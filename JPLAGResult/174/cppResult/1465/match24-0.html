/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void getsum (TreeNode *root) {
        if (root->left) {
            getsum(root->left);
            root->val += root->left->val;
        }
        if (root->right) {
            getsum(root->right);
            root->val += root->right->val;
        }
    }
    long long solve (TreeNode *root, int &total) {
        long long ret = LLONG_MIN;
        if (root->left) ret = max(ret,solve(root->left,total));
        if (root->right) ret = max(ret,solve(root->right,total));
        if (root->val != total) ret = max(ret,(root->val)*1LL*(total-(root->val)));
        return ret;
    }
    int maxProduct(TreeNode* root) {
        getsum(root);
        int total = root->val;
        return solve(root,total)%1000000007;
    }
};
