/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    long long res = 0;    
    long long sum = 0;    
    
    void search(TreeNode* cur)
    {
        sum += cur->val;
        if (cur->left)
            search(cur->left);
        if (cur->right)
            search(cur->right);        
    }
    
    long long calc(TreeNode* cur)
    {
        if (!cur->left && !cur->right)
            return cur->val;
        
        long long tmp = cur->val;
        if (cur->right)
        {
            long long tmp0 = calc(cur->right);
            res = max(res, (sum - tmp0) * tmp0);
            tmp += tmp0;
        }
            
        if (cur->left)
        {
            long long tmp0 = calc(cur->left);
            res = max(res, (sum - tmp0) * tmp0);
            tmp += tmp0;
        }
        return tmp;
    }
    
    int maxProduct(TreeNode* root) {        
        search(root);
        calc(root);
        return (res % 1000000007);
    }
};
