/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxProduct(TreeNode* root) {
        long long res=0;
        sum(root);
        help(root,res,root->val);
        return res%1000000007;
    }
    int sum(TreeNode * root){
        if(!root)
            return 0;
        root->val+=sum(root->left)+sum(root->right);
        root->val;
        return root->val;
    }
    void help(TreeNode* root,long long &res,int s){
        if(!root)
            return ;
        res=max((long long)root->val*(s-root->val),res);
        help(root->left,res,s);
        help(root->right,res,s);
    }
};
