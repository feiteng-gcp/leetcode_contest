/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    long minn = INT_MAX;
    long maxSum = 0, res = 0;
    const int MOD = 1e9 + 7;
    int maxProduct(TreeNode* root) {
        maxSum = sumOfTree(root);
        helper(root);
        long ans = ((maxSum - res) * res)% MOD;
        return ans;
    }
    int sumOfTree(TreeNode *root){
        if (root == NULL) return 0;
        int summ = root->val + sumOfTree(root->left) + sumOfTree(root->right);
        return summ;
    }
    int helper(TreeNode *root){
        if (root == NULL) return 0;
        int summ = root->val + helper(root->left) + helper(root->right);
        if (minn > abs(maxSum / 2 - summ))
        {
            res = summ;
            minn = abs(maxSum / 2 - summ);
        }
        return summ;
    }
};
