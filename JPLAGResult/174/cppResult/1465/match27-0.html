/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    long long total = 0;
    long long max = -1;
    void find(TreeNode* root) {
        if (root == NULL) return ;
        total += root->val;
        find(root->left);
        find(root->right);
    }

    int get(TreeNode* root) {
        if (root == NULL) return 0;
        
        long long count = root->val;
        count += get(root->left);
        count += get(root->right);
        
        long long temp = count * (total - count);
        if (temp > max) max = temp;
        
        return count;
    }
    
    
    int maxProduct(TreeNode* root) {
        find(root);
        //cout << total << endl;
        get(root);
        return max % (1000000007);
    }
};
