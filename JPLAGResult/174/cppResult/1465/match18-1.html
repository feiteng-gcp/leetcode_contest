long long mod = 1e9 + 7;
class Solution {
public:
    long long dfs1(TreeNode *u) {
        if(u == NULL) return 0;
        long long ans = u->val;
        ans += dfs1(u->left);
        ans += dfs1(u->right);
        return ans;

    }

    long long dfs2(TreeNode *u, long long sum, long long &ans) {
        if(u == NULL) return 0;
        long long l = dfs2(u->left, sum, ans);
        long long r = dfs2(u->right, sum, ans);
        
        ans = max(ans, l * (sum - l));
        ans = max(ans, r * (sum - r));

        return l + r + u->val;
    }

    int maxProduct(TreeNode* root) {
        long long sum = dfs1(root);        
        long long ans = 0;
        dfs2(root, sum, ans);
        return ans % mod;
    }
};
