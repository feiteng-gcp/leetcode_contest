/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    const long long mod = 1e9 + 7;
    int total = 0;
public:
    int maxProduct(TreeNode* root) {
        total = getTotal(root);
        int MAX = 0;
        int res = getSum(root, MAX);
        long long r = (long long)MAX * ((long long)total - MAX) % mod;
        return (int)r;
    }
    
private:
    int getSum(TreeNode* root, int& MAX) {
        if (root == NULL)    return 0;
        int sum = root->val + getSum(root->left, MAX) + getSum(root->right, MAX);
        // cout<<sum<<endl;
        if (sum <= total / 2)    MAX = max(MAX, sum);
        if (total - sum <= total / 2)    MAX = max(MAX, total - sum);
        return sum;
    }
    
    int getTotal(TreeNode* root) {
        int sum = 0;
        queue<TreeNode*> que;
        que.push(root);
        while (!que.empty()) {
            auto out = que.front();    que.pop();
            sum += out->val;
            if (out->left != NULL)    que.push(out->left);
            if (out->right != NULL)    que.push(out->right);
        }
        return sum;
    }
    
    
};
