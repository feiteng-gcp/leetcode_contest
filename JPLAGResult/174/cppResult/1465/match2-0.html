/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    long long total = 0;
    unordered_map<TreeNode *, long long> m;
    long long res = INT_MIN;
    long long MOD = 1e9 + 7;
    
public:
    int maxProduct(TreeNode* root) {
        helper(root);
        //cout << total << endl;
        helper2(root->left);
        helper2(root->right);
        return res % MOD;
    }
    
    int helper(TreeNode *root)
    {
        if(!root) return 0;
        total += root->val;
        auto l = helper(root->left);
        auto r = helper(root->right);
        m[root] = l + r + root->val;
        return m[root];
    }
    
    void helper2(TreeNode *root)
    {
        if(!root) return;
        //cout << root->val << endl;
        res = max(res, (total - m[root]) * m[root]);
        //cout << res << " " << total - m[root] << " " << m[root] << endl;
        helper2(root->left);
        helper2(root->right);
        return;
    }
};
