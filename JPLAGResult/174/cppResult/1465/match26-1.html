/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
const int mod = 1e9 + 7;
class Solution {
public:
    int maxProduct(TreeNode* root) {
        vector<long> arr;
        long sum = getSum(root, arr);

        long res = 0L;
        for ( auto x : arr ) {
            res = max(res, x * (sum - x));
        }
        return res % mod;
    }
    
    long getSum(TreeNode* root, vector<long>& arr) {
        if ( root == nullptr ) return 0;
        long res =  (long)(root -> val) + getSum(root -> left, arr) + getSum(root -> right, arr);
        arr.push_back(res);
        return res;
    }
};
