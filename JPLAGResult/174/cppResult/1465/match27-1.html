/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    long long Sum,Ans;

    void dfs(TreeNode *x)
    {
        if (x==NULL) return ;
        Sum+=x->val;
        dfs(x->left);
        dfs(x->right);
    }
    int dfs2(TreeNode *x)
    {
        if (x==NULL) return 0;
        long long sum=dfs2(x->left)+dfs2(x->right)+x->val;
        Ans=max(Ans,(Sum-sum)*sum);
        return sum;
    }
public:
    int maxProduct(TreeNode* root) {
        dfs(root);
        dfs2(root);
        int MOD=1e9+7;
        return Ans%MOD;
    }
};
