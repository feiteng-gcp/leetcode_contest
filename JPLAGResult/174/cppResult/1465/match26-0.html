/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    unordered_set<long long> m;
    long long all;
    int maxProduct(TreeNode* root) {
        all = getsum(root);
        long long mm = 0;
        for (auto& i: m) {
            mm = max(mm, i*(all -i));
        }
        return mm % 1000000007;
    }
    long long getsum(TreeNode* root) {
        if(!root) return 0;
        long long t = root->val + getsum(root->left) + getsum(root->right);
        m.insert(t);
        return t;
    }
};
