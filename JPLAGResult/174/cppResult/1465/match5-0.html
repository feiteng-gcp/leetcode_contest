/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    typedef long long ll;
    ll ans=0;
    void dfs(TreeNode* rt){
        if(!rt)return;
        ans+=rt->val;
        dfs(rt->left);
        dfs(rt->right);
    }
    ll ret=0;
    ll dfs1(TreeNode* rt){
        if(!rt)return 0;
        ll l=dfs1(rt->left);
        ll r=dfs1(rt->right);
        ret=max(ret,(ans-(l+r+rt->val))*(l+r+rt->val));
        return (l+r+rt->val);
    }
    int maxProduct(TreeNode* root) {
        dfs(root);
        dfs1(root);
        ll md=1e9+7;
        return ret%md;
    }
};
