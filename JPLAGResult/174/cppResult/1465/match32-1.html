/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    unordered_map<TreeNode*, long long> sum;
    int dfsHelper(TreeNode* root) {
        if (!root) return 0;
        sum[root] = dfsHelper(root->left) + dfsHelper(root->right) + root->val;
        return sum[root];
    }
public:
    int maxProduct(TreeNode* root) {
        dfsHelper(root);

        long long half = 0;
        for (auto& kv : sum) {
            if ((sum[root] - kv.second) * kv.second > (sum[root] - half) * half) {
                half = kv.second;
            }
        }
        return (sum[root] - half) * half % 1000000007;
    }
};
