/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    static const int mod = 1e9 + 7;
    long long ans = 0;

    int sum(TreeNode *x) {
        if(!x) return 0;
        return x->val + sum(x->left) + sum(x->right);
    }
    int solve(TreeNode *x, int tot) {
        if(!x) return 0;
        int cur = x->val + solve(x->left, tot) + solve(x->right, tot);
        ans = max(ans, 1LL * cur * (tot - cur));
        return cur;
    }
    int maxProduct(TreeNode* root) {
        int tot = sum(root);
        solve(root, tot);
        return ans % mod;
    }
};
