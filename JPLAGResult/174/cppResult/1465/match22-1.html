/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
typedef long long ll;
class Solution {
public:

    ll compute(vector<ll>& cnt, TreeNode* root) {
        ll sum = root->val;
        if (root->left != NULL) sum += compute(cnt, root->left);
        if (root->right != NULL) sum += compute(cnt, root->right);
        
        cnt.push_back(sum);
        return sum;
    }
    int maxProduct(TreeNode* root) {
        vector<ll> cnt;
        ll ans = 0;
        ll total = compute(cnt, root);
        for (ll val : cnt) {
            ans = max(ans, (total-val)*val);
        }
        return ans%1000000007;
    }
};
