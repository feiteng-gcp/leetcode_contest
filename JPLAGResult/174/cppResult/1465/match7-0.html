/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*, int> count; 
    long long ans=0;
    const int T = 1e9+7;
    
    long long countTree(TreeNode* root) {
        if (root==NULL)
            return 0;
        count[root] = countTree(root->left) + countTree(root->right) + root->val;
        return count[root];
    }
    
    int maxProduct(TreeNode* root) {
        count[root] = countTree(root->left) + countTree(root->right) + root->val;
        
        long long sum=count[root];
        
        queue<TreeNode*> q;
        q.push(root);
        while (!q.empty()) {
            TreeNode* r=q.front();
            q.pop();
            
            if (r->left) {
                q.push(r->left);
                // cout<<sum-count[r->left]<<' '<<count[r->left]<<' '<< sum-count[r->left])*count[r->left]<<endl;
                ans=max(ans, (sum-count[r->left]) * count[r->left]);
            }
            
            if (r->right) {
                q.push(r->right);
                // cout<<count[r->right]<<endl;
                ans=max(ans, (sum-count[r->right]) * count[r->right]);
            }
        }
    
        return ans % T;
    }
};
