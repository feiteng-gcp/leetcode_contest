/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
typedef long long ll;
const ll MOD = 1000000007LL;
class Solution {
public:
    ll calSum(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }
        return root->val + calSum(root->left) + calSum(root->right);
    }
    
    ll solve(TreeNode* root, ll sum) {
        if (root == nullptr) {
            return 0;
        }
        ll lsum = solve(root->left, sum);
        ll rsum = solve(root->right, sum);
        ll cur = root->val + lsum + rsum;
        if (cur != sum) {
            ans = max(ans, 1LL * (sum - cur) * cur);
        }
        return cur;
    }
    int maxProduct(TreeNode* root) {
        ll sum = calSum(root);
        ans = 0;
        solve(root, sum);
        return ans % MOD;
    }
    ll ans;
};
