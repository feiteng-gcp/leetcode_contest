/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    long ma = 0;
    map<TreeNode*, long> m;
    int maxProduct(TreeNode* root) {
       f2(root, f1(root));
		return static_cast<int>(ma % 1000000007);
    }
    
    long f1(TreeNode* root) {
		if (root == nullptr) {
			return 0;
		} else {
			m[root] =  root->val + f1(root->left) + f1(root->right);
			return m[root];
		}
	}

	void f2(TreeNode* root, long sum) {
		if (root != nullptr) {
			ma = max(ma, m[root] * (sum - m[root]));
			f2(root->left, sum);
			f2(root->right, sum);
		}
	}
    
    
    
};
