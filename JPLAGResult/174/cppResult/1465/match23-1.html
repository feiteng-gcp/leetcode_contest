class Solution {
public:
    long long ans,all;
    void d(TreeNode* root) {
       all += root->val;
       if(root->left) d(root->left);
       if(root->right) d(root->right);         
    }
    long long dfs(TreeNode* root) {

       long long now = root->val;
       if(root->left) {
           long long t=dfs(root->left);
           now += t;
           ans = max(ans,t*(all-t));
       }
       if(root->right) {
            long long t=dfs(root->right);
            now +=t;
            ans = max(ans,t*(all-t));           
       }
       return now;
    }
    int maxProduct(TreeNode* root) {
        ans = all = 0;
        const int M = 1e9+7;
        d(root);
        dfs(root);
        return ans%M;
    }
};
