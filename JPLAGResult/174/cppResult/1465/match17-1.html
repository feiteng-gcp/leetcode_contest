/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    long long prime = 1000000007;
    long long sum = 0;
    long long ans = 0;
    
    void get_sum(TreeNode* node) {
        if (!node) return;
        sum += node->val;
        get_sum(node->left);
        get_sum(node->right);
    }
    
    long long dfs(TreeNode* node) {
        if (!node) return 0;
        long long tree_sum = dfs(node->left) + dfs(node->right) + node->val;
        ans = max(ans, (sum - tree_sum) * tree_sum);
        return tree_sum;
    }
    
    int maxProduct(TreeNode* root) {
        get_sum(root);
        dfs(root);
        return ans % prime;
    }
};
