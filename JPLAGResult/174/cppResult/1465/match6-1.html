/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
#define MOD 1000000007

class Solution {
public:
    unsigned long DFS_sum(TreeNode *root, map<TreeNode *, unsigned long> &ms) {
        if (!root) return 0;
        unsigned long ans = DFS_sum(root->left, ms) + DFS_sum(root->right, ms) + root->val;    
        ms[root] = ans;
        return ans;
    }
    
    int maxProduct(TreeNode* root) {
        map<TreeNode *, unsigned long> ms;
        unsigned long sum = DFS_sum(root, ms);
        unsigned long maxP = 0;
        
        for (auto &e: ms) {
            unsigned long sonSum = e.second;
            unsigned long otherSum = (sum - sonSum);
            unsigned long prod = (sonSum * otherSum);
            maxP = max(maxP, prod);
        }
        
        return maxP % MOD;
    }
};
