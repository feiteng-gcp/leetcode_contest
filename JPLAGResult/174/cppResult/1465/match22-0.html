/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
typedef long long ll;
const ll mod = 1e9+7;

class Solution {
public:
    int k = 0;
    vector<ll> d;
    
    ll f(TreeNode* t){
        ll sum = t->val;
        if(t->left != NULL)     sum += f(t->left);
        if(t->right != NULL)    sum += f(t->right);
        d.push_back(sum);
        return sum;
    }
    
    int maxProduct(TreeNode* root) {
        ll sum = f(root);
        ll ret = 0;
        for(ll x : d)   ret = max(ret, x*(sum-x));
        return ret%mod;
    }
};
