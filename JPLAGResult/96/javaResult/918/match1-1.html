class Solution {
    public int reachableNodes(int[][] edges, int M, int N) {
        Node[] nodes = new Node[N];
        int inf = (int)1e9 + 1;
        for(int i = 0; i < N; i++)
        {
            nodes[i] = new Node();
            nodes[i].dist = inf;
        }
        
        List<Edge> edgeList = new ArrayList(edges.length);
        for(int[] edge : edges)
        {
            Node a = nodes[edge[0]];
            Node b = nodes[edge[1]];
            
            Edge ab = new Edge();
            ab.src = a;
            ab.dst = b;
            ab.length = edge[2];
            a.edgeList.add(ab);
            b.edgeList.add(ab);
            
            edgeList.add(ab);
        }
        
        int count = 0;
        Deque<Node> queue = new ArrayDeque<>(N);
        nodes[0].dist = 0;
        nodes[0].inque = true;
        queue.addLast(nodes[0]);
        while(!queue.isEmpty())
        {
            Node head = queue.removeFirst();
            head.inque = false;
            for(Edge edge : head.edgeList)
            {
                Node other = edge.getOther(head);
                
                edge.setLength(head, M - head.dist);
                
                if(M - head.dist < edge.length + 1)
                {
                    continue;
                }
                
                if(head.dist + edge.length + 1 >= other.dist)
                {
                    continue;
                }
                other.dist = head.dist + edge.length + 1;
                if(other.inque)
                {
                    continue;
                }
                
                other.inque = true;
                queue.addLast(other);
            }
        }
        
        
for(Node node : nodes)
        {
            if(node.dist != inf)
            {
                count++;
            }
        }
        
        
        for(Edge edge : edgeList)
        {
            count += Math.min(edge.length, edge.srcLength + edge.dstLength);
        }
        
        return count;
    }
    
    public static class Node{
        List<Edge> edgeList = new ArrayList(1);
        int dist;
        boolean inque;
    }
    
    public static class Edge{
        Node src;
        Node dst;
        int length;
        int srcLength;
        int dstLength;
        
        public Node getOther(Node me)
        {
            return src == me ? dst : src;
        }
        
        public void setLength(Node me, int length)
        {
            if(me == src)
            {
                srcLength = Math.max(srcLength, length);
            }
            else
            {
                dstLength = Math.max(dstLength, length);
            }
        }
    }
}
