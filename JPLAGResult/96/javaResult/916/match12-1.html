class Solution {
    public String decodeAtIndex(String S, int K) {
      
        return aaaa(S, K);
     
    }
    public String aaaa(String S, int K){
        long k = K;
            char[] a = S.toCharArray();
        char[][] s = new char[101][101];
        long[][] b = new long[102][2];
        long[] c = new long[102];
        int n = S.length();
        String ans = "";
        int l = 0;
        for(int i = 0; i < n; i++){
            if(a[i] <'2' || a[i]>'9'){
                l++;
                b[l][0] = 1;
                int z = 0;
                s[l][z] = a[i];
                int j = i+1;
                for(; j < n; j++){
                    if(a[j]>='2' && a[j]<='9') break;
                    z++;
                    s[l][z] = a[j];
                }
                b[l][1] = j-i;
                i = j-1;
            }
            else{
                l++;
                b[l][0] = 2;
                b[l][1] = a[i]-'0';
            }
        }
        long  max = 0;
        int i = 1;
        for(; i <= l; i++){
           if(b[i][0] == 1){
               if(max + b[i][1] >= k){
                    k -= max;
                   k--;
                    ans = ans + s[i][(int)k];
                   return ans;
               }
               else max+=b[i][1];
               
           }else{
               if(max * b[i][1] >= k){
                   if(k % max == 0){
                       for(int j = i-1; j >= 1; j--){
                           if(b[j][0] == 1){
                               ans = ans + s[j][(int)(b[j][1]-1)] ;
                               return ans;
                           }
                       }
                   }
                   else{
                       
                       k = k%max;
                       
                       return aaaa(S, (int)k);
                   }
               }
               else
                max = max*b[i][1]
;
           }
        }
        return ans;
    }
}
