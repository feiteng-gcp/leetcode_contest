	class Solution {
	    public String decodeAtIndex(String S, int K) {
	        char[] s = S.toCharArray();
	        int n = s.length;
	        long[] dp = new long[n+1];
	        for(int i = 0;i < n;i++){
	        	if(s[i] >= '2' && s[i] <= '9'){
	        		dp[i+1] = dp[i] * (s[i]-'0');
	        	}else{
	        		dp[i+1] = dp[i] + 1;
	        	}
	        }
	        K--;
	        for(int i = n-1;i >= 0;i--){
	        	K %= dp[i+1];
	        	if(K+1 == dp[i+1] && !(s[i] >= '2' && s[i] <= '9')){
	        		return "" + s[i];
	        	}
	        }
	        return null;
	    }
	}
