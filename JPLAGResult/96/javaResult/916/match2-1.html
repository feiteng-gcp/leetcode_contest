class Solution {
    public String decodeAtIndex(String S, int K) {
        long len = 0;
        for(int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if(c >= 'a' && c <= 'z') {
                len++;
                if(len == (long)K) {
                    return String.valueOf(c);
                }
            } else {
                long initialLen = len;
                len = initialLen * (c - '0');
                if(len >= (long)K) {
                    long mul = (long) K / initialLen;
                    if(K - mul * initialLen > 0) {
                        K -= mul * initialLen;
                    } else {
                        K -= (mul-1) * initialLen;
                    }
                    
return decodeAtIndex(S.substring(0, i), K);
                }
            }
        }
        return "";
    }
}
