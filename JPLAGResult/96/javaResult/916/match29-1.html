class Solution {
    public String decodeAtIndex(String S, int k) {
        char[] str = S.toCharArray();
        int ptr = 0, prev = 0;
        while(k > 1){
            char c = str[ptr];
            if(Character.isDigit(c)){
                int count = Character.getNumericValue(c);
                while(count-- > 1){
                    if(k - prev > 0) k-=prev;
                    else return decodeAtIndex(S, k);
                }
                prev = Character.getNumericValue(c)*prev;
            } else {
                prev++; k--;
            }
            ptr++;
        }
        if(Character.isLetter(str[ptr])) return str[ptr]+"";
        else return str[0]+"";
    }
}
