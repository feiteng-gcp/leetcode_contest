class Solution {
    public String decodeAtIndex(String S, int K) {
        int len = S.length();
        long curr = 0;
        int i = 0;
        while (i < len) {
            char c = S.charAt(i++);
            if (c >= 'a' && c <= 'z') {
                curr++;
            } else {
                curr *= c - '0';
            }
            if (curr >= K) {
                break;
            }
        }
        String str = S.substring(0, i);
        int index = str.length() - 1;
        while (Character.isDigit(str.charAt(index)) || curr != K) {
            char c = str.charAt(index--);
            if (c >= 'a' && c <= 'z') {
                curr--;
            } else {
                int count = c - '0';
                curr /= count;
                K = (int) (K % curr == 0 ? curr : K % curr);
            }
        }
        return "" + str.charAt(index);
    }
    
    
}
