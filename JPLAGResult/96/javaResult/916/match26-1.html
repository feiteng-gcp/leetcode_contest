class Solution {
    public String decodeAtIndex(String S, int K) {
        // System.out.println(S + "," + K);
        if(K == 1) return S.charAt(0) + "";
        long len = 0;
        
for(int i = 0; i < S.length(); i++) {
            long clen = len;
            if('2' <= S.charAt(i) && S.charAt(i) <= '9') {
                len *= (S.charAt(i) - '0');
            } else {
                len++;
                if(len == K) return S.charAt(i) + "";
            }
            if(clen <= K && K <= len) {
                long left = (long) K % clen;
                if(left == 0) left = clen;
                return decodeAtIndex(S.substring(0, i + 1), (int) left);
            }
        }
        return null;
    }
}
