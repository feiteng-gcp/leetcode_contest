class Solution {
    public String decodeAtIndex(String S, int K) {
        return dh(S,K);
    }
    
    public String dh(String S, long K){
        long len = 0;
        long Kt = K;
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if(c >= '0' && c <= '9'){
                long x = Integer.parseInt(c + "");
                if(len *  x >= Kt){
                    long mod = (Kt+len)%len;
                    if(mod == 0){
                        mod = len;
                    }
                    return dh(S.substring(0,i),(int)mod);
                }else{
                    len *= x;
                }
            }else{
                len++;
                if(K == len){
                    return c + "";
                }
            }
        }
        
        return "-";
    }
}
