class Solution {
    public String decodeAtIndex(String S, int K) {
        long l = 0;
        for(int i = 0; i < S.length(); i++) {
            if(Character.isLetter(S.charAt(i))) {
                l++;
            }
            else {
                l *= Character.getNumericValue(S.charAt(i));
            }
        }
        
        long K2 = l - K;
        
        for(int i = S.length() - 1; i >= 0; i--) {
            if(Character.isLetter(S.charAt(i))) {
                if(K2 > 0) {
                    K2--;
                    l--;
                }
                else {
                    return "" + S.charAt(i);
                }
            }
            else {
                long len = l/Character.getNumericValue(S.charAt(i));
                if(K2 < len) {
                    l = len;
                }
                else {
                    K2 -= Math.floor(K2/len)*len;
                    l = len;
                }
            }
        }
        return "dude, idk what happened";
    }
}
