class Solution {

    public String decodeAtIndex(String S, int K) {
    
    	long total = 0;
    	int i , length = S.length();
    	for (i = 0;i < length;i ++) {
    		if (S.charAt(i) >= '2' && S.charAt(i) <= '9') {
    			total = total * (int) (S.charAt(i) - '0');
    		} else {
    			total ++;
    		}
    	}
    	return Character.toString(solve(S , length - 1 , total , K));
    	
    }
    
    private char solve(String S , int current , long total , long K) {
    
    	if (S.charAt(current) >= '2' && S.charAt(current) <= '9') {
    		long per = total / (int) (S.charAt(current) - '0');
    		K = K % per;
    		if (K == 0) {
    			K = per;
    		}
    		return solve(S , current - 1 , per , K);
    	} else {
    		if (total == K) {
    			return S.charAt(current);
    		} else {
    			return solve(S , current - 1 , total - 1 , K);
    		}
    	}
    	
    }
}
