class Solution {
    
    public String decodeAtIndex(String S, int K) {
        if (K == 1) return S.substring(0,1);
        
        --K;
        
        int n = S.length();
        long len = 0;
        for (int i = 0; i < n; i++) {
            char c = S.charAt(i);
            if (c > '0' && c <= '9') {
                long l = len * (c - '0');
                if (l > K) return decodeAtIndex(S.substring(0, i), 1 + K % (int)len);
                
                len = l;
            }else {
                // a letter
                if (len == K) return "" + c;
                ++len;                
            }
        }
        return "";
    }
}
