class Solution {
    public String decodeAtIndex(String S, int K) {
    		long len = 0;
    		String last = S.substring(0, 1);
        for(int i = 0; i < S.length(); ++i) {
        		long pre = len;
        		if(S.charAt(i) >= '0' && S.charAt(i) <= '9') {
        			len *= S.charAt(i) - '0';
        		} else {
        			len++;
        			last = S.substring(i, i + 1);
        		}
        		if(len >= K){
                    if(len == K) return last;
                    if(K % pre == 0) return decodeAtIndex(S.substring(0, i), (int)pre);
                    else return decodeAtIndex(S.substring(0, i), (int)(K % pre));
                }
        }
        return last;
    }
}
