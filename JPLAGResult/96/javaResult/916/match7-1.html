class Solution {
    public String decodeAtIndex(String S, int K) {
    	long count = 0;
    	char lastLetter = 'z';
    	for (int
 i = 0; i < S.length(); i++) {
    		if (S.charAt(i) >= '2' && S.charAt(i) <= '9') {
    			int digit = S.charAt(i) - '2' + 2;
    			long total = count * digit;
    			if (total >= K) {
    				long r = K % count;
    				if (r == 0) {
    					return lastLetter + "";
    					// r = count;
    				}
    				return decodeAtIndex(S, (int) r);
    			}
    			count = total;
    			
    		} else {
    			lastLetter = S.charAt(i);
    			count++;
    			if (count == K) {
    				return lastLetter + "";
    			}

    		}
    	}
   	    return "z";
	}
}
