class Solution {
    long[] dp;
    String SS;
    public String decodeAtIndex(String S, int K) {
        if (S == null || S.length() == 0) {
            return S;
        }
        SS = S;
        int len = S.length();
        dp = new long[len + 1];
        
        for (int i = 0; i < len; i++) {
            char c = S.charAt(i);
            if (Character.isDigit(c)) {
                dp[i + 1] = dp[i] * (c - '0'); 
            } else {
                dp[i + 1] = dp[i] + 1;
            }
          //  System.out.print(dp[i + 1] + " ");
            if (dp[i + 1] >= K) {
                break;
            }
        }
        
        return "" + find(K);
    }
    
    private char find(long k) {
        int len = dp.length;
        int ith = -1;
        for (int i = 0; i < len; i++) {
            if (dp[i] >= k) {
                ith = i;
                break;
            }
        }
        if (k == 1) {
            return SS.charAt(ith - 1);
        }
        char c = SS.charAt(ith - 1);
        if (Character.isDigit(c)) {
            long prevCnt = dp[ith - 1];
            int ntimes = c - '0';
            long newK = ((k % prevCnt - 1 + prevCnt) % prevCnt) + 1;
            return find(newK);
        } 
        

        return SS.charAt((int)(ith - 1 - (dp[ith] - k)));
    }
    
    
}
