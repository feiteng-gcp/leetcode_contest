class Solution {
    public String decodeAtIndex(String S, int K) {
        int n = S.length();
        int count = 0;
        char lastC = '0';
        for (int i = 0; i < n; i++) {
            char c = S.charAt(i);
            if (Character.isDigit(c)) {
                int d = c - '0';
                for (int j = 2; j <= d; j++) {
                    int newCount = count * j;
                    if (newCount == K) {
                        return lastC + "";
                    }
                    if (newCount > K) {
                        int newK = K - (newCount - count);
                        return decodeAtIndex(S, newK);
                    }
                }
                count *= d;
            } else {
                lastC = c;
                count += 1;
                if (count == K) {
                    return c + "";
                
}
            }
        }
        return null;
    }

}
