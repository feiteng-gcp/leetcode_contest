class Solution {
    public String decodeAtIndex(String S, long K) {

        if (K == 0) {

            return "" + S.charAt(0);
        }
        
String prevS = "";

        long length = 0;
        for (int i = 0; i < S.length(); i++) {
            Character c = S.charAt(i);

            if (Character.isDigit(S.charAt(i))) {
                long newLength = length * Integer.parseInt("" + S.charAt(i));
                if (newLength >= K) {
                    return decodeAtIndex(S, K - length);
                } else if (newLength == K) {
                    return prevS;

                } else {
                    length = length * Integer.parseInt("" + S.charAt(i));
                }


            } else {
                length++;
                prevS = S.charAt(i) + "";
            }

            if (K == length) {

                return "" + 
S.charAt(i);
            }

        }

        return "";

    }
}
