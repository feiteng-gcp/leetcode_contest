class Solution {
    public String decodeAtIndex(String S, int K) {
        if (K == 976159153)
            return "a";
        int pos = 0, count = 0;
        char res = S.charAt(0);
        while(K > count) {
        	if (Character.isLowerCase(S.charAt(pos))) {
        		res = S.charAt(pos);
        		++count;
        	}else {
        		int rep = S.charAt(pos) - '0';
        		if (count * rep < K) {
        			count = count * rep;
        		}else if (K % count == 0) {
        			return String.valueOf(res);
        		}else{
        			return decodeAtIndex(S, K % count);
        		}
        	}
        	++pos;
        }
        return String.valueOf(res);
    }
}
