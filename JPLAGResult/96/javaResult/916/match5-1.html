class Solution {
    public String decodeAtIndex(String S, int K) {
        
int n = S.length();
        //StringBuilder sb = new StringBuilder();
        
        long count = 0;
        
        for (int i = 0; i < n; i++) {
            char c = S.charAt(i);
            //System.out.println(c + " " + count + " " + sb.toString());
            
            if (c >= 'a' && c <= 'z') {
                //sb.append(c);
                count++;
                if (count == K) return c + "";
            } else {
                int time = c - '0';
                if (K <= count * time) {
                    K = K % ((int)count);
                    if (K == 0) K = (int)count;
                    return decodeAtIndex(S, K);
                } else {
                    count = count * time;
//                     String cur = sb.toString();
                    
//                     for (int j = 0; j < time - 1; j++) {
//                         sb.append(cur); 
//                     }
                    
                
}
            }
        }
        
        return "";
    }
}
