class Solution {
    public String decodeAtIndex(String S, int K) {
        char[] arr = S.toCharArray();
        long k = 1L * (K-1);
        int n = arr.length;
        long[] dp = new long[n];
        for (int i = 1; i < n; ++i){
            if (arr[i] >= '0' && arr[i] <= '9'){
                if (arr[i-1] >= '0' && arr[i-1] <= '9'){
                     dp[i] = dp[i-1]*(long)(arr[i-1] - '0');
                }
                else {
                    dp[i] = (dp[i-1]+1L);
                }
            }else{
                if (arr[i-1] >= '0' && arr[i-1] <= '9'){
                    dp[i] = dp[i-1]*(long)(arr[i-1] - '0');
                }else{
                    dp[i] = dp[i-1] + 1L;
                }
            }
            // System.out.print(dp[i] + " ");
        }
        // System.out.println();
        for (int i = n-1; i>= 0; --i){
            if (arr[i] >= '0' && arr[i] <= '9'){
                k %= dp[i];
            }else{
                if (dp[i] == k) return arr[i] + "";
            }
            // System.out.println(arr[i] + " k: " + k);
        }
        return "";
    }
}
