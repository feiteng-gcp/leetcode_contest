class Solution {
    public String decodeAtIndex(String S, int K) {
        String[] aData = new String[S.length()];
        long[][] dData = new long[S.length()][2];
        int aIdx = 0;
        int dIdx = 0;
        long total = 0;
        StringBuilder strBuilder = new StringBuilder();
        
for (char ch : S.toCharArray()) {
            if (ch >= '0' && ch <= '9') {
                //digit
                aData[aIdx++] = strBuilder.toString();
                dData[dIdx][1] = ch - '0';
                total *= dData[dIdx][1];
                dData[dIdx++][0] = total;
                strBuilder = 
new StringBuilder();
            } else {
                total++;
                strBuilder.append(ch);
            }
        }
        if (strBuilder.length() > 0) {
            aData[aIdx++] = strBuilder.toString();
        }

        if (aIdx > dIdx) {
            if (dIdx >= 1) {
                if (dData[dIdx - 1][0] < K) {
                    K -= dData[dIdx - 1][0];
                    return String.valueOf(aData[aIdx - 1].charAt(K - 1));
                }
            } else {
                return String.valueOf(aData[0].charAt(K - 1));
            }
            aIdx--;
        }

        dIdx--;
        aIdx--;

        while (aIdx >= 0) {
            if (dIdx >= 1) {
                if (dData[dIdx - 1][0] < K) {
                    long subK = K % (dData[dIdx][0] / dData[dIdx][1]);
                    if (subK == 0) {
                        subK = dData[dIdx][0] / dData[dIdx][1];
                    }
                    if (subK > dData[dIdx - 1][0]) {
                        subK -= dData[dIdx - 1][0];
                        return String.valueOf(aData[aIdx].charAt((int) subK - 1));
                    }
                    K = (int) subK;
                }
            } else {
                long subK = K % (dData[dIdx][0] / dData[dIdx][1]);
                if (subK == 0) {
                    subK = dData[dIdx][0] / dData[dIdx][1];
                }
                return String.valueOf(
aData[aIdx].charAt((int) subK - 1));
            }
            dIdx--;
            aIdx--;
        }
        return "";
    }
}
