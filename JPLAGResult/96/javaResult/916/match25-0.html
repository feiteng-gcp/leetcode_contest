class Solution {
    public String decodeAtIndex(String S, int K) {
        if(S==null||K<=0){
            return null;
        }
        long totalLen=0;
        long lenWhenStop=-1;
        int targetIndex=-1;
        for(int index=0;index<S.length();index++){
            char ch=S.charAt(index);
            if(Character.isLetter(ch)){
                totalLen++;
                if(totalLen==K){
                    return ch+"";
                }
            }
            else{
                lenWhenStop=totalLen;
                totalLen=totalLen*(long)(ch-'0');
                if(totalLen>=K){
                    int netLen=(int)(K%lenWhenStop);
                    return decodeAtIndex(S.substring(0,index),netLen==0?(int)lenWhenStop:netLen);
                    //targetIndex=index;
                    //break;
                }
            }
        }
        long multi=(long)(S.charAt(targetIndex)-'0');
        for(long i=1;i<=multi;i++){
            if(i*lenWhenStop>K){
                return decodeAtIndex(S.substring(0,targetIndex),(int)(K-(i-1)*lenWhenStop));
            }
        }
        return null;
    }
}
