class Solution {
    class Node {
        StringBuilder s;
        long rep, len;

        public Node(StringBuilder s, long rep, long len) {
            this.s = s;
            this.rep = rep;
            this.len = len;
        }
    }

    public String decodeAtIndex(String S, int K) {
        K--;
        Node curr = new Node(new StringBuilder(), 1, 0);
        Deque<Node> stk = new ArrayDeque<>();
        stk.push(curr);
        for (char ch : S.toCharArray()) {
            if (ch >= '0' && ch <= '9') {
                curr.rep = ch - '0';
                curr.len *= curr.rep;
                stk.push(curr = new Node( new StringBuilder(), 1, curr.len));
            }
            else {
                curr.s.append(ch);
                curr.len += 1;
            }
        }
        while (!stk.isEmpty()) {
            curr = stk.pop();
            if (K >= curr.len) return "impossible";
            long len = curr.len / curr.rep;
            K %= len;
            long prevLen = len - curr.s.length();
            if (K >= prevLen) {
                return "" + curr.s.charAt((int)(K - prevLen ));
            }
        }
        return "error";
    }
}
