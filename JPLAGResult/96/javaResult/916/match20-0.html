class Solution {
    public String decodeAtIndex(String S, int K) {
        int n = S.length();
        long[] lens = new long[n];
        lens[0] = 1;
        for (int i = 1; i < n; i++) {
            if (Character.isDigit(S.charAt(i))) {
                lens[i] = lens[i - 1] * (S.charAt(i) - '0');
            } else {
                lens[i] = lens[i - 1] + 1;
            }
        }
        int i = 0;
        while (lens[i] < K) {
            i++;
        }
        while (Character.isDigit(S.charAt(i)) || lens[i] != K) {
            K %= lens[i - 1];
            if (K == 0) {
                K = (int)lens[i - 1];
            }
            i--;
        }
        return S.charAt(i) + "";
    }
}
