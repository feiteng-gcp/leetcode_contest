class Solution {
    public String decodeAtIndex(String S, int K) {
        int n = S.length();
        
        long[] len = new long[n];
        len[0] = 1;
        for(int i = 1; i < n; i++)
        {
            char c = S.charAt(i);
            
            if(c >= '2' && c <= '9')
            {
                len[i] = len[i - 1] * (c - '0');
            }
            else
            {
                len[i] = len[i - 1] + 1;
            }
            
            if(len[i] > K)
            {
                n = i + 1;
                break;
            }
        }
        
        for(int i = n - 1; i >= 1; i--)
        {
            if(len[i] >= K && len[i - 1] < K)
            {
                char c = S.charAt(i);
                if(c >= '2' && c <= '9')
                {
                    K %= len[i - 1];
                    if(K == 0)
                    {
                        K = (int)len[i - 1];
                    }
                }
                else
                {
                    return "" + c;
                }
            }
        }
        
        return "" + S.charAt(0);
    }
}
