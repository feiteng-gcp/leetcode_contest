class Solution {
    public String decodeAtIndex(String S, int K) {
            long curLen = 0;
            long[] lll = new long[S.length()];
            K--;
            for (int i = 0; i < S.length(); i++) {
                long l = curLen;
                if (S.charAt(i) >= 'a') {
                    l++;
                    if (l == K + 1) {
                        return S.substring(i, i + 1);
                    }
                } else {
                    int v = S.charAt(i) - '0';
                    l = v * curLen;
                    if (curLen <= K && l > K) {
                        long idx = K % curLen;
                        return recurF(S, i - 1, (int) idx, lll);
                    }
                }
                curLen = l;
                lll[i] = curLen;
            }
            return "";
        }

        public String recurF(String s, int idx, int plus, long[] lll) {
            if (s.charAt(idx) >= 'a') {
                if (lll[idx] == plus + 1) {
                    return s.substring(idx, idx + 1);
                } else {
                    return recurF(s, idx - 1, plus, lll);
                }
            } else {
                int prevL = (int)lll[idx - 1];
                return recurF(s, idx - 1, plus % prevL, lll);
            }
        }
}
