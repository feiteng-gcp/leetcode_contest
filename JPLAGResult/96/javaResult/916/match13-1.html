class Solution {
    public String decodeAtIndex(String S, int K) {
        List<String> str = new ArrayList<>();
        List<Integer> num = new ArrayList<>();
        
String cur = "";
        for(int i = 0; i < S.length(); i++) {
            char tmp = S.charAt(i);
            if(tmp >= '2' && tmp <= '9') {
                str.add(cur);
                num.add(tmp - '0');
                cur = "";
            } else {
                cur += tmp;
            }
        }
        if(cur.length() != 0) {
            str.add(cur);
        }
        
        String str1 = "";
        long prev = 0;
        for(int i = 0; i < str.size(); i++) {
            if(i == num.size()) {
                return "" + str.get(i).charAt((int)(K - prev - 1));
            } else {
                long curLen = prev + str.get(i).length();
                long tmp = curLen * num.get(i);
                if(tmp >= K) {
                    long tmpLen = curLen;
                    long tmpPrev = prev;
                    for(int j = i; j >= 0; j--) {
                        // System.out.println(tmpLen);
                        // System.out.println(tmpPrev);
                        long index = K % tmpLen;
                        K = (int)index;
                        if(index == 0)
                            index = tmpLen;
                        // System.out.println(index);
                        if(index > tmpPrev)
                            return "" + str.get(j).charAt((int)(index - tmpPrev - 1));
                        else {
                            tmpLen = (tmpLen - str.get(j).length()) / num.get(j - 1);
                            tmpPrev = tmpLen  - str.get(j - 1).length();
                        }
                    }
                } else {
                    prev = tmp;
                
}
            }
        }
        return "";
    }
}
