class Solution {
    public int projectionArea(int[][] grid) {
        int n = grid.length, result = 0;
        int[] xs = new int[n], ys = new int[n];
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] > 0) {
                    result++;
                    xs[i] = Math.max(xs[i], grid[i][j]);
                    ys[j] = Math.max(ys[j], grid[i][j]);
                
}
            }
        }
        for (int i = 0; i < n; i++) {
            result += xs[i];
            result += ys[i];
        }
        return result;
    }
}
