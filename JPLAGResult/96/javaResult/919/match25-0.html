class Solution {
    public int projectionArea(int[][] grid) {
        int totalArea = 0;
        if(grid == null || grid.length == 0 || grid[0].length == 0) return totalArea;
        int n = grid.length;
        int[] rows = new int[n];
        int[] cols = new int[n];
        
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                if(grid[i][j] != 0) totalArea++;
                rows[i] = Math.max(rows[i], grid[i][j]);
                cols[j] = Math.max(cols[j], grid[i][j]);
            }
        }
        
        for(int i=0; i<n; i++) {
            totalArea += rows[i]+cols[i];
        }
        
        return totalArea;
    }
}
