class Solution {
    public int projectionArea(int[][] grid) {
        int ans = 0;
        int row = grid.length;
        int col = grid[0].length;

        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (grid[i][j] == 0) {
                    continue;
                }
                ans++;
            }
        }

        for (int i = 0; i < row; i++) {
            int maxHeight = 0;
            for (int j = 0; j < col; j++) {
                maxHeight = Math.max(maxHeight, grid[i][j]);
            }
            ans += maxHeight;
        }

        for (int j = 0; j < col; j++) {
            int maxHeight = 0;
            for (int i = 0; i < row; i++) {
                maxHeight = Math.max(maxHeight, grid[i][j]);
            }
            ans += maxHeight;
        }

        return ans;
    }
}
