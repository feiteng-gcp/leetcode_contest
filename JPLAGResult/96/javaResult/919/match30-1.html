class Solution {
    public int projectionArea(int[][] grid) {
        
int a = 0, b = 0, c = 0;
        int[] amax = new int[grid.length];
        int[] bmax = new int[grid[0].length];
        for(int i=0;i<grid.length;i++) {
            for(int j=0;j<grid.length;j++) {
                amax[i] = Math.max(amax[i], grid[i][j]);
                bmax[j] = Math.max(bmax[j], grid[i][j]);
                if(grid[i][j]!=0) c++
;
            }
        }
        for(int i=0;i<grid.length;i++) a+=amax[i];
        for(int j=0;j<grid[0].length;j++) b+=bmax[j];
        return a+b+c;
    }
}
