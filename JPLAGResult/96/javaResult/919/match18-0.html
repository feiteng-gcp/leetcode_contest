class Solution {
    public int projectionArea(int[][] grid) {
        int area = 0, n = grid.length;        
        int[] front = new int[n];       
        int[] side = new int[n];        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                front[i] = Math.max(front[i], grid[i][j]);
                side[j] = Math.max(side[j], grid[i][j]);
                if (grid[i][j] > 0) {
                    area++;
                }
            }
        }
        
        for (int i = 0; i < n; i++) {
            area += front[i] + side[i];
        }
        return area;
    }
}
