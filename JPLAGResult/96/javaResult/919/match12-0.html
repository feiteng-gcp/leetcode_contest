class Solution {
    public int projectionArea(int[][] grid) {
        int rows = grid.length, cols = grid[0].length;
        int xy = 0, xz = 0, yz = 0;
        
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < cols; c++) {
                if (grid[r][c] > 0) {
                    xy++;
                }
            }
        }
        
        for (int r = 0; r < rows; r++) {
            int max = -1;
            for (int c = 0; c < cols; c++) {
                max = Math.max(max, grid[r][c]);
            }
            xz += max;
        }
        
        for (int c = 0; c < cols; c++) {
            int max = -1;
            for (int r = 0; r < rows; r++) {
                max = Math.max(max, grid[r][c]);
            }
            yz += max;
        }
        
        return xy + xz + yz;
    }
}
