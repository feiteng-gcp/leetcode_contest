class Solution {
    public int projectionArea(int[][] grid) {
        
int sum = 0;
        
        int[] z = new int[grid.length];
        int[] y = new int[grid.length];
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] != 0) sum++;
                z[i] = Math.max(z[i], grid[i][j]);
                y[j] = Math.max(y[j], grid[i][j]);
            
}
        }                
        
        for (int i = 0; i < grid.length; i++) {
            sum += z[i];
            sum += y[i];
        }
        return sum;
    }
}
