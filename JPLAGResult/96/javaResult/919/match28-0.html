class Solution {
    public int projectionArea(int[][] grid) {
        if (grid.length == 0 || grid[0].length == 0) {
            return 0;
        }
        
        int m = grid.length, n = grid[0].length, area = 0;
        
        for (int i = 0; i < m; i++) {
            int max = 0;
            for (int j = 0; j < n; j++) {
                max = Math.max(grid[i][j], max);
                if (grid[i][j] > 0) {
                    area++;
                }
            }
            area += max;
        }
        
        for (int j = 0; j < n; j++) {
            int max = 0;
            for (int i = 0; i < m; i++) {
                max = Math.max(grid[i][j], max);
            }
            area += max;
        }
        
        return area;
    }
}
