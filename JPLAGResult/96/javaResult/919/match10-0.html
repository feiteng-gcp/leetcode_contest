class Solution {
    public int projectionArea(int[][] g) {
        int n = g.length;
        int xy = 0;
        int[] xz = new int[n];
        int[] yz = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int v = g[i][j];
                xy += Math.min(1, v);
                xz[i] = Math.max(xz[i], v);
                yz[j] = Math.max(yz[j], v);
            }
        }
        int sum = xy;
        for (int v: xz) sum += v;
        for (int v: yz) sum += v;
        return sum;
    }
}
