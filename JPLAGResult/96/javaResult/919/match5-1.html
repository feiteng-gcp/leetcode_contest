	class Solution {
	    public int projectionArea(int[][] grid) {
	        int n =
 grid.length, m = grid[0].length;
	        int ret = 0;
	        for(int i = 0;i < n;i++){
	        	for(int j = 0;j < m;j++){
	        		if(grid[i][j] > 0)ret++;

	        	}
	        }
	        for(int i = 0;i < n;i++){
	        	int max = 0;
	        	for(int j = 0;j < m;j++){
	        		max = Math.max(max, grid[i][j]);
	        	}
	        	ret += max;
	        }
	        for(int i = 0;i < m;i++){
	        	int max = 0;
	        	for(int j = 0;j < n;j++){
	        		max = Math.max(max, grid[j][i]);
	        	}
	        	ret += max;
	        }
	        return ret;
	    }
	}
