class Solution {
    public int projectionArea(int[][] grid) {
        int x = grid.length;
        int y = grid[0].length;
        int result = 0;
        for (int i = 0; i < x; i ++) {
            int max = 0;
            for (int j = 0; j < y; j ++) {
                if (grid[i][j] > 0) {
                    result += 1;
                }
                max = Math.max(max, grid[i][j]);
            }
            result += max;
        }
        
        for (int j = 0; j < y; j ++) {
            int max = 0;
            for (int i = 0; i < x; i ++) {
                max = Math.max(max, grid[i][j]);
            }
            result += max;
        }
        
        return result;
    }
}
