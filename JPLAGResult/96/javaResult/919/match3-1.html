class Solution {
	public int projectionArea(int[][] grid) {

		int ans = 0;

		for (int r = 0; r < grid.length; r++) {

			int max = 0;
			for (int c = 0; c < g
rid[r].length; c++) {
				if (grid[r][c] > 0)
				    ans++;
			}
		}

		for (int r = 0
; r < grid.length; r++) {

			int max = 0;
			for (int c = 0; c < grid[r].length; c++) {
				max = Math.max(max, grid[r][c]);
			}
			ans += max;
		}
		for (int c = 0
; c < grid[0].length; c++) {

			int max = 0;
			for (int r = 0; r < grid.length; r++) {
				max = Math.max(max, grid[r][c]);
			}
			ans += max;
		}
		return ans;
	}
}
