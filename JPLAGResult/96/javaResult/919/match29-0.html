class Solution {
    public int projectionArea(int[][] grid) {
        if (grid == null)
            return 0;
        int xy =0, xz = 0, yz = 0;
        int n = grid.length;
        for (int i = 0; i < n; i++){
            int maxxz = 0;
            for (int j = 0; j < n; j++){
                if (grid[i][j] != 0)
                    xy += 1;
                maxxz = Math.max(maxxz, grid[i][j]);
            }
            xz += maxxz;
        }
        for (int i = 0; i < n; i++){
            int maxxz = 0;
            for (int j = 0; j < n; j++){
                maxxz = Math.max(maxxz, grid[j][i]);
            }
            yz += maxxz;
        }
        return xy+xz+yz;
    }
}
