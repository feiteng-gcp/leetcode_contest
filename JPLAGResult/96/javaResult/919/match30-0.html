class Solution {
    public int projectionArea(int[][] grid) {
        // z - how many grid is not null
        // x (i): sum of biggest num in i
        // y (j): sum of bigges num in j
        if(grid == null || grid.length==0 || grid[0].length==0)return 0;
        int LenI = grid.length;
        int LenJ = grid[0].length;
        int ans = 0;
        
        int z = 0;
        int[] x = new int[LenI];
        int[] y = new int[LenJ];
        
        for(int i=0; i<grid.length;i++){
            for(int j=0;j<grid[i].length;j++){
                int num = grid[i][j];
                if(num>0)z++;
                //position[i,j]
                if(num>x[i])x[i]=num;
                if(num>y[j])y[j]=num;
            }
        }
        
        ans+=z;
        for(int i=0;i<LenI;i++)ans+=x[i];
        for(int j=0;j<LenJ;j++)ans+=y[j];
        
        return ans;
    }
}
