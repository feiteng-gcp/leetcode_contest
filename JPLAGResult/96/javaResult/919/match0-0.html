class Solution {
    public int projectionArea(int[][] grid) {
         int top = 0;
        for(int i = 0 ; i < grid.length ; i++){
            for(int j = 0 ; j < grid[0].length ; j++){
                if(grid[i][j] != 0) top += 1;
            }
        }
        int left = 0;
        for(int i = 0 ; i < grid.length ; i++){
            int max = 0;
            for(int j = 0 ; j < grid[i].length ; j++){
                max = Math.max(max, grid[i][j]);
            }
            left += max;
        }
        int front = 0;
        for(int j = 0 ; j < grid[0].length ; j++){
            int max = 0;
            for(int i = 0 ; i < grid.length ; i++){
                max = Math.max(max, grid[i][j]);
            }
            front += max;
        }
        return left+front+top;
    }
}
