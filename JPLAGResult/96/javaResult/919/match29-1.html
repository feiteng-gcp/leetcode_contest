class Solution {
    public int projectionArea(int[][] grid) {
        int row = grid.length;
        if(row == 0)
            return 0;
        
int col = grid[0].length;
        int sum = row * col;
        for(int i = 0; i < row; i++) {
            int max = 0;
            for(int j = 0; j < col; j++) {
                max = Math.max(max, grid[i][j]);
                if(grid[i][j] == 0)
                    sum--;
            
}
            sum += max;
        }
        
        for(int i = 0; i < col; i++) {
            int max = 0;
            for(int j = 0; j < row; j++) {
                max = Math.max(max, grid[j][i]);
            }
            sum += max;
        }
        return sum;
    }
}
