class Solution {
    public int projectionArea(int[][] grid) {
        int n = grid.length;
        int bot = 0;
        int[] row = new int[n], col = new int[n];
        for(int i = 0; i<n; ++i){
            for(int j = 0; j<n; ++j){
                if(grid[i][j] > 0) bot++;
                
row[i] = Math.max(row[i], grid[i][j]);
                col[j] = Math.max(col[j], grid[i][j]);
            }
        }
        int sum = bot;
        for(int r : row) sum+= r;
        for(int c : col) sum+= c;
        return sum;
    }
}
