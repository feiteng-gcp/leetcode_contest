class Solution {
    public int numRescueBoats(int[] people, int limit) {
        Arrays.sort(people);
        int count = 0;
        int l = 0, h = people.length-1;
        while(l < h){
            if(people[l] + people[h] > limit) {
                count++;
                h--;
            } else {
                count++;
                h--; l++;
            }
        }
        return l==h ? count+1:count;
    }
}
