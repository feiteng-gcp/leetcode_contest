class Solution {
    public int numRescueBoats(int[] people, int limit) {
        Arrays.sort(people);
        int i = 0, j = people.length - 1;
        int boat = 0;
        while(j>=i) {
            if(people[i] + people[j] <= limit) {
                i += 1;
            }
            j -= 1;
            boat += 1;
        }
        return boat;
    }
}
