class Solution {
    public int numRescueBoats(int[] people, int limit) {
        Arrays.sort(people);
        int num = 0;
        int start = 0;
        int end = people.length - 1;
        while (start < end) {
            if (people[start] + people[end] <= limit) {
                num += 1;
                start += 1;
                end -= 1;
            } else {
                end -= 1;
                num += 1;
            }
        }
        return start == end ? num + 1 : num;
    }
}
