import java.util.*;

class Solution {
    public int numRescueBoats(int[] people, int limit) {

        Arrays.sort(people);

        int lo = 0;
        int hi = people.length-1;
        int numboats = 0;

        while(lo <= hi) {

            // special case: lo == hi:
            if (lo == hi) {
                numboats++;
                break;
            }

            // see if hi can share with lo
            if (people[hi] + people[lo] <= limit) {
                numboats++;
                hi--;
                lo++;
            }
            else {
                // hi will go on his own then.
                numboats++;
                hi--;
            }

        }


        return numboats;
    }
}
