class Solution {
public:
    int projectionArea(vector<vector<int>>& v) {
        int ans = 0;
        vector<int> vv = vector<int>(v[0].size());
        for (int i = 0; i < v.size(); i++) {
            int ma = 0;
            for (int j = 0; j < v[i].size(); j++) {
                if (v[i][j])
                    ans++;
                ma = max(ma, v[i][j]);
                vv[j] = max(vv[j], v[i][j]);
            }
            ans += ma;
        }
        for (int i = 0; i < vv.size(); i++)
            ans += vv[i];
        return ans;
    }
};
