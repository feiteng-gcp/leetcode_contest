class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int area = 0;
        unordered_map<int, int> x_to_h, y_to_h;
        for (int y = 0; y < grid.size(); ++y) {
            for (int x = 0; x < grid[y].size(); ++x) {

                if (grid[y][x] > 0) ++area;
                x_to_h[x] = max(x_to_h[x], grid[y][x]);
                y_to_h[y] = max(y_to_h[y], grid[y][x]);
            }
        }
        for (auto &elem : x_to_h) area += elem.second;
        for (auto &elem : y_to_h) area += elem.second;
        return area;
    }
};
