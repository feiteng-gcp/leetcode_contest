class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {

        int top = 0, front = 0, right = 0, n = grid.size(), m = grid.front().size();
        for (int i = 0; i < n; i++) {
            int max = grid[i][0];
            for (int j = 0; j < m; j++) {
                if (grid[i][j] > 0) {
                    top++;
                }

                if (grid[i][j] > max) {
                    max = grid[i][j];
                }
            }
            front += max;
        }
        for (int j = 0; j < m; j++) {
            int max = grid[0][j];
            for (int i = 0; i < n; i++) {
                if (grid[i][j] > max) {
                    max = grid[i][j];
                }
            }
            right += max;
        }
        return right + top + front;
    }
};
