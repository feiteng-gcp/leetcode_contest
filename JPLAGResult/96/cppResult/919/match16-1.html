class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        
        int n = grid.size();
        int m = grid[0].size();
        
        int res = 0;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                res += (grid[i][j] > 0);
            }
        }
        

        for (int i = 0; i < n; i++) {
            int curr_max = 0;
            for (int j = 0; j < m; j++) {
                curr_max = ((grid[i][j] > curr_max) ? grid[i][j] : curr_max);
            }
            res += curr_max;
        }
        

        for (int j = 0; j < m; j++) {
            int curr_max = 0;
            for (int i = 0; i < n; i++) {
                curr_max = ((grid[i][j] > curr_max) ? grid[i][j] : curr_max);
            }
            res += curr_max;
        }
        
        
        return res;
        
    }
};
