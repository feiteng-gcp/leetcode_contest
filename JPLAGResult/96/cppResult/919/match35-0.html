class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int n = grid.size();
        int bottom = 0;
        int left = 0, back = 0;
        for(int i = 0; i < n; i++){
            int max_i = 0;
            for(int j = 0; j < n; j++){
                if(grid[i][j] > 0)
                    bottom++;
                max_i = max(max_i, grid[i][j]);
            }
            left+= max_i;
        }
        for(int j = 0; j < n; j++){
            int max_j = 0;
            for(int i = 0; i < n; i++){
                max_j = max(max_j, grid[i][j]);
            }
            back += max_j;
        }
        return bottom + left + back;
    }
};
