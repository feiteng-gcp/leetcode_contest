class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int m,n;
        int ans = 0;
        m = grid.size();
        if(m<=0)return ans;
        n = grid[0].size();
        
        for(int i=0;i<m;i++)
        {
            int mx = INT_MIN;
            for(int j=0;j<n;j++)
            {
                mx = max(mx,grid[i][j]);
                if(grid[i][j]>0)ans++;
            }
            ans+=mx;
        }
        
        for(int j=0;j<n;j++)
        {
            int mx = INT_MIN;
            for(int i=0;i<m;i++)
            {
                mx = max(mx,grid[i][j]);
            }
            ans+=mx;
        }
        return ans;
    }
};
