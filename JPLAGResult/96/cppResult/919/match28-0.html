class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int n = grid.size();
        int ret = 0;
        for (int i = 0; i < n; i++) {
            int rmax = 0;
            for (int j = 0; j < n; j++) {
                if (grid[i][j]) {
                    ret++;  /* For top view */
                }
                rmax = max(rmax, grid[i][j]);
            }
            ret += rmax; /* Side View */
        }
        
        for (int j = 0; j < n; j++) {
            int cmax = 0;
            for (int i = 0; i < n; i++) {
                cmax = max(cmax, grid[i][j]);   
            }
            ret += cmax;
        }
        return ret;
    }
};
