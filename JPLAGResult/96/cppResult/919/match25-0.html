class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int xy = 0, xz = 0, yz = 0;
        for (int i = 0; i < grid.size(); i++) {
            int tmp = 0;
            for (int j = 0; j < grid[0].size(); j++) {
                xy += grid[i][j] > 0;
                tmp = max(tmp, grid[i][j]);
            }
            xz += tmp;
        }
        for (int j = 0; j < grid[0].size(); j++) {
            int tmp = 0;
            for (int i = 0; i < grid.size(); i++) {
                tmp = max(tmp, grid[i][j]);
            }
            yz += tmp;
        }
        return xy + xz + yz;
    }
};
