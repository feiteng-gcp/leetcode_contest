class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int sum = 0;
        int n = grid.size();
        int m = grid[0].size();
        for (int i = 0; i < n; ++i) {
            int val = 0;
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] > 0)  ++sum;
                val = max(val, grid[i][j]);
            }
            sum += val;
        }
        for (int j = 0; j < m; ++j) {
            int val = 0;
            for (int i = 0; i < n; ++i) {
                val = max(val, grid[i][j]);
            }
            sum += val;
        }
        return sum;
    }
};
