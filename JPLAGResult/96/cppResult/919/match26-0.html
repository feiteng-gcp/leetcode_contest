class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        
        int n = grid.size();
        
        if (n == 0) return 0;
        
        int ans = 0;
        
        for (int i = 0; i < n; ++i) {
            int maxt = INT_MIN;
            for (int j = 0; j < n; ++j) {
                
                if (grid[i][j]) ans ++;
                if (maxt < grid[i][j]) {
                    maxt = grid[i][j];
                }                
            }
            
            ans += maxt;
        }
        
        
        for (int i = 0; i < n; ++i) {
            int maxt = INT_MIN;
            for (int j = 0; j < n; ++j) {
                
                if (maxt < grid[j][i]) {
                    maxt = grid[j][i];
                }                
            }
            
            ans += maxt;
        }
        
        return ans;
    }
};
