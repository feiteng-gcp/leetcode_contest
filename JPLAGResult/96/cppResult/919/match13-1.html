class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {

        int r = 0;
        for (int i = 0; i < grid.size(); i++) {
            int y = 0;
            for (int j = 0; j < grid[i].size(); j++) {
                if (grid[i][j]) r++;
                y = max(y, grid[i][j]);
            }
            r += y;
        }
        for (int i = 0; i < grid[0].size(); i++) {
            int y = 0;
            for (int j = 0; j < grid.size(); j++) {
                y = max(y, grid[j][i]);
            }
            r += y;
        }
        return r;
    }
};
