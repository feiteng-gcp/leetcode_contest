class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int M = grid.size();

        int N = grid[0].size();
        int xy = 0;
        for(int x = 0; x < M; x++){
            for(int y = 0; y < N; y++){
                xy += grid[x][y]!=0;
            }
        }
        int xz = 0;

        for(int x = 0; x < M; x++){
            int maxy = 0;
            for(int y = 0; y < N; y++){
                maxy = max(maxy,grid[x][y]);
            }
            xz += maxy;
        }
        int yz = 0;

        for(int y = 0; y < N; y++){
            int maxx = 0;
            for(int x = 0; x < N; x++){
                maxx = max(maxx,grid[x][y]);
            }
            yz += maxx;
        }
        return xy + yz + xz;
    }
};
