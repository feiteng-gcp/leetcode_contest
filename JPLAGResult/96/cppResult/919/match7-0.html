class Solution {
public:
    int projectionArea(vector<vector<int>>& grid) {
        int N = grid.size();
        int M = grid.at(0).size();
        int ans = 0;
        for(int i=0; i<N; i++) {
            int ma = 0;
            for(int j=0; j<M; j++) {
                ma = max(ma, grid.at(i).at(j));
            }
            ans += ma;
        }
        for(int j=0; j<M; j++) {
            int ma = 0;
            for(int i=0; i<N; i++) {
                ma = max(ma, grid.at(i).at(j));
            }
            ans += ma;
        }
        for(int i=0; i<N; i++) {
            for(int j=0; j<M; j++) {
                if (grid.at(i).at(j) > 0) {
                    ans++;
                }
            }
        }
        return ans;
    }
};
