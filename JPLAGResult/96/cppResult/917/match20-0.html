class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int max_pointer = people.size() - 1;
        int min_pointer = 0;
        sort(people.begin(), people.end());
        int num_boat = 0;
        while(min_pointer < max_pointer)
        {
            if(people[min_pointer] + people[max_pointer] <= limit)
            {
                num_boat++;
                max_pointer--;
                min_pointer++;
            }
            else
            {
                num_boat++;
                max_pointer--;
            }
        }
        if(max_pointer == min_pointer)
        {
            num_boat++;
        }
        return num_boat;
    }
};
