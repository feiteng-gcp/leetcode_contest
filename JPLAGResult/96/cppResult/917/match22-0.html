class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        sort(people.begin(),people.end());
        int left = 0,right = static_cast<int>(people.size()) - 1,result = 0;
        while(left < right) {
            if(people[left] + people[right] <= limit) {
                result++;
                left++;
                right--;
            }else {
                result++;
                right--;
            }
        }
        if(left == right) {
            result++;
        }
        return result;
    }
};
