class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        sort(people.begin(), people.end());
        int n = people.size();
        int j= n-1, i=0, cnt=0;
        while(i<=j){
            if(i==j){
                cnt++;
                i++;
                j--;
            }
            else if(people[i]+people[j]>limit){
                cnt++;
                j-=1;
            }
            else{
                cnt++;
                i+=1;
                j-=1;
            }
        }
        return cnt;
        
    }
};
