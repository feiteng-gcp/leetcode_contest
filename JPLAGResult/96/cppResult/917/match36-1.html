class Solution {
public:
    int numRescueBoats(vector<int>& p, int l) {
        sort(p.begin(), p.end());
        int e = p.size()-1, s = 0, res = 0;
        while (s <= e) {
            if (p[s]+p[e] <= l) {
                s++;
            }
            e--;
            res++;
        }
        return res;
    }
};
