class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int ans = 0;
        vector<int> temp;
        for(auto p:people)
        {
            if(p>limit)ans++;
            else temp.push_back(p);
        }
        sort(temp.begin(),temp.end());
        int l = 0,r= temp.size()-1;
        
        while(l<=r)
        {
            if(temp[l]+temp[r]<=limit)
            {
                ans++;
                l++;
                r--;
            }
            else
            {
                ans++;
                r--;
            }
        }
        return ans;
    }
};
