class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int n = people.size();
        int left = 0;
        int right = n-1;
        int ans = 0;
        sort(people.begin(), people.end());
        while (left <= right) {
            int rem = limit - people[right];
            if (people[left] <= rem)
                left++;
            right--;
            ans++;
        }
        return ans;
    }
};
