class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        if (people.empty()) return 0;
        sort(people.begin(), people.end());
        int left = 0; int right = people.size() - 1;
        int ret = 0;
        while(left <= right){
            int w_left = people[left];
            int w_right = people[right];

            if (w_left + w_right <= limit) {
                ret ++;
                left ++; right --;
            }
            else{
                ret ++;
                right --;
            }
        }
        return ret;
    }
};
