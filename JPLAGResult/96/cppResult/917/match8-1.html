class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        sort(people.begin(), people.end());
        int left = 0, right = people.size()-1;
        int res = 0;
        while(left <= right){
            if(left == right){
                right--;
                res++;
            }
            else if(people[left] + people[right] <= limit){
                left++;
                right--;
                res++;
            }else{
                right--;
                res++;
            }
        }
        return res;
    }
};
