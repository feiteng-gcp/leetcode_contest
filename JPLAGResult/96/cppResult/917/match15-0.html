class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        if (people.empty()) return 0;
        sort(people.begin(), people.end());
        int left = 0;
        int right = people.size() - 1;
        int res = 0;
        while (left < right) {
            if (people[left] + people[right] <= limit) {
                left++;
                right--;
                res++;
            } else {
                right--;
                res++;
            }
        }
        if (left == right) res++;
        return res;
    }
};
