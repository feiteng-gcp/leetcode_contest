class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        sort(people.begin(), people.end());
        int ans = 0;
        int n = people.size();
        if (n <= 1) return n;
        int i = 0, j = n - 1;
        while (i <= j) {
            if (people[i] + people[j] <= limit) {
                i++; j--;
                ans++;
            } else {
                j--;
                ans++;
            }
        }
        return ans;
    }
};
