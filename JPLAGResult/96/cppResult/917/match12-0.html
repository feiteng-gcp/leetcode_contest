class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int ret = 0;
        sort(people.begin(), people.end());
        int p = 0, q = (int)people.size() - 1;
        while (p <= q) {
            if (p == q) {
                ret ++;
                break;
            }
            if (people[p] + people[q] <= limit) {
                p ++;
                q --;
                ret ++;
            } else {
                q --;
                ret ++;
            }
        }
        return ret;
    }
};
