class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        sort(begin(people),end(people));
        int r=people.size()-1,l=0,rst=0;
        while(r>=l){
            if(people[r]+people[l]>limit){
                --r;
                ++rst;
            }
            else{
                --r;
                ++l;
                ++rst;
            }
        }
        return rst;
    }
};
