class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int num_rides = 0;
        sort(people.begin(), people.end());
        int i = 0;
        int j = people.size() - 1;
        
        while (i < j) {
            if(people[i] + people[j] <= limit) {
                i++;
                j--;
            } else {
                j--;
            }
            num_rides++;
        }
        
        return i==j ? num_rides + 1 : num_rides;
    }
};
