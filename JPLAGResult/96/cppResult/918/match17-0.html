struct PQE {
    int idx, val;
    bool operator<(const PQE&b) const {
        return val > b.val;
    }
    PQE(int _i=0, int _v=0): idx(_i), val(_v) {}
};
struct EDGE {
    int idx, val;
    EDGE(int _i=0, int _v=0): idx(_i), val(_v) {}
};
class Solution {
public:
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        int INF = 1000000007;
        vector<int> dst(N, INF);
        vector<vector<EDGE>> ee(N, vector<EDGE>());
        for(auto edge: edges) {
            ee[edge[0]].push_back(EDGE(edge[1], edge[2]+1));
            ee[edge[1]].push_back(EDGE(edge[0], edge[2]+1));
        }
        priority_queue<PQE> pq;
        pq.push(PQE(0,0));
        while(!pq.empty()) {
            PQE now = pq.top();
            pq.pop();
            //cout << "--" << now.idx << ": " << now.val << endl;
            if(dst[now.idx] != INF) continue;
            dst[now.idx] = now.val;
            for(auto edge: ee[now.idx]) {
                if(dst[edge.idx] == INF) {
                    pq.push(PQE(edge.idx, now.val+edge.val));
                }
            }
        }
        int ans = 0;
        for(int i=0; i<N; i++) {
            //cout << i << ": " << dst[i] << endl;
            if(dst[i] <= M) {
                ans++;
            }
        }
        for(auto edge: edges) {
            int nans = 0;
            nans += std::max(std::min(edge[2], M-dst[edge[0]]), 0);
            nans += std::max(std::min(edge[2], M-dst[edge[1]]), 0);
            nans = std::min(nans, edge[2]);
            //cout << edge[0] << "-" << edge[1] << ": " << nans << endl;
            ans += nans;
        }
        return ans;
    }
};
