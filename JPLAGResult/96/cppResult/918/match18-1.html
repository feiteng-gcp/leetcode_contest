vector<pair<int, int> > G[3005];
bool vis[3005];
int d[3005];
class Solution {
public:
    void D(){
        priority_queue<pair<int, int> , vector<pair<int,int> > , greater<pair<int, int> > > pq;
        memset(vis,0,sizeof(vis));
        memset(d, 0x3f, sizeof(d));
        d[0] = 0;
        pq.push(make_pair(0, 0));
        while(!pq.empty()){
            pair<int, int> p = pq.top(); pq.pop();
            int u = p.second;
            int dis = p.first;
            if(vis[u]) continue;
            vis[u] = true;
            for(auto edge: G[u]){
                int v = edge.first;
                int cost = edge.second;
                if(d[u] + cost < d[v]){
                    d[v] = d[u] + cost;
                    pq.push(make_pair(d[v], v));
                }
            }
        }
    }
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        for(int i = 0;i < N;i++){
            G[i].clear();
        }
        for(int i = 0;i < edges.size();i++){
            int u = edges[i][0], v = edges[i][1], cost = edges[i][2];
            G[u].push_back(make_pair(v, cost + 1));
            G[v].push_back(make_pair(u, cost + 1));

        }
        D();
        
        int ans = 0;
        for(int i = 0;i < N;i++){
            if(d[i] <= M) ans++;
        }
        for(int i = 0;i < edges.size();i++){
            int u = edges[i][0], v = edges[i][1], cost = edges[i][2];
            ans += min(max(0, M - d[u]) + max(0, M - d[v]), cost);
        }
        return ans;
    }
};
