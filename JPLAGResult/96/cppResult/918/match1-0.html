class Solution {
 public:
  int reachableNodes(vector<vector<int>>& edges, int M, int N) {
    vector<vector<pair<int, int>>> g(N);
    map<int, map<int, int>> rem;
    for (auto& e : edges) {
      g[e[0]].emplace_back(e[1], e[2]);
      g[e[1]].emplace_back(e[0], e[2]);
      rem[e[0]][e[1]] = e[2];
    }
    vector<int> dist(N, INT_MAX);
    priority_queue<pair<int, int>, vector<pair<int, int>>,
                   greater<pair<int, int>>>
        pq;
    dist[0] = 0;
    pq.emplace(0, 0);
    int ans = 0;
    while (!pq.empty()) {
      int u = pq.top().second;
      int d = pq.top().first;
      pq.pop();
      if (d > dist[u]) {
        continue;
      }
      ++ans;
      for (auto& p : g[u]) {
        int v = p.first;
        int w = p.second;
        int& r = rem[min(u, v)][max(u, v)];
        int del = min(r, M - dist[u]);
        r -= del;
        ans += del;
        if (dist[v] > dist[u] + w + 1) {
          dist[v] = dist[u] + w + 1;
          if (dist[v] <= M) {
            pq.emplace(dist[v], v);
          }
        }
      }
    }
    return ans;
  }
};
