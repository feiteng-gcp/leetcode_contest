using PII = pair<int, int>;
const int inf = 2e9;
class Solution {
  vector<PII> g[3010];
  bool inq[3010];
  int d[3010];
public:
  int reachableNodes(vector<vector<int>> &edges, int M, int N) {
    for (int i = 0; i < N; ++i) {
      g[i].clear();
      inq[i] = false;
      d[i] = inf;
    }
    for (auto &e : edges) {
      g[e[0]].push_back({e[1], e[2] + 1});
      g[e[1]].push_back({e[0], e[2] + 1});
    }
    queue<int> q;
    q.push(0);
    d[0] = 0;
    while (q.size()) {
      int x = q.front(); q.pop(); inq[x] = false;
      for (auto &e : g[x]) {
        int nw = d[x] + e.second;
        if (nw < d[e.first]) {
          d[e.first] = nw;
          if (!inq[e.first]) {
            inq[e.first] = true;
            q.push(e.first);
          }
        }
      }
    }
    int ans = 0;
    for (int i = 0; i < N; ++i) {
      if (d[i] <= M) ans++;
    }
    for (auto &e : edges) {
      long long ca = max(0, M - d[e[0]]);
      long long cb = max(0, M - d[e[1]]);
      ans += min(ca + cb, (long long)e[2]);
    }
    return ans;
  }
};
