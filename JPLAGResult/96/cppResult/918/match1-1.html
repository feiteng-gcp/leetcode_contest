class Solution {
public:
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        vector<int> dis(N, 1000000001);
        vector<vector<pair<int,int>>> e(N, vector<pair<int, int>>());
        for (auto& item : edges) {
            int u = item[0];
            int v = item[1];
            int c = 1 + item[2];
            e[u].push_back({v, c});
            e[v].push_back({u, c});
        }
        dis[0] = 0;
        queue<int> Q;
        Q.push(0);
        while (!Q.empty()) {
            int u = Q.front(); Q.pop();
            for (auto item : e[u]) {
                int v = item.first;
                int d = dis[u] + item.second;
                if (d < dis[v]) {
                    dis[v] = d;
                    Q.push(v);
                }
            }
        }
        int cnt = 0;
        for (int i = 0; i < N; ++i) {
            if (dis[i] <= M)  ++cnt;
        }
        for (auto& item : edges) {
            int u = item[0];
            int v = item[1];
            int c = item[2];
            if (c == 0)  continue;
            if (dis[u] >= M && dis[v] >= M)  continue;
            if (dis[u] >= M) {
                cnt += min(c, M - dis[v]);
                continue;
            }
            if (dis[v] >= M) {
                cnt += min(c, M - dis[u]);
                continue;
            }
            int val = M - dis[v] + M - dis[u];
            cnt += min(c, val);
        }
        return cnt;
    }
};
