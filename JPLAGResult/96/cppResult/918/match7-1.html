class Solution {
public:
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        vector<vector<pair<int,int> > > eds(N, vector<pair<int,int> >());
        for (vector<int> &e : edges) {
            eds[e[0]].push_back(make_pair(e[1], e[2]+1));
            eds[e[1]].push_back(make_pair(e[0], e[2]+1));
        }
        vector<bool> vis(N, false);
        vector<int> dis(N, M+1);
        dis[0] = 0;
        queue<int> q;
        for (q.push(0); !q.empty();) {
            int x = q.front();
            q.pop();
            vis[x] = false;
            for (int i = 0; i < eds[x].size(); ++i) {
                int y = eds[x][i].first;

                int d = dis[x] + eds[x][i].second;
                if (d < dis[y]) {
                    dis[y] = d;
                    if (!vis[y]) {
                        vis[y] = true;
                        q.push(y);
                    }
                }
            }
        }
        int res = 0;
        for (int i = 0; i < N; ++i) {
            if (dis[i] <= M) ++res;
        }
        for (vector<int> &e : edges) {
            res += min(e[2], max(0, M-dis[e[0]]) + max(0, M-dis[e[1]]));
        }
        return res;
    }
};
