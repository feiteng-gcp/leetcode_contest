typedef long long LL;

class Solution {
public:
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        vector<vector<int> > G(N);
        vector<vector<int> > L(N);
        vector<LL> D(N,-1);
        set< pair<LL,int> > Q;
        for (auto e : edges) {
            G[e[0]].push_back(e[1]); L[e[0]].push_back(e[2]);
            G[e[1]].push_back(e[0]); L[e[1]].push_back(e[2]);
        }
        D[0]=0;
        Q.insert({D[0],0});
        while(Q.size()) {
            int x = Q.begin()->second;
            Q.erase(Q.begin());
            for (int i = 0; i < G[x].size(); ++i) {
                int y = G[x][i];
                int l = L[x][i]+1;
                if (D[y]==-1 || D[y]>D[x]+l) {
                    Q.erase({D[y],y});
                    D[y]=D[x]+l;
                    Q.insert({D[y],y});
                }
            }
        }
        int res = 0;
        for (auto e : edges) {
            int a = e[0];
            int b = e[1];
            int c = e[2];
            
            if (D[a]==-1 || D[b]==-1) continue;
            //cout << "(" << a << "," << b << " = "<< c << ") " << D[a] << " :: " << D[b] << endl; 
            int aa = max(0ll,M-D[a]);
            int bb = max(0ll,M-D[b]);
            res += min(c,aa+bb);
            //cout << "added: " << min(c,aa+bb) << endl;
        }
        for (int i = 0; i < N; ++i) if (D[i]!=-1 && D[i]<=M) ++res;
        return res;
     }
};
