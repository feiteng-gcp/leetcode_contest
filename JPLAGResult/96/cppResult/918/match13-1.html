typedef long long LL;

struct State {
    int u;
    LL v;
    State() {}

    State(int u_, LL v_) : u(u_), v(v_) {}
    bool operator < (const State& that) const {
        return v > that.v;
    }
};

class Solution {
public:
    int reachableNodes(vector<vector<int>>& edges, int m, int n) {
        vector<LL> dist(n, 1LL << 60);
        dist[0] = 0;
        priority_queue<State> que;
        que.push(State(0, 0));
        vector<vector<pair<int, int>>> g(n);
        for (const auto& e : edges) {
            g[e[0]].push_back(make_pair(e[1], e[2] + 1));
            g[e[1]].push_back(make_pair(e[0], e[2] + 1));
        }
        while (!que.empty()) {
            State curr = que.top();
            que.pop();
            for (const auto e : g[curr.u]) {
                State next(e.first, curr.v + e.second);
                if (next.v > m) continue;
                if (next.v < dist[next.u]) {
                    dist[next.u] = next.v;
                    que.push(next);
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) if (dist[i] <= m) ++ret;
        for (const auto& e : edges) {
            int i = e[0], j = e[1], k = e[2];
            int x = 0, y = 0;
            if (dist[i] < m) x = min((LL)k, m - dist[i]);
            if (dist[j] < m) y = min((LL)k, m - dist[j]);
            ret += min(x + y, k);
        }
        return ret;
    }
};
