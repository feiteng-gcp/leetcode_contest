class Solution {
public:
    int reachableNodes(vector<vector<int>>& edges, int M, int N) {
        map<int, map<int, int>> graph;
        for (auto edge : edges) {
            int &u = edge[0], &v = edge[1], &w = edge[2];
            graph[u][v] = w;
            graph[v][u] = w;
        }
        vector<int> moves(N, 0);
        moves[0] = M;
        vector<bool> visited(N, false), reachable(N, false);

        visited[0] = reachable[0] = true;
        queue<int> q;
        q.push(0);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (auto it : graph[u]) {
                int v = it.first;
                int w = it.second + 1;
                if (w <= moves[u]) {
                    reachable[v] = true;
                    if (moves[v] < moves[u] - w) {
                        moves[v] = moves[u] - w;
                        if (!visited[v]) {
                            visited[v] = true;
                            q.push(v);
                        }
                    }
                }
            }
            visited[u] = false;
        }
        int num = 0;
        for (int i = 0; i < N; ++i) {
            num += reachable[i];
        }
        for (auto edge : edges) {
            int &u = edge[0], &v = edge[1], &w = edge[2];
            num += min(moves[u] + moves[v], w);
        }
        return num;
    }
};
