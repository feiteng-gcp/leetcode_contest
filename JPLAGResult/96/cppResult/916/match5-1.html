class Solution {
public:
    #include<string>
    string decodeAtIndex(string S, int K) {
        long long o=0;
        for (int i=0;i<S.length();i++)
        if (S[i]>='0'&&S[i]<='9') o*=S[i]-'0';else o++;
        for (int i=S.length()-1;i>=0;i--)
        if (S[i]>='0'&&S[i]<='9'){
            o/=S[i]-'0';
            while (K>o) K-=o;
        }else if (K==o) return S.substr(i,1);else o--;
    }
};
