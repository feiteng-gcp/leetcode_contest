class Solution {
public:
    string decodeAtIndex(string s, int K) {
        int n = s.size();
        vector<long long> a(n);
        a[0] = 1;
        for (int i = 1; i < n; ++i) {
            if (isdigit(s[i])) {
                a[i] = a[i - 1] * (s[i] - '0');
            } else {
                a[i] = a[i - 1] + 1;
            }
        }

        for (int i = n - 1; i >= 0; --i) {
            if (islower(s[i])) {
                if (K == a[i]) {
                    return string(1, s[i]);
                }
                assert(a[i] > K);
                continue;
            }

            // digit
            if (K == a[i]) {
                for (int j = i; j >= 0; --j) if (islower(s[j])) {
                    return string(1, s[j]);
                }
            }

            long long len = a[i - 1];
            K %= len;
            if (K == 0)
                K = len;
        }
    }
};
