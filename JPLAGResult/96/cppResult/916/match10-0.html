class Solution {
public:
    string decodeAtIndex(string S, int K) {
        long n = S.length(), k = 0;
        for (int i=0; i<n; i++) {
            if (islower(S[i])) {
                if (++k==K) return S.substr(i,1);
            } else {
                k *= (S[i]-'0');
            }
            if (k>=K) {
                return decodeAtIndex(S.substr(0,i+1),K-k/(S[i]-'0'));
            }
        }
        return S;
    }
};
