class Solution {
    
public:
    string decodeAtIndex(string S, int K) {
        
        long long count = 0;
        for (char ch : S) {
            //cout<<"Count:("<<ch<<"): "<<count<<endl;
            if (isdigit(ch)) {
                long long mult = ch - '0';
                if (count * mult >= K) {
                    mult = K % count;
                    //cout<<K<<" "<<count<<endl;            
                    if (mult == 0) {
                        mult = count;
                    }
                    return decodeAtIndex(S, mult);
                }
                count *= mult;
            }else {
                count++;
            }
            if (count == K) {
                return string(1, ch);
            }
        }
        return "";
    }
};
