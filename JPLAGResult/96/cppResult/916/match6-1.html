class Solution {
public:
    string decodeAtIndex(string S, int K) {
        

        unsigned long curr_len = 0;
        for (int i = 0; i < S.size(); i++) {
            if (S[i] >= 'a' && S[i] <= 'z') {
                curr_len++;
                if (curr_len == K)
                    return string(1, S[i]);
            }
            else {
                int n = S[i] - '0';
                if (curr_len * n < K)
                    curr_len *= n;
                else {
                    return decodeAtIndex(S, K - curr_len);
                }
            }
        }
        
    }
};
