class Solution {
public:
    string decodeAtIndex(string S, int K) {
        if (K == 1)
            return string(1, S[0]);
        long long int tot = 1;
        for (int i = 1; i < S.length(); i++) {
            char c = S[i];
            if (c > '1' && c <= '9') {
                int d = c-48;
                tot = tot*d;
                if (tot >= K) {
                    if (K%(tot/d) == 0) {
                        while (S[i] <= '9')
                            i--;
                        return string(1, S[i]);
                    } else {
                        return decodeAtIndex(S, K%(tot/d));
                    }
                }
            } else {
                tot++;
                if (tot == K) {
                    return string(1, S[i]);
                }
            }
        }
    }
};
