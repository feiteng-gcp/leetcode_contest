class Solution {
public:
    string decodeAtIndex(string S, int K) {
        int64_t l = 0;
		K--;
        for (int i = 0; i < S.size(); i++) {
            char c = S[i];
            if (isalpha(c)) {
                if (l == K)
                    return string() + c;
                l++;
            }
            else if (l * (c - '0') <= K) {
                l *= c - '0';
            }
            else {
                K %= l;
                i = -1;
                l = 0;
            }
        }
        return "";
    }
};
