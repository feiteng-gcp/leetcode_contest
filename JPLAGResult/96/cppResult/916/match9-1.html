class Solution {
public:
    string decodeAtIndex(string S, int K) {
        long long total_len = 0;
        for(auto c: S) {
            if(isdigit(c)) {
                total_len *= c - '0';
            } else {
                ++total_len;
            }
            if(total_len >= K) {
                if(isdigit(c)) {
                    total_len /= c - '0';
                    int new_index = K % total_len;
                    if(new_index == 0) {
                        new_index = total_len;
                    }
                    return decodeAtIndex(S, new_index);
                } else {
                    return string(1, c);
                }
            }
        }
    }
};
