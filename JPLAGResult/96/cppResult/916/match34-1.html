class Solution {
public:
    string decodeAtIndex(string S, int K) {
        if (K == 1) return S.substr(0,1);
       
        const int n = S.size();
        vector<long> pos(n, 1);
        

        for (int i = 1; i < n; i++) {
            if (S[i] >= '2' && S[i] <= '9') {
                pos[i] = pos[i-1]*(S[i]-'0');
            } else {
                pos[i] = pos[i-1]+1;
            }
        }
        

        int j = n-1;
        
        while (1) {
            while (j > 0 && pos[j-1] >= K) j--;
            if (K == pos[j] && S[j] >= 'a' && S[j] <= 'z') {
                return string(1, S[j]);
            }
            j--;
            K = K % pos[j];
            if (K == 0) K = pos[j];
        }

    }
};
