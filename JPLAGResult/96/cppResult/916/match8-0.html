class Solution {
public:
    string s;
    
    string find(long long k) {
        long long len = 0;
        for (int i = 0 ; i<s.size() ; i++) {
            if (s[i] <='z' && s[i]>='a') {
                len ++;
                if (k == len) return s.substr(i,1);
            }
            else {
                int x = s[i] - 48;
                if (len * x < k) {
                    len *= x;
                    continue;
                }
                
                if (k % len == 0) {
                    for (int j = i;j>=0;j--)
                        if (s[j]<='z' && s[j]>='a') return s.substr(j,1);
                }
                
                if (len * x > k) {
                    return find(k % len);
                }
            }
        }
    }
    
    string decodeAtIndex(string S, int K) {
        this->s = S;
        return find(K);
    }
};
