class Solution {
public:
	
	char hehehe(string s,int n,int k)
    {
    	long long t=0;
    	for(int i=0;i<n;i++)
    	{
    		if(s[i]>='2'&&s[i]<='9')
    		{
    			if(t*(s[i]-'0')>=k)
    			{
    				if(k%t==0)return hehehe(s,i+1,t);
    				else return hehehe(s,i+1,k%t);
				}
				t*=(s[i]-'0');
			}
			else
			{
				t++;
				if(t==k)return s[i];
			}
		}
	}
    string decodeAtIndex(string S, int K) {
		string tem="";
        tem+=hehehe(S,S.size(),K);
        return tem;
    }
};
