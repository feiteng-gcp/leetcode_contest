class Solution {
public:
    string decodeAtIndex(string S, int K) {
        long long now = 0;
        for(int i = 0;i < S.length();i++){
            if(isdigit(S[i])){
                int d = S[i] - '0';
                if(d * now >= K){
                    string S1 = S.substr(0, i);
                    return decodeAtIndex(S1, (K - 1) % now + 1);
                }else{
                    now = now * d;
                }
            }else{
                now++;
                if(now == K) {
                    string temp = "";
                    temp += S[i];
                    return temp;
                }
            }
        }
        return 0;
    }
};
