class Solution {
public:
    string decodeAtIndex(string S, int K) {
        long long len = 0;
        
        // stack<pair<string,int>> stk;
        // stack<pair<int,int>> stk;
        // vector<vector<int>> stk;
        int n = S.length();
        // cout << S << " " << K << endl;
        for (int i = 0; i < n; ++i) {
            char c = S[i];
            if (c >= '2' && c <= '9') {
                // stk.emplace_back(vector<int>{len, c - '0', i);
                int cnt = c - '0';
                if (len * cnt >= K) {
                    return decodeAtIndex(S.substr(0, i), (K - 1) % len + 1);   
                }
                len *= cnt;
            } else {
                ++len;
                if (len == K) {
                    return string(1, c);
                }
            }
            // cout << len << endl;
        }
        
        return "";
    }
};
