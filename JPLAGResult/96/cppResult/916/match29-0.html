class Solution {
public:
    string decodeAtIndex(string S, int K) {
        vector<string> box;
        vector<long long> num;
        int i,j;
        string input;
        for(i=0;i<S.length();i++){
            if(S[i]>='a'&&S[i]<='z'){
                input += S[i];
            }
            else{
                box.push_back(input);
                input = "";
                num.push_back(S[i]-'0');
            }
        }
        
        int temp = K-1;
        if(num.empty()){
            string res;
            res += input[temp];
            return res;
        }
        for(i=num.size()-1;i>=0;i--){
            long long sum = 0;
            for(j=0;j<i;j++){
                sum += box[j].length();
                sum*=num[j];
            }
            sum += box[i].length();
            temp = (temp%sum);
            int item = sum-box[i].length();
            if(temp>=sum-box[i].length()&&temp<sum){
                string res;
                res += box[i][temp-item];
                return res;
            }
        }
        
    }
};
