class Solution {
public:
    string decodeAtIndex(string S, int K) {
        long i, j, d = 0, begin = 0, len;
        string t = "", p;
        len = 0;
        for(i = 0; i < S.size(); i++)
        {
            char ch = S[i];
            if(ch >= '1' && ch <= '9')
            {
                d = ch - '0';
                if(len * d >= K)
                {
                    return decodeAtIndex(S.substr(0, i), (K - 1) % len + 1);
                }
                len *= d;
            }
            else
            {
                len++;
                if(len == K)
                    return string(1, ch);
            }
        }
        return string(1, S.back());
    }
};
