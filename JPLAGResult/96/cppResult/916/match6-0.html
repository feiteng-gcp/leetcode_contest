class Solution {
public:
    string decodeAtIndex(string S, int K) {
        int64_t len = 0;
        for (int i = 0; i < S.size(); i++) {
            if (S[i] >= 'a' && S[i] <= 'z') {
                len++;
                if (len == K) return string(1, S[i]);
            } else {
                int repeat = S[i] - '0';
                if (len * repeat >= K) {
                    return decodeAtIndex(S, (int) ((K - 1) % len + 1));
                } else {
                    len *= repeat;
                }
            }
        }
    }
};
