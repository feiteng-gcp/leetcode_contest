class Solution {
public:
    string decodeAtIndex(string S, int K) {
        vector<long long> len(S.length(), 0);
        len[0] = 1;
        for (int i = 1; i < S.length(); ++i) {
            if ('2' <= S[i] && S[i] <= '9') {
                len[i] = 1ll * len[i - 1] * (S[i] - '0');
            } else {
                len[i] = len[i - 1] + 1;
            }
        }
        // cout << "K=" << K << endl;
        for (int i = 0; i < S.length(); ++i) {
            // cout << len[i] << endl;
            if (len[i] == K) {
                if ('a' <= S[i] && S[i] <= 'z') {
                    string ret(1, S[i]);
                    // cout << ret << endl;
                    return ret;
                } else {
                    return decodeAtIndex(S, K / (S[i] - '0'));
                }
            } else if (len[i] > K) {
                if (K % len[i - 1] == 0)  return decodeAtIndex(S, len[i - 1]);
                else  return decodeAtIndex(S, K % len[i - 1]);
            }
        }
        return "";
    }
};
