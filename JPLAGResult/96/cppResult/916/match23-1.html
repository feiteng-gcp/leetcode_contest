class Solution {
public:
	string decodeAtIndex(string S, int K) {
		long long tt = 0;
		int i = 0;
		for (; i < S.size(); i++) {
			if (isdigit(S[i])) {
				tt = tt * (S[i] - '0');
			} else {
				tt += 1;
			}
			if (tt >= K)
				break;
		}
		for (; i >= 0; --i) {
			if (isdigit(S[i])) {
				int tms = S[i] - '0';
				tt /= tms;

				K = (K-1) % tt + 1;
			} else {
				tt -= 1;
				if (tt < K) {
					return S.substr(i, 1);
				}
			}
		}
		return S.substr(0, 1);
	}
};
