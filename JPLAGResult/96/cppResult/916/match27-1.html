class Solution {
public:
    string decodeAtIndex(string S, int K) {
		long long cnt = 0, len = S.size();
		if (K == 1) return S.substr(0, 1);
		stringstream ss;

		for (int i=0;i<len;i++){
			char c = S[i];
			if (isalpha(c)) {
				cnt++;
				if (cnt==K) {
					ss << c;
					return ss.str();
				}
			}
			else {
				int num = c-'0';
				if (cnt*num>=K){
					return decodeAtIndex(S.substr(0,i),K%cnt==0? cnt:K%cnt);
				}
				cnt *= num;
			}
		}
		return ss.str();
    }
};
