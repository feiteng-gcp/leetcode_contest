class Solution {
public:
    string decodeAtIndex(string S, int K) {
        vector<long long> lens(S.size());

        lens[0]=1;
        for(int i=1;i<S.size();i++){
            if(S[i]>='2'&&S[i]<='9'){
                lens[i]=(S[i]-'0')*lens[i-1];
            }else{
                lens[i]=lens[i-1]+1;
            }
        }
        
        long long n=K-1;
        for(int i=S.size()-1;i>=0;i--){
            if(S[i]>='2'&&S[i]<='9'){
                int d=S[i]-'0';
                n=n%lens[i-1];
            }else{
                if(n+1==lens[i]){
                    return string(1,S[i]);
                }
            }
        }
        return "";
    }
};
