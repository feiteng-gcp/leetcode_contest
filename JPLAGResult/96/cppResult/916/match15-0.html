class Solution {
public:
    string decodeAtIndex(string S, int K) {
        
        int i = 0;
        long n = 0;
        vector<int> v;
        while (n < K) {
            if (isdigit(S[i])) {
                int d = S[i] - '0';
                n *= d;
                v.push_back(d);
            }
            else {
                n++;
                v.push_back(1);
            }
            
            i++;
        }
        
        i--;
        for (int j = v.size()-1; j >= 0; j--) {
            if (K == n && isalpha(S[i]))
                return string(1, S[i]);
            
            if (v[j] >= 2) {
                n /= v[j];
                if (K % n == 0)
                    K = n;
                else
                    K %= n;
            }
            else {
                n--;
            }
            
            i--;
        }
        
        return "";
    }
};
