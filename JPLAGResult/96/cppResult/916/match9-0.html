class Solution {
public:
    string decodeAtIndex(string S, int K) {
        vector<long> lens;
        long curr = 0;
        for (auto c: S) {
            if (c >= '2' && c <= '9') {
                curr *= c - '0';
            } else {
                curr++;
            }
            lens.push_back(curr);
            //cout << c << ' ' << lens[lens.size() - 1] << endl;
        }
        
        int r = S.size() - 1;
        long k = K;
        
        while (r >= 0) {
            while (lens[r] > k) {
                r--;
            }
            //cout << k << ' ' << r << ", " << S[r] << ' '<< lens[r] << endl;
            k %= lens[r];
            if (k == 0) {
                if (S[r] >= '2' && S[r] <= '9') {
                    r--;
                    k = lens[r];
                } else {
                    break;
                }
            } else{
                r--;
            }
        }
        
        return string(1, S[r]);
    }
};
