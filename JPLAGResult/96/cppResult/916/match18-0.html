class Solution {
public:
    using int64 = long long;
    string decodeAtIndex(string S, int K) {
        int len = S.length();
        int64 sum = 0;
        for (int i = 0; i < len; i++) {
            if (S[i] >= 'a' && S[i] <= 'z') {
                sum++;
                if (sum == K) {string str = ""; str += S[i]; return str;}
            } else {
                int64 tmp = sum * (S[i] - '0');
                if (K > tmp) {sum = tmp; continue;}
                int mod = K % sum;
                if (mod != 0) return decodeAtIndex(S, K % sum);
                return decodeAtIndex(S, sum);
            }
        }
    }
};
