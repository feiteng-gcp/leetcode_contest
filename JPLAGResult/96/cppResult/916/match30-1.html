typedef unsigned long long uint64;

class Solution {
public:
    vector<uint64> lengths;
    string decodeAtIndex(string S, int K) {
        uint64 count = 0ull;
        uint64 KK = (uint64) K;
        for (int i = 0; i < S.size(); i ++) {
            char c = S[i];
            if (c >= 'a' and c <= 'z') { // letter

                count ++;
            } else {
                count *= (c - '0');
            }
            lengths.push_back(count);
            if (count >= KK) { // K-th letter already generated     
                return decode(S, i, KK - 1);
            }
        }
        return "";
    }
    string decode(string S, int i, uint64 KK) {
        if (i == 0) return string(1, S[0]);
        
        // Backtrack, KK is 0-based
        char c = S[i];
        // cout << S << " " << i << " " << lengths[i - 1] << " " <<  KK << endl;
        if (c >= 'a' and c <= 'z') { // letter
            return (KK == lengths[i] - 1) ? string(1, c) : decode(S, i - 1, KK);
        } else { // number
            int ic = (c - '0');
            return decode(S, i - 1, KK % lengths[i - 1]);
        }
    }
};
