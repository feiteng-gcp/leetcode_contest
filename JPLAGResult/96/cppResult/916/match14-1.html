using ull = unsigned long long;
class Solution {
public:
    string decodeAtIndex(string S, int K) {
        int len = S.length();
        ull res = 0;
        for(int i=0;i<len;i++){
            ull tmp = 0;
            if(S[i]>='1'&&S[i]<='9'){
                tmp=res*(S[i]-'0');
                if(tmp>=K){
                    K%=res;
                    if(!K) K = res;
                    return decodeAtIndex(S,K);
                }
            }
            else{
                tmp=res+1;
                if(tmp==K){
                    string rt;
                    rt+=S[i];
                  
                    return rt;
                }
            }
            res = tmp;
            
        }
    }
};
