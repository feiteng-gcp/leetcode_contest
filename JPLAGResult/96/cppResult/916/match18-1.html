typedef long long LL;

class Solution {
public:

    string decodeAtIndex(string s, int k) {
        int n = s.size();
        vector<LL> a(n);
        LL x = 0;
        for (int i = 0; i < n; ++i) {
            if ('a' <= s[i] && s[i] <= 'z') {
                ++x;
            } else {
                x *= s[i] - '0';
            }
            a[i] = x;
        }
        --k;
        for (int i = n; i-- > 0; ) {
            if ('a' <= s[i] && s[i] <= 'z') {
                if (a[i] == k + 1) return string(1, s[i]);
            } else {
                k %= a[i - 1];
            }
        }
        return "?";
    }
};
