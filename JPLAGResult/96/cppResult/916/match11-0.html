class Solution {
public:
    string decodeAtIndex(string S, int K) {
        long long len = 0;
        for (char ch : S) {
            if (ch >= 'a' && ch <= 'z') {
                ++len;
            } else {
                len *= (ch - '0');
            }
        }

        --K;
        for (int i = S.length() - 1; i >= 0; --i) {
            if (S[i] >= 'a' && S[i] <= 'z') {
                if (len == K + 1) {
                    return S.substr(i, 1);
                } else {
                    --len;
                }
            } else {
                len /= (S[i] - '0');
                K %= len;
            }
        }

        return S.substr(K, 1);
    }
};
