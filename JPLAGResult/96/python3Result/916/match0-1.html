class Solution:
    def decodeAtIndex(self, S, K):
        """
        :type S: str
        :type K: int
        :rtype: str
        """

        t = []
        l = []
        for c in S:
            if ord(c) >= ord('a') and ord(c) <= ord('z'):
                t.append(c)
            else:
                l.append("".join(t))
                t = []
                l.append(int(c))

        if len(t) > 0:
            l.append("".join(t))

        count = 0
        lengths = []
        for e in l:
            if isinstance(e, str):
                count += len(e)
            else:
                count *= e
            lengths.append(count)


        return self.lei(l, lengths, K)


    
def lei(self, l, lengths, K):
        lo = 0
        hi = len(l) - 1
        while lo < hi:
            mid = (lo + hi) // 2
            if lengths[mid] < K:
                lo = mid + 1
            else:
                hi = mid

        #print(lo)

        group_index = lo
        if isinstance(l[group_index], int):
            haha = K % lengths[group_index-1]
            if haha == 0:
                haha = lengths[group_index-1]
            return self.lei(l, lengths, haha)
        else:
            if group_index == 0:
                return l[group_index][K-1]
            return l[group_index][K-1-lengths[group_index-1]]
