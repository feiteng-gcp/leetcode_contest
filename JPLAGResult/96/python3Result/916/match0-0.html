class Solution:
    def decodeAtIndex(self, S, K):
        """
        :type S: str
        :type K: int
        :rtype: str
        """
        ll = [0]
        for i, c in zip(range(len(S)), S):
            l = ll[-1]
            if '2' <= c <= '9':
                l *= int(c)
                ll.append(l)
            else:
                l +=1
                ll.append(l)


        def bs(ll, K):
            a = 0
            b = len(ll)
            while a < b-1:
                m = ( a + b ) // 2 
                if ll[m] < K:
                    a = m
                else:
                    b = m
            return b


        def find_chr(S, K, ll):
            anchor = bs(ll, K)

            if '2' <= S[anchor-1] <= '9': 
                new_K = K % ll[anchor-1]
                if new_K == 0 :
                    while '2' <= S[anchor-1] <= '9': 
                        anchor -= 1
                    return S[anchor-1]
                else:
                    return find_chr(S, new_K, ll)
            else:
                return S[anchor-1]
            
            
        return find_chr(S, K, ll)
