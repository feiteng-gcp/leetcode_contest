class Solution:
    def projectionArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n = len(grid)
        m = len(grid[0])
        a = [0 for i in range(n)]
        b = [0 for i in range(m)]
        ans = 0
        for i in range(n):
            for j in range(m):
                if (grid[i][j]>0):
                    ans = ans + 1
                
if (grid[i][j]>a[i]):
                    a[i] = grid[i][j]
                if (grid[i][j]>b[j]):
                    b[j] = grid[i][j]
        ans = ans + sum(a) + sum(b)
        return ans
