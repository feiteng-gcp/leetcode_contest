class Solution:
    def projectionArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        total = 0
        m = len(grid)
        n = len(grid[0])
        for x in grid:
            for y in x:
                if y:
                    total += 1
        for i in range(m):
            best = 0
            for j in range(n):
                best = max(best, grid[i][j])
            total += best
        for i in range(n):
            best = 0
            for j in range(m):
                best = max(best, grid[j][i])
            total += best
        return total
