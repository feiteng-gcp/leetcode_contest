class Solution:
    def projectionArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """

        m, n = len(grid), len(grid[0])
        s = 0
        for i in range(m):
            mv = -1
            for j in range(n):
                mv = max(mv, grid[i][j])
                if grid[i][j]: s +=
 1
            s += mv
        for j in range(n):
            mv = -1
            for i in range(m):
                mv = max(mv, grid[i][j])
            s += mv
        return s
