class Solution:
    def projectionArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        N = len(grid)
        s_xy, s_xz, s_yz = 0, 0, 0
        
for i in range(N):
            tmp = 0
            for j in range(N):
                if grid[j][i] > tmp:
                    tmp = grid[j][i]
                if grid[i][j] != 0:
                    s_xy += 1
            s_yz += tmp
        for i in range(N):
            s_xz += max(grid[i])
        return s_xy + s_xz + s_yz
