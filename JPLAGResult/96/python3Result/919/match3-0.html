class Solution:
    def projectionArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        N = len(grid)
        area_x = 0
        area_y = 0
        area_z = 0 
        for i in range(N):
            max_y = 0
            for j in range(N):
                if grid[i][j] != 0:
                    area_x += 1
                if grid[i][j] > max_y:
                    max_y = grid[i][j]
            area_y += max_y 
        for j in range(N):
            max_z = 0 
            for i in range(N):
                if grid[i][j] > max_z:
                    max_z = grid[i][j]
            area_z += max_z 
        return area_x + area_y + area_z
