class Solution:
    def projectionArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        area = 0
        n = len(grid)
        m = len(grid[0])
        
for r in range(n):
            for c in range(m):
                if grid[r][c] > 0:
                    area += 1
        
        for r in range(n):
            area += max(grid[r])
        
        for c in range(m):
            area += max([grid[r][c] for r in range(n)])
        
        return area
