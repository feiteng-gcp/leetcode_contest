class Solution:
    def projectionArea(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        m = len(grid)
        if m==0:
            return 0
        n = len(grid[0])
        a = 0
        rowMax = [0] * m
        colMax = [0] * n

        for i in range(m):
            for j in range(n):
                if grid[i][j] > 0:
                    a += 1
                    if grid[i][j] > rowMax[i]:
                        rowMax[i] = grid[i][j]
                    if grid[i][j] > colMax[j]:
                        colMax[j] = grid[i][j]

        for i in range(m):
            a += rowMax[i]
        for i in range(n):
            a += colMax[i]
        return a
