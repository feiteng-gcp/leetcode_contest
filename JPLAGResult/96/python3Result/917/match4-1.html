class Solution:
    def numRescueBoats(self, people, limit):
        """
        :type people: List[int]
        :type limit: int
        :rtype: int
        """
        people.sort()
        left = 0
        right = len(people) - 1
        nb_boats = 0
        while left <= right:
            if left == right:
                nb_boats += 1
                break
            elif people
[right] + people[left] <= limit:
                nb_boats += 1
                right -= 1
                left += 1
            else:
                nb_boats += 1
                right -= 1
        
        return nb_boats
