import heapq

class Solution:
    def numRescueBoats(self, people, limit):
        """
        :type people: List[int]
        :type limit: int
        :rtype: int
        """
        
        if not people:
            return 0

        people = sorted
(people)

        start, end = 0, len(people) - 1
        n_boats = 0
        while start < end:
            if people[start] + people[end] <= limit:
                start += 1
                end -= 1
                n_boats += 1
            else:
                end -= 1
                n_boats += 1
        if start == end:
            n_boats += 1

        return n_boats
