class Solution:
    def numRescueBoats(self, people, limit):
        """
        :type people: List[int]
        :type limit: int
        :rtype: int
        """
        people.sort()

        boats = 0

        i = 0
        j = len(people) - 1

        while i <= j:
            
if people[i] + people[j] <= limit:
                boats += 1
                i += 1
                j -= 1
            else:
                j -= 1
                boats += 1
        return boats
