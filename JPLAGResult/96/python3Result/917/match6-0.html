class Solution:
    def numRescueBoats(self, people, limit):
        """
        :type people: List[int]
        :type limit: int
        :rtype: int
        """
        count = 0
        people.sort()
        j = len(people)-1
        i = 0
        while i <= j:
            if i == j:
                count += 1
                return count
            if people[j] + people[i] <= limit:
                count += 1
                j -= 1
                i += 1
            else:
                count += 1
                j -= 1
        return count
