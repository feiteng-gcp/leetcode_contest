class Solution:
    def numRescueBoats(self, people, limit):
        """
        :type people: List[int]
        :type limit: int
        :rtype: int
        """
        if len(people) == 1: return 1
        people.sort()
        i, j = 0, len(people) - 1
        cnt = 0
        while i < j:
            if people[i] + people[j] <= limit:
                cnt += 1
                i += 1
                j -= 1
            else:
                cnt += 1
                j -= 1
        cnt += (i == j)
        return cnt
