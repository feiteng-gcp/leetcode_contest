class Solution {
    public int[][] transpose(int[][] A) {
        int len = A.length;
        if (len == 0) {
            return A;
        }
        int wid = A[0].length;
        int[][] res = new int[wid][len];
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < wid; j++) {
                res[j][i] = A[i][j];
            }
        }
        return res;
    }
}
