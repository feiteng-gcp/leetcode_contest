class Solution {

	private int[] dx = new int[] {- 1 , 0 , 1 , 0};
	private int[] dy = new int[] {0 , 1 , 0 , - 1};
	
	private
 static class Element {		
		public int x , y , bitmap , dist;
		public Element(int x , int y , int bitmap , int dist) {
			this.x = x;
			this.y = y;
			this.bitmap = bitmap;
			this.dist = dist;
		}
		@Override
		public int hashCode() {
			int ans = 31;
			ans = ans * 37 + x;
			ans = ans * 37 + y;
			ans = ans * 37 + bitmap;
			return ans;
		}
		public boolean equals(Object object) {
			Element e = (Element) object;
			return this.x == e.x && this.y == e.y && this.bitmap == e.bitmap;
		}
	}
	
	private Set<Element> vis = new HashSet<>();
	
    public int shortestPathAllKeys(String[] grid) {
     
    	int i , j , n = grid.length , m = grid[0].length() , cnt = 0;
    	Set<Character> set = new HashSet<>();
    	int x = 0 , y = 0;
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < m;j ++) {
    			if (grid[i].charAt(j) == '@') {
    				x = i;
    				y = j;
    			} else if (grid[i].charAt(j) >= 'a' && grid[i].charAt(j) <= 'z') {
    				set.add(grid[i].charAt(j));
    			}
    		}
    	}
    	// key number
    	cnt = set.size();
    	if (cnt == 0) {
    		return 0;
    	}
    	
    	Queue<Element> queue = new LinkedList<>();
    	vis.add(new Element(x , y , 0 , 0));
    	queue.add(new Element(x , y , 0 , 0));
    	while (!queue.isEmpty()) {
    		Element current = queue.poll();
    		for (i = 0;i < 4;i ++) {
    			int tx = current.x + dx[i];
    			int ty = current.y + dy[i];
    			if (tx >= 0 && tx < n && ty >= 0 && ty < m) {
    				if (grid[tx].charAt(ty) != '#') {
    					if (grid[tx].charAt(ty) >= 'a' && grid[tx].charAt(ty) <= 'z') {
    						int index = grid[tx].charAt(ty) - 'a';
    						int nextBitmap = current.bitmap;
    						if ((nextBitmap & (1 << index)) == 0){
    							nextBitmap = nextBitmap | (1 << index);
    						}
    						int nextDist = current.dist + 1;
    						Element e = new Element(tx , ty , nextBitmap , nextDist);
    						
    						if (e.bitmap == (1 << cnt) - 1) {

    							return e.dist;
    						}
    						
    						if (!vis.contains(e)) {
    							vis.add(e);
    							queue.add(e);
    						}
    					} else if (grid[tx].charAt(ty) >= 'A' && grid[tx].charAt(ty) <= 'Z') {
    						int index = grid[tx].charAt(ty) - 'A';
    						if ((current.bitmap & (1 << index)) != 0) {
    							Element e = new Element(tx , ty , current.bitmap , current.dist + 1);
    							
    							if (!vis.contains(e)) {
    								vis.add(e);
    								queue.add(e);
    							}
    						}
    					} else {
    						Element e = new Element(tx , ty , current.bitmap , current.dist + 1);
    						if (!vis.contains(e)) {
    							vis.add(e);
    							queue.add(e);
    						}
    					}
    				}
    			}
    		}
    	}
    	return - 1;
    	
    }
}
