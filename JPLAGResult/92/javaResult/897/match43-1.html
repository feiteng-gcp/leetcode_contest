import java.util.ArrayList;
class Solution {
    public int primePalindrome(int N) {
        if (N == 1)
            return 2;
        N--;
        while (true) {
            N = nextPalindrome(N);
            if (isPrime(N))
                return N;
        }
    }
    
    int nextPalindrome(int N) {
        if (N == 0)
            return 1;
        N++;
        ArrayList<Integer> digits = new ArrayList<>();
        while (N != 0) {
            digits.add(N % 10);
            N /= 10;
        }
        int left = 0;
        int right = digits.size() - 1;
        while (left < right) {
            int ld = digits.get(left);
            int rd = digits.get(right);
            if (ld == rd) {
                left++;
                right--;
                continue;
            }
            if (ld < rd) {
                digits.set(left, rd);
                left++;
                right--;
                continue;
            }
            // following: ld > rd
            if (left + 1 == right) {
                digits.set(left, rd + 1);
                digits.set(right, rd + 1);
                break;
            }
            int next = digits.get(left + 1);
            if (next != 9) {
                digits.set(left, rd);
                digits.set(left + 1, next + 1);
                left++;
                right--;
                continue;
            }
            int i = left + 2;
            int num = 0;
            while ((num = digits.get(i)) == 9)
                i++;
            digits.set(i, num + 1);
            int factor = 1;
            for (int j = 0; j < i; j++)
                factor *= 10;
            int sum = 0;
            for (int j = i; j < digits.size(); j++) {
                sum += factor * digits.get(j);
                factor *= 10;
            }
            return nextPalindrome(sum);
        }
        int sum = 0;
        int factor = 1;
        for (int j = 0; j < digits.size(); j++) {
            sum += factor * digits.get(j);
            factor *= 10;
        }
        return sum;
    }
    
    boolean isPrime(int N) {
        int sqrt = (int) Math.sqrt(N);
        
for (int i = 2; i <= sqrt; i++)
            if (N % i == 0)
                return false;
        return true;
    }
}
