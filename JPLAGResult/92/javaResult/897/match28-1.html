class Solution {
    
    
    int tm = Integer.MAX_VALUE;
    public int primePalindrome(int N) {
        if(N==1){
            return 2;
        }
        String n = "" + N;
        int l = n.length();
        
       while(tm == Integer.MAX_VALUE){
        if(l%2==0){
            getn("",l/2,N);
            if(tm!=Integer.MAX_VALUE){
                
return tm;
            }
            for(int i=0;i<=9;i++){
                getn(""+i,l/2,N);
            } 
        }else{
            for(int i=0;i<=9;i++){
                getn(""+i,l/2,N);
            } 
            if(tm!=Integer.MAX_VALUE){
                return tm;
            }
            
            getn("",l/2+1,N);
        }
        
            l++;
            
        }
        
        
        
        
return tm;
        
    }
    
    public void getn(String b , int n,int N){
        if(n==0){
            if(b.charAt(0)!='0'){
                int r = Integer.parseInt(b);
                if(r>=N&&r<tm&&isP(r)){
                    tm = r;
                }
            }
            return;
            
        }
        if(n>0){
            for(int i=0;i<=9;i++){
                getn(""+i+b+i,n-1,N);   
            }
        }
    }
    
    
    public boolean  isP(int num){
        for(int i=2;i*i<=num;i++)
{
            if(num%i==0){
                return false;
            }
        }
        return true;
    }
}
