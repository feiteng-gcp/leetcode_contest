class Solution {
    public int primePalindrome(int N) {
        String s = String.valueOf(N);
        String left = s.substring(0, (s.length() + 1) / 2);
        int p = Integer.parseInt(left + new StringBuilder(left.substring(0, s.length() - left.length())).reverse().toString());
        N = (p >= N) ? p : next(p);
        while (true) {
            if (isPrime(N)) {
                return N;
            }
            N = next(N);
        }
    }

    private boolean isPrime(int N) {
        if (N == 1) {
            return false;
        }
        int hi = (int) Math.sqrt(N);
        
for (int i = 2; i <= hi; i++) {
            if (N % i == 0) {
                return false;
            }
        }
        return true;
    }

    private int next(int N) {
        String s = String.valueOf(N);
        if (s.matches("9+")) {
            return (int) Math.pow(10, s.length()) + 1;
        }
        String left = s.substring(0, (s.length() + 1) / 2);
        String nextLeft = String.valueOf(Integer.parseInt(left) + 1);
        String nextRight = new StringBuilder(nextLeft.substring(0, s.length() - nextLeft.length())).reverse().toString();
        return Integer.parseInt(nextLeft + nextRight);
    }

    public static void main(String[] args) {
        Solution s = new Solution();
//        System.out.println(Arrays.deepToString(s.transpose(new int[][] {{1,2,3},{4,5,6}})));
        System.out.println(s.isPrime(1));
        System.out.println(s.isPrime(11));
        System.out.println(s.isPrime(101));
        System.out.println(s.next(8));
        System.out.println(s.next(99));
        System.out.println(s.primePalindrome(6));
        System.out.println(s.primePalindrome(8));
        System.out.println(s.primePalindrome(13));
        System.out.println(s.primePalindrome(1));
        System.out.println(s.primePalindrome(11));
        System.out.println(s);
        System.out.println(s);
    }
}
