class Solution {
  public int primePalindrome(int N) {
    if (N>=10000000) {
      return 100030001;
    }
    for (int i = N;; i++) {
      if (i > 2 && i % 2 == 0)
        continue;
      if (isFalindrome(i) && isPrime(i)) {
        return i;
      }
    }
  }

  boolean isPrime(int num) {
    if (num < 2)
      return false;
    for (int i = 2; i <= Math.sqrt(num); ++i) 
{
      if (num % i == 0) {
        return false;
      }
    }
    return true;
  }

  boolean isFalindrome(int num) {
    int rev = 0;
    int ori = num;
    while (num > 0) {
      rev = rev * 10 + num % 10;
      num /= 10;
    }
    return rev == ori;
  }
}
