class Solution {
     public int primePalindrome(int N) {
            N = findHigherPalindrome(N);
            while (!checkPrim(N)) {
                N = findHigherPalindrome(N + 1);
            }
            return N;
    }
    public boolean checkPrim(int n) {
        if (n == 1) {
            return false;
        }
        
for (int i = 2; i <= n / 2; i++) {
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }
    public int findHigherPalindrome(long origin) {
        String n = Long.toString(origin);
        char[] s = n.toCharArray(); // limit
        int sLen = s.length;
        char[] res = new char[sLen];
        for (int i = 0; i <= sLen / 2; i++) {
            res[sLen - 1 - i] = s[i];
            res[i] = s[i];
        }
        for (int i = 0; i < sLen; i++) {
            if (s[i] < res[i]) {
                return Integer.parseInt(String.valueOf(res));
            } else if (s[i] > res[i]) {
                for (int j = (sLen - 1) / 2; j >= 0; j--) {
                    if (++res[j] > '9') {
                        res[j] = '0';
                    } else {
                        break;
                    }
                }
                for (int k = 0; k < sLen / 2; k++) {
                    res[sLen - 1 - k] = res[k];
                }
                return Integer.parseInt(String.valueOf(res));
            }
        }
        return Integer.parseInt(String.valueOf(res));
    }
}
