class Solution {
    public int primePalindrome(int N) {
        for (int i = N; i < 10; i++) {
            if (isPrime(i)) return i;
        }
        
        if (N <= 11) return 11;
        
        for (int dOne = 1; dOne <= 9; dOne = dOne + 2) {
            for (int dTwo = 0; dTwo <= 9; dTwo++) {
                int palin = 100*dOne + 10*dTwo + dOne;
                if (palin >= N && isPrime(palin)) return palin;
            }
        }
        
        for (int dOne = 1; dOne <= 9; dOne = dOne + 2) {
            for (int dTwo = 0; dTwo <= 9; dTwo++) {
                for (int dThree = 0; dThree <= 9; dThree++) {
                    int palin = 10000*dOne + 1000*dTwo + 100*dThree + 10*dTwo + dOne;
                    if (palin >= N && isPrime(palin)) return palin;
                }
            }
        }
        
        for (int dOne = 1; dOne <= 9; dOne = dOne + 2) {
            for (int dTwo = 0; dTwo <= 9; dTwo++) {
                for (int dThree = 0; dThree <= 9; dThree++) {
                    for (int dFour = 0; dFour <= 9; dFour++) {
                        int palin = 1000000*dOne + 100000*dTwo + 10000*dThree + 1000*dFour + 100*dThree + 10*dTwo + dOne;
                        if (palin >= N && isPrime(palin)) return palin;
                    }

                }
            }
        }
        return 100030001;
    }
    
    public boolean isPrime(int n) {
        if (n < 2) return false;
        for (int i = 2; i <= Math.sqrt(n); i++) {
            if (n % i == 0) return false;
        }
        return true;
    }
}
