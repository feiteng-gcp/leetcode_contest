import java.math.BigInteger;

class Solution {
    private static boolean isPrime(int num) {
        if (num < 2) return false;
        if (num == 2) return true;
        if (num % 2 == 0) return false;
        
for (int i = 3; i * i <= num; i += 2)
            if (num % i == 0) return false;
        return true;
    }
    
    private boolean allNines(int n){
        while (n >= 1) {
            if (n % 10 != 9) {
                return false;
            }
            n /= 10;
        }
        return true;
       
    }
       
    private boolean isParlindrome(int n) {
        String s = String.valueOf(n);
        for(int i = 0; i < s.length()/2; i++) {
            if(s.charAt(i) != s.charAt(s.length() - 1 - i)) {
                return false;
            }
        }
        
        return true;
    }

  
    public static String fnextPalindrome(String num) {
        int len = num.length();
        String left = num.substring(0, len / 2);
        String middle = num.substring(len / 2, len - len / 2);
        String right = num.substring(len - len / 2);

        if (right.compareTo(reverse(left)) < 0)
            return left + middle + reverse(left);

        String next = new BigInteger(left + middle).add(BigInteger.ONE).toString();
        return next.substring(0, left.length() + middle.length())
             + reverse(next).substring(middle.length());
    }

    private static String reverse(String s) {
        return new StringBuilder(s).reverse().toString();
    }


    public int primePalindrome(int N) {
        int n = N;
        while(true) {
            if(!isParlindrome(n)) {
                n = Integer.valueOf(fnextPalindrome(String.valueOf(n)));
            }
            if(!isPrime(n)) {
                n++;
                continue;
            }
            return n;
        }
    }
}
