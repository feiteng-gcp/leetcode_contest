class Solution {
    public int primePalindrome(int N) {
        if (N == 1 || N == 2) return 2;
        if (N >= 9989900) return 100030001;
        int m = N;
        while (true) {
            if (isPalindrome(m) && isPrime(m)) {
                return m;
            }
            m++;
        }
    }

    private boolean isPrime(int m) {
        int p = (int) Math.sqrt(m) + 1;
        int i;
        for (i = 2; i <= p; i++) 
{
            if ((m % i) == 0) {
                return false;
            }
        }

        return true;
    }

    private boolean isPalindrome(int m) {
//        String val = String.valueOf(m);
//        char[] arr = val.toCharArray();
//        int i;
//        for (i = 0; i < arr.length / 2; i++) {
//            if (arr[i] != arr[arr.length - i - 1]) {
//                return false;
//            }
//        }
//        return true;
        int n = m;
        int rev = 0;
        while (m > 0) {
            int dig = m % 10;
            rev = rev * 10 + dig;
            m = m / 10;
        }
        return n == rev;
    }

}
