class Solution {
    public int primePalindrome(int n) {
        if (n == 1) {
            return 2;
        }
        
        if (n > 10000000) {
           return 100030001;
        }
        
        while(!(isPalindrome(n) && isPrime(n))) {
            n++;
        }
        
        return n;
    }
    
    private boolean isPrime(int n) {
        if (n == 2) {
            return true;
        } 
        
        for (int i = 2; i < ((int) Math.sqrt(n) + 1); i++) {
            if (n % i == 0) {
                return false;
            }
        }
        
        return true;
    }
    
    private boolean isPalindrome(int n) {
        int r = 0;
        int tmp = n;
        while (tmp > 0) {
            r = r * 10 + tmp % 10;
            tmp = tmp / 10;
        }
        
        return r == n;
    }
}
