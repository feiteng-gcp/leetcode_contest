class Solution {
    public int primePalindrome(int N) {
        List<Integer> palindrone = getFirstPalindrone(N);
        while (true) {
            int number = toInt(palindrone);
            if (isPrime(number)) {
                return number;
            }
            palindrone = getNextPalindrone(palindrone);
        }
    }

    private int toInt(List<Integer> numbers) {
        int n = 0;
        for (int i = 0; i < numbers.size(); i++) {
            n *= 10;
            n += numbers.get(i);
        }

        return n;
    }

    private List<Integer> toList(int n) {
        List<Integer> numbers = new ArrayList<>();
        while (true) {
            if (n == 0) {
                break;
            }

            int r = n % 10;
            int q = n / 10;
            numbers.add(r);
            n = q;
        }

        return numbers;
    }

    private List<Integer> getFirstPalindrone(int startIndex) {
        int i = startIndex;
        while (true) {
            if (isPalindrone(i)) {
                return toList(i);
            }
            i += 1;
        }
    }

    private List<Integer> getNextPalindrone(List<Integer> numbers) {
        int n = numbers.size();
        if (n % 2 == 0) {
            return getNextPalindroneEven(numbers);
        }
        return getNextPalindroneOdd(numbers);
    }

    private List<Integer> getNextPalindroneEven(List<Integer> numbers) {
        int n = numbers.size();
        int half = n / 2;
        for (int i = 0; i < half; i++) {
            int value = numbers.get(half + i);
            if (value < 9) {
                numbers.set(half + i, value + 1);
                numbers.set(half - 1 - i, value + 1);
                for (int j = half - i; j < (half + i); j++) {
                    numbers.set(j, 0);
                }

                return numbers;
            }
        }

        List<Integer> newNumbers = new ArrayList<>();
        newNumbers.add(1);
        for (int i = 0; i < n - 1; i++) {
            newNumbers.add(0);
        }
        newNumbers.add(1);
        return newNumbers;
    }

    private List<Integer> getNextPalindroneOdd(List<Integer> numbers) {
        int n = numbers.size();
        int half = n / 2;
        int mid = numbers.get(half);
        if (mid < 9) {
            numbers.set(half, mid + 1);
            return numbers;
        }

        for (int i = 0; i < half; i++) {
            mid = numbers.get(half - i - 1);
            if (mid < 9) {
                numbers.set(half - i - 1, mid + 1);
                numbers.set(half + i + 1, mid + 1);
                for (int j = half - i; j <= (half + i); j++) {
                    numbers.set(j, 0);
                }
                return numbers;
            }
        }

        List<Integer> newNumbers = new ArrayList<>();
        newNumbers.add(1);
        for (int i = 0; i < n - 1; i++) {
            newNumbers.add(0);
        }
        newNumbers.add(1);
        return newNumbers;
    }

    private boolean isPalindrone(int n) {
        List<Integer> numbers = toList(n);
        int count = numbers.size();
        int half = count / 2;
        for (int i = 0; i < half; i++) {
            if (numbers.get(i).intValue() != numbers.get(count - 1 - i).intValue()) {
                return false;
            }
        }

        return true;
    }


    private boolean isPrime(int n) {
        if (n < 2) {
            return false;
        }
        
for (int i = 2; i * i <= n; i++) {
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }
}
