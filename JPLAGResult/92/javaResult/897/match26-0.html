class Solution {
    
      private boolean isPrime(int n) {

        if (n > 2 && (n & 1) == 0)
            return false;

        for (int i = 3; i * i <= n; i += 2)
            if (n % i == 0)
                return false;
        return true;
    }

    private static boolean isAlphaNumeric(char c) {
        char newC = Character.toLowerCase(c);
        return (newC >= '0' && newC <= '9') || (newC >= 'a' && newC <= 'z');
    }

    private static boolean isPalindrome(String s) {

        String l = s.toLowerCase();

        int i = 0;
        int j = l.length() - 1;
        while (j > i) {
            while (i < s.length() && !isAlphaNumeric(l.charAt(i))) {
                i++;
            }

            while (j >= 0 && !isAlphaNumeric(l.charAt(j))) {
                j--;
            }

            if (i < s.length() && j >= 0 && l.charAt(i) != l.charAt(j)) {
                return false;
            } else {
                i++;
                j--;
            }
        }

        return true;
    }


    public int primePalindrome(int N) {

        if (N==1){
            return 2;
        }
        
        
        if (N>=9989900){
            return 100030001;
        }
        
        int startNumber = N;
        while (!(isPalindrome(startNumber + ""))||!(isPrime(startNumber)) ) {
            startNumber++;
        }

        //System.out.println(startNumber);
        return startNumber;
    }
}
