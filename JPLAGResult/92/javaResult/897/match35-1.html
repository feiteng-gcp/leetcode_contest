class Solution {
    private boolean isPrime(int N) {
        if (N <= 1) {
            return false;
        }
        int sq = (int) Math.sqrt(N) + 1;
        
for (int i = 2; i <= sq && i < N; i++) {
            if (N % i == 0) {
                return false;
            }
        }
        return true;
    }

    public int primePalindrome(int N) {
        int num = N;
        while (true) {
            int next = Integer.parseInt(nearestPalindromic(num + ""));
            if (isPrime(next)) {
                return next;
            }
            num = next + 1;
        }
    }
    
    public String nearestPalindromic(String n) {
        if (n.equals("1"))
            return "1";

        String a = mirroring(n);
        if (a.equals(n)) {
            return n;
        }

        StringBuilder s = new StringBuilder(n);
        int i = (s.length() - 1) / 2;
        while (i >= 0 && s.charAt(i) == '0') {
            s.replace(i, i + 1, "9");
            i--;
        }
        if (i == 0 && s.charAt(i) == '1') {
            s.delete(0, 1);
            int mid = (s.length() - 1) / 2;
            s.replace(mid, mid + 1, "9");
        } else
            s.replace(i, i + 1, "" + (char)(s.charAt(i) - 1));
        String b = mirroring(s.toString());

        s = new StringBuilder(n);
        i = (s.length() - 1) / 2;
        while (i >= 0 && s.charAt(i) == '9') {
            s.replace(i, i + 1, "0");
            i--;
        }
        if (i < 0) {
            s.insert(0, "1");
        } else
            s.replace(i, i + 1, "" + (char)(s.charAt(i) + 1));
        String c = mirroring(s.toString());
        long[] tmp = new long[3];
        tmp[0] = Long.parseLong(a);
        tmp[1] = Long.parseLong(b);
        tmp[2] = Long.parseLong(c);
        long nn = Long.parseLong(n);
        long res = Long.MAX_VALUE;
        for (int ii = 0; ii < 3; ii++) {
            if (tmp[ii] >= nn) {
                res = Math.min(res, tmp[ii]);
            }
        }
        return res + "";
    }
    
    private String mirroring(String s) {
        String x = s.substring(0, (s.length()) / 2);
        return x + (s.length() % 2 == 1 ? s.charAt(s.length() / 2) : "") + new StringBuilder(x).reverse().toString();
    }
    
}
