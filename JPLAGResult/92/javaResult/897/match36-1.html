class Solution {
    public int primePalindrome(int N) {
        if (N == 1) return 2;
        int len = 0;
        int tmp = N;
        while (tmp > 0) {
            tmp /= 10;
            len++;
        }
        while (len <= 9) {
            List<Long> list = generateNDigitPalindrome(len);
            // System.out.println(len+" "+list);
            for (long item : list) {
                if (item >= N && checkPrime(item)) {
                    return (int) item;
                }
            }
            len++;
        }
        return -1;
    }
    private List<Long> generateNDigitPalindrome(int n) {
        int len = n / 2;
        List<Long> res = new ArrayList<>();
        if (n % 2 == 0) {
            dfs(res, len, "");
        } else {
            for (int i = 0; i <= 9; i++) {
                dfs(res, len, String.valueOf(i));
            }
        }
        Collections.sort(res);
        return res;
    }
    private void dfs(List<Long> res, int len, String tmp) {
        if (tmp.length() / 2 == len) {
            if (tmp.charAt(0) != '0')
                res.add(Long.valueOf(tmp));
            return;
        }
        for (int i = 0; i <= 9; i++) {
            dfs(res, len, i + tmp + i);
        }
    }
    private boolean checkPrime(long n) {
        for (int i = 2; i <= Math.sqrt(n); i++) 
{
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }
}
