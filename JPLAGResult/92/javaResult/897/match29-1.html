class Solution {
    static TreeSet<Integer> set = new TreeSet<>();
    static {
        boolean prime[] = new boolean[100030002];
        Arrays.fill(prime, true);
        prime[0] = false;
        prime[1] = false;
        for (int p = 2; p * p <= 100030001; p++) {
            if (prime[p] == true) {
                for (int i = p * 2; i <= 100030001; i += p) {
                    prime[i] = false;
                }
            }
        }
        for (int i = 0; i < prime.length; i++) {
            if (prime[i] && isPalindromeNumber(i)) {
                set.add(i);
            }
        }

    }

    
static public boolean isPalindromeNumber(int num) {
        int copy = num;
        int total = 0;
        while(copy != 0) {
            total = total * 10 + (copy % 10);
            copy /= 10;
        }
        return total == num;
    }





    public int primePalindrome(int N) {
        return set.ceiling(N);
    }
}
