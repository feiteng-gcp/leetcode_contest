class Solution {
    public int primePalindrome(int N) {
        int n = N;
        int len = 0;
        while (n > 0){
            len++;
            n /= 10;
        }
        for (int i = len; i <= 9; ++i){
            int[] arr = new int[i];
            long z = find(0, i, (long)N, arr);
            if (z != -1L) return (int)z;
        }
        return 0;
    }
    
    private long find(int idx, int len, long N, int[] arr){
        int mid = len / 2;
        if (idx > mid){
            long k = 0L;
            for (int num : arr) k = k * 10L + (long) num;
            if (k >= N && isPrime(k)) return k;
            return -1L;
        }
        
        int start = idx == 0 ? 1 : 0;
        for ( ; start < 10; ++start){
            arr[idx] = start;
            arr[len - idx-1] = start;
            long z = find(idx+1,len, N, arr);
            if (z != -1L) return z;
        }
        return -1L;
    }
    
    private boolean isPrime(long k){
        // System.out.println(k);
        if (k < 2L) return false;
        
for (long i = 2L; i * i <= k; i++){
            if (k % i == 0L) return false;
        }
        return true;
    }
}
