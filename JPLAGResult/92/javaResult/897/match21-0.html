class Solution {
    public int primePalindrome(int N) {
        int digit = 0;
        int V = N;
        while (N > 0) {
            digit++;
            N /= 10;
        }
        while (digit < 32) {
            List<Integer> curAllPalindrome = new ArrayList<>();
            getPalindrome(digit, 0, curAllPalindrome, 0);
            for (int pld : curAllPalindrome) {
                if (pld >= V && checkPrime(pld)) {
                    return pld;
                }
            }
            digit ++;
        }
        return -1;
    }
    
    private void getPalindrome(int n, int level, List<Integer> res, int cur) {
        if (n % 2 == 0 && level + 1 == n / 2){
            for (int i = 1; i <= 9; i++) {
                int resValue = cur + i * getTenPower(level);
                resValue += i * getTenPower(n - level -1);
                res.add(resValue);
            }
            return ;
        }
        
        if (n % 2 == 1 && level == n / 2) {
            for (int i = 0; i <= 9; i++) {
                int resValue = cur + i * getTenPower(level);
                res.add(resValue);
            }
            return ;
        }
        
        
        if (level == 0) {
            for(int i = 1; i <= 9; i++) {
                int curValue = cur + i * getTenPower(level);
                curValue += i * getTenPower(n - level - 1);
                getPalindrome(n, level + 1, res, curValue);
            }
        } else {
            for(int i = 0; i <= 9; i++) {
                int curValue = cur + i * getTenPower(level);
                curValue += i * getTenPower(n - level - 1);
                getPalindrome(n, level + 1, res, curValue);
            }
        } 
        
        
    }
    
    private int getTenPower(int n) {
        if (n == 0) {
            return 1;
        }
        if (n == 1) {
            return 10;
        }
        int half = n / 2;
        return getTenPower(half) * getTenPower(n - half);
    }
    
    private boolean checkPrime(int n) { 
        if (n < 2) {
            return false;
        }
        if (n == 2) {
            return true;
        }
        for (int i = 2; i * i <= n; i++) {
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }
}
