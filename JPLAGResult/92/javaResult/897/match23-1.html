class Solution {
    public int primePalindrome(int N) {
    	int ans = N;
    	if(N == 1) return 2;
    	if(N == 2) return 2;
        if(N >= 10000000) return 100030001;
    	if(N % 2 == 0) N += 1;
    	for(ans = N; ; ans+=2) {
    		//System.out.println(ans);
    		if(checkPalindrome(ans) && checkPrime(ans)) return ans;
    	}
    }
    
    private boolean checkPrime(int N) {
    	for
(int i = 2; i * i <= N; i++) {
    		if(N % i == 0) return false;
    	}
    	return true;
    }
    
    private boolean checkPalindrome(int N) {
    	String Q = N + "";
    	int
 s = 0;
    	int e = Q.length()-1;
    	
    	while(s <= e) {
    		if(Q.charAt(s) != Q.charAt(e)) return false;
    		s++;
    		e--;
    	}
    	return true;
    }
}
