/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private Map<TreeNode, Integer> map;
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        if(root == null)
            return root;
        //using a hashmap to store all leaf nodes and it's depth
        map = new HashMap<>();
        return dfs(root, 1);
    }
    public TreeNode dfs(
TreeNode node, int depth) {
		if (node.left == null && node.right == null) {
			map.put(node, depth);
			return node;
		}
		if (node.left == null)
			return dfs(node.right, depth + 1);
		if (node.right == null)
			return dfs(node.left, depth + 1); 

        TreeNode l = dfs(node.left, depth + 1);
        TreeNode r = dfs(node.right, depth + 1);
        //same depth
        if(map.get(l) == map.get(r)) {
            map.put(node, map.get(l));
            return node;
        }
        return map.get(l) > map.get(r) ? l : r;
    }
}
