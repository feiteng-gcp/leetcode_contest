/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    TreeNode res = null;
    int resDepth = 0;
    int numMaxDepth = 0;
    int maxDepth = -1;
    HashSet<TreeNode> set = new HashSet<>();
    
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        res = root;
        markDepth(root, 0);
        
findNode(root, 0);
        return res;
    }
    
    private void markDepth(TreeNode node, int depth) {
        if (node==null) {
            return;
        }
        if (depth > maxDepth) {
            maxDepth = depth;
            numMaxDepth = 1;
            set = new HashSet<TreeNode>();
            set.add(node);
        }
        else if (depth == maxDepth) {
            numMaxDepth++;
            set.add(node);
        }
        markDepth(node.left, depth+1);
        markDepth(node.right, depth+1);
    }
    
    private int findNode(TreeNode node, int depth) {
        if (node==null) {
            return 0;
        }
        
        int self = (set.contains(node)) ? 1 : 0;
        int left = findNode(node.left, depth+1);
        int right = findNode(node.right, depth+1);
        int sum = self+left+right;
        
if (sum == numMaxDepth) {
            if (depth > resDepth) {
                res = node;
                resDepth = depth;
            }
        }
        return sum;
    }
}
