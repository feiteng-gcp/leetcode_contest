/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        Map<TreeNode, TreeNode> parentByChild = new HashMap<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        Queue<TreeNode> lastLevel = new LinkedList<>();
        while(!q.isEmpty()) {
            int size = q.size();
            lastLevel = new LinkedList<>(q);
            for (int i = 0; i < size; i++) {
                TreeNode n = q.poll();
                if (n.left != null) {
                    q.add(n.left);
                    parentByChild.put(n.left, n);
                }
                if (n.right != null) {
                    q.add(n.right);
                    parentByChild.put(n.right, n);
                }
            }
        }
        TreeNode prevParent = null;
        while(lastLevel.size() > 1) {
            TreeNode child = lastLevel.poll();
            TreeNode parent = parentByChild.get(child);
            if (prevParent != parent) {
                lastLevel.add(parent);
                prevParent = parent;
            }
        }
        return lastLevel.peek();
    }
}
