/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        find(root, 0);
        return ret;
    }
    
int maxdeep = 0;
    TreeNode ret = null;
    public int find(TreeNode root, int deep) {
        if (root == null) {return deep;}
        int left = find(root.left , deep+1);
        int right = find(root.right, deep+1);
        if (left == right) {
            if (maxdeep <= left) {
                maxdeep = left;
                ret = root;
            }
        }
        return Math.max(right, left);
    }
}
