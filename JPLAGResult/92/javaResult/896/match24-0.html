/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        if (root == null) return null;
        return dfs(root).node;
    }
    
    private Value dfs(TreeNode root) {
        if (root == null) return null;
        Value lV = dfs(root.left), rV = dfs(root.right);
        if (lV == null && rV == null) return new Value(1, root);
        else if (lV == null || rV != null && rV.depth > lV.depth) {
            rV.depth = rV.depth + 1;
            return rV;
        } else if (rV == null || lV != null && lV.depth > rV.depth) {
            lV.depth = lV.depth + 1;
            return lV;
        } else { //lV.depth == rV.depth
            return new Value(lV.depth+1, root);
        }
    }
    
    private static class Value {
        int depth;
        TreeNode node;
        Value(int depth, TreeNode node) {
            this.depth = depth;
            this.node = node;
        }
    }
}
