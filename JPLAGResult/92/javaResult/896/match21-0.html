/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    class Inter{
        int depth;
        TreeNode node;
        public Inter(int depth,TreeNode node){
            this.depth = depth;
            this.node = node;
        }
    }
    
    private TreeNode result;
    
    private Inter searchDeep(TreeNode node){
        if (node==null) return new Inter(0,null);
        Inter leftInter  = searchDeep(node.left);
        Inter rightInter  = searchDeep(node.right);
        if (leftInter.depth>rightInter.depth){
            return new Inter(leftInter.depth+1,leftInter.node);
        } else if (leftInter.depth<rightInter.depth){
            return new Inter(rightInter.depth+1,rightInter.node);
        }  else {  
            return new Inter(leftInter.depth+1,node);
        }
    }
    
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        return searchDeep(root).node;
    }
}
