/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    class Result {
        TreeNode solution;
        int maxdepth;
        Result(TreeNode solution, int maxdepth) { this.solution = solution; this.maxdepth = maxdepth;}
    }
    
    public TreeNode subtreeWithAllDeepest(TreeNode root) {

        Result res = aux(root);
        return res.solution;

    }
    
    private Result aux(TreeNode node) {
        
        if (node == null) { return new Result(null, 0); }
        
        if (node.left == null && node.right == null) { return new Result(node, 1); }
        else if (node.left == null) {
            Result resRight = aux(node.right);
            resRight.maxdepth++;
            return resRight;
        }
        else if (node.right == null) {
            Result resLeft = aux(node.left);
            resLeft.maxdepth++;
            return resLeft;
        }
        
        // Have left and right.
        else {

            Result resLeft = aux(node.left);
            Result resRight = aux(node.right);
            
            if (resLeft.maxdepth == resRight.maxdepth) {
                return new Result(node, resLeft.maxdepth+1);
            }
            else if (resLeft.maxdepth > resRight.maxdepth) {
                resLeft.maxdepth++;
                return resLeft;
            }
            else { // resLeft.maxdepth < resRight.maxDepth
                resRight.maxdepth++;
                return resRight;
            }
            
        }
        
        
        
        
    }
}
