/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        HashMap<TreeNode, TreeNode> t = new HashMap<>();
        Deque<TreeNode> q = new LinkedList<>();
        q.push(root);
        ArrayList<TreeNode> l = new ArrayList<>();
        while(!q.isEmpty()){
            int s = q.size();
            l.clear();
            //System.out.println(s+"---");
            
for(int i = 0;i<s;i++){
                TreeNode n = q.poll();
            if(n.left!=null) {t.put(n.left,n);q.offer(n.left);}
            if(n.right!=null) {t.put(n.right,n);q.offer(n.right);}
            l.add(n);
                //System.out.println(n.val);
            }
        }
        if(l.size()==1) return l.get(0);
        TreeNode all = t.get(l.get(0));
        while(all!=null){
            int flag = 0;
            for(TreeNode i:l){
                //System.out.println(t.get(i).val);
                if(!t.get(i).equals(all)){
                    
                    flag=1;
                break;
                }
            }
            //System.out.println(all.val);
            if(flag==0) return all;
            for(TreeNode i:l){
                t.put(i,t.get(t.get(i)));
            }
                all=t.get(l.get(0));
        }
        return root;
        }
        
        
    
}
