/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    
    class Info{
        TreeNode n;
        int d;
        Info(int depth, TreeNode node){
            this.n = node;
            d = depth;
        }
    }
    
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        
return query(root).n;
    }
    
    Info query(TreeNode root){
        if(root == null){
            return new Info(0, null);
        
}
        Info left = query(root.left), right = query(root.right);
        if(left.d == right.d){
            return new Info(right.d + 1, root);
        } else if(left.d > right.d){
            left.d = left.d + 1;
            return left;
        } else{
            right.d = right.d + 1;
            return right;
        }
        
    }
}
