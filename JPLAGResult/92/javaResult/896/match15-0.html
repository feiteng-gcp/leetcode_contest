/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    class ResultType {
        int maxDepth;
        TreeNode subtree;
        
        ResultType(int maxDepth, TreeNode subtree) {
            this.maxDepth = maxDepth;
            this.subtree = subtree;
        }
    }
    
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        if (root == null) {
            return null;
        }
        
        ResultType rt = helper(root);
        
        return rt.subtree;
    }
    
    private ResultType helper(TreeNode root) {
        if (root == null) {
            return new ResultType(0, null);
        }
        
        ResultType left = helper(root.left);
        ResultType right = helper(root.right);
        
        if (left.maxDepth == right.maxDepth) {
            return new ResultType(left.maxDepth + 1, root);
        } else if (left.maxDepth < right.maxDepth) {
            return new ResultType(right.maxDepth + 1, right.subtree);
        } else {
            return new ResultType(left.maxDepth + 1, left.subtree);
        }   
    }
}
