/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
public TreeNode subtreeWithAllDeepest(TreeNode root) {
        Result result = getResult(root);
        return result.node;
    }

    private Result getResult(TreeNode root) {
        if (root == null) {
            return new Result(0, null);
        }

        
Result leftResult = getResult(root.left);
        Result rightResult = getResult(root.right);
        if (leftResult.depth < rightResult.depth) {
            return new Result(rightResult.depth + 1, rightResult.node);
        }

        if (leftResult.depth > rightResult.depth) {
            return new Result(leftResult.depth + 1, leftResult.node);
        }

        return new Result(rightResult.depth + 1, root);
    }

    
private static class Result {
        int depth;
        TreeNode node;

        public Result(int depth, TreeNode node) {
            this.depth = depth;
            this.node = node;
        }
    }
}
