/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    
private class Pair {
        TreeNode n;
        int d;
        public Pair(TreeNode n, int d) {
            this.n = n;
            this.d = d;
        }
    }
    
    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        if (root == null) {
            return null;
        
}
        return helper(root).n;
    }
    
    public Pair helper(TreeNode node) {
        if (node == null) {
            return null;
        }
        Pair l = helper(node.left);
        Pair r = helper(node.right);
        if (l == null && r == null) {
            return new Pair(node, 0);
        }
        else if (l == null) {
            r.d = r.d + 1;
            return r;
        }
        else if (r == null) {
            l.d = l.d + 1;
            return l;
        }
        else 
if (r.d > l.d) {
            r.d = r.d + 1;
            return r;
        }
        else if (r.d < l.d) {
            l.d = l.d + 1;
            return l;
        }
        else {
            return new Pair(node, l.d + 1);
        }
    }
}
