/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        if (root==nullptr) return nullptr;
        int depth=0;
        return s2(root, depth);
    }
    
    TreeNode* s2(TreeNode* root, int& depth) {
        // root cannot be nullptr
        if (root->left==nullptr && root->right==nullptr) {
            depth=0;
            return root;
        }
        int l=-1;
        TreeNode* ln=nullptr;
        if (root->left) {
            ln=s2(root->left, l);
        }
        int r=-1;
        TreeNode* rn=nullptr;
        if (root->right) {
            rn=s2(root->right, r);
        }
        if (l>r) {
            depth=l+1;
            return ln;
        } else if (l==r) {
            depth=l+1;
            return root;
        } else {
            // l<r
            depth=r+1;
            return rn;
        }
    }
};
