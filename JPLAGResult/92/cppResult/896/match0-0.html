/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* help(TreeNode * node, int& depth){
        if(!node){
            depth=0;
            return NULL;
        }
        int left_depth=0;
        TreeNode * left_node=help(node->left,left_depth);
        int right_depth=0;
        TreeNode * right_node=help(node->right,right_depth);
        if(left_depth==right_depth){
            depth=left_depth+1;
            return node;
        }
        if(left_depth<right_depth){
            depth=right_depth+1;
            return right_node;
        }else{
            depth=left_depth+1;
            return left_node;
        }
    }
    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        int depth=0;
        return help(root,depth);
    }
};
