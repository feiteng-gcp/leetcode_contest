/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        return f(root).first;
    }
    
    pair<TreeNode*, int> f(TreeNode* node) {
        if (node == NULL) {
            return make_pair((TreeNode*)NULL, 0);
        }
        auto left = f(node->left);
        auto right = f(node->right);
        if (left.second == right.second) {
            return make_pair(node, left.second + 1);

        } else if (left.second > right.second) {
            return make_pair(left.first, left.second + 1);;
        } else {
            return make_pair(right.first, right.second + 1);
        }
    }
};
