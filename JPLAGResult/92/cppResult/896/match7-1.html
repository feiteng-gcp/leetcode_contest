/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    TreeNode* iterFind(TreeNode* root, int& depth) {
        if (root == NULL) {
            depth = 0;
            return root;
        }
        int depthl, depthr;
        TreeNode* tl = iterFind(root->left, depthl);
        TreeNode* tr = iterFind(root->right, depthr);

        if (depthl == depthr) {
            depth = depthl + 1;
            return root;
        } else if (depthl > depthr) {
            depth = depthl + 1;
            return tl;
        } else {
            depth = depthr + 1;
            return tr;
        }
    }
public:
    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        int depth;
        return iterFind(root, depth);
    }
};
