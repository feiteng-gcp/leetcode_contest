/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    pair<TreeNode*, int> getMaxDepthNode(TreeNode* root, int depth) {
        if (root->left == NULL && root->right == NULL) {
            return {root, depth};
        } else if (root->left == NULL) {
            return getMaxDepthNode(root->right, depth + 1);
        } else if (root->right == NULL) {
            return getMaxDepthNode(root->left, depth + 1);
        } else {
            auto p1 = getMaxDepthNode(root->left, depth + 1);
            auto p2 = getMaxDepthNode(root->right, depth + 1);
            if (p1.second == p2.second) {
                return {root, p1.second};
            } else if (p1.second > p2.second) {
                return p1;
            } else {
                return p2;
            }
        }
    }
    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        if (root == NULL)
            return NULL;
        
        auto p = getMaxDepthNode(root, 0);
        return p.first;
    }
};
