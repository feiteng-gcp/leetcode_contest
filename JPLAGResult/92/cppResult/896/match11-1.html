/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    pair<TreeNode*, int> rec(TreeNode* root) {

        if (root->right && root->left) {
            auto p1 = rec(root->right);
            auto p2 = rec(root->left);
            if (p1.second == p2.second) {
                return {root, p2.second+1};
            } else if (p1.second > p2.second) {
                p1.second += 1;
                return p1;
            } else {
                p2.second += 1;
                return p2;
            }
        } else if (root->right) {
            auto p = rec(root->right);
            p.second += 1;
            return p;
        } else if (root->left) {
            auto p = rec(root->left);
            p.second += 1;
            return p;
        } else {
            return {root, 0};
        }
    }
    
    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        if (!root) {
            return nullptr;
        }
        return rec(root).first;
    }
};
