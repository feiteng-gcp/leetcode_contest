/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int max_dpth;
    map<TreeNode*,TreeNode*> parents;
    TreeNode* lca(TreeNode *l, TreeNode* r){
        auto pl = parents[l];
        auto pr = parents[r];
        if(pl == pr){
            return pl;
        }
        return lca(pl, pr);
    }
    void getMaxDepth(TreeNode* root, int lvl){
        if(!root){
          return;
        } 
        max_dpth = max(max_dpth, lvl);  
        if(root->left){
            getMaxDepth(root->left, lvl+1);
        }
        if(root->right){
            getMaxDepth(root->right, lvl+1);
        }        
    }
    
public:
    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        getMaxDepth(root, 0);
        if(!root) return nullptr;
        
        queue<TreeNode*> q;
        q.push(root);
        int num_curr = 1;
        int num_next = 0;
        int lvl = 0;
      //  cout << max_dpth << "\n";
        while(q.size()){
          // cout << lvl << " " << q.size() << "\n";
            if(lvl==max_dpth){
                TreeNode* fr = q.front();
                TreeNode* bk = q.back();
                if(fr == bk) return fr;
                return lca(fr,bk);
            }
            auto pr = q.front();

            q.pop();
            num_curr--;
            
            
            if(pr->left){
                parents[pr->left] = pr;
                q.push(pr->left);
                num_next++;
            }
            if(pr->right){
                parents[pr->right] = pr;
                q.push(pr->right);
                num_next++;
            }   
            if(num_curr == 0){
                num_curr= num_next;
                num_next = 0;
                lvl++;
            }
        }
        return nullptr;
    }
};
