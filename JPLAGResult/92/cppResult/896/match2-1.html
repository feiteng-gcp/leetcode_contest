/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* subtreeWithAllDeepest(TreeNode* root, int& dep) {
        if(!root){
            dep = 0;
            return NULL;
        }
        TreeNode* left, *right;
        int ld=0, rd=0;
        if(root->left)
             left = subtreeWithAllDeepest(root->left, ld);
        if(root->right)
             right = subtreeWithAllDeepest(root->right, rd);

        if(ld == rd){
            dep = ld +1;
            return root;
        }else if(ld > rd){
            dep = ld + 1;
            return left;
        }else{
            dep = rd + 1;
            return right;
        }
    }
    TreeNode* subtreeWithAllDeepest(TreeNode* root){
         int dep;
         return  subtreeWithAllDeepest(root, dep);
    }
};
