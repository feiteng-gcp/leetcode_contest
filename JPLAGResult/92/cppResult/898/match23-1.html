class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& A) {

        vector<vector<int>> At(A[0].size());

        for(int i = 0; i < At.size(); i++){
            At[i].resize(A.size());
        }
        
        for(int i = 0; i < A.size(); i++){
            for(int j = 0; j < A[0].size(); j++){
                At[j][i] = A[i][j];
            }
        }
        return At;
    }
};
