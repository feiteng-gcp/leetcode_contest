class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& A) {
        int N = A.size();
        int M = A[0].size();
        
        vector<vector<int>> result(M, vector<int>(N));
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                result[j][i] = A[i][j];
            }
        }
        
        return result;
    }
};
