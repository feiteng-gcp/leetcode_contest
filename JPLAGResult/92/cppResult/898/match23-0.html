class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& A) {
        vector <vector <int> > a = A, b;
        b.resize(a[0].size());
        for (int i = 0; i < a[0].size(); ++i)
        {
            b[i].resize(a.size());
        }
        for (int i = 0; i < a.size(); ++i)
        {
            for (int j = 0; j < a[0].size(); ++j)
                b[j][i] = a[i][j];
        }
        return b;
}};
