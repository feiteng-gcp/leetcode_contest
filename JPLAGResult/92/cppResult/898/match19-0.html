class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& A) {
        int m=A.size(), n = A[0].size();
        vector<vector<int>> a(n);
        for(int i=0;i<n;i++) {
            for(int j=0;j<m;j++) {
                a[i].push_back(A[j][i]);
            }
        }
        return a;
    }
};
