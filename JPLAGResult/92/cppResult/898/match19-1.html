class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& A) {
        vector<vector<int>> result;

        vector<int> temp;

        for(int i = 0;i < A[0].size();++i){
            for(int j = 0;j < A.size();++j) {
                temp.push_back(A[j][i]);
            }

            result.push_back(temp);
            temp.clear();
        }

        return result;
    }
};
