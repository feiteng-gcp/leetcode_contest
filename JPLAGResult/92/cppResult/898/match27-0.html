class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& a) {
        vector<vector<int>> out;
        
        if (a.size() != 0)
        {
            int rows = a.size();
            int cols = a[0].size();
            
            out.resize(cols);
            
            for (int c=0;c<cols;++c)
            {
                out[c].resize(rows);
                for (int r=0;r<rows;++r)
                {
                    out[c][r] = a[r][c];
                }
            }
        }
        return out;
    }
};
