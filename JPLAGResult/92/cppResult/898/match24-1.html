class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& A) {
        vector<vector<int>> t;
        for (int i = 0; i < A[0].size(); i++) {
            vector<int> new_row;
            for (int j = 0; j < A.size(); j++) {
                new_row.push_back(A[j][i]);
            }
            t.push_back(new_row);
        }
        return t;
    }
};
