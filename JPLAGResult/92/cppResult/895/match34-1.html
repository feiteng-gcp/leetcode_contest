int neigh[4][2]={{0,1},{1,0},{0,-1},{-1,0}};
int d[40][40][1<<6];

class Solution {
public:
    int shortestPathAllKeys(vector<string>& a) {
      int i,j,k,m,n,x,y,bit,xx,yy,bit1,all;
      vector<pair<pair<int,int>,int> > q;
      m=a.size();
      n=a[0].length();
      all=0;
      for (i=0;i<m;i++)
      {
        for (j=0;j<n;j++)
        {
          if (a[i][j]=='@')
          {
            x=i;y=j;
          }
          if ((a[i][j]>='a')&&(a[i][j]<='z'))
          {
            all|=(1<<(a[i][j]-'a'));
          }
        }
      }
      //cout<<all<<endl;
      memset(d,-1,sizeof(d));
      q.clear();
      q.push_back(make_pair(make_pair(x,y),0));
      d[x][y][0]=0;
      for (i=0;i<q.size();i++)
      {
        x=q[i].first.first;
        y=q[i].first.second;
        bit=q[i].second;
        if (bit==all) return d[x][y][bit];
        for (j=0;j<4;j++)
        {
          xx=x+neigh[j][0];
          yy=y+neigh[j][1];
          if ((xx>=0)&&(xx<m)&&(yy>=0)&&(yy<n)&&(a[xx][yy]!='#'))
          {
            if ((a[xx][yy]>='A')&&(a[xx][yy]<='Z'))
            {
              if ((bit&(1<<(a[xx][yy]-'A')))==0) continue;
            }
          //cout<<xx<<" "<<yy<<endl;
            bit1=bit;
            if ((a[xx][yy]>='a')&&(a[xx][yy]<='z'))
            {
              bit1=(bit1|(1<<(a[xx][yy]-'a')));
            }
            if (d[xx][yy][bit1]==-1)
            {
              //cout<<xx<<" "<<yy<<" "<<bit1<<endl;
              d[xx][yy][bit1]=d[x][y][bit]+1;
              q.push_back(make_pair(make_pair(xx,yy),bit1));
            }
          }
        }
      }
      return -1;
    }
};
