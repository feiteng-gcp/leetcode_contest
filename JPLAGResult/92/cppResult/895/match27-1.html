#include <bits/stdc++.h>

using namespace std;

struct State {
    int r, c;
    int keys, keyCount;
    int steps;

    State(int R, int C, int k, int kc, int s): r(R), c(C), keys(k), keyCount(kc), steps(s) {}

    int encode() {
        return keys * 900 + r * 30 + c;
    }
};

class Solution {
    private:
        int dr[4] = {1, -1, 0, 0};
        int dc[4] = {0, 0, 1, -1};
        queue<State> Q;
        bool visited[90000];

        void enqueue(int r, int c, int keys, int keyCount, int steps = 0) {
            State s(r, c, keys, keyCount, steps);
            int code = s.encode();
            if (!visited[code]) {
                visited[code] = true;
                Q.push(s);
            }
        }

    public:
        int shortestPathAllKeys(vector<string>& grid) {
            int startingR, startingC;
            int totalKeys = 0;


            for (int i = 0; i < grid.size(); ++i) {
                for (int j = 0; j < grid[i].size(); ++j) {
                    if (grid[i][j] == '@') {
                        startingR = i, startingC = j;
                    } else if (grid[i][j] <= 'z' && grid[i][j] >= 'a') {
                        totalKeys++;
                    }
                }
            }

            enqueue(startingR, startingC, 0, 0);

            while (!Q.empty()) {
                State current = Q.front();
                Q.pop();

                //cout << '(' << current.r << ',' << current.c << "), keys=" << current.keys << ", keyCount=" << current.keyCount << ", steps=" << current.steps << endl;
                ++current.steps;

                for (int k = 0; k < 4; ++k) {
                    int nr = current.r + dr[k], nc = current.c + dc[k];

                    if (nr < 0 || nr >= grid.size() || nc < 0 || nc >= grid[nr].size()) {
                        continue;
                    }

                    char ch = grid[nr][nc];
                    int newKeys = current.keys;
                    int newKeyCount = current.keyCount;

                    if (ch == '#') {
                        continue;
                    } else if (ch <= 'Z' && ch >= 'A') {
                        int lock = ch - 'A';
                        if (!(current.keys & (1 << lock))) {
                            continue;
                        }
                    } else if (ch <= 'z' && ch >= 'a') {
                        int key = ch - 'a';
                        if (!(current.keys & (1 << key))) {
                            newKeys |= (1 << key);
                            ++newKeyCount;
                            if (newKeyCount == totalKeys) {
                                return current.steps;
                            }
                        }
                    }

                    enqueue(nr, nc, newKeys, newKeyCount, current.steps);
                }
            }

            return -1;
        }
};
