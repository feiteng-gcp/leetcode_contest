int dir[4][2] = {{1, 0}, {-1, 0}, {0,1},{0, -1}};

class Solution {
public:
    int shortestPathAllKeys(vector<string>& grid) {
        set<tuple<int, int, int>> visited;
        deque<tuple<int, int, int, int>> q;
        int n = grid.size();
        int m = grid[0].size();
        int target = 0;
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == '@') {
                    q.emplace_back(i, j, 0, 0);
                }
                
                if (islower(grid[i][j])) {
                    target |=  1 << (grid[i][j] - 'a');
                }
            }
        }
        
        while (!q.empty()) {
            int xx, yy, ss, dep;
            tie(xx, yy, ss, dep) = q.front();
            q.pop_front();
            
            for (int i = 0; i < 4; ++i) {
                int x = xx + dir[i][0];
                int y = yy + dir[i][1];
                
                if (x < 0 || y < 0) continue;
                if (x >= n || y >= m) continue;
                if (grid[x][y] == '#') continue;
                
                bool can_walk = grid[x][y] == '.' || grid[x][y] == '@';
                int s = ss;
                
                if (isupper(grid[x][y])) {
                    int k = grid[x][y] - 'A';
                    if (s & (1 << k)) {
                        can_walk = true;
                    }
                }
                
                if (islower(grid[x][y])) {
                    can_walk = true;
                    s |= 1 << (grid[x][y] - 'a');
                    
                    if (s == target) {
                        return dep + 1;
                    }
                }
                
                if (can_walk && !visited.count(make_tuple(x, y, s))) {
                    visited.insert(make_tuple(x, y, s));
                    q.emplace_back(x, y, s, dep+1);
                }
                
            }
            
        }
        return -1;
    }
    
    
};
