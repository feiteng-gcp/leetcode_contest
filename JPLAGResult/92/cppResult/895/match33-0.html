int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};
class Solution {
public:
    
    struct Node {
        int x, y, s, t;
    }st;
    bool vis[32][32][1<<6];
    int n, m;
    bool isin(int x, int y) {
        if (x >= 0 && x < n && y >= 0 && y < m) return true;
        return false;
    }
    int shortestPathAllKeys(vector<string>& grid) {
        int k = 0;
        n = grid.size(); m = grid[0].size();
        for(int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[i].size(); j++) {
                if (grid[i][j] == '@') {
                    st.x = i; st.y = j; st.s = st.t = 0;
                }
                if (grid[i][j] >= 'a' && grid[i][j] <= 'z') {
                    k = max(k, grid[i][j] - 'a' + 1);
                }
            }
        }
        memset(vis, false,sizeof(vis));
        queue<Node> que; que.push(st); vis[st.x][st.y][st.s] = true;
        while(!que.empty()) {
            Node now = que.front(); que.pop();
            if (now.s == ((1<<k) - 1) ) return now.t;
            for (int i = 0; i < 4; i++) {
                Node nxt = now;
                nxt.x += dx[i]; nxt.y += dy[i];
                if (!isin(nxt.x, nxt.y)) continue;
                if (grid[nxt.x][nxt.y] == '#') continue;
                if (grid[nxt.x][nxt.y] >= 'A' && grid[nxt.x][nxt.y] <= 'Z') {
                    int ch = grid[nxt.x][nxt.y] - 'A';
                    if (!(nxt.s & (1 << ch))) continue;
                }
                if (grid[nxt.x][nxt.y] >= 'a' && grid[nxt.x][nxt.y] <= 'z') {
                    int ch = grid[nxt.x][nxt.y] - 'a';
                    nxt.s |= (1<<ch);
                }
                if (vis[nxt.x][nxt.y][nxt.s]) continue;
                vis[nxt.x][nxt.y][nxt.s] = true;
                nxt.t++; que.push(nxt);
            }
        }
        return -1;
    }
};
