class Solution {
public:
    int visited[31][31][1<<6];
    
    int shortestPathAllKeys(vector<string>& grid) {
        queue< pair<pair<int,int>,int> > Q;
        int R=grid.size();
        int C=grid[0].size();
        memset(visited,-1,sizeof(visited));

        int keys=0;
        cout<<R<<" "<<C<<endl;
        for(int i=0;i<R;i++)
            for(int j=0;j<C;j++){
                if(grid[i][j]=='@'){
                    Q.push(make_pair(make_pair(i,j),0));
                    visited[i][j][0]=0;
                }
                if(grid[i][j]>='a' && grid[i][j]<='z'){
                    keys++;
                }
            }

        int dev=-1;
        int di[4]={-1,1,0,0};
        int dj[4]={0,0,1,-1};
        
        while(!Q.empty()){
            pair<pair<int,int>,int>P =Q.front();
            Q.pop();
            pair<int,int>pos=P.first;
            int mask=P.second;
            
            if(mask+1==(1<<keys)){
                dev=visited[pos.first][pos.second][mask];
                break;
            }
            
            int x=pos.first;
            int y=pos.second;
            
            for(int i=0;i<4;i++){
                int X=x+di[i];
                int Y=y+dj[i];
                if(X>=0 && X<R & Y>=0 && Y<C){
                    char ch= grid[X][Y];
                    int mask2=mask;
                    if(ch>='a' && ch<='z'){
                        mask2=mask2|(1<<(ch-'a'));    
                    }
                    
                    if(ch>='A' && ch<='Z'){
                        if( (mask2&(1<<(ch-'A')))!=0 ){
                            if( grid[X][Y]!='#' && visited[X][Y][mask2]==-1 ){
                                visited[X][Y][mask2]=visited[x][y][mask]+1;
                                Q.push(make_pair(make_pair(X,Y),mask2));
                            }        
                        } 
                    }else{
                        
                        if( grid[X][Y]!='#' && visited[X][Y][mask2]==-1 ){
                            visited[X][Y][mask2]=visited[x][y][mask]+1;
                            
                                Q.push(make_pair(make_pair(X,Y),mask2));
                        }
                    
                        
                    }
                    
                    
                }
                
            }
            
            
        }
        
        return dev;
    }
};
