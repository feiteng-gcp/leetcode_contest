struct node {
    int x, y, key;
    node(int x, int y, int key) {
        this->x = x;
        this->y = y;
        this->key = key;
    }
};

int res[30][30][1<<7];
class Solution {
public:
    queue<node> q;
    int dir[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    int encode[20];
    
    bool check(int xx, int yy, int n, int m, int key, vector<string>& grid) {
        if (xx < 0 || yy < 0 || xx >= n || yy >= m) return false;
        if (grid[xx][yy] == '#') return false;
        if (grid[xx][yy] >= 'A' && grid[xx][yy] <= 'Z') {
            int t = grid[xx][yy] - 'A';
            t = encode[t];
            int lock = 1 << t;
            
            if ((key & lock) == 0) {
                return false;
            }
            else {
                return true;
            }
        }
    
        return true;
    }
    
    
    
    int shortestPathAllKeys(vector<string>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int cnt = 0;
        
        
        int INF = 1e9+7;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < (1 << 7); k++) {
                    res[i][j][k] = INF;
                }
            }
        }
        
        memset(encode, -1, sizeof encode);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == '@') {
                    res[i][j][0] = 0;
                    q.push(node(i, j, 0));
                }
                
                if (grid[i][j] >= 'a' && grid[i][j] <= 'z') {
                    if (encode[grid[i][j] - 'a'] == -1) {
                        encode[grid[i][j] - 'a'] = cnt++;
                    }
                }
                
            }
        }
        
        
        while(!q.empty()) {
            int x = q.front().x;
            int y = q.front().y;
            int key = q.front().key;

            q.pop();
            
            if (key == ((1 << cnt) - 1)) {
                return res[x][y][key];
            }
            
            
            for (int i = 0; i < 4; i++) {
                int xx = x + dir[i][0];
                int yy = y + dir[i][1];
                
                if (check(xx, yy, n, m, key, grid)) {
                    if (grid[xx][yy] >= 'a' && grid[xx][yy] <= 'z') {
                        int t = grid[xx][yy] - 'a';
                        t = encode[t];
                        int k = 1 << t;
                        
                        if (res[xx][yy][k | key] > res[x][y][key] + 1) {
                            res[xx][yy][k | key] = res[x][y][key] + 1;
                            q.push(node(xx, yy, k | key));
                        }
                    }
                    
                    #printf("%d %d %d\n", res[xx][yy][key], res[x][y][key], key);
                    if (res[xx][yy][key] > res[x][y][key] + 1) {
                        res[xx][yy][key] = res[x][y][key] + 1;
                        q.push(node(xx, yy, key));
                    }
                }
            }
            
        }
        
        return -1;
        
    }
};
