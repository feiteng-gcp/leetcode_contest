typedef long long llong;
typedef vector<int> VI;
typedef vector<VI> VVI;


struct Pos {
   int row, col;
   int key_mask;
};

int dr[] = {-1, 0, 0, 1};
int dc[] = {0, -1, 1, 0};

const int INF = 0x3f3f3f3f;
int D[32][32][1<<6];

class Solution {
public:
   int shortestPathAllKeys(vector<string>& grid) {
      const int R = grid.size();

      const int C = grid[0].size();
   // cerr << R << ' ' << C << endl;

      Pos start;
      int num_keys = 0;
      for (int r = 0; r < R; ++r) {
         for (int c = 0; c < C; ++c)
            if (grid[r][c] == '@') {
               start = { r, c, 0 };
               grid[r][c] = '.';
            }
            else if (islower(grid[r][c]))
               num_keys++;
      }
   // cerr << start.row << "," << start.col << endl;
   // cerr << num_keys << endl;

      memset(D, 0x3f, sizeof(D));
      D[start.row][start.col][start.key_mask] = 0;
      queue<Pos> q;
      q.push(start);
      while (!q.empty()) {
         Pos cur = q.front();
         q.pop();
         int dist = D[cur.row][cur.col][cur.key_mask];
      // fprintf(stderr, "%d,%d %d : %d\n", cur.row, cur.col, cur.key_mask, dist);
         if (cur.key_mask == (1<<num_keys)-1)
            return dist;
         for (int k = 0; k < 4; ++k) {
            Pos nxt = { cur.row + dr[k], cur.col + dc[k], cur.key_mask };
            if (nxt.row < 0 || nxt.row >= R || nxt.col < 0 || nxt.col >= C)
               continue;
            char ch = grid[nxt.row][nxt.col];
            if (ch == '#')
               continue;
            if (isupper(ch) && 0 == (cur.key_mask & (1 << (ch-'A'))))
               continue;
            if (islower(ch))
               nxt.key_mask |= 1<<(ch-'a');
            if (D[nxt.row][nxt.col][nxt.key_mask] > dist + 1) {
               D[nxt.row][nxt.col][nxt.key_mask] = dist + 1;
               q.push(nxt);
            }
         }
      }

      return -1;
   }
};
