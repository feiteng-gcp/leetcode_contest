const int dx[] = {-1, 0, 1, 0};
const int dy[] = {0, 1, 0, -1};

class Solution {
public:
    int have[32][32][64];
    int shortestPathAllKeys(vector<string>& grid) {
        int m = grid.size(), n = grid[0].size(), x, y, all = 0;

        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == '@') {
                    x = i;
                    y = j;
                }
                else if (isupper(grid[i][j])) {
                    ++all;
                }
            }
        }
        memset(have, 0xff, sizeof(have));
        int want = (1 << all) - 1;
        priority_queue<pair<int, int> > q;
        q.push(make_pair(0, (x << 20) | (y << 10)));
        while (!q.empty()) {
            int t = -q.top().first;
            x = q.top().second >> 20;
            y = (q.top().second >> 10) & 1023;
            int z = q.top().second & 1023;
            //cout << "x = " << x << "y = " << y << "z = " << z << endl;
            q.pop();
            if (have[x][y][z] >= 0) {
                continue;
            }
            have[x][y][z] = t;
            if (z == want) {
                return t;
            }
            for (int i = 0; i < 4; ++i) {
                int xx = x + dx[i];
                int yy = y + dy[i];
                int zz = z;
                if (xx < 0 || xx >= m || yy < 0 || yy >= n || grid[xx][yy] == '#') {
                    continue;
                }
                if (isupper(grid[xx][yy]) && (((z & (1 << (grid[xx][yy] - 'A')))) == 0)) {
                    continue;
                }
                if (islower(grid[xx][yy])) {
                    zz |= (1 << (grid[xx][yy] - 'a'));
                }
                q.push(make_pair(-t - 1, (xx << 20) | (yy << 10) | zz));
            }
            
        }
        return -1;
    
        
    }
};
