class Solution {
public:


	int dirs[4][2] = { { 0,1 },{ 0, -1 },{ 1,0 },{ -1, 0 } };

	int shortestPathAllKeys(vector<string>& grid) {
		if (grid.size() == 0 || grid[0].size() == 0)
		{
			return -1;
		}

		int m = grid.size();
		int n = grid[0].size();

		int need = 0;


		int sx = -1;
		int sy = -1;

		for (int i = 0; i<m; i++)
		{
			for (int j = 0; j<n; j++)
			{
				if (grid[i][j] == '@')
				{
					sx = i;
					sy = j;
				}
				else if (grid[i][j] >= 'A' && grid[i][j] <= 'F')
				{
					int idx = grid[i][j] - 'A';
					need |= (1 << idx);
				}
			}
		}

		if (sx == -1)
		{
			return -1;
		}

		if (need == 0)
		{
			return 0;
		}

		queue<pair<int, int>> q;
		unordered_map<int, unordered_set<int>> visit;

		q.push({ sx*n + sy, 0 });
		visit[sx*n + sy].insert(0);

		int dist = 0;

		while (q.size() > 0)
		{
			int size = q.size();
			while (size--)
			{
				auto f = q.front(); q.pop();

				int x = f.first / n;
				int y = f.first % n;
				int keys = f.second;

				if (keys == need)
				{
					return dist;
				}

				for (int d = 0; d<4; d++)
				{
					int nx = x + dirs[d][0];
					int ny = y + dirs[d][1];

					if (nx <0 || nx >= m || ny<0 || ny >= n)
					{
						continue;
					}

					if (grid[nx][ny] == '#')
					{
						continue;
					}

					//get keys
					if (grid[nx][ny] >= 'a' && grid[nx][ny] <= 'f')
					{
						int idx = grid[nx][ny] - 'a';
						int nk = keys | (1 << idx);

						int nxy = nx * n + ny;
						if (visit[nxy].find(nk) != visit[nxy].end())
						{
							continue;
						}

						if (nk == need)
						{
							return dist + 1;
						}

						q.push({ nxy, nk });
						visit[nxy].insert(nk);

						continue;
					}

					if (grid[nx][ny] == '.'|| grid[nx][ny] == '@')
					{
						int nxy = nx * n + ny;
						if (visit[nxy].find(keys) != visit[nxy].end())
						{
							continue;
						}
						q.push({ nxy, keys });
						visit[nxy].insert(keys);

						continue;
					}

					//A - F
					int idx = grid[nx][ny] - 'A';
					if (keys & (1 << idx))
					{
						int nxy = nx * n + ny;
						if (visit[nxy].find(keys) != visit[nxy].end())
						{
							continue;
						}
						q.push({ nxy, keys });
						visit[nxy].insert(keys);
					}
				}
			}

			dist++;
		}

		return -1;
	}
};
