struct Node{
    int x, y, key, t;
    Node(){
        x=0,y=0,key=0,t=0;
    }

};

class Solution {
public:
    
    int shortestPathAllKeys(vector<string>& grid) {
        int m = grid.size(), n=grid[0].size();
        int dx[] = {-1, 1, 0, 0};
        int dy[] = {0, 0, -1, 1};
        Node vn, vw;
        queue<Node> q;

        int c = 0;
        vector<vector<vector<bool> > > vis(m, vector<vector<bool> >(n, vector<bool>(1<<6, false)));
        for(int i=0; i<m; i++) for(int j=0; j<n; j++){
            if(grid[i][j] == '@'){
                vn.x = i, vn.y = j, vn.key = 0, vn.t = 0;
                q.push(vn);
                vis[i][j][0] = true;
            }else{
                if(grid[i][j]>='a' && grid[i][j]<='f'){
                    c ++;
                }
            }
        }
        int limit = (1<<c) - 1;

        while(!q.empty()){
            vw = q.front(); q.pop();
            if(vw.key == limit){
                return vw.t;
            }
            for(int i=0; i<4; i++){
                int x = vw.x + dx[i];
                int y = vw.y + dy[i];
                if(x>=0 && x<m && y>=0 && y<n){
                    if(grid[x][y] != '#'){
                        if(grid[x][y] >= 'a' && grid[x][y] <= 'f'){
                            int idx = grid[x][y] - 'a';
                            vn.x = x, vn.y = y, vn.key = (vw.key|(1<<idx));
                            if(!vis[x][y][vn.key]){
                                vis[x][y][vn.key] = true;
                                vn.t = vw.t + 1;
                                q.push(vn);
                            }
                        }else if(grid[x][y] >= 'A' && grid[x][y]<='F'){
                            int idx = grid[x][y] - 'A';
                            if((vw.key & (1<<idx)) != 0){
                                vn.x = x, vn.y = y, vn.key = vw.key;
                                if(!vis[x][y][vn.key]){
                                    vis[x][y][vn.key] = true;
                                    vn.t = vw.t + 1;
                                    q.push(vn);
                                }
                            }
                        }else{
                            vn.x = x, vn.y = y, vn.key = vw.key;
                            if(!vis[x][y][vn.key]){
                                vis[x][y][vn.key] = true;
                                vn.t = vw.t + 1;
                                q.push(vn);
                            }
                        }
                    }
                }
            }
        }
        return -1;
    }
};
