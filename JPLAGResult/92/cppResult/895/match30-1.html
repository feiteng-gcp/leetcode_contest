class Solution {
public:
    int dp[35][35][1<<6];
    int shortestPathAllKeys(vector<string>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        pair<int,int> start;
        int tot=0;
        for(int i=0;i<grid.size();i++)
            for(int j=0;j<grid[0].size();j++){
                if(grid[i][j]=='@'){
                    start=make_pair(i,j);
                }
                if(grid[i][j]>='A'&&grid[i][j]<='Z'){
                    tot++;
                }
            }
        memset(dp,-1,sizeof(dp));
        typedef pair<pair<int,int>,int> pii;
        #define mp make_pair
        #define x first
        #define y second
        queue<pii> q;
        dp[start.x][start.y][0]=0;
        q.push(mp(mp(start.x,start.y),0));
        int X[4]={0,0,1,-1};
        int Y[4]={1,-1,0,0};
        while(!q.empty()){

            pii p =q.front();
            if(p.y==((1<<tot)-1)){
                return dp[p.x.x][p.x.y][p.y];
            }
            q.pop();
            for(int i=0;i<4;i++){
                int xx=X[i]+p.x.x;
                int yy=Y[i]+p.x.y;
                int zz=p.y;
                int ok=1;
                if(xx<0||xx>=n||yy<0||yy>=m)
                    continue;
                if(grid[xx][yy]>='a'&&grid[xx][yy]<='z')
                    zz|=1<<(grid[xx][yy]-'a');
                else if(grid[xx][yy]>='A'&&grid[xx][yy]<='Z'){
                    ok=zz&(1<<(grid[xx][yy]-'A'));
                }
                if(dp[xx][yy][zz]==-1&&ok&&grid[xx][yy]!='#'){
                    dp[xx][yy][zz]=dp[p.x.x][p.x.y][p.y]+1;
                    q.push(mp(mp(xx,yy),zz));
                }
            }
        }
        return -1;
    }
};
