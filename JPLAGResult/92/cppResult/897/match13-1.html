typedef long long ll;
int ishuiwen(int i){	//判断回文
	int m,n;
	n = i;
	m = 0;
	while(n){
		m = m * 10 + n % 10;
		n /= 10;
	}
	if(m == i)
		return 1;
	return 0;
}

int isprime(ll m){
	ll i;

	for(i=3;i*i<=m;i+=2){	//+=2直接避开偶数
		if( m % i ==0 )
			return 0;
	}
	return 1;
}

int word_len(int x){		//求数字长度
	int res = 0;
	while(x){
		res++;
		x /= 10;
	}
	return res;
}

int ten_up(int m){			//变下一个10的倍数
	int i,s = 10;			//求10^m
	for(i=1;i<m;i++)	
		s*=10;
	return s;
}

class Solution {
public:
    int primePalindrome(int N) {
        for(ll i=N;;i++){
			if( i % 2 == 0 && i != 2 )	//除2外偶数不是素数
				continue;
			if( i % 5 == 0 && i != 5)	//除5外5倍数不是素数
				continue;
			if(!ishuiwen(i))			
				continue;	
			int len = word_len(i);
			if( len % 2 == 0 && i != 11){	//除11外数字长度为偶数的不是回文数
				i = ten_up(len);			//这招牛，从偶数长度跳到奇数长度
				continue;					//下一轮是因为i++到了101之类带1而不是0结尾的
			}
            if(i == 1)
                i = 2;
			if(isprime(i))		
                return i;
		}
        
        
    }
};
