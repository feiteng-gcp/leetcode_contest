class Solution {
public:
    bool isPrime(int N) {
        for (int i = 2; i <= sqrt(N); i++) {
            if (N % i == 0)
                return false;
        }
        return true;
    }
    
    int primePalindrome(int N) {
        if (N == 1)
            return 2;
        int nDigits = 0;
        int K = N;
        while (K != 0) {
            K /= 10;
            nDigits++;
        }
        string nStr = to_string(N);
        int firstHalf;
        string sFirstHalf;
        bool isEven;
        if (nDigits%2 == 0) {
            sFirstHalf = nStr.substr(0, nDigits/2);
            firstHalf = stoi(sFirstHalf);
            isEven = true;
        } else {
            sFirstHalf = nStr.substr(0, nDigits/2 + 1);
            firstHalf = stoi(sFirstHalf);
            isEven = false;
        }
        while (true) {
            string half1 = to_string(firstHalf);
            if (half1.length() == sFirstHalf.length()) {
                if (isEven) {
                    string half2 = half1;
                    reverse(half2.begin(), half2.end());
                    string sFull = half1 + half2;
                    int full = stoi(sFull);
                    if (full < N) {
                        sFirstHalf = to_string(firstHalf);
                        firstHalf++;
                        continue;
                    }
                    if (isPrime(full)) {
                        return full;
                    }
                } else {
                    string half2 = half1.substr(0, half1.length() - 1);;
                    reverse(half2.begin(), half2.end());
                    string sFull = half1 + half2;
                    int full = stoi(sFull);
                    if (full < N) {
                        sFirstHalf = to_string(firstHalf);
                        firstHalf++;
                        continue;
                    }
                    if (isPrime(full)) {
                        return full;
                    }
                }
            } else {
                if (isEven) {
                    isEven = false;
                    sFirstHalf = to_string(firstHalf);
                    continue;
                } else {
                    isEven = true;
                    firstHalf /= 10;
                    sFirstHalf = to_string(firstHalf);
                    continue;
                }
            }
            sFirstHalf = to_string(firstHalf);
            firstHalf++;
        }
        
    }
};
