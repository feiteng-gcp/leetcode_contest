class Solution {
public:
  bool isPalindrome(int n) {
    string v = to_string(n);
    const int len = v.size();
    for (int i = 0; i < len / 2; ++i)
      if (v[i] != v[len - i - 1])
        return false;
    return true;
  }

  bool isPrime(int n) {
    if (n == 2) return true;

    if (n == 1 || n % 2 == 0) return false;
    for (int i = 3; (i * i) <= n; i += 2)
      if (n % i == 0)
        return false;
    return true;
  }
  int primePalindrome(int N) {
    if (N <= 2) return 2;
    if (N % 2 == 0) ++N;
    if (N >= 9989900)
      N = 100030001;

    for (int i = N; i < INT_MAX; i += 2) {
      if (isPalindrome(i) && isPrime(i))
        return i;
    }
    return -1;
  }
};
