class Solution {
public:
    int primePalindrome(int N) {
        int x = N;
        cout << "ori:" << N << endl;
        if (! isPalin(x)) {
            x = nextPalin0(x);
            cout << x << endl;
        }
        while (!isPrime(x)) {
            x = nextPalin(x);
            cout << x << endl;
        }
        return x;
    }
    
    bool isPrime(int x) {
        if (x < 2) return false;
        if (x == 2) return true;
        for (int i = 2; i*i <= x; ++i) {
            if (x % i == 0) return false;
        }
        return true;
    }
    
    // assert x is not palin
    int nextPalin0(int x) {
        while (! isPalin(++x)) ;
        return x;
    }
    
    bool isPalin(int x) {
        int d = calcDigits(x);
        for (int i = 1; i <= d; ++i) {
            if (getDigit(x, i) != getDigit(x, d+1-i)) return false;
        }
        return true;
    }
    
    int calcDigits(int x) {
        int d = 0;
        
        if (x > 0) {
            int tmp = x;
            while (tmp > 0) {
                d++;
                tmp /= 10;
            }
        } else {
            d = 1;
        }
        
        return d;
    }
    
    // assert x is palin
    int nextPalin(int x) {
        int d = calcDigits(x);
        
        // calc next palin with same digits
        bool isFind = false;
        for (int i = (d+1)/2; i >= 1; --i) {
            int j = d+1-i;
            if (i == j) {
                if (getDigit(x, i) < 9) {
                    incDigit(x, i);
                    isFind = true;
                } else {
                    zeroDigit(x, i);
                }
            } else {
                assert(getDigit(x, i) == getDigit(x, j));
                if (getDigit(x, i) < 9) {
                    incDigit(x, i);
                    incDigit(x, j);
                    isFind = true;
                } else {
                    zeroDigit(x, i);
                    zeroDigit(x, j);
                }
            }
            
            if (isFind) {
                break;
            }
        }
        if (isFind) {
            return x;
        } else {
            return pow(10,d)+1;
        }
    }
    
    int getDigit(int x, int i) {
        assert(i >= 1);
        return x%(int)(pow(10,i))/(int)(pow(10,i-1));
    }
    
    void incDigit(int &x, int i) {
        x += pow(10, i-1);
    }
    
    void zeroDigit(int &x, int i) {
        x -= getDigit(x, i) * pow(10, i-1);
    }    
};
