class Solution {
public:
    
    int primePalindrome(int N) {
        if (N <= 2) return 2;
        for (int i = N; i < INT_MAX; ++i){
            if (i % 2 == 0 || i % 3 == 0 || i % 5 == 0 || i % 7 == 0 || i % 11 == 0){
                if (i == 2 || i == 3 || i == 5 || i == 7 || i == 11){
                    return i;
                }
                continue;
            }
            if ( checkPalin(i) && checkPrime(i) ){
                return i;
            }
        }
        return -1;
    }
    
    bool checkPrime(int n){
        int sq = sqrt(n);
        for (int i = 2; i <= sq; ++i){
            if (n % i == 0){
                return false;
            }
        }
        return true;
    }
    
    bool checkPalin(int k){
        int b = k;
        int t = 0;
        while (b > 0){
            t *= 10;
            t += (b % 10);
            b /= 10;
        }
        if (t == k){
            return true;
        }
        return false;
    }
    

    
};
