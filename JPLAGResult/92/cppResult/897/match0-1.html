class Solution {
public:
    bool check(int x){
        if(x==1){
            return false;
        }
        if(x==2){
            return true;
        }

        for(int i=2; (i-1)*(i-1)<x; i++){
            if(x%i==0 && x!=i){
                return false;
            }
        }
        return true;
    }
    bool checkp(int x){
        int curr=x;
        int ret=0;
        while(curr!=0){
            ret*=10;
            ret+=curr%10;
            curr/=10;
        }
        return ret==x;
    }
    int primePalindrome(int N) {
        if(N<=2){
            return 2;
        }
        if(N==3){
            return 3;
        }
        int n=(N+1)/6;
        while(true){
            int i=n*6-1;
            if(checkp(i) && check(i) && i>=N){
                return i;
            }
            i=n*6+1;
            if(checkp(i) && check(i) && i>=N){
                return i;
            }
            n++;
        }
        /*
        int i = (N%2==1 ? N : N+1);
        while(!(checkp(i) && check(i))){
            i+=2;
        }
        return i;
        */
    }
};
