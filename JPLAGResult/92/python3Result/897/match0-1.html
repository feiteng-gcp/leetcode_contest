import math
class Solution:
    def primePalindrome(self, N):
        """
        :type N: int
        :rtype: int
        """
        if N > 9989899:
            return 100030001
        for i in range(N, 9989899):
            if self.is_palindromic(i) and self.is_prime(i):
                return i
        
    def is_palindromic(self, a):
        temp = a
        b = 0
        while temp != 0:
            b *= 10
            b += temp % 10
            temp //= 10 
        return a == b
    
    def is_prime(self, a):
        
if a < 2:
            return False
        for i in range(2, int(math.sqrt(a)) + 1):
            if a % i == 0:
                return False
        return True
