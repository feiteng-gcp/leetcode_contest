class Solution:
    def check_prime(self, n):
        if n<= 1 : 
            return False
        if n <= 3:
            return True
        if n%2 ==0:
            return False
        if n%3 == 0:
            return False
        num = 5 
        while num*num <= n :
            if n% num == 0 or n % (num+2) == 0:
                return False
            num+=6
        return True
    def get_next_number(self, num):
        num_str = str(num)
        length = len(num_str)
        left_part = num_str[:length//2]
        middle_number = num_str[(length-1)//2]
        is_odd=(length%2!=0)
        if is_odd:
            inc=pow(10, length//2)
            result=int(left_part+middle_number+left_part[::-1])
        else:
            inc=int(1.1*pow(10, length//2))
            result=int(left_part+left_part[::-1])
        if result > num:
            return result
        if middle_number!='9':
            return result + inc
        else:
            inc=pow(10,((length//2)+1))
            return self.get_next_number(((num//inc)+1)*inc)
 
    def primePalindrome(self, N):
        """
        :type N: int
        :rtype: int
        """
        next_number = self.get_next_number(N-1)
        while not self.check_prime(next_number):
            next_number = self.get_next_number(next_number)
        return next_number
