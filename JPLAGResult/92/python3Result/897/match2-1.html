class Solution:
    def primePalindrome(self, N):
        """
        :type N: int
        :rtype: int
        """
        
        def check_is_palindrone(N):
            str_N = str(N)
            length = len(str_N)
            for i in range(length // 2):
                if str_N[i] != str_N[length-1-i]:
                    return False
            return True
        
        def next_palindrone(N, is_palindrone=False):
            str_N = str(N)
            length = len(str_N)
            change_index = (length - 1) // 2
            list_N = list(str_N)
            if not is_palindrone:
                for i in range(change_index+1, length):
                    list_N[i] = str_N[length-1-i]
                tN = int("".join(list_N))
                if tN > N:
                    return tN

            if str_N[change_index] != '9' and length % 2 == 1:
                list_N[change_index] = str(int(str_N[change_index]) + 1)
                return int("".join(list_N))
            else:
                new_N = (N // 10**(length-1-change_index) + 1) * 10**(length-1-change_index)
                return next_palindrone(new_N, is_palindrone=check_is_palindrone(new_N))
                
                
        def is_prime(N):
            
            if N == 1:
                return False
            if N == 2:
                return True
            
            if N % 2 == 0:
                return False
            
            end = int((N+1)**0.5) + 2
            
            for i in range(3, min(end+2, N), 2):
                if N % i == 0:
                    
return False
            return True
            
        if check_is_palindrone(N) and is_prime(N):
            return N
        
        next_pa = next_palindrone(N, check_is_palindrone(N))
        while True:
            if is_prime(next_pa):
                return next_pa
            next_pa = next_palindrone(next_pa, True)
