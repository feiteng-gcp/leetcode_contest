class Solution:
    def primePalindrome(self, N):
        """
        :type N: int
        :rtype: int
        """
        if N == 1:
            return 2
        p 
= self.nextPalindrome(N-1)
        # print(p)
        while not self.isPrime(p):
            p = self.nextPalindrome(p)
            # print(p)
        return p
    
    def isPrime(self, p):
        for i in range(2, int(math.sqrt(p)+1)):
            if p % i == 0:
                return False
        else:
            return True
        
    def nextPalindrome(self, N):
        if N < 9:
            return N+1
        else:
            digits = format(N)
            num = len(digits)
            if num % 2 == 0:
                firstHalf = digits[:num//2]
                newDigits = firstHalf + firstHalf[::-1]
                if int(newDigits) > N:
                    return int(newDigits)
                else:
                    # print(firstHalf)
                    firstHalf = format(int(firstHalf)+1)
                    # print(firstHalf)
                    if len(firstHalf) == num//2:
                        return int(firstHalf + firstHalf[::-1])
                    else:
                        return int(''.join(['1', '0'*(num-1), '1']))
            else:
                firstHalf = digits[:(num//2+1)]
                newDigits = firstHalf + firstHalf[-2::-1]
                if int(newDigits) > N:
                    return int(newDigits)
                else:
                    firstHalf = format(int(firstHalf)+1)
                    if len(firstHalf) == num//2+1:
                        return int(firstHalf + firstHalf[-2::-1])
                    else:
                        return int(''.join(['1', '0'*(num-1), '1']))
