class Solution:
    def primePalindrome(self, N):
        """
        :type N: int
        :rtype: int
        """
        def isPrime(n):
            if n < 2:
                return False
            if n == 2 or n == 3:
                return True
            for i in range(2, int(math.ceil(math.sqrt(n))) + 1):
                if n % i == 0:
                    return False
            return True
        
        def getNextPalindrome(n):
            def isEdge(L):
                for num in L:
                    if num != '9':
                        return False
                return True
            
            def assembleEven(L):
                r = L[::-1]
                L1 = L + r
                return int(''.join(L1))
            
            def assembleOdd(L):
                r = L[:len(L) - 1]
                r = r[::-1]
                L1 = L + r
                return int(''.join(L1))
            
            if n < 9:
                return n + 1
            elif n == 9 or n == 10:
                return 11
            
            l = list(str(n))
            firstL = l[:(len(l) + 1) // 2]
            
            print(firstL)
            
            if isEdge(firstL):
                firstL = list(str( int(''.join(firstL)) + 1 ))
                
                return assembleOdd(firstL)
                
            else:
                firstL = list(str( int(''.join(firstL)) + 1 ))
                if len(l) % 2 == 0:
                    return assembleEven(firstL)
                else:
                    return assembleOdd(firstL)
            
            
        
        
        def check(n):
            return isPrime(n)
        
        def isPalindrome(n):
            L = list(str(n))
            for i in range(len(L) // 2):
                if L[i] != L[len(L) - 1 - i]:
                    return False
            return True
        
        if check(N) and isPalindrome(N):
            return N
        
        N = getNextPalindrome(N)
        
        while not check(N):
            N = getNextPalindrome(N)
            
        return N
