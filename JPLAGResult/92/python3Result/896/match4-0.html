class Solution:
    def subtreeWithAllDeepest(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        
        p = [root]

        while p:
            l = []
            last_p = p[:]
            while p:
                node = p.pop()
                if node:
                    if node.left:
                        node.left.parent = node
                        l.append(node.left)
                    if node.right:
                        node.right.parent = node
                        l.append(node.right)
            p = l


        while last_p:
            if len(last_p) == 1:
                return last_p[0]
            p = []
            while last_p:
                node = last_p.pop()
                p.append(node.parent)
            last_p = p
            if len(set(p)) == 1:
                return list(set(p))[0]
