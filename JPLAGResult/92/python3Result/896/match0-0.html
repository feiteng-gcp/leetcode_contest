# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def subtreeWithAllDeepest(self, root):
        def util(node):
            if not node:
                return (0, None)
            resLeft = util(node.left)
            resRight = util(node.right)
            if resLeft[0] == resRight[0]:
                return (resLeft[0] + 1, node)
            elif resLeft[0] > resRight[0]:
                return (resLeft[0] + 1, resLeft[1])
            else:
                return (resRight[0] + 1, resRight[1])
        return util(root)[1]
        
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
