# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def subtreeWithAllDeepest(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        root.parent = None
        self.addParent(root)
        self.maxD = 1
        self.dnodes = []
        self.getDNode(root, 1)
        return self.findAnc()
        
    def getDNode(self, node, d):
        if d > self.maxD:
            self.maxD = d
            self.dnodes = [node]
        elif d == self.maxD:
            self.dnodes.append(node)
            
        if node.left:
            self.getDNode(node.left, d+1)
        if node.right:
            self.getDNode(node.right, d+1)
            
    def findAnc(self):
        nodes = self.dnodes[:]
        # print(len(nodes))
        while len(nodes)>1:
            levelnodes = set()
            for n in nodes:
                levelnodes.add(n.parent)
            nodes = list(levelnodes)
        return nodes[0]
        
    def addParent(self, root):
        
if root.left:
            root.left.parent = root
            self.addParent(root.left)
        if root.right:
            root.right.parent = root
            self.addParent(root.right)
