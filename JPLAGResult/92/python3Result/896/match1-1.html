# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    
def getNode(self, root):
        if root is None:
            return 0, None
            
        left_depth, left_node = self.getNode(root.left)
        right_depth, right_node = self.getNode(root.right)
        
        if left_depth == right_depth:
            return left_depth+1, root
        elif left_depth > right_depth:
            return left_depth+1, left_node
        else:
            return right_depth+1, right_node
    
    def subtreeWithAllDeepest(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        depth, node = self.getNode(root)
        return node
