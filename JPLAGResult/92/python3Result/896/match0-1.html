# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def subtreeWithAllDeepest(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        def s(cur):
            if not cur:
                return [0]
            l,r=s(cur.left),s
(cur.right)
            if l[0]==r[0]:
                return [l[0]+1,cur]
            elif l[0]>r[0]:
                return [l[0]+1,l[1]]
            else:
                return [r[0]+1,r[1]]
            
        return s(root)[1]
