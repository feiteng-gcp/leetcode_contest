# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def subtreeWithAllDeepest(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        if not root: return root
        q = [root]
        lastL = []
        while q:
            level = []
            for x in q:
                if x.left:
                    x.left.par = x
                    level.append(x.left)
                if x.right:
                    x.right.par = x
                    level.append(x.right)
                lastL = q
            q = level
        s = set(lastL)
        while len(s) != 1:
            x = set()
            for i in s:
                x.add(i.par)
            s = x
        return next(iter(s))
