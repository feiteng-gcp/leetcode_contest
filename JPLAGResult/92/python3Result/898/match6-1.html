class Solution:
    def transpose(self, A):
        """
        :type A: List[List[int]]
        :rtype: List[List[int]]
        """
        ans = []
        R = len(A)
        if R==0:
            return ans
        C=len(A[0])
        if C==0:
            return ans
        
        
for c in range(C):
            cur = []
            for r in range(R):
                cur.append(A[r][c])
                
            ans.append(cur)
            
        return ans
