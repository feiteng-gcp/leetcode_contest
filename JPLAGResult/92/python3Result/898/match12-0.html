class Solution:
    def transpose(self, A):
        """
        :type A: List[List[int]]
        :rtype: List[List[int]]
        """
        row = len(A)
        col = len(A[0])

        res = [[0 for j in range(row)] for i in range(col)]

        for i in range(row):
            for j in range(col):
                res[j][i] = A[i][j]

        return res
