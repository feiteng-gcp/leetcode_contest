class Solution:
    def transpose(self, A):
        """
        :type A: List[List[int]]
        :rtype: List[List[int]]
        """
        m, n = len(A), len(A[0])
        ret = []
        for i in range(n):
            tmp = []
            for j in range(m):
                tmp.append(A[j][i])
            ret.append(tmp)
        return ret
