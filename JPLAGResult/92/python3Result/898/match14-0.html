class Solution:
    def transpose(self, A):
        """
        :type A: List[List[int]]
        :rtype: List[List[int]]
        """
        if len(A) == 0 or len(A[0]) == 0:
            return []
        row, coln = len(A), len(A[0])
        result = []
        for i in range(coln):
            array = []
            for j in range(row):
                array.append(A[j][i])
            result.append(array)
        return result
