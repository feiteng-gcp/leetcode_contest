class Solution:
    def transpose(self, A):
        """
        :type A: List[List[int]]
        :rtype: List[List[int]]
        """
        l = len(A[0])
        re = [[0 for j in range(len(A))] for i in range(l)]
        for i in range(len
(A)):
            for j in range(l):
                re[j][i] = A[i][j]
        return re
