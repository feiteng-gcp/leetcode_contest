class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        if len(A)<=1:
            return 0
        str_len = len(A[0])
        
        minDel = 0
        for k in range(str_len):
            for word, next_word in zip(A[:-1], A[1:]):
                if ord(word[k]) > ord(next_word[k]):
                    minDel += 1
                    break
        
        return minDel
