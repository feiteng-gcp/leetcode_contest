class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        if len(A)==0:
            return 0
        l=len(A[0])
        ans=0
        for i in range(l):
            for j in range(len(A)-1):
                if A[j][i]>A[j+1][i]:
                    ans+=1
                    break
        return ans
