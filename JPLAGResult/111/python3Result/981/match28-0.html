class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        if len(A)<=1 or not A[0] :
            return 0
        ans = 0
        for j in range(len(A[0])) :
            for i in range(1,len(A)) :
                if A[i][j]<A[i-1][j] :
                    ans+=1
                    break
        return ans
