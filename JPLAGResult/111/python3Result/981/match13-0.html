class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        cnt = 0
        for j in range(len(A[0])):
            for i in range(len(A)-1):
                if A[i+1][j] < A[i][j]:
                    cnt += 1
                    break
        return cnt
