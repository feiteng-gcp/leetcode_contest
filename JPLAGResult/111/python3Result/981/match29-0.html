class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        res = 0
        n, m = len(A), len(A[0])
        for i in range(m):
            flag = True
            for j in range(1,n):
                if A[j][i]<A[j-1][i]:
                    flag = False
                    break
            if not flag:
                res += 1
        return res
