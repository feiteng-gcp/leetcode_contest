class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        M = len(A)
        N = len(A[0])
        D = []
        
for i in range(N):
            flag = True
            for j in range(M-1):
                if A[j][i] > A[j+1][i]:
                    flag = False
                    break
            if not flag:
                D.append(i)
        return len(D)
