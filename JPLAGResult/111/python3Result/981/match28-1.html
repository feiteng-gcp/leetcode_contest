class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """

        n = len(A)
        if n <= 1: return 0
        
        ans = 0
        for i in range(len(A[0])):
            for j in range
(1, n):
                if A[j][i] < A[j-1][i]:
                    ans += 1
                    break
        return ans
