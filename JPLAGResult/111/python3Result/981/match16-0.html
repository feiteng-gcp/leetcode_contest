class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        r, c = len(A), len(A[0])
        ans = 0
        for j in range(c):
            for i in range(1,r):
                if A[i][j] < A[i-1][j]:
                    ans += 1
                    break
        return ans
