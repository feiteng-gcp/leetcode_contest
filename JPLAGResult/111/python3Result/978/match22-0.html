class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) < 3:
            return False
        
        inc = True
        if A[0] >= A[1]:
            return False
        for i in range(1, len(A)):
            if A[i-1] == A[i]:
                return False
            if inc:
                if A[i-1] > A[i]:
                    inc = False
            else:
                if A[i-1] < A[i]:
                    return False
        
        if inc:
            return False
        return True
