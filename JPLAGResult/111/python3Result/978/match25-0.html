class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A)>=3:
            i=1 
            while i<len(A)-1 and A[i]>A[i-1]:
                i+=1
            j=len(A)-2
            while j>0 and A[j]>A[j+1]:
                j-=1
            if i-1==j+1:
                return True
        return False
