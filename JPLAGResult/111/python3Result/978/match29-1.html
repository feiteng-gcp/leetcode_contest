class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) < 3 or A[0] >= A[1]:
            return False
        dec = False
        for i in range(len(A) - 1):
            if not dec:
                if A[i] == A[i+1]:
                    return False
                elif A
[i] > A[i+1]:
                    dec = True
            else:
                if A[i] <= A[i+1]:
                    return False
        return dec
