import sys


class Solution:
    def validMountainArray(self, A):
        n = len(A)
        maxv = -sys.maxsize
        idx = -1
        for i in range(n):
            if A[i] > maxv:
                maxv = A[i]
                idx = i
                
        if idx == 0 or idx == n-1:
            return False
        else:
            for i in range(idx):
                if A[i] >= A[i+1]:
                    return False
            for i in range(idx, n-1):
                if A[i] <= A[i+1]:
                    return False
            return True
