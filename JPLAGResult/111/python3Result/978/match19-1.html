class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A)<3: return False
        if A[1]<=A[0]: return False
        idx = -1
        for i in range(len(A)-1):
            if A[i+1]<=A[i]:
                idx = i
                break
        if idx==-1: return False
        for i in range(idx, len(A)-1):
            if A[i+1]>=A[i]:
                return False
        return True
