class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) <= 2: return False
        status = True
        if A[0] >= A[1]: return False
        for i in range(1,len(A)-2):
            if A[i] < A[i+1]:
                if status == False: return False
            elif A[i] > A[i+1]:
                if status == True: status = False
            else:
                return False
        if A[-1] >= A[-2]: return False
        return True
