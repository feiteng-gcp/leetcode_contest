class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        maxindex = 0
        
        for i in range (len(A)-1):
            if A[i] < A[i+1]:
                maxindex = i+1
        if maxindex<1 or maxindex>=len(A)-1:
            return False
        for i in range (maxindex-1):
            if A[i] >= A[i+1]:
                return False
        for i in range (maxindex, len(A)-1):
            if A[i] <= A[i+1]:
                return False
        return True
