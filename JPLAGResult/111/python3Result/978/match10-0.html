class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) == 0:
            return False
        maxi = 0
        for i in range(1, len(A)):
            if A[i] > A[maxi]:
                maxi = i
        if maxi == 0 or maxi == len(A)-1:
            return False
        for i in range(1, maxi+1):
            if not A[i] > A[i-1]:
                return False
        for i in range(maxi+1, len(A)):
            if not A[i] < A[i-1]:
                return False
        return True
