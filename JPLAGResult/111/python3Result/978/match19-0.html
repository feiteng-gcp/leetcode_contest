class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) < 3:
            return False
        if A[0] > A[1]:
            return False
        increase = True
        for i in range(1, len(A)):
            if A[i] == A[i-1]:
                return False
            if increase:
                if A[i] > A[i-1]:
                    continue
                else:
                    increase = False
            else:
                if A[i] > A[i-1]:
                    return False
                else:
                    continue
        if increase:
            return False
        return True
