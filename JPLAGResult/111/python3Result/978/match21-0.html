class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) < 3:
            return False
        
        increase = True
        if A[0] >= A[1] or A[-2] <= A[-1]:
            return False
        for i in range(1, len(A)-2):
            if A[i] == A[i+1]:
                return False
            if increase:
                if A[i] > A[i+1]:
                    increase = False
            else:
                if A[i] < A[i+1]:
                    return False
        return True
