class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) < 3:
            return False
        i = A.index(max(A))
        
if i == 0 or i == len(A) - 1:
            return False
        for j in range(i - 1, -1, -1):
            if A[j] >= A[j + 1]:
                return False
        for j in range(i + 1, len(A)):
            if A[j] >= A[j - 1]:
                return False
        return True
