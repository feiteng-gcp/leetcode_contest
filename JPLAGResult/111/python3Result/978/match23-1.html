class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        n = len(A)
        i = 0
        if n<3:
            return False
        while i<n-1 and A[i]<A[i+1]:
            i
+=1
        if i==0 or i==n-1:
            return False
        while i<n-1:
            if A[i]<=A[i+1]:
                return False
            i+=1
        return True
