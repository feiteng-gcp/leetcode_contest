class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) < 3:
            return False
        i = 0
        while i < len(A)-1 and A[i] < A[i+1]:
            i += 1
        j = len(A)-1
        while j > 0 and A[j-1] > A[j]:
            j -= 1
        return 0 < i == j < len(A)-1
