class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        n = len(A)
        if n < 3: return False
        k = A.index(max(A))
        if not 0 < k < n - 1: return False
        
for i in range(k):
            if A[i] >= A[i+1]: return False
        for i in range(k, n - 1):
            if A[i] <= A[i+1]: return False
        return True
