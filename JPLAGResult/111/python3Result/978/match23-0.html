class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        N = len(A)
        i = 0
        while i<N-1:
            if A[i]>=A[i+1]:
                break
            i += 1
        if i==0 or i==N-1:
            return False
        while i<N-1:
            if A[i]<=A[i+1]:
                return False
            i+=1
        return True
