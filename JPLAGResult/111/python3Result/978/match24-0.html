class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        n = len(A)
        if n<3: return False
        if A[0]>=A[1]: return False
        i = 1
        while i<n and A[i-1]<A[i]:
            i += 1
        if i == n: return False
        while i<n and A[i-1]>A[i]:
            i += 1
        return i == n
