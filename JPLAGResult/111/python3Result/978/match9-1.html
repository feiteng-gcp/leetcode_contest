class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if not A:
            return False
        r = A.index(max(A))
        
if r == 0 or r + 1 == len(A):
            return False
        for i in range(r):
            if A[i] >= A[i + 1]:
                return False
        for i in range(r + 1, len(A)):
            if A[i] >= A[i - 1]:
                return False
        return True
