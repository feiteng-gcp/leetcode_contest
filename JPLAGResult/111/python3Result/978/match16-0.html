class Solution:
    def validMountainArray(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A)<3:
            return False
        
        k = 0
        while k+1 < len(A) and A[k+1] > A[k]:
            k = k+1
        if k==0 or k==len(A)-1:
            return False
        while k+1 < len(A) and A[k+1] < A[k]:
            k = k+1
        if k == len(A)-1:
            return True
        else:
            return False
