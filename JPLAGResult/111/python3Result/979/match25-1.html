class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        numD = 0
        for ch in S:
            if ch=='D':
                numD+=1
        out 
= [numD]
        D_start, I_start = numD-1, numD+1
        
        for ch in S:
            if ch is 'D':
                out.append(D_start)
                D_start -= 1
            elif ch is 'I':
                out.append(I_start)
                I_start += 1
                
        return out
