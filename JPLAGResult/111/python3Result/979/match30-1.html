class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        res = []
        low = S.count('D')
        res.append(low)
        low, high 
= low-1, low+1
        for c in S:
            if c == 'I':
                res.append(high)
                high += 1
            else:
                res.append(low)
                low -= 1
        return res
