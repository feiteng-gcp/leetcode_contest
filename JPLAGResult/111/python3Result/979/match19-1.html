class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        r = [0]
        x1 = 1
        x2 = -1
        for c in S:
            if c == "I":
                r.append(x1)
                x1 += 1
            else:
                r.append(x2)
                x2 -= 1
        m = min(r)
        return [x - m for x in r]
