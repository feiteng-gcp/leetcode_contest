class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        left,right = 0,len(S)
        ans = []
        for c in S :
            if c=='I' :
                ans.append(left)
                left += 1
            else :
                ans.append(right)
                right -= 1
        ans.append(left)
        return ans
