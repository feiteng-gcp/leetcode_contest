class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        
        if len(S) == 0:
            return []
        
        large = len(S)
        small 
= 0
        result = []
        
        for char in S:
            if char == 'I':
                result.append(small)
                small += 1
            else:
                result.append(large)
                large -= 1
        
        return result + [small]
