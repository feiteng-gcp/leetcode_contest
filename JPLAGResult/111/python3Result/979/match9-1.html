class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        mi = 0
        ma = len(S)
        res = []
        for x in S:
            if x == 'I':
                res.append(mi)
                mi += 1
            else:
                res.append(ma)
                ma -= 1
        res.append(mi)
        return res
