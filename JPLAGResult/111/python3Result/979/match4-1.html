class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        s = 0
        e = len(S)
        
        ret 
= []
        for c in S:
            if c == "I":
                ret.append(s)
                s+=1
            else:
                ret.append(e)
                e-=1
        ret.append(s)
        return ret
