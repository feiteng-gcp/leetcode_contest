class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """

        n = len(S)
        min_v = 0
        max_v = n
        res = []
        for s in S:
            if s == 'I':
                res.append(min_v)
                min_v += 1
            else:
                res.append(max_v)
                max_v -= 1
        res.append(min_v)
        return res
