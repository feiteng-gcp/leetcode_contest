class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        N = len(S)
        #used = {i for i in range(N+1)}
        min_used = 0
        max_used = N

        ans = []
        
        for s in S:
            if s == 'I':
                ans.append(min_used)
                min_used += 1
            else:
                ans.append(max_used)
                max_used -= 1
        ans.append(max_used)
        return ans
