class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        m, M = -1, 1
        ans = [0]
        for ch in S:
            if ch == 'I':
                ans.append(M)
                M += 1
            else:
                ans.append(m)
                m -= 1
        return [x-m-1 for x in ans]
