class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        mi = 0
        ma = len(S)
        
        ans = []
        for char in S:
            if char == 'I':
                ans.append(mi)
                mi += 1
            else:
                ans.append(ma)
                ma -= 1
        return ans + [mi]
