class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        mi = ma = 0
        
        res = [0]
        for x in S:
            if x == 'I':
                res.append(ma + 1)
                ma += 1
            else:
                res.append(mi - 1)
                mi -= 1

        return [x - mi for x in res]
