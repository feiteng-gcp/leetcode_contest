class Solution:
    def diStringMatch(self, S):
        """
        :type S: str
        :rtype: List[int]
        """
        rtn = []
        s = 0
        f = len(S)
        for i in S:
            if i == "I":
                rtn.append(s)
                s = s + 1
            else:
                rtn.append(f)
                f = f - 1
        rtn.append(s)
        return rtn
