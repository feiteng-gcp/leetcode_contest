class Solution:
    def shortestSuperstring(self, A):
        """
        :type A: List[str]
        :rtype: str
        """
        n = len(A)
        lsum = [[0] * n for _ in range(n)]
        for i in range(n):
            for j in range(n):
                if i != j:
                    minlen 
= min(len(A[i]), len(A[j]))
                    best = 0
                    for k in range(minlen - 1, 0, -1):
                        if A[i].endswith(A[j][:k]):
                            best = k
                            break
                    lsum[i][j] = len(A[j]) - best
        
        memo = [0] * ((1 << n) * (n + 1))
        memo2 = [0] * ((1 << n) * (n + 1))
        
        def solve(last, left):
            if left == 0:
                return 0
            key = (last << n) + left
            if memo[key] == 0:
                best = 1000000
                for i in range(n):
                    if left & (1 << i):
                        left ^= 1 << i
                        cur = solve(i, left) + lsum[last][i]
                        if cur < best:
                            best = cur
                            memo2[key] = i
                        left ^= 1 << i
                memo[key] = best
            return memo[key]
        
        best = 1000000
        res = ""
        for i in range(n):
            left = ((1 << n) - 1) ^ (1 << i)
            cur = solve(i, left) + len(A[i])
            if cur < best:
                best = cur
                res = A[i]
                w = i
                for j in range(n - 1):
                    z = memo2[(w << n) + left]
                    res += A[z][len(A[z]) - lsum[w][z]:]
                    left ^= 1 << z
                    w = z
        return res
