class Solution {
    
    String [] state;
    boolean [] vis;
    
    public String calc(String s1, String s2) {
        for(int s = Math.min(s1.length(), s2.length()) ; s>0; s--) {
            String p1 = s2.substring(0, s);
            if(s1.endsWith(p1)) {
                return s2.substring(s);
            }
        
}
        return s2;
    }
    
    public String shortestSuperstring(String[] a) {
        String [][] diff = new String[a.length][a.length];
        for(int s = 0 ; s < a.length ; s++) {
           for(int j = 0 ; j < a.length ; j++) {
               diff[s][j] = calc(a[s], a[j]);
           } 
        }
        state = new String[(1 << a.length) * a.length];
        vis = new boolean[(1 << a.length) * a.length];
        LinkedList<Integer> next = new LinkedList<>();
       
        
        for(int r = 0 ; r < a.length ; r++) {
            state[(1 << r) * a.length + r] = a[r];
            next.add((1 << r) * a.length + r);
        }
        while(next.size() > 0) {
            int m = next.removeFirst();
            if(vis[m]) continue;
            vis[m] = true;
            int cur = m % a.length;
            int mask = m / a.length;
            for(int i = 0 ; i < a.length; i++) {
                if((mask & (1 << i)) == 0) {
                    int ni = (mask | (1 << i)) * a.length + i; 
                
                    if(state[ni] == null || state[ni].length() > state[m].length() + diff[cur][i].length()) {
                        state[ni] = state[m] + diff[cur][i];  
                        next.add(ni);
                    }
                }
            }
            
        }
        
        int m = (1 << a.length);
        m--;
        m *=  a.length;
        String best = null;
        for(int t = 0 ; t < a.length ; t++) {
            
if(best == null || best.length() > state[m + t].length()) {
                best = state[m + t];
            }
        }
        return best;
    }
}
