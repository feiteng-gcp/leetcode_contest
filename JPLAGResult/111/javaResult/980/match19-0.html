class Solution {
    public String shortestSuperstring(String[] A) {
        int n = A.length;
        int[][] ds = new int[n][n];
        for(int i=0; i<n; ++i) {
            for(int j=0; j<n; ++j) {
                if(i == j) { ds[i][j] = Integer.MAX_VALUE; continue; }
                ds[i][j] = distance(A[i], A[j]);
                /*System.out.printf("ds[%s][%s] = %s: %s + %s = %s\n", 
                                  i, j, ds[i][j], 
                                  A[i], A[j], A[i] + A[j].substring(A[j].length() - ds[i][j], A[j].length()));*/
            }
        }  
        return findEulerianPath(A, ds);
    }
    private String findEulerianPath(String[] A, int[][] ds) {
        String best = "";
        for(int i=0; i < A.length; ++i) {
            String word = findEulerianPath(A, ds, (1 << i), i);
            if(best.isEmpty() || word.length() + A[i].length() < best.length())
                best = A[i] + word;
        }
        return best;
    }
    
    Map<Integer, String> cache = new HashMap<>();
    private String findEulerianPath(String[] A, int[][] ds, int vertex, int current) {
        if(cache.containsKey(vertex*16+current))
            return cache.get(vertex*16+current);
            
        String best = "";
        for(int i=0; i<ds.length; ++i) {
            if((vertex & (1 << i)) != 0) continue;
            int next = vertex | (1 << i);
            
            String word = findEulerianPath(A, ds, next, i);
            if(best.isEmpty() || best.length() > ds[current][i] + word.length()) {
               //System.out.println(A[i].substring(A[i].length() - ds[current][i], A[i].length()) + " for " + A[i]);
                best = A[i].substring(A[i].length() - ds[current][i], A[i].length()) + word;
            }
        }
        cache.put(vertex*16+current, best);
        return best;
    }
    
    private int distance(String first, String second) {
        for(int i=0; i<first.length(); ++i) {
            if(isShift(first, second, i))
                return second.length() - (first.length() - i);
        }
        return second.length();
    }
    
    private boolean isShift(String first, String second, int k) {
        for(int j=0; j<second.length() && k + j < first.length(); ++j) {
            if(first.charAt(k+j) != second.charAt(j))
                return false;
        }  
        return true;
    }
}
