import javafx.util.*;

class Solution {
    public String shortestSuperstring(String[] A) {
        if (A.length == 0) {
            return "";
        }
        
        Set<String> startSet = new HashSet<>();
        for (String s : A) {
            startSet.add(s);
        }
        
        Queue<Set<String>> queue = new LinkedList<>();
        queue.add(startSet);
        
        String res = null;
        while (!queue.isEmpty()) {
            Set<String> set = queue.remove();
            
            while (set.size() > 1) {
                List<Pair<String, String>> overlapped = new ArrayList<>();
                int maxOverlap = -1;

                for (String a : set) {
                    for (String b : set) {
                        if (a == b) {
                            continue;
                        }
                        int currOverlap = overlap(a, b);
                        if (currOverlap > maxOverlap) {
                            maxOverlap = currOverlap;
                            overlapped.clear();
                            overlapped.add(new Pair<>(a, b));
                        } else if (currOverlap == maxOverlap) {
                            overlapped.add(new Pair<>(a, b));
                        }
                    }
                }
                
                for (int i = 1; i < overlapped.size(); i++) {
                    Pair<String, String> pair = overlapped.get(i);
                    Set<String> newset = new HashSet<>(set);
                    newset.remove(pair.getKey());
                    newset.remove(pair.getValue());
                    newset.add(combine(pair.getKey(), pair.getValue()));
                    queue.add(newset);
                }

                Pair<String, String> fpair = overlapped.get(0);
                set.remove(fpair.getKey());
                set.remove(fpair.getValue());
                set.add(combine(fpair.getKey(), fpair.getValue()));
            }
            
            String curr = set.toArray(new String[1])[0];
            if (res == null || curr.length() < res.length()) {
                res = curr;
            }
        }
        return res;
    }
    
    private int overlap(String a, String b) {
        for (int i = Math.min(a.length(), b.length()); i > 0; i--) {
            int j = 0;
            for (; j < i; j++) {
                if (a.charAt(a.length() - i + j) != b.charAt(j)) {
                    break;
                }
            }
            if (j == i) {
                return i;
            }
        }
        return 0;
    }
    
    private String combine(String a, String b) {
        for (int i = Math.min(a.length(), b.length()); i > 0; i--) {
            int j = 0;
            for (; j < i; j++) {
                if (a.charAt(a.length() - i + j) != b.charAt(j)) {
                    break;
                }
            }
            if (j == i) {
                return a + b.substring(i);
            }
        }
        return a + b;
    }
}
