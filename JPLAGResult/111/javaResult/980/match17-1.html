class Solution {
    public String shortestSuperstring(String[] A) {
        int n = A.length;
        int[][] shared = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i==j)
                    continue;
                for (int k = 1; k < Math.min(A[i].length(), A[j].length()); ++k) {
                    if (A[i].substring(A[i].length()-k).equals(A[j].substring(0,k)))
                        shared[i][j] = k;
                }
            }
        
}
        int[] dp = new int[n*(1<<n)];
        int[] last = new int[n*(1<<n)];
        boolean[] vis = new boolean[n*(1<<n)];
        for (int i = 0; i < n*(1<<n); i++) {
            dp[i] = 1000000;
            last[i] = -1;
        }
        ArrayDeque<Integer> q = new ArrayDeque<Integer>();
        for (int i = 0; i < n; i++) {
            q.add((1<<i)*n+i);
            vis[(1<<i)*n+i] = true;
            last[(1<<i)*n+i] = -1;
            dp[(1<<i)*n+i] = A[i].length();
        
}
        while (!q.isEmpty()) {
            int u = q.removeFirst();
            int lu = u%n;
            int vals = u/n;
            int d = dp[u];
            for (int i = 0; i < n; i++) {
                if ((vals&(1<<i))==0) {
                    int nd = d+A[i].length()-shared[lu][i];
                    int v = (vals|(1<<i))*n+i;
                    if (nd < dp[v]) {
                        dp[v] = nd;
                        last[v] = u;
                        if (!vis[v]) {
                            q.add(v);
                            vis[v] = true;
                        }
                    }
                }
            
}
        }
        int ans = 1000000;
        int val = 0;
        for (int i = 0; i < n; i++) {
            if (dp[n*((1<<n)-1)+i] < ans) {
                ans = dp[n*((1<<n)-1)+i];
                val = n*((1<<n)-1)+i;
            }
        }
        int[] order = new int[n];
        for (int i = n-1; i >= 0; i--) {
            order[i] = val%n;
            val = last[val];
        }
        String ret = A[order[0]];
        for (int i = 1; i < n; i++) {
            ret += A[order[i]].substring(shared[order[i-1]][order[i]]);
        }
        return ret;
    }
}
