class Solution {
    private String helper(String[] A, boolean[] used, int[][] indexes, int prevIndex, int numUsed, String sb) {
        if (A.length == numUsed) {
            return sb;
        }
        
        int sbLen = sb.length();
        int min = Integer.MAX_VALUE;
        String minS = null;
        
        if (numUsed == 0) {
            for (int i = 0; i < A.length; i++) {
                if (used[i]) continue;
                used[i] = true;
                //sb.append(numUsed == 0 ? A[i] : A[i].substring(indexes[prevIndex][i]));
                String temp = helper(A, used, indexes, i, numUsed+1, A[i]);
                //System.out.println(temp);
                if (temp.length() < min) {
                    min = temp.length();
                    minS = temp;
                }
                //sb.delete(sbLen, sb.length());
                used[i] = false;
            }
        } else {
            int max = -1;
            List<Integer> maxIndexes = new LinkedList<>();
            int i = 0;
            for (; i < A.length; i++) {
                if (!used[i]) {
                    if (indexes[prevIndex][i] > max) {
                        max = indexes[prevIndex][i];
                        maxIndexes = new LinkedList<>();
                        maxIndexes.add(i);
                    } else if (indexes[prevIndex][i] == max) {
                        maxIndexes.add(i);
                    }
                }
            }
            
            for (int maxIndex : maxIndexes) {
                used[maxIndex] = true;
                String temp = helper(A, used, indexes, maxIndex, numUsed+1, sb + A[maxIndex].substring(indexes[prevIndex][maxIndex]));
                if (temp.length() < min) {
                    min = temp.length();
                    minS = temp;
                }
                used[maxIndex] = false;
            }
        }
        return minS;
    }
    private int findIndex(String s1, String s2) {
        int len1 = s1.length();
        int len2 = s2.length();
        for (int i = Math.max(1, len1 - (len2 - 1)); i < len1; i++) {
            int j = 0;
            while (i + j < len1 && s1.charAt(i+j) == s2.charAt(j)) {
                j++;
            }
            if (i + j == len1) {
                return j;
            }
        }
        return 0;
    }
    public String shortestSuperstring(String[] A) {
        int aLen = A.length;
        int[][] indexes = new int[aLen][aLen];
        for (int i = 0; i < aLen; i++) {
            for (int j = 0; j < aLen; j++) {
                if (i == j) continue;
                indexes[i][j] = findIndex(A[i], A[j]);
            }
            //System.out.println(Arrays.toString(indexes[i]));
        }
        String result = helper(A, new boolean[aLen], indexes, 0, 0, "");
        //System.out.println(result.length());
        return result;
    }
}
