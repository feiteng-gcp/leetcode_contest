class Solution {
    public String shortestSuperstring(String[] A) {
        if (A.length == 0) return null;
        if (A.length == 1) return A[0];
        String first = A[0];
        String second = A[1];
        int max = 0;
        int index = -1;
        int m = A.length;
        for (int i = 0; i < A.length - 1; i++) {
            for (int j = i+1; j < A.length; j++) {
                int[] overlap = getOverlapPoint(A[i], A[j]);
                if (overlap[0] > max) {
                    first = A[i];
                    second = A[j];
                    max = overlap[0];
                    index = overlap[1];
                }
            }
        }
        String combine = combine(first, second, index);
        String[] B = new String[m-1];
        int j = 0;
        for (int i = 0; i < m; i++) {
            if (!A[i].equals(first) && !A[i].equals(second)) {
                B[j++] = A[i];
            }
        }
        B[m-2] = combine;
        return shortestSuperstring(B);
    }
    
    public String combine(String a, String b, int index) {
        String first = a;
        String second = b;
        if (a.length() > b.length()) {
            first = b;
            second = a;
        }
        int m = first.length();
        int n = second.length();
        if (index < m) {
            return first.substring(0, m - 1 - index) + second;
        }
        if (index >= m && index < n) {
            return second;
        }
        return second.substring(0, index-m+1) + first;
        
    }
    public int[] getOverlapPoint(String a, String b) {
        String first = a;
        String second = b;
        if (a.length() > b.length()) {
            first = b;
            second = a;
        }
        int max = 0;
        int index = -1;
        int m = first.length();
        int n = second.length();
        boolean flag = true;
        for (int i = 0; i < m + n; i++) {
            if (i < m) {
                if (first.substring(m-1-i, m).equals(second.substring(0, i + 1))) {
                    if (i + 1 > max) {
                        max = i + 1;
                        index = i;
                    }
                }
            }
            if (i >= m && i < n) {
                
                if (first.equals(second.substring(i-m, i))) {
                    if (m > max) {
                        max = m;
                        index = i;
                    }
                }
            }
            if (i >= n) {
                if (first.substring(0, m-i+n-1).equals(second.substring(i-m+1, n))) {
                    if (m-i+n-1 > max) {
                        max = m-i+n-1;
                        index = i;
                    }
                }
            }
        }
        return new int[]{max, index};
    }
}
