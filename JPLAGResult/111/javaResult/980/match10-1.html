class Solution {
    private class Pair {
        String s;
        int i;
        int j;
        Pair(String s, int i, int j) {
            this.s = s;
            this.i = i;
            this.j = j;
        }
    }
    
    public String shortestSuperstring(String[] A) {
        if (A.length == 0) return "";
        if (A.length == 1) return A[0];
        
        int max = Integer.MIN_VALUE;
        Set<Pair> set = new HashSet<>();
        for (int i=0; i<A.length; i++) {
            for (int j=i+1; j<A.length; j++) {
                String s1 = concat(A[i], A[j]);
                String s2 = concat(A[j], A[i]);
                String s = s1.length() < s2.length() ? s1 : s2;
                int overlap = A[i].length() + A[j].length() - s.length();
                if (overlap > max) {
                    set.clear();
                    set.add(new Pair(s, i, j));
                    max = overlap;
                } else if (overlap == max) {
                    set.add(new Pair(s, i, j));
                
}
            }
        }
        
        String res = null;
        for (Pair pair : set) {
            String[] arr = new String[A.length-1];
            arr[0] = pair.s;
            int arri = 1;
            for (int i=0; i<A.length; i++) {
                if (i != pair.i && i != pair.j) {
                    arr[arri++] = A[i];
                }
            }
            String s = shortestSuperstring(arr);
            
if (res == null || s.length() < res.length()) {
                res = s;
            }
        }
        
        return res;
    }
    
    private String concat(String s1, String s2) {
        String res = s1+s2;
        for (int i=0; i<s1.length(); i++) {
            String sub = s1.substring(i, s1.length());
            if (
s2.startsWith(sub)) {
                String s = s1.substring(0, i) + s2;
                if (s.length() < res.length()) {
                    res = s;
                }
            }
        }
        return res;
    }
}
