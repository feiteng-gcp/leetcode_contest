class Solution {
    String[] dp;
    StringBuilder sb;
    
int n;
    String[] A;
    public String shortestSuperstring(String[] A) {
        n = A.length;
        this.A = A;
        String[][] comb = new String[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j) {
                    comb[i][j] = A[i];
                }
                else {
                    comb[i][j] = add(A[i], A[j]);
                }
            }
        }

        dp = new String[1 << n];
        sb = new StringBuilder();
        for (int i = 0; i < 250; i++) {
            sb.append("}");
        }

        Arrays.fill(dp, sb.toString());
        String res = recusive((1 << n) - 1);
        return res;
    }

    String recusive(int mask) {
        if (mask == 0) {
            return "";
        }
        else if (dp[mask].length() != 250) {
            return dp[mask];
        }

        String res = sb.toString();
        
for (int i = 0; i < n; i++) {
            if (((1 << i) & mask) != 0) {
                String cur = A[i];
                String other = recusive(mask - (1 << i));
                String add1 = add(cur, other);
                if (res.length() > add1.length()) {
                    res = add1;
                }

                add1 = add(other, cur);
                if (res.length() > add1.length()) {
                    res = add1;
                }
            }
        }

        dp[mask] = res;
        return res;
    }

    String add(String s1, String s2) {
        int n1 = s1.length();
        int n2 = s2.length();
        int idx = -1;
        for (int i = n1; i >= 0; i--) {
            String suff = s1.substring(i);
            int len = n1 - 1 - i + 1;
            if (len <= s2.length()) {
                String pref = s2.substring(0, len);
                if (suff.equals(pref)) {
                    idx = i;
                }
            }
        }

        if (idx == -1) {
            return s1 + s2;
        }

        int len = n1 - idx;
        return s1 + s2.substring(len);
    }
}
