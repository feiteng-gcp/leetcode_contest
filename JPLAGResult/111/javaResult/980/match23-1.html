class Solution {
    public int dd(String a, String b) {
            int maxlen = 0;
            for (int i = 1; i <= Math.min(a.length(), b.length()); i++) {
                boolean correct = true;
                for (int j = 0; j < i; j++) {
                    if (a.charAt(a.length() - i + j) != b.charAt(j)) correct = false;
                }
                if (correct) maxlen = i;
            }
            return maxlen;
        }
        public static int[] tsp(int[][] dist) {
            int n = dist.length;
            int[][] dp = new int[1 << n][n];
            for (int[] d : dp)
                Arrays.fill(d, Integer.MAX_VALUE / 2);
            for (int i = 0; i < n; i++)
                dp[1 << i][i] = 0;
            for (int mask = 0; mask < 1 << n; mask++) {
                for (int i = 0; i < n; i++) {
                    if ((mask & 1 << i) != 0) {
                        for (int j = 0; j < n; j++) {
                            if ((mask & 1 << j) != 0) {
                                dp[mask][i] = Math.min(dp[mask][i], dp[mask ^ (1 << i)][j] + dist[j][i]);
                            }
                        }
                    }
                }
            }
            int res = Integer.MAX_VALUE;
            for (int i = 0; i < n; i++) {
                res = Math.min(res, dp[(1 << n) - 1][i]);
            }
            int cur = (1 << n) - 1;
            int[] order = new int[n];
            int last = -1;
            for (int i = n - 1; i >= 0; i--) {
                int b = -1;
                for (int j = 0; j < n; j++) {
                    if ((cur & 1 << j) != 0
                            && (b == -1
                            || dp[cur][b] + (last == -1 ? 0 : dist[b][last]) > dp[cur][j] + (last == -1 ? 0 : dist[j][last]))) {
                        b = j;
                    }
                }
                order[i] = b;
                cur ^= 1 << b;
                last = b;
            }
            return order;
        }

        public String shortestSuperstring(String[] A) {
            int n = A.length;
            
int[][] dist = new int[n][n];
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if (i == j) {
                        dist[i][j] = Integer.MAX_VALUE / 2;
                    } else {
                        dist[i][j] = -dd(A[i], A[j]);
                    }
                }
            }
            int[] order = tsp(dist);
            StringBuilder sb = new StringBuilder();
            int skip = 0;
            for (int i = 0; i < n; i++) {
                sb.append(A[order[i]].substring(skip));
                if (i < n - 1)
                skip = -dist[order[i]][order[i + 1]];
            }
            return sb.toString();
        }
}
