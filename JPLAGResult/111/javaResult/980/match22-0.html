public class Solution {
    private int n;
    private String[] a;
    public String shortestSuperstring(String[] A) {
        n = A.length;
        a = A;
        String[] dp = new String[1 << n];
        return search(dp, (1 << n) - 1);
    }

    private String search(String[] dp, int bits) {
        if (dp[bits] != null) return dp[bits];
        int candLen = Integer.MAX_VALUE;
        String cand = "";
        for (int i = 0; i < n; ++i) {
            if ((bits & (1 << i)) != 0) {
                int pre = bits ^ (1 << i);
                String preStr = search(dp, pre);
                String str = a[i];
                int n1 = preStr.length(), n2 = str.length();
                if (n1 + n2 < candLen) {
                    candLen = n1 + n2;
                    cand = preStr + str;
                }
                int len = Math.min(n1, n2);
                l: for (int offset = len; offset >= 1; --offset) {
                    if (n1 + n2 - offset >= candLen) break;
                    if (str.substring(0, offset).equals(preStr.substring(n1 - offset)))
                    {
                        candLen = n1 + n2 - offset;
                        cand = preStr + str.substring(offset);
                        break l;
                    }
                    if (preStr.substring(0, offset).equals(str.substring(n2 - offset)))
                    {
                        candLen = n1 + n2 - offset;
                        cand = str + preStr.substring(offset);
                        break l;
                    }
                }
            }
        }
        return dp[bits] = cand;
    }
}
