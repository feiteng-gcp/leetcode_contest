class Solution {
    
    class Node{
        int max;
        String tmp;
        public Node(int max, String tmp){
            this.max = max;
            this.tmp = tmp;
        }
    }
    
    private boolean compareStr(String s1, int s1s, int s1e, String s2, int s2s, int s2e){
        for(int i = 0 ; i < s1e - s1s ; i++){
            if(s1.charAt(s1s + i) != s2.charAt(s2s + i)) return false;
        }
        return true;
    }
    
    private Node findOverlappingPair(String s1, String s2){
        int max = Integer.MIN_VALUE;
        int len1 = s1.length();
        int len2 = s2.length();
        String tmp = null;
        
        for(int i = 1 ; i <= Math.min(len1, len2); i++){
            if(compareStr(s1, len1 - i, len1, s2, 0, i)){
                if(max < i){
                    
max = i;
                    tmp = s1 + s2.substring(i);
                }
            }
        }
        
        for(int i = 1 ; i <= Math.min(len1, len2) ; i++){
            if(compareStr(s1, 0, i, s2, len2 - i, len2)){
                if(max < i){
                    
max = i;
                    tmp = s2 + s1.substring(i);
                }
            }
        }
        return new Node(max, tmp);
    }
    
    public String shortestSuperstring(String[] A) {
        Arrays.sort(A, (a, b) -> b.compareTo(a));
        int n = A.length;
        while(n != 1){
            int max = Integer.MIN_VALUE;
            int l = 0, r = 0;
            String resString = null;
            for(int i = 0 ; i < n ; i++){
                for(int j = i + 1 ; j < n ; j++){
                    Node tmpNode = findOverlappingPair(A[i], A[j]);
                    int res = tmpNode.max;
                    String str = tmpNode.tmp;
                    if(max < res){
                        max = res;
                        resString = new String(str);
                        l = i;
                        r = j;
                    }
                }
            }
            
            n--;
            if(max == Integer.MIN_VALUE)
                A[0] += A[n];
            else{
                A[l] = resString;
                A[r] = A[n];
            }
        }
        return A[0];
    }
}
