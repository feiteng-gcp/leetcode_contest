class Solution {
    
    
    // Represents a subset of indices, with the last index specified
    public class DPStructure {
        int mask;
        int last;
        
        public DPStructure(List<Integer> indices) {
            mask = 0;
            for(int ind : indices) {
                last = (1 << ind);
                mask += last;
            }
        }
        
        private DPStructure(int mask, int last) {
            this.mask = mask; this.last = last;
        }
        
        public int hashCode() {
            return mask;
        }
        
        public int logLast() {
            return log(last);
        }
        
        public int log(int powerof2) {
            int i = 0, j = 1;
            while(j != powerof2) {
                j <<= 1;
                i += 1;
            }
            return i;
        }
        
        public boolean equals(Object o) {
            if (o instanceof DPStructure) {
                DPStructure other = (DPStructure) o;
                return (mask == other.mask && last == other.last);
            }
            return false;
        }
        
        public List<DPStructure> getSubstructures() {
            LinkedList<DPStructure> ret = new LinkedList();
            for (int i=1; i<=mask; i<<=1) {
                if (i == last) {
                    continue;
                }
                if ((i&mask) == 0) {
                    continue;
                }
                ret.add(new DPStructure(mask-last, i));
            }
            return ret;
        }
        
        public String toString() {
            StringBuffer ret = new StringBuffer().append("DPStructure{{");
            int i = 1;
            while(i<=mask && ((i&mask) == 0)) {
                i <<= 1;
            }
            ret.append(log(i));
            for(i<<=1;i<=mask;i<<=1) {
                if ((i&mask) != 0) {
                    ret.append(" ").append(log(i));
                }
            }
            ret.append("}, ").append(logLast()).append("}");
            return ret.toString();
        }
    }
    
    public HashMap<DPStructure, String> dp;
    public String[] strs;
    
    public String shortestSuperstring(String[] A) {
        dp = new HashMap();
        strs = A;
        LinkedList<DPStructure> finalStructures = new LinkedList();
        int n = A.length;
        LinkedList<Integer> list = new LinkedList();
        for(int i=0;i<n;i++) {
            list.add(i);
        }
        String ret = null;
        for(int i=0;i<n;i++) {
            String tmp = getDPResult(new DPStructure(list));
            if (ret == null || tmp.length() < ret.length()) {
                ret = tmp;
            }
            Integer first = list.removeFirst();
            list.add(first);
        }
        return ret;
    }
    
    private String getDPResult(DPStructure structure) {
        String ret = dp.get(structure);
        if (ret == null) {
            ret = computeDPResult(structure);
            dp.put(structure, ret);
        }
        return ret;
    }
    
    private String computeDPResult(DPStructure structure) {
        List<DPStructure> subStructures = structure.getSubstructures();
        int last = structure.logLast();
        if (subStructures.size() == 0) {
            return strs[last];
        }
        String ret = null;
        for (DPStructure subStructure : subStructures) {
            String tmp = getShortestMerge(getDPResult(subStructure), strs[last]);
            if (ret == null || ret.length() > tmp.length()) {
                ret = tmp;
            }
        }
        // System.out.println("dp[" + structure + "]=" + ret);
        return ret;
    }
    
    private String getShortestMerge(String suffix, String appendix) {
        return new String(getShortestMerge(suffix.toCharArray(), appendix.toCharArray()));
    }
    
    private char[] getShortestMerge(char[] suffix, char[] appendix) {
        // Can actually do KMP preprocess for all @appendix ... but don't remember that thing right now. No time to look up.
        int i = suffix.length - appendix.length;
        if (i < 0) {
            i = 0;
        }
        while (true) {
            int j = 0, k = i;
            while (k < suffix.length) {
                if (suffix[k] != appendix[j]) {
                    break;
                }
                k += 1;
                j += 1;
            }
            if (k == suffix.length) {
                break;
            }
            i += 1;
        }
        
        //now return suffix[0..i-1] + appendix
        char[] ret = new char[i + appendix.length];
        for(int k=0;k<i;k++) {
            ret[k] = suffix[k];
        }
        for(int k=0;k<appendix.length;k++) {
            ret[k+i] = appendix[k];
        }
        return ret;
    }
}
