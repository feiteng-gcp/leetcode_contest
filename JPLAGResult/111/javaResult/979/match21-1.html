class Solution {
public int[] diStringMatch(String S) {
		int[] res = new int[S.length() + 1];

		int hi = S.length();
		int lo = 0;

		if (S.charAt(0) == 'I') {
			res[0] = lo++;
		} else {
			res[0] = hi--;
		}

		for (int i = 1; i < S.length(); i++) {

			if (S.charAt(i) == 'I') {
				res[i] = lo++;
			} else {
				res[i] = hi--;
			}
		}
		res[S.length()] = hi;

		return res;
	}
}
