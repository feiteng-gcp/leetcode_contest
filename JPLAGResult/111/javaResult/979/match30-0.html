class Solution {
    public int[] diStringMatch(String S) {
        int num = 0;
        int[] result = new int[S.length() + 1];
        Arrays.fill(result, -1);
        for(int i = 0; i < S.length(); i++){
            if(S.charAt(i) == 'I'){
                //System.out.println(i + " " + num);
                result[i] = num++;
            }
        }
        
        num = S.length();
        for(int i = 0; i < S.length(); i++){
            if(S.charAt(i) == 'D'){
                //System.out.println(i + " " + num);
                result[i] = num--;
            }
        }
        for(int i = 0; i < result.length; i++){
            if(result[i] == -1){
                result[i] = num;
            }
        }
        return result;
    }
}
