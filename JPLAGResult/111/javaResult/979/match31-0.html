class Solution {
    public int[] diStringMatch(String S) {
        int max = S.length();
        int min = 0;
        int[] ans = new int[S.length()+1];
        for(int i=0;i<S.length();i++){
            if(S.charAt(i)=='D'){
                ans[i] = max;
                max --;
            }
            else{
                ans[i] = min;
                min ++;
            }
        }
        ans[S.length()] = max;
        return ans;
    }
}
