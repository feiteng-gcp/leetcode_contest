class Solution {
  public int[] diStringMatch(String S) {
            int n = S.length();
            int[] ret = new int[n + 1];
            int countDown = 0, countUp = 0;
            
for (int i = 0; i < n; i++) {
                if (S.charAt(i) == 'D') countDown++;
                else countUp++;
            }
            
int lo = countDown;
            int hi = lo;
            ret[0] = lo;
            for (int i = 0; i < n; i++) {
                if (S.charAt(i) == 'I') {
                    hi++;
                    ret[i + 1] = hi;
                } else {
                    lo--;
                    ret[i + 1] = lo;
                }
            }
            return ret;
        }
}
