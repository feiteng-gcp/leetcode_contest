class Solution {
    public int[] diStringMatch(String S) {
        int n = S.length();
        int[] res = new int[n + 1];
        Arrays.fill(res, -1);

        int start = 0;
        for (int i = S.length() - 1; i >= 0; i--) 
{
            if (S.charAt(i) == 'D') {
                res[i + 1] = start;
                start++;
            }
        }

        
for (int i = 0; i < res.length; i++) {
            if (res[i] == -1) {
                res[i] = start;
                start++;
            }
        }

        return res;
    }
}
