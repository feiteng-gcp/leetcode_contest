class Solution {
    public int[] diStringMatch(String S) {
        int[] result = new int[S.length() + 1];
        int num = result.length - 1;
        
for (int i = 0; i < S.length(); i++) {
            if (S.charAt(i) == 'D') {
                result[i] = num;
                num--;
            }
        }
        num = 0;
        for (int i = 0; i < result.length; i++) {
            if (result[i] == 0) {
                result[i] = num;
                num++;
            }
        }
        
        return result;
        
       
    }
}
