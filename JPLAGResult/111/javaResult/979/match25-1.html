class Solution {
    public int[] diStringMatch(String S) {
        int[] res = new int[S.length()+1];
        for(int i = 0; i <= S.length(); i++) {
            res[i] = i;
        }
        int index = 0;
        while(index < S.length()) {
            int count = 0;
            while(index <S.length() && S.charAt(index) == 'I') {
                index++;
            }
           // index++;
            while(index <S.length() && S.charAt(index) == 'D') {
                index++;
                count++;
            }
            if(count != 0)
                reverse(res,index-count,index)
;
            //index++;
        }
        return res;
    }
    
    private void reverse(int[] num, int lo, int hi) {
        while(lo < hi) {
            int val = num[lo];
            num[lo] = num[hi];
            num[hi] = val;
            lo++;hi--;
        }
    }
}
