class Solution {
    public int[] diStringMatch(String S) {
        int N = S.length();
        int start = 0, end = N;
        int[] ret = new int[N+1];
        for (int i = 0; i < N; ++i) {
            if (S.charAt(i) == 'I') {
                ret[i] = start;
                start += 1;
            } else {
                ret[i] = end;
                end -= 1;
            }
        }
        ret[N] = start;
        return ret;
    }
}
