class Solution {
    public int[] diStringMatch(String S) {
        int lowerIdx = 0;
        int highIdx = S.length();
        int[] nums = new int[S.length() + 1];
        int numIdx = 0;
        for (char idx : S.toCharArray()) {
            if (idx == 'I') {
                nums[numIdx++] = lowerIdx;
                lowerIdx++;
            } else {
                nums[numIdx++] = highIdx;
                highIdx--;
            }
        }
        nums[numIdx] = lowerIdx;
        return nums;
    }
}
