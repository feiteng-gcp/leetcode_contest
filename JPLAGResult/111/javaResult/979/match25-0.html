class Solution {
    public int[] diStringMatch(String S) {
        int[] res = new int[S.length() + 1];
        for(int i = 0; i <= S.length(); i++){
            res[i] = i;
        }
        int i = 0;
        while(i < S.length()){
            if(S.charAt(i) == 'D'){
                int j = i;
                while(i < S.length() && S.charAt(i) == 'D'){
                    i++;
                }
                swap(res, j, i);
            }else{
                i++;
            }
        }
        return res;
    }
    
    public void swap(int[] res, int left, int right){
        while(left < right){
            int tmp = res[left];
            res[left] = res[right];
            res[right] = tmp;
            left++;
            right--;
        }
    }
}
