public class Solution {
	public int[] diStringMatch(String S) {
		int lo = 0, hi = S.length();
		int[] ans = new int[S.length() + 1];
		for (int i = 0; i < S.length(); ++i) {
			char ch = S.charAt(i);
			if (ch == 'D') ans[i] = hi--;
			else ans[i] = lo++;
		}
		ans[S.length()] = lo;
		return ans;
	}
}
