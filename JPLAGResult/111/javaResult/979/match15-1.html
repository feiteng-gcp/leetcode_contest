class Solution {
    public int[] diStringMatch(String S) {
        int len = S.length() + 1;
        int lo = 0, hi = len - 1;
        int[] res = new int[len];
        int i = 0;
        for (char c: S.toCharArray()) {
            if (c == 'D') 
{
                res[i++] = hi--;
            }
            else {
                res[i++] = lo++;
            }
        }
        res[i] = hi;
        return res;
    }
}
