class Solution {
    public int[] diStringMatch(String S) {
        int N = S.length(), p = 0, n = 0;
        int[] result = new int[N + 1];
        result[0] = 0;
        for (int i = 0; i < N; i++) {
            if (S.charAt(i) == 'I') {
                result[i + 1] = ++p;
            } else {
                result[i + 1] = --n;
            }
        }
        for (int i = 0; i < N + 1; i++) {
            result[i] -= n;
        }
        return result;
    }
}
