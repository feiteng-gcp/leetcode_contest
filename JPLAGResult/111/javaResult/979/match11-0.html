class Solution {
    public int[] diStringMatch(String S) {
        int n = S.length() + 1;
        int[] a = new int[n];
        Arrays.fill(a, -1);
        char[] cs = S.toCharArray();
        int acc = 0;
        for (int i = 0; i < n-1; i++) {
            if (S.charAt(i) == 'I') {
                a[i] = acc++;
            }
        }
        a[n-1] = acc++;
        for (int i = n-2; i >= 0; i--) {
            if (a[i] == -1) {
                a[i] = acc++;
            }
        }
        return a;
    }
}
