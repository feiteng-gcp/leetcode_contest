class Solution {
    public int[] diStringMatch(String S) {
        int n = S.length();
        int i = 0, j = n;
        int[] res = new int[n+1];
        for(int k = 0; k < n; k++) {
            if(S.charAt(k) == 'I'){
                res[k] = i++;
            }else {
                res[k] = j--;
            }
        }

        res[res.length - 1] = j;
        return res;

    }
}
