	class Solution {
	    public int[] diStringMatch(String S) {
	        int n = S.length() + 1;
	        int[] ret = new int[n];
	        int v = n-1;
	        int pre = 0;
	        for(int i = 0;i < n-1;i++){
	        	if(S.charAt(i) == 'D'){
	        		for(int j = i;j >= pre;j--){
	        			ret[j] = v;
	        			v--;
	        		}
	        		pre = i+1;
	        	}
	        }
    		for(int j = n-1;j >= pre;j--){
    			ret[j] = v;
    			v--;
    		}
    		return ret;
	    }
	}
