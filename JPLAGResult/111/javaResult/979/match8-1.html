class Solution {
    public int[] diStringMatch(String S) {
        int maxNum = 0;
        int minNum = 0;
        int[] perm = new int[S.length() + 1];
        perm[0] = 0;
        for (int i = 0; i < S.length(); i ++) {
            if (S.charAt(i) == 'I') {
                maxNum ++;
                perm[i+1] = maxNum;
            } else {
                minNum --;
                perm[i+1] = minNum;
            }
        }
        
        int adjustNum = 0 - minNum;
        if (adjustNum == 0) {
            return perm;
        }
        
        
for (int i = 0; i < perm.length; i ++) {
            perm[i] += adjustNum;
        }
        
        return perm;
    }
}
