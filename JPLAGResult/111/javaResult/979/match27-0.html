class Solution {
    public int[] diStringMatch(String S) {
        int i = 0;
        int d = S.length();
        
        int[] res = new int[S.length() + 1];
        int index = 0;
        
        for (char ch : S.toCharArray()) {
            if (ch == 'I') {
                res[index++] = i++;
            } else {
                res[index++] = d--;
            }
        }
        res[index++] = i;
        
        return res;
    }
}
