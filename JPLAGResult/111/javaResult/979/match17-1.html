class Solution {
    public int[] diStringMatch(String S) {
        
int l = 0, r = S.length(), n = r;
        int[] res = new int[r+1];
        for (int i = 0; i < n; i++ ) {
            if (S.charAt(i) == 'I') {
                res[i] = l;
                l++;
            } else {
                res[i] = r;
                r--;
            }
        }
        res[n] = l;
        return res;
    }
}
