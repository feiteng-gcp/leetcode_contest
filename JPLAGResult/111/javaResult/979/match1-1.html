class Solution {
    public int[] diStringMatch(String S) {
        int idx = 0;
        
int len = S.length();
        int in = 0, de = len;
        int[] ans = new int[len+1];
        for(int i = 0 ; i < len ; i ++){
            if(S.charAt(i) == 'D'){
                ans[idx] = de;
                de--;
            }else{
                ans[idx] = in;
                in++;
            }
            idx++;
        }
        ans[len] = de;
        return ans;
    }
}
