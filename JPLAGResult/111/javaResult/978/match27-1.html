class Solution {
    public boolean validMountainArray(int[] b) {
        
int indM =0;
        for(int s = 1 ; s < b.length ; s++) {
            if(b[s] > b[indM]) {
                indM = s;
            }
        }
        
for(int r = indM + 1 ; r < b.length ; r++) {
            if(b[r] >= b[r - 1]) {
                return false;
            }
        }
        for(int r = indM - 1 ; r >= 0; r--) {
            if(b[r] >= b[r + 1]) {
                return false;
            }
        }
        if(indM == 0 || indM == b.length - 1) return false;
        
        return true;
        
    }
}
