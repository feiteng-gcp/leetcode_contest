class Solution {
    public boolean validMountainArray(int[] A) {
        if (A.length <= 2) {
            return false;
        }
        if (A[1] <= A[0]) return false;
        int i = 1;
        while (i + 1 < A.length && A[i + 1] > A[i]) {
            i++;
        }
        if (i == A.length - 1) {
            return false;
        }
        while (i + 1 < A.length) {
            if (A[i + 1] >= A[i]) {
                return false;
            }
            i++;
        }
        return true;
    }
}
