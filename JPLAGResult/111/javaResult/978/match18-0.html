class Solution {
    public boolean validMountainArray(int[] A) {
        if (A == null || A.length < 3) return false;
        boolean inc = true;
        boolean isInc = false;
        for (int i = 1; i < A.length; i++) {
            if (A[i] == A[i - 1]) return false;
            if (A[i] < A[i - 1]) {
                inc = false;
            } else {
                if (!inc) return false;
                isInc = true;
            }
        }
        return inc == false && isInc;
    }
}
