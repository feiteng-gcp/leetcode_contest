class Solution {
    public boolean validMountainArray(int[] A) {
        if (A == null || A.length < 3) {
            return false;
        }
        int n = A.length;
        int i = 0;
        while (i < n - 1 && A[i] < A[i + 1]) {
            i++;
        }
        if (i == n - 1) {
            return false;
        }
        int j = n - 1;
        while (j > 0 && A[j] < A[j - 1]) {
            j--;
        }
        if (j == 0) {
            return false;
        }
        return i == j;
    }
}
