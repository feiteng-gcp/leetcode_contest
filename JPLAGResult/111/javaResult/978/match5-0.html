class Solution {
    public boolean validMountainArray(int[] A) {
        if(A.length < 3){
            return false;
        }
        int idx = 0;
        for(int i = 0; i < A.length; i++){
            if(A[i] > A[idx]){
                idx = i;
            }
        }
        if(idx == 0 || idx == A.length - 1){
            return false;
        }
        for(int i = 0; i < idx; i++){
            if(A[i] >= A[i + 1]){
                return false;
            }
        }
        for(int i = idx + 1; i < A.length; i++){
            if(A[i] >= A[i - 1]){
                return false;
            }
        }
        return true;
    }
}
