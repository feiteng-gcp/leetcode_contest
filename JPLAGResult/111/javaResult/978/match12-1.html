class Solution {
    public boolean validMountainArray(int[] A) {
           if (A.length < 3) {
                   return false;
           } 
           int n = A.length; 
           int index = -1;
            
           for (int i = 1; i < n; i++) {
                   if (A[i] > A[i-1]) {
                           continue;
                   }else {
                           index = i-1;
                           break;
                   
}
           } 
           if (index == -1 || index == 0) {
                   return false;
           }
           for (int i = index + 1; i < n; i++) {
                   if (A[i] < A[i-1]) {
                           continue;
                   }else 
{
                           return false;
                   }
           } 
           return true; 
    }
}
