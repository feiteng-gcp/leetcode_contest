class Solution {
    public boolean validMountainArray(int[] A) {
        int n = A.length;
        
if(n < 3) return false;
        int i = 0;
        while(i < n - 2 && A[i] < A[i + 1]) i++;
        while(i > 0 && i < n - 1 && A[i + 1] < A[i]) i++;
        return i == n - 1;
        
    }
}
