class Solution {
    public boolean validMountainArray(int[] A) {
        if (A.length < 3) return false;
        if (A[0] >= A[1]) return false;
        boolean findPeek = false;
        for (int i = 1; i < A.length - 1; i++) {
            if (A[i] == A[i+1]) return false;
            else if (A[i] > A[i+1]) {
                findPeek = true;
            
} else {
                if (findPeek) return false;
            }
        }
        return findPeek;
        
    }
}
