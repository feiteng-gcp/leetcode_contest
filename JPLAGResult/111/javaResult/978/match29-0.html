class Solution {
    public boolean validMountainArray(int[] A) {
        if(A.length < 3) return false;
        if(A[0] > A[1] || A[A.length - 2] < A[A.length - 1]) return false;
        
        int i = 0;
        while(i + 1 < A.length && A[i] < A[i+1]) ++i;
        ++i;
        while(i < A.length && A[i - 1] > A[i]) ++i;
        return i == A.length;
    }
}
