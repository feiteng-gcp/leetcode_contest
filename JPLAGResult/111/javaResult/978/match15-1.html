class Solution {
    public boolean validMountainArray(int[] A) {
        if (A == null || A.length < 3)
            return false;
        boolean findPeak = false;
        for (int i = 1; i < A.length; i++) {
            if (A[i] == A[i - 1])
                return false;
            else if (A[i] > A[i - 1]) {
                if (findPeak)
                    return false;
            } else {
                if (i == 1)
                    return false;
                findPeak = true;
            
}
        }
        return findPeak;
    }
}
