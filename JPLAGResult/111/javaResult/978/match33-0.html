import java.util.*;
import java.util.stream.*;
import java.math.*;

public class Solution {


    public boolean validMountainArray(int[] a) {
        if (a == null || a.length < 3) return false;
        int i = 0, j = a.length - 1;
        while (i < j) {
            if (a[i] < a[i + 1]) i++;
            else if (a[j - 1] > a[j]) j--;
            else return false;
        }
        return i > 0 && i < a.length - 1;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.validMountainArray(new int[]{2, 1}));
        System.out.println(sol.validMountainArray(new int[]{3, 5, 5}));
        System.out.println(sol.validMountainArray(new int[]{0, 3, 2, 1}));
        System.out.println(sol.validMountainArray(new int[]{4, 3, 2, 1}));

    }
}
