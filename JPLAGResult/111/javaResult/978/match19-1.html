class Solution {
    public boolean validMountainArray(int[] A) {
        
if (A.length < 3)
            return false;
        int index = 1;
        while (index < A.length && A[index-1] < A[index])
            index++;
        if (index == 1 || index == A.length)
            return false;
        while (index < A.length && A[index-1] > A[index])
            index++;
        return index == A.length;
    }
}
