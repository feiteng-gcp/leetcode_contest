class Solution {
    public boolean validMountainArray(int[] A) {
        int n = A.length;
        if (n < 3) {
            return false;
        }
        int max = 0;
        int index = 0; 
        for (int i = 0; i < n; i++) {
            if (A[i] > max) {
                max = A[i];
                index = i; 
            }
        }
        if (index == 0 || index == n - 1) {
            return false; 
        }
        for (int i = index - 1; i >= 0; i--) {
            if (A[i] >= A[i + 1]) {
                return false;
            }
        }
        for (int i = index + 1; i < n; i++) {
            if (A[i - 1] <=  A[i]) {
                return false;
            }
        }
        return true; 
    }
}
