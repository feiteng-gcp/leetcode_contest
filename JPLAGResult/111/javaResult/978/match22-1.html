class Solution {
    public boolean validMountainArray(int[] A) {
        if(A.length<3) return false;
        int count=0;
        int index=-1;
        for(int i=0;i<A.length-1;i++){
            if(A[i]<A[i+1]) count++;
            else
{
                index=i;
                break;
            }
        }
        if(count==0 || count==A.length-1) return false;
        for(int i=index;i<A.length-1;i++){
            if(A[i]>A[i+1]) count++;
            else break;
        }
        return count==A.length-1;
    }
}
