class Solution {
    public boolean validMountainArray(int[] A) {
        
        if (A.length < 3) { return false; }
        
        // find the first peak before dropping
        int peak = 0;
        for (; peak < A.length - 1; peak ++) {
            if (A[peak + 1] <= A[peak]) { break; 
}
        }
        
        if (peak == 0 || peak == A.length - 1) { return false; }
        
        // check if all remaining array is in strict decending order
        for (; peak < A.length - 1; peak ++) 
{
            if (A[peak + 1] >= A[peak]) { return false; }
        }
        
        return true;
    }
}
