class Solution {
    public boolean validMountainArray(int[] a) {
        if (a.length == 1) {
            return false;
        }
        boolean isIncrease = false;
        boolean isDecrease = false;
        
int i = 0;
        while (i < a.length - 1 && a[i] < a[i + 1]) {
            i++;
        }
        if (i == 0 || i == a.length - 1) {
            return false;
        }
        while (i < a.length - 1 && a[i] > a[i + 1]) {
            i++;
        }
        return i == a.length - 1;
    }
}
