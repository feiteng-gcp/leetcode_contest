class Solution {
    public boolean validMountainArray(int[] A) {
        if (A.length < 3) return false;
        int l = 0, r = A.length - 1;
        while (l < r) {
            if (A[l] < A[l+1]) l++;
            else if (A[r] < A[r-1]) r--;
            else return false;
        }
        if (l == 0 || r == A.length-1) return false;
        return true;
    }
}
