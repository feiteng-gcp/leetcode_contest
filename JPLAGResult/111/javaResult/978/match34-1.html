class Solution {
    public boolean validMountainArray(int[] A) {
        if(A.length < 3)    return false;
        int n = A.length, i = 1;
        while(i < n && A[i] > A[i-1])
        {
            i++;
        }
        if(i == A.length || i == 1)   return false;
        while(i < n && A[i] < A[i-1])
            i++;
        return i == A.length;
    }
}
