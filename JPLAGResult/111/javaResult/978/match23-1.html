class Solution {
    public boolean validMountainArray(int[] A) {
        if (A.length < 3) return false;
        
int i=1;
        while (i<A.length-1 && A[i] > A[i-1]) {
            i++;
        }
        if ((i==A.length-1 && A[i] > A[i-1]) || (i==1 && A[i] < A[i-1])) return false;
        while (i<A.length && A[i] < A[i-1]) {
            i++;
        }
        return i == A.length;
    }
}
