class Solution {
    public boolean isIncreasing(int[] A){

        for(int i=0;i<A.length-1;i++){
            if(A[i] > A[i+1])
                return false;
        }

        return true;
    }

    public boolean isDecreasing(int[] A){

        for(int i=0;i<A.length-1;i++){
            if(A[i] < A[i+1])
                return false;
        }

        return true;
    }

    public boolean validMountainArray(int[] A) {

        if(A == null || A.length < 3)
            return false;

        
int peekPoint = 0;

        for(int i=0;i<A.length-1;i++){
            if(A[i] >= A[i+1]){
                peekPoint = i;
                break;
            }
        }

        if(peekPoint == 0 || peekPoint == A.length-1)
            return false;

        for(int i=peekPoint;i<A.length-1;i++){
            if(A[i] <= A[i+1]){
                return false;
            }
        }

        return true;

    }
}
