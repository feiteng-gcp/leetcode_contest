class Solution {
    public boolean validMountainArray(int[] A) {
        if(A.length < 3) return false;
        if(A[1] < A[0]) return false;
        boolean inc = true;
        
        for(int i = 1; i < A.length; i++) {
        	if(A[i] == A[i-1]) return false;
        	else if(A[i] < A[i-1]) {
        		if(inc) inc = false;
        	}else {
        		if(!inc) return false;
        	}
        }
        if(inc) return false;
        return true;
    }
}
