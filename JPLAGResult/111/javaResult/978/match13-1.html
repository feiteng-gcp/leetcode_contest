class Solution {
	public boolean validMountainArray(int[] A) {
		if (A.length < 3) {
			return false;
		}

		int max = 0;

		for (int i = 1; i < A.length; i++) {
			if (A[i] > A[max]) {
				max = i;
			}
		}

		if (max == 0 || max == A.length - 1) {
			return false;
		}

		for (int i = m
ax + 1; i < A.length; i++) {
			if (A[i - 1] <= A[i]) {
				return false;
			}
		}

		for (int i = 1; i <= max; i++) {
			if (A[i - 1] >= A[i]) {
				return false;
			}
		}

		return true;
	}
}
