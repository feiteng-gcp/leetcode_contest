class Solution {
    public boolean validMountainArray(int[] A) {
                if(A.length < 3) return false;
        int best = 0;
        for (int i = 0; i < A.length; i++) {
            if (A[i] > A[best]) {
                best = i;
            }
        }
        
        if (best == 0 || best == A.length - 1)
            return false;
        
        for (int i = 1; i <= best; i++) {
            if (A[i-1] >= A[i]) {
                return false;
            }
        }
        
        for (int i = best + 1; i < A.length; i++) {
            if (A[i-1] <= A[i])
                return false;
        }
        
        return true;
    }
}
