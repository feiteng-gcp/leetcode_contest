class Solution {
    public boolean validMountainArray(int[] A) {
        if (A.length < 3) return false;
        int s = A[0], i = 1;
        while (i < A.length && A[i] > A[i-1]) {
            i++;
        }
        if (i == 1 || i == A.length) return false;
        int p = A[i-1];
        while (i < A.length) {
            if (A[i] >= A[i-1]) return false;
            i++;
        }
        return true;
    }
}
