	class Solution {
	    public boolean validMountainArray(int[] A) {
	        if(A.length < 3)return false;
	        int n = A.length;
	        int pre = n-1;
	        for(int i = 0;i < n-1;i++){
	        	if(A[i] >= A[i+1]){
	        		pre = i;
	        		break;
	        	}
	        }
	        if(pre == 0 || pre == n-1)return false;
	        for(int i = pre;i < n-1;i++){
	        	if(A[i] <= A[i+1]){
	        		return false;
	        	}
	        }
	        return true;
	    }
	}
