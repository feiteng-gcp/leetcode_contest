class Solution {
    public int minDeletionSize(String[] A) {
        int num = A.length;
        if(num == 0) return 0;
        int cols = A[0].length();
        int ans = 0;
        for(int i=0;i<cols;i++){
            for(int j=0;j<num-1;j++){
                if(A[j].charAt(i)>A[j+1].charAt(i)){
                    ans ++;
                    break;
                }
            }
        }
        return ans;
    }
}
