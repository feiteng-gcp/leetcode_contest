class Solution {
    public int minDeletionSize(String[] A) {
        int r = A.length;
        int c = A[0].length();
        int del = 0;
        for(int i=0;i<c;i++)
        {
            for(int j=0;j<r-1;j++)
            {
                if(A[j].charAt(i) > A[j+1].charAt(i))
                {
                    ++del;
                    break;
                }
            }
        }
        return del;
    }
}
