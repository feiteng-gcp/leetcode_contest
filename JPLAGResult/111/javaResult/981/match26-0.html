class Solution {
    public int minDeletionSize(String[] A) {
        if(A == null || A.length < 2) return 0;
        int ans = 0;
        int len = A[0].length();
        for(int i = 0 ; i < len ; i ++){
            for(int j = 1 ; j < A.length ; j ++){
                if(A[j].charAt(i) < A[j-1].charAt(i)){
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
}
