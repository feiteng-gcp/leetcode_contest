class Solution {
    public int minDeletionSize(String[] A) {
        int len = A.length;
        if (len == 0) return 0;
        int strLen = A[0].length(), r = 0;
        for (int i = 0; i < strLen; i++) {
            for (int j = 1; j < len; j++) {
                if (A[j - 1].charAt(i) > A[j].charAt(i)) {
                    r++;
                    break;
                }
            }
        }
        return r;
    }
}
