class Solution {
    public int minDeletionSize(String[] A) {
        int len = A.length;
        if(len == 0) return 0;
        
        int width = A[0].length();
        
        int count = 0;
        for(int i = 0; i < width; i++) {
            for(int j = 0; j < len - 1; j++) {
                if(A[j].charAt(i) > A[j + 1].charAt(i)) {
                    count++;
                    break;
                }   
            }
        }
        return count;
    }
}
