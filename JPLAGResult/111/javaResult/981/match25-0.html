class Solution {
    public int minDeletionSize(String[] A) {
        
        // base cases
        if (A.length <= 1) {
            return 0;
        }
        
        int numDelete = 0;
        int strLen = A[0].length();
        for (int i = 0; i < strLen; i ++) {
            for (int j = 0; j < A.length - 1; j ++) {
                if (A[j].charAt(i) > A[j + 1].charAt(i)) {
                    numDelete ++;
                    break;
                }
            }
        }
        
        return numDelete;
    }
}
