class Solution {
    public int minDeletionSize(String[] A) {
        if(A == null || A.length < 2){
            return 0;
        }
        int n = A.length;
        int m = A[0].length();
        int cnt = 0;
        
for(int col = 0; col < m; col++){
            for(int row = 1; row < n; row++){
                if(A[row].charAt(col) < A[row-1].charAt(col)){
                    cnt++;
                    break;
                }
            }
        }
        return cnt;
    }
}
