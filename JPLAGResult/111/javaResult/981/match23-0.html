class Solution {
           public int minDeletionSize(String[] A) {
            int h=A.length,w=A[0].length();
            int ans=0;
            for (int i=0;i<w;++i) {
                for (int j=1;j<h;++j) {
                    if (A[j].charAt(i)<A[j-1].charAt(i)) {
                        ++ans;
                        break;
                    }
                }
            }
            return ans;
        }
}
