class Solution {
    public int minDeletionSize(String[] A) {
        int m = A.length;
        if (m == 0) return 0;
        int n = A[0].length();
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m - 1; j++) {
                if (A[j].charAt(i) > A[j+1].charAt(i)) {
                    res++;
                    break;
                }
            }
        }
        return res;
    }
}
