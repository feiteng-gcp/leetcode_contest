class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        int n = A.size();
        if (n < 3) return false;
        int i = 1;
        while (i < n && A[i] > A[i - 1]) i++;
        if (i == 1 || i >= n) return false;
        while (i < n && A[i] < A[i - 1]) i++;
        if (i < n) return false;
        return true;
    }
};
