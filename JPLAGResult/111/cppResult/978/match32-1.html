class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        if (A.size() < 3) {
            return false;
        }
        int peak = -1;
        for (int i = 1; i < A.size(); ++i) {
            if (A[i] == A[i - 1]) {
                return false;
            }
            if (A[i] < A[i - 1]) {
                if (peak == -1 && i != 1) {
                    peak = i - 1;
                    break;
                } else {
                    return false;
                }
            }

        }
        for (int i = peak + 1; i < A.size(); ++i) {
            if (A[i] >= A[i - 1]) {
                return false;
            }
        }
        return true;
    }
};
