class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        int flag = 0;
        int n = A.size();

        if (n < 3) return false;
        
        int peak = -1;
        for (int i=1; i<n-1; i++) {
            if (A[i] > A[i-1] && A[i] > A[i+1]) {
                peak = i;
                break;

            }
        }
        
        if (peak == -1) 
            return false;
        
        for (int i=1; i<=peak; i++) {
            if (A[i] <= A[i-1]) {
                return false;
            }
        }
        for (int i=peak; i<n-1; i++) {
            if (A[i] <= A[i+1]) {
                return false;
            }
        }
        return true;
    }
};
