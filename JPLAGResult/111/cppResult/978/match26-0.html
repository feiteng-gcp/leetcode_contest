class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        int n=A.size();
        int mx=-1, id=-1;
        for(int i=0; i<n; i++){
            if(A[i]>mx){
                id=i;
                mx=A[i];
            }
        }
        if(id==0 || id==n-1) return false;
        
        for(int i=0; i<id; i++){
            if(A[i]>=A[i+1]) return false;
        }
        
        for(int i=id; i<n-1; i++){
            if(A[i]<=A[i+1]) return false;
        }
        
        return true;
    }
};
