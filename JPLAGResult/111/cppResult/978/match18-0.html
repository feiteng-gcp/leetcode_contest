class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        int N = A.size();
        if(N < 3) return false;
        int i = 0;
        while(i+1 < N && A[i] < A[i+1]) i ++;
        if(0 == i || i+1 == N) return false;
        while(i+1 < N && A[i] > A[i+1]) i ++;
        return (i+1 == N) ? true : false;
    }
};
