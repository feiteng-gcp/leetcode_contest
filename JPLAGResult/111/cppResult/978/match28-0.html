#include <bits/stdc++.h>
using namespace std;
using i64 = int64_t;
using vi = vector<i64>;
using vvi = vector<vi>;

class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        if (A.size() < 3) return false;
        int nax = -2e9, id;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] > nax) {
                nax = A[i];
                id = i;
            }
        }
        if (id == 0 || id == A.size() - 1) return false;

        for (int i = 0; i < id; i++) {
            if (A[i] >= A[i + 1]) {
                return false;
            }
        }

        for (int i = id + 1; i < A.size(); i++) {
            if (A[i] >= A[i - 1]) {
                return false;
            }
        }
        return true;
    }
};
