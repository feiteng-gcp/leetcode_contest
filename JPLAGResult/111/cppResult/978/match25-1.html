class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        if(A.size()<3) return false;

		int ln=A.size();
		int maxv=-1, maxi=-1;

		for(int i=0; i<ln;  i++){
			if(A[i]>maxv){
				maxv=A[i];
				maxi=i;
			}
		}
		if(maxi==0 || maxi==ln-1) return false;

		bool f=true;

		for(int i=0; i<maxi; i++){
			if(A[i]>=A[i+1]) return false;
		}
		for(int i=maxi; i<ln-1; i++){
			if(A[i]<=A[i+1]) return false;
		}
		return true;

    }
};
