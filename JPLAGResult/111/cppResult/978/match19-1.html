class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        //A = {3, 5, 5};
        //A = {0, 3, 2, 1};
        //A = {0, 1, 2, 3};
        if(A.size() < 3) return false;
        int i = 0;       
        for(; i < A.size() - 1; i++)
        {
            if(A[i] == A[i + 1]) return false;

            else if(A[i] > A[i + 1]) break;
        }
        if(i == 0 || i == A.size() - 1) return false;        
        
        for(; i < A.size() - 1; i++)
        {
            if(A[i] <= A[i + 1]) return false;
        }
        return true;
    }
};
