class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        int n = A.size();
        if( n < 3 ) return false;
        int Maxnum = *max_element(A.begin(), A.end());
        int pos = 0;
        for( int i = 0; i < Maxnum; ++ i )
        {
            if( A[ i ] == Maxnum )
            {
                pos = i;
                break;
            }
        }
        if( pos == 0 || pos == n - 1 ) return false;
        for( int i = 1; i <= pos; ++ i )
        {
            if( A[ i ] <= A[ i - 1 ] ) return false;
        }
        for( int i = n - 2; i >= pos; -- i )
        {
            if( A[ i ] <= A[ i + 1 ] ) return false;
        }
        return true;
    }
};
