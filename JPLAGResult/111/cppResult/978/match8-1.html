class Solution {
public:
    bool validMountainArray(vector<int>& a) {
        int  as=a.size();
        if(as < 3)
            return false;
        bool isPeak=false;
        for(int i=1;i<as;i++)
        {
            if(a[i]>a[i-1])
            {
                if(isPeak == false)
                {
                    if(i == as -1)
                        return false;
                    continue;
                }else
                {
                    return false;
                }
            }else if(a[i]<a[i-1])
            {
                if(isPeak == false)
                {
                    if(i==1)
                        return false;
                    isPeak=true;
                    continue;
                }else
                {
                    continue;
                }
            }else
            {
                return false;
            }
        }
        return true;        
    }
};
