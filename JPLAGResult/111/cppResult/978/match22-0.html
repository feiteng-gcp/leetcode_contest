class Solution {
public:
    bool validMountainArray(vector<int>& A) {
        if (A.size() < 3)
            return false;
        int mid = -1;
        for(int i = 1; i < A.size(); i++) {
            if (A[i-1] == A[i])
                return false;
            else if (A[i-1] > A[i]) {
                mid = i;
                break;
            }                
        }
        if (mid == -1 || mid == 1)
            return false;
        for(int i = mid; i < A.size(); i++) {
            if (A[i-1] <= A[i])
                return false;
        }
        return true;
    }
};
