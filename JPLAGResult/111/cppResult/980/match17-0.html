int f[12][1<<12];
int p[12][1<<12];
class Solution {
public:
    
    int w[15][15];
    
    bool check(string s1, string s2) {
        for (int i = 0; i < s1.size(); i++) {
            if (s1[i] != s2[i]) {
                return false;
            }
        }
        return true;
    }
    
    void process(string t1, string t2, int id1, int id2) {
        
        for (int len = min(t1.size(), t2.size()); len > 0; len--) {
            
            string s1;
            string s2;
            
            for (int i = t1.size() - len; i < t1.size(); i++) {
                s1 += t1[i];
            }
            for (int i = 0; i < len; i++) {
                s2 += t2[i];
            }
            
            if (check(s1, s2)) {
                w[id1][id2] = len;
                return;
            }
        }
        
    }
    
    string shortestSuperstring(vector<string>& A) {
        
        memset(w, 0, sizeof w);
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A.size(); j++) {
                if (i == j) continue;
                
                process(A[i], A[j], i, j);
            }
        }
        
        int n = A.size();
        /*
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                printf("%d %d %d\n", i, j, w[i][j]);
            }
        }
        */
        
        memset(f, -1, sizeof f);
        memset(p, -1, sizeof p);
        
        for (int i = 0; i < n; i++) {
            f[i][1<<i] = 0;
        }
        
        for (int i = 0; i < (1<<n); i++) {
            for (int j = 0; j < n; j++) {
                if (f[j][i] == -1) continue;
                
                for (int k = 0; k < n; k++) {
                    if ((i & (1 << k)) != 0) continue;
                    
                    int nxt = i | (1 << k);
                    
                    if (f[k][nxt] == -1) {
                        f[k][nxt] = f[j][i] + w[j][k];
                        p[k][nxt] = j;
                    }
                    else if (f[k][nxt] < f[j][i] + w[j][k]){
                        f[k][nxt] = f[j][i] + w[j][k];
                        p[k][nxt] = j;
                    }   
                }
            }
        }
        
        
        int mx = 0, id = 0 ;
        for (int i = 0; i < n; i++) {
            if (mx < f[i][(1<<n)-1]) {
                mx = f[i][(1<<n)-1];
                id = i;
            }
        }
        
        
        int t = (1<<n)-1;
        vector<int> res;
        res.push_back(id);
        while(p[id][t] != -1) {
            int id2 = p[id][t];
            t -= (t & (1 << id));
            
            id = id2;
            res.push_back(id);
        }
        
        
        string ans = "";
        for (int i = res.size()-1; i >= 0; i--) {
            if (i == res.size()-1) {
                ans += A[res[i]];
            }
            else {
                for (int j = w[res[i+1]][res[i]]; j < A[res[i]].size(); j++) {
                    ans += A[res[i]][j];
                }
            }
        }
        
        return ans;
    }
};
