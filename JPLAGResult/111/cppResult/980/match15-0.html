class Solution {
    bool match(const string & a, const string & b, int d) {
        int na = a.size(), nb = b.size();
        int ia = na - d;
        int ib = 0;
        while(ib < d) {
            if(a[ia] != b[ib])
                return false;
            ia++;
            ib++;
        }
        return true;
    }
    int getdist(const string & a, const string & b) {
        int na = a.size(), nb = b.size();
        for(int i = min(na, nb); i > 0; i--) {
            if(match(a, b, i))
                return nb - i;
        }
        return nb;
    }
    int n;
    int best = INT_MAX;
    vector<vector<int>> dist;
    vector<int> temp;
    vector<int> res;
    vector<int> visited;
    void dfs(int i, int cur) {
        if(temp.size() == n) {
            if(cur < best) {
                best = cur;
                res = temp;
            }
            return;
        }
        if(cur >= best)
            return;
        visited[i] = true;
        for(int j = 0; j < n; j++) {
            if(j == i || visited[j])
                continue;
            temp.push_back(j);
            dfs(j, cur + dist[i][j]);
            temp.pop_back();
        }
        visited[i] = false;
    }
public:
    string shortestSuperstring(vector<string>& A) {
        
        n = A.size();
        dist = vector<vector<int>>(n, vector<int>(n, 0));
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                if(i == j)
                    continue;
                int temp = getdist(A[i], A[j]);
                dist[i][j] = temp;
            }
        }
        visited = vector<int>(n, 0);
        for(int i = 0; i < n; i++) {
            temp.push_back(i);
            dfs(i, A[i].size());
            temp.pop_back();
        }
        /*
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                cout << dist[i][j] << ' ';
            }
            cout << endl;
        }
        cout << endl;
        for(int i = 0; i < n; i++)
            cout << res[i] << ' ';
        cout << endl;
        */
        string result = A[res[0]];
        for(int i = 1; i < n; i++) {
            string temp(A[res[i]].begin() + int(A[res[i]].size()) - dist[res[i - 1]][res[i]], A[res[i]].end());
            result.insert(result.end(), temp.begin(), temp.end());
            //cout << temp << ' ' << result << endl;
        }
        return result;
    }
};
