class Solution {
public:
    string shortestSuperstring(vector<string>& A) {
        string ret = "";
        int n = A.size();
        vector<vector<int>> f(1 << n, vector<int>(n, -1));
        vector<vector<int>> g(1 << n, vector<int>(n, -1));
        vector<vector<int>> e(n, vector<int>(n, 0));
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < n; ++ j) {
                e[i][j] = A[j].size();
                if (i != j) {
                    for (int l = 0; l <= min(A[i].size(), A[j].size()); ++ l)
                        if (A[i].substr(A[i].size() - l, l) == A[j].substr(0, l)) e[i][j] = min(e[i][j], (int)A[j].size() - l);
                }
            }
        for (int i = 0; i < n; ++ i) {
            f[1 << i][i] = A.size();
            g[1 << i][i] = -1;
        }
        for (int i = 0; i < (1 << n); ++ i)
            for (int j = 0; j < n; ++ j)
                if (f[i][j] != -1)
                    for (int k = 0; k < n; ++ k)
                        if ((i & (1 << k)) == 0) {
                            int ti = (i | (1 << k)), tj = k;
                            if (f[ti][tj] == -1 || f[ti][tj] > f[i][j] + e[j][k]) {
                                f[ti][tj] = f[i][j] + e[j][k];
                                g[ti][tj] = j;
                            }
                        }
        for (int j = 0; j < n; ++ j) {
            int i = (1 << n) - 1;
            if (f[i][j] != -1) {
                string temp = "";
                int qi = i, qj = j;
                while (qj != -1) {
                    int pi = qi - (1 << qj), pj = g[qi][qj];
                    if (pj == -1) {
                        temp = A[qj] + temp;
                    } else {
                        temp = A[qj].substr(A[qj].size() - e[pj][qj]) + temp;
                    }
                    qi = pi;
                    qj = pj;
                }
                if (ret == "" || temp.size() < ret.size()) ret = temp;
            }
        }
        return ret;
    }
};
