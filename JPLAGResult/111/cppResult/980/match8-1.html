const int N = 12;
int dp[1 << N][N];
int path[1 << N][N];
int g[N][N];

class Solution {
public:
    int calc(const string& s, const string& t) {
        int n = s.size(), m = t.size();
        for (int k = 1; k < n; ++k) {
            if (n - k <= m && s.substr(k) == t.substr(0, n - k)) return m - n + k;
        }
        return m;
    }

    int solve(int mask, int k, int n) {
      if (mask == (1 << n) - 1) return 0;
        int& ret = dp[mask][k];
        if (ret >= 0) return ret;
        ret = 1 << 30;
        for (int i = 0; i < n; ++i) {
            if (mask & (1 << i)) continue;
            if (ret > g[k][i] + solve(mask | (1 << i), i, n)) {
              ret = g[k][i] + solve(mask | (1 << i), i, n);
              path[mask][k] = i;
            }
        }
        return ret;
    }
    string shortestSuperstring(vector<string>& s) {
        int n = s.size();

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i == j) continue;
                g[i][j] = calc(s[i], s[j]);
                // cout << i << " " << j << " " << s[i] << " " << s[j] << " " << g[i][j] << endl;
            }
        }
        memset(dp, 255, sizeof(dp));
        int ret = 1 << 30, best_i = -1;
        for (int i = 0; i < n; ++i) {
          int cur = (int)s[i].size() + solve(1 << i, i, n);
          if (ret > cur) {
            ret = cur;
            best_i = i;
          }
        }
        int mask = 1 << best_i;
        string ans = s[best_i];
        for (int i = best_i; mask != (1 << n) - 1; ) {
          int j = path[mask][i];
          ans += s[j].substr(s[j].size() - g[i][j]);
          mask |= 1 << j;
          i = j;
        }
        return ans;
    }
};
