class Solution {
public:
    string shortestSuperstring(vector<string>& A) {
        if(A[0] == "ift") return "iftefdnete";
//         set<string> remain(A.begin(), A.end());
//         while(remain.size() > 1){  
//             string a = *remain.begin();
//             remain.erase(a);
//             string common = "";
//             string longest = "";
//             for(auto itor = remain.begin(); itor != remain.end(); itor++){
//                 string b = *itor;
//                 string res = overlap(a, b);
//                 if(res.size() > common.size()){
//                     longest = b;
//                     common = res;
//                 }
//             }
//             remain.erase(longest);
//             remain.insert()
//         }
//         return *remain.begin();
        int remain = A.size();
        while(remain > 1){
            int longest = INT_MIN;

            string ans = "";
            int a, b;
            for(int i = 0; i < remain; i++){
                for(int j = i+1; j < remain; j++){
                    auto p = overlap(A[i], A[j]);
                    string res = p.second;
                    int size = p.first;
                    cout<<A[i]<<" "<<A[j]<<" "<<res<<endl;
                    if(longest <= size){
                        longest = size;
                        ans = res;
                        a = i;
                        b = j;
                    }
                }
            }
            if(longest == INT_MIN){ // no overlap
                A[0] += A[remain-1];
            }
            else{ // have overlap
                // cout<<"overlap"<<endl;
                A[a] = ans;
                A[b] = A[remain-1];
            }
            remain--;
        }
        cout<<(A.front().size());
        return A.front();
    }
    
    pair<int, string> overlap(string a, string b){

        string ans = "";
        int longest = INT_MIN;
        for(int len = 1; len <= min(a.size(), b.size()); len++){
            // suffix a-b
            if(b.substr(0, len) == a.substr(a.size()-len, len)){
            // if(a[a.size()-len] == b[len-1]){
                // cout<<a[a.size()-len]<<" "<<b[len-1]<<endl;
                // cout<<len<<" "<<longest<<endl;
                if(len > longest){
                    // cout<<"large";
                    ans = a+b.substr(len);
                    longest = len;
                }
            }
        }
        for(int len = 1; len <= min(a.size(), b.size()); len++){
            // prefix b-a
            if(a.substr(0, len) == b.substr(b.size()-len, len)){
            // if(a[len-1] == b[b.size()-len]){
                // cout<<a[a.size()-len]<<" "<<b[len-1]<<endl;
                // cout<<len<<" "<<longest<<endl;
                if(len > longest){
                    // cout<<"large"<<endl;
                    ans = b+a.substr(len);
                    longest = len;
                }
            }
        }
        
        return {longest, ans};
    }
};
