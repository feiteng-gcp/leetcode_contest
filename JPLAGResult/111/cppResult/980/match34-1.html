int dp_mem[1 << 12][12];

class Solution {
    vector<string> a_;
    
    int g[16][16];
    
    int go(int rest, int i) {
        int& ret = dp_mem[rest][i];
        if (ret < 0) {
            if (rest > 0) {
                ret = 1000010000;
                for (int j = 0; j < a_.size(); ++j) {
                    if (rest & (1 << j)) {
                        ret = min(ret, (int)(a_[i].size() - g[i][j] + go(rest ^ (1 << j), j)));
                    }
                }
            } else {
                ret = a_[i].size();
            }
        }
        return ret;
    }
    
    bool match(const string& s1, const string& s2, int k) {

        int n = s1.size();
        for (int i = 0; i < k; ++i) {
            if (s1[n - k + i] != s2[i]) return false;
        }
        return true;
    }
    
    string og(int rest, int i, int best) {
        if (!rest) {
            return a_[i];
        }
        for (int j = 0; j < a_.size(); ++j) {
            if (rest & (1 << j)) {
                int x = a_[i].size() - g[i][j] + go(rest ^ (1 << j), j);
                if (x == best) {
                    int y = a_[i].size() - g[i][j];
                    return a_[i].substr(0, y) + og(rest ^ (1 << j), j, best - y);
                }
            }
        }
        return "???";
    }
    
public:
    string shortestSuperstring(vector<string>& a) {
        int n = a.size();
        if (n < 1) return "";
        
        memset(dp_mem, 0xff, sizeof(dp_mem));
        a_ = a;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                for (int k = min(a[i].size(), a[j].size()); k >= 0; --k) {
                    if (match(a[i], a[j], k)) {
                        g[i][j] = k;
                        break;
                    }
                }
            }
        }
        int best = 1000010000;
        int all = (1 << n) - 1;
        for (int i = 0; i < n; ++i) {
            best = min(best, go(all ^ (1 << i), i));
        }
        string ret;
        for (const string& s : a) ret += s;
        for (int i = 0; i < n; ++i) {
            if (best == go(all ^ (1 << i), i)) {
                return og((all ^ (1 << i)), i, best);
            }
        }
        return ret;
    }
};
