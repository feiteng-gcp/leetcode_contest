class Solution {
public:
    int f[10005][22];
    int g[10005][22];
    
    string shortestSuperstring(vector<string>& A) {
        vector <string> v;
        for (int i = 0; i < A.size(); i++) {
            bool exist = false;
            for (int j = 0; j < A.size(); j++) {
                if (j != i && A[j].find(A[i]) != string::npos) {
                    exist = true;
                }
            }
            
            if (!exist) {
                v.push_back(A[i]);
            }
        }
        
        int n = v.size();
        int full = (1 << n);
        for (int i = 0; i < full; i++) {
            for (int j = 0; j < n; j++) {
                f[i][j] = 1e9;
            }
        }
        
        for (int j = 0; j < n; j++) {
            f[1 << j][j] = A[j].size();
            g[1 << j][j] = j;
        }
        

        for (int i = 0; i < full; i++) {
            for (int j = 0; j < n; j++) {
                if (f[i][j] > 1e8) {
                    continue;
                }
                
                for (int k = 0; k < n; k++) {
                    if (!(i & (1 << k))) {
                        int mask = i | (1 << k);

                        int pos = 0;
                        for (int l = A[k].size() - 1; l >= 0; l--) {
                            if (A[j].size() >= l && A[j].substr(A[j].size() - l, l) == A[k].substr(0, l)) {
                                pos = l;
                                break;
                            }
                        }
                        
                        int v = f[i][j] + A[k].size() - pos;
                        if (v < f[mask][k]) {
                            f[mask][k] = v;
                            g[mask][k] = j;
                        }
                    }
                }
            }
        }
        
        int last = 0;
        for (int j = 0; j < n; j++) {
            if (f[full - 1][j] < f[full - 1][last]) {
                last = j;
            }
        }
        
        vector <string> anss;
        int cur = full - 1;
        while (cur > 0) {
            anss.push_back(v[last]);
            int temp = g[cur][last];
            cur -= (1 << last);
            last = temp;
        }
        
        reverse(anss.begin(), anss.end());
        string ans = "";
        for (string s : anss) {
            int pos = 0;
            for (int l = s.size() - 1; l >= 0; l--) {
                if (ans.size() >= l && ans.substr(ans.size() - l, l) == s.substr(0, l)) {
                                pos = l;
                                break;
                }
            }
            ans = ans + s.substr(pos, s.size() - pos);
        }
        
        return ans;
    }
};
