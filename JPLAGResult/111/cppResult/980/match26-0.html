class Solution {
public:
    string shortestSuperstring(vector<string>& A) {
        int n = A.size();
        if (n == 0) {
            return "";
        }
        
        vector<vector<int>> padding(n, vector<int>(n));
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i == j) {
                    continue;
                }
                
                int li = A[i].length();
                int lj = A[j].length();
                int L = min(li, lj);
                bool found = false;
                int commonLen = 0;
                for (int l = L; l >= 1; --l) {
                    int matchs = 0;
                    for (int p = li - l, q = 0; p < li && q < lj; ++p, ++q) {
                        if (A[i][p] != A[j][q]) {
                            break;
                        }
                        ++matchs;
                    }
                    if (matchs == l) {
                        found = true;
                        commonLen = l;
                        break;
                    }
                }
                padding[i][j] = lj - commonLen;
            }
        }
        
        int N = 1 << n;
        const int INF = INT_MAX / 20;
        vector<vector<pair<int, long long>>> dp(N, vector<pair<int,long long>>(n, make_pair(INF, 0)));

        for (int i = 0; i < n; ++i) {
            dp[1 << i][i] = make_pair(A[i].length(), i + 1);
        }
        
        for (int i = 1; i < N; ++i) {
            for (int j = 0; j < n; ++j) {
                auto p = dp[i][j];
                int len = p.first;
                long long path = p.second;
                
                for (int k = 0; k < n; ++k) {
                    if (i & (1 << k)) {
                        continue;
                    }
                    int ni = i | (1 << k);
                    int nlen = len + padding[j][k];
                    if (dp[ni][k].first > nlen) {
                        dp[ni][k].first = nlen;
                        dp[ni][k].second = path << 4 | (k + 1);
                    }
                }
            }
        }
        // cout << "build " << endl;
        int k = -1; 
        for (int i = 0; i < n; ++i) {
            if (k == -1 || dp[N - 1][k].first > dp[N - 1][i].first) {
                k = i;
            }
        }
        
        vector<int> path(n);
        long long code = dp[N - 1][k].second;
        for (int i = 0; i < n; ++i, code >>= 4) {
            path[n - 1 - i] = code & 0xF;
            --path[n - 1 - i];
        }
        
        string res;
        res = A[path[0]];
        for (int i = 1; i < n; ++i) {
            int len = padding[path[i - 1]][path[i]];
            if (len > 0) {
                int beg = A[path[i]].length() - len;
                res += A[path[i]].substr(beg);
            }
        }
        return res;
    }
};
