class Solution {
public:
  int graph[12][12];
  string D[1 << 12][12];
  int longestMatch(string &s1, string &s2) {
    for (int i = 1; i < s1.size(); i++) {
      bool allMatch = true;
      for (int j = 0; i + j < (int)s1.size() && j < s2.size(); j++) {
        if (s1[i + j] != s2[j]) {
          allMatch = false;
          break;
        }
      }
      if (allMatch) return (int)s1.size() - i;
    }
    return 0;
  }
  
  string shortestSuperstring(vector<string>& A) {
    int N = (int)A.size();
    for (int i = 0; i < N; i++) {
      for (int j = 0; j < N; j++) {
        if (i == j) continue;
        graph[i][j] = longestMatch(A[i], A[j]);
      }
    }
    for (int i = 0; i < (1 << N); i++) {
      for (int j = 0; j < N; j++) {
        if (!(i & (1 << j))) continue;
        int k = i ^ (1 << j);
        if (k == 0) {
          D[i][j] = A[j];
          continue;
        }
        int minD = 1 << 30;
        string res = "";
        for (int p = 0; p < N; p++) {
          if (k & (1 << p)) {
            int d = (int)D[k][p].size() + (int)A[j].size() - graph[p][j];
            if (d < minD) {
              minD = d;
              res = D[k][p] + A[j].substr(graph[p][j]);
            }
          }
        }
        D[i][j] = res;
      }
    }
    int minD = 1 << 30;
    string res = "";
    for (int i = 0; i < N; i++) {
      if (D[(1 << N) - 1][i].size() < minD) {
        minD = (int)D[(1 << N) - 1][i].size();
        res = D[(1 << N) - 1][i];
      }
    }
    return res;
  }
};
