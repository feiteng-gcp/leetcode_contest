class Solution {
public:
    void DFS(vector<vector<int>> &cost, vector<string> &A, int index, vector<bool> &visited, string &ret, vector<int> &solve, int curLen) {
        visited[index] = true;
        solve.push_back(index);
    //    printf("visit %d %s %s\n", index, A[index].c_str(), cur.c_str());
        if (solve.size() == A.size()) {
            if (ret.empty() || ret.size() > curLen) {
                ret = A[solve[0]];
                for (int i = 1; i < solve.size(); ++i) 
                    ret += A[solve[i]].substr(A[solve[i]].size() - cost[solve[i - 1]][solve[i]]);
//             printf("update %s\n", ret.c_str());
            }
            visited[index] = false;
            solve.pop_back();
            return;
        }
        
        for (int i = 0; i < A.size(); ++i) {
            if (i == index || visited[i]) continue;
            if (!ret.empty() && curLen + cost[index][i] > ret.size()) continue;
            DFS(cost, A, i, visited, ret, solve, curLen + cost[index][i]);
        }
        visited[index] = false;
        solve.pop_back();
    }
    
    string shortestSuperstring(vector<string>& A) {
    //    A={"catg","ctaagt","gcta","ttca","atgcatc"};
    //    A={"phuutlgczfspygaljkv","fspygaljkvahvuii","csywjodtnkynkjckq","poyykqyrhbvcwvjl","xijupvzzwphuutlg","aljkvahvuiivqbqrw","vahvuiivqbqrwryd","wjodtnkynkjckqurgu","ecdmbshotqbxjqgbou","hvuiivqbqrwrydgnr","ivqbqrwrydgnrubcsywj","wphuutlgczfspyga"};

        vector<vector<int>> cost(A.size(), vector<int>(A.size(), 0));
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < A.size(); ++j) {
                if (i == j) continue;
                for (int k = min(A[i].size(), A[j].size()); k > 0; --k) {
                    if (A[j].substr(0, k) == A[i].substr(A[i].size() - k)) {
                        cost[i][j] = A[j].size() - k;
                        break;
                    }
                }
                if (cost[i][j] == 0) cost[i][j] = A[j].size();
       //         printf("cost %d %s %d %s %d\n", i, A[i].c_str(), j, A[j].c_str(), cost[i][j]);
            }
        }
        
        string ret = "";
        
        for (int i = 0; i < A.size(); ++i) {
       //     printf("start at %d %s\n", i, A[i].c_str());
            vector<bool> visited(A.size(), false);
            vector<int> solve;
            DFS(cost, A, i, visited, ret, solve, A[i].size());
        }
        
        return ret;
    }
};
