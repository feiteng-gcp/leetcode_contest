class Solution {
public:
    string _merge(const string &a, const string &b) {
        int lena = a.length(), lenb = b.length();
        int l = min(lena, lenb);
        for (int i = l; i >= 1; --i) {
            //cout << a.substr(lena - i, i) << ' ' << b.substr(0, i) << endl;
            if (a.substr(lena - i, i) == b.substr(0, i)) 
                return a + b.substr(i, lenb - i);
        }
        return a + b;
    }
    string merge(const string &a, const string &b) {
        string r1 = _merge(a, b);
        string r2 = _merge(b, a);
        //cout << r1 << ' ' << r2 << endl;
        if (r1.length() < r2.length()) return r1;
        return r2;
    }
    string shortestSuperstring(vector<string>& a) {
        string dp[1 << 13];
        bool flag[1 << 13] = {false};
        int n = a.size();
        //string x = string("aa"), y = string("ab");
        //cout << merge(x, y) << endl;
        //return "";
        for (int i = 0; i < n; ++i) dp[1 << i] = a[i];
        int mask = 1 << n;
        for (int i = 1; i < mask; ++i) {
            for (int j = 0; j < n; ++j) if (!(i >> j & 1)) {
                string ns = merge(dp[i], a[j]);
                int ni = i | (1 << j);
                if (!flag[ni] || dp[ni].length() > ns.length()) {
                    dp[ni] = ns;
                    flag[ni] = true;
                }
            }
        }
        return dp[mask - 1];
    }
};
