int b[15][15];
int dp[(1 << 12) + 5][25];
string S[(1 << 12) + 5][25];
class Solution {
public:
    
    string shortestSuperstring(vector<string>& A) {
        memset(b, 0, sizeof(b));
        for(int i = 0;i < A.size();i++){
            for(int j = 0;j < A.size();j++){
                if(i == j) continue;
                string s1 = A[i], s2 = A[j];
                for(int l1 = 1;l1 <= min(s1.length(), s2.length());l1++){
                    string sub1 = s1.substr(s1.length() - l1, l1);
                    string sub2 = s2.substr(0, l1);
                    if(sub1 == sub2)
                        b[i][j] = l1;
                }
            }
        }
        int ans = 0x3f3f3f3f;
        string ret = "";
        for(int i = 0;i < (1 << 12) + 5;i++){
            for(int j = 0;j < 25;j++){
                S[i][j] = "";
            }
        }
        memset(dp, ans, sizeof(dp));
        memset(dp[0], 0, sizeof(dp[0]));
        for(int i = 0;i < (1 << A.size());i++){
            for(int j = 0;j < A.size();j++){
                if(i == (1 << A.size()) - 1){
                    if(dp[i][j] < ans){
                        ans = dp[i][j];
                        ret = S[i][j];
                    }
                }else{
                    for(int k = 0;k < A.size();k++){
                        if(i & (1 << k)) continue;
                        int state = i | (1 << k);
                        int len;
                        if(i == 0){
                            len = dp[i][j] + A[k].length();
                            if(len < dp[state][k]){
                                dp[state][k] = len;
                                S[state][k] = S[i][j] + A[k];
                            }
                        }else{
                            len = dp[i][j] + A[k].length() - b[j][k];
                            if(len < dp[state][k]){
                                dp[state][k] = len;
                                S[state][k] = S[i][j] + A[k].substr(b[j][k], A[k].length() - b[j][k]);
                            }
                        }
                    }
                }
            }
        }
        return ret;
    }
};
