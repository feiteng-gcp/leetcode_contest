#define SZ(x) ((int)x.size())
#define all(a) a.begin(),a.end()
#define allr(a) a.rbegin(),a.rend()
#define clrall(name,val) memset(name,(val),sizeof(name));
#define EPS 10e-9
#define ll long long
#define ull long long unsigned
#define SF scanf
#define PF printf
#define psb(b) push_back((b))
#define ppb() pop_back()
#define oo (1<<28)
#define mp make_pair
#define fs first
#define sc second
#define rep(var,s,n,it) for(var=(s);var<(n);(var)+=it)
#define rev(var,s,n,it) for(var=(n-1);var>(s-1);(var)-=it)
#define Read freopen("in.txt","r",stdin)
#define Write freopen("out.txt","w",stdout)
#define __ std::ios_base::sync_with_stdio (false)

class Solution {
    string dna[20];
    int n,lim;
    int par[105];
    char str[105];
    int splen[20][20];
    int strLen[20];

    int dp[20][(1<<15)+5];
    int vp[20][(1<<15)+5];
    int isSubStr[20][(1<<15)+5];
    int substrmask[20];
    int cas=1;

    int rec(int pre,int mask)
    {
        if(mask==lim) return 0;
        int &ret = dp[pre][mask];
        int &vet = vp[pre][mask];
        if(vet==cas) return ret;
        vet=cas;
        ret=oo;
        int add,npre;
        for(int i=0;i<n;i++)
        {
            if((mask>>i)&1) continue;
            add=strLen[i]-splen[pre][i];
            npre=i;
            if(substrmask[i]&mask) add=0,npre=pre;
            ret=min(ret,add+rec(npre,mask|(1<<i)));
        }
        return ret;
    }

    string ans,tmpstr,minstr;

    void pathrec(int pre,int mask)
    {
        if(mask==lim) return ;
        int ret=rec(pre,mask);
        int add,tmp,id=-1,npre,ii=-1;
        minstr="";
        for(int i=0;i<n;i++)
        {
            if((mask>>i)&1) continue;
            add=strLen[i]-splen[pre][i];
            npre=i;
            if(substrmask[i]&mask) add=0,npre=pre;
            tmp=add+rec(npre,mask|(1<<i));
            tmpstr="";
            if(ret==tmp)
            {
                if(add)
                {
                    tmpstr=dna[i].substr(splen[pre][i]);
                }
                if(minstr=="" || tmpstr<minstr)
                {
                    minstr=tmpstr;
                    id=npre;
                    ii=i;
                }
            }
        }
        if(id!=-1)
        {
            ans+=minstr;
            pathrec(id,mask|(1<<ii));
        }
        return ;
    }

    int kmp(string a,string b)
    {
        int la=SZ(a);
        int lb=SZ(b);
        int k=0;
        par[0]=0;
        for(int i=1;i<lb;i++)
        {
            while(k>0 && b[i] != b[k]) k = par[k-1];
            if(b[i] == b[k]) k++;
            par[i] = k;
        }
        k=0;
        for(int i=0;i<la;i++)
        {
            while(k>0&&a[i]!=b[k]) k=par[k-1];
            if(a[i]==b[k]) k++;
            if(k==lb) return k;
        }
        return k;
    }

    void suffixPrefixMatch()
    {
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(i==j) continue;
                splen[i][j]=kmp(dna[i],dna[j]);
            }
        }
        return ;
    }

    void preProcess(int T)
    {
        ///Optimization Courtesy "Sudipto Chandra Dipu"
        clrall(substrmask,0);
        for(int j=0;j<n;j++)
        {
            for(int k=0;k<n;k++)
            {
                if(splen[k][j]==strLen[j])
                {
                    substrmask[j]|=(1<<k);
                }
            }
        }

        return ;
    }
public:
    string shortestSuperstring(vector<string>& A) {
        n = A.size();
        strLen[n]=0;
        for(int i=0;i<n;i++)
        {
            strLen[i]=A[i].size();
            dna[i]=A[i];
        }
        clrall(splen,0);
        suffixPrefixMatch();
        lim=(1<<n)-1;
        ans.clear();
        preProcess(lim+1);
        pathrec(n,0);
        return ans.c_str();
    }
};
