class Solution {
public:
    string pre,end;
    int length;
    string tmp;
    int max_len(string A,string B){
        int result = 0;
        if(A.length() > B.length()){
            tmp = A;
            A= B;B = tmp;
        }
        if(B.find(A) != string::npos)
            return A.length();
        length = 1;
        while(length < A.length()){
            pre = A.substr(0,length);
            end = A.substr(A.length() -length);
            if(B.substr(B.length()-length) == pre){
                tmp = B + A.substr(length);
                result = length;
            }
            if(B.substr(0,length) == end){
                tmp = A+B.substr(length);
                result = length;
            }
            ++length ;
        }
        return result;
    }
    

    vector<string> merge(vector<string> A){
        // vector<string> res;
        int max = 0;
        int index1,index2;
        string merge_str;
        for(int i=0;i<A.size()-1;++i){
            for(int j=i+1;j<A.size();++j){
                int tmplen = max_len(A[i],A[j]);
                if(tmplen>max){
                    index1 = i;
                    index2 = j;
                    max = tmplen;
                    merge_str = tmp;
                }
            }
        }
        // print(A);
        // cout<<max<<" "<<merge_str<<endl;
        if(max == 0){
            index1 = 0;
            index2 = 1;
            merge_str = A[0]+A[1];
        }
        vector<string>::iterator it = A.begin();
        A.erase(it+index1);
        it = A.begin();A.erase(it+index2-1);
        A.push_back(merge_str);
        return A;
    }
    
    string shortestSuperstring(vector<string>& A) {
        if(A.size() == 1)
            return A[0];
        vector<string> res=merge(A);
        while(res.size()!=1){
            // cout<<res.size()<<endl;
            res = merge(res);
        }
        return res[0];
    }
    
    void print(vector<string>& A){
        for(int i=0;i<A.size();++i)
            cout<<A[i]<<" ";
        cout<<endl;
    }
    
};
