    int sub[15][15];
    int dp[21][(1<<20)+5];
int pre[21][(1<<20) + 5];
class Solution {
public:

    int common(string a, string b)
    {
        int n = a.length();
        int m = b.length();
        int mi = min(n,m);
        int ans;
        for(ans = mi; ans >= 1; ans--)
        {
            int valid = 1;
            int i;
            for(i=0; i<ans; i++)
            {
                if(a[n-ans+i]!=b[i]){
                    valid = 0;
                    break;
                }
            }
            if(valid)break;
        }
        return ans;
    }
    string shortestSuperstring(vector<string>& a) {
        int n = a.size();
        int i,j;
        for(i=0; i<n; i++)
        {
            for(j=0; j<n; j++)
            {
                
                sub[i][j] = a[j].length() - common(a[i],a[j]);
            }
        }
        int tot = 1<<n;
        int mask;
        for(i=0;i<n;i++)
        {
           for(mask = 0; mask <=tot; mask++)
                dp[i][mask] = 100000007;
            
            dp[i][1<<i] = a[i].length();
            pre[i][1<<i] = -1;
        }
        int k;
       // printf("mask is %d\n",tot);
         for(mask=0; mask<tot; mask++)
                {
        for(k=0;k<n;k++)
        {
            for(i=0;i<n;i++)
            {
                if(i==k)continue;
               
                    if((mask & (1<<i))==0)continue;
                    if((mask & (1<<k)) ==0)
                    {
                        // kth string is not visited so we make it end string after ith string
                        int nex = mask | (1<<k);
                        if(dp[k][nex] > dp[i][mask]  + sub[i][k]){
                        dp[k][nex] = dp[i][mask]  + sub[i][k];
                        pre[k][nex] = i;
                        }
                        //printf("ending at %d mask %d is %d\n",k,nex,dp[k][nex]);
                    }
                }
            }
        }
        int res = 1000000007;
        int st = -1;
        for(i=0; i<n; i++)
        {
            if(dp[i][tot-1] < res)
            {
                st = i;
                res = dp[i][tot-1];
            }
        }
        vector<int> v;
        v.push_back(st);
        mask = tot - 1;
        while(1)
        {
            int nex = mask - (1<<st);
            st = pre[st][mask];
            
            if(st==-1)break;
            mask = nex;
            v.push_back(st);
        }
        reverse(v.begin(),v.end());
         string myres = a[v[0]];
        cout<<a[v[0]];
        for(i=1; i<n; i++)
        {
            cout<<" ";
            j = v[i];
            k = v[i-1];
            int m  = a[j].length();
            myres += a[j].substr(m-sub[k][j]);
        }
        cout<<endl;
        return myres;
    }
};
