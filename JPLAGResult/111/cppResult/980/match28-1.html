#include <algorithm>
#include <cassert>
#include <cstring>
#include <iostream>
#include <vector>
using namespace std;

int shortest_add(string A, string B) {
    int m = B.size();

    for (int add = 0; add <= m; add++) {
        string combined = A + B.substr(m - add);

        if ((int) combined.size() >= m && combined.substr(combined.size() - m) == B)
            return add;
    }

    assert(false);
}

const int N_MAX = 12;
const int MASK_MAX = 1 << N_MAX;
const int INF = 1e9 + 5;

int dp[MASK_MAX][N_MAX];
int dp_prev[MASK_MAX][N_MAX];

class Solution {
public:
    string shortestSuperstring(vector<string>& A) {
        int n = A.size();
        vector<vector<int>> shortest_adds(n, vector<int>(n, INF));

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                shortest_adds[i][j] = shortest_add(A[i], A[j]);

        for (int i = 0; i < n; i++)
            shortest_adds[i][i] = A[i].size();

        // for (int i = 0; i < n; i++)
        //     for (int j = 0; j < n; j++)
        //         cerr << shortest_adds[i][j] << (j < n - 1 ? ' ' : '\n');

        memset(dp, 63, sizeof(dp));


        for (int i = 0; i < n; i++) {
            dp[1 << i][i] = A[i].size();
            dp_prev[1 << i][i] = i;
        }

        for (int mask = 0; mask < 1 << n; mask++)
            for (int i = 0; i < n; i++)
                for (int j = 0; j < n; j++)
                    if (dp[mask][i] + shortest_adds[i][j] < dp[mask | 1 << j][j]) {
                        dp[mask | 1 << j][j] = dp[mask][i] + shortest_adds[i][j];
                        dp_prev[mask | 1 << j][j] = i;
                        // cerr << mask << ' ' << i << ' ' << dp[mask][i] << " -> " << (mask | 1 << j) << ' ' << j << ' ' << dp[mask | 1 << j][j] << endl;
                    }

        int best = INF;
        int best_mask = (1 << n) - 1, best_index = -1;

        for (int i = 0; i < n; i++)
            if (dp[best_mask][i] < best) {
                best = dp[best_mask][i];
                best_index = i;
            }

        vector<string> parts;

        while (best_mask != 0) {
            int p = dp_prev[best_mask][best_index];
            int length = shortest_adds[p][best_index];
            // cerr << best << ' ' << best_mask << ' ' << best_index << ' ' << p << endl;
            parts.push_back(A[best_index].substr(A[best_index].size() - length));
            best_mask ^= 1 << best_index;
            best_index = p;
        }

        reverse(parts.begin(), parts.end());
        return accumulate(parts.begin(), parts.end(), string());
    }
};
