class Solution {
public:
    vector<vector<int> > g;
    vector<int> vis;
    vector<int> nans,ans;
    int n,minl;
    void dfs(int dep,int l)
    {
        if(l>=minl) return;
        if(dep==n)
        {
            minl=l;
            ans=nans;
            return;
        }
        for(int i=0;i<n;i++)
            if(!vis[i])
            {
                vis[i]=1;
                nans[dep]=i;
                if(!dep)
                    dfs(dep+1,g[i][i]);
                else
                    dfs(dep+1,l+g[i][i]-g[nans[dep-1]][i]);
                vis[i]=0;
            }
    }
    int maxl(string a,string b)
    {
        int la=a.length();
        int lb=b.length();
        int ans=0;
        for(int i=max(1,la-lb+1);i<la;i++)
        {
            int j=0;
            for(j=0;i+j<la&&j<lb;j++)
                if(a[i+j]!=b[j]) break;
            if(i+j==la) return la-i;
        }
        return 0;
    }
    string shortestSuperstring(vector<string>& A) {
        n=A.size();
        if(n==1) return A[0];
        g.assign(n,vector<int>(n,0));
        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++)
            {
                if(i==j) g[i][j]=A[i].length();
                else g[i][j]=maxl(A[i],A[j]);
            }
        minl=100000;
        vis.assign(n,0);
        nans.assign(n,0);
        dfs(0,0);
        string s=A[ans[0]];
        for(int i=1;i<n;i++)
        {
            for(int j=g[ans[i-1]][ans[i]];j<g[ans[i]][ans[i]];j++)
                s+=A[ans[i]][j];
        }
        return s;
    }
};
