class Solution {
public:
    int N_;
    int cost[50][50] = {0};
    int f[4100][15];
    int dp(int now, int last) {
        int& ret = f[now][last];
        if (now + 1 == (1 << N_)) {
            ret = 0;
            return ret;
        }
        if (ret == -1) {
            ret = 0;
            for (int i = 0; i < N_; ++i) {
                if ((now & (1 << i)) == 0) {
                    ret = max(ret, dp(now ^ (1 << i), i) + (now == 0 ? 0 : cost[i][last]));
                }
            }
        }
        return ret;
    }
    string shortestSuperstring(vector<string>& A) {
        N_ = A.size();
        memset(f, -1, sizeof(f));
        for (int i = 0; i < N_; ++i) {
            for (int j = 0; j < N_; ++j) {
                if (i == j) {
                    continue;
                }
                cost[i][j] = 0;
                for (int k = min(A[i].length(), A[j].length()); k > 0; --k) {
                    if (A[j].substr(A[j].length() - k, k) == A[i].substr(0, k)) {
                        cost[i][j] = k;
                        break;
                    }
                }
                //cout << "cost " << i << ", " << j << ": " << cost[i][j] << endl;
            }
        }
        dp(0, 0);
        int now = 0;
        int last = 0;
        string ans = "";
        while (now + 1 < (1 << N_)) {
            //cout << now << " " << last << endl;
            for (int i = 0; i < N_; ++i) {
                if ((now & (1 << i))) {
                    continue;
                }
                int next = now ^ (1 << i);
                int c = (now == 0) ? 0 : cost[i][last];
                /*cout << "f: " << f[now][last] << endl;
                cout << "f[][] " << f[next][i] << " " << i << endl;
                cout << "c " << c << endl;*/
                if (f[now][last] == f[next][i] + c) {
                    for (int j = c; j < A[i].length(); ++j) {
                        ans += A[i][j];
                    }
                    now = next;
                    last = i;
                    break;
                }
            }
        }
        return ans;
    }
};
