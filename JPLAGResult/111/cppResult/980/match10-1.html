class Solution {
public:
    int dp[1<<12][12], match_len[12][12];
    vector <string> S;
    int call(int mask, int last){
        if(mask == (1<<S.size())-1)
            return 0;
        int &ret = dp[mask][last];
        if(~ret) return ret;
        ret = 1<<30;
        for(int i = 0; i < S.size(); i++){
            if(mask & (1<<i))
                continue;
            ret = min(ret, call(mask|(1<<i), i)+int(S[i].size())-match_len[last][i]);
        }
        return ret;
    }
    
    string get(int mask, int last){
        if(mask == (1<<S.size())-1)
            return "";
        int &ret = dp[mask][last];
        for(int i = 0; i < S.size(); i++){
            if(mask & (1<<i))
                continue;
            if(ret == call(mask|(1<<i), i)+int(S[i].size())-match_len[last][i]){
                return S[i].substr(match_len[last][i])+get(mask|(1<<i), i);
            }
        }
        return "";
    }
    string shortestSuperstring(vector<string>& A) {
        memset(dp, -1, sizeof dp);
        int mn_val = 1<<30, mn_idx = -1;
        sort(A.begin(), A.end());
        for(int i = 0; i < A.size()-1; i++){
            if(A[i] == A[i+1].substr(0, A[i].size()))
                continue;
            S.push_back(A[i]);

        }
        S.push_back(A[A.size()-1]);
        for(int i = 0; i < S.size(); i++){
            for(int j = 0; j < S.size(); j++){
                if(i == j) continue;
                for(int k = 0; k < S[i].size(); k++){
                    if(S[i].size()-k > S[j].size())
                        continue;
                    if(S[i].substr(k) == S[j].substr(0, S[i].size()-k)){
                        match_len[i][j] = max(match_len[i][j], int(S[i].size())-k);
                    }
                }
            }
        }
        for(int i = 0; i < S.size(); i++){
            if(call(1<<i, i)+int(S[i].size()) < mn_val){
                mn_val = call(1<<i, i)+int(S[i].size());
                mn_idx = i;
            }
        }
        cout << mn_val << ' ' << mn_idx << endl;
        string s = S[mn_idx]+get(1<<mn_idx, mn_idx);
        return s;
    }
};
