#include <algorithm>
#include <cassert>
#include <iostream>
#include <vector>
using namespace std;


class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int R = A.size(), C = A[0].size();
        int total = 0;

        for (int c = 0; c < C; c++) {
            bool sorted = true;

            for (int r = 0; r + 1 < R; r++)
                if (A[r][c] > A[r + 1][c])
                    sorted = false;

            if (!sorted)
                total++;
        }

        return total;
    }
};
