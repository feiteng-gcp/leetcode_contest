class Solution {
public:
    int minDeletionSize(vector<string>& a) {
        int as=a.size();
        if(as == 1)
            return 0;
        int ret=0;
        for(int i=0;i<a[0].size();i++)
        {
            if(!is(a,i))
                ret++;
        }
        return ret;
    }
    bool is(vector<string>& a, int c)
    {
        for(int i=1;i<a.size();i++)
        {
            if(a[i][c]<a[i-1][c])
                return false;
        }
        return true;
    }
};
