class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int row = A.size();
        int col = A[0].size();
        int num = 0;
        for(int i=0;i<col;i++){
            bool flag = true;
            for(int j=0;j<row-1;j++){
                if(A[j][i]>A[j+1][i]){
                    flag = false;break;
                }
            }
            if(flag)num++;
        }
        return col-num;
    }
};
