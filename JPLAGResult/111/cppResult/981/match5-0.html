class Solution {
public:
    int minDeletionSize(vector<string>& A) {
     //   A={"a","b"};
     //   A={"zyx","wvu","tsr"};
        int n = 0;
        for (int i = 0; i < A[0].size(); ++i) {
            bool isSorted = true;
            for (int j = 1; j < A.size(); ++j) {
                if (A[j][i] < A[j - 1][i]) {
                    isSorted = false;
                    break;
                }
            }
            if (!isSorted) ++n;
        }
        return n;
    }
};
