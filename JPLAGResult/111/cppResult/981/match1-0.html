class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int size = A.size();
        int col = A[0].size();
        int ans = 0;
      
        for (int i = 0; i < col; i += 1) {
            bool flag = true;
            for (int j = 0; j+1 < size; j += 1) {
                if (A[j][i] > A[j+1][i]) {
                    flag = false;
                    break;
                }
            }
            if (flag == false) ans += 1;
        }
        return ans;
    }
};
