class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size(), m = A[0].size();
        int ans = 0;
        for(int j=0;j<m;j++) {
            bool found = false;
            for(int i=1;i<n;i++) if(A[i][j] < A[i-1][j]) { found = true; break; }
            if(found) ans++;
        }
        return ans;
    }
};
