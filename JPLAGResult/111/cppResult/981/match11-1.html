class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int result = 0;
        for (int j = 0; j < A[0].size(); ++j) {
            for (int i = 1; i < A.size(); ++i) {
                if (A[i][j] < A[i - 1][j]) {
                    result++;
                    break;
                }
            }
        }
        return result;
    }
};
