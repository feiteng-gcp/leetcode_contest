class Solution {
public:
    int minDeletionSize(vector<string>& a) {
        int n = a.size(), m = a[0].size();
        int res = 0;
        for (int c = 0; c < m; ++c) {
            bool ok = true;
            for (int r = 1; r < n; ++r) if (a[r][c] < a[r-1][c])
                ok = false;
            if (!ok) ++res;
        }
        return res;
    }
};
