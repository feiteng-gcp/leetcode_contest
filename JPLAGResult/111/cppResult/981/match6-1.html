class Solution {
public:
    int minDeletionSize(vector<string>& a) {
        int res = 0;
        int len = a[0].length();
        int n = a.size();
        for (int i = 0; i < len; ++i) {
            bool ok = true;
            for (int j = 1; j < n; ++j) {
                if (a[j][i] < a[j - 1][i]) {
                    ok = false;
                    break;
                }
            }
            if (!ok) ++res;
        }
        return res;
    }
};
