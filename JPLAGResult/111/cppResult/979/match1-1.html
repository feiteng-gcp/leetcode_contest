class Solution {
public:
    vector<int> diStringMatch(string S) {
        int N = S.size()+1;
        vector<int> tmp(N, 0);
        int inc = 0, dec = 0;
        for (int i=0; i<N-1; ++i){
            if (S[i] == 'I') {
                tmp[1+i] = ++inc;
            } else {
                tmp[1+i] = --dec;
            }
        }
        for (int i=0; i<N; ++i) tmp[i] -= dec;
        return tmp;
    }
};
