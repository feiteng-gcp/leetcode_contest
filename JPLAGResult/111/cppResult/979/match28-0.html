class Solution {
public:
    vector<int> diStringMatch(string S) {
 //       S="III";
  //      S="DDI";
        vector<int> ret(S.size() + 1, 0);
        int minN = 0, maxN = 0;
        for (int i = 0; i < S.size(); ++i) {
            if (S[i] == 'I') {
                ret[i + 1] = ++maxN;
            } else {
                ret[i + 1] = --minN;
            }
        }
        
        for (int i = 0; i < ret.size(); ++i) {
            ret[i] -= minN;
        }
        
        return ret;
    }
};
