class Solution {
public:
    vector<int> diStringMatch(string S) {
        int n=S.size(), left=0, right=n;
        vector<int> res(n+1,0);
        for(int i=n-1;i>=0;--i){
            if(S[i]=='D') res[i+1]=left++;
            else res[i+1]=right--;
        }
        res[0]=right;
        return res;
    }
};
