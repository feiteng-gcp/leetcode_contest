class Solution {
public:
    vector<int> diStringMatch(string S) {
        int l = S.length();
        int n = l+1;
        vector<int> res(n,-1);
        int cur = 0;
        
        for(int i = 0;i<l;i++)
        {
            if(S[i]=='I')
            {
                res[i] = cur;
                cur++;
            }
        }
        
        for(int i = l;i>=0;i--)
        {
            if(res[i]==-1)
            {
                res[i] = cur;
                cur++;
            }
        }
        
        return res;
    }
};
