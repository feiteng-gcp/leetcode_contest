class Solution {
public:
    vector<int> diStringMatch(string S) {
        int n_I = 0, n_D = 0;
        for(int i = 0; i < S.size(); i++)
        {
            if (S[i] == 'I')
                n_I++;
            else
                n_D++;
        }

        vector<int> result;
        int pointer_D = n_D - 1;
        int pointer_I = n_D + 1;
        result.push_back(n_D);
        for(int i = 0; i < S.size(); i++)
        {
            if (S[i] == 'I')
            {
                result.push_back(pointer_I);
                pointer_I++;
            }
            else
            {
                result.push_back(pointer_D);
                pointer_D--;
            }
        }
        return result;
    }
};
