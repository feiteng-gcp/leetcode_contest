class Solution {
public:
    vector < int > A;
    vector<int> diStringMatch(string S) {
        A.push_back(0);
        
        int up = 1;
        int down = -1;
        for(int i = 0;i < S.size();++i) {
            if(S[i] == 'I') {
                A.push_back(up++);
            } else {
                A.push_back(down--);
            }
        }
        
        for(int i = 0;i < A.size();++i) {
            A[i] += -(down + 1);
        }
        return A;
    }
};
