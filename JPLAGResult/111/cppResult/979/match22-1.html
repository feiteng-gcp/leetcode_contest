class Solution {
public:
    vector<int> diStringMatch(string s) {
        int n = s.length();
        int high = n, low = 0;
        vector<int> res;
        for (int i = n - 1; i >= 0; --i) {
            if (s[i] == 'I') {
                res.push_back(high--);
            } else {
                res.push_back(low++);
            }
        }
        res.push_back(high);
        reverse(res.begin(), res.end());

        return res;
    }
};
