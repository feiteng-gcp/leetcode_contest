class Solution {
public:
    vector<int> diStringMatch(string S) {
        int i = 0;
        int j = S.length();
        
        vector<int> ret;
        for(int k = 0; k<S.length(); k++){
            if(S.at(k) == 'I')
            {
                ret.push_back(i);
                i++;
            }else{
                ret.push_back(j);
                j--;
            }
        }
        
        ret.push_back(i);
        return ret;
    }
};
