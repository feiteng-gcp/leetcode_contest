class Solution {
public:
    vector<int> diStringMatch(string S) {
        int n = S.size();
        vector<int> res(n+1);
        res[n] = 0;

        int low = 0, high = 0;
        for (int i = n-1; i >= 0; i--) {
            if (S[i] == 'I') {
                res[i] = --low;
            } else {
                res[i] = ++high;
            }
        }
        for (int i = 0; i <= n; i++) {
            res[i] = res[i] - low;
        }
        return res;
    }
};
