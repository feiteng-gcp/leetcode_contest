class Solution {
public:
    vector<int> diStringMatch(string S) {
        

        int low = 0;
        int high = S.size();
        vector<int> result;
        
        for(int j=0;j<S.size();j++)
        {
            if(S[j]=='I')
            {
                result.push_back(low);
                low++;
            }
            else
            {
                result.push_back(high);
                high--;
            }
        }
        result.push_back(low);
        return result;
    }
};
