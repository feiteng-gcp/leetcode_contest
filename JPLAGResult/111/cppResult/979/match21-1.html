class Solution {
public:
    vector<int> diStringMatch(string S) {
        vector<int> res(1 + S.size());
        int inc = 0;
        int dec = S.size();
        for (int i = 0; i < S.size(); ++i) {
            if (S[i] == 'D') {
                res[i] = dec--;
            }
            else {
                res[i] = inc++;
            }
        }
        res[S.size()] = inc;
        return res;
    }
};
