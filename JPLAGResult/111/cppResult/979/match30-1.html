class Solution {
public:
    vector<int> diStringMatch(string S) {
        if (S.empty()) return vector<int>{};
        auto ans = vector<int>{0};

        int inc = 0;
        int dec = 0;
        for (int i = 0; i < S.length(); i++) {
            if (S[i] == 'I') {
                ans.push_back(++inc);
            } else {
                ans.push_back(--dec);
            }
        }
        for (int i = 0; i < ans.size(); i++) {
            ans[i] -= dec;
        }
        return ans;
    }
};
