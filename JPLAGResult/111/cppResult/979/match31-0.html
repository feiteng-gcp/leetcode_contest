class Solution {
public:
    vector<int> diStringMatch(string S) {
        
        vector<int> ans;
        
        int t1 = 0;
        int t2 = S.size();
        
        for (int i = 0; i < S.size(); i++) {
            if (S[i] == 'I') {
                ans.push_back(t1);
                t1++;
            }
            else {
                ans.push_back(t2);
                t2--;
            }
        }
        ans.push_back(t1);
        
        return ans;
        
    }
};
