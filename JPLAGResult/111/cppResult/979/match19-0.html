class Solution {
public:
    vector<int> diStringMatch(string S) {
      
      int len = S.length();
      int l = 0, r = len;
      vector <int> a;
      
      for (int i = 0;i < len; i ++)
      {
        if (S[i] == 'I') a.push_back(l ++);
          else a.push_back (r --);
      }
      a.push_back (l);
      return a;
        
    }
};
