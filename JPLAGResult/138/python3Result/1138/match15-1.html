class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        max_num_satisfied = 0
        for i, (customer, g) in enumerate(zip(customers, grumpy)):
            if i <= X - 1 or g == 0:
                max_num_satisfied += customer

        cur_num_satisfied 
= max_num_satisfied
        for rm_idx in range(len(customers) - X):
            if grumpy[rm_idx]:
                cur_num_satisfied -= customers[rm_idx]
            if grumpy[rm_idx + X]:
                cur_num_satisfied += customers[rm_idx + X]
            max_num_satisfied = max(cur_num_satisfied, max_num_satisfied)
        return max_num_satisfied
