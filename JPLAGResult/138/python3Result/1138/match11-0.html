class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        
        m = sum([customers[i] for i in range(len(customers)) if grumpy[i]==0])
        
        maxSave = sum([customers[i] for i in range(X) if grumpy[i]])
        lastSave = maxSave
        for i in range(1,len(customers)-X+1):
            if grumpy[i-1]:
                lastSave -= customers[i-1]
            if grumpy[i+X-1]:
                lastSave += customers[i+X-1]
            maxSave = max(lastSave, maxSave)
        #print(m, maxSave)
        return m + maxSave
