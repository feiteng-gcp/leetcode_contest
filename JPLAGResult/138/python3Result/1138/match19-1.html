class Solution:
    def maxSatisfied(self, customers, grumpy, X) -> int:
        N = len(customers)

        ret = 0
        for i in range(N):
            if grumpy[i] == 0:
                ret += customers[i]

        offset = 0
        
for i in range(X):
            if grumpy[i] == 1:
                offset += customers[i]

        max_offset = offset
        for i in range(X, N):
            if grumpy[i] == 1:
                offset += customers[i]
            if grumpy[i-X] == 1:
                offset -= customers[i-X]

            max_offset = max(max_offset, offset)


        return ret + max_offset
