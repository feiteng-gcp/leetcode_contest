class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], x: int) -> int:
        res = sum(c * (1- g) for c, g in zip(customers, grumpy))
        if x >= len(grumpy):
            return sum(customers)
        d 
= sum(customers[i] for i in range(x) if grumpy[i])
        mx = d
        for j in range(x, len(grumpy)):
            if grumpy[j]:
                d += customers[j]
            if grumpy[j - x]:
                d -= customers[j - x]
            mx = max(mx, d)
        return res + mx
