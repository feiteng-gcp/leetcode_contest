class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        
        res = 0
        for c, g in zip(customers, grumpy):
            if g == 0:
                res += c
                
        cur_incre 
= sum(customers[i] for i in range(X) if grumpy[i] == 1)
        max_incre = cur_incre
        
        for i in range(X, len(customers)):
            if grumpy[i-X] == 1: cur_incre -= customers[i-X]
            if grumpy[i] == 1: cur_incre += customers[i]
            max_incre = max(max_incre, cur_incre)
            
        return res + max_incre
