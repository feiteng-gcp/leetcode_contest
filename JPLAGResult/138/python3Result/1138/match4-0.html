class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        if not customers:
            return 0
        if X >= len(customers):
            return sum(customers)
        
        base = sum([customers[i] for i in range(len(customers)) if not grumpy[i]])
        win_additional = sum(customers[i] for i in range(X) if grumpy[i])
        max_win = win_additional
        
        for i in range(X, len(customers)):
            if grumpy[i-X]:
                win_additional -= customers[i-X]
            if grumpy[i]:
                win_additional += customers[i]
            max_win = max(max_win, win_additional)
        return base + max_win
