class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        total = 0
        n = len(customers)
        for i in range(n):
            if grumpy[i] == 0:
                total += customers[i]
        earn = 0
        max_earn = 0
        for i in range(n):
            j = i - X
            if j >= 0 and grumpy[j] == 1:
                earn -= customers[j]
            if grumpy[i] == 1:
                earn += customers[i]
            max_earn = max(max_earn, earn)
        return total + max_earn
