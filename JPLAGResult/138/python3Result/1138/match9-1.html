class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        res = 0
        L = len(customers)
        for i in range(L):
            if grumpy
[i] == 0:
                res += customers[i]
        summ = 0
        for i in range(X):
            if grumpy[i] == 1:
                summ += customers[i]
        maxx = summ
        for i in range(L - X):
            
if grumpy[i] == 1:
                summ -= customers[i]
            if grumpy[i + X] == 1:
                summ += customers[i + X]
            maxx = max(maxx, summ)
        return res + maxx
