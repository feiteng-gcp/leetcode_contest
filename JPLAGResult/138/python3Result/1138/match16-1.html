class Solution:
    def maxSatisfied(self, cus: List[int], grumpy: List[int], X: int) -> int:
        total = sum(cus[i] for i in range(len(cus)) if grumpy[i] == 0)
        curExtra = sum(cus[i] for i in range(X) if grumpy[i] == 1)
        ans = curExtra
        # print (curExtra)
        for i in range(len(cus)-X):
            curExtra = curExtra - cus[i]*grumpy[i] + cus[i+X]*grumpy[i+X]
            # print (i, i+X, curExtra)
            ans = max(ans, curExtra)
        ans += total
        return ans
