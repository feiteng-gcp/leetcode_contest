class Solution:
    def maxSatisfied(self, A: List[int], B: List[int], X: int) -> int:
        res = 0
        for i in range(len(A)):
            if B[i] == 1:
                A[i] = -A[i]
            else:
                res += A[i]
                
        temp = 0
        for i in range(X):
            if A[i] < 0:
                temp -= A[
i]
        ans = temp
        for i in range(len(A)-X):
            if A[i] < 0:
                temp += A[i]
            if A[i+X] < 0:
                temp += -A[i+X]
            
            ans = max(ans, temp)
        return ans + res
