class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        res = 0
        for num, isgrumpy in zip(customers, grumpy):
            if isgrumpy == 0:
                res += num
        cur = 0
        maxinc 
= 0
        #print(res)
        for i in range(len(grumpy)):
            if grumpy[i] == 1:
                cur += customers[i]
            if i-X >= 0 and grumpy[i-X] == 1:
                cur -= customers[i-X]
            maxinc = max(maxinc, cur)
            #print(cur)
        return res + maxinc
