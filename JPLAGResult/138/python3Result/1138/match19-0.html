class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        res = 0
        cur = 0
        n = len(customers)
        xx = 0
        
        for i in range(n):
            if grumpy[i] == 0:
                xx += customers[i]
                
        
        for i in range(X):
            if grumpy[i] == 1:
                cur += customers[i]
        res = cur

        for i in range(X, n):
            if grumpy[i] == 1:
                cur += customers[i]
            if grumpy[i-X] == 1:
                cur -= customers[i-X]
            res = max(cur, res)
        return res + xx
