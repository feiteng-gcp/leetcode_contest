class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        # corner case
        if X >= len(customers):
            return sum(customers)
        # calculate all non-grumpy elements
        total = 0
        for i in range(len
(customers)):
            if grumpy[i] == 0:
                total += customers[i]
        
        # use sliding window to detect maximum grumpy value
        # sliding window X
        windowValue = 0
        for i in range(X):
            if grumpy[i] == 1:
                windowValue += customers[i]
        maxVal = windowValue
        for i in range(X, len(customers)):
            
if grumpy[i] == 1:
                windowValue += customers[i]
            if grumpy[i - X] == 1:
                windowValue -= customers[i - X]
            maxVal = max(maxVal, windowValue)
        return maxVal + total
