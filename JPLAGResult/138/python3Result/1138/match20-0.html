class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        n = len(customers)
        s = 0
        for i in range(0, n):
            if not grumpy[i]:
                s += customers[i]
        
        actual = sum([customers[j] for j in range (X) if not grumpy[j]])
        window = sum([customers[j] for j in range (X)])
        maxResult = result = s - actual + window 
        # print(f's:{s}')
        # print(f"result: {result}")
        # print(f"actual:{actual}")
        # print(f"window:{window}")
        
        
        for i in range(X, n):
            if (grumpy[i-X]):
                result-=customers[i-X]                
            if (grumpy[i]):
                result+=customers[i]            
            print(f"result: {result}")
            maxResult = max(maxResult, result)
  
        return maxResult
