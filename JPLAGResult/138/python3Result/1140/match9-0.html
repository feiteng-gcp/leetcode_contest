class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        d = {}
        for code in barcodes:
            if code in d:
                d[code] += 1
            else:
                d[code] = 1
        a = sorted(d.items(), key=lambda x:x[1], reverse=True)
        group = [[], []]
        for key, value in a:
            if len(group[0]) <= len(group[1]):
                for _ in range(value):
                    group[0].append(key)
            else:
                for _ in range(value):
                    group[1].append(key)
        if len(group[0]) < len(group[1]):
            group[0], group[1] = group[1], group[0]
        n = len(barcodes)
        half = n // 2
        group[0], group[1] = group[0][:n - half], group[0][n - half:] + group[1]
        ans = []
        for i in range(n):
            if i % 2 == 0:
                ans.append(group[0][i // 2])
            else:
                ans.append(group[1][(i - 1) // 2])
        return ans
