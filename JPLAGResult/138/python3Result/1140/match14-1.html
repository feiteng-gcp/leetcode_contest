class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        counter = collections.Counter(barcodes)
        nums = sorted(barcodes, key=lambda x: (-counter[x], x))
        # print(nums)
        m = (len(nums) + 1) // 2;
        result = []
        for a, b in zip(nums[:m], nums[m:]):
            result
.append(a)
            result.append(b)
        if len(nums) % 2 == 1:
            result.append(nums[m - 1])
        return result
