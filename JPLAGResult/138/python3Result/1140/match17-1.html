class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        f = {}
        for i in barcodes:
            if i in f:
                f[i] += 1
            else:
                f[i] = 1
        fl = []
        for i in f:
            fl.append((i, f[i]))
        fl.sort(key = lambda x: (-x[1], x[0]))
        print(fl)
            
        result = [0] * len(barcodes)
        ctr = 0
        for num, freq in fl:

            while freq > 0:
                result[ctr] = num
                ctr += 2
                if ctr >= len(result):
                    ctr = 1
                freq -= 1
        return result
