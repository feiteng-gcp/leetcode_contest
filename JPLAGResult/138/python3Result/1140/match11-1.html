class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        import heapq        
        h = []
        counts = {}
        for barcode in barcodes:
            counts[barcode] = counts.get(barcode, 0) + 1
        
        for barcode, count in counts.items():
            heapq.heappush(h, (-count, barcode))
        result = []
        while len(h) > 1:
            c1, b1 = heapq.heappop(h)
            c2, b2 = heapq.heappop(h)
            result.append(b1)
            result.append(b2)
            if c2 < -1:
                heapq.heappush(h, (c2+1, b2))
            if c1 < -1:
                heapq.heappush
(h, (c1+1, b1))
                
        if h:
            result.append(h[0][1])
            
        return result
