class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        count = {i:0 for i in set(barcodes)}
        for i in barcodes:
            count[i] += 1
        
        count = list(count.items())
        
        count.sort(key=lambda x: x[1], reverse = True)
        
        ans = [-1] * len(barcodes)
        i = 0
        j = len(ans) - 1
        if j % 2 == 0:
            j -= 1
            
        flag = True
        
        for num, v in count:
            for _ in range(v):
                # print(i, j, flag)
                if i >= len(ans):
                    ans[j] = num
                    j -= 2
                elif j < 0:
                    ans[i] = num
                    i += 2
                elif flag:
                    ans[i] = num
                    i += 2
                else:
                    ans[j] = num
                    j -= 2
            flag = not flag
        return ans
