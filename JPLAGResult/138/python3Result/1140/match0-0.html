import heapq, collections

class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        freq = collections.Counter(barcodes)
        heap = [(-count, barcode) for barcode, count in freq.items()]
        heapq.heapify(heap)
        result = []
        while heap:
            count, barcode = heapq.heappop(heap)
            if result and result[-1] == barcode:
                next_count, next_barcode = heapq.heappop(heap)
                result.append(next_barcode)
                next_count += 1
                if next_count < 0:
                    heapq.heappush(heap, (next_count, next_barcode))
            else:
                result.append(barcode)
                count += 1
            if count < 0:
                heapq.heappush(heap, (count, barcode))
            
        return result
