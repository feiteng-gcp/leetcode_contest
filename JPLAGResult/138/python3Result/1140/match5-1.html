class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        dd = collections.Counter(barcodes)
        q = []
        res = []
        for k, v in dd.items():
            heapq.heappush(q, (-v, k))
        
        last = None
        while q or last:
            nv, k = heapq.heappop(q)
            res.append(k)
            if last:
                heapq.heappush(q, last)
            nn = nv + 1
            last = (nn, k) if nn != 0 else None
        return res
