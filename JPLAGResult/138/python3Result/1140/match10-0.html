from heapq import heapify, heappush, heappop
class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        
        counter = collections.Counter(barcodes)
        
        h = []
        for bar, count in counter.items():
            heappush( h, (-count, bar) )
            
        output = []
        while len(h) > 1:
            
            c1, b1 = heappop(h)
            c2, b2 = heappop(h)
            
            output.append(b1)
            output.append(b2)
            
            if c1 + 1 < 0:
                heappush(h, (c1+1, b1))
            if c2 + 1 < 0:
                heappush(h, (c2+1, b2))
        
        if h:
            output.append(h[0][1])
            
        return output
