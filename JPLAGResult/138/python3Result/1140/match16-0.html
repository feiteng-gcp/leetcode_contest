class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        vals = collections.defaultdict(int)
        for c in barcodes:
            vals[c] += 1
        q = []
        for val, cnt in vals.items():
            heapq.heappush(q, (-cnt, val))
        ret = []
        last_cnt, last_val = None, None
        while len(q):
            cnt, val = heapq.heappop(q)
            cnt = -cnt
            ret.append(val)
            if last_cnt is not None and last_cnt > 0:
                heapq.heappush(q, (-last_cnt, last_val))
            last_cnt, last_val = cnt-1, val
        assert len(ret) == len(barcodes)
        return ret
