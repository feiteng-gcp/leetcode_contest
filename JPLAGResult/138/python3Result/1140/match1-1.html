class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        counter = collections.Counter(barcodes)
        pq = []
        for k, cnt in counter.items():
            heapq.heappush(pq, (-cnt, k))
        res = []
        while pq:
            first = heapq.heappop(pq)
            if res and res[-1] == first[1]:
                if not pq:
                    return []
                second 
= heapq.heappop(pq)
                res.append(second[1])
                if -second[0] > 1:
                    heapq.heappush(pq, (second[0] + 1, second[1]))
                heapq.heappush(pq, first)
            else:
                res.append
(first[1])
                if -first[0] > 1:
                    heapq.heappush(pq, (first[0] + 1, first[1]))
        return res
