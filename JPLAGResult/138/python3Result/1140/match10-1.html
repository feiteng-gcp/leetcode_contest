from heapq import heappush, heappop
class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        cnt = collections.Counter(barcodes)
        ans, pq = [], []
        for k, v in cnt.items():
            heappush(pq, (-v, k))
        while pq:
            v, k = heappop(pq)
            if ans and k == ans[-1]:
                vv, kk = heappop(pq)
                ans.append(kk)
                if vv < -1:
                    heappush(pq, (vv + 1, kk))
                heappush(pq, (v, k))
            else:
                ans.append(k)
                if v < -1:
                    heappush(pq, (v + 1, k))
        return ans
