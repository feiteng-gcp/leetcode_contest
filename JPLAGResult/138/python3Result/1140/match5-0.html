class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        c = collections.Counter(barcodes)
        ans = []
        h = []
        for v, cnt in c.items():
            h.append((-cnt, v)) # min heap
        heapq.heapify(h)
        while len(ans) != len(barcodes):
                if len(h) == 1:
                    c, v = heapq.heappop(h)
                    ans.append(v)
                    break
                c1, v1 = heapq.heappop(h)
                c2, v2 = heapq.heappop(h)
                if ans:
                    if v1 != ans[-1]:
                        ans.append(v1)
                        ans.append(v2)
                    else:
                        ans.append(v2)
                        ans.append(v1)
                else:
                    ans.append(v1)
                    ans.append(v2)   
                if c1 != -1:
                    heapq.heappush(h, (c1+1, v1))
                if c2 != -1:
                    heapq.heappush(h, (c2+1, v2))                    
        return ans
