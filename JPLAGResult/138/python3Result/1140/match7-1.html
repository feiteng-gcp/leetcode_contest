class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        
        numCount = {}
        for n in barcodes:
            if n not in numCount:
                numCount[n] = 1
            else:
                numCount[n] += 1
        # print(numCount)
        heap = []
        for n, count in numCount.items():
            heapq.heappush(heap, (-count, n))
        result = []
        while len(heap) > 1:
            c1m, n1 = heapq.heappop(heap)
            c2m, n2 = heapq.heappop(heap)
            result.append(n1)
            result.append(n2)
            c1m += 1
            if c1m < 0:
                heapq.heappush(heap, (c1m, n1))
            c2m += 1
            if c2m < 0:
                heapq.heappush(heap, (c2m, n2))
        if len(heap) > 0:
            result.append(heapq.heappop(heap)[1])
        return result
