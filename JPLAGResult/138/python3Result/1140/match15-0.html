class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        count = collections.Counter(barcodes)
        pq = [[-count[k], k] for k in count]
        heapq.heapify(pq)
        L = len(barcodes)
        
        res = []
        last = None
        while len(res) < L:
            elem = heapq.heappop(pq)
            temp = None
            if elem[1] == last:
                temp = elem
                elem = heapq.heappop(pq)
            res.append(elem[1])
            last = elem[1]
            if -elem[0] > 1:
                heapq.heappush(pq, [elem[0] + 1, elem[1]])
            if temp:
                heapq.heappush(pq, temp)
        return res
