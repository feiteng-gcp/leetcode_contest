class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        counter = collections.Counter(barcodes)
        heap = [(-v,k) for k,v in counter.items()]
        heapq.heapify(heap)
        ans = []
        while len(ans) < len(barcodes):
            #print(heap, ans)
            v, k 
= heapq.heappop(heap)
            if ans and k == ans[-1]:
                
                v2, k2 = heapq.heappop(heap)
                ans.append(k2)
                v2 += 1
                if v2 < 0:
                    heapq.heappush(heap,(v2,k2))
                heapq.heappush(heap,(v,k))
                    
            else:
                ans.append(k)
                v += 1
                if v < 0:
                    heapq.heappush(heap,(v,k))
        return ans
