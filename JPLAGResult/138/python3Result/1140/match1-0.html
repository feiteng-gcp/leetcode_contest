class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        h = []
        count = collections.Counter(barcodes)
        ret = []
        for k,v in count.items():
            heapq.heappush(h,(-v,k))
        while len(h) > 0:
            top = heapq.heappop(h)
            if len(ret) > 0 and top[1] == ret[-1]:
                top2 = heapq.heappop(h)
                ret.append(top2[1])
                if top2[0]+1 < 0:
                    heapq.heappush(h,(top2[0]+1,top2[1]))
                heapq.heappush(h,(top[0],top[1]))
            else:
                ret.append(top[1])
                if top[0]+1 < 0:
                    heapq.heappush(h,(top[0]+1,top[1]))
        return ret
