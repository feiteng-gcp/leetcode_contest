from heapq import heappush, heappop

class Solution:
    def rearrangeBarcodes(self, barcodes: List[int]) -> List[int]:
        seen = {}
        for barcode in barcodes:
            if barcode in seen:
                seen[barcode] += 1
            else:
                seen[barcode] = 1
        
        re = []
        heap = []
        lastitemkey = (0, -1)
        for key, item in seen.items():
            heappush(heap, (-item, key))
        
        while heap:
            MinusItem, key = heappop(heap)
            re.append(key)
            if lastitemkey[1]>=1 and lastitemkey[0]<0:
                heappush(heap, lastitemkey)
            lastitemkey = (MinusItem+1, key)
        return re
