class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        target = -1
        i = len(A) - 1
        while i >= 1:
            if A[i - 1] > A[i]:
                target = i - 1
                break
            i -= 1
        if target == -1:
            return A
        else:
            # find the largest and swap
            target2 = -1
            for j in range(len
(A) - 1, target, -1):
                if A[j] < A[target]:
                    target2 = j
                    break
            A[target], A[target2] = A[target2], A[target]
            return A
