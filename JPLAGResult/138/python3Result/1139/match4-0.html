class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        
        candidate = -1
        for i in range(n-1, 0, -1):
            if (A[i-1] > A[i]):
                candidate = i-1
                break
        if candidate == -1:
            return A
        
        second = candidate+1
        
        for i in range(candidate+2, n):
            if A[candidate] > A[i]:
                second = i
            else:
                break
        
        A[candidate], A[second] = A[second], A[candidate]
        return A
