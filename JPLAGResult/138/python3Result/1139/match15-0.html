class Solution:
    def prevPermOpt1(self, arr: List[int]) -> List[int]:
        if len(arr) < 2:
            return arr
        
        for k in range(len(arr)-2, -1, -1):
            if arr[k] > arr[k + 1]:
                l = k + 1
                
                for i in range(k+2, len(arr)):
                    if arr[k] > arr[i] > arr[l]:
                        l = i
                
                arr[k], arr[l] = arr[l], arr[k]
                break
        
        return arr
