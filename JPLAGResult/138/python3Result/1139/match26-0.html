from bisect import bisect_left
from typing import List


class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        if len(A) == 1:
            return A

        # item_to_idx = {item: i for i, item in enumerate(A)}
        visited = [A[-1]]
        for i in reversed(range(len(A) - 1)):
            item = A[i]
            if item <= visited[-1]:
                visited.append(item)
            else:  # item > visited[-1]
                # print(visited)
                visited = visited[::-1]
                swap_idx = (bisect_left(visited, item) - 1) + i + 1
                A[i], A[swap_idx] = A[swap_idx], A[i]
                return A
        return A
