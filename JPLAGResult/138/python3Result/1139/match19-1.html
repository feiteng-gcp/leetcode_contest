class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        if A == sorted(A) or len(A) <= 1:
            return A
        n = len(A)
        j = n - 1
        while j > 0:
            if A[j - 1] > A[j]:
                j = j - 1
                break
            j -= 1
        i = n - 1
        while i > j:
            
if A[i] < A[j]:
                break
            i -= 1
        A[i], A[j] = A[j], A[i]
        return A
