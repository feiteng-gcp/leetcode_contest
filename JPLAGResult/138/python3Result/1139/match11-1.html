class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        i = n - 2
        while i >= 0:
            if A[i] > A[i + 1]: break
            i -= 1
        if i < 0: return A
        j, target = n - 1, max(x for x in A[i + 1:] if x < A[i])
        while j > i:
            if A[j] == target: break
            j -= 1
        A[i], A[j] = A[j], A[i]
        return A
