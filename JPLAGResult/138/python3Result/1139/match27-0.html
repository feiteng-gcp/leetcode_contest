class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        first_num = -1
        for i in reversed(range(n-1)):
            if A[i] > A[i+1]:
                first_num = i
                break
        # print("first_num = ", first_num)
        if first_num == -1:
            return A
        closest_num, closest_index = 0 , -1
        for i in range(first_num+1,n):
            if A[i] < A[first_num] and A[i] >= closest_num:
                closest_num = A[i]
                closest_index = i
        # print("closest_index = ", closest_index)
        if closest_index == -1:
            return A
        A[closest_index],A[first_num] = A[first_num], A[closest_index]
        return A
