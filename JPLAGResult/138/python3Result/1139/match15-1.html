class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        swap_first = -1        
        for i in range(len(A) - 1):
            if A[i] > A[i+1]:
                swap_first = i
        if swap_first == -1:
            return A
        swap_second 
= swap_first + 1
        for i in range(swap_first + 1, len(A)):
            if A[swap_second] < A[i] < A[swap_first]:
                swap_second = i
                
        A[swap_first], A[swap_second] = A[swap_second], A[swap_first]
        
        return A
