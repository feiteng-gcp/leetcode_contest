class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        i = n-1
        while i > 0:
            if A[i-1] > A[i]:
                break
            i -= 1
        if i == 0:
            return A
        pivot = i-1
        i = n-1
        while i > pivot:
            if A[i] < A[pivot]:
                break
            i -= 1
        assert i > pivot
        A[pivot], A[i] = A[i], A[pivot]
        return A
