class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        
        for i in reversed(range(1, len(A))):
            if A[i-1] > A[i]:
                break
        else:
            return A
        
        i = i - 1
        j = i + 1
        swap = A[j]
        while j + 1 < len(A):
            if A[j+1] >= A[i]:
                break
            #if A[j+1] == A[j]:
                #break
            j = j + 1
        
        A[i], A[j] = A[j], A[i]
        
        return A
