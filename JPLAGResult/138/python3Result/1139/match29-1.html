class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        if n == 1:
            return A
        j 
= n - 2
        while j >= 0:
            if A[j] > A[j + 1]:
                break
            j -= 1
        if j == -1:
            return A
        cand = None
        pos = n
        for i in range(n - 1, j, -1):
            if A[i] < A[j]:
                if cand is None or A[i] > cand:
                    cand = A[i]
                    pos = i
        B = list(A)
        B[pos], B[j] = B[j], B[pos]
        return B
