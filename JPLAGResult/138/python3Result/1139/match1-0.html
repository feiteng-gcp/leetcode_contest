class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        
        for j in reversed(range(len(A)-1)):
            for i in reversed(range(j+1, len(A))):
                if A[i] < A[j]:
                    A[i], A[j] = A[j], A[i]
                    return A

        #             return A
        return A
