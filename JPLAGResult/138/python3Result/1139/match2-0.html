class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        l = len(A)
        for i in range(l-1, 0, -1):
            if A[i] < A[i-1]:
                for j in range(l-1, i-1, -1):
                    if A[i-1] > A[j]:
                        A[i-1],A[j] = A[j],A[i-1]
                        return A
        return A
