class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        idx1 = -1
        for i in range(len(A) - 2, -1, -1):
            if A[i] > A[i + 1]:
                idx1 = i
                break
        if idx1 < 0:
            return A
        for i in range(len(A) - 1, -1, -1):
            if A[i] < A[idx1]:
                A[idx1], A[i] = A[i], A[idx1]
                return A
