class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        for i in range(n - 1, 0, -1):
            if A[i - 1] > A[i]:
                for j in range
(n - 1, 0, -1):
                    if A[j] < A[i - 1]:
                        A[j], A[i - 1] = A[i - 1], A[j]
                        return A
        return A
