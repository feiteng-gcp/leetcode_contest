class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        maxA = max(A) + 1
        min_after = [maxA for i in range(n)]
        for i in range(n - 2, -1, -1):
            min_after[i] = min(A[i + 1], min_after[i + 1])
        i = n - 1
        while i >= 0 and min_after[i] >= A[i]:
            i -= 1
        if i < 0:
            return A
        k = 0
        pos = -1
        for j in range(i + 1, n):
            if A[j] >= k and A[j] < A[i]:
                k = A[j]
                pos = j
        A[i], A[pos] = A[pos], A[i]
        return A
