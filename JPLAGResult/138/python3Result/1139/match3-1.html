class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        n = len(A)
        l = -1
        for i in range(n-1, 0, -1):
            if A[i] < A[i-1]:
                l = i-1
                break
        else:
            return A
        
        r 
= -1
        for i in range(n-1, l, -1):
            if A[i] < A[l]:
                r = i
                break
        
        A[l], A[r] = A[r], A[l]
        return A
