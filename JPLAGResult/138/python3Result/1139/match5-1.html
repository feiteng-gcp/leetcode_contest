class Solution:
    def prevPermOpt1(self, A: List[int]) -> List[int]:
        if len(A) == 1:
            return A
        for i in range(len(A)-1, 0, -1):
            if A[i] >= A[i-1]:
                pass
            else:
                x = i - 1
                break
        else:
            return A
        if x == len(A) - 1:
            A[-1],A[-2] = A[-2],A[-1]
            return A
        for i in range(len
(A)-1, 0, -1):
            if A[i] < A[x]:
                y = i
                break
        A[x], A[y] = A[y], A[x]
        return A
