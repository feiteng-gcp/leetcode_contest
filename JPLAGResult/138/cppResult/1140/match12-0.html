class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        map<int, int> m;
        for (int i : barcodes)
            m[i]++;
        vector<pair<int, int>> vp;
        for (auto [i, cnt] : m)
            vp.emplace_back(cnt, i);
        sort(vp.begin(), vp.end(), greater<pair<int, int>>());

        vector<int> ret(barcodes.size());
        for (int p = 0, i = 0; i < vp.size(); p += 2) {
            ret[p < barcodes.size() ? p : (p - barcodes.size()) / 2 * 2 + 1] = vp[i].second;
            if (!--vp[i].first)
                i++;
        }
        return ret;
    }
};
