class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& B) {
        map<int, int> m;
        priority_queue<pair<int, int>> pq;
        for (auto b:B) m[b]++;
        for (auto a:m) {
            pq.push(make_pair(a.second, a.first));
        }
        int k=0;
        while (!pq.empty()) {
            int nn=pq.top().first;
            int val=pq.top().second;
            pq.pop();
            for (int p=0; p<nn; p++) {
                B[k]=val;
                k+=2;
                if (k>=B.size()) k=1;
            }
        }
        return B;
    }
};
