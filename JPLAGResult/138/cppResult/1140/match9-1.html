class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int> barcodes) {
        unordered_map<int, int> cnts;
        for (int b : barcodes) { ++cnts[b]; }

        priority_queue<pair<int, int>> pq;
        for (auto pr : cnts) { pq.push({ pr.second, pr.first }); }

        vector<int> ans;
        queue<pair<int, int>> prev;
        while (!pq.empty() || prev.size() > 1)
        {
            if (prev.size() > 1)
            {
                auto cur = prev.front(); prev.pop();
                if(cur.first > 0) pq.push(cur);
            }
            if (!pq.empty())
            {
                auto cur = pq.top(); pq.pop();
                ans.push_back(cur.second);
                --cur.first;
                prev.push(cur);
            }
        }

        return ans;
    }
    
};
