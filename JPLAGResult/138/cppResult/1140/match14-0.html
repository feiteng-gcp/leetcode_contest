class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& b) {
        unordered_map<int, int> temp;
        for (int x : b) {
            temp[x]++;
        }
        
        priority_queue<pair<int,int>> pq;
        for (auto x : temp) {
            pq.push({x.second, x.first});
        }
        vector<int> ans;
        while (!pq.empty()) {
            pair<int,int> z = pq.top();
            pq.pop();
            if (ans.empty() || ans.back() != z.second) {
                ans.push_back(z.second);
                if (z.first > 1) {
                    pq.push({z.first - 1, z.second});
                }
            }
            else {
                pair<int,int> x = pq.top();
                pq.pop();
                ans.push_back(x.second);
                if (x.first > 1) {
                    pq.push({x.first - 1, x.second});
                }
                pq.push(z);
            }
        }
        return ans;
    }
};
