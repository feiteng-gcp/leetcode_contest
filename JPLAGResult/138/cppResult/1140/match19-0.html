class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        unordered_map<int, int> ma;
        for(auto & b : barcodes)
            ma[b]++;
        priority_queue<pair<int, int>> que;
        for(auto& m : ma) {
            que.push({m.second, m.first});
        }
        vector<int> ret;
        while(!que.empty()) {
            pair<int, int> first = que.top();
            que.pop();
            first.first--;
            ret.push_back(first.second);
            
            if(!que.empty()) {
                pair<int, int> second = que.top();
                que.pop();
                second.first--;
                ret.push_back(second.second);
                if(second.first > 0)
                    que.push(second);
            }
            
            if(first.first > 0)
                que.push(first);
        }
        
        return ret;
        
    }
};
