class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        unordered_map<int,int> bc;
        for (auto n : barcodes) ++bc[n];
        multimap<int,int> mult;
        for (auto item : bc) {
            mult.insert({item.second,item.first});
        }
        auto n = barcodes.size();
        vector<int> res(n);
        int pos = 0;
        for (auto it = mult.rbegin(); it != mult.rend(); ++it) {
            for (auto i = 0; i < it->first; ++i) {
                res[pos] = it->second; pos += 2; if (pos >= n) pos = 1;
            }
        }
        return res;
    }
};
