class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        map<int, int> m;
        for (int i = 0;i < barcodes.size();i++) {
            m[barcodes[i]]++;
        }
        priority_queue<pair<int, int>> pq;
        for (auto it : m) {
            pq.push({it.second, it.first});
        }
        vector<int> ret;
        while (!pq.empty()){
            if (ret.empty() || ret.back() != pq.top().second) {
                pair<int, int> p = pq.top();
                pq.pop();
                ret.push_back({p.second});
                p.first--;
                if (p.first != 0) {
                    pq.push(p);
                }
            }
            else {
                pair<int, int> tmp = pq.top();
                pq.pop();
                pair<int, int> p = pq.top();
                pq.pop();
                ret.push_back({p.second});
                p.first--;
                if (p.first != 0) {
                    pq.push(p);
                }
                pq.push(tmp);
            }
        }
        return ret;
    }
};
