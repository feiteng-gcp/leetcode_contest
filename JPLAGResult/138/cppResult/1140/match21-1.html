class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        map<int, int> count;
        for (int num : barcodes) {
            count[num]++;
        }
        vector<pair<int, int>> v;
        for (auto& x : count) {
            v.push_back({x.second, x.first});
        }
        sort(v.begin(), v.end());
        vector<int> ans(barcodes.size());
        int gap = ans.size() / v.back().first ;//+ (ans.size() % v.back().first == 0 ? 0 : 1);
        if (gap == 1) ++gap;
        for (int i = v.size() - 1, j = 0; i >= 0; --i) {
            for (int c = 0; c < v[i].first; ++c) {
                if (j >= ans.size()) j = j % gap + 1;
                assert(ans[j] == 0);
                ans[j] = v[i].second;
                j += gap;
            }
        }
        return ans;
    }
};
