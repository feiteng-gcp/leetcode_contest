class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& b) {
      map<int, int> m;
        for(int i = 0; i < b.size(); i++) {
            m[b[i]]++;
        }
        multimap<int, int, std::greater<int>> mm;
        for(auto u: m) {
            mm.insert(make_pair(u.second, u.first));
        }
        vector<int> ret;
        int last = -1;
        while(ret.size() < b.size()) {
            auto u = mm.begin();
            while(u->second == last) u++;
            last = u->second;
            ret.push_back(u->second);
            auto v = make_pair(u->first-1, u->second);
            mm.erase(u);
            mm.insert(v);
        
        }
        return ret;

    }
};
