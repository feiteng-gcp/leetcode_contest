class Solution {
  public:
    vector<int> rearrangeBarcodes(vector<int> &barcodes) {
        map<int, int> ma;
        for (int i = 0; i < barcodes.size(); ++i) {
            ma[barcodes[i]]++;
        }

        priority_queue<pair<int, int>> pq;
        for (auto p : ma) {
            pq.emplace(p.second, p.first);
        }
        vector<int> ret;
        auto pp = pq.top();
        ret.push_back(pp.second);
        pq.pop();
        if (pp.first != 1)
            pq.emplace(pp.first - 1, pp.second);
        while (pq.size()) {
            auto pp = pq.top();
            pq.pop();
            if (pp.second == ret.back()) {
                auto pp2 = pq.top();
                pq.pop();
                ret.push_back(pp2.second);
                pq.push(pp);
                if (pp2.first != 1)
                    pq.emplace(pp2.first - 1, pp2.second);
            } else {
                ret.push_back(pp.second);
                if (pp.first != 1)
                    pq.emplace(pp.first - 1, pp.second);
            }
        }
        return ret;
    }
};
