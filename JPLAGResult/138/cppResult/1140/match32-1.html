class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& b) {
        map<int, int> cnt;
        int n = b.size();
        for (int i = 0; i < n; ++i)
            cnt[b[i]]++;
        vector<vector<int> > rcnt;
        for(map<int, int>::iterator ite = cnt.begin(); ite != cnt.end(); ++ite) {
            vector<int> tmp(2);
            tmp[0] = ite->first;
            tmp[1] = ite->second;
            rcnt.push_back(tmp);
        }
        sort(rcnt.begin(), rcnt.end(), [](vector<int>& v1, vector<int>& v2) { return v1[1] > v2[1];});
        
        vector<int> res(n);
        int cur = 0;

        for (int i = 0; i < rcnt.size(); ++i) {
            int num = rcnt[i][1];
            int m = rcnt[i][0];
            
            //printf("%d %d\n", m, num);
            for (int j = 0; j < num; ++j) {
                res[cur] = m;
                cur += 2;
                if (cur > n - 1) cur = 1;
            }
        }
        return res;
        
    }
};
