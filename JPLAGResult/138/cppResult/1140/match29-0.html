class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        int N = barcodes.size();
        map<int, int> cou;
        for (auto v : barcodes) cou[v]++;
        
        map<int, vector<int>> cou_inv;
        for (auto pa : cou) cou_inv[-pa.second].push_back(pa.first);
        vector<int> ret(N);
        int i = 0;
        {
            for (auto pa : cou_inv) for (auto w : pa.second)
            {
                
                for (int j = 0; j < -pa.first; j++)
                {
                    int idx = i * 2;
                    if (idx >= N)
                    {
                        if (N % 2 == 0) idx -= (N - 1);
                        else idx -= N;
                    }
                    ret[idx] = w;
                    i++;
                }
            }
        }
        return ret;
    }
};
