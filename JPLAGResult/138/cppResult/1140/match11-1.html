class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        unordered_map<int, int> counter;
        for (int i: barcodes) counter[i]++;

        priority_queue<pair<int, int>> pq;
        for (auto & kv: counter) {
            pq.push({kv.second, kv.first});
        }

        vector<int> res;

        while (pq.size() > 1) {
            pair<int, int> largest = pq.top();
            pq.pop();
            pair<int, int> second_largest = pq.top();
            pq.pop();

            res.push_back(largest.second);
            res.push_back(second_largest.second);
            largest.first--;
            second_largest.first--;

            if (largest.first)
                pq.push(largest);

            if (second_largest.first)
                pq.push(second_largest);
        }

        if (!pq.empty()) {
            res.push_back(pq.top().second);
        }

        return res;
    }
};
