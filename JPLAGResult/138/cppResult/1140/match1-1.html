class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& b) {
        int n = b.size();
        unordered_map<int, int> cnt;
        for(int num: b) cnt[num]++;
        vector<pair<int, int>> tosort;
        for(auto c: cnt) {
            tosort.push_back(make_pair(c.first, c.second));
        }

        sort(tosort.begin(), tosort.end(), [&](const pair<int, int> &p1, const pair<int, int> &p2) {
            return p1.second > p2.second;
        });
        // for(auto p: tosort) {
        //     cout << p.first << " " << p.second << endl;
        // }
        vector<int> res(n, 0);
        int id = 0;
        for(auto p: tosort) {
            int num = p.first, rep = p.second;

            for(int i = 0; i < rep; ++i) {
                res[id] = num;
                id = id + 2;
                if(id >= n) {
                    id = 1;
                }
            }
        }
        
        return res;
    }
};
