bool cmp (pair<int,int> &a , pair<int, int> &b) {
    return a.second > b.second;
}

class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        int n = barcodes.size();
        vector<int> res (n, 0);
        //sort(barcodes.begin(), barcodes.end());
        unordered_map<int, int> cache;
        for (int i = 0; i < n; ++i) {
            cache[barcodes[i]]++;
        }
        vector<pair<int, int>> pairs;
        for (auto & p : cache) {
            pairs.push_back(p);
        }
        sort(pairs.begin(), pairs.end(), cmp);
        
        int i = 0;
        for (auto & p : pairs) {
            int val = p.first;
            int cnt = p.second;
            int j = 0;
            while (i < n && j < cnt) {
                res[i] = val;
                i += 2;
                j++;
            }
            if (j < cnt) {
                i = 1;
                while (i < n && j < cnt) {
                    res[i] = val;
                    i += 2;
                    j++;
                }
            }
        }
        return res;
    }
};
