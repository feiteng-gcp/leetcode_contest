class Solution {
public:
    using pii = pair<int, int>;
    vector<int> rearrangeBarcodes(vector<int>& a) {
        int n = a.size();
        map<int, int> F;
        for (int i = 0; i < n; ++ i)
            F[a[i]] ++;
        vector<pii> v;
        for (auto p : F)
            v.push_back(pii(p.second, p.first));
        sort(v.begin(), v.end());
        reverse(v.begin(), v.end());
        vector<int> ret(n);
        int j = 0;
        for (int i = 0; i < n; i += 2)
        {
            ret[i] = v[j].second;
            if (! -- v[j].first) j ++;
        }
        for (int i = 1; i < n; i += 2)
        {
            ret[i] = v[j].second;
            if (! -- v[j].first) j ++;
        }
        return ret;
    }
};
