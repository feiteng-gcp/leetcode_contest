class Solution {
public:
	vector<int> rearrangeBarcodes(vector<int>& barcodes) {
		int n = barcodes.size();
		map<int, int>ct;
		for (auto it : barcodes)ct[it]++;
		priority_queue<pair<int, int>>Q;
		for (auto it : ct)Q.push({ it.second,it.first });

		vector<int>ret;
		pair<int, int>last = { -1,-1 };
		while (!Q.empty()) {
			auto t = Q.top(); Q.pop();
			ret.push_back(t.second);
			if (last.first > 0)Q.push(last);
			last = { t.first - 1,t.second };
		}
		return ret;
	}
};
