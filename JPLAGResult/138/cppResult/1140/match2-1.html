class Solution {
public:
  vector<int> rearrangeBarcodes(vector<int>& barcodes) {
    map<int, int> cnt;
    for (int v: barcodes) ++cnt[v];
    vector<pair<int, int> > vec;
    for (auto p: cnt) vec.push_back(make_pair(-p.second, p.first));
    sort(vec.begin(), vec.end());
    vector<int> ret = barcodes;

    int p = 0;
    for (int i = 0; i < ret.size(); i += 2) {
      ret[i] = vec[p].second;
      if (!--cnt[ret[i]]) ++p;
    }
    for (int i = 1; i < ret.size(); i += 2) {
      ret[i] = vec[p].second;
      if (!--cnt[ret[i]]) ++p;
    }
    return ret;
  }
};
