class mycomparison
{
      bool reverse;
public:
    mycomparison(const bool& revparam=false)
    {reverse=revparam;}
    bool operator () (const pair<int, int>& lhs, const pair<int, int>&rhs) const

    {
        return (lhs.second<rhs.second);
    }
};


class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        unordered_map<int, int> counter;
        for (auto x: barcodes) {
            counter[x] += 1;
        }
        priority_queue<pair<int, int>, std::vector<pair<int, int>>, mycomparison> heap(mycomparison(false));
        for(auto it = counter.begin(); it != counter.end(); it++) {
            heap.push(make_pair(it->first, it->second));
        }
        vector<int> results;
        results.reserve(barcodes.size());
        while (!heap.empty()) {

            auto pair = heap.top();
            heap.pop();
            if (results.size() > 0 && pair.first == results.back()) {
                auto pair2 = heap.top();
                heap.pop();
                results.push_back(pair2.first);
                pair2.second--;
                if (pair2.second > 0)
                    heap.push(pair2);
                heap.push(pair);
            } else {
                results.push_back(pair.first);
                pair.second--;
                if (pair.second > 0)
                    heap.push(pair);
            }
        }
        return results;
    }
};
