class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        int n = barcodes.size();
        map<int, int> record;
        vector<int> result(n);
        for(int i = 0; i < n; ++i){
            record[barcodes[i]] += 1;
        }

        priority_queue<pair<int, int>> pq;
        for(auto rec : record){
            pq.push({rec.second, rec.first});
        }

        auto top = pq.top();
        pq.pop();
        result[0] = top.second;
        top.first--;
        pq.push(top);
        for(int i = 1; i < n; ++i){
            top = pq.top();
            pq.pop();

            if(result[i-1] == top.second){
                auto temp = pq.top();
                pq.pop();
                pq.push(top);
                top = temp;
            }

            result[i] = top.second;
            top.first--;
            pq.push(top);
        }
        return result;
    }
};
