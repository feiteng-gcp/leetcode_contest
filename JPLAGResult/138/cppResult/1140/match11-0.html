class Solution {
public:
    
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        unordered_map<int, int> ct;
        for(int b : barcodes) ct[b]++;
        vector<pair<int, int>> v;
        for(auto &it : ct) {
            v.push_back({-it.second, it.first});
        }
        sort(begin(v), end(v));
        vector<int> ok;
        for(auto &it : v) {
            for(int i = 0; i < abs(it.first); i++) {
                ok.push_back(it.second);
            }
        }
        vector<int> res(barcodes.size());
        for(int i = 0, j = 0; i < barcodes.size(); i++, j += 2) {
            if(j >= barcodes.size()) j = 1;
            res[j] = ok[i];
        }
        return res;
    }
};
