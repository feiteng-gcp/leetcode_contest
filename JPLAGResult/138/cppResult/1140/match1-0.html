class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        int N = barcodes.size();
        map<int, int> cache;
        set<int> s;
        
        for (auto v : barcodes) {
            cache[v] += 1;
            s.insert(v);
        }
        
        vector<int> elements(s.begin(), s.end());
        
        sort(elements.begin(), elements.end(), [&] (int a, int b) {
           return cache[a] > cache[b]; 
        });
        
        vector<int> result(N);
        
        int at = 0;
        for (auto val : elements) {
            int nr = cache[val];
            for (int i = 0; i < nr; i++) {
                result[at] = val;
                at += 2;
                if (at >= N) {
                    at = 1;
                }
            }
        }
        
        return result;
    }
};
