class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        vector<int> c2 = customers;
        int N = customers.size();
        for (int i = 0; i < N; i++) c2[i] *= grumpy[i];
        int acc = 0, acc_max = 0;
        for (int i = 0; i < N; i++)
        {
            acc += c2[i];
            if (i >= X) acc -= c2[i - X];
            acc_max = max(acc_max, acc);
        }
        return accumulate(customers.begin(), customers.end(), 0) - accumulate(c2.begin(), c2.end(), 0) + acc_max;
    }
};
