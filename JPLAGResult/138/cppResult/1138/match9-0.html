class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int total_points = 0;
        int max_tech_points = 0;
        int current_tech_points = 0;
        
        for(int i=0; i<customers.size(); i++){
            if(grumpy[i] == 0){
                total_points += customers[i];               
            }
            else {
                current_tech_points += customers[i];                
            }            
            if(i>=X && grumpy[i-X]==1){
                current_tech_points -= customers[i-X];   
            }            
            max_tech_points = max(max_tech_points, current_tech_points);
        }
        return total_points + max_tech_points;
    }
};
