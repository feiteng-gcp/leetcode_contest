#include <bits/stdc++.h>
using namespace std;
typedef long long ll;


class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int n = customers.size();
        vector<int> sums(n+1, 0);
        int res = 0;
        for(int i=1; i<=n; i++){
            int ii = i - 1;
            sums[i] = sums[i-1];
            if(grumpy[ii] == 1){
                sums[i] += customers[ii];
            }else{
                res += customers[ii];
            }
        }

        int maxv = 0;
        for(int i=X; i<=n; i++){
            maxv = max(maxv, sums[i] - sums[i - X]);
        }
        res += maxv;
        return res;
        
    }
};
