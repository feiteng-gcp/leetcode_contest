class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        

        int cur = 0;
        for (int i = 0; i < customers.size(); i++) {
            if (!grumpy[i])
                cur += customers[i];
        }
        
        for (int i = 0; i < X; i++) {
            if (grumpy[i])
                cur += customers[i];
        }
        
        int res = cur;
        
        for (int i = 0; i < customers.size() - X; i++) {
            if (grumpy[i])
                cur -= customers[i];
            if (grumpy[i + X])
                cur += customers[i + X];
            
            res = max(res, cur);
        }
        
        return res;
        
    }
};
