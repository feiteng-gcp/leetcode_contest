class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int n = customers.size();
        vector<int> A(n+1);
        vector<int> B(n+1);
        for(int i=1; i<n+1; i++) {

            A[i]=A[i-1]+customers[i-1];
            if(grumpy[i-1]==0) {
                B[i]=B[i-1]+customers[i-1];
            } else {
                B[i]=B[i-1];
            }
        }
        int ans=0;
        for(int i=0; i<n-X+1; i++) {
            ans=max(ans,B[i]+(A[i+X]-A[i])+B[n]-B[i+X]);
        }
        return ans;
    }
};
