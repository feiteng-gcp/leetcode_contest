class Solution {
public:
    //010
    //i: 3, 4, 5, 6, 7
    
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int ans = 0;
        for(int i = 0; i < customers.size(); ++i) {
            if(grumpy[i] == 0) ans += customers[i];
        };
        
        int extra = 0;
        for(int i = 0; i < X; ++i) {
            if(grumpy[i] == 1) extra += customers[i];
        };
        
        int m = ans + extra;
        for(int i = X; i < grumpy.size(); ++i) {
            if(grumpy[i] == 1) {
                extra += customers[i];
            };
            if(grumpy[i - X] == 1) {
                extra -= customers[i - X];
            };
            //std::cout << i << ", " << extra << std::endl;
            m = std::max(m, ans + extra);
        };
        return m;
    }
};
