class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
      int n = customers.size(), now = 0;
      for (int i = 0; i < n; ++i) {
        if (!grumpy[i]) now += customers[i];
      }
      int ret = 0;
      for (int i = 0; i < n; ++i) {
        if (grumpy[i]) now += customers[i];
        if (i - X >= 0 && grumpy[i-X]) now -= customers[i-X];
        ret = max(ret, now);
      }
      return ret;
    }
};
