class Solution {
public:
    int maxSatisfied(vector<int>& c, vector<int>& g, int X) {
        int base = 0;
        for(int i = 0; i < c.size(); i++) {
            if(g[i] == 0) {
                base += c[i];
            }
        }
        int w = 0;
        for(int i = 0; i < X; i++) {
            if(g[i] == 1) {
                w += c[i];
            }
        }
        int ret = base + w;
        for(int i = 1; i + X - 1 < g.size(); i++) {
            if( g[i-1] == 1) w -= c[i-1];
            if(g[i+X-1] == 1) w += c[i+X-1];
            ret = max(ret, base + w);
        }
        return ret;
    }
};
