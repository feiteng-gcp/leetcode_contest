class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int size = customers.size();
        int ret = 0, sum = 0;
        for(int i = 0; i < size; ++i) {
            if(grumpy[i] == 0)
                sum += customers[i];
        }
        vector<int> more(size, 0);
        if(grumpy[0] == 1)
            more[0] = customers[0];
        for(int i = 1; i < size; ++i) {
            more[i] += more[i - 1];
            if(grumpy[i] == 1)
                more[i] += customers[i];
        }
        int maxM = more[X - 1];
        for(int i = X; i < size; ++i) {
            maxM = max(maxM, more[i] - more[i - X]);
        }
        
        return sum + maxM;
    }
};
