class Solution {
public:
    int maxSatisfied(vector<int>& a, vector<int>& b, int X) {
        int n = a.size();
        vector<int> sum(n + 1);
        for (int i = 0; i < n; ++i) sum[i + 1] = sum[i] + a[i];
        vector<int> L(n + 1), R(n + 1);
        for (auto& it : b) it = 1 - it;
        L[0] = a[0] * b[0];
        for (int i = 1; i < n; ++i) L[i] = L[i - 1] + a[i] * b[i];
        R[n] = 0;
        for (int i = n - 1; i >= 0; --i) R[i] = R[i + 1] + a[i] * b[i];
        cout << n << endl;
        int ret = sum[X] + R[X];

        if (n - X - 1 >= 0) ret = max(ret, L[n - X - 1] + sum[n] - sum[n - X]);
        for (int i = 1; i + X < n; ++i) {
            // [i, i + X)
            int cur = L[i - 1] + sum[i + X] - sum[i] + R[i + X];
            ret = max(ret, cur);
        }
        return ret;
    }
};
