class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int maxSave = 0;
        int N = customers.size();

        for(int i=0; i<X&&i<N; i++){
            if(grumpy[i]==1)
              maxSave += customers[i];
        }
        int curSave = maxSave;
        for(int i=X; i<N; i++){
            if(grumpy[i-X]==1)
                curSave -= customers[i-X];
            if(grumpy[i]==1)
                curSave += customers[i];
            maxSave = max(maxSave, curSave);
        }
        int totalSat = 0;
        for(int i=0; i<N; i++){
            if(grumpy[i]==0)
              totalSat+=customers[i];
        }
        return totalSat+maxSave;
    }
};
