class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int localDiff = 0;
        for(int i = 0; i < min(X, (int)grumpy.size()); i++){
            if(grumpy[i] == 1){
                localDiff += customers[i];
            }
        }
        int maxDiff = localDiff;
        for(int i = X, j = 0; i < grumpy.size(); i++, j++){
            if(grumpy[i] == 1){
                localDiff += customers[i];
            }
            if(grumpy[j] == 1){
                localDiff -= customers[j];
            }
            maxDiff = max(maxDiff, localDiff);
        }
        int amount = maxDiff;
        for(int i = 0; i < grumpy.size(); i++){
            if(grumpy[i] == 0)
                amount += customers[i];
        }
        return amount;
    }
};
