class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int r = 0;
        int n = customers.size();
        vector<int> zz(n+1);
        for (int i=0; i<n; ++i) {
            r+= customers[i] * (1-grumpy[i]);

            zz[i+1] = zz[i] + customers[i] * (grumpy[i]);
        }
        int g  = 0;
        for (int i=0; i+X<=n; ++i) {
            g = max(g, zz[i+X] - zz[i]);
            
        }
        return g+ r;
    }
};
