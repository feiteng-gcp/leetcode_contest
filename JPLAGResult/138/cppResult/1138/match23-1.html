class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int l = customers.size();
        int sat = 0;
        
        for(int i = 0;i < l;i ++)
        {
            if(grumpy[i] == 0) sat += customers[i];
        }
        
        //cout << sat << endl;
        int max_sat = 0;

        for(int i = 0;i < X;i ++)
        {
            if(grumpy[i] == 1) max_sat += customers[i];
        }
        int last_sat = max_sat;
        for(int i = 1;i < l-X+1;i ++)
        {
            if(grumpy[i-1] == 1) 
            {
                last_sat -= customers[i-1];
            }
            if(grumpy[i+X-1] == 1)
            {
                last_sat += customers[i+X-1];
            }
            if(max_sat < last_sat)
            {
                max_sat = last_sat;
            }
        }
        return max_sat+sat;
    }
};
