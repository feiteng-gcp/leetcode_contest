class Solution {
 public:
  int maxSatisfied(vector<int> &customers, vector<int> &grumpy, int x) {
    int happies = 0;
    int m = customers.size();
    int best_more = 0;

    int more = 0;
    for (int i = 0; i < m; ++i) {
      if (grumpy[i]) {
        more += customers[i];
      } else {
        happies += customers[i];
      }
      if (i >= x && grumpy[i-x]) {
        more -= customers[i-x];
      }
      best_more = max(best_more, more);
    }
    return happies + best_more;
  }
};
