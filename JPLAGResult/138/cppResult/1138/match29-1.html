class Solution {
public:
    int maxSatisfied(vector<int>& a, vector<int>& b, int X) {

        int n = a.size();
        vector<int> vec(n + 1), v2(n + 1);
        for (int i = 0; i < n; i++) {
            vec[i + 1] = vec[i] + a[i] * (1 - b[i]);
            v2[i + 1] = v2[i] + a[i];
        }
        int ans = vec[n];
        for (int i = X; i <= n; i++) {
            ans = max(ans, vec[n] - vec[i] + vec[i - X] + v2[i] - v2[i - X]);
        }
        return ans;
    }
};
