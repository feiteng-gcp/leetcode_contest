class Solution {
public:
  int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
    int N = customers.size();
    int ans = 0;
    for (int i = 0; i < N; i++) if (grumpy[i] == 0) ans += customers[i], customers[i] = 0;
    partial_sum(customers.begin(), customers.end(), customers.begin());
    int best = customers[X - 1];
    for (int i = X; i < N; i++) best = max(best, customers[i] - customers[i - X]);
    return best + ans;
  }
};
