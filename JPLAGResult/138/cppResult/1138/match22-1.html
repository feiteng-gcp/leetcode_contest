class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int res=0;
        for (int i=0;i<(int)customers.size();++i){

            if (grumpy[i]==0) res+=customers[i];
        }
        int t=0;
        for (int i=0;i<X;++i){
            if (grumpy[i]==1) t+=customers[i];
        }
        int ans=res+t;
        for (int i=X;i<(int)customers.size();++i){
            if (grumpy[i]==1) t+=customers[i];
            if (grumpy[i-X]==1) t-=customers[i-X];
            ans=max(ans,res+t);
        }
        return ans;
    }
};
