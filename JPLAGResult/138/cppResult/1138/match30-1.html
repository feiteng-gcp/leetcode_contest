class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int n = customers.size();
        int cnt = 0;
        for (int i = 0; i < X && i < n; ++ i) cnt += customers[i];
        
        
        int pre = 0, suf = 0;

        for (int i = X; i < n; ++ i) if (grumpy[i] == 0) suf += customers[i];
        int ans = cnt + suf;
        for (int i = X; i < n; ++ i) {
            cnt -= customers[i - X];
            cnt += customers[i];
            if (grumpy[i - X] == 0) pre += customers[i - X];
            if (grumpy[i] == 0) suf -= customers[i];
            ans = max(ans, cnt + pre + suf);
        }
        return ans;
    }
};
