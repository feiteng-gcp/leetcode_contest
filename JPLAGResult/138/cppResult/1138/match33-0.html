class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int base = 0;
        for (int i = 0; i < customers.size(); ++i) {
            if (grumpy[i] == 0) {
                base += customers[i];
            }
        }
        int maxNonGrumpy = 0;
        int nonGrumpy = 0;
        for (int i = 0; i < customers.size(); ++i) {
            if (i >= X && grumpy[i-X] == 1) {
                nonGrumpy -= customers[i-X]; 
            }
            if (grumpy[i] == 1) {
                nonGrumpy += customers[i];
            }
            maxNonGrumpy = max(maxNonGrumpy, nonGrumpy);
        }
        return base + maxNonGrumpy;
    }
};
