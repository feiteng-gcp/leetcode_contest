class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int n = customers.size();
        int cnt = 0, cur = 0, total = 0;
        for (int i = 0; i < n; ++i) {
            if (grumpy[i]) {
                cur += customers[i];
            } else {
                total += customers[i];
            }
            if (i-X >= 0 && grumpy[i-X]) {
                cur -= customers[i-X];
            }
            cnt = max(cnt, cur);
        }
        return total + cnt;
    }
};
