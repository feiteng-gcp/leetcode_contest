class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        // 
    	int n = customers.size();

    	int m = 0;	// max value;
    	int sum = 0;	// total non grumpy sum;
    	int cur = 0;	// current X range grumpy sum

    	// compute non grumpy sum.
    	for (int i = 0; i < n; ++i) {
    		if (grumpy[i] == 0) {
    			sum += customers[i];
    		}
    	}

    	// initialize max
    	for (int i = 0; i < X; ++i) {
    		if (grumpy[i] == 1) {
    			cur += customers[i];
    		}
    	}
    	m = cur;

    	// sum + cur represents total sat for range [0, X-1]

    	// DP
    	for (int i = X; i < n; ++i) {
    		// removing the 0th
    		if (grumpy[i-X] == 1) {
    			cur -= customers[i-X];
    		}
    		// adding the Xth
    		if (grumpy[i] == 1) {
    			cur += customers[i];
    		}
    		m = max(m, cur);
    	}

    	return (m + sum);

    }
};
