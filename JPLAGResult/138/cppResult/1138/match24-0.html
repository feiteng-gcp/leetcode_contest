class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int n=customers.size();
        int ans=0;
        for(int i=0;i<n;i++)
        {
            if(!grumpy[i])ans=ans+customers[i];
        }
        for(int i=0;i<X;i++)
        {
            if(grumpy[i])ans=ans+customers[i];
        }
        int now=ans;
        for(int i=X;i<n;i++)
        {
            if(grumpy[i])now=now+customers[i];
            if(grumpy[i-X])now=now-customers[i-X];
            ans=max(ans,now);
        }
        return ans;
    }
};
