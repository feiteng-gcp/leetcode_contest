class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int n = customers.size();
        int ans = 0;
        vector<int> loss(n, 0);
        for(int i = 0; i < n; i++) {
            if(grumpy[i])
                loss[i] = customers[i];
            else
                ans = ans + customers[i];
        }

        int mx = 0, sum = 0;
        for(int i = 0; i < n; i++) {
            sum = sum + loss[i];
            if(i >= X)
                sum = sum - loss[i - X];
            mx = max(mx, sum);
        }
        ans = ans + mx;
        return ans;
    }
};
