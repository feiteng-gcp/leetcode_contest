class Solution {
public:
int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
    int n = customers.size();
    int res = 0;
    for (int i = 0; i < n; i++) {
        int customer = customers[i];
        int status = grumpy[i];
        if (status == 0) {
            res += customer;
        }
    }
    
    
    int total = 0;
    int cnt = 0;
    for (int i = 0; i < X; i++) {
        if (grumpy[i] == 1) {
            cnt += customers[i];
        }
    }
    total = cnt;
    for (int i  = X; i < n; i++) {
        if (grumpy[i - X] == 1) {
            cnt -= customers[i - X];
        }
        if (grumpy[i] == 1) {
            cnt += customers[i];
        }
        
        total = max(total, cnt);
    }
    
    return res + total;
}
};
