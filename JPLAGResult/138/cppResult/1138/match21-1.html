class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int base = 0;
        for (int i = 0; i < customers.size(); ++i) {
            if (!grumpy[i]) base += customers[i];
        }
        int increased = 0;
        for (int i = 0; i < X; ++i) {
            if (grumpy[i]) increased += customers[i];
        }
        int max_increased = increased;
        for (int i = X; i < customers.size(); ++i) {
            if (grumpy[i-X]) increased -= customers[i-X];
            if (grumpy[i]) increased += customers[i];
            if (max_increased < increased) max_increased = increased;
        }
        return max_increased + base;
    }
};
