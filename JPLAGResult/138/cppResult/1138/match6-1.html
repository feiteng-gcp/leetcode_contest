class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
        int n = customers.size();
        

        int sat = 0;
        for (int i = 0; i < n; ++i) {
            if (grumpy[i] != 1) {
                sat += customers[i];
            }
        }
        
        int bonus = 0;
        for (int i = 0; i < X; ++i) {
            if (grumpy[i] == 1) {
                bonus += customers[i];
            }
        }
        
        int maxb =bonus;
        
        for (int i = X; i < n; ++i) {
            int j = i - X;

            if (grumpy[j] == 1) {
                bonus -= customers[j];
            }
            if (grumpy[i] == 1) {
                bonus += customers[i];
            }
            
            maxb = max(maxb, bonus);
        }
        
        return sat + maxb;
    }
};
