class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size();

        for( int i = n-2; i >= 0; i-- ) {
            if( A[i] > A[i+1] ) {
                for( int j = n-1; j >= i+1; j-- ) {
                    if( A[i] > A[j] ) {
                        swap( A[i], A[j] );
                        return A;
                    }
                }
            }
        }
        return A;
    }
};
