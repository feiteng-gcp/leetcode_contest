class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size();
        map<int, int> map;
        map[A[n - 1]] = n - 1;

        for (int i = n - 2; i >= 0; --i) {
            auto jt = map.lower_bound(A[i]);
            if (jt != map.begin()) {
                --jt;
                swap(A[i], A[jt->second]);
                break;
            }
            map[A[i]] = i;
        }
        return A;
    }
};
