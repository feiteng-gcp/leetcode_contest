class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int N=A.size();
        int t=-1;
        int r=100000;
        for(int i=N-1;i>=0;i--){
            if(r<A[i]){
                t=i;
                break;
            }
            r=min(A[i],r);
        }
        if(t==-1) return A;
        int c=-1;
        for(int i=t+1;i<N;i++){
            if(A[i]<A[t]){
                if(c==-1||A[i]>A[c]) c=i;
            }
        }
        swap(A[t],A[c]);
        return A;
    }
};
