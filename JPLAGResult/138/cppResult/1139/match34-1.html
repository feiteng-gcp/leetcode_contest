class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        // smaller, so swap large in left with small in right
        // largest, large num needs to be as left as possible, so small num need to be as large as possible
        int N = A.size();
        for (int i = N-1; i >= 0; --i) {
            int smaller = -1;
            for (int j = i+1; j < N; ++j) {
                if (A[j] < A[i]) {
                    smaller = std::max(smaller, A[j]);
                }
            }
            
            if (smaller != -1) {
                for (int j = i+1; j < A.size(); ++j) {
                    if (A[j] == smaller) {
                        A[j] = A[i];
                        break;
                    }
                }
                A[i] = smaller;
                break;
            }
        }
        return A;
        
        // 3, 1, 2, 4, 5, 6, ==> 2, 1, 3, 4, 5, 6
        // largest number smaller from right
        
    }
};
