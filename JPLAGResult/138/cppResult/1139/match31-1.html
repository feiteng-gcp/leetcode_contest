class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size();
        for (int i = n-1; i > 0; --i) {
            for (int j = i-1; j >= 0; --j) {
                if (A[j] > A[i]) {
                    if (!helper(A, j+1, i))
                        swap(A[j], A[i]);
                    return A;
                }
            }
        }
        return A;
    }
    bool helper(vector<int> &A, int s, int e) {
        if (!(e-s) || e-s == 1)
            return false;
        for (int i = e-1; i > s; --i) {
            for (int j = i-1; j >= s; --j) {
                if (A[j] > A[i]) {
                    if (!helper(A, j+1, i)) {
                        swap(A[j], A[i]);
                    }
                    return true;
                }
            }
        }
        return false;
    }
};
