class Solution {
public:
    bool twoDiff(vector<int> &a, vector<int> &b) {
        int cnt = 0;
        for (int i = 0; i < a.size(); ++i) {
            if (a[i] != b[i]) cnt += 1;
        }
        return cnt == 2;
    }
    vector<int> prevPermOpt1(vector<int>& A) {

        bool yes = true;
        for (int i = 0; i < A.size()-1; ++i) {
            if (A[i] > A[i+1]) {
                yes = false;
                break;
            }
        }
        if (yes) {
            return A;
        }
        auto copy = A;
        while (next_permutation(A.begin(), A.end(), [](int a, int b){return a>b;})) {
            if (twoDiff(A, copy)) {
                return A;
            }
        }
        return A;
    }
};
