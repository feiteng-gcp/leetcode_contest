class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size(), i, j, k;
        
        for (i = n-2; i >= 0; --i) {
            if (A[i] > A[i+1]) break;
        }
        
        if (i == -1) return A;
        
        k = lower_bound(A.begin() + i + 1, A.end(), A[i]) - A.begin() - 1;
        
        swap(A[k], A[i]);
        
        return A;
    }
};
