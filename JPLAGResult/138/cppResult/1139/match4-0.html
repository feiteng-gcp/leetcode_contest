class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size();
        for (int i = n - 1; i >= 0; --i) {
            // find j with min value. If tied, choose the latest
            int k = -1;
            for (int j = i + 1; j < n; ++j) {
                if (A[j] < A[i] and (k == -1 or A[j] > A[k])) {
                    k = j;
                }
            }
            if (k > i) {
                swap(A[i], A[k]);
                break;
            }
        }
        
        return A;
    }
};
