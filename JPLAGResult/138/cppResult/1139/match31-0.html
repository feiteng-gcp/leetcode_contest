class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n=A.size();
        for(int i=n-1;i>=0;i--)
        {
            for(int j=n-1;j>i;j--)
            {
                if (A[i]>A[j])
                {
                    swap(A[i],A[j]);
                    return A;
                }
            }
        }
        return A;
    }
};
