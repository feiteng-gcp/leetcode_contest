class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int N = A.size();
        
        int idx = -1;
        for (int i = N - 1; i >= 1; i--) {
            if (A[i - 1] > A[i]) {
                idx = i - 1;
                break;
            }
        }
        
        if (idx == -1) {
            return A;
        }
        
        for (int i = N - 1; i >= idx; i--) {
            if (A[i] < A[idx]) {
                swap(A[i], A[idx]);
                return A;
            }
        }
        
        return A;
    }
};
