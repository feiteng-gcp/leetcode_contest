class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n=A.size();
        int l=n-1;
        for (int i=n-1;i>0; i--)
        {
            if (A[i]<A[i-1])
			{
                l=i-1;
				break;
			}
        }
        vector<int> res(A);
        if (l==n-1)
            return res;

        for (int i=n-1;i>l; i--)
        {
            if (A[i]<A[l])
            {
                swap(res[i], res[l]);
                return res;
            }
        }
        return res;
    }
};
