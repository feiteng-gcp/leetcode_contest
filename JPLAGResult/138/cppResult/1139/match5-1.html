class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        //printf("======\n");

        int j = A.size()-1;
        //printf("j=%d\n",j);
        for (int i=A.size()-2; i>=0; i--) {
            //printf("%d %d >? %d\n",i,A[i],A[i+1]);
            if (A[i] > A[i+1]) {
                j = i;
                //printf("j=%d\n",j);
                break;
            }

        }
        //printf("%d\n", j);
        if (j == A.size()-1) return A;
        int k = j+1;
        for (int i=j+1; i<A.size(); i++) {
            if (A[i] < A[j]) {
                k = i;
            }
        }
        swap(A[j],A[k]);
        return A;
    }
};
