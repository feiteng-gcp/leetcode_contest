class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size();
        if (n <= 1) return A;
       
        map<int, int> vis;
        vis[A[n - 1]] = n - 1;
        for (int i = n - 2; i >= 0; -- i) {
            auto it = vis.lower_bound(A[i]);
            if (it != begin(vis)) {
                -- it;
                swap(A[i], A[it->second]);
                break;
            }
            
            vis[A[i]] = i;
        }
        
        return A;
    }
};
