class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int startIndex = -1;
        for (int i = 0; i < A.size() - 1; i++) {
            if (A[i] > A[i + 1]) { //
                startIndex = i;
            }
        }
        if (startIndex == -1) return A;
        int endIndex = -1;
        for (int i = startIndex + 1; i < A.size(); i++) {
            if (A[i] < A[startIndex]) {
                endIndex = i;
            }
        }
        swap(A[startIndex], A[endIndex]);
        return A;
    }
};
