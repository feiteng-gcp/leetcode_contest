class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size();
        int i = -1, j = -1;
        for (int k = 0; k < n; ++k) {
            if (k < n-1 && A[k] > A[k+1]) {
                i = k;
                j = k+1;
            }
            if (i >= 0 && A[k] < A[i] && A[k] >= A[j])
                j = k;
        }
        if (i >= 0)
            swap(A[i], A[j]);
        return A;
    }
};
