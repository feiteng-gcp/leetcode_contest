class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {

        int pivot = -1;
        for(int i = A.size() - 1; i > 0; i--){
            if(A[i - 1] > A[i]){
                pivot = i - 1;
                break;
            }
        }
        if(pivot >= 0){
            for(int i = A.size() - 1; i > pivot; i--){
                if(A[pivot] > A[i]){
                    swap(A[pivot], A[i]);
                    break;
                }
            }
        }
        return A;
    }
};
