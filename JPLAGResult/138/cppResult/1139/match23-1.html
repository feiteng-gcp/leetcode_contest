class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
      map<int, int> bst;
      for (int i = A.size() - 1; i >= 0; --i) {
        auto it = bst.lower_bound(A[i]);
        if (it != bst.begin()) {
          --it;
          swap(A[i], A[it->second]);
          break;
        }
        bst[A[i]] = i;
      }
      return A;
    }
};
