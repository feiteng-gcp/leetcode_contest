class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n=A.size(), i, j;
        for (i=n-1; i>0; --i) {
            if (A[i-1]>A[i]) break;
        }
        if (i==0) return A;
        for (j=i; j<n-1; ++j) {
            if (A[j+1]>=A[i-1]) break;
        }
        swap(A[i-1], A[j]);
        return A;
    }
};
