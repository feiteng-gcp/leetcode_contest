class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size()-1;
        for(int i = n; i > 0; --i){
            if(A[i-1]>A[i]){
                int idx = lower_bound(A.begin()+i, A.end(), A[i-1]) - A.begin() - 1;
                swap(A[i-1], A[idx]);
                break;
            }
        }
        return A;
    }
};
