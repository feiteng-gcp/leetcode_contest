class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int N = A.size();
        int lloc = -1;
        for (int i = 0; i + 1 < N; i++)
        {
            if (A[i] > A[i+1])
                lloc = i;
        }
        if (lloc == -1)
            return A;
        int ca = N - 1;
        while (A[ca] >= A[lloc])
            ca--;
        swap (A[ca], A[lloc]);
        return A;
    }
};
