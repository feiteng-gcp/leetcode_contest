class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size(), i = n-1;
        while (i > 0 && A[i] >= A[i-1]) i--;
        if (i == 0) return A;
        int j = n-1;
        while (A[j] >= A[i-1]) j--;
        swap(A[i-1], A[j]);
        return A;
    }
};
