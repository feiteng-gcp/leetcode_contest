class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {

        int len = (int)A.size(), pos = len - 2;
    while(pos >= 0 && A[pos] <= A[pos + 1]) pos--;
    if(pos < 0) return A;
    int p2 = pos;
    while(p2 + 1 < len && A[p2 + 1] < A[pos]) p2++;
    swap(A[pos], A[p2]);
    return A;
    }
};
