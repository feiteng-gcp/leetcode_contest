class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        map<int, int> previous;
        for (int i = A.size() - 1; i >= 0; --i) {
            // cout << "Looking at " << A[i] << endl;
            auto largest_smaller = previous.lower_bound(A[i]);
            if (largest_smaller != previous.cbegin()) {
                // cout << "It has previous greater " << *prev(largest_smaller);
                int idx = prev(largest_smaller)->second;
                // cout << "swapping " << A[i] << " and " << A[idx] << endl;
                swap(A[i], A[idx]);
                return A;
            }
            auto existing = previous.find(A[i]);
            if (existing == previous.cend()) previous[A[i]] = i;
        }
        return A;
    }
};
