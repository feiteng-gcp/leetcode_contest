class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int L=A.size();
        int last = A[L-1], first = -1;
        for (int i=L-2; i>=0; --i){
            if (A[i] <= last) { last = A[i]; continue; }
            first = i; break;
        }
        if (first == -1) return A;
        for (int i=L-1; i>first; --i) {
            if (A[i] >= A[first]) continue;
            // if (i>=1 && A[i-1] == A[i]) continue;
            swap(A[i], A[first]);
            break;
        }
        return A;
    }
};
