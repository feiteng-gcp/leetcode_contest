class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int idx=-1;
        int n = A.size();
        for(int i=0; i<n-1; i++) {
            if(A[i]>A[i+1])
                idx=i;
        }
        if(idx==-1)
            return A;
        int j=idx+1;
        for(int i=idx+2; i<n; i++) {
            if(A[idx]>A[i])
                j=i;
        }
        int t=A[idx];
        A[idx]=A[j];
        A[j]=t;
        return A;
    }
};
