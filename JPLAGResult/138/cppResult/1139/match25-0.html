class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int n = A.size(), i;
        for (i = n-2; i >= 0; --i) {
            if (A[i] > A[i+1]) break;
        }
        if (i < 0) return A;
        
        int j;
        for (j = n-1; j > i; --j) {
            if (A[j] < A[i]) break;
        }
        iter_swap(A.begin()+i, A.begin()+j);
        return A;
    }
};
