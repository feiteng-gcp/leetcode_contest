class Code {
    int val;
    int count;
    Code(int val, int count) {
        this.val = val;
        this.count = count;
    }
}
class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        Map<Integer, Code> map = new HashMap<>();
        int n = barcodes.length;
        for (int v: barcodes) {
            Code c = map.getOrDefault(v, new Code(v, 0));
            c.count ++;
            
map.put(v, c);
        }
        PriorityQueue<Code> q = new PriorityQueue<Code>((c1, c2) -> c2.count - c1.count);
        for (Code c: map.values()) {
            q.offer(c);
        }
        int cur = 0;
        while (!q.isEmpty()) {
            Code c = q.poll();
            if (cur == 0 || barcodes[cur - 1] != c.val) {
                barcodes[cur] = c.val;
                c.count --;
                if (c.count != 0) {
                    q.offer(c);
                }
            } else 
{
                Code c2 = q.poll();
                barcodes[cur] = c2.val;
                c2.count --;
                if (c2.count != 0) {
                    q.offer(c2);
                }
                q.offer(c);
            }
            cur ++;
        }
        return barcodes;
    }
}
