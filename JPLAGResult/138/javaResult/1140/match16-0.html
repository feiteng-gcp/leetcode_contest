class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((o1, o2) -> (o2[1] - o1[1]));
        Arrays.sort(barcodes);
        for (int i = 0; i < barcodes.length; ++i) {
            int times = 1;
            while (i + 1 < barcodes.length && barcodes[i + 1] == barcodes[i]) {
                ++times;
                ++i;
            }
            System.out.println(barcodes[i] + " " + times);
            pq.offer(new int[]{barcodes[i], times});
        }
        int i = 0;
        while (!pq.isEmpty()) {
            int[] a = pq.poll();
            if (i == 0 || a[0] != barcodes[i - 1]) {
                barcodes[i] = a[0];
                --a[1];
                if (a[1] != 0) pq.offer(a);
            } else {
                int[] b = pq.poll();
                barcodes[i] = b[0];
                --b[1];
                if (b[1] != 0) pq.offer(b);
                pq.offer(a);
            }
            ++i;
        }
        return barcodes;
    }
}
