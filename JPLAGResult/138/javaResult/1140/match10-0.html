class Solution {
    private class Pair {
        int bc;
        int count;
        public Pair(int bc, int count) {
            this.bc = bc;
            this.count = count;
        }
    }
    
    public int[] rearrangeBarcodes(int[] barcodes) {
        PriorityQueue<Pair> pq = new PriorityQueue<>(new Comparator<Pair>(){
            @Override
            public int compare(Pair p1, Pair p2) {
                if (p1.count == p2.count) return 0;
                return p1.count < p2.count ? 1 : -1;
            }
        });
        
        Map<Integer, Integer> map = new HashMap<>();
        for (int bc : barcodes) {
            map.put(bc, map.getOrDefault(bc, 0) + 1);
        }
        
        for (Integer key : map.keySet()) {
            pq.offer(new Pair(key, map.get(key)));
        }
        
        int[] res = new int[barcodes.length];
        int idx = 0;
        int prev = -1;
        while (!pq.isEmpty()) {
            Pair p = pq.poll();
            Pair next = null;
            if (prev == p.bc) {
                next = pq.poll();
            }
            
            if (next != null) {
                res[idx] = next.bc;
                next.count--;
                if (next.count != 0) {
                    pq.offer(next);
                }
            } else {
                res[idx] = p.bc;
                p.count--;
            }
            
            if (p.count != 0) {
                pq.offer(p);
            }
            
            prev = res[idx++];
        }
        
        return res;
    }
}
