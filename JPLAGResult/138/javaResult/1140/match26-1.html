class Solution {
    static class Item implements Comparable<Item> {
        int value;
        int count;
        public Item(int value, int count) {
            this.value = value;
            this.count = count;
        }
        public int compareTo(
Item other) {
            return other.count - count;
        }
    }
    public int[] rearrangeBarcodes(int[] barcodes) {
        Map<Integer, Integer> map = new HashMap<>();
        for(int i = 0; i < barcodes.length; ++i) {
            int count = map.getOrDefault(barcodes[i], 0);
            map.put(barcodes[i], count+1);
        }
        
        PriorityQueue<Item> pq = new PriorityQueue<Item>();
        for(Map.Entry<Integer, Integer> entry: map.entrySet()) {
            pq.add(new Item(entry.getKey(), entry.getValue()));
        }
        
        int[] ret = new int[barcodes.length];
        for(int i = 0; i < ret.length; i++) {
            Item a = pq.poll();
            if(i == 0 || a.value != ret[i-1]) {
                ret[i] = a.value;
                if(a.count > 1) {
                    pq.add(new Item(a.value, a.count-1));
                }
            }
            else { // a.value == ret[i-1]
                Item b = pq.poll(); // it must exist 
                ret[i] = b.value;
                if(b.count > 1) {
                    pq.add(new Item(b.value, b.count-1));
                }
                
pq.add(a);
            }
        }
        return ret;
    }
}
