class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        int n = barcodes.length;
        Map<Integer, Integer> map = new HashMap<>();
        for (int b : barcodes) {
            map.put(b, 
map.getOrDefault(b, 0) + 1);
        }
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> b[1] - a[1]);
        for (int m : map.keySet()) {
            pq.add(new int[]{m, map.get(m)});
        }
        int[] res = new int[n];
        
int index = 0;
        while (!pq.isEmpty()) {
            int[] first = pq.poll();
            if (index == 0 || first[0] != res[index-1]) {
                res[index] = first[0];
                index++;
                if (--first[1] > 0) {
                    pq.add(first);
                }
            } else {
                int[] second = pq.poll();
                res[index] = second[0];
                index++;
                if (--second[1] > 0) {
                    pq.add(second);
                }
                pq.add(first);
            }
        }
        return res;
    }
}
