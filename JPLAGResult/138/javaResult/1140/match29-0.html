class Solution {
    public int[] rearrangeBarcodes(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        PriorityQueue<Node> pq = new PriorityQueue<>((n1, n2) -> {
            if (n1.count == n2.count) {
                return Integer.compare(n1.value, n2.value);
            } else {
                return Integer.compare(n2.count, n1.count);
            }
        });
        for (int key : map.keySet()) {
            Node node = new Node(key, map.get(key));
            pq.offer(node);
        }
        int[] res = new int[nums.length];
        int index = 0;
        while (!pq.isEmpty()) {
            Node first = pq.poll(), second = pq.poll();
            if (index > 0 && res[index - 1] == first.value) {
                res[index++] = second.value;
                second.count--;
            } else {
                res[index++] = first.value;
                first.count--;
            }
            if (first.count > 0) pq.offer(first);
            if (second != null && second.count > 0) pq.offer(second);
        }
        return res;
    }
    class Node {
        int value;
        int count;
        public Node(int value, int count) {
            this.value = value;
            this.count = count;
        }
    }
}
