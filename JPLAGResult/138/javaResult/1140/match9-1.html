class Solution {
    private class Pair {
        int barcode, times;
        public Pair(int barcode, int times) {
            this.barcode = barcode;
            this.times = times;
        }
    }
    public int[] rearrangeBarcodes(int[] barcodes) {
        int n = barcodes.length;
        if(n == 1)  return barcodes;
        Map<Integer, Integer> map = new HashMap<>();
        for(int barcode : barcodes) map.put(barcode, map.getOrDefault(barcode, 0) + 1);
        PriorityQueue<Pair> pQueue = new PriorityQueue<Pair>(new Comparator<Pair>() {
            public int compare(Pair a, Pair b) {
                return b.times - a.times;
            }
        });
        for(int barcode : map.keySet()) pQueue.add(new Pair(barcode, map.get(barcode)))
;
        int[] ans = new int[n];
        int idx = 0;
        while(pQueue.size() >= 2) {
            Pair p1 = pQueue.poll();
            Pair p2 = pQueue.poll();
            ans[idx++] = p1.barcode;
            ans[idx++] = p2.barcode;
            p1.times -= 1;
            p2.times -= 1;
            if(p1.times > 0)    pQueue.add(p1);
            if(p2.times > 0)    pQueue.add(p2);
        }
        if(pQueue.size() > 0)   ans[idx++] = pQueue.poll().barcode;
        return ans;
    }
}
