class Solution {
    public class HeapElement implements Comparable<HeapElement> {
        public int number;
        public int occurrence;
        public HeapElement(int a, int b) {
            number = a;
            occurrence = b;
        }
        @Override
        public int compareTo(HeapElement other) {
            return this.occurrence - other.occurrence;
        }
        
    }
    public int[] rearrangeBarcodes(int[] barcodes) {
        
HashMap <Integer, Integer> occurrences = new HashMap<Integer, Integer>();
        for (int currentNumber : barcodes) {
            if (occurrences.containsKey(currentNumber)) occurrences.put(currentNumber, occurrences.get(currentNumber) + 1);
            else occurrences.put(currentNumber, 1);
        }
        PriorityQueue<HeapElement> heap = 
new PriorityQueue<HeapElement>();
        for (int currentNumber : occurrences.keySet()) {
            HeapElement h = new HeapElement(currentNumber, occurrences.get(currentNumber) * -1);
            heap.add(h);
        }
        Arrays.fill(barcodes, -1);
        HeapElement lastElement = null;
        for (int i = 0; i < barcodes.length; i++) {
            HeapElement currentMin = heap.peek();
            if (currentMin != lastElement) {
                currentMin = heap.poll();
                lastElement = currentMin;
                currentMin.occurrence++;
                barcodes[i] = currentMin.number;
                heap.add(currentMin);
            }
            else {
                HeapElement backup = heap.poll();
                HeapElement toAdd = heap.poll();
                toAdd.occurrence++;
                lastElement = toAdd;
                barcodes[i] = toAdd.number;
                heap.add(toAdd);
                heap.add(backup);
            }
        }
        return barcodes;
    }
}
