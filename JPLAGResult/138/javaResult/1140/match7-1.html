class Solution {
        public int[] rearrangeBarcodes(int[] barcodes) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->b[0]-a[0]);
        int [] temp = new int[10001];
        for(int n:barcodes){
            temp[n]++;
        }
        for(int i=0;i<temp.length;++i){
            if(temp[i]>0){
                pq.add(new int[]{temp[i],i});
            }
        }
        int[] ans = new int[barcodes.length];
        int p = 0;
        while(pq.size()>0){
            int[] curr = pq.poll();
            if(p==0||curr[1]!=ans[p-1]){
                curr[0]--;
                ans[p++] = curr[1];
                if(curr[0]>0)
                    pq.add(curr);
                continue;
            }
            List<int[]> ls = new ArrayList<>();
            while(curr[1]==ans[p-1]){
                ls.add(curr);
                curr=pq.poll();
            }
            curr[0]--;
            ans[p++] = curr[1];
            if(curr[0]>0)
                pq.add(curr);
            for(int i=0;i<ls.size();++i){
                pq.add(ls.get(i));
            }

        }
        return ans;
    }
}
