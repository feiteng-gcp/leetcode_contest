class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        HashMap<Integer, Integer> count = new HashMap<>();
        for (int num : barcodes) {
            count.put(num, count.getOrDefault(num, 0) + 1);
        }
        // <num, count>
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((a1, a2) -> {
            return a2[1] - a1[1];
        });
        for (Integer key : count.keySet()) {
            pq.add(new int[] {key, count.get(key)});
        }
        
        for (int i = 0; i < barcodes.length; i++) {
            int[] item = pq.poll();
            if (i == 0) {
                barcodes[i] = item[0];
                item[1]--;
                if (item[1] > 0) {
                    pq.add(item);
                }
                continue;
            }
            
            // need to pop one more number
            if (item[0] == barcodes[i - 1]) 
{
                int[] nextItem = pq.poll();
                barcodes[i] = nextItem[0];
                nextItem[1]--;
                if (nextItem[1] > 0) {
                    pq.add(nextItem);
                }
                pq.add(item);
            } else {
                barcodes[i] = item[0];
                item[1]--;
                if (item[1] > 0) {
                    pq.add(item);
                }
            }
        }
        
        return barcodes;
    }
}
