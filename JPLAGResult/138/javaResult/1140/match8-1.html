class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        int n = barcodes.length;
        TreeSet<Integer> ts = new TreeSet<Integer>();
        int[] ans = new int[n];
        
int[] counts = new int[10001];
        for (int i = 0; i < n; i++) {
            counts[barcodes[i]]++;
        }
        for (int i = 0; i <= 10000; i++) {
            if (counts[i]>0) {
                ts.add(counts[i]*10001+i);
            }
        }
        int index = 0;
        while (ts.size() > 1) {
            ans[index++] = ts.last()%10001;
            ans[index++] = ts.lower(ts.last())%10001;
            int one = ts.pollLast();
            int two = ts.pollLast();
            
if (one>20001)
               ts.add(one-10001);
            if (two>20001)
               ts.add(two-10001);
        }
        if (ts.size()==1)
            ans[index++] = ts.last()%10001;
        return ans;
    }
}
