/*
Insert one by one
MaxHeap <barcode, count>
*/
class Solution {
  public int[] rearrangeBarcodes(int[] barcodes) {
    Map<Integer, Integer> count = new HashMap<>();
    for (int b : barcodes) {
      count.put(b, count.getOrDefault(b, 0) + 1);
    }
    PriorityQueue<int[]> maxHeap = new PriorityQueue<>((a, b)->(b[1] - a[1]));
    for (Integer b : count.keySet()) {
      maxHeap.offer(new int[] {b, count.get(b)});
    }
    int i = 0;
    while (maxHeap.size() > 1) {
      int[] max = maxHeap.poll();
      barcodes[i++] = max[0];
      max[1]--;
      int[] secondMax = maxHeap.poll();
      barcodes[i++] = secondMax[0];
      secondMax[1]--;
      if (max[1] > 0) {
        maxHeap.offer(max);
      }
      if (secondMax[1] > 0) {
        maxHeap.offer(secondMax);
      }
    }
    if (!maxHeap.isEmpty()) {
      barcodes[i] = maxHeap.poll()[0];
    }
    return barcodes;
  }
}
