class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        if (barcodes == null || barcodes.length == 0) {
            return barcodes;
        }
        
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        
        for (int i = 0; i < barcodes.length; i++) {
            map.put(barcodes[i], map.getOrDefault(barcodes[i], 0) + 1);
        }
        
        PriorityQueue<Pair> queue = new PriorityQueue<Pair>((a, b) -> a.count == b.count ? a.value - b.value : b.count - a.count);
        
        for (int key : map.keySet()) {
            queue.offer(new Pair(key, map.get(key)));
        }
        
        int[] result = new int[barcodes.length];
        int index = 0;
        
        while (!queue.isEmpty()) {
            Pair first = queue.poll();
            
            if (queue.isEmpty()) {
                result[index] = first.value;
                break;
            }
            
            Pair second = queue.poll();
            
            result[index++] = first.value;
            result[index++] = second.value;
            
            first.count--;
            second.count--;
            
            if (first.count > 0) {
                queue.offer(first);
            }
            
            if (second.count > 0) {
                queue.offer(second);
            }
        }
        
        return result;
    }
    
    class Pair {
        int value;
        int count;
        
        public Pair(int value, int count) {
            this.value = value;
            this.count = count;
        }
    }
}
