class Solution {
    
class Boob{
        int x, y;
        public Boob(int x, int y){
            this.x = x;
            this.y = y;
        }
    }
    public int[] rearrangeBarcodes(int[] barcodes) {
        PriorityQueue<Boob> pq = new PriorityQueue<>((a,
b)->(b.y-a.y));
        Map<Integer, Integer>map = new HashMap<>();
        for (int i : barcodes) {
            map.put(i,map.getOrDefault(i, 0)+1);
        }
        for (int i : map.keySet()){
            pq.add(new Boob(i, map.get(i)));
        }
        
        int [] res = new int [barcodes.length];
        int idx = 0;
        Boob prev = null;
        while (!pq.isEmpty()) {
            Boob cur = pq.poll();
            System.out.println(cur.x + " " + cur.y);
            cur.y--;
            res[idx++] = cur.x;
            if (prev != null && prev.y > 0) {
                pq.add(prev);
            }
            if (cur.y > 0){
                prev = cur;
            } else {
                prev = null;
            }
        }
        return res;
    }
}
