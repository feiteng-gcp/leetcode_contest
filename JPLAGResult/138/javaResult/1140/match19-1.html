class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        
Map<Integer, Integer> map = new HashMap<>();
        for(int code : barcodes) {
            map.put(code, map.getOrDefault(code, 0) + 1);
        }
        
        PriorityQueue<Item> queue = new PriorityQueue<>(barcodes.length, (Item a, Item b)-> b.freq - a.freq);
        for(Map.Entry<Integer, Integer> entry : map.entrySet()) {
            queue.offer(new Item(entry.getKey(), entry.getValue()));
        }
        
        Item prev = new Item(0, -1);
        int index = 0;
        while(queue.size() != 0) {
            Item curr = queue.poll();
            barcodes[index++] = curr.num;
            if(prev.freq > 0) {
                queue.offer(prev);
            }
            curr.freq--;
            prev = curr;
        
}
        return barcodes;
    }
    class Item {
        int freq;
        int num;
        public Item(int num, int freq) {
            this.num = num;
            this.freq = freq;
        }
    }
}
