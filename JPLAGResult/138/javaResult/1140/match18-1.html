class Solution {
    public int[] rearrangeBarcodes(int[] b) {
        
        Queue<Pair> q = new PriorityQueue<>();
        int n = b.length;
        
Map<Integer, Integer> map = new HashMap<>(); 
        for (int i = 0; i < n; i++) {
            map.put(b[i], map.getOrDefault(b[i], 0) + 1);
        }
        
for (int k : map.keySet()) {
            q.add(new Pair(k, map.get(k)));
        }
        int[] ret = new int[n];
        int idx = 0; 
        while (q.size() > 1) {
            
            Pair pa = q.poll();
            Pair pb = q.poll(); 
            ret[idx++] = pa.num;
            pa.cnt--;
            if (pa.cnt > 0) {
                q.add(pa);
            }
            ret[idx++] = pb.num;
            pb.cnt--;
            if (pb.cnt > 0) {
                q.add(pb);
            }
        }
        if (!q.isEmpty()) {
            ret[idx] = q.poll().num;
        }
        
return ret;
    }
    
    private class Pair implements Comparable<Pair>{
        int num;
        int cnt; 
        public Pair(int num, int cnt) {
            this.num = num;
            this.cnt = cnt;
        }
        
        public int compareTo(Pair o) {
            if (o.cnt == cnt) {
                return num - o.num;
            }
            return o.cnt - cnt; 
        }
    }
}
