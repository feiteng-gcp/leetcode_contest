class Solution {
    class FreqData {
        final int val;
        final int freq;

        FreqData(int val, int freq) {
            this.val = val;
            this.freq = freq;
        }
    }
    
    public int[] rearrangeBarcodes(int[] barcodes) {
        Map<Integer, Integer> counts = new HashMap<>();
        for (int b : barcodes) {
            counts.put(b, counts.getOrDefault(b, 0) + 1);
        }

        List<FreqData> freqData = new ArrayList<>();
        
        for (Map.Entry<Integer, Integer> e : counts.entrySet()) {
            freqData.add(new FreqData(e.getKey(), e.getValue()));
        }
        freqData.sort((d1, d2) -> d2.freq - d1.freq);
        
        
        List<List<Integer>> groups = new ArrayList<>();
        FreqData top = freqData.get(0);
        

        for (int i = 0; i < top.freq; i++) {
            groups.add(new ArrayList<>());
        }
        
        int g = 0;
        for (FreqData fd : freqData) {
            for (int i = 0; i < fd.freq; i++) {
                groups.get(g).add(fd.val);
                g = (g + 1) % groups.size();
            }
        }
        
        
        int[] result = new int[barcodes.length];
        int current = 0;
        
        for (List<Integer> group : groups) {
            for (Integer e : group) {
                result[current++] = e;
            }
        }
        return result;
    }
}
