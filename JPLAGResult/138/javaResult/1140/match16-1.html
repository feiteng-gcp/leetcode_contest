class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        int index = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((a,b)-> b[1]-a[1]);
        Map<Integer, Integer> map = new HashMap();
        for(int num: barcodes) map.put(num, map.getOrDefault(num, 0)+1);
        for(int num: map.keySet()) pq.offer(new int[]{num, map.get(num)});
        
while(!pq.isEmpty()){
            int[] most = pq.poll();
            if(index==0 || barcodes[index-1]!=most[0]){
                barcodes[index++] = most[0];   
                if(--most[1]!=0) pq.offer(most);
            }
            else{
                int[] less = pq.poll();
                barcodes[index++] = less[0];
                if(--less[1]!=0) pq.offer(less);
                pq.offer(most);
            }
        }
        return barcodes;
    }
}
