class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        int MOD = 10001;
        Arrays.sort(barcodes);
        int length = barcodes.length;
        int[] counts = new int[MOD];
        for (int num : barcodes)
            counts[num]++;
        Integer[] newArray = new Integer[length];
        for (int i = 0; i < length; i++) {
            int num = barcodes[i];
            int count = counts[num];
            newArray[i] = count * MOD + num;
        }
        Arrays.sort(newArray, new Comparator<Integer>() {
        	public int compare(Integer num1, Integer num2) {
				return num2 - num1;
			}
        });
        for (int i = 0; i < length; i++)
            barcodes[i] = newArray[i] % MOD;
        int[] ret = new int[length];
        if (length % 2 == 0) {
            int halfLength = length / 2;
            for (int i = 0; i < halfLength; i++) {
                ret[i * 2] = barcodes[i];
                ret[i * 2 + 1] = barcodes[i + halfLength];
            }
        } else {
            int halfLength = length / 2;
            for (int i = 0; i <= halfLength; i++)
                ret[i * 2] = barcodes[i];
            for (int i = 0; i < halfLength; i++)
                ret[i * 2 + 1] = barcodes[i + halfLength + 1];
        }
        return ret;
    }
}
