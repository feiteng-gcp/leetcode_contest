		@Override
		public boolean equals(Object o) {

			Pear p = (Pear) o;
			return (this.k==p.k && this.count==p.count);

		@Override
		public int hashCode() {
			return 42*this.k + 69*this.count;
		}
		@Override
		public int compareTo(Pear p) {
			return (p.count-this.count);
		}
