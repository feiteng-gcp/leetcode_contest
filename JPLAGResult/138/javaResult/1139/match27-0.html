class Solution {
    public int[] prevPermOpt1(int[] A) {
        if(A.length < 2) return A;
        int i = A.length - 1;
        while(i > 0 && A[i] >= A[i-1]) i--;
        if(i == 0) return A;
        
        int val = -1, index = -1;
        for(int j = A.length-1; j >= i; j--){
            if(A[j] > val && A[j] < A[i-1]){
                val = A[j];
                index = j;
            }
        }
        swap(A, i-1, index);
        return A;
    }
    
    private void swap(int[] arr, int i, int j){
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
}
