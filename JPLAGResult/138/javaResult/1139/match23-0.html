class Solution {
    public int[] prevPermOpt1(int[] A) {
        Stack<Integer> stack = new Stack<>();
        for(int i=A.length-1; i>=0; i--){
            if(i==A.length-1 || A[i]<=A[i+1]){
                stack.push(i);
            }else{
                int j=stack.pop();
                while(!stack.isEmpty() && A[i]>A[stack.peek()]){
                    j = stack.pop();
                }
                int tmp = A[i];
                A[i] = A[j];
                A[j] = tmp;
                return A;
            }
        }
        return A;
    }
}
