class Solution {
    public int[] prevPermOpt1(int[] A) {
        int n = A.length;
        
for (int i = n-1; i >= 0; i--) {
            int swap = -1;
            int sn = 0;
            for (int j = i+1; j < n; j++) {
                if (A[j]<A[i]) {
                    if (A[j] > sn) {
                        sn = A[j];
                        swap = j;
                    }
                }
            }
            if (swap > -1) {
                int t = A[i];
                A[i] = A[swap];
                A[swap] = t;
                return A;
            }
        }
        return A;
    }
}
