class Solution {
    public int[] prevPermOpt1(int[] A) {
        
        int n = A.length;
        
        for(int i = n-2; i >= 0; i--){
            
            if(A[i] > A[i+1]){
                
                int j = i+1;
                
                while(j < n && A[j] < A[i]){
                    
                    j++;
                    
                }
              
                
int tmp = A[i];
                A[i] = A[j-1];
                A[j-1] = tmp;
                
                return A;
                
            }
            
        }
        
        return A;
        
    }
}
