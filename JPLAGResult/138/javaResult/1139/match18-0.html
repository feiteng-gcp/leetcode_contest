class Solution {
    public int[] prevPermOpt1(int[] A) {
        int left=-1;
        for(int i=A.length-1;i>0;i--){
            if(A[i]<A[i-1]){
                left=i-1;
                int j=A.length-1;
                while(A[j]>=A[left])
                    j--;
                int temp=A[j];
                A[j]=A[left];
                A[left]=temp;
                break;
            }
        }
        return A;
    }
}
