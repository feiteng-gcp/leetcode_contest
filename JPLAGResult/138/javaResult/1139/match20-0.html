class Solution {
    
    public int[] prevPermOpt1(int[] A) {
        int min = Integer.MAX_VALUE;
        for (int i = A.length - 1; i >= 0; i--) {
            int cur = A[i];
            if (cur <= min) {
                min = Math.min(cur, min);
            } else {
                // cur > min
                // for i + 1, find most closest number to cur
                int maxi = -1;
                for (int j = i + 1; j < A.length; j++) {
                    if (A[j] < cur) {
                        if (maxi == -1 || A[j] >= A[maxi]) {
                            maxi = j;
                        }
                    }
                }
                // swap i and maxi
                int bak = A[i];
                A[i] = A[maxi];
                A[maxi] = bak;
                return A;
            }
        }
        return A;
    }
}
