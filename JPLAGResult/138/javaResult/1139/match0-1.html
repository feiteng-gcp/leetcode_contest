class Solution {
        public int[] prevPermOpt1(int[] A) {
        boolean change =true;
        for(int i=1;i<A.length;++i){
            if(A[i-1]>A[i]){
                change = false;
                break;
            }
        }
        if(change) return A;
        int[] ans = A.clone();
        TreeMap<Integer,Integer> tmap = new TreeMap<>();
        for(int i=A.length-1;i>=0;--i){
            Map.Entry<Integer,Integer> last = tmap.lowerEntry(A[i]);
            if(last==null){
                if(!tmap.containsKey(A[i])){
                    tmap.put(A[i],i);
                }
                continue;
            }

            int idx = last.getValue();
            ans[i] = ans[idx];
            ans[idx] = A[i];
            return ans;
        }
        return ans;



    }
}
