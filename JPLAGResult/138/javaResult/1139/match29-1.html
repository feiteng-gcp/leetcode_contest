class Solution {
    public int[] prevPermOpt1(int[] A) {
        int len = A.length;
        
for (int i = len-2; i >= 0; i--){
            int max = -1;
            for (int j = i+1; j < len; j++){
                if (A[i] > A[j]) {
                    if (max == -1)
                        max = j;
                    else if (A[max] < A[j]) 
                        max = j
;
                }
            }
            if (max != -1){
                int temp = A[max];
                A[max] = A[i];
                A[i] = temp;
                return A;
            }
        }
        return A;
    }
}
