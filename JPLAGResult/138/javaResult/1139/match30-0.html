class Solution {
    public int[] prevPermOpt1(int[] a) {


        TreeMap<Integer, Integer> lastIndex = new TreeMap<>();
        for (int i = a.length - 1; i >= 0; i--) {
            Map.Entry<Integer, Integer> lowerEntry = lastIndex.lowerEntry(a[i]);
            if (lowerEntry != null) {
                int[] result = a.clone();
                int tmp = result[i];
                result[i] = result[lowerEntry.getValue()];
                result[lowerEntry.getValue()] = tmp;
                return result;
            }
            lastIndex.put(a[i], i);
        }

        return a;

    }

    public static void main(String[] args) {
        Solution sol = new Solution();

        int[] res;
        res = sol.prevPermOpt1(new int[] { 3, 2, 1 });
        res = sol.prevPermOpt1(new int[] { 1, 1, 5 });
        res = sol.prevPermOpt1(new int[] { 1, 9, 4, 6, 7 });

        System.out.println();
    }
}
