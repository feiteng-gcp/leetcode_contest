class Solution {
    public int[] prevPermOpt1(int[] A) {
        int N = A.length;
        int i = N-2;
        while (i >= 0 && A[i] <= A[i+1]) i--;
        if (i < 0) return A;
        int j = i+1;
        while (j+1 < N && A[i] > A[j+1]) j++;
        int t = A[i];
        A[i] = A[j];
        A[j] = t;
        return A;
    }
}
