class Solution {
    public int[] prevPermOpt1(int[] a) {
        int l = a.length;
        int b[] = Arrays.copyOf(a,l);
        Arrays.sort(b);
        if(Arrays.equals(b,a)){
            return a;
        }
        
        for(int i=l-1;i>=0;--i){
            
            int r = -1;
            for(int j=i+1;j<l;++j){
                if(a[j]<a[i]){
                    
                    if(r==-1||a[r]<=a[j]){
                        r = j;
                    }
                    
                }
            }
            if(r>=0){
                
                int t = a[i];
                a[i] = a[r];
                a[r] = t;
                return a;
                
            }
            
            
        }
        return a;
        
        
        
    }
}
