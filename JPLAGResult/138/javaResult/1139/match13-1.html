public class Solution {
	public int[] prevPermOpt1(int[] A) {
		int N = A.length;
		if (N < 2) return A;
		int start = N 
- 1;
		while (start > 0 && A[start - 1] <= A[start]) start--;
		if (start == 0) return A;
		int last = start - 1;
		while (start + 1 < N && A[start + 1] < A[last]) start++;
		int tmp = A[start];
		A[start] = A[last];
		A[last] = tmp;
		return A;
	}
}
