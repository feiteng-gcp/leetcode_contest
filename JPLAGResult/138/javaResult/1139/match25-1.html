class Solution {
    public int[] prevPermOpt1(int[] A) {
        
for(int i = A.length-2; i >= 0; --i) {
            int index = -1;
            for(int j = A.length-1; j > i; --j) {
                if(A[j] < A[i]) {
                    if(index == -1) {
                        index = j;
                    }
                    else if(A[j] > A[index]) {
                        index = j;
                    
}
                }
            }
            if(index != -1) {
                int t = A[i];
                A[i] = A[index];
                A[index] = t;
                return A;
            }
        }
        return A;
    }
}
