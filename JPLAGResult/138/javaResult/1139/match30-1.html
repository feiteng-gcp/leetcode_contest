class Solution {
    public int[] prevPermOpt1(int[] A) {
        TreeMap<Integer, Integer> map = new TreeMap();
        for(int i=A.length-1;i>=0;i--){
            Integer lower = map.lowerKey(A[i]);
            if(lower!=null){
                int index = map.get(lower);
                int tmp = A[index];
                A[index] = A[i];
                A[i] = tmp;
                
return A;
            }
            map.put(A[i], i);
        }
        return A;
    }
}
