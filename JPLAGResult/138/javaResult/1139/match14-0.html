class Solution {
    public int[] prevPermOpt1(int[] A) {
        int length = A.length;
        int beginIndex = -1, endIndex = -1;
        for (int i = 1; i < length; i++) {
            int num1 = A[i - 1];
            int num2 = A[i];
            if (num1 > num2) {
                beginIndex = i - 1;
                endIndex = i;
                continue;
            }
            if (beginIndex >= 0) {
                if (A[endIndex] <= num2 && A[beginIndex] > num2) {
                    endIndex = i;
                }
            }
        }
        if (beginIndex >= 0) {
            int temp = A[beginIndex];
            A[beginIndex] = A[endIndex];
            A[endIndex] = temp;
        }
        return A;
    }
}
