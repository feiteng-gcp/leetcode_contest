class Solution {
public int[] prevPermOpt1(int[] A) {
		int n = A.length;
		for (int i = n - 2; i >= 0; i--) {
			if (A[i] > A[i+1]) {
				int j = i+1;
				for (; j < n-1; j++) {
					if (A[j+1] >= A[i]) break;
				}
				int tmp = A[i];
				A[i] = A[j];
				A[j] = tmp;
				break;
			}
		}
		return A;
	}
}
