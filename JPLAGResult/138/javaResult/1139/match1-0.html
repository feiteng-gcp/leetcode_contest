class Solution {
    public int[] prevPermOpt1(int[] A) {
        int len=A.length;
        for(int i=len-2;i>=0;i--){
            if(A[i+1]<A[i]){
                for(int j=len-1;j>i;j--){
                    if(A[j]<A[i]){
                        int tmp=A[j];
                        A[j]=A[i];
                        A[i]=tmp;
                        return A;
                    }
                }
            }
        }
        return A;
    }
}
