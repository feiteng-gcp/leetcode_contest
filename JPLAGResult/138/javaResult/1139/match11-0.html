class Solution {
    public int[] prevPermOpt1(int[] A) {
        int len = A.length, idx = A.length - 2;
        while(idx >= 0 && A[idx] <= A[idx + 1])
            idx--;
        if(idx < 0)
            return A;
        int cIdx = len - 1;
        while(cIdx >= idx && A[cIdx] >= A[idx])
            cIdx--;
        int tmp = A[cIdx];
        A[cIdx] = A[idx];
        A[idx] = tmp;
        return A;
    }
}
