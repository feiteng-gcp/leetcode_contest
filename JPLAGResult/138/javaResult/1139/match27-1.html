class Solution {
    public int[] prevPermOpt1(int[] A) {
        if (A == null || A.length < 2) return A;
        int currentIndex = A.length - 1;
        while (currentIndex > 0 && A[currentIndex] >= A[currentIndex - 1]) currentIndex--;
        if (currentIndex == 0) return A;
        if (currentIndex == A.length - 1) {
            swap(A, A.length - 1, A.length - 2);
            return A;
        }
        currentIndex--;
        int indexToSwap = A.length - 1;
        while (A[indexToSwap] >= A[currentIndex]) indexToSwap--;
        
swap(A, currentIndex, indexToSwap);
        return A;
    }
    
    private void swap(int[] A, int from, int to) {
        int temp = A[from];
        A[from] = A[to];
        A[to] = temp;
    }
}
