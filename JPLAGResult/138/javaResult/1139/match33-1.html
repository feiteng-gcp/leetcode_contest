class Solution {
    public int[] prevPermOpt1(int[] nums) {
        int index = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] > nums[i + 1]) {
                index = i;
                break;
            }
        }
        if (index == -1) {
            return nums;
        }
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[index] > nums[i]) {
                
int tmp = nums[index];
                nums[index] = nums[i];
                nums[i] = tmp;
                break;
            }
        }
        return nums;
    }
}
