class Solution {
    public int[] prevPermOpt1(int[] A) {
        boolean flag = true;
        for (int i=0; i<A.length - 1; i++) {
            if (A[i] > A[i+ 1]) 
                flag = false;
        }
        
        if (flag == true)
            return A;
        
        for (int i = A.length - 1; i >= 0; i --) {
            int max = -1;
            int index = -1;
            for (int j = i + 1; j < A.length; j++) {
                if (A[i] > A[j]) {
                    if (A[j] >= max) { max = A[j]; index = j;}
                }
                // System.out.format("%d, %d\n", max, index);
            }
            
            if (max != -1) {
                int temp = A[i]; A[i] = A[index]; A[index] = temp;
                return A;
            }
        }
        return A;
    }
}
