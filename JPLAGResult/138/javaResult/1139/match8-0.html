class Solution {
    public int[] prevPermOpt1(int[] A) {
        int n = A.length;
        if(n == 1)  return A;
        int idx1 = n-2;
        while(idx1 >= 0 && A[idx1] <= A[idx1+1])  idx1 -= 1;
        if(idx1 < 0)    return A;
        int idx2 = idx1 + 1;
        while(idx2+1 < n && A[idx2] <= A[idx2+1] && A[idx2+1] < A[idx1])    idx2 += 1;
        // System.out.println("index1: " + idx1 + "; index2: " + idx2);
        int tmp = A[idx1];
        A[idx1] = A[idx2];
        A[idx2] = tmp;
        
        return A;
    }
}
