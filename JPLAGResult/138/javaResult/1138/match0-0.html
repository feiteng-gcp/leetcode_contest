class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int ans=0,len=grumpy.length;
        for(int i=0;i<len;i++){
            if(grumpy[i]==0) ans+=customers[i];
        }
        int val=0,tmp=0;
        for(int i=0;i<X;i++){
            if(grumpy[i]==1) tmp+=customers[i];
        }
        val=tmp;
        for (int left=1,right=X;right<len;left++,right++){
            if(grumpy[left-1]==1) tmp-=customers[left-1];
            if(grumpy[right]==1) tmp+=customers[right];
            val=Math.max(val,tmp);
        }
        return ans+val;
    }
}
