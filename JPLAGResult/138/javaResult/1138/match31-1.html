class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int n = customers.length;
        int sum = 0;
        for(int i = 0; i < n; i++){
            if(grumpy[i] == 0){
                sum += customers[i];
            }
        }
        int tmp = 0;
        for(int i = 0; i < X; i++){
            if(grumpy[i] == 1){
                tmp += customers[i];
            }
        
}
        int bonus = tmp;

        for(int i = 1; i + X - 1 < n; i++){
            if(grumpy[i-1] == 1){
                tmp -= customers[i-1];
            }
            if(grumpy[i+X-1] == 1){
                tmp += customers[i+X-1];
            
}
//            System.out.println(tmp);
            bonus = Math.max(tmp, bonus);
        }

        return bonus + sum;

    }
}
