class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int n = customers.length;
        int[] prefixCustomers = new int[n+1], prefixGrumpy = new int[n+1];
        for(int i = 0; i < n; ++i) {
            prefixCustomers[i+1] = prefixCustomers[i] + customers[i];
            prefixGrumpy[i+1] = prefixGrumpy[i] + (grumpy[i] == 0 ?  customers[i] : 0);
        }
        
int max = prefixGrumpy[n];
        for(int i = 0; i <= n-X; ++i) {
            max = Math.max(max, prefixGrumpy[i]-prefixGrumpy[0] + prefixCustomers[i+X]-prefixCustomers[i] + prefixGrumpy[n]-prefixGrumpy[i+X]);
        }
        return max;
    }
}
