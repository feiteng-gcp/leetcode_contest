class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int res = 0;
        for (int i = 0; i < customers.length; ++i) {
            if (grumpy[i] == 0) {
                res += customers[i];
            }
        }
        int curr = 0;
        for (int i = 0; i < X; ++i) {
            if (grumpy[i] == 1) {
                curr += customers[i];
            }
        }
        
int maxCover = curr;
        //System.out.println(curr);
        for (int i = X; i < customers.length; ++i) {
            if (grumpy[i - X] == 1) {
                curr -= customers[i - X];
            }
            if (grumpy[i] == 1) {
                curr += customers[i];
            
}
            maxCover = Math.max(curr, maxCover);
            //System.out.println(curr);
        }
        return res + maxCover;
    }
}
