class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int len=customers.length;
        int ans=0;
        for(int i=0;i<len;i++){
            if(grumpy[i]==0)
                ans+=customers[i];
        }
        int g=0;
        int m=0;
        for(int i=0;i<X;i++){
            if(grumpy[i]==1)
                g+=customers[i];
        }
        m=g;
        for(int i=X;i<len;i++){
            if(grumpy[i-X]==1)
                g-=customers[i-X];
            if(grumpy[i]==1)
                g+=customers[i];
            m=Math.max(m,g);
        }
        return m+ans;
    }
}
