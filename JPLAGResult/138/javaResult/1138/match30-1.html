public class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int N = customers.length;
        int base = 0, maxAdd = 0, delta = 0;
        for (int i = 0; i < N; ++i) {
            if (grumpy[i] == 0) base += customers[i];
        }
        for (int i = 0; i < X; ++i) {
            if (grumpy[i] == 1) delta += customers[i];
        }
        maxAdd = delta;
        for (int i = X; i < N; ++i) {
            if (grumpy[i] == 1) delta += customers[i];
            if (grumpy[i - X] == 1) delta -= customers[i - X]
;
            maxAdd = Math.max(maxAdd, delta);
        }
        return base + maxAdd;
    }
}
