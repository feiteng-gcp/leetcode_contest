class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int or = 0;
        for(int i=0;i<customers.length;++i){
            if(grumpy[i]==0){
                or+=customers[i];
            }
        }
        int ans = or;
        if(X==0)
            return ans;
        int left = 0;
        for(int i=0;i<customers.length;++i){
            if(grumpy[i]==1){
                or+=customers[i];
            }
            if(i-left>=X){
                if(grumpy[left]==1)
                    or-=customers[left];
                ++left;
            }
            ans = Math.max(or,ans);
        }
        return ans;
    }
}
