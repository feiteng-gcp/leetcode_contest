class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int n = customers.length;
        int[] ori = new int[n + 1], all = new int[n + 1];
        for (int i = 0; i < n; ++i) { // [i, j] = p[j + 1] - p[i]
            ori[i + 1] = ori[i] + (grumpy[i] == 0 ? customers[i] : 0);
            all[i + 1] = customers[i] + all[i];
        }
        if (X >= n) return all[n];
        int res = Integer.MIN_VALUE;
        for (int i = 0; i <= n - X; ++i) {
            res = Math.max(ori[i] + all[i + X] - all[i] + ori[n] - ori[i + X], res);
        }
        return res;
    }
}
