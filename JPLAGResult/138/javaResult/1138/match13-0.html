class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int initialSatisfied = 0;
        int minutes = customers.length;
        
        for(int i=0;i<minutes;i++){
            if(grumpy[i]==0){
                initialSatisfied+=customers[i];
            }
        }
        
        int grumpyWindowCustomers = 0;
        
        for(int i=0;i<X;i++){
            if(grumpy[i]==1){
                grumpyWindowCustomers += customers[i];
            }
        }
        
        int maxGrumpyWindowCustomers = grumpyWindowCustomers;
        int beginIndex = 0;
        for(int i=X;i<minutes;i++,beginIndex++){
            if(grumpy[i]==1){
                grumpyWindowCustomers += customers[i];
            }
            if(grumpy[beginIndex]==1){
                grumpyWindowCustomers -= customers[beginIndex];
            }
            maxGrumpyWindowCustomers = Math.max(grumpyWindowCustomers,maxGrumpyWindowCustomers);
        }
        return (maxGrumpyWindowCustomers + initialSatisfied);
    }
}
