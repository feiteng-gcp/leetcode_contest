/*
Calculate the affecter customers
Slinding X to find max
*/
class Solution {
  public int maxSatisfied(int[] customers, int[] grumpy, int X) {
    int customerTotal = 0;
    int notSatisfiedTotal = 0;
    int currentConverted = 0;
    int maxConverted = 0;
    for (int i = 0; i < customers.length; i++) {
      customerTotal += customers[i];
      if (i >= X) {
        currentConverted -= grumpy[i - X] * customers[i - X];
      }
      currentConverted += grumpy[i] * customers[i];
      maxConverted = Math.max(maxConverted, currentConverted);
      notSatisfiedTotal += grumpy[i] * customers[i];
    }
    return customerTotal - (notSatisfiedTotal - maxConverted);
  }
}
