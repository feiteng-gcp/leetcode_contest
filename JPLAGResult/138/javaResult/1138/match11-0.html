class Solution {
    public int maxSatisfied(int[] customers, int[] count, int x) {
        int sum = 0;
        for (int i = 0; i < customers.length; i++) {
            if (count[i] == 0) {
                sum += customers[i];
            }
        }
        for (int i = 0; i < x; i++) {
            if (count[i] == 1) {
                sum += customers[i];
            }
        }
        int res = sum;
        for (int i = x; i < customers.length; i++) {
            if (count[i - x] == 1) {
                sum -= customers[i - x];
            }
            if (count[i] == 1) {
                sum += customers[i];
            }
            res = Math.max(res, sum);
        }
        return res;
    }
}
