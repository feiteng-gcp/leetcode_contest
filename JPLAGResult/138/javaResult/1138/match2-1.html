class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int length = customers.length;
        int[] customersSatisfied = new int[length];
        int[] customersUnsatisfied = new int[length];
        int ret = 0;
        for (int i = 0; i < length; i++) {
            int customer = customers[i];
            if (grumpy[i] == 0) {
                customersSatisfied[i] = customer;
                ret += customer;
            } else
                customersUnsatisfied[i] = customer;
        }
        int maxSum = maxSubarraySum(customersUnsatisfied, X);
        ret += maxSum;
        return ret;
    }

    public int maxSubarraySum(int[] nums, int X) {
        int length = nums.length;
        int sum = 0;
        for (int i = 0; i < X; i++)
            sum += nums[i];
        int maxSum = sum;
        for (int i = 0, j = X; j < length; i++, j++) {
            sum -= nums[i];
            sum += nums[j];
            maxSum = Math.max(maxSum, sum);
        }
        return maxSum;
    }
}
