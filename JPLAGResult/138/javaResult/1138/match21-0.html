class Solution {
 public int maxSatisfied(int[] customers, int[] grumpy, int X) {
	    	int n = customers.length;
	    	int sum1=0;
	        for(int i=0;i<n;++i) {
	        	if(grumpy[i]==0) {
	        		sum1+=customers[i];
	        	}
	        }
	        int temp=sum1;
	        int max=0;
	        for(int i=0;i<X;++i) {
	        	if(grumpy[i]==1) {
	        		temp+=customers[i];
	        	}
	        }
	        max=temp;
	        n-=X;
	        //n-=1;//减去第一次
	        for(int i=0;i<n;++i) {
	        	if(grumpy[i]==1) {
	        		temp-=customers[i];
	        	}
	        	if(grumpy[i+X]==1) {
	        		temp+=customers[i+X];
	        	}
	        	max = Math.max(max, temp);
	        }
	        return max;
	    }
    
}
