class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int X) {
        int ori = 0, max = 0;
        int len = customers.length;
        for (int i = 0; i < len; ++i) {
        	if (grumpy[i] == 0) {
        		ori += customers[i];
        	}
        }
        int mem = 0;
        for (int i = 0; i < len; ++i) {
        	if (grumpy[i] == 1) {
        		mem += customers[i];
        	}
        	if (i >= X && grumpy[i - X] == 1) {
        		mem -= customers[i - X];
        	}
        	max = Math.max(max, mem);
        }
        return ori + max;
    }
}
