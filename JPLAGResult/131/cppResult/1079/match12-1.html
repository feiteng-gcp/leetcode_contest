/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int sumRootToLeaf(TreeNode* root) {
        int ans = 0;
        sumRootToLeaf(root, 0, ans);
        return ans;
    }
    void sumRootToLeaf(TreeNode *node, int cur, int &ans) {
        const int MOD = 1000000007;
        if (!node) return;
        cur = (((cur * 2) % MOD) + node->val) % MOD;
        if (!node->left and !node->right) {
            ans = (ans + cur) % MOD;
            return;
        }
        sumRootToLeaf(node->left, cur, ans);
        sumRootToLeaf(node->right, cur, ans);
    }
};
