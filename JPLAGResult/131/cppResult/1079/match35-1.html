/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int sumRootToLeaf(TreeNode* root) {
        queue<TreeNode*> nod;
        queue<int> tot;
        TreeNode* tmp;
        int now, cnt, ans = 0;
        const int MOD = 1e9 + 7;
        nod.push(root);
        tot.push(root -> val);
        while(!nod.empty())
        {
            tmp = nod.front();

            now = tot.front();
            now %= MOD;
            nod.pop();
            tot.pop();
            cnt = 0;
            if(tmp -> left != NULL)
            {
                cnt++;
                nod.push(tmp -> left);
                tot.push(2 * now + tmp -> left -> val);
            }
            if(tmp -> right != NULL)
            {
                cnt++;
                nod.push(tmp -> right);
                tot.push(2 * now + tmp -> right -> val);
            }
            if(cnt == 0)
            {
                ans += now;
                ans %= MOD;
            }
        }
        return ans;
    }
};
