/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int sumRootToLeaf(TreeNode* root) {
        int res = 0;
        DFS(root, 0, res);
        return res;
    }
    void DFS(TreeNode *p, int curvalue, int &res) {
        int nowvalue = curvalue + p->val;
        nowvalue %= 1000000007;
        if (!p->left && !p->right) {
            res += nowvalue;
            res %= 1000000007;
            return;
        }
        if (p->left) DFS(p->left, (2 * nowvalue) % 1000000007, res);
        if (p->right) DFS(p->right, (2 * nowvalue) % 1000000007, res);
    }
};
