/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int ans;
    int mod;
    

    void f(TreeNode* r, int sum) {
        if (!r) return;
        
        sum = (sum * 2 + r->val) % mod;
        if (!r->left && !r->right) {
            ans = (ans + sum) % mod;
            return;
        }
        
        f(r->left, sum);
        f(r->right, sum);
    }
    
    int sumRootToLeaf(TreeNode* root) {
        ans = 0;
        mod = 1E9+7;
        f(root,0);
        return ans;
    }
};
