/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    long ans;
    
    int sumRootToLeaf(TreeNode* root) {
        ans=0;
        dfs(root,0);
        return ans;
        
    }
    
    void dfs(TreeNode* root, long pre)
    {
        if(root==NULL)return;
        pre=(pre*2+root->val)%1000000007;
        if(root->left==NULL&&root->right==NULL)
        {
            ans+=pre;
            ans=ans%1000000007;
            return;
        }
        if(root->left!=NULL)dfs(root->left,pre);
        if(root->right!=NULL)dfs(root->right,pre);
    }
    
    
};
