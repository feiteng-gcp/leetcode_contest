/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    long long MOD = 1000000007;
    long long res = 0;
    void DFS(TreeNode* cur, long long num) {

        long long next = ((num << 1) + cur->val) % MOD;
        if (!cur->left && !cur->right) {
            res = (res + next) % MOD;
            return;
        }
        if (cur->left) {
            DFS(cur->left, next);
        }
        if (cur->right) {
            DFS(cur->right, next);
        }
    }
public:
    int sumRootToLeaf(TreeNode* root) {
        long long num = 0;
        if (!root) {
            return 0;
        }
        DFS(root, num);
        return res;
    }
};
