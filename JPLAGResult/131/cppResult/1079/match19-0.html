/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
const int mod = 1000000007;
class Solution {
private:
    int ret;
    void dfs(TreeNode* root, int fa){
        int cur = ((fa << 1) + root->val)%mod;
        if(root->left==NULL&&root->right==NULL){
            ret=(ret+cur)%mod;
            return;
        }
        if(root->left!=NULL){
            dfs(root->left, cur);
        }
        if(root->right!=NULL){
            dfs(root->right, cur);
        }
    }
public:
    int sumRootToLeaf(TreeNode* root) {
        ret=0;
        dfs(root, 0);
        return ret;
    }
};
