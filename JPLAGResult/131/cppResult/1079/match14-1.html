/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    void dfs(TreeNode* p, int x)
    {
        if (p == NULL) return;
        x = (x * 2 + p->val) % mod;
        if (p->left == NULL && p->right == NULL)
        {
            ret = (ret + x) % mod;
            return;
        }
        dfs(p->left, x);
        dfs(p->right, x);
    }
    
    int sumRootToLeaf(TreeNode* root) {
        ret = 0;
        dfs(root, 0);
        return ret;
    }
    
private:
    int ret;
    int mod = 1000000007;
};
