/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int sumRootToLeaf(TreeNode* root) {
        int mod=1000000007, res=0;
        helper(res,mod,0,root);
        return res;
    }
    
    void helper(int& res, int& mod, int cur, TreeNode* root){
        if(root==NULL) return;
        cur=(2*cur+(root->val))%mod;
        if(root->left==NULL && root->right==NULL) res=(res+cur)%mod;
        helper(res,mod,cur,root->left);
        helper(res,mod,cur,root->right);
        return;
    }
};
