/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    const static LL MOD = 1E9 + 7;
    LL total;
    

    void dfs(TreeNode* root, LL prefix) {
        if(!root) return;
        LL delta = (prefix * 2 + root->val) % MOD;
        if(!root->left and !root->right) {
            total = (total + delta) % MOD;
        }
        dfs(root->left, delta);
        dfs(root->right, delta);
    }
    
public:
    int sumRootToLeaf(TreeNode* root) {
        total = 0;
        dfs(root, 0);
        return int(total % MOD);
    }
};
