/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int sumRootToLeaf(TreeNode* root) {
        int sum = 0;
        PreOrder(root, 0, sum);
        return sum;
    }
    
    void PreOrder(TreeNode* root, int parent_val, int& sum){
        if(root == NULL) return;
        
        // find own value
        int self_val = (parent_val * 2) % 1000000007 + root->val;
        
        // add to sum if it's a leaf node
        if(!root->left && !root->right) sum = (sum + self_val) % 1000000007;
        
        PreOrder(root->left, self_val, sum);
        PreOrder(root->right, self_val, sum);
    }
};
