ll ans=0;

void dfs(TreeNode* t, ll x) {
  x+=t->val;
  x%=MAX;
  bool f=0;
  if(t->right!=NULL) {
    dfs(t->right,x*2);
    f=1;
  }
  if(t->left!=NULL) {
    dfs(t->left,x*2);
    f=1;
  }
  if(!f) {
    ans+=x;
    ans%=MAX;
  }
}

class Solution {
public:
  int sumRootToLeaf(TreeNode* root) {
    ans=0;
    dfs(root,0);
    return ans;
  }
};
