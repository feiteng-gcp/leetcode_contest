class Solution {
public:
    const int mod = 1e9 + 7;
    int total = 0;
    
    int sumRootToLeaf(TreeNode* root) {
        if (!root) return 0;
        traverse(root, 0);
        return total;
    }
    
    void traverse(TreeNode *node, int n) {
        if (!node) return;
        
        n = (n * 2 + node->val) % mod;
        if (node->left == node->right) {
            total = (total + n) % mod;
            return;
        }
        
        traverse(node->left, n);
        traverse(node->right, n);
    }
};
