/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int res, mod;
    void func(auto* a, int v) {
        if (a->left==NULL and a->right==NULL) {
            res=(res+(2L*v+a->val))%mod;
            return;
        }
        if (a->left!=NULL)
            func(a->left, (2*v+a->val)%mod);
        if (a->right!=NULL)
            func(a->right, (2*v+a->val)%mod);
    }
    int sumRootToLeaf(TreeNode* root) {
        res=0;
        mod=1e9+7;
        func(root, 0);
        return res;
    }
};
