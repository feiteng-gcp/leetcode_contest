/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int sumRootToLeaf(TreeNode* root) {
        queue<pair<TreeNode*, int>> q;
        if(!root){
            return 0;
        }
        int total = 0;
        q.push(make_pair(root, 0));
        while(!q.empty()){
            auto p = q.front();
            q.pop();
            //int val = p.second*2 + root->val;
            int val = p.second*2 + p.first->val;
            val%=1000000007;
            //cout<<root->val<<" "<<val<<endl;
            //cout<<p.first->val<<" "<<val<<endl;
            //leaf node.
            if(!p.first->left&&!p.first->right){
                //cout<<"leaf:"<<val<<endl;
                total = (total + val)%1000000007;
            }
            else{
                if(p.first->left){
                    q.push(make_pair(p.first->left, val));
                }
                if(p.first->right){
                    q.push(make_pair(p.first->right, val));
                }
            }
        }
        return total;
    }
};
