/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    long result = 0, m = pow(10, 9) + 7;;
    void help(TreeNode* root, long n) {
        if (!root) {
            return;
        }
        
        n = (n + n + root->val) % m;
        
        if (!root->left && !root->right) {
            result = (result + n) % m;
            return;
        }
        
        if (root->left) {
            help(root->left, n);
        }
        
        if (root->right) {
            help(root->right, n);
        }
    }
public:
    int sumRootToLeaf(TreeNode* root) {
        help(root, 0);
        return result;
    }
};
