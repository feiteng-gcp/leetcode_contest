/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    int sum = 0;
    int mod = 1000000007;
public:
    void dfs(TreeNode *rt, int t) {
        if (!rt)
            return;
        t = t * 2 + rt->val;

        t %= mod;
        if (!rt -> left && !rt -> right) {
            sum += t;
            sum %= mod;
            return;
        }
        dfs(rt->left, t);
        dfs(rt->right, t);
    }
    int sumRootToLeaf(TreeNode* root) {
        sum = 0;
        dfs(root,0);
        return sum;
    }
};
