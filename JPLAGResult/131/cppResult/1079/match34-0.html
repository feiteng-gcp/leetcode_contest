/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

const int MOD = 1e9 + 7;

class Solution {
    long long answer;
    void go(TreeNode* cur, long long sum) {
        if (cur == nullptr) return;
        sum = (sum * 2 + cur->val) % MOD;
        if (cur->left == nullptr && cur->right == nullptr) {
            answer = (answer + sum) % MOD;
        }
        go(cur->left, sum);
        go(cur->right, sum);
    }
public:
    int sumRootToLeaf(TreeNode* root) {
        answer = 0;
        go(root, 0);
        return answer;
    }
};
