/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ret=0,mod=1000000007;
    void helper(TreeNode* root,int x)
    {
        x*=2;x+=root->val;x%=mod;
        if(root->left==NULL&&root->right==NULL)
        {
            ret+=x;
            ret%=mod;
            return;
        }
        if(root->left!=NULL)
            helper(root->left,x);
        if(root->right!=NULL)
            helper(root->right,x);
    }
    int sumRootToLeaf(TreeNode* root) {
        helper(root,0);
        return ret;
    }
};
