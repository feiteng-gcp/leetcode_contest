/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans = 0;
    const int M = 1000000007;
    void rec(TreeNode* r, int now) {
        now = (now * 2 + r->val) % M;
        if(r->left == NULL && r->right == NULL) {

            (ans += now) %= M;
        }
        else {
            if(r->left != NULL)  rec(r->left, now);
            if(r->right != NULL) rec(r->right, now);
        }
    }
    
    int sumRootToLeaf(TreeNode* root) {
        ans = 0;
        rec(root, 0);
        return ans;
    }
};
