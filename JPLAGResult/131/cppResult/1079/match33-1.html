string to_string(string s){return s;}

template<class T>
string to_string(vector<T>v){
	string res;
	res+="{";
	rep(i,v.size()){
		if(i)res+=",";
		res+=to_string(v[i]);
	}
	res+="}";
	return res;
}

template<class T>
ostream&operator<<(ostream&os,vector<T>v){
	cout<<to_string(v);
	return os;
}

ostream&operator<<(ostream&os,TreeNode*t){
	map<int,int>mp;
	function<void(TreeNode*,int)>dfs=[&](TreeNode*t,int k){
		if(!t)return;
		mp[k]=t->val;
		dfs(t->left,k*2+1);
		dfs(t->right,k*2+2);
	};
	dfs(t,0);
	int Max=0;
	for(auto p:mp)Max=max(Max,p.first);
	vector<string>ans(Max+1,"null");
	for(auto p:mp)ans[p.first]=to_string(p.second);
	os<<ans;
	return os;
}

class Solution {
public:
	int ans=0;
	void dfs(TreeNode*t,int sum){
		sum=((sum<<1)+t->val)%MOD;

		if(!(t->left)&&!(t->right)){
			(ans+=sum)%=MOD;
			return;
		}
		if(t->left)dfs(t->left,sum);
		if(t->right)dfs(t->right,sum);
	}
    int sumRootToLeaf(TreeNode* root) {
        ans=0;
        dfs(root,0);
        return ans;
    }
};

#ifdef DEBUG
int main(){
	Solution sol;
	TreeNode*root=new TreeNode(1);
	root->left=new TreeNode(0);
	root->right=new TreeNode(1);
	root->left->left=new TreeNode(0);
	root->left->right=new TreeNode(1);
	root->right->left=new TreeNode(0);
	root->right->right=new TreeNode(1);

	auto res=sol.sumRootToLeaf(root);
	cout<<res<<endl;
}
