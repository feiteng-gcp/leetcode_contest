/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    long long ans=0,modd=pow(10,9)+7;
    void work(TreeNode* root,long long now){

        now%=modd;
        if(root->left==NULL&&root->right==NULL){
            ans+=now*2+root->val;
            ans%=modd;
            return;
        }
        if(root->left!=NULL) work(root->left,now*2+root->val);
        if(root->right!=NULL) work(root->right,now*2+root->val);
    }
    int sumRootToLeaf(TreeNode* root) {
        work(root,0);
        return ans;
    }
};
