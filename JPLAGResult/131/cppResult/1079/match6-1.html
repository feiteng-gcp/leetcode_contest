/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

const long long MODULO = 1000000007;
long long ans;
class Solution {
public:
    void calc(TreeNode* i, long long num) {

        if (i->left == NULL && i->right == NULL) {
            ans = (ans + num) % MODULO;
            return;
        }
        if (i->left) calc(i->left, (num * 2 + i->left->val) % MODULO);
        if (i->right) calc(i->right, (num * 2 + i->right->val) % MODULO);
    }
    int sumRootToLeaf(TreeNode* root) {
        ans = 0;
        calc(root, root->val);
        return ans;
    }
};
