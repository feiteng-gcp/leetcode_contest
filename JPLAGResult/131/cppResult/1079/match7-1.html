/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int M = 1000000007;
    long long ans = 0;

    void dfs(TreeNode *p, long long cur) {
        cur = ((cur << 1) + p->val) % M;
        if (p->left == NULL && p->right == NULL) {
            ans = (ans + cur) % M;
            return;
        }
        if (p->left) dfs(p->left, cur);
        if (p->right) dfs(p->right, cur);
    }
    int sumRootToLeaf(TreeNode* root) {
        dfs(root, 0);
        return ans;
    }
};
