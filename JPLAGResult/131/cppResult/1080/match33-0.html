class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res_vec;
        for(string& q: queries) {
            int i = 0, j = 0;
            bool res = true;
            while(i < len(q)) {
                if(isupper(q[i]) and j == len(pattern)) {
                    res = false;
                    break;
                }
                if(q[i] == pattern[j]) {
                    i++; j++;
                } else if(islower(q[i])) {
                    i++;
                } else {
                    break;
                }
            }
            res_vec.push_back(res and j == len(pattern));
        }
        return res_vec;
    }
};
