class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        int n = queries.size();
        vector<bool> res(n, false);
        for(int i = 0; i < n; ++i)
            res[i] = judge(queries[i], pattern);
        return res;
    }
    bool judge(string a, const string &b)
    {
        int j = 0;
        for(int i = 0; i < a.size(); ++i)
        {
            if(a[i] == b[j])
                ++j;
            else if(a[i] >= 'A' && a[i] <= 'Z')
                return false;
        }
        return j == b.size();
    }
};
