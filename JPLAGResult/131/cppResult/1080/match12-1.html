class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res;
        for (string& query : queries) {
            res.push_back( isMatch(query, pattern) );
        }
        return res;
    }
private:
    bool isMatch(string& query, string& pattern) {
        int n = query.size();
        int m = pattern.size();
        if (m > n) return false;
        if (n == m) return query == pattern;
        int i = 0;

        for (char c : pattern) {
            // find a match in query
            bool hasMatch = false;
            while (i < n) {
                if (c == query[i]) {
                    hasMatch = true;
                    ++i;
                    break;
                }
                if (isupper(query[i])) return false;
                ++i;
            }
            if (!hasMatch) return false;
        }
        while (i < n) {
            if (isupper(query[i])) return false;
            ++i;
        }
        return true;
    }
};
