class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> ret;
        int n = queries.size(), m = pattern.length();
        ret.resize(n, 0);
        for (int i = 0; i < n; i++)
        {
            int l = queries[i].length(), j = 0, k = 0;
            for (; j < l; j++)
            {
                if (queries[i][j] == pattern[k]) k++;
                else if (queries[i][j] >= 'A' && queries[i][j] <= 'Z')
                {
                    k = -1;
                    break;
                }
            }
            if (k == m) ret[i] = 1;
        }
        return ret;
    }
};
