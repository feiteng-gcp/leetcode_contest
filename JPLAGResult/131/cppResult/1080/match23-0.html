class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        int n = queries.size();
        vector<bool> res (n, false);
        for (int i = 0; i < n; ++i)
            if (match(queries[i], pattern))
                res[i] = true;
        return res;
    }
    bool match(string a, string b) {
        int i = 0, j = 0;
        while (i < a.size() && j < b.size()) {
            if (a[i] == b[j]) {
                i++;
                j++;
            } else if (islower(a[i])) {
                i++;
            } else 
                return false;
        }
        if (j != b.size()) return false;
        while (i < a.size()) {
            if (isupper(a[i++]))
                return false;
        }
        return true;
    }
};
