class Solution {
public:
    bool helper(string q, string p){
        if((int)q.length()<(int)p.length()){
            return false;
        }
        
        int p1 = 0;

        int q1 = 0;
        
        while(p1<p.length()&&q1<q.length()){
            if(p[p1]==q[q1]){
                p1++;
                q1++;
            }
            else{
                if(isupper(q[q1])){
                    return false;
                }
                else{
                    q1++;
                }

            }
        }
        
        if(p1<p.length()){
            return false;
        }
        
        for(;q1<q.length();q1++){
            if(isupper(q[q1])){
                return false;
            }
        }
        
        return true;
    }
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res;
        
        for(int i = 0;i<queries.size();i++){
            res.push_back(helper(queries[i],pattern));
        }
        
        return res;
    }
};
