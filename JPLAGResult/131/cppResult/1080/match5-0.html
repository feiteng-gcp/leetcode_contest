class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> ans;
        for(auto &q:queries)ans.push_back(match(pattern,q));
        return ans;
    }
    
    bool match(string& pattern,string& q)
    {
        int i=0,j=0;
        while(i<pattern.size()&&j<q.size())
        {
            if(pattern[i]==q[j])
            {
                i++,j++;
            }
            else
            {
                if(!(q[j]>='a'&&q[j]<='z'))return false;
                j++;
            }
        }
        if(i!=pattern.size())return false;
        for(;j<q.size();j++)if(!(q[j]>='a'&&q[j]<='z'))return false;
        return true;
    }
};
