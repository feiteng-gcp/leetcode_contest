class Solution {
public:
    bool helper(string query, string pattern) {
        int pos = 0;
        for(int j = 0; j < query.length(); ++j) {
            if(pos == pattern.length()) {
                if(query[j] > 'Z')
                    continue;
                else return false;
            }
            if(pattern[pos] <= 'Z') {
                if(query[j] != pattern[pos]) {
                    if(query[j] > 'Z')
                        continue;
                    else
                        return false;
                }else {
                    pos++;
                }
            }else {
                if(query[j] != pattern[pos]) {
                    if(query[j] > 'Z')
                        continue;
                    else
                        return false;
                }else {
                    pos++;
                }
            }
        }
        if(pos == pattern.length()) return true;
        else return false;
    }
    
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> result;
        for(int i = 0; i < queries.size(); ++i) {
            result.push_back(helper(queries[i], pattern));
        }
        return result;
    }
};
