class Solution {
public:
    
    bool judge(string s, string pattern) {
        int i = 0, j = 0;
        while (i < s.size() && j < pattern.size()) {
            if (s[i] == pattern[j]) {
                i++;
                j++;
            } else {
                if (s[i] <= 'Z' && s[i] >= 'A') {
                    return false;
                }
                i++;
            }
        }
        if (j < pattern.size()) {
            return false;
        } 
        for (;i < s.size();i++) {
            if (s[i] <= 'Z' && s[i] >= 'A') {
                return false;
            }
        }
        return true;
    }
    
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> v;
        for (int i = 0;i < queries.size();i++) {
            string s = queries[i];
            
            v.push_back(judge(s, pattern));
        }
        return v;
    }
};
