class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res;
        for (auto query: queries) {
            int p = 0, i = 0;
            while (p < pattern.size() && i < query.size()) {
                if (pattern[p] == query[i]) {
                    p++;
                    i++;
                }
                else if (query[i] >= 'a' && query[i] <= 'z') {
                    i++;
                }
                else
                    break;
                if (p == pattern.size()) {
                    while (i < query.size() && query[i] >= 'a' && query[i] <= 'z')
                        i++;
                }
            }
            if (p == pattern.size() && i == query.size())
                res.push_back(true);
            else
                res.push_back(false);
        }
        return res;
    }
};
