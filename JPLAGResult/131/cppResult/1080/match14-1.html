class Solution {
    bool solve(string q, string p) {

        int i = 0, j = 0;
        while (i < q.length() && j < p.length()) {
            if (q[i] == p[j]) {
                ++i;
                ++j;
            } else {
                if (q[i] >= 'A' && q[i] <= 'Z') return false;
                ++i;
            }
        }
        if (j < p.length()) return false;
        while (i < q.length()) {
            if (q[i] >= 'A' && q[i] <= 'Z') return false;
            ++i;
        }
        return true;
    }
public:
    vector<bool> camelMatch(vector<string>& q, string p) {
        vector<bool> res;
        for (string qq : q) {
            res.push_back(solve(qq, p));
        }
        return res;
    }
};
