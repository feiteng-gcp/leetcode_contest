class Solution {
public:
    bool check(string q, string p) {
        int i, j = 0;
        for (i = 0; i < q.size(); i++) {
            if (j < p.size() && q[i] == p[j]) {
                j++;
            } else if (!(q[i] >= 'a' && q[i] <= 'z'))
                return false;
        }
        return j == p.size();
    }
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        int i, n, m = queries.size();
        vector<bool>ans;
        for (i = 0; i < m; i++) {
            ans.push_back(check(queries[i], pattern));
        }
        return ans;
    }
};
