class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {

        int len = queries.size();
        vector<bool> ans(len, false);
        for (int i = 0; i < len; i++) {
            int idx = 0;
            bool valid = true;
            for (auto & c : queries[i]) {
                if (idx < pattern.size() && c == pattern[idx])
                    idx++;
                else if (isupper(c)) {
                    valid = false;
                    break;
                }
            }
            if (valid && idx == pattern.size())
                ans[i] = true;
        }
        return ans;
    }
};
