class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string P) {
        
        vector<bool> ans; 
        for(auto &s: queries) {
            
            int j=0; 
            int num_upper = 0; 
            for(int i=0; i<s.size(); i++) {
                if(j<P.size() && s[i] == P[j]) {
                    j++; 
                }
                else if('A'<=s[i] && s[i]<='Z' ){
                    num_upper++; 
                }
            }
            
            if(num_upper>0 || j!=P.size()) {
                ans.push_back(false); 
                // cout<<s<<" "<<num_upper<<" "<<j<<endl; 
            }
            else {
                ans.push_back(true); 
            }
        }
        
        return ans; 
        
    }
};
