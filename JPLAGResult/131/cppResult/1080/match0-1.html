class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res(queries.size());
        for (int i = 0; i < res.size(); ++i) {
            res[i] = match(queries[i], pattern);
        }
        return res;
    }
private:
    int count_cap(string s) {
        int res = 0;
        for (auto& c : s) {
            if ('A' <= c && c <= 'Z') {
                ++res;
            }
        }
        return res;
    }
    
    bool match(string q, string p) {
        // cout << "[CAP COUNT]" << q << ": " << count_cap(q) << endl;
        
        if (count_cap(q) != count_cap(p)) {
            return false;
        }
        
        int pos = 0;
        for (char& c : q) {
            if (c == p[pos]) {
                ++pos;
            }
            if (pos == p.length()) {
                return true;
            }
        }
        return false;
    }
};
