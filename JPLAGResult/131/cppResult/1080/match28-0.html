class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res;
        int n = pattern.length();
        for(string s : queries) {
            int l = 0;
            bool ok=false;
            for(int i=0; i<s.length(); ++i) {
                if(s[i] == pattern[l]&&l<n) {
                    ++l;
                    if(l==n) {
                        ok = true;
                    }
                } else if(s[i]>='A'&&s[i]<='Z') {
                    ok = false;
                    break;
                }
            } 
            res.push_back(ok);
        }
        return res;
    }
};
