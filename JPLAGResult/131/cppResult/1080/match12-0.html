class Solution {
private:
    bool cmp(string& a, string& p){
        int pos=0;
        for(int i=0;i<p.size();i++){
            bool find = false;
            while(pos<a.size()){
                if(a[pos]==p[i]){
                    find=true;
                    pos++;
                    break;
                }
                if(a[pos]>='A'&&a[pos]<='Z')return false;
                pos++;
            }
            if(find==false)return false;
        }
        while(pos<a.size()){
            if(a[pos]>='A'&&a[pos]<='Z')return false;
            pos++;
        }
        return true;
    }
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool>ret;
        for(int i=0;i<queries.size();i++){
            ret.push_back(cmp(queries[i], pattern));
        }
        return ret;
    }
};
