class Solution {
public:
    bool isUpperCase(char a)
    {
        return (int(a)>=int('A') && int(a)<=int('Z')); 
    }
    bool match(string A,string pattern)
    {

        int i=0,j=0;
        while (i<A.length())
        {
            if (j<pattern.length() && A[i]==pattern[j])
            {
                i++;
                j++;
            }
            else
            {
                if (isUpperCase(A[i])) return false;
                i++;
            }
        }
        return (j>=pattern.length());
    }
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> result;
        int i;
        for (i=0; i<queries.size(); i++)
        {
            result.push_back(match(queries[i],pattern));
        }
        return result;
    }
};
