class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> ans;
        for (auto& query : queries) { ans.emplace_back(Matches(query, pattern)); }
        return ans;
    }
    
    bool Matches(const string& query, const string& pattern) {
        int p = 0;
        for (auto& ch : query) {
            if (p >= pattern.size()) {
                if (isupper(ch)) return false;
            }
            if (ch == pattern[p]) {
                ++p;
            } else if (isupper(ch)) {
                return false;
            }
        }
        return p >= pattern.size();
    }
};
