class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        int len=pattern.size();
        vector<bool> res;
        int n=queries.size();
        for(int i=0;i<n;i++)
        {
            int ll=queries[i].size();

            int j=0,k=0;
            while(j<ll&&k<len)
            {
                if(queries[i][j]==pattern[k])k++;
                else if(queries[i][j]>='A'&&queries[i][j]<='Z')break;
                j++;
            }
            if(k<len)res.push_back(false);
            else
            {
                while(j<ll)
                {
                    if(queries[i][j]>='A'&&queries[i][j]<='Z')
                    {
                        res.push_back(false);
                        break;
                    }
                    j++;
                }
                if(j==ll)res.push_back(true);
            }
        }
        return res;
    }
};
