class Solution {
    bool match(const string & s, const string & t) {
        int ns = s.size(), nt = t.size();
        int i = 0, j = 0;
        while(i < ns) {
            if(j < nt && s[i] == t[j]) {
                i++;
                j++;
            }
            else {
                if('A' <= s[i] && s[i] <= 'Z')
                    return false;
                i++;
            }
        }
        return j == nt;
    }
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res;
        for(auto query : queries) {
            res.push_back(match(query, pattern));
        }
        return res;
    }
};
