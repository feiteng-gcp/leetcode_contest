class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> result;
        for (auto q : queries) {
            result.push_back(check(q, pattern));
        }
        return result;
    }
private:
    bool check(const string& q, const string& p) {
        int pi = 0;

        int qi = 0;
        while (qi < q.size()) {
            if (pi < p.size() && q[qi] == p[pi]) {
                qi++;
                pi++;
            } else if (islower(q[qi])) {
                qi++;
            } else {
                return false;
            }
        }
        
        if (pi < p.size()) return false;
        return true;
    }
};
