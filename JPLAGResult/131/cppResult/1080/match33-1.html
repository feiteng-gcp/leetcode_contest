class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> ans;
        for (string s: queries) {
            int i = 0, j = 0;
            bool valid = true;
            while (i < s.size()) {
                if (j < pattern.size() && s[i] == pattern[j]) {
                    j++;
                }
                else if (s[i] < 'a') { // upper case
                    valid = false;
                    break;
                }
                i++;
            }
            if (j < pattern.size()) {
                valid = false;
            }
            ans.push_back(valid);
        }
        return ans;
    }
};
