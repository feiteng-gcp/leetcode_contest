class Solution {
public:
    
    bool match(string &s, string &p)
    {
        int j = 0;
        for (char &c : s)

        {
            if (j < p.size() && p[j] == c)
            {
                j++;
            }
            else if ('A' <= c && c <= 'Z') return false;
        }
        return j == p.size();
    }
    
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        int n = queries.size();
        vector<bool> ans(n);
        for (int i = 0; i < n; ++i) ans[i] = match(queries[i], pattern);
        return ans;
    }
};
