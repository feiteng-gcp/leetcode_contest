class Solution {
public:
    
    int check(string s1, string s2) {
        int n = s1.length();

        int m = s2.length();
        int i = 0, j = 0;
        
        while (i < n && j < m) {
            if (s1[i] == s2[j]) {
                i++;
                j++;
            }
            else {
                if (s1[i] <= 'Z' && s1[i] >= 'A')
                    return 0;
                i++;
            }
        }
        while (i < n) {
            if (s1[i] <= 'Z' && s1[i] >= 'A')
                return 0;
            i++;
        }
        return j == m;
    }
    
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> ret;
        
        for (int i = 0;i < queries.size();i++) {
            ret.push_back(check(queries[i], pattern));
        }
        return ret;
    }
};
