class Solution {
public:
    bool match(string &query, string &pattern){
        int m = query.size();
        int n = pattern.size();
        if(m<n){
            return false;
        }
        int i = 0, j = 0;
        while(i<m&&j<n){
            if(query[i]==pattern[j]){
                ++i,++j;
            }
            else if(islower(query[i])){
                ++i;
            }
            else{
                return false;
            }
        }
        if(j!=n){
            return false;
        }
        while(i<m&&islower(query[i])){
            ++i;
        }
        return i==m;
    }
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool>ans;
        ans.reserve(queries.size());
        for(auto &query:queries){
            ans.emplace_back(match(query, pattern));
        }
        return ans;
    }
};
