class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int begin = 0, cnt = 0;
        while(begin < T) {
            int max_end = -1;
            for(auto& p: clips) {
                if(p[0] <= begin) {
                    asmax(max_end, p[1]);
                }
            }
            if(max_end <= begin) return -1;
            begin = max_end;
            cnt++;
        }
        return cnt;
    }
};
