class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int n = clips.size();
        int k = 0;
        unordered_set<int> used;
        int res = 0;
        
        while (k < T) {
            int best = k, best_id = -1;
            for (int i = 0; i < n; ++i) {
                if (used.count(i) == 0 && clips[i][0] <= k) {
                    if (clips[i][1] > best) {
                        best = clips[i][1];
                        best_id = i;
                    }
                }        
            }
            // cout << best_id << " " << best << endl;
            if (best > k) {
                ++res;
                used.insert(best_id);
                k = best;
            } else {
                return -1;
            }
        }
        
        return res;
    }
};
