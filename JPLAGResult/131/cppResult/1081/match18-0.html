class Solution {
public:
    int videoStitching(vector<vector<int>>& a, int t) {
        int n = a.size();
        sort(a.begin(),a.end(), [](const vector<int>& a, const vector<int>& b)->bool {return a[0]<b[0];});
        int cur = 0, ret  = 0;
        while(cur<t){
            int mx = 0;
            for(int i=0;i<n;i++){
                if(a[i][0]>cur)break;
                mx=max(mx, a[i][1]);
            }
            if(mx<=cur)return -1;
            cur=mx;
            ret++;
        }
        return ret;
    }
};
