class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int pos = 0,cnt = 0;
        while(pos < T) {
            int epos = -1;
            for (int i = 0;i < clips.size();i++) {
                if (clips[i][0] <= pos && pos <= clips[i][1]) epos = max(epos,clips[i][1]);
            }
            if (epos == -1 || pos == epos) return -1;
            pos = epos;
           // cout << pos << endl;
            cnt++;
        }
        return cnt;
    }
};
