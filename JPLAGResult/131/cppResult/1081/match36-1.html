class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int dp[200];
        dp[T] = 0;
        
        for (int t = T-1; t >= 0; t--) {
            dp[t] = 1000000;
            for (const auto& c : clips) {
                if (c[0] <= t && c[1] >= t+1) {
                    dp[t] = min(dp[t], 1 + dp[min(c[1], T)]);
                    //printf("  clip %d,%d  => %d\n", c[0], c[1], 1 + dp[min(c[1], T)]);
                }
            }
            //printf("dp[%d] = %d\n", t, dp[t]);
        }
        
        return dp[0] <= clips.size() ? dp[0] : -1;
    }
};
