class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        sort(clips.begin(),clips.end());
        int t = 0, cnt = 0;
        while (t < T) {
            int max_ = t;
            for (auto c : clips) {
                if (c[0] <= t) max_ = max(max_, c[1]);
            }
            if (max_ > t) t = max_;
            else return -1;
            /*auto it = lower_bound(clips.begin(),clips.end(),{t+1,0}) - 1;
            if (it->second > t) t = it->second;
            else return -1;*/
            ++cnt;
        }
        return cnt;
    }
};
