bool myLess(const vector<int>& a, const vector<int>& b) {
    if (a[0] == b[0]) return a[1] < b[1];
    return a[0] < b[0];
}


class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        sort(clips.begin(), clips.end(), myLess);
        int ans = 0;
        int end = 0;
        int i = 0;
        while (end < T) {
            int nexte = end;
            while (i < clips.size() && clips[i][0] <= end) {
                nexte = max(nexte, clips[i][1]);
                if (nexte >= T) {
                    break;
                }
                i++;
            }
            if (nexte <= end || i == clips.size() ) {
                return -1;   
            }
            ans++;
            end = nexte;
        }
        return ans;

    }
};
