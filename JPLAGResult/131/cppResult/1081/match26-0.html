class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        vector<int> dp(T+1, -1);
        dp[0] = 0;
        for (int i=1; i<=T; ++i) {
            for (auto &c: clips) {
                if (c[0] > i || c[1] < i) { continue; }
                for (int j=c[0]; j<i; ++j) {
                    if (dp[j] < 0) { continue; }
                    if (dp[i] == -1 || dp[j]+1 < dp[i]) {
                        dp[i] = dp[j] + 1;
                    }   
                }   
            }   
        }   

        return dp[T];
    }   
};
