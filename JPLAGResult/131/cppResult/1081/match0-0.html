class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        sort(clips.begin(), clips.end(), [](const auto& i1, const auto& i2){
            return i1[0] < i2[0];
        });
        int i = 0;
        int start = 0;
        int count = 0;
        while(i < clips.size()) {
            if (clips[i][0] > start) {
                break;
            }
            int furthest = numeric_limits<int>::min();
            while(i < clips.size() && clips[i][0] <= start) {
                if( clips[i][1] > start) {
                    if (clips[i][1] > furthest) {
                        furthest = clips[i][1];
                    }
                }
                ++i;
            }
            ++count;
            
            if (furthest >= T) return count;
            start = furthest;
        }
        return -1;
    }
};
