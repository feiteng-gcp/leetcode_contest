class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int ans = 0, cur = 0;
        
        while(cur < T)
        {
            int far = cur;
            for(auto& i : clips)
                if(i[0] <= cur) far = max(far, i[1]);
            //cout<<"far = "<<far<<endl;
            if(far == cur)
                return -1;
            else
            {
                cur = far;
                ans++;
            }
        }
        
        return ans;
    }
};
