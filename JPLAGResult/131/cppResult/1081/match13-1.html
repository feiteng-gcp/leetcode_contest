class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int r = 0, p = 0;
        sort(clips.begin(), clips.end());

        while (true) {
            int t = p;
            for (int i = 0; i < clips.size(); i++) {
                if (clips[i][0] > t)
                    break;
                p = max(clips[i][1], p);
            }
            if (t == p) return -1;
            r++;
            if (p >= T) break;
        }
        return r;
    }
};
