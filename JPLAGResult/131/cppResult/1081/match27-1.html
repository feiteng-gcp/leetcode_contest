class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        vector<pair<int, int>> vp;
        REP(i, sz(clips)) {
            vp.pb(MP(clips[i][0], clips[i][1]));
        }
        SORT(vp);
        
        int curr = 0;

        int res = 0;
        while(curr < T) {
            int ind = -1;
            int mx = curr;
            for(int i=0; i<sz(vp); i++) {
                if(vp[i].first <= curr && vp[i].second > mx) {
                    mx = vp[i].second;
                    ind = i;
                }
            }
            if(ind == -1) return -1;
            curr = mx;
            res++;
        }
        return res;
    }
};
