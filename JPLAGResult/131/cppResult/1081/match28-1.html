class Solution {
public:
    static bool comp(vector<int> &a, vector<int> &b){
        if(a[0] == b[0]) return a[1]>b[1];
        else return a[0] < b[0];
    }
    
    int videoStitching(vector<vector<int>>& clips, int T) {
        sort(clips.begin(),clips.end(), comp);
        int eT = 0;
        int count = 0;
        int maxEnd = 0;
        for(int i = 0; i < clips.size(); i++){
            if(clips[i][0] <= eT){
                if(maxEnd < clips[i][1]){
                    maxEnd = clips[i][1];
                }
            }
            else{
                count++;
                eT = maxEnd;
                if(eT>=T) return count;
                if(clips[i][0] > eT)return -1;
                maxEnd = max(maxEnd,clips[i][1]);
                
            }
        }
        count++;
        eT = maxEnd;
        if(eT>=T) return count;
        else return -1;
    }
};
