class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        vector<int> dp(T + 1, INT_MAX / 2);
        dp[0] = 0;
        for (int i = 0; i <= T; ++i) {
            for (auto & clip : clips) {
                if (clip[0] <= i && clip[1] >= i) {
                    for (int j = i + 1; j <= clip[1] && j < dp.size(); ++j) {
                        dp[j] = std::min(dp[j], dp[i] + 1);
                    }
                }
            }
        }
        return dp[T] == INT_MAX / 2 ? -1 : dp[T];
    }
};
