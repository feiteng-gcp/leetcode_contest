class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int r = 0;
        int res = 0;
        while (true) {
            int right = r;
            for (auto clip: clips) {
                if (clip[0] <= r)
                    right = max(right, clip[1]);
            }
            res++;
            if (right == r)
                break;
            r = right;
            if (r >= T)
                break;
        }
        if (r >= T)
            return res;
        else
            return -1;
    }
};
