class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int ans = 0, cur = 0;
        while (cur < T) {
            int last = 0;
            for (auto clip : clips) {
                if (clip[0] <= cur && clip[1] >last) last = clip[1];
            }
            if (last > cur) {
                cur = last;
                ans ++;
            }
            else return -1;
        }
        return ans;
    }
};
