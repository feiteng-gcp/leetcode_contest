class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int cnt = 0;
        vector<vector<int> > res;
        int cur = 0;
        while (cur < T) {
            int best = cur;
            int ind = -1;
            for (int i = 0; i < clips.size(); i++) {
                if (clips[i][0] <= cur && clips[i][1] > best) {
                    best = clips[i][1];
                    ind = i;
                }
            }
            if (ind == -1) {
                return -1;
            }
            cnt++;
            cur = best;
        }
        return cnt;
    }
};
