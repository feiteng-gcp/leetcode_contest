class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int last = 0;
        sort(clips.begin(), clips.end());
        
        int i = 0;
        int res = 0;
        while( last < T) {
            int newLast = -1;
            while(i < clips.size() && clips[i][0] <= last){
                newLast = max(newLast, clips[i][1]);
                i ++;
            }
            res ++;
            if (newLast == last) return -1;
            last = newLast;
        }
        return res;
    }
};
