class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int ans = 0;
        int cur = 0;
        int n = clips.size();
        vector<bool> used(n, false);
        while (cur < T) {
            int ma = -1;
            int j = -1;
            for (int i = 0; i < n; i++) {
                if (clips[i][0] <= cur && !used[i]) {
                    if (clips[i][1] > ma) {
                        ma = clips[i][1];
                        j = i;
                    }
                }
            }
            if (ma <= cur) {
                return -1;
            }
            ans += 1;
            cur = ma;
            used[j] = true;
        }
        return ans;
    }
};
