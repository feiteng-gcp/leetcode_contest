class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        vector<bool> flag;
        
        for (int i = 0; i < clips.size(); i++) {
            flag.push_back(false);
        }
        
        int now = 0;
        int cnt = 0;
        while(now < T) {
            
            int next = now;
            int id = -1;
            for (int i = 0; i < clips.size(); i++) {
                if (flag[i] == false && clips[i][0] <= now && clips[i][1] > next) {
                    next = clips[i][1];
                    id = i;
                }
            }
            
            if (now == next) {
                return -1;
            }
            
            flag[id] = true;
            now = next;
            cnt++;
        }
        
        return cnt;
    }
    
    
};
