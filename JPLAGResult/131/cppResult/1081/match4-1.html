class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {

        int maxV = 0, count = 0;
        while (maxV < T) {
            int next = maxV;
            for (auto & v:clips) {
                if (v[0]<=maxV) next = max(next, v[1]);
            }
            if (next==maxV) return -1;
            maxV = next;
            count += 1;
        }
        return count;
    }
};
