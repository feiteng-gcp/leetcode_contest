class Solution {
public:
    static bool cmp(vector<int>a,vector<int> b){
        if(a[0]!=b[0])return a[0]<b[0];
        else return a[1]<b[1];
    }
    int videoStitching(vector<vector<int>>& a, int T) {
        sort(a.begin(),a.end(),cmp);
        int ans=0;
        int n=a.size();
        int l=0;
        for(int i=0;i<n;i++){
            if(l>=T)break;
            if(a[i][0]>l)return -1;
            if(a[i][1]<=l)continue;
            int maxr=0,maxi=i;
            for(int j=i;j<n && a[j][0]<=l;j++){
                if(a[j][1]>=maxr){
                    maxr=a[j][1];
                    maxi=j;
                }
            }
            i=maxi;
            ans++;
            l=a[i][1];
            if(l>=T)break;
        }
        if(l<T)return -1;
        return ans;
    }
};
