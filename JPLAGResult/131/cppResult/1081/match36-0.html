class Solution {
public:
    int videoStitching(vector<vector<int>>& clips, int T) {
        int N = clips.size();
        for (int i = 0; i < N; ++i) {
            clips[i][0] = min(T, clips[i][0]);
            clips[i][1] = min(T, clips[i][1]);
        }
        const int INFTY = 1000;
        vector<int> f(105, INFTY);
        sort(clips.begin(), clips.end());
        f[0] = 0;
        for (int i = 0; i < N; ++i) {
            int x = INFTY;
            for (int j = clips[i][0]; j < clips[i][1]; ++j)
                x = min(f[j], x);
            f[clips[i][1]] = min(f[clips[i][1]], x + 1);
        }
        return f[T] < INFTY ? f[T] : -1;
    }
};

bool operator <(const vector<int>& A, const vector<int>& B) {
    return A[1] < B[1];
}
