class Solution {
public:
    string removeOuterParentheses(string S) {
        int state = 0;
        string result = "";
        for(int i = 0; i < S.length(); ++i) {
            if(S[i] == '(') {
                if(state != 0) result += "(";
                state++;
                
            }else {
                state--;
                if(state != 0) result += ")";
            }
        }
        return result;
    }
};
