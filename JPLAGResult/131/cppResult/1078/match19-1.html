class Solution {
public:
    string removeOuterParentheses(string S) {
        int last = -1;
        int cnt = 0;
        string ans;
        for (int i = 0; i < S.length(); ++i) {
            if (S[i] == '(') cnt ++;

            else cnt --;
            
            if (cnt == 0) {
                //cout << last + 1 << ' ' << i << endl;
                for (int j = last + 2; j <= i - 1; ++j) ans += S[j];
                last = i;
            }
            
        }
        return ans;
    }
};
