class Solution {
public:
    string removeOuterParentheses(string S) {
        string ans;

        int lr=0;
        for(auto c:S){
            if(c=='('){
                if(lr>0) ans+=c;
                lr++;
            }else{
                lr--;
                if(lr>0) ans+=c;
            }
        }
        return ans;
    }
};
