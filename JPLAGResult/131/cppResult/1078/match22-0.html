class Solution {
public:
    string removeOuterParentheses(string S) {
        if (S.size() < 2) return "";
        int counter = 0, start = 0;
        for (int i = 0; i < S.size(); i++) {
            if (S[i] == '(') counter++;
            else counter--;
            if (counter == 0) {
                S.replace(i,1,"");
                S.replace(start,1,"");
                start = i-1;
                i -= 2;
            }
        }
        return S;
    }
};
