class Solution {
public:
    string removeOuterParentheses(string S) {
        int n=S.size();
        if(n==0) return "";
        string res;

        int count=0, start=0;
        for(int i=0;i<n;++i){
            if(S[i]=='(') count++;
            else count--;
            if(count==0){
                res+=S.substr(start+1,i-start-1);
                start=i+1;
            }
        }
        return res;
    }
};
