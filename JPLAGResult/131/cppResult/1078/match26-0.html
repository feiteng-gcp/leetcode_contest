class Solution {
public:
    string removeOuterParentheses(string S) {
        int count = 0, start = 0;
        string res = "";
        for (int i = 0; i < S.size(); i++) {
            if (S[i] == '(') {
                count++;
            } else if (S[i] == ')') {
                count--;
            }
            if (count == 0) {
                res += S.substr(start+1, i-1-start);
                start = i+1;
            }
        }
        return res;
    }
};
