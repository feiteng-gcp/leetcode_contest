class Solution {
public:
    string removeOuterParentheses(string S) {
        int n = S.size();
        if (n == 0) return "";
        string res = "";
        int sum = 0;
        
        int j = 0;
        for (int i = 0; i < n; ++i) {
            if (S[i] == '(') ++sum;
            else --sum;
            if (sum == 0) {
                res += S.substr(j+1, i-j-1);
                j = i + 1;
            }
        }
        
        return res;
    }
};
