class Solution {
public:
    string removeOuterParentheses(string S) {
        string t = "";

        int cnt = 0;
        for (int i = 0; i < S.length(); i++) {
            if (S[i] == '(')  {
                cnt++;
                if (cnt != 1) {
                    t += S[i];
                }
            }
            else {
                cnt--;
                if (cnt != 0) {
                    t += S[i];
                }
            }
        }
        return t;
    }
};
