class Solution {
public:
    string removeOuterParentheses(string s) {
      string ans;

      int start = 0, paren = 0;
      for (int i = 0; i < s.size(); ++i) {
        if (s[i] == '(') {
          ++paren;
        } else {

          --paren;
        }
        if (paren == 0) {
          ans += s.substr(start + 1, i - start - 1);
          start = i + 1;
        }
      }
      return ans;
    }
};
