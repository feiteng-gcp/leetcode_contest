class Solution {
public:
    string removeOuterParentheses(string S) {
        vector<string> v;
        int prev = 0;
        int l = 0;
        for (int i = 0;i < S.length();i++) {
            if (S[i] == '(')
                l++;
            else {
                l--;
                if (l == 0) {
                    v.push_back(S.substr(prev, i-prev+1));
                    prev = i+1;
                }
            }
        }
        string ret;
        for (int i = 0;i < v.size();i++) {
            ret += v[i].substr(1, v[i].length()-2);
        }
        return ret;
    }
};
