class Solution {
public:
    string removeOuterParentheses(string s) {
        string tmp, res;

        int cnt = 0;
        for (char ch : s) {
            if (ch == '(') {
                ++cnt;
            } else {
                --cnt;
            }
            tmp.push_back(ch);
            if (!cnt) {
                res += tmp.substr(1, tmp.length() - 2);
                tmp = "";
            }
        }
        
        return res;
    }
};
