class Solution {
public:
    string removeOuterParentheses(string S) {
        int istart = 0;
        int depth = 0;
        string result;
        for (int i = 0; i < S.size(); ++i) {
            if (S[i] == '(')
                depth += 1;
            else {
                depth -= 1;
                if (depth == 0) {
                    result.append(S.substr(istart + 1, i - istart - 1));
                    istart = i + 1;
                }
            }
        }
        return result;
    }
};
