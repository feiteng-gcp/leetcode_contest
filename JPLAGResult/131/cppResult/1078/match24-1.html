class Solution {
public:
    string removeOuterParentheses(string S) {
        string ans = "";

        int cnt = 0;
        for (int i = 0;i < S.size();i++) {
            if (S[i] == '(') cnt++;
            else cnt--;
            if (S[i] == '(' && cnt == 1) continue;
            if (S[i] == ')' && cnt == 0) continue;
            ans += S[i];
        }
        return ans;
    }
};
