class Solution {
public:
    string removeOuterParentheses(string S) {
        int p = 0;
        string r;
        for (char c : S) {
            if (c == '(') {
                if (p > 0) r += c;
                p++;
            }
            else {
                p--;
                if (p > 0) r += c;
            }
        }
        return r;
    }
};
