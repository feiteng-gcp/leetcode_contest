class Solution {
public:
    string removeOuterParentheses(string S) {
        if(S.empty()) {
            return "";
        }
        string st = "";
        int cnt = 0;
        
        string res;
        for(char c : S) {

            if(c == '(') {
                cnt ++;
            }
            else {
                cnt --;
            }
            st.push_back(c);
            if(cnt == 0) {
                st.pop_back();
                res += st.substr(1);
                st = "";
            }
        }
        
        return res;
    }
};
