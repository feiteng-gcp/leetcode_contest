class Solution {
public:
    string removeOuterParentheses(string S) {
        string res = "";
        int count = 0;
        int start = 0;
        for (int i = 0; i < S.length(); ++i) {
            if (S[i] == '(') {
                count ++;
            } else if (S[i] == ')') {
                count --;
            }
            if (count == 0) {

                res += S.substr(start+1, i-start-1);
                start = i +1;
                count = 0;
            }
        }
        return res;
    }
};
