class Solution {
public:
    string removeOuterParentheses(string S) {
        int L = S.length();
        int x = 0;
        string ans = "";
        for (int i = 0; i < L; ++i) {
            if (S[i] == '(') {
                ++x;
                if (x != 1)
                    ans += S[i];
            } else {
                --x;
                if (x != 0)
                    ans += S[i];
            }
        }
        return ans;
    }
};
