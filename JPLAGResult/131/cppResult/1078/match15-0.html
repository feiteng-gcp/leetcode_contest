class Solution {
public:
    string removeOuterParentheses(string S) {
        string ret;
        int level = 0;
        for (int i = 0; i < S.length(); i++) {
            if (S[i] == '(') {
                level++;
                if (level != 1) {
                    ret += '(';
                }
            } else {
                level --;
                if (level != 0) {
                    ret += ')';
                }
            }
        }
        return ret;
    }
};
