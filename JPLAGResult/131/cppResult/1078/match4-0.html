class Solution {
public:
    string removeOuterParentheses(string S) {
        int s = 0, c = 0;
        string result;
        for (int i = 0; i < S.length(); i++) {
            if (S[i] == '(') {
                c++;
            } else {
                c--;
            }
            if (c == 0) {
                result += S.substr(s + 1, i - s - 1);
                s = i + 1;
            }
        }
        return result;
    }
};
