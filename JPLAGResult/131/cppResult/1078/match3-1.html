class Solution {
public:
    string removeOuterParentheses(string S) {
        vector<string> toks;
        int start = 0;
        int x = 0;
        for (int i = 0; i < S.length(); ++i) {
            if (S[i] == '(') ++x;
            else --x;
            
            if (x == 0) {
                toks.push_back(S.substr(start, i - start + 1));
                start = i + 1;
            }
        }
        
        string res = "";
        for (auto& t : toks) {
            for (int i = 1; i + 1 < t.length(); ++i) {
                res += t[i];
            }
        }
        
        return res;
    }
};
