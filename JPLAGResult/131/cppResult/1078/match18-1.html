class Solution {
public:
    string removeOuterParentheses(string S) {
        int len=S.length();
        string ans;

        int cur=0;
        int last=-1;
        for(int i=0; i<len; i++) {
            if (S[i]=='(') cur++;
            else cur--;
            if (cur==0) {
                for(int j=last+2; j<=i-1; j++) ans+=S[j];
                last=i;
            }
        }
        return ans;
    }
};
