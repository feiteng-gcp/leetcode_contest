class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if len(clips) == 0:
            return -1
        if T == 0:
            for clip in clips:
                if clip[0] == 0:
                    return 1
            
return -1
        end = 0
        count = 0
        while end < T:
            new_end = end
            for clip in clips:
                if clip[0] <= end:
                    new_end = max(new_end, clip[1])
            if new_end == end:
                return -1
            end = new_end
            count += 1
        return count
