class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        clips=sorted(clips,key=lambda x: x[0])
        res=0
        start=0
        final=0
        for i in range(len(clips)):
            if clips[i][0]<=start:
                final=max(final,clips[i][1])
                if final>=T:
                    break
            else:
                res+=1
                start=final
                if clips[i][0]<=start:
                     final=max(final,clips[i][1])
                if final>=T:
                    break
        if final>=T:
            return res+1
        else:
            return -1
