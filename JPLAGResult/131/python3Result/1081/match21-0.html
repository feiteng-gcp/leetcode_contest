class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if T == 0:
            return 0

        clips.sort()
        c = 0
        left = 0
        right = 0
        for i in range(len(clips)):
            if clips[i][0] > left:
                if clips[i][0] > right:
                    return -1
                c += 1
                left = right
                right = clips[i][1]
            elif clips[i][0] <= left:
                if clips[i][1] > right:
                    right = clips[i][1]
            if right >= T:
                return c + 1
        return -1
