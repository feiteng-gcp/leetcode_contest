class Solution:
    def videoStitching(self, clips, T: int) -> int:
        clips.sort(key=lambda x:x[1],reverse=True)
        clips
.sort(key=lambda x:x[0])

        ANS=0
        NOW=0
        NEXT=0

        #print(clips)

        for i,j in clips:
            if i>NOW and i>NEXT:
                return -1
            if NEXT<j:
                NEXT=j

                if NEXT>=T:
                    ANS+=1
                    break

            if i>=NOW:
                ANS+=1
                NOW=NEXT

        if NEXT<T:
            return -1

        return ANS
