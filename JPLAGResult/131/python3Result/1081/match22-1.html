class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        arr = sorted(clips + [[T, T]], key=lambda a:(a[0], -a
[1]))
        if arr[0][0] > 0:
            return -1
        # print(arr)
        cur = ans = mx = 0
        for i in range(len(arr)):
            x, y = arr[i][0], arr[i][1]
            if x > cur:
                if mx < x:
                    return -1
                ans += 1
                cur = mx
                if mx >= T:
                    break
            if mx < y:
                mx = y
        return ans
