class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        clips  = [(i[0],i[1]) for i in clips]
        clips.sort()
        now, p, l= 0, 0, len(clips) 
        ans = 0
        while p<l:
            best = -1
            
            while(p<l and clips[p][0]<=now):
                best = max(best,clips[p][1])
                p+=1
            
            if best == -1:
                return -1
            now = best 
            ans +=1
            
            if now >= T :
                return ans 
        return -1
