class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if T == 0:
            return 0
        clips.sort()
        n = len(clips)
        ret = 0
        i = 0
        currEnd = 0
        # print(clips)
        while currEnd < T and i < n and clips[i][0] <= currEnd:
            newCurrEnd = currEnd
            while i < n and clips[i][0] <= currEnd:
                newCurrEnd = max(newCurrEnd, clips[i][1])
                i += 1
            currEnd = newCurrEnd
            ret += 1

        if currEnd < T:
            return -1
        return ret
