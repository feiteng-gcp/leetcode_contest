class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        s = 0
        e = 0
        ans = 0
        clips.sort(key = lambda x:x[0])
        for c in clips:
            if c[0] > s:
                s = e
                ans += 1
            if c[0] <= s:
                if e < c[1]:
                    e = c[1]
                if e >= T:
                    ans += 1
                    break
        return ans if e >= T else -1
