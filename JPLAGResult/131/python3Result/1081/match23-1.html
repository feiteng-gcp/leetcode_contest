class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        def function(cl):
            result=[]
            for i in clips:
                result.append(i)
                count=[]
                for j in range(len(result)-1):
                    if result[j][0]<=i[0] and result[j][1]>=i[1]:
                        result.pop()
                        break
                    if result[j][0]>=i[0] and result[j][1]<=i[1]:
                        count.append(j)
                for j in range(len(count)-1,-1,-1):
                    del result[count[j]]
            return result

        def function2(cl,T):
            cl.sort(key=lambda x:x[1])
            if cl[-1][1]<T:
                return -1
            cl.sort
(key=lambda x:x[0])
            if cl[0][0]!=0:
                return -1
            count=0
            vmax=0
            recordmax=0
            for i in range(len(cl)):
                if vmax>=cl[i][0]:
                    if cl[i][1]>recordmax:
                        recordmax=cl[i][1]
                    if i<=len(cl)-2:
                        if cl[i+1][0]>vmax:
                            count+=1
                            vmax=recordmax
                            recordmax=0
                    else:
                        if cl[i][1]>vmax:
                            count+=1
                            vmax=cl[i][1]
                if vmax>=T:
                    return count
            return -1

        z=function(clips)
        return function2(z,T)
