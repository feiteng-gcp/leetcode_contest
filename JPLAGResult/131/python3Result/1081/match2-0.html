class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if T == 0:
            return 0
        
        count = 0
        covered = 0
        
        while covered < T:
            e = covered
            
            for c in clips:
                if c[0] <= covered:
                    e = max(e, c[1])
            
            if e == covered:
                return -1
            covered = e
            count += 1
            
        return count
