class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if T == 0: return 0

        clips.sort()
        N = len(clips)
        
        i = 0
        ans = 0
        last = 0
        while i < N:
            ans += 1
            nlast 
= last
            while i < N and clips[i][0] <= last:
                nlast = max(nlast, clips[i][1])
                i += 1
            if nlast >= T:
                return ans
            elif nlast == last:
                return -1
            last = nlast
        return -1
