class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        n = len(clips)
        no_used = [False] * len(clips)
        for i in range(n):
            for j in range(i + 1, n):
                if clips[i][0] <= clips[j][0] and clips[j][1] <= clips[i][1]:
                    no_used[j] = True
                elif clips[j][0] <= clips[i][0] and clips[i][1] <= clips[j][1]:
                    no_used[i] = True
        used_clips = []
        for i in range(n):
            if no_used[i] == False:
                used_clips.append(clips[i])
        used_clips = sorted(used_clips, key=lambda x: x[0])
        if used_clips[0][0] != 0:
            return -1
        
        maxLen = 0
        index = 0
        ans = 0
        while index < len(used_clips):
            ans += 1
            maxLen = used_clips[index][1]
            if maxLen >= T:
                return ans
            if index + 1 == len(used_clips):
                return -1
            index += 1
            oldIndex = index
            while index < len(used_clips) and used_clips[index][0] <= maxLen:
                index += 1
            if index == oldIndex:
                return -1
            index -= 1
        return -1
