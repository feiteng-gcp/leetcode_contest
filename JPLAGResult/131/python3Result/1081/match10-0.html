class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if T == 0:
            return 0
        clips.sort(key=lambda x:x[0])
        prev_end = 0
        end = 0
        count = 1
        
        for s, e in clips:
            if s > end:
                return -1
            if s > prev_end:
                prev_end = end
                count += 1
            
            end = max(end, e)
            
            if end >= T:
                return count
        
        return -1
