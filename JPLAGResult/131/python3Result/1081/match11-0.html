class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if T == 0:
            return 0
        clips.sort(key = lambda x:(x[0], -x[1]))
        stack = []
        for x, y in clips:
            if stack and stack[-1][0] <= x and y <= stack[-1][1]:
                continue
            stack.append([x,y])
        if stack[0][0] > 0:
            return -1
        if stack[-1][1] < T:
            return -1
        ans = 1
        i = 1
        right = stack[0][1]
        pre = 0
        while i < len(stack):
            if right >= T:
                return ans
            while i < len(stack) and right >= stack[i][0]:
                pre = i
                i += 1
            if right == stack[pre][1]:
                return -1
            ans += 1
            right = stack[pre][1]
        if right >= T:
            return ans
        else:
            return -1
