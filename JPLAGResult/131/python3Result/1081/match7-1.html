class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        clips = sorted(clips)
        if clips[0][0] != 0:
            return -1
        i = 0
        pre_end = -1
        ans = 1
        while i < len(clips) and clips[i][0] == 0:
            pre_end = clips[i][1]
            i += 1
        if pre_end >= T:
            return ans
        #print(clips)
        while i < len(clips):
            tmp_end 
= 0
            #print("tmp_end " + str(pre_end) )
            savei = i
            while savei < len(clips) and clips[savei][0] <= pre_end: 
                tmp_end = max(tmp_end, clips[savei][1])
                savei += 1
            if savei == i:
                return -1
            else:
                i = savei
            #print(tmp_end)
            ans += 1
            pre_end = tmp_end
            if pre_end >= T:
                return ans
            
        return -1 if pre_end < T else ans
