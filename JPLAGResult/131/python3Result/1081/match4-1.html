class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        sortedclips = sorted(clips)
        start = end = count = pos = 0
        while end < T:
            maxend = end
            prevpos = pos
            
while pos < len(sortedclips) and sortedclips[pos][0] <= end:
                maxend = max(sortedclips[pos][1], maxend)
                pos += 1
            if prevpos == pos:
                return -1
            end = maxend
            count += 1
        return count
