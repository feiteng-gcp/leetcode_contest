class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        clips = sorted(clips, key = lambda x:x[0])
        stack = []
        p = 0
        r = 0
        c = None
        # print(clips)
        reached = False
        for clip in clips:
            if clip[0] <= p:
                if clip[1] > r:
                    r = clip[1]
                    c = clip
            else:
                if c == None:
                    return -1
                stack += [c]
                c = clip
                p = r
                r = clip[1]
                if p >= T:
                    reached = True
                    break
        if not reached:
            stack += [c]
            p = r
        # print(stack, p)
        if stack[0][0] > 0 or p < T:
            return -1
        
        return len(stack)
