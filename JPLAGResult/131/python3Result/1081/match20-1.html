class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if len(clips)==0:
            if T==0:
                return 0
            else:
                return -1
            
        clips.sort()
        print(clips)
        
if clips[-1][1]<T:
            return -1
        if clips[0][0]>0:
            return -1
        i = 0
        last = 0
        cnt = 0
        while(i<len(clips)):
            if clips[i][0]>last:
                return -1
            idx = i
            cov = clips[i][1]
            while(i<len(clips) and clips[i][0]<=last):
                if cov<clips[i][1]:
                    cov = clips[i][1]
                    idx = i
                i+=1
            print(clips[idx])
            cnt+=1
            last = clips[idx][1]
            if last>=T:
                break
        return cnt
