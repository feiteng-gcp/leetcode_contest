class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        if len(clips) == 0:
            return -1
        if T == 0:
            return 0
        
        cs = sorted(clips)
        res = 0
        r = rr = 0
        i, n = 0, len(cs)
        while i < n:
            if cs[i][0] > rr:
                return -1
            while i<n and r >= cs[i][0]:
                rr = max(rr,cs[i][1])
                i += 1
            res += 1
            r = rr
            if r>=T:
                return res
        return -1
