class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        clips = sorted(clips, key = lambda x: (x[0],-x[1]))
        ans = 0
        s = 0
        stack = []
        for u,v in clips:
            if len(stack) <= 1:
                stack.append((u,v))
                continue
            while len(stack) >= 2 and u <= stack[-2][1] and v >= stack[-1][1]:
                stack.pop()
            if v > stack[-1][1]:
                stack.append((u,v)) 
        print(stack)
        start = 0
        ans = 0
        for u,v in stack:
            if u > start:
                return -1
            start = v
            ans += 1
            if start >= T:
                return ans

        return -1
