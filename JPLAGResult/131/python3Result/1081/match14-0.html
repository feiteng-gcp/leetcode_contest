class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        clips = sorted(clips)
        num = 0
        end = 0
        nextend = -1
        i = 0
        while(end < T and i < len(clips)):
            while i < len(clips) and (clips[i][0] <= end):
                nextend = max(clips[i][1], nextend)
                i += 1
            if(nextend == -1):
                return -1
            num += 1
            end = nextend
            nextend = -1
        if(end < T):
            return -1
        return num
