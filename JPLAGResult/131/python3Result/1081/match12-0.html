class Solution:
    def videoStitching(self, clips: List[List[int]], T: int) -> int:
        clips = sorted(clips)
        start, end = 0, 0
        res = 0
        endpre = end
        while(end < T):
            maxend = end
            for clip in clips:
                if clip[0] <= end:
                    maxend = max(clip[1], maxend)
            end = maxend
            res += 1
            if end == endpre:
                return -1
            endpre = end
        return res
