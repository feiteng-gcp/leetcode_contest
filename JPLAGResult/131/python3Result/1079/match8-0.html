# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        def traverse(summ, node, current):
            current = (current*2 + node.val)% (10**9+7)
            if node.left is None and node.right is None:
                summ[0] = summ[0] + current
                return
            if node.left is not None:
                traverse(summ, node.left, current)
            if node.right is not None:
                traverse(summ, node.right, current)
        summ = [0]
        traverse(summ, root, 0)
        return summ[0]% (10**9+7)
