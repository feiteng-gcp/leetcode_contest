# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        def travel(upper, node, res):  # leaf
            if node.left is None and node.right is None:
                res.append(upper * 2 + node.val)
            else:
                if node.left:
                    travel(upper * 2 + node.val, node.left, res)
                if node.right:
                    travel(upper * 2 + node.val, node.right, res)

        res = []
        travel(0, root, res)
        return sum(res) % (10 ** 9 + 7)
