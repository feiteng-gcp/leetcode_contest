# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        def function(root,result="",count=0):
            result+=str(root.val)
            if root.left is None and root.right is None:
                return count+int(result,2)
            
if root.left is not None:
                count=function(root.left,result,count)
            if root.right is not None:
                count=function(root.right,result,count)
            return count
        return function(root)%(pow(10,9) + 7)
