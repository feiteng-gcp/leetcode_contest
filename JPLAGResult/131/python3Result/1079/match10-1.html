# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        self.ans = 0
        self.mod = 1000000007
        
        def dfs(r, s):
            s = s * 2 + r.val
            if r.left == None and r.right == None:
                self.ans += s
                
return
            if r.left:
                dfs(r.left, s)
            if r.right:
                dfs(r.right, s)
        
        dfs(root, 0)
        return self.ans % self.mod
