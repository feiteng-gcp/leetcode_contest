# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        mod = int(1e9 + 7)
        def walk(node, cur, ans):
            cur = cur * 2 + node.val
            if not node.left and not node.right:
                ans[0] = (ans[0] + cur) % mod
            if node.left:
                walk(node.left, cur, ans)
            if node.right:
                walk(node.right, cur, ans)
        ans = [0]
        walk(root, 0, ans)
        return ans[0]
