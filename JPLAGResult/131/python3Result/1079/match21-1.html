# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        s = self.DFS(root, 0)
        return s % (pow(10, 9) + 7)
    def DFS(self, root, s):
        
        s = (s << 1) + root.val
        # print(s)
        
if root.left is None and root.right is None:
            return s
        ls, rs = 0, 0
        if root.left:
            ls = self.DFS(root.left, s)
        if root.right:
            rs = self.DFS(root.right, s)
            
        return ls + rs
