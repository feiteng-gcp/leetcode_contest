# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        if root == None:
            return 0
        
        self.res = 0
        self._trav(0,root)
        return self.res
    
    def _trav(self,num, r):
        num = num*2 + r.val
        if r.left == r.right and r.left == None:
            self.res += num
            self.res %= (10**9+7)
            return
        
        if r.left:
            self._trav(num, r.left)
        if r.right:
            self._trav(num, r.right)
