# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        output = []
        q = [(root, root.val)]
        while q:
            q, buf = [], q[:]
            for node, val in buf:
                if not node.left and not node.right:
                    output.append(val)
                    continue
                if node.left:
                    q.append((node.left, val * 2 + node.left.val))
                if node.right:
                    q.append((node.right, val * 2 + node.right.val))
        return sum(output) % (10**9 + 7)
