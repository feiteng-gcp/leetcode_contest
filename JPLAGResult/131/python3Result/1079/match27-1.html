# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

from collections import deque
mod=10**9+7

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        QUE=deque()
        QUE.append([root,[root.val]])
        ANS=0

        while QUE:
            x,NOW=QUE.pop()

            if x.left==None and x.right==None:
                ANS=(ANS+int("".join(map
(str,NOW)),2))%mod
                continue

            if x.left!=None:
                QUE.append([x.left,NOW+[(x.left).val]])

            if x.right!=None:
                QUE.append([x.right,NOW+[(x.right).val]])

        return ANS
