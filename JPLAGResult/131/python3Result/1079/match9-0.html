class Solution:
    def sumRootToLeaf(self, root: "TreeNode") -> int:
        self.ans=0
        def search(root,legacy):
            if not root.left and not root.right:
                self.ans+=root.val+legacy
                self.ans%=10**9+7
                return
            newleg=(legacy+root.val)*2
            if root.left:
                search(root.left,newleg)
            if root.right:
                search(root.right,newleg)

        search(root,0)
        return self.ans
