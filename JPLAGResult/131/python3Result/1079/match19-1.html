# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        self.sum = 0
        
        def dfs(r, carry):
            if not r:
                return
            v = carry * 2 + r.val
            if not r.left and not r.right:
                self.sum += v
            else:
                dfs(r.left, v)
                dfs(r.right, v)
            
            
        dfs(root, 0)
        
        return self.sum % 1000000007
