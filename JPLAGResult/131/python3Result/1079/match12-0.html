# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

MOD = 1000000000 + 7

class Solution:
    def pathToNumber(self, curPath: List) -> int:
        res = 0
        for i in range(len(curPath)):
            res = res * 2 + curPath[i]
        return res
    
    def helper(self, root: TreeNode, curPath, ans):
        if root is None:
            return
        curPath.append(root.val)
        if root.left is None and root.right is None:
            curNumber = self.pathToNumber(curPath)
            ans[0] = (ans[0] + curNumber) % MOD
            curPath.pop()
            return
        
        self.helper(root.left, curPath, ans)
        self.helper(root.right, curPath, ans)
        curPath.pop()      
        
    def sumRootToLeaf(self, root: TreeNode) -> int:
        ans = [0]
        self.helper(root, [], ans)
        return ans[0]
