# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

_MOD = 10**9+7
class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        self.result = 0
        self.dfs(root, 0)
        return self.result
        
    def dfs(self, root, prefix):
        if not root:
            return
        current = (prefix*2 + root.val) % _MOD
        if not root.left and not root.right:
            self.result = (self.result + current) % _MOD
        self.dfs(root.left, current)
        self.dfs(root.right, current)
