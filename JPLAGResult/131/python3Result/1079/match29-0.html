# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        ret = 0
        MOD = 10 ** 9 + 7
        c = 0
        
        def s(p):
            nonlocal ret, c
            if p is None:
                return
            c = c * 2 + p.val
            if p.left is None and p.right is None:
                ret = (ret + c) % MOD
            s(p.left)
            s(p.right)
            c //= 2
            
        s(root)
        return ret
