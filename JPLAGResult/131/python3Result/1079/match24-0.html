# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.ans = 0
    def topdown(self, root, pre):
        value = (pre<<1) + root.val
        if not root.left and not root.right:
            self.ans += value
            self.ans %= (10**9 + 7)
            return
        if root.left:
            self.topdown(root.left, value)
        if root.right:
            self.topdown(root.right, value)
    def sumRootToLeaf(self, root: TreeNode) -> int:
        # 自顶向下
        self.topdown(root, 0)
        return self.ans
