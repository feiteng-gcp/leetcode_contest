# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

modv = 10**9 + 7

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        return self.dfs(root, 0)
        
    def dfs(self, root, val):
        val <<= 1
        val %= modv
        val += root.val
        val %= modv
        if root.left is None and root.right is None:
            return val
        ans = 0
        if root.left:
            ans += self.dfs(root.left, val)
        if root.right:
            ans += self.dfs(root.right, val)
        ans %= modv
        return ans
