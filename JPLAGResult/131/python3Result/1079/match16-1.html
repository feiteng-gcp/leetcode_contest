# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        res 
= [0]
        self.dfs(root, 0, res)
        return res[0]
        
    def dfs(self, node, pathSum, res):
        if not node:
            return
        pathSum = (pathSum * 2 + node.val) % 1000000007
        if not node.left and not node.right:
            res[0] = (res[0] + pathSum ) % 1000000007
            return
        self.dfs(node.left, pathSum, res)
        self.dfs(node.right, pathSum, res)
