# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        def helper(node, cur):
            if not node:
                return 
            
            cur 
= (cur << 1) + node.val

            if not node.left and not node.right:
                self.res += cur % (10**9 + 7)
                
            helper(node.left, cur)
            helper(node.right, cur)
        
        self.res = 0
        helper(root, 0)
        
        return self.res % (10**9 + 7)
