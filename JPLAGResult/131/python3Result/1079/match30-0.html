# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        def dfs(root, s):
            if not root:
                return
            
            s = s * 2 + root.val
            if not root.left and not root.right:
                self.res += s
                return
            if root.left:
                dfs(root.left, s)
            if root.right:
                dfs(root.right, s)
        
        self.res = 0
        dfs(root, 0)
        return self.res % (10 ** 9 + 7)
