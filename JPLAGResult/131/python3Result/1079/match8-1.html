# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        if not root:
            return 0
        
        mod = 10 ** 9 + 7
        self.ans = 0
        
        
def helper(node, pre):
            cur = pre * 2 + node.val
            if not node.left and not node.right:
                self.ans = (self.ans + cur) % mod
                return
            if node.left:
                helper(node.left, cur)
            if node.right:
                helper(node.right, cur)
        
        helper(root, 0)
        return self.ans
