# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        def h(root, num):
            if root is None:
                return 0
            if root.left is None and root.right is None:
                return (num << 1) + root.val
            
            return h(root.left, (num << 1) + root.val) + h(root.right, (num << 1) + root.val)
        return h(root, 0) % (10**9+7)
