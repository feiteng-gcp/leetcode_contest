# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        res = 0
        def dfs(node, pre=''):
            nonlocal res
            pre += str(node.val)
            if not node.left and not node.right:
                res += int(pre, 2)
                return
            if node.left:
                dfs(node.left, pre)
            if node.right:
                dfs(node.right, pre)
        dfs(root)
        return res % (10 ** 9 + 7)
