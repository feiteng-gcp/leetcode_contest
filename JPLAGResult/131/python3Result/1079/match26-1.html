# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        MOD = 10**9 + 7
        ans = 0
        
        def dfs(node, cur):
            nonlocal ans
            if not node.left and not node.right:
                ans = (ans + cur * 2 + node.val) % MOD
            else:
                
if node.left:
                    dfs(node.left, (cur * 2 + node.val) % MOD)
                if node.right:
                    dfs(node.right, (cur * 2 + node.val) % MOD)
        
        dfs(root, 0)
        return ans
