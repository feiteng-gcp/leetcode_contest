# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        def pre_order(node, res, ans):
            if node.left == None and node.right == None:
                #print("res = {}".format(res))
                ans[0] += res * 2 + node.val
                ans[0] %= 10 ** 9 + 7
            else:
                
if node.left: pre_order(node.left, res * 2 + node.val, ans)
                if node.right: pre_order(node.right, res * 2 + node.val, ans)
        ans = [0]
        pre_order(root, 0, ans)
        return ans[0]
