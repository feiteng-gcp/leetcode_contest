# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.max_val = 1000000007
    def sumRootToLeaf(self, root: TreeNode) -> int:
        return self.dfs(root)[0]
    
def dfs(self, root):
        if not root:
            return 0, 0
        if not root.left and not root.right:
            return root.val, 1 # sum, how many branches in total
        sum1, b1 = self.dfs(root.left)
        sum2, b2 = self.dfs(root.right)
        b3 = ((b1+b2)<<1) % self.max_val
        sum3 = (sum1+sum2+b3*root.val) % self.max_val
        return sum3, b3
