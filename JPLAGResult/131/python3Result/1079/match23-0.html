# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        b = (pow(10, 9) + 7)
        
        def sumSubTree(node, carry):
            if node.left == None and node.right == None:
                return carry * 2 + node.val    
            
            new_carry = carry * 2 + node.val
            left_sum = 0
            right_sum = 0
            if node.left != None:
                left_sum = sumSubTree(node.left, new_carry)
            if node.right != None:
                right_sum = sumSubTree(node.right, new_carry)
                
            return (left_sum + right_sum) % b
            
        return sumSubTree(root, 0)
