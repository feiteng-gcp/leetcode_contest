class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        res=[False]*len(queries)
        for i in range(len(queries)):
            k=0
            for j in range(len(queries[i])):
                if queries[i][j]!=pattern[k]:
                    if queries[i][j].isupper():
                        break
                else:
                    k+=1
                    if k==len(pattern):
                        break
            x=[a for a in queries[i][j+1:] if a.isupper()]
            if len(x)==0 and k==len(pattern):
                    res[i]=True
                    
        return res
