class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        def check(s, p):
            p1 = p2 = 0
            while p1 < len(s):
                if p2 < len(p) and s[p1] == p[p2]:
                    p1 += 1
                    p2 += 1
                elif s[p1].islower():
                    p1 += 1
                else:
                    return False
            return p2 >= len(p)

        return [check(i, pattern) for i in queries]
