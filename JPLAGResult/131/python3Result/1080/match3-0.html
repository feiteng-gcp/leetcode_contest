class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        
        def isMatch(query, pattern):
            i = 0
            for l in query:
                
                if i < len(pattern) and pattern[i] == l:
                    i += 1
                elif l.isupper():
                    return False
                    
            return i == len(pattern)
        
        return [isMatch(q, pattern) for q in queries]
