class Solution:
    def camelMatch(self, queries: List[str], p: str) -> List[bool]:
        res = []
        for q in queries:
            idx = 0
            for i,c in enumerate(q):
                if c == p[idx]:
                    idx += 1
                elif c.upper() == c:
                    # res.append(False)
                    break
                if idx == len(p):
                    q = q[i+1:]
                    res.append(q.lower()==q)
                    break
            if idx < len(p):
                res.append(False)
        return res
