class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        res = []
        ct = 0
        for c in pattern:
            if ord(c) >= ord('A') and ord(c) <= ord('Z'):
                ct += 1 
        
for q in queries:
            #print(q, pattern)
            res.append(self.helper(q, pattern, ct))
        return res
    
    def helper(self, query, pattern, ct):
        if len(query) < len(pattern):
            return False
        count1 = 0
        count2 = 0
        i = 0 # query
        j = 0 # pattern
        while i < len(query) and j < len(pattern):
            if ord(query[i]) >= ord('A') and ord(query[i]) <= ord('Z'):
                count1 += 1
            #if ord(pattern[j]) >= ord('A') and ord(pattern[j]) <= ord('Z'):
            #    count2 += 1 
                           
            if query[i] == pattern[j]:
                j+=1
   
            i+=1
        
        while i != len(query):
            if ord(query[i]) >= ord('A') and ord(query[i]) <= ord('Z'):
                count1 += 1
            i += 1
        #print(i,j,count1,count2)
        return j == len(pattern) and count1 == ct
