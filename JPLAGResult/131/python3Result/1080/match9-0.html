class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        def match(x, y):
            yi=0
            xi=0
            while yi<len(y) or xi<len(x):
                if yi==len(y):
                    if x[xi].isupper(): return False
                    xi+=1
                elif xi==len(x): return False
                elif x[xi]==y[yi]:
                    xi+=1
                    yi+=1
                elif x[xi].islower():
                    xi+=1
                else:
                    return False
            return True
        return [match(x, pattern) for x in queries]
