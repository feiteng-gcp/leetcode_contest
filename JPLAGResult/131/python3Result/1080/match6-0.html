def does_match(q, p):
    i = 0
    for ch in q:
        if ch.isupper():
            if i >= len(p) or p[i] != ch:
                return False
            
            i += 1
        else:
            if i < len(p) and p[i] == ch:
                i += 1
    
    return i == len(p)


class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        res = [False] * len(queries)
        for i, q in enumerate(queries):
            res[i] = does_match(q, pattern)
            
        return res
