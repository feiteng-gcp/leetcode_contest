class Solution:
    def solve(self,s,p):
        n = len(s)
        cnt = 0;
        m = len(p)
        j = 0
        for c in p:
            while(j<n and s[j]!=c):
                if (s[j]>='a' and s[j]<='z'):
                    j+=1
                else:
                    return False
            if j<n:
                j+=1
            else:
                return False
        while( j<n):
            if (s[j]>='a' and s[j]<='z'):
                j+=1
            else:
                return False
        return True
    
def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        
        res=[]
        for s in queries:
            res.append(self.solve(s,pattern))
        return res
