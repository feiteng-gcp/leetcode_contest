class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        res = []
        for q in queries:
            i = 0
            for c in q:
                if i == len(pattern):
                    if c.isupper():
                        res.append(False)
                        break
                else:
                    if c.isupper() and c != pattern[i]:
                        res.append(False)
                        break
                    if c == pattern[i]:
                        i += 1
            else:
                if i == len(pattern):
                    res.append(True)
                else:
                    res.append(False)
        return res
