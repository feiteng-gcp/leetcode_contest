class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        
        def match(query, pattern):
            p = 0
            for ch in query:
                if ord(ch) < ord('a'):
                    if p >= len(pattern) or ch != pattern[p]:
                        return False
                    p += 1
                else:
                    if p < len(pattern) and ch == pattern[p]:
                        p += 1
            return p == len(pattern)
        
        
        ans = []
        for query in queries:
            ans.append(match(query, pattern))
        return ans
