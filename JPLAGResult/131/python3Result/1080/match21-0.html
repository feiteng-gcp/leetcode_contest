class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        if len(queries)==0:
            return [True]
        def substring(p, q):
            i = 0
            j = 0
            while(i<len(p) and j<len(q)):
                if p[i]==q[j]:
                    i+=1
                    j+=1
                elif q[j]==q[j].upper():
                    return False
                else:
                    j+=1
            if i>=len(p):
                while(j<len(q)):
                    if q[j]==q[j].upper():
                        return False
                    j+=1
                return True
            else:
                return False
        lst = []
        for q in queries:
            if substring(pattern, q):
                lst.append(True)
            else:
                lst.append(False)
        return lst
