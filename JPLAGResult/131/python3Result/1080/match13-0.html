class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        res = []
        for q in queries:
            res.append(self.match(pattern, q))
        return res            
        
    def match(self, p, w):
        step = 0
        for i, c in enumerate(w):
            if c == p[step]:
                step += 1
                if step >= len(p):
                    return i == len(w)-1 or w[i+1:].islower()
            elif not c.islower():                
                return False
        return False
