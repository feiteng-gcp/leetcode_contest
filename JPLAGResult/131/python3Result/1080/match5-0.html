import re
class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        ll = '[a-z]*'
        newPattern = []
        newPattern += ll
        for c in pattern:
            newPattern += c
            newPattern += ll

        newPattern = ''.join(newPattern)
        newPattern = '^' + newPattern + '$'

        results = []
        for q in queries:
            if re.match(newPattern, q) is None:
                results.append(False)
            else:
                results.append(True)

        return results
