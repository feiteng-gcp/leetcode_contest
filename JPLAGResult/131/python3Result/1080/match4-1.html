class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        res = []
        for query in queries:
            if self.check(query, pattern):
                res.append(True)
            else:
                res.append(False)
        return res
    
    
def check(self, query, pattern):
        n = len(pattern)
        idx = 0
        for c in query:
            if idx < n and c == pattern[idx]:
                idx += 1
            elif not c.islower():
                return False
        return idx == n
