class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        ans = []
        for query in queries:
            i, j 
= 0, 0
            fail = False
            while i < len(query) and j < len(pattern):
                if query[i] == pattern[j]:
                    i += 1
                    j += 1
                elif query[i] == query[i].lower():
                    i += 1
                elif query[i] == query[i].upper() and pattern[j] == pattern[j].lower():
                    fail = True
                    break
                else:
                    fail = True
                    break
            if fail:
                ans += [False]
            elif i == len(query) and j < len(pattern):
                ans += [False]
            else:
                while i < len(query):
                    if query[i] == query[i].upper():
                        fail = True
                        break
                    i += 1
                if fail:
                    ans += [False]
                else:
                    ans += [True]
        return ans
