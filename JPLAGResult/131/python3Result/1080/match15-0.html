class Solution:
    def camelMatch(self, queries: "List[str]", pattern: str) -> "List[bool]":
        def match(s,p):
            spos=ppos=0
            while ppos<len(p):
                if spos>=len(s): return False
                if s[spos]==p[ppos]:
                    ppos+=1;spos+=1
                elif s[spos]>="A" and s[spos]<="Z" and p[ppos]>="a":
                    return False
                elif p[ppos]>="A" and p[ppos]<="Z" and s[spos]>="a":
                    while spos<len(s) and s[spos]>="a":
                        spos+=1
                elif p[ppos]>="A" and p[ppos]<="Z" and s[spos]<"a":
                    return False
                else:
                    spos+=1
            for i in range(spos,len(s)):
                if s[i]>="A" and s[i]<="Z":
                    return False
            return True

        ans=[]
        for st in queries:
            ans.append(match(st,pattern))
        return ans
