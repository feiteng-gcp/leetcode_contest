from collections import Counter
class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        ret = []        
        for query in queries:
            pattern_index = 0
            cnt = Counter(query)
            query_index = 0
            result = True
            while query_index < len(query) and pattern_index < len(pattern):
                if query[query_index] == pattern[pattern_index]:
                    cnt[query[query_index]] -= 1
                    pattern_index += 1
                query_index += 1
            if not (pattern_index == len(pattern)):
                ret.append(False)
                continue
            for k,v in cnt.items():
                if v < 0:
                    result = False 
                    break
                if v > 0 and k.isupper():
                    result = False 
                    break
            ret.append(result)
        return ret
