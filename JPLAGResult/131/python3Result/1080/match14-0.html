class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        ans = []
        for i in queries:
            ans.append(self.dfs(0, 0, i, pattern))
        return ans
    
    
    def dfs(self, i, j, q, p):
        if i == len(q) and j == len(p):
            return True
        if i>len(q) or j>len(p):
            return False
        
        if i < len(q) and q[i].islower():
            if self.dfs(i+1, j, q, p):
                return True
        if j < len(p) and i < len(q) and q[i] == p[j]:
            if self.dfs(i+1, j+1, q, p):
                return True
        return False
