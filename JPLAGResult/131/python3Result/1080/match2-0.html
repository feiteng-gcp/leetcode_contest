class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        ans = []
        ctp = collections.Counter([c for c in list(pattern) if 'A' <= c <= 'Z'])
        for q in queries:
            ctq = collections.Counter([c for c in list(q) if 'A' <= c <= 'Z'])
            if ctq != ctp:
                ans.append(False)
                continue
            r = True
            for c in pattern:
                if c in q:
                    q = q[q.index(c) + 1:]
                else:
                    r = False
                    break
            ans.append(r)
        return ans
