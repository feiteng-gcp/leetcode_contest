class Solution:
    def match(self, string , pattern):
        lowletter = set('qwertyuiopasdfghjklzxcvbnm')
        i, j = 0, 0
        while i < len(string) and j < len(pattern):
            if string[i] == pattern[j]:
                i += 1
                j += 1
            else:
                if string[i] not in lowletter:
                    return False
                i += 1
        if j < len(pattern):
            return False
        while i < len(string):
            if string[i] not in lowletter:
                return False
            i += 1
        return True
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        ans = []
        for string in queries:
            ans.append(self.match(string, pattern))
        return ans
