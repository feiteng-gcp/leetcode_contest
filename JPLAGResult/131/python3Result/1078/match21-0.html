class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        ans = ''
        in_pare = 0
        
        for c in S:
            if c=='(':
                in_pare+=1
                if in_pare>1:
                    ans+=c
            elif c==')':
                in_pare-=1
                if in_pare>0:
                    ans+=c
        return ans
