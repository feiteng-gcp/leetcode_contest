class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        if len(S)==0:
            return S
        s = ''
        last = 0
        idx = 0
        for i in range(len(S)):
            if S[i]=='(':
                idx-=1
            else:
                idx+=1
            if idx==0:
                # print(i, last)
                s+=S[last+1:i]
                last = i+1
        return s
