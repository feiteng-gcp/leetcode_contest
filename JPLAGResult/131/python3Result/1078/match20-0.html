class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        res = ''
        count = 0
        start = 0
        for i, p in enumerate(S):
            if p == '(':
                count += 1
            elif p == ')':
                count -= 1
                
            
            if i > 0 and count == 0:
                # find valid from start = i
                res += S[start+1:i]
                start = i+1
                #count = 0
                
        return res
