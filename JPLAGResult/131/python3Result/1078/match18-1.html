class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        if S:
            res=str()
            stack
=0
            j=0
            for i in range(len(S)):
                if S[i]=='(':
                    stack+=1
                else:
                    stack-=1
                if stack==0:
                    res+=S[j+1:i]
                    j=i+1
            return res
