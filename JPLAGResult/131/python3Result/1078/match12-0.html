class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        l = 0
        r = 0
        result = ""
        for x in S:
            if x == "(":
                l += 1
                if l != 1:
                    result += x
            if x == ")":
                l -= 1
                if l != 0:
                    result += x
        return result
