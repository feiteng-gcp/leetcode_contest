class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        ans = []
        c = 0
        for s in S:
            if s == '(':
                if c == 0:
                    c += 1
                else:
                    c += 1
                    ans
.append('(')
            elif s == ')':
                c -= 1
                if c != 0:
                    ans.append(')')

        return "".join(ans)
