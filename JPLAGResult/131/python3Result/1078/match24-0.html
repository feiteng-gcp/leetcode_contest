class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        stack = ""
        res = []
        level = 0
        for s in S:
            stack += s
            if s == '(':
                level += 1
            elif s == ')':
                level -= 1
                if level == 0:
                    res.append(stack[1:-1])
                    stack = ""
        return "".join(res)
