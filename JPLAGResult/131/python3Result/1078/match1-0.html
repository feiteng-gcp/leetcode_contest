class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        if S=='':
            return ''
        
        left_cnt , right_cnt = 0,0
        l = 1
        res = ''
        for r,c in enumerate(S):
            if c == '(': 
                left_cnt += 1
            else:
                right_cnt += 1
            if left_cnt == right_cnt:
                left_cnt = right_cnt = 0
                res += S[l:r]
                l = r+2
        return res
