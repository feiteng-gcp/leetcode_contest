class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        l=0
        ans=""
        for c in S:
            if c=="(":
                l+=1
            else:
                l-=1
            if (c=="(" and l>1) or (c==")" and l>0):
                ans+=c
        return ans
