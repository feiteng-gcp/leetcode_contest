class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        level = 0
        result = ""
        for c in S:
            if c == "(":
                if level != 0: result += c
                level += 1
            elif c == ")":
                level -= 1
                if level != 0: result += c
        return result
