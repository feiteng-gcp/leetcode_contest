class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        SUM=0
        ANS=[]
        NOW=""

        for s in S:
            if s=="(":
                SUM+=1
                NOW+="("
            else:
                SUM
-=1
                NOW+=")"

                if SUM==0:
                    ANS.append(NOW[1:-1])
                    NOW=""

        return "".join(ANS)
