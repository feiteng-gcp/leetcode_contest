class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        if not S: return S
        s=S;remain
=0
        ansArr=[]
        for i,ch in enumerate(s):
            if ch=="(":
                if remain!=0:
                    ansArr.append(ch)
                remain+=1
            else:
                if remain!=1:
                    ansArr.append(ch)
                remain-=1
        return "".join(ansArr)
