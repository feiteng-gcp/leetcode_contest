class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        if len(S)==0:
            return ''
        stack=0
        ans=[]
        for i in range(len(S)):
            if S[i]=='(':
                if stack:
                    ans.append('(')
                stack+=1
            else:
                stack-=1
                if stack:
                    ans.append(')')
        return ''.join(ans)
