class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        primitive = []
        cnt = 0
        start = 0
        ans = []
        
        for i, c in enumerate(S):
            
if c == '(':
                cnt += 1
            else:
                cnt -= 1
                if cnt == 0:
                    ans.append(S[start + 1:i])
                    start = i + 1
                    
        return ''.join(ans)
