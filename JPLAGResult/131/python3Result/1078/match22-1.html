class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        left = start = 0
        res = []
        
        for end, char in enumerate(S):
            if char == '(':
                left += 1
            else:
                left -= 1
            
            if left == 0:
                res.append(S[start + 1: end])
                start = end + 1
        
        res.append(S[start + 1:len(S) -1 ])
        return ''.join(res)
