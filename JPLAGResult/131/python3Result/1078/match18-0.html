class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        left = 0
        prev = -1
        ret = ""
        for i in range(len(S)):
            if S[i] == '(':
                left += 1
            else:
                left -= 1
            if left == 0:
                ret += S[prev+2:i]
                prev = i
        return ret
