class Solution:
    def removeOuterParentheses(self, s: str) -> str:
        balance = 0
        start = 0
        
        out = []
        for index, char in enumerate(s):
            if char == '(':
                balance += 1
            else:
                balance -= 1
                
            if balance == 0:
                out.append(s[start+1:index])
                start = index+1
            
        return ''.join(out)
