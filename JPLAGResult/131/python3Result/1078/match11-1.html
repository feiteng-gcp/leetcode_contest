class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        split 
= []
        prev, cnt = 0, 0
        for i, paren in enumerate(S):
            if paren == '(':
                cnt += 1
            else:
                cnt -= 1
            if cnt == 0:
                split.append(S[prev:i+1])
                prev = i + 1
                
        res = ''
        for s in split:
            res += s[1:-1]
        return res
