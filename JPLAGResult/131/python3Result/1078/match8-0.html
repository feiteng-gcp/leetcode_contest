class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        t = 0
        res = ""
        for i in S:
            if i == '(':
                if t != 0:
                    res+=(i)
                t+=1
            else:
                t-=1
                if t != 0:
                    res+=(i)
        return res
