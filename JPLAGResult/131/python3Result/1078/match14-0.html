class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        if not S:
            return ''
        left, right = 0, 0
        ans = ''
        p = 0
        for i in range(len(S)):
            if S[i] == '(':
                left += 1
            else:
                right += 1
            if left == right:
                ans += S[p+1:i]
                p = i+1
        return ans
