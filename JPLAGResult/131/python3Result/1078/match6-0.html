class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        balance = 0
        cur = ''
        ans = ''
        for c in S:
            cur += c
            if c == '(':
                balance += 1
            else:
                balance -= 1
            if balance == 0:
                ans += cur[1:-1]
                cur = ''
        return ans
