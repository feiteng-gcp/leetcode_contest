class Solution:
    def removeOuterParentheses(self, S: str) -> str:
        ans=[]
        count=0
        for i in S:
            ans.append(i)
            if i=='(':
                count+=1
                if count==1:
                    ans.pop()
            else:
                count-=1
                if count==0:
                    ans.pop()
        return ''.join(ans)
