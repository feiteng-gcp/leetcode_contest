/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private static final int mod = (int)1e9+7;
    public int sumRootToLeaf(TreeNode root) {
        return helper(root, 0);
    }
    private int helper(TreeNode root, int parent){
        if(root==null) return parent;
        int res = 0;
        parent = (parent*2+root.val)%mod;
        
if(root.left==null && root.right==null) return parent;
        if(root.left!=null) res+=helper(root.left, parent);
        if(root.right!=null) res+=helper(root.right, parent);
        return res%mod;
    }
}
