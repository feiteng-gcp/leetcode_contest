/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int sumRootToLeaf(TreeNode root) {
        if(root==null) return 0;
        StringBuilder sb=new StringBuilder();
        List<String> list=new ArrayList<>();
        int ans=0;
        dfs(root,sb,list);
        for(String str:list){
            int num=0;
            for(char c:str.toCharArray()){
                num=2*num+(c-'0');
                num%=1000000007;
            }
            ans+=num;
            ans%=1000000007;
        }
        return ans;
    }
    void dfs(TreeNode tn,StringBuilder sb,List<String> list){
        sb.append(tn.val);
        if(tn.left==null&&tn.right==null){
            list.add(sb.toString());
        }else{
            if(tn.left!=null) dfs(tn.left,sb,list);
            if(tn.right!=null) dfs(tn.right,sb,list);
        }
        sb.delete(sb.length()-1,sb.length());
    }
}
