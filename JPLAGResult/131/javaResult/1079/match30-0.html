/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private long res;
    public int sumRootToLeaf(TreeNode root) {
        if (root==null)return 0;
        go(root,0);
        return (int) res;
    }
    
    private void go(TreeNode node, long prev) {
        long xx = (prev * 2 + node.val)%1000000007;
        if (node.left==null&&node.right==null) {
            res = ((res+xx) % 1000000007);
        }
        if (node.left!=null)
            go(node.left,xx);
        if (node.right!=null)
            go(node.right,xx);
    }
}
