/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long result = 0;
	public static int maxN = 1000000007;

	public int sumRootToLeaf(TreeNode root) {
		if (root == null) return 0;
		visit(root, (long)root.val);
		return (int)(result % 1000000007);
	}

	private void visit(TreeNode root, long val) {
		if (root.left == null && root.right == null) {
			result = (result + val) % maxN;
			return;
		}
		if (root.left != null){
			visit(root.left, (val * 2 + root.left.val) % maxN);
		}
		if (root.right != null) {
			visit(root.right, (val * 2 + root.right.val) % maxN);
		}
	}
}
