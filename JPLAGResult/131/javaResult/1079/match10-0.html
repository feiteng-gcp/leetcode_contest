/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int MOD = 1000000007;
    public int sumRootToLeaf(TreeNode root) {
        return sum(root, 0) % MOD;
    }
    
    public int sum(TreeNode cur, int curSum) {
        curSum = ((curSum * 2) + cur.val) % MOD;
        if (cur.left == null && cur.right == null) {
            return curSum;
        }
        int sum = 0;
        if (cur.left != null) sum += (sum(cur.left, curSum)  % MOD);
        if (cur.right != null) sum += (sum(cur.right, curSum)  % MOD);
        return sum;
    }
}
