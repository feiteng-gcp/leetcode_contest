/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long sum = 0;
    int MOD = (int)Math.pow(10, 9) + 7;
    public int sumRootToLeaf(TreeNode root) {
        
if (root == null) return 0;
        sumToLeaf(root, 0);
        return (int)sum;
    }
    
    void sumToLeaf(TreeNode node, long parent) {
        parent = (parent << 1) + node.val;
        parent %= MOD; 
        if (node.left == null && node.right == null) {
            //System.out.println(parent);
            sum += parent;
            sum %= MOD;
        
} else {
            if (node.left != null){
                sumToLeaf(node.left, parent);
            } 
            if (node.right != null)
            {
                sumToLeaf(node.right, parent);
            }
        }
        
        
    }
}
