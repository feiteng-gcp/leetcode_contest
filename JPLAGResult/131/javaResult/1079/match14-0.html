/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    int s = 0;
    public int sumRootToLeaf(TreeNode root) {
        if(root==null) return 0;
        go(root,0);
        
        return s;
    }
    
    public void go(TreeNode r, int v){
        int cur = v*2+r.val;
        cur %= 1000000007;
        if(r.left==null&&r.right==null){
            s += cur;
            s %= 1000000007;
            
            return ;
        }
        
        if(r.left!=null){
            go(r.left,cur);
        }
        if(r.right!=null){
            go(r.right,cur);
        }
        
    }
}
