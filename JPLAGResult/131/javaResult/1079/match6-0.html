/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int sumRootToLeaf(TreeNode root) {
        int md = 1000000000+7;
        int[] sum = {0};
        help5017(root,0,sum,md);
        return sum[0];
    }
    private void help5017(TreeNode root,int cur,int[] sum,int md){
        if(root==null)
            return;
        cur=cur*2+root.val;
        cur%=md;
        if(root.left==null&&root.right==null){
            sum[0]+=cur;
            sum[0]%=md;
            return;
        }
        help5017(root.left,cur,sum,md);
        help5017(root.right,cur,sum,md);
    }
}
