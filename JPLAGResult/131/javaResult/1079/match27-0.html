/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    static long sum;
    public int sumRootToLeaf(TreeNode root) {
        sum = 0;
        if (root == null) return 0;
        recurse(root, (long)0);
        return (int)sum;
    }
    
    public static void recurse(TreeNode curr, long val){
        val *= 2;
        val += curr.val;
        val = val % 1000000007;
        if (curr.left == null && curr.right == null) {
            sum += val;
            sum = sum % 1000000007;
        }
        else{
            if (curr.left != null){
                recurse(curr.left, val);
            }
            if (curr.right != null){
                recurse(curr.right, val);
            }
        }
    }
}
