/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private int ans = 0;
    private int MOD = (int)(1e9 + 7);
    public int sumRootToLeaf(TreeNode root) {
        DFS(root, 0);
        return ans;
    }
    private void DFS(TreeNode r, int sum) {
        if (r == null) return;
        sum = ((sum << 1) + r.val) % MOD;
        if (r.left == null && r.right == null) {
            ans += sum;
            ans %= MOD;
            
return;
        }
        if (r.left != null) {
            DFS(r.left, sum);
        }
        if (r.right != null) {
            DFS(r.right, sum);
        }
    }
}
