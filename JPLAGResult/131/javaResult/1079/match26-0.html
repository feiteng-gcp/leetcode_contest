/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private int sum=0;
    private final int modulo = 1000000007;
    public int sumRootToLeaf(TreeNode root) {
        dfs(root, 0);
        return sum;
    }
    
    private void dfs(TreeNode root, int number){
        //if(root == null) return;
        number = number <<1;
        number = (number+root.val) % modulo;
        if(root.left == null && root.right == null){
            sum = (sum+number)%modulo;
            return;
        }
        if(root.left != null){
            dfs(root.left, number);
        }
        if(root.right != null){
            dfs(root.right, number);
        }
    }
}
