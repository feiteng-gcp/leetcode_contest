/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
  private static final int MOD = 1000000007;

  public int sumRootToLeaf(TreeNode root) {
    return sum(root, 0);
  }

  private int sum(TreeNode root, int agg) {
    if (root == null) {
      return 0;
    }
    int updatedAgg = (agg * 2 + root.val) % MOD;
    if (root.left == null && root.right == null) {
      return updatedAgg;
    }
    return (sum(root.left, updatedAgg) + sum(root.right, updatedAgg)) % MOD;
  }
}
