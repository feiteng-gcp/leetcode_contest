/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private static final int MOD = (int)(1e9 + 7 + 1e-6);
    
    public int sumRootToLeaf(TreeNode root) {
        return dfs(0, root);
    }
    
    private int dfs(int cur, TreeNode root){
        if(root == null){
            return 0;
        }
        cur = (cur << 1 | root.val) % MOD;
        
if(root.left == null && root.right == null){
            //System.out.println(cur);
            return cur;
        }
        return (dfs(cur, root.left) + dfs(cur, root.right)) % MOD;
    }
}
