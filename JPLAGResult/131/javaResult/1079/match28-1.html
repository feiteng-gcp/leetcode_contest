/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int mod = (int)1e9 + 7;
    public int sumRootToLeaf(TreeNode root) {
        return sumRootToLeaf(root, 0);
    }
    
    public int sumRootToLeaf(TreeNode root, int prod) {
        if(root == null)
        {
            return 0;
        }
        prod = (prod * 2 + root.val) % mod;
        if(root.left == null && root.right == null)
        
{
            return prod;
        }
        else
        {
            return (sumRootToLeaf(root.left, prod) + sumRootToLeaf(root.right, prod)) % mod;
        }
    }
}
