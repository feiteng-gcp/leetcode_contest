/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    long mod = 1000000000+7, ans = 0;
    public int sumRootToLeaf(TreeNode root) {
        dfs(root,0);
        ans %= mod;
        return (int) ans;
    }
    void dfs(TreeNode cur, long prev){
        if(cur==null) return;
        prev = (prev*2+cur.val)%mod;
        if(cur.left==null&&cur.right==null){ // being a leaf
            ans = (ans+prev)%mod;
            return;
        }
        dfs(cur.left,prev);
        dfs(cur.right,prev);
    }
}
