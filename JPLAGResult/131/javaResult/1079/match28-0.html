/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private long MOD = 1000000000 + 7;

    public int sumRootToLeaf(TreeNode root) {
        return (int) sumRootToLeaf(root, 0);
    }

    private long sumRootToLeaf(TreeNode root, long parentValue) {
        if (Objects.isNull(root)) {
            return 0;
        }
        long value = (parentValue * 2 + root.val) % MOD;
        if (Objects.isNull(root.left) && Objects.isNull(root.right)) {
            return value;
        } else {
            return (sumRootToLeaf(root.left, value) + sumRootToLeaf(root.right, value)) % MOD;
        }
    }
}
