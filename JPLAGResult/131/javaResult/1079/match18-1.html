/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int res=0;
    int M= 1000000007;
    public int sumRootToLeaf(TreeNode root) {
        if(root == null) return 0;
        dfs(root, 0);
        return res;
    }
    
    private void dfs(TreeNode root, int cur) {
        //leaf
        int newVal = ((cur%M)*2 + root.val)%M;
        if(root.left == null && root.right == null) {
            res = (res + newVal)%M;
            
return;
        } 
        
        if(root.left != null) dfs(root.left, newVal);
        if(root.right != null) dfs(root.right, newVal);
    }
}
