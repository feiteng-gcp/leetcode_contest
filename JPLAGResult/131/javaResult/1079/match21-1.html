/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    private static final int MOD = 1_000_000_007;
    long result;
    public int sumRootToLeaf(TreeNode root) {
        result = 0;
        recursive(root, 0);
        
        return (int) result;
    }
    
    private void recursive(TreeNode root, long before){
        before = ((before << 1) + root.val) % MOD; 
        if(root.left == null && root.right == null){
            result += before;
            result %= MOD;
            
return;
        }
        
        if(root.left != null){
            recursive(root.left, before);
        }
        
        if(root.right != null){
            recursive(root.right, before);
        }
    }
}
