/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int sum = 0;
    
final int MOD = 1000000007;
    public int sumRootToLeaf(TreeNode root) {
        if (root == null) return 0;
        
        dfs(root, 0);
        return sum;
    }
    
    private void dfs(TreeNode root, int pre) {
        pre = pre * 2 + root.val;
        if (pre >= MOD)
            pre -= MOD;
        if (pre >= MOD)
            pre -= MOD;
        if (root.left == null && root.right == null) {
            sum += pre;
            if(sum >= MOD)
                sum -= MOD;
            return;
        
}
        
        if (root.left != null)
            dfs(root.left, pre);
        if (root.right != null)
            dfs(root.right, pre);
    }
}
