/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int sum;
    final int mod = 1000000007;
    public int sumRootToLeaf(TreeNode root) {
        this.sum = 0;
        dfs(root, 0);
        
        return this.sum;
    }
    
    void dfs(TreeNode root, int prev) {
        
        int val = ((prev * 2)%mod + root.val)%mod;
        
        if(root.left == null && root.right == null) {
            sum = (sum+val)%mod;
            return;
        }
        
        if(root.left != null) {
            dfs(root.left, val);
        }
        
        if(root.right != null) {
            dfs(root.right, val);
        }
        
    }
}
