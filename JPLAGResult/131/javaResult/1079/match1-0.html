/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int ans =0;
    int mod = 1000000007;
    public int sumRootToLeaf(TreeNode root) {
        sum(root, 0);
        return ans;
    }
    void sum(TreeNode node, int v){
        if(node == null) return ;
         v = v*2%mod + node.val;
        if(node.left == null && node.right == null){
            ans = (ans + v)%mod;
        }else{
            sum(node.left, v);
            sum(node.right, v);
        }
    }
}
