class Solution {
    long mod = 1000000007;
    public int sumRootToLeaf(TreeNode root) {
        return (int)helper(root, 0L);
    }
    public long helper(TreeNode root, long sum){
        if(root == null){
            return 0;
        }
        //System.out.println(mod);
        //long val = root.val % mod;
        sum = sum % mod;
        sum = ((2 * sum) % mod  + root.val) % mod;
        
if(root.left == null && root.right == null){
            return sum % mod;
        }
        return (helper(root.left, sum) + helper(root.right, sum)) % mod;
    }
}
