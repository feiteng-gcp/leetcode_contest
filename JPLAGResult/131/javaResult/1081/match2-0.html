class Solution {
    public int videoStitching(int[][] clips, int T) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->a[0]==b[0]? b[1]-b[0]-a[1]+a[0] : a[0]-b[0]);
        for(int[] clip : clips){
            pq.offer(clip);
        }
        int res = 0, cur=0;
        while(cur<T && !pq.isEmpty()){
            if(pq.peek()[0]>cur) break;
            cur = pq.poll()[1];
            while(!pq.isEmpty()&&pq.peek()[0]<=cur){
                if(pq.peek()[0]==cur && pq.peek()[1]>cur) break;
                int[] poll = pq.poll();
                
                if(poll[1]>cur) {
                    poll[0] = cur;
                    pq.offer(poll);
                }
            }
            res++;
        }
        return cur>=T ? res : -1;
    }
}
