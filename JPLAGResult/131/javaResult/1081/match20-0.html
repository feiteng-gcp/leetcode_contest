class Solution {
    public int videoStitching(int[][] clips, int T) {
        Arrays.sort(clips, (a, b)->{
            return a[0] == b[0] ? b[1] - a[1] : a[0] - b[0];
        });
        
        int step = 0, des = 0, idx = 0;
        if(clips[0][0] != 0)
            return -1;
        
        while(des < T) {
            int next = des;
            while(idx < clips.length && clips[idx][0] <= des) {
                next = Math.max(next, clips[idx][1]);
                idx++;
            }
            if(next == des)
                return -1;
            des = next;
            step++;
        }
        
        return step;
    }
}
