class Solution {
    public int videoStitching(int[][] clips, int T) {
        Arrays.sort(clips, (a, b) -> a[0] == b[0] ? a[1] - b[1] : a[0] - b[0]);
        
        int i = 0;
        int farest = 0;
        int count = 0;
        
        while(i < clips.length && farest < T) {
            if(clips[i][0] > farest) {
                return -1;
            }
            
            int pre = farest;
            
            while(i < clips.length && clips[i][0] <= pre) {
                farest = Math.max(farest, clips[i][1]);
                i++;
            }
            
            count++;
            //i++;
        }
        
        if(farest < T) {
            return -1;
        }
        
        return count;
    }
}
