class Solution {
    public int videoStitching(int[][] clips, int T) {

		Map<Integer, Integer> map = new HashMap<>();

		for (int[] c : clips) {
			Integer y = map.get(c[0]);
			if (y == null) {
				map.put(c[0], c[1]);
			} else {
				if (c[1] > y) {
					map.put(c[0], c[1]);
				}
			}
		}

		int s = 0;
		int e = 0;
		int res = 0;
		while (true) {

			int max = -1;
			for (int i = s; i <= e; i++) {
				Integer y = map.get(i);
				if (y != null && y > max) {

					max = y;
				}
			}
			if (max == -1) {
				return -1;
			}
			res++;
			s++;
			e = max;
			if (max >= T) {
				break;
			}
		}

		return res;

	}
}
