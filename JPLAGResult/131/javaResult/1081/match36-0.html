class Solution {
  public int videoStitching(int[][] clips, int T) {
    int count = 0;
    for (int i = 0; i < T; count++) {
      int max = 0;
      for (int[] clip : clips) {
        if (clip[0] <= i) max = Math.max(max, clip[1]);
      }
      if (max <= i) return -1;
      i = max;
    }
    return count;
  }
}
