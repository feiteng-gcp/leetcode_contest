class Solution {
    public int videoStitching(int[][] clips, int T) {
        if(T == 0) return 0;
        if(clips == null || clips.length == 0) return -1;
        int[] dp = new int[T+1];
        int[] maxDis = new int[T+1];
        Arrays.fill(dp, Integer.MAX_VALUE);
        dp[0] = 0;
        for(int[] clip : clips) {
            if(clip[0] > T) continue;
            maxDis[clip[0]] = Math.max(maxDis[clip[0]], clip[1]);
        }
        if(maxDis[0] == 0) return -1;
        for(int i = 0; i < T+1; i++){
            for(int j = i+1; j <= maxDis[i] & j < T+1; j++){
                dp[j] = Math.min(dp[i]+1, dp[j]);
            }
        }
        if(dp[T] == Integer.MAX_VALUE) return -1;
        return dp[T];
    }
}
