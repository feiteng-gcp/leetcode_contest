class Solution {
    public int videoStitching(int[][] clips, int T) {
		int n = clips.length;
		int[] f = new int[n];
		Arrays.fill(f, n +1);
		
		for(int i = 0; i < n; i++) {
			for(int j = i + 1; j < n; j++) {
				if (clips[i][1] > clips[j][1]) {
					int[] clip = clips[i];
					clips[i] = clips[j];
					clips[j] = clip;
				}
			}
		}		
		for(int i = 0; i < n; i++) {
			if (clips[i][0] <= 0) f[i] = 1;
		}
		for(int i = 0; i < n; i++) {
			for(int j = i - 1; j >= 0; j--) {
				if (clips[i][0] > clips[j][0] && clips[i][0] <= clips[j][1]) {
					f[i] = Math.min(f[i], f[j] + 1);
				}
			}
		}
		int result = n + 1;
		for(int i = 0; i < n; i++) {
			if (clips[i][1] >= T) {
				result = Math.min(result, f[i]);
			}
		}
		return  result >= n + 1 ? -1 : result;
	}
}
