class Solution {
    public int videoStitching(int[][] cp, int T) {
        
        
        Arrays.sort(cp,(x,y)->{ 
            if(x[0]==y[0]){
                return y[1]-x[1];
            }                   
            return x[0]-y[0];
                               });
        
      //  for(int u[]:cp){
    //        System.out.println(u[0]+","+u[1]);
     //   }
        
        
        
        if(T==0) return 0;
        
        if(cp[0][0]>0) return -1;
        
        int le = cp[0][0];
        int ri = cp[0][1];
        
        int cur = 1;
        
        int cnt = 1;
        while(cur<cp.length&&ri<T){
            
            int tri = -1;
           // cur = cp.length;
            int j=cur;
            for(;j<cp.length;++j){
                if(cp[j][0]<=ri){
                    tri = Math.max(tri,cp[j][1]);
                }else{
                    break;
                }
            }
            
                    cur = j;
            System.out.println(tri);
            
            if(tri>ri){
                ri = tri;
                cnt++;
            } else{
                break;
            }
            
            
        }
        
        if(ri>=T){
            return cnt;
        }else{
            return -1;
        }
        
        
        
        
    }
}
