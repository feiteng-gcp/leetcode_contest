class Solution {
	public int videoStitching(int[][] clips, int T) {
		int last = 0;
		int cnt = 0;
		while (last < T) {
			int[] max = null;
			for (int[] cl : clips)
				if (cl[0] <= last && cl[1] > last) {
					if (max == null || max[1] < cl[1])
						max = cl;
				}
			if (max != null) {
				last = max[1];
				cnt++;
			}
			else
				return -1;
		}
		return cnt;
	}
}
