class Solution {
    public int videoStitching(int[][] clips, int T) {
        int ans = 0;
        int need = 0;
        while(need < T){
            int longest = -1;
            for(int[] clip : clips){
                if(clip[0] < clip[1] && clip[0] <= need && clip[1] > need && clip[1] > longest){
                    longest = clip[1];
                }
            }
            if(longest == -1){
                return -1;
            }
            ans++;
            need = longest;
            //System.out.println(need);
        }
        return ans;
    }
}
