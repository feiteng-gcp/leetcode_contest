class Solution {
   public int videoStitching(int[][] clips, int T) {
		int lastend = 0;
		int res = 0;
		while (lastend < T) {
			// 找start<=lastend的所有，定位到end最大的
			int maxEnd = lastend;

			for (int i = 0; i < clips.length; i++) {
				if (clips[i][0] <= lastend) {
					maxEnd = Math.max(maxEnd, clips[i][1]);
				}
			}
			if (maxEnd == lastend) {
				return -1;
			}
			lastend = maxEnd;
			res++;
		}
		return res;
	}
}
