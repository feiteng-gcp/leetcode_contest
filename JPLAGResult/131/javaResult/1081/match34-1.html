class Solution {
    public int videoStitching(int[][] clips, int T) {
        if (T == 0) {
            return 0;
        }
        TreeMap<Integer, Integer> bst = new TreeMap<>();
        for (int[] a : clips) {
            int start = a[0];
            int end = a[1];
            Integer cur = bst.get(start);
            if (cur == null) {
                bst.put(start, end);
            } else if (cur < end) {
                bst.put(start, end);
            }
        }
        
        if (bst.get(0) == null) {
            return -1;
        }
        
        int[] dp = new int[T + 1];
        dp[0] = bst.get(0);
        for (int i = 1; i <= T; i++) {
            Integer cur = bst.get(i);
            dp[i] = dp[i - 1];
            if (cur != null) {
                dp[i] = Math.max(dp[i], cur);
            }
        }
        
        int cur = 0;
        int ret = 0;
        while (cur < T) {
            int cur1 = dp[cur];
            ret++;
            
if (cur1 <= cur) {
                return -1;
            }
            cur = cur1;
        }
        
        return ret;
    }
}
