class Solution {
    public int videoStitching(int[][] clips, int T) {
		if (T == 0) {
			return 0;
		}
		int[] res = new int[T+1];
		for (int[] clip : clips) {
			if (clip[1] > T) {
				clip[1] = T;
			}
			if (clip[0] <= T) {
				res[clip[0]] = Math.max(res[clip[0]], clip[1]);
			}
		}
		int count = 0;
		int reach = 0;
		int preReach = -1;
		while (reach < T && preReach != reach) {
			int max = reach;
			for (int i = preReach + 1; i <= reach; i++) {
				max = Math.max(res[i], max);
			}
			preReach = reach;
			reach = max;
			count++;
		}
		if (reach == T) {
			return count;
		} else {
			return -1;
		}
    }
}
