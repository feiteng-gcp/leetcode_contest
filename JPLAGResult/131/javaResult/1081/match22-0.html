class Solution {
    public int videoStitching(int[][] clips, int T) {
		int[] dp = new int[110];
		for (int i = 0; i < dp.length; i++) dp[i] = -1;
		for (int i = 0; i < clips.length; i++) {
			clips[i][0] ++;
			clips[i][1] ++;
		}
		T++;
		dp[1] = 0;
		Arrays.sort(clips, new Comparator<int[]>() {
			@Override
			public int compare(int[] o1, int[] o2) {
				return o1[0] - o2[0];
			}
		});
		int inf = 100000000;
		for (int i = 0; i < clips.length; i++) {
			int s = clips[i][0];
			int e = clips[i][1];
			int tmp = inf;
			for (int j = s; j < e; j++) if (dp[j] != -1) {
				tmp = Math.min(tmp, dp[j] + 1);
			}
			if (dp[e] == -1 || dp[e] > tmp) dp[e] = tmp;
		}
		int res = inf;
		for (int i = 105; i >= T; i --) if (dp[i] != -1) res = Math.min(res, dp[i]);
		if (res == inf) return -1;
		return res;
	}
}
