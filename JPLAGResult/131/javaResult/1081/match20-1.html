class Solution {
    public int videoStitching(int[][] clips, int T) {
        
        Comparator<int[]> cmptor = new Comparator<int[]>() {
            public int compare(int[] c1, int[] c2) {
                return c1[0] - c2[0];                
            }
        };
        
        Arrays.sort(clips, cmptor);
        int n = clips.length;
        int ptr = 0;
        int next = 0;
        int res = 0;
        while (next < T) {
            
            int tmp = next;
            
            while (ptr < n && clips[ptr][0] <= next) {
                if (clips[ptr][1] > tmp) tmp = clips[ptr][1];
                ++ptr;
            
}
            if (tmp == next) return -1;
            next = tmp;
            ++res;
            
        } 
        return res;
        
    }
}
