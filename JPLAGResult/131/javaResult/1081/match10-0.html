class Solution {
    int inf = (int)1e8;
    public int videoStitching(int[][] clips, int T) {
        int[] dp = new int[T];
        for(int i = 0; i < T; i++)
        {
            dp[i] = inf;
            for(int[] clip : clips)
            {
                if(clip[0] <= i && clip[1] > i)
                {
                    if(clip[0] == 0)
                    {
                        dp[i] = Math.min(1, dp[i]);
                    }
                    for(int j = Math.max(clip[0] - 1, 0); j < i; j++)
                    {
                        dp[i] = Math.min(dp[i], dp[j] + 1);
                    }
                }
            }
        }
         return dp[T - 1] >= inf ? -1 : dp[T - 1];
    }
    
}
