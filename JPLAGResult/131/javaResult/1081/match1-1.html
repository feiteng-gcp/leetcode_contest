class Solution {
    public int videoStitching(int[][] clips, int T) {
        int ans=0;
        Arrays.sort(clips, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[0]-o2[0];
            }
        });
        int left=0,right=0,max=0,idx=0;
        for(int[] clip : clips){
            if(clip[0]>right) {
                if(idx==0) return -1;
                left=clips[max][0];
                right=clips[max][1];
                ans++;
            }
            if(right>=T) return ans;
            if(clip[0]>=left&&clip[0]<=right){
                max=clips[idx][1]>clips[max][1]?idx:max;
            }
            idx++;
        }
        return clips[max][1]>=T?(ans+1):-1;
    }
}
