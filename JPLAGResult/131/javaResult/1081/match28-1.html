class Solution {
    public int videoStitching(int[][] clips, int T) {
        Arrays.sort(clips, (a,b) -> (a[0] - b[0]));
        
        int start=0;
        int end=0;
        int cnt=0;
        if(clips.length == 0 || clips[0][0] > 0) return -1;
        for(int i=0; i<clips.length; i++) {
            int tmp = end;
            while(i<clips.length && clips[i][0]<=start) {
                if(clips[i][1] > end) {
                    end = clips[i][1];
                }
                i++;
            }
            if(end>tmp) cnt++;
            if(end >= T) return cnt;
            
if(i>=clips.length || clips[i][0] > end) return -1;
            start=end;
            i--;
        }
        
        return -1;
    }
}
