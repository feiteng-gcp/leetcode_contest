class Solution {
    public int videoStitching(int[][] clips, int T) {
        Arrays.sort(clips, (a,b) -> ((a[0]==b[0])?(a[1]-b[1]):(a[0]-b[0])));
        int totalClips = 0;
        int start = 0;
        int end = -1;
        for (int i=0; i< clips.length; i++) {
            if (clips[i][0] <= start) {
                if (end < clips[i][1]) {
                    end = clips[i][1];
                }
                //System.out.println(clips[i][0]+" "+clips[i][1]+" "+start+" "+end+" "+totalClips);
            }
            else {
                start = end;
                totalClips++;
                if (end >= T) return totalClips;
                if (clips[i][0] > start) return -1;
                i--;
            }
        }
        //System.out.println(totalClips+" "+end+" "+T);
        if (end >= T) return totalClips+1;
        return -1;
    }
}
