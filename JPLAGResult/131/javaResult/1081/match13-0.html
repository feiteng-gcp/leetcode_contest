class Solution {
    public int videoStitching(int[][] clips, int T) {
        Arrays.sort(clips, (a, b) -> a[0] == b[0] ? b[1] - a[1] : a[0] - b[0]);
        if(clips[0][0] != 0)
            return -1;
        int end = clips[0][1];
        if(end >= T)
            return 1;
        int max = end;

        int cnt = 1;
        for(int i = 1; i < clips.length; i++){
            if(clips[i][0] <= end){
                max = Math.max(max, clips[i][1]);
            }else{
                if(end != max){
                    cnt++;
                    end = max;
                }
                if(end >= T)
                    return cnt;
            }
        }
        if(end != max){
            end = max;
            cnt++;
        }
        return end >= T ? cnt : -1;
    }
}
