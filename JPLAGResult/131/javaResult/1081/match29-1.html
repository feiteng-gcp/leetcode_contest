class Solution {
    public int videoStitching(int[][] clips, int T) {
        Arrays.sort(clips, (a,b)-> (a[0] - b[0]));
        TreeMap<Integer, Integer> map = new TreeMap();
        for(int[] clip : clips){
            Map.Entry<Integer,Integer> pre = map.floorEntry(clip[0]);
            if(pre == null || pre.getValue() < clip[1] )
               map.put(clip[0],clip[1]);
        }
        Integer pos = 0;
         int cnt = 0;
        while(pos < T){
            Map.Entry<Integer,Integer> next = map.floorEntry(pos);
            
if(next == null || pos == next.getValue()) return -1;
            pos = next.getValue();
            cnt++;
        }
        return cnt;
    }
}
