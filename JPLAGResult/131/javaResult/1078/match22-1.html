class Solution {
    public String removeOuterParentheses(String S) {
        if (S.length() <= 2) return "";
        int count = 0;
        StringBuilder sb = new StringBuilder();
        
for (int i = 0; i < S.length(); i++) {
            if (S.charAt(i) == '(') {
                count++;
                if (count > 1) 
                    sb.append(S.charAt(i));
            } else if (S.charAt(i) == ')') {
                if (count > 1) 
                    sb.append(S.charAt(i));
                count--;
            }
            
        }
        return sb.toString();
    }
}
