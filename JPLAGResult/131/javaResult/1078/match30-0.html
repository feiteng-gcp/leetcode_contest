class Solution {
        public String removeOuterParentheses(String S) {
        StringBuilder sb = new StringBuilder();
        int count = 0;
        boolean start = false;
        for(char c:S.toCharArray()){

            if(c==')'){
                --count;
                if(count==0 && start == true)
                    start=false;
            }
            if(start){
                sb.append(c);
            }
            if(c=='(') {
                ++count;
                if (count == 1 && start == false)
                    start = true;
            }
        }
        return sb.toString();
    }
}
