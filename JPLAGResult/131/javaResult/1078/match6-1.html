class Solution {
    public String removeOuterParentheses(String S) {
        int len = S.length();
        StringBuffer sb = new StringBuffer();
        int lastpos = 0;
        int stack = 1;
        for(int i=1;i<len;i++){
            if(S.charAt(i)=='(')stack++;
            else if(S.charAt(i)==')') stack--;
            if(stack==0){
                sb.append(S.substring(lastpos+1,i));
                stack = 1;
                i = i+1;
                
lastpos = i;
            }
        }
        return sb.toString();
    }
}
