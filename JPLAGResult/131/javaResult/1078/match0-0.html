class Solution {
    public String removeOuterParentheses(String S) {
        List<String> list = new ArrayList<>();
        int i = -1;
        int cnt = 0;
        for(int j = 0; j < S.length(); j++) {
            if(S.charAt(j) == '(') {
                cnt++;
            } else {
                cnt--;
            }
            
            if(cnt == 0) {
                list.add(S.substring(i + 1, j + 1));
                i = j;
            }
        }
        
        StringBuilder sb = new StringBuilder();
        for(String str : list) {
            sb.append(str.substring(1, str.length() - 1));
        }
        
        return sb.toString();
        
    }
}
