class Solution {
    public String removeOuterParentheses(String S) {

		List<String> res = new ArrayList<>();
		
		int c = 0;
		int lastStart = 0;
		for (int i = 0; i < S.length(); i++) {
			if (S.charAt(i) == '(') {
				c++;
			} else {
				c--;
			}
			if (c == 0) {
				res.add(S.substring(lastStart + 1, i));
				lastStart = i + 1;
			}
		}
		
		return res.stream().collect(Collectors.joining());
		
	}
}
