class Solution {
    public String removeOuterParentheses(String S) {
        StringBuilder result = new StringBuilder();
        int count = 0;
        int start = 0;
        for(int i = 0; i < S.length(); i++){
            if(S.charAt(i) == '('){
                count++;
            }
            else{
                count--;
                if(count == 0){
                    result.append(S.substring(start + 1, i));
                    start = i + 1;
                }
            
}
        }
        
        return result.toString();
    }
}
