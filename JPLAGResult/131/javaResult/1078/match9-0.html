	class Solution {
	    public String removeOuterParentheses(String S) {
	        char[] s = S.toCharArray();
	        int h = 0;
	        StringBuilder sb = new StringBuilder();
	        for(char c : s){
	        	if(c == '('){
	        		if(h > 0){
	        			sb.append(c);
	        		}
	        		h++;
	        	}else{
	        		if(h > 1){
	        			sb.append(c);
	        		}
	        		h--;
	        	}
	        }
	        return sb.toString();
	    }
	}
