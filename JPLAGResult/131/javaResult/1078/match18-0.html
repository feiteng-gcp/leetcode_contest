class Solution {
    public String removeOuterParentheses(String S) {
        StringBuilder sb = new StringBuilder();
        int sum=0, ileft=0;
        for(int i=0; i<S.length(); i++){
            char ch = S.charAt(i);
            if(ch == '('){
                sum++;
            }else{
                sum--;
            }
            if(sum == 0){
                int iright = i;
                sb.append(S.substring(ileft+1, iright));
                ileft=iright+1;
            }
        }
        return sb.toString();
    }
}
