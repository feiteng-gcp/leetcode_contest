class Solution {
    public String removeOuterParentheses(String S) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0, left = 0, cnt = 0; i < S.length(); i++) {
            if (S.charAt(i) == '(') {
                cnt ++;
            } else {
                cnt --;
                if (cnt == 0) {
                    sb.append(S.substring(left + 1, i));
                    left = i + 1;
                }
            }
        }
        return sb.toString();
    }
}
