class Solution {
    public String removeOuterParentheses(String s) {
        char cs[] = s.toCharArray();
        StringBuilder sb = new StringBuilder();
        
        int n = 0;
        int l = 0;
        for(int i=0;i<cs.length;++i){
            char c = cs[i];
            if(c=='('){
                ++n;
            }else{
                --n;
            }
            if(n==0){
                sb.append(s.substring(l+1,i));
                l = i+1;
            }
        }
        return sb.toString();
        
    }
}
