class Solution {
    public String removeOuterParentheses(String S) {
        StringBuilder builder = new StringBuilder();
        int depth = 0;
        for(char c : S.toCharArray())
        {
            if(c == '(')
            {
                depth++;
                if(depth != 1)
                {
                    builder.append('(');
                }
            }
            else
            {
                depth--;
                if(depth != 0)
                {
                    builder.append(')');
                }
            
}
        }
        return builder.toString();
    }
}
