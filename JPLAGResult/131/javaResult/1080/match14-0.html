class Solution {
    public List<Boolean> camelMatch(String[] queries, String pattern) {
        List<Boolean> list = new ArrayList<>(queries.length);
        for(String q: queries){
            list.add(testMatch(q, pattern));
        }
        return list;
    }
    
    private boolean testMatch(String q, String pattern){
        int i=0, j=0;
        while(i<q.length() && j<pattern.length()){
            if(q.charAt(i) == pattern.charAt(j)){
                i++; j++;
            }else if(q.charAt(i) >= 'a' && q.charAt(i) <= 'z'){
                i++;
            }else{
                return false;
            }
        }
        if(j<pattern.length()){
            return false;
        }else{ //j==pattern.length(), pattern ends
            while(i<q.length()){
                if(q.charAt(i) >= 'A' && q.charAt(i) <= 'Z'){
                    return false;
                }
                i++;
            }
        }
        return true;
    }
}
