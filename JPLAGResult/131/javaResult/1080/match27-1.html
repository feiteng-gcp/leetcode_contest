class Solution {
    public List<Boolean> camelMatch(String[] queries, String pattern) {
        if (queries.length == 0) {
            return Collections.emptyList();
        }
        
List<Boolean> matchedResult = new ArrayList<>(queries.length);
        for (String query : queries) {
            matchedResult.add(canMatch(query, pattern));
        }
        return matchedResult;
    }
    
    private boolean canMatch(String query, String pattern) {
        int j = 0;
        for (int i = 0; i < pattern.length(); i++) {
            while (j < query.length() && query.charAt(j) != pattern.charAt(i) && query.charAt(j) >= 'a' && query.charAt(j) <= 'z') {
                j++;
            }
            if (j == 
query.length()) {
                return false;
            }
            if (query.charAt(j) == pattern.charAt(i)) {
                j++;
                continue;
            }
            return false;
        }
        
while (j < query.length()) {
            if (query.charAt(j) < 'a' || query.charAt(j) > 'z') {
                return false;
            }
            j++;
        }
        return true;
    }
}
