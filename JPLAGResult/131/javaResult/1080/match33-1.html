class Solution {
    public List<Boolean> camelMatch(String[] queries, String pattern) {
        List<Boolean> r = new ArrayList();
        for(String q : queries)
        {
            r.add(insertable(q, pattern));
        }
        return r;
    }
    
    public static boolean insertable(String q, String p)
    {
        int i = q.length() - 1;
        int j = p.length() - 1;
        while(i >= 0 && j >= 0)
        {
            if(q.charAt(i) != p.charAt(j))
            {
                if(q.charAt(i) < 'a' || q.charAt(i) > 'z')
                {
                    return false;
                }
                i--;
                
            }
            else
            {
                i--;
                j--;
            }
        }
        while(i >= 0)
        
{
            if(q.charAt(i) < 'a' || q.charAt(i) > 'z')
                {
                    return false;
                }
                i--;
        }
        return j == -1;
    }
}
