class Solution {
    
    public boolean ok(String q, String p){
        
        char qs[] = q.toCharArray();
        
        char ps[] = p.toCharArray();
        
        int g = 0;
        
        int len = ps.length;
        
        for(char c:qs){
            if(c>='a'&&c<='z'){
                
                if(g==len){
                    
                }else{
                    
                    if(c==ps[g]){
                        ++g;
                    }else{
                        
                    }
                    
                    
                } 
                
                
            }else{
                
                if(g==len) return false;
                
                if(c!=ps[g]){
                    return false;
                }else{
                    ++g;
                }
                
                
            }
            
        }
        
        return g==len;
        
    }
    
    
    public List<Boolean> camelMatch(String[] qs, String p) {
        
        List<Boolean> li = new ArrayList<>();
        
        int l = qs.length;
        
        
        for(String q:qs){
            if(ok(q,p)){
                li.add(true);
            }else{
                li.add(false);
            }
        }
        
        return li;
    }
}
