class Solution {
private boolean check(String query , String pattern) {
		int i = 0 , j = 0 , length1 = query.length() , length2 = pattern.length();
		while (i < length1 && j < length2) {
			if (query.charAt(i) == pattern.charAt(j)) {
				i ++;
				j ++;
			} else {
				if (Character.isLowerCase(query.charAt(i)) == false) {
					return false;					
				}
				i ++;				
			}
		}
		if (j == length2) {
			while (i < length1 && Character.isLowerCase(query.charAt(i))) {
				i ++;
			}
			return i == length1;
		} else {
			return false;
		}
	}
	
    public List<Boolean> camelMatch(String[] queries, String pattern) {
    	List<Boolean> ans = new ArrayList<>();
    	for (String query : queries) {
    		if (check(query , pattern)) {
    			ans.add(true);
    		} else {
    			ans.add(false);
    		}
    	}
    	return ans;
    }
    
}
