class Solution {
    public List<Boolean> camelMatch(String[] queries, String pattern) {
        List<Boolean> result=new ArrayList<>();
        
        for(int i=0;i<queries.length;i++){
            if(check(queries[i],pattern)){
                result.add(true);
            } else
                result.add(false);
        
}
        
        return result;
    }
    
    public boolean check(String q, String p){
        if(q.length()<p.length()) return false;
        int i=0;
        int j=0;
        
        while(i<q.length()&&j<p.length()){
            char c1=q.charAt(i);
            char c2=p.charAt(j);
            if(c1>='A'&&c1<='Z'&&c2>='A'&&c2<='Z'&&c1!=c2) return false;
            else if(c1>='A'&&c1<='Z'&&c2>='A'&&c2<='Z'&&c1==c2){
                i++;
                j++;
            }else if(c1==c2&&c1>='a'&&c1<='z'&&c2>='a'&&c2<='z'){
                i++;
                j++;
            }else if(c1!=c2&&c1>='a'&&c1<='z'){
                i++;
            }else
                return false
;
        }
        
        while(i<q.length()){
            char c1=q.charAt(i);
            if(c1>='A'&&c1<='Z')
                
return false;
            i++;
        }
        
        return j==p.length();
    }
    
    
    
    
}
