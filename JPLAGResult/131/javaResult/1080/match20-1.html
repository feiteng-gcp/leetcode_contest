class Solution {
    
public List<Boolean> camelMatch(String[] queries, String pattern) {
        List<Boolean> ret = new ArrayList<>();
        for (String s : queries) {
            if (isOK(s, pattern)) {
                ret.add(true);
            } else {
                ret.add(false);
            }
        }
        return ret;
    }
    
    
private boolean isOK(String s, String p) {
        int sstart = 0;
        int pstart = 0;
        int slen = s.length();
        int plen = p.length();
        while (pstart < plen) {
            while (sstart < slen && s.charAt(sstart) != p.charAt(pstart)) 
{
                if (s.charAt(sstart) >= 'A' && s.charAt(sstart) <= 'Z') {
                    return false;
                }
                sstart++;   
            }
            if (sstart == slen) {
                return false;
            }
            sstart++;
            pstart++;
        }
        while (sstart < slen) {
                if (s.charAt(sstart) >= 'A' && s.charAt(sstart) <= 'Z') {
                    return false;
                }
                sstart++;            
        }
        return true;
    }
}
