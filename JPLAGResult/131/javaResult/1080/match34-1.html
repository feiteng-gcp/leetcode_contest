class Solution {
    public List<Boolean> camelMatch(String[] queries, String pattern) {


		List<Boolean> res = new ArrayList<>();


		for (String q : queries) {

			int i = 0;
			int j = 0;
			boolean isValid = true;
			while (i < q.length() && j < pattern.length()) {

				char x = q.charAt(i);
				char y = pattern.charAt(j);
				boolean isUpper = Character.isUpperCase(x);

				if (x == y) {
					i++;
					j++;
				} else {
					if (isUpper) {

						isValid = false;
						break;
					}
					i++;
				}
			}
			if (!isValid) {
				res.add(Boolean.FALSE);
			} else {
				if (j != pattern.length()) {
					res.add(false);
				} else {
					boolean b = true;
					while (i < q.length()) {

						if (Character.isUpperCase(q.charAt(i))) {
							b = false;
							break;
						}
						i++;
					}
					res.add(b);
				}
			}

		}
		return res;
	}
}
