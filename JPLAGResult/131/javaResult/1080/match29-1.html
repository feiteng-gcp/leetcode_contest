class Solution {
    public List<Boolean> camelMatch(String[] queries, String pattern) {
        List<Boolean> result = new ArrayList<>(queries.length);
        
        for(int i = 0; i < queries.length; ++i) {
            result.add(test(pattern, queries[i]));
            
            
//             for(int j = 0; j < q.length(); j++) {
                
//                 if()
//             }
        
}
        
        return result;
    }
    
    private Boolean test(String pattern, String q) {
        int pChar = 0;
        for(Character c : q.toCharArray()) {
            Boolean isUpper = Character.isUpperCase(c);
            if(pChar == pattern.length()) {
                if(isUpper) return false;
            } else if(c == pattern.charAt(pChar)) {
                pChar++;
            } else if(isUpper) {
                return false;
            
}
        }
        return pChar == pattern.length();
    }
}
