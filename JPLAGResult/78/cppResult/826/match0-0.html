class Solution {
public:
    double helper(int xa, int xb, map<pair<int, int>, double> &dp){
        //cout << xa << " " << xb << endl;
        if(xa < 0 && xb > 0) return 1;
        else if(xa == 0 && xb > 0) return 1;
        else if(xa <= 0 && xb <= 0) return 0.5;
        else if(xa >= 0 && xb <= 0) return 0;
        pair<int, int> p = make_pair(xa, xb);
        if(dp.find(p) != dp.end()) return dp[p];
        dp[p] = 0.25 * (helper(xa-100, xb, dp) + helper(xa-75, xb-25, dp) + helper(xa-50, xb-50, dp) + helper(xa-25, xb-75, dp));
        //cout << dp[p] << endl;
        return dp[p];
    }
    double soupServings(int N) {
        if(N > 9999) return 1;
        map<pair<int, int>, double> dp;
        return helper(N, N, dp);
    }
};
