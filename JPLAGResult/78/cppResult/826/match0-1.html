class Solution {
public:
    double dfs(map<pair<int, int>, double>& tab, int a, int b)
{
    pair<int, int> cur = {a, b};
    if(tab.count(cur)) return tab[cur];
    if(!a && b) return 1;
    else if(!a && !b) return 0.5;
    else if(!b) return 0;
    double& ans = tab[cur];
    return ans = (dfs(tab, max(0, a - 100), b) + dfs(tab, max(0, a - 75), max(0, b - 25)) + dfs(tab, max(0, a - 50), max(0, b - 50)) + dfs(tab, max(0, a - 25), max(0, b - 75))) / 4;

}

double soupServings(int N)
{
    if(N >= 1e4) return 1;
    map<pair<int, int>, double> tab;
    return dfs(tab, N, N);
}
};
