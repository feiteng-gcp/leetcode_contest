class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int re = 0;
        vector<pair<char, int>> v;
        char lastChar = S[0]; int count = 1; 
        for (int index = 1; index < S.size(); ++index) {
            if (S[index] == lastChar) count++;
            else {
                v.push_back(make_pair(lastChar, count));
                lastChar = S[index];
                count = 1;
            }
        }
        v.push_back(make_pair(lastChar, count));
        
        for (auto word : words) {
            if (word.size() > S.size()) continue;
            char lastChar2 = word[0]; int count2 = 1; int vIndex = 0;
            bool flag = true;
            for (int index = 1; index < word.size(); ++index) {
                if (word[index] == lastChar2) count2++;
                else {
                    if (vIndex < v.size() && lastChar2 == v[vIndex].first && (v[vIndex].second == count2 || (v[vIndex].second > count2 && v[vIndex].second >= 3))) {
                        vIndex++;
                    }
                    else {
                        flag = false;
                        break;
                    }
                    lastChar2 = word[index];
                    count2 = 1;
                }
            }
            if (flag && vIndex < v.size() && lastChar2 == v[vIndex].first && (v[vIndex].second == count2 || (v[vIndex].second > count2 && v[vIndex].second >= 3))) {
                vIndex++;
            } else flag = false;
            if (vIndex != v.size()) flag = false;
            if (flag) {
                //cout << word << endl;
                re++;
            }
        }
        return re;
    }
};
