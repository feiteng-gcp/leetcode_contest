class Solution {
public:
    
    
    int expressiveWords(string a, vector<string>& w) {
        
        
        vector<char> sca;
        vector<int> sfa;
        rle(a, sca, sfa);
        
        int n = w.size();
        int ret = 0;
        for(int i=0; i<n; i++){
            string word = w[i];
            
            vector<char> wca;
            vector<int> wfa;
            
            rle(word, wca, wfa);

            if(wca.size() != sca.size()) continue;
            bool ok = true;
            
            for(int j=0; j<sca.size(); j++){
                //goto endloop;
                if(wca[j] != sca[j] || wfa[j] > sfa[j] || (wfa[j] == 1 && sfa[j] == 2)) ok = false; 
            }
            if(!ok) continue;
            ret++;
            
            //endloop:
            
        }
        return ret;
        
    }
    
    void rle(string s, vector<char>& ca, vector<int>& fa){
        int n = s.size();
        
        int i=0;
        
        while(i < n){
            char c = s[i];
            int j = i;
            while(j < n && s[j] == c){
                j++;
            }
            
            ca.push_back(c);
            fa.push_back(j-i);
            i = j;
        }
    }
};
