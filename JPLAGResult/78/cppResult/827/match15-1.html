class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        if (S.size()==0) return 0;
        vector<pair<char,int> > inVec;
        inVec.push_back(pair<char,int>(S[0],1));
        for(int i=1;i<S.size();++i) {
            if (S[i]==inVec.back().first) ++(inVec.back().second);
            else inVec.push_back(pair<char,int>(S[i],1));
        }
        
        int ret=0;
        for(int i=0;i<words.size();++i) {
            vector<pair<char,int> > curVec;
            string curS = words[i];
            
            if (curS.size()==0) continue;
                        
            curVec.push_back(pair<char,int>(curS[0],1));
            for(int i=1;i<curS.size();++i) {
                if (curS[i]==curVec.back().first) ++(curVec.back().second);
                else curVec.push_back(pair<char,int>(curS[i],1));
            }
            

            if (curVec.size()!=inVec.size()) continue;
            bool flag = true;
            for(int i=0;i<curVec.size();++i) {
                if (curVec[i].first!=inVec[i].first || curVec[i].second > inVec[i].second || (inVec[i].second < 3 && curVec[i].second < inVec[i].second)) {flag = false; break;}                
            }
            if (flag) ++ret;
        }
        return ret;
    }
};
