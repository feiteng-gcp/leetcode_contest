class Solution {
    vector<pair<char,int>> Parse(string S){
        char c = S[0];
        int count = 1;
        vector<pair<char,int>> vec;
        for(int  i=1; i<S.size();i++){
            if(S[i] == S[i-1]){
                count++;
            }
            else
            {
                vec.push_back({c,count});
                c = S[i];
                count = 1;
            }
        }
        vec.push_back({c, count});
        return vec;
    }
    bool Match(vector<pair<char,int>> vec, vector<pair<char,int>> vec2)

    {
        if(vec.size() != vec2.size())
            return false;
        for(int j =0 ; j <vec.size();j++){
            if(vec[j].first != vec2[j].first)
                return false;
            if(vec[j].second == vec2[j].second) continue;
            if(vec[j].second < 3)
                return false;
        }
        return true;
    }
public:
    int expressiveWords(string S, vector<string>& words) {
        if(S.size() == 0) return 0;
        vector<pair<char,int>> vec = Parse(S);
        //for (auto x : vec){
        //    cout<<x.first<<":"<<x.second<<endl;
        //}
        int count = 0;
        for(int i=0; i<words.size();i++){
            string s = words[i];
            vector<pair<char,int>> vec2 = Parse(s);
            if(Match(vec,vec2)) count++;
        }
        return count;
    }
};
