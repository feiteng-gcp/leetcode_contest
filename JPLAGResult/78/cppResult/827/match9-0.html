class Solution {
public:
  struct Seq {
    char c;
    int n;
    
    Seq(char c, int n):
      c(c), n(n) {}
  };
  
  int expressiveWords(string S, vector<string>& words) {
    vector<Seq> seqs = make_seq(S);
    int ret = 0;
    for (const auto &w: words) {
      vector<Seq> seqw = make_seq(w);
      if (valid(seqs, seqw)) ++ret;
    }
    return ret;
  }
  
  bool valid(const vector<Seq> &s, const vector<Seq> &w) {
    if (s.size() != w.size()) return false;
    for (int i = 0; i != s.size(); ++i) {
      if (s[i].c != w[i].c) return false;
      if (s[i].n == w[i].n) continue;
      if (s[i].n < 3 || w[i].n > s[i].n) return false;
    }
    return true;
  }
  
  vector<Seq> make_seq(string s) {
    vector<Seq> seqs;
    int count = 1;
    for (int i = 1; i < s.size(); ++i) {
      if (s[i] == s[i - 1]) {
        ++count;
      } else {
        seqs.emplace_back(s[i - 1], count);
        count = 1;
      }
    }
    seqs.emplace_back(s[s.size() - 1], count);
    return seqs;
  }
};
