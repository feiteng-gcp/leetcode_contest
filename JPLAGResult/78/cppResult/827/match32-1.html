class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int res = 0;
        for(int i = 0; i < words.size(); i++){
            string word = words[i];
            int p = 0;
            int q = 0;
            while(p <S.size()){
                if(q<word.size() && S[p] == word[q]){
                    p ++;
                    q ++;
                    continue;
                }
                if(p == 0 || S[p] != S[p - 1])
                    break;
                int count = 0;
                int k = p;
                while(k > 0 && S[k - 1] == S[k]) k --, count ++;
                while(p < S.size() && S[p+1] == S[p]) p++, count++;
                if(count < 2)
                    break;
                p++;
            }
            if(p == S.size() && q == word.size())
                res ++;
        }
        return res;
    }
};
