class Solution {
public:
    vector<pair<char, int> > stretch(string &s) {
	vector<pair<char,int> > v;
	if (s.size() == 0) {
		return v;
	}
	v.push_back(make_pair(s[0],1));
	for(int i = 1; i < s.size(); i++) {
		if (s[i] == s[i-1]) {
			v[v.size()-1].second++;
		} else {
			v.push_back(make_pair(s[i], 1));
		}
	}
	return v;
}

int expressiveWords(string S, vector<string>& words) {
	if (S.size() == 0) {
		return 0;
	}
	int total = 0;
	vector<pair<char, int> > s =  stretch(S);
	for(auto w: words) {
		vector<pair<char, int> > p = stretch(w);
		if (p.size() == s.size()) {
			bool bad = false;

			for(int i = 0; i < p.size(); i++) {
				// cout<<p[i].first<<"|"<<s[i].first<<"|"<<p[i].second<<"|"<<s[i].second<<endl;
				if (p[i].first != s[i].first) {
					bad = true;
					break;
				} else if ((s[i].second<3 && s[i].second == p[i].second) || (s[i].second >= 3 && s[i].second >= p[i].second)) {
					continue;
				} else {
					bad = true;
					break;
				}
			}
			if (!bad) {
				total++;
			}
		}
	}
	return total;
}
};
