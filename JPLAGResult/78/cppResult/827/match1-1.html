class Solution {
public:
     vector< pair<char,int> > cal(string &s){
        vector< pair<char,int> > v;
        for (char c:s){
            if ( v.size() && v.back().first == c ) v.back().second++;

            else v.push_back({c,1});
        }
         return v;
    }
    
    bool check(vector< pair<char,int> > &V, vector< pair<char,int> >  &v){
        if (V.size() != v.size() ) return 0;
        
        for (int i=0; i<v.size(); i++){
            if ( V[i].first != v[i].first )return 0;
            if ( V[i].second == v[i].second ) continue;
            if ( V[i].second>=3 && V[i].second > v[i].second ) continue;
            return 0;
        }
        
        return 1;
    } 
    
    int expressiveWords(string S, vector<string>& words) {
        auto V = cal(S);
        int ans = 0;
        for ( auto &s:words ){
            auto v = cal(s);
            ans += check(V,v);
        }
        return ans;
    }
};
