class Solution {
public:
    vector<pair<char, int>  > conv(string s){
        vector<pair<char, int>  >  res;
        pair<char, int> x;
        x = make_pair((char)27, -1);
        for(auto c: s){
            if(c!=x.first){
                if(x.second>0) res.push_back(x);
                x = make_pair(c, 1);
            } else
                x.second++;
        }
        if(x.second>0) res.push_back(x);
        return res;
    }
    
    int expressiveWords(string S, vector<string>& words) {
        vector<pair<char, int> > m;
        m = conv(S);
        int ans = 0;
        for(string w:words){
            vector<pair<char, int> > f;
            f = conv(w);
            if (m.size()!=f.size()) continue;
            bool flag=true;
            for(int i =0;i<m.size();i++)
                if(m[i].first!=f[i].first||(m[i].second+f[i].second==3)||(m[i].second<3&&f[i].second>=3)) {flag = false; break;}
            if (flag) ans ++;
        }
        return ans;
    }
};
