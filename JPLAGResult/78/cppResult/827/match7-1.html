class Solution {
public:
    vector<pair<char, int>> conv(const string& s)
    {
        vector<pair<char, int>> res;
        for (int i = 0; i < s.size(); ++i)
        {
            if (i > 0 && s[i] == s[i - 1])
                ++res.back().second;
            else
                res.push_back({s[i], 1});
        }
        return res;
    }
    
    int expressiveWords(string S, vector<string>& words) {
        vector<pair<char, int>> ori = conv(S);
        int res = 0;
        for (const auto& word : words)
        {

            vector<pair<char, int>> cur = conv(word);
            if (cur.size() != ori.size())
                continue;
            int ok = 1;
            for (int i = 0; i < cur.size(); ++i)
            {
                if (cur[i].first != ori[i].first)
                {
                    ok = 0;
                    break;
                }
                if (cur[i].second == ori[i].second || ori[i].second >= 3 && cur[i].second < ori[i].second)
                    continue;
                ok = 0;
            }
            res += ok;
        }
        return res;
    }
};
