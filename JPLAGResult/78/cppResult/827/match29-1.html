class Solution {
    vector<pair<char, int>> split(const string &s) {
        vector<pair<char, int>> vp;
        char c = 0;
        for(int i = 0; i < s.size(); i++) {
            if(s[i] == c) vp.back().second++;
            else {
                c = s[i];
                vp.push_back({c, 1});
            }
        }
        return vp;
    }

public:
    int expressiveWords(string S, vector<string>& words) {
        auto split_s = split(S);
        int count = 0;
        for(auto &w : words) {
            auto split_word = split(w);
            if(split_s.size() == split_word.size()) {
                int flag = 1;
                for(int i = 0; flag && i < split_s.size(); i++) {
                    auto p1 = split_s[i], p2 = split_word[i];
                    if(p1.first != p2.first) {
                        flag = 0;
                    }
                    else if(p1.second < p2.second) {
                        flag = 0;
                    }
                    else if(p1.second == 2 && p2.second == 1) {
                        flag = 0;
                    }
                }
                count += flag;
            }
        }
        return count;
    }
};
