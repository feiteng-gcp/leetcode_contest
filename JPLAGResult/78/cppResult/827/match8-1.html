class Solution {
public:
    
    vector< pair<char,int> > solve(string &s){
        int n = s.size();
        vector< pair<char,int> > ans;
        if(n == 0){
            return ans;
        }
        
        int c = 1;
        for(int i = 1; i <= n; ++i){
            if(i == n || s[i] != s[i-1] ){
                ans.push_back({ s[i-1] , c});
                c = 1;
            }else{
                c++;
            }
        }

        return ans;
    }
    
    int expressiveWords(string S, vector<string>& words) {
        vector< pair<char,int> > main = solve(S);
        int ans = 0;
        for(string elem : words){
            vector< pair<char,int> > current = solve(elem);
            if(main.size() != current.size()){
                continue;
            }
            bool ok = true;
            for(int k = 0; k < current.size(); ++k){
                if(current[k].first != main[k].first){
                    ok = false;
                }
                if(current[k].second == main[k].second){
                    continue;
                }
                if(main[k].second < current[k].second){
                    ok = false;
                }
                if(main[k].second < 3){
                    ok = false;
                }
                
            }
            ans += ok;
        }
        return ans;
    }
};
