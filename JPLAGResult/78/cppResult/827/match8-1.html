class Solution {
public:
    typedef pair<char, int> cnt_t;
    int expressiveWords(string S, vector<string>& words) {
        vector<cnt_t> target = count(S);
        int ret = 0;

        for (auto w : words) {
            vector<cnt_t> cand = count(w);
            if (extendable(target, cand)) ret++;
        }
        return ret;
    }
    bool extendable(const vector<cnt_t>& target, const vector<cnt_t>& cand) {
        if (target.size() != cand.size()) return false;
        for (int i = 0; i < target.size(); ++i) {
            if (cand[i].first != target[i].first) return false;
            if (cand[i].second == target[i].second || (cand[i].second < target[i].second && target[i].second >= 3)) continue;
            return false;
        }
        return true;
    }
    vector<cnt_t> count(const string& s) {
        vector<cnt_t> ret;
        for (int i = 0; i < s.size();) {
            int j = i;
            while (j < s.size() && s[i] == s[j]) j++;
            ret.push_back(cnt_t(s[i], j-i));
            i = j;
        }
        return ret;
    }
};
