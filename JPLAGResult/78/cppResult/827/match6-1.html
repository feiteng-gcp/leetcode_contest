class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int ret = 0;
        for (auto& word : words){
            //cout << "#####" << endl;
            int i = 0, j = 0;
            bool h = true;
            while (true){
                if (i == S.size() && j == word.size()) break;
                if (i >= S.size() || j >= word.size()){
                    h = false;
                    break;
                }
                if (S[i] != word[j]){
                    h = false;
                    break;
                }
                int ci = 1, cj = 1;
                while (i + 1 < S.size() && S[i] == S[i + 1]){
                    i ++;
                    ci ++;
                }

                while (j + 1 < word.size() && word[j] == word[j + 1]){
                    j ++;
                    cj ++;
                }
                //cout << i << ' ' << j << ' ' << ci << ' ' << cj << endl;
                if (ci != cj){
                    if (!(ci >= 3 && ci >= cj)){
                        h = false;
                        break;
                    }
                }
                i ++;
                j ++;
            }
            if (h) ret ++;
        }
        return ret;
    }
};
