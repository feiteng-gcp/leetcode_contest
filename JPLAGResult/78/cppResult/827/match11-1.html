class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        if(S.length()==0||!words.size()) return 0;
        vector<pair<char,int>> dp;
        int i=0;
        while(i<S.length()){
            char ch = S[i];
            i++;
            int num = 1;
            while(i<S.length()&&S[i]==S[i-1]){
                i++;
                num++;
            }
            dp.push_back({ch,num});
        }
        int res = 0;
        for(auto st:words){
            if(!st.length()) continue;
            if(st==S) continue;
            vector<pair<char, int>> cur;
            i = 0;
            while(i<st.length()){
                char ch = st[i];
                i++;
                int num = 1;
                while(i<st.length()&&st[i]==st[i-1]){
                    i++;
                    num++;
                }
                cur.push_back({ch,num});
            }

            if(cur.size()!=dp.size()) continue;
            bool f = true;
            for(int j=0;j<dp.size();j++){
                //cout<<dp[j].second<<" "<<cur[j].second<<endl;
                if(dp[j].first!=cur[j].first||dp[j].second<cur[j].second){
                    f = false;
                    break;
                }
                if((dp[j].second>cur[j].second&&dp[j].second>=3)||dp[j].second==cur[j].second)continue;
                else {
                    f = false;
                    break;
                }
                
            }
            if(f) res++;
        }
        return res;
    }
};
