class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        vector<pair<char, int>> s;
        for (auto c: S)
        {
            if (s.empty() || s.back().first != c)
            {
                s.push_back({c,1});
            }
            else
            {
                s.back().second++;
            }
        }
        int res = 0;
        for (auto word: words)
        {
            vector<pair<char, int>> w;
            for (auto c: word)
            {
                if (w.empty() || w.back().first != c)
                {
                    w.push_back({c, 1});
                }
                else
                {
                    w.back().second++;
                }
            }
            if (w.size() != s.size())
            {
                continue;
            }

            bool shouldAdd = true;
            for (int i = 0; i < s.size(); i++)
            {
                if (s[i].first != w[i].first || s[i].second < w[i].second || (s[i].second < 3 && s[i].second != w[i].second))
                {
                    shouldAdd = false; 
                    break;
                }
            }
            if (shouldAdd)
            {
                res++;
            }
        }
        return res;
    }
};
