class Solution {
public:
    bool canBeStretched(string & S, string & K, int si, int ki, int sl, int kl){
        
        if((ki == kl && si != sl) || (si == sl && ki != kl)){
            return false;
        }
        if(si == sl && ki == kl )
            return true;
        if(S[si] != K[ki])
            return false;
        char sc = S[si];
        int scount = 0;
        while(si < sl && S[si] == sc){
            si++;
            scount++;
        }
        char kc = K[ki];
        int kcount = 0;
        while(ki < kl && K[ki] == kc){

            ki++;
            kcount++;
        }
        if(scount == kcount || scount >=3) {
            //cout<<"si : ki :: "<<si<<","<<ki<<" S[si] : K[ki] :: "<<S[si]<<","<<K[ki]<<endl;
            return canBeStretched(S, K, si, ki, sl, kl);
        }
        
        return false;
    }
    int expressiveWords(string S, vector<string>& words) {
        int sl = S.length();
        int count = 0;
        for(auto k : words){
            cout<<"Doing for word : "<<k<<endl;
            if(canBeStretched(S, k, 0, 0, sl, k.length()))
                count++;
        }
        return count;
        
    }
};
