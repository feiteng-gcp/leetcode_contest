class Solution {
public:
int expressiveWords(string S, vector<string>& words) {
    int len = words.size();
    int slen = S.length();

    int cnt = 0;
    for (int i=0;i<len;i++)
    {
        string word = words[i];
        int l = word.length();
        int j=0,k=0;
        int flag = 0;
        while (j<l && k<slen)
        {
            if (S[k] != word[j])
            {
                flag = 1;
                break;
            }
            int cnt1 = 0;
            char tmp = S[k];
            while (k<slen && S[k] == tmp)
            {
                k++;
                cnt1++;
            }
            int cnt2 = 0;
            while (j<l && word[j] == tmp)
            {
                j++;
                cnt2++;
            }
            if (cnt1-cnt2<0 || (cnt1-cnt2>0 && cnt1<3))
            {
                flag = 1;
                break;
            }
        }
        if (flag || j<l || k<slen)
            cnt--;
        cnt++;
    }
    return cnt;
}
};
