class Solution {
public:
    vector<pair<char, int> > convert(string s){
        char c = s[0];
        int count = 1;
        vector<pair<char, int> > result;

        for(int i = 1; i < s.size(); i++){
            if(s[i] == c) ++count;
            else{
                result.emplace_back(c, count);
                c = s[i];
                count = 1;
            }
        }
        result.emplace_back(c, count);
        return result;
    }
    
    bool match(const vector<pair<char, int> >& v1, const vector<pair<char, int> >& v2){
        int n1 = v1.size();
        int n2 = v2.size();
        if(n1 != n2) return false;
        for(int i = 0; i < n1; i++){
            if(v1[i].first != v2[i].first) return false;
            if(v1[i].second != v2[i].second && v2[i].second < 3) return false;
        }
        return true;
    }
    
    int expressiveWords(string S, vector<string>& words) {
        vector<pair<char, int> > sv = convert(S);
        int result = 0;
        for(auto x : words){
            vector<pair<char, int> > word = convert(x);
            if(match(word, sv)) ++result;
        }
        return result;
    }
};
