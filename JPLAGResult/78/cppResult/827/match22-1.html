class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        const int n = S.size();
        
        if (n == 0) {

            int res = 0;
            for (auto &w : words) {
                if (w.size() == 0) {
                    res++;
                }
            }
            return res;
        }
        
        
        
        
        vector<pair<char, int>> s;
        int i = 0;
        int p = 0;
        while (i < n) {
            while (p < n-1 && S[p+1] == S[p]) {
                ++p;
            }
            s.emplace_back(S[i], p-i+1);
            ++p;
            i = p;
        }
        const int k = s.size();
        
        int res = 0;
        
        for (auto &w : words) {
            int m = w.size();
            int t = 0;
            int i = 0;
            int p = 0;
            bool flag = 0;
            while (i < m) {
                if (t >= k || w[i] != s[t].first) {
                    flag = 1;
                    break;
                }
                while (p < m-1 && w[p+1] == w[p]) {
                    ++p;
                }
                if (s[t].second <= 2) {
                    if (p-i+1 == s[t].second) {
                        ++p;
                        i = p;
                        ++t;
                    } else {
                        flag = 1;
                        break;
                    }
                } else {
                    if (s[t].second >= p-i+1) {
                        ++p;
                        i = p;
                        ++t;
                    } else {
                        flag = 1;
                        break;
                    }
                }
                if (flag) break;
                
            }
            if (flag) continue;
            if (t == k) {
                res++;
            }
        }
        
        return res;
    }
};
