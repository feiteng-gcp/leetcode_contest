class Solution {
public:
	vector<pair<char, int>>count(string s){
		vector<pair<char, int>>x;
		char p = s[0];
		int ct = 1;
		for (int i = 1; i<s.length(); i++){
			if (s[i] != p){
				x.push_back({ p, ct });
				ct = 1;
				p = s[i];
				continue;
			}
			p = s[i];
			ct++;
		}
		x.push_back({ p, ct });
		return x;
	}
	int expressiveWords(string S, vector<string> words) {
		vector<pair<char, int>>a = count(S);
		int ct = 0;
		for (int i = 0; i<words.size(); i++){
			auto x = count(words[i]);
			int j;
			if (a.size() != x.size())continue;
			for (j = 0; j<a.size(); j++){
				if (x[j].second > a[j].second)break;
				if (a[j].second < 3 && x[j].second != a[j].second)break;
				if (a[j].first != x[j].first)break;
			}
			if (j == a.size())ct++;
		}
		return ct;
	}
};
