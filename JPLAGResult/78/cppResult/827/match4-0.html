class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        
        int count = 0;
        
        for (int i = 0; i < words.size(); i++) {
            if (isStretchy(S, words[i])) {
                count++;
            }
        }
        
        return count;
    }
    
    bool isStretchy (string P, string Q) {
        
        int j = 0;
        int i = 0;
        
        while ( i < P.length()) {
            
            int count = 1;
            char c = P[i];
            while (P[++i] == c && i < P.length()) {
                count++;
            }
            
            if (Q[j] != c) {
                return false;
            }
        
            int qCount = 1;
            while (Q[++j] == c && j < Q.length()) {
                qCount++;
            }
        
            if ((qCount != count) && count < 3 || (qCount > count)) {
                return false;
            }
            
            cout << c << endl << count << endl << qCount << endl;
            
        }
        
        if (i != P.length()|| j != Q.length()) {
            return false;
        }
        
        return true;
    }
};
