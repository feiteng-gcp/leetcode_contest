class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int res = 0;
        
        for (auto w : words) {
            if (isStretchy(w, S)) ++res;
            //cout << w << ":" << res;
        }
        return res;
    }
    
    bool isStretchy(const string& s, const string&t) {
        if (s.empty() && t.empty()) return true;
        if (s.empty()) return false;
        if (t.empty()) return false;
        
        if (s[0] != t[0]) return false;
        int i=1;
        while(i<s.size()&& s[i]==s[i-1]) {
            ++i;
        }
        
        int j=1;
        while(j<t.size() && t[j]==t[j-1]) ++j;
        if (j < i) return false;
        
        if (i == j || (j >=3)) {
            auto newS = s.substr(i);
            auto newT = t.substr(j);
            return isStretchy(newS, newT);
        }
        
        return false;
    }
    
};
