class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int res = 0;
        string s = S;
        for (auto& w : words) {
            int n = s.length(), m = w.length();
            bool bExtended = true;
            int i = 0, j = 0;
            for (i = 0, j = 0; i < n && j < m; ++i, ++j) {
                if (s[i] == w[j]) {
                    int groupSize = 1;
                    while (i + 1 < n && s[i + 1] == s[i]) {
                        ++groupSize;
                        ++i;
                    }
                    int actualSize = 1;
                    while (j + 1 < m && w[j + 1] == w[j]) {
                        ++actualSize;
                        ++j;
                    }
                    // cout << w << " " << s[i] << " " << groupSize << " " << actualSize << endl; 
                    if (!(actualSize == groupSize || actualSize < groupSize && groupSize >= 3)) {
                        bExtended = false;
                        break;
                    }
                } else {
                    bExtended = false;
                    break;
                }
            } 
            if (bExtended && i == n && j == m) {
                ++res;
            }
        }
        
        return res;
    }
};
