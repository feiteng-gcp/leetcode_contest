class Solution {
public:
    int expressiveWords(string s, vector<string>& words) {
        auto s_data = get_data(s);
        int result = 0;
        for (auto word : words) {
            auto w_data = get_data(word);
            if (is_valid(s_data, w_data)) {
                result++;
                //cout << word << endl;
            }
        }
        return result;
    }
    
    vector<pair<char, int>> get_data(string &s) {
        vector<pair<char, int>> result;
        int n = s.length();
        if (n == 0) return result;
        pair<char, int> cur_result = {s[0], 1};
        for (int i = 1; i < n; i++) {
            if (s[i] == cur_result.first) {
                cur_result.second++;
            } else {
                result.push_back(cur_result);
                cur_result = {s[i], 1};
            }
        }
        result.push_back(cur_result);
        return result;
    }
    
    bool is_valid(vector<pair<char, int>> &s, vector<pair<char, int>> &query) {
        int slen = s.size(), qlen = query.size();
        if (slen != qlen) return false;
        for (int i = 0; i < slen; i++) {
            if (s[i].first != query[i].first) return false;
            if (s[i].second < query[i].second || (s[i].second == 2 && s[i].second == query[i].second + 1)) return false;
        }
        return true;
    }
};
