class Solution {
public:
    struct fff
{
	char s;
	int cnt;
};
    int expressiveWords(string S, vector<string>& words) {
        int l=S.size(),n=words.size();
	vector<fff> v;
	int num=1;
	for(int i=1;i<l;i++)
	{
		if(S[i]==S[i-1])num++;
		else
		{
			fff news;
			news.s=S[i-1];
			news.cnt=num;
			v.push_back(news);
			num=1;
		}
	}
	fff news;
	news.s=S[l-1];
	news.cnt=num;
	v.push_back(news);
	int tot=v.size();
	int ans=0;
	for(int c=0;c<n;c++)
	{
		int len=words[c].size();
		int num=1;
		vector<fff> now;
		for(int i=1;i<len;i++)
		{
			if(words[c][i]==words[c][i-1])num++;
			else
			{
				fff news;
				news.s=words[c][i-1];
				news.cnt=num;
				now.push_back(news);
				num=1;
			}
		}
		fff news;
		news.s=words[c][len-1];
		news.cnt=num;
		now.push_back(news);
		if(now.size()!=tot)continue;

		int ok=1;
		for(int i=0;i<tot;i++)
		{
			if(v[i].s!=now[i].s)
			{
				ok=0;
				break;
			}
			if(v[i].cnt<now[i].cnt||(v[i].cnt==2&&now[i].cnt==1))
			{
				ok=0;
				break;
			}
		}
		ans=ans+ok;
	}
	return ans;
    }
};
