class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        auto t = trans(S);
      int cnt = 0;
      for (auto v: words) {
        auto p = trans(v);
        if (p.size() == t.size()) {
          for (int i = 0; i <= p.size(); i++) {
            if (i == p.size()) cnt++;
            if (p[i] == t[i]) continue;
            if (isdigit(p[i]) && isdigit(t[i]) && t[i] > p[i] && t[i] - '0' >= 3) continue;
            break;
          }
        }
      }
      return cnt;
    }
  string trans(string S) {
    string t;
        for (int i = 0; i < S.size(); i++) {
          int j = i;
          while (j < S.size() && S[i] == S[j]) j++;
          int len = j - i;
          t += S[i];
            t += to_string(len);
          /*
          if (len >= 3) {
            
          } else {
            t += string(len, S[i]);
          }*/
          i = j - 1;
        }
    return t;
  }
};
