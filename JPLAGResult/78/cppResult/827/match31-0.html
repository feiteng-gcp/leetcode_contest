class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int res = 0;
        for(auto word: words){
            int i = 0, j = 0;
            bool check = true;
            while(i<S.size() && j<word.size()){
                if(S[i]==word[j]){
                    char c = word[j];
                    int cntS = 0, cntW = 0;
                    while(i<S.size() && S[i]==c){
                        cntS++; 
                        i++;
                    }
                    
                    while(j<word.size() && word[j]==c){
                        cntW++;
                        j++;
                    }
                    if((cntS<3 && cntS!=cntW) || cntS<cntW){
                        check = false;
                        break;
                    }
                }
                else break;
            }
            if(i==S.size() && j==word.size() && check) res++;
        }
        return res;
    }
};
