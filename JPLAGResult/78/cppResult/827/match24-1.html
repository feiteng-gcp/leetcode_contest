class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int count = 0;
        for (auto word : words) {
            if (judge(S, word)) {
                count++;
            }
        }
        return count;
    }

    bool judge(string& S, string& word) {
        int len_S = S.size(), len_word = word.size();
        int i = 0, j = 0;
        while (i < len_S && j < len_word) {

            int pre_i = i, pre_j = j;
            while (i + 1 < len_S && S[i] == S[i + 1]) {
                i++;
            }
            while (j + 1 < len_word && word[j] == word[j + 1]) {
                j++;
            }
            if (S[i] != word[j] || (i - pre_i + 1) < (j - pre_j + 1) || (i - pre_i + 1) > (j - pre_j + 1) && (i - pre_i + 1) < 3) {
                return false;
            }
            i++;
            j++;
        }
        return i >= len_S && j >= len_word;
    }
};
