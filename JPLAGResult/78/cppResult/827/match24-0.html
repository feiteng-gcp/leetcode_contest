class Solution {
public:
    bool check(string a, string b) {
        int la = a.length(), lb = b.length();
        int i = 0, j = 0;
        while (i < la && j < lb) {
            if (a[i] != b[j]) {
                return false;
            }
            int ni = i + 1, nj = j + 1;
            while (ni < la && a[ni] == a[i]) {
                ni++;
            }
            while (nj < lb && b[nj] == b[j]) {
                nj++;
            }
            // cout << i << ", " << j << ", " << ni << ", " << nj << endl;
            if ((ni - i >= 3 && ni >= nj) || (ni - i == nj - j)) {
                i = ni;
                j = nj;
            } else {
                return false;
            }
        }
        return i == la && j == lb;
    }
    int expressiveWords(string S, vector<string>& words) {
        int res = 0;
        for (string s : words) {
            if (check(S, s)) {
                res++;
            }
        }
        return res;
    }
};
