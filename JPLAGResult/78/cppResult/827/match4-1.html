class Solution {
public:
    
    int expressiveWords(string S, vector<string>& words) {
        auto target = make(S);
        

        int cnt = 0;
        for (int i = 0; i < words.size(); ++i) {
            if (match(target, make(words[i]))) {
                cnt++;
            }
        }
        
        return cnt;
    }
    
    vector<pair<char, int>> make(string &S) {
        vector<pair<char, int>> table;
        int j= 0;
        for (int i = 0; i < S.size(); ++i) {
            if (S[i] != S[j]) {
                table.emplace_back(S[j], i - j);
                j = i;
            }
        }
        
        table.emplace_back(S[j], S.size() - j);
        
        return table;
    }
    
    bool match(vector<pair<char, int>> &l, vector<pair<char, int>> &&r)
    {
        if (l.size() != r.size()) return false;
        for (int i = 0; i < l.size(); ++i) {
            
            if (l[i].first != r[i].first) return false;
      //      printf("%c %d %d \n", l[i].first, l[i].second, r[i].second);
            if (l[i].second < r[i].second) return false;
            if (l[i].second == 2 && r[i].second == 1) return false;
        }
        
        return true;
    }
};
