class Solution {
public:
    vector<char> letters;
    vector<int> counts;
    

    int expressiveWords(string S, vector<string>& words) {
        tokenize(S, letters, counts);
        int ans = 0;
        for (auto &w : words) {
            if (isExpressive(w)) ans ++;
        }
        return ans;
    }
    
    bool isExpressive(string &s) {
        vector<char> ll;
        vector<int> cc;
        
        tokenize(s, ll, cc);
        if (ll.size() != letters.size()) return false;
        for (int i = 0; i < letters.size(); i ++) {
            if (ll[i] != letters[i]) return false; // must be of same char
            if (cc[i] > counts[i]) return false;   // must not exceed target counts
             // non-extendable counts has to match exactly
            if (counts[i] < 3 and cc[i] != counts[i]) return false;
        }
        return true;
    }
    
    void tokenize(string &s, vector<char>& ll, vector<int>& cc) {
        for (int i = 0; i < s.size() ; i ++) {
            if (ll.empty() or s[i] != ll.back()) {
                ll.push_back(s[i]);
                cc.push_back(1);
            } else {
                cc.back() ++;
            }
        }
        // for (int i = 0; i < ll.size(); i ++) cout << ll[i] << cc[i]; cout << endl;        
    }
};
