class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        int res = 0;
        for (auto s: words) {   
            bool stre = true;
            int cnt = 1;
            int ps = 0;
            char cur = s[0];
            for (int i = 1; i < s.size(); i++) {    
                if (s[i] != s[i - 1]) {
                    if (S[ps] != cur) {
                        stre = false;
                        break;
                    }
                    ps++;
                    int scnt = 1;
                    while (ps < S.size() && S[ps] == S[ps - 1]) {
                        ps++; scnt++;
                    }
                    if (scnt != cnt) {
                        if (scnt < cnt || scnt < 3) {
                            stre = false;
                            break;
                        }
                    }
                    cur = s[i];
                    cnt = 1;
                } else {
                    cnt++;
                }
            }
            if (S[ps] != cur) {
                stre = false;
            }
            ps++;
            int scnt = 1;
            while (ps < S.size() && S[ps] == S[ps - 1]) {
                ps++; scnt++;
            }
            if (scnt != cnt) {
                if (scnt < cnt || scnt < 3) {
                    stre = false;
                }
            }
            if (ps < S.size()) stre = false;
            if (stre == true) res++;
        }
        return res;
    }
};
