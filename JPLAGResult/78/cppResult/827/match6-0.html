class Solution {
public:
    int expressiveWords(string s, vector<string>& words)
    {
        int len=s.size();
        if(len==0) return 0;
        vector<int> cnt(len,0);
        string pat;
        int idx=0;
        cnt[idx]++;
        pat+=s[0];
        for(int i=1;i<len;i++)
        {
            if(s[i]==s[i-1]) cnt[idx]++;
            else
            {
                idx++;
                cnt[idx]++;
                pat+=s[i];
            }
        }


        int ret=0;
        for(auto word : words)
        {
            int l=word.size();
            vector<int> c2(l,0);
            string p1;
            int ii=0;
            c2[ii]++;
            p1+=word[0];
            for(int i=1;i<l;i++)
            {
                if(word[i]==word[i-1]) c2[ii]++;
                else
                {
                    ii++;
                    c2[ii]++;
                    p1+=word[i];
                }
            }

            if(pat==p1)
            {
                bool flag=true;
                for(int i=0;i<pat.size();i++)
                {
                    if(cnt[i]==c2[i]) continue;
                    if(cnt[i]<c2[i]||cnt[i]<3)
                    {
                        flag=false;
                        break;
                    }
                }
                if(flag) ret++;
            }
        }

        return ret;
    }

    
    
};
