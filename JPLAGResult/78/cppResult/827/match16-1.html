class Solution {
public:
    string breakStretch(string S, vector<int>& repeat){//returns diluted string
        

        int i =0, n =S.size();
        string dilute;
        while(i<n){
            char prevChar = S[i++];
            int count = 1;
            while(i<n && S[i] == prevChar){
                ++count;
                ++i;
            }
            
            stringstream ss;
            ss<<prevChar;
            string temp;
            ss>>temp;
            dilute += temp;
            
            repeat.push_back(count);
            
        }
        
        return dilute;
    }
    int expressiveWords(string S, vector<string>& words) {
        vector<int> repeat;
        string dilute = breakStretch(S,repeat);
        
        int res = 0;
        for(int i =0; i<words.size(); ++i){
            string word = words[i];
            vector<int> pos;
            string dilutedW = breakStretch(word,pos);
            
            if(dilute != dilutedW)
                continue;
            bool ok = true;
            for(int j =0; j<repeat.size(); ++j){
                if(repeat[j] < 3){
                    if(repeat[j] != pos[j]){
                        ok = false;
                        break;
                    }
                }
            }
            
            if(!ok)
                continue;
            
            ++res;
        }
        
        return res;
    }
};
