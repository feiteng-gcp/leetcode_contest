class Solution {
public:
    int solve (vector<int> & cnt, string & cond, string & word) {
        string tmpd;
        vector<int> tmpt;

        for (int i = 0; i < word.size(); i++) {
            if (i == 0 || word[i-1] != word[i]) {
                tmpd.push_back(word[i]);
                tmpt.push_back(1);
            }
            else {
                tmpt.back()++;
            }
        }
        if (tmpd != cond) return 0;
        for (int i = 0; i < cond.size(); i++) {
            if (tmpt[i] == cnt[i]) continue;
            if (tmpt[i] > cnt[i]) return 0;
            if (tmpt[i] < cnt[i] && cnt[i] >= 3) continue;
            return 0;
        }
        return 1;
        
    }
    
    int expressiveWords(string s, vector<string>& words) {
        string cond;
        vector<int> cnt;
        for (int i = 0; i < s.size(); i++) {
            if (i == 0 || s[i-1] != s[i]) {
                cond.push_back(s[i]);
                cnt.push_back(1);
            }
            else {
                cnt.back()++;
            }
        }
        int ret = 0;
        for (string & word : words) {
            ret += solve(cnt, cond, word);
        }
        return ret;
        
    }
};
