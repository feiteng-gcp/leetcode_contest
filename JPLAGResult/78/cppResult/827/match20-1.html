const int N = 1e6 + 10;
using ll = long long;
const int INF = 0x3f3f3f3f;
using pii = pair<int, int>;
using vi = vector<int>;
#define rep(i, a, b) for (int i = (a); i < (b); i++)
#define mp make_pair 
#define pb push_back
#define clr(x, v) memset(x, v, sizeof (x))
#define all(x) x.begin(), x.end()
#define fi first
#define se second

class Solution {
public:
    vector<pair<char, int>> get(string S) {
        vector<pair<char, int>> vec;
        int n = S.size();
        int cnt = 1;
        for (int i = 1; i <= n; i++) {
            if (i == n || S[i] != S[i - 1]) {
                vec.push_back(mp(S[i - 1], cnt));
                cnt = 1;
            } else {
                cnt++;
            }
        }
        return vec;
    }
    int expressiveWords(string S, vector<string>& words) {
        if (S == "") {
            int res = 0;
            for (auto word : words) {
                if (word == "") res++;
            }
            return res;
        }
        int res = 0;
        auto v1 = get(S);
        for (auto word : words) {
            if (word == "") continue;
            auto v2 = get(word);
            if (v1.size() != v2.size()) continue;
            bool ok = true;
            // cout << v1.size() << endl;

            rep(i, 0, (int)v1.size()) {
                if (v1[i].fi != v2[i].fi) {
                    ok = false;
                    break;
                }
                if (v1[i].se < v2[i].se) {
                    ok = false;
                    break;
                }
                // cout << v1[i].se << " " << v2[i].se << endl;
                if (v1[i].se > v2[i].se && v1[i].se < 3) {
                    ok = false;
                    break;
                }
            }
            if (ok) res++;
        }
        return res;
    }
};
