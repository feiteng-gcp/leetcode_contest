class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
        
        if (S.size() == 0) return 0;
        if (words.size() == 0) return 0;
        
        vector< pair<char, int> > target;
        GetVec(S, target);
        
        int count = 0;
        for (int i = 0; i < words.size(); ++i)
        {                        
            string curWord = words[i];
            vector< pair<char, int> > cur;
            GetVec(curWord, cur);
            
            if (cur.size() != target.size()) continue;
            
            bool flag = true;
            for (int t = 0; t < cur.size(); ++t)
            {
                if (target[t].first != cur[t].first) 
                {
                    flag = false;
                    break;
                }
                
                if (target[t].second <= 2 && target[t].second != cur[t].second)
                {
                    flag = false;
                    break;
                }
                
                if (target[t].second >= 3 && target[t].second < cur[t].second)
                {
                    flag = false;
                    break;
                }
            }
            
            if (flag == true) ++count;

        }
        
        return count;
    }
    
    void GetVec(string& s, vector< pair<char, int> >& vec)
    {
        for (int i = 0; i < s.size(); ++i)
        {
            if (vec.size() == 0) vec.push_back( {s[i], 1} );
            else
            {
                if (vec.back().first == s[i]) vec.back().second++;
                                         else vec.push_back( {s[i], 1} );
            }
        }
    }
};
