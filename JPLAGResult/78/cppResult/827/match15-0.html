class Solution {
public:
    bool valid(int i, int j) {
        if (j >= 3) {
            return true;
        }
        return i == j;
    }

    bool can_stretch(const string &t, const string &s) {
        int i = 0;
        int j = 0;
        while (i + 1 < t.size() && t[i] == t[i+1]) {
            i++;
        }
        i++;
        while (j + 1 < s.size() && s[j] == s[j+1]) {
            j++;
        }
        j++;
        if (t[0] != s[0]) {
            return false;
        }
        if (!valid(i, j)) {
            return false;
        }
        while (i < t.size() && j < s.size()) {
            int cnt_t = 1;
            int cnt_s = 1;
            while (i + 1 < t.size() && t[i] == t[i+1]) {
                i++;
                cnt_t++;
            }
            while (j + 1 < s.size() && s[j] == s[j+1]) {
                j++;
                cnt_s++;
            }
            if (!valid(cnt_t, cnt_s)) {
                return false;
            }
            if (t[i] != s[j]) {
                return false;
            }
            i++;
            j++;
        }
        if (i < t.size() || j < s.size()) {
            return false;
        }
        return true;
    }

    int expressiveWords(string s, vector<string>& words) {
        int sum = 0;
        for (const string &word : words) {
            if (can_stretch(word, s)) {
                sum++;
            }
        }
        return sum;
    }

};
