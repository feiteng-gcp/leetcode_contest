class Solution {
public:
    void token(string S, vector<string>& tok) {
        tok.clear();
        string sub;
        for (int i = 0; i < S.size(); ++i) {
            if (i == 0 || sub.empty() || S[i] == sub.back()) {
                sub.push_back(S[i]);
            } else {
                tok.push_back(sub);
                sub = S[i];
            }

        }
        if (sub != "") {
            tok.push_back(sub);
        }        
        //for (auto &s: tok) {
        //    cout<<s<<" ";
       // }
        //cout<<endl;
    }
    bool cmp(string &s1, string &base) {
        if (s1[0] != base[0]) return false;
        if (base.size() > s1.size()) return false;
        if (s1.size() == base.size()) return true;
        if (s1.size() >= 3) return true;
        if ( (s1.size() - base.size()) >= 2 ) return true;
        return false;
    }
    int expressiveWords(string S, vector<string>& words) {
        vector<string> tok1;
        vector<string> tok2;
        int cnt = 0;
        token(S, tok1);
        for (auto &s : words) {
            token(s, tok2);
            if (tok1.size() != tok2.size()) continue;
            bool pass = true;
            for (int i = 0; i < tok1.size(); ++i) {
                //cout<<"cmp:"<<tok1[i]<<" - "<<tok2[i]<<" : "<<cmp(tok1[i], tok2[i])<<endl;
                if (!cmp(tok1[i], tok2[i])) {
                    pass = false;
                    break;
                }
            }
            if (pass) {
                ++cnt;
            }
        }
        return cnt;
        
    }
};
