class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {
      if (S.size() <= 2) return 0;
      vector<pair<char,int>> ssign  = signature(S);
      int count = 0;
      for (auto s : words) {
          if (isStretchy(ssign, s)) {
             //out << s << endl;
              count++;
          }
      }
        return count;
    }
    bool isStretchy(const vector<pair<char,int>>& ssign, const string test) {
         vector<pair<char,int>> st = signature(test);
         if (st.size() != ssign.size()) return false;       
         for(int i = 0; i < ssign.size(); i++) {
             if(ssign[i].first != st[i].first) {                 
                 return false;
             }
             if(ssign[i].second != st[i].second) {
                 if (ssign[i].second >= 3 && st[i].second < ssign[i].second) {
                     continue;
                 } else {
                     return false;
                 }
             }
         }
        return true;
    }
    vector<pair<char,int>> signature(string s) {
        vector<pair<char,int>> res;
        int slen = s.size();
        if (slen == 1) {
            res.push_back(pair<char,int>{s[0],1});
        } else {
            char prev = s[0];
            int cnt  = 1;
            for (int i = 1; i < slen; i++) {
                if (s[i] != prev) {
                    res.push_back(pair<char,int>{prev, cnt});
                    prev = s[i];
                    cnt = 1;
                } else {
                    cnt++;
                }
            }
            res.push_back(pair<char,int>{prev, cnt});
        }
        
        //debug
        /*
        cout << "string:: " << s << endl;
        for(auto i : res) {
            cout << i.first << "#" << i.second <<endl;
        }
        */
        return res;
    }
};
