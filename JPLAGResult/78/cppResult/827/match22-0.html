class Solution {
public:
    int expressiveWords(string S, vector<string>& words) {        
        vector<int> N;
        vector<char> C;
        char c = 0;
        int n = 1;
        int G = 0;
        for (int i = 0; i < S.length(); ++i) {
            if (i == 0) {
                c = S[i];
            } else if (c != S[i] && i) {
                G++;
                N.push_back(n);
                C.push_back(c);
                // cout << G << " " << c << " " << n << endl;
                c = S[i];
                n = 1;
            } else {
                n++;
            }
        }
        if (c) {
            G++;
            // cout << G << " " << c << " " << n << endl;
            N.push_back(n);
            C.push_back(c);
        }
        
        int sol = 0;
        for (auto s: words) {
            c = 0;
            n = 1;
            int g = 0;
            bool valid = true;
            
            for (int i = 0; i < s.length() && valid; ++i) {
                if (i == 0) {
                    c = s[i];
                } else if (c != s[i] && i) {
                    // check
                    if (C[g] == c && (N[g] == n || N[g] >= 3)) {
                    } else {
                        // cout << s << " " << g << " " << c << " " << n << endl;
                        valid = false;
                        break;
                    }
                    
                    c = s[i];
                    n = 1;
                    g++;
                } else {
                    n++;
                }
            }
            if (!valid) {
                continue;
            }

            if (g < G) {
                if (C[g] == c && (N[g] == n || N[g] >= 3)) {
                    ++g;
                } else {
                    // cout << s << " " << g << " " << c << " " << n << endl;
                    valid = false;
                }
            }
            
            if (g == G && valid) {
                sol += 1;
            }
        }
        
        return sol;
    }
};
