class Solution {
public:
    vector<string> subdomainVisits(vector<string>& cpdomains) {
        unordered_map<string,int> cnts;
        
        for (auto& s : cpdomains) {
            parse(s, cnts);
        }
        
        vector<string> res;
        for (auto& p :  cnts) {
            string tmp = to_string(p.second) + " " + p.first;
            res.push_back(tmp);
        }
        
        return res;
    }
    
private:
    void parse(const string& s, unordered_map<string,int>& cnts) {
        int cnt = 0;
        int n = s.length();
        string domain;
        for (int i = 0; i < n; ++i) {
            if (isdigit(s[i])) {
                cnt = s[i] - '0';
                while (i + 1 < n && isdigit(s[i + 1])) {
                    ++i;
                    cnt = cnt * 10 + s[i] - '0';
                }
            } else if (isspace(s[i])) {
                int j = n - 1;
                while (j > i && isspace(s[j])) {
                    --j;
                } 
                domain = s.substr(i + 1, j - i);
                break;
            }
        }
        

        n = domain.length();
        
        cnts[domain] += cnt;
        for (int i = 0; i < n; ++i) {
            if (domain[i] == '.') {
                cnts[domain.substr(i + 1)] += cnt;
            }
        }
    }
};
