class Solution {
public:
    vector<string> subdomainVisits(vector<string>& cpdomains) {
        vector<string> res;
        unordered_map<string,int> ht;
        for(auto domain: cpdomains){
            int ind = domain.find(" ");
            int cnt = stoi(domain.substr(0,ind));
            string url = domain.substr(ind+1);
            ht[url] += cnt;
            for(int i=0; i<url.size(); i++){
                if(url[i]!='.') continue;
                ht[url.substr(i+1)] += cnt;
            }
        }
        
        for(auto it: ht){
            string t = to_string(it.second) + " " + it.first;
            res.push_back(t);
        }
        return res;
    }
};
