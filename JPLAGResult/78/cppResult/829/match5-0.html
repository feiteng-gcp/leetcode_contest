class Solution {
public:
    vector<string> subdomainVisits(vector<string>& cpdomains) {
        map<string, int> cnt;
        for (string s: cpdomains) {
            int cp;
            char domain[1111] = {0};
            sscanf(s.c_str(), "%d %s", &cp, domain);
            int n = strlen(domain);
            cnt[string(domain)] += cp;
            for (int i = 0; i < n; i++) {
                if (domain[i] == '.') {
                    cnt[string(domain + i + 1)] += cp;
                }
            }
        }
        vector<string> result;
        char str[1111];
        for (auto p: cnt) {
            sprintf(str, "%d %s", p.second, p.first.c_str());
            result.push_back(str);
        }
        return result;
    }
};
