class Solution {
public:
    vector<string> subdomainVisits(vector<string>& cpdomains) {
        map<string, int> mp;
        for (auto str : cpdomains) {
            int space = str.find(" ");
            int cnt = stoi(str.substr(0, space));
            string d = str.substr(space + 1);
            int dot = d.find(".");
            mp[d] += cnt;
            string dd = d.substr(dot + 1);
            mp[dd] += cnt;
            int dot2 = dd.find(".");
            if (dot2 != string::npos) {
                string ddd = dd.substr(dot2 + 1);
                mp[ddd] += cnt;
            }
        }
        vector<string> res;
        for (auto p : mp) {
            res.push_back(to_string(p.second) + " " + p.first);
        }
        return res;
    }
};
