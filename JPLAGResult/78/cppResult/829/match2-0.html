class Solution {
public:
    vector<string> subdomainVisits(vector<string>& cpdomains) {
        unordered_map<string, int> mm;
        int n = cpdomains.size();
        for(int i = 0; i < n; i++){
            string s = cpdomains[i], rec;
            bool found = false;
            int count;
            for(int j = 0; j < s.size(); j++){
                char c = s[j];
                if(!found && c >= '0' && c <= '9') rec += c;
                if(c == ' ') {
                    found = true;
                    count = stoi(rec);
                    mm[s.substr(j+1)] += count;
                }
                if(c == '.') mm[s.substr(j+1)] += count;
            }
        }
        vector<string> result;
        for(auto it = mm.begin(); it != mm.end(); ++it){
            string s = to_string(it->second) + " " + it->first;
            result.push_back(s);
        }
        return result;
    }
};
