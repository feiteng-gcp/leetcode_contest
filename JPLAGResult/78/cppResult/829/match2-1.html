class Solution {
public:
    vector<string> subdomainVisits(vector<string>& a) {
        
        vector<string> ret;
        
        
        unordered_map<string, int> cnt;
        
        int n = a.size();
        
        for(int i=0; i<n; i++){
            
            
            string ss = a[i];
            
            stringstream buf(ss);
            
            int rep;
            string s;
            
            
            
            buf >> rep >> s;
            
            //cout << "hello" << endl;
            
            
            int m = s.size();
            
            int j=0;
            while(j<m){
                cnt[s.substr(j)]+= rep;
                while(j<m && s[j] != '.'){
                    j++;
                }
                j++;
            }
        }
        
        //cout << "hello" << endl;
    
        for(auto it = cnt.begin(); it!= cnt.end(); it++){
            ret.push_back(to_string(it->second) + " " + it->first);
        }
        
        return ret;
    }
};
