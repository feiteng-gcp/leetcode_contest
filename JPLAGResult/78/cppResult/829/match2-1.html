class Solution {
public:
    vector<string> subdomainVisits(vector<string>& a) {
        map<string, int> d;
        for (auto line : a) {
            int p = line.find(' ');
            int cnt = stoi(line.substr(0, p).c_str());
            line = line.substr(p + 1);
            while (!line.empty()) {
                d[line] += cnt;
                p = line.find('.');
                if (p != string::npos) {
                    line = line.substr(p + 1);
                } else {
                    line = "";
                }
            }
        }
        vector<string> ans;
        for (auto p : d) {
            string res = to_string(p.second) + " " + p.first;
            ans.push_back(res);
        }
        return ans;
    }
};
