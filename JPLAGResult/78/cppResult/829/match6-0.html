class Solution {
public:
    vector<string> splitByP(string& S)
{
    vector<string> ans;
    int len = (int)S.size(), beg = 0;
    for(int i = 0; i <= len; i++)
    {
        if(i == len || S[i] == '.')
        {
            ans.push_back(S.substr(beg));
            beg = i + 1;
        }
    }
    return ans;
}

vector<string> subdomainVisits(vector<string>& cpdomains)
{
    map<string, int> tab;
    int len = (int)cpdomains.size();
    for(int i = 0; i < len; i++)
    {
        int pos = (int)cpdomains[i].find(' ');
        int cnt = stoi(cpdomains[i].substr(0, pos));
        string s = cpdomains[i].substr(pos + 1);
        vector<string> sl = splitByP(s);
        for(int j = 0; j < sl.size(); j++) tab[sl[j]] += cnt;
    }
    vector<string> ans;
    for(auto p = tab.begin(); p != tab.end(); p++)
        ans.push_back(to_string(p->second) + " " + p->first);
    return ans;
}
};
