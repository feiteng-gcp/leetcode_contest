class Solution {
public:
    vector<string> subdomainVisits(vector<string>& cpdomains) {
        vector<string> result;
        unordered_map<string, int> map;
        for (auto s: cpdomains) {
            int num = 0;
            int loc = s.find(' ');
            num = stoi(s.substr(0, loc));
            
            int len = loc;
            s = s.substr(loc + 1, s.size() - len);
            map[s] += num;
            
            while (1) {
                int loc = s.find('.');
                if (loc == s.npos) break;
                
                string domain = s.substr(loc + 1, s.size() - loc);
                map[domain] += num;
                s = domain;
            }
        }
        
        for (auto p: map) {
            result.push_back(to_string(p.second) + " " + p.first);
        }
        return result;
    }
};
