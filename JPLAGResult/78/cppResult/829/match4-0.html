class Solution {
public:
    vector<string> subdomainVisits(vector<string>& cpdomains) {
        unordered_map<string, int> m;
        vector<string> v;
        for (auto s: cpdomains) {
            // url
            string num;
            string tmp;
            int n = 0;
            int i = 0;
            // find num
            while (s[i] != ' ') {
                num += s[i++];
            }
            n = atoi(num.c_str());
            for (int j = s.length() - 1; j > i; --j) {
                if (s[j] == '.') m[tmp] += n;
                tmp = s[j] + tmp;
            }
            m[tmp] += n;
            tmp.clear();
        }
        for (auto it: m) {
            string tmp;
            tmp += to_string(it.second);
            tmp += ' ';
            tmp += it.first;
            // cout << it.first << " " << it.second << endl;
            v.push_back(tmp);
        }
        return v;
    }
};
