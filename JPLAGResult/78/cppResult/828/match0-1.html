class Solution {
public:
    bool xorGame(vector<int>& nums) {
        int val = 0;
        int c=0;
        for(int elem : nums){
            val^=elem;
            if(elem == 0){
                c++;
            }
        }
        if(val == 0 || nums.size() % 2 == 0){
            return true;
        }
        
        return false;
        
    }
};
