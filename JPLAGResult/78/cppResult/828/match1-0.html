class Solution {
public:
    bool xorGame(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int mask = 0;
        for (auto c : nums) {
            mask ^= c;
        }
        int count = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (i < nums.size() - 1 && nums[i] == nums[i+1]) {
                i++;
                continue;
            }
            count++;
        }
        
        return !mask || !(count & 1);
    }
};
