class Solution:
    def subdomainVisits(self, cpdomains):
        """
        :type cpdomains: List[str]
        :rtype: List[str]
        """
        
        domains = {}
        
        for p in cpdomains:
            n, d = p.split(' ')
            n = int
(n)
            
            subList = d.split('.')
            
            for start in range(len(subList)):
                domain = '.'.join(subList[start:])
                if domain not in domains:
                    domains[domain] = 0
                    
                domains[domain] += n
                
        return [str(domains[d]) + ' ' + d for d in domains]
