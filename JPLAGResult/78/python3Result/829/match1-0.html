class Solution:
    def subdomainVisits(self, cpdomains):
        """
        :type cpdomains: List[str]
        :rtype: List[str]
        """
        mem = {}
        for dom in cpdomains:
            print(dom)
            print(dom.split(' '))
            count, url = dom.split(' ')
            url_parts = url.split('.')
            for i in range(len(url_parts)):
                partial_url = '.'.join(url_parts[i:])
                if partial_url not in mem:
                    mem[partial_url] = 0
                mem[partial_url] += int(count)
        result = []
        for url in mem:
            result.append(str(mem[url]) + ' ' + url)
        return result
