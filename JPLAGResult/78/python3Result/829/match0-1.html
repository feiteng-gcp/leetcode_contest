class Solution:
    def subdomainVisits(self, cpdomains):
        """
        :type cpdomains: List[str]
        :rtype: List[str]
        """
        d = {}
        for mail in cpdomains:
            t, m = mail.split(' ')
            t = int(t)
            ms = m.split('.')
            for i in range(len(ms)):
                tt = '.'.join(ms[i:])
                d[tt] = d.get(tt,0)+t
            
        res=[]
        for i in d:
            res.append('%d %s'%(d[i],i))
        return res
    
s=Solution()
print(s.subdomainVisits(["900 google.mail.com", "50 yahoo.com", "1 intel.mail.com", "5 wiki.org"]))
