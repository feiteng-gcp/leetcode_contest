class Solution:
    def soupServings(self, N):
        """
        :type N: int
        :rtype: float
        """
        if N>6000: return 1
        d = {}
        
def f(n,m):
            if (n,m) in d: return d[(n,m)]
            if n<=0 and m<=0: return 0.5
            if n<=0: return 1
            if m<=0: return 0
            t = 0.25*f(n-100,m)+0.25*f(n-75,m-25)+0.25*f(n-50,m-50)+0.25*f(n-25,m-75)
            d[(n,m)]=t
            return t
        return f(N,N)
