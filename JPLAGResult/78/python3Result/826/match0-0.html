class Solution:
    def soupServings(self, N):
        """
        :type N: int
        :rtype: float
        """
        if N > 6666:
            return 1.0
        
        def dfs(A, B):
            if (A,B) in mem:
                return mem[(A,B)]
            if A <= 0 and B > 0:
                return 1
            if A <= 0 and B <= 0:
                return 0.5
            if A > 0 and B <= 0:
                return 0
            else:
                mem[(A,B)] = 0.25 * (dfs(A-100, B) + dfs(A-75, B-25) + dfs(A-50,B-50) + dfs(A-25,B-75))
                return mem[(A,B)]
        
        mem = {}
        return dfs(N,N)
