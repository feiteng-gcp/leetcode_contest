class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        res=0
        l=[]
        if not S:
            return 0
        c=S[0]
        count=1
        for i in range(1,len(S)):
            if S[i]!=c:
                l.append([c,count])
                c=S[i]
                count=1
            else:
                count+=1
        l.append([c,count])
        for s in words:
            n=0
            if not s:
                continue
            c=s[0]
            count=1
            flag=True
            for i in range(1,len(s)):
                if s[i]!=c:
                    if n>=len(l)-1:
                        flag=False
                        break
                    if c!=l[n][0] or count>l[n][1] or (l[n][1]<3 and  l[n][1]!=count):
                        flag=False
                        break
                    else:
                        c=s[i]
                        count=1
                        n+=1
                else:
                    count+=1
                        
            if flag and c==l[-1][0] and (count==l[-1][1] or (count<l[-1][1] and l[-1][1]>2)):
                res+=1
        return res
