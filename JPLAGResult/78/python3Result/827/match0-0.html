class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        def s2wl(S):
            S += '\t'
            N = len(S)
            wList = []
            cnt = 1
            for i in range(N-1):
                if S[i]!=S[i+1]:
                    wList.append((S[i], cnt))
                    cnt = 1
                else:
                    cnt += 1
            return wList

        oList = s2wl(S)
        N = len(oList)
        cnt = 0
        # print(oList)
        for w in words:
            wList = s2wl(w)
            # print(wList)
            if len(wList)!=N:
                continue
            f = True
            for i in range(N):
                if not (wList[i][0]==oList[i][0]
                    and (wList[i][1]==oList[i][1]
                         or (oList[i][1] > wList[i][1] and oList[i][1]>=3)) ):
                    f = False
                    break
            if f:
                cnt += 1

        return cnt
