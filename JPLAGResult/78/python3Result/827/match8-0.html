class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        def dense(S):
            nums = []
            letters = []
            n = len(S)
            p = 0
            cur = 0
            while cur < n:
                letters.append(S[cur])
                nums.append(1)
                while cur + 1 < n and S[cur+1] == letters[p]:
                    nums[p] += 1
                    cur += 1
                p += 1
                cur += 1
            return [nums, letters]
        template = dense(S)
        num = 0
        for word in words:
            w = dense(word)
            if len(template[1]) == len(w[1]):
                for i in range(len(w[1])):
                    if template[1][i] == w[1][i]:
                        if template[0][i] < 3 and template[0][i] != w[0][i]:
                            break
                    else:
                        break
                else:
                    num += 1
        return num
