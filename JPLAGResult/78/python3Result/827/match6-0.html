class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        if not S: return 0
        
        def fmt(s):
            ss = [[s[0], 1]]
            for c in s[1:]:
                if c == ss[-1][0]:
                    ss[-1][1] += 1
                else:
                    ss.append([c, 1])
            return ss
        
        S = fmt(S)
        N = len(S)
        cnt = 0
        
        for w in words:
            if not w: continue
            w = fmt(w)
            if len(w) != N: continue
            for a, b in zip(w, S):
                if a[0] != b[0] or a[1] > b[1] or a[1] < b[1] < 3: break
            else:
                cnt += 1
        return cnt
