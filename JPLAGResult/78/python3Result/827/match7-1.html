class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        
        """
        1. Strict subsequence
        2. if set(b) != set(a), impossible
        
        """
        
        def encode(S):
            if not S:
                return []
            code = [[S[0], 1]]
            i = 1
            while i < len(S):
                if S[i] == code[-1][0]:
                    code[-1][1] += 1
                else:
                    code.append([S[i], 1])
                i += 1
            
            return code
        
        target = encode(S)
        possible_count = 0
        for word in words:
            candidate = encode(word)
            
            if len(candidate) != len(target):
                continue
            is_possible = True
            index = 0
            while index < len
(candidate):
                if target[index][0] != candidate[index][0] or\
                    target[index][1] < candidate[index][1] or\
                    (target[index][1] > candidate[index][1] and target[index][1] < 3):
                    is_possible = False
                    break
                index += 1
            if is_possible:
                possible_count += 1
                
        return possible_count
