class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        cnt = 0
        
        mps = [1]
        lst = [S[0]]
        for i in range(1,len(S)):
            if S[i] == S[i-1]:
                mps[-1] += 1
            else:
                mps.append(1)
                lst.append(S[i])
        # print(mps)
        # print(lst)
        for w in words:
            mps_w = [1]
            lst_w = [w[0]]
            for i in range(1,len(w)):
                if w[i] == w[i-1]:
                    mps_w[-1] += 1
                else:
                    mps_w.append(1)
                    lst_w.append(w[i])
            if len(mps) != len(mps_w):
                continue
            flag = 0
            # print(mps_w)
            # print(lst_w)
            for i in range(len(mps)):
                if lst[i] != lst_w[i]:
                    flag = 1
                    break
                    
                if mps_w
[i] > mps[i] or (mps_w[i] < mps[i] and mps[i]<3):
                    flag = 1
                    break
            if flag == 0:
                cnt += 1
                
        return cnt
