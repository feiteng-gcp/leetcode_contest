class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        s_conv = self.helper(S)
        ans = 0
        for word in words:
            cur = self.helper(word)
            if len(cur) != len(s_conv):
                break
            flag = 1
            for i in range(len(cur)):
                if cur[i][0] == s_conv[i][0] and (cur[i][1] == s_conv[i][1] or  s_conv[i][1] >= 3):
                    continue
                else:
                    flag = 0
                    break
            ans += flag
        return ans

    def helper(self, s):
        ans = []
        for x in s:
            if ans and x == ans[-1][0]:
                ans[-1][1] += 1
            else:
                ans.append([x, 1])
        return ans
