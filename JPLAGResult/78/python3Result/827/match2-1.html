class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        lookup = []
        for c in S:
            if not lookup or lookup[-1][0] != c:
                lookup.append([c, 0])
            lookup[-1][1] += 1
        results = []
        for w in words:
            processed = []
            for c in w:
                if not processed or processed[-1][0] != c:
                    processed.append([c, 0])
                processed[-1][1] += 1
            if len(processed) != len
(lookup):
                continue
            match = True
            for w_pair, s_pair in zip(processed, lookup):
                if w_pair[0] == s_pair[0] and w_pair[1] <= s_pair[1] and (s_pair[1] >= 3 or w_pair[1] == s_pair[1]):
                    pass
                else:
                    match = False
                    break
            if match:
                results.append(w)
        return len(results)
