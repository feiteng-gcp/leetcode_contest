class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        l = len(S)
        if l == 0:
            res = 0
            for w in words:
                if w == '':
                    res += 1
            return res
        def cont(s):
            l = len(s)
            temp = [(s[0], 0)]
            for i in range(l):
                if s[i] == temp[-1][0]:
                    temp[-1] = (s[i], temp[-1][1] + 1)
                else:
                    temp += [(s[
i], 1)]
            return temp
        template = cont(S)
        l = len(template)
        res = 0
        for w in words:
            m = len(w)
            if m == 0:
                continue
            haha = cont(w)
            if len(haha) != l:
                continue
            for i in range(l):
                if haha[i][0] == template[i][0]:
                    if haha[i][1] != template[i][1] and template[i][1] < 3:
                        res -= 1
                        break
                else:
                    res -= 1
                    break
            res += 1
        return res
