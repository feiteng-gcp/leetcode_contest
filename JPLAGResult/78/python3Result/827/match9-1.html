class Solution:
    def extendsTo(self, sParse, wParse):
        sIdx = 0
        wIdx = 0
        
        if len(sParse) != len(wParse):
            return False
        
        while sIdx < len(sParse):
            # Character mismatch
            
if sParse[sIdx][0] != wParse[wIdx][0]:
                return False
                
            if sParse[sIdx][1] == wParse[wIdx][1]:
                sIdx += 1
                wIdx += 1
                continue
                
            if sParse[sIdx][1] > wParse[wIdx][1] and \
               sParse[sIdx][1] >= 3:
               sIdx += 1
               wIdx += 1
               continue
               
            return False
            
        return True

    def parseWord(self, word):
        parse = []
        
        lastChar = None
        lastGroup = 0
        for c in word:
            if lastChar and c == lastChar:
                lastGroup += 1
            else:
                if lastChar:
                    parse.append((lastChar, lastGroup))
                lastChar = c
                lastGroup = 1
                
        if lastGroup > 0:
            parse.append((lastChar, lastGroup))
            
        return parse
        
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        
        sParse = self.parseWord(S)
            
        return sum(self.extendsTo(sParse, self.parseWord(w)) for w in words)
