class Solution:
    def expressiveWords(self, S, words):
        """
        :type S: str
        :type words: List[str]
        :rtype: int
        """
        if not words: return 0
        S_com = self.createList(S)        
        res = 0
        for word in words:
            word_l = self.createList(word)
            res += self.compare(S_com, word_l)
        return res
    
    def createList(self, s):
        if not s: return []
        res, lastch, num = [], s[0], 1
        
for idx in range(1, len(s)):
            if s[idx] != lastch:
                res.append((lastch, num))
                lastch = s[idx]
                num = 1
            else:
                num += 1
        res.append((lastch, num))
        return res
    
    def compare(self, a, b):
        if len(a) != len(b): return 0
        for i in range(len(a)):
            if a[i][0] != b[i][0]: return 0
            if a[i][1] <3:
                if a[i][1] != b[i][1]: return 0
            else:
                if a[i][1] < b[i][1]: return 0
        return 1
