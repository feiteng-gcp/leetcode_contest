class Solution {
    public List<String> subdomainVisits(String[] cpdomains) {
        Map<String, Integer> map = new HashMap<>();
        
        for (String s : cpdomains) {
            int i = 0;
            
            while (i < s.length() && Character.isDigit(s.charAt(i))) i++;
            
            int count = Integer.parseInt(s.substring(0, i));
            
            String[] domains = s.substring(i + 1).split("\\.");
            String domain = "";
            
            for (i = domains.length - 1; i >= 0; i--) {
                domain = domains[i] + (i < domains.length - 1 ? "." : "") + domain;
                
map.put(domain, count + map.getOrDefault(domain, 0));
            }
        }
        
        List<String> res = new ArrayList<>();
        
        for (String key : map.keySet()) {
            res.add(map.get(key) + " " + key);
        }
        
        return res;
    }
}
