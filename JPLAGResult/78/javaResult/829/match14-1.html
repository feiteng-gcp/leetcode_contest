class Solution {
    public List<String> subdomainVisits(String[] cpdomains) {
        Map<String, Integer> map = new HashMap<>();
        for (int i = 0; i < cpdomains.length; i++){
            int count = Integer.parseInt(cpdomains[i].split(" ")[0]);
            String site = cpdomains[i].split(" ")[1];

            if (map.get(site) == null)
                map.put(site, 0);
            map.put(site, map.get(site) + count);
            
            String[] tokens = site.split("\\.");
            if (tokens.length == 2){
                site = tokens[1];
                if (map.get(site) == null)
                    map.put(site, 0);
                map.put(site, map.get(site) + count); 
            }
            else {
                site = tokens[1] + "." + tokens[2];
                if (map.get(site) == null)
                    map.put(site, 0);
                map.put(site, map.get(site) + count);
                site = tokens[2];
                if (map.get(site) == null)
                    map.put(site, 0);
                
map.put(site, map.get(site) + count);
            }
        }
        
        List<String> result = new ArrayList<>();
        
        for (String s : map.keySet()){
            String temp = map.get(s) + " " + s;
            result.add(temp);
        }
        
        return result;
    }
}
