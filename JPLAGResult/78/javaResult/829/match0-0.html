class Solution {
    public List<String> subdomainVisits(String[] cpdomains) {
        HashMap<String, Integer> map = new HashMap<>();
        for(String cpdomain : cpdomains) {
            String[] parts = cpdomain.split(" ");
            int num = Integer.valueOf(parts[0]);
            String[] domain = parts[1].split("\\.");
            String cur = domain[domain.length - 1];
            map.put(cur, map.getOrDefault(cur, 0) + num);
            for(int i = domain.length - 2; i != -1; --i) {
                cur = domain[i] + "." + cur;
                map.put(cur, map.getOrDefault(cur, 0) + num);
            }
        }
        List<String> res = new ArrayList<>();
        for(String key : map.keySet()) {
            res.add(Integer.toString(map.get(key)) + " " + key);
        }
        return res;
    }
}
