class Solution {
    public List<String> subdomainVisits(String[] cpdomains) {
        
        Map<String, Integer> counts = new HashMap<String, Integer>();
        
        for (String domain : cpdomains) {
            String[] parts = domain.split(" ");
            
            int count = Integer.parseInt(parts[0]);
            
            String[] subdomains = parts[1].split("\\.");
            
            if (subdomains.length == 3) {
                String a = subdomains[0] + "." + subdomains[1] + "." + subdomains[2];
                String b = subdomains[1] + "." + subdomains[2];
                String c = subdomains[2];

                counts.put(a, counts.getOrDefault(a, 0) + count);
                counts.put(b, counts.getOrDefault(b, 0) + count);
                counts.put(c, counts.getOrDefault(c, 0) + count);
            } else {
                String b = subdomains[0] + "." + subdomains[1];
                String c = subdomains[1];

                counts.put(b, counts.getOrDefault(b, 0) + count);
                counts.put(c, counts.getOrDefault(c, 0) + count);
            }
        }
        
        List<String> res = new ArrayList<String>();
        for (String key : counts.keySet()) {
            res.add(String.format("%d %s", counts.get(key), key));
        }
        
        return res;
    }
}
