class Solution {
    public List<String> subdomainVisits(String[] cpdomains) {
        HashMap<String, Object> root = new HashMap<String, Object>();
        for(String domain : cpdomains) {
            addDomain(root, domain);
        }
        LinkedList<String> ret = new LinkedList<String>();
        fillResults(ret, root, "");
        return ret;
    }
    
    private void addDomain(HashMap<String, Object> root, String domain) {
        String[] countpair = domain.split(" ");
        Integer count = Integer.parseInt(countpair[0]);
        LinkedList<String> link = domainLink(countpair[1]);
        addCount(root, link, count);
    }
    
    private LinkedList<String> domainLink(String domain) {
        LinkedList<String> ret = new LinkedList<String>();
        for(String s : domain.split("\\.")) {
            ret.add(s);
        }
        return ret;
    }
    
    private void addCount(HashMap<String, Object> root, LinkedList<String> link, int domainCount) {
        if (link.size()==0) {
            return;
        }
        String base = link.removeLast();
        Object o = root.get(base);
        if(o==null) {
            o = new HashMap<String, Object>();
            root.put(base, o);
        }
        validateHashMap(o, "not a hashmap at " + base);
        HashMap<String, Object> map = (HashMap) o;
        Integer count = (Integer) (map.get("count"));
        if (count==null) {
            count = 0;
        }
        map.put("count", count+domainCount);
        Object next = map.get("next");
        if (next==null) {
            next = new HashMap<String, Object>();
            map.put("next", next);
        }
        validateHashMap(next, "not hashmap into " + base);
        addCount((HashMap)next, link, domainCount);
    }
    
    private void validateHashMap(Object o, String errorMessage) {
        if(o instanceof HashMap) {
            return;
        } else {
            System.out.println(o.getClass());
            throw new RuntimeException(errorMessage);
        }
    }
    
    private void fillResults(LinkedList<String> results, HashMap<String, Object> root, String base) {
        for(Map.Entry<String, Object> entry : root.entrySet()) {
            String s = entry.getKey();
            Object o = entry.getValue();
            validateHashMap(o, "did not get a map for " + base + s);
            HashMap<String, Object> map = (HashMap) o;
            Integer count = (Integer) (map.get("count"));
            results.add(count + " " + s + base);
            HashMap<String, Object> next = (HashMap) (map.get("next"));
            if (next==null) {
                continue;
            } else {
                fillResults(results, next, "." + s + base);
            }
        }
    }
}
