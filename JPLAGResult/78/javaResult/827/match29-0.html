	class Solution {
	    public int expressiveWords(String S, String[] words) {
	        int[][] fs = tof(S);
	        int ct = 0;
	        outer:
	        for(String w : words){
	        	int[][] f = tof(w);
	        	if(fs.length != f.length)continue;
	        	for(int i = 0;i < f.length;i++){
	        		if(fs[i][0] != f[i][0])continue outer;
	        		if(fs[i][1] >= 3){
	        			if(f[i][1] <= fs[i][1]){
	        			}else{
	        				continue outer;
	        			}
	        		}else{
	        			if(f[i][1] == fs[i][1]){
	        			}else{
	        				continue outer;
	        			}
	        		}
	        	}
	        	ct++;
	        }
	        return ct;
	    }
	    
	    int[][] tof(String s)
	    {
	    	int n = s.length();
	    	int[][] ret = new int[n][];
	    	int p = 0;
	    	for(int i = 0;i < n;i++){
	    		if(i == 0 || s.charAt(i) != s.charAt(i-1)){
	    			ret[p++] = new int[]{s.charAt(i), 1};
	    		}else{
	    			ret[p-1][1]++;
	    		}
	    	}
	    	return Arrays.copyOf(ret, p);
	    }
	}
