class Solution {
    public int expressiveWords(String S, String[] words) {
        if (words == null || words.length == 0) {
            return 0;
        }
        
int count = 0;
        for (String word : words) {
            if (isExpanded(S, word)) {
                count++;
            }
        }
        return count;
    }
    
    private boolean isExpanded(String S, String word) {
        if (S.equals(word)) {
            return true;
        } else if (S.length() < word.length()) {
            return false;
        }
        
        int nS = S.length();
        int nWord = word.length();
        int iS = 0;
        int iWord = 0;
        while (iS < nS && iWord < nWord) {
            
if (S.charAt(iS) != word.charAt(iWord)) {
                return false;
            }
            int sLength = 0;
            char first = S.charAt(iS);
            while (iS < nS && S.charAt(iS) == first) {
                iS++;
                sLength++;
            }
            int wordLength = 0;
            while (iWord < nWord && word.charAt(iWord) == first) {
                iWord++;
                wordLength++;
            }
            if (sLength == wordLength || sLength > wordLength && sLength >= 3) {
                continue;
            } else {
                return false;
            }
        }
        return iS == nS && iWord == nWord;
    }
}
