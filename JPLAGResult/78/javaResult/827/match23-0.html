import java.util.ArrayList;
import java.util.List;

public final class Solution {

	static class CC
	{
		final char c;
		final int count;
		public CC(char c, int count) {
			super();
			this.c = c;
			this.count = count;
		}
		public String toString(){
			return c + "(" + count + ")";
		}
	}
	static List<CC> parseCC(String s){
		ArrayList<CC> list = new ArrayList<CC>();
		for(int i=0; i<s.length();){
			char c = s.charAt(i++);
			int count = 1;
			for(;i<s.length() && s.charAt(i) == c; i++,count++);
			list.add(new CC(c, count));
		}
		return list;
	}
	static boolean check(List<CC> ext, List<CC> word){
		if(ext.size() != word.size())
			return false;
		for(int i=0; i<ext.size();i++){
			CC e = ext.get(i);
			CC w = word.get(i);
			if(e.c != w.c)
				return false;
			if(e.count == w.count)
				continue;
			if(e.count < w.count || e.count<3)
				return false;
		}
		return true;
	}
	public static int expressiveWords(String S, String[] words) {
		int rs = 0;
		List<CC> list = parseCC(S);
		for(String word : words){
			if(check(list, parseCC(word)))
				rs ++;
		}
		return rs;
	}
}
