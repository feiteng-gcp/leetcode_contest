class Solution {
    public int expressiveWords(String S, String[] words) {
        
int res = 0;
        for(String word : words) {
            if(isStrechy(S, word)) {
                res ++;
            }
        }
        return res;
    }
    
    public boolean isStrechy(String S, String word) {
        int i = 0;
        int j = 0;
        
if(S.length() < word.length()) {
            return false;
        }
        while(i<S.length() && j<word.length()) {
            int count = 0;
            char tmp = ' ';
            while(i<S.length() && (tmp == ' ' || S.charAt(i) == tmp)) {
                
                count++;
                tmp = S.charAt(i);
                i++;
            }
            if(count < 3) {
                for(int l=0;l<count;j++, l++) {
                    if(j>=word.length() || word.charAt(j)!=tmp) {
                        return false;
                    }
                }
            } else {
                int countW = 0;
                while(j<
word.length() && word.charAt(j) == tmp) {
                    j++;
                    countW++;
                }
                if(countW > count || countW==0) {
                    return false;
                }
            }
        }
        return i==S.length() && j==word.length();
    }
}
