class Solution {
    public int expressiveWords(String S, String[] words) {
        int res = 0;
        
        for (String word: words) {
            int s1 = 0, w1 = 0;
            int m = S.length(), n = word.length();
            
            while (s1 < m && w1 < n) {
                int s2 = s1 + 1;
                int w2 = w1 + 1;
                
                while (s2 < m && S.charAt(s2) == S.charAt(s1)) {
                    s2++;
                }
                
                while (w2 < n && word.charAt(w2) == word.charAt(w1)) {
                    w2++;
                }
                
                if (s2 - s1 < w2 - w1) {
                    break;
                }
                
                if (s2 - s1 > w2 - w1 && s2 - s1 < 3) {
                    break;
                }
                s1 = s2;
                w1 = w2;
            }
            
            if(s1 == m && w1 == n) res++;
            
        }
        
        return res;
    }
}
