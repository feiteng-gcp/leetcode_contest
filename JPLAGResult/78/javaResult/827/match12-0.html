class Solution {
    
    class Pair {
        int count;
        char c;
        
        Pair(int count, char c) {
            this.count = count;
            this.c = c;
        }
        
        public String toString() {
            return "[" + c + ": " + count + "]";
        }
    }
    
    public int expressiveWords(String S, String[] words) {
        
        int count = 0;
        
        for (String word : words) {
            boolean res = canStretch(S, word);
            
            if (res) {
                count++;
            }
            
            // System.out.println(word + " " + res);
        }
        
        return count;
    }
    
    private boolean canStretch(String S, String w) {
        if (S.equals(w)) {
            return true;
        }

        if (S.isEmpty() || w.isEmpty()) {
            return false;
        }
        
        List<Pair> aS = toList(S);
        List<Pair> aW = toList(w);
        
        // System.out.println(S + ": " + aS.toString());
        // System.out.println(w + ": " + aW.toString());
        
        if (aS.size() != aW.size()) {
            return false;
        }
        
        for (int i = 0; i < aS.size(); i++) {
            if (aS.get(i).c != aW.get(i).c) {
                return false;
            }
            
            if (aW.get(i).count == aS.get(i).count) {
                continue;
            }
            
            if (aS.get(i).count != 2 && aW.get(i).count <= aS.get(i).count) {
                continue;
            }
            
            return false;
        }
        
        return true;
    }
    
    private List<Pair> toList(String S) {
        List<Pair> aS = new ArrayList<Pair>();
        
        for (char c : S.toCharArray()) {
            if (aS.isEmpty() || aS.get(aS.size() - 1).c != c) {
                aS.add(new Pair(1, c));
            } else {
                aS.get(aS.size() - 1).count++;
            }
        }
        
        return aS;
    }
}
