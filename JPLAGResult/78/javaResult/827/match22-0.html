class Solution {
    public int expressiveWords(String S, String[] words) {
        List<WordGroup> sGroup = groupify(S);
        int count = 0;
        for (String word : words) {
            List<WordGroup> groupedWord = groupify(word);
            if (sGroup.size() != groupedWord.size()) {
                continue; // impossible to stretch
            }
            
            boolean valid = true;
            boolean stretched = false;
            for (int i = 0; i < sGroup.size(); i++) {
                if (sGroup.get(i).c != groupedWord.get(i).c) {
                    valid = false;
                    break;
                }
                int lettersNeeded = sGroup.get(i).count;
                int lettersHave = groupedWord.get(i).count;
                if (lettersHave > lettersNeeded) {
                    valid = false;
                    break;
                }
                int diff = Math.abs(lettersNeeded - lettersHave);
                if (diff == 0) {
                    continue;
                } else {
                    stretched = true;
                }
                if (lettersNeeded < 3) {
                    valid = false;
                    break;
                }
            }
            
            if (valid && stretched) {
                System.out.println(word);
                count++;
            }
        }
        return count;
    }
    
    private List<WordGroup> groupify(String s) {
        List<WordGroup> list = new ArrayList<>();
        if (s.isEmpty()) {
            return list; 
        }
        WordGroup cur = new WordGroup(s.charAt(0));
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(i-1)) {
                cur.count++; 
            } else {
                list.add(cur);
                cur = new WordGroup(s.charAt(i));
            }
        }
        list.add(cur);
        return list;
    }
    
    private class WordGroup {
        char c;
        int count;
        public WordGroup(char c) {
            this.c = c;
            this.count = 1;
        }
    }
}
