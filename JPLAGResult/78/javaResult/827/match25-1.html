class Solution {

	private static class Element {
		public char ch;
		public int cnt;
		public Element(char ch , int cnt) {
			this.ch = ch;
			this.cnt = cnt;
		}
	}
	
	private List<Element> getList(String string){
		
		List<Element> list = new ArrayList<>();
		int i , length = string.length();
		for (i = 0;i < length;i ++) {
			char ch = string.charAt(i);
			int cnt = 0;
			while (i < length && (string.charAt(i) == ch)) {
				i ++;
				cnt ++;
			}
			i --;
			list.add(new Element(ch , cnt));
		}
		return list;
		
	}
	
	private boolean check(List<Element> list1 , List<Element> list2) {
		
		int i = 0 , n1 = list1.size() , n2 = list2.size();
		if (n1 != n2) {
			return false;
		} else {
			for (i = 0;i < n1;i ++) {
				Element e1 = list1.get(i) , e2 = list2.get(i);
				if (e1.ch != e2.ch) {
					return false;
				}
				if (e1.cnt == e2.cnt) {
					continue;
				} else if (e1.cnt < e2.cnt) {
					return false;
				} else {
					if (e1.cnt > 2) {
						continue;
					} else {
						return false;
					}
				}
			}
			return true;
		}
		
	}

	
    public int expressiveWords(String S, String[] words) {
        
    	List<Element> list = getList(S);
    	int ans = 0;    	
    	for (String string : words) {    		
    		List<Element> temp = getList(string);
    		if (check(list , temp)) {
    			ans ++;
    		}
    	}
    	return ans;
    	
    }
}
