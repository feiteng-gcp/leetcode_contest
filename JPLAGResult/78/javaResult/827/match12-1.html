class Solution {
    public int expressiveWords(String S, String[] words) {
        int count = 0;
        for(String word : words){
            int i = 0;
            int j = 0;
            boolean result = true;
            while(i < S.length() && j < word.length()){
                char charS = S.charAt(i);
                char charWord = word.charAt(j);
                if(charS != charWord){
                    result = false;
                    break;
                }
                
int countS = 1;
                int countWord = 1;
                while(i < S.length() - 1 && S.charAt(i + 1) == charS){
                    i++;
                    countS++;
                }
                while(j < word.length() - 1 && word.charAt(j + 1) == charWord){
                    j++;
                    countWord++;
                }
                if(countWord > countS || (countWord < countS) && countS == 2){
                    result = false;
                    break;
                }
                i++;
                j++;
            }
            if(result && i == S.length() && j == word.length()){
                count++;
            }
        }
        return count;
    }
}
