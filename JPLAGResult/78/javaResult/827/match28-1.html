public class Solution {
	public int expressiveWords(String S, String[] words) {
		List<int[]> list = compress(S);
		int cntRes = 0;
		for (String word : words) {
			List<int[]> newList = compress(word);
			if (newList.size() == list.size() && canExtend(list, newList))
			{
				cntRes++;
			}
		}
		return cntRes;
	}

	private boolean canExtend(List<int[]> list, List<int[]> newList) {
		for (int i = 0; i < list.size(); ++i) {
			int[] before = newList.get(i), after = list.get(i);
			if (before[0] != after[0])
				return false;
			if (!(before[1] == after[1] || (before[1] < after[1] && after[1] >= 3)))
				return false;

		}
		return true;
	}

	private List<int[]> compress(String s) {
		List<int[]> res = new ArrayList<>();
		int pre = -1, cnt = 0;
		for (char ch : s.toCharArray()) {
			int idx = ch - 'a';
			if (idx == pre) {
				cnt++;
			} else {
				if (cnt > 0) {
					res.add(new int[]{pre, cnt});
				}
				pre = idx;
				cnt = 1;
			}
		}
		if (cnt > 0) {
			res.add(new int[]{pre, cnt});
		}
		return res;
	}
}
