class Solution {
    public int expressiveWords(String S, String[] words) {
        int count = 0;
        for(String word : words){
            if(isMatch(S, word)) count++
;
        }
        return count;
    }
    
    private boolean isMatch(String s, String w){
        
int i = 0, j = 0;
        while(i != s.length() && j != w.length()) {
            if(s.charAt(i) != w.charAt(j)) return false;
            int count1 = 0, count2 = 0;
            while(i + count1 < s.length() && s.charAt(i) == s.charAt(i + count1)) count1++;
            while(j + count2 < w.length() && w.charAt(j) == w.charAt(j + count2)) count2++;
            if(count1 < count2 || count1 < 3 && count1 != count2) return false
;
            i += count1;
            j += count2;
        }
        return i == s.length() && j == w.length();
    }
}
