class Solution {
    
    String s;
    int R = 26;
    int[] cnt = new int[R];
    int nS;
    boolean test = false;
    private boolean ok(String w) {
        if(test)
        System.out.println("proc:  " + w);
        int n = w.length();
        if (n > nS) return false;
        
        int[] wcnt = new int[R];
        for (int i = 0; i <n; i++) wcnt[w.charAt(i) - 'a']++;
        
        //can only increase w
        for (int i = 0; i < R; i++) {
            if (wcnt[i] > cnt[i]) return false;
        }
        
        // now process each chunk
        int i, j;
        for (i = 0, j = 0; i < nS && j < n;) {
            if(test)System.out.println("i = " + i + ", j = " + j);
            if (s.charAt(i) != w.charAt(j)) return false;
            char c = s.charAt(i);
            int ih = i+1;
            int jh = j+1;
            while (ih < nS && s.charAt(ih) == c) ih++;
            while (jh < n && w.charAt(jh) == c) jh++;
            int slen = ih - i;
            int wlen = jh - j;
            if(test)System.out.println("slen = " + slen + ", wlen = " + wlen);
            
if (wlen > slen) return false;
            if (wlen < slen && slen < 3) return false; // can not extend
            
            i = ih;
            j = jh;
        }
        return i == nS && j == n;
    }
    
    
    public int expressiveWords(String S, String[] words) {
        if (words == null || words.length == 0 || S == null) return 0;
        
        s = S;
        nS = S.length();
        for (int i = 0; i < nS; i++) cnt[S.charAt(i) - 'a']++;
        
        
int c = 0;
        for (String word : words) {
            if (ok(word)) {
                c++;
            }
        }
        
        return c;
    }
}
