class Solution {
    public int expressiveWords(String S, String[] words) {
        int res = 0;
        for(int i = 0; i < words.length; i++) {
            if(isStretchy(S, words[i]))
                res++
;
        }
        
        return res;
    }
    
    private boolean isStretchy(String src, String des) {
        if(des.length() > src.length())
            return false
;
        
        char[] sc = src.toCharArray();
        char[] dc = des.toCharArray();
        
        int indexS = 0, indexD = 0;
        while(indexS < sc.length && indexD < dc.length) {
            if(sc[indexS] != dc[indexD])
                return false;
            
            int countS = 1;
            indexS++;
            while(indexS < sc.length && sc[indexS] == sc[indexS - 1]) {
                indexS++;
                countS++;
            }
            
            int countD = 1;
            indexD++;
            while(indexD < dc.length && dc[indexD] == dc[indexD - 1]) {
                indexD++;
                countD++;
            }
            
            if(countS < countD)
                return false;
            
            if(countS != countD && countS < 3)
                return false;
        }
        
        return indexS == sc.length && indexD == dc.length;
        
    }
}
