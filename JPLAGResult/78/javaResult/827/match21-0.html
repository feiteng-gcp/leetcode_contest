class Solution {
    public boolean isExtend(String s, String word){
        int i = 0, j = 0;
        while(i<s.length() && j<word.length()){
            char chS = s.charAt(i);
            int count1 = 0;
            while(i<s.length() && s.charAt(i)==chS){
                i++;
                count1++;
            }
            char chW = word.charAt(j);
            int count2 = 0;
            while(j<word.length() && word.charAt(j)==chW){
                j++;
                count2++;
            }
            if(chS!=chW) return false;
            if(count1!=count2 && count1<3) return false;
        }
        if(i!=s.length() || j!=word.length()) return false;
        return true;
    }
    
    public int expressiveWords(String S, String[] words) {
        int res = 0;
        for(String word:words){
            if(isExtend(S, word)){
                res++;
            }
        }
        return res;
    }
}
