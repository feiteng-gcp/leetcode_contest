class Solution {
    public int expressiveWords(String S, String[] words) {
        int cnt = 0;
        for (String t : words) {
            if (isStretchy(S, t)) {
                cnt++;
            }
        }
        return cnt;
    }
    
    boolean isStretchy(String s, String t) {
        int i = 0, j = 0;
        while (j < t.length()) 
{
            if (s.charAt(i) != t.charAt(j)) {
                return false;
            }
            int sta = j;
            while (j < t.length() && t.charAt(j) == t.charAt(sta)) {
                j++;
            }
            int len2 = j - sta;
            sta = i;
            while (i < s.length() && s.charAt(i) == s.charAt(sta)) {
                i++;
            }
            int len1 = i - sta;
            if (len1 == len2) {
                continue;
            }
            if (len1 < len2) {
                return false;
            }
            
if (len1 > len2 && len1 < 3) {
                return false;
            }
        }
        if (i == s.length() && j == t.length()) {
            return true;
        }
        return false;
    }
}
