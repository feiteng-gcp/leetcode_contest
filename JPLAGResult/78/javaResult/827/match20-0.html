class Solution {
    public int expressiveWords(String S, String[] words) {
        List<Element> list = transform(S);
        int count = 0;
        for (String word : words) {
            List<Element> current = transform(word);
            if (check(list, current)) {
                count++;
            }
        }
        return count;
    }
    
    public boolean check(List<Element> a, List<Element> b) {
        if (a.size() != b.size()) {
            return false;
        }
        for (int i = 0; i < a.size(); i++) {
            Element A = a.get(i);
            Element B = b.get(i);
            //System.out.println(A.c + ":" + A.count + "," + B.c + ":" + B.count);
            if (A.c != B.c || (A.count != B.count && (A.count < 3 || A.count < B.count))) {
                return false;
            }
        }
        return true;
    }
    
    public List<Element> transform(String S) {
        List<Element> result = new ArrayList<>();
        if (S.length() == 0) {
            return result;
        }
        char last = S.charAt(0);
        result.add(new Element(last));
        for (int i = 1; i < S.length(); i++) {
            char cur = S.charAt(i);
            if (cur == last) {
                result.get(result.size() - 1).count++;
            } else {
                result.add(new Element(cur));
            }
            last = cur;
        }
        return result;
    }
    
    public static class Element {
        char c;
        int count;
        public Element(char c) {
            this.c = c;
            count = 1;
        }
    }
}
