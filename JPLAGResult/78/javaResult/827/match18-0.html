class Solution {
    public int expressiveWords(String S, String[] words) {
        int count = 0;
        for(String each : words){
            if(can(S,each)){
                count++;
            }
        }
        return count;
    }
    public boolean can(String S, String w){
        int i = 0;
        int j = 0;
        while(i < S.length() && j < w.length()){
            if(S.charAt(i) == w.charAt(j)){
                int x = count(S,i);
                int y = count(w,j);
                if(x < y || (x > y && x < 3)){
                    return false;
                }
                else{
                    i+=x;
                    j+=y;
                }
            }
            else{
                return false;
            }
        }
        if(i == S.length() && j == w.length())return true;
        return false;
    }
    public int count(String x, int index){
        int count = 1;
        while(x.length() > index+1){
            if(x.charAt(index) == x.charAt(index+1)){
                count++;
            }
            else{
                return count;
            }
            index++;
        }
        return count;
    }
}
