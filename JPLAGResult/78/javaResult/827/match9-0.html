class Solution {
    public int expressiveWords(String S, String[] words) {
        int num = 0;
        
        for (String word : words) {
            if (isStrechy(S, word)) {
                num++;
            }
        }
        
        return num;
    }
    
    private boolean isStrechy(String s, String w) {
        if (w.length() == 0) {
            return s.length() == 0;
        }
        
        if (s.length() < w.length()) {
            return false;
        }
        
        int i = 0;
        int j = 0;
        
        while (i < s.length() && j < w.length()) {
            if (s.charAt(i) != w.charAt(j)) {
                return false;
            }

            int a = 1;
            while (i + a < s.length() && s.charAt(i + a) == s.charAt(i)) {
                a++;
            }

            int b = 1;
            while (j + b < w.length() && w.charAt(j + b) == w.charAt(j)) {
                b++;
            }

            if (a < b || (a > b && a < 3)) {
                return false;
            }

            i += a;
            j += b;
        }

        return i == s.length() && j == w.length();
    }
}
