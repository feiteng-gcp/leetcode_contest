class Solution {
    public int expressiveWords(String S, String[] words) {
        
        ArrayList<String> in = splitWord(S);
        
        int ret = 0;
        
        for(String word: words){
            ArrayList<String> cur = splitWord(word);
            if(in.size()!=cur.size()) continue;
            boolean match = true;
            for(int i = 0;i<cur.size() && match;i++){
                if(in.get(i).length()<3){
                    if(!in.get(i).equals(cur.get(i)))
                        match = false;
                }
                else{
                    if(in.get(i).length()<cur.get(i).length()) match = false;
                    
                    else if(!in.get(i).substring(0,cur.get(i).length()).equals(cur.get(i)))
                        match = false;
                }
            }
            
            if(match) ret++;
        }
        
        return ret;
    }
    
    
    ArrayList<String>  splitWord(String s){
        ArrayList<String> groups = new ArrayList();
        StringBuilder word = new StringBuilder();
        int k = 0;
        for(int i = 0;i<s.length();i++){
            if(i+1<s.length() && s.charAt(i)==s.charAt(i+1)){
                word.append(s.charAt(i));
            }
            else{
                word.append(s.charAt(i));
                groups.add(word.toString());
                word = new StringBuilder();
            }
        }
        
        
        return groups;
        
        
    }
}
