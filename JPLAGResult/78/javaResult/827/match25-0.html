class Solution {
    class Data {
        public char ch;
        public int cnt;
        public Data(char ch, int cnt) {
            this.ch = ch;
            this.cnt = cnt;
        }
    }
    public List<Data> encode(String input) {
        List<Data> ret = new ArrayList<>();
        char[] arr = input.toCharArray();
        int cnt = 1;
        for(int i = 1; i < arr.length; i++) {
            if(arr[i] == arr[i - 1]) {
                cnt++;
            } else {
                ret.add(new Data(arr[i - 1], cnt));
                cnt = 1;
            }
        }
        ret.add(new Data(arr[arr.length - 1], cnt));
        
        return ret;
    }
    
    public int expressiveWords(String S, String[] words) {
        List<Data> ls = encode(S);
        int ret = 0;
        for(int i = 0; i < words.length; i++) {
            List<Data> lw = encode(words[i]);
            if(ls.size() == lw.size()) {
                boolean can = true;
                for(int j = 0; j < ls.size(); j++) {
                    if(lw.get(j).ch == ls.get(j).ch && (lw.get(j).cnt == ls.get(j).cnt || (ls.get(j).cnt > lw.get(j).cnt && ls.get(j).cnt != 2))) {
                        
                    } else {
                        can = false;
                        break;
                    }
                }
                if(can) {
                    ret++;
                    
                }
            }
        }
        
        return ret;
    }
}
