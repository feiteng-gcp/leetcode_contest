class Solution {
    public int expressiveWords(String S, String[] words) {
        
        int count = 0 ; 
        for(String s : words){
            if(canBe(S, s))count++; 
        }
        return count; 
    }
    
    public List<String> breaker(String S){
        String temp = ""; 
        int i = 0; 
        List<String> set1 = new ArrayList<>(); 
        while(i < S.length()){
            if(temp.equals("") || S.charAt(i) == S.charAt(i-1)){
                temp += S.charAt(i); 
            }else{
                set1.add(temp); 
                temp = S.charAt(i)+""; 
            }
            ++i; 
        }
        if(!temp.equals(""))set1.add(temp);
        return set1; 
    }
    
    public boolean canBe(String S, String s){
        List<String> set1 = breaker(S); 
        List<String> set2 = breaker(s); 
        
        if(set1.size() != set2.size())return false; 
        
        for(int i = 0 ; i < set1.size() ; ++i){
            String s1 = set1.get(i); 
            String s2 = set2.get(i); 
            
            if(s1.charAt(0) != s2.charAt(0))return false; 
            if(s2.length() > s1.length())return false; 
            
            if(s1.equals(s2))continue; 
            
            if(s1.length() < 3) return false; 
        }
        
        return true; 
    }
}
