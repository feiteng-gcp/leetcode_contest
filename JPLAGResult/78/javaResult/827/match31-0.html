class Solution {
    public int getRunLength (String s, int indx) {
        int i = indx, count = 0;
        while ((i < s.length()) && (s.charAt(i) == s.charAt(indx))) {
            i++;
            count++;
        }
        return count;
    }
    
    public int expressiveWords(String S, String[] words) {
        int ret = 0;
        for (String w: words) { 
            int i = 0, j = 0, n1 = 0, n2 = 0;
            boolean found = true;
            while (i < S.length() && j < w.length()) {
                if (S.charAt(i) != w.charAt(j)) {
                    found = false;
                    break;
                }
                n1 = getRunLength(S, i);
                n2 = getRunLength(w, j);
                
                if (n1 >= 3) {
                    if (n2 > n1) {
                        found = false;
                        break;
                    }
                } else {
                    if (n1 != n2) {
                        found = false;
                        break;
                    }
                }

                i += n1;
                j += n2;
            }
            if (found) {
                if (i == S.length() && j == w.length())
                    ret++;
            }
        }
        
        return ret;
    }
}
