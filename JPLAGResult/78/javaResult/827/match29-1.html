class Solution {
	public int expressiveWords(String S, String[] words) {
		a = gen(S.toCharArray());
		int ans = 0;
		for (String word : words) {
			if (check(word.toCharArray())) 
{
				ans++;
			}
		}
		return ans;
	}

	List<int[]> a;

	boolean check(char[] B) {
		List<int[]> b = gen(B);
		if (a.size() !
= b.size()) {
			return false;
		}
		for (int i = 0; i < a.size(); i++) {
			int[] aa = a.get(i);
			int[] bb = b.get(i);
			if (aa[0] != bb[0]) {
				return false;
			}
			if (aa[1] >= 3) {
				if (aa[1] < bb[1]) {
					return false;
				}
			} else {
				if (aa[1] != bb[1]) {
					return false;
				}
			}
		}

		return true;
	}

	List<int[]> gen(char[] A) {
		List<int[]> ans = new ArrayList<>();
		int pre = Integer.MAX_VALUE;
		int cnt = 0;
		for (int i = 0; i < A.length; i++) {
			if (pre != A[i]) {
				if (cnt > 0) {
					ans.add(new int[] { pre, cnt });
					cnt = 0;
				}
				pre = A[i];
			}
			cnt++;
		}
		if (cnt > 0) {
			ans.add(new int[] { pre, cnt });
		}
		return ans;
	}

}
