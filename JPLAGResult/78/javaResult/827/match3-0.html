class Solution {
    public int expressiveWords(String S, String[] words) {
        int result = 0;
        for (String word : words) {
            int sIndex = 0;
            int wIndex = 0;
            while (sIndex < S.length() && wIndex < word.length() && S.charAt(sIndex) == word.charAt(wIndex)) {
                int endSIndex = sIndex + 1;
                int endWIndex = wIndex + 1;
                while (endSIndex < S.length() && S.charAt(endSIndex) == S.charAt(sIndex)) {
                    endSIndex++;
                }
                while (endWIndex < word.length() && word.charAt(endWIndex) == word.charAt(wIndex)) {
                    endWIndex++;
                }
                int sCount = endSIndex - sIndex;
                int wCount = endWIndex - wIndex;
                if (sCount < wCount || (sCount > wCount && sCount < 3)) {
                    break;
                }
                sIndex = endSIndex;
                wIndex = endWIndex;
            }
            if (sIndex >= S.length() && wIndex >= word.length()) {
                result++;
            }
        }
        return result;
    }
}
