class Solution {
    public int expressiveWords(String S, String[] words) {
        int result = 0;
        for (String word : words) {
            if (helper(S, word)) {
                result++;
            }
        }
        return result;
    }
    private boolean helper(String S, String word) {
        
int i = 0;
        int j = 0;
        while (i < S.length() && j < word.length()) {
            if (S.charAt(i) != word.charAt(j)) {
                return false;
            }
            int count1 = 1;
            int count2 = 1;
            while (i < S.length() - 1 && S.charAt(i + 1) == S.charAt(i)) {
                i++;
                count1++;
            }
            
while (j < word.length() - 1 && word.charAt(j + 1) == word.charAt(j)) {
                j++;
                count2++;
            }
            if (count1 != count2 && count1 < 3) {
                return false;
            }
            i++;
            j++;
        }
        if (i == S.length() && j == word.length()) {
            return true;
        }
        return false;
    }
}
