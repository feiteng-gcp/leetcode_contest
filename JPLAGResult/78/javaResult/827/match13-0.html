class Solution {
    public int expressiveWords(String S, String[] words) {
        List<int[]> groups = new ArrayList<>();
        
        for (int i = 0; i < S.length();) {
            char c = S.charAt(i);
            int j = i;
            
            while (i < S.length() && S.charAt(i) == c) i++;
            
            groups.add(new int[] {c - 'a', i - j});
        }
        
        int res = 0;
        
        for (String word : words) {
            if (isExtendable(groups, word)) {
                res++;
            }
        }
        
        return res;
    }
    
    private boolean isExtendable(List<int[]> groups, String word) {
        int i = 0, k = 0;
        
        while (k < groups.size() && i < word.length()) {
            int[] group = groups.get(k++);
            char c = word.charAt(i);
            
            if (group[0] != c - 'a') return false;
            
            int j = i;
            
            while (i < word.length() && word.charAt(i) == c) i++;
            
            int count = i - j;
            
            if (group[1] < count || (group[1] < 3 && group[1] != i - j)) {
                return false;
            }
        }
        
        return (k == groups.size() && i == word.length());
    }
}
