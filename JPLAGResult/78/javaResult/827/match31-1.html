class Solution {
    
public int expressiveWords(String S, String[] ws) {
        int ar = 0;
        for(String w:ws){
            int lw = w.length();
            int l = S.length();
            int pw = 0;
            int p = 0;
            boolean ok = true;
            while(pw<lw&&p<l){
                int cw = 1;int c = 1;
                while(pw+1<lw&&w.charAt(pw)==w.charAt(pw+1)){
                    pw++;
                    cw++;
                }
                while(p+1<l&&S.charAt(p)==S.charAt(p+1)){
                    p++;
                    c++;
                }
                if(w.charAt(pw)!=S.charAt(p)){
                    ok = false;
                    break;
                
}else{
                    if(cw>c||(c>cw&&3>c)){
                        ok = false;
                        break;
                    }
                    pw++;
                    p++;
                }   
                
                
            }
            if(!ok||pw!=lw||p!=l){
                continue;
            }else{
                ar++;
            }
            
            
            
        }
        return ar;
    }
}
