class Solution {
    public int expressiveWords(String S, String[] words) {
        List<Node> sNodes = toList(S);
        int r = 0;
        for (String word : words) {
            List<Node> nodes = toList(word);
            if (nodes.size() != sNodes.size()) continue;
            int i = 0;
            for (; i < nodes.size(); i++) {
                Node sNode = sNodes.get(i), node = nodes.get(i);
                if (sNode.count < 3 && sNode.count != node.count) break;
                if (sNode.count >= 3 && node.count > sNode.count) break;
            }
            if (i == nodes.size())
                r++;
        }
        return r;
    }

    List<Node> toList(String w) {
        List<Node> nodes = new LinkedList<>();
        int count = 1;
        for (int i = 1; i < w.length(); i++)
            if (w.charAt(i - 1) != w.charAt(i)) {
                nodes.add(new Node(w.charAt(i - 1), count));
                count = 1;
            } else count++;
        nodes.add(new Node(w.charAt(w.length() - 1), count));
        return nodes;
    }

    class Node {
        final char c;
        final int count;

        Node(char c, int count) {
            this.c = c;
            this.count = count;
        }
    }
}
