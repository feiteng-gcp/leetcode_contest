class Solution {
    public int expressiveWords(String S, String[] words) {
        List<String> given = process(S);

        int ct = 0;
        for (String word: words) {
            List<String> now = process(word);
            boolean okay = true;
            if (given.size() == now.size()) {
                for (int i = 0; i < now.size(); i++) {
                    String[] giveN = given.get(i).split(" ");
                    String[] noW = now.get(i).split(" ");
                    if (!giveN[0].equals(noW[0])) {
                        okay = false;
                        break;
                    } else {
                        int Given = Integer.parseInt(giveN[1]);
                        int Now = Integer.parseInt(noW[1]);
                        if (Now > Given || (Now < Given && Given < 3)) {
                            okay = false;
                            break;
                        }
                    }
                }
                if (okay) {
                    // System.out.println(word);
                    ct++;
                }
            }
        }

        return ct;
    }

    public List<String> process(String S) {
        char[] chars = S.toCharArray();

        int[] counts = new int[chars.length];

        List<String> processed = new LinkedList();

        char prevCh = chars[0];
        int prevCount = 1;
        for (int i = 1; i < chars.length; i++) {
            if (chars[i] == prevCh) {
                prevCount++;
            } else {
                processed.add(prevCh + " " + prevCount);
                prevCh = chars[i];
                prevCount = 1;
            }
        }
        processed.add(prevCh + " " + prevCount);
        return processed;
    }
}
