class Solution {
    public int expressiveWords(String S, String[] words) {
        if (words == null || words.length == 0) {
            return 0;
        }
        
List<String> sa = convert(S);
        int ans = 0;
        for (String w : words) {
            List<String> wa = convert(w);
            if (wa.size() != sa.size()) {
                continue;
            }
            boolean extend = true;
            for (int i = 0; i < wa.size(); ++i) {
                if (wa.get(i).length() == sa.get(i).length()) {
                    continue;
                }
                if (wa.get(i).length() > sa.get(i).length() 
                    || sa.get(i).length() < 3) 
{
                    extend = false;
                    break;
                }
            }
            if (extend) {
                ++ans;
            
}
        }
        return ans;
    }
    
    private List<String> convert(String s) {
        List<String> ans = new ArrayList<>();
        if (s.length() == 0) {
            return ans;
        }
        int i = 0;
        StringBuilder sb = new StringBuilder();
        while (i < s.length()) {
            sb.append(s.charAt(i));
            if (i + 1 == s.length() || s.charAt(i + 1) != s.charAt(i)) {
                ans.add(sb.toString());
                // System.out.println(sb.toString());
                sb.setLength(0);
            }
            ++i;
        }
        return ans;
    }
}
