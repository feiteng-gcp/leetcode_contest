class Solution {
    Map<String, Double> map = new HashMap<>();
    public double soupServings(int N) {
        return helper(N, N, 0.25);
    }
    
    public double helper(int A, int B, double flag){
        if(flag < 1e-50) return 1;
        
        String s = A + " " + B;
        if(map.get(s) != null)
            return map.get(s);
        
        double res = 0;
        if(A > 100){
            res += helper(A - 100, B, flag * 0.25);
        }else{
            if(B == 0)
                res += 0.5;
            else
                res += 1;
        }
        
        if(A > 75){
            if(B > 25)
                res += helper(A - 75, B - 25, flag * 0.25);
        }else{
            if(B <= 25)
                res += 0.5;
            else
                res += 1;
        }
        
        if(A > 50){
            if(B > 50)
                res += helper(A - 50, B - 50, flag * 0.25);
        }else{
            if(B <= 50)
                res += 0.5;
            else
                res += 1;
        }      
        
        if(A > 25){
            if(B > 75)
                res += helper(A - 25, B - 75, flag * 0.25);
        }else{
            if(B <= 75)
                res += 0.5;
            else
                res += 1;
        }
        
        map.put(s, res * 0.25);
        return res * 0.25;
    }
}
