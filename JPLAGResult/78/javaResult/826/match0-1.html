class Solution {
      public double soupServings(int N) {
        if (N > 10000) return 1;
        return exp(N, N);
    }

    private static Map<Long, Double> kk = new HashMap<>();
    public double exp(int i1, int i2) {
        if (i1 <= 0 && i2 <= 0) {
            return 0.5;
        }
        if (i2 <= 0) return 0;
        
if (i1 <= 0) return 1;

        long key = i1;
        key *= 1000000000L;
        key += i2;

        if (kk.containsKey(key)) return kk.get(key);

        double res = 0;
        res += exp(i1 - 100, i2) * 0.25;
        res += exp(i1 - 75, i2 - 25) * 0.25;
        res += exp(i1 - 50, i2 - 50) * 0.25;
        res += exp(i1 - 25, i2 - 75) * 0.25;

        kk.put(key, res);
        return res;
    }
}
