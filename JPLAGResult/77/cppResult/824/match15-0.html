class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        vector<int> rs = {1, 0};
        for(auto c : S) {
            if (rs[1] + widths[c - 'a'] <= 100) {
                rs[1] += widths[c - 'a'];
            } else {
                rs[0] ++;
                rs[1] = widths[c - 'a'];
            }
        }
        return rs;
    }
};
