class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        vector<int> re;
        if (S.size() == 0) {
            re = {0, 0};
            return re;
        }
        int lines = 1;
        int currLength = 0;
        for (auto c : S) {
            int charLength = widths[c - 'a'];
            if (currLength + charLength > 100) {
                lines++;
                currLength = charLength;
            } else {
                currLength += charLength;
            }
        }
        re = {lines, currLength};
        return re;
    }
};
