class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        vector<int> ans = {1, 0};
        for (const auto& iter : S) {
            int cur = iter - 'a';

            if (ans[1] + widths[cur] <= 100) {
                ans[1] += widths[cur];
            } else {
                ans[0]++;
                ans[1] = widths[cur];
            }
        }
        return ans;
    }
};
