class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int resNum = 0, resWidth = 0;
        for(int i=0;i<S.length();i++){
            if(100 < resWidth + widths[S[i] - 'a']){
                resNum++;
                resWidth = widths[S[i] - 'a'];
            }
            else{
                resWidth += widths[S[i] - 'a'];
            }
        }
        if(resWidth != 0) resNum++;
        vector<int> res;
        res.push_back(resNum);
        res.push_back(resWidth);
        return res;
    }
};
