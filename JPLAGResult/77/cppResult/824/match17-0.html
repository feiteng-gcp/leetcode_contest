class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int numline = 1;
        int lw = 100;
        int clen = 0;
        for(int i = 0; i < S.size(); ++i){
            int w = widths[S[i]-'a'];
            if(clen + w > 100){
                numline++;
                clen = w;
            }
            else
                clen += w;
        }
        vector<int> res;
        res.push_back(numline);
        res.push_back(clen);
        return res;
    }
};
