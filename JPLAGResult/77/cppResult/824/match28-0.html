class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int n =S.size();
        int vis[26];
        memset(vis,0,sizeof(vis));
        for(int i=0;i<n;i++)
        {
            if(S[i]>='A' &&S[i]<='Z') S[i]=S[i]-'A'+'a';
            vis[S[i]-'a']++;
        }
        int cnt =0;
        int line=1;
        for(int i=0;i<n;i++)
        {
            int tmp = widths[S[i]-'a'];
            if(cnt+tmp>100)
            {
                line ++;
                cnt =tmp;
            }
            else if(cnt+tmp==100)
            {
                line++;
                cnt=0;
            }
            else
            {
                cnt+=tmp;
            }
        }
        vector<int> res;
        res.push_back(line);
        res.push_back(cnt);

        return res;
    }
};
