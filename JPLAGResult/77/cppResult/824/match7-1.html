class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string s) {
        int len = s.size();
        if (len == 0) {
            return {0, 0};
        }
        

        int icur = 0;
        int cnt = 0;
        for (char c : s) {
            if (cnt + widths[c - 'a'] > 100) {
                icur += 1;
                cnt = 0;
            }
            cnt += widths[c - 'a'];
        }
        
        return {icur + 1, cnt};
    }
};
