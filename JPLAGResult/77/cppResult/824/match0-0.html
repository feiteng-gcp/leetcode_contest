class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        vector<int> res;
        res.push_back(0);
        res.push_back(0);
        int x = 0;
        for (int i = 0; i < S.size(); i++) {
            if (x + widths[S[i] - 'a'] <= 100) {
                x = x + widths[S[i] - 'a'];
            } else {
                x = widths[S[i] - 'a'];
                res[0]++;
            }
        }
        if (x > 0) res[0]++;
        res[1] = x;
        return res;
    }
};
