class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int len=0, line=1;
        for (char c:S){
            int idx = c-'a';
            if (len + widths[idx] > 100 ){
                len = widths[idx];
                line++;
            }
            else {
                len += widths[idx];
            }
        }
        return vector<int>{line,len};
    }
};
