class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int lines = 1;
        int cur = 0;
        
        for (auto ch : S) {
            auto w = widths[ch - 'a'];
            if (cur + w > 100) {
                ++lines;
                cur = w;
                //cout << ch << ": " << lines << " " << cur << endl;
            } else {
                cur += w;
            }
        }
        
        return {lines, cur};
        
    }
};
