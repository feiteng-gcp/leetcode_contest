class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int w = 0;
        int lines = 1;
        for (auto ch : S) {
            int idx = ch - 'a';
            w += widths[idx];
            if (w > 100) {
                w = widths[idx];
                lines++;
            }
        }
        return {lines, w};
    }
};
