class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int line = 0, last = 0;

        int cur = 0;
        for (int i = 0; i < S.size(); ++i) {
            if (last + widths[S[i] - 'a'] > 100) {
                line++;
                last = widths[S[i] - 'a'];
            } else {
                last += widths[S[i] - 'a'];
            }
        }
        return {line + 1, last};
    }
};
