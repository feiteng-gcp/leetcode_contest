class Solution {
public:
  vector<int> numberOfLines(vector<int>& widths, string S) {

    int l = 1;
    int restWidth = 100;
    for (auto c: S) {
      int w = widths[c - 'a'];

      if (restWidth < w) {
	restWidth = 100;
	l++;
      }

      restWidth -= w;
    }

    vector<int> ret;
    ret.push_back(l);
    ret.push_back(100 - restWidth);
    return ret;
  }
};
