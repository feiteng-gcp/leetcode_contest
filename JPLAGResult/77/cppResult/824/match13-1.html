class Solution {
public:
    vector<int> numberOfLines(vector<int>& w, string s) {
      int a=0,b=0;
      for(char c:s){
	int k=w[c-'a'];
	b+=k;
	if(b>100) b=k,a++;
      }
      return {a+1,b};
    }
};
