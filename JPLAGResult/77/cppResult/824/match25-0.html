class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int ans_line = 0, ans_char = 0;
        for(auto c: S) {
            if(ans_char == 0) {
                ++ans_line;
            }
            ans_char += widths[c - 'a'];
            if(ans_char > 100) {
                ans_char = widths[c - 'a'];
                ++ans_line;
            }
        }
        vector<int> ans({ans_line, ans_char});
        return ans;
    }
};
