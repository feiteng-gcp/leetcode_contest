class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int now = 0;
        int cnt = 1;

        for(int i = 0;i < S.length();i++){
            int id = S[i] - 'a';
            if(now + widths[id] > 100){
                cnt++;
                now = widths[id];
            }else{
                now += widths[id];
            }
        }
        vector<int> ans;
        ans.push_back(cnt);
        ans.push_back(now);
        return ans;
    }
};
