class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
      if (S.empty()) return vector<int>{0,0};
        int total = 0;
      int res = total % 100;
      int l = 1;
      for (auto c: S) {
        total += widths[c - 'a'];
        if (total > 100) {
          l++;
          total = widths[c - 'a'];
        }
      }
      
      return vector<int>{l, total};
    }
};
