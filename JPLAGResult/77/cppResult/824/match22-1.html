class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {

        int l = 0, i = 0;
        
        for (auto c : S) {
            i += widths[c - 'a'];
            if (i > 100) {
                l++;
                i = widths[c - 'a'];
            }
        }
        
        return {l+1, i};
    }
};
