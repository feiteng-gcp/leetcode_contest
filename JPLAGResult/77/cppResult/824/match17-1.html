class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string S) {
        int num=0,cnt=0;
        for(auto c:S)

        {
            int w=widths[c-'a'];
            if(cnt+w>100)
            {
                num++;
                cnt=w;
            }else cnt+=w;
        }
        vector<int> ans;
        ans.push_back(num+1);
        ans.push_back(cnt);
        return ans;
    }
};
