class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int> >& grid) {
        int n=grid.size();
        int m=grid[0].size();
        int a[m]={0}; //hang
        int b[n]={0}; //lie
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                b[i] =max(b[i],grid[i][j]);
            }
        }
        for(int j=0;j<m;j++)
        {
            for(int i=0;i<n;i++)
            {
                a[j]=max(a[j],grid[i][j]);
            }
        }
        int res=0;
        for(int i=0;i<n;i++) cout<<a[i]<<" ";
        cout<<endl;
        for(int i=0;i<m;i++) cout<<b[i]<<" ";
        cout<<endl;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                res+= min(a[i],b[j])-grid[i][j] ;
            }
        }
        return res;
    }
};
