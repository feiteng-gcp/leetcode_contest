class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) {
        vector<int> rowLine(grid.size(), 0), colLine(grid[0].size(), 0);

        for(int i = 0; i < grid.size(); i++) {
            for(int j = 0; j < grid[i].size(); j++) {
                rowLine[i] = max(rowLine[i], grid[i][j]);
                colLine[j] = max(colLine[j], grid[i][j]);
            }
        }
        int sum = 0;
        for(int i = 0; i < grid.size(); i++) {
            for(int j = 0; j < grid[i].size(); j++) {
                sum += min(rowLine[i], colLine[j]) - grid[i][j];
            }
        }
        return sum;
    }
};
