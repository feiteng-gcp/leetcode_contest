typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T> >;

const ll MOD=1e9+7;
class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& g) {
      int N=g.size(),M=g[0].size();

      vector<int> a(N),b(M);
      for(int i=0;i<N;i++){
	for(int j=0;j<M;j++){
	  a[i]=max(a[i],g[i][j]);
	  b[j]=max(b[j],g[i][j]);
	}
      }

      ll res=0;
      for(int i=0;i<N;i++){
	for(int j=0;j<M;j++){
	  res+=min(a[i],b[j])-g[i][j];
	}
      }
      return res;
    }
};
