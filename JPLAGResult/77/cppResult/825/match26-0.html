class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> maxH(m, 0), maxV(n, 0);
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j] > maxH[i]) maxH[i] = grid[i][j];
                if(grid[i][j] > maxV[j]) maxV[j] = grid[i][j];
            }
        }
        int res = 0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                res += (min(maxH[i], maxV[j]) - grid[i][j]);
            }
        }
        return res;
    }
};
