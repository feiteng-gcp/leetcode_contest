class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) 
    {
        int m=grid.size();
        if(m==0) return 0;
        int n=grid[0].size();
        
        vector<int> tb(n,0);
        vector<int> lr(m,0);
        
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                tb[j]=max(tb[j],grid[i][j]);
                lr[i]=max(lr[i],grid[i][j]);
            }
        }
        
        int ret=0;
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                ret+=min(tb[j],lr[i])-grid[i][j];
            }
        }
        
        return ret;
        
    }
};
