class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) {
        if (grid.size() == 0 || grid[0].size() == 0) return 0;
        int rowNum = grid.size();
        int colNum = grid[0].size();
        vector<int> rowLim(rowNum, 0);
        vector<int> colLim(colNum, 0);
        for (int rowIndex = 0; rowIndex < rowNum; ++rowIndex) {
            for (int colIndex = 0; colIndex < colNum; ++colIndex) {
                rowLim[rowIndex] = max(rowLim[rowIndex], grid[rowIndex][colIndex]);
                colLim[colIndex] = max(colLim[colIndex], grid[rowIndex][colIndex]);
            }
        }
        int re = 0;
        for (int rowIndex = 0; rowIndex < rowNum; ++rowIndex) {
            for (int colIndex = 0; colIndex < colNum; ++colIndex) {
                re += min(rowLim[rowIndex], colLim[colIndex]) - grid[rowIndex][colIndex];
            }
        }
        return re;
    }
};
