class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) {
        int nx = grid.size();
        if (nx == 0) {
            return 0;
        }
        int ny = grid[0].size();
        if (ny == 0) {
            return 0;
        }
        
        vector<int> maxrow(nx);
        vector<int> maxcol(ny);
        for (int i = 0; i < nx; i++) {
            for (int j = 0; j < ny; j++) {
                maxrow[i] = max(maxrow[i], grid[i][j]);
                maxcol[j] = max(maxcol[j], grid[i][j]);
            }
        }
        
        int ret = 0;
        for (int i = 0; i < nx; i++) {
            for (int j = 0; j < ny; j++) {
                int v = grid[i][j];
                int max = min(maxrow[i], maxcol[j]);
                ret += max - v;
            }
        }
        
        return ret;
    }
};
