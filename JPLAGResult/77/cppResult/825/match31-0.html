class Solution {
public:
    int n, m;
    vector<int> row, col;
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        row.resize(n,0);
        col.resize(n,0);
        
        for (int i=0; i<n; i++){
            for (int j=0; j<m; j++){
                row[i] = max( row[i], grid[i][j] );
                col[j] = max( col[j], grid[i][j] );
            }
        }
        
        int ans=0;
        for (int i=0; i<n; i++){
            for (int j=0; j<m; j++){
                int h = min(row[i],col[j]);
                ans += h-grid[i][j];
            }
        }
        return ans;
    }
};
