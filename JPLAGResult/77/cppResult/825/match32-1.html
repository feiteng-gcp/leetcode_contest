class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) {
        if(grid.size() == 0 || grid[0].size() == 0)
            return -1;
        int m = grid.size();
        int n = grid[0].size();
        vector<int> rows(m, 0);

        vector<int> cols(n, 0);
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(grid[i][j] > rows[i])
                    rows[i] = grid[i][j];
                if(grid[i][j] > cols[j])
                    cols[j] = grid[i][j];
            }
        }
        int ret = 0;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(grid[i][j] < min(rows[i], cols[j]))
                    ret += min(rows[i], cols[j])-grid[i][j];
            }
        }
        return ret;
    }
};
