class Solution {
public:
    int maxIncreaseKeepingSkyline(vector<vector<int>>& grid) {
        int n = grid.size(),m = grid[0].size();
        vector<int> a(n,0);
        vector<int> b(m,0);

        for(int j = 0;j < m;j++){
            for(int i = 0;i < n;i++){
                b[j] = max(b[j],grid[i][j]);
            }
        }
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                a[i] = max(a[i],grid[i][j]);
            }
        }
        int ans = 0;
        for(int i = 0;i < n;i++){

            for(int j = 0;j < m;j++){
                cout << min(a[i],b[j]) << " ";
                ans += min(a[i],b[j]) - grid[i][j];
            }
            cout << endl;
        }
        return ans;
    }
};
