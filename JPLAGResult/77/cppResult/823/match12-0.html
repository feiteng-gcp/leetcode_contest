const int N=30,M=300000;
int dp[N][N][M];
class Solution {
public:
    vector<pair<int,int> > get(vector<int>&A,int s,int t)
    {
        int n=t-s;
        vector<pair<int,int> > ans;
        for(int i=0;i<(1<<n);i++)
        {
            int _sum=0,_tot=0;
            for(int j=0;j<n;j++)
                if(i>>j&1)
                {
                    _tot++;
                    _sum+=A[s+j];
                }
            ans.push_back(make_pair(_sum,_tot));
        }
        return ans;
    }
    int sum=0,tot=0;
    bool splitArraySameAverage(vector<int>& A) {
        tot=A.size();
        sum=0;
        for(auto i:A) sum+=i;
        
        if(A.size()==1) return 0;
        int n=tot/2,m=tot-n;
        vector<pair<int,int> > ls=get(A,0,n);
        vector<pair<int,int> > rs= get(A,n,tot);
        set<pair<int,int>> g;
        for(auto i:rs) g.insert(i);
        for(int i=0;i<ls.size();i++)
        {
            int lsum=ls[i].first,ltot=ls[i].second;
            int _s=0,_t=m;
            if(ltot==0) _s=1;
            if(ltot==n) _t=m-1;
            for(int rtot=_s;rtot<=_t;rtot++)
            if((sum*(rtot+ltot))%tot==0)
            {
                int _rsum=sum*(rtot+ltot)/tot-lsum;
                pair<int,int> pp= make_pair(_rsum,rtot);
                if(g.find(pp)!=g.end()) return 1;
            }
        }
        return 0;
    }
};
