class Solution {
public:
    bool backtracking(int index, int BSum, int totalSum, int BSize, int totalSize, vector<int> &A) {
        if (index >= totalSize) return false;
        BSum += A[index];
        BSize += 1;
        int CSum = totalSum - BSum;
        int CSize = totalSize - BSize;
        if (totalSum * CSize == totalSize * CSum) {
            //cout << "trigger" << CSize << endl;
            if (totalSize == BSize) return false;
            return true;
        }
        if (totalSum * CSize > totalSize * CSum) return false;

        for (int ii = index + 1; ii < totalSize; ++ii) {
            if (backtracking(ii, BSum, totalSum, BSize, totalSize, A)) {
                return true;
            }
        }
        return false;
    }
    
    bool splitArraySameAverage(vector<int>& A) {
        if (A.size() <= 1) return false;
        sort(A.begin(), A.end());
        int totalSize = A.size(); 
        int totalSum = 0;
        for (auto num : A) totalSum += num;
        for (int ii = 0; ii < totalSize; ++ii) {
            if (backtracking(ii, 0, totalSum, 0, totalSize, A)) return true;
        }
        return false;
    }
};
