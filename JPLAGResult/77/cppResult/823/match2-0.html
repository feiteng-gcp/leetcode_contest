const int maxN = 32;
const int maxK = 10005;
const int mod = (int)1e9 + 7;
const int inf = (int)1e9;

bool dp[maxN][maxK * maxN];
class Solution {
public:


    bool splitArraySameAverage(vector<int>& A) {
        if(A.size() <= 1){
            return false;
        }
        int n = A.size();
        int total = accumulate(A.begin() , A.end() , 0);

        for(int i = 0; i <= n; ++i){
            for(int k = 0; k <= total; ++k){
                dp[i][k] = false;
            }
        }
        dp[0][0] = true;
        for(int i = 0; i < n; ++i){
            for(int c = i; c >= 0; c--){
                for(int sum = 0; sum <= total; sum++){
                    if(dp[c][sum]){
                        dp[c + 1][sum + A[i]] = true;
                    }
                }
            }
        }

        for(int cant = 1; cant < n; cant++){
            if( (cant * total) % n == 0){
                int sum = (cant * total) / n;
                if(dp[cant][sum]){
                    return true;
                }
            }
        }
        return false;
    }
};
