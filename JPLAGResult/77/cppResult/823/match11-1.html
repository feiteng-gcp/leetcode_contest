class Solution {
public:
    bool splitArraySameAverage(vector<int>& A) {
        int sum=0,n=A.size();
        for(int i=0;i<n;i++) {
            sum+=A[i];
        }
        int n1=n/2,n2=(n+1)/2;
        set<int> S;

        for(int s=0;s<(1<<n1)-1;s++) {
            int cnt=0,tmp=0;
            for(int i=0;i<n1;i++) {
                if((s>>i)&1) {
                    cnt++;
                    tmp+=A[i];
                }
            }
            S.insert(-n*tmp+sum*cnt);
        }
        for(int s=1;s<(1<<n2);s++) {
            int cnt=0,tmp=0;
            for(int i=0;i<n2;i++) {
                if((s>>i)&1) {
                    cnt++;
                    tmp+=A[i+n1];
                }
            }
            if(S.count(n*tmp-sum*cnt)) {
                return true;
            }
        }
        S.clear();
        for(int s=1;s<(1<<n1);s++) {
            int cnt=0,tmp=0;
            for(int i=0;i<n1;i++) {
                if((s>>i)&1) {
                    cnt++;
                    tmp+=A[i];
                }
            }
            S.insert(-n*tmp+sum*cnt);
        }
        for(int s=0;s<(1<<n2)-1;s++) {
            int cnt=0,tmp=0;
            for(int i=0;i<n2;i++) {
                if((s>>i)&1) {
                    cnt++;
                    tmp+=A[i+n1];
                }
            }
            if(S.count(n*tmp-sum*cnt)) {
                return true;
            }
        }
        return false;
    }
};
