class Solution {
private:
    set<pair<pair<int, int>, int> > visited;
    bool find(int need, int pos, int space, vector<int>& A) {
        if(space == 0) {
            if(need != 0) {
                return false;
            } else {
                return true;
            }
        }
        
        auto key = make_pair(make_pair(need, pos), space);
        if(visited.count(key) > 0) {
            return false;
        }
        
        for(int i = pos; i < A.size(); ++i) {
            if(A[i] > need) {
                break;
            }
            if(find(need - A[i], i + 1, space - 1, A) == true) {
                return true;
            }
        }
        
        visited.insert(key);
        return false;
    }
public:
    bool splitArraySameAverage(vector<int>& A) {
        if(A.size() < 2) {
            return false;
        }
        double sum = 0;
        for(auto num: A) {
            sum += num;
        }
        double avg = sum / A.size();
        sort(A.begin(), A.end());
        for(int i = 1; i <= A.size() / 2; ++i) {
            if(i * avg != (int)(i * avg)) {
                continue;
            }
            if(find(i * avg, 0, i, A) == true) {
                return true;
            }
        }
        
        return false;
    }
};
