class Solution {
public:
    //bitset<300010> dp[16];
    bool splitArraySameAverage(vector<int>& A) {
        int su=0;
        int n=A.size();
        //cout<<n<<endl;
        for(int i=0;i<n;++i) su+=A[i];
        for(int i=1;i<=n/2;++i){
            if(su*i%n) continue;

            int k=su*i/n;
            bitset<300010> dp[16];
            dp[0][0]=1;
            for(int j=0;j<n;++j){
                for(int a=i;a>0;--a){
                    for(int b=k;b>=A[j];--b){
                        if(dp[a-1][b-A[j]]) dp[a][b]=1;
                    }
                }
            }
            //cout<<i<<' '<<k<<' '<<dp[i][k]<<endl;
            if(dp[i][k]) return true;
        }
        return false;
    }
};
