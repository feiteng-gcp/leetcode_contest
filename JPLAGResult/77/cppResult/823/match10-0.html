class Solution {
public:
    map<int,bool> mp[16],mp1[16];
    bool splitArraySameAverage(vector<int>& A) {
        if(A.size() == 1) return 0;
        int n = A.size();
        int n1 = n / 2,n2 = n - n1;
        mp[0][0] = true;
        mp1[0][0] = true;
        int S = 0;
        for(int i = 0;i < n;i++){
            S += A[i];
        }
        for(int i = 0;i < (1 << n1);i++){
            int sum = 0;
            int cnt = 0;
            for(int j = 0;j < n1;j++){
                if(i & (1 << j)){
                    sum += A[j];
                    cnt++;
                }
            }
            mp[cnt][sum] = true;
        }
        for(int i = 0;i < (1 << n2);i++){
            int sum = 0;
            int cnt = 0;
            for(int j = 0;j < n2;j++){
                if(i & (1 << j)){
                    sum += A[j + n1];
                    cnt++;
                }
            }
            mp1[cnt][sum] = true;
        }
        for(int i = 1;i < n;i++){
            double aver = S * 1.0 / n;
            double s1 = aver * i;
            int s2 = (int) s1;
            if(1.0 * s2 != s1) continue;
            for(int j = 0;j <= i;j++){
                for(auto x : mp[j]){
                    int s3 = x.first;
                    if(mp1[i - j][s2 - s3]) return true;
                }
            }
        }
        return false;
    }
};
