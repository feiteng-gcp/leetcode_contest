static bool f[30][300005];

class Solution {
public:
    bool splitArraySameAverage(vector<int>& A) {
        int *a = A.data();
        int n = (int)A.size();
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += a[i];
        }
        memset(f, 0, sizeof(f));
        f[0][0] = 1;
        int s = 0;
        for (int i = 0; i < n; i++) {
            int x = a[i];
            s += x;
            for (int cnt = i; cnt >= 0; cnt--) {
                for (int j = s - x; j >= 0; j--) {
                    f[cnt + 1][j + x] |= f[cnt][j];
                }
            }
        }
        for (int i = 1; i < n; i++) {
            if (sum * i % n == 0 && f[i][sum * i / n]) {
                return true;
            }
        }
        return false;
    }
};
