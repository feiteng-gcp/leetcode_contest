class Solution {

public:
    bool splitArraySameAverage(vector<int>& A) {
        int n=A.size();
        int sum = accumulate(A.begin(),A.end(),0);
        sort(A.begin(),A.end());
        for(int i=1;i<=n/2;i++){
            if(((sum*i)%n)!=0) continue;
            else if(backPack((sum*i)/n,A,i)) {
                //cout<<i<<endl;
                return true;
            }
        }
        return false;
    }
    
    bool backPack(int m, vector<int> A , int n) {
        unordered_set<int> s;
        dfs(A,vector<int>(),n,0,s);
        return s.count(m);
       
    }
    
    void dfs(vector<int> A, vector<int> temp, int bit, int index, unordered_set<int> &s){
        if(temp.size()==bit){
            s.insert(accumulate(temp.begin(),temp.end(),0));return;
        }
        for(int i=index;i<A.size();i++){
            temp.push_back(A[i]);
            dfs(A,temp,bit,i+1,s);
            temp.pop_back();
        }
    }
};
