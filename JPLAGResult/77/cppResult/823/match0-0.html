class Solution {
public:
    int sum,n;
    vector<int> a;
    //int f[31][16][150001];
    int ff(int sta,int sum,int m){
        //cout<<sum<<' '<<m<<endl;
        if (m==0&&sum!=0) return 0;
        if (sum==0&&m==0) return 1;
        //if (f[sta][sum][m]==1) return 1;
        for (int i=sta;i<n;i++){
            if (sum-a[i]<0) continue;
            if (ff(i+1,sum-a[i],m-1)) return 1;
        }
        return 0;
    }
    bool splitArraySameAverage(vector<int>& A) {
        sort(A.begin(),A.end());
        n=A.size();
        a=A;
        int sum=0;
        for (int i=0;i<n;i++) sum+=A[i];
        for (int i=1;i<=n/2;i++){
            if (sum*i%n!=0) continue;
            //memset(f,0,sizeof(f));
            //cout<<i<<endl;
            if (ff(0,sum*i/n,i)) {
                //cout<<sum*i/n<<endl;
                return true;
            }
        }
        return false;
    }
};
