class Solution {
private:
    bool helper(vector<int>& A, int idx, int cur_sum, int left, int target){
        if(left == 0 && cur_sum == target)
            return true;
        if(idx >= A.size() || left == 0 || cur_sum > target)
            return false;
        return helper(A, idx+1, cur_sum, left, target) || helper(A, idx+1, cur_sum+A[idx], left-1, target);
    }
public:
    bool splitArraySameAverage(vector<int>& A) {
        int times = A.size()/2;
        int sum_value = 0;
        for(int i = 0; i < A.size(); ++i){
            sum_value += A[i];
        }
        for(int i = 1; i <= times; ++i){
            if(((long)sum_value * (long)i)%A.size() != 0)
                continue;
            else{
                int target = ((long)sum_value * i)/A.size();
                bool tmp = helper(A, 0, 0, i, target);
                if(tmp == true)
                    return true;
            }    
        }
        return false;
    }
};
