class Solution {
public:
    bool splitArraySameAverage(vector<int>& A) {
        int gloSum = 0;
        for(int i=0 ; i<A.size() ; i++){
            gloSum += A[i];
        }
        sort(A.begin(),A.end());
        
        return sameLocAvgFound(A,1,gloSum,A[0],1);       
    }
    

    bool sameLocAvgFound(vector<int>& A, int idx, int gloSum, int locSum, int locCnt){
        if(idx==A.size()){
            if(locCnt!=0 and locCnt!=A.size()){
                return (gloSum*locCnt == locSum*A.size());
            }
            return false;
        }
        // not including current one
        if(sameLocAvgFound(A,idx+1,gloSum,locSum,locCnt)){
            return true;
        }
        
        if(sameLocAvgFound(A,idx+1,gloSum,locSum+A[idx],locCnt+1))
            return true;
        return false;
    }
};
