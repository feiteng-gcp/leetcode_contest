bool dp[16][300000+5];
class Solution {
public:
    bool splitArraySameAverage(vector<int>& A)
    {
        int len=A.size();
        int sum=0;
        for(int i=0;i<len;i++) sum+=A[i];
        double avg=sum/(double)len;

        for(int lb=1;lb<=len/2;lb++)
        {
            int sb=lb*avg;
            if(abs(((double)sb/lb)-avg)>0.00001) continue;
            memset(dp,false,sizeof dp);
            dp[0][0]=true;
            for(int i=0;i<len;i++)
            {
                for(int j=sb;j>=A[i];j--)
                {
                    for(int k=1;k<=len/2;k++)
                    {
                        dp[k][j]=(dp[k][j]||dp[k-1][j-A[i]]);
                    }
                }
            }
            if(dp[lb][sb])
            {
                return true;
            }
        }

        return false;
     }
};
