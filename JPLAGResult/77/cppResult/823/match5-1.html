class Solution {
public:
    bool splitArraySameAverage(vector<int>& A) {
        vector<vector<bool>> f(A.size() + 1, vector<bool>(A.size() * 10000 + 1, false));
        int sum = 0;
        for (const auto& iter : A) {
            sum += iter;
        }

        f[0][0] = true;
        for (int i = 0; i < A.size(); i++) {
            for (int j = i; j >= 0; j--) {
                for (int k = 0; k <= sum ;k++) {
                    if (!f[j][k]) { continue; }
                    int cur = k + A[i];
                    if (cur * (A.size() - j - 1) == (sum - cur) * (j + 1) && sum - cur != 0) { 
                        return true; 
                    }
                    f[j + 1][cur] = true;
                }
            }
        }
        return false;
    }
};
