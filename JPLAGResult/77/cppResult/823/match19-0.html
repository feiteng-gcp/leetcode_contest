class Solution {
public:
    int n, sumA;
    vector<int> Le, Re;
    vector<int> L[20], R[20];
    
    void build(const vector<int> &e, vector<int> Box[20]){
        int m = e.size();
        for (int S=(1<<m)-1; S>=0; S--){
            int sum=0, cnt=0;
            for (int i=0; i<m; i++) if ( (S>>i)&1 ){
                sum += e[i];
                cnt ++;
            }
            Box[cnt].push_back(sum);
        }
        for (int i=0; i<20; i++){
            sort( Box[i].begin(), Box[i].end() );
        }
    }
    
    bool splitArraySameAverage(vector<int>& A) {
        if (A.size()==1) return 0;
        n = A.size();
        sumA = 0;
        for (auto ele:A) sumA += ele;
        
        for (int i=0; i<n/2; i++) Le.push_back(A[i]);
        for (int i=n/2; i<n; i++) Re.push_back(A[i]);
        
        build(Le,L);
        build(Re,R);
        
        for (int i=0; i<20; i++){
            for (int j=0; i+j<n; j++){
                int m = i+j;
                if (m==0) continue;
                if ( sumA*m % n !=0 ) continue;
                int tar = sumA * m / n;
                
                for (int x:L[i]){
                    auto it = lower_bound( R[j].begin(), R[j].end(), tar-x );
                    if ( it != R[j].end() && (*it) == tar-x ) return 1;
                }
            }
        }
        return 0;
    }
};
