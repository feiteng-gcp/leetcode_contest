class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words)
    {
        set<string> s ;
        string mapx[26] = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."} ;

        for(string x : words)
        {
            string temp = "" ;
            for(int i = 0 ; i<x.size() ; i++)
            {
                temp += mapx[x[i]-'a'] ; 
            }
            s.insert(temp) ;
        }
        return s.size() ;
    }
};
