class Solution {
public:
  int uniqueMorseRepresentations(vector<string>& words) {
    unordered_set<string> codes;

    vector<string> map {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
    for (auto w: words) {
      string code;
      for (auto c: w) {
	code += map[c - 'a'];
      }

      codes.insert(code);
    }

    return codes.size();
  }
};
