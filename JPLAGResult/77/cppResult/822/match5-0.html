class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        vector<string> code = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        unordered_set<string> distinct;
        for(auto word : words) {
            string trans = "";
            for(auto c : word) {
                trans += code[c - 'a'];
            }
            distinct.insert(trans);
        }
        return distinct.size();
    }
};
