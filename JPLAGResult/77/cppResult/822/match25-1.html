class Solution {
public:
    vector<string> morse = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};

    int uniqueMorseRepresentations(vector<string>& words) {
         set<string> s;
        for(string w : words){
            string word;
            for(char c : w){
                word += morse[c - 'a'];
            }
            s.insert(word);
        }
        return s.size();
    }
};
