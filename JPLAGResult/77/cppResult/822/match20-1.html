class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {

        vector<string> code = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        unordered_map<string, int> m;
        for(auto w : words)
        {
            string trans;
            for(auto c : w)
            {
                trans += code[c - 'a'];
            }
            m[trans]++;
        }
        
        return m.size();
    }
};
