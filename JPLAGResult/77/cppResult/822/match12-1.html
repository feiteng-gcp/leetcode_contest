const string dicts[]={".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};

class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        set<string>s;

        for(auto& word:words) {
            string tmp="";
            for(auto& c:word) {
                tmp+=dicts[c-'a'];
            }
            s.insert(tmp);
        }
        return s.size();
    }
};
