class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        vector<string> mapping={".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."}; 
        
        unordered_set<string> pool; 
        for(auto w:words) {
            string tmp=""; 
            for(auto c:w) {
                int indx=(int)(c-'a');
                tmp.append(mapping[indx]); 
            }
            pool.insert(tmp); 
        }
        return (int)pool.size(); 

    }
};
