class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        
        vector<string> codes {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        
        unordered_map<string, int> m;
        
        for (auto w : words) {
            string code;
            for (auto ch : w) {
                code += codes[ch - 'a'];
            }
            m[code]++;
        }
        
        return m.size();
        
    }
};
