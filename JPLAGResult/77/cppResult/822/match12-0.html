class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        vector<string> charLib = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        unordered_set<string> s;
        for (auto word : words) {
            string curr = "";
            for (auto c : word) {
                curr += charLib[c - 'a'];
            }
            s.insert(curr);
        }
        return s.size();
        
    }
};
