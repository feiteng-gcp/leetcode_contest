class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        string morse[26]={".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        unordered_map<string, int> hash;
        int ans=0;
        int n=words.size();
        
        hash.clear();
        for (int i=0; i<n; i++) {
            int l=words[i].length();
            string cur;
            cur="";
            for (int j=0; j<l; j++) {
                cur=cur+morse[words[i][j]-'a'];
            }
            
            if (hash.find(cur)==hash.end()) {
                hash.insert(make_pair(cur, 1));
                ans++;
            }
        }
        
        return ans;
    }
};
