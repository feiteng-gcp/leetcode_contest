class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        vector<string> tableCode={".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        set<string> uniqueStr;
        for(int i=0;i<words.size();i++){
            string curCode="";
            for(int j=0;j<words[i].size();j++)
                curCode=curCode+tableCode[words[i][j]-'a'];
            uniqueStr.insert(curCode);
        }
        return (int) uniqueStr.size();
        
    }
};
