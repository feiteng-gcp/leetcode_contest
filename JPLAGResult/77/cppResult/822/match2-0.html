// [".-", "-...", "-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."]

class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        vector<string> dicts = {".-", "-...", "-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};

        unordered_set<string> set;
        for(auto & w : words)
        {
            string code;
            for(auto ch : w)
            {
                code += dicts[ch-'a'];
            }

            set.insert(code);
        }  

        return set.size();
    }
};
