class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        unordered_set<string> s;
        
        vector<string> codes = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        
        for (auto &w : words) {
            string t;
            for (auto c : w) {
                t += codes[c - 'a'];
            }
            s.insert(t);
        }
        
        return s.size();
    }
};
