class Solution {
public:
	int uniqueMorseRepresentations(vector<string>& words) {
		const string ref[26] = { ".-", "-...", "-.-.", "-..", ".", "..-.", "--.", "....", "..", ".---", "-.-", ".-..", "--", "-.", "---", ".--.", "--.-", ".-.", "...", "-", "..-", "...-", ".--", "-..-", "-.--", "--.." };
		set<string> ret;
		for (string s : words)
		{
			string x = "";
			for (char c : s)
				x += ref[c - 'a'];
			ret.insert(x);
		}
		return ret.size();
	}
};
