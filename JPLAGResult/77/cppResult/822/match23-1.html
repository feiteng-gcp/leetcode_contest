class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
    	set<string> ans;
    	string tmp;

    	vector<string> map = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
    	for (string w:words){
    		string tmp = "";
    		for (char ch:w)
    			tmp += map[ch-'a'];
    		ans.insert(tmp);
    	}
    	return ans.size();
    }
};
