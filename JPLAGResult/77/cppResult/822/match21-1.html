class Solution {
public:
    vector<string> tb={".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
    unordered_set<string> s;
    int uniqueMorseRepresentations(vector<string>& words) {
        for (int i=0;i<words.size();i++){
            string st;
            for (int j=0;j<words[i].size();j++){
                st=st+tb[words[i][j]-'a'];
            }
            s.insert(st);
        }
        return s.size();
    }
};
