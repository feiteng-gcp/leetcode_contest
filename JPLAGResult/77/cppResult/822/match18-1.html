class Solution {
public:
    string s[26]={".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
    int uniqueMorseRepresentations(vector<string>& words) {

        int n=words.size();
	set<string> ss;
	int ans=0;
	for(int i=0;i<n;i++)
	{
		int l=words[i].size();
		string cur="";
		for(int j=0;j<l;j++)
		{
			cur=cur+s[words[i][j]-'a'];
		}
		if(ss.find(cur)==ss.end())
		{
			ans++;
			ss.insert(cur);
		}
	}
	return ans;
    }
};
