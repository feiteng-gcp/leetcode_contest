class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) 
    {
        vector<string> m={".-","-...","-.-.","-..",".","..-.","--.","....","..",
                          ".---","-.-",".-..","--","-.","---",".--.","--.-",".-.",
                          "...","-","..-","...-",".--","-..-","-.--","--.."};
        
        unordered_set<string> us;
        
        for(auto word : words)
        {
            string tmp="";
            for(char c : word)
            {
                tmp+=m[c-'a'];
            }
            us.insert(tmp);
        }
        
        return us.size();
    }
};
