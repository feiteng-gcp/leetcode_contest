class Solution {
public:
    string str[26] = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
    int uniqueMorseRepresentations(vector<string>& words) {
        set<string> s;
        for (auto word : words) {
            string res = "";
            for (auto c : word) {
                res += str[c - 'a'];
            }
            s.insert(res);
        }
        return s.size();
    }
};
