class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        string morseCodes[26] = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        set<string> unique;
        for (auto word : words) {
            string repre = "";
            for (auto letter : word) {
                repre.append(morseCodes[letter - 'a']);
            }
            unique.insert(repre);
        }
        return (int)unique.size();
    }
};
