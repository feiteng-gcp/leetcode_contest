class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        vector<string> morse({".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."});
        
        set<string> ans_set;
        for(auto& word: words) {
            string tmp = "";
            for(auto c: word) {
                tmp += morse[c - 'a'];
            }
            ans_set.insert(tmp);
        }
        return ans_set.size();
    }
};
