class Solution {
    public boolean splitArraySameAverage(int[] A) {
        int n = A.length;
        if (n == 1) {
            return false;
        }


        boolean[][] dp = new boolean[n + 1][300001];
        boolean[][] old = new boolean[n + 1][300001];
 
        int sum = 0;
        for (int i = 0; i < n; i++) {
            
            sum += A[i];
        }

        
        dp[0][0] = true;
        int max = 0;
        for (int j = 0; j < n; j++) {
            //old = dp;
            boolean[][] t = old;
            old = dp;
            dp = t;
            
            max += A[j];
            for (int len = 1; len <= j + 1; len++) {
                for (int i = 0; i <= max; i++) {
                    if (old[len][i] || (i - A[j] >= 0 && old[len - 1][i - A[j]])) {
                        dp[len][i] = true;
                    } else {
                        dp[len][i] = false;
                    }
                }
            }
        }

        for (int len = 1; len < n; len++) {
            for (int i = 0; i < 300001; i++) {
                if (dp[len][i] && (n - len) * i == (sum - i) * len) {
                    //System.out.println("len = " + len + " i = " + i);
                    return true;
                }
            }
        }

        return false;

    }
}
