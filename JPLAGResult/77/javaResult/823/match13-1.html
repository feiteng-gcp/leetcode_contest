class Solution {
    
public boolean splitArraySameAverage(int[] A) {
        if(A.length < 2) return false;
        
        double sum = 0;
        for(int num : A) sum += num;
        double avg = sum / A.length;
        
        Map<Integer, Integer> map = new HashMap<>();
        int max = 0;
        for(int i = 1; i < A.length; i++){
            sum -= avg;
            if (Math.abs(sum - Math.floor(sum)) < 0.01 || Math.abs(sum - Math.ceil(sum)) < 0.01){
                map.put((int) sum, A.length - i);
                max = Math.max((int) sum, max);
            }
        }
        
        for(Map.Entry<Integer,Integer> e : map.entrySet()){
            System.out.println(e.getKey() + " " + e.getValue());
        }

        return dfs(A, map, 0, 0, 0, max);
    }
    
    private boolean dfs(int[] A, 
Map<Integer, Integer> map, int index, int cur, int count, int max){
        if(cur > max || index == A.length) return false;
        if(map.get(cur) != null && map.get(cur) == count) return true;
        for(int i = index; i < A.length; i++){
            boolean exist = dfs(A, map, i + 1, cur + A[i], count + 1, max);
            if(exist) return true;
        }
        return false;
    }
}
