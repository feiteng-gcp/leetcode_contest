class Solution {
    public boolean splitArraySameAverage(int[] A) {
        double avr=0;
        List<Set<Integer>> li=new ArrayList<Set<Integer>>();
        for (int i=0;i<A.length;i++)
        {
            avr=avr+A[i];
            li.add(new HashSet<Integer>());
        }
        li.add(new HashSet<Integer>());
        avr=avr/A.length;
        for (int i=0;i<A.length;i++)
        {
            if (A[i]==avr)
                return true;
            
for (int j=i-1;j>=0;j--)
            {
                Set<Integer> source=li.get(j);
                for (Integer x:source)
                {
                    int sum=x+A[i];
                    if (((((double)sum)/(j+2))==avr)&&(j+2!=A.length))
                    {
                        //System.out.println(sum+" "+(j+2));
                        return true;
                    }
                    li.get(j+1).add(sum);
                    //System.out.println(sum+" "+(j+2));
                }
            }
            li.get(0).add(A[i]);
            //System.out.println(A[i]+" 1");
        }
        return false;
    }
}
