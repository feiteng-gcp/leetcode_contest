public final class Solution {
	boolean search(int[] A, int total , int sum, int count, int index){
		if(count == A.length)
			return false;
		if(total * count == sum * A.length)
			return true;
		if(index < 0)
			return false;
		return search(A, total, sum, count, index-1) || search(A, total, sum+A[index], count+1, index-1);
	}
	public boolean splitArraySameAverage(int[] A) {
		if(A.length<=1)
			return false;
        int sum  = 0;
        for(int v : A){
        	sum += v;
        }
        return search(A, sum, A[A.length-1],1,A.length-2);
    }
	
}
