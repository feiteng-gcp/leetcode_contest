class Solution {
    public boolean splitArraySameAverage(int[] A) {
        int sum = 0;
        int n = A.length;
        for (int i = 0; i < n; i++) sum += A[i];
        
        Arrays.sort(A);
        
        for (int i = 1; i <= n - i; i++) {
            if (sum * i % n == 0 && helper(A, i, sum * i / n))
                return true;
        }
        return false;
    }
    
    // whether there are k elem in A, whose sum == t
    private boolean helper(int[] A, int k, int t) {
        int n = A.length;
        boolean[][][] dp = new boolean[n][k+1][t+1];
        
        dp[0][1][A[0]] = true;
        dp[0][0][0] = true;
        for (int i = 1; i < n; i++) {
            // add current element A[i]
            dp[i][0][0] = true;
            for (int j = 1; j <= k; j++) {
                for (int s = 0; s <= t; s++) {
                    if(dp[i-1][j][s] || 
                       s >= A[i] && dp[i-1][j-1][s - A[i]])
                        dp[i][j][s] = true;                    
                }
            }
        }
        return dp[n-1][k][t];
    }
}
