class Solution {
    public boolean splitArraySameAverage(int[] A) {
        Arrays.sort(A);
        int sum = 0;
        for (int i = 0; i < A.length; i++) {
            sum += A[i];
        }
        
for (int i = 1; i <= A.length / 2; i++) {
            if (sum * i % A.length != 0) continue; // can't compose out of 
            if (find(A, 0, i, sum * i, A.length)) return true;
        }
        return false;
    }
    
    private boolean find(int[] A, int index, int count, int sum, int factor) {
        if (index >= A.length) return false;
        if (count == 1) {
            // todo binary search
            for (int i = index; i < A.length; i++) {
                int c = A[i] * factor;
                if (c == sum) return true;
                else if (c > sum) return false; // it's sorted, can't find
            }
            return false;
        } else {
            for (int i = index; i < A.length; i++) {
                int rem = sum - A[i] * factor;
                if (rem < 0) return false;
                if (find(A, i + 1, count - 1, rem, factor)) return true;
            }
            return false;
        }
    }
}
