class Solution {
    public boolean splitArraySameAverage(int[] A) {
        if (A.length == 1) return false;
        if (A.length == 2) return A[0] == A[1];
        Arrays.sort(A);
        int sum = 0;
        for (int n : A) sum += n;
        for (int i = 1; i < A.length; i++) {
            int target = sum * i / A.length;
            if (target * A.length == sum * i) {
                if (helper(A, target, 0, i)) return true;
            }
        }
        return false;
    }
    
    private boolean helper(int[] A, int target, int start, int count) {
        if (count == 0) return false;
        if (target <= 0) return target == 0;
        for (int i = start; i < A.length; i++) {
            if (A[i] == target) return count == 1;
            if (A[i] > target) return false;
            if (helper(A, target - A[i], i + 1, count - 1)) return true;
        }
        return false;
    }
}
