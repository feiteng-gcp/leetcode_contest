class Solution {
    public boolean splitArraySameAverage(int[] A) {
        int n = A.length;
        int sum = 0;
        for (int a: A) sum += a;
        
        Arrays.sort(A);
        
        for (int i = 1; i <= (n>>1); ++i) {
            if ((i*sum)%n != 0) continue;
            int sub = i*sum/n;
            if (
subSum(A, i, sub, 0)) return true;
        }
        return false;
    }
    
    private boolean subSum(int[] A, int n, int sum, int begin) {
        if (n == 0 && sum == 0) return true;
        else if (n == 0 || sum == 0 || begin >= A.length || sum < A[begin]) return false;
        
        return subSum(A, n-1, sum-A[begin], begin+1) || subSum(A, n, sum, begin+1);
    }
}
