class Solution {
    public boolean splitArraySameAverage(int[] A) {
        final int N = A.length;
        
        int total = 0;
        for (int v : A) {
            total += v;
        }
        
        // boolean hasZero = false;
        // for (int v : A) {

        //             System.out.format("i = %d, sum = %d, otherCnt = %d, otherSum = %d\n", i, sum, otherCnt, otherSum);
        //             return true;
        //         }
        //     }
        // }
        // return false;
        
        final int K = N >> 1;
        
        IntSet[] ans = new IntSet[K + 1];
        for (int i = 0; i <= K; ++i) {
            ans[i] = new IntSet();
        }
        ans[0].add(0);
        for (int i = 0; i < N; ++i) {
            for (int j = K; j > 0; --j) {
                IntSet next = new IntSet();
                for (int v : ans[j - 1]) {
                    next.add(v + A[i]);
                }
                ans[j].addAll(next);
            }
        }
        
        for (int i = 1; i <= K; ++i) {
            for (int sum : ans[i]) {
                long otherSum = total - sum;
                long otherCnt = N - i;
                if (sum * otherCnt == otherSum * i) {
                    return true;
                }
            }
        }
        return false;
    }
