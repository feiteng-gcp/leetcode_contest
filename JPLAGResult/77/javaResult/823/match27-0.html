class Solution {
    public boolean splitArraySameAverage(int[] A) {
        int n = A.length, possibleMaxSum = (n/2) * 10000, m = possibleMaxSum;
        boolean[][] canAchieve = new boolean[n/2][m];
        for(int i=0;i<n/2;i++) Arrays.fill(canAchieve[i], false);
        for(int i=0;i<n;i++) {
            
            int maxNumberOfGroup = Math.min(i+1, n/2);
            for(int j=maxNumberOfGroup-2;j>=0;j--) {
                int tmp = (j+1) * 10000;
                for(int v=0;v<tmp;v++) {
                    if(canAchieve[j][v]) {
                        canAchieve[j+1][v+A[i]] = true;
                    }
                }
            }
            canAchieve[0][A[i]] = true;
        }
        
        int sum = 0;
        for(int i : A) sum += i;
        
        for(int i=0;i<n/2;i++) {
            if((sum*(i+1))%n != 0) {
                continue;
            }
            int total = sum*(i+1) / n;
            if(canAchieve[i][total]) {
                // System.out.println("" + (i+1) + " numbers can have sum " + total);
                return true;
            }
        }
        return false;
    }
}
