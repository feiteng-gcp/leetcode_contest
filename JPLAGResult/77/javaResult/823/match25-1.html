class Solution {
    public boolean splitArraySameAverage(int[] A) {
        // ave(B) == ave(C) ? then == ave(A);
        
        // if there is a sub(A) ave == ave(A)? subA > 0 and < lenA
        // if backtrack pickup, complexity is high 2^n
        
        // can we sort A (nlogn)?
        // after sort there is a N^2 algorithm?
        // could be faster??
        
        int len = A.length;
        if (len < 2) return false;
        
        int nave = 0;
        for (int i = 0; i < len; ++i) {
            nave += A[i];
            A[i] *= len; // extend to use ave as int
        }
        
        List<Integer> negList = new ArrayList<>();
        List<Integer> posList = new ArrayList<>();
        
        for (int i = 0; i < len; ++i) {
            int v = A[i] - nave;
            if (v == 0) return true;
            if (v > 0) posList.add(v);
            else negList.add(v);
        }
        if (posList.isEmpty() || negList.isEmpty()) return false;
        
        System.out.println(negList);
        System.out.println(posList);

        
        Set<Integer> negSet = calculateSet(negList);
        Set<Integer> posSet = calculateSet(posList);
        
        System.out.println(negSet);
        System.out.println(posSet);

        for (Integer neg : negSet) {
            
if (posSet.contains(-neg)) return true;
        }
        return false;
    }
    
    Set<Integer> calculateSet(List<Integer> negList) {
        Set<Integer> negSet = new HashSet<>();
        int sum = 0;
        for (Integer neg : negList) {
            sum += neg;
            Set<Integer> temp = new HashSet<>();
            for (Integer curr : negSet) {
                temp.add(curr + neg);
            }
            negSet.addAll(temp);
            negSet.add(neg);
        }
        negSet.remove(sum);
        return negSet;
    }
}
