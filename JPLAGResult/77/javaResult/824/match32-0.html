class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int lines = 1, last = 0;
        for (char c: S.toCharArray()) {
            if (last + widths[c - 'a'] > 100) {
                lines++;
                last = widths[c - 'a'];
            } else {
                last += widths[c - 'a'];
            }
        }
        return new int[]{lines, last};
    }
}
