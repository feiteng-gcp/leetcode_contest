class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int L = 100;
        int n = S.length();
        int nline = 0;
        int rem = 0;
        for (int i = 0; i < S.length(); i++) {
            int w = widths[S.charAt(i) - 'a'];
            if (rem >= w) {
                rem -= w;
            }else {
                // start a new line
                nline++;
                rem = L - w;
            }
        }
        return new int[]{nline, L - rem};
        
    }
}
