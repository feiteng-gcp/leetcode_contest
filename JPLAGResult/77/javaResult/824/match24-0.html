class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        char[] array = S.toCharArray();
        int[] result =  new int[2];
        int lineCount = 1;
        int curWidth = 0;
        for (char c : array) {
            if (curWidth + widths[c - 'a'] > 100) {
                lineCount++;
                curWidth = widths[c - 'a'];
            } else {
                curWidth += widths[c - 'a'];
            }
        }
        result[0] = lineCount;
        result[1] = curWidth;
        return result;
    }
}
