class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        if(S==null||S.length()==0){
            return new int[2];
        }
        int countLine=1;
        int lastLine=0;
        for(int i=0;i<S.length();i++){
            int len=widths[S.charAt(i)-'a'];
            if(lastLine+len>100){
                countLine++;
                lastLine=len;
            
}
            else{
                lastLine=lastLine+len;
            }
        }
        return new int[]{countLine,lastLine};
    }
}
