class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int l = 0;
        int cur = 0;
        for(int i= 0; i<S.length(); i++){
            int pos = S.charAt(i)-'a';
            if(widths[pos] > cur){
                l++;
                cur = 100;
                cur -= widths[pos];
            }
            else{
                cur -= widths[pos];
            }
        }
        return new int[]{l,100-cur};
    }
}
