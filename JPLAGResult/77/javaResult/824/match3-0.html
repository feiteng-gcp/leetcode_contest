class Solution {

	
	public int[] numberOfLines(int[] widths, String S) {
		
        if (S.isEmpty()) {
            return new int[] { };
        }
		int line = 1 , last = 0;
		for (int i = 0;i < S.length();i ++) {
			int index = S.charAt(i) - 'a';
			if (last + widths[index] <= 100) {
				last += widths[index];
			} else {
				line ++;
				last = widths[index];
			}
		}
	    return new int[] {line , last};
	
	}
	
}
