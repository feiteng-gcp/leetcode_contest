class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int lines = 1, curWidth = 0;
        char[] c = S.toCharArray();
        
        for(int i = 0; i < c.length; i++) {
            int index = c[i] - 'a';
            if(curWidth + widths[index] > 100) {
                lines++;
                curWidth = widths[index];
            }
            else {
                curWidth += widths[index];
            
}
        }
        
        return new int[]{lines, curWidth};
    }
}
