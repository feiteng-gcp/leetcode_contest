class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int count = 1;
        int left = 100;
        for(char c : S.toCharArray()){
            if(left >= widths[c - 'a']){
                left -= widths[c - 'a'];
            } else {
                count++;
                left = 100 - widths[c - 'a'];
            }
        }
        return new int[]{count, 100 - left};
    }
}
