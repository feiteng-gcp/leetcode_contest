class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int used=0;
        int lines=1;
        for (int i=0;i<S.length();i++)
        {
            int willbe=used+widths[(int)(S.charAt(i)-'a')];
            if (willbe>100)
            {
                used=widths[(int)(S.charAt(i)-'a')];
                lines++;
            }
            else
                used=willbe;
        }
        int[] r=new int[2];
        r[0]=lines;
        r[1]=used;
        return r;
    }
}
