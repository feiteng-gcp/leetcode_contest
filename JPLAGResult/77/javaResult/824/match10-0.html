class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        if (S == null || S.length() < 1) return new int[]{0, 0};
        int lineCount = 1, widthCount = 0;
        for (char c : S.toCharArray()) {
            int w = widths[c - 'a'];
            if (widthCount + w > 100) {
                lineCount++;
                widthCount = 0;
            }
            widthCount += w;
        }
        return new int[]{lineCount, widthCount};
    }
}
