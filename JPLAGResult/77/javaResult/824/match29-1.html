class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int[] ans = new int[2];
        for(int i = 0; i < S.length(); i++){
            char c = S.charAt(i);
            
int width = widths[c - 'a'];
            if(ans[1] + width > 100){
                ans[0]++;
                ans[1] = width;
            }else{
                ans[1] += width;
            }
        }
        if(ans[1] > 0){
            ans[0]++;
        }
        return ans;
    }
}
