class Solution {
    public int[] numberOfLines(int[] widths, String S) {
        int n = 0, rest = 0;
        for (char c: S.toCharArray()) {
            int w = widths[c - 'a'];
            if (w <= rest) {
                rest -= w;
            } else {
                rest = 100 - w;
                n++;
            
}
        }
        return new int[]{n, 100 - rest};
    }
}
