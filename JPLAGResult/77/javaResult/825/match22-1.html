	class Solution {
	    public int maxIncreaseKeepingSkyline(int[][] a) {
	        int n = a.length, m = a[0].length;
	        int[] mr = new int[n];
	        for(int i = 0;i < n;i++){
	        	int max = 0;
	        	for(int j = 0;j < m;j++){
	        		max = Math.max(max, a[i][j]);
	        	}
	        	mr[i] = max;
	        }
	        int[] mc = new int[m];
	        for(int i = 0;i < m;i++){
	        	int max = 0;
	        	for(int j = 0;j < n;j++){
	        		max = Math.max(max, a[j][i]);
	        	}
	        	mc[i] = max;
	        }

	        int ans = 0;
	        for(int i = 0;i < n;i++){
	        	for(int j = 0;j < m;j++){
	        		ans += Math.min(mr[i], mc[j]
) - a[i][j];
	        	}
	        }
	        return ans;
	    }
	}
