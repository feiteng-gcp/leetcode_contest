class Solution {

	private int[] rowMax;
	private int[] colMax;
	
    public int maxIncreaseKeepingSkyline(int[][] grid) {
        
    	int i , j , n = grid.length , m = grid[0].length;
    	rowMax = new int[n];
    	colMax = new int[m];
    	for (i = 0;i < n;i ++) {
    		int max = - 1;
    		for (j = 0;j < m;j ++) {
    			if (grid[i][j] > max) {
    				max = grid[i][j];
    			}
    		}
    		rowMax[i] = max;
    	}
    	for (j = 0;j < m;j ++) {
    		int max = - 1;
    		for (i = 0;i < n;i ++) {
    			if (grid[i][j] > max) {
    				max = grid[i][j];
    			}
    		}
    		colMax[j] = max;
    	}
    	int ans = 0;
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < m;j ++) {
    			int temp = Math.min(rowMax[i
] , colMax[j]);
    			ans += temp - grid[i][j];
    		}
    	}
    	return ans;
    	
    }
	
}
