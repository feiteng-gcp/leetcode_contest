class Solution {
    public int maxIncreaseKeepingSkyline(int[][] grid) {
        if(grid == null || grid.length == 0) return 0;
        
int m = grid.length;
        int n = grid[0].length;
        int[] topBottom = new int[n];
        int[] leftRight = new int[m];
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                int num = grid[i][j];
                topBottom[j] = Math.max(topBottom[j], num);
                leftRight[i] = Math.max(leftRight[i], num);
            }
        }
        int ans = 0;
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                int num = grid[i][j];
                int max = Math.min(topBottom[j], leftRight[i]);
                ans += 
Math.max(0, max - num);
            }
        }
        return ans;
    }
}
