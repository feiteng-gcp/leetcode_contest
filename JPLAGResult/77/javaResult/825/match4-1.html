class Solution {
    public int maxIncreaseKeepingSkyline(int[][] grid) {
        
int r = grid.length, c = grid[0].length, res = 0;
        int[] left = new int[r], top = new int[c];
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                left[i] = Math.max(left[i], grid[i][j]);
                top[j] = Math.max(top[j], grid[i][j]);
            }
        }
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                res += Math.min(left[i], top[j]) - grid[i][j];
            }
        }
        return res;
    }
}
