class Solution {
    public int maxIncreaseKeepingSkyline(int[][] grid) {
        // increase to the min of (max_row, max_col)
        int m = grid.length, n = grid[0].length;
        int sum = 0;
        int[] top = new int[n];
        int[] left = new int[m];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] > left[i])
                    left[i] = grid[i][j];
                if (grid[i][j] > top[j])
                    top[j] = grid[i][j];
            }
        }
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                sum += Math.min(left[i], top[j]) - grid[i][j];
            }
        }
        
        return sum;
    }
}
