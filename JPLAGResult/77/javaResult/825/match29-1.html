class Solution {
    public int maxIncreaseKeepingSkyline(int[][] grid) {
        int[] skyx=new int[grid[0].length];
        int[] skyy=new int[grid.length];
        for (int y=0;y<grid.length;y++)
            for (int x=0;x<grid[0].length;x++)
            {
                if (grid[y][x]>skyx[x])
                    skyx[x]=grid[y][x];
                if (grid[y][x]>skyy[y])
                    skyy[y]=grid[y][x];
            }
        int incs=0;
        for (int y=0;y<grid.length;y++)
            for (int x=0;x<grid[0].length;x++)
            {
                if (skyx[x]<skyy[y])
                    incs=incs+skyx[x]-grid[y][x];
                else
                    incs=incs+skyy[y]-grid[y][x];
            }
        return incs;
    }
}
