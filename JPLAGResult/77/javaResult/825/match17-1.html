class Solution {
    public int maxIncreaseKeepingSkyline(int[][] grid) {
        
int n = grid.length;
        int[] top = new int[n], left = new int[n];
        for (int r = 0; r < n; r++) {
            for (int c = 0; c < n; c++) {
                top[c] = Math.max(top[c], grid[r][c]);
                left[r] = Math.max(left[r], grid[r][c]);
            }
        }
        int increment = 0;
        for (int r = 0; r < n; r++) {
            for (int c = 0; c < n; c++) {
                int h = Math.min(top[c], left[r]) - grid[r][c];
                if (h > 0) increment+= h;
            }
        }
        return increment;
    }
}
