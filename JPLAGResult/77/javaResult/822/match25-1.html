class Solution {
    public int uniqueMorseRepresentations(String[] words) {
       String[] codes=new String[]{".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        HashSet<String> set=new HashSet<>();
       for(String w:words){
           StringBuilder sb=new StringBuilder();
            for(char ch:w.toCharArray()){
                sb.append(codes[ch-'a']);
            
}
            set.add(sb.toString());
       }
       //
        return  set.size();

    }
}
