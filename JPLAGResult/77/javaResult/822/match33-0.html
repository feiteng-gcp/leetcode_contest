class Solution {
    public int uniqueMorseRepresentations(String[] words) {
        String[] replace = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        HashSet<String> hs = new HashSet<>();
        for(int i = 0; i < words.length; i++){
            String a = "";
            for(int j = 0; j < words[i].length(); j++){
                a += replace[words[i].charAt(j) - 'a'];
            }
            hs.add(a);
        }
        return hs.size();
    }
}
