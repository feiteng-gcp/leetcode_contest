class Solution {
    public int uniqueMorseRepresentations(String[] words) {
                String[] dict = new String[]{".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        Set<String> set = new HashSet<>();
        for(String w :words) {
        	String key = "";
        	for(char c : w.toCharArray()) {
        		key += dict[c - 'a'];
        	}
        	set.add(key);
        }
        return set.size();
    }
}
