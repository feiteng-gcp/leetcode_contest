class Solution {
    public int uniqueMorseRepresentations(String[] words) {
        // can use bit to represent
        
        // or directly use string set
        
        String[] code = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        
        Set<String> count = new HashSet<>();
        
        for (String word : words) {
            String trans = "";
            for (char c : word.toCharArray()){
                trans += code[c - 'a'];
            }
            count.add(trans);
        }

        return count.size();
    }
}
