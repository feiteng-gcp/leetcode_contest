class Solution {
    public int uniqueMorseRepresentations(String[] words) {
        String[] tab = new String[]{".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        TreeSet<String> s = new TreeSet<String>();
        for(String w:words){
            String m = "";
            for(int i=0; i<w.length(); i++){
                m += tab[w.charAt(i)-'a'];
            
}
            s.add(m);
        }
        return s.size();
    }
}
