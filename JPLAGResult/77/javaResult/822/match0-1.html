class Solution {
    public int uniqueMorseRepresentations(String[] words) {
        String[] codes = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        Set<String> set = new HashSet<>();
        for (String w : words) {
            String code = "";
            for (char c : w.toCharArray()) {
                code += codes[c - 'a'];
            }
            set.add(code);
        }
        return set.size();
    }
}
