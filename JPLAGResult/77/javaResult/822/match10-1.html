class Solution {
    public int uniqueMorseRepresentations(String[] words) {
       String[] table = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        HashSet<String> set = new HashSet<>();
        for (String s: words) {
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < s.length(); i++) {
                sb.append(table[s.charAt(i)- 'a']);
            }
            set.add(sb.toString());
        }
        return set.size();
    }
}
