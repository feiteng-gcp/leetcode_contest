class Solution:
    def uniqueMorseRepresentations(self, words):
        """
        :type words: List[str]
        :rtype: int
        """
        morses = [".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."]
        morse_map = dict()
        for i, morse in enumerate(morses):
            character = chr(ord('a') + i)
            morse_map[character] = morse
        
        transformations = set()
        
        
for word in words:
            result = ""
            for character in word:
                result += morse_map[character]
            transformations.add(result)
        
        return len(transformations)
