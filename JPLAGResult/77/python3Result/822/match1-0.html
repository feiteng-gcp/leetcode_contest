import string
class Solution:
    def uniqueMorseRepresentations(self, words):
        """
        :type words: List[str]
        :rtype: int
        """
        string_set = set()
        morse = [".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."]
        letters = string.ascii_lowercase
        morse_dict = dict()
        for i in range(26):
            morse_dict[letters[i]] = morse[i]
        for subword in words:
            tmp = ""
            for i in subword:
                tmp += morse_dict[i]
            string_set.add(tmp)
        return len(string_set)
