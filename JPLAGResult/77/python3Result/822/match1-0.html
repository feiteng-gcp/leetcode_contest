class Solution:
    def uniqueMorseRepresentations(self, words):
        """
        :type words: List[str]
        :rtype: int
        """
        if not words:
            return 0
        
        morris = [".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."]
        
        cache = set()
        
        for word in words:
            code = []
            for char in word:
                t = ord(char) - ord('a')
                code += [morris[t]]
            
            cache.add(''.join(code))
            
        return len(cache)
