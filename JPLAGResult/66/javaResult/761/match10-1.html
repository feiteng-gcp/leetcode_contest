/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        
List<Interval> intervals = new ArrayList<>();
        for (List<Interval> its : avails) {
            for (Interval i : its) {
                intervals.add(i);
            }
        }
        
        List<Interval> occupy = merge(intervals);
        return free(occupy);
    }
    
    public List<Interval> merge(List<Interval> intervals) {
        if (intervals == null || intervals.size() == 0) {
            return intervals;
        }
        
        List<Interval> res = new ArrayList<>();
        
        
Collections.sort(intervals, new Comparator<Interval>(){
            public int compare(Interval a, Interval b) {
                return a.start - b.start;
            }
        });
        
        Interval prev = null;
        for (Interval interval : intervals) {
            if (prev == null || interval.start > prev.end) {
                res.add(interval);
                prev = interval;
            
} else {
                prev.end = Math.max(prev.end, interval.end);
            }
        }
        
        return res;
    }
    
    public List<Interval> free(List<Interval> intervals) {
        List<Interval> res = new ArrayList<>();
        
for (int i = 0; i < intervals.size() - 1; i++) {
            res.add(new Interval(intervals.get(i).end, intervals.get(i+1).start));
        }
        return res;
    }
}
