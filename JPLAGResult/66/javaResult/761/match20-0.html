/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Integer> start = new ArrayList<>(), end = new ArrayList<>();
            List<Interval> res = new ArrayList<>();
            for (List<Interval> person : avails) {
                for (Interval p : person) {
                    start.add(p.start);
                    end.add(p.end);
                }
            }
            Collections.sort(start);
            Collections.sort(end);
            List<Interval> in = new ArrayList<>();
            for (int i = 0, j = 0; i < start.size(); ++i) {
                if (i == start.size() - 1 || start.get(i + 1) > end.get(i)) {
                    in.add(new Interval(start.get(j), end.get(i)));
                    j = i + 1;
                }
            }
            for (int i = 1; i < in.size(); ++i) {
                res.add(new Interval(in.get(i - 1).end, in.get(i).start));
            }
            return res;
    }
}
