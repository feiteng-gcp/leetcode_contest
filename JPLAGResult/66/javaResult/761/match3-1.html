/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */

class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        Set<Integer> set = new HashSet<Integer>();
        for(List<Interval> intervals : avails){
            for(Interval interval: intervals){
                set.add(interval.start);
                set.add(interval.end);
            }
        }
        List<Integer> list = new ArrayList<>(set);
        Collections.sort(list);
        Map<Integer, Integer> map = new HashMap<>();
        for(int i = 0; i<list.size(); i++){
            map.put(list.get(i), i);
        }
        boolean[] mark = new boolean[list.size()-1];
        for(List<Interval> intervals : avails){
            for(Interval interval: intervals){
                int start = map.get(interval.start);
                int end = map.get(interval.end);
                for(int i =start; i<end; i++){
                    mark[i] = true;
                }
            }
        }
        List<Interval> result = new ArrayList<>();
        boolean isLastIntervalBusy = true;
        int beginFree = 0;
        for(int i = 0; i< mark.length; i++){
            if(isLastIntervalBusy&&!mark[i]){
                beginFree = i;
                isLastIntervalBusy = false;
            } else if(!isLastIntervalBusy&&mark[i]){
                Interval interval = new Interval(list.get(beginFree), list.get(i));
                result.add(interval);
                isLastIntervalBusy = true;
            }
        }
        if(!isLastIntervalBusy){
            Interval interval = new Interval(list.get(beginFree), list.get(list.size()-1));
        }
        return result;
    }
}
