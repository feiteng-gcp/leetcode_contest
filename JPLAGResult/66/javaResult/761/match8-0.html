/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        TreeMap<Integer, Integer> sche = new TreeMap<>();
        
        for(List<Interval> employee: avails){
            for(Interval slot: employee){
                sche.put(slot.start, sche.getOrDefault(slot.start, 0)+1);
                sche.put(slot.end, sche.getOrDefault(slot.end, 0)-1);
            }
        }
        
        List<Interval> res = new ArrayList<>();
        int busy = 0;
        int start = Integer.MIN_VALUE;
        for(int key: sche.keySet()){
            int update = sche.get(key);
            if(busy == 0 && update >0 && start != Integer.MIN_VALUE){
                res.add(new Interval(start, key));
            }
            
            busy += update;
            if(busy == 0 && update <0){
                start = key;    
            }
        }
        
        
        return res;
    }
}
