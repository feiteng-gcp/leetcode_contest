/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> list = new ArrayList<>();
        for(List<Interval> tmp : avails){
            list.addAll(tmp);
        }
        
        
        Collections.sort(list, (a , b)->{
            if(a.start != b.start)
                return a.start - b.start;
            else
                return b.end - a.end;
        });
        
        LinkedList<Interval> merge = new LinkedList<>();
        for(Interval inter : list){
            if(merge.size() == 0 || inter.start > merge.getLast().end){
                merge.add(inter);
            }else{
                merge.getLast().end = Math.max(merge.getLast().end, inter.end);
            }
        }
        
        List<Interval> res = new LinkedList<Interval>();
        for(int i = 1; i < merge.size(); i ++){
            Interval tmp = new Interval(merge.get(i - 1).end, merge.get(i).start);
            res.add(tmp);
        }
        
        return res;
    }
}
