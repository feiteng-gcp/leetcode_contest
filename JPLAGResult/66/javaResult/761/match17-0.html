/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<int[]> list = new ArrayList<>();
        for (int i = 0; i < avails.size(); i++) {
            for (Interval interval: avails.get(i)) {
                list.add(new int[] {interval.start, interval.end});
            }
        }
        
        list.sort(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        
        List<int[]> listMerge = merge(list);
        List<Interval> res = new ArrayList<>();
        for (int i = 1; i < listMerge.size(); i++) {
            res.add(new Interval(listMerge.get(i - 1)[1], listMerge.get(i)[0]));
        }
        return res;
    }
    
    private List<int[]> merge(List<int[]> list) {
        List<int[]> res = new ArrayList<>();
        int left = list.get(0)[0];
        int right = list.get(0)[1];
        int i = 1;
        while (i < list.size()) {
            if (list.get(i)[0] <= right) {
                right = Math.max(right, list.get(i)[1]);
            } else {
                System.out.println(left + " " + right);
                res.add(new int[] {left, right});
                left = list.get(i)[0];
                right = list.get(i)[1];
            }
            i++;
        }
        if (i == list.size()) {
            res.add(new int[] {left, right});
        }
        return res;
    }
}
