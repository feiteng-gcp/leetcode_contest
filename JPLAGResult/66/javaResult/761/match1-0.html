/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> inter = new ArrayList<>();
        for (List<Interval> in: avails) {
            for (int i = 0; i < in.size(); i++) {
                inter.add(in.get(i));
            }
        }
        Collections.sort(inter, new Comparator<Interval>() {
            public int compare(Interval i1, Interval i2) {
                return i1.start - i2.start;
            }
        });
        List<Interval> merge = new ArrayList<>();
        for (int i = 0; i < inter.size();) {
            int start = inter.get(i).start;
            int end = inter.get(i).end;
            i++;
            while (i < inter.size() && end >= inter.get(i).start) {
                end = Math.max(end, inter.get(i).end);
                i++;
            }
            merge.add(new Interval(start, end));
        }
        List<Interval> res = new ArrayList<>();
        if (merge.size() == 0) {
            return res;
        }
        
        int pre = merge.get(0).end;
        for (int i = 1; i < merge.size(); i++) {
            // System.out.println(merge.get(i).start + " " + merge.get(i).end);
            res.add(new Interval(pre, merge.get(i).start));
            pre = merge.get(i).end;
        }
        return res;
    }
}
