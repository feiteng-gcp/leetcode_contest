/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        HashSet<Integer> point = new HashSet();
        for (int i = 0; i < avails.size(); ++i)
            for (int j = 0; j < avails.get(i).size(); ++j) 
{
                point.add(avails.get(i).get(j).start);
                point.add(avails.get(i).get(j).end);
            }
        ArrayList<Integer> points = new ArrayList(point);
        Collections.sort(points);
        int[] index = new int[avails.size()];
        
List<Interval> result = new ArrayList();
        for (int i = 0; i < points.size() - 1; ++i) {
            int start = points.get(i);
            int end = points.get(i + 1);
            int j = 0;
            for (;j < avails.size(); ++j) {
                if (index[j] == avails.get(j).size()) continue;
                Interval current = avails.get(j).get(index[j]);
                if (current.end <= start) {
                    index[j]++;
                    --j;
                } else if (current.start < end) {
                    break;
                }
            }
            if (j == avails.size()) {
                result.add(new Interval(start, end));
            }
        }
        if (result.size() == 0) return result;
        List<Interval> f = new ArrayList();
        int s = result.get(0).start;
        int e = result.get(0).end;
        for (int i = 1; i < result.size(); ++i) {
            if (result.get(i).start != e) {
                f.add(new Interval(s, e));
                s = result.get(i).start;
            }
            e = result.get(i).end;
        }
        f.add(new Interval(s, e));
        return f;
    }
}
