/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> result = new ArrayList<>();
        for(List<Interval> list:avails){
            result.addAll(list);
        }
        Collections.sort(result,new Comparator<Interval>(){
           public int compare(Interval f,Interval s){
               return f.start==s.start?f.end-s.end:f.start-s.start;
           } 
        });
        Interval ref = new Interval();
        List<Interval> new_result = new ArrayList<>();
        for(int i=0;i<result.size();i++){
            if(i == 0){
                ref = result.get(i);
            }else{
                if(result.get(i).start > ref.end){
                    new_result.add(new Interval(ref.end,result.get(i).start));
                    ref = result.get(i);
                }
                else if(result.get(i).start <= ref.end){
                    ref.end = Math.max(ref.end,result.get(i).end);
                }
               
            }
        }
       // new_result.add(ref);
        return new_result;
    }
}
