/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> all = new ArrayList<>();
        List<Interval> res = new ArrayList<>();
        avails.forEach(x -> all.addAll(x));
        int curS = Integer.MIN_VALUE, curE = Integer.MIN_VALUE;
        all.sort(Comparator.comparingInt(o -> o.start));
        for (int i = 0, len = all.size(); i < len; i++) {
            int e = all.get(i).end;
            int s = all.get(i).start;
            if (s > curE) {
                if (curE != Integer.MIN_VALUE)
                    res.add(new Interval(curE, s));
                curS = s;
                curE = e;
            } else {
                curE = Integer.max(curE, e);
            }
        }
        return res;
    }
}
