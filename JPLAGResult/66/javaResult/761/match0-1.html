/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */

class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> res = new LinkedList<>();
        List<Interval> list = new ArrayList<>();
        // int min = Integer.MAX_VALUE;
        for (int i = 0; i < avails.size(); i++) {
            for (int j = 0; j < avails.get(i).size(); j++) {
                // if (j == 0) min = Math.min(min, avails.get(i).get(j).start);
                list.add(avails.get(i).get(j));
            }
        }
        Collections.sort(list, (l1, l2) -> (l1.start - l2.start));
        for (int i = 0; i < list.size(); i++) {
            if (i + 1 < list.size() && list.get(i + 1).start <= list.get(i).end) {
                list.get(i).end = Math.max(list.get(i).end, list.get(i + 1).end);
                list.remove(i + 1);
                i--;
            }
        }
        for (int i = 1; i < list.size(); i++) {
            res.add(new Interval(list.get(i - 1).end, list.get(i).start));
        }
        
        return res;
    }
}
