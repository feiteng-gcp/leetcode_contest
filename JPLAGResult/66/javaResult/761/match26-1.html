/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */

class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> sortList = new ArrayList<>();
        for(List<Interval> list : avails) {
            for(Interval interval : list) {
                sortList.add(interval);
            }
        }
        sortList.sort(new Comparator<Interval>() {
            @Override
            public int compare(Interval o1, Interval o2) {
                return o1.start - o2.start;
            }
        });
        int start = sortList.get(0).start;
        int end = sortList.get(0).end;
        List<Interval> ans = new ArrayList<>();
        for(Interval interval : sortList) {
            if(interval.start > end) {
                ans.add(new Interval(end,interval.start));
                start = interval.start;
                end = interval.end;
            } else if(interval.end > end) {
                end = interval.end;
            
}
        }
        return ans;
    }
}
