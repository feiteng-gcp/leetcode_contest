/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> res = new ArrayList<>();
        if (avails == null || avails.size() == 0) {
            return res;
        }
        
        TreeMap<Integer, Integer> cache = new TreeMap<>();
        for (List<Interval> list : avails) {
            for (Interval i : list) {
                int s = i.start;
                int e = i.end;
                cache.put(s, cache.getOrDefault(s, 0) + 1);
                cache.put(e, cache.getOrDefault(e, 0) - 1);
            }
        }
        
        int cnt = 0;
        for (Integer s : cache.keySet()) {
            cnt += cache.get(s);
            if (cnt == 0) {
                Integer e = cache.higherKey(s);
                if (e != null) {
                    Interval i = new Interval(s, e);
                    res.add(i);
                }
            }
        }
        
        return res;
    }
}
