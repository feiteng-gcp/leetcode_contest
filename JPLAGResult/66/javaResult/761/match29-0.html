/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> av) {
        List<Interval> ls = kMerge(av);
        List<Interval> ans = new ArrayList<>();
        
        for(int i = 1; i < ls.size(); ++i) {
            ans.add(new Interval(ls.get(i - 1).end, ls.get(i).start));
        }
        return ans;
    }
    private List<Interval> kMerge(List<List<Interval>> av) {
        PriorityQueue<Node> pr = new PriorityQueue<>((a,b)->(a.in.start == b.in.start? a.in.end - b.in.end : a.in.start - b.in.start));
        int[] rec = new int[av.size()];
        for(int i = 0; i < rec.length; ++i) {
            pr.add(new Node(av.get(i).get(rec[i]), i));
        }
        List<Interval> ls = new ArrayList<>();
        while(pr.size() > 0) {
            Node node = pr.poll();
            if(ls.size() > 0 && ls.get(ls.size() - 1).end >= node.in.start) {
                ls.get(ls.size() - 1).end = Math.max(node.in.end, ls.get(ls.size() - 1).end);
            } else {
                ls.add(node.in);
            }
            int id = node.id;
            rec[id]++;
            if(rec[id] < av.get(id).size()) {
                pr.add(new Node(av.get(id).get(rec[id]), id));
            }
        }
        return ls;
    }
    class Node{
        Interval in;
        int id;
        public Node(Interval i, int x) {
            in = i;
            id = x;
        }
    }
}
