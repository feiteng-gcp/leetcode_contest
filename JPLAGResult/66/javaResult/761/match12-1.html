class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> ans = new LinkedList<>();
        List<List<Interval>> table = new LinkedList<>();
        for (List<Interval> tmpl: avails) {
            List<Interval> tmp = new LinkedList<>();
            tmp.addAll(tmpl);
            table.add(tmp);
        }
        int end = -1;
        while (table.size() != 0) {
            boolean flag = true;
            int minimum = Integer.MAX_VALUE;
            int e;
            List<List<Interval>> n = new LinkedList<>();
            for (List<Interval> tmp: table) {
                Interval i = tmp.get(0);
                if (i.start <= end) {
                    flag = false;
                    tmp.remove(0);
                    end = Math.max(end, i.end);
                    if (tmp.size() != 0) n.add(tmp);
                }else {
                    minimum = Math.min(minimum, i.start);
                    n.add(tmp);
                }
            }
            table = n;
            if (flag) {
                if (end != -1) ans.add(new Interval(end, minimum));
                end = minimum;
            }
        }
        return ans;
    }
}
