/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> list = new ArrayList<>();
        for(int i = 0;  i < avails.size(); i++){
            for(int j = 0; j < avails.get(i).size(); j++) list.add(avails.get(i).get(j));
        }
        
        Collections.sort(list, (a,b) ->(a.start == b.start ? a.end - b.end : a.start - b.start));
        List<Interval> res = new LinkedList<>();
        if(list.size() == 0) return res;
        int pre = list.get(0).end;
        
for(int i = 1; i < list.size(); i++){
            Interval it = list.get(i);
            if(it.start > pre){
                res.add(new Interval(pre, it.start));
                
pre = it.end;
            }
            else{
                pre = Math.max(pre, it.end);
            }
        }
        return res;
    }
}
