/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */

class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> res = new ArrayList<>();
        if (avails.size() == 0) {
            return res;
        }

        TreeMap<Integer, Integer> ivs = new TreeMap<>();
        for (List<Interval> list : avails) {
            if (list == null || list.isEmpty()) {
                continue;
            }

            for (Interval iv : list) {
                
merge(ivs, iv.start, iv.end);
            }
        }

        Map.Entry<Integer, Integer> prev = null;
        for (Map.Entry<Integer, Integer> entry : ivs.entrySet()) {
            if (prev != null && entry.getKey() > prev.getValue()) {
                res.add(new Interval(prev.getValue(), entry.getKey()));
            }
            prev = entry;
        }
        return res;
    }
    private void merge(TreeMap<Integer, Integer> ivs, int start, int end) {
        Integer prev = ivs.floorKey(end);
        while (prev != null && ivs.get(prev) >= start) {
            start = Math.min(prev, start);
            end = Math.max(ivs.get(prev), end);
            ivs.remove(prev);
            prev = ivs.floorKey(end);
        }
        ivs.put(start, end);
    }
}
