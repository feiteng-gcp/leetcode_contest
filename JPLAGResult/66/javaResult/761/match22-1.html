/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        int sz = 0;
        for(int i = 0; i < avails.size(); i++){
            sz+=avails.get(i).size();
        }
        
        Integer[][] darr = new Integer[sz][2]; 
        int cnt = 0;
        for(int i = 0; i < avails.size(); i++){
            for(int j = 0; j < avails.get(i).size(); j++){
                darr[cnt][0]=avails.get(i).get(j).start;
                darr[cnt][1]=avails.get(i).get(j).end;
                cnt++;
            }
        }
        
        Arrays.sort(darr, new Comparator<Integer[]>() {
            @Override
            public int compare(Integer[] o1, Integer[] o2) {
                Integer quantityOne = o1[0];
                Integer quantityTwo = o2[0];
                // reverse sort on quantity
                return quantityOne.compareTo(quantityTwo);
            }
        });
        
        List<Interval> res= new ArrayList<>();
        if(sz == 0) return res;
        
        // if(darr[0][0] > 1)
        //     res.add(new Interval(1,darr[0][0]));
        
        int mn = darr[0][0];
        int mx = darr[0][1];

        
        for(int i = 1; i < darr.length; i++){
            // System.out.println(mn + " " + mx);
            if(mx < darr[i][0]){
                //add2res and update
                res.add(new Interval(mx,darr[i][0]));
                
mn = darr[i][0];
                mx = darr[i][1];
            }
            else{
                mx=Math.max(mx, darr[i][1]);
            }
        }
        return res;
    }
