/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */

class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> list = new ArrayList<>();
        for(List<Interval> avail : avails){
            list.addAll(avail);
        }
        List<Interval> l2 = merge(list);
        List<Interval> res = new ArrayList<>();
        int start = l2.get(0).end;
        for(int i = 1; i < l2.size(); i++){
            res.add(new Interval(start, l2.get(i).start));
            start = 
l2.get(i).end;
        }
        // Map<Integer, Integer> map = new TreeMap<>();
        // for(List<Interval> avail : avails){
        //     for(Interval intv : avail){
        //         map.put(intv.start, map.getOrDefault(intv.start, 0) + 1);
        //         map.put(intv.end + 1, map.getOrDefault(intv.end, 0) - 1);
        //     }
        // }
        // System.out.println(map);
        return res;
    }
    
    public List<Interval> merge(List<Interval> intervals) {
        if(intervals == null || intervals.size() == 0) 
            return intervals;
        List<Interval> list = new ArrayList<>();
        Collections.sort(intervals, (a, b)->{
            return a.start - b.start;
        });
        for(Interval interval : intervals){
            if(list.size() == 0 || interval.start > list.get(list.size() - 1).end)
                list.add(interval);
            list.get(list.size() - 1).end = Math.max(interval.end, list.get(list.size() - 1).end);
        }
        return list;
    }
}
