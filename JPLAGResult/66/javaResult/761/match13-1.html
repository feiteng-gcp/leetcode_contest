/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> res = new ArrayList<>();
        if(avails == null || avails.size() == 0 || avails.get(0).size() == 0) {
            return res;
        }
        List<Interval> merged = mergeInterval(avails);
        res = findCommon(merged);
        return res;
    }
    public List<Interval> mergeInterval(List<List<Interval>> avails) {
        List<Interval> temp = new ArrayList<>();
        
List<Interval> merged = new ArrayList<>();
        //Interval pre = avails.get(0).get(0);
        for(List<Interval> avail : avails) {
            for(Interval interval : avail) {
                temp.add(interval);
            }
        }
        Collections.sort(temp, (a, b) -> {
            if(a.start == b.start) {
                return a.end - b.end;
            } else {
                return a.start - b.start;
            }
        });
        Interval pre = temp.get(0);
        
for(Interval cur : temp) {
            if(cur.start <= pre.end) {
                pre.end = Math.max(cur.end, pre.end);
            } else {
                merged.add(new Interval(pre.start, pre.end));
                pre = cur;
            }
        }
        merged.add(new Interval(pre.start, pre.end));
        return merged;
    }
    
public List<Interval> findCommon(List<Interval> merged) {
        List<Interval> res = new ArrayList<>();
        if(merged.size() == 0) {
            return res;
        }
        for(int i = 0; i < merged.size() - 1; i++) {
            Interval pre = merged.get(i);
            Interval cur = merged.get(i + 1);
            if(pre.end == cur.start) {
                continue;
            }
            res.add(new Interval(pre.end, cur.start));
        }
        return res;
    }
}
