/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> res=new ArrayList<Interval>();
        if(avails==null||avails.size()==0||avails.get(0).size()==0){
            return res;
        }
        
        
List<Interval> list=new ArrayList<Interval>();
        for(List<Interval> list1:avails){
            list.addAll(list1);
        }
        
        Collections.sort(list, (a, b)->(a.start==b.start?a.end-b.end:a.start-b.start));
        
        for(Interval intv:list){
            if(res.isEmpty()||res.get(res.size()-1).end<intv.start){
                res.add(intv);
            }else{
                res.get(res.size()-1).end=
Math.max(res.get(res.size()-1).end, intv.end);
            }
        }
        
        List<Interval> next=new ArrayList<Interval>();
        for(int i=0;i<res.size()-1;i++){
            if(res.get(i+1).start>res.get(i).end){
                next.add(new Interval(res.get(i).end, res.get(i+1).start));
            }
        }
        
        return next;
    }
}
