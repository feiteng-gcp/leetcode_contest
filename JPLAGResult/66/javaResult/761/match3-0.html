/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> busyTime = new ArrayList<>();
        for (List<Interval> ava : avails) {
            int idx = 0;
            List<Interval> list = new ArrayList<>();
            for (Interval inter : ava) {
                while (idx < busyTime.size() && busyTime.get(idx).end < inter.start)
                    list.add(busyTime.get(idx++));
                if (idx >= busyTime.size() || inter.end < busyTime.get(idx).start)
                    list.add(new Interval(inter.start, inter.end));
                else {
                    Interval cur = new Interval(Math.min(inter.start, busyTime.get(idx).start), Math.max(inter.end, busyTime.get(idx).end));
                    idx++;
                    while (idx < busyTime.size() && cur.end >= busyTime.get(idx).start)
                        cur.end = Math.max(cur.end, busyTime.get(idx++).end);
                    list.add(cur);
                }
            }
            while (idx < busyTime.size())
                list.add(busyTime.get(idx++));
            busyTime = new ArrayList<>();
            Interval merge = null;
            for (Interval inter : list) {
                if (merge == null)
                    merge = inter;
                else {
                    if (inter.start > merge.end) {
                        busyTime.add(merge);
                        merge = inter;
                    }
                    else
                        merge.end = Math.max(merge.end, inter.end);
                }
            }
            busyTime.add(merge);
        }
        List<Interval> res = new ArrayList<>();
        for (int i = 0; i + 1 < busyTime.size(); i++)
            if (busyTime.get(i).end < busyTime.get(i + 1).start)
                res.add(new Interval(busyTime.get(i).end, busyTime.get(i + 1).start));
        return res;
    }
}
