/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        List<Interval> result = new ArrayList<>();
        Map<Integer, Integer> events = new TreeMap<>();
        for (List<Interval> intervals : avails) {
            for (Interval busy : intervals) {
                events.put(busy.start, events.getOrDefault(busy.start, 0) + 1);
                events.put(busy.end, events.getOrDefault(busy.end, 0) - 1);
            }
        }
        
        int start = -1;
        int count = 0;
        for (Map.Entry<Integer, Integer> e : events.entrySet()) {
            int time = e.getKey();
            count += e.getValue();
            
            if (count > 0 && start != -1 && time > start) {
                result.add(new Interval(start, time));
                start = -1;
            }
            
            if (count == 0) {
                start = time;
            }
        }
        return result;
    }
}
