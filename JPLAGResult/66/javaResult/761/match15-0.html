/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public List<Interval> employeeFreeTime(List<List<Interval>> avails) {
        TreeMap<Integer, Integer> events = new TreeMap<>();
        for (List<Interval> intervals : avails) {
            for (Interval interval : intervals) {
                add(events, interval.start, 1);
                add(events, interval.end, -1);
            }
        }
        List<Interval> result = new ArrayList<>();
        int current = 0;
        Integer start = null;
        for (Map.Entry<Integer, Integer> e : events.entrySet()) {
            if (current == 0 && start != null) result.add(new Interval(start, e.getKey()));
            current += e.getValue();
            if (current == 0) start = e.getKey();
        }
        return result;
    }
    
    private static void add(TreeMap<Integer, Integer> events, int p, int v) {
        if (!events.containsKey(p)) events.put(p, 0);
        events.put(p, events.get(p) + v);
    }
}
