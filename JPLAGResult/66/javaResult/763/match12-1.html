class Solution {
    public String makeLargestSpecial(String S) {
		List<String> res = findAllSubSpecial(S, 0);
		if (res.size() == 1) {
			return makeLargestSpecial(S, 0);
		}
		String s = "";
		for (String r : res) {
			s += makeLargestSpecial(r);
		}
        return makeLargestSpecial(s, 0);
    }
	
	private List<String> findAllSubSpecial(String s, int start) {
		List<String> res = new ArrayList<>();
		int zeroCnt = 
0;
		int oneCnt = 0;
		for (int i = start; i < s.length(); i++) {
			if (s.charAt(i) == '1') {
				oneCnt++;
			} else {
				zeroCnt++;
			}
			if (oneCnt < zeroCnt) {
				break;
			} else if (oneCnt == zeroCnt) {
				res.add(s.substring(start, i + 1));
				start = i + 1;
			}
		}
		return res;
	}
	
	private List<String> findAllSpecial(String s, int start) {
		List<String> res = new ArrayList<>();
		int zeroCnt = 0;
		int oneCnt = 0;
		for (int i = start; i < s.length(); i++) {
			if (s.charAt(i) == '1') {
				oneCnt++;
			} else {
				zeroCnt++;
			}
			if (oneCnt < zeroCnt) {
				break;
			} else if (oneCnt == zeroCnt) {
				res.add(s.substring(start, i + 1));
			}
		}
		return res;
	}
	
	private String makeLargestSpecial(String s, int start) {
		if (start == s.length()) {
			return s;
		}
		List<String> allSpecials = findAllSpecial(s, start);
		if (allSpecials.isEmpty()) {
			return makeLargestSpecial(s, start + 1);
		}
		String replMax = null;
		String currMax = "00";
		for (String k : allSpecials) {
			List<String> res = findAllSpecial(s, start + k.length());
			for (String r : res) {
				if (k.compareTo(r.substring(0, Math.min(r.length(), k.length()))) < 0) {
					// found a potential larger one
					if (currMax.compareTo(r.substring(0, Math.min(r.length(), currMax.length()))) < 0) {
						replMax = k;
						currMax = r;
					}
				}
			}
		}
		if (replMax != null) {
			s = s.substring(0, start) + currMax + replMax + s.substring(start + currMax.length() + replMax.length());
			return makeLargestSpecial(s, start);
		} else {
			return makeLargestSpecial(s, start + 1);
		}
	}
}
