class Solution {
    public int[] anagramMappings(int[] A, int[] B) {
           int[] res = new int[A.length];
           Set<Integer> set = new HashSet<>();
           for (int i = 0; i < A.length; i++) {
               for (int j = 0; j < B.length; j++) {
                   if (A[i] == B[j] && !set.contains(j)) {
                       res[i] = j;
                       set.add(j);
                       break;
                   }
               }
           }
           return res;
    }
}
