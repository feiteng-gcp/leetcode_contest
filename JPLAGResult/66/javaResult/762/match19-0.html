class Solution {
    public int[] anagramMappings(int[] A, int[] B) {
        Map<Integer, Integer> masterMap = new HashMap<>(B.length);
        for (int i = 0; i < B.length; i++) {
            masterMap.put(B[i], i);
        }
        int[] out = new int[A.length];
        for (int i = 0; i < out.length; i++) {
            out[i] = masterMap.get(A[i]);
        }
        return out;
    }
}
