class Solution {
    public int[] anagramMappings(int[] A, int[] B) {
        int[] result = new int[A.length];
        HashMap<Integer, Integer> temp = new HashMap<>();
        for(int i = 0; i < B.length; i++) {
            temp.put(B[i], i);
        }
        for(int j = 0; j < A.length; j++) {
            result[j] = temp.get(A[j]);
        }
        return result;
    }
}
