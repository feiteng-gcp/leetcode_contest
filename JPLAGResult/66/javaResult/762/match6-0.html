class Solution {
    public int[] anagramMappings(int[] A, int[] B) {
        int[] index = new int[A.length];
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < B.length; i++) {
            map.put(B[i], i);
        }
        for (int i = 0; i < A.length; i++) {
            index[i] = map.get(A[i]);
        }
        return index;
    }
}
