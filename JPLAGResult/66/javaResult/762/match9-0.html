class Solution {
    public int[] anagramMappings(int[] A, int[] B) {
        int[] sol = new int[A.length];
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        for(int i=0; i<B.length; i++){
            map.put(B[i], i);
        }
        for(int i=0; i<A.length; i++){
            sol[i] = map.get(A[i]);
        }
        return sol;
    }
}
