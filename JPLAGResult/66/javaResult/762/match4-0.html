class Solution {
    public int[] anagramMappings(int[] A, int[] B) {
        //if(A.length != B.length) return null
        int[] res = new int[A.length];
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < B.length; i++) {
            map.put(B[i], i);
        }
        for (int i = 0; i < A.length; i++) {
            res[i] = map.get(A[i]);
        }
        return res;
    }
}
