class Solution {
    public int[] anagramMappings(int[] a, int[] b) {
        
        int ans[]=new int[b.length];
        HashMap<Integer,Integer> mp=new HashMap(b.length);
        for(int i=0;i<b.length;i++){
            mp.put(b[i],i);
        }
        
        for(int i=0;i<b.length;i++){
            ans[i]=mp.get(a[i]);
        }
        return ans;
        
        
    }
}
