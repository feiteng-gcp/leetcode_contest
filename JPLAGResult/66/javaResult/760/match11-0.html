class Solution
{
    public String boldWords(String[] words, String S)
    {
        int[] bold=new int[S.length()+1];
        for (String word : words)
        {
            int begin=S.indexOf(word);//begin=1
            while(begin!=-1)
            {
                for (int i = begin; i < begin+word.length(); i++)//ab
                {
                    bold[i+1]=1;
                }
                begin=S.indexOf(word,begin+1);
            }
        }
        StringBuilder sb=new StringBuilder();
        int iBuf=0;
        for (int i = 0; i < bold.length; i++)
        {
            if(i>0)
            {
                if(iBuf==0 && bold[i]==1)
                {
                    sb.append("<b>");
                }
                else if(iBuf==1 && bold[i]==0)
                {
                    sb.append("</b>");
                }
                sb.append(S.charAt(i-1));

            }
            iBuf=bold[i];
        }
        if(iBuf==1)
        {
            sb.append("</b>");
        }
        return sb.toString();
    }
}
