class Solution {
    public String boldWords(String[] words, String S) {
        boolean [] setBold = new boolean[S.length() + 2];
        for (String word : words) {
            int index = S.indexOf(word, 0);
            while (index != -1) {
                Arrays.fill(setBold, index + 1, index + word.length() + 1, true);
                index = S.indexOf(word, index + 1);
            }
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 1; i < setBold.length; i++) {
            if (setBold[i] && !setBold[i - 1]) {
                sb.append("<b>");
            }
            if (!setBold[i] && setBold[i - 1]) {
                
sb.append("</b>");
            }
            if (i < S.length() + 1) sb.append(S.charAt(i - 1));
        }
        return sb.toString();
    }
}
