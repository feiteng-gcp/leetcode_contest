class Solution {
private boolean isMatch(String S , int startIndex , String word) {
		
		int i = startIndex , j = 0;
		while (i < S.length() && j < word.length() && (S.charAt(i) == word.charAt(j))) {
			i ++;
			j ++;
		}
		return j == word.length();
		
	}
	
    public String boldWords(String[] words, String S) {
        	
    	int i , length = S.length();
    	int[] cnt = new int[length + 100];
    	boolean[] flag = new boolean[length];
    	for (i = 0;i < length;i ++) {
    		for (String word : words) {
    			if (isMatch(S , i , word)) {
    				cnt[i] ++;
    				cnt[i + word.length()] --;
    			}
    		}
    	}
    	int value = 0;
    	for (i = 0;i < length;i ++) {
    		value += cnt[i];
    		if (value > 0) {
    			flag[i] = true;
    		}
    	}
    	StringBuilder builder = new StringBuilder();
    	for (i = 0;i < length;i ++) {
    		if (flag[i]) {
    			builder.append("<b>");
    			while (i < length && flag[i]) {
    				builder.append(S.charAt(i));
    				i ++;
    			}
    			i --;
    			builder.append("</b>");
    		} else {
    			builder.append(S.charAt(i));
    		}
    	}
    	return builder.toString();
    	
    }
	
}
