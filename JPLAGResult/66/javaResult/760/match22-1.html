class Solution {
    public String boldWords(String[] words, String S) {
        boolean[] appear = new boolean[S.length()];
        for(int i=0;i<S.length();i++){
            String x = S.substring(i);
            for(String s : words){
                if(
x.startsWith(s)){
                    for(int n=i;n<i+s.length();n++){
                        appear[n] = true;
                    }
                }
            }
        
}
        StringBuffer ans = new StringBuffer();
        for(int i=0;i<S.length();i++){
            if(appear[i]){
                int index = i;
                while(index<S.length()&&appear[index]){
                    index++;
                }
                ans.append("<b>");
                for(int t=i;t<index;t++){
                    ans.append(S.charAt(t));
                }
                ans.append("</b>");
                i = index - 1;
            }else{
                ans.append(S.charAt(i));
            }
        }
        return ans.toString();
    }
}
