class Solution {
    public String boldWords(String[] words, String S) {
        // 应该是尽可能的长吧...
        //这个不就是昨天做过的那个题么...
        int n = S.length();
        boolean[] contained = new boolean[n];
        for (int i = 0; i < n; i++) {
            int len = 0;
            
for (String word : words) {
                if (S.startsWith(word, i)) {
                    len = Math.max(len, word.length());
                }
            }
            for (int j = i; j < len + i; j++) {
                contained[j] = true;
            }
        }
        for (int i = 0; i < n; i++) {
            System.out.print(contained[i] + "| ");
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (contained[i] && (i == 0 || !contained[i - 1])) sb.append("<b>");
            
            
            sb.append(S.charAt(i));
            
            if ((i == n - 1 && contained[i]) || (contained[i] && i < n - 1 && !contained[i + 1])) sb.append("</b>");
        }
        return sb.toString();
    }
}
