class Solution {
    public String boldWords(String[] words, String S) {
        if (words == null || words.length == 0) return "";
        
        Set<Integer> indices = new HashSet<>();
        for (String word : words) {
            markIndices(S, word, indices);
        }    
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < S.length(); i++) {
            if (indices.contains(i) && (i == 0 || !indices.contains(i - 1))) {
                sb.append("<b>");
            }
            sb.append(S.charAt(i));
            
            if (indices.contains(i) && (i == S.length() - 1 || !indices.contains(i + 1))) {
                sb.append("</b>");
            }
        }
        return sb.toString();
    }
    
    void markIndices(String S, String word, Set<Integer> indices) {
        for (int i = 0; i <= S.length() - word.length(); i++) {
            String substr = S.substring(i, i + word.length());
            if (substr.equals(word)) {
                for (int j = i; j < i + word.length(); j++) {
                    indices.add(j);
                }
            }
        }
    }
}
