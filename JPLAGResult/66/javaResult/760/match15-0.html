class Solution {
    public String boldWords(String[] words, String S) {
        boolean[] tag = new boolean[S.length()];
        for (String w : words) {
            int from = 0, index = -1;
            while (-1 != (index = S.indexOf(w, from))) {
                for (int i = 0; i < w.length(); ++i) {
                    tag[index + i] = true;
                }
                from++;
            }            
        }
        int i = 0;
        StringBuilder sb = new StringBuilder();
        while (i < S.length()) {
            if (tag[i]) {
                sb.append("<b>");
                while (i < S.length() && tag[i]) {
                    sb.append(S.charAt(i));
                    i++;
                }
                sb.append("</b>");
            } else {
                sb.append(S.charAt(i));
                i++;
            }
        }
        return sb.toString();
    }
}
