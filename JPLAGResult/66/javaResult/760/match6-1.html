class Solution {
    public String boldWords(String[] words, String S) {
        final int n = S.length();
        boolean[] mark = new boolean[n];
        
        for (String word: words) {
            final int wlen = word.length();
            for (int i = 0; i+wlen <= n; ++i) {
                boolean match = true;
                for (int j = 0; j < wlen; ++j) 
{
                    if (S.charAt(i+j) != word.charAt(j)) {
                        match = false;
                        break;
                    }
                }
                if (match) {
                    for (int j = 0; j < wlen; ++j) 
{
                        mark[i+j] = true;
                    }
                }
            }
        }
        
        StringBuilder sb = new StringBuilder();
        boolean inB = false;
        for (int i = 0; i < n; ++i) {
            if (mark[i]) {
                if (!inB) {
                    inB = true;
                    sb.append("<b>");
                }
            } else {
                if (inB) {
                    inB = false;
                    sb.append("</b>");
                }
            
}
            sb.append(S.charAt(i));
        }
        if (inB) sb.append("</b>");
        return sb.toString();
    }
}
