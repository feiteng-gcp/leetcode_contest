class Solution {
    public String boldWords(String[] words, String S) {
        if(words==null||words.length==0){
            return S;
        }       
        char[] array=S.toCharArray();
        boolean[] b=new boolean[array.length];
       
        
for(String str:words){
            int index=S.indexOf(str);
            while(index>=0){
                
                for(int i=index;i<=index+str.length()-1;i++){
                    b[i]=true;
                }
                index=S.indexOf(str, index+1);
            }
        }
        
        int i=0;
        StringBuilder sb=new StringBuilder();
        boolean flag=false;
        while(i<array.length){
            if(!b[i]){
                if(flag){
                    sb.append("</b>");
                    flag=false;
                }
                sb.append(array[i]);
            }else{
                if(!flag){
                    sb.append("<b>");
                    flag=true;
                }
                sb.append(array[i]);
            }
            i++;
        }
        
        if(flag){
            sb.append("</b>");
        }
        
        return sb.toString();
        
    }
}
