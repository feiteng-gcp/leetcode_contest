class Solution {
    public String boldWords(String[] words, String S) {
        Arrays.sort(words, (a, b)-> Integer.compare(b.length(), a.length()));
        
Map<Integer, Integer> matchLengthAtIdx = new HashMap<>();

        for (int i = 0; i < S.length(); i++) {
            for (int j = 0; j < words.length; j++) {
                if (i + words[j].length() <= S.length() && 
                        S.substring(i, i + words[j].length()).equals(words[j])) {
                    matchLengthAtIdx.put(i, i + words[j].length());
                    break;
                }
            }
        }

        StringBuilder sb = new StringBuilder();
        int matchUntilIdx = -1;
        for (int i = 0; i < S.length(); i++) {
            if (matchLengthAtIdx.containsKey(i)) {
                
if (matchUntilIdx < i) {
                    sb.append("<b>");
                }
                matchUntilIdx = Math.max(matchUntilIdx, matchLengthAtIdx.get(i));
            }
            
if (matchUntilIdx == i) {
                sb.append("</b>");
            }
            sb.append(S.charAt(i));
        }

        if (matchUntilIdx == S.length()) {
            sb.append("</b>");
        }

        return sb.toString();
    }
}
