class Solution {
    class Interval {
      int start;
      int end;
      Interval() { start = 0; end = 0; }
      Interval(int s, int e) { start = s; end = e; }
    }
    public String boldWords(String[] words, String S) {
        ArrayList<Interval> arr = new ArrayList<>();
        for (String word : words) {
            int index = 0;
            while (index<S.length() && index != -1) {
                int s = S.indexOf(word, index);
                int e = s + word.length();
                if (s != -1) arr.add(new Interval(s, e));
                index = s == -1 ? -1 : s+1;
            
}
        }
        ArrayList<Interval> list = merge(arr);
        StringBuilder sb = new StringBuilder();
        int index = 0;
        for (Interval i : list) {
            int s = i.start;
            int e = i.end;
            // System.out.println(s+","+e);
            while (index < s) sb.append(S.charAt(index++));
            sb.append("<b>");
            while (index < e) sb.append(S.charAt(index++));
            sb.append("</b>");
        }
        for (int j = index; j < S.length(); ++j) {
            
sb.append(S.charAt(j));
        }
        return sb.toString();
    }
    
    public ArrayList<Interval> merge(ArrayList<Interval> intervals) {
        if (intervals.size() == 0 || intervals.size() == 1) return intervals; 
        PriorityQueue<Interval> pq = new PriorityQueue<Interval>(intervals.size(), 
new Comparator<Interval>() {
            @Override
            public int compare(Interval i1, Interval i2) {
                if (i1.start < i2.start) {
                    return -1;
                } else if (i1.start == i2.start) {
                    return 0;
                } else {
                    return 1;
                }
            }
        });
        for (Interval i : intervals) {
            pq.add(i);
        }
        ArrayList<Interval> result = new ArrayList<Interval>();
        Interval tmp = pq.poll();
        int min = tmp.start, max = tmp.end;
        while (!pq.isEmpty()) {
            tmp = pq.poll();
            if (tmp.start > max) {
                result.add(new Interval(min, max));
                min = tmp.start;
                max = tmp.end;
            } else {
                min = Math.min(min, tmp.start);
                max = Math.max(max, tmp.end);
            }
        }
        result.add(new Interval(min, max));
        return result;
    }
}
