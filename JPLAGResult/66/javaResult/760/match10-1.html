class Solution {
    public String boldWords(String[] words, String S) {
        boolean[] mark = new boolean[S.length()];
        for(String w: words){
            String sub = S;
            
int i = sub.indexOf(w);
            while(i!=-1){
                for(int j = i; j< i+w.length(); j++){
                    mark[j]=true;
                }
                i = sub.indexOf(w, i+1);
            }
        }
        String tagBegin = "<b>";
        String tagEnd = "</b>";
        StringBuilder sb = new StringBuilder();
        boolean isLastCharBold = false;
        for(int i=0; i<mark.length; i++){
            if(!isLastCharBold&&mark[i]){
                sb.append(tagBegin);
                sb.append(S.charAt(i));
                isLastCharBold = true;
            } else if(isLastCharBold&&!mark[i]){
                sb.append(tagEnd);
                sb.append(S.charAt(i));
                isLastCharBold = false;
            } else {
                sb.append(S.charAt(i));
            }
        
}
        if(isLastCharBold){
            sb.append(tagEnd);
        }
        return sb.toString();
    }
}
