class Solution {
    public String boldWords(String[] words, String S) {
        int n = S.length();
        if (n==0)
            return "";
        int w = words.length;
        boolean[] bold = new boolean[n];
        for (int i = 0; i < w; i++) {
            for (int j = 0; j <= n-words[i].length(); j++) {
                boolean bw = true;
                for (int k = 0; k < words[i].length(); k++) {
                    if (words[i].charAt(k)!=S.charAt(j+k)) {
                        bw = false;
                        break;
                    }
                }
                if (bw) {
                    for (int k = 0; k < words[i].length(); k++) {
                        bold[j+k] = true;
                    }
                }
            }
        }
        StringBuilder ans = new StringBuilder();
        if (bold[0])
            ans.append("<b>");
        ans.append(S.charAt(0));
        for (int i = 1; i < n; i++) {
            if (!bold[i-1]&&bold[i])
                ans.append("<b>");
            if (bold[i-1]&&!bold[i])
                ans.append("</b>");
            ans.append(S.charAt(i));
        }
        if (bold[n-1])
            ans.append("</b>");
        return ans.toString();
    }
}
