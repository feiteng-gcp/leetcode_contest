class Solution {
    public String boldWords(String[] words, String S) {
        boolean[] bold = new boolean[S.length()];
        for(String word:words){
            int totalAway= 0;
            int index = S.indexOf(word);
            totalAway+=index;
            while(index!=-1){
                for(int i = index; i<word.length()+index; i++){
                    bold[i] = true;
                }
                int nextInd = S.substring(index+1).indexOf(word);
                if(nextInd==-1){
                    index = -1;
                }
                else{
                    index = nextInd + index+1;
                }
            }
        }
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i<S.length(); i++){
            int cur = i;
            while(cur<S.length()&&bold[cur]){
                cur++;
            }
            if(cur==i){
                sb.append(S.charAt(i));
            }
            else{
                sb.append("<b>");
                for(int j = i; j<cur; j++){
                    sb.append(S.charAt(j));
                }
                sb.append("</b>");
                i = cur-1;
            }
        }
        return sb.toString();
    }
}
