class Solution {
    public String boldWords(String[] words, String S) {
 
        
        boolean []wordArr = new boolean[S.length()];
        for( String word : words){
            int wordLen = word.length();
            for(int i = 0; i <= S.length() - wordLen;i++){
                if(S.substring(i, i + wordLen).equals(word)){
                    for( int j = i ; j < i+wordLen; j++){
                        wordArr[j] = true;
                    }
                }
            }
        }
        
        StringBuilder boldWords = new StringBuilder();
        for(int i=0; i <S.length();){
            if(!wordArr[i]){
                boldWords.append(S.charAt(i));
                i++;
            }
            else{
                String tmp = "";
                while( i < S.length() && wordArr[i]){
                    tmp = tmp + S.charAt(i);
                    i++;
                }
                boldWords.append("<b>").append(tmp).append("</b>");
            }
            
        }
        
        return boldWords.toString();
    }
}
