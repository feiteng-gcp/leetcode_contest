class Solution {
    public String boldWords(String[] words, String S) {
        boolean[] bold = new boolean[S.length()];
        for (String word: words) {
            setBold(word, S, bold);
        }
        
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < bold.length;) {
            int j = i;
            while (j < bold.length && bold[j] == bold[i]) j++;
            if (bold[i]) {
                sb.append("<b>");
                sb.append(S.substring(i, j));
                sb.append("</b>");
            } else {
                sb.append(S.substring(i, j));
            }
            i = j;
        }
        return sb.toString();
    }
    
    private void setBold(String key, String S, boolean[] bold) {
        for (int i = 0; i <= S.length() - key.length(); i++) {
            boolean isBold = true;
            for (int j = 0; j < key.length(); j++) {
                if (S.charAt(i + j) != key.charAt(j)) {
                    isBold = false;
                    break;
                }
            }
            if (isBold) 
{
                for (int j = 0; j < key.length(); j++) {
                    bold[i + j] = true;
                }
            }
        }
    }
}
