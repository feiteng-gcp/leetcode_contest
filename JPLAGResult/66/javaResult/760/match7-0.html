class Solution {
    public String boldWords(String[] words, String S) {
        int N = S.length();
        int[] P = new int[N + 1];
        for (String word : words) {
        	int len = word.length();
        	for (int i = 0; i + len <= S.length(); ++i) {
        		if (S.startsWith(word, i)) {
        			P[i]++;
        			P[i + len]--;
        		}
        	}
        }
        StringBuilder sb = new StringBuilder();
        int sum = 0;
        boolean begin = false;
        for (int i = 0; i < S.length(); ++i) {
        	if (P[i] == 0) {
        		sb.append(S.charAt(i));
        		continue;
        	} else {
        		sum += P[i];
        		if (sum > 0 && !begin) {
        			begin = true;
        			sb.append("<b>");
        		} else if (sum == 0) {
        			begin = false;
        			sb.append("</b>");
        		} 
        		sb.append(S.charAt(i));
        	}
        }
        if (begin) sb.append("</b>");
        return sb.toString();
    }
    
    public static void main(String[] args) {
    	Solution solution = new Solution();
    	String[] words = {"ccb","b","d","cba","dc"};
    	String S = "eeaadadadc";
    	System.out.println(solution.boldWords(words, S));
    }
}
