class Solution {
    public String boldWords(String[] words, String S) {
        char[] c = S.toCharArray();
        int n = c.length;
        if(n==0) return S;
        
        boolean[] isBold = new boolean[n];
        Arrays.fill(isBold, false);
        for(String word : words) {
            for(int i=0;i<=n-word.length();i++) {
                if(word.equals(new String(c, i, word.length()))) {
                    for(int j=0;j<word.length();j++) {
                        isBold[i+j] = true;
                    }
                }
            }
        }
        
        StringBuffer ret = new StringBuffer();
        for(int i=0;i<=n;i++) {
            boolean curBold = getIsBold(isBold, i), prevBold = getIsBold(isBold, i-1);
            if(curBold &! prevBold) ret.append("<b>");
            if(prevBold &! curBold) ret.append("</b>");
            if(i!=n) ret.append(c[i]);
        }
        return ret.toString();
    }
    
    public boolean getIsBold(boolean[] isBold, int index) {
        int n = isBold.length;
        if (index<0 || index>=n) return false;
        else return isBold[index];
    }
}
