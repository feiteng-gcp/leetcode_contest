class Solution {
    public String boldWords(String[] words, String S) {
        boolean[] bold = new boolean[S.length()];
        for(String str : words) {
            int fromIdx = 0, startIdx = 0;
            while(fromIdx < S.length() && startIdx != -1) {
                startIdx = S.indexOf(str, fromIdx);
                if(startIdx != -1) {
                    for(int i = 0; i < str.length(); i++) {
                        bold[startIdx + i] = true;
                    }
                    fromIdx = startIdx + 1;
                }
            
}   
        }
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < bold.length; i++) {
            if(bold[i]) {
                int j = i + 1;
                for(; j < bold.length && bold[j]; j++);
                sb.append("<b>")
                    .append(S.substring(i, j))
                    .append("</b>");
                i = j - 1;
            }
            else {
                sb.append(
S.charAt(i));
            }
        }
        return sb.toString();
    }
}
