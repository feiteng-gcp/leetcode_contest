class Solution {
    public String boldWords(String[] words, String S) {
        //这个就是找子串了，
        //找到子串，然后把连续的都加粗就可以了
        int[] flag = new int[S.length()];
        for(int i=0;i<words.length;i++){
            int index = S.indexOf(words[i], 0);
            while(-1!=index){
                Arrays.fill(flag, index, index+words[i].length(),1);
                
                index = S.indexOf(words[i], index+1);
            } 
        }
        StringBuilder sb = new StringBuilder();
        
        //011011  110011 1
        for(int i=0;i<flag.length;i++){
            System.out.println(flag[i]);
            // 什么时候加<b>
            if(flag[i]==1 && (i==0 || flag[i-1] == 0) ){
                sb.append("<b>");
            }
            //什么时候加</b>
            if(flag[i]==0 && i>0 && flag[i-1] == 1){
                sb.append("</b>");
            }
            
            sb.append(S.charAt(i));
            // 最特殊的情况，末尾是1
            if(i==flag.length-1 && flag[i] == 1){
                sb.append("</b>");
            }
            
                
            
        }
        
        return sb.toString();
        
    }
}
