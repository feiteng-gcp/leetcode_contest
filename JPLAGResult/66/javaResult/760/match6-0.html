class Solution {
    public String boldWords(String[] words, String S) {
        int[] flags = new int[S.length()];
        for (String w : words) {
            helper(w, S, flags);
        }
        
        boolean open = false;
        StringBuilder sb = new StringBuilder();
        for (int i=0; i<S.length(); i++) {
            if (flags[i] == 0 && open) {
                sb.append("</b>");
                open = false;
            }
            if (flags[i] == 1 && !open) {
                open = true;
                sb.append("<b>");
            }
            sb.append(S.charAt(i));
        }
        if (open) {
            sb.append("</b>");
        }
        
        return sb.toString();
    }
    
    private void helper(String w, String S, int[] flags) {
        for (int i=0; i<S.length(); i++) {
            if (i+w.length()-1 < S.length()) {
                String temp = S.substring(i, i+w.length());
                if (temp.equals(w)) {
                    for (int j=i; j<i+w.length(); j++) {
                        flags[j] = 1;
                    }
                }
            }
        }
    }
}
