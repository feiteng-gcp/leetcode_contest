class Solution {
    public String boldWords(String[] words, String S) {
        
        int mark[] = new int[S.length()];
        for(String w: words) {
            markChars(S, w, mark);
        }
        
        StringBuffer ans = new StringBuffer();
        
        for(int i=0; i<S.length(); i++) {
            if(mark[i] == 1) {
                ans.append("<b>");

                while(i<S.length() && mark[i] == 1) {
                    ans.append(S.charAt(i));
                    i++;
                }
            
                ans.append("</b>");
            }
            
            if(i<S.length())
                ans.append(S.charAt(i));
     
        }
        
        return ans.toString();
    }
    
    void markChars(String S, String w, int mark[]) {
        
        for(int i=0; i<=S.length()-w.length(); i++) {
            if(S.substring(i, i+w.length()).equals(w)) {
                for(int j=i; j<i+w.length(); j++) {
                    mark[j] = 1;
                }
            }
        }
    }
}
