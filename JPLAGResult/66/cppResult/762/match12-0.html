class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        unordered_map<int, int> anagramValueToIndexMap;
        for(int i = 0; i < B.size(); ++i) {
            anagramValueToIndexMap[B[i]] = i;
        }
        
        vector<int> result(A.size());
        for(int j = 0; j < A.size(); ++j) {
            result[j] = anagramValueToIndexMap[A[j]];
        }
        
        return result;
    }
};
