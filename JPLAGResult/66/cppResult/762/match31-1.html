class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        vector<int> res(A.size());
        unordered_multimap<int, int> m;
        for (int i = 0; i < B.size(); i++)
            m.insert(make_pair(B[i], i));
        

        for (int i = 0; i < A.size(); i++) {
            auto it = m.find(A[i]);
            res[i] = it->second;
            m.erase(it);
        }
        return res;
    }
};
