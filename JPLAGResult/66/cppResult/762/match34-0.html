class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        vector<int> ret;
        unordered_map<int, vector<int>> hash;
        for (int i = 0; i < B.size(); ++ i)
            hash[B[i]].push_back(i);
        for (int i = 0; i < A.size(); ++ i) {
            ret.push_back(hash[A[i]].back());
            hash[A[i]].pop_back();
        }
        return ret;
    }
};
