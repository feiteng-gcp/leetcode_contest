class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        unordered_map<int, int> B_val_idx;
        for (int i = 0; i < B.size(); ++i) {
            B_val_idx[B[i]] = i;
        }
        vector<int> res(A.size());
        for (int i = 0; i < A.size(); ++i) {
            res[i] = B_val_idx[A[i]];
        }
        return res;
    }
};
