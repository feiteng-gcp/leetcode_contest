class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        map<int,vector<int>> m;
        vector<int> ret;
        for(int i=0;i<B.size();i++){
            m[B[i]].push_back(i);
        }
        for(int i=0;i<A.size();i++){
            ret.push_back(m[A[i]].back());
            m[A[i]].pop_back();
        }
        return ret;
    }
};
