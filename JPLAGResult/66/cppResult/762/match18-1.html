class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        map<int, vector<int> > index;
        for(int i = 0; i < B.size(); ++i) {
            index[B[i]].push_back(i);
        }
        vector<int> ans;
        for(int i = 0; i < A.size(); ++i) {
            ans.push_back(index[A[i]].back());
            index[A[i]].pop_back();
        }
        return ans;
    }
};
