class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        int n = A.size();
        if (n == 0) return {};
        unordered_map<int, vector<int>> mp;
        for (int i = 0; i < n; i++) {
            mp[B[i]].push_back(i);
        }
        vector<int> res(n,-1);
        for (int i = 0; i < n; i++) {
            res[i] = mp[A[i]].back();
            mp[A[i]].pop_back();
        }
        return res;
    }
};
