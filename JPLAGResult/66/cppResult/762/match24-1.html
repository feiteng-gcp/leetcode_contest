class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        vector<bool> visited(B.size(), false);
        vector<int> result(B.size(), 0);
        
        for (int i = 0; i < B.size(); ++i) {
            for (int j = 0; j < B.size(); ++j) {
                if (A[i] == B[j] && visited[j] == false) {
                    visited[j] = true;
                    result[i] = j;
                    break;
                }
            }
        }
        
        return result;
    }
};
