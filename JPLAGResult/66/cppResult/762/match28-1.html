class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        map<int, queue<int> > pos;

    int len = (int)A.size();
    for(int i = 0; i < len; i++) pos[B[i]].push(i);
    vector<int> ans;
    for(int i = 0; i < len; i++)
    {
        ans.push_back(pos[A[i]].front());
        pos[A[i]].pop();
    }
    return ans;
    }
};
