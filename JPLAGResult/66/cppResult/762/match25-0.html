class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        vector<int> result;
        unordered_map<int, vector<int>> ht;
        for (int i = 0; i < B.size(); i++) {
            ht[B[i]].push_back(i);
        }
        for (int i = 0; i < A.size(); i++) {
            result.push_back(ht[A[i]].back());
            ht[A[i]].pop_back();
        }
        return result;
    }
};
