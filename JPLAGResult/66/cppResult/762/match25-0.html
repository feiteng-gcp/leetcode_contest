class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        vector<int> ret(A.size(), -1);
        for (int i = 0; i < B.size(); ++i) {
            for (int j = 0; j < A.size(); ++j) {
                if (A[j] == B[i] && ret[j] == -1) {
                    ret[j] = i;
                    break;
                }
            }
        }
        return ret;
    }
};
