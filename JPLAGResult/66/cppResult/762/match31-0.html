class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        vector<int> res(A.size());
        unordered_multimap<int, int> mp;
        for (int i = 0; i < B.size(); ++i)
            mp.insert({B[i], i});
        for (int i = 0; i < A.size(); ++i)
        {
            auto it = mp.find(A[i]);
            res[i] = it->second;
            mp.erase(it);
        }
        return res;
    }
};
