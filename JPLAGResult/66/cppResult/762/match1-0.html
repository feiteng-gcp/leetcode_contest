class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        int n = A.size();
        vector<bool> visit(n);
        vector<int> ret(n);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (A[i] == B[j] && !visit[j]) {
                    ret[i] = j;
                    visit[j] = true;
                    break;
                }
            }
        }
        return ret;
        
    }
};
