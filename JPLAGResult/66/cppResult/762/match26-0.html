class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        if(A.size()==0 || B.size()==0)
            return vector<int>();
        int s1 = A.size();
        int s2 = B.size();
        vector<int> ans;
        
        for(int i=0;i<s1;i++)
        {
            for(int j=0;j<s2;j++)
            {
                if(A[i]==B[j])
                {
                    ans.push_back(j);
                    break;
                }
            }
        }
        return ans;
    }
};
