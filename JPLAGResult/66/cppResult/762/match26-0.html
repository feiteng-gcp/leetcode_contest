class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        map<int, vector<int> > b;
        for (int i = 0; i < B.size(); i++) {
            b[B[i]].push_back(i);
        }
        vector<int> a;
        for (int i = 0; i < A.size(); i++) {
            auto it = b.find(A[i]);
            a.push_back(*(it->second.begin()));
            it->second.erase(it->second.begin());
        }
        return a;
    }
};
