class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        unordered_map<int, vector<int> > pos;
        for (int i = 0; i < B.size(); ++i) {
            pos[B[i]].push_back(i);
        }
        vector<int> result;
        for (int i = 0; i < A.size(); ++i) {
            int num = A[i];
            result.push_back(pos[num].back());
            pos[num].pop_back();
        }
        return result;
    }
};
