class Solution {
public:
    vector<int> anagramMappings(vector<int>& A, vector<int>& B) {
        map<int, vector<int>> mp;
        for(int i=0;i<B.size();++i){
            mp[B[i]].push_back(i);
        }
        
        vector<int> P;
        for(int i=0;i<A.size();++i){
            P.push_back(mp[A[i]].back());
            mp[A[i]].pop_back();
        }
        return P;
    }
};
