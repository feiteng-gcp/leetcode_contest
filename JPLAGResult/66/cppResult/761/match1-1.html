/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<pair<int, int>> ts;
      for (auto& avail: avails) {
        for (auto& d: avail) {
          ts.push_back(make_pair(d.start, 1));
          ts.push_back(make_pair(d.end, -1));
        }
      }
      sort(ts.begin(), ts.end());
      int c = 0;
      int prev = -1;
      vector<Interval> ans;
      for (auto& p: ts) {
        c += p.second;
        if (p.second == 1) {
          if (prev != -1 && c == 1 && prev != p.first) {
            ans.push_back(Interval(prev, p.first));
            prev = -1;
          }
        } else if (p.second == -1) {
          if (c == 0) {
            prev = p.first;
          }
        }
      }
      return ans;
    }
};
