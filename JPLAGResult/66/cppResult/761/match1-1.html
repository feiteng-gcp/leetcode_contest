/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<Interval> ss;
        vector<Interval> svec;

        for(int i = 0; i < avails.size(); i++)
        {
            for(int j = 0; j < avails[i].size(); j++)
            {
                svec.push_back(avails[i][j]);
            }
        }
        sort(svec.begin(), svec.end(), [](Interval a, Interval b){ if(a.start < b.start) return true;
                                                                 if(a.start == b.start && a.end > b.end)
                                                                 return true;
                                                                 return false;});
        int current1 = svec[0].start;
        int current2 = svec[0].end;
        for(int i = 0; i < svec.size(); i++)
        {
            if(svec[i].end < current2)
            {
                continue;
            }
            if(svec[i].start < current2)
            {
                current2 = svec[i].end;
                continue;
            }
            else
            {
                Interval tmp;
                tmp.start = current2;
                tmp.end = svec[i].start;
                if(tmp.start != tmp.end)
                {
                    ss.push_back(tmp);
                }
                current1 = svec[i].start;
                current2 = svec[i].end;
            }
        }
        return ss;
    }
};
