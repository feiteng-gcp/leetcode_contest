/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<Interval> free;
        for (int i = 0; i < avails.size(); ++i)
            for (int j = 0; j < avails[i].size(); ++j)
                free.push_back(avails[i][j]);
        

        auto cmp = [](const Interval& a, const Interval& b) {
            return a.start < b.start || (a.start == b.start && a.end < b.end);
        };
        sort(free.begin(), free.end(), cmp);
        
        vector<Interval> ans;
        int end = free[0].end;
        for (int i = 0; i < free.size() -1; ++i) {
            if (free[i +1].start > end) {
                Interval it;
                it.start = end;
                it.end = free[i +1].start;
                ans.push_back(it);
            }
            end = max(end, free[i +1].end);
        }
        
        return ans;
    }
};
