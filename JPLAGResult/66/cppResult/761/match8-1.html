/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<pair<int,int>> timeline;
        for (auto e : avails) {
            for (auto i : e) {
                timeline.push_back(make_pair(i.start, i.end));
            }
        }
        
        sort(timeline.begin(), timeline.end());
        
        vector<Interval> result;
        
        //for(auto p : timeline) cout << p.first << " " << p.second << ", ";
            
        if (!timeline.empty()) {
            int end = timeline[0].second;
            for (int i = 1; i < timeline.size(); ++i) {
                if (timeline[i].first > end) {
                    result.push_back(Interval(end, timeline[i].first));
                }
                end = max(end, timeline[i].second);
            }
        }
        
        return result;
    }
};
