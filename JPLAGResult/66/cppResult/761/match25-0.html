/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        int len = avails.size();
        
        vector<Node> p;
        
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < avails[i].size(); j++) {
                p.push_back(Node(avails[i][j].start, true));
                p.push_back(Node(avails[i][j].end, false));
            }
        }
        
        vector<Interval> ret;
        sort(p.begin(), p.end());
        int cnt = 1;
        int start = 1;
        while (start < p.size()) {
            if (p[start++].isStart == true) {
                cnt++;
            } else {
                cnt--;
            }
           // cout << cnt << ":" << (cnt == 0) ;
            if (cnt == 0 && start < p.size()) {
             //   cout << " here" << endl;
                ret.push_back(Interval(p[start - 1].t, p[start].t));
            }
        }
        
        return ret;
    }
private:
    class Node {
      public:
        int t;
        bool isStart;
        Node(int tt, bool flag): t(tt), isStart(flag) {};
        bool operator<(const Node & rhs) const{
            return t < rhs.t || (t == rhs.t && isStart == true);
        }
    };
};
