/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) 
    {
        vector<Interval> arr;
        for (auto& intervals : avails)
            for (auto& i : intervals)
                arr.push_back(i);
        sort(arr.begin(), arr.end(), [](const Interval& v1, const Interval& v2) {
            return v1.start < v2.start;

        });

        vector<Interval> ret;
        int max_val = -1;
        for (auto& i : arr)
        {
            if (max_val > 0 && i.start > max_val)
                ret.push_back({max_val, i.start});
            max_val = max(i.end, max_val);
        }
        return ret;
    }
};
