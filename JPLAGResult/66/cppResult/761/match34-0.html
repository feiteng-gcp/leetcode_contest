/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    static bool Cmp(Interval a, Interval b)
    {
        return a.start<b.start;
    }
    vector<Interval> Union(vector<Interval>& a,vector<Interval>& b)
    {
        if(a.size()==0) return b;
        if(b.size()==0) return a;
        vector<Interval> inters(a.begin(),a.end()),res;
        inters.insert(inters.end(),b.begin(),b.end());
        sort(inters.begin(),inters.end(),Cmp);
        int s=inters[0].start,e=inters[0].end;
        for(int i=1;i<inters.size();i++)
        {
            if(inters[i].start<=e)
            {
                e=max(e,inters[i].end);
            }
            else
            {
                Interval tmp(s,e);
                res.push_back(tmp);
                s=inters[i].start;
                e=inters[i].end;
            }
        }
        Interval tmp(s,e);
        res.push_back(tmp);
        return res;
    }
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<Interval> unionedIntervals=avails[0],res;
        for(int i=1;i<avails.size();i++)
        {
            unionedIntervals=Union(unionedIntervals,avails[i]);
        }
        for(int i=0;i<unionedIntervals.size()-1;i++)
        {
            int s=unionedIntervals[i].end,e=unionedIntervals[i+1].start;
            if(s!=e)
            {
                Interval tmp(s,e);
                res.push_back(tmp);
            }
        }
        return res;
    }
};
