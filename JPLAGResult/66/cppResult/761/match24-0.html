/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<Interval> rs;
        if(avails.size()==0)
            return rs;
        vector<Interval> inters;
        for(auto p:avails){
            for(auto q:p)
                inters.push_back(q);
        }
        sort(inters.begin(), inters.end(), [](const auto a, const auto b){return a.start<b.start;});
        int t=inters[0].end;
        for(int i=1;i<inters.size();i++){
            if(inters[i].start>t){
                rs.emplace_back(t, inters[i].start);
                t=inters[i].end;
            }
            else{
                t=max(t, inters[i].end);
            }
        }
        return rs;
    }
};
