/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        map<int, int> mp;
        for (auto &i : avails) {
            for (auto &j : i) {
                mp[j.start]++;
                mp[j.end]--;
            }
        }
        int cur = 0;
        vector<Interval> res;
        int last = 0;
        bool flag = 0;
        for (auto &i : mp) {
            cur += i.second;
            if (cur == 0) {
                last = i.first;
                flag = 1;
            }
            else if (cur>0 && flag == 1){
                res.push_back(Interval(last, i.first));
                flag = 0;
            }
            
        }
        return res;
    }
};
