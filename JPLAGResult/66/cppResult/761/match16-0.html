/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        
        map<int,int> lookup;
        vector<Interval> result;
        
        for(auto avail : avails)
        {
            for(auto currAvail : avail)
            {
                lookup[currAvail.start]++;
                lookup[currAvail.end]--;
            }
        }
        
        //int iCounter = 0;
        int counter = 0;
        bool isOpen = false;
        int start = -1;
        for(auto kv : lookup)
        {
            //iCounter++;
            counter += kv.second;
            if(isOpen)
            {
                //Interval inter(start, kv.first);
                result.push_back(Interval(start, kv.first));
                isOpen = false;
            }
            else if(counter == 0)
            {
                isOpen = true;
                start = kv.first;
            }
        }
        return result;
    }
};
