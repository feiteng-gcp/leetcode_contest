/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) 
    {
        map<int,int>Map;
        for (int i=0; i<avails.size(); i++)
        {
            for (int j=0; j<avails[i].size(); j++)
            {
                int left = avails[i][j].start;
                int right = avails[i][j].end;
                
                int leftBound = left;
                auto pos1=Map.lower_bound(left);
                if (pos1!=Map.begin() && prev(pos1,1)->second >= left)  //左边界部分重合
                leftBound = prev(pos1,1)->first;
    
                int rightBound = right;
                auto pos2=Map.upper_bound(right);
                if (pos2!=Map.begin() && prev(pos2,1)->first <= right)  //右边界部分重合
                rightBound = max(right, prev(pos2,1)->second);
    
                Map.erase(pos1,pos2); // 删除一个前闭后开的迭代器区间
                Map[leftBound]=rightBound;
            }
        }
                
        vector<Interval>result;
        for (auto a:Map)
        {
            if (result.size()!=0)
                result.back().end = a.first;
            result.push_back(Interval(a.second,0));
        }
        
        if (result.size()!=0)
            result.pop_back();
        
        return result;
    }
};
