/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:

    vector<Interval> employeeFreeTime(const vector<vector<Interval>>& avails) {
        map<int, int> t;
        for (auto& p : avails) {
            for (auto& i : p) {
                ++t[i.start];
                --t[i.end];
            }
        }
        int mark = 0;
        vector<Interval> res;
        for (auto p = t.begin(), e = prev(t.end()); p != e; ++p) {
            mark += p->second;
            if (mark == 0) res.emplace_back(p->first, next(p)->first);
        }
        return res;
    }
};
