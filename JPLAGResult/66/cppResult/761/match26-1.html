/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
bool cmp(const Interval &a, const Interval &b)
{
    if(a.start < b.start || (a.start == b.start && a.end <= b.end))
        return true;
    return false;
}
class Solution 
{
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) 
    {
        vector<Interval> res;
        int n = avails.size();
        if(n == 0)
            return res;
        vector<Interval> inv;
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < avails[i].size(); j++)
                inv.push_back(avails[i][j]);
        }
        sort(inv.begin(), inv.end(), cmp);
        //for(int i = 0; i < inv.size(); i++)
            //cout<<inv[i].start<<" "<<inv[i].end<<endl;

        int n1 = inv.size();
        int j = 0;
        for(int i = 1; i < n1; i++)
        {
            if(inv[j].end >= inv[i].start)
                inv[j].end = max(inv[j].end, inv[i].end);
            else
            {
                j++;
                inv[j] = inv[i];
            }
        }
        inv.resize(j + 1);
        //for(int i = 0; i < inv.size(); i++)
          //  cout<<inv[i].start<<" "<<inv[i].end<<" xxx"<<endl;
        n1 = j + 1;
        for(int i = 1; i < n1; i++)
            res.push_back(Interval(inv[i-1].end, inv[i].start));
        return res;
    }
};
