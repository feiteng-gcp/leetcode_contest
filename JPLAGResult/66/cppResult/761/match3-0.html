/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

struct Event {
    int x;
    int type;
    int id;
};

bool operator<(const Event &a, const Event &b) {
    if (a.x != b.x) return a.x < b.x;
    return a.type < b.type;
};


class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        int n = avails.size();
        vector<Event> events;
        
        for (int i = 0; i < n; ++i) {            
            for (auto temp: avails[i]) {
                Event start;
                start.x = temp.start;
                start.type = 0;
                start.id = i;
                events.push_back(start);
                Event end;
                end.x = temp.end;
                end.type = 1;
                end.id = i;
                events.push_back(end);
            };
            
        }
        sort(events.begin(), events.end());
        vector<Interval> ans;
        int m = events.size();
        unordered_set<int> cur;        
        
        for (int i = 0; i < n; ++i) cur.insert(i);
        
        int left = INT_MIN;        
        
        for (int i = 0; i < m; ++i) 
            if (events[i].type == 1) {
                cur.insert(events[i].id);
                if (cur.size() == n) {
                    left = events[i].x;
                };
            }
            else if (events[i].type == 0) {
                if (cur.size() == n) {
                    int right = events[i].x;
                    if (right != left) {
                        if (left != INT_MIN) {
                            Interval temp(left, right);
                            ans.push_back(temp);                            
                        }
                    };                      
                };
                cur.erase(events[i].id);
            };
        
        return ans;
        
        
        
    }
};
