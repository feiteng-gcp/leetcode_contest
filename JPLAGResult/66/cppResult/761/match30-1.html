/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        if (avails.empty()) return vector<Interval>{};
        vector<Interval> res, ins, ans;
        for(int i = 0; i < avails.size(); i++){
            for(int j = 0; j < avails[i].size(); j++){
                ins.push_back(avails[i][j]);
            }
        }
        sort(ins.begin(), ins.end(), [](Interval a, Interval b){return a.start < b.start;});
        res.push_back(ins[0]);
        for (int i = 1; i < ins.size(); i++) {
            if (res.back().end < ins[i].start) res.push_back(ins[i]);
            else
                res.back().end = max(res.back().end, ins[i].end);
        }
        for(int i = 0; i < res.size() - 1; i++) {
            Interval t(res[i].end, res[i+1].start);
            ans.push_back(t);
        }
        return ans;
    }
};
