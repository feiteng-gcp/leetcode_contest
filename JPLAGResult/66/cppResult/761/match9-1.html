/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    
    vector<Interval> merge(vector<Interval>& ins) {
        if (ins.empty()) 
            return vector<Interval>{};
        vector<Interval> res;
        sort(ins.begin(), ins.end(), [](Interval a, Interval b){return a.start < b.start;});
        res.push_back(ins[0]);
        for (int i = 1; i < ins.size(); i++) {
            if (res.back().end < ins[i].start) res.push_back(ins[i]);
            else
                res.back().end = max(res.back().end, ins[i].end);
        }

            return res;
    }
    
    
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        
        vector<Interval> flatList;
        vector<Interval> mergedIntervals;
        vector<Interval> ans;
        
        //create a flatList of all the intervals
        for(int i=0;i<avails.size();i++){
            for(int j=0;j<avails[i].size();j++){
                flatList.push_back(avails[i][j]);
            }
        }
        
        mergedIntervals = merge(flatList);
        Interval tmpAns;
        
        if(mergedIntervals.size()<=1)
            return ans;
        
        for(int i=0;i<mergedIntervals.size()-1;i++){
            Interval a = mergedIntervals[i];
            Interval b = mergedIntervals[i+1];
            tmpAns.start = a.end;
            tmpAns.end = b.start;
            ans.push_back(tmpAns);
        }
        
        return ans;
    }
};
