/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class RangeModule {
public:

	map<int, int> r;

	RangeModule() {

	}

	void merge(map<int, int>::iterator& m, int left, int right) {
		if (m->first > right || m->second < left) {
			return;
		}
		else if (left <= m->first) {
			int nr = max(m->second, right);
			pair<int, int> p = { left, max(m->second,right) };
			r.erase(m++);
			while (m != r.end() && m->first <= right) {
				nr = max(m->second, right);
				r.erase(m++);
			}
			p.second = nr;
			r.insert(p);
		}
		else if (left > m->first && left <= m->second) {
			int nr = max(m->second, right);
			pair<int, int> p = { m->first, nr };
			r.erase(m++);
			while (m != r.end() && m->first <= right) {
				nr = max(m->second, right);
				r.erase(m++);
			}
			p.second = nr;
			r.insert(p);
		}
	}

	void addRange(int left, int right) {
		if (r.empty()) {
			r[left] = right;
			return;
		}
		auto it = r.lower_bound(left);
		if (it == r.begin()) {
			if (it->first > right) {
				r.insert({ left,right });
			}
			else {
				merge(it, left, right);
			}
		}
		else if (it != r.begin()) {
			--it;
			if (it->second >= left) {
				merge(it, left, right);
			}
			else {
				++it;
				if (it == r.end() || it->first > right) {
					r.insert({ left,right });
				}
				else {
					merge(it, left, right);
				}
			}
		}
	}

	bool queryRange(int left, int right) {
		if (r.empty()) return false;
		auto it = r.lower_bound(left);
		if (it != r.begin()) {
			--it;
			if (it->first <= left && it->second >= right) return true;
			++it;
		}
		if (it != r.end()) {
			if (it->first <= left && it->second >= right) return true;
			++it;
		}
		if (it != r.end()) {
			if (it->first <= left && it->second >= right) return true;
		}
		return false;
	}

	void splitRange(map<int, int>::iterator& m, int left, int right) {
		if (m->second < left || m->first > right) {
			m++;
		}
		else if (left <= m->first && m->second <= right) {
			r.erase(m++);
		}
		else if (left > m->first) {
			pair<int, int> p0 = { m->first,left },
				p1 = { right,m->second };
			r.erase(m++);
			r.insert(p0);
			if (right < p1.second) {
				r.insert(p1);
			}
		}
		else if(left <= m->first) {
			pair<int, int> p0 = { right,m->second };
			r.erase(m++);
			r.insert(p0);
		}
	}

	void removeRange(int left, int right) {
		auto it = r.lower_bound(left);
		auto itl = it, itr = it;
		if (it != r.begin()) {
			--it;
		}
		while (it != r.end() && it->first <= right) {
			// this iterates the iterator since the map needs the 
			// r.erase(it++) kludge when deleting from a map
			splitRange(it, left, right);
		}
	}

};


class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& a) {
        // re-using code from my RangeModule problem
        if(a.size()==0) return {};
        RangeModule rm;
        rm.addRange(-2,100000002);
        for(int i=0;i<a.size();++i){
            for(int j=0;j<a[i].size();++j){
                rm.removeRange(a[i][j].start,a[i][j].end);
            }
        }
        vector<Interval> v;
        for(auto x:rm.r){
            if(x.first==-2 || x.second==100000002) continue;
            v.push_back(Interval(x.first,x.second));
        }
        return v;
    }
};
