/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        map<int, int> mp;
        for (auto& emp : avails) {
            for (auto& inter : emp) {
                ++mp[inter.start];
                --mp[inter.end];
            }
        }
        int cnt = 0;
        vector<int> res;
        for (auto& kv : mp) {
            if (cnt == 0) res.push_back(kv.first);
            cnt += kv.second;
            if (cnt == 0) res.push_back(kv.first);
        }
        vector<Interval> ans;
        for (int i = 1; i < res.size() - 1; i += 2) {
            ans.push_back({res[i], res[i+1]});
        }
        return ans;
    }
};
