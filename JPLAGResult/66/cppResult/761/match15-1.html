/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
	vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
		int l, i, n, s = 0;
		vector<pair<int, int>> intvs;
		for (auto vec : avails) {
			l = vec.size();
			for (i = 0; i < l; i++)
				intvs.push_back({ vec[i].start, 1 }), intvs.push_back({ vec[i].end, -1 });			
		}
		sort(intvs.begin(), intvs.end());
		n = intvs.size();
		bool start = false;
		int i1, i2;
		vector<Interval> res;
		for (i = 0; i < n; i++) {
			int t = intvs[i].first;
			s += intvs[i].second;

			while (i + 1 < n && intvs[i + 1].first == t) s += intvs[++i].second;
			if (!s && !start) { start = true; i1 = t; }
			else if (s && start) { start = false; res.push_back(Interval(i1, t)); }
		}
		return res;
	}
};
