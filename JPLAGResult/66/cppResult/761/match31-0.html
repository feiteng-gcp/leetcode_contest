/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector <Interval> r;
        map<int, int> s;
        
        for (int i = 0; i < avails.size(); ++i) {
            for (int j = 0; j < avails[i].size(); ++j) {
                auto a = avails[i][j];
                s[a.start] = s[a.start] < a.end ? a.end : s[a.start];
            }
        }
        
        while(!s.empty()) {
            auto start = s.begin();
            auto end = s.end();
            int cur_end = start->second;

            while(true) {
                end = s.upper_bound(cur_end);
                int next_end = cur_end;
                for (auto it = start; it != end; ++it) {
                    if (it->second > next_end) {
                        next_end = it->second;
                    }
                }
                cout << "cur: " << cur_end << ", next:" << next_end << endl;
                if (cur_end != next_end) {
                    cur_end = next_end;
                    continue;
                }

                break;
            }
            cout << "start: " << start->first << ", end: " << cur_end << endl;
            r.push_back({start->first, cur_end});
            s.erase(start, end);
        }
        
        if (r.size() < 2) {
            return {};
        }
        
        vector <Interval> result;

        for (int i = 1; i < r.size(); ++i) {
            if (r[i].start != r[i-1].end) {
                result.push_back({r[i-1].end, r[i].start});
            }
        }
        
        return result;
    }
};
