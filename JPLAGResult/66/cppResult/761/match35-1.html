/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
     vector<Interval> employeeFreeTime(vector<vector<Interval>> avails) {
        vector<Interval> tmp;
        for (auto& ls : avails) for (auto& l : ls) tmp.push_back(l);
        tmp = merge(tmp);
        vector<Interval> res;

        for (int i = 0; i < tmp.size()-1; i++) {
            res.push_back({ tmp[i].end, tmp[i+1].start});
        }
        return res;
    }

    vector<Interval> merge(vector<Interval> ls) {
        int n = ls.size();
        if (n < 2) return ls;
        sort(ls.begin(), ls.end(), [](Interval& l1, Interval& l2) { return l1.start < l2.start; });
        vector<Interval> res;
        auto pre = ls[0];
        for (int i = 1; i < n; i++) {
            auto cur = ls[i];

            if (pre.end < cur.start) {
                res.push_back(pre);
                pre = cur;
            }
            else {
                pre.start = min(pre.start, cur.start);
                pre.end = max(pre.end, cur.end);
            }
        }
        res.push_back(pre);
        return res;
    }
};
