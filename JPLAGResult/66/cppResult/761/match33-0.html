/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        map<int, int> mp;
        for (const auto& avail : avails)
        for (const auto& itv : avail)
        {
            auto it = mp.upper_bound(itv.start);
            if (it == mp.begin())
            {
                auto ret = mp.insert({itv.start, itv.end});
                it = ret.first;
            }
            else
            {
                --it;
                if (it->second < itv.start)
                {
                    auto ret = mp.insert({itv.start, itv.end});
                    it = ret.first;
                }
                else
                {
                    it->second = max(it->second, itv.end);
                }
            }
            for (auto it2 = next(it); it2 != mp.end() && it2->first <= it->second;)
            {
                it->second = max(it->second, it2->second);
                it2 = mp.erase(it2);
            }
        }
        vector<Interval> res;
        for (auto it1 = mp.begin(), it2 = next(it1); it2 != mp.end(); ++it1, ++it2)
            res.push_back({it1->second, it2->first});
        return res;
    }
};
