/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        map<int, int> event;
        for (auto & times : avails)
            for (auto& itv : times) {
                ++event[itv.start];
                --event[itv.end];
            }
        
        int cnt = 0, pre = INT_MIN;
        vector<Interval> ans;
        for (auto& kv : event) {
            // cout << kv.first << " " << kv.second << endl;
            if (cnt==0 && kv.second>0) {
                if (pre!=INT_MIN) ans.push_back(Interval(pre, kv.first));
            } else if (cnt>0 && cnt+kv.second==0) 
                pre = kv.first;
            cnt += kv.second;
        }
        
        return ans;
    }
};
