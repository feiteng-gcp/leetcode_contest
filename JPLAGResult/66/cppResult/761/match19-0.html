/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<pair<int, int>> data;
        for(int i = 0; i < avails.size(); i ++){
            for(int j = 0; j < avails[i].size(); j ++){
                Interval in = avails[i][j];
                data.push_back(pair<int, int>(in.start, 1));
                data.push_back(pair<int, int>(in.end, -1));
            }
        }
        sort(data.begin(), data.end());
        int count = 0;
        int pre = -1;
        vector<Interval> ret;
        Interval ii;
        for(int i = 0; i < data.size(); i ++){
            pair<int, int> cur = data[i];
            pre = count;
            if(cur.second == -1){
                count --;
            }else count ++;
            if(pre == 1 && count == 0){
                //start
                ii = Interval();
                ii.start = cur.first;
            }
            if(i > 1 && pre == 0 && count == 1){
                ii.end = cur.first;
                if(ii.start != ii.end)
                    ret.push_back(ii);
            }
        }
        return ret;
    }
};
