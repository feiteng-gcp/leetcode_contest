/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
struct point {
    int val, ide, idt;
    bool ifstart;
    point(int _val, int _ide, int _idt, bool _ifstart):
        val(_val), ide(_ide), idt(_idt), ifstart(_ifstart) {};
};

bool operator < (point &a, point &b) {
    return a.val < b.val || (a.val == b.val && a.ifstart);
}


class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<point> a;
        int n = avails.size();
        for (int i = 0;i < n;++i) {
            for (int j = 0;j < avails[i].size();++j) {
                a.push_back(point(avails[i][j].start, i, j, true));
                a.push_back(point(avails[i][j].end, i, j, false));
            }
        }
        sort(a.begin(), a.end());
        
        vector<Interval> ans;
        int tot = n, last = INT_MIN;
        for (auto & p: a) {
            if (p.ifstart) {
                if (tot == n && last != INT_MIN && last < p.val) {
                    ans.push_back(Interval(last, p.val));
                }
                --tot;
            }else {
                ++tot;
                last = max(last, p.val);
            }
        }
        
        return ans;
    }
};
