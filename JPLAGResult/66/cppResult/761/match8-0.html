/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        map<int, int> T;
        for (auto employee : avails) {
            for (auto interval : employee) {
                T[interval.start]++;
                T[interval.end]--;
            }
        }
        int cnt = 0;
        Interval tmp;
        bool fill = false;
        vector<Interval> ans;
        for (auto entry : T) {
            cnt += entry.second;
            if (cnt == 0) {
                tmp.start = entry.first;
                fill = true;
            }
            if (cnt > 0 && fill) {
                tmp.end = entry.first;
                ans.push_back(tmp);
                fill = false;
            }
        }
        return ans;
    }
};
