/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:

    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
      std::map<int, int> timeline;
      for (int emp = 0; emp < avails.size(); ++emp) {
      	for (int intervalIndex = 0; intervalIndex < avails[emp].size(); ++intervalIndex) {
      		const auto& interval = avails[emp][intervalIndex];
      		auto res = timeline.insert(std::make_pair(interval.start, 1));
      		if (res.second == false) {
      			++timeline[interval.start];
      		}

      		res = timeline.insert(std::make_pair(interval.end, -1));
			     if (res.second == false) {
      			--timeline[interval.end];
      		}      		
      	}
      }


	
	    vector<Interval> result;
      const int invalid = 200000000;

      int curActive = 0;
      int inactiveTimeStart = invalid;
      for (auto it = timeline.begin(); it != timeline.end(); ++it) {
      	if (inactiveTimeStart != invalid) {
      		if (inactiveTimeStart - it->first != 0) {
      			result.push_back(Interval{inactiveTimeStart, it->first});
      		}

      		inactiveTimeStart = invalid;
      	}
      	curActive += it->second;
      	if (curActive == 0) {
      		inactiveTimeStart = it->first;
      	}
      }

      return result;
    }
};
