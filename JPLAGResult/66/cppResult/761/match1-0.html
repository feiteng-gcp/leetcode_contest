/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<pair<int,int>> tmp;
        for (auto vec : avails) {
            for (auto v : vec) {
                tmp.push_back(make_pair(v.start, v.end));
            }
        }
        
        sort(tmp.begin(), tmp.end());
        vector<pair<int,int>> merged;
        if (tmp.size() > 0) merged.push_back(tmp[0]);
        for (int i = 1; i < tmp.size(); ++i) {
            pair<int, int>& top = merged[merged.size()-1];
            auto curr = tmp[i];
            if (curr.first <= top.second) {
                top.second = max(curr.second, top.second);
            }
            else {
                merged.push_back(curr);
            }
        }
        
        vector<Interval> result;
        for (int i = 0; i < merged.size()-1; ++i) {
            int left = merged[i].second;
            int right = merged[i+1].first;
            if (left < right) {
                result.push_back(Interval(left, right));
            }
        }
        return result;
    }
};
