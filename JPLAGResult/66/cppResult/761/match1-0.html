/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<pair<int, int> > v;
        for(int i = 0; i < avails.size(); i++) {
            for(int j = 0; j < avails[i].size(); j++) {
                v.push_back(make_pair(avails[i][j].start, avails[i][j].end));
            }
        }
        
        sort(v.begin(), v.end());
        vector<pair<int, int> > a;
        int l = v[0].first, r = v[0].second;
        for(int i = 1; i < v.size(); i++) {
            if(v[i].first <= r) r = max(r, v[i].second);
            else {
                a.push_back(make_pair(l, r));
                l = v[i].first;
                r = v[i].second;
            }
        }
        
        a.push_back(make_pair(l, r));
        vector<Interval> ans;
        for(int i = 0; i + 1 < a.size(); i++) {
            int lf = a[i].second;
            int rg = a[i + 1].first;
            if(lf != rg) ans.push_back(Interval(lf, rg));
        }
        
        return ans;
    }
};
