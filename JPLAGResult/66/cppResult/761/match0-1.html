/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        map<int, int> mp;
        for (auto emp: avails) for (auto x: emp) {
            mp[x.start]++; mp[x.end]--;
        }
        int l = -1, s = 0;
        vector<Interval> R;
        for (auto pr: mp) {
            if (s == 0 && l != -1 && pr.second) {
                R.push_back(Interval(l, pr.first));
            }
            s += pr.second;
            l = s ? -1 : pr.first;
        }
        return R;
    }
};
