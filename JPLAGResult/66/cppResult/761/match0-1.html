/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<Interval> ans;
        int n = avails.size();
        vector<int> points(n, 0);
        int last = -1;
        while (true) {

            bool check = false;
            int target = -1;
            for (int i = 0; i < n; i++) {
                if (points[i] >= avails[i].size()) { continue; }
                if (target == -1 || avails[i][points[i]].start < avails[target][points[target]].start) {
                    target = i;
                    check = true;
                }
            }
            if (!check) { break; }
            if (last == -1) {
                last = avails[target][points[target]].end;
            } else {
                if (last < avails[target][points[target]].start) {
                    Interval temp(last, avails[target][points[target]].start);
                    ans.push_back(temp);
                }
            }
            last = max(last, avails[target][points[target]].end);
            points[target]++;
        }
        return ans;
    }
};
