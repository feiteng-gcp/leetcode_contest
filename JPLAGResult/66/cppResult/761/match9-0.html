/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    static int cmp(Interval a, Interval b){
        if(a.start==b.start)return a.end<b.end;
        return a.start<b.start;
    }
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        //先合在一起
        vector<Interval>allnums;
        for(int i=0;i<avails.size();i++){
            for(int j=0;j<avails[i].size();j++){
                allnums.push_back(avails[i][j]);
            }
        }
        sort(allnums.begin(),allnums.end(),cmp);
        vector<Interval>nums;
        //vector<Interval>tmp;
        //tmp.start=allnums[0].start;
        //tmp.start=allnums[0].end;
        nums.push_back(Interval(allnums[0].start, allnums[0].end));
        int index=0;
        for(int i=1;i<allnums.size();i++){
            if(allnums[i].start<=nums[index].end){
                if(allnums[i].end>nums[index].end){      
                    nums[index].end=allnums[i].end;
                }
            }else{
                nums.push_back(Interval(allnums[i].start, allnums[i].end));
                index++;
            }
        }
        vector<Interval>result;
        for(int i=0;i<nums.size();i++){
            if(i==0){
                ;
                //if(nums[i].start>1){
                //    result.push_back(Interval(1,nums[i].start));
                //}
            }else{
                if(nums[i].start>nums[i-1].end){
                    result.push_back(Interval(nums[i-1].end,nums[i].start));
                }
            }
        }
        return result;
        
    }
};
