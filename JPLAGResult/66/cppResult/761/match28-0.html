/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        if (avails.empty()) return {};
        vector<Interval> total;
        for (auto& vec : avails) {
            for (auto& item : vec) {
                total.emplace_back(item);
            }
        }
        sort(total.begin(), total.end(), [](const Interval& a, const Interval& b) {
            if (a.start == b.start) {
                return a.end < b.end;
            }
            return a.start < b.start;
        });
        vector<Interval> result;
        result.emplace_back(total[0]);
        for (int i = 1; i < total.size(); ++i) {
            if (total[i].start <= result.back().end) {
                result.back().end = max(result.back().end, total[i].end);
            } else {
                result.emplace_back(total[i]);
            }
        }
        // for (auto& item : result) {
        //     cout << item.start << ' ' << item.end << endl;
        // }
        vector<Interval> em;
        for (int i = 1; i < result.size(); ++i) {
            int left = result[i - 1].end;
            int right = result[i].start;
            if (right - left > 0) {
                em.push_back({left, right});
            }
        }
        return em;
    }
};
