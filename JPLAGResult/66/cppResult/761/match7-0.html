/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<pair<int, int> > T;
        for (int i = 0; i <avails.size(); i++) {
            for (int j = 0 ; j < avails[i].size(); j++) {
                T.push_back(make_pair(avails[i][j].start, 1));
                T.push_back(make_pair(avails[i][j].end, 0));
            }
        }
        sort(T.begin(), T.end());
        int prev = INT_MIN, cnt = 0;
        vector<Interval> res;
        for (int i = 0; i < T.size(); i++) {
            if(T[i].second == 1) {
                cnt++;
                if (cnt == 1 && prev != INT_MIN && prev != T[i].first) {
                    res.push_back(*new Interval(prev, T[i].first));
                }
            } else {
                cnt--;
                prev = T[i].first;
            }
        }
        return res;
    }
};
