/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        vector<Interval> ret;
        if(avails.empty())
        {
            return ret;
        }
        vector<pair<int, int>> m;
        for(auto& em : avails)
        {
            for(auto& t : em)
            {
                m.push_back({t.start, t.end});
            }
        }
        
        sort(m.begin(), m.end());
        int curS = m[0].first;
        int curE = m[0].second;
        for(int i = 1; i < m.size(); ++i)
        {
            if(m[i].first <= curE)
            {
                curE = max(curE, m[i].second);
            }
            else
            {
                Interval inter(curE, m[i].first);
                ret.push_back(inter);
                curS = m[i].first;
                curE = m[i].second;
            }
        }
        
        return ret;
    }
};
