/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */

class Solution {
public:
    vector<Interval> employeeFreeTime(vector<vector<Interval>>& avails) {
        map<int, int> timeline;
        for (const auto& avail: avails)
            for (const auto& interval: avail){
                timeline[interval.start]++;
                timeline[interval.end]--;
        }
        int count=0, last=-1;
        vector<Interval> freeTime;
        for (const auto& time: timeline) {
            if (count==0 && last>=0 && last<time.first) freeTime.push_back(Interval(last, time.first));
            count+=time.second;
            if (count==0) last=time.first;
        }
        return freeTime;
    }
};
