class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<bool> bold(S.size(), false);
        for (int i=0; i<S.size(); i++)
            for (const auto& word: words) if (i+word.size()<=S.size())
                if (S.substr(i,word.size())==word)

                    for (int j=0; j<word.size(); j++) bold[i+j]=true;
        string ans="";
        bool b=false;
        for (int i=0; i<S.size(); i++) {
            if (b && !bold[i]) ans+="</b>";
            if (!b && bold[i]) ans+="<b>";
            ans+=S[i];
            b=bold[i];
        }
        if (b) ans+="</b>";
        return ans;
    }
};
