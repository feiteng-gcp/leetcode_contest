class Solution {
public:
    string boldWords(vector<string>& words, string S) 
    {
        unordered_set<string> dict;
        for (auto &str : words) dict.insert(str);
        int n = S.length();
        vector<int> dp(n, -1);
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j <= i; j++)
            {
                string str = S.substr(j, i-j+1);
                if (dict.count(str))
                {
                    dp[j] = i;
                    break;
                }
            }
        }
        int m = -1;
        string re = "";
        bool open = false;
        for (int i = 0; i < n; i++)
        {
            if (dp[i] >= 0) m = max(m, dp[i]);
            if (i <= m)
            {
                if (!open)
                {
                    open = true;
                    re += "<b>";
                }
            }
            else
            {
                if (open)
                {
                    open = false;
                    re += "</b>";
                }
            }
            re += S[i];
        }
        if (open) re += "</b>";
        return re;
    }
};
