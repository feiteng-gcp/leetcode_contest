 class Solution {
 public:
     string boldWords(vector<string>& words, string S) {
        vector<int> bold(S.size(),0);
        for(string word: words){
            for(int i=0;i+word.size() <= S.size();i++){
                if(S.substr(i, word.size()) == word){
                    for(int j=i;j<i+word.size();j++) bold[j] =1;
                }
            }
        }
        
        string res = "";
        for(int i=0;i<S.size();){
            if(bold[i]){
                res = res + "<b>";
                while(i < S.size() && bold[i]) res = res + S[i++];
                res = res + "</b>";
            }else{
                res = res + S[i++];
            }
        }
        return res;
        
     }
} ;
