class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<vector<string>> Map(26, vector<string>());
        for (auto &word : words) {
            Map[word[0] - 'a'].push_back(word);
        }
        // cout<<Map[0][0]<<"   "<<Map[1][0]<<endl;
        vector<bool> bolded(S.size(), false);
        for (int i = 0; i < S.size(); i++) {
            if (Map[S[i] - 'a'].empty()) continue;
            // cout<<i<<endl;

            for (int j = 0; j < Map[S[i] - 'a'].size(); j++) {
                // cout<<Map[S[i] - 'a'][j]<<"    "<<S.substr(i, Map[S[i] - 'a'][j].size())<<endl;
                if (Map[S[i] - 'a'][j] == S.substr(i, Map[S[i] - 'a'][j].size())) {
                    for (int k = 0; k < Map[S[i] - 'a'][j].size(); k++) {
                        bolded[i + k] = true;
                    }
                }
            }
        }
        // cout<<bolded[0]<<bolded[1]<<bolded[2]<<bolded[3]<<bolded[4]<<endl;
        for (int c = bolded.size() - 1; c >= 0; c--) {
            if (bolded[c] && (c == bolded.size() - 1 || !bolded[c + 1])) S.insert(c + 1, "</b>");
            if (bolded[c] && (c == 0 || !bolded[c - 1])) S.insert(c, "<b>");
        }
        return S;
    }
};
