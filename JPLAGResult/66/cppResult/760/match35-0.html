class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n = S.size();
        vector<pair<int, int> > v;
        for(int i = 0; i < n; i++) {
            int mx = -1;
            for(int j = 0; j < words.size(); j++) {
                if(i + words[j].size() - 1 >= n) continue;
                bool ok = true;
                for(int k = 0; k < words[j].size(); k++) if(S[i + k] != words[j][k]) ok = false;
                if(ok) mx = max(mx, i + int(words[j].size()) - 1);
            }
            
            if(mx != -1) v.push_back(make_pair(i, mx + 1));
        }
        
        if(v.size() == 0) return S;
        vector<int> open(555, 0), close(555, 0);
        int l = v[0].first, r = v[0].second;
        for(int i = 1; i < v.size(); i++) {
            if(v[i].first <= r) r = max(r, v[i].second);
            else {
                open[l] = 1;
                close[r] = 1;
                l = v[i].first;
                r = v[i].second;
            }
        }
        
        open[l] = close[r] = 1;
        
        string res = "";
        for(int i = 0; i < n + 1; i++) {
            if(open[i]) res += "<b>";
            if(close[i]) res += "</b>";
            if(i >= 0 && i < n) res += S[i];
            //if(close[i] && !open[i + 1]) res += "</b>";
            //if(close[i] && !open[i + 1]) 
        }
        
        return res;
    }
};
