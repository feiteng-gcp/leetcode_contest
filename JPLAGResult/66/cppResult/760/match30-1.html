class Solution {
public:
    string boldWords(vector<string>& words, string S) {        
        vector<bool> b(S.size(), 0);
        unordered_set<string> dic(words.begin(), words.end());
        
        for(int i = 0; i < S.size(); ++i)
        {
            for(int j = 1; i + j <= S.size(); ++j)
            {
                string t = S.substr(i, j);
                if (dic.count(t))
                {
                    for(int k = i; k < i + j; ++k)

                        b[k] = 1;
                }
            }            
        }
        
        string ans;
        bool bb = 0;
        for(int i = 0; i < S.size(); ++i)
        {
            if (b[i])
            {
                if (!bb)

                    ans += "<b>";                
                bb = 1;
            }
            else if (bb)
            {
                bb = 0;
                ans += "</b>";
            }
            
            ans += S[i];
        }
        
        if (bb)
            ans += "</b>";
        
        return ans;
    }
};
