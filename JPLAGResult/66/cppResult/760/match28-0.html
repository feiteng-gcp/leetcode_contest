class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<bool> boldMap(S.size(),false);
        int maxLen;
        bool isMatch;
        for(int i=0 ; i<S.size() ; i++){
            maxLen = 0;
            for(int j=0 ; j<words.size() ; j++){
                isMatch = true;
                for(int k=0 ; k<words[j].size() ; k++){
                    if(S[i+k] != words[j][k]){
                        isMatch = false;
                    }
                }
                if(isMatch and maxLen<words[j].size()) maxLen = words[j].size();
            }
            for(int j=i ; ((j<i+maxLen)and j<S.size()) ; j++)
                boldMap[j] = true;
        }

        /*
        for(int i=0 ; i<S.size() ; i++){
            cout << boldMap[i] << "-";
        }
        cout << endl;
        */
        string ans;
        bool boldOn = false;
        for(int i=0 ; i<boldMap.size() ; i++){
            if(boldMap[i]){
                if(!boldOn) {
                    boldOn = true;
                    ans += "<b>";
                }
                ans += S[i];
            }
            else{
                if(boldOn){
                    boldOn = false;
                    ans+= "</b>";
                }
                ans += S[i];
            }
        }
        if(boldOn)
            ans+= "</b>";
        return ans;
    }
};
