class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        string result;
        
        bool* array = new bool[S.size()];
        memset(&array[0], false, sizeof(bool) * S.size());
        
        int pos = 0;
        for (const auto &w : words)
        {
            pos = 0;
            while ((pos = S.find(w, pos)) != string::npos)
            {
                memset(&array[pos], true, sizeof(bool) * w.size());
                ++pos;
            }
        }
        
        bool add = false;
        for (int i = 0; i < S.size(); ++i)
        {
            if (array[i] && !add)
            {
                result += "<b>";
                add = true;
            }
            if (!array[i] && add)
            {
                add = false;
                result += "</b>";
            }
            result += S[i];
        }
        
        if (add)
        {
            result += "</b>";
        }
        
        delete[] array;
        array = 0;
        
        return result;
    }
};
