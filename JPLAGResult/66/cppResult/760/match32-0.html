class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        if (S.size() == 0) {
            return "";
        }
        bool indexes[S.size()] = {false};
        for(auto word: words) {
            for( int i = 0; i < S.size() - word.size() + 1; i++) {    
                if(S.substr(i, word.size()) == word) {
                    for(int k = i; k < i+ word.size(); k++) {
                        indexes[k] = true;
                    }
                }
            }
        }
        string res = "";
        for( int i = 0; i < S.size(); i++) {
            if (indexes[i] == true) {
                res += "<b>";
                res += S[i];
                i++;
                while(i < S.size() && indexes[i] == true) {
                    res += S[i];
                    i++;
                }
                res += "</b>";
                i--;
            } else {
                res += S[i];
            }
            
        }
        return res;
    }
};
