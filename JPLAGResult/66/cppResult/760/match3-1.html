class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        sort(words.begin(), words.end());
        reverse(words.begin(), words.end());
        bool bold[S.size() + 1];
        memset(bold, false, sizeof(bold));

        int last_idx = 0, len;
        for (int i = 0; i < S.size(); i++) {
            bool match = false;
            for (int j  = 0; j < words.size(); j++) {
                int pos = 0;
                while(i + pos < S.size() && pos < words[j].size() && S[i + pos] == words[j][pos]) pos++;
                if (pos < words[j].size()) continue;
                match = true;
                len = words[j].size();
                break;

            }
            if (!match)  {
                continue;
            }
            for (int j = 0; j < len; j++)   {
                bold[i + j] = true;
            }
            
        }
        string res = "";

        bool outside = true;
         for (int i = 0; i < S.size(); i++) {
            if (bold[i] && outside) {
                res += "<b>";
                outside = false;
            } else if (!bold[i] && !outside) {
                res += "</b>";
                outside = true;
            }
             res += S[i];
         }
        if (!outside) res += "</b>";
        return res;
    }

};
