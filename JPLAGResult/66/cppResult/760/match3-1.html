class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n = S.size();
        if (n == 0)
            return "";
        vector<int> bold(n, 0);

        for (auto word : words) {
            for (int i = 0; i <= n - word.size(); i++) {
                if (S.substr(i, word.size()) == word) {
                    for (int j = i; j < i + word.size(); j++) {
                        bold[j] = 1;
                    }
                }
            }
        }
        string result = "";
        if (bold[0] == 1) {
            result += "<b>";
            
        }
        result.push_back(S[0]);
        for (int i = 1; i < n; i++) {
            if (bold[i - 1] == 1 && bold[i] == 0) {
                result += "</b>";
                result.push_back(S[i]);
            } else
            if (bold[i - 1] == 0 && bold[i] == 1) {
                result += "<b>";
                result.push_back(S[i]);
            } else {
                result.push_back(S[i]);
            }
        }
        if (bold[n - 1]) {
            result += "</b>";
        }
        return result;
    }
};
