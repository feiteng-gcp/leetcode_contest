class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        map<string, int> d;
        for (auto x : words) d[x] = x.size();
        int n = S.size();
        vector<int> st(n, 0), ed(n + 1, 0);

        for (int i = 0; i < n; ++i) {
            for (int j = i; j < n; ++j) {
                string sub = S.substr(i, j - i + 1);
                if (d.count(sub)) {
                    st[i]++;
                    ed[j + 1]++;
                }
            }
        }
        string ans;

        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            if (st[i]) {
                if (cnt == 0) ans += "<b>";
                cnt += st[i];
            }
            if (ed[i]) {
                cnt -= ed[i];
                if (cnt == 0) ans += "</b>";
            }
            ans += S[i];
        }
        if (ed[n])
            ans += "</b>";
        return ans;
    }
};
