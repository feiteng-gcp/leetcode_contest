class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        if(S.size() == 0) return S;
        vector<bool> bold(S.size());
        for(auto& t : words)
            bolden(S, t, bold);
        
        string ret = "";
        for(int i=0;i<S.size();i++){
            if(bold[i] == 1){
                if(i == 0 || !bold[i-1]) ret += "<b>";
            }
            else{
                if(i > 0 && bold[i-1]) ret += "</b>";
            }
            ret += S[i];
        }
        if (bold[S.size() - 1]) ret += "</b>";
        
        return ret;
    }
private:
    void bolden(string& s, string& t, vector<bool>& mark){
        size_t found = s.find(t);
        while(found != string::npos){
            for(int i=found;i<found+t.size();i++){
                mark[i] = true;
            }
            found = s.find(t, found+1);
        }
    }
};
