class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n = S.size();
        vector<bool> bold(n, 0);
        for (auto &w : words) {
            int size = w.size();
            for (int i = 0; i <= n-size; i++) {
                if (S.substr(i,size) == w) {
                    for (int j = i; j < i+size; j++) {
                        bold[j] = 1;
                    }
                }
            }
        }
        // cout<<"fff"<<endl;
        string res;
        if (bold[0] == 1) {
            res += "<b>";
        }
        res += S[0];
        for (int i = 1; i < n; i++) {
            if ((bold[i-1] == 1 && bold[i] == 1) || (bold[i-1] == 0 && bold[i] == 0)) {
                res += S[i];
            }
            else if (bold[i-1] == 1 && bold[i] == 0) {
                res += "</b>";
                res += S[i];
            }
            else {
                res += "<b>";
                res += S[i];
            }
        }
        if (bold[n-1] == 1) {
            res += "</b>";
        }
        return res;
        
    }
};
