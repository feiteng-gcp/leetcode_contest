class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        unordered_set<string> mm(words.begin(), words.end());
        int N = S.size();
        vector<int> index(N, 0);
        for(int i = 0; i < N; i ++){
            for(int j = 1; j <= N; j ++){
                if(i + j <= N){
                    string str = S.substr(i, j);
                    if(mm.find(str) != mm.end()){
                        for(int k = i; k < i + j; k ++){
                            index[k] = 1;
                        }
                    }
                }
            }
        }
        index.insert(index.begin(), 0);
        index.push_back(0);
        for(int i = 0; i < index.size(); i ++){
            cout << index[i] << '\t';
        }
        cout <<endl;
        string ret = "";
        for(int i = 0; i < N + 2; i ++){
            if(i != 0 && i != N + 1) ret.append(1, S[i - 1]);
            if(index[i] == 0){
                
                if(i < N + 1 && index[i + 1] == 1){
                    ret.append("<b>");
                }
            }else{
                
                if(i < N + 1 && index[i + 1] == 0){
                    ret.append("</b>");
                    
                }
                
            }
        }
        
        return ret;
    }
};
