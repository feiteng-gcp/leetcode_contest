class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<pair<int, int>> f;
        for (int i = 0; i < S.length(); i++) {
            for (const auto& iter : words) {
                if (iter.length() <= i + 1 && S.substr(i - iter.length() + 1, iter.length()) == iter) {
                    f.push_back(make_pair(i - iter.length() + 1, i));

                }
            }
        }
        sort(f.begin(), f.end(), [](const pair<int, int>& a, const pair<int, int>& b) {
            return (a.first < b.first);
        });
        string ans;
        int cur = 0;
        for (int i = 0; i < S.length(); i++) {
            if (cur >= f.size() || i < f[cur].first) {
                ans += S[i];
            } else {
                int last = f[cur].second;
                while (cur < f.size() && f[cur].first <= last + 1) {
                    last = max(last, f[cur++].second);
                }
                ans += "<b>" + S.substr(i, last - i + 1) + "</b>";
                i = last;
            }
        }
        return ans;
    }
};
