class Solution {
public:
    string boldWords(vector<string>& words, string s) {
        if(s.size()==0)return s;
        vector<bool>visited(s.size(),0);
        
        for(int i=0;i<words.size();i++){
            string h=words[i];
            
            for(int j=0;j<s.size();j++){
                if(h[0]==s[j]){
                    int pos=0;
                    while(pos<h.size() && j+pos<s.size()&& h[pos]==s[j+pos]){
                        pos++;
                    }
                    
                    if(pos==h.size()){
                        for(int k=j;k<j+h.size();k++)
                            visited[k]=1;
                    }
                }
            }
            
        }
        
        

        string dev="";
        
        for(int i=0;i<s.size();i++){
            if(visited[i]==0){
                dev+=s[i];
            }else{

                dev+="<b>";
                while(i<s.size() && visited[i]==1){
                    dev+=s[i];
                    i++;
                }
                dev+="</b>";
                i--;
            }
        }
        
        return dev;
    }
};
