class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        // violent
        int k[510] = {0};
        for (auto w: words) {
            for (int i = 0; i < S.length() - w.length() + 1; ++i) {
                
                bool match = true;
                for (int j = 0; j < w.length(); ++j) {
                    if (S[i + j] != w[j]) {
                        match = false;
                        break;
                    }
                }
                if (match) {
                    for (int j = 0; j < w.length(); ++j) {
                        k[i + j] = 1;
                    }
                }
            }
        }
        
        string ans;
        bool isink = false;
        for (int i = 0; i < S.length(); ++i) {
            cout << k[i];
            if (!isink && k[i]) {
                ans += "<b>";
                isink = true;
            }
            else if (isink && !k[i]) {
                ans += "</b>";
                isink = false;
            }
            ans += S[i];
        }
        if (isink) {
            ans += "</b>";
        }
        return ans;
    }
};
