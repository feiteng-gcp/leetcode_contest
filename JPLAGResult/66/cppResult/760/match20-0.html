class Solution {
public:
  string boldWords(vector<string>& words, string S) {
    int n = S.length();
    vector<bool> bold(n, false);
    for (auto& word: words) {
      int l = word.length();
      for (int i = 0; i <= n - l; ++i) {
        if (S.substr(i, l) == word) {
          for (int j = i; j < i + l; ++j) {
            bold[j] = true;
          }
        }
      }
    }
    string ans = "";
    for (int i = 0; i < n;) {
      if (!bold[i]) {
        ans += S[i];
        ++i;
      } else {
        ans += "<b>";
        for (; i < n && bold[i]; ++i) {
          ans += S[i];
        }
        ans += "</b>";
      }
    }
    return ans;
  }
};
