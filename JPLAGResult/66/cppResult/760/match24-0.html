class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int size = S.size();
        bool match[size] = {false};
        for(int i = 0; i < words.size(); i++){
            for(int j = 0; j <= size - words[i].size(); j++){
                if(S.substr(j,words[i].size()) == words[i]){
                    for(int k = j; k < j + words[i].size(); k++){
                        match[k] = true;
                    }
                }
            }
        }
        string output = "";
        for(int i = 0; i < size;){
            if(match[i]){
                output += "<b>";
                output += S[i];
                i++;
                while(i < size && match[i] == true){
                    output += S[i];
                    i++;
                }
                output += "</b>";
            } else {
                output += S[i];
                i++;
            }
        }
        return output;
    }
};
