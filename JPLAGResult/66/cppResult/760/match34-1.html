class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        string result ;
        int slen = S.size() ;
        cout << slen << endl ;
        int save[slen] ;
        memset(save, 0, sizeof(save)) ;
        int num = words.size() ;
        for(int iter = 0 ; iter < num ; iter++){
            string b = words[iter] ;
            int len = b.size() ;
            for(int i = 0 ; i+len <= slen ; i++){
                string pick = S.substr(i, len) ;
                //cout << pick << endl ;

                if(pick==b){
                    for(int j = 0 ; j < len ; j++)
                        save[i+j] = 1 ;
                }
            }
        }
        bool flag = 0 ;
        for(int i = 0 ; i < slen ; i++)
        {
            if(flag==0){
                if(save[i]==0)
                    result += S[i] ;
                else{
                    result += "<b>" ;
                    result += S[i] ;
                    flag = 1 ;
                }
            }
            else{
                if(save[i]==0){
                    result += "</b>" ;
                    result += S[i] ;
                    flag = 0 ;
                }
                else
                    result += S[i] ;
            }
        }
        if(flag==1)
            result += "</b>" ;
        return result ;
    }
};
