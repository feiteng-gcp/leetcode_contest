class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n(S.size()), m(words.size());
        vector<bool> bold(n, false);
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                bool find(true);
                int len(words[j].size());
                if(i+len > n) find = false;
                for(int k=0;k<len&&find;++k)
                    if(words[j][k] != S[i+k]) find = false;
                if(find)
                    for(int k=0;k<len;++k) bold[i+k] = true;
            }
        }
        bool ok(false);
        string ans("");
        for(int i=0;i<n;++i){
            if(!bold[i]){
                if(ok) ans.append("</b>");
                ans.append(1, S[i]);
                ok = false;
            }
            else{
                if(!ok){
                    ans.append("<b>");
                    ans.append(1, S[i]);
                    ok = true;
                }
                else ans.append(1, S[i]);
            }
        }
        if(ok) ans.append("</b>");
        return ans;
    }
};
