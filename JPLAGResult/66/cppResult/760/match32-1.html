class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int a[505];
        for (int i = 0; i < 505; i++) {
            a[i] = 0;
        }
        
        for (string s : words) {
            for (int i = 0; i + s.length() - 1 < S.length(); i++) {
                bool match = true;
                for (int j = 0; j < s.length(); j++) {
                    if (S[i+j] != s[j]) {
                        match = false;
                        break;
                    }
                }
                

                if (match) {
                    for (int j = 0; j < s.length(); j++) {
                        a[i+j] = 1;
                    }
                }
            }
        }

        string ans = "";
        if (a[0] == 1) {
            ans = ans + "<b>" + S[0];
        } else {
            ans = ans + S[0];
        }
        
        for (int i = 1; i < S.length(); i++) {
            if (a[i] == 1 && a[i-1] == 0) {
                ans = ans + "<b>" + S[i];
            } else if (a[i] == 0 && a[i-1] == 1) {
                ans = ans + "</b>" + S[i];
            } else {
                ans = ans + S[i];
            }
        }
        
        if (a[S.length() - 1] == 1) {
            ans = ans + "</b>";
        }
        
        return ans;
    }
};
