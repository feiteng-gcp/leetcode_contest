class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<bool> bs(S.size());
        for (int i = 0; i < S.size(); i++) {
            for (int j = 0; j < words.size(); j++) {
                auto& s = words[j];
                int k = s.size();
                if (S.substr(i, k) == s) {
                    fill(bs.begin() + i, bs.begin() + i + k, true);
                }

            }
        }
//        d(bs);
        string res;
        for (int i = 0; i < S.size(); i++) {
            if (bs[i]) {
                if (i == 0 || !bs[i-1]) res += "<b>";
            } else {
                if (i > 0 && bs[i-1]) res += "</b>";
            }
            res.push_back(S[i]);
        }
        if (bs[S.size() - 1]) res += "</b>";
        return res;
    }
};
