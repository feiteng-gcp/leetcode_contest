class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int len = S.length();
        if(len == 0) return S;
        
        if(words.size() == 0) return S;
        
        vector<int> flag(len, 0);
        
        for(int i = 0; i < len; i++){
            for(int j = 0; j < words.size(); j++){
                string word = words[j];
                if(S[i] != word[0]) continue;
                
                string tmp = S.substr(i, word.length());
                if(tmp != word) continue;
                
                for(int k = 0; k < word.length(); k++){
                    flag[i+k] = 1;
                }
            }
        }
        
        string res = "";
        for(int i = 0; i < len; i++){
            if(flag[i] == 1){
                res += "<b>";
                int j = i;
                while(j < len && flag[j] == 1){
                    res += S[j];
                    j++;
                }
                res += "</b>";
                i = j-1;
            }
            else{
                res += S[i];
            }
        }
        
        return res;
    }
};
