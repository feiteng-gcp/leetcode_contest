class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n = S.size();
        vector<bool> r(n, false);
        for (string w : words) {
            int d = w.size();
            for (int i = 0; i <= n - d; ++i) {
                if (S.substr(i, d) == w) {
                    for (int k = i; k < i + d; ++k) r[k] = true;
                }
            }
        }
        string res = "";
        for (int i = 0; i < n; ++i) {
            if (r[i]) {
                res += "<b>";
                while (i < n && r[i])
                    res += S[i++];  
                --i;
                res += "</b>";
            }
            else res += S[i];
        }
        return res;
    }
};
