class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<bool> dp(S.size(), false);

        for (int i = 0; i < S.size(); ++i)
        {
            for (string str : words)
            {
                if (i + str.size() <= S.size())

                {
                    if (S.substr(i, str.size()) == str)
                    {
                        for (int j = i; j < i + str.size(); ++j)
                        {
                            dp[j] = true;
                        }
                    }
                }
            }
        }
        
        bool flag = false;
        string ret;
        for (int i = 0; i < S.size(); ++i)
        {
            if (flag)
            {
                if (!dp[i])
                {
                    flag = false;
                    ret += "</b>";
                }
                ret.push_back(S[i]);
            }
            else
            {
                if (dp[i])
                {
                    flag = true;
                    ret += "<b>";
                }
                ret.push_back(S[i]);
            }
        }
        if (flag)
        {
            ret += "</b>";
        }
        
        return ret;
    }
};
