class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        unordered_set<string> dict(words.begin(),words.end());
        bool isKeywords[S.size()];
        for(int i=0;i<S.size();i++)
        {
            isKeywords[i]=false;
        }
        for(int len=1;len<=min(10,(int)S.size());len++)
        {
            for(int i=0;i<=S.size()-len;i++)
            {

                string str=S.substr(i,len);
//                cout<<"str:"<<str<<endl;
                if(dict.count(str)>0)
                {
                    for(int j=0;j<len;j++)
                        isKeywords[i+j]=true;
                }
            }
        }
//        for(int i=0;i<S.size();i++)
//            cout<<isKeywords[i]<<"\t";
//        cout<<endl;
        string rslt=S,ltag="<b>",rtag="</b>";
        bool flag=false;
        int cnt=0;
        for(int i=0;i<rslt.size();i++)
        {
            if(isKeywords[i-cnt])
            {
                rslt.insert(i,ltag);
                i=i+ltag.size();
                cnt+=ltag.size();
//                cout<<"before: "<<i<<endl;
                while(i<rslt.size() && isKeywords[i-cnt]) i++;
//                cout<<"after: "<<i<<endl;
                rslt.insert(i,rtag);
                i=i+rtag.size();
                cnt+=rtag.size();
            }
        }
        return rslt;
    }
};
