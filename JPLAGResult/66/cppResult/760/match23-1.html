class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<bool> bold(S.length(), false);
        for( const auto &keyword: words ) {
            if( keyword.empty() )
                continue;
            int start_on = 0;            
            while( start_on < S.length() ) {
                const auto r = S.find(keyword, start_on);
                if( r == string::npos )
                    break;
//                cout << r << " " << keyword << endl;
                for( int i = r; i < r + keyword.length(); ++i )
                    bold[i] = true;
//                start_on = r + keyword.length();

                start_on = r + 1;
            }
        }
        
        string output;
        bool is_bold = false;
        for( int i = 0; i < S.length(); ++i ) {
            if( bold[i] && !is_bold ) {
                output += "<b>";
                is_bold = true;
            }
            else if( !bold[i] && is_bold ) {
                output += "</b>";
                is_bold = false;                
            }            
            output += S[i];
        }
        
        if( is_bold )
            output += "</b>";            

        return output;    
    }
};
