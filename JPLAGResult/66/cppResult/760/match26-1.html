class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int a[505];
        memset(a,0,sizeof(a));
        int len = words.size();
        for (int i = 0;i < len;i++) {
            string s = words[i];
            int slen = s.size();
            int index = 0,last = -1;
            last = S.find(s,index);
            while(last >= 0) {
                for (int j = last;j < last + slen;j++) {
                    a[j] = 1;
                }
                index = last + 1;
                last = S.find(s,index);
            }
        }
        len = S.size();

        bool bold = false;
        string ans;
        for (int i = 0;i < len;i++) {
            if (a[i] && !bold) {
                ans += "<b>";
                bold = true;
            }
            else if (!a[i] && bold) {
                ans += "</b>";
                bold = false;
            }
            ans += S[i];
        }
        if (bold) {
            ans += "</b>";
        }
        return ans;
    }
};
