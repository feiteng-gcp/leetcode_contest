class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        vector<bool> bold(S.size(), false);
        for (string & word : words) {
            for (int i = 0; i+word.size() <= S.size(); i++) {
                if (word == S.substr(i, word.size())) {
                    for (int j = 0; j < word.size(); j++) {
                        bold[i+j] = true;
                    }
                }
            }
        }
        bool flag = false;
        string ret;
        for (int i = 0; i < S.size(); i++) {
            if (bold[i] && !flag) {
                ret.append("<b>");
                flag = true;
            }
            if (!bold[i] && flag) {
                ret.append("</b>");
                flag = false;
            }
            ret.push_back(S[i]);
        }
        if (flag) ret.append("</b>");
        return ret;
        
    }
};
