const int SIZE = 1e6+10;
class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n=SZ(S);
        VI d(n);
        for(string s: words){
            FOR(i,0,SZ(S)-SZ(s)){
                if(S.substr(i,SZ(s))==s){
                    REP(j,SZ(s))d[i+j]=1;

                }
            }
        }
        string an="";
        for(int i=0,j;i<SZ(S);i=j){
            if(!d[i]){
                an+=S[i];
                j=i+1;
            }
            else{
                an+="<b>";
                for(j=i;j<n&&d[j];j++){
                    an+=S[j];
                }
                an+="</b>";
            }
        }
        return an;
    }
};
