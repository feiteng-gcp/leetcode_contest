class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n = S.size();
        vector<bool> visit(n);
        for (auto& s : words) {
            int m = s.size();

            for (int i = 0; i + m <= n; ++i) {
                if (S.substr(i, m) == s) {
                    for (int j = 0; j < m; ++j) {
                        visit[i + j] = true;
                    }
                }
            }
        }
        string ret;
        for (int i = 0, j; i < n; i = j) {
            for (; i < n && !visit[i]; ++i) ret += S[i];
            if (i == n) break;
            for (j = i; j < n && visit[j]; ++j);
            ret += "<b>";
            ret += S.substr(i, j - i);
            ret += "</b>";
        }
        return ret;
    }
};
