class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        unordered_set<string> dict = {words.begin(), words.end()};
        vector<bool> bold(S.size());
        for (int i = 0; i < S.size(); i++) {
            for (int k = 10; k > 0; k--) {
                if (i + k <= S.size() && dict.count(S.substr(i, k))) {
                    for (int j = 0; j < k; j++) {
                        bold[i+j] = true;
                    }
                    break;
                }
            }
        }
        string ans;
        bool inbold = false;
        for (int i = 0; i < S.size(); i++) {
            if (bold[i] && !inbold) {
                ans += "<b>";
                inbold = true;
            } else if (!bold[i] && inbold) {
                ans += "</b>";
                inbold = false;
            }
            ans.push_back(S[i]);
        }
        if (inbold) {
            ans += "</b>";
        }
        return ans;
    }
};
