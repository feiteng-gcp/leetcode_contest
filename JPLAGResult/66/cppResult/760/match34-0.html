class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        unordered_set<string> all(words.begin(), words.end());
        vector<bool> bold(S.size(), false);
        for(int i = 0; i < S.size(); i++)
        {
            for(int j = 1; j + i <= S.size() && j <= 10; j++)
            {
                auto temp = S.substr(i, j);
                if (all.count(temp))
                {
                    for(int k = i; k < i + j; k++)
                        bold[k] = true;
                }
            }
        }
        string ans;
        bool current_bold = false;
        for(int i = 0; i < S.size(); i++)
        {
            if (bold[i] && !current_bold)
            {
                ans += "<b>";
                current_bold = true;
            }
            if (!bold[i] && current_bold)
            {
                ans += "</b>";
                current_bold = false;
            }
            ans += S[i];
        }
        if (current_bold) ans += "</b>";
        return ans;
    }
};
