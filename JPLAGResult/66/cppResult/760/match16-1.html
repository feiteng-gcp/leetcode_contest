class Solution {
public:
    string boldWords(vector<string>& words, string S) {
        int n = S.size();
        vector<bool> bold(n, false);
        for(auto w: words){
            int l = w.size();
            for(int i = 0; i < S.size()-l+1; i++){
                if(S.substr(i, l) == w){
                    for(int j = i; j < i+l; j++) bold[j] = true;
                }
            }
        }
        string res;
        for(int i = 0; i < n; i++){
            //cout<<bold[i];
            if(bold[i]){
                if(i == 0 || !bold[i-1]) res += "<b>";
            }
            else{
                if(i > 0 && bold[i-1]) res += "</b>";
            }
            res += S[i];
        }
        if(bold[n-1]) res += "</b>";
        return res;
            
    }
};
