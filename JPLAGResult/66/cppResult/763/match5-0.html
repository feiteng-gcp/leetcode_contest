class Solution {
public:
    string str;
    int n;
    bool flag[50][50];
    string dp[50][50];
    string f(int a, int b) {
        if (flag[a][b])
            return dp[a][b];
        if (a > b) {
            dp[a][b] = "";
            flag[a][b] = true;
            return "";
        }
        string best;
        int bal = 0;
        int last = a - 1;
        vector<string> v;
        for (int i = a; i <= b; i++) {
            if (str[i] == '1')
                bal++;
            else
                bal--;
            if (bal == 0) {
                if (i != b || last != a - 1) {
                    v.push_back(f(last + 1, i));
                    last = i;
                }
            }
        }
        if (v.empty()) {
            best = str[a];
            best += f(a + 1, b - 1);
            best += str[b];
        } else {
            for (int i = 0; i < v.size(); i++) {
                int p = i;
                for (int j = i + 1; j < v.size(); j++)
                    if (v[j] > v[p])
                        p = j;
                swap(v[i], v[p]);
            }
            best = "";
            for (int i = 0; i < v.size(); i++)
                best += v[i];
        }
        //cout << "a = " << a << ", b = " << b << ", best = " << best << endl;
        dp[a][b] = best;
        flag[a][b] = true;
        return best;
    }
    string makeLargestSpecial(string S) {
        str = S;
        n = str.size();
        return f(0, n - 1);
    }
};
