string s;

void doit(int x)
{
  int i,now,st;
  if (x==s.length()) return;
  if (s[x]=='0') return;
  vector<string> a;
  a.clear();
  string tmp;
  string tt;
  now=0;
  tmp="";
  //cout<<x<<endl;
  st=x;
  for (i=x;i<s.length();i++)
  {
    if (s[i]=='1') now++;
    else now--;
    tmp.push_back(s[i]);
    if (now<0)
    {
      break;
    }
    if (now==0)
    {
      //cout<<tmp<<endl;
      doit(st+1);
      tmp=s.substr(st,tmp.length());
      a.push_back(tmp);
      tmp="";
      st=i+1;
    }
  }
  if (i==s.length()) i--;
  i-=tmp.length();
  //cout<<"i:"<<i<<endl;
  sort(a.begin(),a.end());
  reverse(a.begin(),a.end());
  tmp=s.substr(0,x);
  tt=s.substr(i+1);
  for (i=0;i<a.size();i++)
    tmp.append(a[i]);
  tmp.append(tt);
  s=tmp;
}

class Solution {
public:
    string makeLargestSpecial(string ss) {
      s=ss;
      doit(0);
      return s;
    }
};
