class Solution {

public:
	string makeLargestSpecial(string s) {
		int n = s.length();
		vector<string> best(n + 1);
		best[n] = "";
		vector<string> ss;
		string s2;

		int i, j, k, a, b, c0, c1;

		for (i = n - 1; i >= 0; i--) {
			if (s[i] == '0') best[i] = '0' + best[i + 1];
			else {
				best[i] = '1' + best[i + 1];
				s = s.substr(0, i) + best[i];
				ss.clear();
				for (a = i; a < n; a++) {

					c0 = 0; c1 = 0;
					for (b = a; b < n; b++) {
						if (s[b] == '0') c0++;
						else c1++;
						if (c0 > c1) {
							sort(ss.begin(), ss.end());
							reverse(ss.begin(), ss.end());
							s2 = "";
							for (auto sss : ss) s2 = s2 + sss;
							s2 = s2 + (s.substr(a));
							goto outer;
						}
						if (c0 == c1) {
							ss.push_back(s.substr(a, b + 1 - a));
							a = b;
							break;
						}
					}
				}
				sort(ss.begin(), ss.end());
				reverse(ss.begin(), ss.end());
				s2 = "";
				for (auto sss : ss) s2 = s2 + sss;
				a = 0;
			outer:
				best[i] = s2;
			}
		}

		return best[0];
	}
};
