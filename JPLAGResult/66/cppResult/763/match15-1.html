class Solution {
public:
  string makeLargestSpecial(string S) {
    if (S.length() == 0) { return S; }
    string ans = S;
    bool flag;
    do {
      flag = false;
      int st[55], tail = 0;
      int f[55];
      memset(f, 0, sizeof(f));
      for (int i = 0; i < ans.length(); ++i) {
        if (ans[i] == '1') {
          st[tail++] = i;
        } else {
          int h = st[--tail];
          int d = i - h + 1;

          f[h] = d;
          f[i] = -d;
        }
      }
      for (int i = 0; i < ans.length() - 1; ++i) {
        if (f[i] < 0 && f[i + 1] > 0) {
          string a = ans.substr(0, i - (-f[i]) + 1);
          string b = ans.substr(i - (-f[i]) + 1, -f[i]);
          string c = ans.substr(i + 1, f[i + 1]);
          string d = ans.substr(i + 1 + f[i + 1], -1);
          if (c > b) {
            ans = a + c + b + d;
            flag = true;
            break;
          }
        }
      }
    } while (flag);
    return ans;
  }
};
