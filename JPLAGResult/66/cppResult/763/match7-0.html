class Solution {
public:
    string solve(string S) {
        if (S.size() == 0) return S;
        
        if (S.size() == 2) return S;
        
        int n = S.size();    
        int i = 0;
        int cur = 0;
        vector<string> list;
        string temp;
        while (i < n) {
            temp.push_back(S[i]);
            if (S[i] == '1') 
                cur++;
            else
                cur--;
            if (cur == 0) {
                list.push_back("1"+solve(temp.substr(1, temp.size()-2))+"0");
                temp.clear();
            };
            i++;
        };
        sort(list.begin(), list.end());
        reverse(list.begin(), list.end());
        string ans;
        for (int i = 0; i < list.size(); ++i)
            ans.append(list[i]);
        return ans;        
    };
    
    string makeLargestSpecial(string S) {
        int n = S.size();
        if (n == 0) return "";
        vector<vector<string>> f(n+1, vector<string>(n+1));
        //vector<vector<int>> g(n+1, vector<int>(n+1, -1));
        
        for (int i = 0; i < n; ++i)
            f[i][i] = "";
        
        return solve(S);
        
        
        
        
        for (int len = 2; len <= n; ++len)
            for (int i = 0; i < n-len+1; ++i) {
                int j = i+len-1;
                //cout << i << " " << j << endl;
                
                string ans;
                int cur = 0;      
                bool valid = true;
                for (int k = i; k <= j; ++k) 
                    if (S[k] == '1') cur++;
                    else {
                        cur--;
                        if (cur < 0) {
                            valid = false;
                            break;
                        }
                    };
                if (!valid) {
                    f[i][j] = "";
                    continue;
                };
                
                f[i][j] = S.substr(i, len);
                if (len > 2) {
                    if (f[i+1][j-1].size() > 0) {
                        string c = "1"+f[i+1][j-1]+"0";
                        if (c > f[i][j])
                            f[i][j] = c;
                    };                        
                }                              
                
                cur = 0;
                for (int k = i; k < j; ++k) {
                    if (S[k] == '1') cur++;
                    else cur--;                    
                    if (cur == 0) {
                        string a = f[i][k];
                        string b = f[k+1][j];
  //                      if (i == 2 && j == 11 && k == 5)
//                            cout << a << " " << b << endl;
                        
                        if (a.size() > 0 && b.size() > 0) {
                            string c = a+b;
                            string d = b+a;
                            if (c > f[i][j]) f[i][j] = c;
                            if (d > f[i][j]) f[i][j] = d;
                        };
                    };
                };
                
            };
        
        //cout << f[2][11] << " " << f[6][11] << endl;
        cout << f[0][3] << " " << f[7][n-1] << endl;
        
        
        return f[0][n-1];
        
        
                
        
    }
};
