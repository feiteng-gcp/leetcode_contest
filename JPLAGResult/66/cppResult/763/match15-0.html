class Solution {
public:
	string calc(string s) {
		//cerr << "s == " << s << endl;
		if (s.size() == 0) return s;
		vector<int> rg(s.size());
		stack<pair<char, int> > st;
		for (int i = 0; i < s.size(); i++) {
			if (s[i] == '1') st.push(make_pair('1', i));
			else {
				assert(!st.empty());
				int pos = st.top().second;
				st.pop();
				rg[pos] = i;
			}
		}

		//cerr << "rgs == " << endl;
		//forn(i, sz(s)) cerr << rg[i] << ' '; cerr << endl;
		assert(st.empty());
		if (rg[0] == s.size() - 1) {
			string mid = "";
			for (int i = 1; i + 1 < s.size(); i++) mid += s[i];
			return "1" + calc(mid) + "0";
		}
		else {
			string s1 = "", s2 = "";
			int mid = rg[0];
			for (int i = 0; i <= mid; i++) s1 += s[i];
			for (int i = mid + 1; i < s.size(); i++) s2 += s[i];
			string ans1 = calc(s1);
			string ans2 = calc(s2);
			cerr << "input s == " << s << endl;
			cerr << "s1 ans1 == " << s1 << ' ' << ans1 << endl;
			cerr << "s2 ans2 == " << s2 << ' ' << ans2 << endl;
			if (ans1 < ans2) swap(ans1, ans2);
			cerr << "res == " << max(ans1 + ans2, ans2 + ans1) << endl;
			return max(ans1 + ans2, ans2 + ans1);
		}
	}

	string makeLargestSpecial(string s) {
		//return calc(s);
		set<string> used;
		queue<string> q;
		q.push(s);
		used.insert(s);
		string ans = s;
		int ops = 0;

		while (!q.empty()) {
			string s = q.front();
			q.pop();
		//	cerr << "s == " << s << endl;
			ops++;
			if (ops % 100 == 0 && double(clock()) / CLOCKS_PER_SEC > 1.7) break;
			//cerr << "ops == " << ops << endl;
			ans = max(ans, s);
			vector<int> rg(s.size());
			stack<pair<char, int> > st;
			for (int i = 0; i < s.size(); i++) {
				if (s[i] == '1') st.push(make_pair('1', i));
				else {
					assert(!st.empty());
					int pos = st.top().second;
					st.pop();
					rg[pos] = i;
				}
			}

			for (int i = 0; i < s.size(); i++) {
				if (rg[i] != 0 && rg[i] + 1 < s.size() && rg[rg[i] + 1] != 0) {
					string to = "";
					for (int j = 0; j < i; j++) to += s[j];
					int l1 = i, r1 = rg[i];
					int l2 = r1 + 1;
					int r2 = rg[l2];
					for (int j = l2; j <= r2; j++) to += s[j];
					for (int j = l1; j <= r1; j++) to += s[j];
					for (int j = r2 + 1; j < s.size(); j++) to += s[j];
					if (used.count(to) == 0 && to > s) {
						used.insert(to);
						q.push(to);
					}
				}
			}
		}

		return ans;
	}
};
