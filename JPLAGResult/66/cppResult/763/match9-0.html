class Solution {
public:
    string makeLargestSpecial(string S) {
        int i, j, k, l;
        l = S.size();
        while (1) {
            string ns = S;
            for (i = 0; i + 1 < l; i++) {
                if (S[i] == '0' && S[i + 1] == '1') {
                    int c0, c1;
                    c0 = c1 = 0;
                    for (j = i; j >= 0; j--) {
                        if (S[j] == '0') {
                            c0++;
                        }
                        else {
                            c1++;
                        }
                        if (c0 == c1) {
                            break;
                        }
                    }
                    c0 = c1 = 0;
                    for (k = i + 1; k < l; k++) {
                        if (S[k] == '0') {
                            c0++;
                        }
                        else {
                            c1++;
                        }
                        if (c0 == c1) {
                            break;
                        }
                    }
                    //cout << j << i << k << endl;
                    string a, b;
                    a = S.substr(j, i - j + 1);
                    b = S.substr(i + 1, k - i);
                    //cout << a << " " << b << endl;
                    if (a.compare(b) < 0) {
                        ns = S.substr(0, j) + b + a + S.substr(k + 1, l - k - 1);
                        //cout << ns << endl;
                        break;
                    }
                }
            }
            if (S.compare(ns) == 0) {
                break;
            }
            S = ns;
        }
        return S;
    }
};
