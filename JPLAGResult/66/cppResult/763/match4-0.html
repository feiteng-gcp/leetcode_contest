class Solution {
public:
    string makeLargestSpecial(string S) {
        string result = S;
        for (int i=0; i<S.size(); i++)
        {
            //start : i
            int one_num = 0;
            for (int j=i; j<S.size(); j++)
            {
                if (S[j] == '1')
                {
                    one_num ++;
                }
                if (j - i + 1 - one_num > one_num)
                {
                    break;
                }
                if (j - i + 1 - one_num == one_num)
                {
                    int one_num2 = 0;
                    //start : j+1
                    for (int k=j+1; k<S.size(); k++)
                    {
                        if (S[k] == '1')
                        {
                            one_num2 ++;
                        }
                        if (k - j - 1 + 1 - one_num2 > one_num2)
                        {
                            break;
                        }
                        if (k - j - 1 + 1 - one_num2 == one_num2)
                        {
                            //swap
                            string s2 = S.substr(0, i) + S.substr(j+1, k-j-1+1) + S.substr(i, j - i + 1) + S.substr(k+1);
                            //cout << i << " " << j << " " << k << endl;
                            //cout << s2 << endl;
                            if (s2 > result)
                            {
                                result = s2;
                                
                            }
                        }
                    }
                }
            }
        }
        if (S == result)
        {
            return result;
        }
        else
        {
            return makeLargestSpecial(result);
        }
    }
};
