class Solution {
public:
    bool ck(string s) {
        int n = s.size();
        int cur = 0;
        for (int i = 1; i < n - 1; i++) {
            if (s[i] == '1') cur++;
            else cur--;
            if (cur < 0) return false;
        }
        return true;
    }
    string solve(string s) {
        if (s == "") return "";
        if (s == "10") return "10";
        //cout << s << endl;
        
        int n = s.size();
        int foo = 0, bar = n - 1;
        if (ck(s)) foo++, bar--;
        string res = "";
        int cur = 0;
        int pre = foo;
       // cout << s << " " << endl;
       // cout << foo << " " << bar << endl;
        vector<string> tmp;
        for (int i = foo; i <= bar; i++) {
            if (s[i] == '1') cur++;
            else cur--;
            if (cur == 0) {
                tmp.push_back(solve(s.substr(pre, i - pre + 1)));
                pre = i + 1;
            }
        }
        sort(tmp.begin(), tmp.end());
        for (int i = (int)tmp.size() - 1; i >= 0; i--) {
            res += tmp[i];
        }
        for (int i = 0; i < foo; i++) res = "1" + res;
        for (int i = 0; i < foo; i++) res += '0'; 
        return res;
    }
    string makeLargestSpecial(string S) {
        return solve(S);
    }
};
