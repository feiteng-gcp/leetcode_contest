class Solution {
public:
    string makeLargestSpecial(string S) {
        string ret = S;
        unordered_set<string> hash;
        hash.insert(S);
        deque<string> dq;
        dq.push_back(S);
        while (!dq.empty()) {
            string x = dq.front();
            //cout << "#" << x << endl;
            ret = max(ret, x);
            unordered_map<int, vector<int>> arr;
            for (int i = 0; i < x.size(); ++ i) {
                int cnt0 = 0, cnt1 = 0;
                for (int j = i; j < x.size(); ++ j) {
                    if (x[j] == '0') {
                        cnt0 ++;
                    } else {
                        cnt1 ++;
                    }
                    if (cnt0 > cnt1) break;
                    if (cnt0 == cnt1) {
                        //cout << i << ' ' << j << endl;
                        arr[i].push_back(j);
                    }
                }
            }
            for (auto& p : arr) {
                int x1 = p.first;
                for (auto& y1 : p.second) {
                    int x2 = y1 + 1;
                    if (arr.count(x2) == 0) continue;
                    //cout << "####" << x1 << ' ' << y1 << endl;
                    for (auto& y2 : arr[x2]) {
                        string s1 = x.substr(0, x1),
                        s2 = x.substr(x1, y1 - x1 + 1),
                        s3 = x.substr(x2, y2 - x2 + 1),
                        s4 = x.substr(y2 + 1, S.size() - y2 - 1);
                        string tmp = s1 + s3 + s2 + s4;
                        if (hash.count(tmp) == 0 && tmp > ret) {
                            /*
                            cout << x1 << ' ' << y1 << ' ' << x2 << ' ' << y2 << endl;
                            cout << x << endl;
                            cout << tmp << endl;
                            cout << "########" << endl;
                            */
                            dq.push_back(tmp);
                            hash.insert(tmp);
                        }
                    }
                }
            }
            dq.pop_front();
        }
        return ret;
    }
};
