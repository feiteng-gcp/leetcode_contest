class Solution {
private:
    bool check(string S)
    {
        int c1 = 0;
        int c0 = 0;
        for(auto c : S)
        {
            c1 += c == '1';
            c0 += c == '0';
            if (c1 < c0) return false;
        }
        return c1 == c0;
    }
public:
    string makeLargestSpecial(string S) {
        bool swap = true;
        while(swap)
        {
            swap = false;
            for(int i = 0; i < S.size(); i++)
            {
                for (int l = 2; i + l <= S.size(); l += 2)
                {
                    auto first = S.substr(i, l);
                    if (check(first))
                    {
                        for(int j = 2; i + l + j <= S.size(); j += 2)
                        {
                            auto second = S.substr(i + l, j);
                            if (check(second) && first < second)
                            {
                                if (l < j && second.substr(0, l) == first) continue;
                                swap = true;
                                string new_S = "";
                                if (i > 0) new_S = S.substr(0, i);
                                new_S += second;
                                new_S += first;
                                if (i + l + j < S.size()) new_S += S.substr(i + l + j);
                                S = new_S;
                            }
                        }
                    }
                }
            }
        }
        return S;
    }
};
