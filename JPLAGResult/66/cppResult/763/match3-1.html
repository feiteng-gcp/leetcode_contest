class Solution {
public:

  bool special(const string &s) {
    int ones = 0, zeros = 0;
    for (int i = 0; i < s.size(); ++i) {
      ones += s[i] == '1';
      zeros += s[i] == '0';
      if (ones < zeros) {
        return false;
      }
    }
    return ones == zeros;
  }
  string makeLargestSpecial(string S) {
    set<string> vis{S};
    priority_queue<string> q;
    q.push(S);
    while (!q.empty()) {
      auto s = q.top();
      q.pop();
      for (int i = 0; i < s.size(); ++i) {
        for (int j = i; j < s.size(); ++j) {
          auto s1 = s.substr(i, j - i + 1);
          if (!special(s1)) {
            continue;
          }
          for (int k = j + 1; k < s.size(); ++k) {
            auto s2 = s.substr(j + 1, k - j);
            if (!special(s2)) {
              continue;
            }
            auto p1 = s.substr(0, i);
            auto p2 = s.substr(k + 1);
            auto v = p1 + s2 + s1 + p2;
            if (v <= s) {
              continue;
            }
            if (*(vis.rbegin()) < v) {
              q.push(v);
              vis.insert(v);
            }
          }
        }
      }
    }
    return *vis.rbegin();
  }
};
