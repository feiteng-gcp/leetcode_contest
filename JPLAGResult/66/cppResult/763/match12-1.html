class Solution {
public:
    string makeLargestSpecial(string S) {
        string ans = makeLargestSpecial1(S);
        while (true) {
            string temp = makeLargestSpecial1(ans);
            if (temp == ans) { break; }
            ans = temp;
        }
        return ans;
    }
    string makeLargestSpecial1(string S) {
        string ans;
        int n = S.length();
        if (Check(S)) {
            ans = S;
        }
        for (int i = 0; i < n; i++) {

            for (int j = 0; j <= i; j++) {
                for (int k = i + 1; k < n; k++) {
                    if (!Check(S.substr(i + 1, k - i)) || !Check(S.substr(j, i - j + 1))) { continue; }
                    string temp = S.substr(0, j) + S.substr(i + 1, k - i) + S.substr(j, i - j + 1);
                    if (k + 1 < n) {
                        temp += S.substr(k + 1, n - k - 1);
                    }
                    if (Check(temp)) {
                        if (ans.empty() || ans < temp) {
                            ans = temp;
                        }
                    }
                }
            }
        }
        if (ans.empty()) { ans = S; }
        return ans;
    }
    
    bool Check(string s) {
                    int n = s.length();
                    bool check = true;
                    for (int l = 0, one = 0; l < n; l++) {
                        if (s[l] == '1') {
                            one++;
                        } else {
                            one--;
                        }
                        if (one < 0) {
                            check = false;
                            break;
                        }
                        if (l == n - 1 && one != 0) { return false; } 
                    }
        return check;
    }
};
