class Solution {
public:
    
    string makeLargestSpecial(string S) {
        auto preS = S;
        int n = S.size();
        vector<vector<int>> alllens;
        vector<int> lens;
        while (true) {
            alllens.clear();

            for (int i=0; i<n; ++i) {
                int cnt = 0;
                lens.clear();
                for (int j=i; j<n; ++j) {
                    cnt += S[j]=='1' ? 1 : -1;
                    // printf("%d %d %d\n", i, j, cnt);
                    if (cnt<0) break;
                    if (cnt==0) 
                        lens.push_back(j-i+1);
                        // cout << strs.back().first << endl;
                }
                alllens.push_back(lens);
            }
            
            preS = S;
            for (int i=0; i<n; ++i) 
                for (auto x : alllens[i])
                    if (i+x<n)
                        for (auto y : alllens[i+x]) {
                            // printf("%d %d %d ",i,x,y); 
                            auto nxt = preS.substr(0, i) + preS.substr(i+x, y) + preS.substr(i,x) + preS.substr(i+x+y);
                            if (nxt>S) 
                                S = nxt;
                            // cout << nxt<<endl;
                        }
            if (S==preS) break;
            // cout << S << endl;
        }
        return S;
    }
};
