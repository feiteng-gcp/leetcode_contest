class Solution {
public:
    string makeLargestSpecial(string S) {
        int n = S.size();
        set<string> d;
        d.insert(S);
        queue<string> q;
        q.push(S);
        while (!q.empty()) {
            auto cur = q.front(); q.pop();
            for (int i = 0; i < n; ++i) {
                for (int j = i + 1; j < n; j += 2) if (Check(cur, i, j)) {
                    for (int k = j + 2; k < n; k += 2) if (Check(cur, j + 1, k)) {
                        string next = cur.substr(0, i) + cur.substr(j + 1, k - j) + cur.substr(i, j - i + 1) + (k + 1 < n ? cur.substr(k + 1) : "") ;
                        if (d.count(next) == 0 && next > *d.rbegin()) {
                            d.insert(next);
                            q.push(next);
                        }
                    }
                }
            }
        }
        return *d.rbegin();
    }
    bool Check(string& s, int l, int r) {
        int c1 = 0, c0 = 0;
        for (int i = l; i <= r; ++i) {
            s[i] == '0' ? ++c0 : ++c1;
            if (c1 < c0)
                return false;
        }
        return c0 == c1;
    }
};
