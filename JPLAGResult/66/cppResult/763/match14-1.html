class Solution {
public:
    string makeLargestSpecial(string S) {
        int cnt[55];
	cnt[0]=0;
	int len=S.size();
	for(int i=0;i<len;i++)
	{
		int last;
		last=cnt[i];
		if(S[i]=='1')cnt[i+1]=last+1;
		else cnt[i+1]=last-1;
	}
	cnt[len+1]=-1;
	vector<int> v;
	int place[55];

	for(int c=len/2;c>=0;c--)
	{
		/*printf("c:%d\n",c);
		for(int i=0;i<=len+1;i++)
		{
			printf("%d ",cnt[i]);
		}
		printf("\n");*/
		for(int i=0;i<=len+1;i++)
		{
			if(cnt[i]<c)
			{
				int num=v.size();
				if(num>=3)
				{
					/*printf("WA\n");
					for(int i=0;i<num;i++)
					{
						printf("%d ",v[i]);
					}
					printf("\n");*/
					int f=v[0]+1;
					int ff=f;
					for(int j=0;j<num-1;j++)
					{
						place[j]=j;
					}
					for(int p=0;p<num-1;p++)
					{
						for(int q=p+1;q<num-1;q++)
						{
							int a=v[place[p]]+1,b=v[place[q]]+1;
							int sw=0;
							while(a<=v[place[p]+1]&&b<=v[place[q]+1])
							{
								if(cnt[a]>cnt[b])break;
								if(cnt[a]<cnt[b])
								{
									sw=1;
									break;
								}
								a++;
								b++;
							}
							if(sw)
							{
								int tmp=place[p];
								place[p]=place[q];
								place[q]=tmp;
							}
						}
					}
					int cntt[55];
					string ss=S;
					for(int cc=0;cc<num-1;cc++)
					{
						for(int j=v[place[cc]]+1;j<=v[place[cc]+1];j++)
						{
							cntt[f]=cnt[j];
							ss[f-1]=S[j-1];
							f++;
						}
					}
					for(int j=ff;j<f;j++)
					{
						cnt[j]=cntt[j];
						S[j-1]=ss[j-1];
					}
				}
				v.clear();
			}
			else if(cnt[i]==c)v.push_back(i);
		}
	}
	return S;
    }
};
