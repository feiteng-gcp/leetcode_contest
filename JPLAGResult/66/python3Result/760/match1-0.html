def find_all(a_str, sub):
    start = 0
    while True:
        start = a_str.find(sub, start)
        if start == -1:
            break
        yield start
        start += 1


class Solution:
    def boldWords(self, words, s):
        """
        :type words: List[str]
        :type s: str
        :rtype: str
        """

        if not s or not words:
            return s

        all_ranges = []

        for word in words:
            for b in find_all(s, word):
                all_ranges.append((b, b + len(word)))

        all_ranges.sort()

        ranges = []
        for r in all_ranges:
            if not ranges:
                ranges.append(r)
            else:
                prev_r = ranges.pop()
                if r[0] <= prev_r[1]:
                    ranges.append((prev_r[0], max(r[1], prev_r[1])))
                else:
                    ranges.append(prev_r)
                    ranges.append(r)

        res = ''
        i = 0
        for r in ranges:
            res += s[i: r[0]]
            res += '<b>'
            res += s[r[0]: r[1]]
            res += '</b>'
            i = r[1]

        if i < len(s):
            res += s[i:]

        return res
