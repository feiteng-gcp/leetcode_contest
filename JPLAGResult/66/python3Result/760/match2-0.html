class Solution:
    def boldWords(self, words, S):
        kwrd = [False]*len(S)
        for i in range(len(S)):
            for w in words:
                if i+len(w) <= len(S):
                    if S[i:i+len(w)] == w:
                        for j in range(i, i+len(w)):
                            kwrd[j] = True
        sol = []
        prev = False
        for i in range(len(S)):
            if kwrd[i] and not prev:
                sol.append('<b>')
                prev = True
            elif not kwrd[i] and prev:
                sol.append('</b>')
                prev = False
            sol.append(S[i])
        if prev:
            sol.append('</b>')
        return ''.join(sol)
