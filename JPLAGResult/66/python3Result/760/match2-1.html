class Solution:
    def boldWords(self, words, S):
        """
        :type words: List[str]
        :type S: str
        :rtype: str
        """
        bold = [False] * len(S)
        
        for w in words:
            for i in range
(len(S)):
                if S[i:i+len(w)] == w:
                    for j in range(i, i+len(w)):
                        bold[j] = True
        
        answer = ''
        bolded = False
        for i in range(len(S)):
            if bold[i]:
                if not bolded:
                    answer += '<b>'
                    bolded = True
            if not bold[i] and bolded:
                bolded = False
                answer += '</b>'
                
            answer += S[i]
        if bolded:
            answer += '</b>'
        return answer
