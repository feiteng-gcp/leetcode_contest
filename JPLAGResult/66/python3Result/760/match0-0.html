class Solution:
    def boldWords(self, words, S):
        """
        :type words: List[str]
        :type S: str
        :rtype: str
        """
        mask = [0]*len(S)

        for w in words:
            idx = 0
            while idx < len(S):
                idx = S.find(w, idx)
                if idx == -1:
                    break
                for i in range(len(w)):
                    mask[idx + i] = 1
                idx += 1

        ret = ""
        on_flag = False
        for i, s in enumerate(S):
            if mask[i] == 1 and on_flag == False:
                ret += '<b>'
                on_flag = True
            elif mask[i] == 0 and on_flag == True:
                ret += '</b>'
                on_flag = False
            ret += s
        if on_flag:
            ret += '</b>'

        return ret
