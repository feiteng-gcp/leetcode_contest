# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        idx = [1 for _ in range(len(avails))]
        lim = [len(av) for av in avails]
        heap = [((av[0].start, av[0].end), i) for i, av in enumerate(avails)]
        res = []
        import heapq as hq
        hq.heapify(heap)
        while heap:
            top, i = hq.heappop(heap)
            if not res or top[0] > res[-1][1]:
                res.append(top)
            elif top[1] > res[-1][1]:
                a = res.pop()
                res.append((a[0], top[1]))
            if idx[i] < lim[i]:
                av = avails[i][idx[i]]
                hq.heappush(heap, ((av.start, av.end), i))
                idx[i] += 1
        
        # print(res)
        
        free = []
        for i in range(1, len(res)):
            free.append(Interval(res[i - 1][1], res[i][0]))
            
        return free
