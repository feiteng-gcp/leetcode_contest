# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        q = []
        for avail in avails:
            for cur in avail:
                heapq.heappush(q, (cur.start, cur.end))
                
        res = []
        last = None
        while q:
            start, end = heapq.heappop(q)
            if not last:
                last = end
            if start > last:
                res.append(Interval(last, start))
            if end > last:
                last = end
        return res
