# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def merge(self, intervals):
        """
        :type intervals: List[Interval]
        :rtype: List[Interval]
        """
        sorted_ints = sorted(intervals, key = lambda x: x.start)
        res = []
        for i in sorted_ints:
            if not res or i.start > res[-1].end:
                res.append(i)
            else:
                res[-1] = (Interval(res[-1].start, max(res[-1].end, i.end)))

        return res
    
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        if len(avails) == 1:
            return avails[0]
        
        ans = self.merge([i for a in avails for i in a])
        
        # print([(a.start, a.end) for a in ans])
        res 
= []
        
        for i in range(len(ans)-1):
            res.append(Interval(ans[i].end, ans[i+1].start))    
            
        return res
