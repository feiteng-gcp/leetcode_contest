# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e


class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        
        ava = []
        for av in avails:
            for a in av:
                ava.append((a.start, a.end))
        
        # ava.sort(key=lambda tup: tup[0])
        ava.sort()
        
        ans = []
        left = ava[0][0]
        cur = ava[0][1]
        for a in ava:
            if a[0] <= cur:
                cur = max(cur, a[1])
            else:
                ans.append([left, cur])
                left = a[0]
                cur = a[1]
        
        ans.append([left, cur])
        
        res = []
        for i in range(len(ans)-1):
            res.append(Interval(ans[i][1], ans[i+1][0]))
        return res
