# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        temp = []
        n = len(avails)
        for i in range(n):
            for j in range(len(avails[i])):
                temp.append((avails[i][j].start,avails[i][j].end))
        temp.sort(key = lambda x:x[0])
        j=1
        #print(temp)
        ans = [temp[0]]
        while j<len(temp):
            if temp[j][0]<=ans[-1][1]:
                ans[-1] = min(ans[-1][0],temp[j][0]),max(temp[j][1],ans[-1][1])
            else:
                ans.append(temp[j])
            j += 1
        #print(ans)
        ret = []
        for i in range(1,len(ans)):
            ret.append([ans[i-1][1],ans[i][0]])
        return ret
