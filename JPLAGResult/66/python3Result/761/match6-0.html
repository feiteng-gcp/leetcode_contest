class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        result = []
        intervals = [item for sublist in avails for item in sublist]
        intervals = sorted(intervals, key=lambda e: e.start)
        merged = [intervals[0]]

        for i in range(1, len(intervals)):
            e = intervals[i]
            if merged[-1].end < e.start:
                merged.append(e)
            else:
                merged[-1].end = max(merged[-1].end, e.end)

        for j in range(len(merged)-1):
            result.append(Interval(s=merged[j].end, e=merged[j+1].start))
        return result
