# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        intervals = []
        
        for t in avails:
            intervals += t
        intervals.sort(key=lambda x: x.start)
        
        rets = []
        for i in intervals:
            
if rets and i.start <= rets[-1].end:
                rets[-1].end = max(rets[-1].end, i.end)
            else:
                rets.append(i)
        ans 
= []
        for idx in range(len(rets)-1):
            ans.append([rets[idx].end, rets[idx+1].start])
        return ans
