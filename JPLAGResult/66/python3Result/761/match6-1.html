# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        def enqueue(Q, I):
            
if not Q or Q[-1].end < I.start:
                Q.append(I)
            else:
                Q[-1].end = max(Q[-1].end, I.end)
        
        if not avails:
            return []
        filtered = collections.deque(avails[0])
        for i in range(1, len(avails)):
            intrvls = collections.deque(avails[i])
            temp = collections.deque()
            while filtered or intrvls:
                if not filtered:
                    enqueue(temp, intrvls.popleft())
                elif not intrvls:
                    enqueue(temp, filtered.popleft())
                elif filtered[0].start < intrvls[0].start:
                    enqueue(temp, filtered.popleft())
                else:
                    enqueue(temp, intrvls.popleft())
            filtered = temp
        s = filtered.popleft().end
        res = []
        while filtered:
            f = filtered.popleft()
            res.append(Interval(s, f.start))
            s = f.end
        return res
