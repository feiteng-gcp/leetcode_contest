# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def employeeFreeTime(self, avails):
        """
        :type avails: List[List[Interval]]
        :rtype: List[Interval]
        """
        total = sorted((n for arr in avails for n in arr), key=lambda n: n.start)
        t = -1
        r = []
        for n in total:
            if n.start > t:
                r.append(Interval(t, n.start))
            t = max(t, n.end)
        return r[1:]
