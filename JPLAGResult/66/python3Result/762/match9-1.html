class Solution:
    def anagramMappings(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        d = {}
        for i in range(len(B)):
            if B[i] not in d: d[B[i]]=[]
            d[B[i]].append(
i)
        ret = []
        for i in A:
            ret.append(d[i].pop())
        return ret
    
s=Solution()
print(s.anagramMappings(A = [12, 28, 46, 32, 50], B = [50, 12, 32, 46, 28]))
