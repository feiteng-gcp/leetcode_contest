from collections import defaultdict
class Solution:
    def anagramMappings(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        index_dict = defaultdict(list)
        for i,num in enumerate(B):
            index_dict[num].append(i)
            
        result = []
        for num in A:
            result.append(index_dict[num].pop())
        return result
