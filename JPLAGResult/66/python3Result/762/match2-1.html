class Solution:

    def anagramMappings(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        # d = {j:i for i, j in enumerate(B)}
        d = collections.defaultdict
(list)
        
        for i, j in enumerate(B):
            d[j].append(i)

        res = []
        for i in A:
            res.append(d[i][-1])
            d[i] = d[i][:-1]
        
        return res
