class Solution:
    def anagramMappings(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        my_dict = collections.defaultdict(set)
        for i, b in enumerate(B):
            my_dict[b] = my_dict[b] | {i}

        P = []
        for a in A:
            P.append(my_dict[a].pop())
        
        return P
