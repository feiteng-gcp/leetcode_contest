class Solution:
    def anagramMappings(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        mem = dict()
        for i in range(len(B)):
            mem[B[i]] = i
        res = []
        for i in range(len(A)):
            res.append(mem[A[i]])
        return res
