class Solution:
    def anagramMappings(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        dict_b = {}
        for i, b in enumerate(B):
            dict_b[b] = dict_b.get(b, []) + [i]

        res = []
        for a in A:
            res.append(dict_b[a][-1])
            dict_b[a] = dict_b[a][:-1]
        return res
