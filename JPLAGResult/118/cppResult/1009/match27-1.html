class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> ans;
        for (int i = 0; i < A.size(); i++) {
            int k = distance(A.begin(), max_element(A.begin(), A.end() - i)) + 1;
            if (k == A.size() - i) continue;
            if (k > 1) ans.push_back(k);
            ans.push_back(A.size() - i);
            reverse(A.begin(), A.begin() + k);
            reverse(A.begin(), A.end() - i);
        }
        return ans;
    }
};
