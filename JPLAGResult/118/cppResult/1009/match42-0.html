class Solution {
public:
    vector<int> pancakeSort(vector<int>& a) {
        int n = a.size();
        vector<int> res;
        for(int i = n; i >= 2; --i){
            int pos = -1;
            for(int j = 0; j < i; ++j){
                if(a[j] == i) {
                    pos = j;
                    break;
                }
            }
            if(pos + 1 > 1) {
                res.emplace_back(pos + 1);
                reverse(a.begin(), a.begin() + pos + 1);
            }
            res.emplace_back(i);
            reverse(a.begin(), a.begin() + i);
        }
        return res;
    }
};
