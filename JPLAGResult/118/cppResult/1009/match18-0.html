class Solution {
    void flip(vector<int>& A, int k) {
        int i = 0, j = k - 1;
        while (i < j) {
            int tmp = A[i];
            A[i] = A[j];
            A[j] = tmp;
            i++;
            j--;
        }
    }
    int find(vector<int>& A, int target) {
        int n = A.size();
        for (int i = 0; i < n; i++) {
            if (A[i] == target) return i;
        }
        return -1;
    }
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> ans;
        int n = A.size();
        for (int i = 0; i < n - 1; i++) {
            int idx = find(A, n - i);
            if (idx == n - i - 1) continue;
            flip(A, idx + 1);
            flip(A, n - i);
            ans.push_back(idx + 1);
            ans.push_back(n - i);
        }
        return ans;
    }
};
