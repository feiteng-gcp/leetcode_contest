class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        int n = A.size();
        vector<int> ans;
        for(int i=n;i>=1;i--)
        {
            // find biggest element from 0 to i-1 and its idx
            
            if(A[i-1]==i)continue;
            int idx = -1;
            for(int j=0;j<i;j++)
            {
                if(A[j]==i)
                {
                    idx = j;
                    break;
                }
            }
            
            reverse(A.begin(),A.begin()+idx+1);
            ans.push_back(idx+1);
            reverse(A.begin(),A.begin()+i);
            ans.push_back(i);
        }
        return ans;
    }
};
