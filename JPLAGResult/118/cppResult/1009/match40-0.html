class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> res;
        int n = A.size();
        
        for(int i = n; i > 1; i--) {
            
            int maxnum = maxsize(A, i);
            
            if(maxnum != i-1) {
                flip(A, maxnum);
                flip(A, i - 1);
                res.push_back(maxnum + 1);
                res.push_back(i);
            }
        }
        
        return res;
        
    }
    
    int maxsize(vector<int>& A, int size) {
        int ind = 0;
        for(int i = 1; i < size; i++) {
            if(A[i] > A[ind])   ind = i;
        }
        return ind;
    }
    
    
    void flip(vector<int>& A, int i) 
    { 
        int start = 0; 
        while (start < i) 
        { 
            swap(A[start], A[i]);
            start++; 
            i--; 
        } 
    } 
};
