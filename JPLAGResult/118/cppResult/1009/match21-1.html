class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> result;

        for (auto len = A.size(); len > 0; len--) {
            int max_idx = 0;
            for (int i = 0; i < len; i++) {
                if (A[i] >= A[max_idx]) {
                    max_idx = i;
                }
            }
            if (max_idx + 1 == len) {
                continue;
            }
            if (max_idx > 0) {
                result.push_back(max_idx + 1);
            }
            result.push_back(len);
            reverse(A.begin(), A.begin() + max_idx + 1);
            reverse(A.begin(), A.begin() + len);
        }
        return result;
    }
};
