class Solution {
public:
    vector<int> pancakeSort(vector<int>& arr) {
        vector<int> ans;
        

        for (int i = arr.size()-1; i >= 0; --i) {
            int maxi = i;
            for (int j = 0; j < i; ++j) {
                if (arr[j] > arr[maxi]) maxi = j;
            }
            // cout << maxi << endl;
            if (maxi == i) continue;
            
            ans.push_back(maxi+1);
            reverse(arr.begin(), arr.begin()+maxi+1);
            ans.push_back(i+1);
            reverse(arr.begin(), arr.begin()+i+1);

        }
        
        return ans;
    }
};
