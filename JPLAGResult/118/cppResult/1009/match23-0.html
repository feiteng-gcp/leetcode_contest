class Solution {
public:
    vector<int> A, flips;

    void flip(int k) {
        if (k >= 2) flips.push_back(k);
        std::reverse(A.begin(), A.begin() + k);
        //printf("%d:", k); for (int i = 0; i < A.size(); i++) printf(" %d", A[i]); printf("\n");
    };    

    vector<int> pancakeSort(vector<int>& A_) {
        A = A_;
        flips.clear();
        
        for (int m = A.size(); m > 0; m--) {
            int maxi = 0;
            for (int i = 0; i < m; i++) {
                if (A[i] > A[maxi]) maxi = i;
            }
            if (maxi == m - 1) continue;
            
            
            flip(maxi + 1);
            flip(m);
        }
        
        return flips;
    }
};
