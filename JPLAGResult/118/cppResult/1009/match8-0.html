class Solution {
public:
    vector<int> pancakeSort(vector<int>& a) {
        int n = a.size();
        vector<int> ret;
        for (int i = n; i > 0; --i) {
            int x = 0;
            for (int j = 0; j < i; ++j) if (a[j] > a[x]) x = j;
            ret.push_back(x + 1);
            reverse(a.begin(), a.begin() + x + 1);
            // for (int y : a) cout << y << ' '; cout << endl;
            ret.push_back(i);
            reverse(a.begin(), a.begin() + i);
            // for (int y : a) cout << y << ' '; cout << endl;
        }
        return ret;
    }
};
