class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> ans;
        int n = A.size();
        for (int i = n; i >= 1; i--)
        {
            int j = 0;
            while (A[j] != i) j++;
            ans.push_back(j + 1);
            int k = 0;
            while (k < j)
            {
                int t = A[k];
                A[k] = A[j];
                A[j] = t;
                j--;
                k++;
            }
            ans.push_back(i);
            j = i - 1;
            k = 0;
            while (k < j)
            {
                int t = A[k];
                A[k] = A[j];
                A[j] = t;
                j--;
                k++;
            }
            // for (int x : A) cout << x << " ";
            // cout << "\n";
        }
        return ans;
    }
};
