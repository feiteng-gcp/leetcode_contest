class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        int n = A.size();
        vector<int> res;
        for (int i = n - 1; i >= 0; --i) {
            if (A[i] != i + 1) {

                int k = 0;
                for (int j = 0; j < n; ++j) {
                    if (A[j] == i + 1) {
                        k = j;
                        break;
                    }
                }
                
                if(k > 0) {
                    res.push_back(k + 1);   
                    A = flip(A, k);
                }
                
                res.push_back(i + 1);
                A = flip(A, i);
            }
        }
        
        return res;
    }
private:
    vector<int> flip(vector<int>& A, int pos) {
        vector<int> res(A.size());
        for (int i = 0; i <= pos; ++i) {
            res[i] = A[pos - i];
        }
        for (int i = pos + 1; i < A.size(); ++i) {
            res[i] = A[i];
        }
        return res;
    }
};
