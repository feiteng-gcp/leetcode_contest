class Solution {
public:
    void flip(vector<int> &a, int sz){
        int l = 0, r = sz;
        while(l < r){
            swap(a[l++], a[r--]);
        }
    }

    vector<int> pancakeSort(vector<int>& A) {
        int n = A.size();
        vector<int> ans;
        for(int sz = n; sz >= 1; sz--){
            int mx_ind = -1;
            int mx = -1;
            for(int i = 0; i < sz; i++){
                if(A[i] > mx){
                    mx = A[i];
                    mx_ind = i;
                }
            }
            ans.push_back(mx_ind + 1);
            flip(A, mx_ind);
            ans.push_back(sz);
            flip(A, sz - 1);
        }
        return ans;
    }
};
