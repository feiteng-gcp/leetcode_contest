class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> ret;
        while(A.size()>1){
            int max_i = 0;
            for(int i=1; i<A.size(); ++i){
                if(A[i] > A[max_i])
                    max_i = i;
            }
            ret.push_back(max_i+1);
            ret.push_back(A.size());
            vector<int> next(A.rbegin(), A.rbegin()+A.size()-max_i-1);
            next.insert(next.end(), A.begin(), A.begin()+max_i);
            A = next;
        }
        return ret;
    }
};
