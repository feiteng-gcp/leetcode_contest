class Solution 
{
public:
    vector<int> pancakeSort(vector<int>& A)
    {
        vector<int> ans;
        int size = A.size();
        for(int l = size;l > 0;l--)
        {
            int i = 0;
            for(;i < l;i++)
                if(A[i] == l)
                    break;
            if(i == l-1)
                continue;
            if(i != 0)
            {
                ans.push_back(i+1);
                reverse(A.begin(), A.begin()+i+1);
            }
            ans.push_back(l);
            reverse(A.begin(), A.begin()+l);
        }
        return ans;
    }
};
