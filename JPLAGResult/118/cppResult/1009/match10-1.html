class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int>v;
        int n=A.size();
        for(int i=0;i<n;i++){
            int x=-1,idx;

            for(int j=0;j<n-i;j++){
                if(x<A[j]){
                    x=A[j];
                    idx=j;
                }
            }
            v.push_back(idx+1);
            v.push_back(n-i);
            reverse(A.begin(),A.begin()+idx+1);
            //cout<<idx<<endl;
            //cout<<A[0]<<A[1]<<A[2]<<A[3]<<endl;
            reverse(A.begin(),A.begin()+n-i);
            //cout<<idx<<endl;
            //cout<<A[0]<<A[1]<<A[2]<<A[3]<<endl;
        }
        return v;
    }
};
