class Solution {
private:
    int find_max_idx(vector<int>& A, int pos) {
        int res = 0;
        for (int i = 1; i < pos; i++) {
            if (A[i] > A[res]) {
                res = i;
            }
        }
        return res;
    }
    void reverse(vector<int>& A, int pos) {
        int begin = 0, end = pos-1;
        while (begin < end) {
            int t = A[begin];
            A[begin] = A[end];
            A[end] = t;
            begin++;
            end--;
        }
    }
    void print(vector<int>& A) {
        for (int i = 0; i < A.size(); i++) {
            cout << A[i] << " ";
        }
        cout << endl;
    }

public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> res;
        int l = A.size();
        for (int i = 0; i < l - 1; i++) {
            int step = find_max_idx(A, l - i);
            res.push_back(step+1);
            res.push_back(l-i);
            reverse(A, step+1);
            //print(A);
            reverse(A, l-i);
            //print(A);
        }
        //print(A);
        return res;
    }
};
