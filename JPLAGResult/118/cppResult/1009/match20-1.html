    vector<int> pancakeSort(vector<int>& v) {
        vector<int> ans;
        for (int n = v.size(); n > 0; n--) {
            for (int i = 0; i < n; i++)
                if (v[i] == n) {
                    ans.push_back(i+1);
                    reverse(v.begin(), v.begin()+i+1);
                    //rreverse(v, i+1);
                    ans.push_back(n);
                    reverse(v.begin(), v.begin()+n);
                    //rreverse(v, n);
                }
        }
        for (int i = 0; i < v.size(); i++)
            cout << v[i] << " ";
        
        return ans;
    }
};
