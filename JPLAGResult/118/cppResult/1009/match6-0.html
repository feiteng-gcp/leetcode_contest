class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> B = A;
        vector<int> ans;
        for (int i = B.size()-1; i >= 0; i--) {
            int mx = -1;
            int idx = 0;
            for (int j = 0; j <= i; j++) {
                if (B[j] > mx) {
                    mx = B[j];
                    idx = j;
                }
            }
            ans.push_back(idx+1);
            ans.push_back(i+1);
            reverse(B.begin(), B.begin()+idx+1);
            reverse(B.begin(), B.begin()+i+1);
            
        }

        return ans;
    }
};
