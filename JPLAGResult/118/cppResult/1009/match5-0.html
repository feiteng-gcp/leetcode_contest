class Solution {
public:
    void pancake_flip(vector<int>& A, int t) {
        int i = 0, j = t - 1;
        while (i < j) {
            swap(A[i], A[j]);
            i ++;
            j --;
        }
    }
    vector<int> pancakeSort(vector<int>& A) {
        int n = A.size();
        vector<int> ret;
        for (int i = 0;i < n;i ++) {
            int m = n - i;
            int k = 0;
            for (int j = 1;j < m;j ++) {
                if (A[j] > A[k]) k = j;
            }
            ret.push_back(k + 1);
            pancake_flip(A, k + 1);
            ret.push_back(m);
            pancake_flip(A, m);
        }
        return ret;
    }
};
