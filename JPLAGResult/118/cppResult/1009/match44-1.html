class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        int n = A.size();
        vector<int> output;
        for (int i = n; i >= 2; --i) {

            int index = -1;
            for (int j = 0; j < i; ++j) {
                if (A[j] == i) {
                    index = j;
                    break;
                }
            }
            if (index != i - 1) {
                if (index > 0) {
                    output.push_back(index + 1);
                    for (int k = 0; k < index; k++) {
                            int tmp = A[k];
                            A[k] = A[index];
                            A[index] = tmp;
                            index--;
                    }
                }
                output.push_back(i);
                int m = i - 1;
                for (int k = 0; k < m; k++) {
                    int tmp = A[k];
                    A[k] = A[m];
                    A[m] = tmp;
                    m--;
                }
            }
        }
        return output;
    }
};
