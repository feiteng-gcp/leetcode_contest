class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> ans;
        int len = A.size();
        for (int cur = len - 1; cur > 0; --cur){
            int idx = 0;
            for (int j = 1; j <= cur; ++j){
                if (A[j] > A[idx]) idx = j;  
            }
            
            if (idx < cur) {
                ans.push_back(idx + 1);
                ans.push_back(cur + 1);
            }
            
            for (int i = 0, j = idx; i < j; ++i,--j) {
                swap(A[i], A[j]);
            }
            for (int i = 0, j = cur; i < j; ++i,--j) {
                swap(A[i], A[j]);
            }
        }
        return ans;
    }
};
