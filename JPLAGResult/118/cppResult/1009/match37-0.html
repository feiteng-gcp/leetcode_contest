class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> ans;
        int n=A.size();
        for(int i=n-1;i>0;i--)
        {
            int maxv=0,maxi;
            int j;
            for(j=0;j<=i;j++)
            {
                if(A[j]>maxv)maxv=A[j],maxi=j;
            }
            //cout<<maxv<<endl;
            if(maxi+1!=1)ans.push_back(maxi+1);
            if(i+1!=1)ans.push_back(i+1);
            reverse(A.begin(),A.begin()+maxi+1);
            reverse(A.begin(),A.begin()+i+1);
            //for(auto &v:A)cout<<v<<" ";cout<<endl;
        }
        return ans;
    }
};
