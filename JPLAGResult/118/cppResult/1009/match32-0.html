class Solution {
public:
    vector<int> pancakeSort(vector<int>& A) {
        vector<int> ans;
        int n = A.size();
        for (int i = n - 1; i > 0; -- i) {
            int maxVal = A[0], maxId = 0;
            for (int j = 0; j <= i; ++ j) {
                if (maxVal < A[j]) {
                    maxVal = A[j];
                    maxId = j;
                }
            }
            
            flip(A, maxId + 1);
            flip(A, i + 1);
            ans.push_back(maxId + 1);
            ans.push_back(i + 1);
        }
        return ans;
    }
    
    void flip(vector<int> &A, int k) {
        for (int i = 0; i < k / 2; ++ i) {
            swap(A[i], A[k - i - 1]);
        }
    }
};
