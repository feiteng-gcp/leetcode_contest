class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        vector<int> results;
        for(int i=0;pow(x,i)<=bound;i++){
            for(int j=0;pow(y,j)<=bound;j++){
                if(pow(x,i)+pow(y,j)<=bound)
                    if(find(results.begin( ), results.end( ),pow(x,i)+pow(y,j))==results.end())
                        results.push_back(pow(x,i)+pow(y,j));
                if(y==1) break;
            }
            if(x==1) break;
        }
        return results;
    }
};
