class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {

        unordered_set<int> res;
        for(int i = 0;; i ++){
            int powx = pow(x, i);
            if(powx > bound) break;
            for(int j = 0;;j ++) {
                int t = powx + pow(y, j);
                if(t > bound) break;
                res.insert(t);
                if(y == 1) break;
            }
            if(x == 1) break;
        }
        return vector<int>(res.begin(), res.end());
    }
};
