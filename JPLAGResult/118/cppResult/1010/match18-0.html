class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        unordered_set<int> result;
        for(int xi = 1; xi<bound; xi*=x){
            for(int yi = 1; xi+yi<=bound; yi*=y){
                //cout<<xi<<" "<<yi<<" "<<xi+yi<<endl;
                result.insert(xi+yi);
                if(y==1)break;
            }
            if(x==1)break;
        }
        return vector<int>(result.begin(), result.end());
    }
};
