int power(int x, int i){
    if(i == 0)
        return 1;
    if(i == 1)
        return x;
    int m = i%2==1? x: 1;
    int a = power(x, i/2);
    return m*a*a;
}


class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        set<int> s;
        for(int i = 0; i < 20; ++i){
            int a = power(x, i);
            if(a >= bound)
                break;
            for(int j = 0; j < 20; ++j){
                int b = power(y, j);
                if(a+b > bound)
                    break;
                if(a+b <= bound)
                    s.insert(a+b);
            }
        }
        vector<int> v;
        copy(s.begin(), s.end(), back_inserter(v));
        return v;
    }
    

};
