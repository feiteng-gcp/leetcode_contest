class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        set<int> ans;
        for (int xx = 1; xx <= bound; xx *= x) {
            for (int yy = 1; xx + yy <= bound; yy *= y){
                ans.insert(xx + yy);

                if (y == 1) break;
            }
            if (x == 1) break;
        }
        vector<int> res;
        for (int a : ans) res.push_back(a);
        return res;
    }
};
