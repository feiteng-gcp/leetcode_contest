#include <bits/stdc++.h>
using namespace std;
using i64 = int64_t;
using vi = vector<i64>;
using vvi = vector<vi>;


class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        vector<int> ret;
        for (i64 a = 1; a <= 1e14; a *= x) {
            for (i64 b = 1; b <= 1e14; b *= y) {
                if (a + b <= bound) {
                    ret.push_back(a + b);
                }
                if (b == b * y) break;
            }
            if (a == a * x) break;
        }
        sort(ret.begin(), ret.end());
        ret.erase(unique(ret.begin(), ret.end()), ret.end());
        return ret;
    }
};
