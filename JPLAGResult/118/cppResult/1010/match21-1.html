class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        set<int> ret;
        long long f = 1;
        for(int i = 0; f <= bound; ++i, f *= x){
            long long s = 1;
            for(int j = 0; s <= bound; ++j, s *= y){
                if(f + s <= bound)
                    ret.insert(f + s);
                if(y == 1) break;
            }
            if(x == 1) break;
        }
        return vector<int>(ret.begin(), ret.end());
    }
};
