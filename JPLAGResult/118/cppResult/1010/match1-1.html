class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        unordered_set<int> s;
        for (int i = 0;; ++i) {
            int r1 = pow(x, i);
            for (int j = 0;; ++j) {
                int r = r1 + pow(y, j);
                if (r <= bound) s.insert(r);
                if (r > bound || y == 1) break;
            }
            if (r1 >= bound || x == 1) break;
        }
        vector<int> res;
        for (int num: s) res.push_back(num);
        return res;
    }
};
