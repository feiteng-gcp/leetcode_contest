class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        vector<int> ret;
        for (int i = 0; pow(x, i) <= bound && i <= 32; i++) {
            for (int j = 0; pow(y, j) + pow(x, i) <= bound && j <= 32; j++) {
                ret.push_back((int)(pow(x, i) + pow(y, j)));
            }
        }
        sort(ret.begin(), ret.end());
        ret.erase(unique(ret.begin(), ret.end()), ret.end());
        return ret;
    }
};
