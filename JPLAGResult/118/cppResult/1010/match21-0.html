class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        set<int> res;
        
        long long xx = 1;
        for (int i = 0; xx <= bound; i++, xx *= x) {
            long long yy = 1;
            for (int j = 0; yy <= bound; j++, yy *= y) {
                if (xx + yy <= bound) {
                    res.insert(xx + yy);
                }
                if (y == 1 && j > 1) break;
            }
            if (x == 1 && i > 1) break;
        }
        
        return std::vector<int>(res.begin(), res.end());
        
        
    }
};
