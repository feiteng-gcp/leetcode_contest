class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        set<int> S;
        for (int i = 0; ; i++) {
            int d = pow(x, i);
            if (d > bound) break;
            for (int j = 0; ; j++) {
                int k = pow(x, i) + pow(y, j);
                if (k > bound) break;
                S.insert(k);
                if (y == 1) break;
            }
            if (x == 1) break;
        }
        vector<int> ans;
        for (int i : S) {
            ans.push_back(i);
        }
        return ans;
    }
};
