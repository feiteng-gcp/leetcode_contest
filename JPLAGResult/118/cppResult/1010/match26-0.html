class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        vector<int> ans;
       for(int i=0;;i++)
       {
           for(int j=0;;j++)
           {
               // cout<<i<<" "<<j<<endl;
               int temp = pow(x,i)+pow(y,j);
               if(temp<=bound&&j<=bound){
                   
                   ans.push_back(temp);}
               else
                   break;    
           }
           if(pow(x,i)>bound||i>=bound)
               break;
           
       } 
        sort(ans.begin(),ans.end());
        auto iter = unique(ans.begin(), ans.end());
        ans.erase(iter, ans.end());
        return ans;
    }
};
