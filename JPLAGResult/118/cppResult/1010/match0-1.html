class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        int a = 1 , b = 1;
        set<int> ret;
        for(;a<=bound; a*=x){
            for(b=1; a+b<=bound; b*=y){
                ret.insert(a+b);
                if(y == 1) break;
            }
            if(x == 1) break;
        }
            
        return vector<int>(ret.begin(), ret.end());
    }
};
