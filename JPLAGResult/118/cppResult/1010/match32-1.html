class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        vector<int> res;
        for (int xx = 1; xx <= bound; xx *= x) {
            for (int yy = 1; yy <= bound; yy *= y) {
                if (xx + yy <= bound) {
                    bool flag = false;
                    for (int i = 0; i < res.size(); i ++) {
                        if (xx + yy == res[i]) {
                            flag = true;
                            break;
                        }
                    }
                    if (!flag) {
                        res.push_back(xx + yy);
                    }
                }
                if (y == 1) {
                    break;
                }
            }
            if (x == 1) {
                break;
            }
        }
        return res;
        
    }
};
