class Solution {
public:
    vector<int> powerfulIntegers(int x, int y, int bound) {
        set<int> unique;
        for(int i = 1;i < bound;i *= x){
            for (int j = 1;j + i <= bound;j *= y) {
                unique.insert(j + i);
                if (y == 1) break;
            }
            if (x == 1) break;
        }
        vector<int> result;
        for (auto&& u : unique) {
            result.emplace_back(u);
        }
        return result;
    }
};
