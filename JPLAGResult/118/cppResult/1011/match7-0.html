/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool helper(TreeNode* root, vector<int>& pre, int& cur, vector<int>& res) {
        if (!root || cur==pre.size()) return true;
        
        if (root->val != pre[cur]) return false;

        int tmp = cur+1;
        if (!(helper(root->left, pre, tmp, res))) {
            if (!root->right || root->right->val != pre[cur+1])
                return false;
            else {
                TreeNode* tmp = root->left;
                root->left = root->right;
                root->right = tmp;
                res.push_back(root->val);
            }
        }
        cur += 1;
        return helper(root->left, pre, cur, res) && helper(root->right, pre, cur, res);
    }
    
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> res;
        int cur = 0;
        if (helper(root, voyage, cur, res))
            return res;
        else
            return {-1};
    }
};
