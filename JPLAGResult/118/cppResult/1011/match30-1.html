/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int helper(vector<int> & r, TreeNode * root, vector<int> & voyage, int idx){
        if(!root) return idx;
        if(root->val != voyage[idx]) return -1;
        if(root->left && root->right){
            if(root->left->val == voyage[idx+1]){
                int idx_t = helper(r, root->left, voyage, idx+1);
                if(idx_t != -1){
                    idx_t = helper(r, root->right, voyage, idx_t);
                    return idx_t;
                }else{
                    return -1;
                }
            }else{
                r.push_back(root->val);
                int idx_t = helper(r, root->right, voyage, idx+1);
                if(idx_t != -1){
                    idx_t = helper(r, root->left, voyage, idx_t);
                    return idx_t;
                }else{
                    return -1;
                }
            }
        }else if(root->left){
            return helper(r, root->left, voyage, idx+1);
        }else if(root->right){
            return helper(r, root->right, voyage, idx+1);
        }else{
            return idx+1;
        }
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> r;
        int idx_t = helper(r, root, voyage, 0);
        if(idx_t != -1){
            return r;
        }else{
            return {-1};
        }
    }
};
