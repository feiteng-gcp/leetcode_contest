/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    vector<int>ret;
    bool f;
    int j;
    void dfs(TreeNode* root, vector<int>&a){

        if(!f||root==NULL||j>=a.size())return;
        if(root->val!=a[j]){
            f=false;
            return;
        }
        j++;
        if(root->left!=NULL&&root->right!=NULL){
            if(j+1>=a.size()){
                f=false;
                return;
            }
            if(root->left->val!=a[j]&&root->right->val!=a[j]){
                f=false;
                return;
            }
            if(root->left->val!=a[j]){
                ret.push_back(root->val);
                TreeNode* tmp=root->left;
                root->left=root->right;
                root->right=tmp;
            }
            dfs(root->left,a);
            dfs(root->right,a);
        }else{
            if(j>=a.size()&&(root->left!=NULL||root->right!=NULL)){
                f=false;
                return;
            }
            dfs(root->left,a);
            dfs(root->right,a);
        }
    }
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& a) {
        ret.clear();
        f=true;
        j=0;
        dfs(root,a);
        if(!f||j<a.size()){
            ret.clear();
            ret.push_back(-1);
            return ret;
        }
        return ret;
    }
};
