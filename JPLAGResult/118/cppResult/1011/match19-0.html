#include <bits/stdc++.h>
using namespace std;

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool rec(TreeNode* root, vector<int>& voyage, int& x, vector<int>& ans) {
        if (root == NULL || x == voyage.size()) return true;
        if (root->val != voyage[x]) return false;
        int N = voyage.size();
        x++;
        if (rec(root->left, voyage, x, ans))
            return rec(root->right, voyage, x, ans);
        else {
            ans.push_back(root->val);
            if (rec(root->right, voyage, x, ans))
                return rec(root->left, voyage, x, ans);
            else 
                return false;
        }
    }
    
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> ans;
        int x = 0;
        if (!rec(root, voyage, x, ans)) {
            ans.clear();
            ans.push_back(-1);
        }
        return ans;
    }
};
