/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    public:
        vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
            if(helper(root, voyage, 0, voyage.size()))
                return ans;
            return {-1};

        }
    private:
        vector<int> ans;
        bool helper(TreeNode *root, vector<int>& voyage, int start, int end) {
            if(root == NULL && start == end) return true;
            else if(root == NULL || start > end) return false;

            if(root->val != voyage[start]) return false;
            if(root->left == NULL) return helper(root->right, voyage, start + 1, end);
            if(root->right == NULL) return helper(root->left, voyage, start + 1, end);

            int leftroot = -1, rightroot = -1;
            for(int i = start; i < end; ++i) {
                if(voyage[i] == root->left->val)
                    leftroot = i;
                if(voyage[i] == root->right->val)
                    rightroot = i;
            }
            if(leftroot == -1 || rightroot == -1) return false;
            if(leftroot < rightroot)
                return helper(root->left, voyage, leftroot, rightroot) && helper(root->right, voyage, rightroot, end);
            else {
                ans.push_back(root->val);
                return helper(root->left, voyage, leftroot, end) && helper(root->right, voyage, rightroot, leftroot);
            }
        }
};
