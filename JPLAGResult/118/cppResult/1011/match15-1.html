/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool dfs(TreeNode* root, int lb, int rb, vector<int>&voy, vector<int>&ret) {
        if(root == NULL) {
            return (lb==rb);
        }
        if(lb==rb) return false;
        if(root->val != voy[lb]) return false;
        if(root->right == NULL) {
            return dfs(root->left, lb+1, rb, voy, ret);
        }
        if(root->left == NULL) {
            return dfs(root->right, lb+1, rb, voy, ret);
        }

        int pl=-1, pr=-1;
        for(int i=lb; i<rb; i++) {
            if(voy[i] == root->left->val) pl = i;
            if(voy[i] == root->right->val) pr = i;
        }
        if(pl == -1 || pr == -1) return false;
        if(pl == lb+1) {
            return dfs(root->left, lb+1, pr, voy, ret) && dfs(root->right, pr, rb, voy, ret);
        }
        if(pr == lb+1) {
            ret.push_back(root->val);
            return dfs(root->right, lb+1, pl, voy, ret) && dfs(root->left, pl, rb, voy, ret);
        }
        return false;
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> ret;
        int N = voyage.size();
        bool pos = dfs(root, 0, N, voyage, ret);
        if(pos) return ret;
        else return vector<int>{-1};
    }
};
