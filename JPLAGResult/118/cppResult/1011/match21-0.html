/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> ans;
        if(helper(root,0,voyage.size()-1,voyage,ans))return ans;
        return vector<int>({-1});
    }
    
    bool helper(TreeNode* root,int l,int r,vector<int>& voyage,vector<int>& ans)
    {
        //cout<<root->val<<" "<<l<<" "<<r<<" "<<voyage[l]<<endl;
        if(root->val!=voyage[l])return false;
        if(l==r)return true;
        int leftIndex=-1,rightIndex=-1;
        for(int i=l+1;i<=r;i++)
        {
            if(root->left&&voyage[i]==root->left->val)leftIndex=i;
            if(root->right&&voyage[i]==root->right->val)rightIndex=i;
        }
        //cout<<root->val<<" "<<leftIndex<<" "<<rightIndex<<endl;
        if(root->left&&leftIndex==-1||root->right&&rightIndex==-1)return false;
        if(root->left==nullptr&&root->right==nullptr)return false;
        else if(root->left==nullptr)
        {
            if(rightIndex!=l+1)return false;
            return helper(root->right,rightIndex,r,voyage,ans);
        }
        else if(root->right==nullptr)
        {
            if(leftIndex!=l+1)return false;
            return helper(root->left,leftIndex,r,voyage,ans);
        }
        else
        {
            if(leftIndex==l+1)
            {
                return helper(root->left,leftIndex,rightIndex-1,voyage,ans)&&helper(root->right,rightIndex,r,voyage,ans);
            }
            else if(rightIndex==l+1)
            {
                ans.push_back(root->val);
                //cout<<helper(root->left,rightIndex,leftIndex-1,voyage,ans)<<" "<<helper(root->right,leftIndex,r,voyage,ans)<<endl;
                return helper(root->right,rightIndex,leftIndex-1,voyage,ans)&&helper(root->left,leftIndex,r,voyage,ans);
                //return helper(root->left,leftIndex,rightIndex-1,voyage,ans)&&helper(root->right,rightIndex,r,voyage,ans);
            }
            else
            {
                return false;
            }
        }
        
    }
};
