/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    unordered_map<TreeNode*, int> size;
    int getNum(TreeNode* root) {
        if (size.find(root) != size.end())
            return size[root];
        else {
            if (!root) return 0;
            else {
                size[root] = getNum(root->left) + getNum(root->right) + 1;
                return size[root];
            }
        }
    }
    vector<int> helper(TreeNode* root, vector<int>& vo, int a, int b) {
        int n = getNum(root);
        if (n != b - a) return {-1};
        if (n > 0) {
            int nl = getNum(root->left), nr = getNum(root->right);
            if (vo[a] != root->val) return {-1};
            vector<int> hl = helper(root->left, vo, a+1, a+1+nl), hr = helper(root->right, vo, a+1+nl, b);
            vector<int> hl1 = helper(root->left, vo, a+1+nr, b), hr1 = helper(root->right, vo, a+1, a+1+nr);
            int len = ((hl.size() && hl[0] == -1) || (hr.size() && hr[0] == -1)) ? INT_MAX : hl.size() + hr.size();
            int len1 = ((hl1.size() && hl1[0] == -1) || (hr1.size() && hr1[0] == -1)) ? INT_MAX : hl1.size() + hr1.size();
            if (len == INT_MAX && len1 == INT_MAX) return {-1};
            else if (len <= len1) {
                for (int r : hr) hl.push_back(r);
                return hl;
            }
            else {
                for (int r : hr1) hl1.push_back(r);
                hl1.push_back(root->val);
                return hl1;
            }
        }
        else return b == a ? vector<int>(0) : vector<int>(1, -1);
    }
public:

    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& vo) {
        int N = getNum(root);
        if (vo.size() != N) return {-1};
        return helper(root, vo, 0, N);
    }
};
