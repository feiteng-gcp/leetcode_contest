/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool match(TreeNode* root, const vector<int>& voyage, int start, int range) {
        if (range < 0) return false;
        if (root->val != voyage[start]) return false;
        if (root->left == nullptr && root->right == nullptr) return range == 1;
        if (root->left == nullptr) return match(root->right, voyage, start + 1, range - 1);
        if (root->right == nullptr) return match(root->left, voyage, start + 1, range - 1);
        if (root->left->val == voyage[start + 1]) {
            int j = -1;
            for (int k = 2;k < range;k ++) {
                if (voyage[start + k] == root->right->val) {
                    j = k;
                    break;
                }
            }
            if (j == -1) return false;
            return match(root->left, voyage, start + 1, j - 1) && match(root->right, voyage, start + j, range - j);
        } else if (root->right->val == voyage[start + 1]) {
            to_flip.insert(root->val);
            int j = -1;
            for (int k = 2;k < range;k ++) {
                if (voyage[start + k] == root->left->val) {
                    j = k;
                    break;
                }
            }
            if (j == -1) return false;
            return match(root->right, voyage, start + 1, j - 1) && match(root->left, voyage, start + j, range - j);
        } else return false;
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        if (!match(root, voyage, 0, voyage.size())) {
            return {-1};
        }
        vector<int> ret;
        for (const auto& r : to_flip) {
            ret.push_back(r);
        }
        return ret;
    }
    set<int> to_flip;
};
