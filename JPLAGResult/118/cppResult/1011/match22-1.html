/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int gs(TreeNode* root) {
        if (!root) return 0;
        return 1 + gs(root->left) + gs(root->right);
    }
    
    bool go(vector<int> &res, TreeNode* root, vector<int>& V, int a, int b) {
        if (!root) {
            if (a==b) return 1;
            else return 0;
        }
        if (a>=b) return 0;
        if (root->val != V[a]) return 0;
        ++a;
        int sl = gs(root->left);
        int sr = gs(root->right);
        if (go(res,root->left,V,a,a+sl) && go(res,root->right,V,a+sl,b)) {
            ; //ok
        } else {
            res.push_back(root->val);
            if (!go(res,root->right,V,a,a+sr) || !go(res,root->left,V,a+sr,b)) {
                return 0;
            }

        }
        return 1;
    }
    
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> res;
        if (!go(res,root,voyage, 0, voyage.size())) {
            return {-1};
        }
        return res;
    }
};
