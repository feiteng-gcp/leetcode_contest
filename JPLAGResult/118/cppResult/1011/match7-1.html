/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool helper(TreeNode* root, vector<int>& voyage, int &index, vector<int> & ret)
    {
        if (root==nullptr)
            return true;
        if (index>=voyage.size() || root->val!=voyage[index])
            return false;
        int left=index+1;
        if (helper(root->left, voyage, left, ret))
        {
            int right=left;
            if (helper(root->right, voyage, right, ret))
            {
                index=right;
                return true;
            }
            return false;
        }
        int right=index+1;
        if (helper(root->right, voyage, right, ret))
        {
            int left=right;
            if (helper(root->left, voyage, left, ret))
            {
                index=left;
                ret.push_back(root->val);
                return true;
            }
            return false;
        }
        return false;
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> ret;
        int index=0;
        if (helper(root, voyage, index, ret) && index==voyage.size())
            return ret;
        return vector<int>{-1};
    }
};
