/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vector<int> voy, ans;
    bool impossible;
    int idx;
    

    void dfs(TreeNode* root) {
        if (!root) return;
        if (root->val != voy[idx]) {
            // cout << root->val <<  "  asdf\n";
            impossible = true;
            return;
        }
        if (!root->left && !root->right) {
            idx++;
            return;
        }
        if (!root->left) {
            if (root->right->val != voy[idx+1]) {
                impossible = true;
                return;
            }
            idx++;
            dfs(root->right);
            return;
            
        }
        if (!root->right) {
            if (root->left->val != voy[idx+1]) {
                impossible = true;
                return;
            }
            idx++;
            dfs(root->left);
            return;
        }
        
        if (root->left->val != voy[idx+1] && root->right->val != voy[idx+1]) {
            impossible = true;
            return;
        }
        if (root->left->val != voy[idx+1]) {
            ans.push_back(root->val);
            idx++;
            dfs(root->right);
            dfs(root->left);
            return;
        }
        idx++;
        dfs(root->left);
        dfs(root->right);
    }
    
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        voy = voyage;
        ans.clear();
        idx = 0;
        impossible = false;
        dfs(root);
        if (impossible) {
            vector<int> a = {-1};
            return a;
        }
        return ans;
    }
};
