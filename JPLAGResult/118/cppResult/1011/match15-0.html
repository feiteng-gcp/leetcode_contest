/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> ans;
    vector<int> v;
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {        
        v = voyage;
        if (!DFS(root, 0, v.size() - 1)) return {-1};
        return ans;
    }
    
    bool DFS(TreeNode* root, int l, int r) {        
        //printf("--%d %d %d\n", root->val, l, r);
        if (root->val != v[l]) return false;
        if (root->left == NULL && root->right == NULL) return l == r;
        if (root->left == NULL || root->right == NULL) return DFS(root->left == NULL ? root->right : root->left, l + 1, r);
        int left = -1, right = -1;
        for (int i = l + 1; i <= r; i++) {
            if (v[i] == root->left->val) left = i;
            if (v[i] == root->right->val) right = i;
        }
        if (left == -1 || right == -1) return false;
        //printf("%d : %d %d\n", root->val,left, right);
        if (left == l + 1) {
            return DFS(root->left, left, right - 1) && DFS(root->right, right, r);            
        } else if (right == l + 1) {
            ans.push_back(root->val);
            return DFS(root->right, right, left - 1) && DFS(root->left, left, r);
        }
        return false;
    }
};
