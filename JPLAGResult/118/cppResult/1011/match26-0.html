/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> result;
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        result.clear();
        flip(root, voyage, 0);
        return result;
    }
    int flip(TreeNode* root, vector<int>& voyage, int id){
        if (root->val != voyage[id]) {
            result = {-1};
            return -1;
        }
        if (root->left && root->right) {
            if (root->left->val != voyage[id + 1]) {
                swap(root->left, root->right);
                result.emplace_back(root->val);
            }
            auto res = flip(root->left, voyage, id + 1);
            if (res == -1) return -1;
            res = flip(root->right, voyage, res);
            return res;
        } else if (root->left) {
            return flip(root->left, voyage, id + 1);
        } else if (root->right) {
            return flip(root->right, voyage, id + 1);
        } else {
            return id + 1;
        }
    }
};
