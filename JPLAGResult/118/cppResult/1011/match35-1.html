/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        if(!root) return vector<int>();
        
        if(root -> val != voyage[0]) return vector<int>({-1});
        
        voyage = vector<int>(voyage.begin() + 1, voyage.end());
        
        if(root -> left == NULL) swap(root -> left, root -> right);
        
        vector<int> cr;
        
        if(root -> left && root -> left -> val != voyage[0]){
            cr.push_back(root -> val);
            swap(root -> left, root -> right);
        }
        
        vector<int> ret = flipMatchVoyage(root -> left, voyage);
        if(ret.size() && ret[0] == -1) return ret;
        
        vector<int> ot = flipMatchVoyage(root -> right, voyage);
        if(ot.size() && ot[0] == -1) return ot;
        
        for(int el: ret) cr.push_back(el);
        for(int el: ot) cr.push_back(el);
        
        return cr;
        
    }
};
