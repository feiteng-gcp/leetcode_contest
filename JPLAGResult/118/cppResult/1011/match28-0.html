/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int getSize(TreeNode *cur) {
        if (cur == nullptr) return 0;
        return 1 + getSize(cur->left) + getSize(cur->right);
    }
    
    bool dfs(TreeNode *cur, vector<int>& voy, vector<int>& ans, int i) {
        if (cur==nullptr) return true;
        if (i >= (int)voy.size()) return false; // index out of bounds
        if (cur->val != voy[i]) return false;
        int left = getSize(cur->left);
        int right = getSize(cur->right);
        if (dfs(cur->left, voy, ans, i+1) && dfs(cur->right, voy, ans, i+1+left)) {
            return true;
        }
        if (dfs(cur->right, voy, ans, i+1) && dfs(cur->left, voy, ans, i+1+right)) {
            ans.push_back(voy[i]);
            return true;
        }
        return false;
    }
    
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        
        vector<int> ans;
        
        if (dfs(root, voyage, ans, 0)) return ans;
        return {-1};
    }
};
