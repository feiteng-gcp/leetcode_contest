/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        if (count(root) != voyage.size())
            return {-1};
        
        if (!root)
            return {};
        
        vector<int> result;
        int * tmp = &voyage[0];

        bool br = dfs(root, tmp, result);
        if (!br)
            return {-1};
        return result;
    }
    
    bool dfs(TreeNode * root, int *& voyage, vector<int> & result) {
        if (!root)
            return true;
        
        if (root->val != *voyage)
            return false;
        
        ++voyage;
        
        if (root->right && root->right->val == *voyage) {
            if (root->left)
                result.push_back(root->val);
            bool br = dfs(root->right, voyage, result);
            if (!br)
                return false;
            return dfs(root->left, voyage, result);
        } else {
            bool br = dfs(root->left, voyage, result);
            if (!br)
                return false;
            return dfs(root->right, voyage, result);
        }
    }
    
    int count(TreeNode * node) {
        if (!node)
            return 0;
        return count(node->left) + count(node->right) + 1;
    }
};
