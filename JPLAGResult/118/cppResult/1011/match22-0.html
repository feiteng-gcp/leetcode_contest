/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
bool preOrder(TreeNode* root, vector<int>& voyage, vector<int>& result) {
    if (!root) {
        return voyage.empty();
    }
    
    if (voyage.empty() || voyage[0] != root->val) {
        return false;
    }
    
    if (root->left && !root->right) {
        vector<int> temp(voyage.begin() + 1, voyage.end());
        return preOrder(root->left, temp, result);
    }
    
    if (!root->left && root->right) {
        vector<int> temp(voyage.begin() + 1, voyage.end());
        return preOrder(root->right, temp, result);
    }
    
    if (!root->left && !root->right) {
        return voyage.size() == 1;
    }
    
    int index1 = find(voyage.begin() + 1, voyage.end(), root->left->val) - voyage.begin();
    int index2 = find(voyage.begin() + 1, voyage.end(), root->right->val) - voyage.begin();
    if (index1 == -1 || index2 == -1) {
        return false;
    }
    
    if (index1 == 1) {
        vector<int> temp1(voyage.begin() + 1, voyage.begin() + index2);
        vector<int> temp2(voyage.begin() + index2, voyage.end());
        return preOrder(root->left, temp1, result) && preOrder(root->right, temp2, result);
    }
    
    if (index2 == 1) {
        result.push_back(root->val);
        vector<int> temp1(voyage.begin() + 1, voyage.begin() + index1);
        vector<int> temp2(voyage.begin() + index1, voyage.end());
        return preOrder(root->left, temp2, result) && preOrder(root->right, temp1, result);
    }
    
    return false;
}
    

    
    
vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
    vector<int> result;
    if (!preOrder(root, voyage, result)) {
        return {-1};
    }
    return result;
}
};
