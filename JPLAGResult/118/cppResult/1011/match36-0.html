/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
	vector<int> Result;
	
	bool Travel(TreeNode* root, vector<int>& V, int& I)
	{
		if(root)
		{
			//cout << root->val << " " << V[I] << " " << I << endl;
			
			if(root->val == V[I])
			{
				I++;
				
				if((root->left) && (root->right))
				{
					if(root->left->val == V[I])
					{
						return (Travel(root->left, V, I) && Travel(root->right, V, I));
					}
					else
					{
						Result.push_back(root->val);
						return (Travel(root->right, V, I) && Travel(root->left, V, I));
					}
				}
				else
				{
					return (Travel(root->left, V, I) && Travel(root->right, V, I));
				}
			}
			else
			{
				return false;
			}
		}
		
		return true;
	}
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
		int i = 0;
		
		if(Travel(root, voyage, i))
		{
			return Result;
		}
		else
		{
			return vector<int>(1, -1);
		}
    }
};
