/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> res;
    vector<int> vis;
    bool dfs(TreeNode *root, const vector<int> &voyage){
        if(root == nullptr){
            return true;
        }
        if(root->val != voyage[vis.size()]){
            return false;
        }
        vis.push_back(root->val);
        size_t size = vis.size();
        if(dfs(root->left, voyage) && dfs(root->right, voyage)){
            return true;
        }
        vis.resize(size);
        swap(root->left, root->right);
        if(dfs(root->left, voyage) && dfs(root->right, voyage)){
            res.push_back(root->val);
            return true;
        }
        vis.resize(size);

        return false;
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        if(root == nullptr){
            return vector<int>();
        }
        if(dfs(root, voyage)){
            return res;
        }
        return vector<int>(1, -1);
    }
};
