/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void dfs(TreeNode* node, vector<int>::iterator &vo, vector<int> &res, bool &res_flag) {
        if (node == NULL) {
            return;
        }
        if (node->val != *vo) {
            printf("false, node->val = %d\n", node->val);
            res_flag = false;
            return;
        }
        vo ++;
        if (node->left == NULL && node->right == NULL) {
            return;
        } else if (node->left == NULL) {
            dfs(node->right, vo, res, res_flag);

        } else if (node->right == NULL) {
            dfs(node->left, vo, res, res_flag);
        } else {
            if (node->left->val == *vo) {
                dfs(node->left, vo, res, res_flag);
                dfs(node->right, vo, res, res_flag);
            } else {
                res.push_back(node->val);
                dfs(node->right, vo, res, res_flag);
                dfs(node->left, vo, res, res_flag);
            }
        }
    }
    
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> res;
        bool res_flag = true;
        vector<int>::iterator vo = voyage.begin();
        dfs(root, vo, res, res_flag);
        if (res_flag == false) {
            res.clear();
            res.push_back(-1);
        }
        return res;
    }
};
