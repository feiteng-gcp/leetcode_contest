/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

int dfs(TreeNode *root, vector<int>& result, vector<int>& newvec, vector<int>& voyage, int index)
{
	if (!root)
	{
		return index;
	}
	if (index >= voyage.size())
	{
		return index;
	}
	newvec.push_back(root->val);
	index += 1;
	if (root->left && root->left->val != voyage[index])
	{
		index = dfs(root->right, result, newvec, voyage, index);
		index = dfs(root->left, result, newvec, voyage, index);
		result.push_back(root->val);
	}
	else
	{
		index = dfs(root->left, result, newvec, voyage, index);
		index = dfs(root->right, result, newvec, voyage, index);
	}
	return index;
}

vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
	vector<int> result, newvec;
	dfs(root, result, newvec, voyage, 0);
	if (newvec.size() != voyage.size())
	{
		return vector<int>{-1};
	}
	for (int i = 0; i < newvec.size(); ++i)
	{
		if(newvec[i]!=voyage[i])
		{
			return vector<int>{-1};
		}
	}
	return result;
}
};
