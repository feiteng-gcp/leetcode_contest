/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> a;
    int now = 0;
    bool flag = true;
    void dfs(TreeNode* root, vector<int>& voyage){
        if(!flag){
            return;
        }
        int exp = voyage[now];
        if(root->val !=exp){
            a.clear();
            a.push_back(-1);
            flag = false;
            return;
        }
        if(now+1 >= voyage.size()){
            return;
        }
        now++;
        if(!root->left && !root->right){
            return;
        }
        else if(!root->left){
            dfs(root->right, voyage);
        }
        else if(!root->right){
            dfs(root->left, voyage);
        }
        else{
            if(root->left->val == voyage[now]){
                dfs(root->left, voyage);
                dfs(root->right, voyage);
            }
            else{
                a.push_back(root->val);
                dfs(root->right,voyage);
                dfs(root->left, voyage);
            }
        }
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        dfs(root, voyage);
        return a;
    }
};
