/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> ret;
    bool can;
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        can = true;
        int ind = 0;
        traverse(root, voyage, ind);
        if (!can || ind < voyage.size()) {
            vector<int> rr;
            rr.push_back(-1);
            return rr;
        }
        return ret;
    }
    
    void traverse(TreeNode* root, vector<int>& voyage, int& ind) {
        if (root == NULL || ind >= voyage.size()) return;
        if (voyage[ind++] != root->val) {
            can = false;
            return;
        }
        if (root->left == NULL) {
            traverse(root->right, voyage, ind);
        } else if (root->right == NULL) {
            traverse(root->left, voyage, ind);
        } else {
            if (root->left->val == voyage[ind]) {
                traverse(root->left, voyage, ind);
                traverse(root->right, voyage, ind);
            } else {
                ret.push_back(root->val);
                traverse(root->right, voyage, ind);
                traverse(root->left, voyage, ind);
            }
        }
    }
};
