/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> ret;
    int tarversal(TreeNode* root, vector<int>& voyage, int i){
        if(root == nullptr) return i;
        if(i == -1 || i>voyage.size() || root->val != voyage[i]) return -1;
        ++i;
        int j = tarversal(root->left, voyage, i);
        if(j != -1)
            j = tarversal(root->right, voyage, j);
        else{
            ret.push_back(root->val);
            j = tarversal(root->right, voyage, i);
            j = tarversal(root->left, voyage, j);
        } 
        return j;
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        int res = tarversal(root, voyage, 0);
        if(res == -1 || res!=voyage.size())
            return {-1};
        return ret;
    }
};
