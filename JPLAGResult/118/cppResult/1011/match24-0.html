/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    bool flag;
    int n;
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& v) {
        vector<int> res;
        if (!root) return res;
        n = v.size();
        int i = 0;
        flag = true;
        helper(root, v, i, res);
        //cout<<i<<endl;
        if (flag && i == n - 1) return res;
        return {-1};
    }
    
    void helper(TreeNode* root, vector<int>& v, int& i, vector<int>& res) {
        if (!root || !flag) return;    
        if (i >= n) {
            flag = false;
            return;
        }
        if (i < n && root->val != v[i]) {
            flag = false;
            return;
        }
        
        if (root->left) {
            if (i >= n - 1) {
                flag = false;
                return;
            }
            if (root->left->val == v[i + 1]) {
                ++i;
                helper(root->left, v, i, res);
            }
            else {
                if (root->right && root->right->val == v[i + 1]) {
                    //cout<<i<<endl;
                    swap(root->left, root->right);
                    res.emplace_back(root->val);
                    ++i;
                    helper(root->left, v, i, res);
                }
                else {
                    flag = false;
                    return;
                }
            }
        }
        if (root->right) {
            if (i >= n - 1) {
                flag = false;
                return;
            }
            else if (root->right->val != v[i + 1]) {
                flag = false;
                return;
            }
            else {
                ++i;
                helper(root->right, v, i, res);
            }
        }
        
    }
};
