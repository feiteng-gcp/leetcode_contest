/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void traverse(TreeNode* x, vector<int>* out) {
        if (x == nullptr) return;
        out->push_back(x->val);
        traverse(x->left, out);
        traverse(x->right, out);
    }

    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        if (root == nullptr && voyage.size() == 0) { return vector<int>(); }
        if (root == nullptr || root->val != voyage[0]) { return vector<int>(1, -1); }

        for (int flip = 0; flip <= 1; flip++, swap(root->left, root->right)) {
            vector<int> tr_left;
            traverse(root->left, &tr_left);

            vector<int> voy_left(voyage.begin() + 1, voyage.begin() + (1 + tr_left.size()));
            vector<int> voy_right(voyage.begin() + (1 + tr_left.size()), voyage.end());

            set<int> s1(tr_left.begin(), tr_left.end());
            set<int> s2(voy_left.begin(), voy_left.end());
            if (s1 != s2) continue;

            vector<int> res1 = flipMatchVoyage(root->left, voy_left);
            if (!res1.empty() && res1[0] == -1) continue;

            vector<int> res2 = flipMatchVoyage(root->right, voy_right);
            if (!res2.empty() && res2[0] == -1) continue;
            
            vector<int> res;
            for (int x : res1) res.push_back(x);
            for (int x : res2) res.push_back(x);
            if (flip) res.push_back(root->val);
            return res;
        }
        
        return vector<int>(1, -1);
    }
};
