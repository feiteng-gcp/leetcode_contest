/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    bool isValid = true;
    void dfs(TreeNode* root, vector<int>& voyage, int begin, int end, vector<int>& res) {
        if (!isValid) {
            return;
        }
        if (root->left == nullptr && root->right == nullptr) {

            return;
        } else if (root->left == nullptr) {
            int rightIdx = -1;
            for (int i = begin; i < end; ++i) {
                if (root->right->val == voyage[i]) {
                    rightIdx = i;
                }
            }
            if (begin+1 != rightIdx) {
                isValid = false;
                return;
            }
            dfs(root->right, voyage, begin+1, end, res);

        } else if (root->right == nullptr) {
            int leftIdx = -1;
            for (int i = begin; i < end; ++i) {
                if (root->left->val == voyage[i]) {
                    leftIdx = i;
                }
            }
            if (begin+1 != leftIdx) {
                isValid = false;
                return;
            }
            dfs(root->left, voyage, begin+1, end, res);
        } else {
            int leftIdx = -1, rightIdx = -1;
            for (int i = begin; i < end; ++i) {
                if (root->left->val == voyage[i]) {
                    leftIdx = i;
                }
                if (root->right->val == voyage[i]) {
                    rightIdx = i;
                }
            }
            if (leftIdx < 0 || rightIdx < 0) {
                isValid = false;
                return;
            }
            if (begin + 1 == leftIdx) {
                dfs(root->left, voyage, leftIdx, rightIdx, res);
                dfs(root->right, voyage, rightIdx, end, res);
            } else if (begin + 1 == rightIdx) {
                res.push_back(root->val);
                dfs(root->right, voyage, rightIdx, leftIdx, res);
                dfs(root->left, voyage, leftIdx, end, res);
            } else {
                isValid = false;
            }
        }
        
    }
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> res;
        dfs(root, voyage, 0, voyage.size(), res);
        if (!isValid) {
            res.clear();
            res.push_back(-1);
        }
        return res;
    }
};
