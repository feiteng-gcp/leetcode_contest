/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        if (root == nullptr)
            return vector<int>(0);
        
        if (root -> val != voyage[0])
            return vector<int> (1, -1);
        
        if (root -> left == nullptr && root -> right == nullptr)
            return vector<int> (0);
        
        if (root -> left == nullptr) {
        
            vector<int> temp = vector<int> (voyage.begin() + 1, voyage.end());
            vector<int> result = flipMatchVoyage(root -> right, temp);
            if (result.size() > 0 && result[0] == -1) return vector<int> (1, -1);
            return result;
        }
        
        if (root -> right == nullptr) {
            vector<int> temp = vector<int> (voyage.begin() + 1, voyage.end());
            vector<int> result = flipMatchVoyage(root -> left, temp);
            if (result.size() > 0 && result[0] == -1) return vector<int> (1, -1);
            return result;
        }
        
        if (root -> left -> val== voyage[1]) {
            int start = -1;
            for (int i = 2; i < voyage.size(); i++) {
                if (voyage[i] == root -> right -> val) {
                    start = i;
                    break;
                }
            }
            if (start == -1) return vector<int> (1, -1);
            vector<int> temp = vector<int> (voyage.begin() + 1, voyage.begin() + start);
            vector<int> result1 = flipMatchVoyage(root -> left, temp);
            temp = vector<int> (voyage.begin() + start, voyage.end());
            vector<int> result2 = flipMatchVoyage(root -> right, temp);
            for (auto r1 : result1) {
                if (r1 == -1) return vector<int> (1, -1);
            }
            for (auto r2 : result2) {
                if (r2 == -1)
                    return vector<int> (1, -1);
                result1.push_back(r2);
                    
            }
            return result1;
        }
        
        if (root -> right -> val == voyage[1]) {
            int start = -1;
            for (int i = 2; i < voyage.size(); i++) {
                if (voyage[i] == root -> left -> val) {
                    start = i;
                    break;
                }
            }
            if (start == -1) return vector<int> (1, -1);
            vector<int> temp = vector<int> (voyage.begin() + 1, voyage.begin() + start);
            vector<int> result1 = flipMatchVoyage(root -> right, temp);
            temp = vector<int> (voyage.begin() + start, voyage.end());
            vector<int> result2 = flipMatchVoyage(root -> left, temp);
            for (auto r1 : result1) {
                if (r1 == -1) return vector<int> (1, -1);
            }
            for (auto r2 : result2) {
                if (r2 == -1)
                    return vector<int> (1, -1);
                result1.push_back(r2);
                
            }
            result1.push_back(root -> val);
            return result1;
        }
        return vector<int> (1, -1);
    }
};
