/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int N,cur;
    vector<int> ans;
    bool traverse(TreeNode* root, vector<int>& voyage)
    {
        if(root==NULL)return true;
        if(root->val!=voyage[cur])return false;
        cur++;
        if(root->left!=NULL&&root->right!=NULL)
        {
            if(voyage[cur]==root->right->val)
            {
                ans.push_back(root->val);
                return traverse(root->right,voyage)&&traverse(root->left,voyage);
            }
            else return traverse(root->left,voyage)&&traverse(root->right,voyage);
        }
        else
        {
            return traverse(root->left,voyage)&&traverse(root->right,voyage);
        }
    }
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        N=voyage.size();
        cur=0;
        ans.clear();
        bool ok=traverse(root,voyage);
        if(ok)return ans;
        else
        {
            ans.clear();
            ans.push_back(-1);
            return ans;
        }
    }
};
