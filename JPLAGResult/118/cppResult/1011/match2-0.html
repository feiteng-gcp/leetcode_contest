/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> ans; 
    vector<int> order;
    int x = 0;
    bool impossible = false;
    vector<int> flipMatchVoyage(TreeNode* root, vector<int>& voyage) {
        vector<int> m1;
        m1.push_back(-1);
        order = voyage;
        /*
        stack<TreeNode*> s;
        s.push(root);
        int i = 0;
        while(!s.empty()) {
            TreeNode* node = s.pop();
            if (node->val != voyage[i++]) return m1;
            
        }
        */
        dfs(root);
        if (impossible) return m1;
        else return ans;
    }
    
    void dfs(TreeNode* node) {
        if (impossible) return;
        if (node == NULL) return;
        if (x >= order.size() || node->val != order[x++]) {
            cout << order.size() << " order size \n";
            cout << node->val << " " << x << endl;
            impossible = true;
            return;
        }
        if (node->left != NULL && node->left->val != order[x]) {
            cout << "here";
            ans.push_back(node->val);
            TreeNode* temp = node->left;
            node->left = node->right;
            node->right = temp;
        }
        dfs(node->left);
        dfs(node->right);
    }
};
