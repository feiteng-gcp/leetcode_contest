/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
     typedef TreeNode* pTreeNode;
   private:
    void helper(pTreeNode root, const vector<int>& voyage, int voyst,
                vector<int>& ret) {
        if (root == nullptr) return;
        if (!voyage.empty() && voyage.back() == -1) return;

        if (root->val != voyage[voyst]) {
            ret.push_back(-1);
            return;
        }

        // if (voyst == voyage.size() - 1) return;
        if (root->left == nullptr && root->right == nullptr) return;

        if (root->left != nullptr) {
            if (root->left->val == voyage[voyst + 1]) {
                helper(root->left, voyage, voyst + 1, ret);
                if (root->right != nullptr) {
                    auto it = find(voyage.begin() + voyst, voyage.end(),
                                   root->right->val);
                    if (it != voyage.end()) {
                        int idx = distance(voyage.begin(), it);
                        helper(root->right, voyage, idx, ret);
                    } else {
                        ret.push_back(-1);
                        return;
                    }
                }
            } else {
                if (root->right != nullptr) {
                    if (root->right->val == voyage[voyst + 1]) {
                        ret.push_back(root->val);
                        helper(root->right, voyage, voyst + 1, ret);
                        auto it = find(voyage.begin() + voyst, voyage.end(),
                                       root->left->val);
                        if (it != voyage.end()) {
                            int idx = distance(voyage.begin(), it);
                            helper(root->left, voyage, idx, ret);

                        } else {
                            ret.push_back(-1);
                            return;
                        }
                    } else {
                        ret.push_back(-1);
                        return;
                    }
                } else {
                    ret.push_back(-1);
                    return;
                }
            }
        } else {
            if (root->right->val == voyage[voyst + 1]) {
                helper(root->right, voyage, voyst + 1, ret);
            } else {
                ret.push_back(-1);
                return;
            }
        }
    }

   public:
    vector<int> flipMatchVoyage(pTreeNode root, vector<int>& voyage) {
        vector<int> ret;
        helper(root, voyage, 0, ret);

        if (!ret.empty()) {
             if(find(ret.begin(), ret.end(), -1)==ret.end())
                return ret;
            else
                return {-1};
        } else
            return ret;
    }
};
