class Solution {
public:
    struct Infor {
        int a, b, c;
        long long fz, fm;
        
        void get(string v) {
            a = b = c = 0;
            int dot = -1, p = -1;
            long long fm1 = 1, fm2 = 1;
            
            for (int i = 0; i < v.size(); i++) {
                if (v[i] == '.') {
                    dot = i;
                } else if (v[i] == '(') {
                    p = i;
                } else if (v[i] != ')') {
                    if (dot == -1) {
                        a = a * 10 + v[i] - 48;
                    } else if (p == -1) {
                        b = b * 10 + v[i] - 48;
                        fm1 *= 10;
                    } else {
                        c = c * 10 + v[i] - 48;
                        fm2 *= 10;
                    }
                }
            }
            
            if (fm2 != 1) {
                fm2--;
            }
            
            fm2 *= fm1;
            fm = fm1 * fm2;
            fz = fm2 * b + fm1 * c;
            long long g = __gcd(fm, fz);
            fm /= g;
            fz /= g;
            fz += fm * a;
            cout << fz << '/' << fm << endl;
        }
    } s, t;
    
    bool isRationalEqual(string S, string T) {
        s.get(S);
        t.get(T);
        
        if (s.fz == t.fz && s.fm == t.fm) {
            return true;
        } else {
            return false;
        }
    }
};
