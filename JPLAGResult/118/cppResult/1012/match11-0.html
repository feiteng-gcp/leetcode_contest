class Solution {
public:
    void parse(string& str, vector<int>& inter, vector<int>& dec) {
        int i = 0;
        for (; i < str.size() && str[i] != '.'; ++i) {
            inter.push_back(str[i] - '0');
        }
        ++i;
        for (; i < str.size(); ++i) {
            if (str[i] == '(') {
                break;
            }
            dec.push_back(str[i] - '0');
        }
        if (i < str.size()) {
            ++i;
            int j = i;
            while (dec.size() < 100) {
                dec.push_back(str[i] - '0');
                ++i;
                if (str[i] == ')') {
                    i = j;
                }
            }
        }
        if (inter.size() == 0) {
            inter.push_back(0);
        }
        while (dec.size() < 100) {
            dec.push_back(0);
        }
    }
    
    vector<int> dif(vector<int> l, vector<int> s) {
        vector<int> res;
        int c = 0;
        int j = l.size() - 1;
        for (int i = s.size() - 1; i >= 0; --i, --j) {
            l[j] -= s[i] + c;
            if (l[j] < 0) {
                l[j] += 10;
                c = 1;
            } else {
                c = 0;
            }
            res.push_back(l[j]);
        }
        reverse(res.begin(), res.end());
        return res;
    }
    
    int toNum(vector<int>& num) {
        int res = 0;
        for (auto&n: num) {
            res *= 10;
            res += n;
        }
        return res;
    }
    
    bool equal(int n, vector<int> ld, vector<int> sd) {
        ld.insert(ld.begin(), n);
        sd.insert(sd.begin(), 0);
        vector<int> res = dif(ld, sd);
        for (int i = 0; i < res.size() - 1; ++i) {
            if (res[i] != 0) {
                return false;
            }
        }
        return res.back() < 2;
    }
    
    bool isRationalEqual(string S, string T) {
        vector<int> is, it, ds, dt;
        parse(S, is, ds);
        parse(T, it, dt);
        int si = toNum(is);
        int ti = toNum(it);
        if (abs(si - ti) > 1) {
            return false;
        } if (is > it) {
            return equal(si - ti, ds, dt);
        } else if (is < it) {
            return equal(ti - si, dt, ds);
        } else {
            for (int i = 0; i < ds.size(); ++i) {
                if (ds[i] < dt[i]) {
                    return equal(0, dt, ds);
                } else if (ds[i] > dt[i]) {
                    return equal(0, ds, dt);
                }
            }
            return true;
        }
    }
};
