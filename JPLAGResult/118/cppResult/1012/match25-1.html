class Solution {
public:

	string extend(string str) {
		string ret="";
        string rep="";
		int len=str.length();
        int pos, pos2;
		for(int i=0; i<len; i++) if (str[i]=='(') pos=i;
		for(int i=0; i<len; i++) if (str[i]==')') pos2=i;
		for(int i=pos+1; i<=pos2-1; i++) rep+=str[i];
		for(int i=0; i<len; i++) {
			if (str[i]=='('||str[i]==')') continue;
			else ret+=str[i];
		}
		if (rep=="") rep="0";
		for(int i=0; i<1000; i++) ret+=rep;
		return ret;
	}

	string dot(string str) {

		int len=str.length();
		for(int i=0; i<len; i++) {
			if (str[i]=='.') return str;
		}
		return str+".";
	}

	int Pos(string& str, char ch) {
		int len=str.length();
		for(int i=0; i<len; i++) {
			if (str[i]==ch) return i;
		}
		return -1;
	}

	int norm(string& str) {
		string ret="";
		int pos=Pos(str, '.');
		int num=0;
        int len=str.length();
		for(int i=0; i<pos; i++) {
			num=num*10+str[i]-'0';
		}
		for(int i=pos+1; i<len; i++) {
			ret+=str[i];
			if (ret.length()==500) break;
		}
		str=ret;
		return num;
	}

	bool all(string& str, char ch) {
		int len=str.length();
		for(int i=0; i<len; i++) {
			if (str[i]!=ch) return false;
		}
		return true;
	}
    
    bool all(string& str, char ch, int st) {
		int len=str.length();
		for(int i=st; i<len; i++) {
			if (str[i]!=ch) return false;
		}
		return true;
	}
    
    bool check(string s1, string s2) {
        int pos1=-1, pos2=-1;
        for(int i=0; i<500; i++) if (all(s1, '9', i)) {pos1=i;break;}
        for(int i=0; i<500; i++) if (all(s2, '0', i)) {pos2=i;break;}
        if (pos1==-1||pos2==-1) return false;
        if (pos1!=pos2) return false;
        if (pos1&&pos2&&s2[pos2-1]-s1[pos1-1]==1) return true;
    }

    bool isRationalEqual(string S, string T) {
        S=dot(S);
        T=dot(T); 
        cout<<S<<' '<<T<<endl;
        S=extend(S);
        T=extend(T);
        int a=norm(S);
        int b=norm(T);
        if (a==b&&S==T) return true;
        if (a-b==1&&all(S, '0')&&all(T, '9')) return true;
        if (b-a==1&&all(T, '0')&&all(S, '9')) return true;
        if (a==b) {
            if (check(S, T)) return true;
            if (check(T, S)) return true;
        }
        return false;
    }
};
