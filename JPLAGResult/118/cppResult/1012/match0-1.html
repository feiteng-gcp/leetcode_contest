class num{
public:
    string ip;
    string nrp;
    string rp;
    num(string S){
        ip = "", nrp = "", rp = "";
        int i = 0;
        int start = 0;
        while(i<S.size() && S[i] != '.')
            ++i;
        ip = S.substr(start, i-start);
        start = ++i;
        while(i<S.size() && S[i] != '(')
            ++i;
        if(i>start)
            nrp = S.substr(start, i-start);
        
        start = ++i;
       
        while(i<S.size() && S[i] != ')')
            ++i;
        if(i>start)
            rp = S.substr(start, i-start);
        
    }
    void print(){
        cout<<ip<<"."<<nrp<<"("<<rp<<")"<<endl;
    }
    void modify(){
        if(rp.size() == 4)
            if(rp.substr(0,2) == rp.substr(2,2))
                rp = rp.substr(0,2);
        if(rp.size() == 2)
            if(rp[0] == rp[1])
                rp = rp.substr(0,1);
        if(rp.size() == 3)
            if(rp[0] == rp[1] && rp[1] == rp[2])
                rp = rp.substr(0,1);
        while(nrp.size() && nrp.back() == rp.back()){
            rp.pop_back();
            rp.insert(rp.begin(), nrp.back());
            nrp.pop_back();
        }
        if(rp == "0")
            rp = "";
        if(nrp == "0" && rp =="")
            nrp = "";
        if(rp == "9"){
            rp = "";
            if(nrp == "")
                ip = to_string(stoi(ip)+1);
            else{
                int count_zero = 0;
                for(int i=0; i<nrp.size() && nrp[i] == '0'; ++i, ++count_zero);
                nrp = to_string(stoi(nrp)+1);
                for(;count_zero>0;--count_zero)
                    nrp.insert(nrp.begin(), '0');
            }
                
        }
    }
};
class Solution {
public:
    bool isRationalEqual(string S, string T) {
        num a(S), b(T);
        // a.print(),b.print();
        a.modify(),b.modify();
        // a.print(),b.print();
        
        return a.ip == b.ip && a.rp == b.rp && a.nrp == b.nrp;
    }
};
