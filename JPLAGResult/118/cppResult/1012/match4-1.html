#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 1000005

class Solution {
public:
  pair<string, pair<string, string>> parse(string &x) {

    int step = 0;
    string v, nonrp, rp;
    REP(i,0,x.size()) {
      if (x[i] == '.') {
        step = 1;
      } else if (x[i] == '(') {
        step = 2;
      } else if (x[i] == ')') {}
      else {
        if (step == 0) v += x[i];
        else if (step == 1) nonrp += x[i];
        else rp += x[i];
      }
    }
    return make_pair(v, make_pair(nonrp, rp));
  }
  bool isRationalEqual(string S, string T) {
    pair<string, PSS> s = parse(S), t = parse(T);
    PSS a = s.second, b = t.second;
    if (a.second.empty() && b.second.empty()) {
      if (s.first != t.first) return false;
      string p = a.first, q = b.first;
      while(p.back() == '0') p.pop_back();
      while(q.back() == '0') q.pop_back();
      return p == q;
    }
    if (abs((int)s.first.size() - (int)t.first.size()) > 1) return false;
    if (s.first.size() != t.first.size()) {
      if (s.first.size() < t.first.size()) s.first = "0" + s.first;
      else t.first = "0" + t.first;
    }
    string x = s.first + a.first, y = t.first + b.first;
    REP(i,0,100) x += a.second;
    REP(j,0,100) y += b.second;
    while (x.size() < 100) x += '0';
    while (y.size() < 100) y += '0';
    int len = min(x.size(), y.size());
    if (x.substr(0, len) != y.substr(0, len)) {
      string xb = x, yb = y;
      REP(t,0,2) {
        x = xb; y = yb;
        if (t == 1) swap(x, y);
        for (int i = x.size()-1; i>=0; i--) {
          if (x[i] == '9') x[i] = '0';
          else {
            x[i]++;
            break;
          }
        }
        if (x.substr(0, len) == y.substr(0, len)) return true;
      }
      return false;
    }
    return true;
  }
};
