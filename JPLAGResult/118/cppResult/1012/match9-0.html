class Solution {
public:
    bool isRationalEqual(string S, string T) {
        vector<string> ps = splitParts(S), pt = splitParts(T);
        auto rs = rebuild(ps, 20), rt = rebuild(pt, 20);
        if(rs.first != rt.first) return false;
        cout<<rs.second<<endl;
        cout<<rt.second<<endl;
        for(int i = 0; i < min(rs.second.length(), rt.second.length()); i++){
            if(rs.second[i] != rt.second[i]) return false;
        }
        return true;
    }
    
    vector<string> splitParts(string s){
        vector<string> res;
        int dot = s.find_first_of("."), n = s.length();
        if(dot == -1){
            res.push_back(s);
            return res;
        }
        res.push_back(s.substr(0, dot));
        int bracket = s.find_first_of("(");
        if(bracket == -1){
            res.push_back(s.substr(dot + 1));
            return res;
        }
        else{
            res.push_back(s.substr(dot + 1, bracket - dot - 1));
            res.push_back(s.substr(bracket + 1, n - bracket - 2));
        }
        return res;
    }
    
    pair<int, string> rebuild(vector<string>& parts, int len){
        string res = "";
        int integer = stoi(parts[0]);
        if(parts.size() == 1){
            for(int i = 0; i < len; i++){
                res += "0";
            }
        }
        else if(parts.size() == 2){
            res += parts[1];
            for(int i = parts[1].size(); i < len; i++){
                res += "0";
            }
        }
        else{
            if(allnine(parts[2])){
                if(allnine(parts[1])){
                    ++integer;
                    for(int i = 0; i < len; i++){
                        res += "0";
                    }
                }
                else{
                    int nonre = stoi(parts[1]), n0 = 0;
                    for(char c : parts[1]){
                        if(c != '0') break;
                        else n0++;
                    }
                    ++nonre;
                    for(int i = 0; i < n0; i++){
                        res += "0";
                    }
                    // prefix 0
                    res += to_string(nonre);
                    for(int i = parts[1].size(); i < len; i++){
                        res += "0";
                    }
                }
            }
            else{
                res += parts[1];
                for(int i = parts[1].size(); i < len; i += parts[2].size()){
                    res += parts[2];
                }
            }
        }
        return {integer, res};
    }
    
    bool allnine(string s){
        bool all9 = true;
        for(char c : s){
            if(c != '9'){
                all9 = false;
                break;
            }
        }
        return all9;
    }
};
