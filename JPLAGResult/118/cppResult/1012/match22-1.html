class Solution {
public:
    bool isRationalEqual(string S, string T) {
        pair<long long, long long> p1 = extract(S);
        pair<long long, long long> p2 = extract(T);
        
        //cout << "S: {" << p1.first << ", " << p1.second << "}\n";
        //cout << "T: {" << p2.first << ", " << p2.second << "}\n";
        
        return pariCmp(p1, p2);        
    }
private:
    pair<long long, long long> extract(string s) {
        string a, b, c;
        int p1 = s.find(".");
        int p2 = s.find("(");
        int p3 = s.find(")");
        
        if (p1 == string::npos) {
            a = s;
            b = "";
            c = "";
        }
        else if (p2 == string::npos){
            a = s.substr(0, p1);
            b = s.substr(p1 + 1);
            c = "";
        }
        else {
            a = s.substr(0, p1);
            b = s.substr(p1 + 1, max(0, p2 - p1 - 1));
            c = s.substr(p2 + 1, max(0, p3 - p2 - 1));
        }
        
        if (c.length() == 0 || stoi(c) == 0) {
            if (b.length() == 0 || stoi(b) == 0) {
                return {1LL * stoi(a), 1LL};
            }
            else {
                return {1LL * stoi(b) + pow10(b.length()) * stoi(a), pow10(b.length())};
            }
        }
        else {
            if (b.length() == 0) {
                return {(pow10(c.length()) - 1) * stoi(a) + stoi(c), pow10(c.length()) - 1};
            }
            else {
                long long xx = pow10(b.length());
                long long yy = pow10(c.length()) - 1;
                int ia = stoi(a);
                int ib = stoi(b);
                int ic = stoi(c);
                return {1LL * ia * xx * yy + 1LL * ib * yy + 1LL * ic, xx * yy};
            }
        }
    }
    
    long long pow10(int n) {
        long long res = 1;
        for (int i = 0; i < n; ++i) {
            res *= 10;
        }
        return res;
    }
    

    int gcd(long long a, long long b) {
        return b ? gcd(b, a % b) : a;
    }
    
    bool pariCmp(pair<long long, long long> p1, pair<long long, long long> p2) {
        long long g1 = gcd(p1.first, p1.second);
        long long g2 = gcd(p2.first, p2.second);
        
        p1.first /= g1;
        p1.second /= g1;
        p2.first /= g2;
        p2.second /= g2;
        
        return p1.first == p2.first && p1.second == p2.second;
    }
};
