class Solution 
{
    struct Num
    {
        string intp;
        string nr;
        string re;
        Num(): intp(""), nr(""), re(""){}
    };
    void f(string s, Num &n)
    {
        string temp = "";
        int state = 0;
        for(char c : s)
        {
            if(state == 0)
            {
                if(c != '.')
                {
                    temp += c;
                }
                else
                {
                    n.intp = temp;

                    temp = "";
                    state = 1;
                }
            }
            else if(state == 1)
            {
                if(c != '(')
                {
                    temp += c;
                }
                else
                {
                    n.nr = temp;
                    temp = "";
                    state = 2;
                }
            }
            else if(state == 2)
            {
                if(c != ')')
                {
                    temp += c;
                }
                else
                {
                    n.re = temp;
                    temp = "";
                    state = 3;
                    break;
                }
            }
        }
        if(state == 0)
        {
            n.intp = temp;
            n.nr = n.re = "";
        }
        else if(state == 1)
        {
            n.nr = temp;
            n.re = "";
        }
        else if(state == 2)
        {
            n.re = temp;
        }
        while(!n.nr.empty() && !n.re.empty())
        {
            if(n.nr[n.nr.length()-1] == n.re[n.re.length()-1])
            {
                n.re = n.nr[n.nr.length()-1] + n.re;
                n.re.pop_back();
                n.nr.pop_back();
            }
            else
            {
                break;
            }
        }
        bool all0 = true;
        for(char c : n.nr)
        {
            if(c != '0')
            {
                all0 = false;
                break;
            }
        }
        if(all0)
            n.nr = "";
        all0 = true;
        for(char c : n.re)
        {
            if(c != '0')
            {
                all0 = false;
                break;
            }
        }
        if(all0)
            n.re = "";
        if(n.re.empty())
        {
            while(!n.nr.empty() && n.nr[n.nr.length()-1] == '0')
                n.nr.pop_back();
        }
        return;
    }
    bool re9(Num n)
    {
        if(n.re.empty())
            return false;
        for(char c : n.re)
            if(c != '9')
                return false;
        return true;
    }
    void aaa(Num &a, Num &b)
    {
        int num = 0;
        if(a.re.empty() || b.re.empty())
            return;
        while(b.re.length() != a.re.length())
        {
            b.re += b.re[num++];
        }
        return;
    }
    bool bbb(Num &a, Num &b)
    {
        int n1, n2;
        if(a.nr == "")
        {
            if(b.nr != "")
                return false;
            n1 = stoi(a.intp);
            n2 = stoi(b.intp);
            return (n2-n1 == 1);
        }
        else
        {
            n1 = stoi(a.nr);
            if(b.nr.empty())
                n2 = 0;
            else
                n2 = stoi(b.nr);
            n1++;
            while(n1 % 10 == 0 && n1 > 0)
                n1 /= 10;
            if(n1 != n2)
                return false;
            if(n1 != 0)
                return a.intp == b.intp;
            n1 = stoi(a.intp);
            n2 = stoi(b.intp);
            return (n2-n1 == 1);
        }
    }
public:
    bool isRationalEqual(string S, string T)
    {
        Num a, b;
        f(S, a);
        f(T, b);
        if(a.re.length() > b.re.length())
            aaa(a, b);
        else
            aaa(b, a);
        if(a.intp == b.intp && a.nr == b.nr && a.re == b.re)
            return true;
        if(re9(a) && b.re == "")
            return bbb(a, b);
        else if(re9(b) && a.re == "")
            return bbb(b, a);
        else
            return false;
    }
};
