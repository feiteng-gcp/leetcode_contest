class Solution {
public:
    int EXPAND = 100;
    bool isRationalEqual(string S, string T) {
        vector<string> s = Trans(S);
        vector<string> t = Trans(T);
       // cout << ": " << s[0] << "-" << s[1] << endl;
       // cout << ": " << t[0] << "-" << t[1] << endl;
        return s == t;
    }
    
    vector<string> Trans(string s) {
        vector<string> ret(2);
        int dot = -1, left = -1, right = -1;
        for (int i = 0; i < s.size(); i++)
            if (s[i] == '.') {
                dot = i;
            } else if (s[i] == '(') {
                left = i;
            } else if (s[i] == ')') {
                right = i;            
            }        
        ret[0] = s.substr(0, dot == -1 ? s.size() : dot);
        if (dot == -1 || dot + 1 == s.size()) return ret;
        string non = s.substr(dot + 1, (left == -1 ? s.size() : left - dot - 1));
        cout << non << endl;
        string rec = "";
        if (left != -1) {
            rec = s.substr(left + 1, right - left - 1);            
            if (Only(rec, '9')) {                
                if (Only(non, '9')) {
                    ret[0] = to_string(stoi(ret[0]) + 1);
                    return ret;
                }
                for (int i = non.size() - 1,advanced = 1; i >= 0 && advanced > 0; i--) {
                    int val = non[i] - '0' + advanced;
                    advanced = val / 10;
                    non[i] = '0' + val % 10;
                }                
                rec = "";
            }
            //cout << "11rec = " << rec << endl;
            if (Only(rec, '0')) rec = "";
            //cout << "1221rec = " << rec << endl;
        }      
        if (rec == "" && Only(non, '0')) return ret;
        //cout << "rec = " << rec << endl;
        ret[1] = non;
        if (rec != "") {
            while (ret[1].size() < 100) ret[1].append(rec);
            ret[1].resize(100);
        }
        if (Only(ret[1], '0')) ret[1] = "";
        return ret;
    }
    
    
    bool Only(string &s, char ch) {
        for (auto &i : s) if (i != ch) return false;
        return true;
    }
};
