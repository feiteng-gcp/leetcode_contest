class Solution {
public:
    bool isRationalEqual(string S, string T) {
        auto ss = numberToFraction(S);
        auto tt = numberToFraction(T);
        return ss == tt;
    }
    
    pair<int, int> numberToFraction(const string & s) {
        // a.b(c)
        string a, b, c;
        splitNumber(s, a, b, c);
        //cout << a << " " << b << " " << c << endl;
        
        
        pair<int, int> r = {std::stoi(a), 1};
        if (!b.empty()) {
            r = addFraction(r, {std::stoi(b), power(10, b.size())});
        }
        if (!c.empty()) {
            pair<int, int> cc = {std::stoi(c), power(10, c.size()) - 1};
            cc.second *= power(10, b.size());
            r = addFraction(r, cc);
        }
        
        //cout << r.first << "/" << r.second << endl;
        
        return r;
    }
    
    void splitNumber(const string & s, string & a, string & b, string & c) {
        int ip = s.find_first_of('.');
        if (ip == string::npos) {
            a = s;
            return;
        }
        
        a = s.substr(0, ip);
        
        int ib = s.find_first_of('(');
        if (ib == string::npos) {
            b = s.substr(ip + 1);
            return;
        }
        
        b = s.substr(ip + 1, ib - ip - 1);
        
        c = s.substr(ib + 1, s.size() - ib - 2);
    }
    
    pair<int, int> addFraction(const pair<int, int> & a, const pair<int, int> & b) {
        pair<int64_t, int64_t> r = {(int64_t)a.first * b.second + b.first * a.second, (int64_t)a.second * b.second};
        int64_t g = gcd(r.first, r.second);
        return {r.first / g, r.second / g};
    }
    
    int64_t gcd(int64_t x, int64_t y) {
        while (y != 0) {
            int64_t tmp = x % y;

            x = y;
            y = tmp;
        }
        return x;
    }
    
    int power(int base, int n) {
        int x = 1;
        for (int i = 0; i < n; ++i) {
            x *= base;
        }
        return x;
    }
};
