class Solution {
public:
    vector<int> toFrac(string S) {
        int index = 0;
        int state = 0;
        int i = 0, n = 0, r = 0;
        int ic = 0, nc = 1, rc = 0;
        while (index < S.size()) {
            if (S[index] == '.') state = 1;
            else if (S[index] == '(') state = 2;
            else if (S[index] == ')') state = 3;
            else {
                if (state == 0)
                    i = i * 10 + S[index] - '0';

                else if (state == 1) {        
                    nc *= 10;
                    n = n * 10 + S[index] - '0';
                }
                else {
                    r = r * 10 + S[index] - '0';
                    rc = rc * 10 + 9;
                }
            }
            index++;
        }
        rc *= nc;
        if (r == 0 && n == 0)
            return vector<int> {i, 1};
        if (r == 0)
            return vector<int> {nc * i + n, nc};
        if (n == 0)
            return vector<int> {rc * i + r, rc};
        return vector<int> {i * nc * rc + n * rc + r * nc, nc * rc};
    }
    
    bool isRationalEqual(string S, string T) {
        vector<int> s = toFrac(S);
        vector<int> t = toFrac(T);
        cout << s[0] << " " << s[1] << " ";
        cout << t[0] << " " << t[1];
        if (s[0] * t[1] == s[1] * t[0])
            return true;
        else
            return false;
    }
};
