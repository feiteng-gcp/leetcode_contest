#define EACH(i,c) for(__typeof((c).begin()) i=(c).begin(),i##_end=(c).end();i!=i##_end;++i)

template<class T> inline void amin(T &x, const T &y) { if (y<x) x=y; }
template<class T> inline void amax(T &x, const T &y) { if (x<y) x=y; }
template<class Iter> void rprintf(const char *fmt, Iter begin, Iter end) {
    for (bool sp=0; begin!=end; ++begin) { if (sp) putchar(' '); else sp = true; printf(fmt, *begin); }
    putchar('\n');
}

class Solution {
public:
    pair<LL, LL> normal(pair<LL, LL> a) {
	LL g = __gcd(a.first, a.second);
	a.first /= g;
	a.second /= g;
	return a;
    }

    LL number(string s) {
	LL ret = 0;
	REP (i, s.size()) {
	    ret = ret * 10 + s[i] - '0';
	}
	return ret;
    }
    pair<LL, LL> parse(string s) {
	int dot = s.find('.');
	if (dot == s.npos) {
	    // dddd
	    return make_pair(number(s), 1LL);
	}

	LL a = number(s.substr(0, dot));
	int left = s.find('(');
	if (left == s.npos) {
	    // dddd.dddd
	    LL b = number(s.substr(dot+1));
	    LL de = 1;
	    REP (i, (int)s.size() - dot - 1) {
		a *= 10;
		de *= 10;
	    }
	    return normal(make_pair(a + b, de));
	}

	// dddd.dddd(dddd)
	int len1 = left - dot - 1;
	int len2 = s.size() - left - 2;
	LL x = number(s.substr(dot+1, len1));
	LL y = number(s.substr(left+1, len2));
	LL de = 1;

	REP (i, len1) {
	    a *= 10;
	    de *= 10;
	}
	a += x;

	LL a1 = a;
	LL de1 = de;
	REP (i, len2) {
	    a1 *= 10;
	    de1 *= 10;
	}
	a1 += y;

	return normal(make_pair(a1 - a, de1 - de));
    }

    bool isRationalEqual(string S, string T) {
	pair<LL, LL> s = parse(S);
	pair<LL, LL> t = parse(T);
        
//	eprintf("%lld / %lld : %s\n", s.first, s.second, S.c_str());
//	eprintf("%lld / %lld : %s\n", t.first, t.second, T.c_str());
	return s == t;
    }
};
