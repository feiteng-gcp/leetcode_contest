class Solution {
public:
bool isRationalEqual(string S, string T) {
	int repeattime = 16;
	double snum, tnum;
	string Snew, Tnew;
	int left = -1, right = -1;
	for (int i = 0; i < S.size(); ++i)
	{
		if (S[i] == '(')
		{
			left = i;
		}
		else if (S[i] == ')')
		{
			right = i;
		}
		else
		{
			Snew += S[i];
		}
	}
	if (left != -1)
	{
		string repeat = S.substr(left + 1, right - left - 1);
		for (int i = 0; i < repeattime; ++i)
		{
			Snew += repeat;
		}
	}
	left = -1, right = -1;
	for (int i = 0; i < T.size(); ++i)
	{
		if (T[i] == '(')
		{
			left = i;
		}
		else if (T[i] == ')')
		{
			right = i;
		}
		else
		{
			Tnew += T[i];
		}
	}
	if (left != -1)
	{
		string repeat = T.substr(left + 1, right - left - 1);
		for (int i = 0; i < repeattime; ++i)
		{
			Tnew += repeat;
		}
	}
	snum = stod(Snew);
	tnum = stod(Tnew);
	if (abs(snum - tnum) < 2.2204460492503131e-016)
	{
		return true;
	}
	return false;
}
};
