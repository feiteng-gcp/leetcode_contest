class Solution {
    typedef long long int64;
    typedef pair<int64, int64> pll;
public:
    int64 gcd(int64 x, int64 y) {
        while (y != 0) {
            int64 t = x % y;
            x = y;
            y = t;
        }
        return x;
    }
    pll simple(pll x) {
        int64 g = gcd(x.first, x.second);
        return make_pair(x.first / g, x.second / g);
    }
    pll add(pll x, pll y) {
        return simple(make_pair(x.first * y.second + x.second * y.first, x.second * y.second));
    }
    pll trans(string s) {
        int sp = -1, sq = -1;

        pll sa = make_pair(0ll, 1ll);
        pll sb = make_pair(0ll, 1ll);
        pll sc = make_pair(0ll, 1ll);
        for (int i = 0; i < s.length(); i ++) {
            if (s[i] == '.') {
                sp = i;
                break;
            } else {
                sa.first = sa.first * 10 + (s[i] - '0');
            }
        }
        printf("sa = %lld / %lld\n", sa.first, sa.second);
        if (sp == -1) {
            return sa;
        }
        for (int i = sp + 1; i < s.length(); i ++) {
            if (s[i] == '(') {
                sq = i;
                break;
            } else {
                sb.first = sb.first * 10 + (s[i] - '0');
                sb.second *= 10;
            }
        }
        printf("sb = %lld / %lld\n", sb.first, sb.second);
        if (sq == -1) {
            sb = simple(sb);
            return add(sa, sb);
        }
        for (int i = sq + 1; i < s.length(); i ++) {
            if (s[i] == ')') {
                break;
            } else {
                sc.first = sc.first * 10 + (s[i] - '0');
                sc.second *= 10;
            }
        }
        sc.second = (sc.second - 1) * sb.second;
        printf("sc = %lld / %lld\n", sc.first, sc.second);
        sb = simple(sb);
        sc = simple(sc);
        printf("sc = %lld / %lld\n", sc.first, sc.second);
        return add(sa, add(sb, sc));
        
    }
    bool isRationalEqual(string S, string T) {
        pll s = trans(S);
        printf("S = %lld / %lld\n", s.first, s.second);
        pll t = trans(T);
        printf("T = %lld / %lld\n", t.first, t.second);
        return s == t;
    }
};
