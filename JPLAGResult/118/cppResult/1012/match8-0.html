class Solution {
public:
    bool isRationalEqual(string S, string T) {
        string newS = trans(S);
        string newT = trans(T);
        double s = stod(newS);
        cout << s << endl;
        double t = stod(newT);
        cout << t << endl;
        if (abs(s - t) < 0.00000001) return true;
        return false;
    }
    
    string trans(string S) {
        int i = 0;
        string res;
        while (i < S.size()) {
            while (S[i] != '(' && i < S.size()) {
                res.push_back(S[i]);
                i++;
            }
            if (i < S.size()) {
                i++;
                int start = i;
                while (S[i] != ')') {
                    i++;
                }
                string tmp = S.substr(start, i - start);
                for (int i = 0; i < 10; i++) {
                    res += tmp;
                }
            }
        }
        return res;
    }
};
