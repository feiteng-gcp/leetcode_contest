class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        B = sorted(A)
        if A == B:
            return []
        res = []
        for i in range(len(A)-1, -1, -1):
            if A[i] == B[i]:
                continue
            for j in range(i):
                if A[j] == B[i]:
                    if j != 0:
                        res.append(j+1)
                        A = A[:j+1][::-1] + A[j+1:]
                    res.append(i+1)
                    A = A[:i+1][::-1] + A[i+1:]
        return res
