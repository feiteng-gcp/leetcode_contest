class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        arranges = []
        for i in range(len(A)-1,0,-1):
            idx = A.index(i+1)
            if idx!=i:
                arranges
.append(idx+1)
                arranges.append(i+1)
                A[:idx+1] = A[:idx+1][::-1]
                A[:i+1] = A[:i+1][::-1]
                print(A)
        return arranges
