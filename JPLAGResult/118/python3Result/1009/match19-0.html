class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        n=len(A)
        k=[]
        while n>1:
            ind=A.index(max(A[0:n]))
            if ind==n-1:
                n-=1
                A.pop(-1)
                continue
            if ind==0:
                k.append(n)
                A.reverse()
                A.pop(-1)
                n-=1
                continue
            k.append(ind+1)
            k.append(n)
            b=A[0:(ind+1)]
            b.reverse()
            A=b+A[(ind+1):]
            A.reverse()
            A.pop(-1)
            n-=1
        return k
