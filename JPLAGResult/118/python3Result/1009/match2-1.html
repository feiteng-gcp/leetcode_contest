class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        ans = []
        
        N = len(A)
        for tgt in range(1,N+1):
            idx = A.index(tgt)
            dest_idx = N-tgt
            ans
.append(idx+1)
            #print(idx, dest_idx)
            #print(A[:idx+1][::-1], A[idx+1:])
            A = A[:idx+1][::-1]+A[idx+1:]
            #print(A)
            #print(ans)
            ans.append(dest_idx+1)
            A = A[:dest_idx+1][::-1]+A[dest_idx+1:]
            #print(A)
            #print(ans)
        ans.append(N)
        return ans
