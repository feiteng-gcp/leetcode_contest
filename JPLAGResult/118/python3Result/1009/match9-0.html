class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """

        ans = []
        for n in reversed(range(1, len(A) + 1)):
            i = A.index(n)
            if i + 1 == n:
                continue
            ans.append(i + 1)
            ans.append(n)
            A = A[:i + 1][::-1] + A[i + 1:]
            A = A[:n][::-1] + A[n:]
            
        return ans
