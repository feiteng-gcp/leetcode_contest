class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        need = 0
        for i in range(1, len(A)):
            if A[i] < A[i-1]:
                need = 1
                break
        if need == 0:
            return []
        result = []
        for i in range(0, len(A) - 1):
            length = len(A) - i
            index = 0
            value = A[0]
            #print(value)
            #print()
            for j in range(0, length):
                #print(A[i])
                if A[j]>value:
                    #print("bigger")
                    value = A[j]
                    index = j
            result
.append(index+1)
            result.append(length)
            #print(result)
            A[0:index+1] = A[0:index+1][::-1]
            A[0:length] = A[0:length][::-1]
        return result
