class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """

        n = len(A)

        if n <= 1:
            return []

        res = []
        cur = n
        while cur > 1:
            i = cur-1
            if A[i] == cur:
                cur -= 1
                continue

            while A[i] != cur:
                i -= 1
            if i != 0:
                res.append(i+1)
                self.flip(A, i+1)
            res.append(cur)
            self.flip(A, cur)
            cur -= 1

        return res

    def flip(self, A, k):
        i = 0
        j = k-1

        while i < j:
            A[i], A[j] = A[j], A[i]
            i += 1
            j -= 1
