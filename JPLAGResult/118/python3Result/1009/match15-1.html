class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        ret=[]
        
for i in range(len(A)):
            #print(i, A)
            curr=i
            for j in range(i, len(A), 1):
                if A[j]<A[curr]:
                    curr=j
            if curr!=i:
                ret.append(curr+1)
                X=A[:curr+1]
                X.reverse()
                Y=A[curr+1:]
                A=X+Y
                if i>0:
                    ret.append(curr-i+1)
                    X=A[:curr-i+1]
                    X.reverse()
                    Y=A[curr-i+1:]
                    A=X+Y
                    ret.append(curr+1)
                    X=A[:curr+1]
                    X.reverse()
                    Y=A[curr+1:]
                    A=X+Y
        return ret
