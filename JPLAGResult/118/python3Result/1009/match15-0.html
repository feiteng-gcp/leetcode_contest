class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        res = []
        A.reverse()
        
        for i in range(len(A)):
            idx = i
            for j in range(i+1, len(A)):
                if A[j] > A[idx]:
                    idx = j
            if idx == i:
                continue
            res.append(len(A) - idx)
            res.append(len(A) - i)
            A[idx:] = A[idx:][::-1]
            A[i:] = A[i:][::-1]
            
        return res
