class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        l = len(A)
        ans = []
        if l<=1:
          return ans
        b = sorted(A)
        j = l-1
        while j>=0:
          if A[j] == b[j]:
            j -= 1
          else:
            ind = A.index(b
[j])
            if ind>0:
              ans.append(ind+1)
              A = A[:ind+1][::-1]+A[ind+1:]
            ans.append(j+1)
            A = A[:j+1][::-1]+A[j+1:]
            #print(ind,j,A)
        return ans
