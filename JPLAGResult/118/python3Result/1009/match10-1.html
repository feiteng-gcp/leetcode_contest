class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        res = []
        for i in range(len(A),1,-1):
            #print(A)
            t = A.index(i)
            if t == i-1:
                A.pop()
                continue
            if t == 0:
                A  = A[::-1]
                A.pop()
                res.append(i)
                continue
            res.append(t+1)
            res.append(i)
            k = A[t+1:i]
            k.reverse()
            D = k+A[:t]
            A= D
        return res
