class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        # 每次把最大值放在首位，然后翻转放到最后，之后去处理前n-1个数
        n = len(A)
        if n == 1:
            return []
        ans = []
        for cur in range(n,1,-1):
            position = A.index(cur)
            if position == cur-1:
                continue
            else:
                ans
.append(position + 1)
                ans.append(cur)
                A = A[:position+1][::-1] + A[position+1:]
                A = A[:cur][::-1] + A[cur:]
        return ans
