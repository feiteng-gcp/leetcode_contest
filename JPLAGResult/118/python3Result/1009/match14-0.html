class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        def find(A, i, target):
            for j in range(i+1):
                if A[j] == target:
                    return j
        
        def reverse(A, i):
            start, end = 0, i
            while start < end:
                A[start], A[end] = A[end], A[start]
                start+=1
                end-=1
        
        n = len(A)
        res = []
        for i in range(n-1, -1, -1):
            if A[i] != i+1:
                j = find(A, i, i+1)
                reverse(A, j)
                reverse(A, i)
                res.append(j+1)
                res.append(i+1)
                
        return res
