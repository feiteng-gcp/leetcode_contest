class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        res 
= []
        l = len(A)
        for i in range(l):
            mi = A.index(l-i)
            self.addRes(A, res, mi)
            self.addRes(A, res, l-i-1)
        return res
    
    def addRes(self, A, res, i):
