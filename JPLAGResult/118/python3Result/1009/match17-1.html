class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        ls = []

        def findMax(size):
            maxidx = 0
            for i in range(size):
                if A[i] > A[maxidx]:
                    maxidx = i
            return maxidx

        
def flip(A, i):
            start = 0
            while start < i:
                temp = A[start]
                A[start] = A[i]
                A[i] = temp
                start += 1
                i -= 1

        curr_size = len(A)

        while curr_size > 1:
            maxidx = findMax(curr_size)
            if maxidx != curr_size - 1:
                flip(A, maxidx)
                ls.append(maxidx + 1)
                flip(A, curr_size-1)
                ls.append(curr_size)
            curr_size -= 1
            
        return ls
