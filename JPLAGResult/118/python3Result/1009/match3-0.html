class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        res = []
        n = len(A)
        while n > 0:
            i = A.index(n)
            res.append(i+1)
            A = A[:i+1][::-1] + A[i+1:]
            res.append(n)
            A = A[:n][::-1] + A[n:]
            n -= 1
        return res
