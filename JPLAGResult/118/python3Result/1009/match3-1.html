class Solution:
    def pancakeSort(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        if not A:
            return []
        
        sortA = sorted(A)
        
        ret = []
        
        j = len(A) - 1
        
        while j >= 0:
            if A[j] != sortA[j]:
                i = A[:j + 1].index(sortA[j])
                ret
.append(i + 1)
                A = A[:i+1][::-1] + A[i+1:]
                ret.append(j + 1)
                A = A[:j+1][::-1] + A[j+1:]
            j -= 1
            
        return ret
