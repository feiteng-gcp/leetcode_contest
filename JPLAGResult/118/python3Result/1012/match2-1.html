class Solution:
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        n = 20
        ns 
= self.process(S, 20)
        nt = self.process(T, 20)
        if abs(nt-ns) <= 1:
            return True
        else:
            return False
