class Solution:
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        def lst(S):
            nrp = "0"
            rp = "0"

            if "." in S:
                a = S.split(".")
                ip = a[0]
                if "(" in a[1]:
                    b = a[1].split("(")
                    nrp = b[0]
                    rp = b[1][:-1]
                else:
                    nrp = a[1]

            else:
                ip = S

            res = ip + "." + nrp + rp + rp + rp + rp + rp + rp + rp + rp + rp + rp + rp + rp
            if rp == "9" or rp == "99" or rp == "999" or rp == "9999":
                temp = res[:len(ip) + 5]
                t = (float(temp) + 0.0001) * 10000
                ip = str(int(t / 10000))
                nrp0 = str(int(t % 10000))
                l = len(nrp0)
                nrp = "0" * (4-l) + nrp0
                rp = "0"
                res = ip + "." + nrp + rp + rp + rp + rp + rp + rp + rp + rp + rp + rp + rp + rp

            return res

        sl = lst(S)
        tl = lst(T)

        for i in range(14):
            if sl[i] != tl[i]:
                return False
        return True
