class Solution:
    def change(self, S):
        first = S.split('(')[0]
        print(first)
        if S.find('(')!=-1:
            second = S.split('(')[-1]
            second = second.split(')')[0]
        else:
            second = '0'
        print(second)
        res = first
        if res.find('.')==-1:
            res = res+'.'
        while len(res)<20:
            res = res + second
        print(res)
        res =float(res)
        print(res)
        return res
        
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        tS = self.change(S)
        tT = self.change(T)
        if(abs(tS-tT)<1e-12):
            return True
        else:
            return False
