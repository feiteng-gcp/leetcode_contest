class Solution:
    def isRationalEqual(self, S, T):
        def separate(S):
            for a in range(len(S)):
                try:
                    int(S[a])
                except:
                    if S[a]!='.':
                        S[a]=""
                    continue
                S[a]=int(S[a])
            #print(S)
            a=0
            while a<len(S)-1:
                if S[a+1]:
                    S[a]=str(S[a])+str(S[a+1])
                    S.pop(a+1)
                    continue
                elif S[a+1]==0:
                    S[a]=str(S[a])+str(S[a+1])
                    S.pop(a+1)
                    continue
                else:
                    S[a]=str(S[a])
                S.pop(a+1)
                a=a+1
            return S
        
        def rational(n):
            if len(n)>=2:
                a=n[0]
                while len(a)<20:
                    a=a+n[1]
                if len(a)>20:
                    return a[:20]
                return a
            else:
                return n[0]
        
        try:
            float(S)
            float(T)
        except:
            num=[]
            S=list(S)
            T=list(T)
            #print(S)
            #print(T)
            S=separate(S)
            T=separate(T)
            #print(S)
            #print(T)
            S=rational(S)
            T=rational(T)
            print(S)
            print(T)
            if S==T:
                return True
            if round(float(S),10)==round(float(T),10):
                return True
            return False
        if float(S)==float(T):
            return True
        return False
        
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
