class Solution:
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        sint = int(S.split('.')[0])
        tint = int(T.split('.')[0])
        
        if len(S.split('.')) > 1:
            if S.split('.')[1].split('(')[0] == '':
                snrp = None
            else:
                snrp = S.split('.')[1].split('(')[0]
                
            if len(S.split('.')[1].split('(')) > 1:
                if S.split('.')[1].split('(')[1].split(')')[0] == '':
                    srp = '0'
                else:
                    srp = S.split('.')[1].split('(')[1].split(')')[0]
            else:
                srp = '0'
        else:
            snrp = None
            srp = '0'
        
        if len(T.split('.')) > 1:
            if T.split('.')[1].split('(')[0] == '':
                tnrp = None
            else:
                tnrp = T.split('.')[1].split('(')[0]
            
            if len(T.split('.')[1].split('(')) > 1:
                if T.split('.')[1].split('(')[1].split(')')[0] == '':
                    trp = '0'
                else:
                    trp = T.split('.')[1].split('(')[1].split(')')[0]
            else:
                trp = '0'
        else:
            tnrp = None
            trp = '0'
        
        Snew = ''
        if snrp is not None:
            Snew += snrp
        while(len(Snew) < 12):
            Snew += srp
        
        Tnew = ''
        if tnrp is not None:
            Tnew += tnrp
        while(len(Tnew) < 12):
            Tnew += trp
        
        Snew = int(Snew[:12]) + sint * 1000000000000
        Tnew = int(Tnew[:12]) + tint * 1000000000000
        
        if (abs(Snew - Tnew) <= 1):
            return True;
        return False;
