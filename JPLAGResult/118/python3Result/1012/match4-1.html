def int_part(s):
    x = s.find('.')
    if (x == -1):
        return int(s)
    return int(s[:x])

LEN = 30
def dig_part(s):
    x = s.find('.')
    if x == -1:
        return '0'*LEN
    
    y = s.find('(')
    if y == -1:
        base = s[x+1:]
        return base + '0' * (LEN - len(base))
    
    base = s[x+1:y]
    repeat = s[y+1:-1]
    
    return (base + repeat * 100)[:LEN]
    

class Solution:
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        x 
= int_part(S)
        y = int_part(T)
        z = dig_part(S)
        w = dig_part(T)
        a = x + float('0.'+z)
        b = y + float('0.'+w)
        if abs(a - b) < 1e-30:
            return True
        return False
