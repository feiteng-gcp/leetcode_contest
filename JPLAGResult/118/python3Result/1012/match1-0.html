class Solution:
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        def f(x):
            if '.' not in x or '(' not in x:
                return float(x)
            a = x.index('(')
            b = x.index(')')
            y = x[:a]+x[a+1:b]*10
            return float(y)
        a = f(S)
        b = f(T)
        if abs(a-b) < 10e-10:
            return True
        else:
            return False
