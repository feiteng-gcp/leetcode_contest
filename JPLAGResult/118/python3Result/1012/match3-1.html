class Solution:
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        def parse(n):
            if '.' in n:
                res=n.split('.')
                i=int(res
[0])
                if '(' in res[1]:
                    res=res[1].split('(')
                    n=res[0]
                    r=res[1][:-1]
                    return i, n, r
                else:
                    return i, res[1], "0"
            else:
                return int(n), "0", "0"
        def ext(n, r):
            res=n
            for i in range(len(n), 20, 1):
                res+=str(r[(i-len(n))%len(r)])
            return res
        si, sn, sr = parse(S)
        snr=ext(sn, sr)
        ti, tn, tr = parse(T)
        tnr=ext(tn, tr)
        #print (si, sn, sr, snr)
        #print (ti, tn, tr, tnr)
        if si==ti:
            if snr==tnr:
                return True
            pos=0
            for x in range(20):
                if snr[x]!=tnr[x]:
                    pos=x
                    break
            ds=int(snr[:pos+1])
            dt=int(tnr[:pos+1])
            #print(ds, dt)
            if ds==dt+1 and snr[pos+1:]=="0"*(20-pos-1) and tnr[pos+1:]=="9"*(20-pos-1):
                return True
            if ds+1==dt and snr[pos+1:]=="9"*(20-pos-1) and tnr[pos+1:]=="0"*(20-pos-1):
                return True
        if si==ti+1 and snr=="00000000000000000000" and tnr=="99999999999999999999":
            return True
        if si+1==ti and tnr=="00000000000000000000" and snr=="99999999999999999999":
            return True
        return False
