import re


class Solution:
    def isRationalEqual(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        S = list(re.findall(r'(\d*)(\.?)(\d*)(\(\d+\))?', S)[0])
        T = list(re.findall(r'(\d*)(\.?)(\d*)(\(\d+\))?', T)[0])
        
        S[0] = 0 if S[0] == '' else int(S[0])
        T[0] = 0 if T[0] == '' else int(T[0])
            
        def process(i, nonrep, rep):
            c = 0
            if rep:
                rep = rep[1:-1]
                if rep == rep[0] * len(rep):
                    rep = rep[0]
                if len(rep) == 1 and rep[0] == '9':
                    c = 1
                    rep = '0' * 10000
                else:
                    rep = rep * (10000 // len(rep) + 1)
            else:
                rep = '0' * 10000
                
            if c == 1:
                if not nonrep or nonrep == '9' * len(nonrep):
                    return (i + 1, '0' * 10000)
                else:
                    nonrep = str(int(nonrep) + 1).zfill(len(nonrep))
            full = nonrep + rep
            full = full[:10000]
            
            return (i, full)
        
        # print(process(S[0], S[2], S[3]))
        # print(process(T[0], T[2], T[3]))
        return process(S[0], S[2], S[3]) == process(T[0], T[2], T[3])
