# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def helper(self, root, voyage):
        if root.val != voyage[0]:
            return [-1]
        ans = []
        if root.left:
            left = root.left.val
            if left not in voyage:
                return [-1]
            pl = voyage.index(left)
        if root.right:
            right = root.right.val
            if right not in voyage:
                return [-1]
            pr = voyage.index(right)
        if root.left and root.right:
            if pl > pr:
                ans.append(root.val)
                ans += self.flipMatchVoyage(root.left, voyage[pl:])
                ans += self.flipMatchVoyage(root.right, voyage
[pr:pl])
            else:
                ans += self.flipMatchVoyage(root.left, voyage[pl:pr])
                ans += self.flipMatchVoyage(root.right, voyage[pr:])
        elif root.left and not root.right:
            ans += self.flipMatchVoyage(root.left, voyage[pl:])
        elif root.right and not root.left:
            ans += self.flipMatchVoyage(root.right, voyage[pr:])
        return ans
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        ans = self.helper(root, voyage)
        if -1 in ans:
            return [-1]
        else:
            return ans
