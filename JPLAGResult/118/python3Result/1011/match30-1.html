class Solution:

    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        if not root and not voyage:
            return []

        if not root or not voyage:
            return [-1]

        res = self.internal(root, voyage)
        if res != None:
            return res
        else:
            return [-1]

    
def internal(self, root, voyage):
        if not root and not voyage:
            return []

        if not root or not voyage:
            return None
        if root.val != voyage[0]:
            return None

        if not root.left and not root.right and len(voyage) == 1:
            return []

        if not root.left and not root.right and len(voyage) != 1:
            return None

        if not root.left:
            return self.internal(root.right, voyage[1:])
        if not root.right:
            return self.internal(root.left, voyage[1:])

        if voyage[1] == root.left.val:
            ri = -1
            for x in range(1, len(voyage)):
                if voyage[x] == root.right.val:
                    ri = x
                    break
            if ri == -1:
                return None
            fl = self.internal(root.left, voyage[1:ri])
            fr = self.internal(root.right, voyage[ri:])
            if fl != None and fr != None:
                return fl+fr
            return None
        elif voyage[1] == root.right.val:
            li = -1
            for x in range(1, len(voyage)):
                if voyage[x] == root.left.val:
                    li = x
                    break
            if li == -1:
                return None
            fr = self.internal(root.right, voyage[1:li])
            fl = self.internal(root.left, voyage[li:])
            if fl != None and fr != None:
                return [root.val]+fl+fr
            return None
        else:
            return None
