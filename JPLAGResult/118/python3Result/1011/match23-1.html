# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        if not root:
            return []
        if root.val != voyage[0]:
            return [-1]
        voyage.pop(0)
        if not root.left:
            return self.flipMatchVoyage(root.right, voyage)
        if not root.right:
            return self.flipMatchVoyage(root.left, voyage)
        ans = []
        
if root.left.val != voyage[0]:
            ans.append(root.val)
            root.left, root.right = root.right, root.left
        left = self.flipMatchVoyage(root.left, voyage)
        right = self.flipMatchVoyage(root.right, voyage)
        if left == [-1] or right == [-1]:
            return [-1]
        return ans + left + right
