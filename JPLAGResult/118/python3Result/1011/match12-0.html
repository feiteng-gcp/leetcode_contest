# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        n = len(voyage)
        if not root and n > 0:
            return [-1]
        
        i = 0
        st = [root]
        res = []
        
        while i < n:
            cur = st.pop(-1)
            if (not cur and i < n) or (cur and i >= n):
                return [-1]
            if cur.val != voyage[i]:
                return [-1]
            # left
            if cur.left:
                if i + 1 < n and cur.left.val != voyage[i + 1]:
                    res.append(cur.val)
                    cur.left, cur.right = cur.right, cur.left
            if cur.right: st.append(cur.right)
            if cur.left: st.append(cur.left)
            i += 1
        
        return res
