# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        
        def flips(node, voyage, ans):
            if node.val != voyage[0]:
                return -1
            
            if not node.left and not node.right:
                return 1

            if not node.left:
                ret = flips(node.right, voyage[1:], ans)
                if ret == -1:
                    return -1
                return 1 + ret

            if not node.right:
                ret = flips(node.left, voyage[1:], ans)
                if ret == -1:
                    return -1
                return 1 + ret
            
            if node.left.val != voyage[1]:
                ans.append(node.val)
                node.left, node.right = node.right, node.left
            
            lret = flips(node.left, voyage[1:], ans)
            if lret == -1:
                return -1
            
            rret = flips(node.right, voyage[1 + lret:], ans)
            if rret == -1:
                return -1
            
            return 1 + lret + rret

        ans = []
        ret = flips(root, voyage, ans)
        if ret == -1:
            return [-1]
        return ans
