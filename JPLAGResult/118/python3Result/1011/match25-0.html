# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        def f(root,voyage):
            if root==None:
                if voyage==[]:
                    return 0
                else:
                    return -1
            if voyage==[]:
                return -1
            if root.val!=voyage[0]:
                return -1
            if root.left==None:
                return f(root.right,voyage[1:])
            if root.right==None:
                return f(root.left,voyage[1:])
            if root.left.val not in voyage:
                return -1
            else:
                lr=voyage.index(root.left.val)
            if root.right.val not in voyage:
                return -1
            else:
                rr=voyage.index(root.right.val)
            if (lr>1) and (rr>1):
                return -1
            if lr==1:
                a=f(root.left,voyage[1:rr])
                b=f(root.right,voyage[rr:])
                if (a==-1) or (b==-1):
                    return -1
                return a+b
            a=f(root.right,voyage[1:lr])
            b=f(root.left,voyage[lr:])
            if (a==-1) or (b==-1):
                return -1
            k.append(root.val)
            return a+b+1
        k=[]
        a=f(root,voyage)
        if a==-1:
            return [-1]
        return k
