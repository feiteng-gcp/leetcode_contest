# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        
        def dfs(node,s):
            if not node:
                return []
            if not s or node.val!=s[0]: return None
            if node.left:
                if node.right:
                    if len(s)<3:return None
                    if s[1]==node.left.val:
                        if node.right.val not in s: return None
                        idx = s.index(node.right.val)
                        l 
= dfs(node.left, s[1:idx]) 
                        if l is None:return None
                        r = dfs(node.right,s[idx:])
                        if r is None:return None
                        return l+r
                    else:
                        if node.left.val not in s: return None
                        idx = s.index(node.left.val)
                        l=dfs(node.left, s[idx:]) 
                        if l is None:return None
                        r=dfs(node.right,s[1:idx])
                        if r is None:return None
                        return [node.val] + l+r
                else:
                    return dfs(node.left,s[1:])
            else:
                if node.right:
                    return dfs(node.right,s[1:])
                else:
                    return []
                
        a = dfs(root,voyage)
        return [-1] if a is None else a
