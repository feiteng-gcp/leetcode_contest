# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        #if root.val != voyage[0]: return [-1]
        N = len(voyage)
        #if N <= 2: return []
        res = []
        
        def recursion(node):
            if node.val != voyage.pop(0):
                
return False
            if voyage and node.left and node.right and voyage[0] == node.right.val:
                node.left, node.right = node.right, node.left
                res.append(node.val)
            ret = True
            if node.left:
                ret = ret and recursion(node.left)
            if node.right:
                ret = ret and recursion(node.right)
            return ret
            
        return [-1] if not recursion(root) else res
