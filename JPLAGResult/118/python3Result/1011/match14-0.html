# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        stack = [root]
        l = len(voyage)
        j = 0
        ans = []
        while stack:
          temp = stack.pop()
          #print(temp.val)
          if temp.val != voyage[j]:
            return [-1]
          if j+1<l and temp.right and temp.right.val == voyage[j+1] and temp.left:
            ans.append(temp.val)
            temp.left,temp.right = temp.right,temp.left
          if temp.right:
            stack.append(temp.right)
          if temp.left:
            stack.append(temp.left)
          j += 1
        return ans
