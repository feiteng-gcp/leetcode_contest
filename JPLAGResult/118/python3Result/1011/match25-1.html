# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        self.res = []
        def flip(root, voyage):
            if not root:
                return True
            if root.val != voyage[0]:
                return False
            if len(voyage) == 1:
                
return True
            if not root.left:
                return flip(root.right, voyage[1:])
            if not root.right:
                return flip(root.left, voyage[1:])
            if voyage[1] == root.left.val:
                for i in range(2, len(voyage)):
                    if voyage[i] == root.right.val:
                        return flip(root.left, voyage[1:i]) and flip(root.right, voyage[i:])
            if voyage[1] == root.right.val:
                self.res.append(root.val)
                for i in range(2, len(voyage)):
                    if voyage[i] == root.left.val:
                        return flip(root.right, voyage[1:i]) and flip(root.left, voyage[i:])
            return False
        
        if flip(root, voyage):
            return self.res
        else:
            return [-1]
