# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        result = self.flipMatchVoyageH(root, voyage)
        result = list(filter(lambda x: x != -2, result))
        if not result:
            return []
        if -1 in result:
            return [-1]
        else:
            return result
        
    def flipMatchVoyageH(self, root, voyage):
        result = []
        result.append(-2)
        if root.left == None:
            if root.right == None:
                return result
            else:
                rightie = root.right.val
                if voyage.index(rightie) < voyage.index(root.val):
                    result.append(-1)
                    return result
                else:
                    return self.flipMatchVoyageH(root.right, voyage)
        elif root.right == None:
            leftie = root.left.val
            if voyage.index(leftie) < voyage.index(root.val):
                result.append(-1)
                return result
            else:
                return self.flipMatchVoyageH(root.left, voyage)
        leftie = root.left.val
        rightie = root.right.val 
        curr = root.val
        if voyage.index(leftie) < voyage.index(curr) or voyage.index(rightie) < voyage.index(curr):
            return [-1]
        else:
            if voyage.index(leftie) > voyage.index(rightie):
                result.append(curr)
                temp = root.left
                root.left = root.right
                root.right = temp
                
return result + self.flipMatchVoyageH(root.left, voyage) + self.flipMatchVoyageH(root.right, voyage)
            else:
                return self.flipMatchVoyageH(root.left, voyage) + self.flipMatchVoyageH(root.right, voyage)
