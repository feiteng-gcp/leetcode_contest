# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        # print(root, voyage)
        if root is None:
            if voyage == []:
                return []
            else:
                return [-1]
        if len(voyage) == 0:
            return [-1]
        if root.val != voyage[0]:
            return [-1]
        if root.left is None and root.right is None:
            if len(voyage) == 1:
                return []
            else:
                return [-1]
        if root.left is not None and root.right is not None:
            if root.left.val not in voyage or root.right.val not in voyage:
                return [-1]
            left_index = voyage.index(root.left.val)
            right_index = voyage.index(root.right.val)
            # print(left_index, right_index)
            if left_index < right_index:
                if left_index != 1:
                    ans = [-1]
                else:
                    ans = self.flipMatchVoyage(root.left, voyage[left_index:right_index]) + self.flipMatchVoyage(root.right, voyage[right_index:])
            else:
                if right_index != 1:
                    ans = [-1]
                else:
                    ans = self.flipMatchVoyage(root.right, voyage[right_index:left_index]) + self.flipMatchVoyage(root.left, voyage[left_index:])
                    ans += [root.val]
        elif root.left is None:
            ans = self.flipMatchVoyage(root.right, voyage[1:])
        else:
            ans = self.flipMatchVoyage(root.left, voyage[1:])
        if -1 in ans:
            return [-1]
        else:
            return list(set(ans))
