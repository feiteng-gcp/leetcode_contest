# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        ans = [False, 0, []]
        def dfs(r, p):
            
if p[0] or not r:
                return
            if r.val != voyage[p[1]]:
                p[0] = True
                p[2] = [-1]
                return
            p[1] += 1
            if r.left and r.left.val == voyage[p[1]]:
                dfs(r.left, p)
                dfs(r.right, p)
            elif not r.left and r.right and r.right.val == voyage[p[1]]:
                dfs(r.right, p)
            elif not r.left and not r.right:
                return
            else:
                p[2].append(r.val)
                r.left, r.right = r.right, r.left
                dfs(r.left, p)
                dfs(r.right, p)
        dfs(root, ans)
        return ans[2]
