# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        reverse = []
        def transverse(root, l, r):
            if voyage[l]!=root.val:
                return False
            l+=1
            if root.left==None and root.right==None:
                if l>r:
                    return True
                else:
                    return False
            if root.left!=None and root.right==None:
                return transverse(root.left,l,r)
            if root.left==None and root.right!=None:
                return transverse(root.right,l,r)
            try:
                lp = voyage.index(root.left.val,l,r+1)
                rp = voyage.index(root.right.val,l,r+1)
            except:
                return False
            if lp==l:
                return transverse(root.left,lp,rp-1) and transverse(root.right,rp,r)
            elif rp==l:
                reverse.append(root.val)
                return transverse(root.right,rp,lp-1) and transverse(root.left,lp,r)
            else:
                return False
        if not transverse(root, 0, len(voyage)-1):
            return [-1]
        else:
            return reverse
