# class TreeNode:
#     def __init__(self, x):
#         self.val = x

#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        res = [0]
        self.helper(root, voyage, 0, len(voyage), res)
        print
(res)
        if res[0] == -1:
            return [-1]
        else:
            return res[1:]
        
    def helper(self, root, voyage, l, r, res):
        # print('call', root.val, voyage, l, r, res)
        if res and res[0] == -1:
            return
        if not root:
            if l == r:
                return
            else:
                res[0] = -1
        if l >= r:
            res[0] = -1
            return
        if root.val != voyage[l]:
            res[0] = -1
            return
        if root.left and root.right:
            if root.left.val != voyage[l+1]:
                res.append(root.val)
                rvi = voyage.index(root.left.val)
                self.helper(root.right, voyage, l+1, rvi, res)
                self.helper(root.left, voyage, rvi, r, res)
            else:
                rvi = voyage.index(root.right.val)
                self.helper(root.left, voyage, l+1, rvi, res)
                self.helper(root.right, voyage, rvi, r, res)
        elif root.right:
            # res.append(root.val)
            self.helper(root.right, voyage, l+1, r, res)
        elif root.left:
            self.helper(root.left, voyage, l+1, r, res)
        else:
