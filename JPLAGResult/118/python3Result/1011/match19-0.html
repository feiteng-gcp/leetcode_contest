# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        Box = collections.namedtuple('Box', ('ind'))
        holder = [0]
        result = []
        
        def flipTraversal(node, holder):
            if not node or node.val != voyage[holder[0]]:
                return False
            
            if node.left is not None:
                if node.left.val != voyage[holder[0] + 1]:
                    node.left, node.right = node.right, node.left
                    result.append(node.val)
                holder[0] += 1
                if not flipTraversal(node.left, holder):
                    return False
            if node.right is not None:
                holder[0] += 1
                return flipTraversal(node.right, holder)
            return True
        
        
        if not root:
            return len(voyage) > 0
        elif flipTraversal(root, holder):
            return result
        else:
            return [-1]
