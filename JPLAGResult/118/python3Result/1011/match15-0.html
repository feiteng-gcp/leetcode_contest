# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
  def flipMatchVoyage(self, root, voyage):
    """
    :type root: TreeNode
    :type voyage: List[int]
    :rtype: List[int]
    """
    self.index = 0
    self.result = []
    # rootNode = self._buildTree(root)
    self.index = 0
    self._canDo = True
    self._traverse(root, voyage)
    if self._canDo:
      return self.result
    return [-1]
  
  def _traverse(self, node, voyage):
    if self._canDo == False:
      return
    if node == None:
      return
    
    if node.val != voyage[self.index]:
      print(self.index)
      print(node.val)
      self._canDo = False
      
    if node.left != None and node.right != None and node.right.val == voyage[self.index + 1]:
      node.right, node.left = node.left, node.right
      self.result.append(node.val)
    
    self.index += 1
    if node.left:
      self._traverse(node.left, voyage)
    if node.right:
      self._traverse(node.right, voyage)
