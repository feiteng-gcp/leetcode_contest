# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        
        self.count = -1
        ans = []
        
        def flip(node):
            if not node:
                return True
            self.count += 1
            
            
if node.val != voyage[self.count]:
                return False
            if node.left:
                if node.left.val != voyage[self.count + 1]:
                    node.left, node.right = node.right, node.left
                    ans.append(node.val)
                
            
            if not flip(node.left):
                return False
            if not flip(node.right):
                return False
            return True
        
        return ans if flip(root) else [-1]
