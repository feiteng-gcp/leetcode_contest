# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        if root.val != voyage[0]:
            return [-1]
        
        self.res = []
        self.stop = False
        self.i = 0
        def _flip(self, cur):
            print(cur.val)
            if self.stop:
                return
            if cur.val != voyage[self.i]: # For right
                self.res = [-1]
                self.stop = True
                return
            self.i += 1
            if cur.left:
                if cur.left.val != voyage[self.i]:
                    cur.left, cur.right = cur.right, cur.left
                    if cur.left and cur.left.val == voyage[self.i]:
                        self.res.append(cur.val)
                    else:
                        self.res = [-1]
                        self.stop = True
                        return
                        
                
            if cur.left:
                _flip(self, cur.left)
                
            if cur.right:
                _flip(self, cur.right)

        _flip(self, root)
        return self.res
