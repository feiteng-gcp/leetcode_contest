# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        rcd = []
        def dfs(root, pos):
            if not root: return 0
            # if root.val == 2:
                # print(pos)
            if voyage[pos] != root.val:
                 return -1
            if root.left and root.right and root.left.val != voyage[pos + 1]:
                root.left, root.right = root.right, root.left
                rcd.append(root.val)
            rst = dfs(root.left, pos + 1)
            if rst == -1: return -1
            rst2 = dfs(root.right, pos +  rst + 1)
            if rst2 == -1: return -1
            return 1 + rst + rst2
        temp = dfs(root, 0)
        return rcd if temp != -1 else [-1]
