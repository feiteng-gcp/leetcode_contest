# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        self.ans = []
        self.faild = False
        self.idx = 0
        self.dfs(root, voyage)
        if self.faild:
            return [-1]
        else:
            return self.ans
    
    def dfs(self, root, v):
        if not root or self.faild:
            return 
        
        if root.val != v[self.idx]:
            self.faild = True
            return
        
        if root.left and root.left.val == v[self.idx + 1]:
            self.idx += 1
            self.dfs(root.left, v)
            self.dfs(root.right, v)
        elif root.left:
            self.ans.append(root.val)
            self.idx += 1
            self.dfs(root.right, v)
            self.dfs(root.left, v)
        else:
            self.idx += 1
            self.dfs(root.right, v)
