# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        self.voyage=voyage[::-1]
        self.ret = []
        if not self.pre(root) or self.voyage:
            return [-1]
        return self.ret
    def pre(self,nd):
        if not nd :return True
        if not self.voyage :return False
        n = self.voyage.pop()
        if nd.val!=n:return False
        if not (nd.left or nd.right):
            return True
        
        if not self.voyage:return False
        if (nd.left and nd.left.val!=self.voyage[-1]) or (not nd.left and nd.right.val!=self.voyage[-1]):
            self.ret .append(n)
            return self.pre(nd.right) and self.pre(nd.left)
        else:       
            return self.pre(nd.left) and self.pre(nd.right)
