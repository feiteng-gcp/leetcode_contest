# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def flipMatchVoyage(self, root, voyage):
        """
        :type root: TreeNode
        :type voyage: List[int]
        :rtype: List[int]
        """
        n = len(voyage)
        r 
= []
        def find(node, i):
            if voyage[i] != node.val:
                return -1
            i += 1
            if node.left or node.right:
                if i == n:
                    return -1
                if node.left and node.right and voyage[i] == node.right.val:
                    r.append(node.val)
                    i = find(node.right, i)
                    if i < 0: return i
                    return find(node.left, i)
                if node.left:
                    i = find(node.left, i)
                    if i < 0: return i
                if node.right:
                    i = find(node.right, i)
            return i
            
        if find(root, 0) != n:
            return [-1]
        return r
