class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        if bound == 0:
            return []
        import math
        xi = int(math.log(bound, x)+1) if x > 1 else 1
        yj = int(math.log(bound, y)+1) if y > 1 else 1
        s  = set()
        for i in range(xi):
            for j in range
(yj):
                r = x**i + y**j
                if r <= bound:
                    s.add(r)
        return list(s)
