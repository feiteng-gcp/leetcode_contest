class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        res = {0}
        res.remove(0)
        for i in range(101):
            f = x ** i
            if f >= bound:
                break
            for j in range(101):
                s 
= y ** j
                if f + s > bound:
                    break
                else:
                    res.add(f+s)
        return list(res)
