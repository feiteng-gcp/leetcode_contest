class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        ans = set()
        for i in range(20):
            for j in range(20):
                pow_num = x ** i + y ** j
                if pow_num <= bound:
                    ans.add(pow_num)
        return list(ans)
