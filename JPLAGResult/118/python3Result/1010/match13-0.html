class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        list1, list2 = [], []
        i = 1
        while i < bound:
            list1.append(i)
            i *= x
            if x == 1:
                break
        i = 1
        while i < bound:
            list2.append(i)
            i *= y
            if y == 1:
                break
        rst = set()
        for i in range(len(list1)):
            for j in range(len(list2)):
                if list1[i] + list2[j] > bound:
                    break
                rst.add(list1[i] + list2[j])
        return list(rst)
