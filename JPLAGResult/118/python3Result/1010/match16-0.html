import math


class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        if bound == 0:
            return []
        canx = [1] 
        cany = [1]
        if x != 1:
            logx = int(math.log(bound, x))
            for i in range(1, logx+1):
                canx.append(int(math.pow(x, i)))
        if y != 1:        
            logy = int(math.log(bound, y))
            for i in range(1, logy+1):
                cany.append(int(math.pow(y, i)))

        dic = set()
        for xx in canx:
            for yy in cany:
                if xx+yy <= bound:
                    dic.add(xx+yy)
        return list(dic)
