from math import log, ceil

class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        if bound < 2:
            return []
        
        s = set()
        
        if x == 1:
            ri = 1
        else:
            ri = int(ceil(log(bound,x)))
        
        if y == 1:
            rj = 1
        else:
            rj = int(ceil(log(bound,y)))      
        
        
        for i in range(ri):
            for j in range
(rj):
                t = x**i + y**j
                if (t <= bound):
                    s.add(t)
        
        return list(s)
