class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        res = set()
        tmpx, tmpy = [], []
        if x == 1:
            tmpx.append(1)
        else:
            i = 0
            while x ** i <= bound:
                tmpx.append(x ** i)
                i += 1
        if y == 1:
            tmpy.append(1)
        else:
            i = 0
            while y ** i <= bound:
                tmpy.append(y ** i)
                i += 1
        
        
for i in range(len(tmpx)):
            for j in range(len(tmpy)):
                if tmpx[i] + tmpy[j] > bound:
                    break
                res.add(tmpx[i] +tmpy[j])
        
        return list(res)
