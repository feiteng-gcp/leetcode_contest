class Solution:
    def get_list(self, a, bound):
        if a == 1:
            return [1]

        a_list = [1]
        cnt = 1
        while cnt < bound:
            a_list.append(cnt)
            cnt *= a
        return a_list


    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        ans = set()

        x_list = self.get_list(x, bound)
        y_list = self.get_list(y, bound)

        for a in x_list:
            for b in y_list:
                if a + b <= bound:
                    ans.add(a + b)

        return list(ans)
