class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        if bound < 2:
            return []
        
        ans_set = set()
        for pow_x in range(int(math.log(bound, x)) + 1 if x > 1 else 1):
            for pow_y in range(int(math.log(bound, y)) + 1 if y > 1 else 1):
                val = x**pow_x + y**pow_y
                if val <= bound and val not in ans_set:
                    ans_set.add(val)
        
        return list(ans_set)
