class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        ans=[]
        xn=1
        while xn<bound:
            yn=1
            while xn+yn<=bound:
                ans
.append(xn+yn)
                yn*=y
                if yn==1:break
            xn*=x
            if xn==1:break
        return list(set(ans))
