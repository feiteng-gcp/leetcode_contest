class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        ans = set()
        l = 1
        while l <= bound:
            r = 1
            while r + l <= bound:
                ans
.add(r + l)
                r *= y
                if y == 1:
                    break
            l *= x
            if x == 1:
                break
            
        
        return list(ans)
