class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        
        ans = []
        
        if bound < 2:
            return []

        if x == 1 and y == 1:
            return [2]
        
        if x == 1:
            j = 0
            while True:
                v = x + y ** j
                if v > bound:
                    break
                ans.append(v)
                j += 1
            return ans

        if y == 1:
            i = 0
            while True:
                v = x ** i + y
                if v > bound:
                    break
                ans.append(v)
                i += 1
            return ans

        ans = set()
        for i in range(bound):
            vx = x ** i
            if vx > bound:
                break
            for j in range(bound):
                v = vx + y ** j
                if v > bound:
                    break
                ans.add(v)

        return list(ans)
