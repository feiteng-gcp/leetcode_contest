class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        if bound <= 1: return []
        
        res = set()
        x_set = set([1])
        y_set = set([1])
        x_curr, y_curr = x, y
        if x != 1:
            while x_curr < bound:
                x_set.add(x_curr)
                x_curr *= x
        
        if y != 1:
            while y_curr < bound:
                y_set.add(y_curr)
                y_curr *= y

        for i in x_set:
            for j in y_set:
                if i+j <= bound:
                    res.add(i+j)
        
        return list(res)
