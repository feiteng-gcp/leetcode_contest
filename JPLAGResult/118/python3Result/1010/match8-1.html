import math
class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        if bound <= 1:
            return []
        bx 
= 1 if x == 1 else int(math.log(bound, x))+1
        by = 1 if y == 1 else int(math.log(bound, y))+1
        res = set()
        for i in range(bx):
            for j in range(by):
                if x**i + y**j <= bound:
                    res.add(x**i+y**j)
                    
        return list(res)
