class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        ans = set()
        idx1 = idx2 = 0
        while x**idx1 <= bound:
            idx2 = 0
            while y**idx2 <= bound - x**idx1:
                if x**idx1 + y**idx2 <= bound:
                    # print(x**idx1 + y**idx2, 'idx1:', idx1, 'idx2:', idx2)
                    ans.add(x**idx1 + y**idx2)
                idx2 
+= 1
                if y == 1: break
            idx1 += 1
            if x == 1: break
        ans = list(ans)
        ans.sort()
        return ans
