class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        if bound < 2:
            return []
        if x == 1 and y == 1:
            return [2]
        res = {2}
        def f(b):
            t = b
            while t+1 <=bound:
                res.add(t+1)
                t*=b
        if x == 1:
            f(y)
            return list(res)
        if y == 1:
            f(x)
            return list(res)
        a 
= 1
        while a < bound:
            b = 1
            while a+b<=bound:
                res.add(a+b)
                b*=y
            a*=x
        return list(res)
