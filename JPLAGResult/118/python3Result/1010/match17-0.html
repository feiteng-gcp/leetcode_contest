class Solution:
    def powerfulIntegers(self, x, y, bound):
        """
        :type x: int
        :type y: int
        :type bound: int
        :rtype: List[int]
        """
        px = list(set([x**i for i in range(21)]))
        py = list(set([y**i for i in range(21)]))
        ans = set()
        for a in px:
            for b in py:
                if a+b <= bound:
                    ans.add(a+b)
        # print(list(ans))
        return list(ans)
