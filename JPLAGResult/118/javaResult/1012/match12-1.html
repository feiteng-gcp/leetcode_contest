class Solution {
    public boolean isRationalEqual(String S, String T) {
        R s = parse(S);
        R t = parse(T);
        int diff = s.a - t.a;
        for (int k = 0; k < 1000; k++) {
            diff = diff * 10 + (s.get(k) - t.get(k));
            if (Math.abs(diff) >= 10) {
                return false;
            }
        }
        return true;
    }
    
    private R parse(String s) {
        int loc = s.indexOf('.');
        if (loc == -1) {
            return new R(Integer.parseInt(s), "", "");
        }
        int a = Integer.parseInt(s.substring(0, loc));
        
        String frac = s.substring(loc + 1);
        loc = frac.indexOf('(');
        if (loc == -1) {
            return new R(a, frac, "");
        }
        String b = frac.substring(0, loc);
        String c = reduce(frac.substring(loc + 1, frac.length() - 1));
        return new R(a, b, reduce(c));
    }
    
    private String reduce(String str) {
        int len = str.length();
        for (int m = 1; m < len; m++) {
            if (len % m == 0) {
                boolean flag = true;
                for (int i = m; i < len; i++) {
                    if (str.charAt(i) != str.charAt(i - m)) {
                        flag = false;
                        break;
                    }
                }
                if (flag) {
                    return str.substring(m);
                }
            }
        }
        return str;
    }
}

class R {
    int a;
    String b, c;
    
    public R(int a, String b, String c) {
        this.a = a;
        this.b = b;
        this.c = c;
    }
    
    public char get(int k) {
        if (k < b.length()) {
            return b.charAt(k);
        }
        if (c.length() == 0) {
            return '0';
        }
        k -= b.length();
        k %= c.length();
        return c.charAt(k);
    }
}
