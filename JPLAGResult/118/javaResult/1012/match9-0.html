class Solution {
    public boolean isRationalEqual(String S, String T) {
        String[] sip = S.split("\\.");
        String[] tip = T.split("\\.");
        int si = Integer.parseInt(sip[0]);
        int ti = Integer.parseInt(tip[0]);
        
        /*System.out.println(si);
        System.out.println(ti);
        
        if (si != ti) {
            return false;
        }*/
        
        String[] sr = get(sip);
        String[] tr = get(tip);
        
        String a = constr(sr, 50);
        String b = constr(tr, 50);
        
        String xa = sip[0] + "." + a;
        String xb = tip[0] + "." + b;
        
        double va = Double.parseDouble(xa);
        double vb = Double.parseDouble(xb);
        return Math.abs(va - vb) < 1e-42;
        /*for (int i = 0; i < Math.max(sr[0].length() + 2 * sr[1].length(), tr[0].length() + 2 * tr[1].length()); i++) {
            if (a.charAt(i) != b.charAt(i)) {
                return false;
            }
        }
        return true;*/
    }
    
    String constr(String[] a, int len) {
        String ret = a[0];
        while (ret.length() < len) {
            ret += a[1];
        }
        return ret;
    }
    
    
    String[] get(String[] y) {
        if (y.length == 1 || y[1].isEmpty()) {
            return new String[] {"0", "0"};
        }
        String x = y[1];
        String[] ret = new String[] {"", ""};
        int i = 0;
        for (; i < x.length(); i++) {
            if (x.charAt(i) == '(') {
                break;
            } 
            ret[0] += x.charAt(i);
        }
        if (i < x.length()) {
            i++;
            while (i < x.length() && x.charAt(i) != ')') {
                ret[1] += x.charAt(i);
                i++;
            }
        } else {
            ret[1] = "0";
        }
        return ret;
    }
}
