class Solution {
    public boolean isRationalEqual(String S, String T) {
        double v1 = parse(S);
        double v2 = parse(T);
        return (Math.abs(v2 - v1) < 0.00000000001);
    }

    public double parse(String s) {
        double res = 0;
        int i = 0;
        while (i < s.length() && Character.isDigit(s.charAt(i))) {
            res *= 10;
            res += s.charAt(i) - '0';
            i++;
        }
        i++;
        double f = 0.1;
        while (i < s.length() && Character.isDigit(s.charAt(i))) {
            res += f * (s.charAt(i) - '0');
            f /= 10;
            i++;
        }
        if (i < s.length()&&s.charAt(i) == '(') {
            i++;
            double ff = 0.1;
            double rep = 0;
            while (i < s.length() && Character.isDigit(s.charAt(i))) {
                rep += ff * (s.charAt(i) - '0');
                ff /= 10;
                i++;
            }
            res += f * 10 * (rep / (1 - ff * 10));
        }
        return res;
    }
}
