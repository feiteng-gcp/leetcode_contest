class Solution {
   public boolean isRationalEqual(String S, String T) {
		int iPartS = ipart(S);
		String nPartS = npart(S);
		String rPartS = rpart(S);
		StringBuilder preS = new StringBuilder();
		preS.append(nPartS);
		for (int i = 0; i < 10; i++) preS.append(rPartS);

		int iPartT = ipart(T);
		String nPartT = npart(T);
		String rPartT = rpart(T);
		StringBuilder preT = new StringBuilder();
		preT.append(nPartT);
		for (int i = 0; i < 10; i++) preT.append(rPartT);

		if (Math.abs(iPartS - iPartT) > 1) return false;
		if (iPartS == iPartT) {

			if (allNine(rPartS)) {
				if (allNine(rPartT)) {
					if (preS.length() > preT.length()) {
						return preS.toString().startsWith(preT.toString());
					} else {
						return preT.toString().startsWith(preS.toString());
					}
				}
				if (!allZero(rPartT)) return false;
				while (preS.charAt(preS.length() - 1) == '9') preS.deleteCharAt(preS.length() - 1);
				if (!allZero(preT.substring(preS.length()))) return false;
				int preSi = Integer.parseInt(preS.toString());
				int preTi = Integer.parseInt(preT.substring(0, preS.length()));
				return preSi + 1 == preTi;
			}

			if (allNine(rPartT)) {
				if (!allZero(rPartS)) return false;
				while (preT.charAt(preT.length() - 1) == '9') preT.deleteCharAt(preT.length() - 1);
				if (!allZero(preS.substring(preT.length()))) return false;
				int preTi = Integer.parseInt(preT.toString());
				int preSi = Integer.parseInt(preS.substring(0, preT.length()));
				return preTi + 1 == preSi;
			}

			if (preS.length() < 10 || preT.length() < 10) {
				if (preS.length() == 0) {
					return allZero(preT.toString());
				}
				if (preT.length() == 0) {
					return allZero(preS.toString());
				}
				return preS.toString().equals(preT.toString());
			}

			if (preS.length() > preT.length()) {
				return preS.toString().startsWith(preT.toString());
			} else {
				return preT.toString().startsWith(preS.toString());
			}
		} else if (iPartS > iPartT) {
			if (!allZero(preS.toString())) return false;
			if (!allNine(preT.toString()) || !allNine(rPartT)) return false;
		} else {
			if (!allZero(preT.toString())) return false;
			if (!allNine(preS.toString()) || !allNine(rPartS)) return false;
		}
		return true;
	}

	private boolean allZero(String S) {
		for (int i = 0; i < S.length(); i++) if (S.charAt(i) != '0') return false;
		return true;
	}

	private boolean allNine(String S) {
		if (S.length() == 0) return false;
		for (int i = 0; i < S.length(); i++) if (S.charAt(i) != '9') return false;
		return true;
	}

	private int ipart(String S) {
		String[] sp = S.split("\\.");
		return Integer.parseInt(sp[0]);
	}

	private String npart(String S) {
		String[] sp = S.split("\\.");
		if (sp.length != 2) return "";
		String[] sp2 = sp[1].split("\\(");
		return sp2[0];
	}

	private String rpart(String S) {
		String[] sp = S.split("\\.");
		if (sp.length != 2 || sp[1].indexOf('(') < 0) return "";
		String[] sp2 = sp[1].split("\\(");
		return sp2[1].replace(")", "");
	}


}
