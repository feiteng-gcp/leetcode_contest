class Solution {
	public boolean isRationalEqual(String S, String T) {
		double d1 = Double.parseDouble(convert(S));
		double d2 = Double.parseDouble(convert(T));
		return d1 == d2;
	}

	private String convert(String s) {
		String[] a = s.split("[()]");
		if (a.length > 1) {
       // tr(a);
			s = a[0] + a[1];
			for (int i = 0; i < 20; i++)
				s = s + a[1];
	//		s = s + a[2];
		}
        // tr(s);
		return s;
	}
    	public static void tr(Object... o) {
		System.out.println(Arrays.deepToString(o));
	}
}
