	    public boolean isRationalEqual(String S, String T) {
	    	int[] a = go(S);
	    	int[] b = go(T);
	    	java.math.BigInteger A, B;
	    	{
		    	StringBuilder sb = new StringBuilder();
		    	for(int v : a){
		    		sb.append(v);
		    	}
		    	A = new java.math.BigInteger(sb.toString());
	    	}
	    	{
		    	StringBuilder sb = new StringBuilder();
		    	for(int v : b){
		    		sb.append(v);
		    	}
		    	B = new java.math.BigInteger(sb.toString());
	    	}
	    	java.math.BigInteger cha = A.subtract(B).abs();
	    	return cha.compareTo(java.math.BigInteger.TEN.pow(100)) <= 0;
	    }
