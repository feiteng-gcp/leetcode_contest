class Solution {
  double convertToDouble(String s) {
        int index = s.indexOf('(');
        if (index < 0) {
           // System.out.println("s = " + s);
            return Double.valueOf(s);
        } else {
            StringBuilder sb = new StringBuilder(s.substring(0, index));
            int rightIndex = s.indexOf(')');
            String repeat = s.substring(index + 1, rightIndex);

            while (sb.length() < 50) {
                sb.append(repeat);
            }
           // System.out.println("s = " + sb.toString());
            return Double.valueOf(sb.toString());
        }
    }

    public boolean isRationalEqual(String S, String T) {
        return Math.abs(convertToDouble(S)  - convertToDouble(T)) < 1e-8;
    }
}
