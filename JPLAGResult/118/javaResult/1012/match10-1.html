import java.util.regex.Matcher;

import java.util.regex.Pattern;

class Solution {
    public boolean isRationalEqual(String S, String T) {
        Pattern pattern = Pattern.compile("^([0-9]*)[.]?([0-9]*)([(]([0-9]+)[)])?$");
        String s1, s2, s3, t1, t2, t3;
        Matcher matcher = pattern.matcher(S);
        matcher.find();
        s1 = matcher.group(1) != null ? matcher.group(1) : "";
        s2 = matcher.group(2) != null ? matcher.group(2) : "";
        s3 = matcher.group(4) != null ? matcher.group(4) : "";
        matcher = pattern.matcher(T);
        matcher.find();
        t1 = matcher.group(1) != null ? matcher.group(1) : "";
        t2 = matcher.group(2) != null ? matcher.group(2) : "";
        t3 = matcher.group(4) != null ? matcher.group(4) : "";

        long sa, sb, ta, tb;
        sa = _intFromString(s2);
        sb = (long)Math.pow(10, s2.length());
        if (s3.length() != 0) {
            sa *= Math.pow(10, s3.length()) - 1;
            sa += _intFromString(s3);
            sb = (long)Math.pow(10, s2.length() + s3.length()) - sb;
        }
        sa += sb * _intFromString(s1);

        ta = _intFromString(t2);
        tb = (long)Math.pow(10, t2.length());
        if (t3.length() != 0) {
            ta *= Math.pow(10, t3.length()) - 1;
            ta += _intFromString(t3);
            tb = (long)Math.pow(10, t2.length() + t3.length()) - tb;
        }
        ta += tb * _intFromString(t1);

        return sa * tb == sb * ta;
    }

    private int _intFromString(String s) {
        if (s == null || s.length() == 0) return 0;
        return Integer.valueOf(s);
    }
}
