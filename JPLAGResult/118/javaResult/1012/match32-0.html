class Solution {
    public boolean isRationalEqual(String S, String T) {
        S = getStr(S);
        T = getStr(T);

        if (!S.contains("(") && !T.contains("(")) {
            double ds = Double.valueOf(S);
            double dt = Double.valueOf(T);
            return ds == dt;
        } else if (S.contains("(") && T.contains("(")) {
            int indexPointS = S.indexOf(".");
            int indexPointT = T.indexOf(".");



            String ss = S.substring(0, indexPointS);
            String st = T.substring(0, indexPointT);
            if (ss.equals(st)) {
                int indexKS = S.indexOf("(");
                int indexKT = T.indexOf("(");
                ss = S.substring(indexPointS + 1, indexKS);
                st = T.substring(indexPointT + 1, indexKT);
                String ks = S.substring(indexKS+1, S.length() - 1);
                String kt = T.substring(indexKT+1, T.length() - 1);
                int ls = ss.length();
                int lt = st.length();
                if (ls < lt) {
                    int m = lt - ls;
                    for (int i = 0; i < m; ++i) {
                        ss = ss + ks.charAt(i % ks.length());
                    }
                    String t = "";
                    for (int i = m; i < m + ks.length(); ++i) {
                        t += ks.charAt(i % ks.length());
                    }
                    ks = t;
                } else if (ls > lt) {
                    int m = ls - lt;
                    for (int i = 0; i < m; ++i) {
                        st = st + kt.charAt(i % kt.length());
                    }
                    String t = "";
                    for (int i = m; i < m + kt.length(); ++i) {
                        t += kt.charAt(i % kt.length());
                    }
                    kt = t;
                }
                if (ss.equals(st)) {
                    int l1 = ks.length();
                    int l2 = kt.length();
                    String sS = "";
                    for (int i = 0; i < l2; ++i) {
                        sS += ks;
                    }
                    String sT = "";
                    for (int i = 0; i < l1; ++i) {
                        sT += kt;
                    }
                    if (sT.equals(sS)) {
                        return true;
                    } else {
                        return false;
                    }
                } else {
                    return false;
                }
            } else {
                return false;
            }

        } else {
            if (S.contains("(")) {
                String t = S.substring(S.indexOf("(") + 1, S.length() - 1);
                int n = Integer.valueOf(t);
                if (n == 0) {
                    S = S.substring(0, S.indexOf("("));
                    double ds = Double.valueOf(S);
                    double dt = Double.valueOf(T);
                    return ds == dt;
                }
            }
            if (T.contains("(")) {
                String t = T.substring(T.indexOf("(") + 1, T.length() - 1);
                int n = Integer.valueOf(t);
                if (n == 0) {
                    T = T.substring(0, T.indexOf("("));
                    double ds = Double.valueOf(S);
                    double dt = Double.valueOf(T);
                    return ds == dt;
                }
            }
            return false;
        }
    }

    private String getStr(String S) {
        if (S.contains("(")) {
            int index = S.indexOf("(");
            String s = S.substring(index+1, S.length() - 1);
            boolean flag = false;
            for (int i = 0; i < s.length(); ++i) {
                if (s.charAt(i) != '9') {
                    flag = true;
                    break;
                }
            }
            if (!flag) {
                String t = S.substring(0, index);
                int i;
                for (i = t.length() - 1; i >= 0; --i) {
                    if (t.charAt(i) != '9' && t.charAt(i) != '.') {
                        break;
                    }
                }
                S = "";
                if (i == -1) {
                    S += "1";
                    for (i = 1; i < t.length(); ++i) {
                        if (t.charAt(i) == '.') {
                            S += '.';
                        } else {
                            S += "0";
                        }
                    }
                } else {
                    S = t.substring(0, i);
                    char c = t.charAt(i);
                    S += (char) (c + 1);
                    for (int j = i + 1; j < t.length(); ++j) {
                        if (t.charAt(j) == '.') {
                            S += '.';
                        } else {
                            S += "0";
                        }
                    }
                }
            }
        }
        return S;
    }
}
