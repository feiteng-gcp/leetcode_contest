class Solution {
    public boolean isRationalEqual(String S, String T) {
        double d1 = convert(S);
        double d2 = convert(T);
        if(Math.abs(d1-d2)<1e-9) return true;
        return false;
    }
    public double convert(String s){
        int i=0;
        for(i=0; i<s.length(); i++){
            if(s.charAt(i)=='(')
                break;
        }
        double temp1 = Double.parseDouble(s.substring(0,i));
        int j=i;
        boolean flag  = false;
        for(j=i; j<s.length();j++){
            if(s.charAt(j)==')'){ flag = true; break;}
        }
        int ct = 0;
        StringBuffer sb = new StringBuffer(s.substring(0, i));
        if(flag){
        while(ct<15){
            sb.append(s.substring(i+1, j));
            ct+=(j-i-1);
        }
        }
        // System.out.println(sb.toString());
        return Double.parseDouble(sb.toString());
    }
}
