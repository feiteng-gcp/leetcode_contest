class Solution {
    public boolean isRationalEqual(String S, String T) {
        long n1 = 0, m1 = 0, n2 = 0, m2 = 0;
        if (S.indexOf(".") == -1) {
            n1 = Integer.valueOf(S);
            m1 = 1;
        } else if (S.indexOf("(") == -1) {
            String str = S.substring(0, S.indexOf(".")) + S.substring(S.indexOf(".") + 1);
            n1 = Integer.valueOf(str);
            m1 = (int)Math.pow(10, S.substring(S.indexOf(".") + 1).length());
        } else {
            int time1 = S.substring(S.indexOf(".") + 1, S.indexOf("(")).length();
            
int time2 = time1 + S.substring(S.indexOf("(")).length() - 2;
            String str = S.substring(0, S.indexOf("("));
            String str1 = str.substring(0, str.indexOf(".")) + str.substring(str.indexOf(".") + 1);
            int v1 = str1.length() == 0 ? 0 : Integer.valueOf(str1);
            long v2 = v1 * (long)Math.pow(10, time2-time1) + Integer.valueOf(S.substring(S.indexOf("(")+1, S.length()-1));
            m1 = (long)Math.pow(10, time2) - (long)Math.pow(10, time1);
            n1 = v2 - v1;
            // System.out.println(time1 + "  " + time2);
        }
        if (T.indexOf(".") == -1) {
            n2 = Integer.valueOf(T);
            m2 = 1;
        } else if (T.indexOf("(") == -1) {
            String str = T.substring(0, T.indexOf(".")) + T.substring(T.indexOf(".") + 1);
            n2 = Integer.valueOf(str);
            m2 = (int)Math.pow(10, T.substring(T.indexOf(".") + 1).length());
        } else {
            int time1 = T.substring(T.indexOf(".") + 1, T.indexOf("(")).length();
            int time2 = time1 + T.substring(T.indexOf("(")).length() - 2;
            String str = T.substring(0, T.indexOf("("));
            String str1 = str.substring(0, str.indexOf(".")) + str.substring(str.indexOf(".") + 1);
            int v1 = str1.length() == 0 ? 0 : Integer.valueOf(str1);
            long v2 = v1 * (long)Math.pow(10, time2-time1) + Integer.valueOf(T.substring(T.indexOf("(")+1, T.length()-1));
            m2 = (long)Math.pow(10, time2) - (long)Math.pow(10, time1);
            n2 = v2 - v1;
        }
        System.out.println(n1 + "  " + m1 + "  " + n2 + "  " + m2);
        return n1 * m2 == m1 * n2;
    }
}
