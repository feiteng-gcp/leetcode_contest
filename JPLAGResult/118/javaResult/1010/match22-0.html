class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        List<Integer> xp = new ArrayList<>();
        List<Integer> yp = new ArrayList<>();
        int c = 0;
        while (true) {
            if ((int) Math.pow(x, c) > bound)
                break;
            xp.add((int) Math.pow(x,c++));
            if (x == 1) break;
        }
        c = 0;
        while (true) {
            if ((int) Math.pow(y, c) > bound)
                break;
            yp.add((int) Math.pow(y,c++));
            if (y == 1) break;
        }
        HashSet<Integer> allSums = new HashSet<>();
        for (int i : xp) {
            for (int j : yp) {
                if (i+j <= bound)
                allSums.add(i + j);
            }
        }
        List<Integer> ret = new ArrayList<>();
        for (int i : allSums) {
            ret.add(i);
        }
        return ret;
    }
}
