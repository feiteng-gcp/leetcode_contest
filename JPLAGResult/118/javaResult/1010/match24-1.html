class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        if(bound < 2) {
            return Collections.<Integer>emptyList();
        }
        
List<Integer> powX = buildPowers(x, bound);
        List<Integer> powY = buildPowers(y, bound);
        Set<Integer> res = new HashSet<>();
        for(int i = 0; i < powX.size(); i++) {
            for(int j = 0; j < powY.size(); j++) {
                int value = powX.get(i) + powY.get(j);
                if(value <= bound) {
                    
res.add(value);
                }
            }
        }
        
        return new ArrayList<Integer>(res);
    }
    
    private List<Integer> buildPowers(int x, int bound) {
        if(x == 1) {
            return Arrays.asList(1);
        }
        int value = 1;
        List<Integer> res = new ArrayList<>();
        while(value <= bound) {
            res.add(value);
            value *= x;
        }
        return res;
    }
    
}
