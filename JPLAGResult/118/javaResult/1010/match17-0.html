class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> set = new HashSet();
        long[] xp = new long[32];
        xp[0] = 1;
        int xn = 0;
        long[] yp = new long[32];
        int yn = 0;
        yp[0] = 1;
        if(x != 1)
        {
            for(; xp[xn] <= bound; xn++)
            {
            xp[xn + 1] = xp[xn] * x;
            }
        }
        
        if(y != 1)
        {
        for(; yp[yn] <= bound; yn++)
        {
            yp[yn + 1] = yp[yn] * y;
        }
        }
        
        for(int i = 0; i <= xn; i++)
        {
            for(int j = 0; j <= yn; j++)
            {
                if(xp[i] + yp[j] <= bound)
                {
                    int val = (int)(xp[i] + yp[j]);
                    set.add(val);
                }
            }
        }
        
        return new ArrayList(set);
    }
}
