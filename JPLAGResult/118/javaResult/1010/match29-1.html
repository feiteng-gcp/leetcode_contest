class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        ArrayList<Integer> ans = new ArrayList<Integer>();
        ArrayList<Double> allx = new ArrayList<Double>();
        ArrayList<Double> ally = new ArrayList<Double>();
        double topx = (Math.log10(bound) / Math.log10(x));
        double topy = (Math.log10(bound) / Math.log10(y));
        if (x == 1 || x == 0) {
            topx = 1;
        }
        if (y == 1 || y == 0) {
            topy = 1;
        }
        for (int i = 0; i < topx; i++) {
            double res = Math.pow(x,i);
            if (!allx.contains(res)) {
                allx.add(res);
            }
        }
        for (int i = 0; i < topy; i++) {
            double res = Math.pow(y,i);
            if (!ally.contains(res)) {
                ally.add(res);
            }
        }
        
for (int i = 0; i < allx.size(); i++) {
            for (int j = 0; j < ally.size(); j++) {
                int res = (int)Math.round(allx.get(i)+ally.get(j));
                if (!ans.contains(res) && res <= bound) {
                    
ans.add(res);
                }
            }
        }
        return ans;
    }
}
