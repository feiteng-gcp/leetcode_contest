class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; ; i++) {
            int t = (int) Math.pow(x, i);
            if (t > bound) {
                break;
            }
            for (int j = 0; ; j++) {
                int u = (int) Math.pow(y, j);
                int r = u + t;
                if (r <= bound) {
                    set.add(r);
                } else {
                    break;
                }
                if (y == 1) {
                    break;
                }
            }
            if (x == 1) {
                break;
            }
        }
        List<Integer> list = new ArrayList<>(set);
        return list;
    }
}
