class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        HashSet<Integer> ret=new HashSet<Integer>();
        for(int i=0;i<bound;i++)
        {
            int bou=bound-(int)Math.pow(x,i);
            if(bou<0)break;
            for(int j=0;j<bound;j++)
            {
                int kk=bou-(int)Math.pow(y,j);
                if(kk<0)break;
                else
                {
                    ret.add((int)(Math.pow(x,i)+
Math.pow(y,j)));
                }
            }
        }
        List<Integer> setList = new ArrayList<Integer>(ret);
        return setList;
    }
}
