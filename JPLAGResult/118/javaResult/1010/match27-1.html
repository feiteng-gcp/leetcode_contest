class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        List<Integer> list = new ArrayList<>();
        
        Set<Integer> set = new HashSet<>();
        int candidateX = 1;
        for(int i=0; candidateX<=bound;i++){
            int candidateY =1;
            for(int j=0; candidateX + candidateY<=bound;j++){
                set.add(candidateX + candidateY);
                
if(candidateY * y == candidateY) break;
                candidateY *= y;
            }
            if(candidateX * x == candidateX) break;
            candidateX *= x;
        }
        
        for(int num: set) list.add(num);
        return list;
    }
}
