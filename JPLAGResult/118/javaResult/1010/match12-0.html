class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> uniq = new HashSet<>();
        for(int xx=1; xx<=bound; xx *= x) {
            for(int yy=1; yy<=bound; yy *= y) {
                int sum = xx + yy;
                if (sum <= bound) {
                    uniq.add(sum);
                }
                if (y == 1) break;
            }
            if (x == 1) break;
        }
        List<Integer> ret = new ArrayList<>(uniq);
        return ret;
    }
}
