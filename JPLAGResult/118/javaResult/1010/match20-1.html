	class S
olution {
	    public List<Integer> powerfulIntegers(int x, int y, int bound) {
	        TreeSet<Integer> ts = new TreeSet<>();
	        for(int i = 0;i <= 40;i++){
	            int u = (int)Math.pow(x, i);
	            if(u > bound)break;
	            for(int j = 0;j <= 40;j++){
	                int v = (int)Math.pow(y, j);
	                if(u+v > bound)break;
	                ts.add(u+v);
	            }
	        }
	        return new ArrayList<Integer>(ts);
	    }
	}
