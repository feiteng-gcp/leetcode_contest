class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        // x = 3; y = 5; bound = 15;
        Set<Integer> set = new HashSet<>();
        for (int i = 1; i < bound; i *= x) {
            for (int j = 1; j + i <= bound; j *= y) {
                set.add(i + j);
                
if (y == 1) break;
            }
            if (x == 1) break;
        }
        List<Integer> res = new ArrayList<>(set);
        Collections.sort(res);
        return res;
    }
}
