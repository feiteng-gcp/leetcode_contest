class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> set = new HashSet<>();
        if (x == 1 && y == 1) {
            if (2 <= bound)
                set.add(2);
        } else if (x == 1) {
            for (int j = 0; (int)Math.pow(y, j) + 1 <= bound; j++)
                set.add(1 + (int)Math.pow(y, j));
        } else if (y == 1) {
            for (int i = 0; (int)Math.pow(x, i) + 1 <= bound; i++)
                set.add(1 + (int)Math.pow(x, i));
        } else {
            
for (int i = 0; (int)Math.pow(x, i) < bound ; i++) {
                int n1 = (int)Math.pow(x, i);
                for (int j = 0; n1 + (int)Math.pow(y, j) <= bound; j++) 
                    set.add(n1 + (int)Math.pow(y, j));
            }
        }
        
List<Integer> res = new LinkedList<>();
        for (int s : set)
            res.add(s);
        return res;
    }
}
