class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        List<Integer> res = new ArrayList();
        if (x == 1 && y == 1) {
            if (bound <= 1) {
                return res;
            } else {
                res.add(2);
                return res;
            }
        }
        if (x == 1 || y == 1) {
            int multi = (x == 1 ? y : x);
            for (int i = 1; i <= bound; i *= multi) {
                if (i + 1 <= bound) {
                    res.add(i + 1);
                } else {
                    break;
                }
            }
            return res;
        }
        Set<Integer> set = new HashSet();
        for (int i = 1; i <= bound; i *= x) {
            for (int j = 1; j <= bound; j *= y) {
                int sum = i + j;
                if (sum <= bound) {
                    set.add(sum);
                } else {
                    break;
                }
            }
        }
        return new ArrayList(set);
    }
}
