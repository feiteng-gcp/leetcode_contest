class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        List<Integer> xp = get(x, bound);
        List<Integer> yp = get(y, bound);
        Set<Integer> ret = new HashSet<>();
        for (int i = 0; i < xp.size(); i++) {
            for (int j = 0; j < yp.size(); j++) {
                int sum = xp.get(i) + yp.get(j);
                if (sum > bound) {
                    break;
                } else {
                    ret.add(sum);
                }
            }
        }
        return new ArrayList<>(ret);
    }
    
    List<Integer> get(int x, int bound) {
        List<Integer> ret = new ArrayList<>();
        ret.add(1);
        if (x == 1) 
            return ret;
        for (int i = x; i <= bound; i *= x) {
            ret.add(i);
        }
        return ret;
    }
}
