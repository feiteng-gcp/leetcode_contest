class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
    	Set
<Integer> ans = new HashSet<>();
    	
    	int xx = 1;
    	for(int i = 0; i < 100000; i++) {
    		int yy = 1;
    		for(int j = 0; j < 100000; j++) {
    			int next = xx + yy;
    			if(next <= bound) ans.add(next);
    			else break;
    			if(y == 1) break;

    			yy *= y;
    		}
    		if(x == 1 || xx > bound) break;
    		xx *= x;
    	}
    	List<Integer> aans = new ArrayList<>();
    	aans.addAll(ans);
    	return aans;
        
    }
}
