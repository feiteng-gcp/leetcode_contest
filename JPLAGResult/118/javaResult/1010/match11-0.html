class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        List<Integer> xPow = new ArrayList<>();
        List<Integer> yPow = new ArrayList<>();
        xPow.add(1);
        yPow.add(1);
        if (x != 1) {
            int num = x;
            while (num < bound) {
                xPow.add(num);
                num *= x;
            }
        }
        if (y != 1) {
            int num = y;
            while (num < bound) {
                yPow.add(num);
                num *= y;
            }
        }
        Set<Integer> set = new HashSet<>();
        for (int i : xPow) {
            for (int j : yPow) {
                if (i + j <= bound) {
                    set.add(i + j);
                } else {
                    break;
                }
            }
        }
        return new ArrayList<>(set);
    }
}
