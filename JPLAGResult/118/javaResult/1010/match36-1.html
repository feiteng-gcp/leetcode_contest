class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        if(x==1&&y==1){
            if(x+y<=bound){
                List<Integer> a = new ArrayList<>();
                a.add(2);
                return a;
            }else{
                return new ArrayList<>();
                
            }
        }
        
        if(x==1){
            List<Integer> a = new ArrayList<>();
            for(int i=0;(int)Math.pow(y,i)<bound;i++){
                a.add((int)Math.pow(y,i)+1);   
                
            }
            return a;
            
        }
        
        if(y==1){
            List<Integer> a = new ArrayList<>();
            for(int i=0;(int)Math.pow(x,i)<bound;i++){
                a.add((int)Math.pow(x,i)+1);   
                
            }
            return a;
            
        }
        
        
        
        
HashSet<Integer> st  =new HashSet<>();
        for(int i=0;(int)Math.pow(x,i)<bound;i++){
            int cur = (int)Math.pow(x,i);
            for(int j=0;cur+(int)Math.pow(y,j)<=bound;j++){
                st.add(cur+(int)Math.pow(y,j));
            }
        }
        List<Integer> a = new ArrayList<>();
        for(int aa:st){
            a.add(aa);
        }
        return a;
    }
}
