class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        
        Set<Integer> res = new HashSet();
        
        int valuex = 1;         
        for (int i=0; i<21; i++){            
            int valuey = 1;
            for (int j=0; j<21; j++){
                if (valuex+valuey <= bound) res.add(valuex+valuey);
                valuey *= y;
                if (valuey > bound) break;
            }
            
            valuex *= x;
            if (valuex > bound) break;
        }
        
        List<Integer> answer = new ArrayList();
        answer.addAll(res);
        return answer;
    }
}
