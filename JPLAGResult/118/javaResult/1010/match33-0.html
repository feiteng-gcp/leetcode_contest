class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i <= 32; i++) {
            for (int j = 0; j <= 32; j++) {
                double v = Math.pow(x, i) + Math.pow(y, j);
                if (v <= bound) {
                    set.add((int)v);
                }
            }
        }
        return new ArrayList<Integer>(set);
    }
}
