class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        HashSet<Integer> ans = new HashSet<>();
        
        for(int i=0;Math.pow(x,i)<bound+1&&i<25;i++){
            int now = (int) Math.pow(x,i);
            for(int j=0;Math.pow(y,j)+now<bound+1&&j<25;j++){
                ans.add(now+(int)Math.pow(y,j));
            }
        }
        LinkedList<Integer> res = new LinkedList<>();
        for(int w:ans) res.add(w);
        return res;
    }
}
