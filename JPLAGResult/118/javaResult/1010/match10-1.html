class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        ArrayList<Integer> xval = new ArrayList<>();
        
xval.add(1);
        if (x > 1) {
            int xi = x;
            while (xi <= bound) {
                xval.add(xi);
                xi *= x;
            }
        }
        ArrayList<Integer> yval = new ArrayList<>();
        yval.add(1);
        
if (y > 1) {
            int yi = y;
            while (yi <= bound) {
                yval.add(yi);
                yi *= y;
            }
        }
        HashSet<Integer> good = new HashSet<>();
        for (int xi : xval) {
            for (int yi : yval) {
                if (xi + yi <= bound) {
                    good.add(xi + yi);
                
}
            }
        }
        return new ArrayList<Integer>(good);
    }
}
