class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> res = new HashSet<>();
        int I = log(bound, x), J = 
log(bound, y);
        for (int i = 0; i <= I; i++) {
            for (int j = 0; j <= J; j++) {
                int sum = (int) Math.pow(x, i) + (int) Math.pow(y, j);
                if (sum <= bound) res.add(sum)
;
            }
        }
        return new ArrayList(res);
    }
    
    private int log(int n, int b) {
        if (b == 1) return n;
        int e = 0, c = 1;
        while (c < n) {
            c *= b;
            e++;
        }
        return e;
    }
}
