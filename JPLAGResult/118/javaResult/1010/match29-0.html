class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        List<Integer> ans = new ArrayList<> ();
        List<Integer> xb = new ArrayList<> ();
        List<Integer> yb = new ArrayList<> ();
        Set<Integer> s = new HashSet<> ();
        
        int temp = 1;
        while (temp < bound) {
            xb.add(temp);
            int temp1 = temp * x;
            if (temp1 == temp)
                break;
            else
                temp = temp1;
        }
        
        temp = 1;
        while (temp < bound) {
            yb.add(temp);
            int temp1 = temp * y;
            
            if (temp1 == temp)
                break;
            else
                temp = temp1;
        }
        
        for (int i = 0; i < xb.size(); ++i) {
            for (int k = 0; k < yb.size(); ++k) {
                int sum = xb.get(i) + yb.get(k);
                
                if (sum <= bound && !s.contains(sum)) {
                    s.add(sum);
                    ans.add(sum);
                }
            }
        }
        
        return ans;
    }
}
