class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        int x_bound = x==1?bound:(int)(Math.log(bound)/Math.log(x))+1;
        int y_bound = y==1?bound:(int)(Math.log(bound)/Math.log(y))+1;
        
Set<Integer> st = new HashSet<Integer>();
        for(int i=0; i<=x_bound; i++){
            for(int j=0; j<=y_bound; j++){
                int temp = (int)(Math.pow(x, i)+Math.pow(y, j));
                if(temp<=bound)
                    st.add(temp)
;
            }
        }
        List<Integer> res = new ArrayList<Integer>();
        for(int key: st)
            res.add(key);
        return res;
    }
}
