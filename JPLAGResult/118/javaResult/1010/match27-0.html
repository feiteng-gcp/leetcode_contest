class Solution {
  public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> set = new HashSet<>();

        long xx = 1;
        long yy = 1;
        for (int i = 0; i < 50; i++) {

            yy = 1;
            for (int j = 0; j < 50; j++) {
                if (xx + yy <= bound) {
                    set.add((int) (xx + yy));
                }

                yy *= y;
                if (yy > bound) break;

            }


            xx *= x;

            if (xx > bound) break;
        }

        return new ArrayList<>(set);
    }
}
