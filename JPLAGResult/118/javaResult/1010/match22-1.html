class Solution {
    public List<Integer> powerfulIntegers(int x, int y, int bound) {
        Set<Integer> resultSet = new HashSet<>();
        List<Integer> xRes = new ArrayList<>();
        List<Integer> yRes = new ArrayList<>();
        int curr = 1;
        while(curr < bound) {
            
xRes.add(curr);
            curr *= x;
            if(curr == 1) break;
        }
        curr = 1;
        while(curr < bound) {
            yRes.add(curr);
            curr *= y;
            if(y == 1) break;
        }
        
for(Integer inX : xRes) {
            for(Integer inY : yRes) {
                if(inX + inY <= bound) resultSet.add(inX + inY);
            }
        }
        List<Integer> result = new ArrayList<>(resultSet);
        return result;
    }
}
