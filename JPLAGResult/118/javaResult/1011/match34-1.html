/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
        public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> res = new ArrayList<>();
        if (root == null) {
            if (voyage.length == 0) {
                return res;
            } else {
                res.add(-1);
                return res;
            }
        }
        if (helper(root, voyage, 0, res) == voyage.length) {
            return res;
        } else {
            res = 
new ArrayList<>();
            res.add(-1);
            return res;
        }
    }

    public int helper(TreeNode node, int[] v, int i, List<Integer> res) {
        if (i >= v.length) return 100;
        if (node.val != v[i]) return 100;
        if (node.right == null && node.left == null) return 1;
        if (node.left == null || node.right == null) {
            if (node.left != null) {
                return 1+ helper(node.left, v, i+1, res);
            } else {
                return 1+helper(node.right, v, i+1, res);
            }
        }
        if (i+1 < v.length && node.left.val == v[i+1]) {
            int leftnum = helper(node.left, v, i+1, res);
            return 1 +leftnum+ helper(node.right, v, i+1+leftnum, res);
        } else {
            res.add(node.val);
            TreeNode temp = node.left;
            node.left = node.right;
            node.right = temp;
            int leftnum = helper(node.left, v, i+1, res);
            return 1 +leftnum+ helper(node.right, v, i+1+leftnum, res);
        }
    }

}
