/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Set<Integer> visited = new HashSet<>();
    List<Integer> list = new ArrayList<>();
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        if (travel(root, voyage, 0))
            return list;
        else{
            list = new ArrayList<>();
            list.add(-1);
            return list;
        }
    }
    
    private boolean travel(TreeNode node, int[] voyage, int idx){
        if (node == null && idx == -1)
            return true;
        else if (node == null || idx == -1)
            return false;
        
        
        if (visited.contains(idx))
            return false;
        
        if (node.val != voyage[idx])
            return false;
        
        int lidx = -1;
        int ridx = -1;
        for (int i = idx+1; i < voyage.length; i++){
            if (node.left!=null && node.left.val == voyage[i])
                lidx = i;
            if (node.right!=null && node.right.val == voyage[i])
                ridx = i;
        }
        //System.out.println(node.val + " " + lidx + " " + ridx);
        //if one of the next is lying on idx+1
        //if no need to swap
        if (lidx == idx+1){
            visited.add(idx);
            if (travel(node.left, voyage, lidx) && travel(node.right, voyage, ridx))
                return true;
            visited.remove(idx);
            return false;
        }
        else if (ridx == idx+1){
            visited.add(idx);
            if (lidx!=-1)
                list.add(node.val);
            if (travel(node.left, voyage, lidx) && travel(node.right, voyage, ridx))
                return true;
            visited.remove(idx);
            if (lidx!=-1)
                list.remove(list.size()-1);
            return false;   
        }
        else if ((lidx == -1 && ridx == -1)&&(node.left == null && node.right == null))
                return true;
        else
            return false;
    }
}
