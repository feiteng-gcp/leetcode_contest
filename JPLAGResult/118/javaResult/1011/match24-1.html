/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        return fmv(root, voyage, 0, voyage.length);
    }
    public List<Integer> fmv(TreeNode root, 
int[] voyage, int start, int end) {
        ArrayList<Integer> ans = new ArrayList<Integer>();
        if (root==null)
            return ans;
        if (root.val!=voyage[start]) {
            ans.add(-1);
            return ans;
        }
        if (root.left == null)
            return fmv(root.right, voyage, start+1, end);
        if (root.right == null)
            return fmv(root.left, voyage, start+1, end);
        int cl = count(root.left);
        int cr = count(root.right);
        if (voyage[start+1]==root.left.val) {
            List<Integer> ll = fmv(root.left, voyage, start+1, start+1+cl);
            List<Integer> rl = fmv(root.right, voyage, start+1+cl, end);
            if (ll.size()>0&&ll.get(0)==-1) {
                ans.add(-1);
                return ans;
            }
            if (rl.size()>0&&rl.get(0)==-1) {
                ans.add(-1);
                return ans;
            }
            ans.addAll(ll);
            ans.addAll(rl);
            return ans;
        } else {
            List<Integer> rl = fmv(root.right, voyage, start+1, start+1+cr);
            List<Integer> ll = fmv(root.left, voyage, start+1+cr, end);
            if (ll.size()>0&&ll.get(0)==-1) {
                ans.add(-1);
                return ans;
            }
            if (rl.size()>0&&rl.get(0)==-1) {
                ans.add(-1);
                return ans;
            }
            ans.addAll(ll);
            ans.addAll(rl);
            ans.add(root.val);
            
return ans;
        }
    }
    public int count(TreeNode root) {
        if (root==null)
            return 0;
        return count(root.left)+count(root.right)+1;
    }
}
