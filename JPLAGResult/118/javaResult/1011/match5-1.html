	class Solution {
		int p = 0;
		List<Integer> fed;
		
	    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
	    	p = 0;
	    	fed = new ArrayList<>();
	        if(dfs(root, voyage)){
	        	return fed;
	        }else{
	        	fed.clear();
	        	fed.add(-1);
	        	return fed;
	        }
	    }
	    
	    boolean dfs(TreeNode cur, int[] voyage)
	    {
	    	if(cur == 
null)return true;
	    	if(voyage[p] != cur.val)return false;
	    	p++;
	    	if(cur.left == null){
	    		return dfs(cur.right, voyage);
	    	}
	    	if(cur.right == null){
	    		return dfs(cur.left, voyage);
	    	}
	    	
	    	if(voyage[p] == cur.left.val){
	    		boolean res = dfs(cur.left, voyage);
	    		if(!res)return false;
	    		return dfs(cur.right, voyage);
	    	}else{

	    		fed.add(cur.val);
	    		boolean res = dfs(cur.right, voyage);
	    		if(!res)return false;
	    		return dfs(cur.left, voyage);
	    	}
	    }
	}
