/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> res = new ArrayList<Integer>();
        if(helper(res, root, voyage, new TreeNode(0))){
            return res;
        }else{
            res.clear();
            res.add(-1);
            return res;
        }
    }
    
    boolean helper(List<Integer> res, TreeNode node, int[] voyage, TreeNode cur){
        if(node == null) return true;
        if(node.val != voyage[cur.val]){
            return false;
        }
        if(node.left!=null&&node.right!=null){
            if(node.left.val!=voyage[cur.val+1]){
                res.add(node.val);
                TreeNode temp = node.left;
                node.left = node.right;
                node.right = temp;
            }
            cur.val++;
            boolean left = helper(res, node.left, voyage, cur);
            cur.val++;
            boolean right = helper(res, node.right, voyage, cur);
            return left&&right;
        }else if(node.left != null && node.right == null){
            cur.val++;
            return helper(res, node.left, voyage, cur);
        }else if(node.left == null && node.right != null){
            cur.val++;
            return helper(res, node.right, voyage, cur);
        }else{
            return true;
        }
    }
}
