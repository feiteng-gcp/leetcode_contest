/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private int i = 0;
    private boolean b = true;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        // voyage = new int[]{1,2,3};
        // root = new TreeNode(1);
        // root.left = new TreeNode(2);
        // root.right = new TreeNode(3);
        List<Integer> res = new ArrayList<>();
        if (root == null) return res;
        if (root.val != voyage[0]) {
            res.add(-1);
            return res;
        }
        flipMatchVoyage(root, voyage, res);
        if (!b) {
            res = new ArrayList<>();
            res.add(-1);
        
}
        //System.out.println(b);
        return res;
    }
    private boolean flipMatchVoyage(TreeNode nd, int[] voyage, List<Integer> res) {
        if (!b) return false;
        if (nd == null) return true;
        if (nd.val != voyage[i]) return false;
        i++;
        if (flipMatchVoyage(nd.left, voyage, res)) {
            if (flipMatchVoyage(nd.right, voyage, res)) return true;
            b = false;
            return false;
        }
        res.add(nd.val);
        if (flipMatchVoyage(nd.right, voyage, res)) {
            if (flipMatchVoyage(nd.left, voyage, res)) return true;
        }
        b = false;
        return false;
    }
}
