/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private int cur;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> list = new ArrayList<Integer>();
        cur = 0;
        if(dfs(root, voyage, list)) return list;    
        return Arrays.asList(-1);
    
}
    
    private boolean dfs(TreeNode node, int[] voyage, List<Integer> list){
        if(node == null) return true;
        if(cur == voyage.length) return false;
        if(node.val != voyage[cur]) return false;
        if(node.right != null && node.right.val == voyage[cur+1] && node.left != null){
            list.add(node.val);
            TreeNode left = node.left;
            TreeNode right = node.right;
            node.left = right;
            node.right = left;
        }
        cur++;
        boolean resLeft = dfs(node.left, voyage, list);
        if(!resLeft) return false;
        return dfs(node.right, voyage, list);
    }
}
