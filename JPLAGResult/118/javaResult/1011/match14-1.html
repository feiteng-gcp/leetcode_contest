/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private List<Integer> ans;
    
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        int n = getCount(root);
        if (n != voyage.length) {
            return Arrays.asList(-1);
        }
        ans = new ArrayList<>();
        if (flip(root, 0, voyage) == -1) {
            return 
Arrays.asList(-1);
        }
        return ans;
    }
    
    private int flip(TreeNode node, int k, int[] voyage) {
        if (node == null) {
            return 0;
        }
        if (node.val != voyage[k]) {
            return -1;
        }
        int left = flip(node.left, k + 1, voyage);
        if (left == -1) {
            ans.add(node.val);
            int right = flip(node.right, k + 1, voyage);
            if (right == -1) {
                return -1;
            }
            left = flip(node.left, k + 1 + right, voyage);
            if (left == -1) {
                return -1;
            }
            return left + right + 1;
        } else {
            int right = flip(node.right, k + 1 + left, voyage);
            if (right == -1) {
                return -1;
            }
            return left + right + 1;
        }
    }
    
    private int getCount(TreeNode node) {
        if (node == null) {
            return 0;
        } else {
            return 1 + getCount(node.left) + getCount(node.right);
        }
    }
}
