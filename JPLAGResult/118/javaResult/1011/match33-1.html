/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int index = 0;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> res = new ArrayList<>();
        dfs(root,null,voyage,res);
       // System.out.println(index);
        if(index == voyage.length) return res;
        else {
            List<Integer> temp = new ArrayList<>();
            temp.add(-1);
            return temp;
        }
    
}
    
    private void dfs(TreeNode node, TreeNode parent, int[] voyage,List<Integer> swap) {
        if(node == null) return;
        if(index >= voyage.length || node.val == voyage[index]) {
            index++;
        } else {
            if(parent == null || parent.right == null || parent.right.val != voyage[index] || node == parent.right) {
                index = voyage.length+1;
                return;
            } else {
                swap.add(parent.val);
                parent.left = parent.right;
                parent.right = node;
                index++;
                node = parent.left;
            }
        }
        dfs(node.left,node,voyage,swap);
        dfs(node.right,node,voyage,swap);
    }
}
