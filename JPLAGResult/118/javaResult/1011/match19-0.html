/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    private boolean isGood = true;
    private int index = 0;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        isGood = true;
        index = 0;
        
        List<Integer> flipped = new ArrayList();
        visit(root, voyage, flipped);
        if (!isGood){
            flipped.clear();
            flipped.add(-1);
        }
        
        return flipped;
    }    
    
    private void visit(TreeNode node, int[] voyage, List<Integer> flipped){
        if (node == null) return;
        
        System.out.println(node.val + " " + index + " " + voyage[index]);
        
        if (node.val != voyage[index]){ 
            isGood = false;
            return;
        }
        
        if (node.left != null && node.left.val != voyage[index+1]){ 
            //swap
            TreeNode temp = node.left;
            node.left = node.right;
            node.right = temp;
            
            flipped.add(node.val);
        } 
        
        index++;
        
        visit(node.left, voyage, flipped);
        visit(node.right, voyage, flipped);
    }
}
