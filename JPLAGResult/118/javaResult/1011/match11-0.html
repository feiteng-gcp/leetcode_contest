/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int index = 0;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        int N = voyage.length;
        List<Integer> res = new ArrayList();
        if (!helper(root, voyage, res)) {
            res.clear();
            res.add(-1);
        }
        return res;
    }
    
    private boolean helper(TreeNode root, int[] voyage, List<Integer> res) {
        if (index == voyage.length) {
            return true;
        }
        if (root == null) {
            return true;
        }
        if (root.val != voyage[index]) {
            return false;
        }
        index++;
        if (root.left != null && root.right != null) {
            if (root.left.val != voyage[index]) {
                res.add(root.val);
                TreeNode temp = root.left;
                root.left = root.right;
                root.right = temp;
            }
        }
        if (helper(root.left, voyage, res) && helper(root.right, voyage, res)) {
            return true;
        } else {
            return false;
        }
    }
}
