/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> list = new LinkedList<>();
        preOrder(root, voyage, list);
        return list;
    }
    int [] pre = new int[1];
    public void preOrder(TreeNode node, int[] voyage, List<Integer> list){
        if(node==null){ return;
        }
            
        if(node.val!=voyage[pre[0]]){
            list.clear();
            list.add(-1);
            return;
        }
        if(node.left!=null && node.left.val!=voyage[pre[0]+1]){
            list.add(voyage[pre[0]]);
            TreeNode temp = node.left;
            node.left = node.right;
            node.right = temp;
        }
        if(node.left!=null){
            pre[0] ++;
            preOrder(node.left, voyage, list);
        }
        if(node.right!=null){
            pre[0] ++;
            preOrder(node.right, voyage, list);      
        }
        
    }
}
