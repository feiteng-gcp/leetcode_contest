/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < voyage.length; i++) {
            map.put(voyage[i], i);
        }
        List<Integer> list = new ArrayList<>();
        if (dfs(list, root, voyage, map, 0, voyage.length - 1)) {
            return list;
        }
        list = new ArrayList<>();
        list.add(-1);
        return list;
    }
    
    private boolean dfs(List<Integer> list, TreeNode root, int[] voyage, Map<Integer, Integer> map, int left, int right) {
        if (root == null || left > right) {
            return (root == null && left > right) ? true : false;
        }
        if (root.val == voyage[left]) {
            
if (root.left == null && root.right != null) {
                return dfs(list, root.right, voyage, map, left + 1, right);
            }
            if (root.right == null && root.left != null) {
                return dfs(list, root.left, voyage, map, left + 1, right);
            }
            if (root.left == null && root.right == null) {
                return left == right;
            }
            if (left + 1 >= voyage.length) {
                return false;
            }
            int pos1 = map.get(root.left.val);
            int pos2 = map.get(root.right.val);
            if (voyage[left + 1] == root.left.val) {
                return dfs(list, root.left, voyage, map, left + 1, pos2 - 1)
                    && dfs(list, root.right, voyage, map, pos2, right);
            } else if (voyage[left + 1] == root.right.val) {
                list.add(root.val);
                
return dfs(list, root.right, voyage, map, left + 1, pos1 - 1)
                    && dfs(list, root.left, voyage, map, pos1, right);
            } else {
                return false;
            }
        }
        return false;
    }
}
