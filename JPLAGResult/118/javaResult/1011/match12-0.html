/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        Map<Integer, Integer> indexMap = new HashMap();
        for(int i = 0, until = voyage.length; i < until; i++)
        {
            indexMap.put(
                voyage[i],
                (int)i
            );
        }
        
        List<Integer> result = new ArrayList();
        if(!rotate(root, indexMap, result))
        {
            return Arrays.asList(-1);
        }
        return result;
    }
    
    public static boolean rotate(TreeNode root, Map<Integer, Integer> indexMap, List<Integer> result)
    {
        if(root.left != null)
        {
            if(indexMap.get(root.val) > indexMap.get(root.left.val))
            {
                return false;
            }
            if(!rotate(root.left, indexMap, result))
            {
                return false;
            }
        }
        if(root.right != null)
        {
            if(indexMap.get(root.val) > indexMap.get(root.right.val))
            {
                return false;
            }
            if(!rotate(root.right, indexMap, result))
            {
                return false;
            }
        }
        
        if(root.left != null && root.right != null)
        {
            if(indexMap.get(root.left.val) > indexMap.get(root.right.val))
            {
                result.add(root.val);
            }
        }
        
        return true;
    }
}
