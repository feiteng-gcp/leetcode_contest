/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int i=0;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> res = new ArrayList<Integer>();
        if(dfs(root, voyage,res)) return res;
        res = new ArrayList<Integer>();
        res.add(-1);
        
return res;
    }
    public boolean dfs(TreeNode root, int[] voyage, List<Integer> res){
        if(root==null){
            return true;
        }
        // System.out.println(root.val+","+i+","+voyage[i]);
        if(root.val!=voyage[i]) return false;
        i++;
        if(dfs(root.left, voyage, res)&&dfs(root.right, voyage, res)) return true;
        // System.out.println(root.val);
        res.add(root.val);
        if(dfs(root.right, voyage, res)&&
dfs(root.left, voyage, res)) return true;
        // res.add(i);
        res.remove(res.size()-1);
        return false;
    }
}
