/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> cannot = new LinkedList();
        cannot.add(-1);
       List<Integer> ans =  flipMatchVoyage( root, voyage, 0);
        if( ans == null)
            return cannot;
        return ans;
        
    }
    
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage, int s) {
        
        List<Integer> ans = new LinkedList();
        if( root == null )
            return ans;
        // System.out.println(root.val);
        
        if(root.val == voyage[s]){
            List<Integer> left = flipMatchVoyage(root.left, voyage, s+1);
            
            List<Integer> right = null;
            if( left != null){
                right = flipMatchVoyage(root.right, voyage, s+ getSize(root.left) +1);
            }  else {
                // System.out.println("Here"+ root.val);
                
                ans.add(root.val);
                left = flipMatchVoyage(root.right, voyage, s + 1);
                // print(left);
                if( left != null)
                    right = flipMatchVoyage(root.left, voyage, s + getSize(root.right) +1);
                // print(left);
            }
            if(left == null || right == null)
                return null;
            
            ans.addAll(left);
            ans.addAll(right);
            return ans;
        } else {
            return null;
        }
    }
    
    public int getSize(TreeNode root){
        if( root == null)
            return 0;
        return getSize(root.left) + getSize(root.right) + 1;
    
    }
    
    public void print(List<Integer> l){
        if( l == null){
            System.out.println("null");
            return;
        }
        for( int i :l)
            System.out.print(i + " ");
        System.out.println("-----");
    }
    
}
