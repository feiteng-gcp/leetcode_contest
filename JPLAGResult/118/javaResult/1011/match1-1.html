/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private int[] voyage;
    private List<Integer> res;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        res = new ArrayList<>();
        this.voyage = voyage;
        flip(root, 0);
        return res;
    }
    
    private int flip(
TreeNode node, int pos) {
        if(node == null) {
            return pos;
        }
        if(node.val != voyage[pos]) {
            res.clear();
            res.add(-1);
            return -1;
        }
        pos++;
        if(node.left != null && node.right != null && voyage[pos] == node.right.val) {
            res.add(node.val);
            TreeNode tmp = node.left;
            node.left = node.right;
            node.right = tmp;
        }
        pos = flip(node.left, pos);
        if(pos == -1) {
            return pos;
        } else {
            return flip(node.right, pos);
        }
    }
}
