/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int index;
    
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> res = new ArrayList<>();
        index = 0;
        
if (!helper(root, voyage, res)) {
            res.clear();
            res.add(-1);
        }
        return res;
    }
    
    private boolean helper(
TreeNode root, int[] voyage, List<Integer> res) {
        if (root == null) {
            return true;
        }
        if (root.val != voyage[index]) {
            return false;
        }
        index++;
        if (helper(root.left, voyage, res)) {
            if (helper(root.right, voyage, res)) {
                return true;
            }
            return false;
        } else {
            res.add(root.val);
            if (!helper(root.right, voyage, res)) {
                return false;
            }
            if (!helper(root.left, voyage, res)) {
                return false;
            }
        }
        return true;
    }
}
