/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int p = 0;
    
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> res = new ArrayList<>();
        traverse(root, root, voyage, res);
        if (!isSameTree(root, voyage)) {
            res.clear();
            res.add(-1);
        }
        return res;
    }
    
    private boolean isReversed(TreeNode node, int[] voyage) {
        if (node == null) return false;
        if (node.left == null || node.right == null) return false;
        int leftIndex = findIndex(voyage, node.left.val);
        int rightIndex = findIndex(voyage, node.right.val);
        return leftIndex > rightIndex;
    }
    
    private void traverse(TreeNode root, TreeNode node, int[] voyage, List<Integer> list) {
        if (isSameTree(root, voyage)) {
            return;
        }
        if (node == null) {
            return;
        }
        if (isReversed(node, voyage)) {
            list.add(node.val);
            TreeNode tmp = node.left;
            node.left = node.right;
            node.right = tmp;
        }
        traverse(root, node.left, voyage, list);
        traverse(root, node.right, voyage, list);
    }
    
    private boolean isSameTreeHelper(TreeNode root, int[] voyage) {
        if (root == null) return true;
        if (root.val != voyage[p]) return false;
        p++;
        return isSameTreeHelper(root.left, voyage) && isSameTreeHelper(root.right, voyage);
    }
    
    private boolean isSameTree(TreeNode root, int[] voyage) {
        p = 0;
        return isSameTreeHelper(root, voyage);
    }
    
    private int findIndex(int[] arr, int val) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == val) return i;
        }
        return -1;
    }
    
}
