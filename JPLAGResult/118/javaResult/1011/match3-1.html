/**
 * Definition for a binary tree node.
 */
public class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode(int x) { val = x; }
}

class Solution {
    
    int x;
    List<Integer> ans;
    
    List<Integer> getNone() {
        List<Integer> non = new ArrayList<>();
        
non.add(-1);
        return non;
    }
    
    boolean dfs(TreeNode node, int[] voyage) {
        if (node == null) return true;
        if (node.val == voyage[x]) {
            x++;
        } else {
            return false;
        }
        boolean b = dfs(node.left, voyage);
        if (b) {
            return dfs(node.right, voyage);
        } else {
            ans.add(node.val);
            b = dfs(node.right, voyage);
            if (b) {
                return dfs(node.left, voyage);
            } else {
                return false;
            }
        }
    }
    
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        if (root == null || root.val != voyage[0]) {
            return getNone();
        }
        ans = new ArrayList<>();
        if (dfs(root, voyage)) 
{
            return ans;
        } else {
            return getNone();
        }
    }
}
