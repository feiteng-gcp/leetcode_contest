/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<Integer> res = new ArrayList<>();
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        boolean valid = tra(root, voyage, 0, voyage.length-1);
        if (!valid) {
            List<Integer> ret = new ArrayList<>();
            ret.add(-1);
            return ret;
        }
        return res;
    }
    boolean tra(TreeNode r, int[] v, int s, int e) {
        if (r == null && s > e) return true;
        if (r == null || s > e) return false;
        if (r.val != v[s]) return false;
        if (r.left == null && r.right == null) return s == e;
        if (r.left == null) {
            return tra(r.right, v, s+1, e);
        }
        if (r.right == null) {
            return tra(r.left, v, s+1, e);
        }
        //if (idx + 2 <= v.length) return false;
        if (r.left.val == v[s+1]) {
            int rs = search(v, s+2, e, r.right.val);
            if (rs == -1) return false;
            return tra(r.left, v, s+1, rs-1) && tra(r.right, v, rs, e);
        } else if (r.right.val == v[s+1]) {
            res.add(r.val);
            int ls = search(v, s+2, e, r.left.val);
            if (ls == -1) return false;
            return tra(r.right, v, s+1, ls-1) && tra(r.left, v, ls, e);
        } else {
            return false;
        }
    }
    int search(int[] v, int s, int e, int t) {
        int res = -1;
        for (int i = s; i <= e; i++) {
            if (v[i] == t) {
                res = i;
                break;
            }
        }
        return res;
    }
}
