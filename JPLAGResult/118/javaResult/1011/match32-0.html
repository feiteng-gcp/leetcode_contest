/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int[] v;
    List<Integer> ans;
    
    
    boolean checkRoot(TreeNode root, int i, int j) {
        if (root.left != null) {
            if (j <= i) return false;
            if (root.left.val == v[i+1]) {
                if (root.right == null) return check(root.left, i + 1, j);
                int k = i + 2;
                for (; k <= j; k++)
                    if (v[k] == root.right.val) break;
                if (k > j) return false;
                return check(root.left, i + 1, k-1) && check(root.right, k, j);
            } else {
                if (root.right == null) return false;
                if (root.right.val != v[i+1]) return false;
                int k = i + 2;
                for (; k <= j; k++)
                    if (v[k] == root.left.val) break;
                if (k > j) return false;
                ans.add(root.val);
                return check(root.right, i + 1, k-1) && check(root.left, k, j);
            }
        }
        return true;
    }
    
    boolean check(TreeNode root, int i, int j) {
        if (root == null) return true;
        if (root.val != v[i]) return false;
        if (root.left != null) return checkRoot(root, i, j);
        TreeNode t = root.left;
        root.left = root.right;
        root.right = t;
        return checkRoot(root, i, j);
    }
    
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        ans = new ArrayList<>();
        v = voyage;
        if (check(root, 0, v.length-1)) return ans;
        ans.clear();
        ans.add(-1);
        return ans;
    }
}
