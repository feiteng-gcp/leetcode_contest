/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] v) {
        List<Integer> a = new ArrayList<>();
        Stack<TreeNode> st = new Stack<>();
        st.push(root);
        int y = 0;
        while(!st.empty()){
            TreeNode cur = st.pop();
            if(cur.val!=v[y]){
                List<Integer> e = new ArrayList<>();
                e.add(-1);
                return e;
            }
            y++;
            if(cur.left!=null){
                if(cur.left.val==v[y]){
                    if(cur.right!=null){
                        st.push(cur.right);
                    }
                    st.push(cur.left);
                }else{
                    a.add(cur.val);
                    st.push(cur.left);
                    if(cur.right!=null){
                        st.push(cur.right);
                    }
                    
                    
                }
                
            }else if(cur.right!=null){
                st.push(cur.right);   
            }
            
        }
        if(y==v.length){
            return a;
        }else{
            List<Integer> e = new ArrayList<>();
                e.add(-1);
                return e; 
        }
    }
}
