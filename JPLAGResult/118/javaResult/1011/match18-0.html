/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int p = 0;
    List<Integer> re;
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        p = 0;
        re = new ArrayList<>();
        if(!check(root, voyage)) {
            re = new ArrayList<>();
            re.add(-1);
        }
        return re;
    }
    boolean check(TreeNode root, int[] voyage) {
        if(root.val != voyage[p]) return false;
        p++;
        if(root.left != null && root.right != null) {
            if(root.left.val != voyage[p]) {
                if(root.right.val != voyage[p]) return false;
                re.add(root.val);
                if(!check(root.right, voyage)) return false;
                if(!check(root.left, voyage)) return false;
            }
            else {
                if(!check(root.left, voyage)) return false;
                if(!check(root.right, voyage)) return false;
            }
            return true;
        }
        else if(root.left != null) {
            root.right = root.left;
        }
        if(root.right != null) {
            return check(root.right, voyage);
        }
        return true;
    }
}
