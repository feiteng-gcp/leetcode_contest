/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> result = new ArrayList<>();
        int returnVal = dfs(result, root, voyage, 0);
        if(returnVal == -1) {
            result.clear();
            result.add(-1);
        }
        return result;
    }
    
    public int dfs(List<Integer> result, TreeNode root, int[] voyage, int index) {
        if(root == null) return index - 1;
        if(root.val != voyage[index]) return -1;
        int leftReturnVal = index, rightReturnVal = index;
        if(root.left == null) {
            rightReturnVal = dfs(result, root.right, voyage, index + 1);
            return rightReturnVal;
        } else if(root.right == null) {
            leftReturnVal = dfs(result, root.left, voyage, index + 1);
            return leftReturnVal;
        } else if(root.left.val == voyage[index + 1]) {
            leftReturnVal = dfs(result, root.left, voyage, index + 1);
            if(leftReturnVal == -1) return -1;
            rightReturnVal = dfs(result, root.right, voyage, leftReturnVal + 1);
            return rightReturnVal;
        } else {
            result.add(root.val);
            rightReturnVal = dfs(result, root.right, voyage, index + 1);
            if(rightReturnVal == -1) return -1;
            leftReturnVal = dfs(result, root.left, voyage, rightReturnVal + 1);
            return leftReturnVal;
        }
    }
}
