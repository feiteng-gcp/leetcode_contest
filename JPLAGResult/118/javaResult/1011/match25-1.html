/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
List<Integer> ans = new ArrayList<>();

	public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
		if (solve(root, voyage, 0, voyage.length))
			return ans;
		else {
			ans.clear();
			ans.add(-1);
			return ans;
		}
	}


	boolean solve(TreeNode root, int[] voyage, int start, int end) {
		if (root == null) {
			return start == end;
		}
		if (start >= end) return false;
		if (root.val != voyage[start]) {
			return false;
		}
		int rindex = end;
		int lindex = start + 1;
		if (root.right != null) {
			for (int i = start; i
 < end; i++) {
				if(voyage[i] == root.right.val) {
					rindex = i;
					break;
				}
			}
		}
		if (root.left != null) {
			for (int i = start; i
 < end; i++) {
				if(voyage[i] == root.left.val) {
					lindex = i;
					break;
				}
			}
		}

		if (lindex == start + 1) {
			if (!solve(root.left, voyage, start + 1, rindex)) return false;
			if (!solve(root.right, voyage, rindex, end)) return false;
		} else {
			if (rindex != start + 1) return false;
			ans.add(root.val);
			if (!solve(root.left, voyage, lindex, end)) return false;
			if (!solve(root.right, voyage, rindex, lindex)) return false;
		}
		return true;
	}
}
