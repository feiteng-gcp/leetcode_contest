/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> ret = new ArrayList<>();
        
        if (root == null) return ret;
        
        int traversed = preorder(root, voyage, 0, ret);
        if (traversed != -1) return ret;
        
        ret.clear();
        ret.add(-1);
        return ret;
    }
    
    private int preorder(TreeNode node, int[] voyage, int idx, List<Integer> flipped) {
        if (node == null) return 0;
        if (node.val != voyage[idx]) return -1;
        if (node.right != null && node.left != null && node.left.val != voyage[idx + 1]) {
            TreeNode tmp = node.left;
            node.left = node.right;
            node.right = tmp;
            flipped.add(node.val);
        }
        int lTraversed = preorder(node.left, voyage, idx + 1, flipped);
        if (lTraversed == -1) return -1;
        int rTraversed = preorder(node.right, voyage, idx + 1 + lTraversed, flipped);
        if (rTraversed == -1) return -1;
        return lTraversed + rTraversed + 1;
    }
}
