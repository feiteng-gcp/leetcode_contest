/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> result = flipHelper(root, voyage, 0, voyage.length - 1);
        
        if (result == null) {
            result = new ArrayList<Integer>();
            result.add(-1);
        }
        
        return result;
    }
    

    public static List<Integer> flipHelper(TreeNode root, int[] voyage, int beg, int end) {
        if (root == null) {
            if (beg > end) {
                return new ArrayList<Integer>();
            }
            
            return null;
        }
        
        List<Integer> result = new ArrayList<>();
        // check root
        if (root.val != voyage[beg]) {
            return null;
        }
        
        if (root.left == null) {
            return flipHelper(root.right, voyage, beg + 1, end);
        }
        
        if (root.right == null) {
            return flipHelper(root.left, voyage, beg + 1, end);
        }
        
        // both left and right is not null
        if (beg + 1 > end) {
            return null;
        }
        
        if (root.left.val == voyage[beg + 1]) {
            // may not flip
            int index = -1;
            for (int i = beg + 2; i <= end; i++) {
                if (voyage[i] == root.right.val) {
                    index = i;
                    break;
                }
            }
            
            if (index == -1) {
                return null;
            }
            
            // [beg + 1, index - 1], [index, end]
            List<Integer> leftSide = flipHelper(root.left, voyage, beg + 1, index - 1);
            List<Integer> rightSide = flipHelper(root.right, voyage, index, end);
            if (leftSide == null || rightSide == null) {
                return null;
            }
            result.addAll(leftSide);
            result.addAll(rightSide);
            return result;
        } else if (root.right.val == voyage[beg + 1]) {
            // may flip
            int index = -1;
            for (int i = beg + 2; i <= end; i++) {
                if (voyage[i] == root.left.val) {
                    index = i;
                    break;
                }
            }
            
            if (index == -1) {
                return null;
            }
            
            // [beg + 1, index - 1], [index, end]
            List<Integer> rightSide = flipHelper(root.right, voyage, beg + 1, index - 1);
            List<Integer> leftSide = flipHelper(root.left, voyage, index, end);
            if (leftSide == null || rightSide == null) {
                return null;
            }
            result.add(root.val);
            result.addAll(leftSide);
            result.addAll(rightSide);
            return result;
        } else {
            return null;
        }
    }
}
