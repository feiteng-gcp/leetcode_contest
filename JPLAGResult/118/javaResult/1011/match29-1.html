/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        Map<Integer, Integer> map = new HashMap<>();
        for(int i=0; i<voyage.length; i++) {
            map.put(voyage[i], i);
        }
        List<Integer> res = new ArrayList<>();
        
helper(root, map, voyage, res);
        if(!find) {
            res.clear();
            res.add(-1);
        }
        return res;
    }
    boolean find = true;
    
void helper(TreeNode root, Map<Integer, Integer> map, int[] voyage, List<Integer> res) {
        if(root == null)
            return;
        Integer rtv = map.get(root.val);
        if(root.left != null) {
            Integer lv = map.get(root.left.val);
            if(rtv > lv) {
                find = false;
                return;
            }
        }
        if(root.right != null) {
            Integer rv = map.get(root.right.val);
            if(rtv > rv) {
                find = false;
                return;
            } 
        }
        if(root.left != null && root.right != null) {
            Integer lv = map.get(root.left.val);
            Integer rv = map.get(root.right.val);
            if(lv > rv) {
                res.add(root.val);
            }
        }
        helper(root.left, map, voyage, res);
        helper(root.right, map, voyage, res);
        
    }
}
