/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {

    private List<Integer> flips;
    
private List<Integer> traversal;
    private Map<Integer, Integer> voyageMap;

    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        this.flips = new ArrayList<>();
        this.traversal = new ArrayList<>();
        this.voyageMap = new HashMap<>();

        for (int i = 0; i < voyage.length; i++) {
            voyageMap.put(voyage[i], i);
        }
        flip(root);
        preorderTraverse(root);
        if (Arrays.stream(voyage).boxed().collect(Collectors.toList()).equals(this.traversal)) 
{
            return flips;
        } else {
            return Arrays.asList(-1);
        }
    }

    private void flip(TreeNode treeNode) {
        if (Objects.nonNull(treeNode)) {
            if (Objects.nonNull(treeNode.left) && Objects.nonNull(treeNode.right)
                    && voyageMap.get(treeNode.left.val) > voyageMap.get(treeNode.right.val)) {
                flips.add(treeNode.val);
                TreeNode temp = treeNode.left;
                treeNode.left = treeNode.right;
                treeNode.right = temp;
            }
            flip(treeNode.left);
            flip(treeNode.right);
        }
    }

    private void preorderTraverse(TreeNode treeNode) {
        if (Objects.nonNull(treeNode)) {
            traversal.add(treeNode.val);
            preorderTraverse(treeNode.left);
            preorderTraverse(treeNode.right);
        }
    }
}
