/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> flipMatchVoyage(TreeNode root, int[] voyage) {
        List<Integer> res = flip(root, voyage, 0, voyage.length - 1);
        return res;
    
}


    List<Integer> flip(TreeNode root, int[] voyage, int start, int end) {
        if (start > end) {
            List<Integer> res = new ArrayList<>();
            res.add(-1);
            return res;
        }
        if (root.val != voyage[start]) {
            List<Integer> res = new ArrayList<>();
            res.add(-1);
            return res;
        }
        if (root.left == null && root.right == null) {
            return new ArrayList<>();
        
} else if (root.left == null) {
            return flip(root.right, voyage, start + 1, end);
        } else if (root.right == null) {
            return flip(root.left, voyage, start + 1, end);
        } else {
            List<Integer> res = new ArrayList<>();
            if (root.left.val == voyage[start + 1]) {
                int rIndex = start + 1;
                while (rIndex <= end && voyage[rIndex] != root.right.val) {
                    rIndex++;
                }
                List<Integer> lres = flip(root.left, voyage, start + 1, rIndex - 1);
                if (lres.size() == 1 && lres.get(0) == -1) {
                    return lres;
                }
                List<Integer> rres = flip(root.right, voyage, rIndex, end);
                if (rres.size() == 1 && rres.get(0) == -1) {
                    return rres;
                }
                lres.addAll(rres);
                return lres;
            } else {
                if (root.right.val != voyage[start + 1]) {
                    res.add(-1);
                    return res;
                }
                res.add(root.val);
                int lIndex = start + 1;
                while (lIndex <= end && voyage[lIndex] != root.left.val) {
                    lIndex++;
                }
                List<Integer> lres = flip(root.left, voyage, lIndex, end);
                if (lres.size() == 1 && lres.get(0) == -1) {
                    return lres;
                }
                res.addAll(lres);
                List<Integer> rres = flip(root.right, voyage, start + 1, lIndex - 1);
                if (rres.size() == 1 && rres.get(0) == -1) {
                    return rres;
                }
                res.addAll(rres);
            }
            return res;
        }
    }
}
