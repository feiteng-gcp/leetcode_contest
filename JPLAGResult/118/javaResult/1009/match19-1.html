class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> res = new LinkedList<>();
        if (A == null || A.length == 0) {
        	return res;
        }
        int len = A.length;
        int[] pos = new int[len];
        
for (int i = 0; i < len; ++i) {
        	pos[A[i] - 1] = i;
        }
        for (int i = len - 1; i >= 0; --i) {
        	if (pos[i] != i) {
        		if (pos[i] != 0)
        			res.add(pos[i] + 1);
            	rev
erse(pos, len, pos[i]);
            	res.add(i + 1);
            	reverse(pos, len, i);
        	}
        }
        return res;
    }
    
    private void reverse(int[] pos, int len, 
int max) {
    	for (int i = 0; i < len; ++i) {
    		if (pos[i] <= max) {
    			pos[i] = max - pos[i];
    		}
    	}
    }
}
