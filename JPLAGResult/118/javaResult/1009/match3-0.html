class Solution {
    
    void reverse(int[] A, int len) {
        for (int i = 0, j = len - 1; i < j; i++, j--) {
            int t = A[i];
            A[i] = A[j];
            A[j] = t;
        }
    }
    
  public List<Integer> pancakeSort(int[] A) {

        List<Integer> ans = new ArrayList<>();
        for (int len = A.length; len > 1; len--) {

            int index = 0;
            for (int i = 0; i < len; i++) {
                if (A[i] > A[index]) {
                    index = i;
                }
            }

            ans.add(index + 1);
            ans.add(len);

            reverse(A, index + 1);
            reverse(A, len);
        }
        //System.out.println(Arrays.toString(A));
        return ans;
    }
}
