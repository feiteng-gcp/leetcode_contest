class Solution {
    public List<Integer> pancakeSort(int[] A) {
        final int N = A.length;
        ArrayList<Integer> ans = new ArrayList<>();
        for (int v = N; v >= 1; --v) {
            int idx = find(A, v);
            if (idx == v - 1) {
                continue;
            }
            
ans.add(idx + 1);
            flip(A, idx);
            ans.add(v);
            flip(A, v - 1);
        }
        return ans;
    }
    
    private static int find(int[] A, int val) {
        for (int i = 0; ; ++i) {
            if (A[i] == val) {
                return i;
            }
        }
    
}
    
    private static void flip(int[] A, int k) {
        for (int i = 0, j = k; i < j; ++i, --j) {
            int tmp = A[i];
            A[i] = A[j];
            A[j] = tmp;
        }
    }
}
