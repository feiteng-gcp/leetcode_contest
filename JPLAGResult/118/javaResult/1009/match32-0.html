class Solution {
    
    private void flip(int[] A, int frmIdx, int toIdx) {
        
        for (int i = frmIdx; i < frmIdx + (toIdx - frmIdx) / 2; i ++) {
            int temp = A[i];
            A[i] = A[toIdx - 1 - (i - frmIdx)];
            A[toIdx - 1 - (i - frmIdx)] = temp;
        }
    }
    
    private int largestIdx(int[] A, int frmIdx, int toIdx) {
        
        int idx = frmIdx;
        int largest = Integer.MIN_VALUE;
        for (int i = frmIdx; i < toIdx; i ++) {
            if (A[i] > largest) {
                largest = A[i];
                idx = i;
            }
        }
        return idx;
    }
    
    public List<Integer> pancakeSort(int[] A) {
        
        List<Integer> flips = new ArrayList<>();
        
        for (int i = A.length; i > 1; i --) {
            int idx = largestIdx(A, 0, i);
            flip(A, 0, idx + 1);
            flip(A, 0, i);
            flips.add(idx + 1);
            flips.add(i);
        }
        
        return flips;
    }
}
