class Solution {
    public List<Integer> pancakeSort(int[] A) {
        int n = A.length;
        List<Integer> res = new ArrayList<>();
        for (int i = n - 1; i >= 0; i--) {
            int idx = -1;
            for (int j = 0; j < n; j++) {
                if (A[j] == i + 1) {
                    idx = j;
                    break;
                }
            }
            
            res.add(idx + 1);
            swap(A, 0, idx);
            res.add(i + 1);
            swap(A, 0, i);
        }
        
        return res;
    }   
    
    void swap(int[] A, int low, int high) {
        for (int i = low, j = high; i < j; i++, j--) {
            int tmp = A[i];
            A[i] = A[j];
            A[j] = tmp;
        }
    }
}
