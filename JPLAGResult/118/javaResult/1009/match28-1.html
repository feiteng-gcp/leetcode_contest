class Solution {
    public List<Integer> pancakeSort(int[] A) {
        int[] S = Arrays.copyOf(A, A.length);
        Arrays.sort(S);
        
        // input: [5, -10, 20, 2, 5, 1]
        
        //  k=3:  [20, -10, 5, 2, 5, 1] ~ prepare 20 to move to the end
        //  k=6:  [1, 5, 2, 5, -10, 20] ~ moved 20
        
        //  k=4:  [5, 2, 5, 1, -10, 20] ~ prepare 5 to move to the end
        //  k=5:  [-10, 1, 5, 2, 5, 20] ~ moved 5
        
        //  k=3:  [5, 1, -10, 2, 5, 20] ~ prepare another 5
        //  k=4:  [2, -10, 1, 5, 5, 20]
        
        //  k=3:  [1, -10, 2, 5, 5, 20] ~ just moved 2
        
        //  k=2:  [-10, 1, 2, 5, 5, 20] ~ moved 1
        
        // sorte: [-10, 1, 2, 5, 5, 20]
        
        List<Integer> ret = new ArrayList<>();
        
        for(int last=A.length-1; last >= 0; last--) {
            if (S[last] == A[last]) continue;
            int idx = findIdx(A, S[last]);
            
            if (idx > 0) {
                int k = idx + 1;
                ret.add(k);
                flip(A, idx);
            }
            
            int k = last + 1;
            
ret.add(k);
            flip(A, last);
        }
        
        return ret;
    }
    
    private int findIdx(int[] A, int el) {
        for(int i=0; i<A.length; i++) {
            if (A[i] == el) return i
;
        }
        return -1;
    }
    
    private void flip(int[] A, int en) {
        int be = 0;
        while(be < en) {
            int tmp = A[be];
            A[be] = A[en];
            A[en] = tmp;
            be++;
            en--;
        }
    }
}
