import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class Solution {
    public List<Integer> pancakeSort(int[] A) {
        // move the last element to the right place always
        int[] sorted = Arrays.copyOf(A, A.length);
        Arrays.sort(sorted);
        
        ArrayList<Integer> moves = new ArrayList<>();
        for(int i = A.length - 1; i >= 0; i--) {
            // for the last index, what should go here?
            // find index of thing to sort
            int should = sorted[i];
            int idx = find(A, should);
            
            // to move to the last, first move to front...
            moves.add(idx + 1);
            reverse(A, idx + 1);
            moves.add(i + 1);
            reverse(A, i + 1);
        }
        
        return moves;
    }
    
    int find (int[] A, int target) {
        for(int i = 0; i < A.length; i++) {
            if(A[i] == target) {
                return i;
            }
        }
        
        return -1;
    }

    void reverse (int[] A, int target) {
        int[] tmp = new int[target];
        for(int i = 0; i < target ; i++) {
            tmp[i] = A[i];
        }

        for(int i = 0; i < target ; i++) {
            A[i] = tmp[target - 1 - i];
        }
    }
}
