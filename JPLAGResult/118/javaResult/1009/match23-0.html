class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> list = new ArrayList<>();
        //flip max to 0, then flip it to the last
        for (int i = A.length; i >= 1; i--){
            int currIdx = find(A, i);
            //System.out.println(i + " " + currIdx);
            if (currIdx == i-1)
                continue;
            list.add(currIdx+1);
            list.add(i);
            flip(A, currIdx+1);
            flip(A, i);
            //System.out.println(Arrays.toString(A));
        }
        return list;
    }
    
    private int find(int[] A, int target){
        for (int i = 0; i < A.length; i++){
            if (A[i] == target)
                return i;
        }
        return 0;
    }
    
    private void flip(int[] A, int k){
        int i1 = 0;
        int i2 = k-1;
        while(i1 < i2){
            swap(A, i1, i2);
            i1++;
            i2--;
        }
    }
    
    private void swap(int[] A, int i1, int i2){
        int temp = A[i1];
        A[i1] = A[i2];
        A[i2] = temp;
    }
}
