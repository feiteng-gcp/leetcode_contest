class Solution {
    
public List<Integer> pancakeSort(int[] A) {
        List<Integer> res = new ArrayList<>();
        for (int curr_size = A.length; curr_size > 1; --curr_size) 
        { 
            int mi = findMax(A, curr_size); 
            if (mi != curr_size-1) 
            { 
                res.add(mi + 1);
                flip(A, mi); 
                res.add(curr_size);
                flip(A, curr_size-1); 
            } 
        }
        
        return res;
    }
    
    static void flip(int arr[], int i) 
    { 
        int start = 0; 
        while (start < i) 
        { 
            int temp = arr[start]; 
            arr[start] = arr[i]; 
            arr[i] = temp; 
            start++; 
            i--; 
        } 
    }
    
    
static int findMax(int arr[], int n) 
    { 
        int mi, i; 
        for (mi = 0, i = 0; i < n; ++i) 
            if (arr[i] > arr[mi]) 
                mi = i; 
        return mi; 
    } 
}
