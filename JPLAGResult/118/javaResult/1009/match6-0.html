class Solution {
    public List<Integer> pancakeSort(int[] a) {
        List<Integer> list = new ArrayList<>();
        dfs(a, a.length, list);
        return list;
    }
    
    public void dfs(int[] a, int to, List<Integer> cur) {
        if(to == 1) {
            return;
        }
        int i = 0;
        for(; i < to; ++i) {
            if(a[i] == to) {
                break;
            }
        }
        if(i != to-1) {
            if(i != 0) {
                cur.add(i+1);
                flip(a, i+1);    
            }
            cur.add(to);
            flip(a,to);            
        }
        
        dfs(a,to-1, cur);
    }
    
    void flip(int[] a, int k) {
        int i = 0, j = k-1;
        while(i < j) {
            int t = a[i];
            a[i] = a[j];
            a[j] = t;
            i++;
            j--;
        }
    }
    
}
