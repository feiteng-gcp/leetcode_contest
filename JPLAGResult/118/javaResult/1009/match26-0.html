class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> ret= new ArrayList<Integer>();
        for(int i=A.length-1;i>=0;i--)
        {
            int max=i;
            for(int j=0;j<=i;j++)
            {
                max=A[j]>A[max]?j:max;
            }
            swap(A,max);
            swap(A,i);
            ret.add(max+1);
            ret.add(i+1);
        }
        System.out.println(Arrays.toString(A));
        return ret;
    }
    void swap(int[] A,int k)
    {
        int i=0,j=k;
        while(i<j)
        {
            int temp=A[i];
            A[i]=A[j];
            A[j]=temp;
            i++;
            j--;
        }
    }
}
