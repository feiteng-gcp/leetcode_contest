class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> res = new ArrayList<>();
        helper(A, A.length, res);
        return res;
    }
    
    void helper(int[] A, int end, List<Integer> res) {
        if(end == 0) return;
        int max = Integer.MIN_VALUE;
        int idx = 0;
        for(int i=0; i<end; i++) {
            if(A[i]>max) {
                max = A[i];
                idx = i;
            }
        }
        if(idx != end-1) {
            flip(A, idx);
            flip(A,end-1);
            
res.add(idx+1);
            res.add(end);
        }
        helper(A, end-1, res);
    }
    
    void flip(int[] A, int e) {
        int i=0, j=e;
        while(i<j) {
            int t = A[i];
            A[i] = A[j];
            A[j] = t;
            i++;
            j--;
        }
    }
}
