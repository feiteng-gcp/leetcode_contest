class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> ret = new ArrayList<>();
        int n = A.length;
        for (int i = n - 1; i >= 0; i--) {
            if (A[i] == i + 1)
                continue;
            int idx = find(A, i - 1, i + 1);
            //System.out.println("for " + (i + 1) + " found idx " + idx);
            ret.add(idx + 1);
            rev(A, idx);
            ret.add(i + 1);
            rev(A, i);
            //System.out.println(Arrays.toString(A));
        }
        return ret;
    }
    
    void rev(int[] a, int lim) {
        //System.out.println("rev till " + lim);
        for (int i = 0, j = lim; i <= j; i++, j--) {
            int tmp = a[i];
            a[i] = a[j];
            a[j] = tmp;
        }
    }
    
    int find(int[] a, int lim, int v) {
        //System.out.println("find lim " +lim + " v " + v);
        for (int i = 0; i <= lim; i++) {
            if (a[i] == v) {
                return i;
            }
        }
        return -1;
    }
}
