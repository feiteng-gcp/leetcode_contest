class Solution {
    public List<Integer> pancakeSort(int[] A) {
        int n = A.length;
        List<Integer> ans = new ArrayList<>();
        for (int i = n; i >= 2; i--) {
            int j = 0;
            while (A[j] != i) {
                j++;
            }
            if (j != i - 1) {
                if (j != 0) {
                    ans.add(j + 1);
                    flip(A, 0, j);
                }
                ans.add(i);
                flip(A, 0, i - 1);
            }
        }
        return ans;
    }
    
    private void flip(int[] A, int left, int right) {
        while (left < right) {
            int tmp = A[left];
            A[left] = A[right];
            A[right] = tmp;
            left++;
            right--;
        }
    }
}
