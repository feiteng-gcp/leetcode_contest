class Solution {
        
public void reverse(int k, int[] a) {
        int i = 0;
        int j = k;
        while (i < j) {
            int temp = a[i];
            a[i] = a[j];
            a[j] = temp;
            i ++;
            j --;
        }
    }


    public List<Integer> pancakeSort(int[] A) {
        int len = A.length;
        
List<Integer> res = new ArrayList<>();
        for (int i = len; i > 0; i--) {
            int max = A[0];
            int maxj = 0;
            for (int j = 0; j < i; j++) {
                if (A[j] > max) {
                    max = A[j];
                    maxj = j;
                }
            }
            if (maxj > 0) {
                reverse(maxj, A);
                res.add(maxj+1);
            }
            if (i > 1) {
                reverse(i - 1, A);
                res.add(i);
            }
        }
        return res;
    }
}
