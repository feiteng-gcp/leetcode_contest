class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> result = new ArrayList<>();
        dfs(A, A.length, result);
        return result;
    }
    
    private void dfs(int[] A, int target, List<Integer> result) {
        if (target == 1) return;
        
        int i = 0;
        while (A[i] != target) i++;
        
        if (i != target - 1) {
            if (i != 0) {
                pancakeflip(A, i + 1); // target at first
                result.add(i + 1);
            }
            
pancakeflip(A, target);
            result.add(target);
        }
        
        dfs(A, target - 1, result);
    }
    
    private void pancakeflip(int[] A, int len) {
        int i = 0, j = len - 1;
        while (i < j) {
            int tmp = A[i];
            A[i] = A[j];
            A[j] = tmp;
            i++;
            j--;
        }
    }
}
