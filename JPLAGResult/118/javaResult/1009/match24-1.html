import java.util.*;


class Solution {

    public List<Integer> pancakeSort(int[] A) {
        List<Integer> output = new ArrayList<>();
        for (int i = A.length; i > 1; i--) {
            Optional<Integer> optionalInteger = pancakeSort(A, i);
            if (optionalInteger.isPresent()) {
                output.add(
optionalInteger.get());
                output.add(i);
            }
        }
        return output;
    }

    private Optional<Integer> pancakeSort(int[] A, int n) {
        int i = indexOf(A, n);
        if (i == n - 1) {
            return Optional.empty();
        } else {
            reverse(A, i + 1);
            reverse(A, n);
            return Optional.of(i + 1);
        }
    
}

    private void reverse(int[] A, int n) {
        for (int i = 0; i < n / 2; i++) {
            int tmp = A[i];
            A[i] = A[n - 1 - i];
            A[n - 1 - i] = tmp;
        }
    }

    private int indexOf(int[] A, int n) {
        for (int i = n - 1; i >= 0; i--) {
            if (A[i] == n) {
                return i;
            }
        }
        throw new RuntimeException();
    }
}
