class Solution {

public List<Integer> pancakeSort(int[] A) {
		List<Integer> res = new ArrayList<>();
		for (int num = A.length; num > 0; num--) {
			int index = 0;
			for (int i = 0; i < A.length; i++) {
				if (A[i] == num) {
					index = i;
					break;
				}
			}
			if (index == num - 1) continue;
			res.add(index+1);

			res.add(num);
			solve(A, index+1);
			solve(A, num);
		}
		return res;
	}

	private void solve(int[] A, int k) {
		for (int i = 0; i <
 k/2; i++) {
			int t = A[i];
			A[i] = A[k - 1 - i];
			A[k - 1 - i] = t;
		}
	}
}
