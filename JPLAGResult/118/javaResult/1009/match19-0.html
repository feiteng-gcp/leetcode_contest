class Solution {
    int[] pos;
    
    private void flip(int s) {
        for (int i = 0; i < pos.length; ++i) {
            if (pos[i] <= s) {
                pos[i] = s - pos[i];
            }
        }
    }
    
    public List<Integer> pancakeSort(int[] A) {
        pos = new int[A.length];
        List<Integer> ans = new ArrayList<> ();
        
        for (int i = 0; i < A.length; ++i) {
            pos[A[i] - 1] = i;
        }
        
        for (int i = pos.length - 1; i >= 0; --i) {
            if (pos[i] != i) {
                ans.add(pos[i] + 1);
                ans.add(i + 1);
                flip(pos[i]);
                flip(i);
            }
        }
        
        return ans;
    }
}
