class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> output = new ArrayList<>();
        for (int i = 0; i < A.length; i++) {
            int index = 0;
            for (int j = 0; j < A.length; j++)
                if (A[j] == A.length - i)
                    index = j;
            if (index == A.length - i) continue;
            // System.out.println(Arrays.toString(A));
            flip(A, index);
            // System.out.println(Arrays.toString(A));
            if (index + 1 != 1)
                output.add(index + 1);
            flip(A, A.length - i - 1);
                        // System.out.println(Arrays.toString(A));
            if (A.length - i != 1)
            output.add(A.length - i);
        }
        return output;
    }
    
    public void flip(int[] A, int index) {
        // flip 0 to index
        for (int x = 0; x < (index + 1)/2; x++) {
            int temp = A[index - x];
            A[index - x] = A[x];
            A[x] = temp;
        }
    }
}
