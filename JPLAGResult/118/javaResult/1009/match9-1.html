class Solution {
    
    void flip(int[] arr, int r) {
        for (
int i=0; i<r/2; i++) {
            int tmp = arr[i];
            arr[i] = arr[r-i-1];
            arr[r-i-1] = tmp;
        }
    }
    
    int find(
int[] arr, int n) {
        for (int i=0; i<n; i++) {
            if (arr[i] == n) {
                return i;
            }
        }
        return -1;
    }
    
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> ans = new ArrayList<>();
        for (int i=0; i<A.length-1; i++) {
            
int index = find(A, A.length-i);
            ans.add(index+1);
            ans.add(A.length-i);
            flip(A, index+1);
            flip(A, A.length-i);
        }
        return ans;
    }
}
