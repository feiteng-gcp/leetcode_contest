class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> result = new LinkedList<>();
        
        for (int i=A.length; i>0; i--) {
            int idx = getIndex(A, i);
            if (idx != i) {
                if (idx > 1) {
                    result.add(idx);
                    flip(A, idx);
                }
                result.add(i);
                flip(A, i);
            }
            
            // display(A);
        }
        
        return result;
    }
    
    private void display(int[] A) {
        for(int i=0; i<A.length; i++)
            System.out.print(A[i] + " ");
        System.out.println();
    }
    
    private void flip(int[] A, int n) {
        for(int i=0; i<n/2; i++) {
            int tmp = A[i];
            A[i] = A[n-i-1];
            A[n-i-1] = tmp;
        }
    }
    
    private int getIndex(int[] A, int k) {
        for(int i=0; i<A.length; i++)
            if(k == A[i])
                return i+1;
        
        return -1;
    }
}
