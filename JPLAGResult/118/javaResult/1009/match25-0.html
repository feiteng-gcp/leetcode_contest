class Solution {
    public List<Integer> pancakeSort(int[] A) {
        int n = A.length;
        List<Integer> result = new ArrayList(n * 2);
        for(int i = n; i >= 1; i--)
        {
            int index = indexOf(A, i);
            result.add(index + 1);
            result.add(i);
            reverse(A, index + 1);
            reverse(A, i);
        }
        
        return result;
    }
    
    public void reverse(int[] A, int k)
    {
        int l = 0;
        int r = k - 1;
        while(l < r)
        {
            int t = A[l];
            A[l] = A[r];
            A[r] = t;
            l++;
            r--;
        }
    }

    
    public static int indexOf(int[] A, int a)
    {
        for(int i = 0, until = A.length; i < until; i++)
        {
            if(A[i] == a)
            {
                return i;
            }
        }
        return -1;
    }
}
