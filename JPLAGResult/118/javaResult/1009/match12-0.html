class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> ans = new LinkedList<>();
        for(int i=A.length; i>=1; i--){
            // find i
            int index = find(A, i);
            if(index==i-1)
                continue;
            ans.add(index+1);
            reverse(A, index);
            ans.add(i);
            reverse(A, i-1);
        }
        return ans;
    }
    public int find(int[] A, int target){
        for(int i=0; i<A.length; i++){
            if(A[i]==target)
                return i;
        }
        return -1;
    }
    public void reverse(int[] A, int index){
        int s=0,e=index;
        while(s<e){
            int temp = A[s];
            A[s] = A[e];
            A[e] = temp;
            s++;
            e--;
        }
        return;
    }

}
