class Solution {
    public List<Integer> pancakeSort(int[] A) {
        int N = A.length;
        List<Integer> res = new ArrayList<>();
        if (N == 1) {
            return res;
        }
        
for (int i = N - 1; i >= 0; i--) {
            int max = A[i];
            int maxIndex = i;
            for (int j = i; j >= 0; j--) {
                if (max < A[j]) {
                    max = A[j];
                    maxIndex = j;
                }
            }
            if (maxIndex != i) {
                if (maxIndex != 0) {
                    res.add(maxIndex + 1);
                    flip(A, maxIndex + 1);
                }
                res.add(i + 1);
                flip(A, i + 1);
            }
        
}
        return res;
    }

    void flip(int[] A, int n) {
        int i = 0;
        int j = n - 1;
        while (i < j) {
            int temp = A[i];
            A[i] = A[j];
            A[j] = temp;
            i++;
            j--;
        }
    }
}
