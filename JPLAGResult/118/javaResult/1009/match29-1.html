class Solution {
    public List<Integer> pancakeSort(int[] A) {
        int n = A.length;
        List<Integer> res = new ArrayList<>();
        while(n > 0) {
            int posOfMax = findPosOfMax(A, n);
            if(posOfMax != n - 1) {
                
if(posOfMax != 0) {
                    res.add(posOfMax + 1);
                    flip(A, posOfMax + 1);
                }
                res.add(n);
                flip(A, n);
            }
            n--;
        }
        return res;
    }
    
    private int findPosOfMax(
int[] A, int n) {
        int res = 0;
        for(int i = 1; i < n; i++) {
            if(A[i] >= A[res]) {
                res = i;
            }
        }
        
        if(res != n - 1 && A[res] == A[0]) {
            return 0;
        }
        
        return res;
    }
    
    private void flip(int[] A, int k) {
        for(
int i = 0; i < k/2; i++) {
            int tmp = A[i];
            A[i] = A[k - 1 - i];
            A[k - 1 - i] = tmp;
        }
    }
}
