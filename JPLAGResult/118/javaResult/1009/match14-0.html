class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> ret = new ArrayList<>();
        for (int i = 0; i < A.length; i++)
        {
            int max = -1, index = -1;
            for (int j = 0; j < A.length - i; j++)
            {
                if (A[j] > max)
                {
                    max = A[j];
                    index = j;
                }
            }
            if (index == A.length - i - 1) continue;
            if (index != 0)
                ret.add(index + 1);
            ret.add(A.length - i);
            flip(A, index);
            flip(A, A.length - i - 1);
        }
        return ret;
    }
    
    private void flip(int[] A, int index)
    {
        int left = 0, right = index;
        while (left < right)
        {
            int temp = A[left];
            A[left] = A[right];
            A[right] = temp;
            left++;
            right--;
        }
    }
}
