class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> res = new ArrayList<Integer>();
        int n = A.length;
        for(int i=n; i>1; i--){
            int mi= find_max(A, i);
            // System.out.println(i+","+mi);
            if(mi!=i-1){
                if(mi>0)
                    res.add(mi+1);
                res.add(i);
                flip(A, mi);
                flip(A,i-1);
            }
        }
        return res;
    }
    public int find_max(int[] A, 
int cur_size){
        int res = Integer.MIN_VALUE;
        int idx = -1;
        for(int i=0; i<cur_size; i++){
            if(A[i]>res){
                res = A[i];
                idx = i;
            }
        
}
        return idx;
    }
    public void flip(int[] A, int k){
        int i=0; 
        int j= k;
        while(i<j){
            int temp = A[i];
            A[i] = A[j];
            A[j] = temp;
            i++;
            j--;
        }
    }
}
