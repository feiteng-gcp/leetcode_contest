class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> list = new ArrayList<>();
        for (int i = A.length - 1; i > 0; i--) {
            flip(list, A, i);
        }
        return list;
    }
    
    void flip(List<Integer> list, int[] a, int last) {
        int m = 0;
        for (int i = 1; i <= last; i++) {
            if (a[i] > a[m]) {
                m = i;
            }
        }
        if (m == last) {
            return;
        }
        if (m > 0) {
            list.add(m + 1);
            flip(a, m);
        }
        list.add(last + 1);
        flip(a, last);
    }
    
    void flip(int[] a, int last) {
        for (int i = 0, j = last; i < j; i++, j--) {
            int temp = a[i];
            a[i] = a[j];
            a[j] = temp;
        }
    }
    
}
