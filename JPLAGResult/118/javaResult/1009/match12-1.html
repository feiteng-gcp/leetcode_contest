class Solution {
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> re = new ArrayList<>();
        for(int i = A.length; i > 1; i--) {
            int p = get(A, i);
            if(p == i - 1) continue;
            re.add(p + 1);
            re.add(i);
            update(A, i, p);
        
}
        return re;
    }
    int get(int[] A, int i) {
        for(int j = 0; j < A.length; j++) if(A[j] == i) return j;
        return 0;
    }
    void update(int[] A, int end, int p) {
        reverse(A, p + 1);
        reverse(A, end);
    
}
    void reverse(int[] A, int end) {
        int e = end - 1, s = 0;
        while(s < e) {
            int temp = A[s];
            A[s] = A[e];
            A[e] = temp;
            s++;
            e--;
        }
    }
}
