class Solution {
    
    
    public int[] reverse(int num,int[] A){
        int b = 0;
        int e = num-1;
        while(b<e){
            int t = A[b];
            A[b] = A[e];
            A[e] = t;
            b++;
            e--;
        }
        
        int l = A.length;
        int pos[] = new int[A.length+1];
        for(int i=0;i<l;i++){
            pos[A[i]] = i+1;
        }
        return pos;
    }
    
    
    
    public List<Integer> pancakeSort(int[] A) {
        List<Integer> r  =new ArrayList<>();
        int l = A.length;
        int pos[] = new int[A.length+1];
        for(int i=0;i<l;i++){
            pos[A[i]] = i+1;
        }
        for(int i=1;i<l+1;i++){
            r.add(pos[i]);
            pos = reverse(pos[i],A);
            r.add(l-i+1);
            pos = reverse(l-i+1,A);
        }
        
        r.add(l);
        return r;
        
        
        
    }
}
