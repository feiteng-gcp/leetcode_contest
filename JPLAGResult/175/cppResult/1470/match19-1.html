class TweetCounts {
public:
  map<string, vector<int> > mp;
  TweetCounts() {
    mp.clear();
  }

  void recordTweet(string tweetName, int time) {
    mp[tweetName].emplace_back(time);
  }

  vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
    int delta=0;
    if(freq=="minute") delta=60;
    if(freq=="hour") delta=60*60;
    if(freq=="day") delta=60*60*24;
    endTime++;

    int num=(endTime-startTime+delta-1)/delta;
    vector<int> res(num,0);
    for(int v:mp[tweetName])
      if(startTime<=v && v<endTime)
        res[(v-startTime)/delta]++;

    return res;
  }
};
