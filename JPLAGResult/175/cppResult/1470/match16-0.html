class TweetCounts {
public:
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        rec[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int itvl = 0;
        if (freq == "minute") {
            itvl = 60;
        } else if (freq == "hour") {
            itvl = 3600;
        } else if (freq == "day") {
            itvl = 86400;
        }
        vector<int> ans;        
        while (true) {
            auto it = rec[tweetName].lower_bound(startTime);            
            int nxt_startTime = min(startTime+itvl, endTime+1);
            auto it2 = rec[tweetName].lower_bound(nxt_startTime);
            int cnt = 0;
            while (it != it2) {
                cnt += 1;
                ++it;
            }
            ans.push_back(cnt);
            startTime = nxt_startTime;
            if (nxt_startTime == endTime+1)
                break;
        }
        return ans;
        
    }
    unordered_map<string, multiset<int>> rec;
};
