class TweetCounts {

public:
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        M[tweetName].insert(time);   
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int P = 0;

        if (freq[0] == 'm') {
            P = 60;
        } else if (freq[0] == 'h') {
            P = 60 * 60;
        } else {
            P = 24 * 60 * 60;
        }
        
        vector<int> res;
        if (M.count(tweetName) == 0) {
            for (int i = startTime; i <= endTime; i += P) {
                res.push_back(0);
            }
        } else {
            auto& S = M[tweetName];
            auto it = S.lower_bound(startTime);
            for (int i = startTime; i <= endTime; i += P) {
                int l = i, r = min(endTime + 1, i + P);
                int cnt = 0;
                while (it != S.end()) {
                    if (*it < l) {
                        ++it;
                    } else if (*it < r) {
                        ++cnt;
                        ++it;
                    } else {
                        break;
                    }
                }
                res.push_back(cnt);
            }
        }
        return res;
    }
    
private:
    map<string,multiset<int>> M;
};
