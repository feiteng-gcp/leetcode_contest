class TweetCounts {
    unordered_map<string, multiset<int>> rec;
public:
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        rec[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int l;

        if(freq == "minute"){
            l = 60;
        }else if(freq == "hour"){
            l = 3600;
        }else{ // day
            l = 86400;
        }
        int L = endTime - startTime + 1;
        int size = L / l + (L % l != 0);
        vector<int> result(size);

        if(rec.find(tweetName) == rec.end()){
            return result;
        }
        
        auto begin = rec[tweetName].lower_bound(startTime);
        auto end = rec[tweetName].upper_bound(endTime);
        
        for(auto it = begin; it != end; ++it){
            ++result[(*it - startTime) / l];
        }
        
        return result;
    }
};
