class TweetCounts {
public:
    TweetCounts(): ms() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        ms[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        vector<int> ret;
        if (ms.find(tweetName) == ms.end()) return {};
        
        auto m = ms[tweetName];
        int delta = 1;
        if (freq == "hour") delta = 3600;
        else if (freq == "minute") delta = 60;
        else if (freq == "day") delta = 86400;
        
        for (int t = startTime; t <= endTime; t += delta) {
            auto it_start = m.lower_bound(t);
            auto it_end = m.lower_bound(min(endTime, t + delta - 1) + 1);
            ret.push_back(distance(it_start, it_end));
        }
        return ret;
    }
    
    map<string, multiset<int>> ms;
};
