class TweetCounts {
public:
    TweetCounts() {
    }
    
    void recordTweet(string tweetName, int time) {
        hash[tweetName].push_back(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int del;
        if (freq[0]=='m') del=60;
        else if (freq[0]=='h') del=3600;
        else if (freq[0]=='d') del=3600*24;
        vector<int> res((endTime-startTime)/del+1,0);
        for (int t:hash[tweetName]) 
            if (startTime<=t && t<=endTime) {
                res[(t-startTime)/del]++;
            }
        return res;
    }
private:
    map<string,vector<int>> hash;
};
