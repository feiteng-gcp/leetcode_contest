class TweetCounts {
public:
    TweetCounts() {}
    
    void recordTweet(string tweetName, int time) {
        m[tweetName].emplace(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        vector<int> result;
        const auto& timestamp = m[tweetName];
        auto itr = timestamp.lower_bound(startTime);
        int interval;
        if (freq == "minute") {
            interval = 60;
        } else if (freq == "hour") {
            interval = 3600;
        } else {
            interval = 86400;
        }
        int count = 0;
        while (startTime <= endTime) {
            while (itr != timestamp.end() && *itr <= min(endTime, startTime + interval - 1)) {
                count++;
                itr++;
            }
            result.emplace_back(count);
            count = 0;
            startTime += interval;
        }
        return result;
    }
private:
    unordered_map<string,multiset<int>> m;
};
