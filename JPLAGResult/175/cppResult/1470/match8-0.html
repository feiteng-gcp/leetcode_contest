class TweetCounts {
private:
    map<string,multiset<int>> h;
    
public:
    TweetCounts() {
        h.clear();
    }
    
    void recordTweet(string tweetName, int time) {
        h[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int f;
        if(freq == "minute") f = 60;
        else if(freq == "hour") f = 3600;
        else f = 3600*24;
        
        vector<int> ans;
        vector<int> g(h[tweetName].begin(),h[tweetName].end());
        
        
        // for(auto x:g) printf("%d ",x);
        // puts("");
        for(int i = startTime ; i <= endTime ; i += f) {
            ans.push_back(lower_bound(g.begin(),g.end(),min(endTime+1,i+f))-lower_bound(g.begin(),g.end(),i));
        }
        return ans;
    }
};

 * obj->recordTweet(tweetName,time);
