class TweetCounts {
public:
            unordered_map<string,unordered_map<int,int> > m;
    TweetCounts() {
    }
    
    void recordTweet(string name, int time) {
        m[name][time]++;

    }
    
    vector<int> getTweetCountsPerFrequency(string fre, string name, int startTime, int endTime) {
        int intval;
        if(fre == "minute")
            intval = 60;
        else if(fre == "hour")
            intval = 3600;
        else
            intval = 3600 * 24;

        unordered_map<int,int> &m1 = m[name];
                vector<int> w( (endTime - startTime) / intval + 1,0);
        for(auto p:m1){
            if(p.first >= startTime && p.first <= endTime){
                int index = (p.first - startTime)/intval;
                w[index]+=p.second;
            }
        }
        return w;       
    }
};
