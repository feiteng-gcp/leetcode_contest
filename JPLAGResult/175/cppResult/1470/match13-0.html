class TweetCounts {
public:
    map<string, vector<int>> mp;
    TweetCounts() {
        mp.clear();
    }
    
    void recordTweet(string tweetName, int time) {
        mp[tweetName].push_back(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int delta = 60;
        if (freq == "hour") delta = 3600;
        if (freq == "day") delta = 3600 * 24;
        vector<int> ret((endTime - startTime + delta) / delta);
        for (auto t : mp[tweetName]) {
            if (t < startTime or t > endTime) continue;
            int i = (t - startTime) / delta;
            ret[i]++;
        }
        return ret;
    }
};
