class TweetCounts {

public:
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        times[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int interval = 0;

        if (freq == "minute") {
            interval = 60;
        } else if (freq == "hour") {
            interval = 60 * 60;
        } else {
            interval = 60 * 60 * 24;
        }
        auto& ts = times[tweetName];
        auto it = ts.begin();
        while (it != ts.end() && *it < startTime) {
            ++it;
        }
        vector<int> freqs;
        for (int t = startTime; t <= endTime; t += interval) {
            if (it == ts.end() || *it > endTime) {
                freqs.push_back(0);
                continue;
            }
            int num = 0;
            while (it != ts.end() && *it < min(t + interval, endTime + 1)) {
                ++it;
                ++num;
            }
            freqs.push_back(num);
        }
        return freqs;
    }
    
private:
    map<string, multiset<int>> times;
};
