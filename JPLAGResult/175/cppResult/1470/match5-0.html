class TweetCounts {
	map<string, vector<int>> plist;
public:
	TweetCounts() {
		plist.clear();
	}

	void recordTweet(string tweetName, int time) {
		plist[tweetName].push_back(time);
	}

	vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
		int r, i, j, k, a, b;
		if (freq == "minute") r = 60;
		if (freq == "hour") r = 3600;
		if (freq == "day") r = 3600 * 24;
		vector<int> ans;
		vector<int> v = plist[tweetName];
		sort(v.begin(), v.end());

		ans.clear();
		while (startTime <= endTime) {
			a = lower_bound(v.begin(), v.end(), startTime) - v.begin();
			b = upper_bound(v.begin(), v.end(), min(endTime, startTime + r - 1)) - v.begin() - 1;
			ans.push_back(b - a + 1);
			startTime += r;
		}
		return ans;
	}
};
