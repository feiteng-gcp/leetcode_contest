class TweetCounts
{
public:
    unordered_map<string, set<int> > tab;
    TweetCounts()
    {
        tab.clear();
    }
    
    void recordTweet(string tweetName, int time)
    {
        tab[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime)
    {
        int T = 0;
        if(freq == "minute") T = 60;
        else if(freq == "hour") T = 3600;
        else T = 86400;
        
        vector<int> ans;
        for(int t = startTime; t <= endTime; t += T)
        {
            if(!tab.count(tweetName))
            {
                ans.push_back(0);
                continue;
            }

            auto p = tab[tweetName].lower_bound(t);
            int cur = 0;
            while(p != tab[tweetName].end() && *p < t + T && *p <= endTime)
            {
                cur++;
                p++;
            }
            ans.push_back(cur);
        }
        return ans;
    }
};
