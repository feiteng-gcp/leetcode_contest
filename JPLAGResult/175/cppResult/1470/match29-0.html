class TweetCounts {
public:
    map <string, vector <int> > v;
    
    TweetCounts() {
        v.clear();
    }
    
    void recordTweet(string tweetName, int time) {
        v[tweetName].push_back(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int f = 0;
        if (freq == "minute") {
            f = 60;
        } else if (freq == "hour") {
            f = 60 * 60;
        } else {
            f = 60 * 60 * 24;
        }
        vector <int> ans((endTime - startTime) / f + 1, 0);
        for (int i = 0; i < v[tweetName].size(); i++) {
            if (v[tweetName][i] >= startTime && v[tweetName][i] <= endTime) {
                ans[(v[tweetName][i] - startTime) / f]++;
            }
        }
        return ans;
    }
};
