class TweetCounts {
public:
    
    map<string,multiset<int> > m;
    
    TweetCounts() {
        m.clear();
    }
    
    
    void recordTweet(string tweetName, int time) {
        m[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int d=0;
        if(freq[0]=='m')
            d=60;
        if(freq[0]=='h')
            d=60*60;
        if(freq[0]=='d')
            d=24*60*60;
        
        multiset<int> st=m[tweetName];
        multiset<int> :: iterator it;
        vector<int> ans;
        
        it=st.begin();
        
        if(st.empty())
            return ans;
        
        int was=0;
        int have=0;
        
        while(it!=st.end() && *it<startTime)
        {
            it++;
            have++;
            was++;
        }
        
        for(int i=startTime;i<=endTime;i+=d)
        {
            int l=i;
            int r=min(i+d-1,endTime);
                        
            while(it!=st.end() && *it<=r)
            {
                it++;
                have++;
            }
            
            ans.push_back(have-was);
            was=have;
        }
        return ans;
        
        
    }
};
