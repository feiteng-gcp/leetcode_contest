class TweetCounts {
private:

    unordered_map<string, set<int>> m;
public:
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        m[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        set<int> st = m[tweetName];

        vector<int> v(st.begin(), st.end());
        int delta;
        if(freq.compare("minute") == 0){
            delta = 60;
        } else if(freq.compare("hour") == 0){
            delta = 3600;
        } else{
            delta = 86400;
        }
        vector<int> ans;
        int end = startTime + delta;
        while(end <= endTime){
            int idx1 = lower_bound(v.begin(), v.end(), startTime) - v.begin();
            if(idx1 < v.size() && v[idx1] < end){
                int idx2 = lower_bound(v.begin(), v.end(), end - 1) - v.begin();
                if(idx2 < v.size()){
                    if(v[idx2] < end){
                        ans.push_back((idx2 - idx1) + 1);
                    } else{
                        ans.push_back(idx2 - idx1);
                    }
                } else{
                    ans.push_back(((int)(v.size())) - idx1);
                }
            } else{
                ans.push_back(0);
            }
            startTime = end;
            end = startTime + delta;
        }
        endTime++;
        int idx1 = lower_bound(v.begin(), v.end(), startTime) - v.begin();
        if(idx1 < v.size() && v[idx1] < endTime){
            int idx2 = lower_bound(v.begin(), v.end(), (endTime - 1)) - v.begin();
            if(idx2 < v.size()){
                if(v[idx2] < endTime){
                    ans.push_back((idx2 - idx1) + 1);
                } else{
                    ans.push_back(idx2 - idx1);
                }
            } else{
                ans.push_back(((int)(v.size())) - idx1);
            }
        } else{
            ans.push_back(0);
        }
        return ans;
    }
};
