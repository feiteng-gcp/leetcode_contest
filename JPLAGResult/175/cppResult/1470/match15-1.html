class TweetCounts {
public:
    map<string, multiset<int>> tw;

    map<string, int> maxtime;
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        tw[tweetName].insert(time);
        //maxtime[tweetName] = max(maxtime[tweetName], time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int interval = 1;
        if (freq == "minute") {
            interval = 60;
        }
        else if (freq == "hour") {
            interval = 60 * 60;
        }
        else if (freq == "day") {
            interval = 60 * 60 * 24;
        }

        int sz = ((endTime - startTime) / interval) + 1;
        vector<int> ans(sz);
        for (int t : tw[tweetName]) {
            if (startTime <= t && t <= endTime) {
                ans[(t - startTime) / interval]++;
            }
        }
        return ans;        
    }
};
