class TweetCounts {
public:
    
    unordered_map<int, vector<string>> mp;
    
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
         mp[time].push_back(tweetName);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
               
        int val = 0;
        if (freq[0] == 'm')
            val = 60;
        if (freq[0] == 'h')
            val = 3600;
        if (freq[0] == 'd')
            val = 3600 * 24;
        
        int x = (endTime - startTime) / val;
        vector<int> ans(x + 1, 0);
        
        for (int i = startTime; i <= endTime; i++)
        {
            if (mp[i].size() == 0)
                continue;
            int cnt = count(mp[i].begin(), mp[i].end(), tweetName);
            //printf("i:%d cnt:%d\n", i, cnt);
            ans[(i - startTime) / val] += cnt;
        }
        
        return ans;
    }
};
