class TweetCounts {    
    map<string, multiset<int>> m;
public:
    TweetCounts() {
    }
    
    void recordTweet(string tweetName, int time) {
        m[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int t = freq == "minute" ? 60 : freq == "hour" ? 3600 : 24 * 3600;
        vector<int> ret;
        int s = startTime;
        while (s <= endTime) {
            int e = s + t - 1;
            if (e >= endTime) {
                e = endTime;
            }
            auto a = m[tweetName].lower_bound(s);
            auto b = m[tweetName].upper_bound(e);
            ret.push_back(distance(a, b));
            s += t;            
        }
        return ret;
    }
};
