class TweetCounts {
    public:
        unordered_map < string, set < int > > T;
        TweetCounts() {

        }

        void recordTweet(string tweetName, int time) {
            T[tweetName].insert(time);
        }

        vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
            int d = 60;
            if (freq == "hour") d = 3600;
            if (freq == "day") d = 86400;

            int n = ((endTime - startTime)/d)+1;
            vector < int > res(n, 0);

            auto &S = T[tweetName];
            auto it = S.lower_bound(startTime);
            while (it != S.end()) {
                if (*it > endTime) break;
                res[(*it-startTime)/d]++;
                it++;
            }

            return res;
        }
    };
