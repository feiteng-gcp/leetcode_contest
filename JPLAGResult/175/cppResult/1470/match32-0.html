class TweetCounts {
    unordered_map<string, multiset<int>> h_;
public:
    TweetCounts() {
        
    }
    
    void recordTweet(string tweetName, int time) {
        h_[tweetName].insert(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string tweetName, int startTime, int endTime) {
        int delta;
        vector<int> output;

        if (freq == "minute") {
            delta = 60;
        } else if (freq == "hour") {
            delta = 3600;
        } else {
            delta = 86400;
        }
        
        output.resize((endTime - startTime) / delta + 1, 0);
        auto it = h_[tweetName].lower_bound(startTime);
        while (it != h_[tweetName].end() && *it <= endTime) {
            output[(*it - startTime) / delta]++;
            it++;
        }
        return output;
    }
};
