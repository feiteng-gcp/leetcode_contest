class TweetCounts {
public:
    map<string,vector<int>> mp;
    TweetCounts() {
        
    }
    
    void recordTweet(string n, int time) {
        mp[n].push_back(time);
    }
    
    vector<int> getTweetCountsPerFrequency(string freq, string name, int s, int t) {
        int d;
        if(freq[0] == 'm') d = 60;
        if(freq[0] == 'h') d = 3600;
        if(freq[0] == 'd') d = 24 * 3600;
        int n = (t-s)/d+1;
        vector<int> ret(n);
        for(int i : mp[name]) {
            if(s <= i && i <= t) {
                ret[(i-s)/d]++;
            }
        }
        return ret;
    }
};
