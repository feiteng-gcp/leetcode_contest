class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int m = (int)seats.size(), n = (int)seats[0].size();
    vector<vector<int> > dp(m + 1, vector<int>(1 << n, -1));
    dp[0][0] = 0;
    for(int i = 0; i < m; i++)
        for(int j = 0; j < (1 << n); j++) if(dp[i][j] != -1)
        {

            for(int k = 0; k < (1 << n); k++)
            {
                bool ok = true;
                int cnt = 0;
                for(int l = 0; l < n; l++)
                    if((k >> l) & 1)
                    {
                        cnt++;

                        if(seats[i][l] == '#' ||
                           (l > 0 && (((j >> (l - 1)) & 1) || ((k >> (l - 1)) & 1))) ||
                           (l < n - 1 && (((j >> (l + 1)) & 1) || ((k >> (l + 1)) & 1))))
                            ok = false;
                    }
                if(ok) dp[i + 1][k] = max(dp[i + 1][k], dp[i][j] + cnt);
            }
        }
    int ans = 0;
    for(int i = 0; i < (1 << n); i++) ans = max(ans, dp[m][i]);
    return ans;
    }
};
