class Solution {
public:
    int n, m;
    int dp[8][1 << 8][1 << 8];
    int f[8][1 << 8];
    int maxStudents(vector<vector<char>>& seats) {
        n = seats.size();
        m = seats[0].size();
        memset(dp, 0, sizeof dp);

        memset(f, 0,sizeof f);
        for (int i = 0; i < (1 << m); i++){
            bool ok = true;
            int cnt = 0;
            for (int j = 0; j < m; j++){
                if (i & (1 << j)){
                    cnt++;
                    if (seats[0][j] == '#'){
                        ok = false;
                        break;
                    }
                    if (j > 0){
                        if (i & (1 << (j - 1))){
                            ok = false;
                            break;
                        }
                    }
                }
            }
            if (ok){
                dp[0][0][i] = cnt;
                f[0][i] = cnt;
                cout << i << " " << cnt << endl;
            }
        }
        for (int i = 1; i < n; i++){

            for (int j = 0; j < (1 << m); j++){
                
                
              //  cout << i - 1 << " " << j << " " << f[i - 1][j] << endl;
                
                bool ok = true;
                int cnt = 0;
                for (int k = 0; k < m; k++){
                    if (j & (1 << k)){
                        cnt++;
                        if (seats[i][k] == '#'){
                            ok = false;
                            break;
                        }
                        if (k > 0){
                            if (j & (1 << (k - 1))){
                                ok = false;
                                break;
                            }
                            
                        }
                        
                    }
                }
                if (ok){
                    cout << i << " " << j << endl;
                    for (int k = 0; k < (1 << m); k++){
                        //dp[k][j];
                        
                        bool ok1 = true;
                        for (int l = 0; l < m; l++){
                            if (k & (1 << l)){
                                if (l > 0){
                                    if (j & (1 << (l - 1))){
                                        ok1 = false;
                                        break;
                                    }
                                }
                                if (l < m - 1){
                                    if (j & (1 << (l + 1))){
                                        ok1 = false;
                                        break;
                                    }
                                }
                            }
                        }
                        if (ok1){
                            dp[i][k][j] = max(dp[i][k][j], f[i - 1][k] + cnt);
                            f[i][j] = max(f[i][j], dp[i][k][j]);
                            //cout << "## " << i << " " << j << " " << f[i][j] << endl;
                        }
                    }
                }

            }
        }
        int ans = 0;
        for (int i = 0; i < (1 << m); i++)
            ans = max(ans, f[n-1][i]);
        //for (int i = 0; i < (1 << m); i++)
           // cout << "!:" << f[n][i] << endl;
        return ans;
    }
};
