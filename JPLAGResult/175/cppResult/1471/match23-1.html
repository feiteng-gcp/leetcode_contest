class Solution {
public:
    int dp[10][1 << 8];
    void ck(int &x, int y) {
        if (x < y) x = y;
    }
    int maxStudents(vector<vector<char>>& seats) {
        int n = seats.size(), m = seats[0].size();

        memset(dp, -1, sizeof(dp));
        dp[0][0] = 0;
        for (int i = 0; i < n; i++) {
            for (int s = 0; s < (1 << m); s++) {
                if (dp[i][s] == -1) continue;
                for (int s2 = 0; s2 < (1 << m); s2++) {
                    bool sol = true;
                    int cnt = 0;
                    for (int j = 0; j < m; j++) {
                        if (s2 & (1 << j)) {
                            if (seats[i][j] == '#') {
                                sol = false;
                                break;
                            }
                            cnt++;
                            if (i > 0) {
                                if (j > 0 && (s & (1 << (j - 1)))) sol = false;
                                if (j + 1 < m && (s & (1 << (j + 1)))) sol = false;
                            }
                            if (j > 0 && (s2 & (1 << (j - 1)))) sol = false;
                            if (j + 1 < m && (s2 & (1 << (j + 1)))) sol = false;

                            if (!sol) break;
                        }
                    }
                    if (sol) {
                        ck(dp[i + 1][s2], dp[i][s] + cnt);
                    }
                }
            }
        }
        int res = 0;
        for (int i = 0; i < (1 << m); i++) {
            res = max(res, dp[n][i]);
        }
        return res;
    }
};
