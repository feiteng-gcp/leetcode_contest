class Solution {
public:
    int maxStudents(vector<vector<char>>& a) {
        int n = a.size(), m = a[0].size();
        vector<vector<int>> dp(n, vector<int>((1 << m), 0));

        int ans = 0;

        for(int i = 0; i < n; i++) {
            for(int j = 0; j < (1 << m); j++) {
                bool ok = true;
                int cnt = 0;
                for(int k = 0; k < m; k++) {
                    if(((j >> k) & 1)) {
                        if(a[i][k] != '.') ok = false;
                        if(k > 0 && ((j >> (k - 1)) & 1)) ok = false;
                        if(k < n - 1 && ((j >> (k + 1)) & 1)) ok = false;
                        cnt += 1;
                    }
                }

                if(ok) {
                    if(i == 0) dp[i][j] = cnt;

                    else {
                        for(int l = 0; l < (1 << m); l++) {
                            bool can = true;
                            for(int k = 0; k < m; k++) {
                                if((j >> k) & 1) {
                                    if(k > 0 && ((l >> (k - 1)) & 1)) can = false;
                                    if(k < n - 1 && ((l >> (k + 1)) & 1)) can = false;
                                }
                            }
                            if(can) {
                                dp[i][j] = max(dp[i][j], dp[i - 1][l] + cnt);
                            }
                        }
                    }
                    ans = max(ans, dp[i][j]);
                }
            }
        }
        return ans;
    }
};
