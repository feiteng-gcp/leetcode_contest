class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int n = seats.size(), m = seats[0].size();
        
        int ans = 0;
        
        int dp[10][1<<8] = {0};
        memset(dp, -1, sizeof(dp));
        dp[0][0] = 0;
        for (int i = 0; i < n; ++i) {
            for (int mask = 0; mask < (1 << m); ++mask) {
                if (dp[i][mask] == -1) continue;
                for (int nmask = 0; nmask < (1 << m); ++nmask) {
                    bool isGood = true;
                    int cnt = 0;
                    for (int j = 0; j < m && isGood; ++j) {
                        if (nmask & (1 << j)) {
                            ++cnt;
                            if (seats[i][j] == '#') isGood = false;
                            if (j - 1 >= 0 && (nmask & (1 << (j - 1)))) isGood = false;
                            if (j + 1 < m && (nmask & (1 << (j + 1)))) isGood = false;
                            if (j - 1 >= 0 && (mask & (1 << (j - 1)))) isGood = false;
                            if (j + 1 < m && (mask & (1 << (j + 1)))) isGood = false;
                        }
                    }
                    if (isGood) {
                        dp[i + 1][nmask] = max(dp[i + 1][nmask], dp[i][mask] + cnt);
                        ans = max(ans, dp[i + 1][nmask]);
                    }
                }
            }
        }
        
        return ans;
    }
};
