class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int n = seats.size();
        int m = seats[0].size();
        
        vector<int> valid;
        for(auto x:seats) {
            int t = 0;
            for(int i = 0 ; i < m ; i++) {
                t <<= 1;
                if(x[i] == '.') t |= 1;
            }
            valid.push_back(t);
        }
        
        static int dp[8][1<<8];
        
        memset(dp,-1,sizeof dp);
        
        function<bool(int)> check_nei = [&] (int p) {
            for(int k = 0 ; k < m-1 ; k++)
                if((p>>k&1) && (p>>(k+1)&1)) return false;
            return true;
        };
        
        function<int(int)> get_ones = [&] (int j) {
            int cnt = 0;
            while(j) {
                if(j &1) cnt++;
                j >>= 1;
            }
            return cnt;
        };
        
        int ans = 0;
        
        for(int j = 0 ; j < 1<<m ; j++){
            if(~valid[0]&j) continue;
            if(!check_nei(j)) continue;
            dp[0][j] = get_ones(j);
        }
        
        for(int i = 1 ; i < n ; i++) {
            for(int j = 0 ; j < 1<<m ; j++) {
                if(~valid[i]&j) continue;
                if(!check_nei(j)) continue;
                
                int kk = 0;
                for(int ii = 0 ; ii < m ; ii++) {
                    if(j>>ii&1) {
                        if(ii-1>=0) kk |= 1<<(ii-1);
                        if(ii+1<m) kk |= 1<<(ii+1);
                    }
                }
                
                

                for(int k = 0 ; k < 1<< m ; k++) {
                    if(dp[i-1][k]==-1) continue;
                    if(kk&k) continue;
                    dp[i][j] = max(dp[i][j], dp[i-1][k] + get_ones(j));
                }
            }
        }
        
        for(int i = 0 ; i < 1<<m ; i++)
            ans = max(dp[n-1][i],ans);
        return ans;
    }
};
