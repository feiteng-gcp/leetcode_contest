class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int m = seats.size();
        int n = seats[0].size();
        int mask = (1 << n) - 1;
        vector<int> dp(mask + 1, -1);
        dp[0] = 0;
        vector<int> next_dp(mask + 1, -1);
        for (int r = 0; r < m; r++) {
            for (int i = 0; i <= mask; i++) {
                next_dp[i] = -1;
                int new_seat;
                if (!check_valid(i, seats[r], &new_seat)) {
                    continue;
                }
                for (int j = 0; j <= mask; j++) {
                    if (dp[j] == -1) continue;
                    if (can_cheat(i, j, n)) continue;
                    int tmp = new_seat + dp[j];
                    next_dp[i] = max(next_dp[i], tmp);
                    //printf("row %d, i = 0x%x, try j 0x%x, tmp = %d\n", r, i, j, tmp);
                }
                //printf("row %d, dp[0x%0x] = %d\n", r, i, next_dp[i]);
            }
            dp = next_dp;
        }
        int res = 0;
        for (int i = 0; i <= mask; i++) {
            res = max(res, dp[i]);
        }
        return res;
    }
    
    bool check_valid(int mask, vector<char>& seats, int* new_seat) {
        *new_seat = 0;
        for (int i = 0; i < seats.size(); i++) {
            if (mask & (1 << i)) {
                if (seats[i] != '.') return false;
                if (i + 1 < seats.size() && (mask & (1 << (i + 1))))
                    return false;
                ++*new_seat;
            }
        }
        return true;
    }
    
    bool can_cheat(int mask, int prev_mask, int n) {
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i)) {
                if (i > 0 && (prev_mask & (1 << (i - 1)))) return true;
                if (i + 1 < n && (prev_mask & (1 << (i + 1)))) return true;
            }
        }
        return false;
    }
};
