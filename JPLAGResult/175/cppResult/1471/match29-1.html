class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int n = seats.size();
        int m = seats[0].size();

        vector<vector<int>> dp(n, vector<int>((1 << m), -1));
        int result = -1;
        for (int i = 0; i < n; i++) {
            for (int st = 0; st < (1 << m); st++) {

                bool valid = true;
                for (int j = 0; j < m; j++) {
                    if (st >> j & 1) {
                        if (seats[i][j] == '#') {
                            valid = false;
                            break;
                        }
                        if (j > 0 && (st >> (j - 1) & 1)) {
                            valid = false;
                            break;
                        }
                        if (j < m - 1 && (st >> (j + 1) & 1)) {
                            valid = false;
                            break;
                        }
                    }
                }
                if (!valid) {
                    continue;
                }
                
                int cnt = 0;
                for (int j = 0; j < m; j++) {
                    if (st >> j & 1) {
                        cnt++;
                    }
                }
                
                if (i == 0) {
                    dp[i][st] = cnt;
                } else {
                    for (int prevSt = 0; prevSt < (1 << m); prevSt++) {
                        if (dp[i - 1][prevSt] == -1) {
                            continue;
                        }
                        bool validPrev = true;
                        for (int j = 0; j < m; j++) {

                            if (st >> j & 1) {
                                if (j > 0 && (prevSt >> (j - 1) & 1)) {
                                    validPrev = false;
                                    break;
                                }
                                if (j < m - 1 && (prevSt >> (j + 1) & 1)) {
                                    validPrev = false;
                                    break;
                                }
                            }
                        }
                        if (validPrev) {
                            dp[i][st] = max(dp[i][st], dp[i - 1][prevSt] + cnt);
                        }
                    }
                }
                
                
                if (i == n - 1) {
                    result = max(result, dp[i][st]);
                }
            }
        }
        return result;
    }
};
