class Solution {
public:
    int maxStudents(vector<vector<char>>& s) {
        int n = s.size();
        int m = s[0].size();
        int maxn = 1<<m;
        int dp[2][256];
        for(int i=0;i<256;i++){
           dp[0][i]=dp[1][i]=0;
        }
        for(int i=0;i<n;i++){
            for(int k=0;k<maxn;k++){
                bool f = true;
                int cnt = 0;
                    for(int j=0;j<m;j++){
                        if((k>>j)&1) cnt++;   
                    }
                for(int j=0;j<m;j++){
                    if(((k>>j)&1)&&s[i][j]=='#'){
                        f = false;
                        break;
                    }
                    if(((k>>j)&1)&&((k>>(j+1))&1)){
                        f = false;
                        break;
                    }
                    if(((k>>(j+1))&1)&&((k>>(j))&1)){
                        f = false;
                        break;
                    }
                }
                if(f)
                for(int w=0;w<maxn;w++){
                    f = true;
                    for(int j=0;j<m;j++){
                        if(((w>>j)&1)&&((k>>(j+1))&1)){
                            f = false;
                            break;
                        }
                         if(((w>>(j+1))&1)&&((k>>(j))&1)){
                            f = false;
                             break;
                        }
                    }
                    if(f){
                        
                        dp[i%2][k]=max(dp[(i+1)%2][w]+cnt,dp[i%2][k]);
                    }

                }
                
            }
        }
        int ans = 0;
        for(int i=0;i<maxn;i++){
            ans=max(dp[(n-1)%2][i],ans);
        }
        return ans;
    }
};
