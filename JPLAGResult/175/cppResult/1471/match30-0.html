class Solution {
public:
  int maxStudents(vector<vector<char>>& seats) {
    int m = seats.size();
    int n = seats[0].size();
    vector<int> cur(1<<n);
    
    for (auto seat: seats) {
      vector<int> next(1<<n);
      int seat_mask = 0;
      for (auto t: seat) seat_mask = seat_mask * 2 + (t == '#');
      for (int to = 0; to < (1<<n); to++) {
        if (to&(to>>1)) continue;
        if (to&seat_mask) continue;
        for (int from = 0; from < (1<<n); from++) {
          if (from & (to>>1)) continue;
          if ((from>>1) & to) continue;
          next[to] = max(next[to], cur[from] + __builtin_popcount(to));
        }
      }
      cur = next;
    }
    return *max_element(cur.begin(), cur.end());
  }
};
