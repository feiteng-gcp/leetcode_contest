class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int m = seats.size(), n = seats[0].size();
        vector<int> brk(m);
        for(int i = 0; i < m; i++) {
            int x = 0;
            for(int j = 0; j < n; j++) {
                if(seats[i][j] == '#') x |= (1 << j);
            }
            brk[i] = x;
        }
        

        vector<int> cnt_1(1 << n);
        for(int x = 0; x < (1<<n); x++) {
            for(int i = 0; i < 9; i++) {
                if(x & (1 << i)) cnt_1[x]++;
            }
        }
        
        // dp[i][j]: 行i填j的情况下，行[i,m-1]最大学生数
        vector<vector<int>> dp(m, vector<int>(1 << n));
        for(int i = 0; i < (1<<n); i++) {
            if((i & brk[m-1]) || (i & (i << 1))) continue;
            dp[m-1][i] = cnt_1[i];
            // cout << "i: " << i << ", dp[i]: " << dp[m-1][i] << endl;

        }
        
        for(int j = m-2; j >= 0; j--) {
            for(int i = 0; i < (1<<n); i++) {
                if((i & brk[j]) || (i & (i << 1))) continue;

                dp[j][i] = cnt_1[i];
                for(int k = 0; k < (1<<n); k++) {
                    if(dp[j+1][k] > 0 && (i & (k << 1)) == 0 && ((i << 1) & k) == 0) {
                        dp[j][i] = max(dp[j][i], cnt_1[i] + dp[j+1][k]);
                    }
                }
            }
        }
        return *max_element(dp[0].begin(), dp[0].end());
    }
    
};
