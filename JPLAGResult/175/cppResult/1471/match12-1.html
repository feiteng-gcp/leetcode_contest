class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int N = seats.size();
        int M = seats[0].size();
        int masks[12];
        memset(masks, 0, sizeof(masks));
        

        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (seats[i][j] == '#') {
                    masks[i] |= 1 << j;
                }
            }
        }
        
        static int dp[12][256];
        memset(dp, 0, sizeof(dp));
        
        int res = 0;
        

        for (int i = N-1; i >= 0; i--) {
            for (int seating = 0; seating < (1 << M); seating++) {
                if (seating & masks[i]) continue;
                
                int pop = 0; // __builtin__popcount(seating);
                for (int i = 0; i < M; i++) pop += (seating >> i) & 1;

                int& best = dp[i][seating];
                
                for (int prev = 0; prev < (1 << M); prev++) {
                    if (prev & masks[i+1]) continue;
                    
                    //if (prev & seating) continue;
                    if (prev & (seating << 1)) continue;
                    if ((prev << 1) & seating) continue;
                    if (seating & (seating << 1)) continue;
                    if ((seating << 1) & seating) continue;
                    
                    best = max(best, dp[i+1][prev] + pop);
                }
                
                if (i == 0) {
                    res = max(res, best);
                }
            }
        }
       
        
        return res;
    }
};
