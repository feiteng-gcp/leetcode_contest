int dp[9][1<<8];

class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        memset(dp, 0, sizeof(dp));
        int seat[8] = {0};
        int r = seats.size(), c = seats[0].size();
        for(int i=0;i<r;++i){
            for(int j=0;j<c;++j){
                if(seats[i][j] == '#'){
                    seat[i] |= 1<<j;
                }
            }
        }
        for(int i=1;i<=r;++i){
            for(int k=0;k<(1<<c);++k){
                if((k&seat[i-1]) || ((k>>1)&k) || ((k<<1)&k)) continue;
                int x = __builtin_popcount(k);
                for(int j=0;j<(1<<c);++j){
                    if(((j>>1)&k) || ((j<<1)&k)) continue;
                    dp[i][k] = max(dp[i][k], dp[i-1][j] + x);
                }
            }
        }        
        int ans = 0;
        for(int i=0;i<(1<<c);++i) ans = max(ans, dp[r][i]);
        return ans;
    }
};
