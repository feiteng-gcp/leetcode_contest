class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        const int m = seats.size();
        const int n = seats[0].size();
        
        vector<int> forbids(m);
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (seats[i][j] == '#') {
                    forbids[i] |= (1 << j);
                }
            }
        }
        
        vector<int> layer(1 << n, -1);
        layer[0] = 0;
        for (int state = 1; state < layer.size(); ++state) {
            int bit = (state & (-state));
            if (!(state & (bit << 1)) && layer[state & ~bit] >= 0) {
                layer[state] = layer[state & ~bit] + 1;
            }
        }
        
        vector<int> last_dp(1 << n, -1);
        vector<int> cur_dp(1 << n, -1);
        for (int state = 0; state < cur_dp.size(); ++state) {
            if (!(forbids[0] & state)) {
                cur_dp[state] = layer[state];
            }
        }
        for (int i = 1; i < m; ++i) {
            swap(cur_dp, last_dp);
            for (int state = 0; state < cur_dp.size(); ++state) {
                if (forbids[i] & state) {
                    cur_dp[state] = -1;
                    continue;
                }
                cur_dp[state] = 0;
                for (int last_state = 0; last_state < last_dp.size(); ++last_state) {
                    if (last_dp[last_state] >= 0 && layer[last_state | state] >= 0) {
                        cur_dp[state] = max(cur_dp[state], last_dp[last_state] + layer[state]);
                    }
                }
            }
        }
        return *max_element(cur_dp.begin(), cur_dp.end());
    }
};
