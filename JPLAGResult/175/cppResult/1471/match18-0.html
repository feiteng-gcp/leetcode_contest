class Solution {
public:
    
    int dp[9][(1<<9)];
    int n,m;
    
    bool ok(int mask,vector<char> v)
    {
        for(int i=0;i<v.size();i++)
            if(mask&(1<<i))
            {
                if(v[i]=='#')
                    return 0;
            }
        return 1;
    }
    
    int cnt(int mask)
    {
        int res=0;
        for(int i=0;i<m;i++)
            if(mask&(1<<i))
                res++;
        return res;
    }
    
    bool ok(int mask1,int mask2)
    {
        for(int i=0;i<m;i++)
        {
            if(mask1&(1<<i))
            {
                if(i && (mask2&(1<<(i-1))))
                    return 0;
                if(i+1<m && (mask2&(1<<(i+1))))
                    return 0;
            }
        }
        return 1;
    }
    
    bool ok(int mask)
    {
        for(int i=1;i<m;i++)
            if((mask&(1<<i)) && (mask&(1<<(i-1))))
                return 0;
        return 1;
    }
    
    int maxStudents(vector<vector<char>>& seats) {
        memset(dp,0,sizeof(dp));
        
        reverse(seats.begin(),seats.end());
        n=seats.size();
        m=seats[0].size();
        
        for(int mask=0;mask<(1<<m);mask++)
            if(ok(mask,seats[0]) && ok(mask))
            {
                dp[0][mask]=cnt(mask);
            }

        for(int i=1;i<n;i++)
            for(int mask1=0;mask1<(1<<m);mask1++)
                for(int mask2=0;mask2<(1<<m);mask2++)
                {
                    if(ok(mask1,seats[i-1]) && ok(mask2,seats[i]) && ok(mask1) && ok(mask2) && ok(mask1,mask2))
                        dp[i][mask2]=max(dp[i][mask2],dp[i-1][mask1]+cnt(mask2));
                }
        
        int ans=0;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<(1<<m);j++)
            {
                ans=max(ans,dp[i][j]);
                
            }
        }
        return ans;
        
    }
};
