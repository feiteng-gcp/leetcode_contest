class Solution {
public:
    int f[10][(1<<8)]={0};
    int is1(int n,int k){
        return ((n>>k)&1);
    }
    int maxStudents(vector<vector<char>>& seats) {
        int m=seats.size(),n=seats[0].size(),all=(1<<n);
        //init first line
        for(int i=0;i<all;++i){

            int ok=1,stu=0;
            for(int j=0;j<n;++j){
                if(is1(i,j)){
                    stu++;
                    if(seats[0][j]=='#' || (j-1>=0&&is1(i,j-1)) ||(j+1<n&&is1(i,j+1)) ) {ok=0;break;}
                }
            }
            if(ok) f[0][i]=stu;

        }
        
        for(int i=0;i<m-1;++i){

            for(int j=0;j<all;++j){
               // if(f[i][j]){
                    for(int jj=0;jj<all;++jj){
                        bool ok=1;
                        int stu=0;
                        for(int k=0;k<n;++k){
                            if(is1(jj,k)){
                                stu++;
                                if(seats[i+1][k]=='#' ||   (k-1>=0&&is1(jj,k-1)) ||(k+1<n&&is1(jj,k+1))||
                                  (k-1>=0&&is1(j,k-1)) || (k+1<n&&is1(j,k+1))  )
                                {ok=0;break;}
                            }
                        }
                        if(ok){

                               f[i+1][jj]=max(f[i+1][jj],f[i][j]+stu);}
                    }
                //}
            }
        }
        
        int ans=0;
        for(int i=0;i<all;++i) 
            ans=max(ans,f[m-1][i]);
        return ans;
    }
};
