class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int n = seats.size();
        if ( n == 0 ) return 0 ;
        int m = seats[0].size();
        if ( m == 0 ) return 0 ;
        int cnt = 1<<m ;
        vector<int> fa(cnt,0),fb(cnt,0); 
        
        for (int i = 0 ; i < cnt ; i ++) {
            int tmp = 0; 
            bool chk = true; 
            for ( int j = 0 ; j < m ; j ++ ) {
                if ( (i&(1<<j)) == 0 ) continue ;
                if (seats[0][j] == '#') chk = false ;
                if ( (j>0) && ( (i&(1<<(j-1))) != 0) ) chk=false; 
                tmp++;
            }
            if (chk) fa[i] = max(fa[i],tmp); 
            for ( int j = 0 ; j < m; j ++ ) if (fa[i]>fa[i|(1<<j)]) fa[i|(1<<j)]=fa[i]; 
        }
        
        for ( int l = 1 ; l < n ; l ++ ) {
            for ( int i = 0 ; i < cnt ; i ++ ) fb[i] = 0 ; 
            for ( int i = 0 ; i < cnt ; i ++ ) {
                bool chk = true; 
                int num = 0 ; 
                for ( int j = 0 ; j < m ; j ++ ) {
                    if ( (i&(1<<j)) == 0 ) continue ;
                    if ( seats[l][j] == '#') chk = false;
                    if ( (j>0) && ( (i&(1<<(j-1))) != 0)) chk = false;
                    num ++ ; 
                }
                if (not chk) continue ; 
                
                int tmp = (1<<m) -1;  
                tmp = tmp - (tmp&(i<<1));
                tmp = tmp - (tmp&(i>>1)); 
                
                fb[i] = max(fb[i],fa[tmp]+num);  
                //cout<<i<<" "<<tmp<<" "<<fb[i]<<endl;;
            }
            for ( int i = 0 ; i < cnt ; i ++ ) {
                for ( int j = 0 ; j < m; j ++ ) if (fb[i]>fb[i|(1<<j)]) fb[i|(1<<j)]=fb[i];     
            }
            for ( int i = 0 ; i < cnt ; i ++ ) fa[i] = fb[i]; 
        }
        int ans = 0 ; 
        for ( int i = 0 ; i < cnt ; i ++ ) ans = max(fa[i],ans);
        return ans; 
    }
};
