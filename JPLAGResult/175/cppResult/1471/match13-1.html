class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int n=seats.size(), m=seats[0].size();
        vector<int> bs={0};
        for(int i=0; i<n; ++i) {
            bs.push_back(0);
            for(int j=0; j<m; ++j)
                if(seats[i][j]=='.')
                    bs.back()|=1<<j;
        }
        vector<vector<int>> dp(n+1, vector<int>(1<<m, -1e9));

        dp[0][0]=0;
        for(int i=0; i<n; ++i) {
            for(int j=0; j<1<<m; ++j) {
                if((j&bs[i])!=j)
                    continue;
                for(int k=0; k<1<<m; ++k) {
                    if((k&bs[i+1])!=k)
                        continue;
                    bool ok=1;
                    for(int l=0; l+1<m&&ok; ++l)
                        ok=(k>>l&1^1)||(k>>l+1&1^1);
                    for(int l=0; l+1<m&&ok; ++l)
                        ok=(j>>l&1^1)||(k>>l+1&1^1);
                    for(int l=0; l+1<m&&ok; ++l)
                        ok=(j>>l+1&1^1)||(k>>l&1^1);

                    if(!ok)
                        continue;
                    dp[i+1][k]=max(dp[i+1][k], dp[i][j]+__builtin_popcount(k));
                }
            }
        }
        int ans=0;
        for(int i=0; i<1<<m; ++i)
            ans=max(ans, dp[n][i]);
        return ans;
    }
};
