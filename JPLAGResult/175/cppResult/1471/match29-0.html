class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        // if (seats.size())
        int R = seats.size();
        int C = seats[0].size();
        int ALLC = 1<<C;
        int ret = 0;
        
        vector< vector <int> > dp(R+1, vector<int>(ALLC, -1));
        dp[0][0] = 0;
        
        for (int i = 1; i <= R; ++i) {
            for (int c = 0; c < ALLC; ++c) {
                if (dp[i-1][c] != -1) {
                    for (int nc = 0; nc < ALLC; ++nc) {
                        //check validity
                        bool pass = true;
                        int count = 0;
                        int prev = 0;
                        
                        for (int k = 0; k < C; ++k) {
                            if (nc & (1<<k)) {
                                if (seats[i-1][k] == '#' || prev) {
                                    pass = false;
                                    break;
                                }
                                prev = 1; //no left right
                                ++count;
                            } else {
                                prev = 0;
                            }
                        }
                        
                        if (i > 1) {                            
                            for (int k = 0; k < C && pass; ++k) {
                                int l = k-1;
                                int r = k+1;                                
                                
                                if (nc & (1<<k)) {                             
                                    if (l >= 0 && (c & (1<<l))) {                                     
                                        pass = false;
                                        break;   
                                    }
                                    if (r < C && (c & (1<<r))) {
                                        pass = false;
                                        break;  
                                    }
                                }
                            }
                        }
                        
                        if (pass) {
                            //cout << "OK ! " << i << " " << nc << " " << c << endl; 
                            dp[i][nc] = max(dp[i][nc], dp[i-1][c] + count);
                        }
                    }
                }
            }
        }
        
        for (int i = 0; i < ALLC; ++i) {
            ret = max(ret, dp[R][i]);
        }
            
        return ret;
    }
};
