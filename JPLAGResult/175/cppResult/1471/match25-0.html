int f[8][1<<8];
class Solution {
public:
    int maxStudents(vector<vector<char>>& a) {
        int n = a.size();
        int m = a[0].size();
        memset(f,-1,sizeof(f));
        for(int i=0;i<(1<<m);i++){
            bool sign = 0;
            int cnt = 0;
            for(int j=0;j<m;j++){
                if(a[0][j]=='#'&&(i&(1<<j))){
                    sign=1;
                    break;
                }
                if(j&&(i&(1<<j))&&(i&(1<<(j-1)))){
                    sign=1;
                    break;
                }
                if(i&(1<<j))cnt++;
            }
            if(!sign){
                f[0][i]= cnt;
            }
        }
        for(int i=1;i<n;i++)
            for(int j=0;j<(1<<m);j++){
                if(f[i-1][j]==-1)continue;
                for(int k=0;k<(1<<m);k++){
                    bool sign = 0;
                    int cnt =0;
                    for(int r=0;r<m;r++){
                        if(a[i][r]=='#'&&(k&(1<<r))){
                            sign=1;
                            break;
                        }
                        if(r&&(k&(1<<r))&&(k&(1<<(r-1)))){
                            sign=1;
                            break;
                        }
                        if(r&&(k&(1<<r))&&(j&(1<<(r-1)))){
                            sign=1;
                            break;
                        }
                        if(r+1<m&&(k&(1<<r))&&(j&(1<<(r+1)))){
                            sign=1;
                            break;
                        }
                        if(k&(1<<r))cnt++;
                    }
                    if(!sign)f[i][k]=max(f[i][k],f[i-1][j]+cnt);
                }
            }
        int ret=0;
        for(int i=0;i<(1<<m);i++)ret=max(ret,f[n-1][i]);
        return ret;
    }
};
