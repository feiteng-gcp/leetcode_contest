class Solution {
public:
    
    bool valid_status(int s, vector<char>& seat) {
        int n = seat.size();
        for(int i = 0; i < n; i++) {
            if(seat[i] == '#' && (s & (1 << (n - 1 - i)))) return false;
        }
        return true;
    }
    
    bool judge(int s, int p, int n) {

        vector<vector<int>> seat(2, vector<int>(n, 0));
        for(int i = 0; i < n; i++) {
            if(p & (1 << (n - 1 - i))) {
                seat[0][i] = 1;
            }
        }
        
        for(int i = 0; i < n; i++) {
            if(s & (1 << (n - 1 - i))) {
                seat[1][i] = 1;
            }
        }
        
        for(int i = 0; i < n; i++) {
            if(i > 0 && seat[1][i-1] == 1 && seat[1][i] == 1) return false;
            if(i < n - 1 && seat[1][i] == 1 && seat[1][i+1] == 1) return false;
            if(i > 0 && seat[0][i-1] == 1 && seat[1][i] == 1) return false;
            if(i < n - 1 && seat[0][i+1] == 1 && seat[1][i] == 1) return false;
        }
        return true;
    }
    

    int row_num(int s, int n) {
        int ans = 0;
        for(int i = 0; i < n; i++) {
            if(s & (1 << (n - 1 - i))) ans++;
        }
        return ans;
    }
    
    int maxStudents(vector<vector<char>>& seats) {
        int m = seats.size(), n = seats[0].size();
        vector<vector<int>> dp(m, vector<int>((1<<n), 0));
        
        for(int s = 0; s < (1 << n); s++) {
            if(!valid_status(s, seats[0])) continue;
            if(!judge(s, 0, n)) continue;
            int cnt = row_num(s, n);
            dp[0][s] = cnt;
        }
        
        for(int i = 1; i < m; i++) {
            for(int s = 0; s < (1 << n); s++) {
                if(!valid_status(s, seats[i])) continue;
                
                for(int p = 0; p < (1 << n); p++) {
                    if(!valid_status(p, seats[i-1])) continue;
                    if(judge(s, p, n)) {
                        int cnt = row_num(s, n);
                        dp[i][s] = max(dp[i][s], dp[i-1][p] + cnt);
                    }
                }

            }
        }
        
        int ans = 0;
        for(int s = 0; s < (1 << n); s++) {
            ans = max(ans, dp[m-1][s]);
        }
        return ans;
    }
};
