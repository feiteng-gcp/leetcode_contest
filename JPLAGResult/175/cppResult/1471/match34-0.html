int ds[4][2] = {{-1,-1}, {-1,1}, {0,-1}, {0,1}};

class Solution {
public:
    int R, C;
    vector<vector<char>> g;

    bool can(int r, int s){
        for(int i=0; i<C; i++){
            if(((1<<i)&s) == 0) continue;
            if(((1<<i) & s) > 0 && g[r][i]=='#') return false;

            if(i-1>=0 && ((1<<(i-1)) & s)>0) return false;

            if(i+1<C && ((1<<(i+1)) & s)>0) return false;
        }
        return true;
    }

    bool canp(int p, int s){
        for(int i=0; i<C; i++){
            if((s & (1<<i)) > 0){
                if(i-1>=0 && ((1<<(i-1)) & p)>0) return false;
                if(i+1<C && ((1<<(i+1)) & p)>0) return false;
            }
        }
        return true;
    }
    
    int cnt(int s){
        int a = 0;
        for(int i=0; i<C; i++){
            if((s & (1<<i)) > 0) a++;
        }
        return a;
    }

    int maxStudents(vector<vector<char>>& seats) {
        R = seats.size(); C = seats[0].size();
        g = seats;
        vector<vector<int>> dp(R, vector<int>(1<<C, 0));
        for(int s=0; s<(1<<C); s++){
            if(can(0, s)){
                dp[0][s] = cnt(s);
            }
            
            //cout<<s<<" "<<dp[0][s]<<endl;
        }

        for(int i=1; i<R; i++){
            for(int s=0; s<(1<<C); s++){
                if(!can(i, s)) continue;
                int res = 0;
                for(int p=0; p<(1<<C); p++){
                    if(canp(p, s)){
                        res =max(res, dp[i-1][p] + cnt(s));
                    }
                }
                dp[i][s] = res;
            }
        }

        int res = 0;
        for(int s=0; s<(1<<C); s++){
            res = max(res, dp[R-1][s]);
        }

        return res;

    }
};
