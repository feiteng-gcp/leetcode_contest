class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int m = seats.size();
        int n = seats[0].size();
        vector<int> dp((1 << n), 0);
        for (int i = 0; i < m; ++i) {
            vector<int> new_dp((1 << n), 0);
            int row_mask = 0;
            for (int j = 0; j < n; ++j) {
                if (seats[i][j] == '#') row_mask |= (1 << j);
            }
            for (int new_mask = 0; new_mask < (1 << n); new_mask++) {
                if (new_mask & row_mask) continue;
                if (new_mask & (new_mask << 1)) continue;
                int mask_count = 0;
                int x = new_mask;
                while (x) {
                    if (x % 2) ++mask_count;
                    x /= 2;
                }
                new_dp[new_mask] = mask_count;

                for (int mask = 0; mask < (1 << n); ++mask) {
                    // Checks if compatible.
                    if (mask & (new_mask << 1)) continue;
                    if (new_mask & (mask << 1)) continue;
                    new_dp[new_mask] = max(new_dp[new_mask], dp[mask] + mask_count);
                }
            }
            dp = new_dp;
        }
        int ans = 0;
        for (int x : dp) ans = max(ans, x);
        return ans;
    }
};
