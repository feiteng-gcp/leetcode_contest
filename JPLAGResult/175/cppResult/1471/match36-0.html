class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        vector<int> bit_seats;
        for (const auto& row : seats) {
            int n = 0;
            for (char c : row) {
                if (c == '#') {
                    n = n * 2 + 1;
                } else {
                    n *= 2;
                }
            }
            bit_seats.push_back(n);
        }
        vector<int> max_s(1 << seats[0].size());
        vector<int> temp(max_s);
        for (int i = 0; i < max_s.size(); i++) {
            max_s[i] = checkRow(i, bit_seats.back());
        }
        int row = bit_seats.size() - 1;
        for (row--; row >= 0; row--) {
            swap(max_s, temp);
            print(temp);
            max_s = vector<int>(1 << seats[0].size(), -1);
            for (int i = 0; i < max_s.size(); i++) {
                for (int j = 0; j < max_s.size(); j++) {
                    if (temp[j] != -1) {
                        int c = check2Rows(i, j, bit_seats[row]);
                        if (c == -1) {
                            continue;
                        }
                        if (max_s[i] == -1) {
                            max_s[i] = c + temp[j];
                        } else {
                            max_s[i] = max(max_s[i], c + temp[j]);
                        }
                    }
                }
            }
        }
        int m = -1;
        for (int n : max_s) {
            m = max(m, n);
        }
        print(max_s);
        return m == -1 ? 0 : m;
    }
private:
    void print(const vector<int>& v) {
        return;
        for (int n : v) {
            cout << n << " ";
        }
        cout << endl;
    }
    int check2Rows(int front, int back, int bit_seats) {
        int front_count = checkRow(front, bit_seats);
        if (front_count == -1) {
            return -1;
        }
        if (front & (back << 1)) {
            return -1;
        }
        if (front & (back >> 1)) {
            return -1;
        }
        return front_count;
    }
    int checkRow(int bits, int bit_seats) {
        if (bits & bit_seats) {
            return -1;
        }
        if (bits & (bits << 1)) {
            return -1;
        }
        return count(bits);
    }
    int count(int bits) {
        int c = 0;
        while (bits) {
            c += bits & 1;
            bits >>= 1;
        }
        return c;
    }
};
