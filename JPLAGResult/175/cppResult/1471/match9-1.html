class Solution {
public:

    int cnt(int x)
    {
        int c=0;
        while(x)
        {
            c+=x&1;
            x>>=1;
        }
        return c;
    }
    
    int maxStudents(vector<vector<char>>& seats) {
        int m=seats.size(),n=seats[0].size();
        int mask[m];
        for(int i=0;i<m;i++)
        {
            int temp=0;
            for(auto x:seats[i])temp=temp*2+(x=='#'?1:0);
            mask[i]=temp;
        }
        int Y=1<<n,f[m][1<<n];
        for(int i=0;i<Y;i++)f[0][i]=((i&mask[0])||(i&(i>>1)))?0:cnt(i);
        for(int i=1;i<m;i++)
        {
            for(int j=0;j<Y;j++)
            {
                f[i][j]=0;
                if((j&mask[i])||(j&(j>>1)))continue;
                for(int k=0;k<Y;k++)
                {
                    if((k&((j<<1)|(j>>1)))==0)f[i][j]=max(f[i][j],f[i-1][k]);
                }
                f[i][j]+=cnt(j);

            }
        }
        int ans=0;
        for(int i=0;i<Y;i++)ans=max(ans,f[m-1][i]);
        return ans;
    }
};
