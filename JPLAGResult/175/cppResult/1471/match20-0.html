class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int N = seats.size();
        int M = seats[0].size();
        vector<int> last, next;
        
        next = vector<int>(1<<M);
        for (int i=0;i<N;++i) {
            last = next;
            next = vector<int>(1<<M);
            for (int j=0;j<(1<<M);++j) {
                int badmask = (j<<1) | (j >> 1);
                int bc = 0;
                bool ok = true;
                if (j & badmask) {
                    ok = false;
                }
                for (int k=0;k<M;++k) {
                    if (j & (1<<k)) {
                        if (seats[i][k]=='#') {
                            ok = false;
                        }
                        ++bc;
                    }
                }
                if (!ok) continue;
                
                for (int k=0;k<(1<<M);++k) {
                    if (k & badmask) {
                        continue;
                    }
                    next[j] = max(next[j], last[k]+bc);
                }
            }
        }
        int best = 0;
        for (int i=0;i<(1<<M);++i) {
            best = max (best, next[i]);
        }
        
        return best;
    }
};
