class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
        int n = seats.size();
        int m = seats[0].size();
        int dp[10][1<<8];
        memset(dp, 0, sizeof(dp));

        int ans = 0;
        for (int i = 0; i < n; ++ i) {
            for (int j = 0; j < (1<<m); ++ j) {
      
                bool flag = true;
                for (int k = 0; k < m; ++ k) {
                    if (seats[i][k] == '#' && (j & (1<<k))) {
                        flag = false;
                        break;
                    }
                    if (k>0 && (j & (1<<k)) && (j & (1<<(k-1)))) {
                        flag = false;
                        break;
                    }
                    if (k+1<m && (j & (1<<k)) && (j & (1<<(k+1)))) {
                        flag = false;
                        break;
                    }
                }
                if (!flag) continue;
                
                for (int k = 0; k < (1<<m); ++ k) {
                    flag = true;
                    for (int t = 0; t < m; ++ t) if (j & (1<<t)) {
                        if (t > 0 && (k&(1<<(t-1)))) {
                            flag = false;
                            break;
                        }
                        if (t + 1 < m && (k&(1<<(t+1)))) {
                            flag = false;
                            break;
                        }
                    }
                    if (flag) dp[i][j] = max(dp[i][j], (i == 0 ? 0 : dp[i - 1][k]) + __builtin_popcount(j));
                }
                if (i == n - 1) ans = max(ans, dp[i][j]);
            }
        }
        return ans;
    }

};
