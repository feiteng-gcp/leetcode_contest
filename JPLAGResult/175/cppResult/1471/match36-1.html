class Solution {
    bool ok(int n, vector<char>& seats, int last, int now) {
        for (int i = 0; i < n; ++i) {

            if (seats[i] == '#' && (now & 1 << i)) {
                return 0;
            }
            if (i > 0 && (now & 1 << i - 1) && (now & 1 << i)) {
                return 0;
            }
            if (i > 0 && (now & 1 << i) && (last & 1 << i - 1)) {
                return 0;
            }
            if (i +1 < n && (now & 1 << i) && (last & 1 << i + 1)) {
                return 0;
            }

        }
        return 1;
    }
    
    int count(int x) {
        int ans = 0;
        while (x) {
            ++ans;
            x -= x & -x;
        }
        return ans;
    }
public:
    int maxStudents(vector<vector<char>>& seats) {
        int m = seats.size(), n = seats[0].size();
        int f[m][1 << n], ans = 0;
        memset(f, 0, sizeof(f));
        for (int i = 0; i < 1 << n; ++i) {
            if (ok(n, seats[0], 0, i)) {
                f[0][i] = count(i);
            }
        }
        for (int i = 1; i < m; ++i) {

            for (int j = 0; j < 1 << n; ++j) {
                for (int k = 0; k < 1 << n; ++k) {
                    if (ok(n, seats[i], j, k)) {
                        f[i][k] = max(f[i][k], f[i - 1][j] + count(k));
                        ans = max(f[i][k], ans);
                    }
                }
            }
        }
        return ans;
    }
};
