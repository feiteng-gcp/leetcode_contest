class Solution {
    
public:
    int maxStudents(vector<vector<char>>& seats) {
        int n = seats[ 0 ].size();
        int m = seats.size();
        vector< vector< int > > dp( m, vector< int >( 1 << n, 0 ) );

        for( int i = 0; i < ( 1 << n ); i ++ )
        {
            if( Check( 0, i, seats ) ) dp[ 0 ][ i ] = getnum( i );
        }
        int result = 0;
        for( int i = 1; i < m; i ++ )
        {
            for( int j = 0; j < ( 1 << n ); j ++ )
            {
                if( Check( i, j, seats ) == false ) continue;
                for( int k = 0; k < ( 1 << n ); k ++ )
                {
                    if( Check( i - 1, k, seats ) == false ) continue;
                    if( Judge( j, k ) == false ) continue;
                    dp[ i ][ j ] = max( dp[ i ][ j ], dp[ i - 1 ][ k ] + getnum( j ) ); 
                }
                if( i == m - 1 ) result = max( result, dp[ i ][ j ] );
            }

        }
        return result;
    }
    
    
    int getnum( int num )
    {
        int cnt = 0;
        for( int i = 9; i >= 0; i -- )
        {
            if( ( num >> i ) & 1 ) cnt ++;
        }
        return cnt;
    }
    
    bool Check( int dep, int val, vector<vector<char>>& seats )

    {
        for( int i = 9; i >= 0; i -- )
        {
            if( ( val >> i ) & 1 )
            {
                if( seats[ dep ][ i ] == '#' ) return false;
                if( ( val >> ( i + 1 ) ) & 1 ) return false;
            }
        }
        return true;
    }
    
    bool Judge( int j, int k )
    {
        vector< int > v;
        for( int i = 9; i >= 0; i -- )
        {
            if( ( k >> i ) & 1 ) v.push_back( i );
        }
        for( auto x: v )
        {
            int l = x - 1;
            if( l >= 0 && ( ( j >> l ) & 1 ) ) return false;
            int r = x + 1;
            if( ( j >> r ) & 1 ) return false;
        }
        return true;
    }
    
};
