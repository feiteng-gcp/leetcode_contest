class Solution {
public:
	int n, m, currow;
	int a[11];
	int f[11][1 << 9];

	void DFSinit(int cur, int last, int stat, int val) {
		if (cur == n) {
			f[0][stat] = val;
			return;
		}
		if (((a[0] >> cur) & 1) == 0 && !last) {
			DFSinit(cur + 1, 1, stat | 1 << cur, val + 1);
		}
		DFSinit(cur + 1, 0, stat, val);
	}

	void DFSupdate(int cur, int last, int stat, int val, int lstat) {
		if (cur == n) {
			f[currow][stat] = max(f[currow][stat], f[currow - 1][lstat] + val);
			return;
		}
		DFSupdate(cur + 1, 0, stat, val, lstat);
		if (((a[currow] >> cur) & 1) == 0 && !last) {
			if (cur > 0 && ((lstat >> (cur - 1)) & 1)) {
				return;
			}
			if ((lstat >> (cur + 1)) & 1) {
				return;
			}
			DFSupdate(cur + 1, 1, stat | (1 << cur), val + 1, lstat);
		}
	}

    int maxStudents(vector<vector<char> >& seats) {
		memset(f, 0, sizeof(f));
		n = seats[0].size();
		m = seats.size();
		for (int i = 0; i < m; ++i) {
			a[i] = 0;
			for (int j = 0; j < n; ++j) {
				if (seats[i][j] == '#') {
					a[i] |= (1 << j);
				}
			}
		}
		DFSinit(0, 0, 0, 0);
		for (int i = 1; i < m; ++i) {
			currow = i;
			for (int j = 0; j < (1 << n); ++j) {
				DFSupdate(0, 0, 0, 0, j);
			}
		}
		int ans = 0;
		for (int i = 0; i < (1 << n); ++i) {
			ans = max(ans, f[m - 1][i]);
		}
		return ans;
    }
};
