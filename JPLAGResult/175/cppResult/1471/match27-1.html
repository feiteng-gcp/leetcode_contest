const int N = 8;
int dp[1 << N][N];
class Solution {
    int m,n;
    vector<vector<int> > b;
    vector<vector<int> > t;
    bool legal(int all) {
        for (int i = 1; i < n; ++i) {

            if (((all >> i) & 1) && (all >> i - 1 & 1)) { return false;}
        }
        return true;
    }
    int init(int all) {
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            if (all >> i & 1) {
                ++ret;
                t[0][i] = 2;
            }
        } 
        return ret;
    }
    
    int cal(int n) {
        int ret = 0;
        while(0 != n) {
            ++ret;
            n = n & n - 1;
        }
        return ret;
    }
    
    bool conflict(int prev, int cur) {
        if ((cur >> 1 & prev) || (cur << 1 & prev)) { return true; }
        return false;
    }
    
    bool ok(int r, int c) {
        if (0 != t[r][c]) { return false; }
        if (c > 0) {
            if (t[r][c - 1] == 2 || 2 == t[r - 1][c - 1]) { return false; }
        }
        if (c < n - 1) {
            if (2 == t[r][c + 1] || 2 == t[r - 1][c + 1]) { return false;}

        }
        return true;
    }
    
    bool isok(int s, int r) {
        for (int i = 0; i < n; ++i) {
            if (b[r][i] == 1 && (1 == (s >> i & 1))) { return false;}
        }
        return true;
    }

public:

    int maxStudents(vector<vector<char>>& seats) {
        m = seats.size();
        n = seats[0].size();
        b = vector<vector<int> >(m, vector<int>(n, 0));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if ('.' == seats[i][j]) {
                    b[i][j] = 0;
                }
                else { b[i][j] = 1; }
            }
        }
        // int first = 0;
        // for (int j = 0; j < n; ++j) {
        //     if (b[0][j] == 1) { first = first | (1 << j); }
        // }
        // int all = ((1 << n) - 1) ^ first;
        // // cout << "first: " << first <<  " all: " << all << endl;
        // int mask = all;
        // int sub = mask;
        // int ans = 0;
        // do {
        //     ans = max(ans, solve(sub));
        //     sub = (sub - 1) & mask;
        //     // cout << sub << ": " << solve(sub) << endl;
        // }while(0 != sub);
        
        int all = (1 << n);
        memset(dp, 0, sizeof(dp));
        for (int r = 0; r < m; ++r) {
            for (int i = 0; i < all; ++i) {
                if (isok(i, r) && legal(i)) {
                    if (0 != r) {
                        for (int j = 0; j < all; ++j) {
                            if ( legal(j) && isok(j, r - 1) && !conflict(i, j)) {
                                dp[i][r] = max(dp[i][r], cal(i) + dp[j][r - 1]);
                            }
                        }
                    }
                    else {
                        dp[i][r] = cal(i);
                    }
                }

            }
        }
        
        int ans = 0;
        for (int i = 0; i < all; ++i) {
            ans = max(ans, dp[i][m - 1]);
        }
        return ans;
    }
};
