class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) {
       // seats = vector<vector<char>>(8, vector<char>(8, '#'));
        int m = seats.size(), n = seats[0].size();
        vector<vector<int>> dp(m, vector<int>((1 << n), 0));
        
            for (int i = m - 1; i >= 0; --i) {     
                for (int k = 0; k < 1 << n; ++k) {
                    int count = 0;
                    bool flag1 = true;
                    for (int l = 0; l < n; ++l) {
                        if ((k & (1 << l)) && seats[i][l] == '#') {flag1 = false; break;}
                        if (l > 0 && (k & (1 << l)) && (k & (1 << (l - 1)))) {flag1 = false; break;}
                        if (k & (1 << l)) ++count;
                    }    
                    if (!flag1) continue;
                    
                    for (int j = 0; j < 1 << n; ++j) {
                        bool flag2 = true;
                        for (int l = 0; l < n; ++l) {
                            if (k & (1 << l)) {
                                if (l > 0 && (j & (1 << (l - 1)))) {flag2 = false;break;}
                                if (l < n - 1 && (j & (1 << (l + 1)))) {flag2 = false;break;}
                            }   
                        }                       
                        if (flag2) dp[i][j] = max(dp[i][j], count + ((i == m - 1)?0:dp[i + 1][k]));
                    }
                }
            }    
            
        
        
        int ret = 0;
        for (int i = 0; i < 1 << n; ++i) {
            ret = max(ret, dp[0][i]);
        }
        
        return ret;
    }
};
