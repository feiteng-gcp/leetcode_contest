class Solution {
public:
    int dp[1<<8][8];
    
    bool ok(vector<vector<char>>& seats, int up, int cur, int row){
        for(int i = 0; i < seats[row].size(); i++){
            if((cur & (1<<i)) == 0)
                continue;
            if(seats[row][i] == '#')
                return false;

            if(i && (up & (1<<i-1)))
                return false;
            if(i && (cur & (1<<i-1)))
                return false;
            if(i<seats[row].size()-1 && (up & (1<<i+1)))
                return false;
            if(i<seats[row].size()-1 && (cur & (1<<i+1)))
                return false;
        }
        return true;
    }
    
    int call(vector<vector<char>>& seats, int up, int row){
        if(row == seats.size())
            return 0;
        int &ret = dp[up][row];
        if(~ret) return ret;

        ret = 0;
        for(int i = 0; i < (1<<seats[0].size()); i++){
            if(ok(seats, up, i, row))
                ret = max(ret, call(seats, i, row+1) + __builtin_popcount(i));
        }
        return ret;
    }
    
    int maxStudents(vector<vector<char>>& seats) {
        memset(dp, -1, sizeof dp);
        return call(seats, 0, 0);
    }
};
