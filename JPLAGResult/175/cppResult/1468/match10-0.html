class Solution {
public:
    bool checkIfExist(vector<int>& arr) {
        unordered_set<int> rec;
        for(int i: arr){
            if(rec.find(2*i) != rec.end()
              || (i%2 == 0 && rec.find(i/2) != rec.end()))
                return true;
            rec.insert(i);
        }
        return false;
    }
};
