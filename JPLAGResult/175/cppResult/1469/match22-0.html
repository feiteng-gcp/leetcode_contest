class Solution {
public:
    int minSteps(string s, string t) {
        int f1[26] = {0};
        int f2[26] = {0};
        
        for (int i = 0; i < s.size(); i++) {
            f1[ s[i] - 'a']++;
        }
        
        for (int i = 0; i < t.size(); i++) {
            f2[ t[i] - 'a']++;
        }
        
        int sum = 0;
        
        for (int i = 0; i < 26; i++) {
            sum += abs(f1[i] - f2[i]);
        }
        return sum / 2;
    }
};
