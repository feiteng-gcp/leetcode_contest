class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> cnt(26);
        for (char c : s) {
            cnt[c-'a']++;
        }
        for (char c : t) {
            cnt[c-'a']--;
        }
        int ret = 0;
        for (int num : cnt) {
            if (num > 0) ret += num;
        }
        return ret;
        
    }
};
