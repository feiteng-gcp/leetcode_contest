class Solution {
public:
    int minSteps(string s, string t) {
        map<char, int> ms, mt;
        for (auto c : s) ms[c]++;
        for (auto c : t) mt[c]++;
        int ret = 0;
        for (int i = 0; i < 26; i++) {
            char c = 'a' + i;
            ret += abs(ms[c] - mt[c]);
        }
        return ret / 2;
    }
};
