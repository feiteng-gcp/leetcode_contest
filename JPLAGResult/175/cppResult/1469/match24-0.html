class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> ss(256, 0), tt(256, 0);
        for (int i = 0 ; i < s.size(); ++i)
        {
            ++ss[s[i]], ++tt[t[i]];
        }
        int dif = 0;
        for (int i = 0; i < 256; ++i)
        {
            dif += abs(ss[i] - tt[i]);
        }
        return dif % 2 ? (dif / 2 + 1) : dif / 2;
    }
};
