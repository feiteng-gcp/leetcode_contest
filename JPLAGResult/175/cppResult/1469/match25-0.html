class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> sl(26);
        vector<int> tl(26);
        for (int i = 0; i < s.length(); i++) {
            sl[s[i]-'a']++;
        }
        for (int i = 0; i < t.length(); i++) {
            tl[t[i]-'a']++;
        }
        int ans = 0;
        for (int i = 0; i < 26; i++) {
            ans += abs(sl[i] - tl[i]);
        }
        return ans / 2;
    }
};
