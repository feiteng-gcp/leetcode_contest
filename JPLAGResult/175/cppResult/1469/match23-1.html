#include <bits/stdc++.h>

using namespace std;


class Solution {
	int freq_s[256];
	int freq_t[256];

public:
    int minSteps(string s, string t) {
    	memset(freq_s, 0, sizeof(freq_s));
    	memset(freq_t, 0, sizeof(freq_t));

        for (int i = 0; i < s.size(); i++) {
        	freq_s[(int)s[i]]++;

        	freq_t[(int)t[i]]++;
        }

        int ans = 0;

        for (int i = 'a'; i <= 'z'; i++) {
        	ans += abs(freq_s[i] - freq_t[i]);
        }

        return ans / 2;
    }
};
