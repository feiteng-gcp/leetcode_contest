class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> ss(26, 0), tt(26, 0);
        for (int i = 0; i < s.size(); ++i) {
            ss[s[i] - 'a']++;
            tt[t[i] - 'a']++;
        }
        
        int count = 0;
        for (int i = 0; i < 26; ++i) {
            if (tt[i] > ss[i])
                count += tt[i] - ss[i];
        }
        return count;
    }
};
