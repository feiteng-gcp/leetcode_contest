class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> c(26, 0);
        for (int i=0;i<s.length();++i) {
            c[s[i]-'a'] ++;
            c[t[i]-'a'] --;
        }
        int cnt = 0;
        for(int i=0;i<26;++i) {
            cnt += max(c[i], 0);
        }
        return cnt;
    }
};
