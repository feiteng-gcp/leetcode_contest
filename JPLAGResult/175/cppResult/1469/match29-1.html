class Solution {
public:
    int minSteps(string s, string t) {
        sort(s.begin(), s.end());
        sort(t.begin(), t.end());
        map <char, int> s1, s2;
        for (int i = 0; i < s.size(); i++) {
            s1[s[i]]++;
            s2[t[i]]++;
        }
        int ans = s.size();
        for (char c = 'a'; c <= 'z'; c++) {
            ans -= min(s1[c], s2[c]);
        }
        return ans;
    }
};
