class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> cnt(26);
        for (char c : s) {
            ++cnt[c - 'a'];
        }
        for (char c : t) {
            --cnt[c - 'a'];
        }
        int res = 0;
        for (int c : cnt) {
            if (c > 0)
                res += c;
        }
        return res;
    }
};
