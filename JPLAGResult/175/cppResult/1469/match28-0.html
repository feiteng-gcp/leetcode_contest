class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> cnt(256,0);
        for(char c:s)
            cnt[c]++;
        for(char c:t)
            cnt[c]--;
        int sum = 0;
        for(int i:cnt)
            sum+=abs(i);
        return sum/2;
    }
};
