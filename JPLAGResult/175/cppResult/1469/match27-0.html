class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> cnt1(26,0),cnt2(26,0);
        for(auto c:s){
            cnt1[c-'a']++;
        }
        for(auto c:t){
            cnt2[c-'a']++;
        }
        int res=0;
        for(int i=0;i<26;i++)
        {
            res+=(cnt1[i]>cnt2[i]? cnt1[i]-cnt2[i]:0);
        }
        return res;
    }
};
