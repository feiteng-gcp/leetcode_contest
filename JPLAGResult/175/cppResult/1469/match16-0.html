class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> sCount( 26, 0 ), tCount( 26, 0 );
        for( char c : s ) {
            sCount[ c - 'a' ]++;
        }
        for( char c : t ) {
            tCount[ c - 'a' ]++;
        }
        int res = 0;
        for( int i = 0; i < sCount.size(); i++ ) {
            res += abs( sCount[i] - tCount[i] );
        }
        return res / 2;
    }
};
