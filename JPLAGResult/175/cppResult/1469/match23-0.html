class Solution {
    int sum[26];
    int sum2[26];
public:
    int minSteps(string s, string t) {
        memset(sum,0,sizeof(sum));
        memset(sum2,0,sizeof(sum2));
        for(int i=0;i<s.length();i++){
            sum[s[i]-'a']++;
            sum2[t[i]-'a']++;
        }
        int ans=0;
        for(int i=0;i<26;i++){
            ans+=abs(sum[i]-sum2[i]);
        }
        return ans/2;
    }
};
