class Solution {
public:
    int minSteps(string s, string t) {
        unordered_map<char, int> c;
        for (char ch: s) {
            ++c[ch];
        }
        for (char ch: t) {
            --c[ch];
        }
        int ans = 0;
        for (auto& [k, v]: c) {
            ans += abs(v);
        }
        return ans / 2;
    }
};
