class Solution {
public:
    int minSteps(string s, string t) {
        map<char,int> m1,m2;
        
        for(int i=0;i<s.size();i++)
        {
            m1[s[i]]++;
            m2[t[i]]++;
        }
        
        int same=0;
        
        for(char c='a';c<='z';c++)
            same+=min(m1[c],m2[c]);
        
        return s.size()-same;
    }
};
