class Solution:
    def checkIfExist(self, arr: List[int]) -> bool:
        s = set(arr)
        for i in s:
            if i * 2 in s and i != 0:
                return True
        times = 0
        for i in arr:
            if i == 0:
                times += 1
        return times > 1
