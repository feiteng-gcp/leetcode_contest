class TweetCounts:

    def __init__(self):
        self.memo = collections.defaultdict(list)
        self.flen = {"minute":60,"hour":3600,"day":86400}

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.memo[tweetName] += [time]

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        delta = self.flen[freq]
        res = [0] * ((endTime - startTime + delta) //delta)
        for n in self.memo[tweetName]:
            if startTime <= n <=endTime:
                res[(n-startTime)//delta] += 1
        return res
