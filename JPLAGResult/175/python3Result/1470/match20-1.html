    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        arr = self.tweets[tweetName]
        
        div_met = 1
        if freq == 'minute':
            div_met = 60
        elif freq == 'hour':
            div_met = 60 * 60
        elif freq == 'day':
            div_met = 60 * 60 * 24
        
        res = [0] * ((endTime - startTime) // div_met + 1)
        
        pos = bisect.bisect_left(self.tweets[tweetName], startTime)
        
        
        while pos < len(arr) and arr[pos] <= endTime:
            res[(arr[pos] - startTime) // div_met] += 1
            pos += 1
        return res
