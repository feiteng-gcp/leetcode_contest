class TweetCounts:

    def __init__(self):
        self.cache = collections.defaultdict(list)
        

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.cache[tweetName].append(time)
        

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        freq = {"minute": 60, "hour": 3600, "day": 86400}[freq]
        self.cache[tweetName] = sorted(self.cache[tweetName])
        endTime += 1
        def foo(c):
            li = 0
            for l in range(startTime, endTime, freq):
                while li < len(c) and c[li] < l:
                    li += 1
                r = min(l + freq, endTime)
                ri = li
                while ri < len(c) and c[ri] < r:
                    ri += 1
                yield ri - li
                li = ri
        return list(foo(self.cache[tweetName]))
