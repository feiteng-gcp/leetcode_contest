class TweetCounts:

    def __init__(self):
        from collections import defaultdict
        self.m = defaultdict(list)
        self.t = {"day":86400, "minute":60, "hour":3600}        

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.m[tweetName].append(
time)
        

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        freq = self.t[freq]
        ret = [0] * ((endTime - startTime) // freq + 1)
        for v in self.m[tweetName]:
            if startTime <= v <= endTime:
                ret[(v - startTime) // freq] += 1
        return ret
