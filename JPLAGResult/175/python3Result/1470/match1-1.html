class TweetCounts:

    def __init__(self):
        self.tweets = collections.defaultdict(set)

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.tweets[tweetName].add(time)

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        if freq == 'minute':
            delta = 60
        elif freq == 'hour':
            delta = 60 * 60
        else:
            delta = 60 * 60 * 24
        ret = [0] * ((endTime - startTime) // delta + 1)
        for time in self.tweets[tweetName]:
            if startTime <= time <= endTime:
                ret[(time - startTime) // delta] += 1
        return ret
