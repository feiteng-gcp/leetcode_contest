class TweetCounts:

    def __init__(self):
        self.database = collections.defaultdict(list)
        return None


    def recordTweet(self, tweetName: str, time: int) -> None:
        self.database[tweetName].append(time)
        return None
        

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        d = {"minute": 60,
            "hour": 60*60,
            "day": 24*60*60}
        interval = d[freq]
        
        res = [0] * ((endTime-startTime)//interval + 1)
        for time in self.database[tweetName]:
            if time <= endTime and time >= startTime:
                res[(time - startTime)//interval] += 1
        return res
