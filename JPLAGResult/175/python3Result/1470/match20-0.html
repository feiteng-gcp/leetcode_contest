class TweetCounts:

    def __init__(self):
        self.tweet = collections.defaultdict(list)

    def recordTweet(self, tweetName: str, time: int) -> None:
        bisect.insort(self.tweet[tweetName], time)
    
    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        tgt = self.tweet[tweetName]
        dt = {'minute':60,'hour':3600,'day':86400}[freq]
        L,R = bisect.bisect_left(tgt, startTime), bisect.bisect_left(tgt, endTime+1)
        ans = [0]*math.ceil((endTime-startTime+1)/dt) 
        for t in tgt[L:R]:
            ans[(t-startTime)//dt]+=1
        return ans
