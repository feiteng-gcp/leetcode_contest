from collections import defaultdict

class TweetCounts:

    def __init__(self):
        self.data = defaultdict
(list)

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.data[tweetName].append(time)
        

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        delta = None
        if freq == 'minute':
            delta = 60
        elif freq == 'hour':
            delta = 60 * 60
        elif freq == 'day':
            delta = 24 * 60 * 60
        num_buckets = (endTime + 1 - startTime + delta - 1) // delta
        ret = [0] * num_buckets
        for t in self.data[tweetName]:
            if t < startTime:
                continue
            if t > endTime:
                continue
            bucket = (t - startTime) // delta
            ret[bucket] += 1
        return ret
