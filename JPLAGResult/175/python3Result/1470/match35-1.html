class TweetCounts:

    def __init__(self):
        self.tc = collections.defaultdict(list)
        

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.tc[tweetName].append(time)
        

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        sf = {"minute": 60, "hour": 60*60, "day": 24*60*60}
        t = self.tc[tweetName]
        t.sort()
        n, i = len(t), 0
        while i < n and t[i] < startTime:
            i += 1
        
        ret = []
        offset = startTime
        while offset <= endTime:
            tend = min(offset + sf[freq]-1, endTime)
            j = i
            while j < n and t[j] <= tend:
                j += 1
            ret.append(j-i)
            offset, i = tend+1, j
        return ret
