    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        if freq == 'minute':
            delta = 60
        elif freq == 'hour':
            delta = 3600
        else:
            delta = 86400
        
        endTime += 1
        
        res = []
        s = startTime
        e = startTime + delta
        while e < endTime:
            idx1 = bisect.bisect_left(self.mem[tweetName], s)
            idx2 = bisect.bisect_left(self.mem[tweetName], e)
            res.append(idx2 - idx1)
            s = e
            e = s + delta
        idx1 = bisect.bisect_left(self.mem[tweetName], s)
        idx2 = bisect.bisect_left(self.mem[tweetName], endTime)
        res.append(idx2 - idx1)
        return res
