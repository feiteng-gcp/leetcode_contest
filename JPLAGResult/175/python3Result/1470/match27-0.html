class TweetCounts:

    def __init__(self):
        self.records = collections.defaultdict(set)
        return

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.records[tweetName].add(time)
        return

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int):
        if freq == "minute":
            freq_time = 60
        elif freq == "hour":
            freq_time = 3600
        elif freq == "day":
            freq_time = 86400
        n = int((endTime - startTime) / freq_time) + 1
        res = [0 for i in range(n)]
        for time in self.records[tweetName]:
            if time >= startTime and time <= endTime:
                index = int((time - startTime) / freq_time)
                res[index] += 1
        return res
