class TweetCounts:

    def __init__(self):
        self.count_dict = collections.defaultdict(list)
        self.delta_dict = {'minute': 60, 'hour': 3600, 'day': 86400}

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.count_dict[tweetName].append(time)

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        delta = self.delta_dict[freq]
        n = (endTime - startTime) // delta + 1
        res = [0 for _ in range(n)]
        for x in self.count_dict[tweetName]:
            if x < startTime or x > endTime:
                continue
            ind = (x-startTime) // delta
            res[ind] += 1
        return res
