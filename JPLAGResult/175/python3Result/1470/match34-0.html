class TweetCounts:

    def __init__(self):
        self.m = collections.defaultdict(list)

    def recordTweet(self, tweetName: str, time: int) -> None:
        self.m[tweetName].append(time)

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        endTime += 1
        if freq == 'minute':
            freq = 60
        elif freq == 'hour':
            freq = 60 * 60
        else:
            freq = 60 * 60 * 24
        ans = [0] * ((endTime - startTime - 1) // freq + 1)
        for t in self.m[tweetName]:
            if endTime > t >= startTime:
                ans[(t - startTime) // freq] += 1
        return ans
