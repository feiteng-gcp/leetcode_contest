class TweetCounts:

    def __init__(self):
        self.record = collections.defaultdict(list)

    def recordTweet(self, tweetName: str, time: int) -> None:
        bisect.insort(self.record[tweetName], time)

    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        if freq == 'minute':
            span = 60
        elif freq == 'hour':
            span = 3600
        elif freq == 'day':
            span = 3600*24
        
        query = []
        for st in range(startTime, endTime+1, span):
            query.append((st,min(st+span-1, endTime)))
        # print(query)
        ans = []
        for st, en in query:
            left = bisect.bisect_left(self.record[tweetName], st)
            right = bisect.bisect_right(self.record[tweetName], en)
            ans.append(right-left)
        return ans
