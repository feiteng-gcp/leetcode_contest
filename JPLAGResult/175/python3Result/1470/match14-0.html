    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        
        if freq == 'minute':
            t = 60
        elif freq == 'hour':
            t = 60 * 60
        elif freq == 'day':
            t = 60 * 60 * 24
            
        arr = []
        
        st_time = startTime
        
        while 1:
            
            ed_time = min( st_time + t - 1, endTime)
            
            p = bisect.bisect_left(self.d[tweetName], st_time)
            
            # print (self.d[tweetName])
            # print (p)
            # print (st_time, ed_time)
            
            c = 0
            while p < len(self.d[tweetName]) and st_time <= self.d[tweetName][p] <= ed_time:
                c += 1
                p += 1
            
            arr.append(c)
            
            if ed_time == endTime:
                break
            else:
                st_time = ed_time + 1
            
        return arr
