import math

class TweetCounts:

    def __init__(self):
        self.tweet_record = {}

    def recordTweet(self, tweetName: str, time: int) -> None:
        if tweetName not in self.tweet_record:
            self.tweet_record[tweetName] = [time]
        else:
            self.tweet_record[tweetName].append(time)
            
    def getTweetCountsPerFrequency(self, freq: str, tweetName: str, startTime: int, endTime: int) -> List[int]:
        if tweetName not in self.tweet_record:
            return []
        else:
            print("Path 2")
            if freq == "minute":
                delta = 60
            if freq == "hour":
                delta = 3600
            if freq == "day":
                delta = 3600*24
            
            num_intervals = math.ceil( (endTime - startTime) // delta )
            output = [0 for _ in range(num_intervals + 1)]
            for time in self.tweet_record[tweetName]:
                if time >= startTime and time <= endTime:
                    time_shifted = time - startTime
                    output[ time_shifted // delta ] += 1
        return output
