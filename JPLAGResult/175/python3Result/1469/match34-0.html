from collections import Counter

class Solution:
    def minSteps(self, s: str, t: str) -> int:
        cs = Counter(s)
        ct = Counter(t)

        res = 0
        for c in cs: 
            if c in ct: 
                res += abs(ct[c] - cs[c])
            else: 
                res += cs[c]
        
        for c in ct: 
            if c not in cs: 
                res += ct[c]
        
        return res // 2
