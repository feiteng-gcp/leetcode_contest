from collections import Counter

class Solution:
    def minSteps(self, s: str, t: str) -> int:
        cnt1, cnt2 = Counter(s), Counter
(t)
        res = 0
        for o in range(26):
            c = chr(ord('a') + o)
            res += max(0, cnt1[c] - cnt2[c])
        return res
