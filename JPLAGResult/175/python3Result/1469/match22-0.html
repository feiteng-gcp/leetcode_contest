from collections import Counter
class Solution:
    def minSteps(self, s: str, t: str) -> int:
        
        cntr1 = Counter(s)
        cntr2 = Counter(t)
        
        t = 0
        for i in range(26):
            ch = chr(ord('a') + i)
            
            t += min(cntr1[ch], cntr2[ch])
        
        return len(s) - t
