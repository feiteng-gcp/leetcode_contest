class Solution:
    def minSteps(self, s: str, t: str) -> int:
        
        cs = collections.Counter(s)
        ct = collections.Counter
(t)
        
        c = 0
        
        for i in ct:
            if i in cs:
                
                c += abs(ct[i] - cs[i])
                
            else:
                c += ct[i]
                
        for i in cs:
            if i not in ct:
                c += cs[i]
                
        return c // 2
