class Solution:
    def minSteps(self, s: str, t: str) -> int:
        ss = [0] * 26
        tt 
= [0] * 26
        
        for c in s:
            ss[ord(c) - ord('a')] += 1
        
        for c in t:
            tt[ord(c) - ord('a')] += 1
        
        r1 = 0
        for i in range(26):
            r1 += max(ss[i] - tt[i], 0)
        r2 = 0
        for i in range(26):
            r2 += max(tt[i] - ss[i], 0)
        
        return min(r1, r2)
