class Solution:
    def minSteps(self, s: str, t: str) -> int:
        ds = collections.Counter()
        dt = collections.Counter()
        
        for x in s:
            ds[x] += 1
        
        for x in t:
            dt[x] += 1
        
        res = 0
        for k in ds.keys():
            res += abs(ds[k] - dt[k])
        
        for k in dt.keys():
            if k not in ds.keys():
                res += abs(dt[k] - ds[k])
        
        # print(res)
        return res // 2
