class Solution:
    def minSteps(self, s: str, t: str) -> int:
        from collections import Counter
        sDict = Counter(s)
        tDict = Counter(t)
        count = 0 
        for key in sDict:
            if key in tDict:
                count+=min(sDict[key], tDict[key])
        return len(s)-count
