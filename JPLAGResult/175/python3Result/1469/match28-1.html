class Solution:
    def minSteps(self, s: str, t: str) -> int:
        cs = collections.Counter(s)
        ct = collections.Counter(t)
        
        a, b = 0, 0
        for x, tx in ct.items():
            sx = 0 if x not in cs else cs[x]
            if sx > tx:
                a += sx - tx
            else:
                b += tx - sx
        return max(a, b)
