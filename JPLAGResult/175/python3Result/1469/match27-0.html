class Solution:
    def minSteps(self, s: str, t: str) -> int:
        
        S=[0 for _ in range(26)]
        T=[0 for _ in range(26)]
        
        
        for c in s:
            S[ord(c)-ord('a')]+=1
        
        for c in t:
            T[ord(c)-ord('a')]+=1
        
        res=0
        
        for i in range(26):
            if S[i]-T[i]<0:
                res+=T[i]-S[i]
        
        return res
