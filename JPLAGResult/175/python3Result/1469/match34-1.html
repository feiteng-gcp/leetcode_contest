from collections import Counter

class Solution:
    def minSteps(self, s: str, t: str) -> int:
        cnt_s, cnt_t = Counter(s), Counter
(t)
        diff = 0
        for c in cnt_s:
            if c in cnt_t:
                diff += max(cnt_s[c] - cnt_t[c], 0)
            else:
                diff += cnt_s[c]
        return diff
