class Solution:
    def minSteps(self, s: str, t: str) -> int:
        cnt1 = collections.Counter(s)
        cnt2 = collections.Counter(t)
        
        diff = 0
        for c in set(cnt1.keys()).union(set(cnt2.keys())):
            diff += abs(cnt1[c] - cnt2[c])
        
        return (diff + 1) // 2
