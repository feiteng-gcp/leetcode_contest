class Solution:
    def minSteps(self, s: str, t: str) -> int:
        s_cnt = {}
        t_cnt = {}
        for c in range(ord('a'), ord('z')+1):
            s_cnt[chr(c)] = 0
            t_cnt[chr(c)] = 0
        for c in s:
            s_cnt[c] += 1
        for c in t:
            t_cnt[c] += 1
        
        step 
= 0
        for c in s_cnt:
            _s = s_cnt[c]
            _t = t_cnt[c]
            step += max(0, _t - _s)
        
        return step
