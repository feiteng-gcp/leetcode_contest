class Solution:
    def minSteps(self, s: str, t: str) -> int:
        x = [0] * 26
        orda = ord('a')
        for c in s:
            x[ord(c) - orda] += 1
        for c in t:
            x[ord(c) - orda] -= 1
        result = 0
        for i in range(26):
            if x[i] > 0:
                result += x[i]
        return result
