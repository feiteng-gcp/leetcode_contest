class Solution:
    def minSteps(self, s: str, t: str) -> int:
        ds = collections.defaultdict(int)
        dt = collections.defaultdict(int)
        for a in s:
            ds[a] += 1
        for b in t:
            dt[b] += 1
        res 
= 0
        for k in dt.keys():
            res += max(dt[k] - ds[k], 0)
        return res
