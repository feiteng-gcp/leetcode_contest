class Solution:
    def minSteps(self, s: str, t: str) -> int:
        n = len(s)
        char_map1 = [0]*26
        char_map2 = [0]*26
        for c in s:
            char_map1[ord(c)-ord('a')] +=1
        for c in t:
            char_map2[ord(c)-ord('a')] +=1
        delx = 0
        for i in range(26):
            delx += abs(char_map1[i] - char_map2[i])
        return int(delx/2)
