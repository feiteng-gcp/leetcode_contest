class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        
        s = []
        s.append(['#'] * (len(seats[0]) + 2) )
        for i in seats:
            s.append(['#'] + i + ['#'])
                
        m, n = len(s), len(s[0])
        
        arr = []
        
        chk = set()
        
        
for i in range(m):
            for j in range(n):
                if s[i][j] == '.':
                    arr.append((i, j))
                    
        if not arr:
            return 0
                    
        ans = 0        
        count = 0
        
        halt = 0
        
        def ftry(i, h = False):
            
            if h:
                nonlocal halt
                halt += 1
                if halt > 1000000:
                    return
            
            nonlocal count
            nonlocal ans
            
            if i == len(arr):
                return
                        
            x, y = arr[i]
            
            cheat = False
            
            for x0, y0 in ((0, -1), (0, 1), (-1, -1), (-1, 1)):
                u = x + x0
                v = y + y0
                
                if (u, v) in chk:
                    cheat = True
                    
                    
            if not cheat:
                
                
                
                
                chk.add((x, y))
                count += 1
                ans = max(ans, count)
                
                ftry(i + 1, h)
                
                count -= 1
                chk.remove((x, y))
                
                if h:
                    ftry(i + 1, h)
                
                
                
            else:
                ftry(i + 1, h)
                
                
        ftry(0, True)
        ftry(1)
                
        return ans
