class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        rows = len(seats)
        cols = len(seats[0])
        pre = []
        valid_next = []
        
        for row in seats:
            c = 0
            for x in row:
                c <<= 1
                if x == '#':
                    c |= 1
            pre.append(c)
        #print(pre)
        for n in range(1 << cols):
            valid = True
            for c in range(cols):
                if (n & (1 << c)) > 0:
                    if c > 0 and (n & (1 << (c - 1))) > 0:
                        valid = False
                    if c + 1 < cols and (n & (1 << (c + 1))) > 0:
                        valid = False
            if valid:
                valid_next.append(n)
        
        cache = {}
        def go(row, mask):
            key = (row, mask)
            if key in cache:
                return cache[key]
            if row == rows:
                return 0
            
            best = go(row + 1, 0)
            for n in valid_next:
                if (pre[row] & n) > 0:
                    continue
                valid = True
                count = 0
                for c in range(cols):
                    if (n & (1 << c)) > 0:
                        count += 1
                        if c > 0 and (mask & (1 << (c - 1))) > 0:
                            valid = False
                        if c + 1 < cols and (mask & (1 << (c + 1))) > 0:
                            valid = False
                if valid:
                    best = max(count + go(row + 1, n), best)
                    #print(key, n, count + go(row + 1, n))
        
            #print(key, best)
            cache[key] = best
            return cache[key]
        
        return go(0, 0)
