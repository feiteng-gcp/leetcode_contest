class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        m = len(seats)
        n = len(seats[0])
        dp = [[-1 for j in range(2**n)] for i in range(m)]
        condition = [0] * m
        for i in range(m):
            s = seats[i]
            for ch in s:
                if ch == ".":
                    condition[i] = condition[i] * 2 + 1
                else:
                    condition[i] = condition[i] * 2 + 0
        
        goodLine = [True for i in range(2**n)]
        num = [0 for i in range(2**n)]
        for i in range(2**n):
            pattern = 3
            for j in range(max(n-1, 1)):
                if i & pattern == pattern:
                    goodLine[i] = False
                pattern *= 2
            num[i] = str(bin(i)).count("1")
            if goodLine[i] == True:
                if i & condition[0] == i:
                    dp[0][i] = num[i]
                    
        for i in range(1, m, 1):
            for j in range(2**n):
                if (j & condition[i] == j) and (goodLine[j] == True):
                    for k in range(2**n):
                            if ((j * 2) & k ==0 ) and ((j // 2) & k == 0):
                                dp[i][j] = max(dp[i-1][k] + num[j], dp[i][j])
        ans = max(dp[m-1])
        return ans
