class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        m = len(seats)
        n = len(seats[0])
        
        def translate(state):
            p = []
            for i in range(n):
                p.append(bool(state % 2))
                state //= 2
            return p
        
        def valid(state, line):
            p = translate(state)
            for i in range(n):
                if p
[i] and line[i] == '#':
                    return False
            for i in range(1, n):
                if p[i] and p[i - 1]:
                    return False
            return True
        
        def count(state):
            return sum(translate(state))
        
        def check(state0, state1):
            p = translate(state0)
            q = translate(state1)
            for i in range(n):
                if not q[i]:
                    continue
                if i > 0 and p[i - 1]:
                    return False
                if i < n - 1 and p[i + 1]:
                    return False
            return True
        
        dp = [[-1 for j in range(2 ** n)] for i in range(m)]
        for j in range(2 ** n):
            if valid(j, seats[0]):
                dp[0][j] = count(j)
            else:
                dp[0][j] = -1
        for i in range(1, m):
            for j in range(2 ** n):
                if not valid(j, seats[i]):
                    dp[i][j] = -1
                    continue
                max_value = -1
                for k in range(2 ** n):
                    if check(k, j) and dp[i - 1][k] > max_value:
                        max_value = dp[i - 1][k]
                if max_value >= 0:
                    dp[i][j] = max_value + count(j)
                else:
                    dp[i][j] = -1
        return max(dp[-1])
