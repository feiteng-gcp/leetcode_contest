import numpy as np
from itertools import product
from functools import lru_cache


class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        row_length = len(seats[0])
        
        seats = np.array([[int(s == ".") for s in row] for row in seats])
        # print(seats)
        
        # all combinations of first row
        all_comb = [np.array(x) for x in product([0, 1], repeat = row_length)][::-1]
        best_res = [0 for _ in range(len(seats) + 1)]
        
        def allowable(comb_seating, seat_row, front_row):
            # print("function allowable")
            # print(comb_seating, seat_row)
            # print(all(comb_seating <= seat_row))
            
            if not all(comb_seating <= seat_row):
                return False
            
            prev_occ = 0
            for seat in comb_seating:
                if seat + prev_occ == 2:
                    return False
                prev_occ = seat
                
            for i,seat in enumerate(comb_seating):
                if i-1 >= 0 and comb_seating[i] + front_row[i-1] == 2:
                    return False
                if i+1 <= row_length-1 and comb_seating[i] + front_row[i+1] == 2:
                    return False
                
            return True
        
        def possibilities(front_row, seat_row):
            # print("function possibilities")
            # print(front_row, seat_row)
            res = []
            for comb_seating in all_comb:
                if allowable(comb_seating, seat_row, front_row):
                    res.append(comb_seating)
            return np.array(res)
        
        def iterate(fronts, remaining):
            res_sum = np.sum(fronts)
            if res_sum <= best_res[len(remaining)] - 2:
                return 0
            best_res[len(remaining)] = max(best_res[len(remaining)], res_sum)
            if remaining.size == 0:
                # print(res_sum)
                return res_sum
            res = 0
            # print("possibilities result")
            # print(possibilities(fronts[-1], remaining[0]))
            for poss in possibilities(fronts[-1], remaining[0]):
                # break  # TODO
                res = max(res, iterate(np.concatenate((fronts, [poss])), 
                                       remaining[1:]))
            return res
            
        fronts_init = np.array([[0]*row_length])
        result = iterate(fronts_init, seats)
        print("RESULT")
        print(result)
        return result
