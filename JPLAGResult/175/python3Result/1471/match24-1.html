class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        m = len(seats)
        n = len(seats[0])
        status = (1<<n)-1
        code_seat = [0]*m
        for i in range(m):
            for j in range(n):
                code_seat[i] = (code_seat[i]<<1) + int(seats[i][j]=='.')
        
        dp = [[0]*status for i in range(m)]
        one_count = [0] * status
        for i in range(1, status):
            one_count[i] = one_count[i - (i&(-i))]+1
            
        for i in range(m):
            for j in range(status):
                if j & (j>>1):
                    continue
                if (j | code_seat[i])^code_seat[i]:
                    continue
                maxn = 0
                ct = one_count[j]
                if i == 0:
                    dp[0][j]=ct
                    continue                
                for k in range(status):
                    if dp[i-1][k] > maxn and (k<<1)&j==0 and (k>>1)&j==0:
                        maxn = dp[i-1][k]
                dp[i][j]=maxn+ct
   
        return max(dp[-1])
