class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        
        def check(a, b):
            # print(len(a), len(b))
            for i in range(1, len(a) - 1):
                if a[i] and (b[i - 1] or b[i + 1]):
                    return False
            return True
        
        m = len(seats)
        n = len(seats[0])
        
        all_prevs = [[0 for _ in range(n + 2)]]
        all_prev_res = [0]
    
        for cur_row in range(m):
            all_nexts = []
            all_next_res = []
            
            for config in range(1 << n):
                cur = [0]
                tmp = config
                flag = True
                for i in range(n):
                    # print(cur_row, i, tmp, m, n)
                    if tmp & 1 and seats[cur_row][i] == '#':
                        flag = False
        
                    cur.append(tmp & 1)
                    tmp = tmp >> 1
                cur.append(0)
            
                if not flag or not check(cur, cur):
                    continue

                mx = 0
                for cnt, prev in zip(all_prev_res, all_prevs):
                    if not check(cur, prev):
                        continue
                    
                    mx = max(mx, cnt + sum(cur))
                    
                all_nexts.append(cur)
                all_next_res.append(mx)
                
            all_prevs = all_nexts
            all_prev_res = all_next_res
            
        return max(all_prev_res)
