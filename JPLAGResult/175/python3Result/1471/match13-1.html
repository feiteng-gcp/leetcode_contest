class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        memo = dict()
        M = len(seats)
        N = len(seats[0])
        
        def find_next(i,j):
            nonlocal seats
            ii = i
            jj = j+1
            while True:
                if jj == N:
                    jj = 0
                    ii += 1
                if ii == M:
                    return None, None
                if seats[ii][jj] == '.':
                    return ii,jj
                jj += 1
        
        def dfs(i,j):
            nonlocal seats, memo
            key = ''.join([''.join(row) for row in seats[i:]])
            
            if key in memo:
                return memo[key]
            
            taken = 1
            seats[i][j] = '#'
            mark = []
            
for ii,jj in [(i+1,j-1),(i+1,j+1),(i,j-1),(i,j+1)]:
                if 0 <= ii < M and 0 <= jj < N and seats[ii][jj] == '.':
                    seats[ii][jj] = '#'
                    mark.append((ii,jj))
            
            next_i, next_j = find_next(i,j)
            if next_i is not None and next_j is not None:
                taken += dfs(next_i, next_j)
            
            not_taken = 0
            seats[i][j] = '.'
            for ii,jj in mark:
                seats[ii][jj] = '.'
                
            next_i, next_j = find_next(i,j)
            if next_i is not None and next_j is not None:
                not_taken += dfs(next_i, next_j)
            
            ans = max(taken, not_taken)
            memo[key] = ans
            return ans
        
        ii,jj = find_next(0,-1)
        
        if ii is None and jj is None:
            return 0
    
        return dfs(ii,jj)
