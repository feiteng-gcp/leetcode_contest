class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        
        m = len(seats)
        n = len(seats[0])
        
        dp = [[-1 for _ in range(2 ** n)] for _ in range(m)]
        
        def stateToBinArr(state):
            str1 = "{0:b}".format(state)
            
            return [0] * (n - len(str1)) + [int(x) for x in str1]
        
        binArr_cache = [stateToBinArr(x) for x in range(2 ** n)]
        
        valid_states = []
        for state in range(2 ** n):
            
            arr1 = binArr_cache[state]
            flag = True
            for i in range(1, n):
                if arr1[i] == 1 and arr1[i - 1] == 1:
                    flag = False
                    break
            
            if flag:
                valid_states.append(state)
        
        def canSeat(state, row):
            binArr = binArr_cache[state]
            for i in range(n):
                if binArr[i] == 1 and row[i] == '#':
                    return False
            return True
        
        def notConflict(state_previousRow, state_curRow):
            binArr_previous = binArr_cache[state_previousRow]
            binArr_cur = binArr_cache[state_curRow]
            
            for i in range(n):
                if binArr_cur[i] == 1:
                    if i > 0 and binArr_previous[i - 1] == 1:
                        return False
                    if i < n - 1 and binArr_previous[i + 1] == 1:
                        return False
            return True
        
        states_previousRow = []
        for i in range(m):
            states_curRow = []
            for state in valid_states:
                if canSeat(state, seats[i]):
                    if i == 0:
                        dp[i][state] = sum(binArr_cache[state])
                        states_curRow.append(state)
                    else:
                        for state_previousRow in states_previousRow:
                            if notConflict(state_previousRow, state):
                                dp[i][state] = max(dp[i][state], dp[i - 1][state_previousRow] + sum(binArr_cache[state]))
                        
                        if dp[i][state] >= 0:
                            states_curRow.append(state)
            
            states_previousRow = states_curRow
        
        return max(dp[m - 1])
