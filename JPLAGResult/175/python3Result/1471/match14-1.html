import functools
class Solution:
    state2num = {}
    def maxStudents(self, seats: List[List[str]]) -> int:
        state = ';'.join([','.join(x) for x in seats])
        if state in self.state2num:
            return self.state2num[
state]
        res = 0
        m = len(seats)
        n = len(seats[0])
        for i, x in enumerate(seats):
            for j, y in enumerate(x):
                if y == '#':
                    continue
                seats[i][j] = '#'
                tmp_inds = [(i, j)]
                res = max([res, self.maxStudents(seats)])
                for (i0, j0) in [(i, j+1), (i+1, j-1), (i+1, j+1)]:
                    if 0 <= i0 < m and 0 <= j0 < n and seats[i0][j0] == '.':
                        seats[i0][j0] = '#'
                        tmp_inds.append((i0, j0))         
                res = max([res, self.maxStudents(seats)+1])
                for (i0, j0) in tmp_inds:
                    seats[i0][j0] = '.'
                self.state2num[state] = res
                return res
        self.state2num[state] = res
        return res
