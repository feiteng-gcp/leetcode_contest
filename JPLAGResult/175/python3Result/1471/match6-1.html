class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        n, m = len(seats), len(seats[0])
        M = 2**m
        
        def seatOk(x, seatsR):
            xb = [(x >> i & 1) for i in range(m)]
            for i in range(1, m):
                if xb
[i] and xb[i-1]:
                    return False
            for i in range(m):
                if xb[i] and seatsR[i] != ".":
                    return False
            return True
        
        def cheatConflict(x, y):
            xb = [(x >> i & 1) for i in range(m)]
            yb = [(y >> i & 1) for i in range(m)]
            for i in range(1, m):
                if xb[i] and yb[i-1]:
                    return True
            for i in range(0, m-1):
                if xb[i] and yb[i+1]:
                    return True
            return False
        
        def seatCount(x):
            return sum(int(c) for c in bin(x)[2:])
        
        dp = [0 for _ in range(M)]
        for i in range(n):
            # print ("dp", seats[i])
            ndp = [0 for _ in range(M)]
            for x in range(M):
                if not seatOk(x, seats[i]): continue
                # print ("try x:", x, bin(x)[2:])
                for y in range(M):
                    if not cheatConflict(x, y):
                        ndp[x] = max(ndp[x], dp[y] + seatCount(x))
            #for x in range(M):
            #    if ndp[x] != 0:
            #        print (x, bin(x)[2:], ndp[x])
            dp = ndp
        return max(dp)
