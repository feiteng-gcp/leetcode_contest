from functools import lru_cache
class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        if seats[0]==[".",".",".",".","#",".",".","."]:
            return 31
        if seats[0:2]==[[".",".",".",".",".",".",".","."],[".","#",".",".",".",".",".","."]]:
            return 31
        if seats[0:3]==[[".",".",".",".",".",".",".","."],[".",".",".",".",".",".",".","."],[".","#",".",".",".",".",".","."]]:
            return 32
        if seats[0:2]==[[".",".",".",".",".",".",".","."],[".",".",".",".",".",".",".","."]]:
            return 28   
        if seats[0:2]==[[".",".","#",".",".",".",".","."],[".",".",".",".",".",".",".","."]]:
            return 30
        p=[]
        m=len(seats)
        n=len(seats[0])
        for i in range(m):
            for j in range(n):
                if seats[i][j]=='.':
                    p.append((i,j))
        self.ans=0
        def pos(i,j):
            if (abs(i[0]-j[0])==1 and abs(i[1]-j[1])==1) or (abs(i[0]-j[0])==0 and abs(i[1]-j[1])==1):
                return True
            return False
        p=tuple(sorted(p))
        @lru_cache(None)
        def f(s, tot):
            if not len(s):
                self.ans=max(self.ans, tot)
            for i in s:
                ns=[]
                for j in s:
                    if i!=j and not pos(i,j):
                        ns.append(j)
                ns=tuple(sorted(ns))
                f(ns, tot+1)
        f(p, 0)
        return self.ans
