class Solution:
    def maxStudents(self, seats: 'List[List[str]]') -> int:
        dp={}
        m,n=len(seats),len(seats[0])
        def toarr(num,n):
            arr=[0]*n
            k=num;i=0
            while k>0:
                arr[i]=k%2
                k//=2
                i+=1
            arr.reverse()
            return tuple(arr)
        
        dp[toarr(0,n)]=0
        def valid1(a1,seat1):
            l=len(a1)
            for i in range(1,l):
                if a1[i]==1 and a1[i-1]==1:
                    return False 
            for i in range(l):
                if a1[i]==1 and seat1[i]=='#':
                    return False 
            return True
        def valid2(a1,a2):
            l=len(a1)
            
for i in range(l):
                if a1[i]==1 and i>=1 and a2[i-1]==1:
                    return False
                if a1[i]==1 and i+1<l and a2[i+1]==1:
                    return False 
            return True
        for idx in range(m):
            newdp={}
            for i in range(2**n):
                cur=toarr(i,n)
                if not valid1(cur,seats[idx]):
                    continue
                for prev in dp:
                    if valid2(cur,prev):
                        if cur not in newdp:
                            newdp[cur]=0
                        newdp[cur]=max(newdp[cur],dp[prev]+sum(cur))
            dp=newdp
        ans=[dp[k] for k in dp]
        return max(ans)
