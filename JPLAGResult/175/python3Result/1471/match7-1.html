class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        N, M = len(seats), len(seats[0])
        ALL = 1 << M
        def SeatMast(row):
            mask = 0
            for x in row:
                x = 1 if x == '.' else 0
                mask = (mask << 1) | x
            return mask
        def OkRow(row):
            check = 0b11
            while check <= row:
                if check & row == check:
                    return False
                check <<= 1
            return True
        def CountRow(row):
            cnt = 0
            while row:
                cnt += 1
                row &= row - 1
            return cnt

        ok_rows = {row:CountRow(row) for row in range(ALL) if OkRow(row)}
        seats = [SeatMast(row) for row in seats]
        dp = {0:0}
        for i, seat in enumerate(seats):
            ndp = {}
            for s, c in dp.items():
                for ns, dc in ok_rows.items():
                    if ns & seat != ns: continue
                    if s & (ns << 1) or s & (ns >> 1): continue
                    ndp[ns] = max(c + dc, ndp.get(ns, 0))
            dp = ndp
        return max(dp.values())
