class Solution:
    def _plan2idx(self, plan: List[bool]):
        idx = 0
        for i in range(len(plan)):
            if plan[i]:
                idx += (2 ** i)
        return idx
    
    def _idx2plan(self, idx: int, N: int):
        plan = []
        for i in range(N):
            if idx & (2 ** i) > 0:
                plan.append(True)
            else:
                plan.append(False)
        return plan
    
    def maxStudents(self, seats: List[List[str]]) -> int:
        M = len(seats)
        N = len(seats[0])
        
        avb_map = [[True if c == '.' else False for c in row] for row in seats]
        # print(avb_map)
        
        IMP = -9999
        mx = [IMP for _ in range(2 ** N)]
        
        # row 0
        for idx in range(2 ** N):
            plan = self._idx2plan(idx, N)
            avb = True
            for j in range(N):
                if plan[j] and not avb_map[0][j]:
                    avb = False
                    break
                if j > 0 and (plan[j] and plan[j-1]):
                    avb = False
                    break
            if avb:
                mx[idx] = sum(plan)
                # print(plan)
        
        # row 1 ~ M-1
        for i in range(1, M):
            nmx = [IMP for _ in range(2 ** N)]
            for idx in range(2 ** N):
                # check avb
                plan = self._idx2plan(idx, N)
                avb = True
                for j in range(N):
                    if plan[j] and not avb_map[i][j]:
                        # broke
                        avb = False
                        break
                    if j > 0 and (plan[j] and plan[j-1]):
                        # cheat
                        avb = False
                        break
                if not avb:
                    continue
                    
                sum_plan = sum(plan)
                
                # Find avb prev plan
                for pidx in range(2 ** N):
                    if mx[pidx] == IMP:
                        continue
                        
                    pplan = self._idx2plan(pidx, N)
                    # print(pidx, pplan)
                    avb = True
                    for j in range(N):
                        if plan[j] and ((j > 0 and pplan[j-1]) or (j < N-1 and pplan[j+1])):
                            # cheat
                            avb = False
                            break
                    if avb:
                        # print(i, plan, pplan, sum_plan + mx[pidx])
                        nmx
[idx] = max(nmx[idx], sum_plan + mx[pidx])
            
            mx = nmx
        
        return max(mx)
