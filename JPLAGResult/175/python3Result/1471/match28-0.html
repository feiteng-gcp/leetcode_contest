class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        '''
        8 columns, 256 possible assignments per column
        consider each column from leftmost to rightmost
        - some of the 256 assignments is impossible due to broken seats (x & y == 0)
        - some are impossible due to previous column (x & y == 0) and ((x>>1) & y == 0)
        f(i, m) := max students can seat in first i columns
                   where ith column has seating mask m
        f(0, *) = 0
        f(i, x) := 0 if x contains a broken seat
                   max of {bits in x + f(i-1, y) where (x & y == 0) and ((x>>1) & y == 0)}
        '''
        rows = len(seats)
        cols = len(seats[0])
        broken = [0] * cols
        for i, row in enumerate(seats):
            for j, c in enumerate(row):
                if c == '#': broken[j] |= 1 << i
        t = 1 << rows
        best = [0]*t
        for broken_seats in broken:
            new_best = [0]*t
            for x in range(t):
                if x & broken_seats: continue
                student_count = bin(x).count('1')
                for y in range(t):
                    if x & y: continue
                    if (x>>1) & y: continue
                    if (x<<1) & y: continue
                    new_best[x] = max(new_best[x], student_count + best[y])
            best = new_best
        return max(best)
