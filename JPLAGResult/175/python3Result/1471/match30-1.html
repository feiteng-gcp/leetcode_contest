from typing import List

class Solution:
    def maxStudents(self, seats: List[List[str]]) -> int:
        def n_to_ls(num, n):
            ls = []
            for _ in range(n):
                ls.append(num % 2)
                num //= 2
            return ls[::-1]

        def ls_to_n(ls):
            num = 0
            for i in ls:
                num *= 2
                num += i
            return num

        def feasible(ls):
            for i in range(n-1):
                if ls[i] and ls[i+1]: return False
            return True

        def feasible2(ls1, ls2):
            for i in range
(n):
                if ls2[i] and (i>0 and ls1[i-1] or i<n-1 and ls1[i+1]): return False
            return True

        m, n = len(seats), len(seats[0])
        dp = dict()
        ls = []
        cnt = 0
        for i,s in enumerate(seats[0]):
            if s == '.':
                ls.append(i)
                cnt += 1
        for i in range(2**cnt):
            ls2 = n_to_ls(i, cnt)
            ls3 = [0] * n
            for ii,j in enumerate(ls):
                ls3[j] = ls2[ii]
            if feasible(ls3):
                num = ls_to_n(ls3)
                dp[num] = sum(ls3)
        for i in range(1, m):
            ls = []
            cnt = 0
            t = dp
            dp = dict()
            for i, s in enumerate(seats[i]):
                if s == '.':
                    ls.append(i)
                    cnt += 1
            for ii in range(2 ** cnt):
                ls2 = n_to_ls(ii, cnt)
                ls3 = [0] * n
                for iii,j in enumerate(ls):
                    ls3[j] = ls2[iii]
                if feasible(ls3):
                    for a in t:
                        num = ls_to_n(ls3)
                        a_ls = n_to_ls(a, n)
                        if feasible2(a_ls, ls3):
                            if num not in dp:
                                dp[num] = t[a] + sum(ls3)
                            else:
                                dp[num] = max(dp[num], t[a] + sum(ls3))

        res = 0
        for k in dp:
            res = max(res, dp[k])
        return res
