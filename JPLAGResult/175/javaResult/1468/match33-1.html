class Solution {
    public boolean checkIfExist(int[] arr) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < arr.length; i++) {
            int curr = arr[i];
            
if (set.contains(curr))
                return true;
            set.add(curr * 2);
            if (curr % 2 == 0)
                set.add(curr / 2);
        }
        return false;
    }
}
