class Solution {
    public boolean checkIfExist(int[] arr) {
        Set<Integer> set = new HashSet<>();
        int c = 0;
        for (int i : arr) {
            set.add(i);
            if (i == 0) {c++;}
            if (c == 2) {return true;}
        }
        for (int i : arr) {
            if (i != 0 && set.contains(2 * i)) {return true;}
        }
        return false;
    }
}
