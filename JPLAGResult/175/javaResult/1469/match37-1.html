class Solution {
    public int minSteps(String s, String t) {
        int freq[] = new int[30];
        for (char c: s.toCharArray()){
            freq[c-'a']++;
        }
        
for (char c: t.toCharArray()){
            freq[c-'a']--;
        }
        int ans = 0;
        for (int i = 0; i<=25; i++){
            ans += Math.abs(freq[i]);
        }
        return ans/2;        
    }
}
