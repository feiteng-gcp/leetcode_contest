class Solution {
    public int minSteps(String s, String t) {
        int[] cnt = new int[26];
        for(char cs : s.toCharArray()){
            cnt[cs-'a']++;
        }
        for(char ct: t.toCharArray()){
            cnt[ct - 'a']--;
        }
        int total = 0;
        for(int c: cnt){
            total += Math.abs(c);
        }
        return total/2;
    }
}
