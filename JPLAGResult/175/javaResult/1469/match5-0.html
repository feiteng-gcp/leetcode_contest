class Solution {
    public int minSteps(String s, String t) {
        int[] freqsS = getFreq(s);
        int[] freqsT = getFreq(t);
        int res = 0;
        for (int i = 0; i < 26; i++) {
            res += Math.max(0, freqsS[i] - freqsT[i]);
        }
        return res;
    }
    
    private int[] getFreq(String s) {
        int[] freqs = new int[26];
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            freqs[ch - 'a']++;
        }
        return freqs;
    }
}
