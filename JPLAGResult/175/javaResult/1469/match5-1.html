class Solution {
    public int minSteps(String s, String t) {
        int[] sc = count(s);
        int[] tc = count(t);
        
        int ans = 0;
        for (int i = 0; i < 26; i++) {
            ans += Math.abs(sc[i] - tc[i]);
        }
        return ans/2;
        
    }
    
    private int[] count(String s) {
        int[] sc = new int[26];
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            sc[c-'a']++;
        }
        return sc;
    }
}
