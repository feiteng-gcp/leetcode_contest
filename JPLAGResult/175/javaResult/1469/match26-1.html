class Solution {
    public int minSteps(String s, String t) {
        int[] sn = new int[26];
        int[] tn = new int[26];
        for(char ss: s.toCharArray()) {
            ++(sn[ss - 'a']);
        }
        for(char tt: t.toCharArray()) {
            ++(tn[tt - 'a']);
        }
        int same = 0;
        for(int i = 0; i < 26; ++i) {
            same += Math.min(sn[i], tn[i]);
        }
        return s.length() - same;
    }
}
