class Solution {
    public int minSteps(String s, String t) {
		int[] sa = new int[26];
		int[] ta = new int[26];
		for (char c : s.toCharArray()) {
			sa[c-'a']++;
		}
		for (char c : t.toCharArray()) {
			ta[c-'a']++;
		}
		int count = 0;
		for (int i = 0; i < 26; i++) {
			count += Math.abs(sa[i] - ta[i]);
		}
		return count / 2;
    }
}
