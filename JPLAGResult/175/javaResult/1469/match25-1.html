class Solution {
    
    public int minSteps(String s, String t) {
    
        int freq[]=new int[26];
        
        for(char item: s.toCharArray())
            freq[item-'a']+=1;
        
        for(char item: t.toCharArray())
            freq[item-'a']-=1;
        
        int sum=0;
        for(int i=0; i<26; i++)
            sum+=Math.abs(freq[i]);
        
        return sum/2;
    }
}
