class Solution {
    public int minSteps(String s, String t) {
        int n = s.length();
        int[] countMap = new int[26];
        for (int i = 0; i < n; ++i) {
            countMap[s.charAt(i) - 'a'] += 1;
            countMap[
t.charAt(i) - 'a'] -= 1;
        }
        int count = 0;
        for (int i = 0; i < 26; ++i) {
            count += Math.abs(countMap[i]);
        }
        return count / 2;
    }
}
