class Solution {
    public int minSteps(String s, String t) {
        int[] cntT = new int[26];
        for(char c: t.toCharArray()){
            cntT[c - 'a']++;
        }
        int[] cntS = new int[26];
        for(char c: s.toCharArray()){
            cntS[c - 'a']++;
        }
        int res = 0;
        for(int i = 0 ; i < 26; i++){
            res += Math.max(0, cntT[i] - cntS[i]);
        }
        return res;
    }
}
