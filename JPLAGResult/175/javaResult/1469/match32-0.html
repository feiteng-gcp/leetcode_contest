class Solution {
    public int minSteps(String s, String t) {
        int[] tChars = new int[128];
        int[] sChars = new int[128];
        for (int i = 0; i < s.length(); i++) {
            sChars[s.charAt(i)]++;
            tChars[t.charAt(i)]++;
        }
        int res = 0;
        for (int i = 'a'; i <= 'z'; i++) {
            res += Math.max(0, sChars[i] - tChars[i]);
        }
        return res;
    }
}
