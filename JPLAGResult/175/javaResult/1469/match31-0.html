class Solution {
    public int minSteps(String s, String t) {
        int[] a = new int[256], b = new int[256];
        for (int i=0; i<s.length(); i++) {
            a[s.charAt(i)]++;
            b[t.charAt(i)]++; // Fixed-1: s->t
        }
        
        int total = 0;
        for (int i=0; i<a.length; i++) {
            // System.out.println(a[i] + "\t" + b[i]);
            total += Math.abs(a[i] - b[i]);
        }
        
        return total / 2;
    }
}
