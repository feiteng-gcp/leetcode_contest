	class Solution {
	    public int minSteps(String s, String t) {
	        int[] f = new int[26];
	        for(char c : s.toCharArray()){
	        	f[c-'a']++;
	        }
	        for(char c : t.toCharArray()){
	        	f[c-'a']--;
	        }
	        int ret = 0;
	        for(int i = 0;i < 26;i++){
	        	ret += Math.max(0, f[i]);
	        }
	        return ret;
	    }
	}
