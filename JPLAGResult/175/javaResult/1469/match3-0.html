class Solution {
    public int minSteps(String s, String t) {
        int[] countS = new int[26], countT = new int[26];
        
        for (int i = 0; i < s.length(); i++) {
            countS[s.charAt(i) - 'a']++;
        }
        
        for (int i = 0; i < t.length(); i++) {
            countT[t.charAt(i) - 'a']++;
        }
        
        int dif = 0;
        for (int i = 0; i < 26; i++) {
            dif += Math.max(countS[i] - countT[i], 0);
        }
        
        return dif;
    }
}
