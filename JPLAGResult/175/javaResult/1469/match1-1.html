class Solution {
    public int minSteps(String s, String t) {
        int d1 [] = new int[26];
        int d2 [] = new int[26];
        
        for(int i=0;i<s.length();++i){
            d1[s.charAt(i)-'a']++;
        }
        for(int i=0;i<t.length();++i){
            d2[t.charAt(i)-'a']++;
        }
        int ss = 0;
        for(int i=0;i<26;++i){
            ss += Math.abs(d1[i]-d2[i]);
        }
        return ss/2;
        
    }
}
