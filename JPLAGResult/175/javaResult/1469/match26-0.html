class Solution {
    public int minSteps(String s, String t) {
        int[] c1 = new int[26];
        int[] c2 = new int[26];
        for (char c : s.toCharArray()) {
            c1[c-'a']++;
        }
        for (char c : t.toCharArray()) {
            c2[c-'a']++;
        }
        int count = 0;
        for (int i = 0; i < 26; i++) {
            count += Math.min(c1[i], c2[i]);
        }
        return s.length() - count;
    }
}
