class Solution {
    public int minSteps(String s, String t) {
        int len = s.length();
        int[] dp1 = new int[26];
        for (int i = 0; i < len; i++) {
            dp1[s.charAt(i) - 'a']++;
        }

        int[] dp2 = new int[26];
        for (int i = 0; i < len; i++) {
            dp2[t.charAt(i) - 'a']++;
        }

        int sum = 0;
        for (int i = 0; i < 26; i++) {
            sum += Math.abs(dp1[i] - dp2[i]);
        }
        return sum / 2;

    }
}
