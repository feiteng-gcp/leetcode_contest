class Solution {
	public int minSteps(String s, String t) {
		int[] scnt = new int[26];
		int[] tcnt = new int[26];
		for (int i = 0; i < s.length(); i++) {
			scnt[s.charAt(i) - 'a']++;
			tcnt[t.charAt(i) - 'a']++;
		}
		int res = 0;
		for (int i = 0; i < 26; i++) {
			res += Math.abs(scnt[i] - tcnt[i]);
		}
		return res / 2;
	}
}
