class Solution {
    public int maxStudents(char[][] seats) {
        int m = seats.length;
        int n = seats[0].length;
        int[] dp = new int[1<<n];
        for (int i = 0; i < m; i++) {
            int[] tmp = new int[1<<n];
            char[] arr = seats[i];
            int check = 0;
            for (int j = 0; j< n; j++) {
                if (arr[j] == '#') {
                    check |= 1 << (n-1-j);
                }
            }
            for (int k = 0; k < (1<<n); k++) {
                if ((k & check) != 0 || ((k>>1) & k) != 0 || ((k<<1) & k) != 0) {
                    tmp[k] = -1;
                    continue;
                }
                for (int l = 0; l < (1<<n); l++) {
                    if (dp[l] != -1 && ((k>>1) & l) == 0 && ((k<<1) & l) == 0) {
                        tmp[k] = Math.max(tmp[k], dp[l] + Integer.bitCount(k));
                    }
                }
            }
            dp = tmp;
        }

        int max = 0;
        for (int v : dp) {
            max = Math.max(max, v);
        }
        return max;
    }
}
