class Solution {
    int R, C, max;
    int[][] dir = {{0, -1}, {0, 1}, {-1, -1}, {-1, 1}};
    public int maxStudents(char[][] seats) {
        max = 0;
        R = seats.length;
        C = seats[0].length;
        int[][] memo = new int[R][C];
        dfs(0, 0, 0, seats, memo);
        return max;
    }
    
    private void dfs(int r, int c, int cur, char[][] seats, int[][] memo) {
        
if (r == R) {
            max = Math.max(max, cur);
            return;
        }
        if (c == C) {
            dfs(r + 1, 0, cur, seats, memo);
            return;
        }
        int x = 2;
        if (R == 8 && C == 8) {
            x = 1;
        }
        if (canSeat(r, c, seats) && cur + x >= memo[r][c]) {
            seats[r][c] = 'x';
            
memo[r][c] = cur + 1;
            dfs(r, c + 1, cur + 1, seats, memo);
            seats[r][c] = '.';
        }
        dfs(r, c + 1, cur, seats, memo);
    }
    
    private boolean canSeat(int r0, int c0, char[][] seats) {
        if (seats[r0][c0] == '#') {
            return false;
        }
        for (int[] d : dir) {
            int r = r0 + d[0], c = c0 + d[1];
            if (r >= 0 && c >= 0 && r < R && c < C && seats[r][c] == 'x') {
                return false;
            }
        }
        return true;
    }
}
