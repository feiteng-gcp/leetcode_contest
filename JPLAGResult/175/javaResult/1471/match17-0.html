class Solution {
    public int maxStudents(char[][] seats) {
        int m = seats.length, n = seats[0].length;
        int[][] a = new int[m][1<<n];
        int[][] cnt = new int[m][1<<n];
        for (int k=0; k<m; k++) {
            for (int x=0; x<(1<<n); x++) {
                boolean valid = true;
                int count = 0;
                for (int j=0; j<n; j++) {
                    int val = x & (1<<j);
                    if (val == 0 && seats[k][j] == '#') {
                        valid = false;
                        break;
                    }                               
                    if (val != 0 && seats[k][j] == '.') {
                        if (j > 0 && (x & (1 << (j-1))) != 0 && seats[k][j-1] == '.') {
                            valid = false;
                            break;
                        }
                        count++;
                    }
                }
                if (!valid) cnt[k][x] = -10000;
                else cnt[k][x] = count;               
            }
        }
        
        int res = 0;
        for (int k=0; k<m; k++) {
            for (int x=0; x<(1<<n); x++) {
                if (k == 0) a[k][x] = cnt[k][x];
                else {
                    for (int y=0; y<(1<<n); y++) {
                        boolean valid = true;
                        for (int j=0; j<n; j++) {
                            if ((x & (1 << j)) != 0 && seats[k][j] == '.') {
                                if (j > 0 && (y & (1<< (j-1))) != 0 && seats[k-1][j-1] == '.' ||
                                    j < n-1 && (y & (1<<(j+1))) != 0 && seats[k-1][j+1] == '.') {
                                    valid = false;
                                    break;
                                } 
                            }
                        }
                        if (valid) {
                            a[k][x] = Math.max(a[k][x], a[k-1][y] + cnt[k][x]);
                        }
                    }
                }
                res = Math.max(res, a[k][x]);
            }
        }
        return res;
    }
}
