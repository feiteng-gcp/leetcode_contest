class Solution {
    boolean[][] take;
    int[][][] dp;
    int n;
    int m;
    
    public int maxStudents(char[][] seats) {
        n = seats.length;
        m = seats[0].length;
        take = new boolean[n][m];
        dp = new int[8][8][512];
        
for(int i = 0; i < 8; i++){
            for(int j = 0; j < 8; j++){
                Arrays.fill(dp[i][j], -1);
            }
        }
        
        int res = dfs(seats, n - 1, m - 1);
        return res;
    }
    
    private int toKey(int x, int y){
        int res = 0;
        if(x == n - 1){
            for(int i = y + 1; i < m; i++){
                res = res << 1;
                res += take[n - 1][i] ? 1 : 0;
            }
            return res;
        }
        
        if(y == m - 1){
            for(int i = 0; i < m; i++){
                res = res << 1;
                res += take[x + 1][i] ? 1 : 0;
            }
            return res;
        }
        
        for(int i = 0; i <= y + 1; i++){
            res = res << 1;
            res += take[x + 1][i] ? 1 : 0;
        }
        
        for(int i = y + 1; i < m; i++){
            res = res << 1;
            res += take[x][i] ? 1 : 0;
        }
        
        return res;
    }
    
    private int dfs(char[][] seats, int x, int y){
        if(x == -1){
            return 0;
        }
        
        int key = toKey(x, y);
        if(dp[x][y][key] == -1){
            int nextX = y == 0 ? x - 1 : x;
            int nextY = y == 0 ? m - 1 : y - 1;
            
            int max = dfs(seats, nextX, nextY);
            
            if(check(seats, x, y)){
                take[x][y] = true;
                max = Math.max(max, 1 + dfs(seats, nextX, nextY));
                take[x][y] = false;
            }
            dp[x][y][key] = max;
        }
        
        return dp[x][y][key];
    }
    
    
    
    private boolean check(char[][] seats, 
int x, int y){
        if(seats[x][y] == '#'){
            return false;
        }
        
        if(y != 0 && take[x][y - 1]){
            return false;
        }
        
        if(y != m - 1 && take[x][y + 1]){
            return false;
        }
        
        if(x != n - 1){
            
if(y != 0 && take[x + 1][y - 1]){
                return false;
            }
            
            if(y != m - 1 && take[x + 1][y + 1]){
                return false;
            }
        }
        
        return true;
    }
}
