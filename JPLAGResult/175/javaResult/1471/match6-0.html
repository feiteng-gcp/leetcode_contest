class Solution {
    public int maxStudents(char[][] seats) {
        int l1 = seats.length;
        int l2 = seats[0].length;
        
        int dp[][] =new int[l1+1][1<<l2];
        
        for(int u[]:dp){
            Arrays.fill(u,-1000000);
        }
        
        for(int st1 = 0;st1<(1<<l2);++st1){
            dp[0][st1] = 0;   
        }
        
        
        for(int i=1;i<=l1;++i){
            ot:for(int st1 = 0;st1<(1<<l2);++st1){
                
                for(int j=1;j<l2;++j){
                    if(st1<<~j<0&&st1<<~(j-1)<0){
                        continue ot;
                    }
                }
                for(int j=0;j<l2;++j){
                     if(st1<<~j<0 && seats[i-1][j]=='#'){
                         continue ot;
                     }   
                }
                
                
                
                ot1:for(int st2 = 0;st2<(1<<l2);++st2){
                
                    // for(int j=1;j<l2;++j){
                    //     if(st2<<~j<0&&st2<<~(j-1)<0){
                    //         continue ot1;
                    //     }
                    // }
                    // for(int j=0;j<l2;++j){
                    //  if(st2<<~j<0 && i>=2&&seats[i-2][j]=='#'){
                    //      continue ot1;
                    //  }   
                    // }   
                    
                    for(int j=0;j<l2;++j){
                     if(st1<<~j<0 && ((j-1>=0&&st2<<~(j-1)<0)||(j+1<l2&&st2<<~(j+1)<0))){
                         continue ot1;
                     }   
                    }
                    int num  =Integer.bitCount(st1);
                    
                    dp[i][st1] = Math.max(dp[i][st1],dp[i-1][st2]+num);
                
                
                
                }
        }
        
    }
        int r = 0;
        for(int st2 = 0;st2<(1<<l2);++st2){
            r=  Math.max(r,dp[l1][st2]);
        }
        return r;
        
        
}
}
