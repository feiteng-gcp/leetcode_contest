class Solution {
    
    int r, c;
    int[][] memo;
    List<Integer> masks;
    
    public int maxStudents(char[][] seats) {
        r = seats.length;
        c = seats[0].length;
        memo = new int[r][1<<c];
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < r; i++) {
            Arrays.fill(memo[i], -1);
            for (int j = 0; j < c; j++) {
                if (seats[i][j] == '.') {
                    list.add(getId(i, j));
                }
            }
        }
        
        return getMax(seats, 0, 0);
    }
    
    private int getMax(char[][] seats, int curRow, int mask) {
        if (curRow == r) {
            return 0;
        }
            //System.out.println(String.format("%s-%s", curRow, mask));
        if (memo[curRow][mask] != -1){
            return memo[curRow][mask];
        }
        masks = new LinkedList<>();
        backtrack(seats[curRow], 0, mask, 0);
            //System.out.println(String.format("[!!!!]%s-%s", curRow, masks.size()));
        int res = 0;
        for (int m : masks) {
            res = Math.max(res, Integer.bitCount(m) + getMax(seats, curRow + 1, m));
        }
        memo[curRow][mask] = res;
        return res;
    }
    
    private void backtrack(char[] seats, int cur, int prev, int mask) {
        if (cur == c) {
            masks.add(mask);
            //System.out.println(String.format("[!]%s %s %s", cur, Integer.toBinaryString(prev), Integer.toBinaryString(mask)));
            return;
        }
        if (seats[cur] == '#') {
            backtrack(seats, cur + 1, prev, mask);
            return;
        }
        // left - cur - 1, right cur + 1
        if (cur > 0 && (((mask & (1 << (cur-1))) > 0) || (prev & (1 << (cur-1))) > 0)) {
                backtrack(seats, cur + 1, prev, mask);
        } else if (cur < r - 1 && ((prev & (1 << (cur+1))) > 0)) {
                backtrack(seats, cur + 1, prev, mask);
        } else {
            //System.out.println(String.format("[1]%s %s %s", cur, Integer.toBinaryString(prev), Integer.toBinaryString(mask)));
            mask |= (1 << (cur));
            backtrack(seats, cur + 1, prev, mask);
            mask ^= (1 << (cur));
            backtrack(seats, cur + 1, prev, mask);
        }
    }
    
    private int getId(int x, int y) {
        return c * x + y;
    }
}
