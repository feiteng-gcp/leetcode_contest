class Solution {
    int[][] map;
    int n, m;
    int[] right;
    int[] left;
    
    boolean[] visit;
    
    int[] fx = {-1, 0, 0, -1, 1, 1};
    int[] fy = {1, 1, -1, -1, 1, -1};
    
    public boolean dfs(int u) {
        for(int v = 0; v < m; v++) {
            if(map[u][v] == 1 && !visit[v]) {
                visit[v] = true;
                if(right[v] == -1 || dfs(right[v])) {
                    right[v] = u;
                    left[u] = v;
                    return true;
                
}
            }
        }
        return false;
    }
    
    public int maxStudents(char[][] seats) {
        n = 0;
        m = 0;
        
int[][] mark = new int[seats.length][seats[0].length];
        
        
        for(int j = 0; j < seats[0].length; j++) {
            for(int i = 0; i < seats.length; i++) {
                if(seats[i][j] == '.') {
                    if(j % 2 == 0) {
                        mark[i][j] = n;
                        n++; 
                    } else {
                        
mark[i][j] = m;
                        m++;
                    }
                }
            }
        }
        
        map = new int[n][m];
        int cnt1 = 0, cnt2 = 0;
        for(int i = 0; i < seats.length; i++) {
            for(int j = 0; j < seats[0].length; j+=2) {
                if(seats[i][j] == '.') {
                    for(int k = 0; k < 6; k++) {
                        int u = i + fx[k], v = j + fy[k];
                        if(u >= 0 && u < seats.length && v >= 0 && v < seats[0].length && seats[u][v] == '.') {
                            map[mark[i][j]][mark[u][v]] = 1;
                        }
                    }
                }
            }
        }
        
        left = new int[n];
        right = new int[m];
        for(int i = 0; i < n; i++) left[i] = -1;
        for(int i = 0; i < m; i++) right[i] = -1;
        
        int cnt = 0;
        for(int i = 0; i < n; i++) {
            if(left[i] == -1) {
                visit = new boolean[m];
                if(dfs(i)) cnt++
;
            }
        }
        
        return n + m - cnt;
    }
}
