class Solution {
    int cr, cc, l;
    char[][] mt;
    public int maxStudents(char[][] seats) {
        cr = seats.length;
        cc = seats[0].length;
        l = cr * cc;
        mt = seats;
        // dp = new int[l];
        return dfs(0, 0, 0);
    
}
    int dfs(int i, int acc, int ava){
        if(i == l) return 0;
        int r = i / cc, c = i % cc;
        if(mt[r][c] == '#'){
            // nothing to do move to next
            return dfs(i+1, acc, ava);
        }
        else{
            // check the previous seats
            int max = 0;
            if(has(r-1, c-1) && has(r-1, c+1) && has(r, c-1)){
                mt[r][c] = 'S';
                max = Math.max(max, dfs(i+1, acc + 1, ava + 1) + 1);
                mt[r][c] = '.';
            }
            if(acc * 2 + 2 >= ava){
                max = Math.max(max, dfs(i+1, acc, ava + 1));
            }
            return max;
        }
    }
    boolean has(int r, int c){
        return r < 0 || r >= cr || c < 0 || c >= cc || mt[r][c] != 'S';
    }
}
