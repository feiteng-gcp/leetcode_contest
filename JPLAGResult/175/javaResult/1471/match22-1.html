class Solution {
    public int maxStudents(char[][] seats) {
        int m = seats.length;
        int n = seats[0].length;
        int limit = 1 << n;
        int[] prevRow = new int[limit];
        
prevRow[0] = 1;
        for (int i = 0; i < m; i++) {
            int[] newRow = new int[limit];
            for (int mask = 0; mask < limit; mask++) {
                int added = 0;
                boolean valid = true;
                for (int j = 0; j < n; j++) {
                    if ((mask & (1 << j)) == 0) continue;
                    added++;
                    if (seats[i][j] == '#' || (mask & (1 << (j + 1))) != 0) {
                        valid = false;
                        break;
                    }
                }
                if (valid) {
                    for (int k = 0; k < limit; k++) if (prevRow[k] > 0 && prevRow[k] + added > newRow[mask]) {
                        boolean valid2 = true;
                        for (int j = 0; j < n; j++) {
                            if ((k & (1 << j)) == 0) continue;
                            if (j > 0 && (mask & (1 << (j - 1))) != 0 || (mask & (1 << (j + 1))) != 0) {
                                valid2 = false;
                                break;
                            }
                        }
                        if (valid2) newRow[mask] = prevRow[k] + added;
                    }
                }
            
}
            prevRow = newRow;
        }
        int result = 0;
        for (int i = 0; i < limit; i++) result = Math.max(result, prevRow[i]);
        return result - 1;
    }
}
