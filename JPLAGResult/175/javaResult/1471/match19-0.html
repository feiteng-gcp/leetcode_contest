class Solution {
    private boolean isValidRow(int x, int n) {
        int pat = 3;
        for (int i = 0; i < n; i++) {
            if ((x&pat) == pat) {
                return false;
            }
            pat <<= 1;
        }
        return true;
    }
    
    private boolean isGoodNext(int x, int y, int n) {
        for (int i = 0; i < n; i++) {
            if ((y&(1<<i)) == 0) {
                continue;
            }
            int t = i-1;
            if (t >= 0 && (x&(1<<t)) > 0) {
                return false;
            }
            
            t = i+1;
            if (t < n && (x&(1<<t)) > 0) {
                return false;
            }
        }
        return true;
    }
    
    private int countBit(int x) {
        int ret = 0;
        while(x > 0) {
            ret++;
            x &= x-1;
        }
        return ret;
    }
    
    public int maxStudents(char[][] seats) {
        int m = seats.length;
        int n = seats[0].length;
        List<Integer> validRows = new LinkedList<>();
        int[] masks = new int[m];
        int[] counters = new int[1<<n];
        int[][] dp = new int[m][1<<n];
        
        /*
        int maxx = 0;
        for (int i = 0; i < 256; i++) {
            int cnt = 0;
            if (isValidRow(i, 8) == true) {
                validRows.add(i);
            }
            
            for (int j = 0; j < 256; j++) {
                if (i != j && isGoodNext(i, j, 8) == true) {
                    cnt++;
                }
            }
            maxx = Math.max(cnt, maxx);
        }*/
        
        for (int i = 0; i < m; i++) {
            int mask = 0;
            for (int j = 0; j < n; j++) {
                mask <<= 1;
                if (seats[i][j] == '#') {
                    mask++;
                } 
            }
            masks[i] = mask;
        }
        
        for (int i = 0; i < (1<<n); i++) {
            counters[i] = countBit(i);
            if (isValidRow(i, n) == true) {
                validRows.add(i);
                if ((masks[0]&i) == 0) {
                    dp[0][i] = counters[i];
                }
            }
        }
        
        for (int lvl = 1; lvl < m; lvl++) {
            for (int i = 0; i < validRows.size(); i++) {
                Integer integerY = validRows.get(i);
                int y = integerY.intValue();
                int counter = counters[y];
                if ((masks[lvl]&y) != 0) {
                    continue;
                }
                for (int j = 0; j < validRows.size(); j++) {
                    Integer integerX = validRows.get(j);
                    int x = integerX.intValue();
                    if ((masks[lvl-1]&x) == 0 && isGoodNext(x, y, n) == true) {
                        dp[lvl][y] = Math.max(dp[lvl-1][x] + counter, dp[lvl][y]);
                    }
                }
                
                // System.out.println(lvl + ": " + y + " : " + dp[lvl][y]);
            }
        }
        
        int ans = 0;
        for (int i = 0; i < (1<<n); i++) {
            if (isValidRow(i, n) == true) {
                ans = Math.max(ans, dp[m-1][i]);
            }
        }
        

        // System.out.println(Arrays.toString(masks));
        // System.out.println(validRows.size() + " " + ans);
        return ans;
    }
}
