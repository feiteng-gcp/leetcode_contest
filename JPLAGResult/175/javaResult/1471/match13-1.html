class Solution {

    public int maxStudents(char[][] seats) {
      int n = seats.length, m = seats[0].length;
      boolean[][] isSeat = new boolean[n][m];
      
Map<String, Integer> memo = new HashMap<>();
      return dfs(seats, isSeat, 0, 0, memo);
    }

    private int dfs(char[][] seats, boolean[][] isSeat, int r, int c, Map<String, Integer> memo) {
      int n = seats.length, m = seats[0].length;
      if (c == m) {
        c = 0;
        r++;
      }
      if (r == n) {
        return 0;
      }
      String mark = Arrays.toString(isSeat[r]);
      if (r - 1 >= 0) {
        mark += Arrays.toString(isSeat[r - 1]);
      }
      mark += r + " " + c;
      if(memo.containsKey(mark)){
        return memo.get(mark);
      }
      int res = dfs(seats, isSeat, r, c + 1, memo);
      if (seats[r][c] == '.' && isValid(seats, isSeat, r - 1, c - 1) && isValid(seats, isSeat,
          r - 1, c + 1) && 
isValid(seats, isSeat, r, c - 1)) {
        isSeat[r][c] = true;
        res = Math.max(res, 1 + dfs(seats, isSeat, r, c + 1, memo));
        isSeat[r][c] = false;
      }
      
memo.put(mark, res);
      return res;
    }

    private boolean isValid(char[][] seats, boolean[][] isSeat, int r, int c) {
      int n = seats.length, m = seats[0].length;
      return r < 0 || c < 0 || r >= n || c >= m || seats[r][c] == '#' || !isSeat[r][c];
    }
 }
