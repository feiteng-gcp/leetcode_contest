class Solution {

	private boolean contain(int set, int j) {
		return (set & (1 << j)) > 0;
	}
	private int countOne(int set) {
		int res = 0;
		while (set > 0) {
			res++;
			set &= set - 1;
		}
		return res;
	}

	int[][] dp;
	char[][] seats;
	int n, m;

	public int maxStudents(char[][] seats) {
		n = seats.length;
		m = seats[0].length;
		this.seats = seats;

		dp = new int[n][1 << m];
		for (int i = 0; i < n; i++) Arrays.fill(dp[i], -1);


		int res = 0;
		for (int set = 0; set < (1 << m); set++) {
			if (!check(0, 0, set)) continue;
			res = Math.max(res, s
olve(0, set));
		}
		return res;
	}

	private boolean check(int i, int pset, int set) {
		for (int j = 0; j < m; j++) if (contain(set, j)) {
			if (seats[i][j] == '#')
				return false;
			if (j-1 >= 0 && (contain(pset, j-1) || contain(set, j-1))) return false;
			if (j+1 < m && (contain(pset, j+1) || contain(set, j+1))) return false;
		}

		return true;
	}

	private int solve(int idx, int pset) {
		if (idx == n-1) {
			return countOne(pset);
		}
		if (dp[idx][pset] != -1) return dp[idx][pset];

		int cnt = countOne(pset);
		int res = cnt;
		for (int set = 0; set < (1 << m); set++) {
			if (!check(idx+1, pset, set)) continue;
			res = Math.max(res, solve(idx+1, set) + cnt);
		}
		return dp[idx][pset] = res;
	}
}
