class Solution {
    public int maxStudents(char[][] seats) {
        Map<String, Integer> memo = new HashMap<>();
        return calc(seats, 0, "", memo);
    }
    
    private int calc(char[][] seats, int row, String prev, Map<String, Integer> memo) {
        if (row >= seats.length) return 0;
        
        String key = row + prev;
        
        if (memo.containsKey(key)) return memo.get(key);
        
        int max = backtrack(seats, row, 0, memo);
        
        memo.put(key, max);
        return max;
    }
    
    private int backtrack(char[][] seats, int row, int start, Map<String, Integer> memo) {
        if (start == seats[0].length) {
            return calc(seats, row + 1, new String(seats[row]), memo);
        }
        
        int max = 0;
        if (canSeat(seats, row, start)) {
            seats[row][start] = 's';
            max = Math.max(max, backtrack(seats, row, start + 1, memo) + 1);
            seats[row][start] = '.';
        }
        
        max = Math.max(max, backtrack(seats, row, start + 1, memo));
        return max;
    }
    
    private boolean canSeat(char[][] seats, int r, int c) {
        return seats[r][c] == '.' && ok(seats, r, c - 1) && ok(seats, r, c + 1)
            && ok(seats, r - 1, c - 1) && ok(seats, r - 1, c + 1);
    }
    
    private boolean ok(char[][] seats, int r, int c) {
        int rows = seats.length, cols = seats[0].length;
        if (r < 0 || c < 0 || r >= rows || c >= cols) return true;
        return seats[r][c] != 's';
    }
}
