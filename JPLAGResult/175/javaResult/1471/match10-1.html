class Solution {
    public int maxStudents(char[][] seats) {
        int R = seats.length;
        
int C = seats[0].length;
        int size = (1 << C);
        int[] lastRow = new int[size];
        for (int r = 0; r < R; r++) {
            int[] row = new int[size];
            int mask = toMask(seats[r]);
            for (int i = 0; i < size; i++) {
                if (!hasConsecutiveBits(i) && Integer.bitCount(mask | i) == Integer.bitCount(mask)) {
                    for (int j = 0; j < size; j++) {
                        if (!hasConsecutiveBits(i | j)) 
{
                            row[i] = Math.max(row[i], lastRow[j] + Integer.bitCount(i));
                        }
                    }
                }
            }
            lastRow = row;
        }
        return Arrays.stream(lastRow).max().getAsInt();
    }

    private boolean hasConsecutiveBits(int n) {
        for (int i = 0; i < 8; i++) {
            if ((n & (1 << i)) > 0 && (n & (1 << (i + 1))) > 0) {
                return true;
            }
        }
        return false;
    }

    private int toMask(char[] a) {
        
int m = 0;
        for (int i = 0; i < a.length; i++) {
            m <<= 1;
            if (a[i] == '.') {
                m |= 1;
            }
        }
        return m;
    }
}
