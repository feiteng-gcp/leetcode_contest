class Solution {
    public int maxStudents(char[][] seats) {
        final int N = seats.length;
        final int M = seats[0].length;
        
        int[] dp = new int[1 << M];
        for (int i = 0; i < N; ++i) {
            int[] next = new int[1 << M];
            for (int j = 0; j < (1 << M); ++j) {
                if (!valid(j, seats[i])) {
                    continue;
                }
                for (int k = 0; k < (1 << M); ++k) {
                    if (!validRows(k, j, M)) {
                        continue;
                    }
                    next[j] = Math.max(next[j], dp[k] + Integer.bitCount(j));
                }
            }
            dp = next;
        }
        int best = 0;
        for (int v : dp) {
            best = Math.max(best, v);
        }
        return best;
    }
    
    private static int getBit(int x, int b) {
        if (b < 0) {
            return 0;
        }
        return (x >> b) & 1;
    }
    
    private static boolean validRows(int front, int back, int M) {
        for (int i = 0; i < M; ++i) {
            if (getBit(back, i) == 0) {
                continue;
            }
            int neighbors = getBit(front, i - 1) + getBit(front, i + 1);
            if (neighbors > 0) {
                return false;
            }
        }
        return true;
    }
    
    private static boolean valid(int mask, char[] row) {
        for (int i = 0; i < row.length; ++i) {
            int b = (mask >> i) & 1;
            if (row[i] == '#' && getBit(mask, i) == 1) {
                return false;
            }
        }
        for (int i = 0; i < row.length; ++i) {
            if (getBit(mask, i) == 0) {
                continue;
            }
            int neighbors = getBit(mask, i - 1) + getBit(mask, i + 1);
            if (neighbors > 0) {
                return false;
            }
        }
        return true;
    }
}
