class Solution {

	private boolean validateRow(char[][] seats , int row , int bitmap) {
		int m = seats[0].length;
		for (int i = 0;i < m;i ++) {
			if ((bitmap & (1 << i)) != 0) {
				if (seats[row][i] == '#') {						
					return false;
				} 
				if (i - 1 >= 0 && (bitmap & (1 << (i - 1))) != 0 && seats[row][i - 1] == '.') {
					return false;
				}
				if (i + 1 < m && (bitmap & (1 << (i + 1))) != 0 && seats[row][i + 1] == '.') {
					return false;
				}
			}
		}
		return true;
	}
	
	private boolean validate(char[][] seats , int row , int bitmap1 , int bitmap2) {
		if (validateRow(seats , row , bitmap1) == false) {
			return false;
		}
		int m = seats[0].length;		
		for (int i = 0;i < m;i ++) {
			if ((bitmap2 & (1 << i)) != 0) {
				if (i - 1 >= 0 && (bitmap1 & (1 << (i - 1))) != 0) {
					return false;
				}
				if (i + 1 < m && (bitmap1 & (1 << (i + 1))) != 0) {
					return false;					
				}
			}
		}
		return true;		
	}		
	
	private int getTotal(int bitmap , int m) {
		int ans = 0;
		for (int i = 0;i < m;i ++) {
			if ((bitmap & (1 << i)) != 0) {
				ans ++;
			}
		}
		return ans;		
	}
	
	private int[] dp1 , dp2;	
	
    public int maxStudents(char[][] seats) {
    	int i , j , n = seats.length , m = seats[0].length;
    	dp1 = new int[(1 << m)];
    	dp2 = new int[(1 << m)];
    	Arrays.fill(dp1 , - 1);
    	Arrays.fill(dp2 , - 1);
    	for (i = n - 1;i >= 0;i --) {
    		if (i == n - 1) {
	    		for (j = 0;j < (1 << m);j ++) {
	    			int cnt = getTotal(j , m);    			
	    			if (validateRow(seats , i , j)) {
	    				dp2[j] = cnt;	    				
	    			}
	    		}
    		} else {
    			for (int bitmap1 = 0;bitmap1 < (1 << m);bitmap1 ++) {
    				for (int bitmap2 = 0;bitmap2 < (1 << m);bitmap2 ++) {
    					int cnt = getTotal(bitmap2 , m);    					
    					if (dp1[bitmap1] >= 0 && validate(seats , i , bitmap2 , bitmap1)) {
    						dp2[bitmap2] = Math.max(dp2[bitmap2] , dp1[bitmap1] + cnt);    						
    					}
    				}
    			}
    		}
    		for (j = 0;j < (1 << m);j ++) {
    			dp1[j] = dp2[j];
    			dp2[j] = - 1;
    		}
    	}
    	int ans = 0;
    	for (i = 0;i < (1 << m);i ++) {
    		ans = Math.max(ans , dp1[i]);
    	}
    	return ans;    	
    }
}
