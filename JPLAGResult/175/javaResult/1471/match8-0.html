class Solution {
    public int maxStudents(char[][] seats) {
        boolean[][] validCross = new boolean[256][256];
        boolean[] valid = new boolean[256];
        int[] num = new int[256];
        for (int i = 0; i < 256; i++) {
            num[i] = numOfOne(i);
            valid[i] = canCheatSame(i);
            for (int j = 0; j < 256; j++) {
                validCross[i][j] = canCheat(i, j);
            }
        }
        int n = seats.length;
        int m = seats[0].length;
        int[] rows = new int[n];
        for (int i = 0; i < n; i++) {
            int r = 0;
            for (int j = 0; j < m; j++) {
                r = r * 2 + (seats[i][j] == '#' ? 1 : 0);
            }
            rows[i] = r;
        }
        int mm = 1 << m;
        // System.out.println(mm);
        int[][] dp = new int[n][mm];
        for (int i = 0; i < mm; i++) {
            if (valid[i] && ((i & rows[0]) == 0)) {
                dp[0][i] = num[i];
            }
        }
        // System.out.println("here");
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < mm; j++) {
                if (!valid[j] || ((j & rows[i]) != 0)) {
                    continue;
                }
                for (int k = 0; k < mm; k++) {
                    if (!valid[k] || ((k & rows[i - 1]) != 0) || !validCross[j][k]) {
                        continue;
                    }
                    dp[i][j] = Math.max(dp[i][j], dp[i - 1][k] + num[j]);
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < mm; i++) {
            ans = Math.max(ans, dp[n - 1][i]);
        }
        return ans;
    }
    
    public boolean canCheat(int x, int y) {
        int[] sx = new int[8];
        int[] sy = new int[8];
        for (int i = 0; i < 8; i++) {
            sx[i] = x % 2;
            sy[i] = y % 2;
            x /= 2;
            y /= 2;
        }
        for (int i = 0; i < 8; i++) {
            if (sx[i] == 1) {
                if (i > 0 && sy[i - 1] == 1) {
                    return false;
                }
                if (i < 7 && sy[i + 1] == 1) {
                    return false;
                }
            }
        }
        return true;
    }
    
    public boolean canCheatSame(int x) {
        int[] sx = new int[8];
        for (int i = 0; i < 8; i++) {
            sx[i] = x % 2;
            x /= 2;
        }
        for (int i = 1; i < 8; i++) {
            if (sx[i] == 1 && sx[i - 1] == 1) {
                return false;
            }
        }
        return true;
    }
    
    public int numOfOne(int x) {
        int ans = 0;
        while (x > 0) {
            ans += x % 2;
            x /= 2;
        }
        return ans;
    }
}
