class Solution {
    int ans;
    int[][] count;
    Map<String, Integer> map = new HashMap<>();
    public int maxStudents(char[][] seats) {
        count = new int[seats.length][seats[0].length];
        
return dfs(seats, seats.length-1, 0, 0);
    }
    
    private int dfs(char[][] seats, int i, int j, int curr) {
        //System.out.println(i + " " + j + " " + curr);
        if (i == -1) {
            return 0;
        }
        //end of line
        if (j == seats[i].length) 
{
            return dfs(seats, i-1, 0, curr);
        }
        
        
        
        String rest = create(seats, i, j);
        if (map.containsKey(rest)) {
            return curr + map.get(rest);
        }
        
        int next = 0;
        if (seats[i][j] != '.' || count[i][j] != 0) {
            next = dfs(seats, i, j+1, curr);
        }
        else {
            //put
            put(seats, i, j);
            next = 1 + dfs(seats, i, j+1, curr);
            remove(seats, i, j);
            next = Math.max(next, dfs(seats, i, j+1, curr));
        
}
    
        map.put(rest, next);
        
        return next;
    }
    
    private String create(char[][] seats, int i, int j) {
        StringBuilder sb = new StringBuilder();
        for (int idx = j; idx < seats[0].length; idx++) {
            sb.append(seats[i][idx]);
            sb.append(count[i][idx]);
        }
        for (int idx = i-1; idx >= 0; idx--) {
            sb.append(Arrays.toString(seats[idx]));
            sb.append(Arrays.toString(count[idx]));
        }
        return sb.toString();
    }
    
    private void put(char[][] seats, int i, int j) {
        count[i][j]++;
        if (j+1 < seats[0].length && seats[i][j+1] == '.') {
            count[i][j+1]++;
        }
        if (i-1 >= 0 && j-1 >= 0 && seats[i-1][j-1] == '.') {
            count[i-1][j-1]++;
        }
        if (i-1 >=0 && j+1 < seats[0].length && seats[i-1][j+1] == '.') {
            count[i-1][j+1]++;
        }
        
    }
    
    private void remove(char[][] seats, int i, int j) {
        count[i][j]--;
        if (j+1 < seats[0].length && seats[i][j+1] == '.') {
            count[i][j+1]--;
        }
        if (i-1 >= 0 && j-1 >= 0 && seats[i-1][j-1] == '.') {
            count[i-1][j-1]--;
        }
        if (i-1 >=0 && j+1 < seats[0].length && seats[i-1][j+1] == '.') {
            count[i-1][j+1]--;
        }
    }
