class Solution {
    int rows;
    int cols;
    char[][] seats;
    int[][] dp = new int[9][256];
    int maxRes = 0;
    public int maxStudents(char[][] seats) {
        rows = seats.length;
        cols = seats[0].length;
        this.seats = seats;
        for (int r = 0; r < rows; r ++) {
            Arrays.fill(dp[r], -1);
        }
        for (int row = rows - 1; row >= 0; row --) {
            int[] prev = dp[row + 1];
            for (int state = 0; state < 256; state ++) {
                if (prev[state] == -1) continue;
                backtrack(row, 0, state, 0, prev[state]);
            }
        }
        return maxRes;
    }

    private void backtrack(int row, int col, int prevState, int curState, int res) {
        maxRes = Math.max(res, maxRes);
        
if (col == cols) {
            dp[row][curState] = Math.max(dp[row][curState], res);
            return;
        }
        if (seats[row][col] == '#') {
            backtrack(row, col + 1, prevState, curState, res);
            return;
        }
        // place
        if (!hasStudent(curState, col - 1) && !hasStudent(prevState, col - 1) && !hasStudent(prevState, col + 1)) {
            backtrack(row, col + 1, prevState, curState + (1 << col), res + 1);
        
}
        // not place
        backtrack(row, col + 1, prevState, curState, res);

    }

    private boolean hasStudent(int state, int col) {
        if (col < 0 || col >= cols) return false;
        state = state >> col;
        return (state & 1) == 1;
    }

}
