class Solution {
    public int maxStudents(char[][] seats) {
        int m = seats.length, n = seats[0].length;
        return dfs(seats, 0, m, n, new HashMap<>());
    
}
    private int dfs(char[][] seats, int pos, int m, int n, Map<String, Integer> cache) {
        if (pos == m * n) {
            return 0;
        }
        int i = pos / n, j = pos % n;
        String key = generate(seats, m, n, pos);
        
if (cache.containsKey(key)) {
            return cache.get(key);
        }
        int leave = dfs(seats, pos + 1, m, n, cache);
        if (!validate(seats, m, n, i, j)) {
            return leave;
        }
        seats[i][j] = 'O';
        int take = 1 + dfs(seats, pos + 1, m, n, cache);
        seats[i][j] = '.';
        int res = Math.max(leave, take);
        cache.put(key, res);
        return res;
    }
    private boolean validate(char[][] seats, int m, int n, int i, int j) {
        if (seats[i][j] == '#') {
            return false;
        }
        int[][] candidates = new int[4][2];
        candidates[0] = new int[]{i, j - 1};
        candidates[1] = new int[]{i, j + 1};
        candidates[2] = new int[]{i - 1, j - 1};
        candidates[3] = new int[]{i - 1, j + 1};
        for (int[] c : candidates) {
            if (0 <= c[0] && 0 <= c[1] && c[1] < n && seats[c[0]][c[1]] == 'O') {
                return false;
            }
        }
        
return true;
    }
    private String generate(char[][] seats, int m, int n, int pos) {
        StringBuilder key = new StringBuilder();
        key.append(pos);
        key.append(",");
        int start = Math.max(0, pos - pos % n - n);
        for (int k = start; k <= pos; k++) {
            int i = k / n, j = k % n;
            key.append(seats[i][j]);
            key.append(",");
        }
        return key.toString();
    }
}
