class Solution {
    public int maxStudents(char[][] seats) {
        int m = seats.length;
        int n = seats[0].length;
        int[][] dp = new int[m][1 << n];
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < (1 << n); ++j) {
                if(!vd(seats, i, j)) dp[i][j] = 0;
                else {
                    if(i == 0) {
                        dp[i][j] = cnt(j);
                    }
                    else {
                        int t = cnt(j);
                        int max = t;
                        for(int k = 0; k < (1 << n); ++k) {
                            if(dp[i - 1][k] != 0 && cvd(seats, k, j)) {
                                int nt = dp[i - 1][k] + t;
                                if(nt > max) max = nt;
                            }
                        }
                        dp[i][j] = max;
                    }
                }
            }
        }
        int max = 0;
        for(int x : dp[m - 1]) if(x > max) max = x;
        return max;
    }
    boolean vd(char[][] A, int i, int j) {
        int n = A[0].length;
        int[] arr = new int[n];
        for(int k = 0; k < n; ++k) {
            arr[k] = j % 2;
            j >>= 1;
        }
        for(int k = 0; k < n; ++k) {
            if(arr[k] == 1) {
                if(A[i][k] != '.') return false;
                if(k - 1 >= 0 && arr[k - 1] == 1) return false;
                if(k + 1 < n && arr[k + 1] == 1) return false;
            }
        }
        return true;
    }
    int cnt(int x) {
        int res = 0;
        while(x != 0) {
            ++res;
            x &= x - 1;
        }
        return res;
    }
    boolean cvd(char[][] A, int x, int y) {
        int n = A[0].length;
        int[] arr1 = new int[n];
        int[] arr2 = new int[n];
        for(int k = 0; k < n; ++k) {
            arr1[k] = x % 2;
            x >>= 1;
            arr2[k] = y % 2;
            y >>= 1;
        }
        for(int k = 0; k < n; ++k) {
            if(arr2[k] == 1) {
                if(k - 1 >= 0 && arr1[k - 1] == 1) return false;
                if(k + 1 < n && arr1[k + 1] == 1) return false; 
            }
        }
        return true;
    }
}
