class Solution {
    int m, n;
    char[][] s;
    int ans;
    public int maxStudents(char[][] seats) {
        s = seats;
        m = seats.length;
        n = seats[0].length;
        ans = 0;
        calc(0, 0, 0);
        return ans;
    }
    void calc(int x, int y, int cnt) {
        if (y == n && (m - x - 1) * ((n + 1) / 2)  + cnt < ans) return;
        if (y == n) {
            y = 0;
            ++x;            
        }
        if (x == m) {
            ans = Math.max(ans, cnt);
            return;
        }
        if (s[x][y] == '#') {
            calc(x, y + 1, cnt);
            return;
        }
        boolean f = true;
        if (isOccupied(x, y - 1) || isOccupied(x, y + 1) || isOccupied(x - 1, y - 1) || isOccupied(x - 1, y + 1)) f = false;
        if (f) {
            s[x][y] = 'x';
            calc(x, y + 1, cnt + 1);
            s[x][y] = '.';
        }
        calc(x, y + 1, cnt);
    }
    boolean isOccupied(int x, int y) {
        if (x < 0 || y < 0 || x >= m || y >= n) return false;
        return s[x][y] == 'x';
    }
}
