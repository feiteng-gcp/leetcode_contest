class Solution {
    public int maxStudents(char[][] seats) {
        int m = seats.length;
        int n = seats[0].length;
        int[] dp = new int[1 << m];
        int[] b = new int[n];
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if (seats[i][j] == '#') {
                    b[j] += (1 << i);
                
}
            }
        }
        
        int pos = (1 << m); 
        for(int i = 0; i < pos; ++i) {
            if ((i & b[0]) == 0) {
                dp[i] = Integer.bitCount(i);
            }
        }
        for(int i = 1; i < n; ++i) {
            // System.out.println("For " + i + " colum");
            // test(dp);
            int[] ndp = new int[pos];
            for(int j = 0; j < pos; ++j) {
                if ((j & b[i]) == 0) {
                    int num = Integer.bitCount(j);
                    int nb = (j | (j >> 1) | (j << 1));
                    for(int k = 0; k < pos; ++k) {
                        
if ((k & nb) == 0) {
                            ndp[j] = Math.max(ndp[j], dp[k] + num);
                        }
                    }
                }
            }
            dp = ndp;
        }
        int res = 0;
        for(int i = 0; i < pos; ++i) {
            res = Math.max(res, dp[i]);
        }
        return res;
    }
    private void test(int[] dp) {
        for(int i = 0; i < dp.length; ++i) {
            if (dp[i] > 0) {
                System.out.println(Integer.toBinaryString(i) + " " + dp[i]);
            }
        }
    }
}
