class TweetCounts {
    
    Map<String, TreeSet<Integer>> map = new HashMap<>();

    public TweetCounts() {
    
    }
    
    public void recordTweet(String tweetName, int time) {
        TreeSet<Integer> set = map.get(tweetName);
        if (set == null) {
            set = new TreeSet<>();
            map.put(tweetName, set);
        }
        
        set.add(time);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        List<Integer> ans = new ArrayList<>();
        int count = 0;
        
        TreeSet<Integer> set = map.get(tweetName);
        if (set == null) return ans;
        
        int interval = 0;
        if ("minute".equals(freq)) {
            interval = 60;
        } else if ("hour".equals(freq)) {
            interval = 3600;
        } else {
            interval = 86400;
        }
        
        for (int t = startTime; t <= endTime; t += interval) {
            int s = t;
            int e = Math.min(t + interval, endTime + 1);
            
            ans.add(set.subSet(s, e).size());
        }
        
        return ans;
    }
}
