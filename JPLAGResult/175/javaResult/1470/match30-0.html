class TweetCounts {
    
    private Map<String, TreeMap<Integer, Integer>> map;

    public TweetCounts() {
        this.map = new HashMap();
    }
    
    public void recordTweet(String tweetName, int time) {
        TreeMap<Integer, Integer> tweets = this.map.getOrDefault(tweetName, new TreeMap<Integer, Integer>());
        tweets.put(time, tweets.getOrDefault(time, 0) + 1);
        this.map.put(tweetName, tweets);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        int duration = 0;
        List<Integer> res = new ArrayList();
        if (!this.map.containsKey(tweetName)) {
            return res;
        }
        if (freq.equals("minute")) {
            duration = 60;
        } else if (freq.equals("hour")) {
            duration = 3600;
        } else if (freq.equals("day")) {
            duration = 86400;
        } else {
            return res;
        }
        while (startTime <= endTime) {
            Map<Integer, Integer> tweets = this.map.get(tweetName).subMap(startTime, Math.min(startTime + duration, endTime + 1));
            int f = 0;
            for (int v : tweets.values()) {
                f += v;
            }
            res.add(f);
            startTime += duration;
        }
        return res;
    }
}
