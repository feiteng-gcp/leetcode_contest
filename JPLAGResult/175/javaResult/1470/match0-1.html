class TweetCounts {

  private class Node {

    int time;
    Map<String, Integer> tweetNames;
    Node next;

    public Node(int time, Map<String, Integer> tweetNames, Node next) {
      this.time = time;
      this.tweetNames = tweetNames;
      this.next = next;
    }
  }

  private TreeMap<Integer, Node> map;

  public TweetCounts() {
    map = new TreeMap<>();
    map.put(-1, new Node(-1, new HashMap<>(), null));
  }

  public void recordTweet(String tweetName, int time) {
    Node pre = map.lowerEntry(time).getValue();
    Node cur = pre.next;
    if (cur != null && cur.time == time) {
      cur.tweetNames.put(tweetName, cur.tweetNames.getOrDefault(tweetName, 0) + 1);
    } else {
      Map<String, Integer> tweets = new HashMap<>();
      tweets.put(tweetName, 1);
      Node node = new Node(time, tweets, pre.next);
      pre.next = node;
      map.put(time, node);
    }
  }

  public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime,
      int endTime) {
    int step = freq.equals("hour") ? 3600: freq.equals("minute")? 60: 3600 * 24;
    List<Integer> res = new ArrayList<>();
    int start = startTime, end = Math.min(start + step - 1, endTime);
    Node pre = map.lowerEntry(startTime).getValue();
    Node cur = pre.next;
    while(start <= endTime){
      int count = 0;
      if(cur != null){
        while(cur != null && cur.time <= end){
          count += cur.tweetNames.getOrDefault(tweetName, 0);
          cur = cur.next;
        }
      }
      res.add(count);
      start = end + 1;
      end = Math.min(start + step - 1, endTime);
    }
    return res;
  }
}
