class TweetCounts {

    Map<String, TreeMap<Integer, Integer>> map = new HashMap<>();
    
TreeMap<Integer, Integer> EMPTY_TREE_MAP = new TreeMap<>();

    public TweetCounts() {

    }

    public void recordTweet(String tweetName, int time) {
        map.computeIfAbsent(tweetName, key -> new TreeMap<>()).merge(time, 1, Integer::sum);
    }

    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        TreeMap<Integer, Integer> treeMap = map.getOrDefault(tweetName, EMPTY_TREE_MAP);
        int delta = 60;
        
if ("hour".equals(freq)) {
            delta = 3600;
        } else if ("day".equals(freq)) {
            delta = 86400;
        }
        int[] res = new int[(endTime - startTime) / delta + 1];
        int time = startTime - 1;
        Map.Entry<Integer, Integer> entry;
        while (time <= endTime) {
            entry = treeMap.ceilingEntry(time + 1);
            if (entry == null) break;
            time = entry.getKey();
            if (time <= endTime) res[(time - startTime) / delta] += entry.getValue();
        }

        return Arrays.stream(res).boxed().collect(Collectors.toList());
    }
}
