class TweetCounts {
    Map<String, List<Integer>> map;
    public TweetCounts() {
        map = new HashMap();
    }
    
    public void recordTweet(String tweetName, int time) {
        List<Integer> list = 
map.getOrDefault(tweetName, new ArrayList<Integer>());
        list.add(time);
        map.put(tweetName, list);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        List<Integer> ans = new ArrayList();
        List<Integer> tlist = map.getOrDefault(tweetName, new ArrayList());
        int len = endTime- startTime+1;
        int bsize = 0;
        if(freq.equals("minute")) bsize = 60;
        if(freq.equals("hour")) bsize = 60*60;
        if(freq.equals("day")) bsize = 24*60*60;
        int nbin = (int)Math.ceil(len*1.0/bsize);
        int[] A = new int[nbin];
        for(int t: tlist){
            if(t> endTime || t<startTime) continue; 
            int bid = (t-startTime)/bsize;
            A[bid]++;
        
}
        for(int a: A){
            ans.add(a);
        }
        return ans;
    }
}
