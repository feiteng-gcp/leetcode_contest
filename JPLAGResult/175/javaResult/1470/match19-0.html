class TweetCounts {
    HashMap<String, TreeMap<Integer, Integer>> map;

    public TweetCounts() {
        map = new HashMap<>();
    }
    
    public void recordTweet(String tweetName, int time) {
        if (!map.containsKey(tweetName)) {
            map.put(tweetName, new TreeMap<>());
        }
        
        TreeMap<Integer, Integer> treeMap = map.get(tweetName);
        treeMap.put(time, treeMap.getOrDefault(time, 0) + 1);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        List<Integer> list = new ArrayList<>();
        int gap = 0;
        
        if (freq.equals("minute")) {
            gap = 60;
        } else if (freq.equals("hour")) {
            gap = 60 * 60;
        } else if (freq.equals("day")) {
            gap = 60 * 60 * 24;
        }
        
        TreeMap<Integer, Integer> treeMap = map.getOrDefault(tweetName, new TreeMap<Integer, Integer>());
        
        while (startTime < endTime + 1) {
            Map<Integer, Integer> subMap = treeMap.subMap(startTime, true, (int)Math.min(startTime + gap, endTime + 1), false);
            
            int sum = 0;
            for (int c : subMap.keySet()) {
                sum += subMap.get(c);
            }
            
            list.add(sum);
            startTime += gap;
        }
        
        return list;
    }
}
