class TweetCounts {
   Map<String, TreeMap<Integer, Integer>> map;

	public TweetCounts() {
		map = new HashMap<>();
	}

	public void recordTweet(String tweetName, int time) {
		if (!map.containsKey(tweetName)) {
			map.put(tweetName, new TreeMap<>());
		}
		TreeMap<Integer, Integer> treeMap = map.get(tweetName);
		treeMap.put(time, treeMap.getOrDefault(time, 0) + 1);
	}

	public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
		Map<Integer, Integer> subMap;
		if (map.containsKey(tweetName)) {
			TreeMap<Integer, Integer> treeMap = treeMap = map.get(tweetName);
			subMap = treeMap.subMap(startTime, true, endTime, true);
		} else {
			subMap = Collections.emptyMap();
		}

		int delta = buildFreq(freq);
		int[] res = new int[((endTime - startTime) / delta) + 1];
		for (Map.Entry<Integer, Integer> entry : subMap.entrySet()) {
			int interval = (entry.getKey() - startTime) / delta;
			res[interval] += entry.getValue();
		}
		List<Integer> list = new ArrayList<>();
		for (int i = 0; i < res.length; i++) {
			list.add(res[i]);
		}
		return list;
	}

	private int buildFreq(String freq) {
		int delta = 0;
		if (freq.equals("minute")) {
			delta = 60;
		} else if (freq.equals("hour")) {
			delta = 3600;
		} else {
			delta = 24 * 60 * 60;
		}
		return delta;
	}
}
