class TweetCounts {
    Map<String,TreeSet<Integer>> mp = new HashMap<>();
    public TweetCounts() {
        
    }
    
    public void recordTweet(String name, int time) {
        mp.putIfAbsent(name,new TreeSet<>());
        mp.get(name).add(time);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String name, int st1, int et) {
        
        List<Integer> li = new ArrayList<>();
        int ck = 1;
        
if(freq.equals("minute")){
            ck = 60;
        }else if(freq.equals("hour")){
            ck = 3600;
        }else{
            ck = 3600*24;
        }
        TreeSet<Integer> st = mp.get(name);
        //SortedSet<Integer> r = st.tailSet(st1,true);
        
        while(st1<et+1){
            int up = Math.min(et+1,st1+ck);
            int ct = 0;
            ct = st.subSet(st1,true,up,false).size();
            li.add(ct);
            st1 = up;
        }
        return li;
        
        
    }
}
