class TweetCounts {

    Map<String, List<Integer>> data = new HashMap<>();
    public TweetCounts() {

    }

    public void recordTweet(String tweetName, int time) {
        data.computeIfAbsent(tweetName, k -> new ArrayList<>()).add(time);
    }

    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        List<Integer> times = data.get(tweetName);
        List<Integer> ans = new ArrayList<>();

        
int interval = 60;
        if (freq.equals("hour")) {
            interval = 3600;
        } else if (freq.equals("day")) {
            interval = 60 * 60 * 24;
        }
        for (int i = startTime; i <= endTime; i += interval) {
            ans.add(0);
        }

        if (times == null) return ans;

        for (int t : times) 
{
            if (t >= startTime && t <= endTime) {
                int index = (t - startTime) / interval;
                ans.set(index, ans.get(index) + 1);
            }
        }
        return ans;
    }
}
