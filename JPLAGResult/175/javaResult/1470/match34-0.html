class TweetCounts {
    HashMap<String, TreeMap<Integer, Integer>> m;
    public TweetCounts() {
        m = new HashMap<>();
    }
    
    public void recordTweet(String tweetName, int time) {
        TreeMap<Integer, Integer> cur = m.getOrDefault(tweetName, new TreeMap<Integer, Integer>());
        cur.put(time, cur.getOrDefault(time, 0) + 1);
        m.put(tweetName, cur);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        TreeMap<Integer, Integer> cur = m.getOrDefault(tweetName, new TreeMap<Integer, Integer>());
        List<Integer> res = new ArrayList<>();
        int dur = 60;
        if(freq.equals("day")){
            dur = 3600 * 24;
        }
        if(freq.equals("hour")){
            dur = 3600;
        }
        int s = startTime, e = s + dur;
        while(s <= endTime){
            e = s + dur;
            if(e > endTime)e = endTime + 1;
            Map<Integer, Integer> rm = cur.subMap(s, e);
            int count = 0;
            for(int v: rm.values()){
                count += v;
            }
            res.add(count);
            s += dur;
        }
        return res;
    }
}
