class TweetCounts {
    private Map<String, TreeSet<Integer>> map;

    public TweetCounts() {
        this.map = new HashMap<>();
    }
    
    public void recordTweet(String tweetName, int time) {
        if (!map.containsKey(tweetName)) {
            map.put(tweetName, new TreeSet<>());
        }
        
map.get(tweetName).add(time);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        List<Integer> res = new ArrayList<>();
        if (!map.containsKey(tweetName)) {
            return res;
        }
        TreeSet<Integer> total = map.get(tweetName);
        int base = 0;
        
if (freq.equals("minute")) {
            base = 60;
        } else if (freq.equals("hour")) {
            base = 3600;
        } else {
            base = 86400;
        }
        int start = startTime;
        while (start <= endTime) {
            int curS = start;
            int curE = Math.min(curS + base, endTime + 1);
            res.add(total.subSet(curS, true, curE, false).size());
            start = curE;
        }
        return res;
    } 
}
