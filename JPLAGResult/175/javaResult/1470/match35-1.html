class TweetCounts {
    Map<String,ArrayList<Integer>> mp = new HashMap<>();
    public TweetCounts() {
        
    }
    
    public void recordTweet(String name, int time) {
        if(!mp.containsKey(name)){
            mp.put(name,new ArrayList<>());
        }
        
mp.get(name).add(time);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String name, int st1, int et) {
        
        List<Integer> li = new ArrayList<>();
        
int ck = 1;
        if(freq.equals("minute")){
            ck = 60;
        }else if(freq.equals("hour")){
            ck = 3600;
        }else{
            ck = 3600*24;
        }
        List<Integer> r = mp.get(name);
        Collections.sort(r);

        int sz  =r.size();
        while(st1<et+1){
            int up = Math.min(et+1,st1+ck);
            
            int lo = 0;
            int hi = sz;
            
            while(lo<hi){
                int mid = lo+hi; mid>>=1;
                if(r.get(mid)>=st1){
                    hi = mid;
                }else{
                    lo = mid+1;
                }
            }
          //  System.out.println(st1);
           // System.out.println(up);
            
             int lo1 = -1;
            int hi1 = sz-1;
            
            while(lo1<hi1){
                int mid = lo1+hi1+1; mid>>=1;
                if(r.get(mid)>=up){
                    hi1 = mid-1;
                }else{
                    lo1 = mid;
                }
            }
            int ct = lo1-lo+1;
           // System.out.println(lo1);
           // System.out.println(lo);
         
                li.add(ct);
           
            
            
            
            
            
            
            st1 = up;
        }
        return li;
        
        
    }
}
