class TweetCounts {
    
    Map<String, List<Integer>> map;

    public TweetCounts() {
        map = new HashMap<>();
    
}
    
    public void recordTweet(String tn, int time) {
        map.putIfAbsent(tn, new ArrayList<>());
        map.get(tn).add(time);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String f, String tn, int startTime, int endTime) {
        List<Integer> ans = new ArrayList<>();
        
        
List<Integer> list = map.get(tn);
        
        int gap = 0;
        if (f.charAt(0) == 'm') {gap = 60;}
        else if (f.charAt(0) == 'h') {gap = 3600;}
        else {gap = 3600 * 24;}
        int n = (endTime - startTime + 1) / gap + ((endTime - startTime + 1) % gap == 0 ? 0 : 1);
        int[] res = new int[n];
        
        // Collections.sort(list);
        
for (int time : list) {
            if (time >= startTime && time <= endTime) {
                res[(time - startTime) / gap]++;
            }
        }
        
        for (int i : res) {ans.add(i);}
        return ans;
    }
}
