class TweetCounts {
    Map<String, TreeMap<Integer, Integer>> map = new HashMap<>();
    public TweetCounts() {
        
    }
    
    public void recordTweet(String tweetName, int time) {
        map.computeIfAbsent(tweetName, k -> new TreeMap<>()).merge(time, 1, Integer::sum);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        int diff = 86400;
        if ("minute".equals(freq)) {
            diff = 60;
        } else if ("hour".equals(freq)) {
            diff = 3600;
        }
        
        List<Integer> list = new ArrayList<>();
        TreeMap<Integer, Integer> treemap = map.get(tweetName);
        
        for (int i = startTime; i <= endTime; i += diff) {
            int sum = 0;
            for (int v : treemap.subMap(i, true, Math.min(i+diff, endTime+1), false).values()) {
                sum += v;
            }
            list.add(sum);
        }
        
        return list;
    }
}
