class TweetCounts {

    Map<String, TreeMap<Integer, Integer>> hmap;
    public TweetCounts() {
        hmap = new HashMap();
    }
    
    public void recordTweet(String tweetName, int time) {
        TreeMap<Integer, Integer> treeMap = hmap.getOrDefault(tweetName, new TreeMap());
        treeMap.put(time, treeMap.getOrDefault(time, 0) + 1);
        hmap.put(tweetName, treeMap);
    }
    
    public List<Integer> getTweetCountsPerFrequency(String freq, String tweetName, int startTime, int endTime) {
        List<Integer> result = new ArrayList();
        if (!hmap.containsKey(tweetName)) return result;
        TreeMap<Integer, Integer> tMap = hmap.get(tweetName);
        
        List<int[]> intervals = new ArrayList();
        int intervalSize = 0;
        if (freq.equals("minute")) {
            intervalSize = 60;
        } else if (freq.equals("hour")) {
            intervalSize = 3600;
        } else {
            intervalSize = 86400;
        }
        
        for (int i=startTime;i<=endTime;i+=intervalSize) {
            int[] arr = new int[2];
            arr[0] = i;
            arr[1] = Math.min(i+intervalSize, endTime+1);
            intervals.add(arr);
        }
        for (int[] interval: intervals) {
            Map<Integer, Integer> subMap = tMap.subMap(interval[0], interval[1]);
            int res = 0;
            for (Integer num: subMap.values()) {
                res+=num;
            }
            result.add(res);
        }
        return result;
    }
}
