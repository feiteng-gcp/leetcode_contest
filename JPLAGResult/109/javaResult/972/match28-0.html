class Solution {
    int[][] A = new int[][]{
            {4, 6}, {6, 8}, {7, 9}, {4, 8}, {3, 9, 0}, {}, {1, 7, 0}, {2, 6}, {1, 3}, {2, 4}
    };
    int M = 1000000007;
   
    public int knightDialer(int N) {
        if (N == 1) return 10;
        int result = 0;
        int[][] dp = new int[N][10];
        for (int i = 0; i < 10; ++i) {
            dp[0][i] = 1;
        }
        for (int i = 1; i < N; ++i) {
            for (int j = 0; j < 10; ++j) {
                int[] arr = A[j];
                for (int a : arr) {
                    dp[i][j] += dp[i-1][a];
                    dp[i][j] %= M;
                }
            }
        }
        for (int i = 0; i < 10; ++i) {
            result += dp[N-1][i];
            result %= M;
        }
        return result;
    }
}
