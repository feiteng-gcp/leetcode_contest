class Solution {
    public int knightDialer(int N) {
        long MOD = 1000000007;
        
int[][] j = {{4, 6}, {6, 8}, {7, 9}, {4, 8}, {3, 9, 0}, {}, {1, 7, 0}, {2, 6}, {1, 3}, {2, 4}};

        long[][] dp = new long[10][N];

        for (int i = 0; i < 10; i++) {
            dp[i][0] = 1;
        }
        for (int i = 1; i < N; i++) {
            
for (int k = 0; k < 10; k++) {
                for (int e : j[k]) {
                    dp[k][i] = (dp[k][i] + dp[e][i - 1] + MOD) % MOD;
                }
            }
        
}
        
        
        long ans = 0;
        for (int i = 0; i < 10; i++) {
            ans = (ans + MOD + dp[i][N-1])%MOD;
        }
        return (int)ans;
    }
}
