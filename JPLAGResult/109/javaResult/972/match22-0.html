class Solution {
    private static int MOD = (int) 1e9 + 7;

    public int knightDialer(int N) {
        long[][] dp = new long[10][N];
        for (int digit = 0; digit < 10; digit++) {
            dp[digit][0] = 1;
        }
        for (int hop = 1; hop < N; hop++) {
            dp[0][hop] = (dp[4][hop - 1] + dp[6][hop - 1]) % MOD;
            dp[1][hop] = (dp[6][hop - 1] + dp[8][hop - 1]) % MOD;
            dp[2][hop] = (dp[7][hop - 1] + dp[9][hop - 1]) % MOD;
            dp[3][hop] = (dp[4][hop - 1] + dp[8][hop - 1]) % MOD;
            dp[4][hop] = (dp[0][hop - 1] + dp[3][hop - 1] + dp[9][hop - 1]) % MOD;
            dp[5][hop] = 0;
            dp[6][hop] = (dp[0][hop - 1] + dp[1][hop - 1] + dp[7][hop - 1]) % MOD;
            dp[7][hop] = (dp[2][hop - 1] + dp[6][hop - 1]) % MOD;
            dp[8][hop] = (dp[1][hop - 1] + dp[3][hop - 1]) % MOD;
            dp[9][hop] = (dp[2][hop - 1] + dp[4][hop - 1]) % MOD;
        }
        int sum = 0;
        for (int digit = 0; digit < 10; digit++) {
            sum += (int) dp[digit][N - 1];
            sum %= MOD;
        }
        return sum;
    }
}
