class Solution {
    public int knightDialer(int N) {
        long p = 1000000007;
        long dp[][] = new long[10][N+1];
        
int t[][] = {{4,6},{6,8},{7,9},{4,8},{3,9,0},{},{0,1,7},{2,6},{1,3},{2,4}};
        
        
for(int i=0;i<=9;i++){
           dp[i][0] = 1;   
        }
    for(int j=0;j<N;j++){
        for(int i=0;i<=9;i++){
                if(t[i].length>0){
                    for(int jp:t[i]){
                        if(j+1<N){
                            dp[jp][j+1] += dp[i][j]; 
                            dp[jp][j+1] %= p ;
                        }
                    }
                }
            }
        }
        long sum = 0;
        
for(int i=0;i<=9;i++){
            sum += dp[i][N-1];
            sum %= p;
            
        }
        
        return (int) sum;
    }
}
