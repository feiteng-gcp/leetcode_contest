class Solution {
    public int knightDialer(int N) {
        if(N == 1) return 10;
        
long mod = 1000000007;
        int[][] map = new int[][]{{4,6}, {6,8}, {7,9}, {4,8}, {3,9, 0}, {}, {1,7, 0}, {2, 6}, {1,3}, {2,4}};
        long[] dp = new long[10];
        for(int i = 0 ; i < 10 ; i++){
            dp[i] = 1;
        }
        long[] tmp;
        for(int i = 1 ; i < N  ; i++){
            tmp = 
new long[10];
            for(int j = 0 ; j < 10 ; j++){
                for(int k = 0 ; k < map[j].length ; k++){
                    tmp[map[j][k]] += dp[j];
                    tmp[map[j][k]] %= mod;
                }
            }
            dp = tmp;
        }
        long count = 0l;
        for(int i = 0 ; i < 10 ; i++){
            count += dp[i];
            count %= mod;
        }
        return (int)count;
    }
}
