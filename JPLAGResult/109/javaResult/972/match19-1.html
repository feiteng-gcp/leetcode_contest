class Solution {
    
private static final int[][] moves = {{4, 6}, {8, 6}, {7, 9}, {4, 8}, {3, 0, 9}, {}, {1, 7, 0}, {2, 6}, {1, 3}, {2, 4}};
    public int knightDialer(int N) {
        int[] dp = new int[10];
        Arrays.fill(dp, 1);
        
        for(int index = 1; index < N; index++) {
            int[] newRow = new int[10];
            for(int i = 0; i < newRow.length; i++) {
                for(int pre : moves[i]) {
                    newRow[i] = (newRow[i] + dp[pre]) % 1000000007;
                
}
            }
            dp = newRow;
        }
        int res = 0;
        for(int num : dp) {
            res = (res + num) % 1000000007;
        }
        return res;
    }
}
