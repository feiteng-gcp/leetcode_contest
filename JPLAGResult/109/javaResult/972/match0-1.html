class Solution {
    public int knightDialer(int N) {
        HashMap<Integer, List<Integer>> map = new HashMap<>();
        for(int i = 0; i < 10; i++){
            map.put(i, new ArrayList<Integer>());
        }
        int mod = 1000000007;
        
map.get(0).add(4);
        map.get(0).add(6);
        map.get(1).add(6);
        map.get(1).add(8);
        map.get(2).add(7);
        map.get(2).add(9);
        map.get(3).add(8);
        map.get(3).add(4);
        map.get(4).add(3);
        map.get(4).add(9);
        map.get(4).add(0);
        map.get(6).add(1);
        map.get(6).add(7);
        map.get(6).add(0);
        map.get(7).add(2);
        map.get(7).add(6);
        map.get(8).add(1);
        map.get(8).add(3);
        map.get(9).add(2);
        map.get(9).add(4);
        int[] p = new int[10];
        for(int i = 0; i < 10; i++) p[i] = 1;
        for(int i = 0; i < N - 1; i++){
            int[] t = new int[10];
            
for(int j = 0; j < 10; j++){
                for(int k : map.get(j)){
                    t[j] += p[k];
                    t[j] %= mod;
                }
            }
            p = t;
        
}
        int ans = 0;
        for(int i : p){
            ans += i;
            ans%=mod;
        }
        return ans;

    }
}
