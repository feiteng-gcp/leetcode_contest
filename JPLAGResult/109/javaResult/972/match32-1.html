class Solution {
  // 0: 2 (04, 06)
  // 1: 2 (16, 18)
  // 2: 2 (27, 29)
  // 3: 2 (34, 38)
  // 4: 3 (43, 49, 40)
  // 5: 0
  // 6: 3 (61, 67, 60)
  // 7: 2 (72, 76)
  // 8: 2 (81, 83)
  // 9: 2 (92, 94)
  public int knightDialer(int N) {
    int MOD = 1_000_000_000 + 7;
    List[] numWays = new ArrayList[10];
    numWays[0] = new ArrayList<Integer>();
    numWays[0].add(4);
    numWays[0].add(6);
    
    numWays[1] = new ArrayList<Integer>();
    numWays[1].add(6);
    numWays[1].add(8);
    
    numWays[2] = new ArrayList<Integer>();
    numWays[2].add(7);
    numWays[2].add(9);
    
    numWays[3] = new ArrayList<Integer>();
    numWays[3].add(4);
    numWays[3].add(8);
    
    numWays[4] = new ArrayList<Integer>();
    numWays[4].add(3);
    numWays[4].add(9);
    numWays[4].add(0);
    
    numWays[5] = new ArrayList<Integer>();
    
    numWays[6] = new ArrayList<Integer>();
    numWays[6].add(1);
    numWays[6].add(7);
    numWays[6].add(0);
    
    numWays[7] = new ArrayList<Integer>();
    numWays[7].add(2);
    numWays[7].add(6);
    
    numWays[8] = new ArrayList<Integer>();
    numWays[8].add(1);
    numWays[8].add(3);
    
    numWays[9] = new ArrayList<Integer>();
    numWays[9].add(2);
    numWays[9].add(4);
    
    
int[] numsCount = new int[10];
    Arrays.fill(numsCount, 1);
    
    for (int move = 0; move < N - 1; ++move) {
      int[] nextNumsCount = new int[10];
      
      for (int num = 0; num < 10; ++num) {
        for (int i = 0; i < numWays[num].size(); ++i) {
          nextNumsCount[(int)(numWays[num].get(i))] += numsCount[num];
          nextNumsCount[(int)(numWays[num].get(i))] %= MOD;
        
}
      }
      
      numsCount = nextNumsCount;
    }
    
    int total = 0;
    for (int num = 0; num < 10; ++num) {
      total += numsCount[num];
      total %= MOD;
    }
    return total;
  }
}
