class Solution {
    public int knightDialer(int N) {
        int[][] record = new int[2][10];
        int mod = 1000000007;
        for (int i=0; i<=9; i++) {
            record[0][i] = 1;
        }
        int curr = 0;
        for (int i=0; i<N-1;i++) {
            int next = 1-curr;
            
record[next][0] = (record[curr][4]+record[curr][6]) % mod;
            record[next][1] = (record[curr][6]+record[curr][8]) % mod;
            record[next][2] = (record[curr][7]+record[curr][9]) % mod;
            record[next][3] = (record[curr][4]+record[curr][8]) % mod;
            record[next][4] = ((record[curr][3]+record[curr][9]) % mod + record[curr][0] ) % mod;
            record[next][5] = 0;
            record[next][6] = ((record[curr][1]+record[curr][7] ) % mod+record[curr][0]) % mod;
            record[next][7] = (record[curr][2]+record[curr][6]) % mod;
            record[next][8] = (record[curr][1]+record[curr][3]) % mod;
            record[next][9] = (record[curr][2]+record[curr][4]) % mod;
            curr = next;
        }
        int total = 0;
        for (int i=0; i<=9;i++) {
            total = (total+record[curr][i]) % mod;
        }
        return total;
    }
}
