class Solution {
    public int knightDialer(int N) {
        int[] prev = new int[10];
        int mod = 1000000007;
        Arrays.fill(prev, 1);
        int[][] numWays = new int[][]{{4,6,-1},{6,8,-1},{7,9,-1},{4,8,-1},{0,3,9},{-1,-1,-1},{0,1,7},{2,6,-1},{1,3,-1},{2,4,-1}};
        for (int i = 2; i <= N; i++) {
            int[] cur = new int[10];
            for (int j = 0; j < 10; j++) {
                int[] ways = numWays[j];
                for (int k = 0; k < 3; k++) {
                    if (ways[k] < 0) break;
                    cur[j] = (cur[j] + prev[ways[k]]) % mod;
                }
            }
            prev = cur;
        }
        
        int result = 0;
        for (int num : prev)
            result = (result + num) % mod;
        return result;
    }
}
