class Solution {
    static long[][] dp;
    
static int[][] next = { {4, 6}, {6, 8}, {7, 9}, {4, 8}, {3, 9, 0}, {}, {1, 7, 0}, {2, 6}, {1, 3}, {2, 4}};
    public int knightDialer(int N) {
        dp = new long[5000][10];
        for (int i = 0; i < 10; i++) {
            dp[0][i] = 1;
        }
        
for (int i = 1; i < N; i++) {
            for (int j = 0; j < 10; j++) {
                for (int k : next[j]) {
                    dp[i][j] += dp[i - 1][k] % 1000000007L;
                    dp[i][j] = dp[i][j] % 1000000007L;
                }
            }
        }
        
        long sum = 0;
        for (int i = 0; i < 10; i++) {
            sum += dp[N - 1][i] % 1000000007L;
            sum = sum % 1000000007L;
        }
        
        return (int) sum;
    }
}
