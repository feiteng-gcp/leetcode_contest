class Solution {
    public int knightDialer(int N) {
        long m = 1000000007;
        long[][] dp = new long[10][N];
        for (int i = 0; i < 10; ++i) { dp[i][0] = 1; }
        for (int i = 1; i < N; ++i) {
            dp[0][i] = (dp[4][i - 1] + dp[6][i - 1]) % m;
            dp[1][i] = (dp[6][i - 1] + dp[8][i - 1]) % m;
            dp[2][i] = (dp[7][i - 1] + dp[9][i - 1]) % m;
            dp[3][i] = (dp[4][i - 1] + dp[8][i - 1]) % m;
            dp[4][i] = (dp[0][i - 1] + dp[3][i - 1] + dp[9][i - 1]) % m;            
            dp[6][i] = (dp[0][i - 1] + dp[1][i - 1] + dp[7][i - 1]) % m;
            dp[7][i] = (dp[2][i - 1] + dp[6][i - 1]) % m;
            dp[8][i] = (dp[1][i - 1] + dp[3][i - 1]) % m;
            dp[9][i] = (dp[2][i - 1] + dp[4][i - 1]) % m;
        }
        long ans = 0;
        for (int i = 0; i < 10; ++i) ans = (ans + dp[i][N - 1]) % m;
        return (int) ans;
    }
}
