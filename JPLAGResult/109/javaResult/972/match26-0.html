class Solution {
    private static int[][] moves = {
        {4, 6},
        {6, 8},
        {7, 9},
        {4, 8},
        {0, 3, 9},
        {},
        {0, 1, 7},
        {2, 6},
        {1, 3},
        {2, 4}
    };
        
    public int knightDialer(int N) {
        int[][] dp = new int[N][10];
        for(int i=0; i<10; i++) {
            dp[0][i] = 1;
        }

        if(N > 1) {
            for(int i=0; i<10; i++) {
                dp[1][i] = moves[i].length;
            }
        }
        
        for(int i=2; i<N; i++) {
            for(int j=0; j<10; j++) {
                for(int p : moves[j]) {
                    dp[i][j] = (dp[i][j] + dp[i-1][p]) % 1000000007;
                }
            }
        }
        
        int res = 0;
        for(int i=0; i<10; i++) {
            res = (res + dp[N-1][i]) % 1000000007;
        }
        
        return res;
    }
}
