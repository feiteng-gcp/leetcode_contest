class Solution {
   public int knightDialer(int N) {
        int mod = 1_000_000_007;
        int[][] edges = {
                {4, 6},
                {6, 8},
                {9, 7},
                {8, 4},
                {3, 9, 0},
                {},
                {1, 7, 0},
                {2, 6},
                {1, 3},
                {2, 4}
        };

        int[] dp = new int[10];

        for (int i = 0; i <= 9; i++) {
            dp[i] = 1;
        }

        int[] t = new int[10];
        for (int i = 1; i <= N - 1; i++) {
            Arrays.fill(t, 0);
            for (int j = 0; j <= 9; j++) {

                if (dp[j] > 0) {
                    for (int k = 0; k < edges[j].length; k++) {
                        int v = edges[j][k];
                        t[v] = (t[v] + dp[j]) % mod;
                    }
                }
            }

            for (int j = 0; j <= 9; j++) {
                dp[j] = t[j];
            }
        }

        int ans = 0;
        for (int j = 0; j <= 9; j++) {
            ans = (ans + dp[j]) % mod;
        }
        return ans;
    }

}
