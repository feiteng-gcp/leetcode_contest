class Solution {
    public int knightDialer(int N) {
        ArrayList<ArrayList<Integer>> g = new ArrayList<>();
        for (int i = 0; i < 10; i++) {
            g.add(new ArrayList<>());
        }
        g.get(0).add(4);
        g.get(0).add(6);
        g.get(1).add(6);
        g.get(1).add(8);
        g.get(2).add(7);
        g.get(2).add(9);
        g.get(3).add(4);
        g.get(3).add(8);
        g.get(4).add(0);
        g.get(4).add(3);
        g.get(4).add(9);
        g.get(6).add(0);
        g.get(6).add(1);
        g.get(6).add(7);
        g.get(7).add(2);
        g.get(7).add(6);
        g.get(8).add(1);
        g.get(8).add(3);
        g.get(9).add(2);
        g.get(9).add(4);
        int[][] dp = new int[N+1][10];
        for (int i = 0; i < 10; i++)
            dp[1][i] = 1;
        for (int i = 2; i <= N; i++) {
            for (int u = 0; u < 10; u++) {
                for (int v : g.get(u)) {
                    dp[i][u] += dp[i-1][v];
                    dp[i][u] %= 1000000007;
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < 10; i++) {
            ans += dp[N][i];
            ans %= 1000000007;
        }
        return ans;
    }
}
