class Solution {
    public int knightDialer(int N) {
        int[][] jumpIdx = {{4,6}, {6,8}, {7,9}, {4,8}, {3,9,0}, {}, {1,7,0}, {2,6}, {1,3}, {2,4}};
        long[] dp = new long[10];
        Arrays.fill(dp, 1);
        int MOD = 1000000007;
        while (N > 1) {
            long[] newDP = new long[10];
            for (int i = 0; i < 10; i++) {
                for (int fromIdx : jumpIdx[i]) {
                    newDP[i] += (long) dp[fromIdx];
                    newDP[i] %= MOD;
                }
            }
            dp = newDP;
            N -= 1;
        }
        long total = 0;
        for (long k : dp) {
            total += k;
            total %= MOD;
        }
        return (int) total;
    }
}
