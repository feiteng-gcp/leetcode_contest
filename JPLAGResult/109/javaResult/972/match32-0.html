class Solution {
    public int knightDialer(int N) {
        String[] maps = new String[]{"46", "68", "79", "48", "390", "", "170", "26", "13", "24"};
        
        int[] dp = new int[10];
        Arrays.fill(dp, 1);
        
        for (int i = 1; i < N; i++) {
            int[] temp = new int[10];
            
            for (int j = 0; j < 10; j++) {
                
                for (char ch: maps[j].toCharArray()) {
                    int parent = ch-'0';
                    temp[j] += dp[parent];
                    temp[j] %= 1000000007;
                }
            }
            dp = temp;
        }
        
        int res = 0;
        for (int i = 0; i < dp.length; i++) {
            res += dp[i];
            res %= 1000000007;
        }
        return res;
        
    }
}
