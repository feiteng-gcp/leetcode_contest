class Solution {
    private static final int MOD = 1000000007;
    private static final int[][] moves = {
        {4, 6},
        {6, 8},
        {7, 9},
        {4, 8},
        {3, 9, 0},
        {},
        {1, 7, 0},
        {2, 6},
        {1, 3},
        {2, 4},
    };
    
    public int knightDialer(int N) {
        long[] dp = new long[10];
        Arrays.fill(dp, 1);
        for (long i = 1; i < N; ++i) {
            long[] next = new long[10];
            for (int u = 0; u < 10; ++u) {
                for (int v : moves[u]) {
                    next[v] += dp[u];
                    next[v] %= MOD;
                }
            }
            dp = next;
        }
        long sum = 0;
        for (int i = 0; i < 10; ++i) {
            sum += dp[i];
            sum %= MOD;
        }
        return (int)sum;
    }
}
