class Solution {
    int[][] a = {{4, 6},
                 {6, 8},
                 {7, 9},
                 {4, 8},
                 {3, 9, 0},
                 {},
                 {1, 7, 0},
                 {2, 6},
                 {1, 3},
                 {2, 4}};
    
    long M = 1000000007;
    
    public int knightDialer(int N) {
        long[][] dp = new long[N][10];
        for (int j = 0; j < 10; j++) {
            dp[0][j] = 1;
        }
        
        for (int i = 1; i < N; i++) {
            for (int j = 0; j < 10; j++) {
                for(int k = 0; k < a[j].length; k++) {
                    dp[i][j] += dp[i - 1][a[j][k]];
                    dp[i][j] %= M;
                }
            }
        }
        
        long res = 0;
        for (int j = 0; j < 10; j++) {
            res += dp[N - 1][j];
            res %= M;
        }
        return (int)res;
    }
}
