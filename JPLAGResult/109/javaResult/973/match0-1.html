class Solution {
    public int[] movesToStamp(String stamp, String target) {
        char[] targetC = target.toCharArray();
        char[] stampC = stamp.toCharArray();
        int remain_length = targetC.length;
        ArrayList<Integer> steps = new ArrayList();
        while (remain_length > 0) {
            boolean found_match = false;
            
for (int i=0; i<= targetC.length - stampC.length; i++) {
                boolean match = true;
                boolean hasNonQ = false;
                for (int j=0; j<stampC.length; j++) {
                    if (targetC[i+j] == '?') {
                        continue;
                    }
                    if (targetC[i+j] != stampC[j]) {
                        match = false;
                        break;
                    }
                    hasNonQ = true;
                }
                if (match && hasNonQ) {
                    found_match = true;
                    steps.add(i);
                    for (int j=0; j<stampC.length; j++) {
                        if (targetC[i+j] == '?') {
                            continue;
                        }
                        targetC[i+j] = '?';
                        remain_length --;
                    }
                    break;
                }
            }
            if (!found_match) {
 /*               for (int i=0; i< targetC.length; i++) {
                    System.out.print(targetC[i]);
                }*/
                return new int[0];
            }
        }
        int[] result = new int[steps.size()];
        for (int i=0; i<result.length; i++) {
            result[i] = steps.get(result.length-1-i);
        }
        return result;
    }
}
