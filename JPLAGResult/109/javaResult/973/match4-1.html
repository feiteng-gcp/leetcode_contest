	class Solution {
		int cor(char[] s, char[] t, int f)
		{
			int ex = 0;
			for(int i = 0;i < s.length;i++){
				if(t[i+f] == '.')continue;
				ex = 1;
				if(s[i] != t[i+f])return 0;
			}
			if(ex == 0)return 2;
			return 1;
		}
		
	    public int[] movesToStamp(String stamp, String target) {
	    	char[] t = target.toCharArray();
	    	char[] s = stamp.toCharArray();
	    	
	    	Deque<Integer> route = new ArrayDeque<>();
	    	boolean[] done = new boolean[t.length-s.length+1];
	    	while(true){
	    		boolean up = false;
		    	for(int i = 0;i < t.length-s.length+1;i++){
		    		if(done[i])continue;
		    		int cor = cor(s, t, i);
		    		if(cor == 1){
		    			route.addFirst(i);
		    			for(int j = 0;j < s.length;j++){
		    				t[j+i] = '.';
		    			}
		    			up = true;
		    			done[i] = true;
		    		}else if(cor == 2){
		    			done[i] = true;
		    		}
		    	}
		    	if(!up)break;
	    	}
	    	for(int i = 0;i < t.length;i++){
	    		if(t[i] != '.')return new int[0];
	    	}
	    	
	    	int[] ret = new int[route.size()];
	    	int p = 0;

	    	for(int x : route){
	    		ret[p++] = x;
	    	}
	    	return ret;
	    }
	}	//
