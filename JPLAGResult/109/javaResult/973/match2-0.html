class Solution {
    public int[] movesToStamp(String stamp, String target) {
        char[] ntarget = new char[target.length()];
        char[] nstamp = new char[stamp.length()];
        for (int i = 0; i < ntarget.length; i++) {
            ntarget[i] = target.charAt(i);
        }
        for (int i = 0; i < nstamp.length; i++) {
            nstamp[i] = stamp.charAt(i);
        }
        int [] ret = new int[10*ntarget.length];
        int count = 0;
        while (true) {
            if (checkDone(ntarget)) {
                break;
            }
            int x = findStamp(ntarget, nstamp);
            // System.out.println(Arrays.toString(ntarget));
            ret[count++] = x;
            if (x == -1)
                return new int[]{};
        }
        int[] rr = new int[count];
        int y = 0;
        for (int i = count - 1; i >= 0; i--) {
            rr[y++] = ret[i];
        }
        return rr;
        
    }
    
    public boolean checkDone(char[] arr) {
        for (int  i = 0; i < arr.length; i++) {
            if (arr[i] != '?') {
                return false;
            }
        }
        return true;
    }
    
    public int findStamp(char[] arr, char[] stamp) {
        // optimization
        boolean optimize = true;
        for (int i = 0; i < stamp.length; i++) {
            if (arr[i] != '?')
                optimize = false;
        }
        int start = 0;
        if (optimize) {
            while(true) {
                if (arr[start + stamp.length] == '?') {
                    start++;
                } else {
                    break;
                }
            }
        }
        System.out.println("start" + start);
        for (int i = start; i < arr.length - stamp.length + 1; i++) {
            if (arr[i] == stamp[0] || arr[i] == '?') {
                // System.out.println(i + "poop");
                boolean valid = false;
                if (arr[i] != '?')
                    valid = true;
                int j = 1;
                boolean fail = false;
                while (j < stamp.length) {
                    if (stamp[j] != arr[i + j] && arr[i + j] != '?') {
                        fail = true;
                        break;
                    } else {
                        if (arr[i + j] != '?') {
                            valid = true;
                        }
                    }
                    j++;
                }
                // System.out.println(fail + " " + valid);
                if (!fail && valid) {
                    for (int k = i; k < i + j; k++) {
                        arr[k] = '?';
                    }
                    return i;
                }
            }
        }
        return -1;
    }
}
