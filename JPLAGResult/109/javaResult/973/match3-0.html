class Solution {
    public int[] movesToStamp(String stamp, String target) {
        ArrayList<Integer> revMoves = new ArrayList<Integer>();
        char[] ca = target.toCharArray();
        char[] cs = stamp.toCharArray();
        int n = target.length();
        int sn = stamp.length();
        int covered = 0;
        boolean more = true;
        while (more) {
            more = false;
            for (int i = 0; i <= n-sn; i++) {
                boolean poss = true;
                boolean something = false;
                for (int j = 0; j < sn; j++) {
                    if (ca[i+j] != '?' && cs[j] != ca[i+j]) {
                        poss = false;
                        break;
                    } else if (ca[i+j] != '?') {
                        something = true;
                    }
                }
                if (poss && something) {
                    for (int j = i; j < i+sn; j++) {
                        if (ca[j]!='?') {
                            ca[j] = '?';
                            covered++;
                            more = true;
                        }
                    }
                    revMoves.add(i);
                }
            }
        }
        int[] m = new int[revMoves.size()];
        for (int i = revMoves.size()-1; i >= 0; i--) {
            m[m.length-1-i] = revMoves.get(i);
        }
        return covered==n?m:new int[0];
    }
}
