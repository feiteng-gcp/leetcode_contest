class RecentCounter {

    private LinkedList<Integer> calls;
    
    public RecentCounter() {
        calls = new LinkedList<>();
    }
    
    public int ping(int t) {
        while (!calls.isEmpty() && calls.getFirst() < t - 3000)
            calls.removeFirst();
        
        calls.addLast(t);
        
        return calls.size();
    }
    
}
