class RecentCounter {
    Queue<Integer> q;
    public RecentCounter() {
        q = new LinkedList<>();
    }
    
    public int ping(int t) {
        int cutoff = t - 3000;
        while (q.size() != 0 && q.peek() < cutoff)
            q.remove();
        q.add(t);
        return q.size();
    }
}
