class RecentCounter {
    Deque<Integer> pings;
    public RecentCounter() {
        pings = new LinkedList();
    }
    
    public int ping(int t) {
        while (!pings.isEmpty() && pings.peek() < t - 3000) pings.poll();
        pings.add(t);
        return pings.size();
    }
}
