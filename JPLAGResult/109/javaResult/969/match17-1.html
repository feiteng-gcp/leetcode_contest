class RecentCounter {
    PriorityQueue<Integer> pq;

    public RecentCounter() {
        pq = new PriorityQueue<>();
    }
    
    public int ping(int t) {
        pq.add(t);
        while (t - pq.peek() > 3000) pq.poll();
        return pq.size();
    }
}
