class RecentCounter {

    Queue<Integer> q;
    
    public RecentCounter() {
        q = new LinkedList<>();
    }
    
    public int ping(int t) {
        int start = t - 3000;
        
while (!q.isEmpty() && q.peek() < start) q.poll();
        q.add(t);
        return q.size();
    }
}

 * RecentCounter obj = new RecentCounter();
