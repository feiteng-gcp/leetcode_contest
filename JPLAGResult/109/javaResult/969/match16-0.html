class RecentCounter {
    Deque<Integer> dq;
    public RecentCounter() {
        dq = new LinkedList<>();
    }
    
    public int ping(int t) {
        dq.addFirst(t);
        while(t - dq.peekLast() > 3000){
            dq.pollLast();
        }
        return dq.size();
    }
}
