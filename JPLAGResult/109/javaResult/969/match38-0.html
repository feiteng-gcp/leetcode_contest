class RecentCounter {
    LinkedList<Integer> mono;
    
    public RecentCounter() {
        mono = new LinkedList<>();
    }
    
    public int ping(int t) {
        while (!mono.isEmpty() && t - mono.getFirst() > 3000)
            mono.removeFirst();
        
        mono.addLast(t);
        return mono.size();
    }
}

 * RecentCounter obj = new RecentCounter();
 * int param_1 = obj.ping(t);
