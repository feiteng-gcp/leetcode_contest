class RecentCounter {

    private Queue<Integer> Q;

    public RecentCounter() {
        Q = new LinkedList<>();
    }

    public int ping(int t) {
        Q.add(t);
        while (Q.peek() + 3000 < t) 
{
            Q.poll();
        }

        return Q.size();
    }
}

 * RecentCounter obj = new RecentCounter();
