class RecentCounter {
    
    Queue<Integer> q;

    public RecentCounter() {
        q = new LinkedList<>();
    }
    
    public int ping(int t) {
        q.add(t);
        while(!q.isEmpty() && q.peek() < t - 3000){
            q.poll();
        }
        return q.size();
    }
}

 * RecentCounter obj = new RecentCounter();
 * int param_1 = obj.ping(t);
