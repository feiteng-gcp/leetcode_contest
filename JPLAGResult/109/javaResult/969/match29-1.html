class RecentCounter {
    Deque<Integer> deque;
    public RecentCounter() {
        deque = new ArrayDeque<>();
    }
    
    public int ping(int t) {
        deque.offerLast(t);
        while (!deque.isEmpty() && deque.peekFirst() < t - 3000)
            deque.pollFirst();
        return deque.size();
    }
}
