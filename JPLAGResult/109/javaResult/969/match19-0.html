	class RecentCounter {
		PriorityQueue<Integer> pq = new PriorityQueue<>();

	    public RecentCounter() {
	        
	    }
	    
	    public int ping(int t) {
	        pq.add(t);
	        while(!pq.isEmpty() && pq.peek() < t-3000){
	        	pq.poll();
	        }
	        return pq.size();
	    }
	}	//
