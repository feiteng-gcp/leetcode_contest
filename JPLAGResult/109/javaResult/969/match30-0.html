class RecentCounter {
    Queue<Integer> pings;
    public RecentCounter() {
        pings = new LinkedList<Integer>();
    }
    
    public int ping(int t) {
        pings.add(t);
        
        while (pings.size() > 0 && t - pings.peek() > 3000) {
            pings.poll();
        }
        
        return pings.size();
    }
}
