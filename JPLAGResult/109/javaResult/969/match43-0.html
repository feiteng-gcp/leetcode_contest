class RecentCounter {
    Queue<Integer> q;
    public RecentCounter() {
        q = new LinkedList<>();
    }
    
    public int ping(int t) {
        q.add(t);
         while(!q.isEmpty() && t - q.peek() > 3000){
            q.remove();
        }
        return q.size();
    }
}
