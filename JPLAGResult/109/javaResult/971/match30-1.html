class Solution {
  public int shortestBridge(int[][] A) {
    int rc = A.length;
    int cc = A[0].length;
    
    int[][] group = new int[rc][cc];
    
int curGroup = 1;
    for (int r = 0; r < rc; ++r) {
      for (int c = 0; c < cc; ++c) {
        if (A[r][c] == 1 && group[r][c] == 0) {
          dfs(A, r, c, group, curGroup);
          ++curGroup;
        }
      }
    }
    
    
    int res = Integer.MAX_VALUE;
    for (int r = 0; r < rc; ++r) {
      for (int c = 0; c < cc; ++c) {
        if (group[r][c] == 1) {
          
for (int nr = 0; nr < rc; ++nr) {
            for (int nc = 0; nc < cc; ++nc) {
              if (group[nr][nc] == 2) {
                res = Math.min(res, Math.abs(r - nr) + Math.abs(c - nc) - 1);
              }
            }
          }
        
}
      }
    }
    
    return res;
  }
  
  void dfs(int[][] A, int r, int c, int[][] group, int curGroup) {
    int rc = A.length;
    int cc = A[0].length;
    
if (r < 0 || r >= rc || c < 0 || c >= cc) {
      return;
    }
    
    if (A[r][c] == 0 || group[r][c] != 0) {
      return;
    }
    
    group[r][c] = curGroup;
    dfs(A, r - 1, c, group, curGroup);
    dfs(A, r + 1, c, group, curGroup);
    dfs(A, r, c - 1, group, curGroup);
    dfs(A, r, c + 1, group, curGroup);
  }
}
