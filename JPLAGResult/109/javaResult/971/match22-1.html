class Solution {
    public int shortestBridge(int[][] A) {
        init(A);
        Queue<Point> q = new LinkedList<>();
        int N = A.length, M = A[0].length;
        Set<String> set = new HashSet<>();
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (A[i][j] == 2) {
                    q.offer(new Point(i, j));
                
}
            }
        }
        int step = 0;
        while (!q.isEmpty()) {
            int size = q.size();
            while (size > 0) {
                Point cur = q.poll();
                for (int[] d : dirs) {
                    int nx = cur.x + d[0], ny = cur.y + d[1];
                    if (nx < 0 || ny < 0 || nx >= N || ny >= M) continue ;
                    if (A[nx][ny] == 2) continue;
                    if (A[nx][ny] == 1) return step;
                    A[nx][ny] = 2;
                    q.offer(new Point(nx, ny));
                
}
                size--;
            }
            step++;
        }
        return 0;
    }
    
    class Point {
        int x, y;
        public Point(){}
        public Point(int x, int y){
            this.x = x; this.y = y;
        }
    }
    
    void init(int[][] A) {
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[i].length; j++) {
                if (A[i][j] == 1) {
                    dfs(A, i, j);
                    return ;
                }
            }
        }
    }
    
    
int[][] dirs = new int[][]{
        {0, 1}, {1, 0}, {0, -1}, {-1, 0}
    };
    void dfs (
int[][] A, int x, int y) {
        if (x < 0 || y < 0 || x >= A.length || y >= A[0].length) {
            return ;
        }
        if (A[x][y] == 0 || A[x][y] == 2) return ;
        A[x][y] = 2;
        
for (int[] d : dirs) {
            dfs(A, x + d[0], y + d[1]);
        }
    }
}
