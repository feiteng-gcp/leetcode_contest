class Solution {
    Set<String> set1;
    Set<String> set2;
    public int shortestBridge(int[][] A) {
        set1 = new HashSet<>();
        set2 = new HashSet<>();
        Set<String> set3 = new HashSet<>();
        
boolean[][] visited = new boolean[A.length][A[0].length];
        boolean flag = false;
        for(int i = 0; i < A.length; i++){
            for(int j = 0; j < A[0].length; j++){
                if(A[i][j] == 1 && visited[i][j] == false){
                    if(!flag){
                        dfs(i, j, false, visited, A);
                        flag = true;
                    } else {
                        dfs(i, j, true, visited, A);
                    }
                }
            }
        }
        Queue<String> qu = new LinkedList<>();
        for(String s : set1){
            qu.add(s);
            //set3.add(s);
        }
        int level = -1;
        
int[][] dirs = new int[][]{{-1, 0},{0, -1},{0, 1}, {1, 0}};
        
while(!qu.isEmpty()){
            int sz = qu.size();
            //System.out.println(sz);
            for(int i = 0; i < sz; i++){
                String s = qu.poll();
                if(set3.contains(s)) continue;
                set3.add(s);
                if(set2.contains(s)) return level;
                String[] xxx = s.split(",");
                int x = Integer.parseInt(xxx[0]);
                int y = Integer.parseInt(xxx[1]);
                for(int[] dir : dirs){
                    int xx = x + dir[0];
                    int yy = y + dir[1];
                    if(xx >= 0 && xx < visited.length && yy >= 0 && yy < visited[0].length){
                        qu.add(xx + "," + yy);
                    
}
                }
            }
            level++;
        }
        return 1000;
    }
    
    private void dfs(int x, int y, boolean bl, boolean[][] visited, int[][] A){
        int[][] dirs = new int[][]{{-1, 0},{0, -1},{0, 1}, {1, 0}};
        if(visited[x][y]) return;
        if(bl == false) set1.add(x + "," + y);
        else set2.add(x + "," + y);
        visited[x][y] = true;
        
for(int[] dir : dirs){
            int xx = x + dir[0];
            int yy = y + dir[1];
            if(xx >= 0 && xx < visited.length && yy >= 0 && yy < visited[0].length && A[xx][yy] == 1){
                dfs(xx, yy, bl, visited, A);
            }
        }
    }
}
