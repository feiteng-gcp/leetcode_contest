class Solution {
    private static final int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    
    public int shortestBridge(int[][] A) {
        int res = 0;
        int m = A.length;
        int n = A[0].length;
        Queue<int[]> que = new LinkedList<>();
        
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(A[i][j] == 1) {
                    que.offer(new int[]{i, j});
                    dfs(A, i, j, que);
                    break;
                }
            }
            // 注意跳出两层循环！这里也要有break
            if(!que.isEmpty()) {
                break;
            }
        }
        // System.out.println(que);
        // System.out.println(que.size());
        // while(!que.isEmpty()) {
        //     for(int num : que.poll()) {
        //         System.out.print(num + " ");
        //     }
        //     System.out.println();
        // }
        
        while(!que.isEmpty()) {
            int size = que.size();
            for(int index = 0; index < size; index++) {
                int[] cur = que.poll();
                for(int[] dir : dirs) {
                    int nextX = cur[0] + dir[0];
                    int nextY = cur[1] + dir[1];
                    if(nextX < 0 || nextX >= m || nextY < 0 || nextY >= n) {
                        continue;
                    }
                    if(A[nextX][nextY] == 1) {
                        return res;
                    }
                    if(A[nextX][nextY] == 0) {
                        A[nextX][nextY] = 2;
                        que.offer(new int[]{nextX, nextY});
                    }
                }
            }
            res++;
        }
        
        return res;
    }
    
    private void dfs(int[][] A, int x, int y, Queue<int[]> que) {
        A[x][y] = 2;
        int m = A.length;
        int n = A[0].length;
        
        for(int[] dir : dirs) {
            int nextX = x + dir[0];
            int nextY = y + dir[1];
            if(nextX < 0 || nextX >= m || nextY < 0 || nextY >= n || A[nextX][nextY] != 1) {
                continue;
            }
            que.offer(new int[]{nextX, nextY});
            dfs(A, nextX, nextY, que);
        }
    }
}
