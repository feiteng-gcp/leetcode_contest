class Solution {
    int[] dx = {0, 0, -1, 1};
    int[] dy = {-1, 1, 0, 0};
        public int shortestBridge(int[][] A) {
        int n = A.length;
        int r1 = -1;
        int c1 = -1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1) {
                    r1 = i;
                    c1 = j;
                    break;
                }
            }
        }

        boolean[][] vis = new boolean[n][n];
        dfs(A, vis, r1, c1);
        Deque<Integer> rows = new LinkedList<>();
        Deque<Integer> cols = new LinkedList<>();
        int[][] dist = new int[n][n];
        for (int i = 0; i < n; i++) {
            Arrays.fill(dist[i], Integer.MAX_VALUE);
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (vis[i][j]) {
                    rows.add(i);
                    cols.add(j);
                    dist[i][j] = 0;
                }
            }
        }

        while (!rows.isEmpty()) {
            int curRow = rows.removeFirst();
            int curCol = cols.removeFirst();
            for (int i = 0; i < dx.length; i++) {
                int nr = curRow + dx[i];
                int nc = curCol + dy[i];
                if (nr < 0 || nr >= n || nc < 0 || nc >= n) {
                    continue;
                }

                if (dist[nr][nc] > dist[curRow][curCol] + 1) {
                    dist[nr][nc] = 1 + dist[curRow][curCol];
                    rows.add(nr);
                    cols.add(nc);
                }
            }
        }

        int res = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1 && !vis[i][j]) {
                    res = Math.min(res, dist[i][j] - 1);
                }
            }
        }

        return res;
    }

    void dfs(int[][] A, boolean[][] vis, int row, int col) {
        int n = A.length;
        if (row < 0 || row >= n || col < 0 || col >= n) {
            return;
        }

        if (vis[row][col]) {
            return;
        }

        if (A[row][col] == 0) {
            return;
        }

        vis[row][col] = true;
        for (int i = 0; i < dx.length; i++) {
            int nr = row + dx[i];
            int nc = col + dy[i];
            dfs(A, vis, nr, nc);
        }
    }
}
