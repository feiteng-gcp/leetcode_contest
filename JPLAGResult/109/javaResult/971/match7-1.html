class Solution {
    public int shortestBridge(int[][] A) {
        int n = A.length;
        boolean flag = false;
        Queue<Integer> queue = new LinkedList<>();
        boolean[][] seen = new boolean[n][n];
        
for(int i = 0 ; i < A.length ; i++){
            for(int j = 0 ; j < A[0].length ; j++){
                if(A[i][j] == 1){
                    dfs(i, j, queue, A, seen);
                    flag = true;
                    break;
                }
            }
            if(flag) break;
        }
        int count = 0;
        while(!queue.isEmpty()){
            int m = queue.size();
            //System.out.println("before:" + queue);
            for(int i = 0 ; i < m ; i++){
                int pos = queue.poll();
                int x = pos / n;
                int y = pos % n;
                int[] dx = {0, 1, 0, -1};
                int[] dy = 
{1, 0, -1, 0};
                for(int j = 0 ; j < dx.length ; j++){
                    int nx = x + dx[j];
                    int ny = y + dy[j];
                    if(isValid(nx, ny, A)) {
                        if(A[nx][ny] == 0 && !seen[nx][ny]){
                            seen[nx][ny] = true;
                            queue.offer(nx * n + ny);
                        }else if(A[nx][ny] == 1 && !seen[nx][ny]){
                            return count;
                        }
                    
}
                }
            }
            //System.out.println("after:" + queue);
            count++;
        }
        return count;
    }
    
    private void dfs(int i, int j, Queue<Integer> queue, int[][] A, boolean[][] seen){
        if(!isValid(i, j, A) || A[i][j] == 0 || seen[i][j]) return;
        seen[i][j] = true;
        queue.offer(i * A.length + j);
        dfs(i + 1, j, queue, A, seen);
        dfs(i - 1, j, queue, A, seen);
        dfs(i, j + 1, queue, A, seen);
        dfs(i, j - 1, queue, A, seen);
    }
    
    private boolean isValid(int i, int j, int[][] A){
        if(i < 0 || i >= A.length || j < 0 || j >= A[0].length) return false;
        return true;
    }
}
