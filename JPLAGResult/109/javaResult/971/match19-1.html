class Solution {
    public int shortestBridge(int[][] A) {
        Queue<int[]> list = new LinkedList<>();
        
boolean flag = false;
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                if (A[i][j] == 1) {
                    helper(A, i, j, list);
                    // System.out.println(list.size());
                    flag = true;
                    break;
                }
            }
            if (flag)
                break;
        }
        
int res = 0;
        // System.out.println(list.size());
        int[][] dir = new int[][]{{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        while (!list.isEmpty()) {
            for (int s = list.size(); s > 0; s--) {
                int[] node = list.poll();
                
for (int[] d : dir) {
                    int r = node[0] + d[0], c = node[1] + d[1];
                    if (r >= 0 && r < A.length && c >= 0 && c < A[0].length) {
                        if (A[r][c] == 1) {
                            return res;
                        } else if (A[r][c] == 0) {
                            list.offer(new int[]{r, c});
                            A[r][c] = 2;
                        }
                    }
                }
            
}
            res++;
        }
        return -1;
    }
    
    public void helper(int[][] A, int i, int j, Queue<int[]> list) {
        int row = A.length, col = A[0].length;
        if (i < 0 || i >= row || j < 0 || j >= col || A[i][j] != 1)
            return;
        list.offer(new int[]{i, j});
        A[i][j] = 2;
        helper(A, i+1, j, list);
        helper(A, i, j+1, list);
        helper(A, i-1, j, list);
        helper(A, i, j-1, list);
    }
}
