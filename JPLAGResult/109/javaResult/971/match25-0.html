class Solution {
    public int shortestBridge(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        int[][] dict = {{-1,0}, {1,0},{0,1},{0,-1}};
        boolean finish = false;
        List<List<Integer>> l = new java.util.ArrayList();
        for(int i = 0; i < m && !finish; i++){
            for(int j = 0; j < n && !finish; j++) {
                if(A[i][j] == 1) {
                    form1(A, dict, i, j, l);
                    finish = true;
                }
            }
        }
        int res = 1;
        while(l.size() != 0) {
            int size = l.size();
            while(size > 0){
                List<Integer> temp = l.get(0);
                size--;
                for(int a = 0; a < 4; a++) {
                    int r = temp.get(0) + dict[a][0];
                    int c = temp.get(1) + dict[a][1];
                    if(r < 0 || c < 0 || r == A.length || c == A[0].length) continue;
                    if(A[r][c] == 1) return res;
                    else if(A[r][c] == 0) {
                        A[r][c] = -1;
                        l.add(Arrays.asList(r,c));
                    }
                }
                l.remove(0);
            }
            res++;
        }
        return res;
    }
    public void form1(int[][] A, int[][] dict, int i, int j, List<List<Integer>> l) {
        A[i][j] = 2;
        for(int a = 0; a < 4; a++) {
            int r = i + dict[a][0];
            int c = j + dict[a][1];
            if(r < 0 || c < 0 || r == A.length || c == A[0].length) continue;
            if(A[r][c] == 1) form1(A, dict, r, c, l);
            else {
                A[r][c] = -1;
                l.add(Arrays.asList(r, c));
            };
        }  
    }
}
