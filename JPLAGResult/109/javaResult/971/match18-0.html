class Solution {
    int[][] offset = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int shortestBridge(int[][] grid) {
        List<int[]> list1 = new ArrayList<>();
        List<int[]> list2 = new ArrayList<>();
        int m = grid.length;
        int n = grid[0].length;
        int count = 0;
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(grid[i][j] == 1){
                    List<int[]> list = new ArrayList<>();
                    merge(grid, i, j, list);
                    count++;
                    if(count == 1){
                        list1 = new ArrayList<>(list);
                    }else{
                        list2 = new ArrayList<>(list);
                    }
                }
            }
        }
        int min = Integer.MAX_VALUE;
        // System.out.println(list1);
        // System.out.println(list1);
        for(int[] x : list1){
            for(int[] y : list2){
                min = Math.min(min, Math.abs(x[0] - y[0]) + Math.abs(x[1] - y[1]) - 1);
            }
        }
        return min;
    }
    public void merge(int[][] grid, int i, int j, List<int[]> list){
        int m = grid.length;
        int n = grid[0].length;
        if(i < 0 || i >= m || j < 0 || j >= n || grid[i][j] == 0){
            return;
        }
        grid[i][j] = 0;
        list.add(new int[]{i, j});
        for(int[] xy : offset){
            int x = i + xy[0];
            int y = j + xy[1];
            merge(grid, x, y, list);
        }
    }
}
