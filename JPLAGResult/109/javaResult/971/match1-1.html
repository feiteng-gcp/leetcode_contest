class Solution {

    public int shortestBridge(int[][] A) {
        
int x = 0, y = 0;
        for (int r = 0; r < A.length; r++) {
            for (int c = 0; c < A[0].length; c++) {
                if (A[r][c] == 1) {
                    x = r;
                    y = c;
                    break;
                }
            }
        }
        for (int move = 1; ; move++) {
            boolean[][] visited = new boolean[A.length][A[0].length];
            visit(x, y, A, visited);

            
for (int r = 0; r < A.length; r++) {
                for (int c = 0; c < A[0].length; c++) {
                    if (visited[r][c]) {
                        expand(r, c, A);
                    }
                }
            }
            if (connected(x, y, A)) {
                return move;
            }
        }
    }

    private void expand(int x, int y, int[][] A) {
        int[][] dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        for (int[] dir : dirs) {
            int r = x + dir[0];
            int c = y + dir[1];
            if (r >= 0 && r < A.length && c >= 0 && c < A[0].length) {
                A[r][c] = 1;
            }
        }
    }

    private boolean connected(int x, int y, int[][] A) {
        boolean[][] visited = new boolean[A.length][A[0].length];
        visit(x, y, A, visited);
        for (int r = 0; r < A.length; r++) {
            for (int c = 0; c < A[0].length; c++) {
                if (A[r][c] == 1 && !visited[r][c]) {
                    return false;
                
}
            }
        }
        return true;
    }

    private void visit(int r, int c, int[][] A, boolean[][] visited) {
        
if (r < 0 || r >= A.length || c < 0 || c >= A[0].length || A[r][c] == 0 || visited[r][c]) {
            return;
        }
        visited[r][c] = true;
        visit(r - 1, c, A, visited);
        visit(r + 1, c, A, visited);
        visit(r, c - 1, A, visited);
        visit(r, c + 1, A, visited);
    }
}
