class Solution {
    
    int[][] dirs = new int[][]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    
    int m, n;
    
    boolean inRange(int a, int b) {
        return a >= 0 && a < m && b >= 0 && b < n;
    }
    
    public int shortestBridge(int[][] A) {
        
        m = A.length;
        n = A[0].length;
        
        Queue<int[]> qu = new LinkedList<>();
        
        boolean flag = false;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1) {
                    DFS(A, i, j, qu);
                    flag = true;
                    break;
                }
            }
            if (flag) break;
        }
        
        int step = 1;
        while (!qu.isEmpty()) {
            int level = qu.size();
            for (int i = 0; i < level; i++) {
                
                int[] loc = qu.poll();
                
                // System.out.println(loc[0]+" "+loc[1]);
                for (int[] dir: dirs) {
                    int r = loc[0]+dir[0];
                    int c = loc[1]+dir[1];
                    
                    if (inRange(r, c) && A[r][c] == 0) {
                        A[r][c] = -1;
                        
                        for (int[] dir2: dirs) {
                            int nr = r +dir2[0];
                            int nc = c+ dir2[1];
                            
                            if (inRange(nr, nc) && A[nr][nc] == 1) {
                                return step;
                            }
                        }
                        qu.offer(new int[]{r, c});
                    }
                }
                
            }
            step++;
        }
        
        return -1;
        
    }
    
    void DFS(int[][] A, int r, int c, Queue<int[]> qu) {
        A[r][c] = -1;
        
        qu.offer(new int[]{r, c});
        
        for (int[] dir: dirs) {
            int nr = r +dir[0];
            int nc = c + dir[1];
            if (inRange(nr, nc) && A[nr][nc] == 1) {
                
                DFS(A, nr, nc, qu);
                
            }
        }
    }
}
