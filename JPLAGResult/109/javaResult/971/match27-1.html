class Solution {
    
    int[][] dirs = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int shortestBridge(int[][] A) {
        Set<Integer> first = new HashSet<>();
        boolean find = false;
        for(int i = 0; i < A.length;i ++) {
            for(int j = 0; j < A[0].length;j ++) {
                if (A[i][j] == 1) {
                    dfs(A, i, j, first);
                    find = true;
                    break;
                }
            }
            if (find) {
                break;
            }
        }
        // System.out.println(first);
        
        
Set<Integer> second = new HashSet<>();
        for(int i = 0; i < A.length;i ++) {
            for(int j = 0; j < A[0].length;j ++) {
                if (A[i][j] == 1) {
                    dfs(A, i, j, second );
                    break;
                }
            }
        }    
        
        // System.out.println(second);
        LinkedList<Integer> q = new LinkedList<>(first);
        // System.out.println(q);
        int cnt = 0;
        
while(!q.isEmpty()) {
            int size = q.size();
            cnt++;
            for(int i = 0;i < size;i ++) {
                int cur = q.pollFirst();
                
int row = cur / A[0].length;
                int col = cur % A[0].length;
                for(int[] dir : dirs) {
                    int ii = row + dir[0];
                    int jj = col + dir[1];
                    if (ii >= 0 && ii < A.length && jj >= 0 && jj < A[0].length) {
                        int next = ii * A[0].length + jj;
                        if (second.contains(next)) {
                            return cnt - 1;
                        }
                        if (A[ii][jj] == 0 && !first.contains(next)) {
                            first.add(next);
                            q.offerLast(next);
                        }
                    
}
                }
            }
        }
        return -1;
    }
    
    
    private void dfs(int[][] A, int i, int j, Set<Integer> res) {
        if (i >= 0 && i < A.length && j >= 0 && j < A[0].length && A[i][j] == 1) {
            A[i][j] = 2;
            if (side(i, j, A)) {
                res.add(i * A[0].length + j);
            }
            dfs(A, i + 1, j , res);
            dfs(A, i - 1, j , res);
            dfs(A, i, j + 1 , res);
            dfs(A, i, j - 1 , res);
        }
    }
    
    
    
    private boolean side(int i, int j, int[][] A) {
        for(int[] dir : dirs) {
            int ii = i + dir[0];
            int jj = j + dir[1];
            if (ii >= 0 && ii < A.length && jj >= 0&& jj < A[0].length && A[ii][jj] == 0) {
                return true;
            }
        }
        return false;
    }
}
