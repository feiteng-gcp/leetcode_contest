class Solution {
    static int[][] ops = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
    public int shortestBridge(int[][] A) {
        Set<Integer> bset = new HashSet(), aset = new HashSet();
        
Queue<Integer> q = new LinkedList();
        for (int i = 0; i < A.length; ++i) {
            for (int j = 0; j < A.length; ++j) {
                if (1 == A[i][j]) {
                    int k = i * 100 + j;                    
                    if (q.isEmpty()) { 
                        q.offer(k);
                        aset.add(k);
                    } else {
                        
bset.add(k);
                    }
                }
            }
        }
        while (!q.isEmpty()) 
{
            int p = q.poll();
            int i = p / 100;
            int j = p % 100;
            for (int[] op : ops) {
                int x = i + op[0];
                int y = j + op[1];
                if (0 <= x && x < A.length && 0 <= y && y < A.length && 1 == A[x][y] && aset.add(x * 100 + y)) {
                    q.offer(x * 100 + y);
                    bset.remove(x * 100 + y);
                }
            }
        }
        int dist = Integer.MAX_VALUE;
        for (int a : aset) {
            int i = a / 100;
            int j = a % 100;
            for (int b : bset) {
                int x = b / 100;
                int y = b % 100;
                dist = Math.min(dist, Math.abs(i - x) + Math.abs(j - y));
            }
        }
        return dist - 1;
    }
}
