class Solution {
    int[][] direct = 
new int[][]{{1,0},{-1,0},{0,1},{0,-1}};
    public int shortestBridge(int[][] A) {
        
List<int[]> island1 = new ArrayList();
        List<int[]> island2 = new ArrayList();
        for(int i = 0;i < A.length;i++) {
            for(int j = 0;j < A[0].length;j++) {
                if(A[i][j] == 0 ||A[i][j] == 2) continue;
                dfs(A, i, j, island1.size() == 0?island1:island2);
            
}
        }
        int dis = Integer.MAX_VALUE;
        for(int[] i: island1) {
            for(int[] j: island2) {
                dis = Math.min(dis, Math.abs(i[0]-j[0])+
Math.abs(i[1]-j[1]));
            }
        }
        return dis-1;     
    }
    public void dfs(int[][] A, int i, int j, List<int[]> island) {
        if(isEdge(A, i, j)) island.add(new int[]{i,j});
        A[i][j] = 2;
        for(int[] d: direct) {
            int x = d[0]+i;
            int y = d[1]+j;
            if(x < 0 || y < 0 || x >= A.length || y >= A[0].length ||A[x][y] == 0 || A[x][y] == 2) continue;
            dfs(A, x, y, island);
        }
    }
    public boolean isEdge(int[][] A, int r, int c) {
        for(int[] d: direct) {
            int x = d[0]+r;
            int y = d[1]+c;
            if(x < 0 || y < 0 || x >= A.length || y >= A[0].length) continue;
            if(A[x][y] == 0) return true;
        }
        return false;
    }
}
