class Solution {
    int[] dx = {1, 0, -1, 0};
    int[] dy = {0, 1, 0, -1};
    public int shortestBridge(int[][] A) {
        int n = A[0].length;
        
        int x = 0, y = 0;
        for (int i = 0; i < n; i++) {
            if (A[x][y] == 1) break;
            
for (int j = 0; j < n; j++) {
                if (A[i][j] == 1) {
                    x = i;
                    y = j;
                    break;
                }
            }
        }
        
        ArrayDeque<Integer> queuex = new ArrayDeque<>();
        ArrayDeque<Integer> queuey = new ArrayDeque<>();
        boolean[][] visited = new boolean[n][n];
        dfs(A, x, y, visited, queuex, queuey);
        
        int res = 0;
        
        while (!queuex.isEmpty()) {
            ArrayDeque<Integer> queuex2 = new ArrayDeque<>();
            ArrayDeque<Integer> queuey2 = new ArrayDeque<>();
            
            while (!queuex.isEmpty()) {
                x = queuex.poll();
                y = queuey.poll();
                if (A[x][y] == 1) return res;
                
for (int i = 0; i < 4; i++) {
                    int x2 = x + dx[i];
                    int y2 = y + dy[i];
                    if (x2 < 0 || x2 >= n || y2 < 0 || y2 >= n || visited[x2][y2]) {
                        continue;
                    }
                    
                    visited[x2][y2] = true;
                    queuex2.offer(x2);
                    queuey2.offer(y2);
                }
                
                
            }
            queuex = queuex2;
            queuey = queuey2;
            res++;
        }
        
        return res;
        
    }
    
    void dfs(int[][] A, int x, 
int y, boolean[][] visited, ArrayDeque<Integer> queuex, ArrayDeque<Integer> queuey) {
        int n = A[0].length;
        
        if (x < 0 || x >= n || y < 0 || y >= n || visited[x][y]) {
            return;
        }
        visited[x][y] = true;
        if (A[x][y] == 0) {
            queuex.add(x);
            queuey.add(y);
            return;
        }
        
for (int i = 0; i < 4; i++) {
            int x2 = x + dx[i];
            int y2 = y + dy[i];
            dfs(A, x2, y2, visited, queuex, queuey);
        }
        
    }
}
