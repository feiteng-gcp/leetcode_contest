class Solution {
    public int shortestBridge(int[][] A) {
        boolean[][] visited = new boolean[A.length][A[0].length];
        Queue<int[]> queue = new LinkedList();//firstIsland
        out:for(int i=0;i<A.length;i++){
            for(int j=0;j<A.length;j++){
                if(A[i][j]==1){
                    flood(A, visited, i, j, queue);
                    break out;
                }
            }
        }
        
        
int[][] sur = new int[][]{{-1,0},{1,0},{0,-1},{0,1}};
        
while(!queue.isEmpty()){
            int[] cur = queue.poll();
            for(int i=0;i<sur.length;i++){
                int col=cur[0]+sur[i][0], row=cur[1]+sur[i][1];
                if(col<0||col>=A.length||row<0||row>=A[0].length||visited[col][row]) continue;
                if(A[col][row]==1){
                    return cur[2];
                
}else{
                    visited[col][row]=true;
                    queue.offer(new int[]{col,row,cur[2]+1});
                }
            
}
        }
        
        return 1;
    }
    
    private void flood(int[][] A, boolean[][] visited, int i, int j, Queue<int[]> queue){
        if(i<0||i>=A.length||j<0||j>=A[0].length||A[i][j]==0||visited[i][j]) return;
        
        visited[i][j]=true;
        queue.offer(new int[]{i,j,0});
        flood(A, visited, i-1, j, queue);
        flood(A, visited, i+1, j, queue);
        flood(A, visited, i, j-1, queue);
        flood(A, visited, i, j+1, queue);
    }
}
