class Solution {

    private int[][] A;
    private int r;
    private int c;
    
    private int[][] dirs;
    
    public int shortestBridge(int[][] A) {
        this.A = A;
        r = A.length;
        c = A[0].length;
        
        dirs = 
new int[][] {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        
        outer:
        
for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                if (A[i][j] == 1) {
                    mark(i, j);
                    break outer;
                }
            }
        }
        
        LinkedList<int[]> queue = new LinkedList<>();
        for (int i = 0; i < r; i++) 
            for (int j = 0; j < c; j++)
                if (A[i][j] == 1) queue.addLast(new int[] {i, j});
        
        int res = 0;
        while (true) {
            int size = queue.size();
            for (int i = 0; i < size; i++) 
{
                int[] arr = queue.removeFirst();
                for (int[] dir: dirs) {
                    int ni = arr[0] + dir[0];
                    int nj = arr[1] + dir[1];
                    if (ni >= 0 && ni < r && nj >= 0 && nj < c && A[ni][nj] != 1) {
                        if (A[ni][nj] == 2) return res;
                        
                        
A[ni][nj] = 1;
                        queue.addLast(new int[] {ni, nj});
                    }
                }
            }
            
            res++;
        }
    }
    
    private void mark(int i, int j) {
        if (i < 0 || i >= r || j < 0 || j >= c || A[i][j] != 1) return;
        
        A[i][j] = 2;
        
        for (int[] dir: dirs)
            mark(i + dir[0], j + dir[1]);
    }
    
}
