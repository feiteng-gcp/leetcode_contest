class Solution {
     public int shortestBridge(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        int[][] color = new int[m][n];

        int finalColor = 1;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1 && color[i][j] == 0) {
                    dfs(A, i, j, color, finalColor++);
                }
            }
        }

        List<int[]> color2 = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (color[i][j] == 2) {
                    color2.add(new int[]{i, j});
                }
            }
        }
        int[][] c2 = new int[color2.size()][2];
        int len= 0;
        for (int[] e : color2) {
            c2[len][0] = e[0];
            c2[len][1] = e[1];
            len++;
        }

        int ans = Integer.MAX_VALUE;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (color[i][j] == 1) {
                    for (int k = 0; k < c2.length; k++) {
                        int x = c2[k][0];
                        int y = c2[k][1];
                        int t = Math.abs(i - x) + Math.abs(j - y);
                        ans = Math.min(ans, t);
                    }
                }
            }
        }
        return ans - 1;
    }

    void dfs(int[][] A, int i, int j, int[][] color, int finalColor) {
        int m = A.length;
        int n = A[0].length;
        if (!(i >= 0 && i < m && j >= 0 && j < n)) return;
        if (A[i][j] != 1 || color[i][j] > 0) return;
        color[i][j] = finalColor;

        dfs(A, i + 1, j, color, finalColor);
        dfs(A, i - 1, j, color, finalColor);
        dfs(A, i, j + 1, color, finalColor);
        dfs(A, i, j - 1, color, finalColor);
    }

}
