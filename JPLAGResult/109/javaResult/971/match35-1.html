class Solution {
    
private int[][] dir = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    public int shortestBridge(int[][] A) {
        int wid = A.length, len = A[0].length;
        
Queue<int[]> q = new LinkedList<>();
        boolean found = false;
        for (int j = 0; j < wid; j++) {
            if (found) {
                break;
            }
            for (int i = 0; i < len; i++) {
                if (A[j][i] == 1) {
                    bfs(A, j, i, q);
                    found = true;
                    break;
                }
            }
        }
        int res = 0;
        Set<String> visited = new HashSet<>();
        while (!q.isEmpty()) {
            int size = q.size();
            for (int l = 0; l < size; l++) 
{
                int[] cur = q.poll();
                for (int k = 0; k < 4; k++) {
                    int j = cur[0] + dir[k][0], i = cur[1] + dir[k][1];
                    if (j < 0 || j >= A.length || i < 0 || i >= A[0].length || A[j][i] == 2 || visited.contains(j + "," + i)) {
                        continue;
                    }
                    if (A[j][i] == 1) {
                        return res;
                    }
                    visited.add(j + "," + i);
                    q.offer(new int[]{j, i});
                }
            }
            res++;
        
}
        return 0;
    }
    
    private void bfs(int[][] A, int staj, int stai, Queue<int[]> q) {
        Queue<int[]> tmpq = new LinkedList<>();
        Set<String> visited = new HashSet<>();
        tmpq.offer(new int[]{staj, stai});
        A[staj][stai] = 2;
        q.offer(new int[]{staj, stai});
        visited.add(staj + "," + stai);
        while (!tmpq.isEmpty()) {
            int[] cur = tmpq.poll();
            for (int k = 0; k < 4; k++) {
                int j = cur[0] + dir[k][0], i = cur[1] + dir[k][1];
                if (j < 0 || j >= A.length || i < 0 || i >= A[0].length || A[j][i] != 1 || visited.contains(j + "," + i)) {
                    continue;
                }
                A[j][i] = 2;
                visited.add(j + "," + i);
                tmpq.offer(new int[]{j, i});
                q.offer(new int[]{j, i});
            }
        }
    }
    
    private void print(int[][] A) {
        for (int j = 0; j < A.length; j++) {
            for (int i = 0; i < A[0].length; i++) {
                System.out.print(A[j][i]);
            }
            System.out.println();
        }
        System.out.println();
    }
}
