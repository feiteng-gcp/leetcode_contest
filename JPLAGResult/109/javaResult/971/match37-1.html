class Solution {
    int[] X = {1, -1, 0, 0};
    int[] Y = {0, 0, 1, -1};
    public int shortestBridge(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        boolean found = false;
        Queue<int[]> queue = new LinkedList<>();
        
for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1) {
                    if (!found) {
                        dfs(A, i, j);
                        found = true;    
                    } else {
                        queue.offer(new int[]{i, j});
                    
}
                }
            }
        }
        int step = 0;
        
while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int[] cur = queue.poll();
                for (int j = 0; j < 4; j++) {
                    int row = cur[0] + X[j];
                    int col = cur[1] + Y[j];
                    if (row < 0 || row >= m || col < 0 || col >= n || A[row][col] == 1) continue;
                    if (A[row][col] == 2) return step;
                    A[row][col] = 1;
                    queue.offer(new int[]{row, col});
                
}
            }
            step++;
        }
        return -1;
    }
    public void dfs(int[][] A, int row, int col) {
        int m = A.length;
        int n = A[0].length;
        if (row < 0 || row >= m || col < 0 || col >= n || A[row][col] != 1) return;
        A[row][col] = 2;
        for (int i = 0; i < 4; i++) {
            dfs(A, row + X[i], col + Y[i]);
        }
    }
}
