class Solution {
    int[][] arr = new int[][]{
            {0, 1}, {0,-1}, {1, 0}, {-1, 0}
    };
    public int shortestBridge(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        boolean[][] visited = new boolean[m][n];
        int a = 0;
        
int b = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (A[i][j] == 1) {
                    a = i;
                    b = j;
                    break;
                }
            }
        }
        Queue<int[]> queue = new LinkedList<>();
        initFirstIsland(visited, a, b, A, queue);
        int result = 0;
        
while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; ++i) {
                int[] pos = queue.poll();
                for (int[] tmpArr : arr) {
                    int r = pos[0] + tmpArr[0];
                    int c = pos[1] + tmpArr[1];
                    if (r < 0 || r >= A.length || c < 0 || c >= A[0].length || visited[r][c]) {
                        continue;
                    }
                    if (A[r][c] == 1) {
                        return result;
                    }
                    queue.add(new int[]{r,c});
                    visited[r][c] = true;
                
}
            }
            result++;
        }
        return result;
    }

    private void initFirstIsland(boolean[][] visited, int a, int b, int[][] A, Queue<int[]> queue) {
        if (a < 0 || a >= A.length || b < 0 || b >= A[0].length || visited[a][b]) {
            return;
        }
        if (A[a][b] == 1) {
            visited[a][b] = true;
            queue.add(new int[]{a, b});
            initFirstIsland(visited, a - 1, b, A, queue);
            initFirstIsland(visited, a + 1, b, A, queue);
            initFirstIsland(visited, a, b - 1, A, queue);
            initFirstIsland(visited, a, b + 1, A, queue);
        }
    }
}
