class Solution {
    
    int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        
    public int shortestBridge(int[][] A) {
        Set<String> setA = new HashSet<>();
        Set<String> setB = new HashSet<>();
        int m = A.length, n = A[0].length;
        boolean[][] visited = new boolean[m][n];
        int turn = 0;
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(A[i][j] == 1 && !visited[i][j]){
                    if(turn == 0){
                        dfs(setA, A, visited, i, j);
                        turn = 1;
                    }else{
                        dfs(setB, A, visited, i, j);
                    }
                }
            }
        }
            
        int res = 0;
        visited = new boolean[m][n];
        Queue<int[]> q = new LinkedList<>();
        for(String s :setA){
            String[] ss = s.split(":");
            q.add(new int[]{Integer.valueOf(ss[0]), Integer.valueOf(ss[1])});
        }
        while(!q.isEmpty()){
            int size = q.size();
            for(int i = 0; i < size; i++){
                int[] pos = q.poll();
                for(int[] dir : dirs){
                    int x = dir[0] + pos[0];
                    int y = dir[1] + pos[1];
                    if(x >= 0 && x < A.length && y >= 0 && y < A[0].length){
                        if(A[x][y] == 1){
                            if(setB.contains(x + ":" + y)){
                                return res;
                            }
                        }else{
                            if(visited[x][y]){
                                continue;
                            }else{
                                visited[x][y] = true;
                                q.add(new int[]{x, y});
                            }
                        }
                    }
                }
            }
            res++;
        }
        return 0;
    }
    
    public void dfs(Set<String> set, int[][] A, boolean[][] visited, int i, int j){
        if(visited[i][j]){
            return;
        }
        visited[i][j] = true;
        set.add(i + ":" + j);
        for(int[] dir : dirs){
            int x = dir[0] + i;
            int y = dir[1] + j;
            if(x >= 0 && x < A.length && y >= 0 && y < A[0].length && A[x][y] == 1){
                dfs(set, A, visited, x, y);
            }
        }
    }
}
