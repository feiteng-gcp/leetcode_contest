class Solution {
    public int shortestBridge(int[][] A) {
        boolean shouldBreak = false;
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                if (A[i][j] == 1) {
                    mark(A, i, j);
                    shouldBreak = true;
                    break;
                }
            }
            if (shouldBreak) {
                break;
            }
        }
        Queue<int[]> q = new LinkedList<>();
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                if (A[i][j] == 1) {
                    q.offer(new int[] {i, j});
                }
            }
        }
        q.offer(null);
        int res = 0;
        int[][] ds = {{1,0},{-1,0},{0,1},{0,-1}};
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            if (cur == null) {
                res++;
                q.offer(null);
                continue;
            }
            for (int[] d : ds) {
                int row = cur[0] + d[0], col = cur[1] + d[1];
                if (row >= 0 && row < A.length && col >= 0 && col < A[0].length && A[row][col] != 1) {
                    if (A[row][col] == 2) {
                        return res;
                    } else {
                        A[row][col] = 1;
                        q.offer(new int[] {row, col});
                    }
                }
            }
        }
        return -1;
    }
    
    private void mark(int[][] A, int i, int j) {
        if (i < 0 || i >= A.length || j < 0 || j >= A[0].length || A[i][j] != 1) {
            return;
        }
        A[i][j] = 2;
        mark(A, i - 1, j);
        mark(A, i + 1, j);
        mark(A, i, j - 1);
        mark(A, i, j + 1);
    }
    
}
