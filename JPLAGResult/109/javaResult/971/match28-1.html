class Solution {
    
private static int[][] dirs = {
        {0, 1}, {0, -1}, {1, 0}, {-1, 0}
    };
    
    private void island(int[][] A, List<int[]> points, int r, int c, boolean[][] v) {
        List<int[]> Q = new LinkedList<int[]>();
        Q.add(new int[] {r, c});
        
        while(!Q.isEmpty()) {
            int[] n = Q.remove(0);
            int nr = n[0], nc = n[1];
            if(v[nr][nc]) {
                continue;
            }
            
            points.add(n);
            v[nr][nc] = true;
            
            for(int i=0; i<dirs.length; i++) {
                int tr = nr+dirs[i][0];
                int tc = nc+dirs[i][1];
                
                if(tr >= 0 && tr < A.length && tc >=0 && tc < A[0].length && !v[tr][tc] && A[tr][tc] == 1) {
                    Q.add(new int[] {tr, tc});
                
}
            }
        }
    }
    
    public int shortestBridge(int[][] A) {
        List<int[]> points1 = new LinkedList<int[]>();
        List<int[]> points2 = new LinkedList<int[]>();
        boolean[][] v = new boolean[A.length][A[0].length];

        
boolean first = true;
        
        outer: for(int r=0; r<A.length; r++) {
            for(int c=0; c<A[0].length; c++) {
                if(v[r][c] || A[r][c] == 0) {
                    continue;
                }
                
                if(first) {
                    island(A, points1, r, c, v);
                    first = false;
                } else {
                    island(A, points2, r, c, v);
                    break outer;
                
}
            }
        }
        
        int min = Integer.MAX_VALUE;
        for(int[] p1 : points1) {
            for(int[] p2 : points2) {
                int d;
                if(p1[0] == p2[0]) {
                    d = Math.abs(p1[1]-p2[1])-1;
                } else if(p1[1] == p2[1]) {
                    d = Math.abs(p1[0]-p2[0])-1;
                } else {
                    d = Math.abs(p1[0]-p2[0]) + Math.abs(p1[1]-p2[1]) - 1;
                }
                
                min = 
Math.min(d, min);
            }
        }
        
        return min;
    }
}
