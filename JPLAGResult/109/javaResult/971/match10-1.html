class Solution {
     public int shortestBridge(int[][] A) {
        int N = A.length;
        List<int[]> isLand1 = new ArrayList<>(10000);
        List<int[]> isLand2 = new ArrayList<>(10000);
        
boolean first = true;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (A[i][j] == 1) {
                    if (first) {
                        first = false;
                        isLand1.add(new int[]{i, j});
                        A[i][j] = 0;
                        floodFill(i, j, A, isLand1);
                    } else {
                        isLand2.add(new int[]{i, j});
                        A[i][j] = 0;
                        floodFill(i, j, A, isLand2);
                    }
                
}
            }
        }
        int result = 2 * N;
        for (int[] p1 : isLand1) {
            for (int[] p2 : isLand2) {
                result = Integer.min(result, Math.abs(p1[0] - p2[0]) + Math.abs(p1[1] - p2[1]) - 1);
            }
        }
        return result;
    }

    private void floodFill(int x, int y, int[][] A, List<int[]> list) {
        Deque<int[]> deque = new LinkedList<>();
        
int[][] NSWE = new int[][]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        deque.add(new int[]{x, y});
        
while (!deque.isEmpty()) {
            int[] cur = deque.pollFirst();
            for (int i = 0; i < NSWE.length; i++) {
                int[] next = add(cur, NSWE[i]);
                if (next[0] >= 0 && next[0] < A.length && next[1] >= 0 && next[1] < A[0].length) {
                    if (A[next[0]][next[1]] == 1) {
                        deque.offerLast(next);
                        list.add(next);
                        A[next[0]][next[1]] = 0;
                    
}
                }
            }
        }
    }

    private int[] add(int[] v1, int[] v2) {
        return new int[]{v1[0] + v2[0], v1[1] + v2[1]};
    }
}
