class Solution {
public:
    vector<int> movesToStamp(string stamp, string target) {
        string qm(target.length(), '?');
        if (!isSameLetters(stamp, target)) {
            return {};
        }
        if (target.front() != stamp.front() || target.back() != stamp.back()) {
            return {};
        }
        vector<int> indices;
        int s = find(target, target.length() - stamp.length(), stamp);
        while(s >= 0) {
            for (int i = 0; i < stamp.length(); i++) {
                target[s + i] = '?';
            }
            indices.push_back(s);
            if (target == qm) {
                break;
            }
            if (s == 0) {
                s = find(target, target.length() - stamp.length(), stamp);
                if (s == -1) {
                    return {};
                }
            } else {
                s = find(target, s - 1, stamp);
                if (s == -1) {
                    s = find(target, target.length() - stamp.length(), stamp);
                    if (s == -1) {
                        return {};
                    }
                }
            }
        }
        reverse(indices.begin(), indices.end());
        return indices;
    }
private:
    int find(const string& h, int s, const string& n) {
        for (; s >= 0; s--) {
            if (match(h, s, n)) {
                return s;
            }
        }
        return -1;
    }
    bool match(const string& h, int s, const string& n) {
        bool allQ = true;
        for (int i = 0; i < n.length(); i++) {
            if (h[s + i] != '?' && h[s + i] != n[i]) {
                return false;
            }
            if (h[s + i] != '?') {
                allQ = false;
            }
        }
        return !allQ;
    }
    bool isSameLetters(const string& stamp, const string& target) {
        unordered_set<char> charset(stamp.begin(), stamp.end());
        for (char c : target) {
            if (charset.find(c) == charset.end()) {
                return false;
            }
        }
        return true;
    }
};
