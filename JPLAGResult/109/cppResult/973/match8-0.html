class Solution {
public:

	string munch(string in, string stamp, int idx) {
		string ret = in;
		for (int i=0; i<int(stamp.size()); i++) {
			if (ret[idx+i] == '?')
				ret[idx + i] = stamp[i];
		}
		return ret;
	}

	int MunchCount(string target, string stamp, string curr, int idx) {
		int ret = 0;
		for (int i=0; i<int(stamp.size()); i++) {
			int ii = idx + i;
			if (curr[ii] == '?') {
				if (stamp[i] == target[ii]) {  ret ++; }
				else return 0;
			} else {
				{ }
			}
		}
		return ret;
	}

    vector<int> movesToStamp(string stamp, string target) {
        string curr = "";
		for (int i=0; i<int(target.size()); i++) curr += "?";

		vector<int> ret;
		int attempt = 0;
		while (attempt < target.size()*10) {
			bool moved = false;
			for (int i=0; i<target.size() - stamp.size() + 1; i++) {
				int mc = MunchCount(target, stamp, curr, i);
				if (mc > 0) {
					curr = munch(curr, stamp, i);
					//printf("curr=%s, munch at %d\n", curr.c_str(), i);
					ret.push_back(i);
					moved = true;
					break;
				}
			}

			if (curr == target) break;
			if (moved == false) {
				return vector<int>();
			}
		}
		reverse(ret.begin(), ret.end());
		return ret;
    }
};
