class Solution {
public:
    vector<int> movesToStamp(string stamp, string target) {

        vector<int> res;
        int matched = 0;
        while (matched < target.size()) {
            bool found = false;
            for (int i = 0; i <= target.size() - stamp.size(); ++i) {
                int j;
                for (j = 0; j < stamp.size(); ++j) {
                    if (target[i + j] != '?' && stamp[j] != target[i + j]) {
                        break;
                    }
                }
                if (j == stamp.size()) {
                    int cnt = 0;
                    for (int k = 0; k < stamp.size() ; ++k) {
                        if (target[i + k] != '?') {
                            target[i + k] = '?';
                            ++matched;
                            ++cnt;
                        }
                    }
                    if (cnt > 0) {
                        res.push_back(i);
                        found = true;
                        break;
                    }
                }
            }
            if (!found) {
                res.clear();
                return res;
            }
        }
        vector<int> r;
        for (auto it = res.rbegin(); it != res.rend(); ++it) {
            r.push_back(*it);
        }
        return r;
    }
};
