class Solution {
public:
    //if the sequence can be stamped, then we ought to be able to recognize at least one occurrence of the stamp in the target
    //so we reconstruct the steps from the last one
    //stamping can never exclude any possibilities, so we can try to greedily stamp
    int N,M;
    string S,T;
    vector<int> movesToStamp(string stamp, string target) {
        S=stamp; T=target;
        N=S.size();
        M=T.size();
        if(N>M)return {};
        vector<int> A;
        for(int moves=0;;++moves){
            bool found=false;
            for(int i=0;i<M-N+1;++i){
                bool f=true;
                bool nonzero=false;
                for(int s=0;s<N && i+s<M;++s){
                    if(T[i+s]!='@') nonzero=true;
                    if(T[i+s]!='@' && S[s]!=T[i+s]){
                        f=false;
                        break;
                    }
                }
                if(f && nonzero){
                    //for(int s=0;s<N;++s){
                    for(int s=0;s<N && i+s<M;++s){
                        T[i+s]='@';
                    }
                    A.push_back(i);
                    found=true;
                }
            }
            //printf("%s %s\n",S.c_str(),T.c_str());
            //printf("%d\n",moves);
            if(!found)break;
        }
        for(auto x:T){
            if(x!='@')return {};
        }
        reverse(A.begin(),A.end());
        return A;
    }
};
