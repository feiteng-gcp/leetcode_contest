class Solution {
public:
    vector<int> movesToStamp(string stamp, string target) {
        int n = stamp.size();
        int m = target.size();
        vector<int> res;
        while (true) {
            int pos = -1;
            for (int i = 0; i + n - 1 < m; ++i) {
                bool match = true;
                bool character = false;
                for (int j = 0; j < n; ++j) {
                    if (target[i + j] != '.') {
                        character = true;
                        if (stamp[j] != target[i + j]) {
                            match = false;
                            break;
                        }
                    }
                }
                if (!character) continue;
                if (!match) continue;
                pos = i;
                break;
            }
            if (pos == -1) break;
            // cout << target << endl;
            res.push_back(pos);
            for (int i = 0; i < n; ++i) {
                target[i + pos] = '.';
            }
        }
        for (int i = 0; i < m; ++i) {
            if (target[i] != '.') {
                return vector<int>();
            }
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
