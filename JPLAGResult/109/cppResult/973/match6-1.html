const int N = 1e3 + 10;


class Solution {
public:
    vector<int> movesToStamp(string stamp, string target) {
        int n = target.size(), m = stamp.size();
        vector<int> L(n), R(n);
        for (int i = 0; i < n; ++i) {
            int j;
            for (j = 0; i + j < n && j < m && target[i + j] == stamp[j]; ++j);
            L[i] = j;
            for (j = 0; i - j >= 0 && j < m && target[i - j] == stamp[m - 1 - j]; ++j);
            R[i] = j;
        }
        int cnt = 0;
        vector<int> ret;
        while (true) {
            int x = -1;
            for (int i = 0; i + m <= n; ++i) {
                int j;
                bool found = false;
                for (j = 0; j < m; ++j) {
                    if (target[i + j] != '?') found = true;
                    if (!(target[i + j] == '?' || target[i + j] == stamp[j])) break;
                }
                if (found && j == m) {
                    x = i;
                    break;
                }
            }
            // cout << target << " " << x << " " << cnt << endl;
            if (x < 0) return {};
            ret.push_back(x);
            for (int i = 0; i < m; ++i) {
                if (target[x + i] != '?') ++cnt;
                target[x + i] = '?';
            }
            if (cnt == n) break;
        }
        reverse(ret.begin(), ret.end());
        return ret;


    }
};
