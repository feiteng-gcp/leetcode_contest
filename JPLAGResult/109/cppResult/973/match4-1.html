class Solution {
 public:
  vector<int> movesToStamp(string stamp, string target) {
    const int m = stamp.size(), n = target.size();
    auto eq_target = [&](int start) -> bool {
      bool has = false;
      for (int i = 0; i < m; ++i) {
        if (stamp[i] != target[start + i] && target[start + i] != '?') {
          return false;
        }
        if (target[start + i] != '?') {
          has = true;
        }
      }
      return has;
    };
    vector<int> ans;

    while (true) {
      int x = -1;
      for (int i = 0; i + m <= n; ++i) {
        if (eq_target(i)) {
          x = i;
          break;
        }
      }
      if (x == -1) {
        return {};
      }
      ans.push_back(x);
      for (int i = x; i < x + m; ++i) {
        target[i] = '?';
      }
      bool endgame = true;
      for (int i = 0; i < n; ++i) {
        if (target[i] != '?') {
          endgame = false;
          break;
        }
      }
      if (endgame) {
        reverse(ans.begin(), ans.end());
        return ans;
      }
    }
  }
};
