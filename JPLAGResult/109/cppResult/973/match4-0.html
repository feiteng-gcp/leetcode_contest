class Solution {
    map<int, vector<int>> cache;
public:
    vector<int> dfs(const string& s, const string& t, int tBeg, int tEnd) {
        int key=tBeg*t.size()+tEnd;
        auto it=cache.find(key);
        if (it!=cache.end())return it->second;
        for (int i=max(0, tBeg-(int)s.size()+1);i<tEnd&&i+s.size()<=t.size();++i){
            int ok=1;
            for (int j=0;j<s.size();++j) {
                if (i+j>=tBeg && i+j<tEnd&&t[i+j]!=s[j]){
                    ok=0;
                    break;
                }
            }
            if (ok) {
                if (i<=tBeg&&i+s.size()>=tEnd){
                    return vector<int>(1, i);
                }
                vector<int>l=i>tBeg?dfs(s,t,tBeg,i):vector<int>();
                if(l.size()&&l[0]==-1)continue;
                vector<int>r=i+s.size()>=tEnd?vector<int>():dfs(s,t,i+s.size(),tEnd);
                if (r.size()&&r[0]==-1)continue;
                vector<int>ans;
                for (int k:l)ans.push_back(k);
                for (int k:r)ans.push_back(k);
                ans.push_back(i);
                cache[key]=ans;
                return ans;
            }
        }
        return cache[key]= vector<int>(1,-1);
    }
    vector<int> movesToStamp(string stamp, string target) {
            vector<int> ans=dfs(stamp, target, 0, target.size());
            if (ans.size()&&ans[0]==-1)return {};
            return ans;
    }
};;
