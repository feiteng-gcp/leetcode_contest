class Solution {
public:
    vector<int> movesToStamp(string stamp, string target) {
        vector<int> res;
        int n = target.size();
        int m = stamp.size();
        int cnt=0;

        while(cnt!=n) {
            int flag1=true;
            for(int i=0;i<n-m+1;i++) {
                bool flag=true, check=false;
                for(int j=0;j<m;j++) {
                    if(target[i+j]!='?' && target[i+j]!=stamp[j]) {
                        flag = false;
                        break;
                    }
                    if(target[i+j]==stamp[j]) {
                        check=true;
                    }
                }
                if(flag && check) {
                    flag1=false;
                    res.push_back(i);
                    for(int j=0;j<m;j++) {
                        if(target[i+j]!='?') {
                            cnt++;
                            target[i+j]='?';
                        }
                    }
                    i=i+m;
                }
            }
            if(flag1==true) {
                vector<int> e;
                return e;
            }
            
        }
        vector<int> a;
        for(int i=res.size()-1;i>=0;i--)
            a.push_back(res[i]);
        return a;
    }
};
