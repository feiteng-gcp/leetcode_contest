int dy8[] = {-2,-1,1,2,2,1,-1,-2};
    int dx8[] = {1,2,2,1,-1,-2,-2,-1};
    int dp[5003][4][3];

class Solution {
public:
    const int MOD = 1000000007;
    
    
    int knightDialer(int N) {
        for(int i = 0; i < 5003; ++i) for(int j = 0; j < 4; ++j) for(int k = 0; k < 3; ++k) dp[i][j][k] = 0;
        

        for(int i = 0; i < 3; ++i){
            for(int j = 0; j < 3; ++j){
                dp[0][i][j] = 1;
            }
        }
        dp[0][3][1] = 1;
        
        for(int k = 0; k < N-1; ++k){
            for(int i = 0; i < 4; ++i){
                for(int j = 0; j < 3; ++j){
                    if(i == 3 && (j == 0 || j == 2)) continue;
                    for(int d = 0; d < 8; ++d){
                        int dy = i + dy8[d];
                        int dx = j + dx8[d];
                        if(dy < 0 || dy >= 4 || dx < 0 || dx >= 3) continue;

                        (dp[k + 1][dy][dx] += dp[k][i][j]) %= MOD;
                    }
                }
            }
        }
        int ans = 0;
        for(int i = 0; i < 4; ++i){
            for(int j = 0; j < 3; ++j){
                if(i == 3 && (j == 0 || j == 2)) continue;
                (ans += dp[N-1][i][j]) %= MOD;
            }
        }
        return ans;
    }
};
