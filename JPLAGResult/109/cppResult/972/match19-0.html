class Solution {
public:
    //0,1,2,3,7,8,9,0 - 2, 4,6 - 3
    
    int knightDialer(int N) {
        vector<vector<long>> mem_vec(N, vector<long>(10,0));
        map<int, vector<int>> hop_map;
        hop_map[0] = {4,6};
        hop_map[1] = {6,8};
        hop_map[2] = {7,9};
        hop_map[3] = {4,8};
        hop_map[4] = {3,9,0};
        hop_map[5] = {};
        hop_map[6] = {1,7,0};
        hop_map[7] = {2,6};
        hop_map[8] = {1,3};
        hop_map[9] = {4,2};
        for(int i = 0; i < 10; ++i){
            mem_vec[0][i] = 1;
        }
        for(int i = 1; i < N; ++i){
            for(int j = 0; j < 10; ++j){
                for(auto each : hop_map[j]){
                    mem_vec[i][j] = (mem_vec[i][j] + mem_vec[i-1][each])%(1000000000 + 7) ;
                }
            }
        }
        long val_sum = 0;
        for(int i = 0; i < 10; ++i){
            val_sum = (val_sum + mem_vec[N-1][i])%(1000000000 + 7);
        }
        return val_sum;
    }
};
