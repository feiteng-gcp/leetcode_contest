class Solution {
public:
    int knightDialer(int N) {
        unsigned long long M = 1e9 + 7;
        vector<vector<unsigned long long>> nxt = {
            {4, 6}, // 0
            {6, 8}, // 1
            {7, 9}, // 2
            {4, 8}, // 3
            {3, 9, 0}, // 4
            {}, // 5
            {1, 7, 0}, // 6
            {2, 6}, // 7
            {1, 3}, // 8
            {2, 4} // 9
        };
        vector<vector<unsigned long long>> ans(5001, vector<unsigned long long>(11, 0));
        for(int i = 0; i < 10; ++i) ans[1][i] = 1;
        for(int i = 2; i <= N; ++i) {
            for(int j = 0; j < 10; ++j) {
                for(int d : nxt[j]) {
                    ans[i][j] += ans[i-1][d];
                    ans[i][j] %= M;
                }
            }
        }
        unsigned long long ret = 0;
        for(int i = 0; i < 10; ++i) {
            ret += ans[N][i];
            ret %= M;
        }
        return ret;
    }
};
