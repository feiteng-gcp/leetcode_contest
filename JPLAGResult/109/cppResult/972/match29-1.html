class Solution {
public:
    int knightDialer(int N) {

        vector<vector<int>> jump{{4,6},{8,6},{7,9},{4,8},{0,3,9},{},{0,1,7},{2,6},{1,3},{2,4}};
        int mod = 1e9+7;

        vector<long> dp(10,1);
        for(int k=1; k<N; ++k){
            vector<long> nw(10,0);
            for(int i=0; i<10; ++i){
                for(auto j:jump[i])
                    nw[i] = (nw[i]+dp[j])%mod;
            }
            dp = nw;
        }
        
        long res = 0;
        for(auto i:dp) res += i;
        return res%mod;
    }
};
