class Solution {
public:
    int knightDialer(int N) {
        int mod = pow(10, 9)+7;
        vector<vector<int>> next = {{4, 6}, {6, 8}, {7, 9}, {4, 8}, {0, 3, 9}, {}, {0, 1, 7}, {2, 6}, {1, 3}, {2, 4}};
        vector<long> count(10, 1);
        for (int i = 1; i < N; i++) {
            vector<long> newcount(10, 0);
            for (int n = 0; n < count.size(); n++) {
                for (auto nn : next[n]) {
                    newcount[n] += count[nn];
                    newcount[n] %= mod;
                }                
            }
            count = newcount;
        }
        
        long result = 0;
        for (auto c : count) {
            result += c;
            result %= mod;
        }
        return result;
    }
};
