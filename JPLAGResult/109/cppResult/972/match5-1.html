class Solution {
public:
    int knightDialer(int N) {
        int mod = 1000000007;

        vector<vector<int>> move = {{4,6},{6,8},{7,9},{4,8},{3,9,0},{},{1,7,0},{2,6},{1,3},{2,4}};
        vector<vector<int>> dp(10, vector<int>(N + 1, 0));
        for(int i =0;i<10;i++) {
            dp[i][1] = 1;
        }

        for(int j = 2; j <= N; j++) {
            for(int i = 0;i < 10;i++){
                for(int k = 0; k < move[i].size(); k++) {

                    dp[i][j] = ((dp[i][j] % mod) + (dp[move[i][k]][j-1] % mod)) % mod;
                }
            }
        }
        
        long long sum = 0;
        for(int i = 0;i<10;i++){
            sum = ((sum % mod) + (dp[i][N] % mod)) % mod;
        }
        
        return sum;
    }
};
