class Solution {
public:
    int knightDialer(int N) {
        int modNo = 1000000007;
        vector<int> dp1(10);
        vector<int> dp2(10);
        for (int i = 0; i < 10; i++) {
            dp1[i] = 1;
        }

        for (int i = 2; i <= N; i++) {
            dp2[0] = dp1[4] + dp1[6];
            dp2[1] = dp1[8] + dp1[6];
            dp2[2] = dp1[7] + dp1[9];
            dp2[3] = dp1[4] + dp1[8];
            dp2[4] = ((dp1[0] + dp1[9]) % modNo) + dp1[3];
            dp2[5] = 0;
            dp2[6] = ((dp1[0] + dp1[7]) % modNo) + dp1[1];
            dp2[7] = dp1[2] + dp1[6];
            dp2[8] = dp1[1] + dp1[3];
            dp2[9] = dp1[4] + dp1[2];
            for (int& num : dp2) {
                num %= modNo;
            }
            dp1 = dp2;
        }
        int result = 0;
        for (int& num : dp1) {
            result += num;
            result %= modNo;
        }
        return result;
    }
};
