class Solution {
public:
    int knightDialer(int N) {
        const int MOD = 1000000000 + 7;
        int a[2][10] = {0};
        for (int i = 0; i < 10; i++) a[0][i] = 1;
        int t = 0;

        for (int i = 1; i < N; i++) {
            t = 1 - t;
            a[t][0] = (a[1 - t][4] + a[1 - t][6]) % MOD;
            a[t][1] = (a[1 - t][6] + a[1 - t][8]) % MOD;
            a[t][2] = (a[1 - t][7] + a[1 - t][9]) % MOD;
            a[t][3] = (a[1 - t][4] + a[1 - t][8]) % MOD;
            a[t][4] = ((a[1 - t][3] + a[1 - t][9]) % MOD + a[1 - t][0]) % MOD;    
            a[t][5] = 0;
            a[t][6] = ((a[1 - t][1] + a[1 - t][7]) % MOD + a[1 - t][0]) % MOD;
            a[t][7] = (a[1 - t][2] + a[1 - t][6]) % MOD;
            a[t][8] = (a[1 - t][1] + a[1 - t][3]) % MOD;
            a[t][9] = (a[1 - t][2] + a[1 - t][4]) % MOD;
            //printf("%d %d %d %d %d %d %d %d %d %d \n", a[t][0], a[t][1], a[t][2], a[t][3], a[t][4], a[t][5], a[t][6], a[t][7], a[t][8], a[t][9]);
        }
        int ans = 0;
        for (int i = 0; i< 10; i++) ans = (ans + a[t][i]) % MOD;
        return ans;
    }
};
