const int M = 1000000007;
class Solution {
public:
    int knightDialer(int n) {
        vector<int> to[10] = {
            {4, 6},
            {6, 8},
            {7, 9},
            {4, 8},
            {0, 3, 9}, 
            {},
            {0, 1, 7},
            {2, 6},
            {1, 3},
            {2, 4}
        };
        vector<vector<int>> w(n + 1, vector<int>(10));
        for (int i = 0; i < 10; ++i) w[0][i] = 1;
        for (int i = 0; i < n - 1; ++i) {
            for (int j = 0; j < 10; ++j) {
                for (int x : to[j]) {
                    (w[i + 1][x] += w[i][j]) %= M;
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < 10; ++i) (ret += w[n - 1][i]) %= M;
        return ret;
    }
};
