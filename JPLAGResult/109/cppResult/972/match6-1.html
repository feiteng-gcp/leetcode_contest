const int mod = 1e9 + 7;
class Solution {
public:
    int knightDialer(int N) {

        vector<int> g[10] = {{4, 6}, {6, 8}, {7, 9}, {4, 8}, {0, 3, 9}, {}, {0, 1, 7}, {2, 6}, {1, 3}, {2, 4}};
        int dp[5010][10] = {};

        for (int i = 0; i <= 9; ++i) dp[1][i] = 1;
        for (int j = 2; j <= N; ++j) {
            for (int i = 0; i <= 9; ++i) {
                for (int x : g[i]) {
                    dp[j][i] += dp[j - 1][x];
                    if (dp[j][i] >= mod)

                        dp[j][i] -= mod;
                }
            }
        }
        long long s = 0;
        for (int i = 0; i <= 9; ++i) {
            s += dp[N][i];
        }
        return s % mod;
    }
};
