int dp[5051][4][3];
const int MOD = 1000000007;

int dx[] = {1, 1, -1, -1, 2, 2, -2, -2};
int dy[] = {2, -2, 2, -2, 1, -1, 1, -1};

class Solution {
public:
    int knightDialer(int N) {
        memset(dp, 0, sizeof(dp));
        for (int i = 0; i < 3; ++i) {
            for (int j = 0; j < 3; ++j) {
                dp[0][i][j] = 1;
            }
        }
        dp[0][3][1] = 1;
        for (int step = 1; step <= N; ++step){
            for (int i = 0; i < 4; ++i) {
                for (int j = 0; j < 3; ++j) {
                    if (!dp[step - 1][i][j]) continue;
                    for (int k = 0; k < 8; ++k) {
                        int nx = dx[k] + i;
                        int ny = dy[k] + j;
                        if (nx < 0 || nx >= 4 || ny < 0 || ny >= 3) continue;
                        if (nx == 3 && ny != 1) continue;
                        dp[step][nx][ny] = (dp[step][nx][ny] + dp[step - 1][i][j]) % MOD;
                    }
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < 4; ++i) {
            for (int j = 0; j < 3; ++j) {
                ret = (ret + dp[N - 1][i][j]) % MOD;
            }
        }
        return ret;
    }
};
