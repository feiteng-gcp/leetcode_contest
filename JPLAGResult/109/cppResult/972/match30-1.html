class Solution {
public:
    int knightDialer(int N) {

        vector<vector<int>> next{
            {4, 6},
            {6, 8},
            {7, 9},
            {4, 8},
            {0, 3, 9},
            {},
            {0, 1, 7},
            {2, 6},
            {1, 3},
            {2, 4},
        };


        int64_t mod = 1e9+7, r = 0;
        vector<int64_t> cnt(10, 1);
        for (int i = 1; i < N; i++) {
            vector<int64_t> tmp(10);
            for (int j = 0; j < 10; j++) {
                for (int k : next[j]) {
                    tmp[k] = (tmp[k] + cnt[j]) % mod;
                }
            }
            swap(cnt, tmp);
        }
        for (auto i : cnt)
            r = (r + i) % mod;
        return r;
    }
};
