class Solution {
public:
    int knightDialer(int N) {
	unsigned long long mod = 1e9 + 7;

	vector<vector<int>> moves = {
		{ 4, 6 },
		{ 6, 8 },
		{ 7, 9 },
		{ 4, 8 },
		{ 0, 3, 9 },
		{},
		{ 0, 1, 7 },
		{ 2, 6 },
		{ 1, 3 },
		{ 2, 4 }
	};


	if (N == 1)
		return moves.size();

	vector<vector<unsigned long long>> dp(N + 1, vector<unsigned long long>(10));

	for (int i = 0; i < 10; i++)
		dp[2][i] = moves[i].size();

	for (int i = 3; i <= N; i++){
		for (int j = 0; j < 10; j++){
			for (int m : moves[j]){
				dp[i][j] = (dp[i][j]+dp[i - 1][m])% mod;
			}
		}
	}

	unsigned long long count = 0;
	for (int i = 0; i < 10; i++)
		count = (count+dp[N][i])%mod;
	return count;
    }
};
