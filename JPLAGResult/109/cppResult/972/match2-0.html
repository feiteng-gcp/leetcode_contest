class Solution {
public:
    int knightDialer(int N) {
        for (int i = 0; i < 10; i++) ans[1][i] = 1;
        for (int n = 2; n <= N; n++) {
            ans[n][0] = (ans[n - 1][4] + ans[n - 1][6]) % MOD;
            ans[n][1] = (ans[n - 1][6] + ans[n - 1][8]) % MOD;
            ans[n][2] = (ans[n - 1][7] + ans[n - 1][9]) % MOD;
            ans[n][3] = (ans[n - 1][4] + ans[n - 1][8]) % MOD;
            ans[n][4] = ((ans[n - 1][0] + ans[n - 1][3]) % MOD + ans[n - 1][9]) % MOD;
            ans[n][5] = 0;
            ans[n][6] = ((ans[n - 1][0] + ans[n - 1][1]) % MOD + ans[n - 1][7]) % MOD;
            ans[n][7] = (ans[n - 1][2] + ans[n - 1][6]) % MOD;
            ans[n][8] = (ans[n - 1][1] + ans[n - 1][3]) % MOD;
            ans[n][9] = (ans[n - 1][2] + ans[n - 1][4]) % MOD;
        }
        int ret = 0;
        for (int i = 0; i < 10; i++) {
            ret = (ret + ans[N][i]) % MOD;
        }
        return ret;
    }
    
    int ans[5001][10];
    const int MOD = 1e9+7;
};
