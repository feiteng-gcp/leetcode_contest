class Solution {

public:
    int knightDialer(int N) {
        int dp[10], tmp[10];
        
        for(int i = 0;i < 10;i++)
            dp[i] = 1;
        for(int i = 2;i <= N;i++){
            for(int j = 0;j < 10;j++){
                tmp[j] = dp[j];
            }

            dp[0] = (tmp[4] + tmp[6]) % 1000000007;
            dp[1] = (tmp[6] + tmp[8]) % 1000000007;
            dp[2] = (tmp[7] + tmp[9]) % 1000000007;
            dp[3] = (tmp[4] + tmp[8]) % 1000000007;
            dp[4] = (((tmp[3] + tmp[9]) % 1000000007) + tmp[0]) % 1000000007;
            
            dp[6] = (((tmp[1] + tmp[7]) % 1000000007) + tmp[0]) % 1000000007;
            dp[7] = (tmp[2] + tmp[6]) % 1000000007;
            dp[8] = (tmp[1] + tmp[3]) % 1000000007;
            dp[9] = (tmp[2] + tmp[4]) % 1000000007;
        }
        int ret = 0;
        for(int i = 0;i < 10;i++){
            ret = (ret + dp[i]) % 1000000007;
        }
        if(N == 1)
            return 10;
        else
            return ret-1;
    }
};
