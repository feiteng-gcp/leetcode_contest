class Solution {
public:
    int knightDialer(int N) {

        unordered_map<int, vector<int>> next;
        next[1] = {6, 8};
        next[2] = {7, 9};
        next[3] = {4, 8};
        next[4] = {3, 9, 0};
        next[5] = {};
        next[6] = {1, 7, 0};
        next[7] = {2, 6};
        next[8] = {1, 3};
        next[9] = {2, 4};
        next[0] = {4, 6};
        int res = 0;
        vector<vector<int>> dp(N+1, vector<int>(10, 0));
        if(N == 1) return 10;

        for(int j = 0; j < 10; j++){
            dp[1][j] = 1;
        }
        for(int i = 2; i <= N; i++){
            for(int j = 0; j < 10; j++){
                for(int x: next[j]){
                    dp[i][j] = (dp[i][j] + dp[i-1][x])%1000000007;
                }
                //cout<<j<<" "<<dp[i][j]<<endl;
                if(i == N) res = (res + dp[i][j])%1000000007;
            }
        }
        return res;
    }
    
    
};
