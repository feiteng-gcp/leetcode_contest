template <typename T>
void print_seq(T&& v) {
    cout << "[ ";
    for (auto& x : v) {
        cout << x << " ";
    }
    cout << "]\n";
}

class Solution {
public:
    int knightDialer(int N) {
        unordered_map<int, vector<int>> board = {
            {0, {4, 6}},
            {1, {6, 8}},
            {2, {7, 9}},
            {3, {4, 8}},
            {4, {0, 3, 9}},
            {5, {}},
            {6, {0, 1, 7}},
            {7, {2, 6}},
            {8, {1, 3}},
            {9, {2, 4}}
        };
        int mod = 1000000007;
        
        vector<vector<int>> dp(N + 1, vector<int>(10));
        for (int i = 0; i < 10; ++i) {
            dp[1][i] = 1;
        }
        for (int n = 2; n <= N; ++n) {
            for (int i = 0; i < 10; ++i) {
                dp[n][i] = 0;
                for (int j : board[i]) {
                    dp[n][i] = (dp[n][i] + dp[n-1][j]) % mod;
                }
            }
        }
        
        //print_seq(dp[2]);
        //print_seq(dp[3]);
        
        int sum = 0;
        for (int i = 0; i < 10; ++i) {
            sum = (sum + dp[N][i]) % mod;
        }
        
        return sum;
    }
};
