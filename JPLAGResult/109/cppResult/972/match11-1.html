typedef long long int64;
const int MOD = 1e9 + 7;
class Solution {
public:
    int knightDialer(int n) {
        vector<int64> d(10), e(10);

        for (int i = 0; i < 10; ++i) d[i] = 1;
        for (int i = 1; i < n; ++i) {
            fill(e.begin(), e.end(), 0);
            e[1] = (d[6] + d[8]) % MOD;
            e[2] = (d[7] + d[9]) % MOD;
            e[3] = (d[4] + d[8]) % MOD;
            e[4] = (d[3] + d[9] + d[0]) % MOD;
            e[5] = 0;
            e[6] = (d[1] + d[7] + d[0]) % MOD;
            e[7] = (d[2] + d[6]) % MOD;
            e[8] = (d[1] + d[3]) % MOD;
            e[9] = (d[2] + d[4]) % MOD;
            e[0] = (d[4] + d[6]) % MOD;
            swap(d, e);
        }
        
        int ret = 0;
        for (int i = 0; i < 10; ++i) ret = (ret + d[i]) % MOD;
        return ret;
    }
};
