class Solution {
public:

    int knightDialer(int N) {
        vector<vector<int>> nxts= {
            {4,6},
            {6, 8},
            {7,9},
            {4,8},
            {3,9,0},
            {},
            {1,7,0},
            {2,6},
            {1,3},
            {2, 4}
        };
        int P=1000000007;
        vector<vector<long long>> f(10, vector<long long>(5010, 0));
        for (int i=1;i<=N;++i) {
            for (int j=0;j<=9;++j) {
                if (i==1)f[j][i]=1;
                else {
                    for (int k: nxts[j]) {
                        f[j][i]+=f[k][i-1];
                    }

                    f[j][i]%=P;
                }
            }
        }
        long long ans=0;
        for (int i=0;i<=9;++i) {
            ans+=f[i][N];
        }
        ans%=P;
        return (int)ans;


    }
};
