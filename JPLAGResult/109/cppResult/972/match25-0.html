class Solution {
    const int P=1000000007;
public:
    int knightDialer(int N) {
        array<int,10> dp,dp2;
        fill(dp.begin(),dp.end(),1);
        for(int i=1;i<N;i++){
            dp2[0]=(dp[4]+dp[6])%P;
            dp2[1]=(dp[6]+dp[8])%P;
            dp2[2]=(dp[7]+dp[9])%P;
            dp2[3]=(dp[4]+dp[8])%P;
            dp2[4]=((dp[3]+dp[9])%P+dp[0])%P;
            dp2[5]=0;
            dp2[6]=((dp[1]+dp[7])%P+dp[0])%P;
            dp2[7]=(dp[2]+dp[6])%P;
            dp2[8]=(dp[1]+dp[3])%P;
            dp2[9]=(dp[2]+dp[4])%P;
            dp=dp2;
        }
        int ans=0;
        for(int i=0;i<10;i++)ans=(ans+dp[i])%P;
        return ans;
    }
};
