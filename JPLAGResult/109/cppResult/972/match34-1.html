class Solution {
public:
    int knightDialer(int N) {
        long mod = 1000000007;
        vector<long> f(10, 1);
        vector<long> g(10, 1);
        for (int i = 1; i < N; i++) {
            g[0] = f[4] + f[6];

            g[1] = f[6] + f[8];
            g[2] = f[7] + f[9];
            g[3] = f[4] + f[8];
            g[4] = f[0] + f[3] + f[9];
            g[5] = 0;
            g[6] = f[0] + f[1] + f[7];
            g[7] = f[2] + f[6];
            g[8] = f[1] + f[3];
            g[9] = f[2] + f[4];
            for (int i = 0; i < 10; i++) {
                f[i] = g[i] % mod;
            }
        }

        long res = 0;
        for (int i = 0; i < 10; i++) {
            res += f[i];
            res %= mod;
        }
        return res;
    }
};
