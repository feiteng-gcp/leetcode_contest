class Solution {
public:
    int knightDialer(int N) {
        int mod = 1e9+7;
        vector<vector<int>> dp(N, vector<int>(10, 0));

        for(int i = 0; i <= 9; i++) 
            dp[0][i] = 1;
        for(int n = 1; n < N; n++) {
            dp[n][1] = (dp[n-1][6]+dp[n-1][8])%mod;
            dp[n][2] = (dp[n-1][7]+dp[n-1][9])%mod;
            dp[n][3] = (dp[n-1][4]+dp[n-1][8])%mod;
            dp[n][4] = ((dp[n-1][3]+dp[n-1][9])%mod + dp[n-1][0])%mod;
            dp[n][5] = 0;
            dp[n][6] = ((dp[n-1][1]+dp[n-1][0])%mod + dp[n-1][7])%mod;
            dp[n][7] = (dp[n-1][2]+dp[n-1][6])%mod;
            dp[n][8] = (dp[n-1][1]+dp[n-1][3])%mod;
            dp[n][9] = (dp[n-1][2]+dp[n-1][4])%mod;
            dp[n][0] = (dp[n-1][4]+dp[n-1][6])%mod;
        }
        int res = 0;
        for(int i = 0; i <= 9; i++) {
            res = (res+dp[N-1][i])%mod;
        }
        return res;
    }
};
