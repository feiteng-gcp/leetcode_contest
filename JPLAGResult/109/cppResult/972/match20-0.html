class Solution {
public:
    int knightDialer(int N) {
        vector<vector<int>> pairs{{4, 6}, {6, 8}, {7, 9}, {4, 8}, {0, 3, 9}, {}, {0, 1, 7}, {2, 6}, {1, 3}, {2, 4}};
        
        vector<vector<long long>> dp(N+1, vector<long long>(10, 0));
        
        long long inf = 1e9 + 7;
        
        for(int i = 0; i < 10; ++i) dp[1][i] = 1;
        
        for(int i = 2; i <= N; ++i){
            for(int j = 0; j < 10; ++j){
                for(auto p : pairs[j]){
                    dp[i][j] += dp[i-1][p];
                    dp[i][j] %= inf;
                }
            }
        }
        
        long long ans = 0;
        for(int i = 0; i < 10; ++i){
            ans += dp[N][i];
            ans %= inf;
        }
        
        return ans;
    }
};
