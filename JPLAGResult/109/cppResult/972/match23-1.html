vector<vector<int>> graph {
    {4, 6}, {6, 8}, {7, 9}, {4, 8}, {0, 3, 9}, {}, {0, 1, 7}, {2, 6}, {1, 3}, {2, 4}
};

class Solution {
public:
    int knightDialer(int N) {
        int buffer[5000][10] = {0};
        for (int i = 0; i < 10; i++) {
            buffer[0][i] = 1;
            buffer[1][i] = graph[i].size();
        }

        for (int i = 2; i < N; i++) {
            for (int j = 0; j < 10; j++) {
                for (int k = 0; k < graph[j].size(); k++) {

                    buffer[i][j] += buffer[i-1][graph[j][k]];
                    buffer[i][j] %= 1000000007;
                }
                // printf("%d ", buffer[i][j]);
            }
            // printf("\n");
        }
        int sum = 0;
        for (int i = 0; i < 10; i++) {
            sum += buffer[N-1][i];
            sum %= 1000000007;
        }
        return sum;
    }
};
