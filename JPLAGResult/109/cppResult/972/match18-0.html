class Solution {
public:
    int mod=1e9+7;
    int knightDialer(int N) {
        long long ans[5005][10];
        memset(ans,0,sizeof(ans));
        for(int i=0;i<10;i++)
            ans[1][i]++;
        for(int i=2;i<=N;i++){
            ans[i][1]=(ans[i-1][8]+ans[i-1][6])%mod;
            ans[i][2]=(ans[i-1][7]+ans[i-1][9])%mod;
            ans[i][3]=(ans[i-1][4]+ans[i-1][8])%mod;
            ans[i][4]=(ans[i-1][3]+ans[i-1][0]+ans[i-1][9])%mod;
            ans[i][6]=(ans[i-1][1]+ans[i-1][0]+ans[i-1][7])%mod;
            ans[i][7]=(ans[i-1][2]+ans[i-1][6])%mod;
            ans[i][8]=(ans[i-1][1]+ans[i-1][3])%mod;
            ans[i][9]=(ans[i-1][2]+ans[i-1][4])%mod;
            ans[i][0]=(ans[i-1][4]+ans[i-1][6])%mod;
        }
        long long ret=0;
        for(int i=0;i<10;i++)
            ret+=ans[N][i];
        return ret%mod;
    }
};
