class RecentCounter {
public:
    RecentCounter(): m_pings() {
        
    }
    
    int ping(int t) {
        m_pings.push_back(t);
        while (!m_pings.empty() && m_pings.front() < t - 3000) {
            m_pings.pop_front();
        }
        return m_pings.size();
    }
    
private:
    list<int> m_pings;
};

/**
 * Your RecentCounter object will be instantiated and called as such:
 * RecentCounter* obj = new RecentCounter();
 * int param_1 = obj->ping(t);
