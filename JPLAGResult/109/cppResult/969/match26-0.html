class RecentCounter {
public:
    list<int> ls;
    RecentCounter() {
        
    }
    
    int ping(int t) {
        ls.push_back(t);
        while (ls.front() < t - 3000) ls.pop_front();
        return ls.size();
    }
};

/**
 * Your RecentCounter object will be instantiated and called as such:
 * RecentCounter* obj = new RecentCounter();
 * int param_1 = obj->ping(t);
 */
