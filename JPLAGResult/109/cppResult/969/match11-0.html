class RecentCounter {
    queue<int> qq;
public:
    RecentCounter() {
        
    }
    
    int ping(int t) {
        while(!qq.empty()&&qq.front()+3000<t) qq.pop();
        qq.push(t);
        return qq.size();
    }
};

/**
 * Your RecentCounter object will be instantiated and called as such:
 * RecentCounter* obj = new RecentCounter();
 * int param_1 = obj->ping(t);
 */
