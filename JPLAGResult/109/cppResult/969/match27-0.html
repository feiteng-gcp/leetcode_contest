class RecentCounter {
public:
    deque<int> Q;
    RecentCounter() {
        Q.clear();
    }
    
    int ping(int t) {
        while(!Q.empty() && Q.front()<t-3000)Q.pop_front();
        Q.push_back(t);
        return Q.size();
    }
};

/**
 * Your RecentCounter object will be instantiated and called as such:
 * RecentCounter* obj = new RecentCounter();
 * int param_1 = obj->ping(t);
 */
