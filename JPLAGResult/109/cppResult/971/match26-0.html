class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        int m = A.size(), n = A[ 0 ].size();
        vector< vector< int > > work( A );
        bool found = false;
        queue< pair< int, int > > island;
        for( int i = 0; i < m && !found; i++ ) {
            for( int j = 0; j < n; j++ ) {
                if( A[ i ][ j ] ) {
                    getIsland( work, i, j, island );
                    found = true;
                    break;
                }
            }
        }    
        vector< int > dir{ 0, 1, 0, -1, 0 };
        for( int step = 1; ;step++ ) {            
            int size = island.size();
            for( int i = 0; i < size; i++ ) {                
                int x = island.front().first, y = island.front().second;
                island.pop();
                for( int k = 0; k < 4; k++ ) {
                    int nx = x + dir[ k ], ny = y + dir[ k + 1 ];
                    if( nx >= 0 && nx != m && ny >= 0 && ny != n ) {
                        if( work[ nx ][ ny ] == 1 ) return step - 1;
                        if( work[ nx ][ ny ] == 0 ) {
                            work[ nx ][ ny ] = -step;
                            island.emplace( nx, ny );
                        }
                    }
                }
            }
        }
        return -1;
    }
    void getIsland( vector< vector< int > > &lands, int x, int y, queue< pair< int, int > > &island ) {
        if( x < 0 || x == lands.size() || y < 0 || y == lands[ 0 ].size() ) {
            return;
        }
        if( lands[ x ][ y ] != 1 ) {
            return;
        }
        lands[ x ][ y ] = -1;
        island.emplace( x, y );
        getIsland( lands, x - 1, y, island );
        getIsland( lands, x + 1, y, island );
        getIsland( lands, x, y - 1, island );
        getIsland( lands, x, y + 1, island );
        
    }
};
