class Solution {
public:
    vector<pair<int, int>> island1, island2;

    int shortestBridge(vector<vector<int>>& A) {
        // find the first island

        int flag = 0;
        for(int i = 0; i < A.size(); i++) {
            for(int j = 0; j < A[0].size(); j++) {
                if(A[i][j] == 1) {
                    dfs(A, i, j, 1);
                    flag = 1;
                    break;
                }
            }
            if(flag)
                break;
        }

        flag = 0;
        for(int i = 0; i < A.size(); i++) {
            for(int j = 0; j < A[0].size(); j++) {
                if(A[i][j] == 1) {
                    dfs(A, i, j, 2);
                    flag = 1;
                    break;
                }
            }
            if(flag)
                break;
        }
        
        // compare
        int dist = 10000;
        for(auto p1: island1) {
            for(auto p2: island2) {
                dist = min(dist, abs(p1.first-p2.first)+abs(p1.second-p2.second)-1);
            }
        }
        return dist;
    }
    
    void dfs(vector<vector<int>>& A, int x, int y, int id) {
        if(id == 1)
            island1.push_back({x, y});
        else
            island2.push_back({x, y});
        A[x][y] = 0;

        int dirs[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        for(int i = 0; i < 4; i++) {
            int nx = x+dirs[i][0], ny = y+dirs[i][1];
            if(nx >= 0 && nx < A.size() && ny >= 0 && ny < A[0].size() && A[nx][ny] != 0) {
                dfs(A, nx, ny, id);
            }
        }
    }
};
