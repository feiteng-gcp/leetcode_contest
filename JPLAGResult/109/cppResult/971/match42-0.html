class Solution {
public:
    
    int dir[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    int vis[105][105];
    vector<pair<int, int>> res[2];
    
    bool check(int x, int y, int n) {
        if (x < 0 || y < 0 || x >= n || y >= n) {
            return false;
        }
        return true;
    }
    
    bool is(vector<vector<int>>& A, int n, int x, int y) {
        if (x == 0 || y == 0 || x == n-1 || y == n-1) {
            return true;
        }
        
        int flag = 0;
        for (int i = 0; i < 4; i++) {
            int xx = x + dir[i][0];
            int yy = y + dir[i][1];
            
            if (A[xx][yy] == 0) flag = 1;
        }
        
        if (flag == 1) return true;
        else return false;
    }
    
    void dfs(vector<vector<int>>& A, int n, int x, int y, int flag) {
        
        if (is(A, n, x, y)) {
            res[flag].push_back(make_pair(x, y));
        }
        
        for (int i = 0; i < 4; i++) {
            int xx = x + dir[i][0];
            int yy = y + dir[i][1];
            if (check(xx, yy, n) && vis[xx][yy] == 0 && A[xx][yy] == 1) {
                vis[xx][yy] = 1;
                dfs(A, n, xx, yy, flag);
            }
        }
        
        
    }
    
    int shortestBridge(vector<vector<int>>& A) {
        int n = A.size();
        
        memset(vis, 0, sizeof vis);
        int flag = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (vis[i][j] == 0 && A[i][j] == 1) {
                    vis[i][j] = 1;
                    dfs(A, n, i, j, flag);
                    flag = 1;
                }
            }
        }
        
        int ans = 100000;
        
        for (int i = 0; i < res[0].size(); i++) {
            for (int j = 0; j < res[1].size(); j++) {
                
                int x = res[0][i].first;
                int y = res[0][i].second;
                
                int xx = res[1][j].first;
                int yy = res[1][j].second;
                
                int t = abs(xx - x) + abs(yy - y) - 1;

                if (ans > t) {
                    ans = t;
                }
            }
        }
        
        return ans;
    }
};
