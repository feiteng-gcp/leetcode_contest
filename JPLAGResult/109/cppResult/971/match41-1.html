class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {

        int x;
        int y;
        bool found = false;
        for (int i = 0; !found && i < A.size(); ++i) {
            for (int j = 0; j < A[i].size(); ++j) {
                if (A[i][j] == 1) {
                    found = true;
                    x = i;
                    y = j;
                    break;
                }
            }
        }
        int dx[] = {0, 1, 0, -1};
        int dy[] = {1, 0, -1, 0};
        unordered_set<int> visited;
        queue<pair<int, int>> q1;
        queue<pair<pair<int, int>, int>> q2;
        visited.insert(encode(x, y));
        q1.push(make_pair(x, y));
        q2.push(make_pair(make_pair(x, y), 0));
        A[x][y] = 0;
        while (!q1.empty()) {
            int x = q1.front().first;

            int y = q1.front().second;
            q1.pop();
            for (int i = 0; i < 4; ++i) {
                int xx = x + dx[i];
                int yy = y + dy[i];
                if (xx < 0 || xx >= A.size() || yy < 0 || yy >= A[0].size()) {
                    continue;
                }
                if (visited.count(encode(xx, yy)) > 0 || A[xx][yy] == 0) {
                    continue;
                }
                visited.insert(encode(xx, yy));
                q1.push(make_pair(xx, yy));
                q2.push(make_pair(make_pair(xx, yy), 0));
                A[xx][yy] = 0;
            }
        }
        while (!q2.empty()) {
            int x = q2.front().first.first;
            int y = q2.front().first.second;

            int d = q2.front().second;
            q2.pop();
            if (A[x][y] == 1) {
                return d - 1;
            }
            for (int i = 0; i < 4; ++i) {
                int xx = x + dx[i];
                int yy = y + dy[i];
                if (xx < 0 || xx >= A.size() || yy < 0 || yy >= A[0].size()) {
                    continue;
                }
                if (visited.count(encode(xx, yy)) > 0) {
                    continue;
                }
                visited.insert(encode(xx, yy));
                q2.push(make_pair(make_pair(xx, yy), d + 1));
            }
        }
        return 0;
    }
private:
    int encode(int x, int y) {
        return x * 1000 + y;
    }
};
