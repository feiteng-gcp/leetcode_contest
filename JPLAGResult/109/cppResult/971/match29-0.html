class Solution {
private:
    class Item {
    public:
        Item(int _i, int _j, int _level) {
            i = _i;
            j = _j;
            level = _level;
        }
        
        int i;
        int j;
        int level;
    };
    
public:
    int shortestBridge(vector<vector<int>>& A) {
        queue<pair<int, int>> q;
        bool found = false;
        for(int i = 0; i<A.size(); ++i) {
            if(found) break;
            for(int j = 0; j<A.size(); ++j) {
                if(A[i][j]) {
                    A[i][j] = 2;
                    found = true;
                    q.push(make_pair(i, j));
                    break;
                }
            }
        }
        
        queue<Item> q2;
        // Color one component as 2
        while(q.size()) {
            auto it = q.front();
            int i = it.first;
            int j = it.second;
            q.pop();
            if(i>0 && A[i-1][j] == 0 || i<A.size()-1 && A[i+1][j] == 0 || j<A.size()-1 && A[i][j+1] == 0 || j>0 && A[i][j-1] == 0) 
               q2.push(Item(i, j, 0));
            
            
            if(i>0 && A[i-1][j] == 1) {
                A[i-1][j] = 2;
                q.push(make_pair(i-1,j));
            }
            if(i<A.size()-1 && A[i+1][j] == 1) {
                A[i+1][j] = 2;
                q.push(make_pair(i+1,j));
            }
            if(j<A.size()-1 && A[i][j+1] == 1) {
                A[i][j+1] = 2;
                q.push(make_pair(i,j+1));
            }
            if(j>0 && A[i][j-1] == 1) {
                A[i][j-1] = 2;
                q.push(make_pair(i,j-1));
            }
        }
        
        /*for(int i = 0; i<A.size(); ++i) {
            for(int j = 0; j<A.size(); ++j) {
                cout<<i<<" "<<j<<" "<<A[i][j]<<endl;
            }
        }*/
        
        while(q2.size()) {
            Item item = q2.front();
            q2.pop();
            int i = item.i;
            int j = item.j;
            int level = item.level;
            
            if(i>0 && A[i-1][j] == 1 || i<A.size()-1 && A[i+1][j] == 1 || j<A.size()-1 && A[i][j+1] == 1 || j>0 && A[i][j-1] == 1) return level; 
                
            if(i>0 && A[i-1][j] == 0) {
                A[i-1][j] = 2;
                q2.push(Item(i-1, j, level+1));
            }
            if(i<A.size()-1 && A[i+1][j] == 0) {
                A[i+1][j] = 2;
                q2.push(Item(i+1, j, level+1));
            }
            if(j<A.size()-1 && A[i][j+1] == 0) {
                A[i][j+1] = 2;
                q2.push(Item(i, j+1, level+1));
            }
            if(j>0 && A[i][j-1] == 0) {
                A[i][j-1] = 2;
                q2.push(Item(i, j-1, level+1));
            }
        }
        
        return 0;
    }
};
