int d[4][2] = {
    {1,0},
    {-1,0},
    {0,1},
    {0,-1},
};
class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        int n = A.size();
        int m = A.back().size();
        
        vector<vector<bool>> v(n, vector<bool>(m, false));
        deque<tuple<int,int,int>> q;
        
        function<void (int, int)> update = [&](int a, int b) {
            for (int i = 0; i < 4; ++i) {
                int x = a + d[i][0];
                int y = b + d[i][1];
                
                if (x < 0 || y < 0) continue;
                if (x >= n || y >= m) continue;
                
                if (A[x][y] && !v[x][y]) {
                    v[x][y] = 1;
                    q.emplace_back(x, y, 0);
                    update(x, y);
                }
            }
        };
        
        
        for (int i = 0; i < n; ++i) {
            if (!q.empty()) break;
            for (int j = 0; j < m; ++j) {
                if (A[i][j]) {
                    q.emplace_back(i, j, 0);
                    v[i][j] = true;
                    update(i, j);
                    break;
                }
            }
        }
        
        
        
        
        while (!q.empty()) {
            int a, b, dep;
            tie(a,b, dep) = q.front();
            q.pop_front();
            
            for (int i = 0; i < 4; ++i) {
                int x = a + d[i][0];
                int y = b + d[i][1];
                
                if (x < 0 || y < 0) continue;
                if (x >= n || y >= m) continue;
                
                if (!v[x][y]) {
                    if (A[x][y] == 1) {
                        return dep;
                    }
                    v[x][y] = 1;
                    q.emplace_back(x, y, dep + 1);
                }
            } 
        }
     
        return -1;
    }
    
};
