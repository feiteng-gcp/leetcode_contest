class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        int s = A.size();
        set<pair<int, int>> island[2];

        for (int i=0; i<s; i++) {
            for (int j=0; j<s; j++) {
                if (A[i][j]==1) {
                    dfs(A, s, i, j, island, 0);
                    i=s;
                    break;
                }
                
            }

        }
        for (int i=0; i<s; i++) {
            for (int j=0; j<s; j++) {
                if (A[i][j]==1&&island[0].find(pair<int, int>(i, j))==island[0].end()) {
                    dfs(A, s, i, j, island, 1);
                    i=s;
                    break;
                }
            }
        }
        int m = 2 * s;
        for (auto p1:island[0]) {
            for (auto p2:island[1]) {
                int t = abs(p1.first-p2.first)-1 + abs(p1.second-p2.second);
                m = min(m, t);
            }
        }
        return m;
    }
    void dfs(vector<vector<int>>& A, int s, int row, int col, set<pair<int, int>>* island, int is) {
        if (A[row][col]==0 || island[is].find(pair<int, int>(row, col)) != island[is].end()) {
            return;
        }
        island[is].insert(pair<int, int>(row, col));

        if (row>0) {
            dfs(A, s, row-1, col, island, is);
        }
        if (col>0) {
            dfs(A, s, row, col-1, island, is);
        }
        if (row<s-1) {
            dfs(A, s, row+1, col, island, is);
        }
        if (col<s-1) {
            dfs(A, s, row, col+1, island, is);
        }
        
    }
};
