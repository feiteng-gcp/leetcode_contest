typedef pair<int, int> ii;

const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

class Solution {
public:
    int shortestBridge(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        int sx = -1, sy = -1;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (a[i][j] == 1) {
                    sx = i;
                    sy = j;
                    break;
                }
            }
            if (sx != -1) break;
        }
        vector<vector<int>> visit(n, vector<int>(m, -1));
        vector<vector<bool>> flag(n, vector<bool>(m));
        queue<ii> Q, Q2;
        Q.push({sx, sy});
        flag[sx][sy] = true;
        while (!Q.empty()) {
            int x = Q.front().first, y = Q.front().second;
            // cout << x << " " << y << endl;
            visit[x][y] = 0;
            a[x][y] = 0;
            Q2.push({x, y});
            Q.pop();

            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0];
                int ny = y + d[k][1];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
                if (a[nx][ny] == 0) continue;
                if (flag[nx][ny]) continue;
                Q.push({nx, ny});
                flag[nx][ny] = true;
            }
        }
         // cout << sx << " " << sy << endl;
        // for (int i = 0; i < n; ++i) {
        //     for (int j = 0; j < m; ++j) {
        //         cout << visit[i][j] << " ";
        //     }
        //     cout << endl;
        // }
        while (!Q2.empty()) {
            int x = Q2.front().first, y = Q2.front().second;
            if (a[x][y] == 1) return visit[x][y] - 1;
            Q2.pop();

            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0];
                int ny = y + d[k][1];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
                if (visit[nx][ny] >= 0) continue;
                Q2.push({nx, ny});
                visit[nx][ny] = visit[x][y] + 1;
                // cout << nx << " " << ny << " " << visit[nx][ny] << endl;
            }
        }
        return -1;
    }
};
