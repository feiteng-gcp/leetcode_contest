class Solution {
public:
    set<vector<int>> change(vector<vector<int>>& A, int r0, int c0, int value) {
        int n = A.size();
        int m = A[0].size();
        vector<pair<int, int>> dirs = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
        vector<pair<int, int>> dq = {{r0, c0}};
        set<vector<int>> seen;
        seen.insert({r0, c0});
        int dr, dc, rr, cc;
        while(!dq.empty()) {
            vector<pair<int, int>> new_dq;            
            for(auto& p: dq) {
                for(auto& dir: dirs) {
                    tie(dr, dc) = dir;
                    rr = p.first + dr;
                    cc = p.second + dc;
                    if (rr < 0 || rr >= n || cc < 0 || cc >= m) continue;
                    if (A[rr][cc] == 1 && seen.count({rr, cc}) == 0) {
                        seen.insert({rr, cc});
                        new_dq.push_back({rr, cc});
                    }
                }                
            }
            dq.swap(new_dq);
        }
        set<vector<int>> ans;
        for(auto& p: seen) {
            A[p[0]][p[1]] = value;
            for(auto& dir : dirs) {
                tie(dr, dc) = dir;
                rr = p[0] + dr;
                cc = p[1] + dc;
                if (rr < 0 || rr >= n || cc < 0 || cc >= m) continue;
                if (A[rr][cc] == 0) {
                    ans.insert(p);
                    break;
                }    
            }        
        }
        return ans;
    }
    int shortestBridge(vector<vector<int>>& A) {
        int n = A.size();

        int m = A[0].size();
        vector<set<vector<int>>> grp;
        int value;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if (A[i][j] == 1) {
                    grp.push_back(change(A, i, j, ++value));
                    break;

                }
            }
        }
        int ans = INT_MAX;
        for(auto& p1: grp[0]) {
            for(auto& p2: grp[1]) {
                ans = min(ans, abs(p1[0] - p2[0]) + abs(p1[1] - p2[1]) - 1);
            }
        }
        return ans;
    }
};
