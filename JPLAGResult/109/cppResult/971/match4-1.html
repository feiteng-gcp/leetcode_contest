class Solution {
private:
    void dfs(vector<vector<int>>& A, int i, int j, int l1, int l2, vector<pair<int, int>>& land) {
        if (i >= 0 && j >= 0 && i < l1 && j < l2 && A[i][j] == 1) {
            pair<int, int> p(i, j);
            land.push_back(p);
            A[i][j] = 0;
            dfs(A, i+1, j, l1, l2, land);
            dfs(A, i-1, j, l1, l2,land);
            dfs(A, i, j+1, l1, l2,land);
            dfs(A, i, j-1, l1, l2,land);
        }
    }
public:
    int shortestBridge(vector<vector<int>>& A) {
        if (A.empty() || A[0].empty()) return 0;
        int l1 = A.size();
        int l2 = A[0].size();
        bool found = false;
        vector<pair<int, int>> land1;
        vector<pair<int, int>> land2;

        for (int i = 0; i < l1; i++) {
            for (int j = 0; j < l2; j++) {
                if (A[i][j] == 1) {
                    found = true;
                    dfs(A, i, j, l1, l2, land1);
                    break;
                }
            }
            if (found) break;
        }
        
        found = false;
        for (int i = 0; i < l1; i++) {
            for (int j = 0; j < l2; j++) {
                if (A[i][j] == 1) {
                    found = true;
                    dfs(A, i, j, l1, l2, land2);
                    break;
                }
            }
            if (found) break;
        }
        int res = INT_MAX;
        for (auto p1 : land1) {
            for (auto p2 : land2) {
               res = min(res, abs(p1.first-p2.first) + abs(p1.second - p2.second)); 
            }
        }
        return res-1;
    }
};
