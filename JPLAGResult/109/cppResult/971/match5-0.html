class Solution {
public:
    void findIsland(vector<vector<int>>& A, int i, int j, vector<pair<int, int>>& island)
    {
        island.push_back(pair<int, int>(i, j));
        A[i][j] = 0;
        if (i>0&&A[i-1][j]==1)
            findIsland(A, i-1, j, island);
        if (i<A.size()-1&&A[i+1][j]==1)
            findIsland(A, i+1, j, island);
        if (j>0&&A[i][j-1]==1)
            findIsland(A, i, j-1, island);
        if (j<A[0].size()-1&&A[i][j+1]==1)
            findIsland(A, i, j+1, island);
    }
    int shortestBridge(vector<vector<int>>& A) {
        vector<pair<int, int>> fst, sec;
        int i, j;
        bool flag = false;
        for (i=0;i<A.size();i++)
        {
            for (j=0;j<A[0].size();j++)
            {
                if (A[i][j]==1)
                {
                    findIsland(A, i, j, fst);
                    flag = true;
                    break;
                }
            }
            if (flag)
                break;
        }
        flag = false;
        for (i=0;i<A.size();i++)
        {
            for (j=0;j<A[0].size();j++)
            {
                if (A[i][j]==1)
                {
                    findIsland(A, i, j, sec);
                    flag = true;
                    break;
                }
            }
            if (flag)
                break;
        }
        int res=INT_MAX;
        for (auto p1: fst)
        {
            for (auto p2 : sec)
            {
                res = min(res, abs(p1.first-p2.first)+abs(p1.second-p2.second));
            }
        }
        return res-1;
    }
};
