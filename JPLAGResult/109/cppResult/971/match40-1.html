class Solution {
    void FindStart(vector<vector<int>>& A, int& startI, int& startJ) {

        startI = -1;
        startJ = -1;
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[0].size(); j++) {
                if (A[i][j] == 1) {
                    startI = i;
                    startJ = j;
                    break;
                }
            }
            if (startI != -1) {
                break;
            }
        }
    }
    bool isSafe(const vector<vector<int>>& A, const int i, const int j) {
        return i >= 0 && i < A.size() && j >= 0 && j < A[0].size();
    }
    void Color(vector<vector<int>>& A, const int startI, const int startJ, const int colorTo) {
        A[startI][startJ] = colorTo;

        if (isSafe(A, startI - 1, startJ) && A[startI - 1][startJ] == 1) {
            Color(A, startI - 1, startJ, colorTo);
        }
        if (isSafe(A, startI + 1, startJ) && A[startI + 1][startJ] == 1) {
            Color(A, startI + 1, startJ, colorTo);
        }
        if (isSafe(A, startI, startJ - 1) && A[startI][startJ - 1] == 1) {
            Color(A, startI, startJ - 1, colorTo);
        }
        if (isSafe(A, startI, startJ + 1) && A[startI][startJ + 1] == 1) {
            Color(A, startI, startJ + 1, colorTo);
        }
    }
    bool Color2(vector<vector<int>>& A, const int near) {
        bool found1 = false;

        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[0].size(); j++) {
                if (A[i][j] == near) {
                    if (isSafe(A, i - 1, j) && (A[i - 1][j] == 0 || A[i - 1][j] == 1)) {
                        found1 |= (A[i - 1][j] == 1);
                        A[i - 1][j] = near + 1;
                    }
                    if (isSafe(A, i + 1, j) && (A[i + 1][j] == 0 || A[i + 1][j] == 1)) {
                        found1 |= (A[i + 1][j] == 1);
                        A[i + 1][j] = near + 1;
                    }
                    if (isSafe(A, i, j - 1) && (A[i][j - 1] == 0 || A[i][j - 1] == 1)) {
                        found1 |= (A[i][j - 1] == 1);
                        A[i][j - 1] = near + 1;
                    }
                    if (isSafe(A, i, j + 1) && (A[i][j + 1] == 0 || A[i][j + 1] == 1)) {
                        found1 |= (A[i][j + 1] == 1);
                        A[i][j + 1] = near + 1;
                    }
                }
            }
        }
        return found1;
    }
public:
    int shortestBridge(vector<vector<int>>& A) {
        int startI;
        int startJ;
        int count = 2;

        FindStart(A, startI, startJ);
        Color(A, startI, startJ, 2);
        
        while (!Color2(A, count)) {
            count++;
        }
        return count - 2;
    }
};
