class Solution {
private:
    vector<vector<int>> color;
    int n;
    int shift_x[4] = {0, 0, 1, -1};
    int shift_y[4] = {1, -1, 0, 0};
    vector<vector<int>> dist;
    void dfs(int x, int y, int cur_color, vector<vector<int>>& A) {
        color[x][y] = cur_color;
        for(int i = 0; i < 4; ++i) {
            int x_new = x + shift_x[i];
            int y_new = y + shift_y[i];
            if(x_new < 0 || x_new >= n
              || y_new < 0 || y_new >= n) {
                continue;
            }
            if(color[x_new][y_new] != -1) {
                continue;
            }
            if(A[x_new][y_new] != 1) {
                continue;
            }
            dfs(x_new, y_new, cur_color, A);
        }
    }
    int bfs(int x, int y) {
        queue<pair<int, int>> q;
        int pos = x * 1000 + y;
        q.push(make_pair(pos, 0));
        vector<vector<int>> visit(n, vector<int>(n, -1));
        visit[x][y] = 1;
        while(!q.empty()) {
            auto top = q.front();
            q.pop();
            y = top.first % 1000;
            x = top.first / 1000;
            
            if(color[x][y] == 1) {
                return top.second - 1;
            }
            for(int i = 0; i < 4; ++i) {
                int x_new = x + shift_x[i];
                int y_new = y + shift_y[i];
                if(x_new < 0 || x_new >= n
                  || y_new < 0 || y_new >= n) {
                    continue;
                }
                if(color[x_new][y_new] == 1) {
                    return top.second;
                }
                if(color[x_new][y_new] == 0) {
                    continue;
                }
                if(visit[x_new][y_new] == 1) {
                    continue;
                }
                pos = x_new * 1000 + y_new;
                q.push(make_pair(pos, top.second + 1));
                visit[x_new][y_new] = 1;
            }
        }
        return -1;
    }
public:
    int shortestBridge(vector<vector<int>>& A) {
        n = A.size();
        color = vector<vector<int>>(n, vector<int>(n, -1));
        dist = vector<vector<int>>(n, vector<int>(n, -1));
        int cur_color = 0;
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < n; ++j) {
                if(A[i][j] != 1) {
                    continue;
                }
                if(color[i][j] != -1) {
                    continue;
                }
                dfs(i, j, cur_color, A);
                ++cur_color;
            }
        }
        
        int ans = INT_MAX;
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < n; ++j) {
                if(color[i][j] == 0) {
                    int res = bfs(i, j);
                    if(res != -1) {
                        ans = min(ans, res);
                    }
                }
            }
        }
        return ans;
    }
};
