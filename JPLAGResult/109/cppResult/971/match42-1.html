class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        int R = A.size(), C = A[0].size();
        vector<int> island1, island2;
        vector<vector<int>> visited(R, vector<int>(C, 0));

        vector<vector<int>> dir = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        for(int i = 0; i < R; i++)
            for(int j = 0; j < C; j++)
                if(A[i][j] == 1 && visited[i][j] == 0)
                {
                    visited[i][j] = 1;
                    queue<int> qu;
                    qu.push(i * C + j);
                    vector<int> land;
                    while(!qu.empty())
                    {
                        int x = qu.front() / C;
                        int y = qu.front() % C;
                        land.push_back(qu.front());
                        qu.pop();

                        for(int k = 0; k < 4; k++)
                        {
                            int nx = x + dir[k][0];
                            int ny = y + dir[k][1];
                            if(nx >= 0 && nx < R && ny >= 0 && ny < C && A[nx][ny] && visited[nx][ny] == 0)
                            {
                                visited[nx][ny] = 1;
                                qu.push(nx * C + ny);
                            }
                        }
                    }
                    
                    if(island1.empty())
                        island1 = land;
                    else
                        island2 = land;
                }
        
        int flip = INT_MAX;
        for(int i = 0; i < island1.size(); i++)

            for(int j = 0; j < island2.size(); j++)
            {
                int x1 = island1[i] / C;
                int y1 = island1[i] % C;
                int x2 = island2[j] / C;
                int y2 = island2[j] % C;
                
                flip = min(flip, abs(x1 - x2) + abs(y1 - y2) - 1);
       //         cout<<x1<<", "<<y1<<"; "<<x2<<", "<<y2<<endl;
       //         cout<<"flip = "<<flip<<endl;
            }
        return flip;
    }
};
