class Solution {
public:
	int shortestBridge(vector<vector<int>>& A) {
		int N = A.size();
		vector<vector<int>> island; // k x 2
		vector<bool> visited(N*N, false);

		for (int i = 0; i < N; ++i) {
			for (int j = 0; j < N; ++j) {
				if (A[i][j] == 1) {
					dfs(A, i, j, island, visited);

					for (int k = 0; k < N * 2; ++k) {
						if (!spread(A, island, visited)) {
							return k;
						}
					}
				}
			}
		}

		return -1;
	}
private:
	void dfs(vector<vector<int>>& A, int i, int j, vector<vector<int>>& island, vector<bool>& visited) {
		int N = A.size();
		island.push_back({ i, j });
		visited[pos2id(i, j, N)] = true;
		if (i - 1 >= 0 && A[i - 1][j] == 1 && !visited[pos2id(i - 1, j, N)]) dfs(A, i - 1, j, island, visited);
		if (i + 1  < N && A[i + 1][j] == 1 && !visited[pos2id(i + 1, j, N)]) dfs(A, i + 1, j, island, visited);
		if (j - 1 >= 0 && A[i][j - 1] == 1 && !visited[pos2id(i, j - 1, N)]) dfs(A, i, j - 1, island, visited);
		if (j + 1  < N && A[i][j + 1] == 1 && !visited[pos2id(i, j + 1, N)]) dfs(A, i, j + 1, island, visited);
	}

	bool spread(vector<vector<int>>& A, vector<vector<int>>& island, vector<bool>& visited) {
		int N = A.size();
		vector<vector<int>> new_island = island;
		for (auto& p : island) {
			int i = p[0];
			int j = p[1];
			if (i - 1 >= 0 && !visited[pos2id(i - 1, j, N)]) {
				if (A[i - 1][j] == 0) {
					new_island.push_back({ i - 1, j });
					visited[pos2id(i - 1, j, N)] = true;
				}
				else {
					return false;
				}
			}
			if (i + 1 < N && !visited[pos2id(i + 1, j, N)]) {
				if (A[i + 1][j] == 0) {
					new_island.push_back({ i + 1, j });
					visited[pos2id(i + 1, j, N)] = true;
				}
				else {
					return false;
				}
			}
			if (j - 1 >= 0 && !visited[pos2id(i, j - 1, N)]) {
				if (A[i][j - 1] == 0) {
					new_island.push_back({ i, j - 1 });
					visited[pos2id(i, j - 1, N)] = true;
				}
				else {
					return false;
				}
			}
			if (j + 1 < N && !visited[pos2id(i, j + 1, N)]) {
				if (A[i][j + 1] == 0) {
					new_island.push_back({ i, j + 1 });
					visited[pos2id(i, j + 1, N)] = true;
				}
				else {
					return false;
				}
			}
		}
		island = new_island;
		return true;
	}

	int pos2id(int i, int j, int N) {
		return i * N + j;
	}
};
