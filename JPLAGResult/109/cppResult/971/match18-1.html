class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {

        bool done = false;
        for (int i = 0; i < A.size() && !done; i++) {
            for (int j = 0; j < A[0].size() && !done; j++) {
                if (A[i][j] == 1) {
                    color(A, i, j);
                    done = true;
                }
            }
        }
        vector<pair<int, int>> tmp[2];
        
        // hole
        int dx[4] = {1, -1, 0, 0};

        int dy[4] = {0, 0, 1, -1};
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[0].size(); j++) {
                if (A[i][j] > 0) {

                    bool flag = true;
                    for (int t = 0; t < 4; t++) {
                        int nx = i + dx[t];
                        int ny = j + dy[t];
                        if (nx >=0 && nx < A.size() && ny >= 0 && ny < A.size()) {
                            if (A[nx][ny] != A[i][j]) {
                                flag = false;
                            }
                        }
                    }
                    if (flag == false) {
                        tmp[A[i][j] - 1].push_back(make_pair(i, j));
                    }
                }
            }
        }
        
        // out(A);
        // cout << tmp[0].size() << endl;
        // cout << tmp[1].size() << endl;
        
        // find result
        int res = 1e8;
        
        for (int i = 0; i < tmp[0].size(); i++) {
            for (int j = 0; j < tmp[1].size(); j++) {
                int dis = abs(tmp[0][i].first - tmp[1][j].first) + abs(tmp[0][i].second - tmp[1][j].second);
                res = min(res, dis);
            }
        }
        

        return res - 1;
        
    }
    
    void color(vector<vector<int>>& A, int i, int j) {
        int dx[4] = {1, -1, 0, 0};
        int dy[4] = {0, 0, 1, -1};
        

        A[i][j] = 2;
        
        for (int t = 0; t < 4; t++) {
            int nx = i + dx[t];
            int ny = j + dy[t];
            if (nx >=0 && nx < A.size() && ny >= 0 && ny < A.size()) {
                if (A[nx][ny] == 1) {
                    color(A, nx, ny);
                }
            }
        }
        
    }
    
    void out(vector<vector<int>>& A) {
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[0].size(); j++) {
                cout << A[i][j] << " ";
            }
            cout << endl;
        }
    }
    
    
    
};
