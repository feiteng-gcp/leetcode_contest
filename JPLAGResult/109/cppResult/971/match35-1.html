class Solution {
public:
    vector<int> row1, row2, col1,col2;
    int shortestBridge(vector<vector<int>>& A) {
		int N = A.size();
		//res标记为1的元素，统计个数。

		int mark = 1;
		//如果发现为1的元素，就去感染他周围（山下左右）的元素
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (A[i][j] == 1) {					
					infect(A, i, j, N, mark);
                    mark++;
				}
			}
		}
		return get_min_distance(N);
    }
    
    int get_min_distance(int N){

        int min_dis = N + N;
        for(int i = 0;i < row1.size();i++){
            for(int j = 0;j < row2.size();j++){
                int dis = abs(row1[i] - row2[j]) + abs(col1[i] - col2[j]) - 1;
                if(dis < min_dis)
                    min_dis = dis;
            }
        }
        return min_dis;
    }
    

    void infect(vector<vector<int>>& A, int i, int j, int N, int mark) {
		//元素下标越界或！=1退出
		if (i < 0 || i >= N || j < 0 || j >= N || A[i][j] != 1) {
			return;
		}
        if(mark == 1){
            row1.push_back(i);
            col1.push_back(j);
        }
        else if(mark == 2){
            row2.push_back(i);
            col2.push_back(j);
        }
		A[i][j] = 2;
		infect(A, i + 1, j, N, mark);//上
		infect(A, i - 1, j, N, mark);//下
		infect(A, i, j - 1, N, mark);//作
		infect(A, i, j + 1, N, mark);//右
		
	}
};
