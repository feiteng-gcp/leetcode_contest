class Solution {
public:
    int dir[4][2] = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
    vector<vector<int>> vis, g;
    int m, n;
    
    void dfs(int r, int c){
        vis[r][c] = 1;
        for(int i = 0; i < 4; ++i){
            int rr = r + dir[i][0];
            int cc = c + dir[i][1];
            if(rr < 0 || rr >= m || cc < 0 || cc >= n || g[rr][cc] == 0 || vis[rr][cc]) continue;
            dfs(rr, cc);
        }
    }
    
    int shortestBridge(vector<vector<int>>& A) {
        m = A.size();
        n = A[0].size();
        g = A;
        vis.assign(m, vector<int>(n, 0));
        
        bool f = false;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(A[i][j] == 1){
                    dfs(i, j);
                    f = true;
                    break;
                }
            }
            
            if(f) break;
        }
        
        queue<pair<int,int>> q;
        
        auto vis1 = vis;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(vis1[i][j]){
                    q.push({i, j});
                }
            }
        }
        
        int layer = 0;
        while(!q.empty()){
            int size = q.size();
            for(int i = 0; i < size; ++i){
                auto cur = q.front();
                q.pop();
                int r = cur.first, c = cur.second;
                
                if(g[r][c] == 1 && vis[r][c] == 0){
                    return layer - 1;
                }
                
                for(int i = 0; i < 4; ++i){
                    int rr = r + dir[i][0];
                    int cc = c + dir[i][1];
                    if(rr < 0 || rr >= m || cc < 0 || cc >= n || vis1[rr][cc]) continue;
                    //cout<<rr<<" "<<cc<<endl;
                    q.push({rr, cc});
                    vis1[rr][cc] = 1;
                }
            }
            
            layer++;
        }
        
        return 0;
    }
};
