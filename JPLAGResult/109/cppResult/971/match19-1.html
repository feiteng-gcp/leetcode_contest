class Solution {
public:
    struct Node {
        Node(int i, int j, int len) {
            this->i = i;
            this->j = j;
            this->len = len;
        }
        int i, j, len;
    };
    
    void dfs(vector<vector<int>>& color, vector<vector<int>>& a, int i, int j, int clr) {

        int dir[][2] = {{1,0}, {-1,0}, {0,1}, {0,-1}};
        color[i][j] = clr;
        for (int k = 0; k < 4; k++) {
            int r = i + dir[k][0];
            int c = j + dir[k][1];
            if (r >= 0 && r < a.size() && c >= 0 && c < a[0].size() && color[r][c] == 0 && a[r][c] == 1) {
                dfs(color, a, r, c, clr);
            }
        }
    }
    
    int shortestBridge(vector<vector<int>>& A) {
        vector<vector<int>> color(A.size(), vector<int>(A[0].size(), 0));

        int k = 1;
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[i].size(); j++) {
                if (A[i][j] == 1 && color[i][j] == 0) {
                    dfs(color, A, i, j, k++);
                }
            }
        }
        //cout << "qqq " << k << endl;
        queue<Node> que;

        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[i].size(); j++) {
                if (color[i][j] == 1) {
                    que.push(Node(i, j, 0));
                }
            }
        }
        
        while (!que.empty()) {
            Node t = que.front();
            que.pop();
            int dir[][2] = {{1,0}, {-1,0}, {0,1}, {0,-1}};
            //cout << t.i << " " << t.j << endl;

            for (int k = 0; k < 4; k++) {
                int r = t.i + dir[k][0];
                int c = t.j + dir[k][1];
                if (r >= 0 && r < A.size() && c >= 0 && c < A[0].size() && color[r][c] != 1) {
                    if (color[r][c] == 2) {
                        return t.len;
                    } else {
                        color[r][c] = 1;
                        que.push(Node(r, c, t.len + 1));
                    }
                }
            }
        }
        return -1;
    }
};
