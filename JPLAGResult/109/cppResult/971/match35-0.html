class Solution {
public:
void colorsLand(vector<vector<int>>& A, vector<vector<int>>& colors,int i,int j,int color)
{
	if (colors[i][j] == color)
	{
		return;
	}
	colors[i][j] = color;
	if (i > 0 && A[i - 1][j] == 1)
	{
		colorsLand(A, colors, i - 1, j, color);
	}
	if (j > 0 && A[i][j-1] == 1)
	{
		colorsLand(A, colors, i, j-1, color);
	}
	if (i < A.size()-1 && A[i+1][j] == 1)
	{
		colorsLand(A, colors, i+1, j, color);
	}
	if (j < A.size() - 1 && A[i][j+1] == 1)
	{
		colorsLand(A, colors, i, j + 1, color);
	}
}

int searchLand(vector<vector<int>>& A, vector<vector<int>>& colors, vector<vector<int>>& minCounts, int i, int j,int count)
{
	if (colors[i][j] == 2)
	{
		return count -1;
	}
	if (count >= minCounts[i][j])
	{
		return INT_MAX;
	}
	if (count < minCounts[i][j])
	{
		minCounts[i][j] = count;
	}
	int count1 = INT_MAX, count2 = INT_MAX, count3 = INT_MAX, count4 = INT_MAX;
	if (i > 0 && colors[i - 1][j] !=1)
	{
		count1 = searchLand(A, colors, minCounts, i - 1, j, count + 1);
	}
	if (j > 0 && colors[i][j -1] != 1)
	{
		count2 = searchLand(A, colors, minCounts, i, j - 1, count + 1);
	}
	if (i < A.size() - 1 && colors[i + 1][j] != 1)
	{
		count3 = searchLand(A, colors, minCounts, i + 1, j, count + 1);
	}
	if (j < A.size() - 1 && colors[i][j + 1] != 1)
	{
		count4 = searchLand(A, colors, minCounts, i, j + 1, count + 1);
	}
	return min(count1, min(count2, min(count3, count4)));
}

int shortestBridge(vector<vector<int>>& A) {
	int size = A.size();

	vector<vector<int>> colors(size, vector<int>(size, 0));
	int color = 1;
	for (int i = 0; i < size; ++i)
	{
		for (int j = 0; j < size; ++j)
		{
			if (A[i][j] == 1 && colors[i][j] == 0)
			{
				colorsLand(A, colors, i, j, color);
				++color;
			}
			if (color == 3)
			{
				break;
			}
		}
		if (color == 3)
		{
			break;
		}
	}
	//for (int i = 0; i < A.size(); ++i)
	//{
	//	for (int j = 0; j < A.size(); ++j)
	//	{
	//		cout << A[i][j] << " ";
	//	}
	//	cout << endl;
	//}
	//cout << "------------" << endl;
	//for (int i = 0; i < A.size(); ++i)
	//{
	//	for (int j = 0; j < A.size(); ++j)
	//	{
	//		cout << colors[i][j] << " ";
	//	}
	//	cout << endl;
	//}

	vector<vector<int>> minCounts(size, vector<int>(size, 2*size));
	int count = INT_MAX;
	for (int i = 0; i < size; ++i)
	{
		for (int j = 0; j < size; ++j)
		{
			if (A[i][j] == 1 && colors[i][j] == 1)
			{
				int tmp = searchLand(A, colors, minCounts, i, j, 0);
				count = min(tmp, count);
			}
		}
	}
	return count;
}
