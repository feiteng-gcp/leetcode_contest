class Solution {
public:
    using pii = pair<int, int>;
    vector<int> root;
    int dx[4] = {-1, 0, 1, 0};
    int dy[4] = {0, 1, 0, -1};
    int find(int node){
        if (node == root[node]) return node;
        return root[node] = find(root[node]);
    }
    
    void conn(int x, int y){
        int rx = find(x);
        int ry = find(y);
        if (rx != ry){
            root[ry] = rx;
        }
    }
    
    int shortestBridge(vector<vector<int>>& A) {
        int n = A.size(); int m = A[0].size();
        for (int i = 0; i < n * m; i++) root.push_back(i);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (A[i][j] == 1) {
                    if (i + 1 < n && A[i + 1][j] == 1) conn(i * m + j, (i + 1) * m + j);
                    if (j + 1 < m && A[i][j + 1] == 1) conn(i * m + j + 1, i * m + j);
                }
            }
        }
        unordered_map<int, vector<pii>> mp;
        int f = -1, s = -1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (A[i][j] == 1) {
                    int parent = find(i * m + j);
                    if (f == -1) f = parent;
                    else if (parent != f) s = parent;
                    for (int k = 0; k < 4; k++) {
                        int x = i + dx[k]; int y = j + dy[k];
                        if (x >= 0 && x < n && y >= 0 && y < m) {
                            //if (i == 0 && j == 4) cout << x << " " << y << " " << A[x][y] << endl;
                            if (A[x][y] == 0) {mp[parent].push_back({i, j}); break;}
                        }
                    }
                }
                
            }
        }
        int res = INT_MAX;
        for (auto num1 : mp[f]) {
            //cout << num1.first << " " << num1.second << endl;
            for (auto num2 : mp[s]) {
                res = min(res, abs(num1.first - num2.first) + abs(num1.second - num2.second));
            }
        }
        
        return res - 1;
               
    }
};
