class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        queue<pair<int, int>> q1, q2;
        int n = A.size();

        int ds[][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};

        bool found = false;
        for (int i = 0; i < n && !found; i++)
        {
            for (int j = 0; j < n && !found; j++)
            {
                if (A[i][j] == 1)
                {
                    found = true;
                    A[i][j] = 2;
                    q2.push({i, j});
                    while (!q2.empty())
                    {
                        auto cur = q2.front();
                        q2.pop();
                        for (auto d: ds)
                        {
                            int nx = cur.first + d[0], ny = cur.second + d[1];
                            if (nx >= 0 && nx < n && ny >= 0 && ny < n && A[nx][ny] == 1)
                            {
                                q2.push({nx, ny});
                                A[nx][ny] = 2;
                            }
                        }
                    }
                }
            }
        }

        int count1 = 0, count2 = 2;
        for (int i = 0; i< n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (A[i][j] == 1) count1++;
                else if (A[i][j] == 2) count2++;
            }
        }
        int cand = count1 < count2 ? 1 : 2;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (A[i][j] == cand)
                {
                    bool shouldAdd = false;
                    for (auto d: ds)
                    {
                        int nx = i + d[0], ny = j + d[1];
                        
                        if (nx >= 0 && nx < n && ny >= 0 && ny < n && A[nx][ny] == 0)
                        {
                            shouldAdd = true;
                        }
                    }

                    if (shouldAdd) q1.push({i, j});
                }
            }
        }
        int res = 0;
        while (!q1.empty())
        {
            int size = q1.size();
            for (int i = 0; i < size; i++)
            {
                auto cur = q1.front();
                q1.pop();
                for (auto d: ds)
                {
                    int nx = cur.first + d[0], ny = cur.second + d[1];
                    if (nx >= 0 && nx < n && ny >= 0 && ny < n)
                    {
                        if (A[nx][ny] == 3-cand) return res;
                        if (A[nx][ny] == 0)
                        {
                            A[nx][ny] = cand;
                            q1.push({nx, ny});
                        }
                    }
                }
            }
            res++;
        }
        return res;
    }
};
