class Solution {
public:
    vector<int> parent;
    int n, m;
    int dirs[4][2] = {{1,0}, {-1,0}, {0,1}, {0, -1}};
    int get_idx(int i, int j) {
        return i * m + j;
    }
    pair<int,int> get_pair(int x) {
        return make_pair(x / m, x % m);
    }
    int get_parent(int x) {
        return x == parent[x] ? x : parent[x] = get_parent(parent[x]);
    }
    void unite(int x, int y) {
        int px = get_parent(x);
        int py = get_parent(y);
        parent[px] = parent[py];
    }
    int shortestBridge(vector<vector<int>>& A) {
        n = A.size();
        m = A[0].size();
        parent = vector<int>();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m;j ++) {
                parent.push_back(get_idx(i, j));
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (A[i][j] == 0) continue;
                int id1 = get_idx(i, j);
                for (int k = 0; k < 4; k++) {
                    int newi = i + dirs[k][0];
                    int newj = j + dirs[k][1];
                    if (newi < 0 || newi >= n || newj < 0 || newj >= m || A[newi][newj] == 0) continue;
                    int id2 = get_idx(newi, newj);
                    unite(id1, id2);
                }
            }
        }
        int land1 = -1, land2 = -1;
        vector<pair<int,int>> v1;
        vector<pair<int,int>> v2;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                get_parent(get_idx(i, j));
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (A[i][j] == 1) {
                    int p = get_parent(get_idx(i, j));
                    if (land1 == -1 || land1 == p) {
                        land1 = p;
                        v1.push_back(make_pair(i, j));
                    } else {
                        land2 = p;
                        v2.push_back(make_pair(i, j)); 
                    }
                    
                }
            }
        }
        int ans = INT_MAX;
        for (int i = 0; i < v1.size(); i++) {
            for (int j = 0; j < v2.size(); j++) {
                int d = fabs(v1[i].first - v2[j].first) + fabs(v1[i].second - v2[j].second) - 1;
                ans = min(ans, d);
            }
        }
        return ans;
    }
};
