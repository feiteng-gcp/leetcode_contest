class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        int n = A.size(), m = A[0].size();
        vector<int> dirs = {0, 1, 0, -1, 0};
        queue<pair<int, int>> q_;
        for (int i = 0; i < n; ++i) {
            bool found = false;
            for (int j = 0; j < m; ++j) {
                if (A[i][j] == 1) {
                    queue<pair<int, int>> q;
                    q.push({i, j});
                    A[i][j] = 2;
                    while (!q.empty()) {

                        auto pos = q.front(); q.pop();
                        q_.push(pos);
                        for (int d = 0; d < 4; ++d) {
                            int x = pos.first + dirs[d];
                            int y = pos.second + dirs[d+1];
                            if (x >= 0 && x < n && y >= 0 && y < m && A[x][y] == 1) {
                                A[x][y] = 2;
                                q.push({x, y});
                            }
                        }
                    }
                    found = true;
                    break;
                }
            }
            if (found) break;
        }
        int steps = 0;
        while (!q_.empty()) {
            int k = q_.size();
            for (int i = 0; i < k; ++i) {

                auto pos = q_.front(); q_.pop();
                for (int d = 0; d < 4; ++d) {
                    int x = pos.first + dirs[d];
                    int y = pos.second + dirs[d+1];
                    if (x >= 0 && x < n && y >= 0 && y < m) {
                        if (A[x][y] == 1) {
                            return steps;
                        } else if (A[x][y] == 0) {
                            A[x][y] = 2;
                            q_.push({x, y});
                        }
                    }
                }
            }
            steps += 1;
        }
        return -1;
    }
};
