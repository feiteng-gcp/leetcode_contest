class Solution {
private:
    vector<pair<int, int>> group1;
    vector<pair<int, int>> group2;
    int len;
public:
    int shortestBridge(vector<vector<int>>& A) {
        int res = INT_MAX;
        len = A.size();
        bool finish_dfs = false;
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < len; j++) {
                if (A[i][j] == 1) {
                    dfs(i, j, A);
                    finish_dfs = true;
                    break;
                }
            }
            if (finish_dfs) {
                break;
            }
        }
        
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < len; j++) {
                if (A[i][j] == 1) {
                    group2.push_back({i, j});
                    for (auto node : group1) {
                        res = min(res, abs(node.first - i) + abs(node.second - j) - 1);
                    }
                }
            }
        }
        return res;
    }
    void dfs(int x, int y, vector<vector<int>>& A) {
        A[x][y] = -1;       // colored
        group1.push_back({x, y});
        if (x > 0 && A[x-1][y] == 1) {
            dfs(x-1, y, A);
        }
        if (x < len - 1 && A[x+1][y] == 1) {
            dfs(x+1, y, A);
        }
        if (y > 0 && A[x][y-1] == 1) {
            dfs(x, y-1, A);
        }
        if (y < len - 1 && A[x][y+1] == 1) {
            dfs(x, y+1, A);
        }
    }
};
