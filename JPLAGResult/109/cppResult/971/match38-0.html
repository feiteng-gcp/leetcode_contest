class Solution {
    vector<pair<int, int>> set1, set2;
    vector<vector<int>> visited;
    void dfs(int i, int j, int n, vector<vector<int>>& A, vector<pair<int, int>>& s) {
        if(i < 0 || j < 0 || i >= n || j >= n || visited[i][j] || !A[i][j])
            return;
        visited[i][j] = 1;
        s.emplace_back(i, j);
        dfs(i + 1, j, n, A, s);
        dfs(i, j + 1, n, A, s);
        dfs(i - 1, j, n, A, s);
        dfs(i, j - 1, n, A, s);
    }
public:
    int shortestBridge(vector<vector<int>>& A) {
        int n = A.size();
        visited = vector<vector<int>>(n, vector<int>(n, 0));
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                if(A[i][j]) {
                    if(set1.empty()) {
                        dfs(i, j, n, A, set1);
                    }
                    else {
                        dfs(i, j, n, A, set2);
                    }
                }
            }
        }
        int res = 2 * n;
        for(auto & p : set1) {
            for(auto & q : set2) {
                res = min(res, abs(p.first - q.first) + abs(p.second - q.second));
            }
        }
        return res - 1;
    }
};
