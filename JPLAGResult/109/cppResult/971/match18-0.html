class Solution {
public:
    typedef pair<int, int> pii;
    int shortestBridge(vector<vector<int>>& A) {
        int sz = A.size();
        int r0 = -1;
        int c0 = -1;
        for (int r=0; r < sz; ++r) {
            for (int c=0; c < sz; ++c) {
                if (A[r][c] == 1) {
                    r0 = r;
                    c0 = c;
                    break;
                }
            }
        }
        int dr[4] = { -1, 0, 1, 0 };
        int dc[4] = { 0, -1, 0, 1 };
        
        queue<pii> q;
        A[r0][c0] = 2;
        q.push(pii(r0, c0));
        while(q.size() > 0) {
            pii cur = q.front();
            q.pop();
            int r1 = cur.first;
            int c1 = cur.second;
            for (int i=0; i < 4; ++i) {
                int r2 = r1 + dr[i];
                int c2 = c1 + dc[i];
                if (r2 >= 0 && r2 < sz && c2 >= 0 && c2 < sz && A[r2][c2] == 1) {
                    A[r2][c2] = 2;
                    q.push(pii(r2, c2));
                }
            }
        }
        //q.clear();
        //vector<vector<int>> B(sz, vector<int>(sz, 0));
        for (int r=0; r<sz; ++r) {
            for (int c=0; c < sz; ++c) {
                if (A[r][c] == 2) {
                    for (int i=0; i < 4; ++i) {
                        int r2 = r + dr[i];
                        int c2 = c + dc[i];
                        if (r2 >= 0 && r2 < sz && c2 >= 0 && c2 < sz && A[r2][c2] == 0) {
                            A[r2][c2] = 3;
                            q.push(pii(r2, c2));
                        }
                    }
                }
            }
        }
        int d = 1;
        while(q.size() > 0) {
            pii cur = q.front();
            int r1 = cur.first;
            int c1 = cur.second;
            q.pop();
            d = A[r1][c1];
            for (int i=0; i < 4; ++i) {
                int r2 = r1 + dr[i];
                int c2 = c1 + dc[i];
                if (r2 >= 0 && r2 < sz && c2 >= 0 && c2 < sz) {
                    if(A[r2][c2] == 0) {
                        A[r2][c2] = d+1;
                        q.push(pii(r2, c2));
                    } else if (A[r2][c2] == 1)
                        return d - 2;
                }
            }
        }
        return d-2;
    }
};
