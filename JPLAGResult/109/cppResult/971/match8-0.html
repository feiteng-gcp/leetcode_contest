class Solution {
public:
    int vis[110][110];
    int dist(const pair<int , int>& p1, const pair<int , int>& p2) {
        return abs(p1.first - p2.first) + abs(p1.second - p2.second);
    }
    int dir[4][2] = {
        {-1, 0},
        {0, -1},
        {0, 1},
        {1, 0}
    };
    void find_c(const vector<vector<int>>& A, int i, int j, set<pair<int , int>>& c) {
        vis[i][j] = true;
        for (int k = 0; k < 4; ++k) {
            int x = i + dir[k][0];
            int y = j + dir[k][1];
            if (x >= 0 && x < A.size() && y >= 0 && y < A[0].size()) {
                if (A[x][y] == 0) c.emplace(i, j);
                if (A[x][y] == 1 && !vis[x][y]) find_c(A, x, y, c);
            }
        }
    }
    int shortestBridge(vector<vector<int>>& A) {
        memset(vis, false, sizeof(vis));
        int row = A.size();
        int col = A[0].size();
        set<pair<int, int>> c1, c2;
        bool flag_c1 = false;
        for (int i = 0; i < row; ++i) {
            for (int j = 0; j < col; ++j) {
                if (!vis[i][j] && A[i][j] == 1) {
                    if (flag_c1) find_c(A, i, j, c2);
                    else {
                        find_c(A, i, j, c1);
                        flag_c1 = true;
                    }
                }
            }
        }
    /*    {
            for (auto p : c1) cout << p.first << ' ' << p.second << endl;
            cout << endl;
            for (auto p : c2) cout << p.first << ' ' << p.second << endl;
        }*/
        int ans = 100 * 100;
        for (auto& p1 : c1) {
            for (auto& p2 : c2) {
                ans = min(ans, dist(p1, p2) - 1);
            }
        }
        
        return ans;
    }
};
