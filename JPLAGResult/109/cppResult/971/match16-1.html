class Solution {
 public:
  int shortestBridge(vector<vector<int>>& A) {
    const int n = A.size();
    vector<int> dx = {-1, 1, 0, 0};
    vector<int> dy = {0, 0, -1, 1};
    auto find_first = [&]() -> pair<int, int> {

      for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
          if (A[i][j] == 1) {
            return {i, j};
          }
        }
      }
    };
    auto p = find_first();
    queue<pair<int, int>> prv, cur;
    cur.push(p);
    vector<vector<bool>> vis(n, vector<bool>(n));
    vis[p.first][p.second] = true;
    while (!cur.empty()) {
      queue<pair<int, int>> nxt;
      while (!cur.empty()) {

        pair<int, int> u = cur.front();
        prv.push(u);
        cur.pop();
        for (int i = 0; i < 4; ++i) {
          int x = u.first + dx[i];
          int y = u.second + dy[i];
          if (x < 0 || x >= n || y < 0 || y >= n) {
            continue;
          }
          if (vis[x][y] || A[x][y] == 0) {
            continue;
          }
          vis[x][y] = true;
          nxt.emplace(x, y);
        }
      }
      cur = move(nxt);
    }
    int ans = 0;
    bool endgame = false;
    while (!prv.empty()) {
      while (!prv.empty()) {
        pair<int, int> u = prv.front();
        prv.pop();

        for (int i = 0; i < 4; ++i) {
          int x = u.first + dx[i];
          int y = u.second + dy[i];
          if (x < 0 || x >= n || y < 0 || y >= n) {
            continue;
          }
          if (vis[x][y]) {
            continue;
          }
          vis[x][y] = true;
          cur.emplace(x, y);
          if (A[x][y] == 1) {
            endgame = true;
          }
        }
      }
      if (endgame) {
        return ans;
      }
      ++ans;
      swap(prv, cur);
    }
  }
};
