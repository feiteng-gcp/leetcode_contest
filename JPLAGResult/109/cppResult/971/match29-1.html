class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        int n = A.size();
        int m = A[0].size();
        int i,j,x,y,s;
        int first_i = 0;
        int first_j = 0;
        int flag = 0;
        for(i = 0; i < n; i++){
            for(j = 0; j < m; j++){
                if(A[i][j] != 0){
                    first_i = i;
                    first_j = j;
                    flag = 1;
                    break;
                }
            }
            if(flag) break;
        }
        queue<int> qx;
        queue<int> qy;
        qx.push(first_i);
        qy.push(first_j);
        int vis[n][m];
        memset(vis, 0, sizeof(int)*n*m);
        queue<int> bx;
        queue<int> by;
        queue<int> bs;

        while(!qx.empty()){
            x = qx.front();
            qx.pop();
            y = qy.front();
            qy.pop();
            A[x][y] = -1;
            // Check border
            if((x>0 && A[x-1][y] == 0) || (x<n-1 && A[x+1][y] == 0) || (y>0 && A[x][y-1] == 0) || (y<m-1 && A[x][y+1]==0)){
                bx.push(x);
                by.push(y);
                bs.push(0);
            }
            // Add new point

            if(x>0 && !vis[x-1][y] && A[x-1][y] == 1){
                vis[x-1][y] = 1;
                qx.push(x-1);
                qy.push(y);
            }
            if(x<n-1 && !vis[x+1][y] && A[x+1][y] == 1){
                vis[x+1][y] = 1;
                qx.push(x+1);
                qy.push(y);
            }
            if(y>0 && !vis[x][y-1] && A[x][y-1] == 1){
                vis[x][y-1] = 1;
                qx.push(x);
                qy.push(y-1);
            }
            if(y<m-1 && !vis[x][y+1] && A[x][y+1] == 1){
                vis[x][y+1] = 1;
                qx.push(x);
                qy.push(y+1);
            }
        }
        while(!bx.empty()){
            // Check Border
            x = bx.front();
            y = by.front();
            s = bs.front();
            bx.pop();by.pop();bs.pop();
            if((x>0 && A[x-1][y] == 1) || (x<n-1 && A[x+1][y] == 1) || (y>0 && A[x][y-1] == 1) || (y<m-1 && A[x][y+1]==1)){
                return s;
            }
            // Add new point

            if(x>0 && !vis[x-1][y] && A[x-1][y] == 0){
                vis[x-1][y] = 1;
                bx.push(x-1);
                by.push(y);
                bs.push(s+1);
            }
            if(x<n-1 && !vis[x+1][y] && A[x+1][y] == 0){
                vis[x+1][y] = 1;
                bx.push(x+1);
                by.push(y);
                bs.push(s+1);
            }
            if(y>0 && !vis[x][y-1] && A[x][y-1] == 0){
                vis[x][y-1] = 1;
                bx.push(x);
                by.push(y-1);
                bs.push(s+1);
            }
            if(y<m-1 && !vis[x][y+1] && A[x][y+1] == 0){
                vis[x][y+1] = 1;
                bx.push(x);
                by.push(y+1);
                bs.push(s+1);
            }
        }
        return -1;
    }
};
