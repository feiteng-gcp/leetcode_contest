class Solution {
    vector<int> d = {1, 0, -1, 0, 1};
    void mark(vector<vector<int>>& A, int val) {
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[0].size(); j++) {
                if (A[i][j] == 1) {
                    A[i][j] = val;
                    queue<pair<int, int>> q({{i, j}});
                    while (q.size() > 0) {
                        for (int k = 0; k < d.size() - 1; k++) {
                            int x = q.front().first + d[k];
                            int y = q.front().second + d[k+1];
                            if (x >= 0 && x < A.size() && y >= 0 && y < A[0].size()) {
                                if (A[x][y] == 1) {
                                    A[x][y] = val;
                                    q.push({x, y});
                                }
                            }
                        }
                        q.pop();
                    }
                    return;
                }
            }
        }
    }
    
public:
    int shortestBridge(vector<vector<int>>& A) {
        mark(A, -1);
        
        queue<pair<int, int>> q;
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A[0].size(); j++) {
                if (A[i][j] == 1) {
                    q.push({i, j});
                }
            }
        }
        
        while (q.size() > 0) {
            for (int i = 0; i < d.size() - 1; i++) {
                int x = q.front().first + d[i];
                int y = q.front().second + d[i+1];
                if (x >= 0 && x < A.size() && y >= 0 && y < A[0].size()) {
                    if (A[x][y] == -1) {
                        return A[q.front().first][q.front().second] - 1;
                    } else if (A[x][y] == 0) {
                        A[x][y] = A[q.front().first][q.front().second] + 1;
                        q.push({x, y});
                    }
                }
            }
            q.pop();
        }
        
        return -1;
    }
};
