class Solution {
    int dx[4] = {0, 1, 0, -1};
    int dy[4] = {1, 0, -1, 0};
    int m, n;
public:
    int shortestBridge(vector<vector<int>>& A) {
        m = A.size();
        n = m == 0? 0: A[0].size();
        
        queue<pair<int, int>> q;
        
        for(int i = 0; i < m && q.empty(); i ++)
        {
            for(int j = 0; j < n; j ++)
            {
                if(A[i][j])
                {
                    ChangeAndEnqueue(A, q, i, j);
                    break;
                }
            }
        }
        
        int res = 0;
        
        while(!q.empty())
        {
            int cnt = q.size();
            
            while(cnt-- > 0)
            {
                pair<int, int> cur = q.front(); q.pop();
                
                for(int d = 0; d < 4; d++)
                {
                    int i = cur.first  + dx[d];
                    int j = cur.second + dy[d];
                    if (i >= 0 && i < m && j >= 0 && j < n)
                    {
                        if(A[i][j] == 1)
                        {
                            return res;
                        }
                        else if(A[i][j] == 0)
                        {
                            A[i][j] = 2;
                            q.push({i, j});
                        }
                    }
                }
            }
            res++;
        }
        
        
        return res;
    }
    
    void ChangeAndEnqueue(vector<vector<int>>& A, queue<pair<int, int>>& q, int i, int j)
    {
        q.push({i, j});
        A[i][j] = 2;
        
        for(int d = 0; d < 4; d ++)
        {
            int ii = i + dx[d];
            int jj = j + dy[d];
            
            if(ii >= 0 && ii < m && jj >= 0 && jj < n && A[ii][jj] == 1)
            {
                ChangeAndEnqueue(A, q, ii, jj);
            }
        }
        
    }
    
};
