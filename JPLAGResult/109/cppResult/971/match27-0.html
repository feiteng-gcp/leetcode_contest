class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        int m = A.size();
        if (m == 0 || A[0].size() == 0) return 1;//TBD
        int n = A[0].size();       
        queue<pair<int,int>> q;
        bool isMarked = false;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                //cout << "i,j =" << i << "," << j << endl;
                if (!isMarked && A[i][j] == 1) {
                    markIsland(i,j,A, q, m, n);
                    isMarked = true;
                    break;
                }
            }
            if (isMarked) break;
        }
        //now bfs from the island1 queue;
       // cout << "island one size = " << q.size() << endl;
        int step = 0;
        while (step < m + n) {
            for(auto size = q.size(); size > 0; size--) {
                pair<int,int> p = q.front(); q.pop();
                int i = p.first;
                int j = p.second;
                for (int k = 0; k < 4; k++) {
                    int ii = i + dir_[k];
                    int jj = j + dir_[k + 1];
                    if (ii >= 0 && ii < m && jj >= 0 && jj < n) {
                        if (A[ii][jj] == 1) {
                           //cout << "island two at" << ii << "," << jj << endl;
                          return step;//reached island2  
                        }
                        if (A[ii][jj] == 0) {
                            A[ii][jj] = -1; //expand to island1
                            q.push(pair<int,int>{ii,jj});
                        }
                    }
                }
            }
            step++;
        }
        return step;
    }
private:
    vector<int> dir_{0, -1, 0, 1, 0};
    void markIsland(int i, int j, vector<vector<int>>& A, queue<pair<int,int>>& q
                   ,const int m, const int n) {
            A[i][j]= -1;
            //cout << "adding island" << i << "," <<j << endl;
            q.push(pair<int,int>{i,j});
            for (int k = 0; k < 4; k++) {
                int ii = i + dir_[k];
                int jj = j + dir_[k + 1];
                if (ii >= 0 && ii < m && jj >= 0 && jj < n && A[ii][jj] == 1) {
                    //cout << "add neighbor " << ii << "," << jj << endl;
                    markIsland(ii,jj,A, q, m, n);
                }
            }
    }
};
