class Solution {
public:
    void dfsHelper(vector<vector<int>>& A, int x, int y, vector<pair<int, int>>& rec) {
        rec.emplace_back(x, y);
        A[x][y] = 2;
        vector<int> dir = { 0, 1, 0, -1, 1, 0, -1, 0 };
        for (int i = 0; i < 4; i++) {
            int new_x = x + dir[2 * i];
            int new_y = y + dir[2 * i + 1];
            if (new_x >= 0 && new_x < A.size() && new_y >= 0 && new_y < A[new_x].size() && A[new_x][new_y] == 1) {
                dfsHelper(A, new_x, new_y, rec);
            }
        }
    }
    
    int shortestBridge(vector<vector<int>>& A) {

        vector<pair<int, int>> one, two;
        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A.size(); j++) {
               if (A[i][j] == 1) {
                   if (one.empty()) {
                       dfsHelper(A, i, j, one);
                   }
                   else if (two.empty()) {

                       dfsHelper(A, i, j, two);
                   }
               } 
            }
        }
        int ans = 10000;
        for (auto& p1 : one) {
            for (auto& p2 : two) {
                if (abs(p1.first - p2.first) + abs(p1.second - p2.second) < ans) {
                    ans = abs(p1.first - p2.first) + abs(p1.second - p2.second);
                }
            }
        }
        return ans - 1;
    }
};
