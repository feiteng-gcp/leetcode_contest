struct Pt{
    int x,y;
    Pt(){}
    Pt(int y,int x):x(x),y(y){}
};
#define eprintf(format, ...) do{}while(0)

class Solution {
public:
    int shortestBridge(vector<vector<int>>& a) {
        queue<Pt> q;
        int n=a.size();

        for (int y=0;y<n;++y) {
            for (int x=0;x<n;++x) {
                if (a[y][x]) {
                    q.push(Pt(y,x));
                    a[y][x]=2;
                    break;
                }
            }
            if (!q.empty())break;
        }

        int dirs[]={1,0,-1,0,1};
        while (!q.empty()) {
            Pt cur=q.front();
            q.pop();
            int y=cur.y,x=cur.x;
            for (int k=0;k<4;++k) {
                int ny=y+dirs[k],nx=x+dirs[k+1];
                if (ny>=0&&ny<n&&nx>=0&&nx<n&&a[ny][nx]==1) {
                    a[ny][nx]=2;
                    q.push(Pt(ny,nx));
                }
            }
        }
        for (int y=0;y<n;++y) for (int x=0;x<n;++x)if (a[y][x]==2)q.push(Pt(y,x));
        int step=0;
        while (!q.empty()) {
            ++step;
            int remain=q.size();
            while (remain-->0) {

                Pt cur=q.front();
                q.pop();
                for (int k=0;k<4;++k) {
                    int ny=cur.y+dirs[k],nx=cur.x+dirs[k+1];
                    if (ny>=0&&ny<n&&nx>=0&&nx<n) {
                        if (a[ny][nx]==1)return step-1;
                        if (a[ny][nx]==0){
                            a[ny][nx]=2;
                            q.push(Pt(ny,nx));
                        }
                    }
                }
            }
        }
        return -1;
    }
};
