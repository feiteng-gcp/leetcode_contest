#define abs(x) (0 < (x) ? (x) : -(x))

class Solution {
public:
    vector<int> island1x; vector<int> island1y;
    vector<int> island2x; vector<int> island2y;
    
    int inline distance(int x1, int y1, int x2, int y2) {
        return abs(x1 - x2) + abs(y1 - y2) - 1;
    }
    
    void updateIsland(vector<vector<int>>& A, vector<int> & xs, vector<int> & ys, int x, int y) {
        xs.push_back(x);
        ys.push_back(y);
        A[x][y] = 0;
        if (x > 0 && A[x-1][y]) updateIsland(A, xs, ys, x-1, y);
        if (x + 1 < A.size() && A[x+1][y]) updateIsland(A, xs, ys, x+1, y);
        if (y > 0 && A[x][y-1]) updateIsland(A, xs, ys, x, y-1);
        if (y + 1 < A.size() && A[x][y+1]) updateIsland(A, xs, ys, x, y+1);
    }
    void createIsland(vector<vector<int>>& A, vector<int> & xs, vector<int> & ys) {
        for(int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A.size(); j++){
                if (A[i][j] == 1) {
                    updateIsland(A, xs, ys, i, j);
                    return;
                }
            }
        }
    }
    int shortestBridge(vector<vector<int>>& A) {
        createIsland(A, island1x, island1y);
        createIsland(A, island2x, island2y);
        // cout << island1x.size() << island2x.size() <<endl;
        int min_dist = A.size();
        for(int i = 0; i < island1x.size(); i++) {
            for(int j = 0; j < island2x.size(); j++) {
                int dist = distance(island1x[i], island1y[i], island2x[j], island2y[j]);
                min_dist = dist < min_dist ? dist : min_dist;
            }
        }
        return min_dist;
    }
};
