class Solution {
public:
    int shortestBridge(vector<vector<int>>& A) {
        queue<pair<int,int>>Q;
        vector<pair<int,int>>lhs, rhs;
        vector<int>dr{-1,1,0,0},dc{0,0,-1,1};
        int m = A.size(), n = m==0?0:A[0].size();

        vector<vector<int>>visit(m, vector<int>(n, 0));
        for(int i=0;i<m;++i){
            bool flag = false;
            for(int j=0;j<n;++j){
                if(A[i][j]){
                    Q.push(make_pair(i,j));
                    visit[i][j] = 1;
                    flag = true;
                    break;
                }
            }
            if(flag)break;
        }

        while(!Q.empty()){
            auto p = Q.front();
            Q.pop();
            lhs.push_back(p);
            for(int k=0;k<4;++k){
                int nr = p.first + dr[k];
                int nc = p.second + dc[k];
                if(nr>=0&&nr<m&&nc>=0&&nc<n&&!visit[nr][nc]&&A[nr][nc]){
                    visit[nr][nc] = 1;
                    Q.push(make_pair(nr, nc));
                }
            }
        }
        for(int i=0;i<m;++i){
            bool flag = false;
            for(int j=0;j<n;++j){
                if(A[i][j]&&!visit[i][j]){
                    Q.push(make_pair(i,j));
                    visit[i][j] = 1;
                    flag = true;
                    break;
                }
            }
            if(flag)break;
        }

        while(!Q.empty()){
            auto p = Q.front();
            Q.pop();
            rhs.push_back(p);
            for(int k=0;k<4;++k){
                int nr = p.first + dr[k];
                int nc = p.second + dc[k];
                if(nr>=0&&nr<m&&nc>=0&&nc<n&&!visit[nr][nc]&&A[nr][nc]){
                    visit[nr][nc] = 1;
                    Q.push(make_pair(nr, nc));
                }
            }
        }
        int ans = INT_MAX;
        for(auto &p:lhs){
            for(auto &q:rhs){
                ans = min(ans,abs(p.first-q.first)+abs(p.second-q.second)-1);
            }
        }
        return ans;
    }
};
