class Solution {
public:
    int x[4] = {-1,1,0,0};
    int y[4] = {0,0,-1,1};
    int shortestBridge(vector<vector<int>>& A) {
        
        bool first_island = false;
        int m = A.size();
        int n = A[0].size();
        vector<vector<bool>> visited(m, vector<bool> (n, false));
        vector<pair<int,int>> v;
        vector<pair<int,int>> v2;
        for(int i=0; i<m ; i++) {
            for(int j=0; j<n; j++) {
                if(!first_island && A[i][j]) {
                    fill_first(A,visited, v,i,j,m,n);
                    first_island = true;
                }
                else if(A[i][j] && !visited[i][j]) {
                    v2.push_back({i,j});
                }
            }
        }
        int dis = INT_MAX;
        for(int i=0;i<v.size();i++) {
            for(int j=0;j<v2.size();j++) {
                dis = min(dis, abs(v2[j].second-v[i].second) + abs(v2[j].first-v[i].first));
            }
        }
        return dis-1;
        
    }
    
    void fill_first(vector<vector<int>>& A, vector<vector<bool>> &visited, vector<pair<int,int>> &v, int i, int j, int m, int n) {
        if(!(i>=0 && i<m && j>=0 && j<n)) {
            return;
        }
        if(A[i][j] != 1) return;
        A[i][j]=0;
        visited[i][j] = true;
        v.push_back({i,j});
        for(int k = 0; k < 4; k++) {
            fill_first(A,visited, v, i+x[k], j+y[k], m, n);
        }
    }
    
};
