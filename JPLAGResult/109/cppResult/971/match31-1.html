class Solution {
    int dir[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};

    void dfs(int y, int x, vector<vector<int>> &field, vector<pair<int,int>> &t){
        check[y][x] = 1;
        t.push_back(make_pair(y,x));
        
        for(int i=0; i<4; ++i){
            int X = x+dir[i][0], Y = y+dir[i][1];
            if(X>=0 && X<size && Y>=0 && Y<size && check[Y][X] == 0 && field[Y][X] == 1){
                check[Y][X] = 1;
                dfs(Y,X,field,t);
            }
        }
    }
    
    vector<vector<int>> check;
    int size;
    vector<pair<int,int>> island_1, island_2;
    
public:
    int shortestBridge(vector<vector<int>>& A) {
        size = A.size();

        check = vector<vector<int>>(size, vector<int>(size,0));
        
        int t = 1;
        
        for(int i=0; i<size; ++i){
            for(int j=0; j<size; ++j){
                if(A[i][j] == 1 && check[i][j] == 0){
                    if(t==1){
                        dfs(i,j,A,island_1);
                        t = 2;
                    }
                    else{
                        dfs(i,j,A,island_2);
                    }
                }

            }
        }
        
        int m = 1e9;
        
        for(int i=0; i<island_1.size(); ++i){
            
            for(int j=0; j<island_2.size(); ++j){
                int k = abs(island_1[i].first - island_2[j].first) + abs(island_1[i].second - island_2[j].second) - 1;
                m = min(m, k);
            }
        }
        
        
        
        return m;
        
    }
};
