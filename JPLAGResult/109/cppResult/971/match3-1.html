class Solution {
public:
	int dx[4] = { 1, 0, 0, -1 };
	int dy[4] = { 0, 1, -1, 0 };
	vector<vector<int>>color;
	int curr;
	int n;

	void ff(int x, int y) {
		color[x][y] = curr;
		for (int i = 0; i < 4; i++) {
			int xx = x + dx[i],yy=y+dy[i];
			if (xx >= 0 && xx < n && yy >= 0 && yy < n&&color[xx][yy] == 1)ff(xx, yy);
		}
	}
	int shortestBridge(vector<vector<int>>& A) {
		curr = 2;
		n = A.size();

		color = A;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (color[i][j] == 1)ff(i, j),curr++;
			}
		}
		vector<pair<int, int>>a, b;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (color[i][j]==0)continue;
				if (color[i][j] == 2)a.push_back({ i,j });

				if (color[i][j] == 3)b.push_back({ i,j });
			}
		}
		int ret = 3 * n;
		for (int i = 0; i < a.size(); i++) {
			for (int j = 0; j < b.size(); j++) {
				ret = min(ret, abs(a[i].first - b[j].first) + abs(a[i].second - b[j].second));
			}
		}
		return ret-1;
	}
};
