class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        table = [1] * 10
        
        MOD 
= 10 ** 9 + 7
        
        for i in range(1, N):
            nt = [0] * 10
            
            nt[0] = (table[4] + table[6]) % MOD
            nt[1] = (table[6] + table[8]) % MOD
            nt[2] = (table[7] + table[9]) % MOD
            nt[3] = nt[1]
            nt[4] = (table[3] + table[9] + table[0]) % MOD
            nt[6] = nt[4]
            nt[7] = (table[2] + table[6]) % MOD
            nt[9] = nt[7]
            nt[8] = (table[1] + table[3]) % MOD
            
            table = nt
        return sum(table) % MOD
