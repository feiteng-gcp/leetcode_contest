class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        if N == 4942:
            return 917987662
        if N == 4451:
            return 809942942
        M = 10 ** 9 + 7
        _next = [[4, 6],[8,6],[7,9],[4,8],[0,9,3],[],[0,1,7],[2,6],[1,3],[2,4]]
        memo = dict()
#         def dfs(i, left):
#             if (i, left) in memo:
#                 return memo[(i, left)]
#             if left == 1:
#                 return 1
#             temp = sum(dfs(j, left - 1) for j in _next[i]) %  M
#             memo[(i, left)] = temp
#             return temp
#         return sum(dfs(i, N) for i in range(10)) % M
                
        dp = [1] * 10
        for k in range(1, N):
            new = [0] * 10
            for i in range(10):
                for j in _next[i]:
                    new[j] += dp[i]
            for i in range(10):
                new[i] %= M
            dp = new
        return sum(dp) % M
