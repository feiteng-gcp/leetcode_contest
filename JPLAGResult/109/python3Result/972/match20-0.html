class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        if N==1:
            return 10
        ans = [1 for _ in range(10)] 
        ans[5] = 0
        for _ in range(N-1):
            t = tuple(ans)
            ans[0] = t[4]+t[6] % (1000000000 + 7)
            ans[1] = t[6]+t[8] % (1000000000 + 7)
            ans[2] = t[7]+t[9] % (1000000000 + 7)
            ans[3] = t[4]+t[8] % (1000000000 + 7)
            ans[4] = t[3]+t[9]+t[0] % (1000000000 + 7)
            ans[6] = t[1]+t[7]+t[0] % (1000000000 + 7)
            ans[7] = t[2]+t[6] % (1000000000 + 7)
            ans[8] = t[1]+t[3] % (1000000000 + 7)
            ans[9] = t[2]+t[4] % (1000000000 + 7)
        
        return sum(ans) % (1000000000 + 7)
