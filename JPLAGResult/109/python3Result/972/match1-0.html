class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        num=[[0 for i in range(10)] for j in range(N+3)]
        for i in range(10):
            num[1][i]=1
        for i in range(2,N+2):
            num[i][0]=(num[i-1][6]+num[i-1][4])%(10**9+7)
            num[i][1]=(num[i-1][6]+num[i-1][8])%(10**9+7)
            num[i][2]=(num[i-1][7]+num[i-1][9])%(10**9+7)
            num[i][3]=(num[i-1][4]+num[i-1][8])%(10**9+7)
            num[i][4]=(num[i-1][3]+num[i-1][9]+num[i-1][0])%(10**9+7)
            num[i][5]=0
            num[i][6]=(num[i-1][1]+num[i-1][7]+num[i-1][0])%(10**9+7)
            num[i][7]=(num[i-1][2]+num[i-1][6])%(10**9+7)
            num[i][8]=(num[i-1][1]+num[i-1][3])%(10**9+7)
            num[i][9]=(num[i-1][2]+num[i-1][4])%(10**9+7)
        s=0
        for i in range(10):
            s=(s+num[N][i])%(10**9+7)
        return s
