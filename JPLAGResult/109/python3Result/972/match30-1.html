class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        nxt={1:[6,8],2:[7,9],3:[4,8],4:[3,9,0],5:[],6:[1,7,0],7:[2,6],8:[1,3],9:[2,4],0:[4,6]}
        mod=10**9+7
        curs=[1]*10
        for i in range(1,N):
            nxts=[0]*10
            for j in range(10):
                for k in nxt[j]:
                    nxts[j]+=(curs[k])%mod
            curs=nxts
        return sum(curs)%mod
