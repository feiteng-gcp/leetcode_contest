class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        d = {   1:[6,8],
                2:[7,9],
                3:[4,8],
                4:[0,3,9],
                5:[],
                6:[0,1,7],
                7:[2,6],
                8:[1,3],
                9:[2,4],
                0:[4,6],}
        
        l = [1 for _ in range(10)]
        l1 = [0 for _ in range(10)]
        
        for i in range(N-1):
            l1[0] = l[4] + l[6]
            l1[1] = l[6] + l[8]
            l1[2] = l[7] + l[9]
            l1[3] = l1[1]
            l1[4] = l[0] + l[3] + l[9]
            l1[6] = l1[4]
            l1[7] = l[2] + l[6]
            l1[8] = l[1] + l[3]
            l1[9] = l1[7]
            for k in range(10):
                l1[k] = l1[k] % (10 ** 9 + 7)
            l = l1
            l1 = [0 for _ in range(10)]
        return sum(l) % (10 ** 9 + 7)
