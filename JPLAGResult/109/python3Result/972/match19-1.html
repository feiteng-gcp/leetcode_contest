class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        jump = {}
        jump[0] = [4, 6]
        jump[1] = [6, 8]
        jump[2] = [7, 9]
        jump[3] = [4, 8]
        jump[4] = [0, 3, 9]
        jump[5] = []
        jump[6] = [0, 1, 7]
        jump[7] = [2, 6]
        jump[8] = [1, 3]
        jump[9] = [2, 4]

        n = N
        dp = [[0 for _ in range(10)] for _ in range
(n)]

        for j in range(10):
            dp[0][j] = 1
        for i in range(1, n):
            dp[i][0] = (dp[i - 1][4] + dp[i - 1][6]) % 1000000007
            dp[i][1] = (dp[i - 1][6] + dp[i - 1][8]) % 1000000007
            dp[i][2] = (dp[i - 1][7] + dp[i - 1][9]) % 1000000007
            dp[i][3] = (dp[i - 1][4] + dp[i - 1][8]) % 1000000007
            dp[i][4] 
= (dp[i - 1][0] + dp[i - 1][3] + dp[i - 1][9]) % 1000000007
            dp[i][5] = 0
            dp[i][6] = (dp[i - 1][0] + dp
[i - 1][1] + dp[i - 1][7]) % 1000000007
            dp[i][7] = (dp[i - 1][2] + dp[i - 1][6]) % 1000000007
            dp[i][8] = (dp[i - 1][1] + dp[i - 1][3]) % 1000000007
            dp[i][9] = (dp[i - 1][2] + dp[i - 1][4]) % 1000000007
        return sum(dp[n - 1]) % 1000000007
