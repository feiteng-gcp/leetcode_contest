class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        m 
= 10 ** 9 + 7
        next = [[4, 6], [8, 6], [7, 9], [4, 8], [3, 9, 0], [], [1, 7, 0], [2, 6], [1, 3], [4, 2]]
        if N == 1:
            return 10
        num = [1] * 10
        num[5] = 0
        
for i in range(N - 1):
            num2 = [0] * 10
            for j in range(10):
                for n in next[j]:
                    num2[j] += num[n]
                num2[j] %= m
            num = num2
        return sum(num) % m
