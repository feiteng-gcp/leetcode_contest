class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        M = 1000000000 + 7
        ss = [[4,6],[8,6],[7,9],[8,4],[3,9,0],[],[1,7,0],[2,6],[1,3],[2,4]]
        cnt = [1] * 10
        
if N == 0:
            return 1
        for i in range(1,N):
            ncnt = [0] * 10;
            for j in range(0,10):
                for k in ss[j]:
                    ncnt[j] = (ncnt[j] + cnt[k]) % M
            cnt = ncnt
        return sum(cnt) % M
