class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        dp = [[0]*2 for _ in range(10)]
        for ii in range(10):
            dp[ii][0] = 1
        for ii in range(1,N):
            dp[1][ii%2] = dp[6][(ii-1)%2]+dp[8][(ii-1)%2]
            dp[2][ii%2] = dp[7][(ii-1)%2]+dp[9][(ii-1)%2]
            dp[3][ii%2] = dp[4][(ii-1)%2]+dp[8][(ii-1)%2]
            dp[4][ii%2] = dp[3][(ii-1)%2]+dp[9][(ii-1)%2]+dp[0][(ii-1)%2]
            dp[6][ii%2] = dp[1][(ii-1)%2]+dp[7][(ii-1)%2]+dp[0][(ii-1)%2]
            dp[5][ii%2] = 0
            dp[7][ii%2] = dp[2][(ii-1)%2]+dp[6][(ii-1)%2]
            dp[8][ii%2] = dp[1][(ii-1)%2]+dp[3][(ii-1)%2]
            dp[9][ii%2] = dp[2][(ii-1)%2]+dp[4][(ii-1)%2]
            dp[0][ii%2] = dp[4][(ii-1)%2]+dp[6][(ii-1)%2]
        output = 0
        for ii in range(10):
            output += dp[ii][(N-1)%2]
        return output%(10**9+7)
