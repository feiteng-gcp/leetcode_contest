class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        if N == 1:
            return 10
        
        # N == 2 solution
        pad = [2, 2, 2, 2, 3, 0, 3, 2, 2, 2]
        
        mod = 10 ** 9 + 7
        
        upstreams = [
            [4, 6], # 0
            [6, 8], # 1
            [7, 9],
            [4, 8],
            [3, 9, 0],
            [],
            [1, 7, 0],
            [2, 6],
            [1, 3],
            [2, 4], # 9
        ]
        
        for _ in range(2, N):
            newpad = [0] * 10
            for i in range(10):
                for u in upstreams[i]:
                    newpad[i] += pad[u]
                newpad[i] %= mod
            pad = newpad
            
        return sum(pad) % mod
