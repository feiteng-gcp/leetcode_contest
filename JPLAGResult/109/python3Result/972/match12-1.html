class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        
        a = [
            [4,6],
            [6,8],
            [7,9],
            [4,8],
            [0,3,9],
            [],
            [0,1,7],
            [2,6],
            [1,3],
            [2,4],
        ]
        
        if N == 1:
            return 10;
        
        f = [1 for i in range(0, 10)]
        
        
for _ in range(1, N):
            g = [0] * 10
            for i in range(0, 10):
                for j in a[i]:
                    g[i] += f[j]
                    g[i] %= 10 ** 9 + 7
            f = g
           
            
        return sum(f) % (10 ** 9 + 7)
