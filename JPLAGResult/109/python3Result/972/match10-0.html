MOD = 10**9+7

class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        d = {0:[4,6], 1:[6,8], 2:[7,9], 3:[4,8], 4:[0,3,9], 5:[], 6:[0,1,7], 7:[2,6], 8:[1,3], 9:[2,4]}
        if N == 1:
            return 10
        counts = [1]*10
        for _ in range(1, N):
            tmp = [0]*10
            for i in range(10):
                for prev in d[i]:
                    tmp[i] += counts[prev]
                    tmp[i] %= MOD
            counts[:] = tmp
        return sum(counts)%MOD
