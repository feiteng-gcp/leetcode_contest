class Solution:
    def knightDialer(self, N):
        """
        :type N: int
        :rtype: int
        """
        lst = [1] * 10
        for i in range(N-1):
            new_lst = [0] * 10
            new_lst[0] = lst[4] + lst[6]
            new_lst[1] = lst[6] + lst[8]
            new_lst[2] = lst[7] + lst[9]
            new_lst[3] = lst[4] + lst[8]
            new_lst[4] = lst[3] + lst[9] + lst[0]
            new_lst[5] = 0
            new_lst[6] = lst[1] + lst[7] + lst[0]
            new_lst[7] = lst[2] + lst[6]
            new_lst[8] = lst[1] + lst[3]
            new_lst[9] = lst[2] + lst[4]
            lst = new_lst

        return sum(lst) % 1000000007
