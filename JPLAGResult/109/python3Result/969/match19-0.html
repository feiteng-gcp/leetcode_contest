from collections import deque 
class RecentCounter:

    def __init__(self):
        self.stk = deque()
        

    def ping(self, t):
        """
        :type t: int
        :rtype: int
        """
        while self.stk and self.stk[0]<t-3000:
            self.stk.popleft()
        self.stk.append(t)
        return len(self.stk)
