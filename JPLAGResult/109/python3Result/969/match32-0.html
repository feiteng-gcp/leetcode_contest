from collections import deque
class RecentCounter:

    def __init__(self):
        self.data = deque()

    def ping(self, t):
        """
        :type t: int
        :rtype: int
        """
        while self.data and t - self.data[0] > 3000:
            self.data.popleft()
        self.data.append(t)
        return len(self.data)
