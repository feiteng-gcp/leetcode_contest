from collections import deque


class RecentCounter:

    def __init__(self):
        self.past = deque(
)

    def ping(self, t):
        """
        :type t: int
        :rtype: int
        """
        while self.past and t - 3000 > self.past[0]:
            self.past.popleft()
        self.past.append(t)
        return len(self.past)
