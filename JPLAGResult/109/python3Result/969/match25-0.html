import queue


class RecentCounter:

    def __init__(self):
        self.d = queue.Queue()

    def ping(self, t):
        """
        :type t: int
        :rtype: int
        """
        self.d.put(t)
        while (self.d.queue)[0] < t-3000:
            self.d.get()
        return self.d.qsize()
