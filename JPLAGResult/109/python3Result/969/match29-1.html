class RecentCounter:

    def __init__(self):
        self.pins=
[]
        

    def ping(self, t):
        """
        :type t: int
        :rtype: int
        """
        self.pins.append(t)
        while self.pins and self.pins[0]<t-3000:
            self.pins.pop(0)
        return len(self.pins)
