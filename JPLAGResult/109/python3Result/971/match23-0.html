class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        # BFS
        # When in need, run bidirectional BFS
        
        from collections import deque
        
        m = len(A)
        n = len(A[0])
        visited = [[0] * n for _ in range(m)]
        # print(m, n)
        
        # for connected component
        q1 = deque()
        # for BFS
        q2 = deque()
        
        directions = [
            (1, 0),
            (0, 1),
            (-1, 0),
            (0, -1)
        ]
        
        # find first connected component (via 1's)
        find = False
        for i in range(m):
            for j in range(n):
                # print(i, j)
                if A[i][j] == 1:
                    # print('find', i, j)
                    find = True
                    break
            if find:
                # find (i, j) with 1
                break
        # print(i, j)
        q1.append((i, j))
        visited[i][j] = 1
        while q1:
            (i, j) = q1.popleft()
            for d in directions:
                ii = i + d[0]
                jj = j + d[1]
                if 0 <= ii and ii < m and 0 <= jj and jj < n:
                    if not visited[ii][jj]:
                        visited[ii][jj] = 1
                        if A[ii][jj] == 1:
                            q1.append((ii, jj))
                        else:
                            q2.append((ii, jj, 1))
        
        # print(q2)
        # find the bridge (via 0's)
        while q2:
            (i, j, l) = q2.popleft()
            for d in directions:
                ii = i + d[0]
                jj = j + d[1]
                if 0 <= ii and ii < m and 0 <= jj and jj < n:
                    if not visited[ii][jj]:
                        # print(i, j, l, '/', ii, jj)
                        visited[ii][jj] = 1
                        if A[ii][jj] == 1:
                            return l
                        else:
                            q2.append((ii, jj, l + 1))
                    
        # not reaching here
        assert False
