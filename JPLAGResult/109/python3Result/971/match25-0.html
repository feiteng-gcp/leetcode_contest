class Solution:
    def bfs(self, A, dist, x, y, n, color):
        A[x][y] = color
        #print(x,y,color)
        if (A[max(x-1, 0)][y] == 0) or (A[min(x+1, n-1)][y] == 0) or (A[x][max(y-1, 0)] == 0) or (A[x][min(y+1, n-1)] == 0):
            dist.add(x * n + y)
        if x > 0:
            if A[x-1][y] == 1:
                self.bfs(A, dist, x-1, y, n, color)
        if x < n-1:
            if A[x+1][y] == 1:
                self.bfs(A, dist, x+1, y, n, color)
        if y > 0:
            if A[x][y-1] == 1:
                self.bfs(A, dist, x, y-1, n, color)
        if y < n-1:
            if A[x][y+1] == 1:
                self.bfs(A, dist, x, y+1, n, color)
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        n = len(A)
        color = 2
        dist1 = set()
        dist2 = set()
        dist = dist1
        for i in range(n):
            for k in range(n):
                if A[i][k] == 1:
                    self.bfs(A, dist, i, k, n, color)
                    color += 1
                    dist = dist2
        ans = 200
        #print(dist1)
        #print(dist2)
        for left in dist1:
            for right in dist2:
                x1, y1 = divmod(left, n)
                x2, y2 = divmod(right, n)
                ans = min(ans, abs(x1-x2) + abs(y1-y2))
        return ans-1
