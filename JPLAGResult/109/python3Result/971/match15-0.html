class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        h = len(A)
        w = len(A[0])
        
        q = []
        
        for y in range(0, h):
            for x in range(0, w):
                if len(q) > 0:
                    break
                if A[y][x] == 1:
                    A[y][x] = 2
                    q.append((y, x))
            
            
        dy = [0, 1, 0, -1]
        dx = [1, 0, -1, 0]
        
        q3 = []
        
        while len(q) > 0:
            q2 = []
            for (y, x) in q:
                for d in range(0, 4):
                    y2 = y + dy[d]
                    x2 = x + dx[d]
                    if y2 >= 0 and y2 < h and x2 >= 0 and x2 < w:
                        if A[y2][x2] == 1:
                            A[y2][x2] = 2
                            q2.append((y2, x2))
                        elif A[y2][x2] == 0:
                            A[y2][x2] = 3
                            q3.append((y2, x2))
            q = q2
        
        ans = 1
        
        q = q3
        
        while len(q) > 0:
            q2 = []
            #print(A)
            for (y, x) in q:
                for d in range(0, 4):
                    y2 = y + dy[d]
                    x2 = x + dx[d]
                    if y2 >= 0 and y2 < h and x2 >= 0 and x2 < w:
                        if A[y2][x2] == 1:
                            return ans
                        elif A[y2][x2] == 0:
                            A[y2][x2] = 3
                            q2.append((y2, x2))
                            
            q = q2
            
            ans += 1
