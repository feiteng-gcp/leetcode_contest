class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        
        N, M = len(A), len(A[0])
        R = float('inf')
        vis = {}
        for i in range(N):
            for j in range(M):
                if A[i][j] == 1 and (i, j) not in vis:
                    bfs = [(i, j)]
                    vis[(i, j)] = 0
                    
for x, y in bfs:
                        for X, Y in [(x,y+1),(x,y-1),(x+1,y),(x-1,y)]:
                            if 0 <= X < N and 0 <= Y < M and (X, Y) not in vis and A[X][Y] == 1:
                                vis[(X, Y)] = 0
                                bfs.append((X, Y))
                    bfs = list(vis.keys())
                    for x, y in bfs:
                        for X, Y in [(x,y+1),(x,y-1),(x+1,y),(x-1,y)]:
                            if 0 <= X < N and 0 <= Y < M and (X, Y) not in vis:
                                if A[X][Y] == 1 and vis[(x, y)] > 0:
                                    return vis[(x, y)]
                                else:
                                    vis[(X, Y)] = vis[(x, y)] + 1
                                    bfs.append((X, Y))
                    return 1
        return 1
