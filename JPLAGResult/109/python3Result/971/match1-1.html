from collections import deque

class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        m = len(A)
        n = len(A[0])
        for i in range(m):
            flag = False
            for j in range(n):
                if A[i][j] == 1:
                    visited = [[False for _ in range(n)] for _ in range(m)]
                    self.dfs_mark_island(i,j,A,visited)
                    flag = True
                    break
            if flag:
                break
        
        visited = [[False for _ in range(n)] for _ in range(m)]
        d = deque()        
        
for i in range(m):
            for j in range(n):
                if A[i][j] == 2:
                    d.append((i,j,0))
                    visited[i][j] = True
        
        def nextstep(i,j,step,visited,d):
            if i<0 or i>=m or j<0 or j>=n:
                return
            if visited[i][j]:
                return
            visited[i][j] = True
            d.append((i,j,step))
        
        while len(d)>0:
            (i,j,step) = d.popleft()
            if A[i][j] == 1:
                return step-1
            nextstep(i+1,j,step+1,visited,d)
            nextstep(i-1,j,step+1,visited,d)
            nextstep(i,j+1,step+1,visited,d)
            nextstep(i,j-1,step+1,visited,d)
            
            
        
        
        return 0
                    
        
        
    
    def dfs_mark_island(self, i,j,A, visited):
        m = len(A)
        n = len(A[0])
        if i<0 or i>=m or j<0 or j>=n:
            return
        if visited[i][j]:
            return
        visited[i][j] = True
        if A[i][j]!=1:
            return
        A[i][j] = 2
        self.dfs_mark_island(i+1,j,A,visited)
        self.dfs_mark_island(i-1,j,A,visited)
        self.dfs_mark_island(i,j+1,A,visited)
        self.dfs_mark_island(i,j-1,A,visited)
