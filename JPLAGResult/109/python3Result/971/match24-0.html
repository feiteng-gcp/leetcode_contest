import queue

class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        m = len(A)
        n = len(A[0])
        directions = [(0,1),(0,-1),(1,0),(-1,0)]
        modified = False
        for ii in range(m):
            for jj in range(n):
                if A[ii][jj] == 1 and not modified:
                    q = queue.Queue()
                    A[ii][jj] = 2
                    q.put((ii,jj))
                    while not q.empty():
                        x,y = q.get()
                        for dx,dy in directions:
                            if 0<=x+dx<m and 0<=y+dy<n:
                                if A[x+dx][y+dy] == 1:
                                    q.put((x+dx,y+dy))
                                    A[x+dx][y+dy] = 2
                    modified = True
        
        
        output = None
        
        q = queue.Queue()
        for ii in range(m):
            for jj in range(n):
                if A[ii][jj] == 2:
                    q.put((ii,jj,0))
        visited = set()
        while not q.empty():
            x,y,d = q.get()
            for dx,dy in directions:
                if 0<=x+dx<m and 0<=y+dy<n and (x+dx,y+dy) not in visited:
                    if A[x+dx][y+dy] == 0:
                        visited.add((x+dx,y+dy))
                        q.put((x+dx,y+dy,d+1))
                    elif A[x+dx][y+dy] == 1:
                        return d
