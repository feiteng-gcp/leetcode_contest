class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        
        visited = set()
        glob_visited = {}
        R = len(A)
        C = len(A[0])
        def findComponents(i, j):
            if i >= 0 and i < R and j >= 0 and j < C and A[i][j] == 1 and (i, j) not in glob_visited:
                visited.add((i, j))
                glob_visited[i, j] = 1
                findComponents(i + 1, j)         
                findComponents(i - 1, j)
                findComponents(i, j + 1)
                findComponents(i, j - 1)
        c = 0
        compo = {}
        for i in range(R):
            for j in range(C):
                if A[i][j] == 1 and (i, j) not in glob_visited:
                    findComponents(i, j)    
                    compo[c] = visited
                    visited = set()
                    c += 1
        
        if len(compo[0]) < len(compo[1]):
            s_compo = 0
        else:
            s_compo = 1
        
        moves = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        def bfs(c):
            q = []
            for cell in compo[c]:
                q.append((cell[0], cell[1], 0))
            while q:
                e_i, e_j, level = q.pop(0)
                for m_i, m_j in moves:
                    i = e_i + m_i
                    j = e_j + m_j
                    if i >= 0 and i < R and j >= 0 and j < C:
                        if A[i][j] == 0 and (i, j) not in visited:
                            visited[i, j] = 1
                            q.append((i, j, level + 1))
                        elif A[i][j] == 1 and (i, j) in compo[1 - c]:
                            return level
            return float("inf")            
        
        visited = {}
        return bfs(s_compo)
