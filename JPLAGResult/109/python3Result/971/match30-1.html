class Solution:
  def shortestBridge(self, A):
    """
    :type A: List[List[int]]
    :rtype: int
    """
    m, n = len(A), len(A[0])
    DIRS = [(0, 1), (1, 0), (-1, 0), (0, -1)]
    
    def Dfs(i, j):
      A[i][j] = 2
      for d in DIRS:
        ni, nj = i + d[0], j + d[1]      
        if 0 <= ni < m and 0 <= nj < n and A[ni][nj] == 1:
          Dfs(ni, nj)
    
    def Bfs():
      dist = {}
      sources 
= []
      for i in range(m):
        for j in range(n):
          if A[i][j] == 1:
            sources.append((i, j))
            dist[(i, j)] = 0

      q = collections.deque(sources)
      while q:
        i, j = q.popleft()
        for d in DIRS:
          ni, nj = i + d[0], j + d[1] 
          if 0 <= ni < m and 0 <= nj < n:
            if A[ni][nj] == 2:
              return dist[(i, j)]
            if A[ni][nj] == 0:
              A[ni][nj] = 1
              q.append((ni, nj))
              dist[(ni, nj)] = dist[(i, j)] + 1
          
    
    for i in range(m):
      for j in range(n):
        if A[i][j] == 1:
          Dfs(i, j)
          return Bfs()
