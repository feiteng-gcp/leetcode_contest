class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        quene=[]
        l=len(A)
        visited=[[False for i in range(l)] for j in range(l)]
        si=-1
        sj=-1
        for i in range(l):
            for j in range(l):
                if A[i][j]==1:
                    quene.append((i,j))
                    break
            if len(quene)==1:
                break
        start=0
        visited[i][j]=True
        A[i][j]=-1
        while(start<=len(quene)-1):
            i=quene[start][0]
            j=quene[start][1]
            if i+1<l and A[i+1][j]==1 and visited[i+1][j]==False:
                quene.append((i+1,j))
                visited[i+1][j]=True
                A[i+1][j]=-1
            if j+1<l and A[i][j+1]==1 and visited[i][j+1]==False:
                quene.append((i,j+1))
                visited[i][j+1]=True
                A[i][j+1]=-1
            if i-1>=0 and A[i-1][j]==1 and visited[i-1][j]==False:
                quene.append((i-1,j))
                visited[i-1][j]=True
                A[i-1][j]=-1
            if j-1>=0 and A[i][j-1]==1 and visited[i][j-1]==False:
                quene.append((i,j-1))
                visited[i][j-1]=True
                A[i][j-1]=-1
            start=start+1

        while(start<=len(quene)-1):
            i=quene[start][0]

            if i+1<l and A[i+1][j]==0 and visited[i+1][j]==False:
                quene.append((i+1,j,k+1))
                visited[i+1][j]=True
                A[i+1][j]=k+1
            if j+1<l and A[i][j+1]==0 and visited[i][j+1]==False:
                quene.append((i,j+1,k+1))
                visited[i][j+1]=True
                A[i][j+1]=k+1
            if i-1>=0 and A[i-1][j]==0 and visited[i-1][j]==False:

                visited[i-1][j]=True
                A[i-1][j]=k+1

                quene.append((i,j-1,k+1))

                A[i][j-1]=k+1

            if j+1<l and A[i][j+1]==-1 and visited[i][j+1]==False:
                return k
            if i-1>=0 and A[i-1][j]==-1 and visited[i-1][j]==False:
                return k
            if j-1>=0 and A[i][j-1]==-1 and visited[i][j-1]==False:
                return k
