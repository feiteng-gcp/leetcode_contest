class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        m = len(A)
        n = len(A[0])

        flag = False
        bfs 
= []
        visited = {}
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    bfs.append((i, j))
                    visited[(i, j)] = 0
                    flag = True
                    break
            if flag:
                break
        index = 0
        direction = [(0, 1), (1, 0), (-1, 0), (0, -1)]
        while index < len(bfs):
            x, y = bfs[index]
            A[x][y] = -1
            for dire in direction:
                xx = x + dire[0]
                yy = y + dire[1]
                if 0 <= xx < m and 0 <= yy < n:
                    if A[xx][yy] == 1 and (xx, yy) not in visited.keys():
                        bfs.append((xx, yy))
                        visited[(xx, yy)] = 0
            index += 1

        index = 0
        while index < len(bfs):
            x, y = bfs[index]
            for dire in direction:
                xx = x + dire[0]
                yy = y + dire

                if 0 <= xx < m and 0 <= yy < n:
                    if A[xx][yy] == 1:
                        return visited[(x, y)]
                    if A[xx][yy] == 0 and (xx, yy) not in visited.keys():

= -1
                        bfs.append((xx, yy))
                        visited[(xx, yy)] = visited[(x, y)] + 1
            index += 1
