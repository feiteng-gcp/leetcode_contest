from collections import deque
class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        L = len(A)
        W = len(A[0])
        dirs = [[1,0], [0,1], [-1,0], [0,-1]]
        
        def InitQueue(x,y,  queue):
            A[x][y] = -1
            for dx, dy in dirs:
                newx, newy = x + dx, y + dy
                if 0<= newx < L and 0 <= newy < W and A[newx][newy] == 1:
                    InitQueue(newx, newy,queue)
                elif 0<= newx < L and 0 <= newy < W and A[newx][newy] == 0:
                    A[newx][newy] = -1
                    queue.append((newx,newy))
        
        def addQueue(x,y,queue):
            for dx, dy in dirs:
                newx, newy = x + dx, y + dy
                if 0<= newx < L and 0 <= newy < W and A[newx][newy] == 1:
                    return True
                elif 0<= newx < L and 0 <= newy < W and A[newx][newy] == 0:
                    A[newx][newy] = -1
                    queue.append((newx,newy))
            return False        
                    
        
        queue = []
        flag = False
        for i in range(L):
            for j in range(W):
                if A[i][j] == 1:
                    InitQueue(i,j, queue)
                    flag = True
                    break
            if flag:        
                break
                    
        distance = 0
        while True:
            distance += 1
            newqueue = []
            flag = False
            for x, y in queue:
                if addQueue(x,y,newqueue):
                    flag = True
                    break
            if flag: break
            queue = newqueue   
            
        return distance
