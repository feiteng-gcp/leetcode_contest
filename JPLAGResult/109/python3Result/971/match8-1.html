import queue

class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        m, n = len(A), len(A[0])
        q = queue.Queue()
        
for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    A[i][j] = -1
                    q.put((i, j))
                    break
            if not q.empty():
                break

        moves = [(-1, 0), (0, -1), (0, 1), (1, 0)]
        while not q.empty():
            x, y = q.get()
            for dx, dy in moves:
                new_x, new_y = x + dx, y + dy
                if 0 <= new_x < m and 0 <= new_y < n and A[new_x][new_y] == 1:
                    A[new_x][new_y] = -1
                    q.put((new_x, new_y))

        q = queue.Queue()
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    A[i][j] = -2
                    q.put((i, j, 0))

        while not q.empty():
            x, y, steps = q.get()
            for dx, dy in moves:
                new_x, new_y = x + dx, y + dy
                if 0 <= new_x < m and 0 <= new_y < n:
                    if A[new_x][new_y] == -1:
                        return steps
                    elif A[new_x][new_y] == 0:
                        A[new_x][new_y] = steps+1
                        q.put((new_x, new_y, steps+1))
        return 0
