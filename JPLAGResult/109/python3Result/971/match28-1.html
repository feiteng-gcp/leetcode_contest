class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        def getOneInIsland(A):
            N = len(A)
            for i in range(N):
                for j in range(N):
                    if A[i][j] == 1:
                        return (i, j)

        def colorOneIsland(A, l):
            N = len(A)

            while len(l) > 0:
                t = l.pop()
                i = t[0]
                j = t[1]

                if A[i][j] == 2: continue

                A
[i][j] = 2
                for t in ((i-1, j), (i+1, j), (i, j-1), (i, j+1)):
                    x = t[0]
                    y = t[1]

                    if x < 0 or x >= N: continue
                    if y < 0 or y >= N: continue
                    if A[x][y] == 1:
                        l.append((x, y))

        def getResult(A):

            l = []
            for i in range(N):
                for j in range(N):
                    if A[i][j] == 2:
                        l.append((i, j))

            step = 0
            done = False
            while not done:
                l2 = []
                for one in l:
                    i = one[0]
                    j = one[1]
                    for t in ((i-1, j), (i+1, j), (i, j-1), (i, j+1)):
                        x = t[0]
                        y = t[1]

                        if x < 0 or x >= N: continue
                        if y < 0 or y >= N: continue

                        if A[x][y] == 1:
                            return step

                        if A[x][y] == 0:
                            A[x][y] = 2
                            l2.append((x, y))
                step += 1
                l = l2

        N = len(A)
        t = getOneInIsland(A)
        l = [t]
        colorOneIsland(A, l)
        return getResult(A)
