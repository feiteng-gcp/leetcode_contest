from collections import deque
class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        if not A or not A[0]:
            return 0
        m, n = len(A), len(A[0])
        q = deque()
        
        def dfs(i, j):
            A[i][j] = -1
            for d in [[0,1],[1,0],[0,-1],[-1,0]]:
                x = i + d[0]
                y = j + d[1]
                if x >= 0 and x < m and y >= 0 and y < n:
                    if A[x][y] == 1:
                        dfs(x, y)
                    elif A[x][y] == 0:
                        q.append((x, y))
                        A[x][y] = 2
        
        find = False
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    dfs(i, j)
                    find = True
                    break
            if find:
                break
        print(q)
        while q:
            i, j = q.popleft()
            for d in [[0,1],[1,0],[0,-1],[-1,0]]:
                x = i + d[0]
                y = j + d[1]
                if x >= 0 and x < m and y >= 0 and y < n:
                    if A[x][y] == 0:
                        q.append((x, y))
                        A[x][y] = A[i][j]+1
                    elif A[x][y] == 1:
                        # print(i, j, x, y, A)
                        return A[i][j]-1
        return 0
