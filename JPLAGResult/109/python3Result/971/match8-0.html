class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        # 标记岛2
        m = len(A)
        n = len(A[0])
        dot = []
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    A[i][j] = 2
                    dot.append([i, j])
                    break
            if len(dot) > 0:
                break

        while len(dot) > 0:
            num = len(dot)
            for i in range(num):
                [x, y] = dot[i]
                # print(x, y, A)
                if x-1 >= 0 and A[x-1][y] == 1:
                    A[x - 1][y] = 2
                    dot.append([x-1, y])
                if x+1 < m and A[x+1][y] == 1:
                    A[x + 1][y] = 2
                    dot.append([x+1, y])
                if y-1 >= 0 and A[x][y-1] == 1:
                    A[x][y-1] = 2
                    dot.append([x, y-1])
                if y+1 < n and A[x][y+1] == 1:
                    A[x][y+1] = 2
                    dot.append([x, y+1])
            dot = dot[num:]
        # 拓展岛2
        count = 0
        B = [[0]*n for x in range(m)]
        for i in range(m):
            for j in range(n):
                B[i][j] = A[i][j]
        while count < m+n:
            for x in range(m):
                for y in range(n):
                    # print(x, y)
                    # for a in B:
                    #     print(a)
                    if B[x][y] == 2:
                        if x - 1 >= 0:
                            if B[x - 1][y] == 1:
                                return count
                            elif B[x - 1][y] == 0:
                                A[x - 1][y] = 2
                        if x + 1 < m:
                            if B[x + 1][y] == 1:
                                return count
                            elif B[x + 1][y] == 0:
                                A[x + 1][y] = 2
                        if y - 1 >= 0:
                            if B[x][y-1] == 1:
                                return count
                            elif B[x][y-1] == 0:
                                A[x][y-1] = 2
                        if y + 1 < n:
                            if B[x][y+1] == 1:
                                return count
                            elif B[x][y+1] == 0:
                                A[x][y+1] = 2
            for i in range(m):
                for j in range(n):
                    B[i][j] = A[i][j]
            count += 1
        return max(m, n)
