class Solution:
    def shortestBridge(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        que = []
        def dfs(i, j):
            if A[i][j] == -1:
                return
            if A[i][j] == 0:
                que.append([i, j])
                A[i][j] = -1
                return
            A[i][j] = -1
            for x, y in [[i + 1, j],[i - 1, j], [i, j + 1], [i, j - 1]]:
                if  x in range(len(A)) and y in range(len(A[0])):
                    dfs(x, y)
            
        for i in range(len(A)):
            ptr = 0
            for j in range(len(A[0])):
                
if A[i][j]:
                    dfs(i, j)
                    ptr = 1
                    break
            if ptr:
                break
        cnt = 1
        # print(A)
        while que:
            new = []
            for i, j in que:
                for x, y in [[i + 1, j],[i - 1, j], [i, j + 1], [i, j - 1]]:
                    if x in range(len(A)) and y in range(len(A[0])):
                        if A[x][y] == 0:
                            new.append([x, y])
                        if A[x][y] == 1:
                            return cnt
                        A[x][y] = -1
            cnt += 1
            que = new
