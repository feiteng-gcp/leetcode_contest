class Solution:
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        N = len(points)

        edges = {}

        for i in range(N):
            p1 = points[i]
            for j in range(i+1, N):
                p2 = points[j]

                p = [0, 0]  # coordinate of mid point, but multiple by 2
                p[0] = p1[0] + p2[0]
                p[1] = p1[1] + p2[1]

                d2 = (p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2 # length ** 2

                t = (p[0], p[1], d2)
                if t not in edges:
                    edges[t] = []
                edges[t].append((i ,j))

        ret = 0

        for key in edges:
            l = edges[key]
            n = len(l)
            if n < 2:
                continue

            for i in range(n):
                p1 = points[l[i][0]]
                p3 = points[l[i][1]]
                for j in range(i+1, n):
                    p2 = points
[l[j][0]]

                    l12 = (p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2
                    l32 = (p3[0] - p2[0]) ** 2 + (p3[1] - p2[1]) ** 2

                    S = (l12 * l32) ** 0.5

                    if ret == 0 or ret > S:
                        ret = S

        return ret
