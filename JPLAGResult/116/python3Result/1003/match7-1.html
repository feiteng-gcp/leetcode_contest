import math
class Solution:
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        maps = collections.defaultdict(list)
        nei = collections.defaultdict(list)
        for i in range(len(points)):
            for j in range(i+1,len(points)):
                lens = (points[i][0]-points[j][0])**2 + (points[i][1]-points[j][1])**2
                
                maps[lens].append((i,j))
                
        visited = set()
        mins = float('inf')
        for i in range(len(points)):
            for j in range(i+1,len(points)):
                lens = (points[i][0]-points[j][0])**2 + (points[i][1]-points[j][1])**2
                if lens not in visited:
                    visited.add(lens)
                    if len(maps[lens])<2:continue
                    for m in range(len(maps[lens])):
                        for n in range(m+1,len(maps[lens])):
                            
                            a,b,c,d = maps[lens][m][0],maps[lens][m][1],maps[lens][n][0],maps[lens][n][1]
                            if len(set([a,b,c,d]))!=4:continue
                            # print(a,b,c,d)
                            lenac = (points
[a][0]-points[c][0])**2 + (points[a][1]-points[c][1])**2
                            lenad = (points[a][0]-points[d][0])**2 + (points[a][1]-points[d][1])**2
                            lenbc = (points
[b][0]-points[c][0])**2 + (points[b][1]-points[c][1])**2
                            lenbd = (points[b][0]-points[d][0])**2 + (points[b][1]-points[d][1])**2    
                            
                                
                            if lenad > lenac:
                                if lenac == lenbd and lenac+lens == lenad:
                                    mins = min(mins,math.sqrt(lenac*lens))
                                    # print('d',lenac,lenad,lenbc,lenbd,lens,mins)
                            if lenac > lenad:
                                if lenad == lenbc and lenad+lens == lenac:
                                    mins = min(mins,math.sqrt(lenad*lens))
                                    # print('a',lenac,lenad,lenbc,lenbd,lens,mins)
        if mins == float('inf'): return 0
        return mins
