class Solution:
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        res = 99999999.0
        kk = True
        cs = set()
        for e in points:
            cs.add((e[0],e[1]))
        le = len(points)
        for i in range(le-3):
            a = points[i]
            for j in range(i+1,le-2):
                b = points[j]
                for k in range(j+1,le-1):
                    c = points[k]
                    if (b[0]-a[0])*(c[0]-a[0])+(b[1]-a[1])*(c[1]-a[1]) == 0:
                        d = (c[0]+b[0]-a[0],c[1]+b[1]-a[1])
                        
                        if d in cs:
                            s = (((b[0]-a[0])**2+(b[1]-a[1])**2)*((c[0]-a[0])**2+(c[1]-a[1])**2))**0.5
                            
                            if kk or s < res:
                                res = s
                                kk = False
                    elif (b[0]-c[0])*(a[0]-c[0])+(b[1]-c[1])*(a[1]-c[1]) == 0:
                        d = (a[0]+b[0]-c[0],a[1]+b[1]-c[1])
                        
                        if d in cs:
                            s = (((b[0]-c[0])**2+(b[1]-c[1])**2)*((a[0]-c[0])**2+(a[1]-c[1])**2))**0.5
                            
                            if kk or s < res:
                                res = s
                                kk = False
                    elif (a[0]-b[0])*(c[0]-b[0])+(a[1]-b[1])*(c[1]-b[1]) == 0:
                        d = (c[0]+a[0]-b[0],c[1]+a[1]-b[1])
                        
                        if d in cs:
                            s = (((a[0]-b[0])**2+(a[1]-b[1])**2)*((c[0]-b[0])**2+(c[1]-b[1])**2))**0.5
                            
                            if kk or s < res:
                                res = s
                                kk = False
                                
                
            
        if kk == True:
            return 0.0
        else:
            return res
