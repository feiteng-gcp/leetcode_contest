class Solution:
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        seg = set()
        for i in range(len(points)):
            for j in range(i+1, len(points)):
                t = ((points[i][0], points[i][1]), (points[j][0], points[j][1]))
                seg.add(t)
                
        rec = {}
        for s in seg:
            o = ((s[0][0] + s[1][0])/2, (s[0][1] + s[1][1])/2)
            if o not in rec:
                rec[o] = {}
            lsquare = (s[0][0] - s[1][0]) * (s[0][0] - s[1]
[0]) + (s[0][1] - s[1][1]) * (s[0][1] - s[1][1])
            if lsquare not in rec[o]:
                rec[o][lsquare] = []
            rec[o][lsquare].append(s)
        res = 1600000001
        for o in rec:
            circle = rec[o]
            for lsquare in circle:
                segs = circle[lsquare]
                for i in range(len(segs)):
                    for j in range(i+1, len(segs)):
                        a = segs[i][0]
                        b = segs[i][1]
                        c = segs[j][0]
                        s = abs((b[0]-a[0]) * (c[1] - a[1]) - (c[0] - a[0]) * (b[1] - a[1]))
                        res = s if s < res else res
        return res if res != 1600000001 else 0
