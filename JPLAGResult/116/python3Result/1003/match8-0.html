class Solution:
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        points = list(map(tuple, points))
        s = set(points)
        n = len(points)
        d = [[0] * n for _ in range(n)]
        for i in range(n):
            for j in range(n):
                d[i][j] = (points[i][0] - points[j][0]) * (points[i][0] - points[j][0]) + (points[i][1] - points[j][1]) * (points[i][1] - points[j][1])
        found = False
        r = 0
        for p1 in range(n):
            for p2 in range(n):
                if p2 == p1: continue
                d1 = d[p1][p2]
                for p3 in range(n):
                    if p3 == p1 or p3 == p2: continue
                    d2 = d[p2][p3]
                    d3 = d[p1][p3]
                    if d3 == d1 + d2:
                        p4 = (points[p1][0] + points[p3][0] - points[p2][0], points[p1][1] + points[p3][1] - points[p2][1])
                        if p4 != points[p2] and p4 in s:
                            cr = (d1 * d2) ** 0.5
                            if not found or cr < r:
                                found = True
                                r = cr
        return r if found else 0
