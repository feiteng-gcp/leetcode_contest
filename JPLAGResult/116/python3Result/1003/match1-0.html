class Solution:
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        pointsDict = {}
        for point in points:
            pointsDict[(point[0], point[1])] = 1
        
        ans = float('inf')
        length = len(points)
        for i in range(length - 1):
            for j in range(i + 1, length):
                for k in range(length):
                    if k == i or k == j:
                        continue
                    dirOne = [points[i][0] - points[k][0], points[i][1] - points[k][1]]
                    dirTwo = [points[j][0] - points[k][0], points[j][1] - points[k][1]]
                    if dirOne[0] * dirTwo[0] + dirOne[1] * dirTwo[1] == 0:
                        mid = [(points[i][0] + points[j][0]), (points[i][1] + points[j][1])]
                        nextOne = (mid[0] - points[k][0], mid[1] - points[k][1])
                        # print(points[i], points[j], dirOne, dirTwo, nextOne)
                        if nextOne in pointsDict:
                            # print(points[i], points[j], dirOne, dirTwo, nextOne)
                            distOne = (dirOne[0] * dirOne[0] + dirOne[1] * dirOne[1]) ** 0.5
                            distTwo = (dirTwo[0] * dirTwo[0] + dirTwo[1] * dirTwo[1]) ** 0.5
                            # print(distOne, distTwo)
                            ans = min(ans, distOne * distTwo)
        
        if ans == float('inf'):
            return 0
        else:
            return ans
