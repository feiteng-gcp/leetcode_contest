class Solution:
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        # (x,y) (a,b)
        #  x*a + y*b == 0
        #  x*b + y*a 
        P = set([(x, y) for x, y in points])
        points = sorted(list(P))
        #print(points)
        
        N = len(points)
        if N < 4: return 0
        
        res = float('inf')
        for i in range(N):
            for j in range(i + 1, N):
                for k in range(j + 1, N):
                    x = points[j][0] - points[i][0]
                    y = points[j][1] - points[i][1]
                    a = points[k][0] - points[i][0]
                    b = points[k][1] - points[i][1]
                    if x*a + y*b == 0:
                        l = points[i][0] + x + a
                        r = points[i][1] + y + b
                        if (l, r) in P:
                            res = min(res, abs(x*b - y*a))
        return 0 if res >= float('inf') else res
