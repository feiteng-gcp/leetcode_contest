class Solution:
    import math
    def checktriangle(self, p1,p2,p3,MinArea):
        d12
=(p1[0]-p2[0])**2+(p1[1]-p2[1])**2
        d23=(p2[0]-p3[0])**2+(p2[1]-p3[1])**2
        d13=(p1[0]-p3[0])**2+(p1[1]-p3[1])**2
        if d12+d23==d13:
            Area=(d12*d23)**0.5
            ind=2
        elif d12+d13==d23:
            Area=(d12*d13)**0.5
            ind=1
        elif d23+d13==d12:
            Area=(d23*d13)**0.5
            ind=3
        else:
            return [0,0]
        if MinArea==0:
            return [ind,Area]
        elif MinArea>Area:
            return [ind,Area]
        else:
            return [0,0]
    def checkrectangle(self,p1,p2,p3,p4,check):
        q4=p4
        if check==1:
            q1=p1
            q2=p2
            q3=p3
        elif check==2:
            q1=p2
            q2=p3
            q3=p1
        elif check==3:
            q1=p3
            q2=p1
            q3=p2
        if q2[1]-q1[1]!=q4[1]-q3[1]:
            return False
        elif q2[0]-q1[0]!=q4[0]-q3[0]:
            return False
        else:
            return True
    def minAreaFreeRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: float
        """
        N=len(points)
        MinArea=0
        if N<=3:
            return 0
        for i in range(N-3):
            for j in range(i+1,N-2):
                for k in range(j+1,N-1):
                    check=self.checktriangle(points[i],points[j],points[k],MinArea)
                    if check[0]==0:
                        continue
                    else:
                        for l in range(k+1,N):
                            if self.checkrectangle(points[i],points[j],points[k],points[l],check[0])==True:
                                #print(i,j,k,l)
                                if MinArea==0:
                                    MinArea=check[1]
                                elif MinArea>check[1]:
                                        MinArea=check[1]
        return MinArea
