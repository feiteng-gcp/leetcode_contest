class Solution:
    def maxWidthRamp(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        N=len(A)
        M=0
        B=[0]*N
        C=[0]*N
        B[0]=A[0]
        for i in range(1,N):
            if A[i]<B[i-1]:
                B[i]=A[i]
            else:
                B[i]=B[i-1]
        C[N-1]=A[N-1]
        for i in range(N-2,0,-1):
            
if A[i]>C[i+1]:
                C[i]=A[i]
            else:
                C[i]=C[i+1]
        i=0
        j=1
        while i<N and j<N:
            if C[j]>=B[i]:
                if M<j-i:
                    M=j-i
                j=j+1
            else:
                i=i+1
        return M
