class Solution:
    def maxWidthRamp(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        newA = [(num, i) for i, num in enumerate(A)]
        newA.sort(key=lambda x: x[0])
        newL = [num[1] for num in newA]
        length = len(A)
        
        dp = [0] * length
        dp[-1] = newL[-1]
        for i in range(length - 1)[::-1]:
            if newL[i] < dp[i + 1]:
                dp[i] = dp[i + 1]
            else:
                dp[i] = newL[i]
        
        ans = 0
        for i in range(length):
            ans = max(ans, dp[i] - newL[i])
        
        return ans
