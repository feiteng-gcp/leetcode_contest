class Solution:
    def maxWidthRamp(self, A):
        """
        :type A: List[int]
        :rtype: int
        """

        N = len(A)
        l = [i for i in range(N)]
        l = list(zip(A, l))
        l.sort()

        l = [e[1] for e in l]

        mins = []
        curr = N + 100
        for e in l:
            curr = min(curr, e)
            mins.append(
curr)


        ret = 0
        for i in range(1, N):
            ret = max(ret, l[i] - mins[i-1])

        return ret
