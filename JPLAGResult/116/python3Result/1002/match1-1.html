class Solution:
    def maxWidthRamp(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        A_i = [(a, i) for i, a in enumerate(A)]
        A_i.sort(key = lambda x: x[0])
        
        max_stack = []
        for a, i in A_i[::-1]:
            if not max_stack or max_stack[-1] < i:
                max_stack.append(i)
        
        res = 0
        for a, i in A_i[:-1]:
            if i == max_stack[-1]:
                max_stack.pop()
            if i < max_stack[-1]:
                res = max(res, max_stack[-1] - i)
        
        return res
