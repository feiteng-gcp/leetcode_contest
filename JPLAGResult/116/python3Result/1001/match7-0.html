class Solution:
    def repeatedNTimes(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        counter = dict()
        for x in A:
            if x not in counter:
                counter[x] = 1
            else:
                counter[x] += 1
        n = len(A) // 2
        for k, v in counter.items():
            if v == n:
                return k
        return -1
