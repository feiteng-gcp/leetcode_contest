#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;
typedef long long ll;

vector <long long> v;
vector <pair <int, int> > dp;

class Solution {
public:
    int leastOpsExpressTarget(int x, int target) {
        v.clear();
        v.push_back(1);
        dp.push_back(make_pair (0, 0));
        while (v.back() <= 1e9)
        {
            v.push_back(v.back() * x);
            dp.push_back(make_pair (0, 0));

        }

        ll nv = v.back() - target;
        dp[0].first = 2 * (target % x);
        dp[0].second = 2 * (nv % x);
        //cout << dp[0].first << " " << dp[0].second << "\n";
        for (int i = 1; i < v.size(); i++)
        {
            int nn = (target / v[i]) % x;
            int n2 = (nv / v[i]) % x;
            dp[i].first = min (dp[i-1].first + i * nn, dp[i-1].second + i * (nn + 1));
            dp[i].second = min (dp[i-1].second + i * n2, dp[i-1].first + i * (n2 + 1));

            //cout << dp[i].first << " " << dp[i].second << "\n";
        }

        return dp[v.size() - 1].first - 1;
    }
};
