typedef pair<int, int> ii;
typedef long long int64;

class Solution {
public:
  vector<ii> A;
  // map<ii, int> cache;
  int solve(int x, int target, int k) {
    if (target == 0) return 0;
    // if (cache.count({target, k})) return cache[{target, k}];
    // cout << x << " " << target << " " << k << endl;
    int ret = 1 << 30;
    for (int i = -x + 1; i < x; ++i) {
      int T = target + i;
      if (T % x) continue;
      if (T % x * A[k].second + i * A[k].second >= ret) break;
      int cur = solve(x, T / x, k + 1) + T % x * A[k].second;

      ret = min(ret, cur + abs(i) * A[k].second);
    }
    //cout << k << " " << target << " " << ret << endl;
    // cache[{target, k}] = ret;
    return ret;
  }
  int leastOpsExpressTarget(int x, int target) {
    A.clear();
    A.emplace_back(1, 2);
    int64 p = x;
    for (int k = 1; p < 1LL << 60; ++k) {
      A.emplace_back(p, k);
      p *= x;
    }
    //reverse(A.begin(), A.end());
    // cache.clear();
    int ret = solve(x, target, 0);
    return ret - 1;
  }
};
