class Solution {
public:
    map<int, int> m[40];
    int xx;
    int f(int a, int b) {
        // printf("%d %d\n", a, b); fflush(stdout);
        if (a == 0)
            return 0;
        auto it = m[b].insert(make_pair(a, INT_MAX)).first;
        if (it->second != INT_MAX)
            return it->second;
        if (a % xx == 0) {
            it->second = f(a / xx, b + 1);
        } else {
            it->second = abs(b - 1) * (a % xx) + (a % xx - 1);
            if (a / xx)
                it->second += f(a / xx, b + 1) + 1;
            if (b + 1 < 40)
            it->second = min(it->second, abs(b - 1) * (xx - a % xx) + (xx - a % xx - 1) + f(a / xx + 1, b + 1) + 1);
        }
        return it->second;
    }
    int leastOpsExpressTarget(int x, int target) {
        xx = x;
        for (int i = 0; i < 40; i++)m[i].clear();
        return f(target, 0);
    }
};
