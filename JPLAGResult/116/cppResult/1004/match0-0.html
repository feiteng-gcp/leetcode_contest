typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)2E9;
#define MAXN 100005

int ds[40], dp[40][2];
class Solution {
public:
  int leastOpsExpressTarget(int x, int target) {
    int v = target, n = 0;
    while (v > 0) {
      ds[n++] = v % x;
      v /= x;
    }
    REP(i,0,n+1) REP(t,0,2) dp[i][t] = INF;
    dp[n][1] = n - 1;
    dp[n][0] = -1;
    for (int i = n-1; i>=0; i--) {
      int c = ds[i];
      int cost = i > 0 ? i : 2;
      if (dp[i+1][0] != INF) {
        dp[i][0] = min(dp[i][0], dp[i+1][0] + c * cost);
        dp[i][1] = min(dp[i][1], dp[i+1][0] + (c + 1) * cost);
      }
      if (dp[i+1][1] != INF) {
        dp[i][0] = min(dp[i][0], dp[i+1][1] + (x - c) * cost);
        dp[i][1] = min(dp[i][1], dp[i+1][1] + (x - c - 1) * cost);
      }
    }
    return dp[0][0];
  }
};
