#include <cmath>
class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {

        int N = points.size();
        double nbest = 1e12;
        for (int i = 0; i < N; i++)
            for (int j = i + 1; j < N; j++)
                for (int k = i + 1; k < N; k++)
                    for (int l = j + 1; l < N; l++)
                    {
                        if (i == j || i == l || j == k || j == l || k == l)
                            continue;
                        pair <double, double> v1, v2, v3, v4;
                        v1.first = points[i][0];
                        v2.first = points[j][0];
                        v3.first = points[k][0];
                        v4.first = points[l][0];
                        
                        if (v1.first + v3.first != v2.first + v4.first)
                            continue;
                        v1.second = points[i][1];
                        v2.second = points[j][1];
                        v3.second = points[k][1];
                        v4.second = points[l][1];
                        if (v1.second + v3.second != v2.second + v4.second)
                            continue;
                        if ((v2.second - v1.second) * (v3.second - v2.second) + (v2.first - v1.first) * (v3.first - v2.first) != 0)
                            continue;
                        
                        double dd = v1.first * v2.second + v2.first * v3.second + v3.first * v4.second + v4.first * v1.second;
                        double dr = v1.second * v2.first + v2.second * v3.first + v3.second * v4.first + v4.second * v1.first;
                        nbest = min (nbest,  abs (dd - dr) / 2);
                    }
        if (nbest > 1e10)
            return 0;
        return nbest;
    }
};
