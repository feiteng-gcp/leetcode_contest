class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        unordered_set<pair<int, int>, PairHash> p_set;
        for (const auto &point : points) {
            p_set.emplace(point[0], point[1]);
        }
        auto norm = [](const vector<int> &a, const vector<int> &b) -> double {
            return sqrt((b[0] - a[0]) * (b[0] - a[0]) + (b[1] - a[1]) * (b[1] - a[1]));
        };
        double result = 0;
        for (int i = 0; i + 2 < points.size(); ++i) {
            for (int j = i + 1; j + 1 < points.size(); ++j) {
                if (points[i] == points[j]) {
                    continue;
                }
                for (int k = j + 1; k < points.size(); ++k) {
                    if (points[k] == points[i] || points[k] == points[j]) {
                        continue;
                    }
                    if ((points[k][0] - points[i][0]) * (points[j][0] - points[i][0]) +
                       (points[k][1] - points[i][1]) * (points[j][1] - points[i][1]) == 0) {
                        pair<int, int> p(points[k][0] + points[j][0] - points[i][0],
                                         points[k][1] + points[j][1] - points[i][1]);
                        if (p_set.count(p) > 0) {
                            double area = norm(points[i], points[j]) * norm(points[i], points[k]);
                            if (result == 0 || area < result) {
                                result = area;
                            }
                        }
                    }
                }
            }
        }
        return result;
    }
private:
    struct PairHash {
        size_t operator()(const pair<int, int> &p) const {
            return 31 * hash<int>()(p.first) + hash<int>()(p.second);
        }
    };
};
