class Solution {
public:
    
    
    bool Ortho(vector<int> &p1, vector<int> &p2, vector<int> &p3) {
        return (p1[1] - p2[1]) * (p2[1] - p3[1]) + (p1[0] - p2[0]) * (p2[0] - p3[0]) == 0;
    }
    
    double getArea(vector<int> &p1, vector<int> &p2, vector<int> &p3, vector<int> &p4) {
        vector<vector<int>> p = {p1, p2, p3, p4};
        sort(p.begin(), p.end(), [&](vector<int> &_p1, vector<int> &_p2) {
           return (_p1[0] < _p2[0]) || (_p1[0] == _p2[0] && _p1[1] < _p2[1]); 
        });
        
        if (p[0][0] + p[3][0] != p[1][0] + p[2][0]) return INT_MAX;
        if (p[0][1] + p[3][1] != p[1][1] + p[2][1]) return INT_MAX;
        if (p[0][0] == p[1][0]) {
            if (p[0][1] == p[2][1]) return (p[3][0] - p[0][0]) * (p[3][1] - p[0][1]);
            else return INT_MAX;
        } else {
            if ((p[3][1] - p[1][1]) * (p[1][1] - p[0][1]) + (p[3][0] - p[1][0]) * (p[1][0] - p[0][0]) == 0) {
                return sqrt((p[3][1] - p[1][1]) * (p[3][1] - p[1][1]) + (p[3][0] - p[1][0]) * (p[3][0] - p[1][0])) * 
                       sqrt((p[1][1] - p[0][1]) * (p[1][1] - p[0][1]) + (p[1][0] - p[0][0]) * (p[1][0] - p[0][0])); 
            } else
                return INT_MAX;
        }
    }
