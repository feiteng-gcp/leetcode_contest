class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& ps) {
        unordered_map<string,int> mp;

        for(int i=0;i<ps.size();i++){
            mp[to_string(ps[i][0])+"_"+to_string(ps[i][1])]=i;
        }
        int n =ps.size();
        double mina = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(i==j)
                    continue;
                for(int k=0;k<n;k++){
                    if(k==i || k==j){
                        continue;
                    }
                    int x1 = ps[j][0]-ps[i][0];
                    int y1 = ps[j][1]-ps[i][1];
                    
                    int x2 = ps[k][0]-ps[j][0];
                    int y2 = ps[k][1]-ps[j][1];
                    
                    if(x1*x2+y1*y2!=0)
                        continue;
                    
                    // double mx = (ps[i][0]+ps[k][0])/2;
                    // double my = (ps[i][1]+ps[k][1])/2;
                    
                    int tx = (ps[i][0]+ps[k][0]-ps[j][0]);
                    int ty = (ps[i][1]+ps[k][1]-ps[j][1]);
                    string key = to_string(tx)+"_"+to_string(ty);
                    if(mp.count(key)==0)
                        continue;
                    if(mp[key]==i || mp[key]==j || mp[key]==k)
                        continue;
                    
                    double area = sqrt(x1*x1+y1*y1) * sqrt(x2*x2+y2*y2);
                    if(mina ==0 || area<mina){
                        mina = area;
                    }
                    
                }
            }
        }
        
        return mina;
    }
};
