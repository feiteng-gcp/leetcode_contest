struct MyPoint
{
    long long x, y;
    MyPoint(long long x, long long y) : x(x), y(y) {}
  
};

    MyPoint operator-(MyPoint cur, const MyPoint& other) 
    {
        cur.x -= other.x;
        cur.y -= other.y;
        return cur;
    }

double dist(MyPoint a, MyPoint b)
{
    a = a - b;
    return sqrt(pow((double)a.x, 2.0) + pow((double)a.y, 2.0));
}

double area(MyPoint p1, MyPoint p2, MyPoint p3)
{

    double a = dist(p1, p2);
    double b = dist(p1, p3);
    double c = dist(p2, p3);
    double s = ( a + b + c) / 2;
	double Area = sqrt(s * (s - a) * (s - b) * (s - c)); 
    return Area * 2;
}

class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n = points.size();

        double ans = double(40000) * 40000.0 * 40000;
        bool hans = false;
        for (int i = 0; i < n; ++i)
        {
            MyPoint a(points[i][0], points[i][1]);
            for (int j = i + 1; j < n; ++j)
            {
                MyPoint b(points[j][0], points[j][1]);
                for (int k = j + 1; k < n; ++k)
                {
                    MyPoint c(points[k][0], points[k][1]);
                    for (int l = k + 1; l < n; ++l)
                    {
                        MyPoint d(points[l][0], points[l][1]);
                        
                        if (isRectangle(a.x, a.y,
                                        b.x, b.y,
                                       c.x, c.y,
                                       d.x, d.y))
                        {
                            ans = min(ans, area(a, b, c)); 
                            hans = true;
                        }
                    }
                }
            }
        }
        return hans ? ans : 0.0;
    }
    
private:
    inline bool isRectangle(int x1, int y1,
                     int x2, int y2,
                     int x3, int y3,
                     int x4, int y4)
    {
        int cx,cy;
        long long dd1,dd2,dd3,dd4;

        cx=(x1+x2+x3+x4);
        cy=(y1+y2+y3+y4);
        
        x1 *= 4;
        x2 *= 4;
        x3 *= 4;
        x4 *= 4;
        
        y1 *= 4;
        y2 *= 4;
        y3 *= 4;
        y4 *= 4;        

        dd1=sqr(cx-x1)+sqr(cy-y1);
        dd2=sqr(cx-x2)+sqr(cy-y2);
        dd3=sqr(cx-x3)+sqr(cy-y3);
        dd4=sqr(cx-x4)+sqr(cy-y4);
        return dd1==dd2 && dd1==dd3 && dd1==dd4;
    }
    
    long long sqr(int a)
    {
        return (long long)a * (long long)a;
    }
};
