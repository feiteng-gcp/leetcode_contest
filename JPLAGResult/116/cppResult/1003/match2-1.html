class Solution {
public:
    long long point(pair<int,int> x,pair<int,int> y) {
        return 1LL*x.first*y.first+1LL*x.second*y.second;
    }
    long long d2(int x1,int y1,int x2,int y2) {
        int xx=x1-x2,yy=y1-y2;
        return 1LL*xx*xx+1LL*yy*yy;
    }
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n=points.size();

        double ans=2000000000;
        for (int a=0; a<n; a++) {
            for (int b=a+1; b<n; b++) {
                for (int c=b+1; c<n; c++) {
                    for (int d=c+1; d<n; d++) {
                        pair<int,int> ab=make_pair(points[a][0]-points[b][0],points[a][1]-points[b][1]);
                        pair<int,int> ac=make_pair(points[a][0]-points[c][0],points[a][1]-points[c][1]);
                        pair<int,int> ad=make_pair(points[a][0]-points[d][0],points[a][1]-points[d][1]);
                        pair<int,int> bc=make_pair(points[b][0]-points[c][0],points[b][1]-points[c][1]);
                        pair<int,int> bd=make_pair(points[b][0]-points[d][0],points[b][1]-points[d][1]);
                        pair<int,int> cd=make_pair(points[c][0]-points[d][0],points[c][1]-points[d][1]);

                        long long ab1=d2(points[a][0],points[a][1],points[b][0],points[b][1]);
                        long long ac1=d2(points[a][0],points[a][1],points[c][0],points[c][1]);
                        long long ad1=d2(points[a][0],points[a][1],points[d][0],points[d][1]);
                        long long bc1=d2(points[b][0],points[b][1],points[c][0],points[c][1]);
                        long long bd1=d2(points[b][0],points[b][1],points[d][0],points[d][1]);
                        long long cd1=d2(points[c][0],points[c][1],points[d][0],points[d][1]);
                        if (point(ab,bc)==0 && point(bc,cd)==0 && point(cd,ad)==0) ans=min(ans,sqrt(ab1*bc1));
                        if (point(ab,bd)==0 && point(bd,cd)==0 && point(cd,ac)==0) ans=min(ans,sqrt(ab1*bd1));
                        if (point(ac,bc)==0 && point(bc,bd)==0 && point(bd,ad)==0) ans=min(ans,sqrt(ac1*bc1));
                    }
                } 
            }
        }
        return (ans==2000000000)?0:ans;
    }
};
