class Solution {

public:
    double minAreaFreeRect(vector<vector<int>>& p) {
        double ans=0;
        int N=p.size();
        for(int i=0;i<N;i++)

            for(int j=0;j<N;j++){
                if(j==i) continue;
                for(int k=0;k<N;k++){
                    if(k==i||k==j) continue;
                    if((p[k][1]-p[i][1])*(p[j][1]-p[i][1])+(p[k][0]-p[i][0])*(p[j][0]-p[i][0])!=0) continue;
                    for(int m=0;m<N;m++){
                        if(m==i||m==j||m==k) continue;
                        if(p[k][1]+p[j][1]==p[i][1]+p[m][1]&&p[k][0]+p[j][0]==p[i][0]+p[m][0]){
                            double area=sqrt((p[k][1]-p[i][1])*(p[k][1]-p[i][1])+(p[k][0]-p[i][0])*(p[k][0]-p[i][0]))*sqrt((p[j][1]-p[i][1])*(p[j][1]-p[i][1])+(p[j][0]-p[i][0])*(p[j][0]-p[i][0]));
                            if(ans==0) ans=area;
                            else
                                ans=min(ans,area);
                        }
                    }
                }
            }
        return ans;
    }
};
