long long dis[55][55],center[55][55][2];
class Solution {
public:
    bool check(int a,int b,int c,int d)
{
	if(center[a][b][0]!=center[c][d][0]||center[a][b][1]!=center[c][d][1])return false;
	if(dis[a][b]!=dis[c][d])return false;
	return true;
}
double cal(int a,int b,int c,int d)
{
	return sqrt(1.0*dis[a][c])*sqrt(1.0*dis[a][d]);
}
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n=points.size();
	int f=0;
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			dis[i][j]=(long long)(points[j][0]-points[i][0])*(points[j][0]-points[i][0])+(long long)(points[j][1]-points[i][1])*(points[j][1]-points[i][1]);
			center[i][j][0]=points[j][0]+points[i][0];
			center[i][j][1]=points[j][1]+points[i][1];
		}

	}
	double ans=40000.0*40001.0;
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			for(int k=j+1;k<n;k++)
			{
				for(int l=k+1;l<n;l++)
				{
					if(check(i,j,k,l))
						{
							f=1;
							ans=min(ans,cal(i,j,k,l));
						}
					if(check(i,k,j,l))
						{
							f=1;
							ans=min(ans,cal(i,k,j,l));
						}
					if(check(i,l,j,k))
						{
							f=1;
							ans=min(ans,cal(i,l,j,k));
						}
				}
			}
		}
	}
	if(!f)return 0;
	return ans;
    }
};
