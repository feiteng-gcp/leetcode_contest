class Solution {
public:
    bool check(int ax, int bx, int cx) {
        return bx - ax + cx - bx == cx - ax;
    }
    
    bool degree90(int bx, int by, int ax, int ay, int cx, int cy) {
        long long xx = bx - ax;
        long long yy = by - ay;
        long long xxx = cx - ax;
        long long  yyy = cy - ay;
        return xx * xxx + yy * yyy == 0;
    }
    
    long long getLen(int ax, int ay, int bx, int by) {
        return (bx - ax) * 1ll * (bx - ax) + (by - ay) * 1ll * (by - ay);
    }
    
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n = points.size();
        double mini = 0;
        set < pair <int, int> > S;

        for (int i=0;i<points.size();i++) S.insert(make_pair(points[i][0], points[i][1]));
        for (int i=0;i<n;i++) {
            for (int j=0;j<n;j++) {
                for (int k=0;k<n;k++) {
                        if (i == j || i == k || j == k) continue;

                        int ax = points[i][0];
                        int ay = points[i][1];
                        
                        int bx = points[j][0];
                        int by = points[j][1];
                        
                        int cx = points[k][0];
                        int cy = points[k][1];
                                            
                        int dx = ax + cx - bx;
                        int dy = ay + cy - by;
                            
                        if (dx == ax && dy == ay) continue;
                        if (dx == bx && dy == by) continue;
                        if (dx == cx && dy == cy) continue;
                        if (S.find(make_pair(dx, dy)) == S.end()) continue;
                        
                        
                        if (!check(ax, bx, cx)) continue;
                        if (!check(ay, by, cy)) continue;
                        if (!check(ax, dx, cx)) continue;
                        if (!check(ay, dy, cy)) continue;
                        
                        if (!degree90(ax, ay, bx, by, cx, cy)) continue;
                        if (!degree90(bx, by, cx, cy, dx, dy)) continue;
                        if (!degree90(cx, cy, dx, dy, ax, ay)) continue;
                        if (!degree90(dx, dy, ax, ay, bx, by)) continue;
                        
                        long long ab = getLen(ax, ay, bx, by);
                        long long bc = getLen(bx, by, cx, cy);
                        long long cd = getLen(cx, cy, dx, dy);
                        long long da = getLen(dx, dy, ax, ay);
                        
                        if (ab == cd && bc == da) {
                            double ans = sqrt(ab * 1.0) * sqrt(bc * 1.0);
                            if (mini == 0.0 || ans < mini) {
                                mini = ans;
                            }
                        }
                    }
                }
            }
        return mini;
    }
};
