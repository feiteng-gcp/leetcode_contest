class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        unordered_set<vector<int>, HashPoint> table;
        for (int i = 0; i < points.size(); ++i) {
            table.insert(points[i]);
        }
        double minValue = 1e15;
        for (int i = 0; i < points.size() - 1 ; ++i) {
            for (int j = i + 1; j < points.size(); ++j) {
                vector<int> vec1{points[j][0] - points[i][0], points[j][1] - points[i][1]};
                for (int k = 0; k < points.size(); ++k) {
                    /*
                    cout << "i: (" << points[i][0] << ", " << points[i][1] << ")" << endl;
                    cout << "j: (" << points[j][0] << ", " << points[j][1] << ")" << endl;
                    cout << "k: (" << points[k][0] << ", " << points[k][1] << ")" << endl;
                    */
                    if (k != i && k != j) {
                        vector<int> tempVec{points[k][0] - points[i][0], points[k][1] - points[i][1]};
                        // cout << "tempVec: (" << tempVec[0] << ", " << tempVec[1] << ")" << endl;
                        if (tempVec[0] * vec1[0] + tempVec[1] * vec1[1] == 0) {
                            // cout << "Inner product == 0" << endl;
                            vector<int> targetVec{points[j][0] + tempVec[0], points[j][1] + tempVec[1]};
                            if (table.count(targetVec) != 0) {
                                minValue = min(minValue, getNorm(vec1) * getNorm(tempVec));
                            }
                        }
                    }
                }
            }
        }
        return isSameNumber(minValue, 1e15)? 0.0: minValue;
    }
private:
    double getNorm(const vector<int>& vec) {
        return sqrt(vec[0] * vec[0] + vec[1] * vec[1]);
    }
    /*

        double normVec1 = sqrt(vec1[0] * vec1[0] + vec1[1] * vec1[1]);
        double normVec2 = sqrt(vec2[0] * vec2[0] + vec2[1] * vec2[1]);
        vector<int> unitVec1 = vec1;
        vector<int> unitVec2 = vec2;

        unitVec1[1] = unitVec1[1] / normVec1;
        
        unitVec2[0] = unitVec2[0] / normVec2;
        unitVec2[1] = unitVec2[1] / normVec2;
        
        return isSameNumber(unitVec1[0], unitVec2[0]) && isSameNumber(unitVec1[1], unitVec2[1]);
    }
    */
    bool isSameNumber(double d1, double d2) {
        return fabs(d1 - d2) < 1e-5; 
    }
    
    struct HashPoint {
        size_t operator()(const vector<int>& vec) const {
            size_t hash_x = hash<int>{}(vec[0]);
            size_t hash_y = hash<int>{}(vec[1]);
            return hash_x ^ hash_y;
        }
    };
};
