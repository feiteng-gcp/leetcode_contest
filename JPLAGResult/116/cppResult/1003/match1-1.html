class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n = points.size();
        double best = std::numeric_limits<double>::max();
        int cc = 0;
        for (int a = 0; a < n-3; ++a) {
            for(int b = a+1; b < n-2; ++b) {
                for (int c = b+1; c < n-1;++c) {
                    for (int d = c+1; d < n; ++d) {
                        double res = area(points[a], points[b], points[c], points[d]);
                        if (res > 0) {
                            ++cc;
                            best = min(best, res);
                        }
                    }
                }
            }
        }
        if (cc == 0) {
            return 0;
        }
        return best;
    }
private:
    long long sq(vector<int>& a, vector<int>& b) {
        return (a[0]-b[0])*(a[0]-b[0]) + (a[1]-b[1])*(a[1]-b[1]);
    }
    double area(vector<int>& a, vector<int>& b, vector<int>& c, vector<int>& d) {
        if (sq(a,b)==sq(c,d) && sq(a,c)==sq(b,d) && sq(a,d)==sq(b,c)) {
            long long x = sq(a,b);
            long long y = sq(a,c);
            long long z = sq(a,d);
            vector<long long> pp = {x,y,z};
            sort(pp.begin(), pp.end());
            //cout << pp[0] << " " << pp[1] << " " << pp[2] << endl;
            return sqrt(1.0*pp[0]) * sqrt(1.0*pp[1]);            
        }
        else {
            return -1;
        }
    }
};
