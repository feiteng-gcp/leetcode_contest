class Solution {
public:
    
    double min(double a, double b)
    {
        return a < b ? a : b;
    }
    
    double getDist(double x1, double y1, double x2, double y2)
    {
        return sqrt(pow(x2 - x1, 2) + pow(y2 - y1, 2));
    }
    
    #define REASONABLE_DIST 0.01
    
    bool within(double dist1, double dist2, double diff)
    {
        return abs(dist2 - dist1) < diff;
    }
    
    double minAreaFreeRect(vector<vector<int>>& points) {
        
        bool rectFound = false;

        double minArea = 2000000000;
        
        for (int i = 0; i < points.size(); i++)
        {
            for (int j = 0; j < i; j++)
            {
                for (int k = 0; k < j; k++)
                {
                    for (int l = 0; l < k; l++)
                    {
                        // cout << i << " " << j << " " << k << " " << l << endl;   
                        double xval = ((double) points[i][0]) + points[j][0] + points[k][0] + points[l][0];
                        xval /= 4;
                        
                        double yval = ((double) points[i][1]) + points[j][1] + points[k][1] + points[l][1];
                        yval /= 4;
                        
                        double dist = getDist(xval, yval, points[i][0], points[i][1]);
                        
                        if (!within(dist, getDist(xval, yval, points[j][0], points[j][1]), REASONABLE_DIST))
                        {
                            continue;   
                        }
                                                
                        if (!within(dist, getDist(xval, yval, points[k][0], points[k][1]), REASONABLE_DIST))
                        {
                            continue;   
                        }
                                                
                        if (!within(dist, getDist(xval, yval, points[l][0], points[l][1]), REASONABLE_DIST))
                        {
                            continue;   
                        }
                        
                        double testx1 = (((double) points[i][0]) + points[j][0])/2;
                        double testy1 = (((double) points[i][1]) + points[j][1])/2;
                        
                        double area = 0.0;
                        
                        if (within(testx1, xval, REASONABLE_DIST) && within(testy1, yval, REASONABLE_DIST))
                        {
                            // cout << "Diag1" << endl;
                            // i , j are a diagonals
                            area = getDist(points[i][0], points[i][1], points[k][0], points[k][1]) 
                                        * getDist(points[i][0], points[i][1], points[l][0], points[l][1]);
                        }
                        else
                        {
                            
                            testx1 = (((double) points[i][0]) + points[k][0])/2;
                            testy1 = (((double) points[i][1]) + points[k][1])/2;
                            
                            cout << testx1 << " " << testy1 << endl;
                            if (within(testx1, xval, REASONABLE_DIST) && within(testy1, yval, REASONABLE_DIST))
                            {
                                // cout << "Diag2" << endl;
                                // i, k are diagonals
                                area = getDist(points[i][0], points[i][1], points[j][0], points[j][1]) 
                                        * getDist(points[i][0], points[i][1], points[l][0], points[l][1]);
                            }
                            else
                            {
                                // cout << "Diag3" << endl;
                                // i, l are diagonals
                                area = getDist(points[i][0], points[i][1], points[j][0], points[j][1]) 
                                        * getDist(points[i][0], points[i][1], points[k][0], points[k][1]);
                            }
                        }

                        rectFound = true;
                        minArea = min(area, minArea);
                        
                        // cout << "DETECTED RECTANGLE WITH AREA " << area << endl;
                    }
                }
            }
        }
        return rectFound ? minArea : 0;
    }
};
