class Solution {
public:
    double clac( map<pair< double , double >,vector< pair< int ,int >>>&check ,double right,pair<int,int>tempp,int x, int y,double k ){
        
        double difxx = tempp.first - x ;  
        double difyy = tempp.second - y ; 
        if( difxx == 0 ) 
            return INT_MAX  ; 
        double tempk = ( difyy )/( difxx ) ; 

        if( abs(tempk*k + 1 ) <1e-5 )
           return sqrt(right)*(sqrt(difyy*difyy+difxx*difxx ) )  ; 
        return INT_MAX ; 
    }
    int old( vector<vector<int>>& points){
        map< pair< int ,int > , int > x ;
        int ret = INT_MAX ; 
        for( int i=0 ; i< points.size() ; i++){
            for( int j=i+1; j< points.size() ; j++){
                if( points[j][0] == points[i][0] ){
                    int left = max(points[j][1] , points[i][1]  );
                    int right = min( points[j][1] , points[i][1] )  ;
                    if( x.count({left,right}) ){
                        ret = min( ret, (left-right)*(points[i][0] - x[{left,right}]) ) ; 
                    }
                    x[ {left, right}] =  points[i][0] ; 
                }
            }
        }
        return ret;
    }
    double minAreaFreeRect(vector<vector<int>>& points) {
        sort( points.begin() , points.end() , 
             [](vector< int > & a , vector< int > & b ){
                 return a[0] < b[0] ; 
             }
        );
        map< pair< double , double >  ,  vector< pair< int ,int > >   > check ; 
        double ans = INT_MAX; 
        ans = min( double (old( points)) ,ans ) ; 
        for( int i=0 ; i< points.size() ; i++){
            for( int j=i+1; j< points.size() ; j++){
                double difx = points[j][0] - points[i][0] ; 
                double dify = points[j][1] - points[i][1] ; 
                double right = dify*dify  + difx*difx  ; 
                if( points[j][0] == points[i][0] ){
                    ;
                }
                else {
                    double k = (dify ) / ( difx ) ; 
                    if( check.count( {k,right}) ) {
                        for( auto &x:check[{k,right}] ) 
                            ans = min( ans, clac( check , right , x ,points[i][0],points[i][1] ,k) ) ; 
                    }
                    check[{k,right}].push_back({points[i][0],points[i][1]}); 
                }
            }
        }
        return ans == INT_MAX?0:ans ; 
    }
};
