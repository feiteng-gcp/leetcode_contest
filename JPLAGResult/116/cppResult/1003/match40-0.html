#include <cmath>

class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        sort(points.begin(), points.end(), [](const vector<int>& x1, const vector<int>& x2){
            return x1[0] < x2[0] || (x1[0] == x2[0] && x1[1] >= x2[1]);
        });
        bool found = false;
        double ans = 0.0;
        for (int i = 0; i < points.size(); i++) {
            for (int j = i+1; j < points.size(); j++) {
                for (int k = j+1; k < points.size(); k++) {
                    for (int v = k+1; v < points.size(); v++) {
                        int line1_x = points[j][0] - points[i][0];
                        int line1_y = points[j][1] - points[i][1];
                        int line2_x = points[k][0] - points[i][0];
                        int line2_y = points[k][1] - points[i][1];
                        int line3_x = points[v][0] - points[k][0];
                        int line3_y = points[v][1] - points[k][1];
                        int line4_x = points[v][0] - points[j][0];
                        int line4_y = points[v][1] - points[j][1];
                        // ping xing
                        // cout << points[i][0] << " " << points[i][1] << endl;
                        // cout << points[j][0] << " " << points[j][1] << endl;
                        // cout << points[k][0] << " " << points[k][1] << endl;
                        // cout << points[v][0] << " " << points[v][1] << endl;
                        // if (points[i][0] == 2 && points[i][1] == 3)
                        //     cout << "(" << line1_x << ", " << line1_y << ") (" << line2_x << ", " << line2_y << ") (" << line3_x << ", " << line3_y << ") (" << line4_x << ", " << line4_y << ")" << endl;
                        if ((line1_x == line3_x && line1_y == line3_y) && (line2_x == line4_x && line2_y == line4_y)) {
                            int xx = line1_x * line2_x;
                            int yy = line1_y * line2_y;
                            if (((xx >= 0 && yy <= 0) || (xx <= 0 && yy >= 0)) && xx + yy == 0) {
                                int x1 = static_cast<double>(line1_x);
                                int y1 = static_cast<double>(line1_y);
                                int x2 = static_cast<double>(line2_x);
                                int y2 = static_cast<double>(line2_y);
                                double rec = sqrt(x1*x1 + y1*y1) * sqrt(x2*x2 + y2*y2);
                                // rec = sqrt(rec);
                                if (found) {
                                    ans = min(ans, rec);
                                } else {
                                    found = true;
                                    ans = rec;
                                }
                            }
                        }
                    }
                }
            }
        }
        return ans;
    }
};
