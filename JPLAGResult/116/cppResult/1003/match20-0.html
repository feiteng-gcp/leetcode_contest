map<pair<int,int>,int> mp;
class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n=points.size();
        mp.clear();
        for(int i=0;i<n;i++){
            mp[make_pair(points[i][0],points[i][1])]=1;
        }
        int ok=0;
        double ans=2e9;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(i==j)
                    continue;
                int nx=points[j][0]-points[i][0];
                int ny=points[j][1]-points[i][1];
                for(int k=0;k<n;k++){
                    if(i==k||j==k)
                        continue;
                    int fx=points[k][0]+nx;
                    int fy=points[k][1]+ny;
                    if(mp[make_pair(fx,fy)]==0)
                        continue;
                    int l1=nx*nx+ny*ny;
                    int l2=(points[k][0]-points[i][0])*(points[k][0]-points[i][0])+(points[k][1]-points[i][1])*(points[k][1]-points[i][1]);
                    int l3=(points[k][0]-points[j][0])*(points[k][0]-points[j][0])+(points[k][1]-points[j][1])*(points[k][1]-points[j][1]);
                    if(l1+l2!=l3)
                        continue;
                    ans=min(ans,sqrt((double)l1)*sqrt((double)l2));
                    ok=1;
                }
            }
        }
        if(!ok)
            return 0;
        return ans;
    }
};
