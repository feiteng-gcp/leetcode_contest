class Solution {
public:
    long long enc(vector<int>& p) {
        return p[0]*40001LL+p[1];
    }
    double minAreaFreeRect(vector<vector<int>>& points) {
        unordered_set<long long> s;
        for(auto& p: points) {
            s.insert(enc(p));
        }
        double best = 1e100;
        int n = points.size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                for (int k = j+1; k < n; k++) {
                    if (i == k) continue;
                    vector<int> &A = points[i];
                    vector<int> &B = points[j];
                    vector<int> &C = points[k];
                    if ((A[0]-B[0])*1LL*(A[0]-C[0]) + (A[1]-B[1])*1LL*(A[1]-C[1]) == 0) {
                        // inner product = 0
                        vector<int> D = {B[0]+C[0]-A[0], B[1]+C[1]-A[1]};
                        if (s.find(enc(D)) != s.end()) {
                            long long v = abs((C[0]-A[0])*1LL*(B[1]-A[1]) - (C[1]-A[1])*1LL*(B[0]-A[0]));
                            if (v > 0) {
                                best = min(best, (double)v);
                            }
                        }
                    }
                }
            }
        }
        if (best > 1e90) best = 0;
        return best;
            
    }
};
