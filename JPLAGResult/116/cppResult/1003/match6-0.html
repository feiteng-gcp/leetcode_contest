class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n = points.size();
        
        // key: (sum of x, sum of y, square of length)
        map<tuple<int, int, int>, vector<vector<int>>> map;
        
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int sum_x = points[i][0] + points[j][0];
                int sum_y = points[i][1] + points[j][1];
                int square_length = square(points[i][0] - points[j][0]) + square(points[i][1] - points[j][1]);
                map[make_tuple(sum_x, sum_y, square_length)].push_back({points[i][0], points[i][1], points[j][0], points[j][1]});
            }
        }
        
        double res = 0;
        
        for (auto p : map) {
            vector<vector<int>>& pts = p.second;
            int m = pts.size();
            
            for (int i = 0; i < m; i++) {
                for (int j = i + 1; j < m; j++) {
                    double square_area = (square(pts[i][0] - pts[j][0]) + square(pts[i][1] - pts[j][1]))
                        * (square(pts[i][0] - pts[j][2]) + square(pts[i][1] - pts[j][3]));
                    double area = sqrt(square_area);
                    
                    if (res == 0 || area < res) {
                        res = area;
                    }
                }
            }
        }
        
        return res;
    }
    
private:
    double square(int a) {
        return a * a;
    }
};
