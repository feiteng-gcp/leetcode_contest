class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        #define M (40000 + 10)
        #define HASH(x, y) ((x) * M + (y))
        
        int N = points.size();
        
        unordered_set<int> p_set;
        for(int i = 0; i < N; i ++){
            p_set.insert(HASH(points[i][0], points[i][1]));
        }
        
        double area = M * M;
        for(int i = 0; i < N; i ++){
            for(int j = 0; j < N; j ++){
                if(i == j) continue;
                for(int k = 0; k < N; k ++){
                    if(j == k or i == k) continue;
                    
                    int dx1 = points[j][0] - points[i][0], dy1 = points[j][1] - points[i][1];
                    int dx2 = points[k][0] - points[i][0], dy2 = points[k][1] - points[i][1];
                    
                    if(dx1 * dx2 + dy1 * dy2 == 0){
                        int x3 = points[j][0] + dx2, y3 = points[j][1] + dy2;
                        if(x3 >= 0 && y3 >= 0 && p_set.find(HASH(x3, y3)) != p_set.end()){
                            double _area = sqrt(1.0 * (dx1 * dx1 + dy1 * dy1) * (dx2 * dx2 + dy2 * dy2));
                            if(_area < area){
                                area = _area;
                            }
                        }
                    }
                }
            }
        }
        if(area != M * M) return area;
        else return 0;
    }
};
