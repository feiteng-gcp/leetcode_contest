typedef long long int64;
class Solution {
public:
    int64 sqr(int64 x) {
        return x * x;
    }
    int64 dis(vector<int>& a, vector<int>& b) {
        return sqr(a[0] - b[0]) + sqr(a[1] - b[1]);
    }
    double minAreaFreeRect(vector<vector<int>>& p) {
        int n = p.size();
        double ret = 1e300;
        for (int i = 0; i < n; ++i) {

            for (int j = i + 1; j < n; ++j) {
                for (int u = 0; u < n; ++u) {
                    if (u == i || u == j) continue;
                    for (int v = u + 1; v < n; ++v) {
                        if (v == i || v == j) continue;
                        int X1 = p[i][0] + p[j][0];
                        int Y1 = p[i][1] + p[j][1];
                        int X2 = p[u][0] + p[v][0];
                        int Y2 = p[u][1] + p[v][1];
                        if (X1 != X2 || Y1 != Y2) continue;
                        int64 D1 = dis(p[i], p[j]);
                        int64 D2 = dis(p[u], p[v]);
                        if (D1 == D2) {
                            ret = min(ret, sqrt(dis(p[i], p[u])) * sqrt(dis(p[i], p[v])));
                        }
                    }
                }
            }
        }
        if (ret > 1e200) ret = 0;
        return ret;
    }
};
