class Solution {
    int64_t dist(vector<int> &v1, vector<int> &v2) {
        return int64_t(v1[0] - v2[0]) * (v1[0] - v2[0]) + int64_t(v1[1] - v2[1]) * (v1[1] - v2[1]);
    }
    vector<int> sym(vector<int> &i, vector<int> &j, vector<int> &k) {
        return {j[0] + (k[0] - i[0]), j[1] + (k[1] - i[1])};
    }
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        int n = points.size();

        map<int, set<int>> m;
        for (auto &pt : points)
            m[pt[0]].insert(pt[1]);

        double mn = INT_MAX;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    auto a = dist(points[i], points[j]);
                    auto b = dist(points[i], points[k]);
                    auto c = dist(points[j], points[k]);
                    vector<int> d;
                    double area = 0;
                    if (a + b == c) // for right angle
                        d = sym(points[i], points[j], points[k]), area = sqrt(a) * sqrt(b);
                    else if (a + c == b)
                        d = sym(points[j], points[k], points[i]), area = sqrt(a) * sqrt(c);
                    else if (b + c == a)
                        d = sym(points[k], points[i], points[j]), area = sqrt(b) * sqrt(c);
                    else continue;
                    if (m.count(d[0]) && m[d[0]].count(d[1])) {
                        mn = min(area, mn);
                    }
                }
            }
        }
        return mn == INT_MAX ? 0 : mn;
    }
};
