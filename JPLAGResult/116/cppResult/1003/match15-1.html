class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        sort(points.begin(), points.end());
        int n = points.size();
        double ans = 40000 * 40000;

        bool found = 0;
        for (int i = 0; i < n - 3; i++) {
            for (int j = i + 1; j < n - 2; j++) {
                for (int k = j + 1; k < n - 1; k++) {
                    for (int l = k + 1; l < n; l++) {
                        long long x1 = points[i][0], y1 = points[i][1];
                        long long x2 = points[j][0], y2 = points[j][1];
                        long long x3 = points[k][0], y3 = points[k][1];
                        long long x4 = points[l][0], y4 = points[l][1];
                        if (x2 - x1 != x4 - x3) continue;
                        if (y2 - y1 != y4 - y3) continue;
                        if ((x2 - x1) * (x3 - x1) + (y2 - y1) * (y3 - y1)) continue;
                        double a = sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
                        double b = sqrt((x3 - x1) * (x3 - x1) + (y3 - y1) * (y3 - y1));
                        ans = min(ans, a * b);
                        found = 1;
                    }
                }
            }
        }
        return found ? ans : 0;
    }
};
