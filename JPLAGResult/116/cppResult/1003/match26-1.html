const double EPS = 1e-6;

class Solution {
public:
    double sqr(double a)
    {
        return a*a;
    }
    bool isRectangle(vector<int>&a, vector<int>&b, vector<int>&c, vector<int>&d){
        vector<double> centroid(2);
        vector<double> dis(4);

        for (int i = 0; i <= 1; i++)
            centroid[i]=(a[i]+b[i]+c[i]+d[i])/4.0;
        
        dis[0]=sqr(centroid[0]-a[0])+sqr(centroid[1]-a[1]);
        dis[1]=sqr(centroid[0]-b[0])+sqr(centroid[1]-b[1]);
        dis[2]=sqr(centroid[0]-c[0])+sqr(centroid[1]-c[1]);
        dis[3]=sqr(centroid[0]-d[0])+sqr(centroid[1]-d[1]);
        
        return abs(dis[0]-dis[1]) < EPS && abs(dis[0]-dis[2]) < EPS && abs(dis[0]-dis[3]) < EPS;
    }
    
    double distance_square(vector<int>& a, vector<int>& b)
    {
        return sqr(a[0]-b[0]) + sqr(a[1]-b[1]);
    }
    

    double minAreaFreeRect(vector<vector<int>>& points) {
        int n = points.size();
        double ans = 40001 * 40001;
        for (int i = 0; i < n; i++)
            for (int j = i + 1; j < n; j++)
                for (int k = j + 1; k < n; k++)
                    for (int l = k + 1; l < n; l++)
                        if (isRectangle(points[i], points[j], points[k], points[l]))
                        {
                            printf("find rect %d %d %d %d\n", i, j, k, l);
                            double disj = distance_square(points[j], points[i]);
                            double disk = distance_square(points[k], points[i]);
                            double disl = distance_square(points[l], points[i]);
                            double e0, e1;
                            if (disl >= disj && disl >= disk)
                            {
                                e0 = disj;
                                e1 = disk;
                            }
                            else if (disk >= disj && disk >= disl)
                            {
                                e0 = disj;
                                e1 = disl;
                            }
                            else if (disj >= disk && disj >= disl)
                            {
                                e0 = disk;
                                e1 = disl;
                            }
                            
                            double area = sqrt(e0) * sqrt(e1);
                            if (area < ans)
                                ans = area;
                        }
                    
        return ans == 40001 * 40001 ? 0 : ans;
    }
};
