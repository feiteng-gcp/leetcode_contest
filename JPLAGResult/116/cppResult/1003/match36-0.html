class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {

        vector<pair<int, int>> ps;
        for(const vector<int>& p: points)
            ps.push_back(make_pair(p[0], p[1]));
        sort(ps.begin(), ps.end());

        int n = points.size();
        double res = INT_MAX;
        for(int i = 0; i < n; i ++)
            for(int j = i + 1; j < n; j ++)
                for(int k = j + 1; k < n; k ++)
                    for(int l = k + 1; l < n; l ++){
                        vector<pair<int, int>> vec = {ps[i], ps[j], ps[k], ps[l]};
                        res = min(res, get_area(vec));
                    }
        return res == INT_MAX ? 0 : res;
    }

private:
    double get_area(vector<pair<int, int>>& ps){

        double res = INT_MAX;
        res = min(res, get_area(ps[0], ps[1], ps[2], ps[3]));
        res = min(res, get_area(ps[0], ps[1], ps[3], ps[2]));
        res = min(res, get_area(ps[0], ps[2], ps[1], ps[3]));
        res = min(res, get_area(ps[0], ps[2], ps[3], ps[1]));
        res = min(res, get_area(ps[0], ps[3], ps[1], ps[2]));
        res = min(res, get_area(ps[0], ps[3], ps[2], ps[1]));
        return res;
    }

    double get_area(const pair<int, int>& a, const pair<int, int>& b,
                    const pair<int, int>& c, const pair<int, int>& d){
        if(!p(c, b, a) ||
           !p(d, c, b) ||
           !p(a, d, c) ||
           !p(b, a, d))
            return INT_MAX;

        int d1 = dis(b, a),
            d2 = dis(c, b),
            d3 = dis(d, c),
            d4 = dis(a, d);
        if(d1 == d3 && d2 == d4)
            return sqrt((long long)d1 * (long long)d2);
        return INT_MAX;
    }

    bool p(const pair<int, int>& a, const pair<int, int>& b, const pair<int, int>& c){
        pair<int, int> x = make_pair(b.first - a.first, b.second - a.second);
        pair<int, int> y = make_pair(c.first - b.first, c.second - b.second);
        return x.first * y.first + x.second * y.second == 0;
    }

    int dis(const pair<int, int>& a, const pair<int, int>& b){
        return (a.first - b.first) * (a.first - b.first) +
               (a.second - b.second) * (a.second - b.second);
    }
};
