class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        stable_sort(points.begin(), points.end(),
                   [](const vector<int>& a, const vector<int>& b) {
                       if (a[0] != b[0]) return a[0] < b[0];
                       else return a[1] < b[1];
                   });
        double sol = 1.1E10;
        int n = points.size();

        map<pair<int, int>, vector<vector<int>>> mymap;
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                int dx = points[j][0] - points[i][0], dy = points[j][1] - points[i][1];
                mymap[make_pair(dx, dy)].push_back({i, j});
            }
        }
        for (const auto &ah : mymap) {
            // for each angle

            const vector<vector<int>>& ag = ah.second;
            for (int i = 0; i < ag.size(); i++) {
                for (int j = i+1; j < ag.size(); j++) {
                    double thissol = findsol(points[ag[i][0]], points[ag[i][1]], points[ag[j][0]], points[ag[j][1]]);
                    sol = min(sol, thissol);
                }
            }
        }
        if (sol < 1E10) return sol;
        else return 0;
    }
    
    double findsol(const vector<int>& a, const vector<int>& b, const vector<int>& c, const vector<int>& d) {
        if ((c[0] - a[0])*(b[0]-a[0]) + (c[1] - a[1])*(b[1]-a[1]) == 0) return (double)abs((c[0] - a[0])*(b[1] - a[1]) - (c[1] - a[1])*(b[0] - a[0]));
        else return 1.1E10;
    }
};
