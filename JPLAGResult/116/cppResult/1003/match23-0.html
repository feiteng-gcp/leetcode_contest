class Solution {
public:
    struct Point{
        double x, y;
        Point(double x_, double y_){
            x = x_; y=y_;
        }
        Point(){}
        bool operator== (Point& point){
            return (x==point.x) && (y==point.y);
        }
        void operator=(Point point){
            x = point.x; y = point.y;
        }
    };
    static double getDistance(Point& start, Point& end){
            return sqrt((start.x-end.x)*(start.x-end.x)+(start.y-end.y)*(start.y-end.y));
    }
    struct Line{
        public:
        double length;
        Point start;
        Point end;
        Line(Point s, Point e){
            start = s;
            end = e;
            length = getDistance(start, end);
        }
        Point getCenter(){
            return Point((start.x+end.x)/2, (start.y+end.y)/2);
        }
    };
    static bool compare(Line line1, Line line2){
        return line1.length<line2.length;
    }
    double minAreaFreeRect(vector<vector<int>>& points) {
        vector<Line> lines;
        int n = points.size();
        for (int i=1; i<n; ++i){
            for (int j=0; j<i; ++j){
                lines.push_back(Line(Point(points[i][0], points[i][1]), Point(points[j][0], points[j][1])));
            }
        }
        sort(lines.begin(), lines.end(), compare);
        n = lines.size();
        double minArea = 0;
        for (int i=0; i<n; ++i){
            for (int j=i+1; j<n; ++j){
                if (lines[j].length!=lines[i].length){
                    break;
                }else{
                    Point center1 = lines[i].getCenter();
                    Point center2 = lines[j].getCenter();
                    if (center1==center2){
                        // is rectangle. get area
                        double area = getDistance(lines[i].start, lines[j].start)*getDistance(lines[i].end, lines[j].start);
                        if (minArea==0 || minArea>area){
                            minArea = area;
                        }
                    }
                }
            }
        }
        return minArea;
    }
};
