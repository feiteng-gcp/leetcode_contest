typedef pair<int, int> PII;
set<PII> p;
struct Pt {
    int x, y;
    Pt(int x=0, int y=0): x(x), y(y){}
};
Pt pv[55];
// bool cmp(const Point& p1, const Point& p2) {
//     if (p1.x == p2.x) return p1.y < p2.y;
//     return p1.x < p2.x;
// }

class Solution {
    int dot(Pt& p1, Pt& p2) {
        return p1.x * p2.x + p1.y * p2.y;
    }
    int cross(Pt& p1, Pt& p2) {
        return p1.x * p2.y - p1.y * p2.x;
    }
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        p.clear();
        int n = points.size();
        for (int i = 0; i < n; i++) {
            pv[i].x = points[i][0];
            pv[i].y = points[i][1];
            p.insert({points[i][0], points[i][1]});
        }
        double ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    int nx = pv[j].x + pv[k].x - pv[i].x;
                    int ny = pv[j].y + pv[k].y - pv[i].y;
                    // printf("%d %d\n", nx, ny);
                    if (!p.count({nx, ny})) continue;
                    Pt l1 = Pt(pv[j].x - pv[i].x, pv[j].y - pv[i].y);
                    Pt l2 = Pt(pv[k].x - pv[i].x, pv[k].y - pv[i].y);
                    if (dot(l1, l2) == 0) {
                        int area = abs(cross(l1, l2));
                        if (ans > area || ans < 0.5) ans = area;
                    }
                }
            }
        }
        return ans;
    }
};
