class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        double ans = -1;
        for (int i = 0; i < points.size(); ++i) {
            for (int j = i + 1; j < points.size(); ++j) {
                long long dxij = points[i][0] - points[j][0];
                long long dyij = points[i][1] - points[j][1];
                long long dij = dxij * dxij + dyij * dyij;
                for (int p = 0; p < points.size(); ++p) {
                    if (p == i || p == j) continue;
                    for (int q = p + 1; q < points.size(); ++q) {
                        if (q == i || q == j) continue;
                        long long dxpq = points[p][0] - points[q][0];
                        long long dypq = points[p][1] - points[q][1];
                        long long dpq = dxpq * dxpq + dypq * dypq;
                        if (dij == dpq) {
                            long long cx1 = points[i][0] + points[j][0];
                            long long cy1 = points[i][1] + points[j][1];
                            long long cx2 = points[p][0] + points[q][0];
                            long long cy2 = points[p][1] + points[q][1];
                            if (cx1 == cx2 && cy1 == cy2) {
                                long long dxip = points[i][0] - points[p][0];
                                long long dyip = points[i][1] - points[p][1];
                                long long dxiq = points[i][0] - points[q][0];
                                long long dyiq = points[i][1] - points[q][1];
                                long long dipq = dxip * dxiq + dyip * dyiq;
                                if (dipq == 0) {
                                    double l = sqrt(dxip * dxip + dyip * dyip);
                                    double w = sqrt(dxiq * dxiq + dyiq * dyiq);
                                    if (ans < 0 || l * w < ans) ans = l * w;
                                }
                            }
                        }
                    }
                }
            }
        }
        return ans < 0 ? 0 : ans;
    }
};
