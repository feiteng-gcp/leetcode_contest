class Solution {
public:
    double minAreaFreeRect(vector<vector<int>>& points)
    {
        int n = points.size();
        if (n == 0)
        {
            return 0;
        }
        double ret = 1e10;
        for (int i = 0; i < n; ++i)
        {
            for (int j = i + 1; j < n; ++j)
            {
                for (int k = j + 1; k < n; ++k)
                {
                    for (int l = k + 1; l < n; ++l)
                    {
                        vector<int> dot;
                        dot.push_back((points[j][0] - points[i][0]) * (points[k][0] - points[i][0]) + (points[j][1] - points[i][1]) * (points[k][1] - points[i][1]));
                        dot.push_back((points[l][0] - points[i][0]) * (points[k][0] - points[i][0]) + (points[l][1] - points[i][1]) * (points[k][1] - points[i][1]));
                        dot.push_back((points[j][0] - points[i][0]) * (points[l][0] - points[i][0]) + (points[j][1] - points[i][1]) * (points[l][1] - points[i][1]));
                        sort(dot.begin(), dot.end());
                        /*for (auto x : dot)
                        {
                            cout << x << " ";
                        }
                        cout << endl;*/
                        if (dot[0] != 0 || dot[1] == 0 || dot[2] == 0)
                        {
                            continue;
                        }
                        vector<double> length;
                        length.push_back(pow(points[i][0] - points[j][0], 2) + pow(points[i][1] - points[j][1], 2));
                        length.push_back(pow(points[i][0] - points[k][0], 2) + pow(points[i][1] - points[k][1], 2));
                        length.push_back(pow(points[j][0] - points[k][0], 2) + pow(points[j][1] - points[k][1], 2));
                        length.push_back(pow(points[i][0] - points[l][0], 2) + pow(points[i][1] - points[l][1], 2));
                        length.push_back(pow(points[j][0] - points[l][0], 2) + pow(points[j][1] - points[l][1], 2));
                        length.push_back(pow(points[k][0] - points[l][0], 2) + pow(points[k][1] - points[l][1], 2));
                        sort(length.begin(), length.end());
                        if (length[0] == length[1] && length[2] == length[3] && length[4] == length[5] && length[0] + length[2] == length[4])
                        {
                            /*cout << i << " " << j << " " << k << " " << l << endl;
                            for (auto x : length)
                            {
                                cout << x << " ";
                            }
                            cout << endl;*/
                            ret = min(ret, sqrt(length[0] * length[2]));
                        }
                        else
                        {
                            continue;
                        }
                    }
                }
            }
        }
        return ret == 1e10 ? 0 : ret;
    }
};
