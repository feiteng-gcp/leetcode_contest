class Solution {
    const int M = 40001;
public:
    double minAreaFreeRect(vector<vector<int>>& points) {
        int N = points.size();
        if (N < 4) return 0.0;
        
        unordered_set<int> us;
        for (auto& p: points)
            us.insert(p[0] * M + p[1]);
        
        double area = 1.0e20;
        for (int i=0; i<N; i++) {
            for (int j=i+1; j<N; j++) {
                for (int k=j+1; k<N; k++) {
                    int p = (points[j][0] + points[k][0] - points[i][0]) * M + points[j][1] + points[k][1] - points[i][1];
                    double a0 = points[j][0] - points[i][0], a1 = points[j][1] - points[i][1];
                    double b0 = points[k][0] - points[i][0], b1 = points[k][1] - points[i][1];
                    if (us.find(p) == us.end()) continue;
                    if (a0 * b0 + a1 * b1 != 0.0) continue;
                    double tmpArea = fabs(a0 * b1 - a1 * b0);
                    area = min(area, tmpArea);
                }
            }
        }
        return area == 1.0e20 ? 0.0 : area;
    }
};
