class Solution {
public:
    int repeatedNTimes(vector<int>& A) {
        int N = A.size()/2;
        map<int,int> cnt;
        for(auto a: A) {
            cnt[a]++;
        }
        for(auto c: cnt) {
            if(c.second == N) {
                return c.first;
            }
        }
        return 0;
    }
};
