#include <bits/stdc++.h>

using namespace std;

template<typename Ptr, typename CMP>
void Sort(Ptr begin, Ptr end, CMP cmp){
	if(begin == end){
		return ;
	}
	Ptr key = begin;
	Ptr i = begin, j = end; --j;
	while(i < j){
		for(;i < j && cmp(*key, *j); --j){
			;
		}
		if(i < j){
			swap(*key, *j);
		}
		for(;i < j && cmp(*i, *key); ++i){
			;
		}
		if(i < j){
			swap(*key, *i);
		}
	}
	if(begin != i){
		sort(begin, i, cmp);
	}
	++j;
	if(end != j){
		sort(j, end, cmp);
	}
}

template<typename Ptr>
void Sort(Ptr begin, Ptr end){
	typename std::remove_reference<decltype((*begin))>::type tmp;
	Sort(begin, end, std::less<typename std::remove_reference<decltype((*begin))>::type>());
}

class Solution {
public:
	int repeatedNTimes(vector<int>& A) {
		unordered_map<int, int> cnt;
		for(auto p : A){
			cnt[p]++;
			if(cnt[p] == A.size() / 2){
				return p;
			}
		}
		return 0;
	}
};
