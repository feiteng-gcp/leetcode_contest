const int N = 1e4 + 7;

bool vis[N];


class Solution {
public:
    int repeatedNTimes(vector<int>& A) {
		memset(vis, false, sizeof(vis));    
		for (auto x : A) {
			if (vis[x]) return x;
			vis[x] = true;
		}
		return -1;
    }
};
