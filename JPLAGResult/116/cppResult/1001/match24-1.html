class Solution {
public:
    int repeatedNTimes(vector<int>& A) {
        int mid = (A.size() - 1) / 2;
        
        nth_element(A.begin(), A.begin() + mid, A.end());
        
        if (A[mid] == A[mid - 1]) {
            return A[mid];
        } else {
            return A[mid + 1];
        }
    }
};
