class Solution {
public:
    set<int> x;

    vector<pair<int, int> > y;
    int maxWidthRamp(vector<int>& A) {
        x.clear();
        y.clear();
        for(int i = 0;i < A.size();i++){
            y.push_back(make_pair(A[i], i));
        }
        sort(y.begin(), y.end());
        int l = A.size();
        int ans = 0;
        for(int i = 0;i < y.size();i++){
            int val = y[i].first;
            int pos = y[i].second;
            if(pos >= l){
                ans = max(ans, pos - l);
            }else{
                l = pos;
            }
        }
        return ans;
    }
};
