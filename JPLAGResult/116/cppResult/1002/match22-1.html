class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        vector<int> ans(A.size());
        ans[0] = 0;
        for (int i = 1; i<A.size(); i++) {
            if (A[i] == A[i-1]) {
                ans[i] = ans[i-1] + 1;

            } else if (A[i] > A[i-1]) {
                int end_pos = i-1 - ans[i-1];
                for (int j = 0; j <= end_pos; j++) {
                    if (A[j] <= A[i]) {
                        ans[i] = i-j;
                        break;
                    }
                }
            } else {
                int end_pos = i-1 - ans[i-1];
                // cout << i << " " << end_pos << endl;
                for (int j = end_pos; j <= i; j++) {
                    if (A[j] <= A[i]) {
                        ans[i] = i-j;
                        break;
                    }
                }
            }
        }
        int res = 0;
        for (auto e: ans) {
            // cout << e << " ";
            res = max(res, e);
        }
        return res;
    }
};
