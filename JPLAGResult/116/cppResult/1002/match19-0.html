class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        vector<int> a;
        int maxlen = 0;
        for(int i=0;i<A.size();i++)
        {
            if(a.size()==0 || A[i]<A[a.back()])
            {
                a.push_back(i);
            }
            else
            {
                for(auto it:a)
                {
                    if(A[it]<=A[i])
                    {
                        maxlen = max(maxlen, i-it);
                        break;
                    }
                }
            }
            
        }
        return maxlen;
    }
};
