struct E {
    int v, idx;
    E(int v=0, int idx=0): v(v), idx(idx) {}
};
bool cmp (const E& e1, const E& e2) {
    if (e1.v == e2.v) {
        return e1.idx < e2.idx;
    }
    return e1.v < e2.v;
}
const int maxn = 50000 + 10;
E a[maxn];
class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        int n = A.size();
        for (int i = 0; i < n; i++) {
            a[i].v = A[i];
            a[i].idx = i;
        }
        sort(a, a + n, cmp);
        int ans = 0;
        int d = a[0].idx;
        for (int i = 1; i < n; i++) {
            if (d < a[i].idx && a[i].idx - d > ans) {
                ans = a[i].idx - d;
            }
            d = min(d, a[i].idx);
        }
        return ans;
    }
};
