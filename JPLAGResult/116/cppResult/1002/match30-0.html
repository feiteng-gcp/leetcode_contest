class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        vector<int> num;
        vector<int> idx;
        int res = 0;
        for (int i = 0; i < A.size(); ++i) {
            auto lb = lower_bound(num.begin(), num.end(), A[i], greater<int>());
            if (lb == num.end()) {
                num.push_back(A[i]);
                idx.push_back(i);
            } else {
                res = max(res, i - idx[(int((lb - num.begin())))]);
            }
        }
        return res;
    }
};
