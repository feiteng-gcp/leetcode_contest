class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        int n = A.size();
        vector<int> prefixmin(n, 100000);
        prefixmin[0] = A[0];
        for (int i = 1; i < n; ++i) {
            prefixmin[i] = min(prefixmin[i - 1], A[i]);
        }
        int ans = 0;
        for (int j = 1; j < n; ++j) {
            // [0, j - 1]
            int l = 0;
            int r = j - 1;
            while (l < r) {
                int mid = l + (r - l) / 2;
                if (prefixmin[mid] > A[j]) {
                    l = mid + 1;
                } else {
                    r = mid;
                }
            }
            if (prefixmin[l] <= A[j]) {
                ans = max(ans, j - l);
            }
        }
        return ans;
    }
};
