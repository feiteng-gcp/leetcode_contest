class Solution {
public:
  int maxWidthRamp(vector<int>& a) {
    int ans = 0;
    int n = a.size();
    vector<pair<int,int>> b;
    for(int i = 0; i < n; ++i) {
      b.emplace_back(a[i], i);
    }
    
    int last = 2*n;
    
    sort(b.begin(), b.end());
    for(auto const& p : b) {
      int idx = p.second;
      
      ans = max(ans, idx - last);
      
      last = min(idx, last);
    }

    return ans;
  }
};
