class Solution {
public:
   

int maxWidthRamp(vector<int>& a) {
    int res = 0;
    int n = a.size();
    vector<pair<int, int>> v;
    for (int i = 0; i < n; i++)
    	v.push_back({a[i], i});
    sort(v.begin(), v.end());
    int t = v[0].second;
    for (auto x : v) {
    	res = max(res, x.second - t);
    	t = min(t, x.second);
    }
    return res;
}
};
