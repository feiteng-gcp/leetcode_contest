class Solution {
public:
    int bin(vector<int>&p ,int l,int r,int t){

        while(l<=r){
            int m = l+(r-l)/2;
            if(p[m]>t){
                l=m+1;
            }else{
                r=m-1;
            }
        }
        return r;
    }
    int maxWidthRamp(vector<int>& A) {
        int n = A.size();
        if(n==0)
            return 0;

        vector<int> p(n,0);
        p[n-1]=A[n-1];
        for(int i = n-2;i>=0;i--){
            p[i]=max(p[i+1],A[i]);
        }
        int maxv = 0;
        for(int i=0;i<n-1;i++){
            int d = bin(p,i+1,n-1,A[i]-1);
            //printf("%d %d %d %d\n",i,d,A[i],A[d]);
            if(d-i>maxv)
                maxv = d-i;
        }
        
        return maxv;
    }
};
