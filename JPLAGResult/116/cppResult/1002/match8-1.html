class Solution {
public:
    int maxWidthRamp(vector<int>& a) {
        int n = a.size();
        vector<tuple<int,int>> v;
        for(int i=0; i<n;i++){
            v.push_back(make_tuple(a[i],i));
        }
        sort(begin(v),end(v));
        int min_idx = get<1>(v[0]);
        int mx = INT_MIN;
        for(int i=1;i<n;i++){
            mx = max(mx, get<1>(v[i])-min_idx);
            min_idx = min(min_idx,get<1>(v[i]));
        }
        if (mx < 0){
            return 0;
        }
        return mx;
    }
};
