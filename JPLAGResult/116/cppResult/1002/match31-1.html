class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        int ret = INT_MIN;
        int index = INT_MAX;
        map<int, vector<int>> help;
        for(int i=0; i<A.size(); i++)
        {
            if(help.find(A[i]) == help.end())
            {
                vector<int> tmp;
                help[A[i]] = tmp;
            }
            help[A[i]].push_back(i);
        }
        
        map<int, vector<int>>::iterator it = help.begin();
        for(; it!=help.end(); it++)
        {
            vector<int> &tmp = it->second;
            for(int i=0; i<tmp.size(); i++)
            {
                if(tmp[i]<index)
                    index = tmp[i];
                else
                    ret = max(ret, tmp[i]-index);
            }
        }
        return ret==INT_MIN?0:ret;
    }
};
