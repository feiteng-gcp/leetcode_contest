class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
     //   A={9,8,1,0,1,9,4,0,4,1};
        vector<pair<int, int>> p(50001, make_pair(50001, 0));
        for (int i = 0; i < A.size(); ++i) {
            p[A[i]].first = min(p[A[i]].first, i);
            p[A[i]].second = max(p[A[i]].second, i);
        }
        

        int ret = 0;
        for (int i = 49999; i >= 0; --i) {
            p[i].second = max(p[i].second, p[i + 1].second);
            ret = max(ret, p[i].second - p[i].first);
        }
            
        return ret;
    }
};
