class Solution {
public:
    vector<pair<int,int> > v;
    int maxWidthRamp(vector<int>& A) {
        v.clear();
        for(int i=0;i<A.size();i++){
            v.push_back(make_pair(A[i],i));
        }
        sort(v.begin(),v.end());
        int mint=INT_MAX;
        int ans=-1;
        for(int i=0;i<v.size();i++){
            if(mint!=INT_MAX){
                ans=max(ans,v[i].second-mint);
            }
            mint=min(mint,v[i].second);
        }
        if(ans<0)return 0;
        return ans;
    }
};
