class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        stack<int> decStack;
        int maxWidth = 0;
        
        for (int i = 0; i < A.size(); i++) {
            if (decStack.empty() || A[i] < A[decStack.top()]) {
                decStack.push(i);
            }
        }
        
        for (int i = A.size() - 1; i >= 0; i--) {
            while (!decStack.empty() && A[decStack.top()] <= A[i]) {
                maxWidth = max(maxWidth, i - decStack.top());
                decStack.pop();
            }
        }
        
        return maxWidth;
    }
};
