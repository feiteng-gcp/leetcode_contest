class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        vector<pair<int, int>> _A;
        int N = A.size();
        for(int i = 0; i < N; i ++){
            _A.push_back(make_pair(A[i], i));
        }
        sort(_A.begin(), _A.end());
        
        priority_queue<int, vector<int>, greater<int>> Q;

        int width = 0;
        Q.push(_A[0].second);
        
        for(int i = 1; i < N; i ++){
            width = max(width, _A[i].second - Q.top());
            Q.push(_A[i].second);
        }
        
        return width;
    }
};
