struct fff
{
	int num;
	int id;
};

class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        int n=A.size();
	vector<fff> v;
	for(int i=0;i<n;i++)
	{
		fff a;
		a.num=A[i];
		a.id=i;
		v.push_back(a);
	}
	sort(v.begin(),v.end(),cmp);
	int ans=0;
	int minm=n+1;
	for(int i=0;i<n;i++)
	{
		ans=max(ans,v[i].id-minm);
		minm=min(minm,v[i].id);
	}
	return ans;
    }
private:
    static bool cmp(fff a,fff b)
{
	if(a.num==b.num)return a.id<b.id;
	return a.num<b.num;
}
};
