class Solution {
public:
    int maxWidthRamp(vector<int>& a) {
      int n=a.size();
      vector<pair<int,int>> ar;
        for(int i=0;i<n;i++)
        {
            ar.push_back(make_pair(a[i],i));
        }
        sort(ar.begin(),ar.end());
        int o=n;
        int ans=0;
        for(int i=0;i<n;i++)
        {
            o=min(ar[i].second,o);
            ans=max(ans,ar[i].second-o);
        }
        return ans;
    }
};
