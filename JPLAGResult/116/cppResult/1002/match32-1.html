class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
	int n=A.size();
	vector<pair<int,int>> vp;
	for(int i=0; i<n; i++){
		vp.push_back({A[i],i});
	}
	sort(vp.begin(), vp.end());
	vector<int> val(n);
	int s=-1;
	for(int i=n-1; i>=0; i--){
		s=max(s,vp[i].second);
		val[i]=s;
	}

	int ans=0;
	for(int i=0; i<n; i++){
		ans=max(ans,val[i]-vp[i].second);
	}

	return ans;
}};
