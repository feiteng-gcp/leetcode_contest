class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        vector<pair<int,int>> vp;
        for(int i=0;i<A.size();i++) vp.push_back({A[i],i});
        sort(vp.begin(),vp.end());
        int best = 0, maxj = -1;
        for(int i=vp.size()-1;i>=0;i--){
            maxj=max(maxj,vp[i].second);
            best=max(best,maxj-vp[i].second);
        }
        return best;
    }
};
