class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        deque<int> st;
        st.push_back(0);
        int mx = 0;
        for (int i = 1; i < A.size(); i++) {
            if (A[i] < A[st.back()]) {
                st.push_back(i);
            }
            else {
                for (int j = st.size() - 1; j >= 0; j--) {
                    if (A[i] >= A[st[j]])
                        mx = max(mx, i - st[j]);
                    else
                        break;
                }
            }
        }
        return mx;
    }
};
