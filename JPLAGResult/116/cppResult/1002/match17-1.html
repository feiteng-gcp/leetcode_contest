class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        int mn = 1<<30;
        int best = 0;
        vector<pair<int,int>> B;
        for (int i = 0; i < A.size(); ++i) {
            B.push_back({A[i],i});
        }
        sort(B.begin(), B.end());
        for (auto p : B) {
            best = max(best,p.second-mn);
            mn = min(p.second,mn);
        }
        return best;
    }
};
