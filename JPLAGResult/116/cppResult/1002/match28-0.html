class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
        int res = 0;
        unordered_map<int,int> m;
        int minnum = INT_MAX;
        for(int i=0;i<A.size();i++){
            if(A[i] >= minnum)continue;
            minnum = A[i];
            int j = A.size()-1;
            while(i<j){
                if(A[i]<=A[j]){
                    res = max(res,j-i);
                    break;
                }
                j--;
            }
        }
        return res;
    }
};
