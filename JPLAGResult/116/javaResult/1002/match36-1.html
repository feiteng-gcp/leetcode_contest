class Solution {
    public int maxWidthRamp(int[] A) {
        int max = 0;
        for (int i = 0; i < A.length - 1; i++) {
            for (int j = A.length - 1; j > i + max; j--) {
                if (A[i] <= A[j]) {
                    max = j - i;
                    
break;
                }
            }
        }
        return max;
    }
}
