class Solution {
    public int maxWidthRamp(int[] A) {
        if (A.length < 2) return 0;
        int result = 0;
        for (int i = 1; i < A.length; i++){
            for (int j = 0; j < i; j++){
                if (A[i] >= A[j]){
                    result = Math.max(result, i - j);
                    break;
                }    
            }
        }
        return result;
    }
}
