class Solution {
    public int maxWidthRamp(int[] A) {
        int ans = 0;
        for (int i = 0; i < A.length; ++i){
            for (int j = A.length - 1; i + ans < j; --j){
                if (A[i] <= A[j]){
                    ans = j - i;
                    break;
                }
            }
        }
        return ans;
    }
}
