class Solution {
    public int maxWidthRamp(int[] A) {
        TreeMap<Integer,Integer> map = new TreeMap<>();
        int gap = 0;
        for (int i = 0; i < A.length; i++) {
            Map.Entry<Integer,Integer> entry = map.floorEntry(A[i]);
            if(entry == null){
                map.put(A[i],i);
            }else {
                gap = Math.max(gap, i - entry.getValue());
            }
        }
        return gap;
    }
}
