class Solution {
    public int maxWidthRamp(int[] A) {
        int len = 0;
        for (int i=0; i<A.length; i++) {
            for (int j=0; j<i; j++) {
                if (A[j] <= A[i]) {
                    len = Math.max(len, i-j);
                    break;
                }
            }
        }
        return len;
    }
}
