class Solution {
    public int maxWidthRamp(int[] a) {
        if (a == null) return 0;
        int m = 0;
        for (int i = 1; i < a.length; i++) {
            for (int j = 0; j < i; j++) {
                if (a[j] <= a[i]) {
                    m = Math.max(m, i - j);
                    break;
                }
            }
        }
        
        return m;
    }
}
