class Solution {
    public int maxWidthRamp(int[] A) {
        List<Integer> list = new LinkedList<>();
        int result = 0;
        for (int i = 0; i < A.length; i++) {
            if (list.size() == 0 || A[list.get(list.size() - 1)] > A[i]) {
                list.add(i);
            } else {
                int j = list.size() - 1;
                while (j >= 0 && A[list.get(j)] <= A[i]) {
                    result = Math.max(result, i - list.get(j));
                    j--;
                }
            }
        } 
        return result;
    }
}
