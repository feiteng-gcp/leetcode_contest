class Solution {
    public int maxWidthRamp(int[] A) {
		TreeMap<Integer, Integer> tmap = new TreeMap<>();
		int max = 0;
		for (int i = 0; i < A.length; i++) {
			int val = A[i];
			Map.Entry<Integer, In
teger> floorEntry = tmap.floorEntry(val);
			if (floorEntry == null) {
				tmap.put(val, i);
			}
			else {
			// we found a ramp
			max = Math.max(max, i - floorEntry.getValue());
			}
		}
		return max;
    }
}
