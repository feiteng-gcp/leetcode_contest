class Solution {
    public int maxWidthRamp(int[] A) {
        int max = 0;
        for (int l = 0; l < A.length; l++) {
            for (int r = l + max + 1; r < A.length; r++) {
                if (A[l] <= A[r]) {
                    max = r - l;
                }
            }
        }
        return max;
    }
}
