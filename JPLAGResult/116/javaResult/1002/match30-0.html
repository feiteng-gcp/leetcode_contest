class Solution {
    public int maxWidthRamp(int[] A) {
        int maxvalue = 0;
        int maxindex = 0;
        for(int i = 0 ; i < A.length ; i ++){
        
            for(int j = A.length-1 ; j>=i&&(j-i>=maxindex) ; j--){
                if(A[j]>=A[i] && (j-i>maxindex)){
                    //maxvalue = A[j] - A[i];
                    maxindex = j-i;
                }
                    
            }
            
            
            
        }
        return maxindex;
    }
}
