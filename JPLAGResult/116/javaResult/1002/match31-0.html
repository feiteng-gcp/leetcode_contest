class Solution {
    public int maxWidthRamp(int[] A) {
        int len = A.length;
        int max = 0;
        for (int i = 0; i < len - max; i++) {
            for (int j = len - 1; j > i; j--) {
                if (j - i <= max) break;
                if (A[i] <= A[j]) {
                    max = Math.max(max, j - i);
                    break;
                }
            }
        }
        return max;
    }
}
