class Solution {
    public int maxWidthRamp(int[] A) {
        int ret = 0;
        for (int i = 0; i < A.length; ++i) {
            for (int j = A.length - 1; j > i; --j) {
                if (A[j] >= A[i]) {
                    ret = Math.max(ret, j - i);
                    break;
                }
            }
        }
        return ret;
    }
}
