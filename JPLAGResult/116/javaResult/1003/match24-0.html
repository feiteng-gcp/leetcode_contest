class Solution {
    public double minAreaFreeRect(int[][] points) {
        int len = points.length;
        if(len<=3){
            return 0;
        }
        double minArea = 0;
        boolean find = false;
        for(int i = 0; i<=len-1;i++){
            for(int j = 1;j<=len-1; j++){
                for(int k = 1;k<=len-1;k++){
                    for(int l = 1;l<=len-1;l++){
                        if(i==j || i==k || i==l || j==k || j==l || k==l){
                            continue;
                        } else {
                            if (points[j][0]-points[i][0] == points[k][0]-points[l][0]
                               && points[j][1]-points[i][1] == points[k][1]-points[l][1]
                               && points[j][0]-points[i][0] == points[k][0]-points[l][0]
                               && points[j][1]-points[i][1] == points[k][1]-points[l][1]){
                                double s1 = dis(points[i][0],points[i][1],points[j][0],points[j][1]);
                                double s2 = dis(points[j][0],points[j][1],points[k][0],points[k][1]);
                                double d1 = dis(points[i][0],points[i][1],points[k][0],points[k][1]);
                                if(d1-s1-s2<0.0001 && d1-s1-s2>-0.0001){
                                    //System.out.println(i+", "+j+", "+k+", "+l);
                                    double curArea = Math.sqrt(s1)*Math.sqrt(s2);
                                    if(!find){
                                        find = true;
                                        minArea = curArea;
                                    }
                                    else if(curArea<minArea){
                                        minArea = curArea;
                                    }
                                    //System.out.println(points[j][0]+"," + points[i][0]+", "+points[k][0] +", " +points[l][0]+", " +curArea);
                                }
                            }
                        }
                    }
                }
            }
        }
        if(find){
            return minArea;
        } else {
            return 0;
        }
    }
    
    private double dis(int x1,int y1, int x2, int y2){
        return (x2-x1)*(x2-x1)+(y2-y1)*(y2-y1);
    }
    
}
