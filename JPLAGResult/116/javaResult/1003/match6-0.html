class Solution {
    public double minAreaFreeRect(int[][] points) {
        double ans = -1;
        Set<Integer> set = new HashSet<>();
        for (int[] p : points) {
            set.add((p[0] << 16) | p[1]);
        }
        
        int n = points.length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int x0 = points[j][0] - points[i][0];
                int y0 = points[j][1] - points[i][1];
                if (i != j) {
                    for (int k = 0; k < n; k++) {
                        if (i != k && j != k) {
                            int x1 = points[k][0] - points[i][0];
                            int y1 = points[k][1] - points[i][1];
                            if (x0 * x1 + y0 * y1 == 0) {
                                int x = points[j][0] + points[k][0] - points[i][0];
                                int y = points[j][1] + points[k][1] - points[i][1];
                                if (set.contains((x << 16) | y)) {
                                    double cur = Math.sqrt(x0 * x0 + y0 * y0) * Math.sqrt(x1 * x1 + y1 * y1);
                                    if (ans < 0 || ans > cur) {
                                        ans = cur;
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        return ans < 0 ? 0 : ans;
    }
}
