class Solution {
        public double minAreaFreeRect(int[][] points) {
            if (points == null || points.length < 4) {
                return 0;
            }
            
double result = 0;
            for(int i=0;i<points.length-3;i++) {
                for(int j=i+1;j<points.length-2;j++) {
                    for(int k=j+1;k<points.length;k++) {
                        for(int z=k+1;z<points.length;z++) {
                            int[][] sides = isRect(points[i], points[j], points[k], points[z]);
                            if (sides != null) {
                                double area = area(sides[0], sides[1]);
                                result = (result==0||result>area)?area:result;
                            
}
                        }
                    }
                }
            }

            return result;

        
}

        private int[][] isRect(int[] points0,int[] points1,int[] points2,int[] points3) {
            int[] a=getVector(points0,points1);
            int[] b=getVector(points0,points2);
            int[] c=getVector(points0,points3);
            if (dotMultiply(a, b)==0) {
                //1,2
                return isEqual(b,getVector(points1, points3))?new int[][]{a, b}:null;
            } else if (dotMultiply(a, c) == 0) {
                //1,3
                return isEqual(c, getVector(points1, points2))?new int[][]{a, c}:null;

            } else if (dotMultiply(b, c) == 0) {
                //2,3
                return isEqual(c, getVector(points2, points1))?new int[][]{b, c}:null;
            
}
            return null;
        }

        private int dotMultiply(int[] a, int[] b) {
            return a[0]*b[0] + a[1]*b[1];
        }

        private boolean isEqual(int[] a, int[] b) {
            return a[0] == b[0] && a[1] == b[1];
        }

        private int[] getVector(int[] a, int[] b) {
            return new int[]{b[0] - a[0], b[1] - a[1]};
        }

        private double area(int[] a, int[] b) {
            return Math.abs(a[0] * b[1] - a[1] * b[0]);
        }

}
