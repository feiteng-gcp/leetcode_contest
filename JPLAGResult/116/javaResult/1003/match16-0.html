class Solution {
    public double minAreaFreeRect(int[][] points) {
        Arrays.sort(points, (x, y) -> x[0] - y[0] != 0 ? x[0] - y[0] : x[1] - y[1]);
        double min = 1.7e9;
        int len = points.length;
        double temp;
        for(int i = 0; i < len - 3; i++) {
            for(int j = i + 1; j < len - 2; j++) {
                for(int k = j + 1; k < len - 1; k++) {
                    for(int w = k + 1; w < len; w++) {
                        temp = areaIfRec(points[i], points[j], points[k], points[w]);
                        if(temp != -1) {
                            min = Math.min(min, temp);
                            break;
                        }
                    }
                }
            }
        }
        return min == 1.7e9 ? 0 : min;
    }
    
    public double areaIfRec(int[] p1, int[] p2, int[] p3, int[] p4) {
        double w = dist(p1, p2), h = dist(p1, p3);
        if(w == dist(p3, p4) && h == dist(p2, p4) && dist(p1, p4) == dist(p2, p3)) {
            return Math.sqrt(w) * Math.sqrt(h);
        }
        return -1;
    }
    
    public double dist(int[] p1, int[] p2) {
        return Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2);
    }
}
