class Solution {
    public double minAreaFreeRect(int[][] points) {
        List<Point> list = new ArrayList<>();
        Set<Point> set = new HashSet<>();
        for (int[] p : points) {
            Point point = new Point(p[0], p[1]);
            list.add(point);
            set.add(point);
        }

        double res = Double.MAX_VALUE;
        Collections.sort(list);
        //System.out.println(list);
        //System.out.println(set);
        int n = list.size();
        for (int i = 0 ; i < n; i++) {
            Point p1 = list.get(i);
            for (int j = i + 1; j < n; j++) {
                Point p2 = list.get(j);
                for (int k = j + 1; k < n; k++) {
                    Point p3 = list.get(k);
                    if (p1.x == p2.x) {
                        if (p3.y == p1.y && set.contains(new Point(p3.x, p2.y))) {
                            System.out.println(Arrays.asList(i, j, k));
                            res = Math.min(res, (p2.y - p1.y) * (p3.x - p1.x));
                        }
                    } else {
                        int v = Math.abs((p2.y - p1.y) * (p3.y - p1.y));
                        if ( v % (p2.x - p1.x) == 0) {
                            v /= (p2.x - p1.x);
                            //System.out.println(Arrays.asList(i, j, k));
                            //System.out.println(Arrays.asList(v, p3.x - p1.x));
                            if (v % (p3.x - p1.x) == 0) {
                                v /= (p3.x - p1.x);
                                
                                if (v == 1 || v == -1) {
                                    //System.out.println(new Point(p3.x + p2.x - p1.x, p3.y + p2.y - p1.y));
                                    Point point = new Point(p3.x + p2.x - p1.x, p3.y + p2.y - p1.y);
                                    //Point point = new Point(2, 1);
                                    if (set.contains(point)) {
                                        System.out.println(Arrays.asList(i, j, k));
                                        double a = Math.sqrt((p2.y - p1.y) * (p2.y - p1.y) + (p2.x - p1.x) * (p2.x - p1.x));
                                        double b = Math.sqrt((p3.y - p1.y) * (p3.y - p1.y) + (p3.x - p1.x) * (p3.x - p1.x));
                                        res = Math.min(res, a * b);
                                    }
                                }
                            }
                        }
                        // double k1 = (p2.y - p1.y) * 1.0 / (p2.x - p1.x);
                        // double k2 = (p3.y - p1.y) * 1.0 / (p3.x - p1.x);
                        
                    }
                }
            }
        }
        if (res == Double.MAX_VALUE) return 0;
        return res;
    }
    
    static class Point implements Comparable<Point> {
        int x, y;
        public Point(int i, int j) {
            x = i;
            y = j;
        }
        
        public int hashCode() {
            //System.out.println(31 * x + y);
            return x + y;
        }
        
        //@override
        public boolean equals(Object obj) {
            Point that = (Point) obj;
            //System.out.println("equal" + that.toString());
            return this.x == that.x && this.y == that.y;
        }
        
        public int compareTo(Point that) {
            //System.out.println("compare");
            if (this.x != that.x) {
                return this.x - that.x;
            } else {
                return this.y - that.y;
            }
        }
        
        public String toString() {
            return x + "," + y;
        }
    }
}
