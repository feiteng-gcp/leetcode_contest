class Solution {
    public double minAreaFreeRect(int[][] points) {
        
        HashSet<int[]> set = new HashSet<>();
        for (int[] n : points) set.add(n)
;
        double res = Double.MAX_VALUE;
        for (int i = 0;i < points.length;i++) {
            for (int j = 0;j < points.length;j++) {
                for (int k = 0;k < points.length;k++) {
                    if (i != j && j != k && i != k) 
{
                        int[] a = points[i], b = points[j], c = points[k];
                        int x1 = (a[0] - b[0]), y1 = (a[1] - b[1]);
                        int x2 = (a[0] - c[0]), y2 = (a[1] - c[1]);
                        int mul = x1*x2 + y1*y2;
                        
                        if (mul != 0) continue;
                        
                        
                        int dx = c[0] - (a[0] - b[0]);
                        int dy = c[1] - (a[1] - b[1]);
                        long l1 = x1*x1 + y1*y1;
                        long l2 = x2*x2 + y2*y2;
                        // System.out.println(l1 + " " + l2);
                        int[] tmp = {dx, dy};
                        
                        for (int[] p : points) {
                            if (p[0] == dx && p[1] == dy) {
                                // System.out.println("(" + a[0] + "," + a[1] + ")");
                                // System.out.println("(" + b[0] + "," + b[1] + ")");
                                // System.out.println("(" + c[0] + "," + c[1] + ")");
                                // System.out.println("(" + tmp[0] + "," + tmp[1] + ")");
                                // System.out.println("----------");
                                res = Math.min(Math.sqrt(l1)*Math.sqrt(l2), res);
                            }
                        }
                    
}
                }
            }
        }
        return res == Double.MAX_VALUE ? 0 : res;
    }
}
