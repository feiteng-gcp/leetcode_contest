class Solution {
    class cmp implements Comparator<int[]> {
        public int compare(int[] a, int[] b) {
            if (a[0] < b[0]) {
                return -1;
            } else if (a[0] == b[0] && a[1] < b[1]) {
                return -1;
            }
            return 1;
        }
    }
    public int dis(int[] a, int[] b) {
        int d1 = (a[0] - b[0])*(a[0] - b[0]) + (a[1] - b[1]) * (a[1] - b[1]);
        return d1;
    }
    public double minAreaFreeRect(int[][] points) {
        
int n = points.length;
        double res = 0;
        for(int i = 0; i < n - 3; ++i) {
            for(int j = i + 1; j < n - 2; ++j) {
                for(int u = j + 1; u < n - 1; ++u) {
                    for(int v = u + 1; v < n; ++v) {
                        int[][] work = new int[4][2];
                        work[0] = points[i];
                        work[1] = points[j];
                        work[2] = points[u];
                        work[3] = points[v];
    
                        Arrays.sort(work, new cmp());
                        
                        // for(int k = 0; k < 4; ++k) {
                        //     System.out.println(work[k][0]+" "+work[k][1]);
                        // }
                        
                        int d1 = dis(work[3], work[0]);
                        int d2 = dis(work[2], work[1]);
                        int dp1 = dis(work[0], work[1]);
                        
int dp2 = dis(work[2], work[3]);
                        int dp3 = dis(work[0], work[2]);
                        int dp4 = dis(work[1], work[3]);
                        if (d1 == d2 && dp1 == dp2 && dp3 == dp4) {
                            
                            double p1 = Math.sqrt(dp1);
                            double p2 = Math.sqrt(dp3);
                            if (res < 1e-5 || res > p1 * p2) {
                                res = p1 * p2;
                            
}
                        }
                    }
                }
            }
        }
        return res;
    }
}
