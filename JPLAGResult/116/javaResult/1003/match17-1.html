class Solution {
    public double minAreaFreeRect(int[][] points) {
        
double result = 0.0;
        
        for (int i = 0; i < points.length - 3; i++) {
            for (int j = i + 1; j < points.length - 2; j++) {
                for (int m = j + 1; m < points.length - 1; m++) {
                    for (int n = m + 1; n < points.length; n++) {
                        double area = getA(points[i], points[j], points[m], points[n]);
                        
                        if (area != -1 && (result == 0.0 || area < result)) {
                            result = area;
                        
}
                    }
                }
            }
        }
        
        return result;
    }
    
    private double getA(int[] p1, int[] p2, int[] p3, int[] p4) {
        if (dis(p1, p2) == dis(p3, p4) && mid(p1, p2, p3, p4)) {
            return Math.sqrt(dis(p1, p3)) * Math.sqrt(dis(p1, p4));
        }
        
        if (dis(p1, p3) == dis(p2, p4) && mid(p1, p3, p2, p4)) {
            return Math.sqrt(dis(p1, p2)) * Math.sqrt(dis(p1, p4));
        }
        
        if (dis(p1, p4) == dis(p2, p3) && mid(p1, p4, p2, p3)) {
            return Math.sqrt(dis(p1, p3)) * 
Math.sqrt(dis(p1, p2));
        }
        
        return -1;
    }
    
    private boolean mid(int[] p1, int[] p2, int[] p3, int[] p4) {
        if (p1[0] + p2[0] == p3[0] + p4[0] && p1[1] + p2[1] == p3[1] + p4[1]) {
            return true;
        }
        
        return false;
    }
    
    private double dis(int[] p1, int[] p2) {
        return (p1[0] - p2[0]) * (p1[0] - p2[0]) + (p1[1] - p2[1]) * (p1[1] - p2[1]);
    }
}
