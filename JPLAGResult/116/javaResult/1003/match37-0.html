class Solution {
    public double minAreaFreeRect(int[][] points) {
        int len = points.length;
        double result = Double.MAX_VALUE;
        for(int i = 0; i < len; ++i) {
            for (int j = i+1; j < len; ++j) {
                for (int k = j+1; k < len; ++k) {
                    if (isZhijiao(points[i], points[j], points[k])) {
                        for (int l = k + 1; l < len; ++l) {
                            result = Math.min(result, isSquare(points[i], points[j], points[k], points[l]));
                        }
                    }
                }
            }
        }
        if (result == Double.MAX_VALUE) {
            return 0;
        } else {
            return result;
        }
    }

    public boolean isZhijiao(int p1[], int p2[], int p3[]) {
        int p[][] = {{p1[0], p1[1]}, {p2[0], p2[1]}, {p3[0], p3[1]}};
        int cnt = 0;
        double len[][] = new double[3][3];
        for (int i = 0; i <= 2; i++) {
            for (int j = i + 1; j <= 2; j++) {
                double[] arr = new double[3];
                arr[0] = (p[i][0] - p[j][0]) * (p[i][0] - p[j][0]) + (p[i][1] - p[j][1]) * (p[i][1] - p[j][1]);
                arr[1] = i;
                arr[2] = j;
                len[cnt++] = arr;
            }
        }
        Arrays.sort(len, new Comparator<double[]>() {
            @Override
            public int compare(double[] o1, double[] o2) {
                return Double.compare(o1[0], o2[0]);
            }
        });
        if (len[2][0] == len[0][0] + len[1][0]) {
            return true;
        }
        return false;
    }


    public double isSquare(int p1[], int p2[], int p3[], int p4[]) {
        int p[][] = {{p1[0], p1[1]}, {p2[0], p2[1]}, {p3[0], p3[1]}, {p4[0], p4[1]}};
        int cnt = 0;
        double len[][] = new double[6][3];
        for (int i = 0; i <= 3; i++) {
            for (int j = i + 1; j <= 3; j++) {
                double[] arr = new double[3];
                arr[0] = (p[i][0] - p[j][0]) * (p[i][0] - p[j][0]) + (p[i][1] - p[j][1]) * (p[i][1] - p[j][1]);
                arr[1] = i;
                arr[2] = j;
                len[cnt++] = arr;
            }
        }
        Arrays.sort(len, new Comparator<double[]>() {
            @Override
            public int compare(double[] o1, double[] o2) {
                return Double.compare(o1[0], o2[0]);
            }
        });
        if (len[0][0] == len[1][0] && len[2][0] == len[3][0]
                && len[4][0] == len[5][0] && len[4][0] == len[2][0] + len[1][0]
                &&(len[4][1] != len[5][1] && len[4][1] != len[5][2] && len[5][2] != len[4][2] && len[5][1] != len[4][2])) {
            return Math.sqrt(len[0][0] * len[2][0]);
        }
        return Double.MAX_VALUE;
    }
}
