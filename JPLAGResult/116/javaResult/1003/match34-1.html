class Solution {
    public double minAreaFreeRect(int[][] points) {
        int len = points.length;
        Set<String> set = new HashSet<>();
        for (int[] p: points) {
            set.add(p[0] + "-" + p[1]);
        }
        
double min = 0;
        boolean has = false;
        for (int i = 0; i < len-2; i++) {
            for (int j = i+1; j < len-1; j++) {
                for (int k = j+1; k < len; k ++) {
                    if ((points[i][0]-points[j][0]) * (points[k][0]-points[i][0]) == (points[j][1]-points[i][1]) * (points[k][1]-points[i][1])) {
                        int px = points[j][0] + points[k][0] - points[i][0];
                        int py = points[j][1] + points[k][1] - points[i][1];
                        
if (set.contains(px + "-" + py)) {
                            int sl1 = Math.abs(points[i][0]-points[j][0]);
                            int sl2 = Math.abs(points[i][1]-points[j][1]);
                            double l1 = Math.sqrt(sl1*sl1+sl2*sl2);
                            sl1 = Math.abs(points[k][0]-points[i][0]);
                            sl2 = Math.abs(points[k][1]-points[i][1]);
                            double l2 = Math.sqrt(sl1*sl1+sl2*sl2);
                            double cur = l1*l2;
                            if (!has) {min = cur; has=true;}
                            else {
                                if (cur < min) min = cur;
                            }
                        
}
                    }
                }
            }
        }
        return min;
    }
}
