class Solution {
    public double minAreaFreeRect(int[][] points) {
        int N = points.length;
        double res = Long.MAX_VALUE;
        
boolean f = false;
        for (int i = 0; i < N; i++) {
            for (int j = i + 1; j < N; j++) {
                for (int k = j + 1; k < N; k++) {
                    for (int l = k + 1; l < N; l++) {
                        if (isRect(points[i], points[j], points[k], points[l])) {
                            long d1 = dis(points[i], points[j]);
                            long d2 = dis(points[i], points[k]);
                            long d3 = dis(points[i], points[l]);
                            f = true;
                            if (d1 > d2 && d1 > d3) {
                                res = Math.min(Math.sqrt(d2) * Math.sqrt(d3), res);
                            } else if (d2 > d1 && d2 > d3) {
                                res = Math.min(Math.sqrt(d1) * Math.sqrt(d3), res);
                            } else {
                                res = Math.min(Math.sqrt(d1) * Math.sqrt(d2), res);
                            }
                        
}
                    }
                }
            }
        }
        if (f) {
            return res;
        }
        
return 0;
    }

    boolean isRect(int[] a, int[] b, int[] c, int[] d) {
        if (dis(a, b) != dis(c, d)) {
            return false;
        }
        if (dis(a, c) != dis(b, d)) {
            return false;
        }
        if (dis(a, d) != dis(b, c)) 
{
            return false;
        }
        return true;
    }

    long dis(int[] a, int[] b) {
        return (long) (a[0] - b[0]) * (a[0] - b[0]) + (a[1] - b[1]) * (a[1] - b[1]);
    }
}
