class Solution {
    public double minAreaFreeRect(int[][] p) {
        Arrays.sort(p, (a, b) -> a[1] == b[1] ? a[0]-b[0] : b[1]-a[1]);
        
        double m = 40000*40000;
        int n = p.length;
        for (int a = 0; a <= n - 4; a++) {
            for (int b = a+1; b <= n-3; b++) {
                int ab = dist(p[a], p[b]);
                double dab = Math.sqrt(ab);
                for (int c = b+1; c <= n-2; c++) {
                    for (int d = c+1; d <= n-1; d++) {
                        int cd = dist(p[c], p[d]);
                        int ac = dist(p[a], p[c]),
                            bc = dist(p[b], p[c]),
                            ad = dist(p[a], p[d]),
                            bd = dist(p[b], p[d]);
                        
                        if (ac == bd && ab == cd && ab + bd == ad && ab + ac == bc) {
                            // System.out.println(a + " " + b + " " + c + " " + d + " = " + ab + " * " + ac + " || ");
                            m = 
Math.min(m, dab * Math.sqrt(ac));
                        }
                    }
                }
            }
        }
        return m == 40000*40000 ? 0 : m;
    }
    
    int dist(int[] a, int[] b) {
        return (a[0]-b[0]) * (a[0]-b[0]) + (a[1]-b[1]) * (a[1]-b[1]);
    }
}
