class Solution {
    public double minAreaFreeRect(int[][] points) {
        Arrays.sort(points, (a, b) -> a[0] != b[0] ? a[0] - b[0] : a[1] - b[1]);
        Double result = null;
        int n = points.length;
        for (int i = 0; i < n; i++) {
            int[] p1 = points[i];
            for (int j = i + 1; j < n; j++) {
                int[] p2 = points[j];
                for (int k = j + 1; k < n; k++) {
                    int[] p3 = points[k];
                    if (!isVertical(p2, p1, p1, p3)) {
                        continue;
                    }
                    for (int l = k + 1; l < n; l++) {
                        int[] p4 = points[l];
                        if (isVertical(p2, p4, p4, p3) && isVertical(p4, p3, p3, p1)) {
                            result = result == null ? area(p1, p2, p4, p3) : Math.min(result, area(p1, p2, p4, p3));
                        } 
                    }
                }
            }
        }
        return result == null ? 0 : result;
    }
    private boolean isVertical(int[] p1, int[] p2, int[] p3, int[] p4) {
        int x1 = p1[0], y1 = p1[1];
        int x2 = p2[0], y2 = p2[1]; 
        int x3 = p3[0], y3 = p3[1];
        int x4 = p4[0], y4 = p4[1];
        
        int line1x = x2 - x1;
        int line1y = y2 - y1;
        int line2x = x4 - x3;
        int line2y = y4 - y3;
        
        return line1x * line2x + line1y * line2y == 0;
    }
    private double area(int[] p1, int[] p2, int[] p3, int[] p4) {
        int x1 = p1[0] - p2[0];
        int y1 = p1[1] - p2[1];
        int x2 = p4[0] - p2[0];
        int y2 = p4[1] - p2[1];
        return Math.abs((double) x1 * y2 - x2 * y1);
    }
}
