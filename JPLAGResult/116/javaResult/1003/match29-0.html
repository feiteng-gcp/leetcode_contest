class Solution {
    public double minAreaFreeRect(int[][] points) {
        int l = points.length;
        double re = 1600000001;
        for(int i = 0; i < l; i++) {
            for(int j = i + 1; j < l; j++) {
                for(int k = j + 1; k < l; k++) {
                    for(int m = k + 1; m < l; m++) {
                        if(isRect(points, i, j, k, m)) {
                            re = Math.min(re, get(points, i, j, k, m));
                        }
                    }
                }
            }
        }
        if(re == 1600000001) return 0.0;
        return re;
    }
    boolean isRect(int[][] points, int i, int j, int k, int m) {
        if(isorth(points, i, j, m)) return isorth(points, m, i, k) && isorth(points, j, i, k);
        if(isorth(points, i, j, k)) return isorth(points, k, i, m) && isorth(points, j, i, m);
        if(isorth(points, i, m, k)) return isorth(points, k, i, j) && isorth(points, m, i, j);
        return false;
    }
    boolean isorth(int[][] points, int i, int j, int k) {
        return (points[i][0] - points[j][0]) * 1l * (points[i][0] - points[k][0]) + (points[i][1] - points[j][1]) * 1l * (points[i][1] - points[k][1]) == 0;
    }
    double get(int[][] points, int i, int j, int k, int m) {
        if(isorth(points, i, j, m)) {
            k = j;
        }
        else if(isorth(points, i, j, k)) m = j;
        //if(isorth(points, i, m, k))
        double dx1 = points[i][0] - points[m][0];
        double dx2 = points[i][0] - points[k][0];
        double dy1 = points[i][1] - points[m][1];
        double dy2 = points[i][1] - points[k][1];
        return Math.sqrt((dx1 * dx1 + dy1 * dy1) * (dx2 * dx2 + dy2 * dy2));
    }
}
