class Solution {
    public double minAreaFreeRect(int[][] points) {
        double area = Double.MAX_VALUE;
        int n = points.length;
        
        Set<Long> cl = new HashSet<Long>();
        
        for (int[] p: points) {
            long key = (long)p[0] * 40000 + p[1];
            cl.add(key);    
        }
        
        
for (int p1 = 0; p1 < n; ++p1) {
            for (int p2 = 0; p2 < n; ++p2) {
                if (p2 == p1) continue;
                for (int p3 = 0; p3 < n; ++p3) {
                    if (p3 == p1 || p3 == p2) continue;
                    int x1 = points[p1][0] - points[p2][0];
                    int y1 = points[p1][1] - points[p2][1];
                    int x2 = points[p3][0] - points[p2][0];
                    int y2 = points[p3][1] - points[p2][1];
                    long dot = dotProd(x1, y1, x2, y2);
                    if (dot == 0) { // Perpendicular angle
                        int dx23 = points[p3][0] - points[p2][0];
                        int dy23 = points[p3][1] - points[p2][1];
                        int p4x = points[p1][0] + dx23;
                        int p4y = points[p1][1] + dy23;
                        long p4Key = (long)p4x * 40000 + p4y;
                        if (cl.contains(p4Key)) { // parallelogram
                            double cp = cross(x1, y1, x2, y2);
                            double curArea = Math.abs(cp);
                            area = Math.min(area, curArea);            
                        }
                    }
                }
            }
        }
        
        return area < Double.MAX_VALUE ? area : 0;
    }
    
    long dotProd(int x1, int y1, int x2, int y2) {
        return (long)x1 * x2 + y1 * y2;
    }
    
    int cross(int x1, int y1, int x2, int y2) {
        
return y2 * x1 - x2 * y1;
    }
    
    double det(int p1, int p2, int p3, int[] points) {
        return 0;
    }
}
