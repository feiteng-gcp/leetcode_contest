class Solution {
    public double minAreaFreeRect(int[][] points) {
        double min = Double.MAX_VALUE;
        boolean flag = false;
        for (int i = 0; i < points.length; i++) {
            for (int j = 0; j < points.length; j++) {
                if (i == j) continue;
                for (int k = 0; k < points.length; k++) {
                    if (i == k || k == j) continue;
                    int w1 = distance(points[i][0], points[i][1], points[j][0], points[j][1]);
                    int w2 = distance(points[i][0], points[i][1], points[k][0], points[k][1]);
                    int w3 = distance(points[k][0], points[k][1], points[j][0], points[j][1]);
                    if (w1 + w2 == w3) {
                        for (int q = 0; q < points.length; q++) {
                            if (q == i || q == j || q == k) continue;
                            if (points[q][0] - points[k][0] == - points[i][0] + points[j][0] &&
                               points[q][1] - points[k][1] == - points[i][1] + points[j][1]) {
                                min = Math.min(Math.sqrt(w1) * Math.sqrt(w2), min);
                                //System.out.println("w1 = " +w1+ "w2 =" + w2 + "w3= "+ w3);
                                flag = true;
                                break;
                            }
                        }                       
                    }                  
                }
            }
        }
        if (flag) return min;
        return 0.0;
    }
    
    public int distance(int i1, int j1, int i2, int j2) {
        return (i1 - i2) * (i1 - i2) + (j1 - j2) * (j1 - j2);
    }
}
