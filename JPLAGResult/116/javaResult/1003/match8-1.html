class Solution {
	public double minAreaFreeRect(int[][] points) {
		double minarea = Double.MAX_VALUE;
		for (int i = 0; i < points.length - 1; i++) {
			for (int j = i + 1; j < points.length; j++) {
				double x1 = points[j][0] - points[i][0];
				double y1 = points[j][1] - points[i][1];
				for (int k = i + 1; k != j && k < points.length; k++) {
					double x2 = points[k][0] - points[j][0];
					double y2 = points[k][1] - points[j][1];
					if (x1 * x2 + y1 * y2 == 0) {
						double x = points[i][0] + x2;
						double y = points[i][1] + y2;
						for (int[] elem : points) {
							if (Math.abs(elem[0] - x) < 0.0001 && Math.abs(elem[1] - y) < 0.0001) {
								double area = Math.sqrt(x1 * x1 + y1 * y1) * Math.sqrt(x2 * x2 + y2 * y2);
								if (area < minarea) {
									minarea = area;
									System.out.println(i + " " + j + " " + k);
								}
								break;
							}

						}
					}
				}
			}
		}
		if (minarea < Double.MAX_VALUE)
			return minarea;
		else
			return 0;
	}
}
