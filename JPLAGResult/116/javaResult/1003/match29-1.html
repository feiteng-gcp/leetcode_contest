class Solution {
    public double minAreaFreeRect(int[][] points) {
        HashSet<List<Integer>> pointSet = new HashSet();
        for(int[] point : points){
            List<Integer> lspoint = new ArrayList(2);
            lspoint.add(point[0]);lspoint.add(point[1]);
            pointSet.add(lspoint);
        }
        
        if(points == null || points.length < 4) return 0;
        
        
double minArea = Integer.MAX_VALUE;
        
        for(int i=0;i<points.length;++i){
            for(int j=i+1;j<points.length;++j){
                for(int k=j+1;k<points.length;++k){
                        double area = getArea(points,i,j,k,pointSet);
                        if(area <= 0.0001) continue;
                        // if(area - 1.0 < 0.00001) return area;
                        if(area < minArea) minArea = area;
                
}
            }
        }
        if(minArea == Integer.MAX_VALUE) return 0;
        return minArea;
    }
    private int getDisSqr(int[][] A, int i, int j){
        int aSqr = (A[i][0]-A[j][0])*(A[i][0]-A[j][0]);
        int bSqr = (A[i][1]-A[j][1])*(A[i][1]-A[j][1]);
        
return aSqr + bSqr;
    }
    
    private double calcu(int[][] A, int i, int j, int k){
        int disA = getDisSqr(A, i, j);
        int disB = getDisSqr(A, i, k);
        return Math.sqrt(disA)*Math.sqrt(disB);
    
}
    
    private double getArea(int[][] A, int i, int j, int k, HashSet<List<Integer>> pointSet){
        //point muti for three points
        //if one of them is 0, it exists a valid rect
        int a = (A[i][0]-A[j][0])*(A[i][0]-A[k][0]) + (A[i][1]-A[j][1])*(A[i][1]-A[k][1]);
        int b = (A[j][0]-A[i][0])*(A[j][0]-A[k][0]) + (A[j][1]-A[i][1])*(A[j][1]-A[k][1]);
        int c = (A[j][0]-A[i][0])*(A[k][0]-A[j][0]) + (A[k][1]-A[i][1])*(A[k][1]-A[j][1]);
        
        if(a == 0 || b == 0 || c == 0){
            //valid rect
            //judge if the other point exists
            if(a == 0){
                List<Integer> otherPoint = new ArrayList(2);
                int vx = A[j][0] - A[i][0];
                int vy = A[j][1] - A[i][1];
                int x = A[k][0] + vx;
                int y = A[k][1] + vy;
                otherPoint.add(x);otherPoint.add(y);
                if(!pointSet.contains(otherPoint)) return -1;
                else return calcu(A,i,j,k);
            } else if(b == 0){
                List<Integer> otherPoint = new ArrayList(2);
                int vx = A[i][0] - A[j][0];
                int vy = A[i][1] - A[j][1];
                int x = A[k][0] + vx;
                int y = A[k][1] + vy;
                otherPoint.add(x);otherPoint.add(y);
                if(!pointSet.contains(otherPoint)) return -1;
                else return calcu(A,j,i,k);
            } else {
                List<Integer> otherPoint = new ArrayList(2);
                int vx = A[i][0] - A[k][0];
                int vy = A[i][1] - A[k][1];
                int x = A[j][0] + vx;
                int y = A[j][1] + vy;
                otherPoint.add(x);otherPoint.add(y);
                if(!pointSet.contains(otherPoint)) return -1;
                else return calcu(A,k,i,j);
            }
        } else return -1;
        
    }
}
