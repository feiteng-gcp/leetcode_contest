class Solution {
	public double minAreaFreeRect(int[][] points) {
		if (points.length < 4)
			return 0;
		double min = D
ouble.MAX_VALUE;
		for (int i = 0; i < points.length; i++) {
			for (int j = i + 1; j < points.length; j++) {
				for (int p = j + 1; p < points.length; p++) {
					for (int q = p + 1; q < points.length; q++) {
						if (isRec(points[i], points[j], points[p], points[q]))
							min = Math.min(min, calcArea(points[i], points[j], points[p], points[q]));

					}
				}
			}
		}


		return min == Double.MAX_VALUE ? 0 : min;
	}

	boolean isRec(int[] a, int[] b, int[] c, int[] d) {
		double cx, cy;
		double dd1, dd2, dd3, dd4;

		cx = (a[0] + b[0] + c[0] + d[0]) / 4.0;
		cy = (a[1] + b[1] + c[1] + d[1]) / 4.0;

		dd1 = sqr(cx - a[0]) + sqr(cy - a[1]);
		dd2 = sqr(cx - b[0]) + sqr(cy - b[1]);
		dd3 = sqr(cx - c[0]) + sqr(cy - c[1]);
		dd4 = sqr(cx - d[0]) + sqr(cy - d[1]);
		return dd1 == dd2 && dd1 == dd3 && dd1 == dd4;
	}

	double sqr(double i) {
		return i * i;
	}
	
	double calcArea(int[] a, int[] b, int[] c, int[] d) {
		double len1 = sqr(a[0] - b[0]) + sqr(a[1] - b[1]);
		double len2 = sqr(a[0] - c[0]) + sqr(a[1] - c[1]);
		double len3 = sqr(b[0] - c[0]) + sqr(b[1] - c[1]);
		List<Double> list = Arrays.asList(len1, len2, len3);
		Collections.sort(list);
		return Math.sqrt(list.get(0)) * Math.sqrt(list.get(1));
	}
}
