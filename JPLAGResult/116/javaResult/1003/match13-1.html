class Solution {
    public double minAreaFreeRect(int[][] points) {
        double min = Double.MAX_VALUE;
        HashMap<String, List<int[]>> map = new HashMap<>();
        
        if (points == null || points.length < 4) {
            return 0;
        }
        
        
int n = points.length;
        for (int i = 0; i < points.length; i++) {
            for (int j = i+1; j < points.length; j++) {
                int ydiff = points[i][1] - points[j][1];
                int xdiff = points[i][0] - points[j][0];
                
                int gcd = getGCD(xdiff, ydiff);
                if (gcd != 0) {
                    xdiff /= gcd;
                    ydiff /= gcd;
                }
                
                
int len = ydiff * ydiff + xdiff * xdiff;
                
                String key = xdiff + ":" + ydiff + ":" + len;
                
                if (!map.containsKey(key)) {
                    map.put(key, new ArrayList<int[]>());
                }
                
                if (points[i][0] < points[j][0]) {
                    map.get(key).add(new int[]{i, j});
                } else {
                    map.get(key).add(new int[]{j, i});
                }
            
}
        }
        
        for (Map.Entry<String, List<int[]>> entry : map.entrySet()) {
            List<int[]> list = entry.getValue();
            
            for (int i = 0; i < list.size(); i++) {
                for (int j = i+1; j < list.size(); j++) {
                    int[] A = points[list.get(i)[0]];
                    int[] B = points[list.get(i)[1]];
                    int[] C = points[list.get(j)[0]];
                    int[] D = points[list.get(j)[1]];
                    if (isRect(A, B, C, D)) {
                        min = Math.min(
getArea(A, B, C, D), min);
                    }
                }
            }
        }
        
        return min == Double.MAX_VALUE ? 0 : min;
    }
    
    public double getArea(int[] A, int[] B, int[] C, int[] D) {
        double len1 = (A[0] - B[0]) * (A[0] - B[0]) + (A[1] - B[1]) * (A[1] - B[1]);
        double len2 = (A[0] - C[0]) * (A[0] - C[0]) + (A[1] - C[1]) * (A[1] - C[1]);
      //  System.out.println(len1);
      //  System.out.println(len2);
        
        return Math.sqrt(len1) * Math.sqrt(len2);
    }
    
    public boolean isRect(int[] A, int[] B, int[] C, int[] D) {
        
        double x = (A[0] + B[0] + C[0] + D[0]) / 4.0;
        double y = (A[1] + B[1] + C[1] + D[1]) / 4.0;
        
        double dd1 = (x - A[0]) * (x - A[0]) + (y - A[1]) * (y - A[1]);
        double dd2 = (x - B[0]) * (x - B[0]) + (y - B[1]) * (y - B[1]);
        double dd3 = (x - C[0]) * (x - C[0]) + (y - C[1]) * (y - C[1]);
        double dd4 = (x - D[0]) * (x - D[0]) + (y - D[1]) * (y - D[1]);
        
        return dd1 == dd2 && dd2 == dd3 && dd3 == dd4;
    }
    
    private int getGCD(int a,int b){
    
        	if (b==0) return a;
        	else return getGCD(b,a%b);
        	
    }
}
