class Solution {
    
class Point {
		int x;
		int y;
		Point(int x, int y) {
			this.x = x;
			this.y = y;
		}
		
		@Override
		public boolean equals(Object obj) {
			if (!(obj instanceof Point)) {
				return false;
			}
			Point p = (Point) obj;
			return x == p.x && y == p.y;
		}
		
		@Override
		public int hashCode() {
			return x * 40001 + y;
		}
		
	}
	
	long distanceSqr(Point A, Point B) {
		return (long)(Math.pow(A.x - B.x, 2) + Math.pow(A.y - B.y, 2));
	}
	
	Point findLastPoint(Point A, Point B, Point C) {
		return new Point(B.x + C.x - A.x, B.y + C.y - A.y);
	}
	
	public 
double minAreaFreeRect(int[][] points) {
        List<Point> pList = new ArrayList<>();
        Set<Point> pSet = new HashSet<>();
        for (int[] p : points) {
        	Point point = new Point(p[0], p[1]);
        	pList.add(point);
        	pSet.add(point);
        }
        double area = Math.pow(40000, 2);
        boolean found = false;
        
for (int firstIdx = 0; firstIdx < points.length - 2; firstIdx++) {
        	Point pFirst = pList.get(firstIdx);
        	for (int secondIdx = firstIdx + 1; secondIdx < points.length - 1; secondIdx++) {
        		Point pSecond = pList.get(secondIdx);
        		long d12 = distanceSqr(pSecond, pFirst);
        		for (int thirdIdx = secondIdx + 1; thirdIdx < points.length; thirdIdx++) {
        			Point pThird = pList.get(thirdIdx);
        			long d13 = distanceSqr(pThird, pFirst);
        			long d23 = distanceSqr(pThird, pSecond);
        			long max = Math.max(d12,  Math.max(d13,  d23));
        			if (d12 + d13 + d23 == max * 2) {
        				// a triangle with right angle
        				if (d12 + d13 == d23 && pSet.contains(findLastPoint(pFirst, pSecond, pThird))) {
        					found = true;
        					area = Math.min(area, Math.sqrt(d12 * d13));
        				} else if (d12 + d23 == d13 && pSet.contains(findLastPoint(pSecond, pFirst, pThird))) {
        					found = true;
        					area = Math.min(area, Math.sqrt(d12 * d23));
        				} else if (d23 + d13 == d12 && pSet.contains(findLastPoint(pThird, pFirst, pSecond))) {
        					found = true;
        					area = Math.min(area, Math.sqrt(d13 * d23));
        				}

        			}
        		}
        	}
        }
        return found ? area : 0;
    }
}
