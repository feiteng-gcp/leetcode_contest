class Solution {
    public double minAreaFreeRect(int[][] points) {
        
int n = points.length;
        double result = 40_000.0 * 40_001.0;
        for(int i1 = 0; i1 < n; i1++) {
            for(int i2 = 0; i2 < n; i2++) {
                for(int i3 = 0; i3 < n; i3++) {
                    for(int i4 = 0; i4 < n; i4++) {
                        if(correctRotation(points, i1, i2, i3) && correctRotation(points, i2, i3, i4) &&
                          correctRotation(points, i3, i4, i1) && correctRotation(points, i4, i1, i2)) {
                            //System.out.println("CorrectRotation: " + i1 + " " + i2 + " " + i3 + " " + i4);
                            if(correctAngle(points, i1, i2, i3) && correctAngle(points, i2, i3, i4) &&
                          correctAngle(points, i3, i4, i1) && correctAngle(points, i4, i1, i2)) {
                                result = Math.min(result, Math.sqrt(1.0*len2(points, i1, i2)) * Math.sqrt(1.0 *len2(points, i2, i3)));
                            
}
                        }
                    }
                }
            }
        }
        if(result > 40_000.0 * 40_000.0) {
            return 0;
        } else {
            return result;
        }
    }
    
    private boolean correctRotation(int[][] points, int i1, int i2, int i3) {
        return points[i1][0] * points[i2][1] - points[i1][1] * points[i2][0] + 
            points[i2][0] * points[i3][1] - points[i2][1] * points[i3][0] + 
            points[i3][0] * points[i1][1] - points[i3][1] * points[i1][0] > 0;
    }
    
    private boolean correctAngle(int[][] points, int i1, int i2, int i3) {
        return len2(points, i1, i2) + len2(points, i2, i3) == len2(points, i1, i3);
    }
    
    private int len2(int[][] points, int i1, int i2) {
        return (points[i1][0] - points[i2][0]) * (points[i1][0] - points[i2][0]) +
            (points[i1][1] - points[i2][1]) * (points[i1][1] - points[i2][1]);
    }
}
