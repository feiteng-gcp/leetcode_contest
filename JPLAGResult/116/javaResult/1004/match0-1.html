class Solution {
    public int leastOpsExpressTarget(int x, int target) {
        
return leastOpsExpressTarget0(x, target);
    }
    
    
    public int leastOpsExpressTarget0(int x, int target)
    {
        int[] parts = new int[32];
        for(int i = 0; i < 32; i++)
        {
            parts[i] = target % x;
            target /= x;
        }
        
        int[][] dp = new int[32][2];
        dp[31][0] = 0;
        dp[31][1] = 31;
        
        for(int i = 30; i >= 0; i--)
        {
            int signNeed = i;
            if(i == 0)
            {
                signNeed = 1 + 1;
            }
            
            int need = parts[i];
            
            int p0 = dp[i + 1][0] == 0 ? -1 : dp[i + 1][0];
            int p1 = dp[i + 1][1] == 0 ? -1 : dp[i + 1][1];
            
            dp[i][0] = Math.min(p0 + signNeed * need,
                               p1 + signNeed * (x - need));
            dp[i][0] = Math.max(dp[i][0], 0);
            dp[i][1] = Math.min(p0 + signNeed * (need + 1), 
                               p1 + signNeed * (x - need - 1));
            dp[i][1] = Math.max(dp[i][1], 0);
        }
        System.out.println(Arrays.toString(parts));
        System.out.println(Arrays.deepToString(dp));
        
        return dp[0][0];
    }
}
