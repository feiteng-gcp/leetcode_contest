class Solution {
    public int leastOpsExpressTarget(int x, int target) {
        
        List<Integer> exp = convert(x, target);
        int n = exp.size();
        Integer[] cache = new Integer[n];
        
        return dp(exp, x, 0, cache)-1;
        
    }
    
    private List<Integer> convert(int x, int target) {
        // convert target into x-ary
        List<Integer> res = new ArrayList<Integer>();
        while (target != 0) {
            res.add(target % x);
            target /= x;
        }
        return res;
    }
    
    private int dp(List<Integer> exp, int x, int start, Integer[] cache) {
        // return the number of operations needed to generate exp[start:]
        int n = exp.size();
        if (start >= n) return 0;
        
        if (cache[start] == null) {
            
            int min = Integer.MAX_VALUE;
            int sum = 0;
            for (int i = start; i <= n; ++i) {
                // i: the first location to add 
                // first, if add
                int cur = 0; // the number of operation needed for current digit
                if (i == 0) cur = 1; 
                else cur = i - 1;
                
                int t = (i == n?0 : exp.get(i)); // the number of additions
                if (i != start) t += 1;
                int tmp = sum + t * cur + t + dp(exp, x, i+1, cache); // the number of operations to add
                if (tmp < min) min = tmp;
                if (i == n) continue;
                
                // second,  minus current location
                int t2 = x - 1 - exp.get(i); // the number of subtraction
                if (i == start) t2 += 1;
                sum += t2 + t2*cur;
            }
            cache[start] = min;
        }
        return cache[start];
    }
}
