class Solution {
    public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
        Map<Integer,Set<Integer>> xy = new HashMap();
        Map<Integer,Set<Integer>> yx = new HashMap();
        Map<Integer,Set<Integer>> left_axis = new HashMap();
        Map<Integer,Set<Integer>> right_axis = new HashMap();
        for(int[] lamp: lamps){
            
if (!xy.containsKey(lamp[0])) {
                xy.put(lamp[0],new HashSet<Integer>());
            }
            xy.get(lamp[0]).add(lamp[1]);
            if (!yx.containsKey(lamp[1])) {
                yx.put(lamp[1],new HashSet<Integer>());
            }
            yx.get(lamp[1]).add(lamp[0]);
            if (!left_axis.containsKey(lamp[1]+lamp[0])) {
                left_axis.put(lamp[1]+lamp[0],new HashSet<Integer>());
            }
            left_axis.get(lamp[1]+lamp[0]).add(lamp[0]);
            if (!right_axis.containsKey(lamp[1]-lamp[0])) {
                right_axis.put(lamp[1]-lamp[0],new HashSet<Integer>());
            }
            right_axis.get(lamp[1]-lamp[0]).add(lamp[0]);
        }
        int[] result = new int[queries.length];
//        for (int[] query: queries){
        for (int index=0; index<queries.length; index++) {
            int x = queries[index][0];
            int y = queries[index][1];
            if (xy.containsKey(x) || yx.containsKey(y) || right_axis.containsKey(y-x) || left_axis.containsKey(y+x)) {
                result[index] =1;
                for (int i=-1;i<=1;i++){
                    
for (int j=-1;j<=1;j++){
                        int x1 = x+i;
                        int y1 = j+i;
                        if (xy.containsKey(x1) && xy.get(x1).contains(y1)) {
                            
xy.get(x1).remove(y1);
                            if (xy.get(x1).size()==0) {
                                xy.remove(x1);
                            }
                        }
                        if (yx.containsKey(y1) && yx.get(y1).contains(x1)) {
                            yx.get(y1).remove(x1);
                            if (yx.get(y1).size()==0) {
                                yx.remove(y1);
                            }
                        }
                        if (left_axis.containsKey(y1+x1) && left_axis.get(y1+x1).contains(x1)) {
                            left_axis.get(y1+x1).remove(x1);
                            if (left_axis.get(y1+x1).size()==0) {
                                left_axis.remove(y1+x1);
                            }
                        }
                        if (right_axis.containsKey(y1-x1) && right_axis.get(y1-x1).contains(x1)) {
                            right_axis.get(y1-x1).remove(x1);
                            
if (right_axis.get(y1-x1).size()==0) {
                                right_axis.remove(y1-x1);
                            }
                        }
                    }
                }
            }
            else {
                result[index] = 0;
            }

        }
        return result;
        
    }
}
