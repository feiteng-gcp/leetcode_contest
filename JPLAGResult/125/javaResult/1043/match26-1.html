class Solution {
    
    
public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
        int[] solution = new int[queries.length];
        
        Set<String> lampSet = new HashSet<>();
        Map<Integer, Integer> xMap = new HashMap<>();
        Map<Integer, Integer> yMap = new HashMap<>();
        Map<Integer, Integer> sumMap = new HashMap<>();
        Map<Integer, Integer> diffMap = new HashMap<>();
        for (int[] l: lamps){
            lampSet.add(ctos(l[0], l[1]));
            xMap.put(l[0], xMap.getOrDefault(l[0], 0) + 1);
            yMap.put(l[1], yMap.getOrDefault(l[1], 0) + 1);
            sumMap.put(l[0]+l[1], sumMap.getOrDefault(l[0]+l[1], 0) + 1);
            diffMap.put(l[0]-l[1], diffMap.getOrDefault(l[0]-l[1], 0) + 1);
        
}
        
        for (int i = 0; i < queries.length; i++){
            /*System.out.println(xMap);
            System.out.println(yMap);
            System.out.println(sumMap);
            System.out.println(diffMap);*/
            int[] q = queries[i];
            if (xMap.getOrDefault(q[0],0) > 0 || yMap.getOrDefault(q[1],0) > 0 || sumMap.getOrDefault(q[0]+q[1],0) > 0 || diffMap.getOrDefault(q[0]-q[1],0) > 0)
                solution[i] = 1;
            
            // Turn off
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0]-1, q[1]-1);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0]-1, q[1]);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0]-1, q[1]+1);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0], q[1]-1);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0], q[1]);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0], q[1]+1);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0]+1, q[1]-1);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0]+1, q[1]);
            turnOff(lampSet, xMap, yMap, sumMap, diffMap, q[0]+1, q[1]+1);
        
}
        
        return solution;
        
    }
    
    public String ctos(int x, int y){
        return Integer.toString(x) + " " + Integer.toString(y);
    }
    
    public void turnOff(Set<String> lamps, Map<Integer, Integer> xMap, Map<Integer, Integer> yMap, Map<Integer, Integer> sumMap, Map<Integer, Integer> diffMap, int x, int y){
        if (lamps.contains(ctos(x, y))){
            lamps.remove(
ctos(x, y));
            xMap.put(x, xMap.get(x) - 1);
            yMap.put(y, yMap.get(y) - 1);
            sumMap.put(x+y, sumMap.get(x+y) - 1);
            diffMap.put(x-y, diffMap.get(x-y) - 1);
        }
    }
}
