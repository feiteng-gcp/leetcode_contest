class Solution {
    public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
        Map<Integer, Set<List<Integer>>> row = 
new HashMap<>();
        Map<Integer, Set<List<Integer>>> col = new HashMap<>();
        Map<Integer, Set<List<Integer>>> dia = new HashMap<>();
        Map<Integer, Set<List<Integer>>> rev = new HashMap<>();
        Set<List<Integer>> lampsOn = new HashSet<>();
        for (int[] l : lamps) {
            int r = l[0];
            int c = l[1];
            List<Integer> point = Arrays.asList(r, c);
            Set<List<Integer>> tmpSet;
            tmpSet = row.getOrDefault(r, new HashSet<>());
            tmpSet.add(point);
            row.put(r, tmpSet);
            tmpSet = col.getOrDefault(c, new HashSet<>());
            tmpSet.add(point);
            col.put(c, tmpSet);
            tmpSet = dia.getOrDefault(r + c, new HashSet<>());
            tmpSet.add(point);
            dia.put(r + c, tmpSet);
            tmpSet = rev.getOrDefault(r - c, new HashSet<>());
            tmpSet.add(point);
            rev.put(r - c, tmpSet);
            lampsOn.add(point);
        }
        
int[][] dir = {{-1, -1}, {-1, 0}, {-1, 1}, {0, -1}, {0, 0}, {0, 1}, {1, -1}, {1, 0}, {1, 1}};
        int[] ret = 
new int[queries.length];
        for (int i = 0; i < queries.length; ++i) {
            int[] q = queries[i];
            int r = q[0];
            int c = q[1];
            if (row.getOrDefault(r, new HashSet<>()).size() != 0 || 
               col.getOrDefault(c, new HashSet<>()).size() != 0 ||
               dia.getOrDefault(r + c, new HashSet<>()).size() != 0||
               rev.getOrDefault(r - c, new HashSet<>()).size() != 0) {
                ret[i] = 1;
            } else 
{
                ret[i] = 0;
            }
            for (int[] d : dir) {
                int rr = r + d[0];
                int cc = c + d[1];
                List<Integer> point = Arrays.asList(rr, cc);
                
if (lampsOn.contains(point)) {
                    lampsOn.remove(point);
                    row.get(rr).remove(point);
                    col.get(cc).remove(point);
                    dia.get(rr + cc).remove(point);
                    rev.get(rr - cc).remove(point);
                }
            }
        }
        
        
        
        return ret;
    }
}
