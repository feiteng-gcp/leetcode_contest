class Solution {
    private void turnoff(Map<String, int[]> map,Map<Integer, Integer> map1,Map<Integer, Integer> map2,Map<Integer, Integer> map3,Map<Integer, Integer> map4, int x, int y, int N) {
        if (x < 0 || x == N || y < 0 || y == N) return;
        String key = "" + x + "," + y;
        if (map.containsKey(key)) {
            map.remove(key);
            int v = map1.get(x);
            if (v == 1) map1.remove(x);
            else map1.put(x, v - 1);
            v = map2.get(y);
            if (v == 1) map2.remove(y);
            else map2.put(y, v - 1);
            v = map3.get(x + y);
            if (v == 1) map3.remove(x + y);
            else map3.put(x + y, v - 1);
            v = map4.get(x - y);
            if (v == 1) map4.remove(x - y);
            else map4.put(x - y, v - 1);
        }
    }
    private int[][] ds = {{-1,-1},{-1,0},{-1,1},{0,-1},{0,0},{0,1},{1,-1},{1,0},{1,1}};
    public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
        Map<Integer, Integer> map1 = new HashMap<>();
        Map<Integer, Integer> map2 = new HashMap<>();
        Map<Integer, Integer> map3 = new HashMap<>();
        Map<Integer, Integer> map4 = new HashMap<>();
        Map<String, int[]> map = new HashMap<>();
        for (int[] l : lamps) {
            int x = l[0], y = l[1];
            map1.put(x, map1.getOrDefault(x, 0) + 1);
            map2.put(y, map2.getOrDefault(y, 0) + 1);
            map3.put(x + y, map3.getOrDefault(x + y, 0) + 1);
            map4.put(x - y, map4.getOrDefault(x - y, 0) + 1);
            map.put("" + x + "," + y, l);
        }
        int[] res = new int[queries.length];
        int i = 0;
        for (int[] q : queries) {
            int x = q[0], y = q[1];
            if (map1.containsKey(x) || map2.containsKey(y) || map3.containsKey(x + y) || map4.containsKey(x - y)) {
                res[i] = 1;
            }
            for (int[] d : ds) {
                turnoff(map, map1, map2, map3, map4, x + d[0], y + d[1], N);
            }
            i++;
        }
        return res;
    }
}
