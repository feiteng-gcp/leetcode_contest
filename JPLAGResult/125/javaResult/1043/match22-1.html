class Solution {
   	priv
ate int[] shifts = new int[] { -1, 0, 1 };

	public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
		int[] res = new int[queries.length];
		Map<Long, Inte
ger> xCntMap = new HashMap<>();
		Map<Long, Integer> yCntMap = new HashMap<>();
		Map<Long, Integer> dCntMap = new HashMap<>();
		Map<Long, Integer> drCntMap = new HashMap<>();
		Set<Long> lSet = new HashSet<>();
		for (int[] lamp : lamps) {
			long x = lamp[0];
			long y = lamp[1];
			incre(x, xCntMap);
			incre(y, yCntMap);
			incre(x - y, dCntMap);
			incre(x + y, drCntMap);
			lSet.add((x << 30) + y);
		}
		for (int i = 0; i < queries.length; i++) {
			long x = queries[i][0
];
			long y = queries[i][1];
			if (xCntMap.containsKey(x) || yCntMap.containsKey(y) || dCntMap.containsKey(x - y)
					|| drCntMap.containsKey(x + y)) {
				res[i] = 1;
			}
			for (int xs : shifts) {
				for (int ys : shifts) {
					long xd = x + xs;
					long yd = y + ys;
					long xy = (xd << 30) + yd;
					if (lSet.contains(xy)) {
						lSet.remove(xy);

						decre(xd, xCntMap);
						decre(yd, yCntMap);
						decre(xd - yd, dCntMap);
						decre(xd + yd, drCntMap);
					}
				}
			}
		}
		return res;
	}

	private void incre(long x, Map<Long, Integer> xCntMap) {
		xCntMap.put(x, xCntMap.getOrDefault(x, 0) + 1);
	}

	private void decre(long x, Map<Long, Integer> xCntMap) {
		xCntMap.put(x, xCntMap.get(x) - 1);
		if (xCntMap.get(x) == 0) {
			xCntMap.remove(x);
		}
	}
}
