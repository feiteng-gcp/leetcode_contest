class Solution {
    public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
        Map<Integer,Map<Integer,Integer>> xMap=new HashMap<>();
        Map<Integer,Set<Integer>> yMap=new HashMap<>(),minMap=new HashMap<>(),sumMap=new HashMap<>();
        for(int i=0;i<lamps.length;i++){
            int[] lamp=lamps[i];
            if(!xMap.containsKey(lamp[0])){
                xMap.put(lamp[0],new HashMap<>());
            }
            xMap.get(lamp[0]).put(lamp[1],i);
            if(!yMap.containsKey(lamp[1])){
                yMap.put(lamp[1],new HashSet<>());
            }
            yMap.get(lamp[1]).add(i);
            if(!minMap.containsKey(lamp[0]-lamp[1])){
                minMap.put(lamp[0]-lamp[1],new HashSet<>());
            }
            minMap.get(lamp[0]-lamp[1]).add(i);
            if(!sumMap.containsKey(lamp[0]+lamp[1])){
                sumMap.put(lamp[0]+lamp[1],new HashSet<>());
            }
            sumMap.get(lamp[0]+lamp[1]).add(i);
        }
        int[] result=new int[queries.length];
        int[][] ds=new int[][]{{-1,-1},{-1,0},{-1,1},{0,-1},{0,0},{0,1},{1,-1},{1,0},{1,1}};
        for(int i=0;i<queries.length;i++){
            int[] query=queries[i];
            if(xMap.containsKey(query[0])||yMap.containsKey(query[1])||minMap.containsKey(query[0]-query[1])||sumMap.containsKey(query[0]+query[1])){
                result[i]=1;
            }
            for(int[] d:ds){
                int x=query[0]+d[0],y=query[1]+d[1];
                if(xMap.containsKey(x)&&xMap.get(x).containsKey(y)){
                    int t=xMap.get(x).remove(y);
                    yMap.get(y).remove(t);
                    minMap.get(x-y).remove(t);
                    sumMap.get(x+y).remove(t);
                    if(xMap.get(x).size()==0){
                        xMap.remove(x);
                    }
                    if(yMap.get(y).size()==0){
                        yMap.remove(y);
                    }
                    if(minMap.get(x-y).size()==0){
                        minMap.remove(x-y);
                    }
                    if(sumMap.get(x+y).size()==0){
                        sumMap.remove(x+y);
                    }
                }
            }
        }
        return result;
    }
}
