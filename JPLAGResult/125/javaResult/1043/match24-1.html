class Solution {
    
public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
        Map<Integer, Integer> mapI = new HashMap<>();
        Map<Integer, Integer> mapJ = new HashMap<>();
        Map<Integer, Integer> mapK1 = new HashMap<>();
        Map<Integer, Integer> mapK2 = new HashMap<>();
        Set<String> set = new HashSet<>();
        
        //light lamp
        for (int[] lamp : lamps){
            int i = lamp[0];
            int j = lamp[1];
            mapI.put(i, mapI.getOrDefault(i,0)+1);
            mapJ.put(j, mapJ.getOrDefault(j,0)+1);
            int k1 = j-i;
            mapK1.put(k1, mapK1.getOrDefault(k1,0)+1);
            int k2 = j+i;
            mapK2.put(k2, mapK2.getOrDefault(k2,0)+1);
            
            String curr = String.valueOf(i) + "#" + String.valueOf(j);
            set.add(curr);
        }
        
        //query
        int[] dir1 = new int[] {0,0,1,-1,1,1,-1,-1};
        int[] dir2 = new int[] {1,-1,0,0,1,-1,1,-1};
        int[] ans = new int[queries.length];
        for (int k = 0; k < ans.length; k++){
            int i = queries[k][0];
            int j = queries[k][1];
            if (mapI.containsKey(i) && mapI.get(i) > 0 || mapJ.containsKey(j) && mapJ.get(j)>0 || 
mapK1.containsKey(j-i)&&mapK1.get(j-i)>0 || mapK2.containsKey(j+i) && mapK2.get(j+i) > 0){
                ans[k] = 1;
            }
            
            //turn off
            for (int next = 0; next < dir1.length; next++){
                int nexti = i + dir1[next];
                int nextj = j + dir2[next];
                if (nexti < 0 || nexti >= N || nextj < 0 || nextj >= N)
                    continue;
                String curr = String.valueOf(nexti) + "#" + String.valueOf(nextj);
                if (set.contains(curr)){
                    //System.out.println("remove");
                    set.remove(curr);
                    mapI.put(nexti, mapI.get(nexti)-1);
                    mapJ.put(nextj, mapJ.get(nextj)-1);
                    int k1 = nextj-nexti;
                    mapK1.put(k1, mapK1.get(k1)-1);
                    int k2 = nextj+nexti;
                    
mapK2.put(k2, mapK2.get(k2)-1);
                    
                }
            }
            
        }
        return ans;
    }
}
