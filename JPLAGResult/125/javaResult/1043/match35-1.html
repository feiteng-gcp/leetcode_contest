class Solution {
    
public int[] gridIllumination(int N, int[][] lamps, int[][] queries) {
        Set<String> set = new HashSet();
        HashMap<Integer, Integer> map_x = new HashMap();
        HashMap<Integer, Integer> map_y = new HashMap();
        HashMap<Integer, Integer> map_xPy = new HashMap();
        HashMap<Integer, Integer> map_xDy = new HashMap();
        
        for(int i = 0; i < lamps.length; i++) {
            int x = lamps[i][0];
            int y = lamps[i][1];
            set.add(x + "&" + y);
            map_x.put(x, map_x.getOrDefault(x, 0) + 1);
            map_y.put(y, map_y.getOrDefault(y, 0) + 1);
            map_xPy.put(x+y, map_xPy.getOrDefault(x+y, 0) + 1);
            map_xDy.put(x-y, map_xDy.getOrDefault(x-y, 0) + 1);
        }
        
        int[] res = new int[queries.length];
        for(int i = 0; i < queries.length; i++) {
            int x = queries[i][0];
            int y = queries[i][1];
            
            if(map_x.getOrDefault(x, 0) != 0 || map_y.getOrDefault(y, 0) != 0
            || map_xPy.getOrDefault(x+y, 0) != 0 || map_xDy.getOrDefault(x-y, 0) != 0) {
                res[i] = 1;
            } else {
                res[i] = 0;
            }
            
            
String str = x + "&" + y;
            if(set.contains(str)) {
                set.remove(str);
                map_x.put(x, map_x.get(x) - 1);
                map_y.put(y, map_y.get(y) - 1);
                map_xPy.put(x+y, map_xPy.get(x+y) - 1);
                map_xDy.put(x-y, map_xDy.get(x-y) - 1);                
            }
            
            if(x-1 >= 0) {
                int xx = x - 1;
                int yy = y;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }
            
            if(x+1 < N) {
                int xx = x + 1;
                int yy = y;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }
            
            if(y-1 >= 0) {
                int xx = x;
                int yy = y - 1;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }
            
            if(y+1 < N) {
                int xx = x;
                int yy = y + 1;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }     

            if(x-1 >= 0 && y - 1 >= 0) {
                int xx = x - 1;
                int yy = y - 1;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }   
            
            if(x-1 >= 0 && y + 1 < N) {
                int xx = x - 1;
                int yy = y + 1;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }   
            
            if(x+1 < N && y - 1 >= 0) {
                int xx = x + 1;
                int yy = y - 1;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }     
            
            if(x+1 < N && y + 1 < N) {
                int xx = x + 1;
                int yy = y + 1;
                str = xx + "&" + yy;
                if(set.contains(str)) {
                    set.remove(str);
                    map_x.put(xx, map_x.get(xx) - 1);
                    map_y.put(yy, map_y.get(yy) - 1);
                    map_xPy.put(xx+yy, map_xPy.get(xx+yy) - 1);
                    map_xDy.put(xx-yy, map_xDy.get(xx-yy) - 1);                
                }                
            }             
        }
        
        return res;
    }
}
