class Solution {
    public int findJudge(int N, int[][] trust) {
        int[] beTrustNum = new int[N + 1];
        boolean[] trustOther = new boolean[N + 1];
        for(int i = 0; i < trust.length; i ++) {
            trustOther[trust[i][0]] = true;
            beTrustNum[trust[i][1]] ++;
        
}
        for(int i = 1; i <= N; i ++) {
            if(trustOther[i] == false && beTrustNum[i] == N - 1) {
                return i;
            }
        }
        return -1;
    }
}
