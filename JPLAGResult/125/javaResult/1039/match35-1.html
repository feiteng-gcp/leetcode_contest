class Solution {
    public int findJudge(int N, int[][] trust) {
        boolean[] isJudge = new boolean[N+1];
        int[] count = new int[N+1];
        Arrays.fill(isJudge, true);
        
for (int[] t : trust) {
            isJudge[t[0]] = false;
            count[t[1]]++;
        }
        
        for (int i = 1; i <= N; i++) {
            if (count[i] == N-1 && isJudge[i])
                return i;
        }
        
        return -1;
    }
}
