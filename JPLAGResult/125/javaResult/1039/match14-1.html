class Solution {
    public int findJudge(int N, int[][] trust) {
        int[] truster = new int[N + 1];
        int[] trustee = new int[N + 1];
        for (int[] t : trust) {
            truster[t[0]]++;
            trustee[t[1]]++;
        }
        for (int i = 1; i <= N; i++) {
            if (truster[i] == 0 && trustee[i] == N - 1) {
                return i;
            }
        }
        return -1;
    }
}
