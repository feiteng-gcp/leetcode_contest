class Solution {
    public int findJudge(int N, int[][] t) {
        
        int bdp[] = new int[N+1];
        int fdp[] = new int[N+1];
        
        for(int[] f:t){
            bdp[f[1]]++;
            fdp[f[0]]++;
        }
        
        for(int i=1;i<=N;++i){
            if(bdp[i]==N-1&&fdp[i]==0){
                return i;
            }
        }
        return -1;
        
    }
}
