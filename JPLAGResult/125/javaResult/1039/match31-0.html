class Solution {
    public int findJudge(int N, int[][] trust) {
        // N = 4; trust = new int[][]{{1,3},{1,4},{2,3},{2,4},{4,3}};
        boolean[] trust_other = new boolean[N + 1];
        int[] trust_by_other = new int[N + 1];
        for (int[] t : trust) {
            int a = t[0], b = t[1];
            trust_other[a] = true;
            trust_by_other[b]++;
        }
        for (int i = 1; i <= N; i++) {
            if (!trust_other[i] && trust_by_other[i] == N - 1) return i;
        }
        return -1;
    }
}
