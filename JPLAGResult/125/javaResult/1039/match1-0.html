class Solution {
    public int findJudge(int N, int[][] trust) {
        int[] in = new int[N];
        int[] out = new int[N];
        for (int[] rel : trust) {
            in[rel[1] - 1]++;
            out[rel[0] - 1]++;
        }
        
        for (int i = 0; i < N; i++) {
            if (in[i] == N - 1 && out[i] == 0) {
                return i + 1;
            }
        }
        return -1;
    }
}
