class Solution {
    public int findJudge(int N, int[][] trust) {
        int[] degree = new int[N];
        boolean[] ta = new boolean[N];//trust anybody.
        for(int[] t : trust){
            degree[t[1]-1]++;
            ta[t[0]-1] = true;
        
}
        
        
        for(int i=0;i<N;i++) {
            if(degree[i]==N-1&&!ta[i]){
                return i+1;
            }
        }
        
        return -1;
    }
}
