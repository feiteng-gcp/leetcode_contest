class Solution {
    public int findJudge(int N, int[][] trust) {
        int[] cnt = new int[N+1];
        boolean[] notJurger = new boolean[N+1];
        for(int[] tr : trust){
            //if(tr[0] != tr[1])
                cnt[tr[1]]++;
            notJurger[tr[0]] = true;
        }
        for(int i = 1; i <= N; i++){
            if(cnt[i] == N-1 && !notJurger[i]) return i;
        }
        return -1;
    }
}
