/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode insertIntoMaxTree(TreeNode root, int val) {
        TreeNode old = root;
        
        
        if(root==null){
            return new TreeNode(val);
        }
        
        
        return go(root,val,null);
        
        
    }
    
    public TreeNode go(TreeNode root,int val,TreeNode fa){
        if(root==null){
            TreeNode nr = new TreeNode(val);
            fa.right = nr;
            return nr;
        }
        
        if(val>root.val){
            if(fa==null){
           TreeNode nr = new TreeNode(val);
            nr.left = root;
            return nr;
            }else{
                TreeNode nr = new TreeNode(val);
                nr.left = root;
                fa.right = nr;
                
                return nr;
            }
            
            
        }else{
            go(root.right,val,root);
            return root;
        }
        
    }
    
    
}
