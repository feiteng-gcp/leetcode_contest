class Solution {
    public int numRookCaptures(char[][] board) {
        int res = 0, x=0, y=0;
        for(int i=0; i<8; i++){
            for(int j=0; j<8; j++){
                if(board[i][j] == 'R'){
                    x = i;
                    y = j;
                    break;
                }
            }
        }
        int[][] dirs = new int[][]{{0,1},{1,0},{0,-1},
{-1,0}};
        for(int[] dir:dirs){
            int i=x+dir[0], j=y+dir[1];
            while(i>=0 && i<8 && j>=0 && j<8){
                
if(board[i][j] == '.') {
                    i+=dir[0];
                    j+=dir[1];
                }
                else if(board[i][j] == 'p'){
                    res++;
                    break;
                } else {
                    break;
                
}
            }
        }
        return res;
    }
}
