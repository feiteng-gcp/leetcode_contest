class Solution {
    public int numRookCaptures(char[][] board) {
        int[][] dirs = new int[][]{
            {0, 1},
            {0, -1},
            {1, 0},
            {-1, 0}
        };
        int sx = -1, sy = -1;
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                if (board[i][j] == 'R') {
                    sx = i;
                    sy = j;
                    break;
                }
            }
            if (sx >= 0) break;
        }
        int cnt = 0;
        for (int[] d : dirs) {
            int x = sx + d[0], y = sy + d[1];
            while (inside(x, y, board)) {
                char c = board[x][y];
                if (c == 'B') {
                    break;
                }
                if (c == 'p') {
                    cnt++;
                    break;
                }
                x += d[0];
                y += d[1];
            }
        }
        return cnt;
    }
    
    private boolean inside (int x, int y, char[][] board) {
        return x >= 0 && y >= 0 && x < board.length && y < board[0].length;
    }
}
