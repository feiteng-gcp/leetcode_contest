class Solution {
    public int numRookCaptures(char[][] board) {
        int m = board.length;
        int n = board[0].length;
        
int I = 0;
        int J = 0;
        for(int i=0; i<board.length; i++) {
            for(int j=0; j<board[i].length; j++) {
                if(board[i][j]=='R') {
                    I = i;
                    J = j;
                    break;
                
}
            }
        }
        int res = 0;
        for(int i=(I+1)%m; i<m; i++) {
            if(board[i][J]=='.') continue;
            if(board[i][J]=='p') res++
;
            //if(board[i][J]=='B')
            break;
        }
        for(int i=(I-1)%m; i>=0; i--) {
            if(board[i][J]=='.') continue;
            if(board[i][J]=='p') res++
;
            //if(board[i][J]=='B')
            break;
        }
        for(int j=(J+1)%n; j<n; j++) {
            if(board[I][j]=='.') continue;
            if(board[I][j]=='p') res++
;
            //if(board[i][J]=='B')
            break;
        }
        for(int j=(J-1)%n; j>=0; j--) {
            if(board[I][j]=='.') continue;
            if(board[I][j]=='p') res++;
            //if(board[i][J]=='B')
            break;
        }
        
        return res;
    }
}
