class Solution {
    public int numRookCaptures(char[][] board) {
        int rx = 0, ry = 0;
        for(int i = 0; i < 8; i++) {
        	for(int j = 0; j < 8; j++) {
        		if(board[i][j] == 'R') {
        			rx = i; ry = j;
        		}

        	}
        }
        int res = 0;
        for(int i = rx + 1; i < 8; i++) {
        	if(board[i][ry] == 'B') {
        		break;
        	} else if(board[i][ry] == 'p') {
        		res++;
                break;
        	}
        }
        for(int i = rx - 1; i >= 0; i--) {
        	if(board[i][ry] == 'B') {
        		break;
        	} else if(board[i][ry] == 'p') {
        		res++;
                break;
        	}
        }
        for(int i = ry + 1; i < 8; i++) {
        	if(board[rx][i] == 'B') {
        		break;
        	} else if(board[rx][i] == 'p') {
        		res++;
                break;
        	}
        }
        for(int i = ry - 1; i >= 0; i--) {
        	if(board[rx][i] == 'B') {
        		break;
        	} else if(board[rx][i] == 'p') {
        		res++;
                break;
        	}
        }
		return res;
    }
}
