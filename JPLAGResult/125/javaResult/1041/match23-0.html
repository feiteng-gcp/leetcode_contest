import java.util.*;

class Solution {
    public int numRookCaptures(char[][] board) {

        int rrow = -1, rcol = -1;
        final int nrow = board.length;
        final int ncol = board[0].length;
        find: for (int row = 0; row < nrow; row++) {
            for (int col = 0; col < ncol; col++) {
                if (board[row][col] == 'R') {
                    rrow = row;
                    rcol = col;
                    break find;
                }
            }
        }

        int capture = 0;

        final int[] drow = new int[]{-1, +1, 0, 0};
        final int[] dcol = new int[]{0, 0, -1, +1};

        for (int ddir = 0; ddir < 4; ddir++) {

            int row2 = rrow + drow[ddir];
            int col2 = rcol + dcol[ddir];

            searchline: while(row2 >= 0 && row2 < nrow && col2 >= 0 && col2 < ncol) {
                // we are on the board.

                if (board[row2][col2] == '.') {
                    row2 += drow[ddir];
                    col2 += dcol[ddir];
                }
                else if (board[row2][col2] == 'B') {
                    break searchline;
                }
                else if (board[row2][col2] == 'p') {
                    capture++;
                    break searchline;
                }
            }
        }

        return capture;
    }
}
