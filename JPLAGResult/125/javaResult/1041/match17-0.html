class Solution {
    public int numRookCaptures(char[][] board) {
        int x=0,y=0;
        search:
        for(int i=0;i<8;i++){
            for(int j=0;j<8;j++){
                if(board[i][j]=='R'){
                    x = i;
                    y = j;
                    break search;
                }
            }
        }
        
        int res = 0;
        for(int i=x+1;i<=x+4&&i<8;i++){
            if(board[i][y]=='B'){
                break;
            } else if(board[i][y]=='p') {
                res++;
                break;
            }
        }
        for(int i=x-1;i>=x-4&&i>=0;i--){
            if(board[i][y]=='B'){
                break;
            } else if(board[i][y]=='p') {
                res++;
                break;
            }
        }
        for(int j=y+1;j<=y+4&&j<8;j++){
            if(board[x][j]=='B'){
                break;
            } else if(board[x][j]=='p') {
                res++;
                break;
            }
        }
        for(int j=y-1;j>=y-4&&j>=0;j--){
            if(board[x][j]=='B'){
                break;
            } else if(board[x][j]=='p') {
                res++;
                break;
            }
        }
        
        return res;
    }
}
