class Solution {
    public int numRookCaptures(char[][] board) {
        int len = board.length;
        if (len == 0) {
            return 0;
        }
        
int wid = board[0].length;
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < wid; j++) {
                if (board[i][j] == 'R') {
                    return getAns(board, i, j);
                }
            }
        }
        return -1;
    }
    public int getAns(char[][] board, int i, int j) {
        
int res = 0;
        int tmpI = i - 1;
        while (tmpI >= 0) {
            if (board[tmpI][j] >= 'a' && board[tmpI][j] <= 'z') {
                res ++;
                break;
            } else if (board[tmpI][j] >= 'A' && board[tmpI][j] <= 'Z') {
                break;
            }
            tmpI--;
        }
        tmpI = i + 1;
        while (tmpI < board.length) {
            if (board[tmpI][j] >= 'a' && board[tmpI][j] <= 'z') {
                res ++;
                break;
            } else if (board[tmpI][j] >= 'A' && board[tmpI][j] <= 'Z') {
                break;
            }
            tmpI++;
        }
        int tmpJ = j - 1;
        while (tmpJ >= 0) {
            if (board[i][tmpJ] >= 'a' && board[i][tmpJ] <= 'z') {
                res ++;
                break;
            } else if (board[i][tmpJ] >= 'A' && board[i][tmpJ] <= 'Z') {
                break;
            }
            tmpJ--;
        }
        tmpJ = j + 1;
        while (tmpJ < board[0].length) {
            if (board[i][tmpJ] >= 'a' && board[i][tmpJ] <= 'z') {
                res ++;
                break;
            } else if (board[i][tmpJ] >= 'A' && board[i][tmpJ] <= 'Z') {
                break;
            }
            tmpJ++;
        }
        return res;
    }
}
