class Solution {
    public int numRookCaptures(char[][] board) {
        
int x = -1;
        int y = -1;
        for (int i=0; i<8; i++) {
            for (int j=0; j<8; j++){
                if (board[i][j] == 'R') {
                    x=i;
                    y=j;
                    break;
                }
            }
            if (x!=-1) break;
        }
        int[][] dirs = 
new int[][] {{1,0},{-1,0},{0,1},{0,-1}};
        int result = 0;
        for (int[] dir:dirs){
            int x1 = x+dir[0];
            int y1 = y+dir[1];
            while (x1>=0 && x1<=7 && y1>=0 && y1<=7) {
                if (board[x1][y1] == 'B') break
;
                if (board[x1][y1] == 'p') {
                    result ++;
                    break;
                }
                x1 += dir[0];
                y1 += dir[1];
            }
        }
        return result;
    }
}
