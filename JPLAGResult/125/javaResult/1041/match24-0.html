class Solution {
    public int numRookCaptures(char[][] board) {
        int x = 0, y = 0;
        int n = board.length;
        for(int i=0; i<n; i++){
            for(int j=0; j<n; j++){
                if(board[i][j] == 'R')
                {
                    x = i;
                    y = j;
                    break;
                }
            }
        }
        
        int res = 0;
        int[] dirx = new int[]{0, 0, 1, -1};
        int[] diry = new int[]{1, -1, 0, 0};
        for(int i=0; i<4; i++){
            int newx = x;
            int newy = y;
            while(newx >=0 && newx< n && newy >=0 && newy < n){
                char c = board[newx][newy];
                if(c == '.' || c == 'R'){
                    newx += dirx[i];
                    newy += diry[i];
                }else if(c == 'p'){
                    res ++;
                    break;
                }else if(c == 'B'){
                    break;
                }
            }
        }
        
        return res;
    }
}
