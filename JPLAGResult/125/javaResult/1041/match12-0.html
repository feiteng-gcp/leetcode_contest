class Solution {
    public int numRookCaptures(char[][] b) {
        
        int dx[] = new int[]{0,0,1,-1};
        int dy[] = new int[]{1,-1,0,0};
        
        int l = b.length;
        int ll = b[0].length;
        
        int rx = -1;
        int ry = -1;
        int r = 0;
        
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                if(b[i][j]=='R'){
                    
                    for(int g= 0 ;g<4;++g){
                        int gx = dx[g];
                        int gy = dy[g];
                        
                        rx = i+gx;
                        ry = j+gy; 
                        
                        while(rx>=0&&rx<l&&ry>=0&&ry<ll&&b[rx][ry]=='.'){
                            rx += gx;
                            ry += gy;
                        }
                        
                        if(rx>=0&&rx<l&&ry>=0&&ry<ll&&Character.isLowerCase(b[rx][ry])){
                            ++r;
                        }
                        
                        
                    }
                }
            }
        }
        
        return r;
    }
}
