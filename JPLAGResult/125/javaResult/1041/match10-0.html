class Solution {
    public int numRookCaptures(char[][] board) {
        int Ri = 0;
        int Rj = 0;
        int count = 0;
        for (int i = 0; i < 8; ++i) {
            for (int j = 0; j < 8; ++j) {
                if (board[i][j] == 'R') {
                    Ri = i;
                    Rj = j;
                }
            }
        }
        for (int i = Ri - 1; i >= 0; --i) {
            if (board[i][Rj] == 'p') {
                count += 1;
                break;
            } else if (board[i][Rj] == 'B') {
                break;
            }
        }
        
        for (int i = Ri + 1; i < 8; ++i) {
            if (board[i][Rj] == 'p') {
                count += 1;
                break;
            } else if (board[i][Rj] == 'B') {
                break;
            }
        }
        
        for (int j = Rj - 1; j >= 0; --j) {
            if (board[Ri][j] == 'p') {
                count += 1;
                break;
            } else if (board[Ri][j] == 'B') {
                break;
            }
        }
        
        
        for (int j = Rj + 1; j < 8; ++j) {
            if (board[Ri][j] == 'p') {
                count += 1;
                break;
            } else if (board[Ri][j] == 'B') {
                break;
            }
        }
        
        return count;
    }
}
