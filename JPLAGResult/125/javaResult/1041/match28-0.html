class Solution {
	public static int N = 8;

	public int numRookCaptures(char[][] board) {
		int Rrow = -1;
		int Rcol = -1;
		for (int i = 0; i < 8; i++) {
			for (int k = 0; k < 8; k++) {
				if (board[i][k] == 'R') {
					Rrow = i;
					Rcol = k;
				}
			}
		}
		int res = 0;
		for (int k = Rcol - 1; k >= 0; k--) {
			if (board[Rrow][k] == '.')
				continue;
			if (board[Rrow][k] == 'p') {
				res++;
			}
			break;
		}
		for (int k = Rcol + 1; k < 8; k++) {
			if (board[Rrow][k] == '.')
				continue;
			if (board[Rrow][k] == 'p') {
				res++;
			}
			break;
		}
		for (int i = Rrow - 1; i >= 0; i--) {
			if (board[i][Rcol] == '.')
				continue;
			if (board[i][Rcol] == 'p') {
				res++;
			}
			break;
		}
		for (int i = Rrow + 1; i < 8; i++) {
			if (board[i][Rcol] == '.')
				continue;
			if (board[i][Rcol] == 'p') {
				res++;
			}
			break;
		}
		return res;
	}
}
