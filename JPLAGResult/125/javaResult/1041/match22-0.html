class Solution {
    public int numRookCaptures(char[][] board) {
        int ri = -1;
        int rj = -1;
        for (int i = 0; i < board.length; i++){
            for (int j = 0; j < board[i].length; j++){
                if (board[i][j] == 'R'){
                    ri = i;
                    rj = j;
                }
            }
        }
        
        int count = 0;
        int si = ri;
        while (si >= 0){
            if (board[si][rj] == 'p'){
                count++;
                break;
            } else if (board[si][rj] == 'B'){
                break;
            }
            si--;
        }
        
        si = ri;
        while (si < 8){
            if (board[si][rj] == 'p'){
                count++;
                break;
            } else if (board[si][rj] == 'B'){
                break;
            }
            si++;
        }
        
        int sj = rj;
        while (sj >= 0){
            if (board[ri][sj] == 'p'){
                count++;
                break;
            } else if (board[ri][sj] == 'B'){
                break;
            }
            sj--;
        }
        
        sj = rj;
        while (sj < 8){
            if (board[ri][sj] == 'p'){
                count++;
                break;
            } else if (board[ri][sj] == 'B'){
                break;
            }
            sj++;
        }
        
        return count;
    }
}
