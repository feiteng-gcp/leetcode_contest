class Solution {
    public int numRookCaptures(char[][] board) {
        int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        int m = board.length;
        int n = board[0].length;
        
        int x = 0, y = 0;
        
for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(board[i][j] == 'R') {
                    x = i;
                    y = j;
                }
            }
        
}
        
        int res = 0;
        
        for(int[] dir : dirs) {
            int nextx = x + dir[0];
            int nexty = y + dir[1];
            while(nextx >= 0 && nextx < m && nexty >= 0 && nexty < n && board[nextx][nexty] != 'B') {
                
if(board[nextx][nexty] == 'p') {
                    res++;
                    break;
                }
                
                nextx += dir[0];
                nexty += dir[1];
            }
        }
        
        return res;
    }
}
