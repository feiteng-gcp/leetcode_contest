class Solution {
public:
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        TreeNode* new_node = new TreeNode(val);
        if(root -> val < val)
        {
            new_node -> left = root;
            return new_node;
        }
        TreeNode *cur = root;
        while(cur -> right != NULL)
        {
            if(cur -> right -> val < val)
            {
                new_node -> left = cur -> right;
                cur -> right = new_node;
                return root;
            }
            cur = cur -> right;
        }
        cur -> right = new_node;
        return root;
    }
};
