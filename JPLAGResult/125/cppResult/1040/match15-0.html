/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vector<int>A;
public:
    void dfs(TreeNode *rt) {
        if (rt == NULL)
            return;
        dfs(rt -> left);
        A.push_back(rt -> val);
        dfs(rt -> right);
    }
    TreeNode *Construct(int l, int r) {
        if (l > r)
            return NULL;
        int i, pl = l;
        for (i = l + 1; i <= r; i++) {
            if (A[i] > A[pl]) {
                pl = i;
            }
        }
        TreeNode *rt = new TreeNode(A[pl]);
        rt -> left = Construct(l, pl - 1);
        rt -> right = Construct(pl + 1, r);
        return rt;
    }
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        A.clear();
        dfs(root);
        A.push_back(val);
        return Construct(0, A.size() - 1);
    }
};
