/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        if (!root){
            TreeNode *newroot = new TreeNode(val);
            return newroot;
        }
        if (val > root->val){
            TreeNode *newroot = new TreeNode(val);
            newroot->left = root;
            return newroot;
        }

        TreeNode *parent = root;
        while (true){
            TreeNode *here = parent->right;
            if (!here) {
                parent->right = new TreeNode(val);
                return root;
            }
            if (val > here->val) {
                TreeNode *newroot = new TreeNode(val);
                newroot->left = here;
                parent->right = newroot;
                return root;
            }
            parent = here;            
        }
        
    }
};
