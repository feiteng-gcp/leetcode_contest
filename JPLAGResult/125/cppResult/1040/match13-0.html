/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> vec;
    void dfs(TreeNode *u)
    {
        if(u==NULL) return;
        dfs(u->left);
        vec.push_back(u->val);
        dfs(u->right);
    }
    TreeNode* dfs2(int l,int r)
    {
        if(l>r) return NULL;
        if(l==r)
        {
            TreeNode *u=new TreeNode(vec[l]);
            return u;
        }
        int p=l;
        for(int i=l;i<=r;i++)
        {
            if(vec[i]>vec[p])
                p=i;
        }
        TreeNode *u=new TreeNode(vec[p]);
        u->left=dfs2(l,p-1);
        u->right=dfs2(p+1,r);
        return u;
    }
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        dfs(root);
        vec.push_back(val);
        return dfs2(0,vec.size()-1);
    }
};
