struct TreeNode{
	int val;
	TreeNode*left;
	TreeNode*right;
	TreeNode(int x):val(x),left(NULL),right(NULL){}
};
#endif


class Solution{
public:
	TreeNode* insertIntoMaxTree(TreeNode* root, int val){
		if(!root){
			TreeNode*n=new TreeNode(val);
			return n;
		}
		if(root->val<val){
			TreeNode*n=new TreeNode(val);
			n->left=root;
			return n;
		}
		root->right=insertIntoMaxTree(root->right,val);
		return root;
	}
};
