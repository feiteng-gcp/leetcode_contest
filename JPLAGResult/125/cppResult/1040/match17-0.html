/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;
    void dfs(TreeNode *now) {
        if(now == NULL) return;
        dfs(now->left);
        v.push_back(now->val);
        dfs(now->right);
    }
    
    TreeNode* m_tree(int l, int r) {
        if(l>r) return NULL;
        int n = l;
        for(int i = l+1 ; i <= r ; i++) {
            if(v[i] > v[n]) {
                n = i;
            }
        }
        TreeNode* ret = new TreeNode(v[n]);
        ret -> left = m_tree(l, n-1);
        ret -> right = m_tree(n+1, r);
        return ret;
    }
    
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        dfs(root);
        v.push_back(val);
        return m_tree(0, (int)v.size()-1);
    }
};
