/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        if (root == nullptr) {
            return new TreeNode(val);
        }
        if (root->val < val) {
            auto* p = new TreeNode(val);
            p->left = root;
            return p;
        }
        auto* p = root;
        while (p->right != nullptr && p->right->val > val) {
            p = p->right;
        }
        if (p->right == nullptr) {
            p->right = new TreeNode(val);
            return root;
        }
        auto* temp = p->right;
        p->right = new TreeNode(val);
        p->right->left = temp;
        return root;
    }
};
