/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        if (!root) return new TreeNode(val);
        if (val > root->val) {
            TreeNode *new_node = new TreeNode(val);
            new_node->left = root;
            return new_node;
        }
        else {
            TreeNode *sub_root = insertIntoMaxTree(root->right, val);
            root->right = sub_root;
            return root;
        }
    }
};
