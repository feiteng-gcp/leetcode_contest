class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int m = board.size();
        int n = board[0].size();
        int row = 0;
        int col = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 'R') {
                    row = i;col=j;
                }
            }
        }
        int dx[4] = {0, 1, -1, 0};
        int dy[4] = {1, 0, 0, -1};
        int ans = 0;
        for (int i = 0; i < 4; i++) {
            int trow = row;
            int tcol = col;
            while (true) {
                trow += dx[i];
                tcol += dy[i];
                if (trow < 0 || trow >= m) break;
                if (tcol < 0 || tcol >= n) break;
                if (board[trow][tcol] == 'B') break;
                if (board[trow][tcol] == 'p') {
                    ans++;
                    break;   
                }
            }
        }
        return ans;
    }
};
