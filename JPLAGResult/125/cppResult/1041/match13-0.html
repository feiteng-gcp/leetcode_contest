class Solution {
public:
    int solve(vector<vector<char>>& board, int x, int y)
    {
        int res = 0;
        for(int i=x-1;i>=0;i--)
        {
            if(board[i][y] == 'B') break;
            else if(board[i][y] == 'p')  {res++; break;}
        }
        for(int i=x+1;i<8;i++)
        {
            if(board[i][y] == 'B') break;
            else if(board[i][y] == 'p')  {res++; break;}
        }
        for(int j=y-1;j>=0;j--)
        {
            if(board[x][j] == 'B') break;
            else if(board[x][j] == 'p') {res++; break;}
        }
        for(int j=y+1;j<8;j++)
        {
            if(board[x][j] == 'B') break;
            else if(board[x][j] == 'p') {res++; break;}
        }
        return res;
    }
    int numRookCaptures(vector<vector<char>>& board) {
        
        for(int i =0;i<8;i++)
        {
            for(int j=0;j<8;j++)
            {
                if(board[i][j] == 'R')
                    return solve(board, i, j);
            }
        }
        return 0;
    }
};
