class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        vector<vector<int>> dir = {{0, -1}, {0, 1}, {1, 0}, {-1, 0}};
        
        int capture = 0;
        for (int i=0; i<8; ++i) {
            for (int j=0; j<8; ++j) {
                if (board[i][j] == 'R') {
                    for (int d = 0; d < 4; ++d) {
                        int x = i + dir[d][0];
                        int y = j + dir[d][1];
                        
                        while (x >= 0 && x < 8 && y >= 0 && y < 8) {
                            if (board[x][y] == 'p') {
                                capture++;
                                break;
                            } else if (board[x][y] == 'B') {
                                break;
                            }
                            
                            x += dir[d][0];
                            y += dir[d][1];
                        }
                    }
                    
                    break;
                }
            }
        }
        
        return capture;
    }
};
