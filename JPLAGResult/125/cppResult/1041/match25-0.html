class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int res = 0;
        int i = 0, j = 0;
        for(int a = 0;a < 8;a++)
            for(int b = 0;b<8;b++)
                if(board[a][b] == 'R'){
                    i = a;
                    j = b;
            }
        
        for(int left = j-1; left >= 0; left--)
            if(board[i][left] >= 'A' && board[i][left] <= 'Z')
                break;
            else if(board[i][left] == 'p') {
                res++;
                break;
            }
        for(int right = j+1; right < 8; right++)
            if(board[i][right] >= 'A' && board[i][right] <= 'Z')
                break;
            else if(board[i][right] == 'p') {
                res++;
                break;
            }
        for(int up = i-1; up >= 0; up--)
            if(board[up][j] >= 'A' && board[up][j] <= 'Z')
                break;
            else if(board[up][j] == 'p') {
                res++;
                break;
            }
        for(int down = i+1; down < 8; down++)
            if(board[down][j] >= 'A' && board[down][j] <= 'Z')
                break;
            else if(board[down][j] == 'p') {
                res++;
                break;
            }
        return res;
    }
};
