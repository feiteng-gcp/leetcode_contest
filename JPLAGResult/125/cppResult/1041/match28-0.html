class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int row,col;
        for(int i=0;i<8;i++)
            for(int j=0;j<8;j++)
                if(board[i][j]=='R')
                {
                    row=i;
                    col=j;
                    break;
                }
        int i,j;
        int ans=0;
        for(i=row+1,j=col;i<8;i++)
        {
            if(board[i][j]=='p')
            {
                ans++;break;
            }
            else if(board[i][j]=='B')
            {
                break;
            }
        }
        for(i=row-1,j=col;i>=0;i--)
        {
            if(board[i][j]=='p')
            {
                ans++;break;
            }
            else if(board[i][j]=='B')
            {
                break;
            }
        }
        for(i=row,j=col+1;j<8;j++)
        {
            if(board[i][j]=='p')
            {
                ans++;break;
            }
            else if(board[i][j]=='B')
            {
                break;
            }
        }
        for(i=row,j=col-1;j>=0;j--)
        {
            if(board[i][j]=='p')
            {
                ans++;break;
            }
            else if(board[i][j]=='B')
            {
                break;
            }
        }
        return ans;
    }
};
