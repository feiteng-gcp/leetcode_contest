class Solution {
public:
    	int numRookCaptures(vector<vector<char>>& board) {
        	int r = 0, c = 0;
		int n = board.size(), m = board[0].size();

		for(int i = 0; i < n; ++i) {
	for (int j = 0; j < m; ++j) {
	if(board[i][j] == 'R') {
	r = i;
	c = j;

}
}
}
int res = 0;
for (int i = r; i >= 0; i--) {
	if (board[i][c] == 'B') break;

	else if (board[i][c] == 'p') {
	res++;
	break;
} 
}
for (int i = c; i >= 0; i--) {
	if (board[r][i] == 'B') break;

	else if (board[r][i] == 'p') {
	res++;
	break;
} 
}
for (int i = r; i < n; i++) {
	if (board[i][c] == 'B') break;

	else if (board[i][c] == 'p') {
	res++;
	break;
} 
}
for (int i = c; i < m; i++) {
	if (board[r][i] == 'B') break;
	else if (board[r][i] == 'p') {
	res++;
	break;
} 
}
return res;
    	}
};
