int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};

class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int n = board.size();
        int m = board[0].size();
        

        int sx, sy;
        for (int i = 0; i < n; ++ i) {
            for (int j = 0; j < m; ++ j) {
                if (board[i][j] == 'R') {
                    sx = i, sy = j;
                }
            }
        }
        
        int ans = 0;
        for (int k = 0; k < 4; ++ k) {
            int x = sx, y = sy;
            while ( 1 ) {
                x += dx[k];
                y += dy[k];
                if (x < 0 || y < 0 || x >= n || y >= m) break;
                if (board[x][y] != '.') {
                    if (board[x][y] == 'p') ++ ans;
                    break;
                }
            }
        }
        return ans;
    }
};
