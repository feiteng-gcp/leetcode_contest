class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int N = board.size();
        
        int ri, rj;

        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (board[i][j] == 'R') {
                    ri = i;
                    rj = j;
                }
            }
        }
        

        int cnt = 0;
        for (int i = ri + 1; i < N; i++) {
            if (board[i][rj] == 'p') cnt++;
            if (board[i][rj] != '.') break;
        }
        for (int i = ri - 1; i >= 0; i--) {
            if (board[i][rj] == 'p') cnt++;
            if (board[i][rj] != '.') break;
        }
        for (int j = rj + 1; j < N; j++) {
            if (board[ri][j] == 'p') cnt++;
            if (board[ri][j] != '.') break;
        }
        for (int j = rj - 1; j >= 0; j--) {
            if (board[ri][j] == 'p') cnt++;
            if (board[ri][j] != '.') break;
        }
        
        return cnt;
    }
};
