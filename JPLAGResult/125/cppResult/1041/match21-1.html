class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        vector<vector<char>> a=board;

        int x, y;
        for(int i=0; i<8; i++) {
            for(int j=0; j<8; j++) {
                if (a[i][j]=='R') {
                    x=i;
                    y=j;
                }
            }
        }
        int dx[]={-1, 1, 0, 0};

        int dy[]={ 0, 0,-1, 1};
        int ans=0;
        for(int i=0; i<4; i++) {
            for(int j=1; j<=8; j++) {
                int Nx=x+j*dx[i];
                int Ny=y+j*dy[i];
                if (Nx<0||Ny<0||Nx>7||Ny>7) break;;
                if (a[Nx][Ny]=='B') break;
                if (a[Nx][Ny]=='p') {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
