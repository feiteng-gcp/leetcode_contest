class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) 
    {
        int Row = -1, Col = -1;
        for (int i = 0; i < 8; i ++)
        {
            for (int j = 0; j < 8; j ++)
                if (board[i][j] == 'R')
                {
                    Row = i; Col = j; break;
                }
            if (Row != -1) break;
        }
        int res = 0;
        for (int i = Col + 1; i < 8; i ++)
        {
            if (board[Row][i] == 'B')
                break;

            else if (board[Row][i] == 'p')
            {
                res ++;
                break;
            }
        }
        for (int i = Col - 1; i >= 0; i --)
        {
            if (board[Row][i] == 'B')
                break;

            else if (board[Row][i] == 'p')
            {
                res ++;
                break;
            }
        }       
        for (int i = Row + 1; i < 8; i ++)
        {
            if (board[i][Col] == 'B')
                break;

            else if (board[i][Col] == 'p')
            {
                res ++;
                break;
            }
        }       
        for (int i = Row - 1; i >= 0; i --)
        {
            if (board[i][Col] == 'B')
                break;
            else if (board[i][Col] == 'p')
            {
                res ++;
                break;
            }
        }
        return res;
    }
};
