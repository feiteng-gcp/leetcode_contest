class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int ri,rj;
        rep(i,8)rep(j,8){
            if (board[i][j]=='R') {ri=i; rj=j; break; }
        }
        int cnt=0;
        for (int i=ri-1; i>=0; --i){
            if (board[i][rj]=='.') continue;
            if (board[i][rj]=='B') break;
            if (board[i][rj]=='p') {++cnt; break;}
        }
        for (int i=ri+1; i<8; ++i){
            if (board[i][rj]=='.') continue;
            if (board[i][rj]=='B') break;
            if (board[i][rj]=='p') {++cnt; break;}
        }
        for (int j=rj-1; j>=0; --j){
            if (board[ri][j]=='.') continue;
            if (board[ri][j]=='B') break;
            if (board[ri][j]=='p') {++cnt; break;}
        }
        for (int j=rj+1; j<8; ++j){
            if (board[ri][j]=='.') continue;
            if (board[ri][j]=='B') break;
            if (board[ri][j]=='p') {++cnt; break;}
        }
        return cnt;
    }
};
