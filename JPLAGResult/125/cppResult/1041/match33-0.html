class Solution {
public:
    int numRookCaptures(vector<vector<char>>& A) {
        int xs = 0;
        int ys = 0;
        int dx[4] = {-1,1,0,0};
        int dy[4] = {0,0,-1,1};
        for (int i = 0; i < 8; ++i) {
            for (int j = 0; j < 8; ++j) {
                if (A[i][j]=='R') {
                    xs = i;
                    ys = j;
                }
            }
        }
        int r = 0;
        for (int d = 0; d < 4; ++d) {
            int x = xs;
            int y = ys;
            while(1) {
                int xx = x + dx[d];
                int yy = y + dy[d];
                if (xx<0 || yy<0 || xx>7 || yy>7 || A[xx][yy]=='B') break;
                if (A[xx][yy]=='p') {++r; break;}
                x = xx;
                y = yy;
            }
        }
        return r;
    }
};
