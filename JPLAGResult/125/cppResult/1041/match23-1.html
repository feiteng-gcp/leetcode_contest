class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        
        int rx,ry;
        for(int i = 0; i < 8; i++) {
            for(int j = 0; j < 8; j++) {
                if(board[i][j] == 'R') {
                    rx = i;
                    ry = j;
                }
            }
        }
        int dx[4] = {-1,1,0,0};
        int dy[4] = {0,0,-1,1};
        int ans = 0;
        for(int i = 0; i < 4; i++) {
            int x = rx;

            int y = ry;
            while(x < 8 && x >= 0 && y < 8 && y > 0) {
                if(board[x][y] == 'B') break;
                if(board[x][y] == 'p') {
                    ans++;
                    break;
                }
                x += dx[i];
                y += dy[i];
            }
        }
        return ans;
    }
};
