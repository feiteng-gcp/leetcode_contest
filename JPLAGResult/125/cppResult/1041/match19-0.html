#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int numRookCaptures(vector<vector<char>>& b) {
        int x, y;
        for (int i = 0; i < 8; i++) {
            for (int j = 0; j < 8; j++) {
                if (b[i][j] == 'R') {
                    x = i, y = j;
                }
            }
        }
        int cnt = 0;
        for (int i = x + 1; i < 8; i++) {
            if (b[i][y] == '.') continue;
            if (b[i][y] == 'p') {
                cnt++;
                break;
            }
            if (b[i][y] == 'B') break;
        }
        for (int i = x - 1; i >= 0; i--) {
            if (b[i][y] == '.') continue;
            if (b[i][y] == 'p') {
                cnt++;
                break;
            }
            if (b[i][y] == 'B') break;
        }
        for (int i = y + 1; i < 8; i++) {
            if (b[x][i] == '.') continue;
            if (b[x][i] == 'p') {
                cnt++;
                break;
            }
            if (b[x][i] == 'B') break;
        }
        for (int i = y - 1; i >= 0; i--) {
            if (b[x][i] == '.') continue;
            if (b[x][i] == 'p') {
                cnt++;
                break;
            }
            if (b[x][i] == 'B') break;
        }
        return cnt;
    }
};
