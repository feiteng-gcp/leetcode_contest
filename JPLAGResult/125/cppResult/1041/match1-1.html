class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int rx, ry;
        for (int i = 0; i < 8; i++) {
            for (int j = 0; j < 8; j++) {
                if (board[i][j] == 'R') {
                    rx = i;
                    ry = j;
                }

            }
        }
        int cnt = 0;
        for (int j = ry - 1; j >= 0; j--) {
            if (board[rx][j] == 'p') {
                ++cnt;
                break;
            } else if (board[rx][j] == 'B') break;

        }
        for (int j = ry + 1; j < 8; j++) {
            if (board[rx][j] == 'p') {
                ++cnt;
                break;
            } else if (board[rx][j] == 'B') break;

        }
        for (int i = rx - 1; i >= 0; i--) {
            if (board[i][ry] == 'p') {
                ++cnt;
                break;
            } else if (board[i][ry] == 'B') break;

        }
        for (int i = rx + 1; i < 8; i++) {
            if (board[i][ry] == 'p') {
                ++cnt;
                break;
            } else if (board[i][ry] == 'B') break;
        }
        return cnt;
    }
};
