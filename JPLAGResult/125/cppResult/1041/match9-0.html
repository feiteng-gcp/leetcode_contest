#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <string>
#include <map>
#include <set>
#include <list>
#include <vector>
#include <algorithm>
#include <utility>
using namespace std;

class Solution {
 public:
  int numRookCaptures(vector<vector<char>>& b) {
    int ret = 0;
    for (int y = 0; y < 8; ++y) {
      for (int x = 0; x < 8; ++x) {
        if (b[y][x] == 'R') {
          for (int yy = y + 1; yy < 8; ++yy) {
            if (b[yy][x] == '.')
              continue;
            if (b[yy][x] == 'p')
              ++ret;
            break;
          }
          for (int yy = y - 1; yy >= 0; --yy) {
            if (b[yy][x] == '.')
              continue;
            if (b[yy][x] == 'p')
              ++ret;
            break;
          }
          for (int xx = x + 1; xx < 8; ++xx) {
            if (b[y][xx] == '.')
              continue;
            if (b[y][xx] == 'p')
              ++ret;
            break;
          }
          for (int xx = x - 1; xx >= 0; --xx) {
            if (b[y][xx] == '.')
              continue;
            if (b[y][xx] == 'p')
              ++ret;
            break;
          }
        }
      }
    }
    return ret;
  }
};
