int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};

class Solution {
public:
    int numRookCaptures(vector<vector<char>>& b) {
        int x, y;
        

        for (int i = 0; i < 8; ++i) {
            for (int j = 0; j < 8; ++j) {
                if (b[i][j] == 'R') {
                    x = i;
                    y = j;
                }
            }
        }
        
        int res = 0;
        for (int k = 0; k < 4; ++k) {
            int xx = x, yy = y;
            while (xx >= 0 && xx < 8 && yy >= 0 && yy < 8) {

                if (b[xx][yy] == 'B') break;
                if (b[xx][yy] == 'p') {
                    ++res;
                    break;
                }
                xx += dx[k];
                yy += dy[k];
            }
        }
        
        return res;
    }
};
