class Solution {
public:
    int numRookCaptures(vector<vector<char>>& b) {
        int i0,j0;
        for(int i=0;i<8;i++){
            for(int j=0;j<8;j++){
                if(b[i][j]=='R') i0=i,j0=j;

            }
        }
        int cnt=0;
        for(int i=i0+1;i<8;i++){
            if(b[i][j0]=='p'){
                cnt++;break;
            }else if(b[i][j0]!='.'){
                break;
            }

        }for(int i=i0-1;i>=0;i--){
            if(b[i][j0]=='p'){
                cnt++;break;
            }else if(b[i][j0]!='.'){
                break;
            }

        }for(int j=j0+1;j<8;j++){
            if(b[i0][j]=='p'){
                cnt++;break;
            }else if(b[i0][j]!='.'){
                break;
            }

        }for(int j=j0-1;j>=0;j--){
            if(b[i0][j]=='p'){
                cnt++;break;
            }else if(b[i0][j]!='.'){
                break;
            }
        }
        return cnt;
    }
};
