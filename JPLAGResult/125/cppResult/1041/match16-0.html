class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int si, sj;
        
        for (int i = 0; i < board.size(); ++i) {
            for (int j = 0; j < board[0].size(); ++j) {
                if (board[i][j] == 'R') {
                    si = i;
                    sj = j;
                    goto next;
                }
            }
        }
        
        next:
        
        int ans = 0;
        for (int i = si - 1; i >= 0; --i) {
            if (board[i][sj] == 'B') break;
            if (board[i][sj] == 'p') {
                ans++;
                break;
            }
        }
        for (int i = si + 1; i <board.size(); ++i) {
            if (board[i][sj] == 'B') break;
            if (board[i][sj] == 'p') {
                ans++;
                break;
            }
        }
        for (int j = sj - 1; j >= 0; --j) {
            if (board[si][j] == 'B') break;
            if (board[si][j] == 'p') {
                ans++;
                break;
            }
        }
        for (int j = sj + 1; j < board[0].size(); ++j) {
            if (board[si][j] == 'B') break;
            if (board[si][j] == 'p') {
                ans++;
                break;
            }
        }
        return ans;
    }
};
