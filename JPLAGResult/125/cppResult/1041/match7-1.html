class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {

        int res = 0;
        for (int y = 0; y < 8; ++y){
            for (int x = 0; x < 8; ++x){
                if (board[y][x] == 'R'){

                    for (int i = x - 1; i >= 0; --i){
                        if (board[y][i] == 'p'){
                            ++res;
                            break;
                        }
                        if (board[y][i] == 'B') break;
                    }
                    for (int i = x + 1; i < 8; ++i){
                        if (board[y][i] == 'p'){
                            ++res;
                            break;
                        }
                        if (board[y][i] == 'B') break;                        
                    }
                    for (int i = y - 1; i >= 0; --i){
                        if (board[i][x] == 'p'){
                            ++res;
                            break;
                        }
                        if (board[i][x] == 'B') break; 
                    }
                    for (int i = y + 1; i < 8; ++i){
                        if (board[i][x] == 'p'){
                            ++res;
                            break;
                        }
                        if (board[i][x] == 'B') break; 
                    }
                    return res;
                }
            }
        }
        return 0;
    }
};
