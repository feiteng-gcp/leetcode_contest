class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int h = board.size();
        int w = board[0].size();
        int dy[4] = {-1, 0, 1, 0};
        int dx[4] = {0, 1, 0, -1};
        int ry, rx;
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                if (board[i][j] == 'R') {
                    ry = i, rx = j;
                }
            }
        }
        int ans = 0;
        for (int dir = 0; dir < 4; dir++) {
            for (int k = 0; k < 8; k++) {
                int y = ry + dy[dir] * k;
                int x = rx + dx[dir] * k;
                if (y < 0 || x < 0 || y >= h || x >= w) break;
                if (board[y][x] == 'B') break;
                if (board[y][x] == 'p') {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
