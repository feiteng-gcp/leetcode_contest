class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int res = 0, rooki, rookj;
        bool flag = false;
        for (rooki = 0; rooki < 8; ++rooki){
            for (rookj = 0; rookj < 8; ++rookj){
                if(board[rooki][rookj] == 'R'){flag = true; break;}
            }
            if (flag == true)break;
        }

        for(int i = rooki; i >= 0; --i){
            if(board[i][rookj] == 'B')break;
            else if(board[i][rookj] == 'p'){
                ++res;
                break;
            }
        }
        for(int i = rooki; i < 8; ++i){
            if(board[i][rookj] == 'B')break;
            else if(board[i][rookj] == 'p'){
                ++res;
                break;
            }
        }
        for(int j = rookj; j >= 0; --j){
            if(board[rooki][j] == 'B')break;
            else if(board[rooki][j] == 'p'){
                ++res;
                break;
            }
        }
        for(int j = rookj; j < 8; ++j){
            if(board[rooki][j] == 'B')break;
            else if(board[rooki][j] == 'p'){
                ++res;
                break;
            }
        }
        return res;
    }
};
