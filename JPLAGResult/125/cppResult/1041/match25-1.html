class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) 
    {
        int x=-1,y=-1,count=0;
        for(int i=0;i<board.size()&&x==-1&&y==-1;i++)
            for(int j=0;j<board.size()&&x==-1&&y==-1;j++)
                if(board[i][j]=='R')
                    x=i,y=j;

        for(int i=x+1;i<board.size();i++)
            if(board[i][y]=='B')
                break;
            else if(board[i][y]=='p')
            {
                count++;
                break;
            }
        for(int i=x-1;i>=0;i--)
            if(board[i][y]=='B')
                break;
            else if(board[i][y]=='p')
            {
                count++;
                break;
            }
        for(int i=y-1;i>=0;i--)
            if(board[x][i]=='B')
                break;
            else if(board[x][i]=='p')
            {
                count++;
                break;
            }
        for(int i=y+1;i<board.size();i++)
            if(board[x][i]=='B')
                break;
            else if(board[x][i]=='p')
            {
                count++;
                break;
            }
        return count;
    }
};
