class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int res = 0;

        for(int i = 0; i < board.size(); i++){
            for(int j = 0; j < board[0].size(); j++){
                if(board[i][j] == 'R'){

                    for(int ip = i; ip >= 0; ip--){
                        if(board[ip][j] == 'B')
                            break;
                        if(board[ip][j] == 'p'){
                            res++;
                            break;
                        }
                            
                    }
                    for(int ip = i; ip < 8; ip++){
                        if(board[ip][j] == 'B')
                            break;
                        if(board[ip][j] == 'p'){
                            res++;
                            break;
                        }
                            
                    }
                    for(int jp = j; jp >= 0; jp--){
                        if(board[i][jp] == 'B')
                            break;
                        if(board[i][jp] == 'p'){
                            res++;
                            break;
                        }
                            
                    }
                    for(int jp = j; jp <8; jp++){
                        if(board[i][jp] == 'B')
                            break;
                        if(board[i][jp] == 'p'){
                            res++;
                            break;
                        }
                            
                    }
                }
            }
        }
        return res;
    }
};
