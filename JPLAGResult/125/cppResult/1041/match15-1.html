class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        int r=board.size(),c=board[0].size();
        int x,y;
        for(x=0;x<r;x++){
            bool get=false;
            for(y=0;y<c;y++){
                if(board[x][y]=='R'){
                    get=true;
                    break;
                }
            }
            if(get) break;

        }
        int ans=0;
        for(int m=x-1;m>=0;m--){
            if(board[m][y]=='p'){
                ans++;
                break;
            }
            else if(board[m][y]=='B')
                break;
        }
        for(int m=x+1;m<r;m++){
            if(board[m][y]=='p'){
                ans++;
                break;
            }
            else if(board[m][y]=='B')
                break;
        }
        for(int n=y-1;n>=0;n--){
            if(board[x][n]=='p'){
                ans++;
                break;
            }
            else if(board[x][n]=='B')
                break;
        }
        for(int n=y+1;n<c;n++){
            if(board[x][n]=='p'){
                ans++;
                break;
            }
            else if(board[x][n]=='B')
                break;
        }
        return ans;
    }
};
