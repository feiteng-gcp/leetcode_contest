class Solution {
public:
    int numRookCaptures(vector<vector<char>>& board) {
        auto rookPos = findRook(board);
        int answer = 0;
        
        for (int i = 1; rookPos.first+i < board.size(); i++) {
            if (board[rookPos.first+i][rookPos.second] == 'p') answer += 1;
            if (board[rookPos.first+i][rookPos.second] != '.') break;
        }
        
        for (int j = 1; rookPos.second+j < board[rookPos.first].size(); j++) {
            if (board[rookPos.first][rookPos.second+j] == 'p') answer += 1;
            if (board[rookPos.first][rookPos.second+j] != '.') break;
        }
        
        for (int i = -1; rookPos.first+i >= 0; i--) {
            if (board[rookPos.first+i][rookPos.second] == 'p') answer += 1;
            if (board[rookPos.first+i][rookPos.second] != '.') break;
        }
        
        for (int j = -1; rookPos.second+j >= 0; j--) {
            if (board[rookPos.first][rookPos.second+j] == 'p') answer += 1;
            if (board[rookPos.first][rookPos.second+j] != '.') break;
        }
        
        return answer;
    }
    
    pair<int, int> findRook(const vector<vector<char>> &board) {
        for (int i = 0; i < board.size(); i++) {
            for (int j = 0 ; j < board[i].size(); j++) {
                if (board[i][j] == 'R') {
                    return make_pair(i, j);
                }
            }
        }
        return make_pair(-1, -1);
    }
};
