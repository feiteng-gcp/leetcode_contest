class Solution {
public:
    int findJudge(int n, vector<vector<int>>& ts) {
        vector<int> cnt(n+1,0);
        vector<int> gcnt(n+1,0);
        for(auto t: ts){
            cnt[t[1]]++;
            gcnt[t[0]]++;
        }
        int v = -1;
        for(int i=1;i<=n;i++){
            if(cnt[i]==n-1 && gcnt[i]==0){
                if(v==-1){
                    v = i;
                }else{
                    return -1;
                }
            }
        }
        
        return v;
        
    }
};
