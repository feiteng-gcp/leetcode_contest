class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> in(N+1);
        vector<int> out(N+1);
                
        for (auto &p : trust) {
            in[p[1]]++;
            out[p[0]]++;
        }
        
        int judge = -1;
        for (int i=1; i<=N; ++i) {
            if (in[i] == N-1 && out[i] == 0) {
                if (judge == -1) {
                    judge = i;
                }
                else {
                    return -1;
                }
            }
        }
        
        return judge;
    }
};
