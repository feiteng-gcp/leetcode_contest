class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        int trustCount[1001];
        int trustOtherCount[1001];
        
        for (int i = 0; i <= N; ++i) {
            trustCount[i] = trustOtherCount[i] = 0;
        }
        
        int len = trust.size();
        for (int i = 0; i < len; ++i) {
            int n1 = trust[i][0], n2 = trust[i][1];
            
            ++trustCount[n2];
            ++trustOtherCount[n1];
        }
        
        for (int i = 1; i <= N; ++i) {
            if (trustCount[i] == N-1 && trustOtherCount[i] == 0) {
                return i;
            }
        }
        
        return -1;
    }
};
