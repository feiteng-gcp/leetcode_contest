class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> trustCount(N + 1, 0);
        vector<bool> canBeJudge(N + 1, true);
        for (auto& t : trust) {
            trustCount[t[1]]++;
            canBeJudge[t[0]] = false;
        }
        for (int i = 1; i <= N; i++) {
            if (canBeJudge[i] && trustCount[i] == N - 1) {
                return i;
            }
        }
        return -1;
    }
};
