#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <string>
#include <map>
#include <set>
#include <list>
#include <vector>
#include <algorithm>
#include <utility>
using namespace std;

class Solution {
 public:
  int findJudge(int N, vector<vector<int>>& trust) {
    vector<int> trusts(N, 0);
    vector<int> trusted_by(N, 0);
    for (auto& t : trust) {
      ++trusts[t[0] - 1];
      ++trusted_by[t[1] - 1];
    }
    for (int i = 0; i < N; ++i) {
      if (trusts[i] == 0 && trusted_by[i] == N - 1)
        return i + 1;
    }
    return -1;
  }
};
