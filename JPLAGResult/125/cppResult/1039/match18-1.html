class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> in_deg(N);
        vector<int> out_deg(N);
        for(auto& e : trust) {
            out_deg[e[0]-1]++;
            in_deg[e[1]-1]++;
        }
        for(int i=0; i<N; i++) {
            if(in_deg[i]==N-1&&out_deg[i]==0)
                return i+1;
        }
        return -1;
    }
};
