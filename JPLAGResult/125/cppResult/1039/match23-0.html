class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> out(N+1), in(N+1);
        for (auto p : trust) {
            int u = p[0], v = p[1];
            out[u]++;
            in[v]++;
        }
        for (int u = 1; u <= N; ++u) {
            if (out[u] == 0 && in[u] == N-1)
                return u;
        }
        return -1;
    }
};
