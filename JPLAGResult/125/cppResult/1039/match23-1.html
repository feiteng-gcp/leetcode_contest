class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> A(N+1), B(N+1);
        for (auto pr: trust) {
            int x = pr[0], y = pr[1];
            A[x]++;
            B[y]++;
        }
        for (int i = 1; i <= N; ++i) {
            if (A[i] == 0 && B[i] == N-1) return i;
        }
        return -1;
    }
};
