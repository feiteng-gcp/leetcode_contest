class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        map<int, int> numPersonTrustsMap;
        map<int, int> numPersonTrustedByMap;
        
        for (int i = 1; i <= N; i++)
        {
            numPersonTrustsMap[i] = 0;
            numPersonTrustedByMap[i] = 0;
        }
        
        for (vector<int>& trustEdge : trust)

        {
            numPersonTrustsMap[trustEdge[0]]++;
            numPersonTrustedByMap[trustEdge[1]]++;
        }
        
        for (int i = 1; i <= N; i++)
        {
            if ((numPersonTrustsMap[i] == 0) && (numPersonTrustedByMap[i] == (N - 1)))
            {
                return i;
            }
        }
        
        return -1;
    }
};
