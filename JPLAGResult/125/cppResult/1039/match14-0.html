class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
   //     N=4,trust = {{1,3},{1,4},{2,3},{2,4},{4,3}};
        vector<pair<int, int>> t(N + 1, make_pair(0, 0));
        for (auto &v: trust) {
            t[v[0]].first++;
            t[v[1]].second++;
        }
        
        for (int i = 1; i <= N; ++i) {
            if (t[i].first == 0 && t[i].second == N - 1) return i;
        }
        
        return -1;
    }
};
