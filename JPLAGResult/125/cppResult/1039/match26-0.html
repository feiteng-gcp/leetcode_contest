class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> cnts(N + 1, 0);
        vector<bool> trust_other(N + 1, 0);
        for (int i = 0; i < trust.size(); i++) {
            cnts[trust[i][1]]++;
            trust_other[trust[i][0]] = true;
        }
        
        for (int i = 1; i <= N; i++) {
            if (cnts[i] == N - 1 && !trust_other[i]) {
                return i;
            }
        }
        return -1;
    }
};
