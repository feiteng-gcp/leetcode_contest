class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> trust_count(N, 0);
        vector<bool> trust_someone(N, false);
        for (auto v : trust) {
            ++trust_count[v[1]-1];
            trust_someone[v[0]-1] = true;
        }
        for (int i = 0; i < N; ++i) {
            if (trust_count[i] == N -1 && !trust_someone[i]) 
                return i + 1;
        }
        return -1;
    }
};
