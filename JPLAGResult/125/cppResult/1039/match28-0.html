class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        map<int,int> trusted;
        map<int,int> trusts;
        for (auto &vec: trust) {
            int u = vec[0], v = vec[1];
            trusted[v] += 1;
            trusts[u] += 1;
        }
        for (int i=1; i<=1000; ++i) {
            if (trusts[i] == 0 && trusted[i] == N-1) {
                return i;
            }
        }
        return -1;
    }
};
