class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        vector<int> g(N+1,1);
        vector<int> h(N+1,0);
        for(auto& v:trust){
            g[v[0]]=0;
            h[v[1]]++;
        }
        for(int i=1;i<=N;i++){
            if(g[i] && h[i]==N-1){
                return i;
            } 
        }
        
        return -1;
        
    }
};
