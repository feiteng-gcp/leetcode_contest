class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        map<int,int> rows;
        map<int,int> columns;
        map<int,int> leftright;
        map<int,int> rightleft;
        map<int,map<int,int>> grid;
        for(vector<int> lamp:lamps){
            int x = lamp[0];
            int y = lamp[1];
            grid[x][y] = 1;
            rows[x]++;
            columns[y]++;
            leftright[x + y]++;
            rightleft[y - x]++;
        }
        vector<int> res;
        for(vector<int> query:queries){
            int x = query[0];
            int y = query[1];
            if(rows[x] > 0 || columns[y] > 0 || leftright[x + y] > 0 || rightleft[y - x] > 0)res.push_back(1);
            else res.push_back(0);
            for(int i = x - 1;i <= x + 1;i++){
                for(int j = y - 1;j <= y + 1;j++){
                    if(i < 0 || i >= N || j < 0 || j >= N)continue;

                    if(grid[i][j] == 1){
                        rows[i]--;
                        columns[j]--;
                        leftright[i + j]--;
                        rightleft[j - i]--;
                    }
                }
            }
        }
        return res;
        
    }
};
