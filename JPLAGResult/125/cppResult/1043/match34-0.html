class Solution{
public:
	vector<int> gridIllumination(int N, vector<vector<int>> lamps, vector<vector<int>> queries) {
		map<int,int>sx,sy,sa,sb;
		map<P,int>mp;
		for(auto p:lamps){
			sx[p[0]]++;
			sy[p[1]]++;
			sa[p[0]+p[1]]++;
			sb[p[0]-p[1]]++;
			mp[{p[0],p[1]}]++;
		}
		vector<int>ans;
		for(auto p:queries){
			if(sx[p[0]]||sy[p[1]]||sa[p[0]+p[1]]||sb[p[0]-p[1]]){
				ans.push_back(1);
			}
			else ans.push_back(0);
			for(int i=-1;i<=1;i++)for(int j=-1;j<=1;j++){
				int nx=p[0]+i,ny=p[1]+j;
				if(mp[{nx,ny}]){
					mp[{nx,ny}]--;
					sx[nx]--;sy[ny]--;sa[nx+ny]--;sb[nx-ny]--;
				}
			}
		}
		return ans;
	}
};
