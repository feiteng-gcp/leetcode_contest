class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        map<int, int> vrt, hrz, ld, rd;
        map<int, map<int,bool>> has_lamp;
        for (auto&& lamp : lamps) {
            has_lamp[lamp[0]][lamp[1]] = true;
            ++hrz[lamp[0]];
            ++vrt[lamp[1]];
            ++ld[lamp[0]+lamp[1]];
            ++rd[lamp[0]-lamp[1]];
        }
        vector<int> ans(queries.size(), 0);
        for (int i = 0; i < queries.size(); ++i) {
            int qi = queries[i][0], qj = queries[i][1];
            if (hrz.count(qi) == 1 && hrz[qi] > 0) ans[i] = 1;
            if (vrt.count(qj) == 1 && vrt[qj] > 0) ans[i] = 1;
            if (ld.count(qi+qj) == 1 && ld[qi+qj] > 0) ans[i] = 1;
            if (rd.count(qi-qj) == 1 && rd[qi-qj] > 0) ans[i] = 1;
            

            for (int li = qi - 1; li <= qi + 1; ++li) {
                for (int lj = qj - 1; lj <= qj + 1; ++lj) {
                    if (has_lamp.count(li) && has_lamp[li].count(lj)) {
                        --hrz[li];
                        --vrt[lj];
                        --ld[li+lj];
                        --rd[li-lj];
                        has_lamp[li].erase(lj);
                    }
                }
            }
        }
        return ans;
    }
};
