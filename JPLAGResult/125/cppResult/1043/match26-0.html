class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        set<pair<int,int>> L;
        unordered_map<int,int> X;
        unordered_map<int,int> Y;
        unordered_map<int,int> DN;
        unordered_map<int,int> DP;
        for (auto x : lamps) {
            L.insert({x[0],x[1]});
            X[x[0]]++;
            Y[x[1]]++;
            DN[x[0]-x[1]]++;
            DP[x[0]+x[1]]++;
        }
        vector<int> res;
        for (auto q : queries) {
            int x = q[0];
            int y = q[1];
            if (X[x]>0 || Y[y]>0 || DN[x-y]>0 || DP[x+y]>0) res.push_back(1); else res.push_back(0);
            for (int xx = x-1; xx<=x+1; ++xx) {
                for (int yy = y-1; yy<=y+1; ++yy) {
                    if (L.count({xx,yy})) {
                        X[xx]--;
                        Y[yy]--;
                        DN[xx-yy]--;
                        DP[xx+yy]++;
                        L.erase({xx,yy});
                    }
                }
            }
        }
        return res;
    }
};
