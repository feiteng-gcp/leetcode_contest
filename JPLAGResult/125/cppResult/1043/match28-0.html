class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        map<int, int> hor, ver, diag, rdiag;
        map<pair<int, int>, int> points;
        for (auto p : lamps) {
            int x = p[0], y = p[1];
            ++hor[x];
            ++ver[y];
            ++diag[x + y];
            ++rdiag[x - y];
            ++points[make_pair(x, y)];
        }
        
        vector<int> answer;
        for (auto p : queries) {
            int x = p[0], y = p[1];
            int ans = 0;
            if (hor[x] > 0 || ver[y] > 0 || diag[x + y] > 0 || rdiag[x - y] > 0)
                ans = 1;
            answer.push_back(ans);
            for (int i = -1; i <= 1; ++i)
                for (int j = -1; j <= 1; ++j) {
                    int nx = x + i, ny = y + j;
                    auto it = points.find(make_pair(nx, ny));
                    if (it != points.end() && it->second > 0) {
                        --it->second;
                        --hor[nx];
                        --ver[ny];
                        --diag[nx + ny];
                        --rdiag[nx - ny];
                    }
                }
        }
        return answer;
    }
};
