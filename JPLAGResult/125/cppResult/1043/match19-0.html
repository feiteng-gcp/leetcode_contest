class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        set<pair<int, int>> S;
        unordered_map<int, int> litcount1, litcount2, litcountdiag1, litcountdiag2;
        vector<int> ans;
        for (const vector<int>& pos: lamps) {
            ++litcount1[pos[0]];
            ++litcount2[pos[1]];
            ++litcountdiag1[pos[0] + pos[1]];
            ++litcountdiag2[pos[0] - pos[1] + N];
            S.insert(make_pair(pos[0], pos[1]));
        }
        for (const vector<int>& pos: queries) {
            ans.push_back(litcount1[pos[0]]
                          || litcount2[pos[1]]
                          || litcountdiag1[pos[0] + pos[1]]
                          || litcountdiag2[pos[0] - pos[1] + N]);
            for (int x = pos[0] - 1; x <= pos[0] + 1; ++x) {
                for (int y = pos[1] - 1; y <= pos[1] + 1; ++y) {
                    if (S.count(make_pair(x, y))) {
                        S.erase(make_pair(x, y));
                        --litcount1[x];
                        --litcount2[y];
                        --litcountdiag1[x + y];
                        --litcountdiag2[x - y + N];
                    }
                }
            }
        }
        return ans;
    }
};
