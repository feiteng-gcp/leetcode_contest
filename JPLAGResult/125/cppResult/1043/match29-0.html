class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int, int> r, c, p, q;
        set<pair<int, int>> nodes;
        TR(it, lamps) {
            int u = (*it)[0];
            int v = (*it)[1];
            r[u]++;
            c[v]++;
            p[u+v]++;
            q[u-v]++;
            
            nodes.insert(MP(u, v));
        }
        
        
        VI ans;
        TR(it, queries) {
            int u = (*it)[0];
            int v = (*it)[1];
            if (r[u] > 0 || c[v] > 0 || p[u+v] > 0 || q[u-v] > 0) {
                ans.PB(1);
            } else {
                ans.PB(0);
            }
            
            for (int dx = -1; dx <= 1; ++dx) {
                for (int dy = -1; dy <= 1; ++dy) {
                    int uu = u + dx;
                    int vv = v + dy;
                    
                    auto it = nodes.find(MP(uu, vv));
                    if (it != nodes.end()) {
                        --r[uu];
                        --c[vv];
                        --p[uu+vv];
                        --q[uu-vv];
                        nodes.erase(it);
                    }
                }
            }
        }
        return ans;
    }
};
