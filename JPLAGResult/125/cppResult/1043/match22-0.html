class Solution {
public:
    long long make_pairs(int x, int y) {
        return (long long) x * 1000000007 + y;
    }
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int, int> cnt_x, cnt_y, cnt_d0, cnt_d1;        
        unordered_map<long long, bool> exist;
        for (int i = 0; i < lamps.size(); ++i) {
            cnt_x[lamps[i][0]]++;
            cnt_y[lamps[i][1]]++;
            cnt_d0[lamps[i][0] - lamps[i][1]]++;
            cnt_d1[lamps[i][0] + lamps[i][1]]++;
            exist[make_pairs(lamps[i][0], lamps[i][1])] = true;
        }
        vector<int> result;
        for (int i = 0; i < queries.size(); ++i) {
            int ans = 0;
            if (cnt_x[queries[i][0]] || cnt_y[queries[i][1]] || cnt_d0[queries[i][0] - queries[i][1]] || cnt_d1[queries[i][0] + queries[i][1]])
                ans = 1;
            result.push_back(ans);
            for (int j = queries[i][0] - 1; j <= queries[i][0] + 1; ++j) {
                for (int k = queries[i][1] - 1; k <= queries[i][1] + 1; ++k) {
                    if (exist[make_pairs(j, k)]) {
                        exist[make_pairs(j, k)] = false;
                        cnt_x[j] --;
                        cnt_y[k] --;
                        cnt_d0[j - k] --;
                        cnt_d1[j + k] --;
                    }
                }
            }
        }
        return result;
    }
};
