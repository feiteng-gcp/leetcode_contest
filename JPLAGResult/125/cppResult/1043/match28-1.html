class Solution {
public:
    int dx[8]={1,1,1,0,-1,-1,-1,0};
    int dy[8]={1,0,-1,-1,-1,0,1,1};
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        map<int,int> row;
        map<int,int> col;
        map<int,int> plus;

        map<int,int> del;
        map<pair<int,int>,int >light;
        for(auto i:lamps){
            int x=i[0],y=i[1];
            plus[x+y]++;
            col[y]++;
            row[x]++;
            del[x-y]++;
            //代表这里有一盏灯
            light[make_pair(x,y)]++;
        }
        vector<int> ans;
        for(auto i:queries){
            int x=i[0],y=i[1];
            if(plus[x+y] ||row[x] ||col[y]||del[x-y]){
                ans.push_back(1);
            }
            else ans.push_back(0);
            for(int k=0;k<8;k++){
                int m=x+dx[k];
                int n=y+dy[k];
                if(m<0 ||n<0 ||m>=N ||n>=N) continue;

                if(light[make_pair(m,n)]){
                    //
                    light[make_pair(m,n)]--;
                    plus[m+n]--;
                    col[n]--;
                    row[m]--;
                    del[m-n]--;
                }
            }
        }
        return ans;
    }
};
