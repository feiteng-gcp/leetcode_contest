class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        vector< int > ans ; 
        map< int ,int > x ; 
        map< int , int > y ; 
        map< int ,int  > b1 ;
        map< int ,int > b2 ; 
        map< pair< int ,int > , int >  l ; 
        for( auto & lamp : lamps ){
            x[lamp[0]]++;
            y[lamp[1]]++;
            b1[lamp[0]+lamp[1]]++;
            b2[lamp[1]-lamp[0]]++;
            l[{lamp[0],lamp[1]}]++ ; 
        }
        
        
        for( auto & query : queries ){
            
            if( x[query[0]] || y[query[1]] || b1[query[0]+query[1]] || b2[query[1]-query[0]] ) 
                ans.push_back( 1 ) ; 
            else 
                ans.push_back ( 0 ) ; 
            
            
            
            for( int i=-1; i<=1;  i++){
                for( int j=-1;j<=1; j++){        
                    int nx = i+query[0] ; 
                    int ny = j+query[1] ; 
                    if( l[{nx,ny}] ){
                        x[nx]--;
                        y[ny]--;
                        b1[nx+ny]--;
                        b2[ny-nx]--;
                        l[{nx,ny}] --;
                    }
                }
            }

        }
        
        return ans ; 
    }
};
