struct pair_hash {
    inline std::size_t operator()(const std::pair<int,int> & v) const {
        return long(v.first)*31+v.second;
    }
};


class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int, unordered_set<pair<int, int>, pair_hash>> xaxis;
        unordered_map<int, unordered_set<pair<int, int>, pair_hash>> yaxis;
        unordered_map<int, unordered_set<pair<int, int>, pair_hash>> xminusy;
        unordered_map<int, unordered_set<pair<int, int>, pair_hash>> xplusy;
        vector<int> queryResult;
        for (const auto& lamp: lamps) {
            xaxis[lamp[0]].emplace(make_pair(lamp[0], lamp[1]));
            yaxis[lamp[1]].emplace(make_pair(lamp[0], lamp[1]));
            xminusy[lamp[0] - lamp[1]].emplace(make_pair(lamp[0], lamp[1]));
            xplusy[lamp[0] + lamp[1]].emplace(make_pair(lamp[0], lamp[1]));
        }
        

        for (const auto& query: queries) {
            queryResult.push_back(!xaxis[query[0]].empty() || !yaxis[query[1]].empty() || !xminusy[query[0] - query[1]].empty() || !xplusy[query[0] + query[1]].empty());
            
            for (int dx = -1; dx < 2; dx++) {
                for (int dy = -1; dy < 2; dy++) {
                    auto x = query[0] + dx;
                    auto y = query[1] + dy;
                    auto pair = make_pair(x, y);
                    xaxis[x].erase(pair);
                    yaxis[y].erase(pair);
                    xminusy[x - y].erase(pair);
                    xplusy[x + y].erase(pair);
                }
            }
        }
        return queryResult;
    }
};
