class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int, int> row, col, dia1, dia2;
        unordered_map<int, unordered_set<int>> hasLamp;
        for (auto& lamp : lamps) {
            ++row[lamp[0]];
            ++col[lamp[1]];
            ++dia1[lamp[0] + lamp[1]];
            ++dia2[lamp[0] - lamp[1] + N];
            hasLamp[lamp[0]].insert(lamp[1]);
        }
        vector<int> ans;
        for (auto q : queries) {
            int res = row.count(q[0]) || col.count(q[1]) || dia1.count(q[0] + q[1]) || dia2.count(q[0] - q[1] + N);
            if (res) {
                for (int x = q[0] - 1; x <= q[0] + 1; ++x) {
                    for (int y = q[1] - 1; y <= q[1] + 1; ++y) {
                        if (hasLamp.count(x) && hasLamp[x].count(y)) {
                            hasLamp[x].erase(y);
                            if (hasLamp[x].empty()) hasLamp.erase(x);
                            if (--row[x] == 0) row.erase(x);
                            if (--col[y] == 0) col.erase(y);
                            if (--dia1[x + y] == 0) dia1.erase(x + y);
                            if (--dia2[x - y + N] == 0) dia2.erase(x - y + N);
                        }
                    }
                }
            }
            ans.push_back(res);
        }
        return ans;
    }
};
