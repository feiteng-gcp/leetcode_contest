#include <unordered_set>

class Solution {
    int dy[8] = {0,-1,-1,-1,0,1,1,1};
    int dx[8] = {1,1,0,-1,-1,-1,0,1};
    

public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lampsInp, vector<vector<int>>& queries) {
        unordered_map<int,int> r;
        unordered_map<int,int> c;
        unordered_map<int,int> d1;
        unordered_map<int,int> d2;
        set<pair<int,int>> lamps;
        
        for(int i=0;i<lampsInp.size();i++){
            r[lampsInp[i][0]]++;
            c[lampsInp[i][1]]++;
            d1[lampsInp[i][1]-lampsInp[i][0]]++;
            d2[lampsInp[i][1]+lampsInp[i][0]]++;
            lamps.insert(make_pair(lampsInp[i][0],lampsInp[i][1]));
        }
        
        vector<int> result;

        for(int i=0;i<queries.size();i++){
            if(r[queries[i][0]]>0||c[queries[i][1]]>0||d1[queries[i][1]-queries[i][0]]>0||d2[queries[i][1]+queries[i][0]]>0) result.push_back(1);
            else result.push_back(0);
            
            for(int j=0;j<8;j++){
                int ny = queries[i][0] + dy[j];
                int nx = queries[i][1] + dx[j];
                
                auto it = lamps.find(make_pair(ny,nx));
                if(it!=lamps.end()) {
                    lamps.erase(it);
                    //cout<<i<<" "<<ny<<" "<<nx<<endl;
                    r[ny]--;
                    c[nx]--;
                    d1[nx-ny]--;
                    d2[nx+ny]--;
                }
            }
        }
        
        return result;
    }
};
