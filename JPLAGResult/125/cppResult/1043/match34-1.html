class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int,set<pair<int,int>>> a1,a2,a3,a4;
        for(auto p:lamps){

            auto pp=make_pair(p[0],p[1]);
            a1[p[0]].insert(pp);
            a2[p[1]].insert(pp);
            a3[p[0]+p[1]].insert(pp);
            a4[p[0]-p[1]].insert(pp);
        }
        
        vector<int> ans;
        for(auto q:queries){
            if(a1[q[0]].size()>0||a2[q[1]].size()>0||a3[q[0]+q[1]].size()>0||a4[q[0]-q[1]].size()>0){
                ans.push_back(1);
            }else ans.push_back(0);
            for(int i=-1;i<=1;i++)
                for(int j=-1;j<=1;j++){
                    if(i==0&&j==0) continue;
                    int x=q[0]+i;
                    int y=q[1]+j;
                    auto pp=make_pair(x,y);
                    a1[x].erase(pp);
                    a2[y].erase(pp);
                    a3[x+y].erase(pp);
                    a4[x-y].erase(pp);
                }
        }
        return ans;
    }
};
