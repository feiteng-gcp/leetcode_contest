class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        map <int, int> a, b, c, d;
        set < pair <int, int> > s;
        for (auto l: lamps) {
            a[l[0]]++;
            b[l[1]]++;
            c[l[0] + l[1]]++;
            d[l[0] - l[1]]++;
            s.insert(make_pair(l[0], l[1]));
        }
        vector <int> ans;
        for (auto q: queries) {
            int v = (a[q[0]] > 0 || b[q[1]] > 0 || c[q[0] + q[1]] > 0 || d[q[0] - q[1]] > 0);
            ans.push_back(v);

            for (int i = -1; i <= 1; i++) {
                for (int j = -1; j <= 1; j++) {
                    int x = q[0] + i, y = q[1] + j;

                    auto pp = s.find(make_pair(x, y));
                    if (pp != s.end()) {
                        s.erase(pp);
                        a[x]--;
                        b[y]--;
                        c[x + y]--;
                        d[x - y]--;
                    }
                }
            }
        }
        return ans;
    }
};
