class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int, int> rows;
        unordered_map<int, int> cols;
        unordered_map<int, int> a;
        unordered_map<int, int> b;
        set<pair<int, int>> s;
        for (auto &v : lamps) {
            rows[v[0]]++;
            cols[v[1]]++;
            a[v[0]-v[1]]++;
            b[v[0]+v[1]]++;
            s.insert({v[0], v[1]});
        }
        vector<int> ans;
        for (auto &v :queries) {
            int row = v[0];
            int col = v[1];
            if (rows[row] > 0 || cols[col] > 0 || a[row-col] > 0 || b[row+col] > 0) {
                ans.push_back(1);
            } else {
                ans.push_back(0);
            }
            int dx[3] = {-1, 0, 1};
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    int trow = row + dx[i];
                    int tcol = col + dx[j];

                    if (s.find({trow, tcol}) != s.end()) {
                        rows[trow]--;
                        cols[tcol]--;
                        a[trow-tcol]--;
                        b[trow+tcol]--;
                    }
                }
            }
        }
        return ans;
    }
};
