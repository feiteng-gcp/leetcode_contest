class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int, int> row;
        unordered_map<int, int> col;
        unordered_map<int, int> diag;
        unordered_map<int, int> anti;
        
        unordered_map<int, unordered_set<int>> lights;
        for(int i = 0; i < lamps.size(); i++)
        {
            row[lamps[i][0]]++;
            col[lamps[i][1]]++;
            diag[lamps[i][0] + lamps[i][1]]++;
            anti[lamps[i][0] - lamps[i][1]]++;
            lights[lamps[i][0]].insert(lamps[i][1]);
        }
        
        vector<vector<int>> dir = {{-1, -1}, {-1, 0}, {-1, 1}, {0, -1}, {0, 0}, {0, 1}, {1, -1}, {1, 0}, {1, 1}};
        vector<int> res;
        for(int i = 0; i < queries.size(); i++)
        {
            if(row[queries[i][0]] > 0 || col[queries[i][1]] > 0 || diag[queries[i][0] + queries[i][1]] > 0 || anti[queries[i][0] - queries[i][1]] > 0)
                res.push_back(1);
            else
                res.push_back(0);
            for(int j = 0; j < dir.size(); j++)
            {
                int x = queries[i][0] + dir[j][0];
                int y = queries[i][1] + dir[j][1];
                if(lights.find(x) != lights.end() && lights[x].find(y) != lights[x].end())
                {
                    lights[x].erase(y);
                    if(lights[x].empty())
                        lights.erase(x);
                    row[x]--;
                    col[y]--;
                    diag[x + y]--;
                    anti[x - y]--;
                }
            }
        }
        return res;
    }
};
