class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        map<int, set<pair<int,int>>> heng;
        map<int, set<pair<int,int>>> shu;
        map<int, set<pair<int,int>>> pie;
        map<int, set<pair<int,int>>> na;
        for(auto v:lamps){
            int x = v[0];
            int y = v[1];
            pair<int,int> p = make_pair(x,y);
            heng[x].insert(p);
            shu[y].insert(p);
            pie[x+y].insert(p);
            na[x+N-1-y].insert(p);
        }
        vector<int> ans;
        for(auto query:queries){
            int x = query[0];
            int y = query[1];
            if(heng[x].size()>0||shu[y].size()>0||pie[x+y].size()>0||na[x+N-1-y].size()>0){
                ans.push_back(1);
            }else{
                ans.push_back(0);
            }
            for(int i=-1;i<=1;i++)
            {
                for(int j=-1;j<=1;j++){
                    int nextx = x+i;
                    int nexty = y+i;
                    if(nextx>=0&&nextx<N&&nexty>=0&&nexty<N){
                        pair<int,int> nextp = make_pair(nextx,nexty);
                        heng[nextx].erase(nextp);
                        shu[nexty].erase(nextp);
                        pie[nextx+nexty].erase(nextp);
                        na[nextx+N-1-nexty].erase(nextp);
                    }
                }
            }
        }
        return ans;
    }
};
