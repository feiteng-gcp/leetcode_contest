class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_set<string> lampSet;
        
        unordered_map<int, int> rowCount;
        unordered_map<int, int> colCount;
        unordered_map<int, int> diag1Count;
        unordered_map<int, int> diag2Count;
        vector<int> result;
        
        for (auto& lamp : lamps) {
            lampSet.insert(to_string(lamp[0]) + "#" + to_string(lamp[1]));
            rowCount[lamp[0]]++;
            colCount[lamp[1]]++;
            diag1Count[lamp[0] + lamp[1]]++;
            diag2Count[lamp[0] - lamp[1]]++;
        }

        for (auto& query : queries) {
            int i = query[0];
            int j = query[1];
            if (rowCount.find(i) != rowCount.end() || colCount.find(j) != colCount.end() || diag1Count.find(i + j) != diag1Count.end() || diag2Count.find(i - j) != diag2Count.end()) {
                result.push_back(1);
            } else {
                result.push_back(0);
            }
            
            for (int x = i - 1; x <= i + 1; x++) {
                for (int y = j - 1; y <= j + 1; y++) {
                    
                    string sign = to_string(x) + "#" + to_string(y);
                    if (lampSet.find(sign) != lampSet.end()) {
                        lampSet.erase(sign);
                        rowCount[x]--;
                        colCount[y]--;
                        diag1Count[x + y]--;
                        diag2Count[x - y]--;
                        if (rowCount[x] == 0) {
                            rowCount.erase(x);
                        }
                        if (colCount[y] == 0) {
                            colCount.erase(y);
                        }
                        if (diag1Count[x + y] == 0) {
                            diag1Count.erase(x + y);
                        }
                        if (diag2Count[x - y] == 0) {
                            diag2Count.erase(x - y);
                        }
                    }
                }
            }
        }
        return result;
    }
};
