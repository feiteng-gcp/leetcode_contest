class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int, int> xs;
        unordered_map<int, int> ys;
        unordered_map<int, int> as;
        unordered_map<int, int> bs;
        set<pair<int, int>> ml;
        
        for (auto& l : lamps) {
            xs[l[0]]++;
            ys[l[1]]++;
            as[l[0]+l[1]]++;
            bs[l[0]-l[1]]++;
            ml.insert(make_pair(l[0], l[1]));
        }
        
        vector<int> output;
        
        for (auto& q : queries) {
            output.push_back(xs[q[0]] > 0 || ys[q[1]] > 0 || as[q[0]+q[1]] > 0 || bs[q[0]-q[1]] > 0);
            
            for (int i = -1; i <= 1; i++) {
                for (int j = -1; j <= 1; j++) {
                    if (ml.erase(make_pair(q[0]+i, q[1]+j)) != 0) {
                        xs[q[0]+i]--;
                        ys[q[1]+j]--;
                        as[q[0]+q[1]+i+j]--;
                        bs[q[0]-q[1]+i-j]--;
                    }
                }
            }
        }
        
        return output;
    }
};
