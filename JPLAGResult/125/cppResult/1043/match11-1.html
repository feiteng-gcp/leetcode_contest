class Solution {
public:
    vector<int> gridIllumination(int n, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        map<int, int> h, s, xz, xf;
        set<pair<int, int> > st;
        for (auto item : lamps) {
            h[item[0]]++;
            s[item[1]]++;
            xz[item[0] + item[1]]++;
            xf[item[0] - item[1]]++;
            st.insert(make_pair(item[0], item[1]));
        }
        vector<int> res;
        int dx[] = {-1, -1, -1, 0, 0, 0, 1, 1, 1};
        int dy[] = {-1, 0, 1, -1, 0, 1, -1, 0, 1};
        for (auto query : queries) {
            if (h[query[0]] || s[query[1]] || xz[query[0] + query[1]] || xf[query[0] - query[1]]) {
                res.push_back(1);
            } else {
                res.push_back(0);
            }
            for (int dir = 0; dir < 9; dir++) {

                int x = query[0] + dx[dir], y = query[1] + dy[dir];
                if (st.find(make_pair(x, y)) != st.end()) {
                    h[x]--;
                    s[y]--;
                    xz[x + y]--;
                    xf[x - y]--;
                    st.erase(make_pair(x, y));
                }
            }
        }
        return res;
    }
};
