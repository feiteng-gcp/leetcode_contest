class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        multiset<int> x, y, xy1, xy2; 
        set<pair<int,int> > la;
        vector<int> ans;
        for(int i = 0; i < lamps.size(); i++) {
            int curx = lamps[i][0];
            int cury = lamps[i][1];
            la.insert({lamps[i][0],lamps[i][1]}); 
            x.insert(curx);
            y.insert(cury);
            xy1.insert(curx+cury);
            xy2.insert(curx-cury);
        }
        int dx[9] = {-1,-1,-1,0,0,0,1,1,1};
        int dy[9] = {-1,0,1,-1,0,1,-1,0,1};
        for(int i = 0; i < queries.size(); i++) {
            int curx = queries[i][0];
            int cury = queries[i][1];
            if(x.find(curx) != x.end() || y.find(cury) != y.end() || xy1.find(curx+cury) != xy1.end()
              || xy2.find(curx-cury) != xy2.end()) 
                ans.push_back(1);
            else ans.push_back(0);
            for(int j = 0; j < 9; j++) {
                int rx = curx + dx[j];
                int ry = cury + dy[j];
                if(rx < N && rx >= 0 && ry < N && ry >= 0) {
                    if(la.find({rx,ry}) != la.end()) {
                        la.erase(la.find({rx,ry}));
                        x.erase(x.find(rx));
                        y.erase(y.find(ry));
                        xy1.erase(xy1.find(rx+ry));
                        xy2.erase(xy2.find(rx-ry));
                    }
                }
            }
        }
        return ans;
    }
};
