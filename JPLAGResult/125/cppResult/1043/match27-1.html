class Solution {
public:
    vector<int> gridIllumination(int N, vector<vector<int>>& lamps, vector<vector<int>>& queries) {
        unordered_map<int,int>row,col,diagRight,diagLeft;
        multiset<pair<int,int> >S;
        
        
        for(int i=0;i<lamps.size();i++){
            row[lamps[i][0]]++;
            col[lamps[i][1]]++;
            diagRight[lamps[i][1]-lamps[i][0]]++;
            diagLeft[lamps[i][1]+lamps[i][0]]++;
            S.insert(make_pair(lamps[i][0],lamps[i][1]));
            
        }
        
        vector<int>ans;
        
        for(int i=0;i<queries.size();i++){
            if(row[queries[i][0]]>0 || 
               col[queries[i][1]]>0 || 
               diagRight[queries[i][1]-queries[i][0]]>0 || 
               diagLeft[queries[i][0]+queries[i][1]]>0)
                ans.push_back(1);            
            else
                ans.push_back(0);
            
            for(int ii=queries[i][0]-1;ii<=queries[i][0]+1;ii++)
                for(int jj=queries[i][1]-1;jj<=queries[i][1]+1;jj++){
                    // turn off
                    if( S.find(make_pair(ii,jj)) !=S.end() ){
                        if(row[ii]>0)row[ii]--;
                        if(col[jj]>0)col[jj]--;
                        if(diagRight[jj-ii]>0)diagRight[jj-ii]--;
                        if(diagLeft[ii+jj]>0)diagLeft[ii+jj]--;
                    }
                }
            
            
        }
        
        return ans;
    }
};
