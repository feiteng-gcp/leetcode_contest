class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        indeg, outdeg = [0] * N, [0] * N
        for a, b in trust:
            indeg[b - 1] += 1
            outdeg[a - 1] += 1
        R = []
        for i in range(N):
            if indeg[i] == N - 1 and outdeg[i] == 0:
                R.append(i)
        if len(R) == 1:
            return R[0] + 1
        return -1
