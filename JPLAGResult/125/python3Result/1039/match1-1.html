class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        trusts = [False for _ in range(N + 1)]
        trusted = [0 for _ in range(N + 1)]
        
for a, b in trust:
            trusts[a] = True
            trusted[b] += 1
        for i in range(1, N + 1):
            if not trusts[i] and trusted[i] == N - 1:
                return i
        return -1
