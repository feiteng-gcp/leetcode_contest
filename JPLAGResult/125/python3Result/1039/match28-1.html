from collections import defaultdict
class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        st = set()
        mp = defaultdict(set)
        for ls in trust:
            mp[ls[1]].add(ls
[0])
            st.add(ls[0])
        for i in range(1, N+1):
            if len(mp[i]) == N-1 and i not in st:
                return i
        return -1
