class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        ind = [0] * (N+1)
        outd = [0]*(N+1)
        for a,b in trust:
            ind[b]+=1
            outd[a]+=1
        for i in range(1,N+1):
            if ind[i] == N-1 and outd[i] == 0:
                return i
        return -1
