class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        trusting = collections.defaultdict(int)
        trusted = collections.defaultdict(int)
        for i, j in trust:
            trusting[i] += 1
            trusted[j] += 1
        for i in range(1, N+1):
            if trusted[i] == N-1 and trusting[i] == 0:
                return i
        return -1
