from collections import defaultdict

class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        record = defaultdict(set)
        has_trust = set()
        for t in trust:
            has_trust.add(t[0])
            record[t[1]].add(t[0])
        
        for r in range(1, N + 1):
            if len(record[r]) == N - 1 and r not in has_trust:
                return r
        
        return -1
