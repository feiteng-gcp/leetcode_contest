class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        l1 = [0 for _ in range(N+1)]
        l2 = [0 for _ in range(N+1)]
        for li in trust:
            l1[li[1]] += 1
            l2[li[0]] += 1
        for i in range(1, N+1):
            if l1[i] == N-1:
                if l2[i] == 0:
                    return i
        return -1
