class Solution:
    def findJudge(self, N: 'int', trust: 'List[List[int]]') -> 'int':
        outdegree = [0]*(N+1)
        indegree 
= [0]*(N+1)
        for x, y in trust:
            outdegree[x] += 1
            indegree[y] += 1
        ans= []
        for i in range(1, N+1):
            if outdegree[i] == 0 and indegree[i] == N-1:
                ans.append(i)
        if len(ans) > 1 or not ans:
            return -1
        else:
            return ans[0]
