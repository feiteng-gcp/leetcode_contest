class Solution:
    def findJudge(self, N: int, trust) -> int:
        t 
= [0 for i in range(N)]
        f = [0 for i in range(N)]

        for a in trust:
            # print(a)
            t[a[1] - 1] += 1
            f[a[0] - 1] += 1


        for i, a in enumerate(t):
            if a == N - 1:
                if f[i] == 0:
                    return i + 1

        return -1
