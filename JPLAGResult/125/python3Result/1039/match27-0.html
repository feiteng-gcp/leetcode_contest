class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        num_trusts = [0] * N
        for pair in trust:
            num_trusts[pair[1] - 1] += 1
            num_trusts[pair[0] - 1] -= 1
            
        for idx, val in enumerate(num_trusts):
            if val == N - 1:
                return idx + 1
        return -1
