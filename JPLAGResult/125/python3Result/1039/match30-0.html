class Solution:
    def findJudge(self, N: int, trust) -> int:
        to=[set() for _ in range(N+1)]
        frm=[set() for _ in range(N+1)]
        for l in trust:
            to[l[1]].add(l[0])
            frm[l[0]].add(l[1])
        for k in range(1, N+1):
            if len(to[k])==N-1 and len(frm[k])==0:
                return k
        return -1
