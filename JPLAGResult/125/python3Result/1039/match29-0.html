class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        trusts = [0] * N
        trusted = [set() for i in range(N)]
        for t in trust:
            fr = t[0]-1
            to = t[1]-1
            trusts[fr] += 1
            trusted[to].add(fr)
        for i in range(N):
            #print(i, trusts[i], len(trusted[i]))
            if not trusts[i] and len(trusted[i]) == N - 1: return i + 1
        return -1
