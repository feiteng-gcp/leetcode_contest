class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        st, sted = [0] * (N + 1), [0] * (N + 1)
        for (a, b) in trust:
            st[a] += 1
            sted[b] += 1
        for i in range(1, N + 1):
            if st[i] == 0 and sted[i] == N - 1:
                return i
        return -1
