class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        trusting = [0] * N
        trusted = [0] * N
        for a, b in trust:
            trusting[a-1] += 1
            trusted[b-1] += 1
        for i in range(N):
            if trusting[i] == 0 and trusted[i] == N-1:
                return i+1
        return -1
