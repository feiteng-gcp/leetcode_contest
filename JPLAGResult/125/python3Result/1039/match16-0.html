class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        trusted_by = [0] * N
        trusts = [0] * N
        for a, b in trust:
            trusted_by[b - 1] += 1
            trusts[a - 1] += 1

        judges = [i for i in range(N) if trusts[i] == 0 and trusted_by[i] == N - 1]
        if len(judges) != 1:
            return -1
        return judges[0] + 1
