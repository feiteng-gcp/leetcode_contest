class Solution:
    def findJudge(self, N: 'int', trust: 'List[List[int]]') -> 'int':
        board1 = [[] for _ in range(N+1)]
        board2 = [[] for _ in range
(N+1)]
        for t in trust:
            board1[t[0]].append(t[1])
            board2[t[1]].append(t[0])
        # print(board1)
        # print(board2)
        pot = -1
        for i in range(1, 1+N):
            if len(board2[i]) == N-1 and len(board1[i]) == 0:
                if pot == -1:
                    pot = i
                else:
                    return -1
        return pot
