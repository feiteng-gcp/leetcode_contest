class Solution:
    def findJudge(self, N: int, trust: List[List[int]]) -> int:
        outEdge = [0] * (N+1)
        inEdge = [0] * (N+1)
        for t in trust:
            outEdge[t[0]] += 1
            inEdge[t[1]] += 1
        for i in range(1, N+1):
            if outEdge[i] == 0 and inEdge[i] == N-1:
                return i
        return -1
