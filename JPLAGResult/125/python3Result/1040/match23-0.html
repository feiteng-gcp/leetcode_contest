# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def insertIntoMaxTree(self, root: TreeNode, val: int) -> TreeNode:
        def dfs(node):
            if node == None:
                return TreeNode(val)
            if val > node.val:
                res = TreeNode(val)
                res.left = node
                return res
            node.right = dfs(node.right)
            return node
        res = dfs(root)
        return res
