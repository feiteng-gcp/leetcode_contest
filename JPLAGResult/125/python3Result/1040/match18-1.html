# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    
def insertIntoMaxTree(self, root: TreeNode, val: int) -> TreeNode:
        if not root:
            return TreeNode(val)
        if root.val < val:
            valnode = TreeNode(val)
            valnode.left = root
            return valnode
        root.right = self.insertIntoMaxTree(root.right, val)
        return root
