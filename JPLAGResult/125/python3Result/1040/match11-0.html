# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def insertIntoMaxTree(self, root: TreeNode, val: int) -> TreeNode:
        def add(node, v):
            if not node:
                return TreeNode(v)
            if node.val < v:
                new_node = TreeNode(v)
                new_node.left = node
                return new_node
            node.right = add(node.right, v)
            return node
        return add(root, val)
