# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def insertIntoMaxTree(self, root: TreeNode, val: int) -> TreeNode:
        def rebuildA(root):
            if not root:
                return []
            return rebuildA(root.left) + [root.val] + rebuildA(root.right)
        
        a = rebuildA(root)
        a.append(val)
        
        def construct(arr):
            if not arr:
                return None
            
            max_val = max(arr)
            max_idx = arr.index(max_val)
            
            root = TreeNode(max_val)
            root.left = construct(arr[:max_idx])
            root.right = construct(arr[max_idx+1:])
            return root
        
        return construct(a)
