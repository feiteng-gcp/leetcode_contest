# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def insertIntoMaxTree(self, root: TreeNode, val: int) -> TreeNode:
        
        
        
        def insert(n):
            if n == None or val > n.val:
                new = TreeNode(val)
                new.left = n
                return new
            n.right = insert(n.right)
            return n
        
        return insert(root)
