# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def insertIntoMaxTree(self, root: TreeNode, val: int) -> TreeNode:
        def remake_list(root):
            if not root:
                return []
            return remake_list(root.left) + [root.val] + remake_list(root.right)
        
        def make_tree(lst):
            if not lst:
                return None
            m, i = max([(
val, i) for i, val in enumerate(lst)])
            r = TreeNode(m)
            r.left = make_tree(lst[:i])
            r.right = make_tree(lst[i+1:])
            return r
        
        # tree -> list that constructed the tree -> append val -> create tree
        lst = remake_list(root) + [val]
        return make_tree(lst)
