from collections import defaultdict
class Solution:
    def check(self, i, j, N, rows, cols, diag, r_diag):
        res = 0
        if 0 <= i < N and 0 <= j < N:
            if i in rows or j in cols or i - j in diag or i + j in r_diag:
                res = 1
                
        for di in [-1, 0, 1]:
            for dj in [-1, 0, 1]:
                x, y = i + di, j + dj
                if 0 <= x < N and 0 <= y < N:
                    rows[x].discard(y)
                    cols[y].discard(x)
                    diag[x - y].discard((x, y))
                    r_diag[x + y].discard((x, y))
                    
if not rows[x]:
                        del rows[x]
                    if not cols[y]:
                        del cols[y]
                    if not diag[x - y]:
                        del diag[x - y]
                    if not r_diag[x + y]:
                        del r_diag[x + y]
        
        return res
        
    
def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        rows = defaultdict(set) 
        cols = defaultdict(set)
        diag = defaultdict(set)
        r_diag = defaultdict(set)
        for x, y in lamps:
            rows[x].add(y)
            cols[y].add(x)
            diag[x - y].add((x, y))
            r_diag[x + y].add((x, y))
        res = []
        
        for i, j in queries:
            is_on = self.check(i, j, N, rows, cols, diag, r_diag)
            res.append(is_on)
        
        return res
