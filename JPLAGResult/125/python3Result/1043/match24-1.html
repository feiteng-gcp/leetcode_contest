class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        lamp_set = {tuple(loc) for loc in lamps}
        counter_row = collections.Counter(item[0] for item in lamps)
        counter_col = collections.Counter(item[1] for item in lamps)
        counter_diag = collections.Counter(item[0]-item[1] for item in lamps)
        counter_rdiag = collections.Counter(item[0]+item[1] for item in lamps)
        
        res = []
        for x, y in queries:
            # print(counter_row, counter_col, counter_diag, counter_rdiag, lamp_set)
            if counter_row[x] > 0 or counter_col[y] > 0 or counter_diag[x-y] > 0 or counter_rdiag[x+y] > 0:
                res.append(1)
            else:
                res.append(0)
            for diff_x in range(-1, 2):
                tmp_x = diff_x + x
                for diff_y in range(-1, 2):
                    tmp_y 
= diff_y + y
                    if (tmp_x, tmp_y) in lamp_set:
                        lamp_set.remove((tmp_x, tmp_y))
                        counter_row[tmp_x] -= 1
                        counter_col[tmp_y] -= 1
                        counter_diag[tmp_x - tmp_y] -= 1
                        counter_rdiag[tmp_x + tmp_y] -= 1
        return res
