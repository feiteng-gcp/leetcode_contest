from collections import defaultdict
class Solution:
    def gridIllumination(self, N: int, lamps, queries):
        cntx=defaultdict(int)
        cnty=defaultdict(int)
        cntd1=defaultdict(int)
        cntd2=defaultdict(int)
        lampset=set()
        for x, y in lamps:
            lampset.add((x,y))
        lamps=lampset
        for x, y in lamps:
            cntx[x]+=1
            cnty[y]+=1
            cntd1[x+y]+=1
            cntd2[x-y]+=1
        ans=[]
        for x, y in queries:
            if cntx[x]>0 or cnty[y]>0 or cntd1[x+y]>0 or cntd2[x-y]>0:
                ans.append(1)
            else:
                ans.append(0)
            for nx in [x-1,x,x+1]:
                for ny in [y-1,y,y+1]:
                    
if nx>=0 and nx<N and ny>=0 and ny<N and (nx,ny) in lamps:
                        lamps.remove((nx,ny))
                        cntx[nx]-=1
                        cnty[ny]-=1
                        cntd1[nx+ny]-=1
                        cntd1[nx-ny]-=1
        return ans
