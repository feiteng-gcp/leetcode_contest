class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        cols = collections.defaultdict(int)
        rows = collections.defaultdict(int)
        diags = collections.defaultdict(int)
        off_diags = collections.defaultdict(int)
        
        rec = set()
        for y, x in lamps:
            rec.add((y,x))
            rows[y] += 1
            cols[x] += 1
            diags[y-x] += 1
            off_diags[y+x] += 1
        #print(rows, cols, diags, off_diags)
        
        res = [0]*len(queries)
        for i, pair in enumerate(queries):
            y, x = pair
            if rows[y] > 0 or cols[x] > 0 or diags[y-x] > 0 or off_diags[y+x] > 0:
                res[i] = 1
                
            dirs = {(y,x),(y-1,x),(y+1,x),(y,x+1),(y-1,x+1),(y+1,x+1),(y,x-1),(y-1,x-1),(y+1,x-1)}
            for tmp_y, tmp_x in dirs:
                if (tmp_y, tmp_x) in rec:
                    rec.remove((tmp_y, tmp_x))
                    if 0 <= tmp_y < N and rows[tmp_y] > 0:
                        rows[tmp_y] -= 1
                    if 0 <= tmp_x < N and cols[tmp_x] > 0:
                        cols[tmp_x] -= 1
                    if 0 <= tmp_y < N and 0 <= tmp_x < N and diags[tmp_y-tmp_x] > 0:
                        diags[tmp_y-tmp_x] -= 1
                    if 0 <= tmp_y < N and 0 <= tmp_x < N and off_diags[tmp_y+tmp_x] > 0:
                        off_diags[tmp_y+tmp_x] -= 1
            #print(rows, cols, diags, off_diags)
        return res
