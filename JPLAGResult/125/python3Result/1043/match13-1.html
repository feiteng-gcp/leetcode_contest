from collections import defaultdict
class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        result = []
        x = defaultdict(int)
        y = defaultdict(int)
        xpy = defaultdict(int)
        xmy = defaultdict(int)
        lamps_set = set()
        for [i,j] in lamps:
            x[i] += 1
            y[j] += 1
            xpy[i+j] += 1
            xmy[i-j] += 1
            lamps_set.add((i,j))
        for [i,j] in queries:
            if x[i]>0 or y[j]>0 or xpy[i+j]>0 or xmy[i-j]>0:
                result.append(1)
            else:
                result.append(0)
            for ii in (-1,2):
                for jj in (-1,2):
                    ci = i+ii
                    cj = j+jj
                    
if (ci,cj) in lamps_set:
                        x[ci] -= 1
                        y[cj] -= 1
                        xpy[ci+cj] -= 1
                        xmy[ci-cj] -= 1
                        lamps_set.remove((ci,cj))
        return result
