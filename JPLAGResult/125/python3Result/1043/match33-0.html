from collections import defaultdict
class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        rows = defaultdict(set)
        cols = defaultdict(set)
        first_diags = defaultdict(set)
        second_diags = defaultdict(set)
        
        on_lamps = set()
        
        def first_diag_key(r, c):
            ans_c = 0
            ans_r = r - c
            diff = min(0, ans_r)
            ans_r -= diff
            ans_c -= diff
            return (ans_r, ans_c)
        
        def second_diag_key(r, c):
            ans_c = 0
            ans_r = r + c
            diff = max(0, ans_r - N + 1)
            ans_c += diff
            ans_r -= diff
            return (ans_r, ans_c)
            
        for idx, l in enumerate(lamps):
            r, c = l
            v = (r, c)
            on_lamps.add(v)
            rows[r].add(v)
            cols[c].add(v)
            first_diags[first_diag_key(r, c)].add(v)
            second_diags[second_diag_key(r, c)].add(v)
        
        ans = []
        for q in queries:
            r, c = q
            rs = rows[r]
            cs = cols[c]
            fds = first_diags[first_diag_key(r, c)]
            sds = second_diags[second_diag_key(r, c)]
            
            #print("On lambs:", on_lamps)
            cans = [rs, cs, fds, sds]
            #print("Cans:", cans)
            ons = [len(on_lamps.intersection(x)) for x in cans]
            #print("Ons:", ons)
            
            ans.append(1 if sum(ons) > 0 else 0)
            
            for r_diff in range(-1, 2):
                for c_diff in range(-1, 2):
                    row = r + r_diff
                    col = c + c_diff
                    if (row, col) in on_lamps:
                        on_lamps.remove((row, col))
        
        return ans
