from itertools import product
from collections import defaultdict as dd
class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        rows = dd(int)
        cols = dd(int)
        downdiag = dd(int)
        updiag = dd(int)
        positions = {}
        for x,y in lamps:
            downd = x - y
            upd = x + y
            positions[(x,y)] = [downd, upd, y, x]
            downdiag[downd] += 1
            updiag[upd] += 1
            rows[y] += 1
            cols[x] += 1
        
        answers = []
        for x,y in queries:
            if y in rows or x in cols or x + y in updiag or x - y in downdiag:
                answers.append(1)
            else:
                answers.append(0)
            for dx,dy in product([-1,0,1],repeat=2):
                nx,ny = x + dx, y + dy
                try:
                    downd, upd, y, x = positions[(nx,ny)]
                    del positions[(nx,ny)]
                except KeyError:
                    continue
                downdiag[downd] -= 1
                updiag[upd] -= 1
                rows[y] -= 1
                cols[x] -= 1
                if downdiag[downd] == 0:
                    del downdiag[downd]
                if updiag[upd] == 0:
                    del updiag[upd]
                if rows[y] == 0:
                    del rows[y]
                if cols[x] == 0:
                    del cols[x]
        return answers
