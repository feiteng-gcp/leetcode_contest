from collections import defaultdict

class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        
        row = defaultdict(set)
        col = defaultdict(set)
        slash = defaultdict(set)
        bslash = defaultdict
(set)
        
        for i, j in lamps:
            row[i].add((i, j))
            col[j].add((i, j))
            bslash[j-i].add((i, j))
            slash[i+j].add((i, j))
            
        ans = []
        for i, j in queries:
            if len(row[i]) > 0 or len(col[j]) > 0 or len(bslash[j-i]) > 0 or len(slash[i+j]) > 0:
                ans.append(1)
            else:
                ans.append(0)
            
            for x in range(i-1, i+2):
                for y in range(j-1, j+2):
                    if 0 <= x < N and 0 <= y < N:
                        if (x, y) in row[x]:
                            row[x].remove((x, y))
                        if (x, y) in col[y]:
                            col[y].remove((x, y))
                        if (x, y) in bslash[y-x]:
                            bslash[y-x].remove((x, y))
                        if (x, y) in slash[x+y]:
                            slash[x+y].remove((x, y))
        
        return ans
