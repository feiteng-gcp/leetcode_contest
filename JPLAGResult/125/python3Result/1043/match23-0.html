class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        lampson = set()
        ln = len(lamps)
        ron = collections.defaultdict(int)
        con = collections.defaultdict(int)
        d1on = collections.defaultdict(int)
        d2on = collections.defaultdict(int)
        for i, (r, c) in enumerate(lamps):
            lampson.add((r, c))
            ron[r] += 1
            con[c] += 1
            d1on[N - 1 - r + c] += 1
            d2on[r + c] += 1
        res = []
        for r, c in queries:
            res.append(int(ron[r] > 0 or con[c] > 0 or d1on[N - 1 - r + c] > 0 or d2on[r + c] > 0))
            for dr in range(-1, 2, 1):
                for dc in range(-1, 2, 1):
                    rr = r + dr
                    cc = c + dc
                    if (rr, cc) in lampson:
                        lampson.remove((rr, cc))
                        ron[rr] -= 1
                        con[cc] -= 1
                        d1on[N - 1 - rr + cc] -= 1
                        d2on[rr + cc] -= 1
        return res
