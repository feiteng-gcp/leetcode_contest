from collections import defaultdict
class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        row_light = defaultdict(set)
        col_light = defaultdict(set)
        left_light = defaultdict(set)
        right_light = defaultdict(set)
        
        
        
        for lamp in lamps:
            row_light[lamp[0]].add(tuple(lamp))
            col_light[lamp[1]].add(tuple(lamp))
            left_light[lamp[0] - lamp[1]].add(tuple(lamp))
            right_light[lamp[1] + lamp[0]].add(tuple(lamp))
            
        answer = []
        
        for query in queries:
            row, col = query
            if len(row_light[row]) > 0 or len(col_light[col]) > 0 or len(left_light[row - col]) > 0 or len(right_light[row + col]) > 0:
                answer.append(1)
            else:
                answer.append(0)
            for i in range(-1, 2):
                for j in range(-1, 2):
                    nrow = row + i
                    ncol = col + j
                    if 0 <= nrow < N and 0<= ncol < N:
                        if (nrow, ncol) in row_light[nrow]:
                            row_light[nrow].remove((nrow, ncol))
                        if (nrow, ncol) in col_light[ncol]:
                            col_light[ncol].remove((nrow, ncol))
                        if (nrow, ncol) in left_light[nrow - ncol]:
                            left_light[nrow - ncol].remove((nrow, ncol))
                        if (nrow, ncol) in right_light[nrow + ncol]:
                            right_light[nrow + ncol].remove((nrow, ncol))
                            
                            
        return answer
