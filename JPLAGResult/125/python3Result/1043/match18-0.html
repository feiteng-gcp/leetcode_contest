class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        rows = collections.Counter()
        cols = collections.Counter()
        lefts = collections.Counter()
        rights = collections.Counter()
        lampsset = set()
        for l in lamps:
            lampsset.add((l[0],l[1]))
            cols[l[0]] += 1
            rows[l[1]] += 1
            lefts[l[0]+N-1-l[1]] += 1
            rights[l[0]+l[1]] += 1
        result = [0] * len(queries)
        for i,q in enumerate(queries):
            if cols[q[0]] >= 1: result[i] = 1
            elif rows[q[1]] >= 1: result[i] = 1
            elif lefts[q[0]+N-1-q[1]] >= 1: result[i] = 1
            elif rights[q[0]+q[1]] >= 1: result[i] = 1
            for r in range(q[1]-1,q[1]+2):
                for c in range(q[0]-1,q[0]+2):
                    if r >= 0 and r < N and c >= 0 and c < N and (c,r) in lampsset:
                        cols[c] -= 1
                        rows[r] -= 1
                        lefts[c+N-1-r] -= 1
                        rights[c+r] -= 1
                        lampsset.remove((c,r))
        return result
