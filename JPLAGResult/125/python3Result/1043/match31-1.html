class Solution:
    def gridIllumination(self, N: 'int', lamps: 'List[List[int]]', queries: 'List[List[int]]') -> 'List[int]':
        dic1 = {}
        dic2 = {}
        dic3 = {}
        dic4 = {}
        diclamps = set()
        def l2str(l):
            return str(l[0]) + '#' + str(l[1])
        for lamp in lamps:
            diclamps.add(str(lamp[0]) + '#' + str(lamp[1]))
            if lamp[0] in dic1.keys():
                dic1[lamp[0]] += 1
            else:
                dic1[lamp[0]] = 1
            if lamp[1] in dic2.keys():
                dic2[lamp[1]] += 1
            else:
                dic2[lamp[1]] = 1
            if lamp[0] + lamp[1] in dic3.keys():
                dic3[lamp[0] + lamp[1]] += 1
            else:
                dic3[lamp[0] + lamp[1]] = 1
            if lamp[0] + N - 1 - lamp[1] in dic4.keys():
                dic4[lamp[0] + N - 1 - lamp[1]] += 1
            else:
                dic4[lamp[0] + N - 1 - lamp[1]] = 1
        res = []
        for que in queries:
            if que[0] in dic1.keys() or que
[1] in dic2.keys() or que[0]+que[1] in dic3.keys() or que[0]+N-1-que[1] in dic4.keys():
                res.append(1)
            else:
                res.append(0)
            pot = [que[0]-1, que[1]-1]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2
[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
            pot = [que[0]-1, que[1]]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2
[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
            pot = [que[0]-1, que[1]+1]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
            pot = [que[0], que[1]-1]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
            pot = [que[0], que[1]+1]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
            pot = [que[0]+1, que[1]-1]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
            pot = [que[0]+1, que[1]]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
            pot = [que[0]+1, que[1]+1]
            if l2str(pot) in diclamps:
                diclamps.remove(l2str(pot))
                dic1[pot[0]] -= 1
                dic2[pot[1]] -= 1
                dic3[pot[0]+pot[1]] -= 1
                dic4[pot[0]+N-1-pot[1]] -= 1
                if dic1[pot[0]] <= 0:
                    dic1.pop(pot[0])
                if dic2[pot[1]] <= 0:
                    dic2.pop(pot[1])
                if dic3[pot[0]+pot[1]] <= 0:
                    dic3.pop(pot[0]+pot[1])
                if dic4[pot[0]+N-1-pot[1]] <= 0:
                    dic4.pop(pot[0]+N-1-pot[1])
        return res
