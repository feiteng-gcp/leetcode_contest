class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        
        rows = collections.defaultdict(int)
        cols = collections.defaultdict(int)
        diag = collections.defaultdict(int)
        adiag = collections.defaultdict(int)
        
  
        ons = set()
        
        def turn_on(i, j):
            rows[i] += 1
            cols[j] += 1
            diag[i+j] += 1
            adiag[i-j] += 1
            ons.add((i, j))
        
        def turn_off(i, j):
            rows[i] -= 1
            cols[j] -= 1
            diag[i+j] -= 1
            adiag[i-j] -= 1
            ons.remove((i, j))
        
        for i, j in lamps:
            turn_on(i, j)
        
        ans = []
        for i, j in queries:
            if rows[i] or cols[j] or diag[i+j] or adiag[i-j]:
                ans.append(1)
            else:
                ans.append(0)
            
            for ni in range(i-1, i+2):
                for nj in range(j-1, j+2):
                    if (ni, nj) in ons:
                        turn_off(ni, nj)
        
        return ans
