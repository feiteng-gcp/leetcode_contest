class Solution:
    def gridIllumination(self, N: int, lamps: List[List[int]], queries: List[List[int]]) -> List[int]:
        
        ans = []
        
        row_lamp 
= collections.defaultdict(set)
        col_lamp = collections.defaultdict(set)
        diag_lamp = collections.defaultdict(set)
        anti_diag_lamp = collections.defaultdict(set)
        
        for x, y in lamps:
            row_lamp[x].add((x,y))
            col_lamp[y].add((x,y))
            diag_lamp[x-y].add((x,y))
            anti_diag_lamp[x+y].add((x,y))
            
        #print(row_lamp, col_lamp, diag_lamp, anti_diag_lamp)
            
        
for x, y in queries:
            if row_lamp[x] or col_lamp[y] or diag_lamp[x-y] or anti_diag_lamp[x+y]:
                ans.append(1)
            else:
                ans.append(0)
            
            for dx in [-1, 0, 1]:
                for dy in [-1, 0, -1]:
                    nx, ny = x+dx, y+dy
                    if 0 <= nx < N and 0 <= ny < N:
                        row_lamp[nx].discard((nx, ny))
                        col_lamp[ny].discard((nx, ny))
                        diag_lamp[nx-ny].discard((nx, ny))
                        anti_diag_lamp[nx+ny].discard((nx, ny))
            #print(row_lamp, col_lamp, diag_lamp, anti_diag_lamp)
        return ans
