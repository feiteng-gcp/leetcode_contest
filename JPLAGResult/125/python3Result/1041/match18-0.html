class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        sr, sc = None, None
        for i in range(8):
            for j in range(8):
                if board[i][j] == 'R':
                    sr, sc = i, j
                    break
            if sr is not None:
                break
        num = 0
        for dr, dc in zip([0, 1, 0, -1], [1, 0, -1, 0]):
            tr, tc = sr + dr, sc + dc
            while 0 <= tr < 8 and 0 <= tc < 8:
                if board[tr][tc] == 'p':
                    num += 1
                    break
                elif board[tr][tc] != '.':
                    break
                tr += dr
                tc += dc
        return num
