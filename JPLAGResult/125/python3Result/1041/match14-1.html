class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        for i in range(8):
            for j in range(8):
                if board[i][j] == 'R':
                    x0, y0 = i, j
                    break
        ans = 0
        def isBounded(x, y):
            return 0<=x<=7 and 0<=y<=7
        def move(dx, dy):
            nonlocal ans
            x, y = x0, y0
            while isBounded(x,y) and (board[x][y]== 'R' or board[x][y] == '.'):
                x += dx
                y += dy
            if isBounded(x,y) and board[x][y] == 'p':
                ans += 1
        move(-1, 0)
        move(0, -1)
        move(1, 0)
        move(0, 1)
        return ans
