class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        ans=0
        for y in range(8):
            for x in range(8):
                if board[y][x]=='R':
                    for nx in range(x+1,8):
                        if board[y][nx]==".": continue
                        if board[y][nx]=="B": break
                        if board[y][nx]=="p":
                            ans+=1
                            break
                    for nx in range(x-1,-1,-1):
                        if board[y][nx]==".": continue
                        if board[y][nx]=="B": break
                        if board[y][nx]=="p":
                            ans+=1
                            break
                    for ny in range(y+1,8):
                        if board[ny][x]==".": continue
                        if board[ny][x]=="B": break
                        if board[ny][x]=="p":
                            ans+=1
                            break
                    for ny in range(y-1,-1,-1):
                        if board[ny][x]==".": continue
                        if board[ny][x]=="B": break
                        if board[ny][x]=="p":
                            ans+=1
                            break
        return ans
