class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        rook_x = -1
        rook_y 
= -1
        for i in range(0, 8):
            for j in range(0, 8):
                if board[i][j] == 'R':
                    rook_x = i
                    rook_y = j
        capture = 0
        for i in range(rook_x+1, 8):
            if board[i][rook_y] != '.':
                capture += (board[i][rook_y]=='p')
                break
        for i in range(rook_x-1, -1, -1):
            if board[i][rook_y] != '.':
                capture += (board[i][rook_y]=='p')
                break
        for i in range(rook_y+1, 8):
            if board[rook_x][i] != '.':
                capture += (board[rook_x][i]=='p')
                break
        for i in range(rook_y-1, -1, -1):
            if board[rook_x][i] != '.':
                capture += (board[rook_x][i]=='p')
                break
        return capture
