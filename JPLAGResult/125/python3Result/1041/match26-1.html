class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        res = 0
        for i in range(8):
            for j in range(8):
                if board[i][j] == 'R':
                    
                    ni, nj = i-1, j
                    while ni > 0 and board[ni][nj] == '.':
                        ni -= 1
                    if ni >= 0 and board[ni][nj] == 'p':
                        res += 1
                        
                    ni, nj = i, j-1
                    while nj > 0 and board[ni][nj] == '.':
                        nj -= 1
                    if nj >= 0 and board[ni][nj] == 'p':
                        res += 1
                        
                    ni, nj = i+1, j
                    while ni < 7 and board[ni][nj] == '.':
                        ni += 1
                    if ni <= 7 and board[ni][nj] == 'p':
                        res += 1
                        
                    ni, nj = i, j+1
                    while nj < 7 and board[ni][nj] == '.':
                        nj += 1
                    if nj <= 7 and board[ni][nj] == 'p':
                        res += 1
                        
        return res
