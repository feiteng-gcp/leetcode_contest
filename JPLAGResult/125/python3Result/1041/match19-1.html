class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        R = len(board)
        C = len(board[0])
        rx, ry = None, None
        for i in range(R):
            for j in range(C):
                if(board[i][j] == 'R'):
                    rx, ry = i, j
                    break
        if not rx:
            return 0
        res = 0
        for dx, dy in [(-1, 0), (1, 0), (0, 1), (0, -1)]:
            px, py = rx, ry
            while(0<=px+dx<R and 0<=py+dy<C):
                px, py = px+dx, py+dy
                
if(board[px][py] == 'p'):
                    res += 1
                    break
                elif(board[px][py] == 'B'):
                    break
        return res
