class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        
        def search(row, col):
            ans = 0
            flag = True
            for j in range(col + 1, 8):
                if board[row][j] == "B":
                    break
                if board[row][j] == 'p':
                    ans += 1
                    break
            for j in range(row + 1, 8):
                if board[j][col] == "B":
                    break
                if board[j][col] == 'p':
                    ans += 1
                    break
            for j in range(col - 1, -1, -1):
                if board[row][j] == "B":
                    break
                if board[row][j] == 'p':
                    ans += 1
                    break
            for j in range(row - 1, -1, -1):
                if board[j][col] == "B":
                    break
                if board[j][col] == 'p':
                    ans += 1
                    break
            return ans
        
        for i in range(8):
            for j in range(8):
                if board[i][j] == 'R':
                    
                    return search(i, j)
