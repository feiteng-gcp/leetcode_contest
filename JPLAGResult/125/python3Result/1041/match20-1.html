class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        n = len(board)
        
        
for i in range(n):
            for j in range(n):
                if board[i][j] == 'R':
                    x, y = i, j
                    break
        
        res = 0
        for di, dj in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
            i, j = x, y
            while 0 <= i + di < n and 0 <= j + dj < n:
                i += di
                j += dj
                c = board[i][j]
                if c == '.':
                    continue
                if c == 'p':
                    res += 1
                break
        return res
