class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        if not board or not board[0]:
            return 0
        r_i = r_j = None
        n = len(board)
        m = len(board[0])
        
for i in range(n):
            for j in range(m):
                if board[i][j] == 'R':
                    r_i = i
                    r_j = j
        
        dirs = [(-1,0),(1,0),(0,1),(0,-1)]
        res = 0
        
        for d_i,d_j in dirs:
            i,j = r_i,r_j
            while True:
                i += d_i
                j += d_j
                if 0 <= i < n and 0 <= j < m:
                    if board[i][j] == '.':
                        continue
                    if board[i][j] == 'p':
                        res += 1
                break
                
        return res
