def find_rock(b):
    for i in range(8):
        for j in range(8):
            if b[i][j] == 'R':
                return (i, j)


class Solution:
    def numRookCaptures(self, b: List[List[str]]) -> int:
        (x, y) = find_rock(b)
        
        dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        count = 0
        for d in dirs:
            for k in range(1, 9):
                i = x + d[0] * k
                j = y + d[1] * k
                if 0 <= i < 8 and 0 <= j < 8:
                    if b[i][j] == 'p':
                        count += 1
                    
                    if b[i][j] != '.':
                        break
        
        return count
