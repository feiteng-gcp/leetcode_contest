class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        L = 8
        for i in range(L):
            for j in range(L):
                if board[i][j] == 'R':
                    result = 0
                    def F(r,c):
                        nonlocal result
                        if board[r][c] == '.': return False
                        if board[r][c] == 'p': result += 1
                        return True
                    for k in range(i+1,L):
                        if F(k,j): break
                    for k in range(i-1,-1,-1):
                        if F(k,j): break
                    for k in range(j+1,L):
                        if F(i,k): break
                    for k in range(j-1,-1,-1):
                        if F(i,k): break
                    return result
