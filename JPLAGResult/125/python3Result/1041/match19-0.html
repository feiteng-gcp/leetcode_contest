class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        r_pos = None
        row, col = 0, 0
        while row < 8 and not r_pos:
            col = 0
            while col < 8 and not r_pos:
                if board[row][col] == 'R':
                    r_pos = (row, col)
                col += 1
            row += 1
        
        row, col = r_pos
        diff = [[0, 1], [0, -1], [1, 0], [-1, 0]]
        ans = 0
        for d in diff:
            row, col = r_pos
            while 0 <= row < 8 and 0 <= col < 8:
                row += d[0]
                col += d[1]
                
                if row < 0 or row >= 8 or col < 0 or col >= 8:
                    break
                
                if board[row][col] == 'p':
                    ans += 1
                    break
                elif board[row][col] == 'B':
                    break
            
            #print(d, ans)
        return ans
