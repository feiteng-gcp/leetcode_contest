class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        for i in range(8):
            for j in range(8):
                if board[i][j] == 'R':
                    dx = [0, -1, 0, +1]
                    dy = [-1, 0, +1, 0]
                    cnt = 0
                    for k in range(4):
                        x = i
                        y = j
                        while 0 <= x < 8 and 0 <= y < 8:
                            if board[x][y] == 'p':
                                cnt += 1
                                break
                            if board[x][y] == 'B':
                                break
                            x += dx[k]
                            y += dy[k]
                    return cnt
