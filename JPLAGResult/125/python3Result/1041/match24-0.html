class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        r = 0
        sx = 0
        sy = 0
        for i in range(8):
            for j in range(8):
                if board[i][j] == "R":
                    sy = i
                    sx = j
        for dx, dy in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
            for i in range(1, 7):
                x = sx + dx * i
                y = sy + dy * i
                if x < 0 or x >= 8 or y < 0 or y >= 8:
                    break
                c = board[y][x]
                if c == "p":
                    r += 1
                if c != ".":
                    break
        return r
