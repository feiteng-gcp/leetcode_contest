class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        found 
= False
        for r in range(8):
            for c in range(8):
                if board[r][c] == 'R':
                    found = True
                    break
            if found:
                break
        count = 0
        #print (r, c)
        for nr in range(r-1, -1, -1):
            if board[nr][c] in 'pB':
                if board[nr][c] == 'p':
                    count += 1
                break
        for nr in range(r+1, 8):
            if board[nr][c] in 'pB':
                if board[nr][c] == 'p':
                    count += 1
                break
        for nc in range(c+1, 8):
            if board[r][nc] in 'pB':
                if board[r][nc] == 'p':
                    count += 1
                break
        for nc in range(c-1, -1, -1):
            if board[r][nc] in 'pB':
                if board[r][nc] == 'p':
                    count += 1
                break
        return count
