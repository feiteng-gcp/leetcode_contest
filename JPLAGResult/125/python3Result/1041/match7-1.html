class Solution:
    def numRookCaptures(self, board: 'List[List[str]]') -> 'int':
        rook = []
        N = 8
        for x in range(N):
            for y in range(N):
                if board[x][y] == 'R':
                    rook = [x, y]
                    
break
            if rook:
                break
        ans = 0
        for dx, dy in [[-1,0],[1,0],[0,1],[0,-1]]:
            x, y = rook
            x, y = x+dx, y+dy
            while 0<=x<N and 0<=y<N:
                if board[x][y] == 'B':
                    break
                elif board[x][y] == 'p':
                    ans += 1
                    break
                x, y = x+dx, y+dy
        return ans
