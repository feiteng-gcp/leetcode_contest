class Solution:
    def numRookCaptures(self, board: 'List[List[str]]') -> 'int':
        rx, ry = 0, 0
        for i in range(8):
            for j in range(8):
                if board[i][j] == 'R':
                    rx = i
                    ry = j
        res = 0
        for i in range(rx+1, 8):
            
if board[i][ry] == 'B':
                break
            if board[i][ry] == 'p':
                res += 1
                break
        for i in range(rx-1, -1, -1):
            if board[i][ry] == 'B':
                break
            if board[i][ry] == 'p':
                res += 1
                break
        for j in range(ry+1, 8):
            
if board[rx][j] == 'B':
                break
            if board[rx][j] == 'p':
                res += 1
                break
        for j in range(ry-1, -1, -1):
            if board[rx][j] == 'B':
                break
            if board[rx][j] == 'p':
                res += 1
                break
        return res
