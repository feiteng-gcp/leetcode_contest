class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        n = len(board)
        m = len(board[0])
        for i in range(n):
            for j in range(m):
                if board[i][j] == 'R':
                    y = i
                    x = j
        
        res = 0
        travy = y+1
        travx = x
        while travy < m:
            if board[travy][travx] == "B":
                break
            elif board[travy][travx] == "p":
                #print(travy,travx)
                res += 1
                break
            travy += 1
            
        travy = y-1
        travx = x
        while travy >= 0:
            if board[travy][travx] == "B":
                break
            elif board[travy][travx] == "p":
                res += 1
                #print(travy,travx)
                break
            travy -= 1
            
        travy = y
        travx = x+1
        while travx < n:
            if board[travy][travx] == "B":
                break
            elif board[travy][travx] == "p":
                res += 1
                #print(travy,travx)
                break
            travx += 1
            
        travy = y
        travx = x-1
        while travx >= 0:
            if board[travy][travx] == "B":
                break
            elif board[travy][travx] == "p":
                res += 1
                #print(travy,travx)
                break
            travx -= 1
        return res
