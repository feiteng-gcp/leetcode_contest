class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        n = 8
        x, y = None, None
        for i in range(n):
            for j in range(n):
                if board[i][j] == 'R':
                    x, y = i, j
                    break
            if x is not None and y is not None:
                break
        res = 0
        for diff_x, diff_y in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            curr_x, curr_y = x, y
            while 0 <= curr_x < n and 0 <= curr_y < n:
                if board[curr_x][curr_y] == 'p':
                    res += 1
                    break
                elif board[curr_x][curr_y] == 'B':
                    break
                else:
                    curr_x += diff_x
                    curr_y += diff_y
        return res
