class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        dirs = (
            (1, 0),
            (-1, 0),
            (0, 1),
            (0, -1)
        )
        for i in range(8):
            for j in range(8):
                if board[i][j] == "R":
                    rook_i, rook_j = i, j
                    break
        reachable = 0
        for di, dj in dirs:
            i, j = rook_i, rook_j
            while 0 <= i < 8 and 0 <= j < 8 and board[i][j] != "B":
                if board[i][j] == "p":
                    reachable += 1
                    break
                i += di
                j += dj
        return reachable
