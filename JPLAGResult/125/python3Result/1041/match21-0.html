class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        
        rx, ry = 0, 0
        def find_rook():
            for i in range(8):
                for j in range(8):
                    if board[i][j] == 'R':
                        return i, j
        rx, ry = find_rook()
        
        # scan 
        count = 0
        start_points = [(rx, 0), (rx, 7), (0, ry), (7, ry)]
        directions = [(0, +1), (0, -1), (1, 0), (-1, 0)]
        for (x, y), (dx, dy) in zip(start_points, directions):
            last_endpoint = '.'
            while 0 <= x <= 7 and 0 <= y <= 7:
                # print(x, y)
                if (board[x][y], last_endpoint) in [('R', 'p')]:  # , ('p', 'R')]:
                    count += 1
                        
                if board[x][y] in 'pBR':
                    last_endpoint = board[x][y]
                    
                x += dx
                y += dy
                
        return count
