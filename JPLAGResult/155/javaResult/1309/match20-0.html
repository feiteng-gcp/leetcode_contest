class Solution {
    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
        for (int i = 0; i < n; i++) {
            if (group[i] == -1)
                group[i] = m++;
        }
        List<List<Integer>> ais = new ArrayList<>();
        for (int i = 0; i < n; i++)
            ais.add(new ArrayList<>());
        for (int i = 0; i < n; i++) {
            for (int j : beforeItems.get(i))
                ais.get(j).add(i);
        }
        List<List<Integer>> gs = new ArrayList<>();
        for (int i = 0; i < m; i++)
            gs.add(new ArrayList<>());
        for (int i = 0; i < n; i++)
            gs.get(group[i]).add(i);
        List<List<Integer>> sgb = new ArrayList<>();
        List<List<Integer>> sga = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            sgb.add(new ArrayList<>());
            sga.add(new ArrayList<>());
            for (int b : beforeItems.get(i)) {
                if (group[b]==group[i])
                    sgb.get(i).add(b);
            }
            for (int a : ais.get(i)) {
                if (group[a]==group[i])
                    sga.get(i).add(a);
                //System.out.println("pony "+i+" "+a);
            }
        }
        boolean[] vis = new boolean[n];
        for (int i = 0; i < m; i++) {
            if (gs.get(i).size()<=1)
                continue;
            List<Integer> sorted = new ArrayList<>();
            ArrayDeque<Integer> q = new ArrayDeque<>();
            for (int gi : gs.get(i)) {
                if (sgb.get(gi).size()==0) {
                    q.add(gi);
                }
            }
            if (q.isEmpty())
                return new int[0];
            while (!q.isEmpty()) {
                int u = q.removeFirst();
                boolean cont = false;
                if (vis[u])
                    continue;
                for (int t : sgb.get(u)) {
                    if (!vis[t]) {
                        cont = true;
                        break;
                    }
                }
                if (cont)
                    continue;
                sorted.add(u);
                vis[u] = true;
                for (int v : sga.get(u)) {
                    if (vis[v])
                        return new int[0];
                    q.add(v);
                }
            }
            //System.out.println(sorted.size() + " " + gs.get(i).size());
            if (sorted.size() != gs.get(i).size())
                return new int[0];
            gs.set(i, sorted);
        }
        List<Set<Integer>> gb = new ArrayList<>();
        List<Set<Integer>> ga = new ArrayList<>();
        List<List<Integer>> gbl = new ArrayList<>();
        List<List<Integer>> gal = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            gb.add(new HashSet<>());
            ga.add(new HashSet<>());
            gbl.add(new ArrayList<>());
            gal.add(new ArrayList<>());
        }
        for (int i = 0; i < n; i++) {
            for (int b : beforeItems.get(i)) {
                if (group[b]!=group[i]) {
                    if (!gb.get(group[i]).contains(group[b])) {
                        gb.get(group[i]).add(group[b]);
                        gbl.get(group[i]).add(group[b]);
                    }
                }
                //System.out.println("pony2 "+group[i]+" "+group[b]);
            }
            for (int a : ais.get(i)) {
                if (group[a]!=group[i]) {
                    if (!ga.get(group[i]).contains(group[a])) {
                        ga.get(group[i]).add(group[a]);
                        gal.get(group[i]).add(group[a]);
                    }
                }
                //System.out.println("pony3 "+group[i]+" "+group[a]);
            }
        }
        int index = 0;
        int[] ans = new int[n];
        boolean[] vg = new boolean[m];
        ArrayDeque<Integer> q = new ArrayDeque<>();
        for (int i = 0; i < m; i++) {
            if (gbl.get(i).size()==0) {
                //System.out.println(i);
                q.add(i);
            }
        }
        while (!q.isEmpty()) {
            int u = q.removeFirst();
            boolean cont = false;
            if (vg[u])
                continue;
            for (int t : gbl.get(u)) {
                if (!vg[t]) {
                    cont = true;
                    break;
                }
            }
            if (cont)
                continue;
            for (int i : gs.get(u))
                ans[index++] = i;
            vg[u] = true;
            for (int v : gal.get(u)) {
                //System.out.println(u+" "+v);
                if (vg[v])
                    return new int[0];
                q.add(v);
            }
        }
        //System.out.println(index);
        if (index != n)
            return new int[0];
        return ans;
    }
}
