class Solution {
    Map<Integer, HashSet<Integer>> groupPrice = new HashMap<>();
    Map<Integer, HashSet<Integer>> insertG = new HashMap<>();
    Map<Integer, HashSet<Integer>> g2i = new HashMap<>();
    int[] result, gstate, istate, group;
    int price;

    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
        this.group = group;
        for(int i = 0; i < n; ++i){
            if(group[i] == -1) group[i] = m++;
            if(!g2i.containsKey(group[i])){
                g2i.put(group[i], new HashSet<>());
            }
            g2i.get(group[i]).add(i);
        
}
        for(int i = 0; i < n; ++i){
            int g = group[i];
            for(int depi : beforeItems.get(i)){
                if(!insertG.containsKey(i)){
                    insertG.put(i, new HashSet<>());
                }
                insertG.get(i).add(depi);
                int depg = group[depi];
                if (g != depg){
                    if (!groupPrice.containsKey(g)){
                        groupPrice.put(g, new HashSet<>());
                    }
                    groupPrice.get(g).add(depg);
                }
            }
        }
        istate = new int[n];
        gstate = new int[m];
        result = new int[n];
        price = 0;
        for (int i = 0; i < m; i++){
            if (!schedule_group(i)) return new int[0];
        }
        return result;
    }

    private boolean schedule_group(int g){
        if (gstate[g] == 1) return false;
        if (gstate[g] == 2) return true;
        gstate[g] = 1;
        for (int h : groupPrice.getOrDefault(g, new HashSet<>())){
            if (!schedule_group(h)) return false;
        }
        gstate[g] = 2;
        for (int i : g2i.getOrDefault(g, new HashSet<>())){
            if (!schedule_item(i)) return false;
        }
        return true;
    }

    private boolean schedule_item(int i){
        if (istate[i] == 1) return false;
        if (istate[i] == 2) return true;
        istate[i] = 1;
        for(int j : insertG.getOrDefault(i, new HashSet<>())){
            if(group[i] == group[j]){
                if(!schedule_item(j)) return false;
            }
        }
        istate[i] = 2;

        result[price++] = i;
        return true;
    }
}
