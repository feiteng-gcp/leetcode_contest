class Solution {
    List<Integer> topSort(int n, int[] deg, List<Set<Integer>> edges) {
        Queue<Integer> q = new LinkedList<>();
        
        for(int i=0; i<n; ++i) {
            if(deg[i] == 0) {
                q.add(i);
            }
        }
        List<Integer> ret = new ArrayList<>();
        while(!q.isEmpty()) {
            int cur = q.poll();
            ret.add(cur);
            for(Integer to: edges.get(cur)) {
                --deg[to];
                if(deg[to] == 0) {
                    q.add(to);
                }
            }
        }
        return ret;
        
    }
    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
        int maxG = -1;
        for(int g: group) maxG = Math.max(g, maxG);
        int gid = maxG+1;
        
        Map<Integer, List<Integer>> gtoe = new HashMap<>();
        for(int i=0; i<n; ++i) {
            if(group[i] == -1) {
                group[i] = gid++;
                gtoe.put(group[i], Arrays.asList(new Integer[]{i}));
            }else {
                if(!gtoe.containsKey(group[i])) {
                    gtoe.put(group[i], new ArrayList<>());
                }
                gtoe.get(group[i]).add(i);
            }
        }
        int ng = gid;
        int[] gdeg = new int[ng];
        List<Set<Integer>> gedges = new ArrayList<>();
        for(int i=0; i<ng; ++i) {
            gedges.add(new HashSet<>());
        }
        for(int i=0; i<n; ++i) {
            int gi = group[i];
            for(int j: beforeItems.get(i)) {
                int gj = group[j];
                if(gj != gi) {
                    if(!gedges.get(gj).contains(gi)) {
                        ++gdeg[gi];
                        
gedges.get(gj).add(gi);
                    }
                    
                }
            }
        }
        List<Integer> groupOrder = topSort(ng, gdeg, gedges);
        //for(Integer tmp: groupOrder)
            //System.out.println(tmp);
        if(groupOrder.size() != ng) return new int[0];
        
        int[] ret = new int[n];
        int pos = 0;
        for(int ii=0; ii<ng; ++ii) {
            int i = groupOrder.get(ii);
            if(!gtoe.containsKey(i)) continue;
            List<Integer> es = gtoe.get(i);
            int ngg = es.size();
            
int[] ggdeg = new int[ngg];
            int[] ind = new int[n];
            List<Set<Integer>> ggedges = new ArrayList<>();
            for(int j=0; j<ngg; ++j) {
                ggedges.add(new HashSet<>());
            }
            for(int j=0; j<es.size(); ++j) {
                ind[es.get(j)] = j;
            }
            for(int j=0; j<es.size(); ++j) {
                int e = es.get(j);
                for(int k: beforeItems.get(e)) {
                    if(group[k] == i) {
                        ++ggdeg[j];
                        ggedges.get(ind[k]).add(j);
                    }
                }
            }
            List<Integer> sorted = topSort(ngg, ggdeg, ggedges);
            if(sorted.size() != ngg) return new int[0];
            
            for(Integer se: sorted) {
                ret[pos++] = es.get(se);
            }
        }
        
        return ret;
    }
}
