class Solution {
    boolean[] v1;
    boolean[] v2;
    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
        List<Integer>[] graph = new List[n+m];
        for (int i=0; i<n; i++) {
            int x = getGroup(group, i, n);
            if (graph[x] == null)
                graph[x] = new ArrayList<>();
            for (int j : beforeItems.get(i)) {
                int y = getGroup(group, j, n);
                if (x != y)
                    graph[x].add(y);
            }
        }
        List<Integer> sorted = new ArrayList<>();
        v1 = new boolean[n+m];
        v2 = new boolean[n+m];
        for (int i=0; i<n; i++)
            if (!search(graph, sorted, getGroup(group, i, n)))
                return new int[0];
        List<Integer>[] gm = new List[m];
        for (int i=0; i<n; i++) {
            int x = group[i];
            if (x != -1) {
                if (gm[x] == null)
                    gm[x] = new ArrayList<>();
                gm[x].add(i);
            }
        }
        int[] ans = new int[n];
        int i = 0;
        for (int x : sorted) {
            if (x < n) {
                ans[i] = x;
                i++;
            } else {
                List<Integer> gi = gm[x-n];
                List<Integer>[] graph2 = new List[n];
                for (int z : gi) {
                    graph2[z] = new ArrayList<>();
                    for (int y : beforeItems.get(z))
                        if (group[y] == group[z])
                            graph2[z].add(y);
                }
                List<Integer> sorted2 = new ArrayList<>();
                for (int z : gi)
                     if (!search(graph2, sorted2, z))
                        return new int[0];
                for (int z : sorted2) {
                    ans[i] = z;
                    i++;
                }
            }
        }
        return ans;
    }

    private int getGroup(int[] group, int i, int n) {
        return group[i] == -1 ? i : group[i] + n;
    }

    private boolean search(List<Integer>[] graph, List<Integer> ans, int x) {
        if (v1[x])
            return true;
        if (v2[x])
            return false;
        v2[x] = true;
        if (graph[x] == null) {
            v1[x] = true;
            ans.add(x);
            return true;
        }
        for (int y : graph[x]) {
            if (!search(graph, ans, y))
                return false;
        }
        v1[x] = true;
        ans.add(x);
        return true;
    }
}
