class Solution {
    class Graph {
        List<Set<Integer>> nodes = new ArrayList<>();
        int[] cnt;
        int n;
        Graph(int n) {
            for (int i = 0; i < n; i++) {
                nodes.add(new HashSet<>());
            }
            cnt = new int[n];
            this.n = n;
        }
        void addEdge(int u, int v) {
            if (!nodes.get(u).contains(v)) {
                nodes.get(u).add(v);
                cnt[v]++;
            }
        }
        List<Integer> topsort() {
            List<Integer> res = new ArrayList<>();
            Queue<Integer> queue = new LinkedList<>();
            for (int i = 0; i < n; i++) {
                if (cnt[i] == 0) {
                    queue.add(i);
                }
            }
            while (!queue.isEmpty()) {
                int cur = queue.poll();
                res.add(cur);
                for (int v: nodes.get(cur)) {
                    cnt[v]--;
                    if (cnt[v] == 0) queue.add(v);
                }
            }
            if (res.size() != n) return null;
            return res;
        }
    }

    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
        int idx = m;
        int[] cnt = new int[m];
        List<List<Integer>> gs = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            gs.add(new ArrayList<>());
        }
        for (int i = 0; i < n; i++) {
            if (group[i] == -1) {
                group[i] = idx++;
                gs.add(new ArrayList<>());
                gs.get(idx-1).add(i);
            } else {
                cnt[group[i]]++;
                gs.get(group[i]).add(i);
            }
        }
        Graph groupGraph = new Graph(idx);
        List<Graph> graphs = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            Graph g = new Graph(cnt[i]);
            graphs.add(g);
        }

        for (int i = 0; i < n; i++) {
            List<Integer> b = beforeItems.get(i);
            int g1 = group[i];
            int idx1 = Collections.binarySearch(gs.get(g1), i);
            for (int p: b) {
                int g2 = group[p];
                if (g2 == g1) {
                    int idx2 = Collections.binarySearch(gs.get(g2), p);
                    graphs.get(g1).addEdge(idx2, idx1);
                } else {
                    groupGraph.addEdge(g2, g1);
                }
            }
        }
        List<Integer> groupsort = groupGraph.topsort();
        if (groupsort == null) return new int[0];
        int[] res = new int[n];
        int j = 0;
        for (int i : groupsort) {
            List<Integer> grp = gs.get(i);
            if (grp.size() == 1) {
                res[j++] = grp.get(0);
            } else {
                List<Integer> sort = graphs.get(i).topsort();
                if (sort == null) return new int[0];
                for (int k : sort) {
                    res[j++] = grp.get(k);
                }
            }
        }
        return res;
    }
}
