class Solution {
    private int t;
    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
        // List<Integer>[] list = new ArrayList[n + m];
        Set<Integer>[] adj = new HashSet[n + m];
        Map<Integer, List<Integer>>[] before = new HashMap[m];
        for (int i = 0; i < n + m; i++) {
            // list[i] = new ArrayList<>();
            adj[i] = new HashSet<>();
            if (i < m) before[i] = new HashMap<Integer, List<Integer>>();
        }
        boolean[] visited = new boolean[n + m];
        for (int i = m; i < n + m; i++) visited[i] = true;
        for (int i = 0; i < n; i++) {
            if (group[i] != -1) {
                Map<Integer, List<Integer>> map = before[group[i]];
                List<Integer> l = new ArrayList<>();
                map.put(i, l);
                for (int j : beforeItems.get(i)) {
                    if (group[j] == group[i]) {
                        l.add(j);
                    }
                    else if (group[j] == -1) {
                        adj[group[i]].add(j + m);
                    }
                    else {
                        adj[group[i]].add(group[j]);
                    }
                }
            }
            else {
                visited[i + m] = false;
                for (int j : beforeItems.get(i)) {
                    if (group[j] == -1) {
                        adj[i + m].add(j + m);
                    }
                    else {
                        adj[i + m].add(group[j]);
                    }
                }
            }
        }
        boolean[] onstack = new boolean[n + m];
        int[] stack = new int[n + m];
        for (int i = 0; i < n + m; i++) {
            if (!dfs(i, visited, onstack, stack, adj)) return new int[0];
        }
        visited = new boolean[n + m];
        int[] res = new int[n];
        int len = t;
        t = 0;
        for (int i = 0; i < len; i++) {
            if (stack[i] < m) {
                int v = stack[i];
                Map<Integer, List<Integer>> map = before[v];
                for (int k : map.keySet()) {
                    if (!dfs2(k, visited, onstack, res, map)) return new int[0];
                }
            }
            else res[t++] = stack[i] - m;
        }
        return res;
    }
    private boolean dfs(int v, boolean[] visited, boolean[] onstack, int[] stack, Set<Integer>[] adj) {
        if (onstack[v]) return false;
        if (visited[v]) return true;
        onstack[v] = true;
        visited[v] = true;
        for (int w : adj[v]) {
            if (!dfs(w, visited, onstack, stack, adj)) return false;
        }
        onstack[v] = false;
        stack[t++] = v;
        return true;
    }
    private boolean dfs2(int v, boolean[] visited, boolean[] onstack, int[] stack, Map<Integer, List<Integer>> adj) {
        if (onstack[v]) return false;
        if (visited[v]) return true;
        onstack[v] = true;
        visited[v] = true;
        for (int w : adj.get(v)) {
            if (!dfs2(w, visited, onstack, stack, adj)) return false;
        }
        onstack[v] = false;
        stack[t++] = v;
        return true;
    }
}
