class Solution {
	private Map<Integer , Set<Integer>> graph = new HashMap<>();
	
	// index -> virtual index	
	private Map<Integer , Integer> virtualIndexMap = new HashMap<>();	
	private Map<Integer , List<Integer>> virtualGroup = new HashMap<>();
	
	private Map<Integer , Set<Integer>> virtualGraph = new HashMap<>();	
	
    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
    	int i , total = beforeItems.size();
    	for (i = 0;i < n;i ++) {
    		graph.put(i , new HashSet<>());    		
    	}
    	for (i = 0;i < total;i ++) {
    		for (int prev : beforeItems.get(i)) {
    			graph.get(prev).add(i);    			
    		}    		
    	}
    	int nextVirtualIndex = 0;    	
    	Map<Integer , List<Integer>> groupMap = new HashMap<>();
    	for (i = 0;i < group.length;i ++) {
    		if (group[i] < 0) {
    			virtualIndexMap.put(i , nextVirtualIndex);
    			virtualGroup.put(nextVirtualIndex , Arrays.asList(i));
    			nextVirtualIndex ++;    			
    		} else {
    			groupMap.putIfAbsent(group[i] , new ArrayList<>());
    			groupMap.get(group[i]).add(i);
    		}
    	}
    	for (List<Integer> list : groupMap.values()) {
    		virtualGroup.put(nextVirtualIndex , new ArrayList<>());
    		for (int index : list) {
    			virtualIndexMap.put(index , nextVirtualIndex);
    			virtualGroup.get(nextVirtualIndex).add(index);    			
    		}
    		nextVirtualIndex ++;    		
    	}    	
    	for (List<Integer> list : virtualGroup.values()) {
    		if (!sort(list)) {
    			return new int[] {};    			
    		}
    	}    	
    	for (i = 0;i < nextVirtualIndex;i ++) {
    		virtualGraph.put(i , new HashSet<>());
    	}
    	buildVirtualGraph();
    	return finalSort();
    }
    
    private void buildVirtualGraph() {
    	for (Map.Entry<Integer , Set<Integer>> entry : graph.entrySet()) {
    		int from = entry.getKey();
    		for (int to : entry.getValue()) {
    			int index1 = virtualIndexMap.get(from) , index2 = virtualIndexMap.get(to);
    			if (index1 != index2) {
    				virtualGraph.get(index1).add(index2);
    			}    			
    		}
    	}    	
    }
    
    private int[] finalSort() {    		
    	int[] inDegree = new int[virtualGraph.size()];
    	for (Map.Entry<Integer , Set<Integer>> entry : virtualGraph.entrySet()) {
    		for (int to : entry.getValue()) {
    			inDegree[to] ++;
    		}
    	}
    	Queue<Integer> queue = new LinkedList<>();

t<Integer> ans = new ArrayList<>();
    	for (int i = 0;i < inDegree.length;i ++) {
    		if (inDegree[i] == 0) {
    			queue.add(i);
    		}
    	}
    	while (!queue.isEmpty()) {
    		int current = queue.poll();
    		ans.addAll(virtualGroup.get(current));
    		for (int next : virtualGraph.get(current)) {

    			inDegree[next] --;
    			if (inDegree[next] == 0) {
    				queue.add(next);    				
    			}
    		}
    	}
    	for (int i = 0;i < inDegree.length;i ++) {
    		if (inDegree[i] != 0) {
    			return new int[] {};
    		}
    	}
    	int[] array = new int[ans.size()];
    	for (int i = 0;i < ans.size();i ++) {
    		array[i] = ans.get(i);
    	}
    	return array;
    }
    
    private boolean sort(List<Integer> list) {
    	// topo sort based on graph
    	List<Integer> temp = new ArrayList<>();    	
    	int[] inDegree = new int[list.size()];
    	Map<Integer , Integer> map = new HashMap<>();
    	for (int i = 0;i < list.size();i ++) {
    		map.put(list.get(i) , i);    		
    	}
    	for (int i = 0;i < list.size();i ++) {
    		for (int next : graph.get(list.get(i))) {
    			if (map.containsKey(next)) {
    				inDegree[map.get(next)] ++;    				
    			}
    		}
    	}
    	Queue<Integer> queue = new LinkedList<>();
    	for (int i = 0;i < list.size();i ++) {

    		if (inDegree[i] == 0) {    			
    			queue.add(i);    			
    		}
    	}
    	while (!queue.isEmpty()) {
    		int current = queue.poll();
    		temp.add(current);
    		for (int next : graph.get(list.get(current))) {    			
    			if (map.containsKey(next)) {
    				int index = map.get(next);
    				inDegree[index] --;
    				if (inDegree[index] == 0) {
    					queue.add(index);    					
    				}
    			}
    		}
    	}
    	if (temp.size() == list.size()) {
    		List<Integer> newList = new ArrayList<>();
    		for (int i = 0;i < list.size();i ++) {
    			newList.add(list.get(temp.get(i)));
    		}
    		for (int i = 0;i < list.size();i ++) {
    			list.set(i , newList.get(i));
    		}
    		return true;
    	} else {
    		return false;
    	}
    }   
}
