class Solution {
    int[] ans;
    int id;
    int V;
    int[] group;
    Map<Integer, List<Integer>> map;
    List<Integer>[] nbs;
    int n, m;
    int[] indeg;
    
    public int[] sortItems(int n, int m, int[] group, List<List<Integer>> beforeItems) {
        //topological sort
        //(1) grouping
        map = new HashMap<>();
        this.group = group;
        this.n = n;
        this.m = m;
        for (int i = 0; i < n; i++) {
            int gid = group[i];
            if (gid >= 0) {
                if (!map.containsKey(gid)) map.put(gid, new ArrayList<>());
                
                map.get(gid).add(i);
            }
        }
        
        V = n + m;
        
        nbs = new List[V];
        for (int i = 0; i < V; i++) nbs[i] = new ArrayList<>();
        indeg = new int[V];
        
        for (int i = 0; i < n; i++) {
            List<Integer> bf = beforeItems.get(i);
            for (int fr : bf) {
                if (group[fr] < 0 || group[i] < 0) {
                    if(group[fr] < 0 && group[i] < 0) {
                        nbs[fr].add(i);
                        indeg[i]++;
                    }else if (group[fr] < 0) {
                        int f = fr, t = group[i] + n;
                        nbs[f].add(t);
                    indeg[t]++;
                    }else  {
                        int f = group[fr] + n, t = i;
                        nbs[f].add(t);
                    indeg[t]++;
                    }
                }else if (group[i] != group[fr]) {                    
                    int f = group[fr] + n, t = group[i] + n;
                    nbs[f].add(t);
                    indeg[t]++;
                }else {
                    //same group
                    nbs[fr].add(i);
                    indeg[i]++;
                }
            }
        }
        
        
        id = 0;
        ans = new int[n];
        
        Deque<Integer> q = new ArrayDeque<>();
        for (int i = 0; i < V; i++) {
            if (indeg[i] == 0 && (i >= n || (i < n && group[i] < 0))) {                
                q.add(i);
            }
        }
        
        while (!q.isEmpty()) {
            int node = q.remove();
            // System.out.println("process " + node);
            if (node >= n) {
                if (!processGroup(node))
                        return new int[0];
            }else 
                ans[id++] = node;
            
            for (int next : nbs[node]) {
                indeg[next]--;
                if (indeg[next] == 0) q.add(next);
            }
        }
        
        if (id < n) return new int[0];
        
        return ans;        
    }
    
    private boolean processGroup(int gid) {
        gid -= n; // transform to group id
         //System.out.println("process group " + gid);
        if(!map.containsKey(gid)) return true; //empty group
        List<Integer> g = map.get(gid);
        if (g.size() == 0) return true;
        
        int added = 0;
        Deque<Integer> q = new ArrayDeque<>();
        for (int i : g) {
            if (indeg[i] == 0) {
                q.add(i);
            }
        }
        
        while (!q.isEmpty()) {
            int cur = q.remove();
            
                ans[id++] = cur;
            //System.out.println("process and add " + cur);
                added++;
            for (int next : nbs[cur]) {
                indeg[next]--;
                if (indeg[next] == 0) {                    
                    q.add(next);
                }
            }
        }
        
        return added == g.size();
    }
    
    
}
