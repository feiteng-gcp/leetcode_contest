class Solution {
    
    public int nthUglyNumber(long n, long a, long b, long c) {
        
        long lcm_ab=lcm(a, b),
             lcm_bc=lcm(b, c),
             lcm_ac=lcm(a, c),
             lcm_abc=lcm(a, 
lcm(b, c));
        
        long lo=1L, hi=Integer.MAX_VALUE;
        
        while(lo<hi) {
            
            long mid=(lo+hi)/2;
            
            long count=mid/a+mid/b+mid/c-(mid/lcm_ab+mid/lcm_bc+mid/lcm_ac)+mid/lcm_abc;
            // System.out.println(lo+" "+hi+" "+mid+" "+count);
            
            if(count>=n)
                hi=mid;
            else
                lo=mid+1;
        }

        return (int)(lo);
    }
    
    public long gcd(long a, long b) {
        if(b==0)
            return a;
        return gcd(b, a%b);
    }
    
    public long lcm(long a, long b) {
        return (a*b)/gcd(a, b);
    }
}
