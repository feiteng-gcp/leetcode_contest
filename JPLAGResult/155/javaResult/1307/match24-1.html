class Solution {
    
    private long gcd(long a, long b) {  
        if (a == 0)  {
            return b;  
        
}
        
        return gcd(b % a, a);  
    }  
    
    private long lcm(long a, long b) {  
        return (a * b) / gcd(a, b);  
    }  
    
    private long check(long a, long b, long c, long num) { 
        return ((num / a) + (num / b) + (num / c)  
                - (num / lcm(a, b))  
                - (num / lcm(b, c))  
                - (num / lcm(a, c))  
                + (num / lcm(a, lcm(b, c))));  
    }
  
    
public int nthUglyNumber(int n, int a, int b, int c) {
        int l = 1, h = Integer.MAX_VALUE;
        
        while (l < h) {
            int m = l + (h - l) / 2;  
            int count = (int)(check((long)a, (long)b, (long)c, (long)m));
            
            if (count < n) {
                l = m + 1;  
            } else 
{
                h = m;   
            }
        }
        return l;
    }
}
