class Solution {
    long gcd(long a, long b) {
        if (b == 0) {
            return a;
        } else {
            return gcd(b, a % b);
        }
    }
    
    public int nthUglyNumber(int n, int a, int b, int c) {
        long x = (long)a * b / gcd(a, b);
        long y = (long)a * c / 
gcd(a, c);
        long z = (long)b * c / gcd(b, c);
        long t = x * c / (gcd(x, c));
        
        long left = 1, right = 2000000000;
        while (left <= right) {
            long mid = (left + right) / 2;
            long total = mid / a + mid / b + mid / c - 
                mid / x - mid / y - mid / z + mid / t;
            if (total >= n) 
{
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return (int)left;
    }
}
