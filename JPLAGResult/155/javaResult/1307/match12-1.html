class Solution {
    
public long gcd(long a, long b) {
        if (a == 0) return b;
        return gcd(b % a, a);
    }
    
    public long lcm(long a, long b) {
        return a * (b / gcd(a, b));
    }
    
    public long eval(long a, long b, long c, long n) {
        return n / a + n / b + n / c - n / lcm(a, b) - n / lcm(b, c) - n / lcm(a, c) + n / lcm(lcm(a, b), c);
    }
    
    
public int nthUglyNumber(int n, int a, int b, int c) {
        long _n = (long)n;
        long _a = (long)a;
        long _b = (long)b;
        long _c = (long)c;
        long left = 1;
        long right = 2_000_000_000;
        while (left < right) {
            long mid = (left + right) / 2;
            long val = eval(_a, _b, _c, mid);
            // System.out.printf("%d, %d, %d\n", mid, val, n);
            if (val > _n) right = mid - 1;
            else if (val == _n) right = mid;
            else left = mid + 1;
        }
        return (int)left;

    }
}
