class Solution {
    public int _n(int n, int a, int b) {
        long target = n;
        if(b % a == 0) return n *a;
        long l = 1;
            long r = a * n;
            while(l < r) {
                long m = l + (r - l)/2;
                long index = getIndex(a,b,m);
                if(index < target) {
                    l = m+1;
                }else if(index > target) {
                    r = m-1;
                }else {
                    l = m;
                    r = m;
                    break;
                }
            }
            while(getIndex(a,b,l) == target) {
                if(isOK(a,b,l)) return(int)l;
                l--;
            }
            while(getIndex(a,b,r) == target) {
                if(isOK(a,b,r)) return (int)r;
                r++;
            }
            return -1;
    }
    
    public int nthUglyNumber(int n, int a, int b, int c) {
        long target = n;
        int[] arr = new int[]{a,b,c};
        Arrays.sort(arr);
        a = arr[0];
        b = arr[1];
        c = arr[2];
        if(a==b || b==c) {
            if(a == c) {
                return n * a;
            }
            return _n(n,a,c);
        }
        if(b%a == 0) return _n(n,a,c);
        if(c % b == 0) return _n(n,a,b);
        if(c % a == 0) return _n(n,a,b);
        long r = arr[0] * n;
        long l = 1;
        while(l < r) {
            long m = l + (r-l) / 2;
            long index = getIndex(arr[0], arr[1], arr[2], m);
            if(index < target) {
                l = m+1;
            }else if(index > target) {
                r = m-1;
            }else{
                l = m;
                r = m;
                break;
            }
        }
        if(isOK(a,b,c,l)) return (int)l;
        l--;
        while(getIndex(a,b,c,l) == target) {
            if(isOK(a,b,c,l)) return (int) l;
            l--;
        }
        while(getIndex(a,b,c,r) == target) {
            if(isOK(a,b,c,r)) return (int) r;
            r++;
        }
        return -1;
    }
    
    long getIndex(long a, long b, long num) {
        long ab = a * b / help(a,b);
        return num / a + num / b - num / ab;
    }
    
    boolean isOK(long a, long b, long num) {
        return num % a == 0 || num % b == 0;
    }
    
    boolean isOK(long a, long b, long c, long num) {
        return num % a == 0 || num % b == 0 || num % c == 0;
    }
    
    long getIndex(long a, long b, long c, long num) {
        long ab = a * b / help(a,b);
        long bc = b * c / help(b,c);
        long ac = a * c / help(a,c);
        long _ab = help(a,b);
        long abc = a*b*c / help(Math.max(c,_ab), Math.min(c,_ab));
        long cnt = num / a + num / b + num / c - num / ab - num / bc - num / ac + num / abc;
        return cnt;
    }
    
    long help(long a, long b) {
        if(b == 0) return a;
        else return help(b, a % b);
    }
}
