class Solution {
  
public int nthUglyNumber(int n, int a, int b, int c) {
    long s = 0, e = 2000000000;
    while(s < e) {
      long m = (s+e)/2;
      long k = getCount(m, a, b, c);
      if(k < n) {
        s = m+1;
      } else 
if(k > n) {
        e = m-1;
      } else {
        e = m;
      }
    }
    return (int) s;
  }

  
long gcd(long a, long b) {
    if (a == 0)
      return b;
    return gcd(b % a, a);
  }

  long lcm(long a, long b) {
    return (a*b)/gcd(a, b);
  }

  private long getCount(long m, int a, int b, int c) {
    long res = (m/a) + (m/b) + (m/c);
    res -= (m/lcm(a, b));
    res -= (m/lcm(b, c));
    res -= (m/lcm(c, a));
    res += (m/lcm(c, lcm(a, b)));
    return res;
  }
}
