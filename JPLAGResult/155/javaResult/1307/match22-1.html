class Solution {
    public int nthUglyNumber(int n, int a, int b, int c) {
        int lo = 1, hi = Integer.MAX_VALUE;
        long ab1 = gcd(a, b);
        long ab2 = a / ab1 * b;
        long bc1 = gcd(b, c);
        long bc2 = b / bc1 * c;
        long ac1 = gcd(a, c);
        long ac2 = a / ac1 * c;
        long abc1 = gcd(ab2, bc2);
        long abc2 = ab2 / abc1 * bc2;
        // System.out.printf("%d %d %d, %d\n", ab2, bc2, ac2, abc2);
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            int cmp = count(mid, a, b, c, ab2, bc2, ac2, abc2) - n;
            if (cmp < 0) 
lo = mid + 1;
            else hi = mid;
        }
        return lo;
    }
    private long gcd(long m, long n) {
        if (n == 0) return m;
        return gcd(n, m % n);
    }
    private int count(long num, int a, int b, int c, long ab, long bc, long ac, long abc) {
        return (int) (num / a + num / b + num / c - num / ab - num / bc - num / ac + num / abc);
    }
}
