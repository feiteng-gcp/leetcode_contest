class Solution {
    private long gcd(long a, long b) {
        if (b == 0) {
            return a;
        } else {
            return gcd(b, a % b);
        }
    
}
    private long lcm(long a, long b) {
        return a * b / gcd(a, b);
    }
    private long cal(long t, int a, int b, int c) {
        long ret = 0;
        ret += t / a;
        ret += t / b;
        ret += t / c;
        ret -= t / lcm(a, b);
        ret -= t / lcm(a, c);
        ret -= t / lcm(b, c);
        ret += t / lcm(lcm(a, b), c);
        return ret;
    
}
    public int nthUglyNumber(int n, int a, int b, int c) {
        int left = 0, right = 2000000010;
        while (right - left > 1) {
            int mid = Long.valueOf(((long)left + right) / 2).intValue();
            if (cal(mid, a, b, c) >= n) 
{
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }
}
