class Solution {
    public int nthUglyNumber(int n, int a, int b, int c) {
        int start = 1, end = Integer.MAX_VALUE;  
  
        while (start < end) {  
            int mid = start + (end - start) / 2;  
            if (count(a, b, c, mid) < n)  
                start = mid + 1;
            else {
                end = mid;  
            }
        }
  
        return start;  
    }
    
    protected int gcd(int a, int b)  {  
        if (a == 0)  
            return b;  
          
        return gcd(b % a, a);  
    }
   
    protected long lcm(long a, long b)  {  
        return (a * b) / gcd((int)a, (int)b);  
    }
    
    protected int count(int a, int b, int c, int num) {  
        return ((num / a) + (num / b) + (num / c)  
                - (int)((long)num / lcm(a, b))  
                - (int)((long)num / lcm(b, c))  
                - (int)((long)num / lcm(a, c))  
                + (int)((long)num / lcm(a, lcm(b, c))));  
    }
}
