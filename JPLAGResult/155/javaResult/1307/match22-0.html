class Solution {
    public int nthUglyNumber(int n, int a, int b, int c) {
        // v is the nth number  implies there are n ugly number <= v
        int lo = 1, hi = Integer.MAX_VALUE;
        long ab, ac, bc, abc;
        ab = 1L * a * b / gcd(a, b);
        ac = 1L * a * c / gcd(a, c);
        bc = 1L * b * c / gcd(b, c);
        abc = 1l * ab * c / gcd(ab, c);
        
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if(cnt(a, b, c, mid, ab, ac, bc, abc) >= n) hi = mid - 1;
            else lo = mid + 1;
        }
        return lo;
    }
    
    private long gcd(long a, long b){
        if (b == 0) return a;
        return gcd(b, a % b);
    }
    private long cnt(long a, long b, long c, long val, long ab, long ac, long bc, long abc) {
        return val / a + val / b + val / c - val / ab - val / ac - val / bc
            + val / abc;
    }
}
