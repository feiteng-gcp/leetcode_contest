class Solution {   
    final static long INF = 2000000000;
    
    private long getRank3(long num, long a) {
        return num/a;
    }
    
    private long getRank2(long num, long a, long b) {
        long ra = num/a;
        long rb = num/b;
        long rab = num/a/b;
        return ra + rb - rab;
    }
    
    private long getGCD(long a, long b) {
        if (a < b) {
            long tmp = a;
            a = b;
            b = tmp;
        }
        if (b == 0) {
            return a;
        }
        return getGCD(b, a%b);
    }
    
    private long getLCM(long a, long b) {
        long g = getGCD(a, b);
        return a/g*b;
    }
    
    private long getLCM3(long a, long b, long c) {
        return getLCM(getLCM(a, b), c);
    }
    
    private long getRank(long num, long a, long b, long c) {
        long ra = num/a;
        long rb = num/b;
        long rc = num/c;
        long rab = num/getLCM(a, b);
        long rbc = num/getLCM(b, c);
        long rca = num/getLCM(c, a);
        long rabc = num/getLCM3(a, b, c);
        return ra + rb + rc - rab - rbc - rca + rabc;
    }
    
    private long getRankByDiv(long num, int[] div, int cnt) {
        if (cnt == 3) {
            return getRank(num, div[0], div[1], div[2]); 
        } else if (cnt == 2) {
            return getRank2(num, div[0], div[1]);
        } else {
            return getRank3(num, div[0]);
        }
    }
    
    public int nthUglyNumber(int n, int a, int b, int c) {
        long l = 1;
        long r = INF;
        
        //int cnt = 3;
        //int[] div = new int[3];
        //div[0] = a;
        //div[1] = b; 
        //div[2] = c;
        //Arrays.sort(div);
        
        //if (div[2] % div[0] == 0 || div[2] % div[1] == 0) {
        //    cnt--;
        //}
        
        //if (div[1] % div[0] == 0) {
        //    cnt--;
        //    div[1] = div[2];
        //}
        
        while (l < r) {
            long mid = (l+r)/2;
            // System.out.println(mid);
            long rank = getRank(mid, a, b, c);
            // System.out.println(rank);
            if (rank < n) {
                l = mid+1;
            } else {
                r = mid;
            }
        }
        return (int)l;
    }
}
