class Solution {
    public int nthUglyNumber(int n, int aint, int bint, int cint) {
		long a = aint;
		long b = bint;
		long c = cint;
        long min = Math.min(a, b);
        min = Math.min(min, c);
        
        
long max = min * n;
        while (min < max) {
        	long mid = (max + min) / 2;
        	long divCount = getDivCount(a, b, c, mid);
            if (divCount < n) {
                min = mid + 1; 
        	} else {
        		max = mid;
        	}
        }
        return (int) min;
    }
	private
 long getDivCount(long a, long b, long c, long num) {  
        return ((num / a) + (num / b) + (num / c)  
                - (num / lcm(a, b))  
                - (num / lcm(b, c))  
                - (num / lcm(a, c))  
                + (num / lcm(a, lcm(b, c))));  
    }
	private long gcd(long a, long b) {  
        if (a == 0)  
            return b
;  
        return gcd(b % a, a);  
    }
    private long lcm(long a, long b) {  
        return ((long) a * (long) b) / gcd(a, b);  
    } 
}
