class Solution {
    public int nthUglyNumber(int n, int a, int b, int c) {
        long ab = ((long)a)*b/gcd(a,b);
        long ac = ((long)a)*c/gcd(a,c);
        long bc = ((long)b)*c/gcd(b,c);
        long abc = ab*c/gcd(ab,c);
        long min = 1;
        long max = 2000000000;
        while (min != max) {
            long mid = (min+max)/2;
            long ans = mid/a+mid/b+mid/c-mid/ab-mid/ac-mid/bc+mid/abc;
            if (ans < n)
                min = mid+1;
            else
                max = mid;
        }
        return (int)min;
    }
    
    public static long gcd(long a, long b) {
		if (a>b)
			return gcd(b,a);
		long t = a%b;
		while (true) {
			if (t==0)
				return b;
			a = b;
			b = t;
			t = a%b;
		}
	}
}
