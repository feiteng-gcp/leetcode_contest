class Solution {
    
    long gcd(long a, long b) {
        if (a % b == 0) {
            return b;
        } else {
            return gcd(b, a%b);
        }
    }
    
    long lcm(long a, long b) {
        return (a*b/gcd(a,b));
    }
    
    long cal(long t, long[] arr, long[] cm, long ad) {
        long a = arr[0], b = arr[1], c = arr[2];
        return t/a+t/b+t/c-(t/cm[0]+t/cm[1]+t/cm[2])+t/ad;
    }
    
    long getR(long a, long b) {
        if (a % b == 0) {
            return a/b;
        } else {
            return a/b+1;
        }
    }
    
    public int nthUglyNumber(int n, int a, int b, int c) {
        long[] arr = new long[3];
        long[] cm = new long[3];
        arr[0] = (long)a;
        arr[1] = (long)b;
        arr[2] = (long)c;
        cm[0] = lcm(arr[0], arr[1]);
        cm[1] = lcm(arr[1], arr[2]);
        cm[2] = lcm(arr[0], arr[2]);
        long ad = lcm(cm[0], arr[2]);
        long num = (long)n;
        Arrays.sort(arr);
        long l = 1, r = num;
        while (l < r) {
            long mid = (l+r)/2;
            if (cal(mid*arr[0], arr, cm, ad) < num) {
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        long cur = 0;
        while ((cur = cal(l*arr[0], arr, cm, ad)) > num) {
            l--;
        }
        System.out.println(cur);
        System.out.println(l);
        long p1 = l, p2 = getR(l*arr[0], arr[1]), p3 = getR(l*arr[0], arr[2]);
        long ug = 0;
        for (; cur<=num; cur++) {
            ug = Math.min(p1*arr[0], Math.min(p2*arr[1], p3*arr[2]));
            if (p1*arr[0] == ug) {
                p1++;
            }
            if (p2*arr[1] == ug) {
                p2++;
            }
            if (p3*arr[2] == ug) {
                p3++;
            }
        }
        return (int)ug;
    }
}
