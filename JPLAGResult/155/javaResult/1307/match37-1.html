class Solution
{
    
long gcd(long a, long b)                 //自定义函数求最大公约数
    {
        if (a % b == 0)                    //终止条件
            return b;                   //b及为最大公约数
        else
            return gcd(b, a % b);
    }

    long lcm(long a, long b)
    {
        return (a * b) / gcd(a, b);
    
}

    long solve(long A, long B, long C, long D)
    {
        long divisibleA = D / A;
        long divisibleB = D / B;
        long divisibleC = D / C;

        
long divisible_A_B = D / (lcm(A, B));
        long divisible_A_C = D / (lcm(A, C));
        long divisible_B_C = D / (lcm(B, C));

        long divisible_A_B_C = D / lcm(lcm(A, B), C);


        
return divisibleA + divisibleB + divisibleC - divisible_A_B - divisible_A_C - divisible_B_C + divisible_A_B_C;
    }

    public int nthUglyNumber(long n, long a, long b, long c)
    {
        
long l = 1, r = 2_000_000_999;
        while (l < r)
        {
            long m = (l + r) / 2;
            long cn = solve(a, b, c, m);
            if (cn >= n)
            {
                r = m;
            }
            else l = m + 1;
        }
        return (int)l;
    }
}
