class Solution {

    long gcd(long a, long b) {
        if (b == 0) return a;
        return gcd(b, a % b);
    }

    long lcm(long a, long b) {
        return a * b / gcd(a, b);
    }

    long lcm(long a, long b, long c) {
        return lcm(lcm(a, b), c);
    }


    public int nthUglyNumber(int n, int a, int b, int c) {
        long low = 1;
        long high = 2L << 40;

        while (low < high) {
            long mid = (low + high) / 2;

            long count = mid / a + (mid / b - mid / lcm(a, b)) +
                    (mid / c - mid / lcm(a, c) - mid / lcm(b, c) + mid / lcm(a, b, c));

            if (count < n) {
                low = mid + 1;
            } else {
                high = mid;
            }

        }
        return (int) low;
    }
}
