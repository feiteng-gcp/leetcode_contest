class Solution {
    long gcd(long a, long b) {
        return a == 0 ? b : 
gcd(b % a, a);
    }
    long lcm(long a, long b) {
        return (a * b) / gcd(a, b);
    }

    long helper(long a, long b, long c, long num) {
        return ((num / a) + (num / b) + (num / c)
                - (num / lcm(a, b))
                - (num / lcm(b, c))
                - (num / lcm(a, c))
                + (num / lcm(a, lcm(b, c))));
    }
    
    
public int nthUglyNumber(int n, int a, int b, int c) {
        long l = 1, h = Integer.MAX_VALUE, m;

        while (l < h) {
            m = l + (h - l) / 2;
            if (helper((long)a, (long)b, (long)c, m) < n)
                l = m + 1;
            else
                h = m;
        }

        return (int)l;
    }
}
