class Solution {
    public List<List<Integer>> minimumAbsDifference(int[] arr) {
        Arrays.sort(arr);
        List<List<Integer>> output = new ArrayList<>();
        int diff = Integer.MAX_VALUE;
        for(int i = 1; i < arr.length; i++) {
            int d = arr[i] - arr[i - 1];
            if(diff > d) {
                diff = d;
                output.clear();
            }
            if(diff == d) {
                List<Integer> c = new ArrayList<>();
                c.add(arr[i - 1]);
                c.add(arr[i]);
                output.add(c);
            }
        }
        return output;
    }
}
