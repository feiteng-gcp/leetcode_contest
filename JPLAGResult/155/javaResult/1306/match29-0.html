class Solution {
    public List<List<Integer>> minimumAbsDifference(int[] a) {
        Arrays.sort(a);
        int n = a.length;
        int minDiff = a[n-1] - a[0];
        for (int i = 1; i < n; i++) minDiff = Math.min(minDiff, a[i] - a[i-1]);
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 1; i < n; i++) {
            if (a[i] - a[i-1] == minDiff) {
                List<Integer> lst = new ArrayList<>();
                lst.add(a[i-1]);
                lst.add(a[i]);
                ans.add(lst);
            }
        }
        return ans;
    }
}
