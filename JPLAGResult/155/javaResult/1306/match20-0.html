class Solution {
    public List<List<Integer>> minimumAbsDifference(int[] arr) {
        Arrays.sort(arr);
        int min = Integer.MAX_VALUE;
        int len = arr.length;
        for (int i = 1; i < len; i++) {
            min = Math.min(min, arr[i] - arr[i - 1]);
        }
        List<List<Integer>> res = new ArrayList<>();
        for (int i = 1; i < len; i++) {
            if (arr[i] - arr[i - 1] == min) {
                List<Integer> cur = new ArrayList<>();
                cur.add(arr[i - 1]);
                cur.add(arr[i]);
                res.add(cur);
            }
        }
        return res;
    }
}
