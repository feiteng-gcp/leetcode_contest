class Solution {
    Map<Integer, List<Integer>> adj;
    int [] parent;
    public int find(int x){
        if(parent[x] != x){
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    public void merge(int x, int y){
        int xp = find(x);
        int yp = find(y);
        if(xp == yp){
            return;
        }
        parent[yp] = xp;
    }
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        adj = new HashMap<>();
        
        char [] sarr = s.toCharArray();
        parent = new int[sarr.length + 1];
        for(int i = 0; i < parent.length; i++){
            parent[i] = i;
        }
        
        for(List<Integer> cur: pairs){
            int x = cur.get(0);
            int y = cur.get(1);
            merge(x, y);
        }
        
        List<Character> [] values = new ArrayList[parent.length];
        int [] pos = new int[values.length];
        for(int i = 0; i < values.length; i++){
            values[i] = new ArrayList<>();
        }
        for(int i = 0; i < sarr.length; i++){
            int p = find(i);
            values[p].add(sarr[i]);
        }
        
        for(int i = 0; i < values.length; i++){
            Collections.sort(values[i]);
        }
        
        StringBuilder res = new StringBuilder();
        for(int i = 0; i < sarr.length; i++){
            int p = find(i);
            int index = pos[p];
            pos[p]++;
            res.append(values[p].get(index));
        }
        return res.toString();
    }
}
