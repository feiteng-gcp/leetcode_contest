class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        int N = s.length();
        DJSet djSet = new DJSet(N);
        
        for(List<Integer> list : pairs){
            djSet.union(list.get(0), list.get(1));
        }
        
        Map<Integer, List<Character>> graph = new HashMap<>();
        
          for (int i = 0; i < N; i++) {
            int head = djSet.root(i);
              
            if(!graph.containsKey(head)){
                graph.put(head, new ArrayList<>());
            }
            graph.get(head).add(s.charAt(i));
          }
        
          for (List<Character> c : graph.values()) {
            Collections.sort(c, Collections.reverseOrder());
          }

          StringBuilder sb = new StringBuilder(N);
        
          for (int i = 0; i < N; i++) {
            List<Character> c = graph.get(djSet.root(i));
            char currMin = c.remove(c.size() - 1);
            sb.append(currMin);
          }
          return sb.toString();

class DJSet {
    public int[] upper;
    public int max;

    public DJSet(int n) {
        upper = new int[n];
        Arrays.fill(upper, -1);
        max = 1;
    }

    public int root(int x) {
        return upper[x] < 0 ? x : (upper[x] = root(upper[x]));
    }

    public boolean equiv(int x, int y) {
        return root(x) == root(y);
    }

    public boolean union(int x, int y) {
        x = root(x);
        y = root(y);
        if (x != y) {
            if (upper[y] < upper[x]) {
                int d = x;
                x = y;
                y = d;
            }
            upper[x] += upper[y];
            upper[y] = x;
            
            max = Math.max(max, -1 * upper[x]);
        }
        return x == y;
    }

    public int count() {
        int ct = 0;
        for (int u : upper)
            if (u < 0)
                ct++;
        return ct;
    }
}
