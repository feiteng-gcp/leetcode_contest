class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        char[] array = s.toCharArray();
        int len = s.length();
        int[] parent = new int[len];
        for (int i = 0; i < len; i++) {
            parent[i] = i;
        }
        int[] rank = new int[len];
        
        for (List<Integer> pair : pairs) {
            union(pair.get(0), pair.get(1), parent, rank);
        }
        
        Map<Integer, List<Character>> map = new HashMap<>();
        
        for (int i = 0; i < len; i++) {
            int group = find(i, parent);
            map.putIfAbsent(group, new ArrayList<>());
            map.get(group).add(array[i]);
        }
        
        Map<Integer, Queue<Character>> map2 = new HashMap<>();
        
        for (Map.Entry<Integer, List<Character>> e : map.entrySet()) {
            Collections.sort(e.getValue());
            map2.put(e.getKey(), new ArrayDeque<>(e.getValue()));
        }
        
        for (int i = 0; i < len; i++) {
            array[i] = map2.get(find(i, parent)).poll();
        }
        
        return new String(array);
    }
    
    private int find(int x, int[] parent) {
        if (x != parent[x]) {
            parent[x] = find(parent[x], parent);
        }
        return parent[x];
    }
    
    private void union(int x, int y, int[] parent, int[] rank) {
        int xPar = find(x, parent);
        int yPar = find(y, parent);
        if (xPar == yPar) {
            return;
        }
        if (rank[xPar] > rank[yPar]) {
            parent[yPar] = xPar;
        } else if (rank[yPar] > rank[xPar]) {
            parent[xPar] = yPar;
        } else {
            parent[yPar] = xPar;
            rank[xPar]++;
        }
    }
}
