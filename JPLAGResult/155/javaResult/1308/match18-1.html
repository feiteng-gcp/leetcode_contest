class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        int n = s.length();
    
        ArrayList<Integer>[] next = new ArrayList[n]; 
        // initializing 
        for (int i = 0; i < n; i++) { 
            next[i] = new ArrayList<Integer>(); 
        
} 
        for (List<Integer> pair : pairs) {
            next[pair.get(0)].add(pair.get(1));
            next[pair.get(1)].add(pair.get(0));
        }
        
        char[] tmpRes = new char[n];
        
        boolean[] mark = new boolean[n];
        for (int i = 0; i < n; ++i)
            if (!mark[i]) {
                List<Character> chrs = new ArrayList<>();
                List<Integer> curr = new ArrayList<>();
                
                Queue<Integer> q = new LinkedList<>();
                q.offer(i);
                mark[i] = true;
                
                while (!q.isEmpty()) {
                    int j = q.poll();
                    curr.add(j);
                    chrs.add(s.charAt(j));
                    for (int k : next[j]) {
                        if (!mark[k]) {
                            q.offer(k);
                            mark[k] = true;
                        }
                    }
                }
                Collections.sort(chrs);
                
Collections.sort(curr);
                for (int ii = 0; ii < chrs.size(); ++ii) {
                    tmpRes[curr.get(ii)] = chrs.get(ii);
                }
            }
        
        StringBuilder res = new StringBuilder();
        for (char c : tmpRes)
            res.append(c);
        return res.toString();
    }
}
