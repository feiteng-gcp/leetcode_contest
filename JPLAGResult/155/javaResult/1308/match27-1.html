class Solution {

    
class UF {
        private int[] parent;

        public UF(int size) {
            parent = new int[size];
            for (int i = 0; i < parent.length; i++) {
                parent[i] = i;
            }
        }

        public int find(int n) {
            if (parent[n] != n) {
                parent[n] = find(parent[n]);
            }
            return parent[n];
        
}

        public void union(int a, int b) {
            int pa = find(a);
            int pb = find(b);
            if (pa != pb) {
                parent[pb] = pa;
            }
        }

        public int getSet() {
            int cnt = 0;
            for (int i = 0; i < parent.length; i++) {
                if (i == parent[i]) {
                    cnt++;
                }
            }
            return cnt;
        }
    }

    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        UF uf = new UF(s.length());
        char[] res = new char[
s.length()];
        for (List<Integer> pair : pairs) {
            uf.union(pair.get(0), pair.get(1));
        }
        Map<Integer, Set<Integer>> groupMembers = new HashMap<>();
        for (int i = 0; i < uf.parent.length; i++) {
            int key = uf.find(i);
            Set<Integer> set = groupMembers.getOrDefault(key, new HashSet<>());
            set.add(i);
            groupMembers.putIfAbsent(key, set);
        }
        for (Set<Integer> set : groupMembers.values()) {
            List<Integer> indexList = new ArrayList<>(set);
            List<Character> characterList = new ArrayList<>(indexList.size());
            for (int i : indexList) {
                characterList.add(s.charAt(i));
            
}
            Collections.sort(indexList);
            Collections.sort(characterList);
            for (int i = 0; i < indexList.size(); i++) {
                res[indexList.get(i)] = characterList.get(i);
            }
        }
        return new String(res);
    }

}
