class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        int n = s.length();
        
int[] par = new int[n];

        for (int i = 0; i < n; i++)
            par[i] = i;

        for (List<Integer> pair : pairs) {
            int a = pair.get(0), b = pair.get(1);
            int x = find(par, a);
            int y = find(par, b);
            par[y] = x;
        
}

        HashMap<Integer, List<Integer>> map = new HashMap<>();

        for (int i = 0; i < n; i++) {
            int root = find(par, i);
            map.putIfAbsent(root, new ArrayList<>());
            map.get(root).add(i);
        }

        char[] ans = new char[s.length()];
        for (int key : map.keySet()) {
            List<Character> chs = new ArrayList<>();
            List<Integer> idxs = map.get(key);
            for (int i : idxs) chs.add(s.charAt(i));
            Collections.sort(idxs);
            Collections.sort(chs);

            for (int i = 0; i < idxs.size(); i++) {
                int idx = idxs.get(i);
                char ch = chs.get(i);
                ans[idx] = ch;
            
}

        }
        return new String(ans);
    }
    private int find(int[] par, int x) {
        while (x != par[x]) {
            par[x] = par[par[x]];
            x = par[x];
        }
        return x;
    }
}
