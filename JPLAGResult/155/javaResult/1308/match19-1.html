class Solution {

    int getP(int p[], int x) {
        if (p[x] == x) {
            return p[x];
        
}
        return p[x] = getP(p, p[x]);
    }
    
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        char ch[] = s.toCharArray();
        Map<Integer, List<Integer>> map = new HashMap<>();
        int p[] = new int[ch.length];
        for (int i = 0; i < ch.length; i++) {
            p[i] = i;
        
}
        for (int i = 0; i < pairs.size(); i++) {
            List<Integer> list = pairs.get(i);
            int a = list.get(0);
            int b = list.get(1);
            int pa = getP(p, a);
            int pb = getP(p, b);
            if (pa == pb) continue;
            if (pa > pb) {
                pa ^= pb;
                pb ^= pa;
                pa ^= pb;
            }
            p[pb] = p[pa];
        }
        
for (int i = 0; i < ch.length; i++) {
            int pi = getP(p, i);
            if (!map.containsKey(pi)) {
                map.put(pi, new ArrayList<>());
            }
            map.get(pi).add(i);
        }
        char res[] = new char[ch.length];
        for (int key : map.keySet()) {
            List<Integer> list = map.get(key);
            List<Integer> tmp = new ArrayList<>(list);
            tmp.sort(Comparator.naturalOrder());
            list.sort(Comparator.comparingInt(o -> ch[o]));
            for (int i = 0; i < list.size(); i++) {
                res[
tmp.get(i)] = ch[list.get(i)];
            }
        }
        return String.valueOf(res);
    }
}
