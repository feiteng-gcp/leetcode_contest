class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        
        int n = s.length();
        int[] roots = new int[n];
        for(int i = 0; i < s.length(); i++)
        {
            roots[i] = i;
        }
        
        for(List<Integer> pair : pairs)
        {
            union(roots, pair.get(0), pair.get(1));
        }
        
        Map<Integer, List<Integer>> idx = new HashMap<Integer, List<Integer>>();
        Map<Integer, List<Character>> value = new HashMap<Integer, List<Character>>();
        
        for(int i = 0; i < roots.length; i++)
        {
            int root = find(roots, i);
            if(!idx.containsKey(root))
            {
                idx.put(root, new ArrayList<Integer>());
                value.put(root, new ArrayList<Character>());
            }
            
            idx.get(root).add(i);
            value.get(root).add(s.charAt(i));
        }
        
        StringBuilder builder = new StringBuilder(s);
        for(Integer key : idx.keySet())
        {
            List<Integer> id = idx.get(key);
            List<Character> ch = value.get(key);
            Collections.sort(ch);
            for(int i = 0; i < id.size(); i++)
            {
                builder.setCharAt(id.get(i), ch.get(i));
            }
        }
        
        return builder.toString();
    }
    
    private int find(int[] roots, int x)
    {
        if(roots[x] != x)
            roots[x] = find(roots, roots[x]);
        
        return roots[x];
    }
    
    private void union(int[] roots, int x, int y)
    {
        if(find(roots, x) != find(roots, y))
        {
            roots[find(roots, x)] = find(roots, y);
        }
    }
}
