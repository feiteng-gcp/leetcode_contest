class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        
        // union-find
        // find connected subcomponents, and then sore each component
        
        if (s == null || pairs == null) return "";
        
        UnionFind uf = new UnionFind(s.length());
        for (List<Integer> pair : pairs) {
            uf.union(pair.get(0), pair.get(1));
        }
        
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            map.putIfAbsent(uf.find(i), new ArrayList<>());
            map.get(uf.find(i)).add(i);
        }
        char[] arr = s.toCharArray();
        for (int key : map.keySet()) {
            sortChars(arr, map.get(key));
        }
        return String.valueOf(arr);
    }
    
    private void sortChars(char[] chars, List<Integer> indices) {
        
        char[] tmp = new char[indices.size()];
        int i = 0;
        for (int index : indices) {
            tmp[i++] = chars[index];
        }
        Arrays.sort(tmp);
        Collections.sort(indices);
        for (int index = 0; index < indices.size(); index++) {
            chars[indices.get(index)] = tmp[index];
        }
        
    }
    
    class UnionFind {
        
        private int[] fathers;
        public UnionFind(int size) {
            this.fathers = new int[size];
            for (int i = 0; i < size; i++) {
                fathers[i] = i;
            }
        }
        
        public void union(int x, int y) {
            
            int xFather = find(x);
            int yFather = find(y);
            if (xFather != yFather) {
                fathers[xFather] = yFather;
            }            
        }
        
        public int find(int x) {
            int father = fathers[x];
            while (father != fathers[father]) {
                father = fathers[father];
            }
            fathers[x] = father;
            return father;
        }
        
    }
}
