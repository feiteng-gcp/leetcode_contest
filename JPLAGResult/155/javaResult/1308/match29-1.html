class Solution {
    
    // d c a b
    
    // d - b
    // c - a
    
    // finding connected components
    
    
static class UnionFind {
        
        int[] parent;
            
        UnionFind(int n) {
            parent = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
            }
        }
        
        void union(int i, int j) {
            int rootI = findRoot(i);
            int rootJ = findRoot(j);
            parent[rootJ] = rootI;
        }
        
        int find(int v) {
            return findRoot(v);    
        }
        
        private int findRoot(int v) {
            while (parent[v] != v) {
                parent[v] = parent[parent[v]];
                v = parent[v];
            }
            return v;
        }
        
    }
    
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        
UnionFind uf = new UnionFind(s.length());
        for (List<Integer> pair : pairs) {
            uf.union(pair.get(0), pair.get(1));
        }
        Map<Integer, Set<Integer>> groups = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            groups.computeIfAbsent(uf.find(i), k -> new HashSet<>())
                .add(i);
        }
        char[] arr = s.toCharArray();
        for (Set<Integer> group : groups.values()) {
            if (group.size() > 1) {
                sortGroup(arr, group);
            }
        }
        return new String(arr);    
    
}
    
    private void sortGroup(char[] arr, Set<Integer> group) {
//        System.out.println(String.format("arr: %s, group: %s", new String(arr), group));
        char[] subSeq = new char[group.size()];
//        System.out.println("subSeq: " + new String(subSeq));
        int[] indices = new int[group.size()];
        int i = 0;
        for (int arrI : group) {
            subSeq[i] = arr[arrI];
            indices[i] = arrI;
            i++;
        }
        Arrays.sort(indices);
        Arrays.sort(subSeq);
//        System.out.println("sorted subSeq: " + new String(subSeq));
        int sortedSeqI = 0;
        for (int arrI : indices) {
            arr[arrI] = subSeq[sortedSeqI++];
        }
    }
}
