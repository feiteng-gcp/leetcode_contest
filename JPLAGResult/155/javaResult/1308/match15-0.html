class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        UF uf = new UF(s.length());
        for(List<Integer> p : pairs) {
            uf.union(p.get(0), p.get(1));
        }
        
        char[] c = s.toCharArray();
        Map<Integer, List<Integer>> map1 = new HashMap();
        Map<Integer, List<Character>> map2 = new HashMap();
        for(int i=0; i<s.length(); i++) {
            int pi = uf.find(i);
            if(!map1.containsKey(pi)) map1.put(pi, new ArrayList());
            if(!map2.containsKey(pi)) map2.put(pi, new ArrayList());
            
            map1.get(pi).add(i);
            map2.get(pi).add(c[i]);
        }
        
        for(int key : map1.keySet()) {
            if(map1.get(key).size()==1) continue;
            List<Character> lc = map2.get(key);
            Collections.sort(lc);
            int i=0;
            for(int j : map1.get(key)) {
                c[j] = lc.get(i++);
            }
        }
        
        return new String(c);
        
    }
    
    class UF {
        int[] parent;
        
        UF(int n) {
            parent = new int[n];
            for(int i=0; i<n; i++) parent[i] = i;
        }
        
        public int find(int x) {
            if(parent[x]!=x) parent[x] = find(parent[x]);
            return parent[x];
        }
        
        public void union(int x, int y) {
            int px = find(x), py = find(y);
            if(px!=py) {
                parent[py] = px;
            }
        }
    }
}
