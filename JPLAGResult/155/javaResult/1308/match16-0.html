class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        int n = s.length();
        for (int i = 0; i < n; i++)
            map.put(i, new ArrayList<>());
        for (List<Integer> p : pairs) {
            int a = p.get(0);
            int b = p.get(1);
            map.get(a).add(b);
            map.get(b).add(a);
        }

        boolean[] seen = new boolean[n];
        char[] c = s.toCharArray();
        for (int i = 0; i < n; i++) {
            if (!seen[i]) {
                List<Integer> comp = new ArrayList<>();
                dfs(comp, map, seen, i);
                System.out.println(comp);
                PriorityQueue<Character> pq = new PriorityQueue<>();
                for (int j = 0; j < comp.size(); j++) {
                    pq.add(c[comp.get(j)]);
                }
                Collections.sort(comp);
                for (int j = 0; j < comp.size(); j++) {
                    c[comp.get(j)] = pq.poll();
                }
            }
        }

        return new String(c);
    }
    
    void dfs(List<Integer> list, Map<Integer, List<Integer>> map, boolean[] seen, int k) {
        seen[k] = true;
        list.add(k);
        for (Integer nei : map.get(k)) {
            if (!seen[nei]) {
                dfs(list, map, seen, nei);
            }
        }
    }
}
