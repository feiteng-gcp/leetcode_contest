class Solution {
    int[] label;
    Set<Integer> set = new HashSet<>();
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        int n = s.length();
        // if (n >= 100000 && n <= pairs.size()){
        //     char[] chars = s.toCharArray();
        //     Arrays.sort(chars);
        //     return String.valueOf(chars);
        // }
        label = 
new int[n];
        for (int i = 0; i < n; i++){
            label[i] = i;
        }
        //System.out.println(n + " "  + pairs.size());
        
        for (List<Integer> list : pairs){
            union(
list.get(0), list.get(1));
            set.add(list.get(0));
            set.add(list.get(1));
        }
        
        
        Map<Integer, PriorityQueue<Character>> map = new HashMap<>();
        for (int i = 0; i < n; i++){
            if (!set.contains(i))
                continue;
            int idx = find(i);
            if (!map.containsKey(idx)){
                map.put(idx, new PriorityQueue<>());
            }
            
map.get(idx).add(s.charAt(i));
        }
        
        //System.out.println(n + " " + set.size());
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++){
            if (!set.contains(i)){
                sb.append(s.charAt(i));
                continue;
            }
            int idx = find(i);
            PriorityQueue<Character> pq = 
map.get(idx);
            sb.append(pq.remove());
        }
        return sb.toString();
    }
    
    private int find(int i){
        while(label[i] != i)
            i = label[i];
        
return i;
    }
    
    private void union(int i, int j){
        int pi = find(i);
        int pj = find(j);
        label[Math.max(pi, pj)] = Math.min(pi, pj);
    }
}
