class Solution {
    int[] p;
    
    int root(int a) {
        return p[a] == a?a: (p[a] = root(p[a]));
    }
    void union(int a, int b) {
        int ra = root(a);
        int rb = root(b);
        if(ra != rb) {
            p[ra] = rb;
        }
    
}
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        int n = s.length();
        p = new int[n];
        for(int i=0; i<n; ++i) p[i] = i;
        
        for(List<Integer> pair: pairs) {
            union(pair.get(0), pair.get(1));
        }
        
        
Map<Integer, List<Integer>> indMap = new HashMap<>();
        for(int i=0; i<n; ++i) {
            int ri = root(i);
            if(!indMap.containsKey(ri)) {
                indMap.put(ri, new ArrayList<>());
            }
            indMap.get(ri).add(i);
        }
        
        StringBuilder ret = new StringBuilder(s);
        for(int k: indMap.keySet()) {
            List<Integer> inds = indMap.get(k);
            char[] chars = new char[inds.size()];
            for(int i=0, j=0; i<inds.size(); ++i) {
                chars[j++] = s.charAt(inds.get(i));
            }
            Arrays.sort(chars);
            for(int i=0, j=0; i<inds.size(); ++i) {
                
ret.setCharAt(inds.get(j++), chars[i]);
            }
        }
        return ret.toString();
        
    }
}
