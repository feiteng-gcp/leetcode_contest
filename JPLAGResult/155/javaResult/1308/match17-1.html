class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {

        int n = s.length();
        List<Integer>[] adj = new List[n];

        for (int i = 0; i <n; i++) {
            adj[i] = new ArrayList<>();
        }

        for (int i = 0; i < pairs.size(); i++) {
            int x = pairs.get(i).get(0);
            int y = 
pairs.get(i).get(1);

            adj[x].add(y);
            adj[y].add(x);
        }

        boolean[] vst = new boolean[n];

        char[] chars = s.toCharArray();
        char[] ans = new char[n];

        for (int i = 0; i < n; i++) {
            if (vst[i]) continue;

            List<Integer> ids = new ArrayList<>();
            dfs(i, adj, vst, ids);

            List<Integer> original = new ArrayList<>(ids);
            original.sort((a, b) -> a - b);
            ids.sort((a, b) -> {
                int t = (int) s.charAt(a) - (int) s.charAt(b);
                return t;
            });

            
for (int j = 0; j < ids.size(); j++) {
                ans[original.get(j)] = chars[ids.get(j)];
            }
        
}

        return new String(ans);


    }

    private void dfs(int x, List<Integer>[] adj, boolean[] vst, List<Integer> ids) {
        if (vst[x]) return;

        vst[x] = true;
        ids.add(x);

        for (int y : adj[x]) {
            dfs(y, adj, vst, ids);

        }
    }
}
