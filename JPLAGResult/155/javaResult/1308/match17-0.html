class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        char[] arr = s.toCharArray();
        List<Integer>[] adj = new ArrayList[arr.length];
        for (int i = 0; i < arr.length; i++) adj[i] = new ArrayList<>();
        for (List<Integer> p : pairs) {
            adj[p.get(0)].add(p.get(1));
            adj[p.get(1)].add(p.get(0));
        }
        char[] res = new char[arr.length];
        boolean[] visited = new boolean[arr.length];
        for (int i = 0; i < arr.length; i++) {
            if (!visited[i]) {
                List<Integer> stack = new ArrayList<>();
                dfs(i, visited, stack, adj);
                List<Integer> l = new ArrayList<>(stack);
                Collections.sort(stack, new Comparator<Integer>() {
                    public int compare(Integer i, Integer j) {
                        return arr[i] - arr[j];
                    }
                });
                Collections.sort(l);
                for (int j = 0; j < l.size(); j++) {
                    res[l.get(j)] = arr[stack.get(j)];
                }
            }
        }
        return new String(res);
    }
    private void dfs(int v, boolean[] visited, List<Integer> stack, List<Integer>[] adj) {
        if (visited[v]) return;
        visited[v] = true;
        for (int w : adj[v]) {
            dfs(w, visited, stack, adj);
        }
        stack.add(v);
    }
}
