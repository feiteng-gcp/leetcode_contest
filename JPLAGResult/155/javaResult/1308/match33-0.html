class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        int[] p = new int[s.length()];
        char[] ca = s.toCharArray();
        for (int i = 0; i < p.length; i++) {
            p[i] = i;
        }
        for (List<Integer> list : pairs) {
            int p1 = find(list.get(0), p);
            int p2 = find(list.get(1), p);
            p[p2] = p1;
        }
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < p.length; i++) {
            int p1 = find(i, p);
            List<Integer> list = map.getOrDefault(p1, new ArrayList<>());
            list.add(i);
            map.put(p1, list);
        }
        for (int k : map.keySet()) {
            List<Integer> list = map.get(k);
            List<Character> chars = new ArrayList<>();
            for (int i : list) {
                chars.add(ca[i]);
            }
            Collections.sort(list);
            Collections.sort(chars);
            // System.out.println(Arrays.toString(list.toArray(new Integer[0])));
            for (int i = 0; i < list.size(); i++) {
                ca[list.get(i)] = chars.get(i);
            }
        }
        return new String(ca);
    }
    
    private int find(int x, int[] p) {
        while (x != p[x]) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
}
