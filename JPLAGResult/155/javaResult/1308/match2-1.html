class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        if (pairs.size() == 0) return s;
        int n = s.length();
        boolean[] vis = new boolean[n];
        int[] mark = new int[n];
        HashMap<Integer, List<Integer>> g = new HashMap<>();
        for (List<Integer> pair : pairs) {
            if (pair.size() < 2) continue;
            int st = pair.get(0), e = pair.get(1);
            if (!g.containsKey(st)) g.put(st, new ArrayList<>());
            if (!g.containsKey(e)) g.put(e, new ArrayList<>());
            g.get(st).add(e);
            g.get(e).add(st);
        }
        for (int i = 0, index = 0; i < n; i++) {
            if (!vis[i]) dfs(index++, i, g, vis, mark);
        }
        HashMap<Integer, List<Character>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int m = mark[i];
            if (!map.containsKey(m)) map.put(m, new LinkedList<>());
            map.get(m).add(s.charAt(i));
        }
        for (int k : map.keySet()) {
            Collections.sort(map.get(k));
        }
        StringBuilder ret = new StringBuilder();
        for (int i = 0; i < n; i++) {
            List<Character> list = map.get(mark[i]);
            //System.out.println(list);
            char c = list.get(0);
            list.remove(0);
            ret.append(c);
            //System.out.println(ret);
        }
        return ret.toString();
    }
    void dfs(int m, int s, HashMap<Integer, List<Integer>> g, boolean[] vis, int[] mark) {
        if (vis[s]) return;
        vis[s] = true;
        mark[s] = m;
        if (g.containsKey(s)) {
            for (int nxt : g.get(s)) {

                dfs(m, nxt, g, vis, mark);
            }
        }
        
    }
}
