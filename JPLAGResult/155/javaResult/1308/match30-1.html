class Solution {
    int[] p;
    
public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        p = new int[s.length()];
        for (int i = 0; i < p.length; i++) {
            p[i] = -1;
        }
        for (List<Integer> pair : pairs) {
            union(pair.get(0), pair.get(1)); 
        }
        
        for (int i = 0; i < p.length; i++) {
            p[i] = find(i);
        }
        Map<Integer, List<Integer>> pToIdx = new HashMap<>(); 
        Map<Integer, List<Character>> pToChars = new HashMap<>(); 
        
        for (int i = 0; i < p.length; i++) {
            pToIdx.putIfAbsent(p[i], new ArrayList<>());
            pToIdx.get(p[i]).add(i);
            pToChars.putIfAbsent(p[i], new ArrayList<>());
            pToChars.get(p[i]).add(s.charAt(i));
        }
        
        
        //sort chars 
        for (int p : pToIdx.keySet()) {
            if (p == -1) continue;
            
Collections.sort(pToChars.get(p));
        }
        
        StringBuilder sb = new StringBuilder(s);
        
        for (int p : pToIdx.keySet()) {
            List<Integer> idxList = pToIdx.get(p);
            List<Character> charList = pToChars.get(p);
            
            for (int j = 0; j < idxList.size(); j++) {
                sb.setCharAt(idxList.get(j), charList.get(j));
            }
            
        }
        return sb.toString();
    }
    
    private int find(int a) {
        if (p[a] == -1) p[a] = a;
        int pa = p[a];
        
if (pa != a) {
            p[a] = find(pa);
        }
        return p[a];
    }
    
    public void union(int a, int b) {
        int pa = find(a);
        int pb = find(b);
        if (pa != pb) {
            p[pa] = pb;
        }
    }
}
