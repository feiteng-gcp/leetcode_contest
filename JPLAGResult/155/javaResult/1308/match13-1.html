class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        UF uf = new UF(s.length());
        
for (List<Integer> pair : pairs) {
            uf.connect(pair.get(0), pair.get(1));
        }
        char[] sorted = new char[s.length()];
        Map<Integer, List<Integer>> indexesByRoot = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            indexesByRoot.computeIfAbsent(uf.root(i), k -> new ArrayList<>()).add(i);
        }
        for (List<Integer> indexes : indexesByRoot.values()) {
            List<Character> chars = new ArrayList<>();
            for (int index : indexes) {
                chars.add(
s.charAt(index));
            }
            Collections.sort(chars);
            for (int i = 0; i < indexes.size(); i++) {
                sorted[indexes.get(i)] = chars.get(i);
            }
        }
        return new String(sorted);
    }

    
private class UF {

        private int[] root;

        public UF(int n) {
            this.root = new int[n];
            for (int i = 0; i < n; i++) {
                root[i] = i;
            }
        }

        public int root(int i) {
            if (root[i] == i) {
                return i;
            } else {
                root[i] = root[root[i]];
                return root(root[i]);
            }
        
}

        public void connect(int i, int j) {
            root[root(i)] = root(j);
        }

        public boolean isConnected(int i, int j) {
            return root(i) == root(j);
        }
    }
}
