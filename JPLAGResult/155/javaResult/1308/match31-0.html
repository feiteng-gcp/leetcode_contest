class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        int n = s.length();
        UnionFind uf = new UnionFind(n);
        int components = n;
        for (List<Integer> p: pairs) {
            if (uf.union(p.get(0), p.get(1))) --components;
        }
        Map<Integer, StringBuilder> sbs = new HashMap(components);
        for (int i = 0; i < n; ++i) {
            int cc = uf.find(i);
            sbs.computeIfAbsent(cc, x -> new StringBuilder()).append(s.charAt(i));
        }
        Map<Integer, String> sbsSorted = new HashMap(components);
        for (Map.Entry<Integer, StringBuilder> entry: sbs.entrySet()) {
            sbsSorted.put(entry.getKey(), stringSort(entry.getValue().toString()));
        }
        StringBuilder ans = new StringBuilder();
        int[] ccInd = new int[n + 1];
        for (int i = 0; i < n; ++i) {
            int cc = uf.find(i);
            ans.append(sbsSorted.get(cc).charAt(ccInd[cc]++));
        }
        
        return ans.toString();
    }
    
    static String stringSort(String text) {
        char[] str = text.toCharArray();
        Arrays.sort(str);
        return new String(str);
    }
    
    class UnionFind {
        Map<Integer, Integer> parentMap;
        
        public UnionFind(int n) {
            this.parentMap = new HashMap<>();
            for (int i = 0; i < n; ++i) {
                this.add(i);
            }
        }
        
        public void add(int x) {
            this.parentMap.put(x, x);
        }
        
        public boolean union(int x1, int x2) {
            int p1 = find(x1),
                p2 = find(x2);
            if (p1 != p2) {
                parentMap.put(p2, p1);
                return true;
            }
            return false;
        }
        
        public int find(int x) {
            int query = x;
            while (parentMap.get(x) != x) {
                x = parentMap.get(x);
            }
            parentMap.put(query, x); // Note: turned out to be a vital optimization
            return x;
        }
    }
}
