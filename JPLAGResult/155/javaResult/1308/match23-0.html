class Solution {
    private int[] s;
    private int find(int x) {
        if (s[x] == x) {
            return x;
        }
        s[x] = find(s[x]);
        return s[x];
    }
    
    private void merge(int u, int v) {
        int su = find(u);
        int sv = find(v);
        if (su < sv) {
            s[sv] = su;
        } else if (sv < su) {
            s[su] = sv;
        }
    }
    
    public String smallestStringWithSwaps(String str, List<List<Integer>> pairs) {
        int len = str.length();
        s = new int[len];
        int[] cnt = new int[len];
        
        for (int i = 0; i < len; i++) {
            s[i] = i;
        }
        
        for (int i = 0; i < pairs.size(); i++) {
            List<Integer> pair = pairs.get(i);
            merge(pair.get(0).intValue(), pair.get(1).intValue());
        }
        
        List<List<Character>> tbl = new ArrayList<>();
        for (int i = 0; i < len; i++) {
            tbl.add(new ArrayList<>());
        }
        for (int i = 0; i < len; i++) {
            s[i] = find(i);
            List<Character> list = tbl.get(s[i]);
            list.add(str.charAt(i));
        }
        
        for (int i = 0; i < tbl.size(); i++) {
            List<Character> list = tbl.get(i);
            Collections.sort(list);
        }
        
        StringBuilder ans = new StringBuilder("");
        for (int i = 0; i < len; i++) {
            int idx = cnt[s[i]]++;
            Character ch = tbl.get(s[i]).get(idx);
            ans.append(ch.charValue());
        }
        return ans.toString();
    }
}
