class Solution {
    public String smallestStringWithSwaps(String s, List<List<Integer>> pairs) {
        char[] ch = s.toCharArray();
        int n = ch.length;
        int[] root = new int[n];
        for (int i=1; i<n; ++i)
            root[i] = i;
        
        for (List<Integer> pair: pairs) {
            int x = pair.get(0), y = pair.get(1);
            int rx = find (root, x), ry = find (root, y);
            if (rx != ry)
                root[rx] = ry;
        }
        
        Map<Integer, List<Integer>> groups = new HashMap<>();
        for (int i=0; i<n; ++i) {
            int r = find (root, i);
            groups.computeIfAbsent(r, (x) -> new ArrayList<>());
            groups.get(r).add(i);
        }
        
        for (List<Integer> group: groups.values()) {
            int size = group.size();
            char[] alphas = new char[size];
            for (int i=0; i<size; ++i)
                alphas[i] = ch[group.get(i)];
            Arrays.sort(alphas);
            for (int i=0; i<size; ++i)
                ch[group.get(i)] = alphas[i];
        }
        
        
        
        return new String(ch);
    }
    
    private int find (int[] root, int x) {
        if (root[x] != x) {
            root[x] = find (root, root[x]);
        }
        return root[x];
    }
}
