class Solution {
    bool toposort(const map<int, set<int>>& edges, vector<int>* ret) {
        int n = edges.size();
        map<int, int> degree;
        for (const auto& item : edges) degree[item.first] = 0;
        for (const auto& edge : edges) {
            for (int nxt : edge.second) {
                degree[nxt]++;
            }
        }
        queue<int> q;
        for (const auto& item : degree) {
            if (item.second == 0) q.push(item.first);
        }
        ret->clear();
        while (!q.empty()) {
            int cur = q.front(); q.pop();
            ret->push_back(cur);
            auto it = edges.find(cur);
            assert(it != edges.end());
            for (int nxt : it->second) {
                degree[nxt]--;
                if (degree[nxt] == 0) q.push(nxt);
            }
        }
        // for (const auto& edge : edges) {
        //     cout << "point " << edge.first << ": ";
        //     for (int nxt : edge.second) cout << nxt << ' ';
        //     cout << endl;
        // }
        // cout << "ret: ";
        // for (int val : *ret) cout << val << ' ';
        // cout << endl;
        return ret->size() == n;
        
        
    }
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        map<int, vector<int>> g2n;
        for (int i = 0; i < n; i++) {
            if (group[i] == -1) {
                group[i] = m;
                m++;
            }
            g2n[group[i]].push_back(i);
        }
        map<int, vector<pair<int, int>>> g2r;
        map<int,set<int>> edges;
        for (int i = 0; i < m; i++) edges[i] = {};
        for (int i = 0; i < n; i++) {
            int gi = group[i];
            for (int bef : beforeItems[i]) {
                int gb = group[bef];
                if (gb == gi) {
                    g2r[gb].emplace_back(bef, i);
                    continue;
                }
                edges[gb].insert(gi);
            }
        }
        vector<int> g_order;
        if (!toposort(edges, &g_order)) return {};
        vector<int> ret;
        for (int i : g_order) {
            edges.clear();
            for (int n : g2n[i]) {
                edges[n] = set<int>();
            }
            for (const auto& item : g2r[i]) {
                edges[item.first].insert(item.second);
            }
            vector<int> tmp;
            if (!toposort(edges, &tmp)) return {};
            for (int val : tmp) ret.push_back(val);
        }
        return ret;
        
        
    }
};
