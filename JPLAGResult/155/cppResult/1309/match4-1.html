class Solution {
    vector<int> sortg(int n,vector<vector<int>> g){
        queue<int> qq;
        vector<int> in(n,0);
        for(int i=0;i<n;i++){
            for(int p:g[i]){
                in[p]++;
            }
        }
        

        for(int i=0;i<n;i++) if(in[i]==0) qq.push(i);
        vector<int> ans;
        while(!qq.empty()){
            int p=qq.front();
            qq.pop();
            ans.push_back(p);
            for(int pp:g[p]){
                in[pp]--;
                if(in[pp]==0) qq.push(pp);
            }
        }
        return ans;
    }
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<int> newg=group;
        int newm=m;
        for(int i=0;i<n;i++){
            if(group[i]==-1){
                newg[i]=newm++;
            }
        }
        vector<int> idx_in_group(n);
        vector<int> gcnt(newm,0);
        for(int i=0;i<n;i++){
            idx_in_group[i]=gcnt[newg[i]]++;
        }
        
        
        vector<vector<int>> idx_in_group_reverse(newm);
        for(int i=0;i<newm;i++){
            idx_in_group_reverse[i]=vector<int>(gcnt[i]);
        }
        for(int i=0;i<n;i++){
            idx_in_group_reverse[newg[i]][idx_in_group[i]]=i;
        }
        
        vector<vector<int>> g(newm);
        vector<vector<vector<int>>> group_g(newm);
        for(int i=0;i<newm;i++){
            group_g[i]=vector<vector<int>>(gcnt[i]);
        }

        for(int i=0;i<n;i++){
            int g1=newg[i];
            for(int j:beforeItems[i]){
                int g2=newg[j];
                if(g1==g2){
                    int p1=idx_in_group[i];
                    int p2=idx_in_group[j];
                    group_g[g1][p2].push_back(p1);
                }else{
                    g[g2].push_back(g1);
                }
            }
        }
        
        auto a1=sortg(newm,g);
        if(a1.size()!=newm) return vector<int>();
        vector<int> ans;
        for(int g_idx:a1){
            auto a2=sortg(gcnt[g_idx],group_g[g_idx]);
            if(a2.size()!=gcnt[g_idx]) return vector<int>();
            for(int j:a2){
                ans.push_back(idx_in_group_reverse[g_idx][j]);
            }
        }
        return ans;
    }
};
