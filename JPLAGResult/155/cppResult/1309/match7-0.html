class Solution {
public:
  vector<int> top_sort(int n, const vector<vector<int>> &adj) {
    vector<int> ind(n);
    queue<int> q;
    for (int i = 0; i < n; ++i) {
      for (int v : adj[i]) {
        ++ind[v];
      }
    }
    for (int i = 0; i < n; ++i) {
      if (!ind[i]) {
        q.push(i);
      }
    }
    vector<int> res;
    while (!q.empty()) {
      auto s = q.front();
      q.pop();
      res.push_back(s);
      for (int v : adj[s]) {
        if (!--ind[v]) {
          q.push(v);
        }
      }
    }
    if (res.size() != n) {
      res.clear();
    }
    return res;
  }
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
      for (int i = 0; i < n; ++i) {
        if (group[i] == -1) {
          group[i] = m++;
        }
      }
      vector<vector<int>> adj(m);
      vector<vector<int>> group_to_v(m);
      for (int i = 0; i < n; ++i) {
        group_to_v[group[i]].push_back(i);
        for (const auto &v : beforeItems[i]) {
          if (group[v] != group[i]) {
            adj[group[v]].push_back(group[i]);
          }
        }
      }
      auto groups = top_sort(m, adj);
      vector<int> res;
      for (int g : groups) {
        int nn = group_to_v[g].size();
        vector<vector<int>> adj2(nn);
        unordered_map<int, int> rev;
        for (int i = 0; i < nn; ++i) {
          rev[group_to_v[g][i]] = i;
        }
        for (int i = 0; i < nn; ++i) {
          int x = group_to_v[g][i];
          for (int v : beforeItems[x]) {
            if (group[v] == g) {
              adj2[rev[v]].push_back(i);
            }
          }
        }
        auto internal = top_sort(nn, adj2);
        for (int idx : internal) {
          res.push_back(group_to_v[g][idx]);
        }
      }
      // vector<bool> vis(m);
      //for (int i = 0; i < res.size(); ++i) {
      //  if (group[res[i]] == -1) {
      //    continue;
      //  }
      //  if (vis[group[res[i]]]) {
      //    res.clear();
      //    break;
      //  }
      //  vis[group[res[i]]] = true;
      //  int j;
      //  for (j = i; j < res.size() && group[res[i]] == group[res[j]]; ++j);
      //  i = j - 1;
      //}
      //reverse(res.begin(), res.end());
      if (res.size() < n) {
        res.clear();
      }
      return res;
    }
};
