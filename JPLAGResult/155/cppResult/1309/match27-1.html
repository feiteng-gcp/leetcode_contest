class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& be) {
        map<int, vector<int>> mp;
        int count = 0;
        for(int i = 0; i < n; i++)
        {
            if(group[i] == -1)
            {
                group[i] = m + count;
                count++;
            }
            mp[ group[i] ].push_back( i );
        }

        int totGroups = m + count;

        vector<int> indeg(totGroups);

        vector< vector<int> > adj(totGroups);

        vector<int> indegI(n);
        vector< vector<int>> adjI(n);

        for(int i = 0; i < n; i++)
        {
            for(int k = 0; k < be[i].size(); k++)
            {
                int gi = group[i], itemk = be[i][k];

                int gk = group[ itemk ];

                if(gi == gk)
                {
                    indegI[i]++;
                    adjI[itemk].push_back(i);
                }
                else
                {
                    indeg[gi]++;
                    adj[gk].push_back( gi );
                }
            }
        }

        queue<int> q;

        for(map<int, vector<int>>::iterator it = mp.begin(); it != mp.end(); it++)
        {
            int groupi = it->first;

            vector<int> vv = mp[groupi];

            for(int j = 0; j < vv.size() ; j++)
            {
                int item = vv[j];
                if(indegI[item] == 0)
                    q.push(item);
            }

            vector<int> a;
            
            if(q.size() == 0)
            {
                cout <<"-";
                a.clear();
                return a;

            }
                

            while(!q.empty())
            {
                int gi = q.front();

                q.pop();

                for(int i = 0; i < adjI[gi].size(); i++)
                {
                    int gk = adjI[gi][i];
                    indegI[gk]--;

                    if(indegI[gk] == 0)
                        q.push(gk);
                }
                a.push_back( gi );
            }            

            mp[groupi] = a;
        }



        for(int i = 0 ; i < totGroups; i++)
        {
            if(indeg[i] == 0)
                q.push(i);
        }

        vector<int> ans;

        if(q.size() == 0)
        {
            cout <<"---";
            ans.clear();
            return ans;

        }

        while(!q.empty())
        {
            int gi = q.front();

            q.pop();

            for(int i = 0; i < adj[gi].size(); i++)
            {
                int gk = adj[gi][i];
                indeg[gk]--;

                if(indeg[gk] == 0)
                    q.push(gk);
            }

            vector<int> vv = mp[gi];
            for(int i = 0; i < vv.size(); i++)
            {
                ans.push_back(vv[i]);
            }
        }
        if(ans.size() < n)
            ans.clear();
        return ans;
    }
};
