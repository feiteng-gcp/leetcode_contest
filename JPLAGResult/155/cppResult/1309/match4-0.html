class Solution {
    template<typename Nodes>
    vector<int> topo(const vector<Nodes>& adj){
        int n=adj.size();
        vector<int> in(n);
        for(int i=0;i<n;i++)
            for(int j:adj[i])
                in[j]++;
        queue<int> q;
        for(int i=0;i<n;i++)
            if(in[i]==0)q.push(i);
        vector<int> ret;
        while(!q.empty()){
            int u=q.front();
            ret.push_back(u);
            q.pop();
            for(int v:adj[u]){
                in[v]--;
                if(in[v]==0)q.push(v);
            }
        }
        if(ret.size()<n)ret.clear();
        return ret;
    }
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<vector<int>> gps(m);
        for(int i=0;i<n;i++)
            if(group[i]==-1){
                gps.push_back({i});
                group[i]=m++;
            }
            else gps[group[i]].push_back(i);
        vector<unordered_set<int>> gadj(m);
        for(int i=0;i<n;i++){
            int g=group[i];
            for(int j:beforeItems[i]){
                int h=group[j];
                if(h!=g) gadj[h].insert(g);
            }
        }
        vector<int> gtopo=topo(gadj);
        if(gtopo.empty())return {};
        vector<unordered_set<int>> bis;
        for(int i=0;i<n;i++)
            bis.emplace_back(beforeItems[i].begin(),beforeItems[i].end());
        vector<int> ans;
        for(int g:gtopo){
            if(gps[g].empty())continue;
            vector<vector<int>> adj(gps[g].size());
            for(int i=0;i<gps[g].size();i++)
                for(int j=0;j<gps[g].size();j++){
                    if(i==j)continue;
                    int u=gps[g][i],v=gps[g][j];
                    if(bis[v].count(u))
                        adj[i].push_back(j);
                }
            vector<int> tp=topo(adj);
            if(tp.empty())return {};
            for(int i:tp)ans.push_back(gps[g][i]);
        }
        return ans;
    }
};
