class Solution {
    struct Group {
        vector<int> items;
        unordered_map<int, int> in_degrees;
        unordered_map<int, vector<int>> out_items;
    };
    
    vector<int> TopologicalSort(vector<int> nodes, unordered_map<int, int> in_degrees, unordered_map<int, vector<int>> out_nodes) {
        queue<int> q;
        unordered_set<int> vis;
        vector<int> res;
        for (int i : nodes) if (in_degrees[i] == 0) q.push(i), vis.insert(i);
        while (!q.empty()) {
            int i = q.front(); q.pop();
            res.push_back(i);
            for (int j : out_nodes[i]) {
                if (--in_degrees[j] == 0)
                    q.push(j);
            }
        }
        if (res.size() != nodes.size()) return {};
        return res;
    }
    
public:
    vector<int> sortItems(int n, int m, vector<int>& group_id, vector<vector<int>>& beforeItems) {
        vector<Group> groups;
        unordered_map<int, int> in_degrees;
        unordered_map<int, vector<int>> out_groups;
        for (int i=0; i<m; i++) groups.push_back(Group());
        for (int i=0; i<n; i++) {
            if (group_id[i] == -1) {
                group_id[i] = m ++;
                groups.push_back(Group());
            }
            groups[group_id[i]].items.push_back(i);
        }
        for (int i=0; i<n; i++) {
            for (int j : beforeItems[i]) {
                if (group_id[i] == group_id[j]) {
                    groups[group_id[i]].in_degrees[i] ++;
                    groups[group_id[i]].out_items[j].push_back(i);
                } else {
                    in_degrees[group_id[i]] ++;
                    out_groups[group_id[j]].push_back(group_id[i]);
                }
            }
        }
        
        vector<int> group_indices, items;
        for (int i=0; i<groups.size(); i++) group_indices.push_back(i);
        vector<int> group_order = TopologicalSort(group_indices, in_degrees, out_groups);
        if (group_order.empty()) return {};
        for (int i : group_order) {
            vector<int> group_items = TopologicalSort(groups[i].items, groups[i].in_degrees, groups[i].out_items);
            if (group_items.size() != groups[i].items.size()) return {};
            for (int j : group_items) items.push_back(j);
        }
        return items;
    }
};
