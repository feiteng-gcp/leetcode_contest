class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<vector<int>> p(n + m);
        vector<vector<int>> graph(n + m);
        vector<vector<int>> graph2(n);
        
        for (int i = 0; i < n; i++)
            if (group[i] != -1)
                p[group[i] + n].push_back(i);
        
        for (int i = 0; i < n; i++)
            for (int x : beforeItems[i])
                if (group[x] == -1) {
                    if (group[i] == -1)
                        graph[x].push_back(i);
                    else
                        graph[x].push_back(group[i] + n);
                } else {
                    if (group[i] == -1)
                        graph[group[x] + n].push_back(i);
                    else {
                        if (group[x] != group[i])
                            graph[group[x] + n].push_back(group[i] + n);
                        else {
                            graph2[x].push_back(i);
                        }
                    }
                }
        
        vector<int> indeg(n + m, 0);
        
        for (int i = 0; i < n + m; i++)
            for (int j = 0; j < graph[i].size(); j++)
                indeg[graph[i][j]]++;
        
        
        vector<int> indeg2(n, 0);
        for (int i = 0; i < n; i++)
            for (int j = 0; j < graph2[i].size(); j++)
                indeg2[graph2[i][j]]++;


        queue<int> q;

        for (int i = 0; i < n + m; i++)
            if (indeg[i] == 0)
                q.push(i);

        vector<int> tmp;
        while (!q.empty()) {
            int sta = q.front();
            tmp.push_back(q.front());
            
            q.pop();

            for (int i = 0; i < graph[sta].size(); i++) {
                indeg[graph[sta][i]]--;
                if (indeg[graph[sta][i]] == 0) {
                    q.push(graph[sta][i]);
                }
            }
        }
        
        if (tmp.size() != n + m)
            return {};
        
        vector<int> ans;
        for (int i = 0; i < n + m; i++) {
            if (tmp[i] < n && group[tmp[i]] == -1)
                ans.push_back(tmp[i]);
            else {
                int cnt = 0;
                for (int j: p[tmp[i]]) {
                    cnt++;
                    if (indeg2[j] == 0)
                        q.push(j);
                }
                   
                   vector<int> t;
                   while (!q.empty()) {
                       int sta = q.front();
                       t.push_back(q.front());
                       
                       q.pop();

                       for (int i = 0; i < graph2[sta].size(); i++) {
                           indeg2[graph2[sta][i]]--;
                           if (indeg2[graph2[sta][i]] == 0) {
                               q.push(graph2[sta][i]);
                           }
                       }
                   }
                   
                   if (t.size() != cnt)
                       return {};
                   
                   for (int j: t)
                       ans.push_back(j);
            }
                
        }
        
        return ans;
    }
};
