class Solution {
public:
    vector<int> adj[100010];
    vector<int> adjByGroup[100010];
    vector<int> groupElement[100010];
    vector<int> topSort;
    int visited[100010];
    bool cycle = 0;
    void dfs(int x) {
        visited[x] = 1;
        for(auto it : adjByGroup[x]) {
            if(visited[it]==0)  dfs(it);
            if(visited[it]==1) cycle = 1;
        }
        
        topSort.push_back(x);
        visited[x] = 2;
    }
    int color[100010];
    vector<int> topSortByElement;
    
    void dfsAgain(int x) {
        
        color[x] = 1;
        for(auto it : adj[x]) {
            if(color[it]==0)  dfsAgain(it);
            if(color[it]==1) cycle = 1;
        }
        
        topSortByElement.push_back(x);
        color[x] = 2;
        
    }
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        int ind = m;
        for(int i = 0;i<group.size();i++) {
            if(group[i] == - 1) group[i] = ind++;
            groupElement[group[i]].push_back(i);
        }
        for(int i = 0;i<beforeItems.size();i++) {
            for(int j = 0;j<beforeItems[i].size();j++) {
                int x = beforeItems[i][j];
                
                if(group[i]==group[x])
                    adj[x].push_back(i);
                if(group[i]!=group[x]) {
                    adjByGroup[group[x]].push_back(group[i]);
                }
            }
        }
        for(int i = 0;i<ind;i++) {
            if(visited[i]==0) dfs(i);
        }
        if(cycle == true) return {};
        
        reverse(topSort.begin(),topSort.end());
        vector<int> answer;
        for(auto it : topSort) {
             for(int jt : groupElement[it]) {
                if(color[jt]==0) dfsAgain(jt);
            }
            if(cycle) return {};
            for(int j = (int)topSortByElement.size()-1;j>=0;j--) {
                answer.push_back(topSortByElement[j]);
            }
            topSortByElement.clear();
            for(int jt : groupElement[it]) {
                color[jt] = 0;
            }
        }
        if(cycle) return {};
        
        return answer;
        
    }
};
