const int N = 3e4 + 10;
vector<int> a[N], b[N];
bool flag[N], flag2[N];
vector<int> topo, topo2;
vector<int> A[N];
bool instack[N];

bool DFS(int u) {
    if (flag[u]) {
        if (instack[u]) return false;
        return true;
    }
    flag[u] = true;
    instack[u] = true;
    for (auto& v : a[u]) {
        if (!DFS(v)) return false;
    }
    topo.push_back(u);
    instack[u] = false;
    return true;
}

bool DFS2(int u) {
    if (flag2[u]) {
        if (instack[u]) return false;
        return true;
    }
    flag2[u] = true;
    instack[u] = true;
    for (auto& v : b[u]) {
        if (!DFS2(v)) return false;
    }
    topo2.push_back(u);
    instack[u] = false;
    return true;
}

class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        for (int i = 0; i < N; ++i) A[i].clear();
        for (int i = 0; i < n; ++i) {
            if (group[i] < 0) group[i] = m++;
            A[group[i]].push_back(i);
        }
        for (int i = 0; i < m; ++i) a[i].clear();
        for (int i = 0; i < n; ++i) b[i].clear();
        for (int i = 0; i < n; ++i) {
            for (auto& j : beforeItems[i]) {
                int x = group[j], y = group[i];
                if (x != y) {
                    a[x].push_back(y);
                } else {
                    b[j].push_back(i);
                }
            }
        }
        fill(flag, flag + m, false);
        fill(instack, instack + m, false);
        topo.clear();
        for (int i = 0; i < m; ++i) {
            if (!DFS(i)) return {};
        }
        fill(flag2, flag2 + n, false);
        fill(instack, instack + n, false);
        topo2.clear();
        for (int i = 0; i < n; ++i) {
            if (!DFS2(i)) return {};
        }
        reverse(topo.begin(), topo.end());
        reverse(topo2.begin(), topo2.end());
        vector<int> idx(n);
        for (int i = 0; i < n; ++i) idx[topo2[i]] = i;
        
        vector<int> ret;
        for (auto& x : topo) {
            sort(A[x].begin(), A[x].end(),
                 [&](int u, int v) {
                     return idx[u] < idx[v];
                 });
            for (auto& it : A[x]) ret.push_back(it);
        }
        return ret;
    }
};
