class Solution {
public:
    
    vector<int> topo(vector<vector<int>> &G) {
        int n = G.size();
        vector<int> ind(n, 0);
        for (int i=0; i<n; i++) {
            for (int v: G[i]) {
                ind[v]++;
            }
        }
        queue<int> que;
        for (int i=0; i<n; i++) {
            if (ind[i] == 0) {
                que.push(i);
            }
        }
        int m = 0;
        vector<int> ans;
        while (!que.empty()) {
            int u = que.front(); que.pop();
            ans.push_back(u);
            m++;
            for (int v: G[u]) {
                ind[v]--;
                if (ind[v] == 0) {
                    que.push(v);
                }
            }
        }
        return ans;
    }
    
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<int> ref(n);
        int alloc = m;
        for (int i=0; i<n; i++) {
            if (group[i] == -1) {
                ref[i] = alloc++;
            } else {
                ref[i] = group[i];
            }
            // printf("ref[%d] = %d\n", i, ref[i]);
        }
        
        vector<vector<int>> G0(alloc);
        vector<vector<int>> G(n);
        for (int i=0; i<n; i++) {
            for (int x: beforeItems[i]) {
                if (ref[x] != ref[i]) {
                    G0[ref[x]].push_back(ref[i]);
                    // printf("G0 add (%d, %d)\n", ref[x], ref[i]);
                }
                G[x].push_back(i);
            }
        }
        vector<int> gt = topo(G0);
        // printf("gt. size %d\n", gt.size());
        if (gt.size() < alloc)
            return {};
        vector<int> innt = topo(G);
        // printf("innt size %d\n", innt.size());
        if (innt.size() < n)
            return {};
        vector<vector<int>> gps(alloc);
        for (int i=0; i<n; i++) {
            int x=  innt[i];
            gps[ref[x]].push_back(x);
        }
        vector<int> ans;
        for (int x: gt) {
            for (int v: gps[x]) {
                ans.push_back(v);
            }
        }
        return ans;
    }
};
