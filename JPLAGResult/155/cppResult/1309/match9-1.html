class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        for (int & x : group)
            if (x == -1)
                x = m++;
        
        vector<vector<int>> dag(m), ingroup_dag(n);
        vector<int> deg(m), ingroup_deg(n);
        for (int i = 0; i < n; ++i) {
            for (int j : beforeItems[i]) {
                if (group[i] == group[j]) {
                    ingroup_dag[j].push_back(i);
                    ingroup_deg[i] += 1;
                } else {
                    dag[group[j]].push_back(group[i]);
                    deg[group[i]] += 1;
                }
            }
        }
        
        queue<int> zeros;
        for (int i = 0; i < m; ++i) {
            if (deg[i] == 0) {
                zeros.push(i);
            }
        }
        
        vector<int> groupOrd;
        while (!zeros.empty()) {
            int igroup = zeros.front();
            zeros.pop();
            
            for (int j : dag[igroup]) {
                deg[j] -= 1;
                if (deg[j] == 0) {
                    zeros.push(j);
                }
            }
            
            groupOrd.push_back(igroup);
        }
        
        if (groupOrd.size() != m)
            return {};
        
        vector<vector<int>> groupMembers(m);
        for (int i = 0; i < n; ++i) {
            groupMembers[group[i]].push_back(i);
        }
        
        vector<int> result;
        for (int igroup : groupOrd) {
            vector<int> memberOrd;
            queue<int> zero_members;
            
            for (int x : groupMembers[igroup]) {
                if (ingroup_deg[x] == 0)
                    zero_members.push(x);
            }
            

            while (!zero_members.empty()) {
                int i = zero_members.front();
                zero_members.pop();
                
                for (int j : ingroup_dag[i]) {
                    ingroup_deg[j] -= 1;
                    if (ingroup_deg[j] == 0)
                        zero_members.push(j);
                }
                
                memberOrd.push_back(i);
            }
            
            if (memberOrd.size() != groupMembers[igroup].size())
                return {};
            for (int x : memberOrd)
                result.push_back(x);
        }
        
        return result;
    }
};
