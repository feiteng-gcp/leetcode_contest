class Solution {
   public:
    vector<int> sortItems(int n, int m, vector<int>& group,
                          vector<vector<int>>& beforeItems) {
        vector<vector<int>> G(n + n + n);
        vector<int> deg(n + n + n);
        vector<vector<int>> D(m);
        for (int i = 0; i < n; i++) {
            if (group[i] != -1) {
                G[group[i] + n].push_back(i);
                G[i].push_back(group[i] + n + n);
                deg[i]++;
                deg[group[i] + n + n]++;
            }
        }
        for (int i = 0; i < n; i++) {
            auto& v = beforeItems[i];
            for (int j : v) {
                // j show before i
                if (group[i] == -1) {
                    if (group[j] == -1) {
                        G[j].push_back(i);
                        deg[i]++;
                    } else {
                        G[group[j] + n + n].push_back(i);
                        deg[i]++;
                    }
                } else {
                    int u = group[i];
                    if (group[j] == -1) {
                        G[j].push_back(u + n);
                        deg[u + n]++;
                    } else {
                        if (group[j] != group[i]) {
                            G[group[j] + n + n].push_back(u + n);
                            deg[u + n]++;
                        } else {
                            G[j].push_back(i);
                            deg[i]++;
                        }
                    }
                }
            }
        }
        queue<int> Q;
        vector<int> ord;
        for (int i = 0; i < n + n + n; i++) {
            if (deg[i] == 0) Q.push(i);
        }
        while (!Q.empty()) {
            int u = Q.front();
            Q.pop();
            if (u < n) ord.push_back(u);
            for (int v : G[u]) {
                deg[v]--;
                if (deg[v] == 0) {
                    Q.push(v);
                }
            }
        }
        if (ord.size() != n) return {};
        for (int i = 0; i < n; i++) {
            if (group[ord[i]] != -1) D[group[ord[i]]].push_back(ord[i]);
        }
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            int now = group[ord[i]];
            if (now == -1) {
                ans.push_back(ord[i]);
            } else {
                for (int x : D[now]) ans.push_back(x);
                D[now] = {};
            }
        }
        return ans;
    }
};
