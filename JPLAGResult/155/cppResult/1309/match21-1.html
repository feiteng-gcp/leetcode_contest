int pos[30004];

bool cmp(int a,int b)
{
    return pos[a]<pos[b];
}

class Solution {
public:
    
    vector<int> g[30004];
    vector<int> gg[30004];

    vector<int> gr[30004];
    vector<int> top;
    int to_g[30004];
    int used[30004];
    int cnt;
    int cycle;
    
    
    void dfs(int u)
    {
        used[u]=1;
        for(int i=0;i<g[u].size();i++)
        {
            if(used[g[u][i]]==2) continue;
            if(used[g[u][i]]==1) 
            {
                cycle=1;
            }
            else
                dfs(g[u][i]);
            
        }
        top.push_back(u);
        used[u]=2;
    }
    
    
    void dfs2(int u)
    {
        used[u]=1;
        for(int i=0;i<gg[u].size();i++)
        {
            if(used[gg[u][i]]==2) continue;
            if(used[gg[u][i]]==1) 
            {
                cycle=1;
            }
            else
                dfs2(gg[u][i]);
            
        }
        top.push_back(u);
        used[u]=2;
    }
    
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        cnt=m;

        for(int i=0;i<group.size();i++)
        {
            if(group[i]==-1)
            {
                gr[cnt].push_back(i);
                to_g[i]=cnt++;
            }
            else
            {
                gr[group[i]].push_back(i);
                to_g[i]=group[i];
            }
        }
        
        
        for(int i=0;i<beforeItems.size();i++)
            for(int j=0;j<beforeItems[i].size();j++)
            {
                int cur=beforeItems[i][j];
                g[i].push_back(cur);
                if(to_g[i]!=to_g[cur])
                    gg[to_g[i]].push_back(to_g[cur]);
            }
        
        for(int i=0;i<n;i++)
            if(!used[i])
            {
                dfs(i);
                if(cycle)
                {
                    top.clear();
                    return top;
                }
            }
        for(int i=0;i<top.size();i++)
            pos[top[i]]=i;
        for(int i=0;i<cnt;i++)
            sort(gr[i].begin(),gr[i].end(),&cmp);
        
        top.clear();
        memset(used,0,sizeof(used));
        
        for(int i=0;i<cnt;i++)
            if(!used[i])
            {
                dfs2(i);
                if(cycle)
                {
                    top.clear();
                    return top;
                }
            }
        vector<int> ans;
        for(int i=0;i<top.size();i++)
            for(int j=0;j<gr[top[i]].size();j++)
                ans.push_back(gr[top[i]][j]);
        return ans;

        
    }
};
