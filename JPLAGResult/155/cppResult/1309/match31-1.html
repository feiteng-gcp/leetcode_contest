class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<vector<int>> group_members(m);
        for (int i = 0; i < n; ++i) {
            if (group[i] == -1) {
                group[i] = group_members.size();
                group_members.emplace_back(vector<int>{i});
            } else {
                group_members[group[i]].emplace_back(i);
            }
        }
        
        unordered_map<int, unordered_set<int>> group_edges;
        for (int i = 0; i < n; ++i) {
            for (int j : beforeItems[i]) {
                if (group[i] != group[j]) {
                    group_edges[group[j]].emplace(group[i]);
                }
            }
        }
        vector<int> group_states(group_members.size(), 0);
        vector<int> group_order;
        for (int i = 0; i < group_members.size(); ++i) {
            if (group_states[i]) {
                continue;
            }
            if (!dfs(group_edges, i, group_states, group_order)) {
                return {};
            }
        }
        
        vector<int> results;
        results.reserve(n);
        vector<int> member_states(n, 0);
        for (int group_index : group_order) {
            const auto &members = group_members[group_index];
            if (members.empty()) {
                continue;
            } else if (members.size() == 1) {
                results.emplace_back(members[0]);
                continue;
            }
            unordered_map<int, unordered_set<int>> adjLists;
            for (int member : members) {
                for (int before : beforeItems[member]) {
                    if (group[before] == group_index) {
                        adjLists[before].emplace(member);
                    }
                }
            }
            for (int member : members) {
                if (member_states[member]) {
                    continue;
                }
                if (!dfs(adjLists, member, member_states, results)) {
                    return {};
                }
            }
        }
        reverse(results.begin(), results.end());
        return results;
    }
private:
    bool dfs(const unordered_map<int, unordered_set<int>> &adjLists, int i, vector<int> &states,
            vector<int> &results) const {
        if (states[i] == 2) {
            return true;
        } else if (states[i] == 1) {
            return false;
        }
        states[i] = 1;
        if (adjLists.count(i)) {
            for (int j : adjLists.at(i)) {
                if (!dfs(adjLists, j, states, results)) {
                    return false;
                }
            }
        }
        states[i] = 2;
        results.emplace_back(i);
        return true;
    }
};
