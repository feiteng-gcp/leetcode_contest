class Solution {
public:
    int check[60005];
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<int> indeg(n, 0), edge[n], no, revg[m];
        map<int,int> over;
        int groupnum = m;
        for(int i = 0; i < n; i++){
            if(group[i] == -1){
                group[i] = groupnum++;
                over[groupnum-1] = i;
            }else{
                revg[group[i]].push_back(i);
            }
        }
        vector<int> indegbg(groupnum, 0), edgebg[groupnum];
        for(int i = 0; i < n; i++){
            memset(check, 0, sizeof(check));
            for(int j : beforeItems[i]){
                if(check[group[j]]) continue;
                check[group[j]] = true;

                if(group[j] == group[i]){
                    indeg[i]++;
                    edge[j].push_back(i);
                }else{
                    edgebg[group[j]].push_back(group[i]);
                    indegbg[group[i]]++;
                }
            }
        }
        vector<int> chunk[m];
        for(int i = 0; i < m; i++){
            if(revg[i].size() == 0) continue;
            queue<int> topo;
            for(int j : revg[i]){
                if(indeg[j] == 0)   topo.push(j);
            }
            if(topo.empty())    return no;
            while(!topo.empty()){
                int x = topo.front();   topo.pop();
                chunk[i].push_back(x);
                for(int j : edge[x]){
                    indeg[j]--;
                    if(indeg[j] == 0)   topo.push(j);
                }
            }
            if(chunk[i].size() != revg[i].size())   return no;
        }
        
        vector<int> resind;
        queue<int> topo;
        for(int i = 0; i < groupnum; i++){
            if(i < m && revg[i].size() == 0)    continue;

            if(indegbg[i] == 0) topo.push(i);
        }
        
        while(!topo.empty()){
            int x = topo.front();   topo.pop();
            resind.push_back(x);
            for(int j : edgebg[x]){
                indegbg[j]--;
                if(indegbg[j] == 0) topo.push(j);
            }
        }
        
        vector<int> ret;
        for(int i : resind){
            if(i >= m)  ret.push_back(over[i]);
            else        for(int j : chunk[i])   ret.push_back(j);
        }
        
         if(ret.size() == n) return ret;

        return no;
    }
};
