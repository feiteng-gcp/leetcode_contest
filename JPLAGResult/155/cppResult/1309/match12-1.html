class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        itemInDeg = vector<int>(30005, 0);
        groupInDeg = vector<int>(60005, 0);
        for (int i = 0; i < n; ++i) {
            if (group[i] == -1)
                group[i] = i + n;
            groupItems[group[i]].push_back(i);
        }
        for (int i = 0; i < n; ++i) {
            int s = beforeItems[i].size();

            for (int j = 0; j < s; ++j) {
                int u = beforeItems[i][j];
                if (group[u] == group[i]) {
                    itemEdges[u].push_back(i);
                    ++itemInDeg[i];
                }
                else {
                    if (!groupEdges[group[u]].count(group[i])) {
                        groupEdges[group[u]].insert(group[i]);
                        ++groupInDeg[group[i]];

                    }
                }
            }
        }
        vector<int> ans;
        queue<int> groupQ;
        for (int i = 0; i < n + n; ++i) {
            if (groupInDeg[i] == 0 && groupItems[i].size() > 0)
                groupQ.push(i);
        }
        while (!groupQ.empty()) {
            int groupId = groupQ.front();
            groupQ.pop();
            queue<int> itemQ;
            for (auto u: groupItems[groupId]) {
                if (itemInDeg[u] == 0)
                    itemQ.push(u);
            }
            while (!itemQ.empty()) {
                int itemId = itemQ.front();
                itemQ.pop();
                ans.push_back(itemId);
                for (auto u: itemEdges[itemId]) {
                    --itemInDeg[u];
                    if (itemInDeg[u] == 0)
                        itemQ.push(u);
                }
            }
            for (auto u: groupEdges[groupId]) {
                --groupInDeg[u];
                if (groupInDeg[u] == 0)
                    groupQ.push(u);
            }
        }
        if (ans.size() == n)
            return ans;
        return vector<int>();
    }
    vector<int> itemEdges[30005];
    set<int> groupEdges[60005];
    vector<int> groupItems[60005];
    vector<int> itemInDeg;
    vector<int> groupInDeg;
};
