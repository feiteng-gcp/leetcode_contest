class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<int> ret;
        vector<vector<int>> depby(n, vector<int>()), gdepby(m, vector<int>()), glist(m, vector<int>()), gfreelist(m, vector<int>());
        queue<int> freelist;
        vector<int> depcnt(n, 0), depocnt(n, 0), gdepcnt(m, 0);
        queue<int> nextg;
        vector<bool> gvis(m, false);
        for (int i = 0; i < n; i++) {
            depcnt[i] = beforeItems[i].size();
            if (group[i] >= 0) glist[group[i]].push_back(i);
            for (auto &&j: beforeItems[i]) {
                depby[j].push_back(i);
                if (group[i] != group[j]) {
                    if (group[i] >= 0) {
                        if (group[j] >= 0) gdepby[group[j]].push_back(group[i]);
                        gdepcnt[group[i]]++;
                    }
                    depocnt[i]++;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            if (depocnt[i]) continue;
            if (group[i] >= 0) {
                gfreelist[group[i]].push_back(i);
                if (gfreelist[group[i]].size() == glist[group[i]].size()) nextg.push(group[i]);
            } else freelist.push(i);
        }
        // for (int i = 0; i < m; i++) {
        //     if (!gdepcnt[i]) nextg.push(group[i]);
        // }
        while (n) {
            if (!freelist.empty()) {
                int tmp = freelist.front();
                freelist.pop();
                for (auto &&i: depby[tmp]) {
                    depcnt[i]--;
                    if (!--depocnt[i]) {
                        if (group[i] >= 0) {
                            gfreelist[group[i]].push_back(i);
                            if (gfreelist[group[i]].size() == glist[group[i]].size()) nextg.push(group[i]);
                        } else freelist.push(i);
                    }
                }
                ret.push_back(tmp);
                n--;
            } else if (!nextg.empty()) {
                int curg = nextg.front();
                // if (gvis[curg]) continue;
                // gvis[curg] = true;
                nextg.pop();
                int curgl = glist[curg].size();
                queue<int> gnext;
                for (auto &&i: glist[curg]) {
                    if (!depcnt[i]) gnext.push(i);
                }
                while (!gnext.empty()) {
                    int tmp = gnext.front();
                    gnext.pop();
                    for (auto &&i: depby[tmp]) {
                        depcnt[i]--;
                        if (group[i] != curg) {
                            if (!--depocnt[i]) {
                                if (group[i] >= 0) {
                                    gfreelist[group[i]].push_back(i);
                                    if (gfreelist[group[i]].size() == glist[group[i]].size()) nextg.push(group[i]);
                                } else freelist.push(i);
                            }
                        } else {
                            if (!depcnt[i]) gnext.push(i);
                        }
                    }
                    ret.push_back(tmp);
                    n--;
                    curgl--;
                }
                if (curgl) {
                    // cout << curg << "E1" << endl;
                    return {};
                }
                // for (auto &&i: gdepby[curg]) {
                //     if (!--gdepcnt[i]) nextg.push(group[i]);
                // }
            } else {
                // cout << "E2" << endl;
                return {};
            }
        }
        return ret;
    }
};
