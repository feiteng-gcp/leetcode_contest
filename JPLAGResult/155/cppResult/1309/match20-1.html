template <typename Container>
std::vector<int> topo_sort(const std::vector<Container>& graph) {
  static_assert(std::is_integral_v<typename Container::value_type>);
  int n = (int)graph.size();
  std::vector<int> in(n, 0);
  for (int u = 0; u < n; ++u) {
    for (int v : graph[u]) {
      ++in[v];
    }
  }
  std::queue<int> q;
  for (int u = 0; u < n; ++u) {
    if (in[u] == 0) {
      q.push(u);
    }
  }
  std::vector<int> sorted;
  sorted.reserve(n);
  while (!q.empty()) {
    int u = q.front();
    q.pop();
    sorted.push_back(u);
    for (int v : graph[u]) {
      if (--in[v] == 0) {
        q.push(v);
      }
    }
  }
  return sorted;
}


class Solution {
 public:
  vector<int> sortItems(int n, int m, vector<int>& group,
                        vector<vector<int>>& beforeItems) {
    vector<vector<int>> groups(m);
    for (int i = 0; i < n; ++i) {
      if (group[i] == -1) {
        group[i] = m++;
        groups.emplace_back();
      }
      groups[group[i]].push_back(i);
    }
    for (auto& g : groups) {
      sort(g.begin(), g.end());
    }
    vector<vector<int>> graph(m);
    vector<vector<vector<int>>> sub(m);
    for (int i = 0; i < m; ++i) {
      sub[i].resize(groups[i].size());
    }
    for (int v = 0; v < n; ++v) {
      for (int u : beforeItems[v]) {
        if (group[u] == group[v]) {
          auto& gu = groups[group[u]];
          sub[group[u]][lower_bound(gu.begin(), gu.end(), u) - gu.begin()]
              .push_back(lower_bound(gu.begin(), gu.end(), v) - gu.begin());
        } else {
          graph[group[u]].push_back(group[v]);
        }
      }
    }
    auto sorted = topo_sort(graph);
    if (sorted.size() != m) {
      return {};
    }
    vector<int> ans;
    for (int i : sorted) {
      auto subsorted = topo_sort(sub[i]);
      if (subsorted.size() != sub[i].size()) {
        return {};
      }
      for (int j : subsorted) {
        ans.push_back(groups[i][j]);
      }
    }
    return ans;
  }
};
