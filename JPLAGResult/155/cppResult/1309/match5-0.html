class Solution {
public:
    int total;
    
    vector<int> compute_dag(const vector<int>& node, const vector<vector<int>>& edges) {
        vector<int> ans;
        vector<int> count(total, 0);
        for (int i = 0; i < node.size(); ++i) {
            for (int j = 0; j < edges[node[i]].size(); ++j) {
                count[edges[node[i]][j]]++;
            }
        }
        for (int i = 0; i < node.size(); ++i) {
            if (count[node[i]] == 0) {
                ans.push_back(node[i]);
            }
        }
        int head = 0;
        while (head < ans.size()) {
            const int x = ans[head++];
            for (int j = 0; j < edges[x].size(); ++j) {
                if (--count[edges[x][j]] == 0) {
                    ans.push_back(edges[x][j]);
                }
            }
        }
        if (ans.size() != node.size()) ans.clear();
        return ans;
    }
    
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        for (int i = 0; i < n; ++i) {
            if (group[i] == -1) {
                group[i] = m++;
            }
        }
        total = max(n, m);
        vector<vector<int>> internal_edges(n);
        vector<vector<int>> external_edges(m);
        vector<vector<int>> group_id(m);
        vector<vector<int>> group_order(m);
        vector<int> order;
        vector<int> ans;
        for (int i = 0; i < n; ++i) {
            group_id[group[i]].push_back(i);
        }
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < beforeItems[i].size(); ++j) {
                const int y = beforeItems[i][j];
                if (group[i] == group[y]) internal_edges[y].push_back(i);
                else external_edges[group[y]].push_back(group[i]);
            }
        }
        for (int i = 0; i < m; ++i) {
            sort(external_edges[i].begin(), external_edges[i].end());
            external_edges[i].erase(unique(external_edges[i].begin(), external_edges[i].end() ), external_edges[i].end());
        }
        for (int i = 0; i < m; ++i) {
            group_order[i] = compute_dag(group_id[i], internal_edges);
            if (group_order[i].size() != group_id[i].size()) return ans;
        }
        vector<int> index(m);
        for (int i = 0; i < m; ++i) index[i] = i;
        order = compute_dag(index, external_edges);
        if (order.size() != m) return ans;
        for (int i = 0; i < order.size(); ++i) {
            for (int j = 0; j < group_order[order[i]].size(); ++j) {
                ans.push_back(group_order[order[i]][j]);
            }
        }
        return ans;
    }
};
