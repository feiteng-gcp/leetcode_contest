class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) 
    {
        vector<unordered_set<int>>groups(n);
        vector<unordered_set<int>>next(n);
        vector<unordered_set<int>>before(n);
        
        int k = n-1;
        
        for (int i=0; i<n; i++)
        {
            if (group[i]==-1)
            {
                groups[k].insert(i);
                group[i] = k;
                k--;
            }
            else
                groups[group[i]].insert(i);
        }
        
        
        
        for (int i=0; i<n; i++)
            for (int j: beforeItems[i])
            {
                int groupId = group[i];                   
                if (groups[groupId].find(j)!=groups[groupId].end())
                {
                    next[j].insert(i);
                    before[i].insert(j);
                }                    
            }
        
        
        vector<vector<int>>eachGroup(n);
        for (int i=0; i<n; i++)
        {
            if (groups[i].size()==0)
                continue;
            if (groups[i].size()==1)
            {
                eachGroup[i] = {*(groups[i].begin())};
                continue;
            }
            
            cout<<i<<endl;
            eachGroup[i] = helper1(groups[i], before, next);
            if (eachGroup[i].size()==0)
                return {};
        }
        
        cout<<"OK"<<endl;
        
        vector<unordered_set<int>>next2(n);
        vector<unordered_set<int>>before2(n);
        
        for (int i=0; i<n; i++)
            for (int j: beforeItems[i])
            {
                int groupId = group[i];                
                if (groups[groupId].find(j)==groups[groupId].end())
                {
                    next2[group[j]].insert(group[i]);
                    before2[group[i]].insert(group[j]);
                }                    
            }
        
        unordered_set<int>nodes;
        for (int i=0; i<n; i++)
            nodes.insert(i);
        vector<int>groupOrder = helper1(nodes, before2, next2);
        
        if (groupOrder.size()==0)
            return {};
        
        vector<int>rets;
        for (auto groupIdx : groupOrder)
        {
            for (auto member: eachGroup[groupIdx])
                rets.push_back(member);
        }
        
        return rets;
    }
    
    vector<int>helper1(unordered_set<int>& nodes, vector<unordered_set<int>>&before, vector<unordered_set<int>>&after)
    {
        vector<int>rets;
        queue<int>q;
        for (auto node : nodes)
        {
            if (before[node].size()==0)
            {
                q.push(node);
            }

        }
        
        
        while (!q.empty())
        {
            int cur = q.front();
            q.pop();
            rets.push_back(cur);
            
            for (auto next: after[cur] )
            {
                before[next].erase(cur);
                if (before[next].size()==0)
                    q.push(next);
            }
        }
        
        if (rets.size()==nodes.size())
            return rets;
        else
            return {};
    }
    
    
    
};
