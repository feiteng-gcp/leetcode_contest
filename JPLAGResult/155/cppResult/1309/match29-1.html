int d[100000];
vector<int> r[100000];

class Solution {
public:

    vector<int> order(vector<int> p, vector<pair<int, int> > &e) {
        //for(auto u: p) printf("%d\n", u);
        //for(auto iter: e) printf("%d %d\n", iter.first, iter.second);
        //puts("");
        for(auto u: p) d[u] = 0, r[u].clear();

        for(auto iter: e) {
            d[iter.second] += 1;
            r[iter.first].push_back(iter.second);

        }

        vector<int> ans;

        queue<int> zero;
        for(auto u: p) {
            if(d[u] == 0) {
                zero.push(u);
            }
        }

        while(zero.size() > 0) {
            int u = zero.front();
            zero.pop();
            ans.push_back(u);
            for(auto v: r[u]) {
                d[v] -= 1;
                if(d[v] == 0) {
                    zero.push(v);
                }
            }
        }

        if(ans.size() == p.size()) {
            return ans;
        }
        else {
            return vector<int>();
        }
    }
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        for(int i = 0; i < n; i++) {
            if(group[i] == -1) {
                group[i] = m;
                m += 1;
            }
        }
        vector<vector<int> > g(m);
        vector<pair<int, int> > e;
        for(int i = 0; i < (int)beforeItems.size(); i++) {
            int u = group[i];
            g[u].push_back(i);
            for(auto x: beforeItems[i]) {
                int v = group[x];
                if(u != v) {
                    e.push_back(make_pair(v, u));
                }
            }
        }

        vector<int> p(m);
        for(int i = 0; i < m; i++) p[i] = i;
        vector<int> gorder = order(p, e);

        //for(auto u: gorder) printf("%d\n", u);

        vector<int> ans;
        if(gorder.size() == 0) return ans;

        for(int i = 0; i < m; i++) {
            e.clear();
            p = g[gorder[i]];
            for(auto u: p) {
                for(auto x: beforeItems[u]) {
                    if(group[x] == gorder[i]) {
                        e.push_back(make_pair(x, u));
                    }
                }
            }
            vector<int> porder = order(p, e);
            //printf("%d\n", porder.size());
            if(p.size() > 0 && porder.size() == 0) return vector<int>();
            for(auto u: porder) {
                ans.push_back(u);
            }
        }
        return ans;
    }
};
