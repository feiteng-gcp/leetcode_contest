class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<vector<int>> post_dig(n), post_group(m), group_num(m);
        for (int i = 0; i < n; ++i) {
            if (group[i] == -1) {
                group[i] = m++;
                post_group.emplace_back();
                group_num.emplace_back();
            }
            group_num[group[i]].push_back(i);
        }
        vector<int> indeg_dig(n), indeg_group(m);
        for (int i = 0; i < n; ++i) {
            for (int p : beforeItems[i])
                if (group[i] == group[p]) {
                    indeg_dig[i]++;
                    post_dig[p].push_back(i);
                } else {
                    indeg_group[group[i]]++;
                    post_group[group[p]].push_back(group[i]);
                }
        }
        vector<int> result;
        
        queue<int> q;
        for (int i = 0; i < m; ++i)
            if (indeg_group[i] == 0)
                q.push(i);
        
        while (!q.empty()) {
            int g = q.front(); q.pop();
            queue<int> inner_q;
            for (int x : group_num[g])
                if (indeg_dig[x] == 0)
                    inner_q.push(x);
            int prev_idx = result.size();
            while (!inner_q.empty()) {
                int x = inner_q.front(); inner_q.pop();
                result.push_back(x);
                for (int y : post_dig[x])
                    if (--indeg_dig[y] == 0)
                        inner_q.push(y);
            }
            if (result.size() - prev_idx != group_num[g].size())
                return {};
            for (int h : post_group[g])
                if (--indeg_group[h] == 0)
                    q.push(h);
        }
        return result.size() == n ? result : vector<int>();
    }
};
