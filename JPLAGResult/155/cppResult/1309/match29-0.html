class Solution {
private:
    vector<int> ans;
    int ind[30005];
    vector<vector<int>> g;
    vector<vector<int>> g2;
    vector<int> topo(vector<vector<int>> &g, vector<int> &gi) {
        memset(ind, 0, sizeof (ind));
        for (auto u : gi) {
            for (auto v : g[u]) {
                ind[v]++;
            }
        }
        vector<int> ans;
        queue<int> q;
        for (auto u : gi) {
            if (ind[u] == 0) {
                q.push(u);
                ans.push_back(u);
            }
        }
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (auto v : g[u]) {
                ind[v]--;
                if (ind[v] == 0) {
                    ans.push_back(v);
                    q.push(v);
                }
            }
        }
        if (ans.size() == gi.size()) 
            return (ans);
        return (vector<int>());
    }
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        vector<int> ans;
        g.resize(2*n);
        g2.resize(2*n);
        int t = m;
        vector<vector<int>> gg(2*n);
        for (int i = 0; i < n; i++) {
            if (group[i] == -1) {
                group[i] = t++;
            }
            gg[group[i]].push_back(i);
        }
        
        for (int i = 0; i < n; i++) {
            for (auto j : beforeItems[i]) {
                if (group[j] != group[i]) {
                    g2[group[j]].push_back(group[i]);
                } else {
                    g[j].push_back(i);
                }
            }
        }
        for (int i = 0; i < 2*n; i++) {
            if (gg[i].empty())
                continue;
            vector<int> tmp = topo(g, gg[i]);
            if (tmp.empty()) {
                return (ans);
            }
            gg[i] = tmp;
        }
        vector<int> gn;
        for (int i = 0; i < 2*n; i++) {
            if (gg[i].empty())
                continue;
            gn.push_back(i);
        }
        for (int i = 0; i < 2*n; i++) {
            sort(g2[i].begin(), g2[i].end());
            g2[i].erase(unique(g2[i].begin(), g2[i].end()), g2[i].end());
        }
        vector<int> tmp = topo(g2, gn);
        if (tmp.empty()) {
            return (ans);
        }
        for (auto v : tmp) {
            ans.insert(ans.end(), gg[v].begin(), gg[v].end());
        }
        return (ans);
    }
};
