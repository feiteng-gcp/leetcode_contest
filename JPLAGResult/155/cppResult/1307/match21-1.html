typedef long long ll;
class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        ll l=0,r=2000000000;

        ll ab=lcm(ll(a),ll(b)),bc=lcm(ll(b),ll(c)),ac=lcm(ll(c),ll(a)),abc=lcm(ab,ll(c));
        while(r-l>1){
            ll m=l/2+r/2+(l%2+r%2)/2;
            if(m/a+m/b+m/c-m/ab-m/bc-m/ac+m/abc>=n)r=m;
            else l=m;
        }
        return r;
    }
};
