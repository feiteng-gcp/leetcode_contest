typedef long long ll;
class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) 
    {
        return findNthTerm(a, b, c, n);
        
    }
    
    ll gcd(ll a, ll b) 
    { 
        if (a == 0) 
            return b; 
  
        return gcd(b % a, a); 
    } 
  
    ll lcm(ll a, ll b) 
    { 
        return (a * b) / gcd(a, b); 
    } 
  

    ll divTermCount(ll a, ll b, ll c, ll num) 
    { 
        return ((num / a) + (num / b) + (num / c) 
            - (num / lcm(a, b)) 
            - (num / lcm(b, c)) 
            - (num / lcm(a, c)) 
            + (num / lcm(a, lcm(b, c)))); 
    } 
  

    ll findNthTerm(ll a, ll b, ll c, ll n) 
    { 
      
        ll low = 1, high = 1e18, mid; 
  
        while (low < high) { 
            mid = low + (high - low) / 2; 
  
            if (divTermCount(a, b, c, mid) < n) 
                low = mid + 1; 
  
            else
                high = mid; 
        } 
  
        return low; 
    } 
  
};
