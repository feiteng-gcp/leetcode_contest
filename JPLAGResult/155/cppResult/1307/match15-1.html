class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        long long s,e;
        s=1;

        e=2e9;
        long long ab=(1L*a*b)/gcd(a,b);
        long long ac=(1L*a*c)/gcd(a,c);
        long long bc=(1L*b*c)/gcd(b,c);
        long long abc=(1L*ab*c)/gcd(ab,c);
        while(s<e){
            long long m=(s+e)/2;
            int r=m/a+m/b+m/c-m/ab-m/ac-m/bc+m/abc;
            if(r==n){
                while(true){
                    if(m%a==0||m%b==0||m%c==0) return m;
                    m--;
                }

            }else if(r>n){
                e=m-1;
            }else{
                s=m+1;
            }
        }
        
        return s;
    }
};
