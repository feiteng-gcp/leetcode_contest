long long gcd(long long a, long long b) {
    return b == 0 ? a : gcd(b, a % b);
}

long long gdd(long long a, long long b) {
    return a / gcd(a, b) * b;
}

long long find_num(long long a, long long b, long long c, long long x) {
    long long x1 = x / a + x / b + x / c;
    long long x2 = x / gdd(a, b) + x / gdd(a, c) + x / gdd(b, c);
    long long x3 = x / gdd(gdd(a, b), c);
    return x1 - x2 + x3;
}

class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        long long l = 0, r = 1e18;
        while (l < r) {
            long long mid = (l + r) / 2;
            if (find_num(a, b, c, mid) >= n) r = mid;
            else    l = mid + 1;
        }
        return l;
    }
};
