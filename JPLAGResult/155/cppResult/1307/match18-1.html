#define ll long long
class Solution {
public:
    ll gcd(ll a, ll b){ 
        if (a == 0) 
            return b; 

        return gcd(b % a,a); 
    } 
    ll lcm(ll a, ll b){
        return (a * b) / gcd(a, b);
    }
    ll count(ll a, ll b, ll c, ll num){
        return ((num / a) + (num / b) + (num / c) 
            - (num / lcm(a, b)) 
            - (num / lcm(b, c)) 
            - (num / lcm(a, c)) 
            + (num / lcm(a, lcm(b, c)))); 
    }
    int nthUglyNumber(int n, int a, int b, int c) {
        ll x = a, y = b, z = c, k = n;
        ll low = 1, high = 2000000009, mid, cnt;
        while(low<high){
            mid = low+(high-low)/2;

            cnt = count(x, y, z, mid);
            // if(cnt==k) return mid;
            if(cnt<k){
                low = mid+1;
            }else{
                high = mid;
            }
        }
        return low;
    }
};
