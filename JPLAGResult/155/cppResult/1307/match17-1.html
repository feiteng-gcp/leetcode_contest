class Solution {
public:
    typedef long long int ll;
    

    ll lcm(ll a, ll b) { 
        return (a * b) / __gcd(a, b); 
    } 
  
    ll count(ll N, ll a, ll b, ll c) { 
        return ((N / a) + (N / b) + (N / c) 
               - (N / lcm(a, b)) - (N / lcm(b, c)) - (N / lcm(a, c)) 
               + (N / lcm(a, lcm(b, c)))); 
    } 
  
    int nthUglyNumber(int n, int a, int b, int c) {
        

        int low = 1, high = 2 * 1e9, mid; 
  
        while(low < high) { 
            mid = low + (high - low)/2; 
            if(count(mid, a, b, c) < n) {
                low = mid + 1;
            } else {
                high = mid;
            }
        } 
        return low; 
    }
};
