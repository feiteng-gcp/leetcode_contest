class Solution {
    typedef long long ll;
    ll gcd(ll x,ll y)
    {
        return y?gcd(y,x%y):x;
    }
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        ll ab=(ll)a*b/gcd(a,b),bc=(ll)b*c/gcd(b,c),ca=(ll)c*a/gcd(c,a),abc=ab*c/gcd(ab,c),l=0,r=2000000000,mid,s;
        while(l+1<r)
        {
            mid=l+r>>1;
            s=mid/a+mid/b+mid/c-mid/ab-mid/bc-mid/ca+mid/abc;
            if(s<n)l=mid;
            else r=mid;
        }
        return r;
    }
};
