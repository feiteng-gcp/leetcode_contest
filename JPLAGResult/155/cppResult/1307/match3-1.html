class Solution {
 public:
  int nthUglyNumber(long n, long a, long b, long c) {
    if(!(a<=b&&b<=c)) {
        vector<int> vc = {a,b,c};
        sort(vc.begin(), vc.end());
        return nthUglyNumber(n, vc[0], vc[1], vc[2]);
    }
    if(b%a==0&&c%a==0)
        return n*a;
    if(b%a==0)
        return nthUglyNumber(n, a, c);
    if(c%b==0)
        return nthUglyNumber(n, a, b);
    if(c%a==0)
        return nthUglyNumber(n, a, b);
    long ab = gcd(a,b);
    long ac = gcd(a,c);
    long bc = gcd(b,c);
    ab = a*b/ab;
    ac = a*c/ac;
    bc = b*c/bc;
    long abc = gcd(gcd(a,b),c);
    abc = a*b*c/abc;
    long l = 1;
    long r = 1E18;
     long mid = (l+r)/2;

    while(l<=r) {
       mid = (l+r)/2;
      if(mid/a+mid/b+mid/c-mid/ab-mid/bc-mid/ac+mid/abc==n) {
        l = mid;
        break;
      }
      else if(mid/a+mid/b+mid/c-mid/ab-mid/bc-mid/ac+mid/abc<n)
        l = mid+1;
      else
        r = mid-1;

    }
    long prev = mid;
    while(mid/a+mid/b+mid/c-mid/ab-mid/bc-mid/ac+mid/abc==n) {
        prev = mid;
        mid--;
    }
    return prev;
  }
    
  int nthUglyNumber(long n, long a, long b) {
    long ab = gcd(a,b);
    ab = a*b/ab;
    long l = 1;
    long r = 1E18;
     long mid = (l+r)/2;

    while(l<=r) {
       mid = (l+r)/2;
      if(mid/a+mid/b-mid/ab==n) {
        l = mid;
        break;
      }
      else if(mid/a+mid/b-mid/ab<n)
        l = mid+1;
      else
        r = mid-1;

    }
    long prev = mid;
    while(mid/a+mid/b-mid/ab==n) {
        prev = mid;
        mid--;
    }
    return prev;
  }
};
