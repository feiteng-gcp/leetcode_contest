class Solution {
public:
  long long gcd(long long la, long long lb) {
    return lb == 0?la: gcd(lb, la%lb);
  }
  long long lcm(long long la, long long lb) {
    return la * lb / gcd(la, lb); 
  }
    int nthUglyNumber(int n, int a, int b, int c) {
      long long lo = 0, hi = 0x3f3f3f3f3f3f3f3fll;
      long long la = a;
      long long lb = b;
      long long lc = c;
      long long lab = lcm(la, lb);
      long long lbc = lcm(lc, lb);
      long long lac = lcm(la, lc);
      long long labc = lcm(lc, lcm(la, lb));
      while (lo < hi) {
        long long mid = (lo + hi) / 2;
        long long tot = mid / la + mid /lb + mid /lc - mid/lab - mid/lbc - mid/lac + mid/labc;
        if (tot >= n) hi = mid;
        else lo = mid + 1;
      }
      return hi;
    }
};
