class Solution {
    typedef long long ll;
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        ll le = 0, ri = 1e18 + 20, mi;
        ll ab = ll(a) * b / __gcd(a, b), ac = ll(a) * c / __gcd(a, c), bc = ll(b) * c / __gcd(b, c);
        ll abc = ab * c / __gcd(ab, ll(c));
        while(le + 1 < ri) {
            mi = (le + ri) / 2;
            ll cuenta = mi / a + mi / b + mi / c;
            cuenta -= mi / ab + mi / ac + mi / bc;
            cuenta += mi / abc;
            //cout << mi << ' ' << cuenta << '\n';
            if(cuenta >= n)
                ri = mi;
            else
                le = mi;
        }
        return ri;
    }
};
