class Solution {
public:
    using uint64 = long long;
    int nthUglyNumber(int n, int a, int b, int c) {
        uint64 ab = (uint64) a * b / __gcd(a, b);
        uint64 bc = (uint64) b * c / __gcd(b, c);
        uint64 ac = (uint64) a * c / __gcd(a, c);
        uint64 abc = ab * c / __gcd(ab, (uint64) c);
        
        uint64 left = 0, right = 2e9;
        while (left < right) {
            uint64 mid = (left + right) / 2;
            uint64 counts = mid / a + mid / b + mid / c - mid / ab - mid / bc - mid / ac + mid / abc;
            if (counts < n) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        
        return left;
    }
};
