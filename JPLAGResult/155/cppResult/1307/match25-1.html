class Solution {
public:

    using lint = long long int;
    int nthUglyNumber(lint n, lint a, lint b, lint c) {
        lint l = 0, r = 2000000000 + 10;
        lint lab = a * b / __gcd(a, b);
        lint lbc = b * c / __gcd(b, c);
        lint lca = c * a / __gcd(c, a);

        lint labc = lca * b / __gcd(lca, b);
        while (r > l + 1)
        {
            lint x = l + (r - l) / 2;
            lint cou = x / a + x / b + x / c - x / lab - x / lbc - x / lca + x / labc;
            if (cou >= n) r = x;
            else l = x;
        }
        return r;
    }
};
