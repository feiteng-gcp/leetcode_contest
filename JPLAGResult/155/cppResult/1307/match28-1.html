class Solution {
public:
    long long gcd(long long a, long long b) {
        if (a == 0 || b == 0) {
            return a + b;
        }
        if (a > b) {
            return gcd(a % b, b);
        } else {
            return gcd(a, b % a);
        }

    }
    
    long long lcm(long long a, long long b) {
        return a / gcd(a, b) * b;
    }
    
    int nthUglyNumber(int n, int a, int b, int c) {
        long long l = 1, r = 2000000000LL;

        
        long long res = -1;
        long long nn = n;
        
        while (l <= r) {
            long long key = (l + r) / 2;
            
            long long nums = key / a + key / b + key / c - key / lcm(a, b) - key / lcm(a, c) - key / lcm(b, c) + key / lcm(a, lcm(b, c));
            
            if (nums >= nn) {
                r = key - 1;
                res = key;
            } else {
                l = key + 1;
            }
        }
        return res;
    }
};
