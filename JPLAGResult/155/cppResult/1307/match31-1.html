using ll = long long;

class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        ll l = 1, r=2e9;
        ll gab=__gcd(a,b), gac=__gcd(a,c), gbc=__gcd(b,c);
        ll lc = c;
        ll cab = a/gab*b, cac=a/gac*c,cbc=b/gbc*c,cabc=cab/__gcd(cab,lc)*c;
        while(l<=r){
            ll mid = (l+r)>>1;
            ll an = mid / a;
            ll bn = mid / b;
            ll cn = mid / c;
            ll abn = mid / cab;
            ll acn = mid / cac;
            ll bcn = mid / cbc;
            ll abcn = mid / cabc;
            ll x = an+bn+cn-abn-acn-bcn+abcn;
            if(x >= n) r=mid-1;
            else l=mid+1;
        }
        return l;
    }
};
