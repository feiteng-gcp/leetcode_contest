class Solution {
public:

    long long lcm(long long a, long long b)
    {
        return a*(b/__gcd(a,b));
    }
    
    long long countMul(long long n, long long a, long long b, long long c)
    {
        long long A = n/a;
        long long B = n/b;
        long long C = n/c;
        long long AB = n/lcm(a, b);
        long long AC = n/lcm(a, c);
        long long BC = n/lcm(b, c);
        long long ABC = n/lcm(c, lcm(a, b));
        return A + B + C - AB - BC - AC + ABC;
    }
    
    int nthUglyNumber(int n, int a, int b, int c) {
        long long lo = 1, hi = INT_MAX;
        long long mid;
        while(lo < hi)
        {
            mid = lo + (hi-lo)/2;
            if(countMul(mid, a, b, c) < n) lo = mid+1;
            else hi = mid;
        }
        return (int)lo;
    }
};
