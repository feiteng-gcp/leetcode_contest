class Solution {
public:
    long long gcd(long long a, long long b) { 
        return a == 0 ? b : gcd(b % a, a); 
    } 


    long long lcm(long long a, long long b) { 
        return (a * b) / gcd(a, b); 
    } 

    long long cnt(long long a, long long b, long long c, long long num) { 
        return ((num / a) + (num / b) + (num / c) 
                - (num / lcm(a, b)) 
                - (num / lcm(b, c)) 
                - (num / lcm(a, c)) 
                + (num / lcm(a, lcm(b, c)))); 
    } 
    int nthUglyNumber(int n, int a, int b, int c) {

        long long low = 1, high = 2*10e9, mid;
        while (low < high) {
            mid = low + (high-low)/2;
            if (cnt(a, b, c, mid) < n) low = mid+1;
            else high = mid;
        }
        return low;
    }
};
