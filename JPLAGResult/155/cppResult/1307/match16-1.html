class Solution {
public:

    long long lcm(long long x, long long y) {
        return x / __gcd(x, y) * y;
    }

    long long count_ugly(long long n, long long a, long long b, long long c) {
        long long answer = n / a + n / b + n / c;
        answer -= n / lcm(a, b) + n / lcm(b, c) + n / lcm(c, a);
        answer += n / lcm(lcm(a, b), c);
        return answer;
    }


    int nthUglyNumber(int n, int a, int b, int c) {
        long long low = 0, high = 2e9 + 5;

        while (low < high) {
            long long mid = (low + high) / 2;

            if (count_ugly(mid, a, b, c) < n)
                low = mid + 1;
            else
                high = mid;
        }

        return low;
    }
};
