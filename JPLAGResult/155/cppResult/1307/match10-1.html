class Solution {
public:

    int nthUglyNumber(int n, int a, int b, int c) {
        long long lo = 0, hi = 2e9+9;
        long long ab = a*1LL*b/__gcd(a,b);
        long long bc = b*1LL*c/__gcd(b,c);
        long long ac = a*1LL*c/__gcd(a,c);
        long long abc = ab*1LL*c/__gcd(ab,(long long)c);

        while(lo<hi) {
            long long mid = (hi+lo)/2;
            
            long long rank = mid/a+mid/b+mid/c-mid/ab-mid/bc-mid/ac+mid/abc;
            
            if(rank>=n)
                hi = mid;
            else
                lo = mid+1;
        }
        return lo;
    }
};
