class Solution {

	bool Check(long long C, long long n, long long a, long long b, long long c)
	{
		long long Count = 0;
		
		//cout << Count << endl;
		Count += (C / a);
		//cout << Count << endl;
		Count += (C / b);
		Count += (C / c);
		Count -= (C / lcm(a, b));
		Count -= (C / lcm(a, c));
		Count -= (C / lcm(b, c));
		Count += (C / lcm(lcm(a, b), c));
		
		//cout << C << " " << Count << endl;
		
		return (Count >= n);
	}
	
public:

    int nthUglyNumber(int n, int a, int b, int c) {
        long long S = 1;
		long long E = 2000000001;
		
		while(S < E)
		{
			long long C = (S + E) / 2;
			

			if(Check(C, n, a, b, c))
			{
				E = C;
			}
			else
			{
				S = C + 1;
			}
		}
		
		return S;
    }
};
