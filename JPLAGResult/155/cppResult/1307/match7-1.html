class Solution {
public:

long long int gcd(long long int a,long long  int b) { 
        if (a == 0) 
            return b; 
        return gcd(b % a, a); 
} 
long long int lcm(long long int a, long long int b) { 
        return (a * b) / gcd(a, b); 
} 
long long int div(long long int a, long long int b, long long int c, long long int n) { 

    return ((n / a) + (n / b) + (n / c) 
            - (n / lcm(a, b)) 
            - (n / lcm(b, c)) 
            - (n / lcm(a, c)) 
            + (n / lcm(a, lcm(b, c)))); 
} 

int find(int a, int b, int c, int n) { 
    int low = 1, high = INT_MAX, mid; 
  
    while (low < high) { 
        mid = low + (high - low) / 2; 
        if (div(a, b, c, mid) < n) 
            low = mid + 1; 
        else
            high = mid; 
    } 
  
    return low; 
} 
    int nthUglyNumber(int n, int a, int b, int c) {
        return find(a, b, c, n);
    }
};
