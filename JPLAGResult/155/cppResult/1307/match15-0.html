class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        long long lo = 0;
        long long hi = 2000000000;
        
        long long lcmab = 1LL * a * b / __gcd(a, b);
        long long lcmbc = 1LL * b * c / __gcd(b, c);
        long long lcmac = 1LL * a * c / __gcd(a, c);
        long long lcmabc = 1LL * lcmab * c / __gcd(lcmab, 1LL * c);
        
        while (lo + 1 < hi) {
            long long mid = lo + (hi - lo) / 2;
            
            long long rk = mid / a + mid / b + mid / c;
            
            rk -= mid / lcmab;
            rk -= mid / lcmbc;
            rk -= mid / lcmac;
            
            rk += mid / lcmabc;
            
            if (rk >= n) {
                hi = mid;
            } else {
                lo = mid;
            }
        }
        
        return hi;
    }
};
