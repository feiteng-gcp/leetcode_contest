class Solution {
public:
	long long gcd(long long a, long long b){
		return __gcd(a, b);
	}

	long long lcm(long long a, long long b){
		return (a * b) / gcd(a, b);
	}
	long long count(long long n, long long a, long long b, long long c) {
		long long ct1 = n / a; 
		long long ct2 = n / b;
		long long ct3 = n / c;

		long long ct4 = n / (lcm(a, b));     
		long long ct5 = n / (lcm(a, c));
		long long ct6 = n / (lcm(b, c));

		long long all = n / lcm(lcm(a, b), c);   

		return  ct1 + ct2 + ct3 - ct4 - ct5 - ct6 + all;
	}
	int nthUglyNumber(int n, int a, int b, int c) {
		long long l = 1, r = (2e9) + 4;
		while (l <= r) {
			long long mid = (l + r) / 2;
			long long curr = count(mid, a, b, c);
			if (mid % a != 0 && mid % b != 0 && mid % c != 0) {
				if (curr < n)l = mid + 1;
				else r = mid - 1;
			}
			else {
				if (curr == n)return mid;
				if (curr < n)l = mid + 1;
				else r = mid - 1;
			}
		}
		return -1;
	}
};
