class Solution {
public:

    long long lcm(long long a, long long b) {
        return a * b / gcd(a, b);
    }
    long long gcd(long long a, long long b){
    if(b == 0)
        return a;
    else
        return gcd(b, a %b);
    }
    

    int nthUglyNumber(long long n, long long a, long long b, long long c) {
        long long lo = 1, hi = 2 * 1e9;

        while (lo < hi) {
            long long m = (lo + hi) / 2;
            long long count = m / a + m / b + m / c - m / lcm(a, b) - m / lcm(b, c) - m / lcm(a, c) + m / lcm(lcm(a, b), c);
            if (count < n) {
                lo = m + 1;
            } else {
                hi = m;
            }
        }
        return lo;
    }
};
