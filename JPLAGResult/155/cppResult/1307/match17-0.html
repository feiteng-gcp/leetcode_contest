class Solution {
public:
    #define ll long long
    ll gcd(ll a, ll b) {
        while(b!=0) {
            ll t = b;
            b=a%b;
            a=t;
        }
        return a;
    }    
    ll lcm(ll x, ll y) {
        return x*(y/gcd(x,y));
    }
    
    ll Count(ll n, ll a, ll b, ll c) {
        return n/a+n/b+n/c-n/lcm(a,b)-n/lcm(b,c)-n/lcm(a,c)+n/lcm(a,lcm(b,c));
    }
    
    
    int nthUglyNumber(int n, int a, int b, int c) {
        ll low=1;
        ll high=2222222222ll;
        
        while(low < high) {
            ll mid = (low+high)/2;
            if (Count(mid,a,b,c)<n) {
                low = mid+1;
            } else {
                high = mid;
            }
        }
        return low;
        
        
    }
};
