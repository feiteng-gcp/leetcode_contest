class Solution {
public:
    long long closest(long long x,long long a,long long b, long long c){
        return max(x-(x%a),max(x-x%b,x-x%c));
    }
    
     long long lcm(long long a, long long b)  
 {  
    return (1LL*a*b)/__gcd(a, b);  
 }  

    int nthUglyNumber(int n, int a, int b, int c) {
        long long lo = 1 , hi = INT_MAX;
        long long ans;
        while(lo<=hi){
            long long mid = lo + (hi-lo)/2;
            long long divct = mid/(1LL*a) + mid/(1LL*b) + mid/(1LL*c) - mid/(1LL*lcm(a,b)) - mid/(1LL*lcm(c,b)) - mid/(1LL*lcm(a,c)) + mid/(1LL*(lcm(lcm(a,b),c)));
            if(divct>=n){
              //  cout<<"found at "<<mid<<endl;
                ans = mid;//closest(mid,a,b,c);
                hi = mid-1;
            }
            else{
                lo = mid+1;
            }
        }
        return ans;
    }
};
