class Solution {

    long long gcd(long long x, long long y) {
        return y ? gcd(y, x % y) : x;
    }
    long long count(long long n, long long a, long long b, long long c) {
        long long res = n / a + n / b + n / c - n / (a * b / gcd(a, b)) - n / (a * c / gcd(a, c)) - n / (b * c / gcd(b, c)) + n / (a * b / gcd(a, b) * c / gcd(a * b / gcd(a, b), c));
        return res;
    }
public:

    int nthUglyNumber(int n, int a, int b, int c) {
        
//         a /= gcd(a, b);
//         c /= gcd(a, c);
//         c /= gcd(b, c);
        // cout << count(4,2,3,4);
        long long left = 1, right = 2e9, mid;
        while(left < right) {
            mid = left + (right - left) / 2;
            if(count(mid, a, b, c) < n)
                left = mid + 1;
            else
                right = mid;
        }
        return left;
    }
};
