class Solution {
public:
    long long gcd(long long a, long long b) {
        if (b == 0)
            return a;
        return gcd(b, a % b);
    }
    long long lcm(long long a, long long b) {
        return 1ll * a / gcd(a, b) * b;
    }
    long long lcm(long long a, long long b, long long c) {
        return lcm(a, lcm(b, c));
    }
    long long get_value(long long n, long long a, long long b ,long long c) {
        long long ans = 0;
        ans += n / a;
        ans += n / b;
        ans += n / c;
        ans -= n / lcm(a, b);
        ans -= n / lcm(b, c);
        ans -= n / lcm(c, a);
        ans += n / lcm(a, b, c);

        return ans;
    }
    int nthUglyNumber(int n, int a, int b, int c) {
        long long left = 0;
        long long right = 5e18;
        while (right - left > 1) {
            long long mid = (left + right) / 2;
            long long val = get_value(mid, a, b, c);
            if (val >= n) {
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }
};
