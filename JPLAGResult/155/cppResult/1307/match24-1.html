class Solution {
public:
    long long ggcd(long long a, long long b) {
        return !b ? a : gcd(b, a%b);
    }
    int nthUglyNumber(int n, int a, int b, int c) {
        long long lcm_ab = 1ll*a*b/ggcd(a, b), lcm_ac = 1ll*a*c/ggcd(a, c), lcm_bc = 1ll*b*c/ggcd(b, c);
        long long lcm_abc = lcm_ab*c/ggcd(lcm_ab, c);
        //int cnt = labc/a+labc/b+labc/c-labc/lab-labc/lac-labc/lbc+labc/labc;
        //int res = n/cnt*labc;

        long long l = 1, r = 2e9;
        while (l < r) {
            long long mid = (l+r)/2;
            long long now = mid/a+mid/b+mid/c-mid/lcm_ab-mid/lcm_ac-mid/lcm_bc+mid/lcm_abc;
            if (now >= n) r = mid;
            else if (now < n) l = mid+1;
        }
        // cout << lcm_ab << " " << lcm_ac << " " << lcm_bc << " " << lcm_abc << " " <<  l << endl;
        while (true) {
            if (l%a == 0 || l%b == 0 || l%c == 0) break;
            ++l;
        }
        return l;
    }
};
