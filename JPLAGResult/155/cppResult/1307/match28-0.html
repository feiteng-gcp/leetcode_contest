class Solution {
public:
    long long gcd(long long a, long long b)
    {
        if (a < b)
        {
            long long tmp = a;
            a = b;
            b = tmp;
        }
        while (a % b != 0)
        {
            long long tmp = a % b;
            a = b;
            b = tmp;
        }
        return b;
    }
    long long lcm(long long a, long long b)
    {
        return a * b / gcd(a, b);
    }
    int nthUglyNumber(int n, int a, int b, int c) {
        long long la = a;
        long long lb = b;
        long long lc = c;
        long long l = 1;
        long long r = 2000000000;
        while (l < r)
        {
            long long mid = (l + r) / 2;
            long long num = mid / la + mid / lb + mid / lc - mid / lcm(la, lb) - mid / lcm(la, lc) - mid / lcm(lc, lb) + mid / lcm(lcm(la, lb), lc);
            if (num < n)
            {
                l = mid + 1;
            }
            else
            {
                r = mid;
            }
        }
        return l;
    }
};
