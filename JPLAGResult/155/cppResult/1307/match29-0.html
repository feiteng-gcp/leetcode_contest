class Solution {
public:
    long long lcm(long long a, long long b) {
        
        long long gc = __gcd(a,b);
        return a*b/gc;
    }
    
    long long lcm(long long a, long long b,long long c) {
        
        return lcm(lcm(a,b),c);
    }
    int nthUglyNumber(int n, int a, int b, int c) {
        
        long long lo = 1;
        long long hi = (long long)1e13;
        while(lo<=hi) {
            
            long long mid = (lo+hi)/2;
            long long tot = mid/a + mid/b + mid/c - mid/lcm(a,b) - mid/lcm(b,c) - mid/lcm(a,c) + mid/lcm(a,b,c);
            if(tot<n) lo = mid+1;
            else hi = mid-1;
        }
        return lo;
    }
};
