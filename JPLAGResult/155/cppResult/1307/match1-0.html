struct tp{
    int i;
    int j;
    int k;
    int v;
};
class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        vector<int> v{a,b,c};
        sort(v.begin(), v.end());
        if(v[0] == 1) {
            return n;
        }
        a = v[0];
        b = v[1];
        c = v[2];
        int p = 0;
        
        long i = 1;
        long j = 1;
        long k = 1;
       
        long v1 = a * i;
        long v2 = b * k;
        long v3 = c * k;
        
        long ahead = (b/a);
        if(ahead >= n) {
            return a * n;
        }
        else {
            p += (ahead - 1);
            v1  += (ahead - 1) * a;
            i += ahead - 1;
        }
     //   cout<<v1<<" "<<v2<<" "<<v3<<" "<<p<<endl;
        
        for(; p < n; p++) {
            long mi = min(min(v1, v2), v3);
         //   cout<<p <<" "<< v1 << " "<< v2<<" "<< v3<<endl;
            if(p == n-1) return mi;
            
            if(v1 == mi) {
                i++;
                v1 = a *i;
                ahead = (min(v2,v3) - v1) / a;
             //   cout<<(min(v2,v3) - v1)<<" "<<ahead<<endl;
                
                if(ahead < 0) goto loop;
                if(p + ahead >= n - 1) {
                    return v1 + (n-p-2)*a;
                }
                
                v1 += ahead * a;
                p += ahead ;
                i += ahead;
                loop:;
            }
            if(v2 == mi) {
                j++;
                v2 = b * j;
            }
            if(v3 == mi) {
                k++;
                v3 = c * k;
            }
            
        }
        return 0;
        
    }
};
