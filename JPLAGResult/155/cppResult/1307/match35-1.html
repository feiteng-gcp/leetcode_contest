class Solution {

public:
  int nthUglyNumber(int n, int a, int b, int c) {
    ll ab=(ll)a/__gcd(a,b)*b,ac=(ll)a/__gcd(a,c)*c,bc=(ll)b/__gcd(b,c)*c;
    ll abc=ab/__gcd(ab,(ll)c)*c;
    ll l=0,r=MAX*3;
    while(l+1<r) {
      ll m=(l+r)/2;
      ll cnt=m/a+m/b+m/c;
      cnt-=m/ab+m/ac+m/bc;
      cnt+=m/abc;
      if(cnt<n) l=m;
      else r=m;
    }
    return r;
  }
};
