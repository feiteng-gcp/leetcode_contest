typedef long long int LL;
class Solution {
public:
    LL gcd(LL a, LL b){
        return b==0?a:gcd(b, a%b);
    }
    LL help(LL a, LL b){
        LL g = gcd(a, b);
        return a*b/g;
    }

    int nthUglyNumber(LL n, LL a, LL b, LL c) {
        LL ab = help(a, b);
        LL ac = help(a, c);
        LL bc = help(b, c);
        LL abc = help(ab, c);
        LL lhs = 0, rhs = INT_MAX;
        while(rhs-lhs>1){
            LL mid = lhs + (rhs-lhs)/2;
            LL cnt = mid/a + mid/b + mid/c - mid/ab - mid/ac - mid/bc + mid/abc;
            if(cnt>=n){
                rhs = mid;
            }
            else{
                lhs = mid;
            }
        }
        return rhs;
    }
};
