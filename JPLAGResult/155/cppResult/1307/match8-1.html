class Solution {
public:
    
    long long gcd(long long a, long long b) {
        return b == 0 ? a : gcd(b, a % b);
    }
    
    long long lcm(long long a, long long b) {
        return a / gcd(a, b) * b;
    }
    
    int nthUglyNumber(int n, int a, int b, int c) {
        long long l = 0, r = 1e+15;

        while (r - l > 1) {
            long long m = (l + r) >> 1;
            
            long long cnt = (m / a) + (m / b) + (m / c)
                - (m / lcm(a, b))
                - (m / lcm(a, c))
                - (m / lcm(b, c))
                + (m / lcm(a, lcm(b, c)));
            
            // cout << m << "  " << cnt << endl;
            if (cnt < n) l = m;
            else r = m;
        }
        return r;
    }
};
