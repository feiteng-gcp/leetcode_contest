class Solution {
public:
    long long lcm(int x, int y) {

        return 1LL*x*y/__gcd(x, y);
    }
    int nthUglyNumber(int n, int a, int b, int c) {
        long long l = 0, r = 1e10, ans = 0;

        while(l <= r) {
            long long m = (l+r)/2;
            long long now = m/a+m/b+m/c-m/lcm(a, b)-m/lcm(a, c)-m/lcm(b, c)+m/lcm(lcm(a, b), c);
            if(now >= n) {
                ans = m;
                r = m-1;
            }
            else l = m+1;
        }
        return ans;
    }
};
