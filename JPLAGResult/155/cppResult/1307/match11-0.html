class Solution {
public:
    long long lcm(long long a, long long b){ 
        return (a * b) / __gcd(a, b); 
    } 
    long long counter(long long a, long long b, long long c, long long num){ 
        return ((num / a) + (num / b) + (num / c) 
                - (num / lcm(a, b)) 
                - (num / lcm(b, c)) 
                - (num / lcm(a, c)) 
                + (num / lcm(a, lcm(b, c)))); 
    } 
    int nthUglyNumber(int n, int a, int b, int c) {
        int low = 1, high = INT_MAX, mid; 
        while (low < high) { 
            mid = low + (high - low) / 2; 
            if (counter(a, b, c, mid) < n) 
                low = mid + 1; 
            else
                high = mid; 
        } 
  
        return low; 
    }
};
