class Solution {
public:
    int nthUglyNumber(int n, int a, int b, int c) {
        long long l = 1, r = 2e9;
        long long result = r;
        while (l <= r) {
            long long mid = l + (r - l) / 2;
            long long num = getNum(mid, a, b, c);
            if (num >= n) {
                r = mid - 1;
                result = min(result, mid);
            } else {
                l = mid + 1;
            }
        }
        return result;
    }
    
private:
    long long getNum(long long n, long long a, long long b, long long c) {
        return n / a + n / b + n / c - n / lcm(a, b) - n / lcm(b, c) - n / lcm(a, c) + n / lcm(lcm(a, b), c);
    }
    
    long long lcm(long long x, long long y) {
        return x / __gcd(x, y) * y;
    }
};
