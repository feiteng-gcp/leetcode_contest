void dfs(int i,string &S,int node,map<int,list<int> >&M,map<int,int>&mark,vector<char>&V,vector<int>&temp)
{
    mark[node] = i;
    V.push_back(S[node]);
    temp.push_back(node);
    for(auto it:M[node])
    {
        if(mark[it]==0)
            dfs(i,S,it,M,mark,V,temp);
    }        
}
class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        map<int,int> mark;
        map<int,list<int> >M;
        for(int i=0;i<pairs.size();i++)
        {
            M[pairs[i][0]].push_back(pairs[i][1]);
            M[pairs[i][1]].push_back(pairs[i][0]);
        }
        for(int i=0;i<s.length();i++)
        {
            if(mark[i]==0){
                vector<char> V;
                vector<int> temp;
                dfs(i+1,s,i,M,mark,V,temp);
                sort(V.begin(),V.end());
                sort(temp.begin(),temp.end());
                for(int i=0;i<temp.size();i++)
                {
                    s[temp[i]] = V[i];
                }
            }
        }
        return s;
    }
};
