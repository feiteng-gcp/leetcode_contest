class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        const int n = s.size();
        vector<unordered_set<int>> adjLists(n);
        for (const auto &p : pairs) {
            adjLists[p[0]].emplace(p[1]);
            adjLists[p[1]].emplace(p[0]);
        }
        string result(s);
        vector<int> vis(n, 0);
        for (int i = 0; i < n; ++i) {
            if (vis[i]) {
                continue;
            }
            vector<int> indices;
            vector<char> values;
            dfs(s, adjLists, i, vis, indices, values);
            sort(indices.begin(), indices.end());
            sort(values.begin(), values.end());
            for (int j = 0; j < indices.size(); ++j) {
                result[indices[j]] = values[j];
            }
        }
        return result;
    }
private:
    void dfs(const string &s, const vector<unordered_set<int>> &adjLists, int i, vector<int> &vis,
            vector<int> &indices, vector<char> &values) const {
        if (vis[i]) {
            return;
        }
        vis[i] = 1;
        indices.emplace_back(i);
        values.emplace_back(s[i]);
        for (int j : adjLists[i]) {
            dfs(s, adjLists, j, vis, indices, values);
        }
    }
};
