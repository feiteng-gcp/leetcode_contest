class Solution {
public:
    int find(vector<int>& f, int x) {
        return f[x] == f[f[x]] ? f[x] : (f[x] = find(f, f[x]));
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int n = s.size();
        vector<int> f(n);
        for (int i = 0; i < n; ++i) f[i]=i;
        for (auto& p : pairs) {
            int a = find(f, p[0]);
            int b = find(f, p[1]);
            if (a != b) f[max(a, b)] = min(a, b);
        }
        vector<vector<char>> cs(n);
        for (int i = 0; i < n; ++i) {
            int x = find(f, i);
            cs[x].emplace_back(s[i]);
        }
        for (auto& s : cs) sort(s.begin(), s.end());
        vector<int> cur(n, 0);
        for (int i = 0; i < n; ++i) {
            int x = find(f, i);
            s[i] = cs[x][cur[x]++];
        } 
        return s;
    }
};
