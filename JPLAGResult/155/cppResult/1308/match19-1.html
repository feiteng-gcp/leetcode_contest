class Solution {
public:
    vector<vector<int>> edge;

    vector<int> ind;
    vector<deque<int>> gnq;
    string ss;
    void dfs(int i, int gn){
        ind[i]=gn;
        gnq[gn].push_back(ss[i]);
        for(int j:edge[i]){
            if(ind[j]<0){
                dfs(j, gn);
            }
        }
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int n=s.size();
        edge.resize(n);
        ind.resize(n,-1);
        ss = s;

        for(auto &it: pairs){
            edge[it[0]].push_back(it[1]);
            edge[it[1]].push_back(it[0]);
        }
        int cnt=0;
        for(int i=0;i<n;i++){
            if(ind[i]==-1){
                gnq.emplace_back();
                dfs(i, cnt++);
            }
        }
        for(auto &it: gnq){
            sort(begin(it),end(it));
        }
        for(int i=0;i<n;i++){
            s[i] = gnq[ind[i]].front();
            gnq[ind[i]].pop_front();
        }
        return s;
    }
};
