class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int N = s.length();
        for (int i = 0; i < N; ++i)
            f[i] = -1;
        for (auto p: pairs) {
            unite(p[0], p[1]);

        }
        for (int i = 0; i < N; ++i) {
            indices[findFather(i)].push_back(i);
            chars[findFather(i)].push_back(s[i]);
        }
        string ans(N, ' ');
        for (int i = 0; i < N; ++i) {
            int t;
            if (t = indices[i].size()) {
                sort(chars[i].begin(), chars[i].end());
                for (int j = 0; j < t; ++j)
                    ans[indices[i][j]] = chars[i][j];
            }
        }
        return ans;
    }
    int f[100005];
    vector<int> indices[100005];
    vector<char> chars[100005];
    int findFather(int x) {
        return f[x] < 0 ? x : f[x] = findFather(f[x]);
    }
    void unite(int x, int y) {
        int fx = findFather(x), fy = findFather(y);
        if (fx != fy)
            f[fx] = fy;
    }
};
