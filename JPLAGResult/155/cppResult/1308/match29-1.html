struct UnionFind {
    vector<int> parent;
    UnionFind(int n) : parent(n) {
        for (int i = 0; i < n; i++) parent[i] = i;
    }
    int find(int i) {
        if (parent[i] == i) return i;
        return parent[i] = find(parent[i]);

    }
    void uni(int i, int j) {
        int pi = find(i);
        int pj = find(j);
        parent[pi] = pj;
    }
};

class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        UnionFind u(s.size());
        int n = s.size();
        for (auto pair: pairs) {
            u.uni(pair[0], pair[1]);
        }
        unordered_map<int, vector<int>> v;
        for (int i = 0; i < n; i++) {
            int pi = u.find(i);
            v[pi].push_back(i);
        }
        string res = s;
        for (auto p : v) {
            string tmp = "";
            for (int i : p.second) tmp += s[i];
            sort(tmp.begin(), tmp.end());
            for (int i = 0; i < p.second.size(); i++) {
                res[p.second[i]] = tmp[i];
            }
        }
        return res;
    }
};
