class Solution {
public:
    void dfs(int source, vector<vector<int> > &G, vector<bool> &visited, vector<int> &v){
        visited[source] = true;
        for(int i=0; i<G[source].size(); i++){
            if(!visited[G[source][i]]){
                // vector<int> t = dfs(G[source][i], G, visited);
                // for(int j=0; j<t.size(); j++)
                //     res.push_back(t[j]);
                dfs(G[source][i], G, visited, v);
            }
        }
        // res.push_back(source);
        // return res;
        v.push_back(source);
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        vector<vector<int> > G(s.length());
        for(int i=0; i<pairs.size(); i++){
            int so = pairs[i][0], d = pairs[i][1];
            G[so].push_back(d);
            G[d].push_back(so);
        }
        vector<bool> visited(s.length());
        for(int i=0; i<pairs.size(); i++){
            if(!visited[pairs[i][0]]){
                vector<int> v;
                dfs(pairs[i][0], G, visited, v);
                sort(v.begin(), v.end());
                vector<char> temp;
                for(int j=0; j<v.size(); j++)
                    temp.push_back(s[v[j]]);
                sort(temp.begin(), temp.end());
                for(int j=0; j<v.size(); j++)
                    s[v[j]] = temp[j];
            }
        }
        return s;
    }
};
