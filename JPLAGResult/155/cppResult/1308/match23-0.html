class Solution {
public:
    int find(vector<int>& grp, int x) {
        while (grp[x] != x)
            x = grp[x];
        return x;
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        vector<int> grp(s.size(), 0);
        for(int i = 0; i < s.size(); i++)
            grp[i] = i;
        for(auto& p: pairs) {
            int r1 = find(grp, p[0]);
            int r2 = find(grp, p[1]);
            if (r1 != r2)
                grp[r1] = grp[r2] = min(r1, r2);
        }
        unordered_map<int, vector<int>> partition;
        for(int i = 0; i < s.size(); i++) {
            partition[find(grp, i)].push_back(i);
            // cout << find(grp, i) << " ";
        }
        unordered_map<int, string> split_str;
        for(auto& p: partition) {
            string tmp;
            for(int idx: p.second)
                tmp += s[idx];
            sort(tmp.begin(), tmp.end());
            split_str[p.first] = tmp;
        }
        unordered_map<int, int> idx_cnt;
        string ans = "";
        for(int i = 0; i < s.size(); i++) {
            int g = find(grp, i);
            ans += split_str[g][idx_cnt[g]++];
        }
        return ans;
            
    }
};
