class Solution {
   public:
    int findfa(int u, vector<int>& f) {
        return f[u] == -1 ? u : f[u] = findfa(f[u], f);
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int n = s.size();
        vector<int> f(n, -1);
        vector<priority_queue<char>> pq(n);
        for (auto& p : pairs) {
            int u = findfa(p[0], f);
            int v = findfa(p[1], f);
            if (u != v) {
                f[u] = v;
            }
        }
        for (int i = 0; i < n; i++) {
            pq[findfa(i, f)].push(-s[i]);
        }
        string ans(n, 0);
        for (int i = 0; i < n; i++) {
            char c = pq[findfa(i, f)].top();
            pq[findfa(i, f)].pop();
            ans[i] = -c;
        }
        return ans;
    }
};
