vector<int> adj[110000];
vector<char> vals;
vector<int> vtx;
bool vis[110000];
string st;
void dfs(int u){    
    vtx.push_back(u);vals.push_back(st[u]);    
    vis[u] = true;
    for(int i=0; i<adj[u].size(); i++){
        int v = adj[u][i];
        if(!vis[v]){
            dfs(v);
        }
    }
}

class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        st = s;
        for(int i=0; s[i]; i++){
            adj[i].clear();
        }
        for(int i=0; i<pairs.size(); i++){
            int u = pairs[i][0];
            int v = pairs[i][1];
            adj[u].push_back(v);
            adj[v].push_back(u);
            //cout << u << " " << v << endl;
        }
        string r = s;
        memset(vis,0,sizeof vis);
        for(int i=0; s[i]; i++){
            if(!vis[i]){
                vals.clear(); vtx.clear();
                dfs(i);
                sort(vals.begin(), vals.end());
                sort(vtx.begin(), vtx.end());
                for(int j=0; j<vtx.size(); j++) r[vtx[j]] = vals[j];
            }
        }
        return r;
    }
};
