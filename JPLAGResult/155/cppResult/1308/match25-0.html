class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        string res = s;
        int n = s.length();
        for(int i=0;i<n;i++)    res[i] = '#';
        vector<vector<int>> g(n);
        for(auto v : pairs){
            g[v[0]].push_back(v[1]);
            g[v[1]].push_back(v[0]);
        }
        for(int i=0;i<n;i++){
            if(res[i]!='#') continue;
            vector<int> indices;
            string chars;
            queue<int> q;
            q.push(i);
            res[i] = '*';
            indices.push_back(i);
            chars.push_back(s[i]);
            while(!q.empty()){
                int x = q.front(); q.pop();
                for(int j : g[x]){
                    if(res[j]=='#'){
                        q.push(j);
                        res[j] = '*';
                        indices.push_back(j);
                        chars.push_back(s[j]);
                    }
                }
            }
            
            sort(indices.begin(),indices.end());
            sort(chars.begin(),chars.end());
            for(int j = 0 ;j<chars.size();j++)
            {
               res[indices[j]] = chars[j];
            }
            
        }
        return res;
    }
};
