class Solution {
public:
    int parent[100005];
    int find(int x)
    {
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int S = s.size();
        int N = pairs.size();        
        for (int i = 0;i < S; i++)
            parent[i] = i;
        
        for (int i = 0;i < N; i++)
        {
            int x = find(pairs[i][0]);
            int y = find(pairs[i][1]);
            if (x!=y)
                parent[x] = y;
        }
        // cout<<"connected"<<endl;
        unordered_map<int, string> m;
        unordered_map<int, int> pointer;
        for (int i = 0;i < S; i++)
        {
            // cout<<i<<s[i]<<parent[i]<<endl;
            if (m.find(find(i)) == m.end())
            {
                m[parent[i]] = s[i];
                pointer[parent[i]] = 0;
            }                
            else
                m[parent[i]] += s[i];
        }
        // cout<<"connected"<<endl;
        for (auto it = m.begin(); it!=m.end();it++)
            sort(it->second.begin(), it->second.end());
        string ans;
        for (int i = 0;i < S; i++)
        {
            int idx = parent[i];
            ans += m[idx][pointer[idx]++];
        }
        return ans;        
    }
};
