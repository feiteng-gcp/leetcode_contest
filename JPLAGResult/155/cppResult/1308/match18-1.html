class Solution {
public:
    bool vis[100010];

    char ans[100010];
    string tmp;
    vector<int>mp[100010],pos;
    void dfs(int u,string &s){
        vis[u]=1;
        pos.push_back(u);
        tmp.push_back(s[u]);
        for(int i=0;i<mp[u].size();i++){
            int v=mp[u][i];
            if(vis[v])continue;
            dfs(v,s);
        }
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int n=s.size();
        for(int i=0;i<pairs.size();i++){
            int u=pairs[i][0],v=pairs[i][1];
            mp[u].push_back(v);
            mp[v].push_back(u);
        }

        for(int i=0;i<n;i++){
            if(!vis[i]){
                dfs(i,s);
                sort(tmp.begin(),tmp.end());
                sort(pos.begin(),pos.end());
                for(int j=0;j<pos.size();j++)ans[pos[j]]=tmp[j];
                tmp.clear();
                pos.clear();
            }
        }
        string A;
        for(int i=0;i<n;i++)A.push_back(ans[i]);
        return A;
    }
};
