class Solution {
public:
    vector<int> g[100005];
    int used[100005];
    string res;
    vector<int> pos;
    vector<char> c;
    
    void dfs(int u)
    {
        used[u]=1;
        c.push_back(res[u]);
        pos.push_back(u);
        for(int i=0;i<g[u].size();i++)
            if(!used[g[u][i]])
                dfs(g[u][i]);
    }
    
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        res=s;
        memset(used,0,sizeof(used));
        for(int i=0;i<pairs.size();i++)
        {
            int a=pairs[i][0];
            int b=pairs[i][1];
            g[a].push_back(b);
            g[b].push_back(a);
        }
        for(int i=0;i<s.size();i++)
            if(!used[i])
            {
                pos.clear();
                c.clear();
                dfs(i);
                sort(c.begin(),c.end());
                sort(pos.begin(),pos.end());
                for(int j=0;j<pos.size();j++)
                    res[pos[j]]=c[j];
            }
                
        return res;
    }
};
