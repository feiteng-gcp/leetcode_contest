class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        const int n = s.size();
        vector<bool> vis(n, false);
        vector<vector<int>> graph(n);
        for (const auto& p: pairs) {
            graph[p[0]].push_back(p[1]);
            graph[p[1]].push_back(p[0]);
        }
        
        vector<vector<int>> comp;
        for (int i = 0; i < n; ++i) {
            if (vis[i]) continue;
            comp.push_back({});
            vis[i] = true;
            comp.back().push_back(i);
            dfs(i, vis, graph, comp.back());
        }
        
        string result = s;
        for (auto& c: comp) {
            auto cp = c;
            sort(c.begin(), c.end());
            sort(cp.begin(), cp.end(), [&](int i, int j) {return s[i] < s[j];});

            for (int i = 0; i < c.size(); ++i) {
                result[c[i]] = s[cp[i]];
            }
        }
        return result;
    }

private:
    void dfs(int i, vector<bool>& vis, const vector<vector<int>>& graph, vector<int>& comp) {
        for (const int j: graph[i]) {
            if (vis[j]) continue;
            vis[j] = true;
            comp.push_back(j);
            dfs(j, vis, graph, comp);
        }
    }
};
