#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define INF 0x3f3f3f3f
#define maxn 111111
int T,n,m,fa[maxn],a[maxn];
char s[maxn]; 
vector<int>g[maxn],gg[maxn];
void init(int n)
{
    for(int i=0;i<n;i++)g[i].clear(),fa[i]=i,a[i]=0;
}
int find(int x)
{
    if(fa[x]==x)return x;
    return fa[x]=find(fa[x]);
}
void unite(int x,int y)
{
    x=find(x),y=find(y);
    if(x==y)return ;
    fa[x]=y; 
}
class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        n = s.size();
        init(n);
        string rs = "";
        for(auto a: pairs) unite(a[0],a[1]);
        for(int i=0;i<n;i++)g[find(i)].push_back(s[i]);
        for(int i=0;i<n;i++)sort(g[i].begin(),g[i].end());
        for(int i=0;i<n;i++)rs += g[fa[i]][a[fa[i]]++];
        return rs;
    }
};
