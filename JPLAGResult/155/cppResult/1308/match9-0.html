class Solution {
    vector<vector<int>>G;
    vector<bool> v;
    vector<int> res;
    vector<char> vals;
    string s;
    int n;
public:
    string smallestStringWithSwaps(string S, vector<vector<int>>& pairs) {
        
        //res.resize(n);
        s = S;
        n = s.size();
        v.resize(n, 0);
        G.resize(n);
        for (int i = 0; i < pairs.size(); i++) {
            G[pairs[i][0]].push_back(pairs[i][1]);
            G[pairs[i][1]].push_back(pairs[i][0]);
        }
        cout << "here";
        string result = "";
        for (int i = 0; i < n; i++) {
            result.push_back(' ');
        }
        for (int i = 0; i < n; i++) {
            //cout << i <<endl;
            if (!v[i]) {
                //cout << i <<endl;
                res.clear();
                vals.clear();
                dfs(i);
                //cout << i<<" "<<res.size()<<endl;
                sort(res.begin(), res.end());
                sort(vals.begin(), vals.end());
                for (int k = 0; k < res.size(); k++) {
                    result[res[k]] = vals[k];
                }
            }
        }
        return result;
    }
    
    void dfs (int node) {
        //cout << node<<endl;
        v[node] = true;
        for (int i = 0; i < G[node].size(); i++) {
            if (!v[G[node][i]]) 
                dfs(G[node][i]);
        }
       res.push_back(node);
       vals.push_back(s[node]);
    }
};
