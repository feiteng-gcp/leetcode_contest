class Solution {
public:
    int f[100010];
    struct Node{
        int id;
        char ch;
        bool operator<(const Node& rhs)const{
            return ch < rhs.ch;
        }
    };
    vector<Node> a[100010];

    int cnt[100010];
    int Find(int x){
        return x == f[x] ? x : f[x] = Find(f[x]);
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int n = s.length();
        for(int i = 0; i < n; ++i) f[i] = i;
        for(int i = 0; i < n; ++i) cnt[i] = 0;

        for(int i = 0; i < pairs.size(); ++i){
            int t1 = Find(pairs[i][0]);
            int t2 = Find(pairs[i][1]);
            if(t1 != t2){
                f[t1] = t2;
            }
        }
        for(int i = 0; i < n; ++i){
            int t = Find(i);
            a[t].push_back({i, s[i]});
        }
        for(int i = 0; i < n; ++i){
            if(!a[i].empty()) sort(a[i].begin(), a[i].end());
        }
        string ans = "";
        for(int i = 0; i < s.length(); ++i){
            int t = Find(i);
            ans += a[t][ cnt[t] ].ch;
            cnt[t]++;
        }
        return ans;
    }
};
