class Solution {
public:
    vector<int> v[112345];
    int vis[112345];
    vector<int> g;
    

    void dfs(int u) {
        g.push_back(u);
        vis[u] = 1;
        for (int i = 0; i < v[u].size(); i++) {
            if (vis[ v[u][i] ]) continue;
            dfs(v[u][i]);
        }
    }
    
    string smallestStringWithSwaps(string s, vector<vector<int>>& vv) {

        for (int i = 0; i < vv.size(); i++) {
            int a = vv[i][0];
            int b = vv[i][1];
            v[a].push_back(b);
            v[b].push_back(a);
        }
        
        for (int i = 0; i < s.size(); i++) {
            if (vis[i] == 1) continue;
            g.clear();
            dfs(i);
            string ss = "";
            sort(g.begin(), g.end());
            for (int i = 0; i < g.size(); i++)
                ss += s[g[i]];
            sort(ss.begin(), ss.end());
            for (int i = 0; i < g.size(); i++)
                s[ g[i] ] = ss[i];
        }
        return s;
    }
};
