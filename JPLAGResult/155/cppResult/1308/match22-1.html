class Solution {
public:

    int getRank(const vector<int>& ranks, int rank) {
        while (ranks[rank] != rank) rank = ranks[rank];
        return rank;
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        vector<int> ranks(s.size(), 0);
        for (int i = 0; i < s.size(); i++) ranks[i] = i;
        for (auto& p : pairs) {
            ranks[getRank(ranks, p[0])] = getRank(ranks, p[1]);
            ranks[p[0]] = ranks[p[1]] = getRank(ranks, p[1]);
        }
        unordered_map<int, string> dict;
        for (int i = 0; i < ranks.size(); i++) {
            int num = getRank(ranks, i);
            if (!dict.count(num)) dict[num] = "";
            dict[num] += s[i];
        }
        for (auto k : dict) {
            string t = k.second;
            sort(t.begin(), t.end());
            dict[k.first] = t;
            // cout << k.first << ' ' << t << '\n';
        }
        string res = "";
        vector<int> nums(s.size(), 0);

        for (int i = 0; i < ranks.size(); i++) {
            int num = getRank(ranks, i);
            res += dict[num][nums[num]];
            nums[num]++;
        }
        return res;
    }
};
