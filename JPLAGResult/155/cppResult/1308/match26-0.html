class Solution {
public:
    int father[100001];
    
    int find(int x){
        if (father[x]!=x) father[x]=find(father[x]);
        return father[x];  
    }
    
    void unions(int x, int y){
        father[find(x)]=find(y);
    }
    
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int n=s.length();
        int m=pairs.size();
        for (int i=0; i<n; i++) father[i]=i;
        for (int i=0; i<m; i++){
            if (find(pairs[i][0])!=find(pairs[i][1]))
                unions(pairs[i][0], pairs[i][1]);
        }
        vector<vector<char> > groups;
        for (int i=0; i<n; i++){
            vector<char> temp;
            groups.push_back(temp);
        }
        for (int i=0; i<n; i++){
            groups[find(i)].push_back(s[i]);
        }
        for (int i=0; i<n; i++){
            if (!groups[i].empty()){
                sort(groups[i].begin(), groups[i].end());
                reverse(groups[i].begin(), groups[i].end());
            }
        }
        string ans="";
        for (int i=0; i<n; i++){
            ans += groups[find(i)].back();
            groups[find(i)].pop_back();
        }
        return ans;
    }
};
