class Solution {
    
    int getColor(map<int, int> & color, int c) {
        int end = c;
        while (color[end] != end) 
            end = color[end];
        
        int temp = c;
        while (color[temp] != end) {
            int next = color[temp];
            color[temp] = end;
            temp = next;
        }
        
        return end;
    }
    
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        map<int, int> color;
        for (int i=0; i<s.length(); i++)
            color[i] = i;
        
        for (auto v : pairs) {
            int a = getColor(color, v[0]);
            int b = getColor(color, v[1]);
            if (a != b) {
                color[a] = b;
            }
        }
        
        map<int, vector<char>> data;
        for (int i=0; i<s.length(); i++) {
            int c = getColor(color, i);
            if (data.find(c) == data.end())
                data[c] = vector<char>();
            data[c].push_back(s[i]);
        }
        
        for (auto v : data) 
            sort(data[v.first].begin(), data[v.first].end());
        
        string ret = "";
        map<int, int> index;
        for (int i=0; i<s.length(); i++) {
            int c = getColor(color, i);
            ret += data[c][index[c]];
            index[c]++;
        }
        
        return ret;
    }
};
