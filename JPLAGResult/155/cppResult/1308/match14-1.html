class Solution {
public:
    deque<char> C[100005];

    int N, pset[100005];
    int findSet(int x) {return pset[x] == x ? x : pset[x] = findSet(pset[x]); }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int N = s.size();
        for (int i=0;i<N;++i) pset[i] = i;
        for (auto pr : pairs) pset[findSet(pr[0])] = findSet(pr[1]);
        for (int i=0;i<N;++i) C[findSet(i)].push_back(s[i]);
        for (int i=0;i<N;++i) sort(C[i].begin(), C[i].end());
        string ans;

        for (int i=0;i<N;++i) {
            char c = C[findSet(i)].front();
            C[findSet(i)].pop_front();
            ans += c;
        }
        return ans;
    }
};
