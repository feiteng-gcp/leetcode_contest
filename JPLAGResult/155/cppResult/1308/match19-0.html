#define pb push_back
#define ll long long
#define mp make_pair
#define f first
#define s second
#define pii pair < int, int >
#define pll pair < ll, ll >
#define all(s) s.begin(), s.end()
#define sz(s) (int) s.size()
#define forit(it, s) for(__typeof(s.begin()) it = s.begin(); it != s.end(); it++)
#define vi vector < int >

const int inf = (int)1e9;
const int mod = (int) 1e9 + 7;

class Solution {
public:
    vector < int > g[100100];
    multiset < char > S[100100];
    int col[100100];
    int n;
    string s;

    void dfs(int v, int color) {
        col[v] = color;
        S[color].insert(s[v]);
        for (int x : g[v]) {
            if (col[x] == 0) {
                dfs(x, color);
            }
        }
    }

    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        this->s = s;
        int n = s.length();
        for (vector < int > q: pairs) {
            g[q[0]].pb(q[1]);
            g[q[1]].pb(q[0]);
        }
        int cc = 0;
        for (int i=0;i<n;i++) {
            if (col[i] == 0) {
                cc++;
                dfs(i, cc);
            }
        }
        string ans = "";
        for (int i=0;i<s.length();i++) {
            int c = col[i];
            ans += (*S[c].begin());
            S[c].erase(S[c].begin());
        }
        return ans;
    }
};
