int father[100005];
int findset(int x) {
    if(x != father[x]) {
        father[x] = findset(father[x]);
    }
    return father[x];
}
int unionset(int a, int b) {
    int x = findset(a), y = findset(b);

    if(x != y){
        father[y] = x;
    }
    return 0;
}

class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        for (int i = 0; i < s.size(); i++) {
            father[i] = i;
        }
        for (const auto& p: pairs) {
            unionset(p[0], p[1]);
        }
        unordered_map<int, multiset<char>> um;
        for(int i = 0; i < s.size(); i++) {
            um[findset(i)].insert(s[i]);
        }
        string ret;
        for(int i = 0; i < s.size(); i++) {
            auto fa = findset(i);
            auto iter= um[fa].begin();
            ret.push_back(*iter);
            um[fa].erase(iter);
        }
        return ret;
    }
};
