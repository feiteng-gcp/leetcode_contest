class Solution {
public:
    int GetF(int x, int fa[]) { return x == fa[x] ? x : fa[x] = GetF(fa[x], fa); }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int fa[s.length()];
        char res[s.length()+7];
        vector<int> pos[s.length()];
        vector<char> num[s.length()];
        for (int i = 0; i < s.length(); ++i) fa[i] = i;
        for (auto p : pairs) {
            int fx = GetF(p[0], fa), fy = GetF(p[1], fa);
            if (fx != fy) fa[fx] = fy;

        }
        for (int i = 0; i < s.length(); ++i) {
            GetF(i, fa);
            pos[fa[i]].push_back(i);
            num[fa[i]].push_back(s[i]);
        }
        for (int i = 0; i < s.length(); ++i) {
            sort(num[i].begin(), num[i].end());
            for (int j = 0; j < pos[i].size(); ++j) {
                res[pos[i][j]] = num[i][j];
            }
        }
        string ans = "";
        for (int i = 0; i < s.length(); ++i) ans += res[i];
        return ans;
    }
};
