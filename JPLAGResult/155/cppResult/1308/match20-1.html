class Solution {
public:
    void dfs(int u,vector<vector<int> > &g,set<int> &st,vector<bool> &visited)
    {
        visited[u]=true;
        st.insert(u);
        for(int i=0;i<g[u].size();i++)
        {
            if(!visited[g[u][i]])
            {
                //st.push_back(g[u][i]);
                dfs(g[u][i],g,st,visited);
            }
        }
        return ;

    }

    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        vector<vector<int> > g(s.length());
        for(int i=0;i<pairs.size();i++)
        {
            int u=pairs[i][0],v=pairs[i][1];
            g[u].push_back(v);
            g[v].push_back(u);
        }
        vector<bool> visited(s.length(),false);
        vector<set<int> > st;
        //int cnt=0;
        for(int i=0;i<s.length();i++)
        {
            set<int> cst;
            if(!visited[i])
            {
                dfs(i,g,cst,visited);
                st.push_back(cst);
            }
        }
        vector<string> str;
        string ans=s;
        
        for(int i=0;i<st.size();i++)
        {
            string cstr;
            for(auto j: st[i])
                cstr+=s[j];
            sort(cstr.begin(),cstr.end());
            str.push_back(cstr);
        }
        for(int i=0;i<st.size();i++)
        {
            int k=0;
            for(auto j: st[i])
                ans[j]=str[i][k++];
        }
        return ans;
    }
};
