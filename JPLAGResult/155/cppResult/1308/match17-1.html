class Solution {
#define MAX 200005
public:
    int cnt;
    int head[MAX], vet[MAX], Next[MAX];

    void add(int x, int y){
        cnt++;
        Next[cnt] = head[x];
        head[x] = cnt;
        vet[cnt] = y;
    }

    bool vis[MAX];

    vector<int> v, c;
    string str;
    void dfs(int x){
        vis[x] = true;
        v.push_back(x);
        c.push_back(str[x]);
        for(int i = head[x]; i; i = Next[i]){
            int v = vet[i];
            if(vis[v]) continue;
            dfs(v);
        }
    }

    string smallestStringWithSwaps(string s, vector<vector<int>>& p) {
        str = s;
        memset(vis, 0, sizeof(vis));
        for(int i = 0; i < p.size(); i++){
            add(p[i][0], p[i][1]);
            add(p[i][1], p[i][0]);

        }
        for(int i = 0; i < s.length(); i++){
            if(!vis[i]){
                v.clear();
                c.clear();
                dfs(i);
                sort(v.begin(), v.end());
                sort(c.begin(), c.end());
                for(int j = 0; j < v.size(); j++){
                    s[v[j]] = c[j];
                }
            }
        }
        return s;
    }
};
