struct UnionFind{
    vector<int> path;
    UnionFind(int n) {
        path.resize(n);
        for(int i=0; i<n; i++)
            path[i] = i;
    }
    
    int find(int num) {
        while(path[num]!=num) {
            path[num] = path[path[num]];
            num = path[num];
        }
        return num;
    }
    
    void unon(int l, int r) {
        int l_root = find(l);
        int r_root = find(r);
        path[l_root] = r_root;
    }
    
};

class Solution {
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        UnionFind uf(s.size());
        for(auto x:pairs) {
            uf.unon(x[0], x[1]);
        }
        vector<vector<char>> vt(s.size());
        vector<int> count(s.size());
        vector<int> ith(s.size());
        for(int i=0; i<s.size(); i++) {
            ith[i] = count[uf.find(i)]++;
        }
        for(int i=0; i<s.size(); i++) {
            vt[uf.find(i)].push_back(s[i]);
        }
        for(auto & x:vt){
            if(!x.empty())
                sort(x.begin(), x.end());
        }
        string res;
        for(int i=0; i<s.size(); i++) {
            res.push_back(vt[uf.find(i)][ith[i]]);
        }
        return res;
    }
};
