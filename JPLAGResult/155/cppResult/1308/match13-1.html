class Solution {
public:
    vector<int> v;
    int find(int a) {
        if (v[a] != a)    return v[a] = find(v[a]);
        return a;
    }

    void unions(int a, int b) {

        v[find(b)] = find(a);
    }

    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int N = s.size();
        v.resize(N);
        for (int i = 0; i < N; i++) v[i] = i;
        for (auto p : pairs)  unions(p[0], p[1]);
        vector<multiset<char>> group(N);
        for (int i = 0; i < N; i++) {
            group[find(i)].insert(s[i]);
        }
        string ret;

        for (int i = 0; i < N; i++) {
            char cur = *group[find(i)].begin();
            ret += cur;
            group[find(i)].erase(group[find(i)].begin());
        }
        return ret;
    }
};
