class Solution {
public:
    vector<int> father;
    
    int find(int x) {
        if (father[x] != x) father[x] = find(father[x]);
        return father[x];
    }
    
    void merge(int x, int y) {
        father[find(y)] = find(x);
    }
    
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {

        const int n = s.size();
        father.resize(n);
        for (int i = 0; i < n; ++i) father[i] = i;
        for (int i = 0; i < pairs.size(); ++i) {
            merge(pairs[i][0], pairs[i][1]);
        }
        vector<vector<int>> u(n, vector<int>());
        for (int i = 0; i < n; ++i) u[find(i)].push_back(i);
        for (int i = 0; i < n; ++i) {
            if (!u[i].empty()) {
                sort(u[i].begin(), u[i].end());
                string tmp = "";
                for (int j = 0; j < u[i].size(); ++j) tmp += s[u[i][j]];
                sort(tmp.begin(), tmp.end());
                for (int j = 0; j < u[i].size(); ++j) s[u[i][j]] = tmp[j];
            }
        }
        return s;
    }
};
