class Solution {
    vector<vector<int>> e;
    vector<int> vis;
public:
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        e.assign(s.size(), vector<int>());
        vis.assign(s.size(), 0);
        for (auto p : pairs) {
            e[p[0]].push_back(p[1]);
            e[p[1]].push_back(p[0]);
        }
        vector<char> ans;
        ans.assign(s.size(), 0);
        for (int i = 0; i < s.size(); ++i) {
            if (vis[i])
                continue;
            vector<char> ch;
            vector<int> idx;
            vis[i] = 1;
            queue<int> q;
            q.push(i);
            ch.push_back(s[i]);
            idx.push_back(i);

            while (!q.empty()) {
                int cur = q.front(); q.pop();
                for (int to: e[cur]) {
                    if (!vis[to]) {
                        vis[to] = 1;
                        q.push(to);
                        ch.push_back(s[to]);
                        idx.push_back(to);
                     }
                }
            }
            sort(ch.begin(), ch.end());
            sort(idx.begin(), idx.end());
            for (int j = 0; j < idx.size(); ++j) {
                ans[idx[j]] = ch[j];
            }
        }
        return string(ans.begin(), ans.end());
    }
};
