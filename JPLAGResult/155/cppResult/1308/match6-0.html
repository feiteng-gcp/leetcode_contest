class Solution {
public:
    
    void DFS(vector<vector<int>> &adls,vector<bool>&vis,int v, vector<int> & temp){
        vis[v]=true;
        temp.push_back(v);
        for(int i = 0 ; i < adls[v].size(); i++){
            if(vis[adls[v][i]]==false){
                DFS(adls,vis,adls[v][i],temp);
            }
        }
        return;
    }
      
    
    void cc(vector<vector<int>> &v, vector<vector<int> > &p,int n){
        vector<vector<int> > adls(n);
        for(int i = 0 ;i  < p.size();i++){
            adls[p[i][0]].push_back(p[i][1]);
            adls[p[i][1]].push_back(p[i][0]);
        }
        vector<bool> vis(n,false);
        for(int i = 0 ; i < n ; i++){
            if(vis[i]==false){
                vector<int> temp;
                DFS(adls,vis,i,temp);
                sort(temp.begin(),temp.end());
                v.push_back(temp);
            }
        }
    }
    
    string smallestStringWithSwaps(string s, vector<vector<int>>& p) {
        if(p.size()==0 || s.size()<2){
            return s;
        }
        vector<vector<int> > v;
        cc(v,p,s.size());
        for(int i= 0; i < v.size();i++){
            if(v[i].size()>1){
                string temp("");
                for(int j = 0 ; j < v[i].size();j++){
                    temp+=s[v[i][j]];
                }
                sort(temp.begin(),temp.end());
                for(int j = 0 ; j < v[i].size();j++){
                    s[v[i][j]] = temp[j];
                }
            }
        }
        return s;
    }
};
