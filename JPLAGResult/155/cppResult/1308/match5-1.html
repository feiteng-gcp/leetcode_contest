class Solution {
public:
    vector<int> vis;
    vector< vector < int > > g;
    vector<char> cv;
    vector<int> iv;
    string str;

    void dfs(int index)
    {
        vis[index] = 1;
        cv.push_back(str[index]);
        iv.push_back(index);
        for(int i=0;i<g[index].size();i++)
            if(!vis[g[index][i]])
                dfs(g[index][i]);
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        str = s;
        g.clear();
        g.resize(s.size(),{});
        for(int i=0;i<pairs.size();i++)
        {
            g[pairs[i][0]].push_back(pairs[i][1]);
            g[pairs[i][1]].push_back(pairs[i][0]);
        }
        string ans = s;
        vis.clear();
        vis.resize(s.size());

        for(int i=0;i<s.size();i++)
            if(!vis[i])
            {
                cv.clear();
                iv.clear();
                dfs(i);
                sort(iv.begin(),iv.end());
                sort(cv.begin(),cv.end());
                for(int i=0;i<iv.size();i++)
                    ans[iv[i]] = cv[i];
            }
        return ans;
    }
};
