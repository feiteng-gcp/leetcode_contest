class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        sort(arr.begin(), arr.end());
        
        int mn = 1 << 26;
        int n = arr.size();
        for (int i = 0; i + 1 < n; ++i) {
            mn = min(mn, arr[i + 1] - arr[i]);
        }
        
        vector<vector<int>> res;
        
        for (int i = 0; i + 1 < n; ++i) {
            if (arr[i] + mn == arr[i + 1]) {
                res.push_back(vector<int>{arr[i], arr[i + 1]});
            }
        }
        
        return res;
    }
};
