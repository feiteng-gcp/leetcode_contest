class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& A) {
        vector<vector<int>> ans;
        sort(A.begin(), A.end());
        int d = INT_MAX;
        for (int i = 0; i + 1 < A.size(); ++i) {
            d = min(d, A[i+1] - A[i]);
        }
        for (int i = 0; i + 1 < A.size(); ++i) {
            if (d == A[i+1] - A[i]) {
                ans.push_back({A[i], A[i+1]});
            }
        }
        return ans;
    }
};
