class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        sort(arr.begin(), arr.end());
        int m = 9999999;
        for(int i = 1; i < arr.size(); ++i){
            m = min(m, arr[i]-arr[i-1]);
        }
        vector<vector<int>> vec;
        for(int i = 1; i < arr.size(); ++i){
            if(arr[i]-arr[i-1]== m) {
                vec.push_back({arr[i-1], arr[i]});
            }
        }
        return vec;
    }
};
