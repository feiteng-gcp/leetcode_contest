class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        sort(begin(arr),end(arr));
        int mi = INT_MAX;
        int n=arr.size();
        for(int i=0;i<n-1;i++){

            mi = min(mi,arr[i+1]-arr[i]);
        }
        vector<vector<int>> res;
        for(int i=0;i<n-1;i++){
            if(arr[i+1]-arr[i]==mi){
                res.emplace_back(vector<int>{arr[i],arr[i+1]});
            }
        }
        return res;
    }
};
