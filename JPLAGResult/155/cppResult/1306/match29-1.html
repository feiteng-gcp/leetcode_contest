class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        sort(arr.begin(), arr.end());
        
        vector<vector<int>> res;
        int mn = 1e9;
        
        for (int i = 0; i + 1 < arr.size(); ++i) {
            if (arr[i + 1] - arr[i] < mn) {
                res.clear();
                res.push_back({arr[i], arr[i + 1]});
                mn = arr[i + 1] - arr[i];
            } else if (arr[i + 1] - arr[i] == mn) {
                res.push_back({arr[i], arr[i + 1]});
            }
        }
        return res;
    }
};
