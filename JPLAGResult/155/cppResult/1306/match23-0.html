class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& a) {
        vector<vector<int> > ans;
        int minn = 1e9;
        int n = a.size();
        sort(a.begin(), a.end());
        for(int i = 1; i < n; i++) {
            minn = min(minn, a[i] - a[i - 1]);
        }

        for(int i = 1; i < n; i++) {
            if(a[i] - a[i - 1] == minn) {
                vector<int> t;
                t.push_back(a[i - 1]);
                t.push_back(a[i]);
                ans.push_back(t);
            }
        }

        return ans;
    }
};
