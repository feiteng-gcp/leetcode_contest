class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& a) {
        sort(a.begin(), a.end());
        int diff = INT_MAX;
        for (int i=0; i<a.size()-1; i++) diff = min(diff, a[i+1] - a[i]);
        vector<vector<int>> res;
        for (int i=0; i<a.size()-1; i++) {
            if (a[i+1] - a[i] == diff)
                res.push_back({a[i], a[i+1]});
        }
        return res;
    }
};
