class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        sort(arr.begin(), arr.end());
        int minf = arr[1] - arr[0];
        for (int i = 1 ; i < arr.size(); i++) {
            if (minf > arr[i] - arr[i-1])
                minf = arr[i] - arr[i-1];
        }
        vector<vector<int> > ans;
        for (int i = 1 ; i < arr.size(); i++) {
            if (arr[i] - arr[i-1] == minf) {
                vector<int> tmp;
                tmp.push_back(arr[i-1]);
                tmp.push_back(arr[i]);
                ans.push_back(tmp);
            }
        }
        return ans;
    }
};
