class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        int n = arr.size();
        sort(arr.begin(),arr.end());
        int min_diff = 1e9;
        for(int i=0; i<n-1; ++i){

            min_diff = min(min_diff, arr[i+1] - arr[i] );
        }
        vector<vector<int>> ans;
        for(int i=0; i<n-1; ++i){
            if( arr[i+1] - arr[i] == min_diff ){
                vector<int> v = {arr[i], arr[i+1]};
                ans.push_back(v);
            }
        }
        return ans;
    }
};
