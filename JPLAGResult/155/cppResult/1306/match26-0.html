class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        sort(arr.begin(), arr.end());
        int minval = INT_MAX;
        for(int i=0; i<arr.size()-1; i++){
            minval = min(minval, arr[i+1]-arr[i]);
        }
        vector<vector<int> > res;
        for(int i=0; i<arr.size()-1; i++){
            if(arr[i+1]-arr[i]==minval){
                vector<int> t;
                t.push_back(arr[i]);
                t.push_back(arr[i+1]);
                res.push_back(t);
            }
        }
        return res;
    }
};
