class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        vector<vector<int> > ans;
        sort(arr.begin(), arr.end());
        int sz = arr.size();
        int t = 1e9;
        for(int i = 1; i < sz; ++i){
            t = min(t, abs(arr[i] - arr[i-1]));
        }
        for(int i = 1; i < sz; ++i){
            if(abs(arr[i] - arr[i-1]) == t){
                vector<int> res;
                res.push_back(arr[i-1]);
                res.push_back(arr[i]);
                ans.push_back(res);
            }
        }
        return ans;
    }
};
