class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        int N = arr.size(), mn = INT_MAX;
        sort(arr.begin(), arr.end());
        for (int i = 1; i < N; i++) {
            mn = min(mn, arr[i] - arr[i - 1]);
        }
        vector<vector<int>> ret;
        for (int i = 1; i < N; i++) {
            if (arr[i] - arr[i - 1] == mn) {
                ret.push_back({arr[i - 1], arr[i]});
            }
        }
        return ret;
    }
};
