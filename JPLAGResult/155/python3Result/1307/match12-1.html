class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        
        
def gcd(x, y):
            if x % y == 0: return y
            else: return gcd(y, x % y)
        
        def lcm(x, y):
            return x * y // gcd(x, y)
        
        lcms = [lcm(a, b), lcm(b, c), lcm(c, a)]
        lcmabs = lcm(a, lcm(b, c))
        
        def count(x):
            ans = 0
            for y in [a, b, c]: ans += x // y
            for y in lcms: ans -= x // y
            ans += x // lcmabs
            return ans
        
        l, r 
= 1, 2 * (10 ** 9) + 1
        while l < r:
            m = (l + r) // 2
            if count(m) < n:
                l = m + 1
            else:
                r = m
        return l
