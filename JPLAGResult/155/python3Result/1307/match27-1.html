from math import gcd

class Solution:

    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        gcd_a_b = gcd(a, b)
        gcd_b_c = gcd(b, c)
        gcd_a_c = gcd(a, c)
        gcd_a_b_c = gcd(a, gcd_b_c)
        lcm_a_b = a*b//gcd_a_b
        lcm_b_c = b*c//gcd_b_c
        lcm_a_c = a*c//gcd_a_c
        lcm_a_b_c = lcm_a_b*c//gcd(lcm_a_b, c)
        def count_divisible(num):
            cnt = num//a + num// b + num//c
            cnt = cnt - num//lcm_a_b - num//lcm_b_c - num//lcm_a_c  + num//lcm_a_b_c
            #print(num, cnt)
            return cnt
        #print(lcm_a_b, lcm_b_c, lcm_a_c, lcm_a_b_c)
        left = 1
        right = 2*10**9
        while left <= right:
            mid = left + (right - left)//2
            cnt = count_divisible(mid)
            if cnt >= n:
                right = mid - 1
            else:
                left = mid + 1
        return left
