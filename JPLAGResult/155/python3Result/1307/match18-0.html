def gcd(x, y):
    return x if not y else gcd(y, x%y)


def lcm(x, y):
    return x*y//gcd(x, y)


class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def number(n, a, b, c):
            return n//a+n//b+n//c-n//(lcm(a,b))-n//(lcm(a,c))-n//(lcm(c,b))+n//(lcm(a,lcm(b,c)))
        l = 1
        r = 2 * 10**9
        while l < r - 1:
            mid = (l+r)//2
            if number(mid, a, b, c) < n:
                l = mid + 1
            else:
                r = mid
        while number(l, a, b, c) < n:
            l += 1
        return l
