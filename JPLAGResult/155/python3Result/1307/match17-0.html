class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        
        def gcd(a_, b_):
            if b_ == 0:
                return a_
            
            return gcd(b_, a_ % b_)
        
        def lcm(a_, b_):
            return a_ * b_ // gcd(a_, b_)
        
        ab = lcm(a, b)
        ac = lcm(a, c)
        bc = lcm(b, c)
        abc = lcm(ab, c)
        
        def ugly_cnt(k):
            return k // a + k // b + k //c - k // ab - k // bc - k // ac + k // abc
        
        l = 0
        r = 2 * 10 ** 9 + 1
        
        while r - l > 1:
            m = (r + l) // 2
            
            if ugly_cnt(m) < n:
                l = m
            else:
                r = m

        return r
