class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def gcd(a, b):
            if a < b:
                return gcd(b, a)
            if b == 0:
                return a
            return gcd(b, a % b)
        def howMany(x, a, b, c, e, f, g, h):
            return x // a + x // b + x // c - x // e - x // f - x // g + x // h
            
        arr = [a, b, c]
        arr.sort()
        a, b, c = arr
        e 
= a * b // gcd(a, b)
        f = b * c // gcd(b, c)
        g = c * a // gcd(a, c)
        h = e * c // gcd(e, c)
        left = 0
        right = a * n
        while left < right - 1:
            mid = (left + right) // 2
            if howMany(mid, a, b, c, e, f, g, h) >= n:
                right = mid
            else:
                left = mid
        return right
