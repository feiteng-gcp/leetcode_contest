import math
class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def lcm(a, b):
            return abs(a*b) // math.gcd(a, b)
        def count(num):
            return num // a + num // b + num // c - num // lcm(a, b) - num // lcm(a,c) - num // lcm(b, c) + num // lcm(lcm(a, b), lcm(b, c))
        start = 1
        end = 2 * 10**9 + 1
        while start != end:
            mid = (start + end) // 2
            if count(mid) < n: start = mid + 1
            else: end = mid
        return start
            
    
    def nthUglyNumber2(self, n: int, a: int, b: int, c: int) -> int:
        ia, ib, ic = 1, 1, 1
        for i in range(n-1):
            mi = min(a*ia, b * ib, c * ic)
            if a * ia == mi: ia += 1
            if b * ib == mi: ib += 1
            if c * ic == mi: ic += 1
        return min(a*ia, b * ib, c * ic)
