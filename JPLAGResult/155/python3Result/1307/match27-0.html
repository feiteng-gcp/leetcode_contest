def gcd(x,y):
    if y==0: return x
    return gcd(y,x%y)
class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        gab=gcd(a,b)
        gac=gcd(a,c)
        gbc=gcd(b,c)
        gabc=gcd(gab,c)
        ab=a*b//gab
        ac=a*c//gac
        bc=b*c//gbc
        abc=ab*c//gcd(ab,c)
        def work(x):
            ca=x//a
            cb=x//b
            cc=x//c
            cab=x//ab
            cbc=x//bc
            cac=x//ac
            cabc=x//abc
            return ca+cb+cc-cab-cbc-cac+cabc
        L=0
        R=20000000000
        while L+1<R:
            M=(L+R)//2
            if work(M)<n:
                L=M
            else:
                R=M
        return R
