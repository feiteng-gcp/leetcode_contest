from math import *
class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        A = [a, b, c]
        A.sort()
        LCM0 = a * b // gcd(a, b)
        LCM1 = a * c // gcd(a, c)
        LCM2 = b * c // gcd(b, c)
        LCM4 = LCM0 * LCM1 // gcd(LCM0, LCM1)

        # print(LCM0, LCM1, LCM2, LCM4)

        left = 1
        right = 2 * 10 ** 9
        while left < right:
            mid = left + (right - left) // 2
            if mid // a + mid // b + mid // c - mid // LCM0 - mid // LCM1 - mid // LCM2 + mid // LCM4 < n:
                left = mid + 1
            else:
                right = mid
        return right
