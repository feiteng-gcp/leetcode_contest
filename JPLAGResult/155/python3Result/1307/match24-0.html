class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def gcd(a, b):
            return a if b == 0 else gcd(b, a % b)

        def lcm(a, b):
            return a * b / gcd(a, b)
        ab = lcm(a, b)
        ac = lcm(a, c)
        bc = lcm(b, c)
        abc = lcm(ab, c)
        def check(num: int) -> (int, int):
            # ma = num % a
            # mb = num % b
            # mc = num % c
            # num -= min(ma, mb, mc)
            idx = ((num // a) + (num // b) + (num // c) 
                   - (num // ab) - (num // ac) - (num // bc)
                   + (num // abc))
            return idx
        
        def trim(num: int) -> int:
            ma = num % a
            mb = num % b
            mc = num % c
            num -= min(ma, mb, mc)
            return num
        
        l = 0
        r = 2000000000
        while l <= r:
            mid = (l + r) // 2
            i = check(mid)
            if i == n:
                return trim(mid)
            elif i < n:
                l = mid + 1
            else:
                r = mid
