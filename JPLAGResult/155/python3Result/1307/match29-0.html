class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        [a,b,c]=sorted([a,b,c])
        def lcm(a,b):
            big,smal = b,a
            while big%smal:
                big,smal=smal,big%smal
            gcd=smal
            return b*a//gcd
            
        
        def count(num):
            d,e,f=lcm(a,b),lcm(b,c),lcm(a,c)
            g=lcm(d,c)
            return num//a+num//b+num//c-num//d-num//e-num//f+num//g
        
        lo,hi=1,2*10**9
        while lo<hi:
            m=(lo+hi)//2
            if count(m)<n:
                lo=m+1
            else:
                hi=m
        return lo
