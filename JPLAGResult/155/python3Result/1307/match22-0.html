class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def gcd(x, y):
            while y > 0:
                x, y = y, x % y
            return x
        def lcm(x, y):
            return (x * y) // gcd(x, y)
        
        ab = lcm(a, b)
        bc = lcm(b, c)
        ac = lcm(a, c)
        abc = lcm(ab, c)
        minn = min(a, b, c)
        
        #print(ab, bc, ac, abc)
        
        def how_many(k):
            return k // a + k // b + k // c - k // ab - k // bc - k // ac + k // abc
        
        count = 0
        
        low, high = 1, 2 * (10 ** 9)
        while low <= high:
            # print(low, high)
            mid = (low + high) // 2
            if how_many(mid) == n:
                count = mid
                break
            elif how_many(mid) > n:
                high = mid - 1
            else:
                low = mid + 1
        
        while count > 0 and how_many(count-1) == n:
            count -= 1
        return count
