import math


class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def lcm(x, y):
            return (x * y) // math.gcd(x, y)

        ab = lcm(a, b)
        ac = lcm(a, c)
        bc = lcm(b, c)
        abc = lcm(ab, c)

        def calc(k):
            return k // a + k // b + k // c - k // ab - k // ac - k // bc + k // abc

        left = 0
        right = n * a
        while True:
            m = (left + right) // 2
            s = calc(m)
            if right - left < 2 and s == n:
                break
            elif s >= n:
                right = m
            else:
                left = m + 1
        return left
