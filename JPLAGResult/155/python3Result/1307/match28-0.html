from fractions import gcd

class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def lcm(x, y):
            return (x * y) // gcd(x, y)
        
        def count(x):
            return x // a + x // b + x // c - x // lcm(a, b) - x // lcm(b, c) - x // lcm(a, c) + x // lcm(lcm(a, b), c)
        
        left = 1
        right = 2 * 1000000000
        while left <= right:
            mid = (left + right) // 2
            cnt = count(mid)
            if cnt == n:
                x1 = mid % a
                x2 = mid % b
                x3 = mid % c
                return mid - min(x1, x2, x3)
            elif cnt < n:
                left = mid + 1
            else:
                right = mid - 1
        return left
