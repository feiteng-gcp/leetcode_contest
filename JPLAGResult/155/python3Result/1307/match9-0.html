class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        
        def gcd(a, b):
            if a < b:
                return gcd(b, a)
            if b == 0:
                return a
            return gcd(b, a % b)
        
        def lcm(a, b):
            return a * b // gcd(a, b)
        
        def lt(x):
            return x // a + x // b + x // c - x // d - x // e - x // f + x // g
        
        a, b, c = sorted([a, b, c])
        d, e, f = lcm(a, b), lcm(b, c), lcm(c, a)
        g = lcm(d, c)
        ib = a
        ie = 2 * 10 ** 9
        while ib <= ie:
            m = (ib + ie) // 2
            if lt(m) < n:
                ib = m + 1
            else:
                ie = m - 1
        return ib
