class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def gcd(x,y):
            while y!=0:
                x,y=y,x%y
            return x
        def lcm(x,y):
            return x*y/gcd(x,y)
        def cnt(x:int)->int:
            # 1..x , divided by a
            cx = x//a+x//b+x//c-x//lcm(a,b)-x//lcm(b,c)-x//lcm(a,c)+x//lcm(a,lcm(b,c))
            return cx

        L = 1
        R = 2*(10**9)
        while L<R:
            mid = (L+R)//2
            if(cnt(mid)<n):
                L = mid+1
            else:
                R=mid
        return L
