class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def gcd(a, b):
            if a < b: return gcd(b, a)
            if a % b == 0: return b
            return gcd(b, a % b)
        def lcm(a, b):
            return a * b // gcd(a, b)
        
        lcm_ab = lcm(a, b)
        lcm_bc = lcm(b, c)
        lcm_ca = lcm(c, a)
        
        lcm_3 = lcm(lcm(a,b), c)
        
        l, r = 1, 10**18
        while l < r:
            mid = l + (r-l) // 2
            
            x = mid // a + mid // b + mid // c - mid // lcm_ab - mid // lcm_bc - mid // lcm_ca + mid // lcm_3
            # print (l, r, mid, mid // a, mid // b, mid // c, mid // lcm_ab, mid // lcm_bc, mid // lcm_ca, mid // lcm_3)
            if x == n:
                if 0 in [mid % a, mid % b, mid % c]:
                    return mid
                r = mid
            elif x < n:
                l = mid + 1
            else:
                r = mid
                
        return l
