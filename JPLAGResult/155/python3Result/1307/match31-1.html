class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        from fractions import gcd
        
        def lcm(a, b):
            return a // gcd(a, b) * b
        
        def calc(m):
            return m // a + m // b + m // c - m // lcm(a, b) - m // lcm(a, c) - m // lcm(b, c) + m // lcm(lcm(a, b), c)
        
        l 
= 1
        r = 2000000000
        while l < r:
            m = (l + r) // 2
            if calc(m) >= n:
                r = m
            else:
                l = m + 1
        return l
