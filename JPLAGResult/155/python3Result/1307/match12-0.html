class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
        def cnt(n):
            t1=n//a+n//b+n//c
            t1-=n//ab+n//ac+n//bc-n//abc
            return t1
        
        def gcd(x,y):
            if x%y==0:
                return y
            else:
                return gcd(y,x%y)
            
        def lcd(x,y):
            return (x*y)//gcd(x,y)
        
        a,b,c=sorted([a,b,c])
        ab=lcd(b,a)
        ac=lcd(c,a)
        bc=lcd(c,b)
        abc=lcd(max(ab,c),min(ab,c))
        
        left=1
        right=2*(10**9)+1
        t=n
        while left<right:
            mid=(left+right)>>1
            if cnt(mid)<t:
                left=mid+1
            else:
                right=mid
        return left
