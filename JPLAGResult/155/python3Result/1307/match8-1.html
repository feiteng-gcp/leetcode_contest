class Solution:
    def nthUglyNumber(self, n: int, a: int, b: int, c: int) -> int:
 
        def lcm(a, b): 
            return ((a * b) // math
.gcd(a, b)) 
        
        def helper(aa, bb, cc, num): 
            return ((num // aa) + (num // bb) + (num // cc) 
                    - (num // lcm(aa, bb)) 
                    - (num // lcm(bb, cc)) 
                    - (num // lcm(aa, cc)) 
                    + (num // lcm(aa, lcm(bb, cc))))
        l, h 
= 1, 2*(10**9)+1
        while l < h:
            mid = (l+h) // 2
            if (helper(a, b, c, mid) < n):
                l = mid+1
            else:
                h = mid
        return l
