class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr.sort()
        res = []
        min_d = None
        for i in range(len(arr) - 1):
            d = arr[i + 1] - arr[i]
            if min_d is None or d < min_d:
                min_d = d
                res = [[arr[i], arr[i + 1]]]
            elif d == min_d:
                res.append([arr[i], arr[i + 1]])
        return res
