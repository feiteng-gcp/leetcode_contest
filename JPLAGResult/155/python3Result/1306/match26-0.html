class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr.sort()
        ans = list()
        min_abs = 10**9
        for i in range(len(arr) - 1):
            if arr[i + 1] - arr[i] < min_abs:
                min_abs = arr[i + 1] - arr[i]
                ans = [[arr[i], arr[i + 1]]]
            elif arr[i + 1] - arr[i] == min_abs:
                ans.append([arr[i], arr[i + 1]])
        return ans
