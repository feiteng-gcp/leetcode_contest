class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr.sort()
        rez = []
        mn = 10 ** 7
        for i in range(len(arr) - 1):
            if arr[i+1] - arr[i] < mn:
                mn = arr[i+1] - arr[i]
                rez = [[arr
[i], arr[i+1]]]
            elif arr[i+1] - arr[i] == mn:
                rez.append([arr[i], arr[i+1]])
        return rez
