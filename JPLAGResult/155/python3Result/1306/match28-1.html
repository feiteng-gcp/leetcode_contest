class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr.sort()
        n = len(arr)
        mi = float('inf')
        for i in range(n-1):
            mi = min(mi, arr[i+1] - arr[i])
        ans = []
        for i in range(n - 1):
            if arr
[i+1] - arr[i] == mi: ans.append([arr[i], arr[i+1]])
        return ans
