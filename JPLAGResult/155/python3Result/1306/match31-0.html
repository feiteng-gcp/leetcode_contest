class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        minval=10000000
        arr.sort();pairs=[]
        
        for i in range(len(arr)-1):
            minval=min(minval,arr[i+1]-arr[i])
        for i in range(len(arr)-1):
            if minval==arr[i+1]-arr[i]:
                pairs.append([arr[i],arr[i+1]])
        return pairs
