class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr = sorted(arr)
        result = []
        mini = None
        for i in range(1, len(arr)):
            if not mini or arr[i] - arr[i - 1] < mini:
                mini = arr[i] - arr[i - 1]
                result = [[arr
[i - 1], arr[i]]]
            elif arr[i] - arr[i - 1] == mini:
                result.append([arr[i - 1], arr[i]])
        return result
