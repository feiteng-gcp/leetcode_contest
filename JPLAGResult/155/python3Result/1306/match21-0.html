class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr = sorted(arr)
        diff = 3000000
        ans = []
        for i in range(1, len(arr)):
            if arr[i] - arr[i - 1] < diff:
                ans = [[arr[i - 1], arr[i]]]
                diff = arr[i] - arr[i - 1]
            elif arr[i] - arr[i - 1] == diff:
                ans.append([arr[i - 1], arr[i]])
                
        return ans
