class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr.sort()
        min_dis = float('inf')
        for i in range(1,len(arr)):
            min_dis = min(min_dis,arr[i]-arr[i-1])
        ans = []
        for i in range(1,len(arr)):
            if arr[i]-arr[i-1] == min_dis:
                ans.append([arr[i-1],arr[i]])
        return ans
