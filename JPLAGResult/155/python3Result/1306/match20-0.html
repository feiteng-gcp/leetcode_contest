class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr.sort()
        
        ans = []
        mindiff = arr[-1] - arr[0]
        for i in range(len(arr) - 1):
            if arr[i + 1] - arr[i] < mindiff:
                mindiff = arr[i + 1] - arr[i]
                ans = [[arr[i], arr[i + 1]]]
            elif arr[i + 1] - arr[i] == mindiff:
                ans.append([arr[i], arr[i + 1]])
        
        return ans
