from math import inf
class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        min_abs = inf
        arr.sort
()
        for i in range(1, len(arr)):
            if arr[i] - arr[i - 1] < min_abs:
                min_abs = arr[i] - arr[i - 1]
        ans = []
        for i in range(1, len(arr)):
            if arr[i] - arr[i - 1] == min_abs:
                ans.append([arr[i - 1], arr[i]])
        return ans
