class Solution:
    def minimumAbsDifference(self, arr: List[int]) -> List[List[int]]:
        arr.sort()
        m = float("inf")
        for idx, val in enumerate(arr):
            if idx:
                m = min(m, val - arr[idx - 1])
        rs = []
        for idx, val in enumerate(arr):
            if idx and val - arr[idx - 1] == m:
                rs.append([arr[idx - 1], val])
        return rs
