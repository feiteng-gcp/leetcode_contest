from typing import Tuple, List

def topological_sort(graph: List[List[int]]) -> Tuple[bool, List[int]]:
    '''
        graph: directed graph, graphe[u] = [v where u -> v]
        visited[u]: 0 not visited, 1 visiting, 2 visited
    '''

    n, is_possible = len(graph), True
    visited, sorted_array = [0] * n, []
    
    def dfs(u):
        nonlocal is_possible
        
        if not is_possible: return
        if visited[u]:
            is_possible = visited[u] == 2
            return
        
        visited[u] = 1
        for v in graph[u]: dfs(v)
        visited[u] = 2
        sorted_array.append(u)
    
    for u in range(n): dfs(u)
    return (is_possible, sorted_array if is_possible else [])

class Solution:
    def sortItems(self, nNode: int, mGroup: int, g: List[int], e: List[List[int]]) -> List[int]:
        n = mGroup
        for i in range(nNode):
            if g[i] == -1:
                g[i] = n
                n += 1
        count = [0] * n
        subg = [0] * nNode
        inv_g = dict()
        for i in range(nNode):
            subg[i] = count[g[i]]
            count[g[i]] += 1
            inv_g[(g[i], subg[i])] = i
        subGraph = []
        for i in range(n):
            subGraph.append([set() for _ in range(count[i])])
        
        graph = [set() for _ in range(n)]
        for i in range(nNode):
            u = g[i]
            for j in e[i]:
                if u != g[j]:
                    graph[u].add(g
[j])
                else:
                    subGraph[u][subg[i]].add(subg[j])
        
        is_possible, arr = topological_sort(graph)
        if not is_possible: return []
        
        ret = []
        for i in arr:
            is_possible, temp = topological_sort(subGraph[i])
            if not is_possible: return []
            for u in temp: ret.append(inv_g[(i, u)])

        return ret
