class Solution:
    def withinGroup(self, before):
        #print("Toposort",before)
        deg = {k:len(v) for k,v in before.items()}
        after = {k:[] for k in before}
        
        for k,xs in before.items():
            for x in xs: after.setdefault(x,[]).append(k)
        
        free = [k for k,v in deg.items() if v==0]
        R=[]
        
        while len(free):
            u=free[-1]
            del free[-1]
            R
.append(u)
            for x in after[u]:
                deg[x]-=1
                if deg[x]==0: free.append(x)
        
        if any(v>0 for k,v in deg.items()): return None
        else: return R
    
    def sortItems(self, n: int, m: int, group: List[int], beforeItems: List[List[int]]) -> List[int]:
        #now we toposort on groups
        #first, construct group graph
        
        deg=[0 for i in range(n)]
        G=[[] for i in range(m)]
        
        for k,v in enumerate(group):
            if v>=0: G[v].append(k)
            else:
                group[k]=len(G)
                G.append([k])
        
        m=len(G)
        beforeGroup = [set() for i in range(m)]
        for a,xs in enumerate(beforeItems):
            for b in xs:
                #a requires b
                u,v=group[a], group[b]
                if u!=v:
                    beforeGroup[u].add(v)
                    #beforeGroup[v].add(u)
        
        #print(group)
        #print(G)

        beforeGroup = {k:v for k,v in enumerate(beforeGroup)}
        #print(beforeGroup)
        topo = self.withinGroup(beforeGroup)
        if topo is None: return []
        #print(">",topo)
        
        R=[]
        #toposort each group
        for g in topo:
            before = {a:[] for a in G[g]}
            for a in G[g]:
                for b in beforeItems[a]:
                    if group[a]==group[b]:
                        before.setdefault(a,[]).append(b)
            #print(g, before)
            Z = self.withinGroup(before)
            if Z is None: return []
            R.extend(Z)
        
        return R
