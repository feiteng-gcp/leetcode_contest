from collections import defaultdict
class Solution:
    def sortItems(self, n: int, m: int, group: List[int], beforeItems: List[List[int]]) -> List[int]:
        gson_tree = defaultdict(set) # 不同小组前置关系
        gfa_tree = defaultdict(set)
        ison_tree = defaultdict(set) # 同小组 项目前置关系
        ifa_tree = defaultdict(set)
        group_tree = defaultdict(list) 
  
        for index in range(0,n):
            if group[index]==-1:
                group[index] = 90000 + index
            x = group[index]
            group_tree[x].append(index)
            
        for index,info in enumerate(zip(group,beforeItems)):
            gid,bef_list = info
            bef_gid_list = [group[x] for x in bef_list]
            bef_gid_set = set(bef_gid_list)
            for bgid in bef_gid_set:
                if bgid!=gid :
                    gson_tree[gid].add(bgid)
                    gfa_tree[bgid].add(gid)
                    
            for x in bef_list:
                if group[x]==gid:
                    ison_tree[index].add(x)
                    ifa_tree[x].add(index)
                    
        can_list = []
        for gid in set(group):
            if gid not in gson_tree:
                can_list.append(gid)
#         print(gson_tree)
#         print(gfa_tree)
        
#         print("")
#         print(ison_tree)
#         print(ifa_tree)
        i = 0
        print(can_list)
        while i<len(can_list):
            gid = can_list[i]
            if gid in gfa_tree:
                
for sgid in gfa_tree[gid]:
                    gson_tree[sgid].remove(gid)
                    if len(gson_tree[sgid])==0:
                        can_list.append(sgid)
            i+= 1
        print(can_list)
        if len(can_list)!=len(set(group)):
            return []
    
        res_list = []
        for gid in can_list:
            item_list = group_tree[gid]
            
            ican_list = []
            for item in item_list:
                if item not in ison_tree:
                    ican_list.append(item)
            print(gid,ican_list)
            i = 0
            while i < len(ican_list):
                item = ican_list[i]
                if item in ifa_tree:
                    for sitem in ifa_tree[item]:
                        ison_tree[sitem].remove(item)
                        if len(ison_tree[sitem])==0:
                            ican_list.append(sitem)
                i += 1
            if len(ican_list)!=len(item_list):
                return []
            else:
                res_list.extend(ican_list)
        return res_list
