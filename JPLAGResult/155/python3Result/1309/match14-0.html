from itertools import *
from collections import deque
class Solution:
    def sortItems(self, n: int, m: int, group: List[int], beforeItems: List[List[int]]) -> List[int]:
        gps = [[] for _ in range(m)]
        
        for i, g in enumerate(group):
            if g == -1:
                group[i] = len(gps)
                gps.append([i])
            else:
                gps[g].append(i)
        
        links = [[] for _ in range(n)]
        glinks = [set() for _ in range(len(gps))]
        incnt = [0 for _ in range(n)]
        gincnt = [0 for _  in range(len(gps))]
        for i, l in enumerate(beforeItems):
            for j in l:
                links[j].append(i)
                incnt[i] += 1
                if group[j] != group[i]:
                    if group[i] not in glinks[group[j]]:
                        gincnt[group[i]] += 1
                        
                    glinks[group[j]].add(group[i])
                
        for i in range(len(gps)):
            glinks[i] = list(glinks[i])
        
        q = deque()
        for i in range(n):
            # print(i, incnt[i], group[i], gincnt[group[i]])
            if incnt[i] == 0 and gincnt[group[i]] == 0:
                q.append(i)
        # print(incnt, gincnt)
        gpos = {}
        ret = []
        while q:
            # print(q)
            i = q.popleft()
            if group[i] not in gpos:
                gpos[group[i]] = len(ret)
                ret.append([i])
            else:
                ret[gpos[group[i]]].append(i)
                
            for j in links[i]:
                incnt[j] -= 1
            
            nxt = set()
            if len(ret[gpos[group[i]]]) == len(gps[group[i]]):
                # print('hihihi', i, group[i])
                for gj in glinks[group[i]]:
                    gincnt[gj] -= 1
                    if gincnt[gj] == 0:
                        for j in gps[gj]:
                            if incnt[j] == 0 and gincnt[group[j]] == 0:
                                nxt.add(j)

            for j in links[i]:
                if incnt[j] == 0 and gincnt[group[j]] == 0:
                    nxt.add(j)
            q += list(nxt)
        # print(incnt, gincnt) 
        ans = list(chain.from_iterable(ret))
        # print(ans)
        if len(ans) == n:
            return ans
        else:
            return []
