class Solution:
    def sortItems(self, n: int, m: int, group: List[int], beforeItems: List[List[int]]) -> List[int]:
        pm=[[] for i in range(m)]
        for i in range(n):
            if group[i]>=0:
                pm[group[i]].append(i)
        for i in range(m):
            dic={}
            dic.clear()
            dicnext={}
            
            for x in pm[i]:
                dicnext[x]=[]
                dic[x]=0
            for x in pm[i]:
                for y in beforeItems[x]:
                    if group[y]==i:
                        dic[x]+=1
                        dicnext[y].append(x)
                    
            tg=[]
            anspm=[]
            for x in pm[i]:
                if dic[x]==0:
                    tg.append(x)
                    dic[x]=-1
            st=0
            end=len(tg)
            while(st<end):
                x=tg[st]
                anspm.append(x)
                st=st+1
                for y in dicnext[x]:
                    dic[y]=dic[y]-1
                    if dic[y]==0:
                        tg.append(y)
                        end=end+1
                        dic[y]=-1
            if len(anspm)<len(pm[i]):
                return []
            for t in range(len(pm[i])):
                pm[i][t]=anspm[t]
        
        L=len(pm)
        for i in range(n):
            if group[i]==-1:
                pm.append([i])
                group[i]=L
                L=L+1
                
        dic={}
        dicnext={}
        for i in range(len(pm)):
            dicnext[i]={}
            dic[i]=0
        for i in range(len(pm)):
            for x in pm[i]:
                for y in beforeItems[x]:
                    if not group[x]==group[y]:
                        
                        if not (group[x] in dicnext[group[y]]):
                            dicnext[group[y]][group[x]]=1
                            dic[group[x]]+=1
                            
        tg=[]
        ans=[]
        for i in range(len(pm)):
            if dic[i]==0:
                tg.append(i)
                dic[i]=-1
        st=0
        end=len(tg)
        print(pm)
        while(st<end):
            x=tg[st]
            for xx in pm[x]:
                ans.append(xx)
            st=st+1
            for y in dicnext[x]:
                dic[y]=dic[y]-1
                if dic[y]==0:
                    tg.append(y)
                    end=end+1
                    dic[y]=-1
            # print(tg)
            # print(ans)
            # print(st)
            # print(end)
            # print(dic[2])
            # print('\n')
        if len(ans)<n:
            return []
        
        return ans
