class Solution:
    def topoSort(self, dependent):
        queue = []
        for n, (dep, cnt) in dependent.items():
            
if cnt is 0:
                queue.append(n)
        result = []
        while queue:
            q = queue.pop(0)
            result.append(q)
            for n in dependent[q][0]:
                dependent[n][1] -= 1
                if dependent[n][1] is 0:
                    queue.append(n)
        return result
        
    
    def sortItems(self, n: int, m: int, group: List[int], beforeItems: List[List[int]]) -> List[int]:
        group_dict = {}
        result = []
        for i in range(n):
            if group[i] is -1:
                if not beforeItems[i]:
                    result.append(i)
                    continue
                else:
                    group[i] = m
                    m += 1
            g = group[i]
            if g not in group_dict:
                group_dict[g] = [set(), set()]
            group_dict[g][0].add(i)
            group_dict[g][1].update(beforeItems[i])
        
        group_dependent = {k:[set(), 0] for k in group_dict}
        for g, (items, befores) in group_dict.items():
            # sort items
            item_dependent = {k:[set(), 0] for k in items}
            for i in items:
                depends = set(b for b in beforeItems[i] if group[b] == g)
                item_dependent[i][1] = len(depends)
                for d in depends:
                    item_dependent[d][0].add(i)
            item_sorted = self.topoSort(item_dependent)
            if len(item_sorted) != len(items):
                return []
            group_dict[g][0] = item_sorted
            
            # make group depend
            depends = set(group[b] for b in befores)
            depends.discard(-1)
            depends.discard(g)
            group_dependent[g][1] = len(depends)
            for d in depends:
                group_dependent[d][0].add(g)
        
        group_sorted = self.topoSort(group_dependent)
        if len(group_sorted) != len(group_dependent):
            return []
        
        for g in group_sorted:
            result.extend(group_dict[g][0])
            
        return result
