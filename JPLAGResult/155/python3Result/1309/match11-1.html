class Solution:
    def sortItems(self, n: int, m: int, group: List[int], b: List[List[int]]) -> List[int]:
        
        def topo(a, b):
            l = len(a.keys())
            cand = set(i for i, j in b.items() if len(j) == 0)
            ret = []
            while cand:
                p = cand.pop()
                ret.append(p)
                for q in a[p]:
                    bq = b[q]
                    bq.discard(p)
                    if len(bq) == 0:
                        cand.add(q)
            return ret if all(len(b[q]) == 0 for q in b) else None

        bi, bg = [collections.defaultdict(set) for i in range(2)]
        for i in range(n):
            gi = group[i]
            bgi = bg[gi]

                if group[j] == gi:
