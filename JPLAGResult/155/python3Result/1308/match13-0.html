class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        p = [i for i in range(len(s))]
        
        def find(a):
            if not p[a] == a:
                p[a] = find(p[a])
            return p[a]
            
        def union(a, b):
            a_p = find(a)
            b_p = find(b)
            
            if a_p < b_p:
                b_p, a_p = a_p, b_p
            p[a_p] = b_p
            
        for a, b in pairs:
            union(a, b)
        
        c = collections.defaultdict(set)
        for i in range(len(s)):
            x = find(i)
            c[x].add(i)
            
        res = list(s)
        #print(c)
        for k in c:
            a = list(c[k])
            a.sort()
            cur = [s[i] for i in a]
            #print(cur)
            cur.sort()
            for i, r in zip(a, cur):
                res[i] = r
        
        
        return "".join(res)
