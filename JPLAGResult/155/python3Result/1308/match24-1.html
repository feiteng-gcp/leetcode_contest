class UnionFind:
    def __init__(self):
        self.A={}
    def find(self,a):
        if a not in self.A: self.A[a]=a
        while self.A[a] != a:
            u=self.A[self.A[a]]
            self.A[a]=u
            a=u
        
return a
    def merge(self,a,b):
        a=self.find(a)
        b=self.find(b)
        self.A[b] = self.A[a]

class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        A=UnionFind()
        for a,b in pairs:
            A.merge(a,b)
        comps={}
        for a in A.A:
            comps.setdefault(A
.find(a), []).append(a)
        
        R=list(s)
        for k,xs in comps.items():
            xs.sort()
            ch=sorted(s[i] for i in xs)
            B={b:a for a,b in enumerate(xs)}
            for i in xs:
                R[i]=ch[B[i]]
        return "".join(R)
