class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
    
    def get_root(self, i):
        if self.parent[i] != self.parent[self
.parent[i]]:
            self.parent[i] = self.get_root(self.parent[i])
        return self.parent[i]
    
    def is_connected(self, i, j):
        return self.get_root(i) == self.get_root(j)
    
    def union(self, i, j):
        i_root = self.get_root(i)
        j_root = self.get_root(j)
        self.parent[i_root] =
 j_root

class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        uf = UnionFind(len(s))
        for a, b in pairs:
            uf.union(a, b)
        
        ps = {}
        for i in range(len(s)):
            ps.setdefault(uf.get_root(i), set()).add(i)
        
        sorted_str = {}
        for k, ss in ps.items():
            l = [s[i] for i in ss]
            l.sort()
            sorted_str[k] = ''.join(l)
        used_cnt = {}
        result = ''
        for i in range(len(s)):
            c_root = uf.get_root(i)
            c_used_cnt = used_cnt.get(c_root, 0)
            result += sorted_str[c_root][c_used_cnt]
            used_cnt[c_root] = c_used_cnt + 1
        return result
