class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        edges
=collections.defaultdict(set)
        for a,b in pairs:
            edges[a].add(b)
            edges[b].add(a)
        def dfs(idx):
            seen.add(idx)
            group.add(idx)
            for nidx in edges[idx]:
                if nidx not in group:
                    dfs(nidx)
        seen=set()
        idxs=[]
        for i in range(len(s)):
            if i not in seen:
                group=set()
                dfs(i)
                idxs.append(group)
        maps={}
        for group in idxs:
            tmp=sorted(s[i] for i in group)
            tmpIds=sorted(group)
            
for i in range(len(tmp)):
                maps[tmpIds[i]]=tmp[i]
        return "".join([maps[i] for i in range(len(s))])
