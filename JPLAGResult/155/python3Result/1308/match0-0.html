import collections
class Solution:
    def smallestStringWithSwaps(self, s: str, pairs) -> str:
        N = len(s)
        p = { i:i for i in range(N)}
        
        def find(x):
            if x != p[x]:
                p[x] = find(p[x])
            return p[x]
        
        def union(x,y):
            p[find(x)] = find(y)
            
        for x,y in pairs:
            union(x,y)
        
        group = collections.defaultdict(list)
        
        for i in range(N):
            group[find(i)].append(i)
            
#         print(group)
        res = {}
        for g in group.values():
            tmp = sorted( s[i] for i in g )
            for i,a in zip(sorted(g), tmp):
                res[i] = a
        return ''.join( res[i] for i in range(N) )
