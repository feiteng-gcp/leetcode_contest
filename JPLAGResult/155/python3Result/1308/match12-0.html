class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        
        
        dic = collections.defaultdict(list)
        for u,v in pairs:
            dic[u].append(v)
            dic[v].append(u)
        
        s = list(s)
        visited = [False]*len(s)
        for i in range(len(s)):
            if visited[i] == False:
                memo = {i}
                level = {i}
                while level != set():
                    nextlevel = set()
                    for idx in level:
                        for nxtidx in dic[idx]:
                            if nxtidx not in memo:
                                memo.add(nxtidx)
                                nextlevel.add(nxtidx)
                    level = nextlevel
                memo = sorted(memo)
                chs = []
                for idx in memo:
                    visited[idx] = True
                    chs.append(s[idx])
                chs.sort()
                for i in range(len(chs)):
                    s[memo[i]] = chs[i]
        return ''.join(s)
