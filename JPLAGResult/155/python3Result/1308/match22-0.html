class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        parents = {}
        def find(p):
            if p not in parents:
                parents[p] = p
            if parents[p] != p:
                parents[p] = find(parents[p])
            return parents[p]
        def union(p, q):
            i, j = find(p), find(q)
            parents[j] = i
        
        for p, q in pairs:
            union(p, q)
        
        nodes = collections.defaultdict(set)
        for p in parents:
            nodes[find(p)].add(p)
        
        S = list(s)
        
        for root in nodes:
            indices = sorted(nodes[root])
            letters = sorted([s[i] for i in indices])
            for i in range(len(indices)):
                S[indices[i]] = letters[i]
        
        return ''.join(S)
