class UFDS:
    def __init__(self,):
        self.parent = {}
        self.rank = {}
    def add_item(self,item):
        if item not in self.parent:
            self.parent[item] = item
            self.rank[item] = 1
    def get_parent(self,item):
        if (self.parent[item] != item):
            self.parent[item] = self.get_parent(self.parent[item])
        return self.parent[item]
    def merge(self,first,second):
        self.add_item(first)
        self.add_item(second)
        first 
= self.get_parent(first)
        second = self.get_parent(second)
        if (first != second):
            if (self.rank[first] < self.rank[second]):
                self.parent[first] = second
            else:
                self.parent[second] = first
                if (self.rank[first] == self.rank[second]):
                    self.rank[first] += 1

class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        ufds = UFDS()
        for i,j in pairs:
            ufds.merge(i,j)
        for i in range(len(s)):
            ufds.add_item(i)
        sets = {}
        for key in ufds.parent:
            parent = ufds.get_parent(key)
            if parent not in sets:
                sets[parent] = set()
            sets[parent].add(key)
        ans = ['' for i in range(len(s))]
        for key, val in sets.items():
            word = []
            vals = []
            for num in val:
                vals.append(num)
                word.append(s[num])
            word.sort()
            vals.sort()
            for i in range(len(word)):
                ans[vals[i]] = word[i]
        return "".join(ans)
