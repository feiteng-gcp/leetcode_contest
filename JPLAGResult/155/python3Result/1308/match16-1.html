class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        if not s:
            return
        cc= collections.defaultdict(list)
        n = len(s)
        visited = [0 for i in range(n)]
        ss = []
        for i in s:
            ss.append(i)
        g = collections.defaultdict(set)
        for u,v in pairs:
            g[u].add(v)
            g[v].add(u)
        
        def dfs(u,ccn):
            visited[u] = 1
            cc[ccn].append(u)
            for v in g[u]:
                if not visited[v]:
                    dfs(v,ccn)
            visited[u] = 2
        ccn = 0
        for i in range(n):
            if visited[i]==0:
                ccn += 1
                dfs(i,ccn)
        for k in cc.keys():
            ids 
= sorted(cc[k])
            value = sorted([s[i] for i in ids])
            for i in range(len(ids)):
                ss[ids[i]] = value[i] 
        return "".join(ss)
