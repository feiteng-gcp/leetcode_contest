class UnionFind:
    def __init__(self, n):
        self.parents = list(range(n))
        
    def find(self, x):
        cur = x
        while cur != self.parents[cur]:
            cur = self.parents[self.parents[cur]]
        root = cur
        cur = x
        while cur != root:
            nxt = self.parents[cur]
            self.parents[cur] = root
            cur = nxt
        return root
    
    def union(self, a, b):
        root_a, root_b = self.find(a), self.find(b)
        self.parents[root_a] = root_b

class Solution:
