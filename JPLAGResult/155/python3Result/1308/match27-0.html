import collections

class UnionFind:
    def __init__(self, n):
        self.parents = list(range(n))
        self.ranks = [0 for _ in range(n)]

    def get_root(self, x):
        if self.parents[x] == x:
            return x
        self.parents[x] = self.get_root(self.parents[x])
        return self.parents[x]

    def merge(self, x, y):
        x = self.get_root(x)
        y = self.get_root(y)
        if x != y:
            if self.ranks[x] < self.ranks[y]:
                self.parents[x] = y
            else:
                self.parents[y] = x
                if self.ranks[x] == self.ranks[y]:
                    self.ranks[x] += 1


class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        n = len(s)
        uf = UnionFind(n)
        for a, b in pairs:
            uf.merge(a, b)
        ls = collections.defaultdict(lambda: [[], []])
        for i in range(n):
            r = uf.get_root(i)
            ls[r][0].append(i)
            ls[r][1].append(s[i])
        res = ['' for _ in range(n)]
        for i, l in ls.items():
            l[0].sort()
            l[1].sort()
            for j, c in zip(l[0], l[1]):
                res[j] = c
        return ''.join(res)
