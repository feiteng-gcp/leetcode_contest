from typing import List

class UnionFindSet():
    def __init__(self, n):
        self.p = [i for i in range(n)]
        self.rk = [0] * n
    
    def find(self, v):
        if self.p[v] != v: self.p[v] = self.find(self.p[v])
        return self.p[v]
    
    def union(self, v1, v2):
        p1 = self.find(v1)
        p2 = self.find(v2)
        if p1 == p2: return True
        if self.rk[p1] < self.rk[p2]: p1, p2 = p2, p1
        self.p[p2] = p1
        self.rk[p1] += self.rk[p1] == self.rk[p2]
        return False

class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        n = len(s)
        ufs = UnionFindSet(n)
        for a, b in pairs:
            ufs.union(a, b)
        idx = [-1] * n
        d = dict()
        for i in range(n):
            idx[i] = ufs.find(i)
            if idx[i] not in d: d[idx[i]] = [0] * 26
            d[idx[i]][ord(s[i])-ord('a')] += 1
        ret = []
        for i in range(n):
            k = idx[i]
            for j in range(26):
                if d[k][j] > 0:
                    d[k][j] -= 1
                    ret.append(chr(j + ord('a')))
                    break
        return ''.join(ret)
