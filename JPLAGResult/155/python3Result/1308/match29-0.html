class Solution:
    def smallestStringWithSwaps(self, s: str, pairs) -> str:
        N = len(s)
        fa = [i for i in range(N)]

        def find(a):
            # print(a)
            if fa[a] != a:
                fa[a] = find(fa[a])
            return fa[a]

        def union(a, b):
            a = find(a)
            b = find(b)
            fa[b] = a

        for a, b in pairs:
            union(a, b)

        ans = [-1] * N
        mem = {}
        mem_i = {}
        for i, c in enumerate(s):
            # print(i, c)
            f = find(i)
            if f not in mem.keys():
                mem[f] = [c]
                mem_i[f] = [i]
            else:
                mem[f].append(c)
                mem_i[f].append(i)

        for k in mem.keys():
            mem[k].sort()
            mem_i[k].sort()
            for i, index in enumerate(mem_i[k]):
                ans[index] = mem[k][i]
        # print(ans)
        return ''.join(ans)
