class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        # first connect all possible indices - union find
        # then create sorted strings for each groups
        # then for each sorted index of each group insert the next char from the group
        uf = {}
        def find (x):
            if x not in uf: uf[x] = x
            
if x != uf[x]:
                uf[x] = find(uf[x]) # normalize
            return uf[x]
        def union(x, y):
            uf[find(x)] = find(y)
            
        for x, y in pairs:
            union(x, y)
            
        # now create sorted characters
        n = len(s)
        groups = {}
        for i in range(n):
            x = find(i)
            if x not in groups:
                groups[x] = []
            groups[x].append(s[i])
            
        for k in groups.keys():
            groups[k] = sorted(groups[k], reverse=True)
            
        # now pick each indices group from union-find
        # and take the char from the respective bin
        ret = []
        for i in range(n):
            ret.append(groups[find(i)].pop())
            
        return "".join(ret)
