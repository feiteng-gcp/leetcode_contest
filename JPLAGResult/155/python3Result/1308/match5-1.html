class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        
def find(data, i):
            if i != data[i]:
                data[i] = find(data, data[i])
            return data[i]
        
        def union(data, i, j):
            pi, pj = find(data, i), find(data, j)
            if pi != pj:
                data[pi] = pj
        
        n = len(s)
        data = [i for i in range(n)]
        for i, j in pairs:
            union(data, i, j)
        
        groups = collections.defaultdict(list)
        for i in range(n):
            root = find(data, i)
            groups[root].append(i)
            
        res = ['0'] * n
        for key, val in groups.items():
            tmp = [s[_] for _ in val]
            tmp.sort()
            j = 0
            for i in val:
                res[i] = tmp[j]
                j += 1
        
        return ''.join(res)
