from queue import Queue
class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        edge = [[] for i in range(len(s))]
        for u, v in pairs:
            edge[u].append(v)
            edge[v].append(u)
        arr = [ch for ch in s]
        visited = [False for i in range(len(arr))]
        for i in range(len(arr)):
            if visited[i]:
                continue
            visited[i] = True
            queue = Queue()
            queue.put(i)
            l = [(arr[i], i)]
            while not queue.empty():
                u = queue.get()
                for v in edge[u]:
                    if visited[v]:
                        continue
                    visited[v] = True
                    l.append((arr[v], v))
                    queue.put(v)
            chs 
= sorted([p[0] for p in l])
            pos = sorted([p[1] for p in l])
            for k in range(len(chs)):
                arr[pos[k]] = chs[k]
        return ''.join(arr)
