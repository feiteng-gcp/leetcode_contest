class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        n = len(s)
        memo = [i for i in range(n)]
        
        def find(x):
            if x != memo[x]:
                memo[x] = find(memo[x])
            return memo[x]
        
        def union(x,y):
            rx, ry = find(x), find(y)
            if rx != ry:
                memo[rx] = ry
                
        for i,j in pairs:
            union(i,j)
        
        group = collections.defaultdict(list)
        for i in range(n):
            group[find(i)] += s[i],
        
        for j in group.values():
            j.sort()
            
        ss = collections.Counter()
        ans = ["" for _ in range(n)]
        for i in range(n):
            ans[i] = group[find(i)][ss[find(i)]]
            ss[find(i)] += 1
        return "".join(ans)
