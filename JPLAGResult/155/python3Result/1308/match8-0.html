class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        s = list(s)
        n = len(s)
        
        edges = [[] for _ in range(n)]
        for x, y in pairs:
            if x != y:
                edges[x].append(y)
                edges[y].append(x)
        
        v = [False] * n
        groups = []
        
        for i in range(n):
            if v[i]: continue
            groups.append([i])
            q = [i]
            v[i] = True
            while q:
                c = q.pop()
                for j in edges[c]:
                    if v[j]: continue
                    v[j] = True
                    q.append(j)
                    groups[-1].append(j)
            groups[-1].sort()
        
        for g in groups:
            t = sorted([s[i] for i in g])
            for i in range(len(g)):
                s[g[i]] = t[i]
        
        return ''.join(s)
