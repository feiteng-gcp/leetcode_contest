class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        graph = collections.defaultdict(set)
        for e1, e2 in pairs:
            graph[e1].add(e2)
            graph[e2].add(e1)
        visited = set()
        ans = list(s)
        for i, c in enumerate(s):
            if i in visited: continue
            q = collections.deque()
            q.append(i)
            idxs = []
            while q:
                cur  = q.popleft()
                if cur in visited: continue
                visited.add(cur)
                idxs.append(cur)
                for nxt in graph[cur]: q.append(nxt)
            idxs.sort()
            chars  = [s[idx] for idx in idxs]
            chars.sort()
            for j, idx in enumerate(idxs):
                ans[idx] = chars[j]
        return ''.join(ans)
