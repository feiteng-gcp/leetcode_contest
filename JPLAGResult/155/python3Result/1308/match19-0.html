class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        
        def find(x):
            if up[x] != x:
                up[x] = find(up[x])
            return up[x]
        
        def union(x, y):
            a, b = find(x), find(y)
            if a != b:
                up[a] = b   
                branch.discard(a)
        
        n = len(s)
        up = [i for i in range(n)]
        branch = set(range(n))
        for a, b in pairs:
            union(a, b)
        ret = [None] * n
        ids = collections.defaultdict(list)
        for i in range(n):
            if find(i) in branch:
                ids[find(i)].append(i)
        for r in branch:
            buf = sorted(s[i] for i in ids[r])
            for i, j in zip(ids[r], buf):
                ret[i] = j
        return ''.join(ret)
