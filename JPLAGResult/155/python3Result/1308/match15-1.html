class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        n = len(s)
        ret = [s[i] for i in range(n)]
        vis = [False for _ in range(n)]
        curr = []
        letters = []
        g 
= collections.defaultdict(set)
        for a, b in pairs:
            g[a].add(b)
            g[b].add(a)
        def dfs(i):
            if vis[i]:
                return
            vis[i] = True
            curr.append(i)
            letters.append(s[i])
            for nei in g[i]:
                dfs(nei)
            return
        
        for i in range(n):
            if vis[i]:
                continue
            curr = []
            letters = []
            dfs(i)
            letters.sort()
            curr.sort()
            m = len(curr)
            for j in range
(m):
                ret[curr[j]] = letters[j]
                
        return ''.join(ret)
