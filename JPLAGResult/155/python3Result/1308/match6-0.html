class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        def dfs(st,index,chrs):
            chrs.append(s[st])
            index.append(st)
            for adj in graph[st]:
                if not used[adj]:
                    used[adj]=True
                    dfs(adj,index,chrs)
                    
        graph=collections.defaultdict(set)
        for p in pairs:
            graph[p[0]].add(p[1])
            graph[p[1]].add(p[0])
        ret=["0"]*(len(s))
        used=[False]*len(s)
        for i in range(len(s)):
            if not used[i]:
                used[i]=True
                index=[]
                chrs=[]
                dfs(i,index,chrs)
                chrs.sort()
                index.sort()
                for j in range(len(index)):
                    ret[index[j]]=chrs[j]
        return "".join(ret)
