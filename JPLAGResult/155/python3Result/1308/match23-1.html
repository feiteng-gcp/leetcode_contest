class Solution:
    def __init__(self):
        self.p = {}
        self.rank = {}
    
    
def find(self, i):
        if self.p[i] != i:
            self.p[i] = self.find(self.p[i])
        return self.p[i]
    
    def union(self, i, j):
        ir = self.find(i)
        jr = self.find(j)
        if self.rank[ir] < self.rank[jr]:
            ir, jr = jr, ir
        
if self.rank[ir] == self.rank[jr]:
            self.rank[ir] += 1
        self.p[jr] = self.p[ir]
        return
        
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        for pair in pairs:
            i, j = pair
            if not i in self.p:
                self.p[i] = i
                self.rank[i] = 1
            if not j in self.p:
                self.p[j] = j
                self.rank[j] = 1
            self.union(i, j)
        dic = collections.defaultdict(set)
        for pair in pairs:
            i, j = pair
            dic[self.find(i)].add(i)
            dic[self.find(j)].add(j)
        res = [c for c in s]
        for values in dic.values():
            component = sorted(list(values))
            curr = [s[node] for node in component]
            curr = sorted(curr)
            for i in range(len(component)):
                res[component[i]] = curr[i]
        return ''.join(res)
