from collections import defaultdict

class UnionFind:
    
    def __init__(self, indexes):
        self.parents = {i: i for i in indexes}
    
    def find(self, a):
        if self.parents[a] == a:
            return a
        pa = self.find(self.parents[a])
        self.parents[a] = pa
        return pa
    
    def union(self, a, b):
        pa = self.find(a)
        pb = self.find(b)
        self.parents[pa] = pb

class Solution:
    def smallestStringWithSwaps(self, s: str, pairs: List[List[int]]) -> str:
        indexes = set([i for a, b in pairs for i in (a, b)])
        uf = UnionFind(indexes)
        for a, b in pairs:
            uf.union(a, b)
        
        groups = defaultdict(list)
        for a in indexes:
            groups[uf.find(a)].append(a)
        
        lst = list(s)
        for _, g in groups.items():
            g.sort()
            vals = sorted([s[i] for i in g])
            for i, c in zip(g, vals):
                lst[i] = c
        
        return ''.join(lst)
