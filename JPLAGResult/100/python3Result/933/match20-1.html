# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        return self.firstTry(root)

    def firstTry(self, root):
        lst = []
        self.inOrder(root, lst)
        if not root:
            return root
        for i in range(len(lst)-1):
            node = lst[i]
            node.left = None
            node.right = lst[i+1]
        lst[-1].left = None
        lst[-1].right = None
        
return lst[0]


    def inOrder(self, node, lst):
        if not node:
            return
        self.inOrder(node.left, lst)
        lst.append(node)
        self.inOrder(node.right, lst)
