# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.root = None
        self.nodes = []
    
    def walk(self, node, is_left=False):
        if node == None:
            return
        if is_left:
            self.root = self.walk(node.left, True) or node
        else:
            self.walk(node.left)
        self.nodes.append(node)
        self.walk(node.right)
        return self.root
    
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        self.walk(root, True)
        # self.nodes.sort(key=lambda node: node.val)
        n = len(self.nodes)
        root = self.root
        for i in range(0, n):
            if self.nodes[i] == self.root:
                continue
            self.nodes[i].left 
= self.nodes[i].right = root.left = None
            root.right = self.nodes[i]
            root = root.right
        return self.root
