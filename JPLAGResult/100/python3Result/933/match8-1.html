# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        S=[]
        def trans(R):
            
if R.left!=None:
                trans(R.left)
            S.append(R.val)
            if R.right!=None:
                trans(R.right)
        trans(root)
        T=TreeNode(S[0])
        res=T
        for i in range(1,len(S)):
            T.right=TreeNode(S[i])
            T=T.right
        return res
