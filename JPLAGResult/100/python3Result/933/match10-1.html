# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        store = []
        def helper(root, store):
            if root.left:
                helper(root.left, store)
            store.append(root.val)
            if root.right:
                helper(root.right, store)
        helper(root, store)
        cur = new_root = TreeNode(None)
        for item in store:
            cur.right = TreeNode(item)
            cur = cur.right
        return new_root.right
