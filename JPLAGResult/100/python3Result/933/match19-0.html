# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

def inorder(root):
    r = []
    if root.left:
        r += inorder(root.left)
    r.append(root.val)
    if root.right:
        r += inorder(root.right)
    return r

def rightbst(arr):
    if len(arr) == 0:
        return None
    else:
        root = TreeNode( arr[0] )
        base = root
        for x in arr[1:]:
            root.right = TreeNode(x)
            root = root.right
        return base
class Solution:
    def increasingBST(self, root):
        return rightbst(inorder(root))
