# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def convert(self, root):
        if root is None:
            return None, None
        
        if root.left is None and root.right is None:
            return root, root
        if root.left is None:
            rhead, rtail = self.convert(root.right)
            root
.right = rhead
            root.left = None
            return root, rtail
        if root.right is None:
            lhead, ltail = self.convert(root.left)
            ltail.right = root
            root.left = None
            return lhead, root
        
        lhead, ltail = self.convert(root.left)
        rhead, rtail = self.convert(root.right)
        ltail.right = root
        root.left = None
        root.right = rhead
        return lhead, rtail
        
        
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        return self.convert(root)[0]
