# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        
        inOrderTraverse = []
        
        def dfs(root, ret):
            if not root:
                return None
            
            
if root.left:
                dfs(root.left, ret)
            ret.append(root.val)
            if root.right:
                dfs(root.right, ret)
            
        dfs(root, inOrderTraverse)
        #print(inOrderTraverse)
        prev = None
        for j in range(len(inOrderTraverse)-1, -1, -1):
            node = TreeNode(inOrderTraverse[j])
            node.right = prev
            prev = node
        return node
