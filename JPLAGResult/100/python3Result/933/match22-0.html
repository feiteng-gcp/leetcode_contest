# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        if root is None:
            return None
        if root.left:
            new_left = self.increasingBST(root.left)
            tail = new_left
            while tail.right:
                tail = tail.right
            new_right = self.increasingBST(root.right)
            tail.right = root
            root.left = None
            root.right = new_right
            return new_left
        if root.right:
            new_right = self.increasingBST(root.right)
            root.right = new_right
            return root
        return root
