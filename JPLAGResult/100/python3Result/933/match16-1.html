# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        ans = []
        def inOrder(node):
            if node:
                inOrder(node.left)
                ans.append(node.val)
                inOrder(node.right)
        inOrder(root)
        if not ans: return None
        
        root = TreeNode(ans[0])
        node = root
        for i in range(1,len(ans)):
            node.right = TreeNode(ans[i])
            node = node.right
        return root
