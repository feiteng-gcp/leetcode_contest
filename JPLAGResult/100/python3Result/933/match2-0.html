# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        v=[]
        def dfs(node):
            if(node.left is not None):
                dfs(node.left)
            v.append(node.val)
            if(node.right is not None):
                dfs(node.right)
        dfs(root)
   
        vl=len(v)
        
        root=TreeNode(v[0])
        p=root
        for i in range(vl-1):
            a=TreeNode(v[i+1])
            p.right=a
            p=a
        return root
