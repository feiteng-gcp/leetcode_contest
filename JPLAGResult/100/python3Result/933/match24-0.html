class Solution:
    def increasingBST(self, root):
        dummy = TreeNode(0)
        last = dummy
        while root:
            if root.left:
                node = root
                node2 = node.left
                while True:
                    if node2.left:
                        node = node2
                        node2 = node2.left
                    elif node2.right:
                        last.right = node2
                        node.left = node2.right
                        last = last.right
                        last.left = None
                        last.right = None
                        break
                    else:
                        last.right = node2
                        node.left = None
                        last = last.right
                        last.left = None
                        last.right = None
                        break
            elif root.right:
                last.right = root
                root = root.right
                last = last.right
                last.left = None
                last.right = None
            else:
                last.right = root
                root = None
                last = last.right
                last.left = None
                last.right = None
        return dummy.right
