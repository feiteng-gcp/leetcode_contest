# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        nodes = []
        def dfs(node):
            if node.left:
                dfs(node.left)
            nodes.append(node.val)
            if node.right:
                dfs(node.right)
        dfs(root)
        ret = TreeNode(nodes[0])
        node = ret
        for i in range(1, len(nodes)):
            node.right = TreeNode(nodes[i])
            node = node.right
        return ret
