# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        path = []
        self.inorder(root, path)
        root = TreeNode(0)
        temp = root
        for num in path:
            temp.right = TreeNode(num)
            temp = temp.right
        return root.right
        #print(path)
        
    def inorder(self, root, path):
        if root is None:
            return
        self.inorder(root.left, path)
        path.append(root.val)
        self.inorder(root.right, path)
