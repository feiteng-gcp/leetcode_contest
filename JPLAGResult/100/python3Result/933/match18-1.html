# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def increasingBST(self, root):
        """
        :type root: TreeNode
        :rtype: TreeNode
        """
        result = []
        def BST(root):
            if not root:
                return
            
if root.left:
                BST(root.left)
            result.append(root.val)
            
            if root.right:
                BST(root.right)
        BST(root)
        head = None
        node = head
        for i in result:
            if not head:
                head = TreeNode(i)
                node = head
            else:
                node.right = TreeNode(i)
                node = node.right
        return head
