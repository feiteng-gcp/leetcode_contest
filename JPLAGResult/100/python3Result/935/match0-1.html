class Solution:
    def orderlyQueue(self, S, K):
        """
        :type S: str
        :type K: int
        :rtype: str
        """
        if K > 1:
            return ''.join(sorted(S))
        res = S
        tmp 
= S
        for i in range(len(S)):
            tmp = tmp[1:]+tmp[0]
            if tmp < res:
                res = tmp
        return res
