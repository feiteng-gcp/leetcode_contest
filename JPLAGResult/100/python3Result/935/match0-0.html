class Solution:
    def orderlyQueue(self, S, K):
        """
        :type S: str
        :type K: int
        :rtype: str
        """
        if K > 1: return ''.join(sorted(S))
        else:
            mn = S
            for i in range(len(S)):
                S = S[1:] + S[0]
                if mn > S:
                    mn = S
            return mn
