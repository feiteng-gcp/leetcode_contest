class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        ok=True
        for i in range(1,len(A)):
            if A[i-1]>A[i]:
                ok=False
                break
        if ok:
            return True
        ok=True
        for i in  range(1,len(A)):
            if A[i-1]<A[i]:
                ok=False
                break
        return ok
