class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        has_inc = False
        has_dec = False
        for i in range(1, len(A)):
            if A[i] > A[i - 1]:
                has_inc = True
            elif A[i] < A[i - 1]:
                has_dec = True
        return not (has_inc and has_dec)
