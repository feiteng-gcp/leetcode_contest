class Solution:
    
def increasing(self, A):
        n = len(A)
        for i in range(n-1):
            if A[i] > A[i+1]:
                return False
        return True
    def decreasing(self, A):
        n = len(A)
        for i in range(n-1):
            if A[i] < A[i+1]:
                return False
        return True
    
    def isMonotonic(self, A):
        n = len(A)
        if n == 1:
            return True
        return self.increasing(A) or self.decreasing(A)
        
        """
        :type A: List[int]
        :rtype: bool
        """
