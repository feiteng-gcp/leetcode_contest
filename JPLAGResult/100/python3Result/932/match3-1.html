class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        ok = True
        for i in range(len(A)-1):
            if A[i] > A[i+1]:
                ok =
 False
        if ok:
            return True
        ok = True
        for i in range(len(A)-1):
            if A[i] < A[i+1]:
                ok = False
        if ok:
            return True
        return False
