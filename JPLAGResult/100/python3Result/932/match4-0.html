class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if A == []: return True
        increase = True
        for i in range(1,len(A)):
            if A[i] < A[i-1]:
                increase = False
                break
        
        
        decrease = True
        for i in range(1,len(A)):
            if A[i] > A[i-1]:
                decrease = False
                break
        return increase or decrease
