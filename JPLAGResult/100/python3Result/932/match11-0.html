class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) <= 1:
            return True
        
        return self.is_reverse_sorted(A) or self.is_sorted(A)
        
        
    def is_reverse_sorted(self, A):
        for i in range(1, len(A)):
            if A[i - 1] < A[i]:
                return False
        return True
    
    def is_sorted(self, A):
        for i in range(1, len(A)):
            if A[i - 1] > A[i]:
                return False
        return True
