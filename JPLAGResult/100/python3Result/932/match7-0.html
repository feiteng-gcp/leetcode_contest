class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        inc = True
        dec = True
        for i in range(1, len(A)):
            if A[i] > A[i-1]:
                dec = False
            if A[i] < A[i-1]:
                inc = False
        if dec or inc:
            return True
        else:
            return False
