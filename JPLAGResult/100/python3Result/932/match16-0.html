class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        return self.isMonoIncreasing(A) or self.isMonoIncreasing(A[::-1])
        
        
    def isMonoIncreasing(self, arr):
        if len(arr) == 1:
            return True
        
        for i in range(1, len(arr)):
            if arr[i - 1] > arr[i]:
                return False
        
        return True
