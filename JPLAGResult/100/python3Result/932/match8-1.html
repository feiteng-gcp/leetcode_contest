class Solution:
    
def isIncreasing(self, A):
        for i in range(len(A) - 1):
            if A[i] > A[i + 1]:
                return False
        return True
    
    def inDecreasing(self, A):
        for i in range(len(A) - 1):
            if A[i] < A[i + 1]:
                return False
        return True
        
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if not A or len(A) < 2:
            return True
        
        if self.isIncreasing(A) or self.inDecreasing(A):
            return True
        return False
