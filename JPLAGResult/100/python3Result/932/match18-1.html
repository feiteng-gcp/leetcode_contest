class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        
        if len(A) < 2: 
            return True
        
        state = 0 # 0 = unknown, 1 = increasing, 2 = decreasing
        
        for i in range(len(A) - 1):
            if A[i] < A[i + 1]:
                if state == 0:
                    state = 1
                elif state == 2:
                    return False
            if A[i] > A[i + 1]:
                if state == 0:
                    state = 2
                elif state == 1:
                    return False
        return True
