class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        flag1=True
        flag2=True
        for i in range(1,len(A)):
            if A[i]<A[i-1]:
                flag1=False
        
        for i in range(1,len(A)):
            if A[i]>A[i-1]:
                flag2=False
        return (flag1 or flag2)
