class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) <= 2:
            return True
        mult = 1 if A[0] <= A[-1] else -1
        
for i in range(1, len(A)):
            if A[i - 1] * mult > A[i] * mult:
                return False
        return True
