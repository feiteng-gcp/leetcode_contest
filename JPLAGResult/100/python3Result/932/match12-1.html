class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) <= 1:
            return True
        increasing = None
        for i in range(1, len(A)):
            if A[i] > A[i-1]:
                if increasing is None:
                    increasing = True
                elif not increasing:
                    return False
            elif A[i] < A[i-1]:
                if increasing is None:
                    increasing = False
                elif increasing:
                    return False
            
        return True
