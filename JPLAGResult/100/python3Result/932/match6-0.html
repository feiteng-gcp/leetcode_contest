class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) == 0 or len(A) == 1:
            return True
        decreasing = True
        increasing = True
        for i in range(1, len(A)):
            if A[i] < A[i-1]:
                increasing = False
            if A[i] > A[i-1]:
                decreasing = False
        return decreasing or increasing
