class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        def up(a):
            n = len(a)
            for i in range(n - 1):
                if a[i] > a[i + 1]:
                    return False
            return True
        
        return up(A) or up(A[::-1])
