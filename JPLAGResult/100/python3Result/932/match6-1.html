class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        increasing = True
        decreasing = True
        if not A:
            return False
        
        
for i in range(1,len(A)):
            if A[i-1]<A[i]:
                decreasing = False
            
            if A[i-1] > A[i]:
                increasing = False
        
        return increasing or decreasing
