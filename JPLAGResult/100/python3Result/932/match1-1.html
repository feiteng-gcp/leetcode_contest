class Solution:
    def isMonotonic(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        isInc = True
        for i in range(1,len(A)):
            if A[i] < A[i-1]:
                isInc = False
                break
        isDec 
= True
        for i in range(1,len(A)):
            if A[i] > A[i-1]:
                isDec = False
                break
        return isInc or isDec
