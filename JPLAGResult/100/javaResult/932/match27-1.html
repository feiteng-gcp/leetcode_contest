class Solution {
    public boolean isMonotonic(int[] A) {
        int i;
        int pd=0;
        for(i=1;i<A.length;i++)
        
{
            if(A[i]>A[i-1])
            {
                if(pd==2)
                {
                    return false;
                }
                pd=1;
            }else if(A[i]<A[i-1])
            {

                    if(pd==1)
                    {
                        return false;
                    }
                    pd=2;
            }
        }
        return true;
    }
}
