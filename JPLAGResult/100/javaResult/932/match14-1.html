class Solution {
    public boolean isMonotonic(int[] A) {
        if (A.length <= 1) {
            return true;
        }
        
boolean flag = true;
        for (int i = 1; i < A.length; i++) {
            if (A[i] < A[i - 1]) {
                flag = false;
                break;
            }
        }
        if (flag) {
            return true;
        
}
        flag = true;
        for (int i = A.length - 2; i >= 0; i--) {
            if (A[i] < A[i + 1]) {
                flag = false;
                break;
            }
        }
        return flag;
    }
}
