class Solution {
    public boolean isMonotonic(int[] A) {
        if (A.length <= 1) return true;
        
        boolean flag1 = true;
        boolean flag2 = true;
        for (int i = 1; i < A.length; i++) {
            if (A[i] > A[i-1]) flag1 = false;
            if (A[i] < A[i-1]) flag2 = false;
        }
        return flag1 || flag2;

    }
}
