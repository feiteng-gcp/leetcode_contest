class Solution {
    public boolean isMonotonic(int[] A) {
        if (A.length == 1) {
            return true;
        }

        int flag = 0;
        for (int idx = 1; idx < A.length; idx++) {
            if (A[idx] > A[idx - 1]) {
                if (flag < 0) {
                    return false;
                }
                flag = 1;
            } else if (A[idx] < A[idx - 1]) {
                if (flag > 0) {
                    return false;
                }
                flag = -1;
            }
        }
        return true;
    }
}
