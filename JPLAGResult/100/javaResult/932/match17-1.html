class Solution {
    public boolean isMonotonic(int[] A) {
        if(A.length <= 2) return true;
        int n = A.length;
        boolean isIncrease = A[n - 1] >= A[0];
        for(int i = 1; i < A.length; i++){
            if(A[i] == A[i - 1]) continue
;
            if(isIncrease != A[i] > A[i - 1]) return false;
        }
        return true;
    }
}
