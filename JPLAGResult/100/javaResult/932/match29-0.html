class Solution {
    public boolean isMonotonic(int[] A) {
        if(A.length == 1) return true;
        int flag = 0;
        for(int i = 0; i < A.length - 1; i++){
            if(A[i] > A[i + 1]){
                if(flag == 2) return false;
                flag = 1;
            }
            else if(A[i] < A[i + 1]){
                if(flag == 1) return false;
                flag = 2;
            }
            else {
                continue;
            }
        }
        return true;
    }
}
