class Solution {
    public boolean isMonotonic(int[] A) {
        boolean flag = true;
        for (int i = 1;i < A.length; i++) {
            if (A[i] < A[i - 1]) {
                flag = false;
                break;
            }
        }
        if (flag) {
            return true;
        }
        flag = true;
        
for (int i = 1;i < A.length; i++) {
            if (A[i] > A[i - 1]) {
                flag = false;
                break;
            }
        }
        if (flag) {
            return true;
        }
        return false;
    }
}
