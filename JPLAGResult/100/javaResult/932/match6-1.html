class Solution {
    public boolean isMonotonic(int[] A) {
        
boolean mi = true, md = true;
        
        for (int i = 0; i < A.length - 1; i++) {
            if (A[i] > A[i+1]) mi = false;
            if (A[i] < A[i+1]) md = false;
        }
        
        return mi || md;
    }
}
