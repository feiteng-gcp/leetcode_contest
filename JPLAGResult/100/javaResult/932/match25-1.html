class Solution {
    public boolean isMonotonic(int[] A) {
        int n = A.length;
        if (n <= 1) return true;
        
        
Boolean increase = null;
        for (int i = 1; i < n; i++) {
            if (A[i] == A[i - 1]) continue;
            if (A[i] > A[i - 1]) {
                if (increase == null) {
                    increase = true;
                } else 
if (increase == false) return false;
            } else {
                if (increase == null) {
                    increase = false;
                } else 
if (increase) return false;
            }
        }
        return true;
    }
}
