class Solution {
    public boolean isMonotonic(int[] A) {
        int mono = 0;
        for (int i = 1; i < A.length; i++) {
            if (A[i] == A[i - 1]) continue;
            if (A[i] > A[i - 1]) {
                if (mono == 0) mono = 1;
                else if (mono == -1) return false;
            } else {
                if (mono == 0) mono = -1;
                else if (mono == 1) return false;
            }
        }
        return true;
    }
}
