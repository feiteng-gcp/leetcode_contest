class Solution {
    public boolean isMonotonic(int[] A) {
        int n = A.length;
        int sign = 0;
        for (int i = 0; i < n - 1; i++) {
            if (A[i] < A[i + 1]) {
                if (sign < 0) return false;
                sign = 1;
            } else if (A[i] > A[i + 1]) {
                if (sign > 0) return false;
                sign = -1;
            }
        }
        
        return true;
    }
}
