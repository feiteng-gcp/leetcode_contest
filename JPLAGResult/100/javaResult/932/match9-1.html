class Solution {
    public boolean isMonotonic(int[] A) {
        
        int flag = 0;
        for (int i = 1; i < A.length; i++ ){
            if (A[i] == A[i - 1])
                continue;
            else if(A[i] > A[i-1]){
                if (flag == 0)
                    flag = 1;
                else if (flag ==1)
                    continue;
                else
                    
return false;
            }
            else{
                if (flag == 0)
                    flag = -1;
                else if (flag ==-1)
                    continue;
                else
                    
return false;
                
            }
        }
        return true;
        
        
    }
}
