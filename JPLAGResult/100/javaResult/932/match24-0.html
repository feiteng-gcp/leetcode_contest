class Solution {
    public boolean isMonotonic(int[] A) {
        int N = A.length;
        boolean r1 = true, r2 = true;
        for (int i = 1; i < N; ++i) {
        	if (A[i - 1] > A[i]) {
        		r1 = false;
        	}
        	if (A[i - 1] < A[i]) {
        		r2 = false;
        	}
        }
        return r1 || r2;
    }
}
