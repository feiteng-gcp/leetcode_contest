class Solution {
    public boolean isMonotonic(int[] A) {
        int inc = 0;
        for(int i = 1; i < A.length; i++) 
{
            if(A[i] > A[i - 1]) {
                if(inc == 2) {
                    return false;
                }
                inc = 1;
            } else if(A[i] < A[i - 1]) {
                if(inc == 1) {
                    return false;
                }
                inc = 2;
            } 
        }
        return true;
    }
}
