/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode increasingBST(TreeNode root) {
        List<TreeNode> inOrd = new ArrayList<>();
        dfs(root, inOrd);
        TreeNode vHead = new TreeNode(0), cur = vHead;
        
        for (TreeNode node: inOrd) {
            cur.right = new TreeNode(node.val);
            
cur = cur.right;
        }
        
        return vHead.right;
    }
    
    void dfs(TreeNode node, List<TreeNode> inOrd) {
        if (node == null) return;
        dfs(node.left, inOrd);
        inOrd.add(node);
        dfs(node.right, inOrd);
    }
}
