/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
     public TreeNode increasingBST(TreeNode root) {
        if (root == null) {
            return null;
        }
        List<TreeNode> nodeList = new ArrayList<>();
        travel(root, nodeList);
        for (int i = 0; i < nodeList.size() - 1; i++) {
            nodeList.get(i).left = null;
            nodeList.get(i).right = nodeList.get(i + 1);
        }
        nodeList.get(nodeList.size() - 1).left = null;
        return nodeList.get(0);
    }

    void travel(TreeNode root, List<TreeNode> nodeList) {
        if (root == null) {
            return;
        }
        travel(root.left, nodeList);
        nodeList.add(root);
        travel(root.right, nodeList);
    }
}
