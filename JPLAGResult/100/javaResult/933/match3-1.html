/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode increasingBST(TreeNode root) {
        ArrayList<Integer> a = new ArrayList<Integer>();
        pre(root, a);
        TreeNode head = new TreeNode(0);
        TreeNode cur = head;
        for(Integer i : a){
            cur.right = new TreeNode(i);
            
cur = cur.right;
        }
        return head.right;
    }
    
    public void pre(TreeNode root, ArrayList<Integer> s){
        if(root == null)
            return;
        pre(root.left, s);
        s.add(root.val);
        pre(root.right, s);
    }
}
