/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode increasingBST(TreeNode root) {
        
List<Integer> res = new ArrayList<>();
        increase(root, res);
        TreeNode dummy = new TreeNode(0);
        TreeNode cur = dummy;
        for (int x : res) {
            cur.right = new TreeNode(x);
            
cur = cur.right;
        }
        return dummy.right;
    }
    private void increase(TreeNode root, List<Integer> res) {
        if (root == null) return;
        
        increase(root.left, res);
        res.add(root.val);
        increase(root.right, res);
    }
}
