	class Solution {
	    public String orderlyQueue(String S, int K) {
	        int n = S.length();
	        if(K >= 2){
	        	char[] s = S.toCharArray();
	        	Arrays.sort(s);
	        	return new String(s);
	        }else{
	        	String ret = S;
	        	for(int i = 0;i < n;i++){
	        		String T = S.substring(i) + S.substring(0, i);
	        		if(T.compareTo(ret) < 0){
	        			ret = T;
	        		}
	        	}
	        	return ret;
	        }
	    }
	}
