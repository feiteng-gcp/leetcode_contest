class Solution {
        public String orderlyQueue(String S, int K) {
            if(K>1){ 
            char ss[]=S.toCharArray();
            Arrays.sort(ss);
            String str=new String(ss);
            return str;
            }
           else{
            String str=S;
            String minstr=S;
            for(int i=0;i<S.length();i++){
                char o=str.charAt(0);
                String str1=str.substring(1)+o;
                str=str1;
                if(str.compareTo(minstr)<0){
                    minstr=str;
                }
            }
            return minstr;
           }
           
        }
    }
