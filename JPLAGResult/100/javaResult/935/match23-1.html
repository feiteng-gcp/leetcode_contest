class Solution {
    public String orderlyQueue(String S, int K) {
        char[] c = S.toCharArray();
        if(K > 1){
            Arrays.sort(c);
            
        }
        
        if(K == 1){
            String temp = S+S;
            int n = S.length();
            String min = S;
            for(int i = 0; i < n; i++){
                String x = 
temp.substring(i, i+n);
                if(x.compareTo(min)<0){
                    min = x;
                }
            }
            return min;
        }
        return new String(c);
    }
}
