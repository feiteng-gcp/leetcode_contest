class Solution {
    public String orderlyQueue(String S, int K) {
        if (K==1){
            String curMin = S;
            String candidate = S;
            for (int i=0; i<=S.length(); i++){
                String newC = candidate.substring(1) + candidate.charAt(0);
                if (newC.compareTo(curMin)<0) 
                    curMin = newC;
                candidate = newC;
            }            
            return curMin;
        }
        
        int[] counts = new int[26];
        for (char c : S.toCharArray()){
            counts[c-'a']++;
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i=0; i<26; i++){
            for (int n = 0; n<counts[i]; n++){
                sb.append((char)('a'+i));
            }
        }        
        
        return sb.toString();    
    }
}
