class Solution {
    public String orderlyQueue(String S, int K) {
        if(K==1){
        String ns = S+S;
        TreeSet<String> t = new TreeSet<>();
        for(int i=0;i<S.length();i++){
            t.add(ns.substring(i,i+S.length()));
        }
        return t.first();
        }else{
            char as[] = S.toCharArray();
            Arrays.sort(as);
            return new String(as);
        }
    }
}
