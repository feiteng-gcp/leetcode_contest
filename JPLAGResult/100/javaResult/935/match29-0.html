class Solution {
    public String orderlyQueue(String S, int K) {
        if (S == null) return null;
        if (K <= 0) return S;
        if (K == 1) return smallestRotation(S);
        char[] c = S.toCharArray();
        Arrays.sort(c);
        return new String(c);
    }
    
    private String smallestRotation(String s) {
        String min = s;
        for (int i = 1; i < s.length(); ++ i) {
            String s2 = s.substring(i) + s.substring(0, i);
            if (s2.compareTo(min) < 0) min = s2;
        }
        return min;
    }
}
