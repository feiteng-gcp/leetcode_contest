class Solution {
    public String orderlyQueue(String S, int K) {
        if (K == 1) 
{
            String result = S;
            for (int i = 1; i < S.length(); i++) {
                String temp = S.substring(i) + S.substring(0, i);
                if (temp.compareTo(result) < 0) {
                    result = temp;
                }
            }
            return result;
        } else {
            char[] cs = S.toCharArray();
            Arrays.sort(cs);
            return new String(cs);
        }
    }
}
