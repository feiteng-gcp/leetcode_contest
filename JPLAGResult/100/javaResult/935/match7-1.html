class Solution {
    public String orderlyQueue(String S, int K) {
		if (K == 1) {
			String smallest = S;
			for (int i = 0; i < S.length(); i++) {
				char c = S.charAt(0);
				S = S.substring(1) + c;

				if (S.compareTo(smallest) < 0) {
					smallest = S;
				}
			}
			return smallest;
		} else {
			char[] arr = S.toCharArray();
			Arrays.sort(arr);
			return new String(arr);
		}
    }
}
