class Solution {
    public String orderlyQueue(String S, int K) {
        if(K > 1) {
            
return sort(S);
        }
        String min = S;
        for(int i = 0; i < S.length(); i++) {
            String p = S.substring(i, 
S.length()) + S.substring(0, i);
            if(p.compareTo(min) < 0) {
                min = p;
            }
        }
        return min;
    }
    
    private String sort(String s) {
        char[] c = s.toCharArray();
        Arrays.sort(c);
        return new String(c);
    }
}
