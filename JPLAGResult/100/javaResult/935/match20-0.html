class Solution {
    public String orderlyQueue(String S, int K) {
        if(K>=2){
            char[] array=S.toCharArray();
            Arrays.sort(array);
            return String.valueOf(array);
        }
        String res=S;
        for(int i=0;i<S.length();i++){
            String temp=S.substring(i)+S.substring(0,i);
            if(temp.compareTo(res)<0){
                res=temp;
            }
        }
        return res;
    }
    
}
