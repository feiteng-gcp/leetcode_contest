class Solution {
    public String orderlyQueue(String S, int K) {
        if (S.length() == 1) return S;
        
        if (K>=2) {
            char[] a = S.toCharArray();
            Arrays.sort(a);
            return new String(a);
        }
        
        String rv = S;
        
String tmp = S;
        for(int i=0; i<S.length()-1; i++) {
            String newString = tmp.substring(1) + tmp.substring(0, 1);
            if (newString.compareTo(rv) < 0) {
                rv = newString;
            }
            tmp = newString;
        }
        return rv;
    }
}
