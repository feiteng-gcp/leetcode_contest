class Solution
{
    public int subarrayBitwiseORs(int[] a)
    {
        HashSet<Integer> total = new HashSet<>();
        HashSet<Integer> cur = new HashSet<>();
        for (int i : a)
        {
            HashSet<Integer> nc = new HashSet<>();
            nc.add(i);
            for (int j : cur) nc.add(i | j);
            for (int j : nc) total.add(j);
            cur = nc;
        }
        return total.size();
    }
}
