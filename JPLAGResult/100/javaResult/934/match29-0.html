class Solution {
    public int subarrayBitwiseORs(int[] arr) {
            //long ts1 = System.currentTimeMillis();
            Set<Integer> res = new HashSet<>();
            for (int e : arr)
                res.add(e);
//            for (int l = 2; l <= n; l++) {
            int n = arr.length;
            //for (int l = 2; l <= 32; l++) {
            for (; n > 0; ) {
                for (int i = 1; i < n; i++) {
                    arr[i-1] |= arr[i];
                    res.add(arr[i-1]);
                }
                n--;
                int dupes = 0;
                for (int i = 1, j = 0; i < n; i++, j++) {
                    while (i < n && arr[i] == arr[j]) {
                        dupes++;
                        i++;
                    }
                    if (i < n)
                        arr[j+1] = arr[i];
                }
                n -= dupes;
            }
            //System.out.println("Algo:  " + (System.currentTimeMillis()-ts1) + " ms");
            //System.out.println(res);
            return res.size();
        }
}
