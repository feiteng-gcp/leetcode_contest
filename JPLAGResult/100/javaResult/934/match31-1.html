class Solution {
    
    Integer[] hist; 
    
    public int subarrayBitwiseORs(int[] A) {
        List<int[]> list = new ArrayList<int[]>();
        int n = A.length;
        hist = new Integer[32];
        Set<Integer> set = new HashSet<Integer>(); 
        int[] prev = new int[32];
        
        for (int i = 0; i < n; ++i) {
            int[] count = helper(A[i], i, prev);
            list.add(count);
            prev = count;
            set.add(convert(count, new int[32]));
            for (int j = 0; j < 32; ++j) {
                if (hist[j] != null && hist[j] != i) {
                    int tmp = hist[j];
                    set.add(convert(count, 
list.get(tmp)));
                }
            }
        }
        
        return set.size();
    }
    
    private int convert(int[] b1, int[] b2) {
        // convert b1 - b2
        int res = 0;
        for (int i = 0; i < 32; ++i) {
            if (b1[i] - b2[i] > 0) res += 1 << i; 
        }
        return res;
    }
    
    private int[] helper(int num,  int ind, int[] prev) {
        
        int[] res = new int[32];
        for (int i = 0; i < 32; ++i) {
            res[i] = prev[i];
            if (num % 2 == 1) {
                res[i] += 1;   
                hist[i] = ind;
            }
            num >>>= 1;
        }
        return res;
        
    }
}
