class Solution {
    public int subarrayBitwiseORs(int[] A) {
    	Set<Integer> ansSet = new HashSet<Integer>();
    	Set<Integer> lastSet = new HashSet<Integer>();
    	Set<Integer> curSet = new HashSet<Integer>();
    	for(int i = A.length - 1; i >= 0; i--) {
    		curSet = new HashSet<Integer>();
    		for(int n : lastSet) {
    			curSet.add(n | A[i]);
    		}
    		curSet.add(A[i]);
    		ansSet.addAll(curSet);
    		lastSet = curSet;
    	}
    	//System.out.println(ansSet);
        return ansSet.size();
    }
}
