class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> orSet = new HashSet<>();
        Set<Integer> retSet = new HashSet<>();
        for (int idx : A) {
            Set<Integer> tmpSet = new HashSet<>();
            for (int setIdx : orSet) {
                tmpSet.add(setIdx | idx);
            }
            tmpSet.add(idx);
            retSet.addAll(orSet);
            orSet = tmpSet;
        }
        retSet.addAll(orSet);
        return retSet.size();
    }
}
