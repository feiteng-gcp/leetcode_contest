class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> set = new HashSet<>();
        //Arrays.sort(A);
        int[] max = new int[A.length];
        for(int i = 0; i < A.length; i ++) {
            if(i == 0) max[0] = A[0];
            else max[i] = A[i] | max[i - 1];
        }
        
        for(int i = 0; i < A.length; i ++) {
            //if(set.contains(A[i])) continue;
            int num = 0;
            for(int j = i; j >= 0; j --) {
                num |= A[j];
                set.add(num);
                if(num == max[i]) break
;
            }
        }
        return set.size();
    }
}
