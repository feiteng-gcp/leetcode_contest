class Solution {
    public int subarrayBitwiseORs(int[] A) {
        if (A.length == 0) return 0;
        
        Set<Integer> last = new HashSet<>();
        Set<Integer> total = new HashSet<>();
        
        last.add(A[0]);
        total.add(A[0]);
        
        for (int i=1; i<A.length; i++) {
            Set<Integer> current = new HashSet<>();
            for (Integer v : last) {
                Integer t = v | A[i];
                current.add(t);
                total.add(t);
            }
            
            current.add(A[i]);
            total.add(A[i]);
            
            last = current;
        }
        
        return total.size();
    }
}
