class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> set = new HashSet();
        if (A==null) return 0;
        int len=A.length;
        if(len==1) return 1;
        int sum=0;
        for (int i=0; i<len; i++){
            sum= sum|A[i];
            set.add(sum);
        }
        
        int large= (int)Collections.max(set);
        int gap=-1;
        for (int i=large; i>=0; i--){
            if (!set.contains(i)){
                gap=i;
                break;
            }
        }
        for (int i=1; i<len; i++){
            sum=A[i];
            for (int j=i; j<len; j++){
                sum=sum|A[j];
                set.add(sum);
                if (sum>=gap) break;
    
            }
        }
        return set.size();
    }
}
