class Solution {
    public int subarrayBitwiseORs(int[] A) {
        int[] incLast = new int[30];
        Set<Integer> result = new HashSet<>();
        for (int i = 0; i < A.length; i++) {
            result.add(A[i]);
            for (int j = 0; j < 30; j++) {
                if ((A[i] & 1 << j) != 0) {
                    incLast[j] = A[i];
                } else {
                    incLast[j] |= A[i];
                    result.add(incLast[j]);
                }
            }
        }
        return result.size();
    }
}
