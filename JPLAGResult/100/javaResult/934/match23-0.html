class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> set = new HashSet<>();
        Set<Integer> pre = new HashSet<>();
        for(int i = 0; i < A.length; i++){
            Set<Integer> now = new HashSet<>();
            int cur = A[i];
            now.add(cur);
            for(int val : pre) now.add(val | cur);
            for(int val : now) set.add(val);
            pre = now;
        }
        
        return set.size();
    }
}
