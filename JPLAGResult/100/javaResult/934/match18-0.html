class Solution {
    public int subarrayBitwiseORs(int[] A) {
        if(A.length < 1) return A.length;
        Set<Integer> preSet = new HashSet<>();
        Set<Integer> totalSet = new HashSet<>();
        preSet.add(A[0]);
        totalSet.addAll(preSet);
        for(int i = 1; i < A.length; i++){
            Set<Integer> nextPreSet = new HashSet<>();
            for(int pre : preSet) nextPreSet.add(pre | A[i]);
            nextPreSet.add(A[i]);
            preSet = nextPreSet;
            totalSet.addAll(preSet);
        }
        return totalSet.size();
    }
}
