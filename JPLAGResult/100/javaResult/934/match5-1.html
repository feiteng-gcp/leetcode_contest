class Solution {
    public int subarrayBitwiseORs(int[] A) {
        if (A.length < 2) return A.length;
        Set<Integer> ans = new HashSet<>();
        Set<Integer> set = new HashSet<>();
        set.add(A[0]);
        ans.add(A[0]);
        for (int i = 1; i < A.length; i++) {
            Set<Integer> nset = new HashSet<>();
            nset.add(A[i]);
            ans.add(A[i]);
            for (int num: set) {
                nset.add(num | A[i]);
                ans.add(num | A[i]);
            
}
            set = nset;
        }
        return ans.size();
    }
}
