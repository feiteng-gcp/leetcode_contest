class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> set = new HashSet<>();
        int max = 0;
        for (int a : A) {
        	max |= a;
        }
        for (int i = 0; i < A.length; i++) {
        	int s = A[i];
        	set.add(s);
        	for (int j = i + 1; j < A.length; j++) {
        		s |= A[j];
        		set.add(s);
        		if (s == max) {
        			break;
        		}

        	}
        }
        return set.size();
    }
}
