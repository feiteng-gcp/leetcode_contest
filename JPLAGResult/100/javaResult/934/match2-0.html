class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> set=new HashSet<Integer>();
        Set<Integer> total=new HashSet<Integer>();
        for(int i=0;i<A.length;i++){
            Set<Integer> toadd=new HashSet<Integer>();
            toadd.add(A[i]);
            for(int pre:set){
                toadd.add(pre|A[i]);
            }
            total.addAll(set);
            set=toadd;
        }
        total.addAll(set);
        return total.size();
    }
}
