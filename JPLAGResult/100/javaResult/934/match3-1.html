class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> set = new HashSet<>();
        Set<Integer> all = new HashSet<>();
        for (int i : A) {
            Set<Integer> next = new HashSet<>();
            next.add(i);
            set.forEach(j -> next.add(j | i));
            set = next;
            all.addAll(set);
        }
        return all.size();
    }
}
