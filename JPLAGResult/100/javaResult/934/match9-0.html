class Solution {
    public int subarrayBitwiseORs(int[] A) {
        Set<Integer> set0 = new HashSet<>();
        Set<Integer> set1 = new HashSet<>();
        // int sum = 0;
        for (int a: A) {
            Set<Integer> set2 = new HashSet<>();
            set2.add(a);
            for (int b: set1) {
                set2.add(b | a);
            }
            set1 = set2;
            set0.addAll(set2);
                
        }
        //System.out.println(set0);
        return set0.size();
    }
}
