class Solution {
public:
    string orderlyQueue(string S, int K) {
        int size = S.size();
        if (size < 2) return S;
        
        string small = S;

        if (K == 1)
        {
            string temp = S;
            for (int i = 0; i < S.size() - 1; i++)
            {
                // shift S;
                temp = (temp.substr(1, size - 1) + temp.substr(0,1));
             
              //  cout << temp << " " << small << endl;
                if (temp < small) small = temp;
            }
            
            return small;
        }
        else
        {
            sort(S.begin(), S.end());
            return S;
        }
    }
};
