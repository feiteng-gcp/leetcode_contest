class Solution {
public:
    string orderlyQueue(string S, int K) {
        if (K == 1) {
            string minstr = S;
            for (int i = 0; i < S.length(); i ++) {
                S = S.substr(1) + S[0];
                if (S < minstr) {
                    minstr = S;
                }
            }
            return minstr;
        }
        else {
            sort(S.begin(), S.end());
            return S;
        }
    }
};
