class Solution {
public:

    string orderlyQueue(string S, int K) {
        int n = S.size();
        if (K == 1) {
            string T = S + S;
            string ret = S;
            for (int k = 1; k < n; ++k) {
                ret = min(ret, T.substr(k, n));
            }
            return ret;
        }
        sort(S.begin(), S.end());
        return S;
    }
};
