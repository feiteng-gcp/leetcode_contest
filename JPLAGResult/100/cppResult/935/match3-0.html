class Solution {
public:
    string orderlyQueue(string S, int K) {
    if(K>=2){
        sort(S.begin(),S.end());
        return S;
    }
    string ans=S;
    for(int i=1;i<S.size();++i){
        string temp=S.substr(i)+S.substr(0,i);
        if(temp<ans)
            ans=temp;
    }
    return ans;

}
};
