class Solution {
public:
    string orderlyQueue(string S, int K) {
        if (K > 1) {
            sort(S.begin(), S.end());
            return S;
        }
        string res = S;
        for (int i = 1; i < S.size(); ++i) {
            S = S.substr(1) + S.substr(0, 1);
            if (S < res) res = S;
        }
        return res;
    }
};
