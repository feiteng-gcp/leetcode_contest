class Solution {
public:
    string orderlyQueue(string S, int K) {
        string res = S;
        if (K==1) {

            for (int i=1; i<S.length(); i++) {
                string ns = S.substr(i,S.length()-i) + S.substr(0,i);
                if (res > ns) res = ns;
            }
            return res;
        }
        sort(res.begin(), res.end());
        return res;
    }
};
