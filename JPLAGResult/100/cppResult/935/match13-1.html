class Solution {
public:
    string orderlyQueue(string S, int K) {
        if (K > 1) {
            sort(S.begin(), S.end());
            return S;
        }
        string ans=S;
        for (int i=1; i<S.size(); i++) {
            string tmp = S.substr(i) + S.substr(0,i);
            if (tmp<ans)
                ans = tmp;
        }
        return ans;
    }
};
