class Solution {
public:
    bool comp(string a,string b)
{
	int l=a.size();
	for(int i=0;i<l;i++)
	{
		if(a[i]<b[i])return true;
		if(a[i]>b[i])return false;
	}
	return false;
}
    string orderlyQueue(string S, int K) {
        int l=S.size();
	if(K==1)
	{
		string res=S,now=S;
		for(int i=1;i<=l;i++)
		{
			string news="";
			news=news+now[l-1];
			for(int j=0;j<l-1;j++)
			{
				news=news+now[j];
			}
			if(comp(news,res))res=news;
			now=news;
		}
		return res;
	}
	else
	{
		int cnt[26];
		memset(cnt,0,sizeof(cnt));
		for(int i=0;i<l;i++)
		{
			cnt[S[i]-'a']++;
		}
		string res="";
		for(int i=0;i<26;i++)
		{
			for(int j=1;j<=cnt[i];j++)
			{
				char c=i+'a';
				res=res+c;
			}
		}
		return res;
	}
    }
};
