        if (K > 1)

        }/*

        int n=s.size();
        string x = s;
        string ret = x;
        for (int i=0; i<n; i++)

            char temp = x[0];
            x.erase(0, 1);
            x += temp;
            if (ret > x)
                ret = x;
        }
        return ret;
    }
