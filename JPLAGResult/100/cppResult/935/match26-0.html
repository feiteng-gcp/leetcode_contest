class Solution {
public:
    string orderlyQueue(string S, int K) {
        // if (S.length() == 1) return S;
        if (K == 1) {
            string ans = S;
            string tmp = S;
            for (int i = 1; i < S.size(); i++) {
                tmp = tmp.substr(1) + tmp[0];
                if (tmp < ans) ans = tmp;                
            }                        
            return ans;
        }
        sort(begin(S), end(S));
        return S;
    }
};
