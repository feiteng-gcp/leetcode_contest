class Solution {
public:
    string orderlyQueue(string S, int K) {
        if (K == 1) {
            string ans = S;
            for (int i = 1; i < S.size(); i++) {
                S = S.substr(1, S.size() - 1) + S.substr(0, 1);
                ans = min(ans, S);
            }
            return ans;
        } else {
            sort(S.begin(), S.end());
            return S;
        }
    }
};
