class Solution {
public:
    string orderlyQueue(string S, int K) {
        if(K>=2){
            sort(S.begin(),S.end());
            return S;
        }
        string ans=S;
        for(int i=1;i<S.size();i++){
            string tmp;
            for(int j=0;j<S.size();j++){
                tmp.push_back(S[(i+j)%S.size()]);
            }
            if(ans.compare(tmp)>0) ans=tmp;
        }
        return ans;
    }
};
