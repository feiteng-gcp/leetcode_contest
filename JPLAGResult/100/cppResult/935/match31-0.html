class Solution {
public:
    string orderlyQueue(string S, int K) {
        if(K==1){
            string s2=S;
            string ret=s2;
            for(int i=0; i<S.size(); i++){
                char c =s2[0];
                s2=s2.substr(1, s2.size()-1);
                s2+=c;
                ret=min(ret, s2);
            }
            return ret;
        }else{
            string s2=S;
            sort(s2.begin(), s2.end());
            return s2;
        }
    }
};
