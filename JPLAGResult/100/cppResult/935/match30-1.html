class Solution {
public:
    string func(string S){
        int a[26]={0};
        for(int i=0;i<S.size();i++){
            a[S[i]-'a']++;
        }
        string res="";
        for(int i=0;i<26;i++){
            for(int j=0;j<a[i];j++){
                res+='a'+i;
            }
        }
        return res;
    }
    string orderlyQueue(string S, int K) {
        if(S.size()<2) return S;

        if(K>1){
            return func(S);
        }else{
            string res=S;
            for(int i=1;i<=S.size();i++){
                string temp=S.substr(1,S.size()-1)+S[0];
                if(temp<res) res=temp;
                S=temp;
            }
            return res;
        }
    }
};
