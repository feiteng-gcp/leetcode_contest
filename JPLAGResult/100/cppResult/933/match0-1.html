/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
//先中序遍历

class Solution {
public:
    TreeNode* increasingBST(TreeNode* root) {
        if (root==NULL)
            return NULL;
        if (root->right==NULL && root->left==NULL)
            return root;
        TreeNode *p = root;
        stack<TreeNode *> ts;
        vector<TreeNode *> v;
        
        while (p!=NULL || !ts.empty())
        {
            if (p!=NULL)
			{
				ts.push(p);
				p = p->left;
			}
			else //p is null
			{
				TreeNode *t = ts.top();
				ts.pop();
                v.push_back(t);
				p = t->right;
			}
            
        }
    
        for (int i=0; i<v.size()-1; ++i)
        {
            v[i]->left = NULL;
            v[i]->right = v[i+1];
        }
        /*这两句少了会超时*/
       
        v[v.size()-1]->left = NULL;
        v[v.size()-1]->right = NULL;
        return v[0];
        
    }
};
