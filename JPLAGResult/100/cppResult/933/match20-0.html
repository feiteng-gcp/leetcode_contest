/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> T;
    void in(TreeNode* r) {
        if( r == NULL ) return;
        in(r->left);
        T.push_back(r->val);
        in(r->right);
    }
    TreeNode* increasingBST(TreeNode* root) {
        in(root);
        TreeNode* ans_r = new TreeNode(T[0]);
        TreeNode* pre = ans_r;
        TreeNode* cur;
        for( int i = 1 ; i < T.size() ; i++ ) {
            cur = new TreeNode(T[i]);
            pre->right = cur;
            pre = cur;
        }
        return ans_r;
    }
};
