/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* increasingBST(TreeNode* root) {
    if(!root)
        return root;
    vector<int> v;
    getV(root,v);
    TreeNode * ans=new TreeNode(v[0]);
    TreeNode *node=ans;
    for(int i=1;i<v.size();++i){
        node->right=new TreeNode(v[i]);
        node=node->right;
    }
    return ans;
}
    void getV(TreeNode* root,vector<int> &v){
    if(!root)
        return ;
    getV(root->left,v);
    v.push_back(root->val);
    getV(root->right,v);
}
};
