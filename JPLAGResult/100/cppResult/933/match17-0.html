/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* increasingBST(TreeNode* root) {
        vector<int> t;
        if(!root) return NULL;
        dfs(root,t);
        TreeNode *prev = new TreeNode(0),*head = prev;
        for(int i = 0; i < t.size(); i++){
            prev->right = new TreeNode(t[i]);
            prev =prev->right;
        }
        return head->right;
    }
    void dfs(TreeNode* root,vector<int>& t){
        if(!root) return;
        dfs(root->left,t);
        t.push_back(root->val);
        dfs(root->right,t);
    }
};
