/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> vals;
    void search(TreeNode* root){
        if (root!=nullptr){
            search(root->left);
            vals.push_back(root->val);
            search(root->right);
        }
    }
    TreeNode* increasingBST(TreeNode* root) {
        search(root);

        TreeNode* newrt = new TreeNode(vals[0]);
        TreeNode* cur=newrt;
        for (int i=1; i<vals.size(); i++){
            TreeNode* next = new TreeNode(vals[i]);
            cur->right=next;
            cur=next;
        }
        return newrt;
    }
};
