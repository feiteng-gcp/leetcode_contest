/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void indfs(TreeNode* r, vector<int> & vals) {
        if (r == nullptr) return;
        if (r->left) indfs(r->left, vals);
        vals.push_back(r->val);
        if (r->right) indfs(r->right, vals);
    }
    TreeNode* increasingBST(TreeNode* root) {
        if (root == nullptr) return nullptr;
        vector<int> vals;
        indfs(root, vals);
        TreeNode*r = new TreeNode(vals[0]);
        TreeNode*n = r;
        for (int i = 1; i < vals.size(); ++i) {
            n->right = new TreeNode(vals[i]);
            n = n->right;
        }
        return r;
    }
};
