/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void Go(vector<int>& v, TreeNode* root){
        if ( !root )
            return;
        
        Go(v, root->left);
        v.push_back( root->val );
        Go(v, root->right);
        
        
    }
    TreeNode* increasingBST(TreeNode* root) {
        
        if ( !root )
            return NULL;
        
        vector<int> v;
        Go(v, root);
        
        TreeNode * head = new TreeNode(v[0]);
        TreeNode *now = head;
        for ( int i = 1 ; i < v.size() ; i++ ){
            now->right = new TreeNode(v[i]);
            now = now->right;
        }
        
        return head;
        
        
    }
};
