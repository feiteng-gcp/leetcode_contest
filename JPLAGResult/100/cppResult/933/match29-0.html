/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;
    
    void dfs(TreeNode* a) {
        if (a->left)
            dfs(a->left);
        v.push_back(a->val);
        if (a->right)
            dfs(a->right);
    }
    TreeNode* increasingBST(TreeNode* root) {
        v.clear();
        dfs(root);
        TreeNode* la = 0;
        for (int i = v.size() - 1; i >= 0; i--){
            TreeNode* tmp = new TreeNode(v[i]);
            tmp->right = la;
            la = tmp;
        }
        return la;
    }
};
