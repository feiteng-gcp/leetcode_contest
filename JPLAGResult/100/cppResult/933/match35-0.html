/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
int store[105];
int cnt;
class Solution {
public:
    void get(TreeNode* root)
{
	if(root->left!=NULL)get(root->left);
	store[cnt++]=root->val;
	if(root->right!=NULL)get(root->right);
}
    TreeNode* increasingBST(TreeNode* root) {
        cnt=0;
	get(root);
	TreeNode *roots=new TreeNode(store[0]);
	TreeNode *p=roots;
	for(int i=1;i<cnt;i++)
	{
		TreeNode *now=new TreeNode(store[i]);
		p->right=now;
		p=now;
	}
	return roots;
    }
};
