/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* increasingBST(TreeNode* root) {
        vector<int> dict;
        worker(root, dict);
        if(!root) return nullptr;

        TreeNode* newRoot = new TreeNode(dict[0]);
        TreeNode* pRoot = newRoot;
        for(int i = 1; i < dict.size(); i++)
        {
            pRoot->right = new TreeNode(dict[i]);
            pRoot = pRoot->right;
        }
        return newRoot;
    }
    
    void worker(TreeNode* root, vector<int>& dict)
    {
        if(!root) return;
        worker(root->left, dict);
        dict.push_back(root->val);
        worker(root->right, dict);
    }
};
