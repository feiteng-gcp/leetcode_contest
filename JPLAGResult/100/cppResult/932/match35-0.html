class Solution {
public:
  bool isMonotonic(vector<int>& A) {
    if(A.size()<=1) return 1;
    bool good=1;
    for(int i=0;i+1<A.size();i++){
      if(A[i]>A[i+1]){
        good=0;
        break;
      }
    }
    if(good) return 1;
    good=1;
    for(int i=0;i+1<A.size();i++){
      if(A[i]<A[i+1]){
        good=0;
        break;
      }
    }
    if(good) return 1;
    return 0;
  }
};
