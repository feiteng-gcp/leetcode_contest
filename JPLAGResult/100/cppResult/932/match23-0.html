class Solution {
public:
    bool isMonotonic(vector<int>& A) {
        bool isZeng = true;
        for (int index = 0; index < A.size() - 1; index ++) {
            if(A[index + 1] < A[index]) {
                isZeng = false;
                break;
            }
        }
        if(isZeng == true) {
            return true;
        }
        
        bool isJian = true;
        for (int index = 0; index < A.size() - 1; index ++) {
            if(A[index + 1] > A[index]) {
                isJian = false;
                break;
            }
        }
        return isJian;
    }
};
