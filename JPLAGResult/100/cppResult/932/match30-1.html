class Solution {
public:
    bool isMonotonic(vector<int>& A) {
        bool t1,t2;
        t1 = t2 = true;
        for (int i = 1; i < A.size(); i++) {
            if (A[i] > A[i - 1]) 
                t1 = false;
            if (A[i] < A[i - 1]) 
                t2 = false;
        }
        return t1 || t2;
        
    }
};
