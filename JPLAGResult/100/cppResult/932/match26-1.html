class Solution {
public:
    bool isMonotonic(vector<int>& A) {
		int la=A.size();
		if(la<=1) return true;
		

		bool flag=true;
		for(int i=1; i<la; i++){
			if(A[i]<A[i-1]){
				flag=false; break;
			}
		}
		if(flag) return true;

		flag=true;
		for(int i=1; i<la; i++){
			if(A[i]>A[i-1]){
				flag=false; break;
			}
		}
		if(flag) return true;

		return false;
        
    }
};
