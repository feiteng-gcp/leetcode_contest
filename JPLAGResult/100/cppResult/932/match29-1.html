class Solution {
    bool incr(vector<int>& A) {
        int N = A.size();

        for (int i = 1; i < N; ++i) {
            if (A[i-1] > A[i])
                return false;
        }
        return true;
    }
    bool decr(vector<int>& A) {
        int N = A.size();

        for (int i = 1; i < N; ++i) {
            if (A[i-1] < A[i])
                return false;
        }
        return true;
    }
public:
    bool isMonotonic(vector<int>& A) {
        return incr(A) || decr(A);
    }
};
