class Solution {
public:
    bool isMonotonic(vector<int>& A) {
        int len = A.size();
        bool flag = true;
        for (int i = 1; i < len; i++) {
            if (A[i] < A[i - 1]) {
                flag = false;
                break;
            }
        }
        if (flag) return true;
        flag = true;
        for (int i = 1; i < len; i++) {
            if (A[i] > A[i - 1]) {
                flag = false;
                break;
            }
        }
        return flag;
    }
};
