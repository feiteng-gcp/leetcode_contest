class Solution {
public:
    bool isMonotonic(vector<int>& a) {
        bool ok = true;
        for (int i = 1; i < a.size(); ++i) if (a[i] < a[i - 1]) ok = false;
        if (ok) return true;
        ok = true;
        for (int i = 1; i < a.size(); ++i) if (a[i] > a[i - 1]) ok = false;
        return ok;
    }
};
