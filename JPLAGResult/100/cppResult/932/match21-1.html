class Solution {
public:
    bool isMonotonic(vector<int>& a) {

        int n = a.size();
        bool found = false;
        for (int i = 1; i < n; ++i) {
            if (a[i - 1] > a[i]) {
                found = true;
                break;
            }
        }
        if (!found) return true;
        found = false;

        for (int i = 1; i < n; ++i) {
            if (a[i - 1] < a[i]) {
                found = true;
                break;
            }
        }
        if (!found) return true;
        return false;
        
    }
};
