class Solution {
public:
    bool isMonotonic(vector<int>& A) {
        bool flag1 = true, flag2 = true;
        for(int i = 1; i < A.size(); i++) {
            if(A[i - 1] > A[i]) {
                flag1 = false;
                break;
            }
        }
        for(int i = 1; i < A.size(); i++) {
            if(A[i - 1] < A[i]) {
                flag2 = false;
                break;
            }
        }
        return flag1 || flag2;
    }
};
