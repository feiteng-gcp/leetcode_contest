class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        vector<int> last(32,-1);
        unordered_set<int> ans;
        for(int i=0;i<A.size();i++){
            ans.insert(A[i]);
            set<int> ll;
            for(int j=0;j<32;j++){
                if(A[i]&(1<<j)){
                    last[j]=i;
                    continue;
                }
                if(last[j]!=-1){
                    ll.insert(last[j]);
                }
            }
            int tmp=A[i];
            for(auto it=ll.rbegin();it!=ll.rend();it++){
                tmp|=A[*it];
                ans.insert(tmp);
            }
        }
        return ans.size();
    }
};
