class Solution {
public:
    int subarrayBitwiseORs(vector<int>& a) {
        int n=a.size();
        unordered_set<int> u,ans;        
        for(int i=0;i<n;i++){
            unordered_set<int> v;
            v.insert(a[i]);
            for(int x:u)v.insert(a[i]|x);
            ans.insert(v.begin(),v.end());
            u=v;
        }
        return ans.size();
    }
};
