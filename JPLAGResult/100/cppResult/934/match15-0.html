class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        unordered_set<int> end;
        unordered_set<int> nend;
        for(int i=0; i<A.size(); i++){
            unordered_set<int> end_next;
            end_next.insert(A[i]);
            for(const int& x:end){
                end_next.insert(A[i] | x);
            }
            for(const int& x:end){
                nend.insert(x);
            }
            end=end_next;
        }
        for(const int& x:end){
            nend.insert(x);
        }
        return nend.size();
    }
};
