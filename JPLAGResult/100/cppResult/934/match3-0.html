class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        unordered_set<int> s;
        int size = A.size(), total = 0;
            
        for (int i = size - 1; i >= 0; --i) {
            total |= A[i];
        }
        for (int i = size - 1; i >= 0; --i) {
            int ans = 0;
            for (int j = i; j >= 0; --j) {
                ans |= A[j];
                s.insert(ans);
                if (ans == total) {
                    break;
                }
            }
        }
        return s.size();
    }
};
