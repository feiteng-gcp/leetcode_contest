class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        int len=A.size();
        unordered_set<int> res, pre;
        res.insert(A[0]);
        pre.insert(A[0]);
        for (int i=1; i<len; i++)
        {
            unordered_set<int> temp;
            temp.insert(A[i]);
            for (auto j: pre)
            {
                temp.insert(j|A[i]);
            }
            for (auto j: temp)
                res.insert(j);
            pre=temp;
        }
        return res.size();
    }
};
