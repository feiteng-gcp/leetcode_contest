class Solution {
public:
    int subarrayBitwiseORs(vector<int>& a) {
        map<int,int>p;
        int ans=0;
        int n=a.size();
        vector<int>f(n,0);
        f[0]=a[0];
        p[a[0]]=1;
        ans=1;
        for(int i=1;i<n;++i){
            f[i]=f[i-1]|a[i];
            if(!p[a[i]]){
                p[a[i]]=1;
                ++ans;
            }
            if(!p[f[i]]){
                p[f[i]]=1;
                ++ans;
            }
        }
        for(int i=0;i<n;++i){
            int v=a[i];
            for(int j=i+1;j<n;++j){
                v=v^(v&a[j]);
                if(v==0)break;
                f[j]-=v;
                if(!p[f[j]]){
                    p[f[j]]=1;
                    ++ans;
                }
            }
        }
        return ans;
    }
};
