set<int> s;
class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        int n=A.size();
	s.clear();
	int ans=0;
	for(int i=0;i<n;i++)
	{
		int now=0;
		for(int j=i;j<n;j++)
		{
			now=now|A[j];
			if(s.find(now)==s.end())
			{
				ans++;
				s.insert(now);
			}
			if(i>0&&(now|A[i-1])==now)break;
		}
	}
	return ans;
    }
};
