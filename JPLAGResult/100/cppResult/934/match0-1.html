class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        set<int> R;
        set<int> P;
        
        for (auto i : A) {
            set<int> N;
            N.insert(i);
            for (auto p : P) {
                N.insert(i | p);
            }
            
            for (auto n : N) {
                R.insert(n);
            }
            
            P = N;
        }
        
        return R.size();
    }
};
