class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        unordered_set<int> result;
        
        unordered_set<int> last;
        for (auto a : A) {
            unordered_set<int> now;
            
            for (auto l : last) {
                now.insert(l | a);
                result.insert(l | a);
            }
            now.insert(a);
            result.insert(a);
            
            last = now;
        }
        
        return result.size();
    }
};
