class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        set<int> s;
        vector<int> nxt(30, -1);
        for (int i = A.size() - 1; i >= 0; i--) {
            for (int j = 0; j < 30; j++) {
                if (A[i] & (1 << j)) {
                    nxt[j] = i;
                }
            }
            vector<pair<int,int>> tmp;
            for (int j = 0; j  < 30; j++) {
                tmp.push_back({nxt[j], j});
            }
            sort(tmp.begin(), tmp.end());
            int cur = 0;
s.insert(A[i]);
            for (int j = 0; j < 30; ) {
                int k = j;
                while (k < 30 && tmp[k].first == tmp[j].first) {
                    if (tmp[j].first != -1)
                        cur |= 1 << tmp[k].second;
                    k++;
                }
                    if (tmp[j].first != -1) {
                // printf("%d %d\n", i, cur);
                s.insert(cur);}
                j = k;
            }
        }
        return s.size();
    }
};
