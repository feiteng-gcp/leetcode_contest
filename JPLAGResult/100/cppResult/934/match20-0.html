class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        set<int> total;
        set<int>  prev;

        for (int i = 0; i != A.size(); ++i) {
            set<int> curr;
            curr.insert(A[i]);
            for (int n : prev) {
                curr.insert(n | A[i]);
            }
            for (int n : curr) {
                total.insert(n);
            }
            curr.swap(prev);
        }
        return total.size();
    }
};
