class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        unordered_set<int> ret;
        unordered_set<int> p, q;
        for (int i = 0; i < A.size(); ++ i) {
            q.clear();
            q.insert(A[i]);
            ret.insert(A[i]);
            for (auto& x : p) {
                q.insert(x | A[i]);
                ret.insert(x | A[i]);
            }
            swap(p, q);
        }
        return ret.size();
    }
};
