class Solution {
public:
    int subarrayBitwiseORs(vector<int>& a) {
        unordered_set<int> A;
        int n = a.size();
        vector<int> tail(n);
        tail[n - 1] = a[n - 1];
        for (int i = n - 2; i >= 0; --i) {
            tail[i] = a[i] | tail[i + 1];
        }
        for (int i = 0; i < n; ++i) {
            int sum = a[i];
            A.insert(sum);
            for (int j = i + 1; j < n && (tail[j] & sum) != tail[j]; ++j) {
                sum |= a[j];
                A.insert(sum);
            }
        }
        return A.size();
    }
};
