class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        vector<int> power[32];
        unordered_set<int> cnt;
        for ( int i = 0 ; i < A.size() ; i++ ){
            for ( int p = 0 ; p < 31 ; p++ ){
                int n = 1 << p;
                if (A[i] & n)
                    power[p].push_back(i);
            }
        }
        
        int ans = 0;
        for ( int i = 0 ; i < A.size() ; i++ ){
            int a = A[i], ii;
            ii = i;
            cnt.insert(a);
            while (true){
                int ptr = 100000;
         
                for ( int p = 0 ; p < 31 ; p++ )
                    if ( !(a & (1<<p)) ){
                        auto it = lower_bound(power[p].begin(), power[p].end(), ii+1);
                        if ( it != power[p].end() )
                            ptr = min( *it, ptr );
  
                    }
                
                if ( ptr == 100000 )
                    break;
                a = a | A[ptr];
                
                cnt.insert(a);
                ii = ptr ;
                    
  
            }
            
            
        }
        return cnt.size();
        
        
    }
};
