class Solution {
public:
    unordered_set<int> vals;
    
    int subarrayBitwiseORs(vector<int>& A) {
        //we could do this in O(N^2) time by just starting sequences from each position and extending until the end
        //Be we can do it in O(32*N) time by stopping the extension after 32 positions since that is the maximum number of bits

        vector<int> accums = vector<int>(A.size(),0);
        
        for (int shift = 0; shift<32; shift++) {
            for (int pos = 0; pos < A.size(); pos++) {
                if (pos+shift<A.size()) {
                    accums[pos] |= A[pos+shift];
                    vals.insert(accums[pos]);
                }
            }
        }
        return vals.size();
    }
};
