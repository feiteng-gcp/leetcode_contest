class Solution {
public:
    int subarrayBitwiseORs(vector<int>& A) {
        set<int> ans;
        
        vector<int> pos[32];
        for (int i = 0; i < A.size(); ++i){
            for (int m = 0; m < 32; m++)
                if (A[i] & (1<<m))
                    pos[m].push_back(i);
        }
        
        
        for (int i = 0; i < A.size(); ++i){
            priority_queue<int> pq;
            ans.insert(A[i]);
            int mask = A[i];
            
            for (int m = 0; m < 32; m++){
                if (A[i] & (1<<m))
                    continue;
                
                int j = (int)(lower_bound(pos[m].begin(), pos[m].end(), i+1) - pos[m].begin());
                if (j < pos[m].size())
                    pq.push(-pos[m][j]);
            }
            
            while (pq.size()){
                int it = -pq.top();
                pq.pop();
                
                mask |= A[it];
                
                //cout << "mask = " << mask << ", i = " << i << ", it = " << it << endl;
                
                ans.insert(mask);
            }
        }
        
        return ans.size();
    }
};
