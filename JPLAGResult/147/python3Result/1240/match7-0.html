class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        ps = [0]
        for i in range(len(piles)):
            ps.append(ps[-1] + piles[i])
            
        #print(ps)
            
        mem = {}
        
        def dp(i, M):
            if i == len(piles):
                return 0
            if (i, M) in mem:
                return mem[(i, M)]
            res = 0
            for j in range(1, min(len(piles) - i, 2 * M) + 1):
                #print(i, i+j)
                res = max(res, (ps[i+j] - ps[i]) + (ps[-1] - ps[i+j] - dp(i+j, max(M, j))))
            mem[i, M] = res
            return res
        
        ans = dp(0, 1)
        #print(mem)
        return ans
