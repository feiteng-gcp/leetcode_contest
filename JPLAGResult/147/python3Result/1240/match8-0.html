class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        cache = {}
        def cal(pos, m):
            if pos == len(piles):
                return 0
            if (pos, m) in cache:
                return cache[(pos, m)]
            sum = 0
            
            x = pos
            while x < len(piles):
                sum = sum + piles[x]
                x = x + 1
            i = 1
            x = pos
            ret = 0
            while i <= 2 * m and x < len(piles):
                temp = sum - cal(x + 1, max(m, i))
                ret = max(ret, temp)
                i += 1
                x += 1
            cache[(pos, m)] = ret
            return ret
        return cal(0, 1)
