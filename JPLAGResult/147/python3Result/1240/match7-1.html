class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        memo = {}
        n = len(piles)
        S = [0]*(n+1)
        for i in range(n):
            S[i+1] = S[i] + piles[i]
        def dfs(i, m):
            key = (i, m)
            if key in memo:
                return memo[key]
            if i == n:
                return 0
            r = 0
            res = 0
            for x in range(1, 2*m+1):
                if i+x > n:
                    break
                r += piles[i+x-1]
                res = max(res, r + (S
[n] - S[i+x]) - dfs(i+x, max(m, x)))
                #print(i, m, x, r, (S[n] - S[i+x]), dfs(i+x, max(m, x)))
            memo[key] = res
            return res
        r = dfs(0, 1)
        #print(memo)
        return r
