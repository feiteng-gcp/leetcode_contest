class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        N = len(piles)
        dp = {}
        
        def calc(i, t, m):
            if i > N:
                return 0
            if (i, t, m) in dp:
                return dp[(i, t, m)]
            
            if t % 2 == 0:
                tmp = -float('inf')
                for k in range(1, 2 * m + 1):
                    tmp = max(tmp, sum(piles[i:i + k]) + calc(i + k, t ^ 1, max(m, k)))                    
                dp[(i, t, m)] = tmp
                return dp[(i, t, m)]
            else:
                tmp = float('inf')
                for k in range(1, 2 * m + 1):
                    tmp = min(tmp, calc(i + k, t ^ 1, max(m, k))) 
                dp[(i, t, m)] = tmp
                return dp[(i, t, m)]
        return calc(0, 0, 1)
