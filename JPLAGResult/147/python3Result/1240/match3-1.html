class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        self.memo = {}
        weight = [0]
        for i in range(len(piles)):
            weight += [weight[-1] + piles[-1-i]]
        weight.reverse()
        weight.pop()
        def dp(i,j):
            if (i,j) not in self.memo:
                if len(piles) - i <= 2*j:
                    self.memo[(i,j)]= weight[i]
                else:
                    tmp = 0
                    for k in range(1,2*j+1):
                        new = max(j,k)
                        tmp = max(tmp,weight[i]-dp(i+k,new))
                    self.memo[(i,j)] = tmp
            return self.memo[(i,j)]
        return dp(0,1)
