class Solution:

    def dfs(self, k, m):
        if (k, m) in self.mem:
            return self.mem[(k, m)]
        if k >= len(self.A):
            ans = 0
        else:
            ans = 0
            for x in range(1, 2*m+1):
                if x + k > len(self.A):
                    break
                this = self.subsum[k][len(self.A)-1] - self.dfs(k+x, max(m, x))
                ans = max(this, ans)

        self.mem[(k, m)] = ans
        return ans
    def stoneGameII(self, A):
        self.subsum = [[0]*len(A) for i in range(len(A))]
        for i in range(len(A)):
            for j in range(i, len(A)):
                self.subsum[i][j] = sum(A[i:j+1])
        self.A = A
        self.mem = {}
        return self.dfs(0, 1)
