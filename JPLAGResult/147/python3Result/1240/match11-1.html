class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        # dp[m][n] is the solution with M=m and piles[-n:]
        dp = [[0 for _ in range(len(piles) + 1)] for _ in range(len(piles) + 1)]
        
        for m in reversed(range(1, len(piles) + 1)):
            for n in range(1, len(piles) + 1):
                if 2*m >= n:
                    dp[m][n] = sum(piles[-n:])
                else:
                    remaining = sum(piles[-n:])
                    
for x in range(1, min(2*m, n) + 1):
                        dp[m][n] = max(dp[m][n], remaining - dp[max(m, x)][n - x])
                        
        return(dp[1][len(piles)])
