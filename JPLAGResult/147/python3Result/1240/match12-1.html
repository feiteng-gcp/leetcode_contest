class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        self.piles = piles
        self.pre_sum = []
        acc = 0
        for var in piles:
            acc += var
            self.pre_sum.append(acc)
        self.dic_a = {}
        self.dic_b = {}
        max_a, max_b = self.dfs(0, 1, 0)
        # print(max_a, max_b)
        return max_a
        
    def dfs(self, index, M, depth):
        if index >= len(self.piles):
            return (0, 0)
        if depth % 2 == 0 and (index, M) in self.dic_a:
            return self.dic_a[(index, M)]
        elif depth % 2 == 1 and (index, M) in self.dic_b:
            return self.dic_b[(index, M)]
        max_a, max_b 
= 0, 0
        for i in range(1, 2 * M + 1):
            if index + i > len(self.piles):
                break
            acc = self.pre_sum[index + i - 1] - self.pre_sum[index] + self.piles[index]
            a, b = self.dfs(index + i, max(M, i), depth + 1)
            if depth % 2 == 0:
                a += acc
            else:
                b += acc
            if (depth % 2 == 0 and a > max_a) or (depth % 2 == 1 and b > max_b):
                max_a = a
                max_b = b
        if depth % 2 == 0:
            self.dic_a[(index, M)] = (max_a, max_b)
        else:
            self.dic_b[(index, M)] = (max_a, max_b)
        return (max_a, max_b)
