class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        M = 1
        dic = {}
        def dp(startPos, M):
            if (startPos, M) in dic:
                return dic[(startPos, M)]
            if startPos + 2 * M > len(piles):
                dic[(startPos, M)] = sum(piles[startPos:])
                return dic[(startPos, M)]
            best = 0
            for X in range(1, 2 * M + 1):
                tempRes = sum(piles[startPos:]) - dp(startPos + X, max(M, X))
                if tempRes > best:
                    best = tempRes
            dic[(startPos, M)] = best
            return best
        return dp(0, 1)
