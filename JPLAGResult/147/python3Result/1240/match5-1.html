class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        return self.stoneGameIIHelper(piles, 1, 0, {})

    
def stoneGameIIHelper(self, piles, m, i, mem):
        if (m, i) in mem:
            return mem[(m, i)]
        if (i + 2 * m > len(piles)):
            mem[(m, i)] = sum(piles[i:])
            return mem[(m, i)]
        best = float('-inf')
        for ix in range(2 * m):
            x = ix + 1
            stonesNow = sum(piles[i:i + x])

            fewest = float('inf')
            newM = max(m, x)
            for iy in range(2 * newM):
                y = iy + 1
                if y < len(piles):
                    newerM = max(newM, y)
                    fewest = min(fewest, self.stoneGameIIHelper(piles, newerM, i + x + y, mem))
            best = max(best, stonesNow + fewest)
        mem[(m, i)] = best
        return best
