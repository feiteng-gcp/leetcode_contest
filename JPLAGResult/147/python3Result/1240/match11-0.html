class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        n = len(piles)
        ans = [[0]*(n+1) for _ in range(n+1)]
        sufsum = []
        for i in range(n-1, -1, -1):
            if i == n-1:
                sufsum.append(piles[i])
            else:
                sufsum.append(piles[i] + sufsum[-1])
        sufsum.reverse()
        sufsum.append(0)
        for j in range(1, n+1):
            for i in range(j, n+1):
                ans[i][j] = sufsum[n-j]
        for i in range(n-1, 0, -1):
            for j in range(i+1, n+1):
                for x in range(1, min(2*i, j)+1):
                    ans[i][j] = max(ans[i][j],  -ans[max(i, x)][j-x] + sufsum[n-j])
        return ans[1][n]
