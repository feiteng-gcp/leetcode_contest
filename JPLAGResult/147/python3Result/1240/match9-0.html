class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        pilesSum = [0]
        
        for i in range(len(piles)):
            pilesSum.append(pilesSum[-1] + piles[~i])
            
        pilesSum.reverse()
        # print(pilesSum)
        memo = {}
        def maxStones(i, M):
            
            if (i, M) not in memo:
                ans = 0
                for x in range(1, 2*M+1):
                    if i + x <= len(piles):
                        ans = max(ans, pilesSum[i] - maxStones(i+x, max(M, x)))
                memo[i, M] = ans
                
            return memo[i, M]
        
        return maxStones(0, 1)
