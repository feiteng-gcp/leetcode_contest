class Solution:
    def stoneGameII(self, piles: List[int]) -> int:
        if len(piles) <= 2:
            return sum(piles)

        lut = dict()

        def search(pos, M):
            if pos >= len(piles):
                return 0
            if (pos, M) in lut:
                return lut[(pos, M)]
            if pos + 2*M-1 >= len(piles):
                return sum(piles[pos:])

            cur_result = 0

            
for num in range(1, 2*M+1):
                cur_result = max(cur_result, sum(piles[pos:]) - search(pos+num, max(M, num)))

            lut[(pos, M)] = cur_result
            return cur_result

        return search(0, 1)
