class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        mapp = {}
        for i in range(len(board)):
            for j in range(len(board[i])):
                mapp[board[i][j]] = (j,i)
        #print(mapp)
        cur = (0,0)
        res = ""
        for t in target:
            ma = mapp[t]
            hmove = ma[0]-cur[0]
            vmove = ma
[1]-cur[1]
            
            if vmove<0:
                res+='U'*(-vmove)
            if hmove>0:
                res+='R'*hmove
            if hmove<0:
                res+='L'*(-hmove)
            if vmove>0:
                res+='D'*vmove
            res+='!'
            cur = ma
        return res
