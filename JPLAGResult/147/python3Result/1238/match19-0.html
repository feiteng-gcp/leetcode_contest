class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        def cal_rc(ch):
            return divmod(ord(ch) - 97, 5)
        now = 'a'
        r, c = 0, 0
        path = ''
        for x in target:
            t_r, t_c = cal_rc(x)
            if r > t_r:
                path += 'U' * (r - t_r)
                r = t_r
            if c > t_c:
                path += 'L' * (c - t_c)
                c = t_c
            if r < t_r:
                path += 'D' * (t_r - r)
                r = t_r
            if c < t_c:
                path += 'R' * (t_c - c)
                c = t_c
            path += '!'
        return path
