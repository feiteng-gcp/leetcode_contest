class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        pos = {}
        for i in range(len(board)):
            for j in range(len(board[i])):
                pos[board[i][j]] = (i, j)
        cur = (0, 0)
        path = ''
        for c in target:
            p = pos[c]
            if c == 'z':
                if p[1] < cur[1]:
                    path += 'L' * (cur[1] - p[1])
                elif p[1] > cur[1]:
                    path += 'R' * (p[1] - cur[1])
                if p[0] < cur[0]:
                    path += 'U' * (cur[0] - p[0])
                elif p[0] > cur[0]:
                    path += 'D' * (p[0] - cur[0])
            else:
                if p[0] < cur[0]:
                    path += 'U' * (cur[0] - p[0])
                elif p[0] > cur[0]:
                    path += 'D' * (p[0] - cur[0])
                if p[1] < cur[1]:
                    path += 'L' * (cur[1] - p[1])
                elif p[1] > cur[1]:
                    path += 'R' * (p[1] - cur[1])
            path += '!'
            cur = p
        return path
