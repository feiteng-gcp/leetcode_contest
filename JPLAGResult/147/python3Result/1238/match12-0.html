class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        n = len(board)
        mp = dict()
        for i in range(n):
            for j, c in enumerate(board[i]):
                mp[c] = (i, j)
        p = (0, 0)
        r = ''
        for c in target:
            if mp[c] == p:
                r += '!'
            else:
                x, y = mp[c][0] - p[0], mp[c][1] - p[1]
                if x < 0: r += 'U' * (-x)
                if y > 0: r += 'R' * y
                if y < 0: r += 'L' * (-y)
                if x > 0: r += 'D' * x
                r += '!'
                p = (mp[c][0], mp[c][1])
        return r
