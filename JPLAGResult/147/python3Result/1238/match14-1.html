class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        def convert(letter):
            dis = ord(letter) - ord('a')
            x = dis // 5
            y = dis % 5
            return x,y
        x,y = 0,0
        res = ''
        for c in target:
            nx,ny = convert(c)
            if x == 5 and y == 0:
                if nx > x:res += 'D' * (nx-x)
                elif nx < x: res += 'U'*(x-nx)
                if ny > y: res += 'R' * (ny-y)
                elif ny < y: res += 'L'*(y-ny)
            else:
                if ny > y: res += 'R' * (ny-y)
                elif ny < y: res += 'L'*(y-ny)
                if nx > x:res += 'D' * (nx-x)
                elif nx < x: res += 'U'*(x-nx)
            res += '!'
            x,y = nx,ny
        return res
