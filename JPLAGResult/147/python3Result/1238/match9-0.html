class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        prer,prec = 0,0
        res = ''
        for t in target:
            index = ord(t) - ord('a')
            r,c = index // 5,index%5
            rcount,dcount = c-prec,r-prer
            if t == 'z':
                if rcount > 0:
                    res += 'R'*rcount
                else:
                    res += 'L' *(-rcount)
                if dcount > 0:
                    res += 'D'*dcount
                else:
                    res += 'U' *(-dcount)
            else:
                if dcount > 0:
                    res += 'D'*dcount
                else:
                    res += 'U' *(-dcount)
                if rcount > 0:
                    res += 'R'*rcount
                else:
                    res += 'L' *(-rcount)
                
            res += '!'
            prer = r
            prec = c
        return res
