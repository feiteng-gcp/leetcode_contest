class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        res = ""
        idxes = collections.defaultdict(tuple)
        import string
        for ch in string.ascii_lowercase:
            for i, row in enumerate(board):
                if ch in row:
                    idxes[ch] = (i, row.find(ch))
        
        
        pre = (0, 0)
        for tarch in target:
            curidx = idxes[tarch]
            movelat = curidx[1] - pre[1]
            movever = curidx[0] - pre[0]
            # print(pre, curidx)
            # print(tarch, "movever", movever, "movelat", movelat)
            if tarch != "z":
                if movever > 0:
                    res = res + abs(movever) * "D"
                else:
                    res = res + abs(movever) * "U"

                if movelat > 0:
                    res = res + abs(movelat) * "R"
                else:
                    res = res + abs(movelat) * "L"
            else:
                if movelat > 0:
                    res = res + abs(movelat) * "R"
                else:
                    res = res + abs(movelat) * "L"
                if movever > 0:
                    res = res + abs(movever) * "D"
                else:
                    res = res + abs(movever) * "U"
            # print(res)
            res = res + "!"
            pre = curidx
            
        return res
