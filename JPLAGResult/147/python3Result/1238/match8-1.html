from typing import Tuple

class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        
        def get_position(char: str) -> Tuple[int, int]:
            r, c = divmod(ord(char) - ord('a'), 5)
            return r, c
        
        res = ''
        cur = 'a'
        cur_r, cur_c = 0, 0
        for char in target:
            next_r, next_c = get_position(char)
            move_r, move_c = next_r - cur_r, next_c - cur_c
            if cur == 'z':
                res += -move_r * 'U'
                res += move_c * 'R'
            elif char == 'z':
                res += -move_c * 'L'
                res += move_r * 'D'
            else:
                
if move_r >= 0:
                    res += move_r * 'D'
                else:
                    res += -move_r * 'U'
                if move_c >= 0:
                    res += move_c * 'R'
                else:
                    res += -move_c * 'L'
            res += '!'
            cur = char
            cur_r, cur_c = next_r, next_c
        return res
