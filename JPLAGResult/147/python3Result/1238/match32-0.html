class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        a = {}
        b = {}
        a['z'] = 5
        b['z'] = 0
        for i in range(5):
            for j in range(5):
                a[board[i][j]] = i
                b[board[i][j]] = j
        res = ''
        x,y = 0,0
        for e in target:
            xx = a[e]
            yy = b[e]
            if e != 'z':
                if x > xx:
                    res += 'U'*(x-xx)
                elif xx > x:
                    res += 'D'*(xx-x)
                if y > yy:
                    res += 'L'*(y-yy)
                elif yy > y:
                    res += 'R'*(yy-y)
            else:
                if y > yy:
                    res += 'L'*(y-yy)
                elif yy > y:
                    res += 'R'*(yy-y)
                if x > xx:
                    res += 'U'*(x-xx)
                elif xx > x:
                    res += 'D'*(xx-x)
            res += '!'
            x,y = xx,yy
        return res
