def subtract (A, B):
    return (A[0]-B[0], A[1]-B[1])

def translate (char, length):
    if length >= 0:
        return char * length
    else:
        return {'D':'U', 'R':'L', 'U':'D', 'L':'R'}[char] * (-length)

class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        pos = {}
        for row in range (len (board)):
            for col in range (len (board[row])):
                char = board[row][col]
                pos[char] = (row, col)
        curr = (0, 0)
        ans = []
        for char in target:
            newcord = pos[char]
            diff = subtract (newcord, curr)
            (drow, dcol) = diff
            if drow >= 0:
                ans.append (translate ('R', dcol))
                ans.append (translate ('D', drow))
            else:
                ans.append (translate ('D', drow))
                ans.append (translate ('R', dcol))
            ans.append ('!')
            curr = newcord
        return ''.join (ans)
