class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        loc = {}
        for i in range(len(board)):
            for j in range(len(board[i])):
                loc[board[i][j]] = (i,j)
        cur = (0,0)
        res = ''
        for i in target:
            if i == 'z' and cur!=(5,0):
                v = loc['u'][0] - cur[0]
                h = loc['u'][1] - cur[1]
                if v > 0:
                    res += 'D' * v
                if v < 0:
                    res += 'U' * (-v)
                if h > 0:
                    res += 'R' * h
                if h < 0:
                    res += 'L' * (-h)
                res += 'D'
                res += '!'
                cur = (5,0)
            elif cur == (5,0) and i != 'z':
                res += 'U'
                cur = (4,0)
                v = loc[i][0] - cur[0]
                h = loc[i][1] - cur[1]
                if v > 0:
                    res += 'D' * v
                if v < 0:
                    res += 'U' * (-v)
                if h > 0:
                    res += 'R' * h
                if h < 0:
                    res += 'L' * (-h)
                res += '!'
                cur = loc[i]
            else:
                v = loc[i][0] - cur[0]
                h = loc[i][1] - cur[1]
                if v > 0:
                    res += 'D' * v
                if v < 0:
                    res += 'U' * (-v)
                if h > 0:
                    res += 'R' * h
                if h < 0:
                    res += 'L' * (-h)
                res += '!'
                cur = loc[i]
        return res
