class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        alphabet = collections.defaultdict(list)
        for index in range(len(board)):
            for jndex in range(len(board[index])):
                alphabet[board[index][jndex]] = [index,jndex]
        sequence = []
        startpoint = [0,0]
        ans = ""
        for letter in target:
            endpoint = alphabet[letter]
            s1, s2 = startpoint[0], startpoint[1]
            e1, e2 = endpoint[0], endpoint[1]
            while s2 > e2:
                ans += "L"
                s2 -= 1
            while s1 > e1:
                ans += "U"
                s1 -= 1
            while s2 < e2:
                ans += "R"
                s2 += 1
            while s1 < e1:
                ans += "D"
                s1 += 1
            
            ans += "!"
            startpoint = endpoint[:]
        return ans
