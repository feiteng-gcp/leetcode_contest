class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        res = ""
        pos = [0, 0]
        for char in target:
            total = ord(char) - ord('a')
            nextPos = [total // 5, total % 5]
            if pos[0] == 5 and nextPos[0] != 5:
                pos[0] -= 1
                res += "U"
                
            if pos[1] > nextPos[1]:
                res += "L" * (pos[1] - nextPos[1])
            else:
                res += "R" * (nextPos[1] - pos[1])
            
            if pos[0] > nextPos[0]:
                res += "U" * (pos[0] - nextPos[0])
            else:
                res += "D" * (nextPos[0] - pos[0])
            
            res += "!"
            pos = nextPos
        
        return res
