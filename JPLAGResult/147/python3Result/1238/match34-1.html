class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        ret = ''
        x, y = 0, 0
        base = ord('a')
        for c in target:
            idx = ord(c) - base
            i, j = idx // 5, idx % 5
            
while x > i:
                ret += 'U'
                x -= 1
            while y > j:
                ret += 'L'
                y -= 1
            while y < j:
                ret += 'R'
                y += 1
            while x < i:
                ret += 'D'
                x += 1
            if x == i and y == j:
                ret += '!'
        return ret
