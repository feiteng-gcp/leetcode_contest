class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        ans = ''
        cx, cy = 0, 0
        for c in target:
            x, y = (ord(c) - ord('a')) // 5, (ord(c) - ord('a')) % 5
            if y < cy:
                ans += 'L' * (cy - y)
            if x < cx:
                ans += 'U' * (cx - x)
            if x > cx:
                ans += 'D' * (x - cx)
            if y > cy:
                ans += 'R' * (y - cy)
            cx, cy = x, y
            ans += '!'
        return ans
