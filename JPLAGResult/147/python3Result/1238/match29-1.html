class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        def getPosition(c):
            diff = ord(c) - ord('a')
            return diff // 5, diff % 5
        result = ''
        pos = (0, 0)
        for c in target:
            newpos = getPosition(c)
            
            if c == 'z':
                if newpos[1] < pos[1]: result += (pos[1] - newpos[1]) * 'L'
                else:                  result += (newpos[1] - pos[
1]) * 'R'
                if newpos[0] < pos[0]: result += (pos[0] - newpos[0]) * 'U'
                else:                  result += (newpos[0] - pos[0]) * 'D'
            else:
                if newpos[0] < pos[0]: result += (pos[0] - newpos[0]) * 'U'
                else:                  result += (newpos[0] - pos[0]) * 'D'
                if newpos[1] < pos[1]: result += (pos[1] - newpos[1]) * 'L'
                else:                  result += (newpos[1] - pos[1]) * 'R'
                
            result += '!'
            pos = newpos
        return result
