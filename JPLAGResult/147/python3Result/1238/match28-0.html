class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        n = len(target)
        p = 'a'
        pre = (0, 0)
        ans = ''
        for c in target:
            if p == c:
                ans += '!'
                continue
            cur = ((ord(c) - ord('a'))//5, (ord(c) - ord('a'))%5)
            dx, dy = cur[0]-pre[0], cur[1]-pre[1]
            if p == 'z':
                ans += 'U'*(-dx) + 'R'*dy
            elif c == 'z':
                ans += 'L'*(-dy) + 'D'*dx
            else:
                if dx >= 0:
                    ans += 'D'*dx
                else:
                    ans += 'U'*(-dx)
                if dy >= 0:
                    ans += 'R'*dy
                else:
                    ans += 'L'*(-dy)
            ans += '!'
            p = c
            pre = cur
        return ans
