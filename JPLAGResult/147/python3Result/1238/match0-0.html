class Solution:
    def alphabetBoardPath(self, target: str) -> str:
        board = ["abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"]
        pos = {}
        for i in range(0, len(board)):
            for j in range(0, len(board[i])):
                pos[board[i][j]] = (i, j)
        ret = []
        x = 0
        y = 0
        last = 'a'
        for i in range(len(target)):
            c = target[i]
            if last == 'z' and c != 'z':
                ret.append('U')
                x -= 1
            while y < pos[c][1]:
                y += 1
                ret.append('R')
            while y > pos[c][1]:
                y -= 1
                ret.append('L')
            while x < pos[c][0]:
                x += 1
                ret.append('D')
            while x > pos[c][0]:
                x -= 1
                ret.append('U')
            ret.append('!')
            last = c
        return "".join(ret)
