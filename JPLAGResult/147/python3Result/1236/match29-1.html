class Solution:
    def tribonacci(self, n: int) -> int:
        import functools
        @functools.lru_cache(maxsize=None)
        def fib(n):
            if n == 0:
                return 0
            if n <= 2:
                return 1
            else:
                return fib(n - 1) + fib(n - 2) + fib(n-3)
        return fib(n)
