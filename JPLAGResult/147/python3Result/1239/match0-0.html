class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        ret = 0
        n = len(grid)
        m = len(grid[0])
        
        def helper(i, j, l):
            for k in range(l):
                if grid[i][j - k] == 0 or grid[i - l + 1][j - k] == 0:
                    return False
                if grid[i - k][j] == 0 or grid[i - k][j - l + 1] == 0:
                    return False
            return True
        
        for l in range(1, min(n, m) + 1):
            for i in range(l - 1, n):
                if ret == l:
                    break
                for j in range(l - 1, m):
                    if ret == l:
                        break
                    if helper(i, j, l):
                        ret = l
        return ret * ret
