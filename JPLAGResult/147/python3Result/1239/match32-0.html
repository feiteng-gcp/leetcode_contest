class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        nrow = len(grid)
        ncol = len(grid[0])
        # Brute force
        # Find the longest consecutive 1
        dic = [[[1, 1] for _ in range(ncol)] for _ in range(nrow)]
        for i in range(nrow):
            for j in range(ncol):
                if j == 0:
                    dic[i][j][0] = grid[i][j]
                else:
                    dic[i][j][0] = 1 + dic[i][j - 1][0] if grid[i][j] else 0
                    
        for j in range(ncol):
            for i in range(nrow):
                if i == 0:
                    dic[i][j][1] = grid[i][j]
                else:
                    dic[i][j][1] = 1 + dic[i - 1][j][1] if grid[i][j] else 0
        
        # print(dic)
        M = 0
        for i in range(nrow):
            for j in range(ncol):
                possible = min(dic[i][j])
                # print(i, j , possible)
                for minus in range(0, possible):
                    upright = (i - possible + minus + 1, j)
                    downleft = (i, j - possible + minus + 1)
                    # print("  ", upright, downleft)
                    # print("upright", dic[upright[0]][upright[1]][0], "downleft", dic[downleft[0]][downleft[1]][1])
                    if dic[upright[0]][upright[1]][0] >= possible - minus and dic[downleft[0]][downleft[1]][1] >= possible - minus:
                        M = max(M, possible - minus)
                        break
                
        return M * M
