class Solution:
    def largest1BorderedSquare(self, g: List[List[int]]) -> int:
        rows,cols = len(g), len(g[0])
        a = [[0] * cols for _ in range(rows)]
        a[0][0] = [g[0][0] , g[0][0]]
        for i in range(1, rows):
            a[i][0] = [g[i][0], 0 if g[i][0] == 0 else a[i-1][0][1] + 1]
        for j in range(1, cols):
            a[0][j] = [0 if g[0][j] == 0 else a[0][j-1][0] + 1, g[0][j]]
            
        #print(a)
        
        res = max(0, sum(sum(g,[])) > 0)
        for i in range(1, rows):
            for j in range(1, cols):
                a[i][j] = [0, 0]
                if not g[i][j]: continue
                    
                a[i][j][0] = 0 if not g[i][j] else a[i][j-1][0] + 1
                a[i][j][1] = 0 if not g[i][j] else a[i-1][j][1] + 1
                
                res = max(1, res)
                for x in range(min(a[i][j][0], a[i][j][1])-1, 0, -1):
                    if a[i][j-x][1] >= x+1 and a[i-x][j][0] >= x+1:
                        res = max(res, x+1)
                        break
        return res * res
