class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        ans = 0
        m, n = len(grid), len(grid[0])
        left = [[0]*n for _ in range(m)]
        up = [[0]*n for _ in range(m)]
        for y in range(n):
            for x in range(m):
                if x == 0:
                    up[x][y] = grid[x][y]
                elif grid[x][y] == 1:
                    up[x][y] = up[x-1][y] + 1
        for x in range(m):
            for y in range(n):
                if y == 0:
                    left[x][y] = grid[x][y]
                elif grid[x][y] == 1:
                    left[x][y] = left[x][y-1] + 1
        for u in range(m):
            for d in range(u, m):
                length = d-u+1
                if ans >= length**2:
                    continue
                for r in range(length-1, n):
                    if left[u][r] >= length and left[d][r] >= length and up[d][r-length+1] >= length and up[d][r] >= length:
                        if length**2 > ans:
                            ans = length**2
                            break
        return ans
