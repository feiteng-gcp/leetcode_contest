class Solution:
  def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
    n = len(grid)
    if n == 0: return 0
    m = len(grid[0])

    right = [[0] * (m + 1) for _ in range(n + 1)]
    down = [[0] * (m + 1) for _ in range(n + 1)]

    for row in range(n - 1, -1, -1):
      for col in range(m - 1, -1, -1):
        if grid[row][col]:
          right[row][col] = right[row][col + 1] + 1
          down[row][col] = down[row + 1][col] + 1

    ans = 0
    for i in range(n):
      for j in range(m):
        for a in range(min(n - i, m - j) - 1, -1, -1):
          if right[i][j] > a and down[i][j] > a:
            if right[i + a][j] > a and down[i][j + a] > a:
              ans = max(ans, (a + 1) ** 2)
              break

    return ans
