class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        
        up = [[0] * n for i in range(m)]
        left = [[0] * n for i in range
(m)]
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    up[i][j] = (0 if i == 0 else up[i - 1][j]) + 1
                    left[i][j] = (0 if j == 0 else left[i][j - 1]) + 1
        
        res = 0
        for i in range(m):
            for j in range(n):
                maxlength = min(up[i][j], left[i][j])
                for l in range(maxlength, 0, -1):
                    _i, _j = i - l + 1, j - l + 1
                    if up
[i][_j] >= l and left[_i][j] >= l:
                        res = max(res, l ** 2)
                        break
        return res
