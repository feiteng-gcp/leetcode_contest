class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        m,n = len(grid),len(grid[0])
        dp = [[[0,0] for _ in range(n)] for _ in range(m)]
        res = 0
        for i in range(m):
            if grid[i][0] == 1:
                dp[i][0][0] = 1
                if i == 0:
                    dp[i][0][1] = 1
                else:
                    dp[i][0][1] = dp[i-1][0][1] + 1
                res = 1
        for i in range(n):
            if grid[0][i] == 1:
                dp[0][i][1] = 1
                if i == 0:
                    dp[0][i][0] = 1
                else:
                    dp[0][i][0] = dp[0][i-1][0] + 1
                res = 1
        for i in range(1,m):
            for j in range(1,n):
                if grid[i][j] == 1:
                    dp[i][j][0] += dp[i][j-1][0] + 1
                    dp[i][j][1] += dp[i-1][j][1] + 1
                    mn = min(dp[i][j-1][0],dp[i-1][j][1])
                    for k in range(mn,-1,-1):
                        r = i - k
                        c = j - k
                        if dp[r][j][0] >= k + 1 and dp[i][c][1] >=k:
                            res = max(res,(k+1)**2)
        return res
