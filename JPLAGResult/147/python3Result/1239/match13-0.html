class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        res = 0
        
        lefts = [[0 for _ in range(n)] for _ in range(m)]
        rights = [[0 for _ in range(n)] for _ in range(m)]
        downs = [[0 for _ in range(n)] for _ in range(m)]
        ups = [[0 for _ in range(n)] for _ in range(m)]
        
        for i in range(m):
            for j in range(n):
                for k in range(i, -1, -1):
                    if grid[k][j] == 1:
                        ups[i][j] += 1
                    else:
                        break
                for k in range(i, m):
                    if grid[k][j] == 1:
                        downs[i][j] += 1
                    else:
                        break
                        
                for k in range(j, -1, -1):
                    if grid[i][k] == 1:
                        lefts[i][j] += 1
                    else:
                        break
                        
                for k in range(j, n):
                    if grid[i][k] == 1:
                        rights[i][j] += 1
                    else:
                        break
                        
        for i in range(m):
            for j in range(n):
                limit = min(rights[i][j], downs[i][j])
                for s in range(1, limit + 1):
                    if downs[i][j + s - 1] >= s and rights[i + s - 1][j] >= s:
                        res = max(res, s * s)
        return res
