class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        
        m, n = len(grid), len(grid[0])
        cols = [[0]*n for _ in range(m)]
        rows = [[0]*n for _ in range(m)]
        for i in range(m):
            num = 0
            for j in range(n):
                if grid[i][j] == 0: num = 0
                else: num += 1
                cols[i][j] = num
        for j in range(n):
            num = 0
            for i in range(m):
                if grid[i][j] == 0: num = 0
                else: num += 1
                rows[i][j] = num
        res = 0
        for i in range(m)[::-1]:
            for j in range(n)[::-1]:
                size = min(cols[i][j], rows[i][j])
                while size > res:
                    if rows[i][j-size+1] >= size and cols[i-size+1][j] >= size:
                        res = size
                    size -= 1
        return res*res
