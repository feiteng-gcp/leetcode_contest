class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        aux = [[[grid[r][c]] * 4 for c in range(n)] for r in range(m)]
        for r in range(1, m):
            for c in range(1, n):
                if grid[r]
[c]:
                    aux[r][c][0] += aux[r - 1][c][0]
                    aux[r][c][1] += aux[r][c - 1][1]
        for r in range(m - 1)[::-1]:
            for c in range(n - 1)[::-1]:
                if grid[r]
[c]:
                    aux[r][c][2] += aux[r + 1][c][2]
                    aux[r][c][3] += aux
[r][c + 1][3]
        ans = 0
        for r in range(m):
            for c in range(n):
                L1 = min(aux[r][c][:2])
                for d in range(L1)[::-1]:
                    L2 = min(aux[r - d][c - d][-2:])
                    if L2 >= d:
                        ans = max(ans, (d + 1) * (d + 1))
                        break
        return ans
