class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        h = len(grid)
        w = len(grid[0])
        max_h = [[0 for _ in range(w)] for _ in range
(h)]
        for x in range(h):
            cnt = 0
            for y in range(w-1,-1,-1):
                if grid[x][y]:
                    cnt += 1
                else:
                    cnt = 0
                max_h[x][y] = cnt
        max_v = [[0 for _ in range(w)] for _ in range(h)]
        
for y in range(w):
            cnt = 0
            for x in range(h-1,-1,-1):
                if grid[x][y]:
                    cnt += 1
                else:
                    cnt = 0
                max_v[x][y] = cnt

        def test(x, y, siz):
            return (max_h[x][y] >= siz) and (max_v[x][y] >= siz) and (max_h[x+siz-1][y] >= siz) and (max_v[x][y+siz-1] >= siz)
        
        min_hw = min(h,w)
        best = 0
        for x in range(h):
            for y in range(w):
                for s in range(min_hw,best,-1):
                    if test(x,y,s):
                        print(x,y,s)
                        best = s
                        break
        return best**2
