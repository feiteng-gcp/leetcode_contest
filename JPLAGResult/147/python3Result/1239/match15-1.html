class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        up = [[0]*len(grid[0]) for _ in range(len(grid))]
        left = [[0]*len(grid[0]) for _ in range(len(grid))]
        
for r in range(len(grid)):
            for c in range(len(grid[0])):
                if grid[r][c] == 0:
                    left[r][c] = 0
                else:
                    left[r][c] = 1
                    if c > 0:
                        left[r][c] += left[r][c-1]
        for c in range(len(grid[0])):
            for r in range(len(grid)):
                if grid[r][c] == 0:
                    up[r][c] = 0
                else:
                    up[r][c] = 1
                    if r > 0:
                        up[r][c] += up[r-1][c
]
        ans = 0
        for r in range(len(grid)):
            for c in range(len(grid[0])):
                for s in range(1, min(up[r][c], left[r][c])+1):
                    if r-s+1 >= 0 and c-s+1 >= 0 and left[r-s+1][c] >= s and up[r][c-s+1] >= s:
                        ans = max(ans, s*s)
        return ans
