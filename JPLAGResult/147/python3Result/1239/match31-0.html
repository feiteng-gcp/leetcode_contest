class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        up=[[0]*len(grid[0]) for _ in range(len(grid))]
        left=copy.deepcopy(up)
        res=0
        #up
        for i in range(len(grid[0])):
            if grid[0][i]:
                up[0][i]=1
                res=1
        for i in range(1,len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j]:
                    up[i][j]=up[i-1][j]+1
                    res=1
        #left
        for i in range(len(grid)):
            if grid[i][0]:
                left[i][0]=1
                res=1
        for i in range(len(grid)):
            for j in range(1,len(grid[0])):
                if grid[i][j]:
                    left[i][j]=left[i][j-1]+1
                    res=1
        
        for i in range(1,len(grid)):
            for j in range(1,len(grid[0])):
                if grid[i][j]:
                    tmp=min(left[i][j],up[i][j])
                    for k in range(1,tmp):
                        if up[i][j-k]>k and left[i-k][j]>k:
                            res=max(res,(k+1)**2)
        #print(left,up)                
        return res
