class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        left = [row[:] for row in grid]
        up   = [row[:] for row in grid]
        
        
for i in range(len(grid)):
            for j in range(len(grid[i])):
                if j > 0 and left[i][j]: left[i][j] += left[i][j-
1]
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if i > 0 and up[i][j]: up[i][j] += up[i-1][j]
                    
        result = 0
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                for k in range(result + 1, min(left[i][j], up[i][j]) + 1):
                    if left[i-k+1][j] >= k and up[i][j-k+1] >= k:
                        result = max(result, k)
        return result**2
