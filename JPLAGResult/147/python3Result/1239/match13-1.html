class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        R = len(grid)
        C = len(grid[0])
        right 
= [[0 for _ in range(C)] for _ in range(R)]
        down = [[0 for _ in range(C)] for _ in range(R)]
        up = [[0 for _ in range(C)] for _ in range(R)]
        left = [[0 for _ in range(C)] for _ in range(R)]
        
        for i in range(R):
            for j in range(C):
                if grid[i][j]:
                    left[i][j] = grid[i][j] + (left[i][j-1] if j - 1 >= 0  else 0)
                    up[i][j] = grid[i][j] + (up[i-1][j] if i - 1 >= 0 else 0)
        for i in range(R-1, -1, -1):
            for j in range(C-1, -1, -1):
                if grid[i][j]:
                    down[i][j] = grid[i][j] + (down[i+1][j] if i + 1 < R else 0)
                    right[i][j] = grid[i][j] + (right[i][j+1] if j + 1 < C else 0)
        
        ans = 0
        
for i in range(R):
            for j in range(C):
                curr = min(down[i][j], right[i][j])
                for z in range(ans, curr):
                    if i + z >= R or j + z >= C:
                        break                    
                    if min(up
[i+z][j+z], left[i+z][j+z]) >= z:
                        # print(i)
                        # print(j)
                        # print(z)
                        # print(down[i][j])
                        # print(right[i][j])
                        # print(up[i+z][j+z])
                        # print(left[i+z][j+z])
                        # print('---')
                        ans = max(ans, z+1)
        # print(down)
        # print(right)
        # print(up)
        # print(left)
        return ans * ans
