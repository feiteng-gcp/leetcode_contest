class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        up, down, left, right = [[0 for _ in range(n)] for _ in range(m)], [[0 for _ in range(n)] for _ in range(m)], [[0 for _ in range(n)] for _ in range(m)], [[0 for _ in range(n)] for _ in range(m)]
        hasOne = False
        
for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    up[i][j] = up[i - 1][j] + 1 if i > 0 else 1
                    left[i][j] = left[i][j - 1] + 1 if j > 0 else 1
                    hasOne =
 True
        for i in range(m - 1, -1, -1):
            for j in range(n - 1, -1, -1):
                if grid[i][j] == 1:
                    down[i][j] = down[i + 1][j] + 1 if i < m - 1 else 1
                    right[i][j] = right[i][j + 1] + 1 if j < n - 1 else 1
                    hasOne = True
        if not hasOne:
            return 0
        # print(left)
        # print(right)
        # print(up)
        # print(down)
        res = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    continue
                for k in range(0, min(m - i, n - j)):
                    if grid[i + k][j + k] == 0:
                        continue
                    if right[i][j] > k and down[i][j] > k and left
[i + k][j + k] > k and up[i + k][j + k] > k:
                        res = max(res, (k + 1)**2)
        return res
