class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        M, N = len(grid), len(grid[0])
        # [from_top, from_left]
        dp = [[[0, 0] for _ in range(N+1)] for _ in range(M+1)]

        max_len = 0
        for i in range(1, M+1):
            for j in range(1, N+1):
                if grid[i-1][j-1] == 0:
                    # default value is [0, 0]
                    # dp[i][j] = [0, 0]
                    continue

                dp
[i][j][0] = dp[i-1][j][0]+1
                dp[i][j][1] = dp[i][j-1][1]+1
                min_len = min(dp[i][j])
                for cur_len in range(min_len, 0, -1):
                    s_col, s_row = i-cur_len+1, j-cur_len+1
                    if dp[s_col][j][1] >= cur_len and dp[i][s_row][0] >= cur_len:
                        max_len = max(max_len, cur_len**2)
                        break

        return max_len
