class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        memo_h = {}
        memo_v = {}
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                memo_v[(i,j)] = 0
                for k in range(i,len(grid)):
                    if grid[k][j]:
                        memo_v[(i,j)] = k-i+1
                    else:
                        break
                memo_h[(i,j)] = 0
                for k in range(j,len(grid[0])):
                    if grid[i][k]:
                        memo_h[(i,j)] = k - j+1
                    else:
                        break
        res = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                k = 1
                while i+k <= len(grid) and j + k <= len(grid[0]):
                    if memo_v[(i,j)] >= k and memo_h[(i,j)] >= k and memo_v[(i,j+k-1)] >= k and memo_h[(i+k-1,j)] >= k:
                        res = max(res,k*k)
                    k += 1
        return res
