class Solution:
    def largest1BorderedSquare(self, grid: List[List[int]]) -> int:
        row = len(grid)
        col = len(grid[0])
        u = [[0]*(col+1) for _ in range(row+1)]
        l = [[0]*(col+1) for _ in range(row+1)]
        for i in range(row):
            for j in range(col):
                if grid[i][j] == 0:
                    continue
                u[i+1][j+1] = u[i][j+1] + 1
                l[i+1][j+1] = l[i+1][j] + 1
        res = 0
        for i in range(row):
            for j in range(col):
                if grid[i][j] == 0:
                    continue
                t = min(u[i+1][j+1],l[i+1][j+1])
                for k in range(t):
                    if l[i+1-k][j+1] > k and u[i+1][j+1-k] > k and k+1 > res:
                        res = k+1
        #print(u,l)
        return res**2
