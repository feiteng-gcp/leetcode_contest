class Solution {
   public int largest1BorderedSquare(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;

        int[][] left = new int[m][n];
        int[][] up = new int[m][n];

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0) continue;

                if (j - 1 >= 0) left[i][j] = left[i][j - 1] + 1;
                else left[i][j] = 1;

                if (i - 1 >= 0) up[i][j] = up[i - 1][j] + 1;
                else up[i][j] = 1;
            }
        }

        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 1; k <= n && k <= m; k++) {

                    if (up[i][j] >= k && left[i][j] >= k && up[i][j - k + 1] >= k && left[i - k + 1][j] >= k) {
                        ans = Math.max(ans, k * k);
                    }
                }
            }
        }
        return ans;
    }
}
