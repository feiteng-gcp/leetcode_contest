class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int rows = grid.length;
        int cols = grid[0].length;
        int[][] subX = new int[rows][cols + 1];
        int[][] subY = new int[rows + 1][cols];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                subX[i][j + 1] = subX[i][j] + grid[i][j];
                subY[i + 1][j] = subY[i][j] + grid[i][j];
            }
        }
        int max = Integer.min(rows, cols);
        for (int l = max; l > 0; l--) {
            for (int i = 0; i + l <= rows; i++) {
                int ii = i + l - 1;
                for (int j = 0; j + l <= cols; j++) {
                    int jj = j + l - 1;
                    if (subX[i][jj + 1] - subX[i][j] == l &&
                            subX[ii][jj + 1] - subX[ii][j] == l &&
                            subY[ii + 1][j] - subY[i][j] == l &&
                            subY[ii + 1][jj] - subY[i][jj] == l) {
                        return l * l;
                    }
                }
            }
        }
        return 0;
    }
}
