class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int height = grid.length, width = grid[0].length;
        
int[][][] dp = new int[height][width][2];
        for(int i =0 ; i < height; i++){
            for(int j = 0; j < width; j++){
                if(grid[i][j] == 0)
                    continue;
                if(dp[i][j][0] == 0)
                    collectY(grid, dp, i, j);
                if(dp[i][j][1] == 0)
                    collectX(grid, dp, i, j)
;
            }
        }
        // for(int i = 0; i < height; i++){
        //      for(int j = 0; j < width; j++){
        //          System.out.println("y:" + i + " x:" + j + " val:" + Arrays.toString(dp[i][j]));
        //      }
        // }
        
        int max = 0;
        for(int i = 0; i < height; i++){
            for(int j = 0; j < width; j++){
                for(int k = 0; k < dp[i][j][1] && i + k < height && j + k < width; k++){
                    if(grid[i + k][j] == 0 || grid[i][j + k] == 0)
                        break;
                    if(k + 1 < max)
                        continue;
                    if(dp[i + k][j][1] >= k + 1 && dp[i][j + k][0] >= k)
{
                        max = Math.max(k + 1, max);
                    }
                }
            }
        }
        return max * max;
    }
    
    public void collectY(int[][] grid, int[][][] dp, int y, int x){
        int height = grid.length, width = grid[0].length;
        int cnt = 0;
        for(int i = y; i < height; i++){
            if(grid[i][x] == 0)
                break;
            cnt ++;
        }
        for(int i = y; cnt > 0; cnt --, i ++){
            dp[i][x][0] = cnt;
        }
    }
    
    public void collectX(int[][] grid, int[][][] dp, int y, int x){
        int height = grid.length, width = grid[0].length;
        int cnt = 0;
        for(int i = x; i < width; i++){
            if(grid[y][i] == 0)
                break;
            cnt ++;
        }
        for(int i = x; cnt > 0; cnt --, i ++){
            dp[y][i][1] = cnt;
        }
    }
}
