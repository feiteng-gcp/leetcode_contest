class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int x = grid.length, y = grid[0].length;
        int[][] h = new int[x + 1][y + 1], v = new int[x + 1][y + 1];
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                h[i + 1][j + 1] = h[i + 1][j] + grid[i][j];
                v[i + 1][j + 1] = v[i][j + 1] + grid[i][j];
            
}
        }

        int max = 0;
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                if (grid[i][j] == 0) continue;
                for (int k = max + 1; i + k <= x && j + k <= y ; k++) {
                    if (h[i + 1][j + k] - h[i + 1][j] != k) break;
                    if (v[i + k][j + 1] - v[i][j + 1] != k) break;
                    if (h[i + k][j + k] - h[i + k][j] != k) continue;
                    if (v[i + k][j + k] - v[i][j + k] != k) continue;
                    max = k;
                
}
            }
        }
        return max * max;
    }
}
