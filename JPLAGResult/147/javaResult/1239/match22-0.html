class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] leftRight = new int[m][n];
        int[][] topDown = new int[m][n];
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    if (i > 0) topDown[i][j] = topDown[i - 1][j] + 1;
                    else topDown[i][j] = 1;
                    
                    if (j > 0) leftRight[i][j] = leftRight[i][j - 1] + 1;
                    else leftRight[i][j] = 1;
                }
            }
        }
        
        int max = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = j; k >= 0; k--) {
                    int len = j - k + 1;
                    if (grid[i][k] == 0 || i - len + 1 < 0 || grid[i - len + 1][j] != 1) break;
                    if (leftRight[i - len + 1][j] >= len && topDown[i][k] >= len) {
                        max = Math.max(max, len * len);
                    }
                }
            }
        }
        
        return max;
    }
}
