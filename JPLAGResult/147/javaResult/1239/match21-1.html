class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[][] dpr = new int[m][n];
        int[][] dpc = new int[m][n];
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(grid[i][j] == 0) continue;
                if(i == 0) {
                    dpc[i][j] = 1;
                } else {
                    dpc[i][j] = 1 + dpc[i - 1][j];
                }
                if(j == 0) {
                    dpr[i][j] = 1;
                } else {
                    dpr[i][j] = 1 + dpr[i][j - 1];
                
}
            }
        }
        int res = 0;
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(grid[i][j] == 0) continue;
                for(int k = Math.min(i, j) ; k >= 0; --k) {
                    if(dpc[i][j] > k && dpr[i][j] > k && dpc[i][j - k] > k && dpr[i - k][j] > k) {
                        int t = k + 1;
                        if(t * t > res) res = t * t;
                        break;
                    
} 
                }
            }
        }
        return res;
    }
}
