class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int m = grid.length, n = grid[0].length, res = 0;
        int[] colSum = new int[n];
        int[][] rowSum = new int[m][n];
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    colSum[j] = i > 0 ? colSum[j] + 1 : 1;
                    rowSum[i][j] = j > 0 ? rowSum[i][j - 1] + 1 : 1;
                    int maxLen = Math.min(rowSum[i][j], colSum[j]);
                    for (int k = maxLen; k >= 1; k--) {
                        if ((rowSum[i - k + 1][j] >= k) && (colSum[j - k + 1] >= k)) {
                            res = Math.max(res, k * k);
                            break;
                        }
                    }
                    
                } else {
                    colSum[j] = 0;
                }
            }
        }
        return res;
    }
}
