class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int n = grid.length, m = grid[0].length, max = 0;
        int[][] down = new int[n][m], right = new int[n][m];
        
        for (int i = n - 1; i >= 0; --i) {
            for (int j = m - 1; j >= 0; --j) {
                if (grid[i][j] == 1) {
                    down[i][j] = i < n - 1 ? down[i+1][j] + 1 : 1;
                    right[i][j] = j < m - 1 ? right[i][j+1] + 1 : 1;
                }
            }
        }
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 1) {
                    
int maxCand = Math.min(down[i][j], right[i][j]);
                    for (int cand = maxCand; cand >= 1; --cand) {
                        if (down[i][j + cand - 1] >= cand && right[i + cand - 1][j] >= cand) {
                            max = Math.max(max, cand * cand);
                            break;
                        }
                    }
                
}
            }
        }
        
        return max;
    }
}
