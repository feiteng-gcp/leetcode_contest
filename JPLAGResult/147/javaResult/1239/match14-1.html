class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int R = grid.length, C = grid[0].length;
        int[][] rn = new int[R][C];
        int[][] cn = new int[R][C];
        
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (j == 0) rn[i][j] = grid[i][j] == 1 ? 1 : 0;
                else rn[i][j] = grid[i][j] == 1 ? rn[i][j - 1] + 1 : 0;
            
}
        }
        
        for (int j = 0; j < C; j++) {
            for (int i = 0; i < R; i++) {
                if (i == 0) cn[i][j] = grid[i][j] == 1 ? 1 : 0;
                else cn[i][j] = grid[i][j] == 1 ? cn[i - 1][j] + 1 : 0;
            }
        }
        
        int ans = 0;
        
for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                for (int l = 1; i + l - 1 < R && j + l - 1 < C; l++) {
                    if (rn[i][j] > 0 && rn [i][j + l - 1] >= l && 
                       cn[i + l - 1][j] >= l && rn[i + l - 1][j + l - 1] >= l
                       && cn[i + l - 1][j + l - 1] >= l) {
                        ans = Math.max(ans, l * l);
                    
}
                }
            }
        }
        return ans;
    }
}
