class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int xlen = grid.length;
        int ylen = grid[0].length;
        int res = 0;
        for(int i=0 ;i<grid.length ; i++){
            
            for(int j=0 ; j<grid[0].length ; j++){
                
                if(grid[i][j]==0) continue;
                
                int len = Math.min(xlen-i,ylen-j);
                
                if(len*len<=res) continue;
                
                for(int k=0 ; k<len ; k++){
                    if(isSquare(grid,i,j,i+k,j+k)){
                        res = Math.max((k+1)*(k+1),res);
                    }
                }
            }
        }
        return res;
    }
    
    public boolean isSquare(int[][] grid,int startx,int starty,int endx,int endy){
        for(int i=startx; i<=endx;i++){
            if(grid[i][starty]!=1){
                return false;
            }
            if(grid[i][endy]!=1){
                return false;
            }
        }
        
        for(int i=starty; i<=endy;i++){
            if(grid[startx][i]!=1){
                return false;
            }
            if(grid[endx][i]!=1){
                return false;
            }
        }
        return true;
    }
}
