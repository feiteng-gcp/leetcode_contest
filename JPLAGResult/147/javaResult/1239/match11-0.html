class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int result = 0;
        int rows = grid.length;
        int cols = grid[0].length;
        for (int i = 0;i<rows;i++) {
            for (int j = 0;j<cols;j++) {
                if (grid[i][j] == 0) continue;
                for (int k = 0;i+k<rows && j+k<cols;k++) {
                    if (grid[i+k][j+k] == 0) continue;
                    if (check(i, j, k, grid)) {
                        result = Math.max(result, (k+1)*(k+1));
                    }
                }
            }
        }
        return result;
    }
    
    private boolean check(int a, int b, int k, int[][] grid) {
        for (int i = 0;i<=k;i++) {
            // check toprow
            if (grid[a+i][b] == 0) return false;
            if (grid[a][b+i] == 0) return false;
            if (grid[a+k][b+i] == 0) return false;
            if (grid[a+i][b+k] == 0) return false;
        }
        return true;
    }
}
