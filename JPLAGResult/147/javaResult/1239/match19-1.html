class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int len = Math.min(m, n);
        
for (int i = len; i >= 1; i--) {
            for (int j = 0; j + i - 1 < m; j++) {
                for (int k = 0; k + i - 1 < n; k++) {
                    if (check(i, j, k, grid)) return i * i
;
                }
            }
        }
        return 0;
    }
    
    public boolean check(int len, int r, int c, int[][] grid) {
        for (int i = 0; i < len; i++) {
            int row1 = r;
            int row2 = r + len - 1;
            int col = c + i;
            
            int row = r + i;
            int col1 = c;
            int col2 = c + len - 1;
            if (grid[row1][col] == 0) return false;
            if (grid[row2][col] == 0) return false;
            
if (grid[row][col1] == 0) return false;
            if (grid[row][col2] == 0) return false; 
        }
        return true;
    }
}
