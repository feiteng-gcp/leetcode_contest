class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        if(grid==null || grid.length==0 || grid[0].length==0) return 0;
        int n = grid.length;
        int m = grid[0].length;
        
        int[][] rowSum = new int[n][m];
        int[][] colSum = new int[n][m];
        int result = 0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(grid[i][j]==1){
                    if(i==0){
                        rowSum[i][j] = 1;
                    } else {
                        rowSum[i][j] = 1 + rowSum[i-1][j];
                    }
                    if(j==0){
                        colSum[i][j] = 1;
                    } else {
                        colSum[i][j] = 1 + colSum[i][j-1];
                    }
                }
            }
        }
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(grid[i][j]==1){
                    result = Math.max(result, 1);
                    int min = Math.min(i,j);
                    min = Math.min(min,rowSum[i][j]);
                    min = Math.min(min,colSum[i][j]);
                    int offSet = 0;
                    result = Math.max(result, offSet);
                    while(offSet<=min){
                        if(rowSum[i][j-offSet]>offSet && colSum[i-offSet][j]>offSet) result = Math.max(result, (int)Math.pow(1+offSet, 2));
                        offSet++;
                    }
                }
            }
        }
        return result;
    }
}
