class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int row = grid.length;
        int col = grid[0].length;
        int max = 0;
        for (int i = 0; i < row; i++) {
            for (int j = i + max; j < row; j++) {
                for (int m = 0; m < col; m++) {
                    int n = m + (j - i);
                    
if (n >= col) continue;
                    if (check(i, j, m, n, grid)) {
                        max = Math.max(max, j - i + 1);
                    }
                }
            }
        }
        
        return max * max;
    }
    
    boolean check(int i, int j, int m, int n, int[][] grid) {
        for (int col = m; col <= n; col++) {
            if (grid[i][col] != 1) return false;
        }
        for (int col = m; col <= n; col++) {
            if (grid[j][col] != 1) return false;
        }
        for (int row = i; row <= j; row++) {
            if (grid[row][m] != 1) return false;
        }
        for (int row = i; row <= j; row++) {
            
if (grid[row][n] != 1) return false;
        }
        return true;
    }
}
