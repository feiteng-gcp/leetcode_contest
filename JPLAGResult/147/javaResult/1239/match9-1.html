class Solution {
		int [][] grid ;
		int r,c;
		boolean check(int i,int j,in
t fa){
			for(int ii=i;ii<i+fa;ii++){ // d
				if(grid[ii][j]!=1) return false;
			}
			for(int jj=j;jj<j+fa;jj++){ //r
				if(grid[i][jj]!=1) return false;
			}
			
			for(int jj=j;jj<j+fa;jj++){ //r
				if(grid[i+fa-1][jj]!=1) return false;
			}
			
			for(int ii=i;ii<i+fa;ii++){ // d
				if(grid[ii][j+fa-1]!=1) return false;
			}
			return true;
		}
	    public int largest1BorderedSquare(int[][] grid) {
	    	this.grid = grid;
	         r = grid.length;
	         c = grid[0].length;
	        int fa = r<c?r:c;
	        for(int
 i=fa;i>=1;i--){
	        	for(int j=0;j<=r-i;j++){
	        		for(int k=0;k<=c-i;k++){
	        			if(grid[j][k]==1){
	        				boolean rs = check(j,k,i);
	        				if(rs) {
	        					return i*i;
	        				}
	        			}

	        		}
	        	}
	        }
	        return 0;
	    }
	}
