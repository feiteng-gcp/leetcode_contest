class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        int length = 0;
        int[][][] boader = new int[grid.length][grid[0].length][2];
        
for (int rowIdx = grid.length - 1; rowIdx >= 0; rowIdx--) {
            for (int colIdx = grid[0].length - 1; colIdx >= 0; colIdx--) {
                if (grid[rowIdx][colIdx] == 1) {
                    if (rowIdx == grid.length - 1) {
                        boader[rowIdx][colIdx][0] = 1;
                    } else {
                        boader[rowIdx][colIdx][0] = boader[rowIdx + 1][colIdx][0] + 1;
                    }
                    if (colIdx == grid[0].length - 1) {
                        boader[rowIdx][colIdx][1] = 1;
                    } else {
                        boader[rowIdx][colIdx][1] = boader[rowIdx][colIdx + 1][1] + 1;
                    }
                }
            }
        }

        for (int rowIdx = 0; rowIdx + length < grid.length; rowIdx++) {
            for (int colIdx = 0; colIdx + length < grid[0].length; colIdx++) {
                if (grid[rowIdx][colIdx] == 1) {
                    int testL = boader[rowIdx][colIdx][0] > boader[rowIdx][colIdx][1] ? boader[rowIdx][colIdx][1] : boader[rowIdx][colIdx][0];
                    while (testL > length) {
                        if (boader[rowIdx][colIdx + testL - 1][0] >= testL && boader[rowIdx + testL - 1][colIdx][1] >= testL) {
                            length = testL;
                            break;
                        
}
                        testL--;
                    }
                }
            }
        }
        return length * length;
    }
}
