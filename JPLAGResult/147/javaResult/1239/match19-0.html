    int[][] grid;
    public int largest1BorderedSquare(int[][] grid) {
        this.grid=grid;
        int maxX=grid.length;
        int maxY=grid[0].length;
        int maxSize=0;
        for(int i=0;i<maxX;i++){
            for(int j=0;j<maxY;j++){
                for(int step=0; i+step<maxX && j+step<maxY; step++){
                        if(validate(i,j,i+step, j+step))
                            maxSize=Math.max(maxSize, (step+1)*(step+1));
                }
            }
        }
        return maxSize;
    }

    boolean validate(int i, int j, int k, int l){
        for(int x=i;x<=k;x++){
            if(grid[x][j]!=1) return false;
            if(grid[x][l]!=1) return false;
        }
        for(int y=j;y<=l;y++){
            if(grid[i][y]!=1) return false;
            if(grid[k][y]!=1) return false;
        }
        return true;
    }
}
