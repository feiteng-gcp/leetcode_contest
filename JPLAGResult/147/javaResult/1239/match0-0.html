class Solution {
    public int largest1BorderedSquare(int[][] g) {
        
        int l = g.length;
        int ll = g[0].length;
        int res = 0;
        
        for(int i=0;i<l;++i){
            if(i+1<=res){
                continue;
            }
            for(int j=0;j<ll;++j){
                if(j+1<=res){
                    continue;
                }
                int cur = g[i][j];
                if(cur==0) continue;
                
                for(int len=res;i-len>=0&&j-len>=0;len++){
                    boolean f = true;
                    for(int x=1;x<=len;++x){
                        f &= (cur==g[i-x][j]);
                    }
                    for(int x=1;x<=len;++x){
                        f &= (cur==g[i][j-x]);
                    }
                    for(int x=1;x<=len;++x){
                        f &= (cur==g[i-len][j-x]);
                    }
                    for(int x=1;x<=len;++x){
                        f &= (cur==g[i-x][j-len]);
                    }
                    if(f){
                        
                        res = Math.max(res,len+1);
                    }
                    
                }
                
                
                
                
            }
            
        }
        return res*res;
        
        
    }
}
