class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        
int row = grid.length, col = grid[0].length;
        int[][] rowdp = new int[row][col];
        int[][] coldp = new int[row][col];
        
        boolean returnZero = true;
        for (int i = 0; i < row; ++i) {
            for (int j = 0; j < col; ++j) {
                if (i == 0 && j == 0) {
                    rowdp[i][j] = grid[0][0];
                    coldp[i][j] = grid[0][0];
                } else if (i == 0) {
                    rowdp[i][j] = rowdp[i][j - 1] + grid[i][j];
                    coldp[i][j] = grid[i][j];
                } else if (j == 0) {
                    rowdp[i][j] = grid[i][j];
                    coldp[i][j] = coldp[i - 1][j] + grid[i][j];
                } else {
                    rowdp[i][j] = rowdp[i][j - 1] + grid[i][j];
                    coldp[i][j] = coldp[i - 1][j] + grid[i][j];
                }
                if (grid[i][j] != 0) {
                    returnZero = false;
                }
            }
        }
        if (returnZero) {
            return 0;
        }
        
        int res = 1;
        
for (int k = 2; k <= Math.max(row, col); ++k) {
            for (int i = 0; i < row - k + 1; ++i) {
                for (int j = 0; j < col - k + 1; ++j) {
                    if (rowdp[i][j + k - 1] - rowdp[i][j] + grid[i][j] == k &&
                        rowdp[i + k - 1][j + k - 1] - rowdp[i + k - 1][j] + grid[i + k - 1][j] == k &&
                        coldp[i + k - 1][j] - coldp[i][j] + grid[i][j] == k &&
                        coldp[i + k - 1][j + k - 1] - coldp[i][j + k - 1] + grid[i][j + k - 1] == k) {
                        
                        res = k;
                        break;
                    }
                }
                if (res == k) {
                    break;
                
}
            }
        }
        return res * res;
    }
}
