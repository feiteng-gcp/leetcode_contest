class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        if (null == grid || 0 == grid.length || 0 == grid[0].length) return 0;
        int m = grid.length, n = grid[0].length;
        int[][] row = new int[m][n + 1], col = new int[m + 1][n];
        int max = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                row[i][j + 1] = row[i][j] + grid[i][j];
                col[i + 1][j] = col[i][j] + grid[i][j];
                if (0 == max && 1 == grid[i][j]) max = 1;
            }
        }
        for (int k = 2; k <= Math.min(m, n); ++k) {
            
for (int i = 0; i + k <= m; ++i) {                
                for (int j = 0; j + k <= n; ++j) {                    
                    if (k == row[i][j + k] - row[i][j] && k == col[i + k][j] - col[i][j] &&
                        k == row[i + k - 1][j + k] - row[i + k - 1][j] && k == col[i + k][j + k - 1] - col[i][j + k - 1]) {
                        max = k;
                        break;
                    }
                }
                if (max == k) break
;
            }
        }
        return max * max;
    }
}
