class Solution {
    public int largest1BorderedSquare(int[][] grid) {
        if (grid.length == 0 || grid[0].length == 0) return 0;
        
        
int N = grid.length;
        int M = grid[0].length;
        
        // h
        int[][] cnts = new int[N][M];
        // v
        int[][] cnts2 = new int[N][M];
        
        for (int i = 0; i < N; i++) {
            cnts[i][M - 1] = grid[i][M - 1] == 1 ? 1 : 0;
            for (int j = M - 2; j >= 0; j--) {
                cnts[i][j] = grid[i][j] == 1 ? 1 + cnts[i][j + 1] : 0;
            }
        }
        
        for (int j = 0; j < M; j++) {
            cnts2[N - 1][j] = grid[N - 1][j] == 1 ? 1 : 0;
            for (int i = N - 2; i >= 0; i--) {
                cnts2[i][j] = grid[i][j] == 1 ? 1 + cnts2[i + 1][j] : 0;
            
}
        }
        
        int max = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                int maxLen = Math.min(N - i, M - j);
                
for (int k = 1; k <= maxLen; k++) {
                    if (cnts[i][j] >= k 
                        && cnts[i + k - 1][j] >= k 
                        && cnts2[i][j] >= k 
                        && cnts2[i][j + k - 1] >= k) {
                        
                        max = Math.max(max, k * k);
                    }
                    
                    
                    
                }
            }
        }
        
        return max;
    }
}
