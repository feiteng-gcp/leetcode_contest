class Solution {
    public int largest1BorderedSquare(int[][] a) {
        int n = a.length;
        int m = a[0].length;
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int e = Math.min(n-i, m-j) - 1;
                if ((e + 1) * (e+1) < res) continue;
                
                for (int k = e; k >= 0; k--) {
                    if (check(a, i, j, i+k, j+k)) {
                        res = Math.max(res, (k+1) *(k+1));
                        break;
                    
}
                }
            }
        }
        return res;
    }
    
    private boolean check(int[][] a, int i1, int j1, int i2, int j2) {
        for (int k = j1; k <= j2; k++) {
            if (a[i1][k] != 1 || a[i2][k] != 1) return false;
        }
        for (int k = i1; k <= i2; k++) {
            if (a[k][j1] != 1 || a[k][j2] != 1) return false;
        }
        return true;
    }
}
