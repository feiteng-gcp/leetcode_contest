public class Solution {
    public static void main(String[] args) {
        Solution solution=new Solution();
        System.out.println(solution.stoneGameII(new int[]{2,7,9,4,4}));
    }


    int[][] cache;
    int max;
    int[] piles;
    int[] remaining;
    public int stoneGameII(int[] piles) {
        int n=piles.length;
        this.piles=piles;
        cache=new int[1001][1001];
        max=0;

        remaining=new int[n];
        remaining[n-1]=0;
        for(int i=n-2;i>=0;i--){
            remaining[i]=remaining[i+1]+piles[i+1];
        }

//        helper(0,1);

        int res= helper(0,1);
//        Matrix.print(cache);
        return res;
    }

    int helper(int i, int m){
        if(cache[i][m]!=0)
            return cache[i][m];

        int res=0;

        int thisTime=0;
        for(int x=1;x<=2*m;x++){
            if(i+x-1>=piles.length) break;
            thisTime+=piles[i+x-1];
            res=Math.max(res, thisTime+remaining[i+x-1]-helper(i+x, Math.max(x, m)));
        }

        cache[i][m]=res;
        return res;
    }
}
