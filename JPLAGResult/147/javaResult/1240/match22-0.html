class Solution {
    public int stoneGameII(int[] piles) {
        int i;
        int[] a=new int[piles.length];
        a[piles.length-1]=piles[piles.length-1];
        for(i=piles.length-2;i>=0;i--){
            a[i]=a[i+1]+piles[i];
        }
        int[][] b=new int[piles.length][piles.length];
        return helper(piles,0,1,a,b);
    }
    
    public int helper(int[] piles, int now, int M, int[] a, int[][] b){
        int i,r=0,t=0,te;
        if(piles.length-now<=2*M){
            return a[now];
        }
        if(b[now][M]>0) return b[now][M];
        for(i=0;i<2*M;i++){
            t+=piles[now+i];
            te=t+a[now+i+1]-helper(piles,now+i+1,Math.max(M,i+1),a,b);
            if(te>r) r=te;
        }
        b[now][M]=r;
        return r;
    }
}
