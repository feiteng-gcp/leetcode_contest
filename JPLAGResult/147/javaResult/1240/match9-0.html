class Solution {
    public int stoneGameII(int[] piles) {
        int total = 0;
        for (int p : piles) {
            total += p;
        }
        return minimax(piles, 0, 1, total);
    }
    Map<Integer, Integer> cache = new HashMap();
    private int minimax(int[] piles, int index, int M, int remain) {
        int key = index * 1000 + M;
        if (cache.containsKey(key)) {
            return cache.get(key);
        }
        if (index >= piles.length) {
            return 0;
        }
        if (piles.length - index + 1 <= 2 * M) {
            return remain;
        }
        int localMax = 0;
        int take = 0;
        for (int i = 0; i < 2 * M && index + i < piles.length; i++) {
            take += piles[index + i];
            int tmpM = Math.max(M, i + 1);
            int oppo = minimax(piles, index + i + 1, tmpM, remain - take);
            localMax = Math.max(localMax, remain - oppo);
        }
        cache.put(key, localMax);
        return localMax;
    }
}
