class Solution {
    
int[] piles;
    int[][] memo;
    int n;
    public int stoneGameII(int[] piles) {
        n = piles.length;
        memo = new int[n][n];//start i, m=j
        int sum = 0;
        for (int i = 0; i < n; i++){
            Arrays.fill(memo[i], Integer.MIN_VALUE);
            sum += piles[i];
        }
        this.piles = piles;
        int x = backtrack(0, 1);
        //for (int[] m1 : memo)
        //    System.out.println(Arrays.toString(m1));
        
        int ans = (sum-x)/2;
        //System.out.println(sum + " " + ans + " " + x);
        return ans+x;
    
}
    
    private int backtrack(int i, int m){
        //System.out.println(i + " " + m);
        if (i == n)
            return 0;
        if (memo[i][m] != Integer.MIN_VALUE)
            return memo[i][m];
        int ans = Integer.MIN_VALUE;
        int sofar = 0;
        //idx start from i(x =1) to i+2m-1
        for (int idx = i; idx < n && idx <= i+2*m-1; idx++){
            sofar += piles[idx];
            ans = Math.max(ans, sofar - backtrack(idx+1, Math.max(m, idx-i+1)));
        }
        memo[i][m] = ans;
        return ans;
    }
}
