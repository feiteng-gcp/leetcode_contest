class Solution {
    public int stoneGameII(int[] piles) {
        int n = piles.length;
        int[][] dp = new int[n][n];
        for (int i = 1; i < n; i++) piles[i] += piles[i-1];
        
return minMax(piles, n, 1, 0, dp);
    }
    
    public int minMax(int[] piles, int n, int m, int st, int[][] dp) {
        int pre = st == 0 ? 0 : piles[st-1];
        
int sum = piles[n-1] - pre;
        if (n - st <= m * 2) {
            return sum;
        }
        if (dp[m][st] != 0) return dp[m][st];
        int sol = 0;
        for (int i = st; i < Math.min(m * 2 + st, n); i++) {
            sol = Math.max(sol, sum - minMax(piles, n, Math.max(m, i - st + 1), i +  1, dp));
        }
        dp[m][st] = sol;
        return sol;
    }
}
