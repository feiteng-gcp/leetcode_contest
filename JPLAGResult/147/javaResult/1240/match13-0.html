class Solution {
    int[][] dp;
    int[] sum;
    public int stoneGameII(int[] piles) {
        int n = piles.length;
        dp = new int[n + 1][n * 2 + 1];
        sum = new int[n + 1];

        for (int i = n - 1; i >= 0; i--) {
            sum[i] = sum[i + 1] + piles[i];
        }

        for (int i = 0; i < n + 1; i++) {
            for (int j = 0; j < n * 2 + 1; j++) {
                dp[i][j] = -1;
            }
        }
        
        return dfs(0, 1, piles);

    }

    public int dfs(int current, int M, int[] piles) {
        if (current >= piles.length) return 0;
        if (dp[current][M] >= 0) return dp[current][M];

        int sm = 0;
        int ans = -1;
        for (int x = 1; x <= M * 2 && current + x - 1 < piles.length; x++) {
            sm += piles[current + x - 1];
            int t = sm + (sum[current + x] - dfs(current + x, Math.max(M, x), piles));
            ans = Math.max(ans, t);
        }
        
        dp[current][M] = ans;
        return ans;

    }
}
