class Solution {
    private int[][] stoneCache;

    public int stoneGameII(int[] piles) {
        int len = piles.length;
        stoneCache = new int[len][len];
        return stoneGameII(piles, 0, 1, 
Arrays.stream(piles).sum());
    }

    private int stoneGameII(int[] piles, int i, int M, int sum) {
        if (i >= piles.length) return 0;
        if (stoneCache[i][M] == 0) {
            int take = 0, max = 0;
            
for (int j = 1; j <= M * 2 && i + j <= piles.length; j++) {
                take += piles[i + j - 1];
                max = Math.max(max, sum - stoneGameII(piles, i + j, Math.max(j, M), sum - take));
            }
            stoneCache[i][M] = max;
        }
        return stoneCache[i][M];
    }
}
