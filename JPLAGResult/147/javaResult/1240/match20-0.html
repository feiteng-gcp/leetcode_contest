class Solution {
    int[] presum;
    int n;
    int[][] dp;
    int[] piles;
    public int stoneGameII(int[] piles) {
        n = piles.length;
        presum = new int[n+1];
        presum[0] = 0;
        for (int i = 0; i < n; i++) presum[i+1] = presum[i] + piles[i];
        
        dp = new int[n+1][n];
        for (int i = 0; i <= n; i++)
            Arrays.fill(dp[i], -1);
        this.piles = piles;
        return solve(1, 0);
    }
    
    int solve(int M, int idx) {
        if (idx >= n) return 0;
        if (dp[M][idx] != -1) return dp[M][idx];
        
        int res = 0;
        int get = 0;
        for (int x = 1; x <= 2 * M; x++) {
            if (x + idx - 1 >= n) break;
            get += piles[x+idx-1];
            
            int tmp = solve(Math.min(n, Math.max(M, x)), idx+x);
            res = Math.max(res, presum[n] - presum[idx+x] - tmp + get);
        }
        return dp[M][idx] = res;
    }
}
