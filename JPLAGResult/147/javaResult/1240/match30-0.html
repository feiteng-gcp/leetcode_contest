class Solution {
    int[] sum;
	Integer[][] f;
	public int stoneGameII(int[] piles) {

		sum = new int[piles.length];

		for (int i = piles.length - 1; i >= 0; i--) {
			if (i == piles.length - 1) {
				sum[i] = piles[i];
			} else {
				sum[i] = sum[i + 1] + piles[i];
			}
		}

		f = new Integer[piles.length][piles.length + 1];
		return dp(0, 1, piles, sum);

	}

	private int dp(int x, int m, int[] p, int[] sum) {
		if (x == p.length) {
			return 0;
		}
		if (m >= p.length - x) {
			return sum[x];
		}
		if (f[x][m] == null) {

			int n = p.length - x;
			int res = Integer.MIN_VALUE;
			int acc = 0;
			for (int j = 1; j <= 2 * m; j++) {
				if (j > n) {
					break;
				}
				acc += p[x + j - 1];
				int all = x + j == p.length ? 0 : sum[x + j];
				int total = acc + all - dp(x + j, Math.max(j, m), p, sum);
				res = Math.max(res, total);
			}

			f[x][m] = res;

		}
		return f[x][m];
	}
}
