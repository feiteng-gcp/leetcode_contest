class Solution {
        int r =0;
      
    
    int dp[][][] = new int[2][101][202];
    
    public int stoneGameII(int[] ps) {
        
        
        return g(0,ps,0,1);
        
    }
    
    public int g(int a, int[] ps, int p, int m){
        
        if(dp[a][p][m]>0){
            return dp[a][p][m];
        }
        
        int l = ps.length;
        if(p>=l){
            return 0;
        }
        
        int tot = 0;
        
        int tj = Integer.MAX_VALUE;
        
        int sum = 0;
        for(int i=p;i<l;++i){
            sum += ps[i];
        }

        
        for(int i=p;i<l&&i<p+2*m;++i){
            int mm = Math.max(i-p+1,m);
            tot+= ps[i];
            if(a==0){
            int vl = g(a^1,ps,i+1,mm);
            if(vl<tj){
                tj = vl;
            }
            }else{
                int vl = g(a^1,ps,i+1,mm);
            if(vl<tj){
                tj = vl;
            }
                
                
            }
            
        }
        return dp[a][p][m] = sum-tj;
        
        
    }
    
}
