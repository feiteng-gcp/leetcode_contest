class Solution {
    int[][] lut;
    public int stoneGameII(int[] piles) {
        int sum = 0;
        lut = new int[piles.length][piles.length];
        for (int i = 0; i < piles.length; i++)
            for (int j = 0; j < piles.length; j++)
                lut[i][j] = -1;
        for (Integer i : piles)
            sum+=i;
        return getMax(piles, 0, 1, sum);
    }
    public int getMax(int[] piles, int st, int M, int remainSum) {
        if (piles.length-st <= 2*M) return remainSum;
        else if (lut[st][M] != -1) return lut[st][M];
        else {
            int max_v = 0;
            int acc = 0;
            for (int i = 1; i <= 2*M; i++) {
                acc+=piles[st+i-1];
                max_v = Math.max(max_v, remainSum-getMax(piles, st+i, Math.max(M, i), remainSum-acc));
            }
            lut[st][M] = max_v;
            return max_v;
        }
    }
}
