class Solution {
    public int stoneGameII(int[] piles) {
        int n = piles.length;
        int[][] dp = new int[n+1][n+1];
        int[] sums = new int[n+1];
        for(int i=0; i<n; i++) {
            sums[i+1] = piles[i] + sums[i];
        }
        
        dfs(sums, n, 1, 1, dp);
        return dp[1][1];
    
}
    
    private int dfs(int[] sums, int n, int start, int M, int[][] memo) {
        if(start >= n+1) return 0;
        if(memo[start][M] != 0) return memo[start][M];
        int remain = sums[n] - sums[start-1];
        int cur=0;
        for(int i=1; i<= Math.min(n-start+1,2*M); i++) {
            int l = 
dfs(sums, n, start+i, Math.max(M, i), memo);
            if(remain-l > cur) cur = remain-l;
        }
        memo[start][M] = cur;
        return cur;
    }
    
}
