class Solution {
  public int stoneGameII(int[] piles) {
    int n = piles.length;
    int[] sm = new int[n+1];
    sm[0] = 0;
    for(int i=0; i<n; i++) {
      sm[i+1] = sm[i] + piles[i];
    }
    int[][] mem = new int[n][n];
    for(int i=n-1; i>=0; i--) {
      for(int j=0; j<n; j++) {
        if(i == n-1) mem[i][j] = sm[i+1] - sm[i];
        mem[i][j] = Integer.MIN_VALUE;
        for(int k=1; k<=Math.min(n, 2*(j+1)); k++) {
          if(i+k > n) continue;
          else if(i+k == n) mem[i][j] = Math.max(mem[i][j], sm[n] - sm[i]);
          else {
            mem[i][j] = Math.max(mem[i][j] , sm[i+k] - sm[i] - mem[i+k][Math.max(j, k-1)]);
          }
        }
      }
    }
      // System.out.println(Arrays.toString(sm));
      // for(int i=0; i<n; i++) {
      //     System.out.println(Arrays.toString(mem[i]));
      // }
    return (sm[n] + mem[0][0])/2;
  }
}
