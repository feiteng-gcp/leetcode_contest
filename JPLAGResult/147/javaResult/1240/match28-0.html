class Solution {
    public int stoneGameII(int[] piles) {
        int[][][] maxStones = new int[piles.length][piles.length + 1][piles.length + 1];
        int[] sum = new int[piles.length + 1];
        for (int idx = 0; idx < piles.length; idx++) {
            sum[idx + 1] = sum[idx] + piles[idx];
        }
        subStone(piles, sum, maxStones, 0, 1);
        return maxStones[0][1][1] > maxStones[0][1][2] ? maxStones[0][1][1] : maxStones[0][1][2];
    }

    private int subStone(int[] piles, int[] sum, int[][][] maxStones, int idx, int m) {
        if (idx >= piles.length) {
            return 0;
        }

        int max = 0;
        for (int x = 1; x <= (m << 1) && idx + x <= piles.length; x++) {
            if (maxStones[idx][m][x] == 0) {
                int otherBest = subStone(piles, sum, maxStones, idx + x, Math.min(piles.length - idx, Math.max(m, x)));
                maxStones[idx][m][x] = (sum[piles.length] - sum[idx]) - otherBest;
            }
            if (maxStones[idx][m][x] > max) {
                max = maxStones[idx][m][x];
            }
        }
        return max;
    }
}
