class Solution {
    public int stoneGameII(int[] piles) {
        int[] sums = new int[piles.length];
        int curSum = 0;
        for (int i = 0; i < piles.length; i++) {
            curSum += piles[i];
            sums[i] = curSum;
        }
        return max(piles, new Integer[piles.length][piles.length], 0, 1, sums);
    
}
    
    private int max(int[] piles, Integer[][] memo, int idx, int m, int[] sums) {
        if (idx == piles.length) {
            return 0;
        
}
        
        if (memo[idx][m] != null) return memo[idx][m];
        
        int range = 2 * m;
        int max = Integer.MIN_VALUE;
        for (int i = idx; i < Math.min(sums.length, idx + range); i++) {
            int curSum = sums[i] - sums[idx] + piles[idx];
            int oppoMax = max(piles, memo, i + 1, Math.max(m, i - idx + 1), sums);
            int remaining = sums[sums.length - 1] - sums[i] - oppoMax;
            max = Math.max(max, curSum + remaining);
        }
        
        memo[idx][m] = max;
        return memo[idx][m];
    }
}
