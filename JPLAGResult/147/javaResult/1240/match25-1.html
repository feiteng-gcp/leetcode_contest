class Solution {
    public int stoneGameII(int[] piles) {
        int n = piles.length;
        int[] remain = new int[n+1];
        remain[n] = 0;
        for (int i = n - 1; i >= 0; --i) {
            remain[i] = remain[i+1] + piles[i];
        }
        int dp[][] = new int[n][n];
        for (int start = n - 1; start >= 0; start -= 1) {
            for (int m = 1; m < n; ++m) {
                if (2 * m >= n - start) {
                    dp[start][m] = remain[start];
                } else {
                    int max = 0;
                    for (int x = 1; x <= 2 * m; ++x) {
                        int nextStart = start + x;
                        max = Math.max(max, remain[start] - dp[nextStart][Math.max(m, x)]);
                    }
                    dp[start][m] = max;
                }
            }
        }
        return dp[0][1];
    }
}
