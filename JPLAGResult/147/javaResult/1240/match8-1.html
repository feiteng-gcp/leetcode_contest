class Solution {
    public int stoneGameII(int[] piles) {
		int N = piles.length;
		
		//pop totals
        int[] totals = new int[N];
        totals[N-1] = piles[N-1];
        for (int i=N-2; i>=0; i--) {
        	totals[i] = piles[i] + totals[i+1];
        }
        
        //dp [startInd][M]
        int[][] memo = new int[N+1][N+1];
        return sg2Helper(memo, totals, piles, 0, 1);
    }
