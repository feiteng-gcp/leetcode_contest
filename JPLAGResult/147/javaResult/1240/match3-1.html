class Solution {
    public int stoneGameII(int[] piles) {
        int len = piles.length;
        int[] sum = new int[len + 1];
        for(int i = len - 1; i >= 0; i--)
            sum[i] = sum[i + 1] + piles[i];
        // System.out.println(Arrays.toString(sum));
        int[][] dp = new int[len][len * 2 + 1];
        
return recursion(dp, sum, 0, 1);
    }
    
    private int recursion(int[][] dp, int[] sum, int index, int x){
        if(dp[index][x] != 0)
            return dp[index][x];
        if(index + 2 * x >= dp.length)
            return sum[index];
        
for(int i = 1; i <= 2 * x; i++){
            dp[index][x] = Math.max(dp[index][x], sum[index] - recursion(dp, sum, index + i, Math.max(i, x)));
        }
        // System.out.println("i:"+ index + " x:" + x + " val:" + dp[index][x]);
        return dp[index][x];
    }
    
}
