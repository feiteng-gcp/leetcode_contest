class Solution {
    public int stoneGameII(int[] piles) {
        int sum = 0;
        for (int p : piles) {
            sum += p;
        }
        return optimal(piles, sum, 0, 1, new HashMap<>());
    }
    private int optimal(int[] piles, int sum, int start, int m, Map<String, Integer> cache) {
        int limit = 2 * m;
        if (piles.length - start + 1 <= limit) {
            return sum;
        }
        String key = start + "," + m;
        if (cache.containsKey(key)) {
            return cache.get(key);
        
}
        int max = 0, cur = 0;
        for (int i = 1; i <= limit; i++) {
            cur += piles[start + i - 1];
            int tmp = optimal(piles, sum - cur, start + i, 
Math.max(m, i), cache);
            max = Math.max(max, sum - tmp);
        }
        cache.put(key, max);
        return max;
    }
}
