class Solution {
    int[][] dp;
    int[] record;
    public int stoneGameII(int[] piles) {
        
        dp = new int[piles.length][piles.length];
        for(int[] line: dp) Arrays.fill(line, -1);
        record = new int[piles.length];
        record[piles.length-1] = piles[piles.length-1];
        for(int i=piles.length-2; i>=0; i--)
            record[i] = record[i+1] + piles[i]
;
        return dfs(piles, 0, 1);
    }
    
    private int dfs(int[] piles, int idx, int M) {
        if(dp[idx][M] != -1) return dp[idx][M];
        if(piles.length-idx <= 2*M) {
            dp[idx][M] = record[idx];
            return record[idx];
        }
        
int max = 0;
        for(int i=0; i<2*M; i++) {
            int rivals = dfs(piles, idx+i+1, Math.max(M, i+1));
            max = Math.max(max, record[idx]-rivals);
        }
        dp[idx][M] = max;
        return max;
    }
}
