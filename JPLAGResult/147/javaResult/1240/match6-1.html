class Solution {
    
    Integer[][] table;
    
    public int stoneGameII(int[] piles) {
        int N = piles.length;
        table = new Integer[N][2 * N + 1];
        
return dp(piles, 0, 1);
    }
    
    public int dp(int[] piles, int offset, int M) {
        if (offset >= piles.length) return 0;
        
        if (table[offset][M] != null) return table[offset][M];
        
        int stones = 0;
        int max = 0;
        int sum = 0;
        
        for (int i = offset; i < piles.length; i++) {
            sum += piles[i];
        }
        
        
for (int i = offset; i < offset + 2 * M && i < piles.length; i++) {
            stones += piles[i];
            int gain = sum - dp(piles, i + 1, Math.max(M, i - offset + 1));
            max = Math.max(max, gain);
        }
        
        table[offset][M] = max;
        return max;
    }
}
