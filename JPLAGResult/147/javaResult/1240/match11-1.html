class Solution {
    // dp[m][i] := when m, and stones from index i are left, the max value the first picker can get
    // dp[m][i] = max(dp[new m][i - (1 <= x <= 2m)]
    //On each player's turn, that player can take all the stones in the first X remaining piles, 
    //where 1 <= X <= 2M.  Then, we set M = max(M, X).
    public int stoneGameII(int[] piles) {
        int len = piles.length;
        int[] ps = new int[piles.length];
        ps[0] = piles[0];
        for (int i = 1; i < piles.length; i++) {
            ps[i] = ps[i - 1] + piles[i];
        }
        int[][] dp = new int[len / 2 + 2][len];
        for (int i = len - 1; i >= 0; i--) {
            int remain = ps[len - 1] - (i == 0 ? 0 : ps[i - 1]);
            for (int m = 1; m <= Math.min(len / 2 + 1, Math.pow(2, i)); m++) 
{
                if (m * 2 >= len - i) {
                    dp[m][i] = remain;
                    continue;
                }
                for (int take = 1; take <= 2 * m; take++) {
                    int newM = Math.max(m, take);
                    newM = Math.min(newM, len / 2 + 1);
                    dp[m][i] = 
Math.max(dp[m][i], remain - dp[newM][i + take]);
                }
            }
        }
        return dp[1][0];
    }
}
