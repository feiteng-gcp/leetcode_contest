class Solution {
		int [] piles;
		int allPiles;
		int max(int a,int b){
			return a>b?a:b;
		}
		int dp[][];
		int dfs(int now,int m)
		{
			if(dp[now][m]!=-1) return dp[now][m];
			if(now==allPiles-1) return piles[allPiles-1];
			else if(now==allPiles) return 0;
			int ma = 0;
			int tmp = 0;
			for(int i=1;i<=2*m && now+i-1<allPiles;i++){
				tmp += piles[now+i-1];
				ma = max(ma,tmp+sum[allPiles]-sum[now+i]-dfs(now+i,max(i,m)));
			}
			dp[now][m] = ma;
			return ma;
		}
		int sum[] ;
	    public int stoneGameII(int[] piles) {
	    	this.piles = piles;
	    	int n = piles.length;
	    	
	    	this.sum = new int[n+1];
	    	this.dp = new int[101][200];
	    	for(int i=0;i<=100;i++){
	    		for(int j=0;j<200;j++)
	    		{
	    			dp[i][j] = -1;
	    		}
	    	}
	    	for(int i=1;i<=n;i++)
	    	{
	    		sum[i] = sum[i-1] + piles[i-1];
	    	}
	    	allPiles = n;
	        return dfs(0,1);
	    }
	}
