class Solution {
    Map<String, Integer> map = new HashMap<>();
    public int stoneGameII(int[] p) {
        int n = p.length;
        int[] sum = 
new int[n];
        sum[n - 1] = p[n - 1];
        for (int i = n - 2; i >= 0; --i) {
            sum[i] = sum[i + 1] + p[i];
        }
        
        
        return dfs(0, p, sum, 1);
    }
    
    private int dfs(int index, int[] p, int[] sum, int M) {
        String s = index + "," + M;
        if (map.containsKey(s)) return map.get(s);
        if (index >= p.length) return 0;
        
        int res = 0;
        for (int i = 1; i <= 2 * M && index + i <= p.length; ++i) {
            int cur = sum[index];
            if (index + i < p.length) {
                cur -= sum[index + i];
            }
            if (index + i < p.length) {
                cur += sum[index + i] - dfs(index + i, p, sum, Math.max(M, i));
            }
            
res = Math.max(res, cur);
        }
        
        map.put(s, res);
        return res;
    }
}
