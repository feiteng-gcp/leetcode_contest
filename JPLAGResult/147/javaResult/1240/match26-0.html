class Solution {
    public int stoneGameII(int[] piles) {
        int n = piles.length;
        int[][] dp = new int[n+1][n*2+1];
        int[][] sums = new int[n][n];
        for (int i = 0; i < n; i++) {
            sums[i][i] = piles[i];
            for (int j = i+1; j < n; j++) {
                sums[i][j] = sums[i][j-1]+piles[j];
            }
        }
        for (int i = n-1; i >= 0; --i) {
            int rem = n-i;
            for (int j = 1; j <= 2*n; ++j) {
                if (2*j>=rem) {
                    dp[i][j] = sums[i][n-1];
                    continue;
                }
                for (int k = 1; k <= 2*j; ++k) {
                    dp[i][j] = Math.max(dp[i][j], sums[i][n-1]-dp[i+k][Math.max(j, k)]);
                }
            }
        }
        return dp[0][1];
    }
}
