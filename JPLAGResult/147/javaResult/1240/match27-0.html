class Solution {
    
    HashMap<String, Integer> map;
    int N;
    int[] sum;
    
    public int stoneGameII(int[] piles) {
        
        map = new HashMap<>();
        N = piles.length;
        sum = new int[N];
        sum[N - 1] = piles[N - 1];
        for (int i = N - 2; i >= 0; i--) {
            sum[i] = piles[i] + sum[i + 1];
        }
        
        return dfs(piles, 0, 1);
        
    }
    
    private int dfs(int[] piles, int start, int M) {
        if (map.containsKey(start + "," + M)) {
            return map.get(start + "," + M);
        }
        
        int res = 0;
        if (N - start <= 2 * M) {
            res = sum[start];
            map.put(start + "," + M, res);
            return res;
        }
        int iGet = 0;
        for (int X = 1; X <= 2 * M; X++) {
            int end = start + X - 1;
            iGet = sum[start] - sum[end + 1];
            int theOtherGet = dfs(piles, end + 1, Math.max(X, M));
            iGet += sum[end + 1] - theOtherGet;
            res = Math.max(iGet, res);
        }
        map.put(start + "," + M, res);
        return res;
    }
}
