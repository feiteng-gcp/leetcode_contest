class Solution {
    public int tribonacci(int n) {
        int[] fib= new int[Math.max(n+1, 3)];
        fib[0] = 0;
        fib[1] = 1;
        fib[2] = 1;
        for(int i = 3; i<n+1; i++) {
            fib[i] = fib[i-1] + fib[i-2] + fib[i-3];
        }
        return fib[n];
    }
}
