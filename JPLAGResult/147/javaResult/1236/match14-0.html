class Solution {
    public int tribonacci(int n) {
        if (n <= 1) {
            return n;
        }
        if (n == 2) {
            return 1;
        }
        int prev1 = 0;
        int prev2 = 1;
        int next = 1;
        while (n >= 3) {
            int temp = prev1;
            prev1 = prev2;
            prev2 = next;
            next = temp + prev1 + prev2;
            n--;
        }
        return next;
    }
}
