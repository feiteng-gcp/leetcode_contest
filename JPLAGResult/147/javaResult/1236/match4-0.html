class Solution {
    public int tribonacci(int n) {
        if(n==0) return 0;
        if(n==1) return 1;
        if(n==2) return 1;
        int prev=1, prevprev=1, prevprevprev=0, cur=0;
        for(int i=2;i<n;i++){
            cur=prev+prevprev+prevprevprev;
            prevprevprev=prevprev;
            prevprev=prev;
            prev=cur;
        }
        return cur;
    }
}
