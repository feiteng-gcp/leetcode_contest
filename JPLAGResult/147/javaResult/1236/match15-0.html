class Solution {
    public int tribonacci(int n) {
        int[] arr = new int[]{0, 1 ,1};
        int sum = 0;
        if(n < 3){
            return arr[n];
        }
        Deque<Integer> dq = new LinkedList<>();
        for(int i : arr){
            dq.addLast(i);
            sum += i;
        }
        for(int i = 3; i < n; i++){
            dq.addLast(sum);
            sum += sum;
            sum -= dq.pollFirst();
        }
        return sum;
    }
}
