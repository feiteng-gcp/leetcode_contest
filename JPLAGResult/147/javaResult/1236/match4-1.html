class Solution {
    public int tribonacci(int n) {
        if (n == 0) return 0;
        if (n == 1) return 1;
        if (n == 2) return 1;
        int n3 = 0, n0 = 0, n1 = 1, n2 = 1;
        for (int i = 3; i <= n; i++) {
            n3 = n0 + n1 + n2;
            n0 = n1;
            n1 = n2;
            n2 = n3;
            // System.out.println(n2);
        }
        return n3;
    }
}
