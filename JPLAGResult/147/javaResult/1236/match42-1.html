class Solution {
    public int tribonacci(int n) {
        if (n == 0) {
            return 0;
        }
        if (n == 1 || n == 2) {
            return 1;
        }
        
int n1 = 0;
        int n2 = 1;
        int n3 = 1;
        while (n >= 3) {
            int tribo = n1 + n2 + n3;
            n1 = n2;
            n2 = n3;
            n3 = tribo;
            n--;
        }
        return n3;
    }
}
