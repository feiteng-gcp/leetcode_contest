class Solution {
    String[] board = {
            "abcde",
            "fghij",
            "klmno",
            "pqrst",
            "uvwxy",
            "z"};

    public String alphabetBoardPath(String target) {
        StringBuilder path = new StringBuilder();
        // 对于每一步都要找到
        char[] chars = target.toCharArray();
        char[][] limit = {
                {'a', 'e'}, {'f', 'j'}, {'k', 'o'}, {'p', 't'}, {'u', 'y'}, {'z', 'z'}
        };
        int c = 0, r = 0;
        for (int i = 0; i < chars.length; i++) {
            // 对于z，特殊处理，直接向左走到头，向下走到头.
            if (chars[i] == 'z') {
                
while (c != 0) {
                    c--;
                    path.append('L');
                }
                while (r != 5) {
                    r++;
                    path.append('D');
                }
                path.append('!');
                continue;
            }
            // 如果当前字符比光标所在行第一个还小，那么上移.
            while (limit[r][0] > chars[i]) {
                r--;
                path.append('U');
            }
            // 下移
            while (limit[r][1] < chars[i]) {
                r++;
                path.append('D');
            }

            // 此时，光标所在行为目的行.
            // 此时，光标应该左移或者右移.
            while (board[r].charAt(c) > chars[i]) {
                // 左移
                c--;
                path.append('L');
            }
            while (board[r].charAt(c) < chars[i]) 
{
                // 左移
                c++;
                path.append('R');
            }
            // 添加到
            path.append('!');
        }
        return path.toString();

    }
}
