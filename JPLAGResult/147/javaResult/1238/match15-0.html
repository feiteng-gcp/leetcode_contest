class Solution {
    public String alphabetBoardPath(String target) {
        char from = 'a';
        StringBuilder path = new StringBuilder();
        for (char c : target.toCharArray()) {
            path.append(getPath(from, c));
            from = c;
        }
        return path.toString();
    }
    
    public String getPath(char from, char to) {
        if (from == 'z' && to == 'z') {
            return "!";
        }
        if (from == 'z') {
            return 'U' + getPath('u', to);
        }
        if (to == 'z') {
            String temp = getPath(from, 'u');
            return temp.substring(0, temp.length() - 1) + 'D' + '!';
        }
        
        int[] fromPos = getRC(from);
        int[] toPos = getRC(to);
        
        StringBuilder sb = new StringBuilder();
        
        if (toPos[0] >= fromPos[0]) {
            for (int i = 0; i < toPos[0] - fromPos[0]; i++) {
                sb.append('D');
            }
        } else {
            for (int i = 0; i < fromPos[0] - toPos[0]; i++) {
                sb.append('U');
            }
        }
        
        if (toPos[1] >= fromPos[1]) {
            for (int i = 0; i < toPos[1] - fromPos[1]; i++) {
                sb.append('R');
            }
        } else {
            for (int i = 0; i < fromPos[1] - toPos[1]; i++) {
                sb.append('L');
            }
        }
        sb.append('!');
        return sb.toString();
    }
    
    public int[] getRC(char c) {
        int num = c - 'a';
        return new int[]{num / 5, num % 5};
    }
}
