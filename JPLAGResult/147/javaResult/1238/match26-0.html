class Solution {
    public String alphabetBoardPath(String target) {
        StringBuilder sb = new StringBuilder();
        int[] base = new int[2];
        for (char c : target.toCharArray()) {
            int[] pos = getPos(c);
            sb.append(getMove(base, pos));
            sb.append('!');
            base[0] = pos[0];
            base[1] = pos[1];
        }
        return sb.toString();
    }
    
    private int[] getPos(char c) {
        int order = c - 'a';
        int[] pos = new int[2];
        pos[0] = order / 5;
        pos[1] = order % 5;
        return pos;
    }
    
    private String getMove(int[] base, int[] target) {
        StringBuilder sb = new StringBuilder();
        int two = target[1] - base[1];
        int one = target[0] - base[0];
        while (two < 0) {
            sb.append('L');
            two++;
        }
        while (one < 0) {
            sb.append('U');
            one++;
        }
        while (one > 0) {
            sb.append('D');
            one--;
        }
        while (two > 0) {
            sb.append('R');
            two--;
        }
        return sb.toString();
    }
}
