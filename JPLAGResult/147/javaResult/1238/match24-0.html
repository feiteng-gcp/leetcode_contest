class Solution {
    public String alphabetBoardPath(String target) {
        if (target == null) {
            return target;
        }
        StringBuilder sb = new StringBuilder();
        buildPath(target, 0, new int[]{0, 0}, sb);
        return sb.toString();        
    }
    
    private void buildPath(String target, int index, int[] curr, StringBuilder sb) {
        if (index == target.length()) {
            return;
        }
        char c = target.charAt(index);
        int x = (int)(c - 'a') % 5;
        int y = (int)(c - 'a') / 5;
        while (x < curr[0]) {
            sb.append('L');
            curr[0]--;                
        }
        while (y < curr[1]) {
            sb.append('U');
            curr[1]--;
        }
        while (x > curr[0]) {
            sb.append('R');
            curr[0]++;
        }
        while (y > curr[1]) {
            sb.append('D');
            curr[1]++;
        }
        sb.append('!');
        buildPath(target, index + 1, curr, sb);
    }
}
