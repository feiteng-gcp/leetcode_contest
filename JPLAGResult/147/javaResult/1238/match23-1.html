class Solution {
    public String alphabetBoardPath(String target) {
        StringBuilder sb = new StringBuilder();
        int x = 0, y = 0;
        for (char c : target.toCharArray()) {
            int s = (c - 'a') / 5;
            int t = (c - 'a') % 5;
            if (s == 5) {
                int n = y - t;
                while (n-- > 0) {
                    sb.append('L');
                }
                n = s - x;
                while (n-- > 0) {
                    sb.append('D');
                }
            } else {
                
if (s > x) {
                    int n = s - x;
                    while (n-- > 0) {
                        sb.append('D');
                    }
                } else if (s < x) 
{
                    int n = x - s;
                    while (n-- > 0) {
                        sb.append('U');
                    }
                
}
                if (t > y) {
                    int n = t - y;
                    while (n-- > 0) {
                        sb.append('R');
                    }
                } else if (t < y) {
                    int n = y - t;
                    while (n-- > 0) {
                        sb.append('L');
                    }
                }
            
}
            sb.append('!');
            x = s;
            y = t;
        }
        return sb.toString();
    }
}
