class Solution {
    public String alphabetBoardPath(String target) {
        StringBuilder sb = new StringBuilder();
        int[] prev = new int[2];
        int[] cur = new int[2];
        for (int i = 0; i < target.length(); i++) {
            cur = calcPos(target.charAt(i));
            walkFromPrevToCur(prev, cur, sb);
            prev = cur;
        }
        return sb.toString();
    }
    
    int[] calcPos(char c) {
        int idx = c - 'a';
        int row = idx / 5;
        int col = idx % 5;
        return new int[]{row, col};
    }
    
    void walkFromPrevToCur(int[] prev, int[] cur, StringBuilder sb) {
        int x = prev[0];
        int y = prev[1];
        int xx = cur[0];
        int yy = cur[1];
        
        if (prev[0] == 5 && cur[0] != 5) {sb.append('U'); x--;}
        
        
        while (y < yy) {
            sb.append('R');
            y++;
        }
        while (y > yy) {
            sb.append('L');
            y--;
        }
        
        while (x < xx) {
            sb.append('D');
            x++;
        }
        while (x > xx) {
            sb.append('U');
            x--;
        }
        
        
        
        sb.append('!');
    }
}
