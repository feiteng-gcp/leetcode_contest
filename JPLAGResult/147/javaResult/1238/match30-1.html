class Solution {
    public String alphabetBoardPath(String target) {
        int i = 0, j = 0, ti, tj;
        StringBuilder sb = new StringBuilder();
        for (char c : target.toCharArray()) {
            c -= 'a';
            ti = c % 5;
            tj = c / 5;
            
if (j == 5 && tj != 5) {
                sb.append('U');
                j--;
            }
            while (i < ti) {
                sb.append('R');
                i++;
            }
            while (i > ti) {
                sb.append('L');
                i--;
            }
            while (j < tj) {
                sb.append('D');
                j++;
            }
            while (j > tj) {
                sb.append('U');
                j--;
            }
            sb.append('!');
        }
        return sb.toString();
    }
}
