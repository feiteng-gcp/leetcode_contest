class Solution {
    public String alphabetBoardPath(String t) {
        char c[] = t.toCharArray();
        
        int a = 0;
        int b = 0;
        
        StringBuilder sb = new StringBuilder();
        for(char cc:c){
            int v = (int)(cc-'a');
            int y = v%5;
            int x = v/5;
            if(cc=='z'){
                
                if(y>=b){
                for(int i=0;i<y-b;++i){
                  sb.append('R');  
                }
            }else{
                for(int i=0;i<b-y;++i){
                  sb.append('L');  
                }
            }
                
                
                
            if(x>=a){
                for(int i=0;i<x-a;++i){
                  sb.append('D');  
                }
            }else{
                for(int i=0;i<a-x;++i){
                  sb.append('U');  
                }
            }
            
            
                
                
                
            }else{
                
                if(x>=a){
                for(int i=0;i<x-a;++i){
                  sb.append('D');  
                }
            }else{
                for(int i=0;i<a-x;++i){
                  sb.append('U');  
                }
            }
            
            if(y>=b){
                for(int i=0;i<y-b;++i){
                  sb.append('R');  
                }
            }else{
                for(int i=0;i<b-y;++i){
                  sb.append('L');  
                }
            }
                
            }
            
            
            
            sb.append("!");
            a = x;
            b = y;
            
            
            
        }
        return sb.toString();
        
    }
}
