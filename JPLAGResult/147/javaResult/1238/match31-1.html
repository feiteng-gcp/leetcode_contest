class Solution {
    public String alphabetBoardPath(String target) {
        int row = 0;
        int col = 0;
        boolean z = false;
        StringBuilder ans = new StringBuilder();
        for(char c : target.toCharArray()) {
            int currow = (c-'a')/5;
            int currcol = (c-'a')%5;
            if(row == 5 && currow != 5){
                ans.append('U');
                row--;
            }
            
if(currow == 5 && row != 5){
                z = true;
                currow--;
            }
            if(currow < row) {
                for(int i = currow; 
                    i< row; i++)
                {
                    ans.append('U');
                }
            } else {
                for(int i = row; i<currow;i ++){
                    ans.append('D');
                }
            }
            if(currcol < col) {
                for(int i = currcol; i<col;i ++) {
                    ans.append('L');
                }
            } else {
                for(int i = col; i<currcol;i ++) {
                    ans.append('R');
                }
            }
            row = currow;
            col = currcol;
            if(z) {
                ans.append('D');
                z = false;
                row++;
            }
            ans.append('!');
        }
        return ans.toString();
    }
}
