class Solution {
    public String alphabetBoardPath(String target) {
        StringBuilder result = new StringBuilder();
        int[] oldPos = new int[]{0, 0};
        for (int i = 0;i<target.length();i++) {
            int[] newPos = getPos(target.charAt(i));
            // System.out.println(Arrays.toString(newPos));
            result.append(getSteps(oldPos, newPos));
            oldPos = newPos;
        }
        return result.toString();
    }
    private int[] getPos(char c) {
        String alphabhet = "abcdefghijklmnopqrstuvwxyz";
        int pos = alphabhet.indexOf(c);
        return new int[]{pos / 5, pos%5};
    }
    
    private String getSteps(int[] oldPos, int[] newPos) {
        StringBuilder sb = new StringBuilder();
        int ro = oldPos[0];
        int co = oldPos[1];
        int rn = newPos[0];
        int cn = newPos[1];
        if (ro != rn || co != cn) {
            if (rn == 5) {
                rn = 4;
            }
            if (ro > rn) {
                for (int i = rn;i<ro;i++) {
                    sb.append("U");
                }
            } else {
                for (int i = ro;i<rn;i++) {
                    sb.append("D");
                }
            }
            if (co > cn) {
                for (int i = cn;i<co;i++) {
                    sb.append("L");
                }
            } else {
                for (int i = co;i<cn;i++) {
                    sb.append("R");
                }
            }
            if (newPos[0] == 5) {
                sb.append("D");
            }
        }
        sb.append("!");
        return sb.toString();
    }
}
