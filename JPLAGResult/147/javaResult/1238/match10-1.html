class Solution {
    // 01234
    // abcde
    // fghij
    // klmno
    // pqrst
    // uvwxy
    // z
    // -----
    // start at z and end at z are special, compare to other letters.
    // 
    public String alphabetBoardPath(String target) {
        StringBuilder builder = new StringBuilder();
        for(int x = 0, y = 0, i = 0, n = target.length(); i < n; ++i) {
            int index = target.charAt(i) - 'a';
            int nextx = index / 5;
            int nexty = index % 5;
            // prefer L and U to accommodate z.
            
while(y > nexty) {
                builder.append('L');
                --y;
            }
            while(x > nextx) {
                builder.append('U');
                --x;
            }
            while(y < nexty) {
                builder.append('R');
                ++y;
            }
            while(x < nextx) {
                builder.append('D');
                ++x;
            }
            // now x == nextx, y == nexty.
            builder.append('!');
        }
        return builder.toString();
    }
}
