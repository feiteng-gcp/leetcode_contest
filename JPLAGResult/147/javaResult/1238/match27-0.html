class Solution {
    public String alphabetBoardPath(String target) {
        char[] cs = target.toCharArray();
        StringBuilder sb = new StringBuilder();
        int x = 0, y = 0;
        for (char c : cs) {            
            int n = c - 'a', i = n / 5, j = n % 5;            
            if (x == i && y == j) {
                sb.append('!');
            } else {
                if ('z' == c) {
                    i--;
                }
                if (5 == x && 0 == y) {
                    x--;
                    sb.append('U');
                }
                move(sb, x, y, i, j);
                if ('z' == c) {
                    i++;
                    sb.append('D');
                }
                sb.append('!');
                x = i;
                y = j;
            }
        }
        return sb.toString();
    }
    private void move(StringBuilder sb, int x, int y, int i, int j) {
        while (x < i) {
            x++;
            sb.append('D');
        }
        while (x > i) {
            x--;
            sb.append('U');
        }
        while (y < j) {
            y++;
            sb.append('R');
        }
        while (y > j) {
            y--;
            sb.append('L');
        }
    }
}
