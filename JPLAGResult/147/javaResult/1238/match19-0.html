class Solution {
    public String alphabetBoardPath(String target) {
        StringBuilder result = new StringBuilder();
        char current = 'a';
        for (int i = 0; i < target.length(); i++) {
            char to = target.charAt(i);
            result.append(move(current, to));
            result.append('!');
            current = to;
        }
        return result.toString();
    }
    
    private static String move(char from, char to) {
        if (from == to) {
            return "";
        }
        int fromR = getRow(from);
        int fromC = getCol(from);
        int toR = getRow(to);
        int toC = getCol(to);
        StringBuilder result = new StringBuilder();
        if (from == 'z') {
            if (fromR <= toR) {
                for (int i = 0; i < toR - fromR; i++) {
                    result.append('D');
                }
            } else {
                for (int i = 0; i < fromR - toR; i++) {
                    result.append('U');
                }
            }
            if (fromC <= toC) {
                for (int i = 0; i < toC - fromC; i++) {
                    result.append('R');
                }
            } else {
                for (int i = 0; i < fromC - toC; i++) {
                    result.append('L');
                }
            }
        } else {
            if (fromC <= toC) {
                for (int i = 0; i < toC - fromC; i++) {
                    result.append('R');
                }
            } else {
                for (int i = 0; i < fromC - toC; i++) {
                    result.append('L');
                }
            }
            if (fromR <= toR) {
                for (int i = 0; i < toR - fromR; i++) {
                    result.append('D');
                }
            } else {
                for (int i = 0; i < fromR - toR; i++) {
                    result.append('U');
                }
            }
        }
        return result.toString();
    }
    
    private static int getRow(char c) {
        return (c - 'a') / 5;
    }
    
    private static int getCol(char c) {
        return (c - 'a') % 5;
    }
}
