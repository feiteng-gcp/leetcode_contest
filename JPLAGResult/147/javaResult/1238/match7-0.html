class Solution {
    public String alphabetBoardPath(String target) {
        int[] pos = new int[2];
        StringBuilder sb = new StringBuilder();
        for (char c : target.toCharArray()) {
            int[] np = new int[]{(c - 'a') / 5, (c - 'a') % 5};
            if (pos[0] >= np[0] && pos[1] >= np[1]) {
                while (pos[0] > np[0]) {
                    pos[0]--;
                    sb.append('U');
                }
                while (pos[1] > np[1]) {
                    pos[1]--;
                    sb.append('L');
                }
            }
            if (pos[0] >= np[0] && pos[1] < np[1]) {
                while (pos[0] > np[0]) {
                    pos[0]--;
                    sb.append('U');
                }
                while (pos[1] < np[1]) {
                    pos[1]++;
                    sb.append('R');
                }
            }
            if (pos[0] < np[0] && pos[1] >= np[1]) {
                while (pos[1] > np[1]) {
                    pos[1]--;
                    sb.append('L');
                }
                while (pos[0] < np[0]) {
                    pos[0]++;
                    sb.append('D');
                }
            }
            if (pos[0] < np[0] && pos[1] < np[1]) {
                while (pos[1] < np[1]) {
                    pos[1]++;
                    sb.append('R');
                }
                while (pos[0] < np[0]) {
                    pos[0]++;
                    sb.append('D');
                }
            }
            sb.append('!');
        }
        return sb.toString();
    }
}
