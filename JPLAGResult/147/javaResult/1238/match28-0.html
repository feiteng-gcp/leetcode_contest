class Solution {
    char[][] maze = {
        {'a', 'b', 'c', 'd', 'e'},
        {'f', 'g', 'h', 'i', 'j'},
        {'k', 'l', 'm', 'n', 'o'},
        {'p', 'q', 'r', 's', 't'},
        {'u', 'v', 'w', 'x', 'y'},
        {'z'}
    };
    
    Map<Character, int[]> posMap = new HashMap<>();
    
    public String alphabetBoardPath(String target) {
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < maze.length; ++i) {
            for (int j = 0; j < maze[i].length; ++j) {
                posMap.put(maze[i][j], new int[] { i, j });
            }
        }
        
        char prev = 'a';
        for (char ch: target.toCharArray()) {
            sb.append(getPath(prev, ch));
            prev = ch;
        }
        
        return sb.toString();
    }
    
    String getPath(char c1, char c2) {
        if (c1 == c2) return "!"; 
        StringBuilder sb = new StringBuilder();
        
        int[] p1 = posMap.get(c1);
        int[] p2 = posMap.get(c2);
        int yDiff = p2[0] - p1[0], xDiff = p2[1] - p1[1];
        //System.out.println(yDiff + " " + xDiff);
        //System.out.println("    " + Arrays.toString(p1));
        //System.out.println("    " + Arrays.toString(p2));
        StringBuilder horiz = new StringBuilder();
        StringBuilder vert = new StringBuilder();
        if (yDiff < 0) {
            for (int i = 0; i < Math.abs(yDiff); ++i) {
                vert.append('U');
            }
        } else {
            for (int i = 0; i < yDiff; ++i) {
                vert.append('D');
            }
        }
        
        if (xDiff < 0) {
            for (int i = 0; i < Math.abs(xDiff); ++i) {
                horiz.append('L');
            }
        } else {
            for (int i = 0; i < xDiff; ++i) {
                horiz.append('R');
            }
        }
        if (c1 == 'z') {
            sb.append(vert).append(horiz);
        } else if (c2 == 'z') {
            sb.append(horiz).append(vert);
        } else {
            sb.append(horiz).append(vert);
        }
        
        sb.append('!');
        return sb.toString();
    }
}
