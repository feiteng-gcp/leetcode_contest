class Solution {
    public String alphabetBoardPath(String target) {
        Map<Integer, int[]> map = new HashMap<>();
        
for(int i = 0; i < 26; i++){
            map.put(i, new int[]{i / 5, i % 5});
        }
        StringBuilder sb = new StringBuilder();
        int from = 0;
        for(int i = 0; i < target.length(); i++){
            int pos = 
target.charAt(i) - 'a';
            helper(map, from, pos, sb);
            from = pos;
        }
        return sb.toString();
    }
    
    private void helper(Map<Integer, int[]> map, int from, int to, StringBuilder sb){
        int[] fpos = map.get(from), tpos = map.get(to);
        
if(fpos[0] > tpos[0]){
            yMove(sb, fpos[0], tpos[0]);
            xMove(sb, fpos[1], tpos[1]);
        }else{
            xMove(sb, fpos[1], tpos[1]);
            yMove(sb, fpos[0], tpos[0]);
        }
        sb.append('!');
    }
    
    private void yMove(StringBuilder sb, int from, int to){
        if(from > to){
            for(int i = 0; i < from - to; i++)
{
                sb.append('U');
            }
        }else{
            for(int i = 0; i < to - from; i++){
                sb.append('D');
            }
        }
    }
    
    private void xMove(StringBuilder sb, int from, int to){
        if(from > to){
            for(int i = 0; i < from - to; i++)
{
                sb.append('L');
            }
        }else{
            for(int i = 0; i < to - from; i++){
                sb.append('R');
            }
        }
    }
}
