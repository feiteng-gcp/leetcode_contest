class Solution {
    public String alphabetBoardPath(String target) {
        int cx = 0;
        int cy = 0;
        StringBuilder ans = new StringBuilder();
        for (char c : target.toCharArray()) {
            int nx = (c-'a')/5;
            int ny = (c-'a')%5;
            
if (c=='z') {
                while (cy < ny) {
                    ans.append('R');
                    cy++;
                }
                while (cy > ny) {
                    ans.append('L');
                    cy--;
                }
                
while (cx < nx) {
                    ans.append('D');
                    cx++;
                }
                while (cx > nx) {
                    ans.append('U');
                    cx--;
                }
                ans.append('!');
                continue;
            }
            
while (cx < nx) {
                ans.append('D');
                cx++;
            }
            while (cx > nx) {
                ans.append('U');
                cx--;
            }
            
while (cy < ny) {
                ans.append('R');
                cy++;
            }
            while (cy > ny) {
                ans.append('L');
                cy--;
            }
            ans.append('!');
        }
        return ans.toString();
    }
}
