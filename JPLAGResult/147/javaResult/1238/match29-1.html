class Solution {
    public String alphabetBoardPath(String target) {
        if(target == null || target.equals("")) return "";
        char[] cs = target.toCharArray();
        int[] pre = new int[]{0,0};
        StringBuilder sb = new StringBuilder();
        for(char c: cs) {
            int[] cur = calPos(c);
            int[] diff = new int[]{pre[0]-cur[0], pre[1]-cur[1]};
            String[] t = buildString(diff);
            if(c == 'z') {
                sb.append(t[1]);
                sb.append(t[0]);
            } else {
                sb.append(t[0]);
                sb.append(t[1]);
            }
            
sb.append("!");
            pre = cur;
        }
        return sb.toString();
    }
    
    private int[] calPos(char c) {
        int num = c-'a';
        int row = num/5;
        int col = num%5;
        return new int[]{row, col};
    }
    // cur在前
    private String[] buildString(int[] diff) {
        int row = diff[0], col = diff[1];
        StringBuilder sbRow = new StringBuilder();
        
while (row > 0) {
            sbRow.append("U");
            row--;
        }
        while (row < 0) {
            sbRow.append("D");
            row++;
        }
        
        StringBuilder sbCol = new StringBuilder();
        
while (col > 0) {
            sbCol.append("L");
            col--;
        }
        while (col < 0) {
            sbCol.append("R");
            col++;
        }
        return new String[]{sbRow.toString(), sbCol.toString()};
    }
}
