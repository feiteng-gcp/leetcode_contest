class Solution {
public String alphabetBoardPath(String target) {
        String[] board = new String[] {"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"};
        int x = 0;
        int y = 0;
        char cur = 'a';
        
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < target.length(); i++) {
            char ch = target.charAt(i);
            int nx = (ch - 'a') / 5;
            int ny = (ch - 'a') % 5;
            if (x == 5) {
                if (nx < x) {
                    sb.append("U");
                    x--;
                }
            }
            
            if (ny < y) {
                int left = y - ny;
                while (left > 0) {
                    sb.append("L");
                    left--;
                }
            } else {
                int right = ny - y;
                while (right > 0) {
                    sb.append("R");
                    right--;
                }
            }
            
            if (nx < x) {
                int up = x - nx;
                
                while (up > 0) {
                    sb.append("U");
                    up--;
                }   
            } else {
                int down = nx - x;
                
                while (down > 0) {
                    sb.append("D");
                    down--;
                }
            }
            
            sb.append("!");
            x = nx;
            y = ny;
        }
        return sb.toString();
    }
    
}
