class Solution {
    public String alphabetBoardPath(String target) {
        char fir = 'a';
        StringBuilder sb = new StringBuilder();
        for (char sec : target.toCharArray()) {
            help(fir, sec, sb);
            fir = sec;
        }
        return sb.toString();
    }
    
    private void help(char fir, char sec, StringBuilder sb) {
        
int frow = (fir - 'a') / 5, srow = (sec - 'a') / 5;
        int fcol = (fir - 'a') % 5, scol = (sec - 'a') % 5;
        if (fir != 'z') {
            if (fcol > scol) {
                for (int i = 0; i < fcol - scol; ++i) {
                    sb.append('L');
                }
            } else {
                for (int i = 0; i < scol - fcol; ++i) {
                    sb.append('R');
                }
            }
            if (frow > srow) {
                for (int i = 0; i < frow - srow; ++i) {
                    sb.append('U');
                }
            } else {
                for (int i = 0; i < srow - frow; ++i) {
                    sb.append('D');
                }
            }
        } else {
            if (frow > srow) {
                for (int i = 0; i < frow - srow; ++i) {
                    sb.append('U');
                }
            } else {
                for (int i = 0; i < srow - frow; ++i) {
                    sb.append('D');
                }
            }
            if (fcol > scol) {
                for (int i = 0; i < fcol - scol; ++i) {
                    sb.append('L');
                }
            } else {
                for (int i = 0; i < scol - fcol; ++i) {
                    sb.append('R');
                }
            }
        }
        sb.append('!');
    }
}
