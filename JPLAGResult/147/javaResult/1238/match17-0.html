class Solution {
    public String alphabetBoardPath(String target) {
        int x = 0, y = 0;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < target.length(); i++){
            char c = target.charAt(i);
            int next = c - 'a';
            int newx = next / 5;
            int newy = next % 5;
            build(sb, x, y, newx, newy);
            x = newx;
            y = newy;
        }
        return sb.toString();
    }
    
    private void build(StringBuilder sb, int x, int y, int newx, int newy){
        while(newy < y){
            sb.append('L');
            y--;
        }
        while (newx > x){
            sb.append('D');
            x++;
        }
        while(newx < x){
            sb.append('U');
            x--;
        }
        while(newy > y){
            sb.append('R');
            y++;
        }
        
        sb.append('!');
        
    }
}
