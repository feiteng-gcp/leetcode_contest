class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        const int n=grid.size(), m=grid[0].size();
        vector<vector<int>> r(n, vector<int>(m, 0)), d(r);
        for (int i=n-1; ~i; i--)
            for (int j=m-1; ~j; j--) {
                if (grid[i][j]==0) r[i][j]=j-1, d[i][j]=i-1;
                else r[i][j]=j==m-1?j:r[i][j+1], d[i][j]=i==n-1?i:d[i+1][j];
            }
        for (int i=0; i<n; i++)
            for (int j=0; j<m; j++)
                cout << r[i][j] << (j==m-1?'\n':' ');
        for (int i=0; i<n; i++)
            for (int j=0; j<m; j++)
                cout << d[i][j] << (j==m-1?'\n':' ');
        int ans=-1;
        for (int i=0; i<n; i++)
            for (int j=0; j<m; j++)
                for (int k=ans+1; i+k<=d[i][j] && j+k<=r[i][j]; k++) 
                    if (j+k<=r[i+k][j] && i+k<=d[i][j+k]) ans=k;
        return (ans+1)*(ans+1);
    }
};
