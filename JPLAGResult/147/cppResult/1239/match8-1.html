int r[105][105];
int c[105][105]; 

class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        for(int i = 0; i <= n; i++) 
            for(int j = 0; j <= m; j++) {
                r[i][j] = c[i][j] = 0;
            }
        

        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= m; j++) {
                c[i][j] = a[i - 1][j - 1] + c[i][j - 1];
            }
        }
        for(int j = 1; j <= m; j++) {
            for(int i = 1; i <= n; i++) {
                r[i][j] = a[i - 1][j - 1] + r[i - 1][j];
            }
        }

        int ans = 0;
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= m; j++) {
                for(int k = 0; i + k <= n && j + k <= m; k++) {
                    if((r[i + k][j] - r[i - 1][j] == k + 1)
                        && (r[i + k][j + k] - r[i - 1][j + k] == k + 1)
                        && (c[i][j + k] - c[i][j - 1] == k + 1)
                        && (c[i + k][j + k] - c[i + k][j - 1] == k + 1)) {
                        if((k + 1) * (k + 1) > ans) {
                            ans = (k + 1) * (k + 1);
                        }
                    }
                }
            } 
        }
        return ans;
    }
};
