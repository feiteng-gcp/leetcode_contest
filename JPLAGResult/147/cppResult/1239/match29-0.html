class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int ans = 0;
        vector<vector<int>> prefixX, prefixY;
        for(int i = 0; i < grid.size(); i++) {
            vector<int> row;
            row.push_back(0);
            prefixX.push_back(row);
        }
        for(int i = 0; i < grid[0].size(); i++) {
            vector<int> row;
            row.push_back(0);
            prefixY.push_back(row);
        }
        for(int i = 0; i < grid.size(); i++) {
            for(int j = 0; j < grid[i].size(); j++) {
                prefixX[i].push_back(prefixX[i][j] + grid[i][j]);
            }
        }
        for(int j = 0; j < grid[0].size(); j++) {
            for(int i = 0; i < grid.size(); i++) {
                prefixY[j].push_back(prefixY[j][i] + grid[i][j]);
            }
        }
        int maxBound = min(grid[0].size(), grid.size());
        for(int L = 0; L < maxBound; L++) {
            for(int i = 1; i <= grid.size(); i++) {
                for(int j = 1; j <= grid[0].size(); j++) {
                    if(i + L <= grid.size() && j + L <= grid[0].size()) {
                        if(prefixX[i-1][j+L] - prefixX[i-1][j-1] == L + 1 && prefixX[i+L-1][j+L] - prefixX[i+L-1][j-1] == L + 1 && prefixY[j-1][i+L] - prefixY[j-1][i-1] == L + 1 && prefixY[j+L-1][i+L] - prefixY[j+L-1][i-1] == L + 1) {
                            ans = max(ans, (L+1) * (L+1));
                        }
                    }
                }
            }
        }
        return ans;
    }
};
