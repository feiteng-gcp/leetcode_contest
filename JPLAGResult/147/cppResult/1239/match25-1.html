class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int R = grid.size();
        int C = grid[0].size();
        vector<vector<int>> r(R, vector<int>(C, 0));
        vector<vector<int>> d = r;
        vector<vector<int>> l = r;
        vector<vector<int>> u = r;

        for(int i = 0; i < R; ++i)
        {
            for(int j = 0; j < C; ++j)
            {
                if(grid[i][j] == 1)
                {
                    l[i][j] = j > 0? l[i][j-1] + 1 : 1;
                    u[i][j] = i > 0? u[i-1][j] + 1 : 1;

                }
            }
        }
        for(int i = R - 1; i >= 0; --i)
        {
            for(int j = C - 1; j >= 0; --j)
            {
                if(grid[i][j] == 1)
                {
                    r[i][j] = j < C - 1? r[i][j+1] + 1 : 1;
                    d[i][j] = i < R - 1? d[i+1][j] + 1: 1;

                }
            }
        }
        
        int ret = 0;
        for(int i = 0; i < R; ++i)
        {
            for(int j = 0; j < C; ++j)
            {
                if(grid[i][j] == 0)
                {
                    continue;
                }
                
                for(int k = 0; k < min(r[i][j], d[i][j]) && i + k < R && j + k < C; ++k)
                {
                    if(l[i+k][j+k] > k && u[i+k][j+k] > k)
                    {
                        ret = max(ret, (k + 1) * (k + 1));
                    }
                }
            }
        }
        
        return ret;
    }
};
