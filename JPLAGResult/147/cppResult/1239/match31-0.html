class Solution {
public:
	int largest1BorderedSquare(vector<vector<int>>& grid) {

		int n = grid.size();
		int m = grid[0].size();

		vector<vector<int>> l(n, vector<int>(m, 0));
		vector<vector<int>> u(n, vector<int>(m, 0));

		for (int i = 0; i<n; i++)
		{
			for (int j = 0; j<m; j++)
			{
				if (i == 0 && j == 0)
				{
					if (grid[i][j] == 1)
					{
						l[i][j] = 1;
						u[i][j] = 1;
					}
				}
				else if (i == 0)
				{
					if (grid[i][j] == 1)
					{
						l[i][j] = l[i][j - 1] + 1;
						u[i][j] = 1;
					}
				}
				else if (j == 0)
				{
					if (grid[i][j] == 1)
					{
						l[i][j] = 1;
						u[i][j] = u[i - 1][j] + 1;
					}
				}
				else
				{
					if (grid[i][j] == 1)
					{
						l[i][j] = l[i][j - 1] + 1;
						u[i][j] = u[i - 1][j] + 1;
					}
				}
			}
		}
		//cout << "---l" << endl;
		//for (int i = 0; i < n; i++)
		//{
		//	for (int j = 0; j < m; j++)
		//		cout << l[i][j] << '\t';
		//	cout << endl;
		//}
		//cout << "----u" << endl;
		//for (int i = 0; i < n; i++)
		//{
		//	for (int j = 0; j < m; j++)
		//		cout << u[i][j] << '\t';
		//	cout << endl;
		//}
		int ans = 0;
		for (int i = 0; i<n; i++)
		{
			for (int j = 0; j<m; j++)
			{
				if (grid[i][j] == 1)
				{
					for (int len = 1; i - len+1 >= 0 && j - len+1 >= 0; len++)
					{
						if (l[i][j] >= len && u[i][j] >= len && l[i - len+1][j] >= len && u[i][j - len+1] >= len)
						{
							if (len > ans)
							{
								//cout << i << '.' << j << endl;
								ans = len;
							}
						}
					}
				}
			}
		}

		return (ans)*(ans);
	}
};
