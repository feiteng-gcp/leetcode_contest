class Solution {
public:
    vector<vector<int>> cum_row, cum_col;
    
    int largest1BorderedSquare(vector<vector<int>>& grid) {

        int n = grid.size(), m = grid[0].size();
        cum_row.resize(n);
        cum_col.resize(m);
        
        for (int i = 0; i < n; i++) {
            cum_row[i].assign(m + 1, 0);
            for (int j = 0; j < m; j++) {
                cum_row[i][j + 1] = cum_row[i][j] + grid[i][j];
            }

        }
        
        for (int j = 0; j < m; j++) {
            cum_col[j].assign(n + 1, 0);
            for (int i = 0; i < n; i++) {
                cum_col[j][i + 1] = cum_col[j][i] + grid[i][j];

            }
        }
        
        for (int k = min(n, m); k > 0; k--) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                    if (i + k <= n && j + k <= m) {
                        if (cum_row[i][j + k] - cum_row[i][j] == k &&
                            cum_row[i + k - 1][j + k] - cum_row[i + k - 1][j] == k &&
                            cum_col[j][i + k] - cum_col[j][i] == k &&
                            cum_col[j + k - 1][i + k] - cum_col[j + k - 1][i] == k) {
                            return k * k;
                        }
                    }
                }
            }
        }
        
        return 0;
    }
};
