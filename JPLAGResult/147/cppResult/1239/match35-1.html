class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& g) {
        int n = g.size(), m = g[0].size(), ans = 0;
        vector<vector<int>> left(n, vector<int>(m)), up(n, vector<int>(m));
        

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (g[i][j] == 1) {
                    left[i][j] = j == 0 ? 1 : left[i][j-1] + 1;
                    up[i][j] = i == 0 ? 1 : up[i-1][j] + 1;
                    

                    int lim = min(up[i][j], left[i][j]);
                    for (int k = lim; k > ans; --k) {
                        if (left[i-k+1][j] >= k && up[i][j-k+1] >= k) {
                    
                            ans = k;
                            break;
                        }
                    }
                }
            }
        }
        
        return ans * ans;
    }
};
