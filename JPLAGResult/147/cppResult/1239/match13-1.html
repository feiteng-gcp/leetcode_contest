class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        int max_U[m][n], max_D[m][n], max_L[m][n], max_R[m][n], length;
        for (int i = 0; i < m; i++) {

            length = 0;
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    length++;
                }
                else {
                    length = 0;
                }
                max_D[i][j] = length;
            }

            length = 0;
            for (int j = n - 1; j >= 0; j--) {
                if (grid[i][j] == 1) {
                    length++;
                }
                else {
                    length = 0;
                }
                max_U[i][j] = length;
            }
        }
        for (int j = 0; j < n; j++) {

            length = 0;
            for (int i = 0; i < m; i++) {
                if (grid[i][j] == 1) {
                    length++;
                }
                else {
                    length = 0;
                }
                max_L[i][j] = length;
            }

            length = 0;
            for (int i = m - 1; i >= 0; i--) {
                if (grid[i][j] == 1) {
                    length++;
                }
                else {
                    length = 0;
                }
                max_R[i][j] = length;
            }
        }

        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int t = min(max_R[i][j], max_U[i][j]);
                for (int k = 1; k <= t; k++) {
                    if (min(max_L[i + k - 1][j + k - 1], max_D[i + k - 1][j + k - 1]) >= k) {
                        ans = max(ans, k);
                    }
                }
            }
        }
        return ans * ans;
    }
};
