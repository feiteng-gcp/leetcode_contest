class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        
        vector<vector<int>> r_prefix(n, vector<int> (m + 1, 0));
        vector<vector<int>> c_prefix(m, vector<int> (n + 1, 0));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                r_prefix[i][j + 1] = r_prefix[i][j] + grid[i][j]; 
            }
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                c_prefix[i][j + 1] = c_prefix[i][j] + grid[j][i]; 
            }
        }
        
        int largest = min(n, m);
        int res = 0;
        for (int step = 1; step <= largest; ++step) {

            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    int drx = i + step - 1, dry = j + step - 1;
                    if (drx >= n || dry >= m) break;
                    if (r_prefix[i][dry + 1] - r_prefix[i][j] == step &&
                        r_prefix[drx][dry + 1] - r_prefix[drx][j] == step &&
                        c_prefix[j][drx + 1] - c_prefix[j][i] == step &&
                        c_prefix[dry][drx + 1] - c_prefix[dry][i] == step) {
                        res = step * step;    
                        break;
                    }
                }
            }
        }
        return res;
    }
};
