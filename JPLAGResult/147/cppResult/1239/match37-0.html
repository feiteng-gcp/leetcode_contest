class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int h = grid.size();
        int w = grid[0].size();
        int s[101][101], t[101][101];
        for(int i=0; i<h; ++i) {
            s[i][0] = 0;
            for(int j=0; j<w; ++j) {
                s[i][j+1] = s[i][j] + grid[i][j];
            }
        }
        for(int j=0; j<w; ++j) {
            t[0][j] = 0;
            for(int i=0; i<h; ++i) {
                t[i+1][j] = t[i][j] + grid[i][j];
            }
        }
        int ans = 0;
        for(int i=0; i<h; ++i) {
            for(int j=0; j<w; ++j) {
                int k = 1;
                while(i+k <= h && j+k <= w) {
                    if((s[i][j+k] - s[i][j] == k) && (t[i+k][j] - t[i][j] == k) && (s[i+k-1][j+k] - s[i+k-1][j] == k) && (t[i+k][j+k-1] - t[i][j+k-1] == k)) {
                        if(ans < k*k) {
                            ans = k*k;
                        }
                    }
                    ++k;
                }
            }
        }
        return ans;
    }
};
