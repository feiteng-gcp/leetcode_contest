class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& g) {
        int r=g.size(),c=g[0].size(),ans=0;
        vector<vector<int>> h(r, vector<int>(c));
        vector<vector<int>> w(r, vector<int>(c));
        for(int i=0;i<r;++i){
            int pre=0;
            for(int j=0;j<c;++j){
                if(g[i][j]) w[i][j] = ++pre;
                else pre = 0;
            }
        }
        for(int i=0;i<c;++i){
            int pre=0;
            for(int j=0;j<r;++j){
                if(g[j][i]) h[j][i] = ++pre;
                else pre = 0;
            }
        }
        for(int n=min(r,c);n;--n){
            int m=n-1;
            for(int i=r-1;i>=m;--i){
                for(int j=c-1;j>=m;--j){
                    if(h[i][j]>=n && w[i][j]>=n && h[i][j-m]>=n && w[i-m][j]>=n)
                        return n*n;
                }
            }    
        }
        return 0;
    }
};
