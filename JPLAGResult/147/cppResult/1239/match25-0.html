class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>> ud(grid);
        vector<vector<int>> lr(grid);
        
        // ud
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if(grid[i][j]==1){
                    if(i==0){
                        ud[i][j] = 1;    
                    }else{
                        ud[i][j] = 1 + ud[i-1][j]; 
                    }
                }
            }
        }

        // lr
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if(grid[i][j]==1){
                    if(j==0){
                        lr[i][j] = 1;    
                    }else{
                        lr[i][j] = 1 + lr[i][j-1]; 
                    }
                }
            }
        }
        
        int result = 0;
        
        /*
            1 1 1 
            2 0 2 
            3 1 3 

            1 2 3 
            1 0 1 
            1 2 3 
        */
        
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                int size =  min(lr[i][j],ud[i][j]);
                if(size>0){
                    while(size > 0){
                        // check 
                        if(ud[i][j-size+1] >= size && lr[i-size+1][j] >= size){
                            result = max(result,size*size);
                            break;
                        }
                        size--;
                    } 
                }
            }
        }        
        
        
        return result;
    }
};
