class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int> > right(n + 2, vector<int>(m + 2, 0));
        vector<vector<int> > down(n + 2, vector<int>(m + 2, 0));
        
        for (int i = 1; i <= n; ++i)
            for (int j = m; j >= 1; --j) {
                if (grid[i - 1][j - 1] == 1)
                    right[i][j] = right[i][j + 1] + 1;
            }
        
        for (int i = n; i >= 1; --i)
            for (int j = 1; j <= m; ++j) {
                if (grid[i - 1][j - 1] == 1)

                    down[i][j] = down[i + 1][j] + 1;
            }
        
        // for (int i = 1; i <= n; ++i)
        //     for (int j = 1; j <= m; ++j)
        //         cout << i << j << "right: " << right[i][j] << ", down: " << down[i][j] << endl;
        int ans = 0;
        for (int i = 1; i <= n; ++i)
            for (int j = 1; j <= m; ++j)
                for (int len = 1; (i + len - 1 <= n) && (j + len - 1 <= m); ++len) {
                   if (right[i][j] >= len && down[i][j] >= len && down[i][j + len - 1] >= len && right[i + len - 1][j] >= len) {
                       ans = max(ans, len * len);
                   } 
                }
        return ans;
    }
};
