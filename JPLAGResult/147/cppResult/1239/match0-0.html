class Solution {
public:
    int l[110][110],r[110][110],d[110][110],u[110][110];
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int n=grid.size(),m=grid[0].size();
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                if(grid[i-1][j-1]){l[i][j]=l[i][j-1]+1;u[i][j]=u[i-1][j]+1;}
                else {l[i][j]=0;u[i][j]=0;}
            }
        }
        for(int i=n;i>=1;i--){
            for(int j=m;j>=1;j--){
                if(grid[i-1][j-1]){r[i][j]=r[i][j+1]+1;d[i][j]=d[i+1][j]+1;}
                else {r[i][j]=0;d[i][j]=0;}
            }
        }
        int ans=0;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                if(grid[i-1][j-1]){
                    ans=max(ans,1);
                    for(int k=2;k<=min(i,j);k++){
                        if(l[i][j]<k||u[i][j]<k)continue;
                        if(d[i-k+1][j-k+1]<k||r[i-k+1][j-k+1]<k)continue;
                        ans=max(ans,k*k);
                    }
                }
            }
        }
        return ans;
    }
};
