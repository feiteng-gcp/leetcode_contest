class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> hsum(m, vector<int>(n + 1, 0));
        vector<vector<int>> vsum(m + 1, vector<int>(n, 0));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                hsum[i][j+1] = hsum[i][j] + grid[i][j];
            }
        }
        for (int j = 0; j < n; ++j) {
            for (int i = 0; i < m; ++i) {
                vsum[i+1][j] = vsum[i][j] + grid[i][j];
            }
        }
        int ret = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                int maxl = min(i + 1, j + 1);
                for (int l = 1; l <= maxl; ++l) {
                    int l1 = hsum[i][j+1] - hsum[i][j-l+1];
                    int l2 = hsum[i-l+1][j+1] - hsum[i-l+1][j-l+1];
                    int l3 = vsum[i+1][j] - vsum[i-l+1][j];
                    int l4 = vsum[i+1][j-l+1] - vsum[i-l+1][j-l+1];
                    if (l1 == l && l2 == l && l3 == l && l4 == l) {
                        ret = max(ret, l * l);
                    }
                }
            }
        }
        return ret;
    }
};
