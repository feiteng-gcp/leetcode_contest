class Solution {
public:
    int dd[100][100];
    int rr[100][100];
    

public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        memset(dd, 0, sizeof(dd));
        memset(rr, 0, sizeof(rr));
        
        for (int i = m - 1; i >= 0; --i) {
            for (int j = n - 1; j >= 0; --j) {
                if (grid[i][j]) {
                    if (i == m - 1) dd[i][j] = 1; else dd[i][j] = dd[i + 1][j] + 1;
                    if (j == n - 1) rr[i][j] = 1; else rr[i][j] = rr[i][j + 1] + 1;
                }
                else {
                    dd[i][j] = rr[i][j] = 0;
                }
                // cout << "dd, rr = " << i << " " << j << " " << dd[i][j] << " " << rr[i][j] << "\n";

            }
        }
        
        for (int sz = min(m, n); sz >= 1; --sz) {
            for (int i = 0; i <= m - sz; ++i) {
                for (int j = 0; j <= n - sz; ++j) {
                    if (dd[i][j] >= sz && rr[i][j] >= sz && dd[i][j + sz - 1] >= sz && rr[i + sz - 1][j] >= sz) {
                        return sz * sz;
                    }
                }
            } 
        }
        
        return 0;
    }
};
