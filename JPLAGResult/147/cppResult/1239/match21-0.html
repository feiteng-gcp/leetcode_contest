class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int nr = grid.size();
        int nc = grid[0].size();
        vector<vector<int>> h(nr, vector<int>(nc));
        vector<vector<int>> v(nr, vector<int>(nc));
        v[0] = grid[0];
        for (int r = 1; r<nr; r++) {
            for (int c = 0; c<nc; c++) {
                v[r][c] = grid[r][c] ? v[r-1][c] + 1 : 0;
            }
        }
        for (int r = 0; r<nr; r++) {
            h[r][0] = grid[r][0];
        }
        for (int r = 0; r<nr; r++) {
            for (int c = 1; c<nc; c++) {
                h[r][c] = grid[r][c] ? h[r][c-1] + 1 : 0;
            }
        }
        
        for (int s = min(nr,nc); s>0; s--) {
            for (int r = 0; r<nr-s+1; r++) {
                for (int c = 0; c<nc-s+1; c++) {
                    if (h[r][c+s-1]>=s 
                     && h[r+s-1][c+s-1]>=s 
                     && v[r+s-1][c]>=s 
                     && v[r+s-1][c+s-1]>=s)
                    { return s*s; }
                }
            }
        }
        return 0;
    }
};
