class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> horz(m, vector<int>(n));
        vector<vector<int>> vert(m, vector<int>(n));
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                horz[i][j] = (j > 0 ? horz[i][j-1] : 0) + grid[i][j];

            }
        }
        
        for (int j = 0; j < n; j++) {
            for (int i = 0; i < m; i++) {
                vert[i][j] = (i > 0 ? vert[i-1][j] : 0) + grid[i][j];

            }
        }
        
        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int d = 0; d <= i; d++) {
                for (int j = d; j < n; j++) {
                    int x = i-d, y = j-d, d1 = d+1;
                    if (vert[i][j] - (x > 0 ? vert[x-1][j] : 0) != d1) continue;
                    if (vert[i][y] - (x > 0 ? vert[x-1][y] : 0) != d1) continue;
                    if (horz[i][j] - (y > 0 ? horz[i][y-1] : 0) != d1) continue;
                    if (horz[x][j] - (y > 0 ? horz[x][y-1] : 0) != d1) continue;
                    ans = max(ans, d1 * d1);
                }
            }
        }
        return ans;
    }
};
