class Solution {
public:
    int X[105][105], Y[105][105];
    bool can(int x1, int y1, int x2, int y2, int k) {
        //if (k == 3) printf("%d %d %d %d\n", );
        return 
            Y[x1][y2] - Y[x1][y1-1] == k &&
            Y[x2][y2] - Y[x2][y1-1] == k &&
            X[x2][y1] - X[x1-1][y1] == k && 
            X[x2][y2] - X[x1-1][y2] == k;
    }
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int N = grid.size(), M = grid[0].size();
        X[0][0] = Y[0][0] = 0;
        for (int i=1;i<=N;++i)
            for (int j=1;j<=M;++j) {
                X[i][j] = X[i-1][j] + grid[i-1][j-1];
                Y[i][j] = Y[i][j-1] + grid[i-1][j-1];
            }
        int ans = 0;
        for (int k=1;k<=min(N,M);++k)
            for (int i=1;i<=N-k+1;++i)
                for (int j=1;j<=M-k+1;++j)
                    if (can(i,j,i+k-1,j+k-1, k)) ans = k;
        return ans * ans;
    }
};
