class Solution {
public:
    
    void setBorderMap(vector<vector<int>>& m, vector<vector<int>>& right, vector<vector<int>>& down){
        int r = m.size();
        int c = m[0].size();

        if (m[r-1][c-1] == 1){
            right[r-1][c-1] = 1;
            down[r-1][c-1] = 1;
        }
        for (int i = r-2; i != -1; i--){
            if (m[i][c-1] == 1){
                right[i][c-1] = 1;
                down[i][c-1] = down[i+1][c-1] + 1;
            }
        }
        for(int i = c - 2; i != -1; --i){
            if (m[r-1][i] == 1){
                right[r-1][i] = right[r-1][i+1] + 1;
                down[r-1][i] = 1;
            }
        }
        
        for (int i = r-2; i != -1; --i){
            for (int j = c-2; j != -1; j--){
                if (m[i][j] == 1){
                    right[i][j] = right[i][j+1] + 1;
                    down[i][j] = down[i+1][j] + 1;
                }
            }
        }
    }
    

    bool hasSizeOfBorder(int size, vector<vector<int>>& right, vector<vector<int>>& down){
        for (int i = 0; i != right.size()-size+1; i++){
            for (int j = 0; j != right[0].size()-size+1; j++){
                if (right[i][j] >= size && down[i][j] >= size 
                   && right[i+size-1][j] >= size && down[i][j+size-1] >= size){
                    return true;
                }
            }
        }
        return false;
    }
    
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        vector<vector<int>> right(grid.size(), vector<int>(grid[0].size(), 0));
        vector<vector<int>> down(grid.size(), vector<int>(grid[0].size(), 0));
        
        setBorderMap(grid, right, down);

        for (int size = min(grid.size(), grid[0].size()); size != 0; size--){
            if (hasSizeOfBorder(size, right, down)){
                return size*size;
            }
        }
        return 0;
    }
};
