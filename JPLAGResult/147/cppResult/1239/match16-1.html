class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {

        vector<vector<int>> row(grid.size(),vector<int>(grid[0].size(),0));
        vector<vector<int>> col(grid[0].size(),vector<int>(grid.size(),0));
        for(int i=0;i<grid.size();i++){
            int cur=0;
            for(int j=0;j<grid[i].size();j++){

                cur+=grid[i][j];
                row[i][j]=cur;
            }
        }
        for(int i=0;i<grid[0].size();i++){
            int cur=0;
            for(int j=0;j<grid.size();j++){

                cur+=grid[j][i];
                col[i][j]=cur;
            }
        }
        cout<<"?"<<endl;
        int ans=0;
        for(int i1=0;i1<grid.size();i1++)
            for(int j1=0;j1<grid[0].size();j1++)
                for(int i2=i1;i2<grid.size();i2++)
                    for(int j2=j1;j2<grid[0].size();j2++){
                        //cout<<i1<<j1<<i2<<j2<<endl;
                        if(grid[i1][j1]!=1||grid[i1][j2]!=1||grid[i2][j1]!=1||grid[i2][j2]!=1)continue;
                        if(j2-j1!=i2-i1)continue;
                        //cout<<i1<<j1<<i2<<j2<<" "<<row[i2][j2]-row[i2][j1]<<row[i2][j2]-row[i2][j1]<<col[j1][i2]-col[j1][i1]<<col[j2][i2]-col[j2][i1]<<endl;
                        if(row[i1][j2]-row[i1][j1]==j2-j1&&row[i2][j2]-row[i2][j1]==j2-j1&&col[j1][i2]-col[j1][i1]==i2-i1&&col[j2][i2]-col[j2][i1]==i2-i1)ans=max(ans,(i2-i1+1)*(j2-j1+1));
                        //cout<<"!"<<endl;
                    }
        return ans;
    }
};
