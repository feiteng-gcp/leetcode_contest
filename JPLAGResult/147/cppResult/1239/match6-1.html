class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<int>> row(n, vector<int>(m+1, 0));
        vector<vector<int>> col(m, vector<int>(n+1, 0));

        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                row[i][j+1] = row[i][j]+grid[i][j];
            }
        }
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                col[i][j+1] = col[i][j]+grid[j][i];
            }
        }
        int ans = 0;

        for(int x = 0; x < n; x++) {
            for(int y = 0; y < m; y++) {
                for(int r = 1; x+r <= n && y+r <= m; r++) {
                    if(row[x][y+r]-row[x][y] == r && row[x+r-1][y+r]-row[x+r-1][y] == r
                      && col[y][x+r]-col[y][x] == r && col[y+r-1][x+r]-col[y+r-1][x] == r) {
                        ans = max(ans, r*r);
                    }
                }

            }
        }
        return ans;
    }
};
