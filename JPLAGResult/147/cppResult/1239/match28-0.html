class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        const int m = grid.size();
        const int n = m == 0? 0: grid[0].size();
        if(m == 0 || n == 0) return 0;
        
        int res = 0;
        vector<vector<int>> h(m, vector<int>(n)), v(m, vector<int>(n));
        if(grid[0][0] == 1)
        {
            h[0][0] = 1;
            v[0][0] = 1;
        }

        for (int i=0; i<m; i++)
        {
            for (int j=0; j<n; j++)
            {
                if (grid[i][j] == 0)
                {
                    v[i][j] = 0;
                    h[i][j] = 0;
                }
                else
                {
                    h[i][j] = (j==0)? 1: h[i][j-1] + 1;
                    v[i][j] = (i==0)? 1: v[i-1][j] + 1;
                }
            }
        }

        for (int i = m-1; i>=0; i--)
        {
            for (int j = n-1; j>=0; j--)
            { 
                int cur = min(h[i][j], v[i][j]); 

                while (cur*cur > res)
                {
                    if (v[i][j-cur+1] >= cur && h[i-cur+1][j] >= cur) res = cur*cur;
                    cur--;
                }
            }
        }
        return res;
    }
};
