class Solution {
public:
    vector<vector<int>> sums;

    int rect_sum(int r1, int r2, int c1, int c2) {
        return sums[r2][c2] - sums[r1][c2] - sums[r2][c1] + sums[r1][c1];
    }

    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int R = grid.size(), C = grid.empty() ? 0 : grid[0].size();
        sums.assign(R + 1, vector<int>(C + 1, 0));

        for (int r = 0; r < R; r++)
            for (int c = 0; c < C; c++)
                sums[r + 1][c + 1] = sums[r + 1][c] + sums[r][c + 1] - sums[r][c] + grid[r][c];

        bool all_zero = true;

        for (int r = 0; r < R; r++)
            for (int c = 0; c < C; c++)
                if (grid[r][c] == 1)
                    all_zero = false;

        if (all_zero)
            return 0;


        int best = 1;

        for (int r = 0; r < R; r++)
            for (int c = 0; c < C; c++)
                for (int s = best + 1; r + s <= R && c + s <= C; s++)
                    if (rect_sum(r, r + s, c, c + s) - rect_sum(r + 1, r + s - 1, c + 1, c + s - 1) == 4 * (s - 1))
                        best = s;

        return best * best;
    }
};
