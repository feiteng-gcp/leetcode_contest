class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();

        vector<vector<int>> right(n, vector<int>(m)), down(n, vector<int>(m));
        for(int i=0;i<n;i++) {
            int count = 0;
            for(int j=0;j<m;j++) {
                if(grid[i][j] == 0) right[i][j] = count = 0;
                else right[i][j] = ++count;

            }
        }
        for(int j=0;j<m;j++) {
            int count = 0;
            for(int i=0;i<n;i++) {
                if(grid[i][j] == 0) down[i][j] = count = 0;
                else down[i][j] = ++count;
            }
        }
        
        int ans = 0;

        for(int i=0;i<n;i++) for(int j=0;j<m;j++) {
            int m = min(right[i][j], down[i][j]);
            for(int k=m;k>ans;k--) {
                if(down[i][j-k+1] >= k && right[i-k+1][j] >= k) ans = k;
            }
        }
        return ans*ans;
    }
};
