class Solution {
public:
	int largest1BorderedSquare(vector<vector<int>>& grid) {
		int ret = 0;
		int h = grid.size();
		int w = grid[0].size();
		static int sw[100][100];
		static int sh[100][100];
        memset(sw,0,sizeof(sw));
        memset(sh,0,sizeof(sh));
        

		for (int y = 0; y < h; y++) {
			int c = 0;
			for (int x = w-1; x >= 0; x--) {
				if (grid[y][x]) c++;
				else c = 0;
				sw[y][x] = c;
			}
		}
		for (int x = 0; x < w; x++) {
			int c = 0;
			for (int y = h - 1; y >= 0; y--) {
				if (grid[y][x]) c++;
				else c = 0;
				sh[y][x] = c;
			}
		}


		for (int y = 0; y < h;y++) {
			for (int x = 0; x < w;x++) {
				if (grid[y][x] == 0)continue;
				int sub = 1;
				for (int len = 2; y + len <= h && x + len <= w; len++) {
					bool ok = true;

					ok &= sw[y][x] >= len;
					ok &= sw[y+len-1][x] >= len;

					ok &= sh[y][x] >= len;
					ok &= sh[y][x + len - 1] >= len;

					if (ok)sub = len;
				}
				ret = max(ret, sub * sub);
			}
		}
		
		return ret;
	}
};
