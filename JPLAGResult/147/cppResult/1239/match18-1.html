class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int row = grid.size();
        int col = grid[0].size();
        vector<vector<int>> hor(row, vector<int>(col));
        vector<vector<int>> ver(row, vector<int>(col));

        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (grid[i][j] == 0) 
                    ver[i][j] = hor[i][j] = 0; 
                else { 
                    hor[i][j] = (j == 0) ? 1 :  
                    hor[i][j - 1] + 1; 
                    ver[i][j] = (i == 0) ? 1 :  
                    ver[i - 1][j] + 1; 
                } 
            }
        }
        int max = 0;

        for (int i = row - 1; i >= 0; i--) { 
            for (int j = col - 1; j >= 0; j--) { 
                int small = min(hor[i][j], ver[i][j]);   
                while (small > max) { 
                    if (ver[i][j - small + 1] >= small && 
                        hor[i - small + 1][j] >= small) { 
                        max = small; 
                    } 
                    small--; 
                } 
            } 
        }
        return max * max;


    }
};
