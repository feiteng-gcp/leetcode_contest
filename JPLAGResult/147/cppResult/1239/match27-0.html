int a[105][105], b[105][105];

class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        memset(a, 0, sizeof a);
        memset(b, 0, sizeof b);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (j == 0) a[i][j] = grid[i][j];
                else if (grid[i][j] == 0) a[i][j] = 0;
                else a[i][j] = a[i][j-1] + 1;
                
                if (i == 0) b[i][j] = grid[i][j];
                else if (grid[i][j] == 0) b[i][j] = 0;
                else b[i][j] = b[i-1][j] + 1;
                
            }
        }
        
        for (int w = min(n,m); w; --w) {
            for (int i = w-1; i < n; ++i) {
                for (int j = w-1; j < m; ++j) {
                    if (a[i][j] < w || b[i][j] < w) continue;
                    int i0 = i-w+1;
                    int j0 = j-w+1;
                    if (a[i0][j] < w || b[i][j0] < w) continue;
                    return w*w;
                }
            }        
        }
        return 0;
    }
};
