class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int r = grid.size(), c = grid[0].size();
        vector<vector<int>> right(r, vector<int>(c)), down(r, vector<int>(c));

        for(int i = 0; i < r; i++) {
            for(int j = c-1; j >= 0; j--) {
                if(grid[i][j] == 0) {
                    right[i][j] = 0;
                } else {
                    right[i][j] = 1 + (j + 1 < c ? right[i][j+1] : 0);
                }
            }
        }
        for(int i = 0; i < c; i++) {
            for(int j = r-1; j >= 0; j--) {
                if(grid[j][i] == 0) {
                    down[j][i] = 0;
                } else {
                    down[j][i] = 1 + (j + 1 < r ? down[j+1][i] : 0);
                }
            }
        }
        int sidel = 0; // side length of the largest subgrid

        for(int i = 0; i < r; i++) {
            for(int j = 0; j < c; j++) {
                int x = min(right[i][j], down[i][j]);
                while(x > sidel) {
                    if(down[i][j+x-1] >= x && right[i+x-1][j] >= x) {
                        sidel = max(sidel, x);
                        break;
                    } else {
                        x--;
                    }
                }
            }
        }
        return sidel * sidel;
    }
};
