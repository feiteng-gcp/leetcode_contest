class Solution {
    typedef vector<vector<int>> Matrix;
    void build(Matrix & grid, Matrix & d1, Matrix & d2) {
        int m = grid.size(), n = grid[0].size();
        for (int i = 0; i < m; ++i) {
            int count = 0;
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 0) {
                    count = 0;
                } else {
                    count ++;
                }
                d1[i][j] = min(d1[i][j], count);
            }
        }
        for (int j = 0; j < n; ++j) {
            int count = 0;
            for (int i = 0; i < m; ++i) {
                if (grid[i][j] == 0) {
                    count = 0;
                } else {
                    count ++;
                }
                d1[i][j] = min(d1[i][j], count);
            }
        }
        for (int i = 0; i < m; ++i) {
            int count = 0;
            for (int j = n-1; j >= 0; --j) {
                if (grid[i][j] == 0) {
                    count = 0;
                } else {
                    count ++;
                }
                d2[i][j] = min(d2[i][j], count);
            }
        }
        for (int j = 0; j < n; ++j) {
            int count = 0;
            for (int i = m-1; i >= 0; --i) {
                if (grid[i][j] == 0) {
                    count = 0;
                } else {
                    count ++;
                }
                d2[i][j] = min(d2[i][j], count);
            }
        }
    }
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> d1(m, vector<int>(n, INT_MAX)), d2(m, vector<int>(n, INT_MAX)); 
        build(grid, d1, d2);
        int res = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                int len = d2[i][j];
                for (int k = len-1; k >= 0; --k) {
                    if (d1[i+k][j+k] >= k+1) {
                        res = max(res, (k+1)*(k+1));
                        break;
                    }
                }
            }
        }
        return res;
    }
};
