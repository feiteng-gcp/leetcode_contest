class Solution {
public:
    
    int getMin(int x, int y) { return (x<y)? x: y; } 
    
    int findSubSquare(vector<vector<int>>& mat) 
{ 
    int max = 0; 
    int m = mat.size();
    int n = mat[0].size();
  
    
    vector<vector<int>> hor(m,vector<int> (mat[0].size(),0));
    vector<vector<int>> ver(m,vector<int> (mat[0].size(),0));
    
        hor[0][0] = ver[0][0] = (mat[0][0] == 1); 
  
   
    for (int i=0; i<m; i++) 
    { 
        for (int j=0; j<n; j++) 
        { 
            if (mat[i][j] == 0) 
                ver[i][j] = hor[i][j] = 0; 
            else
            { 
                hor[i][j] = (j==0)? 1: hor[i][j-1] + 1; 
                ver[i][j] = (i==0)? 1: ver[i-1][j] + 1; 
            } 
        } 
    } 
        
   /*   for (int i=0; i<m; i++) 
    { 
        for (int j=0; j<n; j++) 
        { 
           cout << ver[i][j] << " ";
        } 
          cout << endl;
    }*/    
 
    for (int i = m-1; i>=0; i--) 
    { 
        for (int j = n-1; j>=0; j--) 
        { 
            
            int small = getMin(hor[i][j], ver[i][j]); 
  
            while (small > max) 
            { 
                if (ver[i][j-small+1] >= small && 
                    hor[i-small+1][j] >= small) 
                { 
                    max = small; 
                } 
                small--; 
            } 
        } 
    } 
    return max*max; 
} 
    
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        
        return findSubSquare(grid); 
    }
};
