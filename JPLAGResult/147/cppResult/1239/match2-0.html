class Solution {
public:
    bool helper(vector<vector<int>>& grid,vector<vector<int>> right,vector<vector<int>> down,int len){
        for (int i = 0; i < grid.size() - len +1; i++) {
			for (int j = 0; j < grid[0].size() - len + 1; j++) {
				if (right[i][j] >= len && down[i][j] >= len
						&& right[i + len - 1][j] >= len
						&& down[i][j + len - 1] >= len) {
					return true;
				}
			}
		}
		return false;
    }
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        int m = grid.size();
		int n = grid[0].size();
        vector<vector<int>> right(m,vector<int>(n,0));
        vector<vector<int>> down(m,vector<int>(n,0));
		if (grid[m - 1][n - 1] == 1) {
			right[m - 1][n - 1] = 1;
			down[m - 1][n - 1] = 1;
		}
		for (int i = m - 2; i >=0; i--) {
			if (grid[i][n - 1] == 1) {
				right[i][n - 1] = 1;
				down[i][n - 1] = down[i + 1][n - 1] + 1;
			}
		}
		for (int i = n - 2; i >=0; i--) {
			if (grid[m - 1][i] == 1) {
				right[m - 1][i] = right[m - 1][i + 1] + 1;
				down[m - 1][i] = 1;
			}
		}
		for (int i = m - 2; i >=0 ; i--) {
			for (int j = n - 2; j >=0; j--) {
				if (grid[i][j] == 1) {
					right[i][j] = right[i][j + 1] + 1;
					down[i][j] = down[i + 1][j] + 1;
				}
			}
		}
        int len=min(m,n);
        //从len开始一直试错
        for(int i=len;i>=1;i--){
            if(helper(grid,right,down,i)){
                return i*i;
            }
        }
        return 0;
        
    }
};
