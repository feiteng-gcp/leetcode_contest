class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& grid) {
      if (grid.empty()) {
        return 0;
      }
      int n = grid.size(), m = grid[0].size();

       vector<vector<int>> down(n, vector<int>(m)), right(n, vector<int>(m));
      for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
          if (grid[i][j]) {
            down[i][j] = right[i][j] = 1;
            if (i && down[i - 1][j]) {
              down[i][j] += down[i - 1][j];
            }
            if (j && right[i][j - 1]) {
              right[i][j] += right[i][j - 1];
            }
          }

        }
      }
      int ans = 0;
      for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
          for (int k = 1; i - k + 1 >= 0 && j - k + 1 >= 0; ++k) {
            if (down[i][j] >= k && right[i][j] >= k && down[i][j - k + 1] >= k && right[i - k + 1][j] >= k) {
              ans = max(ans, k);
            }
          }
        }
      }
      return ans * ans;
    }
};
