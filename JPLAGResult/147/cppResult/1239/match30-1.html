class Solution {
public:
    int row[105][105],col[105][105];
    int largest1BorderedSquare(vector<vector<int>>& grid) {
        memset(row,0,sizeof row);
        memset(col,0,sizeof col);

        int ans=0;
        for(int i=1;i<=grid.size();i++)
        {
            for(int j=1;j<=grid[i-1].size();j++)
            {
                if(j) row[i][j]=row[i][j-1];
                row[i][j]+=grid[i-1][j-1];
                if(i) col[i][j]=col[i-1][j];

                col[i][j]+=grid[i-1][j-1];
                for(int k=min(row[i][j],col[i][j]);k>=1;k--)
                {
                    if(row[i][j]-row[i][j-k]==k && row[i-k+1][j]-row[i-k+1][j-k]==k && col[i][j]-col[i-k][j]==k && col[i][j-k+1]-col[i-k][j-k+1]==k)
                    {
                        ans=max(ans,k*k);
                        break;
                    }
                }
            }
        }
        return ans;
    }
};
