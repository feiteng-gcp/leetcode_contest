class Solution {
public:
    int largest1BorderedSquare(vector<vector<int>>& g) {
        int n = g.size(), m = g[0].size();
        vector<vector<int>> dpl(n, vector<int>(m, 0)), dpu(n, vector<int>(m, 0));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (g[i][j]) {
                    dpl[i][j] = j ? dpl[i][j - 1] + 1 : 1;
                    dpu[i][j] = i ? dpu[i - 1][j] + 1 : 1;
                }
            }
        }
        
        for (int b = min(m, n); b; --b) {
            for (int row = b - 1; row < n; ++row) for (int col = b - 1; col < m; ++col) {
                if (dpl[row][col] >= b && dpu[row][col] >= b && dpl[row - b + 1][col] >= b && dpu[row][col - b + 1] >= b) {
                    return b * b;
                }
            }
        }
        
        return 0;
    }
};
