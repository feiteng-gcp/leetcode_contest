class Solution {
public:
  int stoneGameII(vector<int>& piles) {
    int n = piles.size();
    vector<int> s(n + 1);
    for (int i = 0; i < n; ++i) {
      s[i + 1] = s[i] + piles[i];
    }
    int inf = 1000010000;
    vector<vector<int>> a(n + 1, vector<int>(n + 1, -inf));
    function<int(int, int)> go = [&](int i, int j) -> int {
      int& ret = a[i][j];
      if (ret == -inf) {
        if (i < n) {
          ++ret;
          for (int x = 1; x <= j * 2 && i + x <= n; ++x) {
            ret = max(ret, s[i + x] - s[i] - go(i + x, max(j, x)));
          }
        } else {
          ret = 0;
        }
      }
      return ret;
    };
    return (go(0, 1) + s[n]) / 2;
  }
};
