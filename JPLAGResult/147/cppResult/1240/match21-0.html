int f[105][105], g[105][105], a[105], S[105];

class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n = piles.size();
        memset(f, 0, sizeof f);
        memset(g, 0x3f, sizeof g);
        
        for (int i = 1; i <= n; ++i) a[i] = piles[i-1], S[i] = S[i-1] + piles[i-1];
        for (int i = 1; i <= n; ++i) f[n][i] = a[n], g[n][i] = g[n+1][i] = 0;
        
        for (int i = n-1; i; --i) {
            for (int j = 1; j <= n; ++j) {
                for (int X = 1; X <= 2*j && i+X <= n+1; ++X) {
                    f[i][j] = max(f[i][j], S[i+X-1]-S[i-1]+g[i+X][max(X,j)]);
                    g[i][j] = min(g[i][j], f[i+X][max(X,j)]);
                    
                }
                //printf("f[%d][%d] = %d\n", i, j, f[i][j]);
                //printf("g[%d][%d] = %d\n", i, j, g[i][j]);
            }
        }
        
        return f[1][1];
    }
};
