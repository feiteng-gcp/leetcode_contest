class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        vector<int> totals(piles.size() + 1, 0);
        for (int i = piles.size() - 1; i >= 0; i--) {
            totals[i] = totals[i + 1] + piles[i];
        }
        vector<vector<int>> temp(piles.size() + 1, vector<int>(piles.size() + 1, -1));
        return helper(1, 0, piles, totals, temp);
    }

    int helper(int m, int start, vector<int> &piles, vector<int> &totals, vector<vector<int>> &temp) {
        if (temp[start][m] != -1)
            return temp[start][m];
        int res = 0;
        for (int x = 1; x <= m * 2 && x <= piles.size() - start; x++) {
            int gain = totals[start] - helper(min(max(m, x), (int)piles.size()), start + x, piles, totals, temp);
            res = max(res, gain);
        }
        temp[start][m] = res;
        return res;
    }
};
