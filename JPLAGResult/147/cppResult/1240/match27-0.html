using namespace std;

class Solution {
public:
	int N;
	int P[105];
	int dp[2][105][105];
	int f(bool alx, int l, int m) {
		if (l == N) return 0;
		if (dp[alx][l][m] != -1) return dp[alx][l][m];
		int best;
		if (alx) best = -1;
		else best = 1e9;
		int sum = 0;
		for (int x = 1; x <= 2 * m && l+x <= N; ++x) {
			sum += P[x + l - 1];
			if (alx) {
				best = max(best, f(false, l + x, min(N,max(m, x))) + sum);
			}
			else {
				best = min(best, f(true, l + x, min(N, max(m, x))));
			}
		}
		//cout << alx << " " << l << " " << m << " " << best << endl;
		return dp[alx][l][m] = best;
	}
	int stoneGameII(vector<int>& piles) {
		N = piles.size();
		for (int i = 0; i < piles.size(); ++i) P[i] = piles[i];
		memset(dp, -1, sizeof dp);
		return f(true, 0, 1);
	}
};
