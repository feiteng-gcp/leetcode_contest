class Solution {
public:

    int dp[105][205],s[105];
    int n;
    int dfs(int p,int m)
    {
	    if (p>n) return 0;
	    if (dp[p][m]!=-1) return dp[p][m];
	    int tot=1<<30;
	    for (int i=1;i<=m*2;i++)
	    {
    		if (p+i-1<=n) tot=min(tot,dfs(p+i,max(i,m)));else break;
	    }
	    return dp[p][m]=s[n]-s[p-1]-tot;
    }
    int stoneGameII(vector<int>& piles) {
        n=piles.size();
        for (int i=1;i<=n;i++) s[i]=s[i-1]+piles[i-1];
	    memset(dp,-1,sizeof(dp));
	    return dfs(1,1);
    }
};
