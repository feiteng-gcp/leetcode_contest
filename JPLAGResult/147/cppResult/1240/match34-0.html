class Solution {
public:
    int helper(vector<int>& piles, vector<unordered_map<int, int>>& v, int ps, int i, int m) {
        if (i >= ps) return 0;
        auto f = v[i].find(m);
        if (f != v[i].end()) return f->second;
        int ret = INT_MIN, sm = 0;
        int targ = min(ps, i + m * 2);
        for (int j = i; j < targ; j++) {
            sm += piles[j];
            ret = max(ret, sm - helper(piles, v, ps, j + 1, max(m, j - i + 1)));
        }
        v[i][m] = ret;
        return ret;
    }
    
    int stoneGameII(vector<int>& piles) {
        int ps = piles.size();
        vector<unordered_map<int, int>> v(ps, unordered_map<int, int>());
        int sm = 0;
        for (auto&& n: piles) sm += n;
        return (sm + helper(piles,  v, ps, 0, 1)) / 2;
    }
};
