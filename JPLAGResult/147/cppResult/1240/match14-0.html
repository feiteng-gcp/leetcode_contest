class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n = piles.size();
        prefix.push_back(0);
        for(auto pl : piles) {
            prefix.push_back(prefix.back() + pl);
        }
        // f(M, pos) [pos, n) is remaining
        int m = n;
        vector<vector<int>> tbl(m, vector<int>(n));
        for(int i = m-1; i >= 1; i--) {
            for(int j = n - 1; j >= 0; j--) {
                if(2*i >= n - j) {
                    tbl[i][j] = getSum(j, n-1); // the person can take all remaining stones
                } else {
                    for(int k = 1; k <= 2*i; k++) {
                        tbl[i][j] = max(tbl[i][j], getSum(j, n-1) - tbl[max(i, k)][j+k]);
                    }
                }
            }
        }
        return tbl[1][0];
    }
private:
    vector<int> prefix;
    int getSum(int i, int j) {
        return prefix[j+1] - prefix[i];
    }
};
