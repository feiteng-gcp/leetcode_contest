class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n = piles.size();
        vector<int> cum;
        for (int i = 0; i < n; ++i) {
            if (i == 0) cum.push_back(piles[i]);
            else cum.push_back(piles[i] + cum[i - 1]);
        }
        vector<vector<int> > dp(n + 1, vector<int>(n + 1, 0));
        for (int i = n - 1; i >= 0; --i) {
            for (int M = 0; M <= n; ++M) {
                int cumsum = cum[n - 1];
                if (i > 0) cumsum -= cum[i - 1];
                // Get minimum of dp[i + X][max(X, M)]
                int minsofar = cum[n - 1];
                for (int X = 1; i + X <= n && X <= 2 * M; ++X) {
                    minsofar = min(minsofar, dp[i + X][max(X, M)]);
                }
                dp[i][M] = cumsum - minsofar;
            }
        }
        return dp[0][1];
    }
};
