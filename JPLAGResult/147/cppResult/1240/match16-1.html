class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int N = piles.size();
        int M = 201;
        // res[i][j] is the max stones could get when start at pile[i] with M = j
        vector<vector<int>> res(N, vector<int>(M, 0));
        vector<int> sums(N + 1, 0);
        for(int i = 0; i < N; ++i)
        {
            sums[i+1] = sums[i] + piles[i];
        }
        
        for(int i = N - 1; i >= 0; --i)
        {
            for(int m = 1; m < M; ++m)
            {
                if(N - i <= 2 * m)
                {
                    res[i][m] = sums[N] - sums[i];
                }
                else
                {
                    // try X from 1 to 2m
                    for(int x = 1; x <= 2*m; ++x)
                    {
                        res[i][m] = max(res[i][m], sums[N] - sums[i] - res[i + x][max(x, m)]);
                    }
                }
            }
        }
        
        return res[0][1];
    }
};
