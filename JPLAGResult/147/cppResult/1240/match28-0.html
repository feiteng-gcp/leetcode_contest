class Solution {
    vector<vector<int>> dp;
    vector<int> pre;
    
public:
    int stoneGameII(vector<int>& piles) {
        const int n = piles.size();
        
        dp = vector<vector<int>>(n, vector<int>(n, -1));
        pre = vector<int>(n, 0);
        
        for(int i = n-1; i >= 0; i --)
        {
            pre[i] = piles[i];
            if(i < n-1) pre[i]+=pre[i+1];
        }
        pre.push_back(0);
        
        solve(piles, 0, 1);
        
        return dp[0][1];
    }
    
    int solve(vector<int>& piles, int st, int m)
    {
        if(st >= piles.size()) return 0;
        if(dp[st][m] != -1) return dp[st][m];
        
        dp[st][m] = 0;
        
        for(int i = 1; i <= 2*m && st+i <= piles.size(); i ++)
        {
            dp[st][m] = max(dp[st][m], pre[st]-solve(piles, st+i, max(m, i)));
        }
        
        return dp[st][m];
    }
};
