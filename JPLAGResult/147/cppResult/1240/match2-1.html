class Solution {
public:
    int suf[110],dp[110][210],n;
    int d(int x,int m){
        if(dp[x][m]!=-1)return dp[x][m];
        if(x==n)return suf[n];
        if(x>n)return 0;
        int ans=0;
        for(int r,t=1;t<=m*2;t++){
            r=x+t-1;
            if(r>n)break;
            ans=max(ans,suf[x]-d(r+1,max(m,t)));

        }
        return dp[x][m]=ans;
    }
    int stoneGameII(vector<int>& piles) {
        n=piles.size();
        for(int i=n;i>=1;i--)suf[i]=suf[i+1]+piles[i-1];
        memset(dp,-1,sizeof(dp));
        return d(1,1);
    }
};
