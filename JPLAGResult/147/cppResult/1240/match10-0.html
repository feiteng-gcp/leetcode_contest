class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int N = piles.size();
        vector<int> sum(N + 1, 0);
        for (int i = N - 1; i >= 0; --i)
            sum[i] = sum[i + 1] + piles[i];
        vector<vector<int>> f(N + 1, vector<int>(N + 1, 0));
        for (int i = N - 1; i >= 0; --i) {
            for (int j = 1; j <= N; ++j) {
                f[i][j] = 0;
                for (int k = 1; k <= 2 * j && i + k <= N; ++k) {
                    f[i][j] = max(f[i][j], sum[i] - f[i + k][max(j, k)]);
                }
            }
        }
        return f[0][1];
    }
};
