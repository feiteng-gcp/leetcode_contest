class Solution {
public:
    int dfs(int start, int m, int player, vector<int>& piles, vector<vector<int>>& f) {
        if (start>=piles.size()) return 0;
        if (f[start][m]>=0) return f[start][m];
        f[start][m]=0;
        for (int i=1; i<=m+m && start+i<=piles.size(); i++) {
            int tmp=dfs(start+i, max(m, i), 1, piles, f);
            f[start][m]=max(f[start][m], piles[start]-tmp);
        }
        return f[start][m];
    }
    int stoneGameII(vector<int>& piles) {
        for (int i=piles.size()-2; ~i; i--) piles[i]+=piles[i+1];
        vector<vector<int>> f(piles.size(), vector<int>(piles.size(), -1));
        return dfs(0, 1, 0, piles, f);
    }
};
