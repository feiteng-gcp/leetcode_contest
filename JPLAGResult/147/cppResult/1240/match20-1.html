class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int sum=0;
        for(auto &i:piles)sum+=i;
        //cout<<helper(0,1,0,piles)<<endl;
        dp.assign(piles.size(),vector<vector<int>>(piles.size(),vector<int>(2,INT_MIN)));
        return sum-(sum-helper(0,1,0,piles))/2;
    }
    

    int helper(int p,int m,int turn,vector<int>& piles){
        if(p==piles.size())return 0;
        if(dp[p][m][turn]!=INT_MIN)return dp[p][m][turn];
        int ans=INT_MIN;
        int cur=0;
        for(int x=1;x<=2*m;x++){
            if(p+x-1>=piles.size())break;
            cur+=piles[p+x-1];
            int ta=-helper(p+x,max(m,x),1-turn,piles);
            ans=max(ans,cur+ta);
        }
        return dp[p][m][turn]=ans;
    }
    
    vector<vector<vector<int>>> dp;
};
