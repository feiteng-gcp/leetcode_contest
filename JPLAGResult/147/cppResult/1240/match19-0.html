int n;
vector<int> a;
int dp[105][105];

class Solution {
public:
    int dfs(int p, int m) {
        if(p >= n) return 0;
        if(dp[p][m] != -1) {
            return dp[p][m];
        }
        int s = 0, maxx = 0;
        for(int i = p; i < n; i++) s += a[i];
        for(int i = p; i < min(n, p + 2 * m); i++) {
            maxx = max(maxx, s - dfs(i + 1, max(m, i - p + 1)));
        }
        dp[p][m] = maxx;
        return maxx;
    }
    int stoneGameII(vector<int>& piles) {
        a = piles;
        n = a.size();
        for(int i = 0; i < n; i++) {
            for(int j = 0; j <= n; j++) {
                dp[i][j] = -1;
            }
        }
        dfs(0, 1);
        return dp[0][1];
    }
};
