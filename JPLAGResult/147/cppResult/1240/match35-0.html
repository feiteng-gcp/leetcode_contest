class Solution {
public:
    vector<vector<int>> memo;
    
    int stoneGameII(vector<int>& piles) {
        memo.resize(101, vector<int>(101));
        return helper(piles, 0, 1);
    }
    
    int helper(vector<int> &piles, int start, int m) {
        if (memo[start][m] > 0) return memo[start][m];
        int stones = 0, rem = piles.size()-start;
        int total = accumulate(piles.begin() + start, piles.end(), 0);
        for (int x = 1; x <= min(m*2, rem); x++) {
            stones = max(stones, total - helper(piles, start + x, max(x, m)));
        }
        return memo[start][m] = stones;
    }
};
