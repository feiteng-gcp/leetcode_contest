class Solution {
public:
    vector<int> cum;
    unordered_map<int, unordered_map<int, int>> memo;
    
    int stoneGameII(vector<int>& piles) {
        cum.assign(piles.size() + 1, 0);

        for (int i = 0; i < piles.size(); i++) {
            cum[i + 1] = cum[i] + piles[i];
        }
        return helper(piles, 0, 1);
    }
    
    int helper(vector<int>& piles, int start, int m) {
        int n = piles.size();
        
        if (start >= n) return 0;
        

        if (memo.count(start) && memo[start].count(m)) {
            return memo[start][m];
        }
        
        int ans = 0;
        for (int i = start; i < start + m * 2 && i < n; i++) {
            int v = helper(piles, i + 1, max(m, i - start + 1));
            ans = max(ans, cum[n] - cum[start] - v);
        }
        
        memo[start][m] = ans;
        
        return ans;
    }
};
