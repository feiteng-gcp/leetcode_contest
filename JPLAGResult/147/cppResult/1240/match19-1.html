const int MAXN = 110;
const int INF = 1000000000;

int n;
int a[MAXN];
int F[MAXN][MAXN];

class Solution {
public:
    int stoneGameII(vector<int>& piles) {

        int n = piles.size();
        a[0] = 0;
        for (int i = 0; i < n; ++ i)
            a[i+1] = piles[i];
        for (int i = 1; i <= n; ++ i)
            a[i] += a[i-1];

        for (int i = 1; i <= n; ++ i)
            F[n+1][i] = 0;
        for (int i = n; i >= 1; -- i)
        {
            for (int x = 1; x <= i; ++ x)
            {
                F[i][x] = -INF;
                for (int y = 1; y <= 2*x && i+y-1 <= n; ++ y)
                {
                    F[i][x] = max(F[i][x], a[i+y-1]-a[i-1]-F[i+y][max(x, y)]);
                }
            }
        }
        return (a[n]+F[1][1])/2;
    }
};
