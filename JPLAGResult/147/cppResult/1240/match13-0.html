class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n = piles.size();
        vector<int>  sum(n+1);
        vector<vector<int>> dp(n, vector<int> (n+1, 0));
        for(int m = 1;m<=n;m++){
            dp[n-1][m] = piles.back();
        }
        sum[n-1] = piles.back();
        for(int i = n-1;i>=0;i--){
            sum[i] = sum[i+1] + piles[i];
            for(int m = 1;m<=n;m++){
                int tmp = 0;
                if(2*m >= n-i){
                    dp[i][m] = sum[i];
                }
                else{
                    for(int k = 1;k<=2*m;k++){
                        tmp = max(tmp, sum[i] - dp[i+k][max(k, m)]);
                        
                    }
                    dp[i][m] = tmp;
                }
            }
        }
        return dp[0][1];
    }
};
