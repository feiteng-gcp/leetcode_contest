class Solution {
public:
    int stoneGameII(vector<int>& a) {
        int n = a.size();
        vector<vector<vector<int>>> f(n+2, vector<vector<int>>(n+2, vector<int>(n+2)));
        vector<int> sum = a;
        for (int i = 1; i < n; ++i) {
            sum[i] += sum[i-1];
        }
        

        for (int i = 0; i < n; ++i)
            f[i][i][1] = a[i];
        
        for (int len = 2; len <= n; ++len) {
            for (int st = 0; st <= n - len; ++st) {
                int en = st + len - 1;
                
                for (int m = 1; m <= len; ++m) {
                    int lim = min(m * 2, len);
                    int cur = INT_MIN;
                    for (int k = 1; k <= lim; ++k) {
                        cur = max(cur, -f[st+k][en][min(en-st-k+1, max(k, m))]);
                    }
                    f[st][en][m] = cur + sum[en];
                    if (st > 0)
                        f[st][en][m] -= sum[st-1];
                    
                    // printf("%d %d %d: %d\n", st,en, m, f[st][en][m]);
                }
            }
        }
        
        return f[0][n-1][1];
    }
};
