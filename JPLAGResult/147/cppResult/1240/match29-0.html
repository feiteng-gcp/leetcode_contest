class Solution {
public:
    
    vector<int> v;
    int mem[102][102];
    
    int solve(int pos, int last) {
        if (mem[pos][last] != -1) return mem[pos][last];
        
        int rem = 0;
        for (int i = pos; i < v.size(); ++i)
            rem += v[i];
        
        int ret = 0, sum = 0;
        for (int k = 1; k <= 2 * last && pos + k - 1 < v.size(); ++k) {
            sum += v[pos + k - 1];
            
            ret = max(ret, rem - solve(pos + k, max(k, last)));
        }
        return mem[pos][last] = ret;
    }
    
    int stoneGameII(vector<int>& piles) {
        memset(mem, -1, sizeof(mem));
        v = piles;
        return solve(0, 1);
    }
};
