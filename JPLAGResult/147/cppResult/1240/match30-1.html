class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n=piles.size();
        vector<int> sum(n+1,0);
        for(int i=1;i<=n;i++)
            sum[i]=sum[i-1]+piles[i-1];
        vector<vector<int>> dp(n,vector<int>(n+1,0));
        dp[n-1][1]=piles.back();
        for(int i=n-2;i>=0;i--)
        {
            int left=n-i;
            int leftSum=sum.back()-sum[i];
            for(int m=1;;m++)
            {
                if(2*m>=left) 
                {
                    dp[i][m]=leftSum;
                    break;
                }
                for(int x=1;x<=2*m;x++)
                {
                    int nextM=max(m,x);
                    int nextSum=sum.back()-sum[i+x];
                    if(nextM*2+i+x<n)
                    {
                        nextSum=dp[i+x][nextM];
                    }
                    dp[i][m]=max(dp[i][m],leftSum-nextSum);
                }
            }
        }
        return dp[0][1];
    }
};
