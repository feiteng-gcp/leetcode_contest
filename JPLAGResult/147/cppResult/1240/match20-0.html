class Solution {
public:
    vector< vector<int> > dp;
    
    int stoneGameII(vector<int>& piles) {
        int n = piles.size();
        
        dp = vector< vector<int> >(n, vector<int>(n * 5, -1));
        
        return AlexTurn(piles, 0, 1);
    }
    
    int AlexTurn(vector<int>& piles, int pos, int M)
    {
        if (pos >= piles.size()) return 0;
        if (dp[pos][M] != -1) return dp[pos][M];
        int res = 0;
        int sum = 0;
        for (int x = 1; x <= 2 * M; ++x)
        {
            if (pos + x - 1 >= piles.size()) break;
            
            sum += piles[pos + x - 1];
            int tempRes = LeeTurn(piles, pos + x, max(M, x));
            
            res  = max(res, sum + tempRes);
        }
        
        dp[pos][M] = res;
        return res;
    }
    
    int LeeTurn(vector<int>& piles, int pos, int M)
    {
        if (pos >= piles.size()) return 0;
        int res = INT_MAX;
        int sum = 0;
        for (int x = 1; x <= 2 * M; ++x)
        {
            if (pos + x - 1 >= piles.size()) break;
            
            sum     += piles[pos + x - 1];
            int tempRes  = AlexTurn(piles, pos + x, max(M, x));
            
            res      = min(res, tempRes);
        }
        
        return res;
    }
};
