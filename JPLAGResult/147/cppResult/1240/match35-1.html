class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        auto sum = piles;
        for (int i = sum.size() - 2; i >= 0; i--) {
            sum[i] = sum[i + 1] + piles[i];
        }
        vector<int> h[piles.size()];
        for (int i = 0; i < piles.size(); i++) {
            h[i].resize(piles.size());
        }
        return dp(h, piles, sum, 0, 1);
    }
    
    int dp(vector<int> h[], const vector<int>& piles, const vector<int>& sum, int l, int m) {
        if (piles.size() - l <= m * 2) {
            return sum[l];
        }
        
        if (h[l][m]) {
            return h[l][m];
        }
        

        int ans = 0;
        for (int i = 0; i < m * 2; i++) {
            ans = max(ans, sum[l] - dp(h, piles, sum, l + i + 1, max(m, i + 1)));
        }
             
        return h[l][m] = ans;
    }
};
