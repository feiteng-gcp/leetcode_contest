class Solution {
public:
    static const int maxn = 2e2;
    int dp[2][maxn][maxn];
    int sum[maxn];
    int stoneGameII(vector<int>& piles) {        
        int n = piles.size();
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                dp[0][i][j] = 0;
                dp[1][i][j] = 0x3f3f3f3f;
            }
        }
        for(int i = 0; i < n; i++) {
            sum[i+1] = sum[i]+piles[i];
        }

        for(int i = n-1; i >= 0; i--) {
            for(int j = 1; j <= n; j++) {
                for(int k = 1; k <= 2*j && i+k <= n; k++) {
                    dp[0][i][j] = max(dp[0][i][j], dp[1][i+k][max(j, k)]+(sum[i+k]-sum[i]));
                    dp[1][i][j] = min(dp[1][i][j], dp[0][i+k][max(j, k)]);
                }
            }
        }
        return dp[0][0][1];
    }
};
