class Solution {
public:
    int a1=0;
    int a2=0;
    int n;
    int sum[105];
    int d[105][130];
    int dps(int i,int m,vector<int> p){
        if(i>=n) return 0;
        if(m*2>n-i) return sum[i];
        int tmp=sum[i];
        for(int j=m*2-1;j>=0;j--){
            int k;
            if(d[i+j+1][max(m,j+1)]!=INT_MAX)
                k=d[i+j+1][max(m,j+1)];
            else {k=dps(i+j+1,max(m,j+1),p);d[i+j+1][max(m,j+1)]=k;}
            tmp=min(tmp,k);
        }
        return sum[i]-tmp;
    }
    int stoneGameII(vector<int>& piles) {
        n=piles.size();
        sum[n]=0;
        for(int i=n-1;i>=0;i--)
            sum[i]=sum[i+1]+piles[i];
        for(int i=0;i<101;i++)
            for(int j=0;j<130;j++)
                d[i][j]=INT_MAX;
        return dps(0,1,piles);
    }
};
