class Solution {
public:
    int S[105], dp[105][205];
    int stoneGameII(vector<int>& piles) {
        S[0] = 0;
        int N = piles.size();
        for (int i=1;i<=N;++i) S[i] = S[i-1] + piles[i-1];
        memset(dp,0,sizeof(dp));

        for (int x=N;x>=1;--x)
            for (int m=1;m<=N;++m) {
                for (int k=1;k<=2*m;++k) {
                    int y = min(N, x + k - 1);
                    int s = S[y] - S[x-1];
                    int rem = S[N] - S[y];
                    dp[x][m] = max(dp[x][m], s + rem - dp[y+1][max(k,m)]);
                }
               // printf("dp[%d][%d] = %d\n", x, m, dp[x][m]);
            }
        return dp[1][1];        
    }
};
