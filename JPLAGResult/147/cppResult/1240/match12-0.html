class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n = piles.size();
        vector<vector<int>> dp(n + 1, vector<int>(n + 1, INT_MAX));
        vector<int> sums(n + 1, 0);
        for (int i = n - 1; i >= 0; --i) {
            sums[i] = sums[i+1] + piles[i];
        }
        return f(dp, sums, 1, 0);
    }
    
    int f(vector<vector<int>>& dp, vector<int>& sums, int m, int p) {
        int n = dp.size() - 1;
        if (p == n) {
            return 0;
        }
        if (dp[m][p] != INT_MAX) {
            return dp[m][p];
        }
        int y = 0;
        for (int x = 1; x <= 2*m && p + x <= n; ++x) {
            int r = f(dp, sums, max(m, x), p + x);
            y = max(y, sums[p] - r);
        }
        return dp[m][p] = y;
    }
};
