class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n=piles.size();
        vector<int> sums(n+1,0);
        for(int i=n-1;i>=0;i--){
            sums[i]=piles[i]+sums[i+1];
        }
        vector<vector<int>> dp(n,vector<int>(n+1,0));
        for(int i=n-1;i>=0;i--)
            for(int j=1;j<=n;j++){
                if(2*j>=n-i) dp[i][j]=sums[i];
                else{
                    for(int k=1;k<=2*j;k++){
                        int nextM=max(k,j);
                        dp[i][j] = max(dp[i][j],sums[i]-dp[i+k][nextM]);
                    }
                }
            }
        return dp[0][1];
    }
};
