class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n = piles.size();
        unordered_map<int, unordered_map<int, int>> f;
        unordered_map<int, int> sum;
        for (int i = n - 1; i >= 0; i --) sum[i] = sum[i + 1] + piles[i];
        for (int i = n - 1; i >= 0; i --)
            for (int j = 1; j <= n - i; j ++) {
                for (int k = 1; k <= j; k ++) {
                    int ii = i + k, jj = max(j, k * 2);
                    int pp;
                    if (ii + jj >= n) pp = sum[ii];
                    else pp = f[ii][jj];
                    f[i][j] = max(f[i][j], sum[ii] - pp + sum[i] - sum[ii]);
                }
            }
        return f[0][2];
    }
};
