class Solution {
public:
    int stoneGameII(vector<int>& piles) {
        int n=piles.size();
        //M<100
        vector<int> presum(n+1,0);
        for(int i=1;i<=n;i++){
            presum[i]=presum[i-1]+piles[i-1];
        }
        int M=1;
        //from the end;
        
        vector<vector<int>> dp1(n+1, vector<int>(200,-1)), dp2(n+1, vector<int>(200,INT_MAX));
        for(int i=0;i<200;i++){
            dp1[n][i]=0;
            dp2[n][i]=0;
        }
        for(int i=n-1;i>=0;i--){
            for(int m=1;m<100;m++){
                for(int x=1;x<=m*2 && x+i<=n;x++){
                    dp1[i][m]=max(dp1[i][m], presum[x+i]-presum[i] + dp2[i+x][max(m,x)]);
                }
                for(int x=1;x<=m*2 && x+i<=n;x++){
                    dp2[i][m]=min(dp2[i][m], dp1[i+x][max(m,x)]);
                }
            }
        }
        return dp1[0][1];
