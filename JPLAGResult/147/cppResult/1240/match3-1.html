class Solution {
public:
    int dp[102][102];
    vector<int> p;
    int go(int start, int m, int totalLeft) {
        if (start >= p.size()) return 0;
        if (dp[start][m] != -1) return dp[start][m];

        int s = 0;
        for (int i = 1; i <= 2 * m && start + i <= p.size(); ++i)
        {
            s += p[start + i - 1];
            dp[start][m] = max(dp[start][m], totalLeft - go(start + i, max(m, i), totalLeft - s));
        }
        return dp[start][m];
    }
    int stoneGameII(vector<int>& piles) {
        memset(dp, -1, sizeof(dp));
        p = piles;
        int total = 0;
        for (auto v : p) total += v;
        return go(0, 1, total);
    }
};
