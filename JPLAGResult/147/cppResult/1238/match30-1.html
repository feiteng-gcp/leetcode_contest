class Solution {
public:
    string alphabetBoardPath(string target) {
        vector<string> board = {"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"};
        unordered_map<char,vector<int>> M;
        for(int i=0;i<board.size();i++)
            for(int j=0;j<board[i].size();j++)
                M[board[i][j]]={i,j};
        int x=0,y=0;
        string res="";
        for(char c:target)
        {
            if(c!='z' && board[x][y]=='z')
            {
                res+='U';
                x--;
            }
            int tx=M[c][0];

            int ty=M[c][1];
            while(ty<y)
            {
                res+='L';
                y--;
            }
            while(ty>y)
            {
                res+='R';
                y++;
            }
            while(tx<x)
            {
                res+='U';
                x--;
            }
            while(tx>x)
            {
                res+='D';
                x++;
            }

            res+='!';
        }
        return res;
    }
};
