class Solution {
public:
    //abcde
    //fghij
    //klmno
    //pqrst
    //uvwxy
    //z
    pair<int, int> get(char c) {
        int v = c - 'a';
        return {v / 5, v % 5};
    };
    string alphabetBoardPath(string target) {
        string ans;
        pair<int, int> curr{0, 0};
        for(char c: target) {
            pair<int, int> next = get(c);
            pair<int, int> diff{next.first - curr.first, next.second - curr.second};
            
            if(next.first == 5 && next.second == 0) {
                for(int j = 0; j < abs(diff.second); ++j) {
                    if(0 < diff.second) {
                        ans += "R";
                    } else {
                        ans += "L";
                    }
                };
                for(int i = 0; i < abs(diff.first); ++i) {
                    if(0 < diff.first) {
                        ans += "D";
                    } else {
                        ans += "U";
                    }
                };
            } else {
                for(int i = 0; i < abs(diff.first); ++i) {
                    if(0 < diff.first) {
                        ans += "D";
                    } else {
                        ans += "U";
                    }
                };
                for(int j = 0; j < abs(diff.second); ++j) {
                    if(0 < diff.second) {
                        ans += "R";
                    } else {
                        ans += "L";
                    }
                };
            }
            
            
            ans += "!";
            curr = next;
        };
        return ans;
    }
};
