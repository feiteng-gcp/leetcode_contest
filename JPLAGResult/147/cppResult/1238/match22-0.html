class Solution {
    
public:
    string alphabetBoardPath(string target) {
        string board[10] = {"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"};
        string moves = "";
        int x = 0, y = 0;
        for (int i = 0; i < target.length(); ++i) {
            int nx = 0, ny = 0;
            for (int r = 0; r < 6; ++r) {
                for (int c = 0; c < board[r].length(); ++c) {
                    if (board[r][c] == target[i]) {
                        nx = r, ny = c;
                    }
                }
            }
            if (x < 5) {
            if (ny > y) {
                for (int j = 0; j < ny - y; ++j) moves += 'R';
            } else {
                for (int j = 0; j < y - ny; ++j) moves += 'L';
            }
            if (nx > x) {
                for (int j = 0; j < nx - x; ++j) moves += 'D';
            } else {
                for (int j = 0; j < x - nx; ++j) moves += 'U';
            }
            } else {
                if (nx > x) {
                for (int j = 0; j < nx - x; ++j) moves += 'D';
            } else {
                for (int j = 0; j < x - nx; ++j) moves += 'U';
            }
                if (ny > y) {
                for (int j = 0; j < ny - y; ++j) moves += 'R';
            } else {
                for (int j = 0; j < y - ny; ++j) moves += 'L';
            }
            }
            
            moves += '!';
            x = nx, y = ny;
        }
        return moves;
    }
};
