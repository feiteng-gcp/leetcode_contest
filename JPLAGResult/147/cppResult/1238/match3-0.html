class Solution {
public:
    
    string move(char prev, char c, int &x, int &y){
        int p=c-'a';
        int gx=p/5, gy=p%5;
        string ret="";
        if (prev=='z'){
            while(x<gx) ++x, ret+="D";
            while(x>gx) --x, ret+="U";
            while(y<gy) ++y, ret+="R";
            while(y>gy) --y, ret+="L";
        }else{
            while(y<gy) ++y, ret+="R";
            while(y>gy) --y, ret+="L";
            while(x<gx) ++x, ret+="D";
            while(x>gx) --x, ret+="U";
        }
        return ret;
    }
    
    string alphabetBoardPath(string target) {
        int x=0, y=0, n=target.length();
        char prev='#';
        string ans="";
        for (int i=0; i<n; i++){
            char c=target[i];
            ans+=move(prev, c, x, y);
            prev=c;
            ans+="!";
        }
        return ans;
    }
};
