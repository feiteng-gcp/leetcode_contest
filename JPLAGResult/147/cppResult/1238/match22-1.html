class Solution {
public:
    string alphabetBoardPath(string target) {
        unordered_map<char, int> r, c;
        for (int i = 'a'; i <= 'e'; ++i) {
            r[i] = 0;
            c[i] = i - 'a';
        }
        for (int i = 'f'; i <= 'j'; ++i) {
            r[i] = 1;
            c[i] = i - 'f';
        }
        for (int i = 'k'; i <= 'o'; ++i) {
            r[i] = 2;
            c[i] = i - 'k';
        }
        for (int i = 'p'; i <= 't'; ++i) {
            r[i] = 3;
            c[i] = i - 'p';
        }
        for (int i = 'u'; i <= 'y'; ++i) {
            r[i] = 4;
            c[i] = i - 'u';
        }
        r['z'] = 5; c['z'] = 0;
        int x = 0, y = 0;
        string ans = "";
        for (auto w: target) {

            if (w == 'z') {
                if (c[w] < y) {
                    for (int i = c[w]; i < y; ++i) ans += 'L';
                } else {
                    for (int i = c[w]; i > y; --i) ans += 'R';
                }
                if (r[w] < x) {
                    for (int i = r[w]; i < x; ++i) ans += 'U';
                } else {
                    for (int i = r[w]; i > x; --i) ans += 'D';
                }
            } else {
                if (r[w] < x) {
                    for (int i = r[w]; i < x; ++i) ans += 'U';
                } else {
                    for (int i = r[w]; i > x; --i) ans += 'D';
                }
                if (c[w] < y) {
                    for (int i = c[w]; i < y; ++i) ans += 'L';
                } else {
                    for (int i = c[w]; i > y; --i) ans += 'R';
                }
            }
            ans += '!';
            x = r[w];
            y = c[w];
        }
        return ans;
    }
};
