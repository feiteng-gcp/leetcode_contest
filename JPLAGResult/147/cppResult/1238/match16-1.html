class Solution {
public:
    string alphabetBoardPath(string target) {
        vector<string> board_str = {"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"};

        map<char, pair<int, int>> axis;
        
        for (int i = 0; i < board_str.size(); ++i) {
            for (int j = 0; j < board_str[i].size(); ++j) {
                axis[board_str[i][j]] = {i, j};
            }
        }    
        string path = "";
        int x = 0, y = 0;
        
        for (auto c:target) {
            int nx = axis[c].first, ny = axis[c].second;
            if (y == 0 && x == 5) {
                while (nx > x) {
                    x++;
                    path += 'D';
                }
                while (nx < x) {
                    x--;
                    path += 'U';
                }
            }

            while (ny > y) {
                y++;
                path += 'R';
            }
            while (ny < y) {
                y--;
                path += 'L';
            }
            while (nx > x) {
                x++;
                path += 'D';
            }
            while (nx < x) {
                x--;
                path += 'U';
            }
            path += '!';
        }
        return path;
    }
};
