class Solution {
public:
    string alphabetBoardPath(string target) {
        string ans="";
        

        vector<pair<int,int>> arr;
        for(int i=0; i<6; i++) {
            for(int j=0; j<5; j++) {
                arr.push_back(make_pair(i,j));
                if(i==5 && j==0) break;
            }
        }
        
        int pos1=0, pos2=0;
        for(int i=0; i<target.length(); i++) {
            pair<int,int> pos = arr[target[i] - 'a'];
            

            while(pos1 > pos.first) {
                ans+='U';
                pos1--;
            }
            while(pos2 > pos.second) {
                ans+='L';
                pos2--;
            }
            while(pos1 < pos.first) {
                ans+='D';
                pos1++;
            }
            
            while(pos2 < pos.second) {
                ans+='R';
                pos2++;
            }
            ans+='!';
        }
        return ans;
    }
};
