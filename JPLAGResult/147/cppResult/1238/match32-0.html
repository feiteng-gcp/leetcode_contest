#define ll long long
class Solution {
public:
    string alphabetBoardPath(string target) {
     
        string board[10];
        board[0]="abcde";board[1]="fghij";board[2]="klmno";board[3]="pqrst";board[4]="uvwxy";board[5]="z";
        string ans="";
        ll i,j,k;
        ll l=target.length();
        ll curx=0,cury=0;
        for(i=0;i<l;i++)
        {
            ll posx=-1,posy=-1;
            for(j=0;j<=4;j++)
            {
                for(k=0;k<=4;k++)
                    if(board[j][k]==target[i])
                    {
                        posy=j;
                        posx=k;
                        break;
                    }
                  if(k<=4)
                      break;
            }
            if(posx==-1)
            {
                posy=5;
                posx=0;
            }
           // cout<<posy<<" "<<posx<<"\n";
            while(cury<posy && cury<4)
            {
                ans+='D';
                cury++;
            }
            while(cury>posy)
            {
                ans+='U';
                cury--;
            }
            while(curx<posx)
            {
                ans+='R';
                curx++;
            }
            while(curx>posx)
            {
                ans+='L';
                curx--;
            }
            while(cury<posy)
            {
                ans+='D';
                cury++;
            }
            ans+='!';
        }
        return ans;
    }
};
