class Solution {
public:
    string alphabetBoardPath(string target) {
        int x = 0, y = 0;
        string str = "";
        for (int i = 0; i < target.size(); i++)
        {
            int word = target[i]-'a';
            int x1 = word / 5;
            int y1 = word % 5;
            if (word != 25) {
                if (x < x1)
                    for (int j = x; j < x1; j++)
                        str += "D";
                else
                    for (int j = x; j > x1; j--)
                        str += "U";
                if (y < y1)
                    for (int j = y; j < y1; j++)
                        str += "R";
                else
                    for (int j = y; j > y1; j--)
                        str += "L";
            }
            else
            {
                if (y < y1)
                    for (int j = y; j < y1; j++)
                        str += "R";
                else
                    for (int j = y; j > y1; j--)
                        str += "L";
                if (x < x1)
                    for (int j = x; j < x1; j++)
                        str += "D";
                else
                    for (int j = x; j > x1; j--)
                        str += "U";
            }
            str+="!";
            x = x1;
            y = y1;
        }
        return str;
    }
};
