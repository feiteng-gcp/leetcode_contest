class Solution {
public:
    string alphabetBoardPath(string target) {
      int cur = 0;
      string ans;
      for(char c : target){
        int idx = c-'a';
        if( idx == cur ){ ans += '!'; continue; }
        if( cur == 25 ){ ans += 'U'; cur -= 5;}
        if( idx == cur ){ ans += '!'; continue; }

        while( cur%5 > idx%5 ){ ans += 'L'; cur --; }
        while( cur%5 < idx%5 ){ ans += 'R'; cur ++; }
        while( cur/5 < idx/5 ){ ans += 'D'; cur +=5;}
        while( cur/5 > idx/5 ){ ans += 'U'; cur -=5;}
        ans += '!';
      }
      return ans;  
    }
};
