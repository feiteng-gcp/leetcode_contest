class Solution {
public:
  string alphabetBoardPath(string target) {
    int x = 0, y = 0;
    string s;
    for (auto c: target) {

      int nx = (c - 'a') / 5, ny = (c - 'a') % 5;
      if (c == 'z') {
        while (y < ny) y++, s += 'R';
        while (y > ny) y--, s += 'L';
        while (x < nx) x++, s += 'D';
        while (x > nx) x--, s += 'U';
      } else {
        while (x < nx) x++, s += 'D';
        while (x > nx) x--, s += 'U';
        while (y < ny) y++, s += 'R';
        while (y > ny) y--, s += 'L';
      }
      s += '!';
    }
    return s;
  }
};
