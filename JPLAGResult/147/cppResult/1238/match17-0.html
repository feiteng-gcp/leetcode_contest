class Solution {
public:
    string alphabetBoardPath(string target) {
        vector<string> board = {"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"};
        vector<pair<int,int>> z(26);
        for (int r=0; r<board.size(); r++) {
            for (int c=0; c<board[r].size(); c++) {
                z[board[r][c]-'a'] = {r,c};
            }
        }
        string a;
        int r = 0;
        int c = 0;
        for (char x : target) {
            while (c > z[x-'a'].second) { a += 'L'; c--; }
            while (r > z[x-'a'].first) { a += 'U'; r--; }
            while (c < z[x-'a'].second) { a += 'R'; c++; }
            while (r < z[x-'a'].first) { a += 'D'; r++; }
            a+="!";
        }
        return a;
    }
};
