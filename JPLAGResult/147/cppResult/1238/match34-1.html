class Solution {
public:

    string run(int &x, int y) {
        string res = "";
        while (x / 5 > y / 5) {
            x -= 5;
            res += 'U';
        }
        while (x % 5 > y % 5) {
            x --;
            res += 'L';
        }
        while (x / 5 < y / 5) {
            x += 5;
            res += 'D';
        }
        while (x % 5 < y % 5) {
            x ++;
            res += 'R';
        }
        return res + '!';
    }
    
    string alphabetBoardPath(string target) {
        string ans = "";
        int cur = 0;
        for (auto c : target) ans += run(cur, c - 'a');
        return ans;
    }
};
