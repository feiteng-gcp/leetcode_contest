class Solution {
public:
    string alphabetBoardPath(string target) {
        target = "a"+target;
        string res = "";
        for(int i = 0;i+1 < target.size();i++) {
            vector<int> from{(target[i]-'a')/5,(target[i]-'a')%5};
            vector<int> to{(target[i+1]-'a')/5,(target[i+1]-'a')%5};
            if(from[0] >= to[0]) {
                for(int j = 0;j<from[0]-to[0];j++) {
                    res += 'U';
                }
                if(from[1] >= to[1]) {
                    for(int j = 0;j<from[1]-to[1];j++) {
                        res += 'L';
                    }
                } else {
                    for(int j = 0;j<to[1]-from[1];j++) {
                        res += 'R';
                    }                    
                }
            } else {
                if(from[1] >= to[1]) {
                    for(int j = 0;j<from[1]-to[1];j++) {
                        res += 'L';
                    }
                } else {
                    for(int j = 0;j<to[1]-from[1];j++) {
                        res += 'R';
                    }                    
                }  
                for(int j = 0;j<to[0]-from[0];j++) {
                    res += 'D';
                }
            }
            res += '!';
        }
        return res;
    }
};
