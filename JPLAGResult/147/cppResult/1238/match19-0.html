class Solution {
public:
    string alphabetBoardPath(string target) {
        int x = 0, y = 0, len = target.size();
        string ret;
        for (int i = 0; i < len; ++i) {
            int id = target[i] - 'a';
            int nx = id / 5, ny = id % 5;
            while (x > nx) { ret += "U"; --x; }
            while (y > ny) { ret += "L"; --y; }
            while (y < ny) { ret += "R"; ++y; }
            while (x < nx) { ret += "D"; ++x; }
            
            
            
            ret += "!";
        }
        return ret;
        
    }
};
