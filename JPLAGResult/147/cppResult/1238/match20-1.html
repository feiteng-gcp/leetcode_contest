class Solution {
public:
    string alphabetBoardPath(string s) {
        vector<string> b;
        b.emplace_back("abcde");
        b.emplace_back("fghij");
        b.emplace_back("klmno");
        b.emplace_back("pqrst");
        b.emplace_back("uvwxy");
        b.emplace_back("z");
        
        int ny = 0, nx = 0;
        string ans;
        for(auto& c : s) {
            int p = c - 'a';
            int py = p / 5, px = p % 5;
            if(ny == py && nx == px) {
                ans += "!";
                continue;
            }
            if(ny == 5) {
                --ny;
                ans += "U";
            }

            while(nx < px) {
                ++nx;
                ans += "R";
            }
            while(nx > px) {
                --nx;
                ans += "L";
            }
            while(ny < py) {
                ++ny;
                ans += "D";
            }
            while(ny > py) {
                --ny;
                ans += "U";
            }
            ans += "!";
        }
        return ans;
    }
};
