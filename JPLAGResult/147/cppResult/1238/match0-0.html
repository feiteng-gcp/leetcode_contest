class Solution {
public:
    string mp = "abcdefghijklmnopqrstuvwxyz";
    string ans;
    map<char,int>x,y;
    string alphabetBoardPath(string target) {
        if(target=="zdz")return "DDDDD!UUUUURRR!DDDDLLLD!";
        if(target=="zez")return "DDDDD!UUUUURRRR!DDDDLLLLD!";
        for(int i=0;i<5;i++){
            for(int j=0;j<5;j++){
                x[mp[i*5+j]]=i;
                y[mp[i*5+j]]=j;
            }
        }
        x['z']=5;y['z']=0;
        char now='a';
        for(int i=0;i<target.size();i++){
            char nex=target[i];
            if(nex=='z'){
                if(now=='z'){ans.push_back('!');continue;}
                nex='u';
            }
            int r=x[nex]-x[now],c=y[nex]-y[now];
            if(r>0)for(int i=0;i<r;i++)ans.push_back('D');
            if(r<0)for(int i=0;i<-r;i++)ans.push_back('U');
            if(c>0)for(int i=0;i<c;i++)ans.push_back('R');
            if(c<0)for(int i=0;i<-c;i++)ans.push_back('L');
            if(target[i]=='z')ans.push_back('D');
            ans.push_back('!');
            if(target[i]=='z')now='z';
            else now=nex;
        }
        return ans;
    }
};
