class Solution {
public:
    string alphabetBoardPath(string target) {
        string ans = "";
        vector<vector<char>> grid;
        unordered_map<char,pair<int,int>> m;
        vector<char> row;
        for(int i = 0; i < 26; i++) {
            if(i % 5 == 0) {
                grid.push_back(row);
                row.empty();
            }
            row.push_back('a' + i);
            m['a' + i] = make_pair(i % 5,i/5);
        }
        int startX = 0;
        int startY = 0;
        char prevChar = ' ';
        for(int i = 0; i < target.length(); i++) {
            pair<int,int> pos = m[target[i]];
            char curChar = target[i];
            int posX = pos.first;
            int posY = pos.second;
            int diffX = posX - startX;
            int diffY = posY - startY;
            if(curChar == 'z') {
                for(int j = 0; j < abs(diffX); j++) {
                    if(diffX > 0) {
                        ans += "R";
                    }
                    else {
                        ans += "L";
                    }
                }
                for(int j = 0; j < abs(diffY); j++) {
                    if(diffY > 0) {
                        ans += "D";
                    }
                    else {
                        ans += "U";
                    }
                }
            }
            else if(prevChar == 'z') {
                for(int j = 0; j < abs(diffY); j++) {
                    if(diffY > 0) {
                        ans += "D";
                    }
                    else {
                        ans += "U";
                    }
                }
                for(int j = 0; j < abs(diffX); j++) {
                    if(diffX > 0) {
                        ans += "R";
                    }
                    else {
                        ans += "L";
                    }
                }
            }
            else {
                for(int j = 0; j < abs(diffX); j++) {
                    if(diffX > 0) {
                        ans += "R";
                    }
                    else {
                        ans += "L";
                    }
                }
                for(int j = 0; j < abs(diffY); j++) {
                    if(diffY > 0) {
                        ans += "D";
                    }
                    else {
                        ans += "U";
                    }
                }
            }
            ans += "!";
            startX = posX;
            startY = posY;
            prevChar = curChar;
        }
        return ans;
    }
};
