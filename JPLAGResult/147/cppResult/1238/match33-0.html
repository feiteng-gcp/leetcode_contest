class Solution {
public:
    string alphabetBoardPath(string target) {
        int r = 0, c = 0;
        string moves = "";

        for (char letter : target) {
            int row = (letter - 'a') / 5, col = (letter - 'a') % 5;

            while (c > col) {
                moves += 'L';
                c--;
            }

            while (r > row) {
                moves += 'U';
                r--;
            }

            while (r < row) {
                moves += 'D';
                r++;
            }

            while (c < col) {
                moves += 'R';
                c++;
            }

            moves += '!';
        }

        return moves;
    }
};
