class Solution {
public:
    string alphabetBoardPath(string target) {
        unordered_map<char, vector<int>> d;
        for(char c = 'a';c<='z';c++){
            int t = c-'a';
            d[c] = {t/5,t%5};
            
        }
        vector<int> cur = {0,0};
        string out;
        for(char c:target){
            vector<int> t = d[c];
            int a = t[0] - cur[0], b = t[1] - cur[1];
            if(cur[0] != 5){
            if(b < 0){
                while(b != 0){
                    out += 'L';
                    b++;
                }
            }
            else{
                while(b != 0){
                    out += 'R';
                    b--;
                }
            }
            if(a < 0){
                while(a != 0){
                    out += 'U';
                    a++;
                }
            }
            else{
                while(a != 0){
                    out += 'D';
                    a--;
                }
            }}
            else{
                
            if(a < 0){
                while(a != 0){
                    out += 'U';
                    a++;
                }
            }
            else{
                while(a != 0){
                    out += 'D';
                    a--;
                }
            }
            if(b < 0){
                while(b != 0){
                    out += 'L';
                    b++;
                }
            }
            else{
                while(b != 0){
                    out += 'R';
                    b--;
                }
            }
            }
            out += '!';
            cur = t;
        }
        return out;
    }
};
