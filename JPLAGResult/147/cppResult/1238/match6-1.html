class Solution {
public:
    string alphabetBoardPath(string target) {
        string board[6] = {"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"};
        string ans = "";
        int cur[2] = {0, 0};
        for(int i = 0; i < target.size(); i++){
            if(target[i] == board[cur[0]][cur[1]]) ans += "!";
            else{
                int temp = target[i] -'a';
                int pos[2] = {temp/5, temp%5};

                if(target[i] == 'z'){
                    for(int j = 0; j < abs(pos[1]-cur[1]); j++){
                        if(pos[1]-cur[1]>0){
                            ans += "R";
                        }else{
                            ans += "L";
                        }
                    }
                    for(int j = 0; j < abs(pos[0]-cur[0]); j++){
                        if(pos[0]-cur[0]>0){
                            ans += "D";
                        }else{
                            ans += "U";
                        }
                    }
                }else{
                    for(int j = 0; j < abs(pos[0]-cur[0]); j++){
                        if(pos[0]-cur[0]>0){
                            ans += "D";
                        }else{
                            ans += "U";
                        }
                    }
                    for(int j = 0; j < abs(pos[1]-cur[1]); j++){
                        if(pos[1]-cur[1]>0){
                            ans += "R";
                        }else{
                            ans += "L";
                        }
                    }                    
                }
                ans += "!";
                cur[0] = pos[0];
                cur[1] = pos[1];
            }
        }
        return ans;
    }
};
