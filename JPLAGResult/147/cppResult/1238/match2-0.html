class Solution {
public:
  string alphabetBoardPath(string target) {
    vector<string> st;
    st.emplace_back("abcde");
    st.emplace_back("fghij");
    st.emplace_back("klmno");
    st.emplace_back("pqrst");
    st.emplace_back("uvwxy");
    st.emplace_back("z");

    using P = pair<int, int>;
    vector<P> vp;
    for(int i=0;i<26;i++)
      vp.emplace_back(i/5,i%5);

    string ans;
    int y=0,x=0;
    char p='a';
    for(char c:target){
      int k=c-'a';
      int ny=vp[k].first;
      int nx=vp[k].second;
      //cout<<k<<" "<<x<<" "<<y<<":"<<nx<<" "<<ny<<endl;
      if(p=='z'){
        while(y<ny) ans+='D',y++;
        while(y>ny) ans+='U',y--;
        while(x<nx) ans+='R',x++;
        while(x>nx) ans+='L',x--;
      }else{
        while(x<nx) ans+='R',x++;
        while(x>nx) ans+='L',x--;
        while(y<ny) ans+='D',y++;
        while(y>ny) ans+='U',y--;
      }
      ans+='!';
        p=c;
    }
    return ans;
  }
};
