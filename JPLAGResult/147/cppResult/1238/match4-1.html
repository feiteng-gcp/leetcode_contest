class Solution {
public:
    string alphabetBoardPath(string target) {
        string res = "";
        int x = 0, y = 0;
        for (int i = 0; i < target.size(); i++) {

            int r = (target[i]-'a')/5;
            int c = (target[i]-'a')%5;
            
            if (target[i] != 'z') {
            if (r > x) 
                for (int i = 0; i < r-x; i++) res += 'D';
            else
                for (int i = 0; i < x-r; i++) res += 'U';
            if (c > y) 
                for (int i = 0; i < c-y; i++) res += 'R';
            else
                for (int i = 0; i < y-c; i++) res += 'L';
            res += '!';

            }
            else {
            if (c > y) 
                for (int i = 0; i < c-y; i++) res += 'R';
            else
                for (int i = 0; i < y-c; i++) res += 'L';
            if (r > x) 
                for (int i = 0; i < r-x; i++) res += 'D';
            else
                for (int i = 0; i < x-r; i++) res += 'U';
            res += '!';                
            }
            x = r;y = c;
        }
        return res;
    }
};
