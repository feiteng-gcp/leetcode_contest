class Solution {
public:
    vector<string> board;
    pair<int, int> position(char ch){
        int a=ch-'a';
        return pair<int, int>(a/5, a%5);
    }
        
    string path(pair<int, int> p1, pair<int, int> p2){
        string s="";
        while(p1.second>p2.second)
        {
            s+="L";
            p1.second--;
        }
        while(p1.first>p2.first)
        {
            s+="U";
            p1.first--;
        }
        while(p1.first<p2.first)
        {
            s+="D";
            p1.first++;
        }
        while(p1.second<p2.second)
        {
            s+="R";
            p1.second++;
        }
        return s;
    }
        
        
    string alphabetBoardPath(string target) {
        board=vector<string>({"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"});
        int i, n=target.size();
        pair<int, int> start=pair<int, int>(0,0);
        pair<int, int> end=pair<int, int>(0,0);
        string r;
        for(i=0;i<n;i++)
        {
            start=end;
            end=position(target[i]);
            r+=path(start, end);
            r+="!";
        }
        return r;
    }
};
