class Solution {
public:
    string alphabetBoardPath(string target) {
        int cur_i = 0, cur_j = 0;
        string res;
        for (int i = 0; i < target.size(); ++i) {
            char c = target[i];
            int next_i = (c - 'a') / 5;
            int next_j = (c - 'a') % 5;
            if (c == 'z') {
                for (int j = 0; j < cur_j; ++j) {
                    res += "L";
                }
                for (int j = 0; j < next_i - cur_i; ++j) {
                    res += "D";
                }
            } else {

                if (cur_i > next_i) {
                    for (int j = 0; j < cur_i - next_i; ++j) {
                        res += "U";
                    }
                    if (cur_j > next_j) {   
                        for (int j = 0; j < cur_j - next_j; ++j) {
                            res += "L";
                        }
                    } else {
                        for (int j = 0; j < next_j - cur_j; ++j) {
                            res += "R";
                        }
                    }
                } else {
                    for (int j = 0; j < next_i - cur_i; ++j) {
                        res += "D";
                    }

                    if (cur_j > next_j) {   
                        for (int j = 0; j < cur_j - next_j; ++j) {
                            res += "L";
                        }
                    } else {
                        for (int j = 0; j < next_j - cur_j; ++j) {
                            res += "R";
                        }
                    }
                }
            }
            res += "!";
            cur_i = next_i;
            cur_j = next_j;
        }
        
        return res;
    }
};
