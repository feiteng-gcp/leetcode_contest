class Solution {
public:
    string alphabetBoardPath(string target) {
      vector<string> board = {"abcde", "fghij", "klmno", "pqrst", "uvwxy", "z"};

      map<char, pair<int, int>> ch_to_pos;
      for (int i = 0; i < board.size(); ++i) {
        for (int j = 0; j < board[i].size(); ++j) {
          ch_to_pos[board[i][j]] = {i, j};
        }
      }
      auto mv = [&] (int from_x, int from_y, int to_x, int to_y, string &out) {

        while (from_y > to_y) {
          out += "L";
          --from_y;
        }
        
        while (from_x < to_x) {
          out += "D";
          ++from_x;
        }
        while (from_x > to_x) {
          out += "U";
          --from_x;
        }
        while (from_y < to_y) {
          out += "R";
          ++from_y;
        }
        out += "!";
      };
      int now_x = 0, now_y = 0; 
      string ans;
      for (const auto &ch : target) {
        auto new_pos = ch_to_pos[ch];
        mv(now_x, now_y, new_pos.first, new_pos.second, ans);
        now_x = new_pos.first, now_y = new_pos.second;
      }
      return ans;
    }
    
};
