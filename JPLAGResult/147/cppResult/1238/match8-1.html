class Solution {
public:
    string alphabetBoardPath(string target) {
        int cur = 0;
        string ret;
        int cy=0,cx=0;
        while(cur < target.size()){
            int nxt = target[cur]-'a';

            int ny = nxt/5, nx=nxt%5;
            
            while(cy>ny){ ret += 'U'; cy--; }
            while(cx>nx){ ret += 'L'; cx--; }
            while(cx<nx){ ret += 'R'; cx++; }      
            while(cy<ny){ ret += 'D'; cy++; }
            

            
            ret += '!';
            cur++;
        }
        
        return ret;
    }
};
