# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        if not root:
            return 0
        totalSum = 0
        if L <= root.val <= R:
            totalSum += root.val
            totalSum += self.rangeSumBST(root.left, L, R)
            totalSum += self.rangeSumBST(root.right, L, R)
            return totalSum
        elif root
.val < L:
            return self.rangeSumBST(root.right, L, R)
        else:
            return self.rangeSumBST(root.left, L, R)
