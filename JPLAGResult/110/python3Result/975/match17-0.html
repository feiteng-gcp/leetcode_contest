# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def dfs(self, root, l, r):
        if root is None:
            return
        if l <= root.val <= r:
            self.ans += root.val

        if l < root.val:
            self.dfs(root.left, l, r)
        if r > root.val:
            self.dfs(root.right, l, r)

    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        self.ans = 0
        self.dfs(root, L, R)
        return self.ans
