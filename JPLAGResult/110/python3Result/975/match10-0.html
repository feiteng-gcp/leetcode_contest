# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        if root.val<=R and root.val>=L:
            s=root.val
        else:
            s=0
        if root.left!=None:
            s=s+self.rangeSumBST(root.left,L,R)
        if root.right!=None:
            s=s+self.rangeSumBST(root.right,L,R)
        return s
