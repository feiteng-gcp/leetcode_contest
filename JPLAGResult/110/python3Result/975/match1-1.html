# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        
        cur = 0
        
        def dfs(root, L, R):
            nonlocal cur
            
            if not root: return 
            if L <= root.val <= R:
                cur += root.val
            if L <= root.val:
                dfs(root.left, L, R)
            if root.val <= R:
                dfs(root.right, L, R)
        dfs(root, L, R)
        
        return cur
