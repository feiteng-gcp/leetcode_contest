# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        res = 0
        
        def fun(root):
            nonlocal res
            if root != None:
                if L <= root.val <= R:
                    res += root.val
                fun(root.left)
                fun(root.right)
        
        fun(root)
        return res
