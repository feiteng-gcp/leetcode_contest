# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        result = 0
        
if not root:
            return result
        if root.val <= R and root.val >= L:
            # print("adding",root.val)
            result += root.val
        if root.val < R and root.right:
            result += self.rangeSumBST(root.right, L, R)
        if root.val > L and root.left:
            result += self.rangeSumBST(root.left, L, R)
        return result
