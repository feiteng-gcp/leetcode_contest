# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        total_sum = 0
        
        def traverse(node):
            nonlocal total_sum
            if node is None:
                return
            
            if L <= node.val <= R:
                total_sum += node.val
                
            if node.val >= L:
                traverse(node.left)
            if node.val <= R:
                traverse(node.right)
        
        traverse(root)
        
        return total_sum
