# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def helper(self, cur, L, R):
        
        if(cur == None):
            return 0
        
        som = 0
        if(cur.val >= L and cur.val <= R):
            som += cur.val
            
        som += self.helper(cur.left, L, R)
        som += self.helper(cur.right, L, R)
        return som
    
    def rangeSumBST(self, root, L, R):
        """
        :type root: TreeNode
        :type L: int
        :type R: int
        :rtype: int
        """
        return self.helper(root, L, R)
