class Solution:
    def distinctSubseqII(self, S):
        """
        :type S: str
        :rtype: int
        """
        maps = {}
        dp = [0]*(len(S)+1)
        dp[0] = 1
        for i in range(1,len(S)+1):
            dp[i] = 2*dp[i-1]
            if S[i-1] in maps:
                dp[i] = (dp[i]-dp[maps[S[i-1]]])%1000000007
            maps[S[i-1]] = i-1
        return dp[-1]%1000000007-1
