class Solution:
    def distinctSubseqII(self, S):
        """
        :type S: str
        :rtype: int
        """
        dp = [1] * (len(S) + 1)
        pos = {}
        
        for i in range(1, len(S)+1):
            a = S[i-1]
            dp[i] = dp
[i-1] * 2
            
            if a in pos:
                dp[i] -= dp[pos[a]]
            
            pos[a] = i-1

        return (dp[-1]-1) % (10**9 + 7)
