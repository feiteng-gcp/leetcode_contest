class Solution:
    def distinctSubseqII(self, s):
        """
        :type S: str
        :rtype: int
        """
        last = {}
        n = len(s)
        dp = [0] * (n+1)
        dp[0] = 1
        for i in range
(1, n+1):
            dp[i] = 2 * dp[i-1]
            if s[i-1] in last:
                dp[i] = dp[i] - dp[last[s[i-1]]]
            last[s[i-1]] = i-1
        return (dp[n]-1) % (10**9 + 7)
