import itertools
class Solution:
    def distinctSubseqII(self, S):
        """
        :type S: str
        :rtype: int
        """
        
        #result = set()
        #for i in range(0, ls):
            #for j in range(i+1, ls+1):
                #result.add(S[i:j])
        #print(list(itertools.product(S, repeat=len(S))))
        #print(list(result))
        la = {}
        ls = len(S)
        distinct = [0] * (ls+1)
        distinct[0] = 1
        #la[""] = 0
        
        for i in range(1, ls+1):
            #print(i)
            #print(distinct)
            #print(la)
            distinct[i] = 2 * distinct[i-1]
            
            #if (la[ord(S[i-1])] != -1):
            if S[i-1] in la:
                #print("in")
                distinct[i] = distinct[i] - distinct[la[S[i-1]]]
                #print("n", distinct)
            
            la[S[i-1]] = i-1
        #print(distinct)
        #print("final", distinct)
        return (distinct[ls]-1) % (10**9 + 7)
