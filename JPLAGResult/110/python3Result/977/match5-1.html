class Solution:
    def distinctSubseqII(self, S):
        """
        :type S: str
        :rtype: int
        """
        ht = [0 for _ in S] + [0]
        ht[0] = 1
        seen = {}

        for i in range(1, len(S)+1):
            ht[i] = 2 * ht[i-1]

            c = S
[i-1]
            if c in seen:
                ht[i] -= ht[seen[c]]

            seen[c] = i-1
        
        # Minus off the empty subsequence
        return (ht[len(S)] - 1) % (10**9 + 7)
