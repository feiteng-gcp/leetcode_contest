class Solution:
    def distinctSubseqII(self, S):
        """
        :type S: str
        :rtype: int
        """
        dp = [1]
        last = dict()
        for i, c in enumerate(S):
            dp.append(2 * dp[i])
            if c in last:
                dp[i + 1] -= dp[last[c]]
            last[c] = i
        return (dp[len(S)] - 1) % 1000000007
