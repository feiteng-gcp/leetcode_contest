class Solution:
    def distinctSubseqII(self, s):
        """
        :type S: str
        :rtype: int
        """
        d = {}
        n = len(s)
        dp = [0 for _ in range(n + 1)]
        dp[0] = 1
        for i in range(1, n + 1):
            dp[i] = 2 * dp[i - 1]
            if s[i-1] in d:
                dp[i] -= dp
[d[s[i-1]]]
            d[s[i-1]] = i-1
        return (dp[n] - 1) % (10**9 + 7)
