class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int

        """

        from collections import defaultdict
        d = defaultdict(list)
        for x,y in points:
            d[x].append(y)
        for k in d.keys():
            d[k] = sorted(d[k])



        l = sorted(d.keys())
        res = float("inf")
        for i in range(len(l)):
            for j in range(i+1,len(l)):
                high = l[j] - l[i]
                for a in range(len(d[l[i]])):
                    for b in range(a+1,len(d[l[i]])):
                        if (d[l[i]][b]-d[l[i]][a]) * high >= res:
                            break
                        if d[l[i]][b] in d[l[j]] and d[l[i]][a] in d[l[j]]:
                            res = (d[l[i]][b]-d[l[i]][a]) * high

        if res == float("inf"):
            return 0
        else:
            return res
