class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        s = set([tuple(p) for p in points])
        x2ys = collections.defaultdict(list)
        y2xs = collections.defaultdict(list)
        for p in points:
            x2ys[p[0]].append(p[1])
            y2xs[p[1]].append(p[0])
        '''
        for k, v in x2ys.items():
            x2ys[k] = sorted(v)
        for k, v in y2xs.items():
            x2ys[k] = sorted(v)
        '''
        d0 = x2ys if len(x2ys.keys()) > len(y2xs.keys()) else y2xs
        d = collections.defaultdict(list)
        
        for k, v in d0.items():
            for i in range(len(v)):
                for j in range(i+1, len(v)):
                    dk = min(v[i], v[j])
                    dl = abs(v[i] - v[j])
                    d[(dk, dl)].append(k)
        
        out = 0
        for k, v in d.items():
            if len(v) > 1:
                v2 = sorted(v)
                mingap = min([v2[i + 1] - v2[i] for i in range(len(v2) - 1)])
                area = k[1] * mingap
                out = area if out == 0 else min(out, area)
        return out
