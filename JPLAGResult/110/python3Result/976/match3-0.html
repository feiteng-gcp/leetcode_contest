from collections import defaultdict


class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        x_to_y = defaultdict(set)
        all_x = set()

        for x, y in points:
            x_to_y[x].add(y)
            all_x.add(x)

        all_x = sorted(list(all_x))

        min_area = float('inf')

        for i in range(len(all_x) - 1):
            for other_i in range(i+1, len(all_x)):
                left_x, right_x = all_x[i], all_x[other_i]

                common_y = x_to_y[left_x] & x_to_y[right_x]
                common_y = sorted(list(common_y))

                for j in range(len(common_y) - 1):
                    top_y, bottom_y = common_y[j], common_y[j + 1]

                    area = (bottom_y - top_y) * (right_x - left_x)

                    min_area = min(min_area, area)

        if min_area == float('inf'):
            return 0

        return min_area
