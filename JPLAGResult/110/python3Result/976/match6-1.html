from collections import defaultdict
class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        hl = defaultdict(list)
        # vl = defaultdict(list)
        for p in points:
            hl[p[1]].append(p[0])
            # vl[p[0]].append(p[1])
        nhl = []
        for l in hl:
            nhl.append([l, hl[l]])
        nhl.sort()
        # print(nhl.sort())
        # nvl = []
        # for l in vl:
        #     nvl.append([l, sorted(vl[l])])
        lh = len(nhl)
        # lv = len(nvl)
        ret = 40001 * 40001
        for y1 in range(lh):
            for y2 in range(y1+1, lh):
                candi_x = list(set(nhl[y1][1]) & set(nhl[y2][1]))
                candi_x.sort()
                min_diff = 40001
                # print(y1, y2, nhl[y1][0], nhl[y2][0], candi_x)
                for i in range(len(candi_x) - 1):
                    min_diff = min(min_diff, candi_x[i+1] - candi_x[i])
                if min_diff != 40001:
                    ret 
= min(ret, (nhl[y2][0]-nhl[y1][0])*(min_diff))
        return ret if ret < 1600080001 else 0
