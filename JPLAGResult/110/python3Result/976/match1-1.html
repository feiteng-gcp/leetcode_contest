class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        dp=dict()
        points.sort()
        #print(points)
        last=None
        curr=[]
        ret=None
        for p in points:
            #print(p)
            #if (3, 4) in dp:
            #    print(dp[(3, 4)])
            if p[0]!=last:
                last=p[0]
                curr=[p[1]]
            else:
                for x in curr:
                    if (x, p[1]) in dp:
                        if ret==None or (p
[1]-x)*(p[0]-dp[(x, p[1])])<ret:
                            ret=(p[1]-x)*(p[0]-dp[(x, p[1])])
                    dp[(x, p[1])]=p[0]
                curr.append(p[1])
        if ret==None:
            return 0
        else:
            return ret
