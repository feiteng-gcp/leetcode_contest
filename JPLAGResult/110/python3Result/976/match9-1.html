class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        
        pt = [tuple(i) for i in points]
        pts = set(pt)
        ar = float('inf')
        for i in range(len(pt)):
            iv = pt[i]
            for jv in pt[i+1:]:
                
if (iv[0] != jv[0]) and (iv[1] != jv[1]) and \
                    (iv[0], jv[1]) in pts and (jv[0], iv[1]) in pts:
                    ijar = abs(jv[1] - iv[1]) * abs(jv[0] - iv[0])
                    if ijar < ar:
                        ar = ijar
        if ar == float('inf'):
            return 0
        return ar
