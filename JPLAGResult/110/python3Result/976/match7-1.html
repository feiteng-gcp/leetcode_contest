from collections import defaultdict

class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """

        mpx, mpy = defaultdict(set), defaultdict
(set)
        for p in points:
            mpx[p[0]].add(p[1])
            mpy[p[1]].add(p[0])
        res = (1<<31) - 1
        for p0 in points:
            for p1 in points:
                if p1[0] <= p0[0] or p1[1] <= p0[1]: continue
                if p0[0] in mpy[p1[1]] and p0[1] in mpx[p1[0]]:
                    res = min(res, (p1[0]-p0[0])*(p1[1]-p0[1]))
        return res if res < (1<<31) - 1 else 0
