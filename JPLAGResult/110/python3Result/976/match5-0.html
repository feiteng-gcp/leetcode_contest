from collections import defaultdict
class Solution:
    def minAreaRect(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        if len(points) <= 4:
            return 0
        
        sets = set()
        
        xs = set()
        ys = set()
      #  ysets = defaultdict(set)
       # xs = set()
        for i, (x, y) in enumerate(points):
            sets.add((x, y))
            xs.add(x)
            ys.add(y)
        
        xss = sorted(xs)
        yss = sorted(ys)
        m = len(xss)
        n = len(yss)
        mina = float('inf')
        for l in range(m - 1):
            for r in range(l + 1, m):
                for lo in range(n - 1):
                    for u in range(lo + 1, n):
                        if (xss[l], yss[lo]) in sets and (xss[l], yss[u]) in sets and (xss[r], yss[lo]) in sets and (xss[r], yss[u]) in sets:
                            mina = min(mina, (xss[r] - xss[l]) * (yss[u] - yss[lo]))
        if mina < float('inf'):
            return mina
        return 0
