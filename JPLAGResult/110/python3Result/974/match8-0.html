class Solution:
    def reorderLogFiles(self, logs):
        num = []
        letter = []
        for log in logs:
            if log.split(' ')[1].isdigit():
                num.append(log)
            else:
                letter.append(log)
        letter = sorted(letter, key = lambda log: " ".join(log.split(' ')[1:]))
        return letter + num
        """
        :type logs: List[str]
        :rtype: List[str]
        """
