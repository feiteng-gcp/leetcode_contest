class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        num_log = []
        char_log = []
        for e in logs:
            s = e.split()
            if s[1].isdigit():
                num_log.append(e)
            else:
                char_log.append(e)

        res = sorted(char_log, key = lambda x: x.split()[1:])

        return res + num_log
