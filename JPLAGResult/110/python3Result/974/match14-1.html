class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        letter_logs = []
        digit_logs = []
        for log in logs:
            x = log.split(" ")
            if (x[1][0] >= '0' and x[1][0] <='9' ):
                digit_logs
.append(log)
            else:
                letter_logs.append(log)
        letter_logs = sorted(letter_logs, key = lambda x: " ".join( x.split(" ")[1:]+[(x.split(" ")[0])] ))
        res =letter_logs
        for log in digit_logs:
            res.append(log)
        return res
