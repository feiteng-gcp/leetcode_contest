class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        letter = []
        digit = []
        f = lambda x: x[1]
        
for i in logs:
            a, b, c = i.partition(' ')
            if c[0].isdigit():
                digit.append(i)
            else:
                letter.append((a, c))
        letter.sort(key = f)
        letter2 = [a + ' ' + b for a, b in letter]
        return letter2 + digit
