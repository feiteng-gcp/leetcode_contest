class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        letter, digits 
= [], []
        
        for log in logs:
            if log.split(' ')[1].isalpha():
                letter.append(log)
            else:
                digits.append(log)
        
        return sorted(letter, key=lambda x: x.split(' ')[1:]) + digits
        # return sorted(logs, key=lambda x: not x.split(' ')[1].isalpha())
