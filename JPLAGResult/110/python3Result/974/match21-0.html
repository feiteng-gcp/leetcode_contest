class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        ll,dl = [], []
        for log in logs:
            if log[log.find(" ")+1].isdigit():
                dl.append(log)
            else:
                ll.append(log)
        ll.sort(key=lambda x:x[x.find(" "):])
        return ll+dl
