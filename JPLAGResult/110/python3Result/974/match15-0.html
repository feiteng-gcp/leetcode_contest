class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        list_numbers = []
        res = []
        for log in logs:
            if log[-1].isdigit():
                list_numbers.append(log)
            else:
                idx = log.find(' ')
                res.append(log[idx + 1:] + ' ' + log[:idx])
        res.sort()
        ans = []
        for log in res:
            idx = log.rfind(' ')
            ans.append(log[idx + 1: ] + ' ' + log[:idx])
        ans.extend(list_numbers)
        return ans
