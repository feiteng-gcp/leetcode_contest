class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        
        letter_logs = [log for log in logs if not log.split(" ")[1].isdigit()]
        digit_logs = [log for log in logs if log.split(" ")[1].isdigit()]
        
        return list(sorted(letter_logs, key = lambda entry: ' '.join(entry.split(" ")[1:]) + " ---- " + entry.split(" ")[0])) + digit_logs
