class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        def getKey(log):
            l = log.split()
            newlog = l[1:]+[l[0]]
            return newlog
            
        dlogs = []
        llogs = []
        for i in range(len(logs)):
            if logs[i].split()[1][0]
.isdigit():
                dlogs.append(logs[i])
            else:
                llogs.append(logs[i])
        llogs.sort(key=getKey)
        return llogs + dlogs
