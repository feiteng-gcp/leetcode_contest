class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        if not logs: return []
        
        digits, letters = [], []
        for i, log in enumerate(logs):
            temp = log.split(" ")
            if temp[1].isdigit(): 
                digits.append(i)
            else:
                letters.append((i, temp[1:]))
        letters.sort(key=lambda x: x[1])
        
        return [logs[item[0]] for item in letters] + [logs[item] for item in digits]
