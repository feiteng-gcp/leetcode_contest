class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        log_type_0=[]
        log_type_1=[]
        for s in logs:
            curr=s.split()
            if not curr[1].isdigit():
                to_add=log_type_0
            else:
                to_add=log_type_1
            to_add.append([curr[0]])
            to_add[-1].append(s[len(curr[0]):])
        log_type_0.sort(key=lambda log:log[1]+log[0])
        ret=[]
        for curr in log_type_0 + log_type_1:
            ret.append(curr[0]) 
            ret[-1]+=curr[1]
        return ret
