class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        lt = []
        nm = []
        for i in logs:
            x = i.split()
            if len
(x) > 1 and x[1].isdigit():
                nm.append(i)
            else:
                lt.append(i)
        lt.sort(key=lambda a: a.split()[1:])
        return lt + nm
