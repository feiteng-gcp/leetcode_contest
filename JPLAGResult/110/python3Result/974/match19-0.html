class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        maps = {}
        res = []
        res2=[]
        for l in logs:
            strs = l.split(' ')
            if strs[1].isdigit():
                res2.append(l)
            else:
                maps[' '.join(strs[1:])] = l
                res.append(' '.join(strs[1:]))
                
        res.sort()
        res3 = []
        for r in res:
            res3.append(maps[r])
        return res3 + res2
