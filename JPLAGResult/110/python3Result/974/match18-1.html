class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        letter = []
        digit = []
        for l in logs:
            s = l.split()
            if s[1][0] not in '0123456789':
                letter.append(l)
            else:
                digit.append(l)
        letter.sort(key=lambda x: [' '.join(x.split()[1:]), x.split()[0]])
        return letter + digit
