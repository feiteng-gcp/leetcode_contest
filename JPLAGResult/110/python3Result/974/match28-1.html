class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        digits = []
        letters = {}
        for log in logs:
            temp = log.split(' ')
            if temp[1][0].isdigit():
                digits.append(log)
            else:
                key = ' '.join(temp[1:])
                if key in letters:
                    letters[key].add(log)
                else:
                    letters[key] = [log]
        # print("digits",digits)
        # print("letters",letters)
        result = []
        for key in sorted(letters.keys()):
            # print("at key",key,"value is",letters[key])
            # if len(letters[key]) == 1:
            #     print("adding",letters[key])
            #     result.append(letters[key])
            # else:
            for log in sorted(letters[key]):
                # print("Adding",log)
                result.append(log)
        for log in digits:
            # print("Adding",log)
            result.append(log)
        return result
