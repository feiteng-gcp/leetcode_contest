class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        inp = logs
        digits = []
        letters = []

        for i in inp:
            s = i.split(" ")
            if(s[1].isalpha()):
                letters.append((s[1:], s[0]))
            else:
                digits.append(i)

        letters = sorted(letters)

        new_letters = []
        for l in letters:
            new_letters.append(l[1] + " " + ' '.join(l[0]))

        ans = new_letters + digits
        return ans
