class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        letter_log, digit_log = [], []
        for log in logs:
            items = log.split()
            (Id, x), content = items[:2], ' '.join(items[1:])
            if x.isdigit():
                digit_log.append((content, Id))
            else:
                letter_log.append((content, Id))
        letter_log.sort()
        return [Id + ' ' + log for log, Id in letter_log + digit_log]
