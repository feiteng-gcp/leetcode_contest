class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        l_v = []
        n_v = []
        
        for log in logs:
            words = log.split(" ")
            if 'a'<=words[1][0] <= 'z':
                l_v.append(words[1:] + [words[0]])
            else:
                n_v.append(log)
        # print(l_v)
        l_v.sort()
        res = []
        for l in l_v:
            ll = [l[-1]] + l[:-1]
            res.append(" ".join(ll))
        
        res.extend(n_v)
        return res
