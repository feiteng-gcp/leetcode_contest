class Solution:
    def reorderLogFiles(self, logs):
        """
        :type logs: List[str]
        :rtype: List[str]
        """
        if not logs:
            return []
        
        letter = []
        digit = []
        for log in logs:
            if ord("a") <= ord(log.split(" ")[1][0]) <= ord("z"):
                letter.append(log)
            else:
                digit.append(log)
                
        letter.sort(key=lambda x : x.split(" ")[1:])
        return letter + digit
