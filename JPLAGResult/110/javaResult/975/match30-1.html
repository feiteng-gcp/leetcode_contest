/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public int rangeSumBST(TreeNode root, int L, int R) {
        if (root == null) {
            return 0;
        }
        
        int val = root.val;
        int sum = 0;
        if (L <= root.val && root.val <= R) {
            sum += root.val;
            return sum + rangeSumBST(root.left, L, root.val) + rangeSumBST(root.right, root.val, R);
        
}
        
        if (root.val < L) {
            return rangeSumBST(root.right, L, R);
        }
        
        if (R < root.val ) {
            return rangeSumBST(root.left, L, R);
        }
        
        return 0;
    }
}
