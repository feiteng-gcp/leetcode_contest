/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public int rangeSumBST(TreeNode root, int L, int R) {
           int[] ans = new int[1];
           dfs(root, ans, L, R);
           return ans[0]; 
    }
    public void dfs(TreeNode root, int[] val, int L, int R) {
           if (root == null) {
                   return;
           } 
           if (root.val >= L && root.val <=R) {
                   val[0] += root.val;
           } 
           dfs(root.left, val, L, R);
           dfs(root.right, val, L, R); 
    }    
}
