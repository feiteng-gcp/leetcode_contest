/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int res;
    public int rangeSumBST(TreeNode root, int L, int R) {
        res = 0;
        if (root == null) {
            return res;
        }
        
dfs(root, L, R);
        return res;
    }
    
    private void dfs(TreeNode root, int L, int R) {
        if (root == null) {
            return;
        }
        dfs(root.left, L, R);
        if (root.val >= L && root.val <= R) {
            res += root.val;
        } else if (root.val > R) {
            return;
        }
        dfs(root.right, L, R);
    }
}
