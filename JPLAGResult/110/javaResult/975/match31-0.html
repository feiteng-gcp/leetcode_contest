/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int rangeSumBST(TreeNode root, int L, int R) {
        int[] res = new int[1];
        helper(res, root, L, R);
        return res[0];
    }
    
    private void helper(int[] res, TreeNode cur, int L, int R) {
        if (cur == null) {
            return;
        }
        if (cur.val <= R && cur.val >= L) {
            res[0] += cur.val;
        }
        if (cur.val > L) {
            helper(res, cur.left, L, R);
        }
        if (cur.val < R) {
            helper(res, cur.right, L, R);
        }
    }
}
