class Solution {
    public int minAreaRect(int[][] points) {
        TreeMap<Integer, List<int[]>> data = new TreeMap<>();
        
        for(int[] p : points) {
            data.computeIfAbsent(p[0], ll -> new ArrayList<>()).add(p);
        }
        
        Map<Integer, Integer> lastX = new HashMap<>(); 
        int best = 1_000_000_000;
        for(Map.Entry<Integer, List<int[]>> item : data.entrySet()) {
            List<int[]> pointsOnX = item.getValue();
            for(int s = 0 ; s < pointsOnX.size() ; s++) {
                 for(int h = s + 1 ; h < pointsOnX.size() ; h++) {
                    int s1 = Math.min(pointsOnX.get(s)[1], pointsOnX.get(h)[1]);
                    int s2 = Math.max(pointsOnX.get(s)[1], pointsOnX.get(h)[1]);
                     
                    int key = s1 * 40000 + s2;
                    Integer last = lastX.get(key);
                     if(last != null) {
                         best = Math.min(best, (s2 - s1) * (item.getKey() - last));
                     }
                    lastX.put(key, item.getKey());
                 }
            }
        }
        
        
        if(best == 1_000_000_000) {
            return 0;
        }
        return best;
    }
    
    
}
