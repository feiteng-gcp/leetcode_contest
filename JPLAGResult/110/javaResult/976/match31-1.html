class Solution {
    public int minAreaRect(int[][] points) {
        HashSet<String> set = new HashSet<>();
        for (int[] p : points) {
            set.add(p[0] + "," + p[1]);
        }
        int n = points.length;
        int res = Integer.MAX_VALUE;
        for (int i=0; i<n; i++) {
            for (int j=i+1; j<n; j++) {
                if (exist(set, points[i], points[j])) {
                    // System.out.println(i + " " + j + " exist");
                    res = 
Math.min(res, area(points[i], points[j]));
                }
            }
        }
        return (res==Integer.MAX_VALUE) ? 0 : res;
    }
    
    private int area(int[] x, int[] y) {
        int w = Math.abs(x[0]-y[0]);
        int h = Math.abs(x[1]-y[1]);
        // System.out.println("area: " + w*h);
        return w*h;
    }
    
    private boolean exist(HashSet<String> set, int[] a, int[] b) {
        if (a[0] == b[0] || a[1] == b[1]) {
            return false;
        }
        String p1 = a[0] + "," + b[1];
        String p2 = b[0] + "," + a[1];
        return set.contains(p1) && set.contains(p2);
    }
}
