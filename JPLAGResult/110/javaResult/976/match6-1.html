class Solution {
    public int minAreaRect(int[][] points) {
        HashMap<Integer, Set<Integer>> map = new HashMap<>();
        for (int[] point : points) {
            if (!map.containsKey(point[0])) map.put(point[0], new HashSet<>());
            map.get(point[0]).add(point[1]);
        }
        //a b   
        //c d
        int area = Integer.MAX_VALUE;
        for (int i = 0; i < points.length; i++) {
            for (int j = i+1; j < points.length; j++) {
                int[] p1 = points[i], p2 = points[j];
                if (p1[0] == p2[0] || p1[1] == p2[1]) continue;
                if ((map.containsKey(p2[0]) && map.get(p2[0]).contains(p1[1])) 
                        &&(map.containsKey(p1[0]) && 
map.get(p1[0]).contains(p2[1]))) {
                    
                    area = Math.min(area, Math.abs(p1[0] - p2[0])*Math.abs(p1[1] - p2[1]));
                    //System.out.println(p1[0]+ " " + p1[1]+ " " + p2[0]+ " " + p2[1]+ " " + area);
                }
            }
        }
        return area == Integer.MAX_VALUE ? 0 : area;
    }
}
