class Solution {
    public int minAreaRect(int[][] points) {
        int ret = 1 << 30; 
        Map<Integer, Set<Integer>> map = new HashMap<>(); 
        int n = points.length; 
        for (int i = 0; i < n; i++) {
            if (!map.containsKey(points[i][0])) {
                map.put(points[i][0], new HashSet<>());
            }
            map.get(points[i][0]).add(points[i][1]);
        }
        for (int i = 0; i < n; i++) {
            int x1 = points[i][0];
            int y1 = points[i][1];
            for (int j = i + 1; j < n; j++) {
                int x2 = points[j][0];
                int y2 = points[j][1];
                if (x1 == x2 || y1 == y2) {
                    continue;
                }
                if (map.containsKey(x1) && map.get(x1).contains(y2) && map.containsKey(x2) && map.get(x2).contains(y1)) {
                    ret = Math.min(ret, Math.abs(x2 - x1) * Math.abs(y2 - y1));
                }
            }
        }
        return ret == (1 << 30) ? 0 : ret; 
    }
}
