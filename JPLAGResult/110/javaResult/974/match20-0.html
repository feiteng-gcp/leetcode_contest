class Solution {
    public String[] reorderLogFiles(String[] logs) {
        String[] ans = new String[logs.length];
        List<Integer> digIdx = new ArrayList<>();
        PriorityQueue<String> pq = new PriorityQueue<>(new Comparator<String>(){
           public int compare(String s1, String s2){
               String[] ss1 = s1.split(" ");
               String[] ss2 = s2.split(" ");
               String s1new = s1.substring(ss1[0].length());
               String s2new = s2.substring(ss2[0].length());
               return s1new.compareTo(s2new);
           } 
        });
        for(int i = 0; i < logs.length; i++){
            String[] ss = logs[i].split(" ");
            //if digit
            if(ss[1].charAt(0) <= '9' && ss[1].charAt(0) >= '0'){
                digIdx.add(i);
            }
            else{
                pq.add(logs[i]);
            }
        }
        int i = 0;
        while(!pq.isEmpty()){
            ans[i++] = pq.remove();
        }
        for(int j = 0; j < digIdx.size(); j++){
            ans[i++] = logs[digIdx.get(j)];
        }
        return ans;
    }
}
