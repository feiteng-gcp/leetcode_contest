class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> letterLogs = new ArrayList<>();
        List<String> digitLogs = new ArrayList<>();
        for (String log : logs) {
            if (Character.isLowerCase(
log.charAt(log.length() - 1))) {
                letterLogs.add(log);
            } else {
                digitLogs.add(log);
            }
        }
        letterLogs.sort((
log1, log2) -> {
            String substring1 = log1.substring(log1.indexOf(' ') + 1);
            String substring2 = log2.substring(log2.indexOf(' ') + 1);
            int compare = substring1.compareTo(substring2);
            if (compare == 0) {
                compare = log1.split("\\s+")[0].compareTo(log2.split("\\s+")[0]);
            }
            return compare;
        });
        letterLogs.addAll(digitLogs);
        return letterLogs.toArray(new String[0]);        
    }
}
