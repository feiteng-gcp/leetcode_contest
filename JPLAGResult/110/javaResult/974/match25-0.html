class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> letter = new ArrayList<>(), digit = new ArrayList<>();
        for (String s : logs) {
            int t = s.indexOf(' ');
            if (s.charAt(t + 1) >= '0' && s.charAt(t + 1) <= '9') digit.add(s);
            else letter.add(s);
        }
        Collections.sort(letter, new Comparator<String>() {
            @Override
            public int compare(String o1, String o2) {
                int id1 = o1.indexOf(' '), id2 = o2.indexOf(' ');
                String s1 = o1.substring(id1 + 1), s2 = o2.substring(id2 + 1);
                if (s1 == s2) return (o1.substring(0, id1 + 1).compareTo(o2.substring(0, id2 + 1)));
                else return s1.compareTo(s2);
            }
        });
        String[] res = new String[logs.length];
        int count = 0;
        for (String s : letter) res[count++] = s;
        for (String s : digit) res[count++] = s;
        return res;
    }
}
