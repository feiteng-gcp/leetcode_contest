class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> digitalLog = new ArrayList<>();
        List<String> letterLog = new ArrayList<>();
        for (String log : logs) {
            int i = log.indexOf(" ");
            if (Character.isDigit(log.charAt(i + 1))) {
                digitalLog.add(log);
            } else {
                String a = log.substring(i);
                for (int j = 0; j <= letterLog.size(); j++) {
                    if (j == letterLog.size()) {
                        letterLog.add(j, log);
                        break;

                    }
                    int ii = letterLog.get(j).indexOf(" ");
                    String b = letterLog.get(j).substring(ii);
                    if (a.compareTo(b) > 0) {
                        continue;
                    } else if (a.compareTo(b) == 0) {
                        String ai = log.substring(0, i);
                        String bi = log.substring(0, ii);
                        if (ai.compareTo(bi) > 0) {
                            continue;
                        } else {
                            letterLog.add(j, log);
                            break;
                        }
                    } else {
                        letterLog.add(j, log);
                        break;
                    }
                }
            }
        }
        String[] res = new String[logs.length];
        int i = 0;
        for (String l : letterLog) {
            res[i++] = l;
        }
        for (String l : digitalLog) {
            res[i++] = l;
        }
        return res;
    }
}
