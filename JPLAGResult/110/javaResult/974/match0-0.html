class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> ans1 = new ArrayList<>();
        List<String> ans2 = new ArrayList<>();
        for (String s : logs) {
            String[] info = s.split("\\s+");
            if (Character.isDigit(info[1].charAt(0))) {
                ans1.add(s);
            } else {
                ans2.add(s);
            }
        }
        PriorityQueue<String> pq = new PriorityQueue<>((a, b) -> (a.substring(a.indexOf(" ") + 1).compareTo(b.substring(b.indexOf(" ") + 1))));
        pq.addAll(ans2);
        String[] ret = new String[logs.length];
        int i = 0;
        while (!pq.isEmpty()) {
            ret[i++] = pq.poll();
        }
        for (String s : ans1) {
            ret[i++] = s;
        }
        return ret;
    }
}
