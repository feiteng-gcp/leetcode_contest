class Solution {
    public String[] reorderLogFiles(String[] logs) {
        
PriorityQueue<String> letter = new PriorityQueue<>(new Comparator<String>(){
            public int compare(String a, String b) {
                String A = a.substring(a.indexOf(" ") + 1, a.length());
                String B = b.substring(b.indexOf(" ") + 1, b.length());
                return A.compareTo(B);
            }
        });
        List<String> digit = new ArrayList<>();
        for (String log : logs) {
            char last = log.charAt(log.length() - 1);
            if (last >= '0' && last <= '9') 
{
                digit.add(log);
            } else {
                letter.add(log);
            }
        }
        String[] result = new String[logs.length];
        
int index = 0;
        while (!letter.isEmpty()) {
            result[index++] = letter.poll();
        }
        for (int i = 0; i < digit.size(); i++) {
            result[index + i] = digit.get(i);
        }
        return result;
    }
}
