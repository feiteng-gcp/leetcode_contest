class Solution {
    public String[] reorderLogFiles(String[] logs) {
        LinkedList<String[]> letterLogs = new LinkedList();
        LinkedList<String> digitLogs = new LinkedList();
        for(String log : logs) {
            String[] logTokens = log.split(" ");
            if (isDigitLog(logTokens)) {
                digitLogs.add(log);
            } else {
                letterLogs.add(new String[] {
                    logTokens[0], join(logTokens, 1, logTokens.length, " ")
                });
            }
        }
        
        int i = 0;
        String[][] lLogs = new String[letterLogs.size()][];
        for (String[] log : letterLogs) {
            lLogs[i++] = log;
        }
        Arrays.sort(lLogs, new Comparator<String[]> () {
            public int compare(String[] log1, String[] log2) {
                int cmpr = log1[1].compareTo(log2[1]);
                if (cmpr == 0) {
                    return log1[0].compareTo(log2[0]);
                } else {
                    return cmpr;
                }
            }
        });
        i = 0;
        String[] ret = new String[logs.length];
        for(String[] log : lLogs) 
{
            ret[i++] = join(log, 0, 2, " ");
        }
        for(String log : digitLogs) {
            ret[i++] = log;
        }
        return ret;
    }
    
    private String join(String[] strArray, int begin, int end, String sep) {
        StringBuffer ret = new StringBuffer()
            .append(strArray[begin]);
        for(int i=begin+1; i<end; i++) {
            ret.append(sep).append(strArray[i]);
        }
        return ret.toString();
    }
    
    private boolean isDigitLog(String[] logTokens) {
        for(int i=1; i<logTokens.length; i++) {
            if (logTokens[i].length() == 0) {
                continue;
            }
            char c = logTokens[i].charAt(0);
            if (c >= '0' && c <= '9') {
                return true;
            }
        }
        return false;
        
    }
}
