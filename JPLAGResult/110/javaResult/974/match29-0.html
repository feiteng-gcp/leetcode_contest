class Solution {
    static class log implements Comparable<log>{
        String id,log;
        log(String idd,String logg){
            id=idd;log=logg;
        }
        public int compareTo(log n){
            int c=this.log.compareTo(n.log);
            if(c==0){
                return this.id.compareTo(n.id);
            }
            return c;
        }
    }
    public String[] reorderLogFiles(String[] l) {
        ArrayList<log> s=new ArrayList<>();
        ArrayList<String> d=new ArrayList<>();
        for(String c:l){
            int id=c.indexOf(" ");
            if(c.charAt(id+1)>='0' && c.charAt(id+1)<='9'){
                d.add(c);
            }
            else{
                s.add(new log(c.substring(0,id),c.substring(id+1)));
            }
        }
        Collections.sort(s);
        String ans[]=new String[l.length];
        int i=0;
        for(log c:s){
            ans[i++]=c.id+" "+c.log;
        }
        for(String c:d){
            ans[i++]=c;
        }
        return ans;
    }
}
