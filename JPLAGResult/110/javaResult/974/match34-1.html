class Solution {
    public String[] reorderLogFiles(String[] logs) {
        int n = logs.length;
        if (n == 0) {
            return new String[n];
        }

        String[] res = new String[n];
        
List<String> digits = new ArrayList<>();
        List<String> letter = new ArrayList<>();
        for (String str : logs) {
            int idx = firstIdx(str);
            char c = str.charAt(idx + 1);
            if (Character.isLetter(c)) {
                
letter.add(str);
            }
            else {
                digits.add(str);
            }
        }

        Collections.sort(letter, new Comparator<String>() {
            @Override
            public int compare(String o1, String o2) {
                int id1 = firstIdx(o1);
                int id2 = firstIdx(o2);
                String s1 = o1.substring(id1 + 1);
                String s2 = o2.substring(id2 + 1);
                if (s1.compareTo(s2) != 0) {
                    return s1.compareTo(s2);
                }

                return o1.substring(0, id1).compareTo(o2.substring(0, id2));
            }
        });

        
for (int i = 0; i < letter.size(); i++) {
            res[i] = letter.get(i);
        }

        for (int i = 0; i < digits.size(); i++) {
            res[i + letter.size()] = digits.get(i);
        }

        return res;
    }

    int firstIdx(String str) {
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == ' ') {
                return i;
            }
        }

        return -1;
    }
}
