class Solution {
    class Item {
        String id;
        String content;
        Item(String id, String content) {
            this.id = id;
            this.content = content;
        }
    }
    
public String[] reorderLogFiles(String[] logs) {
        List<Item> ls = new ArrayList<>();
        List<String> ds = new ArrayList<>();
        for (String log : logs) {
            int index = log.indexOf(" ");
            String id = log.substring(0, index);
            String content = log.substring(index + 1);
            if (isLetter(content)) {
                Item i = new Item(id, content);
                
ls.add(i);
            } else {
                ds.add(log);
            }
        }
        Collections.sort(ls, (a, b) -> (a.content.equals(b.content) ? a.id.compareTo(b.id) : a.content.compareTo(b.content)));
        String[] result = new String[logs.length];
        int p = 0;
        
for (Item i : ls) {
            result[p++] = i.id + " " + i.content;
        }
        for (String s : ds) {
            result[p++] = s;
        }
        return result;
    }
    private boolean isLetter(String s) {
        for (char c : s.toCharArray()) {
            if (Character.isLetter(c)) {
                return true;
            }
        }
        return false;
    }
}
