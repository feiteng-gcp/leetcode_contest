class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> num = new ArrayList<String>();
        List<String> ch = new ArrayList<String>();
        for (String s : logs) {
            String[] arr = s.split("\\s+");
            char c = arr[1].charAt(0);
            if (c >= '0' && c <= '9') 
{
                num.add(s);
            } else {
                ch.add(s);
            }
        }
        Collections.sort(ch, new Comparator<String>(){
            @Override
            public int compare(String s1, String s2) {
                String[] arr1 = s1.split("\\s+");
                String[] arr2 = s2.split("\\s+");
                return s1.substring(arr1[0].length() + 1).compareTo(s2.substring(arr2[0].length() + 1));
            }
        });
        int index = 0;
        String[] result = new String[logs.length];
        for (String s : ch) {
            result[index++] = s;
        }
        
for (String s : num) {
            result[index++] = s;
        }
        return result;
    }
}
