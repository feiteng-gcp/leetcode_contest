class Solution {

    public String[] reorderLogFiles(String[] logs) {
        List<String> letterLogs = new ArrayList<>();
        List<String> digitLogs = new ArrayList<>();
        for (String s : logs) {
            String[] split = s.split("\\s+");
            String token = split[1];
            if (Character.isLetter(token.charAt(0))) {
                letterLogs.add(s);
            } else {
                digitLogs.add(s);
            }
        }
        letterLogs.sort((a, b) -> {
            String[] parseA = parseLog(a);
            String[] parseB = parseLog(b);
            if (parseA[1].equals(parseB[1])) {
                return parseA[0].compareTo(parseB[0]);
            } else {
                return parseA[1].compareTo(parseB[1]);
            }
        });
        List<String> result = new ArrayList<>();
        result.addAll(letterLogs);
        result.addAll(digitLogs);
        return result.toArray(new String[0]);
    }

    private String[] parseLog(String log) {
        int index = log.indexOf(" ");
        String[] parse = new String[2];
        parse[0] = log.substring(0, index);
        parse[1] = log.substring(1 + index);
        return parse;
    }
}
