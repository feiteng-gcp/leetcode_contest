class Solution {
    public String[] reorderLogFiles(String[] logs) {
        int len  = logs.length;
        
List<String> digitLogs = new ArrayList<>();
        
        List<String> letterLogs = new ArrayList<>();
        for (String log : logs) {
            String[] logStr = log.split(" ");
            if (Character.isDigit(logStr[1].charAt(0))) {
                digitLogs.add(log);
            } else {
                letterLogs.add(log);
            }
        }
        
        Collections.sort(letterLogs, new LogComparator());
        
        String[] result = new String[len];
        
int count = 0;
        
        for (String cur : letterLogs) {
            result[count++] = cur;
        }
        
        for (String cur : digitLogs) {
            result[count++] = cur;
        }
        return result;
    }
    
    class LogComparator implements Comparator<String> {
        public int compare(String str1, String str2) {
            String[] logs1 = str1.split(" ");
            String[] logs2 = str2.split(" ");
            String id1 = logs1[0];
            String id2 = logs2[0];
            
            int result = str1.substring(id1.length()).compareTo(str2.substring(id2.length()));
            if (result != 0) {
                return result;
            }
            
            return id1.compareTo(id2);
        }
    }
}
