class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> slogs = new ArrayList<>();
        String[] nlogs = new String[logs.length];
        int j = 0;
        for (String s : logs) {
            int i = s.indexOf(" ");
            if (s.charAt(i + 1) < '0' || s.charAt(i + 1) > '9') {
                slogs.add(s);
            }
            else nlogs[j++] = s;
        }
        Collections.sort(slogs, new Comparator<String>() {
            public int compare(String s, String t) {
                int m = s.indexOf(" ");
                int n = t.indexOf(" ");
                return s.substring(m + 1).compareTo(t.substring(n + 1));
            }
        });
        String[] res = new String[logs.length];
        int l = 0;
        for (int k = 0; k < slogs.size(); k++) {
            res[l++] = slogs.get(k);
        }
        for (int k = 0; k < j; k++) {
            res[l++] = nlogs[k];
        }
        return res;
    }
}
