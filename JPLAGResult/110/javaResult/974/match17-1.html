class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> a = new ArrayList<>();
        List<String> n = new ArrayList<>();
        for (String l : logs) {
            if (Character.isAlphabetic(l.charAt(l.length() - 1))) a.add(l);
            else 
n.add(l);
        }
        Collections.sort(a, (l1, l2) -> {
            return helper(l1).compareTo(
helper(l2));
        });
        a.addAll(n);
        return a.toArray(new String[a.size()]);
    }
    
    private String helper(String l) {
        int i = l.indexOf(' ');
        return l.substring(i) + l.substring(0, i);
    }
}
