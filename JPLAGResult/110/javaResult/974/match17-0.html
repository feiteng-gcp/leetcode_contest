class Solution {
   public String[] reorderLogFiles(String[] logs) {
        ArrayList<String> letters = new ArrayList<>();
        ArrayList<String> digits = new ArrayList<>();
        for (int i = 0; i < logs.length; i++) {
            if (isLetterLog(logs[i])) letters.add(logs[i]);
            else digits.add(logs[i]);
        }
        letters.sort((o1, o2) -> {
            int i = o1.indexOf(" ");
            String s1 = o1.substring(i);
            int j = o2.indexOf(" ");
            String s2 = o2.substring(j);
            if(s1.compareTo(s2)==0) return o1.compareTo(o2);
            return s1.compareTo(s2);
        });
        letters.addAll(digits);
        return letters.toArray(new String[letters.size()]);
    }

    private boolean isLetterLog(String log) {
        String[] split = log.split(" ");
        Character s = split[1].toCharArray()[0];
        return !Character.isDigit(s);

    }
}
