class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> dig=new ArrayList<String>();
        List<String[]> letter=new ArrayList<String[]>();
        for(String s:logs){
            int index=isLetter(s);
            if(index==-1){
                dig.add(s);
            }
            else{
                letter.add(new String[]{s.substring(index),s});
            }
        }
        Collections.sort(letter,new Comparator<String[]>(){
            public int compare(String[] s1,String[] s2){
                return s1[0].compareTo(s2[0]);
            }
        });
        String[] res=new String[logs.length];
        int index=0;
        for(String[] s:letter){
            res[index]=s[1];
            index++;
        }
        for(String s:dig){
            res[index]=s;
            index++;
        }
        return res;
    }
    public int isLetter(String s){
        int index=s.indexOf(' ');
        while(index<s.length()){
            char ch=s.charAt(index);
            if(Character.isLetter(ch)){
                return index;
            }
            else if(Character.isDigit(ch)){
                return -1;
            }
            index++;
        }
        return -1;
    }
}
