class Solution {
    public class LogComparator implements Comparator<String> {
        public int compare(String sa1, String sa2) {
            String s1 = sa1.substring(sa1.indexOf(" "));
            String s2 = sa2.substring(sa2.indexOf(" "));
            
String id1 = sa1.split(" ")[0];
            String id2 = sa2.split(" ")[0];
            if(!s1.equals(s2)) {
                return s1.compareTo(s2);
            }
            return id1.compareTo(id2);
        }
    }

    
public String[] reorderLogFiles(String[] logs) {
        ArrayList<String> dlogs = new ArrayList<>();
        ArrayList<String> clogs = new ArrayList<>();
        for(String str : logs) {
            String[] tokens = str.split(" ");
            
if(Character.isDigit((tokens[tokens.length-1].charAt(0)))) {
                dlogs.add(str);
            } else {
                clogs.add(str);
            }
        }
        clogs.sort(new LogComparator());
        clogs.addAll(dlogs);
        return clogs.toArray(new String[]{});
    }
}
