class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> letters = new ArrayList<>();
        List<String> digits = new ArrayList<>();
        
        
       
            
        for(String l : logs) {
            if(l.charAt(l.length() - 1) <= 'z'  && l.charAt(l.length() - 1) >= 'a' ) {
                letters.add(l);
            } else {
                digits.add(l);
            }
        }
        
         String [][] lt = new String[letters.size()][2];
        int j = 0;
        for(String x : letters) {
            int ch = x.indexOf(" ");
            lt[j][0] = x.substring(0, ch);
            lt[j ++ ][1] = x.substring(ch + 1);
        }
        Arrays.sort(lt, (x1, x2) -> {
           if(x1[1].compareTo(x2[1]) == 0) {
               return x1[0].compareTo(x2[0]) ;
           } 
            return x1[1].compareTo(x2[1]) ;
            
        });
        
        
        String[] ans = new String[logs.length];
        int i = 0;
        for(String[] t : lt) {
            ans[i++] = t[0] + " " + t[1];
        }
        for(String t : digits) {
            ans[i++] = t;
        }
        return ans;
    }
}
