class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> digitLogs = new ArrayList<>();
        List<String> letterLogs = new ArrayList<>();
        for(String log : logs){
            String item = log.split(" ")[1];
            if(Character.isDigit(item.charAt(0))){
                digitLogs.add(log);
            }else{
                letterLogs.add(log);
            }
        }
        letterLogs.sort(new Comparator<String>() {
            @Override
            public int compare(String o1, String o2) {
                String identifier1 =  o1.substring(0, o1.indexOf(' '));
                String identifier2 =  o2.substring(0, o2.indexOf(' '));
                String log1 = o1.substring(o1.indexOf(' ') + 1);
                String log2 = o2.substring(o2.indexOf(' ') + 1);
                int res = log1.compareTo(log2);
                if(res != 0){
                    return res;
                }else{
                    return identifier1.compareTo(identifier2);
                }
            }
        });
        letterLogs.addAll(digitLogs);
        return letterLogs.toArray(new String[0]);
    }
}
