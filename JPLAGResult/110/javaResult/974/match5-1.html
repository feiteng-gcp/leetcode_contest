class Solution {
    public String[] reorderLogFiles(String[] logs) {
        String[] ret = new String[logs.length];
        ArrayList<Integer> nums = new ArrayList<>();
        TreeMap<String, Integer> map = new TreeMap<>();
        int index = 0;
        for (String log : logs) {
            if (isNum(log)) {
                nums.add(index++);
                continue;
            }
            map.put(log.substring(log.indexOf(" ")+1), index);
            index++;
        }
        index = 0;
        for (Map.Entry<String, Integer> entry : map.entrySet()) {
            ret[index++] = new String(logs[entry.getValue()]);
        }
        for (int i = 0; i < nums.size(); i++) {
            ret[index++] = new String(logs[nums.get(i)]);
        }
        return ret;
    }
    
    public boolean isNum(String s) {
        return Character.isDigit(s.split(" ")[1].charAt(0));
    }
}
