class Solution {
    public int distinctSubseqII(String S) {
        int mod = 1000000007;
        int[] last = new int[256];
        Arrays.fill(last, -1);
        int n = S.length();
        long[] array = new long[n+1];
        array[0] = 1;
        
for (int i=1; i<=n; i++)
        {
            array[i] = (2*array[i-1])%mod;
            if (last[(int)S.charAt(i-1)] != -1)
                array[i] = (array[i] - array[last[(int)S.charAt(i-1)]])%mod;
                if(array[i] < 0){
                    array[i] = mod + array[i];
                }

            last[(int)S.charAt(i-1)] = (i-1);
        }
        return (int)(array[n]-1);
    }
}
