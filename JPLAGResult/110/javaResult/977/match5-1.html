import java.util.Arrays;


class Solution {
	private static final long M = 1000000007;

	public int distinctSubseqII(String str) {
		int[] last = new int[256];
		Arrays.fill(last, -1);

		int n = str.length();
		long[] dp = new long[n + 1];
		dp[0] = 1;
		for (int i = 1; i <= n; i++) {
			dp[i] = 2 * dp[i - 1];
			if (last[(int) str.charAt(i - 1)] != -1)
				dp[i] = dp[i] - dp[last[(int) str.charAt(i - 1)]];
			last[(int) str.charAt(i - 1)] = (i - 1);
			dp[i] %= M;
		}

		return (int) ((dp[n] - 1 + M * 2) % M);
	}
}
