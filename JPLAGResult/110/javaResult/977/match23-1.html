class Solution {
    public int distinctSubseqII(String str) {
        int MOD = 1000000007;
        int[] last = new int[256];
        Arrays.fill(last, -1);

        int n = str.length();
        int[] dp = new int[n + 1];
        dp[0] = 1;

        for (int i = 1; i <= n; i++) {
            dp[i] = (2 * (dp[i - 1]%MOD))%MOD;
            char c = str.charAt(i - 1);

            if (last[(int) c] != -1) {
                dp[i] = dp[i] - dp[last[(int) c]];
            }
            dp[i] = dp[i]%MOD;
            last[(int) c] = (i - 1);
        }

        return ((dp[n]-1)+MOD)%MOD;
    }
}
