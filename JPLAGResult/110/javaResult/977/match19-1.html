class Solution {
    public int distinctSubseqII(String S) {
        int len = S.length();
        long[] dp = new long[len + 1];
        int[] map = new int[26];
        Arrays.fill(map, -1);
        for (int i = 1; i <= len; ++i) {
            int id = S.charAt(i - 1) - 'a';
            
if (map[id] == -1) {
                dp[i] = (dp[i - 1] * 2 + 1) % 1000000007;
            } else {
                dp[i] = (dp[i - 1] * 2 - dp[map[id] - 1])  % 1000000007;
            }
            map[id] = i;
        }
        return (int)(dp[len] + 1000000007) % 1000000007;
    }
}
