class Solution {
    public int distinctSubseqII(String str) {
         int MAX_CHAR = 256; 
       
        int[] last = new int[MAX_CHAR]; 
        Arrays.fill(last, -1); 
          
        int n = str.length(); 
       
        long[] dp = new long[n+1]; 
       
        dp[0] = 1; 
       int m=(int)1e9+7;
        for (int i=1; i<=n; i++) 
        { 
            dp[i] = (2*dp[i-1])%m; 
       
            if (last[(int)str.charAt(i-1)] != -1) 
                dp[i] = (dp[i] - dp[last[(int)str.charAt(i-1)]]+m)%m; 
       
            last[(int)str.charAt(i-1)] = (i-1); 
        } 
       
        return (int)((dp[n]-1+m)%m); 
    }
}
