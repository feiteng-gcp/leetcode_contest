class Solution {
    int mo=1000000007;
    public int distinctSubseqII(String S) {
        int[] last=new int[26]; 
        Arrays.fill(last,-1); 
        int n=S.length(); 
        int[] f=new int[n+1]; 
        f[0]=1;  
        for (int i=1;i<=n;i++) 
        {  
            int now=S.charAt(i-1)-97;
            f[i]=(f[i-1]<<1)%mo;
            if (last[now]!=-1) f[i]=(f[i]-f[last[now]]+mo)%mo; 
            last[now]=i-1; 
        } 
        return (f[n]-1+mo)%mo; 
    }
}
