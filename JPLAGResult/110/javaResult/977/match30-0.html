class Solution {
    
    static final int MAX_CHAR = 256; 
       
    static int distinctSubseqII(String str){ 
        int mod = 1000000007;
        int[] last = new int[MAX_CHAR]; 
        Arrays.fill(last, -1); 
        int n = str.length(); 
        long[] dp = new long[n + 1]; 
        dp[0] = 1; 
        for (int i = 1 ; i <= n ; i++) { 
            dp[i] = 2 * dp[i - 1];
            if (last[str.charAt(i - 1)] != -1) 
                if(dp[i] < dp[last[str.charAt(i - 1)]]){
                    dp[i] = (dp[i] + mod - dp[last[str.charAt(i - 1)]]) % mod; 
                }else dp[i] = dp[i] - dp[last[str.charAt(i - 1)]]; 
            dp[i] = dp[i] % mod;
            System.out.println(dp[i]);
            last[str.charAt(i - 1)] = (i - 1); 
        } 
        return (int)dp[n] - 1; 
    } 
}
