class Solution {
    
    public int distinctSubseqII(String S) {
        long mod = 1000000007;
        int[] last = new int[26];
        Arrays.fill(last, -1);
        int n = S.length();
        long[] dp = new long[n+1];
        dp[0] = 1l;
        
        for(int i = 1; i <= n; i++){
            dp[i] = 2 * dp[i-1];
            if(last[S.charAt(i-1) - 'a'] != -1){
                dp[i] = dp[i] - dp[last[S.charAt(i-1) - 'a']];
            }
            if(dp[i] < 0){
                dp[i] += mod;
            }
            dp[i] %= mod;
            //System.out.println(dp[i]);
            last[S.charAt(i-1) - 'a'] = (i - 1);
        }
        return (int)((dp[n] - 1) % mod);
    }
}
