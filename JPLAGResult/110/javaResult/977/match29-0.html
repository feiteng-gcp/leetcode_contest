class Solution {
    private static long MOD;
    static {
        MOD = 1;
        for (int i = 0; i < 9; i++) {
            MOD *= 10;
        }
        MOD += 7;
    }
    public int distinctSubseqII(String S) {
        int[] last = new int[256];
        for (int i = 0; i < 256; i++) {
            last[i] = -1;
        }
        int n = S.length();
        long[] cache = new long[n + 1];
        cache[0] = 1;
        for (int i = 1; i < n + 1; i++) {
            
            cache[i] = 2 * cache[i - 1];
            if (last[S.charAt(i - 1)] != -1) {
                cache[i] = (cache[i] - cache[last[S.charAt(i - 1)]]) % MOD;
            }
            last[S.charAt(i - 1)] = i - 1;
        }
        while (cache[n] < 0) {
            cache[n] += MOD;
        }
        int result = (int) (cache[n] - 1);
        return result;
    }
}
