class Solution {
    int MOD = (int)1000000007;
    public int distinctSubseqII(String S) {
        int N = S.length();
        if (N == 0)
            return 0;
        int[] found = new int[26];
        
Arrays.fill(found, -1);
        int[] dp = new int[N+1];
        dp[0] = 1;
        for (int i = 0; i < N; i++) {
            int c_val = S.charAt(i) - 'a';
            dp[i+1] = 2*dp[i] % MOD;
            if (found[c_val] != -1)
                dp[i+1] = (dp[i+1] - dp[found[c_val]] + MOD) % MOD;
            found[c_val] = i;
        }
        return dp[N]-1;
    }
}
