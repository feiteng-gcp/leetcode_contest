class Solution {
    public int distinctSubseqII(String S) {
        int n = S.length(); 
        int[] dp = new int[n + 1]; 
        int[] sum = new int[n + 1];
        int[] hash = new int[26]; 
        Arrays.fill(hash, -1);
        int mod = 1000000007;
        dp[0] = 1; 
        char[] cs = 
S.toCharArray(); 
        for (int i = 1; i <= n; i++) {
            if (hash[cs[i - 1] - 'a'] == -1) {
                dp[i] = sum[i - 1] + 1;
            } else {
                dp[i] = (sum[i - 1] - sum[hash[cs[i - 1] - 'a'] - 1] + mod) % mod;
            }
            if (i == 1) {
                sum[i] = 1; 
            } else {
                sum[i] = sum[i - 1] + dp[i]; 
                sum[i] %= mod; 
            }
            hash[cs[i - 1] - 'a'] = i; 
             // System.out.println(i + " "  + " dp : " + dp[i]);
             // System.out.println(i + " "  + " sum : " + sum[i]);
        }

        return sum[n];
    }
}
