class Solution {
    private final static long MOD = 1000L * 1000L * 1000L + 7;
    public int distinctSubseqII(String S) {
        int alpha = 26;
        int[] prev = new int[alpha];
        Arrays.fill(prev, -1);
        int n = S.length();
        long[] dp = new long[n + 1];
        dp[0] = 1;

        for (int i = 0; i < n; i++) {
            dp[i + 1] = 2 * dp[i];
            dp[i + 1] %= MOD;
            if (prev[S.charAt(i) - 'a'] != -1) {
                dp[i + 1] -= dp[prev[S.charAt(i) - 'a']];
                dp[i + 1] %= MOD;
                if (dp[i + 1] < 0) {
                    dp[i + 1] += MOD;
                }
            }

            prev[S.charAt(i) - 'a'] = i;
        }

        return (int) ((dp[n] - 1 + MOD) % MOD);
    }
}
