/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    
    int rangeSumBST(TreeNode* root, int L, int R) {
        if (root == NULL) {
            return 0;
        }
        int s = rangeSumBST(root->left, L, R);
        if (root->val >= L && root->val <= R) {
            s += root->val;
        }
        s += rangeSumBST(root->right, L, R);
        return s;
    }
};
