/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int rangeSumBST(TreeNode* root, int L, int R) {
        if (root == NULL)
            return 0;
        
        int ret = 0;
        
        if (L <= root->val && root->val <= R)
            ret += root->val;
        
        if (root->val <= R)
            ret += rangeSumBST(root->right, L, R);
        
        if (root->val >= L)
            ret += rangeSumBST(root->left, L, R);
        
        return ret;
    }
};
