/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void traverse(TreeNode* p, int L, int R, int& ret) {
        if (!p) return;
        if (p->val >= L && p->val <= R) ret += p->val;
        traverse(p->left, L, R, ret);
        traverse(p->right, L, R, ret);
    }
    int rangeSumBST(TreeNode* root, int L, int R) {
        int ret = 0;
        traverse(root, L, R, ret);
        return ret;
        
    }
};
