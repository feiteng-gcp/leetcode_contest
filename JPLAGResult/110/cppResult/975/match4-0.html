/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int rangeSumBST(TreeNode* root, int L, int R) {
	    int ret = 0;
	    if(L <= root -> val && root -> val <= R)
		    ret += root -> val;
	    if(root -> left)
		    ret += rangeSumBST(root -> left, L, R);
	    if(root -> right)
		    ret += rangeSumBST(root -> right, L, R);
	    return ret;
    }
};
