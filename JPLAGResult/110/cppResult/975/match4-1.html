/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int rangeSumBST(TreeNode* root, int L, int R) {
        int sum = 0;
        if (root->val <= R && root->val >= L) sum += root->val;
        if (root->left) sum += rangeSumBST(root->left, L, R);
        if (root->right) sum += rangeSumBST(root->right, L, R);
        return sum;
    }
};
