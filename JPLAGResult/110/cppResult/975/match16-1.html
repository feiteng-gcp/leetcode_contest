/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    void dfs(TreeNode* tn,int l,int r,int &ans){
        if(!tn) return;
        dfs(tn->left,l,r,ans);
        if(tn->val>=l&&tn->val<=r) ans+=tn->val;
        dfs(tn->right,l,r,ans);
    }
    int rangeSumBST(TreeNode* root, int L, int R) {
        int ans=0;
        dfs(root,L,R,ans);
        return ans;
    }
};
