class Solution {
    
    bool check(string& l, int i) {
        for (; i < l.size(); i++) {
            if (isdigit(l[i])) return true;
        }
        return false;
    }
    static bool comp(pair<string, string>& s1, pair<string, string>& s2) {
        return s1.second < s2.second;
    }
    
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        
        vector<pair<string, string>> alpha, num;
        
        for (string& l : logs) {
            size_t pos = l.find(" ");
            if (check(l, (int)pos+1)) {
                num.push_back({l, l.substr(pos+1)});
            } else alpha.push_back({l, l.substr(pos+1)});
        }
        //sort(num.begin(), num.end(), comp);
        sort(alpha.begin(), alpha.end(), comp);
        
        vector<string> ans;
        for (auto p : alpha) {
            ans.push_back(p.first);
        }
        for (auto p : num) {
            ans.push_back(p.first);
        }
      
        return ans;
    }
};
