class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {        
        map< pair< string, string >, string > letter_logs;
        vector< string > digit_logs;
        for( string &s: logs ) {
            int pos = s.find( ' ' );
            string id = s.substr( 0, pos );
            if( isdigit( s[ pos + 1 ] ) ) {
                digit_logs.push_back( s );
            } else {
                letter_logs[ make_pair( s.substr( pos + 1 ), id ) ] = s;
            }
        }
        vector< string > reordered;
        for( auto &p: letter_logs ) {
            reordered.push_back( move( p.second ) );
        }
        reordered.insert( reordered.end(),
                          make_move_iterator( digit_logs.begin() ),
                          make_move_iterator( digit_logs.end()   ) );
        return reordered;
    }
};
