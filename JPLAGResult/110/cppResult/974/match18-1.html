class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> letterLogs, digitsLogs;
        for (auto l : logs) {

            if (isdigit(l.back())) {
                digitsLogs.push_back(l);
            } else {
                letterLogs.push_back(l);
            }
        }
      /*  
        for (auto ele : letterLogs) {
            printf("%s\t", ele.c_str());
        }
        printf("\n");
        */
        sort(letterLogs.begin(), letterLogs.end(), cmp);
        /*
        for (auto ele : letterLogs) {
            printf("%s\t", ele.c_str());
        }
        printf("\n");
        */
        
        for (auto ele : digitsLogs) {
            letterLogs.push_back(ele);
        }
        return letterLogs;
    }
    static bool cmp(const string& s1, const string& s2) {
        string new1, new2;
        auto size_t1 = s1.find_first_of(" "), size_t2 = s2.find_first_of(" ");
        new1 = s1.substr(size_t1 + 1);
        new2 = s2.substr(size_t2 + 1);
        //printf("new1 = %s, new2 = %s\n", new1.c_str(), new2.c_str());
        return new1 < new2;
    }
};
