class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        map<string, string> mp;
        vector<string> ds;
        for (auto & log : logs) {
            int idx = 0;
            int len = log.size();
            while (log[idx] != ' ')
                idx++;
            idx++;
            if (isdigit(log[idx]))
                ds.push_back(log);
            else {
                string str = log.substr(idx) + log.substr(0, idx - 1);
                mp[str] = log;
            }
        }
        vector<string> ans;
        for (auto & e : mp)
            ans.push_back(e.second);
        for (auto & str : ds)
            ans.push_back(str);
        return ans;
    }
};
