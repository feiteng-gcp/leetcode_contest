bool cmp(pair<string, string> a, pair<string, string> b) {
    return a.second < b.second || (a.second == b.second && a.first < b.first);
}


class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<pair<string, string>> letter;
        vector<string> num;
        for (string s : logs) {
            int i = 0;
            while (s[i] != ' ') ++i;
            ++i;
            if (s[i] >= '0' && s[i] <= '9') {
                num.push_back(s);
            } else {
                letter.push_back({s.substr(0, i - 1), s.substr(i)});
            }
        }
        
        sort(letter.begin(), letter.end(), cmp);
        vector<string> res;
        for (auto pr : letter) {
            res.push_back(pr.first + " " + pr.second);
        }
        for (string n : num) res.push_back(n);

        return res;
    }
};
