class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> res, digs;
        map<string, string> mp;
        for (string s: logs) {
            stringstream sin(s);
            string str0, str;
            sin >> str0;
            sin >> str;
            if (isdigit(str[0])) {
                digs.push_back(s);
            } else {
                mp[s.substr(str0.length())] = s;

            }
        }
        for (auto item : mp) {
            res.push_back(item.second);
        }
        for (auto s: digs) {
            res.push_back(s);
        }
        return res;
    }
};
