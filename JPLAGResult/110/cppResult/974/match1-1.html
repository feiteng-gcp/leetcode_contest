bool cmp( string & a, string & b ){
    stringstream left(a);
    stringstream right(b) ; 
    string l,r ;
    left>>l ; right>>r;
    while( 1 ) {
        left>>l ; right>>r;
        if( l > r ) 
            return false  ;
        else if( l < r ) 
            return true ;
    }
    return true  ; 
}
class Solution {
public:
    bool legal( string s ){
        stringstream ss (s) ;
        string t ;
        ss >> t ; 
        ss >> t ;
        if( isdigit( t[0] ) )
            return true ;
        return false ; 
    }

    vector<string> reorderLogFiles(vector<string>& logs) {
        vector< string > right ;
        vector< string > left ; 
        for ( auto &s : logs ) {
            if( legal( s ) ) 
                right.push_back( s ) ; 
            else
                left.push_back( s ) ; 
        }
        sort( left.begin() , left.end() , cmp ) ; 
        for( auto &x:right)
            left.push_back( x ) ; 
        return left ;
        
    }
};
