class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        map<string, string> mp;
        vector<string>digit;
        for(auto s:logs){
            int blank=-1;
            bool dig = true;
            for(int i = 0; i < s.size(); i++){
                char c = s[i];
                if(blank== -1 && c==' ')
                    blank = i+1;
                else if(blank!=-1){
                    if(c >='a' && c <= 'z'){
                        dig = false;
                    }
                    break;
                }
            }
            if(dig){
                digit.push_back(s);
            }else{
                mp[s.substr(blank)] = s;

            }
        }
        vector<string>res;
        for(auto i:mp){
            res.push_back(i.second);
        }
        for(auto i:digit){
            res.push_back(i);
        }
        return res;
    }
};
