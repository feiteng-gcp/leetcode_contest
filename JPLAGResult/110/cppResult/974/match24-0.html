class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        map<string, int> s2i;
        vector<int> digs;
        for (int i=0; i<logs.size(); i++){
            if (!CheckLetter(logs[i], s2i, i)) {
                digs.push_back(i);
            }
        }
        vector<string> res;
        for (auto p: s2i) {
            res.push_back(logs[p.second]);
        }   
        for (auto i: digs) {
            res.push_back(logs[i]);
        }
        return res;
    }
    
    bool CheckLetter(string & s, map<string, int> & s2i, int idx) {
        string tag = "";
        for (char c: s){
            if (c!=' ') tag+=c;
            else break;
        }
        if (isdigit(s[tag.size()+1])) return false;
        string tmp = s.substr(tag.size()+1) + tag;
        s2i[tmp] = idx;
        return true;
    }
};
