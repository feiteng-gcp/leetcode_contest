class Solution {
public:
static bool cmp(const string&a, const string&b)
{
    string ta = a.substr(a.find_first_of(' '));
    string tb = b.substr(b.find_first_of(' '));
    
    return ta < tb;
}
vector<string> reorderLogFiles(vector<string>& logs) 
{
    vector<string> num, letter;

    for(auto &i:logs)
    {
        int j = i.size() - 1;
        while (i[j]==' ')
        {
            j--;
        }
        if(i[j]>='0'&&i[j]<='9')
        {
            num.push_back(i);
        }
        else
        {
            letter.push_back(i);
        }
    }
    sort(letter.begin(), letter.end(), cmp);
    
     for (auto&i : num)
        {
            letter.push_back(i);
        }
    return letter;
}
};
