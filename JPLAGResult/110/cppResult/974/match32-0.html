class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> a, b;
        for (auto s : logs) {
            if (s.back() >= '0' && s.back() <= '9') {
                b.push_back(s);
            } else {
                a.push_back(s);
            }
        }
        sort(a.begin(), a.end(), [](string s, string t) {
            return s.substr(s.find(" ")) < t.substr(t.find(" "));
        });
        a.insert(a.end(), b.begin(), b.end());
        return a;
    }
};
