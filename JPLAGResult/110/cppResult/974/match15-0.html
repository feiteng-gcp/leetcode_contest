template<class T>
constexpr T INF = ::numeric_limits<T>::max() / 32 * 15 + 208;

// class
class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<pair<string, string>> a, b;
        vector<string> ret;
        for (auto &&s : logs) {
            int p = 0;
            string S, T;
            int i = 0;
            while(s[i] != ' ') i++;
            S = s.substr(0, i);
            T = s.substr(i+1, 1000);
            if('0' <= T[0] && T[0] <= '9') b.emplace_back(T, S);
            else a.emplace_back(T, S);
        }
        sort(a.begin(), a.end());
        for (auto &&s : a) {
            ret.emplace_back(s.second + ' ' + s.first);
        }
        for (auto &&s : b) {
            ret.emplace_back(s.second + ' ' + s.first);
        }
        return ret;
    }
};
