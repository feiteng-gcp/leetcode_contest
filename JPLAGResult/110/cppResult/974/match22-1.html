class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<pair<string, string>> letter;
        vector<string> digits;
        for (string&s: logs) {
            int i = s.find(' ');
            string id = s.substr(0,i);
            string content = s.substr(i+1,s.size()-i-1);
            if (isdigit(content[0]))
                digits.push_back(s);

            else
                letter.push_back({content,id});
        }
        sort(letter.begin(), letter.end());
        vector<string> result;
        for (auto &s: letter) {
            result.push_back(s.second + " " + s.first);
        }
        for (auto &s: digits) {
            result.push_back(s);
        }
        
        return result;
    }
    
};
