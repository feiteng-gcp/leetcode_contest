class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> letters, digits;
        for(string s: logs){
            int p = s.find_first_of(" ");
            if(isdigit(s[p+1])){
                digits.push_back(s);
            }else{
                letters.push_back(s);
            } 
        }
        sort(letters.begin(), letters.end(), [](string &a, string& b){
            int pa = a.find_first_of(" "), pb = b.find_first_of(" ");
            return a.substr(pa+1) < b.substr(pb+1);
        });
        for(string s: digits){
            letters.push_back(s);
        }
        return letters;
    }
};
