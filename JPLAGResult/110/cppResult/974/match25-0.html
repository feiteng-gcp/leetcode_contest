class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> letters;
        vector<string> digits;
        unordered_map<string, vector<string>> imap;
        for(string log: logs){
            int i= 0;
            while(log[i]!=' ') i++;
            i++;
            if(log[i]>='0'&& log[i]<='9') digits.push_back(log);
            else {
                imap[log.substr(i, log.size()-1)].push_back(log.substr(0, i));
                letters.push_back(log.substr(i, log.size()-1));
            }
        }
        sort(letters.begin(), letters.end());
        for(int i = 0; i<letters.size(); i++){
            string l = letters[i];
            letters[i] = imap[l].back()+l;
            imap[l].pop_back();
        }
        for(string log: digits) letters.push_back(log);
        return letters;
    }
};
