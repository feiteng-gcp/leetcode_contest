class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> d;
        vector<pair<string, string>> l;
        for (const string& log : logs) {
            string id = "";
            string data = "";
            bool flag = false;
            for (int i = 0; i < log.length(); ++i) {
                if (log[i] == ' ' && !flag) {
                    flag = true;
                    continue;
                }
                if (flag) {
                    data += log[i];
                } else {
                    id += log[i];
                }
            }

            if (isdigit(data[0])) {
                d.push_back(id + " " + data);
            } else {
                l.push_back(make_pair(data, id));
            }
        }
        sort(l.begin(), l.end());
        vector<string> result;
        for (const auto& x : l) {
            result.push_back(x.second + " " + x.first);
        }
        for (const auto& x : d) {
            result.push_back(x);
        }
        return result;
    }
};
