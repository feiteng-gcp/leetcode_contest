class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> num;
        vector<string> res;
        
        map<string, set<string>> m;
        
        for(string& s: logs)
        {
            int firstSpace = s.find_first_of(" ");
            string tag = s.substr(0, firstSpace);
            string remain = s.substr(firstSpace+1);
            if(isdigit(remain[0]))
            {
                num.push_back(s);
            }
            else
            {
                m[remain].insert(tag);
            }
        }
        for(auto& p: m)
        {
            for(string s: p.second)
            {
                res.push_back(s+" " +p.first);
            }
        }
        for(string& s: num)
            res.push_back(s);
        
        return res;
    }
};
