class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> res;
        vector<string> digits;
        
        for (auto& s : logs) {
            if (isLetter(s)) {
                res.push_back(s);
            } else {
                digits.push_back(s);
            }
        }
        
        auto cmp = [&](const string& l, const string& r) {
            auto s1 = getLog(l);
            auto s2 = getLog(r);
            
            return s1 == s2 ? l < r : s1 < s2;
        };
        
        sort(res.begin(), res.end(), cmp);
        
        for (auto& s : digits) {
            res.push_back(s);
        }
        
        return res;
    }
    
private:
    bool isLetter(const string& s) {
        auto words = split(s, "     \n\t");
        
        auto word = words[1];
        if (isdigit(word[0])) {
            return false;
        }
        
        return true;
    }
    
    
    string getLog(const string& s) {
        string res;
        auto p = s.find_first_of(" ");
        res = s.substr(p + 1);
        // cout << s << " |" << res << "|" << endl;
        return res;
    }
    
    vector<string> split(const string& s, const string& delimiters) {
        vector<string> res;
        int n = s.length();
        
        for (int i = 0; i < n; ++i) {
            auto j = s.find_first_of(delimiters, i);
            if (j == string::npos) {
                j = n;
            }
            if (j - i > 0) {
                res.emplace_back(s.substr(i, j - i));
            }
            i = j;
        }
        
        return res;
    }
};
