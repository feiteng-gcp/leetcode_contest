#define all(c) (c).begin(),(c).end()
#define tr(c,i) for(typeof((c).begin()) i = (c).begin(); i != (c).end(); i++)
#define ll long long
#define sz(a) int((a).size())
#define pb push_back
#define SORT(x) sort((x).begin(),(x).end());
#define VI vector<int>
#define VII vector < vector <int> >
#define MP make_pair
#define SET(x,byte) memset(x,byte,sizeof(x));
#define I(x) ((x) - 'A')
#define REP(i,mx) for(int i=0;(i)<(mx);i++)
#define FOR(i,mn,mx) for(int i=mn;(i)<=(mx);i++)
#define U(x) (1<<(x))
#define INF 2000000007
#define PI acos(-1.0)
#define endl '\n'
class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<pair<string, string>> let;
        vector<pair<string, string>> dig;
        
        for(string &s: logs) {
            stringstream ss(s);
            string id;
            ss>>id;
            string temp;
            ss>>temp;

            bool isDigit = true;
            REP(i, sz(temp)) {
                if(temp[i] < '0' || temp[i] > '9') {
                    isDigit = false;
                    break;
                }
            }
            if(isDigit) {
                dig.pb(MP(s.substr(sz(id)), id));
            } else let.pb(MP(s.substr(sz(id)), id));
        }
        
        SORT(let);
        vector<string> vs;
        REP(i, sz(let)) {
            vs.pb(let[i].second + let[i].first);
        }
        REP(i, sz(dig)) {
            vs.pb(dig[i].second + dig[i].first);
        }
        
        return vs;
    }
};
