class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<pair<string, string>> letter_logs;
        vector<string> digit_logs, ret;
        for (auto log : logs) {
            string identifier = "";
            int len = log.length();
            for (int i = 0; i < len; ++i) {
                if (log[i] == ' ') {
                    if (isdigit(log[i + 1])) {
                        digit_logs.push_back(log);
                    } else {
                        letter_logs.push_back(make_pair(log.substr(i + 1, len), identifier));
                    }
                    break;
                }
                identifier += log[i];
            }
        }
        sort(letter_logs.begin(), letter_logs.end());
        for (auto log : letter_logs) {
            ret.push_back(log.second + ' ' + log.first);
        }
        for (auto log : digit_logs) {
            ret.push_back(log);
        }
        return ret;
    }
};
