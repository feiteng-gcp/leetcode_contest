class Solution {
public:
    static bool isDigit(char c) { return c<='9' && c>='0'; }
    
    static bool cmp(string a, string b) {
        string t1, t2;
        for (int i = 0; i < a.size(); i ++) {
            if (a[i] == ' ') {
                t1 = a.substr(i + 1, a.size() - i - 1);
                break;
            }
        }
        for (int i = 0; i < b.size(); i ++) {
            if (b[i] == ' ') {
                t2 = b.substr(i + 1, b.size() - i - 1);
                break;
            }
        }
        if (isDigit(t1[0]) && !isDigit(t2[0])) return false;
        if (isDigit(t2[0])) return true;
        return t1 < t2;
    }
    
    vector<string> reorderLogFiles(vector<string>& logs) {
        int n = logs.size();
        for (int i = n; i > 0; i --)
            for (int j = 0; j + 1 < i; j ++)
                if (!cmp(logs[j], logs[j+1])) swap(logs[j], logs[j+1]);
        return logs;
    }
};
