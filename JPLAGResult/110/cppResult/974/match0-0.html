class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> ans,tmp;
        vector<pair<string,string>> vec;
        for(auto s:logs){
            int idx=s.find(" ")+1;
            if(isdigit(s[idx]))
                tmp.push_back(s);
            else vec.push_back(make_pair(s.substr(idx),s.substr(0,idx)));
        }
        sort(vec.begin(),vec.end());
        for(auto p:vec){
            ans.push_back(p.second+p.first);
        }
        for(auto s:tmp) ans.push_back(s);
        return ans;
    }
};
