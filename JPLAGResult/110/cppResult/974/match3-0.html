class Solution {
 public:
  vector<string> reorderLogFiles(vector<string>& logs) {
    vector<string> dig, let;
    for (string& s : logs) {
      int p = s.find(' ');
      if (isdigit(s[p + 1])) {
        dig.push_back(s);
      } else {
        let.push_back(s);
      }
    }
    sort(let.begin(), let.end(),
         [](const string& lhs, const string& rhs) -> bool {
           int p = lhs.find(' '), q = rhs.find(' ');
           if (lhs.substr(p + 1) != rhs.substr(q + 1)) {
             return lhs.substr(p + 1) < rhs.substr(q + 1);
           }
           return lhs.substr(0, p) < rhs.substr(0, q);
         });
    let.insert(let.end(), dig.begin(), dig.end());
    return let;
  }
};
