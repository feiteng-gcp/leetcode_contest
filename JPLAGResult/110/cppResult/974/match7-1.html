class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector <string> num, word;
        for (auto cur : logs) {
            stringstream temp(cur);
            string s;
            temp >> s;
            temp >> s;
            if (isalpha(s[0])) {
                word.push_back(cur);
            } else {
                num.push_back(cur);
            }
        }
        sort(word.begin(), word.end(), [](string a, string b) {
        stringstream temp(a);
        string la;
        temp >> la;
        for (int i = 0; i < a.size(); i++) {
            if (a[i] == ' ') {
                a = a.substr(i + 1);
                break;
            }
        }
        
        stringstream temp_(b);
        string lb;
        temp_ >> lb;
        for (int i = 0; i < b.size(); i++) {
            if (b[i] == ' ') {
                b = b.substr(i + 1);
                break;
            }
        }
        
        if (a == b) {
            return la < lb;
        } else {
            return a < b;
        }        
    });
        for (auto cur : num) {
            word.push_back(cur);
        }
        return word;
    }
};
