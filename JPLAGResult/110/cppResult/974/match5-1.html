class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> ans;
        vector<string> num;
        for(auto &s:logs)
        {
            string t1,t2;
            stringstream ss;
            ss<<s;
            ss>>t1>>t2;
            if(t2[0]>='0'&&t2[0]<='9')
            {
                num.push_back(s);
            }
            else
            {
                ans.push_back(s);
            }
        }
        sort(ans.begin(),ans.end(),cmp);
        for(auto &s:num)ans.push_back(s);
        return ans;
    }
    
    static bool cmp(const string& a,const string& b)
    {
        string aa,bb;
        stringstream ss;
        ss<<a;
        string tmp,tmp1,tmp2;ss>>tmp1;
        ss>>aa;
        while(ss>>tmp)aa+=" "+tmp;
        //aa=ss.str();
        stringstream ss2;
        ss2<<b;
        ss2>>tmp2;
        ss2>>bb;
        while(ss2>>tmp)bb+=" "+tmp;
        //bb=ss.str();
        //cout<<aa<<"|"<<bb<<endl;
        if(aa!=bb)return aa<bb;
        return tmp1<tmp2;
    }
};
