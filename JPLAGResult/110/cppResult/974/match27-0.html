class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector <pair <pair <string, string>, string> > v1;
        vector <string> v2;
        for (string log : logs) {
            int pos = 0;
            for (int i = 0; i < log.size(); i++) {
                if (log[i] == ' ') {
                    pos = i;
                    break;
                }
            }
            
            if (log[pos + 1] >= '0' && log[pos + 1] <= '9') {
                v2.push_back(log);
            } else {
                v1.push_back(make_pair(make_pair(log.substr(pos + 1), log.substr(0, pos)), log));
                //cout << v1.back().first.first << endl << v1.back().first.second << endl;
            }
        }
        
        sort(v1.begin(), v1.end());
        vector <string> ans;
        for (int i = 0; i < v1.size(); i++) {
            ans.push_back(v1[i].second);
        }
        
        for (int i = 0; i < v2.size(); i++) {
            ans.push_back(v2[i]);
        }
        
        return ans;
    }
};
