class Solution {
    bool IsDigitLog(const string& log) {
        istringstream ss(log);
        string s;
        ss >> s;
        ss >> s;
        return isdigit(s[0]);
    }
    static bool MyComp(const string& logA, const string& logB) {
        auto spaceAIdx = logA.find_first_of(' ');
        auto spaceBIdx = logB.find_first_of(' ');
        auto notSpaceAIdx = logA.find_first_not_of(' ', spaceAIdx);
        auto notSpaceBIdx = logB.find_first_not_of(' ', spaceBIdx);
        auto subA = logA.substr(notSpaceAIdx);
        auto subB = logB.substr(notSpaceBIdx);
        return subA < subB;
    }
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> vecDigitLogs;
        vector<string> vecAlphaLogs;
        vector<string> output;
        for (const auto& s : logs) {
            if (IsDigitLog(s)) {
                vecDigitLogs.push_back(s);
            } else {
                vecAlphaLogs.push_back(s);
            }
        }
        sort(vecAlphaLogs.begin(), vecAlphaLogs.end(), MyComp);
        for (const auto& s : vecAlphaLogs) {
            output.push_back(s);
        }
        for (const auto& s : vecDigitLogs) {
            output.push_back(s);
        }
        return output;
    }
};
