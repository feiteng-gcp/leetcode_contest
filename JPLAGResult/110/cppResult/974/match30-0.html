class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<pair<vector<string>, int>> letters;
        vector<int> digits;
        for (int i = 0;i < logs.size();++i) {
            auto pos = logs[i].find(' '), pos_last = pos;
            if (logs[i][pos + 1] >= '0' && logs[i][pos + 1] <= '9') {
                digits.emplace_back(i);
                continue;
            }
            letters.emplace_back();
            letters.back().second = i;
            letters.back().first.emplace_back(logs[i].substr(0, pos));
            while (true) {
                pos = logs[i].find(' ', pos_last + 1);
                if (pos == string::npos) {
                    letters.back().first.emplace_back(logs[i].substr(pos_last + 1));
                    break;
                }
                letters.back().first.emplace_back(logs[i].substr(pos_last + 1, pos - pos_last));
                pos_last = pos;
            }
        }
        sort(letters.begin(), letters.end(), [](const pair<vector<string>, int>&a,
                                               const pair<vector<string>, int>&b){
            for (int i = 1;i < a.first.size() && i < b.first.size();++i) {
                if (a.first[i] != b.first[i]) {
                    return a.first[i] < b.first[i];
                }
            }
            if (a.first.size() != b.first.size()) {
                return a.first.size() < b.first.size();
            }
            return a.first[0] < b.first[0];
        });
        vector<string> result;
        for (auto&& l : letters) {
            result.emplace_back(logs[l.second]);
        }
        for (auto&& d : digits) {
            result.emplace_back(logs[d]);
        }
        return result;
    }
};
