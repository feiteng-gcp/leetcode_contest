class Solution {
public:
    vector<string> reorderLogFiles(vector<string>& logs) {
        vector<string> letters;
        vector<string> digs;

        for(string s:logs){
            int t=s.find(' ');
            if(s[t+1]>='0'&&s[t+1]<='9'){
                digs.push_back(s);
            }else{
                letters.push_back(s);
            }
        }
        
        sort(letters.begin(),letters.end(),[](string a,string b){
            int t1=a.find(' ');
            int t2=b.find(' ');
            string s1=a.substr(t1+1);
            string s2=b.substr(t2+1);
            if(s1==s2) return a<b;
            else return s1<s2;
        });
        
        for(string s:digs) letters.push_back(s);
        return letters;
    }
};
