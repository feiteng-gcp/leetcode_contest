class Solution {
public:

	const int mod = 1e9 + 7;
	int distinctSubseqII(string str) {
		vector<int> last(256, -1); 
		int n = str.length(); 
		int dp[n+1]; 
		dp[0] = 1; 
		for (int i=1; i<=n; i++){ 
			dp[i] = 2*dp[i-1] % mod; 
			if (last[str[i-1]] != -1) 
				dp[i] = dp[i] + mod - dp[last[str[i-1]]];
			dp[i] %= mod;
			last[str[i-1]] = (i-1); 
		} 
		return (dp[n] + mod - 1) % mod; 
	}
};
