class Solution {
public:
    int distinctSubseqII(string S) {
        vector<int> global(S.size()+1, 0), local(S.size()+1, 0);

        unordered_map<char, int> visited;
        for(int i = 0; i < S.size(); i++){
            if(visited.count(S[i]) == 0){
                local[i+1] = global[i] + 1;
                global[i+1] = local[i+1] + global[i];
                visited[S[i]] = i;
            }else{
                local[i+1] = global[i] + 1;
                global[i+1] = global[i] + local[i+1] - local[visited[S[i]] + 1];
                visited[S[i]] = i;
            }
            local[i+1] = local[i+1]%1000000007;
            if(global[i+1] < 0) global[i+1] += 1000000007;
            global[i+1] = global[i+1]%1000000007;
        }
        return global.back();
    }
};
