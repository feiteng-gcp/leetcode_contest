class Solution {
public:
    int MOD = 1000000007;
    
    int distinctSubseqII(string S) {
        vector<int> last(26,-1); 
        int n = S.length(); 
        int dp[n+1]; 
        dp[0] = 1; // empty subsequence
  
        for(int i=1;i<=n;i++) { 
            dp[i] = 2*dp[i-1]; 
            dp[i]%=MOD;
            
            if(last[S[i-1]-'a'] != -1) 
                dp[i] = dp[i] - dp[last[S[i-1]-'a']]; 
            
            if(dp[i]<0) dp[i]+=MOD;
            
            last[S[i-1]-'a'] = (i-1); 
        } 
        
        int ans = dp[n]-1; // (-1) for the non-empty subsequence.
        if(ans<0) ans+=MOD;
        
        return ans;   
    }
};
