class Solution {
public:
    int distinctSubseqII(string str) 
    { 
        int MOD= 1000000007;
        vector<long long int> last(256, -1); 
        int n = str.length(); 
        long long int dp[n+1]; 
        dp[0] = 1; 
        for (int i=1; i<=n; i++) { 
            dp[i] = (2*dp[i-1])%MOD; 
            if (last[str[i-1]] != -1) 
                dp[i] = (dp[i] - dp[last[str[i-1]]])%MOD;
            dp[i] = (dp[i]+MOD)%MOD;
            last[str[i-1]] = (i-1); 
        } 
        return (dp[n]%MOD-1); 
    } 
};
