class Solution {
public:
    int distinctSubseqII(string S) {
        int n = S.size();
        long long mod = 1e9+7;
        unordered_map<char, int> mp;
        vector<long long> f(n+1, 0);

        f[0] = 1;
        for (int i = 0; i < n; ++i) {
            f[i+1] = f[i]*2;
            if (mp.count(S[i]))
                f[i+1] += mod-f[mp[S[i]]-1];
            f[i+1] = (f[i+1]+mod) % mod;
            mp[S[i]] = i+1;
        }
        // for (int i = 0; i < n+1; ++i)
            // cout << f[i] << endl;
        return f[n]-1;
    }
};
