class Solution {
public:
    int distinctSubseqII(string a)
{
    int M = 1000000007;
    long long SubArray[2005] = {0};
    int LastIndex[2005] = {0};
    int iter = 0;
    int n = a.length();
    for(iter = 1; iter <= n; iter++)
    {
        switch(LastIndex[a[iter - 1]])
        {
        case 0:
            {
                SubArray[iter] = (2 * SubArray[iter - 1] + 1) % M;
                break;
            }
        default:
            {
                SubArray[iter] = (2 * SubArray[iter - 1] - SubArray[LastIndex[a[iter - 1]] - 1] + M) % M;
                break;
            }
        }
        LastIndex[a[iter - 1]] = iter;
    }
    return SubArray[n] % M;
}
};
