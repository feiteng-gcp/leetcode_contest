class Solution {
public:
    int distinctSubseqII(string S) {
        int n = S.length();
        long long SubArray[2002] = {0};
        int LastIndex[27] = {0};
        int iter = 0;
        for(iter = 1; iter <= n; iter++)
        {
            if(LastIndex[S[iter - 1]-'a']==0)
                SubArray[iter] = (2 * SubArray[iter - 1] + 1) % 1000000007;
            else
                SubArray[iter] = (2 * SubArray[iter - 1] - SubArray[LastIndex[S[iter - 1]-'a'] - 1] + 1000000007) % 1000000007;
            LastIndex[S[iter - 1]-'a'] = iter;
        }
        
        return SubArray[n] % 1000000007;
    }
};
