class Solution {
public:
    int distinctSubseqII(string str) 
    {

        vector<long> last(256, -1); 
        long M = 1e9+7;
  
        int n = str.length(); 
  
        int dp[n+1]; 
  
        dp[0] = 1; 
  
        for (int i=1; i<=n; i++) 
        { 
            dp[i] = 2*dp[i-1]; 
            dp[i] = dp[i]%M;
  
            if (last[str[i-1]] != -1) 
            {
                dp[i] = (dp[i]+M) - dp[last[str[i-1]]]; 
                dp[i] = dp[i]%M;
            }
                
  
            last[str[i-1]] = (i-1); 
        } 
  
        return dp[n]-1; 
    }
};
