class Solution {
public:
    int distinctSubseqII(string S) {
        int mod = 1000000007;

        vector<int> last(26, -1); 

        int n = S.length(); 

        int dp[n+1]; 

        dp[0] = 1; 

        for (int i=1; i<=n; i++) 
        { 
            dp[i] = (2*dp[i-1] % mod + mod) % mod; 
            if (last[S[i-1] - 'a'] != -1) 
                dp[i] = (dp[i] - dp[last[S[i-1] - 'a']] + mod) % mod;
            last[S[i-1] - 'a'] = (i-1); 
        } 

        return dp[n] - 1; 
    }
};
