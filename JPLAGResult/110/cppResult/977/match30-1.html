class Solution {
public:
    int distinctSubseqII(string S) {
        int n = S.length();
        
        int M = 1e9 + 7;
        
        // vector<vector<vector<int>>> dp(n + 1, vector<vector<long long>>(26, vector<int>(n + 2, 0)));
        // vector<long long> dp2(26, 0);
        vector<vector<long long>> dp(26, vector<long long>(n + 2, 0));
        vector<vector<long long>> nxt = dp;
        
        for (int i = 1; i <= n; ++i) {
            char c = S[i - 1] - 'a';
            
            nxt[c][0] = 0;
            for (int j = c; j <= c; ++j) {
                for (int k = 1; k <= n; ++k) {
                    // if (j != c) {
                    //     nxt[j][k] = dp[j][k];
                    //     continue;
                    // }
                    
                    
                    if (k == 1) {
                        nxt[j][k] = 1;
                        for (int l = 0; l < 26; ++l) {
                            if (l != j) {
                                nxt[j][k] = (nxt[j][k] + dp[l][n]) % M;
                            }
                        }
                    } else {
                        nxt[j][k] = (dp[j][k - 1] - dp[j][k - 2] + M) % M;
                    }
                    
                    // cout << i << " " << j << " " << k << " " << nxt[j][k] << endl;
                }
            }
            

            for (int k = 1; k <= n; ++k) {
                nxt[c][k] = (nxt[c][k]  + nxt[c][k - 1]) % M;
            }
            
            for (int k = 0; k <= n; ++k) {
                dp[c][k] = nxt[c][k];

            }
        }
        
        
        long long res = 0;
        for (int i = 0; i < 26; ++i) {
            res = (res + dp[i][n]) % M;
        }
        
        return res;
    }
};
