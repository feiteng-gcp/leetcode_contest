class Solution {
public:
   int distinctSubseqII(string val) {
      int n; //scanf("%d", &n);
      n = val.size();
      map<int, int> mp;
      typedef long long ll;
      vector<ll> dp(n + 1, 0);
    /*for(int i = 1; i <= n; i++){
        scanf("%d" ,&val[i]);
    } */
       const int mod = 1e9 + 7;

    dp[0] = 1;
    for(int i = 1; i <= n; i++){
        dp[i] = dp[i - 1] << 1, dp[i] %= mod;

        if(mp[val[i - 1]]) {
            dp[i] -= dp[mp[val[i - 1]] - 1];
            dp[i] = (dp[i] % mod + mod) % mod;
        }
        mp[val[i - 1]] = i;
     }
     //printf("%d\n", dp[n] - 1);
    return dp[n] - 1;
}
};
