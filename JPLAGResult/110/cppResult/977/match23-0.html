#define MOD 1000000007
class Solution {
public:
    int distinctSubseqII(string S) {
        vector<int> dp(S.size() + 1, 1);
        vector<int> last(256, -1);
        for (int i = 0; i < S.size(); i++) {
            dp[i + 1] = dp[i] * 2 % MOD;
            if (last[S[i]] != -1)
                (dp[i + 1] += MOD - dp[last[S[i]]]) %= MOD;
            last[S[i]] = i;
        }
        return (dp.back() + MOD - 1) % MOD;
    }
};
