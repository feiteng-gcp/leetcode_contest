class Solution {
public:
    int distinctSubseqII(string S) {
        const long mod = 1000000007;
        vector<int> last(256, -1);

        vector<long> dp(S.size() + 1);
        dp[0] = 1;
        for (int i = 1; i <= S.size(); i++) {
            dp[i] = 2 * dp[i - 1];
            if (last[S[i - 1]] != -1)
                dp[i] -= dp[last[S[i - 1]]];
            last[S[i - 1]] = i - 1;
            dp[i] = (dp[i] % mod + mod) % mod;
        }
        return (dp[S.size()] - 1 + mod) % mod;
    }
};
