class Solution {
public:
    int distinctSubseqII(string s) {
        const int mod = 1e9 + 7;
        vector<long long> f(s.size() + 1);
        f[0] = 1;
        for (int i = 1; i <= s.size(); i++) {
            for (int j = 0; j < i; j++) {
                f[i] = (f[i] + f[j]) % mod;
            }
            for (int j = i - 1; j >= 1; j--) {
                if (s[j - 1] == s[i - 1]) {
                    f[i] = (f[i] - f[j] + mod) % mod;
                }
            }
        }
        long long ans = 0;
        for (int i = 1; i <= s.size(); i++) {
            ans = (ans + f[i]) % mod;
        }
        return ans;
    }
};
