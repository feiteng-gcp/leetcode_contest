#define MOD 1000000007

class Solution {
public:
    int distinctSubseqII(string S) {
   //     S="abc";
    //    S="zchmliaqdgvwncfatcfivphddpzjkgyygueikthqzyeeiebczqbqhdytkoawkehkbizdmcnilcjjlpoeoqqoqpswtqdpvszfaksn";

        unsigned long count[S.size() + 1];
        vector<int> last(26, -1);
        count[0] = 1;
        for (int i = 0; i < S.size(); ++i) {
            
            count[i + 1] = 2 * count[i];
            
            if (last[S[i] - 'a'] != -1) 
                count[i + 1] += MOD - count[last[S[i] - 'a']];
         //   printf("%d %c %d %lu %lu\n", i, S[i], last[S[i] - 'a'], (last[S[i] - 'a'] == -1)?0:count[last[S[i] - 'a']], count[i + 1]);
            count[i + 1] = count[i + 1] % MOD;
            last[S[i] - 'a'] = i;
        }

        return (count[S.size()] + MOD - 1) % MOD;
    }
};
