class Solution {
public:
    int distinctSubseqII(string S) 
    {
        vector<long long> dp(S.size() + 1, 0);
        long b = 1000000007;
        vector<int> last(26, -1);
        for(int i = 0; i < S.size(); ++i)
        {
            if(last[S[i] - 'a'] == -1)
            {
                // first letter, all subseq in the past will have a new seq, plus itself as s seq
                dp[i + 1] = dp[i] * 2 + 1;
                // cout << i << ' ' << dp[i + 1] << endl;
                dp[i + 1] = dp[i + 1] % b;
                last[S[i] - 'a'] = i;
            }
            else
            {
                // 
                dp[i + 1] = dp[last[S[i] - 'a']] + (dp[i] + b - dp[last[S[i] - 'a']]) * 2;
                // if(dp[i] < dp[last[S[i] - 'a']])
                // {
                //     cout << i << ' ' << last[S[i] - 'a'] << ' ' << dp[i] << ' ' <<  dp[last[S[i] - 'a']] << endl;
                // }
                // if(i == 40)
                // {
                //     for(auto x : dp) cout << x << ' ';
                //     cout << endl;
                // }
                // cout << i << ' ' << dp[i + 1] << ' ' << dp[last[S[i] - 'a']] << ' ' << (dp[i] - dp[last[S[i] - 'a']])  << endl;
                // cout << last[S[i] - 'a'] << endl;
                dp[i + 1] = dp[i + 1] % b;
                last[S[i] - 'a'] = i; 
            }
        }
        // for(auto x : dp) cout << x << ' ';
        // cout << endl;
        return dp[S.size()];
    }
};
