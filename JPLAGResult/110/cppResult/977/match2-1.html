class Solution {
public:
    int distinctSubseqII(string S) {
        int len=S.size();
        vector<long long> dp(len+1,1);
        vector<int> visited(256,-1);
        for(int i=1;i<=len;i++){
            
            if(visited[S[i-1]]!=-1){
                dp[i]=(2*dp[i-1]-dp[visited[S[i-1]]-1]+1000000007)%1000000007;
            }
            else dp[i]=(2*dp[i-1])%1000000007;
            visited[S[i-1]]=i;
            dp[i]%=1000000007;
        }
        return dp.back()-1;
    }
};
