class Solution {
public:
    int distinctSubseqII(string S) {
        int mod = 1000000007;
        VI vi(26, -1);
        vector<ll> dp(2002, 0);
        dp[0] = 1;
        
        REP(i, sz(S)) {
            dp[i+1] = 2*dp[i];
            if(vi[S[i]-'a'] > -1) {
                dp[i+1] -= dp[vi[S[i]-'a']];
            }
            vi[S[i]-'a'] = i;
            dp[i+1] %= mod;
        }
        return (int)(((dp[sz(S)]-1) + mod) % mod);
    }
};
