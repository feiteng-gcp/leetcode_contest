class Solution {
    using pii = pair<int, int>;
public:
    int minAreaRect(vector<vector<int>>& points) {
        set<pii> table;
        for (auto point : points) {
            table.emplace(point[0], point[1]);
        }
        int ret = INT_MAX;
        for (int i = 0; i < points.size(); i++) {
            for (int j = i+1; j < points.size(); j++) {
                if (points[i][0] == points[j][0]) continue;
                if (points[i][1] == points[j][1]) continue;
                if (table.find(make_pair(points[i][0], points[j][1])) == table.end()) continue;
                if (table.find(make_pair(points[j][0], points[i][1])) == table.end()) continue;
                ret = min(abs(points[i][0] - points[j][0]) * abs(points[i][1] - points[j][1]), ret);
            }
        }
        if (ret == INT_MAX) ret = 0;
        return ret;
        
    }
};
