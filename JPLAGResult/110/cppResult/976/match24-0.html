class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        set<vector<int>> st;
        for (auto& p: points) {
            st.insert(p);
        }
        int ans = INT_MAX;
        for (auto& p: points) {
            for (auto& q: points) {
                if (p[0] != q[0] && p[1] != q[1]) {
                    if (st.count({p[0], q[1]}) && st.count({q[0], p[1]})) {
                        ans = min(ans, abs(q[1] - p[1]) * abs(q[0] - p[0]));
                    }
                }
            }
        }
        return ans == INT_MAX ? 0 : ans;
    }
};
