#define MAXN 40010
class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        unordered_set<int> ps;
        for (auto it: points) {
            ps.insert(it[0] * MAXN + it[1]);
        }
        
        int sol = INT_MAX;
        int pn = points.size();
        for (int i = 0; i < pn; ++i) {
            for (int j = i + 1; j < pn; ++j) {
                if (points[i][0] == points[j][0]) continue;
                if (points[i][1] == points[j][1]) continue;
                int p3 = points[i][0] * MAXN + points[j][1];
                int p4 = points[j][0] * MAXN + points[i][1];
                if (!ps.count(p3)) continue;
                if (!ps.count(p4)) continue;
                int x = points[i][0] - points[j][0];
                int y = points[i][1] - points[j][1];
                int a = abs(x * y);
                sol = min(sol, a);
            }
        }
        return (sol == INT_MAX) ? 0 : sol;
    }
};
