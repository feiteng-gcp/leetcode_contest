class Solution {
public:
    int minAreaRect(vector<vector<int>>& a) {
		int n = a.size();

		map<int, set<int>> plane;
		for (auto &e : a)
		{
			plane[e[0]].insert(e[1]);
		}
		int ans = INT_MAX;
		for (auto i = plane.begin(); i != plane.end(); ++i)
		{
			for (auto j = plane.begin(); j != i; ++j)
			{
				int last = -1, dx = i->first - j->first;
				for (int y : i->second)
				{
					if (j->second.count(y))
					{
						if (last == -1) last = y;

						else
						{
							ans = min(ans, dx * (y - last));
							last = y;
						}
					}
				}
			}
		}
        if (ans != INT_MAX)
		    return ans;
        return 0;
    }
};
