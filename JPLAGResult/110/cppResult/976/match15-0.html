class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        map<int, set<int>> hori;
        for(auto x: points){
            hori[x[0]].insert(x[1]);
        }
        int ret = INT_MAX;
        for(auto i = hori.begin();i!=hori.end();i++){
            for(auto j = i; j!=hori.end();){
                j++;
                if(j==hori.end()){
                    break;
                }
                auto& left = *i;
                auto& right = *j;
                bool find = false;
                int bottom = 0;
                int width = right.first - left.first;
                for(auto x: left.second){
                    if(right.second.count(x)){
                        if(find){
                            ret = min(ret,(x-bottom) * width);
                            bottom = x;
                        }
                        else{
                            find = true;
                            bottom = x;
                        }
                    }
                }
            }
        }
        if(ret == INT_MAX) return 0;
        return ret;
    }
};
