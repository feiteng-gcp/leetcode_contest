class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        unordered_map< int, unordered_set< int > > pts;
        for( auto &xy: points ) {
            pts[ xy[ 0 ] ].insert( xy[ 1 ] );
        }
        int min_area = INT_MAX;
        for( int i = 0; i < points.size(); i++ ) {            
            for( int j = i + 1; j < points.size(); j++ ) {       
                int xi = points[ i ][ 0 ], yi = points[ i ][ 1 ];
                int xj = points[ j ][ 0 ], yj = points[ j ][ 1 ];
                if( xi == xj || yi == yj ) continue;
                if( xi > xj ) swap( xi, xj );
                if( yi > yj ) swap( yi, yj );                
                if( has( pts, xi, yi ) && has( pts, xi, yj ) && has( pts, xj, yi ) && has( pts, xj, yj ) ) {                    
                    min_area = min( min_area, ( xj - xi ) * ( yj - yi ) );                    
                }                
            }
        }
        return min_area == INT_MAX ? 0 : min_area;
    }
    bool has( unordered_map< int, unordered_set< int > > &pts, int x, int y ) {
        return pts.count( x ) && pts[ x ].count( y );
    }
};
