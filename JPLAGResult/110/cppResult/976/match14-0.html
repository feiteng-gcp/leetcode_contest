class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        int n = points.size();
        set <pair <int, int> > s;
        vector <int> x, y;
        for (int i = 0; i < n; i++) {
            x.push_back(points[i][0]);
            y.push_back(points[i][1]);
            s.insert(make_pair(points[i][0], points[i][1]));
        }
        
        int ans = 2e9;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j || points[i][0] >= points[j][0] || points[i][1] >= points[j][1]) {
                    continue;
                }
                
                int x1 = points[i][0], x2 = points[j][0];
                int y1 = points[i][1], y2 = points[j][1];
                //printf("%d %d %d %d\n", x1, y1, x2, y2);
                if (s.find(make_pair(x1, y2)) != s.end() && s.find(make_pair(x2, y1)) != s.end()) {
                    
                    ans = min(ans, (x2 - x1) * (y2 - y1));
                }  
            }
        }
        
        if (ans > 2e9 - 1) {
            ans = 0;
        }
        return ans;
    }
};
