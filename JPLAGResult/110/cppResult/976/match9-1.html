class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        int n = points.size();
        set<pair<int, int>> pset;
        int minArea = INT_MAX;
        

        for (auto &point : points) {
            pset.insert({point[0], point[1]});
        }
        
        for (int i=0; i<n; i++) {
            for (int j=i+1; j<n; j++) {
                if (points[i][0] != points[j][0] && points[i][1] != points[j][1]) {

                    if (pset.count({points[i][0], points[j][1]}) && pset.count({points[j][0], points[i][1]})) {
                        int area = (abs(points[j][1] - points[i][1])) * (abs(points[j][0] - points[i][0]));
                        if (minArea > area) {
                            minArea = area;
                        }
                    }
                }
            }
        }
        
        
        return minArea == INT_MAX ? 0 : minArea;
    }
};
