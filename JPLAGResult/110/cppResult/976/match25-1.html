class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        unordered_set<string> pts;
        for (auto p : points) {
            string s = to_string(p[0]) + ' ' + to_string(p[1]);
            pts.insert(s);

        }
        int n = points.size(), ans = INT_MAX;
        for (int i = 0; i < n; i++) {
            int x1 = points[i][0], y1 = points[i][1];
            for (int j = i + 1; j < n; j++) {
                int x2 = points[j][0], y2 = points[j][1];
                if (x1 == x2 || y1 == y2) continue;
                string s = to_string(x1) + ' ' + to_string(y2);
                string t = to_string(x2) + ' ' + to_string(y1);
                if (pts.count(s) && pts.count(t)) {
                    ans = min(ans, abs(x1 - x2) * abs(y1 - y2));
                }
            }
        }
        return ans == INT_MAX ? 0 : ans;
    }
};
