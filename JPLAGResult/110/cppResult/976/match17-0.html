class Solution {
public:
    int minAreaRect(vector<vector<int>>& p) {
        
        int n = p.size();
        unordered_map<int, unordered_map<int, bool>> visited;
        for(auto& v: p)
        {
            visited[v[0]][v[1]] = true;
        }
        
        int ans = INT_MAX;
        
        for(int i=0; i<n; i++){
            for(int j=i+1; j<n; j++){
                
                if(visited[p[i][0]][p[j][1]] && visited[p[j][0]][p[i][1]]){
                    int cur = abs((p[i][0]-p[j][0]) * (p[i][1]-p[j][1]));
                    if(cur>0) ans = min(ans, cur);
                }
            }
        }
        
        return ans == INT_MAX? 0: ans;
    }
};
