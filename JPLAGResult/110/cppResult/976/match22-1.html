typedef pair<int, int> ii;

class Solution {
public:
    int minAreaRect(vector<vector<int>>& p) {
        int n = p.size();
        set<ii> visit;
        for (auto& it : p) {
            visit.insert({it[0], it[1]});
        }
        int ret = 2000000000;

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (!(p[i][0] < p[j][0] && p[i][1] < p[j][1])) continue;
                //cout << p[i][0] << " " << p[i][1] << " " << p[j][0] << " " << p[j][1] << endl;
                if (visit.count({p[i][0], p[j][1]}) && visit.count({p[j][0], p[i][1]})) {
                    //cout << p[i][0] << " " << p[i][1] << " " << p[j][0] << " " << p[j][1] << endl;
                    ret = min(ret, (p[j][1] - p[i][1]) * (p[j][0] - p[i][0]));
                }
            }
        }
        if (ret == 2000000000) ret = 0;
        return ret;
    }
};
