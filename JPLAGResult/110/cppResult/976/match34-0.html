#define all(c) (c).begin(),(c).end()
#define tr(c,i) for(typeof((c).begin()) i = (c).begin(); i != (c).end(); i++)
#define ll long long
#define sz(a) int((a).size())
#define pb push_back
#define SORT(x) sort((x).begin(),(x).end());
#define VI vector<int>
#define VII vector < vector <int> >
#define MP make_pair
#define SET(x,byte) memset(x,byte,sizeof(x));
#define I(x) ((x) - 'A')
#define REP(i,mx) for(int i=0;(i)<(mx);i++)
#define FOR(i,mn,mx) for(int i=mn;(i)<=(mx);i++)
#define U(x) (1<<(x))
#define INF 2000000007
#define PI acos(-1.0)
#define endl '\n'
class Solution {
public:
    int minAreaRect(vector<vector<int>>& p) {
        map<pair<int, int>, int> mp;
        REP(i, sz(p)) {
            mp[MP(p[i][0], p[i][1])] = 1;
        }
        
        int mn = INF;
        
        REP(i, sz(p)) {
            FOR(j, i+1, sz(p)-1) {
                int a = p[i][0];
                int b = p[i][1];
                int c = p[j][0];
                int d = p[j][1];
                
                if(a == c || d == b) continue;
                
                if(mp[MP(c, b)] && mp[MP(a, d)]) {
                    mn = min(mn, abs((c-a)*(d-b)));
                }
            }
        }
        
        if(mn == INF) return 0;
        return mn;
    }
};
