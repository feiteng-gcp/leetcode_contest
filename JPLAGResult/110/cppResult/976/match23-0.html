class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        unordered_map <int, set <int> > x;
        for (auto cur : points) {
            x[cur[0]].insert(cur[1]);
        }        
        int prev, ans = INT_MAX;
        for (auto i : x) {
            for (auto j : x) {
                if (i.first != j.first) {
                    prev = -1;
                    for (auto k : i.second) {
                        if (prev != -1 && j.second.find(k) != j.second.end() && j.second.find(prev) != j.second.end()) {
                            ans = min(ans, abs(i.first - j.first) * (k - prev));
                        }
                        prev = k;
                    }
                }
            }
        }
        return ans == INT_MAX ? 0 : ans;
    }
};
