class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        unordered_set<int> cache;
        for (auto& p : points) {
            cache.insert(hash(p));
        }
        
        int c = 0;
        int ans = INT_MAX;
        for (int i = 0; i < points.size() - 1; ++i) {
            for (int j = i + 1; j < points.size(); ++j) {
                int x1 = points[i][0];
                int y1 = points[i][1];
                int x2 = points[j][0];
                int y2 = points[j][1];
                if (x1 != x2 && y1 != y2) {
                    vector<int> p1 = {x1, y2};
                    vector<int> p2 = {x2, y1};
                    if (cache.find(hash(p1)) != cache.end() && cache.find(hash(p2)) != cache.end()) {
                        ++c;
                        ans = min(ans, abs(x1-x2) * abs(y1-y2));
                    }
                }
            }
        }
        
        if (c == 0) {
            return 0;
        }
        else {
            return ans;
        }
    }
private:
    int hash(const vector<int>& p) {
        return p[0] * 40001 + p[1];
    }
};
