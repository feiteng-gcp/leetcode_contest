class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        map<int,vector<int> > cords;
        
        for (int i=0;i<points.size();++i) {
            int x = points[i][0], y = points[i][1];
            cords[x].push_back(y);
        }
        
        int ret = INT_MAX;
        unordered_map<int,int> slideWindow;
        for (auto it = cords.begin(); it != cords.end(); ++it) {
            int curX = it->first;

            vector<int> & curYs = it->second;
            sort(curYs.begin(),curYs.end());
            
            for(int i=0;i<curYs.size();++i) {
                for(int j=i+1;j<curYs.size();++j) {
                    int hash = curYs[i]*40001+curYs[j];
                    int diff = curYs[j] - curYs[i];
                    auto findIt = slideWindow.find(hash);
                    if (findIt==slideWindow.end()) {
                        slideWindow.insert(make_pair(hash,curX));
                    } else {
                        int curArea = diff * (curX - findIt->second);
                        if (curArea < ret) ret = curArea;
                        findIt->second = curX;
                    }
                }
            }
            
        }   
        return ret==INT_MAX ? 0 : ret;
    }
};
