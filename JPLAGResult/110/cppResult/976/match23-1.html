class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
 //       points = {{1,1},{1,3},{3,1},{3,3},{4,1},{4,3}};
        map<int, set<int>> ps;
        for (auto &p: points) {
            ps[p[0]].insert(p[1]);
        }
      
        int minArea = INT_MAX;
     /*   for (auto &x1: ps) { 
            printf("%d\n", x1.first);
            for (auto &y1: x1.second)
                printf("%d ", y1);
            printf("\n");
        }*/
        for (auto &x1: ps) {
            for (auto &x2: ps) {
                if (x1.first >= x2.first) continue;
            //    printf("%d %d\n", x1.first, x2.first);
                vector<int> sp; 
                for (auto &y1: x1.second) {
                    if (x2.second.find(y1) != x2.second.end()) {
                   //     printf("push %d\n", y1);
                        sp.push_back(y1);
                    }
                    
                }
                
                if (!sp.empty()) {
                    for (int i = 0; i < sp.size() - 1; ++i) {
                        minArea = min(minArea, (sp[i + 1] - sp[i]) * (x2.first - x1.first));
                    }
                }
            }
        }
        
        return minArea == INT_MAX?0:minArea;
    }
};
