class Solution {
public:
    int minAreaRect(vector<vector<int>>& a) {
        int ans = 0;
        set<pair<int, int>> se;
        for (int i = 0; i < a.size(); i++) {
            se.insert({a[i][0], a[i][1]});
        }
        for (int i = 0; i < a.size(); i++) {
            for (int j = i + 1; j < a.size(); j++) {
                int x1 = a[i][0], y1 = a[i][1];
                int x2 = a[j][0], y2 = a[j][1];
                if (se.count({x1, y2}) && se.count({x2, y1}) && x1 != x2 && y1 != y2) {
                    int c = abs(x2 - x1) * abs(y2 - y1);
                    if (ans == 0 || c < ans) {
                        ans = c;
                    }
                }
            }
        }
        return ans;
    }
};
