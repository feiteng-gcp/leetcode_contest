class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        map<int, map<int, bool>> pts;
        for (auto& pt: points) {
            pts[pt[0]][pt[1]] = true;
        }

        int res = INT_MAX;
        for (int i = 0; i < points.size(); i++) {
            int xi = points[i][0], yi = points[i][1];
            for (int j = i + 1; j < points.size(); j++) {
                int xj = points[j][0], yj = points[j][1];
                if (xi != xj && yi != yj && pts[xi].find(yj) != pts[xi].end() && pts[xj].find(yi) != pts[xj].end()) {
                    res = min(res, abs(xi - xj) * abs(yi - yj));
                }
            }
        }
        if (res == INT_MAX) {
            return 0;
        }
        return res;
    }
};
