class Solution {
public:
    int minAreaRect(vector<vector<int>>& s) {
        unordered_map<int, unordered_set<int> > m;
        for(auto i: s) {
            m[i[0]].insert(i[1]);
        }

        long long res = LLONG_MAX;
        sort(s.begin(), s.end());
        for(int i=0; i<s.size(); ++i) {
            for(int j=i+1; j<s.size(); ++j) {
                if (s[i][0] == s[j][0]) continue;
                if (s[i][1] == s[j][1]) continue;
                int x1 = s[i][0], y1 = s[j][1];
                int x2 = s[j][0], y2 = s[i][1];
                if (m[x1].find(y1) != m[x1].end() && m[x2].find(y2) != m[x2].end()) {
                    long long d1 = abs(x1 - x2);
                    long long d2 = abs(y2 - y1);
                    res = min(res, d1 * d2);
                }
            }
        }
        return res==LLONG_MAX ? 0: res;
    }
};
