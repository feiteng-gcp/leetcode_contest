class Solution {
public:
    vector<int> X[41234];
    vector<int> Y[41234];
    int minAreaRect(vector<vector<int>>& p) {
        int n = p.size();
        for (int i = 0; i < n; i++) {
            X[ p[i][0] ].push_back(p[i][1]);
            Y[ p[i][1] ].push_back(p[i][0]);
        }
        for (int i = 0; i <= 40000; i++) {
            sort(X[i].begin(), X[i].end());
            sort(Y[i].begin(), Y[i].end());
        }
        
        int res = -1;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int x1 = p[i][0];
                int y1 = p[i][1];
                int x2 = p[j][0];
                int y2 = p[j][1];
                
                if (x1 == x2 || y1 == y2) continue;
                bool ok = false;
                if (binary_search(X[x1].begin(), X[x1].end(), y2) && binary_search(Y[y1].begin(), Y[y1].end(), x2))
                    ok = true;
                if (ok) {
                    int area = abs(x1 - x2) * abs(y1 - y2);
                    if (res == -1) res = area;
                    else if (res > area) res = area;
                }
            }
        }
        if (res == -1) res = 0;
        return res;
    }
};
