class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        bool found = false;
        int ma = 40000 * 40000;
        int col = 40001;
        set<int> idx;
        for (vector<int> v: points) {
            idx.insert(v[0] * col + v[1]);
        }
        int n = points.size();
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int x0 = points[i][0];
                int y0 = points[i][1];
                int x1 = points[j][0];
                int y1 = points[j][1];
                if (x0 == x1 || y0 == y1) continue;
                if (idx.find(x0 * col + y1) != idx.end() && idx.find(x1 * col + y0) != idx.end()) {
                    // printf("%d %d\n", i, j);
                    // printf("%d %d %d %d\n", x0, y0, x1, y1);
                    ma = min(ma, abs(x0 - x1) * abs(y0 - y1));
                    found = true;
                }
            }
        }
        
        if (!found) return 0;
        return ma;
    }
};
