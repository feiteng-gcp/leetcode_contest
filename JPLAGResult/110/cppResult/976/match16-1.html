class Solution {
public:
    int minAreaRect(vector<vector<int>>& P) {
        int best = 2*1000*1000*1000;
        set<pair<int,int>> M;
        for (auto p : P) {
            M.insert({p[0],p[1]});
        }
        for (int i = 0; i < P.size(); ++i) for (int j = 0; j < P.size(); ++j) if (i!=j) {
            int x1 = P[i][0];
            int y1 = P[i][1];
            int x2 = P[j][0];
            int y2 = P[j][1];
            if (x1 != x2 && y1 != y2) {

                if (M.count({x1,y2}) && M.count({x2,y1})) {
                    int v = abs(x2-x1)*abs(y2-y1);
                    if (v && v < best) best = v;
                }
            }
        }
        return best == 2*1000*1000*1000 ? 0 : best;
    }
};
