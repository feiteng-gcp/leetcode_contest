class Solution {
public:
    int encode(const vector<int>& point, int N){
        return point[0] * N + point[1];
    }
    int encode(int x, int y, int N){
        return x * N + y;
    }
    int minAreaRect(vector<vector<int>>& points) {
        const int N = 40001;
        set<int> points_dict;

        for(auto point: points){
            points_dict.insert(encode(point, N));
        }
        int min_area = INT_MAX;
        for(int i=0;i<points.size();++i){
            for(int j=i;j<points.size();++j){
                int x1 = points[i][0];
                int x2 = points[j][0];
                int y1 = points[i][1];
                int y2 = points[j][1];
                if(points_dict.count(encode(x1, y2, N)) > 0 and points_dict.count(encode(x2, y1, N)) > 0){
                    int area = abs((x2 - x1)*(y2 - y1));
                    if(area > 0) min_area = min(area, min_area);
                }
            }
        }
        return min_area == INT_MAX ? 0:min_area;
    }
};
