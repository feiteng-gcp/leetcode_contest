class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        int ret = 40000*40000;
        set<pair<int,int>> s;
        int n=points.size();
        for(int i=0;i<n;i++){
            s.emplace(points[i][0], points[i][1]);
        }
        
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                int x1=points[i][0], y1=points[i][1];
                int x2=points[j][0], y2=points[j][1];
                if(x1!=x2 && y1!=y2){
                    if(s.count(make_pair(x1,y2))&&s.count(make_pair(x2,y1))){
                        ret = min(ret, abs(x1-x2)*abs(y1-y2));
                    }
                }
            }
        }
        return ret==40000*40000 ? 0 : ret;
    }
};
