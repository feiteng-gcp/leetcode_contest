class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        set<pair<int, int> > st;
        int res = INT_MAX;
        for (auto item : points) {
            st.insert(make_pair(item[0], item[1]));
        }
        for (int i = 0; i < points.size(); i++) {
            for (int j = i + 1; j < points.size(); j++) {
                int x1 = min(points[i][0], points[j][0]);
                int x2 = max(points[i][0], points[j][0]);
                int y1 = min(points[i][1], points[j][1]);
                int y2 = max(points[i][1], points[j][1]);
                if (st.find(make_pair(x1, y1)) != st.end() && st.find(make_pair(x1, y2)) != st.end() && st.find(make_pair(x2, y1)) != st.end() && st.find(make_pair(x2, y2)) != st.end()) {
                    // cout << i << ", " << j << endl;
                    if ((x2 - x1) * (y2 - y1) > 0) {
                        res = min(res, (x2 - x1) * (y2 - y1));
                    }
                }
            }
        }
        return res == INT_MAX ? 0 : res;
    }
};
