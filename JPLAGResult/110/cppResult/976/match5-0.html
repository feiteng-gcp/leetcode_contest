class Solution {
    int key(int x, int y) {
        return x*40001 + y;
    }
public:
    unordered_set<int> s;
    int r = INT_MAX; 
    int minAreaRect(vector<vector<int>>& points) {
        int N = points.size();
        for (int i=0; i<N; i++) {
            s.insert(key(points[i][0],points[i][1]));
        }
        for (int i=0; i<N-1; i++) {
            for (int j=i+1; j<N; j++) {
                int ix = points[i][0];
                int iy = points[i][1];
                int jx = points[j][0];
                int jy = points[j][1];
                if (jx==ix || jy==iy) continue;
                if (s.find(key(ix,jy)) == s.end()) continue;
                if (s.find(key(jx,iy)) == s.end()) continue;
                r = min(r,abs(ix-jx)*abs(iy-jy));
            }
        }
        return r == INT_MAX ? 0 : r;
    }
};
