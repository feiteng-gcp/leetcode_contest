class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {

        typedef pair<int,int>P;
        int ans=2e9;
        set<P>st;
        for(int i=0;i<points.size();i++){
            st.insert(P(points[i][0],points[i][1]));
        }
        for(int i=0;i<points.size();i++){
            for(int j=i+1;j<points.size();j++){
               // for(int k=j+1;k<points.size();k++){
                    int a1=points[i][0];
                    int b1=points[i][1];
                    int a2=points[j][0];

                    int b2=points[j][1];
                    if(a1!=a2&&b1!=b2){
                        if(st.count(P(a1,b2))&&st.count(P(a2,b1))){
                            ans=min(ans,abs(a1-a2)*abs(b1-b2));
                        }
                    }
              //  }
            }
        }
        return ans==2e9?0:ans;
    }
};
