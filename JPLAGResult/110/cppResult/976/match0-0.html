class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        int ans=INT_MAX,len=points.size();
        unordered_set<string> st;
        for(auto v:points) 
            st.insert(to_string(v[0])+"-"+to_string(v[1]));
        for(int i=0;i<len;i++){
            for(int j=0;j<i;j++){
                if(points[i][0]==points[j][0]||points[i][1]==points[j][1])
                    continue;
                string s1=to_string(points[i][0])+"-"+to_string(points[j][1]);
                string s2=to_string(points[j][0])+"-"+to_string(points[i][1]);
                if(st.find(s1)!=st.end()&&st.find(s2)!=st.end())
                    ans=min(ans,abs(points[i][0]-points[j][0])*abs(points[i][1]-points[j][1]));
            }
        }
        return ans==INT_MAX?0:ans;
    }
};
