class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        unordered_map<int, unordered_set<int>> hash;
        for (auto& p : points) {
            hash[p[0]].insert(p[1]);
        }
        int area = 0;
        for (int i = 0; i < points.size(); i++) {
            for (int j = i + 1; j < points.size(); j++) {
                if (points[i][0] == points[j][0] || points[i][1] == points[j][1]) {
                    continue;
                }
                if (hash[points[i][0]].find(points[j][1]) != hash[points[i][0]].end() 
                    && hash[points[j][0]].find(points[i][1]) != hash[points[j][0]].end()
                    && (area == 0 || abs(points[i][0] - points[j][0]) * abs(points[i][1] - points[j][1]) < area)) {
                    area = abs(points[i][0] - points[j][0]) * abs(points[i][1] - points[j][1]);
                }
            }
        }
        return area;
    }
};
