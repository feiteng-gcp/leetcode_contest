class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        if (A.size() == 0) {
            return 0;
        }
        if (A.size() == 1) {
            return 1;
        }
        if (A.size() == 2) {
            return 2;
        }
        vector<int> B(A.size()-1);
        
        for (int i = 0; i < A.size()-1;i++) {
            if (A[i+1] > A[i]) {B[i] = 1;}
            if (A[i+1] == A[i]) {B[i] = -1;}
        }
        vector<int> f(B.size());
        f[0] = 1;
        for (int i = 1; i < f.size(); i++) {
            if (B[i] != B[i-1]) {
                f[i] = f[i-1] + 1;
            } else {
                f[i] = 1;
            }
            if (B[i] == -1) {
                f[i] = 0;
            }
        }
        
        int ans = 1;
        for (auto k: f) {
            ans = std::max(ans, k);
        }
        return ans + 1;
    }
};
