class Solution {
public:
    int maxTurbulenceSize(vector<int>& a) {
        vector<int> pre;
        int res = (a.size()==0? 0: 1);
        for(int i = 1; i < a.size(); i++)
        {
            if(a[i] > a[i-1]) pre.push_back(-1);
            else if(a[i] < a[i-1]) pre.push_back(1);
            else pre.push_back(0);
            if(pre[i-1] != 0) res = 2;
        }
        
        int i = 0, j = 0;
        int preSign = 0;
        
        while(j < pre.size())
        {
            if(pre[j] == 0)
            {
                i = j+1;
            }

            else if(preSign == 0)
            {
                i=j;
                preSign = pre[j];
            }
            else if(preSign == -pre[j])
            {
                preSign = pre[j];
                res = max(res, j-i+2);
            }
            else
            {
                i=j;
                preSign = pre[j];
            }
            
            
            j++;
        }
        
        return res;
    }
};
