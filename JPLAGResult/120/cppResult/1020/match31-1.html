class Solution {
public:
    int maxTurbulenceSize(vector<int>& a) {
        int n = a.size();
        int ans = 0, cnt = 0;
        for (int i = 0; i + 1 < n; i++) {
            // cout << i << " " << cnt << endl;
            if ((1+i) & 1) {
                if (a[i] < a[i+1]) {
                    cnt++;
                    continue;
                }
            } else {
                if (a[i] > a[i+1]) {
                    cnt++;
                    continue;
                }
            }
            ans = max(ans, cnt);
            cnt = 0;
        }
        ans = max(ans, cnt);
        cnt = 0;
        for (int i = 0; i + 1 < n; i++) {
            // cout << i << " " << cnt << endl;
            if (i & 1) {
                if (a[i] < a[i+1]) {
                    cnt++;
                    continue;
                }
            } else {
                if (a[i] > a[i+1]) {
                    cnt++;
                    continue;
                }
            }
            ans = max(ans, cnt);
            cnt = 0;
        }
        ans = max(ans, cnt);
        return ans + 1;
    }
};
