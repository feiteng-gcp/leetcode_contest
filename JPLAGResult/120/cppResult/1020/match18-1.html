class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
		int cur=0,ma=0,i;

		FOR(i,A.size()-1) {
			if(i%2==0) {
				if(A[i]<A[i+1]) cur++;
				else cur=0;
				
			}
			else {
				if(A[i]>A[i+1]) cur++;
				else cur=0;
			}
			ma=max(ma,cur);
		}
		cur=0;
		FOR(i,A.size()-1) {

			if(i%2==1) {
				if(A[i]<A[i+1]) cur++;
				else cur=0;
				
			}
			else {
				if(A[i]>A[i+1]) cur++;
				else cur=0;
			}
			ma=max(ma,cur);
		}
		return ma+1;
        
    }
};
