class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        if (A.size() < 3) {
            return A.size();
        }
        int res = 1;
        int pos = 1;
        while(pos + 1 < A.size()) {
            int c = 0;
            while(pos + 1 < A.size() && ((A[pos] > A[pos-1] && A[pos] > A[pos+1]) || (A[pos] < A[pos-1] && A[pos] < A[pos+1]))) {
                //cout << pos << " ";
                ++c;
                ++pos;
            }
            //cout << endl;
            if (c > 0) {
                res = max(res, c + 2);
            }
            ++pos;
        }
        if(res == 1) {
            for (int i = 0; i + 1 < A.size(); ++i) {
                if(A[i] != A[i+1]) {
                    res = 2;
                    break;
                }
            }
        }
        return res;
    }
};
