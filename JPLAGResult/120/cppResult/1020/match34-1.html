class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        int n = A.size();
        

        vector<vector<int>> dp(n + 1, vector<int>(2, 1));
        // dp[0][0] = dp[0][1] = 1;
        for (int i = 1; i < n; ++i) {
            if (A[i] > A[i - 1]) {
                dp[i][0] = max(dp[i][0], dp[i - 1][1] + 1);
            } 
            if (A[i] < A[i - 1]) {
                dp[i][1] = max(dp[i][1], dp[i - 1][0] + 1);
            }
        }
        
        
        int res = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < 2; ++j) {
                res = max(res, dp[i][j]);
            }
        }
        return res;
        
    }
};
