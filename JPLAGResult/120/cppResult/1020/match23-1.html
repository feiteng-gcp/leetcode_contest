class Solution {
public:
    int maxTurbulenceSize(vector<int>& a) {
        int n = a.size();
        if (n == 0) return 0;
        int ans = 1, cnt = 1;
        vector<int> B(n);
        
        for (int i = 1; i < n; ++ i) {
            if (a[i] > a[i - 1]) {
                B[i] = 1;
                if (cnt == 1 || B[i - 1] == -1) ++ cnt;
                else if (B[i - 1] == 1) cnt = 2;
                else cnt = 1;
            } else if (a[i] < a[i - 1]) {
                B[i] = -1;
                if (cnt == 1 || B[i - 1] == 1) ++ cnt;
                else if (B[i - 1] == -1) cnt = 2;

                else cnt = 1;
            } else {
                B[i] = 0;
                cnt = 1;
            }
            ans = max(ans, cnt);
        }
        return ans;
    }
};
