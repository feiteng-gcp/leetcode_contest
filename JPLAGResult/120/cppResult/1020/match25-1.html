int calc(vector<int>& A, int f){

    int len=1,mx=1;
    for(int i=0;i+1<A.size();i++){
        if(A[i]<A[i+1] && (i&1)== f){
            len++;
            mx=max(mx,len);
        }else if(A[i]>A[i+1] && (i&1)== (f^1)){
            len++;
            mx=max(mx,len);      
        }else{
            len=1;
        }
    }
    
    
    return mx;
}

class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        return max(calc(A,0), calc(A,1));
    }
};
