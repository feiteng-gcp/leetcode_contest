class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        int ans = 0;
        int running = 0, s = 0;
        for (int i = 0; i < A.size(); i++) {
            if (running < 1) {
                running++;
            } else if (running < 2) {
                s = sign(A[i] - A[i-1]);
                if (s) running++;
            } else {
                int ns = sign(A[i] - A[i-1]);
                if (ns == -s) running++, s = ns;
                else if (ns) running = 2, s = ns;
                else running = 1, s = 0;
            }
            ans = max(ans, running);
        }
        return ans;
    }
    
    int sign(int a) {
        return a > 0 ? 1 : (a < 0 ? -1 : 0);
    }
};
