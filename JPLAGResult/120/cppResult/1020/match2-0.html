class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        vector<int> downDp(A.size(), 0);
        vector<int> upDp(A.size(), 0);
        downDp[0] = upDp[0] = 1;
        int maxCount = 1;
        for(int i = 1;i< A.size();i++){
            if(A[i] > A[i - 1]){
                upDp[i] = downDp[i - 1] + 1;
                downDp[i] = 1;
            }
            else if(A[i] < A[i - 1]){
                downDp[i] = upDp[i - 1] + 1;
                upDp[i] = 1;
            }
            else{
                downDp[i] = upDp[i] = 1;
            }
            if(upDp[i] > maxCount) maxCount = upDp[i];
            if(downDp[i] > maxCount) maxCount = downDp[i];
        }
        return maxCount;
        
    }
};
