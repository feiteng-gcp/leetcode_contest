#include <iostream>
#include <cassert>
#include <unordered_map>
#include <unordered_set>
#include <set>

#include <fstream>
#include <algorithm>
#include <sstream>
#include <math.h>

using namespace std;

class Solution {
public:
    int maxTurbulenceSize(vector<int> A) {
        int current_length = 0;
        int max_length = 1;
        bool increase = false;

        for (int i = 0; i < A.size() - 1; ++i) {
            if (!((current_length == 0 && A[i] != A[i + 1]) ||
                    (A[i] != A[i + 1] && (A[i] > A[i + 1] == increase)))) {
                current_length = 0;
            }
            if (A[i] != A[i + 1]) {
                increase = A[i] < A[i + 1];
                ++current_length;
                max_length = max(max_length, current_length + 1);
            }
        }
        return max_length;
    }
};
