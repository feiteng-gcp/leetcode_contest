class Solution {
public:
    int get_bit(int x, int y){
        if (x < y)
            return 0;
        if (x > y)
            return 1;
        return 2;
    }
    int maxTurbulenceSize(vector<int>& A) {
        int n = A.size(), l = 0, r = 1;
        int ans = 1, i;
        vector<int>b;
        for (i = 0; i + 1 < n; i++) {
            b.push_back(get_bit(A[i], A[i+1]));
        }
        for (i = 0; i < n - 1; i++) {
            if ((b[i] ^ b[i+1]) != 1) {
                ans = max(ans, i - l + 2);
                l = i + 1;
            }
        }
        return ans;
    }
};
