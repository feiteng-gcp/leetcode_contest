class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        int res = 1, up = 1, down = 1;
        for (int i = 1; i < A.size(); ++i) {
            if (A[i] > A[i - 1]) {
                up = down + 1;
                down = 1;
                res = max(res, up);
            } else if (A[i] < A[i - 1]) {
                down = up + 1;
                up = 1;
                res = max(res, down);
            } else {
                up = 1;
                down = 1;
            }
        }
        return res;
    }
};
