class Solution {
public:
	int maxTurbulenceSize(vector<int>& A) {
		if (A.size() == 1) return 1;
		int cnt1 = 1, cnt2 = 1;
		for (int i = 0, cLen1 = 1, cLen2 = 1; i < (int)A.size() - 1; i++)
		{
			if (i % 2 != 0) {
				if (A[i] > A[i + 1]) cLen1++;
				else cLen1 = 1;
			}
			else
			{
				if (A[i] < A[i + 1]) cLen1++;
				else cLen1 = 1;
			}
			cnt1 = max(cnt1, cLen1);

			if (i % 2 == 0)
			{
				if (A[i] > A[i + 1]) cLen2++;
				else cLen2 = 1;
			}
			else
			{
				if (A[i] < A[i + 1]) cLen2++;
				else cLen2 = 1;
			}
			cnt2 = max(cnt2, cLen2);
		}
		return max(cnt1, cnt2);
	}
};
