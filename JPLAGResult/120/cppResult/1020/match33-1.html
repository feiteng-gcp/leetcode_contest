class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        if (A.empty())
            return 0;
        int maxA = 1, maxB = 1;
        int currA = 1, currB = 1;
        for (int i = 1; i < A.size(); i++) {

            if (i % 2 == 1) {
                if (A[i] > A[i - 1]) {
                    currA++;
                    maxA = max(maxA, currA);
                    currB = 1;
                } else if (A[i] < A[i - 1]) {
                    currB++;
                    maxB = max(maxB, currB);
                    currA = 1;
                } else {
                    currA = 1;
                    currB = 1;
                }
            } else {
                if (A[i] < A[i - 1]) {
                    currA++;
                    maxA = max(maxA, currA);
                    currB = 1;
                } else if (A[i] > A[i - 1]) {
                    currB++;
                    maxB = max(maxB, currB);
                    currA = 1;
                } else {
                    currA = 1;
                    currB = 1;
                }            
            }
        }
        return max(maxA, maxB);
    }
};
