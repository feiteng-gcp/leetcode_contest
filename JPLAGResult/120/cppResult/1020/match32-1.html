class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        vector<vector<int>> f(A.size(), vector<int>(2, 1));

        int ans = 1;
        for (int i = 1; i < A.size(); i++) {
            if (A[i] > A[i - 1]) {
                f[i][0] = f[i - 1][1] + 1;
            } else if (A[i] < A[i - 1]) {
                f[i][1] = f[i - 1][0] + 1;
            }
            ans = max(ans, max(f[i][0], f[i][1]));
        }
        return ans;
    }
};
