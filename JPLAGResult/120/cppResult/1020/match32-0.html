class Solution {
public:
    int maxTurbulenceSize(vector<int>& A);
};


int Solution::maxTurbulenceSize(vector<int>& A) {
    if (A.size() == 1) {
        return 1;
    }
    int n = A.size();
    vector<int> diff(n, 0);
    for (int i = 1; i < n; ++i) {
        diff[i] = A[i] - A[i - 1];
    }
    int res = 1;
    int start = 0;
    for (int i = 1; i < n; ++i) {
        if ((diff[i] > 0 && diff[i - 1] <= 0) || (diff[i] < 0 && diff[i - 1] >= 0)) {
            res = max(res, i - start + 1);
            // cout<<i<<" "<<start<<" "<<res<<endl;
        } else if ((diff[i] > 0 && diff[i - 1] > 0) || (diff[i] < 0 && diff[i - 1] < 0)){
            start = i - 1;
        } else {
            start = i;
        }
    }
    return res;
}
