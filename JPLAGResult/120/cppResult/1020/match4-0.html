class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        int f[40000][2]={0};
        f[0][0] = f[0][1] = 1;
        for(int i=1;i<A.size();i++){
            if(A[i]>A[i-1]){
                f[i][0]=f[i-1][1]+1;
                f[i][1]=1;
            }
            else if(A[i]<A[i-1]){
                f[i][0]=1;
                f[i][1]=f[i-1][0]+1;
            }
            else{
                f[i][0] = f[i][1] = 1;
            }
        }
        int res = 0;
        for(int i=0;i<A.size();i++){
            res = max(res,max(f[i][0],f[i][1]));
        }
        return res;
    }
};
