class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        int m = 1;
        vector<int> len (A.size(), 1);
        if (A.size() == 1) return m;
        len[1] = 2;
        m = 2;
        for (int i = 2; i < A.size(); ++ i) {
            if (A[i-2] > A[i-1] && A[i-1] < A[i]) {
                len[i] = len[i-1] + 1;
            }
            else if (A[i-2] < A[i-1] && A[i-1] > A[i]) {
                len[i] = len[i-1] + 1;
            }
            else {
                len[i] = 2;
            }
            
            m = (len[i] > m ? len[i] : m );
        }  
        return m;
    }
};
