class Solution {
public:
    int maxTurbulenceSize(vector<int>& A) {
        int N = A.size();
        if(N == 1) return 1;
        int len = 1, maxlen = 0;
        int lastdir = 0, curdir = 0;
        for(int i = 1; i < N; i++){
            if(A[i-1] > A[i])
                curdir = 2;
            else if(A[i-1] < A[i])
                curdir = 1;
            else
                curdir = 0;
            if(curdir * lastdir == 2){
                len++;
                maxlen = max(maxlen, len);
            }
            else if(curdir == 0){
                len = 1;
            }
            else{
                len = 2;
                maxlen = max(maxlen, len);
            }
            lastdir = curdir;
        }
        
        return maxlen;
    }
};
