/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int distributeCoins(TreeNode* root) {
        map<TreeNode*, int> dict;
        return dfs(root, dict);
    }
    
    int dfs(TreeNode* root, map<TreeNode*, int>& count) {
        if(!root) return 0;
        if(count.find(root) == count.end())
            count[root] = root->val;
        
        int coinsNum = count[root];
        int res = 0;
        
        res += dfs(root->left, count);
        coinsNum += count[root->left];
        res += abs(count[root->left]);
        
        res += dfs(root->right, count);
        coinsNum += count[root->right];
        res += abs(count[root->right]);
        
        count[root] = coinsNum - 1;
        return res;
    }
};
