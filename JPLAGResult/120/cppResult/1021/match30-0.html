/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans;
    int distributeCoins(TreeNode* root) {
        ans = 0;
        DFS(root);
        return ans;
    }
    
    int DFS(TreeNode* root) {
        if (NULL == root) return 0;
        int ret = root->val + DFS(root->left) + DFS(root->right) - 1;        
        ans += abs(ret);
        return ret;
    }
};
