/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int output;
    int distributeCoins(TreeNode* root) {
        output =0;
        
        helper(root);
        
        return output;
    }
    
    vector<int> helper(TreeNode *root)
    {
        if (!root)
        {
            return {0,0};
        }
        
        vector<int> l = helper(root->left);
        vector<int> r = helper(root->right);
        
        output += l[0] + l[1] + r[0]+r[1];
        
        vector<int> result(2, 0);
        if (root->val >= 1)
        {
            result[0] = root->val-1;
        }
        else
        {
            result[1] = 1;
        }
        
        result[0] += l[0] + r[0];
        result[1] += l[1] + r[1];
        
        if (result[0] > 0 && result[1] > 0)
        {
            if (result[0] >= result[1])
            {
                result[0] -= result[1];
                result[1] = 0;
            }
            else
            {
                result[1] -= result[0];
                result[0] = 0;
            }
        }
        return result;
    }
};
