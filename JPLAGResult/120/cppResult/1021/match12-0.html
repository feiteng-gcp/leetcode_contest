/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int distributeCoins(TreeNode* root) {
        int ans = 0;
        help(root, ans);
        return ans;
    }
    
    int help(TreeNode* root, int& ans) {
        if (root == NULL) {
            return 0;
        }
        
        int needl = help(root->left, ans);
        int needr = help(root->right, ans);
        
        root->val -= needl + needr;
        ans += abs(needl) + abs(needr);
        root->val -= 1;
        return -root->val;
        
    }
    
};
