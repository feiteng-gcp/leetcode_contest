/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int distributeCoins(TreeNode* root) {
        int res = 0;
        postOrder(root,res);
        return res;
    }
private:
    pair<int,int> postOrder(TreeNode* root, int& res){
        if (root == NULL)
            return make_pair(0,0);
        auto left = postOrder(root->left,res);
        auto right = postOrder(root->right,res);
        int curcount = left.first+right.first+1;
        int curv = left.second+right.second+root->val;
        res+=abs(curcount-curv);
        return {curcount,curv};
    }
};
