/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans = 0;
    int mabs(int x){
        return x>=0? x:-x;
    }
    void dfs(TreeNode* root, int &rem){
        if(root == NULL){
            return;
        }
        if(root->val >1){
            rem += root->val-1;
            root->val = 1;
        }
        if(root->val ==0 &&rem>0){
            root->val = 1;
            rem --;
        }
        int tmp = rem;
        dfs(root->left,rem);
        ans += mabs(tmp-rem);
        if(root->val ==0 &&rem>0){
            root->val = 1;
            rem --;
        }
        tmp = rem;
        dfs(root->right,rem);
        ans += mabs(tmp-rem);
        if(root->val ==0 &&rem>0){
            root->val = 1;
            rem --;
        }
        tmp = rem;
        dfs(root->left,rem);
        ans += mabs(tmp-rem);
        return;
    }
    int distributeCoins(TreeNode* root) {
        int rem = 0;
        dfs(root,rem);
        return ans;
    }
};
