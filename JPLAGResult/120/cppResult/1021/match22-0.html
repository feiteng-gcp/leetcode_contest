/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int distributeCoins(TreeNode* root) {
        if (root == nullptr) return 0;
        int moves = 0;
        distributeCoins_helper(root, moves);
        return moves;
    }
    
    // return coins weight
    int distributeCoins_helper(TreeNode* node, int& moves) {
        if (node == nullptr) return 0;
        int left = distributeCoins_helper(node->left, moves);
        int right = distributeCoins_helper(node->right, moves);
        moves += abs(left) + abs(right);
        int ret = left + right;
        ret += node->val - 1;
        return ret;
    }
};
