/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

int ans;

class Solution {
public:
    
    int dfs(TreeNode *root) {
        if (root == NULL)
            return 0;

        int l = dfs(root->left), r = dfs(root->right);
        root->val += l + r;
        ans += abs(root->val - 1);
        return root->val - 1;
    }
    
    int distributeCoins(TreeNode* root) {
        ans = 0;
        dfs(root);
        return ans;
    }
};
