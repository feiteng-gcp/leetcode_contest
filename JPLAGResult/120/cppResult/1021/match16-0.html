class Solution {
public:
  int ans;
  int dfs(TreeNode *root) {
    if (root == NULL) {
      return 0;
    }
    int l = dfs(root->left);
    int r = dfs(root->right);
    ans += max(l, -l);
    ans += max(r, -r);
    return l + r + root->val - 1;    
  }
  int distributeCoins(TreeNode* root) {
    ans = 0;
    dfs(root);
    return ans;
  }
};
