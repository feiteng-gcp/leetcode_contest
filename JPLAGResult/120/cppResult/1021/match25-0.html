   int dfs(TreeNode* root, int& total_move) {
    if(!root) {
      return 0;
    }
    if(!root->left && !root->right) {
      return root->val -1;
    }
    int left = dfs(root->left, total_move);
    int right = dfs(root->right, total_move);
    int sum = 0;
    sum += abs(left);
    sum += abs(right);
    total_move += sum;
    int coin =root->val - 1;
    coin += left;
    coin += right;
    return coin;
   }
    int distributeCoins(TreeNode* root) {
      int total_move = 0;
      dfs(root, total_move);
      return total_move;
    }
};
