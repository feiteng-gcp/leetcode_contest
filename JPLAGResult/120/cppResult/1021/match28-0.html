/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans=0;
    int post(TreeNode* root)
    {
        if(root==NULL)
            return 0;
        int l=post(root->left);
        int r=post(root->right);
        root->val=1-root->val;
        root->val+=l;
        root->val+=r;
        ans+=abs(l)+abs(r);
        return root->val;
        
    }
    int distributeCoins(TreeNode* root) {
        post(root);
        return ans;
    }
};
