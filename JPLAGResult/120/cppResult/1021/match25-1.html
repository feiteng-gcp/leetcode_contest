/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* tn,int &ans){
        if(!tn) return 0;
        if(!tn->left&&!tn->right) return tn->val-1;

        int left=dfs(tn->left,ans);
        int right=dfs(tn->right,ans);
        ans+=(abs(left)+abs(right));
        tn->val+=(left+right);
        return tn->val-1;
    }
    int distributeCoins(TreeNode* root) {
        int ans=0;
        dfs(root,ans);
        return ans;
    }
};
