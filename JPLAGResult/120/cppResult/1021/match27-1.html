/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    unordered_map<TreeNode*, int> cnt;
    int dfs(TreeNode* n) {
        if (n == NULL) return 0;
        if (cnt.find(n) == cnt.end()) {
            cnt[n] = n->val;
        }

        int coinsNum = cnt[n];
        int res = 0;
        
        res += dfs(n->left);
        coinsNum += cnt[n->left];
        res += abs(cnt[n->left]);
            
        res += dfs(n->right);
        coinsNum += cnt[n->right];
        res += abs(cnt[n->right]);
        
        cnt[n] = coinsNum-1;
        return res;
        /*
        for(TreeNode kid : n.children) {
            res += dfs(kid, cnt);
            coinsNum += cnt.get(kid);
            res += Math.abs(cnt.get(kid));
        }
        cnt.put(n, coinsNum - 1);
        return res;*/
    }
    
    int distributeCoins(TreeNode* root) {
        return dfs(root);
    }
};
