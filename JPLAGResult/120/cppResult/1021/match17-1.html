/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int res = 0;

public:
    int distributeCoins(TreeNode* root) {
        helper(root);
        
        return res;
    }
    
    int helper(TreeNode* root)
    {
        if(!root) return 0;
        
        int l = helper(root->left);
        int r = helper(root->right);
        
        res += abs(l);
        res += abs(r);
        
        return root->val + l + r -1;
    }
};
