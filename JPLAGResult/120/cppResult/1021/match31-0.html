/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int dfs(TreeNode *root, TreeNode *pre) {
        if(root == NULL) {
            return 0;
        }
        int ans = dfs(root->left, root) + dfs(root->right, root);
        if(pre) {
            if(root->val <= 0) {
                ans += abs(root->val)+1;
                pre->val += root->val - 1;
            }
            else {
                ans += root->val - 1;
                pre->val += root->val -1;
            }
        }
        return ans;
    }
public:
    int distributeCoins(TreeNode* root) {
        return dfs(root, NULL);   
    }
};
