/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int distributeCoins(TreeNode* root) {
        if (!root)
            return 0;
        int count = 0;
        helper(root, count);
        return count;
    }
    
    pair<int,int> helper(TreeNode *root, int &cnt) {
        if (!root)
            return {0,0};
        

        auto left = helper(root->left, cnt);
        auto right = helper(root->right, cnt);
        int val = root->val+left.first+right.first;
        int size = left.second+right.second+1;
        cnt += abs(val-size);
        return {val, size};
    }
};
