struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};
#endif

class Solution {
public:
    struct Data {
	int sz, has;
	LL ans;
	Data() {
	    sz = 0;
	    has = 0;
	    ans = 0;
	}
    };

    Data rec(TreeNode *x) {
	if (!x) return Data();
	Data l = rec(x->left);
	Data r = rec(x->right);
	Data d;
	d.sz = 1 + l.sz + r.sz;
	d.has = x->val + l.has + r.has;
	d.ans = abs(d.sz - d.has) + l.ans + r.ans;
	return d;
    }

    int distributeCoins(TreeNode* root) {
	Data d = rec(root);
	return d.ans;
    }
};
