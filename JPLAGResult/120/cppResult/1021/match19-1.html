class Solution {
public:
    int get_sum(TreeNode* node, int &nodes) {
        if(node == NULL) return 0;
        nodes += 1;
        return node->val + get_sum(node->left, nodes) + get_sum(node->right, nodes);
    }

    int distributeCoins(TreeNode* root) {
        if(root == NULL) return 0;
        int cur = root->val;
        int lns = 0;

        int rns = 0;
        int l = get_sum(root->left, lns);
        int r = get_sum(root->right, rns);
        int tmp = 0;
        tmp += abs(l-lns);
        tmp += abs(r-rns);
        return tmp + distributeCoins(root->left) + distributeCoins(root->right);
    }
};
