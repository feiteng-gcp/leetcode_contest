/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans;
    // size, coin
    int myfabs(int x) {
        return x < 0 ? -x : x;
    }
    pair<int,int> dfs(TreeNode* rt) {
        if (!rt) return make_pair(0, 0);
        pair<int,int> lef = dfs(rt->left), rig = dfs(rt->right);
        pair<int,int> cur;
        cur.first = lef.first + rig.first + 1;
        cur.second = lef.second + rig.second + rt->val;
        ans += myfabs(cur.first - cur.second);
        return cur;
    }
    int distributeCoins(TreeNode* root) {
        ans = 0;
        dfs(root);
        return ans;
    }
};
