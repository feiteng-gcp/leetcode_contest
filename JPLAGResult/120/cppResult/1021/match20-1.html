/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int distributeCoins(TreeNode* root) {
        return work(root).first;
    }
    

    pair<int, int> work(TreeNode* root) {
        if (!root) return {0, 0};
        auto left = work(root->left);
        auto right = work(root->right);
        int moves = left.first + right.first + abs(left.second) + abs(right.second);
        int balance = left.second + right.second + root->val - 1;
        return {moves, balance};
    }
};
