/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int distributeCoins(TreeNode* root) {
        ret = 0;
        help(root);

        return ret;
    }
    int help(TreeNode *root) {
        if (!root) return 0;
        int l = help(root->left);
        int r = help(root->right);
        int tmp = root->val + l + r - 1;
        ret += abs(tmp);
        return tmp;
    }
private:
    int ret;
};
