/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
	int step;
	//后序遍历  返回硬币-节点数
	int DFS(TreeNode* p) {
		if (NULL == p)return 0;
		int dl = DFS(p->left);
		int dr = DFS(p->right);
		step += abs(dl) + abs(dr);
		return dl + dr + p->val - 1;
	}
public:
	int distributeCoins(TreeNode* root) {
		step = 0;
		DFS(root);
		return step;
	}
};
