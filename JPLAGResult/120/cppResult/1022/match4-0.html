class Solution {
public:
    int n,m;
    int ans;
    int dx[4],dy[4];
    int len;
    void dfs(int x,int y,int l, vector<vector<int>>& grid){
        if(x<0 || x>=n ||y<0||y>=m || grid[x][y] == -1){
            return;
        }
        if(grid[x][y] == 2){
            if(l == len+1){
                ans++;
            }
            
            return;
        }
        int tmp = grid[x][y];
        grid[x][y] = -1;
        for(int i=0;i<4;i++){
            int nx = x+dx[i];
            int ny = y+dy[i];
            dfs(nx,ny,l+1,grid);
        }
        grid[x][y] =tmp;
        return;
    }
    int uniquePathsIII(vector<vector<int>>& grid) {
        int sx,sy,ex,ey;
        len = 0;
        for(int i=0;i<grid.size();i++){
            for(int j=0;j<grid[0].size();j++){
                if(grid[i][j] == 1){
                    sx = i, sy = j;
                }
                if(grid[i][j] == 2){
                    ex = i, ey = j;
                }
                if(grid[i][j] == 0){
                    len++;
                }
            }
        }
        n = grid.size();
        m = grid[0].size();
        cout<<n<<" "<<m<<endl;
        ans = 0;
        dx[0] = 1; dx[1] = -1; dx[2] = dx[3] = 0;
        dy[0] = dy[1] = 0; dy[2]= 1; dy[3] = -1;
        dfs(sx,sy, 0,grid);
        return ans;
    }
};
