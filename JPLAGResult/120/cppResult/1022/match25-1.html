class Solution {
public:
    int a,b,c,d;//index of start and end
    int count;//count of 0s;
    int ans;
    int uniquePathsIII(vector<vector<int>>& grid) {
        count=0;

        ans=0;
        for(int i=0;i<grid.size();i++)
        {
            for(int j=0;j<grid[0].size();j++)
            {
                if(grid[i][j]==0)count++;
                if(grid[i][j]==1){a=i;b=j;} 

                if(grid[i][j]==2){c=i;d=j;}
            }
        }
        tv(grid,0,a,b);
        return ans;
        
    }
    
    void tv(vector<vector<int>> grid,int c,int i,int j){
        
 
        if(grid[i][j]==-1)return;
        if(grid[i][j]==2){
            
                if(c==count)ans++;return;
        }
        if(grid[i][j]==0||grid[i][j]==1)
        {
            
            if(grid[i][j]==0)
            c++;
            grid[i][j]=-1;
            if(i<grid.size()-1)tv(grid,c,i+1,j);
            if(i>0)tv(grid,c,i-1,j);
            if(j>0)tv(grid,c,i,j-1);
            if(j<grid[0].size()-1)tv(grid,c,i,j+1);
                         
        }
        
        
    }
};
