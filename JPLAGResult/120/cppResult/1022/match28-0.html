class Solution {
public:
    int paths;
    int result;
    vector<vector<bool>> visited;
    int n,m;
    int uniquePathsIII(vector<vector<int>>& grid) {
         n = grid.size();
         m = grid[0].size();
        result = 0;
        paths = 0;
        
        int x = -1, y = -1;
        
        int ex = -1, ey = -1;
        
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < m; j++)
            {
                if (grid[i][j] == 0)
                    paths++;
                
                if (grid[i][j] == 1)
                {
                    x = i;
                    y = j;
                }
                
                if (grid[i][j] == 2)
                {
                    ex = i;
                    ey = j;
                }
            }
        }
        
        if (x == -1 || ex == -1)
            return 0;
        
        visited = vector<vector<bool>>(n, vector<bool>(m, false));
        visited[x][y] = true;
        
        int c = 0;
        helper(grid, x, y, c);
        
        return result;
    }
    
    void helper(vector<vector<int>> &grid, int x, int y, int &c)
    {
        if (grid[x][y] == 2)
        {
           // cout<<c <<" "<<paths<<endl;
            if (c-1 == paths)
                result++;
        }
        
        int dirx[] = {0, 0, 1, -1};
        int diry[] = {1, -1, 0, 0};
        
        for (int i = 0; i < 4; i++)
        {
            int nx = x+dirx[i];
            int ny = y+diry[i];
            
            if (nx >=0 && nx <n && 
               ny >=0 && ny < m &&
                grid[nx][ny] != -1 &&
               !visited[nx][ny])
            {
               // cout<<x<<" "<<y<<" -> "<<nx<<" "<<ny<<endl;
                visited[nx][ny] = true;
                c++;
                helper(grid, nx, ny, c);
                c--;
                visited[nx][ny] = false;
            }
        }
    }
    
    
};
