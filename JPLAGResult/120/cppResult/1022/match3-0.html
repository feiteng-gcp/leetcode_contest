class Solution {
public:
    int m, n;
    bool finished(vector<vector<int>>& grid){
        for(int i=0;i<m;++i){
            for(int j=0;j<n;++j){
                if(grid[i][j]!=-1&&grid[i][j]!=2)
                    return false;
            }
        }
        return true;
    }
    int dfs(vector<vector<int>>& grid, int i, int j){
        int ret = 0;
        if(grid[i][j]==2){
            if(finished(grid)) return 1;
            else return 0;
            //return 1;
        }
        grid[i][j] = -1;
        if(i>0&&grid[i-1][j]!=-1){
            ret+=dfs(grid,i-1,j);
        }
        if(i<m-1&&grid[i+1][j]!=-1){
            ret+=dfs(grid,i+1,j);
        }
        if(j>0&&grid[i][j-1]!=-1){
            ret+=dfs(grid,i,j-1);
        }
        if(j<n-1&&grid[i][j+1]!=-1){
            ret+=dfs(grid,i,j+1);
        }
        grid[i][j] = 0;
        return ret;
    }
    int uniquePathsIII(vector<vector<int>>& grid) {
        m = grid.size();
        n = grid[0].size();
        vector<vector<int>> temp = grid;
        int i,j;
        for(i=0;i<m;++i){
            for(j=0;j<n;++j){
                if(grid[i][j]==1){
                    goto out;
                }
            }
        }
        out:
        //cout<<i<<" "<<j<<endl;
        //return i;
        return dfs(temp,i,j);
    }
};
