class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {

        int posi, posj, spaceleft = 0, res = 0;
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid[0].size(); j++){
                if(grid[i][j] == 1){
                    posi = i;
                    posj = j;
                }
                if(grid[i][j] == 0)
                    spaceleft++;
            }
        }
        cout << "space: "  << spaceleft << endl;
        vector<pair<int, int>> path;
        back(grid, res, spaceleft, posi, posj, path);
        return res;
    }
    

    void back(vector<vector<int>>& grid, int& res, int spaceleft, int posi, int posj, vector<pair<int, int>> path){
        if(posi < 0 || posi >= grid.size() || posj < 0 || posj >= grid[0].size() || grid[posi][posj] == -1)
            return;
        if(grid[posi][posj] == 2 && spaceleft == 0){
            for(int k = 0; k < path.size(); k++)
                cout << path[k].first << "  " << path[k].second << endl;
            res++;
            cout << "****************" << endl;
            return;
        }
        else if(grid[posi][posj] == 2)
            return;
        
        int tmp = grid[posi][posj]-1;
        grid[posi][posj] = -1;
        path.push_back(pair<int, int> {posi, posj});
        back(grid, res, spaceleft+tmp, posi-1, posj, path);
        back(grid, res, spaceleft+tmp, posi+1, posj, path);
        back(grid, res, spaceleft+tmp, posi, posj-1, path);
        back(grid, res, spaceleft+tmp, posi, posj+1, path);
        grid[posi][posj] = 0;
    }
};
