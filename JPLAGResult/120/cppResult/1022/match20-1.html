class Solution {
    int m, n;
    int si, sj;
    int ei, ej;
    int cnt;
    int res = 0;
    
    int dx[4] = {1,0,-1,0};
    int dy[4] = {0,1,0,-1};
    
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        cnt = 0;
        m = grid.size();

        n = grid[0].size();
        
        for(int i = 0; i < m; i++)
        {
            for(int j = 0; j < n; j ++)
            {
                if(grid[i][j] == 1)
                {
                    si = i;
                    sj = j;
                }
                else if(grid[i][j] == 2)
                {
                    ei = i;
                    ej = j;
                }
                else if(grid[i][j] == 0)
                {
                    cnt++;
                }
            }
        }
        
        //cout<<si<<" "<<sj<<" "<<ei<<" "<<ej<<" "<<cnt<<endl;
        
        grid[si][sj] = -1;
        grid[ei][ej] = 0;
        cnt++;
        
        helper(grid, si, sj, cnt);
        
        return res;
    }
    
    void helper(vector<vector<int>>& grid, int i, int j, int left)
    {
        if(i == ei && j == ej && left == 0)
        {
            res++;

            return;
        }
        
        for(int d = 0; d < 4; d++)
        {
            int ii = i + dx[d];
            int jj = j + dy[d];
            
            if(valid(ii,jj) && grid[ii][jj] == 0)
            {
                grid[ii][jj] = -1;
                
                helper(grid, ii, jj, left-1);
                
                grid[ii][jj] = 0;
            }
        }
        
    }
    
    bool valid(int i, int j)
    {
        return i >= 0 && j >= 0 && i < m && j < n;
    }
};
