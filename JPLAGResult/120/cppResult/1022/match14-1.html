class Solution {
public:
    vector<vector<int>> grid;
    vector<vector<bool>> visit;

    int m;
    int n;
    const int dx[4] = {0, 0, 1, -1};
    const int dy[4] = {1, -1, 0, 0};
    
    int uniquePathsIII(vector<vector<int>>& grid_) {
        grid = grid_;
        m = grid.size();
        n = grid[0].size();

        visit = vector<vector<bool>>(m, vector<bool>(n, false));
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    visit[i][j] = true;
                    return dfs(i, j);
                }
            }
        }
        return 0;
    }
    
    bool checkEnd() {
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0 && !visit[i][j]) {
                    return false;
                }
            }
        }
        return true;
    }
    
    int dfs(int x, int y) {
        if (grid[x][y] == 2) {
            if (checkEnd()) {
                return 1;
            } else {
                return 0;
            }
        }
        int ret = 0;

        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            
            if (nx >= 0 && nx < m && ny >= 0 && ny < n) {
                if (grid[nx][ny] != -1 && !visit[nx][ny]) {
                    visit[nx][ny] = true;
                    ret += dfs(nx, ny);
                    visit[nx][ny] = false;
                }
                
            }
        }
        return ret;
    }
    
};
