bool vis[20][20];

int dx[4] = {-1, 0, 0, 1}, dy[4] = {0, -1, 1, 0};

class Solution {
public:
    int n, m, cnt, ans;
    
    bool judge(int x, int y) {
        if (x >= 0 && x < n && y >= 0 && y < m)
            return true;
        return false;
    }
    
    void dfs(vector<vector<int>>& grid, int x, int y, int val) {
        if (grid[x][y] == 2) {
            if (val == cnt)
                ++ans;
            return;
        }
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i], ny = y + dy[i];
            if (!judge(nx, ny)) continue;
            if (grid[nx][ny] == -1 || vis[nx][ny]) continue;
            vis[nx][ny] = true;
            dfs(grid, nx, ny, val + 1);
            vis[nx][ny] = false;
        }
        return;
    }
    
    int uniquePathsIII(vector<vector<int>>& grid) {
        memset(vis, false, sizeof(vis));
        n = grid.size(); m = grid[0].size();
        cnt = 0;
        int sx = -1, sy = -1;
        for (size_t i = 0; i < n; ++i) {
            for (size_t j = 0; j < m; ++j) {
                if (grid[i][j] == 1) {
                    sx = i, sy = j;
                } else if(grid[i][j] == -1) {
                    ++cnt;
                }
            }
        }
        cnt = n * m - cnt;
        ans = 0;
        vis[sx][sy] = true;
        dfs(grid, sx, sy, 1);
        return ans;
    }
};
