class Solution {
    int N,M, ret;
    int s, e;//start, end
    bitset<20> cannot, target, cur;
    void dfs(int id) {
        if (target == cur && id == e) ret++;
        int i = id/M;
        int j = id%M;

        int dx[4] = {0, 0, 1, -1};
        int dy[4] = {1, -1, 0, 0};
        for (int d = 0; d < 4; d++) {
            int nx = dx[d] + i;
            int ny = dy[d] + j;
            if (nx >= 0 && ny >=0 && nx < N && ny < M) {
                int nid = nx * M + ny;
                if (cur[nid] == 0 && cannot[nid] == 0) {
                    cur[nid] = 1;
                    dfs(nid);
                    cur[nid] = 0;
                }
            }
        }
    } 
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        N = grid.size();
        if (N == 0) return 0;
        M = grid[0].size();
        ret = 0;

        cannot =  target = cur = bitset<20>();
        for (int i = 0; i <N ;i++) {
            for (int j = 0; j <M; j++) {
                int id = i * M + j;
                
                if (grid[i][j] == -1) cannot[id] = 1;
                else {
                    target[id] = 1;
                    if (grid[i][j] == 1) s = id;
                    if (grid[i][j] == 2) e = id;
                }
            }
        }
        cout << target << endl;
        cout << cur << endl;
        cout << cannot << endl;
        cur[s] = 1;
        dfs(s);
        return ret;
    }
};
