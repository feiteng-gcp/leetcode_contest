class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {

        int res = 0;
        // find the start pos
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[0].size(); j++) {
                if (grid[i][j] == 1) {
                    //cout << "i is "<< i << " j is " << j << endl;
                    vector<vector<bool>> visited(grid.size(), vector<bool>(grid[0].size(), false));
                    visited[i][j] = true;
                    DFS(grid, visited, i, j, res);
                }
            }
        }
        return res;
    }
    
    void DFS(vector<vector<int>>& grid, vector<vector<bool>>& visited, int posx, int posy, int& res) {
        if (grid[posx][posy] == 2) {
            //cout << "end i is "<< posx << " j is " << posy << endl;
            if (checkValid(grid, visited)) res++;
            return;
        }
        if (grid[posx][posy] == -1) return;
        
        // grid[i][[j] = 0
        visited[posx][posy] = true;
        if (posx-1 >= 0 && !visited[posx-1][posy]) {
            DFS(grid,visited, posx-1, posy, res);
        }
        if (posy-1 >= 0 && !visited[posx][posy-1]) {
            DFS(grid,visited, posx, posy - 1, res);
        }
        if (posx+1 < grid.size() && !visited[posx+1][posy]) {
            DFS(grid,visited, posx+1, posy, res);
        }
        if (posy+1 < grid[0].size() && !visited[posx][posy+1]) {
            DFS(grid,visited, posx, posy + 1, res);
        }
        visited[posx][posy] = false;

    }
    
    bool checkValid(vector<vector<int>>& grid, vector<vector<bool>>& visited) {
        for (int i = 0; i < visited.size(); i++) {
            for (int j = 0; j < visited[0].size(); j++) {
                if (grid[i][j] == 0 && !visited[i][j]) return false;
            }
        }
        return true;
    }
};
