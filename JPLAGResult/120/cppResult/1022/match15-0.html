int dx[] = { 1,-1,0,0 };
int dy[] = { 0,0,1,-1 };
long long int dp[1 << 20][20];
class Solution {
public:
	
	int n;
	int m;
	int enc(int x, int y) {
		return x * m + y;
	}

	int uniquePathsIII(vector<vector<int>>& grid) {
		n = grid.size();
		m = grid[0].size();
		memset(dp, 0, sizeof(dp));
		for (int i = 0; i < grid.size(); i++) {
			for (int j = 0; j < grid[0].size(); j++) {
				if (grid[i][j] == 1) {
					dp[1 << enc(i, j)][enc(i, j)] = 1;
				}
			}
		}
		for (int i = 0; i < (1 << n * m); i++) {
			for (int r = 0; r < n*m; r++) {
				if ((i >> r) & 1) {

				}
				else {
					continue;
				}
				if (dp[i][r] == 0)continue;
				int x = r / m;
				int y = r % m;
				for (int k = 0; k < 4; k++) {
					int xx = x + dx[k];
					int yy = y + dy[k];
					if (xx >= 0 && yy >= 0 && xx < n&&yy < m) {
						if (grid[xx][yy] == -1)continue;
						if ((i >> enc(xx, yy)) & 1) {

						}
						else {
							int z = enc(xx, yy);
							dp[i | (1 << z)][z] += dp[i][r];
						}
					}
				}
			}
		}
		int u = 0;
		int goal = 0;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (grid[i][j] != -1) {
					u += (1 << enc(i, j));
				}
				if (grid[i][j] == 2) {
					goal = enc(i, j);
				}
			}
		}
		return dp[u][goal];
	}
};
