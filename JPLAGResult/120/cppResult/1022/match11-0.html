int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};
class Solution {
public:
  vector<vector<int>> vis;
  int ans;
  
  int ed;
  int n, m;
  int tot;
  
  bool inmap(int x, int y) {
    return x >= 0 && y >= 0 && x < n && y < m;
  }
  
  void dfs(vector<vector<int>>& grid, int id, int step = 0) {
   
    if (id == ed) {
      
      ans += step == tot + 1;
      return;
    }
    for (int i = 0; i < 4; i++) {
      int xx = id / m + dx[i];
      int yy = id % m + dy[i];
      if (inmap(xx, yy) && (grid[xx][yy] == 0 || grid[xx][yy] == 2) && vis[xx][yy] == 0) {
        vis[xx][yy] = 1;
        dfs(grid, xx * m + yy, step + 1);
        vis[xx][yy] = 0;
      }
      
    }
  }
  
  int uniquePathsIII(vector<vector<int>>& grid) {
    n = grid.size();
    m = grid[0].size();
    tot = 0;
    
    vis.resize(n, vector<int>(m, 0));
    ans = 0;
    int st;
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < m; j++) {
        if (grid[i][j] == 1) {
          st = i * m + j;
        }
        else if (grid[i][j] == 2) {
          ed = i * m + j;
        }
        else if (grid[i][j] == 0) {
          tot++;
        }
      }
    }
    dfs(grid, st);
    return ans;
  }
};
