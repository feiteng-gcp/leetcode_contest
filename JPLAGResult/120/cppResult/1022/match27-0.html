const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
const int N = 20;
int dp[N][1 << N];

class Solution {
public:
    int n, m;
    int s, t;
    vector<vector<int>> a;
    int run(int p, int mask) {
        if (p == t) {
            if (mask == (1 << (n * m)) - 1) return 1;
            return 0;
        }
        int& ret = dp[p][mask];
        if (ret >= 0) return ret;
        ret = 0;
        int x = p / m, y = p % m;
        for (int k = 0; k < 4; ++k) {
            int nx = x + d[k][0];
            int ny = y + d[k][1];
            if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
            if (a[nx][ny] < 0) continue;
            int np = nx * m + ny;
            if (mask & (1 << np)) continue;
            ret += run(np, mask | (1 << np));
        }
        //cout << p << " " << mask << " " << ret << endl;
        return ret;
    }
    int uniquePathsIII(vector<vector<int>>& a) {
        this->a = a;
        n = a.size(), m = a[0].size();
        int mask = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (a[i][j] == 1) {
                    s = i * m + j;
                } else if (a[i][j] == 2) {
                    t = i * m + j;
                } else if (a[i][j] < 0) {
                    mask |= 1 << (i * m + j);
                }
            }
        }
        
        memset(dp, 255, sizeof(dp));
        int ret = run(s, mask | (1 << s));
        return ret;
    }
};
