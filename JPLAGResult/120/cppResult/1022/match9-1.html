class Solution {
public:
	#define N 22
	bool g[N][N];
	int dir[4][2]={0,1,0,-1,1,0,-1,0};
	int ans,num;
	int n,m;
	bool judge(int x,int y){
		if(x>=0&&x<n&&y>=0&&y<m) return true;
		return false;
	}
	void dfs(int x,int y,int cnt,vector<vector<int>>& grid){
		//cout<<x<<','<<y<<endl;
		if(cnt==num){
			if(grid[x][y]==2){
				++ans;
			}

			return ;
		}
		for(int i=0;i<4;++i){
			int di=x+dir[i][0];
			int dj=y+dir[i][1];
			if(judge(di,dj)&&!g[di][dj]&&grid[di][dj]!=-1){
				g[di][dj]=1;
				dfs(di,dj,cnt+1,grid);
				g[di][dj]=0;
			}
		}
	}
    int uniquePathsIII(vector<vector<int>>& grid) {
    	n=grid.size();
    	m=grid[0].size();
    	bool ok=false;
    	int sx,sy;
    	num=0;
    	for(int i=0;i<n;++i) for(int j=0;j<m;++j) if(grid[i][j]!=-1) ++num;

		for(int i=0;i<n;++i){
			for(int j=0;j<m;++j){
				if(grid[i][j]==1){
					sx=i;
					sy=j;
					ok=true;
					break;
				}
			}
			if(ok) break;
		}
		ans=0;
		g[sx][sy]=1;
		dfs(sx,sy,1,grid);
		return ans;
    }
};
