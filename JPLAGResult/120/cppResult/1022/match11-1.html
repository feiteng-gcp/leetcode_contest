class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid)
    {
        int m = grid.size();
        if (m == 0)
        {
            return 0;
        }
        int n = grid[0].size();
        if (n == 0)
        {
            return 0;
        }

        int count0 = 0;
        int sx, sy, ex, ey;
        for (int i = 0; i < m; ++i)
        {
            for (int j = 0; j < n; ++j)
            {
                if (grid[i][j] == 0)
                {
                    ++count0;
                }
                else if (grid[i][j] == 1)
                {
                    sx = i;
                    sy = j;
                }
            }
        }
        vector<vector<bool>> visited(m, vector<bool>(n, false));
        return DFS(grid, m, n, sx, sy, 0, count0 + 1, visited);
    }
    
    int DFS(vector<vector<int>> &grid, int m, int n, int x, int y, int step, int count, vector<vector<bool>> &visited)
    {
        if (step == count)
        {
            return grid[x][y] == 2 ? 1 : 0;
        }
        int ret = 0;
        visited[x][y] = true;

        vector<vector<int>> offset({{-1, 0}, {1, 0}, {0, -1}, {0, 1}});
        for (int i = 0; i < 4; ++i)
        {
            int nx = x + offset[i][0];
            int ny = y + offset[i][1];
            if (nx < 0 || nx >= m || ny < 0 || ny >= n)
            {
                continue;
            }
            if (visited[nx][ny] || grid[nx][ny] == -1)
            {
                continue;
            }
            ret += DFS(grid, m, n, nx, ny, step + 1, count, visited);
        }
        visited[x][y] = false;
        return ret;
    }
};
