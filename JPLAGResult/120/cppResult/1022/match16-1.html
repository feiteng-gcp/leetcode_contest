class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        pair<int, int> start, end;
        
        int obs_num = 0;
        
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[0].size(); j++) {
                if (grid[i][j] == 1) {
                    start.first = i;
                    start.second = j;

                } else if (grid[i][j] == 2) {
                    end.first = i;
                    end.second = j;
                } else if (grid[i][j] == -1) {
                    obs_num++;
                }
            }
        }
        total_num = grid.size() * grid[0].size() - obs_num;
        
        vector<vector<bool>> isVisited(grid.size(), vector<bool>(grid[0].size(), false));
        isVisited[start.first][start.second] = true;
        
        ans = 0;
        search(start, 1, grid, isVisited);
        return ans;
    }
private:
    int total_num, ans;

    vector<pair<int, int>> dir {{1, 0}, {0, -1}, {-1, 0}, {0, 1}};
    void search(pair<int, int> curr, int visited_num, vector<vector<int>>& grid, vector<vector<bool>>& isVisited) {
        if (grid[curr.first][curr.second] == 2) {
            if (visited_num == total_num) ans++;
            return;
        }
        for (int i = 0; i < 4; i++) {
            pair<int, int> next = {curr.first + dir[i].first, curr.second + dir[i].second};
            if (next.first >= 0 && next.first < grid.size() && next.second >= 0 && next.second < grid[0].size()) {
                if (grid[next.first][next.second] >= 0 && !isVisited[next.first][next.second]) {
                    isVisited[next.first][next.second] = true;
                    search(next, visited_num + 1, grid, isVisited);
                    isVisited[next.first][next.second] = false;
                }
            }
        }
    }
};
