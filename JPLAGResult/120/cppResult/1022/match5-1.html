class Solution {
public:
    int res = 0;
    int h;
    int w;
    int target;

    const int help[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
    void dp(int x, int y, int num, vector<vector<int>>& grid){
        if(x<0||y<0||x>=h||y>=w||grid[x][y]==-1){
            return;
        }
        if(grid[x][y]==2){
            if(num == target){
                res++;
            }
            else return;
        }
        else{
            grid[x][y] = -1;

            num++;
            for(int i=0;i<4;i++){
                int nx = help[i][0]+x;
                int ny = help[i][1]+y;
                dp(nx,ny,num,grid);
            }
            num--;
            grid[x][y] = 0;
        }
    }
    
    
    int uniquePathsIII(vector<vector<int>>& grid) {
        h = grid.size();
        w = grid[0].size();
        res = 0;

        target = 0;
        int x = 0;
        int y = 0;
        for(int i=0;i<h;i++){
            for(int j=0;j<w;j++){
                target += (grid[i][j]==0);
                if(grid[i][j]==1){
                    x = i;
                    y=j;
                }
            }
        }
        target ++;
        dp(x,y,0,grid);
        return res;
        
    }
};
