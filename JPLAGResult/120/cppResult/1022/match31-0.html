class Solution {
public:
    int tot = 0;
    int edx = 0;
    int edy = 0;
    int r = 0;
    int c = 0;
    
    int dx[4] = {-1, 1, 0, 0};
    int dy[4] = {0, 0, -1, 1};
    
    void dfs(vector<vector<int>>& grid, vector<vector<bool>>& vis, int cnt, int x, int y, int &ans) {
        if(x==edx && y==edy) {
            if(cnt == tot) ans += 1;
            return ;
        }
        if(grid[x][y] == 0) ++cnt;
        for(int i=0; i<4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(nx >= 0 && nx < r && ny >= 0 && ny < c && !vis[nx][ny]) {
                if(grid[nx][ny] == -1) continue;
                vis[nx][ny] = true;
                dfs(grid, vis, cnt, nx, ny, ans);
                vis[nx][ny] = false;
            }
        }
    }
    
    int uniquePathsIII(vector<vector<int>>& grid) {
        r = grid.size();
        c = grid[0].size();
        vector<vector<bool>> vis = vector<vector<bool>>();
        for(int i=0; i<r; i++) vis.push_back(vector<bool>(c, false));
        int ans = 0;
        int x=0, y=0;
        for(int i=0; i<r; i++) {
            for(int j=0; j<c; j++) {
                if(grid[i][j] == 1) {x=i; y=j;}
                else if(grid[i][j] == 0) ++tot;
                else if(grid[i][j] == 2) {edx=i; edy=j;}
            }
        }
        vis[x][y] = true;
        dfs(grid, vis, 0, x, y, ans);
        return ans;
    }
};
