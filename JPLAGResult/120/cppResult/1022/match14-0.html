class Solution {
public:
	vector<vector<int> > grid;
	vector<vector<int> > t;
	const int dx[4] = { 0,1,-1,0 };
	const int dy[4] = { 1,0,0,-1 };
	int res = 0;
	int N, M;
	int c;
	void dfs(int i, int j) {
		t[i][j] = 1;
		c--;
		if (c == 0) {
			if (grid[i][j] == 2) {
				res++;
			}
			c++;
			t[i][j] = 0;
			return;
		}
		for (int k = 0; k < 4; k++) {
			int ni = i + dx[k];
			int nj = j + dy[k];
			if (ni >= 0 && ni < N && nj >= 0 && nj < M) {
				if (t[ni][nj] == 0) {
					dfs(ni, nj);
				}
			}
		}
		c++;
		t[i][j] = 0;
	}
	int uniquePathsIII(vector<vector<int> >& _grid) {
		grid = _grid;
		N = grid.size();
		M = grid[0].size();
		t.resize(N, vector<int>(M));
		int x, y;
		c = N * M;
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				if (grid[i][j] == -1) {
					t[i][j] = 1;
					c--;
				}
				else {
					t[i][j] = 0;
					if (grid[i][j] == 1) {
						x = i;
						y = j;
					}
				}
			}
		}
		dfs(x, y);
		return  res;
	}
};
