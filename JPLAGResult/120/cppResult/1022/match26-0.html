class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        if (grid.empty() || grid[0].empty()) return 0;
        M = grid.size(), N = grid[0].size();
        int si, sj, target;
        target = ret = 0;
        for (int i =0 ; i < M; ++ i) {
            for (int j = 0; j < N; ++ j) {
                if (grid[i][j] == 1) {
                    si = i;
                    sj = j;
                }
                if (grid[i][j] == 0) target ++;
            }
        }
        
        set<int> vst;
        vst.insert(si * N + sj);
        vector<vector<int>> path;
        path.push_back({si, sj});
        dfs(grid, si, sj, vst, target, path);
        return ret;
    }
    
    void dfs(vector<vector<int>>& grid, int i, int j, set<int> &vst, int target, vector<vector<int>> &path) {
        // for (auto &p : path) cout << p[0] << "," << p[1] << "  ";
        // cout << endl;
        if (grid[i][j] == 2) {
            if (vst.size() == target + 2) {
                ret ++;
            }
            return;
        }
        
        const static vector<vector<int>> dir = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        for (auto &d : dir) {
            auto ni = i + d[0], nj = j + d[1];
            if (ni >= 0 && ni < M && nj >= 0 && nj < N) {
                if (grid[ni][nj] == -1) continue;
                auto pos = ni * N + nj;
                if (!vst.count(pos)) {
                    vst.insert(pos);
                    // path.push_back({ni, nj});
                    dfs(grid, ni, nj, vst, target, path);
                    // path.pop_back();
                    vst.erase(pos);
                }
            }
        }
    }
    
private:
    int ret;
    int M, N;
};
