class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        int sr, sc;
        int tr, tc;

        int len = 0;
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[0].size(); ++j) {
                if (grid[i][j] == 1) {
                    sr = i;
                    sc = j;
                }
                if (grid[i][j] == 2) {
                    tr = i;
                    tc = j;
                }
                if (grid[i][j] != -1) {
                    ++len;
                }
            }
        }
        
        int res = 0;
        vector<vector<bool>> visited(grid.size(), vector<bool>(grid[0].size(), false));
        dfs(grid, sr, sc, 0, tr, tc, len, visited, res);
        return res;
    }
private:

    void dfs(const vector<vector<int>>& grid, int r, int c, int path, int tr, int tc, int len, vector<vector<bool>>& visited, int& res) {
        if (r == tr && c == tc) {
            if (path == len - 1) {
                ++res;
            }
            else {
                return;
            }
        }
        if (r >= 0 && r < grid.size() && c >= 0 && c < grid[0].size() && grid[r][c] != -1 && !visited[r][c]) {
            visited[r][c] = true;
            dfs(grid, r + 1, c, path + 1, tr, tc, len, visited, res);
            dfs(grid, r - 1, c, path + 1, tr, tc, len, visited, res);
            dfs(grid, r, c + 1, path + 1, tr, tc, len, visited, res);
            dfs(grid, r, c - 1, path + 1, tr, tc, len, visited, res);
            visited[r][c] = false;
        }
    }
};
