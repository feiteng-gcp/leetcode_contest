class Solution {
private:
    int ret;
    int mark[25];
    vector<int>a[25];
    void dfs(int x,int t,int cur,int num){
        if(cur==num&&x==t)ret++;
        for(int i=0;i<a[x].size();i++){
            int y=a[x][i];
            if(mark[y])continue;
            mark[y]=1;
            dfs(y,t,cur+1,num);
            mark[y]=0;
        }
    }
public:
    int uniquePathsIII(vector<vector<int>>& g) {
        int num=0;
        int n=(int)g.size();
        int m=(int)g[0].size();

        int dir[4][2]={{-1,0},{1,0},{0,-1},{0,1}};

        int s=-1,t=-1;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(g[i][j]==-1)continue;
                num++;
                if(g[i][j]==1)s=i*m+j;

                if(g[i][j]==2)t=i*m+j;
                for(int k=0;k<4;k++){
                    int x=i+dir[k][0];
                    int y=j+dir[k][1];
                    if(x<0||y<0||x>=n||y>=m||g[x][y]==-1)continue;
                    a[i*m+j].push_back(x*m+y);
                }
            }
        }
        //cout<<num<<endl;
        ret=0;
        memset(mark,0,sizeof(mark));
        mark[s]=1;
        dfs(s,t,1,num);
        return ret;
    }
};
