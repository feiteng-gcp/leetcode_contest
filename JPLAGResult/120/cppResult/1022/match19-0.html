class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        m = grid.size();
        if (m == 0) return 0;
        n = grid[0].size();
        int sr=0,sc=0,er=0,ec=0,count =0;
        for(int i =0; i <m;i++){
            for(int j =0; j <n;j++){
                if (grid[i][j] == 0)
                    count++;
                else if (grid[i][j] == 1){
                    sr = i;
                    sc = j;
                }else if (grid[i][j] == 2){
                    er = i;
                    ec = j;
                }
            }
        }
        vector<vector<bool>> seen(m,vector<bool> (n,false));
        int res = 0;
        dfs(grid,sr,sc,er,ec,-1,count,res,seen);
        return res;
    }
private:
    int m;
    int n;
    void dfs(vector<vector<int>>& grid, int sr, int sc, int er, int ec, int cc, 
             int count, int& res,vector<vector<bool>>& seen){
        if (sr < 0 || sr >= m || sc < 0 || sc >= n || seen[sr][sc] || grid[sr][sc] == -1)
            return;
        if (sr == er and sc == ec){
            if (cc == count)
                res++;
            return;
        }
        seen[sr][sc]= true;
        dfs(grid,sr+1,sc,er,ec,cc+1,count,res,seen);
        dfs(grid,sr-1,sc,er,ec,cc+1,count,res,seen);
        dfs(grid,sr,sc+1,er,ec,cc+1,count,res,seen);
        dfs(grid,sr,sc-1,er,ec,cc+1,count,res,seen);
        seen[sr][sc]= false;
    }
};
