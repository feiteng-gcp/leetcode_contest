int f[20][1 << 20];
class Solution {    
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        memset(f, 0, sizeof(f));
        int n = grid.size();
        int m = grid[0].size();
        //vector<vector<int>> f(n * m, vector<int>(1 << (n * m)));
        int x = 0;
        for (int i = 0; i < n; i++) for (int j = 0; j < m; j++) if (grid[i][j] == 1) x = i * m + j;
        f[x][1 << x] = 1;
        queue<int> q;
        q.push((1 << x) * n * m + x);
        while (!q.empty()) {
            int x = q.front();
            q.pop();
            int st = x / (n * m);
            x %= (n * m);
            const int mov[4][2] = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
            for (int k = 0; k < 4; k++) {
                int i = x / m + mov[k][0];
                int j = x % m + mov[k][1];
                if (i < 0 || i >= n || j < 0 || j >= m || grid[i][j] < 0) continue;
                int y = i * m + j;
                if (st & (1 << y)) continue;
                int new_st = st | (1 << y);
               // printf("%d %d %d %d %d\n", x / m, x % m, i, j, new_st);                
                if (f[y][new_st] == 0) q.push(new_st * n * m + y);
                f[y][new_st] += f[x][st];                                
            }
        }
        int des_st = 0, des = 0;
        for (int i = 0; i < n; i++) for (int j = 0; j < m; j++) {
            if (grid[i][j] >= 0) des_st += 1 << (i * m + j);
            if (grid[i][j] == 2) des = i * m + j;
        }
       // printf("%d %d\n", des, des_st);
        return f[des][des_st];
    }
};
