class Solution {
 public:
  void dfs(const vector<vector<int>>& grid, int r, int c, vector<vector<bool>>& visited,
           int empty_cnt, int total_empty, int& path_num) {
    if (grid[r][c] == 0) {
      empty_cnt++;
    } else if (grid[r][c] == 2) {
      if (empty_cnt == total_empty) {
        path_num++;
      }
      return;
    }
    const int m = grid.size();
    const int n = grid[0].size();
    vector<int> dx{1, 0, -1, 0};

    vector<int> dy{0, 1, 0, -1};
    for (int i = 0; i < 4; i++) {
      int x = r + dx[i];
      int y = c + dy[i];
      if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] == -1 || visited[x][y]) {
        continue;
      }
      visited[x][y] = true;
      dfs(grid, x, y, visited, empty_cnt, total_empty, path_num);
      visited[x][y] = false;
    }
  }
  int uniquePathsIII(vector<vector<int>>& grid) {
    if (grid.empty() || grid[0].empty()) {
      return 0;
    }
    const int m = grid.size();
    const int n = grid[0].size();
    int start_r, start_c;

    int total_empty = 0;
    for (int r = 0; r < m; r++) {
      for (int c = 0; c < n; c++) {
        if (grid[r][c] == 1) {
          start_r = r;
          start_c = c;
        } else if (grid[r][c] == 0) {
          total_empty++;
        }
      }
    }
    int path_num = 0;
    vector<vector<bool>> visited(m, vector<bool>(n, false));
    visited[start_r][start_c] = true;
    dfs(grid, start_r, start_c, visited, 0, total_empty, path_num);
    return path_num;
  }
};
