class Solution {
public:
    bool isValidPath(vector<vector<int>>& grid)
    {
        for(vector<int> grid_i : grid)
        {
            for(int i : grid_i)
            {
                if(i == 0)
                    return false;
            }
        }
        return true;
    }
    

    bool find_start(vector<vector<int>>& grid, int& m, int& n)
    {
        for(int i = 0; i < grid.size(); i++)
        {
            for(int j = 0; j < grid[i].size(); j++)
            {
                if(grid[i][j] == 1)
                {
                    m = i;
                    n = j;
                    return true;
                }

            }
        }
        return false;
    }
    
    int uniquePath_base1(vector<vector<int>>& grid, int m, int n)
    {
        int result = 0;
        if(grid[m][n] == 2)
        {
            if(isValidPath(grid))
                result = 1;
        }
        else if(grid[m][n] == 0)
        {
            result = uniquePath_base(grid, m, n);
        }
        return result;
    }
    
    int uniquePath_base(vector<vector<int>>& grid, int m, int n)
    {
        int result = 0;

        grid[m][n] = -1;
        if(n > 0) // тид
        {
            result += uniquePath_base1(grid, m, n - 1);
        }
        if(n < grid[0].size() - 1)
        {
            result += uniquePath_base1(grid, m, n + 1);
        }
        if(m > 0)
        {
            result += uniquePath_base1(grid, m - 1, n);
        }
        if(m < grid.size() - 1)
        {
            result += uniquePath_base1(grid, m + 1, n);
        }
        grid[m][n] = 0;
        return result;
    }
    
    int uniquePathsIII(vector<vector<int>>& grid) {
        if(grid.size() < 1)
            return 0;
        if(grid[0].size() < 1)
            return 0;
        int m = -1;
        int n = -1;
        if(!find_start(grid, m, n))
            return 0;
        return uniquePath_base(grid, m, n);
    }
};
