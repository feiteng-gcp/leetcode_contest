int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};     

struct State {
    int x, y, mask;
    State() {}
    State(int _x, int _y, int _m) {
        x = _x, y = _y, mask = _m;
    }
};

class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        int ans = 0;
        int sx, sy, ex, ey;
        int n = grid.size(), m = grid[0].size();

        int cnt = 0;
        for (int i = 0; i < n; ++ i) {
            for (int j = 0; j < m; ++ j) {
                if (grid[i][j] == 1) {
                    sx = i, sy = j;
                } else if (grid[i][j] == 2) {
                    ex = i, ey = j;
                }
                if (grid[i][j] != -1) ++ cnt;
            }
        }
        
        auto myHash = [&m](int x, int y) { return x * m + y; };
        
        queue<State> q;
        q.push(State(sx, sy, (1<<myHash(sx, sy))));
        while (!q.empty()) {
            auto cur = q.front(); q.pop();
            if (cur.x == ex && cur.y == ey) {
                if (__builtin_popcount(cur.mask) == cnt)
                    ++ ans;
                continue;

            }
            
            for (int i = 0; i < 4; ++ i) {
                int x = cur.x + dx[i];
                int y = cur.y + dy[i];
                if (x >= 0 && x < n && y >= 0 && y < m && grid[x][y] != -1 && (cur.mask & (1<<myHash(x, y))) == 0) {
                    q.push(State(x, y, cur.mask | (1<<myHash(x, y))));
                }
            }
        }
        return ans;
    }
};
