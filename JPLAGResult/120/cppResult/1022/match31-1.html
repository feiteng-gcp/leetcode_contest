class Solution {
public:
    int uniquePathsIII(vector<vector<int>>& grid) {
        int startx,starty,numEmpty=0;
        for(int i=0;i<grid.size();i++)
            for(int j=0;j<grid[i].size();j++)
            {
                if(grid[i][j]==1)startx=i,starty=j;
                else if(grid[i][j]==0)numEmpty++;
            }
        int ans=0;
        dfs(startx,starty,grid,numEmpty,ans);
        return ans;
    }
    

    int dx[4]={1,-1,0,0};
    int dy[4]={0,0,1,-1};
    
    void dfs(int x,int y,vector<vector<int>>& grid,int rem,int& ans)
    {
        //cout<<x<<" "<<y<<" "<<rem<<endl;
        if(grid[x][y]==2)
        {
            if(rem==0)ans++;
            return;
        }
        if(grid[x][y]==0)grid[x][y]=3;
        for(int i=0;i<4;i++)
        {
            int nx=x+dx[i],ny=y+dy[i];
            if(nx>=0&&nx<grid.size()&&ny>=0&&ny<grid[nx].size()&&(grid[nx][ny]==0||grid[nx][ny]==2))
            {
                if(grid[x][y]==3)dfs(nx,ny,grid,rem-1,ans);
                else dfs(nx,ny,grid,rem,ans);
            }
        }
        if(grid[x][y]==3)grid[x][y]=0;
    }
};
