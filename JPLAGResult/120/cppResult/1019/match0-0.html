class Solution {
public:
    vector<int> sortedSquares(vector<int>& A) {
        // int abs_min = INT_MAX;
        // int abs_min_index = 0;
        // int N = A.size();
        // for(int i=0;i<N; i++){
        //     if(abs(A[i])<abs_min){
        //         abs_min_index = i;
        //         abs_min = abs(A[i]);
        //     }
        // }
        vector<int> res;
        for(int i=0; i<A.size(); i++){
            res.push_back(A[i]*A[i]);
        }
        sort(res.begin(), res.end());
        return res;
    }
};
