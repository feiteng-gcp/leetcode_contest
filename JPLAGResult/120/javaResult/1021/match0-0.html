/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int distributeCoins(TreeNode root) {
        return dfs(root, new HashMap<>());
    }
    
    private int dfs(TreeNode root, Map<TreeNode, Integer> count) {
        if(!count.containsKey(root)) {
            count.put(root, root.val);
        }
        int coinsNum = count.get(root);
        int res = 0;
        if (root.left != null) {
            res += dfs(root.left, count);
            coinsNum += count.get(root.left);
            res += Math.abs(count.get(root.left));
        }
        if (root.right != null) {
            res += dfs(root.right, count);
            coinsNum += count.get(root.right);
            res += Math.abs(count.get(root.right));
        }
        count.put(root, coinsNum - 1);
        return res;
    }
}
