/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int move;
    public int distributeCoins(TreeNode root) {
        move = 0;
        dfs(root);
        return move;
    }
    
    private int dfs(TreeNode node) {
        if (node == null) return 0;
        int diffLeft = dfs(node.left);
        int diffRight = dfs(node.right);
        int diff = node.val - 1 + diffLeft + diffRight;
        move += Math.abs(diff);
        return diff;
    }
}
