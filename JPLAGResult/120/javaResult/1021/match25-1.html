/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int distributeCoins(TreeNode root) {
        int[] result = 
dfs(root);
        return result[2];
    }
    
    private int[] dfs(TreeNode root) {
        if (root == null) {
            return new int[]{0, 0, 0};
        }
        int[] l = dfs(root.left);
        int[] r = dfs(root.right);
        return new int[]{1 + l[0] + r[0], root.val + l[1] + r[1], Math.abs(l[0] - l[1]) + Math.abs(r[0] - r[1]) + l[2] + r[2]};
    }
}
