/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int counter = 0;
    public int distributeCoins(TreeNode root) {
        search(root);
        return counter;
    }
    private int[] search(TreeNode root) {
        // return first the number of nodes, second the number of coins
        if(root == null) return new int[]{0, 0};
        int[] left = search(root.left);
        int[] right = search(root.right);
        
        int root_left = Math.abs(left[1] - left[0]);
        int root_right = Math.abs(right[1] - right[0]);
        counter = counter + root_left + root_right;
        
        int totalCoins = left[0]+right[0] + root.val;
        int totalNodes = left[1]+right[1] + 1;
        return new int[]{totalCoins, totalNodes};
    }
}
