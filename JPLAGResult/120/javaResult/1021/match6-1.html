/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    
    HashMap<TreeNode,Integer> cs = new HashMap<>();
    HashMap<TreeNode,Integer> ns = new HashMap<>();
    
    void dfs(TreeNode root){
        int num = root.val;
        int z = 1;
        
        
        if(root.left!=null){
            dfs(root.left);
            num += cs.get(root.left);
            z += ns.get(root.left);
        }
        
        if(root.right!=null){
            dfs(root.right);
            num += cs.get(root.right);
            z += ns.get(root.right);
        }
        cs.put(root,num);
        ns.put(root,z);
        
    }
    int r = 0;
    void solve(TreeNode root){
        int c = cs.get(root);
        int n = ns.get(root);
        
        if(n>c){
            r+= n-c;
        }else if(c>n){
            r+= c-n;
        }
        
        if(root.left!=null){
            solve(root.left);
            
        }
        
        if(root.right!=null){
            solve(root.right);
            
        }
        
    }
    
    public int distributeCoins(TreeNode root) {
        if(root==null){
            return 0;
        }
        dfs(root);
        solve(root);
        
        return r;
    }
}
