/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int res;

    public int distributeCoins(TreeNode root) {
        res = 0;
        step(root);
        return res;
    }
    
    private int step(TreeNode p) {
        if (p == null) return 0;
        
        int l = step(p.left);
        int r = step(p.right);
        
        int needed = l + r + (1 - p.val);
        
        res += Math.abs(needed);
        return needed;
    }
}
