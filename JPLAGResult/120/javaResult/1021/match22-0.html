/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int count = 0;
    
    public int distributeCoins(TreeNode root) {
        help(root);
        return count;
    }
    
    int[] help(TreeNode treeNode) {
        if(treeNode == null) return new int[]{0,0};
        int[] left = help(treeNode.left), right = help(treeNode.right);
        count += Math.abs(left[0] - left[1]);
        count += Math.abs(right[0] - right[1]);
        return new int[]{left[0] + right[0] + treeNode.val, left[1] + right[1] + 1};
    }
}
