/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int move = 0;
    public int distributeCoins(TreeNode root) {
        
help(root);
        return move;
    }
    
    // help means the number of coins that will pass to the parent node, if it is a negative value, then it means the child requires that number of coins
    private int help(TreeNode node) {
        if(node == null) return 0;
        int left = help(node.left);
        int right = help(node.right);
        move += Math.abs(node.val + left + right - 1);
        return node.val + left + right - 1;
    }
}
