/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public int distributeCoins(TreeNode root) {
        int[] res=new int[1];
        checkDis(root,res);
        return res[0];
    }
    // index0 need dis, index1 need count, index2 offer dis,index3 offer count
    // index0 count index1 step
    public int checkDis(TreeNode curr,int[] res){
        if(curr==null){
            return 0;
        }
        int l=checkDis(curr.left,res);
        int r=checkDis(curr.right,res);
        int totalLeft=curr.val+l+r-1;
        res[0]=res[0]+Math.abs(totalLeft);
        return totalLeft;
    }
}
