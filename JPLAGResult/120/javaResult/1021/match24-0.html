/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int distributeCoins(TreeNode root) {
        ans = 0;
        dfs(root);
        return ans;
    }
    
    private int ans;
    
    public int[] dfs(TreeNode node) {
        if (node == null) {
            return new int[] {0, 0};
        }
        int coin = node.val, count = 1;
        int[] left = dfs(node.left), right = dfs(node.right);
        coin += left[0];
        coin += right[0];
        
        count += left[1];
        count += right[1];
        
        ans += Math.abs(coin - count);
        
        return new int[] {coin, count};
    }
}
