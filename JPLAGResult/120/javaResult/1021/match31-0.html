/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int distributeCoins(TreeNode root) {
        int ans = dcu(root, null);
        ans += dcd(root);
        return ans;
    }
    
    public int count(TreeNode root) {
        if (root == null)
            return 0;
        return count(root.left)+count(root.right)+1;
    }
    
    public int cc(TreeNode root) {
        if (root==null)
            return 0;
        return cc(root.left)+cc(root.right)+root.val;
    }
    
    public int dcu(TreeNode root, TreeNode p) {
        if (root==null)
            return 0;
        int ans = 0;
        int al = dcu(root.left, root);
        ans += al;
        int ar = dcu(root.right, root);
        ans += ar;
        int count = count(root);
        int cc = cc(root);
        if (cc > count) {
            root.val -= cc-count;
            p.val += cc-count;
            ans += cc-count;
        }
        return ans;
    }
    
    public int dcd(TreeNode root) {
        if (root == null)
            return 0;
        int col = count(root.left);
        int cor = count(root.right);
        int ccl = cc(root.left);
        int ccr = cc(root.right);
        int ans = 0;
        if (col > ccl) {
            root.val -= col-ccl;
            root.left.val += col-ccl;
            ans += col-ccl;
        }
        if (cor > ccr) {
            root.val -= cor-ccr;
            root.right.val += cor-ccr;
            ans += cor-ccr;
        }
        ans += dcd(root.left);
        ans += dcd(root.right);
        return ans;
    }
}
