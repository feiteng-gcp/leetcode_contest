class Solution {
    int m = 0;
    int n = 0;
    int[][] neis = new int[][]{{0,-1},{0,1},{-1,0},{1,0}};
    public int uniquePathsIII(int[][] grid) {
        if(grid == null || grid.length == 0 || grid[0].length == 0) {
            return 0;
        }
        int path = 0;// how many empty space 
        m = grid.length;
        n = grid[0].length;
        int[] start = new int[2];
        int[] end = new int[2];
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(grid[i][j] == 0) {
                    path++;
                }
                if(grid[i][j] == 1) {
                    start = new int[]{i,j};
                }
            }
        }
        int[] result = new int[1];
        dfs(grid,start,result,path,0);
        return result[0];
    }
    public void dfs(int[][] grid, int[] start,int[] result, int path, int cur) {
        int x = start[0];
        int y = start[1];
        if(cur > path) {
            return;
        }
        for(int[] nei : neis) {
            int row = x + nei[0];
            int col = y + nei[1];
            if(row >= m || row < 0 || col >= n || col < 0 || grid[row][col] == 1 || grid[row][col] == -1) {
                continue;
            }
            if(grid[row][col] == 2) {
                if(cur == path) {
                    result[0]++;
                }
                continue;
            } 
            grid[row][col] = -1;
            dfs(grid,new int[]{row,col},result,path,cur+1);
            grid[row][col] = 0;
        }
    }
}
