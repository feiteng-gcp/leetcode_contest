class Solution {
    int totalCount;
    int current;
    int n, m;
    boolean[][] isVisited;
    int res;
    int[][] grid;
    
    public int uniquePathsIII(int[][] grid) {
        totalCount = 0;
        current = 0;
        n = grid.length;
        m = grid[0].length;
        isVisited = new boolean[n][m];
        this.grid = grid;
        
        int x = -1;
        int y = -1;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0) totalCount++;
                else if (grid[i][j] == 1) {
                    x = i;
                    y = j;
                }
            }
        }
        
        dfs(x, y);
        
        return res;
    }
    
    private void dfs(int i, int j) {
        if (i < 0 || i >= n || j < 0 || j >= m || grid[i][j] == -1 || isVisited[i][j]) return;
        
        if (grid[i][j] == 2) {
            if (current == totalCount) res++;
            return;
        }
        
        if (grid[i][j] == 0) current++;
        isVisited[i][j] = true;
        dfs(i + 1, j);
        dfs(i - 1, j);
        dfs(i, j + 1);
        dfs(i, j - 1);
        isVisited[i][j] = false;
        if (grid[i][j] == 0) current--;
    }
}
