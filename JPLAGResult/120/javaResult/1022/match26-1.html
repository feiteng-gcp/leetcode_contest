public class Solution {
    private int res;
    private int pCnt;
    private int m, n;

    public int uniquePathsIII(int[][] grid) {
         m = grid.length;
         n = grid[0].length;
        int sm = 0, sn = 0, em = 0, en = 0;
        res = 0;
        pCnt = 0;
        for(int i = 0; i < m; i++ ){
            for (int j = 0; j < n; j++) {
                
if(grid[i][j] == 1){
                    sm = i;
                    sn = j;
                }
                else if(grid[i][j] == 2){
                    em = i;
                    en = j;
                }
                if(grid[i][j] != -1){
                    pCnt ++;
                
}
            }
        }
        boolean[][] vis = new boolean[m][n];
        vis[sm][sn] = true;
        helper(sm, sn, em, en, grid, vis, 1);
        System.out.println(res);
        return res;
    }

    private void helper(int sm, 
int sn, int em, int en, int[][] grid, boolean[][] vis, int cnt){
        if(sm == em && sn == en){
            if(cnt == pCnt){
                res ++;
            }
            return;
        }
        int i, j;
        i = sm +1;
        j = sn;
        if(i < m && grid[i][j] != -1 && !vis[i][j]){
            vis[i][j] = true;
            helper(i, j, em, en, grid, vis, cnt +1);
            vis[i][j] = false;
        }
        i = sm -1;
        j = sn;
        if(i >=0 && grid[i][j] != -1 && !vis[i][j]){
            vis[i][j] = true;
            helper(i, j, em, en, grid, vis, cnt +1);
            vis[i][j] = false;
        }
        i = sm;
        j = sn + 1;
        if(j < n && grid[i][j] != -1 && !vis[i][j]){
            vis[i][j] = true;
            helper(i, j, em, en, grid, vis, cnt +1);
            vis[i][j] = false;
        }
        i = sm;
        j = sn - 1;
        if(j >= 0 && grid[i][j] != -1 && !vis[i][j]){
            vis[i][j] = true;
            helper(i, j, em, en, grid, vis, cnt +1);
            vis[i][j] = false;
        }
    }
}
