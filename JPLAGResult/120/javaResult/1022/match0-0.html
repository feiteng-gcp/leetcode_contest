class Solution {
    int l  =-1;
    int ll = -1;
    int r = 0;
    int sx = -1;
        int sy = -1;
        int all = 0;
        int ex = -1;
        int ey = -1;
    
    void dfs(int cl,int i,int j,int[][] g){
        
        if(g[i][j]==2){
            if(cl==r){
                all++;
            }
            return;
        }
        
        int d1[] = new int[]{1,-1,0,0};
        int d2[] = new int[]{0,0,1,-1};
        for(int ii=0;ii<4;ii++){
            int nx = i+d1[ii];
            int ny = j+d2[ii];
            if(nx>=0&&nx<l&&ny>=0&&ny<ll){
                if(g[nx][ny]==0){
                    g[nx][ny] =-1;
                    dfs(cl+1,nx,ny,g);   
                    g[nx][ny] = 0;
                }else if(g[nx][ny]==2){
                    dfs(cl,nx,ny,g);   
                    
                }
            }
        }
        
    }
    
    public int uniquePathsIII(int[][] g) {
        l = g.length;
        ll = g[0].length;
        
        
        for(int i=0;i<l;i++){
            for(int j=0;j<ll;j++){   
                if(g[i][j]==0){
                    r++;
                }
                if(g[i][j]==1){
                    sx = i;
                    sy = j;
                    
                }
                if(g[i][j]==2){
                    ex = i;
                    ey = j;
                    
                }
            }
        }
        
        
        dfs(0,sx,sy,g);
        return all;
        
        
        
    }
}
