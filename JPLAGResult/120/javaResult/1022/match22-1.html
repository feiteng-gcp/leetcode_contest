class Solution {
    public int uniquePathsIII(int[][] grid) {
        int[] res={0};
        
int[] start=null;
        int[] end=null;
        int count=0;
        for(int i=0;i<grid.length;i++){
            for(int j=0;j<grid[0].length;j++){
                if(grid[i][j]==1){
                    start=new int[]{i,j};
                    count++;
                    grid[i][j]=0;
                }
                else if(grid[i][j]==2){
                    end=new int[]{i,j};
                    count++;
                    grid[i][j]=0;
                }
                else if(grid[i][j]==0){
                    count++;
                
}
            }
        }
        findPath(grid,start[0],start[1],end,new boolean[grid.length][grid[0].length],res,count,0);
        return res[0];
    }
    public void findPath(int[][] grid,int r,int c,int[] end,boolean[][] visited,int[] res,int count,int len){
        if(grid[r][c]==-1||visited[r][c]){
            return;
        }
        visited[r][c]=true;
        len++;
        if(len>count){
            visited[r][c]=false;
            
return ;
        }
        if(len==count){
            if(r==end[0]&&c==end[1]){
                res[0]++;
            }
            visited[r][c]=false;
            return ;
        }
        if(r-1>=0&&!visited[r-1][c]){
            findPath(grid,r-1,c,end,visited,res,count,len);
        }
        if(r+1<grid.length&&!visited[r+1][c]){
            findPath(grid,r+1,c,end,visited,res,count,len);
        }
        if(c-1>=0&&!visited[r][c-1]){
            findPath(grid,r,c-1,end,visited,res,count,len);
        }
        if(c+1<grid[0].length&&!visited[r][c+1]){
            findPath(grid,r,c+1,end,visited,res,count,len);
        }
        visited[r][c]=false;
    }
}
