class Solution {    
    private int ans = 0;
    private int zero;
    private int[][] grid;
    
    public int uniquePathsIII(int[][] grid) {
        this.grid = grid;
        int zero = 0;
        int[] start = new int[2];
        boolean[][] marked = new boolean[grid.length][grid[0].length];
        
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0)
                    zero++;
                else if (grid[i][j] == 1) {
                    start[0] = i;
                    start[1] = j;
                }
            }
        } 
        this.zero = zero;
        
        dfs(start[0], start[1], 0, marked);
        return ans;
    }
    
    private void dfs(int i, int j, int zero, boolean[][] marked) {        
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length || grid[i][j] == -1 || marked[i][j]) return;
        marked[i][j] = true;
        if (grid[i][j] == 2) {
            if (zero == this.zero)
                ans++;
        } else {
            if (grid[i][j] == 0)
                zero++;
            dfs(i - 1, j, zero, marked);
            dfs(i + 1, j, zero, marked);
            dfs(i, j - 1, zero, marked);
            dfs(i, j + 1, zero, marked);            
        }
        marked[i][j] = false;
    }
}
