class Solution {
    int sumOfPath;
    public int uniquePathsIII(int[][] grid) {
        sumOfPath = 0;
        
        int m = grid.length;
        int n = grid[0].length;
        int[] start = {0,0}, end = {0,0};
        
        
int count = 0;
        for (int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(grid[i][j] == 0) {
                    count++;
                } else if (grid[i][j] == 1) {
                    start[0] = i;
                    start[1] = j;
                } else if (grid[i][j] == 2) {
                    end[0] = i;
                    end[1] = j;
                
}
            }
        }
        boolean[][] visited = new boolean[m][n];
        visited[start[0]][start[1]] = true;
        dfs(start, end, 0, count, visited, grid);
        return sumOfPath;
    }
    private static int[] dir = {-1,0,1,0,-1};
    private void dfs(int[] cur, 
int[] end, int path, int count, boolean[][] visited, int[][] grid) {
        if(cur[0] == end[0] && cur[1] == end[1]) {
            if(path == count + 1) {
                sumOfPath++;
            }
            return;
        }
        for(int d = 0;d < 4; d++) {
            int nx = cur[0] + dir[d];
            int ny = cur[1] + dir[d + 1];
            if(nx <0 || nx >= grid.length || ny<0 || ny >= grid[0].length) {
                continue;
            }
            if(visited[nx][ny]) {
                continue;
            }
            if(grid[nx][ny] == -1) {
                continue;
            }
            visited[nx][ny] = true;
            dfs(new int[]{nx, ny}, end, path + 1, count, visited, grid);
            visited[nx][ny] = false;
        }
    }
}
