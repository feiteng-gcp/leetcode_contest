class Solution {
    
    private static int[][] D = new int[][] {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    
    public int uniquePathsIII(int[][] grid) {
        this.N = grid.length;
        this.M = grid[0].length;
        this.f = new boolean[N][M];
        this.count = new int[N][M];
        this.ans = 0;
        this.target = 0;
        this.grid = grid;
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (grid[i][j] != -1) {
                    this.target++;
                    for (int[] d : D) {
                        int nx = i + d[0], ny = j + d[1];
                        if (nx < N && ny < M && nx >= 0 && ny >= 0 && grid[nx][ny] != -1) {
                            count[i][j]++;
                        }
                    }
                }
            }
        }
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (grid[i][j] == 1) {
                    f[i][j] = true;
                    dfs(i, j, 1);
                }
            }
        }
        return ans;
    }
    
    private void dfs(int x, int y, int sum) {
        if (grid[x][y] == 2) {
            if (sum == target) {
                ans++;
            }
        } else {
            List<int[]> list = new ArrayList<>();
            for (int[] d : D) {
                int nx = x + d[0], ny = y + d[1];
                if (nx < N && ny < M && nx >= 0 && ny >= 0 && grid[nx][ny] != -1 && !f[nx][ny]) {
                    list.add(new int[] {nx, ny});
                }
            }
            for (int i = 0; i < list.size(); i++) {
                int nx = list.get(i)[0], ny = list.get(i)[1];
                boolean flag = true;
                for (int j = 0; j < list.size(); j++) {
                    if (i != j && count[list.get(j)[0]][list.get(j)[1]] <= 1) {
                        flag = false;
                    }
                }
                if (flag) {
                    for (int j = 0; j < list.size(); j++) {
                        count[list.get(j)[0]][list.get(j)[1]]--;
                    }
                    f[nx][ny] = true;
                    dfs(nx, ny, sum + 1);
                    f[nx][ny] = false;
                    for (int j = 0; j < list.size(); j++) {
                        count[list.get(j)[0]][list.get(j)[1]]++;
                    }
                }
            }
        }
    }
    
    private int[][] grid;
    private int[][] count;
    private boolean[][] f;
    private int N, M, ans, target;
}
