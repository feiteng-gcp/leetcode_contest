class Solution {
    public int uniquePathsIII(int[][] grid) {
        int start = 0, end = 0, length = 0;
        
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    start = i * grid[0].length + j;
                } else if (grid[i][j] == 2) {
                    end = i * grid[0].length + j;
                } else if (grid[i][j] == 0) {
                    length++;
                }
            }
        }
        
        Set<Integer> set = new HashSet<>();
        set.add(start);
        
        return dfs(start, end, length, grid, set);
    }
    
    private int dfs(int index, int end, int length, int[][] grid, Set<Integer> path) {
        if (index == end) {
            return path.size() == length + 2 ? 1 : 0;
        }
        int[] dirRow = {-1, 1, 0, 0};
        int[] dirCol = {0, 0, -1, 1};
        int row = index / grid[0].length;
        int col = index % grid[0].length;
        int result = 0;
        
        for (int i = 0; i < 4; i++) {
            int nextRow = row + dirRow[i];
            int nextCol = col + dirCol[i];
            int nextIndex = nextRow * grid[0].length + nextCol;
            if (nextRow >= 0 && nextRow < grid.length && nextCol >= 0 && nextCol < grid[0].length &&
                grid[nextRow][nextCol] != -1 && path.add(nextIndex)
               ) {
                result += dfs(nextIndex, end, length, grid, path);
                path.remove(nextIndex);
            }
        }
        
        return result;
    }
}
