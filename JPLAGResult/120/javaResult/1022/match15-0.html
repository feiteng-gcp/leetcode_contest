class Solution {
    public int uniquePathsIII(int[][] grid) {
        int count0 = 0;
        int[] start = null;
        for(int i = 0 ; i < grid.length;i++) {
            for(int j= 0 ; j < grid[0].length;j++) {
                if(grid[i][j] == 0) count0++;
                else if(grid[i][j] == 1) start = new int[]{i,j};
            }
        }
        return dfs(start[0],start[1],grid,new boolean[grid.length][grid[0].length],count0);
    }
    
    int[] dirs = new int[]{-1,0,1,0,-1};
    int dfs(int x, int y, int[][] grid, boolean[][] visited, int count0) {
        if(x >=0 && x < grid.length && y >=0 && y < grid[0].length && !visited[x][y]) {
            if(grid[x][y] == 0 || grid[x][y] == 1) {
                visited[x][y] = true;
                if(grid[x][y] == 0) count0--;
                int result = 0;
                for(int i = 0 ; i < 4;i++) {
                    result += dfs(x + dirs[i], y + dirs[i+1], grid,visited,count0);
                }
                visited[x][y] = false;
                return result;
            }else if(grid[x][y] == 2) {
                if(count0 == 0) return 1;
                else return 0;
            }
        }
        return 0;
    }
}
