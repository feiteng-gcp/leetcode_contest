class Solution {
    int empties = 0;
    int counter = 0;
    public int uniquePathsIII(int[][] grid) {
        int[] start = new int[]{0,0};
        int[] end = new int[]{0,0};
        
for(int i = 0; i < grid.length; i++) {
            for(int j = 0; j < grid[0].length; j++) {
                if(grid[i][j] == 1) start = new int[]{i, j};
                if(grid[i][j] == 2) end = new int[]{i, j};
                if(grid[i][j] == 0) empties++;
            }
        }
        List<int[]> path = new LinkedList<>();
        path.add(start);
        Set<Integer> visited = new HashSet<>();
        visited.add(code(start));
        search(grid, start[0], start[1], path, visited);
        return counter;
    }
    private int code(int[] input) {
        
return input[0]*30 + input[1];
    }
    private void search(int[][] grid, int x, int y, List<int[]> path, Set<Integer> visited) {
        if(grid[x][y] == 2) {
            if(path.size() == empties + 2) counter++;
            return;
        }
        int[][] dirs = new int[][]{{0,1}, {0,-1}, {1,0}, {-1,0}};
        for(int[] dir: dirs) {
            int xx = x + dir[0]; int yy = y + dir[1];
            if(xx >= 0 && yy >= 0 && xx < grid.length && yy < grid[0].length && grid[xx][yy] != -1) {
                int[] next = new int[]{xx, yy};
                if(!visited.contains(code(next))) {
                    visited.add(code(next));
                    path.add(next);
                    search(grid, xx, yy, path, visited);
                    visited.remove(code(next));
                    path.remove(path.size()-1);
                }
            }
        }
    }
}
