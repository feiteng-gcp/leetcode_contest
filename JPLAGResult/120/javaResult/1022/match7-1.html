class Solution {
    public int uniquePathsIII(int[][] grid) {
        int h = grid.length;
        int w = grid[0].length;
        
int n = h * w;
        int m = 1 << n;
        int[][] dp = new int[m][n];
        int start = 0;
        int end = 0;
        int visitTotal = 2;
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                if (grid[i][j] == 1) {
                    start = i * w + j;
                }
                if (grid[i][j] == 2) {
                    end = i * w + j;
                }
                if (grid[i][j] == 0) {
                    visitTotal++;
                }
            }
        }

        dp[addCell(0, start)][start] = 1;
        
int[] nextX = new int[]{0,0,1,-1};
        int[] nextY = new int[]{1,-1,0,0};
        for (int set = 0; set < m; set++) {
            for (int i = 0; i < n; i++) {
                int x = i / w;
                int y = i % w;
                if (dp[set][i] > 0) {
                    for (int k = 0; k < 4; k++) {
                        int x1 = x + nextX[k];
                        int y1 = y + nextY[k];
                        if (x1 < 0 || y1 < 0 || x1 >= h || y1 >= w || grid[x1][y1] == -1) continue;
                        int j = x1 * w + y1;
                        if (!hasCell(set, j)) {
                            // visit j
                            dp[addCell(set, j)][j] += dp[set][i];
                        
}
                    }
                }
            }
        }
        return dp[(1 << visitTotal) - 1][end];
    }

    public int addCell(int set, int i) {
        return set | (1 << i);
    }

    public boolean hasCell(int set, int i) {
        return ((set >> i) & 1) == 1;
    }
}
