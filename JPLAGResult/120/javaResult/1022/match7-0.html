	class Solution {
	    public int uniquePathsIII(int[][] grid) {
	        int n = grid.length, m = grid[0].length;
	        int[][] dp = new int[1<<n*m][n*m];
	        int st = -1, en = -1;
	        int all = 0;
	        for(int i = 0;i < n;i++){
	        	for(int j = 0;j < m;j++){
	        		if(grid[i][j] == 1){
	        			st = i*m+j;
	        		}
	        		if(grid[i][j] == 2){
	        			en = i*m+j;
	        		}
	        		if(grid[i][j] >= 0){
	        			all |= 1<<i*m+j;
	        		}
	        	}
	        }
	        
	        int[] dr = { 1, 0, -1, 0 };
			int[] dc = { 0, 1, 0, -1 };
	        for(int i = 0;i < 1<<n*m;i++){
	        	if(Integer.bitCount(i) == 1){
	        		if(i == 1<<st){
	        			dp[i][st] = 1;
	        		}
	        	}else{
	        		for(int j = 0;j < n*m;j++){
	        			if(i<<~j<0 && grid[j/m][j%m] >= 0){
	        				int r = j/m, c = j%m;
	        				for(int t = 0;t < 4;t++){
	        					int nr = r + dr[t], nc = c + dc[t];
	        					if(nr >= 0 && nr < n && nc >= 0 && nc < m){
	        						int k = nr*m+nc;
	        						if(i<<~k<0 && grid[nr][nc] >= 0){
		        						dp[i][j] += dp[i^1<<j][k];
	        						}
	        					}
	        				}
	        			}
	        		}
	        	}
	        }
	        return dp[all][en];
	    }
	}
