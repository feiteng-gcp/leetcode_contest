class Solution {
    boolean[][] vis;
    int sx;
    int sy;
    int ex;
    int ey;
    int zeros;
    int total;
    int[][] grid;
    int[] dx = {0, 0, -1, 1};
    int[] dy = {-1, 1, 0, 0};
    public int uniquePathsIII(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        this.grid = grid;
        List<Integer> ls = new ArrayList<>();
        vis = new boolean[n][m];
        zeros = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    sx = i;
                    sy = j;
                }

                if (grid[i][j] == 2) {
                    ex = i;
                    ey = j;
                }

                if (grid[i][j] == 0) {
                    zeros++;
                }
            }
        }

        total = 0;
        dfs(0, sx, sy);

        return total;
    }

    void dfs(int cnt, int row, int col) {
        if (row < 0 || row >= grid.length || col < 0 || col >= grid[0].length) {
            return;
        }

        if (vis[row][col]) {
            return;
        }

        if (grid[row][col] == -1) {
            return;
        }

        if (row == ex && col == ey) {
            if (cnt == zeros + 1) {
                total++;
            }

            return;
        }

        vis[row][col] = true;
        for (int i = 0; i < dx.length; i++) {
            int nrow = row + dx[i];
            int ncol = col + dy[i];
            dfs(cnt + 1, nrow, ncol);
        }

        vis[row][col] = false;
    }
}
