class Solution {
    public int uniquePathsIII(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        Set<Integer> remaining = new HashSet<>();
        
int startI = 0, startJ = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0) {
                    remaining.add(i * m + j);
                
} else if (grid[i][j] == 1) {
                    startI = i;
                    startJ = j;
                }
            }
        }
        return uniquePathsIII(grid, remaining, startI, startJ);
    }
    
    private int uniquePathsIII(int[][] grid, Set<Integer> remaining, int i, int j) {
        int count = 0;
        count += checkAndFind(grid, remaining, i+1, j);
        count += checkAndFind(grid, remaining, i, j+1);
        count += checkAndFind(grid, remaining, i-1, j);
        count += 
checkAndFind(grid, remaining, i, j-1);
        return count;
    }
    
    private int checkAndFind(int[][] grid, Set<Integer> remaining, int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length) return 0;
        if (remaining.isEmpty() && grid[i][j] == 2) return 1;
        
        int index = i * grid[0].length + j;
        if (remaining.remove(index)) {
            int count = uniquePathsIII(grid, remaining, i, j);
            remaining.add(index);
            return count;
        }
        return 0;
    }
}
