class Solution {
    int count = 0;
    public int uniquePathsIII(int[][] A) {
        int x0 = 0, y0 = 0;
        int num = 0;
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                if (A[i][j] == 1) {
                    x0 = i;
                    y0 = j;
                }
                if (A[i][j] != -1) {
                    num++;
                }
            }
        }

        boolean[][] visited = new boolean[A.length][A[0].length];
        visited[x0][y0] = true;
        dfs(A, x0, y0, num - 1, visited);
        return count;
    }
    int[][] dir = new int[][]{{0, -1}, {-1, 0}, {0, 1}, {1, 0}};
    public void dfs(int[][] A, int x, int y, int num, boolean[][] visited) {
        if (A[x][y] == 2) {

            if (num == 0) {
                count++;
            }
            return;
        }
        
        for (int i = 0; i < 4; i++) {
            int xx = x + dir[i][0];
            int yy = y + dir[i][1];
            if (xx < 0 || yy < 0 || xx >= A.length || yy >= A[0].length) {
                continue;
            }
            if (visited[xx][yy] || A[xx][yy] == -1) {
                continue;
            }
            visited[xx][yy] = true;
            dfs(A, xx, yy, num - 1, visited);
            visited[xx][yy] = false;
        }
    }
}
