class Solution {
    private int result;
    int rows, cols;
    
    public int uniquePathsIII(int[][] grid) {
        rows = grid.length;
        cols = grid[0].length;
        
int startY = 0, startX = 0;
        int count = 0;
        for (int y = 0; y < rows; y++) {
            for (int x = 0; x < cols; x++) {
                if (grid[y][x] == 1) {
                    startY = y;
                    startX = x;
                } else if (grid[y][x] == 0) {
                    count++;
                }
            }
        }
        
        boolean[][] visited = new boolean[rows][cols];
        visited[startY][startX] = true;
        result = 0;
        dfs(grid, startY, startX, visited, count);
        return result;
    }
    
    
    private int[] dx = new int[]{1, 0, -1, 0};
    private int[] dy = new int[]{0, -1, 0, 1};
    private void dfs(
int[][] grid, int y, int x, boolean[][] visited, int count) {
        if (grid[y][x] == 2) {
            if (count == -1) result++;
            return;
        }
        
        
for (int i = 0; i < 4; i++) {
            int nextY = y + dy[i];
            int nextX = x + dx[i];
            if (nextY >= 0 && nextY < rows && nextX >= 0 && nextX < cols 
                && !visited[nextY][nextX] && grid[nextY][nextX] != -1) {
                visited[nextY][nextX] = true;
                dfs(grid, nextY, nextX, visited, count - 1);
                visited[nextY][nextX] = false;
            }
        }
    }
}
