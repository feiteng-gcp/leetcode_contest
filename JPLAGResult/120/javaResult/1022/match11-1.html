class Solution {
    int total;
    public int uniquePathsIII(int[][] grid) {
        if(grid == null || grid.length == 0) return 0;
        total = 0;
        int targetStep = 0;
        
int[] start = new int[2];
        for(int i = 0; i < grid.length; i++) {
            for(int j = 0; j < grid[0].length; j++) {
                if(grid[i][j] == 0) targetStep++;
                else if(grid[i][j] == 1) {
                    start[0] = i;
                    start[1] = j;
                }
            }
        }
        help(0, start[0]+1, start[1], grid, targetStep);
        help(0, start[0]-1, start[1], grid, targetStep);
        help(0, start[0], start[1]+1, grid, targetStep);
        
help(0, start[0], start[1]-1, grid, targetStep);
        return total;
    }
    
    private void help(int steps, int x, int y, int[][] grid, int targetStep) {
        if(x >= grid.length || x < 0 || y >= grid[0].length || y < 0) return;
        if(steps == targetStep && grid[x][y] == 2) {total++; return;}
        if(grid[x][y] != 0) return;
        grid[x][y] = -1;
        help(steps+1, x+1, y, grid, targetStep);
        help(steps+1, x-1, y, grid, targetStep);
        help(steps+1, x, y+1, grid, targetStep);
        help(steps+1, x, y-1, grid, targetStep);
        grid[x][y] = 0;
    }
}
