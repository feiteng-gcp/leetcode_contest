class Solution {
    private int[] dr = new int[]{0, 1, 0, -1};
    private int[] dc = new int[]{1, 0, -1, 0};
    
private int[][] grid;
    private int R = 0;
    private int C = 0;
    private int res = 0;
    private int squareNum = 0;
    
    public int uniquePathsIII(int[][] grid) {
        this.grid = grid;
        R = grid.length;
        C = grid[0].length;
        int[] start = new int[2];
        
for (int r = 0; r < R; ++r) {
            for (int c = 0; c < C; ++c) {
                if (grid[r][c] == 0) squareNum++;
                else if (grid[r][c] == 1) start = new int[]{r, c}
;
            }
        }
        dfs(start[0], start[1], squareNum);
        return res;
    }
    
    private void dfs(int r, int c, int remain) {
        if (grid[r][c] == 2) {
            if (remain == 0) res++;
            return;
        }
        grid[r][c] = -1;
        
        for (int i = 0; i < 4; ++i) {
            int nextR = r + dr[i];
            int nextC = c + dc[i];
            if (isValid(nextR, nextC)) {
                dfs(nextR, nextC, grid[nextR][nextC] == 0 ? remain - 1 : remain);
            }
        }
        
        grid[r][c] = 0;
    }
    
    private boolean isValid(int r, int c) {
        return r >= 0 && r < R && c >= 0 && c < C && grid[r][c] != -1;
    }
}
