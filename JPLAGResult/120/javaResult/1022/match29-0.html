class Solution {
    private final static int[][] DIR = new int[][] { { 0, 1 }, { 0, -1 }, { 1, 0 }, { -1, 0 } };
	private int[][] grid;
	private boolean[][] visited;
	private int n;
	private int m;

	public int uniquePathsIII(int[][] grid) {
		this.grid = grid;
		this.n = grid.length;
		this.m = grid[0].length;
		this.visited = new boolean[n][m];
		int length = 0;
		int startI = -1;
		int startJ = -1;
		int endI = -1;
		int endJ = -1;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (grid[i][j] == -1) {
					continue;
				}
				length++;
				if (grid[i][j] == 1) {
					startI = i;
					startJ = j;
					visited[i][j] = true;
				} else if (grid[i][j] == 2) {
					endI = i;
					endJ = j;
				}
			}
		}
		int total = uniquePaths(startI, startJ, endI, endJ, length, 1);
		return total;
	}

	private int uniquePaths(int curI, int curJ, int endI, int endJ, int length, int curLeng) {
		if (length == curLeng) {
			if (curI == endI && curJ == endJ) {
				return 1;
			} else {
				return 0;
			}
		}
		if (curI == endI && curJ == endJ) {
			return 0;
		}
		int cur = 0;
		for (int[] dir : DIR) {
			int nextI = curI + dir[0];
			int nextJ = curJ + dir[1];
			if (nextI >= 0 && nextJ >= 0 && nextI < n && nextJ < m && grid[nextI][nextJ] != -1
					&& !visited[nextI][nextJ]) {
				visited[nextI][nextJ] = true;
				cur += uniquePaths(nextI, nextJ, endI, endJ, length, curLeng + 1);
				visited[nextI][nextJ] = false;
			}
		}
		return cur;
	}
}
