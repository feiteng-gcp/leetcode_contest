class Solution {
    
    private static int[] DIR = {-1, 0, 1, 0, -1};
    
    private int ans = 0;
    
    public int uniquePathsIII(int[][] grid) {
        int n = grid.length, m = grid[0].length, empty = 0;
        int x = -1, y = -1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0) {
                    empty ++;
                } else if (grid[i][j] == 1) {
                    x = i;
                    y = j;
                }
            }
        }
        // System.out.println(empty);
        helper(grid, x, y, 0, empty, new boolean[n][m]);
        return ans;
    }
    
    private void helper(int[][] grid, int x, int y, int cnt, int empty, boolean[][] used) {
        int n = grid.length, m = grid[0].length;
        if (x < 0 || x >= n || y < 0 || y >= m) {
            return;
        }
        if (used[x][y] || grid[x][y] == -1) {
            return;
        }
        // System.out.println(x + ", " + y + ", " + grid[x][y] + ", " + cnt);
        if (grid[x][y] == 2) {
            if (cnt == empty) {
                ans ++;
            }
            return;
        }
        used[x][y] = true;
        if (grid[x][y] == 0) {
            cnt ++;
        }
        for (int k = 0; k < DIR.length - 1; k ++) {
            int xx = x + DIR[k], yy = y + DIR[k + 1];
            helper(grid, xx, yy, cnt, empty, used);
        }
        used[x][y] = false;
        if (grid[x][y] == 0) {
            cnt --;
        }
    }
}
