class Solution {
    int ans = 0;
    private int[][] grid;
    private boolean[][] seen;
    public int uniquePathsIII(int[][] grid) {
        if (grid.length == 0 || grid[0].length == 0) return 0;
        
        int[] start = new int[2];
        int[] end = new int[2];
        int zeroes = 0;
        this.grid = grid;
        this.seen = new boolean[grid.length][grid[0].length];
        
        for(int i = 0; i < grid.length; i++) {
            for(int k = 0; k < grid[0].length; k++) {
                if (grid[i][k] == 1) start = new int[] {i, k};
                if (grid[i][k] == 2) end = new int[] {i, k};
                if (grid[i][k] == 0) zeroes++;
            }
        }
        
        dfs(start, end, start[0], start[1], zeroes + 1);
        return ans;
    }
    
    private void dfs(int[] start, int[] end, int r, int c, int zeroes) {
        if (r < 0 || r >= grid.length || c < 0 || c >= grid[0].length) {
            return;
        }
        
        System.out.println("WTF "  + r + "  " + c +  "   " + zeroes);
        if (r == end[0] && c == end[1]) {
            if (zeroes == 0) ans++;
            // System.out.println("WTF "  + x + "  " + y +  "   " + zeroes);
            return;
        }
        
        
    
        
        if (seen[r][c] || grid[r][c] == -1) return;
        
        seen[r][c] = true;
        
        dfs(start, end, r - 1, c, zeroes - 1);
        dfs(start, end, r + 1, c, zeroes - 1);
        dfs(start, end, r, c - 1, zeroes - 1);
        dfs(start, end, r, c + 1, zeroes - 1);
        seen[r][c] = false;
    }
}
