class Solution {
    int rows;
    int cols;
    int res;
    int startCoor;
    int endCoor;
    public int uniquePathsIII(int[][] grid) {
        rows = grid.length;
        cols = grid[0].length;
        res = 0;
        Set<Integer> set = new HashSet<>();
        Set<Integer> visited = new HashSet<>();
        for(int i = 0; i < rows; i++) {
            for(int j = 0; j < cols; j++) {
                if(grid[i][j] == 1) {
                    startCoor = calcCoor(i, j);
                } else if(grid[i][j] == 2) {
                    endCoor = calcCoor(i, j);
                    set.add(endCoor);
                } else if(grid[i][j] == 0) {
                    set.add(calcCoor(i, j));
                }
            }
        }
        // System.out.println(startCoor + " " + endCoor);
        dfs(startCoor, endCoor, set);
        return res;
    }
    private void dfs(int start, int end, Set<Integer> set) {
        // System.out.println(start / cols + " " + start % cols + " " + set.size());
        if(start == end && set.size() == 0) {
            res++;
            return;
        }
        int[][]dirs = new int[][]{{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
        int row = start / cols;
        int col = start % cols;
        for(int[] dir: dirs) {
            int newR = row + dir[0];
            int newC = col + dir[1];
            if(newR >= 0 && newR < rows && newC >= 0 && newC < cols) {
                int newCoor = calcCoor(newR, newC);
                if(set.contains(newCoor)) {
                    set.remove(newCoor);
                    dfs(newCoor, end, set);
                    set.add(newCoor);
                }
            }
        }
    }
    private int calcCoor(int row, int col) {
        return row * cols + col;
    }
}
