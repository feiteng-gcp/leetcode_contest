class Solution {
    int N0;
    
    int M;
    int N;
    
    public int uniquePathsIII(int[][] grid) {
        M = grid.length;
        if(M == 0) return 0;
        
N = grid[0].length;
        int I = 0;
        int J = 0;
        for(int i = 0;i < grid.length; i++) {
            for(int j = 0;j < grid[0].length;j++) {
                if(grid[i][j] == 0) N0++;
                if(grid[i][j] == 1) {
                    I = i;
                    J = j;
                
}
            }
        }
        dfs(grid, I, J, 0);
        return all;
    }
    int all = 0;
    void dfs(int[][] grid, int i, 
int j, int count) {
        if(i < 0 || j < 0 || i >= M || j >= N ||grid[i][j] < 0) return;
        if(grid[i][j] == 2) {
            // System.out.println(i + " " + j);
            if(count == N0 + 1) all++;
            return;
        }
        int tmp = grid[i][j];
        grid[i][j] = -2;
        dfs(grid, i, j + 1, count+1);
        dfs(grid, i, j - 1, count+1);
        dfs(grid, i + 1, j, count+1);
        dfs(grid, i - 1, j, count+1);
        grid[i][j] = tmp;        
    }
}
