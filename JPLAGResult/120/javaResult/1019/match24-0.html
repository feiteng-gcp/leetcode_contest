class Solution {
    public int[] sortedSquares(int[] A) {
        // first calc squares, then sort, this is nlogn
        // find 0, square, then merge sort (the negative one need to be read in reverse order), this is longn + n = n
        for (int i = 0; i < A.length; i++) {
            A[i] = A[i] * A[i];
        }
        Arrays.sort(A);
        return A;
    }
}
