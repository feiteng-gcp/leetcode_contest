class Solution {
    public int[] sortedSquares(int[] A) {
        int n = A.length;
        int[] ans = new int[n];
        int p1 = 0, p2 = n - 1;
        int index = n-1;
        while (p1 <= p2) {
            int s1 = A[p1] * A[p1];
            int s2 = A[p2] * A[p2];
            
if (s1 < s2) {
                ans[index--] = s2;
                p2--;
            } else {
                ans[index--] = s1;
                p1++;
            }
        }
        return ans;
    }
}
