class Solution {
    public int maxTurbulenceSize(int[] A) {
        if(A.length < 2) return A.length;
        int count = 1, max = 1;
        int prev = A[1] - A[0];
        for(int i = 1; i < A.length; i++) {
            if(A[i] > A[i - 1]) {
                if(prev > 0) {
                    count = 1;
                } else if(prev < 0) {
                    count++;
                } else {
                    count = 1;
                }
            } else if(A[i] < A[i - 1]) {
                if(prev > 0) {
                    count++;
                } else if(prev < 0) {
                    count = 1;
                } else {
                    count = 1;
                }                
            } else {
                count = 0;
            }
            prev = A[i] - A[i - 1];
            max = Math.max(max, count);
        }
        return max + 1;
    }
}
