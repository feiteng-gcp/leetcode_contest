	class Solution {
	    public int maxTurbulenceSize(int[] A) {
	        int n = A.length;
	        int dp = 0;
	        int ret = 1;
	        for(int i = 0;i < n-1;i++){
	        	if(A[i] < A[i+1]){
	        		if(i > 0 && A[i-1] > A[i]){
	        			dp++;
	        		}else{
	        			dp = 2;
	        		}
	        	}else if(A[i] > A[i+1]){
	        		if(i > 0 && A[i-1] < A[i]){
	        			dp++;
	        		}else{
	        			dp = 2;
	        		}
	        	}else{
	        		dp = 1;
	        	}
	        	ret = Math.max(ret, dp);
	        }
	        return ret;
	    }
	}
