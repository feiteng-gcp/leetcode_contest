class Solution {
    public int maxTurbulenceSize(int[] A) {
        int max = 0;
        int count1 = 0, count2 = 0;
        for(int i = 0;i<A.length-1;i++) {
            if(i % 2 == 0) {
                if(A[i] > A[i+1]) {
                    count1++;
                    count2 = 0;
                    max = Math.max(max,count1);
                }else if(A[i] < A[i+1]) {
                    count1 = 0;
                    count2 ++;
                    max = Math.max(max,count2);
                }else {
                    count1 = 0;
                    count2 = 0;
                }
            }else {
                if(A[i] < A[i+1]) {
                    count1++;
                    count2 = 0;
                    max = Math.max(max,count1);
                }else if(A[i] > A[i+1]) {
                    count1=0;
                    count2++;
                    max = Math.max(max,count2);
                }else {
                    count1=0;
                    count2=0;
                }
            }
        }
        return max+1;
    }
}
