class Solution {
    public int maxTurbulenceSize(int[] A) {
        if(A == null || A.length == 0) return 0;
        if(A.length == 1) return 1;
        
        
int maxLen = 1;
        int len = 1;
        int inc = 0;
        // inc: 0 means equal, 1 means inc, 2 means dec
        for(int i = 1; i < A.length; i++) {
            if(A[i] == A[i-1]) {
                maxLen = Math.max(maxLen, len);
                len = 1;
                inc = 0;
            
} else if(A[i] > A[i-1]) {
                if(inc == 1) {
                    maxLen = Math.max(maxLen, len);
                    len = 2;
                } else if(inc == 0) {
                    len = 2;
                    inc = 1;
                } else {
                    inc = 1;
                    len++;
                }
            } else {
                if(inc == 0) {
                    len = 2;
                    inc = 2;
                } else 
if(inc == 2) {
                    maxLen = Math.max(maxLen, len);
                    len = 2;
                } else {
                    inc = 2;
                    len++;
                }
            }
        }
        return Math.max(len, maxLen);
    }
}
