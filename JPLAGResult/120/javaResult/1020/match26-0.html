class Solution {
    public int maxTurbulenceSize(int[] A) {
        int n = A.length;
        int isSmall = 0;
        int result = 1;
        int curResult = 1;
        for(int i = 0; i < n - 1; i++){
            if(A[i] == A[i + 1]){
                result = Math.max(result, curResult);
                isSmall = 0;
                curResult = 1;
                continue;
            }
            
            if(isSmall == 0){
                if(A[i] < A[i + 1]){
                    isSmall = 1;
                }
                else{
                    isSmall = -1;
                }
                
                curResult++;
            }
            else if(isSmall == 1){
                if(A[i] < A[i + 1]){
                    result = Math.max(result, curResult);
                    curResult = 2;
                }
                else{
                    isSmall = -1;
                    curResult++;
                }
            }
            else{
                if(A[i] > A[i + 1]){
                    result = Math.max(result, curResult);
                    curResult = 2;
                }
                else{
                    isSmall = 1;
                    curResult++;
                }
            }
        }
        
        result = Math.max(result, curResult);
        return result;
    }
}
