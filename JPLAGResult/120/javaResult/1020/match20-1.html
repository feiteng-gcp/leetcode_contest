class Solution {
    public int maxTurbulenceSize(int[] A) {
        int n = A.length;
        if (n <= 1) {
            return n;
        }
        int max = 1;
        int prevUp = 1, prevDown = 1;
        for (int i = 1; i < n; i++) {
            int curUp = 1, curDown = 1;
            
if (A[i] > A[i - 1]) {
                curUp = prevDown + 1;
            } else if (A[i] < A[i - 1]) {
                curDown = prevUp + 1;
            }
            prevUp = curUp;
            prevDown = curDown;
            max = Math.max(Math.max(prevUp, prevDown), max);
        }
        return max;
    }
}
