class Solution {
    public int maxTurbulenceSize(int[] A) {
        if(A.length <= 2) {
            return A.length;
        }
        int[] up = new int[A.length];
        int[] down = new int[A.length];
        up[0] = 1;
        down[0] = 1;
        int res = 1;
        for(int i = 1; i < A.length; i++) {
            if(A[i] > A[i - 1]) {
                up[i] = down[i - 1] + 1;
                down[i] = 1;
            } else if(A[i] < A[i - 1]) {
                down[i] = up[i - 1] + 1;
                up[i] = 1;
            } else {
                down[i] = 1;
                up[i] = 1;
            }
            
res = Math.max(res, up[i]);
            res = Math.max(res, down[i]);
        }
        return res;
    }
}
