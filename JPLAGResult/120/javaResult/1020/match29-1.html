class Solution {
    public int maxTurbulenceSize(int[] A) {
        // dp[i][0] -- ending as bigger number than next
        // dp[i][1] -- ending as smaller number than next
        int[][] dp = new int[A.length][2]; 
        dp[0][0] = 1;
        dp[0][1] = 1;
        int max = 1;
        for(int i = 1; i < A.length; i++) {
            int n = A[i];
            
if(n > A[i-1]) {
                dp[i][0] = dp[i-1][1] + 1; 
            } else {
                dp[i][0] = 1;
            }
            if(n < A[i-1]) {
                dp[i][1] = dp[i-1][0] + 1;
            } else {
                dp[i][1] = 1;
            
}
            max = Math.max(max, Math.max(dp[i][1], dp[i][0]));         
        }
        return max;
    }
}
