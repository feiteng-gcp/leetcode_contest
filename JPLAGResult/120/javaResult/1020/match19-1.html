class Solution {
    public int maxTurbulenceSize(int[] A) {
        int n = A.length;
        if (n <= 2) {
            return n;
        }
        
int max = 2;
        int[] first = new int[n], second = new int[n];
        Arrays.fill(first, 1);
        Arrays.fill(second, 1);
        for (int k = 1; k < n; k++) {
            if (k % 2 == 1 && A[k] > A[k - 1]) {
                first[k] = 1 + first[k - 1];
            } else if (k % 2 == 0 && A[k] < A[k - 1]) {
                first[k] = 1 + first[k - 1];
            }
            max = Math.max(max, first[k]);
        }
        
for (int k = 1; k < n; k++) {
            if (k % 2 == 1 && A[k] < A[k - 1]) {
                second[k] = 1 + second[k - 1];
            } else if (k % 2 == 0 && A[k] > A[k - 1]) {
                second[k] = 1 + second[k - 1];
            }
            max = Math.max(max, second[k]);
        }
        return max;
    }
}
