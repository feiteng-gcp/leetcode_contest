class Solution {
    public int maxTurbulenceSize(int[] A) {
        int n = A.length;
        int[] dpup = new int[n];
        int[] dpdown = new int[n];
        Arrays.fill(dpup, 1);
        Arrays.fill(dpdown, 1);
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (i > 0 && A[i] > A[i - 1]) dpup[i] = dpdown[i-1] + 1;
            if (i > 0 && A[i] < A[i - 1]) dpdown[i] = dpup[i-1] + 1;
            ans = Math.max(ans, dpup[i]);
            ans = Math.max(ans, dpdown[i]);
        }
        return ans;
    }
}
