class Solution {
    public int maxTurbulenceSize(int[] A) {
        int res = 1;
        int cur1 = 1;
        int cur2 = 1;
        for (int i = 1; i < A.length; i++) {
            if (i % 2 == 0) {
                
if (A[i] >= A[i-1]) {
                    cur1 = 0;
                }
                if (A[i] <= A[i-1]) {
                    cur2 = 0;
                }
            } else {
                if (A[i] <= A[i-1]) {
                    cur1 = 0;
                }
                if (A[i] >= A[i-1]) {
                    cur2 = 0;
                }
            }
            cur1++;
            cur2++;
            
res = Math.max(res, cur1);
            res = Math.max(res, cur2);
        }
        return res;
    }
}
