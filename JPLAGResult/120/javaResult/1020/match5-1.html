class Solution {
    public int maxTurbulenceSize(int[] A) {
        
int res = 1;
        int curr = 1;
        
        for(int i = 0; i < A.length - 1; i++){
            if((i % 2) == 0){
                if(A[i] < A[i+1]){
                    curr++;
                } else {
                    curr = 1;
                }
            } else {
                if(A[i] > A[i+1]){
                    curr++;
                } else {
                    curr = 1;
                }
            }
            
            res = Math.max(res, curr);
        }
        
        curr = 1;
        
        for(int i = 0; i < A.length - 1; i++){
            
if((i % 2) == 1){
                if(A[i] < A[i+1]){
                    curr++;
                } else {
                    curr = 1;
                }
            } else {
                if(A[i] > A[i+1]){
                    curr++;
                } else {
                    curr = 1;
                }
            
}
            
            res = Math.max(res, curr);
        }
        return res;
    }
}
