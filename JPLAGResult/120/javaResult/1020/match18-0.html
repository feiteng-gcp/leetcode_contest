class Solution {
    public int maxTurbulenceSize(int[] A) {
        int[] min = new int[A.length];
        int[] max = new int[A.length];
        int res = 1;
        max[0] = 1;
        min[0] = 1;
        
        for(int i = 1; i < A.length; i++) {
            if(A[i] == A[i-1]) {
                max[i] = 1;
                min[i] = 1;
            } else if(A[i] > A[i-1]) {
                max[i] = 1 + min[i-1];
                min[i] = 1;
                res = Math.max(res, max[i]);
            } else {
                min[i] = max[i-1] + 1;
                max[i] = 1;
                res = Math.max(res, min[i]);
            }
        }
        return res;
    }
}
