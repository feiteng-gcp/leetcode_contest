class Solution {
    public int maxTurbulenceSize(int[] A) {
        if (A.length == 1) return 1;
        int ans = 1;
        int[] dp = new int[A.length];
        dp[0] = 1;
        dp[1] = 2;
        for (int i = 2; i < A.length; i++) {
            if (A[i - 2] < A[i - 1] && A[i - 1] > A[i] || A[i - 2] > A[i - 1] && A[i - 1] < A[i])
                dp[i] = dp[i - 1] + 1;
            else
                dp[i] = 2;
            ans = Math.max(ans, dp[i]);
        }
        return ans;
    }
}
