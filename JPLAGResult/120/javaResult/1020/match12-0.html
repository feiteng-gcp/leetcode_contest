class Solution {
    public int maxTurbulenceSize(int[] A) {
        int N = A.length;
        if(N == 0) return 0;
        int[][] dp = new int[N][2];
        dp[0][1] = 1;
        dp[0][0] = 1;
        int max = 1;
        // 0 >
        // 1 < 
        for(int i = 1;i < A.length;i++) {
            if(A[i] > A[i - 1]) {
                dp[i][0] = dp[i-1][1] + 1;
                dp[i][1] = 1;
            } else if(A[i] < A[ i - 1]) {
                dp[i][1] = dp[i-1][0] + 1;
                dp[i][0] = 1;
            } else {
                dp[i][0] = 1;
                dp[i][1] = 1;


            }
            // System.out.println(Arrays.toString(dp[i]));
            max = Math.max(max, Math.max(dp[i][0], dp[i][1]));
        }
        return max;
    }
}
