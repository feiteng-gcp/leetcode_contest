class Solution {
    public int maxTurbulenceSize(int[] A) {
        if (A == null || A.length == 0) {
            return 0;
        }
        int res = 1;
        for (int i = 0, t1 = 1, t2 = 1; i < A.length - 1; i++) {
            
if (A[i] > A[i + 1]) {
                if (i % 2 == 0) {
                    t1 = 1;
                    t2++;
                } else {
                    t1++;
                    t2 = 1;
                }
            } else if (A[i] < A[i + 1]) {
                if (i % 2 == 0) {
                    t1++;
                    t2 = 1;
                
} else {
                    t1 = 1;
                    t2++;
                }
            } else {
                t1 = 1;
                t2 = 1;
            }
            // System.out.println(i + ", t1 = " + t1 + ", t2 = " + t2);
            res = Math.max(res, Math.max(t1, t2));
        }
        return res;
    }
}
