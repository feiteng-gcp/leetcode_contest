class Solution {
    public int maxTurbulenceSize(int[] A) {
        if (A == null || A.length == 0) {
            return 0;
        }
        if (A.length == 1) {
            return 1;
        }
        // for A[i], it can be < A[i] or > A[i]
        // > A[i-1] < A[i]
        // < A[i-1] > A[i]
        int max = 1;
        int endWithL = 1;
        int endWithR = 1;
        for (int i = 1; i < A.length; i++) {
            if (A[i] == A[i - 1]) {
                endWithR = 1;
                endWithL = 1;
            } else if (A[i] > A[i-1]) {
                endWithR = endWithL + 1;
                endWithL = 1; // stopped
                max = Math.max(max, endWithR);
            } else {
                endWithL = endWithR + 1;
                endWithR = 1; // stopped
                max = Math.max(max, endWithL);
            }
        }
        return max;
    }
}
