class Solution {
    public int maxTurbulenceSize(int[] A) {
        int sign = 0;
        int count = 0;
        int result = 0;//2  1  3  3  4  2
        if(A.length == 1) {
return 1;}
        
for(int i = 0; i < A.length - 1; i++) {
            if(A[i+1] - A[i] > 0) {
                if(sign != 1 || i == 0) {
                    count++;
                } else {
                    count = 1;
                }
                sign = 1;
            
} else if(A[i+1] - A[i] < 0) {
                if(sign != -1 || i == 0) {
                    count++;//i is the result;
                } else {
                    count = 1;
                }
                sign = -1;
            } else{
                sign = 0;
                
count = 0;
            }
            result = Math.max(result,count + 1);
        }
        return result;
    }
}
