class Solution {
    public int maxTurbulenceSize(int[] A) {
        int max = 0;
        if (A.length <= 1) return A.length;
        
        int dir = 0;
        int cur = 2;
        if (A[0] < A[1]) {
            dir = 1; // Increasing
        } else 
if (A[0] > A[1]) {
            dir = 2; // Decreasing
        } else {
            cur = 1;
        }
        
        for (int i = 1; i < A.length - 1; i++) {
            if (A[i] == A[i+1]) {
                dir = 0;
                max = Math.max(max, cur);
                cur = 1;
            } else if (A[i] < A[i+1]) {
                if (dir == 2) {
                    cur++;
                } else 
if (dir == 1) {
                    max = Math.max(max, cur);
                    cur = 2;
                } else {
                    cur = 2;
                }
                dir = 1;
            } else {
                if (dir == 1) {
                    cur++;
                } else 
if (dir == 2) {
                    max = Math.max(max, cur);
                    cur = 2;
                } else {
                    cur = 2;
                }
                dir = 2;
            }
        }
        
        return Math.max(max, cur);
    }
}
