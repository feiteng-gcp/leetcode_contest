class Solution {
    public int maxTurbulenceSize(int[] A) {
        int start1=0;
        int start2=0;
        int maxLen=1;
        for(int i=1;i<A.length;i++){
            if(i%2==0){
                if(A[i]>A[i-1]){
                    maxLen=Math.max(i-start1+1,maxLen);
                    start2=i;
                }
                else if((A[i]<A[i-1])){
                    maxLen=Math.max(i-start2+1,maxLen);
                    start1=i;
                
}
                else{
                    start1=i;
                    start2=i;
                }
            }
            else{
                if(A[i]<A[i-1]){
                    maxLen=Math.max(i-start1+1,maxLen);
                    start2=i;
                }
                else if((A[i]>A[i-1])){
                    maxLen=Math.max(i-start2+1,maxLen);
                    start1=i;
                
}
                else{
                    start1=i;
                    start2=i;
                }
            }
        }
        return maxLen;
    }
}
