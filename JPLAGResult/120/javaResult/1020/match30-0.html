class Solution {
    public int maxTurbulenceSize(int[] A) {
        int n = A.length;
        int[] smallLeft = new int[n];
        int[] largeLeft = new int[n];
        Arrays.fill(smallLeft, 1);
        Arrays.fill(largeLeft, 1);
        for (int i = 1; i < n; i++) {
            if (A[i] < A[i - 1]) {
                smallLeft[i] = 1 + largeLeft[i - 1];
            }
            else if (A[i] > A[i - 1]) {
                largeLeft[i] = 1 + smallLeft[i - 1];
            }
        }

        int[] smallRight = new int[n];
        int[] largetRight = new int[n];
        Arrays.fill(smallRight, 1);
        Arrays.fill(largetRight, 1);
        for (int i = n - 2; i >= 0; i--) {
            if (A[i] < A[i + 1]) {
                smallRight[i] = 1 + largetRight[i + 1];
            }
            else if (A[i] > A[i + 1]) {
                largetRight[i] = 1 + smallRight[i + 1];
            }
        }

        int res = 0;
        for (int i = 0; i < n; i++) {
            res = Math.max(res, smallLeft[i] + smallRight[i] - 1);
            res = Math.max(res, largeLeft[i] + largetRight[i] - 1);
        }

        for (int i = 0; i < n; i++) {
            res = Math.max(res, smallLeft[i]);
            res = Math.max(res, largeLeft[i]);
            res = Math.max(res, smallRight[i]);
            res = Math.max(res, largetRight[i]);
        }

        return res;
    }
}
