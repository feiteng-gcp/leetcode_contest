class Solution {
    public int maxTurbulenceSize(int[] A) {
        int[] dph = new int[A.length];
        int[] dpl = new int[A.length];
        
        dph[0] =1;
        dpl[0] =1;
        
        int ma = 1;
        for(int i=1;i<A.length;i++)
{
            if(A[i]>A[i-1]){
                dph[i] = Math.max(1,dpl[i-1]+1);
                dpl[i] = 1;
                
            }else if(A[i]<A[i-1]){
                dpl[i] = Math.max(1,dph[i-1]+1);
                dph[i] = 1;
            }else{
                 dpl[i] = 1;
                 dph[i] = 1;
            }
            
            ma = Math.max(ma,dph[i]);
            ma = Math.max(ma,dpl[i]);
        }
        
        return ma;
    }
}
