class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        if A[0] >= 0:
            return [x**2 for x in A]
        if A[-1] <= 0:
            return [x**2 for x in A[::-1]]
        k = 0
        while A[k] < 0:
            k += 1
        
        A1 = [x**2 for x in A[:k]]
        A1.reverse()
        A2 = [x**2 for x in A[k:]]
        ans 
= []
        i, j = 0, 0
        while i < len(A1) and j < len(A2):
            if A1[i] > A2[j]:
                ans.append(A2[j])
                j += 1
            else:
                ans.append(A1[i])
                i += 1
        if i == len(A1):
            ans += A2[j:]
        if j == len(A2):
            ans += A1[i:]
        return ans
