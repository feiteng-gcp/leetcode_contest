class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        pos = []
        neg = []
        for num in A:
            if num >= 0:
                pos.append(num * num)
            else:
                neg.append(num * num)
        res = []
        posInd = 0
        negInd = len(neg) - 1
        while posInd < len(pos) or negInd >= 0:
            if posInd >= len(pos):
                res.append(neg[negInd])
                negInd -= 1
            elif negInd < 0:
                res.append(pos[posInd])
                posInd += 1
            else:
                
if pos[posInd] >= neg[negInd]:
                    res.append(neg[negInd])
                    negInd -= 1
                else:
                    res.append(pos[posInd])
                    posInd += 1
        return res
