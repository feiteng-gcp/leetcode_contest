class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """

        n = len(A)
        l, r = -1, 0
        for i in range(n-1):
            if A[i]<=0 and A[i+1]>=0:
                l = i
                r = i + 1
                break
        res = []
        while l>=0 or r<n:
            if l<0 or r<n and abs(A
[l])>=abs(A[r]):
                res.append(A[r]**2)
                r += 1
            else:
                res.append(A[l]**2)
                l -= 1
        return res
