class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        pos, neg = [], []
        for a in A:
            arr = pos if a >= 0 else neg
            arr.append(a ** 2)
        neg = list(reversed(neg))
        # print(pos)
        # print(neg)
        ans = []
        p, n = 0, 0
        
while p < len(pos) and n < len(neg):
            if pos[p] <= neg[n]:
                ans.append(pos[p])
                p += 1
            else:
                ans.append(neg[n])
                n += 1
        ans += pos[p:] + neg[n:]
        return ans
