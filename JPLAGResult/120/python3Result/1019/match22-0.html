class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        res = []
        
        left = 0
        right = 0
        
        while right < len(A) and A[right] < 0:
            right += 1
        
        left = right - 1
        
        while left >= 0 or right < len(A):
            if left >= 0 and (right >= len(A) or abs(A[left]) < abs(A[right])):
                res.append(A[left] * A[left])
                left -= 1
            else:
                res.append(A[right] * A[right])
                right += 1
        
        return res
