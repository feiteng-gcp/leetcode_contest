class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        if A[-1]<0:
            return [x*x for x in A[::-1]]
        for i in range(len(A)):
            if A[i]>=0:
                pid=i
                break
        nid=i-1
        ans=[]
        while pid<len(A) or nid>=0:
            if pid==len(A):
                ans.append(A[nid]*A[nid])
                nid-=1
            elif nid<0:
                ans.append(A[pid]*A[pid])
                pid+=1
            elif A[pid]+A
[nid]>0:
                ans.append(A[nid]*A[nid])
                nid-=1
            else:
                ans.append(A[pid]*A[pid])
                pid+=1
        return ans
