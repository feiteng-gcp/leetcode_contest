class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        negative = list(filter(lambda x: x<0, A))
        positive = list(filter(lambda x: x>=0, A))
        negative = negative[::-1]
        combined = []
        i, j = 0, 0
        while i < len(positive) and j < len(negative):
            if positive[i] < -negative[j]:
                combined.append(positive[i])
                i += 1
            else:
                combined.append(negative[j])
                j += 1
        combined.extend(positive[i:])
        combined.extend(negative[j:])
        return [v**2 for v in combined]
