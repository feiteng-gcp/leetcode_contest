import bisect


class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        pivot = bisect.bisect_left(A, 0)
        left = [-x for x in A[:pivot][::-1]]
        right = A[pivot:]
        i, j = 0, 0
        res = list()
        while i < len(left) and j < len(right):
            if left[i] < right[j]:
                res.append(left[i]**2)
                i += 1
            else:
                res.append(right[j]**2)
                j += 1
        res.extend([x**2 for x in left[i:]])
        res.extend([x**2 for x in right[j:]])
        return res
