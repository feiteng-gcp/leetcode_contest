class Solution:
    def sortedSquares(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        i = self.binarySearch(A, 0)
        j = i-1
        res = []
        while 0<=j<len(A) and 0<=i<len(A):
            if A[i] < abs(A[j]):
                res.append(A[i]**2)
                i += 1
            else:
                res.append(A[j]**2)
                j -= 1
        if i < len(A):
            for k in range(i, len(A)):
                res.append(A[k]**2)
        if j >= 0:
            for k in range(j, -1, -1):
                res.append(A[k]**2)
        return res
        
    def binarySearch(self, lst, target):
        start = 0
        end = len(lst)-1
        while start < end:
            m = (start+end) // 2
            if lst[m] < target:
                start = m+1
            else:
                end = m
        return start
