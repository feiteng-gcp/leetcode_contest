class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if len(A)<=2:
            return len(A)
        res,temp = 0,2
        for i,j,k in zip(A,A[1:],A[2:]):
            if (i-j)*(j-k)<0:
                temp += 1
            else:
                res = max(res,temp)
                temp = 2
        return max(res,temp)
