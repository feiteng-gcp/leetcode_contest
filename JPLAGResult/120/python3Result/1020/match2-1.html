class Solution:
    def maxTurbulenceSize(self, A):
        
        n=len(A)
        if n<2:
            return n
        ans=1
        l=1
        flag=0
        for i in range(n-1):
            if flag==0:
                if A[i]<A[i+1]:
                    l+=1
                    flag=1
                elif A[i]>A[i+1]:
                    l
+=1
                    flag=-1
            elif flag==1:
                if A[i]>A[i+1]:
                    l+=1
                    flag=-1
                elif A[i]<A[i+1]:
                    l=2
                    flag=1
                else:
                    l=1
                    flag=0
            else:
                if A[i]<A[i+1]:
                    l+=1
                    flag=1
                elif A[i]>A[i+1]:
                    l=2
                    flag=-1
                else:
                    l=1
                    flag=0
            ans=max(ans,l)
        return ans

        :type A: List[int]
        :rtype: int
