class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if len(A) <= 2: return len(A)
        maxi = 0
        cur = 1
        prev = 0
        for cnt in range(1,len(A)):
            if A[cnt] > A[cnt-1]:
                if prev == 0 or prev == -1:
                    cur +=1
                    prev = 1
                else:
                    maxi = max(maxi,cur)
                    cur = 2
                    prev = 1
            elif A[cnt] < A[cnt-1]:
                if prev == 0 or prev == 1:
                    cur +=1
                    prev = -1
                else:
                    maxi = max(maxi,cur)
                    cur = 2
                    prev = -1
            else:
                maxi = max(maxi,cur)
                cur =1
                prev = 0
                    
        return max(maxi,cur)
