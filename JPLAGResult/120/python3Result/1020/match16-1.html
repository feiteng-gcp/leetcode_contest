class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if len(A) < 3:
            return len(A)
        
        length = 0
        max_length = 0
        up = A[1] > A[0]
        
for i in range(len(A) - 1):
            up = not up
            # print (up, A[i] < A[i+1], length)
            if ((up and A[i] < A[i+1]) or (not up and A[i] > A[i+1])):
                length += 1
            else:
                # start a new check
                up = A[i+1] > A[i]
                length = 2
            max_length = max(length, max_length)
            
        return max(length, max_length)
