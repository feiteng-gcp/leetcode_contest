class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        res 
= 1
        token = -1
        curr = 1
        for i in range(len(A)-1):
            if token == 1:
                if A[i] < A[i+1]:
                    curr = curr + 1
                    token = 0
                elif A[i] > A[i+1]:
                    token = 1
                    curr = 2
                else:
                    token = -1
                    curr = 1
                    
            elif token == 0:
                if A[i] > A[i+1]:
                    curr = curr + 1
                    token = 1
                elif A[i] < A[i+1]:
                    token = 0
                    curr = 2
                else:
                    token = -1
                    curr = 1
            else:
                if A[i] < A[i+1]:
                    token = 0
                    curr = 2
                elif A[i] > A[i+1]:
                    token = 1
                    curr = 2
                else:
                    curr = 1
            if curr > res:
                res = curr
        return res
