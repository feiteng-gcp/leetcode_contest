class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if not A:
            return 0
        res = 1
        odd = True
        oddScore = 1
        evenScore = 1
        
        for i in range(len(A)-1):
            odd = not odd
            if (odd and A[i] < A[i+1]) or (not odd and A[i] > A[i+1]):
                oddScore += 1
                evenScore = 1
            elif (odd and A[i] > A[i+1]) or (not odd and A[i] < A[i+1]):
                oddScore = 1
                evenScore += 1
            else:
                oddScore = 1
                evenScore = 1
                    
            res = max(res, oddScore, evenScore)
            
        return res
