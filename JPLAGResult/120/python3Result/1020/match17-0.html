class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        l = len(A)
        if l <= 1:
            return l
        if l == 2:
            return 1 if l[0] == l[1] else 2
        # l >= 3
        
        max_ = 1
        cur_max = 1
        pre = 0
        for i in range(1, l):
            cur = A[i] - A[i-1]
            
            # print(cur, pre)
            if pre != 0:
                
                if cur * pre > 0:
                    cur_max = 2
                elif cur == 0:
                    cur_max = 1
                else:
                    cur_max += 1
                    max_ = max(cur_max, max_)
            else:
                if cur == 0:
                    cur_max = 1
                else:
                    cur_max = 2
            max_ = max(cur_max, max_)
            pre = cur
            
        return max_
