class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        n = len(A)
        if n==0:return 0
        mx = 1
        ct = 1
        flag = None
        for i in range(1,n):
            if A[i]==A[i-1]:
                mx = max(mx,ct)
                ct=1
                falg = None
                continue
            b = A[i]>A[i-1]
            if flag is None or (b and not flag) or (not b and flag):
                ct+=1
            else:
                mx = max(mx,ct)
                ct=2
            flag = b
        return max(mx,ct)
