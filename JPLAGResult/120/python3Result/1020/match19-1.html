class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        l = 0
        max_ = 0
        s = 0
        i = 0
        while i<len
(A)-1:
            if s==0:
                if A[i]<A[i+1]:
                    l+=1
                    s = 1
                elif A[i]>A[i+1]:
                    l+=1
                    s = 2
            elif s==2:
                if A[i]<A[i+1]:
                    s = 1
                    l+=1
                else:
                    s = 0
                    l = 0
                    i-=1
            elif s==1:
                if A[i]>A[i+1]:
                    s = 2
                    l+=1
                else:
                    s = 0
                    l = 0
                    i-=1
            i+=1
            max_ = max(l, max_)
        return max_+1
