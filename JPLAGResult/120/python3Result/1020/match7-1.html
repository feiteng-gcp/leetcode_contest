class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if not A:
            return 0
        tbl_u = [1] * len(A)
        tbl_u[0] = 1
        tbl_d = [1] * len(A)
        tbl_d[0] = 1
        ans = 1
        
for i in range(1, len(A)):
            if A[i-1] < A[i]:
                tbl_u[i] = tbl_d[i-1]+1
            if A[i-1] > A[i]:
                tbl_d[i] = tbl_u[i-1]+1
            ans = max(ans, tbl_u[i])
            ans = max(ans, tbl_d[i])
        return ans
