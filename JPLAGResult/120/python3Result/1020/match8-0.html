class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        # 双状态动态规划
        # O(n)
        n = len(A)
        inc, dec = [1]*n, [1]*n
        for i in range(1, n):
            if A[i] > A[i-1]:
                inc[i] = dec[i-1]+1
            elif A[i] < A[i-1]:
                dec[i] = inc[i-1]+1
        return max(max(inc), max(dec))
