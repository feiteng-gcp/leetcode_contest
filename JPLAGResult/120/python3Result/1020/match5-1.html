class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if len(A) <= 1:
            return len(A)
        dp1 = [1] * len(A) # if A[i] > A[i-1]
        dp2 = [1] * len(A) # if A[i] < A[i-1]
        #dp1[0], dp2[0] = 1, 1
        for i in range(1, len
(A)):
            if A[i] > A[i-1]:
                dp1[i] = dp2[i-1] + 1
            if A[i] < A[i-1]:
                dp2[i] = dp1[i-1] + 1
        return max(max(dp1), max(dp2))
