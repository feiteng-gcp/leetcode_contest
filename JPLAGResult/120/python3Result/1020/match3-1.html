class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        ans 
= 1
        tmp = 1
        flg = 0
        for i in range(1, len(A)):
            if flg == 0:
                if A[i] > A[i-1]:
                    flg = 2
                    tmp = 2
                elif A[i] < A[i-1]:
                    flg = 1
                    tmp = 2
                if tmp > ans:
                    ans = tmp
                continue
            if flg == 1 and A[i] > A[i-1]:
                tmp += 1
                flg = 2
                if tmp > ans:
                    ans = tmp
            elif flg == 2 and A[i] < A[i-1]:
                tmp += 1
                flg = 1
                if tmp > ans:
                    ans = tmp
            elif flg == 1 and A[i] < A[i-1] or flg == 2 and A[i] > A[i-1]:
                tmp = 2
            else:
                tmp = 1
                flg = 0
        return ans
