class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        n=len(A)
        if n<=1: return 1
        if n==2: return 1 if A[0]==A[1] else 2
        count=ans=1
        i=j=0
        flag=True
        while j<n-1 and i<n-1:
            i=j
            while i<n-1:
                if (flag and A[i+1]<A[i]) or (not flag and A[i+1]>A[i]):
                    flag=not flag
                    count+=1
                    #print(count,A[i+1])
                    ans=max(ans,count)
                    i+=1
                elif A[i+1]==A[i]:
                    j=i+1
                    count=1
                    break
                else:
                    flag=not flag
                    count=1
                    j=i
                    break
        return ans
