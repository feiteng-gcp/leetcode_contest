class Solution:
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        
        if len(A) <= 1:
            return len(A)
        
        flag = 0
        max_lgth = 1
        cur_lgth = 1
        for i in range(1, len(A)):
            if flag == 0:
                if A[i] > A[i-1]:
                    flag = 1
                    cur_lgth = 2
                elif A[i] < A[i-1]:
                    flag = -1
                    cur_lgth = 2
                else:
                    flag = 0
                    cur_lgth = 1
            elif flag == 1:
                if A[i] > A[i-1]:
                    cur_lgth = 2
                    flag = 1
                elif A[i] < A[i-1]:
                    cur_lgth += 1
                    flag = -1
                else:
                    cur_lgth = 1
                    flag = 0
            elif flag == -1:
                if A[i] < A[i-1]:
                    cur_lgth = 2
                    flag = -1
                elif A[i] > A[i-1]:
                    cur_lgth += 1
                    flag = 1
                else:
                    cur_lgth = 1
                    flag = 0
            max_lgth = max(max_lgth, cur_lgth)
        return max_lgth
