class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        def dfs(grid, cur, dst, total, visited):
            #print(cur, total)
            if cur==dst:
                if total==0:
                    nonlocal ret
                    ret += 1
                return
            if total==0:
                return
            m, n = len(grid), len(grid[0])
            dirs = [[-1,0],[1,0],[0,-1],[0,1]]
            for each in dirs:
                i = cur[0]+each[0]
                j = cur[1]+each[1]
                if i>=0 and i<m and j>=0 and j<n and (grid[i][j]==0 or grid[i][j]==2) and not visited[i][j]:
                    visited[i][j] = True
                    dfs(grid, [i,j], dst, total-1, visited)
                    visited[i][j] = False
        
        m = len(grid)
        if m==0:
            return 0
        n = len(grid[0])
        total = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j]==1:
                    src = [i,j]
                elif grid[i][j]==2:
                    dst = [i,j]
                elif grid[i][j]==0:
                    total += 1
        ret = 0
        visited = [[False]*n for _ in range(m)]
        visited[src[0]][src[1]] = True
        dfs(grid, src, dst, total+1, visited)
        return ret
