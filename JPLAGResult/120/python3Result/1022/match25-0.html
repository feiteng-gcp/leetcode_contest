class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        if not grid or not grid[0]:
            return 0
        
        height = len(grid)
        width = len(grid[0])
        
        memory = {}
        
        count_empty = 1
        target = (0, 0)
        start = (0, 0)
        for i in range(height):
            for j in range(width):
                if grid[i][j] == 0:
                    count_empty += 1
                if grid[i][j] == 2:
                    target = (i, j)
                if grid[i][j] == 1:
                    start = (i, j)
                    grid[i][j] = 0
        
        
        def dfs(m, empty, y, x):
            if x < 0 or x > width - 1 or y < 0 or y > height - 1:
                return 0
            
            if empty == 0 and m[y][x] == 2:
                return 1
            
            if m[y][x] != 0:
                return 0
            
            # print (empty, y, x, m)
            
            result = 0
            m[y][x] = 3
            result += dfs(m, empty - 1, y+1, x)
            result += dfs(m, empty - 1, y-1, x)
            result += dfs(m, empty - 1, y, x+1)
            result += dfs(m, empty - 1, y, x-1)
            m[y][x] = 0
            
            return result
            
        return dfs(grid, count_empty, start[0], start[1])
