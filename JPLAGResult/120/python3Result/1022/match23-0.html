class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        if len(grid) == 0 or len(grid[0]) == 0:
            return 0
        M, N = len(grid), len(grid[0])
        visit = [[0] * N for _ in range(M)]
        
        def count(i, j, step):
            if i >= M or j >= N or i < 0 or j < 0 or grid[i][j] == -1 or visit[i][j] == 1:
                return 0
            if grid[i][j] == 2:
                if step == zeros + 1:
                    return 1
                return 0
            
            visit[i][j] = 1
            a = count(i - 1, j, step + 1)            
            b = count(i + 1, j, step + 1)        
            c = count(i, j - 1, step + 1)        
            d = count(i, j + 1, step + 1)
            visit[i][j] = 0
            return a + b + c +d
        
        zeros = starti = startj = 0
        for i in range(M):
            for j in range(N):
                if grid[i][j] == 1:
                    starti, startj = i, j
                elif grid[i][j] == 0:
                    zeros += 1
                
        return count(starti, startj, 0)
