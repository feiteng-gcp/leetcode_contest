class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        def dfs(i,j, squares):
#            print(squares, i, j)
            if end == (i,j) and not squares:
#                print('hi')
                self.count += 1
                return 
            if not squares or end == (i,j):
                return
                
            for r, c in ((i+1,j),(i-1,j),(i,j+1),(i,j-1)):
                if -1<r<n and -1<c<m and (r,c) in squares or (r,c) == end:
                    squares.remove((r,c))
                    dfs(r,c, squares)
                    squares.add((r,c))
            
        n = len(grid)
        m = len(grid[0])
        
        start = None
        end = None
        squares = set()
        
for row in range(n):
            for col in range(m):
                if grid[row][col] == 1:
                    start = (row,col)
                elif grid[row][col] == 2:
                    end = (row,col)
                elif grid[row][col] == 0:
                    squares.add((row,col))
        squares.add(end)
        self.count = 0
        dfs(start[0],start[1], squares)
        return self.count
