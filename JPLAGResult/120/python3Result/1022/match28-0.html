class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        
        m = len(grid)
        n = len(grid[0])
        count = 0
        start = None
        end = None
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    count += 1
                elif grid[i][j] == 1:
                    start = (i, j)
        
        self.res = 0
        
        def dfs(i, j, cur_count):
            tmp = grid[i][j]
            grid[i][j] = -1
            
            next_ij = [(i, j+1), (i, j-1), (i+1, j), (i-1, j)]
            for nij in next_ij:
                ni, nj = nij
                if 0 <= ni < m and 0 <= nj < n:
                    if cur_count == count and grid[ni][nj] == 2:
                        self.res += 1
                    elif grid[ni][nj] == 0:
                        dfs(ni, nj, cur_count + 1)
            grid[i][j] = tmp
            
        dfs(start[0], start[1], 0)
        
        return self.res
