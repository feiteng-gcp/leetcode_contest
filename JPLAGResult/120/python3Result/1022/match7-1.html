class Solution:
    def dfs(self, visited, grid, i, j, count, target, result):
        if grid[i][j] == 2 and count == target:
            result[0] += 1
            return
        if i+1 < len(grid) and grid[i+1][j] != -1 and visited[i+1][j] == 0:
            visited[i+1][j] = 1
            self.dfs(visited, grid, i+1, j, count+1, target, result)
            visited[i+1][j] = 0
        if j+1 < len(grid[0]) and grid[i][j+1] != -1 and visited[i][j+1] == 0:
            visited[i][j+1] = 1
            self.dfs(visited, grid, i, j+1, count+1, target, result)
            visited[i][j+1] = 0
        
if i-1 >= 0 and grid[i-1][j] != -1 and visited[i-1][j] == 0:
            visited[i-1][j] = 1
            self.dfs(visited, grid, i-1, j, count+1, target, result)
            visited[i-1][j] = 0
        if j-1 >= 0 and grid[i][j-1] != -1 and visited[i][j-1] == 0:
            visited[i][j-1] = 1
            self.dfs(visited, grid, i, j-1, count+1, target, result)
            visited[i][j-1] = 0
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        target = 0
        si 
= 0
        sj = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0 or grid[i][j] == 2:
                    target += 1
                if grid[i][j] == 1:
                    si = i
                    sj = j
        visited = [[0 for i in range(len(grid[0]))] for j in range(len(grid))]
        visited[si][sj] = 1
        result = [0]
        self.dfs(visited, grid, si, sj, 0, target, result)
        return result[0]
