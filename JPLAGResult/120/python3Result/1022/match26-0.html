class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        if not grid:
            return 0
        
        num = 0
        start = None
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0:
                    num += 1
                if grid[i][j] == 1:
                    start = (i, j)
        
        ret = 0
        
        visited = set()
        def bt(i, j):
            nonlocal ret
            if (i, j) in visited:
                return
            if i < 0 or j < 0 or i >= len(grid) or j >= len(grid[0]):
                return
            if grid[i][j] == -1:
                return
            if grid[i][j] == 2:
                if len(visited) == num + 1:
                    ret += 1
                else:
                    return
            visited.add((i, j))
            
            bt(i, j + 1)
            bt(i, j - 1)
            bt(i + 1, j)
            bt(i - 1, j)
            
            visited.remove((i, j))
            
        bt(start[0], start[1])
        
        return ret
