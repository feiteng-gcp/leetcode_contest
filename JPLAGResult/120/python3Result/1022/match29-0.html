class Solution:
    def uniquePathsIII(self, grid):
        n=len(grid)
        m=len(grid[0])
        self.ans=0
        zero=0
        one=[]
        two=[]
        di=[[0,1],[0,-1],[-1,0],[1,0]]
        for i in range(n):
            for j in range(m):
                if grid[i][j]==0:
                    zero+=1
                elif grid[i][j]==1:
                    one=[i,j]
                elif grid[i][j]==2:
                    two=[i,j]
        
        self.g=grid
        
        def dfs(p,cz):
            for d in di:
                x,y=d[0]+p[0],d[1]+p[1]
                
                if x<0 or x>=n or y<0 or y>=m or self.g[x][y]==-1:
                    continue
                    
                #print([x,y],self.g[x][y])
                if self.g[x][y]==2 and cz==zero:
                    self.ans+=1
                elif self.g[x][y]==0:
                    self.g[x][y]=3
                    dfs([x,y],cz+1)
                    self.g[x][y]=0
        
        dfs(one,0)
        
        return self.ans        
        
                
        """
        :type grid: List[List[int]]
        :rtype: int
        """
