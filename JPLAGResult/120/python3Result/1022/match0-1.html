class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        from copy import deepcopy

        r,c = len(grid),len(grid[0])
        lim = sum([i.count(0) for i in grid ])
        def grab(row,col,path,grid):
            grd = deepcopy(grid)
            if 0<=row<r and 0<=col<c and grd[row][col] != -1:
                if grd[row][col] == 0:
                    grd[row][col] = 1
                    return sum(list(map(grab,[row,row,row+1,row-1],[col+1,col-1,col,col],[path+1]*4,[grd]*4)))
                if grd[row][col] == 2 and path == lim:
                    return 1
                else:
                    return 0
            else:
                return 0

        for idx,i in enumerate(grid):
            if 1 in i:
                rs = idx
                cs = i.index(1)
                break

        res = sum(list(map(grab,[rs,rs,rs+1,rs-1],[cs+1,cs-1,cs,cs],[0]*4,[grid]*4)))
        
        return res
