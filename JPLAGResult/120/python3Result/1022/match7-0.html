class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        self.res = 0
        self.count =0
        start,end = 0,0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0:
                    self.count +=1
                if grid[i][j] ==1:
                    start,end = i,j
        # print(self.count)
        self.DFS(start, end, grid, 0)
        return self.res

    def neighbor(self, i, j, grid):
        R = len(grid)
        C = len(grid[0])
        for rr,cc in ((i+1,j), (i-1,j), (i,j+1), (i,j-1)):
            if 0<= rr < R and 0<= cc < C:
                yield (rr,cc)

    def DFS(self, i, j, grid, count):
        if grid[i][j] == 2:
            # print(count)
            if count-1 == self.count:
                self.res +=1
            return
        tmp = grid[i][j]
        for (rr,cc) in self.neighbor(i, j, grid):
            if grid[rr][cc] != 10 and grid[rr][cc] != -1:
                grid[i][j] = 10
                self.DFS(rr, cc, grid, count+1)
                grid[i][j] = tmp
