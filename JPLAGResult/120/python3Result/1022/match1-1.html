class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n, m = len(grid), len(grid[0])
        
        visited = [[False for _ in range(m)] for _ in range(n)] 
        # sol = []
        # solutions = []
        sols = 0
        
        req 
= 0
        startx, starty = None, None
        for i in range(n):
            for j in range(m):
                if grid[i][j] >= 0:
                    req += 1
                if grid[i][j] == 1:
                    startx, starty = i, j
                
        
        def search(x, y, count):
            # print(x, y, req, grid[x][y], count, req)
            
            if (count == req):
                # solutions.append(sol[:])
                nonlocal sols
                sols += 1
                return
            if grid[x][y] == 2:
                return
            
            for dx, dy in ((+1, 0), (-1, 0), (0, +1), (0, -1)):
                nx, ny = x+dx, y+dy
                if 0 <= nx < n and 0 <= ny < m and grid[nx][ny] >= 0 and not visited[nx][ny]:
                    visited[nx][ny] = True
                    search(nx, ny, count+1)
                    visited[nx][ny] = False 
                    
        visited[startx][starty] = True
        search(startx, starty, 1)
        return sols
