class Solution:

        if i<0 or i>m-1 or j<0 or j>n-1 or (i,j) in path:
            return
        else:
            # print("Exploring",i,j,"from:",path)               
            q.append((i,j,path))

        target = 0

                    q.append((i,j,set()))
                    # q.append((i,j,[]))

        s = 0
        while q:
            i,j,p = q.popleft()

            if grid[i][j] == 2:
                # print("Reached destination:path",p)

                continue
            # p.add((i,j))

            # print("path updated to",p)
            temp = p.union(set([(i,j)]))
            # print("path updated to",temp)            
            for a,b in [(i,j-1),(i,j+1),(i-1,j),(i+1,j)]:
                # self.helper(a,b,m,n,p+[(i,j)],q,grid)
                self.helper(a,b,m,n,temp,q,grid)
