class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        res = [0]
        
        moves = [[1,0],[-1,0],[0,1],[0,-1]]
        
        m = len(grid)
        n = len(grid[0])
        
        visited = [[False]*n for i in range(m)]
        
        x,y = -1,-1
        count = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j]==1:
                    x,y = i,j
                if grid[i][j]!=-1:
                    count+=1

        def DFS(i,j,remain):
            if 0<=i<m and 0<=j<n and not visited[i][j] and grid[i][j]!=-1:
                if grid[i][j] == 2:
                    if remain==1:
                        res[0]+=1
                    return 
                else:
                    visited[i][j] = True
                    for mx , my in moves:
                        DFS(i+mx,j+my,remain-1)
                    visited[i][j] = False
        
        DFS(x,y,count)
        return res[0]
