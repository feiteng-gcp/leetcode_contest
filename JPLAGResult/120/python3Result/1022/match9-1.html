class Solution:
    def uniquePathsIII(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n=len(grid)
        if n==0: return 0
        m=len(grid[0])
        if m==0: return 0
        x,y
=0,0
        cells=0
        for i in range(n):
            for j in range(m):
                if grid[i][j]==0: cells+=1
                if grid[i][j]==1: x,y=i,j
        
        self.ans=0
        def func(i,j,n,m,grid,visited,ln,cells):
            #print(i,j)
            if grid[i][j]==2 and cells==ln:
                self.ans+=1
                return
            elif grid[i][j]==2:
                return
            elif grid[i][j]==1:
                visited.add((i,j))
                for a,b in [(i,j+1),(i+1,j),(i-1,j),(i,j-1)]:
                    if a>=0 and a<=n-1 and b>=0 and b<=m-1 and grid[a][b]!=-1 and (a,b) not in visited:
                        func(a,b,n,m,grid,visited,ln,cells)                
            elif grid[i][j]==0:
                visited.add((i,j))
                for a,b in [(i,j+1),(i+1,j),(i-1,j),(i,j-1)]:
                    if a>=0 and a<=n-1 and b>=0 and b<=m-1 and grid[a][b]!=-1 and (a,b) not in visited:
                        func(a,b,n,m,grid,visited,ln+1,cells)
                visited.remove((i,j))
        func(x,y,n,m,grid,set([]),0,cells)
        return self.ans
