# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """

        self.total(root)
        self.visit(root)

ret
    def total(self,nd):
        if not nd:return 0,0
        l1,l2 = self.total(nd.left)
        r1,r2 = self.total(nd.right)
        nd.total = l1+r1+nd.val
        nd.size  = l2+r2+1
        return nd.total,nd.size
    def visit(self,nd):
        if nd:
            self.ret+=abs(nd.total-nd.size)
            self.visit(nd.left)
            self.visit(nd.right)
