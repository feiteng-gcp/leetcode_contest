# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans=0
        
        
def find(p):
            if not p:
                return 0,0
            lc,lm=find(p.left)
            rc,rm=find(p.right)
            tc=lc+rc+p.val-1
            tm=lm+rm+abs(lc)+abs(rc)
            return tc,tm
        return find(root)[1]
