# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        def grab(node):
            if not node:
                return [0,0]
            le,ri = grab(node.left),grab(node.right)
            coins = le[0]+ri[0]+node.val-1
            step = le[1]+ri[1]+abs(coins)
            return [coins,step]
        return grab(root)[1]
