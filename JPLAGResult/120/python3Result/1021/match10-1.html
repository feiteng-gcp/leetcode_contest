# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root):
        d, b = self.dfs(root)
        return d
        
    
def dfs(self, node):
        if not node:
            return 0, 0
        ld, lb = self.dfs(node.left)
        rd, rb = self.dfs(node.right)
        return abs(lb) + abs(rb) + ld + rd, node.val + lb + rb - 1
