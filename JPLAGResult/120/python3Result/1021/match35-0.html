# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.total = 0
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        n, c = self.traverse(root)
        return self.total
        
    def traverse(self, node):
        if node is None:
            return 0, 0
        nl, cl = self.traverse(node.left)
        nr, cr = self.traverse(node.right)
        val = node.val
        if nl > cl:
            # if val + nl - cl > 1:
            self.total += nl - cl #node.val + nl - cl - 1
            val -= nl - cl
        else:
            self.total += cl - nl
            val += cl - nl

        if nr > cr:
            # self.total += nr - cr
            # if val + nr - cr > 1:
            self.total += nr - cr #val + nr - cr - 1
            val -= nr - cl
        else:
            self.total += cr - nr
            val += cr - nr

        return nl + nr + 1, cl + cr + node.val
