# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.ans = 0
    def helper(self, root):
        # 自底向上
        # 正数表示多的硬币，负数表示缺硬币，0表示不多不少
        if not root:
            return 0
        left = self.helper(root.left)
        right = self.helper(root.right)
        self.ans += abs(left) + abs(right)
        cur = root.val + left + right
        return cur-1
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.helper(root)
        return self.ans
