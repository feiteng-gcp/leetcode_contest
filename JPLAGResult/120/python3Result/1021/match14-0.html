# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        res = self.helper(root)
        return res[1]
        
    def helper(self, root):
        if not root:
            return (0, 0)
        rebalenceL, stepL = self.helper(root.left)
        rebalenceR, stepR = self.helper(root.right)
        return (root.val - 1 + rebalenceL + rebalenceR, stepL + stepR + abs(rebalenceL) + abs(rebalenceR))
