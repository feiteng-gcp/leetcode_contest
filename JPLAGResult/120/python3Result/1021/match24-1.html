# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.x=0
        def func(root):
            if not root: return 0
            l=func(root.left)
            r=func(root.right)
            self.x+=abs(l)
            self.x+=abs(r)
            ans=l+r-1+root.val
            return ans
        func(root)
        return self.x
