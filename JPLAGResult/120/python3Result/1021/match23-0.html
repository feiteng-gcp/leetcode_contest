# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if root is None:
            return 0
        def move_coin(node):
            if node is None:
                return 0, 0, 0
            l_coin, l_node, l_move = move_coin(node.left)
            r_coin, r_node, r_move = move_coin(node.right)
            coin = l_coin + r_coin + node.val
            node = l_node + r_node + 1
            move = l_move + r_move + abs(l_coin - l_node) + abs(r_coin - r_node)
            return coin, node, move
        return move_coin(root)[2]
