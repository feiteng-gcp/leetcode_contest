# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def helper(self,node):
        if not node:
            return 0,0
        ld,lt = self.helper(node.left)
        rd,rt = self.helper(node.right)
        diff = ld+rd+node.val-1
        tot = lt+rt+abs(diff)
        return diff,tot
    
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        return self.helper(root)[1]
