class Solution:
    def helper(self, root):
        if root == None:
            return (0, 0)
        lbalance, lcost = self.helper(root.left)
        rbalance, rcost = self.helper(root.right)
        balance = rbalance + lbalance + root.val - 1
        cost = lcost + rcost + abs(balance)
        return balance, cost

    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        return self.helper(root)[1]
