# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    
def sumcoin(self, root):
        if not root:
            return 0, 0
        a = self.sumcoin(root.left)
        b = self.sumcoin(root.right)
        return (root.val + a[0] + b[0], 1 + a[1] + b[1])
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if not root:
            return 0
        elif not root.left and not root.right:
            return 0
        elif not root.left:
            tmp = root.val-1
            root.val = 1
            root.right.val += tmp
            return abs(tmp) + self.distributeCoins(root.right)
        elif not root.right:
            tmp = root.val-1
            root.val = 1
            root.left.val += tmp
            return abs(tmp) + self.distributeCoins(root.left)
        else:
            leftcoin, leftnode = self.sumcoin(root.left)
            rightcoin, rightnode = self.sumcoin(root.right)
            root.left.val += leftnode-leftcoin
            root.right.val += rightnode-rightcoin
            root.val = 1
            return abs(leftnode-leftcoin) + abs(rightnode-rightcoin) + self.distributeCoins(root.left) + self.distributeCoins(root.right)
