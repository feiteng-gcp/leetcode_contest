# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        def extraCoins(root):
            if root is None:
                return 0
            left = extraCoins(root.left)
            right = extraCoins(root.right)
            nonlocal ret
            ret += abs(left)+abs(right)
            return root.val+left+right-1
        
        ret = 0
        extraCoins(root)
        return ret
