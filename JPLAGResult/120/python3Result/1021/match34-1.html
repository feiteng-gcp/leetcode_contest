# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        
        def surplus(node):
            if not node:
                return 0
            ans = node.val-1
            if node.left:
                ans += surplus(node.left)
            if node.right:
                ans += surplus(node.right)
            return ans
        
        
if not root:
            return 0
        
        left_s = surplus(root.left)
        right_s = surplus(root.right)
        ans = abs(left_s) + abs(right_s)
        if root.left:
            root.left.val -= left_s
        if root.right:
            root.right.val -= right_s
        return ans + self.distributeCoins(root.left) + self.distributeCoins(root.right)
