# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def distributeCoins(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        # L, R, P = {}, {}, {}
        self.ans = 0
        
        def dfs(node):
            if not node:
                return 0
            l = dfs(node.left)
            r = dfs(node.right)
            res = l + r + node.val - 1
            # print(node.val, l, r, res)
            self.ans += abs(res)
            return res
        
        dfs(root)
        return self.ans
