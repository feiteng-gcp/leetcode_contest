class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> set = new HashSet<>();
        for (String email : emails) {
            String[] arr = email.split("@");
            
            int plusInd = arr[0].indexOf("+");
            if (plusInd >= 0) {
                arr[0] = arr[0].substring(0, plusInd);
            }
            arr[0] = arr[0].replace(".", "");
            
            set.add(arr[0] + '@' + arr[1]);
        }
        
        return set.size();
    }
}
