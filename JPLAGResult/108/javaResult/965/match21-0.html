class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> set = new HashSet<>();
        for (String s : emails) {
            int l = s.indexOf('@');
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < l; ++i) {
                if (s.charAt(i) == '.') continue;
                if (s.charAt(i) == '+') break;
                sb.append(s.charAt(i));
            }
            set.add(sb.toString() + s.substring(l));
        }
        return set.size();
    }
}
