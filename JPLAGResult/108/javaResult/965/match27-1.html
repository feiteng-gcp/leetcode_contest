class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> set = new HashSet<>();
        for (String email : emails) {
            StringBuilder sb = new StringBuilder();
            String[] temp = email.split("@");
            for (char c : temp[0].toCharArray()) {
                
if (c == '+') {
                    break;
                } else if (c == '.') {
                    continue;
                } else {
                    sb.append(c);
                }
            }
            sb.append("@");
            sb.append(temp[1]);
            set.add(sb.toString());
        }
        // System.out.println(set);
        return set.size();
    }
}
