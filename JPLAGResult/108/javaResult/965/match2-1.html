class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> allEmail = new HashSet<>();
        for (String email:emails) {
            String[] parts = email.split("@");
            parts[0] = parts[0].replaceAll("\\.","");
            int firstPlus = parts[0].indexOf('+');
            if (firstPlus!=-1) {
                parts[0] = parts[0].substring(0,firstPlus);
            }
            allEmail.add(parts[0]+"@"+parts[1]);
        }
        return allEmail.size();
    }
}
