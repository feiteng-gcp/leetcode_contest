class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> set = new HashSet<>();
        for (String email : emails) {
            int a = email.indexOf('@');
            String suffix = email.substring(a);
            String prefix = email.substring(0, a);
            int pl = prefix.indexOf('+');
            if (pl > -1) {
                prefix = prefix.substring(0, pl);
            }
            prefix = prefix.replaceAll("\\.", "");
            set.add(prefix + suffix);
        }
        // System.out.println(set);
        return set.size();
    }
}
