class Solution {
    public int numUniqueEmails(String[] emails) {
        HashSet<String> set = new HashSet<>();
        for (String s : emails)
        {
            int indexDot = s.indexOf('.');
            int indexAt = s.indexOf('@');
            int indexPlus = s.indexOf('+');
            String temp = "";
            if (indexPlus >= 0 && indexAt > indexPlus)
                temp = s.substring(0, indexPlus);
            else
                temp = s.substring(0, indexAt);
            temp = temp.replaceAll("\\.", "");
            temp += s.substring(indexAt, s.length());
            //System.out.println(temp);
            set.add(temp);
        }
        return set.size();
    }
}
