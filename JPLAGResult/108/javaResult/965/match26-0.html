class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> s = new HashSet<>();
        for(String e : emails) {
            String[] cur = e.split("@");
            String to = cur[0].replaceAll("\\.","");
            int idx  = to.indexOf('+');
            if(idx > 0) to = to.substring(0,idx);
            //System.out.println(to);
            s.add(to+"@"+cur[1]);
        }
        return s.size();
    }
}
