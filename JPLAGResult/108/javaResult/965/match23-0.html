class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> set = new HashSet<>();
        for (String add : emails) {
            StringBuilder s = new StringBuilder();
            for (char c : add.toCharArray()) {
                if (c == '+' || c == '@') break;
                if (c == '.') continue;
                s.append(c);
            }
            int index = add.indexOf("@");
            s.append(add.substring(index));
            set.add(s.toString());
        }
        return set.size();  
    }
}
