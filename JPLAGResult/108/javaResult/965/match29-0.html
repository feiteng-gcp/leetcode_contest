class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> set = new HashSet<>();
        for (String s : emails) {
            String sign = s.substring(s.lastIndexOf("@"), s.length());
            String loc = s.substring(0, Math.min(s.lastIndexOf("@"), s.indexOf("+")));
            loc = loc.replace(".", "");
            set.add(loc + sign);
        }
        return set.size();
    }
}
