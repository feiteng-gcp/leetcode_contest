class Solution {
    public int numUniqueEmails(String[] emails) {
        HashSet<String> set = new HashSet<>();
        for (String email : emails) {
            String[] hostDomain = email.split("@");
            String host = hostDomain[0];
            String domain = hostDomain[1];
            String[] rem = host.split("\\+");
            String[] remm = rem[0].split("\\.");
            String part = "";
            for (String t : remm) {
                part += t;
            }
            set.add(part+"@"+domain);

        }
        return set.size();
    }
}
