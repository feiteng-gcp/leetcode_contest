class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> uniqMail = new HashSet();
        for(String email: emails){
            String[] localDomain = email.split("@");
            localDomain[0] = localDomain[0].split("\\+")[0];
            localDomain[0] = String.join("", localDomain[0].split("\\."));
            String mail = String.join("@", localDomain);
            uniqMail.add(mail);
        }
        return uniqMail.size();
    }
}
