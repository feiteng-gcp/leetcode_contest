class Solution {
    public int numUniqueEmails(String[] emails) {
        HashMap<String, Integer> m = new HashMap<>();
        for (String s: emails) {
            String[] parts = s.split("@");
            String localName = parts[0].split("\\+")[0].replaceAll("\\.","");
            String address = localName + "@" + parts[1];
            
m.put(address, m.getOrDefault(address, 0) + 1);
        }
        // int res = 0;
        // for (String key: m.keySet()) {
        //     if (m.get(key) > 1) {
        //         res++;
        //     }
        // }
        return m.size();
    }
}
