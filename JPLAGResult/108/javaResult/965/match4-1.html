import java.util.HashSet;
import java.util.Set;


class Solution {
    public int numUniqueEmails(String[] emails) {
    	Set<String> res = new HashSet<String> ();
    	for (String e:emails) {
    		int j = e.indexOf('@');
    		String l = e.substring(0,j);
    		l = l.replace(
".", "");
//    		Helper.tr(l);
    		int i = l.indexOf('+');
    		if (i>=0) {
    			l = l.substring(0, i);
    		}
    		e = l+e.substring(j);
//    		Helper.tr(e);
    		res.add(e);
    	}
		return res.size();
    	
    }
}
