class Solution {
    public int numUniqueEmails(String[] emails) {
        HashSet<String> ref = new HashSet<>();
        for (String s : emails) {
            int indexOfAt = s.indexOf('@');
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < indexOfAt; i++) {
                if (s.charAt(i) == '+') {
                    break;
                } else if (s.charAt(i) != '.') {
                    sb.append(s.charAt(i));
                }
            }
            sb.append(s.substring(indexOfAt));
            ref.add(sb.toString());
        }
        return ref.size();
    }
}
