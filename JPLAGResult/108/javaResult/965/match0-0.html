class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> set = new HashSet<>();
        for (String e : emails) {
            String[] ss = e.split("@");
            ss[0] = ss[0].replaceAll("\\.", "");
            int index = ss[0].indexOf("+");
            if (index >= 0) {
                ss[0] = ss[0].substring(0, index);
            }
            // System.out.println(ss[0]);
            set.add(ss[0] + "@" + ss[1]);
        }
        return set.size();
    }
}
