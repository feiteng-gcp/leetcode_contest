olution {
	    public int numUniqueEmails(String[] emails) {
	        Set<String> es = new HashSet<>();
	        for(String e : emails){
	        	String[] s = e.split("@");
	        	s[0] = s[0].replace(".", "");
	        	int ind = s[0].indexOf("+");
	        	if(ind >= 0){
	        		s[0] = s[0].substring(0, ind);
	        	}
	        	es.add(s[0] + "@" + s[1]);
	        }
	        return es.size();
	    }
	}
