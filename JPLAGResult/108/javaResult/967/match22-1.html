class Solution {
    public int minFallingPathSum(int[][] A) {
        if(A == null || A.length == 0 || A[0] == null || A[0].length == 0) return 0;
        
int len = A.length;
        int[][] paths = new int[len][len];
        int res = Integer.MAX_VALUE;
        for(int i = len - 1; i >= 0; i--){
            for(int j = 0; j < len; j++){
                if(i == len - 1){
                    paths[i][j] = A[i][j];
                } else {
                    paths[i][j] = paths[i + 1][j];
                    if(j != 0){
                        paths[i][j] = Math.min(paths[i][j], paths[i + 1][j - 1]);
                    }
                    if(j != len - 1){
                        paths[i][j] = Math.min(paths[i][j], paths[i + 1][j + 1]);
                    }
                    paths[i][j] += A[i][j];
                }
                if(i == 0){
                    res = Math.min(res, paths[i][j]);
                }
            }
        }
        return res;
    }
}
