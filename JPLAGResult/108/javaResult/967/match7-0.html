class Solution {
    public int minFallingPathSum(int[][] A) {
        int n = A.length;
        long dp[][] = new long[n][n];
        for (int i = 0; i < n; i++) {
            dp[0][i] = A[0][i];
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < n; j++) {
                dp[i][j] = dp[i - 1][j] + A[i][j];
                if (j > 0) {
                    dp[i][j] = Math.min(dp[i][j], dp[i - 1][j - 1] + A[i][j]);
                }
                if (j < n - 1) {
                    dp[i][j] = Math.min(dp[i][j], dp[i - 1][j + 1] + A[i][j]);
                }
            }
        }
        long ans = dp[n - 1][0];
        for (int j = 1; j < n; j++) {
            ans = Math.min(ans, dp[n - 1][j]);
        }
        return (int) ans;
    }
}
