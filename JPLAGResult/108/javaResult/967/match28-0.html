class Solution {
    public int minFallingPathSum(int[][] A) {
    	int n = A.length;
    	if (n == 0) return 0;
    	int m = A[0].length;
    	if (m == 0) return 0;
    	int[][] dp = new int[n][m];
    	for (int i = 0; i < n; ++i) Arrays.fill(dp[i], 0x3f3f3f3f);
    	for (int i = 0; i < m; ++i) {
    		dp[0][i] = A[0][i];
    	}
    	for (int i = 1; i < n; ++i) {
    		for (int j = 0; j < m; ++j) {
    			dp[i][j] = Math.min(dp[i][j], dp[i - 1][j] + A[i][j]);
    			if (j - 1 >= 0) dp[i][j] = Math.min(dp[i][j], dp[i - 1][j - 1] + A[i][j]);
    			if (j + 1 <  m) dp[i][j] = Math.min(dp[i][j], dp[i - 1][j + 1] + A[i][j]);
    		}
    	}
    	int max = 0x3f3f3f3f;
    	for (int i = 0; i < m; ++i) {
    		max = Math.min(max, dp[n - 1][i]);
    	}
    	return max;
    }
}
