class Solution {
    public int minFallingPathSum(int[][] A) {
        int n = A.length;
        
int[][] dp = new int[n][n];
        // initialize the first row
        for (int i = 0; i < n; i++) {
            dp[0][i] = A[0][i];
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (j == 0) { dp[i][j] = Math.min(dp[i - 1][j], dp[i - 1][j + 1]) + A[i][j]; }
                else if (j == n - 1) { dp[i][j] = Math.min(dp[i - 1][j], dp[i - 1][j - 1]) + A[i][j]; }
                else {
                    dp[i][j] = Math.min(Math.min(dp[i - 1][j], dp[i - 1][j - 1]), dp[i - 1][j + 1]) + A[i][j];
                }
            }
        }
        
        int min = dp[n - 1][n - 1];
        for (int i = 0; i < n; i++) {
            min = Math.min(dp[n - 1][i], min);
        }
        
        return min;
    }
}
