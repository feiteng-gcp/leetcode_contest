class Solution {
    public int minFallingPathSum(int[][] A) {
        
int n = A.length;
        int[] dp = new int[n];
        for (int i = 0; i < n; i++) {
            dp[i] = A[0][i];
        }
        for (int i = 1; i < n; i++) {
            int[] newDp = new int[n];
            
for (int j = 0; j < n; j++) {
                int candidate1 = j == 0 ? Integer.MAX_VALUE : dp[j - 1];
                int candidate2 = dp[j];
                int candidate3 = j == n - 1 ? Integer.MAX_VALUE : dp[j + 1];
                newDp[j] = Math.min(candidate1, Math.min(candidate2, candidate3)) + A[i][j];
            }
            dp = newDp;
        
}
        int res = dp[0];
        for (int num : dp) {
            res = Math.min(res, num);
        }
        return res;
    }
}
