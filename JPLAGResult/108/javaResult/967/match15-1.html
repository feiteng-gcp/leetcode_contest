class Solution {
    public int minFallingPathSum(int[][] A) {
        int n = A.length;
        int[][] dp = new int[n][n];
        for (int i = 0; i < n; i++) {
            dp[0][i] = A[0][i];
        }
        
        for (int i = 1; i < n; i++) {
            
for (int j = 0; j < n; j++) {
                int left = j == 0 ? Integer.MAX_VALUE : dp[i - 1][j - 1] + A[i][j];
                int mid = dp[i - 1][j] + A[i][j];
                int right = j == n - 1 ? Integer.MAX_VALUE : dp[i - 1][j + 1] + A[i][j];
                dp[i][j] = Math.min(left,
                                    Math.min(mid, right));
            }
        
}
        int res = dp[n - 1][0];
        for (int i = 1; i < n; i++) {
            res = Math.min(dp[n - 1][i], res);
        }
        return res;
    }
}
