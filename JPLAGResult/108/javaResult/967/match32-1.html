class Solution {
    public int minFallingPathSum(int[][] A) {
        int[] dp = new int[A.length];
        for (int i = 0; i < dp.length; i++) {
            dp[i] = A[0][i];
        }
        for (int i = 1; i < A.length; i++) {
            int[] tmpDp = new int[dp.length];
            Arrays.fill(tmpDp, Integer.MAX_VALUE);
            for (int j = 0; j < dp.length; j++) {
                tmpDp[j] = Math.min(tmpDp[j], dp[j] + A[i][j]);
                
if (j - 1 >= 0) {
                    tmpDp[j] = Math.min(tmpDp[j], dp[j - 1] + A[i][j]);
                }
                if (j + 1 < A.length) {
                    tmpDp[j] = Math.min(tmpDp[j], dp[j + 1] + A[i][j]);
                }
            }
            dp = tmpDp;
        
}
        int res = Integer.MAX_VALUE;
        for (int tmp : dp) {
            res = Math.min(res, tmp);
        }
        return res;
    }
}
