class Solution {
    public int minFallingPathSum(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        
int[][] sum = new int[m][n];
        
        for (int i = 0; i < n; i++) {
            sum[0][i] = A[0][i];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                sum[i][j] = sum[i - 1][j];
                if (j > 0) {
                    sum[i][j] = Math.min(sum[i][j], sum[i - 1][j - 1]);
                }
                if (j + 1 < n) {
                    sum[i][j] = Math.min(sum[i][j], sum[i - 1][j + 1]);
                }
                sum[i][j] += A[i][j];
            }
        }
        int min = sum[m - 1][0];
        for (int i = 1; i < n; i++) {
            min = Math.min(min, sum[m - 1][i]);
        }
        return min;
    }
}
