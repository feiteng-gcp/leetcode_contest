class Solution {
    // 1 2 3
    // 4 5 6
    // 7 8 9
    public int minFallingPathSum(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        int[][] dp = new int[m][n];
        
        for(int j = 0; j < n; j++) {
            dp[0][j] = A[0][j];
        }
        
        for(int i = 1; i < m; i++) {
            for(int j = 0; j < n; j++) {
                int c1 = j == 0 ? Integer.MAX_VALUE : dp[i - 1][j - 1];
                int c2 = dp[i - 1][j];
                int c3 = j == n - 1 ? Integer.MAX_VALUE : dp[i - 1][j + 1];
                
                dp[i][j] = Math.min(c1, Math.min(c2, c3)) + A[i][j];
            }
        }
        
        int res = Integer.MAX_VALUE;
        for(int j = 0; j < n; j++) {
            res = Math.min(res, dp[m - 1][j]);
        }
        
        return res;
        
    }
}
