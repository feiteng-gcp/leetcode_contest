class Solution {
    public int minFallingPathSum(int[][] A) {
        int[][] path = new int[A.length][A[0].length];
        for (int j = 0; j < A[0].length; j++) {
            path[0][j] = A[0][j];
        }
        for (int i = 1; i < A.length; i++) {
            path[i][0] = Math.min(path[i-1][0], path[i-1][1])+A[i][0];
            for (int j = 1; j < A[0].length-1; j++) {
                path[i][j] = Math.min(path[i-1][j+1], Math.min(path[i-1][j], path[i-1][j-1]))+A[i][j];
            }
            path[i][A[0].length-1] = Math.min(path[i-1][A[0].length-2], path[i-1][A[0].length-1])+A[i][A[0].length-1];
        }
        int res = path[A.length-1][0];
        for (int i = 1; i < A[0].length; i++) {
            res = Math.min(res, path[A.length-1][i]);
        }
        return res;
    }
}
