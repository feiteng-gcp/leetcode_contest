class Solution {
    public int[] beautifulArray(int N) {
        if (N == 1) {
            return new int[]{1};
        }
        
        int[] a = new int[N];
        int[] odd = beautifulArray((N + 1)/ 2);
        int[] even = beautifulArray(N / 2);
        int i = 0;
        for (int j = 0; j < odd.length; j++) {
            a[i++] = odd[j] * 2 - 1;
        }
        for (int j = 0; j < even.length; j++) {
            a[i++] = even[j] * 2;
        }
        return a;
    }
}
