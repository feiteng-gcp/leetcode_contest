class Solution {
    public int[] beautifulArray(int N) {
        int[] res = beautiful(N);
        return res;
    }

    private int[] beautiful(int N) {
        if (N == 1) {
            return new int[]{1};
        } else {
            int[] result = new int[N];
            int[] odd = beautiful((N + 1) / 2);
            int[] even = beautiful(N / 2);
            
for (int i = 0; i < (N + 1) / 2; i++) {
                result[i] = odd[i] * 2 - 1;
            }
            for (int i = 0; i < N / 2; i++) {
                result[i + odd.length] = even[i] * 2;
            }
            return result;
        }
    }
}
