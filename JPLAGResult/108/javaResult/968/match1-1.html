class Solution {
    // 1 2 3 4
    // 2 1 4 3
    
    
    // 1 2 3 4 5
    // 3 1 2 4 5
    
    // 1 2 3 4 5 6 7 8 9
    // 5 1 2 3 4 6 7 8 9
    
    public int[] beautifulArray(int N) {
        int[] res = new int[N]; // 1 ~ N
        if(N == 1) {
            res[0] = 1;
            return res;
        }

        if(N % 2 == 0) {
            int[] subarr = beautifulArray(N / 2);
            int idx = 0;
            for(int i = 0; i < subarr.length; i++ ) {
                res[idx++] = subarr[i] * 2 - 1;
            }
            for(int i = 0; i < subarr.length; i++) {
                res[idx++] = subarr[i] * 2;
            }
        } else {
            
int[] left = beautifulArray(N - N / 2);
            int[] right = beautifulArray(N / 2);
            int idx = 0;
            for(int i = 0; i < left.length; i++) {
                res[idx++] = 2 * left[i] - 1;
            }
            for(int i = 0; i < right.length; i++) {
                res[idx++] = 2 * right[i];
            }
        }

        // System.out.println(Arrays.toString(res));

        return res;
    }
    
}
