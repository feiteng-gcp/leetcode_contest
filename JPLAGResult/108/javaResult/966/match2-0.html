class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        Map<Integer, Integer> pre = new HashMap<>();
        pre.put(0, 1);
        int curSum = 0;
        int sum = 0;
        for (int i = 0; i < A.length; i++) {
            curSum += A[i];
            if (pre.containsKey(curSum - S)) {
                sum += pre.get(curSum - S);
            }
            pre.put(curSum, pre.getOrDefault(curSum, 0) + 1);
        }
        return sum;
    }
}
