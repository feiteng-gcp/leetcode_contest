class Solution {
    
    public int numSubarraysWithSum(int[] A, int S) {
    	int count = 0;
    	for(int i = 0; i < A.length; i++) {
            int sum = 0;
    		for(int j = i; j < A.length && sum <= S; j++) {
    			if(sum + A[j] == S) {
    				count++;
    			}
    			sum += A[j];
    		}
    	}
    	return count;

    }
    public int numSubarraysWithSum1(int[] A, int S) {
    	int count = 0;
    	int left = 0, sum = 0;
    	for(int i = 0; i < A.length; ) {
    		if(sum + A[i] == S) {
    			count++;
    			sum += A[i];
    			i++;
    		} else if(sum + A[i] > S) {
    			sum -= A[left];
    			left++;
    		} else {
    			sum += A[i];
    			i++;
    		}
    	}
    	while(left < A.length) {
			sum -= A[left];
			if(sum == S) {
				count++;
			}
			left++;
    	}
    	return count;        
    }
}
