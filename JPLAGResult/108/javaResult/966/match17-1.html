class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        if (A == null || A.length == 0 || S < 0) {
            return 0;
        }
        int left = -1;
        int right = 0;
        int sum = 0;
        int count = 0;
        for (right = 0; right < A.length; right++) {
            sum += A[right];
            while (left < right && sum > S) {
                left++;
                sum -= A[left];
            }
            int curr = left;
            int currSum = sum;
            while (curr < right && currSum == S) {
                count++;
                curr++;
                currSum -= A[curr];
            }
        }
        return count;
    }
}
