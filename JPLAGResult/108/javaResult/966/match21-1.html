class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        HashMap<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int cur = 0;
        int res = 0;
        for (int i = 0; i < A.length; i++) {
            cur += A[i];
            int t = cur - S;
            res += map.getOrDefault(t, 0);
            map.put(cur, map.getOrDefault(cur, 0) + 1);
            
        }
        return res;
    }
}
