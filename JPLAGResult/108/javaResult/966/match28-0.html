class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        // int i = 0, j = 0;

        // int res = 0;
        // while (i < A.length && j < A.length) {
        //     if (A[j] == 1) sum++;

        //     while (i < j && sum == S) {
        //         if (A[i] == 1) sum--;
        //         i++;
        //         res++;
        //     }
        //     j++;
        // }
        // // if (sum == S) res++;
        // return res;
        int[] dp = new int[A.length + 1];
        for (int i = 1; i <= A.length; i++) {
            dp[i] = dp[i - 1] + A[i - 1];
        }
        int count = 0;
        for (int i = 0; i <= A.length; i++) {
            for (int j = 0; j < i; j++) {
                if (dp[i] - dp[j] == S) count++;
            }
        }
        return count;
    }
}
