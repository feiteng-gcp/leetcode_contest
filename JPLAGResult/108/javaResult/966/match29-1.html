class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int res = 0;
        int sum = 0;
        for(int num : A) {
            sum += num;
            if (map.containsKey(sum - S)) {
                res += map.get(sum - S);
            
}
            map.put(sum, map.getOrDefault(sum, 0)  +1);
        }
        return res;
    }
}
