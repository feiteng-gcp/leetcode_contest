class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        if (A == null || A.length == 0) {
            return 0;
        }
        int[] preSum = new int[A.length+1];
        preSum[0] = 0;
        
for (int i = 1; i < A.length+1; i++) {
            preSum[i] = preSum[i-1] + A[i-1];
        }
        int count = 0;
        for (int i = 1; i < preSum.length; i++) {
            for (int j = 0; j < i; j++) {
                if (preSum[i] - preSum[j] == S) {
                    count++;
                }
            }
        }
        return count;
    }
}
