class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int acc = 0;
        int res = 0;

        for(int i = 0; i < A.length; i++) {
            int curr = A[i];
            acc += curr;

            // look for S - curr
            res += map.getOrDefault(acc - S, 0);


            map.put(acc, map.getOrDefault(acc, 0) + 1);
        }

        return res;      
    }
}
