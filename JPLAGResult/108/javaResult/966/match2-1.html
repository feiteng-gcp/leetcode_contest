class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        // preSum, count
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int preSum = 0;
        int count = 0;
        for (int num : A) {
            preSum += num;
            if (map.containsKey(preSum - S)) {
                count += map.get(preSum-S);
            }
            map.put(preSum, map.getOrDefault(preSum, 0)+1);
        }
        return count;
        
    }
}
