class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        if (A == null || A.length == 0) {
            return 0;
        }
        
        int[] dp = new int[A.length + 1];
        for (int i = 1; i < dp.length; i++) {
            dp[i] = dp[i - 1] + A[i - 1];
        }
        
        int cnt = 0;
        for (int i = 0; i < dp.length; i++) {
            for (int j = i + 1; j < dp.length; j++) {
                if (dp[j] - dp[i] == S) {
                    cnt++;
                }
            }
        }
        
        return cnt;
    }
}
