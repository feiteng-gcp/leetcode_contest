class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        int result = 0;
        int currSum = 0;
        Map<Integer, Integer> sumCount = new HashMap<>();
        sumCount.put(0, 1);
        for (int i = 0; i < A.length; i++) {
            currSum += A[i];
            int cnt = sumCount.getOrDefault(currSum, 0);
            result += sumCount.getOrDefault(currSum - S, 0);
            sumCount.put(currSum, cnt + 1);
        }
        return result;
    }
}
