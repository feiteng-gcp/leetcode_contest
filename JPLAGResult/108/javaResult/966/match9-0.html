class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        Map<Integer,Integer> map = new HashMap<>();
        map.put(0,1);
        int sum = 0;
        int cnt = 0;
        for(int i : A){
            sum += i;
            cnt += map.getOrDefault(sum-S,0);
            map.put(sum,map.getOrDefault(sum,0)+1);
        }
        return cnt;
    }
}
