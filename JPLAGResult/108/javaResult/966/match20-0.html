class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        Map<Integer, Integer> ref = new HashMap<>();
        int sum = 0, ans = 0;
        ref.put(0, 1);
        for (int a : A) {
            sum += a;
            if (sum >= S) {
                ans += ref.containsKey(sum - S) ? ref.get(sum - S) : 0;
            }
            ref.put(sum, ref.getOrDefault(sum, 0) + 1);
        }
        return ans;
    }
}
