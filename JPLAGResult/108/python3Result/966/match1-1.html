class Solution:
    def numSubarraysWithSum(self, A, S):
        """
        :type A: List[int]
        :type S: int
        :rtype: int
        """
        one_index = []
        
        for i,num in enumerate(A):
            if num == 1:
                one_index.append(i)
        
        if len(one_index) < S:
            return 0
        
        one_index.insert(0,-1)
        one_index.append(len(A))

        if S == 0:
            ans = 0
            for i in range(len(one_index)-1):
                ans += (one_index[i+1]-one_index[i])*(one_index[i+1]-one_index[i]-1)//2
            return ans 
        
        start,end = 1,S
        ans 
= 0
        
        while end < len(one_index)-1:
            ans += (one_index[start]-one_index[start-1]) * (one_index[end+1]-one_index[end])
            start += 1
            end += 1
        
        return ans
