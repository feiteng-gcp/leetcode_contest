class Solution:
    def numSubarraysWithSum(self, A, S):
        """
        :type A: List[int]
        :type S: int
        :rtype: int
        """
        if sum(A) < S or not A:
            return 0
        index = [-1]
        for i in range(len(A)):
            if A[i] == 1:
                index.append(i)
        index.append(len(A))
        prev, left, right, next = 0, 1, S, S + 1
        ans = 0
        if S == 0:
            left, right = 0, 1
            while right < len(index):
                ans += (1 + index[right] - index[left] - 1) * (index[right] - index[left] - 1) // 2
                left += 1
                right += 1
            return ans
        elif S == 1:
            left, mid, right = 0, 1, 2
            while right < len(index):
                ans += (index[mid] - index[left]) * (index[right] - index[mid]) 
                left += 1
                mid += 1
                right += 1
            return ans
        else:
            while next < len(index):
                if index[right] > index[left]:
                    ans += (index[left] - index[prev]) * (index[next] - index[right])
                prev += 1
                left += 1
                right += 1
                next += 1
            return ans
