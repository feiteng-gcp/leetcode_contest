class Solution:
    def numSubarraysWithSum(self, A, S):
        """
        :type A: List[int]
        :type S: int
        :rtype: int
        """
        A_sum = sum(A)
        N = len(A)
        if A_sum<S:
            return 0
        pos = [0] * (A_sum+2)
        cnt = 0
        for i, e in enumerate(A):
            if e==1:
                cnt += 1
                pos[cnt] = i
        pos[0] = -1
        pos[cnt+1] = N
        ttl = 0
        for i in range(S, cnt+1):
            if S>0:
                ttl += (pos[i-S+1] - pos[i-S]) * (pos[i+1] - pos[i])
            else:
                ttl += (pos[i+1] - pos[i])*(pos[i+1] - pos[i]-1)//2
        # print(pos)
        return ttl
