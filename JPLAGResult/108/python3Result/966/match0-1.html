class Solution:
    def numSubarraysWithSum(self, A, S):
        """
        :type A: List[int]
        :type S: int
        :rtype: int
        """
        ones = [-1]
        ans = 0
        for i, b in enumerate(A+[1]):
            if b==0:
                continue
            ones.append(i)
            if len
(ones) == S+2:
                #print(ones)
                if S>0:
                    ans += (ones[1]-ones[0])*(ones[-1]-ones[-2])
                else:
                    ans += (ones[1]-ones[0]-1)*(ones[1]-ones[0])//2
                ones.pop(0)
        return ans
