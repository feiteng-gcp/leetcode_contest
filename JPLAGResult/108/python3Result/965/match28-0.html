class Solution:
  def numUniqueEmails(self, emails):
    """
    :type emails: List[str]
    :rtype: int
    """
    sent = set()
    for email in emails:
      name, domain = email.split("@")
      name = name[:name.find("+")]
      name = ''.join(name.split("."))
      sent.add(name + "@" + domain)
    print(sent) 
    return len(sent)
