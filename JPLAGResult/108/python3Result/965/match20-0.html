class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        addrs = set()
        for email in emails:
            name, domain = email.split("@")
            name = name.replace(".","")
            name = name.split("+")[0]
            final = name +"@"+domain
            addrs.add(final)
        #print(addrs)
        return len(addrs)
