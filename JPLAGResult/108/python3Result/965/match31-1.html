class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        ans = 0
        memo
=set()
        for i in emails:
            name,domain = i.split('@')
            name = name.replace('.','')
            idx = name.find('+')
            if(idx !=-1):
                name = name[0:idx]
            if((name,domain) in memo):
                pass
            else:
                memo.add((name,domain))
                ans=ans+1
        return ans
