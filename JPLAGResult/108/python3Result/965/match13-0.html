class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        e_set = set()
        for email in emails:
            w, domain = email.split('@', 1)
            w = w.split('+', 1)[0]
            w = w.replace('.', '')
            e_set.add(w + '@' + domain)
        # print(e_set)
        return len(e_set)
