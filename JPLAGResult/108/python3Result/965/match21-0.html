class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        email_set = set()
        
        for email in emails:
            a, b = email.split('@')
            a = a.replace(".","")
            ind = a.find("+")
            if ind != -1:
                a = a[:ind]
                
            key = (a, b)
            if key not in email_set:
                email_set.add(key)
        
        return len(email_set)
