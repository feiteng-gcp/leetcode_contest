class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        ans = set()
        for s in emails:
            arr = s.split('@')
            pre, aft = arr[0], arr[1]
            pre_arr = pre.split('+')
            pre = pre_arr[0]
            pre_arr = pre.split('.')
            pre = ''.join(pre_arr)
            ans.add(pre + '@' + aft)
        
        return len(ans)
