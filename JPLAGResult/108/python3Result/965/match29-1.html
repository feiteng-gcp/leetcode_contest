class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        ret = set()
        for item in emails:
            local, domain = item.split('@')
            local = "".join(local.split('.'))
            local = local.split('+')[0]
            ret.add((local, domain))
        return len(ret)
