class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        ans = set()
        for email in emails:
            a, b = email.split('@')
            a = a.split('+')[0]
            a = a.replace('.', '')
            ans.add(a + '@' + b)
        return len(ans)
