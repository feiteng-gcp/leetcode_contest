import re

class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        actual = set()
        for e in emails:
            loc, dom = e.split('@', 1)
            loc, = re.findall('^[^+]*', loc)
            loc = loc.replace('.', '')
            #print(loc + '@' + dom)
            actual.add(loc + '@' + dom)
        return len(actual)
