class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        def process(addr):
            loc, domain 
= addr.split("@")
            loc = loc.split('+')[0]
            loc = ''.join(loc.split('.'))
            return loc+domain

        ans = set()
        for email in emails:
            ans.add(process(email))
        return len(ans)
