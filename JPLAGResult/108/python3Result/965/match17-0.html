class Solution:
    def numUniqueEmails(self, emails):
        """
        :type emails: List[str]
        :rtype: int
        """
        email_set = set()
        
        for email in emails:
            name,domain = email.split('@')
            name = name.split('+')[0]
            name = name.replace('.', '')
            email_set.add((name,domain))
        
        return len(email_set)
