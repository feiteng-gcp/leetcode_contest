class Solution:
    def minFallingPathSum(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        
        a = A
        n = len(a)

        f = [[None for _ in range(n)] for _ in range(n)]
        for i in range(n):
            f[0][i] = a[0][i]

        for i in range(1, n):
            for j in range(n):
                f[i][j] = f[i - 1][j] + a[i][j]
                if j != 0:
                    f[i][j] = min(f[i][j], f
[i - 1][j - 1] + a[i][j])
                if j != n - 1:
                    f[i][j] = min(f[i][j], f[i - 1][j + 1] + a[i][j])
        return min(f[-1])
