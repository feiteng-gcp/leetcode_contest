class Solution:
    def minFallingPathSum(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        M = len(A[0])
        N = len(A)
        results = [[0] * (M)]
        for i in range(N):
            result = [0] * (M)
            for j in range(M):
                result[j] = A[i][j] + results[i][j]
                if j > 0:
                    result[j] = min(result[j], A[i][j] + results[i][j-1])
                if j < M - 1:
                    result[j] = min(result[j], A[i][j] + results[i][j+1])
            results.append(result)
        return min(results[N])
