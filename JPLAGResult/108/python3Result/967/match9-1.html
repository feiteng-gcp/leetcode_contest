class Solution:
    def minFallingPathSum(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        n=len(A)
        dp=[[100000000]*n for i in range(n)]
        for i in range(n):
            dp[0][i]=A[0][i]
        nex={0,-1,1}
        for deep in range(1,n):
            for i in range(n):
                for j in nex:
                    p = j+i
                    if(p>-1 and p<n):
                        dp[deep][i]=min(dp[deep][i],dp[deep-1][p])
                dp[deep][i]=dp[deep][i]+A[deep][
i]
        ans= 1000000
        for i in range(n):
            ans = min(ans,dp[n-1][i])
        #print(dp)
        return ans
