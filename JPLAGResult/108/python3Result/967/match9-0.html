class Solution:
    def minFallingPathSum(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        le = len(A)
        lel = len(A[0])
        for i in range(1,le):
            
            for j in range(lel):
                x = A[i][j]
                A[i][j] = x + A[i-1][j]
                if j-1>=0:
                    A[i][j] = min(A[i][j],x + A[i-1][j-1])
                if j + 1 < lel:
                    A[i][j] = min(A[i][j],x + A[i-1][j+1])
        ans = 1000000000
        for i in range(lel):
            ans = min(ans,A[le-1][i])
        return ans
