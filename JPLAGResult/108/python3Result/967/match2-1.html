class Solution:
    def minFallingPathSum(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        n=len(A)
        for i in range(1,n):
            A[i][0]+=min(A
[i-1][0],A[i-1][1])
            A[i][-1]+=min(A[i-1][-1],A[i-1][-2])
            for j in range(1,n-1): A[i][j]+=min(A[i-1][j-1],A[i-1][j],A[i-1][j+1])
        return min(A[-1])
