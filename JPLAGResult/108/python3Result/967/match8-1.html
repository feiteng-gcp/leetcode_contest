class Solution:
    def minFallingPathSum(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        n, m = len(A), len(A[0])
        
        f = [0] * m
        for i in range(n-1, -1, -1):
            nxt_f = [+float('inf')] * m
            for j in range(m):
                nxt_f[j] = min(nxt_f[j], A[i][j]+f[j])
                if j < m-1:
                    nxt_f[j] = min(nxt_f
[j], A[i][j]+f[j+1])
                if j > 0:    
                    nxt_f[j] = min(nxt_f[j], A[i][j]+f[j-1])
            f = nxt_f
            # print(f)
        return min(f)
