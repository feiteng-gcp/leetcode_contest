class Solution:
    def minFallingPathSum(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        N = len(A)
        if N == 1:
            return A[0][0]


        for i in range(1, N):
            for j in range(N):
                if j == 0:
                    A[i][j] += min(A
[i-1][j], A[i-1][j+1])
                elif j == N - 1:
                    A[i][j] += min(A[i-1][j-1], A[i-1][j])
                else:
                    A[i][j] += min(A[i-1][j-1], A[i-1][j], A[i-1][j+1])

        return min(A[N-1])
