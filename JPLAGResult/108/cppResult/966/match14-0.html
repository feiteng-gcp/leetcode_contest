class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        unordered_map<int, int> preSum;
        preSum[0] = 1;
        int res = 0, curr = 0;
        for(int i = 0; i < A.size(); i++){
            curr += A[i];
            if(preSum.count(curr-S)) res += preSum[curr-S];
            preSum[curr]++;
        }
        return res;
    }
};
