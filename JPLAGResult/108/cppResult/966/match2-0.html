class Solution {
public:
    int numSubarraysWithSum(vector<int>& v, int target) {
        unordered_map<int,int>ct;
        ct[0]=1;
        int sum=0;
        int res=0;
        for(int i=0;i<v.size();i++){
            // cout<<"index: "<<i<<endl;
            sum+=v[i];
            if(ct.count(sum-target)==1){
                res+=ct[sum-target];
            }
            ct[sum]++;
            // cout<<"res: "<<res<<endl;
        }
        return res;
    }
};
