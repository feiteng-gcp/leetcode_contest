class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        vector<int> v;
        int l = -1;
        for(int i = 0; i <= A.size(); i++){
            if(i == A.size() || A[i] == 1){
                v.push_back(i - l - 1);
                l = i;
            }
        }
        int ans = 0;
        if(S == 0){
            for(int i = 0; i < v.size(); i++){
                ans += (v[i] + 1) * v[i] / 2; 
            }
        } else {
            for(int i = 0; i < v.size(); i++){
                if(i + S < v.size()){
                    ans += (v[i] + 1) * (v[i + S] + 1);
                }
            }
        }
        return ans;
    }
};
