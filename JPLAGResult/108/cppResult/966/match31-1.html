class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        unordered_map< int, int > sums;
        sums[ 0 ] = 1;
        int num = 0;
        for( int i = 0, ss = 0; i < A.size(); i++ ) {
            ss += A[ i ];            
            num += sums.count( ss - S ) ? sums[ ss - S ] : 0;
            sums[ ss ]++;
        }
        return num;
    }
};
