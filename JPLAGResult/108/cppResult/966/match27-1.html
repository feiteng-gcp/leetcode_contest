class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {

        int cnt = 0;
        map<int, int> m;
        m[0] = 1;
        int sum = 0;
        for (int i=0; i<A.size(); ++i) {
            sum += A[i];
            if (m.count(sum-S))
                cnt += m[sum-S];
            m[sum] += 1;
        }
        return cnt;
    }
};
