class Solution {
public:
    int numSubarraysWithSum(vector<int>& a, int S) {
        int cnt = 0;
        int pre = 0;
        unordered_map<int, int> m;
        m[0] = 1;
        
        for(int i: a)
        {
            pre += i;
            
            int target = pre-S;
            
            cnt += m[target];
            
            m[pre] ++;
        }
        
        
        
        return cnt;
    }
};
