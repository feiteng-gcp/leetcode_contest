class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        map<int, int> mp;
        mp[0] = 1;
        int ans = 0;
        for (int i = 0, now = 0; i < A.size(); i++) {
            now += A[i];
            ans += mp[now - S];
            mp[now]++;
        }
        return ans;
    }
};
