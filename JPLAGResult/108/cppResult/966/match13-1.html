class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        map<int,int>hax;
        hax[0] = 1;
        int pre = 0;

        int ans = 0;
        for (int i = 0; i < A.size(); i++) {
            pre += A[i];
            ans += hax[pre-S];
            hax[pre]++;
        }      
        return ans;
    }
};
