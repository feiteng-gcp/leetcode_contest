class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        vector<int> sum(A.size()+1,0);
        unordered_map<int,int> mp;
        mp[0]=1;
        int ans=0;
        for(int i=1;i<=A.size();i++){
            sum[i]=A[i-1]+sum[i-1];
            if(mp[sum[i]-S]) ans+=mp[sum[i]-S];
            mp[sum[i]]++;
        }
        
        return ans;
    }
};
