class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        int n = A.size();
        vector<int> sums(n+1, 0);
        for(int i = 0; i < n; i++) sums[i+1] = sums[i] + A[i];

        unordered_map<int, int> mp;
        mp[0] = 1;
        int ans = 0;
        for(int i = 1; i <= n; i++){
            ans += mp[sums[i]-S];
            mp[sums[i]]++;
        }
        return ans;
    }
};
