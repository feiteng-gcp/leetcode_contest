class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        map< int ,int > check ; 
        check[0]++; 
        int ans =0 ; 
        int sum =0 ; 
        for( int i=0 ; i <A.size() ; i++){
            sum += A[i] ; 
            if( check[sum - S ] ) {
                ans += check[sum-S] ;  
            }
            check[sum]++;
        }
        return ans ; 
    }
};
