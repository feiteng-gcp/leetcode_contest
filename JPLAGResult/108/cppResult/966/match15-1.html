class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {

        unordered_map<int, int> map;
        int result = 0;
        int sum = 0;
        map[0] = 1;
        for (int i = 0; i < A.size(); ++i) {
            sum += A[i];
            result += map[sum - S];
            ++map[sum]; 
        }
        return result;
    }
};
