class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        map<int, int> pre_sum;
        int sum = 0;
        int ans = 0;
        ++pre_sum[0];
        for(auto n: A) {
            sum += n;
            ans += pre_sum[sum - S];
            ++pre_sum[sum];
        }
        return ans;
    }
};
