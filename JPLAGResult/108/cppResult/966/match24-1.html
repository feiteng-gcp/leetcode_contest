class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        unordered_map<int, int> mp;
        int ans = 0;
        int sum = 0;
        mp[0] = 1;
        for (auto & val : A) {
            sum += val;
            if (mp.find(sum - S) != mp.end())
                ans += mp[sum - S];
            mp[sum]++;
        }
        return ans;
    }
};
