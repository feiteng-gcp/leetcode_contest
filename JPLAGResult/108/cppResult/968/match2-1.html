class Solution {
public:
    vector<int> dfs(vector<pair<int,int>> vct, bool asc) {
        if (vct.size() == 1) {
            vector<int> res = {vct[0].second};
            return res;
        } else {
            vector<pair<int,int>> lvct;
            vector<pair<int,int>> rvct;
            for (auto p : vct) {
                if (p.first % 2 == 0) {
                    lvct.emplace_back(p.first/2, p.second);
                } else {
                    rvct.emplace_back(p.first/2, p.second);
                }
            }
            
            vector<int> lres = dfs(lvct, !asc);
            vector<int> rres = dfs(rvct, asc);
            vector<int> res;
            if (asc) {
                for (auto v : lres) {
                    res.push_back(v);
                }
                for (auto v : rres) {
                    res.push_back(v);
                }
            } else {
                for (auto v : rres) {
                    res.push_back(v);
                }
                for (auto v : lres) {
                    res.push_back(v);
                }
            }
            return res;

        }
    }
    
    vector<int> beautifulArray(int N) {
        vector<pair<int,int>> vct;
        for (int i = 0; i < N; i++) {
            vct.emplace_back(i+1, i+1);
        }
        return dfs(vct, false);
    }
};
