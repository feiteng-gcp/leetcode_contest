int a[1005];
void work(int l, int r, int n) {
    if (n==1) {
        a[l] = 1;
        return;
    }
    else {
        int m = n>>1;
        work(l, l+m-1, m), work(l+m, r, n-m);
        for (int i = l; i<=l+m-1; ++i) a[i] <<= 1;
        for (int i = l+m; i<=r; ++i) a[i] = a[i]*2-1;
    }
}

class Solution {
public:
    vector<int> beautifulArray(int N) {
        work(0, N-1, N);
        vector<int> A;
        for (int i = 0; i<N; ++i) A.push_back(a[i]);
        return A;
    }
};
