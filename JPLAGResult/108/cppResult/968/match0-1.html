class Solution {
public:
    vector<int> s;
    void go(int n, int al, int ar) {
        assert(ar - al + 1 == n);
        if (n == 1) {
            s[al] = 0;
            return;
        }
        int odd = (n + 1) / 2;

        int even = n - odd;
        
        go(odd, al, al + odd - 1);
        go(even, al + odd, ar);
        for (int i = 0; i < odd; ++i) {
            s[al + i] = s[al + i] * 2;
        }
        for (int i = 0; i < even; ++i) {
            s[al + odd + i] = s[al + odd + i] * 2 + 1;
        }
    }
    
    vector<int> beautifulArray(int n) {
        s.resize(n);
        go(n, 0, n - 1);
        for (int i = 0; i < n; ++i) {
            ++s[i];
        }
        return s;
    }
};
