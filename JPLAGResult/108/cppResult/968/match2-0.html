class Solution {
public:
    vector<int> ans;
    void solve(vector<pair<int, int>>& a) {
        if (a.size() == 1) {
            ans.push_back(a[0].second);
        }
        if (a.size() <= 1) return;
        vector<pair<int, int>> b, c;
        for (int i = 0; i < a.size(); i++) {   
            if (a[i].first & 1) {
                a[i].first /= 2;
                b.push_back(a[i]);
            } else {
                a[i].first /= 2;
                c.push_back(a[i]);
            }
        }
        solve(b), solve(c);
    }
    
    vector<int> beautifulArray(int N) {
        ans.clear();
        vector<pair<int, int>> a;
        for (int i = 0; i < N; i++) {
            a.emplace_back(i + 1, i + 1);
        }
        solve(a);
        return ans;
    }
};
