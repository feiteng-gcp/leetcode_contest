class Solution {
public:
    vector<int> beautifulArray(int N) {
        vector<int> v;
        for(int i = 1; i <= N; i ++) v.push_back(i);
        return dfs(v);
    }
    vector<int> dfs(vector<int> & v) {
        if(v.size() <= 2) return v;
        vector<int> vOdd;
        vector<int> vEven;
        for(int i = 0; i < v.size(); i ++) {
            if(i & 1) vEven.push_back(v[i]);
            else vOdd.push_back(v[i]);
        }
        vector<int> v1 = dfs(vOdd);
        vector<int> v2 = dfs(vEven);
        vector<int> ans;
        for(int i = 0; i < v1.size(); i ++) ans.push_back(v1[i]);
        for(int i = 0; i < v2.size(); i ++) ans.push_back(v2[i]);
        return ans;
    }
};

int verify(vector<int> & ans) {
    int N = ans.size();
    vector<bool> visited(N + 1, false);
    for(int i = 0; i < ans.size(); i ++) {
        printf("%d ", ans[i]);
        visited[ans[i]] = true;
    }
    printf("\n");
    for(int i = 1; i <= N; i ++) {
        if(!visited[i]) printf("ERROR - A\n");
    }
    for(int i = 0; i < N; i ++) {
        for(int j = i + 1; j < N; j ++) {
            for(int k = i + 1; k < j; k ++) {
                if(ans[k] * 2 == ans[i] + ans[j]) printf("ERROR - B\n");
            }
        }
    }
    return 0;
}
