const ll MOD=1e9+7;
class Solution {
public:
  void solve(vector<int>& v,int a,int b){
    int n=b-a;
    if(n<=0) return;
    if(n==1){
        v[a]=1;
        return;
    }
    int c=(a+b)/2;
    solve(v,a,c);
    for(int i=a;i<c;i++){
      v[i]*=2;
    }
    solve(v,c,b);
    for(int i=c;i<b;i++){
      v[i]=v[i]*2-1;
    }
  }
  vector<int> beautifulArray(int N) {
    vector<int> v(N);
    solve(v,0,N);
    return v;
  }
};
