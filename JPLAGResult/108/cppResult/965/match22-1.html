class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> S;
        for (string s : emails) {
            string post;
            for (int i = 0; i < s.size(); i ++) {
                if (s[i] == '@') {
                    post = s.substr(i, s.size() - i);
                    s = s.substr(0, i);
                    break;
                }
            }
            for (int i = 0; i < s.size(); i ++)
                if (s[i] == '+') {
                    s = s.substr(0, i);
                    break;
                }
            string tmp = "";
            for (char c : s) {
                if (c != '.') tmp += c;
            }
            S.insert(tmp + post);
        }
        return S.size();
    }
};
