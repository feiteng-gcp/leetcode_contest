class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> seen;
        for (auto& e: emails) {
            string local = e.substr(0, e.find('@'));
            string domain = e.substr(e.find('@') + 1);
            string newLocal;
            for (auto& c: local) {
                if (c == '.') {
                    continue;
                } else if (c == '+') {
                    break;
                }
                newLocal += c;
            }
            seen.insert(newLocal + "@" + domain);
        }
        return seen.size();
    }
};
