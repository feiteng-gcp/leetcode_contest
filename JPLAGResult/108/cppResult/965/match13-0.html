class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> se;
        for (auto s : emails) {
            string t;
            for (int i = 0; i < s.size(); i++) {
                if (s[i] == '.') {
                    continue;
                } else if (s[i] == '+') {
                    break;
                } else {
                    t += s[i];
                }
            }
            for (int i = 0; i < s.size(); i++) {
                if (s[i] == '@') {
                    t += s.substr(i);
                }
            }
            se.insert(t);
        }
        return se.size();
    }
};
