class Solution {
public:
    int numUniqueEmails(vector<string>& e) {
        set<string>st;
        for(int i=0;i<e.size();i++){
            string s="";
            for(int j=0;j<e[i].size();j++){
                if(e[i][j]=='@')
                    break;
                if(e[i][j]=='.')
                    continue;
                s+=e[i][j];
            }
            string t="";
            for(int j=0;j<s.size();j++){
                if(s[j]=='+')
                    break;
                t+=s[j];
            }
            bool f=false;
            for(int j=0;j<e[i].size();j++){
                if(e[i][j]=='@')
                    f=true;
                if(f)
                    t+=e[i][j];
            }
            //cout<<t<<endl;
            st.insert(t);
        }
        return (int)st.size();
    }
};
