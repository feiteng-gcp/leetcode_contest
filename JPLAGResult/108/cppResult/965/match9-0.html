class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        
        set<string> hash;
        for (auto str: emails){
            string s = "";
            for (int i = 0; i < str.size(); ++i){
                if (str[i] == '+' || str[i] == '@'){
                    int j;
                    for (j = 0 ; j < str.size();  ++j)
                        if (str[j] == '@') break;
                   // cout<<s + str.substr(j, str.size() - j + 1)<<endl;
                    hash.insert(s + str.substr(j, str.size() - j + 1) );
                    break;
                    
                }else if (str[i] == '.'){
                    continue;
                    
                }else{
                    s += str[i];
                    
                }
                
            }
            
            
        }
        return hash.size();
        
    }
};
