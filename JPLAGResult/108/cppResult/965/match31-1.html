class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        int n=emails.size();
        int m;
        set<string> ma;
        string t;
        for(int i=0;i<n;i++){
            m=emails[i].length();
            t="";
            vector<int> aaa(m,0);
            for(int j=0;j<m;j++){
                if(emails[i][j]=='.'){
                    aaa[j]=1;
                }else if(emails[i][j]=='+'){
                    for(int k=j;k<m;k++){
                        if(emails[i][k]=='@') break;
                        aaa[k]=1;
                    }
                }
                if(emails[i][j]=='@') break;
            }

            for(int j=0;j<m;j++){
                if(aaa[j]==0) t+=emails[i][j];
            }
            ma.insert(t);
        }
        return ma.size();
    }
};
