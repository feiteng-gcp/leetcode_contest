class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<pair<string, string>> se;
        for(int i = 0; i < emails.size(); i++) {
            string cur = emails[i];
            string ans1,ans2;
            for(int j = 0; j < cur.length(); j++) {
                if(cur[j] == '@' | cur[j] == '+') break;
                else if(cur[j] <= 'z' && cur[j] >= 'a') {
                    ans1 += cur[j];
                }
            }
            for(int j = cur.length()-1; j >= 0; j--) {
                if(cur[j] == '@') break;
                else ans2 += cur[j];
            }
            se.insert({ans1,ans2});
        }
        return se.size();
    }
};
