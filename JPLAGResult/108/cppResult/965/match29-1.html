class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> s;
        for (int i = 0; i < emails.size(); ++i) {
            string n = emails[i];
            string t = "";
            for (int j = 0; j < n.size(); ++j) {
                if (n[j] == '@') {
                    for (int k = 0; k < j; ++k) {
                        if (n[k] == '+') break;
                        if (n[k] != '.') {
                            t += n[k];
                        }
                    }
                    for (int k = j; k < n.size(); ++k) {
                        t += n[k];
                    }
                    break;
                }
            }
            s.insert(t);
        }
        return s.size();
    }
};
