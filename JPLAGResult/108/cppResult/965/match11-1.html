class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> res;
        for (string s : emails) {
            string t = s.substr(0, s.find('@'));
            string d = s.substr(s.find('@'));
            
            int p = t.find('+');
            if (p != string::npos) {
                t = t.substr(0, p);

            }
            
            string norm;
            for (int i = 0; i < t.size(); ++i) {
                if (t[i] != '.') {
                    norm += t[i];
                }
            }
            
            res.insert(norm + d);
        }
        return res.size();
    }
};
