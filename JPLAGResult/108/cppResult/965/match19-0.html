class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> us;
        for(int i = 0; i < emails.size(); i ++) {
            string & one = emails[i];
            string cur;
            int j = 0;
            for(j = 0; '@' != one[j]; j ++) {
                if('.' == one[j]) continue;
                if('+' == one[j]) {
                    while('@' != one[j]) j ++;
                    break;
                }
            }
            while(j < one.size()) {
                cur += one[j];
                j ++;
            }
            us.insert(cur);
        }
        return us.size();
    }
};
