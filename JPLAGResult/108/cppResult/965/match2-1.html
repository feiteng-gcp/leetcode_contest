class Solution {
public:
    string processEmail(string& email) {
        string res;
        int atIdx = -1;
        for (int i=0; i<email.size(); ++i) {
            if (email[i]=='@') {
                atIdx = i;
                break;
            }
        }
        for (int i=0; i<atIdx; ++i) {
            if (i==0 || (i<email.size()-1 && email[i+1]=='@')) {
                res.push_back(email[i]);
                continue;
            }
            if (email[i]=='.')
                continue;
            else if (email[i]=='+')
                break;
            else
                res.push_back(email[i]);

        }
        res.insert(res.end(), email.begin()+atIdx+1, email.end());
        //cout << atIdx << " " << res << endl;
        return res;
    }
    int numUniqueEmails(vector<string>& emails) {
        set<string> s;
        for (int i=0; i<emails.size(); ++i) {
            s.insert(processEmail(emails[i]));
        }
        return s.size();
    }
};
