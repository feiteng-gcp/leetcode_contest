class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> st;
        for(auto &s:emails)
        {
            st.insert(deal(s));
        }
        return st.size();
    }
    
    string deal(string &s)
    {
        string ans;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='.')continue;
            if(s[i]=='+')
            {
                while(s[i]!='@')i++;
                while(i<s.size())ans+=s[i],i++;
                break;
            }
            ans+=s[i];
        }
        //cout<<ans<<endl;
        return ans;
    }
};
