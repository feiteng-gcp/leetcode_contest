class Solution {
public:
    int numUniqueEmails(vector<string>& emails)
{
    set<string> s;

    for (string&i : emails)
    {
        int at = i.find_first_of('@');
        int j = 0;
        while (j < at)
        {
            if (i[j] == '.')
            {
                i.erase(i.begin() + j);
                at--;
            }
            else if (i[j] == '+')
            {
                i.erase(i.begin() + j, i.begin() + at);
                at = j;
            }
            else
            {
                j++;
            }
        }
        s.insert(i);
    }
    return s.size();
}
};
