class Solution {
public:
    set<string> se;
    int numUniqueEmails(vector<string>& emails) {
        int f=0,ff=0;
        se.clear();
        for(string s:emails)
        {
            f=0,ff=0;
            string ns;
            for(char c:s)
            {
                if(c=='@'||f)
                {
                    f=1;
                    ns+=c;
                    continue;
                }
                if(ff) continue;
                if(c=='.')
                    continue;
                else if(c=='+')
                {
                    ff=1;
                    continue;
                }
                else
                    ns+=c;
            }
            se.insert(ns);
        }
        return se.size();
    }
};
