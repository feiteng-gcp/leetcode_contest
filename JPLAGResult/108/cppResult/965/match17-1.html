class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> cnt;
        for (auto str : emails) {
            string tmp;
            bool flag1= true;
            bool flag = false;
            for (auto c : str) {
                if (flag) tmp.push_back(c);
                else if (c == '@') {
                    flag = true;
                    tmp.push_back(c);
                }
                else if (c == '+') {
                    flag1 = false;
                }
                else if (c != '.' && flag1) {
                    tmp.push_back(c);
                }
            }
            cnt.insert(tmp);
        }
        return cnt.size();
    }
};
