class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        int n = emails.size();
        set<string> ans;
        for (int i = 0;i < n;i++) {
            string s = emails[i];
            int m = 0;
            while(s[m++] != '@');
            string tmp = "";
            for (int j = 0;j < m;j++) {
                if (s[j] == '.') continue;
                if (s[j] == '+') break;
                tmp += s[j];
            }
            for (int j = m;j < s.size();j++) tmp += s[j];
            ans.insert(tmp);
        }
        return ans.size();
    }
};
