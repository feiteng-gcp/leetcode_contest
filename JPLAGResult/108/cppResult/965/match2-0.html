class Solution {
public:
    string judge(string a) {
        int at = 0;
        for (int i = 0;i < a.size();i++) {
            if (a[i] == '@') {
                at = i;
            }
        }
        string s = "";
        for (int i = 0;i < at;i++) {
            if (a[i] == '.') {
                
            } else if (a[i] == '+') {
                break;
            } else {
                s += a[i];
            }
        }
        for (int i = at;i < a.size();i++) {
            s += a[i];
        }
        return s;
    }
    int numUniqueEmails(vector<string>& emails) {
        set<string> myset;
        for (int i = 0;i < emails.size();i++) {
            myset.insert(judge(emails[i]));
        }
        return myset.size();
    }
};
