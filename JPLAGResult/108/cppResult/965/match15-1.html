class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> uniqueEmails;
        for( const auto& email : emails )
        {
            auto pos = email.find( "@" );

            string localName = email.substr( 0, pos );
            string domainName = email.substr( pos );
            string finalLocalName;
            for( int i = 0; i < localName.size(); i++ )
            {
                if( localName[i] == '.' )
                    continue;
                if( localName[i] == '+' )
                    break;
                finalLocalName += localName[i];
            }
            uniqueEmails.insert( finalLocalName + "@" + domainName );
        }
        return uniqueEmails.size();
    }
};
