class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_map<string, int> counter;
        for (auto& e : emails) {
            int idx = e.find("@");
            string host = e.substr(0, idx);
            string domain = e.substr(idx + 1);
            string new_host = "";
            for (char& c : host) {
                if (c != '.') {
                    new_host += c;
                }
                if (c == '+') {
                    break;
                }
            }
            ++counter[new_host + "@" + domain];
        }
        return counter.size();
    }
};
