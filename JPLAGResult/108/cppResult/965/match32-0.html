class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
      set<string> ret;
      for (auto& s : emails) {
        ret.insert(solve(s));
      }
      return ret.size();
    }

    string solve(string email) {
      int pat = email.find('@');
      string local = email.substr(0, pat);
      int pplus = local.find('+');
      if (pplus != string::npos) {
        local = local.substr(0, pplus);
        string x;
        for (int i = 0; i < local.length(); ++i)
          if (local[i] != '.') x += string(1, local[i]);
        local = x;
      }
      return local + email.substr(pat);
    }
};
