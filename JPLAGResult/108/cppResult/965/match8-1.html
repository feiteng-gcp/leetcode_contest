class Solution {
 public:
  int numUniqueEmails(vector<string>& emails) {
    set<string> ans;
    for (string& s : emails) {
      string t;
      bool plus = false, at = false;
      for (char c : s) {
        if (at) {
          t += c;
        } else if (c == '@') {
          at = true;
          t += c;
        } else if (plus) {
        } else if (c == '+') {
          plus = true;
        } else if (c == '.') {
        } else {
          t += c;
        }
      }
      ans.insert(t);
    }
    return ans.size();
  }
};
