class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        int n = emails.size();
        set<string> res;
        for (int i=0; i<emails.size(); i++) {
            string cur = "";
            int j=0;
            while (j<emails[i].size()) {
                if (emails[i][j] == '.') {j++;}
                else if (emails[i][j] == '+' || emails[i][j] == '@') {
                    while (emails[i][j] != '@') j++;
                    cur += emails[i].substr(j, emails[i].size()-j);
                    break;
                }
                else {
                    cur.push_back(emails[i][j++]);
                }
            }
            //cout << cur << endl;
            res.emplace(cur);
        }
        
        return res.size();
    }
};
