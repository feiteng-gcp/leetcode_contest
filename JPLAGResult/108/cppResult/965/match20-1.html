class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> set;
        stringstream ss;
        for (string s : emails)
        {
            ss.str("");
            bool domain = false;
            bool plus = false;            
            for (int i = 0; i < s.size(); i++)
            {
                if (s[i] == '@')
                {
                    domain = true;
                }
                else if (s[i] == '+')
                {
                    plus = true;
                }
                
                if (plus && !domain)
                {
                    continue;
                }
                if (!plus && !domain && s[i] == '.')
                {
                    continue;
                }
                ss << s[i];
            }
            set.insert(ss.str());
        }
        return set.size();
    }
};
