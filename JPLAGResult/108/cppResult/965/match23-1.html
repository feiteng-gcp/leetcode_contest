class Solution {
public:
	int numUniqueEmails(vector<string>& emails) {
		set<string> ss;
		string s;
		for (auto sp : emails) {
			string sa = "", sb = "";
			for (int i = 0; i < sp.length(); i++) {
				if (sp[i] == '@') {
					sb = sp.substr(i), sa = sp.substr(0, i);
					break;
				}
			}
			s = "";
			for (auto ch : sa) {
				if (ch == '.') continue;
				if (ch == '+') break;
				s = s + ch;
			}
			ss.insert(s + sb);
		}
		return ss.size();
	}
};
