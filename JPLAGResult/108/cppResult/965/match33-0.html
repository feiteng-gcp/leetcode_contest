class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
		set<string> st;
		for (auto &email : emails) {
			string tmp = "";
			bool domain = false;
			bool plus = false;
			for (auto ch : email) {
				if (ch == '+') {
					plus = true;
					if (domain) tmp += ch;
				} else if (ch == '.') {
					if (domain) tmp += ch;
				} else if (ch == '@') {
					domain = true;
					tmp += ch;
				} else {
					if (domain) tmp += ch;
					else if (!plus) tmp += ch;
				}
			}
			st.insert(tmp);
		}
		return st.size();
    }
};
