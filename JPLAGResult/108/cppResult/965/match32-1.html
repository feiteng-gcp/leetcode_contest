class Solution {
public:
    bool vis[1000];
    vector<string> b={};
    int numUniqueEmails(vector<string>& s) {
        
        int n=s.size();
        
        memset(vis,0,sizeof(vis));
        for (int i=0;i<n;i++)
        {
            int m=s[i].size();
            int j=0;
            string s1="";
            for (j;j<m;j++)
            {
                    if (s[i][j]=='+') break;
                    if (s[i][j]=='.') continue;
                    if (s[i][j]=='@') break;
                    s1=s1+s[i][j];
            }
            for (j;j<m;j++)
                if (s[i][j]=='@')
                    break;
            j++;
            for (j;j<m;j++)
                s1=s1+s[i][j];
            b.push_back(s1);
        }
        int ans=0;
        for (int i=0;i<n;i++)  
            if (vis[i]==false)

            {
                ans++;
                for (int j=i+1;j<n;j++)
                        if (b[i]==b[j])
                            vis[j]=true;
                vis[i]=true;
            }
        return ans;
    }
};
