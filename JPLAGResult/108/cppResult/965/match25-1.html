class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        map<string, bool> mymap;
        for (int i = 0; i < emails.size(); i++) {
            int j = 0;
            string realEmail;
            bool plusShowed = false;

            char c = emails[i].at(0);
            while(c != '@') {
                // cout << c << " ";
                // if (plusShowed || c == '.') continue;
                // if (c == '+') {
                //     plusShowed = true;
                //     continue;
                // }
                if (c == '+') {
                    plusShowed = true;
                }
                else if (!plusShowed && isalpha(c)) {
                    realEmail += c;
                }
                j++;
                c = emails[i].at(j);
            }
            realEmail += emails[i].substr(j);
            mymap[realEmail] = true;
        }
        return mymap.size();
    }
};
