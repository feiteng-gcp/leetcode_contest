class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> count;
        for(int i = 0; i < emails.size(); i++)
        {
            count.insert(refineAddress(emails[i]));
        }
        return count.size();
    }
    
    string refineAddress(const string & s)
    {
        string localName, domainName;
        int atPos = s.find('@');
        localName = s.substr(0, atPos - 0);
        domainName = s.substr(atPos, s.size() - atPos);
        return refine(localName) + domainName;
    }
    
    string refine(const string & s)
    {
        string refine;
        for(char ch : s)
        {
            if(ch == '.') continue;
            if(ch == '+') break;
            refine.push_back(ch);
        }
        return refine;
    }
};
