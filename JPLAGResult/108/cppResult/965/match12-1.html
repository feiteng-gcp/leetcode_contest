class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> s;
        for(auto& e: emails) {
            bool local = true;
            int pos = e.find('@');
            string tmp = "";

            for(int i = 0; i < pos; ++i) {
                if(e[i] == '.') {
                    continue;
                } else if(e[i] == '+') {
                    break;
                }
                tmp += e[i];
            }
            tmp += e.substr(pos);
            s.insert(tmp);
        }
        return s.size();
    }
};
