class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> f;

        for (string &s : emails) {
            bool b = false, p = false;
            string t;
            for (char &c : s) {
                if (c == '@') {
                    t += c;
                    b = true;
                    continue;
                }
                if (c == '.' && !b)
                    continue;
                if (c == '+')
                    p = true;
                if (p && !b)
                    continue;
                t += c;
            }
            f.insert(t);
        }
        return f.size();
    }
};
