class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> e;
        
        for (auto &s: emails) {
            string tmp;
            size_t pos = s.find('@');
            for (int i = 0; i < pos; ++i) {
                if (s[i] == '.') continue;
                else if (s[i] == '+') break;
                else tmp += s[i];
            }
            tmp += s.substr(pos);
            e.insert(tmp);
        }
        
        return e.size();
    }
};
