class Solution {
public:
	int numUniqueEmails(vector<string>& emails) {
		set<string> s;
		for (string str : emails) {
			string new_str = "";
			bool use = true;
			for (int i = 0; i < str.length();i++) {
				char c = str[i];
				if (c == '@') {
					new_str += str.substr(i);
					break;
				}
				if (use) {
					if (c == '.')
						continue;
					if (c == '+') {
						use = false;
						continue;
					}
					new_str += c;
				}
			}
			cout << new_str << endl;
			s.insert(new_str);
		}
		return s.size();
	}
};
