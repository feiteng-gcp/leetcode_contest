class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> unique;

        for (int i = 0; i < emails.size(); ++i)
        {
        	string cur = emails[i];
        	int pos = cur.find('@');

        	string local = cur.substr(0, pos);
        	string domin = cur.substr(pos+1);

        	string addr = shrink(local) + '@' + domin;

        	unique.insert(addr);
        }

        return unique.size();
    }

    string shrink(string s) {
    	string res;
    	for (int i = 0; i < s.length(); ++i)
    		if (s[i] == '+')
    			return res;
    		else
    			if (s[i] != '.')
    				res += s[i];

    	return res;
    }
};
