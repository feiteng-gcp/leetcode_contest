class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> addr;
        for(auto em : emails) {
            int apos = 0;
            while(em[apos] != '@')
                ++apos;
            int ppos = 0;
            while(em[ppos] != '+' && em[ppos] != '@')
                ++ppos;
            string loc;
            for(int i = 0; i < ppos; ++i) {
                if(em[i] != '.')
                    loc += em[i];
            }
            
            addr.insert(loc + em.substr(apos));
        }
        return addr.size();
    }
};
