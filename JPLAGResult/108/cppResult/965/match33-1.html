class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> st;
        for(auto &e: emails) {
            size_t p1 = e.find('+');
            string ss;
            bool y = true;
            bool z = false;
            for(size_t i = 0 ; i < e.length() ; i++) {
                if(!z && e[i] == '.') continue;
                if(e[i] == '+') y = false;
                if(e[i] == '@') y = z = true;
                if(y) ss += e[i];
            }
            st.insert(ss);
        }
        return (int)st.size();
    }
};
