class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        map<string,bool> m;
        int cnt = 0;
        for(int i = 0;i < emails.size();i++) {
            string email = emails[i];
            string format;
            bool ignore = false;
            for(int j = 0;j < email.size();j++) {
                if(email[j] == '@') {
                    format += email.substr(j);
                    break;
                }else if(email[j] == '.') {
                    continue;
                }else if(email[j] == '+') {
                    ignore = true;
                }else if(!ignore) {
                    format.push_back(email[j]);
                }
            }
            if(m.find(format) == m.end()) {
                m[format] = true;
                cnt++;
            }
        }
        return cnt;
    }
};
