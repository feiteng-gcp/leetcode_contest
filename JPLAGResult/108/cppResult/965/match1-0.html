class Solution {
public:
    void preprocess( vector< string > & emails ){
        for ( auto & s : emails ) {
            
            string temp ;
            bool add = false; 
            bool before = true ; 
            for ( auto & c : s ){
                if(  c == '.' && before ) {
                    ;
                }
                else if( c == '+' ){
                    add = true ;
                }
                else if( !add ){
                    temp.push_back( c )  ; 
                }
                else if( c == '@' ){
                    add=  false;  
                    before = false ; 
                }
            }
            s = temp ; 
        }
    }
    int numUniqueEmails(vector<string>& emails) {
        preprocess( emails ) ;
        set< string > check ; 
        for ( auto & s : emails )
            check.insert( s ) ;
        return (int)check.size() ;
    }
};
