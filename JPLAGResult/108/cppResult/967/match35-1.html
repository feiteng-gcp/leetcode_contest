template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
  int minFallingPathSum(vector<vector<int>>& a) {
    int N=a.size();
    int dp[111][111];
    memset(dp,0x3f,sizeof dp);

    for(int j=0;j<N;j++){
      dp[0][j]=a[0][j];
    }
    for(int i=1;i<N;i++){
      for(int j=0;j<N;j++){
        dp[i][j]=min(dp[i][j],a[i][j]+dp[i-1][j]);
        if(j>=1) dp[i][j]=min(dp[i][j],a[i][j]+dp[i-1][j-1]);
        if(j<=N-2) dp[i][j]=min(dp[i][j],a[i][j]+dp[i-1][j+1]);

      }
    }
    int res=1e9;
    for(int j=0;j<N;j++) res=min(res,dp[N-1][j]);
    return res;
  }
};
