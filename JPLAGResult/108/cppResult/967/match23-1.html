class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        vector< vector< int > > min_sum( A );

        for( int i = 1; i < A.size(); i++ ) {
            for( int j = 0; j < A[ 0 ].size(); j++ ) {
                int m = min_sum[ i - 1 ][ j ];                
                if( j > 0 ) m = min( m, min_sum[ i - 1 ][ j - 1 ] );
                if( j + 1 != A[ 0 ].size() ) m = min( m, min_sum[ i - 1 ][ j + 1 ] );
                min_sum[ i ][ j ] += m;
            }
        }
        return *min_element( min_sum.back().begin(), min_sum.back().end() );
    }
};
