class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& a) {

        int n = a.size();
        vector<vector<int>> dp(n, vector<int>(n, 0x3F3F3F3F));
        for (int i = 0; i < n; ++i) {
            dp[0][i] = a[0][i];
        }
        for (int i = 0; i < n - 1; ++i) {
            for (int j = 0; j < n; ++j) {
                for (int k = -1; k <= 1; ++k) {
                    int nj = j + k;
                    if (nj < 0 || nj >= n) continue;

                    dp[i + 1][nj] = min(dp[i + 1][nj], dp[i][j] + a[i + 1][nj]);
                }
            }
        }
        int ans = 0x3F3F3F3F;
        for (int i = 0; i < n; ++i) {
            ans = min(ans, dp[n - 1][i]);
        }
        return ans;
    }
};
