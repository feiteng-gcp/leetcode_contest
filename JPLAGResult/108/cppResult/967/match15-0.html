class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& a) {
        int dp[105][105]={0};
        int n=(int)a.size();
        for(int i=0;i<n;i++){
            dp[0][i]=a[0][i];
        }
        for(int i=1;i<n;i++){
            for(int j=0;j<n;j++){
                dp[i][j]=dp[i-1][j]+a[i][j];
                if(j&&dp[i-1][j-1]+a[i][j]<dp[i][j])
                    dp[i][j]=dp[i-1][j-1]+a[i][j];
                if(j+1<n&&dp[i-1][j+1]+a[i][j]<dp[i][j])
                    dp[i][j]=dp[i-1][j+1]+a[i][j];
            }
        }
        int mx=1e9;
        for(int i=0;i<n;i++)
            mx=min(mx,dp[n-1][i]);
        return mx;
    }
};
