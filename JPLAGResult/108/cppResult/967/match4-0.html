class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& v) {
        int len=v.size();
        vector<vector<int>>dp(v.size(), vector<int>(v[0].size(),0));
        for(int j=0;j<v[0].size();j++){
            dp[len-1][j]=v[len-1][j];
        }
        for(int i=len-2;i>=0;i--){
            for(int j=0;j<v[0].size();j++){
                dp[i][j]=v[i][j]+dp[i+1][j];
                if(j-1>=0) dp[i][j]=min(dp[i][j], v[i][j]+dp[i+1][j-1]);
                if(j+1<v[0].size()) dp[i][j]=min(dp[i][j], v[i][j]+dp[i+1][j+1]);
            }
        }
        int res=INT_MAX;
        for(int j=0;j<v[0].size();j++){
            res=min(res, dp[0][j]);
        }
        return res;
    }
};
