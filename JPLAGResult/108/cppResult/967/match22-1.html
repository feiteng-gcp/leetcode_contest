class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int n = A.size();
        vector<vector<int>> dp(n, vector<int>(n));
        for (int c = 0; c < n; ++c) {
            dp[n-1][c] = A[n-1][c];
        }
        for (int i = n-2; i >= 0; --i) {
            for (int j = 0; j < n; ++j) {
                dp[i][j] = dp[i+1][j];
                if (j-1 >= 0) {
                    dp[i][j] = min(dp[i][j], dp[i+1][j-1]);
                }
                if (j + 1 < n) {
                    dp[i][j] = min(dp[i][j], dp[i+1][j+1]);
                }
                dp[i][j] += A[i][j];
            }
        }
        
        return *min_element(dp[0].begin(), dp[0].end());
        
        //dp[i][j] = min(dp[i+1][j], dp[i+1][j-1], dp[i+1][j+1]) + A[i][j];
    }
};
