class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int n = A.size();
        for (int i=1; i<n; i++) {
            for (int j=0; j<n; j++) {
                int cur = INT_MAX;
                if (j-1>=0) cur = A[i-1][j-1];
                if (j+1<n) cur = min(A[i-1][j+1], cur);
                cur = min(A[i-1][j], cur);
                A[i][j] += cur;
            }
        }
        
        int res =INT_MAX;
        for (int i=0; i<n; i++) {
            res = min(res, A[n-1][i]);
        }
        return res;
    }
};
