class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int size = A.size();
        vector<vector<int>> dp(size, vector<int>(size, INT_MAX));
        for (int i = 0; i < size; ++i) {
            dp[0][i] = A[0][i];
        }
        for (int i = 1; i < size; ++i) {
            for (int j = 0; j < size; ++j) {
                if (j > 0 && dp[i][j] > dp[i - 1][j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1];
                }
                if (dp[i][j] > dp[i - 1][j]) {
                    dp[i][j] = dp[i - 1][j];
                }
                if (j < size - 1 && dp[i][j] > dp[i - 1][j + 1]) {
                    dp[i][j] = dp[i - 1][j + 1];
                }
                dp[i][j] += A[i][j];
            }
        }
        int res = INT_MAX;
        for (int j = 0; j < size; ++j) {
            res = min(res, dp[size - 1][j]);
        }
        return res;
    }
};
