class Solution {
public:
    int f[200][200];
    int minFallingPathSum(vector<vector<int>>& a) {
        int n=a[0].size();
        memset(f,63,sizeof(f));

        for (int i=0;i<n;i++) f[0][i]=a[0][i];
        for (int i=1;i<n;i++)
        {
            for (int j=0;j<n;j++)
            {

                if (j-1>=0) f[i][j]=min(f[i][j],f[i-1][j-1]+a[i][j]);
                if (j+1<n) f[i][j]=min(f[i][j],f[i-1][j+1]+a[i][j]);
                f[i][j]=min(f[i][j],f[i-1][j]+a[i][j]);
            }
        }
        
        int ans=214444444;
        for (int i=0;i<n;i++)
            ans=min(ans,f[n-1][i]);
        return ans;
    }
};
