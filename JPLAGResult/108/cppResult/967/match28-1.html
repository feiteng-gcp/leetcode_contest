class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int n = A.size(), m = A[0].size();
        vector<int> f = A[0], g(m, 0);

        for (int i = 1; i < n; i++) {
            swap(f, g);
            for (int j = 0; j < m; j++) {
                f[j] = g[j];
                if (j > 0) {
                    f[j] = min(f[j], g[j - 1]);
                }
                if (j < m - 1) {
                    f[j] = min(f[j], g[j + 1]);
                }
                f[j] += A[i][j];

            }
        }
        int x = f[0];
        for (int i = 1; i < m; i++) {
            x = min(f[i], x);
        }
        return x;
    }
};
