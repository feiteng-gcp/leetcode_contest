class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        if(A.empty()){
            return 0;
        }
        vector<vector<int>> dp(A.size(), vector<int>(A[0].size()+2, INT_MAX));
        for(int i = A[0].size(); i >= 1; --i){
            dp.back()[i] = A.back()[i-1];
        }
        for(int i = A.size()-2; i >= 0; --i){
            for(int j = 1; j < dp[i].size()-1; ++j){
                dp[i][j] = A[i][j-1] + min(dp[i+1][j-1], min(dp[i+1][j], dp[i+1][j+1]));
            }
        }
        int ans = INT_MAX;
        for(int i = 0; i < dp[0].size(); ++i){
            //cout<<dp[1][i]<<endl;
            ans = min(ans, dp[0][i]);
        }
        return ans;
    }
};
