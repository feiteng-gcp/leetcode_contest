class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {

        vector<vector<int>> fall(A.size(), vector<int>(A.size(), 0));
        for (int i = 0; i < A.size(); ++i) {
            fall[A.size() - 1][i] = A[A.size() - 1][i];
         //   printf("%d %d %d\n", A.size() - 1, i, fall[A.size() - 1][i]);
        }
        
        for (int i = A.size() - 2; i >= 0; --i) {
            for (int j = 0; j < A.size(); ++j) {
                fall[i][j] = fall[i + 1][j];
                if (j > 0) fall[i][j] = min(fall[i][j], fall[i + 1][j - 1]);
                if (j < A.size() - 1) fall[i][j] = min(fall[i][j], fall[i + 1][j + 1]);
                fall[i][j] += A[i][j];
             //   printf("%d %d %d\n", i, j, fall[i][j]);

            }
            
        }
        
        int minS = fall[0][0];
        
        for (int i = 0; i < A.size(); ++i)
            minS = min(minS, fall[0][i]);
    
        return minS;
    }
};
