class Solution {
    int n;
    int zyd[105][105];
public:
    int minFallingPathSum(vector<vector<int>>& A) {

        n = A.size();
        for (int i = 1; i <= n; ++i) 
            for (int j = 1; j <= n; ++j) {
                zyd[i][j] = zyd[i - 1][j];
                if (j > 1)
                    zyd[i][j] = min(zyd[i][j], zyd[i - 1][j - 1]);
                if (j < n)
                    zyd[i][j] = min(zyd[i][j], zyd[i - 1][j + 1]);
                zyd[i][j] += A[i - 1][j - 1];
            }
        int ans = 100 * 100 * 100;
        for (int j = 1; j <= n; ++j)
            ans = min(ans, zyd[n][j]);
        return ans;
    }
};
