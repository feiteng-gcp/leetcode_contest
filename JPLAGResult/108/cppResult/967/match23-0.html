class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        if(A.empty() || A[0].empty()) return 0;
        int n = A.size(), m = A[0].size();
        
        for(int i=n-2;i>=0;i--) {
            for(int j=0;j<m;j++) {
                int mn = A[i+1][j];
                if(j > 0) mn = min(mn, A[i+1][j-1]);
                if(j < m - 1) mn = min(mn, A[i+1][j+1]);
                A[i][j] += mn;
            }
        }
        
        return *min_element(A[0].begin(), A[0].end());
    }
};
