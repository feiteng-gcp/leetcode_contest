class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        const int m = A.size();
        const int n = m == 0? 0 : A[0].size();
        if(m == 0 || n == 0) return 0;
        
        vector<vector<int>> dp(A);
        
        int dx[2] = {-1, 1};
        

        for(int i = 1; i < m; i ++)
        {
            for(int j = 0; j < n; j ++)
            {
                int val = dp[i-1][j];
                for(int d = 0; d < 2; d ++)
                {
                    int jj = j + dx[d];

                    if(jj >= 0 && jj < n)
                    {
                        val = min(val, dp[i-1][jj]);
                    }
                }
                
                dp[i][j] += val;
            }
        }
        
        int res = INT_MAX;
        for(int j = 0; j < n; j ++)
        {
            res = min(res, dp[m-1][j]);
        }
        
        return res;
    }
};
