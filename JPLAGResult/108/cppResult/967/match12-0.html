class Solution {
 public:
  int minFallingPathSum(vector<vector<int>>& A) {
    int n = A.size(), m = A[0].size();
    vector<vector<int>> dp(n, vector<int>(m, 1e9));
    for (int i = 0; i < m; ++i) {
      dp[0][i] = A[0][i];
    }
    for (int i = 1; i < n; ++i) {
      for (int j = 0; j < m; ++j) {
        for (int k = -1; k <= 1; ++k) {
          int jj = j + k;
          if (jj < 0 || jj >= m) {
            continue;
          }
          dp[i][j] = min(dp[i][j], dp[i - 1][jj] + A[i][j]);
        }
      }
    }
    int ans = 1e9;
    for (int i = 0; i < m; ++i) {
      ans = min(ans, dp[n - 1][i]);
    }
    return ans;
  }
};
