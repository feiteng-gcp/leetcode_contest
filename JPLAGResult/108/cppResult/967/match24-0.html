class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                int v = 1e+9;
                for (int d = -1; d <= 1; ++d) {
                    if (0 <= j + d && j + d < m) {
                        v = min(v, a[i - 1][j + d]);
                    }
                }
                a[i][j] += v;
            }
        }
        int ans = 1e+9;
        for (int i = 0; i < m; ++i) {
            ans = min(ans, a[n - 1][i]);
        }
        return ans;
    }
};
