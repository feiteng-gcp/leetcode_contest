class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int n = static_cast<int>(A.size());
        
        vector<vector<int>> f(n + 5, vector<int>(n + 5, 0));
        
        for (int i = 0 ; i < n; ++i) f[0][i] = A[0][i];
        
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                f[i][j] = f[i - 1][j];
                if (j > 0) f[i][j] = min(f[i][j], f[i - 1][j - 1]);
                if (j < n - 1) f[i][j] = min(f[i][j], f[i - 1][j + 1]);
                f[i][j] += A[i][j];
            }
        }
        
        int ans = f[n - 1][0];
        for (int i = 0; i < n; ++i) ans = min(ans, f[n - 1][i]);
        
        return ans;
    }
};
