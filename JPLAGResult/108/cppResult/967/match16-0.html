class Solution {
public:
    int dp[110][110];
    int minFallingPathSum(vector<vector<int>>& A) {
      memset(dp, 0x3f, sizeof(dp));
      int n = A.size();
      for (int i = 1; i <= n; ++i) {
        dp[n][i] = A[n-1][i-1];
      }
      for (int row = n-1; row; --row) {
        for (int col = 1; col <= n; ++col) {
          dp[row][col] = min(dp[row+1][col-1], min(dp[row+1][col], dp[row+1][col+1])) + A[row-1][col-1];
        }
      }
        /*
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; ++j) {
                printf("%d ", dp[i][j]);
            }
            printf("\n");
        }
        */
      int ret = 0x3f3f3f3f;
      for (int i = 1; i <= n; ++i) ret = min(ret, dp[1][i]);
      return ret;
    }
};
