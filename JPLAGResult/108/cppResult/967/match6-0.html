class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int n = A.size();
        vector<vector<int>> dp(n, vector<int>(n));
        for (int i = 0; i < n; i++) {
            dp[0][i] = A[0][i];
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < n; j++) {
                dp[i][j] = min(j == 0 ? INT_MAX : dp[i-1][j-1], min(j == n-1 ? INT_MAX : dp[i-1][j+1], dp[i-1][j])) + 
                    A[i][j];
            }
        }
        return *min_element(dp[n-1].begin(), dp[n-1].end());
    }
};
