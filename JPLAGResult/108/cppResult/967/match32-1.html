class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int n = A.size();
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX));
        for (int j = 0; j < n; j++)
        {
            dp[0][j] = A[0][j];
        }
        for (int i = 1; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                for (int k = j - 1; k <= j + 1; k++)
                {
                    if (k >= 0 && k < n && dp[i - 1][k] + A[i][j] < dp[i][j])
                    {
                        dp[i][j] = dp[i - 1][k] + A[i][j];
                    }
                }
            }
        }
        int min = INT_MAX;
        for (int j = 0; j < n; j++)
        {
            if (dp[n - 1][j] < min)
            {
                min = dp[n - 1][j];
            }
        }
        return min;
    }
};
