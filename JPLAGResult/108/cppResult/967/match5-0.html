class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        vector<vector<int>> f(n, vector<int>(m));
        for (int i = 0; i < m; i++) {
            f[0][i] = a[0][i];
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                f[i][j] = f[i - 1][j] + a[i][j];
                for (int k = max(0, j - 1); k <= min(m - 1, j + 1); k++) {
                    f[i][j] = min(f[i][j], f[i - 1][k] + a[i][j]);
                }
            }
        }
        int ans = f[n - 1][0];
        for (int i = 0; i < m; i++) {
            ans = min(ans, f[n - 1][i]);
        }
        return ans;
    }
};
