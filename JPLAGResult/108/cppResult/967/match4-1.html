class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& A) {
        int n = A.size();
        vector<vector<int>> f(n);

        for (int i = 0; i < n; i++) f[i] = vector<int>(n);
        
        
        for (int j = 0; j < n; j++) {
            f[0][j] = A[0][j];
        }
        
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < n; j++) {
                f[i][j] = f[i-1][j];
                if (j > 0)
                    f[i][j] = min(f[i][j], f[i-1][j-1]);
                
                if (j < n - 1)
                    f[i][j] = min(f[i][j], f[i-1][j+1]);
                
                f[i][j] += A[i][j];

            }
        }
        
        int res = 100000000;
        for (int j = 0; j < n; j++) {
            res = min(res, f[n - 1][j]);
        }
        return res;
    }
};
