class Solution {
public:
    map<int, vector<int>> cnt;
    vector<vector<int>> ans;
    
    vector<vector<int>> groupThePeople(vector<int>& g) {
        for (int i = 0; i < g.size(); ++ i )
        {
            cnt[g[i]].push_back(i);
            if (cnt[g[i]].size() == g[i])
            {
                ans.push_back(cnt[g[i]]);
                cnt[g[i]].clear();
            }
        }
        
        return ans;
    }
};
