class Solution {
public:
    vector<vector<int>> ret;
    map<int,vector<int>> m;
    #define pb push_back
    vector<vector<int>> groupThePeople(vector<int>& v) {
        for(int i=0; i<v.size(); i++){
            m[v[i]].pb(i);
            if(m[v[i]].size() == v[i]){
                ret.pb(m[v[i]]);
                m[v[i]].clear();
            }
        }
        return ret;
    }
};
