class Solution {
public:
    unordered_map<int,vector<int>> rec; 
    unordered_set<int> vis;
    int n = 0;
    unordered_map<int,int> cnt; 
    vector<vector<int>> groupThePeople(vector<int>& a) {
       unordered_map<int,int> t;
        for(auto i : a){
            t[i]++;
        }
        int j = 0;
        for(auto i : t) {
            int k = i.second / i.first;
            for(int l =0;l < k;l++) cnt[j++] = i.first;
            n += k;
        }
        dfs(a,0);
        vector<vector<int>> res;
        for(auto i  :rec) res.push_back(i.second);
        return res;
    }
    bool dfs(vector<int>& a,int i){
        if(i == a.size()){
            
            return true;
        }
        for(int j = 0;j < n;j++){
            bool f = false;
            if(a[i] == cnt[j] && rec[j].size() < cnt[j]) {
                rec[j].push_back(i);
                f =true;
            }
            if(f&& dfs(a,i + 1)) return true;
            if(f) rec[j].pop_back();
        }
        return false;
    }
};
