class Solution {
public:
    vector<vector<int>> groupThePeople(vector<int>& groupSizes) {
        unordered_map<int, vector<int>> umap;
        for (int ii = 0; ii < groupSizes.size(); ii++) {
            umap[groupSizes[ii]].push_back(ii);
        }
        vector<vector<int>> result;
        for (auto& item: umap) {
            int total = item.second.size();
            int nn = item.first;
            for (int ii = 0; ii < total / nn; ii++) {
                result.push_back(vector<int>(item.second.begin() + ii * nn, item.second.begin() + (ii+1) * nn));
            }
        }
        return result;
    }
};
