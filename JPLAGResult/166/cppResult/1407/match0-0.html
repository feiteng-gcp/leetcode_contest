class Solution {
public:
	vector<vector<int>> ans;
	vector<int> t[1005];
    vector<vector<int>> groupThePeople(vector<int>& g) {
        int n=g.size();
        for(int i=0;i<n;i++){
        	t[g[i]].push_back(i);
        	if(t[g[i]].size()==g[i]){
        		ans.push_back(t[g[i]]);
        		t[g[i]].clear();
			}
		}
		return ans;
    }
};
