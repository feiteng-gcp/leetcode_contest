#include<map>
#include<vector>
using namespace std;
class Solution {
public:
    vector<vector<int>> groupThePeople(vector<int>& groupSizes) {
        map<int, vector<int>> m_group;
        vector<vector<int>> ans;
        for (int i = 0; i < groupSizes.size(); ++i)
        {
            m_group[groupSizes[i]].push_back(i);
            if (m_group[groupSizes[i]].size() == groupSizes[i])
            {
                ans.push_back(m_group[groupSizes[i]]);
                m_group[groupSizes[i]].clear();
            }
        }
        return ans;
    }
};
