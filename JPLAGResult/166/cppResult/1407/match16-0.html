class Solution {
public:
    vector<vector<int>> groupThePeople(vector<int>& groupSizes) {
        vector<int> gs[501];
        int n = groupSizes.size();
        for(int i=0; i<n; i++){
            int g = groupSizes[i];
            gs[g].push_back(i);
        }
        
        vector<vector<int>> gg;
        for(int i=1; i<=n; i++){
            vector<int> g;
            if(gs[i].size() <= 0) continue;
            for(int j=0; j<=gs[i].size(); j++){
                if(j % i == 0 && j>0){
                    gg.push_back(g);
                    g = vector<int>(0);
                   // cout<<i<<" "<<gs[i].size()<<endl;
                }
                if(j == gs[i].size()) break;
                g.push_back(gs[i][j]);
            }
        }
        
        return gg;
    }
};
