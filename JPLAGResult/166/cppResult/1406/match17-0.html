class Solution {
public:
    int subtractProductAndSum(int n) {
        int a, b;
        a = 1;
        b = 0;
        while (n) {
            a *= n % 10;
            b += n % 10;
            n = n / 10;
        }
        // cout << a  << " " << b << endl;
        return a - b;
    }
};
