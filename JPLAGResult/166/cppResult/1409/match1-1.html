int ds[4][2] = {{1,0}, {-1, 0}, {0, 1}, {0, -1}};
class Solution {
public:
    int m, n;
    map<vector<vector<int>>, int> dp;
    vector<vector<int>> s0;
    
    
    void flip(vector<vector<int>>& vs, int i, int j){
        for(int d=0; d<4; d++){
            int ii = i + ds[d][0], jj = j + ds[d][1];
            if(ii >=0 && ii<m && jj>=0 && jj<n){
                vs[ii][jj] ^= 1;
            }
        }
        vs[i][j] ^= 1;
    }

    
    int minFlips(vector<vector<int>>& mat) {
        dp.clear();
        m = mat.size(); n = mat[0].size();
        s0 = vector<vector<int>>(m, vector<int>(n, 0));
        queue<vector<vector<int>>> q;
        q.push(mat);
        
        int s = 0;
        while(q.size() > 0){
            int lq = q.size();
            for(int i=0; i<lq; i++){
                vector<vector<int>> s1 = q.front(); q.pop();
                if(s1 == s0) return s;
                for(int i=0; i<m; i++){
                    for(int j=0; j<n; j++){
                        flip(s1, i, j);
                        if(dp.find(s1) == dp.end()){
                            dp[s1] = 0;
                            q.push(s1);
                        }
                        flip(s1, i, j);
                    }
                }
            }
            s++;
        }
        
        return -1;
    }
};
