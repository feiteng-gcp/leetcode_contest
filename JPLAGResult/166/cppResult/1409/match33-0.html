class Solution {
public:
    inline int conv(vector<vector<int>>& mat) {
        int ret = 0;
        int n = mat.size();
        int m = mat[0].size();
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                ret = (ret << 1) + mat[i][j];
            }
        }
        return ret;
    }
    inline void dev(vector<vector<int>>* mat, int s) {
        int n = mat->size();
        int m = (*mat)[0].size();
        for (int i = n-1; i >= 0; --i) {
            for (int j = m-1; j >= 0; --j) {
                (*mat)[i][j] = s % 2;
                s /= 2;
            }
        }
    }
    inline int flip(vector<vector<int>> mat, int i, int j, int n, int m) {
        int dir[4][2] = {{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
        mat[i][j] = (1 - mat[i][j]);
        for (int k = 0; k < 4; ++k) {
            int ni = i +dir[k][0];
            int nj = j + dir[k][1];
            if (ni >= 0 && ni <n && nj >=0 && nj <m) {
                mat[ni][nj] = 1 - mat[ni][nj];
            }
        }
        return conv(mat);
    }
    int minFlips(vector<vector<int>>& mat) {
        int s = conv(mat);
        int n = mat.size();
        int m = mat[0].size();
        queue<int> q;
        //cout << "start=" << s << endl;
        q.push(s);
        std::unordered_map<int, bool> vis;
        std::unordered_map<int, int> dist;
        dist[s] = 0;
        vis[s] = true;
        
        while (!q.empty()) {
            int top = q.front();
            q.pop();
            if (top == 0) return dist[top];
            dev(&mat, top);
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    int next = flip(mat, i, j, n, m);
                    if (!vis[next]) {
                        vis[next] = true;
                        dist[next] = dist[top] + 1;
                        //cout << "ready to push " << next << endl;
                        q.push(next);
                    }
                }
            }
        }
        return -1;
            
        
    }
};
