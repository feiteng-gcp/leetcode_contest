class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        map<int,pair<int,int> > mmp;
        int n=mat.size();
        int m=mat[0].size();
        int all=n*m;
        int cnt=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                mmp[++cnt]={i,j};
            }
        }
        vector<int> ord;
        int ans=1000;
        for(int i=1;i<=cnt;i++) ord.push_back(i);
        int x=(int)ord.size();
        int dir[4][2]={{-1,0},{1,0},{0,1},{0,-1}};
        do{
            int now[3][3];
            for(int i=0;i<n;i++){
                for(int j=0;j<m;j++){
                    now[i][j]=mat[i][j];
                }
            }
            int steps=0;
            for(int x:ord){
                int dx=mmp[x].first;
                int dy=mmp[x].second;
                bool flag=0;
                if(now[dx][dy]==0){
                    for(int i=0;i<4;i++){
                        int nx=dx+dir[i][0];
                        int ny=dy+dir[i][1];
                        if(nx>=0&&nx<n&&ny>=0&&ny<m&&now[nx][ny]){
                            flag=1;break;
                        }
                    }
                }
                if(now[dx][dy]==1||flag){
                    now[dx][dy]=!now[dx][dy];
                    ++steps;
                    for(int i=0;i<4;i++){
                        int nx=dx+dir[i][0];
                        int ny=dy+dir[i][1];
                        if(nx>=0&&nx<n&&ny>=0&&ny<m){
                            now[nx][ny]=!now[nx][ny];
                        }
                    }
                }
            }
            bool jud=1;
            for(int i=0;i<n;i++){
                for(int j=0;j<m;j++){
                    if(now[i][j]) jud=0;
                }
            }
            if(jud) ans=min(ans,steps);
        }while(next_permutation(ord.begin(),ord.end()));
        return (ans==1000?-1:ans);
    }
};
