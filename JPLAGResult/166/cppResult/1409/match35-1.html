class Solution {
    
    int getNum(vector<vector<int>>& mat, int m, int n) {

        int ret = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                ret = ret * 10 + mat[i][j];
            }
        }
        return ret;
    }
    
    vector<vector<int>> getMat(int num, int m, int n) {
        vector<vector<int>> mat(m, vector<int>(n, 0));
        for (int i = m - 1; i >= 0 && num; --i) {
            for (int j = n - 1; j >= 0 && num; --j) {
                mat[i][j] = num % 10;
                num /= 10;
            }
        }
        return mat;
    }
    
public:
    int minFlips(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        
        int goal = getNum(mat, m, n);
        unordered_set<int> visited;
        visited.insert(0);
        queue<int> q;
        q.push(0);
        int levelSize = 1;
        int step = 0;

        vector<pair<int, int>> cord = {{0,1},{1,0},{0,-1},{-1,0}};
        
        while (!q.empty()) {
            int cur = q.front();
            if (cur == goal) break;
            
            q.pop();
            --levelSize;
            vector<vector<int>> curMat = getMat(cur, m, n);
            

            for (int i = 0; i < m; ++i) {
                for (int j = 0; j < n; ++j) {
                    vector<vector<int>> cm(curMat);
                    cm[i][j] = 1-cm[i][j];
                    for (pair<int, int>& p: cord) {
                        int x = i + p.first;
                        int y = j + p.second;
                        if (x >= 0 && x < m && y >= 0 && y < n) {
                            cm[x][y] = 1 - cm[x][y];
                        }
                    }
                    int newN = getNum(cm, m, n);
                    if (!visited.count(newN)) {
                        visited.insert(newN);
                        q.push(newN);
                    }
                }
            }
            
            
            if (levelSize == 0) {
                levelSize = q.size();
                ++step;
            }
        }
        return visited.count(goal) ? step: -1;
    }
};
