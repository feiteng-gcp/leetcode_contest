class Solution {
public:
    bool allzeros(vector<vector<int>>& mat)
    {
        for(int i=0;i<mat.size();i++)
        {
            for(int j=0;j<mat[0].size();j++)
            {
                if(mat[i][j])
                    return false;
            }
        }
        return true;
    }
    vector<vector<int>>  flip(vector<vector<int>> mat,int i,int j)
    {
        int m=mat.size();
        int n=mat[0].size();
        if(i+1<m && j<n)
            mat[i+1][j]=1-mat[i+1][j];
        if(i-1>=0 && j<n)
            mat[i-1][j]=1-mat[i-1][j];
        if(i<m && j+1<n)
            mat[i][j+1]=1-mat[i][j+1];
        if( j-1>=0)
            mat[i][j-1]=1-mat[i][j-1];
        //return m;
          mat[i][j]=1-mat[i][j];  
        return mat;
    }
    int minFlips(vector<vector<int>>& mat) {
        queue<vector<vector<int>>>q;
        set<vector<vector<int>>>s;
        q.push(mat);
        int res=0;
        int m=mat.size();
        int n=mat[0].size();
        while(!q.empty())
        {
            int size=q.size();
            for(int i=0;i<size;i++)
            {
                auto f=q.front();
                q.pop();
                if(allzeros(f))
                    return res;
                for(int j=0;j<m;j++)
                {
                    for(int k=0;k<n;k++)
                    {
                        auto nm=flip(f,j,k);
                        if(s.find(nm)==s.end())
                        {
                            q.push(nm);
                            s.insert(nm);
                        }
                    }
                }
            }
            res++;
        }
        return -1;
    }
};
