class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        if (done(mat)) return 0;
        queue<vector<vector<int>>> q;
        q.push(mat);

        int output = 1;
        unordered_set<string> s;
        while (!q.empty()) {
            int size = q.size();
            for (int i = 0; i < size; ++i) {
                auto cur = q.front();
                q.pop();
                string str = encode(cur);
                if (s.find(str) != s.end()) continue;
                s.insert(str);

                for (int i = 0; i < m; ++i) {
                    for (int j = 0; j < n; ++j) {
                        auto new_m = change(cur, i, j);
                        if (done(new_m)) return output;
                        q.push(new_m);
                    }
                }
            }
            output++;
        }
        
        return -1;
    }
private:
    string encode(vector<vector<int>>& mat) {
        string output = "";
        int m = mat.size();
        int n = mat[0].size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                output += to_string(mat[i][j]);
            }
        }
        return output;
    }
    vector<vector<int>> change(vector<vector<int>>& mat, int i, int j) {
        vector<vector<int>> output = mat;
        output[i][j] = 1 - output[i][j];

        int m = mat.size();
        int n = mat[0].size();
        if (i - 1 >= 0) output[i - 1][j] = 1 - output[i - 1][j];
        if (i + 1 < m) output[i + 1][j] = 1 - output[i + 1][j];
        if (j - 1 >= 0) output[i][j - 1] = 1 - output[i][j - 1];
        if (j + 1 < n) output[i][j + 1] = 1 - output[i][j + 1];
        return output;
    }
    bool done(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();

        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (mat[i][j] == 1) return false;
            }
        }
        return true;
    }
};
