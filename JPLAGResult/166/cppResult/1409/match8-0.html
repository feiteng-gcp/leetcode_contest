class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int best = -1;
        for (int mask = 0; mask < (1 << (n * m)); mask++) {
            int flips = 0;
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                    int state = mat[i][j];
                    static int DY[] = {1,-1,0, 0,0};
                    static int DX[] = {0, 0,1,-1,0};
                    for (int d = 0; d < 5; d++) {
                        int ii = i + DY[d];
                        int jj = j + DX[d];
                        if (ii < 0 || ii >= n || jj < 0 || jj >= m) continue;
                        state ^= (mask >> (ii*m + jj)) & 1;
                    }

                    flips += (mask >> (i*m + j)) & 1;
                    if (state != 0) {
                        flips = -1000000;
                        break;
                    }
                }
            }
            //printf("mask=%d: flips=%d\n", mask, flips);
            if (flips >= 0) {
                best = (best < 0 ? flips : min(flips, best));
            }
        }
        return best;
    }
};
