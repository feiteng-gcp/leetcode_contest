class Solution { 
public:
    int minFlips(vector<vector<int>>& mat) {
        int dx[4]{-1, 0, 1, 0};
        int dy[4]{0, -1, 0, 1};

        int m = mat.size();
        int n = mat[0].size();
        int ans = INT_MAX;
        for (int i = 0; i < (1 << m * n); ++i) {
            bitset<10> bs(i);
            vector<vector<int>> matc(mat);
            for (int j = 0; j < m * n; ++j) {
                if (!bs[j]) continue;

                int r = j / n;
                int c = j % n;
                matc[r][c] = !matc[r][c];
                for (int k = 0; k < 4; ++k) {
                    int nr = r + dy[k];
                    int nc = c + dx[k];
                    if (nr < 0 || nr >= m || nc < 0 || nc >= n) continue;
                    matc[nr][nc] = !matc[nr][nc];

                }
            }
            bool ok = true;
            for (int j = 0; j < m; ++j)
                for (int k = 0; k < n; ++k) {
                    if (matc[j][k]) {
                        ok = false;
                    }
                }
            if (ok) ans = min(ans, (int) bs.count());
        }
        return ans == INT_MAX ? -1 : ans;
    }
};
