class Solution {
public:
  vector<vector<int>> flip(vector<vector<int>> mat, int i, int j) {
    int n = mat.size();
    int m = mat[0].size();
    for (int k = 0; k < n; k++) {
      for (int t = 0; t < m; t++) {
        if ((abs(i - k) == 1 && j == t) || (abs(j - t) == 1 && i == k)) {
          mat[k][t] ^= 1;
        }
      }
    }
    mat[i][j] ^= 1;
    
    return mat;
  }
  
  int hash(const vector<vector<int>>& mat) {
    int h = 0;
    int n = mat.size();
    int m = mat[0].size();
    int p = 1;
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < m; j++) {
        h += mat[i][j] * p;
        p *= 2;
      }
    }
    
    return h;
  }
  
  int minFlips(vector<vector<int>>& mat) {
    int n = mat.size();
    int m = mat[0].size();
    
    vector<int> d(10 * 1000, -1);

    vector<vector<vector<int>>> queue_v = {mat};
    d[hash(mat)] = 0;
    int uk = 0;
  
    while (uk < queue_v.size()) {
      vector<vector<int>> v = queue_v[uk];
//       for (int i = 0; i < n; i++) {
//         for (int j = 0; j < m; j++) {
//           cout << v[i][j];
//         }
//         cout << endl;
//       }
      
//       cout << endl;
      int current_d = d[hash(v)];
      uk++;
      
      for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
          vector<vector<int>> u = flip(v, i, j);
          int h = hash(u);
          if (d[h] == -1) {
            d[h] = current_d + 1;
            queue_v.push_back(u);
          }
        }
      }
    }
    
    return d[0];
  }
};
