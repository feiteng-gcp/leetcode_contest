int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

void flip(vector<vector<int>>& m, int i, int j) {
    m[i][j] ^= 1;
    for (int x = 0; x < 4; ++x) {
        int ni = i + dx[x];
        int nj = j + dy[x];
        if (ni >= 0 && nj >= 0 && ni < m.size() && nj < m[0].size()) {
            m[ni][nj] ^= 1;
        }
    }
}

class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int ans = 100;
        int n = mat.size();
        int m = mat[0].size();
        REP(mask, (1 << (n * m))) {
            auto cur = mat;
            int sc = 0;
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    if ((mask >> (i * m + j)) % 2 == 1) {
                        flip(cur, i, j);
                        ++sc;
                    }
                }
            }
            bool good = true;
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    if (cur[i][j]) {
                        good = false;
                    }
                }
            }
            if (good) {
                ans = min(ans, sc);
            }
        }
        if (ans == 100) ans = -1;
        return ans;
    }
};
