const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
class Solution {
public:
    int minFlips(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        int len = n * m;
        vector<int> dp(1 << len, -1);

        int s = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                s += a[i][j] << (i * m + j);
            }
        }
        dp[s] = 0;
        queue<int> Q;
        Q.push(s);

        while (!Q.empty()) {
            int x = Q.front();
            //cout << "x=" << x << " " << dp[x] << endl;
            if (x == 0) return dp[x];
            Q.pop();
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    int y = x ^ (1 << (i * m + j));
                    for (int k = 0; k < 4; ++k) {
                        int ni = i + d[k][0];
                        int nj = j + d[k][1];
                        if (ni < 0 || ni >= n || nj < 0 || nj >= m) continue;
                        y ^= 1 << (ni * m + nj);   
                    }
                    if (dp[y] < 0) {
                        dp[y] = dp[x] + 1;
                        Q.push(y);
                    }
                }
            }
        }
        //cout << "==" << endl;
        return -1;
    }
};
