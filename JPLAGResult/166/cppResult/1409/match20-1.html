typedef vector<vector<int>> board;

class Solution {
public:

    bool zero(board mat) {
        int n = mat.size();
        int m = mat[0].size();
        for (int i=0; i<n; i++) {
            for (int j=0; j<m; j++) {
                if (mat[i][j]==1) return false;
            }
        }
        return true;
    }
    void f(int& x) {
        x=1-x;
    }
    vector<board> make_flips(board mat) {

        int n = mat.size();
        int m = mat[0].size();
        vector<board> res;
        for (int i=0; i<n; i++) {
            for (int j=0; j<m; j++) {
                board cur = mat;
                f(cur[i][j]);
                if (i-1>=0) f(cur[i-1][j]);
                if (j-1>=0) f(cur[i][j-1]);
                if (i+1<n) f(cur[i+1][j]);
                if (j+1<m) f(cur[i][j+1]);
                res.push_back(cur);

            }
        }
        return res;
    }
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size(); int m = mat[0].size();
        map<board,int> mp;
        queue<pair<board,int>> q;
        q.push({mat, 0});
        while (!q.empty()) {
            pair<board,int> cur = q.front(); q.pop();
            if (zero(cur.first)) return cur.second;
            if (mp.count(cur.first)) continue;
            mp[cur.first] = cur.second;
            int flips = cur.second;
            vector<board> v = make_flips(cur.first);
            for (auto x: v) {
                if (mp.count(x)) continue;
                q.push({x, flips+1});
            }
        }
        return -1;
    }
};
