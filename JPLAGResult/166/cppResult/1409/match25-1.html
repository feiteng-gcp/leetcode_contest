class Solution {
public:
    int tmp[5][5],box[5][5],n,m,f[4][2]={0,1,0,-1,1,0,-1,0};
    bool check(int x,int y){
        if(x<0||x>=n||y<0||y>=m)return false;
        return true;
    }
    int judge(int sta,vector<vector<int>>& mat){
        int cnt=0;
        memset(tmp,0,sizeof(tmp));
        for(int i=0;i<n*m;i++){
            if(sta&(1<<i)){tmp[i/m][i%m]=1;cnt++;}
        }

        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                int sum=0;
                for(int k=0;k<4;k++){
                    int x=f[k][0]+i,y=f[k][1]+j;
                    if(check(x,y))sum+=tmp[x][y];
                }
                sum+=tmp[i][j];
                if(sum&1)box[i][j]=mat[i][j]^1;
                else box[i][j]=mat[i][j];
                if(box[i][j])return 1e9;

            }
        }
        return cnt;
    }
    int minFlips(vector<vector<int>>& mat) {
        n=mat.size(),m=mat[0].size();
        int tp=1<<(n*m),ans=1e9;
        for(int i=0;i<tp;i++){
            ans=min(ans,judge(i,mat));
        }
        if(ans==1e9)return -1;
        return ans;
    }
};
