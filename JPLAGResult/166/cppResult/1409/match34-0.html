class Solution {
public:
    int ans;
    map<vector<vector<int>>, int> s;
    int dfs(vector<vector<int>> mat, int depth) {
        // for (int i = 0; i < mat.size(); i++) {
        //     for (int j = 0; j < mat[0].size(); j++) cout << mat[i][j] << " ";
        //     cout << endl;
        // }
        // cout << "______" << " " << depth << endl;

        // if (depth == 7) return;
        queue<vector<vector<int>>> q;
                            int onecnt = 0;
                            for (int i= 0; i < mat.size(); i++) {
                                for (int j = 0; j < mat[0].size(); j++) {
                                    onecnt += mat[i][j];
                                }
                            }
                            if (onecnt == 0) {
                                return 0;
                            }
        q.push(mat);
        s[mat] = 0;
        int cnt = -1;
        while (!q.empty()) {
            cnt++;
            int qsize = q.size();
            for (int z = 0; z < qsize; z++) {
                auto top = q.front();
                q.pop();
                
                for (int i = 0; i < mat.size(); i++) {
                    for (int j = 0; j < mat[0].size(); j++) {
                        auto temp = top;
                        int dir[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
                        temp[i][j] ^= 1;
                        for (int k = 0; k < 4; k++) {
                            int nx = dir[k][0] + i;
                            int ny = dir[k][1] + j;
                            if (nx < 0 || ny < 0 || nx >= mat.size() || ny >= mat[0].size()) continue;
                            temp[nx][ny] ^= 1;
                        }

                        if (s.find(temp) == s.end()) {
                            
                            int onecnt = 0;
                            for (int i= 0; i < temp.size(); i++) {
                                for (int j = 0; j < temp[0].size(); j++) {
                                    onecnt += temp[i][j];
                                }
                            }
                            if (onecnt == 0) {
                                return cnt + 1;
                            }
                            q.push(temp);
                            s[temp] = cnt;
                        }
                    }
                }
            }
        }
        return -1;

    }
    
    int minFlips(vector<vector<int>>& mat) {
        ans = 2000000;
        return dfs(mat, 0);
        return ans;
    }
};
