class Solution {
public:
    int minFlips(vector<vector<int>> &mat) {
        int count = 100;
        int lenx = mat.size();
        int leny = mat[0].size();
        const int mv[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        for (int i = 0; i < 1 << (lenx * leny); i++) {
            int now = 0;
            vector<vector<int>> tmp = mat;
            for (int j = 0; j < lenx * leny; j++) {
                if (i & (1 << j)) {
                    int x = j / leny;
                    int y = j % leny;
                    tmp[x][y] = 1 - tmp[x][y];
                    now++;
                    for (int k = 0; k < 4; k++) {
                        int xx = x + mv[k][0];
                        int yy = y + mv[k][1];
                        if (xx < 0 || xx >= lenx || yy < 0 || yy >= leny) {
                            continue;
                        }
                        tmp[xx][yy] = 1 - tmp[xx][yy];
                    }
                }
            }
            bool flag = true;
            for (int j = 0; j < lenx; j++) {
                for (int k = 0; k < leny; k++) {
                    if (tmp[j][k]) {
                        flag = false;
                        break;
                    }
                }
            }
            if (flag)
                count = min(count, now);
        }
        if (count == 100)
            return -1;
        return count;
    }
};
