class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int n=mat.size();
        int m=mat[0].size();
        int k = n*m;
        int ans=-1;
        for(int mask=0;mask<(1<<k);mask++) {
            vector<vector<int>> nmat = mat;
            for(int i=0;i<n;i++)
                for(int j=0;j<m;j++) {
                    int ij = i*m+j;
                    if (!((mask>>ij)&1))
                        continue;
                    int dx[]={-1,1,0,0,0};
                    int dy[]={0,0,-1,1,0};
                    for(int d=0;d<5;d++) {
                        int ni=i+dx[d];
                        int nj=j+dy[d];
                        if (ni<0 || ni>=n || nj<0 || nj>=m)
                            continue;
                        nmat[ni][nj]=1-nmat[ni][nj];
                    }
                }
            bool ok=true;
            for(int i=0;i<n;i++)
                for(int j=0;j<m;j++)
                    ok &= (nmat[i][j]==0);
            if (ok) {
                if (ans == -1)
                    ans=__builtin_popcount(mask);
                else
                    ans=min(ans,__builtin_popcount(mask));
            }
        }
        return ans;
        
    }
};
