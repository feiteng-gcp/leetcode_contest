class Solution {
public:
    bool isAllZero(vector<vector<int>>& mat) {
         int m = mat.size();
        int n = mat[0].size();
        
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (mat[i][j] != 0)
                    return false;
            }
        }
        
        return true;
    }
    
    bool isValid(int i, int j, int m, int n)   
    {
        if (i < 0 || i >=m)
            return false;
        
        if (j <0 || j >= n)
            return false;
        
        return true;
            
        
    }
    string  generate(vector<vector<int>>& mat) {
        string result;
        int m = mat.size();
        int n = mat[0].size();
        
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                result += to_string(mat[i][j]);
            }
        }
        
        return result;
    }
    int minFlips(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        
        unordered_set<string> hashset;
        hashset.insert(generate(mat));
        queue<vector<vector<int>>> q;
        q.push(mat);
        int count = 0;
        while (!q.empty())
        {
            int size = q.size();
            
            for (int k = 0; k < size; k++)
            {
                auto thisMat = q.front();
                q.pop();
                
                if (isAllZero(thisMat))
                    return count;
                
                auto newM = thisMat;
                for (int i = 0; i < m; i++)
                {
                    for (int j = 0; j < n; j++)
                    {
                        newM[i][j] = !newM[i][j];
                        
                        if(isValid(i-1, j, m, n))
                            newM[i-1][j] = !newM[i-1][j];
                        
                        if(isValid(i+1, j, m, n))
                            newM[i+1][j] = !newM[i+1][j];
                        
                        if(isValid(i, j-1, m, n))
                            newM[i][j-1] = !newM[i][j-1];
                        
                        if(isValid(i, j+1, m, n))
                            newM[i][j+1] = !newM[i][j+1];
                        
                        
                        string str = generate(newM);
                        //cout << str << endl;
                        if (hashset.count(str) == 0)
                        {
                            hashset.insert(str);
                            q.push(newM);
                        }
                        
                        newM = thisMat;
                    }
                }
            }
            
            count++;
        }
        
        return -1;
    }
};
