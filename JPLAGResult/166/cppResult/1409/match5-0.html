class Solution {
private:
    int matToInt( vector<vector<int>>& mat ) {
        int n = 0;
        int f = 0;
        for( auto &i : mat ) {
            for( auto & j : i ) {
                n += ( j << f );
                f++;;
            }
        }
        return n;
    }
    bool allZero( const vector<vector<int>> & m ) {
        for( auto &i : m ) {
            for( auto & j : i ) {
                if( j != 0 ) {
                    return false;
                }
            }
        }
        return true;
    }
    
    void flip( vector<vector<int>> & m, int i, int j ) {
        m[i][j] = ( m[i][j] + 1 ) % 2;
        if( i + 1 < m.size() ) {
            m[i + 1][j] = ( m[i + 1][j] + 1 ) % 2;
        }
        if( i - 1 >= 0 ) {
            m[i- 1][j] = ( m[i - 1][j] + 1 ) % 2;
        }
        if( j + 1 < m[i].size() ) {
            m[i][j + 1] = ( m[i][j + 1] + 1 ) % 2;
        }
        if( j - 1 >= 0 ) {
            m[i][j - 1] = ( m[i][j - 1] + 1 ) % 2;
        }
    }
    unordered_set<int> matSeen;


public:
    int minFlips(vector<vector<int>>& mat) {
        queue<vector<vector<int>>> mats;
        mats.push( mat );
        int res = 0;
        while( !mats.empty() ) {
            int size = mats.size();
            for( int n = 0; n < size; n++ ) {
                vector<vector<int>> curr = mats.front();
                if( allZero( curr ) ) {
                    return res;
                }
                mats.pop();
                for( int i = 0; i < curr.size(); i++ ) {
                    for( int j = 0; j < curr[i].size(); j++ ) {
                        flip( curr, i, j );
                        int id = matToInt( curr );
                        if( !matSeen.count( id ) ) {
                            //cout << curr[0][0] << " " << curr[0][1] << " " << curr[1][0] << " " <<curr[1][1] << endl;
                            mats.push( curr );
                            matSeen.insert( id );
                        }
                        flip( curr, i, j );
                    }
                }
            }
            res++;
        }
        return -1;
    }
};
