class Solution {
    int m, n;
    const int dir[5][2] = {{0,0}, {-1,0},{0,1},{1,0},{0,-1}};
    int getkey(const vector<vector<int>>& mt){
        int res = 0;
        for(int i = 0; i < m; i++){
            for(int j=0; j < n; j++){
                if(mt[i][j]){
                    res += (1<<((i*n)+j));
                }
            }
        }
        return res;
    }
    inline
    void flipOne(int &key, int x, int y){
        
    }
    
    int flip(int key, int x, int y){
        for(int i =0;i <5; i++){
            int xx = x + dir[i][0];
            int yy = y + dir[i][1];
            if(xx>=0 && xx<m&& yy>=0 && yy<n){
                key ^= (1<<((xx*n)+yy));
            }
        }
        return key;
    }
    
    void bfs(int key, unordered_set<int>& us, queue<int>&q){
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                int newkey = flip(key, i, j);
                if(us.insert(newkey).second){
                    q.push(newkey);
                }
            }
        }
    }
    
public:
    int minFlips(vector<vector<int>>& mt) {
        m = mt.size();
        n = mt[0].size();
        unordered_set<int> us;
        queue<int> q;
        int key= getkey(mt);
        us.insert(key);
        q.push(key);
        int res = 0;
        int cnt = 0;
        while(cnt = q.size()){
            while(cnt--){
                key = q.front();
                if(key == 0){
                    return res;
                }
                q.pop();
                bfs(key, us, q);
            }
            ++res;
        }
        return -1;
    }
};
