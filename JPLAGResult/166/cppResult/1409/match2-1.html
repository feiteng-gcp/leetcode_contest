class Solution {
public:
    int n, m, ans = -1;
    bool judge(vector<vector<int>>& mat) {
        int i, j;
        for (i = 0; i < n; i++) {
            for (j = 0; j < m; j++) {
                if (mat[i][j] == 1) {
                    return false;
                }
            }
        }
        return true;
    }
    void flip(int x, int y, vector<vector<int>>& mat) {
        mat[x][y] ^= 1;
        if (x - 1 >= 0) {
            mat[x - 1][y] ^= 1;
        }
        if (x + 1 < n) {
            mat[x + 1][y] ^= 1;
        }
        if (y - 1 >= 0) {
            mat[x][y - 1] ^= 1;
        }
        if (y + 1 < m) {
            mat[x][y + 1] ^= 1;
        }
    }
    void dfs(int x, int y, vector<vector<int>> mat, int c) {
        // cout << x << " " << y << endl;
        if (x == n) {
            if (judge(mat)) {
                if (ans == -1 || ans < c) {
                    ans = c;
                }
            }
            return ;
        }
        vector<vector<int>> t = mat;
        flip(x, y, t);
        if (y == m - 1) {
            dfs(x + 1, 0, mat, c);
            dfs(x + 1, 0, t, c + 1);
        } else {
            dfs(x, y + 1, mat, c);
            dfs(x, y + 1, t, c + 1);
        }
    }
    int minFlips(vector<vector<int>>& mat) {
        n = mat.size();
        m = mat[0].size();
        dfs(0, 0, mat, 0);
        return ans;
    }
};
