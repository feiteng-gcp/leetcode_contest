class Solution {
void flip(int i, int j, vector<vector<int>> & mat) {
    vector<int> dirX = {1, 0, -1, 0};
    vector<int> dirY = {0, -1, 0, 1};
    
    mat[i][j] = 1 - mat[i][j];
    
    for (int k = 0; k < 4; k++) {
        int newY = i + dirY[k];
        int newX = j + dirX[k];
        
         if (newY >= 0 && newY < mat.size() && newX >= 0 && newX < mat[0].size()) {
            mat[newY][newX] = 1 - mat[newY][newX];
        }
    }
    
}
    
bool zero(vector<vector<int>> & mat) {
	bool z = true;

	for (int i = 0; i < mat.size(); i++) {
		for (int j = 0; j < mat[0].size(); j++) {
			if (mat[i][j]) z = false;
		}
	}

	return z;

}
    
    
public:
    int minFlips(vector<vector<int>>& mat) {
        vector<vector<vector<int>>> vc;
        int cnt = 0;
        
        map<vector<vector<int>>, int> visited;
        vc.push_back(mat);
        visited[mat] = 1;
        
        while(!vc.empty()) {
            vector<vector<vector<int>>> new_vc;
            
            for (vector<vector<int>> v: vc) {
            
                if(zero(v)) return cnt;
                for (int i = 0; i < v.size(); i++) {
                    for (int j = 0; j < v[0].size(); j++) {
                        flip(i, j, v);
                        
              

                        if (visited[v] == 0) {
                            vector<vector<int>> vv = v;
                            new_vc.push_back(vv);
                            visited[vv] = 1;
                        }

                        flip(i, j, v);
                    }
                }
            }
            
            vc = new_vc;
            cnt++;
        }
        
        return -1;
        
    }
};
