class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        unordered_set<int> visited;
        int status = se(mat, m, n);
        if(status == 0)
            return 0;
        queue<int> q;
        q.push(status);
        visited.insert(status);
        int steps = 0;
        while(!q.empty()) {
            int sz = q.size();
            for(int i = 0; i < sz; i++) {
                int st = q.front();
                q.pop();
                if(st == 0)
                    return steps;
                getNextSt(st, visited, q, m, n);
            }
            steps++;
        }
        return -1;
    }
    
private:
    int se(vector<vector<int>>& mat, int m, int n) {
        int d = 0;
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                d <<= 1;
                d |= mat[i][j];
            }
        }
        return d;
    }
    
    void de(int st, int m, int n, vector<vector<int> >& matrix) {
        for(int i = m - 1; i >= 0; i--) {
            for(int j = n - 1; j >= 0; j--) {
                int d = (st&1);
                st >>= 1;
                matrix[i][j] = d;
            }
        }
    }
    
    void getNextSt(int st, unordered_set<int>& visited, queue<int>& q, int m, int n) {
        vector<vector<int> > matrix(m, vector<int>(n));
        de(st, m, n, matrix);
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                //flip
                matrix[i][j] = 1 - matrix[i][j];
                if(i + 1 < m) {
                    matrix[i+1][j] = 1 - matrix[i+1][j];
                }
                if(i - 1 >= 0) {
                    matrix[i-1][j] = 1 - matrix[i-1][j];
                }
                if(j + 1 < n) {
                    matrix[i][j+1] = 1 - matrix[i][j+1];
                }
                if(j - 1 >= 0) {
                    matrix[i][j-1] = 1 - matrix[i][j-1];
                }
                
                int nst = se(matrix, m, n);
                if(visited.find(nst) == visited.end()) {
                    visited.insert(nst);
                    q.push(nst);
                }
                //flip back
                matrix[i][j] = 1 - matrix[i][j];
                if(i + 1 < m) {
                    matrix[i+1][j] = 1 - matrix[i+1][j];
                }
                if(i - 1 >= 0) {
                    matrix[i-1][j] = 1 - matrix[i-1][j];
                }
                if(j + 1 < n) {
                    matrix[i][j+1] = 1 - matrix[i][j+1];
                }
                if(j - 1 >= 0) {
                    matrix[i][j-1] = 1 - matrix[i][j-1];
                }
                
            }
        }
    }
};
