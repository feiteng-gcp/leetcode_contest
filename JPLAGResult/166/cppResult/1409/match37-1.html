class Solution {
public:
	int inf = (1<<30);
	int ans = inf;
	int n, m;
	map<int,int> mp;
	
	int solve(int num, int now){
		int a = mp[now];
		
		if(now == 0) return num;
		
      //  cout<<"num = "<<num<<endl;
       // cout<<(now>>3)%2<<""<<(now>>2)%2<<endl;
         // cout<<(now>>1)%2<<""<<(now>>0)%2<<endl;
		if(a && a-1 <= num) return inf;
		
		mp[now] = num+1;
		

		int ans = inf;
		
		for(int i = 0; i < n; i++){
			for(int j = 0; j < m; j++){
				int t = now ^ (1<<(n * m - (i * m + j) - 1));
				
				for(int k = 0; k < 4; k++){
					int y = i + (k-2)%2, x = j + (1-k)%2;
					
					if(y < 0 || y >= n || x < 0 || x >= m) continue;
					
					t = t ^ (1<<(n *m - (y * m + x) - 1));
				}
				

				ans = min(solve(num+1, t), ans);
			}
		}
		
		return ans;
	}
	
    int minFlips(vector<vector<int>>& mat) {
        n = mat.size();
        m = mat[0].size();
		int now = 0, ans = 0;

		
		for(int i = 0; i < n; i++){
			for(int j = 0; j < m; j++){
				now = now*2 + mat[i][j];
			}
		}
		
		ans = solve(0, now);
		
		if(ans == inf) return -1;
		else return ans;
    }
};
