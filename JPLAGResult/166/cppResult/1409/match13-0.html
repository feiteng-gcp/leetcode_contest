class Solution {
private:
    vector<vector<int>>dir = {{1,0},{-1,0},{0,1},{0,-1}};
public:
    int minFlips(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        int sz = n * m;
        int ret = sz + 1;
        for(int i=0;i<(1<<sz);i++){
            vector<vector<int>> tt = vector<vector<int>>(n, vector<int>(m, 0));
            int tmp = 0;
            for(int j=0;j<sz;j++){
                if(i&(1<<j)){
                    tmp++;
                    int x = j/m;
                    int y = j%m;
                    tt[x][y]^=1;
                    for(int k=0;k<4;k++){
                        int nx = x+dir[k][0];
                        int ny = y+dir[k][1];
                        if(nx>=0&&nx<n&&ny>=0&&ny<m)tt[nx][ny]^=1;
                    }
                }
            }
            bool sign = 0;
            for(int j=0;j<n;j++)
                for(int k=0;k<m;k++){
                    if(a[j][k]^tt[j][k])sign=1;
                }
            if(!sign)ret=min(ret, tmp);
        }
        if(ret==sz+1)ret=-1;
        return ret;
    }
};
