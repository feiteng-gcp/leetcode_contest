class Solution {
public:
    int dx[4] = {1, -1, 0, 0};
    int dy[4] = {0, 0, -1, 1};
    int minFlips(vector<vector<int>>& mat) {
        int N = mat.size();
        int M = mat[0].size();
        int ans = 100;
        for (int i = 0; i < (1 << (N * M)); i++)
        {
            auto temp = mat;
            for (int j = 0; j < N * M; j++)
            {
                if ((i & (1 << j)) == 0) continue;
                int x = j / M;
                int y = j % M;
                temp[x][y] = 1 - temp[x][y];
                for (int k = 0; k < 4; k++)
                {
                    int nx = x + dx[k];
                    int ny = y + dy[k];
                    if (nx < 0 || nx >= N || ny < 0 || ny >= M) continue;
                    temp[nx][ny] = 1 - temp[nx][ny];
                }
            }
            int flag = 1;
            for (int j = 0; j < N; j++)
            {
                for (int k = 0; k < M; k++)
                {
                    if (temp[j][k]) flag = 0;
                }
            }
            if (flag) ans = min(ans, __builtin_popcount(i));
        }
        if (ans < 100) return ans;
        return -1;
    }
};
