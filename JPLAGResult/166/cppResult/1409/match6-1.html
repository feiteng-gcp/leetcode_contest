class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        unordered_set<int> visit;
        queue<int> fringe;
        int origin = toInt(mat);
        visit.insert(origin);
        fringe.push(origin);

        int step = 0;
        while (!fringe.empty()) {
            int size = fringe.size();
            for (int i = 0; i != size; ++i) {
                int node = fringe.front();
                fringe.pop();
                if (node == 0) {
                    return step;
                }

                for (int i = 0; i != mat.size(); ++i) {
                    for (int j = 0; j != mat[0].size(); ++j) {
                        int next = flip(node, i, j, mat.size(), mat[0].size());
                        if (visit.find(next) == visit.end()) {
                            visit.insert(next);
                            fringe.push(next);
                        }
                    }
                }
            }
            ++step;
        }
        return -1;
    }
private:

    int toInt(vector<vector<int>>& mat) {
        int result = 0;
        for (int i = 0; i != mat.size(); ++i) {
            for (int j = 0; j != mat[0].size(); ++j) {
                if (mat[i][j] == 1) {
                    result |= (1 << (i * mat[0].size() + j));
                }
            }
        }
        return result;
    }

    int flip(int matrix, int i, int j, int m, int n) {

        int result = matrix ^ (1 << (i * n + j));
        if (i > 0) {
            result = result ^ (1 << ((i - 1) * n + j));
        }
        if (i + 1 < m) {
            result = result ^ (1 << ((i + 1) * n + j));
        }
        if (j > 0) {
            result = result ^ (1 << (i * n + j - 1));
        }
        if (j + 1 < n) {
            result = result ^ (1 << (i * n + j + 1));
        }
        return result;
    }
};
