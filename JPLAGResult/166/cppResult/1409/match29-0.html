set<vector<vector<int>>> visited;
int dirx[] = {1,0,-1,0};
int diry[] = {0,1,0,-1};
class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        visited.clear();
        int n = mat.size();
        int m = mat[0].size();
        int cur = 0;
        queue<vector<vector<int> >> q;
        visited.insert(mat);
        q.push(mat);
        while(q.size()>0){
            int t = q.size();
            for(int i=0;i<t;i++){
                vector<vector<int>> curmat = q.front();
                q.pop();
                
                bool allZero = true;
                for(auto v1:curmat){
                    for(auto v2:v1){
                        if(v2){
                            allZero = false;
                        }
                    }
                }
                
                if(allZero){
                    return cur;
                }
                
                for(int i=0;i<n;i++){
                    for(int j=0;j<m;j++){
                        vector<vector<int>> next = curmat;
                        next[i][j] = 1-next[i][j];
                        for(int k=0;k<4;k++){
                            int nextx = i+dirx[k];
                            int nexty = j+diry[k];
                            if(nextx>=0&&nextx<n&&nexty>=0&&nexty<m){
                                next[nextx][nexty] = 1-next[nextx][nexty];
                            }
                        }
                        if(visited.count(next)==0){
                            q.push(next);
                            visited.insert(next);
                        }
                    }
                }
                
            }
            cur++;
        }
        return -1;
        
    }
};
