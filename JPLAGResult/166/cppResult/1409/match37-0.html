const int dx[4] = {1, -1, 0, 0};
const int dy[4] = {0, 0, 1, -1};
class Solution {
public:
    int m, n;
    bool valid(int x, int y) {
        return (x >= 0) && (x < n) && (y >= 0) && (y < m);
    }
    int getn(int x, int y) {
        return x * m + y;
    }
    int flp(int S, int x, int y) {
        int msk = 1 << (getn(x, y));
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (valid(nx, ny)) {
                msk += (1 << (getn(nx, ny)));
            }
        }
        return S ^ msk;
    }
    int minFlips(vector<vector<int>>& mat) {
        n = mat.size();
        m = mat[0].size();
        int s = n * m;
        int ss = (1 << s);
        int *f = new int[(1 << s)];
        bool *vis = new bool[(1 << s)];
        int x = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                x = (x << 1) + mat[i][j];
            }
        }
        for (int i = 0; i < ss; ++i) {
            f[i] = -1;
            vis[i] = false;
        }
        f[x] = 0;
        vis[x] = true;
        vector<int> q;
        int qh, qt;
        q.clear();
        q.push_back(x);
        qh = 0; qt = 1;
        while (qh < qt) {
            int S = q[qh];
            for (int x = 0; x < n; ++x) {
                for (int y = 0; y < m; ++y) {
                    int nS = flp(S, x, y);
                    if (vis[nS]) {
                        continue;
                    }
                    f[nS] = f[S] + 1;
                    vis[nS] = true;
                    q.push_back(nS);
                    ++qt;
                }

            }
            ++qh;
        }
        if (vis[0]) {
            return f[0];
        }
        else {
            return -1;
        }
        return 0;
        
    }
};
