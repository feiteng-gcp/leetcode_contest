class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int N = n * m;
        int M = 1 << N;
        
        const int INF = 1 << 30;
        vector<int> dis(M, INF);
        
        int s = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (mat[i][j] == 1) {
                    s |= 1 << (i * m + j);
                }
            }
        }
        int t = 0;
        
        dis[s] = 0;
        queue<int> que;
        que.push(s);
        
        
        vector<pair<int,int>> adjs{{-1, 0}, {1, 0}, {0, 0}, {0, -1}, {0, 1}};
        
        auto inside = [&](int x, int y) {
            return x >= 0 && x < n && y >= 0 && y < m;
        };
        
        vector<vector<int>> MAT(mat);
        vector<vector<int>> MAT2(MAT);
        
        while(!que.empty()) {
            int u = que.front();
            que.pop();
            
            for (int i = 0; i < N; ++i) {
                MAT[i / m][i % m] = (u >> i) & 0x1;
            }
            
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    MAT2 = MAT;
                    for (auto& adj : adjs) {
                        int x = i + adj.first;
                        int y = j + adj.second;
                        if (inside(x, y)) {
                            MAT2[x][y] ^= 1;
                        }
                    }
                    
                    int v = 0;
                    for (int x = 0; x < n; ++x) {
                        for (int y = 0; y < m; ++y) {
                            if (MAT2[x][y] == 1) {
                                v |= 1 << (x * m + y);
                            }
                        }
                    }
                    if (dis[v] > dis[u] + 1) {
                        dis[v] = dis[u] + 1;
                        
                        que.push(v);
                    }
                }
            }
        }
        
        return dis[t] < INF ? dis[t] : -1;
        
    }
};
