class Node{
public:
    vector<vector<int>> mat;
    int op;
    Node(vector<vector<int>> &m,int o){
        op=o;
        mat=m;
    }
    
};

int vX[]={1,-1,0,0};
int vY[]={0,0,1,-1};
class Solution {
public:
    int N,M;
    int getStatus(vector<vector<int>>& mat){

        int res=0;
        for(int i=0;i<N;i++){
            for(int j=0;j<M;j++){
                if(mat[i][j]==1){
                    res|=(1<<(i*M+j));
                }
            }
        }
        return res;
    }
    int minFlips(vector<vector<int>>& mat) {
        N=mat.size();
        M=mat[0].size();
        unordered_set<int> sHave;
        Node tmp(mat,0);
        queue<Node> qNode;
        qNode.push(tmp);
        sHave.insert(getStatus(mat));

        while(!qNode.empty()){
            Node nt=qNode.front();
            qNode.pop();
            if(getStatus(nt.mat)==0){
                return nt.op;
            }
            for(int i=0;i<N;i++){
                for(int j=0;j<M;j++){
                    vector<vector<int>> vT=nt.mat;
                    vT[i][j]^=1;
                    for(int k=0;k<4;k++){
                        int x=i+vX[k];
                        int y=j+vY[k];
                        if(x<0||x>=N||y<0||y>=M)continue;
                        vT[x][y]^=1;
                    }
                    int s=getStatus(vT);
                    if(sHave.find(s)!=sHave.end()){
                        continue;
                    }
                    sHave.insert(s);
                    Node n1(vT,nt.op+1);
                    qNode.push(n1);
                }
            }
        }
        return -1;
        
    }
};
