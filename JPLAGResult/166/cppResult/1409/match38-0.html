class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int code = 0;
        int n = mat.size();
        int m = mat[0].size();
        int dir[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
        for (int i = 0; i<n; i++)
            for (int j = 0; j<m; j++)
                code = code * 2 + mat[i][j];
        if (code == 0)
            return 0;
        int vis[1024];
        for (int i = 0; i<1024; i++)
            vis[i] = 10000;
        vis[code] = 0;
        
        queue<pair<int, vector<vector<int>>>> Q;
        Q.push(pair<int, vector<vector<int>>>(code, mat));
        while (!Q.empty())
        {
            int cur = Q.front().first;
            vector<vector<int>> curmat = Q.front().second;
            Q.pop();
            
            for (int i = 0; i<n; i++)
                for (int j = 0; j<m; j++)
                {
                    vector<vector<int>> tempmat = curmat;
                    tempmat[i][j] = 1-tempmat[i][j];
                    for (int k = 0; k<4; k++)
                    {
                        int tempx = i + dir[k][0];
                        int tempy = j + dir[k][1];
                        if ((tempx>=0) && (tempy>=0) && (tempx<n) && (tempy<m))
                            tempmat[tempx][tempy] = 1-tempmat[tempx][tempy];                        
                    }
                    int tempcode = 0;
                    for (int tempi = 0; tempi<n; tempi++)
                        for (int tempj =0; tempj<m; tempj++)
                            tempcode = tempcode * 2 + tempmat[tempi][tempj];
                    if (vis[tempcode] == 10000)
                    {
                        vis[tempcode] = vis[cur] + 1;
                        if (tempcode == 0)
                            return (vis[cur] + 1);
                        Q.push(pair<int, vector<vector<int>>>(tempcode, tempmat));
                    }
                }
            
        }
        return -1;
    }
};
