class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int ans = -1;
        for(int i =0;i<(1<<(m*n));i++){
            // copy
            vector<vector<int> >m2;
            for(int p=0;p<n;p++){
                m2.push_back({});
                for(int q = 0;q<m;q++){
                    m2[p].push_back(mat[p][q]);
                }
            }

            int cnt =0;
            for(int j=0;j<m*n;j++){
                if(i & (1<<j)){
                    cnt++;
                    int ii = j/m;
                    int jj = j%m;
                    m2[ii][jj] ^=1;
                    int deri[] = {0,1,0,-1};
                    int derj[] = {1,0,-1,0};
                    for(int k=0;k<4;k++){
                        int ni = ii+deri[k];
                        int nj = jj+derj[k];
                        if(ni < 0 || nj <0 || ni ==n || nj==m)continue;
                        m2[ni][nj] ^=1;
                    }
                }             
            }
            // check
            bool ok = true;
            assert(m2.size()==n);
            for(int p=0;p<n;p++){
                assert(m2[p].size()==m);
                for(int q = 0;q<m;q++){
                    if(m2[p][q] != 0){
                        ok = false;
                    }
                }
            }
            if(ok){
                if(ans = -1 || ans > cnt){
                    ans = cnt;
                }
            }
        }
        return ans;
    }
};
