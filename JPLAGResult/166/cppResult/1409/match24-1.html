class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        
        const int dx[] = {1,-1,0,0};
        const int dy[] = {0,0,1,-1};
        int res = -1;
        for(int i = 0; i < (1<<(n*m)); i++) {

            auto a = mat;
            int s = 0;
            for(int x = 0; x < n; x++) {
                for(int y = 0; y < m ;y++) {
                    int z = x*m + y;
                    if((i>>z)&1) {
                        s++;
                        a[x][y] = 1-a[x][y];

                        for(int d = 0; d < 4; d++) {
                            int nx,ny;
                            nx = x + dx[d];
                            ny = y + dy[d];
                            if(nx<0||nx>=n||ny<0||ny>=m) continue;
                            a[nx][ny] = 1-a[nx][ny];
                        }
                    }
                }
            }

            int t = 0;
            for(int x = 0; x < n; x++) {
                for(int y = 0; y < m ;y++) {
                    t += a[x][y];
                }
            }
            if(t==0) {
                if(res==-1 || res > s) {
                    res= s;
                }
            }
        }
        return res;
    }
};
