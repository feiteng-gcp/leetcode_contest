#include <iostream>
#include <cstring>
#include <queue>
#include <vector>
#include <string>
#include <iomanip>
#include <cmath>
#include <climits>
using namespace std;

int n,m;
int mp[20][20],fp[20][20],vis[20][20];
int fx[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};

void change(int i,int j){

    fp[i][j] ^= 1;
    for(int k = 0; k < 4; k++){
        int ni = i + fx[k][0];
        int nj = j + fx[k][1];
        if(ni >= 0 && nj >= 0 && ni < n && nj < m){
            fp[ni][nj] ^= 1;
        }
    }
}

int dfs(int k){
    memcpy(fp, mp, sizeof fp);
    memset(vis, 0, sizeof vis);
    int tot = m-1,res = 0;
    while(k || tot>=0){
        int t = k%2;
        vis[0][tot] = t;
        res+=t;
        if(t == 1)
            change(0,tot);
        tot--;
        k /= 2;
    }
    //cout  << setw(3) << setfill('0') << *fp[0] << endl;

    for(int i = 1; i < n; i++){
        for(int j = 0; j < m; j++){
            if(fp[i-1][j] == 1){
                vis[i][j] = 1;
                res++;
                change(i,j);
            }
        }
    }
    for(int j = 0; j < m; j++){
        if(fp[n-1][j] == 1)
            return 15*15*2;
    }
    return res;
}

class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        n = mat.size();
        if (n != 0) {
            m = mat[0].size();
        }
        else {
            return 0;
        }
        int ans[20][20];
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                mp[i][j] = mat[i][j];
            }
        }
        int num = pow(2,m),minn = INT_MAX;
        for(int i = 0; i < num; i++){
            int ret = dfs(i);
            if(ret < minn){
                memcpy(ans, vis, sizeof vis);
                minn = ret;
            }
        }
        if (minn == 15*15*2) {
            return -1;
        } else {

            int cnt = 0;
           for(int i = 0; i < n; i++){
                for(int j = 0; j < m; j++){
                    if (ans[i][j] == 1) cnt++;
                }
            }
            return cnt;
        }
    }
};
