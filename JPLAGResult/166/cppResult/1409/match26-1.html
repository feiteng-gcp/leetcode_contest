class Solution {
public:

    int go[5][2] = {{-1,0}, {1,0}, {0,1}, {0,-1}, {0,0}};
    int n,m;
    int minFlips(vector<vector<int>>& mat) {
        set<vector<vector<int>>> visited;
        n = mat.size();
        m = mat[0].size();

        int cnt = 0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(mat[i][j]==0)
                    cnt++;
            }
        }
        queue<pair<vector<vector<int>>, int>> q;
        q.push({mat, cnt});
        int step = 0;
        while(!q.empty()){
            int size = q.size();
            for(int i=0; i<size; i++){
                vector<vector<int>> cur = q.front().first;
                int cnt = q.front().second;
                q.pop();
                if(cnt==m*n)
                    return step;

                for(int i=0; i<n; i++){
                    for(int j=0; j<m; j++){
                        vector<vector<int>> next = cur;
                        int tmp = cnt;
                        int newcnt = flip(next, i, j, tmp);
                        if(visited.count(next)>0)
                            continue;
                        visited.insert(next);
                        q.push({next, newcnt});
                        // cout<<"next="<<endl;
                        // for(int i=0; i<n; i++){
                        //     for(int j=0; j<m; j++)
                        //         cout<<next[i][j]<<" ";
                        //     cout<<endl;
                        // }
                        // cout<<"cnt="<<newcnt<<endl;
                    }
                }
            }
            step++;
        }
        return -1;
    }
    

    int flip(vector<vector<int>>& mat, int i, int j, int cnt){
        for(int k=0; k<5; k++){
            int newi = i + go[k][0];
            int newj = j + go[k][1];
            if(newi<0 || newi>=n || newj<0 || newj>=m)
                continue;
            if(mat[newi][newj]==1)
                cnt++;
            else
                cnt--;
            mat[newi][newj] = 1 - mat[newi][newj];
        }
        return cnt;
    }
};
