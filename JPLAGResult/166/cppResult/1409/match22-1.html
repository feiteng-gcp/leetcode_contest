class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int ret = 10;

        for (int i = 0; i < 1 << n * m; i++) {

            vector<vector<int> > mt = mat;
            for (int j = 0; j < n * m; j++) {
                if (i >> j & 1) {
                    int x = j / m;
                    int y = j % m;
                    for (int u = 0; u < n; u++) {
                        for (int v = 0; v < m; v++) {
                            if (abs(u - x) + abs(v - y) <= 1) {
                                mt[u][v] ^= 1;
                            }
                        }

                    }
                }
            }

            bool f = true;
            for (int u = 0; u < n; u++) {
                for (int v = 0; v < m; v++) {
                    if (mt[u][v] == 1) {
                        f = false;
                    }
                }
            }

            if (f) {
                int cnt = 0;
                for (int j = 0; j < n * m; j++) {
                    if (i >> j & 1) {
                        cnt++;
                    }
                }

                ret = min(ret, cnt);
            }
        }

        return ret == 10 ? -1 : ret;
    }
};
