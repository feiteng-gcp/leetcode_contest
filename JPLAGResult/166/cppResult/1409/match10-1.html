class Solution {
public:
    
    struct Node {
        int dist;
        vector<vector<int>> mat;
        
        Node(int dist, const vector<vector<int>>& mat) : dist(dist), mat(mat) {
            
        }
        
        bool all_zero() const {
            int n = mat.size();

            int m = mat[0].size();
            for(int i = 0; i < n; i ++){
                for(int j = 0; j < m; j ++){
                    if(mat[i][j]) return false;
                }
            }
            return true;
        }
    };
    
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size(), m = mat[0].size();
        set< vector<vector<int>>  > visit;
        int dx[4] = {1, -1, 0, 0}, dy[4] = {0, 0, -1, 1};
        queue<Node> q;
        q.push(Node(0, mat));
        visit.insert(mat);
        while(!q.empty()){
            
            Node now = q.front(); q.pop();

            if (now.all_zero()){
                return now.dist;
            }
            
            for(int i = 0; i < n; i ++){
                for(int j = 0; j < m; j ++){
                    vector<vector<int>> next = now.mat;
                    next[i][j] = 1 - next[i][j];
                    for(int k = 0; k < 4; k ++){
                        int ni = i + dx[k];
                        int nj = j + dy[k];
                        if (ni >= 0 && ni < n && nj >= 0 && nj < m){
                            next[ni][nj] = 1 - next[ni][nj];
                        }
                    }
                    if (visit.count(next) == 0){
                        q.push( Node(now.dist + 1, next) );
                        visit.insert(next);
                    }
                }
            }
            
        }
        return -1;
    }
};
