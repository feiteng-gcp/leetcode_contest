class Solution {
public:
    int hash(vector<vector<int>>& mat)
    {
        int row = mat.size();
        int col = mat[0].size();
        int ret = 0;
        for (int i = 0; i < row; i++)
            for (int j = 0; j < col; j++)
            {
                ret <<= 1;
                ret |= mat[i][j];
            }
        return ret;
    }
    
    bool check(vector<vector<int>>& mat)
    {
        for (int i = 0; i < mat.size(); i++)
            for (int j = 0; j < mat[0].size(); j++)
                if (mat[i][j])
                    return false;
        return true;
    }
    void rev(vector<vector<int>>& src, vector<vector<int>>& dest, int i, int j)
    {

        int rows = src.size();
        int cols = src[0].size();
        
        dest[i][j] = !dest[i][j];
        if (i)
            dest[i - 1][j] = !dest[i - 1][j];
        if (i != rows - 1)
            dest[i + 1][j] = !dest[i + 1][j];
        if (j)
            dest[i][j - 1] = !dest[i][j - 1];
        if (j != cols - 1)
            dest[i][j+1]  = !dest[i][j+1];
    }
    
    int minFlips(vector<vector<int>>& mat) {
        queue<vector<vector<int>> >que;
        que.push(mat);
        unordered_set<int> sset;
        int step = 0;
        while (que.size())
        {
            int size = que.size();
            for (int i = 0; i < size; i++)
            {
                vector<vector<int>> cur = que.front();
                que.pop();
                if (check(cur))
                    return step;
                for (int j = 0; j < cur.size(); j++)
                    for (int k = 0; k < cur[0].size(); k++)
                    {
                        vector<vector<int>> dest(cur);
                        rev(cur, dest, j , k);
                        int h = hash(dest);
                        if (sset.find(h) == sset.end())
                        {
                            sset.insert(h);
                            que.push(dest);
                        }
                    }
            }
            step++;
        }
        return -1;
    }
};
