const int maxn = 5; 
int dir[5][2] = {1,0,0,1,0,0,-1,0,0,-1};
int flip[maxn][maxn],opt[maxn][maxn];
int tile[maxn][maxn],n,m;//
    
class Solution {
public:
    
    void init(vector<vector<int>>& mat) {
        for(int i=0;i<m;i++)
            for(int j=0;j<n;j++)
                tile[i][j] = mat[i][j];
    }
    bool check(int x,int y) {
        if(x < 0 || y < 0 || x >= m || y >= n)
            return true;
        return false;
    }
 
    bool get(int x,int y) {
        int cnt = tile[x][y];
        for(int i=0;i<5;i++) {
            int xx = x + dir[i][0];
            int yy = y + dir[i][1];

            if(check(xx,yy))
                continue;
            cnt += flip[xx][yy];
        }
        if(cnt%2)
            return true;
        return false;
    }
 
    int cal() {
        int cnt = 0;
        for(int i=1;i<m;i++) {
            for(int j=0;j<n;j++) {
                if(get(i-1,j))
                    flip[i][j] = 1;
            }
        }
        for(int i=0;i<n;i++) {
            if(get(m-1,i))
                return -1;
        }

        for(int i=0;i<m;i++)
            for(int j=0;j<n;j++) {
                if(flip[i][j])
                    cnt++;
            }
        return cnt;
    }

   
    int minFlips(vector<vector<int>>& mat) {
        m = mat.size(), n = mat[0].size();
        
        int cnt =0 ;
        for (int i = 0;i < m;++i)
            for (int j = 0;j < n;++j)
                if (mat[i][j]==0) ++cnt;
        if (cnt == m*n) return 0;
        
        init(mat);
        int res = INT_MAX;
        for(int i=0;i<(1<<n);i++) {
            memset(flip,0,sizeof(flip));
            for(int j=n-1;j>=0;j--)
                flip[0][j] = (i >> j) & 1;
            int num = cal();
            if(num > 0 && res > num) {
                res = num;
                memcpy(opt,flip,sizeof(flip));
            }
        }
        if (res == INT_MAX){
            return -1;
        }
        else{
            return res;
        }
    }
};
