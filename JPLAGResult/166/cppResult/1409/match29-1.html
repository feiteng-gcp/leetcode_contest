class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int S = n * m;
        int ans = 1e9;
        int dx[5] = {1, 0, -1, 0, 0};
        int dy[5] = {0, 1, 0, -1, 0};
        for(int i = 0;i < (1 << S);i++){
            int tmp = __builtin_popcount(i);
            bool flag = true;
            vector<vector<int>> a = mat;

            for(int j = 0;j < n;j++){
                for(int k = 0;k < m;k++){
                    int id = j * m + k;
                    if(i & (1 << id)){

                        for(int l = 0;l < 5;l++){
                            int nx = j + dx[l];
                            int ny = k + dy[l];
                            if(nx >= 0 and nx < n and ny >= 0 and ny < m){
                                a[nx][ny] ^= 1;
                            }
                        }
                    }
                }
            }
            for(int j = 0;j < n;j++){
                for(int k = 0;k < m;k++){
                    if(a[j][k] != 0)
                        flag = false;
                }
            }
            if(flag){
                ans = min(ans, tmp);
            }
        }
        if(ans == 1e9)
            return -1;
        return ans;
    }
};
