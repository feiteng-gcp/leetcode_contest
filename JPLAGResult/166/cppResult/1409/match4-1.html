class Solution {
public:
    int minFlips(vector<vector<int>>& mat) {
        set<string> state;
        state.insert("000000000");
        queue<pair<string, vector<vector<int>>>> q;
        q.push(make_pair("000000000", mat));
        while (q.size() > 0) {
            pair<string, vector<vector<int>>> cur = q.front();
            q.pop();
            

            bool done = true;
            for (int i=0; i<cur.second.size(); i++) {
                for (int j=0; j<cur.second[i].size(); j++) {
                    if (cur.second[i][j] == 1) {
                        done = false;
                    }
                }
            }

            // cout << cur.first << endl;
            // for (int i=0; i<cur.second.size(); i++) {
            //     for (int j=0; j<cur.second[i].size(); j++) {
            //         cout <<cur.second[i][j] << " ";
            //     }
            //     cout << endl;
            // }

            if (done) {
                return count(cur.first.begin(), cur.first.end(), '1');
            }
            
            for (int i=0; i<mat.size(); i++) {
                for (int j=0; j<mat[i].size(); j++) {
                    if (cur.first[i*3 + j] == '1') continue;
                    

                    vector<vector<int>> cmat(cur.second.begin(), cur.second.end());
                    cmat[i][j] = (cmat[i][j] + 1) % 2;
                    if (i-1 >= 0) {
                        cmat[i-1][j] = (cmat[i-1][j] + 1) % 2;
                    }
                    if (j-1 >= 0) {
                        cmat[i][j-1] = (cmat[i][j-1] + 1) % 2;
                    }
                    if (i+1 < cmat.size()) {
                        cmat[i+1][j] = (cmat[i+1][j] + 1) % 2;
                    }
                    if (j+1 < cmat[i].size()) {
                        cmat[i][j+1] = (cmat[i][j+1] + 1) % 2;
                    }
                    
                    string nxt = cur.first;
                    nxt[i*3 + j] = '1';
                    q.push(make_pair(nxt, cmat));
                }
            }
        }
        
        return -1;
    }
};
