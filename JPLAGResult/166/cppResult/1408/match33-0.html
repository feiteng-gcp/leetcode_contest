class Solution {
public:
  int smallestDivisor(vector<int>& nums, int threshold) {
    int l = 0;
    int r = 1e6 + 10;
    
    while (l + 1 < r) {
      int m = (l + r) / 2;
      
      int sum = 0;
      for (int i = 0; i < nums.size(); i++) {
        sum += (nums[i] + m - 1) / m;
      }
      
      if (sum > threshold) {
        l = m;
      } else {
        r = m;
      }
    }
    
    return l + 1;
  }
};
