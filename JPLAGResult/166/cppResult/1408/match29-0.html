class Solution {
public:
    int smallestDivisor(vector<int>& v, int t) {
        int low = 1;
        int high = 1123456;
        while (low < high) {
            int m = (low + high) / 2;
            int s = 0;
            for (int i = 0; i < v.size(); i++) {
                s += v[i] / m;
                if (v[i] % m) s++;
                if (s > t) s = t + 1;
            }
            //cout << m << " " << s << endl;
            if (s > t) {
                low = m + 1;
            } else {
                high = m;
            }
            //cout << low << " " << high << endl;
        }
        return low;
    }
};
