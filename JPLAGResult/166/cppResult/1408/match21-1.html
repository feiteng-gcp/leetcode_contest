class Solution {
public:
    int smallestDivisor(vector<int>& a, int m) {
        // cout << ceil(3.0) << '\n';
        int l = 1, r = 1000000, mid;
        while (l != r) {

            mid = (l + r) >> 1;
            int tmp = 0;
            for (auto it: a) tmp += ceil(it * 1.0 / mid);
            // printf("mid= %d %d\n", mid, tmp);
            if (tmp <= m) r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
