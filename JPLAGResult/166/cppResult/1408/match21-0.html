class Solution {
public:
  int smallestDivisor(vector<int>& nums, int threshold) {
    int lo = 1, hi = 0x3f3f3f3f;
    while (lo < hi) {
      int mid = (lo + hi) / 2;
      int tot = 0;
      for (auto n: nums) tot += (n + mid - 1) / mid;
      if (tot > threshold) lo = mid + 1;
      else hi = mid;
    }
    return hi;
  }
};
