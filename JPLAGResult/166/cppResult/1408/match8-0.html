class Solution {
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int l = 1;
        int h = 1e6 + 1;
        while (l < h) {
            int mid = l + (h-l)/2;
            int s = 0;
            for (auto i : nums) {
                s += ceil(i*1.0/mid);
            }
            if (s > threshold) {
                l = mid + 1;
            } else {
                h = mid;
            }
        }
        return l;
    }
};
