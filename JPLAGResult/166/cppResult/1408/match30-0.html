class Solution {
    int helper(vector<int>& nums, int divisor) {
        int res = 0;
        for (int num: nums) {
            res += ceil(((float)num) / divisor);
        }
        return res;
    }
    
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int left = 1, right = 1000000;
        while (left < right) {
            if (left == right-1) {
                return helper(nums, left) <= threshold ? left : right;
            }
            int middle = left + (right - left) / 2;
            int result = helper(nums, middle);
            //cout << "try " << middle << ", result = " << result << endl;
            if (result > threshold) {
                left = middle;
            } else {
                right = middle;
            }
        }
        return left;
    }
};
