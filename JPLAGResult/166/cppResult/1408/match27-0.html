class Solution {
public:
    
    int calc(vector<int>& nums, int d) {
        int sum = 0;
        for (int n : nums) {
            sum += n / d;
            sum += (n % d) != 0;
        }
        return sum;
    }
    
    int smallestDivisor(vector<int>& nums, int threshold) {
        int low = 0, high = 1000001;
        while (low + 1 < high) {
            int mid = (low + high) / 2;
            int val = calc(nums, mid);
            
            if (val <= threshold) {
                high = mid;
            } else {
                low = mid;
            }
        }
        return high;
    }
};
