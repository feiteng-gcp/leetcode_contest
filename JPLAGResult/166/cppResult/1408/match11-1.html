class Solution {
public:
    int div(int a, int b){
        return (a + (b-1)) / b;
    }
    int smallestDivisor(vector<int>& nums, int threshold) {
        int l = 1, r = 1e7;
        while(l <= r){
            int m = l + (r-l)/2;
            int s = 0;
            for(int a : nums){
                s += div(a, m);
            }
            
            //cout<<m<<" "<<s<<endl;
            
            if(s <= threshold) r = m - 1;
            else l = m + 1;
        }
        
        return l;
    }
};
