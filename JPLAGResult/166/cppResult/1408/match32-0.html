class Solution {
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int lo = 1;
        int hi = *max_element(nums.begin(), nums.end());
        while (lo < hi) {
            int md = lo + (hi - lo) / 2;
            int sum = 0;
            for (auto x : nums) {
                sum += (x + md - 1) / md;
            }
            if (sum <= threshold) {
                hi = md;
            } else {
                lo = md + 1;
            }
        }
        return hi;
    }
};
