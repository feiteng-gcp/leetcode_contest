class Solution {
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int l = 0;
        int r = *max_element(nums.begin(), nums.end()) + 1;
        while (l + 1 < r) {
            int m = (l + r) / 2;
            int sum = 0;
            for (int &num : nums) {
                sum += ((num + m - 1) / m);
            }
            if (sum <= threshold) {
                r = m;
            }
            else {
                l = m;
            }
        }
        return r;
    }
};
