class Solution {
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int low = 1, high = 1000000;
        while (low < high) {
            int mid = (low + high) >> 1;
            long long sum = 0;
            for (auto it : nums) {
                sum += (it + mid - 1) / mid;
            }
            if (sum <= threshold) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
};
