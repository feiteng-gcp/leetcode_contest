class Solution {
public:
    int smallestDivisor(vector<int>& x, int threshold) {
        int lo = 1, hi = 100000000;
        while (lo < hi) {
            int mi = (lo + hi) / 2;

            long long sum = 0;
            for (int i : x)
                sum += (i + mi - 1) / mi;
            
            if (sum > threshold)
                lo = mi + 1;
            else
                hi = mi;
        }
        return lo;
    }
};
