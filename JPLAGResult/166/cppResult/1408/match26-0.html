class Solution {
public:
    int smallestDivisor(vector<int>& nums, int t) {
        int left = 1, right = INT_MAX;
        while (left < right){
            int mid = left + (right - left) / 2;
           // cout << mid << endl;
            int c = 0;
            for (int x : nums){
                c += x / mid;
                if (x % mid != 0)
                    c++;
            }
            if (c <= t)
                right = mid;
            else
                left = mid + 1;
           // cout << left << " " << right << endl;
        }
        return left;
    }
};
