class Solution {
public:
    bool check(vector<int>& N, int T, int d) {
        int s = 0;
        for (int n : N) {
            s += (n+d-1)/d;

        }
        return s <= T;
    }
    
    
    int smallestDivisor(vector<int>& nums, int threshold) {
        int H = 1000005, L = 0;
        while (H - L > 1) {
            int M = (H+L) / 2;
            if (check(nums, threshold, M)) {
                H = M;
            } else {
                L = M;
            }
        }
        return H;
    }
};
