class Solution {
public:
    int smallestDivisor(vector<int>& a, int b) {
        int lo = 1, hi = 1000000;
        while(lo<hi) {
            int mid=lo+hi>>1;
            int tmp=0;
            for(int i:a) tmp+=(i+mid-1)/mid;
            if(tmp <= b) hi = mid;
            else lo = mid + 1;
        }
        return lo;
    }
};
