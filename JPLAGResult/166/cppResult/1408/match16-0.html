class Solution {
public:
    int check(double k, vector<int>& nums){
        int sum = 0;
        for (int i : nums){
            sum += ceil(i / k);
        }
        return sum;
    }
    
    int smallestDivisor(vector<int>& nums, int threshold) {
        int n = nums.size();
        int left = 1, right = 5e8;
        while (left <= right){
            int mid = left + (right-left) / 2;
            int ret = check(mid, nums);
            if (ret <= threshold){
                right = mid-1;
            }
            else{
                left = mid+1;
            }
        }
        
        return left;
    }
};
