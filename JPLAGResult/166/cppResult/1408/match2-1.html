class Solution {
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int l = 1, r = 1001000;
        int ans = 1001000;
        while (l <= r) {
            int mid = (l + r)/2;
            
            int sum = 0;
            for (int i=0;i<nums.size();i++) {
                sum += (nums[i] + mid - 1) / mid;
            }
            if (sum <= threshold) {
                r = mid - 1;
                ans = mid;
            } else {
                l = mid + 1;
            }
        }
        return ans;
    }
};
