class Solution {
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int l = 0, r = 1e6+1;
        while (r - l > 1) {
            int mid = (l + r) / 2;

            long long s = 0;
            for (int a : nums) {

                s += (a - 1) / mid + 1;
            }

            if (s > threshold) {
                l = mid;
            } else {
                r = mid;
            }
        }

        return r;
    }
};
