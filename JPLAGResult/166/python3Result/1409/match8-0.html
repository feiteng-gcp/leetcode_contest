class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        #print(((1,1)(0,0)) == ((1,1)(0,0)))
        
        m = len(mat)
        n = len(mat[0])
        
        z = [[0 for i in range(n)] for j in range(m)]
        zt = tuple(tuple(zz) for zz in z)
        
        def ttol(t):
            return [list(tt) for tt in t]
        def ltot(l):
            return tuple(tuple(ll) for ll in l)
        
        mt = ltot(mat)
        
        s = set([zt])
        news = set([zt])
        step = 0
        while len(news) > 0:
            if mt in s:
                return step
            step += 1
            newnews = set()
            for t in news:
                
                for i in range(m):
                    for j in range(n):
                        l = ttol(t)
                        l[i][j] = 1 - l[i][j]
                        if i + 1 < m:
                            l[i+1][j] = 1 - l[i+1][j]
                        if i -1 >= 0:
                            l[i-1][j] = 1 - l[i-1][j]
                        if j + 1 < n:
                            l[i][j+1] = 1 - l[i][j+1]
                        if j - 1 >= 0:
                            l[i][j-1] = 1 - l[i][j-1]
                        tp = ltot(l)
                        if tp not in s:
                            newnews.add(tp)
                            if mt in s:
                                return step
            s = s.union(newnews)
            news = newnews
            
        return -1
