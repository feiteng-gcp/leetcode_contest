import queue

class Solution:

            
    
    def minFlips(self, mat: List[List[int]]) -> int:
        
        q = queue.Queue()
        q.put(mat)
        p = {q: None}
        
        while not q.empty():
            current = q.get()
            for r in range(len(current)):
                for c in range(len(current[0])):
                    test = [row.copy() for row in current]
                    test[r][c] = 1 if test[r][c] == 0 else 0
                    
                    if c > 0:
                        test
[r][c - 1] = 1 if test[r][c - 1] == 0 else 0
                    if r > 0:
                        test[r - 1][c] = 1 if test[r - 1][c] == 0 else 0
                    if c < len(test[0]) - 1:
                        test[r][c + 1] = 1 if test[r][c + 1] == 0 else 0
                    if r < len(test) - 1:
                        test[r + 1][c] = 1 if test[r + 1][c] == 0 else 0
                                                
                    if str(test) not in p:
                        q.put(test)
                        p[str(test)] = current

        zero = [[0 for i in range(len(mat[0]))] for j in range(len(mat))]
        if str(zero) not in p:
            return -1
                
        current = zero
        length = 0
        while current != mat:
            current = p[str(current)]
            length += 1
        return length
