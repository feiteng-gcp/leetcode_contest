class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        def check(i,j,n,m):
            if 0<=i<n and 0<=j<m:
                return True
            return False
        def check_mat(mat):
            n,m=len(mat),len(mat[0])
            
for i in range(n):
                for j in range(m):
                    if mat[i][j]==1:
                        return False
            return True
        def flip(mat,i,j):
            if mat[i][j]==1:
                mat[i][j]=0
            else:
                mat[i][j]=1
        def convert(mat):
            s=''
            n,m=len(mat),len(mat[0])
            for i in range(n):
                for j in range(m):
                    s=s+str(mat[i][j])
            return s
        def get(mat):
            if check_mat(mat):
                return 0
            dirs=[[0,0],[0,1],[0,-1],[1,0],[-1,0]]
            q=[[mat,0]]
            n,m=len(mat),len(mat[0])
            vis=collections.defaultdict(int)
            
while q:
                #print(q,'q')
                grid,steps=q.pop(0)
                if check_mat(grid):
                    return steps
                for i in range(n):
                    for j in range(m):
                        temp=[]
                        for _ in range(n):
                            temp.append([k for k in grid[_]])
                        for a,b in dirs:
                            if check(a+i,b+j,n,m):
                                flip(temp,a+i,b+j)
                        y=convert(temp)
                        if y not in vis:
                            q.append([temp,steps+1])
                            vis[y]=1
            return -1
        return get(mat)
