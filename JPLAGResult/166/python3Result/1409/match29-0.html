class Solution:
    def minFlips(self, mat: 'List[List[int]]') -> int:
        arr=[mat]
        ss=sum(sum(row) for row in mat)
        if ss==0:
            return 0
        steps=[0]
        lr,lc=len(mat),len(mat[0])
        ans=float('inf')
        for i in range(lr):
            for j in range(lc):
                #switch
                curl=len(arr)
                for k in range(curl):
                    tmp=[row.copy() for row in arr[k]]
                    tmp[i][j]=1-tmp[i][j]
                    if i>=1:
                        tmp[i-1][j]=1-tmp[i-1][j]
                    if j>=1:
                        tmp[i][j-1]=1-tmp[i][j-1]
                    if i+1<lr:
                        tmp[i+1][j]=1-tmp[i+1][j]
                    if j+1<lc:
                        tmp[i][j+1]=1-tmp[i][j+1]
                    arr.append(tmp)
                    steps.append(steps[k]+1)
                    #if same
                    ss=sum(sum(row) for row in tmp)
                    if ss==0:
                        ans=min(ans,steps[-1])
        if ans==float('inf'):
            ans=-1
        return ans
