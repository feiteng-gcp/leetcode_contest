import copy

class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        seen = set()
        
        def encode(mat, m, n):
            code = 0
            bit = 1
            for i in range(m):
                for j in range(n):
                    if mat[i][j] == 1:
                        code |= bit
                    bit <<= 1
            return code
        
        def decode(num, m, n):
            mat = [[0]*n for _ in range(m)]
            bit = 1
            for i in range(m):
                for j in range(n):
                    if (num & bit) != 0:
                         mat[i][j] = 1
                    bit <<= 1
            return mat
        
        def flip(ma, i, j, m, n):
            mat = copy.deepcopy(ma)
            mat[i][j] = 0 if ma[i][j] == 1 else 1
            for dx, dy in [[0, 1],[1, 0],[0, -1],[-1, 0]]:
                x 
= i+dx
                y = j+dy

                    mat[x][y] = 0 if ma[x][y] == 1 else 1

        seen.add(encode(mat, m, n))
        step = 0
        
        while q:
            size = len(q)

                current = q.popleft()
                if current == 0:
                    return step
                mat = decode(current, m, n)
                
for i in range(m):
                    for j in range(n):
                        flipped = flip(mat, i, j, m, n)
                        nxt = encode(flipped, m, n)
                        if nxt not in seen:
                            q.append(nxt)
                            seen.add(nxt)
            step += 1
        
        return -1
