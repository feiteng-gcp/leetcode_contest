class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        state = [0] * (m * n)
        for i in range(m):
            for j in range(n):
                state[i * n + j] = mat[i][j]
        if sum(state) == 0:
            return 0
        
        memo = set([tuple(state)])
        q = collections.deque([(tuple(state), 0)])
        while q:
            state, step = q.popleft()
            for i in range(m):
                for j in range
(n):
                    nextState = list(state)
                    nextState[i * n + j] = 1 - nextState[i * n + j]
                    for di, dj in ((0, 1), (0, -1), (1, 0), (-1, 0)):
                        ni, nj = i + di, j + dj
                        if ni >= 0 and ni < m and nj >= 0 and nj < n:
                            nextState[ni * n + nj] = 1 - nextState[ni * n + nj]
                            
                    if sum(nextState) == 0:
                        return step + 1
                    nextStateTuple = tuple(nextState)
                    if nextStateTuple in memo:
                        continue
                    memo.add(nextStateTuple)
                    q.append((nextStateTuple, step + 1))
        
        return -1
