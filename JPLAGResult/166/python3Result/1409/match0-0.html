class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        from copy import deepcopy
        row, col = len(mat), len(mat[0])
        def mat_str(m):
            res = ''
            for r in m:
                for c in r:
                   res += str(c)
            return res
        def flip(m, i, j):
            mm = deepcopy(m)
            to = [(i, j), (i+1, j), (i-1, j), (i, j-1), (i, j+1)]
            for x, y in to:
                if 0 <= x < row and 0 <= y < col:
                    mm[x][y] = 1 - m[x][y]
            return mm
        visited = set()
        origin = mat_str(mat)
        if '1' not in origin:
            return 0
        visited.add(origin)
        step = 1
        queue = [mat]
        while True:
            if len(queue) == 0:
                return -1
            next_queue = []
            for mt in queue:
                for x in range(row):
                    for y in range(col):
                        temp = flip(mt, x, y)
                        temp_str = mat_str(temp)
                        if '1' not in temp_str:
                            return step
                        if temp_str not in visited:
                            visited.add(temp_str)
                            next_queue.append(temp)
            queue = next_queue
            step += 1
