class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        m = len(mat)
        n = len(mat[0])
        direct = [(0, -1), (0, 1), (-1, 0), (1, 0)]

        def is_ans(matrix):
            for i in range(m):
                for j in range(n):
                    if matrix[i][j] != 0:
                        return False
            return True

        def get_tuple(matrix):
            res = []
            for row in matrix:
                res.extend(row)
            return tuple(res)

        def flip(matrix, i, j):
            res = copy.deepcopy(matrix)
            res[i][j] = 1 - matrix[i][j]
            for di, dj in direct:
                row = i + di
                col = j + dj
                if 0 <= row < m and 0 <= col < n:
                    res[row][col] = 1 - matrix[row][col]
            return res

        que = collections.deque()
        que.append((mat, 0))
        seen = set()
        t_mat = get_tuple(mat)
        seen.add(t_mat)
        while que:
            matrix, step = que.popleft()
            if is_ans(matrix):
                return step
            for i in range(m):
                for j in range(n):
                    temp = flip(matrix, i, j)
                    tuple_t = get_tuple(temp)
                    if tuple_t not in seen:
                        seen.add(tuple_t)
                        que.append((temp, step + 1))
        return -1
