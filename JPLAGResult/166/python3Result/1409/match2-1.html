class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        def new_matrix(mat):
            res = [[0] * n for i in range(m)]
            for i in range(m):
                for j in range(n):
                    res[i][j] = mat[i][j]
            return res
        def to_tuple(mat):
            rows = []
            for i in range(m):
                rows.append(tuple(mat[i]))
            return tuple(rows)
        
        def get_next(mat, i, j):
            mat = new_matrix
(mat)
            mat[i][j] = 1 - mat[i][j]
            if i > 0:
                mat[i-1][j] = 1 - mat[i-1][j]
            if i < m - 1:
                mat[i+1][j] = 1 - mat[i+1][j]
            if j > 0:
                mat[i][j-1] = 1 - mat[i][j-1]
            if j < n - 1:
                mat[i][j+1] = 1 - mat[i][j+1]
            return to_tuple(mat)
        
        def target(mat):
            for i in range(m):
                for j in range(n):
                    if mat[i][j] == 1:
                        return False
            return True
                
        q = [to_tuple(mat)]
        visited = set(q)
        step = 0
        while len(q) > 0:
            new_q  = []
            for mat in q:
                if target(mat):
                    return step
                for i in range(m):
                    for j in range(n):
                        next_mat 
= get_next(mat, i, j)
                        if next_mat not in visited:
                            visited.add(next_mat)
                            new_q.append(next_mat)
            step += 1
            q = new_q
        return -1
