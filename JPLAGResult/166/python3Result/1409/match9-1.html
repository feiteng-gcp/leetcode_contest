class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        m, n = len(mat), len(mat[0])
        q = collections.deque()
        visit = set()
        
        def ct(mat):
            return tuple(tuple(x) for x in mat)
        
        c = ct(mat)
        q.append(c)
        visit.add(c)
        
        res = 0
        while q:
            for _ in range(len(q)):
                cur = q.popleft()
                
                if sum(sum(x) for x in cur) == 0:
                    return res
                
                for i in range(m):
                    for j in range(n):
                        temp = list(list
(x) for x in cur)
                        temp[i][j] = 1 - temp[i][j]
                        
                        if 0 <= i-1 <= m-1 and 0 <= j <= n-1:
                            temp[i-1][j] = 1 - temp[i-1][j]
                        if 0 <= i+1 <= m-1 and 0 <= j <= n-1:
                            temp[i+1][j] = 1 - temp[i+1][j]
                        if 0 <= i <= m-1 and 0 <= j-1 <= n-1:
                            temp[i][j-1] = 1 - temp[i][j-1]
                        if 0 <= i <= m-1 and 0 <= j+1 <= n-1:
                            temp[i][j+1] = 1 - temp[i][j+1]
                        
                        temp = ct(temp)
                        
                        if temp not in visit:
                            visit.add(temp)
                            q.append(temp)
            
            res += 1
        
        return -1
