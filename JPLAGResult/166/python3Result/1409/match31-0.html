class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        if not mat or not mat[0]:
            return -1
        row, col = len(mat), len(mat[0])
        
        queue = [(mat, 0), ]
        visited = set()
        def toTuple(matrix):
            temp = []
            for r in range(row):
                for c in range(col):
                    temp.append(matrix[r][c])
            return tuple(temp)
        visited.add(toTuple(mat))

        
        def flip(mat, r, c):
            ans = [row[:] for row in mat]
            for dr, dc in (-1, 0), (1, 0), (0, 0), (0, -1), (0, 1):
                if -1 < r + dr < row and -1 < c + dc < col:
                    ans[r + dr][c + dc] =  int(not ans[r + dr][c + dc])
            return ans
                    
        def checkZero(mat):
            for r in range(row):
                for c in range(col):
                    if mat[r][c] == 1:
                        return False
            return True
        
        while queue:
            matrix, move = queue.pop(0)
            if checkZero(matrix):
                return move
            for r in range(row):
                for c in range(col):
                    temp = flip(matrix, r, c)
                    tu_temp = toTuple(temp)
                    if tu_temp not in visited:
                        queue.append((temp, move + 1))
                        visited.add(tu_temp)
        return -1
