import copy
from collections import deque

class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
    
        def compute_next(mat_orig):
            m, n = len(mat_orig), len(mat_orig[0])
            for i in range(m):
                for j in range(n):
                    mat = copy.deepcopy
(mat_orig)
                    mat[i][j] = 1 - mat[i][j]
                    if i > 0: mat[i - 1][j] = 1 - mat[i - 1][j]
                    if j > 0: mat[i][j - 1] = 1 - mat[i][j - 1]
                    if i < m - 1: mat[i + 1][j] = 1 - mat[i + 1][j]
                    if j < n - 1: mat[i][j + 1] = 1 - mat[i][j + 1]
                    yield mat
        
        queue = deque([[mat, 0]])
        visited = set()
        while queue:
            cur_mat, depth = queue.popleft()
            str_rep = str(cur_mat)
            if str_rep in visited:
                continue
            visited.add(str_rep)
            if sum(sum(r) for r in cur_mat) == 0:
                return depth
            for new_mat in compute_next(cur_mat):
                queue.append([new_mat, depth + 1])
        return -1
