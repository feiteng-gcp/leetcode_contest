class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        seen = {}
        q = [(mat, 0)]
        def invert(mat, i, j):
            nmat= []
            for r in range(len(mat)):
                tmp = []
                for c in  range(len(mat[r])):
                    tmp.append(mat[r][c])
                nmat.append(tmp)
            
            nmat
[i][j] ^= 1
            if i > 0:
                nmat[i-1][j] ^= 1
            if j > 0:
                nmat[i][j-1] ^= 1
            if i+1 < len(mat):
                nmat[i+1][j] ^= 1
            if j+1<len(mat[i]):
                nmat[i][j+1] ^= 1
            return nmat
        cnt = 1
        while q:
            if cnt > 10**6:
                break
            cnt += 1
            cur = q.pop(0)
            cur, steps = cur[0], cur[1]
            # print(cur, steps, seen)
            sm = sum(sum(row) for row in cur)
            if sm == 0:
                return steps
            tpcur = tuple(tuple(row) for row in cur)
            if tpcur in seen:
                continue
            seen[tpcur] = True
            for i in range(len(cur)):
                for j in range(len(cur[i])):
                    nmat = invert(cur, i, j)
                    q.append((nmat, steps+1))
        return -1
