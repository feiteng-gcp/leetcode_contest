class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        n, m = len(mat), len(mat[0])
        ans = 0
        def nei(mat):
            for i, row in enumerate(mat):
                for j, c in enumerate(row):
                    newMat = [[x for x in row] for row in mat]
                    newMat
[i][j] ^= 1
                    if i > 0:
                        newMat[i - 1][j] ^= 1
                    if j > 0:
                        newMat[i][j - 1] ^= 1
                    if i + 1 < n:
                        newMat[i + 1][j] ^= 1
                    if j + 1 < m:
                        newMat[i][j + 1] ^= 1
                    yield newMat
        def state(matrix):
            state = cur = 0
            for row in matrix:
                for col in row:
                    if col:
                        state += 1 << cur
                    cur += 1
            return state
        stack = [mat]
        visited = set([state(mat)])
        while stack:
            new = []
            for cur in stack:
                if not any(any(row) for row in cur):
                    return ans
                for next1 in nei(cur):
                    if state(next1) not in visited:
                        visited.add(state(next1))
                        new.append(next1)
            ans += 1
            stack = new
        return -1
