from collections import deque
class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        
        m, n = len(mat), len(mat[0])
        
        state = ''
        for i, r in enumerate(mat):
            for j, num in enumerate(r):
                state += str(num)
                
        seen = set([state])
        q = deque([(state, 0)])
        while q:
            state, step = q.popleft()
            if all(c == '0' for c in state):
                return step
            
            for pos, c in enumerate(state):
                i, j = divmod(pos, n)
                n_state = list(state)
                n_state[pos] = '0' if state[pos] == '1' else '1'
                for ni, nj in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
                    if 0 <= ni < m and 0 <= nj < n:
                        n_pos = ni * n + nj
                        # print(ni, nj)
                        # print(n_pos)
                        n_state[n_pos] = '0' if state[n_pos] == '1' else '1'
                        
                n_state = ''.join(n_state)
                if n_state not in seen:
                    seen.add(n_state)
                    q.append((n_state, step + 1))
        return -1
