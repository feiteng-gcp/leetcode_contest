class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        '''
        search. BFS. max # of state: 2^9. 
        '''
        m, n = len(mat), len(mat[0])
        
        def to_state(mat):
            s = [0] * 9
            for i in range(m):
                for j in range(n):
                    s[i * 3 + j] = mat[i][j]
            return int(''.join([str(d) for d in s]))
        
        
def all_zero(mat):
            for i in range(m):
                for j in range(n):
                    if mat[i][j] != 0:
                        return False
            return True
        
        neighbours = [
            (0, 0),
            (0, 1),
            (0, -1),
            (1, 0),
            (-1, 0),
        ]
        
        if all_zero(mat):
            return 0
        
        visited = set()
        from collections import deque
        q = deque()
        q.append((mat, 0))
        visited.add(to_state(mat))
        from copy import deepcopy
        while q:
            u, l = q.popleft()
            for i in range(m):
                for j in range(n):
                    v = deepcopy(u)
                    for di, dj in neighbours:
                        ii = i + di
                        jj = j + dj
                        if 0 <= ii < m and 0 <= jj < n:
                            v[ii][jj] = 1 - v[ii][jj]
                    if all_zero(v):
                        return l + 1
                    s 
= to_state(v)
                    if s not in visited:
                        visited.add(s)
                        q.append((v, l + 1))
        return -1
