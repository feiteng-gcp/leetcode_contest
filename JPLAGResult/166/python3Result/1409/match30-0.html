class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        n,m=len(mat),len(mat[0])
        zero,ans=[[0]*m]*n,999
        for x in range(2**(n*m)):
            temp,cnt=[s[:] for s in mat],0
            for i in range(n):
                for j in range(m):
                    if (x>>(i*m+j))&1:
                        temp[i][j]^=1
                        if i>0:temp[i-1][j]^=1
                        if j>0:temp[i][j-1]^=1
                        if i<n-1:temp[i+1][j]^=1
                        if j<m-1:temp[i][j+1]^=1
                        cnt+=1
            if temp==zero:ans=min(ans,cnt)
        return -1 if ans==999 else ans
