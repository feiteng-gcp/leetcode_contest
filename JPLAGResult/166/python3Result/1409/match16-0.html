class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:

        
        seen = set()
        count = 0
        curr = {tuple(tuple(row) for row in mat)}
        while curr:
            new_curr = set()
            
            for grid in curr:
                if grid in seen: continue
                if all(all(r == 0 for r in row) for row in grid): return count
                seen.add(grid)

                out = float('Inf')
                for i in range(len(grid)):
                    for j in range(len(grid[0])):
                        temp = [list(row) for row in grid]
                        temp[i][j] = (temp[i][j] +1)%2
                        if i >= 1:
                            temp[i-1][j] = (temp[i-1][j] +1)%2
                        if i < len(grid)-1:
                            temp[i+1][j] = (temp[i+1][j] +1)%2
                        if j >= 1:
                            temp[i][j-1] = (temp[i][j-1] +1)%2
                        if j < len(grid[0])-1:
                            temp[i][j+1] = (temp[i][j+1] +1)%2
                        new_curr.add(tuple(tuple(row) for row in temp))
            count += 1
            curr = new_curr

            
        return -1
