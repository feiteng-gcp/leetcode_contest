class Solution:
    def minFlips(self, mat: List[List[int]]) -> int:
        seen = set()
        target = "0" * (len(mat) * len(mat[0]))
        q = [mat]
        
        
        
        def to_s(m):
            res = ""
            for i in m:
                res += "".join(map(str, i))
            return res
        
        def get_candidate(m):
            res = []
            
for i in range(len(m)):
                for j in range(len(m[0])):
                    copy = [list(r) for r in m]
                    copy[i][j] = 1- copy[i][j]
                    if i + 1 < len(m):
                        copy[i+1][j] = 1- copy[i+1][j]
                    if i - 1 >= 0:
                        copy
[i-1][j] = 1- copy[i-1][j]
                        
                    if j- 1 >= 0:
                        copy[i][j-1] = 1- copy[i][j-1]
                        
                    if j + 1 < len(m[0]):
                        copy[i][j+1] = 1- copy[i][j+1]
                    
                    res.append(copy)
            return res
        
        seen.add(to_s(mat))
        count = 0
        while q:
            next_q = []
            for i in q:
                if to_s(i) == target:
                    return count
                candi = get_candidate(i)
                
                for c in candi:
                    li = to_s(c)
                    
                    if li in seen:
                        continue
                    else:
                        seen.add(li)
                        next_q.append(c)
            
            q = next_q
            count += 1
            
        return -1
