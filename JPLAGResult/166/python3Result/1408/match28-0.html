class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        _max = max(nums)
        n = len(nums)
        l=1
        r = math.ceil(_max/(threshold/n))+1
        #print(l,r)
        def getsum(d):
            return sum(math.ceil(x/d) for x in nums)
        
        while l<r:
            mid = (l+r)//2
            if getsum(mid)>threshold:
                l = mid+1
            else:
                r = mid
        return l
