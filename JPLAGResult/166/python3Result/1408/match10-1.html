class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        
        def cur(x):
            res = 0
            for num in nums:
                res += math.ceil(num / x)
            return res
        
        max_ = max(nums)
        l, r 
= 1, max_
        while l < r:
            mid = (l + r) // 2
            if cur(mid) > threshold:
                l = mid + 1
            else:
                r = mid
        return l
