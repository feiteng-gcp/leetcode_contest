class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        left, right = 1, max(nums)
        while left < right:
            mid = (left + right) // 2
            r = self.getRes(nums, mid)
            if r > threshold:
                left = mid + 1
            else:
                right = mid
        
        return left
            
    def getRes(self, nums, d):
        s = 0
        for n in nums:
            s += n // d if n % d == 0 else (n // d + 1)
        return s
