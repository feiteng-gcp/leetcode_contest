class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        
        left, right 
= 1, max(nums)+1
        while left <= right:
            middle = (left+right)//2
            if sum([(a+middle-1)//middle for a in nums]) <= threshold:
                right = middle - 1
            else:
                left = middle+1
        
        return left
