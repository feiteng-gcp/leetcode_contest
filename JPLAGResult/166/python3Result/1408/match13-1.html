class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        '''
        The bigger the divisor, the less the result,
        try binary search
        '''
        def compute(divisor):
            return sum(math.ceil(n / divisor) for n in nums)
            
        l, h = 1, max(nums)
        while l < h:
            m = (l + h) // 2
            if compute(m) > threshold:
                # m is too small
                l = m + 1
            else:
                h = m
        return h
