import math

class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def get_sum(d):
            return sum(math.ceil(n/d) for n in nums)
        #
        lo, hi = 1, max(nums)
        while lo < hi:
            mid = (lo + hi) // 2
            if get_sum(mid) <= threshold:
                hi = mid
            else:
                lo = mid+1
        return lo
