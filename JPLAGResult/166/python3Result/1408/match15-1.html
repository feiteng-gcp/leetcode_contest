class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def get_value(nums, d):
            ret = 0
            for i in nums:
                if i % d == 0:
                    ret += i // d
                else:
                    ret += (i // d) + 1
            return int(ret)
        
        lo = 1
        hi 
= max(nums)
        while lo < hi:
            m = (lo + hi) // 2
            if get_value(nums, m) > threshold:
                lo = m + 1
            else:
                hi = m
        return hi
