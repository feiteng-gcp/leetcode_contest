class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        low = 1
        high = max(nums)

        def judge_ge(tar):
            s = 0
            for val in nums:
                s += math.ceil(val / tar)
            return s > threshold

        while low <= high:
            mid = (low + high) >> 1
            if judge_ge(mid):
                low = mid + 1
            else:
                high = mid - 1
        return low
