class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        left, right 
= 1, max(nums)
        while left < right:
            total = 0
            mid = left + (right - left) // 2
            for num in nums:
                total += math.ceil(num / mid)
            if total > threshold:
                left = mid + 1
            else:
                right = mid
        return left
