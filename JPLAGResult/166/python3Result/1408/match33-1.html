class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def cnt(mid):
            res=0
            for num in nums:
                if num%mid==0:
                    res+=num//mid
                else:
                    res+=(num//mid)+1
            return res
        low,high
=1,int(1e6)+5
        while low<=high:
            mid=(low+high)>>1
            if cnt(mid)<=threshold:
                high=mid-1
            else:
                low=mid+1
        return low
