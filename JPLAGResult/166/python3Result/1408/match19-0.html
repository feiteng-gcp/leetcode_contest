import math
class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        low = 1
        high = max(nums)
        while low <= high:
            mid = low + (high - low )// 2
            cur = 0
            for num in nums:
                cur += math.ceil(num/mid)
            # print(mid, cur)
            if cur <= threshold:
                high = mid - 1
            else:
                low = mid + 1
        return low
