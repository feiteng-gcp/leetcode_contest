class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def can(k):
            return sum((x+k-1)//k for x in nums) <= threshold
        lo,hi=1,sum(nums)
        while lo<hi:
            mid=(lo+hi)//2
            if can(mid): hi=mid
            else: lo=mid+1
        return lo
