class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        l, r = 1, 10 ** 6
        
        def valid(m, threshold):
            res = 0
            
            for x in nums:
                res += (x // m + 1) if x % m != 0 else x // m
            
            return res <= threshold
        
        while l < r:
            m = (l + r) // 2
            if valid(m, threshold):
                r = m
            else:
                l = m + 1
        
        return l
