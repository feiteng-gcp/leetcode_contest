class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        l = 1
        r = max(nums)
        while l < r:
            mid = (l+r)//2
            s = 0
            for i in nums:
                s += math.ceil(i/mid)
            if s > threshold:
                l = mid + 1
            else:
                r = mid
        return l
