class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def get(k):
            res = 0
            for num in nums:
                res += (num - 1)//k + 1
            return res
        
        p, q 
= 1, max(nums)
        while p < q:
            mid = (p + q) >> 1
            res = get(mid)
            if res <= threshold:
                q = mid
            else:
                p = mid + 1
        return q
