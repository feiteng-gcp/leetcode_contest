class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        
        
        
        if threshold == len(nums):
            return max(nums)
        
        def divsum(nums, m):
            res = 0
            for num in nums:
                res += num // m
                if num % m != 0:
                    res += 1
            return res
        
        s, e = 1, max(nums)
        while s < e:
            mid = (s+e) // 2
            res = divsum(nums, mid)
            if res > threshold:
                s = mid + 1
            else:
                e = mid
        return s
