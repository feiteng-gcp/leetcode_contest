import math


class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        
        def valid(nums, divisor, threshold):
            s = 0
            for num in nums:
                s += math.ceil(num/divisor)
            return s <= threshold
        
        l, r = 1, max(nums)
        
        while l+1 < r:
            mid = (l+r)//2
            if valid(nums, mid, threshold):
                r = mid
            else:
                l = mid
        if valid(nums, l, threshold):
            return l
        return r
