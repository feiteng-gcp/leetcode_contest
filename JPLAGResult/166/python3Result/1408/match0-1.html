from math import ceil
class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def possible(d):
            return sum(ceil(x / d) for x in nums) <= threshold
        
        low, high = 1, max(nums)
        while low < high:
            mid = (low + high) // 2
            if possible(mid):
                high = mid
            else:
                low = mid + 1
        return low
