class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        l,r = 1, sum(nums)
        
def f(x):
            res = 0
            for j in nums:
                res += j // x if j % x == 0 else j // x + 1
            return res > threshold
        while r > l:
            mid = (l + r) // 2
            if f(mid):
                l = mid + 1
            else:
                r = mid
        return l
