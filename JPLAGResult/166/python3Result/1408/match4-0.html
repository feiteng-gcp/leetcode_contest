class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        import math
        def get_sum(n):
            s = 0
            for i in nums:
                s += math.ceil(i / n)
            return s
        lo, hi = 1, 1000000
        while lo <= hi:
            mid = (hi - lo) // 2 + lo
            x = get_sum(mid)
            if x > threshold:
                lo = mid + 1
            else:
                hi = mid - 1
        return lo
