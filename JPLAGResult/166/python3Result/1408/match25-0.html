class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        lo = 1
        hi = max(nums)
        
        while lo < hi:
            mi = (lo+hi)//2
            
            val = sum(math.ceil(x/mi) for x in nums)
            
            if val > threshold:
                lo = mi+1
            else:
                hi = mi
        
        return lo
