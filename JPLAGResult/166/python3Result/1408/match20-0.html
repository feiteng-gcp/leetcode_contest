class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def valid(k):
            cnt = 0
            for num in nums:
                cnt += math.ceil(num*1.0/k)
            return cnt <= threshold
            
            
        l,r = 1, sum(nums)
        while l < r:
            mid = (l + r)//2
            if valid(mid):
                r = mid 
            else:
                l = mid + 1
        return l
