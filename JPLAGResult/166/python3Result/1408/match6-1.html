class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        '''
        monotonic --> bisection
        '''
        
        l, r 
= 1, max(nums)  
        # [l, r]
        # r: always feasible
        
        while l < r:
            d = (l + r) // 2
            s = sum((n + d - 1) // d for n in nums)
            if s > threshold:
                # d is infeasible. must be greater divisor
                l = d + 1
            else:
                # d is feasible
                r = d
        
        return r
