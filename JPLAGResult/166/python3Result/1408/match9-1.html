def div_sum(nums: List[int], div: int) -> int:
    return sum((n + div - 1) // div for n in nums)

class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        l = 1
        r 
= max(nums)
        while l < r:
            m = (l + r) // 2
            ds = div_sum(nums, m)
            if ds <= threshold:
                r = m
            else:
                l = m + 1
        return l
