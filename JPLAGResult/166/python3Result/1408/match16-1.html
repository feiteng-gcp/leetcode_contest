class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        low, high 
= 1, max(nums)
        
        def can(thre):
            res = 0
            for n in nums:
                res += math.ceil(n / thre)
            return res <= threshold
        
        
        while low < high:
            mid = (low + high) // 2
            if can(mid):
                high = mid
            else:
                low = mid + 1
        return low
