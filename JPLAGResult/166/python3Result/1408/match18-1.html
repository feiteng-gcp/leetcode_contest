class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        low, high = 1, 10 ** 6
        
while low < high:
            mid = (low + high) // 2
            total = sum(math.ceil(n / mid) for n in nums)
            if total <= threshold:
                high = mid
            else:
                low = mid + 1
        return low
