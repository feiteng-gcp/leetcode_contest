class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        end = 10**6
        start = 1
        while start<=end:
            mid = start+(end-start)//2
            s = sum([n//mid+(n%mid!=0) for n in nums])
            if s>threshold:
                start = mid+1
            else:
                end = mid-1
        return start
