from math import ceil
class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def helper(divisor):
            return sum(ceil
(n/divisor) for n in nums)
        
        l,r=1,max(nums)
        while l<r:
            mid=(l+r)//2
            if helper(mid)>threshold:
                l=mid+1
            else:
                r=mid
        return l
