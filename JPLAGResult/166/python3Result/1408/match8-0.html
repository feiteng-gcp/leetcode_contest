from math import ceil

class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        def sum_division(n):
            return sum(
                ceil(m / n)
                for m in nums)
        lo, hi = 1, max(nums)
        while lo < hi:
            # print(lo, hi)
            mid = (lo + hi) // 2
            t = sum_division(mid)
            if t > threshold:
                lo = mid + 1
            else:
                hi = mid
        
        return lo
