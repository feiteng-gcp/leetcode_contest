class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        lo, hi = 1, max(nums)
        while lo < hi:
            mid = (lo + hi) // 2
            m = sum((n + mid - 1) // mid for n in nums)
            if m <= threshold:
                hi = mid
            else:
                lo = mid + 1
        return lo
