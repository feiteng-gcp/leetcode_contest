class Solution:
    def groupThePeople(self, groupSizes: List[int]) -> List[List[int]]:
        groups = []
        g = collections.defaultdict(list)
        for i, s in enumerate(groupSizes):
            g[s].append(i)
            if len(g[s]) == s:
                groups.append(g[s])
                g[s] = []
        return groups
