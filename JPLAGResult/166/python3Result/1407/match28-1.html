class Solution:
    def groupThePeople(self, groupSizes: List[int]) -> List[List[int]]:
        d = collections.defaultdict(list)
        for i, size in enumerate(groupSizes):
            d[size].append(i)
        
        ret = []
        for size, persons in d.items():
            tmp = []
            for idx in persons:
                tmp.append(idx)
                if len(tmp) == size:
                    ret.append(tmp)
                    tmp = []
        return ret
