class Solution:
    def groupThePeople(self, groupSizes: List[int]) -> List[List[int]]:
        size = collections.defaultdict(list)
        res = []
        for i, s in enumerate(groupSizes):
            size[s].append(i)
            if len(size[s]) == s:
                res.append(size[s])
                size[s] = []
        return res
