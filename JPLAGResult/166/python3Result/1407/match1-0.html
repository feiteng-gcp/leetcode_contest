class Solution:
    def groupThePeople(self, groupSizes: List[int]) -> List[List[int]]:
        fac = collections.defaultdict(list)
        
        res = []
        for i, s in enumerate(groupSizes):
            fac[s].append(i)
            
            if len(fac[s]) == s:
                res.append(fac[s])
                fac[s] = []
        
        return res
