from collections import defaultdict
class Solution:
    def groupThePeople(self, groupSizes: List[int]) -> List[List[int]]:
        d = defaultdict(list)
        for i, gs in enumerate(groupSizes):
            d[gs].append(i)
        ans = []
        for gs, cands in d.items():
            for i in range(len(cands)//gs):
                ans.append(cands[i*gs:(i+1)*gs])
        return ans
