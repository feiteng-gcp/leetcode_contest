import collections

class Solution:
    def groupThePeople(self, groupSizes: List[int]) -> List[List[int]]:
        ans = []
        d = collections.defaultdict(list)
        for i, size in enumerate(groupSizes):
            d[size].append(i)
            if len(d[size]) == size:
                ans.append(d[size])
                d[size] = []
        return ans
