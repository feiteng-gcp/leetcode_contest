class Solution:
    def groupThePeople(self, groupSizes: List[int]) -> List[List[int]]:
        res = []
        from collections import defaultdict
        d = defaultdict(list)
        for idx, num in enumerate(groupSizes):
            d[num].append(idx)
        for k, v in d.items():
            temp = []
            for i in v:
                temp.append(i)
                if len(temp) == k:
                    res.append(temp[:])
                    temp = []
        return res
