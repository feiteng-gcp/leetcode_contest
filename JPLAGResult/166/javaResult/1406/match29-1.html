class Solution {
    public int subtractProductAndSum(int n) {
        String s = String.valueOf(n);
        int sum = 0;
        int pro = 1;
        for (int i = 0; i < s.length(); i++) {
            int dig  = s.charAt(i) - '0';
            sum += dig;
            pro *= dig;
        }
        return pro - sum;
    }
}
