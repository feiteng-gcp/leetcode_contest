class Solution {
    public int subtractProductAndSum(int n) {
        int p = 1, sum = 0;
        while (n != 0) {
            int cur = (n % 10);
            n /= 10;
            p *= cur;
            sum += cur;
        }
        //System.out.println(p + " " + sum);
        return p - sum;
    }
}
