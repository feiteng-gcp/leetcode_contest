	class Solution {
	    public int smallestDivisor(int[] nums, int threshold) {
	        int low = 0, high = 1000005;
	        while(high - low > 1){
	        	int h = high+low>>1;
	        	long K = 0;
	        	for(int v : nums){
	        		K += (v+h-1)/h;
	        	}
	        	if(K <= threshold){
	        		high = h;
	        	}else{
	        		low = h;
	        	}
	        }
	        return high;
	    }
	}
