class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int left = 1, right = 1000000;
        int res = right;
        while(left <= right){
            int sum = 0;
            int mid = (left + right) >> 1;
            for(int num : nums){
                sum += (num - 1) / mid + 1;
            }
            if(sum <= threshold){
                res = Math.min(res, mid);
                right = mid - 1;
            }
            else
                
left = mid + 1;
        }
        return res;
    }
}
