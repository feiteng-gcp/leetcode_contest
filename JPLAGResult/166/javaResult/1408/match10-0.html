class Solution {
public int smallestDivisor(int[] nums, int threshold) {

        int low = 1;
        int high = 10000000;

        int ans = 10000000;
        while (low <= high) {
            int mid = (low + high) / 2;
            long sum = 0;

            for (int i = 0; i < nums.length; i++) {
                sum += (nums[i] % mid == 0 ? nums[i] / mid : nums[i] / mid + 1);
            }

            if (sum <= threshold) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return ans;
    }
}
