class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int lo = 1;
        int hi = 1000000;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            int sum = 0;
            for (int num : nums) {
                sum += (num - 1) / mid + 1;
            
}
            if (sum > threshold) lo = mid + 1;
            else hi = mid;
        }
        return lo;
    }
}
