class Solution {
    int[] nums;
    public int smallestDivisor(int[] nums, int threshold) {
        this.nums = nums;
        int left = 1; 
        int right = 10000000;
        while (left < right) {
            int mid = (left + right) / 2;
            int sum = getSum(mid);
            if (sum > threshold) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left;
    }
    public int getSum(int divisor) {
        int sum = 0;
        for (int num: nums) {
            sum += num / divisor;
            if (num % divisor != 0) {
                sum++;
            }
        }
        return sum;
    }
}
