class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        // probably a modulo problem
        // binary search?
        int left = 1;
        int right = Integer.MAX_VALUE;
        
        while(left <= right) {
            int mid = (int)(((long)left + right)/2l);
            // System.out.println(left + " " + mid + " " + right);
            int count = 0;
            
            for(int i = 0; i < nums.length; i++) {
                count += Math.ceil((double)nums[i]/(double)mid);
                if(count > threshold) {
                    break;
                }
            }
            
            if(count > threshold) {
                left = mid + 1;
            }
            else {
                right = mid - 1;
            }
        }
        
        return left;
        
    }
}
