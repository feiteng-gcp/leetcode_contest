class Solution {
    public int smallestDivisor(int[] n, int t) {
        int lo = 1, hi = n[0];
        for (int i : n) {hi = Math.max(i, hi);
}
        
        while (lo < hi) {
            int mid = (lo + hi) >>> 1;
            if (valid(mid, n, t)) {
                hi = mid;
            }
            else {lo = mid + 1;}
        }
        return lo;
    }
    
    boolean valid(int m, int[] n, int t) {
        int sum = 0;
        for (int i : n) {
            sum += (int) Math.ceil(((double) i) / m);
            
if (sum > t) return false;
        }
        return true;
    }
}
