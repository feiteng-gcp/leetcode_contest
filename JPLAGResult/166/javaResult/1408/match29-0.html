class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int l = 1, r = 214748364;
        int ans = -1;
        while (l < r) {
            int mid = (l + r) / 2;
            if (sum(nums, mid) <= threshold) {
                ans = mid;
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return ans;
    }
    
    private int sum(int[] nums, int d) {
        int s = 0;
        for (int n : nums) {
            s += (int)(Math.ceil(n * 1.0 / d));
        }
        return s;
    }
}
