class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int low = 1;
        int high = 1000000;
        while (low < high) {
            int mid = (low + high) / 2;
            int sum = 0;
            for (int num : nums) {
                sum += num / mid;
                if (num % mid != 0) {
                    sum += 1;
                
}
            }
            if (sum <= threshold) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
}
