class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        
        int low = 1, high = 1000000;
        //find left most curr <= thres 
        while(low < high) {
            int mid = (low + high) / 2;
            int curr = cal(nums, mid);
            if (curr <= threshold) 
{
                high = mid;
            } else {
                low = mid+1;
            }
            
        }
        return low;
    }
    
    private int cal(int[] nums, int div) {
        int sum = 0;
        for (int num : nums) {
            sum += num / div;
            
            if (num % div != 0)
                sum++;
        }
        //System.out.println("div" + div + " " + sum);
        return sum;
    }
}
