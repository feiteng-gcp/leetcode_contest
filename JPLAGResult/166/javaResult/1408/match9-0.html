class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int l = 1;
        int r = Integer.MAX_VALUE - 5;
        int ans = r;
        while(l <= r){
            int mid = (l + r) / 2;
            if(valid(nums, mid, threshold)){
                ans = mid;
                r = mid - 1;
            }else{
                l = mid + 1;
            }
        }
        return ans;
    }
    
    public boolean valid(int [] nums, int divisor, int threshold){
        int total = 0;
        for(int elem: nums){
            total += (Math.ceil((1.0 * elem / divisor)));
        }
        return total <= threshold;
    }
}
