class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int l = 1, r = 1000000;
        while (l < r) {
            int mid = l + (r-l) / 2;
            if (getResult(nums, mid) <= threshold) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
    
    private int getResult(int[] nums, int divisor) {
        int ans = 0;
        for (int num : nums) {
            ans += (num / divisor);
            if (num % divisor != 0) ans++;
        }
        return ans;
    }
}
