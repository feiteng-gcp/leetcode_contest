class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int start = 1, end = 0;
        for (int n : nums) end = Math.max(end, n);
        
        while (start < end) {
            int mid = (end - start) / 2 + start;
            if (isLessEqual(nums, mid, threshold)) {
                end = mid;
            } else {
                start = mid + 1;
            }
        }
        
        return start;
    }
    
    private boolean isLessEqual(int[] nums, int d, int threshold) {
        int sum = 0;
        for (int n : nums) {
            sum += n / d;
            if (n % d != 0) {
                sum++;
            }
            if (sum > threshold) return false;
        }
        return sum <= threshold;
    }
}
