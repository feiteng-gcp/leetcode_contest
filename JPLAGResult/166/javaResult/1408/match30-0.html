import java.util.Optional;
import java.util.function.IntPredicate;

class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int left = 1;
        int right = 1000001;
        while (left < right) {
            int mid = left+(right-left)/2;
            int sum=0;
            for (int num : nums) {
                sum += op(num, mid);
            }
            if (sum <= threshold) {
                right=mid;
            } else {
                left=mid+1;
            }

        }
        return left;
    }

    public int op(int num, int divisor) {
        return num % divisor == 0 ? num / divisor : num / divisor + 1;
    }

    public static void main(String[] args) {
        System.out.println(new Solution().smallestDivisor(new int[]{1,2,5,9},6));
        System.out.println(new Solution().smallestDivisor(new int[]{2,5,7,3,11},11));
        System.out.println(new Solution().smallestDivisor(new int[]{19},5));
    }
}
