class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int l = 1;
        int r = 1000000;
        
        while(l <= r){
            int mid = l + ((r - l) >> 1);
            if(check(nums, mid, threshold)){
                l = mid + 1;
            }
            else{
                r = mid - 1;
            
}
        }
        
        return l;
    }
    
    private boolean check(int[] nums, int d, int threshold){
        int sum = 0;
        for(int i : nums){
            int add = 
i % d == 0 ? 0 : 1;
            sum += i / d;
            sum += add;
            if(sum > threshold){
                return true;
            }
        }
        
        return false;
    }
}
