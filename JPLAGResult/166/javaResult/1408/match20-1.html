class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        
int l = 1;
        int r = 1000000;

        while (l < r) {
            int m = (l + r) >> 1;
            int v = result(nums, m);
            if (v > threshold) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l;
    }

    private int result(int[] nums, int divide) 
{
        int ans = 0;
        for (int num : nums) {
            ans += num / divide;
            if ((num % divide) > 0) {
                ans++;
            }
        }
        return ans;
    }
}
