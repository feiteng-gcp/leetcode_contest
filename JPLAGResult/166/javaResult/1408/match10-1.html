class Solution {
    public int smallestDivisor(int[] nums, int ts) {
        int low = 1, high = Integer.MAX_VALUE;
        int ans = high;
        while(low <= high) {
            int med = low + (high - low) / 2;
            int sum = 0;
            for (int num : nums) {
                sum += (num % med == 0 ? (num / med) : (num / med + 1));
            }
            if (sum <= ts) {
                ans = Math.min(ans, med);
                
high = med - 1;
            } else {
                low = med + 1;
            }
        }
        return ans;
    }
}
