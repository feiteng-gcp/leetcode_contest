class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        Arrays.sort(nums);
        int n = nums.length;
        
int lo = 1, hi = nums[n - 1];
        while(lo < hi){ 
            int mid = lo + (hi - lo) / 2;
            if(check(nums, mid, threshold)){
                hi = mid;
            }
            else{
                lo = mid + 1;
            }
        }
        return lo;
    }
    
    private boolean check(int[] nums, int div, int threshold){
        int sum = 0;
        for(int n: nums){
            sum += (int)Math.ceil((double) n / (double) div);
        }
        return sum <= threshold;
    }
}
