class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int l = 1, r = 1_000_000;
        while (l < r) {
            int mi = (r - l) / 2 + l;
            if (isPossible(nums, mi, threshold)) 
{
                r = mi;
            } else {
                l = mi + 1;
            }
        }
        return l;
    }

    private boolean isPossible(int[] nums, int x, int t) 
{
        int ans = 0;
        for (int num :
                nums) {
            ans += Math.ceil(num * 1.0 / x);
        }
        
        return ans <= t;
    }
}
