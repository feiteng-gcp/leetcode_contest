class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int l = 1, r = 1;
        for (int i : nums) r = Math.max(i, r);
        while (l < r) {
            int m = l + (r - l) / 2;
            if (succ(nums, threshold, m)) r = m;
            else l = m + 1
;
        }
        return l;
    }
    private boolean succ(int[] nums, int t, int d) {
        int sum = 0;
        for (int i : nums) {
            sum += i / d + (i % d == 0 ? 0 : 1);
            
if (sum > t) return false;
        }
        return true;
    }
}
