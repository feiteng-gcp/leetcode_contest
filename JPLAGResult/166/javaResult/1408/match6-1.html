class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int l = 1;
        int r = nums[nums.length - 1];
        while (l < r) {
            int mid = (l + r) / 2;
            //System.out.println(mid);
            int sum = get(nums, mid);
            //System.out.println(l + " " + r + " " + mid + " " + sum);
            if (sum > threshold) 
{
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        return l;
    }
    private int get(int[] nums, int div) 
{
        int sum = 0;
        for (int n : nums) {
            //System.out.println(Math.ceil(n / div * 1.0));
            sum += (int)(Math.ceil(n * 1.0 / div ));
        }
        return sum;
    } 
}
