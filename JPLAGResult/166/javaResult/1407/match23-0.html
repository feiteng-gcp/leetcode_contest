class Solution {
    public List<List<Integer>> groupThePeople(int[] groupSizes) {
        Map<Integer, List<Integer>> personIdsByGroupSize = new HashMap<>();
        for (int i = 0; i < groupSizes.length; i++) {
            personIdsByGroupSize.computeIfAbsent(groupSizes[i], k -> new ArrayList<>()).add(i);
        }
        List<List<Integer>> output = new ArrayList<>();
        for (Map.Entry<Integer, List<Integer>> entry : personIdsByGroupSize.entrySet()) {
            for (int i = 0; i < entry.getValue().size(); i += entry.getKey()) {
                output.add(entry.getValue().subList(i, i + entry.getKey()));
            }
        }
        return output;
    }
}
