class Solution {
    public List<List<Integer>> groupThePeople(int[] gs) {
        Map<Integer, List<Integer>> data = new HashMap<>();
        for (int i = 0; i < gs.length; i++) {
            List<Integer> list = 
data.getOrDefault(gs[i], new ArrayList<>());
            list.add(i);
            data.put(gs[i], list);
        }
        
        List<List<Integer>> ans = new ArrayList<>();
        for (Map.Entry<Integer, List<Integer>> entry : data.entrySet()) {
            int size = entry.getKey();
            List<Integer> nums = entry.getValue();
            for (int i = 0; i < 
nums.size(); i += size) {
                List<Integer> tmp = new ArrayList<>();
                for (int j = i; j < i + size; j ++) {
                    tmp.add(nums.get(j));
                }
                ans.add(tmp);
            }
        }
        return ans;
    }
}
