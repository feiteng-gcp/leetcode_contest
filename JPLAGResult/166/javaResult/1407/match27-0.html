class Solution {
    public List<List<Integer>> groupThePeople(int[] grp) {
        List<List<Integer>> re = new ArrayList<>();
        Map<Integer, List<Integer>> map = new HashMap<>();
        int n = grp.length;
        for(int i = 0; i < n; i++) {
            int g = grp[i];
            if(!map.containsKey(g)) map.put(g, new ArrayList<>());
            map.get(g).add(i);
        }
        for(int k : map.keySet()) {
            List<Integer> list = map.get(k);
            // if k = 3, pick every 3
            int size = list.size();
            List<Integer> tmp = new ArrayList<>();
            for(int i = 0; i < size; i++) {
                tmp.add(list.get(i));
                if(tmp.size() % k == 0) {
                    re.add(tmp);
                    tmp = new ArrayList<>();
                }
            }
        }
        return re;
    }
}
