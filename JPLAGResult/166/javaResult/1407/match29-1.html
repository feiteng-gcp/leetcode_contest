class Solution {
    public List<List<Integer>> groupThePeople(int[] groupSizes) {
        List<List<Integer>> result = new ArrayList<>();
        Map<Integer, List<Integer>> groupBySize = new HashMap<>();
        for (int size : groupSizes) {
            if (!groupBySize.containsKey(size)) {
                groupBySize.put(size, new ArrayList<Integer>());
            }
        }
        for (int i = 0; i < groupSizes.length; i++) {
            
groupBySize.get(groupSizes[i]).add(i);
            if (groupBySize.get(groupSizes[i]).size() == groupSizes[i]) {
                result.add(groupBySize.get(groupSizes[i]));
                groupBySize.put(groupSizes[i], new ArrayList<Integer>());
            }
        }
        return result;
    }
}
