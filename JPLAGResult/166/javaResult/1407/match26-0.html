class Solution {
    public List<List<Integer>> groupThePeople(int[] groupSizes) {
        Map<Integer, List<Integer>> gs = new HashMap<>();
        for (int i = 0; i < groupSizes.length; i++) {
            int g = groupSizes[i];
            List<Integer> s = new ArrayList<>();
            if (gs.containsKey(g)) {
                s = gs.get(g);
            }
            s.add(i);
            gs.put(g, s);
        }
        List<List<Integer>> ans = new ArrayList<>();
        for (Integer s : gs.keySet()) {
            List<Integer> ids = gs.get(s);
            int len = ids.size();
            List<Integer> one = new ArrayList<>();
            for (int i = 0; i < len; i++) {
                one.add(ids.get(i));
                if (i % s == s - 1) {
                    ans.add(one);
                    one = new ArrayList<>();
                }
            }
        }
        return ans;
    }
}
