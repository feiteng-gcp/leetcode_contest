class Solution {

  private int convert(int[][] m) {
    int res = 0;
    int k = 0;
    for(int i=0; i<m.length; i++) {
      for(int j=0; j<m[i].length; j++) {
        if(m[i][j] == 1) res |= (1<<k)
;
        k++;
      }
    }
    return res;
  }

  private void flip(int[][] m, int i, int j) {
    m[i][j] = (1-m[i][j]);
    if(i+1 < m.length) m[i+1][j] = 1 - m[i+1][j];
    if(j+1 < m[i].length) m[i][j+1] = 1 - m[i][j+1];
    if(i-1 >= 0) m[i-1][j] = 1 - m[i-1][j];
    if(j-1 >= 0) m[i][j-1] = 1 - m[i][j-1];
  }

  private void print(
int[][] m) {
    for (int[] v : m) {
      System.out.println(Arrays.toString(v));
    }
  }

  private int[][] clone(int[][] m) {
    
int [][] res = new int[m.length][m[0].length];
    for(int i=0; i<m.length; i++) {
      for(int j=0; j<m[i].length; j++) {
        res[i][j] = m[i][j];
      }
    }
    return res;
  }

  
public int minFlips(int[][] mat) {
    Queue<int[][]> q = new LinkedList<>();
    q.add(mat);
    Set<Integer> vis = new HashSet<>();
    vis.add(convert(mat));
    int res = 0;
    while(!q.isEmpty()) {
      int sz = q.size();
      for(int z=0; z<sz; z++) {
        int[][] cm = q.poll();
        if(convert(cm) == 0) return res;
        for(int i=0; i<cm.length; i++) {
          for(int j=0; j<cm[i].length; j++) {
            flip(cm, i, j);
//            print(cm);
            int val = convert(cm);
            if(!vis.contains(val)) {
              q.add(clone(cm));
              vis.add(val);
            
}
            flip(cm, i, j);
          }
        }
      }
      res++;
    }
    return -1;
  }
}
