class Solution {
    class Item {
        int[][] m;
        int steps;

        public Item(int[][] m, int steps) {
            this.m = m;
            this.steps = steps;
        }
    }

    public int minFlips(int[][] m) {
        int rows = m.length;
        int cols = m[0].length;
        
Queue<Item> q = new ArrayDeque<>();
        Set<Long> seen = new HashSet<>();
        q.offer(new Item(m, 0));
        while (!q.isEmpty()) {
            Item top = q.poll();
            int[][] topm = top.m;
            if (allzero(topm)) 
{
                return top.steps;
            }
            for (int i = 0; i < rows; i++) {
                for (int j = 0; j < cols; j++) {
                    int[][] flipped = flip(topm, i, j);
                    long code = encode(flipped);
                    if (!seen.contains(code)) {
                        seen.add(code);
                        
q.offer(new Item(flipped, top.steps + 1));
                    }
                }
            }
        }
        
return -1;
    }

    private long encode(int[][] m) {
        int rows = m.length;
        int cols = m[0].length;
        long r = 0;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                r = r * 10 + m[i][j];
            }
        }
        return r;
    }

    int[][] dirs = 
{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

    private int[][] flip(int[][] m, int si, int sj) {
        int rows = m.length;
        int cols = m[0].length;
        int[][] nm = new int[rows][cols];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                nm[i][j] = m[i][j];
            }
        }
        f(nm, si, sj);
        for (int[] d : dirs) {
            int ni = si + d[0];
            int nj = sj + d[1];
            if (ni >= 0 && ni < rows && nj >= 0 && nj < cols) {
                f(nm, ni, nj);
            }
        }
        return nm;
    }

    private void f(int[][] m, int i, int j) {
        m[i][j] = m[i][j] == 1 ? 0 : 1;
    }

    private boolean allzero(int[][] m) {
        int rows = m.length;
        int cols = m[0].length;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (m[i][j] != 0) {
                    return false;
                }
            }
        }
        return true;
    }
}
