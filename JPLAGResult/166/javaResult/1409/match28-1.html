class Solution {
    int n;
    int m;
    public int minFlips(int[][] mat) {
        this.n = mat.length;
        this.m = mat[0].length;
        if (isZeroMat(mat)) {
            System.out.println("???");
            return 0;
        }
        HashSet<Integer> visited = new HashSet<>();
        Queue<int[][]> queue = new LinkedList<>();
        queue.add(mat);
        int step = 0;
        visited.add(convertMatToInt(mat));
        
while (!queue.isEmpty()) {
            int size = queue.size();
            while (size-- > 0) {
                int[][] matrix = queue.poll();
                if(isZeroMat(matrix)) {
                    return step;
                }
                for (int i = 0; i < n; i++) {
                    for (int j = 0; j < m; j++) {
                        int[][] newMat = flip(matrix, i, j);
                        int converted = convertMatToInt(newMat);
                        //System.out.println(converted);
                        if (!visited.contains(converted)) {
                            visited.add(converted);
                            
queue.add(newMat);
                        }
                    }
                }
            }
            step++;
        }
        return -1;
    }
    
    public int[][] flip(int[][] mat, int x, int y) {
        int[][] newMat = 
new int[n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                newMat[i][j] = mat[i][j];
            }
        }
        newMat[x][y] ^= 1;
        if (x > 0) {
            newMat[x - 1][y] ^= 1;
        }
        if (y > 0) {
            newMat[x][y - 1] ^= 1;
        }
        if (x < n - 1) {
            newMat[x + 1][y] ^= 1;
        }
        if (y < m - 1) {
            newMat[x][y + 1] ^= 1;
        }
        return newMat;
    }
    public int convertMatToInt(int[][] mat) {
        int result = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (mat[i][j] == 1) {
                    result += 1 << (i * m + j);
                }
            }
        }
        
return result;
    }
    public boolean isZeroMat(int[][] mat) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (mat[i][j] == 1) {
                    return false;
                }
            }
        }
        return true;
    }
}
