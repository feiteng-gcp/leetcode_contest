class Solution {
    public int minFlips(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        Queue<int[][]> q = new LinkedList<>();
        Set<Integer> his = new HashSet<>();
        q.offer(mat);
        his.add(encode(mat));
        
int step = 0;
        while (!q.isEmpty()) {
            int size = q.size();
            while (size > 0) {
                int[][] cur = q.poll();
                if (solved(cur)) {
                    return step;
                }
                for (int i = 0; i < m; i++) {
                    for (int j = 0; j < n; j++) {
                        int[][] tmp = flip(cur, i, j);
                        if (his.add(encode(tmp))) {
                            q.offer(tmp);
                        }
                    
}
                }
                size--;
            }
            step++;
        }
        return -1;
    }
    
    private int encode(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        int result = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                result <<= 1;
                result |= mat[i][j];
            }
        }
        return result;
    }
    
    int[] rm = new int[]{1,-1,0,0};
    int[] cm = new int[]{0,0,1,-1};
    private int[][] flip(int[][] mat, int r, 
int c) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] result = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                result[i][j] = mat[i][j];
            }
        }
        result[r][c] = mat[r][c] == 0 ? 1 : 0;
        for (int k = 0; k < 4; k++) {
            int nr = r + rm[k];
            int nc = c + cm[k];
            if (nr >= 0 && nc >= 0 && nr < m && nc < n) {
                result[nr][nc] = mat[nr][nc] == 0 ? 1 : 0;
            }
        }
        return result;
    }
    
    private boolean solved(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        
for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (mat[i][j] == 1) {
                    return false;
                }
            }
        }
        return true;
    }
}
