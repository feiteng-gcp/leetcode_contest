class Solution {
    public int minFlips(int[][] mat) {
        Queue<int[][]> q = new LinkedList<>();
        q.add(mat);
        int r = mat.length, c = mat[0].length;
        
        
Set<String> visited = new HashSet<>();
        // System.out.println(helper(mat, r, c));
        visited.add(helper(mat, r, c));
        
        int ans = 0;
        
        while(!q.isEmpty()) {
            int size = q.size();
            
            for(int k=0;k<size;k++) {
                int[][] top = q.poll();
                if(isZero(top, r, c)) {
                    return ans;
                
}
                
                for(int i=0;i<r;i++) {
                    for(int j=0;j<c;j++) {
                        int[][] flipped = flip(top, i, j, r, c);
                        String str = helper(flipped, r, c);
                        // System.out.println(str);
                        if(!visited.contains(str)) {
                            visited.add(str);
                            q.add(flipped);
                        }
                    }
                }
            }
            
            if(q.isEmpty()) return -1;
            
            ans++;
        }
        
        return ans;
    }
    
    private int[][] flip(int[][] mat, int x, int y, int r, int c) {
        int[][] res = new int[r][c];
        
        for(int i=0;i<r;i++) {
            for(int j=0;j<c;j++) {
                
                if(isValid(i, j, x, y)) {
                    res[i][j] = 1-mat[i][j];
                } else {
                    res[i][j] = mat[i][j];
                }
            }
        }
        
        return res;
    }
    
    private boolean isValid(int x1, int y1, int x2, int y2) {
        return (x1==x2 && y1==y2) ||
            (x1==x2-1 && y1==y2) ||
            (x1==x2+1 && y1==y2) ||
            (x1==x2 && y1==y2-1) ||
            (x1==x2 && y1==y2+1);
    }
    
    
    
    private String helper(int[][] mat, int r, 
int c) {
        StringBuilder sb = new StringBuilder();
        
        for(int i=0;i<r;i++) {
            for(int j=0;j<c;j++) {
                sb.append(mat[i][j]).append(",");
            }
        }
        
        return sb.toString();
    } 
    
    
    private boolean isZero(int[][] mat, int r, 
int c) {
        for(int i=0;i<r;i++) {
            for(int j=0;j<c;j++) {
                if(mat[i][j]!=0) return false;
            }
        }
        return true;
    }
}
