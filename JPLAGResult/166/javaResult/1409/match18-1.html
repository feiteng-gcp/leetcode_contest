class Solution {
    public int minFlips(int[][] mat) {
        Map<Integer, Integer> v = new HashMap<>();
        Deque<Integer> q = new ArrayDeque<>();
        int n = mat.length, m = mat[0].length;
        
int fist = getState(mat);
        v.put(fist, 0);
        q.addLast(fist);
        while (!q.isEmpty()) {
            int next = q.removeFirst();
            int step = v.get(next);
            if (next == 0) 
{
                return step;
            }
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                    int other = getState(flip(getMatrix(next, n, m), i, j));
                    if (!v.containsKey(other)) {
                        
v.put(other, step + 1);
                        q.addLast(other);
                    }
                }
            }
        }
        return -1;
    }
    
    public int getState(int[][] mat) {
        int s = 0;
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                s = s * 2 + mat[i][j];
            
}
        }
        return s;
    }
    
    public int[][] getMatrix(int s, int n, int m) {
        int[][] ans = new int[n][m];
        for (int i = n - 1; i >= 0; i--) {
            for (int j = m - 1; j >= 0; j--) {
                ans[i][j] = s % 2;
                s /= 2;
            }
        }
        return ans;
    }
    
    public int[][] flip(int[][] omat, int i, int j) {
        int n = omat.length, m = omat[0].length;
        int[][] mat = getMatrix(getState(omat), n, m);
        
mat[i][j] = 1 - mat[i][j];
        if (i > 0) {
           mat[i - 1][j] = 1 - mat[i - 1][j]; 
        }
        if (i < n - 1) {
            mat[i + 1][j] = 1 - mat[i + 1][j]; 
        }
        if (j > 0) {
            mat[i][j - 1] = 1 - mat[i][j - 1]; 
        }
        if (j < m - 1) {
            mat[i][j + 1] = 1 - mat[i][j + 1]; 
        }
        return mat;
    }
}
