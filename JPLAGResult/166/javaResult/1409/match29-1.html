import java.util.HashSet;
import java.util.LinkedList;
import java.util.Queue;

class Solution {

    private static int[][] direction = new int[][]{{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    
private int m, n;

    public int minFlips(int[][] mat) {
        m = mat.length;
        n = mat[0].length;
        Queue<Integer> queue = new LinkedList<>();
        queue.add(toInt(mat));
        queue.add(0);
        HashSet<Integer> visited = new HashSet<>();
        while (queue.size() != 0) {
            int board = queue.poll();
            int step = queue.poll();
            if (board == 0) {
                return step;
            }
            mat = toMat(board);
            
for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    int next = flip(board, i, j);
                    if (visited.contains(next)) {
                        continue;
                    }
                    if (next == 0) {
                        return step + 1;
                    }
                    visited.add(next);
                    queue.add(next);
                    queue.add(step + 1);
                }
            }
        
}
        return -1;
    }

    private int flip(int board, int x, int y) {
        int[][] mat = toMat(board);
        mat[x][y] = 1 - mat[x][y];
        for (int i = 0; i < 4; i++) {
            int nx = x + direction[i][0];
            int ny = y + direction[i][1];
            if (nx < 0 || nx >= m || ny < 0 || ny >= n) {
                continue;
            }
            mat[nx][ny] = 1 - mat[nx][ny];
        }
        return toInt(mat);
    }

    private int toInt(int[][] mat) {
        
int val = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                val <<= 1;
                val |= mat[i][j];
            }
        }
        return val;
    
}

    private int[][] toMat(int val) {
        int[][] mat = new int[m][n];
        for (int i = m - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {
                mat[i][j] = val & 1;
                val >>= 1;
            
}
        }
        return mat;
    }

    public static void main(String[] args) {
        System.out.println(new Solution().minFlips(new int[][]{{0}}));
    }

}
