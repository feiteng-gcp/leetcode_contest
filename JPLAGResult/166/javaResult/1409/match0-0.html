class Solution {
    public int minFlips(int[][] grid) {
        Set<String> seen = new HashSet<>();
        Deque<int[][]> queue = new ArrayDeque<>();
        queue.add(grid);
        int cnt = 0, n = grid.length, m = grid[0].length;
        while(!queue.isEmpty()){
            int size = queue.size();
            while(size-- > 0){
                int[][] cur = queue.poll();
                if(allZero(cur)){
                    return cnt;
                }
                for(int i = 0 ; i < n; i++){
                    for(int j = 0 ; j < m ; j++){
                        int[][] tmp = copyGrid(cur);
                        flip(tmp, i, j);
                        String mark = gridToString(tmp);
                        if(seen.contains(mark)){
                            continue;
                        }
                        seen.add(mark);
                        queue.offer(tmp);
                    }
                }
            }
            cnt++;
        }
        return -1;
    }
    
    private boolean allZero(int[][] grid){
        for(int i = 0 ; i < grid.length; i++){
            for(int j = 0 ; j < grid[0].length; j++){
                if(grid[i][j] != 0){
                    return false;
                }
            }
        }
        return true;
    }
    
    private String gridToString(int[][] grid){
        StringBuilder res = new StringBuilder();
        for(int i = 0 ; i < grid.length; i++){
            res.append(Arrays.toString(grid[i]));
        }
        return res.toString();
    }
    
    private void flip(int[][] grid, int r, int c){
        int n = grid.length, m = grid[0].length;
        grid[r][c] ^= 1;
        if(r > 0){
            grid[r-1][c] ^= 1;
        }
        if(r < n - 1){
            grid[r + 1][c] ^= 1;
        }
        if(c > 0){
            grid[r][c-1] ^= 1;
        }
        if(c < m - 1){
            grid[r][c + 1] ^= 1;
        }
    }
    
    private int[][] copyGrid(int[][] grid){
        int n = grid.length, m = grid[0].length;
        int[][] newGrid = new int[n][m];
        for(int i = 0 ; i < n ; i++){
            for(int j = 0 ; j < m ; j++){
                newGrid[i][j] = grid[i][j];
            }
        }
        return newGrid;
    }
}
