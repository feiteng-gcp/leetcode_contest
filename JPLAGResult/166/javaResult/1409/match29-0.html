class Solution {
    int m;
    int n;
    public int minFlips(int[][] mat) {
        m = mat.length;
        n = mat[0].length;
        HashSet<Integer> set = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        int d = convert(mat);
        queue.add(d);
        set.add(d);
        int ncnt = 1;
        int steps = 0;
        while (!queue.isEmpty()) {
            int nxt_ncnt = 0;
            for (int cnt = 0; cnt < ncnt; cnt++) {
                int cur = queue.poll();
                if (cur == 0) return steps;
                for (int i = 0; i < m; i++) {
                    for (int j = 0; j < n; j++) {
                        int nxt = flip(cur, i, j);
                        if (!set.contains(nxt)) {
                            set.add(nxt);
                            queue.add(nxt);
                            nxt_ncnt++;
                        }
                    }
                }
            }
            ncnt = nxt_ncnt;
            steps++;
        }
        return -1;
    }
    private int flip(int value, int i, int j) {
        int[][] mat = revert(value);
        mat[i][j] = 1- mat[i][j];
        if (i > 0) mat[i-1][j] = 1 - mat[i-1][j];
        if (j > 0) mat[i][j-1] = 1 - mat[i][j-1];
        if (i+1 < m) mat[i+1][j] = 1 - mat[i+1][j];
        if (j+1 < n) mat[i][j+1] = 1 - mat[i][j+1];
        
        return convert(mat);
    }
    private int[][] revert(int value) {
        int[][] res = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                res[i][j] = value%2;
                value = value/2;
            }
        }
        return res;
    }
    private int convert(int[][] mat) {
        int c = 1;
        int value = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                value+=mat[i][j]*c;
                c*=2;
            }
        }
        return value;
    }
}
