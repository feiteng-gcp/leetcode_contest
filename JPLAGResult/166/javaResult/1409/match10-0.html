class Solution {
    int ans = Integer.MAX_VALUE;
    Set<String> visited;
    public int minFlips(int[][] mat) {
        visited = new HashSet<>();
        Queue<int[][]> q = new LinkedList<>();
        q.offer(mat);
        int moves = 0;
        while(q.size() > 0){
            int size = q.size();
            while(size-- > 0){
                int [][] cur = q.poll();
                if(valid(cur)){
                    return moves;
                }
                for(int i = 0; i < mat.length; i++){
                    for(int j = 0; j < mat[0].length; j++){
                        int [][] next = new int[mat.length][mat[0].length];
                        for(int i2 = 0; i2 < mat.length; i2++){
                            for(int j2 = 0; j2 < mat[0].length; j2++){
                                next[i2][j2] = cur[i2][j2];
                            }
                        }
                        flip(i, j, next);
                        StringBuilder keystr = new StringBuilder();
                        for(int k = 0; k < mat.length; k++){
                            keystr.append(Arrays.toString(next[k]));
                        }
                        if(visited.add(keystr.toString())){
                            q.offer(next);
                        }
                    }
                }

            }
            moves++;
        }
        return -1;
    }
    public boolean valid(int [][] mat){
        for(int i = 0; i < mat.length; i++){
            for(int j = 0; j < mat[0].length; j++){
                if(mat[i][j] == 1){
                    return false;
                }
            }
        }
        return true;
    }
    public void flip(int r, int c, int [][] mat){
        mat[r][c] = mat[r][c] == 1 ? 0 : 1;
        if(r + 1 < mat.length){
            mat[r + 1][c] = mat[r + 1][c] == 1 ? 0 : 1;
        }
        if(c + 1 < mat[0].length){
            mat[r][c + 1] = mat[r][c + 1] == 1 ? 0 : 1;
        }
        if(r - 1 >= 0){
            mat[r - 1][c] = mat[r - 1][c] == 1 ? 0 : 1;
        }
        if(c - 1 >= 0){
            mat[r][c - 1] = mat[r][c - 1] == 1 ? 0 : 1;
        }
    }
}
