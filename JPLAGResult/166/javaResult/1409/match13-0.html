class Solution {
    public int minFlips(int[][] mat) {
        Queue<int[][]> queue = new LinkedList<>();
        queue.offer(mat);
        Set<String> visited = new HashSet<>();
        visited.add(genKey(mat));
        int steps = 0;
        while (!queue.isEmpty()) {
            int sz = queue.size();
            for (int k = 0; k < sz; k++) {
                int[][] currMat = queue.poll();
                if (isAllZero(currMat)) return steps;
                for (int i = 0; i < mat.length; i++) {
                    for (int j = 0; j < mat[0].length; j++) {
                        int[][] nextMat = getNextMat(currMat, i, j);
                        if (!visited.add(genKey(nextMat))) continue;
                        queue.offer(nextMat);
                    }
                }
            }
            steps++;
        }
        return -1;
    }
    
    private boolean isAllZero(int[][] mat) {
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                if (mat[i][j] != 0) return false;
            }
        }
        return true;
    }
    
    private int[][] getNextMat(int[][] mat, int i, int j) {
        int m = mat.length, n = mat[0].length;
        int[][] next = new int[m][n];
        for (int r = 0; r < m; r++) {
            for (int c = 0; c < n; c++) {
                next[r][c] = mat[r][c];
            }
        }
        next[i][j] ^= 1;
        if (i > 0) next[i-1][j] ^= 1;
        if (i < m-1) next[i+1][j] ^= 1;
        if (j > 0) next[i][j-1] ^= 1;
        if (j < n-1) next[i][j+1] ^= 1;
        return next;
    }
    
    private String genKey(int[][] mat) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                sb.append(mat[i][j]).append(",");
            }
        }
        return sb.toString();
    }
}
