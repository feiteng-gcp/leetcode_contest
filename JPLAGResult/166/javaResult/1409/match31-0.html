class Solution {
    public int minFlips(int[][] mat) {
        Set<Integer> set = new HashSet<>();
        set.add(conv(mat));
        //System.out.println(conv(mat));
        if (conv(mat) == 0) return 0;
        Queue<int[][]> q = new LinkedList<>();
        q.add(mat);
        int sol = 1;
        int[][] dir = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
        while (!q.isEmpty()) {
            int len = q.size();
            for (int l = 0; l < len; l++) {
            int[][] cur = q.poll();
            for (int i = 0; i < cur.length; i++) {
                for (int j = 0; j < cur[0].length; j++) {
                    int[][] next = cl(cur);
                    //System.out.println(conv(next));
                    next[i][j] = 1 - next[i][j];
                    for (int k = 0; k < 4; k++) {
                        int ii = i + dir[k][0];
                        int jj = j + dir[k][1];
                        if (ii < 0 || ii >= cur.length || jj < 0 || jj >= cur[0].length) continue;
                        next[ii][jj] = 1 - next[ii][jj];
                    }
                    int nextVal = conv(next);
                    //System.out.println(nextVal);
                    if (nextVal == 0) return sol;
                    if (set.add(nextVal)) {
                        q.add(next);
                    }
                }
            }
            }
            sol++;
        }
        return -1;
    }
    public int[][] cl(int[][] mat) {
        int[][] arr = new int[mat.length][mat[0].length];
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                arr[i][j] = mat[i][j];
            }
        }
        return arr;
    }
    public int conv(int[][] mat) {
        int sol = 0;
        for (int i = 0; i < mat.length; i++) {
            int cur = 0;
            for (int j = 0; j < mat[0].length; j++) {
                cur = cur * 2 + mat[i][j];
            }
            sol = sol * 10 + cur;
        }
        return sol;
    }
}
