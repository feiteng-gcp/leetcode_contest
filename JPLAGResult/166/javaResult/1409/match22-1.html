class Solution {
    
int[][] dirs = new int[][]{{0,1}, {0,-1}, {1,0}, {-1,0}};
    public int minFlips(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        
//         if (m == 1 && n == 1) {
//             return mat[0][0];
//         }
        
//         if (m == 1&& n == 2) {
//             if (m[0][1] ^ m[0][0] == 1) return -1;
//             return (m[0][1] + m[0][0]) / 2;
//         }
        
//         if (n == 1&& m == 2) {
//             if (m[1][0] ^ m[0][0] == 1) return -1;
//             return (m[1][0] + m[0][0]) / 2;
//         }
        
        Set<String> set = new HashSet<>();
        Deque<int[][]> q = new ArrayDeque<>();
        q.add(
cc(mat));
        set.add(ss(mat));
        int steps = 0;
        
        while (!q.isEmpty()) {
            int size = q.size();
            
            for (int kk = 0; kk < size; kk++) {
                int[][] cur = q.poll();
                if (valid(cur)) return steps;
                for (int i = 0; i < m; i++) {
                    for (int j = 0; j < n; j++) {
                        flip(cur, i, j);
                        String ts = ss(cur);
                        if (!set.contains(ts)) {q.add(cc(cur)); set.add(ts);}
                        flip(cur, i, j);
                    
}
                }
            }
            steps++;
        }
        return -1;
        
        
        
    }
    
    boolean valid(int[][] m) {
        for (int i = 0; i < m.length; i++) {
            for (int j = 0; j < m[0].length; j++) {
                if (m[i][j] == 1) return false
;
            }
        }
        return true;
    }
    
    void flip(int[][] m, int i, int j) {
        m[i][j] ^= 1;
        for (int[] d : dirs) {
            int ii = i + d[0];
            int jj = j + d[1];
            
            if (ii >= 0 && ii < m.length && jj >= 0 && jj < m[0].length) {
                m[ii][jj] ^= 1;
            }
        }
    }
    
    String ss (int[][] m) {
        String res = "";
        for (int i = 0; i < m.length; i++) {
            for (int j = 0; j < m[0].length; j++) {
                res += m[i][j];
            }
        }
        return res;
    }
    
    
    int[][] cc (int[][] m) {
        
int[][] res = new int[m.length][m[0].length];
        for (int i = 0; i < m.length; i++) {
            for (int j = 0; j < m[0].length; j++) {
                res[i][j] = m[i][j];
            }
        }
        return res;
    }
}
