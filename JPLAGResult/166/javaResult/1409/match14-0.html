class Solution {
    public int minFlips(int[][] mat) {
        if (check(mat)) {
            return 0;
        }          
        Queue<int[][]> queue = new LinkedList<>();
        Set<String> visited = new HashSet<>();
        queue.offer(mat);
        visited.add(convertToString(mat));
        int result = 0;
      
        while (queue.size() > 0) {
            int size = queue.size();
            // System.out.println(size);
            for (int index = 0; index < size; index++) {
                int[][] temp = queue.poll();
                for (int i = 0; i < temp.length; i++) {
                    for (int j = 0; j < temp[0].length; j++) {
                        int[][] newMatrix = fanzhuan(temp, i, j);
                        if (check(newMatrix)) {
                            return result + 1;
                        }
                        if (visited.add(convertToString(newMatrix))) {
                            // System.out.println(i + " s " + j);
                            queue.offer(newMatrix);
                        }
                    }
                }                
                
            }
            result++;
        }
        return -1;
    }        
    
    public boolean check(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                if (matrix[i][j] != 0) {
                    return false;
                }
            }
        }
        return true;
    }
    
    public int[][] fanzhuan(int[][] mat, int row, int col) {
        int[][] dir = new int[][]{{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
        int[][] r = new int[mat.length][mat[0].length];
        for (int i = 0; i < r.length; i++) {
            for (int j = 0; j < r[0].length; j++) {
                r[i][j] = mat[i][j];
            }
        }
        for (int i = 0; i < dir.length; i++) {
            int newRow = row + dir[i][0];
            int newCol = col + dir[i][1];
            if (newRow < 0 || newRow >= r.length || newCol < 0 || newCol >= r[0].length) {
                continue;
            }
            r[newRow][newCol] = r[newRow][newCol] == 1 ? 0 : 1;
        }
        r[row][col] = r[row][col] == 1 ? 0 : 1;
        return r;
    }
    
    public String convertToString(int[][] arr) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr[0].length; j++) {
                sb.append(arr[i][j]);
                sb.append("$");
            }
            sb.append("*");
        }
        return new String(sb);
    }
}
