class Solution {
    public int minFlips(int[][] mat) {
        HashSet<String> visited = new HashSet<String>();
        Queue<int[][]> q = new LinkedList<int[][]>();
        int layers = 0;
        
q.add(mat);
        while(q.size() > 0) {
            int size = q.size();
            while(size > 0) {
                int[][] top = q.poll();
                visited.add(arrToString(top));
                if(checkGoalState(top)) return layers
;
                for(int i = 0; i < top.length; i++) {
                    for(int j = 0; j < top[i].length; j++) {
                        int[][] newArr = copyArray(top);
                        newArr[i][j] = (newArr[i][j]+1) % 2;
                        for(int k = 0; k < 4; k++) {
                            int nx = i + dx[k];
                            int ny = j + dy[k];
                            if(nx >= 0 && nx < mat.length && ny >= 0 && ny < mat[0].length) {
                                newArr[nx][ny] = (newArr[nx][ny] + 1) % 2;
                            }
                        }
                        String arrStr = arrToString(newArr);
                        if(!visited.contains(arrStr)) {
                            
q.add(newArr);
                        }
                    }
                }
                size--;
            }
            layers++;
        }
        return -1;
    }
    
    private boolean checkGoalState(int[][] mat) {
        for(int i = 0; i < mat.length; i++) {
            for(int j = 0; j < mat[i].length; j++) {
                if (mat[i][j] == 1) return false
;
            }
        }
        return true;
    }
    
    private int[][] copyArray(
int[][] mat) {
        int[][] res = new int[mat.length][mat[0].length];
        for(int i = 0; i < mat.length; i++) {
            for(int j = 0; j < mat[i].length; j++) {
                res[i][j] = mat[i][j];
            }
        }
        return res;
    }
    
    private String arrToString(int[][] mat) {
        String[] res = new String[mat.length * mat[0].length];
        int cnt = 0;
        for(int i = 0; i < mat.length; i++) {
            for(int j = 0; j < mat[i].length; j++) {
                res[cnt++] = 
String.valueOf(mat[i][j]);
            }
        }
        return String.join(",",res);
    }
    
    private int[] dx = new int[] {1,0,-1,0};
    private int[] dy = new int[] {0,1,0,-1};
}
