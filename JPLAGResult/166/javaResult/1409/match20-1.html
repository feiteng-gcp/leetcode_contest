class Solution {
    public int minFlips(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        Set<String> set = new HashSet();
        Queue<int[][]> q = new LinkedList();
        int[][] copy = mat.clone();
        int step = 0;
        q.offer(copy);
        set.add(rep(mat));
        
while (!q.isEmpty()) {
            int size = q.size();
            while (size > 0) {
                int[][] cur = q.poll();
                if (allZero(cur)) {
                    return step;
                }
                for (int i = 0 ; i < m ; i++) {
                    for (int j = 0 ; j < n ; j++) {
                        int[][] temp = change(cur,i,j);
                        if (set.contains(rep(temp))) {
                            continue;
                        } else {
                            q.offer(temp);
                            set.add(
rep(temp));
                        }
                    }
                }
                size--;
            }
            //System.out.println("============");
            step++;
        }
        return -1;
    }
    private boolean allZero(int[][] g) {
        for (int i = 0 ; i < g.length ; i++) {
            for (int j = 0 ; j < g[0].length ; j++) {
                if (g[i][j] != 0) {
                    return false;
                }
            }
        }
        return true;
    }
    private int[][] change(int[][] cur, int row, int col) {
        int[][] res = 
new int[cur.length][cur[0].length];
        for (int i = 0 ; i < cur.length ; i++) {
            for (int j = 0 ; j < cur[0].length ; j++) {
                res[i][j] = cur[i][j];
            }
        }
        res[row][col] = 1 - cur[row][col];
        if (row - 1 >= 0) {
            res[row - 1][col] = 1 - cur[row - 1][col];
        }
        if (row + 1 < cur.length) {
            res[row + 1][col] = 1 - cur[row + 1][col];
        }
        if (col - 1 >= 0) {
            res[row][col - 1] = 1 - cur[row][col - 1];
        }
        if (col + 1 < cur[0].length) {
            res[row][col + 1] = 1 - cur[row][col + 1];
        }
        
        return res;
    }
    private String rep(int[][] g) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0 ; i < g.length ; i++) {
            for (int j = 0 ; j < g[0].length ; j++) {
                sb.append(g[i][j]);
            }
        }
        return sb.toString();
    }
}
