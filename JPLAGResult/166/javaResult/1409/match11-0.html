class Solution {
    private int[] dir = new int[]{0, 1, 0, -1, 0};
    private String encode(int[][] mat) {
        String ret = "";
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[i].length; j++) {
                ret += mat[i][j];
            }
            if (i < mat.length - 1) {
                ret += ":";
            }
        }
        return ret;
    }
    private String filp(String s, int x, int y) {
        String[] tokens = s.split(":");
        int[][] mat = new int[tokens.length][tokens[0].length()];
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                mat[i][j] = tokens[i].charAt(j) - '0';
            }
        }
        mat[x][y] = 1 - mat[x][y];
        for (int i = 0; i < 4; i++) {
            int nx = dir[i] + x, ny = dir[i + 1] + y;
            if (nx >= 0 && ny >= 0 && nx < mat.length && ny < mat[0].length) {
                mat[nx][ny] = 1 - mat[nx][ny];
            }
        }
        return encode(mat);
    }
    
    public int minFlips(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        Queue<String> q = new LinkedList<>();
        Set<String> vis = new HashSet<>();
        String src = encode(mat);
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                mat[i][j] = 0;
            }
        }
        String dest = encode(mat);
        vis.add(src);
        q.offer(src);
        for (int step = 0; step < 1000 && !q.isEmpty(); step++) {
            int len = q.size();
            while (len-- > 0) {
                String cur = q.poll();
                if (cur.equals(dest)) {
                    return step;
                }
                for (int i = 0; i < m; i++) {
                    for (int j = 0; j < n; j++) {
                        String next = filp(cur, i, j);
                        if (!vis.contains(next)) {
                            vis.add(next);
                            q.offer(next);
                        }
                    }
                }
            }
        }
        return -1;
    }
}
