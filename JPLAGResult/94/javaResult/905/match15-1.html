class Solution {
    public int lenLongestFibSubseq(int[] A) {
        int n = A.length;
        
int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int first = A[i];
                int second = A[j];

                int start = j + 1;
                int cnt = 2;
                while (start < n) {
                    int need = first + second;
                    int idx = Arrays.binarySearch(A, start, n, need);
                    if (idx < 0) {
                        break;
                    }
                    else {
                        cnt++;
                        res = Math.max(res, cnt);
                        
start = idx;
                        first = second;
                        second = need;
                    }
                }
            }
        }

        return res;
    }
}
