class Solution {
    public int lenLongestFibSubseq(int[] A) {
        HashSet<Integer> set = new HashSet<>();
        for (int i: A) {
            set.add(i);
        }
        int maximum = 0;
        for (int i = 0; i < A.length; i++) {
            for (int j = i + 1; j < A.length; j++) {
                int a = A[i], b = A[j];
                int c = a + b;
                int len = 2;
                if (set.contains(c)) {
                    while (set.contains(c)) {
                        len++;
                        a = b;
                        b = c;
                        c = a + b;
                    }
                    maximum = Math.max(maximum, len);
                }
            }
        }
        return maximum;
    }
}
