class Solution {
    public int lenLongestFibSubseq(int[] A) {
        if (A.length<3) {
            return 0;
        }
        Set<Integer> set = new HashSet<>();
        for (int num:A) {
            set.add(num);
        }
        int ret = 0;
        for (int i=0;i<A.length-2;i++) {
            for (int j=i+1;j<A.length-1;j++) {
                int temp = dfs(A, A[i], A[j], 0, set);
                if (temp>0) {
                    temp += 2;
                }
                ret = Math.max(ret, temp);
            }
        }
        return ret;
    }
    
    private int dfs(int[] A, int num1, int num2, int length, Set<Integer> set) {
        if (!set.contains(num1+num2)) {
            return length;
        }
        return dfs(A, num2, num1+num2, length+1, set);
    }
}
