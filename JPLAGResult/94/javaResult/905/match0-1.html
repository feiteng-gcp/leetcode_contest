class Solution {
    public int lenLongestFibSubseq(int[] A) {
        int max = 0;
        Set<Integer> set = new HashSet<>();
        for (int x : A) {
            set.add(x);
        }
        for (int j = 0; j < A.length; j++) {
            for (int i = j + 1; i < A.length; i++) {
                int tmp = 2;
                int num1 = A[j], num2 = A[i];
                while (set.contains(num1 + num2)) {
                    tmp++;
                    int tmp1 = num1 + num2;
                    
num1 = num2;
                    num2 = tmp1;
                }
                max = Math.max(max, tmp);
            }
        }
        return (max > 2) ? max : 0;
    }
}
