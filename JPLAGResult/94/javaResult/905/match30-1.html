public class Solution {

	public int lenLongestFibSubseq(int[] a) {

		HashSet<Integer> s = new HashSet<>();
		int n = a.length;
		for (int i = 0
; i < n; i++) {
			s.add(a[i]);
		}

		int res = 0;
		for (int i = 0; i < n; i++) {
			for (int j = i + 1; j < n - 1; j++) {
				int first = a[i];
				int second = a[j];
				int next = first + second; 
				int count = 2;
				if (!s.contains(next)) {
					continue;
				}
				do {
					count++;
					first = second;
					second = next;
					next = first + second;
				} while (s.contains(next));
				res = Math.max(res, count);
			}
		}

		return res;
	}

	public static void main(String[] args) throws Exception {
		long s = System.currentTimeMillis();

		System.out.println(new Solution().lenLongestFibSubseq(new int[] { 1, 3, 5 }));
		System.out.println(new Solution().lenLongestFibSubseq(new int[] { 1, 2, 3, 4, 5, 6, 7, 8 }));
		System.out.println(new Solution().lenLongestFibSubseq(new int[] { 1, 3, 7, 11, 12, 14, 18 }));
		System.out.println("Completed in: " + (System.currentTimeMillis() - s));
	}

}
