class Solution {
    class Pair {
        int n2;
        int n1;
        public Pair(int n2, int n1) {
            this.n2 = n2;
            this.n1 = n1;
        }
        @Override
        public boolean equals(Object obj) {
            Pair pair = (Pair) obj;
            return this.n2 == pair.n2 && this.n1 == pair.n1;
        }
        @Override
        public int hashCode() {
            return 102394 * n1 + n2;
        }
    }

    public int lenLongestFibSubseq(int[] A) {
        Map<Integer, Integer> index = new HashMap<>();
        Set<Pair> visited = new HashSet<>();
        for (int i = 0; i < A.length; i++) {
            
index.put(A[i], i);
        }
        int ans = 0;
        for (int i = A.length - 1; i >= 2; i--) {
            for (int j = i - 1; j >= 1; j--) {
                int n2 = A[i];
                int n1 = A[j];
                if (n2 - n1 >= n1) break;
                int cur = 0;
                if (visited.add(new Pair(n2, n1))) {
                    while (index.containsKey(n2 - n1) && index.get(n2 - n1) < index.get(n1)) {
                        int ix = index.get(n2 - n1);
                        cur++;
                        n2 = n1;
                        n1 = A[ix];
                    }
                }
                ans = Math.max(ans, cur);
            
}
        }
        return ans == 0 ? 0 : ans + 2;
    }
}
