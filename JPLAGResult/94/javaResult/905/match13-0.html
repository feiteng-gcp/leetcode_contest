class Solution {
    public int lenLongestFibSubseq(int[] A) {
        Set<Integer> set = new HashSet<>();
        for( int v:A) set.add(v);
        int max = 0,max2=0;
        for( int i=0 ; i<A.length-2 ; i++){
            for( int j=i+1 ; j<A.length-1 ; j++) {
                int a1 = A[i];
                int a2 = A[j];
                while (set.contains(a1 + a2)) {
                    max = max == 0 ? 3 : max + 1;
                    max2 = Math.max(max, max2);
                    int t = a1 + a2;
                    a1 = a2;
                    a2 = t;
                }
                max = 0;
            }
        }
        return max2;
    }
}
