class Solution {
    public int lenLongestFibSubseq(int[] A) {
        // int[] max = new int[]{0};
        // helper(0, 0, A, max);
        int len = A.length;
        int max = 0;
        for (int i = 0; i < len - 2; i++) {
            
            // System.out.println("f " + A[i]);
            
            for (int j = i + 1; j < len - 1; j++) {
                // System.out.println("s " + A[j]);
                int size = 2;
                int curr = A[i] + A[j];
                
                int lastTwo = A[i];
                int lastOne = A[j];
                
                int nextIndex = helper(j + 1, len - 1, A, curr);
                // System.out.println("curr " + curr + " next " + A[nextIndex]);
                while (nextIndex != -1) {
                    size++;
                    lastTwo = lastOne;
                    curr = lastOne + A[nextIndex];
                    lastOne = A[nextIndex];
                    nextIndex = helper(j + 1, len - 1, A, curr);
                    // System.out.println("curr " + curr + " next " + A[nextIndex]);
                }
                if (size > 2) max = Math.max(max, size); 
            }    
        }
        
        return max;
    }
    
    private int helper(int left, int right, int[] A, int num) {
        // System.out.println("dad " + num );
        while (left + 1 < right) {
            
            int mid = left + (right - left) / 2;
            
            if (A[mid] == num) {
                return mid;
            } else if (A[mid] < num) {
                left = mid;
            } else {
                right = mid;
            }
            
        }
        
        if (A[left] == num) return left;
        if (A[right] == num) return right;
        return -1;
        
    }
        
    
}
