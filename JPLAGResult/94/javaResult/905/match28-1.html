class Solution {
     public int lenLongestFibSubseq(int[] A) {
        int len = 0;
        int[][] dp = new int[A.length][A.length];
        
for (int i = 0; i < A.length; i++) {
            for (int j = i + 1; j < A.length; j++) {
                int max = helper(A, i, j, dp);
//                System.out.println(A[i] + ":" + A[j] + " = " + max);
                if (max > 2) len = Math.max(len, max);
            
}
        }
        return len;
    }

    public int helper(int[] A, int m, int n, int[][] dp) {
        if (dp[m][n] != 0) return dp[m][n];

        int next = A[m] + A[n];
        if (n == A.length - 1) dp[m][n] = 2;

        for (int i = n + 1; i < A.length; i++) {
            if (A[i] > next) break;
            if (A[i] == next) {
                dp[m][n] = 1 + helper(A, n, i, dp);
                return dp[m][n];

            }
        }

        dp[m][n] = 2;

        return dp[m][n];

    }
}
