class Solution {
    public int lenLongestFibSubseq(int[] A) {
		Set<Integer> set = new HashSet<>();
		for (int a : A) {
			set.add(a);
		}
		int max = 0;
		for (int i = 0; i < A.length; i++) {
			for (int j = i + 1; j < A.length; j++) {
				int maxPossible = A.length - j + 1;
				if (maxPossible <= max) {
					break;
				}
				int next = A[i] + A[j];
				if (set.contains(next)) {
					int currLeng = 3;
					int prev = A[j];
					int curr = next;
					while (set.contains(prev + curr)) {
						int tmp = curr;
						curr = prev + curr;
						prev = tmp;
						currLeng++;
					}
					max = Math.max(max, currLeng);
				}
			}
		}
		return max;
    }
}
