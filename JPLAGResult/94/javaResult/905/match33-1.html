class Solution {
    public int lenLongestFibSubseq(int[] A) {
        int[][] dp = new int[A.length][A.length];
        for(int i = 0;i<A.length;i++){
            dp[i][0] = 0;
        
}
        int max = 0;
        for(int i = 1;i<A.length;i++){
            for(int j = 1;j<i;j++){
                int lo = 0;
                int hi = j;
                while(lo+1<hi){
                    int mid = lo + (hi-lo)/2;
                    if(A[mid]+A[j] >= A[i]){
                        hi = mid;
                    }else{
                        lo = mid;
                    }
                    
                }
                int k = lo;
                if(A[k] + A[j] == A[i]) {
                    dp[i][j] = Math.max(dp[j][k]+1,dp[i][j]);}
                 k = hi;
                if(A[k] + A[j] == A[i]) {
                    dp[i][j] = Math.max(dp[j][k]+1,dp[i][j]);
}
                max = Math.max(dp[i][j],max);
            }
        }
        return max>0?max+2:max;
    }
}
