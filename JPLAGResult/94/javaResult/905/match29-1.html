class Solution {
    public int lenLongestFibSubseq(int[] A) {
        
int longest = 0;
        for (int i = 0; i < A.length - 1; i ++) {
            for (int j = i + 1; j < A.length; j ++) {
                int localLongest = findFab(A[i], A[j], A, 2);
                if (localLongest > 2) {
                    longest = Math.max(longest, localLongest);
                }
            
}
        }
        return longest;
    }
    
    private int findFab(int a, int b, int[]A, int l) {
        int index = binarySearch(a + b, A);
        if (index > 0 && index < A.length) {
            return findFab(b, A[index], A, l + 1);
        } else {
            return l;
        }
    }
    
    private int binarySearch(int target, int[] A) {
        int lo = 0;
        int hi = A.length - 1;
        while (lo <= hi) {
            int mid = (lo + hi) >>> 1;
            if (A[mid] == target) {
                return mid;
            }
            if (A[mid] > target) {
                hi = mid - 1;
            } else {
                lo = mid + 1;
            }
        }
        return -1;
    }
}
