class Solution {
    public int lenLongestFibSubseq(int[] A) {
        TreeSet<Long> s = new TreeSet<>();
        for (int i = 0; i < A.length; i++) {
            long v = A[i];
            s.add(v);
        }
        int ans = 0;
        for (int i = 0; i < A.length; i++) {
            loop:
            for (int j = i + 1; j < A.length; j++) {
                long a = A[i], b = A[j], c = a + b;
                int len = 2;
                while (true) {
                    if (!s.contains(c)) {
                        ans = Math.max(ans, len);
                        continue loop;
                    } else {
                        len++;
                        a = b;
                        b = c;
                        c = a + b;
                    }
                }
            }
        }
        return ans >= 3 ? ans : 0;
    }
}
