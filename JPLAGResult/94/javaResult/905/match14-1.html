class Solution {
    public int lenLongestFibSubseq(int[] A) {
        Arrays.sort(A);
        Set<Integer> set = new HashSet<>();
        for(int k : A) set.add(k);
        int n = A.length, re = 0;
        for(int i = 0; i < n; i++)
        {
            
            for(int j = i + 1; j < n; j++)
            {
                int a = A[i], b = A[j];
                if(a == b) continue;
                
int len = 0;
                while(set.contains(a + b))
                {
                    len++;
                    int c = a + b;
                    a = b;
                    b = c;
                }
                if(len > 0) re = Math.max(re, len + 2)
;
            }
        }
        return re;
    }
}
