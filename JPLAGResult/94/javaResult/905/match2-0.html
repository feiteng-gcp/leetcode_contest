class Solution {
    public int lenLongestFibSubseq(int[] A) {
        int l = A.length;
        HashSet<Integer> set = new HashSet<>();
        for (int i : A) set.add(i);
        int res = 0;
        for (int i = 0; i < l; i++) {
            for (int j = i+1; j < l; j++) {
                if (res > l-j+1) break;
                res = Math.max(res, helper(set, A[i], A[j], 2));
            }
        }
        return res <= 2 ? 0 : res;
    }
    private int helper(HashSet<Integer> set, int a1, int a2, int l) {
        int a3 = a1 + a2;
        if (set.contains(a3)) return helper(set, a2, a3, l+1);
        return l;
    }
}
