class Solution {
    public int lenLongestFibSubseq(int[] A) {
        int max = 1;
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0 ; i < A.length ;i++) {
            map.put(A[i], i);
        }
        
for (int i =0; i < A.length -2; i ++) {
            for (int j = i+ 1; j < A.length - 1; j ++) {
                int last = i;
                int cur = j;
                int len = 2;
                while(map.containsKey(A[last] + A[cur]) && map.get(A[last] + A[cur]) > cur) {
                    len ++;
                    int next= map.get(A[last] + A[cur]);
                    last = cur;
                    cur = next;
                }
                max = Math.max(len, max);
            
}
        }
        return max <= 2? 0: max;
    }
}
