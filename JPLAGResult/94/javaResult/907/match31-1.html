class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        int left = 1;
        int right = (int) 1e9;
        while(left <= right) {
            int mid = left + (right - left) / 2;
            
int count = 0;
            for(int pile : piles) {
                count += pile / mid;
                if(pile % mid != 0) {
                    count++;
                }
                if(count > H) {
                    break;
                }
            }
            
            if(count <= H) {
                right = mid - 1;
            
}else {
                left = mid + 1;
            }
        }
        
        return left;
    }
}
