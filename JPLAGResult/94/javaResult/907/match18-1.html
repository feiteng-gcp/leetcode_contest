class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        
        int max = Integer.MIN_VALUE;
        
        for (int i = 0; i < piles.length; i++) {
            max = Math.max(piles[i], max);
        }
        
        
int left = 1, right = max;
        
        while (left + 1 < right) {
            int mid = left + (right - left) / 2;
            if (can(mid, piles, H)) {
                right = mid;
            } else {
                left = mid;
            }
        }
        
        if (can(left, piles, H)) return left;
        if (can(right, piles, H)) return right;
        
return -1;
    }
    
    private boolean can(int k, int[] piles, int H) {
        
        int total = 0;
        for (int i = 0; i < piles.length; i++) {
            
            int div = piles[i] / k;
            int rem = piles[i] % k;
            if (rem != 0) div++;
            total += div;
        }
        
        return total <= H;
    }
    
}
