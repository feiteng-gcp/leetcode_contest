class Solution {
public int minEatingSpeed(int[] piles, int H) {
    	int
 lf = 1;
    	int rt = Integer.MAX_VALUE - 10000;
    	while (lf < rt) {
    		int mid = lf + (rt - lf) / 2;
    		if (!ok(mid, piles, H)) {
    			lf = mid + 1;
    		}
    		else {
    			rt = mid;
    		}
    	}
    	ret
urn rt;
    }
    
    boolean ok(int mid, int[] piples, int H) {
    	int sum = 0;
    	int n = piples.length;
    	for (int i = 0; i < n; ++i) {
    		sum += Math.ceil(piples[i] * 1.0 / mid);
    	}
    	return sum <= H;
    }
}
