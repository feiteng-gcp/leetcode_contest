class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        Arrays.sort(piles);
        if (H == piles.length) {
            return piles[piles.length - 1];
        
}
        int left = 1;
        int right = piles[piles.length - 1];
        while (left <= right) {
            int mid = left + (right - left ) / 2;
            if (check(piles, mid, H)) {
                right = mid - 1;
            
} else {
                left = mid + 1;
            }
        }
        return left;
    }
    public boolean check(int[] piles, int cur, int H) {
        int curNeed = 0;
        for (int i : piles) {
            if (i % cur == 0) {
                curNeed += i / cur;
            } else {
                curNeed += i / cur + 1;
            }
        }
        return curNeed <= H;        
    }
}
