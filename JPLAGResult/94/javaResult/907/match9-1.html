class Solution {
    private boolean canEat(long k, int[] piles, int H) {
        int h = 0;
        for (int p: piles) {
            h +=(int) Math.ceil(p/(double)k);
            if (h > H) {
                return false;
            
}
        }
        return true;
    }
    public int minEatingSpeed(int[] piles, int H) {
        long l = 1;
        long h = 0;
        for (int p : piles) {
            if (p > h) {
                h = p;
            }
        }
        
while(l < h - 1) {
            long mid = (l + h) /2;
            if (canEat(mid, piles,H)) {
                h = mid;
            }
            else {
                l = mid;
            }
        }
        if (canEat(l,piles,H)) {
            return (int)l;
        }
        return (int)h;
    }
}
