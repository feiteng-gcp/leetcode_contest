class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        if (piles == null || piles.length == 0) {
            return 0;
        }
        long lo = 1;
        long hi = Integer.MAX_VALUE;
        while (lo < hi) {
            long mid = lo + (hi - lo) / 2;
            if (check(piles, mid, H)) 
{
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        
return (int)lo;
    }
    
    private boolean check(int[] piles, long speed, int H) {
        long ans = 0;
        for (int p : piles) {
            if ((long)p % speed == 0) {
                ans += (long)p / speed;
            } else 
{
                ans += (long)p / speed + 1;
            }
        }
        return ans <= H;
    }
}
