class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        
        long up = 0;
        
        for(int p:piles){
            up = Math.max(up,(long)p+1); 
        }
        
        long ud = 0;
        
        while(up-ud>1){
            long m = (up+ud)/2;
            long an = 0;
            for(int p:piles){
                long t = (long)p;
                long y = (t+m-1)/m;
                an += y;
            }
            if(an>H)
{
                ud = m;
            }else{
                up = m;
            }

        }
        return (int)up;        
        
        
        
        
    }
}
