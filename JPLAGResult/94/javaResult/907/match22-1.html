class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        int min = piles[0], max = piles[0];
        for (int i = 0; i < piles.length; ++i) {
            min = Math.min(min, piles[i]);
            max = Math.max(max, piles[i]);
        }
        min = 1;
        while (min < max) {
            int mid = min + (max - min) / 2;
            if (possible(piles, mid, H)) max = mid;
            else min = mid + 1
;
        }
        return min;
    }
    
    public boolean possible (int[] piles, int mid, int H) {
        long cnt = 0;
        for (int pile : piles) {
            cnt += pile / mid;
            if (pile % mid > 0) cnt++;
        }
        return cnt <= H;
    }
}
