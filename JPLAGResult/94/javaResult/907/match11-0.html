class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        int start = 1, end = Integer.MAX_VALUE;
        int ans = 0;
        while (start <= end) {
            int m = start + (end - start) / 2;
            int total = 0;
            for (int p :piles) {
                int h = (p + m - 1)/m;
                total += h;
            }
            if (total <= H) {
                ans = m;
                end = m - 1;
            } else start = m + 1;
        }
        return ans;
    }
}
