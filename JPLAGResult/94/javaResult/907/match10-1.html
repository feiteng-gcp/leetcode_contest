class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        Arrays.sort(piles);
        if (piles.length == H) return piles[piles.length - 1];
        
int left = 1;
        int right = Integer.MAX_VALUE;
        while (left < right) {
            int mid = left + (right - left) / 2;
            int hours = 0;
            for (int i : piles) {
                if (i % mid == 0) hours += i / mid;
                else hours += (i / mid) + 1
;
            }
            if (hours <= H) right = mid;
            else left = mid + 1;
        }
        return left;
    }
}
