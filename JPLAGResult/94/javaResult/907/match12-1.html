class Solution {
	private boolean isOK(int[] piles, int H, int val) {
		int cc = 0;
		for (int i = 0; i < piles.length; ++i) {
			cc += piles[i] / val;
			if (piles[i] % val != 0) ++cc;
		}
		return cc <= H;
	}

	public 
int minEatingSpeed(int[] piles, int H) {
		int vmax = Integer.MAX_VALUE - 2;
		int vmin = 1;

		int res = vmax;
		while (vmin <= vmax) {
			int mid = (vmax + vmin) / 2;
			if (isOK(piles, H, mid)) {

				res = mid;
				vmax = mid - 1;
			} else vmin = mid + 1;
		}
		return res;
	}
}
