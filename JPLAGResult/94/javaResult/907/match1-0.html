class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        if (piles == null || piles.length == 0 || H <= 0) {
            return 0;
        }
        
        int min = 1, max = 0;
        for (int i = 0; i < piles.length; i++) {
            if (piles[i] > max) {
                max = piles[i];
            }
        }
        
        while (min + 1 < max) {
            int mid = min + (max - min) / 2;
            boolean ans = checkHelper(mid, piles, H);
            //System.out.println("[" + min + ", " + max + "] " + mid + " => " + ans);
            if (ans) {
                max = mid;
            } else {
                min = mid + 1;
            }
        }
        
        if (checkHelper(min, piles, H)) {
            return min;
        }
        return max;
    }
    
    private boolean checkHelper(int speed, int[] piles, int H) {
        int count = 0;
        for (int i = 0; i < piles.length; i++) {
            int ans = piles[i] / speed;
            if ((piles[i] % speed) > 0) {
                ans++;
            }
            
            //System.out.println(i + " => " + ans);
            
            count += ans;
        }
        return count <= H;
    }
}
