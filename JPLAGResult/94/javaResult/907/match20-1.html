class Solution {
    public int minEatingSpeed(int[] piles, int h) {
       //  [3,6,7,11], H = 8
        //  1 2 2 3
        // 3 7
        int low = 0;
        int high = 0;
        for(int i=0;i<piles.length;i++){
            high = Math.max(piles[i],high);
        }
        int k = high;
        while(low<=high){
            int mid = (low+high)/2;
            if(check(h,piles,mid)){
                k = Math.min(k,mid);
                high = mid-1;
            }
            else 
low = mid+1;
        }
        return k;
    }
    
    public boolean check(int h,int[] piles,int speed){
            double time = 0.0;
            for(int i=0;i<piles.length;i++){
                time+=(int)(Math.ceil((double)piles[i]/speed));
            }
            if(time<=h) return true;
            return false;
    }
}
