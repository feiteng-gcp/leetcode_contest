class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        int lo = 1;
        int hi = 0;
        for (int i = 0; i < piles.length; i ++) {
            hi += piles[i];
            if (hi < 0) {
                hi = Integer.MAX_VALUE;
                break;
            }
        }
        while (lo < hi) {
            int mid = (lo + hi) >>> 1;
            int hrs = countHours(mid, piles);
            System.out.println(mid + " " + hrs);
            if (hrs <= H) 
{
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return lo;
    }
    
    private int countHours(
int k, int[] piles) {
        int cnt = 0;
        for (int i : piles) {
            if (i % k == 0) {
                cnt += i / k;
            } else {
                cnt += i / k + 1;
            }
        }
        return cnt;
    }
}
