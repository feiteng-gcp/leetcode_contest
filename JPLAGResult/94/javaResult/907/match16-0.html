class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        int sum = 0;
        int max = 0;
        for (int each:piles) {
            sum += each;
            max = Math.max(max, each);
        }
        int ret = sum/H;
        if (sum%H!=0) {
            ret++;
        }
        int start = ret;
        int end = max;
        while (start<=end) {
            int mid = start+(end-start)/2;
            if (canFinish(piles, H, mid)) {
                if (mid>start && canFinish(piles, H, mid-1)) {
                    end = mid-1;
                } else {
                    return mid;
                }
            } else {
                start = mid+1;
            }
        }
        return max;
    }
    
    private boolean canFinish(int[] piles, int H, int speed) {
        int time = 0;
        for (int each:piles) {
            time += each/speed;
            if (each%speed!=0) {
                time++;
            }
            if (time>H) {
                return false;
            }
        }
        return time<=H;
    }
}
