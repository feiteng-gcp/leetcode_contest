class Solution {
    public int minEatingSpeed(int[] piles, int H) {
        if(piles.length == 0)
            return 0;
        int max = piles[0];
        // int min = piles[0];
        for(int i = 1; i < piles.length; i++){
            max = Math.max(max, piles[i]);
            // min = Math.min(min, piles[i]);
        }
        int right = max;
        int left = 1;
        while(left < right){
            int mid = (right - left)/2 + left;
            int hour = 0;
            for(int banana: piles){
                hour += banana/mid;
                if(banana % mid != 0)
                    hour++;
            }
            // System.out.println(mid+","+hour);
            if(hour <= H)
                right = mid;
            else
                left = mid+1;
        }
        return left;
    }
}
