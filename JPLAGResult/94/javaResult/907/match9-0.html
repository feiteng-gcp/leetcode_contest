class Solution {
    public boolean isValid(int[] piles, int H, long K) {
        long needH = 0;
        for (int pInt : piles) {
            long p = pInt;
            if (p % K == 0) needH += (p / K);
            else needH += (p / K + 1);
            if (needH > (long)H) return false;
        }
        return true;
    }
    
    public int minEatingSpeed(int[] piles, int H) {
        int n = piles.length;
        if (n == 0) return 0;
        long sum = 0;
        for (int i : piles) sum += i;
        long minBound = sum / (long)H;
        if (minBound == 0) minBound = 1;
        long maxBound = minBound * 2;
        while (!isValid(piles, H, maxBound)) {
            minBound = maxBound;
            maxBound *= 2;
        }
        while (minBound < maxBound) {
            long mid = (minBound + maxBound) / 2;
            if (isValid(piles, H, mid)) {
                maxBound = mid;
            }
            else minBound = mid + 1;
        }
        return ((int) minBound);
    }
}
