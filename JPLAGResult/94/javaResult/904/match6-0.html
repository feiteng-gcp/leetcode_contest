/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        List<Integer> curLeft = new ArrayList();
        List<Integer> curRight = new ArrayList();
        getLeaf(root1, curLeft);
        getLeaf(root2, curRight);
        if (curLeft.size() != curRight.size()) {
            return false;
        }
        for (int i = 0; i < curLeft.size(); i++) {
            if (curLeft.get(i) != curRight.get(i)) {
                return false;
            }
        }
        return true;
    }
    public void getLeaf(TreeNode root, List<Integer> cur) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            cur.add(root.val);
        }
        getLeaf(root.left, cur);
        getLeaf(root.right, cur);
        
    }
}
