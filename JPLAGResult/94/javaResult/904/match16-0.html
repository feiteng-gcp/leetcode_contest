/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    void dfs(TreeNode r, List<Integer> ans) {
        if (r == null) return;
        if (r.left == null && r.right == null) ans.add(r.val);
        dfs(r.left, ans);
        dfs(r.right, ans);
    }
    
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        List<Integer> a = new ArrayList<>(), b = new ArrayList<>();
        dfs(root1, a);
        dfs(root2, b);
        return a.equals(b);
    }
}
