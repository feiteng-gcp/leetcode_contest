/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        Queue<Integer> q1 = new LinkedList<>();
        Queue<Integer> q2 = new LinkedList<>();
        helper(root1, q1);
        helper(root2, q2);
        while (!q1.isEmpty()) {
            int can1 = q1.poll();
            if (q2.isEmpty()) return false;
            int can2 = q2.poll();
            if (can1 != can2) return false;
        }
        
return q2.isEmpty();
    }
    private void helper(TreeNode p, Queue<Integer> q) {
        if (p == null) return
;
        if (p.left == null && p.right == null) {
            q.offer(p.val);
            return;
        }
        helper(p.left, q);
        helper(p.right, q);
    }
}
