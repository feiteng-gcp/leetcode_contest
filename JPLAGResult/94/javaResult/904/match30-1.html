/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        List<TreeNode> l1 = new LinkedList<>();
        List<TreeNode> l2 = new LinkedList<>();
        helper(root1, l1);
        helper(root2, l2);
        if (l1.size() != l2.size()) return false;
        for (int i = 0; i < l1.size(); i++) {
            if (l1.get(i).val != l2.get(i).val) return false;
        }
        return true;
    }
    
    
private void helper(TreeNode root, List<TreeNode> l1) {
        if (root == null) return;
        if (root.left == null && root.right == null) {
            l1.add(root); 
            return;
        }
        if (root.left != null) { 
            helper(root.left, l1);
        }
        if (root.right != null) { 
            helper(root.right, l1);
        }
    }
}
