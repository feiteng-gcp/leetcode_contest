/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        List<Integer> leaves1 = getLeaves(root1);
        List<Integer> leaves2 = getLeaves(root2);
        return isSame(leaves1, leaves2);
    }
    
    private List<Integer> getLeaves(TreeNode root) {
        List<Integer> leaves = new ArrayList<>();
        traverse(root, leaves);
        return leaves;
    }
    
    private void traverse(TreeNode root, List<Integer> leaves) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            leaves.add(root.val);
        }
        traverse(root.left, leaves);
        traverse(root.right, leaves);
    }
    
    private boolean isSame(List<Integer> leaves1, List<Integer> leaves2) {
        if (leaves1.size() != leaves2.size()) {
            return false;
        }
        for (int i = 0; i < leaves1.size(); i ++) {
            if (leaves1.get(i) != leaves2.get(i)) {
                return false;
            }
        }
        return true;
    }
}
