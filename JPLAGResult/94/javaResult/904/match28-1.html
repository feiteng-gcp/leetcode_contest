import java.util.*;


class Solution {
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {

        List<Integer> accum1 = new ArrayList<Integer>();
        List<Integer> accum2 = new ArrayList<Integer>();
        dfs(root1, accum1);
        dfs(root2, accum2);

        if (accum1.size() != accum2.size()) { return false; }
        for (int i = 0; i < accum1.size(); i++) {
            if (accum1.get(i)+1 != accum2.get(i)+1) { return false; }
        }
        return true;

    }

    private void dfs(TreeNode node, List<Integer> accum) {

        if (node == null) { return; }

        // leaf node.
        if (node.left == null && node.right == null) {
            accum.add(node.val);
            return;
        }

        // non-leaf node, could have left or right or both children.
        if (node.left != null) {
            dfs(node.left, accum);
        }
        if (node.right != null) {
            dfs(node.right, accum);
        }

    }
}
