class Solution {
    public int robotSim(int[] commands, int[][] obstacles) {
        int[] dx = {0,1,0,-1};
        int[] dy = {1,0,-1,0};
        Map<Integer, Set<Integer>> map = new HashMap<Integer,Set<Integer>>();
        for(int i = 0;i<obstacles.length;i++){
            if(map.containsKey(obstacles[i][0])){
                map.get(obstacles[i][0]).add(obstacles[i][1]);
            }else{
                HashSet set = new HashSet();
                set.add(obstacles[i][1]);
                map.put(obstacles[i][0],set);
            }
        }
        int x= 0;
        int y = 0;
        int dir = 0;
        int max = 0;
        for(int i = 0;i<commands.length;i++){
            if(commands[i] == -2){
                dir = (dir+3)%4;
            }else if(commands[i] == -1){
                dir = (dir+1)%4;
            }else{
                for(int j = 0;j<commands[i];j++){
                    if(map.containsKey(x+dx[dir]) && map.get(x+dx[dir]).contains(y+dy[dir])) break;
                    x += dx[dir];
                    y += dy[dir];
                    max = Math.max(x*x+y*y,max);
                }
            }
        }
        return max;
    }
}
