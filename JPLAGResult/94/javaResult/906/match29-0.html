class Solution {
    public int robotSim(int[] commands, int[][] obstacles) {
        HashMap<Integer, Set<Integer>> hm = new HashMap<>();
        for(int[] i:obstacles){
            if(!hm.containsKey(i[0])){
                hm.put(i[0], new HashSet<>());
            }
            hm.get(i[0]).add(i[1]);
        }
        int[] dir = new int[]{0, 1};
        int[] po = new int[]{0, 0};
        int max = 0;
        for(int c: commands){
            //System.out.println(po[0] + " "+po[1]);
            //System.out.println(dir[0] + " "+dir[1]);
            if(c == -2){
                if(dir[0] == 0){
                    dir[0] = -dir[1];
                    dir[1] = 0;
                }else{
                    dir = new int[]{dir[1], dir[0]};
                }
            }else if(c == -1){
                if (dir[1] == 0){
                    dir = new int[]{ 0, -dir[0]};
                }else{
                    dir = new int[]{dir[1], 0};
                }
            }else{
                for(int i = 0; i < c; i++){
                    po[0] += dir[0];
                    po[1] += dir[1];
                    if(hm.getOrDefault(po[0], new HashSet<>()).contains(po[1])){
                        po[0] -= dir[0];
                        po[1] -= dir[1];
                        break;
                    }
                }
                max = Math.max(max, po[1]* po[1] + po[0] *po[0]);
            }
        }
        return max;
    }
}
