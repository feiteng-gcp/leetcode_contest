class Solution {
    public int robotSim(int[] commands, int[][] obstacles) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for(int[] obs : obstacles)
        {
            int x = obs[0], y = obs[1];
            if(!map.containsKey(x)) map.put(x, new HashSet<>())
;
            map.get(x).add(y);
        }
        int[] dirs = {0, 1, 2, 3};
        int[] dy = {1, 0, -1, 0}, dx = {0, 1, 0, -1};
        int direction = 0, x = 0, y = 0, re = 0;
        for(int command : commands)
        {
            if(command < 0)
            {
                if(command == -1) direction = (direction + 1) % 4;
                if(command == -2) direction = (direction - 1 + 4) % 4;
            }
            else
            {
                while(command-- > 0)
                {
                    int nx = x + dx[direction], ny = y + dy[direction];
                    if(!map.containsKey(nx) || !map.get(nx).contains(ny))
                    {
                        x = nx;
                        y = ny;
                    }
                }
                re = Math.max(re, x * x + y * y);
            }
        }
        return re;
    }
}
