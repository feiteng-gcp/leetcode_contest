class Solution {
    
public int robotSim(int[] commands, int[][] obstacles) {
        HashSet<String> obSet = new HashSet<>();
        for (int i = 0; i < obstacles.length; i++) {
            
obSet.add(obstacles[i][0] + " " + obstacles[i][1]);
        }
        int x = 0, y = 0, d = 0, ans = 0;
        for (int i = 0; i < commands.length; i++) {
            if (commands[i] == -2) {
                d = (d + 3) % 4;
                continue;
            }
            if (commands[i] == -1) {
                d = (d + 1) % 4;
                continue;
            }
            int tx = x, ty = y;
            if (d == 0) {
                ty += commands[i];
                for (int j = y + 1; j <= ty; j++) {
                    String tmp = tx + " " + j;
                    if (obSet.contains(tmp)) {
                        ty = j - 1;
                        break;
                    }
                }
            }
            if (d == 1) {
                tx += commands[i];
                for (int j = x + 1; j <= tx; j++) {
                    String tmp = j + " " + ty;
                    if (obSet.contains(tmp)) {
                        tx = j - 1;
                        break;
                    }
                }
            }
            if (d == 2) {
                ty -= commands[i];
                for (int j = y - 1; j >= ty; j--) {
                    String tmp = tx + " " + j; 
                    if (obSet.contains(tmp)) {
                        ty = j + 1;
                        break;
                    }
                }
            }
            if (d == 3) {
                tx -= commands[i];
                for (int j = x - 1; j >= tx; j--) {
                    String tmp = j + " " + ty;
                    if (obSet.contains(tmp)) {
                        tx = j + 1;
                        break;
                    }
                }
            }
            x = tx;
            y = ty;
            if (x * x + y * y > ans) ans = x * x + y * y;
        }
        return ans;
    }
}
