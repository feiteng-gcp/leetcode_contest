class Solution {
    
    private Map<Integer, Set<Integer>> map;
    
    private int x, y;
    
    private int[][] directions;
    
    private int index;
    
    private int result;
    
    public int robotSim(int[] commands, int[][] obstacles) {
        if (commands == null || commands.length == 0) {
            return result;
        }
        
        map = new HashMap<Integer, Set<Integer>>();
        if (obstacles != null && obstacles.length > 0) {
            for (int i = 0; i < obstacles.length; i++) {
                int x = obstacles[i][0], y = obstacles[i][1];
                if (!map.containsKey(x)) {
                    map.put(x, new HashSet<Integer>());
                }
                map.get(x).add(y);
            }
        }
        
        directions = new int[4][2];
        directions[0] = new int[]{0, 1};
        directions[1] = new int[]{1, 0};
        directions[2] = new int[]{0, -1};
        directions[3] = new int[]{-1, 0};
        
        for (int i = 0; i < commands.length; i++) {
            moveHelper(commands[i]);
        }
        return result;
    }
    
    private void moveHelper(int c) {
        if (c == -1) {
            index++;
            if (index == 4) {
                index = 0;
            }
        } else if (c == -2) {
            index--;
            if (index == -1) {
                index = 3;
            }
        } else if (c >= 1 && c <= 9) {
            int[] direction = directions[index];
            for (int i = 0; i < c; i++) {
                int nx = x + direction[0];
                int ny = y + direction[1];
                if (map.containsKey(nx) && map.get(nx).contains(ny)) {
                    return;
                }
                
                x = nx;
                y = ny;
                result = Math.max(result, x * x + y * y);
            }
        }
    }
}
