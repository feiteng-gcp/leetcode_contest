public class Solution {
    public class tuple{
        int x;
        int y;
        public tuple(int x,int y){
            this.x = x;
            this.y = y;
        }
        
        @Override
        public int hashCode(){
            return this.x+this.y;
        }
        
        @Override
        public boolean equals(Object o){
            tuple another = (tuple)(o);
            if(another.x==this.x&&another.y==this.y) return true;
            return false;
        }
    }
    
public int robotSim(int[] commands, int[][] Obstacles) {
        Set<tuple> obstacles = new HashSet();
        for(int i=0;i<Obstacles.length;i++){
            obstacles.add(new tuple(Obstacles[i][0],Obstacles[i][1]));
        }
        int direct = 0; // 0 means north 1 means west 2means south 3 means east
        tuple cur = new tuple(0,0);
        int res = 0;
        for(int i=0;i<commands.length;i++){
            int command = commands[i];
            
if(command==-2){
                direct = (direct+1)%4;
            }
            else if(command==-1){
                direct = (direct+3)%4;
            }
            else{
                int move = command;
                while(move>0){
                    tuple next;
                    int x = cur.x;
                    int y = cur.y;
                    if(direct==0){
                        next = new tuple(x,y+1);
                        if(obstacles.contains(next)) break;
                        cur.y +=1;
                    }
                    else if(direct==1){
                        next = new tuple(x-1,y);
                        if(obstacles.contains(next)) break;
                        cur.x -=1;
                    }
                    else if(direct==2){
                         next = new tuple(x,y-1);
                         if(obstacles.contains(next)) break;
                          cur.y -=1;
                    }
                    else{
                        next = new tuple(x+1,y);
                        if(obstacles.contains(next)) break;
                        cur.x +=1;
                    }
                    move--;
                }
            
}
           res = Math.max(cur.x*cur.x+cur.y*cur.y,res);
        }
        return res;
        // x x x x   
        // x 
        // x
        // x
        // x
        //   x
        //   x
        //   x
        //   x          
        // x x  
        // x
        // x
        // x
        // x
    }
}
