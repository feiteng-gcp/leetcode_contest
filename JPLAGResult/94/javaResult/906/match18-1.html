class Solution {
    
public int robotSim(int[] commands, int[][] obstacles) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for(int[] coord: obstacles){
            int x = coord[0];
            int y = coord[1];
            if(!map.containsKey(x))
                map.put(x, new HashSet<Integer>());
            map.get(x).add(y);
        }
        int max = 0;
        
int[][] directions = {{0,1},{1,0},{0,-1},{-1,0}};
      
        int turn = 0;
        int x = 0;
        int y = 0;
        for(int command: commands){
            if(command > 0){
                int xDir = (turn >= 0)? directions[turn][0]:-directions[-turn][0];
                int yDir = (turn >= 0)? directions[turn][1]:directions[-turn][1];
                for(int i=0; i < command; i++){
                    if(!map.containsKey(x+xDir) || !map.get(x+xDir).contains(y+yDir)){
                        x += xDir;
                        y += yDir;
                    }
                }
            }
            else if(command == -1){
                turn++;
                turn = turn%4;
            }
            else if(command == -2){
                turn--;
                turn = turn % 4;
            }
            max = Math.max(max,x*x+y*y);
        }
        return  max;
    }
}
