class Solution
{
    
public int robotSim(int[] commands, int[][] obstacles)
    {
        HashMap<Integer, HashSet<Integer>> obs = new HashMap<>();
        for (int[] o : obstacles)
        {
            HashSet<Integer> set = obs.getOrDefault(o[0], new HashSet<>());
            set.add(o[1]);
            obs.put(o[0], set);
        }
        
int[][] dirVector = new int[][]
        {
            new int[] {0, 1},
            new int[] {1, 0},
            new int[] {0, -1},
            new int[] {-1, 0},
        };
        int dir = 0;
        int ans = 0;
        int x = 0, y = 0;
        for (int c : commands)
        {
            dir += 4;
            if (c == -2) dir--;
            else if (c == -1) dir++;
            dir %= 4;
            while (c > 0)
            {
                int nx = x + dirVector[dir][0];
                int ny = y + dirVector[dir][1];
                if (obs.containsKey(nx) && obs.get(nx).contains(ny)) break;
                else
                {
                    x = nx;
                    y = ny;
                    int curDist = x * x + y * y;
                    if (ans < curDist) ans = curDist;
                }
                c--;
            }
        }
        return ans;
    }
}
