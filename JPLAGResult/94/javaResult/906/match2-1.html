class Solution {
    
public int robotSim(int[] commands, int[][] obstacles) {
        Map<Integer, Map<Integer, Integer>> map = new HashMap<>();
        for (int[] obj : obstacles) {
            if (!map.containsKey(obj[0])) {
                map.put(obj[0], new HashMap<>());
            }
            Map<Integer, Integer> tmp = map.get(obj[0]);
            tmp.put(obj[1], tmp.getOrDefault(obj[1], 0) + 1);
            map.put(obj[0], tmp);
        }
        
        int[] cur = new int[]{0, 0};
        
int[][] dirs = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int i = 0;
        int ans = 0;
        for (int cmd : commands) {
            if (cmd == -2) {
                i = (i - 1 + 4) % 4;
            }
            else if (cmd == -1) {
                i = (i + 1) % 4;
            }
            else {
                for (int j = 1; j <= cmd; ++j) {
                    int nx = cur[0] + dirs[i][0];
                    int ny = cur[1] + dirs[i][1];
                    if (map.containsKey(nx) && 
map.get(nx).containsKey(ny)) {
                        break;
                    }
                    cur[0] = nx;
                    cur[1] = ny;
                    ans = Math.max(ans, cur[0] * cur[0] + cur[1] * cur[1]);
                }
            }
        }
        return ans;
    }
}
