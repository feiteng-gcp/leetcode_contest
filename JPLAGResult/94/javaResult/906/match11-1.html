class Solution {
    public int robotSim(int[] cs, int[][] obs) {
        Set<String> s = new HashSet<>();
        for(int[] b:obs){
            
            s.add(""+b[0]+","+b[1]);
        }
        
int r = 0;
        
        
        // 0 ,1 ,2,3   
        
        int x = 0;int y = 0;int st = 0;
    
        for(int c:cs){
            if(c==-2){
                st = (4+st-1)%4;
            }else if(c==-1){
                st = (4+st+1)%4;
            }else{
                boolean ok =false;
                if(st==0){
                    int ny = y + c;
                    for(int u=y;u<=ny;u++){
                        
                        

                        
                        
                        if(s.contains(""+x+","+u)){
                            y = u-1;
                            ok = true;
                            break;
                        }
                    }
                    
if(!ok){
                    y = ny;
                    }
                    
                    
                }else if(st==1){
                    int nx = x+c;
                    
                    for(int u=x;u<=nx;u++){
                         
                        
                        
                        
                        
                        if(s.contains(""+u+","+y)){
                            x = u-1;
                            ok = true;
                            break;
                        }
                    }
                    if(!ok){
                    x = nx;
                    }
                    
                    
                    
                }else if(st==2){
                    int ny = y-c;
                    for(int u=y;u>=ny;u--){
                                       
                        
                        
                        
                        
                        
                        if(s.contains(""+x+","+u)){
                            y = u+1;
                            ok = true;
                            break;
                        }
                    }
                    
if(!ok){
                    y = ny;
                    }
                    
                    
                    
                }else{
                    int nx = x-c;
                    
                    
                    for(int u=x;u>=nx;u--){
                        
                        
                        
                        
                        if(s.contains(""+u+","+y)){
                            x = u+1;
                            ok = true;
                            break;
                        }
                    }
                    if(!ok){
                    x = nx;
                    }
                    
                
}
                
                
                
            }
            
            r = Math.max(r,x*x+y*y);
            
        }
        
        return r;
        
        
        
        
    }
}
