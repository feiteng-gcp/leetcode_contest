import java.util.*;


class Solution {
    public int robotSim(int[] commands, int[][] obstacles) {
        Set<List<Integer>> obstacleSet = new HashSet<>();
        for(int[] obstacle : obstacles) {
            obstacleSet.add(Arrays.asList(obstacle[0], obstacle[1]));
        }
        int result = 0;
        int dir = 0;
        List<int[]> incTable = Arrays.asList(
                
new int[]{0, 1},
                new int[]{1, 0},
                new int[]{0, -1},
                new int[]{-1, 0}
        );
        int[] pos = new int[]{0, 0};
        
for(int command : commands) {
            if(command == -1) {
                dir = (dir + 1) % 4;
            }else if(command == -2) {
                dir = (dir - 1 + 4) % 4;
            }else {
                int[] inc = incTable.get(dir);
                for(int i = 0; i < command; i++) {
                    List<Integer> nextPos = Arrays.asList(pos[0] + inc[0], pos[1] + inc[1]);
                    if(obstacleSet.contains(nextPos)) {
                        break;
                    }else {
                        pos[0] = nextPos.get(0);
                        pos[1] = nextPos.get(1);
                        result = Math.max(result, pos[0] * pos[0] + pos[1] * pos[1]);
                    }
                
}
            }
        }
        
        return result;
    }
}
