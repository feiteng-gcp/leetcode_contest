class Solution {
    public int robotSim(int[] commands, int[][] obstacles) {
        int ret = 0;
        int[] cur = {0,0,0};
        int[] dirs = {0,1,2,3};
        Set<String> obs = new HashSet<>();
        for (int[] each:obstacles) {
            obs.add(Integer.toString(each[0])+","+Integer.toString(each[1]));
        }
        for (int cmd:commands) {
            if (cmd==-1) {
                cur[2] += 1;
                cur[2] %= 4;
                continue;
            }
            if (cmd==-2) {
                cur[2] -= 1;
                if (cur[2]<0) {
                    cur[2] += 4;
                }
                continue;
            }
            
            if (cur[2]==0) {
                while (cmd>0) {
                    cur[1]++;
                    String pos = Integer.toString(cur[0])+","+Integer.toString(cur[1]);
                    if (obs.contains(pos)) {
                        cur[1]--;
                        break;
                    }
                    cmd--;
                }
            } else if (cur[2]==1) {
                while (cmd>0) {
                    cur[0]++;
                    String pos = Integer.toString(cur[0])+","+Integer.toString(cur[1]);
                    if (obs.contains(pos)) {
                        cur[0]--;
                        break;
                    }
                    cmd--;
                }
            } else if (cur[2]==2) {
                while (cmd>0) {
                    cur[1]--;
                    String pos = Integer.toString(cur[0])+","+Integer.toString(cur[1]);
                    if (obs.contains(pos)) {
                        cur[1]++;
                        break;
                    }
                    cmd--;
                }
            } else {
                while (cmd>0) {
                    cur[0]--;
                    String pos = Integer.toString(cur[0])+","+Integer.toString(cur[1]);
                    if (obs.contains(pos)) {
                        cur[0]++;
                        break;
                    }
                    cmd--;
                }
            }
            ret = Math.max(ret, cur[0]*cur[0]+cur[1]*cur[1]);
        }
        return ret;
    }
}
