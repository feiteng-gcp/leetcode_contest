class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_set<int> s(A.begin(), A.end());

        int res = 0;
        for (int i = 0; i < A.size()-2; i++)
        {
            for (int j = i+1; j < A.size() - 1; j++)
            {
                int cur = 2, a = A[i], b = A[j] , c = a + b;

                while (s.find(c) != s.end())
                {
                    cur++;
                    a = b;
                    b = c;
                    c = a + b;
                }
                res = max(res, cur);
            }
        }
        return res > 2 ? res : 0;            
    }
};
