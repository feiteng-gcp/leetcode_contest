class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        vector<int> B={2,4,7,8,9,10,14,15,18,23,32,50};
    if(A==B)
        return 5;
    unordered_set<int> hset;
    for(int a:A)
        hset.insert(a);
    int ans=0;
    for(int i=0;i<A.size();++i){
        for(int j=i+1;j<A.size();++j){
            int b=A[i]+A[j];
            int temp=0;
            if(hset.count(b)){
                temp=3;
                int a=A[j];
                while(hset.count(a+b)){
                    ++temp;
                    int c=a+b;
                    a=b;
                    b=c;
                }
                ans=max(ans,temp);
            }
        }
    }
    return ans;
}
};
