class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_set<int> X;
        for (int x : A) X.insert(x);
        int res = 0;
        for (int a = 0; a < A.size(); ++a) {
            for (int b = a+1; b < A.size(); ++b) {
                int r = 2;
                int c = A[a]+A[b];
                int aa = A[a];
                int bb = A[b];
                while(X.find(c)!=X.end()) {
                    ++r;
                    aa = bb;
                    bb = c;
                    c = aa+bb;
                }
                //printf("[%d, %d] : %d\n", A[a],A[b], r);
                res = max(r,res);
            }
        }
        if (res < 3) res = 0;
        return res;
    }
};
