class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_set<int> s;
        int res = 0;
        for(auto i:A){
            s.insert(i);
        }

        for(int i=0;i<A.size()-2;i++){
            for(int j=i+1;j<A.size()-1;j++){
                int first = A[i];
                int second = A[j];
                int count = 0;
                while(s.count(first+second)){
                    ++count;
                    int temp = first + second;
                    first = second;
                    second = temp;
                }
                //cout<<i<<" "<<j<<" "<<count<<endl;
                res = max(res,count);
            }
        }
        return res == 0? 0:res+2;
    }
