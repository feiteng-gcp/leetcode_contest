class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        set<int> st;
        for(auto u : A)st.insert(u);
        int ans = 0;
        int n = A.size();
        for(int i = 0; i < n; i++){
            for(int j = i+1; j < n; j++){
                int x = A[i], y = A[j], now = 2;
                if(!st.count(x+y))continue;
                while(st.count(x+y)){
                    int tmp = x+y;
                    x = y, y = tmp;
                    now++;
                    //printf("%d\n", A[i]+A[j]);
                }
                ans = max(ans, now);
            }
        }
        return ans;
    }
};
