class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        #define pii pair<int,pair<int,int>>
        priority_queue<pii,vector<pii>,greater<pii> > pq;
        int ans=0;
        #define x first
        #define y second
        for(int i=0;i<A.size();i++){
            while(!pq.empty()&&pq.top().x<A[i]) pq.pop();
            while(!pq.empty()&&pq.top().x==A[i]){
                pii it = pq.top();
                pq.pop();
                pq.push(make_pair(it.y.x+A[i],make_pair(A[i],it.y.y+1)));
                ans=max(ans,it.y.y+1);
            }
            for(int j=0;j<i;j++){
                pq.push(make_pair(A[i]+A[j],make_pair(A[i],2)));
            }
        }
        return ans;
    }
};
