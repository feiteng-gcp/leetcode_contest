class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        set<int> mis;
        mis.insert(A.begin(),A.end());

        int ml=0;
        int res=0;
        for(int i=1; i < A.size(); i++){
            for(int j=0; j <i; j++){
                int e=A[i];
                int a=A[j];
                int ec=0;
                if(mis.find(e+a)!=mis.end()){
                    ec=2;
                }
                while(mis.find(e+a)!=mis.end()){
                    ec++;
                    int au=e+a;
                    a=e;
                    e=au;
                }
                ml=max(ec,ml);
            }
        }
        return ml;
    }
};
