class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        int n = A.size();
#define ll long long
        unordered_map<ll, int> mp;
        vector<vector<int> > vis(n, vector<int>(n));
        for (int i = 0; i < n; i++) {
            mp[A[i]] = i;
        }
        int res = 2;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (vis[i][j]) {
                    continue;
                }
                int i1 = i, j1 = j, now = 2;
                while (true) {
                    vis[i1][j1] = true;
                    ll val = (ll)A[i1] + A[j1];
                    if (mp.find(val) != mp.end()) {
                        i1 = j1;
                        j1 = mp.at(val);
                        now++;
                        // cout << i1 << ", " << j1 << ", " << now << endl;
                    } else {
                        res = max(res, now);
                        break;
                    }
                }
            }
        }
        return res == 2 ? 0 : res;
    }
};
