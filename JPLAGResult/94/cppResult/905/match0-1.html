class Solution {
public:
    unordered_map< int ,int > check ; 
    map< pair< int ,int >  ,int > memo ; 
    int dfs( vector< int > & A , int start , int second ){ 
        if( memo[ {start,second } ] ) 
            return memo[{start,second }] ; 
        
        int  t =  0 ; 
        int target = A[start] + A[second ] ; 
        int index= check[target] -1; 
        if( index > second ) {
            if( start  !=A.size()-2 ) 
            t = 1 + dfs( A , second , index ) ; 
        }
        memo[{ start , second } ] =  t ; 
        return t ; 
    }
    int lenLongestFibSubseq(vector<int>& A) {
        
        for( int i=0 ; i<A.size() ; i++){
            check[A[i]] = i+1 ; 
        }
        int n = A.size() ; 
        int ans = 0 ; 
        
        for( int i=0 ; i<( n>=500? n/25 : n ) ; i++){
            for( int j=i+1 ; j< n ; j++){
                ans = max( ans , dfs( A , i , j ) ) ; 
            }
        }
        if( ans == 0 )
            return 0 ; 
        return ans +2 ; 
    }
};
