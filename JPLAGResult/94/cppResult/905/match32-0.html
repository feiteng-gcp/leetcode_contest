class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_map<int,int> mapA;
        for(int i=0;i<A.size();++i) mapA[A[i]] = i;
        //unordered_set<int> records;
        int max_length = 0;
        for(int i=0;i<A.size();++i){
            for(int j=i+1;j<A.size();++j){
                int x = i;
                int y = j;
                int length = 2;
                while(y < A.size()){
                    //int code = x * 1000 + y;
                    //if(records.count(code)==1) break;
                    //else records.insert(code);
                    int w = A[x] + A[y];
                    if(mapA.count(w) == 0) break;
                    else{
                        x = y;
                        y = mapA[w];
                        length++;
                    }
                }
                if(length>2) max_length = max(max_length, length);
            }
        }
        return max_length;
    }
};
