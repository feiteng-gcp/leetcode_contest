	LL a, b;

bool operator < (const Node & a, const Node &b) {
	if (a.a < b.a) return true;
	if (a.a > b.a) return false;
	return a.b < b.b;
}

class Solution {
public:
	int lenLongestFibSubseq(vector<int>& A) {
		bool flag[1010][1010];
		map<int, int> index;
		memset(flag, false, sizeof(flag));
		int n = A.size();
		for (int i = 0; i < n; ++i) {
			index[A[i]] = i;
		}

		int res = 0;
		for (int i = 0; i < n; ++i) {
			for (int j = i + 1; j < n; j++) {
				if (flag[i][j]) break;

				int a= i;
				int b = j;
				int acc = 2;
				while (true) {
					int x = A[a] + A[b];
					if (index.count(x) == 0) break;
					a = b;
					b = index[x];
					++acc;
				}
				if (acc > 2) {
					res = max(res, acc);
				}
			}
		}
		return res;
	}
};
