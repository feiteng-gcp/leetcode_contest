class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_set<int> hash;
        
        for(int n: A)
            hash.insert(n);

        int maxLen = 0;
        
        for(int i = 0; i < A.size(); i ++)
        {
            for(int j = i + 1; j < A.size(); j ++)
            {
                int curLen = 2;
                
                int one = A[i], two = A[j];
                
                while(hash.find(one+two) != hash.end())
                {
                    int three = one + two;
                    one = two;
                    two = three;
                    curLen++;
                }
                if(curLen == 2) curLen = 0;
                maxLen = max(maxLen, curLen);
            }
        }
        
        return maxLen;
    }
};
