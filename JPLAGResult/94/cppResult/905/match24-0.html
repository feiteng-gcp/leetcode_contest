class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        set<int> s;
        for (int i = 0; i < A.size(); ++i) s.insert(A[i]);
        
        int ans = 2;
        for (int i = 0; i < A.size() - 1; ++i)
            for (int j = i + 1; j < A.size(); ++j)
            {
                int a1 = A[i], a2 = A[j],len = 2;
                while (s.count(a1 + a2) == 1)
                {
                    a2 += a1;
                    a1 = a2 - a1;
                    ++len;
                }
                ans = max(ans, len);
            }
        if (ans <= 2) ans = 0;
        return ans;
    }
};
