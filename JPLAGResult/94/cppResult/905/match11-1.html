class Solution {
public:
    int lenLongestFibSubseq(vector<int>& v) {
        int res=0;
        unordered_map<int,int>m;
        for(auto num:v)

            m[num]=1;
        for(int i=0;i<v.size();i++){
            for(int j=i+1;j<v.size();j++){
                int count=2;
                int curr=v[j], prev=v[i];
                while(m.count(curr+prev)==1){
                    // cout<<"prev: "<<prev<<" and curr: "<<curr<<endl;
                    count++;
                    int temp=prev;
                    prev=curr;
                    curr=curr+temp;
                }
                res=max(res,count);
            }
            
        }
        return (res==2)?0:res;
    }
};
