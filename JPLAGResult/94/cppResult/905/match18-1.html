class Solution {
public:
int lenLongestFibSubseq(vector<int>& A) {

	int l = A.size();
	int lge = 0;
	for (int i = 0; i < l - 2; ++i) {
		for (int j = i + 1; j < l - 1; ++j) {
			int t1 = A[i], t2 = A[j];
			int count = 2;
			for (int k = j + 1; k < l; ++k) {
                if (A[k] > t1 + t2)
					break;

				if (t1 + t2 == A[k]) {
					t1 = t2;
					t2 = A[k];
					++count;
				}   
			}
			if (count > 2)
				lge = max(lge, count);
		}
	}
	return lge;
}
};
