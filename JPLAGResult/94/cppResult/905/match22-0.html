class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        int ans = 0;
        for (int i = 0; i < A.size(); i++) {
            for (int j = i + 1; j < A.size(); j++) {
                int cur = 2;
                int a = A[i], b = A[j];
                while (std::binary_search(A.begin(), A.end(), a + b)) {
                    int tmp = a;
                    a = b;
                    b = tmp + b;
                    cur += 1;
                }
                ans = max(ans, cur);
            }
        }
        return ans < 3 ? 0 : ans;
    }
};
