class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_set<int> s(A.begin(), A.end());
        int res = 0;
        
        for (int i = 0; i < A.size(); ++i) {
            for (int j = i + 1; j < A.size(); ++j) {
                int a = A[i], b = A[j];
                int len = 1;
                while (s.count(b)) {
                    b = a + b;
                    a = b - a;
                    ++len;
                }
                if (len > 2)
                    res = max(res, len);
            }
        }
        
        return res;
    }
};
