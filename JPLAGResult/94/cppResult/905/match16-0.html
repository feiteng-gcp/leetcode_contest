class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_map<int, int> m;
        for (int i = 0; i < A.size(); ++i) {
            m[A[i]] = i;
        }
        vector<int> count(A.size(), 0);
        int theMax = 0;
        for (int i = 1; i < A.size() - 1; ++i) {
            for (int j = 0; j < i; ++j) {
                int a1 = A[j];
                int a2 = A[i];
                int cur = 2;
                while (m.count(a1+a2)) {
                    ++cur;
                    int a3 = a1 + a2;
                    a1 = a2;
                    a2 = a3;
                    if (a3 > A.back()) break;
                }
                theMax = max(theMax, cur);
            }
        }
        return theMax <= 2 ? 0 : theMax;
    }
};
