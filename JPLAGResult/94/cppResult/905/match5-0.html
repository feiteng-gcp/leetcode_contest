class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) {
        unordered_set<int> nums;
        
        for ( auto a: A )
            nums.insert(a);
        int len = A.size();
        int maxLen = 0;
        for ( int i = 0; i < len; ++i )
        {
            for ( int j = i+1; j < len; ++j )
            {
                int n1 = A[i];
                int n2 = A[j];
                int n3 = n1+n2;
                int sublen = 2;
                while ( nums.find(n3) != nums.end() )
                {
                    ++sublen;
                    n1=n2;
                    n2 = n3;
                    n3 = n1+n2;
                }
                maxLen = max(maxLen, sublen);
            }
        }
        
        return maxLen < 3 ? 0 : maxLen;
    }
};
