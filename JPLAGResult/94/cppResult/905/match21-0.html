template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
    int lenLongestFibSubseq(vector<int>& a) {
        int N=a.size();
        unordered_set<int> s;
        for(int k:a) s.insert(k);
        int res=0;
        for(int i=0;i<N;i++){
            for(int j=i+1;j<N;j++){
                int tmp=2;
                int A=a[i],B=a[j];
                while(s.count(A+B)){
                    tmp++;
                    int C=A+B;
                    A=B,B=C;
                }
                if(tmp>2) res=max(res,tmp);
            }
        }
        return res;
        
    }
};
