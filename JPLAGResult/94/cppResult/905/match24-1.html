class Solution {
public:
    unordered_map<int,int> p;
    int lenLongestFibSubseq(vector<int>& A) {

        for (int i=0;i<A.size();i++) p[A[i]]=i;
        int ans=0;
        for (int i=0;i<A.size();i++)
            for (int j=i+1;j<A.size();j++){
                int a1=i,a2=j,a3,len=2;
                while (true){
                    a3=A[a1]+A[a2];
                    if (p.find(a3)==p.end()) break;
                    a3=p[a3];
                    len++;
                    a1=a2;
                    a2=a3;
                }
                ans=max(ans,len);
            }
        if (ans<3) ans=0;
        return ans;
    }
};
