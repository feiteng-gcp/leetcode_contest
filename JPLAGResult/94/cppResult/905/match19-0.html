class Solution {
public:
    int lenLongestFibSubseq(vector<int>& A) 
    {
        int n = A.size();
        int ans = 0;
        int dp[1000][1000];
        map<int, int> mp;
        for(int i = 0; i < n; i++ ) 
            for(int j = i + 1; j < n; j++ )
                dp[i][j] = 1;
        for( int i = n - 1; i >= 0; i-- )
        {
            for(int j = 0; j <= i-1; j++ )
            {
            map<int,int>::iterator it=mp.find(A[i]+A[j]);
            if( it != mp.end() ){
                dp[j][i] = dp[i][it->second] + 1;
                if( dp[j][i] > ans )
                    ans = dp[j][i];
            }
            
        }
        mp[A[i]]=i;
    }
    if(ans==0)
    {
        return 0;
    }
    else
        return ans + 1;
    }
};
