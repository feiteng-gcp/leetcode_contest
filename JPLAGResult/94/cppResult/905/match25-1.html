class Solution {
public:
int lenLongestFibSubseq(vector<int>& A) {
	int len = A.size();
	vector<unordered_map<int, int>> record;

	vector<int> dp(len, 1);
	unordered_map<int, int> val2idx;
	for (int i = 0; i < len; i++) {
		val2idx[A[i]] = i;
	}
	record.resize(len);
	int ans = 0;
	for (int i = 2; i < len; i++) {
		for (int j = i - 1; j >= 0 && A[i] < A[j] * 2; j--) {
			if (val2idx.find(A[i] - A[j]) != val2idx.end()) {
				record[i][A[j]] = max(record[i][A[j]], 3);
				ans = max(ans, record[i][A[j]]);
			}
			if (record[j].find(A[i] - A[j]) != record[j].end()) {
			//if (val2idx.find(A[i] - A[j]) != val2idx.end() && val2idx[A[i] - A[j]] < j) {
				//dp[i] = max(dp[i], dp[val2idx[A[i] - A[j]]] + 2);
				record[i][A[j]] = max(record[i][A[j]], record[j][A[i] - A[j]] + 1);
				ans = max(ans, record[i][A[j]]);
			}
		}
	}
	return ans;

}
};
