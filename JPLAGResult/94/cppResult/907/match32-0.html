class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int lo = 1, mi, hi = 1000000000;
        while (lo < hi) {
            mi = lo + (hi - lo) / 2;
            if (check(piles, H, mi)) {
                hi = mi;
            } else {
                lo = mi + 1;
            }
        }
        return hi;
    }
    int check(vector<int>& piles, int H, int K) {
        int res = 0;
        for (auto x: piles) {
            res += (x / K) + (x % K > 0) ;
            
        }
        return res <= H;
    }
};
