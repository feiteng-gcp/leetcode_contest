class Solution {
public:

    bool isPossible(vector<int> & piles,long long allowed,int H){
        int size=piles.size();
        int cost=0;
        for(int i=0;i<size;i++){
            cost+=piles[i]/allowed;
            if(piles[i]%allowed!=0){
                cost+=1;
            }
        }
        if(cost<=H){
            return true;
        }else{
            return false;
        }
    }
    
    int minEatingSpeed(vector<int>& piles, int H) {
        int size=piles.size();
        sort(piles.begin(),piles.end());
        long long start=1,end=piles[size-1],mid,answer=piles[size-1];
        
        while(start<=end){
            mid=(start+end)/2;
            bool possible=isPossible(piles,mid,H);
            // cout<<start<<" "<<end<<" "<<mid<<endl;

            if(possible){
                
                answer=mid;
                end=mid-1;
            }else{
                start=mid+1;
            }
        }
        
        return answer;
    }
};
