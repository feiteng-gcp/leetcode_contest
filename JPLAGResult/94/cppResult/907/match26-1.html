class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int maxVal = 0;
        
        for(int pile: piles)
        {
            maxVal = max(pile, maxVal);
        }
        
        int st = 1, end = maxVal;
        
        while(st < end)
        {

            int mid = (st + end)/2;
            
            if(CanFinish(piles, H, mid))
            {
                end = mid;
            }
            else
            {
                st = mid+1;
            }
            
        }
        
        return st;
    }
    
    // given K, can koko finish the piles in H hours.
    bool CanFinish(vector<int>& piles, int H, int K)
    {
        int hour = 0;
        for(int pile: piles)
        {
            int tempHour=0;
            
            if(pile%K)
                tempHour = 1;
            
            tempHour += (pile/K);
            hour += tempHour;
            
            if(hour > H) return false;
        }
        
        return true;
    }
};
