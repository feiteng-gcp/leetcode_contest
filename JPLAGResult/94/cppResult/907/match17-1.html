class Solution {
 public:
  int minEatingSpeed(vector<int>& piles, int H) {
    int maxx = 0;
    for (int i = 0; i < piles.size(); ++i) {
      if (piles[i] > maxx)
        maxx = piles[i];
    }
    int s = 1;
    int e = maxx;

    int res = INT_MAX;
    while (s <= e) {
      int m = (s + e) / 2;
      int cnt = 0;
      for (int i = 0; i < piles.size(); ++i) {
        if (piles[i] % m) {
          cnt += piles[i] / m + 1;
        } else {
          cnt += piles[i] / m;
        }
      }
        //cout << cnt << " " << m << endl;
      if (cnt > H) {
        s = m + 1;
      } else if (cnt <= H) {
        e = m - 1;
        if (res > m)
          res = m;
      }
    }
    if (res == INT_MAX)
      return e;
    return res;
  }
};
