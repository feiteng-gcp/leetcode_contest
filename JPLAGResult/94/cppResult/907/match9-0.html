class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        sort(piles.begin(), piles.end());
        int l = 1, r = piles.back();
        while (l < r)
        {
            int mid = (l + r) / 2;
            int cur = 0;
            for (auto pile: piles)
            {
                cur += pile / mid;
                cur += (pile % mid == 0) ? 0 : 1;
            }
            if (cur <= H)
            {
                r = mid;
            }
            else
            {
                l = mid+1;
            }
        }
        return l;
    }
};
