class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int left = 1, right = 0;
        for (int i = 0; i < piles.size(); ++i) {
            right = max(right, piles[i]);
        }
        while (left < right) {
            int mid = left + (right - left) / 2;
            int times = 0;
            for (int i = 0; i < piles.size(); ++i) {
                times += piles[i] / mid;
                times += (piles[i] % mid == 0) ? 0 : 1;
            }
            if (times > H) {
                left = mid + 1;
            }
            else {
                right = mid;
            }
        }
        return left;
    }
};
