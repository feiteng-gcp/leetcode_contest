class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int n = piles.size();
        sort(piles.begin(), piles.end());
        
        int l = 1, r = piles[n -1];
        while (l <= r) {
            int m = (l + r) /2;
            
            if (verify(piles, m) <= H)  r = m -1;
            else                        l = m +1;
        }
        
        return r +1;
    }
private:
    int verify(vector<int>& p, int speed) {
        int n = p.size();
        int total_h = 0;
        for (int i = 0; i < n; ++i)
            total_h += (p[i] + speed -1) / speed;
        return total_h;
    }
};
