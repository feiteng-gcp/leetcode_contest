class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) 
    {
        
        int left = 1;
        int right = 0;
        for (int i=0; i<piles.size(); i++)
            right = max(right,piles[i]);
        
        while (left<right)
        {
            int mid = (right-left)/2+left;
            if (eat(mid,piles)>H)
                left = mid+1;
            else
                right = mid;
        }
        return left;
    }
    
    int eat(int K, vector<int>& piles)
    {
        int count = 0;
        for (int i=0; i<piles.size(); i++)
        {
            if (piles[i]%K==0)
                count+= piles[i]/K;
            else
                count+= piles[i]/K+1;
        }
        return count;
            
    }
};
