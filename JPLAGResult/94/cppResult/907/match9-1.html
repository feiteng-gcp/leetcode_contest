class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int l = 1, r = 1e9;
        while (l <= r){
            int m = (l + r) / 2;
            int cur = 0;
            for (int p : piles)
                cur += p / m + ((p % m) > 0);
            if (cur > H)
                l = m + 1;
            else
                r = m - 1;
        }
        return l;
        
    }
};
