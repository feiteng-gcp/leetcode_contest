class Solution {
public:
    int timeE(vector<int>& piles, int speed) {
	int c = 0;
	for (int i = 0; i < piles.size(); ++i) {
		c += piles[i] / speed;
		c += (piles[i] % speed) ? 1 : 0;
	}
	return c;
}

int minEatingSpeed(vector<int>& piles, int H) {
	sort(piles.begin(), piles.end());
	int left = 1, right = piles.back();
	while (left <= right - 1) {
		int mid = (left + right) / 2 ;
        cout << mid << endl;
		int t = timeE(piles, mid);
        cout << t << endl;
		if (t > H)
			left = mid + 1;
		else
			right = mid;
	}
	return right;
}
};
