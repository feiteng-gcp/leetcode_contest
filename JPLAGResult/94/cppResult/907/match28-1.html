class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int left = 1, right = 0;
        for(auto& i:piles){
            right= max(right,i);
        }

        while(left<right){
            int mid = left + (right-left) / 2;
            if(valid(piles,H,mid))
                right = mid;
            else
                left = mid+1;
        }
        return left;
    }
    
    bool valid(vector<int>& piles, int H, int speed){
        int count = 0;
        for(auto i:piles){
            if((i%speed)==0){
                count += i/speed;
            }
            else{
                count += i/speed + 1;
            }
        }
        return count<=H;
    }
};
