class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int l = 1, r = *max_element(piles.begin(), piles.end());
        while (l < r)
        {
            int mid = (l + r) >> 1;
            int t = 0;
            for (int i = 0; i < piles.size(); ++i)
                t += piles[i] / mid + bool(piles[i] % mid);
            //cout << l << ' ' << r << ' ' << t << endl;
            if (t <= H) r = mid;
            else l = mid + 1;
        }
        return l;
    }
};
