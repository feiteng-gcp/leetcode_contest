class Solution {
public:
    typedef long long LL;
    int minEatingSpeed(vector<int>& piles, int H) {
      LL low = 1, hight = 10e9, ret = -1LL;
      while (low <= hight) {
        LL mid = (low + hight) / 2LL;
        LL h = 0LL;
        for (auto& x : piles) {

          h += (1LL * x + mid - 1) / mid;
        }
        if (h <= H) {
          ret = mid;
          hight = mid - 1LL;
        } else {
          low = mid + 1LL;
        }
      }
      return ret;
    }
};
