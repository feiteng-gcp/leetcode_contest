class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        if(piles.size()>H)
        {
            return false;
        }
        int maxSpeed=INT_MIN,minSpeed=1;

        int result=maxSpeed;
        for(int i=0;i<piles.size();++i)
        {
            maxSpeed=max(maxSpeed,piles[i]);
        }
        while(minSpeed<=maxSpeed)
        {
            int midSpeed=minSpeed+(maxSpeed-minSpeed)/2;
            int time=0;
            for(int i=0;i<piles.size();++i)
            {
                int tmp=piles[i]/midSpeed;
                time+=piles[i]%midSpeed==0?tmp:tmp+1;
                if(time>H)
                {
                    break;
                }
            }
            if(time>H)
            {
                minSpeed=midSpeed+1;
            }
            else
            {
                result=midSpeed;
                maxSpeed=midSpeed-1;
            }
        }
        return result;
    }
};
