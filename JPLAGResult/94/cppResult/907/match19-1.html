class Solution {
public:
    int minEatingSpeed(vector<int>& v, int maxTime) {
        sort(v.begin(), v.end());
        if(v.size()==maxTime) return v.back();
        int start=1,end=v.back(),mid;
        while(start<end){
            // cout<<"start: "<<start<<" and end: "<<end<<endl;

            mid=(start+end)/2;
            int time=0;
            for(int i=0;i<v.size();i++){
                time+=((v[i]-1)/mid)+1;
            }
            // cout<<"mid: "<<mid<<" and time: "<<time<<endl;
            if(time>maxTime) start=mid+1;
            else end=mid;
        }
        cout<<start<<endl;
        return start;
    }
};
