typedef long long ll;

template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
    int minEatingSpeed(vector<int>& p, int H) {
        int l=1,r=1e9;
        while(l<r){
            int m=l+(r-l)/2;
            ll t=0;
            for(int k:p){
                t+=(k+m-1)/m;
            }
            if(t<=H){
                r=m;
            }else{
                l=m+1;
            }
        }
        return l;
    }
};
