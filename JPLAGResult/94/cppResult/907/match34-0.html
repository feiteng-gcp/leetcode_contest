int check(int x,vector<int> &A,int h) {
    int t=0;
    for(int i=0;i<A.size();++i) {
        t+=A[i]/x;
        if(A[i]%x!=0) t++;
    }
    if(t<=h) return 1;
    return 0;
}

class Solution {
public:
    int minEatingSpeed(vector<int>& A, int H) {
        sort(A.begin(),A.end());  int n=A.size();
        int l=1,r=A[n-1],mid,ans;
        while(l<=r) {
            mid=l+(r-l)/2;
            if(check(mid,A,H)) {
                ans=mid;  r=mid-1;
            }
            else l=mid+1;
        }
        return ans;
    }
};
