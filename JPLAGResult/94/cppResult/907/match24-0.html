class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int mini = 1;
        int maxi = 1e9;
        int res = 1e9;
        while (mini <= maxi) {
            int mid = (mini + maxi) / 2;
            int cur = 0;
            for (int i = 0; i < piles.size(); i++) {
                cur += piles[i] / mid;
                if (piles[i] % mid) cur++;
            }
            if (cur <= H) {
                res = min(res, mid);
                maxi = mid - 1;
            } else {
                mini = mid + 1;
            }
        }
        return res;
    }
};
