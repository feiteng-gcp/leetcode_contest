class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int low = 1;
        int high = 1e9;
        while(low < high){
            int mid = (low) + (high - low)/2;
            int hours = calcualteHours(piles, mid);
            if(hours > H){
                low = mid + 1;
            } else{
                high = mid;
            }
        }
        return low;
    }
    
    int calcualteHours(vector<int>& piles, int K){
        int hours = 0;
        for(int pile:piles){
            if(pile % K == 0) hours += pile/K;
            else hours += (pile/K) +1;
        }
        return hours;
    }
};
