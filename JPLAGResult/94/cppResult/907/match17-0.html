class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int max = 0;
        for(int i = 0; i < piles.size(); i++){
            if(piles[i] > max)
                max = piles[i];
        }
        
        int lo, hi;
        lo = 0; hi = max; // lo can't work, hi can work;
        while(lo < hi-1){
            int mid = lo + (hi-lo)/2;
        
            int sum = 0;
            for(int i = 0; i < piles.size(); i++){
                if(piles[i]%mid == 0)
                    sum += piles[i]/mid;
                else 
                    sum += piles[i]/mid+1;
            }
            
            if(sum > H)
                lo = mid;
            else 
                hi = mid;
        }
        
        return hi;
    }
};
