class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int ave=H/piles.size();

        int up=*max_element(piles.begin(),piles.end())/ave+1,low=1;
        while(low<up){
            int mid=(low+up)/2;
            int times=0;
            for(int x:piles)times+=x/mid+(x%mid!=0);
            if(times<=H)up=mid;
            else low=mid+1;
        }
        return low;
    }
};
