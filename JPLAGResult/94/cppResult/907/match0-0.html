class Solution {
public:
    bool islegal( vector< int > & piles, int k , int h){
        int t = 0 ; 
        for( int i=0 ; i<piles.size() ; i++){
            if( piles[i] < k  ){
                t++;
                continue ; 
            }
            
            t+=(piles[i]/k) ;
            if( k != 1 && piles[i]%k != 0 ){
                t++; 
            }
        }
        //cout << t <<'-'<<k << endl;
        return t <= h ; 
    }
    int minEatingSpeed(vector<int>& piles, int H) {
        int mi = 0 ; 
        for( int i=0 ; i<piles.size() ; i++){
            mi = max( mi, piles[i] ) ; 
        }
        int left=    1  ; 
        int right =  mi*2   ; 
        
        
        while( left <= right ){
            
            int mid = ( right - left ) / 2 + left ; 
            if( islegal( piles , mid , H ) ) {
                right = mid -1 ; 
            }
            else{
                left = mid + 1;  
            }
            
        }
        return left ; 
    }
};
