class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int H) {
        int l=1,r=1<<30,mid;
        while (true){
            if (l>=r) return l;

            mid=(l+r)/2;
            int tot=0;
            for (int i=0;i<piles.size();i++)
                tot+=(piles[i]-1)/mid+1;
            if (tot<=H) r=mid; else l=mid+1;
        }
    }
};
