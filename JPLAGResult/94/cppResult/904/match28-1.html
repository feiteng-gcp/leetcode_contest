/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    
    void calc(vector<int>& leaf, TreeNode* root) {
        if(root==NULL)
            return;
        calc(leaf, root->left);
        if(root->left==NULL && root->right==NULL) {
            leaf.push_back(root->val);
        }
        calc(leaf, root->right);
    } 
    
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector<int> leaf1;
        vector<int>leaf2;
        calc(leaf1,root1);
        calc(leaf2,root2);
        if(leaf1.size()!=leaf2.size())
            return false;
        for(int i=0;i<leaf1.size();i++) {
            if(leaf1[i]!=leaf2[i]) {
                return false;
            }
        }
        return true;
    }
};
