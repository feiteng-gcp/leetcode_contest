/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    void dfs(vector<int>& res, TreeNode* root) {
        if (!root) {
            return;
        }
        
        if (!root->left && !root->right) {
            res.push_back(root->val);
            return;
        }
        
        if (root->left) {
            dfs(res, root->left);
        }
        
        if (root->right) {
            dfs(res, root->right);
        }
    }
    
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector<int> l1, l2;
        dfs(l1, root1);
        dfs(l2, root2);
        
        if (l1.size() != l2.size()) 
            return false;
        
        for (int i = 0;i < l1.size(); ++i) {
            if (l1[i] != l2[i])
                return false;
        }
        
        return true;
    }
};
