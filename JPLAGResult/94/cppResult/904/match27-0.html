/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector<int> s1;
        vector<int> s2;
        traverse(root1, s1);
        traverse(root2, s2);
        return s1 == s2;
    }
    void traverse(TreeNode* n, vector<int> & s) {
        if (n == NULL) return;
        if (n->left == NULL && n->right == NULL) {
            s.push_back(n->val);
        }
        traverse(n->left, s);
        traverse(n->right, s);
    }
};
