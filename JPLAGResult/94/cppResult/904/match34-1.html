/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
 public:
  bool leafSimilar(TreeNode* root1, TreeNode* root2) {
    vector<int> l1, l2;
    Dfs(root1, l1);
    Dfs(root2, l2);
    return l1 == l2;
  }

 private:

  void Dfs(TreeNode* root, vector<int>& v) {
    if (!root) {
      return;
    }
    if (!root->left && !root->right) {
      v.push_back(root->val);
      return;
    }
    if (root->left) {
      Dfs(root->left, v);
    }
    if (root->right) {
      Dfs(root->right, v);
    }
  }
};
