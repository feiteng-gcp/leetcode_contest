/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector<int> s1, s2;
        getLeaves(root1, s1);
        getLeaves(root2, s2);
        return s1 == s2;
    }
    void getLeaves(TreeNode* root, vector<int>& seq) {
        if (!root) {
            return;
        }
        TreeNode* left = root->left, *right = root->right;
        if (!left && !right) {
            seq.push_back(root->val);
        }
        if (left) {
            getLeaves(left, seq);
        }
        if (right) {
            getLeaves(right, seq);
        }
    }
    
};
