/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector<int> s1,s2;
        getLeaf(root1,s1);
        getLeaf(root2,s2);
        return s1==s2;
        
    }
    
    void getLeaf(TreeNode* root,vector<int>& leafseq){
        if(root==nullptr)return;
        if(root->left==nullptr&&root->right==nullptr){
            leafseq.push_back(root->val);
            return;
        }
        if(root->left)getLeaf(root->left,leafseq);
        if(root->right)getLeaf(root->right,leafseq);
    }
};
