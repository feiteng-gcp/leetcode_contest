/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    void f(TreeNode *c, vector <int> &v) {
        if (c == NULL) return;
        if (c->left == NULL && c->right == NULL) v.push_back(c->val);
        f(c->left, v);
        f(c->right, v);
    }
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector <int> v1, v2;
        f(root1, v1);
        f(root2, v2);
        return v1 == v2;
    }
};
