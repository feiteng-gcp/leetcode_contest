/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    void dfs( TreeNode * root , vector< int > & save ){
        if( root ){
            if( root->left == NULL && root->right== nullptr )
            save.push_back( root->val ) ; 
            dfs( root->left ,save ); 
            dfs( root ->right ,save ) ; 
        }
    }
    
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector< int > check; 
        dfs( root1 , check) ;
        vector< int > save ; 
        dfs( root2, save) ;
        if( check.size() !=  save.size() ) 
             return false ; 
        for( int i=0 ; i< save.size() ; i++){
        if( check[i] !=save[i] ) 
            return false;  
        }
        
        return true   ; 
        
        
    }
};
