/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void GetLeafOrder( TreeNode* root, vector<int> &res )
    {
        if ( !root ) return;
        if ( !root->left && !root->right )
        {
            res.push_back(root->val);
            return;
        }    
        if ( root->left )
            GetLeafOrder(root->left, res );
        if ( root->right )
            GetLeafOrder(root->right, res );
    }
    
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        
        vector<int> res1, res2;
        GetLeafOrder(root1, res1 );
        GetLeafOrder(root2, res2 );
        
        return res1 == res2;
    }
};
