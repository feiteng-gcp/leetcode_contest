/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */


void check(TreeNode *root,vector<int> &X) {
    if(root==NULL) return;
    if(root->left==NULL&&root->right==NULL) {
        X.push_back(root->val);  return;
    }
    check(root->left,X);  check(root->right,X);
}

class Solution {
public:

    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        vector<int> X,Y;
        
        check(root1,X);  check(root2,Y);
        if(X.size()!=Y.size()) return false;
        for(int i=0;i<X.size();++i) {
            if(X[i]!=Y[i]) return false;
        }
        return true;
    }
};
