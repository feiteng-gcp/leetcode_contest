/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool leafSimilar(TreeNode* root1, TreeNode* root2) {
        
        vector<int> v1, v2;
        _DFS(root1, v1);
        _DFS(root2, v2);
        return v1 == v2;
        
    }
    
    void _DFS(TreeNode *pNode, vector<int> & vec){
        if(pNode){
            
            if(pNode->left == nullptr && pNode->right == nullptr)
                vec.push_back(pNode->val);
            
            _DFS(pNode->left, vec);
            _DFS(pNode->right, vec);
            
        }
        return;
    }
};
