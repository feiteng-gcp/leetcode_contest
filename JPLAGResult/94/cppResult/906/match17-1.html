class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        unordered_map<int, set<int>> obs;
        for (auto& point : obstacles) {
            obs[point[0]].insert(point[1]);
        }

        vector<int> dir = { 0, 1, -1, 0, 0, -1, 1, 0 };
        int cur_dir = 0;
        int cur_x = 0, cur_y = 0;
        int max_dis = 0;
        for (auto& command : commands) {
            if (command == -2) {
                cur_dir = (cur_dir + 1) % 4;
            }
            else if (command == -1) {
                cur_dir = (cur_dir + 3) % 4;
            }
            else {
                int target_x = cur_x + dir[cur_dir * 2] * command;
                int target_y = cur_y + dir[cur_dir * 2 + 1] * command;
                for (int dis = 1; dis <= command; dis++) {
                    int new_x = cur_x + dir[cur_dir * 2] * dis;
                    int new_y = cur_y + dir[cur_dir * 2 + 1] * dis;
                    if (obs.find(new_x) != obs.end() && obs[new_x].find(new_y) != obs[new_x].end()) {
                        target_x = cur_x + dir[cur_dir * 2] * (dis - 1);
                        target_y = cur_y + dir[cur_dir * 2 + 1] * (dis - 1);
                        break;
                    }
                }
                cur_x = target_x;
                cur_y = target_y;
                max_dis = max(cur_x * cur_x + cur_y * cur_y, max_dis);
            }
        }
        return max_dis;
    }
};
