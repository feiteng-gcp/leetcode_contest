const int dy[4] = {1, 0, -1, 0};
const int dx[4] = {0, 1, 0, -1};


class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
	vector<pair<int, int> > P;
	REP (i, obstacles.size()) {
	    P.emplace_back(obstacles[i][0], obstacles[i][1]);
	}
	sort(P.begin(), P.end());
        
	int curX = 0, curY = 0, curD = 0;

	int ans = 0;
	EACH (e, commands) {
	    if (*e == -1) {
		curD = (curD + 1) % 4;
	    } else if (*e == -2) {
		curD = (curD + 3) % 4;
	    } else {
		REP (t, *e) {
		    int nxtX = curX + dx[curD];
		    int nxtY = curY + dy[curD];
		    pair<int, int> key(nxtX, nxtY);
		    int k = lower_bound(P.begin(), P.end(), key) - P.begin();
		    if (k < (int)P.size() && P[k] == key) {
			break;
		    } else {
			curX = nxtX;
			curY = nxtY;
		    }
		}
	    }
	    amax(ans, curX * curX + curY * curY);
	}
	return ans;
    }
};
