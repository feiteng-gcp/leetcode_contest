class Solution {
public:
  unsigned long long hash(int x, int y) {
    unsigned long long res = unsigned(x);
    res <<= 32;
    res |= unsigned(y);
    return res;
  }

  int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
    vector<pair<int, int>> moves;
    moves.push_back({0, 1});
    moves.push_back({-1, 0});
    moves.push_back({0, -1});
    moves.push_back({1, 0});
    unordered_set<long long> obs;
    for (int i = 0; i < obstacles.size(); ++i) {
      obs.insert(hash(obstacles[i][0], obstacles[i][1]));
        //cout << hash(obstacles[i][0], obstacles[i][1]) << endl;
    }
    int dir = 0;

    int x = 0, y = 0;
    int maxx = 0;
    for (int i = 0; i < commands.size(); ++i) {
      if (commands[i] == -1) {
        dir = (dir + 4 - 1) % 4;
      } else if (commands[i] == -2) {
        dir = (dir + 1) % 4;
      } else {
        for (int j = 0; j < commands[i]; ++j) {
            //cout << dir << " " << x << " " << y << endl;
          int nextx = x + moves[dir].first;
          int nexty = y + moves[dir].second;
          if (obs.find(hash(nextx, nexty)) == obs.end()) {
            x = nextx;
            y = nexty;
            int tmp = x * x + y * y;
            if (tmp > maxx)
              maxx = tmp;
          } else {
              //cout << nextx << " " << nexty << endl;
            break;
          }
        }
      }
    }
    return maxx;
  }
};
