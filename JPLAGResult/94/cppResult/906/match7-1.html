class Solution {
public:
	int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
		unordered_map<int, vector<int>> dictX;
		unordered_map<int, vector<int>> dictY;
		for (const auto& v : obstacles)
		{
			dictX[v[0]].push_back(v[1]);
			dictY[v[1]].push_back(v[0]);
		}
        for(auto& p:dictX)sort(p.second.begin(),p.second.end());
        for(auto& p:dictY)sort(p.second.begin(),p.second.end());
        
		vector<int> dx{ -1,0,1,0 };
		vector<int> dy{ 0,1,0,-1 };
		int dir = 1;

		int x = 0, y = 0;
        int res=0;
		for (auto c : commands)
		{
            //cout<<"("<<x<<" "<<y<<" "<<dir<<")"<<" "<<c<<": ";
			if (c == -2)
			{
				dir = (dir - 1 + 4) % 4;
			}
			else if (c == -1)
			{
				dir = (dir + 1) % 4;
			}
			else
			{
				int x1 = x + c*dx[dir];
				int y1 = y + c*dy[dir];

				if (x == x1)
				{
					y=getNode(y,y1,dictX[x]);
				}
				else
				{
				    x=getNode(x,x1,dictY[y]);
				}
			}
            if(c>=1)
            {
                res=max(res,x*x+y*y);
            }
            // cout<<"("<<x<<" "<<y<<" "<<dir<<")"<<endl;
		}
		return res;
	}

	int getNode(int start, int finish, vector<int>& dict)
	{
        if(start<finish)
        {
            auto p=upper_bound(dict.begin(),dict.end(),start);
            if(p==dict.end() || *p>finish)return finish;
            else return *p-1;
        }
        else
        {
            auto p=lower_bound(dict.begin(),dict.end(),finish);
            auto p1=lower_bound(dict.begin(),dict.end(),start);
            if(p==dict.end() || p1==p)return finish;
            else 
            {
                --p1;
                return *p1+1;
            }
        }
	}
};
