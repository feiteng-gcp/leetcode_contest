class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        set<vector<int> > o;
        for (vector<int> v : obstacles) {
            o.insert(v);
        }
        vector<int> p = {0, 0};
        vector<vector<int> > dirs = {{0, 1}, {1,0}, {0,-1}, {-1, 0}};
        int d = 0;
        int res = 0;
        for (int i = 0; i < commands.size(); i++) {
            if (commands[i] == -1) d++;
            if (commands[i] == -2) d--;
            else {
                for (int j = 0; j < commands[i]; j++) {
                    p[0] += dirs[d][0];
                    p[1] += dirs[d][1];
                    if (o.find(p) != o.end()) {
                        p[0] -= dirs[d][0];
                        p[1] -= dirs[d][1];
                        break;
                    }
                    res = max(res, p[0] * p[0] + p[1] * p[1]);
                }
            }
            
            d += 4;
            d %= 4;
        }
        return res;
    }
};
