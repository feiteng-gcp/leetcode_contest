class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        int x = 0;
        int y = 0;
        int max_distances2 = 0;
        int direction = 0; // north, anti clock
        unordered_map<int, unordered_set<int>> obstacles_map;
        for(auto obstacle:obstacles){
            obstacles_map[obstacle[0]].insert(obstacle[1]);
        }
        for(int command:commands){
            if(command == -2){
                direction = (direction + 1) % 4;
            } else if(command == -1){
                direction = (direction + 3) % 4;
            } else{
                forward(direction, x, y, command, obstacles_map);
            }
            max_distances2 = max(max_distances2, x*x + y*y);
        }
        return max_distances2;
    }
    
    void forward(int direction, int& x, int& y, int command,unordered_map<int, unordered_set<int>>& obstacles_map){
        if(direction == 0){
            for(int i=0;i<command;++i){
                if(obstacles_map[x].count(y+1) == 0) y++;
                else return;
            }
        } else if(direction == 1){
            for(int i=0;i<command;++i){
                if(obstacles_map[x-1].count(y) == 0) x--;
                else return;
            }
        } else if(direction == 2){
            for(int i=0;i<command;++i){
                if(obstacles_map[x].count(y-1) == 0) y--;
                else return;
            }
        } else{
            for(int i=0;i<command;++i){
                if(obstacles_map[x+1].count(y) == 0) x++;
                else return;
            }
        }
    }
};
