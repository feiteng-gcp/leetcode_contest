using namespace std;
typedef long long LL;

 
struct Node { 
	LL a, b;
 
};

bool operator < (const Node & a, const Node &b) {
	if (a.a < b.a) return true;
	if (a.a > b.a) return false;
	return a.b < b.b;
}
Node zz[330];
class Solution {
public:
	int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
		set<Node> sn;
		for (auto & x : obstacles) {
			sn.insert(Node{ x[0], x[1] });
		}

		int d[4][2] = { {0,1}, {1,0}, {0,-1}, {-1,0} };
		LL res = 0, a = 0, b = 0, na , nb;
		Node now;
		int nd = 0;

		for (auto x : commands) {
			if (x == -1) {
				nd = (nd + 1) % 4;
				
			}
			else if (x == -2) {
				nd = (nd +3) % 4;
			}
			else {
				
				for (int i = 0; i < x; i++) {
					now.a = a + d[nd][0];
					now.b = b + d[nd][1];
					if (sn.count(now) >0 ) {
						break;
					}
					a = now.a;
					b = now.b;
				}

				auto t = a*a + b*b;
				res = max(t, res);
			}

		}
		return res;
	}
};
