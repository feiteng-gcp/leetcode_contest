class Solution {
public:
    int dx[4] = {0, 1, 0, -1};
    int dy[4] = {1, 0, -1, 0};
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        set<pair<int, int>> obs;
        for (auto &ob : obstacles) {
            obs.emplace(ob[0], ob[1]);
        }
        int dir = 0;
        int x = 0;
        int y = 0;
        int res = 0;
        for (auto &cmd : commands) {
            if (cmd == -1) {
                dir = (dir + 1) % 4; 
            } else if (cmd == -2) {
                dir = (dir + 3) % 4;
            } else {
                for (int i = 0; i < cmd; i++) {
                    int nx = x + dx[dir];
                    int ny = y + dy[dir];
                    if (obs.find({nx, ny}) != obs.end()) {
                        break;
                    }
                    x = nx;
                    y = ny;
                }
                res = max(res, x * x + y * y);
            }
        }
        return res;
    }
    
    
};
