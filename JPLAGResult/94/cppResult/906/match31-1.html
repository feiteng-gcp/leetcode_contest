typedef long long LL;
class Solution {
public:
    set<pair<LL, LL>> data;
    LL X, Y, ret;
    int dir_idx;

    void move(const int d) {
      if (dir_idx == 0) {
        for (int i = 0; i < d; i++) {
          LL YY  = Y + 1;
          if (data.find(make_pair(X, YY)) == data.end()) {
            Y = YY;
            ret = max(ret, X*X+Y*Y);
          } else return;
        }
      } else if (dir_idx == 1) {
        for (int i = 0; i < d; i++) {
          LL XX = X + 1;
          if (data.find(make_pair(XX, Y)) == data.end()) {
            X = XX;
            ret = max(ret, X*X+Y*Y);
          } else return;
        }
      } else if (dir_idx == 2) {
        for (int i = 0; i < d; i++) {
          LL YY = Y - 1;
          if (data.find(make_pair(X, YY)) == data.end()) {
            Y = YY;
            ret = max(ret, X*X+Y*Y);
          } else return;
        }

      } else {
        for (int i = 0; i < d; i++) {
          LL XX = X - 1;
          if (data.find(make_pair(XX, Y)) == data.end()) {
            X = XX;
            ret = max(ret, X*X+Y*Y);
          } else return;
        }
      }
    }

    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
      for (int i = 0; i < obstacles.size(); i++) {
        int x = obstacles[i][0], y = obstacles[i][1];
        data.insert(make_pair(x, y));
      }

      dir_idx = 0;
      X = 0LL, Y = 0LL, ret = 0LL;
      for (int i = 0; i < commands.size(); i++) {
        if (commands[i] == -2) {
          dir_idx = (dir_idx - 1 + 4) % 4;
        } else if (commands[i] == -1) {
          dir_idx = (dir_idx + 1) % 4;
        } else {
          move(commands[i]);
          // X = XX;
          // Y = YY;
        }
      }

      // cout << "X=" << X << " Y=" << Y << endl;
      return ret;
    }
};
