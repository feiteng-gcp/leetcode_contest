const int p[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        map<int, set<int>> obs;
        for (int i = 0; i < obstacles.size(); ++i)
        {
            obs[obstacles[i][0]].insert(obstacles[i][1]);
        }
        vector<int> pos(2, 0);
        int d = 0, ans = 0;
        for (int i = 0; i < commands.size(); ++i)
        {
            int c = commands[i];
            if (c == -2)
            {
                d = (d + 3) % 4;
            }
            else if (c == -1)
            {
                d = (d + 1) % 4;
            }
            else
            {
                while (c--)
                {
                    vector<int> next(pos);
                    next[0] += p[d][0];
                    next[1] += p[d][1];
                    //cout << next[0] << ' ' << next[1] << endl;
                    if (obs[next[0]].count(next[1]) == 0)
                    {
                        pos = next;
                    }
                    ans = max(ans, pos[0] * pos[0] + pos[1] * pos[1]);
                }
            }
        }
        return ans;
    }
};
