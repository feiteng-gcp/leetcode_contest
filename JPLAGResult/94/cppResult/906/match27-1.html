class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obs) {
        int x = 0, y = 0, dir = 0;

        const int deltas[4][2] = {{0,1},{1,0},{0,-1},{-1,0}};
        unordered_map<int, unordered_set<int>> bx;
        for (auto &ob : obs)
            bx[ob[0]].insert(ob[1]);
        int ret = 0;
        for (int cmd : commands) {
            if (cmd == -2)
                dir = (dir + 3) % 4;
            else if (cmd == -1)
                dir = (dir + 1) % 4;
            else {
                for (int i = 0; i < cmd; i++) {
                    int x2 = x + deltas[dir][0];
                    int y2 = y + deltas[dir][1];
                    if (bx.count(x2) && bx[x2].count(y2))
                        break;
                    x = x2;
                    y = y2;
                    ret = max(ret, x * x + y * y);
                }
            }
            //cout << dir << ' ' << x << ',' << y << endl;
        }
        //cout << endl;
        return ret;
    }
};
