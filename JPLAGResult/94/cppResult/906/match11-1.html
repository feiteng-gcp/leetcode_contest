class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        unordered_map<int, unordered_set<int>> c, m;
        for (auto& p : obstacles) {
            m[p[0]].insert(p[1]);
            // c[p.second].insert(p.first);
        }
        
        int x = 0, y = 0, mx = 0, my = 0;

        vector<pair<int, int>> dirs = {{1,0}, {0,1}, {-1,0}, {0,-1}};
        int d = 0;
        int res = 0;
        for (int n : commands) {

            if (n == -1) {
                d = (d + 1) % 4;
            } else if (n == -2) {
                d = (d - 1 + 4) % 4;
            } else {
                for (int i = 0; i < n; ++i) {
                    auto dir = dirs[d];
                    int newx = x + dir.first;
                    int newy = y + dir.second;
                    if (m[newy].count(newx)) {
                        break;
                    }
                    x = newx;
                    y = newy;
                    
            // cout << x << ", " << y << endl;
                    
                }
            }
            res = max(res, x * x + y * y);
        }
        
        return res;
    }
};
