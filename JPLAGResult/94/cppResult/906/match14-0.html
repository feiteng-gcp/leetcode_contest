class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        int i, j, k, x, y, t, ans = 0;
        x = y = 0;
        k = 0;
        vector<vector<int>> step = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}}; 
        map<pair<int, int>, int> ob;
        for (i = 0; i < obstacles.size(); i++) {
            ob[make_pair(obstacles[i][0], obstacles[i][1])] = 1;
        }
        for (i = 0; i < commands.size(); i++) {
            if (commands[i] == -1) {
                k = (k + 1) % 4;
            }
            else if (commands[i] == -2) {
                k = (k + 3) % 4;
            }
            else {
                for (t = 0; t < commands[i]; t++) {
                    if (ob[make_pair(x + step[k][0], y + step[k][1])] == 1) {
                        break;
                    }
                    x = x + step[k][0];
                    y = y + step[k][1];
                }
                //cout << x << " " << y << endl;
                ans = max(ans, x * x + y * y);
            }
        }
        return ans;
    }
};
