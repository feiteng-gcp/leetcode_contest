class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        pair<int, int> pos{0,0};
        set<pair<int, int>> ob;
        for (int i = 0; i < obstacles.size(); i += 1) {
            ob.insert({obstacles[i][0], obstacles[i][1]});
        }
        vector<pair<int, int>> dir{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int cd = 0;
        int ans = 0;
        for (int i = 0; i < commands.size(); i += 1) {
            if (commands[i] == -2) {
                cd = (cd - 1 + 4) % 4;
            }
            else if (commands[i] == -1) {
                cd = (cd + 1) % 4;
            }
            else {
                int t = 0;
                for (int j = 1; j <= commands[i]; j += 1) {
                    int x = pos.first + j * dir[cd].first;
                    int y = pos.second+ j * dir[cd].second;
                    if (ob.find({x, y}) != ob.end()) {
                        break;
                    }
                    t += 1;
                }
                
                pos.first += t * dir[cd].first;
                pos.second+= t * dir[cd].second;
                ans = max(ans, pos.first * pos.first + pos.second * pos.second);
            }
        }
     //   cout << pos.first << " " << pos.second;
        return ans;
    }
};
