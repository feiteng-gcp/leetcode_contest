class Solution:
    def lenLongestFibSubseq(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        n = len(A)
        dp = [collections.defaultdict(int) for i in range(n)]
        for i in range(n):
            for j in range(i):
                dp[i][A[j]] = 2
        ans = 0
        for i in range(n):
            for j in range(i):
                if A[i] - A[j] in dp[j]:
                    dp[i][A[j]] = max(dp[i][A[j]], dp[j][A[i] - A[j]] + 1)
                    ans = max(ans, dp[i][A[j]])
        return ans
