class Solution:
    def lenLongestFibSubseq(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        dp = [0] * len(A)
        prev = [collections.defaultdict(int) for _ in range(len(A))]
        res = 0
        
        for i in range(1, len(A)):
            for j in range(i-1, -1, -1):
                if A[i] - A[j] in prev[j]:
                    prev[i][A[j]] = prev
[j][A[i] - A[j]] + 1
                    res = max(res, prev[i][A[j]])
                else:
                    prev[i][A[j]] = 2
        
        return res
