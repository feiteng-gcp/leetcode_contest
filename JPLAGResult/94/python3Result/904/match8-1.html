class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        res1 = []
        self.returnLeaves(root1, res1)
        res2 = []
        self.returnLeaves
(root2, res2)
        if len(res1) != len(res2):
            return False
        for i in range(len(res1)):
            if res1[i] != res2[i]:
                return False
        return True
    
    def returnLeaves(self, node, res):
        if not node:
            return
        if not node.left and not node.right:
            res.append(node.val)
        if node.left:
            self.returnLeaves(node.left, res)
        if node.right:
            self.returnLeaves(node.right, res)
