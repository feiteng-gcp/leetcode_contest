# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        
def dfs(node,temp):
            if node is None:return
            if node.left is None and node.right is None:
                temp.append(node.val)
                return
            dfs(node.left,temp)
            dfs(node.right,temp)
        rst1,rst2=[],[]
        dfs(root1,rst1)
        dfs(root2,rst2)
        return rst1==rst2
