# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        s1=[]
        def find1(root):
            if root.right==None and root.left==None:
                s1.append(root.val)
                return
            if root.right!=None:
                find1(root.right)
            if root.left!=None:
                find1(root.left)
        s2=[]
        def find2(root):
            
if root.right==None and root.left==None:
                s2.append(root.val)
                return
            if root.right!=None:
                find2(root.right)
            if root.left!=None:
                find2(root.left)
        find1(root1)
        find2(root2)
        if s1==s2:
            return True
        else:
            return False
