class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        def post(root, leafList):
            childCnt = 0
            if root.left is not None:
                childCnt += 1
                post(root.left, leafList)
            if root.right is not None:
                childCnt +=1
                post(root.right, leafList)
            if childCnt==0:
                leafList.append(root.val)
        leafs1 = []
        leafs2 = []
        if root1 is not None:
            post(root1, leafs1)
        if root2 is not None:
            post(root2, leafs2)
        if len(leafs1) != len(leafs2):
            return False
        else:
            if all(x==y for x, y in zip(leafs1, leafs2)):
                return True
            else:
                return False
