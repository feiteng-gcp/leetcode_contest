# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        def get_leaf(root, res):
            if root is None:
                return
            
            if root.left is None and root.right is None:
                res.append(root.val)
            else:
                get_leaf(root.left, res)
                get_leaf(root.right, res)
                
        list1, list2 = [], []
        get_leaf(root1, list1)
        get_leaf(root2, list2)
        
        if len(list1) != len(list2):
            return False
        
        for i in range(len(list1)):
            if list1[i] != list2[i]:
                return False
            
        return True
