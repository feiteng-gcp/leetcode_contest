# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def allleaf(self,root):
        if root==None:
            return []
        elif root.left==None and root.right==None:
            return [root
.val]
        else:
            return self.allleaf(root.left)+self.allleaf(root.right)
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        return self.allleaf(root1)==self.allleaf(root2)
