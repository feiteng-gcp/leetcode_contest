# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        l1, l2 = [], []
        self.dfs(root1, l1)
        self.dfs(root2, l2)
        return l1 == l2

    
def dfs(self, root, ls):
        if not root: return
        if not root.left and not root.right:
            ls.append(root.val)
            return
        self.dfs(root.left, ls)
        self.dfs(root.right, ls)
