# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        def dfs(root, leafs):
            if not root:
                return
            if not root.left and not root.right:
                leafs.append(root.val)
            else:
                dfs(root.left, leafs)
                dfs(root.right, leafs)
        leafs1, leafs2 = [], []        
        dfs(root1, leafs1)
        dfs(root2, leafs2)
        return leafs1 == leafs2
