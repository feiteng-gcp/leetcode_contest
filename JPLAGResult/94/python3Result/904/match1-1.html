# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        
        def d(r):
            ans=[]
            
def s(n):
                if not n:
                    return
                if n.left is None and n.right is None:
                    ans.append(n.val)
                else:
                    s(n.left)
                    s(n.right)
            s(r)
            return ans
        
        return d(root1)==d(root2)
