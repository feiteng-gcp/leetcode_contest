# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        a, b = [], []
        self.DFS(root1, a, 0)
        self.DFS(root2, b, 0)
        
return a == b
        
    def DFS(self, root, res, l):
        if root is None:
            return
        if root.left is None and root.right is None:
            res.append(root.val)
        else:
            self.DFS(root.left, res, l+1)
            self.DFS(root.right, res, l+1)
