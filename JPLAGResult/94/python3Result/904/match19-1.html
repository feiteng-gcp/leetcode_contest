# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        def leaf_seq(tree):
            if not tree: return []
            seq = []
            
def down(node):
                if not node: return
                if not node.left and not node.right:
                    seq.append(node.val)
                else:
                    down(node.left)
                    down(node.right)
            down(tree)
            return seq

        
        return leaf_seq(root1) == leaf_seq(root2)
