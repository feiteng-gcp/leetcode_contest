# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def getleaf(self, node):
        ret = []
        isleaf = True
        if node.left  is not None:
            isleaf = False
            ret += self.getleaf(node.left)
        if node.right is not None:
            isleaf = False
            ret += self.getleaf(node.right)
        if isleaf:
            return [node.val]
        else:
            return ret
    
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        seq1 = self.getleaf(root1)
        seq2 = self.getleaf(root2)
        if len(seq1) != len(seq2): return False
        return all(a == b for a, b in zip(seq1, seq2))
