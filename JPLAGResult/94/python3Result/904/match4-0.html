# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def helpfun(self, root):
        if root is None:
            return ""
        if root.left is None and root.right is None:
            return str(root.val)
        else:
            return self.helpfun(root.left)+self.helpfun(root.right)
        
    
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        return self.helpfun(root1) == self.helpfun(root2)
