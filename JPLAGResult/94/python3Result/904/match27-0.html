# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def leafSimilar(self, root1, root2):
        """
        :type root1: TreeNode
        :type root2: TreeNode
        :rtype: bool
        """
        seq1 = []
        seq2 = []
        def dfs(node, arr):
            if not node.left and not node.right:
                arr.append(node.val)
            if node.left:
                dfs(node.left, arr)
            if node.right:
                dfs(node.right, arr)

        if root1:
            dfs(root1, seq1)
        if root2:
            dfs(root2, seq2)
        # print(seq1)
        # print(seq2)
        return seq1 == seq2
