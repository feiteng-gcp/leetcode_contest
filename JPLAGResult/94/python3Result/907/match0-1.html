class Solution:
    def can_finish(self, piles, H, K):
        for p in piles:
            t = math.ceil(p / K)
            if H < t:
                return False
            H -= t
        return True
    
    def minEatingSpeed(self, piles, H):
        """
        :type piles: List[int]
        :type H: int
        :rtype: int
        """
        left, right = 1, max(piles)
        
while left < right:
            mid = (left + right) // 2
            if self.can_finish(piles, H, mid):
                right = mid
            else:
                left = mid + 1
        
        return left
