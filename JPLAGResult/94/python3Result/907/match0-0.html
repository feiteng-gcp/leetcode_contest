class Solution:
    def comer(self, piles, m):
        t = 0
        for p in piles:
            t += p//m
            if p % m !=0:
                t+=1
        return t
    def minEatingSpeed(self, piles, H):
        """
        :type piles: List[int]
        :type H: int
        :rtype: int
        """
        h, l = 10**9+5, 1
        while(h>l):
            m=(h+l)//2
            if self.comer(piles, m) <= H:
                h = m
            else:
                l = m+1
        return h
