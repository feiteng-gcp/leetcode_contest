class Solution {
    public int minimumLengthEncoding(String[] words) {
        Arrays.sort(words, new Comparator<String>(){
           public int compare(
String A, String B){
               return B.length() - A.length();
           } 
        });
        
        Queue<String> q = new LinkedList<>();
        
        for(String w : words){
            int size = q.size();
            boolean flag = false;
            
            while(size-- > 0)
{
                String c = q.remove();
                int idx = c.indexOf(w);
                if(idx != -1 && idx + w.length() == c.length()){
                    flag = true;
                }
                q.add(c);
            }
            
            if(!flag) q.add(w);
        }
        
        int res = 0;
        // System.out.println(q.size());
        
        while(!q.isEmpty()){
            res += q.remove().length() + 1;
        }
        return res;
    }
}
