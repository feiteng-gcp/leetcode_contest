class Solution {
    private class Node{
        Node[] nexts = new Node[256];
    }

    Node root = new Node();
    private boolean add(
String word){
        boolean nw = false;
        Node cur = root;
        for(int i = word.length() - 1;i>=0;i--){
            char c = word.charAt(i);
            if(cur.nexts[c] == null){
                nw = true;
                cur.nexts[c] = new Node();
            }
            cur = cur.nexts[c];
        }

        return nw;
    }

    public int minimumLengthEncoding(String[] words) {
        Arrays.sort(words,(o1, o2) -> o2.length() - o1.length());

        int ws = 0,len = 0;
        for(String w : words){
            if(add(w)){
                len += w.length();
                ws++;
            }
        }
        len += ws;
        return len;
    }
}
