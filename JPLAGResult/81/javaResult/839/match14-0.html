class Solution {
    public static class StrCom implements Comparator<String> {
        @Override
        public int compare(String o1, String o2) {
            return o2.length() - o1.length();
        }
    }
    
    public int minimumLengthEncoding(String[] words) {
        StringBuilder encode = new StringBuilder();
        PriorityQueue<String> strSet = new PriorityQueue<String>(new StrCom());
        for (String word: words) {
            strSet.offer(word);
        }
        while (!strSet.isEmpty()) {
            String word = strSet.poll();
            if (encode.length() == 0) {
                encode.append(word);
                encode.append("#");
            } else {
                String encodeStr = encode.toString();
                int index = encodeStr.indexOf(word);
                if (index == -1 || encode.charAt(index + word.length()) != '#') {
                    encode.append(word);
                    encode.append("#");
                }
            }
        }
        return encode.length();
    }
}
