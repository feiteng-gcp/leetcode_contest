class Solution {
    class TrieNode{
        boolean end;
        TrieNode[]children;
        TrieNode(){
            children = new TrieNode[26];
            end = false;
        }
        
    }
    
    public int minimumLengthEncoding(String[] words) {
        TrieNode root = new TrieNode();
        
        // Load all the words in Trie in reverse order
        for(String word : words){
            TrieNode cur = root;
            StringBuilder rev = new StringBuilder(word);
            rev.reverse();
            for(int i = 0;i<rev.length();i++){
                char ch = rev.charAt(i);
                if(cur.children[ch-'a'] == null)
                    cur.children[ch-'a'] = new TrieNode();
                cur = cur.children[ch-'a'];
            }
            
            cur.end = true;
        }
        
        // Get the root to leaf heights of the trie and add all the heights to the ans
        
        int ans = 0;
        return getHeight(root,0);
        
        
    }
    
    int getHeight(TrieNode cur, int ht ){
        if(cur == null) return 0;
        
        int ans = 0;    
        for(TrieNode child : cur.children){
            ans+= getHeight(child,ht+1);
        }
        if(ans==0 && cur.end) {
            // This is a leaf node
            ans = ht+1; // This is for # , so do ht+1
        }
        
        return ans;
    }
}
