class Solution {
    class Trie {
        Trie[] children = new Trie[26];
        boolean noChildren = true;
    }
    Trie root = new Trie();
    public int minimumLengthEncoding(String[] words) {
        for (String s: words) {
            char[] sa = s.toCharArray();
            Trie p = root;
            for (int i = sa.length - 1; i >= 0; i--) {
                char c = sa[i];
                if (p.children[c - 'a'] == null) {
                    p.children[c - 'a'] = new Trie();
                    p.noChildren = false;
                }
                p = p.children[c - 'a'];
            }
        }
        return help(root, 0);
    }
    private int help(Trie p, int l) {
        if (p.noChildren) return l + 1;
        
int res = 0;
        for (int i = 0; i < 26; i++) {
            if (p.children[i] != null)
                res += help(p.children[i], l + 1);
        }
        return res;
    }
}
