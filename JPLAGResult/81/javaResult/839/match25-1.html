class Solution {
    public int minimumLengthEncoding(String[] words) {
        TrieNode root = new TrieNode();
        Set<String> set = new HashSet<>();
        for (String s : words) {
            set.add(s);
        }
        for (String s : set) {
            build(root, s);
        }

        int len = 0;
        for (String s : set) {
            int count = count(root, s);
            if (count == 1) {
                len += s.length() + 1;
            }
        
}
        return len;
    }

    private int count(TrieNode root, String s) {
        for (int i = s.length() - 1; i >= 0; i--) {
            char c = s.charAt(i);
            root = root.next[c - 'a'];
        }
        return root.count;
    }

    private void build(TrieNode root, String s) {
        
for (int i = s.length() - 1; i >= 0; i--) {
            root.count++;
            char c = s.charAt(i);
            if (root.next[c - 'a'] == null) {
                root.next[c - 'a'] = new TrieNode();
            }
            root = root.next[c - 'a'];
        }
        root.count++;
        root.isWord = true;
    }

    private class TrieNode {
        TrieNode[] next = new TrieNode[26];
        int count;
        boolean isWord;
    }
}
