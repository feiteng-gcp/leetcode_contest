import java.util.*;


class Solution {
    public int minimumLengthEncoding(String[] words) {

        // Sort by decreasing length.
        Arrays.sort(words, new Comparator<String>(){
            public int compare(String a, String b) { return b.length()-a.length(); }
        });
        
        Set<String> suffixes = new HashSet<String>();
        int Slen = 0;
        
        for(String word : words) {
            
            // If we already have a suffix containing this word, we would use that without modifying the encoded S returned.
            if (suffixes.contains(word)) { continue; }
            
            // Else we would have to add this word.
            // Emulate making encoded S longer.
            Slen += word.length() + 1; // # symbol
            
            // Memorize the suffixes that we can now support for future words.
            for(int startIdx = 0; startIdx <= word.length()-1; startIdx++) {
                String suffix = word.substring(startIdx, word.length());
                suffixes.add(suffix);
            }
            
        }
        return Slen;
        
    }
}
