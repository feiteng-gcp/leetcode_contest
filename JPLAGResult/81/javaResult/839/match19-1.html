class Solution {
    public int minimumLengthEncoding(String[] words) {
        int len = words.length;
        int[] sub = new int[len];
        for (int i = 0; i < len; ++i) {
            words[i] = new StringBuilder(words[i]).reverse().toString();
        }
        Arrays.sort(words);
        for (int i = 0; i < len - 1; ++i) {
            if (words[i + 1].indexOf(words[i]) == 0) {
                sub[i] = 1;
            }
        }
/*        for (int i = 0; i < len; ++i)
            for (int j = 0; j < len; ++j) {
                if (i == j) continue;
                if (words[i].length() > words[j].length()) continue;
                int k = 0;
                for (; k < words[i].length(); k ++) {
                    if (words[i].charAt(words[i].length() - k - 1) != words[j].charAt(words[j].length() - k - 1)) {
                        break;
                    }
                }
                if (k == words[i].length()) {
                    sub[i] = 1;
                    break;
                }
            } */
        int result = 0;
        
for (int i = 0; i < len; ++i) {
            if (sub[i] != 1) result += words[i].length() + 1;
        }
        return result;
    } 
    
    
}
