class Solution {
    private class TrieNode {
        TrieNode[] nodes;
        int step;
        TrieNode(int step) {
            this.step = step;
            nodes = new TrieNode[26];
        }
    }
    
public int minimumLengthEncoding(String[] words) {
        Arrays.sort(words, (a, b) -> b.length() - a.length());
        TrieNode root = new TrieNode(1);
        int ret = 0;
        for (String word : words) {
            TrieNode current = root;
            for (int i = word.length() - 1; i >= 0; i--) {
                TrieNode next = current.nodes[(int)(word.charAt(i) - 'a')];
                if (next == null) {
                    if (i == 0) ret += current.step + 1;
                    current.nodes[(int)(word.charAt(i) - 'a')] = new TrieNode(current.step + 1);
                    current = current.nodes[(int)(word.charAt(i) - 'a')];
                } else {
                    current = next;
                
}
            }
        }
        return ret;
    }
}
