class Solution {
    public int minimumLengthEncoding(String[] words) {
        int n = words.length;
        Arrays.sort(words, (a, b) -> {
            return a.length() - b.length();
        });
        Set<Integer> set = new HashSet<>();
        
        for (int i = 0; i < n; i++) {
            //if (set.contains(i)) continue;
            int m1 = words[i].length();
            
            for (int j = i + 1; j < n; j++) {
                //if (set.contains(j)) continue;
                int m2 = words[j].length();
                
                if (words[j].substring(m2 - m1).equals(words[i])) {
                    set.add(i);
                    break;
                }
            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (!set.contains(i)) ans += (words[i].length() + 1);
        }
        
        return ans;
        
    }
}
