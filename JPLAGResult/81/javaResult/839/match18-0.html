class Solution {
    private static int total_len = 0;
    
    public int minimumLengthEncoding(String[] words) {
        Trie trie = new Trie();
		
		for (String word : words) {
			String rev = new StringBuilder(word).reverse().toString();
			trie.addWord(rev);
		}
		
		total_len = 0;
		trie.dfs_traversal();
		return total_len;
    }
    
    private static class Trie {
		private class Node {
			Node[] next;
			int depth;
			boolean isWord;
			
			public Node(int depth) {
				this.next = new Node[26];
				this.depth = depth;
				this.isWord = false;
			}
		}
		
		private Node root;
		
		public Trie() {
			this.root = new Node(0);
		}
		
		public void addWord(String word) {
			Node curr = this.root;
			int steps = 0;
			for (char c : word.toCharArray()) {
				if (curr.next[c - 'a'] == null) {
					curr.next[c - 'a'] = new Node(steps);
				}
				curr = curr.next[c - 'a'];
				++steps;
			}
			curr.isWord = true;
			curr.depth = steps;
		}
		
		public void dfs_traversal() {
			dfs(this.root);
		}
		
		private void dfs(Node curr) {
			// base case
			boolean isLeaf = true;
			for (int i = 0; i < 26; ++i) {
				if (curr.next[i] != null) {
					isLeaf = false;
					break;
				}
			}
			if (isLeaf) { 	// reach leaf
				total_len += curr.depth + 1;
				return;
			}
			
			// recurse
			for (int i = 0; i < 26; ++i) {
				if (curr.next[i] == null) continue;
				
				dfs(curr.next[i]);
			}
		}
	}
}
