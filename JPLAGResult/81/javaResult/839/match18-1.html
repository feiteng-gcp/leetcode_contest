class Solution {
    
    class TrieNode {
        TrieNode[] nodes;
        int total;
        TrieNode() {
            nodes = new TrieNode[26];
        }
    }
    
    
public int minimumLengthEncoding(String[] words) {
        TrieNode trie = new TrieNode();
        for (String s : words) {
            int len = s.length();
            TrieNode cur = trie;
            for (int i = len - 1; i >= 0; i --) {
                char c = s.charAt(i);
                if (cur.nodes[c - 'a'] == null) {
                    TrieNode child = new TrieNode();
                    cur.nodes[c - 'a'] = child;
                    cur.total += 1;
                    cur = child;
                } else {
                    cur = cur.nodes[c - 'a'];
                }
            }
        }
        int[] ret = new int[1];
        dfs(trie, ret, 1);
        return ret[0];
    }
    
    private void dfs(TrieNode root, int[] ret, int level) {
        
if (root.total == 0) {
            ret[0] += level;
            return;
        }
        for (int i = 0; i < 26; i ++) {
            if (root.nodes[i] != null) {
                dfs(root.nodes[i], ret, level + 1);
            }
        }
    }
}
