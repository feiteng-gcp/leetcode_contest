class Solution {
     public int minimumLengthEncoding( String[] w ) {
        int i,j,k,n= w.length;
        Set<String> set= new HashSet<>();
         for ( i = 0; i < n; ++i )
             set.add(w[i]);
         i= 0;
         w= new String[set.size()];
         for ( String x: set )
             w[i++]= x;
         n= w.length;
        boolean []covered= new boolean[n];
         Arrays.sort(w, new Comparator<String>() {
             public int compare( 
String x, String y ) {
                 return x.length() - y.length();
             }
         });
        for ( i = 0; i < n; ++i )
            for ( j = 0; j < i && !covered[j]; ++j )
                if ( i != j ) 
                    if ( isSuffix(w[i],w[j]) )
                        covered[j]= true ;
        int ans= 0;
        for ( i= 0; i < n; ++i )
            if ( !covered[i] )
                ans+= w[i].length()+1
;
        return ans;
    }

    private boolean isSuffix(String s, String t) {
        int m= s.length(), n = t.length();
        return m >= n && s.substring(m-n).equals(t);
    }
}
