class Solution {
    public int[] shortestToChar(String S, char C) {
        TreeSet<Integer> idx = new TreeSet<>();
        for (int i = 0; i < S.length(); i++) 
{
            if (S.charAt(i) == C) {
                idx.add(i);
            }
        }

        int[] res = new int[S.length()];
        for (int i = 0; i < S.length(); i++) {
            Integer n = idx.higher(i);
            Integer p = idx.floor(i);
            int dist = Integer.MAX_VALUE;
            if (n != null) {
                dist = Math.min(dist, Math.abs(n - i));
            }
            if (p != null) {
                dist = Math.min(dist, Math.abs(p - i));
            }
            res[i] = dist;
        }
        return res;
    }
}
