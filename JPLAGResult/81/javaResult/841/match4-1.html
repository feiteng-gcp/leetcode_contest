class Solution {
    public int[] shortestToChar(String S, char C) {
        List<Integer> idxs = new ArrayList<>();
        int n=S.length();
        
for(int i=0;i<n;i++){
            if(S.charAt(i)==C){
                idxs.add(i);
            }
        }
        int j = 0,m = idxs.size(),q;
        int[] res = new int[n];
        for(int i=0;i<n;i++){
            int p = Math.abs(i - idxs.get(j));
            if(j+1<m){
                q = idxs.get(j+1);
                if(i==q) {
                    j++;
                }
                p = Math.min(p,q-i);
            }
            res[i] = p;
            
        }
        return res;
    }
}
