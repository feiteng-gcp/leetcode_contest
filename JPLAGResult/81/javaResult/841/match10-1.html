import java.util.ArrayList;
import java.util.List;


class Solution {
    public int[] shortestToChar(String S, char C) {
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < S.length(); i++) {
            if (S.charAt(i) == C) list.add(i);
        }
        int[] ans = new int[S.length()];
        int pos = 0;
        for (int i = 0; i < S.length(); i++) {
            if (i < list.get(0)) ans[i] = Math.abs(i - list.get(0));
            else if (i >= list.get(list.size() - 1)) ans[i] = Math.abs(i - list.get(list.size() - 1));
            else {
                if (i == list.get(pos)) pos++;
                ans[i] = Math.min(Math.abs(i - list.get(pos)), Math.abs(i - list.get(pos - 1)));
            }
        }
        return ans;
    }
}
