class Solution {

	private int upperBound(int index) {

		int low = 0 , high = list.size();
		while (low < high) {
			int mid = (low + high) / 2;
			if (list.get(mid) <= index) {
				low = mid + 1;
			} else {
				high = mid;
			}
		}
		return high - 1;
		
	}
	
	private List<Integer> list = new ArrayList<>();
	
    public int[] shortestToChar(String S, char C) {
     
    	int i , length = S.length();
    	int[] ans = new int[length];
    	for (i = 0;i < length;i ++) {

    		if (S.charAt(i) == C) {
    			list.add(i);
    		}
    	}
    	for (i = 0;i < length;i ++) {
    		int index = upperBound(i) , result = - 1;
    		if (index >= 0) {
    			int pos = list.get(index);
    			int temp = Math.abs(i - pos);
    			if (temp < result || result < 0) {
    				result = temp;
    			}
    		}
    		index ++;
    		if (index >= 0 && index < list.size()) {
    			int pos = list.get(index);
    			int temp = Math.abs(i - pos);
    			if (temp < result ||result < 0) {
    				result = temp;
    			}
    		}
    		ans[i] = result;
    	}
    	return ans;
    		
    }
	
}
