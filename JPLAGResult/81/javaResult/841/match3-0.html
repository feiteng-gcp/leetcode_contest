class Solution {
    public int[] shortestToChar(String S, char C) {
        int n = S.length();
        int[] ans = new int[n];
        TreeSet<Integer> set = new TreeSet<>();
        
        for (int i = 0; i < n; i++) {
            if (S.charAt(i) == C) {
                set.add(i);
            }
        }
        
        
        for (int i = 0; i < n; i++) {
            if (S.charAt(i) != C) {
                int tmp = n;
                Integer low = set.floor(i);
                if (low != null) {
                    tmp = Math.min(i - low, tmp);
                }
                Integer high = set.higher(i);
                if (high != null) {
                    tmp = Math.min(high - i, tmp);
                }
                ans[i] = tmp;
            }
        }
        
        return ans;
    }
}
