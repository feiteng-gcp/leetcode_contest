	class Solution {
	    public int[] shortestToChar(String S, char C) {
	    	int n = S.length();
	    	int[] a = new int[n];
	    	Arrays.fill(a, 9999999);
	    	for(int i = 0;i < n;i++){
	    		if(S.charAt(i) == C){
	    			a[i] = 0;
	    		}
	    	}
	    	for(int i = 1;i < n;i++){
	    		a[i] = Math.min(a[i], a[i-1] + 1);
	    	}
	    	for(int i = n-2;i >= 0;i--){
	    		a[i] = Math.min(a[i], a[i+1] + 1);
	    	}
	    	return a;
	    }
	}
