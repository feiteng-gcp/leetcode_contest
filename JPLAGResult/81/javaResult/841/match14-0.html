class Solution {
    public int[] shortestToChar(String S, char C) {
        int[] res = new int[S.length()];
        List<Integer> q = new ArrayList<>();
        for (int i = 0 ; i < S.length(); i++) {
            if (S.charAt(i) == C) q.add(i);
        }
        int j = 0;
        for (int i = 0; i < S.length(); i++) {
            int dist1 = Math.abs(i - q.get(j));
            int dist2 = j+1 < q.size() ? Math.abs(i - q.get(j+1)) : Integer.MAX_VALUE;
            if (dist2 < dist1) j++;
            res[i] = Math.min(dist1, dist2);
        }
        return res;
    }
}
