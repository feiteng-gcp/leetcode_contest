class Solution {
public:
	int minimumLengthEncoding(vector<string>& words) {
		int n = words.size();
		set<string>x;
		for (int i = 0; i<n; i++) {
			x.insert(words[i]);

		}
		set<string>y;
		for (int i = 0; i<words.size(); i++) {
			if (y.find(words[i]) != y.end())continue;
			string s = words[i];
			for (int j = 1; j<s.length(); j++) {
					string a = words[i].substr(j);
					if (x.find(a) != x.end()) {
						auto it = x.find(a);
						y.insert(*it);
					}
				
			}
		}
		int ret = 0;
		for (auto s:x) {
			if (y.find(s) != y.end())continue;
			ret += s.length();
			ret++;
		}
		return ret;
	}
};
