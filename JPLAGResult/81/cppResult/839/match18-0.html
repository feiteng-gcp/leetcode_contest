class Solution {

public:
    int minimumLengthEncoding(vector<string>& words) {
        int result = 0;
        sort(words.begin(), words.end(), [] (string a, string b){ return a.size()>b.size(); });
        unordered_set<string> to_encode;
        for(auto w:words){
            auto it = to_encode.begin();
            bool found = false;
            while(it!=to_encode.end()){
                if((*it).find(w)!=string::npos){
                    if((*it).find(w)+w.size() == (*it).size()) {
                        found = true;
                        break;
                    }
                }
                it++;
            }
            if(!found){
                to_encode.insert(w);
                result += w.size()+1;
            }
        }
        return result;
    }
};
