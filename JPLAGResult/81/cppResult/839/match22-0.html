class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        unordered_set<string> s;
        for (string a : words) {
            s.insert(a);
        }
        for (string a : words) {
            int l = a.size();
            for (int i = 1; i < l; ++i) {
                string tmp = a.substr(i);
                if (s.count(tmp)) {
                    s.erase(tmp);
                }
            }
        }
        int n = s.size();
        int ans = n;
        for (string a : s) {
            ans += a.size();
        }
        return ans;
        
    }
};
