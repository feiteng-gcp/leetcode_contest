class TrieNode {
public:
    TrieNode* child[26];
    TrieNode(): child({NULL}){}
};
class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        sort(words.begin(), words.end(), [](string& a, string& b) -> bool { return a.size() > b.size(); });
        TrieNode h;
        int ret = 0;
        for (auto w: words) {
            ret += add(&h, w) ? w.size() + 1: 0;
        }
        return ret;
    }
    
    bool add(TrieNode* p, string& word) {
        bool isNew = false;
        for (int i = word.size() - 1; i >= 0; i--) {
            int ch = word[i] - 'a';
            if (p->child[ch] == NULL) {
                p->child[ch] = new TrieNode();
                isNew = true;
            }
            p = p->child[ch];
        }
        return isNew;
    }
};
