class Solution {
public:
    
    bool tail(const string& s, const string& t) {
        for (int i=t.size()-1; i>=0; --i)
            if (t[i]!=s[s.size()-(t.size()-i)])
                return false;
        return true;
    }
    
    int minimumLengthEncoding(vector<string>& a) {
        sort(a.begin(), a.end(), [](const string& x, const string& y) { return x.size() >y.size(); });
        vector<string> b;
        int len = 0;
        for (auto &s : a) {
            bool in = false;
            for (auto& t : b)
                if (tail(t, s)) {
                    in = true;
                    break;
                }
            if (!in) {
                b.push_back(s);
                len += s.size();
            }
        }
        return len + b.size();
    }
};
