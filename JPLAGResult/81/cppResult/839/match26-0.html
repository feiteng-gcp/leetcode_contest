class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        unordered_set<string> hash;
        sort(words.begin(), words.end(), [&](string a, string b) {return a.size() > b.size();});
        int ans = 0;
        auto insert = [&] (const string& str) {
            for (int i = 1; i <= str.size(); ++i) {
                hash.insert(str.substr(str.size() - i, i));
            }
        };
        for (auto &word : words) {
            if (ans == 0) { ans = 1+word.size(); insert(word); continue; }
            else if (hash.count(word) == 0) {
                ++ans;
                ans += word.size();
                insert(word);
            }
        }
        return ans;
    }
};
