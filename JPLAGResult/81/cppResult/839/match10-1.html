class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        unordered_set<string> covered;
        sort(rbegin(words), rend(words), [](const string& a, const string& b) { return a.size() < b.size(); });
        int size = 0;
        for (auto word : words) {
            if (covered.count(word) > 0) {
                continue;
            }
            size += word.size() + 1;
            for (size_t i = 0; i < word.size(); i++) {
                covered.insert(word.substr(i));
            }
        }
        return size;
    }
};
