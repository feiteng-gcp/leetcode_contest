struct Trie {
  Trie() {
      for (int i=0; i<26; ++i) children.push_back(nullptr);
  }    

  vector<Trie*> children;  
};


class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        // only when one is a suffix, you can save.
        int res = 0;
        Trie trie;
        sort(words.begin(), words.end(), [](string a, string b){
                return a.size() > b.size();
            });
        for (auto w : words) {
            if (!add(&trie, w)) res += w.size() + 1;
        }
        
        return res;
    }
    
    bool add(Trie* trie, string w) {
        int n = w.size();
        bool existing = true;
        Trie* cur = trie;
        for (int i=n-1; i>=0; --i) {
            auto ch = w[i];
            if (cur->children[ch-'a'] == nullptr) {
                existing = false;
                cur->children[ch - 'a'] = new Trie();
            }
            //cout << ch << ":" << existing << endl;
            cur = cur->children[ch-'a'];
        }
        //cout << "#####" << existing << endl;
        return existing;
    }
};
