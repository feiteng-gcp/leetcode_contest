class Solution {
private:
    struct TrieNode {
        vector<TrieNode*> ch = vector<TrieNode*>(26);
        bool isKey;
        TrieNode() {
            isKey = false;
        }
    };
    TrieNode* root;
public:
    bool insert(string a, int& deleLen) {
        TrieNode* temp = root;
        int addLen = 0;
        bool needAdd = false;
        for (int i = a.size() - 1; i >= 0; i--) {
            if (temp->ch[a[i] - 'a'] == NULL) {
                temp->ch[a[i] - 'a'] = new TrieNode();
                needAdd = true;
            }
            addLen++;
            temp = temp->ch[a[i] - 'a'];
            if (temp->isKey) {
                deleLen = addLen;
                temp->isKey = false;
            }
        }
        if (!needAdd) return false;
        temp->isKey = true;
        return true;
    }
    
    int minimumLengthEncoding(vector<string>& words) {
        root = new TrieNode();
        int result = 0;
        for (int i = 0; i < words.size(); i++) {
            int d = 0;
            if (insert(words[i], d)) {
                result = result + 1 + words[i].size();
                if (d != 0) result = result - 1 - d;
            }
        }
        return result;
    }
};
