class Solution {
public:
    int minimumLengthEncoding(vector<string>& w) {
        int n = w.size();
        unordered_set<int> s;
        for(int i = 0; i < n; ++i) {
            for(int j = i + 1; j < n; ++j) {
                if(w[i] == w[j]) {
                    s.insert(j);
                    continue;
                }
                if(ew(w[i], w[j])) s.insert(j);
                if(ew(w[j], w[i])) s.insert(i);
            }
        }
        

        int res = 0;
        for(int i = 0; i < n; ++i) {
            if(s.find(i) == s.end()) res += w[i].size() + 1;
        }
        return res;
    }
    
    bool ew(string& a, string& b) {
        if(a.size() < b.size()) return false;
        int i = a.size() - 1, j = b.size() - 1;
        while(j >= 0) {
            if(a[i--] != b[j--]) return false;
        }
        return true;
    }
};
