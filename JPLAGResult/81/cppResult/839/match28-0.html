bool less_than(string &a, string &b)
{
    return a.size()>b.size();
}
string reverse(string &a)
{
    string res = a;
    int i = 0,j = a.size()-1;
    while(i < j)
    {
        char t = res[i];
        res[i] = res[j];
        res[j] = t;
        i++;
        j--;
    }
    return res;
}
class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        int len = 0;
        set<string> setlist; 
        sort(words.begin(), words.end(), less_than);
        for(int i = 0; i < words.size(); i++)
        {
            if(setlist.empty()){
                setlist.insert(reverse(words[i]));
                len += words[i].size()+1;
            }
            else {
                bool flag = false;
                for(string a:setlist)
                {
                    string s = reverse(words[i]);
                    int k = 0;
                    for(; k < s.size(); k++)
                        if(a[k] != s[k])
                            break;
                    if(k == s.size()){
                        flag = true;
                        break;
                    }
                }
                if(flag == false)
                {
                    setlist.insert(reverse(words[i]));
                    len += words[i].size()+1;
                }
            }
        }
        return len;
        
        
    }
};
