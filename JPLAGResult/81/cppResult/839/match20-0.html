class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        unordered_map<int, vector<string>> m;
        for(int i=0; i<words.size();i++){
            string s = words[i];
            m[s.size()].push_back(s);
        }
        unordered_set<string> st;
        for(int i=1;i<=7;i++){
            if(m.find(i) == m.end()) continue;
            vector<string> vs = m[i];
            for(int j=0; j<vs.size();j++){
                if(st.find(vs[j]) == st.end())
                {
                    //cout<<" s "<<vs[j]<<endl;
                    st.insert(vs[j]);
                    for(int k=1; k<vs[j].size();k++){
                        string ss = vs[j].substr(k);
                        //cout<<"ss "<<ss<<endl;
                        if(st.find(ss) != st.end()){
                            st.erase(ss);
                        }
                    }
                }
            }
        }
        int count =  0;
        for(auto x : st){
            count += x.size()+1;
        }
        return count;
    }
};
