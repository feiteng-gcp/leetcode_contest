class Solution {
    
    struct _cmp_by_len {
        bool operator () (const string &a, const string &b) const {
            return a.size() > b.size();
        }
    } cmp_by_len;
    
    bool is_end_with(const string &a, const string &b) {
        int i = a.size();
        int j = b.size();
        while (i > 0 && j > 0) {
            if (a[i - 1] != b[j - 1]) return false;
            --i, --j;
        }
        return true;
    }
    
public:
    int minimumLengthEncoding(vector<string>& words) {
        int N = words.size();
        
        sort(words.begin(), words.end(), cmp_by_len);
        vector<bool> is_encode(N, true);
        
        for (int i = 0; i < N; ++i) {
            if (is_encode[i]) {
                for (int j = i + 1; j < N; ++j) {
                    if (is_encode[j] && is_end_with(words[i], words[j])) {
                        is_encode[j] = false;
                    }
                }
            }
        }
        
        int count = 0;
        for (int i = 0; i < N; ++i) {
            if (is_encode[i]) count += words[i].size() + 1;    
        }
        return count;
    }
};
