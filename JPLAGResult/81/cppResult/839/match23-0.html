class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        vector<bool> v;
        for(int i = 0; i < words.size(); i++) v.push_back(true);
        for(int i = 0; i < words.size(); i++) {
            for(int j = i+1; j < words.size(); j++) {
                int ptr1 = words[i].size()-1;
                int ptr2 = words[j].size()-1;
                while (true) {
                    if (ptr1 < 0) {
                        v[i] = false;
                        break;
                    }
                    else if (ptr2 < 0) {
                        v[j] = false;
                        break;
                    }
                    if (words[i][ptr1] != words[j][ptr2]) break;
                    ptr1--;
                    ptr2--;
                }
            }
        }
        int val = 0;
        for(int i = 0; i < words.size(); i++) {
            if (v[i]) val += words[i].length()+1;
            // cout<<val<<endl;
        }
        return val;
    }
};
