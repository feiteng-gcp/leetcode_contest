class Solution {
public:
    bool is_sub(string a, string b){
        int i=b.size()-1;
        int j=a.size()-1;
        while(i>=0&&j>=0){
            if(a[j]!=b[i])
                return false;
            else{
                i--;
                j--;
            }
        }
        return true;
    }
    void quick_sort(vector<string>& words, int s, int e){
        if(s>=e)
            return;
        int temp=words[e].size();
        int i=s,j=e-1;
        while(i<j){
            while(i<j&&words[i].size()<temp)
                i++;
            while(i<j&&words[j].size()>=temp)
                j--;
            swap(words[i], words[j]);
        }
        if(words[i].size()>=temp)
            swap(words[i], words[e]);
        else
            i++;
        quick_sort(words, s, i-1);
        quick_sort(words, i+1, e);
        return;
    }
    int minimumLengthEncoding(vector<string>& words) {
        int ret=0;
        quick_sort(words, 0, words.size()-1);
        int len=words.size()-1;
        ret+=words[len].size()+1;
        int visit=len;
        for(int i=len-1;i>=0;i--){
            bool sub=false;
            for(int j=len;j>=visit;j--){
                if(is_sub(words[i], words[j])){
                    sub=true;
                    break;
                }
            }
            if(sub==false){
                ret+=words[i].size()+1;
                visit--;
            }
        }
        return ret;
    }
};
