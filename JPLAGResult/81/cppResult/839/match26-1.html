class Trie{
public:
    Trie(){
        next.resize(26,nullptr);
    }
    Trie* add(char ch)
    {
        if(next[ch-'a']==nullptr)
        {
            next[ch-'a']=new Trie;
        }
        return next[ch-'a'];
    }
    vector<Trie*> next;    
};

class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        sort(words.begin(), words.end(), [](string& l, string& r){
            return l.size()>r.size()||l.size()==r.size()&&l<r;
        });
        int count=0;
        auto root= new Trie;
        for(string& word:words)
        {
            bool add=false;
            auto ptr=root;
            for(int i=word.size()-1; i>=0; i--)
            {
                char ch=word[i];
                if(ptr->next[ch-'a']==nullptr)
                {
                    count+=word.size()+1;
                    add=true;
                    break;
                }
                ptr=ptr->next[ch-'a'];
            }
            if(add)
            {
                ptr=root;
                for(int i=word.size()-1; i>=0; i--)
                {
                    char ch=word[i];
                    ptr=ptr->add(ch);
                }
            }
        }
        return count;
    }
};
