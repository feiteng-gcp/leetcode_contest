class Solution {
public:
    int minimumLengthEncoding(vector<string>& words) {
        {
            unordered_set<string> dedup(words.begin(), words.end());
            words = vector<string>(dedup.begin(), dedup.end());
            sort(words.begin(), words.end(), [](const string& s1, const string& s2) { return s1.length() < s2.length(); });
        }
        int sum = 0;
        for (int i = 0; i < words.size(); i++) {
            bool suf = false;

            for (int j = i + 1; j < words.size(); j++) {
                if (isSuffix(words[i], words[j])) {
                    suf = true;
                    break;
                }
            }
            if (!suf) {
                sum += 1 + words[i].length();
            }
        }
        return sum;
    }
private:
    bool isSuffix(const string& s, const string& l) {
        for (int i = 0; i < s.length(); i++) {
            if (s[s.length() - 1 - i] != l[l.length() - 1 - i]) {
                return false;
            }
        }
        return true;
    }
};
