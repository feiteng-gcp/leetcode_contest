class Solution {
public:
    vector<int> shortestToChar(string S, char C) {
        queue<int> que;
        set<int> used;
        vector<int> res(S.length(), 0);
        for(int i=0;i<S.length();i++){
            if(S[i]==C){
                que.push(i);
                used.insert(i);
            }
        }
        int index = 0; 
        while(!que.empty()){
            int s = que.size();
            while(s){
                s--;
                auto it = que.front();
                que.pop();
                res[it] = index;
                if(it+1<S.length()&&used.find(it+1)==used.end()){
                    que.push(it+1);
                    used.insert(it+1);
                }
                
                if(it-1>=0&&used.find(it-1)==used.end()){
                    que.push(it-1);
                    used.insert(it-1);
                }
                
            }
            index++;
        }
        return res;
    }
};
