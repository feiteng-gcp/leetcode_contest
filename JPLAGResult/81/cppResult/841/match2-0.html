class Solution {
public:
    vector<int> shortestToChar(string S, char C) {
        
        vector<int> res(S.size(), 12345);
        for (int i = 0; i < S.size(); i++) {
            if (S[i] == C) {
                res[i] = 0;
                for (int j = i - 1, count = 1; j >= 0 && res[j] > count; j--)
                    res[j] = count++;
                for (int j = i + 1, count = 1; j < S.size() && res[j] > count; j++)
                    res[j] = count++;
            }    
        }
        return res;
    }
};
