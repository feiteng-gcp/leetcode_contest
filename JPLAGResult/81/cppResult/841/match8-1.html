class Solution {
public:
    vector<int> shortestToChar(string S, char C) {

        vector<int> pos;
        vector<int> out;
        for(int i=0;i<S.length();i++){
            if(S[i]==C){
                pos.push_back(i);
            }
        }
        int p=0;
        int i,l;
        l=pos.size();
        for(i=0;i<S.length();i++){
            if(i>pos[p]){
                break;
            }
            out.push_back(pos[p]-i);
        }
        for(;i<S.length();i++){
            if(p<l-1){
                out.push_back((i-pos[p])>(pos[p+1]-i)?(pos[p+1]-i):(i-pos[p]));
                if(i>=pos[p+1]){
                    p++;
                }
            }else{
                break;
            }
        }
        for(;i<S.length();i++){
            out.push_back(i-pos[p]);
        }
        return out;
    }
};
