class Solution {
public:
    vector<int> shortestToChar(string S, char C) {

        int len = S.size(), count = 0, half;
        vector<int> ans(len);
        vector<int> pos;
        for (int i = 0; i < len; ++i){
            if (S[i] == C) pos.push_back(i);
        }
        int p = pos[0], p2;
        for (int i = 0; i <= p; ++i) ans[i] = p - i;
        for (int i = 1; i < pos.size(); ++i){
            p2 = pos[i];
            half = (p2 - p)/2 + p;
            for (int j = p + 1; j <= half; ++j) ans[j] = j - p;
            for (int j = half + 1; j <= p2; ++j) ans[j] = p2 - j;
            p = p2;
        }
        for (int i = p + 1; i < len; ++i) ans[i] = i - p;
        return ans;
    }
};
