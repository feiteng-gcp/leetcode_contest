class Solution {
public:
    vector<int> shortestToChar(string S, char C) {
        vector<int> res;
        vector<int> vec;
        for(int i=0; i<S.size();i++){
            if(S[i] == C) vec.push_back(i);
        }
        for(int i=0; i<S.size();i++){
            if(S[i] == C)
                res.push_back(0);
            else
            {
                int dist = INT_MAX;
                for(int j=0; j<vec.size();j++){
                    dist = min(dist, abs(vec[j] - i));
                }
                res.push_back(dist);
            }
        }
        return res;
    }
};
