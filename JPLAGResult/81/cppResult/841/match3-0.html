class Solution {
public:
    vector<int> shortestToChar(string S, char C) {
        vector<int> table;
        vector<int> res(S.size(),S.size());
        for(int i=0;i<S.size();++i) {
            if(S[i]==C) {
                table.push_back(i);
            }
        }
        int j=0;
        for(int i=0;i<S.size();++i) {
            if(j==table.size()-1) {
                res[i]=abs(table[j]-i);
            } else {
                if(i<=table[j]) {
                    res[i]=table[j]-i;
                } else if(i-table[j]<table[j+1]-i){
                    res[i]=i-table[j];
                } else {
                    j+=1;
                    res[i]=table[j]-i;
                }
            }
        }
        return res;
    }
};
