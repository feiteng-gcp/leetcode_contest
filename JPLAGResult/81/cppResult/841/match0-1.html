class Solution {
public:
    vector<int> shortestToChar(string S, char C) {
        set<int> s;
        for(int i=0; i<S.size(); ++i) {
            if (S[i] == C) {
                s.insert(i);
            }
        }
        s.insert(-10002); s.insert(30002);
        vector<int> res(S.size());
        for(int i=0; i<S.size(); ++i) {
            int p1 = *s.lower_bound(i) - i;
            int p2 =  i -  (*std::prev(s.lower_bound(i)));
            res[i] = min(p1, p2);
        }
        return res;
    }
};
