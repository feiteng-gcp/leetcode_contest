class Solution {
public:
    vector<int> shortestToChar(string S, char C) {
        int n = S.size();
        vector<int> result = vector<int>(n, 0);
        vector<int> pos;
        for (int i = 0; i < n; i++) {
            if (S[i] == C) pos.push_back(i);
        }
        int cur = 0;
        for (int i = 0; i < n; i++) {
            if (S[i] == C) {
                if (pos[cur] != i) cur++;
                continue;
            }
            else {
                if (cur + 1 < pos.size() && abs(pos[cur] - i) > abs(pos[cur + 1] - i))
                    cur++;
                result[i] = abs(pos[cur] - i);
            }
        }
        return result;
    }
};
