class Solution {
public:
    vector<int> shortestToChar(string S, char C) {
    vector<int> pos;
    pos.push_back(-20000);
    for(int i=0;i<S.size();++i){
        if(S[i]==C)
            pos.push_back(i);
    }
    pos.push_back(30000);
    int n=S.size();
    vector<int> ans(n);
    int p=1;
    for(int i=0;i<S.size();++i){
        while(i>=pos[p])
            ++p;
        ans[i]=min(i-pos[p-1],pos[p]-i);
    }
    return ans;
}
};
