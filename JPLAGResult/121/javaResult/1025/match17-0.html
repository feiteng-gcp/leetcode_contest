class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        if (days.length == 0) return 0;
        Map<Integer, Integer> seen = new HashMap<>();
        seen.put(days.length, 0);
        seen.put(days.length - 1, costs[0]);
        return mincostTickets(days, 0, costs, seen);
    }
    
    private int mincostTickets(int[] days, int i, int[] costs, Map<Integer, Integer> seen) {
        if (seen.containsKey(i)) return seen.get(i);
        
        // 1 day
        int min = costs[0] + mincostTickets(days, i + 1, costs, seen);
        // 7 day
        int next = i + 1;
        while (next < days.length && days[next] < days[i] + 7) next++;
        min = Math.min(min, costs[1] + mincostTickets(days, next, costs, seen));
        // 30 day
        while (next < days.length && days[next] < days[i] + 30) next++;
        min = Math.min(min, costs[2] + mincostTickets(days, next, costs, seen));
        
        seen.put(i, min);
        return min;
    }
}
