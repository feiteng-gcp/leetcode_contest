class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        boolean[] dayTripExists = new boolean[366];
        
        for (int day : days) {
            dayTripExists[day] = true;
        }
        
        int[] minCostDp = new int[366];
        
        for (int day = 1; day <= 365; day++) {
            if (!dayTripExists[day]) {
                minCostDp[day] = minCostDp[day - 1];
                continue;
            }
            
            int minCostUpToDay = minCostDp[day - 1] + costs[0];
            
            
if (day >= 7) {
                minCostUpToDay = Math.min(minCostUpToDay, minCostDp[day - 7] + costs[1]);
            } else {
                minCostUpToDay = Math.min(minCostUpToDay, costs[1]);
            }
            
            
if (day >= 30) {
                minCostUpToDay = Math.min(minCostUpToDay, minCostDp[day - 30] + costs[2]);
            } else {
                minCostUpToDay = Math.min(minCostUpToDay, costs[2]);
            }
            
            minCostDp[day] = minCostUpToDay;
        }
                
        return minCostDp[365];
    }
}
