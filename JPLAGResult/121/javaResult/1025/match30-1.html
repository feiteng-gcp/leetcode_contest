class Solution {
    private static final int DAYS_OF_YEAR = 365;

    public int mincostTickets(int[] days, int[] costs) {
        boolean[] hasTripArray = new boolean[DAYS_OF_YEAR + 1];
		for (int day : days)
		    hasTripArray[day] = true;
		int daysLength = days.length;
		int maxDay = days[daysLength - 1];
		int[] dp = new
 int[DAYS_OF_YEAR + 1];
		for (int i = 1; i <= DAYS_OF_YEAR; i++) {
		    if (!hasTripArray[i]) {
		        dp[i] = dp[i - 1];
		        continue;
		    }
		    int minCost;
		    minCost = dp[i - 1] + costs[0];
	        minCost = Math.min(minCost, dp[Math.max(0, i - 7)] + costs[1]);
		    minCost = Math.min((Math.min(minCost, dp[Math.max(0, i - 7)] + costs[1])), dp[Math.max(0, i - 30)] + costs[2]);
		    dp[i] = minCost;
		}
		return dp[maxDay];
    }
}
