class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        return help(days,costs,0,new Integer[days.length]);
    
}
    
    int help(int[] days, int[] costs, int cur, Integer[] cache) {
        if(cur == days.length) return 0;
        if(cache[cur] != null) return cache[cur];
        int result = help(days,costs,cur+1,cache) + costs[0];
        int index7 = cur;
        while(index7 < days.length && days[index7] - days[cur] < 7) index7++;
        result = Math.min(result,help(days,costs,index7,cache)+costs[1]);
        int index30 = cur;
        while(index30 < days.length && days[index30] - days[cur] < 30) index30++
;
        result = Math.min(result,help(days,costs,index30,cache)+costs[2]);
        cache[cur] = result;
        return result;
    }
}
