class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int[] visited = new int[days.length];
        return dfs(days, costs, 0, visited);
    }
    
    // we have NO ticket for cur day
    private int dfs(int[] days, int[] costs, int cur, int[] visited) {
        if (cur >= days.length) {
            return 0;
        }
        if (visited[cur] != 0) {
            return visited[cur];
        }
        int temp = Integer.MAX_VALUE;
        int curDay = days[cur];
        
        //one day pass
        temp = Math.min(temp, costs[0] + dfs(days, costs, cur + 1, visited));
        
        // 7-day pass
        int i = cur;
        while (i < days.length && days[i] < curDay + 7) {
            i++;
        }
        temp = Math.min(temp, costs[1] + dfs(days, costs, i, visited));
        
        // 30-day pass
        i = cur;
        while (i < days.length && days[i] < curDay + 30) {
            i++;
        }
        temp = Math.min(temp, costs[2] + dfs(days, costs, i, visited));
        
        visited[cur] = temp;
        return temp;
    }
}
