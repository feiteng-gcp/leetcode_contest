class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for(int i = 0; i < days.length; i++) {
            map.put(days[i], i);
        }
        int[] dp = new int[days.length + 1];
        for(int i = 1; i < dp.length; i++) {
            dp[i] = dp[i-1] + costs[0];
            
int bound7 = check(map, i, days, 7);
            int bound30 = check(map, i, days, 30);
            dp[i] = Math.min(dp[bound7] + costs[1], dp[i]);
            dp[i] = Math.min(dp[bound30] + costs[2], dp[i]);
            
        }
        return dp[days.length];
    }
    public int check(TreeMap<Integer, Integer> map, int pos, int[] days, int duration) {
            pos = pos - 1;
            if(days[pos] - duration < 0) return 0;
            Integer lower = map.floorKey(days[pos] - duration);
            if(lower == null) return 0;
            return map.get(lower) + 1;
    }
}
