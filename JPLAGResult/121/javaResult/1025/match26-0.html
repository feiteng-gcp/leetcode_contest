class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        TreeMap<Integer, Integer> treeMap = new TreeMap();
        int[] dp = new int[days.length];
        for (int i = 0; i < days.length; i++) {
            treeMap.put(days[i], i);
            dp[i] = Integer.MAX_VALUE;
        }
        for (int i = 0; i < days.length; i++) {
            int prevCost = i == 0 ? 0 : dp[i - 1];
            dp[i] = Math.min(dp[i], prevCost + costs[0]);
            int weekIdx = getIndex(treeMap, days[i] + 7 - 1);
            int monthIdx = getIndex(treeMap, days[i] + 30 - 1);
            dp[weekIdx] = Math.min(dp[weekIdx], prevCost + costs[1]);
            dp[monthIdx] = Math.min(dp[monthIdx], prevCost + costs[2]);
        }
        return dp[dp.length - 1];
    }
    private int getIndex(TreeMap<Integer, Integer> treeMap, int day) {
        int idx;
        if (treeMap.containsKey(day)) {
            idx = treeMap.get(day);
        } else {
            idx = treeMap.get(treeMap.lowerKey(day));
        } 
        return idx;
    }
}
