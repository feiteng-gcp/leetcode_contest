class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        /*
            dp[i] is the minimun cost up to ith day
            if day i in the plan
                dp[i] = min(dp[i - 1] + cost[0], dp[i - 7] + cost[1], dp[i - 30] + cost[2])
            else
                dp[i] = dp[i-1]
        */
        
        int[] dp = new int[365 + 31];
        Set<Integer> plan = new HashSet<Integer>();
        for (int day : days) {
            plan.add(day);
        }
         
        for (int i = 31; i <= 365 + 30; i++) {
            if (!plan.contains(i - 30)) {
                dp[i] = dp[i - 1];
                continue;
            }
            
            int min = dp[i - 1] + costs[0];
            min = Math.min(min, dp[i - 7] + costs[1]);
            min = Math.min(min, dp[i - 30] + costs[2]);
            
            dp[i] = min;
        }
        
        return dp[365 + 30];
    }
}
