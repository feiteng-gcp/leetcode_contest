    public int mincostTickets(int[] days, int[] costs) {
        int len = days.length;
        if ( len == 0 ) return 0;
        // int[] dp = new int[len + 1];
        // int[][] dp = new int[len + 1][3];
        TreeMap<Integer, Integer> map = new TreeMap<>();
        map.put(0, 0);
        for ( int i = 1; i <= len; i++ ){
            int oneday = days[i-1] - 1;
            int sevendays = days[i-1] - 7;
            int thirtydays = days[i-1] - 30;
            Integer onedayKey = map.floorKey(oneday);
            int onedayValue, sevendayValue, thirtydayValue;
            if ( onedayKey == null ) onedayValue = 0;
            else onedayValue = map.get(onedayKey);
            Integer sevendayKey = map.floorKey(sevendays);
            if ( sevendayKey == null ) sevendayValue = 0;
            else sevendayValue = map.get(sevendayKey);
            Integer thirtydayKey = map.floorKey(thirtydays);
            if ( thirtydayKey == null ) thirtydayValue = 0;
            else thirtydayValue = map.get(thirtydayKey);    
            int value = Math.min(onedayValue + costs[0], Math.min(sevendayValue + costs[1], thirtydayValue + costs[2]));
            map.put(days[i-1], value);
        }
        return map.get(days[len - 1]);
        // return 0;
        // ticket t = map.get(days[len-1]);
        // return Math.min(t.one, Math.min(t.seven, t.thirty));
    }
}
