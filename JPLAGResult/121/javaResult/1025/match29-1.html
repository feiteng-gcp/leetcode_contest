class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int[] dp=new int[days.length];
        Arrays.fill(dp,Integer.MAX_VALUE);
        
        for(int i=0;i<dp.length;i++){
            int pastCost=i==0?0:dp[i-1];
            for(int j=i;j<dp.length&&days[i]+1>days[j];j++){
                dp[j]=Math.min(dp[j],pastCost+costs[0]);
            }
            for(int j=i;j<dp.length&&days[i]+7>days[j];j++){
                dp[j]=Math.min(dp[j],pastCost+costs[1]);
            }
            for(int j=i;j<dp.length&&days[i]+30>days[j];j++)
{
                dp[j]=Math.min(dp[j],pastCost+costs[2]);
            }
        }
        
        return dp[dp.length-1];
    }
}
