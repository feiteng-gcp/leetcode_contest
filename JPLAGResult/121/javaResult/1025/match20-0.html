class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int[] dp = new int[366];
        int dayIter = 0;
        int n = days.length;
        dp[0] = 0;
        for(int i = 1; i <= 365 ; i++)
        {
            if(dayIter < n && i == days[dayIter])
            {
                dp[i] = dp[i - 1] + costs[0];
   
                    for(int j = i - 1; j >= i - 7 && j >= 0; j--)
                    {
                        dp[i] = Math.min(dp[i], dp[j] + costs[1]);
                    }
            
                for(int j = i - 1; j >= i - 30 && j >= 0; j--)
                    {
                        dp[i] = Math.min(dp[i], dp[j] + costs[2]);
                    }
                
                dayIter++;
            }
            else
            {
                dp[i] = dp[i - 1];
            }
        }
        return dp[365];
    }
}
