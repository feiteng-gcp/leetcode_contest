class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int[] dp = new int[days.length];
        dp[0] = costs[0];
        for(int i = 1; i < days.length; i++) {
            // try day 1
            dp[i] = dp[i-1] + costs[0];
            // try day 7
            int idx = Arrays.binarySearch(days, days[i] - 6);
            if(idx < 0)
                idx = -(idx+1);
            dp[i] = Math.min(dp[i], idx-1 >= 0 ? dp[idx-1] + costs[1]: costs[1]);;
            // try day 30
            idx = Arrays.binarySearch(days, days[i] - 29);
            if(idx < 0)
                idx = -(idx+1);
            dp[i] = Math.min(dp[i], idx-1 >= 0 ? dp[idx-1] + costs[2]: costs[2]);
        }
        return dp[dp.length-1];
    }
}
