class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int[] dp = new int[days.length];
        Arrays.fill(dp, Integer.MAX_VALUE);
        dp[0] = costs[0];
        Map<Integer, Integer> map = new TreeMap<>();
        map.put(days[0], 0);
        int[] gaps = new int[]{1,7,30};
        for(int i = 1; i < days.length; i++) {
            for(int j = 0; j < gaps.length; j++) {
                int lastKey = findKey(map, days[i], gaps[j]);
                if(lastKey != -1) {
                    dp[i] = Math.min(((lastKey > 0 ? dp[lastKey - 1] : 0) + costs[j]), dp[i]);
                }
                else {
                    dp[i] = dp[i-1] + costs[0];
                }
            }
            
map.put(days[i], i);
        }
        
        return dp[days.length - 1];
    }
    
    private int findKey(Map<Integer, Integer> map, int key, int gap) {
        for(int i = key - gap + 1; i < key; i++) {
            if(map.containsKey(i)) {
                return map.get(i);
            }
        }
        return -1;
    }
}
