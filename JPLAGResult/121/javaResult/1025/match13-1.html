class Solution {
    private Map<Integer, Integer> dMap;
    private Map<Integer, Integer> wMap;
    private Map<Integer, Integer> mMap;

    public int mincostTickets(int[] days, int[] costs) {
        dMap = new HashMap<>();
        wMap = new HashMap<>();
        mMap = new HashMap<>();
        return helper(days, 0, 0, costs);
    }


    private int helper(int[] days, int free, int i, int[] costs) {
        if (i >= days.length) {
            return 0;
        }
        int d = days[i];
        if (d < free) {
            return helper(days, free, i + 1, costs);
        }
        int res = Integer.MAX_VALUE;

        int dCost, wCost, mCost;
        if(dMap.containsKey(i))
{
            dCost = dMap.get(i);
        }
        else {
            dCost = costs[0] + helper(days, d, i + 1, costs);
        }

        if(wMap.containsKey(i))
{
            wCost = wMap.get(i);
        }
        else{
            wCost = costs[1] + helper(days, d + 7, i + 1, costs);
        }

        if(mMap.containsKey(i)){
            mCost = mMap.get(i);
        }
        else{
            mCost = costs[2] + helper(days, d + 30, i + 1, costs);
        }
        dMap.put(i, dCost);
        wMap.put(i, wCost);
        mMap.put(i, mCost);
        return Math.min(dCost, Math.min(wCost, mCost));
    }
}
