class Solution {
    
public int mincostTickets(int[] days, int[] costs) {
        int[] dp = new int[400];
        int[] d = new int[400];
        for (int i = 1; i < 400; i++) {
            dp[i] = 1000000;
        }
        for (int day : days) {
            d[day] = 1;
        }
        for (int i = 0; i < 365; i++) {
            if (d[i+1]==0)
                dp[i+1] = dp[i];
            else {
                dp[i+1] = Math.min(dp[i+1], dp[i]+costs[0]);
                for (int j = 1; j <= 7; j++)
                    dp[i+j] = Math.min(dp[i+j], dp[i]+costs[1]);
                for (int j = 1; j <= 30; j++)
                    dp[i+j] = Math.min(dp[i+j], dp[i]+costs[2]);
            }
        }
        return dp[365];
    }
}
