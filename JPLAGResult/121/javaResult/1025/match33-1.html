class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int i=0; i<days.length; i++) {
            int day = days[i];
            Integer prev1 = map.floorKey(day-1), prev7 = map.floorKey(day-7), prev30 = map.floorKey(day-30);
            int min1 = prev1 == null ? 0 : map.get(prev1), min7 = prev7 == null ? 0 : map.get(prev7), min30 = prev30 == null ? 0 : map.get(prev30);
            
int min = Math.min(min1+costs[0], Math.min(min7+costs[1], min30+costs[2]));
            map.put(day, min);
        }
        return map.get(days[days.length-1]);
    }
}
