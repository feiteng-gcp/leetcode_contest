class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int len = days.length;
        int[] totalCosts = new int[len+1];
        for(int i=0; i<len; i++){
            totalCosts[i+1] = totalCosts[i]+costs[0];
            int weekStart = findStart(days, i, 7);
            totalCosts[i+1] = Math.min(totalCosts[weekStart+1]+costs[1], totalCosts[i+1]);
            int monthStart = findStart(days, i, 30);
            totalCosts[i+1] = Math.min(totalCosts[monthStart+1]+costs[2], totalCosts[i+1]);
        }
        
        return totalCosts[len];
    }
    
    private int findStart(int[] days, int dayIndex, int range){
        for(int i=dayIndex; i>=0; i--){
            if(days[dayIndex]-days[i] >= range) return i;
        }
        
        return -1;
    }
}
