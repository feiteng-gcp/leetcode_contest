class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int[] dp = new int[days[days.length-1]+1];
        Set<Integer> set = new HashSet<>();
        for(int day : days) set.add(day);
        dp[days[0]] = costs[0];
        for(int i = days[0]+1; i < dp.length; i++) {
            // on day i
            if(set.contains(i)) {
                dp[i] = dp[i-1]+costs[0];
                dp[i] = Math.min(dp[i], dp[i-7 >= 0 ? i-7 : 0] + costs[1]);
                dp[i] = Math.min(dp[i], dp[i-30 >= 0 ? i-30 : 0] + costs[2]);
            
} else {
                dp[i] = dp[i-1];
            }
        }
        return dp[dp.length - 1];
    }
}
