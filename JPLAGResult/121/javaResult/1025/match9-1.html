class Solution {
    int M = 365;
    public int mincostTickets(int[] days, int[] costs) {
        int[] ans = new int[M + 1];
        
boolean[] f = new boolean[M + 1];
        for (int d : days) {
            f[d] = true;
        }
        for (int i = 1; i <= M; i++) {
            if (!f[i]) {
                ans[i] = ans[i - 1];
            } else {
                ans[i] = ans[i - 1] + costs[0];
            }
            if (i >= 7) {
                ans[i] = Math.min(ans[i], ans[i - 7] + costs[1]);
            }
            if (i >= 30) {
                ans[i] = Math.min(ans[i], ans[i - 30] + costs[2]);
            }
        }
        return ans[M];
    }
}
