class Solution {
    int[] dc = new int[366];
    int ans = Integer.MAX_VALUE;
    int[] ct;
    public int mincostTickets(int[] days, int[] costs) {
        ct = costs;
        for (int i = 0; i <= 365; i++) {
            dc[i] = Integer.MAX_VALUE;
        }
        dfs(days, 0, 0);
        return ans;
    }
    void dfs(int[] days, int index, int tpcost) {
        if (dc[index] < tpcost) {
            return;
        }
        dc[index] = tpcost;
        if (index == days.length) {
            ans = Math.min(ans, tpcost);
            return;
        }
        dfs(days, findFirBiger(days, days[index] + 1 - 1), tpcost + ct[0]);
        dfs(days, findFirBiger(days, days[index] + 7 - 1), tpcost + ct[1]);
        dfs(days, findFirBiger(days, days[index] + 30 - 1), tpcost + ct[2]);
    }
    int findFirBiger(int[] days, int target){
        int  left = 0, right = days.length;
        while (left < right) {
            int mid = (left + right)/2;
            if (days[mid] <= target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left;
    }
}
