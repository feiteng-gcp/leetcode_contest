class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        boolean[] travel = new boolean[366];
        for(int day : days) travel[day] = true;
        int[] dp = new int[366];
        for(int i = 1; i < 366; i++) {
            if(travel[i]) {
                dp[i] = dp[i-1] + costs[0];
                
if(i >= 7) {
                    dp[i] = Math.min(dp[i-7] + costs[1], dp[i]);
                } else {
                    dp[i] = Math.min(dp[0] + costs[1], dp[i]);
                }
                
if(i >= 30) {
                    dp[i] = Math.min(dp[i-30] + costs[2], dp[i]);
                } else {
                    dp[i] = Math.min(dp[0] + costs[2], dp[i]);
                }
            } else {
                dp[i] = dp[i-1];
            }
        }
        return dp[365];
    }
}
