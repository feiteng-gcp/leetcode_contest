class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int n = days.length;
        int[] memo = new int[n];
        Arrays.fill(memo, Integer.MAX_VALUE);
        memo[0] = costs[0];
        int j = 1;
        while (j < n && days[j] - days[0] <= 6) {
            memo[j] = Math.min(memo[j], costs[1]);
            j++;
        }
        while (j < n && days[j] - days[0] <= 29) {
            memo[j] = Math.min(memo[j], costs[2]);
            j++;
        }
        
        for (int i = 1; i < n; i++) {
            memo[i] = Math.min(memo[i], costs[0] + memo[i - 1]);
            j = i + 1;
            for (; j < n && days[j] - days[i] < 7; j++) {
                memo[j] = Math.min(memo[j], costs[1] + memo[i - 1]);
            }
            for (; j < n && days[j] - days[i] < 30; j++) {
                memo[j] = Math.min(memo[j], costs[2] + memo[i - 1]);
            }
        }
        // System.out.println();
        return memo[n - 1];
    }
}
