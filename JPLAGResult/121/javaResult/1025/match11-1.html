class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        int[] dp = new int[366];

        dp[1] = days[0] == 1 ? costs[0] : 0;

        int dayIndex = days[0] == 1 ? 1 : 0;
        for (int i = 2; i < dp.length; i++) {
            if (dayIndex >= days.length || i < days[dayIndex]) {
                dp[i] = dp[i - 1];
                continue;
            }

            int minCost = costs[0] + dp[i - 1];
            
if (i > 7) {
                minCost = Math.min(minCost, costs[1] + dp[i - 7]);
            } else {
                minCost = Math.min(minCost, costs[1]);
            }
            
if (i > 30) {
                minCost = Math.min(minCost, costs[2] + dp[i - 30]);
            } else {
                minCost = Math.min(minCost, costs[2]);
            }
            dp[i] = minCost;

            dayIndex += 1;
        }

        return dp[365];
    }
}
