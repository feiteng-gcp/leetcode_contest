class Solution {
    public int mincostTickets(int[] days, int[] costs) {
        //365*1000*60
        int[] f = new int[days.length];
        Arrays.fill(f, Integer.MAX_VALUE);
        int[] ds = new int[]{1, 7, 30};
        
        for (int i = 0; i < days.length; i++) {
            int org;
            if (i == 0) org = f[i] == Integer.MAX_VALUE ? 0 : f[i];
            else org = f[i-1];
            for (int j = 0; j < 3; j++) {
                int jump = ds[j];
                
                int t = days[i] + jump-1;
                int l = i, r = days.length-1;
                
                while (l < r) {
                    int mid = (l + r+1)>>1;
                    if (days[mid] <= t) l = mid;
                    else r = mid-1;
                }
                //if (i == 1) System.out.println("*****"+l);
                //if (l == i && f[i] != Integer.MAX_VALUE) continue;
                for (int k = i; k <= l; k++) {
                    
                    if (k != 0) f[k] = Math.min(f[k], org + costs[j]);
                    else {
                        if (f[k] == Integer.MAX_VALUE) f[k] = costs[j];
                        else f[k] = Math.min(f[k], costs[j]);
                    }
                }
            }
            //System.out.println(f[i]);
        }
        return f[days.length-1];
    }
}
