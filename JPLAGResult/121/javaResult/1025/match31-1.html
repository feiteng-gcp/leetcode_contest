class Solution {
	int []dp = new int[380];

	public int mincostTickets(int[] days, int[] costs) {
		for (int i = 0; i < days.length; i++) dp[i] = -1;
		return solve(0, days, costs);
	}

	private int solve(int ind
ex, int[] days, int[] costs) {
		if (index == days.length) return 0;
		if (dp[index] != -1) return dp[index];
		int ans = solve(index + 1, days , costs) + costs[0];
		boolean c7 = true;
		boolean c30 = true;
		for (int i = index + 1; i < days.length; i++) {
			if (c7 && (days[i] - days[index] >= 7)) {
				c7 = false;
				ans = Math.min(ans, solve(i, days, costs) + costs[1]);
			}
			if (c30 && (days[i] - days[index] >= 30)) {
				c30 = false;
				ans = Math.min(ans, solve(i, days, costs) + costs[2]);
			}
		}
		if (c7) {
			ans = Math.min(ans, solve(days.length, days, costs) + costs[1]);
		}
		if (c30) {
			ans = Math.min(ans, solve(days.length, days, costs) + costs[2]);
		}
		return dp[index] = ans;
	}
}
