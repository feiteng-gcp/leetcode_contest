class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        if (A > B) {
            while (A > 0) {
                if (A - B >= 3) {
                    sb.append('a');
                    sb.append('a');
                    sb.append('b');
                    A -= 2;
                    B -= 1;
                } else {
                    A--;
                    sb.append('a');
                    if (B > 0) {
                        sb.append('b');
                        B--;
                    }
                }
                
            }
        } else{
            while (B > 0) {
                if (B - A >= 3) {
                    sb.append('b');
                    sb.append('b');
                    sb.append('a');
                    A -= 1;
                    B -= 2;
                } else {
                    B--;
                    sb.append('b');
                    if (A > 0) {
                        sb.append('a');
                        A--;
                    }
                }
                
            }
        }
        return sb.toString();
    }
}
