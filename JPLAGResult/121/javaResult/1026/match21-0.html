class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        
        while (A > B + 1 && A >= 2 && B > 0) {
            sb.append('a');
            sb.append('a');
            sb.append('b');
            A -= 2;
            B--;
        }
        
        while (B > A + 1 && B >= 2 && A > 0) {
            sb.append('b');
            sb.append('b');
            sb.append('a');
            B -= 2;
            A--;
        }
        
        while (A > 0 || B > 0) {
            if (A > B && A > 0) {
                sb.append('a');
                A--;
            } else if (B > A && B > 0) {
                sb.append('b');
                B--;
            } else if (A == B && A > 0 && B > 0)  {
                sb.append('b');
                B--;
                sb.append('a');
                A--;
            }
        }
        
        return sb.toString();
    }
}
