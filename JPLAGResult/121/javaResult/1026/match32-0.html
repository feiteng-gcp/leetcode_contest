class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb=new StringBuilder();
        int max=Math.max(A,B);
        int min=Math.min(A,B);
        char char1=A>=B?'a':'b';
        char char2=char1=='a'?'b':'a';
        while(sb.length()<A+B){
            if(max>min+1){
                sb.append(char1);
                max--;
                sb.append(char1);
                max--;
            }
            else{
                sb.append(char1);
                max--;
            }
            if(min>0){
                min--;
                sb.append(char2);
            }
        }
        return sb.toString();
    }
}
