class Solution {

	private String helper(int n1, int n2,
 char[] chars) {
		StringBuilder stringBuilder = new StringBuilder();
		while (n1 > 0 && n2 > 0) {
			if (n2 > n1) {
				stringBuilder.append(chars[1]);
				stringBuilder.append(chars[1]);
				n2 -= 2;
				stringBuilder.append(chars[0]);
				n1--;
			} else {
				stringBuilder.append(chars[1]);
				n2 --;
				stringBuilder.append(chars[0]);
				n1--;
			}

		}
		while (n1-- > 0) {
			stringBuilder.append(chars[0]);
		}
		while (n2-- > 0) {
			stringBuilder.append(chars[1]);
		}
		return stringBuilder.toString();
	}

	public String strWithout3a3b(int A, int B) {
		if (A < B) {
			return helper(A, B, new char[]{'a', 'b'});
		} else {
			return helper(B, A, new char[]{'b', 'a'});
		}
	}
}
