class Solution {
public String strWithout3a3b(int A, int B) {
		StringBuilder sb = new StringBuilder();
		while (A > 0 || B > 0) {
			if (A > B) {
				if (A >= 2) {
					sb.append("aa");
					A -= 2;
				} else {
					sb.append("a");
					A -= 1;
				}
				if (B > 0) {
					sb.append("b");
					B -= 1;
				}
			} else if (A == B) {
				sb.append("ab");
				A -= 1;
				B -= 1;
			} else {
				if (B >= 2) {
					sb.append("bb");
					B -= 2;
				} else {
					sb .append("b");
					B -= 1;
				}
				if (A > 0) {
					sb.append("a");
					A -= 1;
				}
			}

		}
		return sb.toString();
	}
}
