class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        int cntA = 0, cntB = 0;
        while(A > 0 || B >0) {
            if(A >= B) {
                if(cntA >=2) {
                    cntA = 0;
                    cntB++;
                    sb.append('b');
                    B--;
                }else {
                    sb.append('a');
                    cntB = 0;
                    cntA++;
                    A--;
                }
            }else {
                if(cntB >=2) {
                    cntB = 0;
                    cntA ++;
                    sb.append('a');
                    A--;
                }else {
                    cntA = 0;
                    cntB++;
                    sb.append('b');
                    B--;
                }
            }
        }
        return sb.toString();
    }
}
