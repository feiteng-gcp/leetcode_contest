class Solution {
    public String strWithout3a3b(int A, int B) {
        if(A==B){
            String res = "";
            for(int i = 0; i < A; i++){
                res += "ab";
            }
            return res;
        }
        return A > B ? helper('a', 'b', A, B) : helper('b', 'a', B, A);
    }

    private String helper(char x, char y, int p, int q) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < p; i++) {
            sb.append(x);
        }

        int i = 0;
        while (q > 0) {
            int n = 2 + i* 3;
            if(n > sb.length()){
                break;
            }
            sb.insert(n, y);
            i++;
            q--;
        }
        i = 0;
        while (q > 0) {
            int n = 3 + i* 4;
            if(n > sb.length()){
                break;
            }
            sb.insert(n, y);
            i++;
            q--;
        }
        return sb.toString();
    }
}
