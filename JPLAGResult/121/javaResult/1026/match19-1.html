class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        boolean firstA = A > B;
        while (A != B && A > 0 && B > 0) {
            if (A > B && A >= 2) 
{
                A -= 2;
                sb.append("aa");
                if (B >= 1) {
                    B--;
                    sb.append('b');
                }
            }
            if (A < B && B >= 2) 
{
                B -= 2;
                sb.append("bb");
                if (A >= 1) {
                    A--;
                    sb.append('a');
                }
            }
        }
        if (A == B) {
            for (int i = 0; i < A; i++) {
                if (firstA) {
                    sb.append("ab");
                } else {
                    sb.append("ba");
                }
            }
        } else {
            for (int i = 0; i < Math.max(A, B); i++) {
                if (A > 0) {
                    sb.append('a');
                } else {
                    sb.append('b');
                }
            }
        }
        
        return sb.toString();
    }
}
