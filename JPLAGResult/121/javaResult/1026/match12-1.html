class Solution {
    public String strWithout3a3b(int A, int B) {        
        StringBuffer sb = new StringBuffer("");
        while(A>0 && B>0) {
            if(A > B) 
{
                if(A > 1) {
                    sb.append("aa");
                    A -= 2;
                } else if(A == 1) {
                    sb.append("a");
                    A--;
                }
                
                if(B > 0) {
                    sb.append("b");
                    B--;
                }
            } else if(B > A) {
                if(B > 1) {
                    sb.append("bb");
                    B -= 2;
                } else if(B == 1) {
                    sb.append("b");
                    B--;
                }
                
                if(A > 0) {
                    sb.append("a");
                    A--;
                }
            } else {
                sb.append("ab");
                A--;
                B--;
            }
        }
        
        while(A > 0) {
            sb.append("a");
            A--;
        }
        while(B > 0) {
            sb.append("b");
            B--;
        }
        return sb.toString();
    }
}
