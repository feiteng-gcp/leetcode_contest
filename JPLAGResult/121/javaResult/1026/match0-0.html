class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        boolean f = A > B;
        while (A > 0 && B > 0) {
            if (A > B) {
                sb.append("aab");
                A -= 2;
                B -= 1;
            } else if (A == B) {
                if (f) {
                    sb.append("ab");
                } else {
                    sb.append("ba");
                }
                A -= 1;
                B -= 1;
            } else {
                sb.append("bba");
                B -= 2;
                A -= 1;
            }
        }
        while (A > 0) {
            sb.append("a");
            A--;
        }
        while (B > 0) {
            sb.append("b");
            B--;
        }
        String res = sb.toString();
        return res;
    }
}
