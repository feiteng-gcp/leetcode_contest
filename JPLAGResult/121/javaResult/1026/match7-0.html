class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        while(A > B && B > 0) {
            sb.append("aa");
            sb.append('b');
            A -= 2;
            B -= 1;
        }
        while(A < B && A > 0) {
            sb.append("bb");
            sb.append('a');
            A -= 1;
            B -= 2;
        }
        if(A == 0) {
            for(int i = 0; i < B; i++) sb.append('b');
        }
        if(B == 0) {
            for(int i = 0; i < A; i++) sb.append('a');
        }
        while(A == B && A > 0) {
            sb.append("ab");
            A--;
            B--;
        }
        return sb.toString();
    }
}
