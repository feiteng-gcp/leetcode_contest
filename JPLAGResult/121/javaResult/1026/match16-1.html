class Solution {
    public String strWithout3a3b(int A, int B) {
        String ans = "";
        while (A>0&&B>0) {
            if (A>B) {
                ans += "aab";
                A-=2;
                B--;
            } else if (B>A) {
                ans += "bba";
                B-=2;
                A--;
            } else {
                ans += "ab";
                A--;
                B--;
            }
        
}
        while (A>0) {
            ans += "a";
            A--;
        }
        while (B>0) {
            ans += "b";
            B--;
        }
        return ans;
    }
}
