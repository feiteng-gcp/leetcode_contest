class Solution {
    public String strWithout3a3b(int A, int B) {
        String str = "";
        int difference = Math.abs(A - B);
        if (difference < 3) {
            if (A >= B) {
                while (B > 0) {
                    str += "ab";
                    A--;
                    B--;
                }
                while (A > 0) {
                    str += "a";
                    A--;
                }
            } else {
                while (A > 0) {
                    str += "ba";
                    A--;
                    B--;
                }
                while (B > 0) {
                    str += "b";
                    B--;
                }
            }
        } else {
            if (A > B) {
                while (A > B && B > 0) {
                    if (A >= 2) {
                        str += "aab";
                        A -= 2;
                        B--;
                    } else {
                        while (A > 0) {
                            str += "a";
                            A--;
                        }
                        str += "b";
                        B--;
                    }
                }
            } else {
                while (B > A && A > 0) {
                    if (B >= 2) {
                        str += "bba";
                        B -= 2;
                        A--;
                    } else {
                        while (B > 0) {
                            str += "b";
                            B--;
                        }
                        str += "a";
                        A--;
                    }
                }
            }
        }
        char lastChar = str.charAt(str.length() - 1);
        String next = (lastChar == 'a' ? "ba" : "ab");
        while (A > 0 && B > 0) {
            str += next;
            A--;
            B--;
        }
        while (A > 0) {
            str += "a";
            A--;
        }
        while (B > 0) {
            str += "b";
            B--;
        }
        return str;
    }
}
