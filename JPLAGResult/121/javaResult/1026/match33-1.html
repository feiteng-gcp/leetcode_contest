class Solution {
        public String strWithout3a3b(int A, int B) {
        String result = "";
        if (A > B) {
            while (A > B && B>0) {
                result += "aab";
                A -= 2;
                B -= 1;
            
}
            while(B>0 && A>0){
                result+="ab";
                B--; A--;
            }
            while (B > 0) {
                result += "b";
                B--;
            }
            while (A > 0) {
                result += "a";
                A--;
            }
            

        } else if(B>A) {
            while (B > A && A>0) {
                result += "bba";
                B -= 2;
                A -= 1;
            }
            while(B>0 &&A>0){
                result+="ab";
                B--; A--;
            }
            while (B > 0) {
                result += "b";
                B--;
            }
            while (A > 0) {
                result += "a";
                A--;
            }
        }else{
            while(B>0){
                result+="ab";
                B--;
            }
        }


        return result;
    }
}
