class Solution {
    public String strWithout3a3b(int A, int B) {
        if (A == 0) {
            if (B == 0) return "";
            if (B == 1) return "b";
            if (B == 2) return "bb";
        } else if (B == 0) {
            if (A == 1) return "a";
            if (A == 2) return "aa";
        }
        
        if (A > B) {
            return "aab" + strWithout3a3b(A - 2, B - 1);
        } else if (B > A) {
            return "bba" + strWithout3a3b(A - 1, B - 2);
        } else {
            return "ab" + strWithout3a3b(A - 1, B - 1);
        }
    }
}
