class Solution {
    public String strWithout3a3b(int A, int B) {
        int n = A;
        int m = B;
        char cMax = 'a';
        char cMin = 'b';
        if (A < B) {
            n = B;
            m = A;
            cMax = 'b';
            cMin = 'a';
        }
        String ans = "";
        while (n > 0 && m > 0) {
            ans += cMax;
            if (n - m >= 3) {
                ans += cMax;
                n --;
            }
            ans += cMin;
            n--;
            m--;
            
        }
        while (n-- > 0) {
            ans += cMax; 
        }
        while (m-- > 0) {
            ans += cMin;
        }
        return ans;
    }
}
