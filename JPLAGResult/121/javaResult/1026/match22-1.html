class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        if (A >= B) {
            while (A > 1 && A > B) {
                sb.append("aa");
                A -= 2;
                if (B-- > 0) sb.append('b');
            }
            if (A < B) sb.append('b');
            while (A-- > 0) {
                sb.append("a");
                if (B-- > 0) sb.append('b');
            }
        } else {
            while (B > 1 && B > A) {
                sb.append("bb");
                B -= 2;
                if (A-- > 0) sb.append('a');
            }
            if (B < A) sb.append('a');
            while (B-- > 0) {
                sb.append("b");
                if (A-- > 0) sb.append('a')
;
            }
        }
        return sb.toString();
    }
}
