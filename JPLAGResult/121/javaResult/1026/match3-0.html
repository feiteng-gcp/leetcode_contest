class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        int aRemain = A;
        int bRemain = B;
        while(aRemain != 0 && bRemain != 0)
        {
            if(aRemain < bRemain)
            {
                sb.append("bba");
                bRemain -= 2;
                aRemain -= 1;
            }
            else if(aRemain > bRemain)
            {
                sb.append("aab");
                aRemain -= 2;
                bRemain -= 1;
            }
            else
            {
                sb.append("ab");
                aRemain--;
                bRemain--;
            }
        }
        while(aRemain > 0)
        {
            sb.append("a");
            aRemain--;
        }
        while(bRemain > 0)
        {
            sb.append("b");
            bRemain--;
        }
        return sb.toString();
    }
}
