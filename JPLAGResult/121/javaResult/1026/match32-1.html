class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder sb = new StringBuilder();
        StringBuilder first = new StringBuilder(), second = new StringBuilder();
        if (A > B) {
           for (int i = 0; i < A; i++) {
                first.append('a');
            }
            while (A > 0) {
                A --;
                sb.append('a');
                if (A > 0) 
{    
                    sb.append('a');
                    A --;
                }
                if (B > 0) {
                    sb.append('b');
                    B--;
                }
            }
            int index = 0;
            while (B > 0) {
                int pos = sb.toString().indexOf("aa", index);
                sb.insert(pos, 'b');
                index = pos + 2;
                B --;
            }
        } else {
           for (int i = 0; i < B; i++) {
                first.append('b');
            }
            while (B > 0) {
                B --;
                sb.append('b');
                if (B > 0) {    
                    sb.append('b');
                    B --;
                }
                if (A > 0) {
                    sb.append('a');
                    A--;
                }
            }
            int index = 0;
            while (A > 0) {
                int pos = sb.toString().indexOf("bb", index);
                sb.insert(pos, 'a');
                index = pos + 2;
                A --;
            }
        }
        return sb.toString();
    }
}
