class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuilder b = new StringBuilder();
        int ra = 0, rb = 0;
        while (A > 0 || B > 0) {
            if (A >= B && ra < 2) {
                b.append('a');
                ra++;
                A--;
                rb = 0;
                continue;
            }
            if (B >= A && rb < 2) {
                ra = 0;
                b.append('b');
                B--;
                rb++;
                continue;
            }
            if (ra > 0) {
                ra = 0;
                rb = 0;
                b.append('b');
                rb++;
                B--;
                continue;
            }
            if (rb > 0) {
                rb = 0;
                ra = 0;
                b.append('a');
                ra++;
                A--;
                continue;
            
}
        }
        return b.toString();
    }
}
