class Solution {
    public String strWithout3a3b(int A, int B) {
        StringBuffer sb = new StringBuffer();
        // handle A is large than B
        if(A > B) {
            while(A > 0 && B > 0 && A - 1 > B) {
                sb.append("aa");
                sb.append('b');
                A = A - 2;
                B = B - 1;
            }
            
            
while( A > 0 && B > 0) {
                sb.append('a');
                sb.append('b');
                A--;
                B--;
            }
            
            while(A > 0) {
                sb.append('a');
                A--;
            }
            return sb.toString();
        }
        // handle B is large than A
        while(A > 0 && B > 0 && A < B - 1) {
            sb.append("bb");
            sb.append('a');
            A--;
            B = B - 2;
        }
        
        while( A > 0 && B > 0) {
            sb.append('a');
            sb.append('b');
            A--;
            B--;
        }
        
        
while(B > 0) {
            sb.append('b');
            B--;
        }
        return sb.toString();
    }
}
