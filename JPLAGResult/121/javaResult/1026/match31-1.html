class Solution {
    public String strWithout3a3b(int A, int B) {
        if (A + B < 3) {
            String ret = "";
            for (int i = 0; i < A; i++) {
                ret += 'a';
            }
            for (int i = 0; i < B; i++) {
                ret += 'b';
            }
            return ret;
        
}
        
        if (A > B) {
            return helper(A, B, 'a', 'b');
        } else if (A < B) {
            return helper(B, A, 'b', 'a');
        } else {
            String ret = "";
            for (int i = 0; i < A; i++) {
                ret += "ab";
            }
            return ret;
        }
    }
    
    private String helper(int A, int B, char a, char b) {
        StringBuilder sb = new StringBuilder();

        while (A > B && B > 0) {
            sb.append(a);
            sb.append(a);
            sb.append(b);
            A -= 2;
            B--;
        }
        //System.out.println(sb.toString());
        
        if (B == 0) {
            while (A > 0) {
                sb.append(a);
                A--;
            }
            return sb.toString();
        }
        
        if (A == B) {
            while (A > 0) {
                sb.append(a);
                sb.append(b);
                A--;
            }
        }
        
        
        return sb.toString();
    }
}
