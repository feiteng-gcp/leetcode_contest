class Solution {
    public int countTriplets(int[] A) {
        Map<Integer, Integer> d = new HashMap<>();
        for (int a : A) {
            for (int b : A) {
                int r = d.getOrDefault(a&b, 0);
                d.put(a&b, r+1);
            }
        }
        System.out.println(d);
        int result = 0;
        for (int a : A) {
            for (int k : d.keySet()) {
                if ((a & k) == 0) {
                    // System.out.println(k + " " + d.get(k));
                    result += d.get(k);
                }
            }
        }
        return result;
    }
}
