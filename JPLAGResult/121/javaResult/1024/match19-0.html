class Solution {
    public int countTriplets(int[] A) {
        int N = 65536;
        int L = A.length;
        int[] counts = new int[N];
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < L; j++) {
                if ((i & A[j]) == 0) {
                    counts[i]++;
                }
            }
        }
        int res = 0;
        for (int i = 0; i < L; i++) {
            for (int j = 0; j < L; j++) {
                res += counts[A[i] & A[j]];
            }
        }
        return res;
    }
}
