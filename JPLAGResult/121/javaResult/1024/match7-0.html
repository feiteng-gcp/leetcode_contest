class Solution {
    public int countTriplets(int[] A) {
        int n = A.length;
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int s = A[i] & A[j];
                map.put(s, map.getOrDefault(s, 0) + 1);
            }
        }
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int key : map.keySet()) {
                if ((key & A[i]) == 0) {
                    res += map.get(key);
                }
            }
        }
        
        return res;
    }
}
