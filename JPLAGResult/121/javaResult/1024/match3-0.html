class Solution {
    public int countTriplets(int[] A) {
        int res = 0;
        for (int i = 0; i < A.length; i++){
            for (int j = i; j < A.length; j++){
                for (int k = j; k < A.length; k++){
                    if ((A[i]&A[j]&A[k]) == 0){
                        if (i == j && j == k){
                            res++;
                        } else if (i == j || i == k || j == k){
                            res += 3;
                        } else {
                            res += 6;
                        }
                    }
                }
            }
        }
        return res;
    }
}
