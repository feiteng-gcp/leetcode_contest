class Solution {
    public int countTriplets(int[] A) {
        
int N = 1 << 16;
        int[] count = new int[N];
        for (int i = 0; i < N; i++) {
            for (int x : A) {
                if ((x & i) == 0) {
                    count[i]++;
                }
            }
        }
        int ans = 0;
        for (int x : A) {
            for (int y : A) {
                ans += count[x & y];
            }
        }
        return ans;
    }
}
