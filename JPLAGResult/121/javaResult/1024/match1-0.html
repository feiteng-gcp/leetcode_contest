class Solution {
    public int countTriplets(int[] A) {
        int result = 0;
        for (int i = 0; i < A.length; i++) {
            for (int j = i; j < A.length; j++) {
                for (int k = j; k < A.length; k++) {
                    if ((A[i] & A[j] & A[k]) == 0) {
                        if (i != j && i != k && j != k) {
                            result += 6;
                        } else if (i == j && j == k) {
                            result += 1;
                        } else if (i == j || i == k || j == k) {
                            result += 3;
                        }
                        
                    }
                }
            }
        }
        
        return result;
    }
}
