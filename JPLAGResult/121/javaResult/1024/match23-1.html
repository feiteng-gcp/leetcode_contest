class Solution {
    public int countTriplets(int[] A) {
        
int dp[] = new int[1 << 16];
        for(int v = 0; v < (1 << 16); v++) {
            for(int i = 0; i < A.length; i++) {
                if((v & A[i]) == 0) {
                    dp[v]++;
                }
            }
        }
        
        int ret = 0;
        for(int i = 0; i < A.length; i++) {
            for(int j = 0; j < A.length; j++) {
                ret += dp[A[i] & A[j]];
            }
        }
        return ret;
    }
}
