class Solution {
    public int countTriplets(int[] A) {
        int count = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for(int i = 0; i < A.length; i++) {
            for(int j = 0; j < A.length; j++) {
                int cur = A[i] & A[j];
                map.put(cur, map.getOrDefault(cur, 0) + 1);
            }
        }
        
        
for(int i = 0; i < A.length; i++) {
            for(int key : map.keySet()){
                if((A[i] & key) == 0) {
                    count += map.get(key);
                }
            }
        }
        
        return count;
    }
}
