class Solution {
    public int countTriplets(int[] A) {
        int count = 0;
        for (int i = 0; i < A.length; i++) {
            for (int j = i; j < A.length; j++) {
                for (int z = j; z < A.length; z++) {
                    if ((A[i] & A[j] & A[z]) == 0) {
                        int repeat = 1;
                        if (i == j && i == z) repeat = 3;
                        else if (i == j || j == z) repeat = 2;
                        
if (repeat == 3) count++;
                        else if (repeat == 2) count += 3;
                        else count += 6;
                    }
                }
            }
        }
        return count;
    }
}
