class Solution {
    public int countTriplets(int[] A) {
        
        int count =0;
        for(int i=0; i<A.length; ++i) {
            for(int j=i; j<A.length; ++j) {
                for(int k=j; k<A.length; ++k) {
                    if((A[i] & (A[j] & A[k])) == 0) {
                        if(i==j && j==k) {
                            ++count;
                        } else if(i==j || j==k) {
                            count = count + 3;
                        } else {
                            count = count + 6;
                        }
                    
}
                }
            }
        }
        
        return count;
    }
}
