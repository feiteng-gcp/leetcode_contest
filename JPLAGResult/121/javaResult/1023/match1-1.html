class TimeMap {
    
    HashMap<String,TreeMap<Integer,String>> mp = new HashMap<>();
    /** Initialize your data structure here. */
    public TimeMap() {
        
    }
    
    public void set(String key, String value, int timestamp) {
        if(!mp.containsKey(key)){
            mp.put(key,new TreeMap<Integer,String>());
        }
        mp.get(key).put(timestamp,value);
        
    }
    
    public String get(String key, int timestamp) {
        if(!mp.containsKey(key)){
            return "";
        }
        TreeMap<Integer,String> r = mp.get(key);
        Integer num = r.floorKey(timestamp);
        if(num==null){
            return "";
        }else{
            return r.get(num);
        }
        
        
        
    }
}
