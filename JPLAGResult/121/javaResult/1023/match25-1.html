class TimeMap {
    Map<String, TreeMap<Integer, String>> content;
    
    /** Initialize your data structure here. */
    public TimeMap() {
        content = new HashMap<>();
    }
    
    public void set(String key, String value, int timestamp) {
        if (!content.containsKey(key)) {
            content.put(key, new TreeMap<>());
        }
        
content.get(key).put(timestamp, value);
    }
    
    public String get(String key, int timestamp) {
        if (!content.containsKey(key)) {
            return "";
        }
        TreeMap<Integer, String> timedValues = content.get(key);
        Map.Entry<Integer, String> floorEntry = timedValues.floorEntry(timestamp);
        if (floorEntry == null) {
            return "";
        }
        return floorEntry.getValue();
    }
}

 * Your TimeMap object will be instantiated and called as such:
