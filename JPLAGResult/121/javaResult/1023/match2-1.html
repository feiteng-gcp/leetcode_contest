class TimeMap {
    class node{
        String val;
        int time;
        node(String a, int b){
            val = a;
            time = b;
        }
    }
    HashMap<String, List<node>> map = new HashMap<>();
    /** Initialize your data structure here. */
    public TimeMap() {
        return;
    }
    
    public void set(String key, String value, int timestamp) {
        if (map.containsKey(key)) {
            map.get(key).add(new node(value, timestamp));
        } else {
            List<node> nd = new ArrayList<>();
            nd.add(new node(value, timestamp));
            map.put(key, nd);
        }
    }
    
    public String get(String key, int timestamp) {
        String ans = "";
        int minrp = Integer.MIN_VALUE;
        if (!map.containsKey(key)) {
            return "";
        } else {
            List<node> nd = map.get(key);
            for (node ndd:nd) {
                if (ndd.time <= timestamp) {
                    if (ndd.time > minrp) {
                        ans = ndd.val;
                        minrp = ndd.time;   
                    }
                }
            }
            return ans;
        }
    }
}
