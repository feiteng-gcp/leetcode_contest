class TimeMap {
    HashMap<String, TreeMap<Integer, String>> core = new HashMap<>();
    /** Initialize your data structure here. */
    public TimeMap() {

    }

    public void set(String key, String value, int timestamp) {
        if (!core.containsKey(key)) {
            core.put(key, new TreeMap<>());
        }
        core.get(key).put(timestamp, value);
    }

    public String get(String key, int timestamp) {
        if (!core.containsKey(key)) {
            return "";
        }
        TreeMap<Integer, String> tree = core.get(key);
        if (tree.floorEntry(timestamp) == null) {
            return "";
        }
        return tree.floorEntry(timestamp).getValue();
    }
}
