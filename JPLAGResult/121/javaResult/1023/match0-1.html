class TimeMap {
    Map<String, List<String>> keyValueMap;
    Map<String, List<Integer>> keyTimeMap;

    /** Initialize your data structure here. */
    public TimeMap() {
        keyValueMap = new HashMap<String, List<String>>();
        keyTimeMap = new HashMap<String, List<Integer>>();
    }
    
    public void set(String key, String value, int timestamp) {
        List<String> valuesList = keyValueMap.getOrDefault(key, new ArrayList<String>());
        List<Integer> timesList = keyTimeMap.getOrDefault(key, new ArrayList<Integer>());
        valuesList.add(value);
        timesList.add(timestamp);
        keyValueMap.put(key, valuesList);
        keyTimeMap.put(key, timesList);
    }
    
    public String get(String key, int timestamp) {
    	List<String> valuesList = keyValueMap.getOrDefault(key, new ArrayList<String>());
        List<Integer> timesList = keyTimeMap.getOrDefault(key, new ArrayList<Integer>());
        int index = binarySearch(timesList, timestamp);
        if (index < 0)
        	return "";
        else
        	return valuesList.get(index);
    }

    private static int binarySearch(List<Integer> list, int key) {
        int low = 0;
		int high = list.size() - 1;
		while (low <= high) {
			int mid = (low + high) / 2;
			if (key < list.get(mid))
				high = mid - 1;
			else if (key == list.get(mid))
				return mid;
			else
				low = mid + 1;
		}
		return low - 1;
    }
}
