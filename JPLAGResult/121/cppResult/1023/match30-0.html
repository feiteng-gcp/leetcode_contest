class TimeMap {
public:
    /** Initialize your data structure here. */
    unordered_map<string, map<int,string>> aloc;
    TimeMap() {}
    
    void set(string key, string value, int timestamp) {
        aloc[key][timestamp] = value;
    }
    
    string get(string key, int timestamp) {
        auto it = aloc[key].upper_bound(timestamp);
        if(it == aloc[key].begin()) return "";
        return (--it)->second;
    }
};
