class TimeMap {
public:
    /** Initialize your data structure here. */
  map<string, map<int, string> > G;
  TimeMap() {
    G.clear();
  }
  
  void set(string key, string value, int timestamp) {
    G[key][timestamp]=value;
  }
  
  string get(string key, int timestamp) {
    if(G[key].empty()) return ""s;
    auto it=G[key].upper_bound(timestamp);
    if(it!=G[key].begin()) --it;
    if(it->first<=timestamp) return it->second;
    return ""s;
  }
};
