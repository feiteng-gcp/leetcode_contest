class TimeMap {
public:
    /** Initialize your data structure here. */
    TimeMap() {
        
    }
    
    void set(string key, string value, int timestamp) {
        m[key][timestamp] = value;
    }
    
    string get(string key, int timestamp) {
        auto f = m.find(key);
        if(f == m.end())
            return "";
        
        auto it = f->second.equal_range(timestamp);
        if(it.first != it.second)
            return it.first->second;
        
        if(it.first == f->second.begin())
            it.first = f->second.end();
        else
            advance(it.first, -1);
        
        return it.first == f->second.end() ? "" : it.first->second;
    }
    
    map<string, map<int, string>> m;
};
