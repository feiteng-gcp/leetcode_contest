class Solution {
public:
    int countTriplets(vector<int>& A) {
        vector<int> h(1 << 16, 0);
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < A.size(); ++j) {
                h[A[i] & A[j]] += 1;
            }
        }
        
        int result = 0;
        for (int i = 0; i < h.size(); ++i) {
            if (h[i] == 0)
                continue;
            for (int j = 0; j < A.size(); ++j) {
                if ((i & A[j]) == 0)
                    result += h[i];
            }
        }
        
        return result;
    }
};
