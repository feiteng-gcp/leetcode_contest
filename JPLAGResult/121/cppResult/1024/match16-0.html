class Solution {
public:

    int countTriplets(vector<int>& A) {
        vector<int> dp(1 << 16, 0);
        for(int i = 0;i < A.size();i++){
            dp[A[i]] += 1;
        }
        for(int i = 0;i < 16;i++){
            for(int j = 0;j < 1 << 16;j++){
                if (j & (1 << i))
                    dp[j] += dp[j^(1 << i)];
            }
        }
        long long res = 0;
        for(int i = 0;i < A.size();i++){
            for(int j = 0;j < A.size();j++){
                res += dp[((1 << 16)-1)^(A[i]&A[j])];
            }
        }
        return res;
    }
};
