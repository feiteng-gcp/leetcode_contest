class Solution {
public:
    int countTriplets(vector<int>& A) {

        int n = A.size();
        vector<int> table(65536, 0);
        for(int a: A)
            for(int i = 0; i < 65536; i ++)
                if(!(a & i))
                    table[i] ++;

        int res = 0;
        for(int i = 0; i < n; i ++)
            for(int j = 0; j < n; j ++)
                res += table[A[i] & A[j]];
        return res;
    }
};
