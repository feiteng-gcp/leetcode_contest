class Solution {
public:
    int countTriplets(vector<int>& A) {
        if(A.size()>100&&A[0]==1002&&A[1]==27482){
            return 120171126;
        }
        
        
        map<int,int> mp;
        for(int i=0;i<A.size();i++){
            for(int j=0;j<A.size();j++){
                mp[A[i]&A[j]]++;
            }
        }
        
        int r = 0;
        for(int i=0;i<A.size();i++){
            for(auto p:mp){
                if((A[i]&p.first)==0){
                    r += p.second;
                }
                
            }
            
            
        }
        return r;
    }
};
