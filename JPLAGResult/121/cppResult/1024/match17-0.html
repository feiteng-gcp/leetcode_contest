class Solution {
public:
    int vis[80007];
    int countTriplets(vector<int>& A) {
        int n=A.size();
        memset(vis, 0, sizeof(vis));
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                vis[A[i]&A[j]]++;
            }
        }
        long long ans=0;
        for(int i=0; i<n; i++) {
            for(int j=0; j<65538; j++) {
                if ((A[i]&j)==0) ans+=vis[j];
            }
        }
        return ans;
    }
};
