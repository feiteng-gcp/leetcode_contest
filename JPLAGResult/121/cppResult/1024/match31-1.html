class Solution {
public:
    int countTriplets(vector<int>& A) {
        int totalCount = 0;
        int count;
        // all same, it must be 0
        for(auto n : A){
            if(!n) totalCount++;
        }
        // 2-1 set
        count = 0;
        for(int i = 0;i<A.size();i++){
            for(int j = i+1 ;j<A.size();j++){
                if((A[i] & A[j]) == 0) ++count;
            }
        }
        totalCount += count * 6;
        // three diffent

        count = 0;
        for(int i = 0;i<A.size();i++){
            for(int j = i + 1  ;j<A.size();j++){
                for(int k = j + 1;k<A.size();k++){
                    if((A[i] & A[j] & A[k]) == 0) ++count;
                }
            }
        }
        totalCount += count * 6; // 3!
        return totalCount;
    }
};
