int C[1<<16];

class Solution {
public:
    int countTriplets(vector<int>& A) {
        memset(C,0,sizeof(C));
        for (int m = 0; m < 1<<16; ++m) {

            for (int a : A) {
                if ((m & a) == 0) {
                    ++C[m];
                }
            }
        }
        int res = 0;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < A.size(); ++j) {
                res += C[A[i]&A[j]];
            }
        }
        return res;
    }
};
