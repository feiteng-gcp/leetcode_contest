int cnt[65536];
class Solution {
public:

    int countTriplets(vector<int>& A) {
        int n=A.size();
	memset(cnt,0,sizeof(cnt));
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cnt[A[i]&A[j]]++;
		}
	}
	int ans=0;
	bitset<1000> b[16];
	for(int i=0;i<16;i++)
	{
		for(int j=0;j<n;j++)
		{
			if((A[j]&(1<<i)))b[i].set(j);
		}
	}

	for(int i=0;i<65536;i++)
	{
		bitset<1000> res;
		for(int j=0;j<16;j++)
		{
			if(i&(1<<j))res=(res|b[j]);
		}
		int cntt=res.count();
		ans=ans+cnt[i]*(n-cntt);
	}
	return ans;
    }
};
