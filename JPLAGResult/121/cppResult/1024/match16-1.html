class Solution {
public:
    int countTriplets(vector<int>& A) {
        int f[65536];
        int ans = 0;
        memset(f, 0, sizeof f);

        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < 65536; j++) {
                if ((A[i] & j) == 0) f[j]++;
            }
        }

        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A.size(); j++) {
                ans += f[A[i] & A[j]];
            }
        }
        return ans;
    }
};
