class Solution {
public:
    int countTriplets(vector<int>& A) {
        int ans = 0;
        int n = A.size();
        for(int i=0; i<n; ++i) {
            for(int j=i; j<n; ++j) {
                for(int k=j; k<n; ++k) {
                    //cout<<i<<","<<j<<","<<k<<endl;
                    if((A[i]&A[j]&A[k])==0) {
                        if(i==j&&j==k) {
                            ++ans;
                        }
                        else if(i==j&&j!=k||i!=j&&j==k){
                            ans += 3;
                        }
                        else ans+=6;
                    }
                }
            }
        }
        return ans;
    }
};
