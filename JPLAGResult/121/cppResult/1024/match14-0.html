class Solution {
public:
    int countTriplets(vector<int>& A) {
        vector<int> cnts(65536);
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < cnts.size(); ++j) {
                if (0 == (A[i] & j)) {
                    ++cnts[j];
                }
            }
        }
        int result = 0;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < A.size(); ++j) {
                result += cnts[A[i] & A[j]];
            }
        }
        return result;
    }
};
