class Solution {
public:
    int countTriplets(vector<int>& A) {
        unordered_map<int, int> m;
        
        int cnt = 0;
        
        for(int i = 0; i < A.size(); i++)
        {
            for(int j = 0; j < A.size(); j ++)
            {
                m[A[i]&A[j]]++;
            }
        }
        //for(auto p: m) cout<<p.first<<" "<<p.second<<endl;
        for(int i = 0; i < A.size(); i++)
        {
            for(auto p: m)
            {
                if((p.first & A[i]) == 0)
                    cnt += p.second;
            }
        }
        
        return cnt;
    }
};
