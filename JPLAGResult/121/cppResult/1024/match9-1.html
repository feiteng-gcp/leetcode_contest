class Solution {
public:
    int countTriplets(vector<int>& A) {
        map< int , int > check ; 
        int ans = 0; 

        for( int i=0 ; i< A.size() ; i++){
            for( int j=0 ; j<A.size() ; j++){
                check[( A[i]&A[j] )]++;
            }
        }

        for( int i =0 ; i<A.size() ; i++){
            for( auto & x: check ){
                if( (A[i] & x.first )  == 0 ) 
                    ans += x.second ; 
            }
        }
        return ans ;
    }
};
