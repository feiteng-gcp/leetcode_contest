const int maxn = (1<<16);
int cnt[maxn];
class Solution {
public:

    int countTriplets(vector<int>& A) {
        int n = A.size();
        memset(cnt, 0, sizeof(cnt));
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < maxn; j++) {

                if(!(A[i]& j)) cnt[j]++; 
            }
        }
        long long ans = 0;
       // cout  << "cnt[0] = " << cnt[0] << endl;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                ans += cnt[A[i]&A[j]];
            }
        }
        return ans;
    }
};
