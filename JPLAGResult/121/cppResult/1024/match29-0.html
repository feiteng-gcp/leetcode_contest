class Solution {
public:
    int table[65536];
    
    void add(int val){
        for(int i = 0; i < 65536; i++){
            if((i & val) == 0){
                table[i] += 1;
            }
        }
    }
    
    int countTriplets(vector<int>& A) {
        int res = 0;
        int zeros = 0;
        for(int i = 0; i < A.size(); i++){
            if(A[i] == 0){
                zeros++;
            }
        }
        
        res += zeros;
        for(int i = 0; i < A.size(); i++){
            for(int j = i + 1; j < A.size(); j++){
                if(A[i] & A[j]){

                } else {
                    res += 6;
                }        
            }
        }
        
        add(A[0]);
        
        for(int i = 1; i < A.size(); i++){
            for(int j = i + 1; j < A.size(); j++){
                int c = A[i] & A[j];
                
                res += table[c] * 6;
            }
            
            add(A[i]);
        }
        
        return res;
    }
};
