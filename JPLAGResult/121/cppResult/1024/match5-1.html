class Solution {
public:
    int countTriplets(vector<int>& A) {
        int max = 1 << 16;
        int n = A.size();
        vector<int> num(max, 0);
        for (int i = 0; i < max; i++)

            for (int j = 0; j < n; j++)
                if ((i & A[j]) == 0)
                    num[i]++;
        int ans = 0;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                ans += num[(A[i] & A[j])];
        return ans;
    }
};
