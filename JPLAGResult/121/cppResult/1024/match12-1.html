class Solution {
public:
    int countTriplets(vector<int>& A) {
        int N = 65536;
        vector<int> count(N, 0);

        for (int& x : A) {
            for (int i = 0; i < N; ++i) {
                if ((x & i) == 0) {
                    ++count[i];
                }
            }
        }
        
        int res = 0;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < A.size(); ++j) {
                res += count[A[i] & A[j]];
            }
        }
        
        return res;
    }
};
