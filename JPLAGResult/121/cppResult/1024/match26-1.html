class Solution {
public:
    int countTriplets(vector<int>& A) {
        int ret = 0;

        unordered_map<int, int> m;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < A.size(); ++j) {
                ++m[(A[i] & A[j])];
            }
        }
        for (auto p: m) {
            int n = p.first;
            int count = p.second;

            for (int i = 0; i < A.size(); ++i) {
                if ((A[i] & n) == 0) ret += count;
            }
        }
        return ret;
    }
};
