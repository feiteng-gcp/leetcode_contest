#include <bits/stdc++.h>
using namespace std;
using vi = vector<int>;
using vvi = vector<vi>;

class Solution {
public:
    int countTriplets(vector<int>& A) {
        int n = A.size();
        vi cnt(1 << 16);

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 1 << 16; j++) {
                if ((A[i] & j) == 0) {
                    cnt[j]++;
                }
            }
        }
        int ret = 0;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int a = A[i] & A[j];
                ret += cnt[a];
            }
        }
        return ret;
    }
};
