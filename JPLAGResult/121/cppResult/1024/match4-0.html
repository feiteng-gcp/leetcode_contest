class Solution {
public:
    int countTriplets(vector<int>& A) {
        int len = (int)A.size(), mask = (1 << 16) - 1;
    vector<int> tab(1 << 16, 0);
    for(int i = 0; i < len; i++)
    {
        int am = A[i] ^ mask;
        for(int j = am; j > 0; j = (j - 1) & am) tab[j]++;
    }
    tab[0] = len;
    int ans = 0;
    for(int i = 0; i < len; i++)
        for(int j = 0; j < len; j++)
            ans += tab[A[i] & A[j]];
    return ans;
    }
};
