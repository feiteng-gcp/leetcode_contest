class Solution {
public:
    int countTriplets(vector<int>& A) {
        unordered_map<int, int> count;

        for (int i = 0; i < A.size(); i++) {
            for (int j = 0; j < A.size(); j++) {
                ++count[A[i]&A[j]];
            }
        }
        int ans = 0;

        for (int i = 0; i < A.size(); i++) {
            for (auto e : count) {
                if (!(e.first & A[i])) ans += e.second;
            }
        }
        return ans;
    }
};
