class Solution {
public:
string strWithout3a3b(int A, int B) {
	int size = A + B;
	int count = abs(A-B);
	string result;
	if (A > B)
	{
		if (B >= count)
		{
			for (int i = 0; i < count; ++i)
			{
				result += "aab";
			}
			for (int i = result.size(); i < size; i += 2)
			{
				result += "ab";
			}
		}
		else
		{
			for (int i = 0; i < B; ++i)
			{
				result += "aab";
			}
			for (int i = result.size(); i < size; ++i)
			{
				result += "a";
			}
		}
	}
	else
	{
		if (A >= count)
		{
			for (int i = 0; i < count; ++i)
			{
				result += "bba";
			}
			for (int i = result.size(); i < size; i += 2)
			{
				result += "ba";
			}
		}
		else
		{
			for (int i = 0; i < A; ++i)
			{
				result += "bba";
			}
			for (int i = result.size(); i < size; ++i)
			{
				result += "b";
			}
		}
	}
	return result;
}
};
