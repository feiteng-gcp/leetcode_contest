class Solution {
public:
    string strWithout3a3b(int A, int B)
{
    if(!B) return string(A, 'a');
    else if(!A) return string(B, 'b');
    string ans = "";
    while(A > 0 && B > 0)
    {
        if(A > B)
        {
            ans += "aab";
            A -= 2; B--;
        }
        else if(A < B)
        {
            ans += "abb";
            A--; B -=2;
        }
        else
        {
            ans += "ab";
            A--;
            B--;
        }
    }
    ans = string(B, 'b') + ans + string(A, 'a');
    return ans;
}
};
