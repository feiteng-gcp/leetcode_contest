class Solution {
public:
    string strWithout3a3b(int A, int B) {
        int go = A>B ? 0: 1;
        string ans;

        while(A || B){
            if(go==0){
                if(A>B && A>=2) {
                    
                    ans+="aa";
                    A-=2;
                }
                else 
                {
                    ans +="a";
                    A--;
                }
            }else{
                if(B>A&& B>=2){
                    ans+="bb";
                    B-=2;
                }else{
                    ans +="b";
                    B--;
                }
            }
            go^=1;
        }
        return ans;
    }
};
