class Solution {
public:
    string strWithout3a3b(int A, int B) {
	string ans;
	char last1 = 0, last2 = 0;
	while (A || B) {
	    char c;

	    if (last1 == 'a' && last2 == 'a') {
		c = 'b';
	    } else if (last1 == 'b' && last2 == 'b') {
		c = 'a';
	    } else if (A > B) {
		c = 'a';
	    } else {
		c = 'b';
	    }
	    if (c == 'a') {
		ans.push_back('a');
		A--;
	    } else {
		ans.push_back('b');
		B--;
	    }
	    last2 = last1;
	    last1 = c;
	}
	return ans;
    }
};
