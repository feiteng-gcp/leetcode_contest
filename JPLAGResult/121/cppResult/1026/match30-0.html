class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string r;
        while (A > 0 || B > 0) {
            if (r.size() >= 2 && r[r.size() - 1] == 'a' && r[r.size() - 2] == 'a') {
                r.push_back('b');
                B -= 1;
            } else if (r.size() >= 2 && r[r.size() - 1] == 'b' && r[r.size() - 2] == 'b') {
                r.push_back('a');
                A -= 1;
            } else if (A >= B) {
                r.push_back('a');
                A -= 1;
            } else {
                r.push_back('b');
                B -= 1;
            }
        }
        return r;
    }
};
