class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string ans="";

	while(A||B)
	{
		if(!A)
		{
			ans=ans+"b";
			B--;
		}
		else if(!B)
		{
			ans=ans+"a";
			A--;
		}
		else if(A>B)
		{
			ans=ans+"aab";
			A=A-2;
			B--;
		}
		else if(A<B)
		{
			ans=ans+"bba";
			B=B-2;
			A--;
		}
		else
		{
			ans=ans+"ab";
			A--;
			B--;
		}
	}
	return ans;
    }
};
