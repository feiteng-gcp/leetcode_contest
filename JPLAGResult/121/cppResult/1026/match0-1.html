class Solution {
public:
    string strWithout3a3b(int A, int B) {
        int rev = 0;
        if(A<B){
             rev  =1;
             swap(A,B);  
        }
            
            string s1 = "";
            int c = 0;
            for(int i=0;i<A;i++){
                c++;
                s1.push_back('a');
                if(c==2&&B>0){
                    s1.push_back('b');
                    B--;
                    c=0;
                }
                    
                
            }
       
        
            string s2 = "";
            for(int i=0;i<s1.size();i++){
                if(s1[i]=='a'){
                    s2.push_back('a');
                }else if(B>0){
                    s2.push_back('b');
                    s2.push_back('b');
                    B--;
                }else{
                    s2.push_back('b');
                }
            }
        
            while(B>0){
                s2.push_back('b');
                B--;
            }
        
        if(rev){
            string s3 = "";
            for(int i=0;i<s2.size();i++){
             if(s2[i]=='a'){
                 s3.push_back('b');
             }else{
                s3.push_back('a');   
             }
                
            }
            return  s3;
            
        }
        
            return s2;
            
        
        
    }
};
