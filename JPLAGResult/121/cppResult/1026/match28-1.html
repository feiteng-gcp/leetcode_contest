class Solution {
public:
    string strWithout3a3b(int A, int B) {
    char a = 'a';
    char b = 'b';

    if (A < B) {
        swap(A, B);
        swap(a, b);
    }

    ostringstream o1;
    o1 << a << a << b;
    string s1 = o1.str();
    ostringstream o2;
    o2 << a << b;
    string s2 = o2.str();
    ostringstream o3;
    o3 << a << a;
    string s3 = o3.str();

    string ans;
    while (A > B && B) {
        ans += s1;
        A -= 2;
        B -= 1;
    }
    while (A != 0 && B != 0) {
        ans += s2;
        --A;
        --B;
    }

    while (A != 0) {
        ans += a;
        --A;
    }
    return ans;
    }
};
