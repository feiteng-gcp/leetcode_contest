class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string resultStr = "";
        if(B - A >= 2) {
            resultStr = "bb";
            B -= 2;
        }
        while (1) {
            if(A == 0) {
                resultStr += string(B,'b');
                break;
            }
            if(B == 0) {
                resultStr += string(A,'a');
                break;
            }
            
            if(A > B) {
                A -= 2;
                B -= 1;
                resultStr += "aab";
            }
            else if (B > A) {
                A -= 1;
                B -= 2;
                resultStr += "abb";
            }
            else {
                A -= 1;
                B -= 1;
                resultStr += "ab";
            }
        }
        return resultStr;
    }
};
