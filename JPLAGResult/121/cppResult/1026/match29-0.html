class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string res;
        while (A != B) {
            if (A > B) {
                res += string(min(2, A), 'a') + string(min(1, B), 'b');
                A -= min(2, A);
                B -= min(1, B);
            } else if (A < B) {
                res += string(min(2, B), 'b') + string(min(1, A), 'a');
                A -= min(1, A);
                B -= min(2, B);
            }
        }
        for (int i = 0; i < A; ++i) {
            res += res.size() && res.back() == 'a' ? "ba" : "ab";
        }
        return res;

    }
};
