class Solution {
public:
    bool legal(int ch){
        int len = ans.length() ;
        if ( len <= 1 ) return true ;
        if ( ch == 1 ){//a
            if ( ans[ len-1 ] == ans[ len-2 ] && ans[ len-1 ] == 'a' )
                return false ;
            else return true ;
        }
        if ( ch == 2 ){//b
            if ( ans[ len-1 ] == ans[ len-2 ] && ans[ len-1 ] == 'b' )
                return false ;
            else return true ;
        }
        else return true ;
    }
    string strWithout3a3b(int A, int B) {
        while ( A > 0 || B > 0 ){
            if ( A > B ){
                if ( legal(1) ){
                    ans  = ans + "a" ;
                    A-- ;
                }
                else{
                    ans = ans + "b" ;
                    B-- ;
                }
            }
            else{
                if ( legal(2) ){
                    ans = ans + "b" ;
                    B-- ;
                }
                else{
                    ans = ans + "a" ;
                    A-- ;
                }
            }
        }
        return ans ;
    }
    string ans = "" ;
};
