class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string s;
        int c = 0;
        while(A || B){
        	if(c == 2){
        		s += 'b';
        		c = - 1;
        		B -= 1;
			}
			else if(c == -2){
				s += 'a';
				c = 1;
				A -= 1;
			}
			else if(A > B){
				s += 'a';
				c = max(c + 1, 1);
				A -= 1;
			}
			else{
				s += 'b';
				c = min(c - 1, -1);
				B -= 1;
			}
		}
		return s;
    }
};
