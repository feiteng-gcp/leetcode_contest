class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string s("");
        if(A == 0 && B == 0) return s;
        if(A == 1 && B == 1) return "ab";
        while(A - B >= 2 && B > 0) {
            s += "aab";
            A -= 2;
            B -= 1;
        }
        while(B - A >= 2 & A > 0) {
            s += "bba";
            A -= 1;
            B -= 2;
        }
        while(A > 1 && B > 1) {
            if(A > B) {
                s += "aba";
                A -= 2;
                B -= 1;
            }
            else{
                s += "bab";
                A -= 1;
                B -= 2;
            }
        }
        while(A) {
            s += "a";
            --A;
        }
            
        while(B) {
            s += "b";
            --B;
        }
            
        return s;
    }
};
