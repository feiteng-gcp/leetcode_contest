class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string s;
        if(A >= B){

           while(A > B && B > 0) {
                s += "aab";
                A-=2;
                --B;
            }
            while(A&&B) {
                s += "ab";
                --A;
                --B;
            }
        } else {
            while(B > A && A > 0) {
                s += "bba";
                --A;
                B-=2;
            }

            while(B && A) {
                s += "ba";
                --B;
                --A;
            }
        }
        while(A) {
            s+="a";
            --A;
        }
        while(B) {
            s+= "b";
            --B;
        }
        return s;
    }
};
