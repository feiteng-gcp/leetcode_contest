class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string result="";
        int leftA=A,leftB=B;
        int bcount=0,acount=0;
        for(int i=0;i<A+B;i++){
            if(leftA<leftB&&bcount<2){
                bcount++;
                acount=0;
                result+="b";
                leftB--;
            }
            else if(leftB>=leftA&&acount<2){
                acount++;
                bcount=0;
                result+="a";
                leftA--;
            }
            else{
                if(acount<2&&leftA>0){
                    acount++;
                    bcount=0;
                    result+="a";
                    leftA--;
                }
                else{
                    bcount++;
                acount=0;
                result+="b";
                leftB--;
                }
                    
            }
                
        }
        return result;
    }
};
