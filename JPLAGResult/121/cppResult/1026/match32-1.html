class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string ans;

        while (A != B && A != 0 && B != 0) {
            if (A > B) {
                ans += "aab";
                A -= 2;
                B -= 1;
            } else {
                ans += "bba";
                B -= 2;
                A -= 1;
            }
        }
        if (A == B) {
            for (int i = 0; i < A; i++) {
                ans += "ab";
            }
        } else if (A > 0) {
            for (int i = 0; i < A; i++) {
                ans += "a";
            }
        } else if (B > 0) {
            for (int i = 0; i < B; i++) {
                ans += "b";
            }
        }
        return ans;
    }
};
