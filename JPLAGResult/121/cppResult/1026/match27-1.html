class Solution
{
public:
    std::string strWithout3a3b(int A, int B)
    {
        std::string S, T;
        if (A < B) {
            if (B > 2 * A) { T = std::string(B - 2 * A, 'b'), B = 2 * A; }

            for (int i = 0; i < B - A; i++) { S += "bba"; }
            for (int i = 0; i < 2 * A - B; i++) { S += "ba"; }
        } else {
            if (A > 2 * B) { T = std::string(A - 2 * B, 'a'), A = 2 * B; }

            for (int i = 0; i < A - B; i++) { S += "aab"; }
            for (int i = 0; i < 2 * B - A; i++) { S += "ab"; }
        }
        return S + T;
    }
};
