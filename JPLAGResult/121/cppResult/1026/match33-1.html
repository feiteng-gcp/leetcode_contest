class Solution {
public:
    string strWithout3a3b(int A, int B) {
        bool reverse = false;
        if (A < B) {
            int temp = B;
            B = A;
            A = temp;
            reverse = true;
        }
        string ret;
        while (A || B) {

            if (A == B) {
                if (A >= 2) {
                    ret += "aabb";
                    A -= 2;
                    B -= 2;
                }
                else {
                    ret += "ab";
                    A -= 1;
                    B -= 1;
                }
            }
            else {
                if (A >= 2) {
                    ret += "aa";
                    A -= 2;
                }
                else {
                    ret += "a";
                    A -= 1;
                }
                if (B) {
                    ret += "b";
                    B -= 1;
                }
            }
        }
        if (reverse) {
            for (char& c: ret) {
                char change;
                if (c == 'a') change = 'b';
                else change = 'a';
                c = change;
            }
        }
        return ret;
    }
};
