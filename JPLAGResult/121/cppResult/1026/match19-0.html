class Solution {
public:
    string strWithout3a3b(int a, int b) {
        string answer = "";
       	if(a > b){
       		while(a > b and a > 1 and b){
       			answer += "aab";
       			--b;
       			a -= 2;
       		}
       		while(a and a == b){
       			--a;--b;
       			answer += "ab";
       		}
       		while(a--)answer += 'a';
       		while(b--)answer += 'b';
       	}
       	else{
       		while(a < b and b > 1 and a){
       			answer += "bba";
       			--a;
       			b -= 2;
       		}
       		while(a and a == b){
       			--a;--b;
       			answer += "ba";
       		}
       		while(b--)answer += 'b';
       		while(a--)answer += 'a';
       	}
       	return answer;
    }
};
