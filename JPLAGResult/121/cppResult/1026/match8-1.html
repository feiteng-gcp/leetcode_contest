class Solution {
public:
    string strWithout3a3b(int A, int B) {

        string ans = "";
        while ((A > 0) && (B > 0)) {
            if (A > B) {
                ans += "aab";
                A -= 2;
                B -= 1;
            } else if (B > A) {
                ans += "bba";
                A -= 1;
                B -= 2;
            } else {
                ans += "ab";
                A--;
                B--;
            }
        }
        while (A > 0) {
            ans += "a";
            A--;
        }
        while (B > 0) {
            ans += "b";
            B--;
        }
        return ans;
    }
};
