class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string S = "";
        if(A > B){

            while(A+B > 0){
                if(A > B){
                    S.push_back('a');
                    A--;
                }
                if(A > 0){
                    S.push_back('a');
                    A--;
                }
                if(B > 0){
                    S.push_back('b');
                    B--;
                }
            }
        }else{

            while(A+B > 0){
                if(B > A){
                    S.push_back('b');
                    B--;
                }
                if(B > 0){
                    S.push_back('b');
                    B--;
                }
                if(A > 0){
                    S.push_back('a');
                    A--;
                }
            }
        }
        return S;
    }
};
