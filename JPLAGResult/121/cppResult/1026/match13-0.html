class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string ans;
        bool flag = A >= B;
        while (A > 0 || B > 0) {
            if (flag) {
                if (A > B) {
                    ans.push_back('a');
                    A--;
                }
                if (A > 0){ans.push_back('a'); A--;}
                if (B > 0){ans.push_back('b'); B--;}
            } else {
                if (B > A) {
                    ans.push_back('b');
                    B--;
                }
               if (B > 0){ans.push_back('b'); B--;}
                if (A > 0){ans.push_back('a'); A--;}             
            }            
        }
        return ans;
    }
};
