class Solution {
public:
    string strWithout3a3b(int A, int B) {
        
        char a = 'a';
        char b = 'b';
        
        if (A < B) {
            swap(a, b);
            swap(A, B);
        }
        
        stringstream ss;
        bool isA = A >= B;
        bool putTwo = (A + 1) / 2 <= (B + 1) / 2 + 1 ;
        
        while (A > 0 && B > 0) {
            ss << a;
            if (A > B) {
                ss << a;
                A--;
            }
            ss << b;
            A--;
            B--;
        }
        
        while (A > 0) {
            ss << a;
            A--;
        }
        
        while (B > 0) {
            ss << b;
            B--;
        }
        
        return ss.str();
    }
};
