class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string s;

        int N=A+B;
        int a=0,b=0;
        for(int i=0;i<N;i++){
            if(b==2 || A>=B && a<2){
                s+='a';
                a++;
                b=0;
                A--;
            } 
            else if(a==2 || B>=A && b<2){
                s+='b';
                b++;
                a=0;
                B--;
            }
        }
        return s;
    }
};
