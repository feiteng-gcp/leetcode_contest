class Solution {
public:

int mincostTickets(vector<int>& d, vector<int>& c) {
    int n = d.size();
    // 1 7 30
    vector<int> dp(n + 1, 0);
    for (int i = 1; i <= n; i++) {
    	dp[i] = dp[i - 1] + c[0];
    	for (int j = i - 1; j >= 1; j--) {
    		if(d[i - 1] - d[j - 1] + 1 <= 7) {
    			dp[i] = min(dp[i], dp[j - 1] + c[1]);
    		}
    		if(d[i - 1] - d[j - 1] + 1 <= 30) {
    			dp[i] = min(dp[i], dp[j - 1] + c[2]);
    		} else break;
    	}
    }
    return dp[n];

}
};
