class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> costOnDay(395, 0);
        int m = INT_MAX;
        for (int i = 30, j = 0; i < 395 && j < days.size(); ++i) {
            if (i - 30 == days[j]) {
                costOnDay[i] = costOnDay[i-1] + costs[0];
                costOnDay[i] = min(costOnDay[i-7] + costs[1], costOnDay[i]);
                costOnDay[i] = min(costOnDay[i-30] + costs[2], costOnDay[i]);
                ++j;
                m = costOnDay[i];
            }
            else {
                costOnDay[i] = costOnDay[i-1];
            }
        }
        return  m;
    }
};
