class Solution {
public:
    
    
    
     
    int r = 12002;
    
    
    int jump(vector<int>& days, vector<int>& costs,int cur,int mon,vector<vector<int>> &dp){
        if(mon>r){
            return 12002;
        }
        
        
        if(dp[cur][mon]<999900){
            return dp[cur][mon];
        }
        
        
        if(cur>=days.size()){
            r = min(r,mon);
            dp[cur][mon] = mon;
            return mon;
        }
        
        int n1 = jump(days,costs,cur+1,mon+costs[0],dp);
        int c = cur+1;
        for(;c<days.size();c++){
            if(days[c]>days[cur]+6){
                break;
            }   
        }
        int n2 = jump(days,costs,c,mon+costs[1],dp);
        
        
        int c1 = cur+1;
        for(;c1<days.size();c1++){
            if(days[c1]>days[cur]+29){
                break;
            }   
        }
        int n3 = jump(days,costs,c1,mon+costs[2],dp);
        
        
        
        dp[cur][mon] = min(n1,min(n2,n3));
        return  dp[cur][mon];
        
    }
    
    
    
    int mincostTickets(vector<int>& days, vector<int>& costs) {
     vector<vector<int>> dp(days.size()+1, vector<int>(12007, 0));
        for(int i=0;i<days.size()+1;i++){
           
            for(int j=0;j<12007;j++){
                   dp[i][j] = 9999000;
            }
            
        }
        
        
        return jump(days,costs,0,0,dp);
    }
};
