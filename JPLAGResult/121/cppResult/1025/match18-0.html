class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int lastDay = days.back();
        vector<int> dp(lastDay + 1);
        vector<bool> dayNeed(lastDay + 1, false);
        for(auto day : days){
            dayNeed[day] = true;
        }
        dp[0] = 0;
        for(int i = 1 ; i<= lastDay; i++){
            if(dayNeed[i]){
                // buy a 1-day for today
                int minValue = costs[0] + dp[i - 1];
                // buy a 7-day
                int prev = max(i - 7, 0);
                minValue = min(minValue, costs[1] + dp[prev]);
                // buy a 30-day
                prev = max(i - 30, 0);
                minValue = min(minValue, costs[2] + dp[prev]);
                dp[i] = minValue;
            }
            else{
                // no need to travel today
                dp[i] = dp[i - 1];
            }
            //cout << "Day " << i << " : " <<  dp[i] << endl;
        }
        return dp[lastDay];
    }
};
