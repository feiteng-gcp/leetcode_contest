class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> d = {1,7,30};

        vector<int> res = days;
        res.push_back(0);
        for (int i = days.size()-1; i>=0; --i) {
            res[i]=1<<30;
            for (int k = 0; k < 3; ++k) {
                int x = i;
                while(x<days.size() && days[x]<days[i]+d[k]) ++x;
                res[i] = min(res[i], res[x] + costs[k]);
            }
        }
        return res[0];
    }
};
