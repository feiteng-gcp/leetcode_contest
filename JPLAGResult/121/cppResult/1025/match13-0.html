class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        // if(costs[0] * 7 < costs[1]) costs[1] = costs[0] * 7;
        // if(costs[0] * 30 < costs[2]) costs[2] = costs[0] * 30;
        // if(costs[1] * 5 < costs[2]) costs[2] = costs[1] * 5;
        // if(costs[0] * 2 + costs[1] * 4 < costs[2]) costs[2] = costs[0] * 2 + costs[1] * 4;
        
        int n = days.size();
        days.push_back(days.back() + 1);
        vector<int> dp(n + 1, INT_MAX);
        dp[n] = 0;
        for(int i = n - 1; i >= 0; i--){
            for(int j = n; j > i; j--){
                if(days[j] - days[i] < 30){
                    dp[i] = min(costs[2] + dp[j + 1], dp[i]);
                }
                if(days[j] - days[i] < 7){
                    dp[i] = min(costs[1] + dp[j + 1], dp[i]);
                }
            }
            dp[i] = min(dp[i + 1] + costs[0], dp[i]);
        }
        // for(int i : dp) cout<<i<<" ";
        return dp[0];
    }
};
