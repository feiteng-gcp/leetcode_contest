class Solution {
public:
    int mincostTickets(vector<int>& D, vector<int>& C) {
        D.push_back(10000);
        int N = D.size();
        vector<int> f(N);
        for (int i = N-2; i >= 0; --i) {
            int t = INT_MAX;
            for (int j = i+1; j < N; ++j) {
                if (D[j-1]-D[i] <= 0) {
                    t = min(t, C[0] + f[j]);
                    continue;
                }
                if (D[j-1]-D[i] <= 6) {
                    t = min(t, C[1] + f[j]);
                    continue;
                }
                if (D[j-1]-D[i] <= 29) {
                    t = min(t, C[2] + f[j]);
                    continue;
                } else {
                    break;
                }
            }
            f[i] = t;
        }
        return f[0];
    }
};
