class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> dp(days.size());
        for (int i = 0; i < days.size(); i++) {
            int best = costs[0] + (i > 0 ? dp[i-1] : 0);
            int j = i - 1;
            while (j >= 0 && days[j] > days[i] - 7) j--;
            best = min(best, costs[1] + (j >= 0 ? dp[j] : 0));
            while (j >= 0 && days[j] > days[i] - 30) j--;
            best = min(best, costs[2] + (j >= 0 ? dp[j] : 0));
            dp[i] = best;
        }
        return dp.back();
    }
};
