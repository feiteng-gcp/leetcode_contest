class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> opt;
        //sort(days.begin(), days.end());
        int min_cost = min(costs[0], costs[1]);

        min_cost = min(min_cost, costs[2]);
        opt.push_back(0);
        for (int i = 0; i < days.size(); ++i) {
            opt.push_back(opt[i] + min_cost);
            for (int j = i - 1; j >= 0; --j) {
                if ((days[i] - days[j] + 1) <= 7) {
                    opt[i + 1] = min(opt[i + 1], opt[j] + costs[1]);
                }
                if ((days[i] - days[j] + 1) <= 30) {
                    opt[i + 1] = min(opt[i + 1], opt[j] + costs[2]);
                } else {
                    break;
                }
            }
        }
        return opt[days.size()];
    }
};
