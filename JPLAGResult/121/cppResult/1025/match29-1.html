class Solution {
    int dp[1000];
    int n;
    
    int INF = 0x3f3f3f3f;
    
    int solve(int x, vector<int>& days, vector<int>& costs){
        if(dp[x]==-1){
            if(x>=n) return dp[x] = 0;
            

            dp[x] = costs[0] + solve(x+1, days, costs);
            
            int nx = x+1;
            while(nx<n&&days[nx]<=days[x]+6) nx++;
            dp[x] = min(dp[x], costs[1] + solve(nx, days, costs));
            
            nx = x+1;
            while(nx<n&&days[nx]<=days[x]+29) nx++;
            dp[x] = min(dp[x], costs[2] + solve(nx, days, costs));
            
        }
        return dp[x];
    }
    
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        memset(dp,-1,sizeof(dp));
        
        n = days.size();
        
        return solve(0,days,costs);
    }
};
