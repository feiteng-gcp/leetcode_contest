class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int dp[500],n=days.size();
        dp[0]=costs[0];
        for(int i=1;i<n;i++)
        {
            dp[i]=costs[0]+dp[i-1];
            int p=i;
            while(p>0 && days[i]-days[p-1]<=6)p--;
            if(p==0)dp[i]=min(dp[i],costs[1]);
            else dp[i]=min(dp[p-1]+costs[1],dp[i]);
            
            p=i;
            while(p>0 && days[i]-days[p-1]<=29)p--;
            if(p==0)dp[i]=min(dp[i],costs[2]);
            else dp[i]=min(dp[p-1]+costs[2],dp[i]);
        }
        return dp[n-1];
    }
};
