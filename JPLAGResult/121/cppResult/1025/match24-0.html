class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int n = days.size();
        vector<int> dp(n + 1);
        dp[n] = 0;
        dp[n - 1] = costs[0];
        for (int id = n - 2; id >= 0; --id) {
            int delta2 = 1;
            int delta3 = 1;
            while(id + delta2 < n && days[id + delta2] < days[id] + 7) ++delta2;
            while(id + delta3 < n && days[id + delta3] < days[id] + 30) ++delta3;
            
            dp[id] = dp[id + 1] + costs[0];
            dp[id] = min(dp[id], dp[id + delta2] + costs[1]);
            dp[id] = min(dp[id], dp[id + delta3] + costs[2]);
        }
        
        
        for (auto& x : dp) {
            cout << x << " ";
        }
        cout << endl;
        
        
        return dp[0];
        
    }
};
