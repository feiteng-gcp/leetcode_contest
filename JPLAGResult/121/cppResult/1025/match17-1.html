class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> f(376, 0);
        int cur = 0;
        for (int i = 1; i <= 375; i++)
        {
            if (cur == days.size() || days[cur] != i)
                f[i] = f[i-1];
            else
            {
                f[i] = min(min(f[max(0, i - 1)] + costs[0], f[max(0, i - 7)] + costs[1]), f[max(0, i - 30)] + costs[2]);
                cur++;
            }
        }
        return f[375];
    }
};
