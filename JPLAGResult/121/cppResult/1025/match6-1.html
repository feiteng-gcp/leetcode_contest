class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> f(366, 0), g(366, 0);
        for(int i = 0; i < days.size(); i++){
            g[days[i]] = 1;
        }
        for(int i = 1; i <= 365; i++){
            if(g[i] == 1){

                f[i] = f[i - 1] + costs[0];
                if(i - 7 >= 0){
                    f[i] = min(f[i], f[i - 7] + costs[1]);
                } else {
                    f[i] = min(f[i], costs[1]);
                }
                if(i - 30 >= 0){
                    f[i] = min(f[i], f[i - 30] + costs[2]);
                } else {
                    f[i] = min(f[i], costs[2]);
                }
            } else {
                f[i] = f[i - 1];
            }
        }
        return f[365];
    }
};
