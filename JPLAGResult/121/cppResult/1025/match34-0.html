class Solution {
public:
	int dp[400];
	bool b[400];
    int mincostTickets(vector<int> days, vector<int> costs) {
        memset(dp,0x3f,sizeof(dp));
        memset(b,0,sizeof(b));
        for(int i:days)b[i-1]=true;
        dp[0]=0;
        rep(i,365){
			if(!b[i])dp[i+1]=min(dp[i+1],dp[i]);
			else{
				dp[i+1]=min(dp[i+1],dp[i]+costs[0]);
				dp[i+7]=min(dp[i+7],dp[i]+costs[1]);
				dp[i+30]=min(dp[i+30],dp[i]+costs[2]);
			}
		}
		int Min=INF;
		for(int i=365;i<400;i++)Min=min(Min,dp[i]);
		return Min;
    }
};
