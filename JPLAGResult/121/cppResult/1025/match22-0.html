class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int dp[days.size() + 1];
        dp[0] = 0;
        for (int i = 0;i < days.size();i ++) {
            int ret = costs[0] + dp[i];
            int j = i - 1;
            while (j >= 0 && days[i] - days[j] < 7) j --;
            ret = min(ret, costs[1] + dp[j + 1]);
            while (j >= 0 && days[i] - days[j] < 30) j --;
            ret = min(ret, costs[2] + dp[j + 1]);
            dp[i + 1] = ret;
        }
        return dp[days.size()];
    }
};
