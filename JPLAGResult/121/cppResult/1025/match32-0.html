class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> dp(days.size());
        dp[0] = costs[0];
        for(int i=1;i<days.size();i++){
            int c30, c7;
            int j =i-1;
            for(j=i-1;j>=0;j--){
                if(days[i] - days[j]>6){
                    break;
                }
            }
            if(j<0){
                c7 = costs[1];
            }
            else{
                c7 = dp[j] + costs[1];
            }
            for(j=i-1;j>=0;j--){
                if(days[i] - days[j]>29){
                    break;
                }
            }
            if(j<0){
                c30 = costs[2];
            }
            else{
                c30 = dp[j] + costs[2];
            }
            dp[i] = min(dp[i-1]+costs[0],min(c30,c7));
        }
        return dp.back();
    }
};
