class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int N = days.size();
        vector<int> cost(N+1, 0);
        for(int day = N-1; day >= 0; day--) {
            auto it0 = upper_bound(days.begin(), days.end(), days[day]);
            cost[day] = costs[0] + cost[it0 - days.begin()];
            auto it1 = upper_bound(days.begin(), days.end(), days[day]+6);
            cost[day] = min(cost[day], costs[1] + cost[it1 - days.begin()]);
            auto it2 = upper_bound(days.begin(), days.end(), days[day]+29);
            cost[day] = min(cost[day], costs[2] + cost[it2 - days.begin()]);
        }
        return cost[0];
    }
};
