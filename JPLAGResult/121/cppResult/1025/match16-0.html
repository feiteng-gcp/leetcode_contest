class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {        
        int n = days.size();
        vector<int> f(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            f[i] = f[i - 1] + costs[0];
            for (int j = i - 1; j > 0; j--) {                
                int dif = 1 + days[i - 1] - days[j - 1]; 
                if (dif <= 7) f[i] = min(f[i], f[j - 1] + costs[1]);
                if (dif <= 30) f[i] = min(f[i], f[j - 1] + costs[2]);
                else break;
            }
        }
        return f[n];
    }
};
