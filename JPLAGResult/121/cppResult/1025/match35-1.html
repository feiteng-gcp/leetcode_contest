class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int rem[ 366 ] ;
        bool ist[ 366 ] ;
        for ( int i = 0 ; i < 366 ; i++ ){
            ist[ i ] = false ;
            rem[ i ] = 2147483647 ;
        }
        for ( int i = 0 ; i < days.size() ; i++ )
            ist[ days[ i ] ] = true ;
        rem[ 0 ] = 0 ;
        for ( int i = 1 ; i <= days[ days.size()-1 ] ; i++ ){
            if ( ist[ i ] == false ){
                rem[ i ] = rem[ i - 1 ] ;
                continue ;
            }
            int Min = rem[ i ] ;
            if ( i >= 1 && rem[ i-1 ] + costs[ 0 ] < Min )
                Min = rem[ i-1 ] + costs[ 0 ] ;
            else if ( i < 1 && costs[ 1 ] < Min )
                Min = costs[ 1 ] ;
            if ( i >= 7 && rem[ i-7 ] + costs[ 1 ] < Min )
                Min = rem[ i-7 ] + costs[ 1 ] ;
            else if ( i < 7 && costs[ 1 ] < Min )
                Min = costs[ 1 ] ;
            if ( i >= 30 && rem[ i-30 ] + costs[ 2 ] < Min )
                Min = rem[ i-30 ] + costs[ 2 ] ;
            else if ( i < 30 && costs[ 2 ] < Min )
                Min = costs[ 2 ] ;
            rem[ i ] = Min ;
        }
        //for ( int i = 0 ; i <= days[ days.size()-1 ] ; i++ )
          //  cout << i << "\t" << rem[ i ] << "\n" ;
        return rem[ days[ days.size()-1 ] ] ;
    }
};
