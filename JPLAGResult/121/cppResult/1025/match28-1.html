class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> ans(days.size(), 40000);
        costs[1] = min(costs[1], costs[2]);
        costs[0] = min(costs[0], costs[1]);
        costs.push_back(40000);
        for(int i = 0; i < days.size(); i += 1)
        	if(i == 0) ans[0] = costs[0];
        	else{
        		int j = i - 1, k = costs[0];
        		while(~j){
        			ans[i] = min(ans[i], ans[j] + k);
        			if(days[i] - days[j] >= 1) k = costs[1];
        			if(days[i] - days[j] >= 7) k = costs[2];
        			if(days[i] - days[j] >= 30) k = costs[3];
        			j -= 1;
				}
				ans[i] = min(ans[i], k);
			}
		return *ans.rbegin();
    }
};
