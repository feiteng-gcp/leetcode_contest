class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> dp = {0};
        
        for (int i = 0; i < days.size(); ++i) {
            int cost = dp.back() + costs[0];
            
            int j = i;
            while (j >= 0 && days[i] - days[j] < 7)
                --j;
            cost = std::min(cost, dp[j + 1] + costs[1]);
            
            while (j >= 0 && days[i] - days[j] < 30)
                --j;
            cost = std::min(cost, dp[j + 1] + costs[2]);
            
            dp.push_back(cost);
        }
        
        return dp.back();
    }
};
