class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int n = days.size();

        int *f = new int[n + 1];
        
        f[0] = 0;
        for (int i = 1; i <= n; ++i) {
            f[i] = INT_MAX;
            for (int j = 0; j < i; ++j) {
                int d = days[i - 1] - days[j] + 1;
                int cost = f[j];
                if (d > 30) cost = INT_MAX;
                else if (d > 7) cost += costs[2];
                else if (d > 1) cost += costs[1];
                else cost += costs[0];
                f[i] = min(f[i], cost);
            }
        }
        
        int ans = f[n];
        
        delete[] f;
        return ans;
    }
};
