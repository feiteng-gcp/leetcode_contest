class Solution {

public:
    using ll = long long;
    const int inf = (int)1e9;
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        vector<int> dp(days.size()+2, inf);
        dp[0] = 0;
        for(int i = 0;i < days.size();i++){
            int idx = lower_bound(days.begin(), days.end(), days[i]+1)-days.begin();
            dp[idx] = min(dp[idx], dp[i]+costs[0]);
            idx = lower_bound(days.begin(), days.end(), days[i]+7)-days.begin();
            dp[idx] = min(dp[idx], dp[i]+costs[1]);
            idx = lower_bound(days.begin(), days.end(), days[i]+30)-days.begin();
            dp[idx] = min(dp[idx], dp[i]+costs[2]);
        }
        return dp[days.size()];
    }
};
