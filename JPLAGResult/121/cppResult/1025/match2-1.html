class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {
        int n = days.size();
        vector<int> dp(n, INT_MAX);
        dp[0] = costs[0];

        for(int i = 1; i < n; i++){
            dp[i] = costs[0] + dp[i-1];
            for(int j = i-1; j >= max(0, i-29); j--){
                if(days[i] - days[j] <= 29){
                    dp[i] = min(dp[i], costs[2] + dp[j-1]);
                }
                if(days[i] - days[j] <= 6){
                    dp[i] = min(dp[i], costs[1] + dp[j-1]);
                }
            }
        }
        return dp[n-1];
    }
};
