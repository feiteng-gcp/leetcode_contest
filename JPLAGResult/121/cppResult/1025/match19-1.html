class Solution {
public:
    int mincostTickets(vector<int>& days, vector<int>& costs) {

        vector<int> V(days.back()+1);
        V[0] = 0;
        for(int i = 1, k = 0; i <= days.back() && k < days.size(); ++i) {
            if(days[k] == i) {
                V[i] = V[i-1]+costs[0];
                V[i] = min(V[i], (i>7?V[i-7]:0)+costs[1]);
                V[i] = min(V[i], (i>30?V[i-30]:0)+costs[2]);
                ++k;
            } else {
                V[i] = V[i-1];
            }
        }
        return V.back();
    }
};
