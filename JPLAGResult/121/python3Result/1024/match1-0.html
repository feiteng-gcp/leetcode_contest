class Solution:
    def countTriplets(self, A: 'List[int]') -> 'int':
        
        pairmap = {}
        
        for i in range(0, len(A)):
            for j in range(i, len(A)):
                
                p = A[i] & A[j]
                if p not in pairmap:
                    pairmap[p] = set()
                    
                pairmap[p].add((i,j))
                pairmap[p].add((j,i))
                
        counter = 0
        
        for k in range(0, len(A)):
            for p in pairmap:
                if p & A[k] == 0:
                    counter+= len(pairmap[p])
                    
        return counter
