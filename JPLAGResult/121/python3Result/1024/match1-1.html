class Solution:
    def countTriplets(self, A: 'List[int]') -> 'int':
        d = dict()
        count 
= 0
        for i in range(len(A)):
            for k in range(len(A)):
                n = A[i] & A[k]
                if n in d:
                    d[n] += 1
                else:
                    d[n] = 1
        for ki in d.keys():
            for Ai in A:
                if ki & Ai == 0:
                    count += d[ki]
        return count
