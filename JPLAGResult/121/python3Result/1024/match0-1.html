class Solution:
    def countTriplets(self, A):
        n = len(A)
        cnt = 0
        res_dict = {}
        for i in range(n):
            for j in range
(n):
                p = A[i] & A[j]
                if p in res_dict:
                    res_dict[p] += 1
                else:
                    res_dict[p] = 1
        for key in res_dict:
            for k in range(n):
                if A[k] & key == 0:
                    cnt += res_dict[key]
        return cnt
