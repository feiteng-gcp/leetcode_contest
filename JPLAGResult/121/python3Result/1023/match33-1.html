class TimeMap:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.tm = {}

    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        if key in self.tm:
            self
.tm[key].insert(0,(value, timestamp))
        else:
            self.tm[key] = [(value, timestamp)]
    def get(self, key: 'str', timestamp: 'int') -> 'str':
        if key in self.tm:
            for val,ts in self.tm[key]:
                if ts <= timestamp:
                    return val
            return ""
        else:
            return ""
