import collections
import bisect



class TimeMap:
    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.kv_store = collections.defaultdict(list)

    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        self.kv_store[key].append((timestamp, value))

    def get(self, key: 'str', timestamp: 'int') -> 'str':
        if key not in self.kv_store:
            return ''
        idx = bisect.bisect(self.kv_store[key], (timestamp, '~'))
        if idx == 0:
            return ''
        else:
            return self.kv_store[key][idx - 1][1]
