class TimeMap:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.hash_map = collections.defaultdict(list)
        

    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        self.hash_map[key].append((timestamp, value))
        # print(self.hash_map)
        

    def get(self, key: 'str', timestamp: 'int') -> 'str':
        if key not in self.hash_map:
            return ""
        elif timestamp < self.hash_map[key][0][0]:
            return ""
        elif timestamp >= self.hash_map[key][-1][0]:
            return self.hash_map[key][-1][1]
        else:
            my_list = [x[0] for x in self.hash_map[key]]
            i = bisect.bisect_right(my_list, timestamp)
            # print(i)
            return self.hash_map[key][i-1][1]
