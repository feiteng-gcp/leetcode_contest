class TimeMap:
    def __init__(self):
        self.keytotime = collections.defaultdict(list)
        self.keytimetoval = {}
        
        
    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        self.keytotime[key].append(timestamp)
        self.keytimetoval[(key, timestamp)] = value
        
        
        
        
    def get(self, key: 'str', timestamp: 'int') -> 'str':
        if key not in self.keytotime:
            return ""
        if len(self.keytotime[key]) == 0:
            return ""
        idx = bisect.bisect_right(self.keytotime[key], timestamp)
        if idx == 0:
            return ""
        else:
            t = self.keytotime[key][idx-1]
            return self.keytimetoval[(key, t)]
