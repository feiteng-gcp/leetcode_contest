import bisect
from collections import defaultdict

class TimeMap:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.k_v = defaultdict(list)
        self.k_v_i = defaultdict(
dict)

    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        self.k_v[key].append(timestamp)
        self.k_v_i[key][timestamp] = value

    def get(self, key: 'str', timestamp: 'int') -> 'str':
        arr = self.k_v[key]
        idx = bisect.bisect_right(arr, timestamp) - 1
        
if idx < 0:
            return ''
        else:
            time = self.k_v[key][idx]
            return self.k_v_i[key][time]
