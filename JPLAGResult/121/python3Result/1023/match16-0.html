class TimeMap:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.data = {}

    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        if key not in self.data: self.data[key] = [(value, timestamp)]
        else: self.data[key].append((value, timestamp))

    def get(self, key: 'str', timestamp: 'int') -> 'str':
        if key not in self.data: return ""
        curr = self.data[key]
        left, right, n = 0, len(curr)-1, len(curr)-1
        
        if timestamp >= curr[-1][1]: return curr[-1][0]
        if timestamp < curr[0][1]: return ""
        while left < right:
            mid = (left + right) // 2
            if timestamp >= curr[mid][1] and timestamp < curr[mid+1][1]: return curr[mid][0]
            if timestamp >= curr[mid+1][1]: left = mid + 1
            if timestamp < curr[mid][1]: right = mid
        return ""
