import bisect


class TimeMap:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.kv = {}
        self.max_value_encountered = ''
        

    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        if key not in self.kv: self.kv[key] = []
        bisect.insort(self.kv[key], [timestamp, value])
        if value > self.max_value_encountered: self.max_value_encountered = value
        

    def get(self, key: 'str', timestamp: 'int') -> 'str':
        if not self.kv[key]: return ''
        index = bisect.bisect_left(self.kv[key], [timestamp, self.max_value_encountered + 'z'])
        if index == 0: return ''
        return self.kv[key][index - 1][1]
