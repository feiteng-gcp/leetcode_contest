class TimeMap:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.dic = collections.defaultdict(list)
        

    def set(self, key: 'str', value: 'str', timestamp: 'int') -> 'None':
        self.dic[key].append([value, timestamp])
        

    def get(self, key: 'str', timestamp: 'int') -> 'str':
        if key not in self.dic or self.dic[key] == [] or timestamp < self.dic[key][0][1]:
            return ""
        timePosition = self.findFirstPositionGorE(timestamp, self.dic[key])
        return self.dic[key][timePosition][0]
        
    def findFirstPositionGorE(self, targetTime, numbers):
        start, end = 0, len(numbers) - 1
        while start + 1 < end:
            mid = start + (end - start) // 2
            if numbers[mid][1] < targetTime:
                start = mid
            elif numbers[mid][1] == targetTime:
                return mid
            else:
                end = mid
        if numbers[end][1] <= targetTime:
            return end
        return start
