class Solution:
    def strWithout3a3b(self, A, B):
        """
        :type A: int
        :type B: int
        :rtype: str
        """
        result = []
        while A + B > 0:
            if len(result) >= 2 and result[-1] == result[-2] == 'b':
                result.append('a')
                A -= 1
            elif len(result) >= 2 and result[-1] == result[-2] == 'a':
                result.append('b')
                B -= 1
            elif A > B:
                result.append('a')
                A -= 1
            else:
                result.append('b')
                B -= 1
        return "".join(result)
