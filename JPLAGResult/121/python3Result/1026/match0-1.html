class Solution:
    def strWithout3a3b(self, A, B):
        """
        :type A: int
        :type B: int
        :rtype: str
        """
        res = ''
        counterA = 0
        counterB = 0
        while A > 0 or B > 0:
            if A > B:
                if counterA >= 2:
                    res += 'b'
                    counterA = 0
                    counterB = 1
                    B -= 1
                else:
                    res += 'a'
                    counterA += 1
                    counterB = 0
                    A -= 1
            else:
                if counterB >= 2:
                    res += 'a'
                    counterA = 1
                    counterB = 0
                    A -= 1
                else:
                    res += 'b'
                    counterA = 0
                    counterB += 1
                    B -= 1
        return res
