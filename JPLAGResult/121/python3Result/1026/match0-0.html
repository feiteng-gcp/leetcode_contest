class Solution:
    def strWithout3a3b(self, A, B):
        """
        :type A: int
        :type B: int
        :rtype: str
        """
        a = 0
        b = 0
        res = ''
        while A > 0 or B > 0:
            if A > B:
                if a == 2:
                    res += 'b'
                    a = 0
                    b += 1
                    B -= 1
                else:
                    a += 1
                    b = 0
                    res += 'a'
                    A -= 1
            else:
                if b == 2:
                    res += 'a'
                    A -= 1
                    b = 0
                    a += 1
                else:
                    b += 1
                    a = 0
                    res += 'b'
                    B -= 1
        return res
