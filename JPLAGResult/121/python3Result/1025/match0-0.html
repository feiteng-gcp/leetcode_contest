class Solution:
    def mincostTickets(self, days: 'List[int]', costs: 'List[int]') -> 'int':
        dp = [0] * len(days)
        dp[0] = costs[0]
        for i in range(1, len(days)):
            duration = [1, 7, 30]
            pos = i
            for j in range(3):
                while pos >= 0 and days[pos] > days[i] - duration[j]:
                    pos -= 1
                new_cost = costs[j] + ( dp[pos] if pos >= 0 else 0 )
                if dp[i] == 0 or new_cost < dp[i]:
                    dp[i] = new_cost
        return dp[-1]
