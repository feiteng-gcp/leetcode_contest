class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        unordered_map<char, char> hash;
        for(int i = 0; i < order.size(); i++){
            hash[order[i]] = 'a' + i;
        }
        for(auto& str: words){
            for(int i = 0; i < str.size(); i++){
                str[i] = hash[str[i]];
            }
        }
        for(int i = 0; i < words.size() - 1; i++)
            if(words[i] > words[i+1]) return false;
        return true;
    }
};
