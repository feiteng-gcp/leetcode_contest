class Solution {
public:
    bool compare(string& str1, string& str2, vector<int>& letters) {
        int len1 = str1.size();
        int len2 = str2.size();
        for (int i=0; i<min(len1, len2); ++i) {
            if (letters[str1[i]-'a'] < letters[str2[i]-'a'])
                return true;
            else if (letters[str1[i]-'a'] > letters[str2[i]-'a'])
                return false;
        }
        return len1 <= len2;
    }
    
    bool isAlienSorted(vector<string>& words, string order) {
        vector<int> letters(26, -1);
        for (int i=0; i<order.size(); ++i)
            letters[order[i]-'a'] = i;
        for (int i=1; i<words.size(); ++i) {
            if (!compare(words[i-1], words[i], letters))
                return false;
        }
        return true;
    }
};
