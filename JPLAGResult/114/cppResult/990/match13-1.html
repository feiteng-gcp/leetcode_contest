class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {

        vector<int> v(26, 0);
        for(int i=0; i<order.length(); i++) {
            v[order[i]-'a'] = i;
        }
        for(int i=0; i<words.size()-1; i++) {
            int m = min(words[i].length(), words[i+1].length());

            bool substr = true;
            for(int j=0; j<m; j++) {
                if(v[words[i][j]-'a'] > v[words[i+1][j]-'a']) return false;
                else if(v[words[i][j]-'a'] < v[words[i+1][j]-'a']) {
                    substr = false;
                    break;
                }
            }
            if(substr) {
                if(words[i].length() > words[i+1].length()) return false;
            }
        }
        return true;
    }
};
