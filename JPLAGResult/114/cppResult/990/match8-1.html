class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        vector<int> mm(256);

        for(int i=0;i<order.size();i++) mm[order[i]]=i;
        vector<string> ww;
        for(int i=0;i<words.size();i++){
            string s="";
            for(int j=0;j<words[i].size();j++){
                s.push_back(mm[words[i][j]]+'a');
            }
            ww.push_back(s);

        }
        for(int i=1;i<ww.size();i++){
            if(ww[i].compare(ww[i-1])<0) return false;
        }
        return true;
    }
};
