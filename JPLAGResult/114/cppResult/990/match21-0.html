class Solution {
public:
	bool isAlienSorted(vector<string>& words, string order) {
		int n = words.size(), i, j, k, m;
		string oo = order;
		for (i = 0; i < 26; i++) {
			oo[order[i] - 'a'] = (char)('a' + i);
		}
		for (i = 0; i < n; i++) {
			for (j = 0; j < words[i].length(); j++) {
				words[i][j] = oo[words[i][j] - 'a'];
			}
		}

		for (i = 1; i < n; i++) {
			if (words[i - 1] > words[i]) return false;
		}

		return true;
	}
};
