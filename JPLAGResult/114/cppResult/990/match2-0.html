class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        vector<int> ord(128);
        for (int i = 0; i < order.size(); i++)
            ord[order[i]] = i + 'a';
        for (auto &s : words)
            for (char &c : s)
                c = ord[c];
        for (int j = 1; j < words.size(); j++)
            if (words[j] < words[j - 1])
                return false;
        return true;
    }
};
