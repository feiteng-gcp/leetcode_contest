class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        char serial[26] = {0};
        for(int i = 0; i < order.size(); i++) {
            serial[order[i]-'a'] = i;
        }

        for(int i = 1; i < words.size(); i++) {
            if(!less(words[i-1], words[i], serial)) {
                return false;
            }
        }

        return true;
    }

    bool less(const string& a,  const string& b, char* p) {
        int n = min(a.size(), b.size());
        for(int i = 0; i < n; i++) {
            int v = p[a[i]-'a'] - p[b[i]-'a'];
            if(v < 0) {
                return true;
            }
            else if(v > 0) {
                return false;
            }
        }

        if(a.size() > n) {
            return false;
        }

        return true;
    }

};
