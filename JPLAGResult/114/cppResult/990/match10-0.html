class Solution {
public:
    int compare(string& a, string& b, unordered_map<char, int>& charPos) {
        for (int i = 0; i < a.size() && i < b.size(); i++) {
            if (charPos[a[i]] < charPos[b[i]]) {
                return -1;
            } else if (charPos[a[i]] > charPos[b[i]]) {
                return 1;
            }
        }
        
        if (a.length() < b.length()) {
            return -1;
        }
        if (a.length() > b.length()) {
            return 1;
        }
        return 0;
    }
    
    bool isAlienSorted(vector<string>& words, string order) {
        unordered_map<char, int> charPos;
        for (int i = 0; i < order.length(); i++) {
            charPos[order[i]] = i;
        }
        
        for (int i = 0; i < words.size() - 1; i++) {
            if (compare(words[i], words[i + 1], charPos) > 0) {
                return false;
            }
        }
        return true;
    }
};
