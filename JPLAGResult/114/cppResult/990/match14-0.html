class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        int pri[26];
        for(int i = 0; i < 26; i++)
        {
            pri[order[i] - 'a'] = i;
        }
        for(int i = 0; i < words.size() - 1; i++)
        {
            if(!before(words[i], words[i + 1], pri))
            {
                return false;
            }
        }
        return true;
    }
    
    bool before(string a, string b, int * pri)
    {
        int i = 0;
        while(a[i] != 0 && b[i] != 0)
        {
            if(pri[a[i] - 'a'] < pri[b[i] - 'a'])
            {
                return true;
            }
            if(pri[a[i] - 'a'] > pri[b[i] - 'a'])
            {
                return false;
            }
            i++;
        }
        if(a[i] == 0 && b[i] != 0)
        {
            return true;
        }
        return false;
    }
};
