class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        vector<int> data(26,0);
        for(int i=0;i<order.size();i++){
            data[order[i]-'a']=i;
        }

        for(auto&s:words){
            for(int i=0;i<s.size();i++){
                s[i]=data[s[i]-'a']+'a';
            }
        }
        for(int i=1;i<words.size();i++){
            if(words[i]<words[i-1])
                return false;
        }
        return true;
    }
};
