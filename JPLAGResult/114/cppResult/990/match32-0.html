class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        map<int, int> mp;
        for (int i = 0; i < order.size(); i++) {
            mp[order[i]] = 'a' + i;
        }
        
        for (int i = 0; i < words.size(); i++) {
            for (int j = 0; j < words[i].size(); j++) {
                words[i][j] = mp[words[i][j]];
            }
        }
        
        vector<string> sw = words;
        sort(sw.begin(), sw.end());
        return sw == words;
    }
};
