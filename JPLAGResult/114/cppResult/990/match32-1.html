class Solution {
public:
    bool isGreater(string a,string b,map<char,int>& dict){
        int p = 0;
        while(true){
            if(p==a.size()&&p==b.size())
                return false;
            if(p==a.size())
                return false;
            if(p==b.size())
                return true;
            if(dict[a[p]]>dict[b[p]])
                return true;
            if(dict[a[p]]<dict[b[p]])
                return false;
            p++;
        }
    }
    bool isAlienSorted(vector<string>& words, string order) {

        map<char,int> dict;
        for(int i = 0 ; i < order.size();i++){
            dict[order[i]]=i;
        }
        for(int i = 0 ; i < words.size();i++){
            for(int j = i+1;j<words.size();j++){
                if(isGreater(words[i],words[j],dict))
                    return false;
            }
        }
        return true;
    }
};
