class Solution {
public:
	bool isAlienSorted(vector<string>& words, string order) {
		bool ret = false;

		vector<string> newwords;
		for (unsigned int i = 0; i < words.size(); i++)
		{
			newwords.emplace_back(ChangeToNormal(words[i], order));
		}
		vector<string> orderedwords(newwords);
		sort(orderedwords.begin(), orderedwords.end());
		ret = true;
		for (int i = 0; i < orderedwords.size(); i++)  //TODO  判断是否有序的更好办法
		{
			if (orderedwords[i] != newwords[i])
			{
				ret = false;
				break;
			}
		}

		return ret;
	}
	string ChangeToNormal(string str, string order)
	{
		string ret = "";
		string normalorder = "abcdefghijklmnopqrstuvwxyz";
		for (unsigned int i = 0; i < str.size(); i++)
		{
			int index = order.find(str[i]);
			ret += normalorder[index];
		}
		return ret;
	}
};
