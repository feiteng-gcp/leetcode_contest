class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        char orderIdx[26];
        for(int i=0;i<26;++i) {
            orderIdx[order[i]-'a'] = (i+'a');
        }
        
        string prev = words[0];
        for (int j=0;j<prev.size();++j) {
            prev[j] = orderIdx[prev[j]-'a'];
        }
        
        for (int i=1;i<words.size();++i) {
            string cur = words[i];
            for (int j=0;j<cur.size();++j) {
                cur[j] = orderIdx[cur[j]-'a'];
            }
            if (prev > cur) return false;
            prev = cur;
        }
        return true;
    }
    
};
