#include<bits/stdc++.h>
using namespace std;

class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        map<char,int> ma;
        for(int i=0;i<26;i++){
            ma[order[i]]=i;
        }
        for(int i=1;i<words.size();i++){
            bool t=0;
            for(int j=0;j<min(words[i].size(),words[i-1].size());j++){
                if(ma[words[i-1][j]]>ma[words[i][j]])return false;

                if(ma[words[i-1][j]]<ma[words[i][j]]){t=1;break;}
            }
            if(words[i-1].size()>words[i].size()&&!t)return false;
        }
        return true;
    }
};
