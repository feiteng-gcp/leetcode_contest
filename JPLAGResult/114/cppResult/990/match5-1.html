class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        unordered_map<char, char> map;
        for (int i = 0; i < order.size(); ++i) {
            map[order[i]] = i + 'a';
        }
        for (string& s : words) {

            for (int i = 0; i < s.size(); ++i) {
                s[i] = map[s[i]];
            }
        }
        for (int i = 0; i + 1 < words.size(); ++i) {
            if (words[i] > words[i+1]) {
                return false;
            }
        }
        return true;        
    }
};
