class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        unordered_map<char, int> dict;
        for (int i = 0; i < order.size(); ++i) {
            dict[order[i]] = i;
        }
        for (int i = 0; i < words.size()-1; ++i) {
            if (!less(dict, words[i], words[i+1])) {
                return false;
            }
        }
        return true;
    }
    
    bool less(unordered_map<char, int>& dict, string& s1, string& s2) {
        int p = 0;
        while (p < s2.size()) {
            if (p == s1.size() || dict[s1[p]] < dict[s2[p]]) {
                return true;
            } else if (dict[s1[p]] > dict[s2[p]]) {
                return false;
            } else {
                ++p;
            }
        }
        return false;
    }
};
