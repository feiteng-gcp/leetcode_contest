class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order) {
        int n=words.size();
        vector<char> cnt(26);

        for(int i=0; i<26; i++){
            cnt[order[i]-'a']=i+'a';
        }
        // for(int i=0; i<26; i++) cout<<char(i+'a')<<" "<<cnt[i]<<endl;
        for(int i=0; i<n; i++){
            for(int j=0; j<words[i].size(); j++){
                words[i][j]=cnt[words[i][j]-'a'];
            }
            // cout<<words[i]<<endl;
        }
        
        for(int i=1; i<n; i++){
            if(words[i]<words[i-1]) return false;
        }
        return true;
    }
};
