class Solution {
public:
    bool isAlienSorted(vector<string>& words, string order)
	{
		map<char,int> ma;
		for(int i=0;i!=order.length();++i)
		{
			ma[order[i]]=i;
		}
		for(int i=0;i!=words.size()-1;++i)
		{
			int la=words[i].length(),
				lb=words[i+1].length();
            bool flag=false;
			for(int j=0;j!=min(la,lb);++j)
			{
				if(ma[words[i][j]]>ma[words[i+1][j]])
					return false;
                else if(ma[words[i][j]]<ma[words[i+1][j]])
                {
                    flag=true; break;
                }
			}
			if(!flag && la>lb) return false;
		}
		return true;
	}

};
