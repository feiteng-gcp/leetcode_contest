class Solution {
public:
    char rep[10005];
    
    bool isAlienSorted(vector<string>& words, string order) {
        for (int i = 0; i < 26; i++) {
            rep[order[i]] = 'a' + i;
        }
        
        for (int i = 0; i < words.size(); i++) {
            for (int j = 0; j < words[i].size(); j++) {
                words[i][j] = rep[words[i][j]];
            }
        }
        
        for (int i = 0; i < words.size() - 1; i++) {
            if (words[i] > words[i + 1]) {
                return false;
            }
        }
        
        return true;
    }
};
