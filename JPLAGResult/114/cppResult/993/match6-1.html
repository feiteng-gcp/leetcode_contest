class Solution {
    int ans = 0;
    vector<int> sf_sums; 
    
    void dfs(const vector<int>& rods, int sum1, int sum2, int i) {
        if(i == rods.size()) {
            if(sum1 == sum2) {
                ans = max(ans, sum1);
            }
        } else {
            int rm = sf_sums[i];
            if(sum1 + sum2 + rm <= ans * 2) return;
            if(abs(sum1 - sum2) > rm) return;
            

            if(sum1 > sum2) {
                dfs(rods, sum1, sum2 + rods[i], i+1);
                dfs(rods, sum1 + rods[i], sum2, i+1);
            } else {
                dfs(rods, sum1 + rods[i], sum2, i+1);
                dfs(rods, sum1, sum2 + rods[i], i+1);
            }
            dfs(rods, sum1, sum2, i+1);
        }
    }

public:
    int tallestBillboard(vector<int>& rods) {
        if(rods.size() <= 1) return 0;
        sort(rods.begin(), rods.end(), std::greater<int>());

        sf_sums = rods;
        for(int i = (int)rods.size() - 2; i >= 0; i--) {
            sf_sums[i] = sf_sums[i+1] + rods[i];
        }
        
        dfs(rods, 0, 0, 0);
        return ans;
    }
};
