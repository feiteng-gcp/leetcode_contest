class Solution {
private:
    int n;
    int sum;
    
public:
    int tallestBillboard(vector<int>& rods) {
        n = rods.size();
        sum = 0;
        for (auto& a: rods)
            sum += a;
        vector<vector<int>> f(n + 1, vector<int>(2 * sum + 1, -1));

        f[0][sum] = 0;
        for (int i = 1; i <=n; i++)
        {
            int rod = rods[i-1];
            for (int j = 0; j <= 2 * sum + 1; j++)
            {
                f[i][j] = f[i-1][j];
                if (j > rod && f[i-1][j-rod] != -1)
                    f[i][j] = max(f[i][j], f[i-1][j-rod] + rod);
                if (f[i-1][j+rod] != -1)
                    f[i][j] = max(f[i][j], f[i-1][j+rod]);
            }
        }
        return f[n][sum] == -1 ? 0 : f[n][sum];
    }
};
