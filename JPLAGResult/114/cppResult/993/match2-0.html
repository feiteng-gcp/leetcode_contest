int pre[11][5001],rear[11][5001];
class Solution {
public:
    int tallestBillboard(vector<int>& rods) {
        int n=rods.size();
	if(n==0)return 0;
	int n1=n/2;
	int n2=n-n1;
	memset(pre,-1,sizeof(pre));
	memset(rear,-1,sizeof(rear));
	pre[0][0]=0;
	rear[0][0]=0;
	for(int i=0;i<n1;i++)
	{
		for(int j=0;j<=2500;j++)
		{
			pre[i+1][j]=max(pre[i+1][j],pre[i][j]);
			if(pre[i][j]==-1)continue;
			int news;
			news=j+rods[i];
			pre[i+1][news]=max(pre[i+1][news],pre[i][j]+rods[i]);
			news=j-rods[i];
			if(news<0)news=-news;
			pre[i+1][news]=max(pre[i+1][news],pre[i][j]+rods[i]);
		}
	}
	for(int i=0;i<n2;i++)
	{
		for(int j=0;j<=2500;j++)
		{
			rear[i+1][j]=max(rear[i+1][j],rear[i][j]);
			if(rear[i][j]==-1)continue;
			int news;
			news=j+rods[n1+i];
			rear[i+1][news]=max(rear[i+1][news],rear[i][j]+rods[n1+i]);
			news=j-rods[n1+i];
			if(news<0)news=-news;
			rear[i+1][news]=max(rear[i+1][news],rear[i][j]+rods[n1+i]);
		}
	}
	int ans=0;
	for(int i=0;i<=2500;i++)
	{
		//printf("%d:%d %d\n",i,pre[n1][i],rear[n2][i]);
		if(pre[n1][i]==-1||rear[n2][i]==-1)continue;
		ans=max(ans,(pre[n1][i]+rear[n2][i])/2);
	}
	return ans;
    }
};
