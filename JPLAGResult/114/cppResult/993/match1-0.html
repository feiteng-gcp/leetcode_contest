class Solution {
public:
    int tallestBillboard(vector<int>& rods) {
        vector<int> max_diff(5000 + 1, -1);
        max_diff[0] = 0;
        for (auto&& r : rods) {
            vector<int> temp = max_diff;
            for (int i = 0;i < temp.size();++i) {
                if (temp[i] == -1) continue;
                max_diff[i + r] = max(max_diff[i + r], temp[i] + r);
                if (i - r >= 0) {
                    max_diff[i - r] = max(max_diff[i - r], temp[i]);   
                } else {
                    max_diff[r - i] = max(max_diff[r - i], temp[i] + r - i);
                }
            }
        }
        return max_diff[0];
    }
};
