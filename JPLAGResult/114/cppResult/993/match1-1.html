class Solution {
public:
  int D[25][5005];
  int tallestBillboard(vector<int>& rods) {
    fill(D[0], D[0] + 5001, -1e9);
    D[0][0] = 0;
    for (int i = 0; i < rods.size(); i++) {
      for (int j = 0; j <= 5000; j++) {
        D[i + 1][j] = D[i][j];
        if (j + rods[i] <= 5000) {
          D[i + 1][j] = max(D[i + 1][j], D[i][j + rods[i]] + rods[i]);
        }

        int k = j - rods[i];
        if (k < 0) {
          D[i + 1][j] = max(D[i + 1][j], D[i][-k] + -k);
        } else {
          D[i + 1][j] = max(D[i + 1][j], D[i][k]);
        }
      }
    }
    return D[(int)rods.size()][0];
  }
};
