class Solution {
public:
    int tallestBillboard(vector<int>& rods) {
        //int a[20][5000];        //len, gap
        if(rods.size()<2) return 0;
        vector<vector<int>> a(20,vector<int>(5000,-1));
        a[0][0]=0;
        a[0][rods[0]]=0;
        for(int i=1;i<rods.size();i++){
            int l=rods[i];
            for(int j=0;j<5000;j++){
                if(a[i-1][j]==-1) continue;
                a[i][j]=max(a[i][j],a[i-1][j]);
                
                a[i][j+l] = max(a[i][j+l],a[i-1][j]);
                if(l<j){
                    a[i][j-l]=max(a[i][j-l], a[i-1][j]+l);
                }
                else{
                    a[i][l-j]=max(a[i][l-j], a[i-1][j]+j);
                }
            }
        }
        int res=0;
        for(int i=0;i<rods.size();i++){
            res=max(res,a[i][0]);
        }
        return res;
    }
};
