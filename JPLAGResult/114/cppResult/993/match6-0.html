class Solution {
private:
	int ans = -1;
	void tall(vector<int>& rods, int a, int b, int i, int sum) {
		if (a == b)
			ans = max(ans, a);
		if (max(a, b) + sum / 2 < ans || i >= rods.size() || abs(a - b) > sum)
			return;
		else {
			if (a > b) {
				tall(rods, a, b + rods[i], i+1, sum - rods[i]);
				tall(rods, a, b, i+1, sum - rods[i]);
				tall(rods, a + rods[i], b, i+1, sum - rods[i]);
			}
			else {
				tall(rods, a + rods[i], b, i+1, sum - rods[i]);
				tall(rods, a, b, i+1, sum - rods[i]);
				tall(rods, a, b + rods[i], i+1, sum - rods[i]);
			}
		}
	}
public:
	int tallestBillboard(vector<int>& rods) {
		sort(rods.rbegin(), rods.rend());
		int sum = 0;
		for (int i = 0; i < rods.size(); i++)
			sum += rods[i];
		tall(rods, 0, 0, 0, sum);
		return ans;
	}
};
