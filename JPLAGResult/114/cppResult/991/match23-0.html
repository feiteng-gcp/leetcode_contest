class Solution {
public:
        bool canReorderDoubled(vector<int>& A) {
	    map<int,int> ma;
	    for(auto it:A)
	    {
		    if(ma.find(it)==ma.end())
		    {
			    ma[it]=1;
		    }
		    else ma[it]++;
	    }
	    for(auto it:ma)
	    {
		    int key=it.first,
			    cnt=it.second;
		    if(cnt>0)
		    {
			    if(key<0)
			    {
				    if(key%2!=0) return false;
				    else
				    {
					    ma[key/2]-=ma[key];
					    ma[key]=0;    
				    }
			    }
			    else if(key>0)
			    {
				    ma[key*2]-=ma[key];
				    ma[key]=0;
			    }
			    else
			    {
				    if(cnt%2) return false;
				    else ma[key]=0;
			    }
		    }
		    else if(cnt<0) return false;   
	    }
	    return true;
    }


};
