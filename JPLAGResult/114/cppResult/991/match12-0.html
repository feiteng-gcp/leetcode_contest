class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int, int> mp1;
        for (auto e : A) mp1[e]++;
        for (auto &e : mp1) {
            //cout<<e.first<<" "<<e.second<<endl;
            if (e.second == 0) continue;
            if (e.first < 0) {
                if (e.first % 2 != 0) return false;
                if (mp1.count(e.first / 2) == 0 || mp1[e.first / 2] < e.second) return false;
                mp1[e.first / 2] -= e.second;
                e.second = 0;
            }
            else if (e.first > 0) {
                if (mp1.count(e.first * 2) == 0 || mp1[e.first * 2] < e.second) return false;
                mp1[e.first * 2] -= e.second;
                e.second = 0;
            }
            else {
                if (e.second % 2 != 0) return false;
                e.second = 0;
            }
        }
        return true;
    }
};
