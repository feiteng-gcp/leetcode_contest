class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        if(A.empty()) return true;
        sort(A.begin(), A.end());
        while(!A.empty())
        {
            int targetInd = -1;
            int curValue = A[0];   
            int halfValue = (curValue % 2 == 0)? curValue/2 : INT_MIN;

            int doubleValue = curValue*2;
            
            for(int i = 1; i < A.size(); i++)
            {
                if((A[i] <0 && A[i] == halfValue) || 
                   (A[i] >= 0 && A[i] == doubleValue))
                {
                    targetInd = i;
                    break;
                }
            }
            if(targetInd > 0)
            {
                A.erase(A.begin() + targetInd);
                A.erase(A.begin());
            }
            else
            {
                return false;
            }
        }
        return true;
        
    }
};
