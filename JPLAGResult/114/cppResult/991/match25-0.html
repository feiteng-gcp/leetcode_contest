class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int, int, std::greater<int>> minus;
        map<int, int> plus;
        int cnt0 = 0; 
        
        for(int i = 0; i<A.size(); i++){
            if(A[i]>0) plus[A[i]]++; 
            else if(A[i]<0) minus[A[i]]++; 
            else cnt0++;
        }
        if (cnt0 & 1) return false; 
        // if(plus.size()%2||minus.size()%2) return false;
        
        for (auto item : plus){
            int val = item.first; 
            int cnt = item.second; 
            if (cnt == 0) continue; 
            if (plus[2 * val] < cnt) return false; 
            plus[2 * val] -= cnt; 
        }

        for (auto item : minus){
            int val = item.first; 
            int cnt = item.second; 
            if (cnt == 0) continue; 
            if (minus[2 * val] < cnt) return false; 
            minus[2 * val] -= cnt; 
        } 
        return true; 
    }
};
