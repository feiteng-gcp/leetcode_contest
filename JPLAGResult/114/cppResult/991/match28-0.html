class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        vector<int> s;
        s.resize(100001);
        for(int i=0;i<A.size();i++)
        {
            if(A[i]<0)
                A[i]=0-A[i];
        }
        sort(A.begin(),A.end());
        int size = 0;       
        for(int i=0;i<A.size();i++)
        {
            if(A[i]%2==0)
            {
                int temp = A[i]/2;
                if(s[temp]>0)
                {
                    s[temp]--;
                    size--;
                    continue;
                }
                
            }
            size++;
            s[A[i]]++;           
        }
        return size==0;
    }
};
