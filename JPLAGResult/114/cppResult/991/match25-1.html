class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int,int> mp1,mp2;
        int z=0;

        for(int n:A){
            if(n==0) z++;
            else if(n>0) mp1[n]++;
            else mp2[-n]++;
        }
        if(z&1) return 0;
        for(auto& pr:mp1){
            int a=pr.first;
            int b=pr.second;
            if(b==0) continue;
            mp1[a*2]-=b;
            if(mp1[a*2]<0) return 0;
        }for(auto& pr:mp2){
            int a=pr.first;
            int b=pr.second;
            if(b==0) continue;
            mp2[a*2]-=b;
            if(mp2[a*2]<0) return 0;
        }
        return 1;
    }
};
