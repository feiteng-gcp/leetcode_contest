class Solution {
public:
  bool canReorderDoubled(vector<int>& A) {

    vector<int> neg(100001), pos(100001);
    int zero = 0;
    for (int n : A) {
      if (n < 0) neg[-n]++;
      else if (n == 0) zero++;
      else pos[n]++;
    }
    if (zero % 2) return false;
    for (int i = 1; i <= 50000; i++) {
      if (neg[i] > 0) {
        if (neg[i] > neg[2 * i]) return false;
        neg[2 * i] -= neg[i];
        neg[i] = 0;
      }
      if (pos[i] > 0) {
        if (pos[i] > pos[2 * i]) return false;
        pos[2 * i] -= pos[i];
        pos[i] = 0;
      }
    }
    for (int i = 1; i <= 100000; i++) {
      if (neg[i] > 0 || pos[i] > 0) return false;
    }
    return true;
  }
};
