class Solution {
public:
    static bool cmp(int a, int b){
        if(a >= 0 && b >= 0){
            return a < b;
        }
        if(a < 0 && b < 0){
            return a > b;
        }
        else{
            return a < b;
        }
    }
    
    bool canReorderDoubled(vector<int>& A) {
        map<int, int> m;
        sort(A.begin(), A.end(), cmp);
        int cnt = 0;
        
        for(int i = 0;i < A.size();i++){
            //cout << A[i] << endl;
            if(A[i] % 2 == 1){
                cnt++;
                m[A[i]]++;

            }
            else{
                if(m.count(A[i]/2) && m[A[i]/2] > 0){
                    m[A[i]/2]--;
                    cnt--;
                }
                else{
                    cnt++;
                    m[A[i]]++;
                }
            }
        }
        return cnt == 0;
    }
};
