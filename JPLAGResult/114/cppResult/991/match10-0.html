class Solution {
public:
    
    void remove(map<int, int>& db, int x) {
        db[x]--;
        if (db[x] == 0)
            db.erase(x);
    }
    bool canReorderDoubled(vector<int>& A) {
        map<int, int> db, db2;
        for (int x : A)
            if (x >= 0)
                db[x]++;
            else
                db2[-x]++;
        while (!db.empty()) {
            int x = db.begin()->first;
            remove(db, x);
            if (db.find(2 * x) == db.end())
                return false;
            remove(db, 2 * x);
        }
        while (!db2.empty()) {
            int x = db2.begin()->first;
            remove(db2, x);
            if (db2.find(2 * x) == db2.end())
                return false;
            remove(db2, 2 * x);
        }
        return true;
    }
};
