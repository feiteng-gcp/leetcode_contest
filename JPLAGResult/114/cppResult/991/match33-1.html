bool cmp(int x, int y) { return abs(x) < abs(y); }

class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int, int> h;
        bool flag = false;
        sort(A.begin(), A.end(), cmp);
        
        for (auto x: A)
            if (h.find(x) != h.end())
                h[x]++;
            else
                h[x] = 1;
        if (h[0] & 1) {
            if (A.size() & 1) flag = true;
            else return false;
        }
        
        int total = 0;
        
        for (int i = 0; i < (int)A.size(); i++) if (A[i] != 0) {
            ++ total;
            if (h[A[i]] > 0) {
                if (h[A[i] * 2] > 0) {
                    h[A[i]] --;
                    h[A[i]*2] --;
                } else {
                    if (!flag) flag = true;
                    else return false;
                }
            }
        }
        
        return true;
    }
};
