class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        sort(begin(A), end(A));
        unordered_map<int, int> numCount;
        int cnt = 0;
        for(int a : A){
            if(a < 0){
                if(numCount[a*2] > 0){
                    numCount[a*2] --;
                    cnt --;
                }else{
                    numCount[a] ++;
                    cnt ++;
                }
            }else{
                if(a%2 == 0 && numCount[a/2] > 0){
                    numCount[a/2] --;
                    cnt --;
                }else{
                    numCount[a] ++;
                    cnt ++;
                }
            }
        }
        return cnt == 0;
    }
};
