class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        if (A.empty())
            return true;
        unordered_map<int, int> ms;
        for (int a : A)

            ms[a]++;
        
        int cnt = 0;
        sort(A.begin(),A.end());
        for (int i = 0; i < A.size(); i++){
            if (ms[A[i]] > 0 && ms.find(2 * A[i]) != ms.end() && ms[2 * A[i]] > 0){
                ms[2 * A[i]]--;
                ms[A[i]]--;
                cnt ++;
            }

        }
        if (cnt == A.size() / 2)
            return true;
        return false;
        
    }
};
