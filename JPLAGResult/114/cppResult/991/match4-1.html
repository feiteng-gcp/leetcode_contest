class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        sort(A.begin(),A.end());
        map<int,int>m;
        for(int i=0;i<A.size();i++)
            m[A[i]]++;
        
        for(int i=0;i<A.size();i++)
            if(m[A[i]]>0 && m[2*A[i]]>0){
                m[A[i]]--;
                m[2*A[i]]--;
            }
        
        for(int i=0;i<A.size();i++)
            if(m[A[i]]>0)return false;
        return true;
    }
};
