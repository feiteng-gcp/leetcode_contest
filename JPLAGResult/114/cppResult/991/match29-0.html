class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        sort(A.begin(), A.end());
        unordered_map<int, int> h;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] > 0) {
                if (h[A[i]] > 0) h[A[i]]--;
                else h[A[i] * 2]++;
            }
        }
        reverse(A.begin(), A.end());
        for (int i = 0; i < A.size(); i++) {
            if (A[i] < 0) {
                if (h[A[i]] > 0) h[A[i]]--;
                else h[A[i] * 2]++;
            }
        }
        for (auto x : h)
            if (x.second != 0) return false;
        return true;
    }
};
