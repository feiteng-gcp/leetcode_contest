class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        sort(A.begin(), A.end(), [](int a, int b){return abs(a) < abs(b) || abs(a) == abs(b) && a < b;});
        unordered_map<int, int> um;
        for (int i : A)
            um[i]++;
        for (int i = 0; i < A.size(); i++) {
            if (!um.count(A[i]))
                continue;
            if (um.count(A[i] * 2)) {
                if (!--um[A[i]]) um.erase(A[i]);
                if (!--um[A[i] * 2]) um.erase(A[i] * 2);
            }
            else {
                return false;
            }
        }
        return um.empty();
    }
};
