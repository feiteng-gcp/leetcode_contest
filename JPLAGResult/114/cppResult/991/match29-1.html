class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        sort(A.begin(), A.end());
        map<int, int> freq;
        
        for (int i = 0; i < A.size(); i++) {
            freq[A[i]]++;
        }
        
        for (int i = 0; i < A.size(); i++) {
            if (freq[A[i]] == 0) continue;
            
            int doubleVal;
            if (A[i] < 0) doubleVal = A[i] / 2;
            else doubleVal = A[i] * 2;

            if (freq[doubleVal] < freq[A[i]]) return false;
            freq[doubleVal] -= freq[A[i]];
            freq[A[i]] = 0;
        }
        
        return true;
    }
};
