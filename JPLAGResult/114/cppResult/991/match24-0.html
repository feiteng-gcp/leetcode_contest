class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        sort(A.begin(), A.end(), [](int a, int b){ return abs(a) < abs(b);});
        map<int, int> num;
        for(auto a : A) num[a]++;

        for(auto a : A){
            if(num[a] == 0) continue;
            num[a]--;
            if(num[2*a] > 0){
                num[2*a]--;
            }else{
                return false;
            }
        }
        return true;
    }
};
