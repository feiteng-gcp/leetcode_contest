class Solution {
public:
   bool canReorderDoubled(vector<int>& A) {
        sort(A.begin(),A.end(),[](const int&a,const int&b)->bool{if (abs(a) == abs(b)) return a < b;return abs(a) < abs(b);});
        map<int,int> vis;
        for (int i = 0 ; i < A.size() ; ++i)
        	++vis[A[i]];
        
		for (int i = 0 ; i < A.size() ; ++i)
            if (vis[A[i]]){
            	if (vis[A[i] * 2]){
            		--vis[A[i]];
					--vis[2 * A[i]];
				}
				else return false;
			}
        return true;
    }
};
