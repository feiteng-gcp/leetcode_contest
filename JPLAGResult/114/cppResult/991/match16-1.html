class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        vector<int> pos;
        vector<int> neg;
        map<int,int> bag;
        for(int i:A){
            if(i>=0)
                pos.push_back(i);
            else
                neg.push_back(i);
            bag[i]++;
        }
        sort(pos.begin(),pos.end());
        sort(neg.begin(),neg.end());
        for(int i:pos){
            if(bag[i]==0)
                continue;
            if(bag[2*i]>0){
                bag[i]--;
                bag[2*i]--;
            }else{
                return false;
            }
        }

        for(int i=neg.size()-1;i>=0;i--){
            if(bag[neg[i]]==0)
                continue;
            if(bag[2*neg[i]]>0){
                bag[neg[i]]--;
                bag[2*neg[i]]--;
            }else{
                return false;
            }
        }
        return true;
    }
};
