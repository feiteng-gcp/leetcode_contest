class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        multiset<int> plus;
        multiset<int> minus;
        int zero = 0;

        for (int i = 0; i < int(A.size()); i++) {
            if (A[i] == 0)
                zero++;
            else if (A[i] > 0)
                plus.insert(A[i]);
            else
                minus.insert(-A[i]);
        }

        if ((zero & 1) || (plus.size() & 1) || (minus.size() & 1))
            return false;

        while (!plus.empty()) {
            int x = *plus.begin();
            auto it = plus.find(x * 2);
            if (it == plus.end())
                return false;
            plus.erase(it);
            plus.erase(plus.begin());
        }

        while (!minus.empty()) {
            int x = *minus.begin();
            auto it = minus.find(x * 2);
            if (it == minus.end())
                return false;
            minus.erase(it);
            minus.erase(minus.begin());
        }

        return true;
    }
};
