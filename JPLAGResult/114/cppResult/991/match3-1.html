class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        multiset<int> pos, neg;
        int zcnt = 0;
        for(int i : A) {
            if(i == 0) zcnt++;
            else if(i < 0) neg.insert(-i);
            else pos.insert(i);
        }
        
        if(zcnt & 1) return false;
        if(neg.size() & 1) return false;

        if(pos.size() & 1) return false;
        
        while(!neg.empty()) {
            int cur = *neg.begin();
            neg.erase(neg.begin());
            auto it = neg.find(2 * cur);
            if(it == neg.end()) return false;
            neg.erase(it);
        }
        
        while(!pos.empty()) {
            int cur = *pos.begin();
            pos.erase(pos.begin());
            auto it = pos.find(cur * 2);
            if(it == pos.end()) return false;
            pos.erase(it);
        }
        
        return true;
    }
};
