bool compAbs(int& a, int& b) { return abs(a) < abs(b); }

class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        sort(A.begin(), A.end(), compAbs);
        unordered_map<int, int> B;
        bool pass = false;
        
        for (int i: A) ++B[i];
        
        if (A.size() % 2 == 1) pass = true;
        
        int score = 0;
        for (int i: A)
        {
            if (B[i] <= 0) continue;
            if (B[2*i] <= 0) continue;
            --B[i];
            --B[2*i];
            ++score;
        }
        return score == A.size() / 2;
    }
};
