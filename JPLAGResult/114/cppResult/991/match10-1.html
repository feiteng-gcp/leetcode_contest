class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        multiset<int> AA;
        multiset<int> BB;
        for( int i=0 ; i<A.size() ; i++ )
            if ( A[i]>=0 ) AA.insert(A[i]);
            else BB.insert(-A[i]);
        if ( AA.size()%2!=0||BB.size()%2!=0 ) return false;

        while( !AA.empty() )
        {
            int a = *(AA.begin());
            AA.erase(AA.begin());
            if ( AA.find(2*a)==AA.end() ) return false;
            AA.erase(AA.find(2*a));
        }
        while( !BB.empty() )
        {
            int a = *(BB.begin());
            BB.erase(BB.begin());
            if ( BB.find(2*a)==BB.end() ) return false;
            BB.erase(BB.find(2*a));
        }
        return true;    
    }
};
