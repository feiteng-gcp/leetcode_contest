class Solution {
public: 
    bool canReorderDoubled(vector<int>& A) {
        if(!A.size()) return true;
        map<int, int> mmp;
        for(int i : A) mmp[i]++;
        
        for(auto i : mmp)
        {
            //cout << i.first << endl;
            while(i.second)
            {
                if(i.first >= 0 && mmp.count(i.first*2) && mmp[i.first*2] > 0)
                {
                    mmp[i.first*2]--;
                    i.second--;
                }
                else if(i.first < 0 && i.first%2 == 0 && mmp.count(i.first/2) && mmp[i.first/2] > 0)
                {
                    mmp[i.first/2]--;
                    i.second--;
                }
                else return false;
            }
        }
        return true;
    }
};
