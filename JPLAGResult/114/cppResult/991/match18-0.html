class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int,int> numCountsPos,numCountsNeg;
        
        for (int i=0;i<A.size();++i) {
            if (A[i]>=0) {
                numCountsPos[A[i]]++;
            } else {
                numCountsNeg[-A[i]]++;
            }
        }
        
        for (auto it = numCountsPos.begin();it!= numCountsPos.end(); ++it) {
            if (it->second > 0) {
                auto it2 = numCountsPos.find(it->first*2);
                if (it2==numCountsPos.end() || it2->second < it->second) return false;
                it2->second -= it->second; it->second = 0;
            }
        }

        for (auto it = numCountsNeg.begin();it!= numCountsNeg.end(); ++it) {
            if (it->second > 0) {
                auto it2 = numCountsNeg.find(it->first*2);
                if (it2==numCountsNeg.end() || it2->second < it->second) return false;
                it2->second -= it->second; it->second = 0;
            }
        }

        return true;
    }
};
