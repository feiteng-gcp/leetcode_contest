class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int, int> pos, neg;
        int zero = 0;
        for (auto&& a : A) {
            if (a > 0) ++pos[a];
            else if (a < 0) ++neg[-a];
            else ++zero;
        }
        if (zero % 2 != 0) return false;
        for (auto&& p : pos) {
            if (p.second > 0) {
                auto iter = pos.find(p.first * 2);
                if (iter == pos.end() || iter->second < p.second) return false;
                iter->second -= p.second;
            }
        }
        for (auto&& p : neg) {
            if (p.second > 0) {
                auto iter = neg.find(p.first * 2);
                if (iter == neg.end() || iter->second < p.second) return false;
                iter->second -= p.second;
            }
        }
        return true;
    }
};
