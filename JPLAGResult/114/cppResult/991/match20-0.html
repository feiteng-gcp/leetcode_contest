class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int,int>mert;
        mert.clear();
        for(int i=0;i<A.size();i++)mert[A[i]]++;
        sort(A.begin(),A.end());
        int ans=0;
        for(int i=0;i<A.size()&&A[i]<=0;i++){
            if(mert[A[i]*2]>0){
                mert[A[i]]--;
                mert[A[i]*2]--;
                ans+=2;
            }
        }
        for(int i=A.size()-1;i>=0&&A[i]>0;i--){
            if(mert[A[i]*2]>0){
                mert[A[i]]--;
                mert[A[i]*2]--;
                ans+=2;
            }
        }
        return (ans==A.size());
    }
};
