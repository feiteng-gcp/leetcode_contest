class Solution {
public:
bool canReorderDoubled(vector<int>& A) {
		bool ret = true;
		//首先正负分开
		vector<int> Pvec;
		vector<int> Nvec;
		for (unsigned int i = 0; i < A.size(); i++)
		{
			if (A[i] < 0)
			{
				Nvec.emplace_back(-1*A[i]);
				//Pvec.emplace_back(-1*A[i]);
			}
			else
			{
				Pvec.emplace_back(A[i]);
			}
		}
		if (Pvec.size() % 2 != 0)
		{
			return false;
		}
		sort(Pvec.begin(), Pvec.end());
		int limit = Pvec.size() / 2;
		for (unsigned int i = 0; i < limit;i++)
		{
			if (find(Pvec.begin() + i + 1, Pvec.end(), 2 * Pvec[i]) != Pvec.end())
			{
				Pvec.erase(find(Pvec.begin() + i + 1, Pvec.end(), 2 * Pvec[i]));
			}
			else
			{
				return false;
			}
		}
		sort(Nvec.begin(), Nvec.end());
		limit = Nvec.size() / 2;
		for (unsigned int i = 0; i < limit;i ++)
		{
			if (find(Nvec.begin() + i + 1, Nvec.end(), 2 * Nvec[i]) != Nvec.end())
			{
				Nvec.erase(find(Nvec.begin() + i + 1, Nvec.end(), 2 * Nvec[i]));
			}
			else
			{
				return false;
			}
		}
		return true;
	}
};
