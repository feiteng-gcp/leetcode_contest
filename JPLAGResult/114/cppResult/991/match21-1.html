class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int,int> mp1;

        map<int,int> mp2;
        int numZero=0;
        for(auto &i:A)
        {
            if(i<0)mp1[i]++;
            else if(i>0)mp2[i]++;
            else numZero++;
        }
        if(numZero%2!=0)return false;
        for(auto it=mp2.begin();it!=mp2.end();it++)
        {
            if(it->second==0)continue;
            if(mp2.find((it->first)*2)==mp2.end()||mp2[(it->first)*2]<it->second)return false;
            else mp2[(it->first)*2]-=it->second;
        }
        for(auto it=mp1.rbegin();it!=mp1.rend();it++)
        {
            if(it->second==0)continue;
            if(mp1.find((it->first)*2)==mp1.end()||mp1[(it->first)*2]<it->second)return false;
            else mp1[(it->first)*2]-=it->second;
        }
        return true;
    }
};
