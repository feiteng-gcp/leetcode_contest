class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int, int> M;
        
        for (int a : A) M[a]++;        
        
        for (map<int,int>::iterator it = M.begin() ; it != M.end() ; ++it) {                    
            int a = it->first, count = it->second;
            //cout << a << " " << count << endl;
            if (count == 0) continue;
            it->second = 0;
            
            if (a == 0) {
                if (count % 2) return false;
                else continue;
            }            
            
            if (a < 0) {
                if (a % 2) return false;
                auto it2 = M.find(a/2);
                if (it2 == M.end() || it2->second == 0) return false;
                
                if (it2->second < count) return false;                
                it2->second -= count;                
            }
            else {
                auto it2 = M.find(a*2);
                if (it2 == M.end() || it2->second == 0) return false;
                if (it2->second < count) return false;
                it2->second -= count;                
            }
        }
        
        for (auto &p : M) {
            if (p.second != 0)
                return false;
        }
        return true;
    }
    
    bool canReorderDoubled2(vector<int>& A) {
        multiset<int> S(A.begin(), A.end());
        
        for (int a : A) {                        
            if (a % 2) {
                auto it = S.find(a*2);
                cout << a <<endl;
                if (it == S.end()) return false;
                S.erase(it);
            }
            else {
                int tmp = a/2;
                auto it = S.find(tmp);
                if (it != S.end()) {
                    S.erase(it);
                }
                else {
                    auto it = S.find(a*2);
                    cout << a <<endl;
                    if (it == S.end()) return false;
                    S.erase(it);                    
                }
            }
        }
        
        return S.empty();
    }
};
