class Solution {
public:
    static bool cmp(int v1, int v2) {
        if (v1<0 && v2<0) return v1 > v2;
        else return v1 < v2;
    }
    
    bool canReorderDoubled(vector<int>& A) {
        int n = A.size();
        if (!n) return true;
        map<int, int> dict;
        for (int i=0; i<n; i++) dict[A[i]] ++;
        sort(A.begin(), A.end(), cmp);
        
        //for (int i=0; i<n; i++) cout << A[i] << " ";
        //cout << endl;
        
        for (int i=0; i<n; i++) {
            if (dict[A[i]] == 0) continue;
            //cout << i <<" : " << A[i] << "  => " << dict[A[i]*2] << endl;
            if (dict.find(A[i]*2) != dict.end() && dict[A[i]*2] != 0) {
                dict[A[i]*2] --;
                dict[A[i]] -- ;
            }
            else {
                //cout << i <<" : " << A[i] << "  => " << dict[A[i]*2] << endl;
                return false;
            }
        }
        return true;
    }
};
