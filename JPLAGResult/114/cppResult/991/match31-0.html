class Solution {
    public:
        bool canReorderDoubled(vector<int>& A) {
            unordered_map<int, int> occur;
            sort(A.begin(), A.end());
            for(int x : A)
                occur[x]++;
            for(int x : A) {
                if(occur[x] > 0 && occur[2 * x] > 0){
                    occur[x]--;
                    occur[2 * x]--;
                }
            }

            int sum = 0;
            for(auto& entry : occur)
                sum += entry.second;

            return sum == 0;

        }
};
