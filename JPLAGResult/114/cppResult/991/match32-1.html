class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        int n=A.size();
        if(n==0) return true;
        vector<int> An, Ap;
        for(int i=0; i<n; i++){
            if(A[i]<0) An.push_back(-A[i]);
            else if(A[i]>0) Ap.push_back(A[i]);
        }

        if(An.size()%2 || Ap.size()%2) return false;
        else return (helper(An) & helper(Ap));
    }
    
    bool helper(vector<int>& A){
        int n=A.size();
        if(n==0) return true;
        sort(A.begin(),A.end());
        //printVec(A);
        map<int,int> mp;
        for(int i=0; i<n; i++){
            if(mp.count(A[i])) mp[A[i]]++;
            else mp[A[i]]=1;
        }
        for(auto it=mp.begin(); it!=mp.end(); ++it){
            //cout << "Now checking " << it->first << endl;
            if(it->second==0) continue;
            else if(mp.count((it->first)*2) && mp[(it->first)*2]>=mp[(it->first)]) 
                mp[(it->first)*2]-=mp[(it->first)];
            else return false;
        }
        return true;
    }
    
    void printVec(vector<int>& A){
        for(int i=0; i<A.size(); i++) cout << A[i] << " ";
        cout << endl;
    }
};
