class Solution {
public:
    bool canReorderDoubled(vector<int>& a) {
        multiset<int>s1,s2;
        for(int i=0;i<a.size();i++)
        {
            if(a[i]>0)
                s1.insert(a[i]);
            else
                s2.insert(-a[i]);
        }
        while(s1.size()>0){
            int num=*s1.begin();
            s1.erase(s1.begin());
            if(s1.find(2*num)==s1.end()){
                return false;
            }
            s1.erase(s1.find(2*num));
        }
        while(s2.size()>0){
            int num=*s2.begin();
            s2.erase(s2.begin());
            if(s2.find(2*num)==s2.end()){
                return false;
            }
            s2.erase(s2.find(2*num));
        }
        return true;
    }
};
