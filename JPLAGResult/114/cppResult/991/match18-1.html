class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        map<int, int> count;
        for (int i = 0; i < A.size(); i++) {
            if (count.find(abs(A[i])) == count.end()) {
                count[abs(A[i])] = 1;
            }
            else {
                count[abs(A[i])] += 1;
            }
        }

        if (count[0] % 2 != 0) {
            return false;
        }
        count.erase(0);

        for (auto it = count.begin(); it != count.end(); it++) {
            int number = it -> first;
            if (count[number * 2] < count[number]) {
                return false;
            }
            count[2 * number] -= count[number];
            count[number] = 0;
        }

        for (auto it = count.begin(); it != count.end(); it++) {
            if (it -> second != 0) {
                return false;
            }
        }

        return true;
    }
};
