class Solution {
public:
    bool canReorderDoubled(vector<int>& v) {
        sort(v.begin(), v.end());
        map<int,int>m;
        for(auto num:v) m[num]++;
        for(auto it=m.begin();it!=m.end();it++){
            if(it->second==0) continue;
            if(it->first>=0 && it->second>0 && m[2*it->first]>=it->second){
                m[2*it->first]-=it->second;
                it->second=0;
            }
            else if(it->first<0 && it->second>0 && (it->first%2)==0 && m[it->first/2]>=it->second){
                m[it->first/2]-=it->second;
                it->second=0;
            }
            else return false;
        }
        return true;
    }
};
