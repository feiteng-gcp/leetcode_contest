bool _c_m_p(int a, int b) {
    return abs(a) < abs(b);
}

class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        unordered_map < int, int > cnt;
        sort(A.begin(), A.end(), _c_m_p);
        for (int i = 0; i < A.size(); i++) cnt[A[i]]++;
        for (int i = 0; i < A.size(); i++) {
            if (cnt[A[i]] == 0) continue;
            if (cnt[2*A[i]] == 0) return false;
            cnt[A[i]]--;
            cnt[2*A[i]]--;
        }
        return true;
    }
};
