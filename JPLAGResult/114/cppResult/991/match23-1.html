class Solution {
public:
    bool canReorderDoubled(vector<int>& A) {
        vector<int> pos(100001, 0);
        vector<int> neg(100001, 0);
        for(auto num : A){
        	if(num >= 0){
        		pos[num] += 1;
        	}
        	else neg[-num] += 1;
        }
        for(int i = 0; i <= 50000; i++){
        	pos[2 * i] -= pos[i];
        	neg[2 * i] -= neg[i];
        	if(pos[2 * i] < 0 || neg[2 * i] < 0)return false;
        	pos[i] = 0;
        	neg[i] = 0;
        }
        for(int i =0; i <= 100000; i++){
        	if(pos[i] != 0 || neg[i] != 0)return false;
        }
        return true;
    }
};
