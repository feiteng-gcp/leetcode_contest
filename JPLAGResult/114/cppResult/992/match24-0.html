class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int ans = 0;
        int size = A.size();
        if(size == 1) return ans;
        int len = A[0].length();
        for(int j = 0; j < len; j++)
        {
            int i = 0;
            while(i < size - 1)
            {
                if(A[i].substr(0, j) == A[i + 1].substr(0, j) && A[i][j] > A[i + 1][j])break;
                i++;
            }
            if(i < size - 1)
            {
                for(i = 0; i < size; i++) A[i][j] = (j == 0)? 'a' : A[i][j - 1];
                ans++;
            }
        }
        return ans;
    }
};
