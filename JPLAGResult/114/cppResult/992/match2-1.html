class Solution {
public:
    int minDeletionSize(vector<string>& A) {

        int res = 0;
        int n = A.size();
        vector<bool> done(n, false);
        for (int i = 0; i < A[0].size(); i++)
        {
            bool needDelete = false;
            for (int j = 1; j < n; j++)
            {
                if (done[j]) continue;
                if (A[j][i] < A[j-1][i])
                {
                    res++;

                    needDelete = true;
                    break;
                }
            }
            if (!needDelete)
            {
                for (int j = 1; j < n; j++)
                {
                    if (A[j][i] > A[j-1][i])
                    {
                        done[j] = true;
                    }
                }
            }
        }
        return res;
    }
};
