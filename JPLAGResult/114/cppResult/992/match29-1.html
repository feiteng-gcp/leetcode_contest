class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        bool vis[101];
        int res = 0;
        memset(vis, false, sizeof(vis));
        if (A.size() == 0) {
            return res;

        }
        for (int i = 0; i < A[0].length(); i++) {
            bool flag = true;
            for (int j = 0; j < A.size() - 1; j++) {
                if (A[j][i] > A[j+1][i] && !vis[j]) {
                    flag = false;
                    break;
                }
            }
            if (!flag) {
                res ++;
            } else {
                flag = true;
                for (int j = 0; j < A.size() - 1; j++) {
                    if (A[j][i] < A[j+1][i]) {
                        vis[j] = true;
                    }
                    if (!vis[j]) {
                        flag = false;
                    }
                }
            }
            if (flag) {
                break;
            }
        }
        return res;
    }
};
