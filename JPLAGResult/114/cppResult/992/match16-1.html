class Solution {
public:
    int minDeletionSize(vector<string>& a) {
        int n = a.size();
        if (n < 1) return 0;
        int m = a[0].size();

        int ret = 0;
        for (int i = 0; i < m; ++i) {
            bool ok = true;
            for (int j = 1; j < n; ++j) {
                if (a[j].substr(0, i + 1) < a[j - 1].substr(0, i + 1)) {
                    ok = false;
                    break;

                }
            }
            if (!ok) {
                for (int j = 0; j < n; ++j) {
                    a[j] = a[j].substr(0, i) + a[j].substr(i + 1);
                }
                --i;
                --m;
                ++ret;
            }
        }
        return ret;
    }
};
