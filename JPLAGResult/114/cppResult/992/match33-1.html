class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        bool visited[1001];
        memset(visited,0,sizeof(visited));
        
        for(int caso=0;caso<100;caso++){

            for(int i=0;i+1<A.size();i++){
                int n=A[i].size();
                for(int k=0;k<n;k++){
                    if(visited[k]==1)continue;
                    
                    if(A[i][k]>A[i+1][k]){
                        visited[k]=1;
                    }
                    
                    if(A[i][k]<A[i+1][k]){
                        break;
                    }
                    
                }
            }
        }
        int cont=0;
        for(int i=0;i<A[0].size();i++)
            if(visited[i]==1)cont++;
        return cont;
    }
};
