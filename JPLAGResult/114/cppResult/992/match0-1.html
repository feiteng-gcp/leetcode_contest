class Solution {
public:
	//dp
	int minDeletionSize(vector<string>& A) {
		int ret = 0;
		vector<string> strs;
		if (A.size() <= 1)
		{
			return 0;
		}
		if (A[0].size() < 1)
		{
			return 0;
		}
		for (unsigned int i = 0; i < A.size(); i++)
		{
			string tmp = "";
			tmp += A[i][0];
			strs.emplace_back(tmp);
		}
		int neednextcheck = 0;
		bool isordered;
		isordered = IsOrdered(strs, neednextcheck);
		if (isordered == true && neednextcheck == 0)
		{
			return ret;
		}
		if (isordered == false)
		{
			ret++;
			
			for (unsigned int i = 0; i < A.size(); i++)
			{
				string tmp = "";	
				strs[i]=tmp;
			}
		}
		for (unsigned int i = 1; i < A[0].size(); i++)
		{
			vector<string> tmpstrs(strs);
			for (unsigned int j = 0; j < A.size(); j++)
			{
				tmpstrs[j] += A[j][i];
			}
			isordered = IsOrdered(tmpstrs, neednextcheck);
			if (isordered == true && neednextcheck == 0)
			{
				return ret;
			}
			if (isordered == false)
			{
				ret++;
			}
			else
			{
				strs = tmpstrs;
			}
		}
		return ret;
	}
	bool IsOrdered(vector<string> strs,int& neednextcheck)
	{
		bool ret = true;
		vector<string> OrderedStrs(strs);
		//string tmp = strs[0];
		neednextcheck = 0;
		for (unsigned int i = 1; i < strs.size(); i++)
		{
			if (strs[i] == strs[i - 1])
			{
				neednextcheck = 1;
				break;
			}
		}
		sort(OrderedStrs.begin(), OrderedStrs.end());		
		for (unsigned int i = 0; i < strs.size(); i++)
		{
			if (OrderedStrs[i] != strs[i])
			{
				return false;
			}
		}

		return ret;
	}
};
