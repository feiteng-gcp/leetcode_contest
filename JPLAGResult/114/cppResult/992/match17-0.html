class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        if (A.empty()) return 0;
        int const wlen = A.front().length();
        if (wlen == 0) return 0;
        int idx = 0;
        int res = 0;
        int const wnum = A.size();
        vector<string> B(wnum, "");
        while (idx < wlen) {
            bool ordered = true;
            for (int i = 1; i < wnum; i++) {
                if ((B[i-1] + A[i-1][idx]) > (B[i] + A[i][idx])) {
                    ordered = false;
                    break;
                }
            }
            if (ordered) {
                for (int i = 0; i < wnum; i++) {
                    B[i] += A[i][idx];
                }
            } else {
                res ++;
            }
            idx ++;
        }
        return res;
    }
};
