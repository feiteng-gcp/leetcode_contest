class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        if(A.size() < 2) return 0;
        

        int n = A.size(), m = A[0].size();
        vector<string> res(n, "");
        for(int j=0;j<m;j++) {
            bool found = false;
            for(int i=1;i<n;i++) {
                if(res[i-1] < res[i] || (res[i-1] == res[i] && A[i-1][j] <= A[i][j])) continue;
                found = true; break;
            }
            if(!found) {
                for(int i=0;i<n;i++) res[i].push_back(A[i][j]);
            }
        }
        return A[0].size() - res[0].size();
    }
};
