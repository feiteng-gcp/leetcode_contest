class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int result = 0;
        unordered_map<string, vector<string>> groups;
        groups[" "] = A;
        for (int j = 0; j < A[0].size(); j++) {
            if (groups.empty()) {
                return result;
            }
            unordered_map<string, vector<string>> newGroups;
            bool terminate = true;
            bool modGroups = true;
            for (auto p : groups) {
                auto s = p.first;
                auto group = p.second;
                if (group.size() == 1) {
                    continue;
                }
                for (int i = 0; i < group.size(); i++) {
                    newGroups[s + group[i][j]].push_back(group[i]);
                    if (i == group.size() - 1) {
                        break;
                    }
                    if (group[i][j] > group[i + 1][j]) {
                        terminate = false;
                        modGroups = false;
                        result++;
                        break;
                    }
                    if (group[i][j] == group[i + 1][j]) {
                        terminate = false;
                    }
                }
                if (modGroups == false) {
                    break;
                }
            }
            if (terminate) {
                break;
            }
            if (modGroups) {
                groups = newGroups;
            }
        }
        return result;
    }
};
