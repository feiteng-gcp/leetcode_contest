class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int rtn = 0, n = A[0].size();
        
        for (int i = 0 ; i < n ; ++i) {
            bool del = false;
            int idx = -1;
            for (int j = 1 ; j < A.size() ; ++j) {
                //cout << A[j] << " " << A[j-1] << endl;
                if (A[j] < A[j-1]) {
                    //cout << "del " << endl;
                    
                    for (int k = 0 ; k < A[j].size() ; ++k) {
                        if (A[j][k] < A[j-1][k]){
                            idx = k;
                            break;
                        }
                    }
                    
                    del = true;
                    break;
                }
            }
            
            if (del) {
                rtn++;
                for (string &a : A) a = a.substr(0, idx) + a.substr(idx+1);
            }
            else 
                return rtn;            
        }
        return rtn;
    }
};
