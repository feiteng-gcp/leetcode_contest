class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        if (A.size() == 0 || A[0].size() == 0)
            return 0;
        
        int mds = 0, ss = A[0].size();
        vector<vector<int>> ordered;
        for (int i = 0; i < A.size()-1; i++)
        {
            vector<int> vec;
            for (int j = 0; j < ss; j++)
            {
                if (A[i][j] < A[i+1][j])
                {
                    vec.push_back(1);
                }
                else if (A[i][j] == A[i+1][j])
                {
                    vec.push_back(0);
                }
                else
                {
                    vec.push_back(-1);
                }
            }
            ordered.push_back(vec);
        }
        
        bool deletec;

        for (int i = 0; i < ss; i++)
        {
            deletec = false;
            for (int j = 0; j < A.size()-1; j++)
            {
                if (ordered[j][i] == -1)
                {
                    mds++;
                    deletec = true;

                    break;
                }
            }
            if (!deletec)
            {
                for (int j = 0; j < A.size()-1; j++)
                {
                    if (ordered[j][i] == 1)
                    {
                        for (int k = i+1; k < ss; k++)
                            ordered[j][k] = 1;
                    }
                }
            }
        }
        
        return mds;
    }
};
