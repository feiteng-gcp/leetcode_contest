class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int N = int(A.size());
        if (N == 0)
            return 0;

        int L = int(A[0].length());
        if (L == 0)
            return 0;

        vector<bool> chk(N);

        int res = 0;
        for (int i = 0; i < L; i++) {
            vector<bool> c = chk;
            bool erased = false;
            for (int j = 1; j < N; j++) {
                if (c[j - 1])
                    continue;

                if (A[j - 1][i] > A[j][i]) {
                    res++;
                    erased = true;
                    break;
                } else if (A[j - 1][i] < A[j][i]) {
                    c[j - 1] = true;
                }
            }
            if (!erased)
                chk = c;
        }

        return res;
    }
};
