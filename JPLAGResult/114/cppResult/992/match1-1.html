class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size();
        int m = A[0].length();
        vector<bool> skip(n, false);
        int ans = 0;
        for (int j = 0; j < m; j++)
        {
            vector<bool> this_skip(n, false);
            bool deleted = false;
            for (int i = 0; i < n - 1; i++)
            {
                if (skip[i])
                    continue;
                if (A[i][j] < A[i+1][j])
                {
                    this_skip[i] = true;
                }
                else if (A[i][j] > A[i+1][j])
                {
                    ans++;
                    deleted = true;
                    break;
                }
            }
            if (!deleted)
            {

                for (int i = 0; i < n - 1; i++)
                {
                    if (this_skip[i])
                        skip[i] = true;
                }
            }
        }
        return ans;
    }
};
