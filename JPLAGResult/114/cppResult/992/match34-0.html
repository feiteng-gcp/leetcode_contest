class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        vector<bool> flag(A.size(), false);
        int ans = 0;
        for(int i = 0; i < A[0].size(); i++) {
            vector<bool> new_flag(flag);
            int j = 1;
            
            for(; j < A.size(); j++) {
                
                if(flag[j]) {
                    new_flag[j] = flag[j];
                    continue;
                }

                char c = A[j-1][i] - A[j][i];
                if(c < 0) {
                    new_flag[j] = true;
                    continue;
                }

                if(c > 0) {
                    break;
                }
            }

            if(j < A.size()) {
                ans++;
                continue;
            }

            flag = new_flag;
        }

        return ans;
    }

};
