class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        size_t N = A.size();
        size_t L = A[0].size();
        vector<char> sorted(N, 0), _sorted(N, 0);
        

        int min_d = 0;
        for(int i = 0; i < L; i ++){
            bool keep = true;
            for(int j = 1; j < N; j ++){
                if(A[j][i] > A[j - 1][i]){
                    _sorted[j] = 1;
                }
                else if(A[j][i] < A[j - 1][i] && !sorted[j]){
                    keep = false;

                    break;
                }
            }
            if(keep){
                for(int i = 0; i < N; i ++){
                    sorted[i] = sorted[i] || _sorted[i];
                }
            }
            else{
                min_d ++;
            }
            fill(_sorted.begin(), _sorted.end(), 0);
            // printf("%d %d\n", i, keep);
        }
        return min_d;
    }
};
