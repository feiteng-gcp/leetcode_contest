class Solution {
public:
  int minDeletionSize(vector<string>& A) {
    int N = (int)A.size();
    int M = (int)A[0].size();
    vector<bool> smaller(N);
    int res = 0;
    int R = (int)A.size(), C = (int)A[0].size();

    for (int i = 0; i < C; i++) {
      bool del = false;
      for (int j = 1; j < R; j++) {
        if (!smaller[j - 1] && A[j - 1][i] > A[j][i]) {
          del = true;
          break;
        } 
      }
      if (del) res++;
      else {

        for (int j = 1; j < R; j++) {
          if (A[j - 1][i] < A[j][i]) smaller[j - 1] = true;
        }
      }
    }
    return res;
  }
};
