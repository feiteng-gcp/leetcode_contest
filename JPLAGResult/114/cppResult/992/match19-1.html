class Solution {
public:
    vector<int> nc;
    int n,m;
    int minDeletionSize(vector<string>& A) {
        n=A.size();
        m=A[0].length();

        int ans=0;
        nc.assign(n-1,1);
        for(int j=0;j<m;j++)
        {
            int f=0;
            for(int i=0;i<n-1;i++)
            {
                if(!nc[i]) continue;
                if(A[i][j]>A[i+1][j])
                {
                    f=1;
                    ans++;
                    break;
                }
            }
            if(!f)
            {
                for(int i=0;i<n-1;i++)
                    if(A[i][j]<A[i+1][j])
                        nc[i]=0;
            }
            
        }
        return ans;
    }
};
