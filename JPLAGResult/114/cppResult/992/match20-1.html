class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        if(A.empty()) return 0;
        int result = 0;

        int m = A.size(), n = A[0].size();
        vector<bool> check(m, true);
        for(int i = 0; i < n; i++) {
            vector<bool> tempcheck(m, false);
            bool delte = false;
            bool end = true;
            for(int j = 1; j < m; j++) {
                if(check[j]==false)
                    continue;
                if(A[j][i] < A[j-1][i]) {
                    delte = true;
                    result++;
                    break;
                }
                else if(A[j][i] == A[j-1][i]) {
                    tempcheck[j] = true;
                    end = false;
                }
            }
            if(delte==false){
                if(end)
                    break;
                check = tempcheck;
            }
        }
        return result;
    }
};
