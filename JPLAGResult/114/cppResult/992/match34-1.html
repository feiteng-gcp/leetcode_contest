class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int res = 0;
        //vector<int> v(A.size(), 0);
        bitset<200> b, tmp;
        for(int i = 0; i < A[0].length(); ++i)
        {
            int flag = 1;
            tmp.reset();
            for(int j = 1; j < A.size(); ++j)
            {
                if(b[j]) continue;
                if(A[j-1][i] < A[j][i]) tmp[j] = 1;
                if(A[j-1][i] > A[j][i] && flag) res++, flag = 0;
            }
            if(flag) b |= tmp;
        }
        return res;
    }
};
