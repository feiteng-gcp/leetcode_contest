class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int ans = 0;
        int ok[105];
        for (int i = 0; i < A.size(); i++) {
            ok[i] = false;
        }
        
        for (int i = 0; i < A[0].size(); i++) {
            bool need = false;
            for (int j = 0; j < A.size() - 1; j++) {
                if (!ok[j] && A[j][i] > A[j + 1][i]) {
                    need = true;
                    break;
                }
            }
            
            if (need) {
                ans++;
            } else {
                for (int j = 0; j < A.size() - 1; j++) {
                    if (A[j][i] < A[j + 1][i]) {
                        ok[j] = true;
                    }
                }
            }
        }
        
        return ans;
    }
};
