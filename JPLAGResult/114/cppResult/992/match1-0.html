class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size();
        int m = A[0].length();
        
        int num_used = 0;
        vector<bool> safe(n-1, false);
        
        for (int i = 0; i < m; i++) {
            bool can_use = true;
            for (int j = 0; j < n - 1; j++) {
                if (safe[j] || A[j][i] <= A[j+1][i])
                    continue;
                can_use = false;
                break;
            }
            
            //cout << "can use " << can_use << endl;
            
            if (can_use) {
                num_used++;
                for (int j = 0; j < n-1; j++) {
                    if (A[j][i] < A[j+1][i])
                        safe[j] = true;
                }
            }
        }
        
        return m - num_used;
    }
};
