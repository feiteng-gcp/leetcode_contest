class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        if (A.size() == 0) return 0;
        
        vector<bool> okAtPos(A.size(), false);

        int res = 0;
        
        for (int i = 0; i < A[0].length(); i++) {
            vector<bool> okRightNow(A.size(), false);
            
            for (int j = 0; j < A.size() - 1; j++) {
                if (!okAtPos[j] && A[j][i] > A[j + 1][i]) {
                    res++;
                    okRightNow.assign(A.size(), false);
                    break;
                } else if (A[j][i] < A[j + 1][i]) {
                    okRightNow[j] = true;
                }
            }
            
            for (int j = 0; j < A.size(); j++) {
                okAtPos[j] = okAtPos[j] or okRightNow[j];
            }
        }
        
        return res;
    }
};
