class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        if (A.empty())
            return 0;
        int ans = A[0].size();
        vector<string> g(A.size());
        for (int idx = 0; idx < A[0].size(); ++idx) {
            string s = "";
            for (int i = 0; i < A.size(); ++i)
                g[i] += A[i][idx];
            if (is_sorted(g.begin(), g.end()))  {
                ans--;
            } else {
                for (auto &x : g)
                    x.pop_back();
            }
        }
        return ans;
    }
};
