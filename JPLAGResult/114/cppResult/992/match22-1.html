class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size();
        vector<string> cur(n,"");
        for(int i = 0 ; i < A[0].size();i++){
            vector<string> newV(cur);
            
            for(int j = 0;j<n;j++){
                newV[j]+=A[j].substr(i,1);
            }
            bool good=true;
            for(int j = 0;j<n-1;j++){
                if(newV[j]>newV[j+1]){
                    good=false;
                    break;
                }
            }
            if(good)
                cur=newV;
        }
        return A[0].size()-cur[0].size();
    }
};
