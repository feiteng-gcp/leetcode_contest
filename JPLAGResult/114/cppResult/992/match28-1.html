class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size(), m = A[0].length();
        vector<bool> del(m);

        int ans = 0;
        for (int i = 0; i < m; ++i) {
        	bool ok = true;
            
        	for (int j = 1; j < n; ++j)
        		if (bad(A[j-1], A[j], i, del)) {
        			ok = false;
        			break;
        		}

        	if (!ok) {
        		++ans;
        		del[i] = true;
        	}
        }

        return ans;
    }

    bool bad(string& a, string& b, int lim, vector<bool>& del) {
    	for (int i = 0; i <= lim; ++i)
    		if (!del[i]) {
    			if (a[i] > b[i])
    				return true;
    			else
    				if (a[i] < b[i])
    					return false;
    		}
        
        return false;
    }
};
