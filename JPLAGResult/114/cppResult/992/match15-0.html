class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int N = A.size(), L = A[0].size();
        int ans = 0;
        bitset<100> diff;
        for(int j=0; j<L; j++){
            bool ok = true;
            for(int i=0; i<N-1; i++){
                if(!diff[i] && A[i][j] > A[i+1][j]){
                    ok = false;
                    break;
                }
            }
            if(ok){
                for(int i=0; i<N-1; i++){
                    if(A[i][j] < A[i+1][j]){
                        diff[i] = true;
                    }
                }
            }else{
                ans++;
            }
        }
        return ans;   
    }
};
