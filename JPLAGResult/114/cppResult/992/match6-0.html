class Solution {
public:
    int less[105];
    int minDeletionSize(vector<string>& A) {
        memset(less, 0, sizeof(less));
        int ans = 0;
        int n = A.size();
        int m = A[0].length();
        for(int i = 0;i < m;i++){
            bool ok = true;
            for(int j = 0;j < n - 1;j++){
                if(less[j] == 1) continue;
                else{
                    if(A[j][i] <= A[j + 1][i])
                        continue;
                    else
                        ok = false;
                }
            }
            if(!ok)
                ans++;
            else{
                for(int j = 0;j < n - 1;j++){
                    if(A[j][i] < A[j + 1][i])
                        less[j] = 1;
                }
            }
        }
        return ans;
    }
};
