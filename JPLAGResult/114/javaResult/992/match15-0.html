class Solution {
    public int minDeletionSize(String[] A) {

		StringBuilder strings[] = new StringBuilder[A.length];

		for (int i = 0; i < A.length; i++) {
			strings[i] = new StringBuilder();
		}

		int n = A[0].length();
		int res = 0;

		for (int i = 0; i < n; i++) {

			boolean flag = false;
			for (int j = 0; j < A.length; j++) {
				if (j > 0) {
					String la = strings[j - 1].toString() + A[j - 1].charAt(i);
					String lb = strings[j].toString() + A[j].charAt(i);
					if (la.compareTo(lb) > 0) {
						flag = true;
						break;
					}
				}
 			}

			if (flag) {
				res++;
			} else {
				for (int j = 0; j < A.length; j++) {
					strings[j].append(A[j].charAt(i));
				}
			}
		}
		return res;

	}
}
