class Solution {
    public int minDeletionSize(String[] A) {
if(A.length == 1) return 0;
        
int len = A[0].length();
        boolean[][] record = new boolean[A.length][A.length];
        int res = 0;
        for(int i=0;i<len;i++){
            boolean delete = false;
            boolean[][] cur = new boolean[A.length][A.length];
            for(int j=1;j<A.length;j++){
                if(record[j-1][j]) continue;
                if(A[j-1].charAt(i) > A[j].charAt(i)) {
                    res++;
                    delete =true;
                    break;
                }
                
if(A[j-1].charAt(i) < A[j].charAt(i)){
                    cur[j-1][j] = true;
                }
            }
            if(!delete){
                for(int j=1;j<A.length;j++){
                    if(cur[j-1][j]) record[j-1][j] = true
;
                }
            }
        }
        return res;
    }
}
