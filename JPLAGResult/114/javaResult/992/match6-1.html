class Solution {
    public int minDeletionSize(String[] A) {
        int res = 0;
        if (A == null || A.length == 0) {
            return 0;
        }
        HashSet<Integer> noNeedToConsider = new HashSet<>();
        for (int j = 0; j < A[0].length(); j++) {
            boolean needDelete = false;
            for (int i = 1; i < A.length; i++) {
                if (!noNeedToConsider.contains(i) && A[i].charAt(j) < A[i - 1].charAt(j)) {
                    needDelete = true;
                    break;
                
}
            }
            if (needDelete) {
                res++;
            } else {
                for (int i = 1; i < A.length; i++) {
                    if (A[i].charAt(j) > A[i - 1].charAt(j)) {
                        if (!needDelete) {
                            noNeedToConsider.add(i);
                        }
                    
}
                }
            }
        }
        return res;
    }
}
