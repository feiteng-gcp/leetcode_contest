class Solution {
    public int minDeletionSize(String[] words) {
        if(words.length == 0) return 0;
        
        Set<Integer> wordsToCompare = new HashSet<>();
        for(int i=1; i<words.length; ++i) wordsToCompare.add(i);
        
        int n = words[0].length();
        int result = 0;
        
        for(int i=0; i<n; ++i) {
            boolean toBeDeleted = false;
            for(int wordIndex : wordsToCompare) {
                if(words[wordIndex-1].charAt(i) <= words[wordIndex].charAt(i)) continue;
                
                toBeDeleted = true;
                break;
            }
            if(toBeDeleted) {
                ++result;
            } else {
                for(int wordIndex : new HashSet<>(wordsToCompare)) {
                    if(words[wordIndex-1].charAt(i) != words[wordIndex].charAt(i))
                        wordsToCompare.remove(wordIndex);
                }
            }
        }
        return result;
    }
}
