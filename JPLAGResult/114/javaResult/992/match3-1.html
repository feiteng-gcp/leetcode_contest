class Solution {
    public int minDeletionSize(String[] A) {
        int n = A.length;
        int m = A[0].length();
        boolean[] f = new boolean[n - 1];
        
int ans = 0;
        for (int i = 0; i < m; i++) {
            boolean del = false;
            for (int j = 0; j < n - 1; j++) {
                if (!f[j] && A[j].charAt(i) > A[j + 1].charAt(i)) {
                    del = true;
                }
            
}
            if (del) {
                ans++;
            } else {
                for (int j = 0; j < n - 1; j++) {
                    if (!f[j] && A[j].charAt(i) < A[j + 1].charAt(i)) {
                        f[j] = true;
                    
}
                }
            }
        }
        return ans;
    }
}
