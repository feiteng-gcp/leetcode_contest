class Solution {
    public int minDeletionSize(String[] A) {
        int count = 0;
        for (int i = 0; i < A[0].length(); i++) {
            if (!isSorted(A, i + 1)) {
                A = delete(A, i);
                count++;
                i--;
            
}
        }
        return count;
    }

    private boolean isSorted(String[] arr, int end) {
        for (int i = 1; i < arr.length; i++) {
            if (arr[i - 1].substring(0, end).compareTo(arr[i].substring(0, end)) > 0) {
                return false;
            
}
        }
        return true;
    }

    private String[] delete(String[] arr, int index) {
        String[] deleted = new String[arr.length];
        for (int i = 0; i < arr.length; i++) {
            deleted[i] = arr[i].substring(0, index) + arr[i].substring(index + 1);
        }
        return deleted;
    }
}
