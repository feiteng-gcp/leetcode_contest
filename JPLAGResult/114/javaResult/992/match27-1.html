class Solution {
    public int minDeletionSize(String[] A) {
        int count = 0;
        int len = A.length, wordLen = A[0].length();
        boolean[] al = new boolean[len];
        al[0] = true;
        r:
    	for (int i = 0; i < wordLen; ++i) {
    		boolean[] temp = new boolean[len];
    		for (int j
 = 1;j < len; ++j) {
    			if (!al[j] && A[j-1].charAt(i) > A[j].charAt(i)) {
    				++count;
    				continue r;
    			}
    			if (A[j-1].charAt
(i) < A[j].charAt(i)) {
    				temp[j] = true;
    			}
    		}
    		for (int j = 0; j < len; ++j) {
    			al[j] = al[j] || temp[j];
    		}
    	}
        return count;
    }
}
