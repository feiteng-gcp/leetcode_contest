class Solution {
    public int minDeletionSize(String[] A) {
        int n = A[0].length();
        
StringBuilder[] tmp = new StringBuilder[A.length];
        
        for (int i = 0; i < A.length; i++) {
            tmp[i] = new StringBuilder();
        }
        
        int delete = 0;
        for (int i = 0; i < n; i++){
            if (isValid(A, i, tmp) == 0) {
                return delete;
            }
            
            if (isValid(A, i, tmp) == -1) 
{
                delete++;
            } else {
                for (int j = 0; j < A.length; j++) {
                    tmp[j].append(A[j].charAt(i));
                }
            }
        }
        
        return delete;
    }
    
    public int isValid(String[] A,
                          int startIndex,
                          StringBuilder[] tmp) {
        int same = 0;
        for (int i = 0; i < A.length - 1; i++) 
{
            String prev = tmp[i].toString() + A[i].charAt(startIndex);
            String next = tmp[i+1].toString() + A[i+1].charAt(startIndex);
            if (prev.compareTo(next) > 0) {
                
                return -1;
            } else if (prev.compareTo(next) == 0) {
                same++;
            }
        }
        
        return same == 0 ? 0 : 1;
    }
}
