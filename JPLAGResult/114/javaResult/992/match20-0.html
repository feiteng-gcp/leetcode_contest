class Solution {
    public int minDeletionSize(String[] a) {
        int n = a.length;
        
        int m = a[0].length();
        
        int[] com = new int[n - 1];
        
        int ans = 0;
        for(int i = 0; i < m; i++)
        {
            boolean poss = true;
            for(int j = 0; j < n - 1; j++)
            {
                if(com[j] == 0 && a[j].charAt(i) > a[j + 1].charAt(i))
                {
                    poss = false;
                break;
                }
            }
            
            if(!poss)
                ans++;
            else
            {
            for(int j = 0; j < n - 1; j++)
            {
                if(com[j] == 0 && a[j].charAt(i) < a[j + 1].charAt(i))
                {
                    com[j] = -1;
                }
            }
            }
        }
        
        return ans;
    }
}
