class Solution {
    public int minDeletionSize(String[] A) {
        boolean[] isTied = new boolean[A.length];
        Arrays.fill(isTied, true);
        int ret = 0;
        for (int j = 0; j < A[0].length(); j++) {
            boolean ordered = true;
            for (int i = 1; i < A.length; i++) {
                if (A[i].charAt(j) < 
A[i - 1].charAt(j) && isTied[i]) {
                    ordered = false;
                    ret++;
                    break;
                }
            }
            if (ordered) {
                for (int i = 1; i < A.length; i++) {
                    if (A[i].charAt(j) != A[i - 1].charAt(j)) {
                        isTied[i] = false;
                    }
                
}
            }
        }
        return ret;
    }
}
