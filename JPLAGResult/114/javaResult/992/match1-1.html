class Solution {
    public int minDeletionSize(String[] A) {
        int n = A.length;
        int len = A[0].length();
        boolean[] delete = new boolean[len];
        int res = 0;
        boolean[] bigger = new boolean[n];

        for(int i = 1; i < n; i++) {
            if(A[i].charAt(0) < A[i - 1].charAt(0)) {
                delete[0] = true;
                res++;
                break;
            }
        }
        if(!delete[0]) {
            
for(int i = 1; i < n; i++) {
                if(A[i].charAt(0) > A[i - 1].charAt(0)) {
                    bigger[i] = true; // no longer need to decide
                }
            }
        }

        for(int i = 1; i < len; i++) {
            boolean shouldDelete = false;

            for(int j = 1; j < n; j++) {
                if(bigger[j]) {
                    continue;
                } else {
                    if(A[j].charAt(i) < A[j - 1].charAt(i)) {
                        shouldDelete = true;
                        break;
                    }
                }
            }

            if(shouldDelete) {
                delete[i] = true;
                res++;
            
} else {
                for(int j = 1; j < n ;j++) {
                    if(!bigger[j] && A[j].charAt(i) > A[j - 1].charAt(i)) {
                        bigger[j] = true;
                    }
                }
            }
        }
        return res;
        
    }
}
