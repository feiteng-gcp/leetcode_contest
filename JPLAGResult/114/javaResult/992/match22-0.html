class Solution {
    public int minDeletionSize(String[] A) {
        int ans = 0;
        
        int n = A.length;
        if (n == 0) return 0;
        int len = A[0].length();
        
        StringBuilder[] sb = new StringBuilder[n];
        for(int i = 0; i < n; i++) {
            sb[i] = new StringBuilder();
        }
        
        for(int i = 0; i < len; i++) {
        
            boolean lex = true;
            for(int k = 1; k < n; k++) {
                String c = sb[k].toString() + A[k].charAt(i);
                String p = sb[k - 1].toString() + A[k - 1].charAt(i);
                
                if (c.compareTo(p) < 0) {
                    ans++;
                    lex = false;
                    break;
                }
            }
            
            if (lex) {
                for(int k = 0; k < n; k++) {
                    sb[k].append(A[k].charAt(i));
                }
            }
            
            // if (lex) {
            //     return ans;
            // }
        }
        
        return ans;
    }
}
