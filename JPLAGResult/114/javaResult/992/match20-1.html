class Solution {
    public int minDeletionSize(String[] A) {
        int len = A[0].length();
        String[] resA = new String[A.length];
        for (int i = 0; i < resA.length; i++) resA[i] = "";
        
int res = 0;
        
        for (int i = 0; i < len; i++) {
            boolean remain = true;
            for (int j = 1; j < A.length; j++) {
                if (resA[j - 1].equals(resA[j]) && A[j - 1].charAt(i) > A[j].charAt(i)) {
                    res++;
                    remain = false;
                    break;
                }
            }
            if (remain) {
                for (int j = 0; j < A.length; j++)
                    resA[j] = resA[j] + A[j].charAt(i)
;
            }
        }
        
        return res;
    }
}
