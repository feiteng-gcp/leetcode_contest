class Solution {
    public boolean isAlienSorted(String[] words, String order) {
         int N = words.length;
        if (N < 2) {
            return true;
        }
        
int[] map = new int[26];
        for (int i = 0; i < 26; i++) {
            map[order.charAt(i) - 'a'] = i;
        }
        for (int i = 0; i < N - 1; i++) {
            if (!isSmaller(words[i], words[i + 1], map)) {
                return false;
            }
        }
        return true;
    }

    boolean isSmaller(String w1, String w2, int[] map) {
        int i = 0;
        while (i < w1.length() && i < w2.length()) {
            int o1 = map[w1.charAt(i) - 'a'];
            int o2 = map[w2.charAt(i) - 'a'];
            if (o1 < o2) {
                return true;
            } else if (o1 > o2) {
                return false;
            } else {
                i++;
            
}
        }
        return i == w1.length();
    }
}
