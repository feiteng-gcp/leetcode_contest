class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < order.length(); i++) {
            map.put(order.charAt(i), i);
        }
        
        for (int i = 0; i < words.length - 1; i++) {
            if (!isValid(words[i], words[i+1], map)) {
                return false;
            }
        }
        
        return true;
    }
    
    public boolean isValid(String prev,
                          String next,
                          HashMap<Character, Integer> map) {
        for (int i = 0; i < Math.min(prev.length(), next.length()); i++) {
            char cprev = prev.charAt(i);
            char cnext = next.charAt(i);
            if (map.get(cprev) > map.get(cnext)) {
                return false;
            } else if (map.get(cprev) < map.get(cnext)) {
                return true;
            }
        }
        
        return prev.length() <= next.length();
    }
}
