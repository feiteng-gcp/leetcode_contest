class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        Map<Character, Integer> map = new HashMap<>();
        for(int i = 0; i < order.length(); i++) {
            map.put(order.charAt(i), i);
        }
        
        for(int i = 1; i < words.length; i++) {
            String pre = words[i - 1];
            String cur = words[i];
            
            
if(!valid(pre, cur, map)) {
                return false;
            }
        }
        
        return true;
        
    }
    
    private boolean valid(String small, String big, Map<Character, Integer> map) {
        int i = 0; 
        for(; i < small.length() && i < big.length() ; i++) {
            if(map.get(small.charAt(i)) > map.get(big.charAt(i))) {
                return false;
            } else if(map.get(small.charAt(i)) < 
map.get(big.charAt(i))) {
                return true;
            }
        }
        
        return i == small.length();
    }
}
