class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        int[] ch = new int[26];
        for(int i = 0; i < order.length(); i++){
            ch[order.charAt(i) - 'a'] = i;
        }
        for(int i = 1; i < words.length; i++){
            String w1 = words[i - 1];
            String w2 = words[i];
            if(!compare(w1, w2, ch)){
                return false;
            
}
        }
        return true;
    }
    
    private boolean compare(String w1, String w2, int[] ch){
        int j = 0; int k = 0;
        while(j < w1.length() && k < w2.length()){
            char c = w1.charAt(j);
            char d = w2.charAt(k);
            if(ch[c - 'a'] < ch[d - 'a']) return true;
            if(ch[c - 'a'] > ch[d - 'a']) return false;
            j++;
            k++;
        }
        return false;
    }
}
