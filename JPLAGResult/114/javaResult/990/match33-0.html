class Solution {
    int[] order = new int[26];
    public boolean isAlienSorted(String[] words, String order) {
        for(int i=0; i<order.length(); ++i)
            this.order[order.charAt(i) - 'a'] = i;
        
        for(int i=1;i<words.length; ++i) {
            if(isLessThen(words[i], words[i-1]))
                return false;
        }
        return true;
    }
    
    boolean isLessThen(String w1, String w2) {
        for(int i=0; i<w1.length() && i < w2.length(); ++i) {
            char s1 = w1.charAt(i);
            char s2 = w2.charAt(i);
            
            if(s1 == s2) continue;
            
            int o1 = order[s1-'a'];
            int o2 = order[s2-'a'];
            
            return o1 < o2;
        }
        return w1.length() < w2.length();
    }
}
