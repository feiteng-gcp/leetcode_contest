class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        HashMap<Character, Integer> map = new HashMap<>();
        for(int i = 0; i < order.length(); i++) map.put(order.charAt(i), i);
        for(int i = 0; i < words.length - 1; i++) if(compare(words[i], words[i+1], map) > 0) return false;
        return true;
    }
    
    public int compare(String s1, String s2, HashMap<Character, Integer> map){
        for(int i = 0; i < Math.min(s1.length(), s2.length()); i++){
            if(s1.charAt(i) == s2.charAt(i)) continue;
            else return map.get(s1.charAt(i)) - map.get(s2.charAt(i));
        }
        return s1.length()-s2.length();
    }
}
