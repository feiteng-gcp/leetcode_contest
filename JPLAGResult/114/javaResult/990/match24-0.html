class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        int[] seq = new int[26];
        for(int i = 0; i < order.length(); i++){
            seq[order.charAt(i) - 'a'] = i;
        }
        for(int i = 0; i < words.length - 1; i++){
            if (!isLess(words[i], words[i+1], seq)) return false;
        }
        return true;
    }
    
    private boolean isLess(String s1, String s2, int[] seq){
        int p1 = 0, p2 = 0;
        while(p1 < s1.length() && p2 < s2.length()){
            if (seq[s1.charAt(p1) - 'a'] < seq[s2.charAt(p2) - 'a']){
                return true;
            }else if (seq[s1.charAt(p1) - 'a'] == seq[s2.charAt(p2) - 'a']){
                p1++;
                p2++;
            }else{
                return false;
            }    
        }
        return p1 == s1.length();
    }
}
