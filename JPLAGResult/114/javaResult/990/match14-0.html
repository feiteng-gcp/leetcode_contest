class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        int[] map = new int[26];
        for (int i=0; i<26; i++) {
            map[order.charAt(i)-'a'] = i;
        }
        for (int i=1; i<words.length; i++) {
            if (!compare(words[i-1], words[i], map)) {
                return false;
            }
        }
        return true;
    }
    
    private boolean compare(String s1, String s2, int[] map) {
        int n1 = s1.length(); int n2 = s2.length();
        int i = 0; int j = 0;
        while (i < n1 && j < n2) {
            int i1 = map[s1.charAt(i)-'a'];
            int i2 = map[s2.charAt(j)-'a'];
            if (i1 < i2) {
                return true;
            } else if (i1 > i2) {
                return false;
            } else {
                i++; j++;
            }
        }
        if (i < n1) {
            return false;
        }
        return true;
    }
}
