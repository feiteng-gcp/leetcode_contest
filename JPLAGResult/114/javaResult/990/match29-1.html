class Solution {
    Map<Character, Integer> orderMap;
    public boolean isAlienSorted(String[] words, String order) {
        orderMap = new HashMap();
        for (int i = 0; i < 26; i++) 
{
            orderMap.put(order.charAt(i), i);
        }
        for (int i = 1; i < words.length; i++) {
            if (!isGood(words[i-1], words[i])) return false;
        }
        return true;
    }
    public boolean isGood(String s1, String s2) {
        for (int i = 0; i < s1.length(); i++) {
            if (i == s2.length() || orderMap.get(s1.charAt(i)) > orderMap.get(s2.charAt(i))) return false;
            if(orderMap.get(s1.charAt(i)) < orderMap.get(s2.charAt(i))) return true;
        }
        return true;
    }
}
