class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        HashMap<Character, Integer> dic = new HashMap<>();
        for(int i = 0; i < 26; i++){
            dic.put(order.charAt(i), i);
        }
        
for(int i = 0; i < words.length - 1; i++){
            if(!compare(words[i], words[i + 1], dic))
                return false;
        }
        return true;
    }
    public boolean compare(String a, String b, HashMap<Character, Integer> dic){
        int lena = a.length();
        int lenb = b.length();
        for(int i = 0; i < 
Math.min(lena, lenb); i++){
            if(dic.get(a.charAt(i)) > dic.get(b.charAt(i))){
                return false;
            }
            else if(dic.get(a.charAt(i)) < dic.get(b.charAt(i)))
                return true;
        }
        if(lena <= lenb)
            return true;
        else
            return false;
    }
}
