class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        Map<Character, Integer> orderMap = new HashMap<>();
        for (int i = 0; i < order.length(); i++) {
            orderMap.put(order.charAt(i), i);
        }
        for (int i = 0; i < words.length - 1; i++) {
            if (compare(words[i], words[i + 1], orderMap) > 0) {
                return false;
            }
        }
        return true;

    }

    public int compare(final String o1, final String o2, Map<Character, Integer> orderMap) {
        int i = 0;
        int j = 0;
        while (i < o1.length() && j < o2.length()) {
            char ch1 = o1.charAt(i);
            char ch2 = o2.charAt(j);
            int n1 = orderMap.get(ch1);
            int n2 = orderMap.get(ch2);
            int diff = n1 - n2;
            if (diff != 0) {
                return diff;
            }
            i++;
            j++;
        }
        if (i < o1.length()) {
            return 1;
        }
        else if (j < o2.length()) {
            return -1;
        }
        else {
            return 0;
        }
    }
}
