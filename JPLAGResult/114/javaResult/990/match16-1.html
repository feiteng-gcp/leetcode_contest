class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        int[] array = new int[26];
        int index = 0;
        for (char c : order.toCharArray()) {
            
array[c - 'a'] = index++;
        }
        
        int n = words.length;
        for (int i = 0; i < n - 1; i++) {
            if (compare(words[i], words[i + 1], array) > 0) {
                return false;
            }
        }
        return true;
    }
    
    private int compare(String w1, String w2, int[] order) {
        int len = Math.min(w1.length(), 
w2.length());
        for (int i = 0; i < len; i++) {
            int idx1 = order[w1.charAt(i) - 'a'];
            int idx2 = order[w2.charAt(i) - 'a'];
            if (idx1 < idx2) {
                return -1;
            } else if (idx1 > idx2) {
                return 1;
            }
        }
        if (w1.length() == w2.length()) {
            return 0;
        }
        return w1.length() < w2.length() ? -1 : 1;
    }
}
