class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        Map<Character, Integer> map = new HashMap<>();
        for(int i = 0; i < order.length(); ++i) {
            char ch = 
order.charAt(i);
            map.put(ch, i);
        }
        
        for(int i = 0; i < words.length-1; ++i) {
            if(!inorder(words[i], words[i+1], map)) {
                return false;
            }
        }
        return true;    
    }
    public boolean inorder(String w1, String w2, Map<Character, Integer> map) {
        for(int i = 0; i < 
w1.length() && i < w2.length(); ++i) {
            char ch1 = w1.charAt(i);
            char ch2 = w2.charAt(i);
            int idx1 = map.get(ch1);
            int idx2 = map.get(ch2);
            if(idx1 < idx2) {
                return true;
            }
            else if(idx1 > idx2) {
                return false;
            }
        }
        return w1.length() <= w2.length();
    }
}
