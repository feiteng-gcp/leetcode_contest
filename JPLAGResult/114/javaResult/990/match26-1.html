class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        Map<Character, Character> mapToInt = new HashMap<>(); 
        
        int j = 0; 
        for (char c: order.toCharArray()) {
            
mapToInt.put(c, (char)('a' + j)); 
            j++; 
        }
        
        for (int i = 1; i < words.length; i++) {
            if (!compareAlien(words[i - 1], words[i], mapToInt)) {
                return false; 
            }
        }
        return true; 
    }
    
    private boolean compareAlien(String a, String b, Map<Character, Character> mapToInt) {
        return (translate(a, mapToInt).compareTo(translate(b, mapToInt)) <= 0); 
    
}
    
    private String translate(String str, Map<Character, Character> mapToInt) {
        StringBuilder sb = new StringBuilder(); 
        for (char c: str.toCharArray()) {
            sb.append(mapToInt.get(c)); 
        }
        return sb.toString(); 
    }
}
