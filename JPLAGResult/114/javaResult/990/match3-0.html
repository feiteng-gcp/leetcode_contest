class Solution {
    public boolean isAlienSorted(String[] words, String orderRaw) {
        Map<Character, Integer> order = new HashMap<>();
        for (int i = 0; i < orderRaw.length(); i++) {
            order.put(orderRaw.charAt(i), i);
        }
        
        for (int i = 1; i < words.length; i++) {
            if (!isOrdered(order, words[i-1], words[i])) {
                return false;
            }
        }
        
        return true;
    }
    
    public boolean isOrdered(Map<Character, Integer> order, String a, String b) {
        for (int i = 0; i < a.length(); i++) {
            if (i == b.length()) {
                return false;
            }
            if (order.get(a.charAt(i)) > order.get(b.charAt(i))) {
                return false;
            } else if (order.get(a.charAt(i)) < order.get(b.charAt(i))) {
                return true;
            }
        }
        
        return true;
    }
}
