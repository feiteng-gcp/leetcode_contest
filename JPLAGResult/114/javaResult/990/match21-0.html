class Solution
{
    private static boolean lessThanOrEqualTo(String a, String b, HashMap<Character, Integer> map)
    {
        int i = 0;
        while (i < Math.min(a.length(), b.length()))
        {
            if (map.get(a.charAt(i)) < map.get(b.charAt(i)))
            {
                return true;
            }
            else if (map.get(a.charAt(i)) > map.get(b.charAt(i)))
            {
                return false;
            }
            i++;
        }
        if (a.length() <= b.length()) return true;
        return false;
    }

    public boolean isAlienSorted(String[] words, String order)
    {
        HashMap<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < order.length(); i++) map.put(order.charAt(i), i);
        for (int i = 0; i < words.length - 1; i++)
        {
            if (!lessThanOrEqualTo(words[i], words[i + 1], map))
            {
                return false;
            }
        }
        return true;
    }
}
