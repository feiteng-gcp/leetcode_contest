class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        char[] a = new char[256];
        for (int i = 0; i < 26; ++i) {
            a[order.charAt(i)] = (char)(i + 'a');
        }
        final int N = words.length;
        String[] newWords = new String[N];
        for (int i = 0; i < N; ++i) {
            newWords[i] = translate(words[i], a);
        }
        for (int i = 1; i < N; ++i) {
            if (newWords[i - 1].compareTo(newWords[i]) > 0) {
                return false;
            }
        }
        return true;
    }
    
    private static String translate(String w, char[] a) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < w.length(); ++i) {
            sb.append(a[w.charAt(i)]);
        }
        return sb.toString();
    }
}
