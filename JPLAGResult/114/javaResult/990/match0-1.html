class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        int[] dict = new int[26];
        for (int i = 0; i < 26; i++) 
{
            dict[order.charAt(i) - 'a'] = i;
        }
        for (int i = 1; i < words.length; i++) {
            if (!checkOrder(words[i- 1], words[i], dict)) {
                return false;
            
}
        }
        return true;
    }
    
    private boolean checkOrder(String a, String b, int[] dict) {
        
int index = 0;
        while (index < a.length() && index < b.length()) {
            if (dict[a.charAt(index) - 'a'] < dict[b.charAt(index) - 'a']) {
                return true;
            } else if (dict[a.charAt(index) - 'a'] > dict[b.charAt(index) - 'a']) {
                
return false;
            }
            index++;
        }
        return index == a.length();
    }
}
