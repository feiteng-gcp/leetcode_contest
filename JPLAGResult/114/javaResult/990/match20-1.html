class Solution {
    
    Map<Character, Integer> cMap = new HashMap<>();
    
    boolean check(
String s1, String s2) {
        int i=0, j=0;
        while (i<s1.length() && j<s2.length()) {
            char c1=s1.charAt(i), c2=s2.charAt(j);
            if (c1 != c2) {
                if (cMap.get(c1) < 
cMap.get(c2)) {
                    return true;
                } else {
                    return false;
                }
            } else {
                i++;
                j++;
            }
        }
        // apple, app
        if (i<s1.length() && j==s2.length()) return false;
        return true;
    }
    
    public boolean isAlienSorted(String[] words, String order) {
        
for (int i=0; i<order.length(); i++) {
            cMap.put(order.charAt(i), i);
        }
        for (int i=0; i<words.length-1; i++) {
            if (!check(words[i], words[i+1])) {
                return false;
            }
        }
        return true;
    }
}
