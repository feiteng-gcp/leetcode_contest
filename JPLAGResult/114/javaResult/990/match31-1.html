class Solution {
    private int M = 26;
    
    public boolean isAlienSorted(String[] words, String order) {
        int[] h = new int[M];
        for (int i = 0; i < order.length(); i++) {
            char ch = order.charAt(i);
            h[ch - 'a'] = i;
        }
        int n = words.length;
        
for (int i = 0; i < n - 1; i++) {
            if (!lessThan(words[i], words[i + 1], h)) {
                return false;
            }
        }
        return true;
    }
    
    private boolean lessThan(String a, String b, int[] h) {
        int nA = a.length();
        int nB = b.length();
        int stop = Math.min(nA, nB);
        for (int i = 0; i < stop; i++) 
{
            int hA = h[a.charAt(i) - 'a'];
            int hB = h[b.charAt(i) - 'a'];
            if (hA < hB) {
                return true;
            } else if (hA > hB) {
                return false;
            }
        }
        return nA <= nB;
    } 
}
