class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        if (words.length < 2) {
            return true;
        }

        int n = words.length;
        int[] loc = new int[26];
        for (int i = 0; i < order.length(); i++) {
            loc[order.charAt(i) - 'a'] = i;
        }

        for (int i = 0; i < n - 1; i++) {
            if (!helper(words[i], words[i + 1], loc)) {
                return false;
            }
        }

        return true;
    }

    boolean helper(String s1, String s2, int[] loc) {
        if (s1.equals(s2)) {
            return true;
        }

        if (s1.startsWith(s2)) {
            return false;
        }

        for (int i = 0; i < Math.min(s1.length(), s2.length()); i++) {
            char c1 = s1.charAt(i);
            char c2 = s2.charAt(i);
            if (loc[c1 - 'a'] > loc[c2 - 'a']) {
                return false;
            }
            else if (loc[c1 - 'a'] < loc[c2 - 'a']) {
                return true;
            }
        }

        return true;
    }
}
