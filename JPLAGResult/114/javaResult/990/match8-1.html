class Solution {

	private boolean check(String s1 , String s2 , Map<Character , Integer> map) {
		
        int i = 0 , length1 = s1.length() , length2 = s2.length();
        while (i < length1 && i < length2) 
{
            int index1 = map.get(s1.charAt(i));
            int index2 = map.get(s2.charAt(i));
            if (index1 < index2) {
                return true;
            } else if (index1 > index2) {
                return false;
            }
            i ++;
        }		
        if (i == length1) 
{
            return true;
        } else {
            return false;
        }
		
	}
	
    
public boolean isAlienSorted(String[] words, String order) {
     
    	Map<Character , Integer> map = new HashMap<>();
    	for (int i = 0;i < order.length();i ++) {
    		map.put(order.charAt(i) , i);
    	}
    	int i , j , n = words.length;
    	for (i = 0;i < n;i ++) {
    		for (j = i + 1;j < n;j ++) {

    			if (!check(words[i] , words[j] , map)) {
    				return false;
    			}
    		}
    	}
    	return true;
    	
    }
}
