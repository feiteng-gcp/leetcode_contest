class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        if (words == null || words.length < 2) {
            return true;
        }
        
        
Map<Character, Integer> map = new HashMap<Character, Integer>();
        for (int i = 0; i < order.length(); i++) {
            map.put(order.charAt(i), i);
        }
        
        for (int i = 1; i < words.length; i++) {
            if (compareHelper(words[i - 1], words[i], map) > 0) {
                return false;
            }
        }
        return true;
    }
    
    private int compareHelper(String w1, String w2, Map<Character, Integer> map) {
        int index1 = 0, index2 = 0;
        while (index1 < w1.length() && index2 < w2.length()) {
            char c1 = w1.charAt(index1);
            char c2 = w2.charAt(index2);
            int pos1 = map.get(c1);
            int pos2 = map.get(c2);
            if (pos1 < pos2) {
                return -1;
            }
            
if (pos1 > pos2) {
                return 1;
            }
            
            index1++;
            index2++;
        }
        if (w1.length() < w2.length()) {
            return -1;
        }
        if (w1.length() > w2.length()) {
            return 1;
        }
        return 0;
    }
}
