class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        for (int i = 0; i < words.length - 1; i++){
            if (!compare(words[i], words[i + 1], order))
                return false;
        }
        return true;
    }
    
    public boolean compare(String o1, String o2, String order) {
        int i = 0;
        while (i < o1.length() && i < o2.length()){
            if (order.indexOf(o1.charAt(i)) > order.indexOf(o2.charAt(i))){
                return false;
            } else if (order.indexOf(o1.charAt(i)) < order.indexOf(o2.charAt(i))){
                return true;
            } else {
                i++;
            }
        }
        return (i == o1.length());
    }
}
