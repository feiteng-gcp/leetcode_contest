class Solution {
    public int tallestBillboard(int[] rods) {
        Arrays.sort(rods);
        int temp;
        for (int i = rods.length / 2 - 1; i >= 0; i--) {
            temp = rods[i];
            rods[i] = rods[rods.length-1-i];
            rods[rods.length-1-i] = temp;
        }
        int sum = Arrays.stream(rods).sum();
        for (int i = sum / 2; i > 0; i--) {
            if (rodsEnable(i, i , rods, 0, sum)) return i;
        }
        return 0;
    }

    public boolean rodsEnable(int left, int right, int[] rods, int i, int remain) {
        if (left == 0 && right == 0) return true;
        if (left + right > remain) return false;
        if (i >= rods.length) return false;
        if (left < 0 || right < 0) return false;
        int rod = rods[i++];
        remain -= rod;
        return rodsEnable(left - rod, right, rods, i, remain) ||
                rodsEnable(left, right - rod, rods, i, remain) ||
                rodsEnable(left, right, rods, i, remain);
    }
}
