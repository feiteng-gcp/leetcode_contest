class Solution {
    public int tallestBillboard(int[] rods) {
        int sum = 0;
        for (int x : rods) {
            sum += x;
        }
        int m = sum / 2;
        boolean[][] f = new boolean[m + 1][m + 1];
        f[0][0] = true;
        sum = 0;
        for (int x : rods) {
            sum += x;
            for (int i = Math.min(m, sum); i >= 1; i--) {
                for (int j = Math.min(i, sum - i); j >= 0; j--) {
                    if (i >= x && f[i - x][j] || j >= x && f[i][j - x]) {
                        f[i][j] = f[j][i] = true;
                    }
                }
            }
        }
        for (int i = m; i >= 0; i--) {
            if (f[i][i]) {
                return i;
            }
        }
        return -1;
    }
}
