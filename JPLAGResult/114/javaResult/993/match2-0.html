class Solution {
    boolean dfs(int[] sticks, int index, int num, int cur, int target, boolean[] vst) {
        if (cur > target) return false;

//        if (target == 10) {
//            System.out.println(Arrays.toString(vst));
//            System.out.println("num = " + num + " cur = " + cur + " index = " + index);
//        }

        for (int i = index; i < sticks.length; ) {
            if (vst[i]) {
                i++;
                continue;
            }
            vst[i] = true;
            if (cur + sticks[i] < target) {

                if (dfs(sticks, i + 1, num, cur + sticks[i], target, vst)) {
                    return true;
                }
            } else if (cur + sticks[i] == target) {
                if (num == 1) return true;
                if (dfs(sticks, 0, num + 1, 0, target, vst)) {
                    return true;
                }
            }
            vst[i] = false;
            int j = i + 1;
            while (j < sticks.length && sticks[j] == sticks[i]) j++;
            i = j;
        }
        return false;
    }

    public int tallestBillboard(int[] rods) {
        if (rods.length <= 1) return 0;
        int sum = 0;
        for (int i = 0; i < rods.length; i++) {
            sum += rods[i];
        }

        Arrays.sort(rods);
        for (int i = 0, j = rods.length - 1; i < j; i++, j--) {
            int t = rods[i];
            rods[i] = rods[j];
            rods[j] = t;
        }

        //System.out.println(Arrays.toString(rods) +  " sum = " + sum);
        int n = rods.length;

        for (int i = sum / 2; i >= rods[n - 1]; i--) {
            if (dfs(rods, 0, 0, 0, i, new boolean[n])) {
                return i;
            }
        }
        return 0;
    }
}
