class Solution {
    private boolean used[];

	private boolean found;

	public int tallestBillboard(int[] rods) {

		used = new boolean[rods.length];
		int sum = 0;
		List<Integer> rodList = new ArrayList<>();
		for (int i = 0; i < rods.length; i++) {
			rodList.add(rods[i]);
			sum += rods[i];
		}

		Collections.sort(rodList, new Comparator<Integer>() {
			@Override
			public int compare(Integer o1, Integer o2) {
				return o2.compareTo(o1);
			}
		});

		for (int x = sum / 2; x >= 0; x--) {
			found = false;
			search(1, 0, x, 0, rodList);
			if (found) {

				return x;
			}
		}

		return 0;


	}

	private void search(int count, int current, int sum, int index, List<Integer> list) {
		if (!found) {
			if (count == 3) {
				found = true;
			} else {
				for (int i = index; i < list.size(); i++) {
					if (!used[i]) {
						if (current + list.get(i) == sum) {
							used[i] = true;
							search(count + 1, 0, sum, 0, list);
							used[i] = false;
							break;
						} else if (current + list.get(i) < sum) {
							used[i] = true;
							search(count, current + list.get(i), sum, i + 1, list);
							used[i] = false;
							while (i < list.size() - 1 && list.get(i) == list.get(i + 1)) {
								i++;
							}
						}
					}
				}
			}
		}
	}
}
