class Solution {
    public boolean canReorderDoubled(int[] A) {
        Arrays.sort(A);
        Map<Integer, Integer> map = new HashMap<>();
        for (int a : A)
            map.put(a, map.getOrDefault(a, 0) + 1);
        for (int i = 0; i < A.length; i++) {
            if (map.get(A[i]) != 0) {
                int count = map.get(A[i]);
                if (A[i] < 0) {
                    if (A[i] % 2 != 0)
                        return false;
                    if (map.getOrDefault(A[i]/2, 0) == 0)
                        return false;
                    map.put(A[i], count-1);
                    map.put(A[i]/2, map.get(A[i]/2)-1);
                } else if (A[i] > 0) {
                    if (map.getOrDefault(A[i]*2, 0) == 0)
                        return false;
                    map.put(A[i], count-1);
                    map.put(A[i]*2, map.get(A[i]*2)-1);
                } else {
                    if (count % 2 != 0)
                        return false;
                    map.put(A[i], 0);
                }
            }
        }
        return true;
    }
}
