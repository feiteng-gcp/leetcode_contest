class Solution {
    public boolean canReorderDoubled(int[] A) {
        TreeMap<Integer, Integer> sort = new TreeMap<>(new Comparator<Integer>() {
			@Override
			public int compare(Integer o1, Integer o2) {
				if ((long) o1 * o2 < 0) {
					return o1 - o2;
				}else {
					return Math.abs(o1) - Math.abs(o2);
				}
			}
        	
        });
        for (int i : A) {
        	sort.put(i, sort.getOrDefault(i, 0)+1);
        }
        while(!sort.isEmpty()) {
        	int now = sort.firstKey();
        	if (sort.containsKey(2*now)) {
        		int c1 = sort.get(now), c2 = sort.get(2*now);
        		if (c1 == 1) {
        			sort.remove(now);
        		}else {
        			sort.put(now, c1-1);
        		}
        		if (c2 == 1) {
        			sort.remove(2*now);
        		}else {
        			sort.put(2*now, c2-1);
        		}
        	}else {
        		return false;
        	}
        }
        return true;
    }
}
