class Solution {
    public boolean canReorderDoubled(int[] A) {
        int[] pos=new int[100001];
        int[] neg=new int[100001];
        int n1=0,n2=0;
        for (int x:A)
        {
            if (x>=0)
            {
                n1++;
                pos[x]++;
            }
            else
            {
                neg[-x]++;
            }
        }
        if (n1%2!=0) return false;
        for (int i=0;i<=100000;i++)
        if (pos[i]>0)
        {
            int j=i<<1;
            if (j>100000) return false;
            if (pos[j]<pos[i]) return false;
            pos[j]-=pos[i];
        }
        for (int i=1;i<=100000;i++)
        if (neg[i]>0)
        {
            int j=i<<1;
            if (j>100000) return false;
            if (neg[j]<neg[i]) return false;
            neg[j]-=neg[i];
        }
        return true;
    }
}
