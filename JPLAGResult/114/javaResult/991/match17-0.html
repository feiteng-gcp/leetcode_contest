class Solution {
    public boolean canReorderDoubled(int[] A) {
        Map<Integer, Integer> map = new TreeMap<>();
        for (int i = 0; i < A.length; i++){
            map.put(A[i], map.getOrDefault(A[i], 0) + 1);
        }
        Map<Integer, Integer> tmp = new TreeMap<>(map);
        for (Map.Entry<Integer, Integer> e : tmp.entrySet()){
            int key = e.getKey();
            // System.out.println(key + "==="+e.getValue());
            if (!map.containsKey(key)) continue;
            if (key > 0){
                if (!map.containsKey(key*2)) return false;
                int reduce = Math.min(map.get(key), map.get(key*2));
                if (map.get(key) != reduce) return false;
                if (map.get(key*2)-reduce == 0){
                     map.remove(key*2);
                } else {
                    map.put(key*2, map.get(key*2)-reduce);
                }
            } else if (key < 0){
                if (!map.containsKey(key/2)) return false;
                int reduce = Math.min(map.get(key), map.get(key/2));
                if (map.get(key) != reduce) return false;
                if (map.get(key/2)-reduce == 0){
                     map.remove(key/2);
                } else {
                    map.put(key/2, map.get(key/2)-reduce);
                }
            } else {
                if (map.get(key) % 2 != 0) return false;
            }
        }
        return true;
    }
}
