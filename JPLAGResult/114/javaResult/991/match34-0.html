class Solution {
    public boolean canReorderDoubled(int[] A) {
        int negCount = 0;
        int posCount = 0;

        TreeSet<Integer> neg = new TreeSet<>();
        TreeSet<Integer> pos = new TreeSet<>();
        HashMap<Integer, Integer> negNums = new HashMap<>();
        HashMap<Integer, Integer> posNums = new HashMap<>();

        for (int i = 0; i < A.length; i++) {
            if (A[i] < 0) {
                negCount++;
                if (neg.contains(A[i])) {
                    int x = negNums.get(A[i]);
                    negNums.put(A[i], x + 1);
                } else {
                    neg.add(A[i]);
                    negNums.put(A[i], 1);
                }
            } else if (A[i] > 0) {
                posCount++;
                if (pos.contains(A[i])) {
                    int x = posNums.get(A[i]);
                    posNums.put(A[i], x + 1);
                } else {
                    pos.add(A[i]);
                    posNums.put(A[i], 1);
                }
            }
        }

        int zeros = A.length - negCount - posCount;
        if (zeros % 2 != 0 || negCount % 2 != 0 || posCount % 2 != 0) {
            return false;
        }

        while (! neg.isEmpty()) {
            int x = neg.last();
            int xn = negNums.get(x) - 1;
            negNums.put(x, xn);

            if (xn == 0) {
                neg.remove(x);
            }

            int y = 2 * x;
            if (! neg.contains(y)) {
                return false;
            }
            int yn = negNums.get(y) - 1;
            negNums.put(y, yn);

            if (yn == 0) {
                neg.remove(y);
            }
        }

        while (! pos.isEmpty()) {
            int x = pos.first();

            int xn = posNums.get(x) - 1;
            posNums.put(x, xn);

            if (xn == 0) {
                pos.remove(x);
            }

            int y = 2 * x;
            if (! pos.contains(y)) {
                return false;
            }
            int yn = posNums.get(y) - 1;
            posNums.put(y, yn);

            if (yn == 0) {
                pos.remove(y);
            }
        }

        return true;
    }
}
