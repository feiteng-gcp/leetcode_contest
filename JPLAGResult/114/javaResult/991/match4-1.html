class Solution {
    public boolean canReorderDoubled(int[] A) {
        Arrays.sort(A);
        Map<Integer, Integer> map = new HashMap();
        for(int i = 0; i < A.length; i++) {
            if (A[i]>=0){
                
if (map.containsKey(A[i])) {
                    int count = map.get(A[i]);
                    if (count == 1){
                        map.remove(A[i]);
                    } else {
                        map.put(A[i], count - 1);
                    }
                } else {
                    map.put(A[i] * 2, map.getOrDefault(A[i] * 2, 0) + 1);
                }
            } else {
                
if (map.containsKey(A[i] * 2)) {
                    int count = map.get(A[i] * 2);
                    if (count == 1){
                        map.remove(A[i]*2);
                    } else {
                        map.put(A[i]*2, count - 1);
                    }
                } else {
                    map.put(A[i], map.getOrDefault(A[i], 0)+1);
                }
            }
        }
        return map.isEmpty();
    }
}
