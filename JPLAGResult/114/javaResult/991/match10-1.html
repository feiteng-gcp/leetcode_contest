class Solution {
    public boolean canReorderDoubled(int[] A) {
        
        int len = A.length;
        
        int TOP = 100000;
        
        
int[] arr1 = new int[TOP+1];
        int[] arr2 = new int[TOP+1];
        
        for(int i=0;i<len;i++){
            if(A[i]>0){
                arr1[A[i]]++;
            }
            if(A[i]<0){
                arr2[-A[i]]++;
            }
        }
        
        for(int i=TOP;i>0;i--){
            
            if((arr1[i]>0 || arr2[i]>0) && i%2==1) return false;
            
  
            
if(arr1[i/2]<arr1[i] || arr2[i/2]<arr2[i]) return false;
            
            arr1[i/2] -= arr1[i];
            
            arr2[i/2] -= arr2[i];
        }
        
        return true;
        
    }
}
