class Solution {
    public boolean canReorderDoubled(int[] A) {
        List<Integer> list1 = new ArrayList<>();
        List<Integer> list2 = new ArrayList<>();
        for(int num : A) {
            if(num >= 0) {
                list1.add(num);
            } else {
                list2.add(-num);
            }
        }
        
        

        return helper(list1) && helper(list2);
    }
    
    private boolean helper(List<Integer> list) {
        Collections.sort(list);
        int n = list.size();
        if(n % 2 == 1) {
            return false;
        }
        Map<Integer, Integer> map = new HashMap<>(); // value: first appearance
        boolean[] visited = new boolean[n];
        
        for(int i = 0; i < n; i++) {
            int num = list.get(i);
            if(!map.containsKey(num)) {
                map.put(num, i);
            }
        }
        
        for(int i = 0; i < n; i ++) {
            if(visited[i]) {
                continue;
            }
            
            int cur = list.get(i);
            int target = 2 * cur;
            
            if(!map.containsKey(target)) {
                return false;
            } else {
                int idx = map.get(target);
                if(idx >= n || list.get(idx) != target) {
                    return false;
                }
                
                visited[idx] = true;
                map.put(target, idx + 1);
            }
        }
        
        
        return true;
    }
}
