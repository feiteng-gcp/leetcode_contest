class Solution {
    public boolean canReorderDoubled(int[] arr) {
		if (arr==null || arr.length==0)
			return true;
        Arrays.sort(arr);
        HashMap<Integer, Integer> map = new HashMap<Integer, Integer>(); //map value to its count
        int firstPositive = arr.length;
        for (int i=0; i<arr.length; i++) {
        	if (arr[i]>=0) {
        		firstPositive = i;
        		break;
        	}
        	//check if double is in map, otherwise add to map
        	//remove entry if count is 0
        	int dub = arr[i] * 2;
        	if (map.containsKey(dub)) {
        		int count = map.get(dub);
        		if (count==1) {
        			map.remove(dub);
        		} else {
        			map.put(dub, count-1);
        		}
        	} else {
        		int count = 0;
        		if (map.containsKey(arr[i])) {
        			count = map.get(arr[i]);
        		}
        		map.put(arr[i], count+1);
        	}
        }
        if (map.size() > 0)
        	return false;
        
        //iterate down
        for (int i=arr.length-1; i>=firstPositive; i--) {
        	int dub = arr[i] * 2;
        	if (map.containsKey(dub)) {
        		int count = map.get(dub);
        		if (count==1) {
        			map.remove(dub);
        		} else {
        			map.put(dub, count-1);
        		}
        	} else {
        		int count = 0;
        		if (map.containsKey(arr[i])) {
        			count = map.get(arr[i]);
        		}
        		map.put(arr[i], count+1);
        	}
        }
        if (map.size() > 0)
        	return false;
        
        return true;
    }
}
