class Solution {
    public boolean canReorderDoubled(int[] A) {
        Arrays.sort(A);
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        for (int num : A) {
            if (num <= 0) {
                if (map.containsKey(2 * num)) {
                    map.put(2 * num, map.getOrDefault(2 * num, 1) - 1);
                    if (map.get(2 * num) == 0) {
                        map.remove(2 * num);
                    }
                } else {
                    map.put(num, 
map.getOrDefault(num, 0) + 1);
                }
            } else {
                if (num % 2 == 0 && map.containsKey(num / 2)) {
                    map.put(num / 2, map.getOrDefault(num / 2, 1) - 1);
                    if (map.get(num / 2) == 0) {
                        map.remove(num / 2);
                    }
                } else {
                    map.put(num, 
map.getOrDefault(num, 0) + 1);
                }
            }
        }
        return map.isEmpty();
    }
}
