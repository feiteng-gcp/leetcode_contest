class Solution {
    public boolean canReorderDoubled(int[] A) {
        Arrays.sort(A);
        
Map<Integer, Integer> freq = new HashMap<>();
        for(int a : A){
            freq.put(a, freq.getOrDefault(a, 0) + 1);
        }
        for(int i = 0; i < A.length; i++){
            int num = A[i];
            
if(!freq.containsKey(num)) continue;
            if(num < 0){
                if(num % 2 != 0 || !freq.containsKey(num / 2)) return false;
                int currFreq = freq.get(num);
                if(currFreq == 1){
                    freq.remove(num);
                }else{
                    freq.put(num, currFreq - 1);
                }
                int pairFreq = freq.get(num / 2);
                if(pairFreq == 1){
                    freq.remove(num / 2);
                }else{
                    
freq.put(num / 2, pairFreq - 1);
                }
            }else if(num > 0){
                if(!freq.containsKey(num * 2)) return false;
                int currFreq = freq.get(num);
                if(currFreq == 1){
                    freq.remove(num);
                }else{
                    freq.put(num, currFreq - 1);
                }
                int pairFreq = freq.get(num * 2);
                if(pairFreq == 1){
                    freq.remove(num * 2);
                }else{
                    freq.put(num * 2, pairFreq - 1);
                }
            }else{
                if(!freq.containsKey(num) || freq.get(num) < 2) return false;
                int currFreq = freq.get(num);
                if(currFreq == 2){
                    freq.remove(num);
                }else{
                    freq.put(num, currFreq - 2);
                
}
            }
        }
        return true;
    }
}
