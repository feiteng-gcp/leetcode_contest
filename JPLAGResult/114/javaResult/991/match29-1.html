class Solution {
    public boolean canReorderDoubled(int[] A) {
        TreeMap<Integer, Integer> sortedPos = new TreeMap<>();
        TreeMap<Integer, Integer> sortedNeg = new TreeMap<>();
        int zeros = 0;
        
        for(int x : A) 
            if(x > 0) sortedPos.put(x, sortedPos.getOrDefault(x, 0) + 1);
            else if(x < 0) sortedNeg.put(-x, sortedNeg.getOrDefault(-x, 0) + 1);
            else zeros++; 
        
        
return zeros%2 == 0 && isValid(sortedPos) && isValid(sortedNeg);
    }
    
    boolean isValid(TreeMap<Integer, Integer> sorted) {        
        while(!sorted.isEmpty()) {
            int min = sorted.firstKey();
            int count = sorted.get(min);
            sorted.remove(min);
            
            System.out.printf("Min = %s (count = %s). Min * 2 = %s\n", min, count, sorted.containsKey(min * 2));
            if(sorted.containsKey(min * 2)) {
                if(sorted.get(min * 2) == count) sorted.remove(min * 2);
                else if(sorted.get(min * 2) < count) return false;
                else sorted.put(min * 2, sorted.get(min * 2) - count);
            }
            else 
                return false;
        }
        return true;
    }
}
