class Solution {
    public boolean canReorderDoubled(int[] b) {
        int n = b.length;
        
        Integer[] a = new Integer[n];
        
        for(int i = 0; i < n; i++)
        {
            a[i] = b[i];
        }
        
        Arrays.sort(a, (a1, a2) -> (int)Math.abs(a1) - (int)Math.abs(a2));
        
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        
        int z = 0;
        for(int i = 0; i < n; i++)
        {
            if(a[i] != 0)
            map.merge(a[i], 1, (x, y) -> x + y);
            else
                z++;
        }
        
        if(z % 2 == 1)
        {
            return false;
        }
        //boolean[] d = new boolean[n];
        
        for(int i = 0; i < n; i++)
        {
            if(a[i] == 0)
                continue;
            if(map.containsKey(a[i]))
            {
                if(map.containsKey(2 * a[i]))
                {
                map.merge(a[i], -1, (x, y) -> x + y);
                map.merge(2 * a[i], -1, (x, y) -> x + y);
                
                if(map.get(a[i]) == 0)
                {
                    map.remove(a[i]);
                }
                
                if(map.get(2 * a[i]) == 0)
                {
                    map.remove(2 * a[i]);
                }
                }
                else
                {
                    return false;
                }
            }
        }
        
        return true;
    }
}
