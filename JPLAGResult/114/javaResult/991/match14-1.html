class Solution {
    public boolean canReorderDoubled(int[] A) {
        int magic = 100_000;
        int[] map = new int[magic * 2 + 1];
        for (int a : A) {
            map[a + magic]++;
        
}
        if (map[magic] % 2 != 0) {
            return false;
        }
        for (int i = magic + 1; i <= 2 * magic; i++) {
            if (map[i] > 0) {
                int twice = (i - magic) * 2 + magic;
                if (twice > 2 * magic) {
                    return false;
                }
                if (map[twice] < map[i]) {
                    return false;
                }
                map[twice] -= map[i];
            }
        }
        for (int i = magic - 1; i >= 0; i--) {
            if (map[i] > 0) {
                int twice = (i - magic) * 2 + magic;
                if (twice < 0) {
                    return false;
                }
                if (map[twice] < map[i]) 
{
                    return false;
                }
                map[twice] -= map[i];
            }
        }
        return true;
    }
}
