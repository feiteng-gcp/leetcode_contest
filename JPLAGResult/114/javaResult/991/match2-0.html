class Solution {
    public boolean canReorderDoubled(int[] A) {
        if (A == null || A.length == 0) { return true; }
        
        Arrays.sort(A);            
        HashMap<Integer, Integer> map = new HashMap<>();
        
        for (int i = 0; i < A.length; i++) {
            // if (A[i] % 2 == 1) { return false; }
                        
            int d = A[i] > 0 ? (A[i] / 2) : (A[i] * 2);
            
            if (A[i] > 0 && (A[i] % 2 == 1)) {
                                
                if (map.containsKey(A[i])) {
                    map.put(A[i], map.get(A[i]) + 1);
                } else {
                    map.put(A[i], 1);
                }
                continue;
            }
            
            //System.out.println("A[i]: " + A[i] + ", d:" + d);
            if (map.containsKey(d)) {
                int e = map.get(d);
                if (e == 1) {
                    map.remove(d);
                } else {
                    map.put(d, e-1);
                }                
            } else {
                if (map.containsKey(A[i])) {
                    map.put(A[i], map.get(A[i]) + 1);
                } else {
                    map.put(A[i], 1);
                }
            }            
            
            //System.out.println("set is: " + map);
        }
        
        return map.size() == 0;
    }
}
