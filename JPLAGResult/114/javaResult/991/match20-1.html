class Solution {
    public boolean canReorderDoubled(int[] A) {
        Arrays.sort(A);
        
        for(int i = 0; i < A.length; i++){
            if (A[i] != 111111){
                
for (int j = i + 1; j < A.length; j++){
                    if (A[j] != 111111 && ((A[i] >= 0 && 2 * A[i] == A[j]) ||
                       (A[i] < 0 && A[i] / 2 == A[j]))){
                        A[i] = 111111;
                        A[j] = 111111;
                       break;
                    }
                    if (j == A.length - 1) return false;
                }
            }
            
        }
        
        for (int i = 0; i < A.length; i++){
            
if (A[i] != 111111) return false;
        }
        
        return true;
    }
}
