class Solution {
    public boolean canReorderDoubled(int[] A) {
        Arrays.sort(A);
        int[] countp = new int[200001];
        int[] countm = new int[200001];
        for (int i : A) {
            if (i>=0) {
                countp[i]++;
            } else {
                countm[-i]++;
            }
        }
        if (countp[0]%2==1)
            return false;
        for (int i = 1; i <= 100000; i++) {
            countp[i*2] -= countp[i];
            countm[i*2] -= countm[i];
            if (countp[i*2]<0||countm[i*2]<0)
                return false;
        }
        return true;
    }
}
