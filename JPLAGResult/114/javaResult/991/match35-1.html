class Solution {
    public boolean canReorderDoubled(int[] A) {
        TreeMap<Integer, Integer> map1 = new TreeMap<>();
        TreeMap<Integer, Integer> map2 = new TreeMap<>();
        for (int i : A) {
            if (i >= 0) map1.put(i, map1.getOrDefault(i, 0) + 1);
            else map2.put(-i, map2.getOrDefault(-i, 0) + 1)
;
        }
        return test(map1) && test(map2);
    }
    private boolean test(TreeMap<Integer, Integer> map) {
        while (!map.isEmpty()) {
            int lo = map.firstKey();
            if (map.get(lo) == 1) map.remove(lo);
            else map.put(lo, map.get(lo) - 1);
            if (!map.containsKey(2*lo)) return false;
            if (map.get(2*lo) == 1) map.remove(2*lo);
            else map.put(2*lo, map.get(2*lo) - 1);
        }
        return true;
    }
}
