class Solution {
  public boolean canReorderDoubled(int[] A) {
    Map<Integer, Integer> freq = new HashMap<>();
    for (int a : A) {
      freq.put(a, 1 + freq.getOrDefault(a, 0));
    }
    Arrays.sort(A);
    for (int i = A.length - 1; i >= 0; i--) {
      int key = A[i];
      if (!freq.containsKey(key)) {
        continue;
      }
      int val = freq.get(key);
      if (val > 1) {
        freq.put(key, val - 1);
      } else {
        freq.remove(key);
      }
      if (key < 0) {
        key *= 2;
      } else {
        if (key % 2 != 0) {
          return false;
        }
        key /= 2;
      }
      if (freq.containsKey(key)) {
        val = freq.get(key);
        if (val > 1) {
          freq.put(key, val - 1);
        } else {
          freq.remove(key);
        }
      } else {
        return false;
      }
    }
    return true;
  }
}
