class Solution {
    public boolean canReorderDoubled(int[] A) {
        if (A == null || A.length == 0) {
            return true;
        }
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int a : A) {
            map.put(a, map.getOrDefault(a, 0) + 1);
        }
        
        int pair = 0;
        for (Map.Entry<Integer, Integer> e : map.entrySet()) {
            int key = e.getKey();
            Integer curVal = e.getValue();
            if (curVal == 0) {
                continue;
            }
            int target = key < 0 ? key / 2 : key * 2;
            Integer targetVal = map.get(target);
            if (targetVal == null || targetVal < curVal) {
                return false;
            }
            
            map.put(target, targetVal - curVal);
            //pair += curVal
            pair += key == target ? curVal / 2 : curVal;

        }
        return pair == A.length / 2;
    }
}
