class Solution {
    public boolean canReorderDoubled(int[] A) {
        Map<Integer, Integer> cnts = new HashMap<>();
        for (int a : A) {
            cnts.put(a, cnts.getOrDefault(a, 0) + 1);
        }
        
        Arrays.sort(A);
        //boolean print  = false;
        //System.out.println(cnts);
        for (int a : A) {
            //if (cnts.values().stream().reduce(0, Integer::sum) < 490)
               // print = true;
            //if (print)
            //System.out.println("Processing " + a);
            if (!cnts.containsKey(a)) 
                continue;
            if (cnts.containsKey(2 * a)) {
                decrement(a, cnts);
                if (!cnts.containsKey(2 * a))
                    return false;
                decrement(2 * a, cnts);
            } else if ((a % 2 == 0) && cnts.containsKey(a / 2)) {
                decrement(a, cnts);
                if (!cnts.containsKey(a / 2))
                    return false;
                decrement(a / 2, cnts);
            } else {
                //if (print)
                //System.out.println("Returning false for " + a);
                return false;
            }
            //if (print)
            //System.out.println("cnts now are " + cnts);
        }
        return cnts.isEmpty();
    }
    
    void decrement(int a, Map<Integer, Integer> cnts) {
        
int rem = cnts.get(a);
        if (rem == 1) {
            cnts.remove(a);
        } else {
            cnts.put(a, cnts.get(a) - 1);
        }
    }
    
}
