class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        m, n, c = len(A), len(A[0]), [0]
        def helper(k, A, passcheck):
            #print(k, passcheck)
            newpasscheck = set(passcheck)
            valid = True
            if k >= n:
                return
            for i in range(1, m):
                if i in passcheck:
                    continue
                if A[i][k] < A[i-1][k]:
                    valid = False
                    break
                elif A[i][k] > A[i-1][k]:
                    newpasscheck.add(i)
            if valid:
                helper(k+1, A, newpasscheck)
            else:
                c[0] += 1
                helper(k+1, A, passcheck)
                
                
                    
        helper(0, A, set())
        return c[0]
