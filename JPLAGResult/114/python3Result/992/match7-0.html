class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        
        le = len(A[0])
        n = len(A)
        l = [1] * (n-1)
        res = 0
        for i in range(le):
            for j in range(n-1):
                if l[j] == 1 and A[j][i] > A[j+1][i]:
                    res += 1
                    break
            else:
                for j in range(0, n-1):
                    if l[j] == 1 and A[j][i] < A[j+1][i]:
                        l[j] = 0
                        
        return res
