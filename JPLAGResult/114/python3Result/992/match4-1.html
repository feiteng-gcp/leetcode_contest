class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """

        n = len(A)
        l = len(A[0])
        res = 0
        inc = [False for x in range(n-1)]
        ninc = 0
        for i in range(l):
            if ninc == n-1:
                break
            removed = False
            for j in range(1, n):
                if inc[j-1]:
                    continue
                if A[j]
[i] < A[j-1][i]:
                    removed = True
                    res += 1
                    break
            if not removed:
                for j in range(1, n):
                    if A[j][i] > A[j-1][i]:
                        inc[j-1] = True
                        ninc += 1
        return res
