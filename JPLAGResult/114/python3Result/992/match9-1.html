class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        n 
= len(A)
        m = len(A[0])
        
        left = list(range(n-1))
        ans = 0
        
        for i in range(m):
            
         
            eql = []
            for j in left:

                if A[j][i] < A[j+1][i]:
                    continue
                if A[j][i] > A[j+1][i]:
                    ans += 1
                    break
                else:
                    eql.append(j)
            else:
                left = eql
                
        return ans
