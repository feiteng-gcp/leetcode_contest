class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        s=0
        L=len(A)
        N=len(A[0])
        dic=[[0 for j in range(L)] for i in range(L)]
        pair=0
        for k in range(N):
            flag=True
            for i in range(L):
                for j in range(i+1,L):
                    if A[i][k]>A[j][k] and dic[i][j]==0:
                        flag=False
                        break
                if flag==False:
                    break
            if flag==False:
                s=s+1
            else:
                for i in range(L):
                    for j in range(i+1,L):
                        if A[i][k]<A[j][k]:
                            dic[i][j]=1
            if pair==L*L:
                break
        return s
