class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        m=len(A)
        n=len(A[0])
        l=[0]*n
        remain=list(range(m-1))
        delete_num=0
        for j in range(n):
            p=True
            l_delete=[]
            for k in range(len(remain)):
                
if A[remain[k]][j]>A[remain[k]+1][j]:
                    delete_num+=1
                    p=False
                    break
                if A[remain[k]][j]<A[remain[k]+1][j]:
                    l_delete.append(k)
            if p:
                for k in range(len(l_delete)-1,-1,-1):
                    remain.pop(l_delete[k])
        return delete_num
