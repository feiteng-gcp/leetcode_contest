class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        a = A
        r = len(a)
        c = len(a[0])

        ans = 0
        need = r - 1
        mark = [False for _ in range(r - 1)]
        for j in range(c):
            buff 
= [False for _ in range(r - 1)]
            usable = True
            for i in range(r - 1):
                if mark[i]:
                    continue
                if a[i][j] > a[i + 1][j]:
                    usable = False
                    break
                if a[i][j] < a[i + 1][j]:
                    buff[i] = True

            if usable:
                for i in range(r - 1):
                    mark[i] = mark[i] or buff[i]
            else:
                ans += 1
        return ans
