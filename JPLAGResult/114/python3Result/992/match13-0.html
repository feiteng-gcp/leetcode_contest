class Solution:
    def minDeletionSize(self, A):
        s=0
        judge=[1]*(len(A)-1)
        for i in range(len(A[0])):
            if(sum(judge)==0):
                return s
            for j in range(len(A)-1):
                if(judge[j]==0):
                    continue
                if(A[j][i]>A[j+1][i]):
                    s+=1
                    break
            else:
                for j in range(len(A)-1):
                    if(A[j][i]<A[j+1][i]):
                        judge[j]=0
        return s
