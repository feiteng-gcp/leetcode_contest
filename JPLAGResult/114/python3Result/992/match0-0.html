class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        m=len(A)
        n=len(A[0])
        if m==1:
            return 0
        L=[1]*(m-1)
        Total=0
        for j in range(n):
            if sum(L)==0:
                return Total
            New=list(L)
            code=0
            for i in range(m-1):
                if New[i]==0:
                    continue
                if A[i][j]>A[i+1][j]:
                    code=1
                    break
                if A[i][j]<A[i+1][j]:
                    New[i]=0
            if code==0:
                L=New
            else:
                Total+=1
        return Total
