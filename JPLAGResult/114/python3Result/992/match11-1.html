class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        N, L = len(A), len(A[0])
        groups = [list(range(N))]
        ans = 0
        
        for j in range(L):
            need_del = False
            for group in groups:
                if need_del: break
                for i in range(1, len
(group)):
                    if A[group[i]][j] < A[group[i-1]][j]:
                        need_del = True
                        break
            if need_del:
                ans += 1
                continue
            
            ngroups = []
            for group in groups:
                ngroups.append([group[0]])
                for i in range(1, len(group)):
                    if A[group[i]][j] == A[group[i-1]][j]:
                        ngroups[-1].append(group[i])
                    else:
                        ngroups.append([group[i]])
            groups = ngroups
        return ans
