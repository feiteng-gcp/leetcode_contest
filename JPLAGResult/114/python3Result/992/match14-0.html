class Solution:
    def minDeletionSize(self, A):
        def helper(B):
            res = 0
            for k in range(len(B[0][0])):
                f1 , f2 = False, False
                subset = []
                for i in range(len(B)):
                    j = 1
                    while j < len(B[i]):
                        if B[i][j][k] < B[i][j-1][k]:
                            f1 = True
                            res += 1
                            break
                        if B[i][j][k] == B[i][j-1][k]:
                            f2 = True
                            tmp = []
                            tmp.append(B[i][j-1][k+1:])
                            tmp.append(B[i][j][k+1:])
                            j += 1
                            while j < len(B[i]) and B[i][j][k] == B[i][j-1][k]:
                                tmp.append(B[i][j][k+1:])
                                j += 1
                            subset.append(tmp)
                            j -= 1
                        j += 1
                    if f1:
                        break
                if not f1 and not f2:
                    return res
                if not f1 and f2:
                    #print (B,subset)
                    return res + helper(subset)
            return res
        return helper([A])
