class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        states = [False] * (len(A) - 1)
        res = 0
        for chars in zip(*A):
            l = len(chars)
            valid = True
            for j in range(l-1):
                if chars[j] > chars[j + 1] and not states[j]:
                    valid = False
                    res += 1
                    break
            if valid:
                for j in range(l-1):
                    if chars[j] < chars[j + 1]:
                        states[j] = True
        return res
