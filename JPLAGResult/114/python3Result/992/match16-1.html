class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        confirmed = [False] * (len(A) - 1)
        result = 0
        for i in range(len(A[0])):
            has_equal = False
            has_deleted = False
            current_confirmed = confirmed[:]
            for j in range(1, len(A)):
                if confirmed[j - 1]:
                    continue
                if A[j - 1][i] < A[j][i]:
                    current_confirmed[j - 1] = True
                if A
[j - 1][i] > A[j][i]:
                    result += 1
                    has_deleted = True
                    break
                if A[j - 1][i] == A[j][i]:
                    has_equal = True
            if not has_deleted:
                if not has_equal:
                    return result
                confirmed = current_confirmed
        return result
