class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        c = collections.Counter(A)
        #print (c)
        for key in sorted(c):
            #print (key, c[key])
            while c[key] > 0:
                print (key) 
                if key >= 0:
                    
                    if c[2*key] <= 0 :
                        return False
                    c[key] -= 1
                    c[2*key] -= 1
                else:
                    if (-1*key) % 2 == 1 or c[key//2] <= 0:
                        return False
                    c[key] -= 1
                    c[key//2] -= 1
                        
        return True
