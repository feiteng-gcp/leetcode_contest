import collections
class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        counter = collections.Counter(A)
        A.sort(key = lambda x: abs(x))
        for a in A:
            if counter[a] == 0:
                continue
            if 2 * a not in counter or counter[2 * a] == 0:
                return False
            counter[2*a] -= 1
            counter[a] -= 1
        return True
