class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        # like a heap with right child is double of left child
        A = sorted(A)
        from collections import Counter
        counts = Counter(A)
        for i in range(len(A)-1, -1, -1):
            a = A[i]
            if a >= 0 and counts[a] > 0:
                if a % 2 == 1: return False # odd number cannot be paired with a half
                b = a // 2
                if counts[b] <= 0: return False # half doen't exist
                counts[a] -= 1
                counts[b] -= 1
            elif a < 0 and counts[a] > 0:
                b = a * 2
                if counts[b] <= 0: return False
                counts[a] -= 1
                counts[b] -= 1
        for k, v in counts.items():
            if v != 0: return False
        return True
