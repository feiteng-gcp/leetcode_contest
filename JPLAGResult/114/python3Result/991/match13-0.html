class Solution:
    def isok(self, arr):
        nums = list(set(arr))
        nums.sort()
        dicts = {}
        for i in arr:
            if i not in dicts:
                dicts[i] = 1
            else:
                dicts[i] += 1
        for i in nums:
            if dicts[i] == 0:
                continue
            elif dicts[i] < 0:
                return False
            else:
                if 2 * i not in dicts:
                    return False
                else:
                    dicts[2 * i] -= dicts[i]
                    dicts[i] = 0
        for k, v in dicts.items():
            if v != 0:
                return False
        return True
        
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        n0 = 0
        neg = []
        pos = []
        for i in A:
            if i < 0:
                neg.append(i)
            elif i > 0:
                pos.append(i)
            else:
                n0 += 1
        if n0 % 2 != 0:
            return False
        if not self.isok(pos):
            return False
        for i in range(len(neg)):
            neg[i] = -neg[i]
        if not self.isok(neg):
            return False
        return True
