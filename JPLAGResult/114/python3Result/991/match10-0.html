class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if not A:
            return True
        c = collections.defaultdict(int)
        for a in A:
            c[a] += 1
#        print("bef " , c)
        for k in sorted(A):
            if c[k] <= 0:
                continue
            k2 = 2*k
            if k2 in c and c[k2]>0:
                c[k2] -= 1
                c[k] -= 1
#            print(c)
#                if c[k2] == 0:
#                    c.pop(k2)
#                if c[k] == 0:
#                    c.pop(k)
        
        for k,v in c.items():
            if v>0:
                return False
        return True
