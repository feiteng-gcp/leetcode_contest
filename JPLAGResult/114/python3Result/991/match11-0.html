from collections import Counter
class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        zero = []
        pos = []
        neg = []
        for i in A:
            if i > 0:
                pos.append(i)
            elif i == 0:
                zero.append(i)
            else:
                neg.append(i)
        if len(zero) %2 or len(pos)%2 or len(neg)%2:
            return False
        posd = dict()
        negd = dict()
        pos.sort()
        neg.sort(reverse=True)
        for i in pos:
            if i%2 == 0 and i/2 in posd and posd[i/2] > 0:
                posd[i/2] -=1
            elif i not in posd:
                posd[i] = 1
            else:
                posd[i] += 1
        for i in neg:
            if i%2 == 0 and i/2 in negd and negd[i/2] > 0:
                negd[i/2] -= 1
            elif i not in negd:
                negd[i] = 1
            else:
                negd[i] += 1
        # print(posd, negd)
        for val in posd.values():
            if val != 0:
                return False
        for val in negd.values():
            if val != 0:
                return False
        return True
