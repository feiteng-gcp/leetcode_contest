import collections

class Solution:

    def _can_reorder(self, A):
        if len(A) == 0:
            return True
        elif len(A) % 2 == 1:
            return False
        counter = collections.Counter(A)
        keys = sorted(counter.keys())

        for k in keys:
            k_count = counter[k]
            if k_count == 0:
                continue
            counter.pop(k)
            if counter[k*2] < k_count:
                return False
            else:
                counter
[k*2] -= k_count
        for k, v in counter.items():
            if v > 0:
                return False
        return True

    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        # grouping
        if len(A) == 0:
            return True
        elif len(A) % 2 == 1:
            return False
        zero_count = 0
        neg = []
        pos = []
        for a in A:
            if a == 0:
                zero_count += 1
            elif a > 0:
                pos.append(a)
            else:
                neg.append(a)
        if zero_count % 2 == 1:
            return False

        return self._can_reorder(pos) and self._can_reorder([-1 * n for n in neg])
