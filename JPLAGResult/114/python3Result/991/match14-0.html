class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        if len(A) == 0:
            return True
        import collections
        c_pos = collections.defaultdict(int)
        c_neg = collections.defaultdict(int)
        zero = 0
        for num in A:
            if num > 0:
                c_pos[num] += 1
            elif num < 0:
                c_neg[num] += 1
            else:
                zero += 1
        if zero%2 != 0:
            return False
        f = True
        f1 = True
        if len(c_pos) != 0:
            k = sorted(c_pos.keys())
            for i in range(len(k)-1):
                if c_pos[k[i]] > 0:
                    if 2*k[i] not in c_pos:
                        return False
                    if c_pos[k[i]] > c_pos[2*k[i]]:
                        return False
                    c_pos[2*k[i]] -= c_pos[k[i]]
            f = (c_pos[k[-1]] == 0)
        if len(c_neg) != 0:
            k1 = sorted(c_neg.keys(), reverse=True)
            for i in range(len(k1)-1):
                if c_neg[k1[i]] > 0:
                    if 2*k1[i] not in c_neg:
                        return False
                    if c_neg[k1[i]] > c_neg[2*k1[i]]:
                        return False
                    c_neg[2*k1[i]] -= c_neg[k1[i]]
            f1 = (c_neg[k1[-1]] == 0)
        return f1 and f
