class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        Ap = [a for a in A if a > 0]
        An = [a for a in A if a < 0]
        
        def f(A):
            s = collections.Counter(A)

            b = sorted(A)
            if b and b[0] < 0:
                b = b[::-1]

            
            for c in b:
                if s[c] == 0:
                    continue
                if s[2*c] == 0:
                    # print(s)
                    # print(c)
                    return False
                s[c] -= 1
                s[2*c] -= 1
            return True
        
        return f(Ap) and f(An)
