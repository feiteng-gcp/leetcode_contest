class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        A_ = sorted(A)
        dic = {}
        for i in A_:
            dic[i] = dic.get(i, 0) + 1
        seq = sorted(list(set(A_)))
        for i in seq:
            # print(i, dic)
            if dic[i] == 0:
                continue
            elif i < 0:
                if i % 2:
                    return False
                else:
                    check = i/2
                    if check not in dic:
                        return False
                    elif dic[i] <= dic[check]:
                        dic[check] -= dic[i]
                        dic[i] = 0
                    else:
                        return False
            else:
                check = i*2
                if check not in dic:
                    return False
                elif dic[i] <= dic[check]:
                    dic[check] -= dic[i]
                    dic[i] = 0
                else:
                    return False
        for i in seq:
            if dic[i] != 0:
                return False
        return True
