class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """

        count = {x:0 for x in A}
        for x in A:
            count[x] += 1
          
        A.sort()
        for x in A:
            # print(x,count)
            if count[x] > 0:
                if (2*x in count) and (count[2*x] > 0):
                    count[x] -= 1
                    count[2*x] -= 1
                elif x!= -1 and x//2 in count and count[x//2] > 0:
                    count[x] -= 1
                    count[x//2] -= 1
                else:
                    return False
            else:
                if 2*x in count and count[2*x]==0:
                    pass
                elif x//2 in count and count[x//2]==0:
                    pass
                else:
                    return False
