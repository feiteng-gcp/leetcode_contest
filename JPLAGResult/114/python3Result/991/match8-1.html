from collections import Counter
class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        cnt = Counter(A)
        A.sort()
        B, C = [], []
        zcnt = 0
        for x in A:
            if x < 0:
                B.append(x)
            elif x == 0:
                zcnt += 1
            else:
                C.append(x)
        if zcnt % 2 != 0:
            return False
        B = B[::-1]
        used = Counter()
        for x in B:
            if used[x] > 0:
                used[x] -= 1
                continue
            if cnt[x*2] <= 0:
                return False
            cnt[x*2] -= 1
            used[x*2] += 1
        used = Counter()
        for x in C:
            if used[x] > 0:
                used[x] -= 1
                
continue
            if cnt[x*2] <= 0:
                return False
            cnt[x*2] -= 1
            used[x*2] += 1
        return True
