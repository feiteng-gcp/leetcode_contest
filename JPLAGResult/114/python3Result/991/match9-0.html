class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        
        
        A.sort()
        negatives = []
        zeros_count = 0
        positives = []
        for a in A:
            if a < 0:
                negatives.append(a)
            elif a == 0:
                zeros_count += 1
            else:
                positives.append(a)
        
        if zeros_count % 2 != 0:
            return False
        else:
            a = self.helper(negatives) 
            b = self.helper(positives[::-1])
            return a and b

    def helper(self, arr):
        m = {}
        for a in arr:
            if a * 2 in m and m[a * 2] > 0:
                m[a * 2] -= 1
                if m[a * 2] == 0:
                    del m[a * 2]
            elif a not in m:
                m[a] = 1
            else:
                m[a] += 1
        
        return len(m.items()) == 0
