from collections import defaultdict

class Solution:
    def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """

        cnt = defaultdict(int)
        for a in A:
            cnt[a] += 1

        A.sort(key=abs)
        n = len(A)
        for i in range(n):
            if cnt[A[i]]:
                
if cnt[2*A[i]] < cnt[A[i]]: return False
                cnt[2*A[i]] -= cnt[A[i]]
                cnt[A[i]] = 0
        return True
