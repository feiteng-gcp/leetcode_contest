class Solution:
    
def canReorderDoubled(self, A):
        """
        :type A: List[int]
        :rtype: bool
        """
        a1 = []
        a2 = []
        a0 = 0
        for a in A:
            if a < 0:
                a1.append(a)
            elif a > 0:
                a2.append(a)
            else:
                a0 += 1
        if a0 % 2 != 0 or len(a1) % 2 != 0 or len(a2) % 2 != 0:
            return False

        a1.sort()
        a1.reverse()
        a2.sort()
        dic = {}
        for a in a1:
            if a not in dic:
                dic[a] = 0
            dic[a] += 1

        for a in a1:
            temp = 2*a
            if dic[a] == 0:
                continue

            if temp in dic and dic[temp] > 0:
                dic[a] -= 1
                dic[temp] -= 1
            else:
                return False

        dic = {}
        for a in a2:
            if a not in dic:
                dic[a] = 0
            dic[a] += 1

        for a in a2:
            if dic[a] == 0:
                continue

            temp = 2*a
            if temp in dic and dic[temp] > 0:
                dic[a] -= 1
                dic[temp] -= 1
            else:
                return False
        return True
