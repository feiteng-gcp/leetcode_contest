class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        lookup = {e: i for i, e in enumerate(order)}

        def compare(a, b):
            n = len(a)
            m = len(b)
            for i in range
(min(n, m)):
                if lookup[a[i]] < lookup[b[i]]:
                    return -1
                if lookup[a[i]] > lookup[b[i]]:
                    return 1
            if n < m:
                return -1
            if n > m:
                return 1
            return 0

        
for i in range(len(words) - 1):
            if compare(words[i], words[i + 1]) == 1:
                return False
        return True
