class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        order = {c:i for i, c in enumerate(order)}
        
        def gte(w1, w2):
            
for c1, c2, in zip(w1, w2):
                if order[c2] < order[c1]:
                    return False
                if order[c2] > order[c1]:
                    return True
            return len(w1) < len(w2)
        
        return all(gte(w1, w2) for w1, w2 in zip(words, words[1:]))
