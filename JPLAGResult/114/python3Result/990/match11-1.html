class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """

        def isBigger(w0, w1, od):
            m = min(len(w0), len(w1))
            for x in range(m):
                if w0[x] == w1[x]: continue
                
if od[w0[x]] > od[w1[x]]: return False
                if od[w0[x]] < od[w1[x]]: return True
            # follow order in m char
            if len(w0) > len(w1): return False
            return True

        od = dict()
        for x, w in enumerate(order):
            od[w] = x

        
for x in range(len(words)-1):
            if not isBigger(words[x], words[x+1], od):
                return False
        return True
