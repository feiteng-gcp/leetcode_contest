class Solution:
    def isAlienSorted(self, words, order):
        m={}
        i=0
        for x in order:
            m[x]=i
            i+=1
        for (i,v) in enumerate(words):
            words[i]=[m[item] for item in v]
        # print(words)
        ori=words[0]
        for x in range(1,len(words)):
            value=words[x]
            if not self.valid(ori,value):
                return False
            ori=value
        return True

    def valid(self, v1,v2):
        mi=min(len(v1),len(v2))
        if len(v1)!=len(v2):
            if v1[:mi]==v2[:mi]:
                return len(v1)==mi
        for x in range(mi):
            if v1[x]<v2[x]:
                return True
            if v1[x]>v2[x]:
                return False
        return True
