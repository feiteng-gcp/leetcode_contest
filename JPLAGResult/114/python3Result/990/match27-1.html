class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        from string import ascii_lowercase
        mapp = {k:v for k, v in zip(order, ascii_lowercase)}
        def trans(word):
          return [mapp[c] for c in word]
        transed = [trans(word) for word in words]
        
for i in range(0, len(transed)-1):
          if transed[i] > transed[i+1]:
            return False
        return True
