class Solution:
    def isAlienSorted(self, words, order):
        def judge(w1,w2,o):
            i=0
            while(i<len(w1) and i < len
(w2)):
                if(o[w1[i]] < o[w2[i]]):
                    return True
                elif(o[w1[i]]>o[w2[i]]):
                    return False
                else:
                    i+=1
            if(i<len(w1)):
                return False
            else:
                return True
        o
={}
        for i in range(len(order)):
            o[order[i]]=i
        for i in range(len(words)-1):
            if(not judge(words[i],words[i+1],o)):
                return False
        return True
