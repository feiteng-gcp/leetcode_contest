class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        dic = collections.defaultdict()
        def bigger(str1, str2):
            for i in range(min(len(str1), len(str2))):
                # print(dic[str1[i]], dic[str2[i]])
                if dic[str1[i]] > dic[str2[i]]:
                    return True
                elif dic[str1[i]] == dic[str2[i]]:
                    continue
                else:
                    return False
            if len(str2) > len(str1):
                return False
            return True
        
        for i in range(len(order)):
            dic[order[i]] = i
            
        
for i in range(1, len(words)):
            if not bigger(words[i], words[i - 1]):
                return False
        return True
