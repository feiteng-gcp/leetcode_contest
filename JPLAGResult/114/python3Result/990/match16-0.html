class Solution:
    def inorder(self, word1, word2, dicts):
        if word1 == word2:
            return True
        w1 = list(word1)
        w2 = list(word2)
        n1 = len(w1)
        n2 = len(w2)
        loc = 0
        while loc < min(n1, n2):
            if dicts[w1[loc]] < dicts[w2[loc]]:
                return True
            elif dicts[w1[loc]] > dicts[w2[loc]]: 
                return False
            loc += 1
        return n1 < n2
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        orders = list(order)
        dicts = {}
        for i in range(len(orders)):
            dicts[order[i]] = i
        for i in range(len(words) - 1):
            if not self.inorder(words[i], words[i + 1], dicts):
                return False
        return True
