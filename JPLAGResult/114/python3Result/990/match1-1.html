class Solution:
    
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        
        order_map = {}
        for index, c in enumerate(order):
            order_map[c] = index
        
        
        def cmp_word(a, b):
            length = min(len(a), len(b))
            for i in range
(length):
                if order_map[a[i]] > order_map[b[i]]:
                    return 1
                elif order_map[a[i]] < order_map[b[i]]:
                    return -1
            
            return len(a) > len(b)
        
        for i in range(1, len(words)):
            if cmp_word(words[i - 1], words[i]) > 0:
                return False
        return True
