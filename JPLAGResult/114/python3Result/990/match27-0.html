class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        trans=str.maketrans(order,'abcdefghijklmnopqrstuvwxyz')
        for i in range(len(words)):
            words[i]=words[i].translate(trans)
        for i in range(len(words)-1):
            if words[i+1]<words[i]:
                return False
        return True
