class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        dict1 = {order[i]:i for i in range(len(order)) }

        def less(w1, w2):
            for i in range(min(len(w1), len(w2))):
                if dict1[w1[i]] < dict1[w2[i]]:
                    return True
                elif dict1[w1[i]] > dict1[w2[i]]:
                    return False
                else:
                    pass
                
            return len(w1) <= len(w2)

            
        for i in range(1, len(words)):
            if not less(words[i-1], words[i]):
                return False
            
        return True
