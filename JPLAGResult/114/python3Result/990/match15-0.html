class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """
        def is_sorted(w1,w2):
            for i in range(min(len(w1),len(w2))):
                if order_d[w1[i]]<order_d[w2[i]]:
                    return True
                if order_d[w1[i]]>order_d[w2[i]]:
                    return False
            if len(w1)>len(w2):
                return False
            return True
        if len(words)<2:
            return True
        order_d={}
        for i in range(len(order)):
            order_d[order[i]]=i
        for i in range(len(words)-1):
            if not is_sorted(words[i],words[i+1]):
                return False
        return True
