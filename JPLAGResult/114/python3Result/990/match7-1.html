class Solution:
    def isAlienSorted(self, words, order):
        """
        :type words: List[str]
        :type order: str
        :rtype: bool
        """

        def bigger(mp, w0, w1):
            n0, n1 = len(w0), len(w1)
            for i in range
(min(n0, n1)):
                if mp[w0[i]] < mp[w1[i]]: return False
                elif mp[w0[i]] > mp[w1[i]]: return True
            return n0 > n1

        mp = {c: i for i, c in enumerate(order)}
        n = len(words)
        for i in range(n-1):
            if bigger(mp, words[i], words[i+1]): return False
        return True
