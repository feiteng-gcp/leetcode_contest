class Solution {
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        int[][] dp = new int[n][n];
        int[][] m = new int[n][n];
        for (int i = 0; i < n; i++) m[i][i] = arr[i]
;
        for (int len = 1; len < n; len++) {
            for (int i = 0; i < n - len; i++) {
                int j = i + len;
                int temp = Integer.MAX_VALUE;
                for (int k = i; k < j; k++) {
                    temp = Math.min(temp, m[i][k] * m[k+1][j] + dp[i][k] + dp[k+1][j]);
                }
                m[i][j] = Math.max(m[i][i], m[i+1][j]);
                dp[i][j] = temp;
            }
        }
        return dp[0][n-1];
    }
}
