class Solution {
    int[][] memo;
    int[][] max; // max element from i to j
    
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        memo = new int[n][n];
        max = new int[n][n];
        return helper(arr, 0, n-1);
    }
    
    // build a tree from index i to j, smallest non-leaf sum
    private int helper(int[] arr, int i, int j) {
        if (i >= j) return 0;
        if (j - i == 1) return arr[i]*arr[j];
        
        if (memo[i][j] > 0) return memo[i][j];
        
        int ans = Integer.MAX_VALUE;
        for (int k = i; k < j; k++) {
            int temp = 0;
            int left = getMax(arr, i, k);
            int right = getMax(arr, k+1, j);
            temp += helper(arr, i, k);
            temp += helper(arr, k+1, j);
            temp += left*right;
            ans = Math.min(ans, temp);
        }
        memo[i][j] = ans;
        return ans;
    }
    
    private int getMax(int[] arr, int i, int j) {
        if (max[i][j] > 0) return max[i][j];
        
        int ans = 0;
        for (int k = i; k <= j; k++) {
            ans = Math.max(ans, arr[k]);
        }
        max[i][j] = ans;
        return ans;
    }
}
