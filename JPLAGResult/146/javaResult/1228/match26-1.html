class Solution {
    // Num of non-leaf nodes are same
    // f[i..j] = (f[i..k] + f[k+1..j]) + max[i..k] * max[k+1..j];
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        long[][] f = new long[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                f[i][j] = Integer.MAX_VALUE;
            }
        }
        for (int i = 0; i < n; i++) {
            f[i][i] = 0;
        }
        
        
int[][] max = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                max[i][j] = arr[i];
                for (int k = i; k <= j; k++) {
                    max[i][j] = Math.max(max[i][j], arr[k]);
                
}
            }
        }
        
        for (int d = 2; d <= n; d++) {
            for (int i = 0; i < n; i++) {
                int j = i + d - 1;
                if (j >= n) continue;
                for (int k = i; k < j; k++) {
                    f[i][j] = Math.min(f[i][j], f[i][k] + f[k+1][j] + max[i][k] * max[k+1][j]);
                
}
            }
        }
        return (int)f[0][n-1];
    }
}
