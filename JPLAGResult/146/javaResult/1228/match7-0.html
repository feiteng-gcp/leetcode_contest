class Solution {
    int[][][] dp;
    public int mctFromLeafValues(int[] arr) {
        int len = arr.length;
        dp = new int[len][len][2];
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < len; j++) dp[i][j][1] = -1;
        }
        int[] ret = helper(arr, 0, arr.length - 1);
        return ret[1];
    }
    public int[] helper(int arr[], int from, int to) {
        if (from == to) {
            return new int[]{arr[from], 0};
        }
        if (dp[from][to][1] != -1) return dp[from][to];
        int min = Integer.MAX_VALUE;
        int leaf = 0;
        for (int i = from; i < to; i++) {
            int[] left = helper(arr, from, i);
            int[] right = helper(arr, i + 1, to); 
            int sum = left[0] * right[0] + left[1] + right[1];
            //System.out.println("" + left[0] + "," + left[1] + "," + right[0] + "," + right[1] + "," + sum);
            if (sum < min) {
                min = sum;
                leaf = Math.max(left[0], right[0]);
            }
        }
        dp[from][to][0] = leaf;
        dp[from][to][1] = min;
        return dp[from][to];
    }
    
}
