class Solution {
    int[][] mx;
    int[][] dp;
    int n;
    public int mctFromLeafValues(int[] arr) {
        n = arr.length;
        mx = new int[n][n];
        for (int i = 0; i < n; i++) for (int j = i; j < n ;j++) {
            int m = arr[i];
            for (int k = i; k <= j; k++) {
                m = Math.max(m, arr[k]);
            }
            mx[i][j] = m;
        }
        dp = new int[n][n];
        for (int i = 0; i < n; i++) Arrays.fill(dp[i], -1);
        return solve(arr, 0, n-1);
    }
    
    int solve(int[]arr, int l, int r) {
        if (l == r) {
            return 0;
        }
        if (dp[l][r] != -1) return dp[l][r];
        
        int res = Integer.MAX_VALUE;
        for (int i = l; i < r; i++) {
            int[] a = new int[i - l + 1];
            int[] b = new int[r - i];
            int ma = mx[l][i];
            int mb = mx[i+1][r];
            
            int tmp = ma * mb;
            tmp += solve(a, l, i);
            tmp += solve(b, i+1, r);
            res = Math.min(res, tmp);
        }
        dp[l][r] = res;
        return res;
    }
    
}
