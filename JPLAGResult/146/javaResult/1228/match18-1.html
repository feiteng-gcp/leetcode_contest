class Solution {
    public int mctFromLeafValues(int[] arr) {
        if(arr == null || arr.length == 0){
            return 0;
        }
        
int len = arr.length;
        
        int[][] dp = new int[len][len];
        int[][] maxVal = new int[len][len];
        
        for(int i = 0; i < len; i++){
            dp[i][i] = 0;
            maxVal[i][i] = arr[i];
        }
        
        for(int distance = 1; distance < len; distance++){
            for(int i = 0; i + distance < len; i++){
                
int j = i + distance;
                dp[i][j] = Integer.MAX_VALUE;
                
                for(int k = i; k < j; k++){
                    dp[i][j] = Math.min(dp[i][j], dp[i][k] + dp[k + 1][j] + maxVal[i][k] * maxVal[k + 1][j]);
                }
                maxVal[i][j] = Math.max(maxVal[i][j - 1], arr[j]);
            }
        }
        
        return dp[0][len - 1];
    }
}
