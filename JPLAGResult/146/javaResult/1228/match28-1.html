class Solution {
     long[][] dp;
    long[][] mx;

    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        mx = 
new long[n][n];

        for (int i = 0; i < n; i++) {
            int max = 0;
            for (int j = i; j < n; j++) {
                max = Math.max(max, arr[j]);
                mx[i][j] = max;
            }
        }

        dp = new long[n + 1][n + 1];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                dp[i][j] = -1;
            }
            dp[i][i] = 0;
        }
        
        return (int) dfs(0, n - 1);
    }

    private long dfs(int i, int j) {
        if (i == j) {
            return 0;
        } else if (dp[i][j] >= 0) {
            return dp[i][j];
        }
        long inf = 1L << 35;
        
long ans = inf;

        for (int k = i; k < j; k++) {
            long l = dfs(i, k);
            long r = dfs(k + 1, j);

            long sum = mx[i][k] * mx[k + 1][j] + l + r;

            ans = Math.min(ans, sum);
        }
        
        dp[i][j] = ans;
        return ans;
    }
}
