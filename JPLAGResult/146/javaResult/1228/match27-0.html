class Solution {
    public int mctFromLeafValues(int[] arr) {
        int [][] maxVal = new int[arr.length][arr.length];
        for (int i = 0; i < arr.length; i++) {
            maxVal[i][i] = arr[i];
        }
        for (int i = 0; i < arr.length; i++) {
            for(int j = i + 1;j < arr.length; j++) {
                maxVal[i][j] = Math.max(maxVal[i][j - 1], arr[j]);
            }
        }
        int [][] dp =  new int[arr.length][arr.length];
        for (int i = 0; i < arr.length; i++) {
            dp[i][i] = arr[i];
        }  
        for (int i = 1; i < arr.length; i++) {
            for(int j = i - 1; j >= 0; j--) {
                dp[j][i] = Integer.MAX_VALUE;
                for(int k = j; k < i; k++) {
                    dp[j][i] = Math.min(dp[j][i], dp[j][k] + dp[k + 1][i] + maxVal[j][k] * maxVal[k + 1][i]);
                }
            }
        }
        int sum = 0;
        for (int i = 0; i <arr.length; i++) {
            sum += arr[i];
        } 
        return dp[0][arr.length - 1] - sum;
    }
}
