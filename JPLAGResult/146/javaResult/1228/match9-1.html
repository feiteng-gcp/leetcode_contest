class Solution {
    public int mctFromLeafValues(int[] arr) {
        int[][] max = new int[arr.length][arr.length];
        
int[][] sum = new int[arr.length][arr.length];
        for(int len = 1 ; len <= arr.length ; len++) {
            for(int start = 0 ; start + len - 1 < arr.length ; start++) {
                int end = start + len - 1;
                if(start == end) {
                    max[start][end] = arr[start];
                    sum[start][end] = 0;
                } else {
                    max[start][end] = Math.max(max[start][end - 1], arr[end]);
                    
int curSum = Integer.MAX_VALUE;
                    for(int mid = start + 1; mid <= end ; mid++) {
                        curSum = Math.min(curSum, max[start][mid - 1] * max[mid][end] + sum[start][mid - 1] + sum[mid][end]);
                    }
                    sum[start][end] = curSum;
                
}
            }
        }
        return sum[0][arr.length - 1];
    }
}
