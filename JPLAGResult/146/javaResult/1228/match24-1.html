class Solution {
    public int mctFromLeafValues(int[] arr)
	{
		ArrayList<Integer> list = new ArrayList<Integer>();
		for (int i : arr)
			list.add(i);
		int sum = 0;
		while (list.size() > 1)
		{
			int index = find(list);
			sum += list.get(index) * list.get(index + 1);
			list.set(index, Math.max(list.get(index), list.get(index + 1)));
			list.remove(index + 1);
		}

		return sum;
	}

	/**
	 * @param list
	 * @return
	 */
	private int find(ArrayList<Integer> list)
	{
		int min = Integer.MAX_VALUE;
		int index = -1;
		for (int i = 0
; i < list.size() - 1; i++)
		{
			if (list.get(i) * list.get(i + 1) < min)
			{
				min = list.get(i) * list.get(i + 1);
				index = i;
			}
		}

		return index;
	}
}
