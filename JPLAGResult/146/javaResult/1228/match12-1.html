class Solution {
    public int mctFromLeafValues(int[] arr) {
        int len = arr.length;
        int[][] rangeMax = 
new int[len][len];
        for(int i = 0; i < arr.length; i++) {
            int currMax = 0;
            for(int j = i; j < arr.length; j++) {
                currMax = Math.max(currMax, arr[j]);
                rangeMax[i][j] = currMax;
            }
        }
        return dfs(arr, 0, arr.length - 1, new int[len][len], rangeMax);
    }
    
    public int dfs(int[] arr, int lo, int hi, 
int[][] memo, int[][] rangeMax) {
        if(lo == hi) return 0;
        if(memo[lo][hi] > 0) return memo[lo][hi];
        int min = Integer.MAX_VALUE;
        for(int i = lo; i < hi; i++) {
            int left = dfs(arr, lo, i, memo, rangeMax);
            int right = dfs(arr, i + 1, hi, memo, rangeMax);
            int sum = left + right + rangeMax[lo][i] * rangeMax[i + 1][hi];
            min = Math.min(min, sum);
        }
        memo[lo][hi] = min;
        return min;
    }
}
