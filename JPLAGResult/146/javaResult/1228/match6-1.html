class Solution {
    public int mctFromLeafValues(int[] arr) {
        
        int n = arr.length;
        
        int[][] map = 
new int[n][n];
        
        return helper(map, arr, 0, n-1);
        
    }
    
    public int helper(int[][] map, int[] arr, int start, int end){
    
        
if(start == end) return 0;
        
        if(map[start][end] != 0) return map[start][end];
        
        int min = Integer.MAX_VALUE;
        
        for(int i = start; i < end; i++){
            
            int left = find(arr, start, i);
            int right = find(arr, i+1, end);
            
            min = Math.min(min, helper(map, arr, start, i) + 
helper(map, arr, i+1, end) + left * right);
        }
        
        map[start][end] = min;
    
        return map[start][end];   
    }
    
    public int find(int[] arr, int start, int end){
        int max = 0;
        for(int i = start; i <= end; i++){
            max = Math.max(max, arr[i]);
        }
        return max;
    }
}
