class Solution {
    public int mctFromLeafValues(int[] arr) {
        int l = arr.length;
        int[][] dp = new int[l][l];
        int[][] max = new int[l][l];
        for (int i = 0; i < l - 1; i++) {
            dp[i][i + 1] = arr[i] * arr[i + 1];
        }
        for (int i = 0; i < l; i++) {
            int t = 0;
            for (int j = i; j < l; j++) {
                t = Math.max(t, arr[j]);
                max[i][j] = t;
            }
        }
        for (int k = 2; k < l; k++) {
            for (int i = 0; i < l - k; i++) {
                int j = i + k, t = Integer.MAX_VALUE;
                for (int m = i; m < j; m++) {
                    // System.out.printf("%d %d %d, %d %d %d\n", i, j, m, dp[i][m] , dp[m + 1][j], max[i][m] * max[m + 1][j]);
                    t = Math.min(t, dp[i][m] + dp[m + 1][j] + max[i][m] * max[m + 1][j]);
                }
                dp[i][j] = t;
                // System.out.printf("%d %d %d, %d \n", i, j, t, dp[i][j]);
            }
        }
        return dp[0][l - 1];
    }
}
