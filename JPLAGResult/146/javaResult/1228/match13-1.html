class Solution {
    Map<List<Integer>, Integer> mem = new HashMap<>();
    
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        int[][] maxmatrix = 
new int[n][n];
        for(int i = 0; i < n; i++) {
            int max = arr[i];
            for(int j = i; j < n; j++){
                max = Math.max(max, arr[j]);
                maxmatrix[i][j] = max;
                
maxmatrix[j][i] = max;
            }
        }
        
        return search(arr, 0, n-1, maxmatrix);
    }
    
    private int search(int[] arr, int l, int r, int[][] maxmatrix){
        if(l == r) return 0;
        if(l + 1 == r) return arr[l] * arr[r];
        List<Integer> key = Arrays.asList(l, r);
        if(mem.containsKey(key)) return mem.get(key)
;
        
        int res = Integer.MAX_VALUE;
        for(int m = l; m < r; m++){
            int lres = search(arr, l, m, maxmatrix);
            int rres = search(arr, m+1, r, maxmatrix);
            int lm = maxmatrix[l][m];
            int rm = maxmatrix[m+1][r];
            res = Math.min(res, lres + rres + lm * rm);
        }
        
        mem.put(key, res);
        return res;
    }
}
