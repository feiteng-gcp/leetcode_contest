class Solution {
    Integer[][] f;

	public int mctFromLeafValues(int[] arr) {

		int n = arr.length;
		f = new Integer[n][n];

		return dp(0, n - 1, arr);

	}

	private int dp(int x, int y, int[] arr) {
		if (f[x][y] == null) {

			if (x == y) {
				f[x][y] = 0;
			} else {
				int res = Integer.MAX_VALUE;
				for (int i = x; i < y; i++) {
					int l = dp(x, i, arr);
					int r = dp(i + 1, y, arr);
					res = Math.min(res, l + r + max(x, i, arr) * max(i + 1, y, arr));
				}
				f[x][y] = res;
			}
		}
		return f[x][y];
	}

	private int max(int x, int y, int[] arr) {
		int res = Integer.MIN_VALUE;
		for (int i = x; i <= y; i++) {
			res = Math.max(arr[i], res);
		}
		return res;
	}
}
