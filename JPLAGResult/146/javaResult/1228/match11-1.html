class Solution {
    public int mctFromLeafValues(int[] arr) {
        //dp[i][j] -> answer for [i] to [j], includej
        //dp[i][j] = loop (dp[i][k] + dp[k+1][j] + max[i][k]*max[k+1][j])
        
        int n = arr.length;
        int[][] dp = new int[n][n];
        int[][] max = new int[n][n];
        
        
for (int i = 0; i < n; i++){
            max[i][i] = arr[i];
            for (int j = i+1; j < n; j++){
                max[i][j] = Math.max(max[i][j-1], arr[j]);
            }
        }
        
        for (int i = 0; i < n; i++){
            dp[i][i] = 0;
        
}
        
        for (int gap = 1; gap < n; gap++){
            for (int i = 0; i+gap < n; i++){
                int j = i+gap;
                dp[i][j] = Integer.MAX_VALUE;
                for (int k = i; k+1 <= j; k++){
                    int curr = dp[i][k] + dp[k+1][j] + max[i][k]*max[k+1][j];
                    
dp[i][j] = Math.min(dp[i][j], curr);
                }
            }
        }
        
        return dp[0][n-1];
        
    }
}
