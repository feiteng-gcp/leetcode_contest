class Solution {
    int N = 41;
    int[] arr;
    int[][] dp = new int[N][N];
    int[][] max = new int[N][N];    
    
    int go(int i, int j) {
        if (i == j) return 0;
        if (dp[i][j] != -1) return dp[i][j];
        int best = Integer.MAX_VALUE;
        for (int k = i; k < j; k++) {
            int sum = go(i, k) + go(k+1, j) + max[i][k]*max[k+1][j];
            best = Math.min(best, sum);
        }
        dp[i][j] = best;
        return best;
    }
    
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        this.arr = arr;
        for (int i = 0; i < n; i++) {
            int m = arr[i];
            for (int j = i; j < n; j++) {
                m = Math.max(m, arr[j]);
                max[i][j] = m;
            }
        }            
        for (int i = 0; i < n; i++) Arrays.fill(dp[i], -1);
        return go(0, n-1);
    }
}
