class Solution {
    
    int[][] mem, memMaxLeaf;
    public int mctFromLeafValues(int[] arr) {
        mem = new int[arr.length + 1][arr.length + 1];
        memMaxLeaf = new int[arr.length + 1][arr.length + 1];
        
        int[] res = dfs(arr, 0, arr.length);

        return res[1];
    }
    
    
    // return: [max, sum]
    int[] dfs(
int[] arr, int from, int to /* non inclusive */) {
        if (to - from == 1) {
            return new int[]{arr[from], 0};
        }

        if (mem[from][to] > 0) {
            
return new int[]{memMaxLeaf[from][to], mem[from][to]};
        }
        
        int minSum = Integer.MAX_VALUE;
        int maxLeaf = 0;
        for (int i = from + 1; i < to; i++) {
            int[] childL = dfs(arr, from, i), childR = dfs(arr, i, to);

            int childSum = childL[1] + childR[1] + childL[0] * childR[0];
            if (childSum < minSum) {
                minSum = childSum;
                maxLeaf = Math.max(childL[0], childR[0]);
            }
            
        }
        
        
mem[from][to] = minSum;
        memMaxLeaf[from][to] = maxLeaf;
        
        return new int[]{memMaxLeaf[from][to], mem[from][to]};
    }
}
