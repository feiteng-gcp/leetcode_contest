class Solution {
    public int mctFromLeafValues(int[] arr) {
        return g(arr,0,arr.length-1)[0];
    }
    
    HashMap<String,int[]> mp = new HashMap<>();
    
    
    public int[] g(int[] arr , int l ,int r){
        String s = l+","+r;
        if(mp.containsKey(s)){
            return mp.get(s);
        }
        
        if(l==r){
            return new int[]{0,arr[l]};
        }
        int ma = -1;
        for(int i=l;i<=r;++i){
            ma = Math.max(ma,arr[i]);
        }
        
        int res = Integer.MAX_VALUE;
        for(int i=l;i<=r-1;++i){
            int c1[] = g(arr,l,i);
            int c2[] = g(arr,i+1,r);
            int cur = c1[1]*c2[1]+c1[0]+c2[0];
            if(cur<res){
                res =cur;
            }
        }
        int x[] =new int[]{res,ma};
        mp.put(s,x);
        return x;
    }
}
