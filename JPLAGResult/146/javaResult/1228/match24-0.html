class Solution {
    public int mctFromLeafValues(int[] arr) {
        List<Integer>list = new LinkedList<>();
        for (int i : arr) list.add(i);
        int sum = 0;
        while (list.size() > 1) {
            int tmp = list.get(0) * list.get(1);
            int index = list.get(0) < list.get(1) ? 0 : 1;
            for (int i = 1; i < list.size() - 1; i++) {
                if (list.get(i) * list.get(i + 1) < tmp) {
                    tmp = list.get(i) * list.get(i + 1);
                    index = list.get(i) < list.get(i + 1) ? i : i + 1;
                }
            }
            sum += tmp;
            list.remove(index);
        }
        return sum;
    }
}
