class Solution {
    public int mctFromLeafValues(int[] arr) {
        
        // time: O(N^3)
        // space: O(N^2)
        
        int n = arr.length;
        int[][] dp = new int[n][n];
        
        for(int i = n - 1; i >= 0; i--)
        {
            for(int j = i + 1; j < n; j++)
            {
                int[] maxVal = new int[n];
                maxVal[j] = arr[j];
                for(int k = j - 1; k >= i; k--)
                    maxVal[k] = Math.max(maxVal[k+1], arr[k]);
                
                int curMax = Integer.MIN_VALUE;
                dp[i][j] = Integer.MAX_VALUE;
                for(int k = i; k < j; k++)
                {
                    curMax = Math.max(curMax, arr[k]);
                    
dp[i][j] = Math.min(dp[i][j], curMax * maxVal[k+1] + dp[i][k] + dp[k+1][j]);
                }
            }
        }
        
        return dp[0][n-1];
    }
}
