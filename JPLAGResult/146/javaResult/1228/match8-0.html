class Solution {
    public int mctFromLeafValues(int[] arr) {
        //edge cases
        if(arr.length == 0) {
            return 0;
        }
        
        int n = arr.length;
        int[][] dp0 = new int[n][n];
        int[][] dp1 = new int[n][n];
        boolean[][] visited = new boolean[n][n];
        
        return Util(arr, 0, n - 1, dp0, dp1, visited)[1];
    }
    
    /*
     *@Returns: int[]; 
     * int[0], largest of [arr[l] -> arr[r]]; cached in dp0
     * int[1], min sum of non-leaf node of[arr[l] -> arr[r]]; cached in dp1 
     */
    private int[] Util(int[] arr, int l, int r, int[][] dp0, int[][] dp1, boolean[][] visited) {
        //cache-hit
        if(visited[l][r]) {
            return new int[]{dp0[l][r], dp1[l][r]};
        }
        //base case
        if(l == r) {
            visited[l][r] = true;
            dp0[l][r] = arr[l];
            dp1[l][r] = 0;
            return new int[]{arr[l], 0};
        }
        //regular case
        int max = Integer.MIN_VALUE;
        int minSum = Integer.MAX_VALUE;
        for(int i = l; i < r; i++) {
            //divide to [l,i] and [i + 1, r]
            int[] left = Util(arr, l, i, dp0, dp1, visited);
            int[] right = Util(arr, i + 1, r, dp0, dp1, visited);
            max = Math.max(left[0], right[0]);
            minSum = Math.min(minSum, left[1] + right[1] + left[0] * right[0]);
        }
        visited[l][r] = true;
        dp0[l][r] = max;
        dp1[l][r] = minSum;
        return new int[]{max, minSum};
    }
}
