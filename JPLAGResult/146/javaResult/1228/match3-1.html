class Solution {
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        int[][] dp = new int[n][n];
        int[][] max = new int[n][n];
        for (int i = 0; i < n; i++) {
            dp[i][i] = arr[i];
            max[i][i] = arr[i];
        }
        for (int len = 1; len < n; len++) {
            
for (int st = 0; st + len < n; st++) {
                max[st][st+len] = Math.max(max[st][st+len-1], arr[st+len]);
            }
        }
        
        for (int len = 2; len <= n; len++) {
            for (int st = 0; st + len - 1 < n; st++) {
                int ed = st + len - 1;
                dp[st][ed] = (1 << 31) - 1;
                for (int k = st; k < ed; k++) {
                    dp[st][ed] = Math.min(dp[st][ed], dp[st][k] + dp[k+1][ed] + max[st][k] * max[k+1][ed]);
                }
            }
        }
        int sum = 0;
        for (int i = 0; i < n; i++) sum += arr[i];
        return dp[0][n-1] - sum;
    }
}
