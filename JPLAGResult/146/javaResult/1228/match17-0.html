class Solution {
    public int mctFromLeafValues(int[] arr) {
        int N = arr.length;
        int[][] dp = new int[N][N];
        int[][] max = new int[N][N];
        
        for (int i = 0; i < N; i ++) {
            int tmp = arr[i];
            for (int j = i; j < N; j ++) {
               max[i][j] = Math.max(tmp, arr[j]);
               tmp = Math.max(tmp, max[i][j]);
            }
        }
        
        for (int L = 1; L < N; L ++) {
            for (int i = 0; i < N; i ++) {
                int j = i + L; if (j >= N) continue;
                if(L == 1) {
                    dp[i][j] = arr[i]*arr[j];
                }
                else {   
                    dp[i][j] = Integer.MAX_VALUE;
                    for (int k = i; k < j ; k ++) {
                        dp[i][j] = Math.min(dp[i][j], dp[i][k] + dp[k+1][j] + max[i][k]*max[k+1][j]);
                    }
                }
            }
        }
        return dp[0][N-1];
        
    }
}
