class Solution {
    Integer[][] memForLeafValue;
    public int mctFromLeafValues(int[] arr) {
        if (arr.length < 2)
            return 0;

        
memForLeafValue = new Integer[arr.length][arr.length];
        return mctFromLeafValues(arr, 0, arr.length-1);
    }

    private int mctFromLeafValues(int[] arr, int begin, int end) {
        if (begin == end)
            return 0;

        if (memForLeafValue[begin][end] != null)
            return memForLeafValue[begin][end];

        int result = Integer.MAX_VALUE;
        for (int i=begin; i<end; i++) {
            int cur = mctFromLeafValues(arr, begin, i) + 
mctFromLeafValues(arr, i+1, end) +
                    getMaxLeafNumber(arr, begin, i) * getMaxLeafNumber(arr, i+1, end);
            result = Math.min(cur, result);
        }
        memForLeafValue[begin][end] = result;
        return result;
    }

    private int getMaxLeafNumber(int[] arr, int begin, int end) {
        int result = arr[begin];
        for (int i=begin+1; i<=end; i++) {
            result = Math.max(arr[i], result);
        }
        return result;
    }
}
