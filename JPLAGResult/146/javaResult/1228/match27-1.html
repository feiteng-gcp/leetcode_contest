class Solution {
    public int mctFromLeafValues(int[] arr) {        
        int n = arr.length;
        if (n == 0) return 0;
        if (n == 1) return arr[0];
        
        int[][] maxVal = new int[n][n];
        for (int i = 0; i < n; ++i) {
            int m = 0;
            for (int j = i; j < n; ++j) {
                m = Math.max(m, arr[j]);
                maxVal[i][j] = m;
            
}
        }
        
        int[][] dp = new int[n][n];
        
        for (int i = 0; i < n - 1; ++i) {
            dp[i][i + 1] = arr[i] * arr[i + 1];
        }
        
        for (int d = 2; d < n; ++d) {
            for (int i = 0; i < n - d; ++i) {
                dp[i][i + d] = Integer.MAX_VALUE;
                for (int j = i; j < i + d; ++j) {
                    // i ... j and j + 1 ... i + d
                    dp[i][i + d] = Math.min(dp[i][i + d], dp[i][j] + dp[j + 1][i + d] + maxVal[i][j] * maxVal[j + 1][i + d]);
                }
            }
        }
        
        return dp[0][n - 1];
    }
}
