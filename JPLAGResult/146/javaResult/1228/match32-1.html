class Solution {
    int [][][] dp;
    
int result = 0;
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        dp = new int[n][n+1][3];
  //      System.out.println(result);
        return helper(arr, 0, n)[0];
    }
    
    int[] helper(int[] arr, int s, int e) {
        if (e - s == 1) {
            dp[s][e][1]= arr[s];
   //         System.out.println(s+","+e+":"+dp[s][e][1]);
            return dp[s][e];
        }

        if(dp[s][e][1] > 0) {
            return dp[s][e];
        }
        int res = Integer.MAX_VALUE;
        
int t = 0;
        for (int i = s+1; i < e; i++) {
            int[] vs1 = helper(arr, s, i);
            int[] vs2 = helper(arr, i, e);
            res = Math.min(res, vs1[1] * vs2[1] + vs1[0] + vs2[0]);
            t = Math.max(t, vs1[1]);
            t = Math.max(t, vs2[1]);
        }
        dp[s][e][0] = res;
        dp[s][e][1] = t;
  //      System.out.println(s+","+e+":"+res+","+t+":");
        
        return dp[s][e];
    }
    
}
