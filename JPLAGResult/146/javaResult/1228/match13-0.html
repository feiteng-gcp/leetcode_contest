class Solution {
    int[][] dp;
    int[][] maxA;
    public int mctFromLeafValues(int[] arr) {
        int n = arr.length;
        dp = new int[n][n];
        maxA = new int[n][n];
        for(int i = 0; i < n; ++i) {
            int max = 0;
            for(int j = i; j < n; ++j) {
                max = Math.max(max, arr[j]);
                maxA[i][j] = max;
            }
        }
        return helper(0, n - 1);
    }
    int helper(int s, int e) {
        if(s == e) return 0;
        if(dp[s][e] != 0) return dp[s][e];
        int min = Integer.MAX_VALUE;
        for(int t = s; t < e; ++t) {
            int l = helper(s, t);
            int r = helper(t + 1, e);
            int tv = l + r + maxA[s][t] * maxA[t + 1][e];
            if(tv < min) min = tv;
        }
        dp[s][e] = min;
        return min;
    }
}
