class Solution {
   public static int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
		ArrayList<Integer>[] red = new ArrayList[n];
		ArrayList<Integer>[] blue = new ArrayList[n];
		for (int i = 0; i < n; i++) {
			red[i] = new ArrayList<>();
			blue[i] = new ArrayList<>();
		}

		for (int i = 0; i < red_edges.length; i++) {
			red[red_edges[i][0]].add(red_edges[i][1]);
		}
		for (int i = 0; i < blue_edges.length; i++) {
			blue[blue_edges[i][0]].add(blue_edges[i][1]);
		}

		int oo = 1000000000;
		int[][] dist = new int[2][n];
		for (int i = 0; i < 2; i++)
			Arrays.fill(dist[i], oo);
		dist[0][0] = 0;
		dist[1][0] = 0;
		
		ArrayDeque<Integer> q = new ArrayDeque<>();
		
		//0 = red, 1 = blue
		q.add(0);
		q.add(0);
		q.add(0);
		q.add(1);
		while(!q.isEmpty()) {
			int node = q.poll();
			int prev = q.poll();
//			System.out.println(node+" "+prev);
			if(prev==1) {
				for(int i : red[node]) {
					if(dist[0][i] > dist[prev][node]+1){
						dist[0][i] = dist[prev][node]+1;
						q.add(i);
						q.add(0);
					}
				}
			} else {
				for(int i : blue[node]) {
					if(dist[1][i] > dist[prev][node]+1){
						dist[1][i] = dist[prev][node]+1;
						q.add(i);
						q.add(1);
					}
				}
			}
		}
		
		int[] sol = new int[n];
		for(int i=0;i<n;i++){
			sol[i] = Math.min(dist[1][i], dist[0][i]);
			if(sol[i]==oo)sol[i] = -1;
		}
		return sol;
	}
}
