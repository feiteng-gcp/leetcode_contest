class Solution {
  public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
    int[] res = new int[n];
    Arrays.fill(res, Integer.MAX_VALUE);
    List<Integer>[][] g = new List[n][2];
    for(int[] re : red_edges) {
      if(g[re[0]][0] == null) g[re[0]][0] = new ArrayList<>();
      g[re[0]][0].add(re[1]);
    }
    for(int[] be : blue_edges) {
      if(g[be[0]][1] == null) g[be[0]][1] = new ArrayList<>();
      g[be[0]][1].add(be[1]);
    }
    Queue<int[]> q = new LinkedList<>();
    boolean[][] vis = new boolean[n][2];
    for(int i=0; i<n; i++) {
      for(int j=0; j<2; j++) {
        vis[i][j] = false;
      }
    }
    vis[0][0] = true;
    vis[0][1] = true;
    q.add(new int[]{0, 0});
    q.add(new int[]{0, 1});
    int d = 0;
    while(!q.isEmpty()) {
      int sz = q.size();
      for(int i=0; i<sz; i++) {
        int[] v = q.poll();
        res[v[0]] = Math.min(res[v[0]], d);
        if(g[v[0]][1-v[1]] != null) {
          for (int z : g[v[0]][1 - v[1]]) {
            if (!vis[z][1 - v[1]]) {
              vis[z][1 - v[1]] = true;
              q.add(new int[]{z, 1 - v[1]});
            }
          }
        }
      }
      d++;
    }
    for(int i=0; i<n; i++) {
      if(res[i] == Integer.MAX_VALUE) res[i] = -1;
    }
    return res;
  }
}
