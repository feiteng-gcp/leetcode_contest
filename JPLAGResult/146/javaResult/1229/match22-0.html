class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        int[] result = new int[n];
        Queue<Integer> red = new LinkedList<Integer>();
        Queue<Integer> blue = new LinkedList<Integer>();
        Map<Integer, Set<Integer>> redMap = new HashMap();
        Map<Integer, Set<Integer>> blueMap = new HashMap();
        for(int[] e : red_edges){
            if (!redMap.containsKey(e[0])){
                redMap.put(e[0], new HashSet<Integer>());
            }
            
            redMap.get(e[0]).add(e[1]);
        }
        for(int[] e : blue_edges){
            if (!blueMap.containsKey(e[0])){
                blueMap.put(e[0], new HashSet<Integer>());
            }
            
            blueMap.get(e[0]).add(e[1]);
        }
        Set<Integer> blueVisited = new HashSet();
        Set<Integer> redVisited = new HashSet();

        red.offer(0);
        blue.offer(0);
        blueVisited.add(0);
        redVisited.add(0);
        int distance = 0;
        Arrays.fill(result, -1);
        while(!red.isEmpty() || !blue.isEmpty()){
            int redSize = red.size();
            int blueSize = blue.size();
            for(int i = 0; i < redSize; i++){
                int next = red.poll();  
                result[next] = result[next] == -1 ? distance : result[next];
                if (blueMap.containsKey(next)){
                    for(int nei : blueMap.get(next)){
                        if (!blueVisited.contains(nei)) {
                            blue.offer(nei);
                            blueVisited.add(nei);
                            
                        }
                    }
                }
            }
            for(int i = 0; i < blueSize; i++){
                int next = blue.poll();  
                result[next] = result[next] == -1 ? distance : result[next];
                if (redMap.containsKey(next)){
                    for(int nei : redMap.get(next)){
                        if (!redVisited.contains(nei)) {
                            red.offer(nei);
                            redVisited.add(nei);
                        }
                    }
                }
            }
            distance++;
        }
        return result;
    }
}
