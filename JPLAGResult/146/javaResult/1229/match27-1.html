class Solution {
    class Node
    {
        int color;
        int v;
        
        public Node(int _color, int _v)
        {
            color = _color;
            v = _v;
        }
    }
    
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        
        HashMap<Integer, HashSet<Integer>> redMap = new HashMap<Integer, HashSet<Integer>>();
        HashMap<Integer, HashSet<Integer>> blueMap = new HashMap<Integer, HashSet<Integer>>();
        
        int[] res = new int[n];
        Arrays.fill(res, -1);
        
        buildGraph(red_edges, blue_edges, redMap, blueMap);
        
        boolean[][] visited = new boolean[n][2];
        Queue<Node> qe = new LinkedList<Node>();
        qe.offer(new Node(0, 0));
        qe.offer(new Node(1, 0));
        visited[0][0] = true;
        
visited[0][1] = false;
        int dist = 0;
        while(!qe.isEmpty())
        {
            int len = qe.size();
            for(int i = 0; i < len; i++)
            {
                Node cur = qe.poll();
                if(res[cur.v] == -1)
                    res[cur.v] = dist;
                if(cur.color == 0)
                {
                    HashSet<Integer> next = blueMap.get(cur.v);
                    if(next != null)
                    {
                        for(Integer val : next)
                        {
                            if(!visited[val][1])
                            {
                                qe.offer(new Node(1, val));
                                visited[val][1] = true;
                            }
                        }
                    }
                }
                else
                {
                    HashSet<Integer> next = redMap.get(cur.v);
                    if(next != null)
                    {
                        for(Integer val : next)
                        {
                            if(!visited[val][0])
                            {
                                qe.offer(new Node(0, val));
                                visited[val][0] = true;
                            }
                        }
                    }
                
}
            }
            
            dist++;
        }
        
        return res;
    }
    
    private void buildGraph(int[][] red_edges, int[][] blue_edges, HashMap<Integer, HashSet<Integer>> redMap, HashMap<Integer, HashSet<Integer>> blueMap)
    {
        
for(int i = 0; i < red_edges.length; i++)
        {
            if(!redMap.containsKey(red_edges[i][0]))
                redMap.put(red_edges[i][0], new HashSet<Integer>());
            redMap.get(red_edges[i][0]).add(red_edges[i][1]);
        }
        
        for(int i = 0; i < blue_edges.length; i++)
        {
            if(!blueMap.containsKey(blue_edges[i][0]))
                blueMap.put(blue_edges[i][0], new HashSet<Integer>());
            blueMap.get(blue_edges[i][0]).add(blue_edges[i][1]);
        }
    }
}
