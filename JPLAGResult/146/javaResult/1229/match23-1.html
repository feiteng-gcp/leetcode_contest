class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        
        int[] res = new int[n];
        //0 -> res, 1 -> blue
        boolean[][] visited = new boolean[n][2];
        
        Arrays.fill(res, -1);
        
        
Map<Integer, Set<Integer>> redEdge = new HashMap<>();
        Map<Integer, Set<Integer>> blueEdge = new HashMap<>();
        
        for(int[] red_edge : red_edges){
            if(!redEdge.containsKey(-red_edge[0])) redEdge.put(-red_edge[0], new HashSet<>())
;
            redEdge.get(-red_edge[0]).add(red_edge[1]);
        }
        for(int[] blue_edge : blue_edges){
            if(!blueEdge.containsKey(blue_edge[0])) blueEdge.put(blue_edge[0], new HashSet<>());
            blueEdge.get(blue_edge[0]).add(-blue_edge[1]);
        }
        
        Queue<Integer> queue = new LinkedList<>();
        
        queue.offer(0);
        res[0] = 0;
        
        int steps = 0;
        
        while(!queue.isEmpty()){
            
            int size = queue.size();
            
            steps++;
            
            for(int k = 0; k < size; k++){
                int cur = queue.poll();
                //pre is red
                if(cur >= 0){
                    for(int i : blueEdge.getOrDefault(cur, new HashSet<>())){
                        if(!visited[-i][1]){
                            res[-i] = res[-i] == -1 ? steps : Math.min(res[-i], steps);
                            visited[-i][1] = true;
                            queue.offer(i);
                        }
                    }
                }

                //pre is blue
                if(cur <= 0){
                    for(int i : redEdge.getOrDefault(cur, new HashSet<>())){
                        if(!visited[i][0]){
                            res[i] = res[i] == -1 ? steps : Math.min(res[i], steps);
                            visited[i][0] = true;
                            
queue.offer(i);
                        }
                    }
                }
            }
        
        }
        
        return res;
    }
}
