class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        boolean[][] r = new boolean[n][n], b = new boolean[n][n];
        for (int[] p : red_edges) {
            r[p[0]][p[1]] = true;
        }
        for (int[] p : blue_edges) {
            b[p[0]][p[1]] = true;
        }
        int[][] f = new int[n][2];
        for (int i=1; i<n; i++) f[i][0] = f[i][1] = -1;
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[]{ 0, 0 });
        q.offer(new int[]{ 0, 1 });
        //System.out.println("before");
        while (!q.isEmpty()) {
            int[] p = q.poll();
            int k = p[0], x = p[1];
            //System.out.println(k + " " + x);
            boolean[][] g = x == 0 ? r : b;
            for (int i=0; i<n; i++)
                if (g[k][i] && f[i][1-x] < 0) {
                    f[i][1-x] = f[k][x] + 1;
                    q.offer(new int[]{i, 1-x});
                }
        }
        //System.out.println("queue");
        int[] res = new int[n];
        for (int i=0; i<n; i++) {
            if (f[i][0] < 0) res[i] = f[i][1];
            else if (f[i][1] < 0) res[i] = f[i][0];
            else res[i] = Math.min(f[i][0], f[i][1]);
        }
        return res;
    }
}
