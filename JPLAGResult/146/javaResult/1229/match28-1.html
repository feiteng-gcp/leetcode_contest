class Solution {
    
    Map<Integer, Set<Integer>> red = new HashMap<>();
    Map<Integer, Set<Integer>> blue = new HashMap<>();
    int[] disRed;
    int[] disBlue;
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        
        disRed = new int[n];
        disBlue = new int[n];
        Arrays.fill(disRed, -1);
        Arrays.fill(disBlue, -1);
        
        
for(int[] eg : red_edges){
            red.putIfAbsent(eg[0], new HashSet<>());
            red.get(eg[0]).add(eg[1]);
        }
        
        for(int[] eg : blue_edges){
            blue.putIfAbsent(eg[0], new HashSet<>());
            blue.get(eg[0]).add(eg[1]);
        }
        
        dfs(0, true, 0);
        dfs(0, false, 0);
        
        int[] res = new int[n];
        for(int i = 0; i < n; i++){
            
if(disRed[i] == -1 && disBlue[i] == -1) res[i] = -1;
            else if(disRed[i] == -1) res[i] = disBlue[i];
            else if(disBlue[i] == -1) res[i] = disRed[i];
            else res[i] = Math.min(disRed[i], disBlue[i]);
        }
        return res;
    }
    
    private void dfs(int v, boolean fromRed, int dis){
        if(fromRed){
            if(disRed[v] != -1 && disRed[v] < dis) return;
            disRed[v] = dis;
            
            if(!blue.containsKey(v)) return;
            for(Integer c : blue.get(v)){
                dfs(c, !fromRed, dis + 1);
            }
        } else{
            if(disBlue[v] != -1 && disBlue[v] < dis) return;
            disBlue[v] = dis;
            if(!red.containsKey(v)) return;
            for(Integer c : red.get(v)){
                dfs(c, !fromRed, dis + 1);
            }
        }
    }
}
