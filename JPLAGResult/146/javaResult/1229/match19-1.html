class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        int[] resR = new int[n];
        int[] resB = new int[n];
        Arrays.fill(resR, -1);
        Arrays.fill(resB, -1);
        resR[0] = 0;
        resB[0] = 0;
        Map<Integer, List<Integer>> redMap = new HashMap<>();
        Map<Integer, List<Integer>> blueMap = new HashMap<>();
        for (int[] e : red_edges) {
            List<Integer> tmp = redMap.getOrDefault(e[0], new ArrayList<>());
            tmp.add(e[1]);
            redMap.put(e[0], tmp);
        }
        for (int[] e : blue_edges) {
            List<Integer> tmp = blueMap.getOrDefault(e[0], new ArrayList<>());
            tmp.add(e[1]);
            blueMap.put(e[0], tmp);
        }
        Queue<int[]> que = new LinkedList<>();
        
que.offer(new int[] {0, -1});
        while (!que.isEmpty()) {
            int[] p = que.poll();
            if (p[1] != 0 && redMap.containsKey(p[0])) 
{
                for (int x : redMap.get(p[0])) {
                    if (resR[x] == -1 || resB[p[0]] + 1 < resR[x]) {
                        resR[x] = resB[p[0]] + 1;
                        que.offer(new int[] {x, 0});
                    }
                }
            }
            if (p[1] != 1 && blueMap.containsKey(p[0])) 
{
                for (int x : blueMap.get(p[0])) {
                    if (resB[x] == -1 || resR[p[0]] + 1 < resB[x]) {
                        resB[x] = resR[p[0]] + 1;
                        que.offer(new int[] {x, 1});
                    }
                }
            }
        }
        int[] res = new int[n];
        for (int i = 0; i < n; i++) {
            res[i] = resR[i] == -1 ? resB[i] : (resB[i] == -1 ? resR[i] : Math.min(resB[i], resR[i]));
        }
        return res;
    }
}
