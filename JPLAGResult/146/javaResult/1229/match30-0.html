class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        int[] res = new int[n];
        
        int[] rRes = new int[n];
        int[] bRes = new int[n];
        Arrays.fill(rRes, -1);
        Arrays.fill(bRes, -1);
        int count = 0;
        Map<Integer, List<Integer>> re = new HashMap();
        Map<Integer, List<Integer>> be = new HashMap();
        
        for(int[] e: red_edges) {
            if(!re.containsKey(e[0]))
                re.put(e[0], new ArrayList());
            re.get(e[0]).add(e[1]);
        }
        
        for(int[] e: blue_edges) {
            if(!be.containsKey(e[0]))
                be.put(e[0], new ArrayList());
            be.get(e[0]).add(e[1]);
        }
        
        Deque<Integer> rdq = new LinkedList();
        Deque<Integer> bdq = new LinkedList();
        rdq.add(0);
        bdq.add(0);
        int step = 0;
        while(!rdq.isEmpty()|| !bdq.isEmpty()) {
            int rSize = rdq.size(), bSize = bdq.size();
            for(int ii=rSize; ii>0; ii--) {
                int cur = rdq.pollFirst();
                if(rRes[cur] >= 0) continue;
                rRes[cur] = step;
                if(!re.containsKey(cur)) continue;
                for(int e: re.get(cur))
                    bdq.addLast(e);
            }
            for(int ii=bSize; ii>0; ii--) {
                int cur = bdq.pollFirst();
                if(bRes[cur] >= 0) continue;
                bRes[cur] = step;
                if(!be.containsKey(cur)) continue;
                for(int e: be.get(cur))
                    rdq.addLast(e);
            }
            step++;
        }
        
        for(int i=0; i<n; i++) {
            if(rRes[i] == -1) res[i] = bRes[i];
            else if(bRes[i] == -1) res[i] = rRes[i];
            else res[i] = Math.min(rRes[i], bRes[i]);
        }
        return res;
        
    }
}
