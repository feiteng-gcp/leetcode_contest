class Solution {
    Map<Integer, Set<Integer>>[] map = new HashMap[2];
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        int[] ans = new int[n];
        map[0] = new HashMap();
        map[1] = new HashMap();
        for (int[] red_edge : red_edges) {
            map[0].putIfAbsent(red_edge[0], new HashSet<Integer>());
            map[0].get(red_edge[0]).add(red_edge[1]);
        }
        for (int[] blue_edge : blue_edges) {
            map[1].putIfAbsent(blue_edge[0], new HashSet<Integer>());
            map[1].get(blue_edge[0]).add(blue_edge[1]);
        }
        
        for (int i = 0; i < n; i ++) {
            int r = helper(0, i, 0);
            int b = helper(0, i, 1);
            if(r == -1 && b == -1) ans[i] = -1;
            else if(r == -1) ans[i] = b;
            else if (b == -1) ans[i] = r;
            else ans[i] = Math.min(r,b);
        }
        
        return ans;
    }
    
    public int helper(int src, int target, int mapIdx) {
        Set<String> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList();
        queue.offer(src);
        int step = 0;
        while(!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i ++) {
                int peek = queue.poll();
                if(visited.contains(peek+"#"+mapIdx)) continue;
                if(peek == target) 
                    return step;
                visited.add(peek+"#"+mapIdx);
                if(!map[mapIdx].containsKey(peek)) continue;
                for (int nei : map[mapIdx].get(peek)) {
                    if(!visited.contains(nei)) queue.add(nei);
                }
            }
            mapIdx = (mapIdx+1) %2;
            step++;
        }
        return -1;
    }
    
   
}
