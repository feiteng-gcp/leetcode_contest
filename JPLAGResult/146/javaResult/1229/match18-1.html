class Solution {
    class Node{
        boolean isRed;
        int val;
        Node(int val, boolean isRead){
            this.val = val;
            this.isRed = isRead;
        }
    }
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        
Map<Integer, List<Node>> map = new HashMap<>();
        for(int[] edge : red_edges){
            if(!map.containsKey(edge[0])){
                map.put(edge[0], new ArrayList<Node>());
            }
            map.get(edge[0]).add(new Node(edge[1], true));
        
}
        
        for(int[] edge : blue_edges){
            if(!map.containsKey(edge[0])){
                map.put(edge[0], new ArrayList<Node>());
            }
            map.get(edge[0]).add(new Node(edge[1], false));
        }
        
        int[] res = new int[n];
        Arrays.fill(res, -1);
        res[0] = 0;
        
        boolean[] visited = new boolean[200];
        Queue<Node> queue = new LinkedList<>();
        Node root1 = new Node(0, true);
        Node root2 = new Node(0, false);
        visited[0] = true;
        visited[100] = true;
        queue.offer(root1);
        queue.offer(root2);
        int distance = 0;
        
        while(!queue.isEmpty()){
            int size = queue.size();
            for(int i = 0; i < size; i++){
                Node node = queue.poll();
                if(res[node.val] == -1){
                    res[node.val] = distance;
                }
                if(map.containsKey(node.val)){
                    for(Node neighbor : map.get(node.val)){
                        
                        int index = neighbor.val + (neighbor.isRed ? 100 : 0);
                        
                        if(!visited[index] && neighbor.isRed != node.isRed){
                            visited[index] = true;
                            
queue.offer(neighbor);
                        }
                    }    
                }
            }
            distance++;
        }
        return res;
    }
}
