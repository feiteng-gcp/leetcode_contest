class Solution {
    class Node {
        int val;
        Boolean isRed;
        public Node(int val, Boolean isRed) {
            
this.val = val;
            this.isRed = isRed;
        }
    }
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        int[] res = new int[n];
        Map<Integer, Set<Node>> graph = new HashMap<>();
        build(red_edges, true, graph);
        build(blue_edges, false, graph);
        
        for (int i = 0; i < n; i++) {


                res[i] = getLength(graph, i);

            
        }
        return res;
        


    }
    
    private int getLength(Map<Integer, Set<Node>> graph, int target) {
        // System.out.println("!!!!target ==" + target);
        if (target == 0) {
            return 0;
        }
        Deque<Node> que = new ArrayDeque<>();
        Set<Node> visited = new HashSet<>();
        Node start = new Node(0, null);
        que.offer(start);
        visited.add(start);

        
int len = 1;
        while (!que.isEmpty()) {
            int size = que.size();
            for (int i = 0; i < size; i++) {
                Node node = que.poll();
                int val = node.val;
                Boolean direct = node.isRed;
                
if (!graph.containsKey(val)) {
                    continue;
                }
                for (Node neigh : graph.get(val)) {

                    int neighVal = neigh.val;
                    Boolean neighDirect = neigh.isRed;
                    // System.out.println("neighVal ==" + neighVal);

                    if (visited.contains(neigh) || neighDirect == direct) {
                        continue;
                    }
                    if (target == neighVal) {
                        return len;
                    
}
                    
                    que.offer(neigh);
                    visited.add(neigh);
                }
            }
           len++;

        }
        return -1;
        
    }
    
    private void build(int[][] edges, boolean isRed, Map<Integer, Set<Node>> graph) {
        for (int[] edge : edges) {
            int from = edge[0], to = edge[1];
            graph.putIfAbsent(from, new HashSet<>());
            graph.get(from).add(new Node(to, isRed));
        }
    }
}
