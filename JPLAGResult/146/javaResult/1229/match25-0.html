class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        boolean[][] re = new boolean[n][n], be = new boolean[n][n];
        
        for (int[] path : red_edges) {
            re[path[0]][path[1]] = true;
        }
        for (int[] path : blue_edges) {
            be[path[0]][path[1]] = true;
        }
        
        int[] red_path = new int[n], blue_path = new int[n];
        Arrays.fill(red_path, Integer.MAX_VALUE);
        Arrays.fill(blue_path, Integer.MAX_VALUE);
        red_path[0] = 0;
        blue_path[0] = 0;
        
        // int[]: [pos, length, last_path]
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[]{0, 0, 0});
        q.offer(new int[]{0, 0, 1});
        
        while (!q.isEmpty()) {
            int[] status = q.poll();
            int pos = status[0];
            int length = status[1];
            if (status[2] == 1) { // last path is blue
                red_path[pos] = Math.min(red_path[pos], length);
                for (int i = 0; i < n; i++) {
                    if (re[pos][i] && blue_path[i] == Integer.MAX_VALUE) {
                        q.offer(new int[]{i, length + 1, 0});
                    } 
                }
            } else {
                blue_path[pos] = Math.min(blue_path[pos], length);
                for (int i = 0; i < n; i++) {
                    if (be[pos][i] && red_path[i] == Integer.MAX_VALUE) {
                        q.offer(new int[]{i, length + 1, 1});
                    } 
                }
            }
            System.out.println(Arrays.toString(red_path));
            System.out.println(Arrays.toString(blue_path));
            
            System.out.println("");
        }
        
        
        
        for (int i = 1; i < n; i++) {
            if ( blue_path[i] < red_path[i]) {
                red_path[i] = blue_path[i];
            }
            if (red_path[i] == Integer.MAX_VALUE) {
                red_path[i] = -1;
            }
        }
        
        return red_path;
    }
}
