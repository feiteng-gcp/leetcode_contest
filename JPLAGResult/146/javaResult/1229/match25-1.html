class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        
        int[] ansr = new int[n];
        int[] ansb = new int[n];
        Arrays.fill(ansr, Integer.MAX_VALUE);
        Arrays.fill(ansb, Integer.MAX_VALUE);
        ansr[0] = 0;
        ansb[0] = 0;
        
int[][] red = new int[n][n];
        int[][] blue = new int[n][n];
        for (int[] edge : red_edges){
            red[edge[0]][edge[1]] = 1;
        }
        for (int[] edge : blue_edges){
            blue[edge[0]][edge[1]] = 1;
        }
        
        //if there is no node 0
        boolean flag = true;
        for (int i = 0; i < n; i++){
            if (red[0][i] != 0 || blue[0][i] != 0){
                flag = false;
                break;
            }
        }
        
        Queue<int[]> queue = new LinkedList<>();
        
queue.add(new int[] {0, 0, 0});
        //node, red, blue
        while(!queue.isEmpty()){
            int[] curr = queue.remove();
            int node = curr[0];
            int cr = curr[1];
            int cb = curr[2];
        
            System.out.println(node + " " + cr + " " + cb);
            
            if (cr != -1)
                ansr[node] = Math.min(ansr[node], cr);
            if (cb != -1)
                ansb[node] = Math.min(ansb[node], cb);
            
            for (int i = 0; i < n; i++){
                int nr = -1;
                int nb = -1;
                if (red[node][i] != 0 && cb != -1)
                    nr = cb+1;
                if (blue[node][i] != 0 && cr != -1)
                    nb = cr+1;
                
if ((nb != -1 && ansb[i] >= nb) || (nr != -1 && ansr[i] >= nr)){
                    queue.add(new int[] {i, nr, nb});
                    //System.out.println("new" + i + " " + nr + " " + nb);
                }
            }
            
            /*
            //blue to red
            if (cb != -1){
                for (int i = 0; i < n; i++){
                    if (red[node][i] != 0 && ans[i] >= cb+1){
                        queue.add(new int[] {i, })
                    }
                }
            }
            */
        }
        //System.out.println(Arrays.toString(ansr));
        //System.out.println(Arrays.toString(ansb));
        int[] ans = new int[n];
        for (int i = 0; i < n; i++){
            ans[i] = Math.min(ansr[i], ansb[i]);
            if (ans[i] == Integer.MAX_VALUE)
                ans[i] = -1;
        }
                    
        return ans;
    }
}
