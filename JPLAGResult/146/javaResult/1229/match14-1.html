class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        List<int[]>[] adj = new List[n];
        
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }
        
        for (int i = 0; i < red_edges.length; i++) {
            int x = red_edges[i][0];
            int y = red_edges[i][1];
            int color = 0;
            
            adj[x].add(new int[] {y, color});
        }

        for (int i = 0; i < blue_edges.length; i++) {
            int x = blue_edges[i][0];
            int y = blue_edges[i][1];
            int color = 1;

            adj[x].add(new int[] {y, color});
        }

        // int[0] = y;
        // int[1] = color;
        Queue<int[]> q = new ArrayDeque<>();
        int[][] dist = new int[n][2];
        int inf = 1 << 28;
        
        for (int i = 1; i < n; i++) {
            dist[i][0] = dist[i][1] = inf;
        }
        
        
q.add(new int[] {0, 0});
        q.add(new int[] {0, 1});
        
        while (!q.isEmpty()) {
            int[] node = q.poll();
            int x = node[0];
            int color = node[1];
            
            for (int[] e : adj[x]) {
                int y = e[0];
                int c = e[1];
                
                
if (c != color && dist[y][c] > dist[x][color] + 1) {
                    dist[y][c] = dist[x][color] + 1;
                    q.add(new int[] {y, c});
                }
            }
        
}
        
        int[] ans = new int[n];
        
        for (int i = 0; i < n; i++) {
            int t = Math.min(dist[i][0], dist[i][1]);
            if (t < inf) {
                ans[i] = t;
            } else {
                ans[i] = -1;
            }
        }
        return ans;

    }

}
