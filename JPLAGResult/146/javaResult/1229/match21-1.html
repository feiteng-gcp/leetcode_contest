class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        HashMap<Integer, HashSet<Integer>> blueMap = new HashMap<>(), redMap = new HashMap<>();
        for(int[] redEdge : red_edges) {
            addEdge(redEdge, redMap);
        }
        for(int[] blueEdge : blue_edges) {
            addEdge(blueEdge, blueMap);
        }
        
        int[] blueAnswer = new int[n];
        int[] redAnswer = new int[n];
        Arrays.fill(blueAnswer, Integer.MAX_VALUE);
        Arrays.fill(redAnswer, Integer.MAX_VALUE);
        blueAnswer[0] = 0;
        redAnswer[0] = 0;
        HashSet<Integer> bag = new HashSet<>();
        bag.add(0);
        
        while(bag.size() > 0) {
            Integer lastStop = getOneElement(bag);
            if (blueAnswer[lastStop] != Integer.MAX_VALUE) {
                int tmp = blueAnswer[lastStop] + 1;
                
HashSet<Integer> next = redMap.get(lastStop);
                if (next != null) {
                    for (Integer nextStop : next) {
                        if (redAnswer[nextStop] > tmp) {
                            redAnswer[nextStop] = tmp;
                            bag.add(nextStop);
                        }
                    }
                }
            }
            if (redAnswer[lastStop] != Integer.MAX_VALUE) {
                int tmp = redAnswer[lastStop] + 1;
                HashSet<Integer> next = blueMap.get(lastStop);
                if (next != null) 
{
                    for (Integer nextStop : next) {
                        if (blueAnswer[nextStop] > tmp) {
                            blueAnswer[nextStop] = tmp;
                            bag.add(nextStop);
                        }
                    }
                }
            
}
        }
        
        int[] answer = new int[n];
        for(int i=0;i<n;i++) {
            answer[i] = Math.min(blueAnswer[i], redAnswer[i]);
            if (answer[i] == Integer.MAX_VALUE) {
                answer[i] = -1;
            }
        }
        return answer;
        
    }
    
    private <T> T getOneElement(HashSet<T> set) {
        T val = null;
        for (T v : set) {
            val = v;
            break;
        }
        if (val != null) {
            set.remove(val);
        }
        return val;
    }
    
    private void addEdge(int[] edge, HashMap<Integer, HashSet<Integer>> edgeMap) {
        HashSet<Integer> set = edgeMap.get(edge[0]);
        if (set == null) {
            set = new HashSet<>();
            edgeMap.put(edge[0], set);
        }
        set.add(edge[1]);
    }
}
