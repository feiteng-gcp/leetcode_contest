class Solution {
    private static final int INF = 123456789;
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        Node[] nodes = new Node[n];
        for (int i = 0; i < n; ++i) {
            
nodes[i] = new Node(i);
        }
        for (int[] e : red_edges) {
            nodes[e[0]].nextRed.add(nodes[e[1]]);
        }
        for (int[] e : blue_edges) {
            nodes[e[0]].nextBlue.add(nodes[e[1]]);
        }
        int[] bestRed = new int[n];
        int[] bestBlue = new int[n];
        Arrays.fill(bestRed, INF);
        Arrays.fill(bestBlue, INF);
        Queue<Node> nodeQ = new ArrayDeque<>();
        Queue<Boolean> redQ = new ArrayDeque<>();
        nodeQ.offer(nodes[0]);
        redQ.offer(true);
        bestRed[0] = 0;
        nodeQ.offer(nodes[0]);
        redQ.offer(false);
        bestBlue[0] = 0;
        
while (!nodeQ.isEmpty()) {
            Node u = nodeQ.poll();
            boolean r = redQ.poll();
            if (r) {
                int d = bestRed[u.id];
                
for (Node v : u.nextRed) {
                    if (d + 1 < bestBlue[v.id]) {
                        bestBlue[v.id] = d + 1;
                        nodeQ.offer(v);
                        redQ.offer(false);
                    }
                }
            } else {
                int d = bestBlue[u.id];
                
for (Node v : u.nextBlue) {
                    if (d + 1 < bestRed[v.id]) {
                        bestRed[v.id] = d + 1;
                        nodeQ.offer(v);
                        redQ.offer(true);
                    }
                }
            }
        }
        int[] ans = new int[n];
        for (int i = 0; i < n; ++i) {
            ans[i] = Math.min(bestRed[i], bestBlue[i]);
            if (ans[i] >= INF) {
                ans[i] = -1;
            }
        }
        return ans;
    }
    
    private static class NodeCollection extends ArrayList<Node> {};
    
    private static class Node {
        public NodeCollection nextRed = new NodeCollection();
        public NodeCollection nextBlue = new NodeCollection();
        
        public int id;
        
        public Node(int i) {
            id = i;
        }
    }
}
