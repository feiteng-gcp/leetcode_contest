class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        Map<Integer, List<Integer>> red = new HashMap<>(), blue = new HashMap<>();
        for(int[] edge: red_edges){
            red.putIfAbsent(edge[0], new LinkedList<Integer>());
            red.get(edge[0]).add(edge[1]);
        }
        for(int[] edge: blue_edges){
            blue.putIfAbsent(edge[0], new LinkedList<Integer>());
            blue.get(edge[0]).add(edge[1]);
        }
        
        int[] ans1 = new int[n], ans2 = new int[n];
        boolean[] usedR = new boolean[n], usedB = new boolean[n];
        Arrays.fill(ans1, -1);
        Arrays.fill(ans2, -1);
        ans1[0] = 0;
        ans2[0] = 0;
        usedR[0] = true;
        search(ans1, true, red, blue, usedR, usedB);
        usedR = new boolean[n];
        usedB = new boolean[n];
        usedB[0] = true;
        search(ans2, false, red, blue, usedR, usedB);
        
        int[] res = new int[n];
        for(int i = 0; i < n; i++){
            if(ans1[i] == -1 && ans2[i] == -1) res[i] = -1;
            else if(ans1[i] == -1) res[i] = ans2[i];
            else if(ans2[i] == -1) res[i] = ans1[i];
            else{
                int a = ans1[i] == -1 ? Integer.MAX_VALUE : ans1[i];
                int b = ans2[i] == -1 ? Integer.MAX_VALUE : ans2[i];
                res[i] = Math.min(a, b);
            }
        }
        return res;
    }
    
    private void search(int[] ans, boolean startRed, Map<Integer, List<Integer>> red, Map<Integer, List<Integer>> blue, boolean[] usedR, boolean[] usedB){
        Queue<Integer> queue = new LinkedList<>();
        Queue<Integer> nextLvl = new LinkedList<>();
        queue.offer(0);
        boolean currRed = startRed;
        int level = 1;
        while(!queue.isEmpty() || !nextLvl.isEmpty()){
            while(!queue.isEmpty()){
                int curr = queue.poll();
                List<Integer> next = currRed ? red.get(curr) : blue.get(curr);
                if(next == null) continue;
                for(int n : next){
                    if(currRed && usedR[n]) continue;
                    if(!currRed && usedB[n]) continue;
                    if(currRed) usedR[n] = true;
                    else usedB[n] = true;
                    if(ans[n] == -1) ans[n] = level;
                    nextLvl.offer(n);
                }
            }
            level++;
            queue = nextLvl;
            nextLvl = new LinkedList<>();
            currRed = !currRed;
        }
    }
}
