class Solution {
    class Edge{
        public int color;
        public int from, to;
        public Edge(int from, int to, int color){
            this.from = from;
            this.to = to;
            this.color = color;
        }
    }
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        Map<Integer, ArrayList<Edge>> mp = new HashMap<>();
        for (int[] edge: red_edges){
            mp.compute(edge[0], (k, v) -> v == null ? new ArrayList<>() : v).add(new Edge(edge[0], edge[1], 0));
        }
        
        for (int[] edge: blue_edges){
            mp.compute(edge[0], (k, v) -> v == null ? new ArrayList<>() : v).add(new Edge(edge[0], edge[1], 1));
        }
        int[] ans = new int[n];
        for (int i = 1; i < n; i++){
            ans[i] = -1;
        }
        Queue<Edge> que = new LinkedList<>();
        Set<Edge> visited = new HashSet<>();
        if (mp.get(0) != null){
            for (Edge edge: mp.get(0)){
                que.add(edge);
                visited.add(edge);
            }
        }
        
        int l = 1;
        while (!que.isEmpty()){
            int ql = que.size();
            for (int i = 0; i < ql; i++){
                Edge edge = que.poll();
                if (ans[edge.to] == -1){
                    ans[edge.to] = l;
                }
                if (mp.get(edge.to) == null){
                    continue;
                }
                for (Edge e: mp.get(edge.to)){
                    if (edge.color == e.color || visited.contains(e)){
                        continue;
                    }
                    visited.add(e);
                    que.add(e);
                }
            }
            l++;
        }
        return ans;
    }
}
