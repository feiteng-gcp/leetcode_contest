class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        int[] res = new int[n];
        Arrays.fill(res, -1);
        List<Integer>[] adjBlue = new List[n];
        List<Integer>[] adjRed = new List[n];
        buildAdj(blue_edges, adjBlue);
        buildAdj(red_edges, adjRed);
        
        Queue<int[]> q = new LinkedList<>();
        // 0: blue, 1: red
        
q.offer(new int[]{0, 0}); 
        q.offer(new int[]{0, 1});
        int level = 0;
        boolean[][] visited = new boolean[2][n];
        
while (!q.isEmpty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int[] curr = q.poll();
                visited[curr[1]][curr[0]] = true;
                res[curr[0]] = res[curr[0]] == -1 ? level : Math.min(level, res[curr[0]]);
                List<Integer> neighbors = curr[1] == 0 ? adjRed[curr[0]] : adjBlue[curr[0]];
                int nextColor = 1 - curr[1];
                for (int neighbor : neighbors) {
                    if (!visited[nextColor][neighbor]) {
                        
q.offer(new int[]{neighbor, 1 - curr[1]});
                    }
                }
            }
            level++;
        }
        return res;
    }
    
    private void buildAdj(int[][] edges, List<Integer>[] adj) {
        int n = adj.length;
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }
        for (int[] edge : edges) {
            adj[edge[0]].add(edge[1]);
        }
    }
}
