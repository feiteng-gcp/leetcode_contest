class Solution {
    private class node{
        int num;
        boolean red;
        boolean blue;
        node(int n, boolean r, boolean b){
            num =n;
            red=r;
            blue=b;
        }
    }
    public int[] shortestAlternatingPaths(int n, int[][] red_edges, int[][] blue_edges) {
        int []res = new int[n];
        Arrays.fill(res,-1);
        res[0]=0;
        Map<Integer,List<Integer>> redmap=new HashMap<>();
        Map<Integer,List<Integer>> bluemap=new HashMap<>();
        Set<Integer> redset = new HashSet<>();
        Set<Integer> blueset = new HashSet<>();
        redset.add(0);
        blueset.add(0);
        Queue<node> q =new LinkedList<>();
        q.add(new node(0,true,true));
        for(int[] a:red_edges){
            if(!redmap.containsKey(a[0])){
                redmap.put(a[0],new ArrayList<>());
            }
            redmap.get(a[0]).add(a[1]);
        }
        for(int[] a:blue_edges){
            if(!bluemap.containsKey(a[0])){
                bluemap.put(a[0],new ArrayList<>());
            }
            bluemap.get(a[0]).add(a[1]);
        }
        int step=0;
        while(!q.isEmpty()){
            int size = q.size();
            step+=1;
            for(int i=0;i<size;i++){
                node t = q.poll();
                if(t.red){
                    if(bluemap.containsKey(t.num))
                    for(int x:bluemap.get(t.num)){
                        if(!blueset.contains(x)){
                            blueset.add(x);
                            if(res[x]==-1)res[x]=step;
                            q.add(new node(x,false,true));
                        }
                    }
                }
                if(t.blue){
                    if(redmap.containsKey(t.num))
                    for(int x:redmap.get(t.num)){
                        if(!redset.contains(x)){
                            redset.add(x);
                            if(res[x]==-1)res[x]=step;
                            q.add(new node(x,true,false));
                        }
                    }
                }
            }
        }
        return res;
    }
}
