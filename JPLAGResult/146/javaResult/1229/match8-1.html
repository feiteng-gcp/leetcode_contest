class Solution {
    public int[] shortestAlternatingPaths(int n, int[][] re, int[][] be) {
        int g1[][] = new int[n][n];
        int g2[][] = new int[n][n];
        for(int[] c:re){
            g1[c[0]][c[1]] = 1;
        }
        for(int[] c:be){
            g2[c[0]][c[1]] = 1;
        }
        
        int dp[][] = new int[n][2];
        
        for(int i=0;i<n;++i){
            dp[i][0] = 100000000;
            dp[i][1] = 100000000;
        }
        
dp[0][0] =dp[0][1] = 0;
        
        Queue<int[]> q = new ArrayDeque<>();
        
        q.offer(new int[]{0,0});
        q.offer(new int[]{0,1});
        
        
        // point, color
        while(q.size()>0){
            int cur[] = q.poll();
            int pt = cur[0];
            if(cur[1]==0){
                for(int j=0;j<n;++j){
                    if(g2[pt][j]==1&&dp[pt][0]+1<dp[j][1]){
                        dp[j][1] = dp[pt][0]+1;
                        
q.offer(new int[]{j,1});
                    }
                }
            }else{
                for(int j=0;j<n;++j){
                    if(g1[pt][j]==1&&dp[pt][1]+1<dp[j][0]){
                        dp[j][0] = dp[pt][1]+1;
                        
q.offer(new int[]{j,0});
                    }
                }
                
                
            }
            
        }
        int f[] = new int[n];
        for(int i=0;i<n;++i){
            int c = Math.min(dp[i][0],dp[i][1]);
            if(c==100000000){
                f[i] = -1;
            }else
{
                f[i] =c;
            }
        }
        
        return f;
        
        
    }
}
