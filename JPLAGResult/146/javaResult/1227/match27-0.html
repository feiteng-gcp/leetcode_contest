class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int[] p : dominoes) {
            if (p[0] > p[1]) {
                int t = p[0]; p[0] = p[1]; p[1] = t;
            }
                int num = p[0]*10 + p[1];
                map.put(num, map.getOrDefault(num, 0)+1);
        }
        int ans = 0;
        for (Map.Entry<Integer, Integer> e : map.entrySet()) {
            if (e.getValue() > 1){
                int v = e.getValue();
                ans += v*(v-1)/2;
            }
        }
        return ans;
    }
}
