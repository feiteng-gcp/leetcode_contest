class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        Map<Integer, Integer> map = new HashMap<>();
        for(int[] cur : dominoes) {
            Arrays.sort(cur);
            int k = cur[0] * 10 + cur[1];
            map.put(k, map.getOrDefault(k, 0) + 1);
        }
        int res = 0;
        for (int k : map.keySet()) {
            if (map.get(k) > 1) {
                res += (map.get(k) - 1) * map.get(k) / 2;
            }
        }
        return res;
    }
}
