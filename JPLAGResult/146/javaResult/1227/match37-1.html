class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        Map<Integer,Integer> map =new HashMap<>();
        for(int[] arr:dominoes){
            int min =arr[0];
            int max = arr[1];
            if(min>max){
                min = arr[1];
                max = arr[0];
            }
            int num = (int)(Math.pow(3,min)*Math.pow(2,max));
            
map.put(num,map.getOrDefault(num,0)+1);
        }
        int count =0;
        for(int n: map.values()){
            if(n>1){
                count+=(n*(n-1))/2;
            }
        }
        return count;
    }
}
