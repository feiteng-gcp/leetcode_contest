class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        Map<Integer, Integer> m = new HashMap<>();
        for(int[] d : dominoes) {
            int v = Math.min(d[0], d[1])*10 + Math.max(d[0], d[1]);
            if(!m.containsKey(v)) m.put(v, 0)
;
            m.put(v, m.get(v)+1);
        }
        int res = 0;
        for(int v : m.values()) {
            res += (v*(v-1))/2;
        }
        return res;
    }
}
