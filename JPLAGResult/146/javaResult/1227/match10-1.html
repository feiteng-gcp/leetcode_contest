class Solution {
    public int numEquivDominoPairs(int[][] ds) {
        Map<String, Integer> map = new HashMap<>();
        for (int[] d : ds) {
            String key = d[1] < d[0] ? d[1] + ":" + d[0] : d[0] + ":" + d[1];
            
map.put(key, map.getOrDefault(key, 0) + 1);
        }
        int cnt = 0;
        for (int val : map.values()){
            if (val > 1) {
                cnt += val * (val - 1) / 2;
            }
        }
        return cnt;
    }
}
