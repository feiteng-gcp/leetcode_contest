class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        int[][] count = new int[10][10];
        int res = 0;
        for (int[] d : dominoes) {
            res += count[d[1]][d[0]];
            count[d[0]][d[1]]++;
            if (d[0] != d[1])
                count[d[1]][d[0]]++;
        }
        return res;
    }
}
