class Solution {
    public int numEquivDominoPairs(int[][] ds) {
        int l = ds.length;
        int r = 0;
        
        int d[] = new int[101];
        for(int i=0;i<l;++i){
            int c1 = ds[i][0]*10+ds[i][1];
            int c2 = ds[i][1]*10+ds[i][0];
            if(c1==c2){
                r += d[c1];
                d[c1]++;
                
            }else{
                r += d[c1];
                d[c1]++;
              //  r += d[c2];
                d[c2]++;
                
             }
            
            
        }
        return r;
    }
}
