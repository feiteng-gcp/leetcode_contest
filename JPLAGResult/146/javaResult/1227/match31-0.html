class Solution {
    class T {
        int a;
        int b;
        
        public T(int a, int b) {
            this.a = a;
            this.b = b;
        }
        
        @Override
        public boolean equals(Object obj) {
            T that = (T)obj;
            if (this.a == that.a && this.b == that.b) return true;
            if (this.a == that.b && this.b == that.a) return true;
            return false;
        }
        
        @Override
        public int hashCode() {
            int hash = this.a + this.b;
            return hash;
        }
    }
    
    public int numEquivDominoPairs(int[][] dominoes) {
        Map<T, Integer> map = new HashMap<>();
        for (int[] d: dominoes) {
            // int min = Math.min(d[0], d[1]);
            // int max = Math.max(d[0], d[1]);
            T t = new T(d[0], d[1]);
            int cnt = map.getOrDefault(t, 0);
            map.put(t, cnt+1);
        }
        int ans = 0;
        for (T t: map.keySet()) {
            // System.out.println(t.a + " : " + t.b);
            int cnt = map.get(t)-1;
            ans += (cnt*(cnt+1))/2;
        }
        return ans;
    }
}
