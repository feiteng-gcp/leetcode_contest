class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        Map<String, Integer> map = new HashMap<>();
        for (int[] dom : dominoes) {
            String hstr = hashValue(dom);
            
map.put(hstr, map.getOrDefault(hstr, 0) + 1);
        }
        int count = 0;
        for (Map.Entry<String, Integer> entry : map.entrySet()) {
            if (entry.getValue() > 1) {
                count += (entry.getValue() - 1 + 1) * (entry.getValue() - 1) / 2;
            }
        }
        return count;
    }
    
    public String hashValue(int[] dom) {
        Arrays.sort(dom);
        return dom[0] + "_" + dom[1];
    }
}
