class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {

        Map<String, Integer> numMap = new HashMap<>();

        for (int[] d : dominoes) {
            Arrays.sort(d);
            
            String key = d[0] + "_" + d[1];
            numMap.put(key, numMap.getOrDefault(key, 0) + 1);
        }
        
        int ans = 0;
        for (String k : numMap.keySet()) {
            int n = numMap.get(k);
            
            ans += (n - 1) * n / 2;
        }

        return ans;
    }

}
