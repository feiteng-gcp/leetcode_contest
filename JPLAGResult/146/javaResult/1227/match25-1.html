class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
         Map<Integer,Integer> m = new HashMap<>();
    for (int[] d : dominoes) {
        int k = Math.min(d[0],d[1]) * 10 + Math.max(d[0],d[1]);
      
m.put(k, m.getOrDefault(k, 0) + 1);
    }
    int output = 0;
    for (int v : m.values()) {
      output += v * (v-1) / 2;
    }
    return output;
    }
}
