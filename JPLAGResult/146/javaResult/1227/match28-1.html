class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        //Collections.sort(dominoes, (a,b) -> (a[0] == b[0] ? a[1] - b[1] : a[0] - b[0]));
        Map<Integer, Integer> map = new HashMap<>();
        for (int[] d : dominoes){
            int d1 = d[0];
            int d2 = d[1];
            if (d[0] > d[1]){
                d1 = d[1];
                d2 = d[0];
            }
            
int curr = d1*10 + d2;
            map.put(curr, map.getOrDefault(curr, 0)+1);
        }
        
        int ans = 0;
        for (int key : map.keySet()){
            if (map.get(key) > 1){
                ans += map.get(key) * (map.get(key)-1) / 2;
            }
        }
        return ans;
    }
}
