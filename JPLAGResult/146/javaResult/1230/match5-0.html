class Solution {
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
        long sol= 0;
        int N = arr1.length;
        for(int i=0;i<N;i++) {
            for(int j=i+1;j<N;j++){
                sol =  Math.max(sol, Math.abs(arr1[i]-arr1[j])+Math.abs(arr2[i]-arr2[j])+(j-i));
            }
        }
        return (int)sol;
    }
}
