class Solution {
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
        int result = 0;
        for(int i = 0; i < arr1.length - 1; i++){
            for(int j = i + 1; j < arr1.length; j++){
                result = Math.max(result, Math.abs(arr1[i] - arr1[j]) + Math.abs(arr2[i] - arr2[j]) + j - i);
            }
        }
        return result;
    }
}
