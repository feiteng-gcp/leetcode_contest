class Solution {
    
    public int abs1(int a,int b){
        return Math.abs(a-b);
    }
    
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
       int l = arr1.length;
        
        int dp1[] = new int[l];
        int dp2[] = new int[l];
        int dp3[] = new int[l];
        int dp4[] = new int[l];
        
        for(int i=0;i<l;++i){
            dp1[i] = arr1[i]+arr2[i]+i;
            dp2[i] = arr1[i]+arr2[i]-i;
            dp3[i] = arr1[i]-arr2[i]+i;
            dp4[i] = arr1[i]-arr2[i]-i;
        }
        
        
        int f1 = g(dp1,l);
        int f2 = g(dp2,l);
        int f3 = g(dp3,l);
        int f4 = g(dp4,l);
        return Math.max(Math.max(f3,f4),Math.max(f1,f2));
        
    }
    
    
    public int g(int dp1[],int l){
        
        int ma1 = dp1[0];
        int mi1 = dp1[0];
        int r1 = 0;
        for(int i=0;i<l;++i){
            int c = Math.max(Math.abs(dp1[i]-ma1),Math.abs(dp1[i]-mi1));
            r1 = Math.max(r1,c);
            ma1 = Math.max(ma1,dp1[i]);
            mi1 = Math.min(mi1,dp1[i]);
        }
            return r1;
        }
    
    
    
}
