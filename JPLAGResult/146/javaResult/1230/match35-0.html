class Solution {
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
        int max = 0;
        for(int i = arr1.length - 1; i >= 0; --i) {
            for(int j = i - 1; j >= 0; --j) {
                int sum = Math.abs(arr1[i] - arr1[j]) + Math.abs(arr2[i] - arr2[j]) + i - j;
                max = Math.max(sum, max);
            }
        }
        
        return max;
    }
}
