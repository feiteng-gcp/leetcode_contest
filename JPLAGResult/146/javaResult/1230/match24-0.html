class Solution {
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
        int length =arr1.length;
        int max = 0;
        
        for(int i =0;i<length-1;i++){
            for(int j =i+1;j<length;j++){
                int n = Math.abs(arr1[i]-arr1[j])+Math.abs(arr2[i]-arr2[j]) + j-i;
                if(n> max){
                    max = n;
                }
                
            }
        }
        return max;
    }
}
