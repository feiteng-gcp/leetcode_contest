class Solution {
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
        int sum = 0;
        int n = arr1.length;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                sum = Math.max(Math.abs(arr1[j] - arr1[i]) + Math.abs(arr2[j] - arr2[i]) + j - i, sum);
            }
        }
        return sum;
    }
}
