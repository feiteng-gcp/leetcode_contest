class Solution {
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
        int len = arr1.length, res = 0;
        for (int i = 0; i < len - 1; ++i){
            for (int j = i + 1; j < len; ++j){
                res = Math.max(res, Math.abs(arr1[i] - arr1[j]) + Math.abs(arr2[i] - arr2[j]) + j - i);
            }
        }
        return res;
    }
}
