class Solution {
    public int maxAbsValExpr(int[] arr1, int[] arr2) {
        int res = 0;
        for(int i = 0; i < arr1.length; i++) {
            for(int j = i+1; j < arr1.length; j++) {
                int num = j - i + Math.abs(arr1[i]-arr1[j]) + Math.abs(arr2[i]-arr2[j]);
                res = Math.max(num, res);
            }
        }
        return res;
    }
}
