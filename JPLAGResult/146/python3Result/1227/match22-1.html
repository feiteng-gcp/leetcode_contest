class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        hash_map = dict()
        res = 0
        for a, b in dominoes:
            if b > a:
                a, b =
 b, a
            if (a, b) in hash_map:
                res += hash_map[(a, b)]
                hash_map[(a, b)] += 1
            else:
                hash_map[(a, b)] = 1
                
        return res
