class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        v2c={}
        for d in dominoes:
            v=min(d[0],d[1]),max(d[0],d
[1])
            v2c[v]=v2c.get(v,0)+1
        ans=0
        for k,v in v2c.items():
            ans+=v*(v-1)//2
        return ans
