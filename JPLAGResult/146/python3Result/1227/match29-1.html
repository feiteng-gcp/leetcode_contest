class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        N=len(dominoes)
        d=dict()
        num=0
        for i in range(N):
            if dominoes[i][0]>dominoes[i][1]:
                dominoes[i]=(dominoes[i][1],dominoes[i][0])
            else:
                dominoes[i]=(dominoes[i][0],dominoes[i][1])
            
if dominoes[i] in d:
                num+=d[dominoes[i]]
                d[dominoes[i]]+=1
            else:
                d[dominoes[i]]=1
        return num
