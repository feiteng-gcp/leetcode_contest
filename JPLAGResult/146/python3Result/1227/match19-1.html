class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        s = collections.Counter()
        for dominoe in dominoes:
            s[(tuple(sorted(dominoe)))] 
+= 1
        ans = 0
        for k, v in s.items():
            if v > 1:
               ans += (v - 1) * v // 2
        return ans
