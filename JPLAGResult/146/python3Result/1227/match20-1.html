class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        tuples = collections.Counter
()
        for i, j in dominoes:
            if i > j:
                i, j = j, i
            tuples[(i, j)] += 1
        res = 0    
        for t in tuples:
            n = tuples[t]
            while n > 1:
                res += n-1
                n -= 1
        return res
