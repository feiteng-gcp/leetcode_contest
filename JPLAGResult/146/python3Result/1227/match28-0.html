from collections import Counter
class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        cnt = Counter()
        for ls in dominoes:
            x, y = min(ls[0], ls[1]), max(ls[0], ls[1])
            cnt[(x, y)] += 1
        res = 0
        for k in cnt:
            if cnt[k] > 1:
                res += cnt[k] * (cnt[k] - 1) // 2
        return res
