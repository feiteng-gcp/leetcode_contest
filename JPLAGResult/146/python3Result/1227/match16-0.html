class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        counter = dict()
        for domino in dominoes:
            a, b = sorted(domino)
            if (a, b) not in counter: counter[(a, b)] = 1
            else: counter[(a, b)] += 1
        rtn = 0
        for k, v in counter.items():
            rtn += (v * (v - 1)) // 2
        return rtn
