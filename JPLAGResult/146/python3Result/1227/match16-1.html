class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        counts = dict()
        for a, b in dominoes:
            m, l = (a, b) if a < b else (b, a)
            
if (m, l) in counts:
                counts[(m, l)] += 1
            else:
                counts[(m, l)] = 1
        total = 0
        for _, count in counts.items():
            total += (count * (count - 1)) >> 1
        return total
