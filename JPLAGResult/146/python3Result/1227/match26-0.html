class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        dic={}
        for (x,y) in dominoes:
            if x>y:
                x,y=y,x
            if (x,y) in dic:
                dic[(x,y)]+=1
            else:
                dic[(x,y)]=1
        
        ans=0
        for X in dic:
            ans+=(dic[X]*(dic[X]-1))//2
        return ans
