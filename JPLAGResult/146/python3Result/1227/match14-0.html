class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        counter = collections.Counter()
        for x, y in dominoes:
            if x > y:
                x, y = y, x
            counter[(x, y)] += 1
        res = 0
        for _, v in counter.items():
            res += v * (v - 1) // 2
        return res
