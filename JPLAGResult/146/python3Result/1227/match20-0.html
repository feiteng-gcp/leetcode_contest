from collections import defaultdict
class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        
        d = defaultdict(int)
        for x, y in dominoes:
            if x > y:
                x, y = y, x
            d[(x, y)] += 1
            
        res = 0
        for x, y in d:
            cnt = d[x, y]
            if cnt > 1:
                res += cnt * (cnt-1) // 2
        return res
