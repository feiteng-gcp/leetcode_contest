class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        count = collections.defaultdict(int)
        for a, b in dominoes:
            t = (a, b) if a <= b else (b, a)
            count[t] += 1
        ans = 0
        for v in count.values():
            if v > 1:
                ans += v * (v - 1) // 2
        return ans
