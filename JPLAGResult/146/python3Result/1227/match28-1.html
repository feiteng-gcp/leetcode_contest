class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        for i in range(len(dominoes)):
            dominoes[i] = tuple(sorted(dominoes[i]))
        
        counts = {}
        for domino in dominoes:
            if domino in counts:
                counts[domino] += 1
            else:
                counts[domino] = 
1
                
        tot = 0
        for domino in counts:
            if counts[domino] > 1:
                tot += (counts[domino] * (counts[domino] - 1)) // 2
        return tot
