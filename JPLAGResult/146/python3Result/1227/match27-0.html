class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        pmap = collections.defaultdict(int)
        for a, b in dominoes:
            a, b = min(a, b), max(a, b)
            pmap[(a, b)] += 1
        
        res = 0
        for i in range(1, 10):
            for j in range(i, 10):
                if pmap[i, j] >= 2:
                    res += (pmap[i, j]) * (pmap[i, j] - 1) // 2
        
        return res
