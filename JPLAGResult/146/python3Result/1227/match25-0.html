class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        # Tuple, number
        pairs = {}
        
        for domino in dominoes:
            if domino[0] < domino[1]:
                temp = (domino[0], domino[1])
            else:
                temp = (domino[1], domino[0])
            
            if temp in pairs:
                pairs[temp] += 1
            else:
                pairs[temp] = 1
        
        total = 0
        for value in pairs.values():
            total += ((value) * (value - 1)) / 2
            
        return int(total)
