class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        visited = set()
        lookup = collections.defaultdict(dict)
        for a, b in dominoes:
            lookup[a][b] = lookup[a].get(b, 0) + 1
        tmpp = collections.defaultdict(dict)
        for k, v in lookup.items():
            for kk, vv in v.items():
                tmpp[k][kk] = v
        res = 0
        for k, v in lookup.items():
            for kk, vv in v.items():
                if (k, kk) not in visited:
                    if k != kk and k in set(tmpp[kk]):
                        if (kk, k) not in visited:
                            res += (lookup[k][kk] + lookup[kk][k]) * (lookup[k][kk] + lookup[kk][k] - 1) // 2
                            visited.add((kk, k))
                            visited.add((k, kk))
                    else:
                        res += lookup[k][kk] * (lookup[k][kk] - 1) // 2
                        visited.add((k, kk))
        return res
