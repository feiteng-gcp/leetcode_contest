class Solution:
  def count(self, num: int) -> int:
    return (num * (num - 1)) / 2

  
def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
    d = collections.defaultdict(dict)
    for a, b in dominoes:
      x, y = min(a, b), max(a, b)
      d[(x, y)] = d.get((x, y), 0) + 1

    return int(sum([self.count(x) for x in d.values()]))
