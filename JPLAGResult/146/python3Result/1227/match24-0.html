class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        dic = collections.defaultdict(int)
        for a,b in dominoes:
            dic[frozenset([a,b])] += 1
        ans = 0
        #print(dic)
        for x in dic.values():
            ans += (x*(x-1))//2
        return ans
