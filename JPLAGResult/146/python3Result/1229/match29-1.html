class Solution:
    def shortestAlternatingPaths(self, n: int, r: List[List[int]], b: List[List[int]]) -> List[int]:
        rr = [float('inf')] * n
        rr[0] = 0
        br = rr[:]
        s = [rr, br]
        e = [[[] for j in range(n)] for i in range
(2)]
        for u, v in r:
            e[0][u].append(v)
        for u, v in b:
            e[1][u].append(v)
        v = [collections.deque([(0, 0)]) for i in range(2)]
        while v[0] or v[1]:
            for i in range(2):
                if not v[i]:
                    continue
                p, step = v[i].popleft()
                for j in e[1 - i][p]:
                    if s[1 - i][j] > s[i][p] + 1:
                        s[1 - i][j] = s[i][p] + 1
                        v[1 - i].append((j, s[1 - i][j]))
        ret = [min(i, j) for i, j in zip(*s)]
        return [i if i < float('inf') else -1 for i in ret]
