class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List
[int]:
        ret = [-1]*n
        red_graph = collections.defaultdict(set)
        blue_graph = collections.defaultdict(set)
        for i, j in red_edges:
            red_graph[i].add(j)
        for i, j in blue_edges:
            blue_graph[i].add(j)
        q = collections.deque([(0, 0, 0), (0, 0, 1)])
        visited = [set([0]), set([0])]
        while q:
            idx, cost, color = q.popleft()
            if ret[idx] == -1 or ret[idx] > cost:
                ret[idx] = cost
            graph = red_graph
            if color == 0:
                graph = blue_graph
            for j in graph[idx]:
                if j not in visited[color^1]:
                    visited[color^1].add(j)
                    q.append((j, cost+1, color^1))
        return ret
