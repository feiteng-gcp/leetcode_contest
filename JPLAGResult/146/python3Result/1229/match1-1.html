class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        red_min = [-1] * n
        blue_min = [-1] * n
        red_min
[0] = 0
        blue_min[0] = 0
        
        q = collections.deque()
        q.append((0, 'blue'))
        q.append((0, 'red'))
        
        red_neigh = []
        blue_neigh = []
        for i in range(n):
            red_neigh.append(set())
            blue_neigh.append(set())
        for i, j in red_edges:
            red_neigh[i].add(j)
        for i, j in blue_edges:
            blue_neigh[i].add(j)
        
        while q:
            k, color = q.popleft()
            if color == 'red':
                for d in blue_neigh[k]:
                    if blue_min[d] == -1 or blue_min[d] > red_min
[k] + 1:
                        blue_min[d] = red_min[k] + 1
                        q.append((d, 'blue'))
            else:
                for d in red_neigh[k]:
                    if red_min[d] == -1 or red_min[d] > blue_min
[k] + 1:
                        red_min[d] = blue_min[k] + 1
                        q.append((d, 'red'))
        
        answer = []
        for i in range(n):
            if blue_min[i] == -1:
                answer.append(red_min[i])
            elif red_min[i] == -1:
                answer.append(blue_min[i])
            else:
                answer.append(min(red_min[i], blue_min[i]))
        return answer
