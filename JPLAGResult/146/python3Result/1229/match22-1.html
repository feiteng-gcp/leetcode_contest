class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        red = [[] for _ in range(n)]
        blue = [[] for _ in range(n)]
        
        for start, end in red_edges:
            red[start].append(end)
        for start, end in blue_edges:
            blue[start].append(end)
        
        d = [[-1] * n, [-1] * n]
        d[0][0] = d[1][0] = 0
        
        q = collections.deque()
        q.append((0, 0))
        q.append((1, 0))
        
        while q:
            color, v = q.popleft()
            
for u in (red if color else blue)[v]:
                if d[color ^ 1][u] == -1:
                    d[color ^ 1][u] = d[color][v] + 1
                    q.append((color ^ 1, u))
                    
        ans = []
        
        for a, b in zip(d[0], d[1]):
            if a == -1 or (b != -1 and a > b):
                ans.append(b)
            else:
                ans.append(a)
                
        return ans
