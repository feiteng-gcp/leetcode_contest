class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        def bfs(st,color):
            deq=collections.deque()
            used=set()
            #çº¢
            if color==0:
                for adj in red_graph[st]:
                    used.add((st,adj,0))
                    deq.append((adj,1,1))
            else:
                for adj in blue_graph[st]:
                    used.add((st,adj,1))
                    deq.append((adj,1,0))
            while deq:
                cnt=0
                size=len(deq)
                while cnt<size:
                    v,step,c=deq.popleft()
                    res[v]=min(res[v],step)
                    if c==1:
                        for adj in blue_graph[v]:
                            if (v,adj,1) not in used:
                                used.add((v,adj,1))
                                deq.append((adj,step+1,0))
                    else:
                        for adj in red_graph[v]:
                            if(v,adj,0) not in used:
                                used.add((v,adj,0))
                                deq.append((adj,step+1,1))
                    cnt+=1
                #print(deq)
                
        red_graph=collections.defaultdict(set)
        blue_graph=collections.defaultdict(set)
        for e in red_edges:
            red_graph[e[0]].add(e[1])
        for e in blue_edges:
            blue_graph[e[0]].add(e[1])
        
        res=collections.defaultdict(int)
        for i in range(n):
            res[i]=float("inf")
            
        bfs(0,0)
        bfs(0,1)
        ret=[-1]*n
        for k in  res:
            if res[k]!=float("inf"):
                ret[k]=res[k]
        ret[0]=0
        return ret
