class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        red = [-1] * n
        blue = [-1] * n
        def get_g(edges):
            adj = [[] for _ in range(n)]
            for u, v in edges:
                adj[u].append(v)
            return adj
        red_g = get_g(red_edges)
        blue_g = get_g(blue_edges)
        red[0] = blue[0] = 0
        que = collections.deque()
        que.append((0, 0))
        que.append((0, 1))
        while que:
            node, co = que.popleft()
            if co == 0:
                for v in blue_g[node]:
                    if blue[v] == -1:
                        blue[v] = red[node] + 1
                        que.append((v, 1))
            elif co == 1:
                for v in red_g[node]:
                    if red[v] == -1:
                        red[v] = blue[node] + 1
                        que.append((v, 0))
        ans = []
        for i in range(n):
            if red[i] == -1:
                ans.append(blue[i])
            elif blue[i] == -1:
                ans.append(red[i])
            else:
                ans.append(min(red[i], blue[i]))
        return ans
