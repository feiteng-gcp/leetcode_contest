class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        red_end=[-1]*n
        red_end[0]=0
        blue_end=[-1]*n
        blue_end[0]=0
        for _ in range(n):
            for i,j in red_edges:
                if blue_end[i]!=-1:
                    if red_end[j]==-1 or red_end[j]>blue_end[i]+1:
                        red_end[j]=blue_end[i]+1
            for i,j in blue_edges:
                if red_end[i]!=-1:
                    if blue_end[j]==-1 or blue_end[j]>red_end[i]+1:
                        blue_end[j]=red_end[i]+1
        res
=[]
        for i in range(n):
            if red_end[i]==-1:
                res.append(blue_end[i])
            elif blue_end[i]==-1:
                res.append(red_end[i])
            else:
                res.append(min(red_end[i],blue_end[i]))
        return res
