from collections import defaultdict, deque
class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges, blue_edges):
        ret = [-1 for x in range(n)]
        ret[0] = 0
        reds = defaultdict(list)
        for s, e in red_edges:
            reds[s].append(e)
        blues = defaultdict(list)
        for s, e in blue_edges:
            blues[s].append(e)
        edges = [reds, blues]
        start = deque([(e, 0, 1) for e in reds[0]] + [(e, 1, 1) for e in blues[0]])
        visited = set() # (e, prev_color)
        while len(start) > 0:
            end, prev_color, length = start.popleft()
            if (end, prev_color) in visited:
                continue
            visited.add((end, prev_color))
            if ret[end] == -1:
                ret[end] = length
            for nxt in edges[1-prev_color][end]:
                start.append((nxt, 1 - prev_color, length + 1))
        return ret
