class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        res = [-1] * n
        suc = [collections.defaultdict(list), collections.defaultdict(list)]
        for i, j in red_edges:
            suc[0][i].append(j)
        for i, j in blue_edges:
            suc[1][i].append(j)
        q = collections.deque()
        q.append((0, 0, 0))
        q.append((0, 1, 0))
        seen = set()
        while q:
            node, color, step = q.popleft()
            if (node, color) in seen:
                continue
            seen.add((node, color))    
            if res[node] == -1 or step <= res[node]:
                res[node] = step
            for j in suc[1-color][node]:
                q.append((j, 1-color, step + 1))
        return res
