from collections import deque
class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        if n == 1:
            return [0]
        red 
= collections.defaultdict(set)
        blue = collections.defaultdict(set)
        for s, d in red_edges:
            red[s].add(d)
        for s, d in blue_edges:
            blue[s].add(d)
        ans = [-1]*n
        queue = deque([(0, -1)]) # -1表示无色，0表示红色，1表示蓝色
        visited = set()
        path = 0
        while queue:
            l = len(queue)
            for _ in range(l):
                v, color = queue.popleft()
                if ans[v] == -1:
                    ans[v] = path
                if color == -1 or color == 0:
                    for nei in blue[v]:
                        if (v, nei, 1) not in visited:
                            visited.add((v, nei, 1))
                            queue.append((nei, 1))
                if color == -1 or color == 1:
                    for nei in red[v]:
                        if (v, nei, 0) not in visited:
                            visited.add((v, nei, 0))
                            queue.append((nei, 0))
            path += 1
        return ans
