class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        if n == 1:
            return [0]
        cs = set()
        l = [[] for _ in range(101)]
        ll = [[] for _ in range
(101)]
        
        for x,y in red_edges:
            l[x].append(y)
        for x,y in blue_edges:
            ll[x].append(y)
        res = [-1]*n
        from collections import deque
        q = deque()
        q.append((0,0,0))
        res[0] = 0
        cs.add((0,1))
        cs.add((0,2))
        while(len(q)):
            x,t,n = q.popleft()
            if (x,t) in cs:
                continue
            if res[x] == -1:
                res[x] = n
            cs.add((x,t))
            if t == 1:
                for e in l[x]:
                    q.append((e,2,n+1))
            elif t == 2:
                for e in ll[x]:
                    q.append((e,1,n+1))
            else:
                
for e in l[x]:
                    q.append((e,2,n+1))
                for e in ll[x]:
                    q.append((e,1,n+1))
        return res
