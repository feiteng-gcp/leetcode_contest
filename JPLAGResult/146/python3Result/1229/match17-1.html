class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        d1, d2 = collections.defaultdict(list), collections.defaultdict
(list)
        for a, b in red_edges:
            d1[a].append(b)
        for a, b in blue_edges:
            d2[a].append(b)
        q, r, seen = [(0, 0, True), (0, 0, False)], {0: 0}, {(0, True), (0, False)}
        while q:
            a, c, isRed = q.pop(0)
            if a not in r: r[a] = c
            for b in d2[a] if isRed else d1[a]:
                if (b, not isRed) not in seen:
                    seen.add((b, not isRed))
                    q.append((b, c+1, not isRed))
        return [r.get(i, -1) for i in range(n)]
