from collections import defaultdict
class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        red = [-1] * n
        blue = [-1] * n
        red[0], blue
[0] = 0, 0
        
        re = defaultdict(set)
        for f, t in red_edges:
            re[f].add(t)
        be = defaultdict(set)
        for f, t in blue_edges:
            be[f].add(t)
        # print(re, be)
        rq, bq = set([0]), set([0])
        rseen, bseen = set(), set()
        while rq or bq:
            nrq, nbq = set(), set()
            rseen |= rq
            bseen |= bq
            for f in rq:
                for t in re[f]:
                    if t not in bseen:
                        nbq.add(t)
                    if blue[f] == -1:
                        continue
                    if red[t] == -1:
                        red[t] = blue[f] + 1
                    else:
                        red[t] = min(red[t], blue[f] + 1)
            for f in bq:
                for t in be[f]:
                    if t not in rseen:
                        nrq.add(t)
                    if red[f] == -1:
                        continue
                    if blue[t] == -1:
                        blue[t] = red[f] + 1
                    else:
                        blue[t] = min(blue[t], red[f] + 1)
            rq = nrq
            bq = nbq
        # print(red)
        # print(blue)
        ret = []
        for i in range(len(red)):
            if red[i] == -1:
                ret.append(blue[i])
            elif blue[i] == -1:
                ret.append(red[i])
            else:
                ret.append(min(red[i], blue[i]))
        return ret
