class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List
[int]:
        graph = []
        graph_r = collections.defaultdict(set)
        graph_b = collections.defaultdict(set)
        for x, y in red_edges:
            graph_r[x].add(y)
        for x, y in blue_edges:
            graph_b[x].add(y)
        graph.append(graph_r)
        graph.append(graph_b)
        # print(graph)
        
        res = [-1 for _ in range(n)]
        res[0] = 0
        queue = collections.deque()
        seen = set()
        
        for color in range(2):
            for neigh in graph[color][0]:
                queue.append((neigh, 1, color))
                seen.add((neigh, color))
        # print(graph)
        # print(queue)
        while queue:
            node, step, color = queue.popleft()
            if res[node] == -1:
                res[node] = step
            for neigh in graph[1 - color][node]:
                if (neigh, 1 - color) in seen:
                    continue
                queue.append((neigh, step + 1, 1 - color))
                seen.add((neigh, 1 - color))
        
        return res
