class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        ans = [float('inf')]*n
        ans[0] = 0
        
        red_adj, blue_adj = collections.defaultdict(set), collections.defaultdict(set)
        red_seen, blue_seen = set(), set
()
        for src, dst in red_edges:
            red_adj[src].add(dst)
        for src, dst in blue_edges:
            blue_adj[src].add(dst)
        
    
        q = collections.deque()
        q.append((0,'r',0))
        while q:
            node, color, dis = q.popleft()
            ans[node] = min(ans[node], dis)
            if color == 'r':
                red_seen.add(node)
                for neighbour in blue_adj[node]:
                    if neighbour not in blue_seen:
                        q.append((neighbour, 'b', dis+1))
            else:
                blue_seen.add(node)
                for neighbour in red_adj[node]:
                    if neighbour not in red_seen:
                        q.append((neighbour, 'r', dis+1))
        
        q.append((0,'b',0))
        red_seen, blue_seen = set(), set()
        while q:
            node, color, dis = q.popleft()
            ans[node] = min(ans[node], dis)
            if color == 'r':
                red_seen.add(node)
                for neighbour in blue_adj[node]:
                    if neighbour not in blue_seen:
                        q.append((neighbour, 'b', dis+1))
            else:
                blue_seen.add(node)
                for neighbour in red_adj[node]:
                    if neighbour not in red_seen:
                        q.append((neighbour, 'r', dis+1))
        
        for i, a in enumerate(ans):
            if a == float('inf'):
                ans[i] = -1
        return ans
