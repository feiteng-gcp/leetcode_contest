import collections
class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        edges = [[[], []] for _ in range
(n)]
        for i, j in red_edges:
            edges[i][0].append(j)
        for i, j in blue_edges:
            edges[i][1].append(j)
        
        result = [[-1] * 2 for _ in range(n)]
        queue = collections.deque()
        queue.append((0, 0))
        queue.append((0, 1))
        result[0][0] = 0
        result[0][1] = 0
        
        
while queue:
            i, color = queue.popleft()
            dist = result[i][color]+1
            color = 1-color
            for j in edges[i][color]:
                if result[j][color] == -1:
                    result[j][color] = dist
                    queue.append((j, color))
        
        return [min(a, b) if min(a, b)!= -1 else max(a, b) for a, b in result]
