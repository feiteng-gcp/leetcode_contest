class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        graph = [{}, {}]
        for i in range(n):
            graph[0][i] = []
            graph[1][i] = []
        
for i, j in red_edges:
            graph[0][i].append(j)
        for i, j in blue_edges:
            graph[1][i].append(j)
        dp = [[None, None] for _ in range(n)]
        level = [[0, 0], [0, 1]]
        level_count = 0
        while level:
            next_level = []
            for node, color in level:
                if dp[node][color] is None:
                    dp[node][color] = level_count
                    for next_node in graph[1 - color][node]:
                        if dp[next_node][1 - color] is None:
                            next_level.append([next_node, 1 - color])
            level_count += 1
            level = next_level
        res = []
        for a, b in dp:
            if a is not None and b is not None:
                res.append(min(a, b))
            elif a is None and b is None:
                res.append(-1)
            elif a is None:
                res.append(b)
            else:
                res.append(a)
        return res
