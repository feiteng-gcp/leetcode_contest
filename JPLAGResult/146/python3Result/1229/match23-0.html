class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        dicblue, dicred = collections.defaultdict(list), collections.defaultdict(list)
        
        ans = collections.defaultdict(int)
        ans[0] = 0
        for a, b in red_edges:
            dicred[a].append(b)
        for a,b in blue_edges:
            dicblue[a].append(b)
        
        stack = []
        visited = set()
        
        for node in dicred[0]:
            stack.append((node, 1, 0))
            visited.add((node,0))
        for node in dicblue[0]:
            stack.append((node,1,1))
            visited.add((node,1))
        #print(stack)

        while stack:
            node, l, color = stack.pop(0) # 1: blue, 0:red lead by this color edge
            if node not in ans:
                ans[node] = l
            else:
                ans[node] = min(ans[node], l)
            #print(ans)
            
            if color == 0:
                temp = dicblue
            elif color == 1:
                temp = dicred
            
                
            for x in temp[node]:
                if (x, color^1) not in visited:
                    stack.append((x,l+1, color^1))
                    visited.add((x,color^1))
        #print(ans)
        res = [-1]*n
        for i, v in ans.items():
            res[i] = v
        return res
