from collections import deque
class Solution:
    def shortestAlternatingPaths(self, N: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        links = [[[] for _ in range(N)], [[] for _ in range(N)]]
        for s, e in red_edges:
            links[0][s].append(e)
        for s,e in blue_edges:
            links[1][s].append(e)
        
        steps = [[1e9 for _ in range(N)], [1e9 for _ in range(N)]]
        steps[0][0] = 0
        steps[1][0] = 0
        q = deque([(0, 0), (1, 0)])
        while q:
            c, n = q.popleft()
            s = steps[c][n]
            cc = (c + 1) & 1
            for nn in links[c][n]:
                if s + 1 < steps[cc][nn]:
                    steps[cc][nn] = s + 1
                    q.append((cc, nn))
                    
        return [min(steps[0][i], steps[1][i]) if min(steps[0][i], steps[1][i]) < 1e7 else -1 for i in range(N)]
