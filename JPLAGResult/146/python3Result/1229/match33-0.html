class Solution:
    
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        
        toList_red = [set() for _ in range(n)]
        toList_blue = [set() for _ in range(n)]

        for s, t in red_edges:
            toList_red[s].add(t)
        for s, t in blue_edges:
            toList_blue[s].add(t)
        dist = [-1]*n
        dist[0] = 0
        q = [(0, 0), (0, 1)]
        visited = set(q)
        step = 1
        while q:
            newq = []
            for cur, prevColor in q:
                toList = toList_red if prevColor else toList_blue
                for to in toList[cur]:
                    newState = (to, prevColor^1)
                    if newState in visited:
                        continue
                    visited.add(newState)
                    newq.append((to, prevColor^1))
                    if dist[to] == -1:
                        dist[to] = step
            step += 1
            q = newq
        return dist
