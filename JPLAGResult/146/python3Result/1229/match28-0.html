class Solution:
  def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
    reds = [[] for _ in range(n)]
    blues = [[] for _ in range(n)]
    for x, y in red_edges:
      reds[x] += y,
    for x, y in blue_edges:
      blues[x] += y,
    inf = 1000010000
    t = [[inf, inf] for _ in range(n)]
    t[0][0] = t[0][1] = 0
    que = [(0, 0), (0, 1)]
    hd = 0
    while hd < len(que):
      x, y = que[hd]
      hd += 1
      if y:
        for u in reds[x]:
          if t[u][0] > t[x][1] + 1:
            t[u][0] = t[x][1] + 1
            que += (u, 0),
      else:
        for u in blues[x]:
          if t[u][1] > t[x][0] + 1:
            t[u][1] = t[x][0] + 1
            que += (u, 1),
    ret = []
    for i in range(n):
      x = min(t[i])
      if x < inf:
        ret += x,
      else:
        ret += -1,
    return ret
