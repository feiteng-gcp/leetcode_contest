from collections import defaultdict, deque
import sys
class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        redGraph = defaultdict(set)
        blueGraph = defaultdict(set)
        graph = [redGraph, blueGraph]
        for u, v in red_edges:
            redGraph[u].add(v)
        for u, v in blue_edges:
            blueGraph[u].add(v)
        ans = [sys.maxsize] * n
        q = deque()
        #ind, isRed, depth
        q.appendleft((0,0,0))
        visited = set([(0,0)])
        while q:
            u, isRed, depth = q.pop()
            ans[u] = min(ans[u],depth)
            for v in graph[isRed][u]:
                if (isRed, v) not in visited:
                    visited.add((isRed, v))
                    q.appendleft((v, 1-isRed, depth+1))
                    
        q.appendleft((0,1,0))
        visited = set([(1,0)])
        while q:
            u, isRed, depth = q.pop()
            ans[u] = min(ans[u], depth)
            for v in graph[isRed][u]:
                if (isRed,v) not in visited:
                    visited.add((isRed,v))
                    q.appendleft((v, 1-isRed, depth+1))
        for i in range(n):
            if ans[i] == sys.maxsize: ans[i] = -1
        return ans
