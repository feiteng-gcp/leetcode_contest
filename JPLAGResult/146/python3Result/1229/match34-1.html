from collections import defaultdict, deque

class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List
[int]:
        # Isn't this just Dijkstra's? but with additional stipulation (red edges, blue edges)
        # Two iterations?
        
        # Get better edges
        red_edges_dict = defaultdict(list)
        blue_edges_dict = defaultdict(list)
        
        for re in red_edges:
            red_edges_dict[re[0]].append(re[1])
        for be in blue_edges:
            blue_edges_dict[be[0]].append(be[1])
            
        # print(red_edges_dict)
        # print(blue_edges_dict)
        
        
        def weird_dijkstra(isRed):
            queue = deque()
            blue_visited = {}
            red_visited = {}
            counter = 0

            queue.append(0)
            if isRed:
                blue_visited[0] = counter
            else:
                red_visited[0] = counter

            while(len(queue) > 0):
                print(queue)
                current_len = len(queue)
                counter += 1
                
                for i in range(current_len):
                    current_node = queue.popleft()

                    if isRed:
                        keys = red_visited.keys()
                        for edge in red_edges_dict[current_node]:
                            if edge not in keys:
                                red_visited[edge] = counter
                                queue.append(edge)
                    else:
                        keys = blue_visited.keys()
                        for edge in blue_edges_dict[current_node]:
                            if edge not in keys:
                                blue_visited[edge] = counter
                                queue.append(edge)

                isRed = not isRed
            
            print(blue_visited)
            print(red_visited)
            best = []
            for i in range(n):
                if i in blue_visited and i in red_visited:
                    best.append(min(blue_visited[i], red_visited[i]))
                elif i in blue_visited:
                    best.append(blue_visited[i])
                elif i in red_visited:
                    best.append(red_visited[i])
                else:
                    best.append(-1)

            return best
        
        
        red_best = weird_dijkstra(True)
        blue_best = weird_dijkstra(False)
            
        # best = [min(x, y) for x, y in zip(red_best, blue_best)]
        best = []
        for x, y in zip(red_best, blue_best):
            if x == -1 and y == -1:
                best.append(-1)
            elif x == -1:
                best.append(y)
            elif y == -1:
                best.append(x)
            else:
                best.append(min(x, y))
            
            
        print(best)
        return best
