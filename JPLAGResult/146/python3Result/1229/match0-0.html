class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        dire_r = collections.defaultdict(list)
        dire_b = collections.defaultdict(list)
        for s,e in red_edges:
            dire_r[s] += [e]
        for s,e in blue_edges:
            dire_b[s] += [e]
        path = [dire_r,dire_b]
        res = [0]
        for i in range(1,n):
            tmp1, tmp2 = 0, 0
            f1, f2 = False, False
            cur = set([0])
            seen = set([0])
            while cur and tmp1 <= len(red_edges)+len(blue_edges):
                tmp_node = []
                for node in cur:
                    for tar in path[tmp1%2][node]:
                        tmp_node += [tar]
                cur = set(tmp_node)
                seen |= cur
                tmp1 += 1
                if i in seen:
                    f1 = True
                    break
            cur = set([0])
            seen = set([0])
            while cur and tmp2 <= len(blue_edges)+len(red_edges):
                tmp_node = []
                for node in cur:
                    for tar in path[(tmp2+1)%2][node]:
                        tmp_node += [tar]
                cur = set(tmp_node)
                seen |= cur
                tmp2 += 1
                if i in seen:
                    f2 = True
                    break
            if not f1:
                tmp1 = 1000
            if not f2:
                tmp2 = 1000
            if f1 or f2:
                res += [min(tmp1,tmp2)]
            else:
                res += [-1]
        return res
