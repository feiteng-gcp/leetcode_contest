class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        res = [-1 for _ in range(n)]
        res[0] = 0
        
        red_graph = collections.defaultdict(list)
        blue_graph = collections.defaultdict(list)
        for a, b in red_edges:
            red_graph[a].append(b)
        for a, b in blue_edges:
            blue_graph[a].append(b)
        

        queue = collections.deque([(0, 0, 'r'), (0, 0, 'b')])
        visited = set()
        
        while queue:
            node, path, color = queue.popleft()
            if (node, color) in visited:
                continue
            visited.add((node, color))
            if res[node] == -1:
                res[node] = path
            if color == 'r':
                for nei in blue_graph[node]:
                    queue.append((nei, path+1, 'b'))
            else:
                for nei in red_graph[node]:
                    queue.append((nei, path+1, 'r'))
                    
        return res
