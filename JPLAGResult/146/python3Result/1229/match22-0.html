class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        dicr=[{} for i in range(n)]
        dicb=[{} for i in range(n)]
        for ed in red_edges:
            i=ed[0];j=ed[1]
            dicr[i][j]=-1
        for ed in blue_edges:
            i=ed[0];j=ed[1]
            dicb[i][j]=-1
        dp=[]
        ans=[[100000,100000] for i in range(n)]
        ans[0][0]=0
        ans[0][1]=0
        tg=0
        for j in dicr[0]:
            if ans[j][0]==100000:
                ans[j][0]=1
                dp.append((j,0))
        for j in dicb[0]:
            if ans[j][1]==100000:
                ans[j][1]=1
                dp.append((j,1))
        while(tg<len(dp)):
            i=dp[tg][0]
            lab=dp[tg][1]
            if lab==0:
                for j in dicb[i]:
                    if ans[j][1]==100000:
                        ans[j][1]=ans[i][0]+1
                        dp.append((j,1))
            else:
                for j in dicr[i]:
                    if ans[j][0]==100000:
                        ans[j][0]=ans[i][1]+1
                        dp.append((j,0))
            tg=tg+1
            #print(ans)
            #print(dp)
        s=[-1 for i in range(n)]
        for i in range(n):
            temp=min(ans[i][0],ans[i][1])
            if temp<100000:
                s[i]=temp
        return s
