class Solution:
    def shortestAlternatingPaths(self, N: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        link = [collections.defaultdict(list), collections.defaultdict(list)]
        for L,R in red_edges:
            link[0][L].append(R)
        for L,R in blue_edges:
            link[1][L].append(R)
        BFS = [(0,0,0), (1,0,0)] # (color, dst, node)
        ans = {}
        for color, dst, n in BFS:
            if (color,n) not in ans: 
                ans[color,n] = dst    
                BFS.extend( (color^1, dst+1, R) for R in link[color][n])
        out = [-1]*N
        for color,n in ans:
            if out[n] == -1:
                out[n]= ans[color,n]
            else:
                out[n]= min(out[n], ans[color,n])
        return out
