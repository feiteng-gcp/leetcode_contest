class Solution:
    def shortestAlternatingPaths(self, n: int, red_edges: List[List[int]], blue_edges: List[List[int]]) -> List[int]:
        red_adj = [[] for _ in range(n)]
        blue_adj = [[] for _ in range(n)]
        for i, j in red_edges:
            red_adj[i].append(j)
        for i, j in blue_edges:
            blue_adj[i].append(j)
            
        end_red  = [1000 for _ in range(n)]
        end_blue = [1000 for _ in range(n)]
        end_red[0] = end_blue[0] = 0
        
        frontier = set([0])
        while frontier:
            new_frontier = set()
            for src in frontier:
                for trg in red_adj[src]:
                    if end_blue[src] + 1 < end_red[trg]:
                        end_red[trg] = end_blue[src] + 1
                        new_frontier.add(trg)
                for trg in blue_adj[src]:
                    if end_red[src] + 1 < end_blue[trg]:
                        end_blue[trg] = end_red[src] + 1
                        new_frontier.add(trg)
            frontier = new_frontier
            
        result = [min(red, blue) for red, blue in zip(end_red, end_blue)]
        result = [-1 if r == 1000 else r for r in result]
        return result
