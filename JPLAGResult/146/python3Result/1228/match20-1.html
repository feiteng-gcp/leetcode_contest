class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        
        memo = {}
        def helper(i, j):
            if (i, j) in memo:
                return memo[(i, j)]
            
            if i + 1 == j:
                memo[(i, j)] = (0, arr[i])
                return memo[(i, j)]
            elif i + 2 == j:
                memo[(i, j)] = (arr[i] * arr[i+1], max(arr[i], arr[i+1]))
                return memo[(
i, j)]
            else:
                ans = float("Inf")
                curr_max = 0
                for k in range(i + 1, j):
                    s1, max1 = helper(i, k)
                    s2, max2 = helper(k, j)
                    if s1 + s2 + max1 * max2 < ans:
                        ans = s1 + s2 + max1 * max2
                        curr_max = max(max1, max2)
                memo[(i, j)] = (ans, curr_max)
                return memo[(i, j)]
        
        res = helper(0, len(arr))[0]
        return res
