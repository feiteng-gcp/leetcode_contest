from collections import defaultdict
class Solution:
    def mctFromLeafValues(self, arr) -> int:
        mi = {}
        for i in range(len(arr)):
            for j in range(i,len(arr)):
                mi[(i,j)] = max(arr[i:j+1])
        memo={}
        def dfs(s,e):
            if s == e:
                return 0
            if s == e-1:
                return arr[s]*arr[e]
            if (s,e) in memo:
                return memo[(s,e)]
            res = 99999999999999
            
for i in range(s,e):
                #print(s,i,e,mi[(s,i)],mi[(i+1,e)])
                res = min(res,dfs(s,i)+dfs(i+1,e)+mi[(s,i)]*mi[(i+1,e)])
            
            #print(s,e,res)
            memo[(s,e)] = res
            #print(memo)
            return res
        #print(memo)
        return dfs(0,len(arr)-1)
