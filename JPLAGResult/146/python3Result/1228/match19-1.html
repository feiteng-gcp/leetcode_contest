class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        result = 0
        while len(arr) > 1:
            prod = 226
            idx = 0
            
for i in range(len(arr) - 1):
                if arr[i] * arr[i+1] < prod:
                    prod = arr[i] * arr[i+1]
                    idx = i
            arr[idx] = max(arr[idx], arr[idx+1])
            del arr[idx+1]
            result += prod
        return result
