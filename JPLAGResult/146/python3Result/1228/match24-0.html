class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        n = len(arr)
        dic = {}
        temp = {}
        for i in range(n):
            for j in range(i+1,n+1):
                temp[(i,j)] = max(arr[i:j])
        #print(temp)
        
        def helper(i,j):
            #print(i,j)
            if j == i+1:
                return 0
            if j == i+2:
                return arr[i]*arr[i+1]
            if (i,j) not in dic:
                ans = float('inf')
                for t in range(i,j-1):
                    ans = min(ans, helper(i, t+1) + helper(t+1,j) + temp[(i,t+1)]*temp[(t+1,j)])
                dic[(i,j)] = ans
            return dic[(i,j)]
        
        return helper(0,n)
