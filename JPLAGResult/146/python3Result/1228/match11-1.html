class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        ans = 0
        while len(arr) > 1:
            i, smallest, idx = 0, 2**31, 0
            for i in range(len(arr) - 1):
                if arr[i] * arr[i+1] < smallest:
                    smallest = arr[i] * arr[i+1]
                    idx = i
            neww = max(arr[idx], arr[idx+1])
            ans += arr[idx] * arr[idx+1]
            del arr[idx+1]
            arr[idx] = neww
        return ans
