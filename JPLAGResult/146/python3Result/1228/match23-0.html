import sys
class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        if not arr:
            return 0
        minSum = {}

        def DFS(i, j):
            #rtype minSum
            if i == j:
                return 0
            if i == j - 1:
                return arr[i] * arr[j]
            if (i, j) in minSum:
                return minSum[(i, j)]
            ans = sys.maxsize
            for k in range(i, j):
                _ans = DFS(i, k) + DFS(k+1, j) + max(arr[i:k+1]) * max(arr[k+1:j+1])
                ans = min(ans,_ans)
            minSum[(i,j)] = ans
            return ans
        ans = DFS(0, len(arr)-1)
        return ans
