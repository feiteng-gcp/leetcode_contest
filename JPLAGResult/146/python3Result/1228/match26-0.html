class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        """
        dp[i][j] = min(dp[i][k]+dp[k+1][j] + max(arr[:k+1])*max(arr[k+1:]))
        """
        N = len(arr)
        dp = [[[0, 0] for _ in range(N)] for _ in range(N)]
        for l in range(1, N+1):
            for i in range(N-l+1):
                j = i+l-1
                if l == 1:
                    dp[i][j][1] = arr[i]
                    continue
                elif l == 2:
                    dp[i][j] = [arr[i]*arr[j], max(arr[i], arr[j])]
                dp[i][j] = [float("inf"), float("inf")]
                for k in range(i, j):
                    if dp[i][k][0]+dp[k+1][j][0] + dp[i][k][1]*dp[k+1][j][1] < dp[i][j][0]:
                        dp[i][j] = [dp[i][k][0]+dp[k+1][j][0] + dp[i][k][1]*dp[k+1][j][1], max(dp[i][k][1], dp[k+1][j][1])]
                    elif dp[i][k][0]+dp[k+1][j][0] + dp[i][k][1]*dp[k+1][j][1] == dp[i][j][0] and \
                    max(dp[i][k][1], dp[k+1][j][1]) < dp[i][j][1]:
                        dp[i][j][1] = max(dp[i][k][1], dp[k+1][j][1])
        return dp[0][-1][0]
