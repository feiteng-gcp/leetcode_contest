class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        seen = {}
        
        def helper(i, j):
            if (i, j) not in seen:
                if i == j:
                    seen[(i, j)] = 0
                else:
                    seen[(i, j)] = float('inf')
                    for k in range(0, j-i):
                        seen[(i, j)] = min(seen[(i, j)], helper(i, i+k) + helper(i+k+1, j) + (max(arr[i: i+k+1]) * max(arr[i+k+1: j+1])))
            return seen[(i, j)]
        
        return helper(0, len(arr) - 1)
