class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        #best = float('inf')
        cache = {}
        def dfs(arr, start, end):
            if (start, end) in cache:
                return cache[(start, end)]
            if start == end:
                return arr[start], 0
            best = float('inf')
            m = arr[start]
            for i in range(start, end):
                lm, la = dfs(arr, start, i)
                rm, ra = dfs(arr, i + 1, end)
                m = max(m, lm, rm)
                best = min(best, la + ra + rm * lm)
            cache[(start, end)] = (m, best)
            return m, best
        _, ans = dfs(arr, 0, len(arr) - 1)
        return ans
