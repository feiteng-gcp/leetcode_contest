from functools import lru_cache
class Solution:
    def mctFromLeafValues(self, arr):
        maxnum = {}
        for i in range(len(arr)):
            for j in range(i+1, len(arr)+1):
                maxnum[(i,j)] = max(arr[i:j])
        
        @lru_cache(None)
        def dp(l,r):
            if r - l == 1: return 0
            if r - l == 2: return arr[l] * arr[l+1]
            return min( dp(l,i) + dp(i,r) +  maxnum[(l,i)] * maxnum[(i,r)] for i in range(l+1, r) )
        return dp(0, len(arr))
