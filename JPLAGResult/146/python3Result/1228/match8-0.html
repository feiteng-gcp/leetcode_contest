class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        def function(s):
            result=[[float("inf")]*len(s) for _ in range(len(s))]
            for i in range(len(s)):
                result[i][i]=0
            for d in range(1,len(s)):
                for i in range(len(s)-d):
                    for k in range(i,i+d):
                        result[i][i+d]=min(result[i][i+d],result[i][k]+result[k+1][i+d]+max(s[i:k+1])*max(s[k+1:i+d+1]))
            return result[0][-1]
        return function(arr)
