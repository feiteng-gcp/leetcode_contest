class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        L=[[float("inf")]* len(arr) for _ in range(len(arr))]
        for i in range(len(arr)):
            L[i][i]=0
        for j in range(1,len(arr)):
            for i in range(len(arr)-j):
                for k in range(j):
                    leaf1=max(arr[i:i+k+1])
                    leaf2=max(arr[i+k+1:i+j+1])
                    L[i][i+j]=min(L[i][i+j],L[i][i+k]+L[i+k+1][i+j]+leaf1*leaf2)
        return L[0][-1]
