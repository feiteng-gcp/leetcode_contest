INF=999999999999999999
class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        '''
        f[x][y]:ans for [x..y]
        f[x][y]=min(f[x,k]+f[x,k+1]+max(x..k)+min(k+1..y))
        '''
        n=len(arr)
        mmm=[[0 for _ in range(n)] for _ in range
(n)]
        for x in range(n): mmm[x][x]=arr[x]
        for w in range(2,n+1):
            for x in range(n):
                y=x+w-1
                if y>=n: break
                mmm[x][y]=max(mmm[x][y-1],arr[y])
        f=[[INF for _ in range(n)] for _ in range(n)]
        for x in range(n): f[x][x]=0
        for w in range(2,n+1):
            for x in range(n):
                y=x+w-1
                if y>=n: break
                
for k in range(x,y):
                    crt=f[x][k]+f[k+1][y]+mmm[x][k]*mmm[k+1][y]
                    f[x][y]=min(f[x][y],crt)
        return f[0][n-1]
