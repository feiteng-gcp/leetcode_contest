class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        n = len(arr)
        f = [[0] * n for _ in range(n)]
        f[0] = arr
        for l in range(1, n):
            for i in range(0, n - l):
                f[l][i] = 1 << 32
                for m in range(l):
                    f[l][i] = min(f[l][i], f[m][i] + f[l - m - 1][i + m + 1] + max(arr[i:i+m+1]) * max(arr[i+m+1:i+l+1]))
        return f[n - 1][0] - sum(arr)
