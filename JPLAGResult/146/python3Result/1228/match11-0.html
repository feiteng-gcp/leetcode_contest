class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        ans = 0
        
        while len(arr) > 1:
            j = 0
            
            for i in range(1, len(arr)-1):
                if arr[i] * arr[i+1] < arr[j] * arr[j+1]:
                    j = i
                    
            ans += arr[j] * arr[j+1]
            arr[j:j+2] = [max(arr[j:j+2])]
            #print(j, arr, ans)
            
        return ans
