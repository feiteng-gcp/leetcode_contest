class Solution:
    def mctFromLeafValues(self, arr: 'List[int]') -> int:
        d={}
        
def search(st,en):
            if st==en: return 0
            if (st,en) in d:
                return d[(st,en)]
            ans=float('inf')
            for i in range(st,en): #st ->en-1
                ans=min(ans,search(st,i)+search(i+1,en)+max(arr[st:i+1])*max(arr[i+1:en+1]))
            d[(st,en)]=ans
            return ans
        return search(0,len(arr)-1)
