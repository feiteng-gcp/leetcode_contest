class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        n = len(arr)
        if n == 2:
            return arr[0]*arr[1]
        maxMatrix = [[0]*n for _ in range(n)]
        for i in range(n):
            maxMatrix[i][i] = arr[i]
            temp = arr[i]
            for j in range(i+1, n):
                temp = max(temp, arr[j])
                maxMatrix[i][j] = temp
        ans = [[1<<31]*n for _ in range(n)]
        for i in range(n):
            ans[i][i] = 0
        for i in range(n-1):
            ans[i][i+1] = arr[i]*arr[i+1]
        for i in range(n-3,-1,-1):
            for j in range(i+2, n):
                for k in range(i, j):
                    ans[i][j] = min(ans[i][j], ans[i][k]+ans
[k+1][j]+maxMatrix[i][k]*maxMatrix[k+1][j])
        return ans[0][n-1]
