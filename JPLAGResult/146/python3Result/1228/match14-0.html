from functools import lru_cache

class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        N = len(arr)
        
        @lru_cache(maxsize=None)
        def find_max(i: int, j: int) -> int:
            return max(arr[i:j+1])
        
        @lru_cache(maxsize=None)
        def recurse(i: int, j: int) -> int:
            if i >= j:
                return 0
            
            res = float('inf')
            for k in range(i, j):
                res = min(res, find_max(i, k) * find_max(k+1, j) + recurse(i, k) + recurse(k+1, j))
            return res
        
        return recurse(0, N-1)
