class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        return self.helper(tuple(arr), {})
        
    
def helper(self, nums, memo):
        if len(nums) == 1:
            return 0
        
        if nums in memo: return memo[nums]
        
        minsum = float("inf")
        for i in range(len(nums) - 1):
            lv = nums[:i + 1]
            rv = nums[i + 1:]
            leftsum = self.helper(lv, memo)
            rightsum = self.helper(rv, memo)
            cursum = leftsum + rightsum + max(lv) * max(rv)
            minsum = min(minsum, cursum)
        
        memo[nums] = minsum
        return minsum
