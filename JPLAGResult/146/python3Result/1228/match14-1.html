from functools import lru_cache

class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        @lru_cache(None)
        def max_leaf(start, end):
            return max(arr[start:end])
        
        @lru_cache(None)
        def solve(start, end):
            if start + 1 >= end:
                return 0
            res = 2**31
            
for split in range(start+1, end):
                res = min(res, solve(start, split) + solve(split, end) + max_leaf(start, split)*max_leaf(split, end))
            return res
        
        return solve(0, len(arr))
