class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        def dfs(left,right):
            if left>=right:
                return 0
            if (left,right) in dp:
                return dp[(left,right)]
            res=float("inf")
            for mid in range(left,right):
                res=min(res,dfs(left,mid)+dfs(mid+1,right)+max(arr[left:mid+1])*max(arr[mid+1:right+1]))
            dp[(left,right)]=res
            return res
        
        dp={}
        return dfs(0,len(arr)-1)
