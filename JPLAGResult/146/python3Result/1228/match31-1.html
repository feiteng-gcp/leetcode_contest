class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        cache = {}
        max_cache = {}
        def solve(l, r):
            state = (l, r)
            if state in cache:
                return cache[state]
            if r - l == 1:
                return arr[l]
            best 
= float('inf')
            for i in range(l+1, r):
                left = solve(l, i)
                max_left = find_max(l, i)
                right = solve(i, r)
                max_right = find_max(i, r)
                best = min(best, left + right + max_left*max_right)
            cache[state] = best
            return best
        def find_max(l, r):
            state = (l, r)
            if state in max_cache:
                return max_cache[state]
            max_num = arr[l]
            for i in range(l, r):
                max_num = max(max_num, arr[i])
            max_cache[state] = max_num
            return max_num
        r = solve(0, len(arr)) 
        return r - sum(arr)
