class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        def t(i, j):
            if i == j:
                return 0, arr[i]
            if (i, j) not in h:
                s = float("inf")
                m = None
                for k in range(i, j):
                    lsum, lmax = t(i, k)
                    rsum, rmax = t(k + 1, j)
                    if lsum + rsum + lmax * rmax < s:
                        s = lsum + rsum + lmax * rmax
                        m = max(lmax, rmax)
                h[(i, j)] = s, m
                    
            return h[(i, j)]
        h = {}
        return t(0, len(arr) - 1)[0]
