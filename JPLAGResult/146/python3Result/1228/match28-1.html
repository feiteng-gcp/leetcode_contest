class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        N = len(arr)
        dp = {}
        def dfs(i, j):
            if i == j: return 0, arr[i]
            if (i, j) in dp: return dp[i, j]
            curr = float('inf')
            maximum = float('-inf')
            
for k in range(i, j):
                left, left_max = dfs(i, k)
                right, right_max = dfs(k+1, j)
                curr = min(curr, left + right + left_max * right_max)
                maximum = max(maximum, left_max, right_max)
            dp[i, j] = (curr, maximum)
            return curr, maximum
        
        return dfs(0, N-1)[0]
