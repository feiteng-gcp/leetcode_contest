class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        n = len(arr)
        dp = [[[0, 0] for _ in range(n)] for __ in range(n)]
        
        for i, a in enumerate(arr):
            dp[i][i][1] = a
            
        for l in range(1, n):
            for start in range(n-l):
                end = start + l
                dp[start][end][1] = max(dp
[start][end-1][1], arr[end])
                dp[start][end][0] = min(dp[start][split][0] + dp[split+1][end][0] + dp[start][split][1]*dp[split+1][end][1] for split in range(start, end))
        
        return dp[0][-1][0]
