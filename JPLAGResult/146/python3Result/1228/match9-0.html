class Solution:
    def mctFromLeafValues(self, arr: List[int]) -> int:
        n = len(arr)
        dp_res = [[0 for i in range(n)] for i in range(n)]
        dp_max = [[0 for i in range(n)] for i in range(n)]
        for i in range(n):
            dp_max[i][i] = arr[i]
        for step in range(1,n):
            for i in range(n-step):
                flag = 0
                for j in range(i, i+step):
                    if flag ==0:
                        dp_res[i][i+step] = dp_res[i][j] + dp_res[j+1][i+step] +dp_max[i][j]*dp_max[j+1][i+step]
                        #print(dp_res[i][i+step])
                        dp_max[i][i+step] = max(dp_max[i][j], dp_max[j+1][i+step])
                        flag = 1
                    elif flag ==1: 
                        dp_res[i][i+step] = min(dp_res[i][i+step], dp_res[i][j] + dp_res[j+1][i+step] +dp_max[i][j]*dp_max[j+1][i+step])
                        #print(dp_res[i][i+step])
        #print(dp_res, dp_max)            
        return dp_res[0][n-1]
