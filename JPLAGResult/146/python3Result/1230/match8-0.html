class Solution:
    def maxAbsValExpr(self, arr1: List[int], arr2: List[int]) -> int:
        pp = []
        pm = []
        mp = []
        mm = []
        for i in range(len(arr1)):
            pp.append(arr1[i]+arr2[i]+i)
            pm.append(arr1[i]+arr2[i]-i)
            mp.append(arr1[i]-arr2[i]+i)
            mm.append(arr1[i]-arr2[i]-i)
        result = max(max(pp)-min(pp), max(pm)-min(pm), max(mp)-min(mp),max(mm)-min(mm))
        return result
