class Solution:
    def maxAbsValExpr(self, arr1: List[int], arr2: List[int]) -> int:
        n = len(arr1)
        nums1, nums2, nums3, nums4 = [], [], [], []
        for i in range
(n):
            nums1.append(arr1[i] + arr2[i])
            nums2.append(arr1[i] - arr2[i])
            nums3.append(-arr1[i] + arr2[i])
            nums4.append(-arr1[i] - arr2[i])
        return max(self.helper(nums1), self.helper(nums2), self.helper(nums3), self.helper(nums4))
    
    def helper(self, nums):
        minEle = maxEle = nums[0]
        for i in range(1, len(nums)):
            if nums[i] + i < minEle:
                minEle = nums[i] + i
            if nums[i] + i > maxEle:
                maxEle = nums[i] + i
        return maxEle - minEle
