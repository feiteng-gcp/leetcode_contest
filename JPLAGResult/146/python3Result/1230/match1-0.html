class Solution:
    def maxAbsValExpr(self, arr1: List[int], arr2: List[int]) -> int:
        arra = [arr1[i] + arr2[i] - i for i in range(len(arr1))]
        c = max(arra) - min(arra)
        arrb = [arr1[i] - arr2[i] - i for i in range(len(arr1))]
        c = max(c, max(arrb) - min(arrb))
        arra = [arr1[i] + arr2[i] + i for i in range(len(arr1))]
        c = max(c, max(arra) - min(arra))
        arrb = [arr1[i] - arr2[i] + i for i in range(len(arr1))]
        c = max(c, max(arrb) - min(arrb))
        return c
