class Solution:
    def maxAbsValExpr(self, A: List[int], B: List[int]) -> int:
        k1, k2, k3, k4 = [-10000000]*4
        ans = 0
        for i in range(len(A)):
            if i > 0:
                ans = max(ans,   A[i] + B[i] + i + k1)
                ans = max(ans,   A[i] - B[i] + i + k2)
                ans = max(ans, - A[i] + B[i] + i + k3)
                ans = max(ans, - A[i] - B[i] + i + k4)                
            if k1 < -A[i]-B[i]-i:
                k1 = -A[i]-B[i]-i
            if k2 < -A[i]+B[i]-i:
                k2 = -A[i]+B[i]-i
            if k3 < A[i]-B[i]-i:
                k3 = A[i]-B[i]-i
            if k4 < A[i]+B[i]-i:
                k4 = A[i]+B[i]-i
        return ans
