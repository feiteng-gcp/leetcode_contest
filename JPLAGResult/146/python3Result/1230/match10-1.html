class Solution:
    def maxAbsValExpr(self, arr1: List[int], arr2: List[int]) -> int:
        z = list(zip(arr1, arr2))
        l = ([
a + b + i for i, (a, b) in enumerate(z)],
            [a + b - i for i, (a, b) in enumerate(z)],
            [a - b - i for i, (a, b) in enumerate(z)],
            [a - b + i for i, (a, b) in enumerate(z)])
        return max(max(i)-min(i) for i in l)
