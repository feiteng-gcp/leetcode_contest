class Solution:
    def maxAbsValExpr(self, arr1: List[int], arr2: List[int]) -> int:
        aux = [[] for _ in range(4)]
        for i, (a, b) in enumerate(zip(arr1, arr2)):
            aux[0].append(a + b + i)
            aux[1].append(a - b + i)
            aux[2].append(-a + b + i)
            aux[3].append(-a - b + i)
        
        ans, max_val, min_val = 0, [-float('inf')] * 4, [float('inf')] * 4
        for i, tup in enumerate(zip(*aux)):
            if i > 0:
                ans = max(
                    ans, 
                    max(v2 - v1 for v1, v2 in zip(tup, max_val)), 
                    max(v1 - v2 for v1, v2 in zip(tup, min_val))
                )
            max_val = [max(v1, v2) for v1, v2 in zip(tup, max_val)]
            min_val = [min(v1, v2) for v1, v2 in zip(tup, min_val)]
        return ans
