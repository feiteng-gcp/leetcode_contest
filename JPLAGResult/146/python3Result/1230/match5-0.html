class Solution:
    def maxAbsValExpr(self, arr1: List[int], arr2: List[int]) -> int:
        minv1=minv2=minv3=minv4=minv5=minv6=float("inf")
        maxv1=maxv2=maxv3=maxv4=maxv5=maxv6=float("-inf")
        for i in range(len(arr1)):
            minv1 = min(minv1, arr1[i]+arr2[i]+i)
            maxv1 = max(maxv1, arr1[i]+arr2[i]+i)
            minv2 = min(minv2, -arr1[i]+arr2[i]+i)
            maxv2 = max(maxv2, -arr1[i]+arr2[i]+i)
            minv3 = min(minv3, arr1[i]-arr2[i]+i)
            maxv3 = max(maxv3, arr1[i]-arr2[i]+i)
            minv4 = min(minv4, arr1[i]+arr2[i]-i)
            maxv4 = max(maxv4, arr1[i]+arr2[i]-i)
            minv5 = min(minv5, -arr1[i]+arr2[i]-i)
            maxv5 = max(maxv5, -arr1[i]+arr2[i]-i)
            minv6 = min(minv6, arr1[i]-arr2[i]-i)
            maxv6 = max(maxv6, arr1[i]-arr2[i]-i)
        return max(maxv1-minv1, maxv2-minv2, maxv3-minv3, maxv4-minv4, maxv5-minv5, maxv6-minv6)
