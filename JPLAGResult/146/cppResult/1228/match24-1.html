class Solution {
public:
    int mctFromLeafValues(vector<int>& a) {
        int n = a.size();
        vector<vector<int>> max_leaf(n, vector<int>(n, 0));
        
        for (int i = 0; i < n; i++) {
            max_leaf[i][i] = a[i];
            for (int j = i + 1; j < n; j++)
                max_leaf[i][j] = max(max_leaf[i][j - 1], a[j]);

        }
        
        vector<vector<int>> cost(n, vector<int>(n, -1));
        
        for (int i = 0; i < n; i++)
            cost[i][i] = 0;
            
        for (int i = n - 1; i >= 0; i--)
            for (int j = i + 1; j < n; j++)
                for (int k = i; k < j; k++) {
                    int candidate = cost[i][k] + cost[k + 1][j] + max_leaf[i][k] * max_leaf[k + 1][j];
                    if (cost[i][j] == -1 || candidate < cost[i][j])
                        cost[i][j] = candidate;
                }
        
        return cost[0][n - 1];
    }
};
