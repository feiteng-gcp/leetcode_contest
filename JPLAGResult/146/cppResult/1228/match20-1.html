class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        vector<vector<int>> dp, dpm;
        dp.resize( arr.size());
        dpm.resize(arr.size());
        for(int i=0; i<arr.size(); ++i){ 
            dp[ i].resize(arr.size()-i);
            dpm[i].resize(arr.size()-i);
        }
        
        for(int i=0; i<arr.size(); ++i){
            dp[ i][0]= 0;
            dpm[i][0]= arr[i];
        }
        

        for(int i=0; i<arr.size(); ++i){
            for(int j=1; i+j<arr.size(); ++j){
                dpm[i][j]= max(dpm[i][j-1], arr[i+j]); 
            }

        }
        
        for(int j=1; j<arr.size(); ++j){
            for(int i=0; i+j<arr.size(); ++i){
                dp[i][j]= 1e9;
                for(int k=1; k<=j; ++k){
                    dp[i][j] = min( dp[i][j], dp[i][k-1] + dp[i+k][j-k] + dpm[i][k-1]*dpm[i+k][j-k] );
                }
            }
        }
        
            
        return dp[0][arr.size()-1];
    }
};
