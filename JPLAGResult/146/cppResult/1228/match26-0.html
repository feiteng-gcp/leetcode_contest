class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        vector<vector<int>> dp(n + 1, vector<int> (n + 1, 0x3f3f3f3f));
        for (int i = 0; i < n; ++i) {
            dp[i][i] = 0;
        }
        
        for (int len = 2; len <= n; ++len) {
            for (int i = 0; i + len - 1 < n; ++i) {
                int j = i + len - 1;
                for (int k = 1; len - k >= 1; ++k) {
                    // if (k % 2 == 1 && len - k % 2 == 1) {
                    int m1 = 0, m2 = 0;
                    for (int u = i; u <= i + k - 1; ++u) m1 = max(m1, arr[u]);
                    for (int u = i + k; u <= j; ++u) m2 = max(m2, arr[u]);
                    //cout << i << " " << k << " " << j <<  endl;
                    //cout << m1 << " " << m2 << endl;
                    dp[i][j] = min(dp[i][j], dp[i][i + k - 1] + dp[i + k][j] + m1 * m2);
                    // }
                }
            } 
        }
        return dp[0][n - 1];
    }
};
