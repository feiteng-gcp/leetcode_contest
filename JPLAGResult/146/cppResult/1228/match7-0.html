class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
		int n = arr.size();
		vector<vector<int>> f(n, vector<int>(n, 1 << 30));
		vector<vector<int>> maxleaf(n, vector<int>(n));
		for (int i = 0; i < n; i++)
		{
			f[i][i] = 0;
			maxleaf[i][i] = arr[i];
		}
		for (int l = 2; l <= n; l++)
			for (int i = 0; i < n - l + 1; i++)
			{
				int j = i + l - 1;
				for (int k = i; k < j; k++)
				{
					f[i][j] = min(f[i][j], f[i][k] + f[k + 1][j] + maxleaf[i][k] * maxleaf[k + 1][j]);
				}
				maxleaf[i][j] = max(arr[i], maxleaf[i + 1][j]);
			}
		return f[0][n - 1];
    }
};
