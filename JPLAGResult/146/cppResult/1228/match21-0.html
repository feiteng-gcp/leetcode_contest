class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int max[41][41];
        int res[41][41];
        int le = arr.size();
        int t;
        for(int i = 0; i < arr.size();i++){
            for(int j = i; j < le; j++){
                if(i==j){
                    max[i][j] = arr[i];
                    
                }else if(arr[j] >= max[i][j-1]){
                    max[i][j] = arr[j];
                }else{
                    max[i][j] = max[i][j-1];
                }
            }
        }
        
        for(int i = 0; i < le;i++){
            for(int j = i+1; j < le; j++){
                res[i][j] = 99999999;
            }
        }
        for(int i = 0; i < le;i++){
            res[i][i] = 0;
        }
        int j;
        for(int l = 1; l < le; l++){
            for(int i = 0; i+l < le;i++){
                j = i+l;
                for(int k = i; k < j; k++){
                    t = res[i][k]+res[k+1][j]+max[i][k]*max[k+1][j];
                    if(t < res[i][j])
                        res[i][j] = t;
                }
            }
        }
        return res[0][le-1];
    }
};
