class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int arr_len = arr.size();
        vector<vector<int>> sm(arr_len, vector<int>(arr_len, 0));
        for (int i = 0; i < arr_len; i++) {
            int min1 = arr[i], min2 = INT_MIN;
            for (int j = i + 1; j < arr_len; j++) {
                if (arr[j] > min1) {
                    min2 = min1;
                    min1 = arr[j];
                } else if (arr[j] > min2) min2 = arr[j];
                sm[i][j] = min1 * min2;
            }
        }
        vector<vector<int>> dp(arr_len, vector<int>(arr_len, INT_MAX));
        for (int i = 0; i < arr_len; i++) dp[i][i] = 0;
        for (int d = 1; d < arr_len; d++)
            for (int i = 0; i + d < arr_len; i++)
                for (int j = i; j < i + d; j++)
                    dp[i][i + d] = min(dp[i][i + d], dp[i][j] + dp[j + 1][i + d] + sm[i][i + d]);
        return dp[0][arr_len - 1];
    }
};
