class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n=arr.size();
        vector<vector<int>> ans(n,vector<int>(n,INT_MAX));
        vector<vector<int>> mm(n,vector<int>(n,0));

        for(int i=0;i<n;i++){
            int t=arr[i];
            for(int j=i;j<n;j++){
                t=max(arr[j],t);
                mm[i][j]=t;
            }
        }
        for(int i=0;i<n;i++){
            for(int j=0;j+i<n;j++){
                if(i==0) ans[j][j+i]=0;
                else{
                    for(int k=j;k<j+i;k++){
                        ans[j][j+i]=min(ans[j][j+i],mm[j][k]*mm[k+1][j+i]+ans[j][k]+ans[k+1][j+i]);
                    }
                }
            }
        }
        return ans[0][n-1];
    }
};
