class Solution {
public:
	int mctFromLeafValues(vector<int>& arr) {


		int n = arr.size();

		if (n == 1)
		{
			return arr[0];
		}
		vector<vector<int>> dp(n, vector<int>(n, -1));

		int ans = dfs(arr, 0, n - 1, dp);
		return ans;

	}
	int dfs(vector<int>& arr, int l, int r, vector<vector<int>> &dp)
	{
		if (l >= r)
			return 0;
		if (l + 1 == r)
		{
			return arr[l] * arr[r];
		}

		if (dp[l][r] != -1) return dp[l][r];
		int result = INT_MAX;
		for (int k = l; k < r; k++)
		{
			result = min(result, dfs(arr, l, k, dp) + dfs(arr, k + 1, r, dp));
		}


		if (r - l + 1 >= 3)
		{
			vector<int> tmp;
			for (int i = l; i <= r; i++)
			{
				tmp.push_back(arr[i]);
			}
			sort(tmp.begin(), tmp.end(), [](int a, int b)
			{
				return a > b;
			});
			return dp[l][r] = result + tmp[0] * tmp[1];
		}
		else
			return dp[l][r] = result;
	}

};
