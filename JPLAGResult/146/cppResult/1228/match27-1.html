class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int f[50][50];
        memset(f, 0x3f, sizeof f);
        int len = arr.size();
        for (int i = 0; i < len; ++i) f[i][i] = 0;
        for (int i = 0; i+1 < len; ++i)f[i][i+1] = arr[i]*arr[i+1];
        

        for (int d = 2; d < len; ++d) {
            for (int i = 0; i+d < len; ++i) {
                int j = i+d;
                for (int k = i; k < j; ++k) {
                    int mx1 = arr[i], mx2 = arr[j];
                    for (int t = i+1; t <= k; ++t) mx1 = max(mx1, arr[t]);
                    for (int t = k+1; t < j; ++t) mx2 = max(mx2, arr[t]);
                    int tmp = f[i][k] + f[k+1][j] + mx1*mx2;
                    f[i][j] = min(f[i][j], tmp);
                }
            }
        }
        
        return f[0][len-1];
        
    }
};
