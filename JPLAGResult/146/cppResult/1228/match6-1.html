class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        long long dp[n][n];
        int minx[n][n];
        for(int i = 0;i < n;i++)

            minx[i][i] = arr[i];
        for(int size = 2;size <= n;size++){
            for(int i = 0;i <= n-size;i++){
                minx[i][i + size - 1] = max(minx[i][i + size - 2],arr[i + size -1]);
            }
        }
        for(int i = 0;i < n;i++){
            dp[i][i] = 0;
        }
        for(int size = 2;size <= n;size++){

            for(int i = 0;i <= n - size;i++){
                dp[i][i + size - 1] = dp[i][i] + dp[i+1][i + size-1] + minx[i][i] * minx[i+1][i+size-1];
                for(int j = i + 2;j <= i + size - 1;j++){
                    dp[i][i + size - 1] = min(dp[i][i+size-1],dp[i][j-1]  +  dp[j][i+size-1] + minx[i][j-1] * minx[j][i+size-1]);
                }
            }
        }
        return dp[0][n-1];
    }
};
