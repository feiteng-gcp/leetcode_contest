class Solution {
public:
    int dp[100][100];
    int cal(vector<int>& arr, int left, int right) {
        if (left == right) {
            return 0;
        }
        if (left > right) {
            return 0;
        }
        if (dp[left][right] != -1) {
            return dp[left][right];
        }
        int ret = 0x7fffffff;
        for (int i = left + 1;i <= right;i++) {
            int x = 0;
            int y = 0;
            for (int j = left;j <= right;j++) {
                if (j < i && x < arr[j]) {
                    x = arr[j];
                }
                if (j >= i && y < arr[j]) {
                    y = arr[j];
                }
            }
            int temp = cal(arr, left, i - 1) + cal(arr, i, right) + x * y;
            if (ret > temp) {
                ret = temp;
            }
        }
        return dp[left][right] = ret;
    }
    int mctFromLeafValues(vector<int>& arr) {
        memset(dp, -1, sizeof(dp));
        return cal(arr, 0, arr.size() - 1);
    }
};
