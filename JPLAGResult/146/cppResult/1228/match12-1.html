class Solution {
public:
  pair<int, int> cache[43][43];
  bool visited[43][43];

  pair<int, int> dfs(vector<int> &A, int L, int R) {
    if (R - L == 1) return {0, A[L]};
    if (R - L == 2) return {A[R - 1] * A[L], max(A[L], A[R-1])};
    if (visited[L][R]) return cache[L][R];
    visited[L][R] = true;

    int ans = 0x3f3f3f3f;
    int ansv = 0;
    for (int i = L + 1; i < R; i++) {
      auto [lk, lv] = dfs(A, L, i); 
      auto [rk, rv] = dfs(A, i, R);
      ans = min(ans, lk + rk + lv * rv);
      ansv = max(lv, rv);
    }
    return cache[L][R] = {ans, ansv};
  }
  int mctFromLeafValues(vector<int>& arr) {
    memset(visited, 0, sizeof(visited));
    return dfs(arr, 0, arr.size()).first;
  }
};
