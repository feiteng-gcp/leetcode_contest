class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();

        int INF = 2147483647;
        vector<vector<int>> dp(n, vector<int>(n, INF));
        for(int i=0; i<n; ++i) {
            dp[i][i] = 0;
        }
        for(int len=1; len<n; ++len) {
            for(int i=0, j=i+len; j<n; ++i, ++j) {

                for(int mid=i; mid<j; ++mid) {
                    int lmax = -1, rmax = -1;
                    for(int k=i; k<=mid; ++k) lmax = std::max(lmax, arr[k]);
                    for(int k=mid+1; k<=j; ++k) rmax = std::max(rmax, arr[k]);
                    dp[i][j] = std::min(dp[i][j], dp[i][mid]+dp[mid+1][j]+lmax*rmax);
                }
                //printf("%d,%d: %d\n", i, j, dp[i][j]);
            }
        }
        return dp[0][n-1];
    }
};
