class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
      int n = arr.size();
      vector<vector<int>> max_num(n, vector<int>(n, 0));
      vector<vector<int>> ret(n, vector<int>(n, 0));
      for (int i = 0; i < n; ++i) {
        max_num[i][i] = arr[i];
        ret[i][i] = 0;
      }
      

      for(int i = 0; i < n; ++i) {
        int tmp_max = arr[i];
        for(int j = i; j < n; ++j) {
          tmp_max = max(tmp_max, arr[j]);
          max_num[i][j] = tmp_max;
        }
      }
      
      
      for (int d = 1; d < n; ++d) {
        for(int i = 0; i < n; ++i) {
          if(i + d >= n)
            break;
          int min_num = 1 << 30;
          int pos = -1;
          for(int j = i; j < i + d; ++j) {
            int tmp = ret[i][j] + ret[j + 1][i + d] + max_num[i][j] * max_num[j + 1][i + d];
            if(tmp < min_num) {
              min_num = tmp;
              pos = j;
            }
          }
          
          ret[i][i + d] = min_num;
          //cout << "i = " << i << " d = " << d << " ret = " << min_num << endl;
        }
      }
      
      return ret[0][n - 1];
    }
};
