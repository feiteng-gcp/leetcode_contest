class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        vector<vector<pair<int,int>>> dp(n, vector<pair<int, int>>(n + 1));
        for (int s = 0; s < n; ++s) {
            dp[s][s+1] = make_pair(arr[s], 0);            
        }
        
        for (int l = 2; l <= n; ++l) {

            for (int s = 0; s + l < n + 1; ++s) {
                int sum = INT_MAX;
                for (int k = 1; k < l; ++k) {
                    sum = min(sum, dp[s][s+k].second + dp[s+k][s+l].second + dp[s][s+k].first * dp[s+k][s+l].first);
                }
                
                dp[s][s+l].first = max(dp[s][s+1].first, dp[s+1][s+l].first);
                dp[s][s+l].second = sum;
            }
        }
        
        return dp[0][n].second;
    }
};
