class Solution {
public:

    int f[45][45];
    int m[45][45];
    
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        for(int i = 0; i < arr.size(); i++) {
            m[i][i] = arr[i];
            for(int j  = i+1;  j < n; j++) {
                m[i][j] = max(m[i][j-1], arr[j]);
            }
        }
        int maxint = 0x7fffffff;
        
        for(int i = 0; i < n; i++) {
            for(int j  = 0; j < n; j++) {
                f[i][j] = 0x7fffffff;
            }
        }
        
        for(int i = 0; i < n; i++) {
            f[i][i] = 0;
        }
        
        for(int i = 0; i < n-1; i++) {
            f[i][i+1] = arr[i] * arr[i+1];
        }
        

        for(int l = 3; l <= n; l++) {
            for(int i = 0; i +l-1<n;i++) {
                for(int j = i+1; j < i+l; j++) {
                    int tmp = f[i][j-1] + f[j][i+l-1] + m[i][j-1] * m[j][i+l-1];
                    f[i][i+l-1] = min(f[i][i+l-1], tmp);
                }
            }
        }
        
        return f[0][n-1];
    }
};
