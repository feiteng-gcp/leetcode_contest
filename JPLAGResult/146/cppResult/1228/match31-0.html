class Solution {
private:
    int find(vector<int>& arr, int left, int right) {
        int res = INT_MIN;
        for(int i = left; i <= right; i++) {
            res = max(res, arr[i]);
        }
        return res;
    }
public:
    int mctFromLeafValues(vector<int>& arr) {
        int size = arr.size();
        vector<vector<int>> dp(size, vector<int>(size, INT_MAX));
        for(int i = 0; i < size; i++) dp[i][i] = 0;
        for(int i = 1; i <= size; i++) {
            for(int j = 0; j + i < size; j++) {
                for(int k = j; k < i + j; k++) {
                    int pro = find(arr, j, k) * find(arr, k + 1, i + j);
                    dp[j][j + i] = min(dp[j][j + i], dp[j][k] + dp[k + 1][i + j] + pro);
                }
            }
        }
        int res = dp[0][size - 1];
        return res;
    }
};
