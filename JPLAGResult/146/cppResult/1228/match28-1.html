class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {

        const int n = arr.size();
        vector<vector<int>> max(n, vector<int>(n, 0));
        for (int i = 0; i < n; ++i) {
            for (int j = i; j < n; ++j) {
                if (arr[j] > max[i][j]) max[i][j] = arr[j];
                if (i < j && max[i][j - 1] > max[i][j]) max[i][j] = max[i][j - 1];
            }
        }
        vector<vector<int>> opt(n, vector<int>(n, numeric_limits<int>::max()));

        for (int i = 0; i < n; ++i) opt[i][i] = 0;
        for (int len = 1; len < n; ++len) {
            for (int i = 0; i + len < n; ++i) {
                for (int j = i; j + 1 <= i + len; ++j) {
                    const int cur = opt[i][j] + opt[j + 1][i + len] + max[i][j] * max[j + 1][i + len];
                    if (cur < opt[i][i + len]) opt[i][i + len] = cur;
                }
            }
        }
        return opt[0][n - 1];
    }
};
