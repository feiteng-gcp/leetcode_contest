class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n=arr.size();
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX));
        vector<vector<int>> max_dp(n, vector<int>(n, INT_MIN));
        for (int i=0;i<n;i++){
            dp[i][i]=0;
            max_dp[i][i]=arr[i];
        }
        for (int j=1;j<n;j++){
            for(int i=j-1;i>=0;i--){
                max_dp[i][j]=max(max_dp[i][i], max_dp[i+1][j]);
                for (int k=i;k<j;k++){
                    dp[i][j]=min(dp[i][j], dp[i][k]+dp[k+1][j]+max_dp[i][k]*max_dp[k+1][j]);
                }
            }
        }
        return dp[0][n-1];
    }
};
