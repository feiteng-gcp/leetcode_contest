class Solution {
private:
    vector<vector<pair<int,int>>> visited;
public:
    int mctFromLeafValues(vector<int>& arr) {
        visited = vector<vector<pair<int,int>>>(arr.size(), vector<pair<int,int>>(arr.size(), {-1,-1}));
        return helper(arr, 0, arr.size() - 1).first - accumulate(arr.begin(), arr.end(), 0);
    }
    
    pair<int,int> helper(vector<int>& arr, int start, int end)// sum, maxNum
    {
        if(start >= end) return {arr[start], arr[start]};
        if(start + 1 == end) return {arr[start] * arr[end] + arr[start] + arr[end], max(arr[start] , arr[end])};
        if(visited[start][end] != make_pair(-1,-1)) return visited[start][end];
        int sum = INT_MAX;
        int maxNumber = -1;
        for(int i = start; i + 1 <= end; ++i)
        {
            auto left = helper(arr, start, i);
            auto right = helper(arr, i + 1, end);
            maxNumber = max(left.second, right.second);
            sum = min(sum, left.first + right.first + left.second * right.second);
        }
        return visited[start][end] = {sum, maxNumber};
    }
};
