class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        vector<vector<int>> mxval(n, vector<int>(n, 0));
        
        for(int i=0;i<n;i++) {
            int mx = arr[i];
            for(int j=i;j<n;j++) {
                mx = max(mx, arr[j]);

                mxval[i][j] = mxval[j][i] = mx;
            }
        }
        
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX));
        for(int i=0;i<n;i++) dp[i][i] = 0;
        for(int len=2;len<=n;len++) for(int start=0;start<=n-len;start++) {
            int end = start + len - 1;
            for(int mid = start;mid<end;mid++) {
                assert(dp[start][mid] < INT_MAX && dp[mid+1][end] < INT_MAX);
                dp[start][end] = min(dp[start][end], dp[start][mid] + dp[mid+1][end] + mxval[start][mid]*mxval[mid+1][end]);
            }
        }
        return dp[0][n-1];
    }
};
