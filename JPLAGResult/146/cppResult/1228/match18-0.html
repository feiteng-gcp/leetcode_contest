class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX)), maxLeaf(n, vector<int>(n, 0));
        for(int l = 1; l <= n; l++){
            for(int i = 0; i + l - 1 < n; i++){
                int j = i + l - 1;
                if(l == 1){
                    //cout<<"len = 1 "<<i<<" "<<j<<endl;
                    dp[i][j] = 0;
                    maxLeaf[i][j] = arr[i];
                    continue;
                }
                for(int k = i; k < j; k++){
                    //cout<<i<<" "<<k<<" "<<j<<endl;
                    //cout<<dp[i][k]<<endl;
                    //cout<<dp[k+1][j]<<endl;
                    //cout<<maxLeaf[i][k]<<endl;
                    //cout<<maxLeaf[k+1][j]<<endl;
                    dp[i][j] = min(dp[i][j], dp[i][k] + dp[k+1][j] + maxLeaf[i][k] * maxLeaf[k+1][j]);
                    maxLeaf[i][j] = max(maxLeaf[i][k], maxLeaf[k+1][j]);
                }
            }
        }
        return dp[0][n-1];
        
    }
};
