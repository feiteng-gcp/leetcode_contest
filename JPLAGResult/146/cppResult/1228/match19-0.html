int maxm[41][41],dp[41][41];
class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n=arr.size();
        for(int i=0;i<n;i++)
        {
            maxm[i][i]=arr[i];
            for(int j=i+1;j<n;j++)
            {
                maxm[i][j]=max(maxm[i][j-1],arr[j]);
            }
        }
        memset(dp,0,sizeof(dp));
        for(int l=1;l<n;l++)
        {
            for(int i=0;i<n-l;i++)
            {
                dp[i][i+l]=2147483647;
                for(int j=i+1;j<=i+l;j++)
                {
                    dp[i][i+l]=min(dp[i][i+l],dp[i][j-1]+dp[j][i+l]+maxm[i][j-1]*maxm[j][i+l]);
                }
            }
        }
        return dp[0][n-1];
    }
};
