class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        vector<vector<int>> dp(n, vector<int>(n, -1));
        vector<vector<int>> ml(n, vector<int>(n, -1));
        for (int l = 1; l <= n; l++) {
            for (int i = 0; i + l <= n; i++) {
                int j = i+l-1;
                if (l == 1) {
                    dp[i][j] = 0;
                    ml[i][j] = arr[i];
                } else {
                    ml[i][j] = max(ml[i][j-1], arr[j]);
                    for (int k = i; k < j; k++) {
                        int t = ml[i][k] * ml[k+1][j] + dp[i][k] + dp[k+1][j];
                        if (dp[i][j] == -1 || dp[i][j] > t) {
                            dp[i][j] = t;
                        }
                    }
                }
                //cout << "dp[" << i << "][" << j << "]=" << dp[i][j] << endl;
                //cout << "ml[" << i << "][" << j << "]=" << ml[i][j] << endl;
            }
        }
        return (dp[0][n-1]);
    }
};
