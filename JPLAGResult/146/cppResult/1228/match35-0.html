class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int mx[41][41];
        for(int i=0;i<arr.size();i++){
            mx[i][i]=arr[i];
            for(int j=i+1;j<arr.size();j++) mx[i][j]=max(mx[i][j-1],arr[j]);
        }
        int dp[41][41];
        memset(dp,0x3f,sizeof(dp));
        for(int i=0;i<arr.size();i++) dp[i][i]=0;
        for(int len=2;len<=arr.size();len++){
            for(int i=0;i<=arr.size()-len;i++){
                int j=i+len-1;
                for(int k=i;k<j;k++){
                    dp[i][j]=min(dp[i][k]+dp[k+1][j]+mx[i][k]*mx[k+1][j],dp[i][j]);
                }
            }
        }
        return dp[0][arr.size()-1];
    }
};
