#define maxn 44
typedef long long ll;
ll dp[maxn][maxn];
ll p[maxn][maxn];
int mx[maxn][maxn];
int a[maxn];
int n;

class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        memset(mx,0,sizeof(mx));
        memset(dp,0,sizeof(dp));
        memset(a,0,sizeof(a));
        memset(p,0,sizeof(p));
        n=arr.size();
        
        for(int i=1;i<=n;i++) a[i]=arr[i-1];
        
        for(int i=0;i<=n;i++)
        {
            for(int j=0;j<=n;j++)
            {
                dp[i][j]=p[i][j]=2147483647;
            }
        }
        
        for(int i=1;i<=n;i++)
        {
            p[i][i]=a[i];
            for(int j=i+1;j<=n;j++)
            {
                p[i][j]=p[i][j-1]*a[j];
                p[i][j]=min(p[i][j],2147483647ll);
            }
        }
        
        for(int i=1;i<=n;i++)
        {

            for(int j=i;j<=n;j++)
            {
                mx[i][j]=a[i];
                for(int k=i;k<=j;k++) mx[i][j]=max(mx[i][j],a[k]);
            }
        }
        
        for(int i=1;i<n;i++) dp[i][i+1]=a[i]*a[i+1];

        for(int i=1;i<=n;i++) dp[i][i]=0;
        
        for(int l=3;l<=n;l++)
        {
            for(int s=1;s<=n;s++)
            {
                int e=s+l-1;
                if (e>n) continue;
                for(int m=s;m<e;m++)
                {
                    dp[s][e]=min(dp[s][e],dp[s][m]+dp[m+1][e]+mx[s][m]*mx[m+1][e]);
                }
            }
        }
        
        int ret=(int)(dp[1][n]);
        return ret;
    }
};
