class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        vector<vector<long long int>> result(arr.size(), vector<long long int>(arr.size(), INT_MAX));
        vector<vector<long long int>> max_value(arr.size(), vector<long long int>(arr.size(), 0));

        for (int i = 0; i < arr.size(); i++) {
            max_value[i][i] = arr[i];
            result[i][i] = 0;
        }
        for (int len = 2; len <= arr.size(); len++) {
            for (int i = 0; i + len - 1 < arr.size(); i++) {
                max_value[i][i + len - 1] = max(max_value[i][i + len - 2], (long long int)arr[i + len - 1]);

                for (int first_right = i + 1; first_right < i + len; first_right++) {
                    result[i][i + len - 1] = min(result[i][i + len - 1], result[i][first_right - 1] + result[first_right][i + len - 1] + max_value[i][first_right - 1] * max_value[first_right][i + len - 1]);
                }
            }
        }
        return result[0][arr.size() - 1];
    }
};
