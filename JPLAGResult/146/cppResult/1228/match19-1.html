class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int N = arr.size();
        vector<vector<int>> maxv(N, vector<int>(N, 0)), dp(N, vector<int>(N));

        for (int i=0; i<N; i++) {
            maxv[i][i] = arr[i];
            for (int j=i+1; j<N; j++) {
                maxv[i][j] = max(maxv[i][j-1], arr[j]);
            }
        }
        int tmp;

        for (int i=N-1; i>-1; i--) {
            dp[i][i] = 0;
            for (int j=i+1; j<N; j++) {
                dp[i][j] = dp[i][j-1] + maxv[j][j] * maxv[i][j-1];
                for (int k=i; k<j; k++) {
                    tmp = dp[i][k] + dp[k+1][j] + maxv[i][k] * maxv[k+1][j];
                    dp[i][j] = min(dp[i][j], tmp);
                }
            }
        }
        return dp[0][N-1];
    }
};
