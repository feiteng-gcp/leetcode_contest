class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        unordered_map<string, int> pairs;
        for(auto &d: dominoes){
            sort(d.begin(), d.end());
            string key = to_string(d[0]) + "|" + to_string(d[1]);
            //cout<<key<<endl;
            pairs[key]++;
        }
        int res = 0;
        for(auto &x: pairs){
            if(x.second < 2) continue;
            res += x.second * (x.second - 1) / 2;
        }
        return res;
    }
};
