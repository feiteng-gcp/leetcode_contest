class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        map<pair<int, int>, int> mp;
        for (auto v: dominoes) {
            mp[{min(v[0], v[1]), max(v[0], v[1])}]++;
        }
        int ans = 0;
        for (auto kv: mp) {
            int cnt = kv.second;
            ans += (cnt * (cnt-1)) / 2;
        }
        return ans;
    }
};
