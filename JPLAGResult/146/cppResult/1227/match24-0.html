class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& a) {
        int cnt[10][10];
        memset(cnt, 0, sizeof(cnt));
        
        int n = a.size(), ans = 0;
        for (int i = 0; i < n; ++i) {
            int x = min(a[i][0], a[i][1]);
            int y = max(a[i][0], a[i][1]);
            
            ++cnt[x][y];
        }
        
        for (int i = 1; i <= 9; ++i)
            for (int j = i; j <= 9; ++j)
                ans += cnt[i][j] * (cnt[i][j] - 1) / 2;
        
        return ans;
    }
};
