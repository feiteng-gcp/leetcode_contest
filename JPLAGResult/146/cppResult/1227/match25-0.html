class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        map<pair<int,int>, int>pairCount;
        for(auto p : dominoes){
            int v1=p[0], v2=p[1];
            if(v1<=v2){
                pairCount[{v1, v2}]++;
            }else{
                pairCount[{v2, v1}]++;
            }
        }
        int res=0;
        for(auto p : pairCount){
            int count=p.second;
            //if(count==1) continue;
            res+=count*(count-1)/2 ;
        }
        return res;
    }
};
