class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        unordered_map<int, int> M1, M2;

        int res = 0;
        for (auto& v : dominoes) {
            int key1 = v[0] * 10 + v[1];
            res += M1[key1];

            if (v[0] != v[1]) {
                int key2 = v[1] * 10 + v[0];
                res += M1[key2];
            }
            ++M1[key1];
        }

        return res;
    }
};
