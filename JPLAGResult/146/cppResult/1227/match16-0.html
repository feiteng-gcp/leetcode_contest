class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        map<vector<int>, int> m;
        for(auto domi: dominoes) {
            sort(domi.begin(), domi.end());
            m[domi]++;
        }
        int res = 0;
        for(auto it: m) {
            res += it.second*(it.second-1)/2;
        }
        return res;
    }
};
