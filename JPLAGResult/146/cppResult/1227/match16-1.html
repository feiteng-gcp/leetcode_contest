class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {

        map<pair<int, int>, int> mp;
        for (auto domino:dominoes) {
            mp[make_pair(min(domino[0], domino[1]), max(domino[0], domino[1]))]++;
        }
        int res = 0;
        for (auto pr:mp) {
            res += pr.second * (pr.second - 1) / 2;
        }
        return res;
    }
};
