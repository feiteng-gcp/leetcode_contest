class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        map<int, int> cnt;
        int ans = 0;
        for (auto &dominoe : dominoes) {
            if (dominoe[0] > dominoe[1]) {
                swap(dominoe[0], dominoe[1]);
            }
            int num = dominoe[0] * 10 + dominoe[1];
            ans += cnt[num];
            ++cnt[num];
        }
        return ans;
    }
};
