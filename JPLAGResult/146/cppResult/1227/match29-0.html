class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        unordered_map<int, unordered_map<int, int>> hash;
        int res = 0;
        for(int i=0; i<dominoes.size(); i++) {
            int a = dominoes[i][0];
            int b = dominoes[i][1];
            if(a > b) {
                int t = a;
                a = b;
                b = t;
            }
            if(hash.find(a)!=hash.end() && hash[a][b]!=0) {
                res += hash[a][b];
            }
            hash[a][b] += 1;
        }
        return res;
    }
};
