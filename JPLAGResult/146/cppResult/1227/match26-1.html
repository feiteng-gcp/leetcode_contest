class Solution {

    map<vector<int>, int> dict;
    public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {

        int nums = 0;
        for(auto x:dominoes) {
            sort(x.begin(), x.end());
            dict[x]+=1;
        }
        for(auto & x:dict) {
            nums+=x.second*(x.second-1)/2;
        }
        return nums;
    }
};
