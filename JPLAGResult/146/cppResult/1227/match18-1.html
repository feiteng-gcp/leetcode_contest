class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        map<vector<int>, int> M;
        int rtn = 0;
        for (auto &d : dominoes) {
            sort(d.begin(), d.end());
            ++M[d];
        }
        
        for (auto &p : M) {
            rtn += p.second * (p.second-1) / 2;
        }
        
        return rtn;
    }
};
