class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        map<vector<int>, int>s;
        int ans = 0;
        for (int i = 0; i < dominoes.size(); ++i) {
            sort(dominoes[i].begin(), dominoes[i].end());
            ans += s[dominoes[i]];
            s[dominoes[i]]++;
        
        }
        return ans;
    }
};
