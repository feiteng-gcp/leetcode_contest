class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        unordered_map<int, int> cnt;
        for (int i = 0; i < dominoes.size(); ++i) {
            sort(dominoes[i].begin(), dominoes[i].end());
        }
        for (int i = 0; i < dominoes.size(); ++i) {
            int x = dominoes[i][0] * 10 + dominoes[i][1];
            cnt[x] += 1;
        }
        int ans = 0;
        for (auto kv : cnt) {
            int n = kv.second;
            ans += n * (n - 1) / 2;
        }
        return ans;
    }
};
