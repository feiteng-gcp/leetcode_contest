class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        map<int, set<int>> red;
        map<int, set<int>> blue;
        for(const auto& r: red_edges) {
            red[r[0]].insert(r[1]);
        }
        for(const auto& r: blue_edges) {
            blue[r[0]].insert(r[1]);
        }
        map<int, int> ret;
        ret[0] = 0;
        queue<pair<int, int>> q;
        queue<int> dep;
        q.push(make_pair(0, 0)); dep.push(0);
        q.push(make_pair(0, 1)); dep.push(0);
        set<pair<int, int>> vis;
        vis.insert(make_pair(0,0));
        vis.insert(make_pair(0,1));
        while(q.size()) {
            int node = q.front().first;
            int color = q.front().second;
            q.pop();
            int d = dep.front(); dep.pop();
            if(ret.count(node) == 0) {
                ret[node] = d;
            }
            if(color == 0) {
                for(int next: blue[node]) {
                    if(vis.count(make_pair(next,1))) {
                        continue;
                    }
                    q.push(make_pair(next, 1)); dep.push(d+1); vis.insert(make_pair(next,1));
                }
            }
            else {
                for(int next: red[node]) {
                    if(vis.count(make_pair(next,0))) {
                        continue;
                    }
                    q.push(make_pair(next, 0)); dep.push(d+1);vis.insert(make_pair(next,0));

                }
            }
        }
        vector<int> r(n);
        for(int i = 0; i < n;i++) {
            if(ret.count(i)) {
                r[i] = ret[i];
            }
            else {
                r[i] = -1;
            }
        }
        return r;
    }
};
