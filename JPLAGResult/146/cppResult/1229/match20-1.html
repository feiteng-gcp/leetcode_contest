class Solution {
public:
    
    void bfs(int last,vector<int>&ans,vector<vector<int> >r,vector<vector<int> >b,int n){
        queue<pair<int,int> >Q;
        Q.push(make_pair(0,last));
        int visited[n+1][2];
        memset(visited,-1,sizeof(visited));        
        visited[0][last]=0;
        
        
        while(!Q.empty()){
            pair<int,int>P=Q.front();
            Q.pop();
            ans[P.first]=min(ans[P.first],visited[P.first][P.second]);
            
            if(P.second==0){
                for(int i=0;i<b.size();i++){
                    if(b[i][0]==P.first){
                        if(visited[b[i][1]][P.second^1] ==-1){
                            visited[b[i][1]][P.second^1]=visited[P.first][P.second]+1;
                            Q.push(make_pair(b[i][1],1));
                        }   

                    }
                }
            }else{
                for(int i=0;i<r.size();i++){
                    if(r[i][0]==P.first){
                        if(visited[r[i][1]][P.second^1] ==-1){
                            visited[r[i][1]][P.second^1] =visited[P.first][P.second]+1;
                            Q.push(make_pair(r[i][1],0));
                        }   
                    }
                }
                
            }
            
            
        }
        
    }
    
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {

        vector<int>ans=vector<int>(n,1<<20);
        
        bfs(0,ans,red_edges,blue_edges,n);
        bfs(1,ans,red_edges,blue_edges,n);
        
        for(int i=0;i<n;i++){
            if(ans[i]==(1<<20))
                ans[i]=-1;
        }
        
        return ans;
    }
};
