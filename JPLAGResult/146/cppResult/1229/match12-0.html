class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
      vector<vector<vector<int>>> adj(2, vector<vector<int>>(n));
      for (const auto &p : red_edges) {
        adj[0][p[0]].push_back(p[1]);
      }
      for (const auto &p : blue_edges) {
        adj[1][p[0]].push_back(p[1]);
      }
      vector<vector<int>> dis(2, vector<int>(n, -1));
      queue<vector<int>> q;
      q.push({0, 0});
      q.push({1, 0});
      dis[0][0] = dis[1][0] = 0;
      while (q.size()) {
        auto s = q.front();
        q.pop();
        //cout << s[0] << " " << s[1] << endl;
        for (const auto &v : adj[s[0]][s[1]]) {
          if (dis[!s[0]][v] == -1) {
            dis[!s[0]][v] = dis[s[0]][s[1]] + 1;
            q.push({!s[0], v});
          }
        }
      }
      vector<int> result(n, -1);
      for (int i = 0; i < n; ++i) {
        // if (result[i] == -1) {
          result[i] = dis[0][i];
       // }
        if (result[i] == -1 || (result[i] > dis[1][i] && dis[1][i] >= 0) ) {
          result[i] = dis[1][i];
        }
        //result[i] = min(dis[0][i], dis[1][i]);
      }
      return result;
    }
};
