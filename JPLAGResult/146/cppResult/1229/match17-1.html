class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        int dis_red[405], dis_blue[405];
        const int inf = 0x3f3f3f3f;
        memset(dis_red, inf, sizeof(dis_red));
        memset(dis_blue, inf, sizeof(dis_blue));
        dis_red[0] = dis_blue[0] = 0;

        for(int i = 0; i < red_edges.size(); i++){
            int v = red_edges[i][0];
            int u = red_edges[i][1];
            if(v == 0){
                dis_red[u] = min(dis_red[u], 1);
            }
        }
        for(int i = 0; i < blue_edges.size(); i++){
            int v = blue_edges[i][0];
            int u = blue_edges[i][1];
            if(v == 0){
                dis_blue[u] = min(dis_blue[u], 1);
            }
        }
        for(int i = 0; i < 2 * n; i++){
            for(int j = 0; j < red_edges.size(); j++){
                int v = red_edges[j][0];
                int u = red_edges[j][1];
                dis_red[u] = min(dis_red[u], dis_blue[v] + 1);
            }
            for(int j = 0; j < blue_edges.size(); j++){
                int v = blue_edges[j][0];
                int u = blue_edges[j][1];
                dis_blue[u] = min(dis_blue[u], dis_red[v] + 1);
            }
        }
        vector<int> ans;
        for(int i = 0; i < n; i++){
            int temp = min(dis_red[i], dis_blue[i]);
            if(temp == inf){
                ans.push_back(-1);
            }
            else{
                ans.push_back(temp);
            }
        }
        return ans;
    }
};
