const int INF = 1000000000;

class Solution {
public:
    using pii = pair<int, int>;
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        int d[n][2];
        for (int i = 0; i < n; ++ i)
            d[i][0] = d[i][1] = INF;
        d[0][0] = d[0][1] = 0;
        
        vector<int> v[n][2];
        for (auto e : red_edges)
            v[e[0]][0].push_back(e[1]);
        for (auto e : blue_edges)
            v[e[0]][1].push_back(e[1]);
        
        queue<pii> Q;
        Q.push(pii(0, 0));
        Q.push(pii(0, 1));
        while (!Q.empty())
        {
            int x, k;
            tie(x, k) = Q.front();
            Q.pop();
            for (auto y : v[x][k])
            {
                if (d[y][k^1] > d[x][k]+1)
                {
                    d[y][k^1] = d[x][k]+1;
                    Q.push(pii(y, k^1));
                }
            }
        }
        
        vector<int> ret(n);
        for (int i = 0; i < n; ++ i)
        {
            ret[i] = min(d[i][0], d[i][1]);
            if (ret[i] == INF) ret[i] = -1;
        }
        return ret;
    }
};
