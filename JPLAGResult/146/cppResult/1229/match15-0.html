const int INF = (1ll << 30);

class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<vector<int>> dist(n, vector<int>(2, INF));
        dist[0][0] = dist[0][1] = 0;
        for (int i = 0; i <= n * 2; i++) {
            for (auto edge: red_edges) {
                dist[edge[1]][0] = min(dist[edge[1]][0], dist[edge[0]][1] + 1);
            }
            for (auto edge: blue_edges) {
                dist[edge[1]][1] = min(dist[edge[1]][1], dist[edge[0]][0] + 1);
            }
        }
        vector<int> ans;
        for (auto i: dist) {
            int j = min(i[0], i[1]);
            if (j == INF) ans.push_back(-1);
            else ans.push_back(j);
        }
        return ans;
    }
};
