class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<int> dst(n+n, 10000);
        vector<vector<int>> edge(n+n, vector<int>());
        for(auto red_edge: red_edges) {
            edge[red_edge[0]].push_back(red_edge[1]+n);
        }
        for(auto blue_edge: blue_edges) {
            edge[blue_edge[0]+n].push_back(blue_edge[1]);
        }
        dst[0] = dst[n] = 0;
        queue<int> que;
        que.push(0);
        que.push(n);
        while(!que.empty()) {
            int now = que.front();
            que.pop();
            for(auto nxt: edge[now]) {
                if(dst[nxt] > dst[now]+1) {
                    dst[nxt] = dst[now]+1;
                    que.push(nxt);
                }
            }
        }
        vector<int> ans(n,0);
        for(int i=0; i<n; ++i) {
            ans[i] = std::min(dst[i], dst[i+n]);
            if(ans[i] == 10000) ans[i] = -1;
        }
        return ans;
    }
};
