int anss[222];
class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        memset(anss,1,sizeof(anss));
        anss[0]=0;
        anss[n]=0;
        int mr=red_edges.size(),mb=blue_edges.size();
        queue<int> q;
        q.push(0);
        q.push(n);
        while(!q.empty())
        {
            int now=q.front();
            //printf("%d\n",now);
            q.pop();
            if(now<n)
            {
                for(int i=0;i<mr;i++)
                {
                    //printf("*%d %d\n",red_edges[i][0],red_edges[i][1]);
                    if(red_edges[i][0]==now&&anss[red_edges[i][1]+n]>anss[now]+1)
                    {
                        anss[red_edges[i][1]+n]=anss[now]+1;
                        q.push(red_edges[i][1]+n);
                    }
                }
            }
            else
            {
                for(int i=0;i<mb;i++)
                {
                    //printf("**%d %d\n",blue_edges[i][0],blue_edges[i][1]);
                    if(blue_edges[i][0]==now-n&&anss[blue_edges[i][1]]>anss[now]+1)
                    {
                        anss[blue_edges[i][1]]=anss[now]+1;
                        q.push(blue_edges[i][1]);
                    }
                }
            }
        }
        vector<int> ans;
        for(int i=0;i<n;i++)
        {
            ans.push_back(min(anss[i],anss[i+n]));
            if(ans[i]>1000)ans[i]=-1;
        }
        return ans;
    }
};
