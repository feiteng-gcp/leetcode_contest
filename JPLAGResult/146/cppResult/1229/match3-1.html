class Solution {
public:
    struct Data {
        int idx;
        int type;//0blue,1red
        int value;
        Data (int _idx = 0, int _type = 0, int _value = 0) {
            idx = _idx;
            type = _type;
            value = _value;
        }
    };
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        int node[200][2];
        memset(node, -1, sizeof(node));
        queue<Data> Q;
        node[0][0] = 0;
        node[0][1] = 0;
        Q.push(Data(0, 0, 0));
        Q.push(Data(0, 1, 0));
        
        while (!Q.empty()) {
            Data data = Q.front();
            Q.pop();
            if (data.type == 0) {
                for (int i = 0;i < red_edges.size();i++) {
                    if (red_edges[i][0] != data.idx) {
                        continue;
                    }
                    int y = red_edges[i][1];

                    if (node[y][1] == -1) {
                        node[y][1] = data.value + 1;
                        Q.push(Data(y, 1, node[y][1]));
                    }
                }
            } else {
                for (int i = 0;i < blue_edges.size();i++) {
                    if (blue_edges[i][0] != data.idx) {
                        continue;
                    }
                    int y = blue_edges[i][1];
                    if (node[y][0] == -1) {

                        node[y][0] = data.value + 1;
                        Q.push(Data(y, 0, node[y][0]));
                    }
                }
            }
        }
        
        vector<int> ret(n, -1);
        for (int i = 0;i < n;i++) {
            if (node[i][0] != -1) {
                if (ret[i] == -1 || ret[i] > node[i][0]) {
                    ret[i] = node[i][0];
                } 
            }
            if (node[i][1] != -1) {
                if (ret[i] == -1 || ret[i] > node[i][1]) {
                    ret[i] = node[i][1];
                } 
            }
        }
        return ret;
    }
};
