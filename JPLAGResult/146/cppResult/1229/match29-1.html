constexpr static int UNVISITED = 9999999;


class Solution {
public:
    vector<int> shortestAlternatingPaths(int N, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        dedup(red_edges);
        dedup(blue_edges);
        vector<vector<int>> rnext(N);
        vector<vector<int>> bnext(N);
        for (const auto& re : red_edges) {
            rnext[re[0]].emplace_back(re[1]);
        }
        for (const auto& be : blue_edges) {
            bnext[be[0]].emplace_back(be[1]);
        }
        
        vector<int> rdist(N, UNVISITED); // dist where next edge is red
        vector<int> bdist(N, UNVISITED); // dist where next edge is blue
        queue<pair<int, bool>> q; // queue of <index, next_edge_is_red>
        rdist[0] = 0;
        bdist[0] = 0;
        q.push(make_pair(0, true));
        q.push(make_pair(0, false));
        int dist = 0;
        while(!q.empty()) {
            ++dist;
            int batch_size = q.size();
            for (int b = 0; b < batch_size; ++b) {
                int cur_node = q.front().first;

                bool next_red = q.front().second;
                q.pop();
                const auto& adj_tab = next_red ? rnext : bnext;
                auto& next_dist_tab = next_red ? bdist : rdist;
                for (int next : adj_tab[cur_node]) {
                    if (next_dist_tab[next] == UNVISITED) {
                        next_dist_tab[next] = dist;
                        q.push(make_pair(next, !next_red));
                    }
                }
            }
        }
        
        vector<int> result(N);
        for (int i = 0; i < N; ++i) {
            result[i] = min(rdist[i], bdist[i]);
            if (result[i] == UNVISITED) {
                result[i] = -1;
            }
        }
        return result;
    }
    
    void dedup(vector<vector<int>>& edges) {
        sort(edges.begin(), edges.end());
        auto new_end = unique(edges.begin(), edges.end());
        edges.resize(new_end - edges.begin());
    }
};
