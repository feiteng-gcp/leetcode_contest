class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        // 0 ~ n-1: start from red, n ~ 2n-1: start from blue
        vector<vector<int>> edges(2*n, vector<int>());
        vector<int> distance(2*n, 1000);
        
        for (auto red_edge: red_edges)
            edges[red_edge[0]].push_back(red_edge[1] + n);
        for (auto blue_edge: blue_edges)
            edges[blue_edge[0] + n].push_back(blue_edge[1]);
        
        // (distance, vertex)
        /*
        priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> pq;
        
        pq.push({0, 0});
        pq.push({0, n});
        
        while(!pq.empty()) {
            auto vec = pq.top();
            pq.pop();
            int dis = vec[0], v1 = vec[1];
            distance[v1] = dis;
            
            for (auto v2: edges[v1]) {
                if (distance[v2] == 1000) 
                    pq.push({dis + 1, v2});
            }
        }
        */
        
        distance[0] = 0, distance[n] = 0;
        queue<int> q;
        q.push(0), q.push(n);

        while(!q.empty()) {
            int v1 = q.front();
            q.pop();
            for (auto v2: edges[v1]) {
                if (distance[v2] == 1000) {
                    distance[v2] = distance[v1] + 1;
                    q.push(v2);
                }
            }
        }
        
        vector<int> res(n, 0);
        for (int i = 0; i < n; i++) {
            res[i] = min(distance[i], distance[i+n]);
            if (res[i] == 1000) res[i] = -1;
        }
        return res;
    }
};
