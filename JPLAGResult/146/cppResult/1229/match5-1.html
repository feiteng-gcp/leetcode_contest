class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<vector<int>> redEdges(n, vector<int>());
        vector<vector<int>> blueEdges(n, vector<int>());
        
        for (auto edge : red_edges) {
            redEdges[edge[0]].push_back(edge[1]);
        }
        
        for (auto edge : blue_edges) {
            blueEdges[edge[0]].push_back(edge[1]);
        }
        
        vector<int> reds(n, -1);
        vector<int> blues(n, -1);
        set<int> candidates;
        reds[0] = blues[0] = 0;
        candidates.insert(0);
        while (!candidates.empty()) {
            vector<int> temp(candidates.begin(), candidates.end());
            candidates.clear();
            for (int from : temp) {
                for (int to : redEdges[from]) {
                    if (blues[from] != -1 && (reds[to] == -1 || blues[from] + 1 < reds[to])) {
                        reds[to] = blues[from] + 1;
                        candidates.insert(to);
                    }
                }
                for (int to : blueEdges[from]) {
                    if (reds[from] != -1 && (blues[to] == -1 || reds[from] + 1 < blues[to])) {
                        blues[to] = reds[from] + 1;
                        candidates.insert(to);
                    }
                }
            }
        }

        vector<int> res(n, -1);

        for (int i = 0; i < n; i++) {
            if (reds[i] == -1 && blues[i] == -1)
                res[i] = -1;
            else if (reds[i] == -1)
                res[i] = blues[i];

            else
                res[i] = min(reds[i], blues[i]);
        }
        return res;
    }
};
