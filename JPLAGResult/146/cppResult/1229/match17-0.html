class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        
        vector<int> redAns, blueAns;
        redAns.push_back(0);
        blueAns.push_back(0);
        for( int i = 1; i < n * n; i++ ) {

            redAns.push_back(1000000);
            blueAns.push_back(1000000);
        }
        
        for( int i = 0; i <= n; i++ ) {

            for( int j = 0; j < red_edges.size(); j++ ) {

                int from = red_edges[j][0];
                int to = red_edges[j][1];
                if( blueAns[from] + 1 < redAns[to] ) {

                    redAns[to] = blueAns[from] + 1;
                }
            }
            
            for( int j = 0; j < blue_edges.size(); j++ ) {

                int from = blue_edges[j][0];
                int to = blue_edges[j][1];
                if( redAns[from] + 1 < blueAns[to] ) {

                    blueAns[to] = redAns[from] + 1;
                } 
            }
        }
        
        vector<int> ans;
        ans.push_back(0);
        for( int i = 1; i < n; i++ ) {

            if( redAns[i] == 1000000 && blueAns[i] == 1000000 ) {

                ans.push_back(-1);
            }
            else if( redAns[i] > blueAns[i] ) {

                ans.push_back(blueAns[i]);
            }
            else {

                ans.push_back(redAns[i]);
            }
        }
       
        return ans;
