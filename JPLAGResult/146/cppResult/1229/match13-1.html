class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        // 0 : red, 1 : blue

        vector<int> edges[2][n];
        for (const auto &red_edge : red_edges) {
            edges[0][red_edge[0]].push_back(red_edge[1]);
        }
        for (const auto &blue_edge : blue_edges) {
            edges[1][blue_edge[0]].push_back(blue_edge[1]);
        }
        int shortest[n][2];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 2; j++) {
                shortest[i][j] = 1000;
            }
        }
        queue<pair<int,int>> to_expand;
        // {node, last_color}
        to_expand.push({0, 0});
        to_expand.push({0, 1});
        int distance = 0;
        while (!to_expand.empty()) {
            int N = to_expand.size();
            for (int i = 0; i < N; i++) {
                auto P = to_expand.front();
                int u = P.first, color = P.second;
                to_expand.pop();
                if (shortest[u][color] == 1000) {
                    shortest[u][color] = distance;
                    for (int v : edges[1 - color][u]) {
                        to_expand.push({v, 1 - color});
                    }
                }
            }
            distance++;

        }
        vector<int> ans(n);
        for (int i = 0; i < n; i++) {
            ans[i] = min(shortest[i][0], shortest[i][1]);
            if (ans[i] == 1000) {
                ans[i] = -1;
            }
        }
        return ans;
    }
};
