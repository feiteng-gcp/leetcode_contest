class Solution {
    queue<int> q;
    vector<int> dis;
    vector<int> ans;
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        q.push(0);
        q.push(n);
        dis = vector<int>(2 * n, 2 * n + 1);
        ans = vector<int>(n);
        dis[0] = dis[n] = 0;
        while (!q.empty()) {
            int now = q.front();
            q.pop();
            if (now < n) {
                for (vector<int> i : blue_edges)
                    if (i[0] == now && dis[now] + 1 < dis[i[1] + n]) {
                        dis[i[1] + n] = dis[now] + 1;
                        q.push(i[1] + n);
                    }
            } else {
                for (vector<int> i : red_edges)

                    if (i[0] == now - n && dis[now] + 1 < dis[i[1]]) {
                        dis[i[1]] = dis[now] + 1;
                        q.push(i[1]);
                    }
            }
        }
        for (int i = 0; i < n; ++i) {
            ans[i] = min(dis[i], dis[i + n]);
            if (ans[i] == 2 * n + 1)
                ans[i] = -1;
        }
        return ans;
    }
};
