const int MAX_DIST = 1000000;

class Node
{
    public:
    int red_cost;
    int blue_cost;
    
    vector<int> link_red;
    vector<int> link_blue;
    
    Node()
    {
        red_cost = MAX_DIST;
        blue_cost = MAX_DIST;
    }
};

class Node_index
{
    public:
    int index;
    bool red;
    
    Node_index(int a,bool r)
    {
        index = a;
        red = r;
    }
};

class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) 
    {
        Node nodes[n];
        for(int i = 0;i<red_edges.size();i++)
        {
            int u = red_edges[i][0];
            int v = red_edges[i][1];
            
            nodes[u].link_red.push_back(v);
            //nodes[v].link_red.push_back(u);
        }
        
        for(int i = 0;i<blue_edges.size();i++)
        {
            int u = blue_edges[i][0];
            int v = blue_edges[i][1];
            
            nodes[u].link_blue.push_back(v);
            //nodes[v].link_blue.push_back(u);
        }
        
        nodes[0].red_cost = 0;
        nodes[0].blue_cost = 0;
        queue<Node_index> process_queue;
        process_queue.push(Node_index(0,true));
        process_queue.push(Node_index(0,false));
        
        while(!process_queue.empty())
        {

            Node_index wip = process_queue.front();
            process_queue.pop();
            
            //cout << "Processing" << endl;
            //cout << wip.index << " " << wip.red << endl;
            
            if(wip.red)//loop through blue links
            {
                for(int link_index = 0;link_index < nodes[wip.index].link_blue.size();link_index++ )
                {
                    if(nodes[nodes[wip.index].link_blue[link_index]].blue_cost == MAX_DIST)
                    {
                        nodes[nodes[wip.index].link_blue[link_index]].blue_cost = nodes[wip.index].red_cost + 1;
                        process_queue.push(Node_index(nodes[wip.index].link_blue[link_index],false));
                        
                        //cout << "Added " << nodes[wip.index].link_blue[link_index] << " " << false << endl;
                    }
                }
            }
            else
            {
                for(int link_index = 0;link_index < nodes[wip.index].link_red.size();link_index++ )
                {
                    if(nodes[nodes[wip.index].link_red[link_index]].red_cost == MAX_DIST)
                    {
                        nodes[nodes[wip.index].link_red[link_index]].red_cost = nodes[wip.index].blue_cost + 1;
                        process_queue.push(Node_index(nodes[wip.index].link_red[link_index],true));
                        
                        //cout << "Added " << nodes[wip.index].link_red[link_index] << " " << true << endl;
                    }
                }
            }
        }
        
        vector<int> answer;
        for(int i = 0;i<n;i++)
        {
            answer.push_back(min(nodes[i].red_cost,nodes[i].blue_cost));
            if(answer[i] == MAX_DIST)
            {
                answer[i] = -1;
            }
        }
        
        return answer;
    }
};
