class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& r, vector<vector<int>>& b) {
        
        vector<int> ans(n);
        
        vector<vector<int>> res(n,vector<int>(2,5000));
        
        //vector<int> label(n,-1);
        
        res[0][0] = 0;
        res[0][1] = 0;
        
        for(int i = 0; i < n; ++i){
            //0
            for(auto e:r){
                if(res[e[0]][1]+1 < res[e[1]][0]){
                    res[e[1]][0] = res[e[0]][1]+1;
                }
            }
            //1
            for(auto e:b){
                if(res[e[0]][0]+1 < res[e[1]][1]){
                    res[e[1]][1] = res[e[0]][0]+1;
                }                
            }
        }
        
        for(int i = 0; i < n; ++i){
            
            ans[i] = min(res[i][0],res[i][1]);
            
            if(ans[i] == 5000)
                ans[i] = -1;
        }
        
        return ans;
        
    }
};
