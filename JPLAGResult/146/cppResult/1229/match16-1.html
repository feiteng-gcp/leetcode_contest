class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        int oo = 1000000000;
        vector<vector<int> > f(n, vector<int>(2, oo));
        f[0][0] = 0;
        f[0][1] = 0;

        for (int i = 0; i < 2 * n; i++) {
            for (int j = 0; j < red_edges.size(); j++) {
                int u = red_edges[j][0];
                int v = red_edges[j][1];
                f[v][1] = min(f[v][1], f[u][0] + 1);
            }
            for (int j = 0; j < blue_edges.size(); j++) {
                int u = blue_edges[j][0];
                int v = blue_edges[j][1];
                f[v][0] = min(f[v][0], f[u][1] + 1);
            }
        }
        vector<int> ret(n, oo);
        for (int i = 0; i < n; i++) {
            ret[i] = min(f[i][0], f[i][1]);
            if (ret[i] == oo) {
                ret[i] = -1;
            }
        }
        return ret;
    }
};
