class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<int> ans(n,-1);
        vector<vector<int>> dp(n,vector<int>(2,-1));
        queue<pair<int,int>> q;
        
        q.push(make_pair(0,-1));
        dp[0][0]=0;
        dp[0][1]=0;
        while(!q.empty())
        {
            
            auto now = q.front();
            q.pop();
            if(now.second == -1 || now.second == 0)//当前路径红色
            {
                for(auto blue_edge : blue_edges)
                {
                    if(blue_edge[0] == now.first)
                    {
                        if(dp[blue_edge[1]][1] == -1 || dp[blue_edge[1]][1] > dp[now.first][0] + 1)
                        {
                            q.push(make_pair(blue_edge[1],1));
                            dp[blue_edge[1]][1] = dp[now.first][0] + 1;
                        }
                    }
                }
            }
            if(now.second == -1 || now.second == 1)//当前路径蓝色
            {
                for(auto red_edge : red_edges)
                {
                    if(red_edge[0] == now.first)
                    {
                        if(dp[red_edge[1]][0] == -1 || dp[red_edge[1]][0] > dp[now.first][1] + 1)
                        {
                            q.push(make_pair(red_edge[1],0));
                            dp[red_edge[1]][0] = dp[now.first][1] + 1;
                        }
                    }
                }
            }
        }
        for(int i = 0;i<n;i++)
        {
            if(dp[i][0] == -1 && dp[i][1] == -1)
                ans[i] = -1;
            else if(dp[i][0] == -1 )
                ans[i] = dp[i][1];
            else if(dp[i][1] == -1 )
                ans[i] = dp[i][0];
            else
                ans[i] = min(dp[i][0],dp[i][1]);
        }
        return ans;
    }
};
