class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<vector<int>> d(n, vector<int>(2, 0xfffffff));
        d[0][0] = d[0][1] = 0; // 0 -> red, 1-> blue
        for (int i = 0; i < n; ++i) {
            for (auto &e : red_edges) {
                if (d[e[0]][1] + 1 < d[e[1]][0]) {
                    d[e[1]][0] = d[e[0]][1] + 1;
                }
            }
            for (auto &e : blue_edges) {
                if (d[e[0]][0] + 1 < d[e[1]][1]) {
                    d[e[1]][1] = d[e[0]][0] + 1;
                }
            }
        }
        
        vector<int> ret;
        for (auto ds : d) {
            int _min = min(ds[0], ds[1]);
            ret.push_back(_min == 0xfffffff ? -1 : _min);
        }
        return ret;
    }
};
