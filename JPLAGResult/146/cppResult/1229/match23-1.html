class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<vector<int>> rr(n);
        vector<vector<int>> bb(n);
        for(auto e:red_edges){
            rr[e[0]].push_back(e[1]);
        }
        for(auto e:blue_edges){
            bb[e[0]].push_back(e[1]);
        }
        vector<int> ans(n,-1);
        ans[0]=0;
        queue<pair<int,int>> qq;
        vector<bool> usedR(n,false);
        vector<bool> usedB(n,false);
        usedR[0]=true;
        usedB[0]=true;
        qq.push(make_pair(0,0));
        qq.push(make_pair(0,1));
        int cnt=1;

        while(!qq.empty()){
            int t=qq.size();
            while(t--){
                auto p=qq.front();
                qq.pop();
                int a=p.first;
                int b=p.second;
                if(b==1){
                    for(auto c:rr[a]){
                        if(usedR[c]) continue;
                        usedR[c]=true;
                        if(ans[c]==-1||ans[c]>cnt) ans[c]=cnt;
                        qq.push(make_pair(c,0));
                    }
                }else{
                    for(auto c:bb[a]){
                        if(usedB[c]) continue;
                        usedB[c]=true;
                        if(ans[c]==-1||ans[c]>cnt) ans[c]=cnt;
                        qq.push(make_pair(c,1));
                    }
                }
            }
            cnt++;
        }
        return ans;
    }
};
