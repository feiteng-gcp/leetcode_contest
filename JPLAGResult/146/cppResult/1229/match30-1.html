class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<vector<vector<int> > >r(n,vector<vector<int> >(2,vector<int>()));
        for(auto v:red_edges){
            r[v[0]][0].push_back(v[1]);
        }
        for(auto v:blue_edges){
            r[v[0]][1].push_back(v[1]);
        }
        vector<vector<int> > res(n,vector<int>(2,-1));
        res[0][0] = res[0][1] = 0;
        queue<pair<int,int> > q;
        q.push({0,0});
        q.push({0,1});
        while(q.size()){
            auto p = q.front();
            q.pop();
            int x = p.first;
            int y = p.second;
            for(int i = 0;i < r[x][1-y].size();i++){
                int y1 = r[x][1-y][i];
                if(res[y1][1-y] == -1 ){
                    res[y1][1-y] = res[x][y] + 1;
                    q.push({y1,1-y});
                }
            }
        }
        vector<int>ans;
        for(int i = 0;i <n;i++){
            if(res[i][0] == -1 && res[i][1] == -1)
                ans.push_back(-1);
            else if(res[i][0] != -1 && res[i][1] != -1){
                ans.push_back(min(res[i][0],res[i][1]));
            } else {
                ans.push_back(res[i][0] == -1?res[i][1]:res[i][0]);
            }
        }
        return ans;
        
    }
};
