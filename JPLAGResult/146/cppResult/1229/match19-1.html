class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
      vector<int> ret_r(n, 500), ret_b(n, 500), ret(n, 500);
      ret[0] = 0, ret_r[0] = 0, ret_b[0] = 0;
      vector<vector<int>> red_graph(n, vector<int>(n, 0));
      vector<vector<int>> blue_graph(n, vector<int>(n, 0));
      vector<int> vis(n, 0);
  
      for(auto& r : red_edges)
        red_graph[r[0]][r[1]] = 1;
      for(auto& b : blue_edges)
        blue_graph[b[0]][b[1]] = 1;
      queue<vector<int>> que;
      que.push({0, 0, 0});
      que.push({0, 0, 1});
      vis[0] = 1;

      while(!que.empty()) {
        vector<int>& top = que.front();
        if(top[2] == 0) {
          for (int i = 0; i < n; ++i) {
            if (ret_b[i] > top[1] + 1 && blue_graph[top[0]][i]) {
              vis[i] = 1;
              ret_b[i] = top[1] + 1;
              que.push({i, ret_b[i], 1});

            }
          }
        } else {
          for (int i = 0; i < n; ++i) {
            if (ret_r[i] > top[1] + 1 && red_graph[top[0]][i]) {
              vis[i] = 1;
              ret_r[i] = top[1] + 1;
              que.push({i, ret_r[i], 0});
            }
          }
        }
        
        que.pop();
      }
      for(int i = 0; i < n; ++i) {
        ret[i] = min(ret_r[i], ret_b[i]);
        if(ret[i] == 500)
          ret[i] = -1;
      }
      return ret;
    }
};
