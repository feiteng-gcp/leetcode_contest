class Solution {
public:
  vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
    vector<unordered_set<int>> r(n);
    vector<unordered_set<int>> b(n);
    for (const auto& e : red_edges)      
      r[e[0]].insert(e[1]);    
    for (const auto& e : blue_edges)
      b[e[0]].insert(e[1]);
    unordered_set<int> seen_r;
    unordered_set<int> seen_b;
    vector<int> ans(n, -1);
    ans[0] = 0;
    queue<pair<int, int>> q;
    q.push({0, 0}); // red
    q.push({0, 1}); // blue    
    while (!q.empty()) {
      int p = q.front().first;
      int is_red = q.front().second & 1;
      int steps = q.front().second >> 1;
      // cout << p << " " << is_red << " " << steps << endl;
      q.pop();
      if (ans[p] >= 0)
        ans[p] = min(ans[p], steps);
      else
        ans[p] = steps;
      const auto& edges = is_red ? r : b;
      auto& seen = is_red ? seen_r : seen_b;
      for (int nxt : edges[p]) {
        if (seen.count(nxt)) continue;
        seen.insert(nxt);
        q.push({nxt, ((steps + 1) << 1) | (1 - is_red)});
      }
    }
    return ans;
  }
};
