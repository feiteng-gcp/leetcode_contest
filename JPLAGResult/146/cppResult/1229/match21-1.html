class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<vector<int>> red(n, vector<int>(0));
        vector<vector<int>> blue(n, vector<int>(0));
        for (int i = 0; i < red_edges.size(); ++i) {
            red[red_edges[i][0]].push_back(red_edges[i][1]);
        }
        for (int i = 0; i < blue_edges.size(); ++i) {
            blue[blue_edges[i][0]].push_back(blue_edges[i][1]);
        }
        queue<pair<int, int>> q;
        const int max = 1000;
        vector<vector<int>> opt(n, vector<int>(2, max));
        opt[0][0] = opt[0][1] = 0;
        q.push(make_pair(0, 0));
        q.push(make_pair(0, 1));
        while (!q.empty()) {
            const int x = q.front().first;
            const int c = q.front().second;
            q.pop();
            if (c == 0) {
                for (int i = 0; i < blue[x].size(); ++i) {
                    const int y = blue[x][i];

                    if (opt[y][1] == max) {
                        opt[y][1] = opt[x][0] + 1;
                        q.push(make_pair(y, 1));
                    }
                }
            } else {
                for (int i = 0; i < red[x].size(); ++i) {
                    const int y = red[x][i];

                    if (opt[y][0] == max) {
                        opt[y][0] = opt[x][1] + 1;
                        q.push(make_pair(y, 0));
                    }
                }
            }
        }
        vector<int> ans(n, -1);
        for (int i = 0; i < n; ++i) {
            if (opt[i][0] < opt[i][1]) {
                if (opt[i][0] != max) ans[i] = opt[i][0];
            } else {
                if (opt[i][1] != max) ans[i] = opt[i][1];
            }
        }
        return ans;
    }
};
