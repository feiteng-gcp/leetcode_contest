class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<vector<vector<int>>> adj(n), blue(n);
        for (auto& r : red_edges) adj[r[0]].push_back({r[1], 0});
        for (auto& b : blue_edges) adj[b[0]].push_back({b[1], 1});
        
        // BFS
        vector<vector<int>> dist(n, vector<int>(3, INT_MAX));
        queue<pair<int, int>> q({{0, 2}});
        dist[0] = {0, 0, 0};
        while (q.size()) {
            int node, color;
            tie(node, color) = q.front(); q.pop();
            for (auto& m : adj[node]) {
                if (m[1] == color) continue;
                if (dist[m[0]][m[1]] > dist[node][color] + 1) {
                    dist[m[0]][m[1]] = dist[node][color] + 1;
                    q.push({m[0], m[1]});
                }
            }
        }
        
        vector<int> res(n);
        for (int i=0; i<n; i++) {
            int r = min(dist[i][0], dist[i][1]);
            res[i] = r == INT_MAX ? -1 : r;
        }
        return res;
    }
};
