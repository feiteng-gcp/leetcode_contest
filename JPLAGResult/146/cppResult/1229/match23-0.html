class Solution {
public:
    vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
        vector<int> red[n];
        vector<int> blue[n];
        for (auto v: red_edges) {
            red[v[0]].push_back(v[1]);
        }
        for (auto v: blue_edges) {
            blue[v[0]].push_back(v[1]);
        }
        vector<int> ans(n, -1);
        queue<pair<int, int>> qu;
        qu.push({0, -1});
        int depth = 0;
        set<pair<int, int>> visRed;
        set<pair<int, int>> visBlue;
        // vis.insert({}) = 1;
        
        while (!qu.empty()) {
            int len = qu.size();
            while (len--) {
                auto pr = qu.front();
                qu.pop();
                int node = pr.first;
                int type = pr.second;

                // cout << "node: " << node << ", type: " << type << ", depth: " << depth << endl;
                if (ans[node] == -1) ans[node] = depth;

                if (type != 0) {
                    for (auto next: red[node]) {
                        if (visRed.count({node, next}) == 0) {
                            visRed.insert({node, next});
                            qu.push({next, 0}); // type = 0: red
                        }
                    }
                }
                if (type != 1) {
                    for (auto next: blue[node]) {
                        if (visBlue.count({node, next}) == 0) {
                            visBlue.insert({node, next});
                            qu.push({next, 1}); // type = 1: blue
                        }
                    }
                }
            }

            depth++;
        }
        
        return ans;
    }
};
