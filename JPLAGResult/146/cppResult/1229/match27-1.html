class Solution {
public:
  vector<int> shortestAlternatingPaths(int n, vector<vector<int>>& red_edges, vector<vector<int>>& blue_edges) {
    vector<int> G[n][2];
    for (auto a: red_edges) {
      G[a[0]][0].push_back(a[1]);
    }
    for (auto a: blue_edges) {
      G[a[0]][1].push_back(a[1]);
    }
    vector<vector<int> >D(n, vector<int>(2, 0x3f3f3f3f));
    queue<pair<int, int>> Q;
    Q.push({0, 0});
    Q.push({0, 1});
    D[0][0] = D[0][1] = 0;
    while (!Q.empty()) {
      auto [s, k] = Q.front(); Q.pop();
      for (int t: G[s][1-k]) {
        if (D[t][1-k] != 0x3f3f3f3f) continue;
        Q.push({t, 1 - k});
        D[t][1-k] = D[s][k] + 1;
      }
    }
    vector<int> ans(n);
    for (int i = 0; i < n; i++) ans[i] = min(D[i][0], D[i][1]);
    for (int i = 0; i < n; i++) if (ans[i] == 0x3f3f3f3f) ans[i] = -1;
    return ans;
  }
};
