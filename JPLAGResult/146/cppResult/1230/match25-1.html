class Solution {
public:
  int gao(vector<int> A) {
    return *max_element(A.begin(), A.end()) - *min_element(A.begin(), A.end());
  }
  int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
    int N = arr1.size();
    vector<int> A[4];

    for (int i = 0; i < 4; i++) A[i].resize(N);
    for (int i = 0; i < N; i++) {
      A[0][i] = arr1[i] + arr2[i] - i;
      A[1][i] = arr1[i] - arr2[i] - i;
      A[2][i] = - arr1[i] - arr2[i] - i;
      A[3][i] = - arr1[i] + arr2[i] - i;
    }
    int ans = 0;
    for (int i = 0; i < 4; i++) {
      ans = max(ans, gao(A[i]));
    }
    return ans;
  }
};
