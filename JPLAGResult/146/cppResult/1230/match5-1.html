class Solution {
public:
    int maxAbsValExpr(vector<int>& a, vector<int>& b) {
        int n = a.size();
        int ans = 0;
        int z[4] = {1000000000, 1000000000, 1000000000, 1000000000};
        for (int i = 0; i < n; i++) {
            z[0] = min(z[0], a[i] + b[i] + i);
            z[1] = min(z[1], a[i] - b[i] + i);
            z[2] = min(z[2], -a[i] + b[i] + i);
            z[3] = min(z[3], -a[i] - b[i] + i);
            ans = max(ans, a[i] + b[i] + i - z[0]);
            ans = max(ans, a[i] - b[i] + i - z[1]);
            ans = max(ans, -a[i] + b[i] + i - z[2]);
            ans = max(ans, -a[i] - b[i] + i - z[3]);
        }
        return ans;
    }
};
