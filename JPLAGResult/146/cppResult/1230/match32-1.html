using namespace std;
using ll = long long;
using ld = double;
typedef pair<int, int> II;
typedef pair<II, int> III;
typedef complex<ld> cd;
typedef vector<cd> vcd;


class Solution {
public:
    int maxAbsValExpr(vector<int>& c, vector<int>& b) {
        int n = SZ(c);
        vector<int> a;
        FOR(i,0,n-1) a.emplace_back(c[i] + b[i] + i);
        sort(ALL(a));
        int res = a[n-1] - a[0];
        
        a.clear();
        FOR(i,0,n-1) a.emplace_back(c[i] + b[i] - i);
        sort(ALL(a));
        res = max(res, a[n-1] - a[0]);
        
        a.clear();
        FOR(i,0,n-1) a.emplace_back(c[i] - b[i] + i);
        sort(ALL(a));
        res = max(res, a[n-1] - a[0]);
        
        a.clear();
        FOR(i,0,n-1) a.emplace_back(c[i] - b[i] - i);
        sort(ALL(a));
        res = max(res, a[n-1] - a[0]);
        
        a.clear();
        FOR(i,0,n-1) a.emplace_back(-c[i] + b[i] + i);
        sort(ALL(a));
        res = max(res, a[n-1] - a[0]);
        
        a.clear();
        FOR(i,0,n-1) a.emplace_back(-c[i] + b[i] - i);
        sort(ALL(a));
        res = max(res, a[n-1] - a[0]);
        
        a.clear();
        FOR(i,0,n-1) a.emplace_back(-c[i] - b[i] + i);
        sort(ALL(a));
        res = max(res, a[n-1] - a[0]);
        
        a.clear();
        FOR(i,0,n-1) a.emplace_back(-c[i] - b[i] - i);
        sort(ALL(a));
        res = max(res, a[n-1] - a[0]);
        
        return res;
    }
};
