class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int N = arr1.size();
        vector<int> pp, pm, mp, mm;
        for (int i = 0; i < N; i++)
        {
            pp.push_back(i + arr1[i] + arr2[i]);
            pm.push_back(i + arr1[i] - arr2[i]);
            mp.push_back(i - arr1[i] + arr2[i]);
            mm.push_back(i - arr1[i] - arr2[i]);
        }
        sort(pp.begin(), pp.end());
        sort(pm.begin(), pm.end());
        sort(mp.begin(), mp.end());
        sort(mm.begin(), mm.end());
        return max({pp.back() - pp[0],
                    pm.back() - pm[0],
                    mp.back() - mp[0],
                    mm.back() - mm[0]
                   });
    }
};
