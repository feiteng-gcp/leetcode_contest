class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size();
        vector<int> xy(8);
        vector<int> sxy(8, INT_MAX);
        
        for (int i = 0; i < n; ++i) {
            xy[0] = max(xy[0], arr1[i] + arr2[i] + i);
            xy[1] = max(xy[1], arr1[i] + arr2[i] - i);
            xy[2] = max(xy[2], arr1[i] - arr2[i] + i);
            xy[3] = max(xy[3], arr1[i] - arr2[i] - i);
            xy[4] = max(xy[4], -arr1[i] + arr2[i] + i);
            xy[5] = max(xy[5], -arr1[i] + arr2[i] - i);
            xy[6] = max(xy[6], -arr1[i] - arr2[i] + i);
            xy[7] = max(xy[7], -arr1[i] - arr2[i] - i);
        }
        
        int best = 0;
        for (int i=0;i<8;++i) {
            if (xy[i] + xy[(i^7)] > best) {
                best = xy[i] + xy[(i^7)];
            }
        }
        return best;
        
    }
};
