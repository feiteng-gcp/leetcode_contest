class Solution {
public:
    int find(vector<int>& arr) {
        int max = arr[0];
        int min = arr[0];
        int ret = 0;
        for (int i = 1;i < arr.size();i++) {
            int x = arr[i] + i;
            int temp;
            temp = abs(x - max);
            if (ret < temp) {
                ret = temp;
            }
            temp = abs(x - min);
            if (ret < temp) {
                ret = temp;
            }
            if (max < x) {
                max = x;
            }
            if (min > x) {
                min = x;
            }
        }
        return ret;
    }
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        vector<int> a(arr1.size(), 0);
        int ret = 0;
        int temp;
        for (int i = 0;i < arr1.size();i++) {
            a[i] = arr1[i] + arr2[i];
        }
        temp = find(a);
        if (ret < temp) {
            ret = temp;
        }
        
        for (int i = 0;i < arr1.size();i++) {
            a[i] = -arr1[i] - arr2[i];
        }
        temp = find(a);
        if (ret < temp) {
            ret = temp;
        }
        
        for (int i = 0;i < arr1.size();i++) {
            a[i] = arr1[i] - arr2[i];
        }
        temp = find(a);
        if (ret < temp) {
            ret = temp;
        }
        
        for (int i = 0;i < arr1.size();i++) {
            a[i] = -arr1[i] + arr2[i];
        }
        temp = find(a);
        if (ret < temp) {
            ret = temp;
        }
        
        return ret;
    }
};
