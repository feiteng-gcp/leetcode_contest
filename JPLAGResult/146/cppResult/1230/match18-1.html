class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n=arr1.size();
        int a1[n],a2[n],a3[n],a4[n];
        for(int i=0;i<n;i++){
            a1[i]=arr1[i]+arr2[i]+i;
            a2[i]=arr1[i]-arr2[i]+i;
            a3[i]=arr1[i]+arr2[i]-i;
            a4[i]=arr1[i]-arr2[i]-i;
        }
        sort(a1,a1+n);
        sort(a2,a2+n);
        sort(a3,a3+n);
        sort(a4,a4+n);
        int mx=0;
        mx=max(mx,a1[n-1]-a1[0]);
        mx=max(mx,a2[n-1]-a2[0]);
        mx=max(mx,a3[n-1]-a3[0]);
        mx=max(mx,a4[n-1]-a4[0]);
        return mx;
    }
};
