class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size();
        vector<int> a(n, 0), b(n, 0), c(n, 0), d(n, 0);
        for(int i = 0; i < n; ++i) {
            a[i] = arr1[i] + arr2[i] + i;
            b[i] = arr1[i] + arr2[i] - i;
            c[i] = arr1[i] - arr2[i] + i;
            d[i] = arr1[i] - arr2[i] - i;
        }
        
        int resa = helper(a);
        int resb = helper(b);
        int resc = helper(c);
        int resd = helper(d);
        return max(resa, max(resb, max(resc, resd)));
    }
    
private:
    int helper(vector<int> &nums) {
        int minv = INT_MAX, maxv = INT_MIN;
        for(int num: nums) {
            minv = min(num, minv);
            maxv = max(num, maxv);
        }
        return maxv - minv;
    }
};
