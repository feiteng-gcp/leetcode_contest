class Solution {
public:
    // a1[i]+a2[i]+i - (a1[j]+a2[j]+j)
    // a1[i]-a2[i]+i - (a1[j]-a2[j]+j)
    // 
    int INF=1e9;

    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n=arr1.size();
        int ans=0;
        int mi1=INF,mi2=INF,mi3=INF,mi4=INF;
        for(int i=0;i<n;i++)
        {
            if(i>=1)
            {
                ans=max(ans,arr1[i]+arr2[i]+i-mi1);
                ans=max(ans,arr1[i]-arr2[i]+i-mi2);
                ans=max(ans,-arr1[i]+arr2[i]+i-mi3);
                ans=max(ans,-arr1[i]-arr2[i]+i-mi4);
            }
            mi1=min(mi1,arr1[i]+arr2[i]+i);
            mi2=min(mi2,arr1[i]-arr2[i]+i);
            mi3=min(mi3,-arr1[i]+arr2[i]+i);
            mi4=min(mi4,-arr1[i]-arr2[i]+i);
        }
        return ans;
    }
};
