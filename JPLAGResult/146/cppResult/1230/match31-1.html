class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n= arr1.size();
        vector<int> extrem;
        extrem.resize(8);
        for(int d1= 0; d1<=1; ++d1){
            for(int d2= 0; d2<=1; ++d2){
                for(int s=0; s<=1; ++s){
                    int bs, bi; 
                    bi=0;
                    bs= score(d1, d2, s, 0, arr1[0], arr2[0]);
                    for(int i= 1; i<n; ++i){
                        int sc= score(d1, d2, s, i, arr1[i], arr2[i]);        
                        if(sc>bs){
                            bs = sc; 
                            bi = i;
                        }    
                    }
                    extrem[4*d1+2*d2+s]= bi;
                }
            }
        }
        int out = 0;
        for(int i=0; i<8; ++i){
            for(int j=0; j<8; ++j){
                out = max(out, score2(extrem[i], extrem[j], arr1, arr2) );
            }
        }
        
        return out;
        
    }
    
    
    int score(int d1, int d2, int s, int i, int v1, int v2){
        int out = 0;
        if(d1 ) out+= v1;
        else  out-=v1;
        if(d2) out+= v2;
        else out-=v2;
        if(s) out+= i;
        else out-=i;
        
        return out;
    }
    
    int score2(int i, int j, vector<int>& a1, vector<int>&a2){
        return abs(a1[i]-a1[j]) + abs(a2[i]-a2[j]) + abs(i-j);
    }
    
    
};
