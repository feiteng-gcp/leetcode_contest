class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
            int res = 0;
        int n = arr1.size();
        
        int s1 = arr1[0] + arr2[0],s2 = arr1[0] -arr2[0],s3=-arr1[0] - arr2[0],s4=-arr1[0] + arr2[0];
        for(int i = 1;i<n;++i)
        {
            int t1 = arr1[i] + arr2[i]-i,t2 = arr1[i] - arr2[i]-i,t3 = arr1[i] + arr2[i] + i,t4 = arr1[i]-arr2[i] + i;
            res = max(res,s1 - t1);
            res = max(res,s2 - t2);
            res = max(res,s3 + t3);
            res = max(res,s4 + t4);
            s1 =  max(s1,t1);
            s2 = max(s2,t2);
            s3 = max(s3,-t3);
            s4 = max(s4,-t4);
        }
        return res;
    }
};
