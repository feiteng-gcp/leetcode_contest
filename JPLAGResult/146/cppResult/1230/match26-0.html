class Solution {
public:
    int maxAbsValExpr(vector<int>& a1, vector<int>& a2) {
        int n = a1.size();
        vector<int>v1,v2,v3,v4;
        for(int i=0;i<n;i++)
        {
            v1.push_back(a1[i]+a2[i]+i);
            v2.push_back(a1[i]+a2[i]-i);
            v3.push_back(a1[i]-a2[i]+i);
            v4.push_back(a1[i]-a2[i]-i);
        }
        sort(v1.begin(),v1.end());
        sort(v2.begin(),v2.end());
        sort(v3.begin(),v3.end());
        sort(v4.begin(),v4.end());
        
        int ans = abs(v1[0] - v1[n-1]);
        ans = max(ans,abs(v2[0]-v2[n-1]));
        ans=max(ans,abs(v3[0]-v3[n-1]));
        ans=max(ans,abs(v4[0]-v4[n-1]));
        return ans;
    }
};
