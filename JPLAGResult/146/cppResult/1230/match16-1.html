class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int ans=0;
        int N=arr1.size();
        int t1=arr1[0]+arr2[0];
        int t2=-arr1[0]-arr2[0];

        int t3=arr1[0]-arr2[0];
        int t4=-arr1[0]+arr2[0];
        for(int i=1;i<N;i++){
            ans=max(ans,t1+i-arr1[i]-arr2[i]);
            ans=max(ans,t2+i+arr1[i]+arr2[i]);
            ans=max(ans,t3+i-arr1[i]+arr2[i]);
            ans=max(ans,t4+i+arr1[i]-arr2[i]);
            t1=max(arr1[i]+arr2[i]-i,t1);
            t2=max(-arr1[i]-arr2[i]-i,t2);
            t3=max(arr1[i]-arr2[i]-i,t3);
            t4=max(-arr1[i]+arr2[i]-i,t4);
        }
        return ans;
    }
};
