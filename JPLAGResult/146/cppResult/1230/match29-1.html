class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        vector<int> mpmp(arr1.size()), mppm(arr1.size()), pmmp(arr1.size()), pmpm(arr1.size());
        for (int i = 0; i < arr1.size(); i++) {
            mpmp[i] = arr1[i] + arr2[i] + i;
            mppm[i] = arr1[i] - arr2[i] + i;
            pmmp[i] = -arr1[i] + arr2[i] + i;
            pmpm[i] = -arr1[i] - arr2[i] + i;
        }
        sort(mpmp.begin(), mpmp.end());
        sort(mppm.begin(), mppm.end());
        sort(pmmp.begin(), pmmp.end());
        sort(pmpm.begin(), pmpm.end());
        return max(mpmp.back() - mpmp.front(), max(mppm.back() - mppm.front(), max(pmmp.back() - pmmp.front(), pmpm.back() - pmpm.front())));
    }
};
