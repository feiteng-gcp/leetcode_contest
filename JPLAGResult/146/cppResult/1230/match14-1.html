class Solution {
public:
    int maxAbsValExpr(vector<int>& A, vector<int>& B) {
        int L = A.size();
        
        int val[8] = {0};

        for (int i=0;i<L;++i) {
            val[0] = max(val[0],  A[i] + B[i] + i);
            val[1] = max(val[1],  A[i] + B[i] - i);
            val[2] = max(val[2],  A[i] - B[i] + i);
            val[3] = max(val[3],  A[i] - B[i] - i);
            val[4] = max(val[4], -A[i] + B[i] + i);
            val[5] = max(val[5], -A[i] + B[i] - i);
            val[6] = max(val[6], -A[i] - B[i] + i);
            val[7] = max(val[7], -A[i] - B[i] - i);
        }
        
        int best = 0;
        for (int i=0;i<8;++i) {
            if (val[i] + val[(i^7)] > best) {
                best = val[i] + val[(i^7)];
            }
        }
        return best;
        
    }
};
