class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {

        int n = arr1.size();
        for(int i = 0; i < n; i++){
            dp[0][i] = arr1[i] + arr2[i] + i;
            dp[1][i] = arr1[i] + arr2[i] - i;
            dp[2][i] = arr1[i] - arr2[i] + i;
            dp[3][i] = arr1[i] - arr2[i] - i;
        }
        int res = 0;
        for(int k = 0; k < 4; k++){
            int minn = dp[k][0];
            int maxx = dp[k][0];
            for(int i = 0; i < n; i++){
                minn = min(minn, dp[k][i]);
                maxx = max(maxx, dp[k][i]);
            }
            res = max(res, maxx - minn);
        }
        return res;
    }
private:
    int dp[4][40005];
};
