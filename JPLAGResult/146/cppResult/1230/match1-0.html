class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int ans = 0;
        int maxv = 10000000;
        for (int s = 0; s < (1 << 3); s++) {
            int minp = maxv;
            int maxp = -maxv;

            for (int i = 0; i < arr1.size(); i++) {
                int tmp = 0;
                if (s & 1)
                    tmp += i;
                else
                    tmp -= i;

                if (s & 2)
                    tmp += arr1[i];
                else
                    tmp -= arr1[i];

                if (s & 4)
                    tmp += arr2[i];
                else
                    tmp -= arr2[i];

                minp = min(minp, tmp);
                maxp = max(maxp, tmp);
            }

            ans = max(ans, maxp - minp);
        }
        return ans;
    }
};
