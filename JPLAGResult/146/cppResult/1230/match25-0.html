class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int len = arr1.size();
        vector<int> num1(len), num2(len), num3(len), num4(len), min1(len), min2(len), min3(len), min4(len);
        for (int i = 0; i < arr1.size(); i++) {
            num1[i] = i + arr1[i] + arr2[i];
            num2[i] = i + arr1[i] - arr2[i];
            num3[i] = i - arr1[i] + arr2[i];
            num4[i] = i - arr1[i] - arr2[i];
        }
        min1[0] = num1[0], min2[0] = num2[0], min3[0] = num3[0], min4[0] = num4[0];
        for (int i = 1 ; i < arr1.size(); i++) {
            min1[i] = min(min1[i - 1], num1[i]);
            min2[i] = min(min2[i - 1], num2[i]);
            min3[i] = min(min3[i - 1], num3[i]);
            min4[i] = min(min4[i - 1], num4[i]);
        }
        int sum = 0;
        for (int i = 1; i < arr1.size(); i++) {
            sum = max(sum, num1[i] - min1[i]);
            sum = max(sum, num2[i] - min2[i]);
            sum = max(sum, num3[i] - min3[i]);
            sum = max(sum, num4[i] - min4[i]);
        }
        return sum;
    }
};
