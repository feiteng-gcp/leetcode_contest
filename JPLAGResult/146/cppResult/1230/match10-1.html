class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size();
        int dp[4];
        for(int i = 0;i < 4;i++)
            dp[i] = 0;
        int ans = 0;
        for(int i = 1;i < arr1.size();i++){
            for(int k = 0;k < 4;k++){
                ans = max(ans,abs(arr1[i] - arr1[dp[k]]) + abs(arr2[i] - arr2[dp[k]]) + i - dp[k]);
            }
            if(dp[0] - i + arr1[i] - arr1[dp[0]] + arr2[i] - arr2[dp[0]] > 0)
                    dp[0] = i;
            if(dp[1] - i + arr1[i] - arr1[dp[1]] + arr2[dp[1]] - arr2[i] > 0)
                    dp[1] = i; 
            if(dp[2] - i + arr1[dp[2]] - arr1[i] + arr2[i] - arr2[dp[2]] > 0)
                    dp[2] = i;   
            if(dp[3] - i + arr1[dp[3]] - arr1[i] + arr2[dp[3]] - arr2[i] > 0)
                    dp[3] = i;                                               
        }
        return ans;
    }
};
