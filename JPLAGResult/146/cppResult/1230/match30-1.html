class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size();
        vector<int> tmp(n, 0);
        for(int i = 0; i < n; i++)
        {
            tmp[i] = arr1[i] + arr2[i];
        }
        
        int ans = helper(tmp);
        
        //tmp.resize(n, 0);
        for(int i = 0; i < n; i++)
        {
            tmp[i] = arr1[i] - arr2[i];
        }
        int cur = helper(tmp);
        ans = max(ans, cur);
        
        //tmp.resize(n, 0);
        for(int i = 0; i < n; i++)
        {
            tmp[i] = arr2[i] - arr1[i];
        }
        cur = helper(tmp);
        ans = max(ans, cur);
        //tmp.resize(n, 0);
        for(int i = 0; i < n; i++)
        {
            tmp[i] = -arr1[i] - arr2[i];
        }
        cur = helper(tmp);
        ans = max(ans, cur);
        return ans;
    }
    
private:
    int helper(vector<int>& tmp) {
        int ans = INT_MIN;
        stack<int> stk;
        int n = tmp.size();
        for(int i = 0; i < n; i++)
        {
            if(stk.empty())
            {
                stk.push(i);
                continue;
            }
            
            if(tmp[i] >= tmp[stk.top()])
            {
                ans = max(ans, tmp[i] - tmp[stk.top()] + i - stk.top());
            }
            else 
            {
                bool cand = false;
                while(!stk.empty() && tmp[stk.top()] - tmp[i] > i - stk.top())
                {
                    stk.pop();
                    cand = true;
                }
                if(cand)
                    stk.push(i);
            }
        }  
        return ans;
    }
};
