class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
       vector<vector<int>> a;
      for (int i = 0; i < arr1.size(); ++i) {
        a.push_back({arr1[i], arr2[i], i});
      }
      int n = arr1.size();
      int ans = 0;
      map<int, int> f; 
      for (int i = 0; i < n; ++i) {
        for (int j = 0; j < 8; ++j) {
          int sum = 0;
          for (int k = 0; k < 3; ++k) {
            int sgn = ((j >> k) & 1) ? 1 : -1;
            sum += sgn * a[i][k];
           
          }
          if (f.count(7 ^ j)) {
            ans = max(ans, sum + f[7 ^ j]);
          }
        }
        for (int j = 0; j < 8; ++j) {
          int sum = 0;
          for (int k = 0; k < 3; ++k) {
            int sgn = ((j >> k) & 1) ? 1 : -1;
            sum += sgn * a[i][k];
           
          }
          if (!f.count(j)) {
            f[j] = sum;
          }
          f[j] = max(f[j], sum);
        }
      }
      return ans;
    }
};
