class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size();
        int ans = 0;
        for (int b = 0; b < 8; b++) {
            int mi = 1e9, mx = -1e9;

            for (int i = 0; i < n; i++) {
                int t = 0;
                if (b & 1) t += arr1[i]; else t -= arr1[i];
                if (b & 2) t += arr2[i]; else t -= arr2[i];
                if (b & 4) t += i; else t -= i;
                mi = min(mi, t);
                mx = max(mx, t);
            }
            ans = max(ans, mx-mi);
        }
        return ans;
    }
};
