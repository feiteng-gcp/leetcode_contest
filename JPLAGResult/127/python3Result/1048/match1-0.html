class Solution:
    def clumsy(self, N: int) -> int:
        if N <= 2:
            return N
        
        if N == 3:
            return N * 2
        
        if N == 4:
            return N * 3 // 2 + 1
        
        res = N * (N - 1) // (N - 2) + (N - 3)
        N -= 4
        while N >= 5:
            res -= N * (N - 1) // (N - 2) 
            res += (N - 3)
            N -= 4
        
        if N == 4:
            res -= 4 * 3 // 2
            res += 1
        if N == 3:
            res -= 3 * 2
        if N <= 2:
            res -= N
        
        return res
