class Solution:
    def clumsy(self, N: int) -> int:
        if N==1:
            return 1
        if N==2:
            return 2
        if N==3:
            return 6
        ret = N*(N-1)//(N-2)+(N-3)
        # print(ret)
        N = N-4
        def process4(X):
            if X==1:
                return -1
            if X==2:
                return -2
            if X==3:
                return -6
            return -(N*(N-1)//(N-2))+(N-3)
        while N>0:
            ret+=process4(N)
            # print(ret,N)
            N = N-4
        return ret
