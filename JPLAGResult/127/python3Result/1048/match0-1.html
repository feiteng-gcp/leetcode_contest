class Solution:
    def clumsy(self, N: int) -> int:
        if N == 3:
            return 6
        if N == 2:
            return 2
        if N == 1:
            return 1
        curr = N
        nos = []
        while curr - 3 > 0:
            temp = curr*(curr-1)//(curr-2)
            nos.append(temp)
            nos.append(curr-3)
            curr = curr - 4
        if curr == 3:
            nos.append(curr*(curr-1)//(curr-2))
        elif curr == 2:
            nos.append(curr*(curr-1))
        elif curr == 1:
            nos.append(curr)
        sign = 1
        res = nos[0]
        for i in range(1,len(nos)):
            res = res + nos[i]*sign
            sign = -sign
        #print(nos)
        return res
