class Solution:
    
    # N = 4
    # 4 * 3 / 2 + 1
    def clumsy(self, N: int) -> int:
        res, tempRes = 0, N
        sign = 1
        op = 0 #[*, /, +, -]
        for n in range(N-1, 0, -1): # n-1 -> 1
            if op == 0: tempRes *= n
            if op == 1: tempRes 
//= n
            if op == 2: # + 
                res = res + tempRes * sign
                tempRes = n
                sign = 1
            if op == 3: # -
                res = res + tempRes * sign
                tempRes = n
                sign = -1
            op = (op + 1) % 4
        res = res + tempRes * sign
        return res
