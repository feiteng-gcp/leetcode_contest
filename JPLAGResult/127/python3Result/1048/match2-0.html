class Solution:
    def clumsy(self, N: int) -> int:
        neg, i = False, N
        result = 0
        
        while i > 0:
            if i > 3:
                if not neg:
                    result += (i * (i-1)) // (i-2)
                    neg = True
                else:
                    result -= (i * (i-1)) // (i-2)
                result += (i-3)
            elif i == 3:
                if not neg:
                    result += (3 * 2 // 1)
                    neg = True
                else:
                    result -= (3 * 2 // 1)
            elif i == 2:
                if not neg:
                    result += (2*1)
                    neg = True
                else:
                    result -= (2*1)
            elif i == 1:
                if not neg:
                    result += 1
                    neg = True
                else:
                    result -= 1
            i -= 4
        return result
