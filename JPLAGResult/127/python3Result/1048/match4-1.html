class Solution:
    
def clumsy(self, N: int) -> int:
        if N == 1:
            return 1
        if N == 2:
            return 2
        if N == 3:
            return 6
        if N == 4:
            return 7
        else:
            result = N * (N-1) // (N-2)
            for i in range(N-4, 0, -4):
                result -= self.muldiv(i)
            for i in range(N-3, 0, -4):
                result += i
            return result
    def muldiv(self, N):
        if N == 1:
            return 1
        if N == 2:
            return 2
        else:
            return N * (N-1) // (N-2)
