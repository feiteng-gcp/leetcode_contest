class Solution:
    def largestSumAfterKNegations(self, A: List[int], K: int) -> int:
        num = 0
        for x in A:
            num = num + 1 if x < 0 else num
        if K <= num:
            A.sort()
            for i in range(K):
                A[i] = -A[i]
            return sum(A)
        else:
            A 
= [abs(x) for x in A]
            if (K - num) % 2 == 0:
                return sum(A)
            else:
                return sum(A) - 2 * min(A)
