class Solution:
    def largestSumAfterKNegations(self, A: List[int], K: int) -> int:
        A = sorted(A)
        for idx, num in enumerate(A):
            if K == 0:
                break
            if num < 0:
                A[idx] = -num
                K -= 1
        A = sorted(A)
        if K % 2:
            A[0] = -A[0]
        return sum(A)
