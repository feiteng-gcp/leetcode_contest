class Solution:
    def largestSumAfterKNegations(self, A: List[int], K: int) -> int:
        A.sort()
        n = len(A)
        if n == 0:
            return 0
        for i in range(0, n):
            if K == 0:
                break
            if A[i] < 0:
                A[i] = -A[i]
                K -= 1
        if K == 0:
            return sum(A)
        A.sort()
        if K % 2 == 1:
            A[0] = -A[0]
        return sum(A)
