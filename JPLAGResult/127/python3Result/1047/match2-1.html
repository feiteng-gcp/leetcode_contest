"""
Idea: Sort numbers.
Choose the most negative numbers. Convert them to positives.
If there are still remaining conversions:
if rem % 2==0 -> no effect
else:
Find the number with the smallest abs value. Convert that to a neg. return ans

"""


class Solution:
    def largestSumAfterKNegations(self, A: List[int], k: int) -> int:
        A.sort()
        i = 0
        for i in range(len
(A)):
            if k == 0:
                break
            if A[i] < 0:
                A[i] = -A[i]
                k -= 1
        
        if k % 2 == 1:
            bi = min(range(len(A)), key=lambda i:abs(A[i]))
            A[bi] = -A[bi]
        return sum(A)
