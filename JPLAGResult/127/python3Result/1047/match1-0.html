class Solution:
    def largestSumAfterKNegations(self, A: List[int], k: int) -> int:
        
        sorted_a = sorted(A)
        la = len(A)
        
        num_negative = 0
        for x in sorted_a:
            if x < 0:
                num_negative += 1
                
        if k <= num_negative:
            cur_sum = 0
            for i in range(la):
                if i < k:
                    cur_sum -= sorted_a[i]
                else:
                    cur_sum += sorted_a[i]
            return cur_sum
        
        else:
            swap = (k - num_negative) % 2 #1 -> swap first non negative number.
            new_arr = [abs(x) for x in sorted_a]
            if swap == 0:
                return sum(new_arr)
            else:
                return sum(new_arr) - min(new_arr) * 2
