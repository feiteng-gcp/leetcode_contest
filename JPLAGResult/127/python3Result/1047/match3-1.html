class Solution:
    def largestSumAfterKNegations(self, A: List[int], K: int) -> int:
        A.sort()
        
for i in range(len(A)):
            if K==0:
                break
            if A[i]<0:
                A[i]=-A[i]
                K-=1
            else:
                if K%2==0:
                    break
                else:
                    if i-1>=0 and A[i-1]<A[i]:
                        A[i-1]=-A[i-1]
                    else:
                        A[i]=-A[i]
                    break
        return sum(A)
