class Solution:
    def largestSumAfterKNegations(self, A: List[int], K: int) -> int:
        A.sort()
        result = 0
        neg = list(filter(lambda x: x < 0, A))
        pos = list(map(lambda x: abs(x), A))
        if K <= len(neg):
            for i in range(len(A)):
                if i < K:
                    result -= A[i]
                else:
                    result += A[i]
            return result
        else:
            if 0 in A:
                for i in A:
                    result += abs(i)
                return result
            else:
                for i in A:
                    result += abs(i)
                if (K - len(neg))%2 == 0:
                    return result
                else:
                    result -= 2*min(pos)
                    return result
