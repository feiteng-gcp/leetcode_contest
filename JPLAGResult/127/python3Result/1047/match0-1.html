class Solution:
    def largestSumAfterKNegations(self, A: List[int], K: int) -> int:
        A=sorted(A)
        n=len(A)
        for i in range(n):
            if K>0 and A[i]<0:
                A[i]=-A[i]
                K
-=1
        A=sorted(A)
        if K%2==1:
            A[0]=-A[0]
        return sum(A)
