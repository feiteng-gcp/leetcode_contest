# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        n = len(preorder)
        if n == 0:
            return None
        val = preorder[0]
        res = TreeNode(val)
        l = 1
        r = 1
        while r < n:
            if preorder[r] >= val:
                break
            r += 1
        res.left = self.bstFromPreorder(preorder[l:r])
        res.right = self.bstFromPreorder(preorder[r:n])
        return res
