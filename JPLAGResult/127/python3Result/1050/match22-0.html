# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        if len(preorder) == 0:
            return None
        val = preorder[0]
        idx = next((i for i in range(1, len(preorder)) if preorder[i] > val), len(preorder))
        ret = TreeNode(val)
        ret.left = self.bstFromPreorder(preorder[1:idx])
        ret.right = self.bstFromPreorder(preorder[idx:])
        return ret
