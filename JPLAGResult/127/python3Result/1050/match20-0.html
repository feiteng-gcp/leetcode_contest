import bisect


class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        if not preorder:
            return None
        x = TreeNode(preorder[0])
        m = bisect.bisect_left(preorder[1:], preorder[0])
        x.left = self.bstFromPreorder(preorder[1:m + 1])
        x.right = self.bstFromPreorder(preorder[m + 1:])
        return x
