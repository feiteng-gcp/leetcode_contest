# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        if len(preorder) == 0:
            return None
        node = TreeNode(preorder[0])
        x = 1
        while x < len(preorder) and preorder[x] < preorder[0]:
            x += 1
        node.left = self.bstFromPreorder(preorder[1:x])
        node.right = self.bstFromPreorder(preorder[x:])
        return node
