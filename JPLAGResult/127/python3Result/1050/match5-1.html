# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        if len(preorder) == 0:
            return None
        
        root = TreeNode(preorder[0])
        
        
        flp = 1
        while flp < len(preorder) and preorder[flp] < preorder[0]:
            flp += 1
        root.left = self.bstFromPreorder(preorder[1:flp])
        root.right = self.bstFromPreorder(preorder[flp:])
        return root
