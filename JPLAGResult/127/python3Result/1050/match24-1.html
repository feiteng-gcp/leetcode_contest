# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def bstFromPreorder(self, preorder) -> TreeNode:
        if len(preorder) == 0:
            return None
        root = TreeNode(preorder[0])
        root
.left = self.bstFromPreorder([x for x in preorder if x < preorder[0]])
        root.right = self.bstFromPreorder([x for x in preorder if x > preorder[0]])
        return root
