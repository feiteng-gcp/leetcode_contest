# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        node = TreeNode(preorder[0])
        if len(preorder) == 1:
            return node
        greater = None
        for i, n in enumerate(preorder):
            if n > preorder[0]:
                greater = i
                break
        lesser = None
        for i, n in enumerate(preorder):
            if n < preorder[0]:
                lesser = i
                break
        if greater:
            if lesser:
                node.left = self.bstFromPreorder(preorder[lesser:greater])
                node.right = self.bstFromPreorder(preorder[greater:])
            else:
                node.right = self.bstFromPreorder(preorder[greater:])
        elif lesser:
                node.left = self.bstFromPreorder(preorder[lesser:])
        return node
