# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        if len(preorder)==0:
            return None
        pivot = preorder[0]
        root = TreeNode(pivot)
        root.left = self.bstFromPreorder([x for x in preorder if x <pivot])
        root.right = self.bstFromPreorder([x for x in preorder if x >pivot])
        return root
