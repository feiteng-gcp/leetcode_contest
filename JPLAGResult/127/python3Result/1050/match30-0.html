# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        if not preorder:
            return None
        
        root = TreeNode(preorder.pop(0))
        if not preorder:
            return root
        
        center = 0
        while center < len(preorder):
            if preorder[center] < root.val:
                center += 1
            else:
                break
        
        root.left = self.bstFromPreorder(preorder[:center])
        root.right = self.bstFromPreorder(preorder[center:])
        
        return root
