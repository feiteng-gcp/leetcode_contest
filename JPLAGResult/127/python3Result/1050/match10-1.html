# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        def helper(i, j):
            if i == j:
                return None
            root = TreeNode(preorder[i])
            
            k = i + 1
            while k < j:
                if preorder[k] > preorder[i]:
                    break
                k += 1
                
            root.left = helper(i+1, k)
            root.right = helper(k, j)
            return root
            
            
        return helper(0, len(preorder))
