# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        if not preorder:
            return None
        n = TreeNode(preorder[0])
        pivot = preorder[0]
        for i in range(1, len(preorder) + 1):
            if i < len(preorder) and preorder[i] > pivot:
                
break
        n.left = self.bstFromPreorder(preorder[1: i])
        n.right = self.bstFromPreorder(preorder[i:])
        return n
