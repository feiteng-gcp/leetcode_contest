# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder: List[int]) -> TreeNode:
        def build(l, r):
            if l >= r:
                return None
            head = TreeNode(preorder[l])
            mid = l + 1
            while mid < r:
                if preorder[mid] > preorder[l]:
                    break
                mid += 1
            head.left = build(l + 1, mid)
            head.right = build(mid, r)
            return head
        return build(0, len(preorder))
