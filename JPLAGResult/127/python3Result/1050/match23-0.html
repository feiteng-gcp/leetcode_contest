# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def bstFromPreorder(self, preorder):
        if not preorder:
            return None
        root=TreeNode(preorder[0])
        l=len(preorder)
        p1=l
        for i in range(1,l):
            if preorder[i]>preorder[0]:
                p1=i
                break
        root.left=self.bstFromPreorder(preorder[1:p1])
        root.right=self.bstFromPreorder(preorder[p1:l])
        return root
