class Solution:
    def minDominoRotations(self, a, b) -> int:
        setA=[set() for i in range(7)]
        setB=[set() for i in range
(7)]
        n = len(a)
        for i in range(n):
            setA[a[i]].add(i)
            setB[b[i]].add(i)
        best=float('inf')
        for i in range(1,7):
            c=setA[i]|setB[i]
            if len(c)==n:
                best=min(best, n-len(setA[i]), n-len(setB[i]))
        return best if best<=n else -1
