class Solution:
    def minDominoRotations(self, A: List[int], B: List[int]) -> int:

        minSwaps = float('inf')
        for chosen in [A[0], B[0]]:
            minSwaps = min(self.countSwapsA(chosen, A, B), minSwaps)
        
        A,B = B,A
        for chosen in [A[0], B[0]]:
            minSwaps = min(self.countSwapsA(chosen, A, B), minSwaps)
        
        if minSwaps == float('inf'):
            return -1
        
        
return minSwaps
        
        
    def countSwapsA(self, num, A, B):
        swaps = 0
        for i in range(len(A)):
            if A[i] != num:
                # need to swap
                if B[i] != num:
                    return float('inf')
                else:
                    swaps += 1
        return swaps
