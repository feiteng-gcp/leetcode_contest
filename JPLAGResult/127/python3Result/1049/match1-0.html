class Solution:
    def minDominoRotations(self, A: List[int], B: List[int]) -> int:
        N=len(A)
        setN=set(list(range(N)))
        Amap={i:set() for i in range(1,7)}
        Bmap={i:set() for i in range(1,7)}
        for i,a in enumerate(A):
            Amap[a].add(i)
        for i,b in enumerate(B):
            Bmap[b].add(i)
        ans=float('inf')
        for n in range(1,7):
            An,Bn=Amap[n],Bmap[n]
            if An|Bn==setN:
                ans=min(len(setN-An),len(setN-Bn))
            continue
        return ans if ans<float('inf') else -1
