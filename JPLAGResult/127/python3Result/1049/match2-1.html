class Solution:
    def minDominoRotations(self, A: List[int], B: List[int]) -> int:
        minStep = float("inf")
        N = len(A)
        # let A all be i
        for i in range(1, 7):
            totalSwap = 0
            fail = False
            for j in range(N):
                # no swap
                if A[j] == i: continue
                
                if B[j] == i: 
                    totalSwap += 1
                else: 
                    fail 
= True
                    break
            if not fail: minStep = min(minStep, totalSwap)
                    
        for i in range(1, 7):
            totalSwap = 0
            fail = False
            for j in range(N):
                # no swap
                if B[j] == i: continue
                
                if A[j] == i: 
                    totalSwap += 1
                else: 
                    fail = True
                    break
            if not fail: minStep = min(minStep, totalSwap)
        
        if minStep != float("inf"): return minStep
        else: return -1
