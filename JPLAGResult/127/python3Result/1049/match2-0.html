class Solution:
    def minDominoRotations(self, a: List[int], b: List[int]) -> int:
        n = len(a)
        res = 1000000
        for k in range(1, 7):
            t = 0
            flag = True
            for i in range(n):
                if (a[i] == k or b[i] == k):
                    if b[i] == k and a[i] != k:
                        t += 1
                else: 
                    flag = False
                    break
            if flag:
                res = min(res, t)
                
        for k in range(1, 7):
            t = 0
            flag = True
            for i in range(n):
                if (a[i] == k or b[i] == k):
                    if a[i] == k and b[i] != k:
                        t += 1
                else: 
                    flag = False
                    break
            if flag:
                res = min(res, t)
        if res == 1000000:
            return -1
        else:
            return res
