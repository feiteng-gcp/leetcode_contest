class Solution:
    def minDominoRotations(self, A: List[int], B: List[int]) -> int:
        d = collections.defaultdict(set)
        N = len(A)
        for i in range(0, N):
            d[A[i]].add(i)
            d[B[i]].add(i)
        res = float('inf')
        for n in range(1, 7):
            if len(d[n]) == N:
                res = min(res, min(N - A.count(n), N - B.count(n)))
        return -1 if res == float('inf') else res
