class Solution:
    def minDominoRotations(self, A: List[int], B: List[int]) -> int:
        n = len(A)
        res = n
        
        for val in range(1, 7):
            countA = 0
            countB = 0
            flag = True
            for i in range(n):
                if val not in [A[i], B[i]]:
                    flag = False
                    break
                if A[i] == val:
                    countA += 1
                if B[i] == val:
                    countB += 1
            if flag:
                res = min(res, n - countA, n - countB)
                
        if res == n:
            return -1
        else:
            return res
