class Solution:
    def minDominoRotations(self, A: List[int], B: List[int]) -> int:
        min_num = -1
        for k in range(1, 7):
            num = self.getNum(A, B, k)
            if num != -1 and (min_num == -1 or num < min_num):
                min_num = num
            num = self.getNum(B, A, k)
            if num != -1 and (min_num == -1 or num < min_num):
                min_num = num
        return min_num
        
    def getNum(self, A, B, k):
        num = 0
        for i in range(len(A)):
            if A[i] != k:
                if B[i] != k:
                    return -1
                num += 1
        return num
