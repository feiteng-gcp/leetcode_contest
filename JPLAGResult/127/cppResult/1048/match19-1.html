class Solution {

public:
    int helper(int N) {
        if (N == 1) {
            return -1;
        }
        if (N == 2) {
            return -2 * 1;
        }
        if (N == 3) {
            return -3 * 2 / 1;
        }
        return -N * (N - 1) / (N - 2) + (N - 3);
    }
    int clumsy(int N) {

        if (N == 1) {
            return 1;
        }
        if (N == 2) {
            return 2 * 1;
        }
        if (N == 3) {
            return 3 * 2 / 1;
        }
        if (N == 4) {
            return 4 * 3 / 2 + 1;
        }
        long long ans = N * (N - 1) / (N - 2) + (N - 3);
        for (int i = N - 4; i > 0; i -= 4) {
            ans += helper(i);
        }
        return (int)ans;
    }
};
