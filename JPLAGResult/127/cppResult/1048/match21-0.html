class Solution {
public:
    int clumsy(int N) {
        int groups = N / 4;
        int remainder = N % 4;
        int ans = 0;
        for(int i = 0; i < groups; ++i) {
            int temp = N * (N-1) / (N-2);
            if(i == 0) {
                ans = temp;
            }else {
                ans -= temp;
            }
            ans += N-3;
            N -= 4;
        }
        if(remainder == 3) {
            ans -= N * (N-1) / (N-2);
        }else if(remainder == 2) {
            ans -= N * (N-1);
        }else if(remainder == 1) {
            ans -= N;
        }
        if(groups == 0) ans = -ans;
        return ans;
    }
};
