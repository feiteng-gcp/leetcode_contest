class Solution {
public:

    long long clumsy2(int N) {
        if (N == 0)
            return 0;
        if (N == 1)
            return 1;
        if (N == 2)
            return 2 - 1;
        if (N == 3)
            return 3 - 2 * 1;
        long long ans = N - (N - 1) * (N - 2) / (N - 3);
        return ans + clumsy2(N - 4);
    }
    
    int clumsy(int N) {
        if (N == 1)
            return 1;
        if (N == 2)
            return 2;
        if (N == 3)
            return 3 * 2 / 1;
        return N * (N - 1) / (N - 2) + clumsy2(N - 3);        
    }
};
