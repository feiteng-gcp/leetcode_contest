class Solution {

    int solve(int n) {
        if (n == 0) {
            return 0;
        } else if (n == 1) {
            return -1;
        } else if (n == 2) {
            return -2;
        } else if (n == 3) {
            return -6;
        } else {
            return -(n * (n - 1) / (n - 2)) + (n - 3) + solve(n - 4);
        }
    }
public:
    int clumsy(int n) {
        if (n == 1) {
            return 1;
        } else if (n == 2) {
            return 2;
        } else if (n == 3) {
            return 6;
        } else {
            return n * (n - 1) / (n - 2) + n - 3 + solve(n - 4);
        }
    }
};
