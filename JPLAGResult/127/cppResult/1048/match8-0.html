class Solution {
public:
    int clumsy(int n) {
        long long ret = 0;
        int sgn = 1;
        while (n >= 4) {
            ret += (n * (n - 1) / (n - 2)) * sgn + (n - 3);
            sgn = -1;
            n -= 4;
        }
        if (n == 1) {
            ret += n * sgn;
        } else if (n == 2) {
            ret += n * (n - 1) * sgn;
        } else if (n == 3) {
            ret += n * (n - 1) / (n - 2) * sgn;
        }
        return ret;
    }
};
