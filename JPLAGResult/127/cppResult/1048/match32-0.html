class Solution {
public:
    int clumsy(int N) {
        long r = 0;
        if (N >= 4) {
            r = N * (N - 1) / (N - 2) + (N - 3);
            for (N -= 4; N >= 4; N -= 4) {
                r = r - N * (N - 1) / (N - 2) + (N - 3);
            }
            if (N == 3) {
                r = r - 3 * 2 / 1;
            }
            else if (N == 2) {
                r = r - 2 * 1;
            }
            else if (N == 1) {
                r = r - 1;
            }
        }
        else if (N == 3) {
            r = 3 * 2 / 1;
        }
        else if (N == 2) {
            r = 2 * 1;
        }
        else if (N == 1) {
            r = 1;
        }
        return r;
    }
};
