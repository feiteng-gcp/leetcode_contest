class Solution {
    int dp(int N){
        if(N==0) return 0;
        if(N==1) return -1;
        if(N==2) return -2;
        if(N==3) return -6;
        return -N*(N-1)/(N-2)+N-3+dp(N-4);
    }
public:
    int clumsy(int N) {
        if(N==1) return 1;
        if(N==2) return 2;
        if(N==3) return 6;
        return N*(N-1)/(N-2)+N-3+dp(N-4);
    }
};
