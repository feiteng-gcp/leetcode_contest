class Solution {
public:
    int clumsy(int N) {
        if(N>2){
            return clumsy_helper(N) + N*(N-1)/(N-2)*2;
        } else if(N==2){
            return 2;
        } else if(N==1){
            return 1;
        } else {
            return 0;
        }
    }
    
    int clumsy_helper(int N){
        if(N>2){
            return -N*(N-1)/(N-2)+(N-3) + clumsy_helper(N-4);
        } else if(N==2){
            return -2;
        } else if(N==1){
            return -1;
        } else {
            return 0;
        }
    }
};
