class Solution {
public:
    int clumsy(int N) {
        if(N==1)
            return 1;
        if(N==2)
            return 2;
        if(N==3)
            return 6;
        if(N==4)
            return 7;
        int res = N * (N-1) / (N-2);
        int n = N - 3;
        if(n%4 == 0)
            res -= 2;
        else if(n%4 == 1 && n>=5)
            res+=0;
        else
            res+=1;
        return res;
    }
};
