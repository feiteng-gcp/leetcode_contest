class Solution {
public:
    int clumsy(int N) {
        if (N == 1) return 1;
        else if (N == 2) return 2;
        else if (N == 3) return 6;
        else if (N == 4) return 7;
        int ans = N + 1;
        int p = (N - 3) % 4;
        if (p == 0) ans -= 2;
        else if (p == 1) ans = ans;
        else if (p == 3 || p == 2) ans++;
        return ans;
    }
};
