class Solution {
public:
    int clumsy(int N) {
        int64_t k = N;
        int64_t r = 0;
        int64_t s = 1;
        while (k>=4) {
            r += s*(k)*(k-1)/(k-2)+(k-3); 

            k-=4;
            s = -1;
        }
        switch (k) {
            case 3:
                r+=s*3*2/1;
                break;
            case 2:
                r+=s*2*1;
                break;
            case 1:
                r+=s;
                break;
        }
        return r;
    }
};
