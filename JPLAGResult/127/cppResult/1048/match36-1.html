class Solution {
    int64_t tripletResult(int64_t x) {
        if (x == 1)
            return 1;
        else if (x == 2)
            return 2;
        else 
            return x * (x - 1) / (x - 2);
    }

public:
    int clumsy(int n) {
        if (n < 1)
            return 0;
        else if (n == 1)
            return 1;
        else if (n == 2)
            return 2;
        else if (n == 3)
            return 6;
        int64_t result = 0;
        bool first = true;
        while (n > 0) {
            int64_t now = tripletResult(n);
            if (first)
                first = false;
            else
                now = -now;
            result += now;
            n -= 3;
            if (n > 0) 
                result += n--;
        }
        return result;
    }
};
