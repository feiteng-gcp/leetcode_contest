class Solution {
public:
    int clumsy(int N) {
        if (N == 1) return 1;
        else if (N == 2) return 2;
        else if (N == 3) return 6;
        else {
             int res = N * (N - 1) / (N - 2) + N - 3;
             N -= 4;
             while(N >= 4) {
                 int temp = N * (N - 1) / (N - 2) - N + 3;
                 res -= temp; 
                 N -= 4;
             }
             if (N == 0) return res;
             else if (N == 1) return res - 1;
             else if (N == 2) return res - 2;
             else if (N == 3) return res - 6;
        }
        return -1;
    }
};
