class Solution {
public:
    int clumsy(int N) {
        int sum = 0, product = N--;

        for (int state = 0; N > 0; --N, state = (state + 1) % 4) {
            if (state == 0) {
                product *= N;
            } else if (state == 1) {
                product /= N;
            } else if (state == 2) {
                sum += product;
                product = N;
            } else {
                sum += product;
                product = -N;
            }
        }
        return sum + product;
    }
};
