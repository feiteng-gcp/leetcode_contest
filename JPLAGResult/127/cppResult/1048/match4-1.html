class Solution {
public:
	int clumsy(int N) {

		if (N == 1) return 1;//1
		if (N == 2) return 2;//2*1
		if (N == 3) return 6;//3*2/1;

		int sum;
		if (N >= 4)
		{

			sum = (N)*(N - 1) / (N - 2) + (N - 3);//4*3/2+1;
			N = N - 4;
			while (N >= 4)
			{
				sum -= (N)*(N - 1) / (N - 2) - (N - 3);
				N -= 4;
			}

			if (N == 3)
				sum -= 6;
			if (N == 2)
				sum -= 2;
			if (N == 1)
				sum -= 1;
		}

		return sum;
	}
};
