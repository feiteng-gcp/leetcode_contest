class Solution {
public:
	int clumsy(int N) {
		int result = 0;
		int prev = N--;
		int sign = 0;
		while (N) {

			if (sign == 0) prev *= N;
			if (sign == 1) prev /= N;
			if (sign == 2) {
				result += prev; prev = N;
			}
			if (sign == 3) {
				result += prev; prev = 0 - N;
			}
			sign = (sign + 1) % 4; N--;
		}
		return result+prev;
	}
};
