class Solution {
    int ans;
public:
    int clumsy(int N) {
        if (N == 1) {
            return 1;
        }
        if (N == 2) {
            return 2;
        }
        if (N == 3) {
            return 6;
        }
        
        ans = 0;
        for (int i = N; i > 3; i -= 4) {
            if (i == N) {
                ans += i * (i - 1) / (i - 2) + (i - 3);
            } else {
                ans += -i * (i - 1) / (i - 2) + (i - 3);
            }
        }
        
        N %= 4;
        if (N == 1) {
            return ans - 1;
        }
        if (N == 2) {
            return ans - 2;
        }
        if (N == 3) {
            return ans - 6;
        }
        return ans;
    }
};
