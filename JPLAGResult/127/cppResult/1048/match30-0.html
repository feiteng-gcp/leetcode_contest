class Solution {
public:
    int clumsy(int N) {
        if (N==1) return 1;
        if (N==2) return 2;
        if (N==3) return 6;
        int cnt=0;
        int ans=N*(N-1)/(N-2);
        for (int i=N-3;i>=1;){
            if (cnt&1){
                if (i-2>=1){
                    ans-=i*(i-1)/(i-2);
                    i-=3;
                }
                else if (i-1>=1){
                    ans-=i*(i-1);
                    i-=2;
                }
                else{
                    ans-=i;
                    i-=1;
                }
            }
            else{
                ans+=i;
                i-=1;
            }
            cnt++;
        }
        return ans;
    }
};
