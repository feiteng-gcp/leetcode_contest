class Solution {
public:
    int clumsy(int N) {
        int sum = 0;
        int flag = 0;
        while(N >= 4){
            
            auto y = - N * (N-1) /(N-2) +(N-3);
            if(flag ==0 ){
                sum+=N * (N-1) /(N-2);
                sum+=N * (N-1) /(N-2);
                flag+=1;
            }
            sum+=y;

            N-=4;
        }
        if(N == 3){
            if(flag ==0 ) sum += (N) * (N-1) / (N-2);
            else sum -= (N) * (N-1) / (N-2);
        }
        if(N == 2){
            if(flag ==0 ) sum += (N) * (N-1);
            else sum -= (N) * (N-1);
        }
        if(N == 1){
            if(flag ==0 ) sum += (N);
            else sum -= (N);
        }
        return sum;
    }
};
