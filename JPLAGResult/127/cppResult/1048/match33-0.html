class Solution {
public:
    long long pclumsy(long long N) {
        if (N == 0) return 0;
        if (N == 1) return N;
        if (N == 2) return N - (N - 1);
        if (N == 3) return N - (N - 1) * (N - 2);
        if (N == 4) return N - (N - 1) * (N - 2) / (N - 3);
        return N - (N - 1) * (N - 2) / (N - 3) + pclumsy(N - 4);
    }
    int clumsy(int N) {
        if (N == 0) return 0;
        if (N == 1) return N;
        if (N == 2) return N * (N - 1);
        if (N == 3) return N * (N - 1) / (N - 2);
        long long n = N;
        return n * (n - 1) / (n - 2) + pclumsy(n - 3);
    }
};
