class Solution {
public:
    int clumsy(int n) {
        int res=0;
        int dn=n-4;
        while(dn-3>=1){
            res-=dn*(dn-1)/(dn-2);
            res+=dn-3;

            dn-=4;
        }
        if(dn>=3){
            res-=dn*(dn-1)/(dn-2);
        }
        else if(dn>=2){
            res-=dn*(dn-1);
        }
        else if(dn>=1){
            res-=dn;
        }
        if(n>=3){
            res+=n*(n-1)/(n-2);

            res+=n-3;
        }
        else if(n>=2){
            res+=n*(n-1);
        }
        else if(n>=1){
            res+=n;
        }
        return res;
    }
};
