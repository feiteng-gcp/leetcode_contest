class Solution {
public:
    int clumsy(int N) {
        if (N == 0) return 0;
        if (N == 1) return 1;
        if (N == 2) return 2;
        if (N == 3) return 6;
        return clumsy2(N) + (N * (N-1) / (N-2)) * 2;
    }
    
    int clumsy2(int N) {
        if (N == 0) return 0;
        if (N == 1) return -1;
        if (N == 2) return -2;
        if (N == 3) return -6;
        return -N * (N-1) / (N-2) + (N-3) + clumsy2(N-4);
    }
};
