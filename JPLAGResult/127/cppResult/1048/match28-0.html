class Solution {
public:
    int clumsy(int N) {
        vector< int > table ; 
        vector< int > add ; 
        while( N ){
            int sum = 0 ; 
            if( N >= 3 ){
                sum +=( N )*(N-1)/(N-2) ;
                N-=3;
            }
            else if( N>=2 ){
                sum += N*(N-1) ; 
                N-=2;
            }
            else{
                sum += N ; 
                N-=1;
            }
            table.push_back( sum ) ; 
            if( N ) {
                add.push_back(N) ; 
                N--;
            }
        }
        int ans = table[0] ;
        if( add.size() )
            ans += add[0] ; 
        for( int i=1 ; i< table.size() ; i++){
            ans -= table[i] ; 
            if( i < add.size() )
                ans += add[i] ; 
        }
        return ans ;
        
    }
};
