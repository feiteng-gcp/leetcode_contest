class Solution {
public:
    int clumsy(int N) {
        if (N == 1) return 1;
        if (N == 2) return 2;
        if (N == 3) return 6;  
        int cycle = N / 4;
        int res = 0;
        for (int i = 0; i < cycle; ++i){
            res += i == 0? (N - i * 4) * (N - i * 4 - 1)/ (N - i * 4 - 2) : - (N - i * 4) * (N - i * 4 - 1)/ (N - i * 4 - 2);
            res += (N - i * 4 - 3);
        }
        int mod = N % 4;
        if (mod == 1) res += -1;
        if (mod == 2) res += -2;
        if (mod == 3) res += -6;
        return res;
    }
};
