class Solution {
public:
    int clumsy(int N) {
        if (N==1) return 1;
        if (N==2) return 2;
        if (N==3) return 6;
        int i = N;
        long long res = i*(i-1)/(i-2) + (i-3);
        i -= 4;
        while (i >= 4) {
            res += -i*(i-1)/(i-2) + (i-3);
            i -= 4;
        }
        if (i == 3) res += -3*2/1;
        if (i == 2) res += -2*1;
        if (i == 1) res += -1;
        return (int)res;
    }
};
