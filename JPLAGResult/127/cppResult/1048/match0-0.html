class Solution {
public:
    int clumsy(int N) {
        int cur=N;
        int i=N;i--;
        int ans=0;
        while(i>=1)
        {
            
            int sign=(N-1-i)%4;
  //          cout<<i<<" "<<sign<<" "<<cur<<" "<<ans<<endl;
            if(sign==0)cur*=i;
            if(sign==1)cur/=i;
            if(sign==2){
                ans+=cur;cur=i;
            }
            if(sign==3){
                ans+=cur;cur=-i;
            }
            i--;
            
        }
        ans+=cur;
        return ans;
    }
};
