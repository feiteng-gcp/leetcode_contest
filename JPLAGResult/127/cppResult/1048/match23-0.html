class Solution {
public:
    int clumsy(int N) {
        if (N == 1) {
            return 1;
        } else if (N == 2) {
            return 2;
        } else if (N == 3) {
            return 3 * 2 / 1;
        } else if (N == 4) {
            return 4 * 3 / 2 + 1;
        }
        
        int64_t result = N * (N-1) / (N-2) * 2;
        while (N > 0) {
            if (N == 1)
                result += -1;
            else if (N == 2)
                result += -2 * 1;
            else if (N == 3)
                result += -3 * 2 / 1;
            else
                result += (int64_t)-N * (N-1) / (N-2) + (N-3);
            N -= 4;
        }
        
        return result;
    }
};
