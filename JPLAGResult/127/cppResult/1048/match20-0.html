class Solution {
public:
    int clumsy(int N) {
        int res = 0, curr = N, p = 1;
        N--;
        while(N > 0){
            if(p%4 == 1){
                curr *= N;
            }else if(p%4 == 2){
                curr /= N;
            }else if(p%4 == 3){
                res += curr;
                curr = N;
            }else{
                res += curr;
                curr = -N;
            }
            p++;
            N--;
        }
        res += curr;
        return res;
        
    }
};
