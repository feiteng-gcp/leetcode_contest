class Solution {
public:
    int clumsy(int N)
{
    if(N == 1) return 1;
    if(N == 2) return 2;
    if(N == 3) return 6;
    int ans = (N * (N - 1) / (N - 2)) + N - 3;
    N -= 4;
    while(N >= 4)
    {
        ans += -(N * (N - 1) / (N - 2)) + N - 3;
        N -= 4;
    }
    switch(N)
    {
        case 3:
            ans += -(N * (N - 1) / (N - 2));
            break;
        case 2:
            ans += -N * (N - 1);
            break;
        case 1:
            ans -= N;
            break;
    }
    return ans;
}
};
