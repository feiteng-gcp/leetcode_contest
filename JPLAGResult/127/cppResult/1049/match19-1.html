class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        unordered_map<int, int> freq;
        for (int i = 0; i < A.size(); i++) {
            freq[A[i]] += 1;
            if (A[i] != B[i]) {
                freq[B[i]] += 1;
            }
        }
        
        int targetNum = -1;
        for (auto p : freq) {
            if (p.second == A.size()) {

                targetNum = p.first;
                break;
            }
        }
        
        if (targetNum == -1) return -1;
        int swapsA = 0, swapsB = 0;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] != targetNum) swapsA += 1;
            if (B[i] != targetNum) swapsB += 1;
        }
        
        return min(swapsA, swapsB);
    }
};
