class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {

        int ans = 50000;
        for (int n = 1; n <= 6; n++) {
            int p = 0;
            bool good = true;
            for (int i = 0; (i < A.size()) && good; i++) {
                if (A[i] != n) {
                    good = B[i] == n;
                    if (good) {
                        p++;
                    }

                }
            }
            if (good) {
                ans = min(ans, p);
            }
            
            good = true;p=0;
            for (int i = 0; (i < B.size()) && good; i++) {
                if (B[i] != n) {
                    good = A[i] == n;
                    if (good) {
                        p++;
                    }
                }
            }
            if (good) {
                ans = min(ans, p);
            }
        }
        if (ans == 50000) {
            ans = -1;
        }
        return ans;
    }
};
