class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int unsames = 0;
        for (int i = 0; i < A.size(); ++i)
            unsames += A[i] != B[i];
        
        int steps1 = 0;
        for (int i = 1; i < A.size(); ++i) {
            if (A[i] != A[0]) {
                if (B[i] == A[0])
                    steps1 += 1;
                else {
                    steps1 = -1;
                    break;
                }
            }
        }
        
        int steps2 = 0;
        for (int i = 1; i < A.size(); ++i) {
            if (B[i] != B[0]) {
                if (A[i] == B[0])
                    steps2 += 1;
                else {
                    steps2 = -1;
                    break;
                }
            }
        }
        
        if (steps1 >= 0)
            steps1 = std::min(steps1, unsames - steps1);
        if (steps2 >= 0)
            steps2 = std::min(steps2, unsames - steps2);
        
        if (steps1 < 0)
            return steps2;
        else if (steps2 < 0)
            return steps1;
        else
            return std::min(steps1, steps2);
    }
};
