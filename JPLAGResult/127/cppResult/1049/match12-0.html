class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        vector<int> cnt(7);
        for (int i = 0; i < A.size(); i++) {
            if (A[i] == B[i]) cnt[A[i]]++;
            else {
                cnt[A[i]]++;
                cnt[B[i]]++;
            }
        }
        int num = -1;
        for (int i = 1; i <= 6; i++) 
            if (cnt[i] == A.size()) {
                num = i; break;
            }
        if (num == -1) return -1;
        
        int diff_A_cnt = 0, diff_B_cnt = 0;
        for (auto & a : A) {
            if (a != num) diff_A_cnt++;
        }
        for (auto & b : B) {
            if (b != num) diff_B_cnt++;
        }
        return min(diff_A_cnt, diff_B_cnt);
    }
};
