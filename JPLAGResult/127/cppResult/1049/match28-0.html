class Solution {
    const int INF = 1000000000;
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int best = INF;
        int N = A.size();
        for (int d = 1; d <= 6; ++d) {
            int cur = 0;
            for (int i = 0; i < N; ++i) {
                if (A[i] == d) continue;
                if (B[i] != d) {
                    cur = INF;
                    break;
                }
                ++cur;
            }
            best = min(best, cur);

            cur = 0;
            for (int i = 0; i < N; ++i) {
                if (B[i] == d) continue;
                if (A[i] != d) {
                    cur = INF;
                    break;
                }
                ++cur;
            }
            best = min(best, cur);
        }
        return best >= INF ? -1 : best;
    }
};
