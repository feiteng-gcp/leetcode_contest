class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        vector<int> tmp1(7,0), tmp2(7,0), tmp3(7,0);
        int n=A.size();
        for(int i=0;i<n;i++){
            if(A[i]==B[i]) tmp1[A[i]]++;

            else tmp1[A[i]]++, tmp2[B[i]]++, tmp1[B[i]]++, tmp3[A[i]]++;;
        }
        int m=INT_MAX;
        for(int i=1;i<6;i++){
            if(tmp1[i]==n) m=min(min(m, tmp2[i]),tmp3[i]);
        }
        if(m==INT_MAX) return -1;
        return m;
    }
};
