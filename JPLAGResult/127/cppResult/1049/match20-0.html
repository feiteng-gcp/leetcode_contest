class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        vector<int> p(10, 0);
        int n = A.size();
        for (int i = 0; i < n; i++) {
            p[A[i]]++;
            p[B[i]]++;
        }
        int val = 0;
        for (int i = 1; i <= 6; i++) {
            if (p[i] >= n) val = i;
        }
        int ans = -1;
        int now = 0;
        int q = 0;
        for (int i = 0; i < n; i++) {
            if (B[i] == val && A[i] != val) {
                now++;
                q++;
            }
            else if (A[i] == val) q++;
        }
        if (q >= n) ans = now;
        now = 0;
        q = 0;
        for (int i = 0; i < n; i++) {
            if (A[i] == val && B[i] != val) {
                now++;
                q++;
            }
            else if (B[i] == val) q++;
        }
        if (q >= n) ans = min(ans, now);
        return ans;
    }
};
