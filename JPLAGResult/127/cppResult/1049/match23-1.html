class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int INF = 1e9;
        int ans = INF;

        int N = A.size();

        for(int t=0; t<2; t++){
            for(int k=1; k<=6; k++){
                int result = 0;
                for(int i=0; i<N; i++){

                    if(A[i] != k){
                        if(B[i] == k){
                            result++;
                        }else{
                            result = INF;
                        }
                    }
                }
                ans = min(ans, result);
            }
            swap(A, B);
        }
        return (ans == INF ? -1 : ans);
    }
};
