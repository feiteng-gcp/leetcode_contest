class Solution {
public:
    int minDominoRotations(vector<int>& a, vector<int>& b) {
        int ans = 1000000;
        for (int i = 1; i <= 6; ++i) ans = min(ans, Cal(a, b, i));
        for (int i = 1; i <= 6; ++i) ans = min(ans, Cal(b, a, i));
        if (ans == 1000000)
            return -1;
        return ans;
    }
    int Cal(vector<int> a, vector<int> b, int v) {
        int ans = 0;
        for (int i = 0; i < a.size(); ++i) if (a[i] != v){
            if (b[i] != v)
                return 1000000;
            ++ans;
        }
        return ans;
    }
};
