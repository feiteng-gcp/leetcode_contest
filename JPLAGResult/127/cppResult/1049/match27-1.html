class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int n = A.size();
        int res = n;
        for (int p = 1; p <= 6; p++)
        {
            int i, a = 0, b = 0;
            for (i = 0; i < n; i++)
            {
                if (A[i] == p && B[i] == p)
                    continue;
                else if (A[i] == p)
                    a++;
                else if (B[i] == p)
                    b++;
                else
                    break;
            }
            
            if (i == n)
                res = min(res, min(a, b));
        }
        
        if (res == n)
            res = -1;
        
        return res;
    }
};
