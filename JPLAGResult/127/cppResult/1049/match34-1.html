class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        vector<int> count(7, 0);
        
        for (int a : A) count[a]++;
        for (int b : B) count[b]++;
        

        int tar = 0;
        for (int i = 1 ; i <= 6 ; ++i) {
            if (count[i] >= A.size()) {
                tar = i;
                break;
            }
        }
        if (tar == 0) return -1;
        
        return min(help(A, B, tar), help(B, A, tar));
    }
    

    int help(vector<int>& A, vector<int>& B, int tar) {
        int rtn = 0;
        for (int i = 0 ; i < A.size() ; ++i) {
            if (A[i] == tar && B[i] != tar)
                rtn++;
            if (A[i] != tar && B[i] != tar)
                return -1;
        }
        return rtn;
    }
};
