class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int n=A.size();
        int ans=99999;
        for(int i=1;i<=6;i++)
        {
            int nowa=0,nowb=0;
            for(int j=0;j<n;j++)
            {
                if(A[j]!=i&&B[j]!=i)
                {
                    nowa=99999;
                    nowb=99999;
                    break;
                }
                else if(A[j]==i&&B[j]!=i)nowb++;
                else if(A[j]!=i&&B[j]==i)nowa++;
            }
            ans=min(ans,min(nowa,nowb));
        }
        if(ans==99999)return -1;
        return ans;
    }
};
