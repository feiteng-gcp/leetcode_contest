class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int ans=100000;
        for(int i=1; i<=6; i++) {

            int cnt=0;
            int ok=1;
            for(int j=0; j<A.size(); j++) {
                if (A[j]==i) continue;
                else if (B[j]==i) cnt++;
                else {
                    ok=0;
                    break;
                }
            }
            if (ok) ans=min(cnt, ans);
        }
        for(int i=1; i<=6; i++) {
            int cnt=0;
            int ok=1;
            for(int j=0; j<A.size(); j++) {
                if (B[j]==i) continue;
                else if (A[j]==i) cnt++;
                else {
                    ok=0;
                    break;
                }
            }
            if (ok) ans=min(cnt, ans);
        }
        if (ans==100000) return -1;
        else return ans;
    }
};
