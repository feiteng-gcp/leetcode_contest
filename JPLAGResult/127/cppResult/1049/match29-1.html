class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        vector<int> cnt = vector<int>(10, 0);
        for (int i = 0; i < A.size(); ++i) {
            ++cnt[A[i]];
            if (A[i] != B[i]) {
                ++cnt[B[i]];
            }
        }

        int val = 0;
        for (int i = 1; i < 7; ++i) {
            if (A.size() == cnt[i]) {
                val = i;
                break;
            }
        }
        if (0 == val) { return -1; }
        int cntA = 0, cntB = 0;
        for (int i = 0; i < A.size(); ++i) {
            if (val == A[i] && val != B[i]) { ++cntA;}
            if (val == B[i] && val != A[i]) { ++cntB;}
        }
        return min(cntA, cntB);
    }
};
