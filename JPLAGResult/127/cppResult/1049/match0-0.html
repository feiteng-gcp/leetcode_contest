class Solution {
    int length, ans;
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        length = A.size();
        ans = length + 1;
        for (int i = 1; i <= 6; ++i) {
            int sum = 0;
            for (int j = 0; j < length; ++j) {
                if (A[j] != i) {
                    if (B[j] == i) {
                        ++sum;
                    } else {
                        sum = length + 1;
                        break;
                    }
                }
            }
            ans = min(ans, sum);
        }
        
        for (int i = 1; i <= 6; ++i) {
            int sum = 0;
            for (int j = 0; j < length; ++j) {
                if (B[j] != i) {
                    if (A[j] == i) {
                        ++sum;
                    } else {
                        sum = length + 1;
                        break;
                    }
                }
            }
            ans = min(ans, sum);
        }
        
        return ans > length ? -1 : ans;
    }
};
