class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int ans = INT_MAX;
        int n = A.size();
        for (int x = 1; x <= 6; ++x) {
            int cur = 0;
            bool ok = true;
            for (int i = 0; i < n; ++i) {
                if (A[i] != x && B[i] != x) ok = false;
                else if (A[i] != x && B[i] == x) ++cur;
            }
            if (!ok) continue;
            ans = min(ans, cur);
            cur = 0;
            for (int i = 0; i < n; ++i)
                if (A[i] == x && B[i] != x) ++cur;
            ans = min(ans, cur);
        }
        return ans == INT_MAX ? -1 : ans;
    }
};
