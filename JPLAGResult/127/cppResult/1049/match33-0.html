class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int res = -1;
        for (int x = 1; x <= 6; ++x) {
            int up = 0;
            int down = 0;
            bool possible = true;
            for (int i = 0; i < A.size(); ++i) {
                if (A[i] != x && B[i] != x) {
                    possible = false;
                    break;
                }
                else {
                    if (A[i] != x) ++up;
                    if (B[i] != x) ++down;
                }
            }
            if (possible) {
                if (res = -1) {
                    res = min(up, down);
                }
                else {
                    res = min(res, min(up, down));
                }
            }
        }
        return res;
    }
};
