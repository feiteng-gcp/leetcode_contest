class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        vector<int> cnts(6);
        for (int i = 0; i < A.size(); ++i) {
            if (A[i] == B[i]) {
                ++cnts[A[i] - 1];

            } else {
                ++cnts[A[i] - 1];
                ++cnts[B[i] - 1];
            }
        }
        int num = -1;
        for (int i = 0; i < 6; ++i) {
            if (cnts[i] == A.size()) {
                num = i + 1;
                break;
            }
        }
        if (num < 0) {
            return -1;
        }
        int upcost = 0, downcost = 0;
        for (int i = 0; i < A.size(); ++i) {
            if (A[i] != num) {
                ++upcost;
            }
            if (B[i] != num) {
                ++downcost;
            }
        }
        return min(upcost, downcost);
    }
};
