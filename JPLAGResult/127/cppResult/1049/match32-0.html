class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        //看看是不是能进行转换成A的第一个数
        bool canTransfromA = true;
        for (int index = 1; index < A.size(); index ++) {
            if(A[index] != A[0] && B[index] != A[0]) {
                canTransfromA = false;
                break;
            }
        }
        //看看是不是能进行转换成B的第一个数
        bool canTransfromB = true;
        for (int index = 1; index < A.size(); index ++) {
            if(A[index] != B[0] && B[index] != B[0]) {
                canTransfromB = false;
                break;
            }
        }
        if(canTransfromA == false && canTransfromB == false) {
            return -1;
        }
        
        int result = INT_MAX;
        //转换成a的第一个元素
        if(canTransfromA == true) {
            //a的第一个元素放到上面还是下面
            int topCount = 0,bottomCount = 0;
            if(A[0] != B[0]) {
                bottomCount = 1;
            }
            int firstANum = A[0];
            for (int index = 1; index < A.size(); index ++) {
                //两个数相同，不需要转换
                if(A[index] == B[index]) {
                    continue;
                }
                //往上
                if(A[index] != firstANum) {
                    topCount ++;
                }
                if(B[index] != firstANum) {
                    bottomCount ++;
                }
            }
            result = min(result,min(topCount,bottomCount));
        }
        if(canTransfromB == true) {
            //b的第一个元素放到上面还是下面
            int topCount = 0,bottomCount = 0;
            if(A[0] != B[0]) {
                topCount = 1;
            }
            int firstANum = B[0];
            for (int index = 1; index < A.size(); index ++) {
                //两个数相同，不需要转换
                if(A[index] == B[index]) {
                    continue;
                }
                //往上
                if(A[index] != firstANum) {
                    topCount ++;
                }
                if(B[index] != firstANum) {
                    bottomCount ++;
                }
            }
            result = min(result,min(topCount,bottomCount));
        }
        return result;
    }
};
