class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
      int ans = 1 << 20;
      for (int i = 1; i <= 6; ++i) {
        int s = calc(A, B, i);
        ans = min(ans, s);
        s = calc(B, A, i);
        ans = min(ans, s);
      }
      if (ans == (1 << 20)) {
        ans = -1;
      }

      return ans;
    }
  int calc(vector<int> a, vector<int> b, int x) {
    int ans = 0;
    for (int i = 0; i < a.size(); ++i) {
      if (a[i] != x) {
        swap(a[i], b[i]);
        ++ans;
        if (a[i] != x) {
          return 1 << 20;
        }
      }
    }
    return ans;
  }
};
