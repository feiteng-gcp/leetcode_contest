typedef long long ll;
class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int len = A.size();
        
        vector<ll> checkSum(7, 0);
        // check if possible
        for(int i = 0; i < len; ++i) {
            if(A[i] == B[i]) {
                checkSum[A[i]] += i + 1;
            } else {
                checkSum[A[i]] += i + 1;
                checkSum[B[i]] += i + 1;
            }
        }
        ll expectedSum = ((1 + len) * len) / 2;
        bool isPossible = false;
        for(int i = 1; i < 7; ++i) {
            if(checkSum[i] == expectedSum) {
                isPossible = true;
                break;
            }
        }
        if(!isPossible) {
            return -1;
        }
        
        int minSwapTimes = 40000;
        for(int i = 1; i < 7; ++i) {
            if(checkSum[i] == expectedSum) {
                int swapTimesA = 0;
                int swapTimesB = 0;
                for(int j = 0; j < len; ++j) {
                    if(A[j] == B[j]) {
                        continue;
                    }
                    if(A[j] != i) {
                        swapTimesA += 1;
                    }
                    if(B[j] != i) {
                        swapTimesB += 1;
                    }
                }
                minSwapTimes = min(minSwapTimes, min(swapTimesA, swapTimesB));
            }
        }
        return minSwapTimes;
    }
};
