class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int n = A.size();
        vector<int> count(n, 0);
        for(int i = 0; i < n; ++i){
            count[i] = ( (1 << A[i]) | (1 << B[i]));
        }
        int res = INT_MAX;
        for(int k = 1; k <= 6; ++k){
            bool find = true;
            int countA = 0;
            int countB = 0;
            for(int i = 0; i < n; ++i){
                if((count[i] & (1 << k)) == 0){
                    find = false;
                    break;
                }
                if(A[i] == B[i]){
                    continue;
                }else{
                    countA += A[i] == k;
                    countB += B[i] == k;
                }
            }
            if(find){
                res = min(min(countA, countB), res);
            }
        }
        return res == INT_MAX ?  -1 : res;
    }
};
