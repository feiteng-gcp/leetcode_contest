class Solution {
public:
    int minDominoRotations(vector<int>& a, vector<int>& b)
    {
        int na = a.size(),nb = b.size(),ans = 1e9;
        for(int i = 1;i <= 6;i++)
        {
            int fail = 0,k = 0;
            for(int j = 0;j < na;j++)
            {
                if(a[j]!= i && b[j]!=i)
                {
                    fail = 1;
                    break;
                }
                else if(b[j] == i && a[j] != i)
                {
                    ++k;
                }
            }
            if(!fail)
            {
                ans = min(ans,k);
            }
            fail = 0,k = 0;
            for(int j = 0;j < na;j++)
            {
                if(a[j] != i && b[j]!=i)
                {
                    fail = 1;
                    break;
                }
                else if(a[j] == i && b[j]!=i)
                {
                    ++k;
                }
            }
            if(!fail)
                ans = min(ans,k);
        }
        if(ans == 1e9)
            return -1;
        else
            return ans;
    }
};
