class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int out = INT_MAX, N = A.size();
        for(int c = 1;c<6; c++){
            int a = 0, b = 0;
            bool pos = true;
            for(int i = 0;i<N;++i){
                if(A[i] != c && B[i] != c){
                    pos = false;
                    break;
                }
                if(A[i] == c && B[i] == c){
                   continue;
               }
                else{
                    if(A[i] == c){
                        b++;
                    }
                    else{
                        a++;
                    }
                }
            }
            if(pos){
                out = min(out, min(a,b));
            }
        }
        return out == INT_MAX? -1:out;
    }
};
