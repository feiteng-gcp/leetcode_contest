class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int best = 1<<30;
        for (int t = 0; t < 2; ++t) {
            

            for (int x = 0; x < 6; ++x) {
                int res = 0;
                int can = 1;
                for (int i = 0; i < A.size(); ++i) {
                    if (A[i]!=x && B[i]!=x) {
                        can = 0;
                    } else {
                        if (A[i]!=x) ++res;
                    }
                } 
                if (can) {
                    best = min(res,best);
                }
            }
            
            swap(A,B);
        }
        return best == (1<<30) ? -1 : best;
    }
};
