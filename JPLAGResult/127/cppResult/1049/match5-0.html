class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int min_r = A.size();
        for(int i=1; i<=6; ++i){
            min_r = min(min_r, possibleRotations(A, B, i));
            min_r = min(min_r, possibleRotations(B, A, i));
        }
        
        return min_r == A.size()? -1: min_r;
    }
    
    int possibleRotations(vector<int>& A, vector<int>& B, int target){
        int num = 0;
        int sz = A.size();
        for(int i=0; i<sz; ++i){
            if(A[i]!=target){
                if(B[i]==target){
                    ++num;
                } else {
                    return sz;
                }
            }
        }
        
        return num;
    }
};
