class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int ans=-1, n=A.size();
        
        for (int c=1; c<=6; c++){
            int rate=0; bool flag=true;
            for (int i=0; i<n; i++){
                if (A[i]==c) continue;
                if (B[i]==c) {
                    ++rate;
                }else flag=false;
            }
            if (flag){
                if (ans==-1 || ans>rate) ans=rate;
            }
        }
        

        for (int c=1; c<=6; c++){
            int rate=0; bool flag=true;
            for (int i=0; i<n; i++){
                if (B[i]==c) continue;
                if (A[i]==c) {
                    ++rate;
                }else flag=false;
            }
            if (flag){
                if (ans==-1 || ans>rate) ans=rate;
            }
        }
        
        return ans;
    }
};
