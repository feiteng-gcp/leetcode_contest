class Solution {
public:
    int minDominoRotations(vector<int>& a, vector<int>& b) {
        unordered_map<int,int> map;
        for(int i=0;i<a.size();i++){
            if(a[i]==b[i]){
                map[a[i]]++;
            }
            else{
                map[a[i]]++;
                map[b[i]]++;
            }
        }
        bool flag=false;
        int maxv=-1;
        for(auto it=map.begin();it!=map.end();it++){
            if(it->second==a.size()){
                flag=true;
                maxv=it->first;
                break;
            }
        }
        if(flag==false)
            return -1;
        int cnta=0,cntb=0;//where maxv in a
        for(int i=0;i<a.size();i++){
            if(a[i]==maxv&&b[i]!=maxv){
                cnta++;
            }
            else if(b[i]==maxv&&a[i]!=maxv){
                cntb++;
            }
        }
        return min(cnta,cntb);
    }
};
