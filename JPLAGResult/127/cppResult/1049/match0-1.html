class Solution {
public:
	int minDominoRotations(vector<int>& A, vector<int>& B) {


		int ans = INT_MAX;
		for (int number = 1; number <= 6; number++)
		{
			int count = 0;
			for (int i = 0; i < A.size(); i++)
			{
				if (A[i] != number)
				{
					if (i<B.size() && B[i] == number)
					{
						count++;
					}
					else
					{
						count = INT_MAX;
						break;
					}
						
				}

			}


			ans = min(ans, count);


			count = 0;
			for (int i = 0; i < B.size(); i++)
			{
				if (B[i] != number)
				{
					if (i<A.size() && A[i] == number)
					{
						count++;
					}
					else
					{
						count = INT_MAX;
						break;
					}

				}

			}

			ans = min(ans, count);

		}

		return ans == INT_MAX ? -1 : ans;
	}
};
