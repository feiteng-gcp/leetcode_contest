class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int ans = 999999;
        for (int i = 1; i <= 6; ++i) {
            ans = min(ans, tryRot(A, B, i));
            ans = min(ans, tryRot(B, A, i));
        }
        return ans < 999999 ? ans : -1;
    }
    int tryRot(const vector<int> &A, const vector<int> &B, int val) {
        int N = A.size();
        int ans = 0;
        for (int i = 0; i < N; ++i) {
            if (A[i] == val) continue;
            else if (B[i] == val) ++ans;
            else return 999999;
        }
        return ans;
    }
};
