class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int mini=1<<30;
        for(int i=1;i<=6;i++){
            bool ok=1;
            int cont=0;
            for(int j=0;j<A.size();j++){
                if(A[j]==i)continue;
                if(B[j]==i){
                    cont++;
                    continue;
                }
                
                ok=0;
            }
            
            if(ok)mini=min(mini,cont);

        }
        
        
        for(int i=1;i<=6;i++){
            bool ok=1;
            int cont=0;
            for(int j=0;j<A.size();j++){
                if(B[j]==i)continue;
                if(A[j]==i){
                    cont++;
                    continue;
                }
                
                ok=0;
            }
            
            if(ok)mini=min(mini,cont);
        }
        
        
        
        
        if(mini==(1<<30))return -1;
        return mini;
    }
};
