class Solution {
public:
    int minDominoRotations(vector<int>& a, vector<int>& b) {
        int res = -1, len = a.size();
        for (int i = 1; i <= 6; ++i) {
            for (int j = 0; j < 2; ++j) {

                int tmp = 0;
                bool ok = true;
                for (int k = 0; k < len; ++k) {
                    if (a[k] != i && b[k] != i) {
                        ok = false;
                        break;
                    } else if ((j == 1 && b[k] != i) || (j == 0 && a[k] != i)) {
                        ++tmp;
                    }
                }
                if (ok) {
                    if (res == -1) {
                        res = tmp;
                    } else {
                        res = min(res, tmp);
                    }
                }
            }
        }
        
        return res;
    }
};
