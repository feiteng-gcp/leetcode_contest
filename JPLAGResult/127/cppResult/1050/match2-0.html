/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder, int l, int r) {
        if (l==r) return nullptr;
        TreeNode* root=new TreeNode(preorder[l]);
        int mid=l;
        while (mid<r && preorder[mid]<=preorder[l]) mid++;
        root->left=bstFromPreorder(preorder, l+1, mid);
        root->right=bstFromPreorder(preorder, mid, r);
        return root;
    }
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return bstFromPreorder(preorder, 0, preorder.size());
    }
};
