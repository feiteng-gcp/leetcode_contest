/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* f(vector<int>& preorder,int s,int e){
        if(s>e)
            return NULL;
        TreeNode* curr=new  TreeNode(preorder[s]);
        int i=s+1;
        for(;i<=e;i++){
            if(preorder[i]>preorder[s])
                break;
        }
        curr->left=f(preorder,s+1,i-1);
        curr->right=f(preorder,i,e);
        return curr;
    }
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return f(preorder,0,preorder.size()-1);
    }
};
