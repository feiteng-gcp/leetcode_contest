/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        int n = preorder.size();
        if (n == 0) return NULL; 
        TreeNode* r = new TreeNode(preorder[0]);
        vector <int> L, R;
        for (int i = 1; i < n; i++) {
            if (preorder[i] < r -> val) {
                L.push_back(preorder[i]);
            } else {
                R.push_back(preorder[i]);
            }
        }
        r -> left = bstFromPreorder(L);
        r -> right = bstFromPreorder(R);
        return r;
    }
};
