/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    
    TreeNode* dfs(int left, int right, vector<int>& arr){
        if (left>right) return NULL;
        TreeNode* rt = new TreeNode(arr[left]);
        int mid=left;
        while(mid+1<=right && arr[mid+1]<arr[left]) ++mid;
        rt->left=dfs(left+1, mid, arr);
        rt->right=dfs(mid+1, right, arr);
        return rt;
    }
    
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        int n=preorder.size();
        return dfs(0, n-1, preorder);
    }
};
