/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> tmp;
    TreeNode* pre(int i,int j){
        if(i==j) return nullptr;

        auto res = new TreeNode(tmp[i]);
        int k;
        for(k=i;k<j;k++){
            if(tmp[k]>tmp[i]) break;
        }
        res->left=pre(i+1,k);
        res->right = pre(k,j);
        return res;
    }
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        tmp=preorder;
        return pre(0,preorder.size());
    }
};
