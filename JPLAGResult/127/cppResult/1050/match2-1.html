/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */


class Solution {
public:
    TreeNode* get(vector<int>& pre, int l, int r) {
        if (l > r - 1) return nullptr;
        TreeNode* rt = new TreeNode(pre[l]);
        int m = l + 1;
        while (m < r && pre[m] < pre[l]) m++;
        
        rt->left = get(pre, l + 1, m);
        rt->right = get(pre, m, r);
        return rt;
    }
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return get(preorder, 0, preorder.size());
    }
};
