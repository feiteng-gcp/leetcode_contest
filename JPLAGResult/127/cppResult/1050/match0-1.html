/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) {
      return construct(preorder, 0, preorder.size() - 1);
    }
  
    TreeNode *construct(vector<int> &preorder, int s, int e) {
      if (s > e) return nullptr;
      TreeNode *node = new TreeNode(preorder[s]);
      int i = s + 1;
      bool find = false;
      for (; i <= e; ++i) {
        if (preorder[i] > preorder[s]) {
          node->left = construct(preorder, s + 1, i - 1);
          find = true;
          break;
        }
      }
      if (!find) {
        node->left = construct(preorder, s + 1, e);
        node->right = nullptr;
      } else {
        node->right = construct(preorder, i, e);        
      }

      return node;
    }
};
