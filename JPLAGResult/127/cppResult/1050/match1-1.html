/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    
    TreeNode* f (vector<int> pre, int* id, 
                                int lo, int hi, int size){ 
        if (*id >= size || lo > hi) 
            return NULL; 
        TreeNode* root = new TreeNode(pre[*id] ); 
        *id = *id + 1; 
  
        if (lo==hi)return root; 
  
        int i; 
        for (i=lo; i<=hi;++i) 
            if ( pre[i] > root->val ) break; 
        root->left = f ( pre, id, *id, i-1, size ); 
        root->right = f ( pre, id, i, hi, size ); 
        return root; 
    } 
    
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        int id = 0; 
        int n=preorder.size();
        return f(preorder, &id, 0, n-1, n); 
    }
};
