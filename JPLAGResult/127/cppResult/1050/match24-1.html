/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return helper(0, preorder.size(), preorder);
    }
    

    TreeNode* helper(int begin, int end, const vector<int>& preorder) {
        if (begin == end) return nullptr;
        int rootVal = preorder[begin];
        TreeNode* root = new TreeNode(rootVal);
        int mid = begin + 1;
        while (mid < end && preorder[mid] < rootVal) mid += 1;
        root->left = helper(begin+1, mid, preorder);
        root->right = helper(mid, end, preorder);
        return root;
    }
};
