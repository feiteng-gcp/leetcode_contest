/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* buildT(vector<int>& A, int l, int r)
{
    if(r == l) return NULL;
    TreeNode* root = new TreeNode(A[l]);
    int p = l + 1;
    while(p < r && A[p] < A[l]) p++;
    root->left = buildT(A, l + 1, p);
    root->right = buildT(A, p, r);
    return root;
}

TreeNode* bstFromPreorder(vector<int>& preorder)
{
    return buildT(preorder, 0, (int)preorder.size());
}
};
