/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode *create(vector<int> &preorder, int start, int end) {
        if (start == end) return NULL;
        TreeNode *root = new TreeNode(preorder[start]);
     //   printf("%d %d %d\n", start, end, preorder[start]);
        int mid = distance(&preorder[0], upper_bound(&preorder[start + 1], &preorder[end], preorder[start]));
        root->left = create(preorder, start + 1, mid);
        root->right = create(preorder, mid, end);
        return root;
    }
    
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return create(preorder, 0, preorder.size());
    }
};
