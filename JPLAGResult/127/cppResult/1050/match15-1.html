/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return build_bst(preorder, 0, preorder.size());
    }
    TreeNode* build_bst(vector<int>& preorder, int l, int r) {
        if (l >= r) return NULL;
        TreeNode* root = new TreeNode(preorder[l]);
        int mid = l + 1;
        while (mid < r && preorder[mid] < root->val) mid++;
        root->left = build_bst(preorder, l + 1, mid);
        root->right = build_bst(preorder, mid, r);
        return root;
    }
};
