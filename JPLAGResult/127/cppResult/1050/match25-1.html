/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* go(vector<int>& a, int l, int r) {
        if (l>=r) return 0;
        int x = a[l];

        TreeNode *rt = new TreeNode(x);
        int m = l+1;
        while(m < a.size() && a[m]<x) ++m;
        rt->left = go(a,l+1,m);
        rt->right = go(a,m,r);
        return rt;
    }
    
    TreeNode* bstFromPreorder(vector<int>& a) {
        return go(a,0,a.size());
    }
};
