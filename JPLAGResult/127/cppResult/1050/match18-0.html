/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        if(preorder.size()==0)
            return NULL;
        TreeNode *root = new TreeNode(preorder[0]);
        int bigindex=preorder.size();
        for(int i=0;i<preorder.size();i++){
            if(preorder[i]>preorder[0]){
                bigindex = i;
                break;
            }  
        }
        //cout << bigindex << endl;
        vector<int> myleft(preorder.begin()+1, preorder.begin()+bigindex);
        vector<int> myright(preorder.begin()+bigindex, preorder.end());
        if(myleft.size()!=0)
            root->left = bstFromPreorder(myleft);
        else
            root->left = NULL;
        if(myright.size()!=0)
            root->right = bstFromPreorder(myright);
        else
            root->right = NULL;
        return root;
    }
};
