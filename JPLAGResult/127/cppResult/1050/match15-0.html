/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) 
    {
        return DFS(preorder,0,preorder.size()-1);
    }
    
    TreeNode* DFS(vector<int>& preorder, int a, int b)
    {
        if (a>b) return NULL;
        TreeNode* root = new TreeNode(preorder[a]);
        if (a==b) return root;
        
        int k = a+1;
        while (k<=b && preorder[k]<preorder[a]) k++;
        
        root->left = DFS(preorder,a+1,k-1);
        root->right = DFS(preorder,k,b);
        return root;
    }
};
