class Solution {
public:
    TreeNode *bstFromPreorder(vector<int> &preorder) {
        int n = preorder.size();

        return Construct(preorder, 0, n - 1);
    }

    TreeNode *Construct(vector<int> &preorder, int start, int end) {
        if (start > end)
            return nullptr;
        auto node = new TreeNode(preorder[start]);
        int j = start + 1;
        while (j <= end && preorder[j] < preorder[start])
            ++j;
        node->left = Construct(preorder, start + 1, j - 1);
        node->right = Construct(preorder, j, end);
        return node;
    }
};
