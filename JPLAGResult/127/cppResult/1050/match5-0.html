/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return helper(0, preorder.size()-1, preorder);
    }
    TreeNode* helper(int st, int en, vector<int>& pre) {
        if (st > en)
            return NULL;
        
        TreeNode *root = new TreeNode(pre[st]);
        if (st == en)
            return root;
        
        int idx = st+1;
        while (idx <= en && pre[idx] < pre[st])
            ++idx;
        
        root->left = helper(st+1, idx-1, pre);
        root->right = helper(idx, en, pre);
        return root;
    }
};
