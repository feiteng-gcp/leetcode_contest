/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& preorder) {
        return dfs(preorder, 0, (int)preorder.size() - 1);

    }
    TreeNode* dfs(vector<int>& a, int lo, int hi) {
        if (lo > hi) return NULL;
        TreeNode *root = new TreeNode(a[lo]);
        if (lo == hi) return root;
        int i = lo + 1;
        while (i <= hi && a[i] < a[lo]) ++i;
        root->left = dfs(a, lo + 1, i - 1);
        root->right = dfs(a, i, hi);
        return root;
    }
};
