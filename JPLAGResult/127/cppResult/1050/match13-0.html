class Solution {
public:
    TreeNode* bstFromPreorder(vector<int>& a) {
        return Build(a, 0, a.size() - 1);
    }
    TreeNode* Build(vector<int>& a, int l, int r) {
       // printf("%d %d\n", l, r);
        if (l > r)
            return nullptr;
        auto root = new TreeNode(a[l]);
        int mid = l + 1;
        while (mid <= r && a[mid] < a[l]) ++mid;
        root->left = Build(a, l + 1, mid - 1);
        root->right = Build(a, mid, r);
        return root;
    }
};
