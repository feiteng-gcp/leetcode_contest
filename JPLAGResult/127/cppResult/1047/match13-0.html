class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        sort(A.begin(), A.end());
        for (int i = 0; i < A.size() && K > 0; i++) {
            if (A[i] < 0) {
                --K;
                A[i] = -A[i];
            }
        }
        K %= 2;
        sort(A.begin(), A.end());
        if (K > 0) {
            A[0] = -A[0];
        }
        
        int sum = 0;
        for (int i = 0; i < A.size(); i++) sum += A[i];
        return sum;
    }
};
