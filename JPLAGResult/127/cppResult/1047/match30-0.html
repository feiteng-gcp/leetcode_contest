class Solution {
public:
    int largestSumAfterKNegations(vector<int>& a, int k) {
        sort(a.begin(), a.end());
        for (int i = 0; i < a.size(); ++i) {
            if (a[i] < 0 && k) {
                a[i] = -a[i];
                --k;
            }
        }
        sort(a.begin(), a.end());
        if (k % 2 == 1) {
            a[0] = -a[0];
        }
        return accumulate(a.begin(), a.end(), 0);
    }
};
