class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        int n = A.size();
        sort(A.begin(),A.end());
        for(int i=0;i<n;i++){
            if(A[i]<0){
                K--;
                A[i]=-A[i];
            }
            if(K==0)break;
        }
        if(K>0){
            sort(A.begin(),A.end());
            if(K&1)A[0]=-A[0];
        }
        int ret = 0;
        for(int i=0;i<n;i++)ret+=A[i];
        return ret;
    }
};
