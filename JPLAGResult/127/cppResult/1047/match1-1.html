class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        sort(A.begin(),A.end());
        int t = 0;
        for(int i =0;i<A.size();i++){
            if(t<K && A[i]<0){
                A[i] *= -1;
                t++;
            }

        }
        sort(A.begin(),A.end());
        if((K-t) % 2){
            A[0] *= -1;
        }
        int sum = 0;
        for(int i =0;i<A.size();i++){
            sum += A[i];
        }
        return sum;
    }
};
