class Solution {
 public:
  int largestSumAfterKNegations(vector<int>& A, int K) {
    multiset<int> st(A.begin(), A.end());
    for (int i = 0; i < K; ++i) {
      int t = *st.begin();
      st.erase(st.begin());
      st.insert(-t);
    }
    return accumulate(st.begin(), st.end(), 0);
  }
};
