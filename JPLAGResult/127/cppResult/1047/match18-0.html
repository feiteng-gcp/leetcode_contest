class Solution {
public:
    static int cmp(int x, int y) {
        return abs(x) > abs(y);
    }
    int largestSumAfterKNegations(vector<int>& A, int K) {
        sort(A.begin(), A.end(), cmp);
        for (int i = 0;i < A.size() && K != 0;i++) {
            if (A[i] < 0) {
                A[i] = -A[i];
                K--;
            }
        }
        if (K > 0) {
            K %= 2;
        }
        if (K > 0) {
            A[A.size() - 1] = -A[A.size() - 1];
        }
        int s = 0;
        for (int i = 0;i < A.size();i++) {
            s += A[i];
        }
        return s;
    }
};
