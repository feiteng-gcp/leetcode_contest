class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        sort(A.begin(), A.end());
        int n = A.size();
        for (int i = 0; i < n; i++) {
            if (A[i] < 0 && K > 0) {
                A[i] = -A[i];
                K--;
            } else {
                break;
            }
        }
        sort(A.begin(), A.end());
        if (K % 2 == 1) {
            A[0] = -A[0];
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            ans += A[i];
        }
        return ans;
    }
};
