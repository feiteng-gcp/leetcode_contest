class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        int min=100;
        int minidx=0;
        for(int i=0;i<K;i++){
            min = 100;
            minidx=0;
            for(int j=0;j<A.size();j++){
                if(A[j]<min) {min=A[j];minidx=j;}
            }
            A[minidx]=-A[minidx];
        }
        int sum=0;
        for(int i=0;i<A.size();i++){
            sum+=A[i];
        }
        return sum;
    }
};
