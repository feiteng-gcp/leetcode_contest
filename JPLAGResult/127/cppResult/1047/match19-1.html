class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        sort(A.begin(), A.end());
        for (int i = 0; i < A.size() && K; i++) {
            if (A[i] < 0) {
                A[i] *= -1;
                K--;
            }
        }
        
        if (K > 0) {
            // if K is even, do nothing
            
            // if K is odd, pick 0 or smallest element
            if (K % 2) {
                sort(A.begin(), A.end());
                A[0] *= -1;
            }

        }
        
        int res = 0;
        for (int i = 0; i < A.size(); i++) {
            res += A[i];
        }
        return res;
    }
};
