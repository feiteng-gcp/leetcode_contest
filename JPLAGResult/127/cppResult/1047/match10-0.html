class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        priority_queue<int, vector<int>, greater<int> > pq;
        for (int x : A)
            pq.push(x);
        for (int k = 0; k < K; ++k) {
            int x = pq.top();
            pq.pop();
            x = -x;
            pq.push(x);
        }
        int res = 0;
        while (!pq.empty()) {
            res += pq.top();
            pq.pop();
        }
        return res;
    }
};
