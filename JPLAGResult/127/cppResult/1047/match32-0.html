class Solution {
public:
    int largestSumAfterKNegations(vector<int>& a, int k) {
        
        sort(a.begin(), a.end());
        
        for(int i = 0; i < a.size() && a[i] < 0 && k > 0; ++i)
            a[i] = -a[i], --k;
        
        if(k > 0 && k % 2 == 1) {
            int mn = INT_MAX, idx = -1;
            for(int i = 0; i < a.size(); ++i)
                if(a[i] < mn)
                    mn = a[i], idx = i;
            a[idx] = -a[idx];
        }
        
        int ans = 0;
        for(auto x : a)
            ans += x;
        
        return ans;
    }
};
