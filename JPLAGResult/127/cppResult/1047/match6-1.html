class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        sort(A.begin(), A.end());
        int n = A.size();
        for (int i = 0; i < n && K > 0; ++i) {
            if (A[i] < 0) {
                A[i] = -A[i];
                K--;
            }

        }
        
        sort(A.begin(), A.end());
        if (K & 1) {
            A[0] = -A[0];
        }
        int res = 0;
        for (int i = 0; i < n; ++i) {
            res += A[i];
        }
        return res;
    }
};
