class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        priority_queue<int, vector<int>, greater<int>> q;
        for (int ele : A) {
            q.push(ele);
        }

        while(K > 0) {
            int temp = -q.top();
            q.pop();
            q.push(temp);
            K--;
        }
        
        int ans = 0;
        while(!q.empty()) {
            ans += q.top();
            q.pop();
        }
        return ans;
    }
};
