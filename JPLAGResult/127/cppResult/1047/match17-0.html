class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int K) {
        sort(A.begin(), A.end());
        for(int i = 0; i < A.size() && K > 0; ++i) {
            if(A[i] < 0) {
                A[i] = -A[i];
                K -= 1;
            }
            if(A[i] == 0) {
                K = 0;
            }
        }
        
        if(K > 0) {
            sort(A.begin(), A.end());
            while(K > 0) {
                A[0] = -A[0];
                K -= 1;
            }
        }
        int sum = 0;
        for(int i = 0; i < A.size(); ++i) {
            sum += A[i];
        }
        return sum;
    }
};
