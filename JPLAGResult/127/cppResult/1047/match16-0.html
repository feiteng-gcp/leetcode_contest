class Solution {
public:
    int largestSumAfterKNegations(vector<int>& A, int k) {
        sort(A.begin(), A.end());
        
        for(int i = 0; i < A.size(); ++i)
        {
            if(k > 0 && A[i] < 0)    
            {
                A[i] = -1 * A[i];
                --k;
            }
        }
        
        k %= 2;
        if(k > 0)
        {
            sort(A.begin(), A.end());            
            A[0] = -1 * A[0];
        }
        int sum = 0;
        for(int i = 0; i < A.size(); ++i)
        {
            sum += A[i];
        }
        
        return sum;
    }
};
