/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        return helper(preorder, 0, preorder.length - 1);
    }
    
    public TreeNode helper(int[] preorder, int l, int r) {
        if (l > r) {
            return null;
        }
        TreeNode node = new TreeNode(preorder[l]);
        int idx = l;
        while (idx <= r && preorder[idx] <= preorder[l]) {
            idx++;
        }
        node.right = helper(preorder, idx, r);
        node.left = helper(preorder, l + 1, idx - 1);
        return node;
    }
}
