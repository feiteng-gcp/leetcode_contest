/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    
    
    public TreeNode bstFromPreorder(int[] preorder) {
        return parse(preorder, 0, preorder.length);        
    }
    
    TreeNode parse(int[] preorder, int l, int r){
        if(l>=r){
            return null;
        }
        TreeNode n = new TreeNode(preorder[l]);
        int j = l + 1;
        while(j < preorder.length && preorder[j] < preorder[l]){
            j++;
        }
        n.left = parse(preorder, l + 1, j);
        n.right = parse(preorder, j, r);
        return n;
    }
    
    
    
}
