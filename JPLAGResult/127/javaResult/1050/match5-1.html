/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
  public TreeNode bstFromPreorder(int[] preorder) {
    return helper(preorder, 0, preorder.length - 1);
  }

  private TreeNode helper(final int[] preorder, int left, int right) {
    if (left > right) {
      return null;
    }
    if (left == right) {
      return new TreeNode(preorder[left]);
    }

    final TreeNode root = new TreeNode(preorder[left]);
    int lastSmallerIndex = left;
    while (lastSmallerIndex <= right) {
      if (preorder[lastSmallerIndex] > preorder[left]) {
        break;
      }
      lastSmallerIndex++;
    
}
    root.left = helper(preorder, left + 1, lastSmallerIndex - 1);
    root.right = helper(preorder, lastSmallerIndex, right);
    return root;
  }
}
