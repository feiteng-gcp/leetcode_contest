/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        return toTree(preorder, 0, preorder.length-1);
    }
    
    private TreeNode toTree(int[] preorder, int left, int right) {
        if (left > right) { return null; }
        TreeNode root = new TreeNode(preorder[left]);
        int rightIdx = right+1;
        for (int i=left+1; i<=right; i++) {
            if(preorder[i] > root.val) {
                rightIdx = i;
                break;
            }
        }
        root.left = toTree(preorder, left+1, rightIdx-1);
        root.right = toTree(preorder, rightIdx, right);
        return root;
    }
}
