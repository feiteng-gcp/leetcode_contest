/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        return construct(preorder, 0, preorder.length-1);
    }
    
    public TreeNode construct (int[] preorder, int begin, int end) {
        if (begin > end) return null;
        int val = preorder[begin];
        TreeNode cur = new TreeNode(val);
        int rightTreeStart = begin+1;
        while (rightTreeStart <= end && preorder[rightTreeStart] < val)
            rightTreeStart++
;
        cur.left = construct(preorder, begin+1, rightTreeStart-1);
        cur.right = construct(preorder, rightTreeStart, end);
        return cur;
    }
}
