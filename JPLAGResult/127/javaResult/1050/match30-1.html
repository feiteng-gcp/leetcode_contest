/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        return help(preorder, 0, preorder.length - 1);
    }
    public TreeNode help(
int[] order, int i, int j) {
        if (i > j) {
            return null;
        }
        TreeNode cur = new TreeNode(order[i]);
        int index = i + 1;
        for (; index <= j; index++) 
{
            if (order[index] > order[i]) {
                break;
            }
        }
        cur.left = help(order, i+ 1, index - 1);
        cur.right = help(order, index, j);
        return cur;
    }
}
