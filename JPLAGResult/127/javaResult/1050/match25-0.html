/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        if (null == preorder || preorder.length == 0) {
            return null;
        }
        return bstFromPreorder(preorder, 0, preorder.length);
    }
    
    private TreeNode bstFromPreorder(int[] preorder, int begin, int end) {
        if (end - begin == 0) {
            return null;
        } else {
            TreeNode node = new TreeNode(preorder[begin]);
            int i = begin + 1;
            while (i < end && preorder[i] < preorder[begin]) {
                i++;
            }
            node.left = bstFromPreorder(preorder, begin + 1, i);
            node.right = bstFromPreorder(preorder, i, end);
            return node;
        }
    }
}
