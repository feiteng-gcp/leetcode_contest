/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        return helper(preorder, 0, preorder.length-1);
    }
    
    private TreeNode helper(int[] preorder, int left, int right) {
        if(left > right) return null;
        if(left == right) return new TreeNode(preorder[left]);
        TreeNode ret = new TreeNode(preorder[left]);
        int index = left+1;
        while(index <= right && preorder[index] < preorder[left]) {
            index++;
        }
        ret.left = helper(preorder, left+1, index-1);
        ret.right = helper(preorder, index, right);
        return ret;
    }
}
