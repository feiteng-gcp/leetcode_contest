/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        return help(preorder,0,preorder.length-1);
    }
    
    private TreeNode help(int[] preorder, int i, int j){
        if(i>j) return null;
        if(i==j) return new TreeNode(preorder[i]);
        TreeNode node = new TreeNode(preorder[i]);
        int mid = i+1;
        while(mid<=j && preorder[mid]<preorder[i]) mid++
;
        node.left=help(preorder,i+1,mid-1);
        node.right=help(preorder,mid,j);
        return node;
    }
}
