class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        // Basically each time we flip the minimum values
        // So we use a min priority queue
        int sum = 0;
        
        
PriorityQueue<Integer> pq = new PriorityQueue<>();
        
        for (int i : A) {
            sum += i;
            pq.offer(i);
        }
        
        while (K-- > 0) {
            int min = pq.remove();
            
            int newNum = (-1) * min;
            
            sum -= 2 * min;
            
            pq.offer(newNum);
        }
        
        return sum;
        
    }
}
