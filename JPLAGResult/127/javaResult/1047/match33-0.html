class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int res = 0;
        int min = Integer.MAX_VALUE;
        for(int i = 0; i < A.length; i++){
            if(A[i] < 0 && K-- > 0)
                A[i] = -A[i];
            
            min = Math.min(min, A[i]);
            res += A[i];
        }
        
        if(K > 0 && K % 2 == 1){
            res -= 2 * min;
        }
        
        return res;
    }
}
