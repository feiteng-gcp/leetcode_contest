class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for(int num : A) pq.offer(num);
        while(K > 0) {
            int cur = pq.poll();
            //System.out.println("cur: " + cur);
            if(cur < 0) {
                pq.offer(-cur);
                K--;
            }else if(cur == 0){
                pq.offer(cur);
                K--;
            }else{
                pq.offer(-cur);
                K--;
            }
        }
        int sum = 0;
        //System.out.println(Arrays.toString(pq));
        while(!pq.isEmpty())  {
            int temp = pq.poll();
            //System.out.println(temp);
            sum += temp;
        }
        return sum;
    }
}
