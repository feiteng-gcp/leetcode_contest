class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int i = 0;
        while(K > 0 && A[i] <= 0){
            if(A[i] < 0){
                A[i] *= -1;
                K --;
            } 
            else K = 0;
            
            i ++;
        }
        
        if(K > 0){
            Arrays.sort(A);
            K = K % 2;
            if(K == 1)
                A[0] = -A[0];   
        
}
        
        int res = 0;
        for(int num : A) res += num;
        return res;
    }
}
