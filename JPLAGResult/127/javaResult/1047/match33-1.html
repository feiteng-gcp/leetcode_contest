class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int N = A.length;
        int first = 0;
        for (int i = 0; i < K; i++) {
            if (first == N) {
                A[first - 1] = -A[first - 1];
            } else {
                if (A[first] > 0) {
                    if (first == 0) {
                        A[first] = -A[first];
                    } else {
                        if (A[first - 1] < A[first]) {
                            first = first - 1;
                        }
                        A[first] = -A[first];
                    }                    
                } else if (A[first] < 0) {
                    A[first] = -A[first];
                    first++;
                } else if (A[first] == 0) {
                    break;
                }
            }
        }
        int res = 0;
        for (int i = 0; i < N; i++) {
            res += A[i];
        }
        return res;
    }
}
