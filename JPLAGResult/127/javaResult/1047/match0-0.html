class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {


        PriorityQueue<Integer> pq = new PriorityQueue<>();

        for (int d : A) {
            pq.add(d);
        }

        for (int i = 0; i < K; i++) {
            int top = pq.poll();
            pq.add(-1 * top);
        }

        int ans = 0;
        while (!pq.isEmpty()) {
            ans += pq.poll();
        }

        return ans;

    }
}
