class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        PriorityQueue<Integer> posQue = new PriorityQueue<>();
        PriorityQueue<Integer> negQue = new PriorityQueue<>();
        int sum = 0;
        
for(int x : A) {
            sum += x;
            if(x >= 0) {
                posQue.add(x);
            } else {
                negQue.add(x);
            }
        }
        
        while(K > 0 && !negQue.isEmpty()) {
            int x = negQue.poll();
            sum -= x * 2;
            posQue.add(- x);
            K --;
        }
        if(K > 0 && K % 2 == 1) {
            sum -= posQue.peek() * 2;
        }
        return sum;
    }
}
