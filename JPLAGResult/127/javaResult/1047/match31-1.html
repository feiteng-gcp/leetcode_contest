class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        PriorityQueue<Integer> minPos = new PriorityQueue();
        PriorityQueue<Integer> minNeg = new PriorityQueue();
        for(int num : A) {
            if (num >= 0) {
                minPos.offer(num);
            } else {
                minNeg.offer(num);
            }
        }
        
        while(K > 0 && !minNeg.isEmpty()) {
            int val = minNeg.poll();
            minPos.offer(-val);
            K--;
        }
        
        int result = minPos.poll();
        
if (K >0) {
            if (K%2==1) {
                result = - result;
            }
        }
        
        for (int num : minPos) {
            result += num;
        }
        
        
for(int num : minNeg) {
            result += num;
        }
        
        return result;
    }
}
