class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int n = A.length;
        int sum = 0;
        int k = 0;
        for(int i=0;i<n;i++){
            if(k < K){
                if(A[i] < 0){
                    A[i] = -A[i];
                    k++;
                }
            }
        }
        
        Arrays.sort(A);
        int rem = K-k;
        if(rem%2 != 0){
            A[0] = -A[0];
        }
        
        for(int i=0;i<n;i++){
            sum += A[i];

        }
        return sum;
    }
}
