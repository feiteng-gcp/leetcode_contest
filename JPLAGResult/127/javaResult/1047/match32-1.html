class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        // revesere all negative to positive if can
        Queue<int[]> queue = new PriorityQueue<int[]>(Comparator.comparingInt(a -> a[0]));
        for (int i = 0; i < A.length; i++) {
            if (A[i] < 0) {
                queue.offer(new int[]{A[i], i});
            }
        }
        int remains = K - queue.size();
        while (K > 0 && !queue.isEmpty()) {
            int[] pos = queue.poll();
            A[pos[1]] *= -1;
            K--;
        }
               
        
        int sum = 0;
        int min = A[0];
        for (int i = 0; i < A.length; i++) {
            sum += A[i];
            
min = Math.min(A[i], min);
        }
        if (remains > 0 && remains % 2== 1) {
            sum -= min * 2;
        } 
        return sum;
        
        // revese the smallest if needed
        
    }
}
