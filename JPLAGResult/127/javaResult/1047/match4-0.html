class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int ans=0,c=0,m=A[0];
        for(int n:A){
            if(n<0){
                c++;
            }
            m=Math.min(Math.abs(m),Math.abs(n));
            ans+=n;
        }
        for(int i=0;K>0&&i<c;i++,K--){
            ans-=(2*A[i]);
        }
        if(K%2==0) return ans;
        return ans-2*m;
    }
}
