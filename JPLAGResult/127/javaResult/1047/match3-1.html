class Solution {
    public int largestSumAfterKNegations(int[] a, int k) {
        Queue<Integer> pq = new PriorityQueue<>(); 
        for (int x : a){
            pq.add(x); 
        }
        while (k > 0) { 
            pq.add(pq.remove() * (-1)); 
            k--;
        } 
        int sum = 0; 
        for (int x : pq){
             sum += x; 
        }
        return sum; 
    }
}
