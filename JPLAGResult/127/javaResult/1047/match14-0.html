class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int t = 0;
        for (int i = 0; i < A.length && t < K; i++) {
            if (A[i] < 0) {
                A[i] = -A[i];
                t++;
            }
            else {
                break;
            }
        }

        Arrays.sort(A);
        int times = (K - t)%2;
        if (times == 1) {
            A[0] = -A[0];
        }

        int sum = 0;
        for (int a : A)
            sum += a;

        return sum;
    }
}
