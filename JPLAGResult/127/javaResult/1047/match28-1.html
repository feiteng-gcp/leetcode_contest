class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        // flip signs for make its the largest
        
        // flip all negative as many as you can
        // else flip only the first one
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        
        for (int a : A) pq.offer(a);
        while (K > 0) {
            int temp = pq.poll();
            pq.offer(-temp);
            K--;
        }
        
        int sum = 0;
        while (!pq.isEmpty()) sum += pq.poll();
        
        
        
        return sum;
        
        
    }
}
