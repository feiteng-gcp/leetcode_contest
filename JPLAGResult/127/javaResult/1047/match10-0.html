class Solution
{
    public int largestSumAfterKNegations(int[] a, int k)
    {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i : a)
        {
            pq.add(i);
        }
        while (k-- > 0)
        {
            int h = pq.poll();
            h = -h;
            pq.add(h);
        }
        int ans = 0;
        while (!pq.isEmpty())
        {
            ans += pq.poll();
        }
        return ans;
    }
}
