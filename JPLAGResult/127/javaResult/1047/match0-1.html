class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        PriorityQueue<Integer> queue = new PriorityQueue<>();
        for (int a: A) {
            queue.add(a);
        }
        
        for (int i=0; i<K; i++) {
            int k = queue.remove();
            queue.add(k*-1);
        }
        int sum = 0;
        while (queue.size() > 0) {
            sum += queue.remove();
        }
        return sum;
    }
}
