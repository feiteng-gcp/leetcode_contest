class Solution {
    public int largestSumAfterKNegations(int[] a, int k) {
        
        Arrays.sort(a);
        int sum = 0;
        for(int i=0;i<a.length;++i){
            if(a[i]<0 && k > 0){
                k--;
                a[i] = -a[i];
            }
        }
        Arrays.sort(a);
        if(k%2==1){
            sum = -a[0];
        }else{
            sum = a[0];
        }
        for(int i=1;i<a.length;++i){
            sum += a[i];
        }
        return sum;
    }
}
