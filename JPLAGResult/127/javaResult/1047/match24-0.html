class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        // neg to pos
        int h = 0;
        for (int i = 0;i < A.length;i++) {
            if (A[i] >= 0) {
                h = i;
                break;
            }
        }
        // System.out.println(h);
        for (int i = 0;i < h;i++) {
            if (K == 0) break;
            A[i] = -A[i];
            K--;
        }
        if (K > 0) {
            // System.out.println(K);
            if (K%2 != 0) {
                // System.out.println(h);
                if (h > 0 && A[h - 1] < A[h]) {
                    A[h - 1] = - A[h - 1];
                } else {
                    A[h] = -A[h];       
                }
            }
        }
        int sum = 0;
        for (int n : A) sum += n;
        
        return sum;
    }
}
