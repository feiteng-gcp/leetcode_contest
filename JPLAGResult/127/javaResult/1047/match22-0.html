class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        int len = A.length;
        Arrays.sort(A);
        
        int index = 0;
        while (K > 0 && index < len && A[index] < 0) {
            A[index] = -A[index];
            index++;
            K--;
        }
        
        Arrays.sort(A);
        
        if (K % 2 == 1) A[0] = -A[0];
        
        int result = 0;
        for (int i : A) result += i;
        
        return result;
    }
}
