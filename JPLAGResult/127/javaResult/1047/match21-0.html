class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int i = 0;
        while (i < K && A[i] <= 0) {
            A[i++] *= -1;
        }
        Arrays.sort(A);
        if ((K - i) % 2 == 1) A[0] = -A[0];
        int sum = 0;
        for (int n : A) sum += n;
        return sum;
    }
}
