class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        int total=0;
        for(int i=0;i<A.length;i++){
            if(K>0&&A[i]<0){
                K--;
                A[i]=-A[i];
            }
        }
        Arrays.sort(A);
        for(int i=0;i<A.length;i++){
            if(K>0){
                K=K%2;
                
if(K==1){
                    K--;
                    total=total-A[i];
                }
                else{
                    total=total+A[i];
                }
            }
            else{
                total=total+A[i];
            }
        }
        return total;
    }
}
