class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        PriorityQueue<Integer> q = new PriorityQueue<>();
        for(int x : A) {
            q.offer(x);
        }
        for(int k = 0; k < K; k++) {
            int temp = q.poll();
            temp = -temp;
            
q.offer(temp);
        }
        int sum = 0;
        while(!q.isEmpty()) {
            sum += q.poll();
        }
        return sum;
    }
}
