class Solution {
    public int largestSumAfterKNegations(int[] A, int K) {
        Arrays.sort(A);
        for(int i=0;i<K;i++){
            help(A);
        }
        
int res = 0;
        for(int each:A){
            res += each;
        }
        return res;
    }
    public void help(int[] A){
        A[0] = -A[0];
        int index = 0;
        for(int i=1;i<A.length;i++){
            if(A[i-1]>A[i]){
                swap(A, i-1, i);
            }
        }
    }
    public void swap(int[] A, int i, int j){
        int tmp = A[i];
        A[i] = A[j];
        A[j] = tmp;
    }
}
