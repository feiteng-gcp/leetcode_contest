	class Solution {
	    public int largestSumAfterKNegations(int[] a, int K) {
	        Arrays.sort(a);
	        for(int i = 0;i < a.length;i++){
	        	if(a[i] < 0 && K > 0){
	        		a[i] = -a[i];
	        		K--;
	        	}
	        }
	        Arrays.sort(a);
	        if(K % 2 == 1){
	        	a[0] = -a[0];
	        }
	        int s = 0;
	        for(int v : a){
	        	s += v;
	        }
	        return s;
	    }
	}
