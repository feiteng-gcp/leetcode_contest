class Solution {
    public int minDominoRotations(int[] A, int[] B) {
        int res = Integer.MAX_VALUE;
        for (int i = 1; i<= 6; i++) {
            int count1 = 0;
            int count2 = 0;
            int j = 0;
            for (; j < A.length; j++) {
                if (A[j] != i && B[j] != i) {
                    break;
                }
                if (A[j] == i) {
                    count1++;
                
} 
                if (B[j] == i) {
                    count2++;
                }
            }
            if (j == A.length) {
                res = Math.min(res, A.length - 
Math.max(count1, count2));
            }
        }
        if (res == Integer.MAX_VALUE) {
            return -1;
        }
        return res;

    }
}
