class Solution {
    public int minDominoRotations(int[] A, int[] B) {
        int[] cnt = new int[7];
        for(int i = 0; i < A.length; i ++) {
            if(A[i] == B[i]) {
                cnt[A[i]] ++;
            } else {
                cnt[A[i]] ++;
                cnt[B[i]] ++;
            
}
        }
        
        int tar = -1;
        for(int i = 0; i < 7; i ++) {
            if(cnt[i] == A.length) {
                tar = i;
            }
        }
        if(tar == -1) {
            return -1;
        }
        
        int aans = 0;
        int bans = 0;
        for(int i = 0; i < A.length; i ++) {
            
if(A[i] != tar) {
                aans ++;
            }
            if(B[i] != tar) {
                bans ++;
            }
        }
        return Math.min(aans, bans);
    }
}
