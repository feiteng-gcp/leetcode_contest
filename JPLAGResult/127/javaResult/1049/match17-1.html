class Solution {
    public int minDominoRotations(int[] A, int[] B) {
        int res = Integer.MAX_VALUE;
        
        for (int num = 1; num <= 6; ++num) {
            int cnt = 0;
            int i;
            for (i = 0; i < A.length; ++i) {
                if (A[i] != num && B[i] != num) {
                    break;
                
}
                if (A[i] != num && B[i] == num) {
                    cnt++;
                }
            }
            
            if (i == A.length) {
                res = Math.min(res, cnt);
            }else {
                continue;
            }
            
            cnt = 0;
            int j;
            for (j = 0; j < B.length; ++j) {
                if (B[j] != num && A[j] != num) {
                    break;
                }
                if (B[j] != num && A[j] == num) {
                    cnt++;
                }
            }
            
            if (j == B.length) {
                
res = Math.min(res, cnt);
            }
        }
        
        if (res == Integer.MAX_VALUE) {
            return -1;
        }
        return res;
    }
}
