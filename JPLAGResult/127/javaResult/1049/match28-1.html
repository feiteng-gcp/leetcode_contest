class Solution {
	public int minDominoRotations(int[] A, int[] B) {
		int res = 1000000;
		for (int i = 1; i <= 6; i ++) {
			int tmp = solve(A, B, i);
			res = Math.min(tmp, res);
			tmp = solve(B, A, i);
			res = Math.min(tmp, res);
		}
		if (res == 1000000) return -1;
		return res;
	}


	private int solve(int[] a, int[] b, int v) {
		int res = 0;
		for (int i = 0; i < a.length; i++) {
			if (a[i] != v && b[i] != v) return  1000000;
			if (a[i] != v) res ++;
		}
		return res;
	}

}
