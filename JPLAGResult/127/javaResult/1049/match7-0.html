class Solution {
    public int minDominoRotations(int[] A, int[] B) {
        int ans = Integer.MAX_VALUE;
        for (int num=1; num<=6; num++) {
            int aSwap = A.length;
            int bSwap = B.length;
            boolean flag = true;
            for (int i=0; i<A.length; i++) {
                if (A[i] != num && B[i] != num) {
                    flag = false;
                    break;
                }
                if (A[i] == num)
                    aSwap--;
                if (B[i] == num)
                    bSwap--;
            }
            if (!flag)
                continue;
            ans = Math.min(ans, Math.min(aSwap, bSwap));
        }
        return (ans == Integer.MAX_VALUE)? -1: ans;
    }
}
