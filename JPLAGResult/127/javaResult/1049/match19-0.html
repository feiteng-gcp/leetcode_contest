class Solution {
  public int minDominoRotations(int[] A, int[] B) {
    int c1 = A[0], c2 = B[0];
    for (int i = 1; i < A.length; i++) {
      if (c1 != A[i] && c1 != B[i]) {
        c1 = -1;
      }
      if (c2 != A[i] && c2 != B[i]) {
        c2 = -1;
      }
    }
    if (c1 == -1 && c2 == -1) {
      return -1;
    }
    int minCount = Integer.MAX_VALUE;
    if (c1 != -1) {
      minCount = Math.min(minCount, getCount(A, B, c1));
    }
    if (c2 != -1) {
      minCount = Math.min(minCount, getCount(A, B, c2));
    }
    return minCount;
  }

  private static int getCount(final int[] a, int[] b, int c1) {
    int countA = 0, countB = 0;
    for (int i = 0; i < a.length; i++) {
      if (a[i] != c1) {
        countA++;
      }

      if (b[i] != c1) {
        countB++;
      }
    }
    return Math.min(countA, countB);
  }
}
