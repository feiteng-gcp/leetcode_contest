class Solution {
    public int minDominoRotations(int[] A, int[] B) {
         if (A.length != B.length) {
            return -1;
        }
        if (A.length == 0) {
            return -1;
        }
        int[] is = new int[] {-1, -1,-1, -1,-1, -1,-1};
        for (int i = 0; i < A.length; i++) {
            int a = A[i];
            int b = B[i];
            if (is[a] == i-1) {
                is[a] = i;
            }
            if (is[b] == i-1) {
                is[b] = i;
            }
        }
        //System.out.println(Arrays.toString(is));
        int finalNumber = -1;
        for (int i =0; i< is.length;i++) {
            if (is[i] == A.length - 1) {
                finalNumber = i;
            }
        }
        //System.out.println(finalNumber);

        if (finalNumber == -1) {
            return -1;
        }
        int countA = 0;
        int countB = 0;
        for (int i = 0; i < A.length; i++) {
            int a = A[i];
            int b = B[i];
            if (a != finalNumber && b == finalNumber) {
                countA++;
            }
            if (b != finalNumber && a==finalNumber) {
                countB++;
            }
        }
        //System.out.println(countA);
        //System.out.println(countB);

        return Math.min(countA, countB);
    }
}
