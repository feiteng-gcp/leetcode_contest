	class Solution {
	    public int minDominoRotations(int[] A, int[] B) {
	    	int cost = 9999999;
	    	outer:
	        for(int i = 1;i <= 6;i++){
	        	int n = A.length;
	        	int ca = 0, cb = 0;
	        	for(int j = 0;j < n;j++){
	        		if(A[j] == i || B[j] == i){
	        			if(A[j] != i)ca++;
	        			if(B[j] != i)cb++;
	        		}else{
	        			continue outer;
	        		}
	        	}
	        	cost = Math.min(cost, Math.min(ca, cb));
	        }
	    	return cost == 9999999 ? -1 : cost;
	    }
	}
