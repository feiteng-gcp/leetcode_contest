class Solution {

    public int minDominoRotations(int[] A, int[] B) {
        int min = A.length + 1;
        for (int i = 1; i <= 6; i++) {
            min = Math.min(min, countRotations(A, B, i));
            min = Math.min(min, countRotations(B, A, i));
        }
        return min > A.length ? -1 : min;
    }

    private int countRotations(int[] target, int[] source, int val) {
        int count = 0;
        for (int i = 0; i < target.length; i++) {
            if (target[i] != val && source[i] != val) {
                return target.length + 1;
            }
            if (target[i] == val) {
                continue;
            } else {
                count++;
            }
        }
        return count;
    }
}
