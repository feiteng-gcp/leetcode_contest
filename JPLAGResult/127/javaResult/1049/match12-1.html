class Solution {
    public int minDominoRotations(int[] A, int[] B) {
        int minStep=Integer.MAX_VALUE;
        for(int i=1;i<=6;i++){
            minStep=Math.min(minStep,minStepToMatch(i,A,B));
            minStep=Math.min(minStep,minStepToMatch(i,B,A));
        }
        return minStep==Integer.MAX_VALUE?-1:minStep;
    }
    public int minStepToMatch(int target,int[] targetArray,int[] suppyArray){
        int count=0;
        for(int i=0;i<targetArray.length;i++){
            if(targetArray[i]!=target){
                if(suppyArray[i]!=target){
                    return Integer.MAX_VALUE;
                }
                count++;
            }
        }
        return count;
    }
}
