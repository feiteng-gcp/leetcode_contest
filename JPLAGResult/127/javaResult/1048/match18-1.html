class Solution {
    public int clumsy(int n) {
        
if (n == 3) {
            return 6;
        } else if (n == 2) {
            return 2;
        } else if (n == 1) {
            return 1;
        }
        long res = n * (n - 1) / (n - 2);
        long i = n - 3;
        for (; i >= 4; i -= 4) {
            res += i - (i - 1) * (i - 2) / (i - 3);
        }
        if (i == 3) {
            res += 3 - 2 * 1;
        } else if (i == 2) {
            res += 2 - 1;
        } else if (i == 1) {
            res += 1;
        }
        return (int) res;
    }
}
