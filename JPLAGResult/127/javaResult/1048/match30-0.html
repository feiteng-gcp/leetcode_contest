	class Solution {
	    public int clumsy(int n) {
	    	int ret = 0;
	        for(int i = n;i >= 1;i-=4){
	        	if(i-4 >= 0){
		        	if(i == n){
		        		ret += i*(i-1)/(i-2);
		        	}else{
		        		ret -= i*(i-1)/(i-2);
		        	}
		        	ret += i-3;
	        	}else if(i == 3){
		        	if(i == n){
		        		ret += i*(i-1)/(i-2);
		        	}else{
		        		ret -= i*(i-1)/(i-2);
		        	}
	        	}else if(i == 2){
		        	if(i == n){
		        		ret += i*(i-1);
		        	}else{
		        		ret -= i*(i-1);
		        	}
	        	}else if(i == 1){
		        	if(i == n){
		        		ret += i;
		        	}else{
		        		ret -= i;
		        	}
	        	}
	        }
	        return ret;
	    }
	}
