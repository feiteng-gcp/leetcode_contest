class Solution {
    public int clumsy(int n) {
        long result = 0;
        int sign = 1;
        while(n > 0){
            if(n == 1){
                result += 1l*sign * n;
                break;
            }else{
                int r = n * (n-1);
                if(n-2>0){
                    r/=(n-2);                    
                }
                result += 1L*sign * r;
                if(n-3 > 0){
                    result += n-3;
                }
                n -= 4;
                sign = -1;                
            }
        }
        return (int)result;
    }
    
}
