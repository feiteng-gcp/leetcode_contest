class Solution {
    public int clumsy(int N) {
        long num=N;
        long total=0;
        char sign='+';
        while(num>0){
            long temp=num;
            num--;
            
if(num>0){
                temp=temp*num;
                num--;
            }
            if(num>0){
                temp=temp/num;
                num--;
            }
            if(sign=='-'){
                temp=-temp;
            }
            if(num>0){
                temp=temp+num;
                num--;
            }
            total=total+temp;
            sign='-';
        }
        return (int)total;
    }
}
