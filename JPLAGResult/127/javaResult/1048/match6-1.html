class Solution {
    public int clumsy(int N) {
        int sum = 0;
        int last = 1;
        while (N > 0) {
            if (last == 1) {
                sum = compute(N, last);
                
                last = -1;
            } else {
                sum = sum + compute(N, last);
            
            }
            N = N - 4;
        }
        return sum;
    }
    
    int compute(int N, 
int d) {
        if (N >= 4) {
            return d * N * (N - 1) / (N - 2) + N - 3;
        } else if (N == 3) {
            return d * N * (N - 1) / (N - 2);
        } else if (N == 2) {
            return d * N * (N - 1);
        } else return d * N;
    }
}
