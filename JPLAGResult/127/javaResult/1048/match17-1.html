class Solution {
    //18 - 4 - 2 = 12
    public int clumsy(int N) {
        
        int sum = 0;
        sum += manipulation(N, true);
        N -= 4;
        
        while (N > 0) {
            sum -= manipulation(N, false);
            N -= 4;
        }
        
        return sum;
    }
    
    private int manipulation(int N, 
boolean first) {
        if (N == 1) {
            return 1;
        }
        
        if (N == 2) {
            return 2 * 1;
        }
        
        
if (N == 3) {
            return 3 * 2 / 1;
        }
        
        if (first) {
            return N * (N - 1) / (N - 2) + (N - 3);
        } else {
            return N * (N - 1) / (N - 2) - (N - 3);
        }
    }
}
