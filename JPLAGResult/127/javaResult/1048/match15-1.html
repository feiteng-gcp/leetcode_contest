class Solution {
    public int clumsy(int N) {
        int ans = ele(N);
        int i = 3;
        while(i < N) {
            if(i % 4 == 0) {
                ans -= ele(N - i);
                i += 3;
            } else {
                ans += (N - i);
                i ++;
            }
        }
        
return ans;
    }
    
    private int ele(int x) {
        if(x >= 3) {
            return x * (x - 1) / (x - 2);
        } else if(x >= 2) {
            return x * (x - 1);
        } else {
            return x;
        }
    }
}
