class Solution {

    public int clumsy(int N) {
    	long ans = N , last = N;    	
    	int type = 0;
    	for (int i = N - 1;i >= 1;i --) {
    		if (type == 0) {
    			ans -= last;
    			ans += (last * i);
    			last *= i;    			
    		} else if (type == 1) {
    			ans -= last;
    			ans += (last / i);
    			last /= i;    			
    		} else if (type == 2) {
    			ans += i;
    			last = i;
    		} else {
    			ans -= i;
    			last = - i;
    		}    		
    		type = (type + 1) % 4;
    	}    
    	return (int) ans;
    }
}
