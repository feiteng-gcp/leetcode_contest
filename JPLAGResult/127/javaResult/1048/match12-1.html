class Solution {
    public int clumsy(int N) {
        if (N == 0) {
            return 1;
        }
        
if (N == 1) {
            return 1;
        } else if (N == 2) {
            return 2 * 1;
        } else if (N == 3) {
            return 3 * 2 / 1;
        } else if (N == 4) {
            return 4 * 3 / 2 + 1;
        }
        res = N - 3;
        return (int) (N * (N - 1) / (N - 2) - getClumsy(N - 4) + res);
    }

    int res = 0;

    private double getClumsy(int N) {
        if (N == 0) {
            return 1;
        }
        if (N == 1) {
            return 1;
        } else if (N == 2) {
            return 2 * 1;
        } else if (N == 3) {
            return 3 * 2 / 1;
        } else if (N == 4) {
            res += 1;
            return 4 * 3 / 2;
        }
        res += (N - 3);
        double r = N * (N -1) / (N -2);
        return r + getClumsy(N - 4);
    }
}
