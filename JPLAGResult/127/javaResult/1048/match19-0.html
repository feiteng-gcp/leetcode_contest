class Solution {
    public int clumsy(int N) {
        int result = 0;
        int pos = 0;
        int pre = N;
        for(int i = N - 1; i >= 1; i--) {
            if(pos == 0) pre *= i;
            else if (pos == 1) pre /= i;
            else if(pos == 2) pre += i;
            else {
                result += pre;
                pre = -i;
            } 
            pos++;
            pos %= 4;
        }
        return result + pre;
    }
}
