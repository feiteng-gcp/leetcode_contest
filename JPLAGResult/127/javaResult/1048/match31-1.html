class Solution {
    public int clumsy(int N) {
        int val = N;
        int res = 0;
        if (N >= 4) {
            res += N * (N - 1) / (N - 2) + (N - 3);
        } else if (N >= 3) {
            res += N * (N - 1) / (N - 2);
        } else if (N >= 2) {
            res += N * (N - 1);
        } else if (N >= 1) {
            res += N;
        }
        N -= 4;
        while (N >= 1) 
{
            if (N >= 4) {
                res -= N * (N - 1) / (N - 2) - (N - 3);
            } else if (N >= 3) {
                res -= N * (N - 1) / (N - 2);
            } else if (N >= 2) {
                res -= N * (N - 1);
            } else if (N >= 1) {
                res -= N;
            }
            N -= 4;
        }
        return res;
    }
}
