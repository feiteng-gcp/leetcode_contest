class Solution {
    public int clumsy(int N) {
        
        return c(N);
    }
    
    public int s(int N){
        if(N==1) return 1;
        if(N==2) return 2-1;
        if(N==3) return 3-2*1;
        if(N==4) return 4-3*2/1;
        if(N==5) return 5-4*3/2+1;
        if(N==6)  return 6-5*4/3+2-1;
        
        
        return N-(N-1)*(N-2)/(N-3)+s(N-4);
    }
    
    public int c(int N ){
        if(N==1) return 1;
        if(N==2) return 2*1;
        if(N==3) return 3*2/1;
        if(N==4) return 4*3/2+1;
        if(N==5) return 5*4/3+2-1;
        
        return N*(N-1)/(N-2)+s(N-3);
    }
    
}
