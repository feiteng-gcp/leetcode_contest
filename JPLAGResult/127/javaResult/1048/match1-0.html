class Solution {
	public static final char[] op = new char[] { '*', '/', '+', '-' };

	public int clumsy(int N) {
		int index = 0;
		int product = N;
		int res = 0;
		for (int i = N - 1; i > 0; i--) {
			if (op[index] == '*') {
				product *= i;
			} else if (op[index] == '/') {
				product /= i;
			} else if (op[index] == '+') {
				res += product;
				product = 0;
				res += i;
			} else {
				product = -i;
			}
			index++;
			index %= 4;
		}
		res += product;
		return res;
	}
}
