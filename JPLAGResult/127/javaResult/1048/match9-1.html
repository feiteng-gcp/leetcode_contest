class Solution {
    public int clumsy(int N) {
        if(N==1)return 1;
        if(N==2)return 2;
        if(N==3)return 6;
        int head=N*(N-1)/(N-2);
        int pos=0,neg=0;
        N-=3;
        int cur=0;
        while (cur<N){
            if(cur%4==0){
                pos+=N-cur;
                cur++;
                continue;
            }
            int p=0;
            while(cur<N&&cur%4!=0){
                if(cur%4==1){
                    p+=N-cur;
                    cur++;
                }else if(cur%4==2){
                    p*=(N-cur);
                    cur++;
                }else {
                    p/=(N-cur);
                    cur++;
                }
            }
            neg+=p;
        }
        return head-neg+pos;
    }
}
