class Solution {
  public int clumsy(int N) {
    if (N == 1) return 1;
    if (N == 2) return 2;
    int ans = N * (N - 1) / (N - 2);
    for (int i = N - 3; i > 0; i -= 4) {
      ans += process(i);
    }
    return (int) ans;
  }

  private long process(int start) {
    if (start == 1) return 1;
    return start - (start - 1) * Math.max(1, start - 2) / Math.max(1, start - 3);
  }
}
