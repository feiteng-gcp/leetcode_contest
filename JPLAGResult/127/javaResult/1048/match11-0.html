class Solution {
    public int clumsy(int N) {
        if (N == 0)
            return 1;
        int op = 0; // *:0, /:1, +:2, -:3
        Stack<Integer> stack = new Stack<>();
        stack.push(N);
        for (int i=N-1; i>0; i--) {
            if (op == 0) {
                stack.push(stack.pop() * i);
            } else if (op == 1) {
                stack.push(stack.pop() / i);
            } else if (op == 2) {
                stack.push(i);
            } else
                stack.push(0 - i);
            
            op = (op + 1) % 4;
        }
        int ans = 0;
        while (!stack.isEmpty())
            ans += stack.pop();
        return ans;
    }
}
