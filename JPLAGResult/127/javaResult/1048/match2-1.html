class Solution {
    public int clumsy(int N) {
        int state = 0;
        int result = 0;
        int multiply = 0;
        int sign = 1;
        
        for (;state < N;state++) {
            if (state > 2) sign = -1;
            int num = N - state;
            
            if (state == 0) {
                multiply = num;
                continue;
            }
            
            int stateMod = state % 4;
            if (stateMod == 0) {
                multiply = num;
            
} else if (stateMod == 1) {
                multiply *= num;
            } else if (stateMod == 2) {
                multiply /= num;
                result += sign * multiply;
                multiply = 0;
            } else {
                result += num;
            
}
        }
        
        return result + sign * multiply;
    }
}
