class Solution {
    public int clumsy(int N) {
        Stack<Integer> s = new Stack<>();
        s.push(N);
        int j = 0;
        
for (int i = N - 1; i >= 1; i--) {
            if (j % 4 == 0) {
                s.push(s.pop() * i);
            } else if (j % 4 == 1) {
                s.push(s.pop() / i);
            } else if (j % 4 == 2) {
                s.push(i);
            } else {
                s.push(-i);
            }
            j++;
        }
        
        int res = 0;
        while (!s.isEmpty()) {
            res += s.pop();
        }
        return res;
    }
}
