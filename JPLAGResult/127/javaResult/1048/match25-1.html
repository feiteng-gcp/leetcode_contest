class Solution {
    public int clumsy(int n) {
        if(n==1) return 1;
        if(n==2) return 2;
        
return n*(n-1)/(n-2) + help(n-3,false);
    }
    
    private int help(int n, boolean neg){
        if(n<=0) return 0;
        if(!neg) return n+help(n-1,true);
        
        int res = 0;
        
        
        if(n==1) res = 1;
        else if(n==2) res = 2;
        else res = n*(n-1)/(n-2);
        
        res = -res;
        return res+help(n-3,false);
    }
}
