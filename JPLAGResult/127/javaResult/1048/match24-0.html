class Solution {
    public int clumsy(int N) {
        int times = N / 4;
        int rest = N % 4;
        int restRes = 0;
        if (rest > 0) {
            if (rest == 1) {
                restRes = 1;
            }
            if (rest == 2) {
                restRes = 2;
            }
            if (rest == 3) {
                restRes = 6;
            }
        }
        int res = 0;
        int num = N;
        for (int i = 0; i < times; ++i) {
            int a = num--;
            int b = num--;
            int c = num--;
            int d = num--;
            res += i == 0 ? a * b / c + d : - a * b / c + d;
        }
        if (rest > 0) {
            if (times == 0) {
                return restRes;
            } else {
                res -= restRes;
            }
        }
        return res;
    }
}
