import java.util.Arrays;

class Solution {
	public int clumsy(int N) {
		return clumsy(N, 1);
	}

	private int clumsy(int n, int sign) {
		if (n <= 0)
			return 0;
		int ans = sign;
		for (int i = 0; i < 3 && n >= 1; i++, n--) {
			if (i <= 1)
				ans *= n;
			if (i == 2)
				ans /= n;
		}
		ans += (n--);

		return ans + clumsy(n, -1);
	}

	public static void tr(Object... o) {
		System.out.println(Arrays.deepToString(o));
	}
}
