class Solution {
    public int clumsy(int N) {
        if (N == 4) {
            return 7;
        } else if (N == 3) {
            return 6;
        } else if (N == 2) {
            return 2;
        } else if (N == 1) {
            return 1;
        }
        int result = N * (N-1) / (N-2) + (N-3);
        N -= 4;
        while (N >= 4) {
            result -= N * (N-1) / (N-2);
            result += (N-3);
            N -= 4;
        }
        if (N == 3) {
            return (result - 6);
        } else if (N == 2) {
            return (result - 2);
        } else if (N == 1) {
            return (result - 1);
        }
        return result;
    }
}
