class Solution {
public:
    int minCostToMoveChips(vector<int>& chips) {
        //sort(chips.begin(), chips.end());
        int even_count = 0;
        int odd_count = 0;
        for (int i : chips) {
            if (i%2) {
                odd_count++;
            } else {
                even_count++;
            }
        }
        return min(odd_count, even_count);
    }
};
