class Solution {
public:
    int minCostToMoveChips(vector<int>& chips) {
        int oddCnt = 0, evenCnt = 0;
        for (auto c : chips) {
            if (c % 2 == 0)
                evenCnt++;
            else
                oddCnt++;
        }
        return min(oddCnt, evenCnt);
    }
};
