class Solution {
public:
    int countVowelPermutation(int n) {

        int64_t a = 1, e = 1, i = 1, o = 1, u = 1, mod = 1e9+7;
        for (int k = 1; k < n; k++) {
            int64_t _a = a, _e = e, _i = i, _o = o, _u = u;
            a = (_e + _i + _u) % mod;
            e = (_a + _i) % mod;
            i = (_e + _o) % mod;
            o = _i % mod;
            u = (_i + _o) % mod;
        }
        return (a + e + i + o + u) % mod;
    }
};
