class Solution {
public:
    int countVowelPermutation(int n) {
        vector<long> a(n, 0);
        vector<long> e(n, 0);
        vector<long> i(n, 0);
        vector<long> o(n, 0);
        vector<long> u(n, 0);
        long maxN = 1000000007;

        a[0] = e[0] = i[0] = o[0] = u[0] = 1;
        for (int x = 1; x < n; x++)
        {
            a[x] = (e[x-1] + i[x-1] + u[x-1]) % maxN;
            e[x] = (a[x-1] + i[x-1]) % maxN;
            i[x] = (e[x-1] + o[x-1]) % maxN;
            o[x] = i[x-1] % maxN;
            u[x] = (i[x-1] + o[x-1]) % maxN;
        }
        return (a[n-1] + e[n-1] + i[n-1] + o[n-1] + u[n-1]) % maxN;
    }
};
