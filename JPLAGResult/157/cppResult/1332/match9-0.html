class Solution {
public:
    int countVowelPermutation(int n) {
        // a : 0, e : 1, i : 2, o : 3, u : 4
        int num[n][5], mod = 1E9 + 7;
        // memset(num, 0, n * 5 * sizeof(int));
        for (int i = 0; i < 5; i++) {
            num[0][i] = 1;
        }
        for (int i = 1; i < n; i++) {
            num[i][0] = ((num[i - 1][1] + num[i - 1][2]) % mod + num[i - 1][4]) % mod;
            num[i][1] = (num[i - 1][0] + num[i - 1][2]) % mod;
            num[i][2] = (num[i - 1][1] + num[i - 1][3]) % mod;
            num[i][3] = num[i - 1][2];
            num[i][4] = (num[i - 1][2] + num[i - 1][3]) % mod;
        }
        int sum = 0;
        for (int i = 0; i < 5; i++) {
            sum = (sum + num[n - 1][i]) % mod;
        }
        return sum;
    }
};
