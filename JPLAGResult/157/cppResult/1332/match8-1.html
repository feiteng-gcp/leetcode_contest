class Solution {
public:
    int mod = 1e9 + 7;
    //aiueo
    
    //n=1, a, i, u, e, o
    //n=2, ae, ea, ei, ia, ie, io, iu, oi, ou, ua
    
    //a->e
    //e->a, i
    //i->a, e, o, u
    //o->i, u
    //u->a
    
    //0, 1, 2, 3, 4
    //a, e, i, o, u
    //
    
    int countVowelPermutation(int n) {
        vector<vector<int>> dp(n+1, vector<int>(5));

        for(int i = 0; i < 5; ++i) {
            dp[1][i] = 1;
        };
        for(int i = 2; i <= n; ++i) {
            dp[i][0] = (dp[i][0] + dp[i-1][1]) % mod;
            dp[i][1] = ((dp[i][1] + dp[i-1][0]) % mod + dp[i-1][2]) % mod;
            dp[i][2] = ((((dp[i][2] + dp[i-1][0]) % mod + dp[i-1][1]) % mod + dp[i-1][3]) % mod + dp[i-1][4]) % mod;
            dp[i][3] = ((dp[i][3] + dp[i-1][2]) % mod + dp[i-1][4]) % mod;
            dp[i][4] = (dp[i][4] + dp[i-1][0]) % mod;
        };
        int ans = 0;
        for(int i = 0; i < 5; ++i) {
            ans = (ans + dp[n][i]) % mod;
        };
        return ans;
    }
};
