class Solution {
public:
    int countVowelPermutation(int n) 
    {
        int M = 1e9+7;
        auto dp = vector<vector<int>>(n, vector<int>(5,0));
        for (int i=0; i<5; i++)
            dp[0][i] = 1;
        for (int i=1; i<n; i++)
        {
            dp[i][0] = ((dp[i-1][1] + dp[i-1][2])%M + dp[i-1][4])%M;
            dp[i][1] = (dp[i-1][0] + dp[i-1][2])%M;
            dp[i][2] = (dp[i-1][1] + dp[i-1][3])%M;
            dp[i][3] = dp[i-1][2];
            dp[i][4] = (dp[i-1][2] + dp[i-1][3])%M;
        }
        
        int sum  = 0;
        for (int i=0; i<5; i++)
            sum = (sum+dp[n-1][i])%M;
        return sum;
    }
};
