class Solution {
public:
    int countVowelPermutation(int n) {
        long long int a[20005]={0},e[20005]={0},i[20005]={0},o[20005]={0},u[20005]={0};
        long long int MOD = 1000000007;
        a[1] = e[1] = i[1] = o[1] = u[1] = 1;
        for (int j=2;j<=n;j++) {
            a[j] = (e[j-1]+i[j-1]+u[j-1])%MOD;
            e[j] = (a[j-1]+i[j-1])%MOD;
            i[j] = (e[j-1]+o[j-1])%MOD;
            o[j] = (i[j-1])%MOD;
            u[j] = (i[j-1]+o[j-1])%MOD;
        }
        return (a[n]+e[n]+i[n]+o[n]+u[n])%MOD;
    }
};
