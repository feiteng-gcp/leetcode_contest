class Solution {
public:
    
    const int mod = 1000000007;
    
    int countVowelPermutation(int n) {
        int dp[5][n + 2];
        for (int i = 0; i < 5; i ++) {
            dp[i][1] = 1;
        }
        for (int j = 2; j <= n; j ++) {
            dp[0][j] = dp[1][j - 1] % mod;
            dp[1][j] = (dp[0][j - 1] + dp[2][j - 1]) % mod;
            dp[2][j] = 0;
            for (int i = 0; i < 5; i ++) {
                if (i != 2) {
                    dp[2][j] = (dp[2][j] + dp[i][j - 1]) % mod;
                }
            }
            dp[3][j] = (dp[2][j - 1] + dp[4][j - 1]) % mod;
            dp[4][j] = dp[0][j - 1] % mod;
        }
        int ans = 0;
        for (int i = 0; i < 5; i ++) {
            ans = (ans + dp[i][n]) % mod;
        }
        return ans;
    }
};
