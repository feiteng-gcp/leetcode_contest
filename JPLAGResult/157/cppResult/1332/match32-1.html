class Solution {
	typedef long long ll;
	ll MOD=1e9+7;
public:
	ll dp[100000][6];
    ll add(int k,int a,int e,int i,int o,int u){
    	ll ans=0;
    	if(a==1) ans=(ans+dp[k-1][1])%MOD;
    	if(e==1) ans=(ans+dp[k-1][2])%MOD;
    	if(i==1) ans=(ans+dp[k-1][3])%MOD;
    	if(o==1) ans=(ans+dp[k-1][4])%MOD;
    	if(u==1) ans=(ans+dp[k-1][5])%MOD;
    	return ans;
    }

    int countVowelPermutation(int n) {
    	dp[1][1]=dp[1][2]=dp[1][3]=dp[1][4]=dp[1][5]=1;
        for(int i=2;i<=n;i++){
        	// for(int j=1;j<=5;j++) 
        	// 	dp[i][1]=(dp[i][1]+dp[i-1][j])%MOD;
        	dp[i][1]=add(i,0,1,1,0,1);
        	dp[i][2]=add(i,1,0,1,0,0);
        	dp[i][3]=add(i,0,1,0,1,0);
        	dp[i][4]=add(i,0,0,1,0,0);
        	dp[i][5]=add(i,0,0,1,1,0);
        }
        ll res=0;
        for(int i=1;i<=5;i++){
        	res=(res+dp[n][i])%MOD;
        }
        return res;
    }
};
