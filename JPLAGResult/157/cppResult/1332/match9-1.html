class Solution {
public:
    int countVowelPermutation(int n) {
        long long int const MOD = 1000000007;
        vector<vector<long long int>> dp(5, vector<long long int>(n + 1, 0LL));

        for (int i = 0; i < 5; i++) {
            dp[i][1] = 1LL;
        }
        for (int l = 2; l <= n; l++) {
            dp[0][l] = dp[1][l - 1];
            dp[1][l] = (dp[0][l - 1] + dp[2][l - 1]) % MOD;
            dp[2][l] = (dp[0][l - 1] + dp[1][l - 1] + dp[3][l - 1] + dp[4][l - 1]) % MOD;
            dp[3][l] = (dp[2][l - 1] + dp[4][l - 1]) % MOD;
            dp[4][l] = dp[0][l - 1];
        }

        long long int res = 0;
        for (int i = 0; i < 5; i++) {
            res = (res + dp[i][n]) % MOD;
        }
        return res;
    }
};
