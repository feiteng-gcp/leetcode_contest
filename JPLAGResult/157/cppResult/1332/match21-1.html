class Solution {
public:
    int countVowelPermutation(int n) {
        int modl = 1000000007;
        int dp[2][5];

        for(int i = 0; i < 5; i++)
            dp[1][i]=1;
        for(int i = 2; i <= n; i++){
            dp[i&1][0]=(((dp[(i-1)&1][1]+dp[(i-1)&1][2])%modl)+dp[(i-1)&1][4])%modl;
            dp[i&1][1]=(dp[(i-1)&1][0]+dp[(i-1)&1][2])%modl;
            dp[i&1][2]=(dp[(i-1)&1][1]+dp[(i-1)&1][3])%modl;
            dp[i&1][3]=dp[(i-1)&1][2];
            dp[i&1][4]=(dp[(i-1)&1][2]+dp[(i-1)&1][3])%modl;
        }
        int ans = 0;
        for(int i = 0; i < 5; i++)
            ans = (ans+dp[n&1][i])%modl;
        return ans;
    }
};
