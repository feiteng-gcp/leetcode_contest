class Solution {
public:
    const static long long MOD = 1e9 + 7;
    
    unordered_map<char, long long> m[20005];
    
    int countVowelPermutation(int n) {
        m[0]['a'] = 1;
        m[0]['e'] = 1;
        m[0]['i'] = 1;
        m[0]['o'] = 1;
        m[0]['u'] = 1;
        
        for(int i = 1; i < n; ++ i)
        {
            m[i]['a'] = m[i - 1]['e'];
            m[i]['e'] = m[i - 1]['a'] + m[i - 1]['i'];
            m[i]['i'] = m[i - 1]['a'] + m[i - 1]['e'] + m[i - 1]['o'] + m[i - 1]['u'];
            m[i]['o'] = m[i - 1]['i'] + m[i - 1]['u'];
            m[i]['u'] = m[i - 1]['a'];
            m[i]['a'] %= MOD;
            m[i]['e'] %= MOD;
            m[i]['i'] %= MOD;
            m[i]['o'] %= MOD;
            m[i]['u'] %= MOD;
        }
        
        return (m[n - 1]['a'] + m[n - 1]['e'] + m[n - 1]['i'] + m[n - 1]['o'] + m[n - 1]['u']) % MOD;
    }
};
