class Solution {
public:
    int countVowelPermutation(int n) {
        vector<vector<long long>> dp(5, vector<long long>(n+1));
        long long M = 1000000007;
        for (int i = 0; i < 5; ++i) dp[i][1] = 1;
        
        for (int l = 2; l <= n; ++l) {
            dp[0][l] = ((dp[1][l-1] + dp[2][l-1]) % M + dp[4][l-1]) % M;
            dp[1][l] = (dp[0][l-1] + dp[2][l-1]) % M;
            dp[2][l] = (dp[1][l-1] + dp[3][l-1]) % M;
            dp[3][l] = dp[2][l-1];
            dp[4][l] = (dp[2][l-1] + dp[3][l-1]) % M;
        }
        
        long long ans = 0;
        for (int i = 0; i < 5; ++i) {
            ans = (ans + dp[i][n]) % M;
        }
        return ans;
    }
};
