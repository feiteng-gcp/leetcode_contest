class Solution {
public:
	long long mo=1e9+7;
//	const int N=20005/;
	long long f[20005][6];

    int countVowelPermutation(int n) {
        f[1][1]=f[1][2]=f[1][3]=f[1][4]=f[1][5]=1ll;
        for(int i=2;i<=n;i++){
        	f[i][1]=(f[i-1][2]+f[i-1][3]+f[i-1][5])%mo;
        	f[i][2]=(f[i-1][1]+f[i-1][3])%mo;
        	f[i][3]=(f[i-1][2]+f[i-1][4])%mo;
        	f[i][4]=f[i-1][3];
        	f[i][5]=(f[i-1][3]+f[i-1][4])%mo;
		}        
		return ((f[n][1]+f[n][2]+f[n][3]+f[n][4]+f[n][5])%mo+mo)%mo;
    }
};
