class Solution {
    // a e i o u
    // 0 1 2 3 4
    long long mod = 1e9+7;

public:
    int countVowelPermutation(int n) {
        vector<vector<long long> > dp(n+2, vector<long long>(5));
        for(int i = 0; i < 5; ++i) dp[1][i] = 1;
        for(int i = 1; i < n; ++i){
            // a -> e
            dp[i+1][1] += dp[i][0];
            // e -> a, i
            dp[i+1][0] += dp[i][1];
            dp[i+1][2] += dp[i][1];
            // i -> a, e, o, u
            dp[i+1][0] += dp[i][2];

            dp[i+1][1] += dp[i][2];
            dp[i+1][3] += dp[i][2];
            dp[i+1][4] += dp[i][2];
            // o -> i, u
            dp[i+1][2] += dp[i][3];
            dp[i+1][4] += dp[i][3];
            // u -> a
            dp[i+1][0] += dp[i][4];
            
            
            for(int j = 0; j < 5; ++j) dp[i+1][j] %= mod;
        }
        long long ans = 0;
        for(int i = 0; i < 5; ++i){
            ans += dp[n][i];
        }
        ans %= mod;
        return (int)ans;
    }
};
