class Solution {
public:
	#define mo 1000000007ll
	#define ll long long
	ll f[20005][10];
    int countVowelPermutation(int n) {
    	f[1][1]=f[1][2]=f[1][3]=f[1][4]=f[1][5]=1;
    	//a        e     i        o       u
        for(int i=1;i<n;i++){
        	f[i+1][2]=(f[i+1][2]+f[i][1])%mo;
        	f[i+1][1]=(f[i+1][1]+f[i][2])%mo;
        	f[i+1][3]=(f[i+1][3]+f[i][2])%mo;
        	f[i+1][1]=(f[i+1][1]+f[i][3])%mo;
        	f[i+1][2]=(f[i+1][2]+f[i][3])%mo;
        	f[i+1][4]=(f[i+1][4]+f[i][3])%mo;
        	f[i+1][5]=(f[i+1][5]+f[i][3])%mo;
        	f[i+1][3]=(f[i+1][3]+f[i][4])%mo;
        	f[i+1][5]=(f[i+1][5]+f[i][4])%mo;
        	f[i+1][1]=(f[i+1][1]+f[i][5])%mo;
		}
		ll ans=(f[n][1]+f[n][2]+f[n][3]+f[n][4]+f[n][5])%mo;
		return {ans};
    }
};
