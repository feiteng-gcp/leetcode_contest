class Solution {
public:
    int countVowelPermutation(int n) {
        vector<vector<long long>> dyn(2, vector<long long>(5, 0));
        for(int i=0;i<5;i++)
            dyn[0][i] = 1;
        
        for(int i=1;i<n;i++){
            int cur = i%2;
            int pre = 1-cur;
            dyn[cur][0] = dyn[pre][1];
            dyn[cur][1] = dyn[pre][0]+dyn[pre][2];
            dyn[cur][2] = dyn[pre][0]+dyn[pre][1]+dyn[pre][3]+dyn[pre][4];
            dyn[cur][3] = dyn[pre][2]+dyn[pre][4];
            dyn[cur][4] = dyn[pre][0];
            for(int j=0;j<5;j++)
                dyn[cur][j] %= MOD;
        }
        long long ans = 0;
        for(int i=0;i<5;i++){
            ans += dyn[1-(n%2)][i];
            ans %= MOD;
        }
        return ans;
    }
    
    int MOD = 1000000000 + 7;
};
