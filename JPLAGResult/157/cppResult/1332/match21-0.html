class Solution {
public:
    int countVowelPermutation(int n) {
        vector<vector<int>> d(n, vector<int>(5, 0));
        
        d[0][0] = 1; d[0][1] = 1; d[0][2] = 1; d[0][3] = 1; d[0][4] = 1;
        for (int i = 1; i < n; i++)
        {
            d[i][0] = d[i-1][1];
            d[i][1] = (d[i-1][0] + d[i-1][2])%1000000007;
            d[i][2] = ((d[i-1][0] + d[i-1][1])%1000000007 + (d[i-1][3] + d[i-1][4])%1000000007)%1000000007;
            d[i][3] = (d[i-1][2] + d[i-1][4])%1000000007;
            d[i][4] = d[i-1][0];
        }
        
        int sum = 0;
        for (int i = 0; i < 5; i++) sum = (sum + d[n-1][i])%1000000007;
        
        return sum; 
    }
};
