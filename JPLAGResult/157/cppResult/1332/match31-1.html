class Solution {
#define ll long long
#define MAX 20005
#define P (ll)(1e9+7)
public:
    ll f[MAX][6];
    int countVowelPermutation(int n) {

        f[n][1] = f[n][2] = f[n][3] = f[n][4] = f[n][5] = 1;
        for(int i = n-1; i >= 1; i--){
            f[i][1] = f[i+1][2];
            f[i][2] = (f[i+1][1]+f[i+1][3])%P;
            f[i][3] = (f[i+1][1]+f[i+1][2]+f[i+1][4]+f[i+1][5])%P;
            f[i][4] = (f[i+1][3]+f[i+1][5])%P;
            f[i][5] = f[i+1][1];
        }
        return (f[1][1]+f[1][2]+f[1][3]+f[1][4]+f[1][5])%P;
    }
};
