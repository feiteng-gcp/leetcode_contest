class Solution {
public:
    const int mod = 1e9+7;
    int countVowelPermutation(int n) {
        vector<long> a(n);
        vector<long> e(n);
        vector<long> i(n);
        vector<long> o(n);
        vector<long> u(n);
        // long a[n] = {};
        // long e[n] = {};
        // long i[n] = {};
        // long o[n] = {};
        // long u[n] = {};
        a[0] = e[0] = i[0] = o[0] = u[0] = 1;
        for(int k = 1; k < n; ++k) {
            a[k] = (e[k-1] + i[k-1] + u[k-1])%mod;
            e[k] = (a[k-1] + i[k-1])%mod;
            i[k] = (e[k-1] + o[k-1])%mod;
            o[k] = (i[k-1])%mod;
            u[k] = (i[k-1] + o[k-1])%mod;
            
        }
        
        long res = a[n-1] + e[n-1] + i[n-1] + o[n-1] + u[n-1];
        res = res%mod;
        return res;
        
        
    }
};
