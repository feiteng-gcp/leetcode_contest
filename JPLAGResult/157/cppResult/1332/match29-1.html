class Solution {
    
    static constexpr long mod = 1e9 + 7;
    
public:

    int countVowelPermutation(int n) {
        assert(n >= 1);
        vector<long> endsWithA(n+1, 0);
        vector<long> endsWithE(n+1, 0);
        vector<long> endsWithI(n+1, 0);
        vector<long> endsWithO(n+1, 0);
        vector<long> endsWithU(n+1, 0);
        endsWithA[1] = endsWithE[1] = endsWithI[1] = endsWithO[1] = endsWithU[1] = 1;
        for (int len = 2; len <= n; ++len) {
            endsWithA[len] = (endsWithE[len-1] + endsWithI[len-1] + endsWithU[len-1]) % mod;
            endsWithE[len] = (endsWithA[len-1] + endsWithI[len-1]) % mod;
            endsWithI[len] = (endsWithE[len-1] + endsWithO[len-1]) % mod;
            endsWithO[len] = (endsWithI[len-1]) % mod;
            endsWithU[len] = (endsWithI[len-1] + endsWithO[len-1]) % mod;
        }
        return (endsWithA[n] + endsWithE[n] + endsWithI[n] + endsWithO[n] + endsWithU[n]) % mod;
    }
};
