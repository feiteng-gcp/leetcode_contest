static const int MOD = 1000000007;


class Solution {
public:
    int countVowelPermutation(int n) {
        int64_t sa = 1, se = 1, si = 1, so = 1, su = 1;
        
        for (int i = 1; i < n; ++i) {
            int64_t na = (se + si + su) % MOD,
                    ne = (sa + si) % MOD,
                    ni = (se + so) % MOD,
                    no = (si) % MOD,
                    nu = (si + so) % MOD;
            sa = na, se = ne, si = ni, so = no, su = nu;
        }
        
        return (sa + se + si + so + su) % MOD;
    }
};
