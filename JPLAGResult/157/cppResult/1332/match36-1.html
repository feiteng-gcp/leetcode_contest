class Solution {
public:
    const int mod = 1e9 + 7;
    int countVowelPermutation(int n) {

        vector<vector<int>> dp(n+1, vector<int>(5, 0));
        for(int i = 0; i < 5; ++i)  dp[1][i] = 1;
        for(int i = 1; i < n; ++i){
            dp[i+1][1] = dp[i][0];
            dp[i+1][0] = dp[i][1];
            dp[i+1][2] = dp[i][1];
            for(int j = 0; j < 5; ++j)

                if(j != 2)  dp[i+1][j] = (dp[i+1][j] + dp[i][2]) % mod;
            dp[i+1][2] = (dp[i+1][2] + dp[i][3]) % mod;
            dp[i+1][4] = (dp[i+1][4] + dp[i][3]) % mod;
            dp[i+1][0] = (dp[i+1][0] + dp[i][4]) % mod;
        }
        int ans = 0;
        for(int i = 0; i < 5; ++i)
            ans = (ans + dp[n][i]) % mod;
        return ans;
    }
};
