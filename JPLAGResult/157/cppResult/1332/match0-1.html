class Solution {
public:
    static const int mod = 1e9+7,N = 2e4+7;
    int dp[N][5];
    int countVowelPermutation(int n) {
        for(int i=0;i<5;i++)dp[1][i]=1;
        for(int i=1;i<n;i++){
            for(int j=0;j<5;j++){
                if(dp[i][j]){
                    if(j==0)dp[i+1][1]=(dp[i+1][1]+dp[i][j])%mod;
                    else if(j==1){
                        dp[i+1][0]=(dp[i+1][0]+dp[i][j])%mod;
                        dp[i+1][2]=(dp[i+1][2]+dp[i][j])%mod;
                    }else if(j==2){
                        for(int k=0;k<5;k++)if(k!=2)dp[i+1][k]=(dp[i+1][k]+dp[i][j])%mod;
                    }else if(j==3){
                        dp[i+1][2]=(dp[i+1][2]+dp[i][j])%mod;
                        dp[i+1][4]=(dp[i+1][4]+dp[i][j])%mod;
                    }else{
                        dp[i+1][0]=(dp[i+1][0]+dp[i][j])%mod;
                    }
                }
            }
        }
        int ans=0;
        for(int i=0;i<5;i++)ans=(ans+dp[n][i])%mod;
        return ans;
    }
};
