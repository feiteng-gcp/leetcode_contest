class Solution {
public:
    int countVowelPermutation(int n) {
        vector<long long> dp(5, 1);
        int res = 0, mod = 1e9+7;
        for(int i=1; i<n; i++) {
            vector<long long> tem(5, 0);
            tem[0] = (dp[1] + dp[2] + dp[4]) % mod;
            tem[1] = (dp[0] + dp[2]) % mod;
            tem[2] = (dp[1] + dp[3]) % mod;
            tem[3] = dp[2] % mod;
            tem[4] = (dp[2] + dp[3]) % mod;
            swap(dp, tem);
        }
        for(auto &it:dp) res = (res + it) % mod;
        return res;
    }
};
