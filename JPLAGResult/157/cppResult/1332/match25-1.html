class Solution {
public:
    const int MOD = 1e9 + 7;
    long long d[20005][5];
    
    int countVowelPermutation(int n) {
        memset(d, 0, sizeof(d));

        for (int i = 0; i <= 4; i++) d[1][i] = 1;
        for (int i = 2; i <= n; i++) {
            d[i][0] = d[i - 1][1];
            d[i][1] = (d[i - 1][0] + d[i - 1][2]) % MOD;
            d[i][2] = (d[i - 1][0] + d[i - 1][1] + d[i - 1][3] + d[i - 1][4]) % MOD;
            d[i][3] = (d[i - 1][2] + d[i - 1][4]) % MOD;
            d[i][4] = d[i - 1][0];
        }
        long long res = 0;
        for (int i = 0; i < 5; i++) {
            res = (res + d[n][i]) % MOD;
        }
        return res;
    }
};
