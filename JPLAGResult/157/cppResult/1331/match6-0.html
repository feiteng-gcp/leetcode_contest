class Solution {
public:
    int n , m;
    int getMaximumGold(vector<vector<int>>& grid) {
        // dfs 试试
        int res = 0;
        n = grid.size(); m=grid[0].size();
        vector<vector<bool>> vis(n, vector<bool>(m,false));
        for ( int i=0; i<n; ++i )
            for ( int j=0; j<m; ++j ){
                if ( grid[i][j]>0 ){
                   // vis[i][j] = true;
                    dfs( res, grid, i, j , vis, 0);
                   // vis[i][j] = false;
                }
            }
        return res;
    }
    
    void dfs( int & res, vector<vector<int>>& grid, int i, int j, vector<vector<bool>> & vis, int cur ){
    //    int dir[4][2] = {{1,0},{0,1},{-1,0},{0,-1}};
        if ( i<0 || i>=n || j<0 || j>=m || grid[i][j]<=0 || vis[i][j] ) return ;
        vis[i][j] = true;
        cur += grid[i][j];
        res = max( res, cur );
        dfs( res, grid, i+1, j, vis, cur );
        dfs( res, grid, i, j+1, vis, cur );
        dfs( res, grid, i-1, j, vis, cur );
        dfs( res, grid, i, j-1, vis, cur );
        vis[i][j] = false;
    }
};
