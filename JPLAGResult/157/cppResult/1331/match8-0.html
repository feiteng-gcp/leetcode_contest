class Solution {
public:
    int dx[4]={-1,1,0,0};
    int dy[4]={0,0,1,-1};
    int dfs(int i,int j,int n,int m,vector<vector<int>>& vis,vector<vector<int>>& grid) {
        vis[i][j]=1;
        int ans=grid[i][j];
        int pans=0;
        for(int k=0;k<4;k++) {
            int x=i+dx[k];
            int y=j+dy[k];
            if(x>=0 && x<n && y>=0 && y<m && grid[x][y]!=0 && vis[x][y]==0) {
                vis[x][y]=1;
                pans=max(pans,dfs(x,y,n,m,vis,grid));
                vis[x][y]=0;
            }
        }
        vis[i][j]=0;
        return ans+pans;
    }
    
    int getMaximumGold(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        vector<vector<int>> vis(n,vector<int>(m,0));
        int ans=0;
        for(int i=0;i<n;i++) {
            for(int j=0;j<m;j++) {
                if(grid[i][j]!=0) {
                    cout<<i<<" "<<j<<" "; 
                    cout<<dfs(i,j,n,m,vis,grid)<<"\n";
                    ans=max(ans,dfs(i,j,n,m,vis,grid));
                    vis[i][j]=false;
                }
            }
        }
        return ans;
    }
};
