class Solution {

public:
    int getMaximumGold(vector<vector<int>>& grid) {
        n = grid.size(), m = grid[0].size();
        v.resize(n, vector<bool> (m, false));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {   
                if (grid[i][j]) dfs(i, j, 0, grid);
            }
        }
        return res;
    }
    
private:
    int n, m, res = 0;
    vector<vector<bool>> v;
    vector<int> dx{0,0,-1,1}, dy{-1,1,0,0};
    
    

    void dfs(int i, int j, int sum, vector<vector<int>>& grid) {
        res = max(res, sum += grid[i][j]);
        v[i][j] = true;
        for (int k = 0; k < 4; ++k) {
            int x = i + dx[k], y = j + dy[k];
            if (x < 0 || y < 0 || x >= n || y >= m) continue;
            if(!v[x][y] && grid[x][y]) dfs(x, y, sum, grid);
        }
        v[i][j] = false;
        sum -= grid[i][j];
    } 
};
