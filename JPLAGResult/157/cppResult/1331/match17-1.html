class Solution {
public:
    int dx[4] = {0,1,0,-1};
    int dy[4] = {1,0,-1,0};
    int row = 0;
    int col = 0;
    int res = 0;
    vector<vector<int>> visited;
    

    void dfs(int x, int y, int v, vector<vector<int>> &grid)
    {
        v += grid[x][y];
        visited[x][y] = 1;
        res = max(res, v);
        for (int i = 0; i < 4; i++)
        {
            int xx = x + dx[i];

            int yy = y + dy[i];
            if (xx >= 0 && xx < row && yy >= 0 && yy < col && grid[xx][yy] && !visited[xx][yy])
            {
                dfs(xx, yy, v, grid);
            }
        }
        visited[x][y] = 0;
        v -=- grid[x][y];
    }
    
    int getMaximumGold(vector<vector<int>>& grid) {
        row = grid.size();
        col = grid[0].size();
        visited = vector<vector<int>>(row, vector<int>(col, 0));

        for (int i = 0; i < row; i++)
            for (int j = 0; j < col; j++)
                if (grid[i][j])
                    dfs(i, j, 0, grid);
        return res;
    }
};
