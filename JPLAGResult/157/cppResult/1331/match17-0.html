const int dx[4] = {-1, 1, 0, 0}, dy[4] = { 0, 0, -1, 1 };
class Solution {
public:
    int n, m;
    vector<vector<bool>> vis;
    int best = 0, acc = 0;
    void dfs(vector<vector<int>> &g, int x, int y) {
        acc += g[x][y];
        best = max(acc, best);
        vis[x][y] = true;
        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i], ny = y + dy[i];
            if (nx >= 0 && nx < n && ny >= 0 && ny < m && !vis[nx][ny] && g[nx][ny] > 0) {
                dfs(g, nx, ny);
            }
        }
        vis[x][y] = false;
        acc -= g[x][y];
    }
    int getMaximumGold(vector<vector<int>>& grid) {
        n = grid.size(); m = grid[0].size();
        vis = vector<vector<bool>>(n, vector<bool>(m));
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++)
                if (grid[i][j] > 0)
                    dfs(grid, i, j);
        return best;
    }
};
