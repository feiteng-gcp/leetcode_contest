class Solution {
public:

    int getMaximumGold(vector<vector<int>>& grid) {
        int res = 0;
        for(int i=0;i<grid.size();i++)
        {
            for(int j=0;j<grid[0].size();j++)
            {
                if(grid[i][j] > 0)
                {
                    int x = DFS(grid, i , j);
                    if(x > res) res = x;
                }
                    
            }
        }
        
        return res;
    }
    

    int DFS(vector<vector<int>>& grid , int i , int j)
    {
        if(i < 0 || j < 0 || i == grid.size() || j == grid[0].size() || grid[i][j] <= 0)
            return 0;
        
        int bk =  grid[i][j];
        int res = 0;
        grid[i][j] = -1;
        
        res = bk + max(max(max(DFS(grid, i, j-1) , DFS(grid, i-1, j)) , DFS(grid, i+1, j)) ,DFS(grid, i, j+1)) ;
        grid[i][j] = bk;
        return res;
    }
};
