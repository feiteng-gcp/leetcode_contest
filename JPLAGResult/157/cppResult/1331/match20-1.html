class Solution {
public:
    int dx[4]={1, -1, 0, 0};
    int dy[4]={0, 0, 1, -1};
    void helper(int x, int y, int g, int& ret, vector<vector<int>>& grid) {
        if (grid[x][y]<500 && grid[x][y]!=0) {
            g+=grid[x][y];
            ret=max(ret, g);

            grid[x][y]+=500;
            for (int k=0; k<4; k++) {
                int xx=x+dx[k];
                int yy=y+dy[k];
                if (xx>=0 && xx<grid.size() && yy>=0 && yy<grid[0].size()) {
                    helper(xx, yy, g, ret, grid);
                }
            }
            grid[x][y]-=500;
            g-=grid[x][y];
        }

    } 
    int getMaximumGold(vector<vector<int>>& grid) {
        int ret=0;
        for (int i=0; i<grid.size(); i++) {
            for (int j=0; j<grid[i].size(); j++) {
                if (grid[i][j]!=0) {
                    helper(i, j, 0, ret, grid);
                }
            }
        }
        return ret;
    }
};
