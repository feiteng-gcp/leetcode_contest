int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};

class Solution {
    int n, m;
    int res;
    void solve(int x, int y, int cur, vector<vector<int>> &v, vector<vector<int>>& g) {
        v[x][y] = 1;
        cur += g[x][y];
        res = max(res, cur);
        for (int i = 0; i < 4; ++i) {
            int xx = x + dx[i];
            int yy = y + dy[i];
            if (xx < 0 || xx >= n || yy < 0 || yy >= m || !g[xx][yy] || v[xx][yy]) continue;
            solve(xx, yy, cur, v, g);
        }
        v[x][y] = 0;
    }
public:
    int getMaximumGold(vector<vector<int>>& g) {
        res = 0;
        n = g.size();
        m = g[0].size();
        vector<vector<int>> v(n, vector<int>(m, 0));
        for (int i = 0; i < n; ++i) for (int j = 0; j < m; ++j) if (g[i][j]) solve(i, j, 0, v, g);
        return res;
    }
};
