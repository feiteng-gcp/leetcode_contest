class Solution {
public:
    vector< vector< int>> g;
    int n, m;
    
    int foo(int x, int y) {
        if (x < 0 or y < 0 or x >= n or y >= m) return 0;

        if (g[x][y] == 0) return 0;
        int rv = 0;
        int here = g[x][y];
        g[x][y] = 0;
        rv = max(rv, here + foo(x - 1, y));
        rv = max(rv, here + foo(x + 1, y));
        rv = max(rv, here + foo(x, y - 1));
        rv = max(rv, here + foo(x, y + 1));
        g[x][y] = here;
        return rv;
    }
    int getMaximumGold(vector<vector<int>>& grid) {
        g = grid;
        n = grid.size();
        m = grid[0].size();
        

        int result = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                result = max(result, foo(i, j));
            }
        }
        return result;
    }
};
