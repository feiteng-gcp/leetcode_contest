class Solution {
public:
    int X[4]={0,0,1,-1};
    int Y[4]={1,-1,0,0};
    int go(vector<vector<int> > &grid,int x,int y){
        int tmp=grid[x][y];
        grid[x][y]=0;

        int ans=0;
        for(int i = 0;i<4;i++){
            int xx=X[i]+x;
            int yy=Y[i]+y;
            if(xx>=0&&xx<grid.size()&&yy>=0&&yy<grid[0].size()&&grid[xx][yy]!=0){
                ans=max(ans,go(grid,xx,yy));
            }
        }
        grid[x][y]=tmp;
        return ans+tmp;
    }
    int getMaximumGold(vector<vector<int>>& grid) {

        int ans=0;
        for(int i = 0;i<grid.size();i++){
            for(int j=0;j<grid[0].size();j++){  
                if(grid[i][j]!=0){
                    ans=max(ans,go(grid,i,j));
                }
            }
        }
        return ans;
    }
};
