class Solution {
public:
    int dx[4] = {1, -1, 0, 0};
    int dy[4] = {0, 0, 1, -1};
    void dfs(const vector<vector<int>>& input, int i, int j, set<pair<int, int>>& visited, int& current_sum, int& ans) {
        current_sum += input[i][j];
        visited.insert({i, j});
        ans = max(ans, current_sum);
        for (int k = 0; k < 4; k ++) {
            int newx = i + dx[k];
            int newy = j + dy[k];
            if (newx >= 0 && newy >= 0 && newx < input.size() && newy < input[0].size() 
               && input[newx][newy] > 0 && visited.find({newx, newy}) == visited.end()) {
                dfs(input, newx, newy, visited, current_sum, ans);
            }
        }
        visited.erase({i, j});
        current_sum -= input[i][j];
    }
    
    int getMaximumGold(vector<vector<int>>& grid) {
        int ans = INT_MIN;
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[i].size(); j++) {
                set<pair<int, int>> visited;
                int tmp_ans = 0;
                int tmp = 0;
                dfs(grid, i, j, visited, tmp, tmp_ans);
                ans = max(ans, tmp_ans);
            }
        }    
        return ans;
    }
};
