typedef pair<int, int> ii;
const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
bool flag[20][20];

class Solution {
public:
    vector<vector<int>> a;
    int n, m;
    int search(int x, int y) {

        flag[x][y] = true;
        int ret = 0;
        for (int k = 0; k < 4; ++k) {
            int nx = x + d[k][0];
            int ny = y + d[k][1];
            if (nx >= 0 && nx < n && ny >= 0 && ny < m && a[nx][ny] && !flag[nx][ny]) {
                ret = max(ret, search(nx, ny));
            }
        }
        flag[x][y] = false;
        return ret + a[x][y];
    }
    int getMaximumGold(vector<vector<int>>& a) {
        this->a = a;
        this->n = a.size(), this->m = a[0].size();
        memset(flag, 0, sizeof(flag));

        int best = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (a[i][j] == 0) continue;
                int cur = search(i, j);
                best = max(best, cur);
            }
        }
        return best;
    }
};
