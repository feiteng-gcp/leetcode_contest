class Solution {
public:
    int n,m;
    int a[20][20];
    int b[20][20];
    int ans;

    int dir[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    

    void dfs(int x, int y, int now){
        now += a[x][y];
        a[x][y] = 0;
        ans = max(ans, now);
        for(int i = 0; i < 4; ++i){
            int xx = x+dir[i][0];
            int yy = y+dir[i][1];
            if(xx >= 1 && xx <= n && yy >= 1 && yy <= m && a[xx][yy]) {
                dfs(xx, yy, now);
            }
        }
        a[x][y] = b[x][y];
    }
    
    
    int getMaximumGold(vector<vector<int>>& grid) {

        n = grid.size();
        m = grid[0].size();
        for(int i = 1; i <= n; ++i){
            for(int j = 1; j <= m; ++j){
                b[i][j] = grid[i-1][j-1];
                a[i][j] = grid[i-1][j-1];
            }
        }
        ans = 0;
        for(int i = 1; i <= n; ++i){
            for(int j = 1; j <= m; ++j){
                if(a[i][j]) {
                    dfs(i, j, 0);
                }
            }
        }
        return ans;
    }
};
