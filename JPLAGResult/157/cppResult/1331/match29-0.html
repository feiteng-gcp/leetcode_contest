class Solution {
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        int M = grid.size();
        int N = grid[0].size();
        int ans = 0;
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                if (grid[i][j])
                    ans = max(ans, dfs(grid, i, j, 0));
            }
        }
        return ans;
    }
    
    int dfs(vector<vector<int>>& grid, int r, int c, int gold) {
        int dirs[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
        gold += grid[r][c];
        int old = grid[r][c];
        grid[r][c] = 0;
        
        int ans = gold;
        for (int d = 0; d < 4; ++d) {
            int rr = r + dirs[d][0];
            int cc = c + dirs[d][1];
            if (rr < 0 || rr == grid.size() || cc < 0 || cc == grid[0].size() || grid[rr][cc] == 0) continue;
            ans = max(ans, dfs(grid, rr, cc, gold));
        }
        grid[r][c] = old;
        
        return ans;
    }
};
