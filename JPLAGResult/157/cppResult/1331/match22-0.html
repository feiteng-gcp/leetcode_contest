class Solution {
public:
    int ans;
    int dir[4][2] = {{0,1}, {1,0}, {-1,0}, {0,-1}};
    

    
    void getMax(vector<vector<int>> &grid, int x, int y, int c){
        int v = grid[x][y];
        ans = max(ans, c+v);
        for(int d = 0; d<4; d++){
            int xx = x+dir[d][0];
            int yy = y+dir[d][1];
            if(xx>=0 && xx<grid.size() && yy>=0 && yy<grid[0].size()){
                if(grid[xx][yy]!=0){
                    grid[x][y] = 0;
                    getMax(grid, xx, yy, c+v);
                    grid[x][y] = v;
                }
            }
        }
    }
    
    int getMaximumGold(vector<vector<int>>& grid) {
        ans = 0;
        for(int  i = 0;i <grid.size(); i++){
            for(int j = 0; j<grid[0].size(); j++){
                if(grid[i][j]!=0)
                    getMax(grid, i, j, 0);
            }
        }
        return ans;
    }
};
