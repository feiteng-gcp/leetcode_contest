class Solution {
 public:
  int getMaximumGold(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    int res = 0;
    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        int cur_res = 0;
        dfs(grid, i, j, 0, cur_res);
        res = max(res, cur_res);
      }
    }
    return res;
  }
 private:
  void dfs(vector<vector<int>> &grid, int i, int j, int gold, int &cur_res) {
    if (i < 0 || i >= grid.size() || j < 0 || j >= grid[0].size() || grid[i][j] == 0) {
      return;
    }
    int temp = grid[i][j];
    gold += temp;
    cur_res = max(cur_res, gold);
    grid[i][j] = 0;
    dfs(grid, i + 1, j, gold, cur_res);
    dfs(grid, i - 1, j, gold, cur_res);
    dfs(grid, i, j + 1, gold, cur_res);
    dfs(grid, i, j - 1, gold, cur_res);
    grid[i][j] = temp;
  }
};
