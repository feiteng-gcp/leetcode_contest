class Solution {
public:
    int n, m;
    int ans = 0;
    int dx[4] = {-1, 1, 0, 0};
    int dy[4] = {0, 0, -1, 1};
    
    void dp(vector<vector<int>>& grid, int sum, int x, int y) {
        ans = max(ans, sum);
        for (int i = 0; i <= 3; i++) {
            int newX = x + dx[i];
            int newY = y + dy[i];
            if (0 <= newX && newX < n && 0 <= newY && newY < m && grid[newX][newY]) {
                int tmp = grid[newX][newY];
                grid[newX][newY] = 0;
                dp(grid, sum + tmp, newX, newY);
                grid[newX][newY] = tmp;
            }
        }
    }
    int getMaximumGold(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j]) {
                    int tmp = grid[i][j];
                    grid[i][j] = 0;
                    dp(grid, tmp, i, j);
                    grid[i][j] = tmp;
                }
            }
        }
        return ans;
    }
};
