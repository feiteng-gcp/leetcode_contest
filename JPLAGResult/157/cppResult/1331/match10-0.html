class Solution {
public:
    vector<int> nei = {1, -1, 0, 0};
    vector<int> nej = {0, 0, 1, -1};
    
    bool check(vector<vector<int>> &grid, int i, int j){
        if((0 <= i) && (i < grid.size()) && (0 <= j) && (j < grid[i].size()) && (grid[i][j] != 0)){
            return true;
        }
        return false;
    }
    
    void solve(vector<vector<int>> &grid, set<pair<int, int>> &s, int i, int j, int &maxi, int curr){
        int val = grid[i][j];
        curr += val;
        s.insert(make_pair(i, j));
        bool visit = false;
        for(int k = 0; k < nei.size(); k++){
            int x = i + nei[k];
            int y = j + nej[k];
            if(check(grid, x, y) && (s.find(make_pair(x, y)) == s.end())){
                visit = true;
                solve(grid, s, x, y, maxi, curr);
            }
        }
        s.erase(make_pair(i, j));
        if(!visit){
            maxi = max(maxi, curr);
        }
    }
    
    int getMaximumGold(vector<vector<int>>& grid) {
        set<pair<int, int>> s;
        int maxi = 0;
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid[i].size(); j++){
                if(grid[i][j] != 0){
                    solve(grid, s, i, j, maxi, 0);
                }
            }
        }
        return maxi;
    }
};
