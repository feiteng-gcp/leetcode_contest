class Solution {
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        int m = 0;
        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[i].size(); j++) {
                m = max(m, getMaximumGold(grid, i, j));
            }
        }
        return m;
    }
private:
    int getMaximumGold(vector<vector<int>>& grid, int x, int y) {
        if (grid[x][y] == 0) {
            return 0;
        }
        int g = grid[x][y];
        grid[x][y] = 0;
        int m = 0;
        for (const auto& d : vector<pair<int, int>>{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}) {
            auto dx = d.first;
            auto dy = d.second;
            if (x + dx < 0 || x + dx >= grid.size()) {
                continue;
            }
            if (y + dy < 0 || y + dy >= grid[x + dx].size()) {
                continue;
            }
            m = max(m, getMaximumGold(grid, x + dx, y + dy));
        }
        grid[x][y] = g;
        return g + m;
    }
};
