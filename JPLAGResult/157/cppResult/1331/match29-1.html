class Solution {
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        int rows = grid.size();
        int cols = grid[0].size();
        int result = 0;
        for (int i = 0; i < rows; ++i) {
            for (int j = 0; j < cols; ++j) {

                result = max(result, step(grid, i, j, 0));
            }
        }
        return result;
    }
    
    int step(vector<vector<int>>& grid, int y, int x, int gold) {
        int rows = grid.size();
        int cols = grid[0].size();
        if (y < 0 || y >= rows || x < 0 || x >=cols || grid[y][x] == 0)
            return gold;
        int val = grid[y][x];
        grid[y][x] = 0;
        gold += val;
        int result = gold;
        result = max(result, step(grid, y - 1, x, gold));
        result = max(result, step(grid, y + 1, x, gold));
        result = max(result, step(grid, y, x - 1, gold));
        result = max(result, step(grid, y, x + 1, gold));
        grid[y][x] = val;
        return result;
    }
};
