class Solution {
private:
    int ans = 0;
    int r, c;
    const int ds[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    
    void dfs(int cur, int x, int y, vector<vector<int>>& grid) {
        int val = grid[x][y];
        cur += val;
        grid[x][y] = 0;
        ans = max(ans, cur);
        for (auto & d : ds) {
            int a = x + d[0];
            int b = y + d[1];
            if (a >= 0 && a < r && b >= 0 && b < c && grid[a][b]) {
                dfs(cur, a, b, grid);
            }
        }
        grid[x][y] = val;
    }
    
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        r = grid.size();
        c = grid[0].size();
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                if (grid[i][j]) {
                    dfs(0, i, j, grid);
                }
            }
        }
        return ans;
    }
};
