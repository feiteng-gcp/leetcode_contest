class Solution {
public:
    int m;
    int n;
    vector<vector<int>> directs = {{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
    // int encode(int i, int j) {
    //     return (i + 1) * 100 + j;
    // }
    // unordered_map<int, int> mem;
    int getMaximumGold(vector<vector<int>>& grid) {
        m = grid.size();
        n = grid[0].size();

        int result = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] != 0) {
                    result = max(result, dfs(i, j, grid));
                }
            }
        }
        return result;
    }
    int dfs(int i, int j, vector<vector<int>>& grid) {
        // int state = encode(i, j);
        // if (mem.count(state)) {
        //     return mem[state];
        // }
        
        
        int gold = grid[i][j];
        int result = gold;
        grid[i][j] = 0;
        

        for (auto const & direct : directs) {
            int ii = direct[0] + i;
            int jj = direct[1] + j;
            if (ii >= 0 && ii < m && jj >= 0 && jj < n && grid[ii][jj] != 0) {
                // cout << ii << "," << jj << ":" << dfs(ii, jj, grid) << endl;
                result = max(result, gold + dfs(ii, jj, grid));
            }
        }
        grid[i][j] = gold;
        // mem[state] = result;
        return result;
    }
};
