class Solution {
public:
    int n,m,ans,f[4][2]={1,0,-1,0,0,1,0,-1};
    bool vis[20][20];
    bool judge(int x,int y){
        if(x<0||x>=n||y<0||y>=m)return false;
        return true;
    }
    void dfs(int x,int y,int now,vector<vector<int>>& grid){
        if(vis[x][y])return;
        vis[x][y]=1;
        now+=grid[x][y];
        if(now>ans)ans=now;
        for(int i=0;i<4;i++){
            int xx=f[i][0]+x,yy=f[i][1]+y;
            if(judge(xx,yy)&&grid[xx][yy]>0)dfs(xx,yy,now,grid);
        }
        vis[x][y]=0;
        now-=grid[x][y];
    }
    int getMaximumGold(vector<vector<int>>& grid) {
        n=grid.size();m=grid[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]>0){
                    dfs(i,j,0,grid);
                }
            }
        }
        return ans;
    }
};
