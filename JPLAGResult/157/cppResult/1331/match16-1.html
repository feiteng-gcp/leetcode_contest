class Solution {
public:
    int n,m;
    int res=0;
    int ans=0;

    vector<vector<int>> dir={{1,0},{-1,0},{0,1},{0,-1}};
    
    void dfs(int i,int j,vector<vector<int>>& grid){
        ans+=grid[i][j];
        //cout<<i<<" "<<j<<" "<<ans<<endl;

        int tmp=grid[i][j];
        grid[i][j]=0;
        for(auto e:dir){
            int x=i+e[0];
            int y=j+e[1];
            if(x>=0&&x<n&&y>=0&&y<m&&grid[x][y]>0){
                dfs(x,y,grid);
            }
        }
        res=max(res,ans);
        grid[i][j]=tmp;
        ans-=tmp;
    }
    

    int getMaximumGold(vector<vector<int>>& grid) {
        n=grid.size();
        m=grid[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]!=0){
                    ans=0;
                    dfs(i,j,grid);
                    res=max(res,ans);
                }
            }
        }
        return res;
    }
};
