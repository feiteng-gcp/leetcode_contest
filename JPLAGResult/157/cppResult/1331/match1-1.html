class Solution {
    int ret = 0;
public:
    int getMaximumGold(vector<vector<int>>& grid) 
    {
        int m = grid.size();
        int n = grid[0].size();
        for (int i=0; i<m; i++)
            for (int j =0; j<n; j++)
            {
                if (grid[i][j]==0)
                    continue;       
                
                DFS(i,j,0,grid);                
            }
        return ret;
    }
    
    void DFS(int i, int j, int cur, vector<vector<int>>& grid)
    {
        int m = grid.size();
        int n = grid[0].size();
        auto dir = vector<pair<int,int>>({{1,0},{-1,0},{0,1},{0,-1}});
        
        cur += grid[i][j];
        int temp = grid[i][j];
        grid[i][j] = 0;
        ret = max(ret, cur);
        
        
        for (int k=0; k<4; k++)
        {
            int x = i+dir[k].first;
            int y = j+dir[k].second;
            
            if (x<0||x>=m||y<0||y>=n)
                continue;            
            if (grid[x][y]==0)
                continue;
            
            DFS(x,y,cur,grid);            
        }
        
        grid[i][j] = temp;
    }
};
