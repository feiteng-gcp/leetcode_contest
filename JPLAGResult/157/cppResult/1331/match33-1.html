class Solution {
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<bool>> visited(n, vector<bool>(m, false));
        ans = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]>0){
                    visited[i][j] = true;
                    dfs(i, j, grid[i][j], grid, visited);
                    visited[i][j] = false;
                }
            }
        }
        return ans;
    }
    

    void dfs(int x, int y, int cur, vector<vector<int>> &grid, vector<vector<bool>> &visited){
        ans = max(ans, cur);
        for(int i=0;i<4;i++){
            int nx = x + xx[i], ny = y + yy[i];
            if(nx >= 0 && nx < grid.size() && ny >= 0 && ny < grid[0].size() && grid[nx][ny] > 0 && !visited[nx][ny]){
                visited[nx][ny] = true;
                dfs(nx, ny, cur+grid[nx][ny], grid, visited);
                visited[nx][ny] = false;
            }
        }
    }
    
    int ans, xx[4] = {-1,0,1,0}, yy[4] = {0,1,0,-1};
};
