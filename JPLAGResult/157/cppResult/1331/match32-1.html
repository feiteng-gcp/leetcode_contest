class Solution {
    int ans = 0;
    vector<int> dir = { 0, 1, 0, -1, 1, 0, -1, 0 };
    void dfsHelper(int x, int y, int total, vector<vector<int>>& grid) {
        total += grid[x][y];
        ans = max(ans, total);

        int tmp = grid[x][y];
        grid[x][y] = 0;
        for (int i = 0; i < 4; i++) {
            int new_x = x + dir[2 * i];
            int new_y = y + dir[2 * i + 1];
            if (new_x >= 0 && new_x < grid.size() && new_y >= 0 && new_y < grid[new_x].size() && grid[new_x][new_y] > 0) {
                dfsHelper(new_x, new_y, total, grid);
            }
        }
        grid[x][y] = tmp;
    }
public:
    int getMaximumGold(vector<vector<int>>& grid) {

        for (int i = 0; i < grid.size(); i++) {
            for (int j = 0; j < grid[i].size(); j++) {
                if (grid[i][j] > 0) {
                    dfsHelper(i, j, 0, grid);
                }
            }
        }
        return ans;
    }
};
