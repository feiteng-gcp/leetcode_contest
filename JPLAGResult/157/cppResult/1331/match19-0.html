class Solution {
    int ans,n,m;
    void dfs(int x,int y,int g,vector<vector<int>>& grid){
        ans=max(ans,g);
        const int dx[4]{1,0,-1,0},dy[4]{0,1,0,-1};
        for(int d=0;d<4;d++){
            int xx=x+dx[d],yy=y+dy[d];
            if(xx>=0&&xx<n&&yy>=0&&yy<m&&grid[xx][yy]>0){
                int tmp=grid[xx][yy];
                grid[xx][yy]=0;
                dfs(xx,yy,g+tmp,grid);
                grid[xx][yy]=tmp;
            }
        }
    }
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        n=grid.size();
        m=grid[0].size();
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                if(grid[i][j]>0){
                    int tmp=grid[i][j];
                    grid[i][j]=0;
                    dfs(i,j,tmp,grid);
                    grid[i][j]=tmp;
                }
        return ans;
    }
};
