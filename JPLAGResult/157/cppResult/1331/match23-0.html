class Solution {
private:
    int ans = 0;
public:
    void dfs(vector<vector<bool>>& v,vector<vector<int>>& g,int r,int c,int x,int y,int val){
        if(x<0||y<0||x>=r||y>=c || v[x][y] || g[x][y] == 0){
            ans = max(ans,val);
            return;
        } 
        v[x][y] = true;
        dfs(v,g,r,c,x+1,y,val+g[x][y]);
        dfs(v,g,r,c,x-1,y,val+g[x][y]);
        dfs(v,g,r,c,x,y+1,val+g[x][y]);
        dfs(v,g,r,c,x,y-1,val+g[x][y]);
        v[x][y] = false;
    }
    int getMaximumGold(vector<vector<int>>& g) {
        int r = g.size();
        int c = g[0].size();
        for(int i = 0;i<r;i++){
            for(int j = 0;j<c;j++){
                vector<vector<bool>> v(r,vector<bool>(c,false));
                dfs(v,g,r,c,i,j,0);
            }
        }
        return ans;
    }
};
