class Solution {
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        int M = grid.size(), N = grid[0].size(), max_res = 0;
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                if (grid[i][j]) {
                    int res = 0;
                    set<pair<int, int>> visited = {};
                    visited.insert({i, j});
                    dfs(grid, i, j, res, grid[i][j], M, N, visited);
                    max_res = max(max_res, res);
                }
            }
        }
        return max_res;
    }
    
private:
    int dirs[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    
    void dfs(vector<vector<int>>& grid, int i, int j, int& res, int gold, int M, int N, set<pair<int, int>>& visited) {
        res = max(res, gold);
        for (int d = 0; d < 4; ++d) {
            int ni = i + dirs[d][0], nj = j + dirs[d][1];
            if (ni >= 0 && ni < M && nj >= 0 && nj < N && grid[ni][nj] && visited.find({ni, nj}) == visited.end()) {
                visited.insert({ni, nj});
                dfs(grid, ni, nj, res, gold + grid[ni][nj], M, N, visited);
                visited.erase({ni, nj});
            }
        }
    }
};
