class Solution {
public:
    int dir[4][2] = {0, 1, 0, -1, 1, 0, -1, 0};
    int getMaximumGold(vector<vector<int>>& grid) {
        vector<vector<int>> visit(grid.size(), vector<int>(grid[0].size(), 0));
        int re = 0;
        for(int i = 0; i < grid.size(); ++i) {
            for(int j = 0; j < grid[0].size(); ++j) {
                if(grid[i][j] != 0) {
                    re = max(re, dfs(i, j, grid, visit));
                }
            }
        }
        return re;
    }
    int dfs(int i, int j, vector<vector<int>>& grid, vector<vector<int>>& visit) {
        int re = grid[i][j];
        visit[i][j] = 1;
        for(int l = 0; l < 4; ++l) {
            int ii = i + dir[l][0];
            int jj = j + dir[l][1];
            if(ii>=0 && ii<grid.size() && jj>=0 && jj<grid[0].size() && grid[ii][jj]!=0 && visit[ii][jj] == 0) {
                re = max(re, dfs(ii, jj, grid, visit) + grid[i][j]);
            }
        }
        visit[i][j] = 0;
        return re;
    }
};
