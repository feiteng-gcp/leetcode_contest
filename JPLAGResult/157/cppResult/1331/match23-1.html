class Solution {
public:
    int getMaximumGold(vector<vector<int>>& grid) {
        int h = grid.size();

        int w = grid[0].size();
        vector<vector<bool>> flag(h, vector<bool>(w, false));
        int res = 0;
        for (int y = 0; y < h; ++y) {
            for (int x = 0; x < w; ++x) {
                if (grid[y][x] == 0) {
                    continue;
                }
                dfs(grid, flag, h, w, y, x, 0, res);
            }
        }
        return res;
    }
    

    void dfs(vector<vector<int>>& grid, vector<vector<bool>>& flag, int h, int w, int starty, int startx, int cur, int& res) {
        if (starty >= h || startx >= w || starty < 0 || startx < 0 || grid[starty][startx] == 0 || flag[starty][startx]) {
            res = max(res, cur);
            return;
        }
        //cout << starty << " " << startx << endl;
        flag[starty][startx] = true;
        cur += grid[starty][startx];
        dfs(grid, flag, h, w, starty-1, startx, cur, res);
        dfs(grid, flag, h, w, starty+1, startx, cur, res);
        dfs(grid, flag, h, w, starty, startx-1, cur, res);
        dfs(grid, flag, h, w, starty, startx+1, cur, res);
        //grid[starty][startx] = 0;
        flag[starty][startx] = false;
    }
};
