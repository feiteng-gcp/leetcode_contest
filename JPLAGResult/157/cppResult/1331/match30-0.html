class Solution {
public:
    
    vector<int> row = {-1,0,1,0};
    vector<int> col = {0,1,0,-1};
    
    int dfs(vector<vector<int>> &grid, int i, int j)
    {
        if(grid[i][j]==0)
        {
            return 0;
        }
        int k = grid[i][j];
        grid[i][j]=0;
        int y=0;
        for(int x=0;x<4;x++)
        {
            if(i+row[x]>=0 && j+col[x]>=0 && i+row[x]<grid.size() && j+col[x]<grid[0].size() && grid[i+row[x]][j+col[x]]!=0)
            {
                y = max(y,dfs(grid,i+row[x],j+col[x]));
            }
        }
        grid[i][j]=k;
        return k+y;
    }
    
    int getMaximumGold(vector<vector<int>>& grid) {
        int ans=0;
        for(int i=0;i<grid.size();i++)
        {
            for(int j=0;j<grid[0].size();j++)
            {
                if(grid[i][j]!=0)
                {
                    ans = max(ans,dfs(grid,i,j));
                }
            }
        }
        return ans;
    }
};
