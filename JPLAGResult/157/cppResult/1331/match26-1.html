class Solution {
public:
    //use a brutal dfs to solve this problem
    int getMaximumGold(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<bool>> visited(n, vector<bool>(m, false));
        int res = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] > 0) {
                    findMaxPath(grid, visited, i, j, res, 0);
                }

            }
        }
        return res;
    }
    
private:
    void findMaxPath(vector<vector<int>>& grid, vector<vector<bool>>& visited, int row, int col, int& res, int cur) {
        int n = grid.size(), m = grid[0].size();
        cur += grid[row][col];
        visited[row][col] = true;
        res = max(res, cur);
        vector<int> dirX{1, 0, -1, 0};
        vector<int> dirY{0, 1, 0, -1};
        for (int i = 0; i < 4; ++i) {
            int r = row + dirX[i];
            int c = col + dirY[i];
            if (r >= 0 && r < n && c >= 0 && c < m && grid[r][c] > 0 && !visited[r][c]) {
                findMaxPath(grid, visited, r, c, res, cur);
            }
        }
        visited[row][col] = false;
    }
};
