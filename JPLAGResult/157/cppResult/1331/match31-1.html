int vis[15][15];

const int mv[4][2] = {{0,1},{1,0},{0,-1},{-1,0}};
class Solution {
public:
	int r,c;

	int dfs(int x, int y, vector<vector<int>>& g){
		if(vis[x][y]) return 0;

		vis[x][y] = 1;
		int m = 0;
		for(int i=0;i<4;++i){
			int nx = x + mv[i][0];
			int ny = y + mv[i][1];
			if(nx >= 0 && ny >= 0 && nx < r && ny < c){
				m = max(m, dfs(nx, ny, g));
			}
		}
		vis[x][y] = 0;
		return g[x][y] + m;
	}

    int getMaximumGold(vector<vector<int>>& grid) {
        r = grid.size();
        c = grid[0].size();
        memset(vis, 0, sizeof(vis));
        for(int i=0;i<r;++i){
        	for(int j=0;j<c;++j)
        		if(grid[i][j] == 0) vis[i][j] = 1;
        }

        int ans = 0;
        for(int i=0;i<r;++i){
        	for(int j=0;j<c;++j){
        		if(vis[i][j] == 0) ans = max(ans, dfs(i,j,grid));
        	}
        }
        return ans;
    }
};
