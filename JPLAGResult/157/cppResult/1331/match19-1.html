class Solution {
private:
    int ans, m, n;
    int dirs[5] = {0, 1, 0, -1, 0};
public:
    void dfs(int x, int y, int glod, vector<vector<int>>& grid){
        ans = max(ans, glod);


        for (int i = 0;i < 4;++i){
            int u = x+dirs[i], v = y+dirs[i+1];
            if (u>=0&&u<m && v>=0&&v<n && grid[u][v]!=0){
                int t = grid[u][v];
                grid[u][v] = 0;
                dfs(u, v, glod+t, grid);
                grid[u][v] = t;
            }
        }

    }

    int getMaximumGold(vector<vector<int>>& grid) {
        ans = 0;

        m = grid.size();
        n = grid[0].size();
        for (int i = 0;i < m;++i)
            for (int j = 0;j < n;++j)
                if (grid[i][j] != 0){
                    int glod = grid[i][j];
                    grid[i][j] = 0;
                    dfs(i, j, glod, grid);
                    grid[i][j] = glod;
                }
        return ans;

    }
};
