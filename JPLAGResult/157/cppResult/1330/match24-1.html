class Solution {
public:
    int longestSubsequence(vector<int>& arr, int difference) {

        map<int, int> pos;
        vector<int> dp(arr.size(), 1);
        for(int i = 0; i < arr.size(); i ++){
            int pre = arr[i] - difference;
            if(pos.count(pre))
                dp[i] = max(dp[i], dp[pos[pre]] + 1);
            pos[arr[i]] = i;
        }
        return *max_element(dp.begin(), dp.end());
    }
};
