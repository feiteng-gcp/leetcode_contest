class Solution {
public:
    int longestSubsequence(vector<int>& arr, int difference) {
        unordered_map<int, int> counts;
        int res = 0;
        for (auto x : arr) {
            int prev = x - difference;
            if (counts.count(prev)) {
                counts[x] = counts[prev] + 1;
            }
            else {
                counts[x] = 1;
            }
            res = max(res, counts[x]);
        }
        return res;
    }
};
