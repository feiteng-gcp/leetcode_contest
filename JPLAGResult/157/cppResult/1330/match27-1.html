class Solution {
public:
    int longestSubsequence(vector<int>& arr, int difference) {
        unordered_map<int,int>dp;
        int ans = 0;
        for(int i = 0 ; i < arr.size(); i++){
            int req = arr[i] - difference;
            dp[arr[i]] = max(dp[arr[i]],dp[req]+1);
            ans = max(ans,dp[arr[i]]);
        }
        return ans;
    }
};
