class Solution {
public:
    int longestSubsequence(vector<int>& arr, int difference) {
        unordered_map<int, int> last_maxlen;
        int max_len = 0;
        for (int n : arr) {
            auto iter = last_maxlen.find(n - difference);
            if (iter == last_maxlen.end()) {
                last_maxlen[n] = max(last_maxlen[n], 1);
            } else {
                last_maxlen[n] = max(last_maxlen[n], 1 + iter->second);
            }
            max_len = max(max_len, last_maxlen[n]);
        }
        return max_len;
    }
};
