class Solution {
public:
    int longestSubsequence(vector<int>& arr, int difference) {
        map<int,int> m;
        int ans=0;
        for(auto it:arr){
            m[it]=max(m[it],m[it-difference]+1);
            ans=max(ans,m[it]);
        }
        return ans;
        
    }
};
