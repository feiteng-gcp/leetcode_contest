class Solution {
public:
    map<int,int>dp;
    int longestSubsequence(vector<int>& arr, int difference) {
        int maxx=1;
        for(int i=0;i<arr.size();i++){
            dp[arr[i]]=dp[arr[i]-difference]+1;
            maxx=max(maxx,dp[arr[i]]);
        }
        return maxx;
    }
};
