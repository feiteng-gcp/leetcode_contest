class Solution {
 public:
  int longestSubsequence(vector<int>& arr, int difference) {
    unordered_map<int, int> cur_len;
    int res = 0;
    for (int num : arr) {
      int prev = num - difference;
      if (cur_len.find(prev) != cur_len.end()) {
        cur_len[num] = cur_len[prev] + 1;
      } else {
        cur_len[num] = 1;
      }
      res = max(res, cur_len[num]);
    }
    return res;
  }
};
