class Solution {
public:
    int longestSubsequence(vector<int>& arr, int difference) {
        map<int, int> m_xl;
        int ans = 0;
        for (auto each : arr)
        {

            if (m_xl.find(each - difference) != m_xl.end())
            {
                m_xl[each] = m_xl[each - difference] + 1;
            }
            else
            {
                m_xl[each] = 1;
            }
            ans = max(ans, m_xl[each]);
        }
        return ans;
    }
};
