class Solution {
public:
    int longestSubsequence(vector<int>& arr, int diff) {
        map<int,int> mp;
        int res = 1;
        for(int i=0; i<arr.size(); i++){
            mp[arr[i]] = max(mp[(arr[i] - diff)]+1, mp[arr[i]]);
            res = max(res,mp[arr[i]]);
        }
        return res;
    }
};
