class Solution {
    int max =0;
    public int getMaximumGold(int[][] grid) {
        
int m = grid.length, n = grid[0].length;
        for(int i =0; i< grid.length; i++) {
            for(int j =0; j< grid[0].length; j++) {
                if( (j==0 || grid[i][j-1]==0) && (i==0 || grid[i-1][j]==0) && grid[i][j]!=0){
                    max = Math.max(helper(grid, new int[m][n], i , j), max);
                }
            }
        }
         for(int i =grid.length - 1; i>=0; i--) {
            for(int j =grid[0].length-1; j>=0; j--) {
                if( (j==n-1 || grid[i][j+1]==0) && (i==m-1 || grid[i+1][j]==0) && grid[i][j]!=0){
                    max = Math.max(helper(grid, new int[m][n], i , j), max);
                
}
            }
        }
        return max;
    }
    
    public int helper(int[][] grid, int[][] vis, int i, int j) {
        int max = grid[i][j];
        int[][] dirs = 
{{0,1}, {-1,0}, {1,0}, {0,-1}};
        vis[i][j]=1;
        for(int[] dir : dirs) 
{
            int x = i + dir[0];
            int y = j + dir[1];
            if(x>=0 && y>=0 && x<grid.length && y<grid[0].length && grid[x][y]!=0 && vis[x][y]==0) {
                max = Math.max(max, grid[i][j] + helper(grid,vis,x,y));
            }
        }
        vis[i][j]= 0;
        return max;
    }
}
