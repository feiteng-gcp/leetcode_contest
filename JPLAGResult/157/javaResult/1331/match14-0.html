class Solution {
    int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    
    public int getMaximumGold(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int res = 0;
        
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                boolean[][] v = new boolean[m][n];
                
                res = Math.max(res, dfs(grid, v, i, j));
            }
        }
        
        return res;
        
    }
    
    private int dfs(int[][] grid, boolean[][] v, int i, int j) {
        if(i < 0 || i >= grid.length || j < 0 || j >= grid[0].length || v[i][j] || grid[i][j] == 0) {
            return 0;
        }

        v[i][j] = true;
        int collect = grid[i][j];
        int next = 0;
        for(int[] dir : dirs) {
            next = Math.max(next, dfs(grid, v, i + dir[0], j + dir[1]));
        }
        
        v[i][j] = false;
        return collect + next;
    }
}
