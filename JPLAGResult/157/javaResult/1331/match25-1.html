class Solution {
    public int getMaximumGold(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        boolean[][] visited = new boolean[m][n];
        int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] > 0) {
                    res = Math.max(res, dfs(grid, visited, m, n, i, j));
                }
            }
        }
        return res;
    }

    
int[][] dirs = new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    private int dfs(int[][] grid, boolean[][] visited, int m, int n, int x, int y) {
        visited[x][y] = true;
        int res = 0;
        for (int[] dir : dirs) 
{
            int newX = dir[0] + x;
            int newY = dir[1] + y;
            if (newX < m && newX >= 0 && newY < n && newY >= 0 && grid[newX][newY] != 0 && !visited[newX][newY]) {
                res = Math.max(res, dfs(grid, visited, m, n, newX, newY));
            }
        }
        res += grid[x][y];
        visited[x][y] = false;
        return res;
    }
}
