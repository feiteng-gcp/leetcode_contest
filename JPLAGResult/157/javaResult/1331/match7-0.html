class Solution {
    private int[][] DIR = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int getMaximumGold(int[][] grid) {
        int res = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0) {
                    continue;
                }
                res = Math.max(res, maxSum(grid, i, j, new boolean[grid.length][grid[0].length]));
            }
        }
        return res;
    }
    
    private int maxSum(int[][] grid, int i, int j, boolean[][] visited) {
        visited[i][j] = true;
        int cur = grid[i][j];
        int max = 0;
        for (int[] dir : DIR) {
            int nextI = i + dir[0];
            int nextJ = j + dir[1];
            if (nextI < 0 || nextI >= grid.length || nextJ < 0 || nextJ >= grid[0].length 
                ||grid[nextI][nextJ] == 0 || visited[nextI][nextJ]) {
                continue;
            }
            visited[nextI][nextJ] = true;
            max = Math.max(max, maxSum(grid, nextI, nextJ, visited));
            visited[nextI][nextJ] = false;
        }
        return cur + max;
    }
}
