class Solution {
    
int[][] dirs = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
    public int getMaximumGold(int[][] grid) {
        int res = 0;
        for(int i=0;i<grid.length;i++)
            for(int j=0;j<grid[0].length;j++)
                if(grid[i][j] !=0)
                    res = Math.max(res, dfs(i,j,grid))
;
        return res;
    }
    public int dfs(int x,int y,int[][] grid){
        if(x<0||y<0||x>=grid.length||y>=grid[0].length) return 0;
        
if(grid[x][y] == 0) return 0;
        int val = grid[x][y];
        grid[x][y] = 0;
        int best = 0;
        for(int[] dir : dirs){
            int nextX = dir[0]+x;
            int nextY = dir[1]+y;
            best = Math.max(best, dfs(nextX,nextY,grid));
        }
        grid[x][y] = val;
        return best+val;
    }
}
