class Solution {
    int[][] directions = new int[][]{{1,0},{-1,0},{0,1},{0,-1}};
    public int getMaximumGold(int[][] grid) {
        boolean[][] visited = new boolean[grid.length][grid[0].length];
        int ans = 0;
        for(int i=0;i<grid.length;i++){
            for(int j=0;j<grid[0].length;j++){
                if(grid[i][j] != 0){
                    ans = Math.max(ans, dfs(i,j, grid, visited));
                }
            }
        }
        return ans;
    }
    private int dfs(int i, int j, int[][] grid, boolean[][] visited){
        visited[i][j] = true;
        int temp = 0;
        for(int[] direction : directions){
            int x = i+direction[0];
            int y = j+direction[1];
            if(x>=0 && x < grid.length && y >=0 && y<grid[0].length && grid[x][y] != 0 && !visited[x][y]){
                temp= Math.max(temp,dfs(x, y, grid, visited));
            }
        }
        visited[i][j] = false;
        return temp+grid[i][j];
    }
}
