class Solution {
    int[][] dir = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    int result = 0;
    public int getMaximumGold(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] != 0) {
                    boolean[][] visited = new boolean[m][n];
                    dfs(grid, visited, i, j, 0);
                }
            }
        }
        return result;
    }
    
    public void dfs(int[][] grid, boolean[][] visited, int x, int y, int val) {
        int m = grid.length;
        int n = grid[0].length;
        val += grid[x][y];
        visited[x][y] = true;
        result = Math.max(result, val);
        
        for (int[] d : dir) {
            int X = x + d[0];
            int Y = y + d[1];
            if (X < 0 || X >= m || Y < 0 || Y >= n || visited[X][Y] || grid[X][Y] == 0) continue;
            dfs(grid, visited, X, Y, val);
        }
        val -= grid[x][y];
        visited[x][y] = false;
    }
}
