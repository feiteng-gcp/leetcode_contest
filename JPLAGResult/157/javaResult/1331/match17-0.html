class Solution {
    static int[][] ops = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
    public int getMaximumGold(int[][] grid) {
        int m = grid.length, n = grid[0].length, max = 0;
        boolean[][] used = new boolean[m][n];
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (0 == grid[i][j]) continue;
                max = Math.max(max, dfs(grid, i, j, used));
            }
        }
        return max;
    }
    private int dfs(int[][] grid, int x, int y, boolean[][] used) {
        int m = grid.length, n = grid[0].length;
        used[x][y] = true;
        int sum = grid[x][y], max = 0;
        for (int[] op : ops) {
            int i = x + op[0], j = y + op[1];
            if (0 <= i && i < m && 0 <= j && j < n && 0 != grid[i][j] && !used[i][j]) {
                max = Math.max(max, dfs(grid, i, j, used));
            }
        }
        sum += max;
        used[x][y] = false;
        return sum;
    }
}
