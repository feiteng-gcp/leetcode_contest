class Solution {
    
    int m;
    int n;
    int[][] nexts = new int[][] {{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
    
    public int getMaximumGold(int[][] grid) {
        int res = 0;
        this.m = grid.length;
        this.n = grid[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                res = Math.max(res, dfs(grid, i, j, new boolean[m][n]));
            }
        }
        return res;
    }
    
    private int dfs(int[][] grid, int i, int j, boolean[][] visited) {
        if (grid[i][j] == 0)
            return 0;
        
        visited[i][j] = true;
        int max = 0;
        
        for (int[] next : nexts) {
            int x = i + next[0];
            int y = j + next[1];
            if (x >= 0 && x < m && y >= 0 && y < n && !visited[x][y]) {
                max = Math.max(max, dfs(grid, x, y, visited));
            }
        }
        
        visited[i][j] = false;
        return max + grid[i][j];
    }
}
