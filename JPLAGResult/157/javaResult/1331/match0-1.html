class Solution {
    
    int dx[] = {0,0,1,-1};
    int dy[] = {1,-1,0,0};
    int res =0;
    
    void dfs(int i,int j,int[][] g,boolean vis[][],int s){
        int l = g.length;
        int ll = g[0].length;
        s += g[i][j];
        res = Math.max(res,s);
        
        for(int y=0;y<4;++y){
            int nx = i+dx[y];
            int ny  =j +dy[y];
            if(nx>=0&&ny>=0&&nx<l&&ny<ll&&g[nx][ny]>0&&!vis[nx][ny]){
                vis[nx][ny] =true;
                dfs(nx,ny,g,vis,s);
                vis[nx][ny] = false;
            }
        }
        
    }
    
    public int getMaximumGold(int[][] g) {
        int l = g.length;
        int ll = g[0].length;
        
        
        for(int i=0;i<l;++i){
            for(int j=0;j<ll;++j){
                if(g[i][j]>0){
                    
                    boolean vis[][] = new boolean [l][ll];
                    vis[i][j] =true;
                    dfs(i,j,g,vis,0);
                    vis[i][j] =false;
                    
                    
                    
                    
                    
                    
                    
                }
            }
        }
        return res;
        
    }
}
