class Solution {
    int[] dx = new int[]{0, 0, 1, -1};
    int[] dy = new int[]{1, -1, 0, 0};
    public boolean isValid(int x, int y, int[][] rooms) {
        if (x >= 0 && x < rooms.length && y >= 0 && y < rooms[0].length) return true;
        return false;
    }
    int max_res = 0;
    public int getMaximumGold(int[][] grid) {
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] != 0) {
                    dfs(new boolean[grid.length][grid[0].length], grid, i, j, 0);
                }
            }
        }
        return max_res;
    }

    public boolean dfs(boolean[][] vis, int[][] bd, int i, int j, int cur) {
        vis[i][j] = true;
        cur += bd[i][j];
        max_res = Math.max(cur, max_res);
        for (int k = 0; k < 4; k++) {
            int x = i + dx[k];
            int y = j + dy[k];
            if (isValid(x, y, bd) && !vis[x][y] && bd[x][y] != 0) {
                if (dfs(vis, bd, x, y, cur)) return true;
            }
        }
        cur -= bd[i][j];
        vis[i][j] = false;
        return false;
    }
}
