class Solution {
    public int getMaximumGold(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        int max = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] != 0) {
                    Integer[][] memo = new Integer[n][m];
                    max = Math.max(max, helper(i, j, grid, memo, new boolean[n][m]));
                }
            }
        }
        return max;
    }
    int helper(int x, int y, int[][] grid, Integer[][] memo, boolean[][] visited) {
        if (x < 0 || y < 0 || x >= grid.length || y >= grid[0].length || grid[x][y] == 0 || visited[x][y]) {
            return 0;
        }
        // if (memo[x][y] != null) {
        //     return memo[x][y];
        // }
        int gold = grid[x][y];
        visited[x][y] = true;
        int max = 0;
        max = Math.max(max, helper(x, y - 1, grid, memo, visited));
        max = Math.max(max, helper(x - 1, y, grid, memo, visited));
        max = Math.max(max, helper(x + 1, y, grid, memo, visited));
        max = Math.max(max, helper(x, y + 1, grid, memo, visited));
        visited[x][y] = false;
        // memo[x][y] = gold + max;
        return gold + max;
        // return memo[x][y];
    }
}
