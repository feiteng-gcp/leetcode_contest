import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    int ans = 0;
    int[][] grid;
    
boolean[][] v1;
    boolean[][] v2;

    int r, c;

    int[] dx = {0, 0, 1, -1};
    int[] dy = {1, -1, 0, 0};

    public int getMaximumGold(int[][] grid) {
        r = grid.length;
        c = grid[0].length;

        v1 = new boolean[r][c];
        v2 = new boolean[r][c];

        ans = 0;
        this.grid = grid;

        
for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                if (grid[i][j] > 0) {
                    dfs(i, j, grid[i][j]);
                }
            }
        }

        return ans;
    }

    void dfs(int i, int j, int current) {
        ans = Math.max(ans, current);

        v2[i][j] = true;
        
for (int k = 0; k < 4; k++) {
            int ii = i + dx[k];
            int jj = j + dy[k];
            if (ii >= 0 && ii < r && jj >= 0 && jj < c && grid[ii][jj] > 0 && !v2[ii][jj]) {
                dfs(ii, jj, current + grid[ii][jj]);
            }
        }
        v2[i][j] = false;
    }


    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(sol.getMaximumGold(new int[][]{
                {0, 6, 0},
                {5, 8, 7},
                {0, 9, 0},
        }));
    }
}
