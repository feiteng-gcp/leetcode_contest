class Solution {
    int res = 0;
    
int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
    
    
public int getMaximumGold(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        for(int i=0; i<m; i++){
            for(int j=0; j<n; j++){
                if(grid[i][j] != 0){
                    
dfs(grid, i, j, 0);
                }
            }
        }
        return res;
    }
    
    void dfs(int[][] grid, int x, int y, int total){
        int m = grid.length, n = grid[0].length;
        int coin = grid[x][y];
        res = Math.max(res, total + coin);
        grid[x][y] = 0;
        
for(int[] d: dirs){
            int nx = x+d[0], ny = y+d[1];
            if(nx<0 || nx>=m || ny<0 || ny>=n || grid[nx][ny] == 0) continue;
            dfs(grid, nx, ny, total + coin);
        }
        grid[x][y] = coin;
    }
}
