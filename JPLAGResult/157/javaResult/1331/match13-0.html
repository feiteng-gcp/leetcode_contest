class Solution {
    public int getMaximumGold(int[][] grid) {
        int max = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] != 0) {
                    max = Math.max(max, dfs(grid, new boolean[grid.length][grid[0].length], i, j));
                }
            }
        }
        return max;
    }

       private int dfs(int[][] grid, boolean[][] see, int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length || see[i][j] || grid[i][j] == 0) return 0;
        see[i][j] = true;
        int r1 = dfs(grid, see, i + 1, j);
        int r2 = dfs(grid, see, i - 1, j);
        int r3 = dfs(grid, see, i, j - 1);
        int r4 = dfs(grid, see, i, j + 1);
        see[i][j] = false;
        return grid[i][j] + Math.max(Math.max(r1, r2), Math.max(r3, r4));
    }
}
