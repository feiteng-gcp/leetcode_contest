class Solution {
    int m = 0;
    int n = 0;
    int max = 0; 
    boolean[][] vis; 
    int[] dx = {1, -1, 0, 0};
    int[] dy = {0, 0, 1, -1};
    public int getMaximumGold(int[][] grid) {
        
        m = grid.length;
        n = grid[0].length;
        vis = new boolean[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] > 0) {
                    dfs(grid, i, j, vis, 0);
                }
            }
        }
        return max;
    }
    private void dfs(int[][] grid, int x, int y, boolean[][] vis, int count) {
        
        vis[x][y] = true;
        count += grid[x][y];
        max = Math.max(count, max);
        for (int k = 0; k < 4; k++) {
            int xx = x + dx[k];
            int yy = y + dy[k];
            if (xx >= 0 && xx < m && yy >= 0 && yy < n && grid[xx][yy] > 0 && !vis[xx][yy]) {
                dfs(grid, xx, yy, vis, count);
            }
        }
        count -= grid[x][y];
        vis[x][y] = false;
    }
}
