class Solution {
    public int getMaximumGold(int[][] grid) {
        //sanity
        int max = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                max = Math.max(max, dfs(i, j, grid, new boolean[grid.length][grid[0].length]));
            
}
        }
        return max;
    }
    private int dfs(int i, int j, int[][] grid, boolean[][] visited) {
        int r = grid.length;
        int c = grid[0].length;
        if (i < 0 || i >= r || j < 0 || j >= c || grid[i][j] == 0 || visited[i][j]) {
            return 0;
        }
        
int total = 0;
        visited[i][j] = true;
        total = Math.max(total, dfs(i + 1, j, grid, visited));
        total = Math.max(total, dfs(i, j + 1, grid, visited));
        total = Math.max(total, dfs(i - 1, j, grid, visited));
        total = Math.max(total, dfs(i, j - 1, grid, visited));
        visited[i][j] = false;
        return total + grid[i][j];
    }
}
