class Solution {
    public int getMaximumGold(int[][] grid) {
		int maxGold = 0;
		boolean[][] visited = new boolean[grid.length][grid[0].length];
		for (int i = 0; i < grid.length; i++) {
			for (int j = 0; j < grid[0].length; j++) {
				maxGold = Math.max(maxGold, getMaximumGold(grid, visited, i, j));
			}
		}
		return maxGold;
    }
	
	private int getMaximumGold(int[][] grid, boolean[][] visited, int i, int j) {
		if (i < 0 || j < 0 || i >= grid.length || j >= grid[0].length || grid[i][j] == 0 || visited[i][j]) {
			return 0;
		}
		int gold = grid[i][j];
		visited[i][j] = true;
		int maxGold = 0;
		maxGold = Math.max(maxGold, getMaximumGold(grid, visited, i + 1, j));
		maxGold = Math.max(maxGold, getMaximumGold(grid, visited, i - 1, j));
		maxGold = Math.max(maxGold, getMaximumGold(grid, visited, i, j + 1));
		maxGold = Math.max(maxGold, getMaximumGold(grid, visited, i, j - 1));
		visited[i][j] = false;
		return gold + maxGold;
	}
}
