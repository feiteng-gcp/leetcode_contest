class Solution {
    private static int[][] neighors = new int[][] {{0, +1}, {-1, 0}, {0, -1}, {+1, 0}};

    public int getMaximumGold(int[][] grid) {
        int R = grid.length;
        int C = grid[0].length;
        int max = 0;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                max = Math.max(max, dfs(grid, i, j, new boolean[R][C]));
            
}
        }
        return max;
    }

    private int dfs(int[][] grid, int r, int c, boolean[][] visited) {
        if (r < 0 || r >= grid.length || c < 0 || c >= grid[0].length || visited[r][c] || grid[r][c] == 0) {
            return 0;
        }
        visited[r][c] = true;
        int output = 0;
        for (int[] neighbor : neighors) {
            int nextR = r + neighbor[0];
            int nextC = c + neighbor[1];
            
output = Math.max(output, grid[r][c] + dfs(grid, nextR, nextC, visited));
        }
        visited[r][c] = false;
        return output;
    }
}
