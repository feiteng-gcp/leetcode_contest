class Solution {
    public int getMaximumGold(int[][] grid) {
        
int max = 0;
        for (int r = 0; r < grid.length; r++) {
            for (int c = 0; c < grid[r].length; c++) {
                if (grid[r][c] != 0) {
                    max = Math.max(max, getMaxGold(grid, r, c));
                
}
            }
        }
        return max;
    }
    
    public int getMaxGold(
int[][] grid, int r, int c) {
        if (r < 0 || r >= grid.length || c < 0 || c >= grid[r].length || grid[r][c] == 0) {
            return 0;
        }
        
        int currGold = grid[r][c];
        grid[r][c] = 0;
        int maxNextGold = 0;
        maxNextGold = Math.max(maxNextGold, getMaxGold(grid, r - 1, c));
        maxNextGold = Math.max(maxNextGold, getMaxGold(grid, r + 1, c));
        maxNextGold = Math.max(maxNextGold, getMaxGold(grid, r, c - 1));
        maxNextGold = Math.max(maxNextGold, getMaxGold(grid, r, c + 1));
        grid[r][c] = currGold;
        return currGold + maxNextGold;
    }
}
