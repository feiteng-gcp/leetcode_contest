class Solution {
    public int getMaximumGold(int[][] grid) {
        int R = grid.length, C = grid[0].length;
        boolean[][] visited = new boolean[R][C];
        
        int ans = 0;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (grid[i][j] != 0) {
                    ans = Math.max(ans, helper(grid, i, j, visited));
                }
            }
        }
        return ans;
    }
    
    private int helper(int[][] grid, int x, int y, boolean[][] visited) {
        int R = grid.length, C = grid[0].length;
        visited[x][y] = true;
        
        int ans = grid[x][y];
        int max = 0;
        for (int[] offset : new int[][] {{-1, 0}, {1, 0}, {0, 1}, {0, -1}}) {
            int nx = x + offset[0], ny = y + offset[1];
            if (nx >= 0 && nx < R && ny >= 0 && ny < C && !visited[nx][ny] && grid[nx][ny] > 0) {
                max = Math.max(max, helper(grid, nx, ny, visited));
            }
        }
        visited[x][y] = false;
        return ans + max;
    }
}
