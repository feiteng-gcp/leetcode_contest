class Solution {
    public int getMaximumGold(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        
int max = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] != 0) {
                    max = Math.max(dfs(grid, i, j), max);
                }
            }
        }
        return max;
    }
    
    private int dfs(int[][] grid, int i, int j) {
        int m = grid.length, n = grid[0].length;
        if (i >= m || j >= n || j < 0 || i < 0 || grid[i][j] == 0) return 0;
        
        int max = 0;
        
        int tmp = grid[i][j];
        grid[i][j] = 0;
        
max = Math.max(dfs(grid, i + 1, j), max);
        max = Math.max(dfs(grid, i - 1, j), max);
        max = Math.max(dfs(grid, i, j - 1), max);
        max = Math.max(dfs(grid, i, j + 1), max);
        grid[i][j] = tmp;
        
        return grid[i][j] + max;
    }
}
