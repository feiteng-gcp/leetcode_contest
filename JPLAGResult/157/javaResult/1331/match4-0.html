class Solution {
    public int getMaximumGold(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int res = 0;
        final int[][] dirs = new int[][]{{0,1},{1,0},{-1,0},{0,-1}};
        for(int i=0; i<m; i++){
            for(int j=0; j<n; j++){
                res = Math.max(res, dfs(grid, i, j, new HashSet<>(), n, dirs));
            }
        }
        return res;
    }
    
    private int dfs(int[][] grid, int x, int y, Set<Integer> visited, int n, int[][] dirs){
        int symbol = x*n+y;
        if(x>=grid.length || y>=n || x<0 || y<0 || grid[x][y]==0 || !visited.add(symbol)) return 0; 
        int res = grid[x][y];
        for(int[] dir: dirs){
            int nx = x+dir[0], ny = y+dir[1];
            res = Math.max(res, grid[x][y]+dfs(grid, nx, ny, visited, n, dirs));
        }
        visited.remove(symbol);
        return res;
    }
}
