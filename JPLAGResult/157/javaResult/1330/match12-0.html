class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0;
        for(int a: arr){
            int cur = map.getOrDefault(a-difference, 0)+1;
            map.put(a, cur);
            res = Math.max(res, cur);  
        }
        return res;
    }
}
