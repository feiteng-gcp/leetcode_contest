class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> map = new HashMap<>();
        int ans = 0;
        for (int i = 0; i < arr.length; ++i) {
            int v = arr[i];
            int old = map.getOrDefault(arr[i], 0);
            int c = 0;
            if (map.containsKey(v - difference)) {
                c = Math.max(c, map.get(v - difference));                
            }
            map.put(v, Math.max(old, c + 1));
            ans = Math.max(ans, map.get(v));
        }
        return ans;
        
    }
}
