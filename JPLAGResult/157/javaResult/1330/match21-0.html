class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0;
        
        for(int i : arr) {
            int last = i - difference;
            if(map.containsKey(last)) {
                map.put(i, map.get(last) + 1);
            } else {
                map.put(i, 1);
            }
            
            res = Math.max(res, map.get(i));
        }
        
        return res;        
    }
}
