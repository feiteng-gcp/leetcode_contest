class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> mem = new HashMap<>();
        int max = 0;
        for (int i = arr.length - 1; i >= 0; --i){
            int tmp = mem.getOrDefault(arr[i] + difference, 0) + 1;
            mem.put(arr[i], tmp);
            max = Math.max(max, tmp);
        }
        return max;
    }
}
