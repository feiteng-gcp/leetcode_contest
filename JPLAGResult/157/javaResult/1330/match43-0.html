class Solution {
      public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0;
        for (int i = 0; i < arr.length; i++) {
            int pre = arr[i] - difference;
            int c = map.getOrDefault(pre, 0);
            map.put(arr[i], c + 1);
            res = Math.max(res, c + 1);
        }
        return res;
    }
}
