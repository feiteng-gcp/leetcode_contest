class Solution {
    public int longestSubsequence(int[] arr, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        
        int n = arr.length;
        int[] f = new int[n];
        int best = 0;
        for (int i = 0; i < n; i++) {
            int prev = arr[i] - k;
            if (map.containsKey(prev)) {
                int j = map.get(prev);
                f[i] = f[j] + 1;
            } else {
                f[i] = 1;
            }
            best = Math.max(best, f[i]);
            map.put(arr[i], i);
        }
        return best;
    }
}
