class Solution {
    public int longestSubsequence(int[] arr, int d) {
        Map<Integer,Integer> mp = new HashMap<>();
        mp.put(arr[0],1);
        int r =1;
        for(int i=1;i<arr.length;++i){
            int ck = arr[i]-d;
            int v = 0;
            if(mp.containsKey(ck)){
                v = mp.get(ck)+1;
            }else{
                v = 1;
            }
            r= Math.max(r,v);
                if(mp.containsKey(arr[i])){
                    mp.put(arr[i],Math.max(mp.get(arr[i]),v));
                }else{
                    mp.put(arr[i],v);
                }
            
        }
        return r;
    }
}
