class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> map = new HashMap<>();
        int ans = 0;
        for(int num : arr) {
            int val = num-difference;
            if(map.containsKey(val)) {
                map.put(num, Math.max((map.get(val)+1), map.getOrDefault(num, 0)));
            
} else {
                map.put(num, 1);
            }
            ans = Math.max(ans, map.get(num));
        }
        
        return ans;
    }
}
