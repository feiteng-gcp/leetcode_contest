class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        int res = 1;
        Map<Integer, Integer> map = new HashMap<>();
        for (int num : arr) {
            if (map.containsKey(num - difference)) {
                map.put(num, map.get(num - difference) + 1);
                res = Math.max(res, map.get(num));
            } else {
                map.put(num, 1);
            }
        }
        return res;
    }
}
