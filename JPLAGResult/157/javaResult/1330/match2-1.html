class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> dp = new HashMap();
        int ans = 0;
                
        for (int j = arr.length - 1; j >= 0; j--) {
            dp.put(arr[j], 1 + dp.getOrDefault(arr[j] + difference, 0));
            ans = Math.max(ans, dp.get(arr[j]));
        }
        
        return ans;
    }
}
