class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        Map<Integer, Integer> map = new HashMap<>();
        int maxLen = 0;
        for (int n : arr) {
            int currLen = map.getOrDefault(n - difference, 0) + 1;
            map.put(n, currLen);
            maxLen = Math.max(maxLen, currLen);
        }
        return maxLen;
    }
}
