class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        HashMap<Integer, Integer> best = new HashMap<>();
        int ans = 0;
        for (int v : arr) {
            int want = v - difference;
            int len = 1 + best.getOrDefault(want, 0);
            best.put(v, len);
            ans = Math.max(ans, len);
        }
        return ans;
    }
}
