class Solution {
    public int countVowelPermutation(int n) {
        if(n==0) return 0;
        long[][] dp = new long[n][5];
        for(int i=0; i<5; i++){
            dp[0][i] = 1;
        }
        int mod = (int)1e9+7;
        for(int i=1; i<n; i++){
            dp[i][0] = (dp[i-1][1]+dp[i-1][4]+dp[i-1][2])%mod;
            dp[i][1] = (dp[i-1][0]+dp[i-1][2])%mod;
            dp[i][2] = (dp[i-1][1]+dp[i-1][3])%mod;
            dp[i][3] = (dp[i-1][2])%mod;
            dp[i][4] = (dp[i-1][2]+dp[i-1][3])%mod;
        }
        long res = 0;
        for(int i=0; i<5; i++){
            res += dp[n-1][i];
            res = res%mod;
        }
        return (int)res;
    }
}
