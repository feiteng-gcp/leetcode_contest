class Solution {
    public int countVowelPermutation(int n) {
		if (n == 1) {
			return 5;
		}
		long[] result = new long[5];
		Arrays.fill(result, 1L);
		for (int i = 2; i <=n; i++) {
			long[] result2 = new long[5];
			// Each vowel 'a' may only be followed by an 'e'.
			result2[1] += result[0];
			// Each vowel 'e' may only be followed by an 'a' or an 'i'.
			result2[0] += result[1];
			result2[2] += result[1];
			// Each vowel 'i' may not be followed by another 'i'.
			result2[0] += result[2];
			result2[1] += result[2];
			result2[3] += result[2];
			result2[4] += result[2];
			// Each vowel 'o' may only be followed by an 'i' or a 'u'.
			result2[2] += result[3];
			result2[4] += result[3];
			// Each vowel 'u' may only be followed by an 'a'.
			result2[0] += result[4];
			for (int j = 0; j <= 4; j++) {
				result[j] = result2[j] % 1_000_000_007;
			}
		}
		long sum = 0;
		for (long r : result) {
			sum = (sum + r) % 1_000_000_007;
		}
		return (int) sum;
    }
}
