class Solution {
    public int countVowelPermutation(int n) {
        int mod = 1_000_000_007;
        long[] a = new long[n], e = new long[n], i = new long[n], o = new long[n], u = new long[n];
        a[0] = 1;
        e[0] = 1;
        i[0] = 1;
        o[0] = 1;
        u[0] = 1;
        for(int k = 1; k < n; k++){
            a[k] = e[k-1] % mod;
            e[k] = (a[k-1] + i[k-1]) % mod;
            i[k] = (a[k-1] + e[k-1] + o[k-1] + u[k-1]) % mod;
            o[k] = (i[k-1] + u[k-1]) % mod;
            u[k] = a[k-1] % mod;
        }
        return (int)((a[n-1] + e[n-1] + i[n-1] + o[n-1] + u[n-1]) % mod);
    }
}
