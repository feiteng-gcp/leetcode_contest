class Solution {
    static int mod = (int) 1e9 + 7;
    public int countVowelPermutation(int n) {
        long[][] dp = new long[5][n];
        for (int i = 0; i < 5; ++i) dp[i][n - 1] = 1;
        for (int j = n - 2; 0 <= j; --j) {
            dp[0][j] = dp[1][j + 1];
            dp[1][j] = (dp[0][j + 1] + dp[2][j + 1]) % mod;
            dp[2][j] = (dp[0][j + 1] + dp[1][j + 1] + dp[3][j + 1] + dp[4][j + 1]) % mod;
            dp[3][j] = (dp[2][j + 1] + dp[4][j + 1]) % mod;
            dp[4][j] = dp[0][j + 1];
        }
        long sum = 0;
        for (int i = 0; i < 5; ++i) sum = (sum + dp[i][0]) % mod;
        return (int) sum;
    }
}
