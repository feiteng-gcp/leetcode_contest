public class Solution {
    private static final int MOD = 1_000_000_007;
    public int countVowelPermutation(int n) {
        long[][] dp = new long[n][5];
        Arrays.fill(dp[0], 1);
        for (int i = 0; i < n - 1; ++i) {
            dp[i + 1][1] += dp[i][0];
            dp[i + 1][0] += dp[i][1];
            dp[i + 1][2] += dp[i][1];
            dp[i + 1][0] += dp[i][2];
            dp[i + 1][1] += dp[i][2];
            dp[i + 1][3] += dp[i][2];
            dp[i + 1][4] += dp[i][2];

            dp[i + 1][2] += dp[i][3];
            dp[i + 1][4] += dp[i][3];
            dp[i + 1][0] += dp[i][4];
            for (int k = 0; k < 5; ++k)
                dp[i + 1][k] %= MOD;
        }
        long ans = 0;
        for (int k = 0; k < 5; ++k) {
            ans += dp[n - 1][k];
        }
        return (int)(ans % MOD);
    }
}
