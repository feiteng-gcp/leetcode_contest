class Solution {
    public int countVowelPermutation(int n) {
        long dp[]=  new long[5];
        Arrays.fill(dp,1);
        
        
        for(int i=2;i<=n;++i){
            long ndp[]=  new long[5];
            for(int j=0;j<=4;++j){
                if(j==0){
                ndp[1] += dp[0];
                }else if(j==1){
                ndp[0] += dp[1];
                ndp[2] += dp[1];
                }else if(j==2){
                
                    ndp[0] += dp[2];
                    ndp[1] += dp[2];
                    ndp[4] += dp[2];
                    ndp[3] += dp[2];
                    
                }else if(j==3){
                
                ndp[2] += dp[3];
                ndp[4] += dp[3];
                }else if(j==4){
                
                ndp[0] += dp[4];
                }
            }
            for(int j=0;j<=4;++j){
                ndp[j] %= 1000000007; 
            }
            dp = ndp;
        }
       
        long s = 0;
        for(int j=0;j<=4;++j){
            System.out.println(dp[j]);
                s += dp[j]; s%= 1000000007; 
            }
        return (int) s;
    }
}
