class Solution {
    public int countVowelPermutation(int n) {
        int a=1;
        int e=1;
        int i=1;
        int o=1;
        int u=1;
        int mod = (int)1e9+7;
        for(int j=1; j<n; j++){
            int aa=a;
            int ee=e;
            int ii = i;
            int oo = o;
            int uu = u;
            a = ee;
            e = (aa+ii)%mod;
            i = ((aa+ee)%mod + (oo+uu)%mod)%mod;
            o = (ii+uu)%mod;
            u = aa;
            
        }
        return (((a+e)%mod + (o+u) %mod)%mod + i)%mod;
    }
}
