class Solution {
    public int countVowelPermutation(int n) {
        long[] A = new long[5];
        Arrays.fill(A, 1);
        for(int i=2; i<=n; i++){
            long[] B = new long[5];
            B[0] = (A[1] + A[2] + A[4]) % 1000000007;
            B[1] = (A[0] + A[2]) % 1000000007;
            B[2] = (A[1] + A[3]) % 1000000007;
            B[3] = A[2] %1000000007;
            B[4] = (A[2] + A[3]) %1000000007;
            A = B;
        }
        int sum = 0;
        for(long i: A){
            sum = (sum + (int)i) %1000000007;
        }
        return sum;
    }
}
