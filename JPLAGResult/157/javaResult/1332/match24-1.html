class Solution {
    public int countVowelPermutation(int n) {
        int mod = (int)1e9 + 7;
        long a = 1, e = 1, i = 1, o = 1, u = 1;
        for (int k = 2; k <= n; k++) {
            long ca = (e + i + u) % mod;
            long ce = (a + i) % mod;
            long ci = (e + o) % mod;
            long co = i % mod;
            long cu = (i + o) % mod;
            a = ca; e = ce; i = ci; o = co; u = cu;
        }
        long res = (a + e + i + o + u) % mod;
        return (int)res;
    }
}
