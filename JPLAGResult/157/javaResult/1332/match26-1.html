/*
0 ae
1 ea  ei
2 ia, ie, io, iu
3 oi, ou
4 ua

*/
class Solution {
    int M = 1000000007;
    public int countVowelPermutation(int n) {
        
long[][] dp = new long[n][5];
        for (int j = 0; j < 5; j++) {
            dp[0][j] = 1;
        }        
        for (int i = 1; i < n; i++) {
            dp[i][0] = (dp[i][0] + dp[i - 1][1]) % M;
            dp[i][0] = (dp[i][0] + dp[i - 1][2]) % M;
            dp[i][0] = (dp[i][0] + dp[i - 1][4]) % M;
            dp[i][1] = (dp[i][1] + dp[i - 1][0]) % M;
            dp[i][1] = (dp[i][1] + dp[i - 1][2]) % M;
            
dp[i][2] = (dp[i][2] + dp[i - 1][1]) % M;
            dp[i][2] = (dp[i][2] + dp[i - 1][3]) % M;
            dp[i][3] = (dp[i][3] + dp[i - 1][2]) % M;
            dp[i][4] = (dp[i][4] + dp[i - 1][2]) % M;
            dp[i][4] = (dp[i][4] + dp[i - 1][3]) % M;
        }
        long ans = 0; 
        for (int j = 0; j < 5; j++) {
            ans = (ans + dp[n - 1][j]) % M;
        }
        return (int)ans;        
    }
}
