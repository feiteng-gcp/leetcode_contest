class Solution {
    public int countVowelPermutation(int n) {
        int mod = 1000000007;
        int[][] dp = new int[5][n+1];
        // a0 e1 i2 o3 u4
        for (int i = 0; i < 5; ++i) {
            dp[i][1] = 1;
        }
        for (int j = 2; j <= n; ++j) {
            dp[0][j] = dp[1][j-1];
            dp[1][j] = (dp[0][j-1] + dp[2][j-1]) % mod;
            dp[2][j] = ((dp[0][j-1] + dp[1][j-1]) % mod + (dp[3][j-1] + dp[4][j-1]) % mod) % mod;
            dp[3][j] = (dp[2][j-1] + dp[4][j-1]) % mod;
            dp[4][j] = dp[0][j-1];
        }
        int ret = 0;
        for (int i = 0; i < 5; ++i) {
            ret = (ret + dp[i][n]) % mod;
        }
        return ret;
    }
}
