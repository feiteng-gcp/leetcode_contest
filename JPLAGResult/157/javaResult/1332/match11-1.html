class Solution {
    public int countVowelPermutation(int n) {
        int[][] dp = new int[n + 1][5]; 
        dp[1][0] = 1;
        dp[1][1] = 1;
        dp[1][2] = 1;
        dp[1][3] = 1;
        dp[1][4] = 1; 
        int mod = (int)1e9 + 7;
        for (int i = 2; i <= n; i++) {
            dp[i][0] = dp[i - 1][1];
            dp[i][4] = dp[i - 1][0]; 
            dp[i][1] = (dp[i - 1][0] + dp[i - 1][2]) % mod;
             dp[i][2] = (((dp[i - 1][0] + dp[i - 1][1]) % mod + dp[i - 1][3]) % mod + dp[i - 1][4]) % mod;
            // System.out.println(dp[i - 1][0] + " " + dp[i - 1][1] + " " + dp[i - 1][3] + " " + dp[i - 1][4]);
            // dp[i][2] = dp[i - 1][0] + dp[i - 1][1] + dp[i - 1][3] + dp[i - 1][4];
            dp[i][3] = (dp[i - 1][2] + dp[i - 1][4]) % mod;
            
        }
        // for (int i = 0; i <= n; i++) {
        //     System.out.println(Arrays.toString(dp[i]));
        // }
        return ((((dp[n][0] + dp[n][1]) % mod + dp[n][2]) % mod + dp[n][3]) % mod + dp[n][4]) % mod; 
    }
}
