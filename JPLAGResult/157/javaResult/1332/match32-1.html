class Solution {
    public int countVowelPermutation(int n) {
        
long a=1,e=1,i=1,o=1,u=1,tempA,tempE,tempI,tempO,tempU;
        int num=1000000007;
        for(int c=1;c<n;c++)
{
            tempA=e;
            tempE=i+a;
            tempI=a+e+o+u;
            tempO=i+u;
            tempU=a;
            a=tempA%num;
            e=tempE%num;
            i=tempI%num;
            o=tempO%num;
            u=tempU%num;
        }
        return (int)((a+e+i+o+u)%num);
    }
}
