class Solution {
    public static final int MOD = 1000000007;
    
    public int countVowelPermutation(int n) {
        if (n == 0) {
            return 0;
        }
        int ap = 1, ep = 1, ip = 1, op = 1, up = 1, a = 0, e = 0, i = 0, o = 0, u = 0, it = 0;
        for (it=2; it<=n; ++it) {
            a = ((ep + ip) % MOD + up) % MOD;
            e = (ap + ip) % MOD;
            i = (ep + op) % MOD;
            o = ip;
            u = (ip + op) % MOD;
            ap = a;
            ep = e;
            ip = i;
            op = o;
            up = u;
        }
        return ((((ap + ep)%MOD + ip)%MOD + op)%MOD + up) % MOD;
    }
}
