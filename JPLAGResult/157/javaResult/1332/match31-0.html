class Solution {
    public int countVowelPermutation(int n) {
        if (n == 1 || n == 2) return 5 * n;
        int[] arr = new int[6];
        int m = 1000000007;
        for (int i = 0; i < arr.length; i++) arr[i] = 1;
        for (int i = 2; i <= n; i++) {
            int[] temp = new int[6];
            temp[2] += arr[1];
            temp[2] = temp[2] % m;
            
            temp[1] += arr[2];
            temp[1] = temp[1] % m;
            temp[3] += arr[2];
            temp[3] = temp[3] % m;
            
            temp[1] += arr[3];
                        temp[1] = temp[1] % m;
            temp[2] += arr[3];
                        temp[2] = temp[2] % m;
            temp[4] += arr[3];
                        temp[4] = temp[4] % m;
            temp[5] += arr[3];
                        temp[5] = temp[5] % m;
            
            temp[3] += arr[4];
                        temp[3] = temp[3] % m;
            temp[5] += arr[4];
                                    temp[5] = temp[5] % m;
            
            temp[1] += arr[5];
                                    temp[1] = temp[1] % m;
            
            arr = temp;
        }
        int res = 0;
        for (int i = 1; i < arr.length; i++) {
            res += arr[i];
            res = res % m;
        }
        return res;
    }
}
