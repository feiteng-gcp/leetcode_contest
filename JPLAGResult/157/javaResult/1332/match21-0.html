class Solution {
    long N = 1000000000 +7;
    public int countVowelPermutation(int n) {
        long[] end = new long[5];
        Arrays.fill(end,1);
        
        for(int i=1;i<n;i++) {
            long[] lat = new long[5];
            lat[0] = end[1] % N;
            lat[1] = (end[0]+ end[2] ) %N;
            lat[2] = (end[0] +end[1] + end[3] +end[4]) %N;
            lat[3] = (end[2] + end[4]) %N;
            lat[4] = (end[0])%N;
            end = lat;
        }
        return (int)((end[0] +end[1] +end[2] +end[3] +end[4])%N);
        
    }
}
