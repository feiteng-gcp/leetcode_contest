class Solution {
    
public int countVowelPermutation(int n) {
        long[] dp = new long[5];
        Arrays.fill(dp, 1);
        for (int i = 1; i < n; i++) {
            long[] dp2 = new long[5];
            dp2[1] += dp[0];
            dp2[0] += dp[1]; dp2[2] += dp[1];
            dp2[0] += dp[2]; dp2[1] += dp[2]; dp2[3] += dp[2]; dp2[4] += dp[2];
            dp2[2] += dp[3]; dp2[4] += dp[3];
            dp2[0] += dp[4];
            dp = dp2;
            
for (int j = 0; j < 5; j++) dp[j] %= 1000000007;
        }
        long res = 0;
        for (long d : dp) {
            res += d;
            res %= 1000000007;
        }
        return (int) res;
    }
}
