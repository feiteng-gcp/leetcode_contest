class Solution {
    public int countVowelPermutation(int n) {
        int MODULE = 1_000_000_007;
        int[] a = new int[n], e = new int[n], i = new int[n], o = new int[n], u = new int[n];

        a[0] = e[0] = i[0] = o[0] = u[0] = 1;
        for (int j = 1; j < n; j++) {
            a[j] = ((e[j - 1] + u[j - 1]) % MODULE + i[j - 1]) % MODULE;
            e[j] = (a[j - 1] + i[j - 1]) % MODULE;
            i[j] = (e[j - 1] + o[j - 1]) % MODULE;
            o[j] = i[j - 1];
            u[j] = (o[j - 1] + i[j - 1]) % MODULE;
        }

        return ((((((a[n - 1] + e[n - 1]) % MODULE) + i[n - 1]) % MODULE) + o[n - 1]) % MODULE + u[n - 1]) % MODULE;
    }
}
