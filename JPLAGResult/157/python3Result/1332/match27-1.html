class Solution:
    def countVowelPermutation(self, n: int) -> int:
        dp = [1 for _ in range(5)]
        MOD = 10**9+7
        for i in range(1, n):
            dp_ = [0,0,0,0,0]
            dp_
[0] = dp[1]
            dp_[1] = (dp[0]+dp[2])%MOD
            dp_[2] = (dp[0]+dp[1]+dp[3]+dp[4])%MOD
            dp_[3] = (dp[2]+dp[4])%MOD
            dp_[4] = dp[0]
            dp = dp_
        return sum(dp)%MOD
