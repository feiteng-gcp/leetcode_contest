class Solution:
    def countVowelPermutation(self, n: int) -> int:

        a = 0
        e = 1
        i = 2
        o = 3
        u = 4
        

        dp = [1, 1, 1, 1, 1]
        
        for _ in range(n - 1):
            tmp = [0] * 5
            tmp[a] += dp[e] + dp[i] + dp[u]
            tmp[e] += dp[a] + dp[i]
            tmp[i] += dp[e] + dp[o]
            tmp[o] += dp[i]
            tmp[u] += dp[i] + dp[o]

            dp = tmp
        
        return sum(dp) % (10 ** 9 + 7)
