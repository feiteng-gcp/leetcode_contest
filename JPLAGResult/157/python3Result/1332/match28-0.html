class Solution:
    def countVowelPermutation(self, n: int) -> int:
        '''
            a,0, -> 1,2,4
            e,1, -> 0,2
            i,2, -> 1,3
            o,3, -> 2
            u,4, -> 2,3
        '''
        dp = [1] * 5
        mod = 10**9 + 7
        for i in range(1, n):
            newdp = [0]*5
            newdp[0] = dp[1] % mod
            newdp[1] = (dp[0] + dp[2]) % mod
            newdp[2] = (dp[0] + dp[1] + dp[3] + dp[4]) % mod
            newdp[3] = (dp[2] + dp[4]) % mod
            newdp[4] = dp[0] % mod
            dp = newdp
        return sum(dp) % mod
