class Solution:
    def countVowelPermutation(self, n: int) -> int:
        #dp
        MOD = (10**9+7)
        if n<1:
            return 0
        dp = {"a":1,"e":1,"i":1,"o":1,"u":1}
        
for i in range(2,n+1):
            newdp = {}
            newdp["a"] = (dp["e"] + dp["i"] + dp["u"])%MOD
            newdp["e"] = (dp["a"] + dp["i"])%MOD
            newdp["i"] = (dp["e"] + dp["o"])%MOD
            newdp["o"] = dp["i"]%MOD
            newdp["u"] = (dp["i"]+dp["o"])%MOD
            dp = newdp
        return sum(dp.values())%MOD
