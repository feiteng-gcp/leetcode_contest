class Solution:
    def countVowelPermutation(self, n: int) -> int:
        dp = [1]*5
        const = 10**9 + 7
        for _ in range(n-1):
            new = [0]*5
            new[0] = dp[1]
            new[1] = dp[0] + dp[2]
            new[2] = dp[0] + dp[1] + dp[3] + dp[4]
            new[3] = dp[2] + dp[4]
            new[4] = dp[0]
            for i in range(5):
                new[i] %= const
            dp = new
        return sum(dp)%const
