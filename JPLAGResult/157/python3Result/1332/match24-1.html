class Solution:
    def countVowelPermutation(self, n: int) -> int:
        c = [1] * 5
        MOD = 10 ** 9 + 7
        for i in range(n - 1):
            c = [c[1] + c[2] + c[4], c[0] + c[2], c[1] + c[3], c[2], c[2] + c[3]]
            for j in range(5):
                c[j] %= MOD
        return sum(c) % MOD
