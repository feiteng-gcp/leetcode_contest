class Solution:
    def countVowelPermutation(self, n: int) -> int:
        mods = 10 ** 9 + 7
        dp = [1] * 5
        for i in range(n-1):
            dp = [dp[1], dp[0] + dp[2], dp[0] + dp[1] + dp[3]+dp[4], dp[2]+dp[4], dp[0]]
            for j in range(5):
                dp[j] = dp[j] % mods
        return sum(dp) % mods
