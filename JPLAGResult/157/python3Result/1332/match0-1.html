class Solution:
    def countVowelPermutation(self, n: int) -> int:
        dp = [1]*5
        MOD 
= 10**9 + 7
        for _ in range(n-1):
            dp2 = [0]*5
            dp2[1] += dp[0]
            dp2[0] += dp[1]
            dp2[2] += dp[1]
            dp2[0] += dp[2]
            dp2[1] += dp[2]
            dp2[3] += dp[2]
            dp2[4] += dp[2]
            dp2[2] += dp[3]
            dp2[4] += dp[3]
            dp2[0] += dp[4]
            dp = [x % MOD for x in dp2]
        return sum(dp) % MOD
