class Solution:
    def countVowelPermutation(self, n: int) -> int:
        MOD = int(1e9 + 7)
        if n <= 0:
            return 0
        states = [1] * 5
        
for i in range(n - 1):
            next_states = [0] * 5
            next_states[0] = (states[1] + states[2] + states[4]) % MOD
            next_states[1] = (states[0] + states[2]) % MOD
            next_states[2] = (states[1] + states[3]) % MOD
            next_states[3] = states[2]
            next_states[4] = (states[2] + states[3]) % MOD
            states = next_states
        return sum(states) % MOD
