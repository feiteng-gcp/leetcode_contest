class Solution:
    def countVowelPermutation(self, n: int) -> int:
        mod = 10 ** 9 + 7
        dp = [[0] * 5 for _ in range(n)]
        dp[0][0] = 1
        dp[0][1] = 1
        dp[0][2] = 1
        dp[0][3] = 1
        dp[0][4] = 1
        
for i in range(n - 1):
            dp[i + 1][0] = dp[i][1]
            dp[i + 1][1] = dp[i][0] + dp[i][2]
            dp[i + 1][2] = dp[i][0] + dp[i][1] + dp[i][3] + dp[i][4]
            dp[i + 1][3] = dp[i][2] + dp[i][4]
            dp[i + 1][4] = dp[i][0]
        return sum(dp[-1]) % mod
