class Solution:
    def countVowelPermutation(self, n: int) -> int:
        dp = {
            "a": 1,
            "e": 1,
            "i": 1,
            "o": 1,
            "u": 1,
        }
        MOD = 1000000007
        for i in range(1, n):
            next_dp = {}
            next_dp["a"] = (dp["e"] + dp["i"] + dp["u"]) % MOD
            next_dp["e"] = (dp["a"] + dp["i"]) % MOD
            next_dp["i"] = (dp["e"] + dp["o"]) % MOD
            next_dp["o"] = dp["i"] % MOD
            next_dp["u"] = (dp["i"] + dp["o"]) % MOD
            dp = next_dp
        return (dp["a"] + dp["e"] + dp["i"] + dp["o"] + dp["u"]) % MOD
