class Solution:
    def countVowelPermutation(self, n: int) -> int:
        
        dp = [1]*5
        
        for _ in range(n-1):
            
            new_dp = [0]*5
            
            #add a
            new_dp[1] += dp[0]
            #add e
            new_dp[0] += dp[1]
            new_dp[2] += dp[1]
            
            #add i
            new_dp[0] += dp[2]
            new_dp[1] += dp[2]
            new_dp[3] += dp[2]
            new_dp[4] += dp[2]
            
            #add o
            new_dp[2] += dp[3]
            new_dp[4] += dp[3]
            
            #add u
            new_dp[0] += dp[4]
            dp = new_dp
            
        return sum(dp)%1000000007
