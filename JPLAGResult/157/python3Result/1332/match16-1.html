class Solution:
    def countVowelPermutation(self, n: int) -> int:
        
        mod = 10 ** 9 + 7
        dp 
= [1] * 5
        for i in range(n-1):
            n_dp = [0] * 5
            n_dp[0] = (dp[1] + dp[2] + dp[4]) % mod
            n_dp[1] = (dp[0] + dp[2]) % mod
            n_dp[2] = (dp[1] + dp[3]) % mod
            n_dp[3] = (dp[2]) % mod
            n_dp[4] = (dp[2] + dp[3]) % mod
            dp = n_dp
        return sum(dp) % mod
