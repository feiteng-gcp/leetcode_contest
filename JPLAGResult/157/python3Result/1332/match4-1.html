class Solution:
    def countVowelPermutation(self, n: int) -> int:
        if n == 1: return 5
        dic = {x: 1 for x in "aeiou"}
        
for i in range(1, n):
            new_dic = {}
            new_dic['a'] = dic['e'] + dic['i'] + dic['u']
            new_dic['e'] = dic['a'] + dic['i']
            new_dic['i'] = dic['e'] + dic['o']
            new_dic['o'] = dic['i']
            new_dic['u'] = dic['i'] + dic['o']
            dic = new_dic
        return sum(dic.values()) % (10**9 + 7)
