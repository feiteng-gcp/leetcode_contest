class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        n = len(grid)
        m = len(grid[0])
        vis = [[False for _ in range(m)] for _ in range(n)]
        dx = [-1,0,0,1]
        dy = [0,-1,1,0]
        z = [0]
        def dfs(x, y):
            best = 0
            for i in range(4):
                nx = x + dx[i]
                ny = y + dy[i]
                if 0 <= nx < n and 0 <= ny < m and not vis[nx][ny] and grid[nx][ny] > 0:
                    vis[nx][ny] = True
                    best = max(best, dfs(nx, ny))
                    vis[nx][ny] = False
            return best + grid[x][y]
                    
            
            
        def startDfs(x, y):
            vis[x][y] = True
            q = dfs(x, y)
            vis
[x][y] = False
            return q
            
        ans = 0
        for i in range(n):
            for j in range(m):
                if grid[i][j]:
                    ret = startDfs(i,j)
                    # print(i)
                    # print(j)
                    # print(ret)
                    ans = max(ans, ret)
        # print(vis)
        return ans
