class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        res = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] > 0:
                    res = max(res, self.getMax(grid, i, j))
                    #print(grid)
        return res
                    
    def getMax(self, grid, i, j):
        res = grid[i][j]
        grid[i][j] = 0
        nexres = 0
        for di, dj in [(1,0),(0,1),(-1,0),(0,-1)]:
            if 0<=i+di<len(grid) and 0<=j+dj<len(grid[0]) and grid[i+di][j+dj] > 0:
                nexres = max(nexres, self.getMax(grid, i+di, j+dj))
        grid[i][j] = res
        return res + nexres
