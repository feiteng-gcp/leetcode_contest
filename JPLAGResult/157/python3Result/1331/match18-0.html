class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        starts = []
        m, n = len(grid), len(grid[0])
        possible = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    starts.append((i, j))
                    possible += grid[i][j]
        def maxStart(i, j, possible):
            curr = grid[i][j]
            possible -= curr
            grid[i][j] = 0
            best = 0
            if i - 1 >= 0 and grid[i - 1][j]:
                best = max(best, maxStart(i - 1, j, possible))
                if best == possible:
                    grid[i][j] = curr
                    return curr + best                    
            if i + 1 < m and grid[i + 1][j]:
                best = max(best, maxStart(i + 1, j, possible))
                if best == possible:
                    grid[i][j] = curr
                    return curr + best
            if j - 1 >= 0 and grid[i][j - 1]:
                best = max(best, maxStart(i, j - 1, possible))
                if best == possible:
                    grid[i][j] = curr
                    return curr + best
            if j + 1 < n and grid[i][j + 1]:
                best = max(best, maxStart(i, j + 1, possible))
            grid[i][j] = curr
            return curr + best
        ans = 0
        for i, j in starts:
            ans = max(ans, maxStart(i, j, possible))
            if ans == possible:
                return ans
        return ans
