class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        ds = [(-1,0),(1,0),(0,-1),(0,1)]
        visited = [[False]*n for _ in range(m)]
        def dfs(i, j):
            if not (0<=i<m and 0<=j<n) or visited[i][j] or grid[i][j]==0:
                return 0
            visited[i][j] = True
            ret = max([dfs(i+d[0], j+d[1]) for d in ds]) + grid[i][j] 
            visited[i][j] = False
            return ret
        ret = 0
        for i in range(m):
            for j in range(n):
                ret = max(ret, dfs(i, j))
        return ret
