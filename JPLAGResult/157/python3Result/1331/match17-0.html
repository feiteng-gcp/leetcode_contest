class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        return solve(grid)


def solve(g):
    rows = len(g)
    cols = len(g[0])
    ret = 0
    v = [[0] * cols for _ in range(rows)]

    def dfs(x, y, c):
        nonlocal ret
        v[x][y] = 1
        c += g[x][y]
        found_next = False
        for dx, dy in ((0, -1), (0, 1), (1, 0), (-1, 0)):
            if (
                0 <= x + dx < rows
                and 0 <= y + dy < cols
                and g[x + dx][y + dy] != 0
                and not v[x + dx][y + dy]
            ):
                dfs(x + dx, y + dy, c)
                found_next = True
        if not found_next:
            ret = max(ret, c)
        v[x][y] = 0
        pass

    for i in range(rows):
        for j in range(cols):
            if g[i][j] != 0:
                dfs(i, j, 0)
    return ret


def _case(*a):
    assert solve(*a[:-1]) == a[-1]
    pass
