class Solution:
    def getMaximumGold(self, A: List[List[int]]) -> int:
        m, n = len(A), len(A[0])
        self.res = 0
        def dfs(curr, x, y):
            nonlocal A
            if x < 0 or x >= m or y < 0 or y >= n or A[x][y] == 0:
                self.res = max(self.res, curr)
                return 
            c = A[x][y]
            A[x][y] = 0
            for dx, dy in((1, 0), (0, 1), (-1, 0), (0, -1)):
                dfs
(curr + c, x + dx, y + dy)
            A[x][y] = c
        for i in range(m):
            for j in range(n):
                dfs(0, i, j)
        return self.res
