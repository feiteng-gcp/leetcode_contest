class Solution:
    def getMaximumGold(self, a: List[List[int]]) -> int:
        h = len(a)
        w = len(a[0])
        
        ans = 0
        
        
        visited = []
        for _ in range(h):
            visited.append([False] * w)

        
def f(y, x):
            if y < 0 or y >= h or x < 0 or x >= w or visited[y][x] or a[y][x] == 0:
                return 0
            visited[y][x] = True
            ret = a[y][x]
            ret += max(
                f(y + 1, x),
                f(y - 1, x),
                f(y, x + 1),
                f(y, x - 1),
            )
            visited[y][x] = False
            return ret
            
        
        
for y in range(h):
            for x in range(w):
                ans = max(ans, f(y, x))
                
        return ans
