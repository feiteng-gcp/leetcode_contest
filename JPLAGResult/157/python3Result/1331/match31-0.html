class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        n1 = len(grid)
        n2 = len(grid[0])
        visits = [[0 for b in range(n2)] for a in range(n1)]
        
        def visit(a, b) :
            if grid[a][b] == 0 :
                return 0
            
            to_ret = 0
            if a > 0 and visits[a-1][b] == 0 :
                visits[a-1][b] = 1
                to_ret = max(to_ret, visit(a-1, b))
                visits[a-1][b] = 0
            if a < n1-1 and visits[a+1][b] == 0 :
                visits[a+1][b] = 1
                to_ret = max(to_ret, visit(a+1, b))
                visits[a+1][b] = 0
            if b > 0 and visits[a][b-1] == 0 :
                visits[a][b-1] = 1
                to_ret = max(to_ret, visit(a, b-1))
                visits[a][b-1] = 0
            if b < n2-1 and visits[a][b+1] == 0 :
                visits[a][b+1] = 1
                to_ret = max(to_ret, visit(a, b+1))
                visits[a][b+1] = 0
            
            return grid[a][b] + to_ret
            
        
        maxt = 0
        for a in range(n1) :
            for b in range(n2) :
                visits = [[0 for b in range(n2)] for a in range(n1)]
                visits[a][b] = 1
                maxt = max(maxt, visit(a, b))
        return maxt
