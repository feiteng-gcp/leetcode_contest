class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        self.ans = -math.inf
        def helper(i, j, sol):
            #print(i, j, sol)
            temp = grid[i][j]
            grid[i][j] = 0
            good = False
            for a, b in ((1, 0), (-1, 0), (0,1), (0,-1)):
                if 0 <= i + a < m and 0 <= j + b < n and grid[i+a][j+b] != 0:
                    #print('ready to in', sol + temp, sol, temp)
                    helper(i + a, j + b, sol + temp)
                    
                    good = True
            grid[i][j] = temp
            if not good:
                self.ans = max(self.ans, sol + temp)
        
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    helper(i, j, 0)
        return self.ans
