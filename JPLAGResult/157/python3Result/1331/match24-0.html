import itertools

class Solution:
    def dfs(self, i, j, m, n, grid, seen, cur_gold, max_gold):
        seen.add((i, j))
        cur_gold = cur_gold + grid[i][j]
        max_gold[0] = max(max_gold[0], cur_gold)

        for ni, nj in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
            if 0 <= ni < m and 0 <= nj < n and grid[ni][nj] != 0 and (ni, nj) not in seen:
                self.dfs(ni, nj, m, n, grid, seen, cur_gold, max_gold)
                seen.remove((ni, nj))

    def getMaximumGold(self, grid) -> int:

        m = len(grid)
        n = len(grid[0])
        gold_positions = [(i, j) for i, j in itertools.product(range(m), range(n)) if grid[i][j] != 0] 

        max_gold = [-1]

        for i, j in gold_positions:

            self.dfs(i, j, m, n, grid, set(), 0, max_gold)
        
        return max_gold[0]
