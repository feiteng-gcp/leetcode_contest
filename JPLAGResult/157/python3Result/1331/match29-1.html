class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        ### naive DFS?
        if not grid or not grid[0]: return 0
        self.res = 0
        M=len(grid)
        N=len(grid[0])
        visited = set()
        def dfs(i,j,g):
            if (i,j) in visited: return
            visited.add((i,j))
            g+=grid[i][j]
            self.res = max
(self.res,g)
            for ii,jj in [(i-1,j),(i+1,j),(i,j-1),(i,j+1)]:
                if 0<=ii<M and 0<=jj<N and (ii,jj) not in visited and grid[ii][jj]!=0:
                    dfs(ii,jj,g)
            visited.remove((i,j))
            return
            
        for i,row in enumerate(grid):
            for j,n in enumerate(grid[i]):
                if n!=0:
                    visited=set()
                    dfs(i,j,0)
                    
        return self.res
