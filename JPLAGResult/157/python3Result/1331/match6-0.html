DIRECTION = [(0, 1), (0, -1), (-1, 0), (1, 0)]
class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        rs = 0
        m = len(grid)
        n = len(grid[0])
        for i in range(m):
            for j in range(n):
                if grid[i][j] > 0:
                    rs = max(rs, self.dfs(grid, i, j))
        return rs
    
    def dfs(self, grid, curX, curY):
        m = len(grid)
        n = len(grid[0])
        value = grid[curX][curY]
        rs = value
        for dx, dy in DIRECTION:
            newX = curX + dx
            newY = curY + dy
            if newX >= 0 and newX < m and newY >= 0 and newY < n and grid[newX][newY] > 0:
                grid[curX][curY] = 0
                rs = max(rs, value + self.dfs(grid, newX, newY))
                grid[curX][curY] = value
        return rs
