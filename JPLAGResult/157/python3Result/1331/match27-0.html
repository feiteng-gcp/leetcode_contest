class Solution:
    def solve(self, grid, row, col):
        m, n = len(grid), len(grid[0])
        if row<0 or col<0 or row>=m or col>=n or grid[row][col]==0:
            return 0
        
        tmp = grid[row][col]
        res = 0
        grid[row][col] = 0
        res = max(res, self.solve(grid, row-1, col))
        res = max(res, self.solve(grid, row, col-1))
        res = max(res, self.solve(grid, row+1, col))
        res = max(res, self.solve(grid, row, col+1))
        grid[row][col] = tmp
        return res + tmp
        
        
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        res = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    continue
                res = max(res, self.solve(grid, i, j))
        return res
