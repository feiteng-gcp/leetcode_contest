class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        
        if not grid or not grid[0]:
            return 0
        
        M, N = len(grid), len(grid[0])
        
        res = 0
        
        def dfs(r,c, tot):
            nonlocal res, M, N
            archive = grid[r][c]
            grid[r][c] = 0
            tot_update 
= tot + archive
            flag = False
            for nr, nc in [(r+1,c),(r,c+1),(r-1,c),(r,c-1)]:
                if 0<=nr<M and 0<=nc<N and grid[nr][nc]:
                    dfs(nr,nc,tot_update)
                    flag = True
            
            #restore
            grid[r][c] = archive
            if not flag: #this is one end.
                res = max(res,tot_update)
