class Solution:
    def getMaximumGold(self, grid: 'List[List[int]]') -> int:
        h = len(grid)
        w = len(grid[0])

        res = 0
        
for i in range(h): 
            for j in range(w): 
                if grid[i][j] != 0: 
                    res = max(res, self.dfs(grid, i, j))
        
        return res
    
    def dfs(self, grid, i, j): 
        h = len(grid)
        w = len(grid[0])

        v = grid[i][j] 
        res = 0

        grid[i][j] = 0
        for di, dj in ((0,1), (1,0), (0,-1), (-1,0)): 
            ni = i + di
            nj = j + dj

            if ni < 0 or ni >= h or nj < 0 or nj >= w: 
                continue
            if grid[ni][nj] == 0: 
                continue 

            res = max(res, self.dfs(grid, ni, nj))
        
        grid[i][j] = v

        return res + v
