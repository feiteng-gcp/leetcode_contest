class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        
        
def dfs(x, y, g, visited):
            visited.add((x,y))
            g += grid[x][y]
            self.res = max(self.res, g)
            for dx,dy in ((0,1),(0,-1),(1,0),(-1,0)):
                if check(x+dx, y+dy) and (x+dx,y+dy) not in visited:
                    dfs(x+dx, y+dy, g, visited)
            visited.remove((x,y))
            
        def check(x,y):
            if 0 <= x < r and 0 <= y < c and grid[x][y] > 0:
                return True
            return False
        
        self.res = 0
        r,c = len(grid), len(grid[0])
        for x in range(r):
            for y in range(c):
                if check(x,y):
                    dfs(x,y,0,set())
        return self.res
