class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        
        def dfs(i,j):
            if grid[i][j]==0:
                return 0
            num = grid[i][j]
            grid[i][j] = 0
            maxi = 0
            for h,v in [[1,0],[-1,0],[0,1],[0,-1]]:
                x, y = i+v, j+h
                if x>=0 and x<m and y>=0 and y<n and grid[x][y]!=0:
                    maxi = max(maxi, dfs(x,y))
            grid[i][j] = num
            return maxi+num
        
        res = 0
        for i in range(m):
            for j in range(n):
                res = max(res, dfs(i,j))
        return res
