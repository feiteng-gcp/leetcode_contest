class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        used = [[False] * n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0: used[i][j] = True
        ret = 0
        
        def dfs(i, j, tot):
            nonlocal ret
            
            used[i][j] = True
            tot += grid[i][j]
            if tot > ret: ret = tot
            for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                if 0 <= i + dx < m and 0 <= j + dy < n and not used[i+dx][j+dy]:
                    dfs(i+dx, j+dy, tot)
            used[i][j] = False
        
        for i in range(m):
            for j in range(n):
                if not used[i][j]: dfs(i, j, 0)
                    
        return ret
