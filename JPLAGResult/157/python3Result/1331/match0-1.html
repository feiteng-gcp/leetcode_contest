class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        ans = 0
        n, m = len(grid), len
(grid[0])
        mark = [[0 for _ in range(m)] for _ in range(n)]
        def dfs(x, y, current):
            mark[x][y] = 1
            current += grid[x][y]
            ans = current
            if (y < m -1) and (grid[x][y+1] > 0) and (mark[x][y+1] < 1):
                ans = max(ans, dfs(x, y+1, current))
            if (y > 0) and (grid[x][y-1] > 0) and (mark[x][y-1] < 1):
                ans = max(ans, dfs(x, y-1, current))
            if (x < n -1) and (grid[x+1][y] > 0) and (mark[x+1][y] < 1):
                ans = max(ans, dfs(x+1, y, current))
            if (x > 0) and (grid[x-1][y] > 0) and (mark[x-1][y] < 1):
                ans = max(ans, dfs(x-1, y, current))
            mark[x][y] = 0
            return ans
            
        
for i in range(n):
            for j in range(m):
                if grid[i][j]>0:
                    ans = max(ans, dfs(i, j, 0))
        return ans
