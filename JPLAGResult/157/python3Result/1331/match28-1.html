class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        res = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] > 0:
                    res = max(res, self.helper(grid, (i, j)))
        return res
    
    def helper(self, grid, current_pos):
        i, j = current_pos
        if grid[i][j] == 0:
            return 0
        current_gold = grid[i][j]
        di = [0, 1, -1, 0]
        dj = [1, 0, 0, -1]
        next_gold = 0
        grid[i][j] = 0
        for k in range(4):
            new_gold = 0
            newi = i + di[k]
            newj = j + dj[k]
            if newi >= 0 and newi < len(grid) and newj >= 0 and newj < len(grid[0]):
                new_gold = self.helper(grid, (newi, newj))
            next_gold = max(next_gold, new_gold)
        
        grid[i][j] = current_gold
        return current_gold + next_gold
