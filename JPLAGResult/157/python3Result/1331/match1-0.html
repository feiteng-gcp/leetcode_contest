from typing import List, Tuple
from collections import defaultdict


class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        visited_arr = [[False for _ in range(n)] for _ in range(m)]

        def dfs(x: int, y: int) -> int:
            visited_arr[x][y] = True
            gold = grid[x][y]
            dfs_gold = 0
            if x > 0 and not visited_arr[x - 1][y] and grid[x - 1][y] != 0:
                dfs_gold = max(dfs_gold, dfs(x - 1, y))
            if y > 0 and not visited_arr[x][y - 1] and grid[x][y - 1] != 0:
                dfs_gold = max(dfs_gold, dfs(x, y - 1))
            if x < m - 1 and not visited_arr[x + 1][y] and grid[x + 1][y] != 0:
                dfs_gold = max(dfs_gold, dfs(x + 1, y))
            if y < n - 1 and not visited_arr[x][y + 1] and grid[x][y + 1] != 0:
                dfs_gold = max(dfs_gold, dfs(x, y + 1))
            visited_arr[x][y] = False

            return gold + dfs_gold

        max_gold = -1
        for i in range(m):
            for j in range(n):
                if grid[i][j] != 0:
                    max_gold = max(max_gold, dfs(i, j))

        return max_gold
