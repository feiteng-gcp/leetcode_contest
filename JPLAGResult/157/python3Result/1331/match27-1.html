class Solution:
    
def getMaximumGold(self, grid: List[List[int]]) -> int:
        R, C = len(grid), len(grid[0])
        max_gold = 0
        
        for r in range(R):
            for c in range(C):
                if grid[r][c] == 0:
                    continue
                visited = set([(r,c)])
                max_gold = max(max_gold, self.dfs(grid, r, c, visited, grid[r][c]))
        
        return max_gold
    
    def dfs(self, grid, r, c, visited, gold):
        R, C = len(grid), len(grid[0])
        max_gold = gold
        for nr, nc in ((r+1, c), (r-1, c), (r, c+1), (r, c-1)):
            if 0 <= nr < R and 0 <= nc < C and grid[nr][nc] != 0 and (nr, nc) not in visited:
                visited.add((nr, nc))
                max_gold = max(max_gold, self.dfs(grid, nr, nc, visited, gold+grid[nr][nc]))
                visited.remove((nr, nc))
        
        return max_gold
