class Solution:
	def getMaximumGold(self, grid):
		m, n = len(grid), len
(grid[0])
		res = 0
		for x in range(m):
			for y in range(n):
				if grid[x][y] > 0:
					res = max(res, self.dfs(x, y, grid))
		return res

	def dfs(self, x, y, grid):
		m, n = len(grid), len(grid[0])
		if x < 0 or x >= m or y < 0 or y >= n or grid[x][y] == 0:
			return 0
		cur 
= grid[x][y]
		res = 0
		grid[x][y] = 0
		for d1, d2 in [[1, 0], [-1, 0], [0, 1], [0, -1]]:
			newX, newY = x + d1, y + d2
			res = max(res, cur + self.dfs(newX, newY, grid))
		grid[x][y] = cur
		return res
