class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        h = len(grid)
        w = len(grid[0])
        ret = [0]
        dis = [(0, 1), (0, -1), (1, 0), (-1, 0)]
        
        def find(x, y, cur, ret):
            if x < 0 or x >= h or y < 0 or y >= w:
                return
            if not grid[x][y]:
                return
            tmp = grid[x][y]
            grid[x][y] = 0
            cur += tmp
            ret[0] = max(ret[0], cur)
            for _x, _y in dis:
                find(x + _x, y + _y, cur, ret)
            grid[x][y] = tmp
        
        for i in range(h):
            for j in range(w):
                find(i, j, 0, ret)
        return ret[0]
