class Solution:
    def getMaximumGold(self, grid):
        h, w = len(grid), len(grid[0])
        ans = 0
        for i in range(h):
            for j in range(w):
                if grid[i][j] > 0:
                    ans = max(ans, self.explore(grid, [[False for j in range(w)] for i in range(h)], i, j, h, w))
        return ans

    def explore(self, grid, visited, y, x, h, w):
        ans = 0
        visited[y][x] = True
        for direction in ((0, 1), (0, -1), (1, 0), (-1, 0)):
            yy, xx = y + direction[0], x + direction[1]
            if 0 <= yy < h and 0 <= xx < w and grid[yy][xx] > 0:
                if not visited[yy][xx]:
                    ans = max(ans, self.explore(grid, visited, yy, xx, h, w))
        visited[y][x] = False
        return ans + grid[y][x]
