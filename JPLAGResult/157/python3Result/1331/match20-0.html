class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        if not grid:
            return 0
        m,n = len(grid), len(grid[0])
        def dfs(i, j, val):
            nonlocal maxval
            if i<0 or i>=m or j<0 or j>=n or visited[i][j] or grid[i][j]==0:
                return
            visited[i][j] = True
            val += grid[i][j]
            maxval = max(maxval, val)
            dfs(i-1,j,val)
            dfs(i+1,j,val)
            dfs(i,j-1,val)
            dfs(i,j+1,val)
            val -= grid[i][j]
            visited[i][j] = False
        res = 0
        for i in range(m):
            for j in range(n):
                maxval = 0
                visited = [[False]*n for _ in range(m)]
                dfs(i, j, 0)
                res = max(res, maxval)
        return res
