class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        if not grid or not grid[0]: return 0
        row, column = len(grid), len(grid[0])
        self.res = 0
        
        def dfs(x, y, prev):
            num = grid[x][y]
            if prev + num > self.res:
                self.res = prev + num
            grid[x][y] = 0
            for xx, yy in [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]:
                if 0<=xx<row and 0<=yy<column and grid[xx][yy]:
                    dfs(xx, yy, prev+num)
            grid[x][y] = num
        
        for i in range(row):
            for j in range(column):
                if grid[i][j]:
                    dfs(i, j, 0)
        return self.res
