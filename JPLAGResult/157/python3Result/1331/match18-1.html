class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        starts = []
        N = len(grid)
        if not N:
            return 0
        M = len(grid[0])
        if not M:
            return 0
        
        for i in range(N):
            for j in range(M):
                if grid[i][j]:
                    starts.append((
i, j))
        
        def dfs(i, j, now):
            # print('i: {}, j: {}, now: {}'.format(i, j, now))
            v = grid[i][j]
            now += v
            grid[i][j] = 0
            m = now
            if i > 0 and grid[i-1][j]:
                m = max(m, dfs(i-1, j, now)) 
            if j > 0 and grid[i][j-1]:
                m = max(m, dfs(i, j-1, now)) 
            if i < N-1 and grid[i+1][j]:
                m = max(m, dfs(
i+1, j, now)) 
            if j < M-1 and grid[i][j+1]:
                m = max(m, dfs(i, j+1, now)) 
            grid[i][j] = v
            return m
        
        
        m = 0
        for i, j in starts:
            m = max(m, dfs(i, j, 0))
        return m
