class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        M, N = len(grid), len(grid[0])
        res = 0
        def dfs(i, j):
            if i < 0 or i >= M or j < 0 or j >= N or grid[i][j] == 0:
                return 0
            att = grid[i][j]
            grid[i][j] = 0
            res = att + max(dfs(i + 1, j), dfs(i - 1, j), dfs(i, j + 1), dfs(i, j - 1))
            grid[i][j] = att
            return res
            
            
        
        for i in range(M):
            for j in range(N):
                if grid[i][j] == 0:
                    continue
                res = max(res, dfs(i, j))
        return res
