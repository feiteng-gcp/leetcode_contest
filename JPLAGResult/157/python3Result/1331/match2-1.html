class Solution:
    def getMaximumGold(self, grid: List[List[int]]) -> int:
        res = 0
        for x in range(len(grid)):
            for y in range(len(grid[0])):
                res = max(res, self.acq(grid, x, y))
        return res
        
    def acq(self, grid, x, y):
        if grid[x][y] == 0: return 0
        cur 
= grid[x][y]
        grid[x][y] = 0
        maxnum = 0 
        for dx, dy in [[1, 0], [-1, 0], [0, 1], [0, -1]]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < len(grid) and 0 <= ny < len(grid
[0]):
                maxnum = max(maxnum, self.acq(grid, nx, ny))
        grid[x][y] = cur
        return cur + maxnum
