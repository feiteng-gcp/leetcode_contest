class Solution:
    def longestSubsequence(self, arr: List[int], difference: int) -> int:
        d
={}
        res=0
        for v in arr:
            lookfor=v-difference
            if lookfor in d:
                d[v]=d[lookfor]+1
            else:
                d[v]=1
            res=max(res,d[v])
        return res
