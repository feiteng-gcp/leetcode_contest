class Solution:
    def longestSubsequence(self, arr, difference: int) -> int:

        hashmap = {}
        max_val = -1

        for val in arr:

            prev_val = val - difference

            if prev_val in hashmap:
                hashmap[val] = hashmap[prev_val] + 1
            else:
                hashmap[val] = 1
            
            max_val = max(max_val, hashmap[val])
        
        return max_val
