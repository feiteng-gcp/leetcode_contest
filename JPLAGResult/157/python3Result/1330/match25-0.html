class Solution:
    def longestSubsequence(self, arr: List[int], difference: int) -> int:
        memo = {}
        for num in arr:
            prev_num = num - difference
            if prev_num in memo:
                memo[num] = memo[prev_num] + 1
            else:
                memo[num] = 1
        return max(memo.values())
