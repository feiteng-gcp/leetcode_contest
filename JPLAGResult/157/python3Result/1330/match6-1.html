class Solution:
    def longestSubsequence(self, arr: List[int], difference: int) -> int:#hash dp[i]代表以i结尾的符合要求的最长序列长度
        dp = {}
        
        for num in arr:
            if num - difference in dp:
                dp[num] = dp[num - difference] + 1
            else:
                dp[num] = 1
        
        return max(dp.values())
