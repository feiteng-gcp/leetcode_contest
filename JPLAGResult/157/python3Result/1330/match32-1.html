class Solution:
    def longestSubsequence(self, arr: List[int], difference: int) -> int:
        if len(arr) == 0:
            return 0
        ans = 1
        dp = dict()
        for num in arr:
            prev = num - difference
            
if prev in dp:
                dp[num] = dp[prev] + 1
            else:
                dp[num] = 1
        
        return max(dp.values())
