class Solution:
    def longestSubsequence(self, arr: 'List[int]', difference: int) -> int:
        
        d = {}

        res 
= 1
        for num in arr: 
            if num - difference in d: 
                d[num] = d[num-difference] + 1
                res = max(res, d[num])
            else: 
                d[num] = 1
        
        #print(d)
        return res
