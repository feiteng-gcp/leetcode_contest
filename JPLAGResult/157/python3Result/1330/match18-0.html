class Solution:
    def longestSubsequence(self, arr: List[int], difference: int) -> int:
        dp = dict()
        res = 1
        for v in arr:
            if v - difference in dp:
                dp[v] = dp[v-difference] + 1
                res = max(res, dp[v])
            else:
                dp[v] = 1
        return res
