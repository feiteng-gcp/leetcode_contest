import collections

class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        n = len(piles)
        dpt = [[0 for j in range(n)] for i in range(n)]
        
for i in range(n):
            dpt[i][i] = -piles[i]
        for k in range(1, n):
            for i in range(0, n - k):
                j = i + k
                if k & 1:
                    dpt[i][j] = max(
                        dpt
[i + 1][j] + piles[i],
                        dpt[i][j - 1] + piles[j]
                    )
                else:
                    dpt[i][j] = min(
                        dpt[i + 1][j] - piles[i],
                        dpt[i][j - 1] - piles[j]
                    )
        return dpt[0][n - 1] > 0
