class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        n=len(piles)
        R=[[0]*n  for i in range(n)]
        for k in range(1,n):
            for i in range(n-k):
                j=i+k
                R
[i][j]=max(piles[i]-R[i+1][j],piles[j]-R[i][j-1])
        return R[0][n-1]>0
