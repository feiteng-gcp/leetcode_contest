class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        n = len(piles)
        dp = [[None] * n for _ in range(n)]
        
        def calc(start, end):
            if dp[start]
[end] is None:
                if start == end:
                    dp[start][end] = piles[start]
                else:
                    dp[start][end] = max(piles[start] - calc(start + 1, end), piles[end] - calc(start, end - 1))
            return dp[start][end]
        
        return calc(0, n - 1) > 0
