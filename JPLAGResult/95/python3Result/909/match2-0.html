class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        l=len(piles)
        dp=[[0 for i in range(l)]for j in range(l+1)]
        s=[0 for i in range(l)]
        s[0]=piles[0]
        for i in range(1,l):
            s[i]=s[i-1]+piles[i]
        for i in range(l):
            dp[1][i]=piles[i]
        for step in range(2,l+1):
            for i in range(l):
                j=i+step-1
                if j>=l:
                    continue
                if i>0:
                    dp[step][i]=max(piles[i]+s[j]-s[i-1]-dp[step-1][i+1],piles[j]+s[j]-s[i-1]-dp[step-1][i])
                else:
                    dp[step][i]=max(piles[i]+s[j]-dp[step-1][i+1],piles[j]+s[j]-dp[step-1][i])
        if dp[l][0]*2>s[l-1]:
            return True
        return False
