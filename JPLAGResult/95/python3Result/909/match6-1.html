class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        cusum=[piles[0]]*len(piles)
        for i in range(1,len(piles)):
            cusum[i]=piles[i]+cusum[i-1]
        dp=[[[piles[i],0] for i in range(len(piles)-l)] for l in range(len(piles))]
        for l in range(1,len(piles)):
            for i in range(len(piles)-l):
                tmp1=max(piles[i]+dp[i+1][l-1][1],piles[i+l]+dp[i][l-1][1])
                if i>0:
                    tmp2=cusum[i+l]-cusum[i-1]-tmp1
                else:
                    tmp2=cusum[i+l]-tmp1
                dp[i][l]=[
tmp1,tmp2]
        return dp[-1][0][0]>dp[-1][0][1]
