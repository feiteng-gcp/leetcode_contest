class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        if not len(piles):return False
        if len(piles)==2:return piles[0]==piles[1]
        rst=[[0]*len(piles) for i in range(len(piles))]
        for i in range(len(piles)-1):
            rst[i][i+1]=max(piles[i:i+2])-min(piles[i:i+2])
        for i in range(2,len(piles)):
            for start in range(len(piles)-i):
                rst[start][start+i]=max(piles[start]-rst[start+1][start+i],piles[start+i]-rst[start][start+i-1])
        return rst[0][-1]>0
