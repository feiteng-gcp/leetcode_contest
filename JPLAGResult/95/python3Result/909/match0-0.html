class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        def func(i,j):
            if (i,j) not in dic:
                if i==j: dic[(i,j)]=piles[i]
                else: dic[(i,j)]=max(piles[i]-func(i+1,j),piles[j]-func(i,j-1))
            return dic[(i,j)]
        dic={}
        return func(0,len(piles)-1)>0
