class Solution:
    def stoneGame(self, piles):
        """
        :type piles: List[int]
        :rtype: bool
        """
        n = len(piles)
        dp = [[[0, 0] for _ in range(n)] for _ in range(n)]
        for i in range(n):
            dp[i][i][0] = piles[i]
        for k in range(1, n):
            for i in range(n - k):
                j = i + k
                dp
[i][j] = max([dp[i + 1][j][1] + piles[i], dp[i + 1][j][0]],
                               [dp[i][j - 1][1] + piles[j], dp[i][j - 1][0]])
        return dp[0][-1][0] >= dp[0][-1][1]
