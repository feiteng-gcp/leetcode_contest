class Solution(object):
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """
        largest_divisor = min(A, B)
        while largest_divisor > 0:
            if A % largest_divisor == 0 and B % largest_divisor == 0:
                break
            else:
                largest_divisor -= 1

        least_multi = A * B // largest_divisor
        each_chunk = (A // largest_divisor) + (B // largest_divisor) - 1
        chunk_num = N // each_chunk
        rest 
= N % each_chunk

        base = chunk_num * least_multi
        count = 0
        a = A
        b = B
        addition = 0
        while count < rest:
            if a < b:
                addition = a
                a += A
            elif a > b:
                addition = b
                b += B
            else:
                addition = a
                a += A
                b += B
            count += 1

        return (base + addition) % 1000000007
