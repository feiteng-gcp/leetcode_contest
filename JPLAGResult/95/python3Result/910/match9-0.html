class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """

        def gcd(a, b):
            if b != 0:
                return gcd(b,a % b)
            else:
                return a
        
        maxY = gcd(A,B)
        minB = A / maxY * B
        
        count = minB / A + minB / B - 1
        
        c = N // count
        
        y = N % count
        
        res = c * minB
        
        start1, start2 = res, res
        while y > 0:
            if start1 + A > start2 + B:
                res = start2 + B
                start2 += B
            else:
                res = start1 + A
                start1 += A
            y -= 1
        return int(res) % (10**9 + 7)
