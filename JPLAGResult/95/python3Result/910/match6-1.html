from math import gcd
class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """
        
        if A > B:
            A, B = B, A
        g = gcd(A, B)
        l = []
        limit = A//g*B
        
for i in range(A, limit+1, A):
            l.append(i)
        for i in range(B, limit+1, B):
            l.append(i)
        l = list(set(l))
        l.sort()
        n = len(l)
        times = N // n
        yu = N % n
        ans = limit*times
        if yu > 0:
            ans += l[yu-1]
        ans = ans % 1000000007
        return ans
