class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """
        lcm = A*B
        values 
= set()
        for x in range(A,lcm+1,A):
            values.add(x)
        for x in range(B,lcm+1,B):
            values.add(x)
        values = list(sorted(values))

        ret = 0
        if N >= len(values):
            ret += lcm * (N // len(values))
        if N % len(values) != 0:
            ret += values[(N % len(values))-1]
            
        return ret % (10**9 + 7)
