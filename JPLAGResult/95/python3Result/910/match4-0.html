import math
class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """
        gcdnum = math.gcd(A, B)
        lcmStep = A // gcdnum + B // gcdnum - 1
        lcm = A * B // gcdnum
        numSteps = N // lcmStep
        N = N - numSteps * lcmStep
        nthNum = (lcm * numSteps) % (10**9 + 7)
        nextNum = [nthNum, nthNum]
        print(nextNum)
        if A > B:
            temp = A
            A = B
            B = temp
        while N > 0:
            if nextNum[0] + A < nextNum[1] + B:
                nextNum[0] += A
            elif nextNum[0] + A == nextNum[1] + B:
                nextNum[0] += A
                nextNum[0] += B
            else:
                nextNum[1] += B
            N -= 1
        return max(nextNum) % (10**9 + 7)
