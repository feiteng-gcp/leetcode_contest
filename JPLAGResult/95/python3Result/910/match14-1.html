class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """

        if A == B or B % A == 0:
            return (A*N) % 1000000007
        if A % B == 0:
            return (B*N) % 1000000007
        lcm = A*B//self.gcd(A, B)
        count = lcm//A + lcm//B-1
        base = (N//count * lcm) % 1000000007
        remain = N % count
        if remain == 0:
            return base
        arr = []
        
for i in range(1, lcm//A):
            arr.append(A*i)
        for i in range(1, lcm//B):
            arr.append(B*i)
        arr.sort()

        res = (base+arr[remain-1]) % 1000000007
        return res

    def gcd(self, a, b):
        a, b = (a, b) if a >= b else (b, a)
        if a % b == 0:
            return b
        else:
            return self.gcd(b, a % b)
