class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """
        if A==B:
            return (N*A)%(10**9+7)
        
        lcm=A*B//math.gcd(A,B)
        c=lcm//A+lcm//B-1
        m=N//c
        ans=m*lcm
        r=N%c
        if r==0:
            return ans%(10**9+7)
        if A>B:
            A,B=B,A
        
        pa,pb
=1,1
        for i in range(r):
            if pa*A<pb*B:
                l=pa*A
                pa+=1
            else:
                l=pb*B
                pb+=1
        return (ans+l)%(10**9+7)
