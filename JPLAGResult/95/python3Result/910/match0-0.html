class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """
        def gcd(a, b):
            while b>0:
                a, b = b, a%b
            return a
        
        def lcm(a, b):
            return a*b//gcd(a, b)
        
        g = gcd(A, B)
        loops = (A+B)//g - 1
        
        extend = lcm(A, B) * (N//loops)
        num = N%loops
        
        base = 0
        a, b = 0, 0
        
        while num>0:
            if a+A<b+B:
                a += A
                base = a
            elif a+A>b+B:
                b += B
                base = b
            else:
                a += A
                b += B
                base = a
            num -= 1
        
        return (base + extend)%(10**9+7)
