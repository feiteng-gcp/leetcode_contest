class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """

        def f(a, b):
            a, b = (a, b) if a >= b else (b, a)
            while b:
                a, b = b, a % b
            return a

        def lcm(a, b):
            return a * b // gcd(a, b)

        gcd = f(A, B)
        t = A * B // gcd
        one = int(t / A + t / B - 1)
        a = N % one
        b = N // one
        tmp = b * t
        t1 = 0
        t2 = 0
        while a != 0:
            if (t1+1) * A < (t2+1)*B:
                t1 += 1
            else:
                t2 += 1
            a -=1
        return (tmp+max(t1*A, t2*B)) % (10**9 + 7)
