class Solution:
    def nthMagicalNumber(self, N, A, B):
        gcd = self.findGCD(A, B)
        lcm = (A * B) // gcd
        single = (lcm // A) + (lcm // B) - 1
        multiple = N // single
        remainder = N % single
        
        if remainder == 0:
            return (multiple * lcm) % (10 ** 9 + 7)
        else:
            rm = self.helper(remainder, A, B, lcm, single, N)
            return (multiple * lcm + rm) % (10 ** 9 + 7)
    
    def helper(self, remainder, A, B, lcm,single, N):

        a = list()
        for i in range(1, lcm//A):
            a.append(i * A)
        for j in range(1, lcm // B):
            a.append(j * B)
        a.sort()
        # print(a)
        return a[remainder-1]
            
            
        
    
    def findGCD(self, A,B):
        mi = min(A, B)
        ma = max(A, B)
        while ma % mi != 0:
            tmp = mi
            mi = ma % mi
            ma = tmp
        return mi
