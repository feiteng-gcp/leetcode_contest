import bisect
class Solution:
    def nthMagicalNumber(self, N, A, B):
        """
        :type N: int
        :type A: int
        :type B: int
        :rtype: int
        """
        def gcd(a,b):
            if a<b:
                a,b=b,a
            
while b:
                a,b=b,a%b
            return a
        def lcm(x,y):
            return x*y//gcd(x,y)
            
        lcm= lcm(A,B)
        pre=lcm//A+lcm//B-1
        count=N//pre
        # return count
        rest=N%pre
        if not rest:return(count*lcm)%(10**9+7)
        mylist=[[A,True],[B,False]]
        mylist.sort()
        while(rest):
            temp=mylist.pop(0)
            # return temp
            if temp[1]:
                bisect.insort(mylist,[temp[0]+A,True])
            else:
                bisect.insort(mylist,[temp[0]+B,False])
            if (not mylist) or mylist[0][0]!=temp[0]:
                rest-=1
                
        return (temp[0]+count*lcm)%(10**9+7)
