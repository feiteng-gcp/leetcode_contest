# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        length = 0
        node = head
        while node is not None:
            length += 1
            node = node.next
        
        nodes_to_take = length // 2
        
        i = 0
        node = head
        while node is not None:
            if i == nodes_to_take:
                return node
            i += 1
            node = node.next
