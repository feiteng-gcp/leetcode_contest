# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        i, j = head, head
        while j.next and j.next.next:
            i = i.next
            j = j.next.next
        else:
            if j.next:
                i = i.next
        return i
