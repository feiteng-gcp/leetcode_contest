# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        cnt = 1
        t = head
        while t.next != None:
            t 
= t.next
            cnt += 1
        i = 0
        t = head
        while i < cnt //2:
            t = t.next
            i += 1
        return t
