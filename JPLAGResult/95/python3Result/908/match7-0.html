# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        cur = head
        tot = 1
        while cur:
            tot += 1
            cur = cur.next
        mid = (tot+1)//2 - 1
        cur = head
        while mid > 0:
            cur = cur.next
            mid -=1
        return cur
