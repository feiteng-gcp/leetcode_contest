# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        if head is None:
            return None
        if head.next is None:
            return head
        
        cnt = 0
        node = head
        while node is not None:
            cnt += 1
            node = node.next
            
        target = cnt // 2 + 1
        node = head
        for _ in range(target - 1):
            node = node.next
        # node.next = None
        return node
