# Definition for singly-linked list.


class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        p = head
        cnt=0
        while p:
            p=p.next
            cnt+=1
            
        t=cnt//2
        p = head
        while t:
            t-=1
            p=p.next
            
        return p
