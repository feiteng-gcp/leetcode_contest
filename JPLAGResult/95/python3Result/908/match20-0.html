# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        sptr = head
        lptr = head.next
        
        while lptr:
            lptr = lptr.next
            if lptr: lptr = lptr.next
            sptr = sptr.next
        return sptr
