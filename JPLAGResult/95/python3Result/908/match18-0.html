# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        p,k=head,0
        while p: p,k=p.next,k+1
        p=head
        for i in range(k//2): p=p.next
        return p
