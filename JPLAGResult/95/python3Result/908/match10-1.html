# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        cur = head
        if cur.next is None:
            return cur
        
        count 
= 0
        while cur is not None:
            count+=1
            cur = cur.next
        cur = head
        pos = 1
        stop_index = (count+2)//2
        while pos <= count:
            if stop_index == pos:
                return cur
            pos +=1
            cur = cur.next
