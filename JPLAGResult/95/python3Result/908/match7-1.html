# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """

        l = 0
        t = head
        while t:
            l += 1
            t = t.next
        t = head
        cnt = l//2
        while cnt:
            t = t.next
            cnt -= 1
        return t
