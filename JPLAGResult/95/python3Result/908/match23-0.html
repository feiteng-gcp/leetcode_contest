# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        if head is None:
            return None
        if head.next is None:
            return head
        length = 1
        tmp = head
        while tmp.next is not None:
            tmp = tmp.next
            length += 1
        mid = length // 2
        tmp = head
        for _ in range(mid):
            tmp = tmp.next
        return tmp
