# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        
        i=0
        node=head
        while node:
            i=i+1
            node=node.next
            
        m=i//2+1
        #print(i,m)
        node=head
        for i in range(m-1):
            node=node.next
        return node
