# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        dd 
= head
        d = head
        while dd is not None and dd.next is not None:
            dd = dd.next.next
            d = d.next
        return d
