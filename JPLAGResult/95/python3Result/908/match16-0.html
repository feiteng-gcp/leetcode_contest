# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        l = 0
        n = head
        while n:
            n = n.next
            l += 1
        cnt = l // 2 + 1
        n = head
        while cnt > 1:
            n = n.next
            cnt += -1
        return n
