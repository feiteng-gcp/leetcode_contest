# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def middleNode(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        n = 1
        curr = head 
        while curr.next:
            n += 1
            curr = curr.next 
        steps = n // 2
        curr = head 
        for _ in range(steps):
            curr = curr.next
        return curr
