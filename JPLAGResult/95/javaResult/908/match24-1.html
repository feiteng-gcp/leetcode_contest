/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

class Solution {
    public ListNode middleNode(ListNode head) {
        ListNode lo = head;
        ListNode hi = head;
        while(hi != null && hi.next != null){
            lo = lo.next;
            hi = hi.next;
            hi = hi.next;
        }
        return lo;
    }
}
