/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

class Solution {
    public ListNode middleNode(ListNode head) {
        ListNode head1 = head; 
        ListNode head2 = head;
        while (head1 != null && head1.next != null) {
            head1 = head1.next.next;
            head2 = head2.next;
        }
        return head2;
    }
}
