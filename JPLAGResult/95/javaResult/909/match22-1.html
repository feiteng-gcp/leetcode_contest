class Solution {
    public boolean stoneGame(int[] piles) {
        int n = piles.length;
        
        int dp[][] = new int[n+1][n+1];
        int prefix[] = new int[n];
        
        
        
        
for(int i = 0 ; i<n ; i++)
            dp[i][i] = piles[i];
        
        for (int gap = 0; gap < n; ++gap)
        {
            for (int i = 0, j = gap; j < n; ++i, ++j)
            {
                int x1 = ((i + 2) <= j) ? dp[i + 2][j] : 0;
                int y1 = ((i + 1) <= (j - 1)) ? dp[i +1 ][j -  1] : 0;
                int z1 = (i <= (j - 2)) ? dp[i][j - 2]: 0;
     
                dp[i][j] = Math.max(piles[i] +
                              Math.min(x1, y1), piles[j] +
                              Math.min(y1, z1));
            }
        }     
        
        return (prefix[n-1]-dp[0][n-1]<dp[0][n-1] ? true : false);
    }
}
