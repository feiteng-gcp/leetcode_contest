class Solution {
    int[][] f = new int[501][501];
    public boolean stoneGame(int[] piles) {
        int n = piles.length;
        for(int len = 1 ; len <= n ; len++){
            for(int i = 0 ; i < n ; i++){
                int j = i + len - 1;
                if(j == n) break;
                if(i == j){
                    f[i][j] = piles[i];
                }else{
                    f[i][j] = Math.max(piles[i] - f[i+1][j] , piles[j] - f[i][j-1]);
                }
            }
        }
        return f[0][n-1] >= 0;
    }
}
