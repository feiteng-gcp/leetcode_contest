class Solution {
    public boolean stoneGame(int[] piles) {
        int l = piles.length;
        int[][] dp = new int[l][l];
        for (int i = 0; i < l; i++) dp[i][i] = piles[i]
;
        for (int len = 1; len < l; len++) {
            for (int i = 0; i < l-len; i++) {
                int j = i + len;
                dp[i][j] = Math.max(piles[i] - dp[i+1][j], piles[j] - dp[i][j-1]);
            }
        }
        return dp[0][l-1] > 0;
    }
}
