class Solution {
   public boolean stoneGame(int[] piles) {
    int sumA = 0;
    int sumB = 0;
    int b = 0, e = piles.length - 1;
    while (b < e) {
      int i = b;
      int j = e;
      while (i < j && piles[i] == piles[j]) {
        i++;
        j--;
      }
      
if (piles[i] <= piles[j]) {
        sumA += piles[e--];
        sumB += piles[b++];
      } else {
        sumA += piles[b++];
        sumB += piles[e--];
      }
    }
    return sumA > sumB;
  }
}
