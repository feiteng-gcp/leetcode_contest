class Solution {
    public boolean stoneGame(int[] piles) {
        int len = piles.length;
        int[][] dp = new int[len+1][len+1];
        int[] psum = new int[len+1];
        int sum = 0;
        for (int i = 0; i < len; i++) {
            sum += piles[i];
            psum[i] = sum;
        }
        for (int i = 0; i < len; i++) dp[i][i] = piles[i];
        for (int i = 1; i < len; i++) {
            for (int j = 0; j < len; j++) {
                if (j + i >= len) continue;                
                int sum1 = psum[j+i-1] -(j > 0 ? psum[j-1] : 0);
                int sum2 = psum[j+i] - psum[j];
                dp[j][j+i] = Math.max(sum1 - dp[j][j+i - 1] + piles[j+i]
                                      , sum2 - dp[j+1][j+i] + piles[j]);
            }
        }
        int lee = psum[len-1] - dp[0][len-1];
        return dp[0][len-1] > lee;
    }
}
