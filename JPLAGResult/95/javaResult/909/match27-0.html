class Solution {
    public boolean stoneGame(int[] piles) {
        int i = 0, j = piles.length - 1;
        int x = 0;
        int[] count = new int[2];
        while (i < j) {
            if (piles[i] >= piles[j]) {
                count[x] += piles[i];
                i++;
            } else {
                count[x] += piles[j];
                j--;
            }
            x %= 2;
        }
        return count[0] > count[1];
    }
}
