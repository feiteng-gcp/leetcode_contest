class Solution {
    public boolean stoneGame(int[] piles) {
        int n = piles.length;
        int[][] sums = new int[n][n];
        for (int i = 0; i < n; i++) {
            sums[i][i] = piles[i];
            for (int j = i+1; j < n; j++) {
                sums[i][j] = sums[i][j-1]+piles[j];
            }
        }
        int[][] dp = new int[n][n];
        for (int i = 0; i < n; i++)
            dp[i][i] = piles[i];
        for (int i = 1; i < n; i++) {
            for (int j = 0; j+i<n; j++) {
                int k = i+j;
                dp[j][k] = sums[j][k]-Math.min(dp[j+1][k],dp[j][k-1]);
            }
        }
        return dp[0][n-1]>sums[0][n-1]-dp[0][n-1];
    }
}
