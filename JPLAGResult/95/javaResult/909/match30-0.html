class Solution {
    public boolean stoneGame(int[] piles) {
        int n = piles.length;
        
        int[][] memo = new int[n][n];
        
        return helper(piles, 0, n - 1, memo) > 0;
    }
    
    int helper(int[] piles, int i, int j, int[][] memo) {
        if (i == j) return piles[i];
        
        if (memo[i][j] != 0) return memo[i][j];
        
        memo[i][j] = Math.max(-helper(piles, i, j - 1, memo) + piles[j], -helper(piles, i + 1, j, memo) + piles[i]);
        return memo[i][j];
    }
}
