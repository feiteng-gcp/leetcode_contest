class Solution {
    public boolean stoneGame(int[] piles) {
        int n=piles.length;
        int[][] f=new int[n][n];
        for (int i=0;i<n;i++) f[i][i]=piles[i];
        for (int l=2;l<=n;l++)
            for (int i=0;i<=n-l;i++)
            {
                int j=i+l-1;
                f[i][j]=Math.max(-f[i][j-1]+piles[j],-f[i+1][j]+piles[i]);
            }
        return f[0][n-1]>0;
    }
}
