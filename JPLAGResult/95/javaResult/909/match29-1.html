class Solution {
    public boolean stoneGame(int[] piles) {
        int[] sums = new int[piles.length + 1];
        for (int i = 0; i < piles.length; i++) {
            sums[i + 1] = sums[i] + piles[i];
        }
        
int n = piles.length;
        int[][] dp = new int[n][n];
        return helper(piles, 0, piles.length - 1, dp) * 2 > sums[n];
    }
    
    public int helper(int[] piles, int l, int r, int[][] dp) {
        if (l == r) return piles[l];
        if (r - l == 1) return Math.max(piles[r],piles[l]);
        if (l > r) return -1;
        if (dp[l][r] > 0) return dp[l][r];
        int getL = piles[l] + Math.min(helper(piles, l + 1, r - 1, dp), helper(piles, l + 2, r, dp));
        int getR = piles[r] + Math.min(helper(piles, l + 1, r - 1, dp), helper(piles, l, r - 2, dp));
        dp[l][r] = Math.max(getL, getR);
        return dp[l][r];
    }
}
