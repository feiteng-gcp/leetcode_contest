class Solution {
    public boolean stoneGame(int[] piles) {
        int n = piles.length;
        int[][] M = new int[n][n];
    
        int sum = 0;
        for(int p : piles){
            sum += p;
        }
    
        for(int l = 0; l < n; l++){        
        //for(int i = 0; i < n; i++){
           
            for(int i = 0; i + l < n; i++){
            //for(int l = 0; l < n - i; l++){
                int j = i + l;
                if(l == 0){
                    M[i][j] = piles[i];
                } else if(l == 1){
                    M[i][j] = Math.max(piles[i], piles[i+1]);
                } else {
                    // Take i
                    M[i][j] = Math.max(M[i][j], piles[i] + Math.min(M[i+2][j], M[i+1][j-1]));
                    // Take j
                    M[i][j] = Math.max(M[i][j], piles[j] + Math.min(M[i+1][j-1], M[i][j-2]));

                }            
                //System.out.println("M["+  i + "]["+ j + "] is " + M[i][j]);
            }
                    //System.out.println("M[0]["+ (n-1) + "] is " + M[0][n-1]);
        }
        //System.out.println("M[0]["+ (n-1) + "] is " + M[0][n-1]);
        return M[0][n-1] > sum/2;
    }
}
