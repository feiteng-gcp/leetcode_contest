class Solution {
    int [][][] memo;
    public boolean stoneGame(int[] piles) {
        memo = new int[piles.length + 1][piles.length + 1][2];
        for(int [][] arr : memo){
            for(int [] bar : arr){
                Arrays.fill(bar, -1);
            }
        }
        
return helper(piles, 0, piles.length - 1, 1) >= 0;
    }
    
    public int helper(int [] piles, int l, int r, int player){
        if(r < l)
            return 0;
        if(memo[l][r][player] != -1)
            return memo[l][r][player];
        int ans = 0;
        if(player == 1){
            ans = Math.max(piles[l] + helper(piles, l + 1, r, 0), piles[r] + helper(piles, l, r - 1, 0));
        }  else {
            ans = Math.min(-piles[l] + helper(piles, l + 1, r, 1), -piles[r] + helper(piles, l, r - 1, 1)); 
        }
        memo[l][r][player] = ans;
        return ans;
    }
}
