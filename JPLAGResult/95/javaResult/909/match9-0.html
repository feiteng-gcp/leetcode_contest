class Solution {
    public boolean stoneGame(int[] piles) {
        int n = piles.length;
        int[] sum = new int[n+1];
        for (int i=1; i<=n; i++) {
            sum[i] = sum[i-1] + piles[i-1];
        }
        int[][] get = new int[n+1][n+1];
        for (int i=n-1; i>=0; i--) {
            for (int j=i+1; j<=n; j++) {
                get[i][j] = Math.max(piles[i] + (sum[j]-sum[i+1]) - get[i+1][j],
                                    piles[j-1] + (sum[j-1]-sum[i]) - get[i][j-1]);
            }
        }
//        System.out.println(Arrays.toString(get[0]));
        
        return get[0][n] > sum[n]-get[0][n];
    }
}
