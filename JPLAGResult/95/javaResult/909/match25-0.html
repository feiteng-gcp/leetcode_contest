class Solution {
    public boolean stoneGame(int[] piles) {
        int[][] dp = new int[piles.length][piles.length];
        int sum = 0;
        for (int i = 0; i < piles.length; i ++) {
            dp[i][i] = piles[i];
            if (i != piles.length - 1) {
                dp[i][i + 1] = Math.max(piles[i], piles[i + 1]);    
            }
            sum += piles[i];
            
        }
        for (int len = 4; len <= piles.length; len += 2) {
            for (int i = 0; i + len - 1 < piles.length; i ++) {
                dp[i][i + len - 1] = Math.max(dp[i + 2][i + len - 1] + piles[i], dp[i + 1][i + len - 2] + piles[i]);
                dp[i][i + len - 1] = Math.max(dp[i][i + len - 1], dp[i][i + len - 3] + piles[i + len -1]);
                dp[i][i + len - 1] = Math.max(dp[i][i + len - 1], dp[i + 1][i + len - 2] + piles[i + len - 1]);
            }
        }
        return dp[0][piles.length - 1] > sum / 2;  
    }
}
