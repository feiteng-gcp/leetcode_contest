class Solution {
    public boolean stoneGame(int[] nums) {
        
        int n = nums.length;
        int[][] dp = new int[n][n];
        for (int i = 0; i < n; i++) {
            dp[i][i] = nums[i];
        }
        
        for (int i = 1; i < n; i++) {
            for (int l = 0; l < n - i; l++) {
                int r = l + i;
                dp[l][r] = Math.max(nums[l] - dp[l + 1][r], nums[r] - dp[l][r - 1]);
            }
        }
        
        return dp[0][n - 1] >= 0;
    }
}
