class Solution {
    public boolean stoneGame(int[] piles) {
        int[][] ar = new int[piles.length + 10][piles.length + 10];

        for (int i = 2; i <= piles.length; i += 2) {
            for (int j = i; j <= piles.length; j++) {
                int js = Math.min(piles[j - i] - piles[j - 1] + ar[i - 2][j - 1], piles[j - i] - piles[j - i + 1] + ar[i - 2][j]);
                int jo = Math.min(piles[j - 1] - piles[j - i] + ar[i - 2][j - 1], piles[j - 1] - piles[j - 2] + ar[i - 2][j - 2]);
                ar[i][j] = Math.max(js, jo);
            }
        }
        if(ar[piles.length][piles.length]>0){
            return true;
        }
        return false;
    }
}
