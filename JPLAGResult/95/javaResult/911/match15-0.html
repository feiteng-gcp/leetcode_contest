class Solution {
    
    private static long modulo = 1000000007L;
    
    public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        return _profitableSchemes(G+1, P, group, profit);
    }
    
    public int _profitableSchemes(int G, int P, int[] group, int[] profit) {
        //Considering only group[0..x-1] and profit[0..x-1],
        //  suppose for each profit value p in [0..P] (p is considered P if p>P)
        //  how many choice are there for each g value (consider g=G if g>G)? Call it dp[p][g]
        //This dp[][] array will change with x
        
        int n = group.length;
        int[][] dp = new int[P+1][G+1];
        for(int j=0;j<=P;j++) for(int k=0;k<=G;k++) {
            dp[j][k] = 0;
        }
        dp[0][0] = 1;
        
        for(int i=0;i<n;i++) {
            int[][] nextdp = new int[P+1][G+1];
            
            for(int j=0;j<=P;j++) for(int k=0;k<=G;k++) {
                nextdp[j][k] = dp[j][k];
            }
            
            for(int j=0;j<=P;j++) for(int k=0;k<=G;k++) {
                int nextj = j + profit[i], nextk = k + group[i];
                if(nextj > P) {
                    nextj = P;
                }
                if(nextk > G) {
                    nextk = G;
                }
                nextdp[nextj][nextk] = (int) (( nextdp[nextj][nextk] + dp[j][k] ) % modulo);
            }
            
            dp = nextdp;
        }
        
        long ret = 0L;
        for(int k=0;k<G;k++) {
            ret += dp[P][k];
            ret %= modulo;
        }
        
        return (int)(ret);
        
    }
}
