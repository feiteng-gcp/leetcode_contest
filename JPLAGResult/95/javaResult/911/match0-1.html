class Solution {
    
public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        int n = group.length;
        
        long[][][] memo = new long[n][G + 1][P + 1];
        
        memo[0][0][0] = 1;
        if (group[0] <= G) {
            memo[0][group[0]][profit[0] > P ? P : profit[0]] = 1;
        }
        
        
for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j <= G; j++) {
                for (int k = 0; k <= P; k++) {
                    memo[i + 1][j][k] = (memo[i][j][k] + memo[i + 1][j][k]) % 1000000007;
                    if (j + group[i + 1] <= G)  {
                        int p = k + profit[i  + 1] > P ? P : k + profit[i  + 1];
                        memo[i + 1][j + group[i + 1]][p] = (memo[i + 1][j + group[i + 1]][p] + memo[i][j][k]) % 1000000007;
                    }
                
}
            }
        }
        
        long res = 0;
        for (int i = 0; i <= G; i++) {
            res = (res + memo[n - 1][i][P])  % 1000000007;
        }
        return (int) res;
    }
}
