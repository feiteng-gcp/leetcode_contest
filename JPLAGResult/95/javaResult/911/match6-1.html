class Solution {
    public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        final long M = 1_000_000_007;
        
int len = profit.length;
        long[][][] dp = new long[101][101][len+1];
        dp[group[0]][profit[0]][0] = 1;
        for (int i = 1; i < len; i++) {
            for (int g = 1; g <= G; g++)
                for (int p = 0; p <= 100; p++) {
                    dp[g][p][i] += dp[g][p][i-1];
                    dp[g][p][i] %= M;
                    int ng = g + group[i];
                    if (ng > G) continue;
                    int np = p + profit[i];
                    if (np > 100) np = 100;
                    dp[ng][np][i] += dp[g][p][i-1];
                    dp[ng][np][i] %= M;
                }
            int g = group[i], p = profit[i];
            dp[g][p][i] += 1;
            dp[g][p][i] %= M;
        }
        long total = 0;
        for (int g = 0; g <= G; g++) 
            for (int p = P; p <= 100; p++) {
                total += dp[g][p][len-1];
                total %= M;
        }
        return (int)total;
    }
}
