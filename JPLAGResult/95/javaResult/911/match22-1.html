class Solution {
    public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        int n = group.length;
        
long[][][] dp = new long[n + 1][G + 1][P + 1];
        dp[0][0][0] = 1;
        for(int i = 1; i < n + 1; i++){
            for(int j = 0; j <= G; j++){
                for(int k = 0; k <= P; k++){
                    for(int p = 0; p < i; p++){
                        if(j + group[i - 1] <= G){
                            dp[i][j + group[i - 1]][Math.min(k + profit[i - 1], P)] += dp[p][j][k];
                            dp[i][j + group[i - 1]][Math.min(k + profit[i - 1], P)] %= 1000000007L;
                        }
                    }
                
}
            }
        }
        long res = 0;
        for(int i = 0; i <= G; i++){
            for(int j = 1; j <= n; j++){
                res += dp[j][i][P];
                res %= 1000000007L;
            }
        }
        return (int)res;
    }
}
