class Solution {
    public int profitableSchemes(int G, int P, int[] g, int[] p) {
        int l = p.length;
         int s = 0;
        for(int pp:p){s+=pp;}
        int dp[][] = new int[s+1][G+1];
        
       
        
        
      //  dp[i][profit][man] =  dp[i-1][profit][man] + dp[i-1][profit-p[i]][man-g[i]];
        long r = 0;
        dp[0][0] = 1 ;
        if(G>=g[0]){
            dp[p[0]][g[0]] = 1;
            // if(p[0]>=P){
            //     r++;
            // }
        }
                
        
for(int i=1;i<l;i++){
            int ndp[][] = new int[s+1][G+1];
            for(int j=0;j<=s;j++){
                for(int k=0;k<=G;k++){
                    if(i>=1){
                        ndp[j][k] = dp[j][k] ;
                        if(j-p[i]>=0&&k-g[i]>=0){
                            ndp[j][k] = (ndp[j][k]+dp[j-p[i]][k-g[i]])%1000000007 ;
                        }
                    }
                    if(i==l-1&&j>=P){
                        r = (r+ndp[j][k])%1000000007;
                    
}
                    
                }
            }
            dp = ndp;
        }

        if(g.length==1){
        
        for(int i=P;i<=s;i++){
            for(int k=0;k<=G;k++){
                r = (r+dp[i][k])%1000000007;
            }
        }
        }
        
        
       
        return (int)r;
        
        
    }
}
