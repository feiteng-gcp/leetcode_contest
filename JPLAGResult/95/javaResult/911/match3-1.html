class Solution {
    public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        int[][] dp = new int[P + 1][G + 1];
        int mod = 1000000000 + 7;
        
dp[0][0] = 1;
        for (int i = 0; i < group.length; i ++) {
            for (int j = G - group[i]; j >= 0; j --) {
                for (int k = 0; k <= P; k ++) {
                    if (dp[k][j] > 0) {
                        if (k + profit[i] >= P) {
                            dp[P][j + group[i]] = (dp[k][j] + dp[P][j + group[i]]) % mod;
                        } else {
                            dp[k + profit[i]][j + group[i]] = (dp[k][j] + dp[k + profit[i]][j + group[i]]) % mod;
                        }
                        
                    }
                
}
            }
        }
        int ans = 0;
        for (int i = 0; i <= G; i ++) {
            System.out.println(dp[P][i]);
            ans = (ans + dp[P][i]) % mod;
        }
        return ans;
    }
}
