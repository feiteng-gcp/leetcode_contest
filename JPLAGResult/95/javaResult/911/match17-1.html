class Solution {
    //dp[i][g][p] for the first i groups, with g members, p profits, how many possibilities
    // dp[i+1][g][p] = dp[i][g][p] + dp[i][g-group[i]][p-profit[i]]
    
    
    public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        int n = group.length;
        long module = 10*10*10*10*10*10*10*10*10 + 7;
        if (n == 0) return 0;
        int maxProfit = 0;
        for (int p : profit) maxProfit += p;
        long[][] dp = new long[G + 1][maxProfit + 1];
        dp[0][0] = 1;
        
for (int i = 1; i <= n; ++i) {
            for (int j = G; j >= 0; --j) {
                for (int k = maxProfit; k >= 0; --k) {
                    if (j - group[i - 1] >= 0 && k - profit[i - 1] >= 0)
                        dp[j][k] += dp[j - group[i - 1]][k - profit[i - 1]];
                    if (dp[j][k] > module) dp[j][k] -= module
;
                }
            }
        }
        
        long ret = 0;
        
        for (int j = 0; j <= G; ++j) {
            for (int k = P; k <= maxProfit; ++k) {
                ret += dp[j][k];
                if (ret > module) ret -= module;
            }
        }
        //System.out.println(ret);
        return (int) (ret % module);
    }
}
