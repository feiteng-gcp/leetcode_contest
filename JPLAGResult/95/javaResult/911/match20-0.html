class Solution {
    private int MOD = 1000000007;
    public int profitableSchemes(int G, int P, int[] group, int[] profits) {
//         long dp[i][people][profit] = dp[i - 1][j - group[i]][profit - profits[i]] + dp[i - 1][group[i]][profit]
        int len = group.length;
        int profit = 0;
        for (int item : profits) {
            profit += item;
        }
        long dp[][][] = new long[2][G + 1][profit + 1];
        
        for (int i = 0; i < len + 1; i++) {
            for (int j = 0; j < G + 1; j++) {
                dp[i%2][j][0] = 1l;
            }
        }
        dp[0][0][0] = 1l;
        long result = 0l;
        
        for (int i = 1; i < len + 1; i++) {
            for (int j = 0; j < G + 1; j++) {
                for (int p = 0; p < profit + 1; p++) {
                    if (j >= group[i - 1] && p >= profits[i - 1]) {
                        dp[i%2][j][p] = (dp[(i - 1)%2][j - group[i - 1]][p - profits[i - 1]] + dp[(i - 1)%2][j][p]) % MOD;

                    } else {
                        dp[i%2][j][p] = dp[(i - 1)%2][j][p];
                    }
                    // System.out.print("i");
                    // System.out.print(i);
                    //     System.out.print("j");
                    // System.out.print(j);
                    //                     System.out.print("p");
                    // System.out.print(p);
                    // System.out.print("value");
                    // System.out.print(dp[i][j][p]);
                    // System.out.println();
                   
                }
            }
        }
        
        for (int i = P; i < profit + 1; i++) {
            result = (result  + dp[len%2][G][i] ) % MOD;
        }
        return (int) (result % MOD);
    }
}
