class Solution {
    public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        // N person with K possible crime .length
        long[][][] res=new long[group.length+1][G+1][101];
        /*
        for(int i=0;i<=G;i++){
            res[0][i][0]=1;
        }
        */
        res[0][0][0]=1;
        for(int i=1;i<=group.length;i++){
            /*for(int j=1;j<=G;j++){
                if(j-group[i-1]<=0){
                    continue;
                }*/
                long[][] last=res[i-1];
            long[][] curr=res[i];
                for(int j=0;j<=G;j++){
                    for(int p=0;p<=100;p++){
                        curr[j][p]=last[j][p];
                    }
                }
                
                
                for(int people=0;people<=G&&people+group[i-1]<=G;people++){
                    for(int pro=0;pro<=100;pro++){
                        int newProfit=Math.min(100,pro+profit[i-1]);
                        curr[people+group[i-1]][newProfit]=curr[people+group[i-1]][newProfit]+last[people][pro];
                        
curr[people+group[i-1]][newProfit]=curr[people+group[i-1]][newProfit]%1000000007;
                    }
                }
            //}
        }
        long count=0;
        for(int h=0;h<=G;h++){
            for(int p=P;p<=100;p++){
                
                count=(count+res[group.length][h][p])%1000000007;
            }
        }
        return (int)count;
    }
}
