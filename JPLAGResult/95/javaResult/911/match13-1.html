class Solution {
    public int profitableSchemes(int G, int P, int[] group, int[] profit) {
        int MOD = 1000000007;
        int n = group.length;
        long[][] dp = new long[G+1][P+1];
        dp[0][0] = 1;
        for (int i = 0; i < n; i++) {
            int g = group[i];
            int p = profit[i];
            for (int j = G-g; j >= 0; --j) {
                for (int k = P; k >= 0 && k+p > P; --k) {
                    dp[j+g][P] += dp[j][k];
                    dp[j+g][P] %= MOD;
                }
                for (int k = P-p; k >= 0; --k) {
                    dp[j+g][k+p] += dp[j][k];
                    dp[j+g][k+p] %= MOD;
                }
            }
        }
        int ans = 0;
        for (int i = 0; i <= G; i++) {
            ans += dp[i][P];
            ans %= MOD;
        }
        return ans;
    }
}
