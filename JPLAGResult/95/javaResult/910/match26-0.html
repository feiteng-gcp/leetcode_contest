class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        int C = common(A, B);
        int D = A * B / C;
        int E = D / A + D / B - 1;
        int res = 0;
        int X = (int) (Math.pow(10, 9) + 7);

        int n = Integer.MAX_VALUE / D;
        while (N >= n * E) {
            int T = (D % X * n) % X;
            res = (res + T) % X;
            N -= n * E;
        }
        while (N >= E) {
            res = (res + D % X) % X;
            N -= E;
        }

        int tmp = res;
        int i = 1, j = 1;
        while (N > 0) {
            if (A * i < B * j) {
                res = (tmp + A * i) % X;
                i++;
            } else {
                res = (tmp + B * j) % X;
                j++;
            }
            N--;
        }
        return res;
    }


    private int common(int x, int y) {

        int m = x, n = y;
        while (y != 0) {
            int z = x % y;
            x = y;
            y = z;
        }
        return x;

    }
}
