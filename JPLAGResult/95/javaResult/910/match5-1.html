class Solution {
    
    
long gcd(long a, long b)
    {
        if (a == 0)
            return b;
        return gcd(b % a, a);
    }
    
    
public int nthMagicalNumber(int n, int a, int b) {
        long h,l,mid=0,c,lcm;
        l = 1;
        h = (long)Math.pow(10,18);
        
lcm = (a * b) / gcd(a, b);

        while(l <= h)
        {
            mid = (l + h) / 2;
            c = (mid / a) + (mid / b) - (mid / lcm);

            if(c < n)
                l = mid + 1;
            if(c > n)
                h = mid - 1;
            if(c == n)
                break;
        }
        while(mid % a !=0&& mid % b!=0)
            mid--;
        return (int)(mid%1000000007);
    }
}
