class Solution {
    
    private static long modulo = 1000000007L;
    
    public int nthMagicalNumber(int N, int A, int B) {
        return magicalNumber(N, A, B);
    }
    
    private int magicalNumber(long N, long A, long B) {
        //This N value is 0-based
        //If N==0 return 0; if N==1, return min(A, B)
        if (N==0L) {
            return 0;
        }
        long lcm = lcm(A,B);
        long magicalNumbersInPeriod = numberOfMagicalNumbersBelow(lcm, A, B, lcm);
        long numberOfPeriods = (N - 1) / magicalNumbersInPeriod;
        N = ( (N-1) % magicalNumbersInPeriod ) + 1;
        long left = 0, right = lcm + 1;
        while(right - left > 1) {
            long middle = (right + left) / 2;
            if (numberOfMagicalNumbersBelow(middle, A, B, lcm) > N) {
                right = middle;
            } else {
                left = middle;
            }
        }
        
        //now want left + lcm * numberOfPeriods;
        
        // System.out.println(String.format("%s %s %s", left, lcm, numberOfPeriods));
        
        lcm %= modulo;
        numberOfPeriods %= modulo;
        
        long ret = lcm * numberOfPeriods + left;
        ret %= modulo;
        return (int)(ret);
        
    }
    
    private long numberOfMagicalNumbersBelow(long n, long a, long b, long lcm) {
        //count how many numbers in [0..n-1] are divisible by either a or b, assume lcm is lcm(a,b)
        return countDivisible(n, a) + countDivisible(n, b) - countDivisible(n, lcm);
    }
    
    private long countDivisible(long n, long a) {
        //return # of values in [0..n-1] that are divisible by a
        if(n==0) {
            return 0;
        } else {
            return (n-1)/a + 1;
        }
    }
    
    private long lcm(long x, long y) {
        return x / gcd(x,y) * y;
    }
    
    private long gcd(long x, long y) {
        while(y != 0) {
            long z = x % y;
            x = y;
            y = z;
        }
        return x;
    }
}
