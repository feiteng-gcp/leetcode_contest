class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        int gcd = getGcd(A, B);
        int max = A * B / gcd;
        int group = max / A + max / B - 1;
        long numGroup = N / group;
        long left = N % group;
        System.out.println(numGroup);
        System.out.println(left);
        if(left == 0)
            return (int)((numGroup * max) % 1000000007);
        long[] num = new long[2];
        num[0] = numGroup * max / A + 1;
        num[1] = numGroup * max / B + 1;
        int count = 0;
        long result = 0;
        while(count < left) {
            long n1 = A * num[0];
            long n2 = B * num[1];
            // System.out.println(n1);System.out.println(n2);
            
            if(n1 < n2) {
                if(n1 % B == 0 && n1 / B < num[1])
                    num[0]++;
                else 
{
                    result = n1;
                    count++;
                    num[0]++;
                }
                
            } else if(n2 < n1) {
                if(n2 % A == 0 && n2 / A < num[0])
                    num[1]++;
                else {
                    result = n2;
                    count++;
                    num[1]++;
                }
            } else {
                num[0]++;
                count++;
                result = n1;
            }
            // System.out.println(result);
            // System.out.println(count);
        
}
        return (int)(result % 1000000007);
    }
    
    public int getGcd(int A, int B) {
        if(B == 0)
            return A;
        return getGcd(B, A % B);
    }
}
