class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        long left = 1l;
        long right = 1l << 60;
        long mid = 0;
        long lcmVal = lcm(A, B);
        
        long ans = right;
        
        while(left <= right){
            mid = (1 + left + right) >> 1;
            long counter = (mid/((long) A)) + (mid/((long) B)) - (mid/lcmVal);
            if(counter >= N){
                ans = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            
}
        }
        return (int) (ans%1000000007l);
    }
    
    public long lcm(long A, long B){
        return A * (B/gcd(A, B));
    }
    
    public long gcd(long A, long B){
        while(B > 0){
            long cur = B;
            B = A % B;
            A = cur;
        }
        return A;
    }
}
