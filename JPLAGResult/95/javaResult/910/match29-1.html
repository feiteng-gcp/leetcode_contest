class Solution {
   public int nthMagicalNumber(int N, int A, int B) {
        long mod = 1000000007;
        if(A > B) {int t = A; A = B; B = t;}
        int gc = gcd(A, B);
        //System.out.println("A :" + A + "  B : " + B + " GC : " + gc);
        long mc = A * B / gc;
        long period = mc / A + mc / B - 1;
        
long nPeriods = N / period;
        long left = N % period;
        long na = 1, nb = 1, count = 0, add = 0;
        while(count < left){
            if(na * A <= nb * B) {
                add = na * A;
                na++;
            }else {
                add = nb * B;
                nb++;
            }
            count++;
        }
        
return (int)((nPeriods * mc + add) % mod);
    }
    
    private int gcd(int a, int b){
        if(b == 0) return a;
        return gcd(b, a % b);
    }
}
