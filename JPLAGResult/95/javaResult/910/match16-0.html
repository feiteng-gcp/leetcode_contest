class Solution {
    int mo=1000000007;
    int gcd(int A,int B)
    {
        if (A>B) return gcd(B,A);
        while (A>0)
        {
            int C=B%A;
            B=A;
            A=C;
        }
        return B;
    }
    int lcm(int A,int B)
    {
        return A/gcd(A,B)*B;
    }
    long pre(long x,int A,int B,int C)
    {
        return x/A+x/B-x/C;
    }
    public int nthMagicalNumber(int N, int A, int B) {
        int C=lcm(A,B);
        long l=1,r=(long)1000000*(long)1000000000;
        while (l+1<r)
        {
            long mid=(l+r)>>1;
            if (pre(mid,A,B,C)<N) l=mid; else r=mid;
        }
        return (int)(r%mo);
    }
}
