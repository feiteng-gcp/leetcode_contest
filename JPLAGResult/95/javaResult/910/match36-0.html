class Solution {
    final int mod = (int) 1e9 + 7;
    public int nthMagicalNumber(int N, int A, int B) {
        if (A > B) {
            return nthMagicalNumber(N, B, A);
        }
        int gcd = gcd(A, B);
        long min = A / gcd * B;
        //System.out.println(min);
        long range1 = min / A;
        long range2 = min / B;
        long range = range1 + range2 - 1L;

        long t = N / range;
        long t1 = N % range;
        //System.out.println(t + "," + t1);
        long x = -1;
        if (t != 0 ) {
            x = helper(t, min);
            //System.out.println(x);
            if (t1 == 0) {
                return (int)x;
            }
        }


        long can = 1;
        for (int i = 1, j = 1; i <= range1 && j <= range1; ) {
            long c1 = A * i;
            long c2 = B * j;
            if (c1 < c2) {
                can = c1;
                i++;
            } else {
                can = c2;
                j++;
            }
            t1--;
            if (t1 == 0) {
                break;
            }
        }
        return (int)(((x == -1 ? 0 : x) + can) % mod);

    }

    private long helper(long t, long min) {
        if (t == 1) {
            return min;
        }
        //System.out.println(t);
        long x = helper(t / 2, min);
        if (t % 2 == 0) {
            return (x * 2) % mod;
        }
        return (x * 2 + min) % mod;
    }

    private int gcd(int A, int B) {
        if (B == 0) {
            return A;
        }
        return gcd(B, A % B);
    }
}
