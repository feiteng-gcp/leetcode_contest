class Solution {
    int gcd(int a, int b) {
        if (a < b) {
            int sp = a; a = b; b = sp;
        }
        
while (b!=0) {
            int r = a % b;
            a = b;
            b = r;
        }
        return a;
    }
    
    static final int mod = 1000000007;
    
    int findnth(int N, int A, int B) {
        System.out.println(N + " " + A + " " + B);
        
        if (N == 0) {
            return 0;
        }
        if (N == A + B - 1) {
            return A * B;
        }
        int i = 1;
        int j = A * B;
        while (i<=j) {
            int mid = (i+j)/2;
            int count = (mid/A + mid/B);
            if (count == N) {
                return mid - Math.min(mid % A, mid % B);
            }
            
if (count < N) {
                i = mid + 1;
            } else {
                j = mid - 1;
            }
        }
        return -1; //  not reachable
    }
    
    long Mod(long x) {
        return x % mod;
    }
    
    
public int nthMagicalNumber(int N, int A, int B) {
        int d = gcd(A, B);
        
//        System.out.println(d);
        
        A /= d; 
        B /= d;
        int c = A + B - 1;
        long ans = Mod(d * Mod((long) N / c * A * B + findnth(N % c, A, B)));
        return (int) ans;
    }
}
