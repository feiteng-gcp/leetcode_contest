class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        long lo = 0, hi = Long.MAX_VALUE, res = -1;
        while(lo <= hi){
            long mid = lo + hi >>> 1;
            long c;
            if(Math.max(A, B) % Math.min(A, B) == 0)
                c = mid / Math.min(A, B);
            else
                c = (mid / A) + (mid / B) - (mid / (A * B / gcd(A, B)));
            if(c >= N){
                hi = mid - 1;
                res = mid;
            }else{
                lo = mid + 1;
            }
        }
        return (int)(res % 1000000007L);
    }
    
    static long gcd(long a, long b){return a == 0 ? b : gcd(b % a, a);}
}
