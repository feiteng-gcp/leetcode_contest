class Solution {

	public int nthMagicalNumber(int N, int A, int B) {
		if (A < B) {
			int sum = A + B;
			A = sum - A;
			B = sum - A;
		}
		int gcd = gcd(A, B);
		long lf = 0;
		long rt = Long.MAX_VALUE / 2;
		while (lf < rt) {
			long mid = lf + (rt - lf + 1) / 2;
			if (valid(mid, A, B, N, gcd)) {
				rt = mid - 1;
			} else {
				lf = mid;
			}
		}
		for (long i = lf; i >= 0; --i) {
			if (i % A == 0 || i % B == 0)
				return (int) (i % (1000000000 + 7));
		}
		return -1;
	}

	int gcd(int m, int n) {
		while (m > 0) {
			int c = n % m;
			n = m;
			m = c;
		}
		return n;

	}

	public boolean valid(long mid, int A, int B, int N, int gcd) {
		long sum = 0;
		if (A % B == 0) {
			sum += mid / B;
		} else {
			sum -= mid / (A / gcd * B / gcd * gcd);
			sum += mid / A;
			sum += mid / B;
		}
		return sum > N;
	}
}
