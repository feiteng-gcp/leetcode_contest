class Solution {
    public int nthMagicalNumber(int n, int a, int b) {
        int mod = 1_000_000_007;
        long high = Long.MAX_VALUE/2, low = 1, mid = 0, cnt;
        int lcm = (a * b) / gcd(a, b);
        while (low <= high) {
            mid = (low + high) / 2;
            cnt = (mid / a) + (mid / b) - (mid / lcm);
            if (cnt < n)
                low = mid + 1;
            if (cnt > n)
                high = mid - 1;
            if (cnt == n)
                break;
        }
        while (mid % a != 0 && mid % b != 0)
            mid--
;
        return (int) (mid % mod);
    }

    int gcd(int a, int b) {
        if (a == 0)
            return b;
        return gcd(b % a, a);
    }
}
