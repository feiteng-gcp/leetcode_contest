	class Solution {
	    public int nthMagicalNumber(int K, int A, int B) {
	        long low = 0, high = 2000000000000000000L;
	        int G = A*B/gcd(A, B);
	        while(high - low > 1){
	        	long h = high+low>>1;
	        	long num = h/A+h/B-h/G;
	        	if(num >= K){

	        		high = h;
	        	}else{
	        		low = h;
	        	}
	        }
	        return (int)(high%1000000007);
	    }
	    
		public int gcd(int a, int b) {
			while (b > 0) {

				int c = a;
				a = b;
				b = c % b;
			}
			return a;
		}
	}
