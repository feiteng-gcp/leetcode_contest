class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        int a = A, b = B, n = N;
        long high = 0L, low = 0L, mid = 0L, cnt = 0L, lcm = 0L;
        low = 1;
        high = 1000000000000000000L;
        lcm = (a * b) / gcd(a, b);

        while(low <= high)
        {
            mid = (low + high) / 2;
            cnt = (mid / a) + (mid / b) - (mid / lcm);

            if(cnt < n)
                low = mid + 1;
            if(cnt > n)
                high = mid - 1;
            if(cnt == n)
                break;
        }
        while((mid % a != 0) && (mid % b != 0))
            mid--
;
        return (int)(mid % 1000000007); 
    }
    
    private long gcd(int a, int b) {
        if (a == 0) return b;
        return gcd(b % a, a);
    }
}
