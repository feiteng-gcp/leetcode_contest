class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        int gcd = getGcd(A, B);
        int lcm = A * B / gcd;
        int numA = lcm / A, numB = lcm / B;
        int n = numA + numB - 1;        //one group
        int group = N / n, offset = N % n;
        int indexA = 1, indexB = 1;
        long res = 0;
        
while (offset > 0) {
            int a = A * indexA, b = B * indexB;
            if (a < b) {
                res = a;
                indexA++;
            } else {
                res = b;
                indexB++;
            
}
            offset--;
        }
        res += (long)(group) * lcm;
        return (int)(res % 1000000007);
    }
    public int getGcd(int A, int B) {
        if (A > B) {
            return getGcd(B, A);
        }
        if (B % A == 0) {
            return A;
        }
        return getGcd(A, B % A);
    }
}
