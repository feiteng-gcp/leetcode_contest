class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        long s = 1, e = 1000000000000000000L, m, count;
        while(s < e) {
            m = (s+e)/2;
            count = m/A + m/B - m/((long)A * B / gcd(A, B));
            if(count < N) s = m+1;
            else e = m;
        }
        return (int)(s%1000000007);
    }
    
    private long gcd(long A, long B) {
        if(B == 0) return A;
        else return gcd(B, A%B);
    }
}
