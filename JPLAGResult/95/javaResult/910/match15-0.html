class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        int common = A * B / getCommon(A,B);
        int n1 = common / A; int n2 = common / B;
        int x = N / (n1 + n2 - 1);
        int y = N % (n1 + n2  - 1);
        int x1 = (int) ((long)common * (long)x % (1e9 + 7));
        int ret = 0; 
        int ac = A;
        int bc = B;
        for (int i = 1; i <= y; i++) {
            if (ac > bc) {
                ret = bc;
                bc += B;
            } else {
                ret = ac;
                ac += A;
            }
        }
        // System.out.println(x1 + "   " + ret);
        return (int)((x1 + ret) % (1e9 + 7));
    }
    
    public int getCommon(int a, int b) {
        if (a < b) return getCommon(b, a);
        while (a % b != 0) {
            int temp = a % b;
            a = b;
            b = temp;
        }
        return b;
    }
}
