class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
        int lcm = A * B / gcd(A, B);
        long left = 1L;
        long right = 40000000000000L;
        while (left < right) {
            long mid = (right+left)/2;
            long count = mid/A + mid/B - mid/lcm;
            if (count == N) {
                if (mid % A ==0 || mid % B==0) {
                    return (int)(mid%(1e9+7));
                }
                else {
                    right = mid-1;
                }
            }
            else if (count > N) {
                right = mid - 1;
            }
            else {
                left = mid + 1;
            }
            System.out.println(left);
            System.out.println(right);
        }
        return (int)(left%(1e9+7));
    }
    
    private int gcd(int a, int b) {
        if (a == 0 || b == 0)
           return 0;
        if (a == b)
            return a;
        if (a > b)
            return gcd(a-b, b);
        return gcd(a, b-a);
    }
    
}
