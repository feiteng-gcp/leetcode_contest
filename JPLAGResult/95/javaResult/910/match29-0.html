class Solution {
	final int MOD = 1000000007;
	
    public int nthMagicalNumber(int N, int A, int B) {
        int g;
        if (A > B)
        	g = gcd(A, B);
        else
        	g = gcd(B, A);
        
        int lcmAB = A * B / g;
        
        
        if (A == g || B == g)
        {
        	int d;
        	if (A == g)
        		d = A;
        	else
        		d = B;
        	
        	long ans = d;
        	ans = ans * N;
        	ans = ans % MOD;
        	
        	return (int)ans;
        }
        int aCount = lcmAB / A;
        int bCount = lcmAB / B;
        
        
        int lcmSectionCount = aCount + bCount - 1;
        int sections = N / lcmSectionCount;
        long ans = sections;
        ans = ans * lcmAB;
        
        int rSection = N % lcmSectionCount;
        int pA, pB;
        long nA = ans, nB = ans;
        int p = 0;
        nA += A;
        nB += B;
        
        while (p < rSection)
        {
        	if (nA > nB)
        	{
        		ans = nB;
        		nB += B;
        	}
        	else
        	{
        		ans = nA;
        		nA += A;
        	}
        	p++;
        }
        
        ans = ans % MOD;
        return (int)ans;
    }

	private int gcd(int a, int b) {
		if (a % b == 0)
			return b;
		else
			return gcd(b, a % b);
	}
}
