class Solution {
    public int nthMagicalNumber(int N, int A, int B) {
		Long lcm = getLcm(Long.valueOf(A), Long.valueOf(B));
		Long ar = lcm / A;
		Long br = lcm / B;
		Long range = ar + br - 1;
		Long round = N / range;
		Long ai = ar * round;
		Long bi = br * round;
		Long i = round * range;
		Long currR = lcm * round;
		while (i < N) {
			Long ra = (ai + 1) * A;
			Long rb = (bi + 1) * B;
			if (ra < rb) {
				ai++;
				i++;
				currR = ra;
			} else if (ra > rb) {
				bi++;
				i++;
				currR = rb;
			} else {
				ai++;
				bi++;
				i++;
				currR = ra;
			}
		}
		System.out.println(currR);
		int res = (int) (currR % 1000000007);
		return res;
	}
	
	// 最大公约数
	public static Long getGcd(Long n1, Long n2) {
		Long gcd = 0L;
		if (n1 < n2) {// 交换n1、n2的值
			n1 = n1 + n2;
			n2 = n1 - n2;
			n1 = n1 - n2;
		}
		if (n1 % n2 == 0) {
			gcd = n2;
		}
		while (n1 % n2 > 0) {
			n1 = n1 % n2;
			if (n1 < n2) {
				n1 = n1 + n2;
				n2 = n1 - n2;
				n1 = n1 - n2;
			}
			if (n1 % n2 == 0) {
				gcd = n2;
			}
		}
		return gcd;
	}

	// 最小公倍数
	public static Long getLcm(Long n1, Long n2) {
		return n1 * n2 / getGcd(n1, n2);
	}
}
