class Solution {
private:
    const int MODDER = 1e9 + 7;
    int dp[110][110][110];
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        int n = static_cast<int>(group.size());

        dp[0][0][0] = 1;
        for(int i = 0;i < n;i++) {
                for(int j = 0;j <= G;j++) {
                    for(int k = 0;k <= P;k++) {
                        dp[i + 1][j][k] = dp[i][j][k];

                    }
                }
            for(int j = 0;j <= G && j + group[i] <= G;j++) {
                for(int k = 0;k <= P;k++) {
                    if(dp[i][j][k] == 0) continue;
                    dp[i + 1][j + group[i]][min(k + profit[i],P)] += dp[i][j][k];
                    dp[i + 1][j + group[i]][min(k + profit[i],P)] %= MODDER;
                }
            }
        }
        int result = 0;
        for(int i = 0;i <= G;i++) {
            result = (result + dp[n][i][P]) % MODDER;
        }
        return result;
    }
};
