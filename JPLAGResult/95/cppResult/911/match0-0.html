class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        vector<vector<long long>> dp(P+1, vector<long long>(G+1, 0L));
        dp[0][0] = 1L;
        long long MOD = 1000000007L;
        for (int i = 0; i < group.size(); i++) {
            int a = group[i];
            int b = profit[i];
            for (int j = P; j >= 0; j--) {
                int t = min(j+b, P);
                for (int k = G; k >= a; k--) {
                    dp[t][k] = (dp[t][k] + dp[j][k-a]) % MOD;
                }
            }
        }
        long long ans = 0;
        for (int i = G; i >= 0; i--) {
            ans = (ans+dp[P][i])%MOD;
        }
        return (int)ans;
    }
};
