#define Long long long
#define MOD (1000000000 + 7)

class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        int n = group.size();
        Long result = 0;
        
        // Count number of subsets <= G people.
        {
            vector<int> f = vector<int>(G + 1, 0);
            f[0] = 1;
            
            for (int i = 0; i < n; ++i) {
                for (int g = G; g >= 0; --g) {
                    int ng = g + group[i];
                    if (ng <= G) {
                        f[ng] = (f[ng] + f[g]) % MOD;
                    }
                }
            }
            
            for (int g = 0; g <= G; ++g) {
                result = (result + f[g]) % MOD;
            }
            
            //for (const int & x : f) cout << x << " ";
            //cout << endl;
        }
        
        //cout << result << endl;
        
        // Count number of subsets <= G people and < P profit.
        {
            vector<vector<int>> f = vector<vector<int>>(P, vector<int>(G + 1, 0));
            f[0][0] = 1;

            for (int i = 0; i < n; ++i) {
                for (int p = P - 1; p >= 0; --p) {
                    for (int g = G; g >= 0; --g) {
                        int np = p + profit[i];
                        int ng = g + group[i];
                        if (np < P && ng <= G) {
                            f[np][ng] = (f[np][ng] + f[p][g]) % MOD;
                        }
                    }
                }
            }

            for (int p = 0; p < P; ++p) {
                for (int g = 0; g <= G; ++g) {
                    result = (result + MOD - f[p][g]) % MOD;
                }
            }
            
            /*for (int p = 0; p < P; ++p) {
                for (int g = 0; g <= G; ++g) {
                    cout << f[p][g] << " ";
                }
                cout << endl;
            }*/
        }
        
        return (int)result;
    }
};
