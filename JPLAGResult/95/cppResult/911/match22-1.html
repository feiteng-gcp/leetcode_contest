class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        int mod=1e9+7;
        int K=group.size(),maxP=0;
        for (int p:profit)
            maxP+=p;
        vector<vector<vector<int> > > f=vector<vector<vector<int> > >(K+1,vector<vector<int> >(G+1,vector<int>(P+1,0)));

        f[0][0][0]=1;
        for (int k=0;k<K;++k){
            int dg=group[k],dp=profit[k];
            for (int g=0;g<=G;++g){
                for (int p=0;p<=P;++p){
                    if (f[k][g][p]==0)
                        continue;
                    f[k+1][g][p]=(f[k+1][g][p]+f[k][g][p])%mod;
                    if (g+dg<=G){
                        int newp=min(P,p+dp);
                        f[k+1][g+dg][newp]=(f[k+1][g+dg][newp]+f[k][g][p])%mod;                       
                    }

                }                    
            }       
        }
        int res=0;
        for (int g=0;g<=G;++g){
            res=(res+f[K][g][P])%mod;                   
        }
        return res;
    }
};
