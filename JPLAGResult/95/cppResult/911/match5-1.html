using ll = long long;
constexpr int MOD = 1e9 + 7;

ll dp[110][110];

class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        int n = group.size();
        memset(dp, 0, sizeof(dp));

        dp[0][0] = 1;
        for (int i = 0; i < n; ++ i) {
            for (int j = P; j >= 0; -- j) {
                for (int k = G - group[i]; k >= 0; -- k) {
                    dp[min(P, profit[i] + j)][group[i] + k] += dp[j][k];
                    dp[min(P, profit[i] + j)][group[i] + k] %= MOD;
                }
            }
        }
        
        ll ans = 0;
        for (int i = 0; i <= G; ++ i) {
            ans = (ans + dp[P][i]) % MOD;
        }
        return ans;
    }
};
