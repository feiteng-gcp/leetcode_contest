class Solution {
public:
    int dp[2][105][10005];
    int MOD = 1e9 + 7;
    void add(int& a, int b) {
        (a += b) %= MOD;
    }
    int pw(int a, int b) {
        int res = 1;
        for (; b; b >>= 1, a = 1ll * a * a % MOD) {
            if (b & 1) {
                res = 1ll * res * a % MOD;
            }
        }
        return res;
    }
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        int n = group.size();
        for (int i = 0; i <= G; i++) {
            for (int j = 0; j < P; j++) {
                dp[0][i][j] = 0;
            }
        }
        int sum = 0;
        for (auto v : profit) sum += v;
        
        dp[0][0][0] = 1;
        int cur = 0;
        for (int _ = 0; _ < n; _++) {
            int w = group[_], v = profit[_];
            for (int i = 0; i <= G; i++) {
                for (int j = 0; j <= sum; j++) {
                    dp[cur ^ 1][i][j] = dp[cur][i][j];
                }
            }
            for (int i = w; i <= G; i++) {
                for (int j = v; j <= sum; j++) {
                    add(dp[cur ^ 1][i][j], dp[cur][i - w][j - v]);
                }
            }
            cur ^= 1;
        }
        int res = 0;
        for (int i = 0; i <= G; i++) {
            for (int j = P; j <= sum; j++) {
                add(res, dp[cur][i][j]);
            }
        }
        // cout << res << endl;
        // return ((pw(2, n) - res) % MOD + MOD) % MOD;
        return res;
    }
};
