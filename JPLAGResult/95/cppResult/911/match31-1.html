const int mod = 1e9 + 7.5;
const int N = 128;
const int G = 128;
const int P = 128;

int f[N][G][P];

class Solution {
public:
    void update(int &x, int &y) {
        x += y;
        x %= mod;
    }

    
    int f[N][G][P];
    
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        vector<int> v;
        

        memset(f, 0, sizeof(f));
        int n = group.size();
        f[0][0][0] = 1;
        
        for (int k = 0; k < group.size(); k++) {
            for (int i = 0; i <= G; i++) {
                for (int p = 0; p <= P; p++) if (f[k][i][p] != 0 && i + group[k] <= G) {
                    update(f[k + 1][min(G, i + group[k])][min(P, p + profit[k])], f[k][i][p]);   
                }
            }
            for (int i = 0; i <= G; i++) {
                for (int p = 0; p <= P; p++) {
                    update(f[k + 1][i][p], f[k][i][p]);
                }
            }
        }
        
        int ret = 0;
        for (int i = 0; i <= G; i++) {
            ret += f[group.size()][i][P];
            ret %= mod;
        }
        
        return ret;
    }
};
