typedef long long ll;
class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        const int mod = 1e9+7;
        ll dp[105][105][105];
        memset(dp,0,sizeof(dp));
        const int maxn = 101;
        int n=group.size();
        for(int i=0;i<n;i++){
            dp[i][group[i]][min(P,profit[i])]=1;
        }

        for(int i=1;i<n;i++){
            int nowg=group[i],nowp=profit[i];
            for(int j=1;j<=G;j++){
                for(int k=0;k<maxn;k++){
                    int mid = nowp;
                    if(k+mid>=P){
                        mid = P-k;
                    }
                    if(j+nowg<=G){
                        dp[i][j+nowg][k+mid]+=dp[i-1][j][k];
                        dp[i][j+nowg][k+mid]%=mod;
                    }
                    dp[i][j][k]+=dp[i-1][j][k];
                    dp[i][j][k]%=mod;
                }
            }
        }
        ll ans=0;
        for(int i=1;i<maxn;i++){
            if(i>G)break;
            ans = (ans + dp[n-1][i][P])%mod;
        }
        return (int)ans;
    }
};
