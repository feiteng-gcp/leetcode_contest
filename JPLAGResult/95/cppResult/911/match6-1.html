class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        const int MOD = 1e9 + 7;
        int n = static_cast<int>(profit.size());
        int totalProfit = 0;
        for (auto p : profit) {
            totalProfit += p;
        }
        vector<vector<int>> dp(totalProfit + 1, vector<int>(G + 1));

        dp[0][0] = 1;
        for (int i = 0; i < n; ++i) {
            for (int j = totalProfit; j >= profit[i]; --j) {
                for (int k = G; k >= group[i]; --k) {
                    dp[j][k] += dp[j - profit[i]][k - group[i]];
                    dp[j][k] %= MOD;
                }
            }
        }
        int nums = 0;
        for (int j = P; j <= totalProfit; ++j) {
            for (int k = 0; k <= G; ++k) {
                nums += dp[j][k];
                nums %= MOD;
            }
        }
        return nums;
    }
};
