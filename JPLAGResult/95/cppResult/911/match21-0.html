class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        vector<vector<int>> gp(G+1,vector<int>(P+1,0));
        const int mod = pow(10,9) + 7;
        gp[0][0]=1;
        for (int i=0;i<group.size();i++)
        {
            int g=group[i], p=profit[i];
            
            for(int j=G;j-g>=0;j--)
            {
                for (int k=P+p;k-p>=0;k--)
                {
                    int prev=gp[j-g][k-p];
                    if (k<P) 
                    {
                        gp[j][k] = (gp[j][k]+prev)%mod;
                    }
                    else
                    {
                        gp[j][P] = (gp[j][P]+prev)%mod;
                    }
                }
            }
        }
        int result=0;
        for (int i=0;i<=G;i++) result = (result+gp[i][P])%mod;
        return result;
    }
};
