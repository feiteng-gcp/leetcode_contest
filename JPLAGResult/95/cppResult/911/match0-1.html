class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        vector<vector<long long>> dp(G+1, vector<long long>(P+1));
        int len = profit.size();
        dp[0][0] = 1;

        int mod = 1e9+7;
        for(int i=0;i<len;i++){
            int g = group[i];
            int p = profit[i];
            for(int x = G-g;x>=0;x--){
                for(int y=P;y>=0;y--){
                    int tp = y+p>=P?P:y+p;
                    dp[x+g][tp]+= dp[x][y];

                    dp[x+g][tp]%=mod;
                }
            }
        }
        long long sum = 0;
        for(int i=0;i<=G;i++){
            sum+=dp[i][P];
        }
        sum%=mod;
        return sum;
    }
};
