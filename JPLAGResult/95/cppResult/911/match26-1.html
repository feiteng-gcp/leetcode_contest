class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        long long MOD = 1e9 + 7;
        long long m[101][10001];

        for (int i = 0; i <= 100; i++) {
            for (int j = 0; j <= 10000; j++) {
                m[i][j] = 0;
            }
        }

        m[0][0] = 1;
        for (int i = 0; i < group.size(); i++) {
            for (int g = 100; g >= 0; g--) {
                for (int p = 200; p >= 0; p--) {
                    long long num = m[g][p];
                    if (g >= group[i] && p >= profit[i]) {
                        int pp = p;
                        int ppp = p - profit[i];
                        if (p >= P) {
                            pp = P;
                        }
                        m[g][pp] += m[g - group[i]][ppp];
                        m[g][pp] %= MOD;
                    }
                }
            }
        }
        long long res = 0;
        for (int g = 0; g <= G; g++) {
            for (int p = P; p <= 10000; p++) {
                res += m[g][p];
                res %= MOD;
            }
        }
        return res;
    }
};
