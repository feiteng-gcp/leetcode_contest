class Solution {
public:
	int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
		vector<vector<int> > scms(G+1, vector<int>(P+1, 0));
		scms[G][P] = 1;
		int mo = 1000000007;

		int len = group.size();
		for (int i = 0; i < len; ++i) {
			for (int j = 0; j+group[i] <= G; ++j) {
				for (int k = 0; k <= P; ++k) {
					int np = max(0, k-profit[i]);
					scms[j][np] = (scms[j][np] + scms[j+group[i]][k]) % mo;
				}
			}
		}
		int ans = 0;
		for (int i = 0; i <= G; ++i) {
			ans = (ans + scms[i][0]) % mo;
		}
		return ans;
	}
};
