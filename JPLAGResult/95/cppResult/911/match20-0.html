typedef long long ll;

ll dp[110][110];
const int mod = 1e9 + 7;

class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& g, vector<int>& p) {
        int n = g.size();
        memset(dp, 0, sizeof(dp));
        dp[0][0] = 1;
        for (int i = 0; i < n; ++i) {
            int _g = g[i], _p = p[i];
            for (int j = G - _g; j >= 0; --j) {
                for (int k = P; k >= 0; --k) {
                    int _j = j + _g;
                    int _k = min(P, k + _p);
                    if (dp[j][k] == 0) continue;
                    dp[_j][_k] += dp[j][k];
                    dp[_j][_k] %= mod;
                    //cout << _j << ' ' << _k << ' ' << dp[_j][_k] << ' ' << j << ' ' << k << ' ' << dp[j][k] << endl;
                }
            }
        }
        ll ans = 0;
        for (int i = 0; i <= G; ++i) ans = (ans + dp[i][P]) % mod;
        return ans;
    }
};
