const int N = 100 + 10;
const int M = 1e4 + 10;
const int MOD = 1e9 + 7;

int dp[N][M];

class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& A, vector<int>& B) {
        memset(dp, 0, sizeof(dp));
        int n = A.size();
        int m = 0;
        dp[0][0] = 1;
        for (int i = 0; i < n; ++i) {
            m = min(M - 1, m + B[i]);
            for (int k = G; k >= A[i]; --k) {
                for (int j = m; j >= B[i]; --j) {
                    dp[k][j] = (dp[k - A[i]][j - B[i]] + dp[k][j]) % MOD;
                }
            }
        }
        int ret = 0;
        for (int i = 1; i <= G; ++i) {
            for (int j = P; j < M; ++j) {
                ret = (ret + dp[i][j]) % MOD;
            }
        }
        return ret;
    }
};
