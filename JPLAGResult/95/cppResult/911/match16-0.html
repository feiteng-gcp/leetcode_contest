class Solution {
public:
    int profitableSchemes(int G, int P, vector<int>& group, vector<int>& profit) {
        vector<vector<int>> f(G + 1, vector<int>(101, 0));
        int mod = 1000000007;
        f[0][0] = 1;
        for (int i = 0; i < group.size(); ++i) {
            vector<vector<int>> g = f;
            for (int j = 0; j <= G; ++j) {
                for (int k = 0; k <= 100; ++k) {
                    if (f[j][k] == 0) { continue; }
                    int jj = j + group[i];
                    int kk = k + profit[i];
                    if (jj > G) { continue; }
                    if (kk > 100) { kk = 100; }
                    g[jj][kk] = (g[jj][kk] + f[j][k]) % mod;
                }
            }
            f = g;
        }
        long long ans = 0;
        for (int i = 1; i <= G; ++i) {
            for (int j = P; j <= 100; ++j) {
                ans += f[i][j];
            }
        }
        return ans % mod;
    }
};
