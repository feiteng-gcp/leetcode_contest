/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        auto ptr = head;
        auto p2 = head;
        
        while (p2 && p2->next) {
            ptr = ptr->next;
            p2 = p2->next->next;
        }
        
        
        return ptr;
    }
};
