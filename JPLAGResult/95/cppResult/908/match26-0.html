/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        int cnt = 0 ;
        auto p = head;
        while (p) {
            cnt++;
            p = p->next;
        }
        p = head;
        cnt /= 2;
        while (cnt--) {
            p = p->next;
        }
        return p;
        
    }
};
