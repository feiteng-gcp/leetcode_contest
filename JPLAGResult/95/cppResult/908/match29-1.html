/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */

class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        int length = 0;
        ListNode* p = head;
        while(p!=nullptr){
            length++;
            p = p->next;
        }
        
        p = head;
        for(int i=0;i<length/2;i++) p = p->next;
        return p;
    }
};
