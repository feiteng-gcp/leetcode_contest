/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */

class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        ListNode *s, *f;
        if (!head || !head->next) return head;
        s = head;
        f = head;
        while (s && f && s->next && f->next) {
            s = s->next;
            f = f->next->next;;
        }
        //cout << s->val << endl;
        return s;
    }
};
