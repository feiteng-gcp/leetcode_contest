/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        int cnt = 0,ind;
        ListNode* cur = head;
        while(cur != NULL){
            cnt++;
            cur = cur->next;
        }
        ind = (cnt+2)/2;
        cur = head;
        for(int i = 1; i < ind; i++){
            cur = cur->next;
        }
        return cur;
    }
};
