/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
		ListNode* p = head;
		int count = 0;
		while (p != nullptr) {
			count++;
			p = p->next;
		}
		p = head;
		for (int i = 1; i < count / 2 + 1; i++) {
			p = p->next;
		}
		return p;
	}
};
