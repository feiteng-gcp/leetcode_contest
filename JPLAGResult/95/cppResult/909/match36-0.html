class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        int table[n][n];
        for (int i = 0; i < n; i += 1) {
            for (int j = 0, k = i; k < n; j += 1, k += 1) {
                int x = ((j + 2) <= k) ? table[j + 2][k] : 0;
                int y = ((j + 1) <= (k - 1)) ? table[j + 1][k - 1] : 0;
                int z = (j <= (k - 2)) ? table[j][k - 2]: 0;
                table[j][k] = max(piles[j] + min(x, y), piles[k] + min(y, z));
            }
        }
        int sum = 0;
        for (int i = 0; i < n; i += 1) sum += piles[i];
        return table[0][n - 1] > (sum / 2);
    }
};
