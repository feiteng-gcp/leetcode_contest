#include <bits/stdc++.h>
class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        int total = 0;
        for(int i=0; i<n; i++) {
            total += piles[i];
        }
        int t[n][n];
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                t[i][j] = 0;
            }
        }
        int g=0;
        int i=0;
        int j=0;
        int x=0;
        int y=0;
        int z=0;


        for (g = 0; g<n; g++)
        {
            for (i=0, j=g; j<n; i++, j++)
            {
                x = ((i+2) <= j) ? t[i+2][j] : 0;
                y = ((i+1) <= (j-1)) ? t[i+1][j-1] : 0;
                z = (i <= (j-2))? t[i][j-2]: 0;

                t[i][j] = max(piles[i] + min(x, y), piles[j] + min(y, z));
            }
        }

        int asum = t[0][n-1];
        int bsum = total - t[0][n-1];
        return asum > bsum;
    }
            
};
