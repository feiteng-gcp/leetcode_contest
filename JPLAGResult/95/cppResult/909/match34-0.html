int dp[550][550];
int sum[550];

class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        sum[0] = piles[0];
        for (int i = 1; i < n; ++ i) sum[i] = sum[i - 1] + piles[i];
        
        memset(dp, -1, sizeof(dp));
        int alex = gao(piles, 0, n - 1);
        return alex > sum[n - 1] - alex;
    }
    
    int gao(vector<int> &piles, int l, int r) {
        if (l > r) return 0;
        if (dp[l][r] != -1) return dp[l][r];
        
        int ans = piles[l] + sum[r] - sum[l] - gao(piles, l + 1, r);
        
        int tmp = piles[r] + (r == 0 ? 0: sum[r - 1]) - (l == 0 ? 0 : sum[l - 1]) - gao(piles, l, r - 1);
        return dp[l][r] = max(ans, tmp);
    }
};
