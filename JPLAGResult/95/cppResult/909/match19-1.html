const int MAXN = 550;
int dp[MAXN][MAXN];

class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                dp[i][j] = INT_MIN;
            }

        }
        for (int i = 0; i < n; i++) {
            dp[i][i] = piles[i];
        }
        for (int len = 2; len <= n; len++) {
            for (int i = 0; i+len <= n; i++) {
                int j = i+len-1;
                dp[i][j] = max(piles[i] - dp[i+1][j], piles[j] - dp[i][j-1]);
            }
        } 
        // cout << dp[0][n-1] << endl;
        return dp[0][n-1] > 0;
        
    }
};
