class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        //piles[0] = 3;
       // piles[1] = 2;
        //piles[2] = 10;
       // piles[3] = 4;

        int a[1000];
        int l = piles.size();
        for(int i = 0; i < l; i++){
            a[i] = -piles[i];
        }
        for(int j = 1; j < l; j++){
            for(int i = 0; i+j < l; i++){
                //cout<<' '<<i<<' '<<a[i]<<' '<<piles[i+1]<<' '<<piles[i]<<endl;
                if(j % 2 == 1)
                    a[i] = max(a[i] + piles[i+j], a[i+1] + piles[i]);
                else
                    a[i] = min(a[i] - piles[i+j], a[i+1] - piles[i]);
                //cout<<a[i]<<endl;
            }
           // for(int i = 0; i+j < l; i++)
            //    cout<<a[i]<<' ';
           // cout<<endl;
        }
        if(a[0] > 0)
            return true;
        else
            return false;
    }
};
