class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        vector<vector<int>> dp(505,vector<int>(505, 0));
        int len = piles.size();
        for(int i=0;i<len;i++){
            dp[i][i] = piles[i];
        }
        for(int i=1;i<len;i++){
            for(int j=0;j<len-i;j++){
                int k = j+i;
                dp[j][k] = max(piles[j]-dp[j+1][k], piles[k]-dp[j][k-1]);
            }
        }
        //cout<<dp[0][len-1]<<endl;
        return dp[0][len-1]>0;
    }
};
