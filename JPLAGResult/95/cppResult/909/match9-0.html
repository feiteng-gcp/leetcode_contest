class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for (int i = 0; i < n; i++) {
            dp[i][i] = piles[i];
        }
        for (int diff = 1; diff < n; diff++) {
            for (int i = 0; i+diff < n; i++) {
                dp[i][i+diff] = max(piles[i]-dp[i+1][i+diff], piles[i+diff]-dp[i][i+diff-1]);
            }
        }
        /*
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                printf("%3d", dp[i][j]);
            }
            printf("\n");
        }
        */
        return (dp[0][n-1] > 0);
    }
};
