class Solution {
public:
    int n;
    int dp[505][505];
    int sum[505];
    bool stoneGame(vector<int>& piles) {
        n = piles.size();
        sum[0] = 0;
        for (int i = 1; i <= n; i++) {
            sum[i] = sum[i - 1] + piles[i];
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = -1;
            }
        }
        for (int k = 1; k <= n; k++) {
            for (int i = 1; i + k - 1 <= n; i++) {
                int j = i + k - 1;
                if (k == 1) {
                    dp[i][j] = piles[i - 1];
                } else {
                    dp[i][j] = max(piles[i - 1] + sum[j] - sum[i] - dp[i + 1][j], piles[j - 1] + sum[j - 1] - sum[i - 1] - dp[i][j - 1]);
                }
            }
        }
        return dp[1][n] > sum[n] - dp[1][n];
    }
};
