class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();

        vector<int> sums = piles, dp = piles;
        
        for (int i = 1; i < n; i++) {
            sums[i] += sums[i - 1];
        }
        
        for (int step = 2; step <= n; step++) {
            for (int start = 0; start <= n - step; start++) {
                int sumOfSubarray = sums[start + step - 1] - sums[start - 1];
                
                // int takeHeadScore = arr[start] + (sumOfSubarray - arr[start] - dp[start + 1]);
                // int takeTailScore = arr[start + step - 1] + (sumOfSubarray - arr[start + step - 1] - dp[start]);
                int takeHeadScore = sumOfSubarray - dp[start + 1];
                int takeTailScore = sumOfSubarray - dp[start];
                dp[start] = max(takeHeadScore, takeTailScore);
            }
        }
        
        return dp[0] > (sums.back() - dp[0]);
    }
};
