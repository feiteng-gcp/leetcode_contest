class Solution {
public:
    bool stoneGame(vector<int>& piles) {
		int sum = 0, subsum = 0;
		sort(piles.begin(), piles.end());
		for (int i = 0; i < piles.size(); i++) {
			if (i < piles.size() / 2) {
				subsum += piles[i];
			}
			sum += piles[i];
		}
		return (double)subsum / sum < 0.49;
	}
};
