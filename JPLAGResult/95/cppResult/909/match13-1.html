class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int sum = 0;
        for (auto &num : piles) {
            sum += num;
        }
        int sz = piles.size();
        vector<vector<int>> dp(sz, vector<int>(sz, 0));
        for (int i = 0; i < sz; i++) {
            dp[i][i] = piles[i];
        }

        for (int i = 0; i + 1 < sz; i++) {
            dp[i][i+1] = max(piles[i], piles[i+1]);
        }
        for (int l = 2; l < sz; l++) {
            for (int i = 0; i + l < sz; i++) {
                int j = i + l;
                dp[i][j] = max(piles[i]+min(dp[i+1][j-1], dp[i+2][j]),
                              piles[j]+min(dp[i+1][j-1], dp[i][j-2]));
            }
        }
        //cout << dp[0][sz-1] << "\n";
        return dp[0][sz-1] * 2 > sum;
    }
};
