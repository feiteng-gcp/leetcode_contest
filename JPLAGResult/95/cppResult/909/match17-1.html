class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int sum = 0;
        for (auto i : piles) {
            sum += i;
        }

        int len = piles.size();
        vector<vector<int>> dp(len, vector<int>(len, 0));
        for (int i = 0; i < len; i++) {
            dp[i][i] = piles[i];
            if (i < len - 1) {
                dp[i][i + 1] = max(piles[i], piles[i + 1]);
            }

        }

        for (int i = len - 1; i >= 0; i--) {
            for (int j = i + 2; j < len; j++) {
                int t1 = min(dp[i + 2][j] + piles[i], dp[i + 1][j - 1] + piles[i]);
                int t2 = min(dp[i][j - 2] + piles[j], dp[i + 1][j - 1] + piles[j]);
                dp[i][j] = max(t1, t2);
            }
        }

        return dp[0][len - 1] > sum / 2;
    }
};
