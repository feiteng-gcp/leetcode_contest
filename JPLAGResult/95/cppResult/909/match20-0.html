const int N = 500 + 10;
int dp[N][N][2];

class Solution {
public:
    bool stoneGame(vector<int>& a) {
        int n = a.size();
        for (int i = 0; i < n; ++i) {
            dp[i][i][0] = a[i];
            dp[i][i][1] = -a[i];
        }
        for (int k = 1; k < n; ++k) {
            for (int i = 0; i + k < n; ++i) {
                int j = i + k;
                dp[i][j][0] = max(a[i] + dp[i + 1][j][1], a[j] + dp[i][j - 1][1]);
                dp[i][j][1] = min(-a[i] + dp[i + 1][j][0], -a[j] + dp[i][j - 1][0]);
            }
        }
        return dp[0][n - 1][0] > 0;
    }
};
