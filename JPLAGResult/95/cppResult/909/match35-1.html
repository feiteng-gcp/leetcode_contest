int f[1001][1001];
int sum[1001];

class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        sum[0] = 0;
        for (int i = 1; i <= n; i++)
            sum[i] = sum[i - 1] + piles[i - 1];
        for (int len = 1; len <= n; len++)
            for (int l = 1; l + len - 1 <= n; l++){
                int r = l + len - 1;
                if (len == 1)
                    f[l][r] = piles[l - 1];
                else
                    f[l][r] = max(piles[l - 1] + sum[r] - sum[l] - f[l + 1][r], 
                                  piles[r - 1] + sum[r - 1] - sum[l - 1] - f[l][r - 1]);
            }
        return f[1][n] > sum[n] / 2;
    }
};
