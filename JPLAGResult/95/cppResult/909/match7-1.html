class Solution {
public:
    bool stoneGame(vector<int>& piles) {

        int dp[piles.size()][piles.size()];
        
        for (int i = 0; i < piles.size() - 1; ++i) {
            dp[i][0] = abs(piles[i] - piles[i + 1]);
         //   printf("dp[%d][%d] = %d\n", i, 0, dp[i][0]);
        }
        
        for (int i = 1; i < piles.size() - 1; ++i) {
            for (int j = 0; j < piles.size() - i - 1; ++j) {
                dp[j][i] = max(piles[j] - dp[j + 1][i - 1], piles[j + i + 1] - dp[j][i - 1]); 
         //       printf("dp[%d][%d] = %d\n", j, i, dp[j][i]);
            }
        }
        

        
        return dp[0][piles.size() - 2] > 0;
    }
};
