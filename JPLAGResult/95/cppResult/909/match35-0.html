class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        sort(piles.begin(), piles.end());
        int alex = 0, lee = 0;
        for (int i = 0; i < piles.size(); i += 2) {
            lee += piles[i];
            alex += piles[i+1];
        }
        return alex > lee;
    }
};
