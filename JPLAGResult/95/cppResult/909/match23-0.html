class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        vector<vector<int>> vi(n);
        vector<vector<int>> vj(n);
        for (int i = 0; i < n; i++) {
            vi[i] = vector<int>(n, 0);
            vj[i] = vector<int>(n, 0);
            vi[i][i] = piles[i];
        }
        
        for (int k = 2; k <= n; k++) {
            for (int i = 0; (i + k - 1) < n; i++) {
                int j = i + k - 1;
                
                int t1 = piles[i] + vj[i + 1][j];
                int t2 = piles[j] + vj[i][j - 1];
                if (t1 > t2) {
                    vi[i][j] = t1;
                    vj[i][j] = vi[i + 1][j];
                }
                else {
                    vi[i][j] = t2;
                    vj[i][j] = vi[i][j - 1];
                }
            }
        }
        
        return (vi[0][n - 1] > vj[0][n - 1]);
    }
};
