class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int n = piles.size();
        if (n <= 2) return true;
        auto total = accumulate(piles.begin(), piles.end(), 0);

        vector<vector<int>> dp(n, vector<int>(n, 0));
        
        for (int i=0; i<n; ++i) {
            dp[i][i] = piles[i];
        }
        
        for (int len = 1; len < n; ++len) {
            for (int i = 0; i+len < n; ++i) {
                int j = i + len;
                
                int third = (i+2 <= j) ? dp[i+2][j] : 0;
                int third2 = (i <= j-2) ? dp[i][j-2] : 0;
                int x = piles[i] + min(third, dp[i+1][j-1]);
                int y = piles[j] + min(dp[i+1][j-1], third2);
                dp[i][j] = max(x, y);
            }
        }
        /*for (int i=0; i<n; ++i) {
            for (int j=0; j<n; ++j)
                cout << dp[i][j] << " ";
            cout << endl;
        }*/
        return 2*dp[0][n-1] > total; 
    }
};
