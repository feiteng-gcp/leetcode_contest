class Solution {
public:
    bool stoneGame(vector<int>& nums) 
    {
        int len=nums.size();
        vector<int> dp(len,0);
        
        for(int i=len-1;i>=0;i--)
        {
            for(int j=i;j<len;j++)
            {
                if(j==i) dp[j]=nums[i];
                else dp[j]=max(nums[i]-dp[j],nums[j]-dp[j-1]);
            }
        }
        
        return dp[len-1]>=0;
    }
    
};
