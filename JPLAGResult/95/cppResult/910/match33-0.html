class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        long long count_A = 1;
        long long count_B = 1;
        long long v_lcm = LCM(A, B);
        int v_mod = pow(10, 9) + 7;
        int cur = 0;
        if (A == B) {
            return (long long)N * A % v_mod;
        }
        long long left = 1;
        long long right = 2;
        while (right / A + right / B - right / v_lcm < N) {
            left <<= 1;
            right <<= 1;
        }
        while (left + 1 < right) {
            long long mid = (right - left) / 2 + left;
            long long temp = mid / A + mid / B - mid / v_lcm;
            if (temp >= N) {
                right = mid;
            }
            else {
                left = mid;
            }
        }
        if (left / A + left / B - left / v_lcm == N) return left % v_mod;
        return right % v_mod;
    }
    
    int GCD(int A, int B) {
        int m = max(A, B);
        int n = min(A, B);
        int temp = n;
        while (m % n){
            temp = m % n;
            m = n;
            n = temp;
        }
        return temp;
    }
    
    long long LCM(int A, int B) {
        return (long long)A * B / GCD(A, B);
    }
};
