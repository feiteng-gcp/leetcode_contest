class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        long long mod = 1e9 + 7.5;
        
        long long l = 1, r = 0x7fffffffffffLL;
        long long ret = 0;
        
        long long lcm = 1LL * A * B / gcd(A, B);
        
        while (l <= r) {
            long long mid = (l + r) / 2;
            long long temp = (mid / A + mid / B - mid / lcm);
            if (temp >= N) {
                ret = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        
        return ret % mod;
    }
    
    int gcd(int a, int b) {
        if (a % b == 0) {
            return b;
        } 
        return gcd(b, a % b);
    }
                     
};
