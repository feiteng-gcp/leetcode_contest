class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {

        long long l = 1, r = 1e16;
        while (l <= r){
            long long m = (l + r) / 2;
            long long cnt = m / A + m / B - m / (A * B / __gcd(A, B));
            if (cnt >= N)
                r = m - 1;
            else
                l = m + 1;
        }
        cout << l << " " << r << endl;
        return l % (int)(1e9 + 7);
    }
};
