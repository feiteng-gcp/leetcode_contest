class Solution {
public:
    long long h(long long a, long long b) {
        if (0 == a % b) {
            return b;
        }
        return h(b, a%b);
    }
    
    long long solve(long long a, long long b, long long t, int N) {
        long long r = N;
        long long l = 0;
        while (l <= r) {
            long long m = l + (r-l)/2;
            long long yy = t*m;
            long long x1 = yy / a;
            long long x2 = yy / b;
            long long x3 = yy / (a*b);
            if (x1+x2-x3 > N) {
                r = m-1;
            } else if (x1+x2-x3 == N) {
                return yy;
            } else {
                l = m+1;
            }
        }
        return -1;
    }
    
    int nthMagicalNumber(int N, int A, int B) {
        long long a = max(A, B), b = min(A, B);
        long long c = h(a, b);
        a /= c;
        b /= c;
        long long MOD = 1000000007;
        long long t1 = solve(a, b, a, N);
        if (t1 >= 0) {
            t1 = t1 % MOD * c % MOD;
            return (int)t1;
        }
        long long t2 = solve(a, b, b, N);
        t2 = t2 % MOD * c % MOD;
        return t2;
    }
};
