class Solution {
public:
    
   using ll = long long;
    
    const int MOD = 1000000007;


   ll gcd (ll a, ll b) {
      if (a == 0) return b;
      return gcd (b % a, a);
   }

   int nthMagicalNumber (int N, int A, int B) {
      ll h, l, m, c, lcm;
      l = 1, h = 1e18;
      lcm = (A * B) / gcd (A, B);

      while (l <= h) {
         m = (l + h) / 2;
         c = (m / A) + (m / B) - (m / lcm);

         if (c < N)  l = m + 1;
         if (c > N)  h = m - 1;
         if (c == N) break;
      }
      
      while (m % A && m % B) m--;

      return m % MOD;
   }
};
