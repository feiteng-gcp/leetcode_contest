class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        int MAXC=1000000007;
        int maxf=maxfct(A,B);
        int cmf=(A*B)/maxf;
        int ucnt=cmf/A+cmf/B-1;
        //cout << maxf << " -> " << cmf << " -> " << ucnt << endl;
        long long cc=((long long)(N/ucnt)*cmf)%MAXC;
        int i=0;
        int j=0;
        int rem=N%ucnt;
        int ret=0;
        while(i+j<rem){
            if((i+1)*A<(j+1)*B){
                i++;
                ret=i*A;
            }
            else{
                j++;
                ret=j*B;
            }
        }
        //cout << cc << " -> " << ret;
        ret=(cc+ret)%MAXC;
        return ret;
    }
    
    int maxfct(int A, int B){
        int tmp=max(A,B);
        int ret=min(A,B);
        while(tmp>1 && tmp%ret){
            tmp=tmp%ret;
            swap(tmp,ret);
        }
        return ret;
    }
};
