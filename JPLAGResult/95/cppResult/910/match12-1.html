class Solution {
public:
    int gcd(int a, int b)
    {
        if (a == 0)
            return b;
        return gcd(b % a, a);
    }
    int nthMagicalNumber(int N, int A, int B) {
        long long int mod = 1000000007;
        long long int high,low,mid,cnt,lcm;

        low = 1;
        high = 1e18;
        lcm = (A*B)/gcd(A,B);

        while(low <= high)
        {
            mid = (low+high)/2;
            cnt = (mid/A)+(mid/B)-(mid/lcm);

            if(cnt<N)
                low=mid+1;
            if(cnt>N)
                high=mid-1;
            if(cnt==N)
                break;
        }
        while(mid%A && mid%B)
            mid--;
        return mid%mod;
    }
};
