class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        long long mod = 1000000007;
        int gcd = __gcd(A,B);

        int a = A/gcd;
        int b = B/gcd;
        int k = N/(a+b-1);
        int remainder = N % (a+b-1);
        long long rst = ((long long)k * a * b) % mod;

        int ma=a;
        int mb=b;
        int residual=0;

        for (int i=0;i<remainder;i++) {
            if (ma<mb) {
                residual=ma;
                ma+=a;
            } else {
                residual=mb;
                mb+=b;
            }
        }
        rst = (rst+residual) % mod;
        rst = (rst * gcd) % mod;
        return (int)rst;
    }
};
