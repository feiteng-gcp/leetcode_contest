class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        int L = lcm(A, B);
        int M = 1000000007;
        
        int res1 = (long long)N / (L/A + L/B - 1) * L % M;
        
        int K = N % (L/A + L/B - 1);
        

        int i = 1;
        int a = 1;
        int b = 1;
        int kth = 0;
        while (i <= K) {
            if (a*A < b*B) {
                kth = a*A;
                ++a;
            }
            else {
                kth = b*B;
                ++b;
            }
            
            ++i;
        }
        
        return (int)(((long long)res1 + (long long)kth) % M);
    }
    
    int lcm (int a, int b) {
        return a*b/gcd(a, b);
    }
    
    int gcd(int a, int b) {
        if (a == 0)
            return b;
        return gcd(b%a, a);
    }
    
    
};
