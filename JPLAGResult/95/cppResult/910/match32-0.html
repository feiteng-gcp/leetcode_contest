class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        long long l = 2;
        long long r = 1000000000000000000;
        long long ans = -1;
        while (l <= r) {
            long long mid = (l + r) >> 1;
            long long cnt = 0;
            if (A == B) {
                cnt = mid / A;
            } else {
                cnt = mid / A + mid / B - mid / lcm(A, B);
            }
            if (cnt > N) {
                r = mid - 1;
            } else if (cnt < N) {
                l = mid + 1;
            } else {
                ans = mid;
                break;
            }
        }
        while (true) {
            if (ans % A == 0) { break; }
            if (ans % B == 0) { break; }
            --ans;
        }
        return ans % 1000000007;
    }
    
    int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
    
    int lcm(int a, int b) {
        return a * b / gcd(a, b);
    }
};
