class Solution {
public:
    long long gcd(long long u, long long v) {
        while ( v != 0) {
            long long r = u % v;
            u = v;
            v = r;
        }
        return u;
    }
    int nthMagicalNumber(int N, int A, int B) {

        long long low = 1;
        long long high = 1e18;
        long long n = N;
        long long a = A;

        long long b = B;
        long long MOD = 1e9 + 7;
        while (low <= high) {
            long long mid = (low + high) / 2;
            long long cnt = 0;
            cnt += mid / a;
            cnt += mid / b;
            cnt -= mid / (a * b / gcd(a, b));
            if (cnt == n) {
                while (mid % a != 0 && mid % b != 0) {
                    mid--;
                }
                return mid % MOD;

            }
            else if (cnt > n) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return -1;
    }
};
