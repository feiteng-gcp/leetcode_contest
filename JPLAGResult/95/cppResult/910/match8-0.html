class Solution {
public:
    long long gcd(long long a, long long b)
    {
        if (a == 0)
            return b;
        return gcd(b % a, a);
    }

    int nthMagicalNumber(int N, int A, int B) {
        long long high,low,mid,count,lcm;
        low = 1;
        high = 1e18;
        lcm = (A * B) / gcd(A, B);

        while(low <= high)
        {
            mid = (low + high) / 2;
            count = (mid / A) + (mid / B) - (mid / lcm);

            if(count < N)
                low = mid + 1;
            if(count > N)
                high = mid - 1;
            if(count == N)
                break;
        }
        while(mid % A && mid % B)
            mid--;

        return mid % long(1E9+7);
    }
};
