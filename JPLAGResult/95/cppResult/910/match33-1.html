class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        long long md=1e9+7;
        long long  g=A*B/__gcd(A,B);
        long long lb=1,ub=1e18;
        while (lb<ub){
            long long mid=(lb+ub)/2;
            if (mid<500)
                cout<<mid/A+mid/B-mid/g<<" "<<mid<<endl;
            if (mid/A+mid/B-mid/g>=N)
                ub=mid;
            else 
                lb=mid+1;
        }
        return lb%md;
    }
};
