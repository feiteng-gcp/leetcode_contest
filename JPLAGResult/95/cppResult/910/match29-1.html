class Solution {
public:
    int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
    int nthMagicalNumber(int N, int A, int B) {
        long long l = 1, r = 1e18;
        long long res;

        long long MOD = 1e9 + 7;
        long long lcm = 1ll * A * B / gcd(A, B);
        while (l <= r) {
            long long m = (l + r) / 2;
            long long cnt = m / A + m / B - m / lcm;
            if (cnt >= N) {
                res = m;
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        return res % MOD;
    }
};
