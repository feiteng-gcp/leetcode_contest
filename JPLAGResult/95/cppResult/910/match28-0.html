class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        int MODDER = 1e9 + 7;
        long long left = min(A,B),right = max(A,B) * 1LL * N;
        long long lcm = A * 1LL * B / __gcd(A,B);
        long long ans = -1;
        while(left <= right) {
            long long mid = (left + right) >> 1;
            long long cnt = mid / A + mid / B - mid / lcm;
            if(cnt < N) {
                left = mid + 1;
            }else {
                if(cnt == N && (mid % A == 0 || mid % B == 0)) {
                    ans = mid;
                    break;
                }
                right = mid - 1;
            }
        }
        return ans % MODDER;
    }
};
