int gcd(int a, int b) {
    int t;
    while (b > 0) {
        t = a;
        a = b;
        b = t % a;
    }
    return a;
}

const int NN = 1000000007;

class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        int g = gcd(A, B);
        long long lcm = A * B / g;
        long long step = lcm / A + lcm / B - 1;
        long long k = (N / step) % NN;
        long long t = N % step;
        if (t == 0) {
            return (k * lcm) % NN;
        }
        
        long long l = 0;
        long long r = lcm;
        while (l < r)  {
            long long m = (l + r) >> 1;
            long long tt = m / A + m / B;
            if (tt == t) {
                if (m % A == 0 || m % B == 0) {
                    return (k * lcm + m) % NN;
                }
                r = m;
            }
            else if (tt < t) {
                l = m + 1;
            }
            else {
                r = m;
            }
        }
        return 0;
    }
};
