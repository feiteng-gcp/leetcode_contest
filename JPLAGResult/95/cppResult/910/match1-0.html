typedef unsigned long long LL;
class Solution {
public:
    LL gcd(LL a, LL b)
    {
        if (a == 0)
            return b;
        return gcd(b % a, a);
    }

    int nthMagicalNumber(int N, int A, int B) {
        LL high,low,mid,cnt,lcm;
        low = 1;
        high = 1e18;
        lcm = (A * B) / gcd(A, B);

        while(low <= high)
        {
            mid = (low + high) / 2;
            cnt = (mid / A) + (mid / B) - (mid / lcm);

            if(cnt < N)
                low = mid + 1;
            if(cnt > N)
                high = mid - 1;
            if(cnt == N)
                break;
        }
        while(mid % A && mid % B)
            mid--;

        return (mid % 1000000007);        
    }
};
