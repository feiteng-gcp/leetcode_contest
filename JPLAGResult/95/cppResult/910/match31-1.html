#define ll long long
class Solution {
public:
    int nthMagicalNumber(int N, int A, int B) {
        long long low = 1, high = 1LL << 60, middle, best = high;
        int lcm = A * B / __gcd(A, B);
        while (low <= high) {

            middle = (low + high + 1) >> 1;
            long long order = middle / A + middle / B - middle / lcm;
            if (order >= N) {
                best = middle;
                high = middle - 1;
            }
            else {
                low = middle + 1;
            }
        }
        return best%1000000007;
    }
};
