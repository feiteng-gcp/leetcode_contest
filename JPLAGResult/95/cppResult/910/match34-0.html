class Solution {
public:

    int gcd(int a,int b)  
    {  
        int t;  
        if(a<b)  
        {  
            t=a;  
            a=b;  
            b=t;  
        }  
        while(t=a%b)  
        {  
            a=b;  
            b=t;  
        }  
        return b;  
    } 

    int nthMagicalNumber(int N, int A, int B) {
        long long low = 2;
        long long high = 5e13L;
        int mod = 1e9+7;
        int c =A*B/ gcd(A, B);
        //cout<<c<<endl;
        while(low<=high){
            long long mid = low+(high-low)/2;
            long long num = mid/A+mid/B-(mid/c);
            if(num<N) low = mid + 1;
            else high = mid - 1;
        }
        return low%mod;
    }
};
