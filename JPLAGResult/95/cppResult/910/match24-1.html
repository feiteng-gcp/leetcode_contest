class Solution {
private:
    int largeNum = 1e9 + 7;
    long long gcd(int a, int b)
    {
        if (b == 0) return a;
        else return gcd(b, a % b);
    }
public:
    int nthMagicalNumber(int n, int a, int b) {
        long long high,low,mid,cnt,lcm;

        low = 1;
        high = 1e18;
        lcm = (a * b) / gcd(a, b);

        while(low <= high)
        {
           mid = (low + high) / 2;
           cnt = (mid / a) + (mid / b) - (mid / lcm);

          if(cnt < n)
            low = mid + 1;
        if(cnt > n)
            high = mid - 1;
        if(cnt == n)
            break;
    }
    while(mid % a && mid % b)
        mid--;
    return mid % largeNum;
    }
};
