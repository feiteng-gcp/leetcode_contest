typedef long long LL;

class Solution {
public:

    long long gcd(LL a, LL b)
    {
        if (a == 0)
            return b;
        return gcd(b % a, a);
    }

    int nthMagicalNumber(int n, int a, int b) {
        LL high,low,mid,cnt,lcm;

        low = 1;
        high = 1e18;
        lcm = (a * b) / gcd(a, b);

        while(low <= high)
        {
            mid = (low + high) / 2;
            cnt = (mid / a) + (mid / b) - (mid / lcm);

            if(cnt < n)
                low = mid + 1;
            if(cnt > n)
                high = mid - 1;
            if(cnt == n)
                break;
        }
        while(mid % a && mid % b)
            mid--;
        return mid % (1000000007);        
    }
};
