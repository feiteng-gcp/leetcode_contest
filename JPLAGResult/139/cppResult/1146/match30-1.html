class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int l1 = str1.size(), l2 = str2.size();

        for (int g = gcd(l1, l2); g > 0; --g) {
            if (l1 % g != 0 || l2 % g != 0) {
                continue;
            }
            bool s1eq = true;
            for (int i = g; i < l1; i += g) {
                if (!equal(g, str1, 0, str1, i)) {
                    s1eq = false;
                    break;
                }
            }
            if (!s1eq) {
                continue;
            }
            bool s2eq = true;

            for (int i = 0; i < l2; i += g) {
                if (!equal(g, str1, 0, str2, i)) {
                    s2eq = false;
                    break;
                }
            }
            if (!s2eq) {
                continue;
            }
            return str1.substr(0, g);
        }
        return "";
    }
    
    bool equal(int len, const string& s1, int p1, const string& s2, int p2) {
        for (int i = 0; i < len; ++i) {
            if (s1[i + p1] != s2[i + p2]) {
                return false;
            }
        }
        return true;
    }
    
    int gcd(int a, int b) {
        if (b == 0) {
            return a;
        }
        return gcd(b, a % b);
    }
};
