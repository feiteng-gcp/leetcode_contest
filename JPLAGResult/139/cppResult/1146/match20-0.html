class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        for(int i=min(str1.size(),str2.size());i>=1;i--){
            if(str1.size()%i||str2.size()%i) continue;
            if(str1.substr(0,i)!=str2.substr(0,i)) continue;
            bool f=false;
            for(int j=1;j<str1.size()/i;j++){
                if(str1.substr((j-1)*i,i)!=str1.substr(j*i,i)) {
                    f=true;
                    break;
                }
            }
            if(f) continue;
            for(int j=1;j<str2.size()/i;j++){
                if(str2.substr((j-1)*i,i)!=str2.substr(j*i,i)) {
                    f=true;
                    break;
                }
            }
            if(!f) {
                return str1.substr(0,i);
            }
        }
        return "";
    }
};
