class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int g = __gcd(str1.size(), str2.size());
        int m = min(str1.size(), str2.size());
        for (int i = m; i >= 1; i--) {
            if (g % i == 0) {
                bool possible = true;

                string pattern = str1.substr(0, i);
                for (int j = 0; j < str1.size(); j += i) {
                    string s = str1.substr(j, i);
                    if (pattern != s) {
                        possible = false;
                        break;
                    }
                }
                for (int j = 0; j < str2.size(); j += i) {
                    string s = str2.substr(j, i);
                    if (pattern != s) {
                        possible = false;
                        break;
                    }
                }
                if (possible) {
                    return pattern;
                }
            }
        }
        return "";
    }
};
