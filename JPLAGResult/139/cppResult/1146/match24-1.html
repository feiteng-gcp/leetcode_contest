class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        string ans;
        int n = str1.size(), m = str2.size();
        for(int i = 1; i <= min(n, m); i++) {
            if(n%i != 0 || m%i != 0)continue;
            string cur = str1.substr(0, i);

            bool flg = 1;
            for(int j = 0; j < n; j++) {
                if(str1[j] != cur[j%i]) {
                    flg = 0;
                    break;
                }
            }
            for(int j = 0; j < m; j++) {
                if(str2[j] != cur[j%i]) {
                    flg = 0;
                    break;
                }
            }
            if(flg)ans = cur;
        }
        return ans;
    }
};
