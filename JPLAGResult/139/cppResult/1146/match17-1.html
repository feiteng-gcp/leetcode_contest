class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int N1 = str1.size();
        int N2 = str2.size();
        for (int len=1000; len >= 1; len--) {
            if (N1 % len != 0 || N2 % len != 0) continue;

            bool ok = true;
            for (int i=len; i<N1; i++) {
                if (str1[i] != str1[i % len])  {
                    ok = false;
                    break;
                }
            }
            if (!ok) continue; 
            for (int i=0; i<N2; i++) {
                if (str2[i] != str1[i % len])  {
                    ok = false;
                    break;
                }
            }
            if (!ok) continue; 
            return str1.substr(0, len);
        }
        return "";
    }
};
