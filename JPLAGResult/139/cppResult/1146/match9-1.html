class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        for (int i = min(str1.size(), str2.size()); i > 0; --i) {
            if (str1.size() % i != 0 || str2.size() % i != 0) continue;

            bool ok = true;
            for (int j = 0; j < str1.size(); ++j) {
                if (str1[j] != str1[j % i]) {
                    ok = false;
                    break;
                }
            }

            for (int j = 0; j < str2.size(); ++j) {
                if (str2[j] != str1[j % i]) {
                    ok = false;
                    break;
                }
            }
            if (ok) return str1.substr(0, i);
        }
        
        return "";
    }
};
