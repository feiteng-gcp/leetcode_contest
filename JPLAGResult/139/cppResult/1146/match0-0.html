class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        string s, t, res;
        if (!funct(str1, s) || !funct(str2,t)) return "";
        if (s != t || s == "") return "";
        int k = gcd(str1.length()/s.length(), str2.length()/s.length());
        while (k--) res += s;
        return res;
    }
    
private:
    int gcd(int a, int b) {
        return b ? gcd(b,a%b) : a;
    }
    
    
    bool funct(string &s, string &ss) {
        int n = s.length();
        vector<int> f(n, 0);
        for (int i = 1, len = 0; i < n; ) {
            if (s[len] == s[i]) f[i++] = ++len;
            else if (len) len = f[len-1];
            else f[i++] = 0;
        }
        ss = s.substr(0, n-f[n-1]); 
        return (n % (n-f[n-1]) == 0);
    }
};
