class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int i, j;
        string temp;
        string ans = "";
        
        for(i = 1; i <= min(str1.length(), str2.length()); ++i) {
            if(str1.length() % i || str2.length() % i) continue;
            
            temp = str1.substr(0, i);
            for(j = 0; j < str1.length() / i; ++j) {
                if(temp.compare(str1.substr(j * i, i))) break;
            }
            if(j < str1.length() / i) continue;
            
            for(j = 0; j < str2.length() / i; ++j) {
                if(temp.compare(str2.substr(j * i, i))) break;
            }
            if(j < str2.length() / i) continue;
            
            ans = temp;
        }
        
        return ans;
    }
};
