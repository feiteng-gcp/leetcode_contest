class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        if (str1.size()<str2.size()) {
            return gcdOfStrings(str2, str1);
        }
        string ret;
        for (int l=1; l<=str2.size(); l++) {
            if (str2.size()%l!=0 || str1.size()%l!=0) {
                continue;
            }
            bool ok=true;
            for (int i=0; i<str1.size(); i++) {
                if (str1[i]!=str2[i%l]) {
                    ok=false;
                    break;
                }
            }
            for (int i=0; i<str2.size(); i++) {
                if (str2[i]!=str2[i%l]) {
                    ok=false;
                    break;
                }
            }
            if (ok) {
                ret=str2.substr(0, l);
            }
        }
        return ret;
    }
};
