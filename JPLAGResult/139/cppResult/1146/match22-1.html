class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int i=0;
        string ans="";
        string res="";
        int s1=str1.size(),s2=str2.size();
        while(i<min(s1,s2)&&str1[i]==str2[i])
        {
            ans+=str1[i];i++;
            if(str1.size()%ans.size()!=0||str2.size()%ans.size()!=0)continue;

            bool match=true;
            for(int j=0;j<str1.size();j+=ans.size()){
            if(str1.find(ans,j)!=j){match=false;break;}
            }

            for(int j=0;j<str2.size();j+=ans.size()){
            if(str2.find(ans,j)!=j){match=false;break;}
            }
            if(match==true)res=ans;
            
        }
        

        return res;
    }
};
