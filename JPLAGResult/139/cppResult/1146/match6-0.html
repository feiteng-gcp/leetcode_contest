class Solution {
   public:
    int n1, n2;
    string gcdOfStrings(string str1, string str2) {
        n1 = str1.size();
        n2 = str2.size();

        if(n1 < n2){
            swap(str1, str2);
            swap(n1, n2);
        }

        int len;
        for(len = 0; len < n2; ++len){
            if(str1[len] != str2[len]){
                break;
            }
        }

        for(int i = len; i > 0; --i){
            if(n1 % i || n2 % i) continue;
            bool success = true;
            for(int j = 0; j < n1; ++j){
                if(str1[j] != str1[j%i]) {
                    success = false;
                    break;
                }
            }
            if(!success) continue;
            for(int j = 0; j < n2; ++j){
                if(str2[j] != str2[j%i]) {
                    success = false;
                    break;
                }
            }
            if(!success)continue;
            return str1.substr(0, i);
        }
        return "";
    }
};
