class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        string tmp;
        int n = min(str1.size(),str2.size());

        for(int i=n,j;i>=1;i--){
            if(str1.size()%i||str2.size()%i)continue;
            tmp = str1.substr(0,i);
            for(j=0;j<str1.size();j++){
                if(tmp[j%i]!=str1[j])break;
            }
            if(j<str1.size())continue;
            for(j=0;j<str2.size();j++){
                if(tmp[j%i]!=str2[j])break;
            }
            if(j>=str2.size())return tmp;
        }
        return "";
    }
};
