class Solution {
public:
    bool check(const string& s, const string& t) {
        int n = s.size(), m = t.size();
        for (int i = 0; i < m; i += n) {
            if (s != t.substr(i, n)) return false;
        }
        return true;
    }
    string gcdOfStrings(string s, string t) {
        int n = s.size(), m = t.size();
        string ret;
        for (int k = 1; k <= n && k <= m; ++k) {
            if (n % k || m % k) continue;
            auto x = s.substr(0, k);
            auto y = t.substr(0, k);
            if (x == y && check(x, s) && check(y, t)) ret = x;
        }
        return ret;
    }
};
