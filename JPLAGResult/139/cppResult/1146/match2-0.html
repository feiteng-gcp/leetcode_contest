class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int len1 = str1.length();
        int len2 = str2.length();
        for(int l = min(len1, len2); l >=1; l--) {
            if(len1 % l != 0 || len2 % l != 0) continue;  
            string common = str1.substr(0, l);
            if(str2.substr(0,l) != common) continue;
            bool isValid = true;
            for(int i = 1; i < len1 / l; i++) {
                string tmp = str1.substr(l*i, l);
                if(common != tmp)  {
                    isValid = false;
                    break;
                }
            }
            for(int i = 1; i < len2 / l; i++) {
                string tmp = str2.substr(l*i, l);
                if(common != tmp)  {
                    isValid = false;
                    break;
                }
            }
            if(isValid) return common;
        }
        return string();
    }
    
};
