class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int n = str1.size(), m = str2.size();
        if (n == m && str1 == str2)
            return str1;
        int g = __gcd(n, m);
        if (n > m) {
            swap(str1, str2);
            swap(n, m);
        }

        for (int i = 1; i <= g; i++) {
            if (n % (i * g) != 0 || m % (i * g) != 0)
                continue;
            bool f = true;
            for (int j = 0; j < n; j++) {
                if (str1[j] != str1[j % (i * g)]) {
                    f = false;
                    break;
                }
            }

            for (int j = 0; j < m; j++) {
                if (str2[j] != str1[j % (i * g)]) {
                    f = false;
                    break;
                }
            }
            if (f)
                return str1.substr(0, i * g);
        }
        return "";
    }
};
