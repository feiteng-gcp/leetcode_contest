class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int size_small = str1.size(), size_large = str2.size();
        string short_string = str1, long_string = str2;
        if(size_small > size_large){
            swap(size_small, size_large); swap(short_string, long_string);
        }
        vector<string> divisor;
        for(int i = 1; i <= size_small; ++i){
            string fragment = short_string.substr(0, i);
            bool if_divisor = true;
            for(int j = i; j < size_small; j += i){
                if(fragment != short_string.substr(j, i)){
                    if_divisor = false; break;
                }
            }
            if(if_divisor) divisor.push_back(fragment);
        }
        for(int i = divisor.size()-1; i >= 0; --i){
            int step = divisor[i].size();
            bool if_divisor = true;
            for(int j = 0; j < size_large; j += step){
                if(divisor[i] != long_string.substr(j, step)){
                    if_divisor = false; break;
                }
            }
            if(if_divisor) return divisor[i];
        }
        return "";
    }
};
