class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int n = str1.size();
        int m = str2.size();
        int p = min(n,m);
        string res;
        for (int i=1; i<=p; ++i) {
            if (n%i || m%i) continue;
            bool ok = true;
            for (int j=0; j<n; ++j) {
                if (str1[j]!= str1[j%i]) {
                    ok = false; break;
                }
            }
            for (int j=0; j<m; ++j) {
                if (str2[j]!= str1[j%i]) {
                    ok = false; break;
                }
            }
            if (ok) res = str1.substr(0,i);
        }
        return res;
    }
};
