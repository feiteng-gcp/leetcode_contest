class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        string r;
        string t;
        for(int i=1;i<=str1.length() && i<=str2.length();i++){

            if(str1.length()%i!=0 || str2.length()%i!=0) continue;
            t=str1.substr(0,i);
            int sign=0;
            for(int j=0;j<str1.length();j+=i){
                if(str1.substr(j,i)!=t) {sign++;break;}
            }
            if(sign>0) continue;
            for(int j=0;j<str2.length();j+=i){
                if(str2.substr(j,i)!=t) {sign++;break;}
            }
            if(sign==0) r=t;
        }
        return r;
    }
};
