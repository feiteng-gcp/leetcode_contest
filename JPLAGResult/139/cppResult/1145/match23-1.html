class Solution {
int subCount(int arr[], int n, int target) 
{
    int k = target;

    unordered_map<int, int> res;
  
    int result = 0;
    int cumSum = 0; 
    for (int i = 0; i < n; i++) { 
        cumSum += arr[i]; 
        if(res.count(cumSum - k)) result += res.at(cumSum - k);
        if(cumSum == k) result++;
        res[cumSum]++;
    } 
    return result; 
} 
  
int countSubmatrix(vector<vector<int>>& mat, int m, int n, int target) 
{ 
    int tot_count = 0;   
    int left, right, i; 

    int temp[n]; 
  
    for (left = 0; left < m; left++) { 
        memset(temp, 0, sizeof(temp)); 
        for (right = left; right < m; right++) { 
            for (i = 0; i < n; ++i) 
                temp[i] += mat[i][right]; 
            
            tot_count += subCount(temp, n, target); 
        } 
    }
    return tot_count; 
} 
  
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        return countSubmatrix(matrix, matrix[0].size(), matrix.size(), target);
    }
};
