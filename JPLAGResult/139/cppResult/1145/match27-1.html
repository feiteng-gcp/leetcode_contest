class Solution {
public:
    int col[330][330],pre[330];
    unordered_map<int,int>mp;
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {

        int n = matrix.size(),m = matrix[0].size();
        for(int i=1;i<=m;i++){
            for(int j=1;j<=n;j++){
                col[i][j] = matrix[j-1][i-1]+col[i][j-1];
            }
        }

        int ans = 0;
        for(int i=1;i<=n;i++){
            for(int j=i;j<=n;j++){
                for(int k=1;k<=m;k++){
                    pre[k] = pre[k-1]+col[k][j]-col[k][i-1];
                }
                mp.clear();
                mp[0] = 1;
                for(int k=1;k<=m;k++){
                    if(mp.count(pre[k]-target))ans += mp[pre[k]-target];
                    mp[pre[k]]++;
                }
            }
        }
        return ans;
    }
};
