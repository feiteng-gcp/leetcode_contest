class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& v, int target) {

        int n = v.size(), m = v[0].size();
        vector<vector<int>> s(n, vector<int>(m, 0));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                s[i][j] += v[i][j];
                if (i) s[i][j] += s[i - 1][j];
                if (j) s[i][j] += s[i][j - 1];
                if (i && j) s[i][j] -= s[i - 1][j - 1];
            }
        }
        
        int ans = 0;

        for (int c1 = 0; c1 < m; ++c1) {
            for (int c2 = c1; c2 < m; ++c2) {
                map<int, int> f;
                f[0]++;
                for (int r = 0; r < n; ++r) {
                    int sum = s[r][c2];
                    if (c1) sum -= s[r][c1 - 1];
                    ans += f[sum - target];
                    f[sum]++;
                }
            }
        }
        return ans;
    }
};
