class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> sums(m + 1, vector<int>(n, 0));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                sums[i+1][j] = sums[i][j] + matrix[i][j];
            }
        }

        int ret = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = i + 1; j <= m; ++j) {
                unordered_map<int, int> scm;
                int sum = 0;
                for (int k = 0; k < n; ++k) {
                    sum += sums[j][k] - sums[i][k];
                    if (sum == target) {
                        ++ret;
                    }
                    if (scm.count(sum - target) != 0) {
                        ret += scm[sum - target];
                    }
                    ++scm[sum];
                }
            }
        }
        return ret;
    }
};
