class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int m = matrix.size();
        int n = matrix[0].size();
        vector<vector<int>> sum(m, vector<int>(n, 0));
        sum[0][0]=matrix[0][0];
        for(int i=1;i<m;i++)
            sum[i][0] = sum[i-1][0] + matrix[i][0];
        for(int j=1;j<n;j++)
            sum[0][j] = sum[0][j-1] + matrix[0][j];
        for(int i=1;i<m;i++)
            for(int j=1;j<n;j++)
                sum[i][j] = sum[i-1][j] + sum[i][j-1] - sum[i-1][j-1] + matrix[i][j];
        
        long long ans=0;
        for(int w=1;w<=n;w++) {
            vector<int> s(m, 0);
            for(int j=w-1;j<n;j++) {
                for(int i=0;i<m;i++)
                    s[i] = sum[i][j] - (j>=w?sum[i][j-w]:0);
                
                // cout<<"sum(w="<<w<<"): "<<endl;
                // for(int i=0;i<m;i++)
                //     cout<<s[i]<<" ";
                // cout<<endl;
                
                unordered_map<int, int> cnt;
                cnt[0]=1;
                for(int i=0;i<m;i++) {
                    if(cnt.count(s[i]-target))
                        ans+=cnt[s[i]-target];
                    cnt[s[i]]++;
                }
            }
        }
        
        return ans;
    }
};
