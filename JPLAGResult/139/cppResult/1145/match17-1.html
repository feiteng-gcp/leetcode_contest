long long p[400][400];

class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int n = matrix.size(), m = matrix[0].size();

        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                p[i][j] = p[i][j - 1] + matrix[i - 1][j - 1];
            }
        }
        int ans = 0;
        for (int j1 = 1; j1 <= m; j1++) {
            for (int j2 = j1; j2 <= m; j2++) {
                unordered_map<long long, int> mp;
                mp[0] = 1;

                long long cnt = 0;
                for (int i = 1; i <= n; i++) {
                    cnt += p[i][j2] - p[i][j1 - 1];
                    ans += mp[cnt - target];
                    // printf("%d %d %d [%d] %d\n", j1, j2, i, cnt, mp[cnt - target]);
                    mp[cnt]++;
                }
            }
        }
        return ans;
    }
};
