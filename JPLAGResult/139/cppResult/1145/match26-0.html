class Solution {
public:
	int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
		int h = matrix.size();
		int w = matrix[0].size();

		vector<vector<long long>> sum(h, vector<long long >(w + 1, 0));
		for (int y = 0; y < h; y++) {
			for (int x = 0; x < w; x++) {
				sum[y][x + 1] = sum[y][x] + matrix[y][x];
			}
		}

		int ret = 0;
		for (int x1 = 0; x1 < w;x1++) {
			for (int x2 = x1; x2 < w; x2++) {
				unordered_map<long long, int> cnt;
				cnt.reserve(h + 1);
				cnt[0] = 1;
				long long s = 0;
				for (int y = 0; y < h; y++) {
					s += sum[y][x2 + 1] - sum[y][x1];
					auto it = cnt.find(s - target);
					if (it != cnt.end()) {
						ret += it->second;
					}
					++cnt[s];
				}
			}
		}

		return ret;
	}
};
