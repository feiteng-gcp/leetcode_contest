class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& mat, int target) {
        int r=mat.size();
        int c=mat[0].size();
        vector<vector<int>> sum(r+1,vector<int>(c+1,0));
        int dSize = 20;
        for(int i=0;i<r;i++) {
            for(int j=0;j<c;j++) {
                sum[i+1][j+1] = sum[i+1][j] + mat[i][j];
            }
        }
        vector<int> s(r+1,0);
        unordered_map<int,int> hmap;
        int res=0;
        for(int c1=1;c1<=c;c1++) {
            for(int c2=c1;c2<=c;c2++) {
                for(int i=1;i<=r;i++) {
                    s[i] = s[i-1] + sum[i][c2] - sum[i][c1-1];
                }
                
                hmap.clear();
                hmap[0]=1;
                
                for(int i=1;i<=r;i++) {
                    if(hmap.count(s[i]-target)) {
                        res+=hmap[s[i]-target];
                    }
                    hmap[s[i]]++;
                }
            }
        }
        return res;
    }
};
