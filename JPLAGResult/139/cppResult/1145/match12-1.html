class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int M = matrix.size();
        int N = matrix[0].size();
        vector<vector<int>> preSum(M + 1, vector<int>(N + 1, 0));

        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                preSum[i + 1][j + 1] = preSum[i + 1][j] + preSum[i][j + 1] - preSum[i][j] + matrix[i][j];
            }
        }
        int count = 0;
        for (int i = 0; i < M; ++i) {
            for (int j = i + 1; j <= M; ++j) {
                unordered_map<int, int> check;
                for (int k = 0; k <= N; ++k) {
                    int sum = preSum[j][k] - preSum[i][k];

                    int prev = sum - target;
                    if (check.find(prev) != check.end()) {
                        count += check[prev];
                    }
                    ++check[sum];
                }
            }
        }
        return count;
    }
};
