#define N 305
int x[N][N];
int y[N][N][N];
unordered_map<int, int> cnt;

class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& ma, int t) {
        memset(x, 0, sizeof(x));
        memset(y, 0, sizeof(y));
        int n = ma.size();
        int m = ma[0].size();
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                x[i][j] = j ? x[i][j - 1] + ma[i][j] : ma[i][0];
                for (int k = 1; k <= j + 1; ++k) {
                    y[i][j][k] = k == j + 1 ? x[i][j] : x[i][j] - x[i][j - k];
                }
            }
        }
        
        int res = 0;
        for (int k = 1; k <= m; ++k) {
            for (int j = k - 1; j < m; ++j) {
                cnt.clear();
                cnt[0] = 1;
                int sum = 0;
                for (int i = 0; i < n; ++i) {
                    sum += y[i][j][k];
                    res += cnt[sum - t];
                    cnt[sum]++;
                }
            }
        }
        
        return res;
    }
};
