    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int x=matrix.size();
        int y=matrix[0].size();
        int ret=0;
        for (int i=0; i<x; i++) {
            vector<int> xsum(y, 0);
            for (int j=i; j<x; j++) {
                unordered_map<int, int> s;
                s[0]=1;
                int ss=0;
                for (int k=0; k<y; k++) {
                    xsum[k]+=matrix[j][k];
                    ss+=xsum[k];
                    ret+=s[ss-target];
                    s[ss]++;
                }
            }
            for (int k=0; k<y; k++) {
                xsum[k]-=matrix[i][k];
            }
        }
        return ret;
    }
};
