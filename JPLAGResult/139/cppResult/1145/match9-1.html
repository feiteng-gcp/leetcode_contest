class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int M = matrix.size();
        int N = matrix[0].size();
        vector<vector<int>> sum_mat(M+1, vector<int>(N+1, 0));
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                sum_mat[i+1][j+1] = matrix[i][j] +  sum_mat[i+1][j] + sum_mat[i][j+1] - sum_mat[i][j];
            }
        }
        
        int count = 0;
        
        for (int k1 = 0; k1 < M+1; k1++) {
            for (int k2 = k1 + 1; k2 < M+1; k2++) {
                unordered_map<int, int> m;
                for (int i = 0; i < N+1; i++) {
                    int expect = sum_mat[k2][i] - sum_mat[k1][i] - target;
                    count += m[expect];
                    m[sum_mat[k2][i] - sum_mat[k1][i]]++;
                }
            }
        }
        return count;
    }
};
