class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& g, int target) {
        int n = g.size(), m = g[0].size(), ans = 0;
        vector<vector<int>> s(n+1, vector<int>(m+1));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j)
                s[i+1][j+1] = s[i+1][j] + g[i][j];
            
            for (int j = 1; j <= m; ++j)
                s[i+1][j] += s[i][j];
        }
        
        for (int i = 1; i <= n; ++i) {
            for (int a = 0; a < i; ++a) {
                unordered_map<int, int> cnt;
                ++cnt[s[i][0] - s[a][0]];
                
                for (int j = 1; j <= m; ++j) {
                    int v = s[i][j] - s[a][j] - target;
                    if (cnt.count(v))
                        ans += cnt[v];
                    
                    ++cnt[s[i][j] - s[a][j]];
                }
            }
        }
        
        return ans;
    }
};
