class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& v, int target) {
        int n = v.size();
        int m = v[0].size();
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            int arr[305] = {0};

            for (int k = i; k < n; k++) {
                for (int j = 0; j < m; j++) {
                    arr[j] += v[k][j];
                }
                
                int cs = 0;
                unordered_map<int, int> freqCnt;
                freqCnt[0] = 1;
                for (int j = 0; j < m; j++) {
                    cs += arr[j];

                    int diff = cs - target;
                    if (freqCnt.find(diff) != freqCnt.end()) {
                        res += freqCnt[diff];
                    }
                    freqCnt[cs]++;
                }                
            }
        }
        return res;
    }
};
