class Solution {
 public:
  int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
    int m = matrix.size(), n = matrix[0].size();
    vector<vector<int>> sum(m + 1, vector<int>(n + 1));
    for (int i = 1; i <= m; ++i) {
      for (int j = 1; j <= n; ++j) {

        sum[i][j] = sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1] +
                    matrix[i - 1][j - 1];
      }
    }
    int ans = 0;
    for (int top = 0; top < m; ++top) {
      for (int bot = top; bot < m; ++bot) {
        map<int, int> mp;
        mp[0] = 1;
        for (int j = 0; j < n; ++j) {
          int pref = get_sum(sum, top, bot, 0, j);
          int need = pref - target;
          if (mp.find(need) != mp.end()) {
            ans += mp[need];
          }
          ++mp[pref];
        }
      }
    }
    return ans;
  }

 private:
  int get_sum(const vector<vector<int>>& sum, int top, int bot, int lft,
              int rgt) {
    return sum[bot + 1][rgt + 1] - sum[bot + 1][lft] - sum[top][rgt + 1] +
           sum[top][lft];
  }
};
