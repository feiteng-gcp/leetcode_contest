class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int n=matrix.size(),m=matrix[0].size();
        vector<vector<int>> sum(m,vector<int>(n));
        for(int j=0;j<m;j++){
            for(int i=0;i<n;i++){
                sum[j][i]=i==0?matrix[i][j]:sum[j][i-1]+matrix[i][j];
            }
        }
        int64_t sol = 0;
        for(int i=0;i<n;i++){
            for(int j=i;j<n;j++){
                map<int64_t,int> mm;
                mm[0]=1;
                int64_t pre=0;
                for(int k=0;k<m;k++){
                    pre=pre+(sum[k][j]-(i==0?0: sum[k][i-1]));
                    sol+=mm[pre-target];
                    mm[pre]++;
                }
            }
        }
        return sol;
    }
};
