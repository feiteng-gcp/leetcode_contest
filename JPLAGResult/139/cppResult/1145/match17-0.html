class Solution {
public:

    int numSubmatrixSumTarget(vector<vector<int>>& m, int target) {
        int r = m.size();
        int c = m[0].size();
        
        vector<vector<int>> dp(r + 1, vector<int>(c + 1, 0));
        //  prefix sum for each col
        for (int i = 0; i < c; ++i) {
            for (int j = 1; j < r + 1; ++j) {
                dp[j][i] = dp[j - 1][i] + m[j - 1][i];
            }
        }
        int sol = 0;
        for (int sr = 0; sr < r; ++sr) {
            for (int er = sr; er < r; ++er) {
                unordered_map<int, int> s;
                s[0] = 1;
                int tmp = 0;
                for (int k = 0; k < c; ++k) {
                    tmp += dp[er + 1][k] - dp[sr][k];
                    sol += s[tmp - target];
                    s[tmp]++;
                }
            }
        }
        
        return sol;
    }
};
