class Solution {
public:
    int numSubmatrixSumTarget(vector<vector<int>>& matrix, int target) {
        int n = sz(matrix), m = sz(matrix[0]), flag = 0;
        if (n < m) flag = 1, swap(n, m);
        vector<vector<int> > sum(n, vector(m + 1, 0));
        rep(i, 0, n) {
        	rep(j, 1, m + 1) if (!flag) 
        		sum[i][j] = sum[i][j - 1] + matrix[i][j - 1];
        	else
        		sum[i][j] = sum[i][j - 1] + matrix[j - 1][i];
        }
        int ans = 0;
        rep(l, 1, m + 1) rep(r, l, m + 1) {
        	map<int, int> R;
        	R[0] = 1;
        	int pre = 0;
        	rep(i, 0, n) {
        		pre += sum[i][r] - sum[i][l - 1];
        		if (R.count(pre - target))
        			ans += R[pre - target];
        		++R[pre];
        	}
        }
        return ans;
    }
};
