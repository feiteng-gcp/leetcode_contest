class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
       map<string,int> mm;
        int n=matrix.size(),m=matrix[0].size();
        for(int i=0;i<n;i++){
            string t1="",t2="";
            for(int j:matrix[i]){
                t1+=to_string(j);
                t2+=to_string(1^j);
            }
            if(mm.count(t1)){
                mm[t1]++;
            }else if(mm.count(t2)){
                mm[t2]++;
            }else {
                mm[t1]++;
            }
        }
        int sol=0;
        for(auto e:mm){
            sol=max(sol,e.second);
        }
        return sol;
    }
};
