class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int H = matrix.size();
        int W = matrix[0].size();
        int ret = 1;
        for (int i=0; i<H; i++) {
            int r = 1;
            for (int j=0; j<H; j++) {
                if (i == j) continue;
                bool f = true;
                for (int k=0; k<W; k++) {
                    if ((matrix[i][0] ^ matrix[i][k] ^ matrix[j][k]) != matrix[j][0]) {
                        f = false;
                        break;
                    }
                }
                if (f) r++;
            }
            ret = max(r, ret);
        }
        return ret;
    }
};
