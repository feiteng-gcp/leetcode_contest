class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int n=matrix.size();
        int m=matrix[0].size();
        bitset<333> M[333];
        for(int i=0;i<n;i++) {
            for(int j=0;j<m;j++) {
                M[i][j]=(matrix[i][j]^matrix[i][0]);
            }

        }
        
        int ans=0;
        for(int i=0;i<n;i++) {
            int ct=0;
            for(int j=0;j<n;j++) {
                ct += ((M[i]^M[j]).count()==0);
            }
            ans=max(ans,ct);
        }
        return ans;
    }
};
