class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& v) {
        int ans = 0;
        map<vector<int>, int> mep; 
        for(int i = 0; i < v.size(); i++) {
            vector<int> v2 = v[i];
            for(int i = 0; i < v2.size(); i++) v2[i] = !v2[i];
            mep[v[i]]++;
            ans = max(ans, mep[v[i]]);
            mep[v2]++;
            ans = max(ans, mep[v2]);
        }
        return ans;
    }
};
