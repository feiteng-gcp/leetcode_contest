class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& A) {
        int n=A.size(),m=A[0].size();
        int i,j,k,a,b,cur,ans=0;
        for(i=0;i<n;i++)
        {
            cur=0;
            for(j=i+1;j<n;j++){
                for(k=0;k<m;k++)
                    if(A[i][k]!=A[j][k]) break;
                if(k==m) cur++;
                else {
                    for(k=0;k<m;k++)
                        if(A[i][k]==A[j][k]) break;
                    if(k==m) cur++;
                }
            }
            if(cur>ans) ans=cur;
        }
        return ans+1;
    }
};
