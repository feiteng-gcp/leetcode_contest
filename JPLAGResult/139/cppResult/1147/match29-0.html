class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int const cols = matrix.front().size();
        int const rows = matrix.size();
        vector<int> tmp(cols);
        auto is_equal = [&tmp, &matrix, cols](int r1, int r2)->int {
            if (r1 == r2) return 1;
            bool eq = (matrix[r1][0] == matrix[r2][0]);
            for (int j = 1; j < cols; j++) {
                bool eq1 = (matrix[r1][j] == matrix[r2][j]);
                if (eq1 != eq) return 0;
            }
            return 1;
        };
        int res = 0;
        for (int i = 0; i < rows; i++) {
            int acc = 0;
            for (int j = 0; j < rows; j++) {
                acc += is_equal(i, j);
            }
            res = max(res, acc);
        }
        return res;
    }
};
