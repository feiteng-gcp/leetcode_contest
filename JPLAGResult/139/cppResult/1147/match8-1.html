class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        using BIT = bitset<320>;
        unordered_map<BIT, int> cnts;
        
        int n = matrix.size();
        if (n == 0) {
            return 0;
        }
        int m = matrix[0].size();
        if (m == 0) {
            return 0;
        }

        for (int i = 0; i < n; ++i) {
            BIT bits;
            BIT bits2;
            for (int j = 0; j < m; ++j) {
                if (matrix[i][j] == 0) {
                    bits.set(j);
                } else {
                    bits2.set(j);
                }
            }
            ++cnts[bits];
            ++cnts[bits2];
        }
        
        int res = 0;
        for (auto& p : cnts) {
            res = max(res, p.second);
        }
        return res;
    }
};
