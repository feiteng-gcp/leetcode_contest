class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        map<vector<int>, int> cnt;
        for(int i = 0 ; i < matrix.size() ; i++){
            int c = matrix[i][0];
            for(int j = 0 ; j < matrix[i].size() ; j++){
                matrix[i][j] ^= c;
            }
            cnt[matrix[i]]++;
        }
        int ans = 0;
        for( auto i : cnt ){
            ans = max(ans, i.second);
        }
        return ans;
    }
};
