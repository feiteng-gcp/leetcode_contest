class Solution {
public:
    bool check(vector<vector<int>> & matrix, vector<int> & sums, int i, int j){
        if(sums[i] + sums[j] != matrix[0].size() && sums[i] != sums[j]) return false;
        bool is_xnor = matrix[i][0] == matrix[j][0] ? false : true;
        for(int k = 1; k < matrix[0].size(); ++k){
            if(matrix[i][k] == matrix[j][k]){
                if(is_xnor) return false;
            }else{
                if(!is_xnor) return false;
            }
        }
        return true;
    }
    
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int maxv = 0;
        vector<int> sums(matrix.size(), 0);
        for(int i = 0; i < matrix.size(); ++i){
            for(int j = 0; j < matrix[i].size(); ++j){
                sums[i] += matrix[i][j];
            }
        }
        for(int i = 0; i < matrix.size(); ++i){
            int oval = 0;
            for(int j = 0; j < matrix.size(); ++j){
                if(i == j) {
                    ++oval;
                    continue;
                }
                if(check(matrix, sums, i, j)){
                    ++oval;
                }
            }
            maxv = max(maxv, oval);
        }
        return maxv;
    }
};
