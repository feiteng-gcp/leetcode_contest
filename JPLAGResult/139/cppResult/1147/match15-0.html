class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        unordered_map<string, int> rec;
        int res = 1;
        int m = matrix.size(), n = matrix[0].size();
        for (int i = 0; i < m; ++i)
        {
            bool flip = matrix[i][0] == 1;
            string s;
            for (int j = 0; j < n; ++j)
            {
                s += flip ? 1 - matrix[i][j] : matrix[i][j];
            }
            rec[s]++;
            res = max(res, rec[s]);
        }
        return res;
    }
};
