class Solution {   
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& mat) {
        int r=mat.size();
        int c=mat[0].size();
        int res=0;
        vector<int> col(c);
        for(int i=0;i<r;i++) {
            //make zero
            for(int j=0;j<c;j++) {
                col[j]=mat[i][j]==0;
            }
            int tot=0;
            for(int j=0;j<r;j++) {
                int totOne=0;
                for(int k=0;k<c;k++) {
                    totOne+=mat[j][k]^col[k];
                }
                tot+=(totOne==0 || totOne==c);
            }
            res=max(res,tot);
        }
        return res;
    }
};
