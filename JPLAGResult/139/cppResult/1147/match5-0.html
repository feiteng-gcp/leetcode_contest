class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        map<vector<int>, int> F;
        for (int i = 0; i < n; ++ i)
        {
            if (a[i][0] == 1)
            {
                for (int j = 0; j < m; ++ j)
                    a[i][j] ^= 1;
            }
            F[a[i]] ++;
        }
        int ans = 0;
        for (auto p : F)
            ans = max(ans, p.second);
        return ans;
    }
};
