class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int x=0;

        auto m= matrix;
        
        for(int i=0; i<m.size(); ++i){
            int k=0;
            for(int j=i+1; j<m.size(); ++j){
                if(m[i]==m[j]||m[i]==flip(m[j])) ++k;
            }
            x= max(x, k);
        }
        return x+1;
        
    }
    
    vector<int> flip(vector<int> v){
        
        for(auto &x :v) x= 1-x;
        return v;
        
    }
    
    
};
