class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        map<string,int> mp;
    int rst=0;
    for(int i=0;i<matrix.size();i++){
        string s="",r="";
        for(int j=0;j<matrix[i].size();j++){
            s.push_back(matrix[i][j]+'0');
            r.push_back((matrix[i][j]+1)%2+'0');
        }
        if(mp.count(s)){    
            mp[s]++;
            if(mp[s]>rst) rst=mp[s];
        }
        else if(mp.count(r)){
            mp[r]++;
            if(mp[r]>rst) rst=mp[r];
        }
        else{
            mp[s]++;
            if(mp[s]>rst) rst=mp[s];
        }
    }
    return rst;
    }
};
