class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int ans = 0;
        auto n = matrix.size();
        for (int i = 0; i < n; ++i) {
            int cc = 1;
            for (int j = 0; j < i; ++j) {
                int c1 = 0, c2 = 0;
                for (int k = matrix[i].size() - 1; k >= 0 && (!c1 || !c2); --k) {
                    if (matrix[i][k] == matrix[j][k]) {
                        ++c1;
                    } else {
                        ++c2;
                    }
                }
                if (!c1 || !c2) {
                    cc += 1;
                }
            }
            ans = max(ans, cc);
        }
        return ans;
    }
};
