class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        
        int ans = 0;
        for(int i = 0; i < m; ++i) {
            for(int j = 1; j < n; ++j) {
                if (matrix[i][0] != matrix[i][j]) {
                    for (int k = 0; k < m; ++k)
                        matrix[k][j] = 1 - matrix[k][j];
                }
            }
            int cnt = 0;
            for(int k = 0; k < m; ++k) {
                int sum = 0;
                for(int j = 0; j < n; ++j) sum += matrix[k][j];
                if (sum == 0 || sum == n) cnt += 1;
            }
            ans = max(ans, cnt);
                
        }
        return ans;
    }
};
