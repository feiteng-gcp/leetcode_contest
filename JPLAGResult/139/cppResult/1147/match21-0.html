class Solution {
public:
    int isSame(vector<int> &mA, vector<int> &mB) {
        bool blFlip = (mA[0] != mB[0]);
        for (int i = 0; i < mA.size(); i++) {
            if ((mA[i] != mB[i]) ^ blFlip) {
                return 0;
            }
        }
        return 1;
    }
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int maxSame = 0;
        for (int i = 0; i < (matrix.size() - 1); i++) {
            int sameCount = 0;
            for (int j = (i + 1); j < matrix.size(); j++) {
                sameCount += isSame(matrix[i], matrix[j]);
            }
            maxSame = (maxSame < sameCount) ? sameCount : maxSame;
        }
        return maxSame + 1;
    }
};
