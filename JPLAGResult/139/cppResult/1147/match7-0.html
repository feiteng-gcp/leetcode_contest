class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& a) {
        int m=a.size(),n=a[0].size();
        int i,j,k,ans=1,t;
        for (i=0;i<m;i++) {
            t=1;
            for (j=i+1;j<m;j++) {
                for (k=0;k<n;k++)
                    if (a[i][0]^a[j][0]!=a[i][k]^a[j][k])
                        break;
                if (k==n)
                    t++;
            }
            ans=max(ans,t);
        }
        return ans;
    }
};
