class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& a) {

        int n = a.size(), m = a[0].size();
        map<vector<int>, int> A;
        for (int i = 0; i < n; ++i) {
            if (a[i][0]) {
                for (int j = 0; j < m; ++j) a[i][j] = 1 - a[i][j];
            }
            A[a[i]]++;
        }
        int ret = 0;
        for (auto& it : A) ret = max(ret, it.second);
        return ret;
    }
};
