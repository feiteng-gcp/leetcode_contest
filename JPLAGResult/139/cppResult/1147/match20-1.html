typedef unsigned long long ull;

class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int n=matrix.size();
        int m=matrix[0].size();
        int ret=0;
        ull a[307], b[307];
        for(int i=0; i<n; i++) {
            a[i]=b[i]=0;
            for(int j=0; j<m; j++) {
                a[i]=a[i]*2333+matrix[i][j]+4654645;
                b[i]=b[i]*2333+(matrix[i][j]^1)+4654645;
            }
        }

        for(int i=0; i<n; i++) {
            // cout<<a[i]<<' '<<b[i]<<endl;
            int ans=1;
            for(int j=i+1; j<n; j++) {
                if (a[i]==a[j]||a[i]==b[j]) ans++;
            }
            ret=max(ret, ans);
        }
        return ret;
    }
};
