class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        for(int i=0; i<m; i++){
            if(matrix[i][0]){
                for(int j=0; j<n; j++)
                    matrix[i][j] = 1-matrix[i][j];
            }
        }
        sort(matrix.begin(), matrix.end());
        int dup = 1;
        int ans = 1;
        for(int i=1; i<m; i++){

            bool sameQ = true;
            for(int j=0; j<n; j++){
                if(matrix[i][j]!=matrix[i-1][j]){
                    sameQ = false;
                    break;
                }
            }
            if(sameQ){
                dup++;
                ans = max(ans, dup);
            }
            else{
                dup = 1;
            }
        }
        return ans;
    }
};
