class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        unordered_map<vector<int>, int, HashVector> cnts;
        for (int i = 0; i < matrix.size(); ++i) {
            vector<int> ones;
            vector<int> zeros;
            for (int j = 0; j < matrix[i].size(); ++j) {
                if (matrix[i][j] == 0) {
                    zeros.emplace_back(j);
                } else {
                    ones.emplace_back(j);
                }
            }
            ++cnts[ones];
            ++cnts[zeros];
        }
        int result = 0;
        for (const auto &item : cnts) {
            result = max(result, item.second);
        }
        return result;
    }
private:
    struct HashVector{
        size_t operator()(const vector<int> &nums) const {
            size_t result = 0;
            for (int i = 0; i < nums.size(); ++i) {
                result = 31 * result + hash<size_t>()(nums[i]);
            }
            return result;
        }
    };
};
