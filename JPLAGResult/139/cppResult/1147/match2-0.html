class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        map<vector<int>, int> m;
        for (int i = 0; i < matrix.size(); i++) {
            vector<int> column;
            vector<int> column2;
            for (int j = 0; j < matrix[0].size(); j++) {
                column.push_back(matrix[i][j]);
                column2.push_back(1 - matrix[i][j]);
            }
            m[column]++;
            m[column2]++;
        }
        int res = 0;
        for (auto&& kv : m) {
            res = max(res, kv.second);
        }
        return res;
    }
};
