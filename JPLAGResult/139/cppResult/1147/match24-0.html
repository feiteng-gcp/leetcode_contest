class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        int ans = 0;
        for (int p = 0; p < n; p++) {
            int cnt = 0;
            for (int i = 0; i < n; i++) {
                int t = 0;
                for (int j = 0; j < m; j++) {
                    t += matrix[i][j] ^ matrix[p][j];
                }
                cnt += (t == 0 || t == m);
            }
            ans = max(ans, cnt);
        }
        return ans;
    }
};
