class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        map<string, int> res;
        int n = matrix.size();
        int m = matrix[0].size();
        for (int i=0; i<n; ++i) {
            string a, b;
            for (int j=0; j<m; ++j) {
                if (matrix[i][j] ==0) {
                    a.push_back('0');
                    b.push_back('1');
                }
                else {
                    a.push_back('1');
                    b.push_back('0');
                }
            }
            res[a]++;
            res[b]++;
        }
        int r = 0;
        for (auto x: res) {
            r = max(r, x.second);
        }
        return r;
    }
};
