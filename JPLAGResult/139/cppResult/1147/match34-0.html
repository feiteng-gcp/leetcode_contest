class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        unordered_map<string, int> m;
        int mx = 0;
        for(int i = 0; i < matrix.size(); ++i) {
            string s1 = "";
            string s2 = "";
            for(auto v: matrix[i]) {
                s1 += string(1, '0' + v);
                s2 += string(1, '0' + (v ^ 1));
            }
            m[s1]++, m[s2]++;
            if(m[s1] > mx) mx = m[s1];
            if(m[s2] > mx) mx = m[s2];
        }
        return mx;
    }
};
