class Solution {
public:
	int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
		int ret = 0;
		int w = matrix[0].size();

		for (int y = 0; y < matrix.size(); y++) {
			for (int b = 0; b < 2;b++) {
				vector<int> flip(w, 0);

				for (int x = 0; x < w; x++) {
					if (matrix[y][x] != b)flip[x] = 1;
				}

				int cnt = 0;
				for (int y2 = 0; y2 < matrix.size(); y2++) {
					bool ok = true;
					int v = matrix[y2][0] ^ flip[0];

					for (int x = 0; x < w; x++) {
						if ((matrix[y2][x] ^ flip[x]) != v) {
							ok = false;
							break;
						}
					}
					if (ok)cnt++;
				}
				ret = max(ret, cnt);
			}

		}


		return ret;
	}
};
