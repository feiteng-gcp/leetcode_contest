class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& m) {
        int r = m.size();
        int c = m[0].size();
        int res = 0;
        // record diff
        vector<int> s(r + 1, 0);
        for (int i = 0; i < r; ++i) {
            // assume row i is the same
            if (s[i]) continue;
            vector<vector<int>> t = m;
            
            // flip first
            for (int j = 0; j < c; ++j) {
                if (m[i][j] != 0) {
                    // flip
                    for (int k = 0; k < r; ++k) {
                        t[k][j] = !t[k][j];
                    }
                }

            }
            
            // check same row
            int tmp = 0;
            for (int j = 0; j < r; ++j) {
                int valid = 1;
                for (int k = 1; k < c; ++k) {
                    if (t[j][k] != t[j][k - 1]) {
                        valid = 0;
                        break;
                    }
                }
                tmp += valid;
                if (valid) s[j] = 1;
            }
            res = max(res, tmp);
        }
        
        return res;
    }
};
