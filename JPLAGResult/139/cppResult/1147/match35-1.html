class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {

        int mask[305];
        int res = 0;
        
        for (int i = 0; i < matrix.size(); ++i) {
            for (int j = 0; j < matrix[i].size(); ++j)
                if (matrix[i][j] == 0)
                    mask[j] = 0;
                else
                    mask[j] = 1;
            
            int cnt = 0;

            for (int k = 0; k < matrix.size(); ++k) {
                int isgood = 1;
                for (int j = 1; j < matrix[k].size(); ++j) {
                    if ((matrix[k][0] ^ mask[0]) != (matrix[k][j] ^ mask[j])) {
                        isgood = 0;
                        break;
                    }
                }
                cnt += isgood;
            }
            
            if (cnt > res)
                res = cnt;
            
        }
        
        return res;
    }
};
