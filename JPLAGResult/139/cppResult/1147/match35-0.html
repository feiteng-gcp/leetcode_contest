class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int M = matrix.size();
        int N = matrix[0].size();
        vector<int> same(M, 0);
        vector<int> oppisite(M, 0);
        int res = 0;
        for (int i = 0; i < M; ++i) {
            for (int j = i + 1; j < M; ++j) {
                if (matrix[i][0] == matrix[j][0]) {
                    int findSame = true;
                    for (int k = 0; k < N; ++k) {
                        if (matrix[i][k] != matrix[j][k]) {
                            findSame = false;
                            break;
                        }
                    }
                    if (findSame) {
                        ++same[i];
                        ++same[j];
                    }
                }
                else {
                    int findOppisite = true;
                    for (int k = 0; k < N; ++k) {
                        if (matrix[i][k] == matrix[j][k]) {
                            findOppisite = false;
                            break;
                        }
                    }
                    if (findOppisite) {
                        ++oppisite[i];
                        ++oppisite[j];
                    }
                }
            }
        }
        for (int i = 0; i < M; ++i) {
            res = max(same[i] + oppisite[i], res);
        }
        return res + 1;
    }
};
