class Solution {
 public:
  int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
    map<string, int> mp;
    for (auto& row : matrix) {
      string s, t;

      for (int i : row) {
        s += to_string(i);
        t += to_string(i ^ 1);
      }
      ++mp[s];
      ++mp[t];
    }
    int ans = 0;
    for (auto& p : mp) {
      ans = max(ans, p.second);
    }
    return ans;
  }
};
