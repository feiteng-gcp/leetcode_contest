class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int ret = 0;
        map<vector<int>, int> m;
        for (int i = 0; i < matrix.size(); i++) {
            for (int j = matrix[i].size() - 1; j >= 0; j--) {
                matrix[i][j] ^= matrix[i][0];
            }
            m[matrix[i]]++;
        }

        for (auto p : m) {
            ret = max(ret, p.second);
        }

        return ret;
    }
};
