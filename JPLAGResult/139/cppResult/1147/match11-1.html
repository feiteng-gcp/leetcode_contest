class Solution {
public:
    int maxEqualRowsAfterFlips(vector <vector <int>> &matrix) {
        map <string, int> count;
        for (auto &row : matrix) {
            string s = "", s_flip = "";

            for (auto &cell : row) {
                s += (cell + '0');
                s_flip += (1 - cell + '0');
            }
            if (count.count(s)) {
                count[s]++;
            }
            else if (count.count(s_flip)) {
                count[s_flip]++;
            }
            else {
                count[s] = 1;
            }
        }
        int ans = 0;
        for (auto it : count) {
            ans = max(ans, it.second);
        }
        return ans;
    }
};
