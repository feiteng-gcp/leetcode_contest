class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        map<vector<int>, int> cnt;
        for(int i = 0; i < n; ++i) {
            int c1 = 0;           
            vector<int> t(m);
            for(int j = 0; j < m ; ++j) {
                c1 += (matrix[i][j] == 0);
                t[j] = matrix[i][j] ^ 1;
            }
            cnt[t]++;
            cnt[matrix[i]]++;
        }
        int ans = 0;
        for(auto i : cnt) {
            ans = max(ans, i.second);
        }
        return ans;
    }
};
