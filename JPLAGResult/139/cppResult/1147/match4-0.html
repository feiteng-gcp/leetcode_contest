class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        unordered_map<string, int> um;
        for (int i = 0; i < n; i++) {
            string a, b;
            for (int j = 0; j < m; j++) {
                a.push_back('0' + matrix[i][j]);
                b.push_back('0' + (1 - matrix[i][j]));
            }
            um[a]++;
            um[b]++;
        }
        int mx = 0;
        for (auto [s, cnt] : um) {
            if (cnt > mx) {
                mx = cnt;
            }
        }
        return mx;
    }
};
