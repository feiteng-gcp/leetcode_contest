class Solution {
public:
    bool isEqual(int i, int j, vector<vector<int>>& matrix)
    {
        int n = matrix[0].size();

        bool flag = true;
        for(int p=0;p<n;p++)
        {
            if(matrix[i][p] != matrix[j][p])
            {
                flag = false;
                break;
            }
        }
        if(flag)
            return true;
        for(int p=0;p<n;p++)
        {
            if(matrix[i][p] == matrix[j][p])
            {
                return false;
            }
        }
        return true;
    }
    
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {

        int ans = INT_MIN;
        for(int i=0;i<matrix.size();i++)
        {
            int count = 1;
            for(int j=i+1;j<matrix.size();j++)
            {
                if(isEqual(i, j, matrix))
                    count = count + 1;
            }
            ans = max(ans, count);
        }
        return ans;
    }
};
