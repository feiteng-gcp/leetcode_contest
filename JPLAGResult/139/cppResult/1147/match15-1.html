class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int m = (int)matrix.size(), n = (int)matrix[0].size();
    int ans = 1;
    map<string, int> cnt;

    for(int i = 0; i < m; i++)
    {
        string t = "";
        for(int j = 0; j < n; j++)
        {
            t += matrix[i][0] == 0 ? '0' + matrix[i][j] : '1' - matrix[i][j];
        }
        cnt[t]++;
        ans = max(ans, cnt[t]);
    }
    return ans;
    }
};
