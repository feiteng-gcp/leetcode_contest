class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        unordered_map<string, int> memo;
                
        for (int i = 0; i < n; ++i) {        
            string temp1(""), temp2("");
            for (int j = 0; j < m; ++j) {
                temp1.push_back('0' + matrix[i][j]);
                temp2.push_back('0' + (1 - matrix[i][j]));
            }
            ++memo[temp1];
            ++memo[temp2];
        }
        
        int res = 0;
        for (auto p = memo.begin(); p != memo.end(); ++p) {
            if (p->second > res)
                res = p->second;
        }
        
        return res;
    }
};
