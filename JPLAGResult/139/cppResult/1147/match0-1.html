class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        int m = matrix[0].size(), res = 0;
        vector<vector<int>> pre = matrix, cur;
        for (int i = 0; i < pre.size(); i++) {
            cur.clear();
            int cnt = 0;
            for (int j = 0; j < m; j++) if (pre[0][j] == 1) {
                for (auto &v : pre) v[j] ^= 1;
            }
            for (auto &v : pre) {
                int sum = 0;
                for (auto &i : v) sum += i;
                if (sum == m || sum == 0) cnt++;
                else cur.push_back(v);
            }
            res = max(res, cnt);
            pre = cur;
        }
        return res;
    }
};
