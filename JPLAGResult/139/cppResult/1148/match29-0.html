class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        int ans[1005];
        memset(ans, 0,sizeof(ans));
        
        int sz1 = arr1.size()-1, sz2 = arr2.size()-1;
        int i = 0;
        for (i = 0;sz1 >= 0 || sz2 >= 0; sz1--, sz2--, i++){
            int tmp = ans[i];
            if (sz1 >= 0){
                tmp += arr1[sz1];
            }
            if (sz2 >= 0){
                tmp += arr2[sz2];
            }
            ans[i] = tmp;
        }
        for (int i=0; i<1005; i++){
            if (ans[i] >= 2){
                if (ans[i+1] >= 1){
                    ans[i+1]--;
                    
                } else {
                    ans[i+1] += 1;
                    ans[i+2] += 1;
                }
                ans[i]-=2;
            }
        }
        int sz = 1004;
        while (ans[sz] == 0 && sz > 0) sz--;
        vector<int> ret;
        for (;sz>=0; sz--){
            ret.push_back(ans[sz]);
        }
        return ret;
    }
};
