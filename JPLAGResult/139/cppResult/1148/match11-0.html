class Solution {
   public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        int n1 = arr1.size();
        int n2 = arr2.size();
        if (n1 < n2) {
            swap(arr1, arr2);
            swap(n1, n2);
        }

        vector<int> res;
        int carry = 0;
        int i = 0;
        for(; i < n2; ++i){
            int sum = arr1[i] + arr2[i] + carry;
            if(sum >= 0){
                carry = -(sum / 2);
            }
            else{
                carry = 1;
                sum += 2;
            }
            res.push_back(sum % 2);
        }
        for(; i < n1; ++i){
            int sum = arr1[i] + carry;
            if(sum >= 0){
                carry = -(sum / 2);
            }
            else{
                carry = 1;
                sum += 2;
            }
            res.push_back(sum % 2);
        }
        while(carry != 0){
            int sum = carry;
            if(sum >= 0){
                carry = -(sum / 2);
            }
            else{
                carry = 1;
                sum += 2;
            }
            res.push_back(sum % 2);
        }
        while(res.size() > 1 && *res.rbegin() == 0){
            res.pop_back();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
