class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        int n=arr1.size(),m=arr2.size();
        if(n<m) return addNegabinary(arr2,arr1);
        vector<int> ans;
        int g=0,i,j;
        for(i=m-1;i>=0;i--) {
            arr1[i-m+n] += arr2[i] + g;
            if(arr1[i-m+n]>=2) {
                arr1[i-m+n]-=2;
                g=-1;
            }
            else if(arr1[i-m+n]==-1) {
                arr1[i-m+n] =1;
                g=1;
            }
            else g=0;
        }
        for(i=n-m-1;i>=0;i--) {
            arr1[i] += g;
            if(arr1[i]>=2) {
                arr1[i]-=2;
                g=-1;
            }
            else if(arr1[i]==-1) {
                arr1[i] =1;
                g=1;
            }
            else g=0;
        }
        if(g==-1) {
            ans.push_back(1);
            ans.push_back(1);
        }
        else {
            for(i=0;i<n;i++)
                if(arr1[i]) break;
            if(i==n) {
                ans.push_back(0);    
            }
            for(;i<n;i++)
                ans.push_back(arr1[i]);
            return ans;
        }
        for(i=0;i<n;i++) ans.push_back(arr1[i]);
        return ans;
    }
};
