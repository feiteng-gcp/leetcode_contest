class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> ans;
        int t=max(arr1.size(),arr2.size());
        int add=0;
        for(int i=0;i<t+5;i++){
            int r=add;
            if(i<arr1.size()) r+=arr1[arr1.size()-i-1];
            if(i<arr2.size()) r+=arr2[arr2.size()-i-1];
            if(r%2==0){
                ans.push_back(0);
                add=r/-2;
            }else{
                ans.push_back(1);
                add=(r-1)/-2;
            }
        }
        
        while(ans.size()>1&&ans.back()==0) ans.pop_back();
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
