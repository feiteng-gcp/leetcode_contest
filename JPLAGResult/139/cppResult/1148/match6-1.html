class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        int n=max(arr1.size(),arr2.size());
        

        vector<int> ans;
        int carry = 0;
        for(int i=0;i<=10*n;i++) {
            
            int x = i < arr1.size() ? arr1[arr1.size()-1-i] : 0;
            int y = i < arr2.size() ? arr2[arr2.size()-1-i] : 0;
            
            
            
            int curr = x+y+carry;
                        
            if (curr >= 2) {
                carry = -1;
                curr %= 2;
            } else if (curr == -1) {
                curr = 1;
                carry = 1;
            } else {
                carry = 0;
            }
            ans.push_back(curr);
        }
        while(ans.size() > 1 && ans.back() == 0) {
            ans.pop_back();
        }
        std::reverse(ans.begin(),ans.end());
        return ans;
    }
};
