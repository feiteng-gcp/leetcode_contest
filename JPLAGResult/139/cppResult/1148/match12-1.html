class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        
        vector<int> v;
        int rem = 0;
        for (int i = 0; i < arr1.size() || i < arr2.size() || rem != 0; ++i) {
            if (i < arr1.size()) rem += arr1[i];
            if (i < arr2.size()) rem += arr2[i];
            if (rem < 0) {
                v.push_back(1);
                rem = -2;
            } else {
                v.push_back(rem % 2);
            }
            rem /= -2;
        }
        while (v.size() > 1 && v.back() == 0) v.pop_back();
        reverse(v.begin(), v.end());
        return v;
    }
};
