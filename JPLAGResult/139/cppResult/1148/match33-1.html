class Solution {
public:
    vector<int> addNegabinary(vector<int>& a, vector<int>& b) {
        for (int i=0; i<a.size()/2; i++) swap(a[i],a[a.size()-i-1]);
        for (int i=0; i<b.size()/2; i++) swap(b[i],b[b.size()-i-1]);
        if (a.size()<b.size()) swap(a,b);
        for (int i=0; i<b.size(); i++) a[i]+=b[i];
        for (int i=0; i<5; i++) a.push_back(0);

        for (int i=0; i<a.size()-3; i++) {
            while (a[i]>=2) {
                a[i]-=2;
                --a[i+1];
            }
            while (a[i]<0) {
                a[i]+=2;
                ++a[i+1];
            }
        }
        int j=a.size()-1;
        while (j>=0 && a[j]==0) --j;
        vector<int> ans;
        for (int i=j; i>=0; i--) ans.push_back(a[i]);
        if (j<0) ans.push_back(0);
        return ans;
    }
};
