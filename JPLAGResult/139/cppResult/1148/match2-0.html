class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> res;
        int add = 0;
        for (int i = 0; i < arr1.size() || i < arr2.size() || add != 0; ++i) {
            int a1 = i < arr1.size() ? arr1[arr1.size()-1-i] : 0;
            int a2 = i < arr2.size() ? arr2[arr2.size()-1-i] : 0;
            int sum = a1+a2+add;
            res.push_back((sum+2) % 2);
            if (sum >= 2) add = -1;
            else if (sum == -1) add = 1;
            else add = 0;
        }
        while(res.size() > 1 && res.back() == 0) res.pop_back(); 
        std::reverse(res.begin(), res.end());
        return res;
    }
};
