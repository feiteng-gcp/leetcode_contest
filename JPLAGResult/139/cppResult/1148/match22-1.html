class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> ans;
        
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        
        int i = 0;
        int c = 0;
        for(; i < min(arr1.size(), arr2.size()); i ++){
            int sum = (arr1[i] + arr2[i] + c);
            if(sum < 0){
                ans.push_back(1);
                c = 1;
            }

            else{
                ans.push_back(sum % 2);
                c = (sum / 2) * (-1);
            }
        }
        
        for(; i < arr1.size(); i ++){
            int sum = arr1[i] + c;
            if(sum < 0){
                ans.push_back(1);
                c = 1;
            }
            else{

                ans.push_back(sum % 2);
                c = (sum / 2) * (-1);
            }
        }
        
        for(; i < arr2.size(); i ++){
            int sum = arr2[i] + c;
            if(sum < 0){
                ans.push_back(1);
                c = 1;
            }
            else{
                ans.push_back(sum % 2);
                c = (sum / 2) * (-1);
            }
        }
        
        if(c == 1) ans.push_back(1);
        if(c == -1){
            ans.push_back(1);
            ans.push_back(1);
        }
        
        while(ans.size() && ans.back() == 0)
            ans.pop_back();
        
        if(!ans.size())
            ans.push_back(0);
        
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
