class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> ret;
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        int value = 0;
        for (int i = 0; i < max(arr1.size(), arr2.size()); i++) {
            if (i < arr1.size()) value += arr1[i];
            if (i < arr2.size()) value += arr2[i];
            if (value == 0 || value == 1) {
                ret.push_back(value);
                value = 0;
            } else if (value == -1) {
                ret.push_back(1);
                value = 1;
            } else if (value == 2) {
                ret.push_back(0);
                value = -1;
            } else if (value == 3) {
                ret.push_back(1);
                value = -1;
            }
        }
        if (value == 1) {
            ret.push_back(value);
        } else if (value == -1) {
            ret.push_back(1);
            ret.push_back(1);
        }
        while (ret.size() > 1) {
            if (ret.back() == 0) ret.pop_back();
            else break;
        }
        reverse(ret.begin(), ret.end());
        return ret;
        
    }
};
