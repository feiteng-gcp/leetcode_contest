class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> results;
        results.reserve(max(arr1.size(), arr2.size()) + 1);
        int i = arr1.size() - 1;
        int j = arr2.size() - 1;
        int k = 0, cin = 0;
        for (; i >= 0 || j >= 0 || cin != 0; --i, --j, ++k) {
            int val = cin;
            if (k % 2 == 0) {
                if (i >= 0) {
                    val += arr1[i];
                }
                if (j >= 0) {
                    val += arr2[j];
                }
                if (val < 0) {
                    results.emplace_back(1);
                    cin = -1;
                } else {
                    results.emplace_back(val % 2);
                    cin = val / 2;
                }
            } else {
                if (i >= 0) {
                    val -= arr1[i];
                }
                if (j >= 0) {
                    val -= arr2[j];
                }
                if (val > 0) {
                    results.emplace_back(1);
                    cin = 1;
                } else {
                    results.emplace_back((-val) % 2);
                    cin = val / 2;
                }
            }
        }
        while (results.size() > 1 && results.back() == 0) {
            results.pop_back();
        }
        return vector<int>(results.rbegin(), results.rend());
    }
};
