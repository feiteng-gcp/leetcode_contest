class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        int c = 0, i, j;
        vector<int> res;
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        for (i = 0, j = 0; i < arr1.size() && j < arr2.size(); ++i, ++j)
        {
            int x = c+arr1[i]+arr2[j];
            if (x<0)
            {
                //  - +
                // -1
                //  1 1
                //  + -
                res.push_back(1);
                c = 1;
            }
            else if (x<2)
            {
                res.push_back(x);
                c = 0;
            }
            else if (x>=2)
            {
                res.push_back(x%2);
                c = -1;
            }
        }
        for (; i < arr1.size(); ++i)
        {
            int x = c+arr1[i];
            if (x<0)
            {
                //  - +
                // -1
                //  1 1
                //  + -
                res.push_back(1);
                c = 1;
            }
            else if (x<2)
            {
                res.push_back(x);
                c = 0;
            }
            else if (x>=2)
            {
                res.push_back(x%2);
                c = -1;
            }
        }
        for (; j < arr2.size(); ++j)
        {
            int x = c+arr2[j];
            if (x<0)
            {
                //  - +
                // -1
                //  1 1
                //  + -
                res.push_back(1);
                c = 1;
            }
            else if (x<2)
            {
                res.push_back(x);
                c = 0;
            }
            else if (x>=2)
            {
                res.push_back(x%2);
                c = -1;
            }
        }
        if (c==-1)
        {
            res.push_back(1);
            res.push_back(1);
        }
        if (c==1)
        {
            res.push_back(1);
        }
        while (res.size()>1&&res.back()==0)
            res.pop_back();
        reverse(res.begin(), res.end());
        return res;
    }
};
