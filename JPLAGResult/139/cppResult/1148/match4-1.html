class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        vector<int> res;
        int c = 0;
        for ( int i = 0; i < max(arr1.size(), arr2.size() ); ++i ) {
            int a = 0; int b = 0;
            if ( i < arr1.size() ) a = arr1[i];
            if ( i < arr2.size() ) b = arr2[i];
            int p = ( a + b + c ) % (-2);
            int q = ( a + b + c ) / (-2);
            if ( p == -1 ) {
                p  += 2;
                q += 1;
            }
            // cout << p << " " << q << endl;
            res.push_back(p);
            c = q;
        }
        if ( c == 1 ) res.push_back(c);
        if ( c == -1 ) {
            res.push_back( 1 );
            res.push_back( 1 );
        }
        while ( res.size() > 1 && res[res.size()-1] == 0 ) {
            res.pop_back();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
