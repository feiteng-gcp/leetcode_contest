class Solution {
public:
    vector<int> addNegabinary(vector<int> &arr1, vector<int> &arr2) {
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        if (arr2.size() < arr1.size()) swap(arr1, arr2);
        while (arr1.size() < arr2.size()) arr1.push_back(0);
        
        int carry = 0;
        int i = 0;
        vector<int> ans;
        while (i < arr1.size() || carry != 0) {
            int sum = carry;
            if (i < arr1.size()) {
                sum += arr1[i] + arr2[i];
            }
            if (sum == -1) {
                ans.push_back(1);
                carry = 1;
            }
            if (sum == 0) {
                ans.push_back(0);
                carry = 0;
            }
            if (sum == 1) {
                ans.push_back(1);
                carry = 0;
            }
            if (sum == 2) {
                ans.push_back(0);
                carry = -1;
            }
            if (sum == 3) {
                ans.push_back(1);
                carry = -1;
            }
            ++i;
        }
        while (ans.back() == 0 && ans.size() > 1) ans.pop_back();
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
