class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> ans;
        reverse(arr1.begin(),arr1.end());reverse(arr2.begin(),arr2.end());
        int i=0,j=0,carry=0;
        while(i<arr1.size()||j<arr2.size()||carry!=0)
        {
            int num=0;
            if(i<arr1.size()&&j<arr2.size()){
                num=carry+arr1[i]+arr2[j];i++;j++;
            }
            else if(i<arr1.size()){
                num=carry+arr1[i];i++;                
            }
            else if(j<arr2.size()){
                num=carry+arr2[j];j++;                
            }

            else num=carry;
            if(num==-1){ans.push_back(1);carry=1;}
            if(num==0){ans.push_back(0);carry=0;}
            if(num==1){ans.push_back(1);carry=0;}
            if(num==2){ans.push_back(0);carry=-1;}
            if(num==3){ans.push_back(1);carry=-1;}            
        }
        while(ans.size()>1&&ans.back()==0)ans.pop_back();
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
