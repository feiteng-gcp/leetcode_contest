class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        int len1 = sz(arr1)-1;
        int len2 = sz(arr2)-1;
        VI res;
        int c = 0;
        while(len1 > -1 || len2 > -1 || c != 0) {
            int temp = c;
            if(len1 > -1) temp += arr1[len1];
            if(len2 > -1) temp += arr2[len2];

            len1--, len2--;
            if(temp == -1) {res.pb(1); c = 1;}
            if(temp == 0) {res.pb(0); c = 0;}
            if(temp == 1) {res.pb(1); c = 0;}
            if(temp == 2) {res.pb(0); c = -1;}
            if(temp == 3) {res.pb(1); c = -1;}            
        }
        while(sz(res) > 1 && res[sz(res)-1] == 0) res.pop_back();
        reverse(all(res));
        return res;
    }
};
