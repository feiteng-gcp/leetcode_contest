class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        if (arr1.size() < arr2.size()) {
            return addNegabinary(arr2, arr1);
        }
        vector<int> res;
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        int of = 0;
        int i = 0;
        for (; i < arr2.size(); ++i) {
            int total = arr1[i] + arr2[i] + of;
            if (total == 3) {
                res.push_back(1);
                of = -1;
            } else if (total == 2) {
                res.push_back(0);
                of = -1;
            } else if (total == 1) {
                res.push_back(1);
                of = 0;
            } else if (total == 0) {
                res.push_back(0);
                of = 0;
            } else if (total == -1) {
                res.push_back(1);
                of = 1;
            }
        }
        for (; i < arr1.size(); ++i) {

            int total = arr1[i] + of;
            if (total == 2) {
                res.push_back(0);
                of = -1;
            } else if (total == 1) {
                res.push_back(1);
                of = 0;
            } else if (total == 0) {
                res.push_back(0);
                of = 0;
            } else if (total == -1) {
                res.push_back(1);
                of = 1;
            }
        }
        if (of == 1) {
            res.push_back(1);
        } else if (of == 0) {
            res.push_back(0);
        } else if (of == -1) {
            res.push_back(1);
            res.push_back(1);
        }
        while(res.size() != 1 && res.back() == 0) {
            res.pop_back();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
