class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> arr1_pos = arr1;
        vector<int> arr1_neg = arr1;
        for(int i = 0; i < arr1.size(); i++) {
            if (i%2==0) {
                arr1_neg[arr1.size()-1-i] = 0;
            } else {
                arr1_pos[arr1.size()-1-i] = 0;
            }
        }
        
        vector<int> arr2_pos = arr2;
        vector<int> arr2_neg = arr2;
        for(int i = 0; i < arr2.size(); i++) {
            if (i%2==0) {
                arr2_neg[arr2.size()-1-i] = 0;
            } else {
                arr2_pos[arr2.size()-1-i] = 0;
            }
        }
        
        vector<int> pos_part = addSimple(arr1_pos, arr2_pos);
        vector<int> neg_part = addSimple(arr1_neg, arr2_neg);
        
        return add(pos_part, neg_part);
    }
    
    int get(vector<int>& a, int i) {
        if (i >= a.size()) return 0;
        return a[a.size() - 1 - i];
    }
    
    vector<int> addSimple(vector<int>& a, vector<int>& b) {
        vector<int> res;
        int carry = 0;
        for (int i=0; i < max(a.size(), b.size()); i++) {
            int now = get(a, i) + get(b, i) + carry;
            res.push_back(now % 2);
            carry = now / 2;
        }
        if (carry > 0) {
            res.push_back(1);
        }
        reverse(res.begin(), res.end());
        return res;
    }
    
    void debug(vector<int>& v) {
        for (auto&& i : v){
            cout<<i<<" ";
        }cout<<endl;
    }
    
    vector<int> add(vector<int>& pos, vector<int>& neg) {
        vector<int> res(2000, 0);
        for (int i=0; i < max(pos.size(), neg.size()); i++) {
            int p = get(pos, i);
            int n = get(neg, i);
            if (p>0 && n>0) {
                pos[pos.size() - 1 - i] = 0;
                neg[neg.size() - 1 - i] = 0;
            }
        }
        debug(pos);
        debug(neg);
        
        for (int i =0; i < pos.size(); ++i) {
            if (get(pos, i)>0) {
                if (i%2 == 0) {
                    res[i]++;
                } else {
                    res[i]++;
                    res[i+1]++;
                }
            }
        }
        for (int i=0; i<neg.size(); ++i) {
            if (get(neg, i)>0) {
                if (i%2 == 1) {
                    res[i]++;
                } else {
                    res[i]++;
                    res[i+1]++;
                }
            }
        }
        for(int i=0; i<10; ++i){
            cout<<res[i]<<" ";
        }cout<<endl;
        
        for (int i = 0; i < 2000; i++) {
            bool fixed = true;
            
            for (int i = 0; i < 2000; i++) {
                if (res[i] > 1) {
                    int del_num = min(res[i+1], res[i]/2);
                    res[i+1] -= del_num;
                    res[i]-=del_num * 2;
                    fixed=false;
                    if (del_num == 0) {
                        res[i+1] += res[i] / 2;
                        res[i+2] += res[i] / 2;
                        res[i] = res[i] % 2;
                    }
                }
            }
            
            if (fixed)break;
        }
        // shrink
        
        vector<int> ans;
        bool leading_zero = true;
        for (int i = 1999; i >= 0; i--) {
            if (leading_zero && res[i]==0) {
                continue;
            }
            if (res[i]>0) {
                leading_zero = false;
            }
            ans.push_back(res[i]);
        }
        if(ans.empty()){
            ans.push_back(0);
        }
        return ans;
    }
};
