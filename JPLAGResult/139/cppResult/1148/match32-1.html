class Solution {
public:
    vector <int> addNegabinary(vector <int> &arr1, vector <int> &arr2) {
        int baseneg2[1024] = {0}, L1 = arr1.size(), L2 = arr2.size();
        for (int i = 0; i < L1; i++) {
            baseneg2[i] += arr1[L1 - 1 - i];
        }

        for (int i = 0; i < L2; i++) {
            baseneg2[i] += arr2[L2 - 1 - i];
        }
        for(int i = 0; i < (1024 - 2); i++)
        {
            while(baseneg2[i + 1] >= 1 && baseneg2[i] >= 2)
            {
                baseneg2[i + 1]--;
                baseneg2[i] -= 2;
            }
            baseneg2[i + 2] += (baseneg2[i] / 2);
            baseneg2[i + 1] += (baseneg2[i] / 2);
            baseneg2[i] %= 2;
        }
        for(int i = 1023; i >= 0; i--)
        {
            if(baseneg2[i] == 1)
            {
                vector <int> ans = {};
                for(int j = i; j >= 0; j--)
                    ans.push_back(baseneg2[j]);
                return ans;
            }
        }
        return {0};
    }
};
