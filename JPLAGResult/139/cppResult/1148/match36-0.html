class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {      
        
        int MAXN = 2005;
        vector<int> a1(MAXN, 0), a2(MAXN, 0), res(MAXN, 0);
        
        for (int i = 0; i < arr1.size(); ++i)
            a1[MAXN - arr1.size() + i] = arr1[i];
        for (int i = 0; i < arr2.size(); ++i)
            a2[MAXN - arr2.size() + i] = arr2[i];
        
        int more = 0;
        for (int i = MAXN-1; i > 1; --i) {
            int x = res[i] + a1[i] + a2[i];
            
            if (x == 0 || x == 1) {
                res[i] = x;
            }
            else if (x > 1) {
                res[i] = x%2;
                res[i-1] -= x/2;
            }
            else {
                int u = ((abs(x)+1)/2)*2;
                res[i] = x + u;
                res[i-1] += u/2;
            }
        }
        
        bool notzero = false;
        for (int i = 0; i < res.size(); ++i)
            if (res[i] != 0) {
                notzero = true;
                break;
            }
        
        if (notzero) {
            int i = 0;
            while (res[i] == 0 && (res.begin() + i) != res.end()) i++;
            res.erase(res.begin(), res.begin()+i);

            return res;
        }
        
        res.clear();
        res.push_back(0);
        return res;
        
    }
};
