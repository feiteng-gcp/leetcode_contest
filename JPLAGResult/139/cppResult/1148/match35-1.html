class Solution {
public:
    vector<int> addNegabinary(vector<int>& a1, vector<int>& a2) {
        vector<int> ans;
        int n = (int)a1.size()-1, m = (int)a2.size()-1, carry = 0;
        while (n >= 0 || m >= 0 || carry != 0) {
            int cur = carry;
            if (n >= 0) {
                cur += a1[n];
                --n;
            } 
            
            if (m >= 0) {
                cur += a2[m];
                --m;
            }
            
            int v = cur / (-2), r = cur % (-2);
            // printf("%d:%d %d\n", cur, v, r);
            
            if (r < 0) {
                r += 2;
                ++v;
            }
            
            ans.push_back(r);
            carry = v;
        }
        
        while (ans.size() > 1 && ans.back() == 0)
             ans.pop_back();
        
        reverse(ans.begin(), ans.end());
        
        
        return ans;
    }
};
