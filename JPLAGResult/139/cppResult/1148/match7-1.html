class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> res{};
        reverse(arr1.begin(), arr1.end());
        reverse(arr2.begin(), arr2.end());
        int const len1 = arr1.size();
        int const len2 = arr2.size();
        int const len = max(len1, len2);

        int c = 0;
        for (int i = 0; i <= (len + 1); i++) {
            int a1 = ((i < len1)?arr1[i]:0);
            int a2 = ((i < len2)?arr2[i]:0);
            int a = c + a1 + a2;
            // cout << i << " " << a << endl;
            if (a == 1 || a == -1 || a == 3) {
                res.push_back(1);
            } else {
                res.push_back(0);
            }
            c = 0;
            if (a == -1) c = 1;
            if (a == 2 || a == 3) c = -1;
        }
        while (res.size() > 1 && res.back() == 0) {
            res.pop_back();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
