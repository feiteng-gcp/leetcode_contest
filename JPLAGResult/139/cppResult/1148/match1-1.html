class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        int i = arr1.size()-1, j = arr2.size()-1;
        if (i < j) return addNegabinary(arr2, arr1);
        vector<int> ans;

        int carry = 0;
        while (j >= 0 || i >= 0 || carry) {
            int v1 = (i >= 0? arr1[i] : 0);
            int v2 = (j >= 0? arr2[j] : 0);
            int s = v1 + v2 - carry;
            
            if (s == -1) {
                carry = -1;
                ans.push_back(1);
            }
            else {
                carry = s / 2;
                ans.push_back(s % 2);
            }
            i--;
            j--;
        }
        while (ans.size() && ans.back() == 0) ans.pop_back();
        if (ans.empty()) ans.push_back(0);
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
