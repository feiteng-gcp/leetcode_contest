class Solution {
public:
    vector<int> addNegabinary(vector<int>& arr1, vector<int>& arr2) {
        vector<int> ans;
        for (int i = 0; i < max(arr1.size(), arr2.size()); i++) {
            int v = (arr1.size() > i ? arr1[arr1.size() - 1 - i] : 0) + (arr2.size() > i ? arr2[arr2.size() - 1 - i] : 0);
            ans.push_back(v);

        }
        ans.push_back(0);
        ans.push_back(0);
        
        for (int i = 0; i < ans.size() - 2; i++) {
            if (ans[i] >= 2) {
                if (ans[i + 1] > 0) {
                    ans[i + 1]--;
                } else {
                    ans[i + 1]++;
                    ans[i + 2]++;
                }
                ans[i] -= 2;
            }
        }
        
        while (!ans.empty() && ans.back() == 0) {
            ans.pop_back();
        }
        if (ans.empty()) {
            ans.push_back(0);
        }
        reverse(ans.begin(), ans.end());
        
        return ans;
    }
};
