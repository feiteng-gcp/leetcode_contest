class Solution:
    def numSubmatrixSumTarget(self, matrix: List[List[int]], target: int) -> int:
        n, m = len(matrix), len(matrix[0])
        ps = [[0] * (m + 1) for _ in range(n + 1)]
        for i in range(n):
            for j in range(m):
                ps[i + 1][j + 1] = matrix[i][j] + ps[i][j + 1] 
        ret = 0
        for i in range(n):
            for j in range(i + 1, n + 1):
                c = collections.Counter([0])
                p = 0
                for k in range(m):
                    p += ps[j][k + 1] - ps[i][k + 1]
                    ret += c[p - target]
                    c[p] += 1
        return ret
