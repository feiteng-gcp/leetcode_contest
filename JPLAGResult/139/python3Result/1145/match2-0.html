from collections import Counter

class Solution:
    def numSubmatrixSumTarget(self, matrix: List[List[int]], tgt: int) -> int:
        def numSubarr(arr, tgt):
            presum = Counter()
            presum[0] = 1
            cur, ans = 0, 0
            
            for x in arr:
                cur += x
                ans += presum[cur-tgt]
                presum[cur] += 1
                
            return ans
        
        def getArr(i, j):

        m, n = len(matrix), len(matrix[0])
        sm = [[0 for j in range(n)] for i in range(m+1)]
        
        for i, row in enumerate(matrix):
            for j, a in enumerate(row):
                sm[i+1][j] = sm[i][j] + a
        
        ans = 0
        
        for i in range(m):
            for j in range(i, m):
                arr = getArr(i, j)
                ans += numSubarr(arr, tgt)
                
        return ans
