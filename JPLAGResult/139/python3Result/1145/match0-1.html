class Solution:
    def numSubmatrixSumTarget(self, matrix: List[List[int]], target: int) -> int:
        S = matrix
        m, n = len(S), len(S[0])
        for i in range(1, m):
            S[i][0] += S[i-1][0]
        for j in range(1, n):
            S[0][j] += S[0][j-1]
        for i in range(1, m):
            for j in range(1, n):
                S[i][j] += (S[i-1][j] + S[i][j-1] - S[i-1][j-1])
        
        S = [[0] * n] + S 
        ans = 0

                    cur = S[i+w][j] - S[i][j]

                    cnt[-cur] += 1
        return ans
