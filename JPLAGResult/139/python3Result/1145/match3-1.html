class Solution:
    def numSubmatrixSumTarget(self, matrix: List[List[int]], target: int) -> int:
        n=len(matrix)
        m=len(matrix[0])
        s=[[0 for i in range(m+1)]for j in range(n+1)]
        
for i in range(1,n+1):
            for j in range(1,m+1):
                s[i][j]=s[i-1][j]+s[i][j-1]-s[i-1][j-1]+matrix[i-1][j-1]
        totals=0
        for a in range(n+1):
            for c in range(a+1,n+1):
                dic={}
                dic.clear()
                for d in range(m+1):
                    num=s[c][d]-s[a][d]
                    if num-target in dic:
                        totals=totals+dic[num-target]
                        #print(str(a)+' '+str(c)+' '+str(d)+' '+str(num)+' '+str(dic[num-target]))
                    try:
                        dic[num]
                    except:
                        dic[num]=0
                    dic[num]=dic[num]+1
        return totals
