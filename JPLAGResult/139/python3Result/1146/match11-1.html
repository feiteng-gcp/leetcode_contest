class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        ans = ''
        for i in range(1, 1 + min(len(str1), len(str2))):
            if len(str1) % i != 0 or len(str2) % i != 0:
                continue
            fnd = True
            p = str1[:i]
            for j in range(i, len
(str1), i):
                if str1[j:j + i] != p:
                    fnd = False
                    break
            if not fnd:
                continue
            for j in range(0, len(str2), i):
                if str2[j:j + i] != p:
                    fnd = False
                    break
            if fnd:
                ans = p
        return ans
