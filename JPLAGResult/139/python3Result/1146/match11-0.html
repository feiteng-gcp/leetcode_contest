class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        l1, l2 = len(str1), len(str2)
        if l1 < l2:
            return self.gcdOfStrings(str2, str1)
        
        for k in range(l2, 0, -1):
            if l2 % k > 0 or l1 % k > 0: continue
            ans = str2[:k]
            
            ok = True
            for i in range(0, l1, k):
                if str1[i:i+k] != ans:
                    ok = False
                    break
            if not ok: continue

            for i in range(0, l2, k):
                if str2[i:i+k] != ans:
                    ok = False
                    break
            if ok: return ans   
                    
        return ''
