class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        la = len(str1)
        lb = len(str2)
        l = min(la,lb)
        def ok(s,subs,k):
          #  print(s+" "+subs+str(k))
            return s==subs*k
        if str1[0:l]!=str2[0:l]:
            return ""
        
        for i in range(l,-1,-1):
            if la%(i+1) == 0 and lb%(i+1)==0 and ok(str1,str1[0:i+1],la//(i+1)) and  ok(str2,str2[0:i+1],lb//(i+1)):
                return str1[0:i+1]
            
        return ""
