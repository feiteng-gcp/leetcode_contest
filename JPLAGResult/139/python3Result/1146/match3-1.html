class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        ans = ''
        def div(s, t):
            if len(s) % len(t):
                return False
            for i in range(len(s) // len(t)):
                if s[i * len(t): (i + 1) * len
(t)] != t:
                    return False
            return True
                
        for i in range(min(len(str1), len(str2)), 0, -1):
            a = str1[:i]
            if div(str1, a) and div(str2, a):
                return a
        return ''
