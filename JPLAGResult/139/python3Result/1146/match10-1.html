class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        if len(str2) > len(str1):
            str1, str2 = str2, str1
            
        size1 = len(str1)
        size2 = len(str2)
        
        def verify_divider(size):
            if size1 % size != 0 or size2 % size != 0:
                return False
            
            if str1[:size] != str2[:size]:
                return False
                
            start = 0
            end = size
            while end <= size1:
                if str1[:size] != str1[start:end]:
                    return False
                
                start = end
                end = end + size
                
            start = 0
            end = size
            while end <= size2:
                if str2[:size] != str2[start:end]:
                    return False
                
                start = end
                end = end + size
            
            
return True
        
        
        for size in range(size2, 0, -1):
            if verify_divider(size):
                return str1[:size]
            
        return ''
