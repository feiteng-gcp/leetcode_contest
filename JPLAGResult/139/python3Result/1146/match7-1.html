class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        def common_factors(l1, l2):
            result = []
            for i in reversed(range(1, min(l1, l2) + 1)):
                if l1 % i == 0 and l2 % i == 0:
                    result.append(i)
            return result
        def is_cycle(s, l):
            return len(s) % l == 0 and s[:l]*(len(s)//l) == s
        
        l_cands = common_factors(len(str1), len(str2))
        for l in l_cands:
            
if is_cycle(str1, l) and is_cycle(str2, l) and str1[:l] == str2[:l]:
                return str1[:l]
        return ""
