class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        len1 = len(str1)
        set1 = set([str1])
        for i in range(2, len1):
            if len1 % i == 0:
                tmp = { str1[j:j+len1//i] for j in range(0,len1,len1//i) }
                if len(tmp) == 1:
                    set1.add(tmp.pop())
        len2 = len(str2)
        for i in range(1, len2):
            if len2 % i == 0:
                tmp = { str2[j:j+len2//i] for j in range(0,len2,len2//i) }
                pop = tmp.pop()
                if len(tmp) == 0 and pop in set1:
                    return pop
        return ''
