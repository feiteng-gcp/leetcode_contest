class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        
        def isDivisible(S, T):
            if len(S) % len(T) != 0:
                return False
            for i in range(len(S)):
                if S[i] != T[i % len(T)]:
                    return False
            return True
        for size in range(min(len
(str1), len(str2)), 0, -1):
            sub = str1[:size]
            if isDivisible(str1, sub) and isDivisible(str2, sub):
                return sub
    
        return ""
