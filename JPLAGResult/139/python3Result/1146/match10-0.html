class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        la = len(str1)
        lb = len(str2)
        mi = min(la,lb)
        t = 0
        for i in range(1,mi+1):
            if la % i == 0 and lb % i == 0:
                t = i
        def f(k):
            x = str1[:k]
            if x*(la//k) == str1 and x*(lb//k) == str2:
                return True
            return False
        for i in range(1, t+1):
            if t % i == 0 and f(t//i):
                return str1[:t//i]
        return ""
