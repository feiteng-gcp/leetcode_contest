class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        if len(str1) > len(str2):
            return self.gcdOfStrings(str2, str1)
        if not str1:
            return str2
        for i in range(len(str1)):
            if str1[i] != str2[i]:
                return ''
        return self.gcdOfStrings(str1, str2[len(str1):])
