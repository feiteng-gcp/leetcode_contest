class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        def compare(s1,s2):
            if s1==s2:
                return s1
            if len(s1)<len(s2):
                return compare(s2,s1)
            while len(s1)>=len(s2):
                if s1.startswith(s2):
                    s1 = s1[len(s2):]
                else:
                    return ""
            if s1=="":
                return s2
            else:
                return compare(s2,s1)
        
        if not str1 or not str2:
            return ""
        if str1 == str2:
            return str1
        return compare(str1,str2)
