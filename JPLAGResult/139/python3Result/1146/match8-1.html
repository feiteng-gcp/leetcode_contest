class Solution:
    def gcdOfStrings(self, str1, str2):
        def helper(s1, s2):
            if s1 == s2: return s1
            if len(s1) > len(s2): return helper(s2, s1)

            N = len(s1)
            for i in range(N):
                if s1[i] != s2[i]:
                    return ""

            return helper(s1, s2[N:])

        return helper(str1, str2)
