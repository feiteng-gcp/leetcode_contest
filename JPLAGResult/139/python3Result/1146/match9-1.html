class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        def isDivisor(t, s):
            for i in range(0, len(s), len(t)):
                if t != s[i:i + len(t)]:
                    return False
            return True
        if len(str1) > len(str2):
            str1, str2 = str2, str1
        for l in range(len(str1), 0, -1):
            if len(str1) % l != 0 or len(str2) % l != 0:
                continue
            
if isDivisor(str1[:l], str1) and isDivisor(str1[:l], str2):
                return str1[:l]
        return ''
