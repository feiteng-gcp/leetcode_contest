class Solution:
    def gcdOfStrings(self, str1: str, str2: str) -> str:
        if len(str1) > len(str2):
            str1, str2 = str2, str1
        def check(s, sub):
            if len(s) % len(sub) != 0:
                return False
            return sub * (len(s) // len(sub)) == s
        for i in range(len(str1), 0, -1):
            sub = str1[:i]
            if check(str1, sub) and check(str2, sub):
                return sub
        return ""
