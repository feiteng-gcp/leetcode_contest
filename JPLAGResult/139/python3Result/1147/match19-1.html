class Solution:
    def maxEqualRowsAfterFlips(self, matrix) -> int:
        
for i in range(len(matrix)):
            if matrix[i][0] == 0:
                pass
            else:
                matrix[i] = [j^1 for j in matrix[i]]
        maxGroups = 0
        used = [False for i in range(len(matrix))]
        for i in range(len(matrix)):
            if not used[i]:
                group = 1
                for j in range(i+1, len(matrix)):
                    if matrix[j] == matrix[i]:
                        group += 1
                        used[j] = True
                maxGroups = max(maxGroups, group)

        # print(matrix)
        return maxGroups
