class Solution:
    def maxEqualRowsAfterFlips(self, matrix):
        toString = lambda nums:''.join(map(str, nums))
        def invert(nums):
            nums = [ 1 - a for a in nums ]
            return toString(nums)
        dic = collections.defaultdict(int)
        res = 0
        for nums in matrix:
            dic[toString(nums)] += 1
            dic[invert(nums)] += 1
            res = max(res, dic[toString(nums)])
        return res
