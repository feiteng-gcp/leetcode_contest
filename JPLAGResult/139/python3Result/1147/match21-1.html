class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        counter = dict()
        for row in matrix:
            eigenvalue = []
            for i in range(len(row) - 1):
                if row[i] == row[i+1]:
                    eigenvalue.append('0')
                else:
                    eigenvalue.append('1')
            eigenvalue = "".join(eigenvalue)
            if eigenvalue in counter:
                counter
[eigenvalue] += 1
            else:
                counter[eigenvalue] = 1
        
        return max(counter.values())
