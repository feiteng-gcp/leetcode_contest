import collections

class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        zerosToOnesPositions = []
        onesToZerosPositions = []
        positions = []

        for rowIndex, row in enumerate(matrix):
            thisZerosToOnesPositions = []
            thisOnesToZerosPositions = []

            for columnIndex, box in enumerate(row):
                if box == 0:
                    thisZerosToOnesPositions.append(columnIndex)
                else:
                    thisOnesToZerosPositions.append(columnIndex)

            zerosToOnesPositions.append(tuple(thisZerosToOnesPositions))
            onesToZerosPositions.append(tuple(thisOnesToZerosPositions))
            positions.append(tuple(thisZerosToOnesPositions))
            positions.append(tuple(thisOnesToZerosPositions))

        # print(zerosToOnesPositions)
        # print(onesToZerosPositions)

        # zerosToOnesCount = len(reduce(lambda a, b: a.intersection(b), zerosToOnesPositions, set(range(len(matrix[0])))))
        # onesToZerosCount = len(reduce(lambda a, b: a.intersection(b), onesToZerosPositions, set(range(len(matrix[0])))))

        # return max(zerosToOnesCount, onesToZerosCount)

        counter = collections.Counter(positions)
        return counter.most_common(1)[0][1]
