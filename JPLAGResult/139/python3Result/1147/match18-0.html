from collections import defaultdict

class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        for i in range(len(matrix)):
            if matrix[i][0] == 1:
                matrix[i] = [1-j for j in matrix[i]]
        d = defaultdict(int)
        for row in matrix:
            d[tuple(row)] += 1
        return max(d.values())
