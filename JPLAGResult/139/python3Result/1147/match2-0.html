class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        strings = [''.join([str(i) for i in row]) for row in matrix]
        orstrings = [''.join([str(0 if i else 1) for i in row]) for row in matrix]
        ans = 0
        for i in range(len(matrix)):
            tmp = 0
            for j in range(i, len(matrix)):
                if strings[i] == strings[j] or strings[i] == orstrings[j]:
                    tmp += 1
            ans = max(ans, tmp)
        return ans
