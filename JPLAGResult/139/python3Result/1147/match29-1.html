class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        memo = []
        whole = set([i for i in range(len(matrix[0]))])
        for row in matrix:
            tmp = set([])
            for i in range(len(row)):
                if row[i]:
                    tmp.add(i)
            if len(tmp) > len(matrix[0])/2:
                tmp = whole - tmp
            memo += [tmp]
        res = 0
        for i in range(len(matrix)):
            cnt = 0
            for j in range(len(matrix)):
                if len(memo[i]) < len(matrix[0])/2 and len(memo[i]) == len(memo[j]) and memo[i]==memo[j]:
                    cnt += 1
                elif len(memo[i]) == len(matrix[0])/2 and (memo
[i] == memo[j] or memo[i] == whole -memo[j]):
                    cnt += 1
            res = max(res,cnt)
        return res
