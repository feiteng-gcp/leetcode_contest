class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        c = collections.Counter()
        for r in matrix:
            zero = []
            one = []
            for i, n in enumerate(r):
                if n == 0:
                    zero.append(i)
                else:
                    one.append(i)
            c[tuple(zero)] += 1
            c[tuple(one)] += 1
        return c.most_common(1)[0][1]
