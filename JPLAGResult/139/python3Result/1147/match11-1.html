class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        edge = (1<<len(matrix[0]))-1
        groups = {}
        for row in matrix:
            n = 0
            for i in row:
                n<<=1
                n|=i
            if n in groups:
                groups[n]+=1
            elif edge^n in groups:
                groups
[edge^n]+=1
            else:
                groups[n]=1
        return max(groups.values())
