class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        flips = {}
        for i in matrix:
            resultS = ""
            if i[0] == 0:
                for j in i:
                    if j == 1:
                        resultS += "0"
                    else:
                        resultS += "1"
            else:
                for j in i:
                    if j == 1:
                        resultS += "1"
                    else:
                        resultS += "0"
            
if resultS in flips:
                flips[resultS] += 1
            else:
                flips[resultS] = 1
        return max(flips.values())
