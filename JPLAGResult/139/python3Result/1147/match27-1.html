class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        h=len(matrix)
        l=len(matrix[0])
        d=[0 for i in range(h)]
        for i in range(h):
            g=0
            for j in range(l):
                g=(g*2+matrix[i][j])
            d[i]=g
        g=2**l
        t=1
        ma=0
        for i in range(h):
            t=1
            for j in range(h):
                if i!=j:
                    if d
[i]==d[j] or d[i]+d[j]==(g-1):
                        t+=1
            ma=max(ma,t)
        return ma
