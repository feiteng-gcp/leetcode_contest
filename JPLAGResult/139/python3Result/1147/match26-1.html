class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        n=len(matrix)
        if n<=1:
            return n
        dict1=collections.defaultdict(int)
        for m in matrix:
            tmpOne=[]
            tmpZero=[]
            for i in range(len(m)):
                if m[i]==0:
                    tmpOne.append(i)
                else:
                    tmpZero.append(i)
            dict1
[tuple(tmpOne)]+=1
            dict1[tuple(tmpZero)]+=1
        return max(dict1.values())
