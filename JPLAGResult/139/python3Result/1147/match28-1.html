class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        memo = collections.defaultdict(list)
        
        
for i, row in enumerate(matrix):
            key0 = []
            key1 = []
            for j , n in enumerate(row):
                if n == 0:
                    key0.append(j)
                else:
                    key1.append(j)
            if len(key1) > len(key0):
                memo[tuple(key0)].append(i)
            elif len(key0) > len(key1):
                memo[tuple(key1)].append(i)
            else:
                memo[tuple(key1)].append(i)
                memo[tuple(key0)].append(i)
        return max( len(x) for x in memo.values())
