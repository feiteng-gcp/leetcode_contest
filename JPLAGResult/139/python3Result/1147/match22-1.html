from collections import Counter

class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        counter = Counter()
        
        for row in matrix:
            ones = tuple(idx for idx, num in enumerate(row) if num == 1)
            zeros = tuple(idx for idx, num in enumerate(row) if num == 0)
            
            if ones in counter:
                counter[ones] += 1
            else:
                counter[ones] = 1
                
            
if zeros in counter:
                counter[zeros] += 1
            else:
                counter[zeros] = 1
                
        return counter.most_common(1)[0][1]
