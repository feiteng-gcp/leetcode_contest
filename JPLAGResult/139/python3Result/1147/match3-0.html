class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        distinct_rows_upto_flip = {}
        for row_index, row in enumerate(matrix):
            str_row = "".join([str(i) for i in row])
            dual_str_row = "".join([str(1-i) for i in row])
            if str_row in distinct_rows_upto_flip:
                distinct_rows_upto_flip[str_row] += 1
            elif dual_str_row in distinct_rows_upto_flip:
                distinct_rows_upto_flip[dual_str_row] += 1
            else:
                distinct_rows_upto_flip[str_row] = 1
        return max(distinct_rows_upto_flip.values())
