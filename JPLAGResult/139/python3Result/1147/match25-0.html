class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        def eq(t1, t2):
            return t1[0]==t2[0] or t1[0]==t2[1]
        
        ls = []
        for r in matrix:
            zeros = []
            ones = []
            for i in range(len(r)):
                if r[i] == 0:
                    zeros.append(i)
                else:
                    ones.append(i)
            ls.append((zeros, ones))
            
        ret = 1
        for i in range(len(ls)):
            cnt = 1
            for j in range(i+1, len(ls)):
                if eq(ls[i], ls[j]):
                    cnt += 1
                    if cnt > ret:
                        ret = cnt
        return ret
