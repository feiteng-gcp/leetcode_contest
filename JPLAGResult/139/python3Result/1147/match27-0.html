class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        
        n = len(matrix[0])
        d = []
        
        for i in matrix:
            b = "".join(map(str, i))
            d.append( int(b, 2) )
            
        maxv = -float('inf')
        
        # print (d)
        
        for i in range(len(d)):
            
            c = 1
            
            for j in range(len(d)):
                
                if i != j:
                    
                    
                    
                    if (d[i] == d[j]) or ( (d[i] & d[j] == 0) and (d[i] | d[j] == 2**n - 1 ) ):
                        c += 1
                        
            maxv = max(maxv, c)
            
        return maxv
