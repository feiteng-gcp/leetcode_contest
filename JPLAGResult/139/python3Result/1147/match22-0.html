from collections import Counter
class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        if not matrix or not matrix[0]:
            return 0
        
        rows = [
            (tuple(c for c in range(len(matrix[r])) if matrix[r][c] == 0), tuple(c for c in range(len(matrix[r])) if matrix[r][c] == 1))
            for r in range(len(matrix))
        ]
        c = Counter()
        for a, b in rows:
            if a in c:
                c[a] += 1
            else:
                c[b] += 1
        return c.most_common(1)[0][1]
