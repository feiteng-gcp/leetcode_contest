from collections import defaultdict
class Solution:
    def maxEqualRowsAfterFlips(self, mat: List[List[int]]) -> int:
        
        d = defaultdict(int)
        m, n = len(mat), len(mat[0])
        
        full = set([i for i in range(n)])
        for i in range(m):
            cur = set()
            for j in range(n):
                if mat[i][j]:
                    cur.add(j)
            if len(cur) > n // 2:
                cur = full - cur
            if n % 2 == 0 and len(cur) == n // 2 and tuple(cur) not in d:
                cur 
= full - cur
            d[tuple(cur)] += 1
            
        # print(d)
            
        return max(d.values())
