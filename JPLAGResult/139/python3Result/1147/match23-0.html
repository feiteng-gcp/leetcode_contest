class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        s = {}
        for i in range(len(matrix)):
            temp = int(''.join([str(y) for y in matrix[i]]), 2)
            c = (1 << len(matrix[i])) - 1 - temp
            if temp in s:
                s[temp]+=1
            elif c in s:
                s[c]+=1
            else:
                s[temp] = 1
        return max(s.values())
