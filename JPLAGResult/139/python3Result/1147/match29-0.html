class Solution:
    def maxEqualRowsAfterFlips(self, matrix: List[List[int]]) -> int:
        m = [''.join([str(v) for v in row]) for row in matrix]
        m2 = [''.join([str(1-v) for v in row]) for row in matrix]
        maxSame = 0
        print(m, m2)
        for i in range(len(matrix)-1):
            curSame = 1
            for j in range(i+1, len(matrix)):
                if m[i] == m[j] or m[i] == m2[j]:
                    curSame += 1
            maxSame = max(maxSame, curSame)
            
        return maxSame
