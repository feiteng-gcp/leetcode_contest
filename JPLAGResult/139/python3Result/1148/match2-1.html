class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        def baseNeg2(N):
            
if N == 0: 
                return [0]
            ans = []
            while N != 0:
                r = N % (-2)
                #print('r=', r)
                N = N // (-2)
                if r < 0:
                    r += 2
                    N += 1
                ans = [r]+ans
            return ans
        
        def baseNeg2toInt(A):
            A = A[::-1]
            ans = 0
            for i in range(len(A)):
                ans += A[i]*((-2)**i)
            return ans
        a = baseNeg2toInt(arr1)
        b = baseNeg2toInt(arr2)
        #print('a = %s, b = %s, a+b = %s'%(a, b, a+b) )
        return baseNeg2(a+b)
