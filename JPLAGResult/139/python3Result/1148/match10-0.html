from itertools import zip_longest

class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        result = []
        carry = 0
        
        for digit1, digit2 in zip_longest(reversed(arr1), reversed(arr2), fillvalue=0):
            res = digit1 + digit2 + carry
            
            if res in (0, 1):
                carry = 0
            elif res == -1:
                carry = 1
            else:
                carry = -1
            
            res = res % 2
            result.append(res)
        
        if carry:
            result.append(1)

        if carry == -1:
            result.append(1)
            
        while len(result) > 1 and result[-1] == 0:
            result.pop()
        
        return list(reversed(result))
