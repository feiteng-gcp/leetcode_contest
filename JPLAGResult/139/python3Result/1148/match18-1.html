def toDec(a: List[int]):
        ans = 0
        for i in a:
            ans 
= ans * -2 + i
        return ans
        
def toM2(n):
    if n == 0:
        return "0"
    s = ""
    while n != 0:
        if n % 2 != 0:
            n -= 1
            s = "1" + s
        else:
            s = "0" + s
        n //= -2
    return s

class Solution:

        A = toDec(arr1)
