class Solution:
    def addNegabinary(self, arr1, arr2):
        arr1.reverse()
        arr2.reverse()

        N = max(len(arr1), len(arr2)) + 2
        arr1.extend([0 for _ in range(N - len(arr1))])
        arr2.extend([0 for _ in range(N - len(arr2))])

        ret = []

        c = 0
        for i in range(N):
            val = arr1[i] + arr2[i] - c
            if val >= 0:
                ret.append(val % 2)
                c = val // 2
            else:
                ret.append(1)
                c = -1

        while len(ret) > 1 and ret[-1] == 0:
            ret.pop()

        ret.reverse()

        return ret
