class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        arr1.reverse()
        arr2.reverse()
        answer = []
        carry = 0
        m = len(arr1)
        n = len(arr2)
        for i in range(max(m, n)):
            num1 = 0
            num2 = 0
            if i < m:
                num1 = arr1[i]
                
            if i < n:
                num2 = arr2[i]
                
            s = num1 + num2 + carry
            if s == 0:
                answer.append(0)
                carry = 0
                
            elif s == 1:
                answer.append(1)
                carry = 0 
                
            elif s == 2:
                answer.append(0)
                carry = -1
                
            elif s == 3:
                answer.append(1)
                carry = -1
                
            elif s == -1:
                answer.append(1)
                carry = 1
                
        if carry == 1:
            answer.append(1)
            
        elif carry == -1:
            answer.extend([1, 1])
            
        answer.reverse()
        for i in range(len(answer)):
            if answer[i] != 0:
                break
                
        return answer[i:]
