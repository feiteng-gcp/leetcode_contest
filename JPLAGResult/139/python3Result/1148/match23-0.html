class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        res = []
        carry = 0
        while arr1 or arr2:
            num1, num2 = 0, 0
            if arr1:
                num1 = arr1.pop()
            if arr2:
                num2 = arr2.pop()
            h = num1 + num2 + carry
            carry = -(h // 2)
            res.append(h % 2)
        if carry == -1:
            res.append(1)
            res.append(1)
        while len(res) > 1 and res[-1] == 0:
            res.pop()
        return res[::-1]
