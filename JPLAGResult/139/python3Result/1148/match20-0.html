class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        x = self.toInt(arr1)
        y = self.toInt(arr2)
        return self.toBaseNeg2(x+y)
        
    
    def toInt(self, s):
        tmp = s[::-1]
        base = 1
        ret = 0
        for c in tmp:
            if c == 1:
                ret += base
            base *= -2
        return ret
        
    def toBaseNeg2(self, N):
        if not N:
            return [0]
        ret = []
        while N:
            if N & 1:
                ret.append(1)
            else:
                ret.append(0)
            N = -(N>>1)
        return ret[::-1]
