class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        dec1, dec2 = 0, 0
        
        n1, n2 = len(arr1)-1, len(arr2)-1
        for i in range(len(arr1)):
            if arr1[i] == 1:
                dec1 += (-2) ** (n1-i)
        
        for i in range(len(arr2)):
            if arr2[i] == 1:
                dec2 += (-2) ** (n2-i)
        
        N = dec1 + dec2
        
        if N == 0:
            return [0]
        if N == 1:
            return [1]
        res = []
        while N != 1:
            N, k = divmod(N, -2)
            if k == -1:
                N += 1
                k = 1
            res.append(k)
            
        res.append(1)
        return res[::-1]
