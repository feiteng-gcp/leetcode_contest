class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        
def decode(arr):
            out = 0
            e = 0
            for i in arr[::-1]:
                if i:
                    out += (-2)**e
                e += 1
            return out
        
        def encode(v):
            if v == 0:
                return [0]
            
            out = []
            while v != 0:
                v,r = divmod(v, -2)
                
                if r < 0:
                    r += 2
                    v += 1
                out = [r] + out
                
            return out
        
        a, b = decode(arr1), decode(arr2)
        x = a + b
        
        #print(a, b, x)
        #print(decode(encode(x)))
        return encode(x)
