class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        if len(arr1) < len(arr2):
            arr1,arr2 = arr2,arr1
        arr1 = arr1[::-1]
        arr2 = arr2[::-1] + [0]*abs(len(arr1)-len(arr2))
        ans = [0]*len(arr1)
        # print(arr1)
        # print(arr2)
        for i in range(len(arr1)):
            ans[i] = ans[i]+arr1[i] + arr2[i]
            while ans[i]>1:
                if i+1 >= len(ans):
                    ans.append(0)
                if ans[i+1]:
                    ans[i+1]-=1
                else:
                    ans[i+1]+=1
                    if i+2 >=len(ans):
                        ans.append(0)
                    ans[i+2]+=1
                ans[i]-=2
        def trim(ans):
            start = 0
            for i in range(len(ans)):
                if ans[i]!=0:
                    return ans[i:]
                    break
            return [0]
        return trim(ans[::-1])
