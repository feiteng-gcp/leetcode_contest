class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        l = []
        t = 0
        while True:
            if len(arr1) == 0 and len(arr2) == 0 and t == 0:
                break
            t *= -1
            if len(arr1) != 0 and arr1.pop() == 1:
                t += 1
            if len(arr2) != 0 and arr2.pop() == 1:
                t += 1
            if t % 2 == 1:
                l.append(1)
            else:
                l.append(0)
            t //= 2
        while len(l) > 1 and l[-1] == 0:
            l.pop()
        return l[::-1]
