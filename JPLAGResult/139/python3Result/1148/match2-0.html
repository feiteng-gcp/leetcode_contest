class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        
        num1, num2 = 0, 0
        for i, num in enumerate(arr1[::-1]):
            num1 += num * ((-2) ** i)
        for i, num in enumerate(arr2[::-1]):
            num2 += num * ((-2) ** i)
        n = num1 + num2
        
        if n == 0: return [0]
        converted = []
        while n != 0:
            rem = n % -2
            n //= -2
            
            if rem < 0:
                rem += (-1) * (-2)
                n += 1
            
            converted = [rem] + converted
        return converted
