import itertools


class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        borrow = 0
        result = []
        for i, j in itertools.zip_longest(reversed(arr1),
                                          reversed(arr2),
                                          fillvalue=0):
            re = i + j + borrow
            if 0 <= re <= 1:
                result.append(re)
                borrow = 0
                continue
            elif re < 0:
                re += 2
                result.append(re)
                borrow = 1
                continue
            else:
                re -= 2
                result.append(re)
                borrow = -1

        if borrow == -1:
            result.append(1)
            result.append(1)
        elif borrow == 1:
            result
.append(1)

        while len(result) > 1 and result[-1] == 0:
            result.pop()

        return list(reversed(result))
