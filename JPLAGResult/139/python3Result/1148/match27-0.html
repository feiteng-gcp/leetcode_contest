class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        if len(arr1) < len(arr2):
            arr1, arr2 = arr2, arr1
        arr = arr1[:len(arr1) - len(arr2)]
        for i in range(len(arr2)):
            arr.append(arr1[len(arr1) - len(arr2) + i] + arr2[i])
        arr.insert(0, 0)
        arr.insert(0, 0)
        for i in range(len(arr) - 1, 1, -1):
            while arr[i] >= 2 and arr[i - 1] >= 1:
                arr[i] -= 2
                arr[i - 1] -= 1
            while arr[i] >= 2:
                arr[i] -= 2
                arr[i - 2] += 1
                arr[i - 1] += 1
        while len(arr) > 1 and arr[0] == 0:
            del arr[0]
        return arr
