class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        res = []
        m, n = len(arr1), len(arr2)
        i, j = m - 1, n - 1
        carry = 0
        while i >= 0 or j >= 0 or carry != 0:
            sum_ = carry
            if i >= 0:
                sum_ += arr1[i]
            if j >= 0:
                sum_ += arr2[j]
            # if sum_ > 0:
            #     sum_ -= carry
            # else:
            #     sum_ = carry
            res.append(abs(sum_) % 2)
            if sum_ < 0:
                sum_ += 2
                carry = 1
            else:
                carry = -(sum_ // 2)
            i -= 1
            j 
-= 1
        while len(res) > 1 and res[-1] == 0:
            res.pop()
        return res[::-1]
