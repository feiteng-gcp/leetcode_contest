class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        
        
        def negB(A):
            
            if A == 0:
                return [0]
            
            res = []
            
            while A:
                r = A % -2
                A = A // -2
                
                if r < 0:
                    r += 2
                    A += 1
                    
                res.append(r)
                
            return res[::-1]
        
        def numb(A):
            c = 0
            for i in range(len(A) - 1, -1, -1):
                c += A[i] * (-2) ** (len(A) - i - 1)
            return c
                    
        c1 = numb(arr1)
        c2 = numb(arr2)
        c3 = c1 + c2
        
        # print (c1, c2)
        
        return negB(c3)
