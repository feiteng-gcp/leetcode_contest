class Solution:
    def addNegabinary(self, arr1, arr2):
        def todec(arr):
            return sum( a*(-2)**i for i,a in enumerate(arr[::-1]) )
        s = todec(arr1) + todec(arr2)
        res = []
        #print(s)
        while s != 0:
            t = abs(s%(-2))
            res.append(t)
            s = (s - t) //(-2)
        return res[::-1] or [0]
