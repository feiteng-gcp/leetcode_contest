class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        def a2n(arr):
            len_a = len(arr)
            _sum = 0
            t = 1
            for i in range(len(arr)):
                _sum += arr[len_a - i - 1] * t
                t *= -2
            return _sum
        def n2a(n):
            res = []
            while n != 0:
                n, c = divmod(n, -2)
                if c != 0:
                    c = 1
                    n += 1
                res.append(c)
            return res[::-1] if res else [0]
        # print(a2n([1,1,0,1]), n2a(-3), n2a(0))
        return n2a(a2n(arr1)+a2n(arr2))
