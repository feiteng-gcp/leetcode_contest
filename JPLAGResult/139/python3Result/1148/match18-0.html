class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        number1 = self.listToInt(arr1)
        number2 = self.listToInt(arr2)
        return list(map(int, list(self.baseNeg2(number1 + number2))))

    def listToInt(self, l: list) -> int:
        res = 0

        for i, v in enumerate(reversed(l)):
            res += v * (-2) ** i

        return res

    def baseNeg2(self, N: int) -> str:
        if N == 0:
            return "0"

        res = []

        while N != 0:
            if N % 2 == 0: # N是偶数
                remainder = 0 # r一定是0
                N = - (N // 2) # 算出d，作为下一轮新的N
            else: # N是奇数
                remainder = 1 # r一定是1
                N = (N - 1) // -2 # 算出d，作为下一轮新的N
            res.append(str(remainder)) # 记录r

        return "".join(res[:: -1]) # 结果是r的倒序
