class Solution:
    def addNegabinary(self, arr1, arr2):
        arr1, arr2 = arr1[::-1], arr2[::-1]
        l1, l2 = len(arr1), len(arr2)
        if l1<l2:
            arr1, arr2, l1, l2 = arr2, arr1, l2, l1
        if l1==0:
            return arr2[::-1]
        if l2==0:
            return arr1[::-1]
        result = [0 for k in range(l1+1)]
        for i in range(l2):
            r = arr1[i] + arr2[i] + result[i]
            if r==3:
                result[i] = 1
                result[i+1] = -1
            elif r==2:
                result[i] = 0
                result[i+1] = -1
            elif r==1:
                result[i] = 1
                result[i+1] = 0
            elif r==0:
                result[i] = 0
                result[i+1] = 0
            else:
                result[i] = 1
                result[i+1] = 1
        for i in range(l2, l1):
            r = arr1[i] + result[i]
            if r==3:
                result[i] = 1
                result[i+1] = -1
            elif r==2:
                result[i] = 0
                result[i+1] = -1
            elif r==1:
                result[i] = 1
                result[i+1] = 0
            elif r==0:
                result[i] = 0
                result[i+1] = 0
            else:
                result[i] = 1
                result[i+1] = 1
        if result[l1]==-1:
            result[l1] = 1
            result.append(1)
        print(result)
        for i in range(len(result)-1, -1, -1):
            if result[i]==1:
                result = result[:i+1]
                break
        else:
            return [0]
        result = result[::-1]
        return result
