class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        arr = [0] * (max(len(arr1), len(arr2)) + 2)
        arr1 = arr1[::-1]
        arr2 = arr2[::-1]
        # print(arr1, arr2)
        for i in range(len(arr) - 2):
            cur = arr[i]
            if i < len(arr1):
                cur += arr1[i]
            if i < len(arr2):
                cur += arr2[i]
            if cur >= 2:
                arr[i] = cur % 2
                arr[i+1] = -1
            elif cur < 0:
                arr[i+1] = 1
                arr[i] = 1
            else:
                arr[i] = cur
        
        if arr[-2] == -1:
            arr[-1] = 1
            arr[-2] = 1
            
        ans = []
        start = False
        for i in range(len(arr) - 1, -1, -1):
            if start or arr[i] == 1:
                start = True
                ans.append(arr[i])
        if len(ans) == 0:
            return [0]
        else:
            return ans
