class Solution:
    
    def toBase(self, arr):
        ans = 0
        base = 1
        for num in arr[::-1]:
            if num == 1:
                ans += base
            base *= -2
        return ans
    
    def fromBase(self, n):
        ans = []
        base = -1
        while n != 0:
            ans.append(n % 2)
            if n % 2 == 1:
                n += base
            base = -base
            n = int(n/2)
        return ans[::-1]
            
    
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        ans = []
        carry = 0
        arr1 = arr1[::-1]
        arr2 = arr2[::-1]
        if len(arr1) > len(arr2):
            arr2 += [0] * (len(arr1) - len(arr2))
        else:
            arr1 += [0] * (len(arr2) - len(arr1))
        for i in range(len(arr1)):
            n = arr1[i] + arr2[i] + carry
            if n == -1:
                carry = 1
                ans.append(1)
            if n == 0:
                carry = 0
                ans.append(0)
            if n == 1:
                carry = 0
                ans.append(1)
            if n == 2:
                carry = -1
                ans.append(0)
            if n == 3:
                carry = -1
                ans.append(1)
            
        if carry == 1:
            ans.append(1)
        if carry == -1:
            ans.append(1)
            ans.append(1)
        while len(ans) > 1 and ans[-1] == 0:
            ans = ans[:-1]
        return ans[::-1]
