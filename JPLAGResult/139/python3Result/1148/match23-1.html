class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        i = 1
        j = 0
        res = []
        l1, l2 = len(arr1), len(arr2)
        if l1 <= l2:
            arr1 = [0] * (l2 - l1) + arr1
        else:
            arr2 = [0] * (l1 - l2) + arr2
        while i <= len(arr1):
            t = arr1[-i] + arr2[-i] + j
            if t == 0 or t == 2:
                res.append(0)
            else:
                res.append(1)
            if t >= 2:
                j = -1
            elif t == -1:
                j = 1
            else:
                j = 0
            i += 1
        if j == 1:
            res.append(1)
        elif j == -1:
            res

            res.append(1)
        while len(res) > 1 and res[-1] == 0:
            res.pop()
        return list(reversed(res))
