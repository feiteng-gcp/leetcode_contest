class Solution:
    def addNegabinary(self, arr1: List[int], arr2: List[int]) -> List[int]:
        if len(arr2) > len(arr1):
            arr1, arr2 = arr2, arr1
        arr1 = [0] * (len(arr1) * 2) + arr1
        arr2 = [0] * (len(arr1) - len(arr2)) + arr2
        for i, n in enumerate(arr2):
            arr1[i] += n
        arr1 = arr1[::-1]
        for i, n in enumerate(arr1):
            if n == 3:
                arr1
[i + 1] -= 1
                arr1[i] = 1
            elif n == 2:
                arr1[i + 1] -= 1
                arr1[i] = 0
            elif n == -1:
                arr1[i + 1] += 1
                arr1[i] = 1
        arr1 = arr1[::-1]
        while len(arr1) > 1 and arr1[0] == 0:
            arr1 = arr1[1:]
        return arr1
