	public boolean yes(String str1, String s) {
		if (str1.length() % s.length() != 0) {
			return false;
		}
		while (str1.length() > 0) {
			if (str1.substring(0, s.length()).equals(s)) {
				str1 = str1.substring(s.length());
			} else {
				return false;
			}
		}
