class Solution {
    public String gcdOfStrings(String str1, String str2) {
        if(!(str1.startsWith(str2) || str2.startsWith(str1))){
            return "";
        }
        String res = "";
        for(int i=0 ; i<str1.length() ; i++){
            String str = str1.substring(0,i+1);
            if(isGcd(str,str1)&&isGcd(str,str2)){
                res = str;
            }
        }
        return res;
    }
    
    public boolean isGcd(String s,String str){
        if(str.length()%s.length()!=0) return false;
        for(int i=0 ; i<str.length() ; i+=s.length()){
            if(!s.equals(str.substring(i,i+s.length()))){
                return false;
            }
        }
        
        return true;
    }
}
