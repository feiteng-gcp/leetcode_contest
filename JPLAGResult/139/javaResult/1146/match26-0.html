class Solution {
    
    
    
    public String gcdOfStrings(String str1, String str2) {
        int l = str2.length();
        int l1 = str1.length();
        
        
        
        
        
        for(int i=0;i<l;++i){
            
            int len = l-i;
            if(l%len==0&&l1%len==0){
                String s = str2.substring(i);
                
                StringBuilder sb = new StringBuilder();
                
                int f = l/len;
                
                for(int u=0;u<f;++u){
                    sb.append(s);
                }
                if(!sb.toString().equals(str2)){
                    continue;
                }
                
                
                StringBuilder sb1 = new StringBuilder();
                
                int f1 = l1/len;
                
                for(int u=0;u<f1;++u){
                    sb1.append(s);
                }
                if(!sb1.toString().equals(str1)){
                    continue;
                }
                
                return s;
                
                
            }
            
        }
        return "";
        
        
    }
}
