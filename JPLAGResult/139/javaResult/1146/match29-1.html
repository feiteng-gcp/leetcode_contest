class Solution {
    public String gcdOfStrings(String str1, String str2) {
        if (str1.indexOf(str2) == -1 && str2.indexOf(str1) == -1) {
            return "";
        }
        int l1 = str1.length();
        int l2 = str2.length();
        
        int l = l1 < l2 ? get(l1, l2) : 
get(l2, l1);
        return str1.substring(0, l);
    }
    
    private int get(int l1, int l2) {
        while (l2 % l1 != 0) {
            int temp = l1;
            l1 = l2 % l1;
            l2 = temp;
        }
        return l1;
    }
}
