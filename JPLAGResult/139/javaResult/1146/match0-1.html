class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int m=str1.length(), n=str2.length(), com=gcd(m, n);
        
        for(int k=com; k>=Math.sqrt(com); k--){
            if(m%k!=0||n%k!=0) continue;
            
if(common(str1.substring(0, k), str1, str2)) return str1.substring(0, k);
        }
        return "";
    }
    
    private int gcd(int m, int n){
        if(m>n) return gcd(n, m);
        if(m==0) return n;
        return gcd(n%m, m);
    }
    
    private boolean common(String tgt, String str1, String str2){
        String tmp = "";
        while(tmp.length()<=str1.length() || tmp.length()<=str2.length()){
            if(tmp.length()==str1.length()&&!tmp.equals(str1)) return false;
            if(tmp.length()==str2.length()&&!tmp.equals(str2)) return false;
            tmp += tgt;
        }
        return true;
    }
}
