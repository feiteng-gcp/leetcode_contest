class Solution {

    public static void main(String[] args) {
        System.out.println(new Solution().gcdOfStrings("ABCABCABCABC", "ABCABC"));
    }
    public String gcdOfStrings(String str1, String str2) {
        if (str1.length() < str2.length()) {
            return gcdOfStrings(str2, str1);
        }

        if (str2.length() == 0) {
            return "";
        }

        if (!str1.contains(str2)) {
            return "";
        }

        if (str1.length() == str2.length()) {
            return str1;
        }
        int gcdv = gcd(str1.length(), str2.length());
        if (gcdv == 0) {
            return "";
        }

        if (test(str1, str1.substring(0, gcdv)) && test(str2, str1.substring(0, gcdv))) {
            return str1.substring(0, gcdv);
        }
        return "";
    }

    private boolean test(String str, String sub) {
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) != sub.charAt(i % sub.length())) {
                return false;
            }
        }
        return true;
    }

    private int gcd(int a, int b) {
        if (b == 0) {
            return a;
        }
        return gcd(b, a % b);
    }
}
