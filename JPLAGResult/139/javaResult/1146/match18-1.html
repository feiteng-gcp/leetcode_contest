class Solution {
    public String gcdOfStrings(String str1, String str2) {
        String s;
        for (int i = Math.min(str1.length(), str2.length()); i > 0; i--) {
            s = str1.substring(0, i);
            
if (str2.split(s).length == 0 && str1.split(s).length == 0) return s;
        }
        return "";
    }
}
