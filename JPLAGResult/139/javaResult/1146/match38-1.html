class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int len1 = str1.length(), len2 = str2.length();
        int len = 
Math.min(len1, len2);
        for(int i = len; i > 0; i--) {
            if(len1 % i == 0 && len2 % i == 0) {
                String sub = str1.substring(0, i);
                if(isGcd(str1, sub) && isGcd(str2, sub)) {
                    return sub;
                
}
            }
        }
        return "";
    }

    private boolean isGcd(String str, String sub) {
        String temp = str.replaceAll(sub, "");
        return temp.length() == 0;
    }
}
