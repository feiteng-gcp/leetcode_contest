class Solution {
    public String gcdOfStrings(String str1, String str2) {
        if (str1.length() == 0 || str2.length() == 0) {
            return "";
        }
        if (str1.equals(str2)) {
            return str1;
        }
        
        if (str1.length() < str2.length()) {
            String temp = str1;
            str1 = str2;
            str2 = temp;
        }
        
        String remain = getRemain(str1, str2);
        return gcdOfStrings(remain, str2);
    }
    
    private String getRemain(String str1, String str2) {
        if (str1.indexOf(str2) != 0) {
            return "";
        }
        while (str1.indexOf(str2) == 0) {
            str1 = str1.substring(str2.length());
        }
        return str1.length() == 0 ? str2 : str1;
    }
}
