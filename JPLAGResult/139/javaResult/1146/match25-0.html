class Solution {
    public String gcdOfStrings(String str1, String str2) {
        //if(str1.length())
        ArrayList<String> s1 = contains(str1);
        ArrayList<String> s2 = contains(str2);
        String res = "";
        for(int x = 0 ; x < s1.size() ; x++){
            if(s2.contains(s1.get(x))){
                if(s1.get(x).length() > res.length()) res = s1.get(x);
            }
        }
        return res;
    }
    
    public ArrayList<String> contains(String s1){
        ArrayList<String> list = new ArrayList<>();
        for(int x = 0 ; x < s1.length() ; x++){
            if(div(s1 , s1.substring(x))) list.add(s1.substring(x));
        }
        return list;
    }
    
    public boolean div(String s1 , String s2){
        if(s1.length() % s2.length() != 0) return false;
        for(int x = 0 ; x < s1.length() ; x++){
            if(s1.charAt(x) != s2.charAt(x%s2.length())) return false;
        }
        return true;
    }
}
