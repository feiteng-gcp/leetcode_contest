class Solution {
    public String gcdOfStrings(String str1, String str2) {
        
String res = "";
        for (int i = 0; i < str2.length(); i++) {
            String str = str2.substring(0, i + 1);
            if (isDivide(str1, str) && isDivide(str2, str)) {
                res = str;
            }
        }
        return res;
    }
    private boolean isDivide(String str, String s) {
        if (str.length() < s.length() || str.length() % s.length() != 0) return false;
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) != s.charAt(i % s.length())) return false;
        }
        return true;
    }
    
}
