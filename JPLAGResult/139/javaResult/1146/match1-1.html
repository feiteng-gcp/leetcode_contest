class Solution {
    public String gcdOfStrings(String str1, String str2) {
        
        for (int i = str2.length() - 1; i >= 0; i--) {
            
String temp = str2.substring(0, i + 1);
            
            if (isValid(temp, str1) && isValid(temp, str2)) {
                return temp;
            }
        }
        
        return "";
    }
    
    private boolean isValid(String m, String n) {
        if (m.length() > n.length()) {
            return false;
        }
        
        if (m.length() == n.length()) {
            return m.equals(n);
        
}
        
        if (n.length() % m.length() != 0) {
            return false;
        }
        
        for (int i = 0; i < n.length() - m.length(); i += n.length()) {
            
for (int j = 0; j < m.length(); j++) {
                if (m.charAt(j) != n.charAt(i + j)) {
                    return false;
                }
            }
        }
        
        return true;
    }
}
