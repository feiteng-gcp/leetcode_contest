class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int n = str1.length(), m = str2.length();
        for (int len = Math.min(n, m); len >= 1; len--) {
            if (n % len != 0 || m % len != 0) continue;
            boolean valid = true;
            for (int i=0; i<n; i++) {
                if (
str1.charAt(i) != str1.charAt(i % len)) {
                    valid = false;
                    break;
                }
            }
            if (!valid) continue;
            for (int i=0; i<m; i++) {
                if (
str2.charAt(i) != str1.charAt(i % len)) {
                    valid = false;
                    break;
                }
            }
            if (!valid) continue;
            return str1.substring(0, len);
        }
        return "";
    }
}
