class Solution {
    public String gcdOfStrings(String str1, String str2) {
        if(str1.length() == str2.length())
            return str1.equals(str2) ? str1 : "";
        int gcd = findGCD(str1.length(), str2.length());
        String common = str1.substring(0, gcd);
        
        if(check(str1, common) && check(str2, common))
            return common;
        return "";
    }
    
    private int findGCD(int a, int b){
        if(b == 0) return a;
        
return findGCD(b, a % b);
    }
    
    private boolean check(String s, String c){
        int lc = c.length();
        for(int i = 0; i < s.length(); i++)
{
            if(s.charAt(i) != c.charAt(i % lc)) return false;
        }
        return true;
    }
}
