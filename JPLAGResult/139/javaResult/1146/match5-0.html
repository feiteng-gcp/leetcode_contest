class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int n = str1.length();
        int m = str2.length();
        boolean f = true;
        int len = Math.min(n,m);
        while (len>=1){
            if (n%len!=0 || m%len!=0) {
                len = len - 1;
                continue;
            }
            f = true;
            for (int r = 0; r<n/len; r++){
                if (str1.substring(r*len,(r+1)*len).compareTo(str1.substring(0,len))!=0){
                    f = false;
                    break;
                }
            }
            for (int r = 0; r<m/len; r++){
                if (str2.substring(r*len,(r+1)*len).compareTo(str1.substring(0,len))!=0){
                    f = false;
                    break;
                }
            }
            if (f) return str1.substring(0,len); else len = len - 1;
        }
        return "";
    }
}
