class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int length1 = str1.length(), length2 = str2.length();
        int maxGcd = gcd(length1, length2);
        for (int i = maxGcd; i >= 1; i--) {
            if (length1 % i != 0 || length2 % i != 0)
                continue;
            String prefix = str2.substring(0, i);
            if (canDiv(str1, prefix) && canDiv(str2, prefix))
                return prefix;
        }
        return "";
    }
    
    private boolean canDiv(String x, String y) {
        if (x.equals(""))
            return true;
        if (x.startsWith(y))
            return canDiv(x.substring(y.length()), y);
        return false;
    }

    private int gcd(int x, int y) {
        return y > 0 ? gcd(y, x % y) : x;
    }
}
