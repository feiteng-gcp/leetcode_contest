class Solution {
    public String gcdOfStrings(String str1, String str2) {
        
        int len1 = str1.length();
        int len2 = str2.length();
        
        int minLen = Math.min(len1,len2);
        
        String str = "";
        
        for(int i=1;i<=minLen;i++){
            
            String st = str1.substring(0,i);
            
            if(!str2.startsWith(st)){
                break;
            }
            
            
if(canDivide(str1, st) && canDivide(str2,st)){
                str = st;
            }
            
        }
        
        return str;
    }
    
    public boolean canDivide(String a, String b){
        
        int lenA = a.length();
        int lenB = b.length();
        
        if(lenA%lenB>0){
            return false;
        }
        
        for(int i=0;i<lenA;i+=lenB){
            String st = a.substring(i,i+lenB);
            if(!
st.equals(b)){
                return false;
            }
        }
        return true;
    }
}
