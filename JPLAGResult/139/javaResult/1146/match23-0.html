	class Solution {
	    public String gcdOfStrings(String str1, String str2) {
	        int n = str1.length();
	        int m = str2.length();
	        inner:
	        for(int i = Math.min(n, m);i >= 1;i--){
	        	if(n % i == 0 && m % i == 0){
	        		if(ok(str1, i) && ok(str2, i)){
	        			for(int k = 0;k < i;k++){
	        				if(str1.charAt(k) != str2.charAt(k)){
	        					continue inner;
	        				}
	        			}
	        			return str1.substring(0, i);
	        		}
	        	}
	        }
	        return "";
	    }
	    
	    boolean ok(String s, int k)
	    {
	    	int n = s.length();
	    	for(int i = 0;i < n-k;i++){
	    		if(s.charAt(i) != s.charAt(i+k))return false;
	    	}
	    	return true;
	    }
	}
