class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int m=str1.length(), n=str2.length(), K=gcd(m, n);
        
        for(int k=K; k>=Math.sqrt(K); k--){
            if(m%k==0&&n%k==0&&isCommon(str1.substring(0, k), str1, str2)) return str1.substring(0, k);
        }
        return "";
    }
    
    private int gcd(int m, int n){
        if(m>n) return gcd(n, m);
        if(m==0) return n;
        return gcd(n%m, m);
    }
    
    private boolean isCommon(String tgt, String str1, String str2){
        String tmp = "";
        while(tmp.length()<=str1.length() || tmp.length()<=str2.length()){
            if(tmp.length()==str1.length()&&!tmp.equals(str1)) return false;
            if(tmp.length()==str2.length()&&!tmp.equals(str2)) return false;
            tmp += tgt;
        }
        return true;
    }
}
