class Solution {
    public String gcdOfStrings(String str1, String str2) {
        String temp = str1;
        if (str1.length()>str2.length()){
            str1 = str2;
            str2 = temp;
        }
        for(int i=str1.length(); i>=1; i--){
            if (str1.length() % i==0)
{
                String pattern = str1.substring(0,i);
                if (check(str1, pattern) && check(str2, pattern)) return pattern;
            }
        }
        
        return "";
    }
    
    boolean check(String s, String pattern){
        if (s.length() % pattern.length()!=0) return false;
        int len = pattern.length();
        for(int i=0; i<=s.length()-len; i+=len){
            String temp = s.substring(i, i+len);
            if (!
temp.equals(pattern)) return false;
        }
        return true;
    }
}
