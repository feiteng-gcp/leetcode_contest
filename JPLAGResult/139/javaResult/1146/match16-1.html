class Solution {
    private boolean divides(String s, String divisor) {
        if (divisor.length() == 0) return true;
        int dl = divisor.length();
        if (s.length() % dl != 0) return false;
        int times = s.length() / dl;
        for (int i = 0; i < times; i++) {
            if (!s.subSequence(i * dl, (i + 1) * dl).equals(divisor)) return false;
        }
        return true;
    }

    
public String gcdOfStrings(String s1, String s2) {
        String gcd = "";

        for (int i = 1; i <= s1.length(); i++) {
            if (s1.length() % i == 0 && s2.length() % i == 0) {
                String prefix = s1.substring(0, i);
                
if (divides(s2, prefix) && divides(s1, prefix)) {
                    gcd = prefix;
                }
            }
        }

        return gcd;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();

//        System.out.println(sol.gcdOfStrings("ABCABC", "ABC"));
//        System.out.println(sol.gcdOfStrings("ABABAB", "ABAB"));
    }
}
