class Solution {
    public String gcdOfStrings(String str1, String str2) {
        if (str1.length() < str2.length())
            return gcdOfStrings(str2, str1);
        int n1 = str1.length();
        int n2 = str2.length();
        if (n1 % n2 == 0 && divisor(str2, str1)) return str2;
        for (int i = n2 - 1; i >= 1; i--) {
            if (n2 % i == 0 && n1 % i == 0) {
                String s = str2.substring(0, i);
                if (divisor(s, str2) && divisor(s, str1)) return s;
            }
        }
        return "";
    }
    private boolean divisor(String a, String b) {
        int n = a.length();
        for (int i = 0; i *n + n <= b.length(); i++) {
            if (!b.substring(i*n, i*n + n).equals(a)) return false;
        }
        return true;
    }
}
