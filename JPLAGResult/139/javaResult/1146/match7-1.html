class Solution {
    boolean canDivide(String S, String T) {
        if (S.length() % T.length() != 0) return false;

        for (int i = 0; i < S.length(); i += T.length()) {
            
for (int j = i; j < i + T.length(); j++) {
                if (S.charAt(j) != T.charAt(j - i)) return false;
            }
        
}
        return true;
    }

    public String gcdOfStrings(String str1, String str2) {
        int ans = 0;
        for (int len = 1; len <= str1.length(); len++) {
            String s = str1.substring(0, len);
            if (canDivide(str1, s) && canDivide(str2, s)) {
                ans = len;
            }
        }

        return str1.substring(0, ans);
    }

}
