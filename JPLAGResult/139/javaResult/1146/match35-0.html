class Solution {
	public static int N, M;

	public String gcdOfStrings(String str1, String str2) {
		N = str1.length();
		M = str2.length();
		for (int i = Math.min(M,N); i >=1; i--) {
			if (N % i == 0 && M % i == 0) {
				String check = str1.substring(0,i);
				String check2 = str2.substring(0,i);
                //System.out.println(check + " " + check2);
				if (check.equals(check2)) {
					//System.out.println(check + " " + check2);
					if (check(i, str1) && check(i, str2)) {
						return str1.substring(0, i);
					}
				}
			}
		}
		return "";
	}

	public static boolean check(int k, String str) {
        //System.out.println(k+" "+str);
		String check = str.substring(0, k);
		int length = str.length();
		for (int i = 0; i < length; i += k) {
			if (!check.equals(str.substring(i, i + k))) {
				return false;
			}
		}
		return true;
	}
}
