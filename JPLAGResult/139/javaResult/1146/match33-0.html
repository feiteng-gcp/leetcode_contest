class Solution {
    public String gcdOfStrings(String str1, String str2) {
        if (str1.equals(str2)) {
            return str1;
        }
        if (!str1.startsWith(str2) && !str2.startsWith(str1)) {
            return "";
        }
        if (str1.length() < str2.length()) {
            return gcdOfStrings(str2, str1);
        }
        for (int len = str2.length(); len >= 1; --len) {
            if (str2.length() % len != 0) {
                continue;
            }
            String factor = str2.substring(0, len);
            if (isFactorOf(str1, factor) && isFactorOf(str2, factor)) {
                return factor;
            }
        }
        return "";
    }
    
    private boolean isFactorOf(String str, String factor) {
        if (str.length() % factor.length() != 0) {
            return false;
        }
        int len = factor.length();
        for (int i = 0; i * len < str.length(); ++i) {
            if (!str.substring(i * len, (i + 1) * len).equals(factor)) {
                return false;
            }
        }
        return true;
    }
