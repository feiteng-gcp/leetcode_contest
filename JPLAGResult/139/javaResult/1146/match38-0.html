class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int gcd = sgcd(str1.length(), str2.length());
        for(int i=gcd; i>=1; i--) {
            if (gcd%i == 0) {
                String common = str1.substring(0, i);
                if (ism(common, str1) && ism(common, str2)) return common;
            }
        }
        return "";
    }
    
    private boolean ism(String s, String l) {
        StringBuilder sb = new StringBuilder();
        while(sb.length() < l.length()) {
            sb.append(s);
        }
        
        if (sb.toString().equals(l)) return true;
        return false;
    }
    
    private int sgcd(int a, int b) {
        if (a > b) return sgcd(b, a);
        
        if (b % a == 0 ) 
            return a;
        else {
            int r = b % a;
            return sgcd(r, a);
        }
    }
}
