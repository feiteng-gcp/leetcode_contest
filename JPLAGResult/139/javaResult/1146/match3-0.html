class Solution {
    public String gcdOfStrings(String str1, String str2) {
        int N1 = str1.length();
        int N2 = str2.length();
        int N = gcd(N1, N2);
        if (!str1.substring(0, N).equals(str2.substring(0, N))) {
            return "";
        }
        String temp = str1.substring(0, N);
        for (int i = 0; i < N1; i += N) {
            if (!str1.substring(i, i + N).equals(temp)) {
                return "";
            }
        }
        for (int i = 0; i < N2; i += N) {
            if (!str2.substring(i, i + N).equals(temp)) {
                return "";
            }
        }
        return temp;
    }


    int gcd(int a, int b) {
        if (a < b) {
            return gcd(b, a);
        }
        if (b == 0) {
            return a;
        }
        return gcd(b, a % b);
    }
}
