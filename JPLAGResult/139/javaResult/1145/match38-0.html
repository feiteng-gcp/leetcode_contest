class Solution {
    private int[][] presum;
    private int rows;
    private int cols;
    
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        fillPresum(matrix);
        
        int count = 0;
        
        for(int x1 = 0; x1 < rows; x1++) {
            for(int y1 = 0; y1 < cols; y1++) {
                for(int x2 = x1; x2 < rows; x2++) {
                    for(int y2 = y1; y2 < cols; y2++) {
                        if(target == sumRegion(x1, y1, x2, y2)) {
                            count++;
                        }
                    }
                }
            }
        }
        
        return count;
    }
    
    public void fillPresum(int[][] matrix) {
        rows = matrix.length;
        
        cols = rows == 0 ? 0 : matrix[0].length;
        
        presum = new int[rows + 1][cols + 1];
        
        for(int i = 0; i < rows; i++) {
            for(int j = 0; j < cols; j++) {
                presum[i + 1][j + 1] = presum[i][j + 1]
                    + presum[i + 1][j]
                    - presum[i][j]
                    + matrix[i][j];
            }
        }
    }
    
    private int sumRegion(int row1, int col1, int row2, int col2) {
        return presum[row2 + 1][col2 + 1]
            - presum[row2 + 1][col1]
            - presum[row1][col2 + 1]
            + presum[row1][col1];
    }
}
