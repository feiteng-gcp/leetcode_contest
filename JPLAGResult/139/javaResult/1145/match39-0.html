class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        if (matrix == null || matrix.length == 0) {
            return 0;
        }
        
        int m = matrix.length;
        int n = matrix[0].length;
        
        int[][] sum = new int[m][n];
        
        sum[0][0] = matrix[0][0];
        
        for (int i = 1; i < m; i++) {
            sum[i][0] = sum[i - 1][0] + matrix[i][0];
        }
        
        for (int j = 1; j < n; j++) {
            sum[0][j] = sum[0][j - 1] + matrix[0][j];
        }
        
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                sum[i][j] = sum[i - 1][j] + sum[i][j - 1] + matrix[i][j] - sum[i - 1][j - 1];
            }
        }
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                System.out.print(sum[i][j] + " ");
            }
            
            System.out.println();
        }
        
        int result = 0;
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int x = i; x < m; x++) {
                    for (int y = j; y < n; y++) {
                        int temp = sum[x][y];
                        
                        if (i != 0) {
                            temp -= sum[i - 1][y];
                        }
                        
                        if (j != 0) {
                            temp -= sum[x][j - 1];
                        }
                        
                        if (i != 0 && j != 0) {
                            temp += sum[i - 1][j - 1];
                        }
                        
                        if (temp == target) {
                            result++;
                        }
                    }
                }
            }
        }
        
        return result;
    }
}
