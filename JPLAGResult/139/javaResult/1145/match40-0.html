class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int[][] copy = new int[matrix.length][matrix[0].length];
        for (int i = 0; i < matrix.length; i++) {
            int lineSum = 0;
            for (int j = 0; j < matrix[0].length; j++) {
                lineSum += matrix[i][j];
                if (i > 0)
                    copy[i][j] = lineSum + copy[i - 1][j];
                else
                    copy[i][j] = lineSum;
            }
        }
        int count = 0;
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                for (int k = i; k < matrix.length; k++) {
                    for (int h = j; h < matrix[0].length; h++) {
                        if (getSum(copy, i, j, k, h) == target)
                            count++;
                    }
                }
            }
        }
        return count;
    }
    
    private int getSum(int[][] copy, int x1, int y1, int x2, int y2) {
        int result = copy[x2][y2];
        if (y1 > 0)
            result -= copy[x2][y1 - 1];
        if (x1 > 0)
            result -= copy[x1 - 1][y2];
        if (x1 > 0 && y1 > 0)
            result += copy[x1 - 1][y1 - 1];
        return result;
    }
}
