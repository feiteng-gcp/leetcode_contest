class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int N = matrix.length;
        int M = matrix[0].length;
        int[][] sum = new int[M][N];
        for (int i = 0; i < M; i++) {
            sum[i][0] = matrix[0][i];
        }
        
for (int i = 1; i < N; i++) {
            for (int j = 0; j < M; j++) {
                sum[j][i] = matrix[i][j] + sum[j][i - 1];
            }
        }
        // for (int i = 0; i < M; i++) {
        //     for (int j = 0; j < N; j++) {
        //         System.out.println(sum[i][j]);
        //     }
        // }
        int cnt = 0;
        for (int i = 0; i < N; i++) {
            for (int j = i; j < N; j++) {
                Map<Integer, Integer> map = new HashMap<>();
                map.put(0, 1);
                int s = 0;
                for(int k = 0; k < M; k++) {
                    s = s + sum[k][j] - sum[k][i] + matrix[i][k];
                    // System.out.println(i + " " + j + " " + k + " " + s);
                    if (map.containsKey(s - target)) cnt += map.get(s - target);
                    if (!map.containsKey(s)) {
                        map.put(s, 1);
                    } else {
                        map.put(s, map.get(s) + 1);
                    }
                
}
            }
        }
        // System.out.println(sum[1][2] + " " + sum[1][1] + " " + matrix[1][1]);
        return cnt;
    }
}
