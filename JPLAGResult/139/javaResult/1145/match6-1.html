class Solution {
    public int numSubmatrixSumTarget(int[][] A, int T) {
        
int M = A.length, N = A[0].length;
        long[][] sum = new long[M + 1][N + 1];
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                sum[i + 1][j + 1] = sum[i][j + 1] + sum[i + 1][j] - sum[i][j] + A[i][j];
            }
        }
        int res = 0;
        
for (int i = 1; i <= M; i++) {
            for (int j = 1; j <= N; j++) {
                long v = sum[i][j];
                for (int k = 0; k < i; k++) {
                    for (int l = 0; l < j; l++) {
                        long s = v - sum[k][j] - sum[i][l] + sum[k][l];
                        if (s == T) {
                            res++;
                        
}
                    }
                }
            }
        }
            
        return res;
    }
}
