class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        
        int m = matrix.length;
        int n = matrix[0].length;
        
        int[][] sum = new int[m][n];
        
        for(int i=0; i<m; i++){
            for(int j=0; j<n; j++){
                if (i-1>=0) sum[i][j] = sum[i-1][j]+matrix[i][j];
                else sum[i][j] = matrix[i][j];
            }
        }
        
        int count = 0;
        for(int i=-1; i<m-1; i++){
            for(int j=i+1; j<m; j++){
                Map<Integer, Integer> map = new HashMap<>();
                map.put(0,1);
                int s = 0, cur = 0;
                for(int k=0; k<n; k++){
                    if (i<0) cur = sum[j][k];
                    else cur = sum[j][k] - sum[i][k];
                    s += cur;
                    count += map.getOrDefault(s-target, 0);
                    map.put(s, map.getOrDefault(s, 0)+1);
                }
            }
        }
        
        return count;
        
    }
}
