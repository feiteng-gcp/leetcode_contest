class Solution {

	public int numSubmatrixSumTarget(int[][] matrix, int target) {
		int sum[][] = new int[matrix.length][matrix[0].length], count = 0;
		for (int i = 0; i < matrix.length; i++) {
			for (int j = 0; j < matrix[0].length; j++) {
				sum[i][j] = matrix[i][j] + (i > 0 ? sum[i - 1][j] : 0) + (j > 0 ? sum[i][j - 1] : 0)
						- (i > 0 && j > 0 ? sum[i - 1][j - 1] : 0);
			}
		}
		for (int i = 0; i < matrix.length; i++) {
			for (int j = i; j < matrix.length; j++) {
				for (int k = 0; k < matrix[0].length; k++) {
					for (int l = k; l < matrix[0].length; l++) {
						if (sum[j][l] - (i > 0 ? sum[i - 1][l] : 0) - (k > 0 ? sum[j][k - 1] : 0)
								+ (i > 0 && k > 0 ? sum[i - 1][k - 1] : 0) == target) {
							count++;
						}
					}
				}
			}
		}
		return count;
	}
}
