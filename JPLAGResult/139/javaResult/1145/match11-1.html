class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        
int[][] sum = new int[matrix.length + 1][matrix[0].length + 1];
        for (int rowIdx = matrix.length - 1; rowIdx >= 0; rowIdx--) {
            for (int colIdx = matrix[0].length - 1; colIdx >= 0; colIdx--) {
                sum[rowIdx][colIdx] = sum[rowIdx][colIdx + 1] + sum[rowIdx + 1][colIdx] - sum[rowIdx + 1][colIdx + 1] + matrix[rowIdx][colIdx];
            }
        }
        int count = 0;
        for (int y1 = 0; y1 < sum.length; y1++) {
            for (int y2 = y1 + 1; y2 < sum.length; y2++) {
                for(int x1 = 0;x1<sum[0].length;x1++) {
                    for(int x2 = x1+1;x2<sum[0].length;x2++) {
                        if(sum[y1][x1]-sum[y2][x1]-sum[y1][x2]+sum[y2][x2]==target) {
                            count++;
                        
}
                    }
                }
            }
        }
        return count;
    }
}
