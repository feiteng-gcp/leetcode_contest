class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int M = matrix.length, N = matrix[0].length;
        int[][] sum = new int[M+1][N+1];
        int count = 0;
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                sum[i+1][j+1] = sum[i][j+1] + sum[i+1][j] - sum[i][j] + matrix[i][j];
            }
        }
        
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                for (int p = i; p < M; ++p) {
                    for (int q = j; q < N; ++q) {
                        if (sum[p+1][q+1] - sum[i][q+1] - sum[p+1][j] + sum[i][j] == target) count++;
                    } 
                }
            }
        }
        return count;
    }
}
