class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] prefixSumMatrix = new int[m + 1][n + 1];
        //System.out.println(Arrays.toString((prefixSumMatrix[0])));
        for (int i = 1; i <= m; ++i) {
            int rowSum = 0;
            for (int j = 1; j <= n; ++j) {
                rowSum += matrix[i - 1][j - 1];
                prefixSumMatrix[i][j] = prefixSumMatrix[i - 1][j] + rowSum;
            }
            //System.out.println(Arrays.toString((prefixSumMatrix[i])));
        }
        int res = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                for (int ip = i; ip < m; ++ip) {
                    for (int jp = j; jp < n; ++jp) {
                        int sum = prefixSumMatrix[ip + 1][jp + 1] 
                            - prefixSumMatrix[i][jp + 1] 
                            - prefixSumMatrix[ip + 1][j] 
                            + prefixSumMatrix[i][j];
                        if (sum == target) {
                            res++;
                        }
                    }
                }
            }
            //System.out.println(Arrays.toString((prefixSumMatrix[i])));
        }
        
        return res;
    }
}
