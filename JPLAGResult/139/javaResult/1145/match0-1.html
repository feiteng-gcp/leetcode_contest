class Solution {
    public int numSubmatrixSumTarget(int[][] m, int t) {
        
        
        int l = m.length;
        int ll = m[0].length;int all = 0;
        
        for(int i=0;i<l;++i){
            long sum[] = new long[ll];
            for(int j=i;j<l;++j){
                
                for(int k=0;k<ll;++k){
                    
                    sum[k] += (long)m[j][k]; 
                }
                
                long s = 0;
                HashMap<Long,Long> mp = new HashMap<>();
                mp.put(0L,1L);
                for(int k=0;k<ll;++k){
                    s += sum[k];
                    long ck = s-(long)t;
                    if(mp.containsKey(ck)){
                        all += mp.get(ck);
                    }
                    mp.put(s,mp.getOrDefault(s,0L)+1L);
                }
                
                
            }
            
            
            
            
        }
        return all;
        
        
        
    }
}
