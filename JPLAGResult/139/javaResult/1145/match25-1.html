class Solution {
    
    public int area(int[][] areaMatrix, int i, int j, int k, int l) {
        return areaMatrix[k][l] + areaMatrix[i][j] - areaMatrix[i][l] - areaMatrix[k][j];
    }
    
    
public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int n = matrix.length;
        int m = matrix[0].length;
        int sum = 0;
        int[][] areaMatrix = new int[n + 1][m + 1];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                areaMatrix[i + 1][j + 1] = matrix[i][j] + areaMatrix[i + 1][j] + areaMatrix[i][j + 1] - areaMatrix[i][j];
            }
        }
        /*
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= m; j++) {
                System.out.print(areaMatrix[i][j]);
            }
            System.out.println();
        } */
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = i + 1; k <= n; k++) {
                    for (int l = j + 1; l <= m; l++) {
                        if (area(areaMatrix, i, j, k, l) == target) sum++
;
                    }
                } 
            }
        } 
        return sum;
    }
}
