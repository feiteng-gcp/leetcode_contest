class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int r = matrix.length;
        int c = matrix[0].length;
        long[][] sum = new long[r][c];
        for (int i=0; i<r; i++) {
            for (int j=0; j<c; j++) {
                long top = i==0? 0 : sum[i-1][j];
                long left = j==0? 0 : sum[i][j-1];
                long topLeft = i>0 && j>0 ? sum[i-1][j-1] : 0;
                sum[i][j] = matrix[i][j] + top + left - topLeft;
            }
        }
        
        /*
        for (int i=0; i<r; i++) {
            for (int j=0; j<c; j++) {
                System.out.print(sum[i][j]);
            }
            System.out.println();
        }*/
        
        int count = 0;
        for (int i=0; i<r; i++) {
            for (int j=0; j<c; j++) {
                for (int m=0; m<=i; m++) {
                    for (int n=0; n<=j; n++) {
                        long top = m == 0? 0 : sum[m-1][j];
                        long left = n == 0? 0 : sum[i][n-1];
                        long topLeft = m>0 && n>0? sum[m-1][n-1] : 0;
                        long area = sum[i][j] - top - left + topLeft;
                        
                        //System.out.println("i,j="+i+","+j + " m,n="+m+","+n + " area="+area + " top,left,topleft="+top+","+left+","+topLeft);
                        if (area == target) { count++; }
                    }
                }
            }
        }
        return count;
    }
}
