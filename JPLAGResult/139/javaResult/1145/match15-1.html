class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int h = matrix.length, w = matrix[0].length;
        int[][] pre = new int[h][w];
        for (int i = 0; i < h; ++i) {
            for (int j = 0; j < w; ++j) {
                pre[i][j] = (j > 0 ? pre[i][j - 1] : 0) + matrix[i][j];
            
}
        }
        // for (int i = 0; i < h; ++i) {
        //     for (int j = 0; j < w; ++j) {
        //         System.out.print(pre[i][j] + ",");
        //     }
        //     System.out.println();
        // }
        int res = 0;
        for (int r = 0; r < w; ++r) {
            for (int l = 0; l <= r; ++l) {
                // System.out.println(r + "; " + l);
                Map<Integer, Integer> map = new HashMap<>();
                map.put(0, 1);
                int cur = 0;
                for (int i = 0; i < h; ++i) {
                    cur += pre[i][r] - (l > 0 ? pre[i][l - 1] : 0);
                    // System.out.print(cur + ",");
                    int prev = cur - target;
                    if (map.containsKey(prev)) res += map.get(prev)
;
                    map.put(cur, map.getOrDefault(cur, 0) + 1);
                }
                // System.out.println();
            }
        }
        return res;
    }
}
