class Solution {
    public int numSubmatrixSumTarget(int[][] a, int t) {
        int n = a.length;
        int m = a[0].length;
        int[][] sum = new int[n + 1][m + 1];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum[i + 1][j + 1] = a[i][j] + sum[i][j + 1] + sum[i + 1][j] - sum[i][j];
            }
        }
//        System.out.println(Arrays.deepToString(sum));
        int ans = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {

                for (int r = 1; r <= i; r++) {
                    for (int c = 1; c <= j; c++) {
                        int s = sum[i][j] + sum[r - 1][c - 1] - sum[i][c - 1] - sum[r - 1][j];
                        if (s == t) ans++;
                    }
                }

            }
        }
        return ans;
    }
}
