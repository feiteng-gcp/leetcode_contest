class Solution {
    public int numSubmatrixSumTarget(int[][] matrix, int target) {
        int rows = matrix.length;
        int cols = matrix[0].length;
        int[][] sum = new int[rows][cols + 1];
        for (int i = 0; i < rows; i++) {
            for (int j = 1; j <= cols; j++) {
                sum[i][j] = matrix[i][j - 1] + sum[i][j - 1];
            }
        }
        int res = 0;
        for (int y1 = 0; y1 < cols; y1++) {
            for (int y2 = y1; y2 < cols; y2++) {
                int[] subSum = new int[rows + 1];
                Map<Integer, Integer> counts = new HashMap<>();
                counts.put(0, 1);
                for (int x = 1; x <= rows; x++) {
                    subSum[x] = subSum[x - 1] + sum[x - 1][y2 + 1] - sum[x - 1][y1];
                    res += counts.getOrDefault(subSum[x] - target, 0);
                    counts.put(subSum[x], counts.getOrDefault(subSum[x], 0) + 1);
                }
            }
        }
        return res;
    }
}
