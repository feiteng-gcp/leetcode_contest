class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        HashMap<String, Integer> map = new HashMap<>();
        int res = 0;
        for (int i=0; i<matrix.length; i++) {
            String[] pair = helper(matrix[i]);
            map.put(pair[0], map.getOrDefault(pair[0], 0) + 1);
            map.put(pair[1], map.getOrDefault(pair[1], 0) + 1);
            res = Math.max(res, map.get(pair[0]));
            
res = Math.max(res, map.get(pair[1]));
        }
        return res;
    }
    
    private String[] helper(int[] arr) {
        String ones = "", zeros = "";
        for (int i=0; i<arr.length; i++) {
            if (arr[i] == 0) {
                zeros += "," + i;
            } else {
                ones += "," + i;
            }
        }
        return new String[]{zeros, ones};
    }
}
