class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        
        int m = matrix.length, n = matrix[0].length;
        int res = 0;
        for(int i = 0; i < m; i++){
            
            int count  = 0;
            
            for(int j = 0; j < m; j++){
                
                if(i != j && sim(matrix, i, j)){
                    
                    count ++;
                }
                
                
            }
            
            res = Math.max(res, count);
            
        }
        
        return res+1;
        
    }
    
    private boolean sim(int[][] matrix, int r1, int r2){
        
        int n = matrix[0].length;
        
        if(matrix[r1][0] == matrix[r2][0]){
            
            for(int i = 0; i < n; i++){
                
                if(matrix[r1][i] != matrix[r2][i])
                    return false;
            
            
            
            }
        
        }
        
        else {
            
            for(int i = 0; i < n; i++){
                
                if(matrix[r1][i] == matrix[r2][i])
                    return false;
            
            
            
            }
            
        }
        
        return true;
        
        
        
    }
}
