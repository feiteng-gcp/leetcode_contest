class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
         Map<String,Integer> map=new HashMap<>();
        for (int[] m:matrix)
        {
            String key=Arrays.toString(m);
            map.put(key,map.getOrDefault(key,0)+1);
            for (int i=0;i<m.length;i++)
            {
                m[i]=m[i]==0?1:0;
            }
            key=
Arrays.toString(m);
            map.put(key,map.getOrDefault(key,0)+1);
        }
        int res=0;
        for (int i:map.values())
        {
            res=Math.max(res,i);
        }
        return res;
    }
}
