class Solution {
  public int maxEqualRowsAfterFlips(int[][] matrix) {
		List<Integer> changs = new ArrayList<Integer>();// 已经改变的
		int ans = 0;
		for (int i = 0; i < matrix.length; i++) {
			if (changs.contains(i)) {
				continue;
			}
			changeto(matrix, i);
			int ts = getAns(matrix);
			if (ts > ans) {
				ans = ts;
			}
		}
		return ans;
	}

	private int getAns(int[][] matrix) {
		int k = 0;

		for (int j = 0; j < matrix.length; j++) {
			if (allZ(matrix[j])) {
				k++;
			}
		}
		return k;
	}

	private boolean allZ(int[] is) {
		for (int i = 1
; i < is.length; i++) {
			if (is[i] != is[0]) {
				return false;
			}
		}
		return true;
	}

	private void changeto(int[][] matrix, int i) {
		for (int j = 1; j < matrix[i].length; j++) {
			if (matrix[i][j] != matrix[i][0]) {
				change(matrix, j);
			}
		}
	}


	private void change(int[][] matrix, int j) {
		for (int i = 0; i < matrix.length; i++) {
			matrix[i][j] = 1 - matrix[i][j];
		}
	}
}
