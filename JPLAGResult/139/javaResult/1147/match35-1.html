class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        Map<String, Integer> map = new HashMap<String, Integer>();
        for (int[] row : matrix) {
            String rowStr = array2String(row);
            map.put(rowStr, map.getOrDefault(rowStr, 0) + 1);
            int[] inverse = getInverse(row);
            String inverseStr = 
array2String(inverse);
            map.put(inverseStr, map.getOrDefault(inverseStr, 0) + 1);
        }
        int ret = 0;
        for (int value : map.values())
            ret = Math.max(ret, value);
        return ret;
    }

    public String array2String(int[] array) {
        String str = "";
        for (int num : array)
            str += num;
        return str;
    }

    public int[] getInverse(int[] array) {
        int length = array.length;
        int[] inverse = new int[length];
        for (int i = 0; i < length; i++)
            inverse[i] = 1 - array[i];
        return inverse;
    }
}
