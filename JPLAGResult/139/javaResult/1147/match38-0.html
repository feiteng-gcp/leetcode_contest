class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        Map<String,Integer> flipCount = new HashMap<>();
        int x = matrix.length;
        int y = matrix[0].length;
        for (int[] ints : matrix) {
            StringBuilder stringBuilder1 = new StringBuilder();
            StringBuilder stringBuilder2 = new StringBuilder();
            for (int anInt : ints) {
                stringBuilder1.append(anInt);
                stringBuilder2.append(anInt^1);
            }

            Integer oneCount = flipCount.getOrDefault(stringBuilder1.toString(), 0);
            oneCount++;
            flipCount.put(stringBuilder1.toString(),oneCount);
            Integer twoCount = flipCount.getOrDefault(stringBuilder2.toString(), 0);
            twoCount++;
            flipCount.put(stringBuilder2.toString(),twoCount);
        }
        return flipCount.values().stream().max(Integer::compareTo).get();
    }
}
