class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
		Map<String, Integer> map = new HashMap<>();
		for (int i = 0; i < matrix.length; i++) {
			String s = "";
			int a = matrix[i][0];
			if (a == 0) {
				for (int j = 0; j < matrix[0].length; j++) {
					s += matrix[i][j];
				}
			} else {
				for (int j = 0; j < matrix[0].length; j++) {
					s += (1 - matrix[i][j]);
				}
			}
			map.put(s, map.getOrDefault(s, 0) + 1);
		}
		
		int max = 1;
		for (int v : map.values()) {
			max = Math.max(max, v);
		}
		return max;
    }
}
