class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        Map<String, Integer> map = new HashMap<>();
        int max = 0;
        for(int[] row: matrix){
            String zeros = getRow(row, 0);
            String ones = getRow(row, 1);
            map.put(zeros, map.getOrDefault(zeros, 0) + 1);
            map.put(ones, map.getOrDefault(ones, 0) + 1);
            max = 
Math.max(max, map.get(zeros));
            max = Math.max(max, map.get(ones));
        }
        
        return max;
    }
    
    private String getRow(int[] row, int num){
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < row.length; i++){
            if(row[i] == num) sb.append(i+",");
        }
        return sb.toString();
    }
}
