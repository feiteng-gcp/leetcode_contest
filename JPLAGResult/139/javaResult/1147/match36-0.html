class Solution {
	public static int N, M;
	public static int[][] map, orig;

	public int maxEqualRowsAfterFlips(int[][] matrix) {
		N = matrix.length;
		M = matrix[0].length;
		map = new int[N][M];
		orig = new int[N][M];
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				map[i][j] = matrix[i][j];
				orig[i][j] = matrix[i][j];
			}
		}
		int res = 0;
		for (int i = 0; i < N; i++) {
			res = Math.max(res, solve(i));
		}
		return res;
	}

	public static int solve(int i) {T color="#f63526">
		for (int j = 0; j < M; j++) {
			if (map[i][j] != 0) {
				flip(j);
			}
		}
        int res = count();
		for (int k = 0; k < N; k++) {
			for (int j = 0; j < M; j++) {
				map[k][j] = orig[k][j];
			}
		}
		return res;
	}

	public static void flip(int j) {
		for (int i = 0; i < N; i++) {
			map[i][j] = (map[i][j] + 1) % 2;
		}
	}

	public static int count() {
		int res = 0;
		for (int[] is : map) {
			int curr = 0;
			for (int i : is) {
				curr += i;
			}
			if (curr == 0 || curr == M)
				res++;
		}
		return res;
	}

	public static void printArray(int[] array) {
		for (int i : array) {
			System.out.print(i + " ");
		}
		System.out.println();
	}

	public static void printMatrix(int[][] matrix) {
		for (int[] is : matrix) {
			printArray(is);
		}
		System.out.println();
	}
}
