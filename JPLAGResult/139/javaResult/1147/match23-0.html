class Solution {

	private Map<String , Integer> cntMap = new HashMap<>();
	
	private String getString(int[] array) {
		StringBuilder builder = new StringBuilder();
		for (int value : array) {
			builder.append(value);
		}
		return builder.toString();
	}
	
	private String getFlipString(int[] array) {
		StringBuilder builder = new StringBuilder();
		for (int value : array) {
			builder.append(1 - value);
		}
		return builder.toString();		
	}

    public int maxEqualRowsAfterFlips(int[][] matrix) {
    	int ans = 0;
    	int i , j , n = matrix.length , m = matrix[0].length;
    	for (i = 0;i < n;i ++) {
    		String string = getString(matrix[i]);
    		cntMap.put(string , cntMap.getOrDefault(string , 0) + 1);
    		string = getFlipString(matrix[i]);
    		cntMap.put(string , cntMap.getOrDefault(string , 0) + 1);
    	}    	
    	for (int cnt : cntMap.values()) {
    		ans = Math.max(ans , cnt);
    	}
    	return ans;
    }
}
