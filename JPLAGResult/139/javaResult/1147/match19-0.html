class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        boolean[] used = new boolean[matrix.length];
        int ans = 1;
        for(int i = 0; i < matrix.length; ++i) {
            int count = 1;
            for(int j = i + 1; j < matrix.length; ++j) {
                if(used[j])
                    continue;
                int k = 0;
                while(k < matrix[0].length && matrix[i][k] == matrix[j][k]) {
                    ++k;
                }
                int l = 0;
                while(l < matrix[0].length && matrix[i][l] != matrix[j][l]) {
                    ++l;
                }
                if(k == matrix[0].length || l == matrix[0].length) {
                    count++;
                    used[j] = true;
                }
            }
            ans = Math.max(ans, count);
        }
        return ans;
    }
}
