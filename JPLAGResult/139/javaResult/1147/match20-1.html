class Solution {
boolean check(int[] a, int[] flag) {
        int[] t = a.clone();
        int n = t.length;
        for (int i = 0; i < n; i++) {
            if (flag[i] == 1) t[i] = 1 - t[i];
        }


        for (int i = 0; i < n - 1; i++) {
            if (t[i] != t[i + 1]) return false;
        }
        return true;
    }

    public int maxEqualRowsAfterFlips(int[][] matrix) {
        
int m = matrix.length;
        
        int ans = 0;
        for (int i = 0; i < m; i++) {
            int count = 0;
            for (int j = 0; j < m; j++) {
                if (check(matrix[j], matrix[i])) {
                    count++;
                
}
            }
            ans = Math.max(ans, count);
        }
        return ans;
    }
}
