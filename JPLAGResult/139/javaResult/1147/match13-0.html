class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        HashMap<String, Integer> map =  new HashMap<>();
        int rv = 0;
        for(int i=0; i<matrix.length; i++) {
            String s = getString(matrix[i]);
            int tmp = map.getOrDefault(s, 0);
            if (tmp+1 > rv) rv = tmp+1;
            map.put(s, tmp+1);
        }
        return rv;
    }
    
    private String getString(int[] row) {
        int f = row[0];
        StringBuilder rv = new StringBuilder();
        for(int i=0; i<row.length; i++) {
            if (f==0) {
                rv.append(Integer.toString(row[i]));
            } else {
                rv.append(Integer.toString(1-row[i]));
            }
        }
        return rv.toString();
    }
}
