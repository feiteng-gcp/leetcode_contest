class Solution {
    public int maxEqualRowsAfterFlips(int[][] matrix) {
        Map<String, Integer> count = new HashMap<>();
        
        for(int[] row : matrix){
            String[] group = getGroups(row);
            count.put(group[0], count.getOrDefault(group[0], 0) + 1);
            
count.put(group[1], count.getOrDefault(group[1], 0) + 1);
        }
        
        int res = 0;
        for(String key : count.keySet()){
            if(count.get(key) > res) res = count.get(key);
        }
        return res;
    }
    
    private String[] getGroups(int[] row){
        StringBuilder sb1 = new StringBuilder();
        StringBuilder sb2 = new StringBuilder();
        sb1.append("-");
        sb2.append("-");
        for(int i = 0; i < row.length; i++){
            if(row[i] == 0) sb1.append(i + "-");
            else sb2.append(i + "-");
        }
        return new String[]{sb1.toString(), sb2.toString()};
    }
}
