	class Solution {
	    public int maxEqualRowsAfterFlips(int[][] matrix) {
	        int n = matrix.length, m = matrix[0].length;
	        int ret
 = 0;
	        for(int i = 0;i < n;i++){
	        	int ct = 0;
	        	inner:
	        	for(int j = i;j < n;j++){
	        		if(Arrays.equals(matrix[i], matrix[j])){
	        			ct++;
	        		}else{
	        			for(int k = 0;k < m;k++){
	        				if(matrix[i][k] + matrix[j][k] != 1)continue inner;
	        			}
	        			ct++;
	        		}

	        	}
		        ret = Math.max(ret, ct);
	        }
		    return ret;
	    }
	}
