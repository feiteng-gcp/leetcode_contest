class Solution {

	public int maxEqualRowsAfterFlips(int[][] matrix) {
		HashMap<String, Integer> map = new HashMap<>();
		for (int[] m : matrix) {
			map.put(m[0] == 0 ? Arrays.toString(m)
					: Arrays.toString(m).replaceAll("0", "-").replaceAll("1", "0").replaceAll("-", "1"),
					map.getOrDefault(
							m[0] == 0 ? Arrays.toString(m)

									: Arrays.toString(m).replaceAll("0", "-").replaceAll("1", "0").replaceAll("-", "1"),
							0) + 1);
		}
		int max = 0;
		for (int value : map.values()) {
			max = Math.max(max, value);
		}
		return max;
	}
}
