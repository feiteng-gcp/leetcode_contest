class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        int n1 = arr1.length;
        int n2 = arr2.length;
        if(n1 < n2)
            return addNegabinary(arr2, arr1);
        
        int pre = 0;
        List<Integer> list = new ArrayList<>();
        
for(int i = n1 - 1, j = n2 - 1; i >= 0 || pre != 0; --i, --j) {
            int now = pre;
            if(i >= 0)
                now += arr1[i];
            if(j >= 0) 
                now += arr2[j];

            if(now == -1) {
                now = 1;
                pre = 1;
            } else if(now >= 2) {
                now -= 2;
                pre = -1;
            } else {
                pre = 0;
            }
            list.add(0, now);
        }
        
while(list.size() > 0 && list.get(0) == 0)
            list.remove(0);
        if(list.size() == 0)
            return new int[]{0};
        int[] ans = new int[list.size()];
        for(int i = 0; i < ans.length; ++i)
            ans[i] = list.get(i);
        return ans;
    }
}
