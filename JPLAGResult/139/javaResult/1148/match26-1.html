class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        int[] result = new int[Math.max(arr1.length, arr2.length) + 2];
        reverse(arr1);
        reverse(arr2);
        
for (int i = 0; i < result.length - 2; i++) {
            if (i < arr1.length) {
                result[i] += arr1[i];
            }
            if (i < arr2.length) {
                result[i] += arr2[i];
            }
            if (result[i] == 2 || result[i] == 3) {
                if (result[i + 1] > 0) {
                    result[i] -= 2;
                    result[i + 1]--;
                }
                else {
                    result[i + 1]++;
                    result[i + 2]++;
                    result[i] -= 2;
                }
            }
            if (result[i] == 4) {
                result[i + 2]++;
                result[i] = 0;
            }
        }
        return helper(result);
                
    }
    
    private int[] helper(int[] result) {
        int length = getLen(result);
        
if (length == 0) {
            return new int[]{0};
        }
        int[] finalRes = new int[length];
        for (int i = 0; i < finalRes.length; i++) {
            finalRes[i] = result[i];
        }
        reverse(finalRes);
        return finalRes;
    }
    
    private int getLen(int[] result) {
        for (int i = result.length - 1; i >= 0; i--) {
            if (result[i] != 0) {
                return i + 1;
            }
        }
        return 0;
    }
    
    private void reverse(int[] arr) {
        for (int i = 0; i < arr.length / 2; i++) {
            swap(arr, i, arr.length - 1 - i);
        }
    }
    
    private void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
