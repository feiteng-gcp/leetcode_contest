class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        // arr1 = new int[]{1};
        // arr2 = new int[]{1,1};
        int len = Math.max(arr1.length, arr2.length) + 2;
        int[] arr = new int[len];
        int i = arr1.length - 1, j = arr2.length - 1;
        int one = 0, k = len;
        while (i > -1 && j > -1) {
            int n = arr1[i] + arr2[j];
            n += one;
            
if (n > 1) {
                n -= 2;
                one = -1;
            }
            else if (n == -1) {
                n = 1;
                one = 1;
            }
            else one = 0;
            arr[--k] = n;
            i--;j--;
        }
        
        while (i > -1) {
            
            int n = arr1[i];
            // System.out.printf("%d %d %d \n", n, i, arr[i]);
            n += one;
            
if (n > 1) {
                n -= 2;
                one = -1;
            }
            else if (n == -1) {
                n = 1;
                one = 1;
            }
            else one = 0;
        // System.out.printf("%d %d %s\n", n, one, i);
            arr[--k] = n;
            i--;
        }
        // System.out.printf("%d %d %d\n",i , j, k);
        while (j > -1) {
            int n = arr2[j];
            n += one;
            if (n > 1) {
                n -= 2;
                
one = -1;
            }
            else if (n == -1) {
                n = 1;
                one = 1;
            }
            else one = 0;
            arr[--k] = n;
            j--;
        }
        
if (one == -1) {
            arr[--k] = 1;
            arr[--k] = 1;
        }
        else if (one == 1) {
            arr[--k] = 1;
        }
        if (k == 0) return arr;
        while (k < len && arr[k] == 0) k++;
        if (k == len) return new int[]{0};
        int[] res = new int[len - k];
        for (int l = 0; l < res.length; l++) {
            res[l] = arr[l + k];
        }
        return res;
    }
}
