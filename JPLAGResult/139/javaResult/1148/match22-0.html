class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        int n = arr1.length;
        int m = arr2.length;
        if (n == 0) {
            return arr2;
        } 
        if (m == 0) {
            return arr1;
        }
        List<Integer> result = new LinkedList<>();
        
        int carry = 0;
        int i = n - 1;
        int j = m - 1;
        while (i >= 0 || j >= 0 || carry != 0) {
            int sum = (i >= 0 ? arr1[i--] : 0) + (j >= 0 ? arr2[j--] : 0) + carry;
            
            carry = 0; // reset
            if (sum == 0) {
                result.add(0, 0);
            } else if (sum == 1) {
                result.add(0, 1);
            } else if (sum == 2) {
                carry = -1;
                result.add(0, 0);
            } else if (sum == -1) {
                carry = 1;
                result.add(0, 1);
            } else if (sum == 3) {
                carry = -1;
                result.add(0, 1);
            }
        }
        
        Iterator<Integer> it = result.iterator();
        while (it.hasNext()) {
            if (it.next() == 0) {
                it.remove();
            } else {
                break;
            }
        }
        if (result.size() == 0) {
            return new int[] {0};
        }
        int[] x = new int[result.size()];
        int index = 0;
        for (int v : result) {
            x[index++] = v;
        }
        return x;
    }
}
