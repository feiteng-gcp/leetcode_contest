class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        List<Integer> list = new LinkedList<Integer>();
        
        int i = arr1.length - 1, j = arr2.length - 1;
        
        int carry = 0, a = 0, b = 0;
        
        while (i >= 0 || j >= 0 || carry != 0) {
            
            if (i >= 0) {
                a = arr1[i];
            } else {
                a = 0;
            }
            
            if (j >= 0) {
                b = arr2[j];
            } else {
                b = 0;
            }
            
            
int sum = carry + a + b;
            
            if (sum == 0) {
                list.add(0);
                carry = 0;
            } else if (sum == 1) {
                list.add(1);
                carry = 0;
            } else if (sum == 2) {
                list.add(0);
                carry = -1;
            } else if (sum == 3) {
                list.add(1);
                carry = -1;
            } else { // sum = -1
                list.add(1);
                carry = 1;
            }
            
            i--;
            j--;
        }
        
        int[] ans = new int[list.size()];
        
        Integer[] intList = list.toArray(new Integer[] { });
        
        for (int k = 0; k < ans.length; k++) {
            ans[list.size() - 1 - k] = intList[k];
        }
        
        int k = 0;
        for (k = 0; k < ans.length; k++) {
            if (ans[k] != 0) {
                break;
            }
        }
        
        
if (k == ans.length) {
            return new int[] {0};
        } else {
            return Arrays.copyOfRange(ans, k, ans.length);
        }
    }
}
