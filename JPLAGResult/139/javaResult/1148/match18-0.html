import java.math.BigInteger;

class Solution {

	public int[] addNegabinary(int[] arr1, int[] arr2) {
		BigInteger x = BigInteger.ZERO, y = BigInteger.ZERO;
		for (int i : arr1) {
			x = x.multiply(BigInteger.valueOf(-2)).add(BigInteger.valueOf(i));
		}
		for (int i : arr2) {
			y = y.multiply(BigInteger.valueOf(-2)).add(BigInteger.valueOf(i));
		}
		char[] c = baseNeg2(x.add(y)).toCharArray();
		int[] result = new int[c.length];
		for (int i = 0; i < result.length; i++) {
			result[i] = c[i] - '0';
		}
		return result;
	}

	public String baseNeg2(BigInteger N) {
		StringBuilder res = new StringBuilder();
		while (!N.equals(BigInteger.ZERO)) {
			res.append(N.and(BigInteger.ONE));
			N = N.shiftRight(1).negate();
		}
		return res.length() > 0 ? res.reverse().toString() : "0";
	}
}
