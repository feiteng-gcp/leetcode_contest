class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        List<Integer> arr = new LinkedList<>();
        int p = arr1.length-1, q = arr2.length-1;
        while(p >= 0 || q >= 0){
            if(p < 0) arr.add(0, arr2[q--]);
            else if(q < 0) arr.add(0, arr1[p--]);
            else arr.add(0, arr1[p--] + arr2[q--]);
        }
        p = arr.size()-1;
        int carry = 0;
        while(p >= 0){
            arr.set(p, arr.get(p) + carry);
            carry = 0;
            if(arr.get(p) > 1){
                carry = -(arr.get(p) / 2);
                arr.set(p, arr.get(p) % 2);
            }
            else if(arr.get(p) < 0){
                while(arr.get(p) < 0){
                    carry++;
                    arr.set(p, arr.get(p) + 2);
                }
            }
            p--;
        }
        
        while(carry != 0){
            int newcarry = 0;
            if(carry < 0){
                while(carry < 0){
                    carry += 2;
                    newcarry++;
                }
            }
            else if(carry > 1){
                while(carry > 1){
                    carry -= 2;
                    newcarry--;
                }
            }
            arr.add(0, carry);
            carry = newcarry;
        }
        
        List<Integer> res = new LinkedList<>();
        int ind = 0;
        while(ind < arr.size() && arr.get(ind) == 0) ind++;
        while(ind < arr.size()) res.add(arr.get(ind++));
        
        if(res.size() == 0) return new int[]{0};
        
        return res.stream().mapToInt(i->i).toArray();
    }
}
