class Solution {
public int[] addNegabinary(int[] arr1, int[] arr2) {
        List<Integer> al = new ArrayList<>();
        int carry = 0, i = arr1.length -1, j = arr2.length-1, cur;
        
        while (i >= 0 || j >= 0 || carry != 0) {
            cur = (i >= 0) ? arr1[i] : 0;
            cur += (j >= 0) ? arr2[j] : 0;
            cur += carry;
            
            System.out.println("carry: " + carry + " cur: " + cur);
            
            al.add(Math.abs(cur % (-2)));
            if (cur == -1) carry = 1;
            else carry = cur/(-2);
            i--; j--;
        }
        
        Collections.reverse(al);
        
        List<Integer> ret = new ArrayList<>();
        for (int elem: al) {
            if (elem == 0 && ret.size() == 0) continue;
            ret.add(elem);
        }
        
if (ret.size() == 0) return new int[] {0};
        return ret.stream().mapToInt(k->k).toArray();
    }
}
