	class Solution {
	    public int[] addNegabinary(int[] arr1, int[] arr2) {
	    	java.math.
BigInteger A = java.math.BigInteger.ZERO;
	    	for(int v : arr1){
	    		A = A.multiply(java.math.BigInteger.valueOf(-2)).add(java.math.BigInteger.valueOf(v));
	    	}
	    	java.math.BigInteger B = java.math.BigInteger.ZERO;
	    	for(int v 
: arr2){
	    		B = B.multiply(java.math.BigInteger.valueOf(-2)).add(java.math.BigInteger.valueOf(v));
	    	}
	    	A = A.add(B);
	    	if(A.signum() == 0){
	    		return new int[]{0};
	    	}
	    	int[] ret = new int[2000];
	    	int last = -1;
	    	for(int i = 0;i < 2000;i++){
	    		if(A.testBit(0)){
	    			ret[i] = 1;
	    			last = i;
	    			A = A.subtract(java.math.BigInteger.ONE);
	    		}
	    		A = A.divide(java.math.BigInteger.valueOf(-2));
	    	}
	    	ret = Arrays.copyOf(ret, last+1);
	        return rev(ret);
	    }
	    
		public int[] rev(int[] a)
		{
			int[] b = new int[a.length];
			for(int i = 0;i < a.length;i++)b[a.length-1-i] = a[i];
			return b;
		}
		

	}
