class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        List<Integer> list = new ArrayList<>();
        int i = arr1.length-1, j = arr2.length-1; int carry = 0;
        while (i >= 0 || j >= 0 || carry != 0) {
            int a = (i >= 0) ? arr1[i] : 0;
            int b = (j >= 0) ? arr2[j] : 0;
            int sum = a + b + carry;
            int digit = 0;
            if (sum == 0) {
                carry = 0;
                
digit = 0;
            } else if (sum == -1) {
                digit = 1;
                carry = 1;
            } else if (sum == -2) {
                digit = 0;
                carry = 1;
            } else if (sum == 1) {
                digit = 1;
                carry = 0;
            } else if (sum == 2) {
                digit = 0;
                carry = -1;
            } else {
                digit = 1;
                carry = -1;
            }
            list.add(digit);
            i--; j--;
        }
        
        // remove 0's
        int index = list.size()-1;
        while (index >= 0 && list.get(index)==0) {
            index--;
        }
        if (index < 0) {
            return new int[]{0};
        }
        
        int[] res = new int[index+1];
        for (i=0; i<res.length; i++) 
{
            res[i] = list.get(index);
            index--;
        }
        return res;
    }
}
