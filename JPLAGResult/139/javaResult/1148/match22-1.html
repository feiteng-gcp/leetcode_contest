class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        
List<Integer> ans = new ArrayList<>();
        int i = arr1.length - 1, j = arr2.length - 1;
        int rem = 0;
        while (i >= 0 || j >= 0 || rem != 0) {
            int a = i >= 0 ? arr1[i] : 0;
            int b = j >= 0 ? arr2[j] : 0;
            int sum = a + b + rem;
            if (sum == 1 || sum == 0) {
                rem = 0;
                ans.add(sum);
            } else if (sum == 2) {
                rem = 1;
                
ans.add(0);
            } else if (sum == 3) {
                rem = 1;
                ans.add(1);
            } else if (sum == -1) {
                rem = -1;
                ans.add(1);
            } else if (sum == -3) {
                rem = -1;
                ans.add(1);
            }
            i--;
            j--;
            // System.out.println(rem);
            rem = -1 * rem;
        }
        for (i = ans.size() - 1; i >= 0; i--) {
            if (ans.get(i) != 0) break;
        }
        if (i == -1) return new int[]{0};
        int[] res = new int[i + 1];
        j = i;
        while (i >= 0) {
            res[j - i] = ans.get(i);
            i--;
        }
        return res;
    }
    
    boolean helper(List<Integer> l ) {
        for (int i : l) {
            if (i != 0) return true;
        }
        return false;
    }
}
