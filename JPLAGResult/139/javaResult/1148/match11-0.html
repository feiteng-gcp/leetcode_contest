class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        Stack<Integer> stack = new Stack<>();
        int res = 0;
        for(int i = arr1.length-1, j = arr2.length-1; i >= 0 || j >= 0; i--, j--){
            int curr = res;
            if(i >= 0) curr += arr1[i];
            if(j >= 0) curr += arr2[j];
            res = - (curr / 2);
            curr = curr % 2;
            if(curr == -1){
                curr = 1;
                res += 1;
            } 
            stack.add(curr);
        }
        while(res != 0){
            int curr = res % 2;
            res = -curr/2;
            if(curr == -1){
                curr = 1;
                res += 1;
            }
            stack.add(curr);
        }
        while(!stack.isEmpty() && stack.peek() == 0) stack.pop();
        if(stack.isEmpty()) return new int[]{0};
        int[] result = new int[stack.size()];
        for(int i = 0; i < result.length; i++) result[i] = stack.pop();
        return result;
    }
    
}
