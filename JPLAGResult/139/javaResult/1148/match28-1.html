class Solution {
   
	public int[] addNegabinary(int[] arr1, int[] arr2) {
		int[] arr = new int[arr1.length > arr2.length ? arr1.length
				: arr2.length];
		int n = 0;
		for (int i = 0; i < arr1.length || i < arr2.length; i++) {
			int a = 0;
			if (i < arr1.length) {
				a = arr1[arr1.length - 1 - i];
			}
			int b = 0;
			if (i < arr2.length) {
				b = arr2[arr2.length - 1 - i];
			}
			int nn = a + b + n;
			arr[i] = (nn + 2) % 2;
			if (nn == 2 || nn == 3) {
				n = -1;
			} else if (nn == -1) {

				n = 1;
			} else {
				n = 0;
			}
		}
		System.out.println(n);
		if (n == -1) {

			int[] anss = new int[arr.length + 2];
			anss[0] = 1;
			anss[1] = 1;
			for (int i = 0; i < arr.length; i++) {
				anss[i + 2] = arr[arr.length - 1 - i];
			}
			return anss;
		}
		if (n == 1) {

			int[] anss = new int[arr.length + 1];
			anss[0] = 1;
			for (int i = 0; i < arr.length; i++) {
				anss[i + 1] = arr[arr.length - 1 - i];
			}
			return anss;
		}
		int end = getE(arr);
		if (end == arr.length) {
			return new int[] { 0 };
		}
		int ans[] = new int[arr.length - end];
		for (int i = 0; i < ans.length; i++) {
			ans[i] = arr[arr.length - end - 1 - i];
		}
		return ans;
	}

	private int getE(int[] arr) {
		int k = 0;
		for (int i = 0; i < arr.length; i++) {
			if (arr[arr.length - 1 - i] == 0) {
				k++;
			}else{
				break;
			}
		}
		return k;
	}
}
