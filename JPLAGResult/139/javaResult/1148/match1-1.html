class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        if (arr1.length < arr2.length) return addNegabinary(arr2, arr1);
        
LinkedList<Integer> list = new LinkedList<>();
        int i = arr1.length - 1;
        int j = arr2.length - 1;
        int c = 0;
        while (i >= 0 && j >= 0) {
            int sum = arr1[i] + arr2[j] + c;
            if (sum < 0) {
                c = 1;
                list.add(0, 2 + sum);
            } else if (sum < 2) {
                c = 0;
                list.add(0, sum);
            } else {
                c = -1;
                list.add(0, sum - 2);
            }
            i--;
            j--;
        }
        
while (i >= 0) {
            int sum = arr1[i] + c;
            if (sum < 0) {
                c = 1;
                list.add(0, 2 + sum);
            } else if (sum < 2) {
                c = 0;
                list.add(0, sum);
            } else {
                c = -1;
                list.add(0, sum - 2);
            }
            i--;
        }
        while (c != 0) {
            if (c < 0) {
                list.add(0, 2 + c);
                c = 1;
            } else if (c < 2) {
                list.add(0, c);
                c = 0;
            } else {
                list.add(0, c - 2);
                c = -1;
            }
        }
        while (list.size() > 1 && list.get(0) == 0) 
            list.remove(0);
        int[] ret = new int[list.size()];
        for (i = 0; i < ret.length; i++) ret[i] = list.get(i);
        return ret;
    }
}
