class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        if (arr1.length < arr2.length) {
            int[] t = arr1.clone();
            arr1 = arr2.clone();
            arr2 = t.clone();
        }
        
        int idx1 = arr1.length - 1, idx2 = arr2.length - 1;
        int[] temp = new int[Math.max(arr1.length, arr2.length)];
        
        while (idx1 >= 0 || idx2 >= 0) {
            if (idx2 < 0) {
                temp[idx1] = arr1[idx1];
                idx1 --;
            } else {
                temp[idx1] = arr1[idx1] + arr2[idx2];
                idx1 --;
                idx2 --;
            }
        }
        
        int flag = 0;
        for (int i = temp.length - 1; i >= 0; i --) {
            temp[i] += flag;
            flag = 0;
            if (temp[i] == 2) {
                temp[i] = 0;
                flag = -1;
            } else if (temp[i] == -1) {
                temp[i] = 1;
                flag = 1;
            } else if (temp[i] == 3) {
                temp[i] = 1;
                flag = -1;
            }
        }
        
        if (flag == 0) {
            int idx = -1;
            for (int i = 0; i < temp.length; i ++) {
                if (temp[i] != 0) {
                    idx = i;
                    break;
                }
            }
            
            if (idx == -1) {
                return new int[]{0};
            }
            return Arrays.copyOfRange(temp, idx, temp.length);
        } else if (flag == 1) {
            int[] rst = new int[temp.length + 1];
            rst[0] = 1;
            for (int i = 0; i < temp.length; i ++) {
                rst[i + 1] = temp[i];
            }
            return rst;
        } else {
            int[] rst = new int[temp.length + 2];
            rst[0] = 1;
            rst[1] = 1;
            for (int i = 0; i < temp.length; i ++) {
                rst[i + 2] = temp[i];
            }
            return rst;
        }
    }
}
