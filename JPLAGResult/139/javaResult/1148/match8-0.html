class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        int N1 = arr1.length;
        int N2 = arr2.length;
        int i = N1 - 1;
        int j = N2 - 1;
        List<Integer> digs = new ArrayList<>();
        int c = 0;
        while (c != 0 || i >= 0 || j >= 0) {
            int t = c;
            if (i >= 0) {
                t += arr1[i--];
            }
            if (j >= 0) {
                t += arr2[j--];
            }
            if (t == 0 || t == 1) {
                digs.add(t);
                c = 0;
            } else if (t == -1) {
                digs.add(1);
                c = 1;
            } else if (t == -2) {
                digs.add(0);
                c = 1;
            } else if (t == 2) {
                digs.add(0);
                c = -1;
            } else if (t == 3) {
                digs.add(1);
                c = -1;
            }
        }
        int N = digs.size();
        while (N > 0 && digs.get(N - 1) == 0) {
            N--;
        }
        if (N == 0) {
            return new int[]{0};
        }
        int[] res = new int[N];
        for (int k = 0; k < N; k++) {
            res[k] = digs.get(N - 1 - k);
        }
        return res;
    }
}
