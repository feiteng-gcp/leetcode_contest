class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        
		ArrayList<Integer> list = new ArrayList<Integer>();
		
		int i1 = arr1.length-1;
		int i2 = arr2.length-1;
		int flag = 0;
		while(i1 >= 0 && i2 >= 0)
		{
			int x = arr1[i1] + arr2[i2] - flag;
			if(x >= 2)
			{
				flag= 1;
				list.add(0, x-2);
			}
			else if(x == -1)
			{
				flag = -1;
				list.add(0, 1);
			}
			else
			{
				flag = 0;
				list.add(0, x);
			}
			i1 --;
			i2 --;
		}
		
		if(i1 >= 0)
		{
			while(i1 >= 0)
			{
				int x = arr1[i1] - flag;
				if(x >= 2)
			{
				flag= 1;
				list.add(0, x-2);
			}
				else if(x == -1)
				{
					flag = -1;
					list.add(0, 1);
				}
				else
				{
					flag = 0;
					list.add(0, x);
				}
				i1 --;
			}
			
		}
		
		if(i2 >= 0)
		{
			while(i2 >= 0)
			{
				int x = arr2[i2] - flag;
				if(x >= 2)
			{
				flag= 1;
				list.add(0, x-2);
			}
				else if(x == -1)
				{
					flag = -1;
					list.add(0, 1);
				}
				else
				{
					flag = 0;
					list.add(0, x);
				}
				i2 --;
			}
		}
		
		flag = -flag;
		if(flag == -1)
		{
			list.add(0, 1);
			list.add(0, 1);
		}
		else if(flag == 1)
			list.add(0, 1);
		
        int index = 0;
		while(index < list.size() && list.get(index) == 0)
			index ++;
		
		if(index == list.size())
			index --;
		
		int[] re = new int[list.size()-index];
		
		for(int i = index ; i < list.size() ; ++ i)
			re[i-index] = list.get(i);
		
		return re;
    }
}
