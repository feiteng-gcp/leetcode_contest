class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        
        int m = arr1.length-1, n = arr2.length-1;
        
        List<Integer> res = new ArrayList();
        
        int count = 0;
        
        while(m >= 0 || n >= 0) {
            
           
                
            int sum = (m >= 0? arr1[m]:0) + (n >= 0? arr2[n]: 0) + count;

            
if(sum >= 0 && sum < 2){

                res.add(0,sum);
                count = 0;

            }

            else if(sum == 2){

                res.add(0, 0);
                count = -1;
            }

            else if(sum < 0){
                
                res.add(0, 1);
                
                count = 1;



            }
            
            else if(sum > 2){
                 
                res.add(0, 1);
                
                count = -1;

                
                
            }
            
            m--;
            n--;


        }
        
        if(count == -1){
            
            res.add(0,1);
            res.add(0,1);
            
        }
        else if(count == 1) {
            
            res.add(0,1);
        }
        
        
        while(res.size()>=2 && res.get(0)==0){
            
            res.remove(0);
            
        }

        
        
        
        
        int[] ans = new int[res.size()];
        
     
        for(int j = 0; j < ans.length; j++){
            
            ans[j] = res.get(j);
        }
        
        return ans;
            
            
               
        
    }
}
