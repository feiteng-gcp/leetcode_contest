class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        if (arr1.length < arr2.length) {
            int[] tmp = arr1;
            arr1 = arr2;
            arr2 = tmp;
        }
        
        List<Integer> res = new ArrayList<>();
        
        int carry = 0;
        int lenDiff = arr1.length - arr2.length;
        for (int i = arr2.length - 1; i >= 0; --i) {
            int cur = arr2[i] + arr1[lenDiff + i] + carry;
            
            if (cur == -1) {
                cur = 1;
                carry = 1;
            } else if (cur == 2) {
                cur = 0;
                carry = -1;
            } else if (cur == 0 || cur == 1) {
                carry = 0;
            } else if (cur == 3) {
                cur = 1;
                carry = -1;
            } else {
                System.out.println("impossible!");
            }
            
            res.add(cur);
        }
        
        for (int i = arr1.length - arr2.length - 1; i >= 0; --i) {
            int cur = arr1[i] + carry;
            if (cur == -1) {
                cur = 1;
                carry = 1;
            } else if (cur == 2) {
                cur = 0;
                carry = -1;
            } else if (cur == 0 || cur == 1) {
                carry = 0;
            } else if (cur == 3) {
                cur = 1;
                carry = -1;
            } else {
                System.out.println("impossible!");
            }
            res.add(cur);
        }
        
        if (carry == 1) {
            res.add(carry);
        } else if (carry == -1) {
            res.add(1);
            res.add(1);
        }
        
        if (res.stream().filter(i -> i == 1).count() == 0)
            return new int[]{0};
        
        Collections.reverse(res);
        int j;
        for (j = 0; j < res.size(); ++j) {
            if (res.get(j) == 1)
                break;
        }
        
        res = res.subList(j, res.size());

        return res.stream().mapToInt(i -> i).toArray();
        
    }
}
