import java.math.BigInteger;

class Solution {
BigInteger zero = BigInteger.ZERO;
public int[] addNegabinary(int[] arr1, int[] arr2) {
    int[] ret;
    BigInteger x = zero;
    BigInteger  y  = x;
    for(int i = -1; i < arr1.length; i++) {
        if(i >= 0)
        x = x.multiply(BigInteger.valueOf(-2)).add(BigInteger.valueOf(arr1[i]));
    }
    for(int i = -1; i < arr2.length; i++) {
        if(i >= 0)
        y = y.multiply(BigInteger.valueOf(-2)).add(BigInteger.valueOf(arr2[i]));
    }
    BigInteger s = x.add(y);
    String str_s = bn(s);
    char[] c = str_s.toCharArray();
    ret = new int[str_s.length()];
    
for (int i = 0; i < c.length; i++) {
      ret[i] = c[i] - '0';
    }
    return ret;
}

public String bn(BigInteger N) {
   StringBuilder res = new StringBuilder();
   while (!N.equals(zero)) {
     res.append(N.and(BigInteger.ONE));
     N = N.shiftRight(1).negate();
   }
   return res.length() > 0 ? res.reverse().toString() : "0";
   }
}
