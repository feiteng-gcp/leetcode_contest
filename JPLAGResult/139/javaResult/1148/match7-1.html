class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        if (arr1.length == 1 && arr1[0] == 0)
            return arr2;
        if (arr2.length == 1 && arr2[0] == 0)
            return arr1;
        List<Integer> resultList = new ArrayList<>(1000);
        int cin = 0;
        for (int i = 0; i < Math.min(arr1.length, arr2.length); i++) {
            int arr1Ptr = arr1.length - i - 1;
            int arr2Ptr = arr2.length - i - 1;
            int sum = arr1[arr1Ptr] + arr2[arr2Ptr] + cin;
            if  (sum >= 2) {
                cin = -1;
                resultList.add(sum % 2);
            } else if (sum == -1) {
                cin = 1;
                resultList.add(1);
            } else {
                cin = 0;
                resultList.add(sum);
            }
        }
        int[] maxArray = arr1.length > arr2.length ? arr1 : arr2;
        int ptr = maxArray.length - Math.min(arr1.length, arr2.length) - 1;
        while (ptr >= 0) {
            int sum = cin + maxArray[ptr];
            if (sum == -1) {
                resultList.add(1);
                cin = 1;
            } else if (sum >= 2) {
                cin = -1;
                resultList.add(sum % 2);
            } else {
                cin = 0;
                resultList.add(sum);
            
}
            ptr--;
        }
        if (cin == -1) {
            resultList.add(1);
            resultList.add(1);
        } else if (cin == 1) {
            resultList.add(1);
        }
        while (resultList.size() > 1 && resultList.get(resultList.size() - 1) == 0)
            resultList.remove(resultList.size() - 1);
        int[] result = new int[resultList.size()];
        for (int i = 0; i < resultList.size(); i++) 
{
            result[result.length - i - 1] = resultList.get(i);
        }
        return result;
    }
}
