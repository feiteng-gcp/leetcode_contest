class Solution {
    
    void rev(int a[]){
        int l = 0;
        int h = a.length-1;
        
        while(l<h){
            int t = a[l];
            a[l] =a[h];
            a[h] = t;
            --h;
            ++l;
            
            
        }
        
    }
    
   
    
    public int[] addNegabinary(int[] a1, int[] a2) {
        rev(a1);rev(a2);
        int res[] = new int[2000];
        
        
        for(int i=0;i<1995;++i){
            if(i<a1.length){
                res[i] += a1[i];
                
            }
            if(i<a2.length){
                res[i] += a2[i];
                
            }
            int jw = res[i]/2;
            res[i+1]+= jw;
            res[i+2] += jw;
            
            res[i] = res[i]%2;
            //System.out.println(res[i]);
            
        }
        

       int p = 1010;
        while(p>=0&&res[p]==0){
            --p;
        }
        if(p==-1){
            return new int[]{0};
        }
        

        int a[] = new int[p+1];
        for(int i=0;i<a.length;++i){
            a[i] = res[p--];
        }
        return a;
        
    }
}
