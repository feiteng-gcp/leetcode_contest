class Solution {
    
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        int carry = 0;
        int n = Math.max(arr1.length, arr2.length);
        List<Integer> l1 = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            int len = arr1.length;
            if (i < len) l1.add(arr1[len - i - 1]);
            else l1.add(0);
        }

        List<Integer> l2 = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            int len = arr2.length;
            if (i < len) l2.add(arr2[len - i - 1]);
            else l2.add(0);
        }

        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            int sum = l1.get(i) + l2.get(i) - carry;
            carry = 0;
            if (sum == -1) {
                sum = 1;
                carry = -1;
            }
            if (sum >= 2) {
                sum -= 2;
                carry = 1;
            
}
            res.add(sum);
        }
        if (carry > 0) {
            res.add(1);
            res.add(1);
        } else if (carry < 0) {
            res.add(1);
        }
        while (res.size() > 1) {
            if (res.get(res.size() - 1) == 0) res.remove(res.size() - 1);
            else break;
        
}
        int[] ans = new int[res.size()];
        for (int i = 0; i < res.size(); i++) {
            ans[i] = res.get(res.size() - i - 1);
        }
        return ans;
    }
}
