class Solution {

	private int[] reverse(int[] array) {
		int[] ans = new int[array.length];
		for (int i = 0;i < array.length;i ++) {
			ans[array.length - i - 1] = array[i];			
		}
		return ans;
	}
	
    public int[] addNegabinary(int[] arr1, int[] arr2) {
    	List<Integer> list = new ArrayList<>();
    	arr1 = reverse(arr1);
    	arr2 = reverse(arr2);
    	int i = 0 , n1 = arr1.length , n2 = arr2.length , c = 0;
    	while (i < n1 && i < n2) {
    		int sum = arr1[i] + arr2[i] + c;
    		if (sum == - 1) {
    			list.add(1);    			
    			c = 1;    			
    		} else if (sum ==
 - 2) {
    			list.add(0);
    			c = 1;    			
    		} else if (sum == 0) {
				list.add(0);    				
				c = 0;
			} else if (sum == 1) {
				list.add(1);
				c = 0;    				
			} else if (sum == 2) {
				list.add(0);    				
				c = - 1;    				
			} else if (sum == 3) {
				list.add(1);
				c = - 1;
			}
    		i ++;    		
    	}
    	while (i < n1) {
    		int sum = arr1[i] + c;    		
    		if (sum == - 1) {
    			list.add(1);    			
    			c = 1;    			
    		} else if (sum == - 2) {
    			list.add(0);
    			c = 1;  			
    		} else if (sum == 0) {
				list.add(0);    				
				c = 0;
			} else if (sum == 1) {
				list.add(1);
				c = 0;    				
			} else if (sum == 2) {
				list.add(0);    				
				c = - 1;    				
			} else if (sum == 3) {
				list.add(1);
				c = - 1;
			}
    		i ++; 
    	}
    	while (i < n2) {
    		int sum = arr2[i] + c;
    		if (sum == - 1) {
    			list.add(1);    			
    			c = 1;    			
    		} else if (sum == - 2) {
    			list.add(0);
    			c = 1;  			
    		} else if (sum == 0) {
				list.add(0);    				
				c = 0;
			} else if (sum == 1) {
				list.add(1);
				c = 0;    				
			} else if (sum == 2) {
				list.add(0);    				
				c = - 1;    				
			} else if (sum == 3) {
				list.add(1);
				c = - 1;
			}
    		i ++; 
    	}
    	while (c != 0) {
    		int sum = c;
    		if (sum == - 1) {
    			list.add(1);    			
    			c = 1;    			
    		} else if (sum == - 2) {
    			list.add(0);
    			c = 1;  			
    		} else if (sum == 0) {
				list.add(0);    				
				c = 0;
			} else if (sum == 1) {
				list.add(1);
				c = 0;    				
			} else if (sum == 2) {
				list.add(0);    				
				c = - 1;    				
			} else if (sum == 3) {
				list.add(1);
				c = - 1;
			}
    		i ++; 
    	}
    	Collections.reverse(list);
    	i = 0;
    	while (i < list.size() && list.get(i) == 0) {
    		i ++;    		
    	}
    	if (i == list.size()) {
    		return new int[] {0};
    	} else {
    		int[] ans = new int[list.size() - i];
    		int next = 0;
    		while (i < list.size()) {    			
    			ans[next] = list.get(i);
    			next ++;
    			i ++;
    		}
    		return ans;
    	}
    }
	
	
}
