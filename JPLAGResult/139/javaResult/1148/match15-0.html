class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
        if(arr1.length < arr2.length) return addNegabinary(arr2, arr1);
        List<Integer> list = new LinkedList<>();
        int len1 = arr1.length - 1, len2 = arr2.length - 1, i = 0, carry = 0;
        // carry can only be -1 and 0. so val can only be between -1 to 2.
        while(len2 - i >= 0){
            int val = arr1[len1 - i] + arr2[len2 - i] + carry;
            if(val == 2){
                carry = -1;
                list.add(0, 0);
            } else if(val == -1){
                carry = 1;
                list.add(0, 1);
            } else if(val == 3){
                carry = -1;
                list.add(0, 1);
            } else {
                carry = 0;
                list.add(0, val);
            }
            i++;
        }
        
        while(len1 - i >= 0){
            int val = arr1[len1 - i] + carry;
            if(val == 2){
                carry = -1;
                list.add(0, 0);
            } else if(val == -1){
                carry = 1;
                list.add(0, 1);
            } else if(val == 3){
                carry = -1;
                list.add(0, 1);
            } else {
                carry = 0;
                list.add(0, val);
            }
            i++;
        }
        
        if(carry == 1) list.add(0, 1);
        else if(carry == -1){
            list.add(0, 1);
            list.add(0, 1);
        } 
        
        int start = 0;
        while(start < list.size() && list.get(start) == 0) start++;
        if(start == list.size()) return new int[]{0};
        
        int[] fAns = new int[list.size() - start];
        for(int k = 0; k < fAns.length; k++){
            fAns[k] = list.get(k + start);
        }
        return fAns;
        
    }
}
