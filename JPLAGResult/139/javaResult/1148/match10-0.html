class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
		int[] arr1r = new int[arr1.length];
		int[] arr2r = new int[arr2.length];
		for (int i = 0; i < arr1.length; i++) {
			arr1r[i] = arr1[arr1.length - i - 1];
		}
		for (int i = 0; i < arr2.length; i++) {
			arr2r[i] = arr2[arr2.length - i - 1];
		}
		List<Integer> result = new ArrayList<>();
		int carry = 0;
		int i = 0;
		for (; i < arr1r.length && i < arr2r.length; i++) {
			int sum = arr1r[i] + arr2r[i] + carry;
			if (sum == 0) {
				result.add(0);
				carry = 0;
			} else if (sum == 1) {
				result.add(1);
				carry = 0;
			} else if (sum == 2) {
				result.add(0);
				carry = -1;
			} else if (sum == 3){
				result.add(1);
				carry = -1;
			} else { // sum == -1
				result.add(1);
				carry = 1;
			}
		}
		int[] arrLonger = arr1r.length >= arr2r.length ? arr1r : arr2r;
		for (; i < arrLonger.length; i++) {
			int sum = arrLonger[i] + carry;
			if (sum == 0) {
				result.add(0);
				carry = 0;
			} else if (sum == 1) {
				result.add(1);
				carry = 0;
			} else if (sum == 2) {
				result.add(0);
				carry = -1;
			} else if (sum == 3){
				result.add(1);
				carry = -1;
			} else { // sum == -1
				result.add(1);
				carry = 1;
			}
		}
		if (carry == 1) {
			result.add(1);
		} else if (carry == -1) {
			result.add(1);
			result.add(1);
		}
		while (result.size() > 0 && result.get(result.size() - 1) == 0) {
			result.remove(result.size() - 1);
		}
		if (result.size() == 0) {
			return new int[]{0};
		}
		int[] r = new int[result.size()];
		for (i = 0; i < result.size(); i++) {
			r[i] = result.get(result.size() - i - 1);
		}
		return r;
    }

}
