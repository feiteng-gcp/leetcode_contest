class Solution {
    public int[] addNegabinary(int[] arr1, int[] arr2) {
    List<Integer> list = new ArrayList<>();
    int base = 0;
    int arr1Length = arr1.length;
    int arr2Length = arr2.length;
    for (int i = 0; i < arr1Length || i < arr2Length; i++) {
      int a1 = i < arr1Length ? arr1[arr1Length - i - 1] : 0;
      int a2 = i < arr2Length ? arr2[arr2Length - i - 1] : 0;
      int sum = a1 + a2 + base;
      if (sum == 0 || sum == 1) {
        base = 0;
      } else if (sum == -1) {
        base = 1;
        sum = 1;
      } else if (sum == 2) {
        base = -1;
        sum = 0;
      } else if (sum == 3) {
        base = -1;
        sum = 1;
      }
      list.add(0, sum);
    }
    
if (base == -1) {
      list.add(0, 1);
      list.add(0, 1);
    } else if (base == 1) {
      list.add(0, 1);
    }
    while(list.get(0) == 0 && list.size() > 1){
      list.remove(0);
    }
    int[] ans = new int[list.size()];
    for (int i = 0; i < list.size(); i++) 
{
      ans[i] = list.get(i);
    }
    return ans;
  }
}
