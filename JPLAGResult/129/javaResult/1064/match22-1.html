class Solution {
    public int smallestRepunitDivByK(int K) {
        if(K==1)return 1;
        if(K%2==0)return -1;
        int size=1;
        HashSet<Integer>hh=new HashSet<>();
        int tens=1,mod=1;
        hh.add(1);
        while(true){
            tens=((10%K)*tens)%K;
            mod=(tens+mod)%K;
            if(mod==0)return size+1;
            if(hh.contains(mod))return -1;
            hh.add(mod);
            size+=1;
        }
    }
}
