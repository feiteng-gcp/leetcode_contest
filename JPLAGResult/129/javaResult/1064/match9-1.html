class Solution {
    public int smallestRepunitDivByK(int K) {
        long n = 0;
        
        for (int i = 1;i <= K;i++) {
            n = n * 10 + 1;
            
            
if (n % K == 0) return i;
            
            n = n % K;
        }
        
        return -1;
    }
}
