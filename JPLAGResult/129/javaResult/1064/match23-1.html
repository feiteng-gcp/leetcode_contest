class Solution {
    public int smallestRepunitDivByK(int K) {
        if(K%2 == 0) return -1;
        if(K%5 == 0) return -1;
        long start = 1;
        int len = 1;
        long end = (long)K * (long)K;
        while(len <= end){
            if(start % K == 0) return len;
            start = (start*10)%K + 1;
            len++;
        }
        return -1;
    }
}
