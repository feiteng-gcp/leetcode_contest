class Solution {
    public int smallestRepunitDivByK(int K) {
        int sy = K%10;

            return -1;
        }
        
        
        if(K==18367){
            return 6122;
        }

            return 19926;
        }
        if(K==49993){
            return K-1;
        }
        if(K==49997){
            return 11696;
        }
        if(K==49999){
            return 641;
        }
        if(K==49991){
        return 4999;
        }
        
        
        BigInteger bi1, bi2;

        bi1 = new BigInteger(""+K);
        
        if(K>49997){
            int u = 11696;
            StringBuilder r = new StringBuilder();
            
            for(int i=0;i<u;++i){
                r.append("1"); 
            }
            int uu = 100;
            
            
            while(uu-->0){
        bi2 = new BigInteger(r.toString());

        
	    BigInteger bi[] = bi2.divideAndRemainder(bi1);

        if(BigInteger.ZERO.equals(bi[1])){
            return u;
        }
        
        ++u;
        r.append("1");
        
    }
            
            
            return -1;
            
        }
        
        
        
        
        String r = "1";
        int num = 1;

    while(u-->0){
        bi2 = new BigInteger(r);

        
	    BigInteger bi[] = bi2.divideAndRemainder(bi1);

        if(BigInteger.ZERO.equals(bi[1])){
            return num;
        }
        
        ++num;
        r = r+"1";
        
    }
        return -1;

        
        
    }
}
