class Solution {
    public int smallestRepunitDivByK(int K) {
        if (K == 1) {
            return 1;
        }
        Set<String> set = new HashSet<>();
        set.add("1,1");
        int mod = 1;
        int totalMod = 1;
        for (int i = 2; true; i++) {
            mod = (10 * mod) % K;
            totalMod = (totalMod + mod) % K;
            if (totalMod == 0) {
                return i;
            }
            String s = String.format("%s,%s", mod, totalMod);
            if (set.contains(s)) {
                return -1;
            }
            set.add(s);
        }
    }
}
