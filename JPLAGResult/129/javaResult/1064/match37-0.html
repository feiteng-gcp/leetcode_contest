class Solution {
  
  public int smallestRepunitDivByK(int K) {
    int k = 1;
    
    if(K% 2 == 0) {
      return -1;
    }
    
    if(K % 5 == 0) {
      return -1;
    }
    int res = 1;
   
    boolean[] visited = new boolean[K];
    
    while(k % K != 0) {
      int next = (k % K) * 10 + 1;
      if(visited[next % K]) {
        return -1;
      }
      
      visited[next % K] = true;
      k = next;
      res++;
    }
    
    return res;
  }
}
