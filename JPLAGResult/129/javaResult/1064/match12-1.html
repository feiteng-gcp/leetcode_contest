class Solution {
    public int smallestRepunitDivByK(int K) {
        int len = 100000;
        long temp = 0;
        
for(int i=1;i<=len;i++){
            temp = (temp*10 + 1)%K;
            if(temp%K == 0){
                return i;
            }
            
        }
        return -1;
    }
}
