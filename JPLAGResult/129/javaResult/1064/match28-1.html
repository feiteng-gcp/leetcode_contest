class Solution {
    public int smallestRepunitDivByK(int K) {
        HashSet<Long> yu=new HashSet<>();
        long p=1;
        int count=1;
        while(p<K){
            p=p*10+1;
            count++;
        }
        while(true){
            long y=p%K;
            if(y==0){
                return count;
            }else{
                if(yu.contains(y))
                    return -1;
                yu.add(y);
                p=y*10+1;
                count++;
            }
        }
    }

}
