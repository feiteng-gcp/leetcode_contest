class Solution {
    public int smallestRepunitDivByK(int K) {
        boolean[] f = new boolean[K];
        int res, sum = 0, p = 1;
        for (res = 1; ; res++) {
            sum = (sum + p) % K;
            if (sum == 0) break
;
            if (f[sum]) return -1;
            f[sum] = true;
            p = p * 10 % K;
        }
        return res;
    }
}
