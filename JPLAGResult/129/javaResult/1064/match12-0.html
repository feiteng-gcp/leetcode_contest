class Solution {
    public int smallestRepunitDivByK(int K) {
        long mods[]=new long[1000000];
        mods[1]=1%K;
        if(K==1){
            return 1;
        }
        for(int i=2;i<1000000;i++){
            mods[i]=((mods[i-1]*10)%K+1%K)%K;
            if(mods[i]==0){
                return i;
            }
        }
        
        return -1;
    }
}
