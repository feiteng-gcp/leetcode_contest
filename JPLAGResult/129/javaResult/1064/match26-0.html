class Solution {
    public int smallestRepunitDivByK(int K) {
        //if (K % 2 == 0) return -1;
        Set<Integer> found = new HashSet<>();
        int remain = 1;
        int num = 1;
        remain = remain % K;
        found.add(remain);
        while (remain != 0) {
            remain = (remain * 10 +1) % K;
            num ++;
            if (found.contains(remain)) return -1;
            found.add(remain);
        }
        return num;
    }
}
