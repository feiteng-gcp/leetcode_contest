class Solution {
    public int smallestRepunitDivByK(int K) {
        HashSet<Integer> visited = new HashSet<>();
        int n = 1,len = 1;
        while (n < K){
            n = n * 10 + 1;
            len++;
        }

        while (true){
            if(!visited.add(n)) return -1;
            int reminder = n % K;
            if(reminder == 0) return len;
            n = reminder * 10 + 1;
            len++;
        }
    }
}
