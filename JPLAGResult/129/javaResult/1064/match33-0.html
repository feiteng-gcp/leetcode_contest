class Solution {
    public int smallestRepunitDivByK(int K) {
    if (K % 10 != 1 && K % 10 != 3 && K % 10 != 7 && K % 10 != 9) {
      return -1;
    }
    int ans = 0;
    int mulResult = 0;
    int lastNum = K % 10;
    while (true) {
      int temp = helper(K, lastNum, (11 - mulResult % 10) % 10);
      if (temp == -1) {
        return -1;
      } else {
        mulResult += temp;
        while (mulResult % 10 == 1) {
          mulResult /= 10;
          ans++;
          if (mulResult == 0) {
            return ans;
          }
        }
      }
    }
  }

  private int helper(int origin, int last, int taget) {
    for (int i = 1; i <= 9; i++) {
      if (last * i % 10 == taget) {
        return origin * i;
      }
    }
    return -1;
  }
}
