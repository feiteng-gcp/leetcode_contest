class Solution {
    public int smallestRepunitDivByK(int K) {
        HashSet<Integer> repeated = new HashSet<>();
        int count = 0;
        int N = 0;
        while(true)
        {
            while(N < K) {
                N = N * 10 + 1;
                count ++;
            }
            N = N % K;
            if(N == 0) break;
            if(repeated.contains(N)) return -1;
            repeated.add(N);
        }
        return count;
    }
}
