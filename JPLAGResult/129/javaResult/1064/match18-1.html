class Solution {
    public int smallestRepunitDivByK(int K) {
        if (K % 2 == 0 || K % 5 == 0) {
            return -1;
        }
        long res = 0;
        long add = 1;
        int i = 1;
        while (true) {
            res = (res + add) % K;
            if (res % K == 0) {
                return i;
            }
            add = (add * 10 % K);
            i++;
        }
    }
}
