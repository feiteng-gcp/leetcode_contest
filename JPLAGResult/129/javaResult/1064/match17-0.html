class Solution {
    public int smallestRepunitDivByK(int K) {
        if (K % 2 == 0 || K % 5 == 0) {
            return -1;
        }
        Set<Integer> set = new HashSet<>();
        int res = 1;
        int t = 1;
        while (!set.contains(t)) {
            set.add(t);
            if (t % K == 0) {
                return res;
            }
            res++;
            t %= K;
            t = t * 10 + 1;
        }
        return -1;
    }
}
