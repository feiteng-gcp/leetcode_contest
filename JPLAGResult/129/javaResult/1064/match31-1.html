class Solution {
    public int smallestRepunitDivByK(int K) {
        if (K % 2 == 0 || K % 5 == 0) {
            return -1;
        }
        
        // There always exists n, m, so that pow(10, n + m) - pow(10, m) is divisible by 9*K,
        // so pow(10, m) * (pow(10, n) - 1) is divisible by 9K.
        // If gcd(K, 10) == 1, then (pow(10, n) - 1) is divisible by 9K.
        // So (pow(10, n) - 1) / 9 is divisible by K. So the return value is not -1.
        
        int digit = 1, remainder = 1 % K;
        while(remainder != 0) {
            digit += 1;
            remainder = (remainder * 10 + 1) % K;
        }
        return digit;
    }
}
