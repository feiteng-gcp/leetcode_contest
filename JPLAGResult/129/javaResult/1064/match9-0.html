class Solution {
    public int smallestRepunitDivByK(int K) {
        if(K%2 == 0)
            return -1;
        int runSum = 1;
        int last = 1;
        for(int i = 1; i <= K; i++) {
            if(runSum % K == 0) 
                return i;
            last = (10 * last) % K;
            runSum += last;
        }
        return -1;
    }
}
