class Solution {
    public int smallestRepunitDivByK(int K) {
        // if k is divisible by 2 or 5, return -1
        if (K % 2 == 0 || K % 5 == 0) {
            return -1;
        }
        
        int result = 1;
        int rem = 1 % K;
        
        while (rem != 0 && result < Integer.MAX_VALUE) {
            result++;
            rem = (rem * 10 + 1) % K;
        }
        
        // If number length exceeds integer limit
        if (result == Integer.MAX_VALUE) {
            return -1;
        }

        return result;
    }
}
