class Solution {
    public int smallestRepunitDivByK(int k) {
        
        long num = 1;
        
        for(int i = 0; i < 1000000; i++)
        {
            if(num % k == 0)
            {
                return i + 1;
            }
            num = num * 10 + 1;
            num %= k;
        }
        
        return -1;
    }
}
