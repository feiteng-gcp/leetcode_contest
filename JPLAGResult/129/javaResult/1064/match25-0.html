class Solution {
    public int smallestRepunitDivByK(int k) {
        if (k == 1) return 1;
        if (k % 2 == 0 || k % 5 == 0) return -1;
        
        boolean[] visited = new boolean[k];
        for (int i = 1, r = 1; i <= k + 2; i++) {
            int rem = r % k;
            if (rem == 0) return i;
            
            if (visited[rem]) return -1;
            visited[rem] = true;
            r = (rem * 10 + 1) % k;
        }
        
        return -1;
    }
}
