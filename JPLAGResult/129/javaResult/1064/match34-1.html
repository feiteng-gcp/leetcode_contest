class Solution {
    public int smallestRepunitDivByK(int K) {
        long ans = 1;
        int count = 1;
        while(count <= K + 1) {
            if (ans % K == 0) 
{
                return count;
            }
            count++;
            ans = ans * 10 + 1;
            ans %= K;
        }
        return -1;
    }
}
