class Solution {
    public int maxScoreSightseeingPair(int[] A) {
        int n = A.length;
        int right = A[n-1]-(n-1);
        int ans = 0;
        for(int i=n-2;i>=0;i--){
            int cur = A[i]+i+right;
            ans = Math.max(ans,cur);
            right = Math.max(right,A[i]-i);
        }
        return ans;
    }
}
