class Solution {
    public int maxScoreSightseeingPair(int[] A) {
        int n = A.length;
        int res = Integer.MIN_VALUE;
        int dp = A[0];
        for(int i = 1; i < n; i ++) {
            dp --;
            int s = A[i] + dp;
            res = Math.max(s, res);
            dp = Math.max(dp, A[i]);
        }
        return res;
    }
}
