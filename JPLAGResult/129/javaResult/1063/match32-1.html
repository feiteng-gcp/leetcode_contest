class Solution {
    public int maxScoreSightseeingPair(int[] A) {
        int f = A[0];
        int N = A.length;
        int res = -1;
        for (int i = 1; i < N; i++) {
            
res = Integer.max(f + A[i] - i, res);
            f = Integer.max(f, A[i] + i);
        }
        return res;
    }
}
