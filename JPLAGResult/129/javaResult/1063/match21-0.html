class Solution {
    public int maxScoreSightseeingPair(int[] A) {
        int res = Integer.MIN_VALUE;
        int n = A.length;
        int max = A[n - 1] - n + 1;
        for (int i = n - 2; i >= 0; i--) {
            res = Math.max(A[i] + i + max, res);
            max = Math.max(A[i] - i, max);
        }
        return res;
    }
}
