class Solution {
    public int maxScoreSightseeingPair(int[] A) {
    	int pre = A[A.length-1]-(A.length-1);
    	int ans = Integer.MIN_VALUE;
        for (int i=A.length-2; i>=0; i--) {
        	ans =Math.max(ans, A[i]+i+pre);
        	pre = Math.max(pre, A[i]-i);
        }
		return ans;
    }
}
