class Solution {
 public int maxScoreSightseeingPair(int[] A) {
     
    	int len = A.length;
    	//A[i] + A[j] + i - j
    	int[] tempi = new int[len];
    	int[] tempj = new int[len];
    	for (int i = 0; i < len; i++) {
    		tempi[i] = A[i] + i;
    		tempj[i] = A[i] -i;
		}
    	
    	int maxNumber= Integer.MIN_VALUE;
    	
    	int maxNumber_tempi = tempi[0];
    	for (int i = 1; i < len; i++) {
    		int curMax = tempj[i] + maxNumber_tempi;
    		if(curMax > maxNumber){
    			maxNumber = curMax;
    		}
    		
    		if(tempi[i] > maxNumber_tempi) maxNumber_tempi = tempi[i];
		}
    	return maxNumber;
    }
}
