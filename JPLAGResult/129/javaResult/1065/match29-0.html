class Solution {
    public boolean queryString(String S, int N) {
       for (int i = N; i >= 1; --i) {
           String binary = getBinary(i);
           if (S.indexOf(binary) == -1) {
               return false;
           }
       }
        return true;
    }
    
    private String getBinary(int num) {
        StringBuilder sb = new StringBuilder();
        while (num != 0) {
            sb.append(num & 1);
            num >>= 1;
        }
        return sb.reverse().toString();
    }
    
}
