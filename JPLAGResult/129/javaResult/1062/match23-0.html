class Solution {
    public boolean canThreePartsEqualSum(int[] A) {
        int sum = 0;
        for (int val : A) sum += val;
        if (sum % 3 != 0) return false;
        
        int cur = 0, cnt = 0;
        for (int val : A) {
            cur += val;
            if (cur == (sum/3)) {
                cur = 0;
                ++cnt;
            }
        }
        return cnt == 3 && cur == 0;
    }
}
