class Solution {
    public boolean canThreePartsEqualSum(int[] a) {
        int n = a.length;
        
        long sum = 0;
        
        for(int val : a)
        {
            sum+= val;
        }
        
        int f = 0;
        
        long csum = 0;
        for(int i = 0; i < n; i++)
        {
            csum += a[i];
            
            if(csum == sum / 3)
            {
                f++;
                csum = 0;
            }
        }
        
        if(f == 3)
        {
            return true;
        }
        
        return false;
    }
}
