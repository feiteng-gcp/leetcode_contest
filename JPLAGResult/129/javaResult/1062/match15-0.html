class Solution {
    public boolean canThreePartsEqualSum(int[] A) {
        int total = 0;
        for (int i = 0; i < A.length;i++){
            total += A[i];
        }
        if (total % 3 != 0) return false;
        int need = total / 3;
        
        int count = 0;
        int sum = 0;
        for (int i = 0; i < A.length;i++){
            sum += A[i];
            if (sum == need){
                count++;
                sum = 0;
            }
        }
        return count == 3;
    }
}
