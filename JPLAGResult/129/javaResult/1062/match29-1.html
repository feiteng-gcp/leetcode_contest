class Solution {
    public boolean canThreePartsEqualSum(int[] A) {
        int n = A.length;
        long sum = 0;
        for(int i=0;i<n;i++){
            sum += (long)A[i];
        }
        
        //System.out.println("sum: " + sum);
        if(sum%3 != 0) return false;
        long req = sum/3;
        
        
int count = 0;
        long curr = 0;
        //System.out.println("req: " + req);
        for(int i=0;i<n;i++){
            curr += (long) A[i];
            //System.out.println("curr: " + curr);
            if(curr == req){
                count++;
                curr = 0;
            }
        }
        
        //System.out.println("count: " + count);
        if(count == 3 && curr == 0){
            return true;
        }
        return false;
    }
}
