class Solution {
 public boolean canThreePartsEqualSum(int[] A) {
    	
    	/**
    	 * [0,2,1,-6,6,-7,9,1,2,0,1]
    	 * 
    	 */
        
    	if(A == null || A.length <3) return false;
    	
    	int sum = 0;
    	for (int i = 0; i < A.length; i++) {
			sum += A[i];
		}
    	
    	if(sum%3 != 0) return false;
    	int blockSum = sum/3;
    	
    	int blockNumber = 0;//块的个数
    	int currentSum = 0;
    	for (int i = 0; i < A.length; i++) {
    		currentSum += A[i];
    		if(currentSum==blockSum){
    			++blockNumber;
    			currentSum = 0;
    		}
		}
    	
    	if(blockNumber ==3) return true;
    	return false;
    	
    }
}
