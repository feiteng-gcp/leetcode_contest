class Solution {
    public boolean canThreePartsEqualSum(int[] A) {
        int sum = 0;
        for (int num: A) {
            sum += num;
        }
        if (sum % 3 != 0) return false;
        System.out.println(sum);
        int target = sum / 3;
        int two = 2 * target;
        boolean sum1 = false, sum2 = false;
        int cur = 0;
        for (int i = 0; i < A.length; i++) {
            cur += A[i];
            if (cur == target && !sum1) {
                sum1 = true;
            } else if (cur == two && sum1) {
                sum2 = true;
            }
        }
        return sum1 && sum2;
    }
}
