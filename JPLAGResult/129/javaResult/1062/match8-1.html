class Solution {
    public boolean canThreePartsEqualSum(int[] A) {
        int sum=0;
        for(int n:A){
            sum+=n;
        }
        if(sum%3!=0) return false;
        sum/=3;
        
int s=0,cnt=0;
        for(int i=0;i<A.length;i++){
            s+=A[i];
            if(s==sum){
                s=0;
                cnt++;
            }
        }
        return cnt==3;
    }
}
