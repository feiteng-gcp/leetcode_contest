class Solution:
    def smallestRepunitDivByK(self, K: int) -> int:
        N = 1
        res = N % K
        modList = set()
        counts = 1
        while res not in modList:
            if res == 0:
                return counts
            modList.add(res)
            res = (res*10+1) % K
            counts += 1
        return -1
