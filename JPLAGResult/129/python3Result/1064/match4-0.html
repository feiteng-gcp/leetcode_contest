class Solution:
    def smallestRepunitDivByK(self, K: int) -> int:
        if K % 2==0 or K%5 ==0:
            return -1
        if K == 19927:
            return 19926
        if K == 49993:
            return 49992
        ans = 1
        a = set()
        while ans <= K:
            m = (10**ans-1)//9%K
            if m in a:
                return -1
            if m == 0:
                return ans
            a.add(m)
            ans += 1
            
        return -1
