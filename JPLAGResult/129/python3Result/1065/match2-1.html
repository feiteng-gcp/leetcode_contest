class Solution:
    def queryString(self, S: str, N: int) -> bool:
        mx_length = len(bin(N)) - 2
        # assume unique substr, all substr is not enough
        if len(S) * mx_length < N:
            return False
        subs 
= set()
        for length in range(1, mx_length + 1):
            for start in range(len(S) + 1 - length):
                subs.add(int(S[start:start + length], 2))
        return len(subs) >= N and subs >= set(range(1, N + 1))
