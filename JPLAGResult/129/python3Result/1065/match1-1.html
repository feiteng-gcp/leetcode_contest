class Solution:
    def queryString(self, S: str, N: int) -> bool:
        setS = set()
        for size in range(len(S)):
            for i in range(len(S)-size+1):
                setS.add(S[i:i+size])
                
        
for i in range(1, N+1):
            b = bin(i)[2:]
            if b not in setS:
                return False
            
        return True
