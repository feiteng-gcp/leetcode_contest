class Solution:
    def maxScoreSightseeingPair(self, A: List[int]) -> int:
        N = len(A)
        ans = 0
        
        max_index = 0
        for i in range(1,N):
            ans = max(ans, A[i]+A[max_index]-i+max_index)
            
            if A[i]+i > A[max_index]+max_index:
                max_index = i
                
        return ans
