class Solution:
    def maxScoreSightseeingPair(self, A: List[int]) -> int:
        mx = A[0]+A[1]-1
        hs = [-1]*(1001)
        arrLen = len(A)
        currMX = A[-1]-(arrLen-1)
        
for i in range(arrLen-2,-1,-1):
            mx = max(mx,A[i]+i+currMX)
            currMX = max(currMX,A[i]-i)
        return mx
