class Solution:
    def maxScoreSightseeingPair(self, A: List[int]) -> int:
        
        best_score = -float('inf')
        
        current_i = 0
        
        for j in range(1, len
(A)):

            best_score = max(best_score, A[current_i]+A[j]+current_i-j)
            if A[j] >= A[current_i] + current_i - j:
                current_i = j
        
        return best_score
