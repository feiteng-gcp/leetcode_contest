class Solution:
    def maxScoreSightseeingPair(self, A: List[int]) -> int:
        """
        :type A: List[int]
        :rtype: int
        """
        n = len(A)
        a, b = A.copy(), A.copy()
        for i in range(n):
            a[i] += i
        for j in range(n):
            b[j] -= j

        max_val = a.copy()
        for i in range(1, n):
            max_val[i] = max(max_val[i], max_val[i - 1])
        ret = 0
        for j in range(1, n):
            ret = max(ret, b[j] + max_val[j - 1])
        return ret
