class Solution:
    def maxScoreSightseeingPair(self, A: List[int]) -> int:
        n = len(A)
        jSum = [A[j] - j for j in range(n)]
        jMaxSum = [0]*n
        jMaxSum[n-1] = jSum[n-1]
        for j in range(n-2,-1,-1):
            jMaxSum[j] = max(jMaxSum[j+1], jSum[j])
        maxPairSum = 0
        for i in range(n-1):
            maxPairSum = max(maxPairSum, A[i]+i+jMaxSum[i+1])
            
        return maxPairSum
