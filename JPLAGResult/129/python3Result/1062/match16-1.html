class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        sum = 0
        for a in A:
            sum += a
        if sum % 3 != 0:
            return False
        cnt 
= 0
        pre = 0
        for a in A:
            pre += a
            if pre == sum / 3:
                cnt += 1
                pre = 0
        if cnt >= 3:
            return True
        else:
            return False
