class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        total = sum(A)
        if sum(A) % 3 != 0: 
return False
        total //= 3
        count = 3
        current = 0
        for num in A:
            current += num
            if current == total:
                current = 0
                count -= 1
        return count <= 0
