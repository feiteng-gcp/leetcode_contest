class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        
        
        
        L = len(A)
        if L < 3:
            return False
        s = sum(A)
        if s % 3 != 0:
            return False
        ss = s // 3
        presum = 0
        ifound, jfound = False, False
        for i,a in enumerate(A):
            presum += a
            if presum == ss:
                if ifound == False:
                    ifound = True
                    presum = 0
                else:
                    if jfound == False:
                        return True
        return False
