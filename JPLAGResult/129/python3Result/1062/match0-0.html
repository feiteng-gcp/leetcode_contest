class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s = sum(A)
        if s % 3 != 0: return False
        target = s // 3
        
        curr_sum = 0
        partitions = 0
        for i in range(len(A)):
            curr_sum += A[i]
            if curr_sum == target:
                partitions += 1
                curr_sum = 0
        return partitions == 3
