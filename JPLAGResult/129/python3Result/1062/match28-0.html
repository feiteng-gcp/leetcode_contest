class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s = sum(A)
        if s % 3: return False
        c = 0
        n = len(A)
        l, r = n, -1
        for i in range(n):
            a = A[i]
            c += a
            if c == s // 3:
                l = i
                break
        c = 0
        for i in range(n-1, -1, -1):
            a = A[i]
            c += a
            if c == s // 3:
                r = i
                break
        return l < r - 1
