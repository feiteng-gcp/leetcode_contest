class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        if sum(A)%3 != 0:
            return False
        target = sum(A)//3
        
        cur = 0
        count = 0
        for a in A:
            cur += a
            
            if cur == target:
                cur = 0
                count += 1
            
        return count==3
