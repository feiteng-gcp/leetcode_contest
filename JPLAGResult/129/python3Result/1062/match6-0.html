class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s=sum(A)
        if s%3 != 0: return False
        s//=3
        cnt=0
        cur=s
        for i,a in enumerate(A):
            cur-=a
            if cur==0:
                cnt+=1
                cur=s
                if cnt==2: return True
        return False
