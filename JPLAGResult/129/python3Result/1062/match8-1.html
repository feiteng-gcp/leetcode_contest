class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s = sum(A)
        if s % 3 != 0:
            return False
        s //= 3
        a = 0
        cnt = 0
        for i in range(len(A)):
            a += A[i]
            if a == s:
                a = 0
                cnt += 1
        return True if cnt == 3 else False
