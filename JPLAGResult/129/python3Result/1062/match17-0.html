class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s = sum(A)
        if(s % 3 != 0):
            return False
        need = s // 3
        cur = 0
        c = 0
        for a in A:
            cur += a
            if cur == need:
                if c == 1:
                    return True
                else:
                    c += 1
                    cur = 0
        return False
