class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        a_sum = sum(A)
        if a_sum % 3 != 0:
            return False
        target = a_sum // 3
        tmp = 0
        counts = 0
        for i, num in enumerate(A):
            tmp += num
            if tmp == target:
                tmp = 0
                counts += 1
        return counts == 3
