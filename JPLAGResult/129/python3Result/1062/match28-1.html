from typing import List



class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s = sum(A)
        if s % 3 != 0:
            return False
        s //= 3

        n = len(A)
        p, sump, q, sumq = -1, 0, -1, 0
        for i in range(n):
            sump += A[i]
            if sump == s:
                p = i
                break

        for i in range(n):
            sumq += A[n - 1 - i]
            if sumq == s:
                q = n - 1 - i
                break

        return -1 < p < q
