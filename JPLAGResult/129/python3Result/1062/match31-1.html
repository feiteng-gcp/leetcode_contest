class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s = sum(A)
        # print(s)
        if s % 3 != 0:
            return False
        t = s // 3
        adds= []
        add = 0
        for a in A:
            add += a
            adds += [add]
        t1 = []
        t2 = []
        for i in range(len(adds)):
            if adds[i] == t:
                t1 += [i]
            elif adds[i] == t * 2:
                t2 += [i]
        p1, p2 = 0, 0
        if len(t1) == 0 or len(t2) == 0:
            return False
        # print(t1, t2)
        while p1 < len(t1):
            while t1[p1] > t2[p2]:
                p2 += 1
                if p2 == len(t2): return False
            p3 = len(adds) - 1
            if p3 == t2[p2]:
                return False
            if adds[-1] - adds[t2[p2]] == t:
                return True
            p1 += 1
        return False
