class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        s = sum(A)
        if s%3 != 0:
            return False
        s = s/3
        count = 3
        cs = 0
        for i in A:
            cs += i
            if cs == s:
                count -=1
                cs = 0
        if count == 0 and cs == 0:
            return True
        else:
            return False
