class Solution:
    def canThreePartsEqualSum(self, A: List[int]) -> bool:
        if not A:
            return False
        total = sum(A)
        if total % 3 != 0:
            
return False
        group_sum = total / 3
        count = 0
        cur_sum = 0
        for num in A:
            cur_sum += num
            if cur_sum == group_sum:
                cur_sum = 0
                count += 1
        if group_sum == 0:
            if cur_sum == 0 and count >= 3:
                return True
            else:
                return False
        else:
            if cur_sum == 0 and count == 3:
                return True
            else:
                return False
