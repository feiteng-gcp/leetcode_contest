class Solution {
public:
    int smallestRepunitDivByK(int K) {
        set<int> S;
        int rem=0;
        int len=0;
        while(1) {
            rem=(rem*10+1)%K;
            len++;
            if (rem==0) return len;
            if (S.count(rem)) break;
            S.insert(rem);
        }
        return -1;
    }
};
