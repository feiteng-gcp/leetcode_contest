class Solution {
public:
    int smallestRepunitDivByK(int K) {
        int x = 1;
        int step = 1;
        while (x < K) { x = x * 10 + 1; ++step; }
        unordered_set<int> exist;
        
        while ((x = x % K) != 0) {
            if (exist.find(x) != exist.end()) {
                return -1;
            }
            exist.insert(x);
            x = x * 10 + 1;
            ++step;
        }
        return step;
    }
};
