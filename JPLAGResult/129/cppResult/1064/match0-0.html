#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int smallestRepunitDivByK(int K) {
        int N = 1;
        int m = 1 % K;
        while (m != 0 && N < 1000000)
        {
            m = (10 * m + 1) % K;
            N += 1;
        }
        if (N == 1000000)
            return -1;
        return N;
    }
};
