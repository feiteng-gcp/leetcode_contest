int found[100005];
class Solution {
public:
    int smallestRepunitDivByK(int K) {
        int m = 1 % K;
        for (int i = 0; i < 100005; i++) {
            found[i] = 0;
        }
        for (int i = 1; ; i++) {
            if (m == 0) {
                return i;
            }
            if (found[m]) {
                return -1;
            }
            found[m] = 1;
            m = (m * 10 + 1) % K;
        }
    }
};
