class Solution {
public:
    int smallestRepunitDivByK(int K) {
        int x = 1%K;

        int n = 1;
        vector<int> was(K,0);
        was[x]=1;
        while(x%K!=0) {
            x = (10*x + 1)%K;
            if (was[x]) return -1;
            was[x]=1;
            ++n;
        }
        return n;
    }
};
