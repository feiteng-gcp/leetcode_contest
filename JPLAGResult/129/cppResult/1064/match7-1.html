class Solution {
public:
    int smallestRepunitDivByK(int K) {
        
        if (K % 2 == 0)
            return -1;
        
        if (K == 1)
            return 1;
        
        

        unordered_set<int> hashset;
        int x = 1;
        int count = 1;
        while (hashset.count(x) == 0)
        {
            hashset.insert(x);
            
            x = 10*x % K + 1;
            count++;
            
            if (x % K == 0)
                return count;
            
            
        }
        return -1;
            
        
    }
};
