class Solution {
public:
    int smallestRepunitDivByK(int K) {
        if (K % 5 == 0 || K % 2 == 0) return -1;
        int r, p, ans;
        r = p = ans = 1;
        while (r % K != 0) {
            ++ans;
            p = p * 10 % K;
            r = (r + p) % K;
        }
        return ans;
    }
};
