class Solution {
public:
    int smallestRepunitDivByK(int K) {
        if (K == 1) return 1;
        int now = 1, tmp = 1;
        for (int i = 2; i <= 2000000; i++)
        {
            tmp = tmp * 10 % K;
            now = (now + tmp) % K;
            if (!now) return i;
        }
        return -1;
    }
};
