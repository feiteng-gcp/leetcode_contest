class Solution {
public:
    int smallestRepunitDivByK(int k) {
        
        if(k % 2 == 0)
            return -1;
        
        set<int> st;
        
        int x = 1, n = 1;
        while(x % k != 0 && st.find(x) == st.end()) {
            st.insert(x);
            x = (x * 10 + 1) % k;
            ++n;
        }
        
        return st.find(x) == st.end() ? n : -1;
    }
};
