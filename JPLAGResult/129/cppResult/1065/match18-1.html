class Solution {
public:
    bool queryString(string S, int N) {
        int len = S.length();
        set<int> s;
        for(int j=1; j<=min(len, 30);++j) {
            for(int i=0; i+j-1<len; ++i) {
                int num = 0;
                for(int k=i; k<i+j; ++k) {
                    num <<= 1;
                    num|=(int)(S[k]-'0');
                }
                //printf("%d\n", num);
                if(num<=N&&num>0) s.insert(num);
            }
        }
        return s.size() == N;
    }
};
