class Solution {
public:
    bool queryString(string s, int N) {
        set<int> A = {0};

        int n = s.size();
        for (int i = 0; i < n; ++i) {
            int cur = s[i] - '0';
            A.insert(cur);
            //cout << i << " " << cur << endl;
            for (int j = i + 1; j < n && j < i + 30; ++j) {
                cur = cur * 2 + (s[j] - '0');
                A.insert(cur);
                //cout << j << " " << cur << endl;
            }
        }
        int last = -1;
        for (auto& it : A) {
            if (it != last + 1) return false;
            ++last;
            if (last == N) break;
        }
        return true;
    }
};
