class Solution {
public:
    bool queryString(string S, int N) {
        int n=S.size();
        int t=n*(n+1)/2;
        if(N>t) return false;
        vector<bool> used(N+1,false);
        for(int i=0;i<n;i++)
        {

            int s=0;
            for(int j=i;j<n;j++){
                s=s*2+(S[j]-'0');
                if(s<=N)
                used[s]=true;
                else break;
            }
        }
        
        for(int i=1;i<=N;i++) if(!used[i]) return false;
        return true;
    }
};
