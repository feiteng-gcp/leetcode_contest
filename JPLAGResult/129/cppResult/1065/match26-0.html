bool vis[300050];

class Solution {
public:
    
    bool queryString(string S, int N) {
        if (N>300000) return false;
        
        for (int i=1; i<=N; i++) vis[i]=false;
        int n=S.length();
        
        for (int i=0; i<n; i++){
            long long rate=0;
            for (int j=0; j<30 && i+j<n; j++){
                rate=rate*2+(S[j+i]-'0');
                if (rate<=N) vis[rate]=true; else break;
            }
        }
        
        for (int i=1; i<=N; i++)
            if (vis[i]==false) return false;
        
        return true;
    }
};
