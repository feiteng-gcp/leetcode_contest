class Solution {
public:
    bool queryString(string S, int N) {
        unordered_set<int> uset;

        for (int i = 0;  i < S.size(); ++i) {
            for (int j = i+1; j-i < 32 && j <= S.size(); ++j) {
                int x = stoi(S.substr(i, j-i), nullptr, 2);
                if (x > 0 && x <= N)
                    uset.insert(x);
            }
        }
        return uset.size() == N;
    }
};
