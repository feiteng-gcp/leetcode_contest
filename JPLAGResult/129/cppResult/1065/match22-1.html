class Solution {
public:
    bool queryString(string S, int N) {
        for (int k=1; k<=N; k++){
            string binary = bitset<32>(k).to_string();
            int idx = 0;
            while (binary[idx]=='0'){
                idx++;
            }
           // string bin2(it, binary.end());
            string bin2(binary.begin()+idx, binary.end());
            size_t found = S.find(bin2);
            if (found==string::npos)
                return false;
        }
        return true;        
    }
};
