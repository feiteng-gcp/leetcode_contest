class Solution {
public:
    bool queryString(string S, int N) {
        int n = S.length();

        unordered_set<int> s;
        for (int i = 0;i < n;i++) {
            int tmp = 0;
            for (int j = i;j < i+30 && j < n;j++) {
                tmp *= 2;
                tmp += S[j] - '0';
                //cout << tmp << endl;
                if (tmp >= 1 && tmp <= N) {
                    s.insert(tmp);
                }
            }
            
        }
        return s.size() == N;
    }
};
