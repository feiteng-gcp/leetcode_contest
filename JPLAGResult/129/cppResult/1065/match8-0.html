class Solution {
public:
    string calc(int v){
        string ans;
        while(v){
            if(v%2) ans = "1" + ans;
            else ans = "0" + ans;
            v /= 2;
        }
        
        return ans;
    }
    
    bool queryString(string S, int N) {
        int n = S.size();
        int maxval = n * (n + 1) / 2;
        if(N > n) return false;
        
        for(int i = N; i > N / 2; --i){
            string str = calc(i);
            if(S.find(str) == string::npos) return false;
        }
        
        return true;
    }
};
