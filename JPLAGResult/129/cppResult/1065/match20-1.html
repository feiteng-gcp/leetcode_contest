typedef long long ll;


class Solution {
 public:
  bool queryString(string S, int N) {
    int cnt = 0;
    set<int> vix;

    for (int i = 0; i < S.size(); i++) {
      ll cur = 0;
      for (int j = 0; j < 30 && i + j < S.size(); j++) {
        cur = cur * 2 + (S[i+j] - '0');
        vix.insert(cur);
      }
    }
    
    if (N > vix.size()) {
      return false;

    }

    for(int i=1;i<=N;i++) {
      if(vix.find(i) == vix.end()) return false;
    }
    return true;
  }
};
