class Solution {
public:
    bool queryString(string S, int N) {
        unordered_set<long long > s;
        for(int i =0;i<S.size();i++){
            long long arg = (S[i] == '1');
            s.insert(arg);
            for(int j =i+1;j<S.size();j++){
                if(((long long)N+1)/2 < arg) break;

                arg = arg * 2 + (S[j] == '1');
                s.insert(arg);
            }
        }
        for(int i=0;i<N;i++){
            if(!s.count(i+1)) return false;
        }
        return true;
    }
};
