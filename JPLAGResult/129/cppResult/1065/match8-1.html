class Solution {
public:
    string convertToBinary(int num) {
        if (num == 0) return "0";
        string res = "";

        while (num != 0) {
            if (num % 2 == 0) res = "0" + res;
            else res = "1" + res;
            num /= 2;
        }
        //cout << num << ": " << res << "\n";
        return res;
    }

    bool queryString(string S, int N) {

        for (int i = 1; i <= N; i++) {
            string str = convertToBinary(i);
            if (S.find(str, 0) == string::npos) return false;
        }
        return true;
    }
};
