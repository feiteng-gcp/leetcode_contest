class Solution {
public:
    bool queryString(string S, int N) {
        set<int> h;
        int M = 1000000000;
        for (int i = 0;i < S.length();i ++) {
            int cur = 0;
            for (int j = i; j < S.length();j ++) {
                cur = cur * 2 + S[j] - '0';
                if (cur > M) break;
                h.insert(cur);
            }
        }
        if (N > h.size()) return false;
        for (int i = 1; i <= N;i ++) {
            if (h.find(i) == h.end()) return false;
        }
        return true;
    }
};
