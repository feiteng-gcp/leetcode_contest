class Solution {
public:
    bool queryString(string S, int N) {
        set<int> Q;
        for (int d = 1; d < 30; ++d) {
            for (int i = 0; i < S.size(); ++i) {
                if (i+d>S.size()) continue;
                int v = 0;
                for (int j = 0; j < d; ++j) {
                    v = (v<<1)|(S[i+j]-'0');
                }
                Q.insert(v);

            }
        }
        if (Q.size() < N) return false;
        for (int i = 1; i <= N; ++i) {
            if (!Q.count(i)) return false;
        }
        return true;
    }
};
