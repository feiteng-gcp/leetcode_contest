class Solution {
public:
    bool queryString(string S, int N) {
        int n = S.size();
        reverse(S.begin(), S.end());
        set<int> arr;
        for(int len = 1; len <= min(n, 31); len++){
            for(int i = 0; i <= n-len; i++){
                int tmp = 0;
                for(int j = 0; j < len; j++){
                    if(S[i+j] == '1')tmp |= (1<<j);
                }
                arr.insert(tmp);
            }
        }
        for(int i = 1; i <= N; i++){
            if(arr.find(i) == arr.end())return false;
        }
        return true;
    }
};
