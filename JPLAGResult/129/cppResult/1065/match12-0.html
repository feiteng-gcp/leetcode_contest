int ok[2050];
class Solution {
public:
    bool queryString(string S, int N) {
        if(N>2048)return false;
        memset(ok,0,sizeof(ok));
        int len=S.size();
        for(int l=1;l<=11;l++)
        {
            int now=0;
            for(int i=0;i<l-1;i++)
            {
                now=now*2+(S[i]-'0');
            }
            for(int i=l-1;i<len;i++)
            {
                now=(now*2+S[i]-'0')%(1<<l);
                ok[now]=1;
            }
        }
        for(int i=1;i<=N;i++)
        {
            if(!ok[i])return false;
        }
        return true;
    }
};
