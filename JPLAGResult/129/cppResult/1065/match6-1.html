class Solution {
public:
  bool queryString(string s, int k) {
    set<int> ss;
    int n=s.size();
    for(int i=0;i<n;i++){
      if(s[i]=='0') continue;

      int res=0;
      for(int j=i;j<n;j++){
        res=res*2+(s[j]-'0');
        if(res>k) break;
        ss.emplace(res);
      }
    }
    return (int)ss.size()==k;
  }
};
