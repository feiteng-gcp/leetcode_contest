class Solution {
public:
    bool queryString(string S, int N) {
        if (N > 1000000) return false;
        unordered_set<int> nums;
        for (int i = 1; i <= N; i++) nums.insert(i);

        for (int i = 0; i < S.size(); i++) {
            int k = 0;
            for (int j = i; j < S.size() && k < N; j++) {
                k *= 2;
                k += S[j] - '0';
                nums.erase(k);
            }
        }
        return nums.empty();
    }
};
