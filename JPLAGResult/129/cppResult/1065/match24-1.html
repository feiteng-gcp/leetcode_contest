class Solution {
public:
    bool queryString(string S, int NN) {

        for(int NA = 1 ; NA <= NN ; NA++){
            string bin = "";
            int N = NA;
            while(N){
                bin += N % 2 + '0';
                N /= 2;
            }
            reverse(bin.begin(), bin.end());
            if(S.find(bin) == -1) return false;
        }
        return true;
    }
};
