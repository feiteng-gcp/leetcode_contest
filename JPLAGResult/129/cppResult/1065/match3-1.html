class Solution {
public:
    bool queryString(string S, int N) {
        unordered_set<int> avail;
        for (int i = 0; i < S.size(); ++i) {

            for (int num = 0, j = i; j < S.size(); ++j) {
                num = num * 2 + (S[j] - '0');
                if (num > 1e9) break;
                avail.emplace(num);
            }
        }
        if (N > avail.size()) return false;
        for (int i = 1; i <= N; ++i) {
            if (!avail.count(i)) return false;
        }
        return true;
    }
};
