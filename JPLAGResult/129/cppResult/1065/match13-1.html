class Solution {
public:
    bool queryString(string s, int N) {
        int n = s.size();
        set<int> H;
        for (int i = 0; i < n; ++ i)
        {
            int x = 0;
            for (int j = 0; j < 30 && i+j < n; ++ j)
            {
                x = x*2+s[i+j]-'0';
                H.insert(x);
            }
        }
        for (int i = 1; i <= N; ++ i)
            if (!H.count(i)) return false;
        return true;
    }
};
