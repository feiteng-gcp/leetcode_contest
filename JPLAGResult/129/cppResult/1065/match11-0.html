class Solution {
public:
    bool queryString(string S, int N) {
        if(N > 2047) return false;
        vector<bool> vis(2048, false);
        int n = S.length();
        for(int i=0;i<n;i++){
            int now = 0;
            for(int j=i;j<i+11 && j<n;j++){
                now = now*2 + (S[j]-'0');
                vis[now] = true;
            }
        }
        for(int i=0;i<=N;i++){
            if(!vis[i])
                return false;
        }
        return true;
    }
};
