class Solution {
    int convertBinaryString(string s, int max) {
        int val = 0;
        for (int i = 0; i < s.size(); i++) {
            val = val * 2 + s[i]-'0';
            if (val > max)
                return INT_MAX;
        }
        
        return val;
    }
public:
    bool queryString(string S, int N) {
        vector<bool> canDo(N+1, false);
        int left = N;
        for (int i = 0; i < S.size(); i++) {
            for (int j = i; j < S.size(); j++) {
                int id = convertBinaryString(S.substr(i,j-i+1), N);
                if (id <= N && id != 0 && canDo[id] == false) {
                    canDo[id] = true;
                    left--;
                    if (left == 0)
                        return true;
                }
            }
        }
        
        return false;
    }
};
