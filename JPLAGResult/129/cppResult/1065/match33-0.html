typedef unsigned long long ull;
class Solution {
public:
    unordered_set<ull> M;
    bool queryString(string S, int N) {
        int len=S.length();
        for(int i=len-1; i>=0; i--) {
            ull tmp=0;
            for(int j=i; j>=0; j--) {
                tmp=tmp*233+(S[j]-'0'+1);
                M.insert(tmp);
            }
        }
        
        if (N>1000000) return false;
        
        for(int i=1; i<=N&&i<=1000000; i++) {
            ull tmp=0;
            int num=i;
            while(num) {
                tmp=tmp*233+(num%2+1);
                num/=2;
            }
            // cout<<tmp<<endl;
            // cout<<tmp<<endl;
            if (!M.count(tmp)) return false;
        }
        
        return true;
    }
};
