class Solution {
public:
    bool queryString(string S, int N) {
        unordered_set<int> nums;
        int slen = S.size();
        for (int i = 0; i < slen; ++i) {
            int num = 0;
            for (int j = 0; j < 30 && i + j < slen; ++j) {
                num = num * 2 + (S[i + j] - '0');
                nums.insert(num);
            }
        }
        
        int max_n = 0;
        for (int j = 1; j < 32000; ++j) {
            if (nums.find(j) == nums.end()) {
                break;
            }
            max_n = j;
        }
        return max_n >= N;
    }
};
