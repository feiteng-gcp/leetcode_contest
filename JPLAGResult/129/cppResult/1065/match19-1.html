bool u[1000000];
class Solution {
public:
    bool queryString(string S, int N) {
        int len = S.length();
        if (N > S.length() * (S.length() + 1) / 2) return false;
        memset(u, 0, sizeof(u));

        for (int i = 0; i < len; i++) {
            int p = 0;
            for (int j = i; j < len; j++) {
                p = p * 2 + S[j] - '0';
                if (p > N) break;
                u[p] = true;
//                printf("%d\n", p);

            }
        }
        for (int i = 1; i <= N; i++) {
            if (!u[i]) return false;
        }
        return true;
    }
};
