class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        if (A.size() < 3) return false;
        int sum = accumulate(A.begin(), A.end(), 0);
        if (sum % 3 != 0)
            return false;
        sum /= 3;
        int sum2 = 0, n = A.size();
        bool b1 = false, b2 = false;
        for (int i = 0; i < n; i++)
        {
            sum2 += A[i];
            if (b1 && !b2 && sum2 == sum * 2 && i != n - 1)
                b2 = true;
            if (!b1 && sum2 == sum)
                b1 = true;
        }
        return b2;
    }
};
