class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = 0;
        for (auto x : A) {
            sum += x;
        }
        if (sum % 3 != 0) {return false;}

        int target = sum / 3;
        int res = 0;
        int cur = 0;
        for (auto x : A) {
            cur += x;
            if (cur == target && res < 2) {
                ++res;
                cur = 0;
            }
        }
        return cur == target;
    }
};
