class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int n = A.size();
        int s = 0;
        for(int v: A) {
            s += v;
        }
        if(s%3)
            return false;
        int d = s/3;
        int sl = 0;
        int posl = n;
        for(int i=0; i<n; i++) {
            sl += A[i];
            if(sl == d) {
                posl = i;
                break;
            }
        }
        int sr=0;
        int posr = -1;
        for(int i=n-1; i>=0; i--) {
            sr += A[i];
            if(sr == d) {
                posr = i;
                break;
            }
        }
        if(posl < posr)
            return true;
        return false;
    }
};
