class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = accumulate(A.begin(), A.end(), 0);
        if (sum % 3) return false;
        int part = sum / 3;

        int subsum = 0;
        int cnt = 0;
        for (int i = 0; i < A.size(); i++) {
            subsum += A[i];
            if (subsum == part) {
                cnt ++;
                subsum = 0;
            }
        }
        return cnt == 3;
        
    }
};
