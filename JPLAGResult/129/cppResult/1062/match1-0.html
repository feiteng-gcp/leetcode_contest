class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum=accumulate(begin(A), end(A), 0);
        if (sum%3) return false;

        int s=0, c=0;
        for(int e: A) {
            s+=e;
            if (s==sum/3) { s=0, ++c; }
        }
        return c==3;
    }
};
