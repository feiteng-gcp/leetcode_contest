class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int s = 0;
        for (int i = 0; i < A.size(); ++i)
            s += A[i];
        
        if (s % 3 != 0) return false;
        int p = s / 3;
        int s1 = 0;
        int p1 = -1, p2 = -1;
        for (int i = 0; i < A.size() - 1; ++i) {
            s1 += A[i];
            if (s1 == p && p1 == -1)
                p1 = i;
            if (s1 == 2 * p && p1 != -1)
                p2 = i;
        }
        
        printf("%d %d %d %d\n", p1, p2, s, p);
        return p1 < p2;
    }
};
