class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = 0;
        for (int i =0 ;i < A.size();i++) {
            sum += A[i];
        }
        if ((sum % 3) != 0)
            return 0;
        int tmp = 0;
        sum /= 3;
        int cnt = 0;
        for (int i = 0;i < A.size();i++) {
            tmp += A[i];
            if (tmp == sum) {
                tmp = 0;
                cnt++;
            }
        }
        //cout << tmp << " " << cnt << endl;
        return (tmp == 0) && (cnt >= 3);
    }
};
