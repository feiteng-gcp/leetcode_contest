class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
      int tot = 0;
      for(int i=0;i<A.size();i++) tot += A[i];

      int cur = 0;

      bool b1=false,b2=false;
      for(int i=0;i<A.size();i++) {
        cur += A[i];
        if (cur * 3 == tot) b1 = true;
        if (b1 && cur * 3 == tot * 2) b2= true;
      }
      return b1 && b2;
    }
};
