class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& a) {
        long long sum=0;
        int n=(int)a.size();
        for(int i=0;i<n;i++)
            sum+=a[i];
        if(sum%3)return false;
        sum/=3;
        int f=-1,g=-1;
        long long cnt=0;
        for(int i=0;i<n;i++){
            cnt+=a[i];
            if(cnt==sum){
                f=i;
                break;
            }
        }
        if(f==-1)return false;
        cnt=0;
        for(int i=f+1;i<n;i++){
            cnt+=a[i];
            if(cnt==sum){
                g=i;
                break;
            }
        }
        if(g==-1||g==n-1)return false;
        return true;
    }
};
