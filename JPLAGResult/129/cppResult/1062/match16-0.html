class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = 0;
        for (auto val : A) {
            sum += val;
        }
        if (sum % 3 != 0) {
            return false;
        }
        int avg = sum / 3;
        int cur = 0;
        int count = 0;
        for (int i = 0; i < A.size(); i++) {
            cur += A[i];
            if (cur == avg) {
                cur = 0;
                count++;
            }
        }
        return count == 3;
    }
};
