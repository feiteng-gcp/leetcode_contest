class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int total = 0;
        for (auto i : A)
            total += i;
        if (total % 3 != 0)
            return false;

        int sum = total / 3;
        int curr = 0;
        
        int count = 0;
        for (int i = 0; i < A.size(); i++) {
            curr += A[i];
            if (curr == sum) {
                curr  = 0;
                count++;
            }
        }
        
        return count >= 3;
    }
};
