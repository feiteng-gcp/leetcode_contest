class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = 0;
        for(auto e : A)
            sum += e;
        
        if(sum / 3 * 3 != sum)
            return false;
        int tmp = 0;
        int cnt = 0;
        for(int i = 0; i < A.size(); i ++){
            tmp += A[i];
            if(tmp == sum / 3){
                cnt ++;
                tmp = 0;
            }
        }
        return cnt == 3;
    }
};
