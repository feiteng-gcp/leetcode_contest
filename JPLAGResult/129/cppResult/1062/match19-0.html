class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = 0;
        for (auto it: A) sum += it;

        if (sum % 3) return false;
        
        int tar = sum / 3;
        int tmp = 0;
        int seg = 0;
        for (auto it: A) {
            tmp += it;
            if (tmp == tar) {
                tmp = 0;
                seg += 1;
            }
        }
        return seg == 3 & tmp == 0;
    }
};
