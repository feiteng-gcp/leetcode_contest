class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
      long long int total_A = 0;
      for(int i=0; i<A.size(); ++i){
        total_A += A[i];
      }
      vector<int> A_pre(A.size()+1, 0);
      vector<int> A_post(A.size()+1, 0);
      int pre_pos=-1;

      int post_pos=-1;
      for(int i=1; i<A.size(); ++i){
        A_pre[i] = A_pre[i-1] + A[i-1];
        if(A_pre[i] == total_A/3){
          pre_pos = i-1;
          break;
        }
      }

      for(int i=A.size()-1; i>=0; --i){
        A_post[i] = A_post[i+1] + A[i];
        if(A_post[i] == total_A/3){
          post_pos = i;
          break;
        }
      }
      
      /*
      for(int i=0; i<A.size(); ++i){
        printf("%d ", A_pre[i]);
      }
      printf("\n");
      for(int i=0; i<A.size(); ++i){
        printf("%d ", A_post[i]);
      }
      */
      
      
      if(pre_pos+1 < post_pos && pre_pos >= 0 && post_pos >= 0) return true;
      
      return false;
    }
};
