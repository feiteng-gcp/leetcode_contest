class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& a) {
        int sum = 0;
        for (auto& it : a) sum += it;
        if (sum % 3) return false;
        sum /= 3;
        cout << sum << endl;
        int n = a.size();

        int L, R, cur = 0;
        for (L = 0; L < n; ++L) {
            cur += a[L];
            if (cur == sum) break;
        }
        if (L == n) return false;
        cur = 0;
        for (R = n - 1; R >= 0; --R) {
            cur += a[R];
            if (cur == sum) break;
        }
        if (R < 0) return false;
        return L + 1 < R;
    }
};
