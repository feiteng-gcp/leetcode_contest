class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int total = accumulate(allof(A), 0);
        if(total % 3) return false;
        int target = total / 3;
        int _sum = 0, cnt = 0;
        for(int i = 0; i < A.size(); i++){
            _sum += A[i];
            if(_sum == target){
                cnt++;
                _sum = 0;
            }
        }
        return _sum == 0 and cnt == 3;
    }
};
