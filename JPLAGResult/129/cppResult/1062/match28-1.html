class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = accumulate(A.begin(), A.end(), 0);
        if (sum % 3 != 0) {
            return false;
        }
        int cap = sum / 3;
        int left = -1, acc = 0;
        for (int i = 0; i < A.size(); ++i) {
            acc += A[i];
            if (acc == cap) {
                left = i;
                break;
            }
        }
        acc = 0;
        for (int i = A.size() - 1; i >= 0; --i) {
            acc += A[i];
            if (acc == cap) {
                if (left + 1 < i) {
                    return true;
                }
                return false;
            }
        }
        return false;
    }
};
