class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum=0, i, j, size=A.size(), partsum;
        for (i=0; i<size; ++i) sum+=A[i];
        if (sum%3!=0) return false;
        partsum = sum/3;
        sum = 0;
        for (i=0; i<size; ++i) {
            sum += A[i];
            if (sum==partsum) break;
        }
        if (i==size) return false;
        sum = 0;
        for (j=i+1; j<size; ++j) {
            sum +=A[j];
            if (sum==partsum) break;
        }
        if (j>=size-1) return false;
        return true;
    }
};
