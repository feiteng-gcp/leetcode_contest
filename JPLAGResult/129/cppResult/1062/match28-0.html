class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        int sum = accumulate(A.begin(), A.end(), 0);
        if (sum % 3 != 0) {
            return false;
        }
        int each = sum / 3;
        int tmp = 0, times = 0;
        for (int i = 0; i < A.size(); i++) {
            tmp += A[i];
            if (tmp == each) {
                if (++times == 2 && i < A.size() - 1) {
                    return true;
                }
                tmp = 0;
            }
        }
        return false;
    }
};
