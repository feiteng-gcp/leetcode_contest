class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {

        int sum = 0;
        for (int i=0; i<A.size();i++){
            sum+= A[i];
        }
        if (sum%3>0)
            return false;
        int cnt = 0;
        int run_sum = 0;
        for (int i=0; i<A.size();i++){
            run_sum += A[i];
            if (run_sum == sum/3){
                run_sum = 0;
                cnt++;
            }
        }
        if (cnt==3 && run_sum==0)
            return true;
        return false;
        
    }
};
