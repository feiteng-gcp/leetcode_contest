class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& A) {
        vector<int> sum;
        int n = A.size();
        sum.resize(n + 1, 0);
        for (int i = 0; i < n; i++)
        {
            sum[i + 1] = sum[i] + A[i];
        }
        if (sum[n] % 3) return 0;
        int w = sum[n] / 3;
        int flag = 0;
        for (int i = 1; i <= n; i++)
        {
            if (sum[i] == w && !flag) flag = 1;
            else if (sum[i] == 2*w && flag == 1) flag = 2;
        }
        return flag == 2;
    }
};
