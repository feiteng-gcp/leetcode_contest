class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& a) 
    {
        int ret=0;
        int m=a[0];
        int len=a.size();
        
        for(int i=1;i<len;i++)
        {
            ret=max(m+a[i]-i,ret);
            m=max(a[i]+i,m);
        }
        
        return ret;
    }
};
