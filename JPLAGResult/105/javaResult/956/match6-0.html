class Solution {
    public int numMusicPlaylists(int N, int L, int K) {
        final long MOD = 1000000007;
        long[][] cnt = new long[N + 1][L + 1];
        for (int nsong = 1; nsong <= N; nsong++) {
            cnt[nsong][1] = nsong; 
            for (int ilen = 2; ilen <= L; ilen++) {
                int nbanned = Math.min(nsong, Math.min(ilen - 1, K));
                cnt[nsong][ilen] = cnt[nsong][ilen - 1] * (nsong - nbanned);
                cnt[nsong][ilen] %= MOD;
            }
        }
        
        long[][] cnk = new long[N + 1][N + 1];
        cnk[1][0] = 1;
        cnk[1][1] = 1;
        for (int i = 2; i <= N; i++) {
            cnk[i][0] = 1;
            cnk[i][i] = 1;
            for (int j = 1; j < i; j++) {
                cnk[i][j] = cnk[i - 1][j - 1] + cnk[i - 1][j];
                cnk[i][j] %= MOD;
            }
        }
        
        long ret = cnt[N][L];
        for (int i = N - 1; i > 0; i--) {
            if ((N - i) % 2 == 1) {
                ret = (ret - (cnk[N][i] * cnt[i][L]) % MOD + MOD) % MOD;
            } else {
                ret += (cnk[N][i] * cnt[i][L]) % MOD;
                ret %= MOD;
            }
        }
        return (int)ret;
    }
}
