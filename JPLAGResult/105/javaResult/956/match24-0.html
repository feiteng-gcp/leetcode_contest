class Solution {
    
    private static final long modulo = 1000000007L;
    
    public int numMusicPlaylists(int N, int L, int K) {
        int[] nums = new int[N+1];
        int[] coef = new int[N+1];
        for(int i=0;i<=N;i++) {
            if(i<=K) nums[i] = 0;
            else nums[i] = numMusicPlaylistsIfNotEverySongNeedsToBePlayed(i, L, K);
            coef[i] = choose(N, i);
        }
        
        boolean shouldAdd = true;
        int ret = 0;
        for(int i=N;i>=0;i--) {
            int tmp = mult(coef[i], nums[i]);
            if(shouldAdd) {
                ret = add(ret, tmp);
            } else {
                ret = subtr(ret, tmp);
            }
            shouldAdd = !shouldAdd;
        }
        return ret;
        
    }
    
    public int numMusicPlaylistsIfNotEverySongNeedsToBePlayed(int N, int L, int K) {
        int ret = 1;
        for(int i=0;i<L;i++) {
            int choice = N - Math.min(i, K); // number of 
            ret = mult(ret, choice);
        }
        return ret;
    }
    
    private int choose(int n, int m) {
        //return @n choose @m modulo @modulo
        int ret = 1;
        for(int i=m+1;i<=n;i++) {
            ret = mult(ret, i);
        }
        for(int i=2;i<=(n-m);i++) {
            ret = divide(ret, i);
        }
        return ret;
    }
    
    private int add(long a, long b) {
        long c = ((a+b) % modulo);
        return (int)(c);
    }
    
    private int subtr(long a, long b) {
        long c = ((a-b + modulo) % modulo);
        return (int)(c);
    }
    
    private int mult(long a, long b) {
        long c = ((a*b) % modulo);
        return (int)(c);
    }
    
    private int divide(long a, int b) {
        return mult(a, invert(b));
    }
    
    private int invert(int v) {
        //coef[0] * modulo + coef[1] * v == 1
        int[] coef = gcdCoef((int)(modulo), v);
        return coef[1];
    }
    
    private int[] gcdCoef(int x, int y) {
        //if a*x + b*y ==  gcd(x, y), return [a, b]
        
        if (y==0) return new int[] {1, 0};
        
        int q = x / y, r = x % y;
        
        //  gcd == v[0] * y + v[1] * r == v[0] * y + v[1] * (x - q * y) == v[1] * x + (v[0] - q * v[1]) * y
        int[] v = gcdCoef(y, r); 
        return new int[] {v[1], v[0] - q * v[1]};
    }
}
