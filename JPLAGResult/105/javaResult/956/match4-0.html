class Solution {
    public int numMusicPlaylists(int n, int p, int m) {
        int mod = (int)(1e9 + 7);
        long[][] dp = new long[p + 1][n + 1];
        dp[0][0] = 1;
        for (int i = 1; i <= p; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = (dp[i][j] + dp[i - 1][j - 1] * (n - j + 1) % mod) % mod;
                if (j > m) dp[i][j] = (dp[i][j] + dp[i - 1][j] * (j - m) % mod) % mod;
            }
        }
        return (int)dp[p][n];
    }
}
