class Solution {
    public int numMusicPlaylists(int N, int L, int K) {
        
long[][] dp = new long[L + 1][N + 1];
        dp[1][1] = 1;
        for (int l = 2; l <= L; l++) {
            for (int n = 1; n <= N; n++) {
                dp[l][n] = mod(dp[l - 1][n] * Math.max(0, n - K) + dp[l - 1][n - 1]);
            }
        }
        long o = mod(dp[L][N] * factorial(N));
        return (int) o;
    }

    private long factorial(long i) {
        long o = 1;
        while (i > 1) {
            o = mod(o * i);
            i--;
        }
        return o;
    }

    private long mod(long l) {
        return l % 1000000007;
    }
