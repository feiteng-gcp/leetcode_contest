class Solution {
    int mo=1000000007;
    long get(int n,int l,int k)
    {
        if (n<=k) return 0;
        long ans=1;
        for (int i=0;i<l;i++)
        {
            if (i<k) ans=(ans*(n-i))%mo;
            else ans=(ans*(n-k))%mo;
        }
        return ans;
    }
    public int numMusicPlaylists(int N, int L, int K) {
        int f=1;
        long ans=0;
        int[][] C=new int[N+1][N+1];
        C[0][0]=1;
        for (int i=1;i<=N;i++)
        {
            C[i][0]=C[i][i]=1;
            for (int j=1;j<i;j++) C[i][j]=(C[i-1][j]+C[i-1][j-1])%mo;
        }
        for (int i=N;i>=1;i--)
        {
            long now=get(i,L,K)*f;
            now=(now*C[N][i])%mo;
            ans=(ans+now+mo)%mo;
            f=-f;
        }
        return (int)ans;
    }
}
