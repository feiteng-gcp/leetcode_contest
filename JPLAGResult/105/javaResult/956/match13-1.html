class Solution {
    private static final int MOD = 1000000007;
    
public int numMusicPlaylists(int N, int L, int K) {
        long[][] dp = new long[N + 1][L + 1];
        dp[0][0] = 1;
        for (int i = 1; i <= L; i++) {
            for (int j = 1; j <= N; j++) {
                dp[j][i] = (dp[j - 1][i - 1] * (N - j + 1) + dp[j][i - 1] * (Math.max(0, j - K))) % MOD;
            }
        }
        return (int)dp[N][L];
    }
}
