class Solution {
    
    
    long [] fact = new long [100 + 1];
    {
        fact[0] = 1;
        for(int m = 1 ; m < fact.length ; m++) {
            fact[m] = (fact[m - 1] * m) %  1_000_000_007;
        }
    }
    long ifact(int m) {
        return pow(fact[m], 1_000_000_007 - 2);
    }
    long pow(long k, int pow) {
        if(pow == 1) return k;
        long r = pow(k, pow / 2);
        r = (r * r) % 1_000_000_007;
        if(pow % 2 == 1) {
            r = (r * k) % 1_000_000_007;
        }
        return r;
    }
    
    public long c(int m, int n) {
        long r = fact[n];
        r = (r * ifact(m)) % 1_000_000_007;
        r = (r * ifact(n - m)) % 1_000_000_007;
        return r;
    }
    
    
    public int numMusicPlaylists(int n, int l, int k) {
        int mm = 1_000_000_007;
        long [] table = new long [n + 1];
        int sum = 0;
        for(int s = 1; s <= n ; s++) {
            table[s] = px(s, l, k);
            for(int rr = 1; rr < s ; rr++) {
                table[s] += mm - (c(rr, s) * table[rr]) % mm;
                table[s] %= mm;
                
            }   
        }
        return (int)(table[n]);
    }
    
    int px(int n, int l, int k) {
        long res = 1;
        long mm = 1_000_000_007;
        int a = n;
        for(int m = 0 ; m < l ; m++) {
            res *= a;
            res %= mm;
            if(m < k) 
{
                a --;    
            }
        }
        return (int)(res);
    }
    
}
