class Solution {
    public int numMusicPlaylists(int N, int L, int K) {
        long[][] dp = new long[L][N + 1];
        dp[0][1] = N;
        long MOD = 1000000007;
        for (int i = 1; i < L; i++) {
            for (int j = 1; j <= N; j++) {
                if (i > K) dp[i][j] += (j >= K  ? j - K : 0) * dp[i - 1][j]
;
                dp[i][j] += (N - (j - 1)) * dp[i - 1][j - 1];
                dp[i][j] %= MOD;
            }
        }
        return (int)dp[L - 1][N];
    }
}
