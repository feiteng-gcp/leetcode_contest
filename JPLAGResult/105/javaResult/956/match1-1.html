class Solution {
    public int numMusicPlaylists(int N, int L, int K) {
        long mod = 1_000_000_007;
        long[][] dp = new long[2][N + 1];
        dp[0][0] = 1;
        for (int i = 1; i <= L; ++i) {
            for (int j = 0; j <= N; ++j) {
                if (i < j) {
                    dp[i & 1][j] = 0;
                } else if (i == j) {
                    dp[i & 1][j] = (dp[(i - 1) & 1][j - 1] * i) % mod;
                } else if (j > K) {
                    dp[i & 1][j] = (dp[(i - 1) & 1][j - 1] * j + dp[(i - 1) & 1][j] * (j - K)) % mod;
                } else {
                    dp[i & 1][j] = 0;
                
}
            }
        }
        return (int)dp[L & 1][N];
    }
}
