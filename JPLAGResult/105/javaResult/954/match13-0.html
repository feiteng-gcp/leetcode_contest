class Solution {
    public int maxSubarraySumCircular(int[] A) {
        if (A == null || A.length == 0) {
            return 0;
        }
        int preSumMin = 0;
        int preSumMax = 0;
        int preSum = 0;
        int sumMin = Integer.MAX_VALUE;
        int sumMax = Integer.MIN_VALUE;
        int len = A.length;
        for (int i = 0; i < len; i++) {
            preSum += A[i];
            sumMax = Math.max(preSum - preSumMin, sumMax);
            if (i != len - 1) {
                sumMin = Math.min(preSum - preSumMax, sumMin);
            }
            preSumMin = Math.min(preSumMin, preSum);
            preSumMax = Math.max(preSumMax, preSum);
        }
        
        return Math.max(sumMax, preSum - sumMin);
    }
}
