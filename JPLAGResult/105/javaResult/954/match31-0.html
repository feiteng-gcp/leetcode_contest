class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int o = Arrays.stream(A).max().getAsInt();
        if (o < 0) {
            return o;
        }
        int sum = Arrays.stream(A).sum();
        int max = maxSubarraySum(A);
        for (int i = 0; i < A.length; i++) {
            A[i] = -A[i];
        }
        int min = -maxSubarraySum(A);
        o = Math.max(o, max);
        o = Math.max(o, sum - min);
        return o;
    }

    private int maxSubarraySum(int[] A) {
        int o = 0;
        int sum = 0;
        for (int i = 0; i < A.length; i++) {
            sum += A[i];
            o = Math.max(o, sum);
            sum = Math.max(0, sum);
        }
        return o;
    }

    public static void main(String[] args) {
        Solution s = new Solution();

//        System.out.println(s.reverseOnlyLetters("ab-cd"));
//        System.out.println(s.reverseOnlyLetters("a-bC-dEf-ghIj"));
//        System.out.println(s.reverseOnlyLetters("Test1ng-Leet=code-Q!"));
        System.out.println(s.maxSubarraySumCircular(new int[] {1,-2,3,-2}));
        System.out.println(s.maxSubarraySumCircular(new int[] {5,-3,5}));
        System.out.println(s.maxSubarraySumCircular(new int[] {3,-1,2,-1}));
        System.out.println(s.maxSubarraySumCircular(new int[] {3,-2,2,-3}));
        System.out.println(s.maxSubarraySumCircular(new int[] {-2,-3,-1}));
        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
    }
}
