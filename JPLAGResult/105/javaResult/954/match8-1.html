class Solution {
    public int maxSubarraySumCircular(int[] a) {
        int max = a[0];
        int cur = a[0];
        
for(int s = 1 ; s < a.length ; s++) {
            cur = Math.max(cur + a[s], a[s]);
            max = Math.max(cur, max);
        }
        
        int [] mp = new int[a.length];
        
mp[0] = a[0];
        int p = a[0];
        for(int k = 1 ; k < a.length ; k++) {
            p += a[k];
            mp[k] = Math.max(mp[k - 1], p);
        }
        int ss = 0;
        
        
for(int t = a.length - 1; t > 0 ; t--) {
            ss += a[t];
            max = Math.max(ss + mp[t - 1], max);
        }
        
        
        return max;
    }
}
