class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int maxSum = getMaxSum(A);
        boolean allNegative = false;
        if (allNegative(A)) {
            allNegative = true;
        }
        
int maxWrap = 0;
        int max = A[0];
        for (int i = 0; i < A.length; i++) {
            max = Math.max(max, A[i]);
            maxWrap += A[i];
            A[i] = -A[i];
        }
        maxWrap += getMaxSum(A);
        if (allNegative) {
            return max;
        }
        return (maxWrap > maxSum) ? maxWrap : maxSum;
    }
    
    private boolean allNegative(int[] nums) {
        for (int num : nums) {
            if (num >= 0) {
                return false;
            }
        }
        return true;
    
}
    
    private int getMaxSum(int[] nums) {
        int curSum = nums[0];
        int max = curSum;
        for (int k = 1; k < nums.length; k++) {
            curSum = Math.max(nums[k], nums[k] + curSum);
            max = Math.max(max, curSum);
        }
        return max;
    }
    
}
