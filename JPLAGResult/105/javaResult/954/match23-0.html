class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int res=Integer.MIN_VALUE;
        int minPrefix=0;
        int total=0;
        int[] leftMax=new int[A.length];
        for(int i=0;i<A.length;i++){
            total=total+A[i];
            res=Math.max(res,total-minPrefix);
            minPrefix=Math.min(minPrefix,total);
            leftMax[i]=Math.max(i==0?0:leftMax[i-1],total);
        }
        int rightMax=A[A.length-1];
        total=0;
        for(int i=A.length-1;i>0;i--){
            total=total+A[i];
            rightMax=Math.max(rightMax,total);
            res=Math.max(res,rightMax+leftMax[i-1]);
        }
        return res;
    }
}
