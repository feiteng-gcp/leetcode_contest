class Solution {
    public int maxSubarraySumCircular(int[] A) {
        //corner cases 
        if(A == null || A.length == 0) return 0;
        
        //start here
        int[] sum = new int[A.length + 1];
        sum[0] = 0; //guaranteed for the existence of index 0
        int total_sum = 0;
        for(int i = 1; i <= A.length; i++){
            sum[i] = sum[i - 1] + A[i - 1];
            total_sum += A[i - 1];
        }
        
        //find the max_sum subarray
        
        int max_sum = Integer.MIN_VALUE;
        int min_sum = Integer.MAX_VALUE;
        int max = sum[0];
        int min = sum[0];
        
        //first find no cycle max
        for(int i = 1; i < sum.length; i++){
            min_sum = Math.min(min_sum, sum[i] - max);
            max_sum = Math.max(max_sum, sum[i] - min);
            if(i != sum.length - 1) max_sum = Math.max(max_sum, total_sum - min_sum);
            max = Math.max(max, sum[i]);
            min = Math.min(min, sum[i]);
        }
        
        //now consider the cycle
        
        
        
        return max_sum;
    }
}
