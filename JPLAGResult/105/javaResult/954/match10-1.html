class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int len = A.length;
        int[] pm = new int[len];
        int[] sm = new int[len];
        pm[0] = A[0];
        sm[len - 1] = A[len - 1];
        
        int s = A[0];
        for (int i = 1; i < len; ++i) {
            s += A[i];
            pm[i] = Math.max(s, pm[i - 1]);
        }
        
        s = A[len - 1];
        for (int i = len - 2; i >= 0; --i) {
            s += A[i];
            sm[i] = Math.max(s, sm[i + 1]);   
        }
        
        int res = pm[0];
        int mi = 0;
        s = 0;
        for (int i = 0; i < len; ++i) {
            s += A[i];
            res = Math.max(res, s - mi);
            
mi = Math.min(mi, s);
        }
        
        for (int i = 0; i < len - 1; ++i) {
            res = Math.max(res, pm[i] + sm[i + 1]);
        }
        return res;
    }
}
