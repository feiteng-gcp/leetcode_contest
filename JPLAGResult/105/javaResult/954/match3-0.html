class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int n=A.length;
        int max=kadane(A);
        int max_wrap  =0; 
        for (int i=0; i<n; i++){ 
            max_wrap+= A[i];
            A[i]= -A[i]; 
        } 
        max_wrap+= kadane(A); 
        if(max_wrap!=0){
            return Math.max(max,max_wrap);
        }else{
            return max;
        }
            
    }

    public static int kadane(int[] a){
        int n=a.length;
        int[]dp=new int[n];
        int maxending=0;
        int maxsofar=0;
        
        for(int i=0; i<n; i++){
            maxending+=a[i];
            if(maxending<0){
                maxending=0;
            }
            if(maxsofar<maxending){
                maxsofar=maxending;
            }
        }
        int max=-31000;
        if(maxsofar==0){
            for(int i=0; i<n; i++){
                max=Math.max(max,a[i]);
            }
            return max;
        }
        return maxsofar;
    
    }
}
