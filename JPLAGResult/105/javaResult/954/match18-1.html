class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int n = A.length;
        
        int max = Integer.MIN_VALUE;
        int dp = 0;
        for (int i = 0; i < n; i++) {
            dp += A[i];
            if (dp > max) max = dp;
            if (dp < 0) dp = 0;
        }
        
        
int[] maxRsum = new int[n];
        int rsum = A[n-1];
        maxRsum[n-1] = rsum;
        for (int i = n - 2; i >= 0; i--) {
            rsum += A[i];
            maxRsum[i] = Math.max(maxRsum[i+1], rsum);
        }
        int lsum = 0;
        for (int i = 0; i < n - 1; i++) {
            lsum += A[i];
            int cur = lsum + maxRsum[i+1];
            if (cur > max) max = cur;
            
        }
        return max;
    }
}
