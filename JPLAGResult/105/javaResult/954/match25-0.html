class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int maxCommon = common(A);
        
        int maxSide = 0;
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < A.length; i++) {
            max = Math.max(max, A[i]);
            maxSide += A[i];
            A[i] = -A[i];
        }
        if(max < 0) {
            return max;
        }
        
        maxSide = maxSide + common(A);
        return maxCommon > maxSide ? maxCommon : maxSide;
    }
    private int common(int A[]) {
        int max = 0;
        int end = 0;
        for(int i = 0; i < A.length; i++) {
            end = end + A[i];
            if(end < 0) {
                end = 0;
            } 
            if(max < end) {
                max = end;
            }
        }
        return max;
    }
}
