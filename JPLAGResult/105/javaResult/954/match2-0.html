class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int N = A.length;
        int cand1 = maxSubArr(A);
        if(allNeg(A)) return cand1;
        int total = 0;
        for(int i = 0; i < N; ++i) {
            total += A[i];
            A[i] = -A[i];
        }
        int cand2 = total + maxSubArr(A);
        return Math.max(cand1, cand2);
    }
    
    private boolean allNeg(int[] A) {
        for(int i = 0; i < A.length; ++i) {
            if(A[i] >= 0) return false;
        }
        return true;
    }
    private int maxSubArr(int[] A) {
        int ans = -30001;
        int maxEndHere = 0;
        for(int i = 0; i < A.length; ++i) {
            maxEndHere += A[i];
            ans = Math.max(ans, maxEndHere);
            if(maxEndHere < 0) 
                maxEndHere = 0;
        }
        return ans;
    }
}
