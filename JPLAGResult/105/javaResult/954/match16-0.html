class Solution {
    public int maxSubarraySumCircular(int[] A) {
        if(A.length == 1){
            return A[0];
        }else{
            int max = A[0];
            int curMax = A[0];
            int curMaxStartIndex = 0;
            for(int i = 1; i<A.length; i++){
                if(curMax > 0){
                    curMax += A[i];
                }else{
                    curMax = A[i];
                    curMaxStartIndex = i;
                }
                max = Math.max(max, curMax);
            }
            int[] fixedIndexMax = new int[A.length];
            int b = A[A.length - 1];
            int bMax = A[A.length - 1];
            fixedIndexMax[A.length - 1] = A[A.length - 1];
            for(int j = A.length - 2; j >= 0; j--){
                b += A[j];
                if( b > bMax ){
                    bMax = b;
                    fixedIndexMax[j] = b;
                }else{
                    fixedIndexMax[j] = bMax;
                }
            }
            
            int a = 0;
            for(int j = 0; j <A.length - 1; j++){
                a += A[j];         
                max = Math.max(max, fixedIndexMax[j + 1] + a);
            }
            return max;
            
        }
    }
}
