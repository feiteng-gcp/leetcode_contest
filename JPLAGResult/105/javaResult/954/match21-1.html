class Solution {
    public int maxSubarraySumCircular(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        boolean allNegative = true;
        int negMax = Integer.MIN_VALUE;
        for (int num : nums) {
            if (num > 0) {
                allNegative = false;
                break;
            }
            negMax = Math.max(negMax, num);
        }
        if (allNegative) {
            return negMax;
        }
        int orgSubMax = maxSubArray(nums);
        int max = 0;
        for (int i = 0; i < nums.length; i++) {
            max += nums[i];
            nums[i] = -nums[i];
        }
        int negSubMax = maxSubArray(nums);
        max += negSubMax;
        return 
Math.max(max, orgSubMax);
    }
    
    private int maxSubArray(int[] nums) {
        int max = nums[0];
        int sum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            sum = 
Math.max(sum + nums[i], nums[i]);
            max = Math.max(max, sum);
        }
        return max;
    }
}
