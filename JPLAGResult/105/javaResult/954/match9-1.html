class Solution {
    public int maxSubarraySumCircular(int[] A) {
        
        if(A.length == 0)
            return 0;
        
        int maxSum=A[0], sumNow= A[0];
        
        for(int i=1; i<A.length; i++){
            if(sumNow + A[i] > A[i]){
                sumNow += A[i];
            }
            else
                sumNow = A[i];
            maxSum = Math.max(maxSum, sumNow);
        }
        
        int[] left = new int[A.length];        
        
int max = A[0], sumTillNow = A[0];
        left[0] = max;
        for(int i=1; i<A.length; i++){
            sumTillNow += A[i];
            max = Math.max(max, sumTillNow);
            left[i] = max;
        }
        
        int[] right = new int[A.length];
        max = A[left.length-1];
        sumTillNow = A[left.length-1];
        right[left.length-1] = max;
        for(int i=left.length-2; i>=0; i--){
            sumTillNow += A[i];
            max = Math.max(max, sumTillNow);
            right[i] = max;
        
}
        
        
        for(int i=1; i<A.length; i++){
            maxSum = Math.max(maxSum, left[i-1]+right[i]);
        }
        return maxSum;
    }
}
