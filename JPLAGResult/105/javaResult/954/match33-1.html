class Solution {
    public int maxSubarraySumCircular(int[] A) {
        int n = A.length;
        int[] left = new int[n], right = new int[n];
        int cur;
        
        left[0] = A[0];
        cur = A[0];
        for(int i=1;i<n;i++) {
            cur += A[i];
            left[i] = Math.max(cur, left[i-1]);
        }
        
        right[n-1] = A[n-1];
        
cur = A[n-1];
        for(int i=n-2;i>=0;i--) {
            cur += A[i];
            right[i] = Math.max(cur, right[i+1]);
        }
        
        int ret = getRegular(A);
        
        for(int i=0;i<n-1;i++) {
            int tmp = left[i] + right[i+1];
            if(tmp>ret) {
                ret = tmp;
            }
        }
        return ret;
    }
    
    private int getRegular(int[] A) {
        int n = A.length;
        int ret = A[0];
        int cur = 0, i = 0, j = 0;
        while(j < n) {
            cur += A[j];
            j += 1;
            if (cur > ret) {
                ret = cur;
            }
            if (cur < 0) {
                cur = 0;
                i = j;
            }
        }
        return ret;
    }
    
    private int[] doubleSize(int[] A) {
        int n = A.length, m = n * 2;
        int[] ret = new int[m];
        for(int i=0;i<n;i++) {
            ret[i] = A[i];
            ret[i+n] = A[i];
        }
        return ret;
    }
    
}
