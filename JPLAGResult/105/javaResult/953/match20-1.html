class Solution {
    public String reverseOnlyLetters(String S) {
        char[] array = S.toCharArray();
        
        int start = 0;
        
int end = S.length() - 1;
        
        while (start < end) {
            while (start < end && !Character.isLetter(array[start])) {
                start++;
            }
            while (start < end && !Character.isLetter(array[end])) {
                end--;
            }
            char temp = array[start];
            array[start] = array[end];
            array[end] = temp;
            start++;
            end--;
        }
        
        return new String(array);
    }
}
