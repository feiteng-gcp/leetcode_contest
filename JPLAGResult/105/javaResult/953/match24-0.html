class Solution {
    public String reverseOnlyLetters(String s) {
        if(s == null || s.length() == 0)
            return s;
        
        char[] c = s.toCharArray();
        int a=0, b=c.length-1;
        
        while(a<b){
            while(a<b && !((c[a]>='a' && c[a]<='z') || (c[a]>='A' && c[a]<='Z')))
                a++;
            while(a<b && !((c[b]>='a' && c[b]<='z') || (c[b]>='A' && c[b]<='Z')))
                b--;
            if(a < b){
                char tmp = c[a];
                c[a] = c[b];
                c[b] = tmp;
                a++; b--;
            }
        }
        return new String(c);
    }
}
