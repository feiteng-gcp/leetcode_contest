	class Solution {
	    public String reverseOnlyLetters(String S) {
	        char[] s = S.toCharArray();
	        for(int i = 0, j = s.length-1;i < j;i++,j--){
	        	while(i < s.length && !ok(s[i]))i++;
	        	while(j >= 0 && !ok(s[j]))j--;
	        	if(i < j){
	        		char d = s[i]; s[i] = s[j]; s[j] = d;
	        	}
	        }
	        return new String(s);
	    }
	    
	    boolean ok(char c)
	    {
	    	return c >= 'A' && c <= 'Z' ||
        			c >= 'a' && c <= 'z';
	    }
	}
