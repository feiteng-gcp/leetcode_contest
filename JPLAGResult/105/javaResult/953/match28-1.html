class Solution {
    public String reverseOnlyLetters(String S) {
        char[] cs = S.toCharArray();
        int l = 0, h = cs.length - 1;
        
while (l < h) {
            while (l < h && !Character.isLetter(cs[l])) l++;
            while (l < h && !Character.isLetter(cs[h])) h--;
            char tmp = cs[l];
            cs[l] = cs[h];
            cs[h] = tmp;
            l++;
            h--;
        }
        return String.valueOf(cs);
    }
}
