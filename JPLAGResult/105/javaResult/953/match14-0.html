class Solution {
    public String reverseOnlyLetters(String S) {
        if (S.length() == 0)  return "";
        int i = 0, j = S.length()-1;
        char[] s = S.toCharArray();
        while (i < j) {
            if (!Character.isLetter(s[i])) i++;
            else if (!Character.isLetter(s[j])) j--;
            else {
                char temp = s[i];
                s[i] = s[j];
                s[j] = temp;
                i++;j--;
            }
        }
        return new String(s);
    }
}
