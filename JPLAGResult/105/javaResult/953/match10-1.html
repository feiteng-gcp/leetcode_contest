class Solution {
    
public String reverseOnlyLetters(String S) {
        char[] A = S.toCharArray();
        int l = 0, r = A.length - 1;
        while (l < r) {
            while(!letter(A[l]) && l < r) l++;
            while(!letter(A[r]) && l < r) r--;
            if (l < r) {
                char ch = A[l];
                A[l] = A[r];
                A[r] = ch;
                l++; r--;
            }
        }
        
        return new String(A);
    }
    
    boolean letter(char ch) {
        return (ch <= 'z' && ch >= 'a') || (ch <= 'Z' && ch >= 'A');
    }
}
