class Solution {
    public String reverseOnlyLetters(String S) {
    	char[] ch  = S.toCharArray() ;
    	int i = 0 , 
    			j = S.length() - 1;
        while ( i < j) {
        	if ( !this.isLetter(ch[i]) ){
        		i ++ ;
        		continue ;
        	}
        	if ( !this.isLetter(ch[j]) ){
        		j--;
        		continue ;
        	}
        	char temp = ch[i] ;
        	ch[i] = ch[j] ;
        	ch[j] = temp ;
            i ++ ;
        	j -- ;

        }
        return new String( ch) ;
    }
    
    private boolean isLetter( char c) {
    	return ( c >= 'a' && c<='z')
    			||
    			( c >= 'A' && c<='Z') ;
    }
}
