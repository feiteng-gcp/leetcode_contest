class Solution {
    public String reverseOnlyLetters(String S) {
        char[] array = S.toCharArray();
        int l = 0, r = S.length() - 1;
        
        while (l < r) {
            while (l < r && !Character.isLetter(array[l])) l++;
            while (l < r && !Character.isLetter(array[r])) r--;
            if (l >= r) break
;
            char tmp = array[l];
            array[l] = array[r];
            array[r] = tmp;
            l++;
            r--;
        }
        
        return new String(array);
    }
}
