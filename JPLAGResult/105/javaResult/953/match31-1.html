class Solution {
    
public String reverseOnlyLetters(String S) {
        char[] C = S.toCharArray();
        int left = 0;
        int right = C.length-1;
        while (left<right) {
            while (left < C.length && !((C[left] >= 'a' && C[left] <= 'z') || (C[left] >= 'A' && C[left] <= 'Z'))) left ++;
            while (right > 0 && !((C[right] >= 'a' && C[right] <= 'z') || (C[right] >= 'A' && C[right] <= 'Z'))) right --;
            if (left >= C.length) break;
            if (right <= 0) break;
            
if (left < right) {
                char temp = C[left];
                C[left] = C[right];
                C[right] = temp;
            }
            left ++;
            right --;
        }
        return new String(C);
    }
}
