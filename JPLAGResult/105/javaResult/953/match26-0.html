class Solution {
    public String reverseOnlyLetters(String S) {
        char[] a = S.toCharArray();
        int i = 0, j = a.length-1;
        while(i < j){
            while(i<a.length && !Character.isLetter(a[i])) i++;
            while(j>= 0 && !Character.isLetter(a[j])) j--;
            if(i>=j || i>= a.length || j<0) break;
            char t = a[i];
            a[i] = a[j];
            a[j] = t;
            i++;
            j--;
        }
        return new String(a);
    }
}
