/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class CBTInserter {
    int dept;
    TreeNode head;
    Deque<TreeNode> gap;

    public CBTInserter(TreeNode root) {
        head = root;
        gap = new ArrayDeque<TreeNode>();
        findGap(root);
    }
    
    void findGap(TreeNode root){
        Deque<TreeNode> level = new ArrayDeque<TreeNode>();
        level.offer(root);
        while(!level.isEmpty()){
            TreeNode cur = level.poll();
            boolean l = cur.left == null;
            boolean r = cur.right == null;
            if(l)
                gap.offer(cur);
            if(r)
                gap.offer(cur);
            if(!l)
                level.offer(cur.left);
            if(!r)
                level.offer(cur.right);
        }
    }
    
    public int insert(int v) {
        TreeNode cur = gap.poll();
        if(cur.left == null){
            cur.left = new TreeNode(v);
            findGap(cur.left);
        }
        else{
            cur.right = new TreeNode(v);
            findGap(cur.right);
        }
        return cur.val;
    }
    
    public TreeNode get_root() {
        return head;
    }
}
