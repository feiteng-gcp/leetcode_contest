/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class CBTInserter {
    TreeNode root;
    Queue<TreeNode> q = new LinkedList<>();
    public CBTInserter(TreeNode root) {
        this.root = root;
        q.offer(root);
        while(true){
            if(q.peek().left==null || q.peek().right==null){
                if(q.peek().left!=null){
                    q.offer(q.peek().left);
                }
                break;
            }
            TreeNode poll = q.poll();
            q.offer(poll.left);
            q.offer(poll.right);
        }
    }
    
    public int insert(int v) {
        TreeNode newNode = new TreeNode(v);
        TreeNode peek = q.peek();
        if(peek.left==null){
            peek.left = newNode;
        }else{
            peek.right = newNode;
            q.poll();
        }
        q.offer(newNode);
        return peek.val;
    }
    
    public TreeNode get_root() {
        return root;
    }
}
