/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class CBTInserter {

    
TreeNode root;
    Queue<TreeNode> q;
    public CBTInserter(TreeNode root) {
        this.root = root;
        q = new LinkedList<TreeNode>();
        q.offer(root);
        TreeNode node = q.peek();
        while(node != null){
            node = q.peek();
            if(node.left == null)
                break;
            q.offer(node.left);
            if(node.right == null)
                break;
            q.offer(node.right);
            q.poll();
        
}
    }
    
    public int insert(int v) {
        TreeNode node = q.peek();
        TreeNode tmp = new TreeNode(v);
        if(node.left == null){
            node.left = tmp;
        }
        else{
            node.right = tmp;
            q.poll();
        }
        q.offer(tmp);
        
return node.val;
    }
    
    public TreeNode get_root() {
        return this.root;
    }
}
