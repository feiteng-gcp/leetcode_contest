/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class CBTInserter {
    
    TreeNode root;
    Queue<TreeNode> queue = new LinkedList<>();
    Set<Integer> inQueue = new HashSet<>();
    
    public CBTInserter(TreeNode root) {
        this.root = root;
        TreeNode node = root;
        
        Queue<TreeNode> q = 
new LinkedList<>();
        q.add(root);
        
        
        while(!q.isEmpty()){            
            TreeNode treeNode = q.poll();
            if(treeNode.left==null||treeNode.right==null){
                queue.add(treeNode);   
                inQueue.add(treeNode.val);
                System.out.println("Adding "+treeNode.val);
            }
            if(treeNode.left!=null){
                q.add(treeNode.left);
            }
            if(treeNode.right!=null){
                q.add(treeNode.right);
            
}
        }                
      //  System.out.println(queue.size());
    }
    
    public int insert(int v) {
        TreeNode tn = new TreeNode(v);
        TreeNode parent = queue.peek();
      //  System.out.println("adding "+v+" to parent "+parent.val);
        if(parent.left==null){
            parent.left = tn;
        }else{
            parent.right = tn;
            queue.poll();
            //System.out.println("queue "+queue.peek().val);
            if(!inQueue.contains(parent.left.val)){
                queue.add(parent.left);
            }
            
queue.add(parent.right);
        }
       // System.out.println(queue.size());
        return parent.val;
    }
    
    public TreeNode get_root() {
        return root;
    }
}
