/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class CBTInserter {
        List<TreeNode> arr = null;
        public CBTInserter(TreeNode root) {
            arr = new ArrayList<>();
            bfs(root, arr);
        }

        private void bfs(TreeNode rt, 
List<TreeNode> arr) {
            Deque<TreeNode> q = new ArrayDeque<>();
            q.offer(rt);
            while (!q.isEmpty()) {
                TreeNode u = q.poll();
                arr.add(u);
                if (u.left != null) q.offer(u.left);
                if (u.right != null) q.offer(u.right);
            }
        }
        public int insert(int v) {
            TreeNode u = new TreeNode(v);
            int idx = arr.size();
            TreeNode p = arr.get((idx - 1) / 2);
            
if (idx % 2 == 1) p.left = u;
            else p.right = u;
            arr.add(u);
            return p.val;
        }

        public TreeNode get_root() {
            return arr.get(0);
        }
    }
