/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class CBTInserter {
    
    TreeNode head;
    
    LinkedList<TreeNode> list;

    public CBTInserter(TreeNode root) {
        head = root;
        list = new LinkedList<TreeNode>();
        LinkedList<TreeNode> bag = new LinkedList<TreeNode>();
        bag.add(root);
        while(bag.size() > 0) {
            TreeNode first = bag.removeFirst();
            if (first==null) {
                continue;
            }
            list.addLast(first);
            bag.addLast(first.left);
            bag.addLast(first.right);
        }
    }
    
    public int insert(int v) {
        TreeNode parent;
        while(true) {
            parent = list.removeFirst();
            if (parent.left == null || parent.right == null) {
                list.addFirst(parent);
                break;
            }
        }
        TreeNode child = new TreeNode(v);
        list.addLast(child);
        if (parent.left == null) {
            parent.left = child;
        } else {
            parent.right = child;
        }
        return parent.val;
    }
    
    public TreeNode get_root() {
        return head;
    }
}
