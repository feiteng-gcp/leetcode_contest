	class CBTInserter {
		int sz;
		TreeNode root;
		
	    public CBTInserter(TreeNode root) {
	        this.root = root;
	        sz = count(root);
	    }
	    
	    public int count(TreeNode x)
	    {
	    	if(x == null)return 0;
	    	return count(x.left) + 1 + count(x.right);
	    }
	    
	    public int insert(int v) {
	    	sz++;
	        TreeNode x = new TreeNode(v);
	        TreeNode cur = root;
	        TreeNode par = null;
	        for(int i = 30-Integer.numberOfLeadingZeros(sz);i >= 0;i--){
        		par = cur;
	        	if(sz<<~i<0){
	        		cur = cur.right;
	        	}else{
	        		cur = cur.left;
	        	}
	        }
	        if(sz<<~0<0){
	        	par.right = x;
	        }else{
	        	par.left = x;
	        }
	        return par.val;
	    }
	    
	    public TreeNode get_root() {
	        return root;
	    }
	}
