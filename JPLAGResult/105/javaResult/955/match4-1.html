import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;


class CBTInserter {
    TreeNode root;
    
Queue<TreeNode> queue;
    boolean left = true;
    public CBTInserter(TreeNode root) {
        this.root = root;
        queue = new LinkedList<>();
        Queue<TreeNode> tmp = new LinkedList<>();
        tmp.add(root);
        while (!tmp.isEmpty()) {
            TreeNode node = tmp.poll();
            if (node.left != null && node.right != null) {
                tmp.add(node.left);
                tmp.add(node.right);
            }else if (node.left != null) {
                queue.add(node);
                left = false;
                tmp.add(node.left);
            }else {
                queue.add(node);
            
}
        }
    }

    public int insert(int v) {
        TreeNode tmp = null;
        if (left) {
            tmp = queue.peek();
            TreeNode node = new TreeNode(v);
            tmp.left = node;
            queue.add(node);
            left = false;
        }else {
            left = true;
            tmp = queue.poll();
            TreeNode node = new TreeNode(v);
            tmp.right = node;
            queue.add(node);
        }
        
return tmp.val;   
    }

    public TreeNode get_root() {
        return root;
    }
}
