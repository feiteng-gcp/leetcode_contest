/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class CBTInserter {
    Queue<TreeNode> qu;
    TreeNode root;
    public CBTInserter(TreeNode root) {
        this.root=root;
        this.qu=new LinkedList<TreeNode>();
        
Queue<TreeNode> tempQu=new LinkedList<TreeNode>();
        tempQu.offer(root);
        while(!tempQu.isEmpty()){
            TreeNode top=tempQu.poll();
            if(top.right==null){
                qu.offer(top);
            }
            if(top.left!=null){
                tempQu.offer(top.left);
            }
            if(top.right!=null){
                tempQu.offer(top.right);
            
}
        }
    }
    
    public int insert(int v) {
        TreeNode top=qu.peek();
        TreeNode curr=new TreeNode(v);
        
if(top.left==null){
            top.left=curr;
        }
        else{
            top.right=curr;
            qu.poll();
        }
        qu.offer(curr);
        return top.val;
    }
    
    public TreeNode get_root() {
        return root;
    }
}
