class Solution {
    const long M = 1000000007;
    long mul(long x, long y){
        return (x*y)%M;
    }
    long add(long x, long y){
        return (x + y)%M;
    }
    long dp[105][105];
    long init;
    long dfs(int L, int N, const int&K){
        if(L<N) return 0;
        if(N<=K) return 0;
        if(N == K+1 && L == N){
            return init;
        }
        if(dp[L][N]>=0) return dp[L][N];
        dp[L][N] = mul(N, dfs(L-1, N-1, K));
        long ad = mul(N-K, dfs(L-1, N, K));
        dp[L][N] = add(dp[L][N], ad);
        return dp[L][N];
    }

public:
    int numMusicPlaylists(int N, int L, int K) {
        memset(dp, -1, sizeof(dp));
        init = 1;
        for(int k=2; k<=(K+1); ++k) init = mul(init, k);
        return dfs(L, N, K);
    }
};
