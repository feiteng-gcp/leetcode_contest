class Solution {
public:
  typedef long long ll;
  
  const int MOD = 1e9 + 7;
  
  ll C[101][101], P[101][101];

  ll pow(int a, int b) {
	  if (b == 0) {
		  return 1;
	  }
	  ll p = pow(a, b / 2);
	  ll res = (p * p) % MOD;
	  if (b % 2) {
		  res = (a * res) % MOD;
	  }
	  return res;
  }
  

  int numMusicPlaylists(int N, int M, int K) {
    C[0][0] = 1;
	  for (int i = 1; i <= N; i++) {
		  C[i][0] = 1;
		  for (int j = 1; j <= i; j++) {
			  C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % MOD;
		  }
	  }
	  for (int i = 1; i <= N; i++) {
		  P[i][0] = 1;
		  for (int j = i; j >= 1; j--) {
			  P[i][i - j + 1] = (j * P[i][i - j]) % MOD;
		  }
	  }
	  ll res = 0;
	  for (int i = 0; i <= N; i++) {
		  int s = (i % 2) ? -1 : 1;
		  res += s * (((C[N][i] * P[N - i][K]) % MOD) * pow(N - K - i, M - K)) % MOD;
		  res %= MOD;
	  }
	  if (res < 0) res += MOD;
    return res;
  }
};
