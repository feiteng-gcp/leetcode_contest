class Solution {
public:
    long long a[101][101];
    const long long M = 1000000007;

    int numMusicPlaylists(int N, int L, int K) {
        a[0][0] = 1;
        for (int i = 1; i <= N; i++) {
            a[i][i] = a[i-1][i-1] * (N - i + 1) % M;
        }
        for (int i = K+2; i <= L; i++) a[i][K+1] = a[i-1][K+1];
        for (int i = K+2; i <= L; i++) {
            for (int j = K+2; j < i && j <= N; j++) {
                a[i][j] = (a[i-1][j-1] * (N - j + 1) % M +
                    a[i-1][j] * (j - K) % M) % M;
            }
        }
        return a[L][N];
    }
};
