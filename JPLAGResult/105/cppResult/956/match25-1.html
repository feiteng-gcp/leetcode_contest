const int N = 107;
const int P = 1e9 + 7;
int f[N], C[N][N];
class Solution {
public:
	void init() {
		rep(i, 0, N) {
			C[i][0] = C[i][i] = 1;
			rep(j, 1, i) C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % P;
		}
	}
	int gao(int N, int L, int K) {
    	memset(f, 0, sizeof(f));
    	f[1] = N;
    	rep(i, 2, L + 1) {
    		int now = N - min(i - 1, K);
    		f[i] = 1ll * f[i - 1] * now % P;
    	}
    	return f[L];
	}
    int numMusicPlaylists(int N, int L, int K) {
    	init();
//    	rep(l, K + 1, L + 1)
//    		dd(l), de(gao(l, L, K));
//    	int ret = (gao(N, L, K) - 1ll * N * gao(N - 1, L, K) % P + P) % P;

    	int ret = 0;
    	rep(l, K + 1, N + 1) {
    		int now = 1ll * C[N][l] * gao(l, L, K) % P;
    		if (~(N - l) & 1) {
    			ret = (ret + now) % P;
    		} else {
    			ret = (ret - now + P) % P;
    		}
    	}
    	return ret;
    }
};
