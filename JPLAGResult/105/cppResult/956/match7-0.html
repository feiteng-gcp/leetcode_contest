class Solution {
public:
#define MOD 1000000007

    int numMusicPlaylists(int N, int L, int K) {
        vector<vector<long long int>> dp(110, vector<long long int>(110, 0));
        dp[0][0] = 1;
        
        for (int i = 1; i <= L; ++i) {
            for (int j = 1; j <= N; ++j) {
                if (j <= N) dp[i][j] += dp[i - 1][j - 1];
                if (j > K)  dp[i][j] += dp[i - 1][j] * (j - K);
                dp[i][j] %= MOD;
            }
        }

        long long int ret = dp[L][N];
        for (int i = 2; i <= N; ++i) {
            ret *= i;
            ret %= MOD;
        }
        return ret;
    }
};
