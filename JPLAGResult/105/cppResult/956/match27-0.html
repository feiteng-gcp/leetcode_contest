class Solution {
public:
    
    int N, L, K;
    const int Mod = 1000000007;
    int dp[102][102][102];
    
    void take_mod(int &x) {
        while(x >= Mod) {
            x -= Mod;
        }
        while(x < 0) {
            x += Mod;
        }
    }
    
    int rec(int idx, int news, int behind) {
        if(idx == L) {
            return news == N;
        }
        
        int &ret = dp[idx][news][behind];
        if(ret != -1) {
            return ret;
        }
        ret = 0;
        if(idx <= K) {
            ret += (1LL * rec(idx + 1, news + 1, behind + 1) * (N - news)) % Mod;
        } else {
            ret += (1LL * rec(idx + 1, news, behind + 1) * (news - behind)) % Mod;   /// one of the old ones
            take_mod(ret);
            for(int i = 0;i < behind - K;++i) {
                ret += (1LL * rec(idx + 1, news, behind - i)) % Mod;   /// one of the old ones
                take_mod(ret);
            }
            
            if(news < N) {
                ret += (1LL * rec(idx + 1, news + 1, behind + 1) * (N - news)) % Mod;
                take_mod(ret);
            }
        }
        return ret;
    }
    
    int numMusicPlaylists(int NN, int LL, int KK) {
        K = KK;
        L = LL;
        N = NN;
        
        memset(dp, 0xff, sizeof(dp));
