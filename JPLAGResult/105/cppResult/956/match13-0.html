class Solution {
public:
    int numMusicPlaylists(int N, int L, int K) {
        long long p[200][200];
        p[0][0] = 1;
        long long MOD = (1e9 + 7);
        for (int i = 1; i < 150; i++) {
            for (int j = 0; j <= i; j++) {
                if (i == j || j == 0) p[i][j] = 1;
                else {
                    p[i][j] = p[i - 1][j - 1] + p[i - 1][j];
                    p[i][j] %= MOD;
                }
            }
        }
        long long res = 0;
        long long s = 1;
        for (int i = 0; i < L; i++) {
            s *= (N - min(i, K));
            s %= MOD;
        }
        res = s;
        long long sign = 1;
        for (int i = 1; i < N; i++) {
            s = p[N][N - i];
            for (int j = 0; j < L; j++) {
                s *= (N - i - min(j, K));
                s %= MOD;
            }
            res = (res + MOD - (sign * s)) % MOD;
            sign *= -1;
        }
        return res;
    }
};
