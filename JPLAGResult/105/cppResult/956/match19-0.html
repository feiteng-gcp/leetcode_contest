#define MAX_N 128

const int MOD = 1e9 + 7.5;

class Solution {
public:
    int numMusicPlaylists(int N, int L, int K) {
        for (int i = 0; i < MAX_N; i++) C[i][0] = 1;
        for (int i = 1; i < MAX_N; i++) {
            for (int j = 1; j < MAX_N; j++) C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % MOD;
        }
        
        int ret = dp(N, L, K);
        for (int i = 1; i < N; i++) {
            if (i % 2 == 0) {
                ret = fix(ret + 1LL * C[N][i] * dp(N - i, L, K));
            } else {
                ret = fix(ret - 1LL * C[N][i] * dp(N - i, L, K));    
            }
               
        }
        return ret;
    }
    
    long long dp(int N, int L, int K) {
        vector<long long> f(L + 1);
        f[1] = N;
        for (int i = 2; i <= L; i++) {
            f[i] = f[i - 1] * max(0, max(N - i + 1, N - K));
            f[i] = fix(f[i]);
        }
        return f[L];
    }
    
    int fix(long long val) {
        return ((val % MOD) + MOD) % MOD;
    }
    
    int C[MAX_N][MAX_N];
};
