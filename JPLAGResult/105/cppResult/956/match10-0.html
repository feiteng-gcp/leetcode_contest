class Solution {
public:
    int numMusicPlaylists(int N, int L, int K) {
        int mod = 1e9+7;
        
        vector<vector<long>> dp(L+1, vector<long>(N+1));
        dp[0][0] = 1;
        
        for (int i = 1; i <= L; i++) {
            for (int j = 1; j <= min(i, N); ++j) {
                dp[i][j] = dp[i - 1][j - 1] * (N - j + 1) % mod;
                
                if (j > K) {
                    dp[i][j] += (dp[i - 1][j] * (j - K)) % mod;
                    dp[i][j] %= mod;
                }
            }
        }
        
        return dp[L][N];
    }
};
