class Solution {
  static const long long mod = 1e9 + 7;
public:
    int numMusicPlaylists(int N, int L, int K) {
      // f[i][j]
      vector<vector<long long>> f(L+1, vector<long long>(N+1, 0));
      f[0][0] = 1;
      for (int l = 0; l < L; ++l )
        for (int n = 0; n <= N; ++n ) {
          if (n < N) (f[l+1][n+1] += f[l][n] * (N-n)) %= mod;
          if (n > K) (f[l+1][n] += f[l][n] * (n-K)) %= mod;
        }
      return f[L][N];
    }
};
