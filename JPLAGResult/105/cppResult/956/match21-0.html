typedef long long LL;
const int mod = 1e9 + 7;
const int MAXN = 123;
vector<LL> fact, inv;
LL power(LL base, int p) {
    LL ret = 1;
    base = base % mod;
    while (p) {
        if (p & 1) ret = ret * base % mod;
        p >>= 1;
        base = base * base % mod;
    }
    return ret;
}
LL combine(int n, int k) {
    LL ret = fact[n] * inv[n-k] % mod;
    return ret * inv[k] % mod;
}
LL a(int n, int k) {
    return fact[n] * inv[n-k] % mod;
}
class Solution {
    LL solve(int n, int L, int k) {
        LL ret = a(n, k);
        for (int i = k+1; i <= L; i++) {
            ret = ret * (n-k) % mod;
        }
        return ret;
    }

public:
    int numMusicPlaylists(int n, int L, int k) {
        if (fact.empty()) {
            fact.push_back(1);
            for (int i = 1; i < MAXN; i++) {
                fact.push_back(fact.back()*i % mod);
            }
            inv.resize(MAXN);
            inv.back() = power(fact[MAXN-1], mod-2);
            for (int i = MAXN-2; i >= 0; i--) {
                inv[i] = inv[i+1] * (i+1) % mod;
            }
        }
        vector<LL> ret(n+1, 0);
        for (int i = k; i <= n; i++) {
            ret[i] = solve(i, L, k);
            for (int j = k; j < i; j++) {
                ret[i] = (ret[i] - combine(i, j)*ret[j]%mod + mod) % mod;
            }
        }
        // for (int i = 0; i <= n; i++) {
        //     cout << ret[i] << ' ';
        // }
        return ret.back();
        

        
        
    }
};
